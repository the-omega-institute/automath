\section{A self-dual synchronisation kernel and its completed determinant}
\label{sec:online-kernel}

We now specialise to a self-dual ten-state synchronisation kernel. We
write $B(u)$ for the weighted adjacency matrix, where the parameter
$u$ records the output-bit potential, and $B:=B(1)$ for the unweighted
matrix.

\subsection{The unweighted spectrum}

\begin{proposition}[Characteristic polynomial and residue constant]
  \label{prop:sync-kernel-spectrum}
  The matrix $B$ is primitive and satisfies
  \[
    \det(I-zB)=(z-1)(z+1)(3z-1)(z^3-z^2+z+1).
  \]
  In particular $\rho(B)=3$, and the Perron residue constant
  \[
    C_B:=\lim_{z\to 1/3}(1-3z)\det(I-zB)^{-1}
  \]
  is given by
  \[
    C_B=\frac{243}{272}.
  \]
\end{proposition}

\begin{proof}
  The determinant factorisation is obtained by direct elimination on
  the explicit $10\times 10$ matrix. Since $(3z-1)$ is the unique
  linear factor whose reciprocal lies outside the unit circle,
  $\rho(B)=3$. For the residue, substitute $z=1/3$ into the cofactor of
  the simple factor $(1-3z)$.
\end{proof}

Together with \Cref{prop:prelim-prime-orbit}, this yields
\[
  p_n(B)=\frac{3^n}{n}+O\!\left(\frac{\max\{\Lambda(B)^n,3^{n/2}\}}{n}\right).
\]

\subsection{Completion and algebraic geometry}

Let
\[
  \Delta(z,u):=\det(I-zB(u)).
\]
The kernel is self-dual in the sense that
\[
  \Delta(z,u)=\Delta(uz,u^{-1}).
\]
Introducing
\[
  u=r^2,\qquad s=r+r^{-1},\qquad w=zr,
\]
one obtains a completed determinant
$\widehat\Delta(w,s)\in\ZZ[s][w]$ characterised by
\[
  \widehat\Delta(w,r+r^{-1})=\Delta(w/r,r^2).
\]

\begin{proposition}[Explicit completed determinant]
  \label{prop:sync-hatdelta-completion}
  The completed determinant is
  \[
    \widehat\Delta(w,s)
    =1-sw-5w^2+3sw^3+(5-s^2)w^4+(s^3-6s)w^5+(s^2-1)w^6.
  \]
  It satisfies the parity symmetry
  \[
    \widehat\Delta(w,-s)=\widehat\Delta(-w,s).
  \]
\end{proposition}

\begin{proof}
  Substitute $u=r^2$ and $z=w/r$ into $\Delta(z,u)$, clear
  denominators, and use the self-duality under $r\mapsto r^{-1}$. The
  resulting Laurent polynomial is invariant under $r\leftrightarrow
  r^{-1}$ and therefore belongs to $\ZZ[r+r^{-1}][w]=\ZZ[s][w]$. A
  direct expansion yields the displayed formula, and the parity
  relation is then immediate by inspection.
\end{proof}

Let $\mathcal X$ be the normalisation of the affine curve
$\widehat\Delta(w,s)=0$. The involution
\[
  \iota:(w,s)\longmapsto(-w,-s)
\]
preserves the curve by \Cref{prop:sync-hatdelta-completion}.

\begin{proposition}[The quotient is a smooth plane quartic]
  \label{prop:sync-hatdelta-quotient}
  The function-field invariants
  \[
    x:=w^2,\qquad y:=sw
  \]
  identify the quotient $\mathcal X/\langle\iota\rangle$ birationally
  with the affine plane curve
  \[
    F(x,y):=
    1-y-5x+3xy+5x^2-xy^2+xy^3-6x^2y+x^2y^2-x^3=0.
  \]
  The projective closure of $F(x,y)=0$ is a smooth plane quartic.
  Consequently
  \[
    g(\mathcal X/\langle\iota\rangle)=3.
  \]
\end{proposition}

\begin{proof}
  Since $\iota$ changes the signs of both $w$ and $s$, the quantities
  $x=w^2$ and $y=sw$ are invariant. Substituting $s=y/w$ and
  $w^2=x$ into $\widehat\Delta(w,s)=0$ and clearing denominators gives
  the displayed equation. Conversely, adjoining $w=\sqrt{x}$ recovers
  the original function field, so the quotient is birational to
  $F(x,y)=0$.

  Homogenising $F$ yields a plane quartic. A direct check of the
  partial derivatives on the affine chart and on the line at infinity
  shows that the projective curve is smooth. The genus formula for a
  smooth plane quartic then gives $g=(4-1)(4-2)/2=3$.
\end{proof}

\begin{proposition}[Two-sheeted cover of genus six]
  \label{prop:sync-hatdelta-double-cover}
  The function field of $\mathcal X$ is
  \[
    \QQ(\mathcal X)=\QQ(\mathcal X/\langle\iota\rangle)(\sqrt{x}),
  \]
  so $\mathcal X\to\mathcal X/\langle\iota\rangle$ is a double cover.
  This cover is branched at exactly two points, namely the point
  $(x,y)=(0,1)$ and one point at infinity. Hence
  \[
    g(\mathcal X)=6.
  \]
\end{proposition}

\begin{proof}
  From $x=w^2$ and $y=sw$ we recover $s=y/w$ once $w=\sqrt{x}$ is
  adjoined, so the extension is quadratic. Branch points are exactly
  the points of the quotient where $x$ has odd valuation. On the affine
  model, $x$ vanishes simply at $(0,1)$. At infinity there is one point
  where $x$ has a simple pole and the remaining points have even
  valuation, hence do not branch. Hurwitz therefore gives
  \[
    2g(\mathcal X)-2
    =2\bigl(2\cdot 3-2\bigr)+2,
  \]
  which simplifies to $g(\mathcal X)=6$.
\end{proof}

\begin{proposition}[Generic Galois group]
  \label{prop:sync-hatdelta-galois-s6-generic}
  Viewed as a polynomial in $w$, the completed determinant
  $\widehat\Delta(w,s)$ has generic Galois group $S_6$ over $\QQ(s)$.
\end{proposition}

\begin{proof}
  The polynomial has degree $6$ in $w$. A direct discriminant
  computation shows that its discriminant is not a square in $\QQ(s)$,
  so the Galois group is not contained in $A_6$. On the other hand,
  explicit specialisations in $s$ produce one factorisation pattern with
  a simple transposition and another with an irreducible quintic factor,
  hence the generic group contains a transposition and a $5$-cycle.
  These elements generate $S_6$, so the generic Galois group is $S_6$.
\end{proof}

\subsection{Cyclotomic resultants and twisted Perron roots}

For $m\ge 2$, let
\[
  s_m:=2\cos(\pi/m),
\]
and let $\Phi_m^+(s)$ be the minimal polynomial of $s_m$. Define the
resultant
\[
  R_m(w):=\Res_s\bigl(\widehat\Delta(w,s),\Phi_m^+(s)\bigr)\in\ZZ[w].
\]
Then the twisted Perron root is
\[
  \rho_m:=\frac{1}{\min\{\abs{w}:R_m(w)=0\}}.
\]

\begin{theorem}[Cyclotomic resultant degree law]
  \label{thm:sync-cyclotomic-degree-law}
  For every $m\ge 4$,
  \[
    \deg_w R_m=3\varphi(2m).
  \]
  Moreover, if $m$ is even then $R_m(w)$ is an even polynomial in $w$.
\end{theorem}

\begin{proof}
  Let $d_m:=\deg \Phi_m^+=\varphi(2m)/2$. The resultant is the
  determinant of the Sylvester matrix of the cubic polynomial
  $s\mapsto\widehat\Delta(w,s)$ and the degree-$d_m$ polynomial
  $\Phi_m^+(s)$. Inspecting the highest $w$-degree term in that
  determinant gives $6d_m=3\varphi(2m)$. When $m$ is even, the minimal
  polynomial $\Phi_m^+(s)$ is even, and the parity relation
  $\widehat\Delta(w,-s)=\widehat\Delta(-w,s)$ from
  \Cref{prop:sync-hatdelta-completion} forces $R_m(-w)=R_m(w)$.
\end{proof}

\begin{corollary}[The worst twist $m=2$]
  \label{cor:rho-m2-closed-form}
  One has
  \[
    R_2(w)=\widehat\Delta(w,0)
    =1-5w^2+5w^4-w^6
    =(1-w^2)(w^4-4w^2+1),
  \]
  and therefore
  \[
    \rho_2=\frac{1}{\sqrt{2-\sqrt3}}=\sqrt{2+\sqrt3}.
  \]
\end{corollary}

\begin{proof}
  Since $s_2=0$, the resultant is obtained by direct substitution. The
  smallest-modulus zero of $R_2$ is $\sqrt{2-\sqrt3}$, hence the stated
  reciprocal formula for $\rho_2$.
\end{proof}

\subsection{Endpoint jets and the large-\texorpdfstring{$m$}{m} asymptotics}

Let $w(s)$ denote the simple branch of $\widehat\Delta(w,s)=0$ near
$(w,s)=(1/3,2)$ and write
\[
  \rho(s):=\frac{1}{w(s)}.
\]

\begin{proposition}[Endpoint jet]
  \label{prop:sync-endpoint-jet}
  The branch $w(s)$ is analytic near $s=2$, with
  \[
    w(2)=\frac13,\qquad
    w'(2)=-\frac{11}{153},\qquad
    \rho'(2)=\frac{11}{17},\qquad
    \rho''(2)=\frac{1351}{9826}.
  \]
  Moreover, if $\delta:=2-s$, then
  \[
    \rho(2-\delta)
    =3-\frac{11}{17}\delta+\frac{1351}{19652}\delta^2
      -\frac{17133}{45435424}\delta^3+O(\delta^4).
  \]
  The associated diffusion constant on the unit circle is
  \[
    \kappa_\infty=\frac{\rho'(2)}{4\rho(2)}=\frac{11}{204}.
  \]
\end{proposition}

\begin{proof}
  A direct computation gives
  \[
    \widehat\Delta(1/3,2)=0,\qquad
    \partial_w\widehat\Delta(1/3,2)\neq 0,
  \]
  so the implicit function theorem yields an analytic branch $w(s)$.
  Differentiating $\widehat\Delta(w(s),s)=0$ once gives
  \[
    w'(2)
    =-\frac{\partial_s\widehat\Delta(1/3,2)}
            {\partial_w\widehat\Delta(1/3,2)}
    =-\frac{11}{153}.
  \]
  Since $\rho=1/w$,
  \[
    \rho'(2)=-\frac{w'(2)}{w(2)^2}=\frac{11}{17}.
  \]
  The second derivative and the cubic Taylor expansion follow by
  differentiating the implicit equation repeatedly. Finally,
  $\kappa_\infty=\rho'(2)/(4\rho(2))$ by definition.
\end{proof}

\begin{theorem}[Asymptotic expansion of the twisted Perron roots]
  \label{thm:rho-gap-m12}
  As $m\to\infty$,
  \begin{align*}
    \rho_m
    &=3-\frac{11\pi^2}{17m^2}
      +\frac{1808\pi^4}{14739m^4}
      -\frac{27872249\pi^6}{2044594080m^6} \\
    &\quad
      -\frac{77645433997\pi^8}{33089710590720m^8}
      +\frac{2976802482983509\pi^{10}}
             {3442653489858508800m^{10}} \\
    &\quad
      +\frac{3832209193654464509\pi^{12}}
             {23878244605658617036800m^{12}}
      +O(m^{-14}).
  \end{align*}
  In particular,
  \[
    3-\rho_m\sim \frac{11\pi^2}{17m^2}.
  \]
\end{theorem}

\begin{proof}
  Since $s_m=2\cos(\pi/m)$ and $s_m\to 2$, the smallest-modulus root of
  $R_m$ is given by the analytic branch $w(s_m)$ for all sufficiently
  large $m$, hence $\rho_m=\rho(s_m)$. The Taylor series of
  $s_m=2\cos(\pi/m)$ is
  \[
    s_m
    =2-\frac{\pi^2}{m^2}+\frac{\pi^4}{12m^4}
      -\frac{\pi^6}{360m^6}+\cdots.
  \]
  A recursive implicit-function computation determines the six-jet of
  $\rho(s)$ at $s=2$. Composing that jet with the expansion of $s_m$
  produces the displayed coefficients through order $m^{-12}$.
  The leading term follows already from \Cref{prop:sync-endpoint-jet}.
\end{proof}

\begin{remark}
  The coefficient $\kappa_\infty=11/204$ is the unit-circle diffusion
  constant for the dominant branch. It is therefore read off from the
  same endpoint geometry that governs the $m^{-2}$ gap law.
\end{remark}

\subsection{Primitive orbit counts under cyclic lift}

\begin{proposition}[Cyclic-lift primitive asymptotics]
  \label{prop:cyclic-lift-primitive-asymptotics}
  Let $M$ be a primitive matrix with Perron root $\lambda$ and let
  \[
    \Lambda(M):=\max\{\abs{\mu}:\mu\in\spec(M),\ \mu\neq\lambda\}.
  \]
  For $q\ge 2$ define $M^{\langle q\rangle}:=M\otimes P_q$. Then
  \[
    p_n(M^{\langle q\rangle})=0
    \qquad (q\nmid n),
  \]
  and for $n=qm$ one has
  \[
    p_{qm}(M^{\langle q\rangle})
    =\frac{1}{m}\sum_{d\mid m}\mu(d)\Tr(M^{qm/d})
    =\frac{\lambda^{qm}}{m}
      +O\!\left(
        \frac{\max\{\Lambda(M)^{qm},\lambda^{qm/2}\}}{m}
      \right).
  \]
\end{proposition}

\begin{proof}
  Since $(M\otimes P_q)^n=M^n\otimes P_q^n$ and
  $\Tr(P_q^n)=q$ if $q\mid n$ and $0$ otherwise,
  \[
    \Tr\bigl((M^{\langle q\rangle})^n\bigr)
    =
    \begin{cases}
      q\,\Tr(M^n), & q\mid n,\\
      0, & q\nmid n.
    \end{cases}
  \]
  Applying Möbius inversion to the trace sequence of
  $M^{\langle q\rangle}$ gives the exact formula for primitive counts.
  The asymptotic estimate follows from
  \Cref{lem:primitive-orbit-asymptotic}.
\end{proof}
