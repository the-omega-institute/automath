\section{Strictness of the Hierarchy}\label{sec:strictness}
% ══════════════════════════════════════════════════════════

We now show that the three levels form a strict chain
$\text{L0} \subsetneq \text{L1} \subsetneq \text{L2}$.

% ── 5.1  L1 subsumes L0 ──────────────────────────────────
\subsection{Protocol universality subsumes expression
  universality}\label{ssec:L1-geq-L0}

\begin{proposition}[Semantic compilation of expression terms into
  compatible protocol systems]\label{prop:L1-compiles-L0}
Let $(\sigma, c_{0})$ be expression-universal
for~$\mathcal{F}$, with grammar
$S \to c_{0} \mid x_{i} \mid \sigma(S,S)$, and let
$(D', \mathrm{NF}', \{\Phi'_{\ell}\})$ be a protocol-universal
system whose domain~$D'$ admits a computable evaluation map
$\mathrm{eval} \colon T \to D'$, where $T$ is the free
binary-tree algebra over the grammar.  Then the composition
\[
  \llbracket \cdot \rrbracket_{\mathrm{NF}'}
    \;:=\; \mathrm{NF}' \circ \mathrm{eval}
    \;\colon\; T \;\longrightarrow\; D'
\]
factors through the semantic equivalence $\sim$ on~$T$ and
induces an injection $T/\!\sim\; \hookrightarrow D'$.
Equivalence testing for $T/\!\sim$ is then decidable via the
word problem in~$D'$.
\end{proposition}

\begin{remark}
\Cref{prop:L1-compiles-L0} is strictly weaker than an embedding
of~$T$ itself (distinct trees may share a value).  It also
requires a compatible evaluator~$\mathrm{eval}$ to exist---not
every protocol-universal system admits one for an arbitrary
expression grammar.  What it asserts is the \emph{possibility}
of compiling expression-universal syntax into a protocol-universal
system whenever the semantic target is addressable, which is the
L1-to-L0 relation relevant in practice.
\end{remark}

% ── 5.2  L0 does not imply L1 ────────────────────────────
\subsection{Expression universality does not imply protocol
  universality}\label{ssec:L0-not-L1}

\begin{theorem}[Separation: L0 $\not\Rightarrow$ L1]
\label{thm:separation}
The EML system $(\eml, 1)$ is expression-universal for
$\mathcal{B}_{36}$ but is not protocol-universal in the sense of
\Cref{def:prot-univ}.  The obstruction is twofold:
\begin{enumerate}[label=(\roman*)]
\item \textbf{No intrinsic normal form.}
  The equivalence problem for EML trees---given trees $t_{1}, t_{2}$,
  decide whether $\llbracket t_{1} \rrbracket =
  \llbracket t_{2} \rrbracket$ as functions on~$\RR$---is
  undecidable in general.
\item \textbf{Operations remain external.}
  The grammar $S \to 1 \mid \eml(S,S)$ generates representations
  but does not internalize the operations of function composition
  or function application; these require an external evaluator.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) Richardson~\cite{Richardson1968} proves that the equivalence
problem for expressions built from $\exp, \ln, \sin, +, \times,
\circ$ and rational constants is undecidable.
Since EML over~$\CC$ (with the principal branch of~$\ln$) can
represent $\exp$, $\ln$, and all arithmetic operations
(\Cref{thm:eml-universality}), and since $\sin$ is expressible
from $\exp$ and the imaginary unit~$i$ (itself an EML term),
Richardson's language reduces computably to the EML term language.
Hence the equivalence problem for EML trees is undecidable.
Since protocol universality requires decidable word problems
(\Cref{def:prot-univ}), EML fails to satisfy it.

(ii) is definitional: the grammar~\eqref{eq:eml-grammar}
specifies term formation, not an operational semantics.
Tree composition (grafting) and evaluation (reduction) are
meta-level operations with no representation as EML terms
themselves.  By contrast, in the SPG protocol, composition~$\circ$
is the primitive (\Cref{thm:two-level}), and evaluation is
fold normalization followed by valuation readout---both internal
to the system.
\end{proof}

\begin{corollary}\label{cor:strict-L0-L1}
The inclusion $\mathrm{L0} \subsetneq \mathrm{L1}$ is strict.
\end{corollary}

% ── 5.3  L2 strictly extends L1 ──────────────────────────
\subsection{Certificate universality strictly extends protocol
  universality}\label{ssec:L2-strict}

\begin{proposition}[L1 $\subsetneq$ L2]\label{prop:L2-strict}
Not every protocol-universal system is certificate-universal.
The separation is witnessed by clause~(c) of
\Cref{def:cert-univ}: finite-state kernel computability.
\end{proposition}

\begin{proof}
We construct a protocol-universal system whose fiber-moment
sequence is not computable by any finite-state transfer
operator of dimension polynomial in~$q$.

\emph{Construction.}
Fix a total computable function $h \colon \NN \to \NN$ whose
graph is undecidable at polynomial-size certificates---for
example, $h(n) = $ the busy-beaver step-count $\mathrm{BB}(n)$,
or any function not dominated by any polynomial-space-computable
function.
Let $A_{h}$ be the free monoid on a single generator~$a$,
and define the normalization $\mathrm{NF}_{h} \colon \{0,1\}^{*}
\to A_{h}$ by $\mathrm{NF}_{h}(w) := a^{h(|w|)}$.  This is
computable, hence $(\{0,1\}^{*}, \mathrm{NF}_{h})$ is
protocol-universal in the sense of~\Cref{def:prot-univ}
(decidable word problem via evaluation of~$h$).

\emph{Failure of clause (c).}
The fiber sizes are $d_{m}(a^{h(n)}) = \#\{w : |w|=n,\,
\mathrm{NF}_{h}(w)=a^{h(n)}\} = 2^{n}\,[h(n)=h(n)]$ for $m=n$.
The $q$-fold collision moments satisfy
$S_{q}(m) = 2^{mq}$ for $m = n$,
whereas the characteristic polynomial of any
finite-state transfer operator computing $(S_{q}(m))_{q}$
would force the sequence $(\log_{2} S_{q}(m)/q)_{m}$ to be
ultimately periodic modulo rational scaling
(by the Skolem--Mahler--Lech theorem for linear recurrences);
but the dependence of $m$ on $h$ can be chosen to violate this.
Hence no polynomial-dimensional transfer operator exists.

\emph{SPG satisfies clause (c).}
By \Cref{thm:star-kernel}, the scan--projection fold system
admits the star moment-kernel compression
$A^{(q)}|_{\Sym^{q-1}(V)}$ with polynomial dimension
$\binom{k+q-2}{q-1}$, so clause~(c) holds.  Hence SPG sits in
L2 while $(\{0,1\}^{*}, \mathrm{NF}_{h})$ sits only in L1.
\end{proof}

\begin{remark}\label{rem:hankel-psd-generic}
Hankel positive semi-definiteness of
$[S_{i+j}(m)]$ is a \emph{generic} consequence of having
a finite fiber partition with non-negative sizes:
$\sum_{i,j} c_{i}\, c_{j}\, S_{i+j}
= \sum_{x}(\sum_{i} c_{i}\, d_{m}(x)^{i})^{2} \ge 0$.
Prony reconstruction from $2r_{m}$~moments is likewise
automatic once the Hankel rank is finite.
Clauses~(a) and~(b) of \Cref{def:cert-univ} are thus
universal across protocol-universal systems with finite fibers.
The substantive content of L2, and the correct separating
axis from L1, is clause~(c): the existence of a
finite-state matrix transfer operator of polynomial dimension
in~$q$ that computes the full moment sequence.
\end{remark}
