\subsubsection{零点敏感正则化缺陷证书}\label{subsubsec:typed-address-biaxial-completion-zero-aware-regularization-gap-certificate}
Haar--Fej\'er--Riesz 对偶把外上包络合成压缩为正三角多项式的对数平均。剩余困难集中在
\[
\mathcal B_{r,M}
=
\frac12\int_{\mathbb T}\log\bigl(W_{r,M}+\Delta_{r,M}^+\bigr)\,dm
-
\frac12\int_{\mathbb T}\log W_r\,dm,
\]
即有限正则化对边界小值区域的放大。该缺陷可写成可核验的 sublevel 证书与局部零型证书。

仍令
\[
f_r(z):=\Phi_a(rz),
\qquad
W_r(e^{i\theta})=|f_r(e^{i\theta})|^2.
\]
若 Fourier 截断满足
\[
\|W_r-W_{r,M}\|_{L^\infty(\mathbb T)}\le\Delta_{r,M}^+,
\qquad
S_{r,M}:=W_{r,M}+\Delta_{r,M}^+,
\]
则
\[
W_r\le S_{r,M}\le W_r+2\Delta_{r,M}^+.
\]
记
\[
\delta:=2\Delta_{r,M}^+.
\]
于是
\[
0\le
\mathcal B_{r,M}
\le
\frac12\int_{\mathbb T}
\log\left(1+\frac{\delta}{W_r(e^{i\theta})}\right)\,dm.
\]
因此，正则化缺陷的控制等价于控制 \(W_r\) 的小值集合。

\begin{lemma}[小值分布恒等式]\label{lem:typed-address-biaxial-completion-sublevel-distribution-identity}
设 \(W\ge0\) 且 \(W>0\) 几乎处处。定义
\[
F_W(u):=m\{\theta:W(e^{i\theta})\le u\},
\qquad u>0.
\]
则对任意 \(\delta>0\)，
\[
\int_{\mathbb T}
\log\left(1+\frac{\delta}{W}\right)\,dm
=
\int_0^\infty
F_W\!\left(\frac{\delta}{e^t-1}\right)\,dt.
\]
\end{lemma}
\begin{proof}
对非负随机变量 \(X=\delta/W\)，有层析恒等式
\[
\mathbb E\log(1+X)
=
\int_0^\infty
\mathbb P(\log(1+X)>t)\,dt.
\]
而
\[
\log(1+\delta/W)>t
\iff
W<\frac{\delta}{e^t-1}.
\]
用 \(\le\) 或 \(<\) 定义分布函数只在至多可数个阈值处不同，不影响积分值。证毕。
\end{proof}

\begin{proposition}[sublevel 型正则化缺陷上界]\label{prop:typed-address-biaxial-completion-sublevel-regularization-gap-bound}
若存在 \(0<\alpha<1\)、\(C_\star>0\)、\(u_0>0\)，使得
\[
F_W(u)\le C_\star u^\alpha
\qquad(0<u\le u_0),
\]
则
\[
\frac12\int_{\mathbb T}
\log\left(1+\frac{\delta}{W}\right)\,dm
\le
\frac12\log\left(1+\frac{\delta}{u_0}\right)
+
\frac12 C_\star
\frac{\pi}{\sin(\pi\alpha)}
\delta^\alpha.
\]
\end{proposition}
\begin{proof}
令
\[
t_0:=\log\left(1+\frac{\delta}{u_0}\right).
\]
当 \(0\le t\le t_0\) 时，用平凡界 \(F_W\le1\)。当 \(t>t_0\) 时，有 \(\delta/(e^t-1)\le u_0\)，故
\[
F_W\!\left(\frac{\delta}{e^t-1}\right)
\le
C_\star\delta^\alpha(e^t-1)^{-\alpha}.
\]
因此
\[
\int_{\mathbb T}
\log\left(1+\frac{\delta}{W}\right)\,dm
\le
t_0+C_\star\delta^\alpha
\int_0^\infty(e^t-1)^{-\alpha}\,dt.
\]
代换 \(x=e^{-t}\) 得
\[
\int_0^\infty(e^t-1)^{-\alpha}\,dt
=
\int_0^1 x^{\alpha-1}(1-x)^{-\alpha}\,dx
=
B(\alpha,1-\alpha)
=
\frac{\pi}{\sin(\pi\alpha)}.
\]
乘以外侧的 \(1/2\) 即得结论。证毕。
\end{proof}

局部零型证书给出更具体的 sublevel 控制。令圆周分解为
\[
\mathbb T=E_0\cup I_1\cup\cdots\cup I_J,
\qquad
I_j=\{\theta:|\theta-\theta_j|\le\rho_j\},
\]
其中 \(m\) 为归一化 Haar 测度。假设在 \(E_0\) 上有
\[
W(e^{i\theta})\ge m_0>0,
\]
并且每个小值弧上有
\[
W(e^{i\theta})
\ge
C_j|\theta-\theta_j|^{2\nu_j},
\qquad
\nu_j\in\mathbb{N}.
\]
则
\[
\frac12\int_{\mathbb T}
\log\left(1+\frac{\delta}{W}\right)\,dm
\le
\frac12m(E_0)\log\left(1+\frac{\delta}{m_0}\right)
+
\sum_{j=1}^J
\frac{a_j}{2\sin\left(\frac{\pi}{2\nu_j}\right)},
\]
其中
\[
a_j:=\left(\frac{\delta}{C_j}\right)^{1/(2\nu_j)}.
\]
该式来自积分估计
\[
\frac1{4\pi}
\int_{-\rho_j}^{\rho_j}
\log\left(1+\frac{\delta}{C_j|x|^{2\nu_j}}\right)\,dx
\le
\frac{a_j}{2\sin\left(\frac{\pi}{2\nu_j}\right)}.
\]
特别地，简单零型小值给出
\[
\frac{a_j}{2\sin(\pi/2)}
=
\frac12\sqrt{\frac{\delta}{C_j}}.
\]

近边界零点可用带 shield 的局部模型记录。若在 \(I_j\) 上有
\[
W(e^{i\theta})
\ge
C_j\left((\theta-\theta_j)^2+\gamma_j^2\right)^{\nu_j},
\qquad
\gamma_j\ge0,
\]
仍令
\[
a_j:=\left(\frac{\delta}{C_j}\right)^{1/(2\nu_j)}.
\]
则该弧贡献可取
\[
B_j^+
:=
\begin{cases}
\displaystyle
\frac{a_j}{2\sin\left(\frac{\pi}{2\nu_j}\right)},
& \gamma_j=0,\\[1.2em]
\displaystyle
\min\left\{
\frac{a_j}{2\sin\left(\frac{\pi}{2\nu_j}\right)},
\frac{\rho_j}{2\pi}
\log\left(1+\left(\frac{a_j}{\gamma_j}\right)^{2\nu_j}\right)
\right\},
& \gamma_j>0.
\end{cases}
\]
第一项把 \((x^2+\gamma_j^2)^{\nu_j}\) 粗略降为 \(|x|^{2\nu_j}\)，第二项使用区间上的均匀下界 \(W\ge C_j\gamma_j^{2\nu_j}\)。当 \(a_j\ll\gamma_j\) 时，第二项呈现线性正则化尺度；当 \(a_j\gtrsim\gamma_j\) 时，贡献退化为边界零型尺度。

\begin{definition}[零点敏感正则化证书]\label{def:typed-address-biaxial-completion-zero-aware-regularization-certificate}
一份正则化缺陷证书记为
\[
\mathsf{ZACert}_{r,M}
:=
\left(
\delta,\,
E_0,m_0,\,
\{I_j,\theta_j,\rho_j,\nu_j,C_j,\gamma_j\}_{j=1}^J
\right),
\]
并要求以下条件均带有严格区间裕度：
\begin{enumerate}
  \item 覆盖：
  \[
  \mathbb T=E_0\cup I_1\cup\cdots\cup I_J.
  \]
  \item 远离小值区：
  \[
  W_r(e^{i\theta})\ge m_0
  \qquad(\theta\in E_0).
  \]
  \item 小值弧局部模型：
  \[
  W_r(e^{i\theta})
  \ge
  C_j\left((\theta-\theta_j)^2+\gamma_j^2\right)^{\nu_j}
  \qquad(\theta\in I_j).
  \]
  \item 正则化幅度：
  \[
  S_{r,M}\le W_r+\delta,
  \qquad
  \delta=2\Delta_{r,M}^+.
  \]
\end{enumerate}
证书输出
\[
B_{\mathrm{ZA}}^+
:=
\frac12m(E_0)\log\left(1+\frac{\delta}{m_0}\right)
+
\sum_{j=1}^J B_j^+.
\]
\end{definition}

\begin{theorem}[零点敏感正则化缺陷证书的正确性]\label{thm:typed-address-biaxial-completion-zero-aware-regularization-certificate-soundness}
若 \(\mathsf{ZACert}_{r,M}\) 通过，则
\[
\mathcal B_{r,M}\le B_{\mathrm{ZA}}^+.
\]
\end{theorem}
\begin{proof}
由 \(S_{r,M}\le W_r+\delta\)，得到
\[
\mathcal B_{r,M}
\le
\frac12\int_{\mathbb T}
\log\left(1+\frac{\delta}{W_r}\right)\,dm.
\]
在 \(E_0\) 上使用 \(W_r\ge m_0\)。在每个 \(I_j\) 上使用局部 shield 下界，并套用 \(B_j^+\) 的两种估计。覆盖条件使这些估计相加后控制整圆周积分。证毕。
\end{proof}

该证书可直接并入 Jensen 排斥阈值。若另有
\[
J_a(r)\le J_r^+,
\qquad
0<A_0^-\le|\Phi_a(0)|\le A_0^+,
\]
则
\[
\frac12\int_{\mathbb T}\log S_{r,M}\,dm-\log A_0^-
\le
J_r^+
+
\log\frac{A_0^+}{A_0^-}
+
B_{\mathrm{ZA}}^+.
\]
因此，一旦
\[
J_r^+
+
\log\frac{A_0^+}{A_0^-}
+
B_{\mathrm{ZA}}^+
<
\log\frac rR,
\]
便有
\[
\frac12\int_{\mathbb T}\log S_{r,M}\,dm-\log A_0^-
<
\log\frac rR.
\]
结合 Haar--Fej\'er--Riesz 谱因子证书，可推出
\[
\Phi_a(w)\ne0
\qquad(|w|\le R).
\]

该不等式给出端点层的尺度律。若主导小值阶数为
\[
\nu_{\max}:=\max_{1\le j\le J}\nu_j,
\]
则 \(B_{\mathrm{ZA}}^+\) 的典型主项按
\[
\delta^{1/(2\nu_{\max})}
\]
衰减。若 dyadic 层 \(r_q=1-\eta2^{-q}\)、\(R_q=1-2^{-q}\) 的剩余排斥裕度为
\[
\mathfrak m_q
:=
\log\frac{r_q}{R_q}
-
J_{r_q}^+
-
\log\frac{A_0^+}{A_0^-},
\]
则需要
\[
\delta
\lesssim
\left(\frac{\mathfrak m_q}{C_{\mathrm{ZA}}}\right)^{2\nu_{\max}}.
\]
在 \(\mathfrak m_q\asymp2^{-q}\) 的自然尺度下，这变成
\[
\delta\lesssim2^{-2\nu_{\max}q}.
\]
简单零型小值要求 \(\delta\lesssim2^{-2q}\)，二重小值要求 \(\delta\lesssim2^{-4q}\)。

若 Fourier 尾部满足
\[
\Delta_{r_q,M}^+\lesssim C\exp(-cM2^{-q}),
\]
则达到 \(\delta\lesssim2^{-2\nu q}\) 的充分尺度为
\[
M
\gtrsim
C_{\eta,c}\,\nu q\,2^q
\]
（常数吸收 \(\log C\) 与 \(\log 2\)）。因此，固定半径 Fourier 截断在端点层破除小值盲区时，自然携带 \(q2^q\) 级复杂度；若高阶小值或近边界 shield 极弱，则常数和有效阶数随之恶化。

综上，有限外谱因子证书的端点瓶颈可写为
\[
\text{Jensen 上界}
\;+\;
\text{基点区间}
\;+\;
\text{零点敏感正则化缺陷}
\;<\;
\text{排斥阈值}.
\]
其中 \(\mathsf{ZACert}_{r,M}\) 负责把 \(\log(W+\delta)-\log W\) 的小值放大效应转化为有限覆盖、局部阶数、局部常数与径向隔离尺度的可审计数据。
