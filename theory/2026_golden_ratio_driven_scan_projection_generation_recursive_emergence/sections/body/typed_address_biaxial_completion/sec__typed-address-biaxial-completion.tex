\section{类型化地址、双轴圆维与完成化闭合：从有限递归到 prime-register 外置账本}\label{sec:typed-address-biaxial-completion}
本章在前文已经建立的稳定折叠、递归地址化、Fibonacci-adic 完成、fold-aware restriction、圆维与残差圆维、prime-register 轴的 pro-圆维、相位圆维/寄存圆维双轴计量，以及 \(\mathrm{Read}_{\mathrm{US}}\) 的类型纪律之上，引入一个统一的完成化接口层。其目的不在于增设新的原语，而在于把有限递归轨道、完成化对象、类型化地址、双轴账本与后续动力学 \(\zeta\) 及有限部接口组织为同一条可审计前端链，从而把第 12 章与后续谱侧章节之间的过渡写成对象层、地址层与审计层同时闭合的结构。

\begin{definition}[类型化相位门证书链]\label{distill:bourgain-multiscale_decoupling_gates-typed-phase-gate-certificate-chain}
称一组有限数据为一条\textbf{类型化相位门证书链}，若它由有限层级
\[
\Sigma_0\xleftarrow{\rho_{1,0}}\Sigma_1\xleftarrow{\rho_{2,1}}\cdots\xleftarrow{\rho_{J,J-1}}\Sigma_J
\]
组成，其中每个 \(\Sigma_j\) 为有限集。对 \(\sigma\in\Sigma_j\) 记
\[
\operatorname{Ch}_j(\sigma):=\rho_{j+1,j}^{-1}(\sigma).
\]
每层配有有限地址集 \(A_j\)、地址读出 \(a_j:\Sigma_j\to A_j\)、整数 \(D_j\ge1\)、相位群
\[
\mu_{D_j}:=\{z\in\mathbb C:z^{D_j}=1\},
\]
相位读出 \(\theta_j:\Sigma_j\to\mu_{D_j}\)，以及投影
\[
\alpha_{j+1,j}:A_{j+1}\to A_j,
\qquad
\nu_{j+1,j}:\mu_{D_{j+1}}\to\mu_{D_j}.
\]
每个 \(\sigma\in\Sigma_j\) 还带有非负归一化门范数 \(\mathfrak N_{j,\sigma}\)。对每个尺度 \(j<J\)，给定例外集 \(X_j\subseteq\Sigma_j\)。对每个非例外 \(\sigma\in\Sigma_j\setminus X_j\)，给定常数 \(q_{j,\sigma},e_{j,\sigma}\ge0\)，sector 与 slab 标签
\[
s_{j,\sigma}:\operatorname{Ch}_j(\sigma)\to S_{j,\sigma},
\qquad
\ell_{j,\sigma}:\operatorname{Ch}_j(\sigma)\to L_{j,\sigma},
\]
对应预算 \(B^{\rm sec}_{j,\sigma}(s)\)、\(B^{\rm slab}_{j,\sigma}(\ell)\)，有限 prefix/Frostman 测试族 \(\mathcal B_{j,\sigma}\subseteq\mathcal P(\operatorname{Ch}_j(\sigma))\) 及预算 \(\beta_{j,\sigma}:\mathcal B_{j,\sigma}\to\mathbb R_{\ge0}\)，并给定 sticky 阈值 \(0<\lambda_{j,\sigma}<1\)。

称非例外 \(\sigma\) 在尺度 \(j\) 上通过类型化相位门审计，若以下条件同时成立：
\begin{enumerate}
  \item 对每个 \(\tau\in\operatorname{Ch}_j(\sigma)\)，有
  \[
  \alpha_{j+1,j}(a_{j+1}(\tau))=a_j(\sigma),
  \qquad
  \nu_{j+1,j}(\theta_{j+1}(\tau))=\theta_j(\sigma).
  \]
  \item 局部门估计成立：
  \[
  \sum_{\tau\in\operatorname{Ch}_j(\sigma)}\mathfrak N_{j+1,\tau}
  \le q_{j,\sigma}\mathfrak N_{j,\sigma}+e_{j,\sigma}.
  \]
  \item sector 与 slab 预算均成立：对所有 \(s\in S_{j,\sigma}\)、\(\ell\in L_{j,\sigma}\)，
  \[
  \sum_{\tau:s_{j,\sigma}(\tau)=s}\mathfrak N_{j+1,\tau}\le B^{\rm sec}_{j,\sigma}(s),
  \qquad
  \sum_{\tau:\ell_{j,\sigma}(\tau)=\ell}\mathfrak N_{j+1,\tau}\le B^{\rm slab}_{j,\sigma}(\ell).
  \]
  \item 任一子地址纤维都不 sticky：对所有 \(a\in A_{j+1}\) 满足 \(\alpha_{j+1,j}(a)=a_j(\sigma)\)，
  \[
  \sum_{\substack{\tau\in\operatorname{Ch}_j(\sigma)\\ a_{j+1}(\tau)=a}}\mathfrak N_{j+1,\tau}
  \le
  \lambda_{j,\sigma}
  \sum_{\tau\in\operatorname{Ch}_j(\sigma)}\mathfrak N_{j+1,\tau}.
  \]
  \item prefix/Frostman 预算均成立：对每个 \(U\in\mathcal B_{j,\sigma}\)，
  \[
  \sum_{\tau\in U}\mathfrak N_{j+1,\tau}\le\beta_{j,\sigma}(U).
  \]
\end{enumerate}
该证书链只是把本节中的有限递归轨道、类型化地址、相位轴与双轴审计账本压缩为同一个有限前端对象；不把任何朴素截断量默认为跨尺度不变量。
\end{definition}

\begin{theorem}[类型化相位门证书链的归一化转移与首次失败分类]\label{distill:bourgain-multiscale_decoupling_gates-typed-phase-gate-certificate-chain-transfer}
设 \(\Sigma_0\leftarrow\cdots\leftarrow\Sigma_J\) 为定义 \ref{distill:bourgain-multiscale_decoupling_gates-typed-phase-gate-certificate-chain} 的类型化相位门证书链。令
\[
\mathsf E_j:=\sum_{\sigma\in\Sigma_j}\mathfrak N_{j,\sigma}.
\]
假设每个非例外 \(\sigma\in\Sigma_j\setminus X_j\) 均通过类型化相位门审计，并且例外 spill 预算满足
\[
\sum_{\sigma\in X_j}\sum_{\tau\in\operatorname{Ch}_j(\sigma)}
\mathfrak N_{j+1,\tau}\le r_j
\qquad(0\le j<J).
\]
定义
\[
Q_j:=
\begin{cases}
\max_{\sigma\in\Sigma_j\setminus X_j}q_{j,\sigma},&\Sigma_j\setminus X_j\ne\varnothing,\\
0,&\Sigma_j\setminus X_j=\varnothing,
\end{cases}
\qquad
G_j:=\sum_{\sigma\in\Sigma_j\setminus X_j}e_{j,\sigma}.
\]
则归一化相位门范数满足
\[
\mathsf E_J
\le
\left(\prod_{j=0}^{J-1}Q_j\right)\mathsf E_0
+
\sum_{k=0}^{J-1}
\left(\prod_{i=k+1}^{J-1}Q_i\right)(G_k+r_k),
\]
其中空乘积取 \(1\)。

反过来，若某条有限证书链在上述审计语义下首次被拒绝，则存在最小尺度 \(j<J\) 与某个 \(\sigma\in\Sigma_j\)，使拒绝见证属于以下有限类型之一：类型化地址或相位投影不相容；局部门估计失败；某个 sector 或 slab 标签纤维超预算；某个子地址纤维 sticky；某个 prefix/Frostman 测试集超预算；或例外 spill 超出 \(r_j\)。因此，在该有限审计语言内，失败不会以未标记异常出现，而必带有一个可枚举的 typed-address、sector/slab、sticky、Frostman 或 spill 见证。
\end{theorem}

\begin{theorem}[构造尾段不降低刚性秩与终端寄存器下界]\label{distill:bourgain-entropy_increment_closure-construction-tail-rigidity-rank-propagation}
固定素数 \(p\)，有限维 \(\mathbb F_p\)-向量空间 \(A\)，以及有限链
\[
E_i\xleftarrow{\rho_{i+1,i}}E_{i+1}\xleftarrow{}\cdots\xleftarrow{\rho_{N,N-1}}E_N
\]
其中每个 \(E_j\) 非空且每个 \(\rho_{j+1,j}\) 满射。设在第 \(i\) 层给定构造 carrier
\[
c_i:E_i\longrightarrow C_i
\]
与刚性标记
\[
\Theta_i:E_i\longrightarrow A.
\]
对 \(j\ge i\) 定义
\[
\rho_{j,i}:=\rho_{i+1,i}\circ\cdots\circ\rho_{j,j-1},
\qquad
c_j:=c_i\circ\rho_{j,i},
\qquad
\Theta_j:=\Theta_i\circ\rho_{j,i}.
\]
对每个 \(c\in c_i(E_i)\)，任选基点 \(e_c\in E_i\cap c_i^{-1}(c)\)，并任选其一个提升 \(\widetilde e_c\in E_N\) 满足 \(\rho_{N,i}(\widetilde e_c)=e_c\)。定义
\[
W_i(c):=\operatorname{Span}_{\mathbb F_p}\{\Theta_i(e)-\Theta_i(e_c):e\in E_i\cap c_i^{-1}(c)\}\subseteq A,
\]
\[
W_N(c):=\operatorname{Span}_{\mathbb F_p}\{\Theta_N(x)-\Theta_N(\widetilde e_c):x\in E_N\cap c_N^{-1}(c)\}\subseteq A.
\]
则
\[
W_N(c)=W_i(c)
\qquad(c\in c_i(E_i)).
\]
特别地，刚性秩
\[
L_i:=\max_{c\in c_i(E_i)}\dim_{\mathbb F_p}W_i(c)
\]
在任何只作上述满射构造提升的尾段中不下降。进一步，若存在终端线性寄存器
\[
r:E_N\longrightarrow (\mathbb F_p)^B
\]
以及线性映射 \(D_c:(\mathbb F_p)^B\to A\)，使得对所有 \(c\in c_i(E_i)\) 与所有 \(x\in E_N\cap c_N^{-1}(c)\) 有
\[
\Theta_N(x)-\Theta_N(\widetilde e_c)
=
D_c\bigl(r(x)-r(\widetilde e_c)\bigr),
\]
则必有
\[
B\ge L_i.
\]
\end{theorem}

\begin{definition}[相对证书的四门基变账本]\label{distill:grothendieck-relative_certificate_base_change_stability-four-gate-certificate-ledger}
设 \(\mathcal B\) 为基范畴，其态射称为允许基变。对每个 \(b\in\mathcal B\)，给定可接受对象域 \(\mathcal A_b\)、证书纤维 \(\mathscr F_b(a)\)、障碍账本 \(\mathrm{Obs}_b(a)\) 与闭合谓词 \(\mathrm{Cl}_b(a)\)。对每个允许基变 \(\nu:b'\to b\)，给定
\[
\nu^\#:\mathcal A_b\to\mathcal A_{b'},\qquad
\nu^*:\mathscr F_b(a)\to\mathscr F_{b'}(\nu^\#a),
\qquad
\nu^\flat:\mathrm{Obs}_b(a)\to\mathrm{Obs}_{b'}(\nu^\#a),
\]
并要求对复合 \(b''\xrightarrow{\mu}b'\xrightarrow{\nu}b\) 有
\[
(\nu\mu)^\#=\mu^\#\nu^\#,\qquad
(\nu\mu)^*=\mu^*\nu^*,\qquad
(\nu\mu)^\flat=\mu^\flat\nu^\flat .
\]
称一条相对证书声明在 \(\nu\) 下通过四门基变，若以下四项成立：构造门 \(\operatorname{Con}_\nu\) 表示 \(\nu^\#a\) 与 \(\nu^*s\) 仍为协议允许对象；刚性门 \(\operatorname{Rig}_\nu\) 表示对象层等价、共同细化等价与读出值类被 \(\nu^*\) 保持；障碍门 \(\operatorname{Obs}_\nu\) 表示 \(\nu^\flat\) 与相应 \v{C}ech coboundary、边残差或二阶障碍算子交换；闭合门 \(\operatorname{Cl}_\nu\) 表示杀死障碍的顶点势、共同细化截面或 untwisting potential 亦可拉回。
\end{definition}

\begin{theorem}[相对证书基变的残差正规形]\label{distill:grothendieck-relative_certificate_base_change_stability-four-gate-certificate-normal-form}
任何形如
\[
\text{局部证书存在}\Longrightarrow \text{全局对象闭合}
\]
的全局声明，若要沿允许基变 \(\nu:b'\to b\) 使用，必须分别指定定义 \ref{distill:grothendieck-relative_certificate_base_change_stability-four-gate-certificate-ledger} 中的四门。并且，在四门语法下，任一基变坏例子都有唯一的首次失败类型
\[
\mathfrak R_{\mathrm{con}},\quad
\mathfrak R_{\mathrm{rig}},\quad
\mathfrak R_{\mathrm{obs}},\quad
\mathfrak R_{\mathrm{cl}} .
\]
若前三门成立而闭合门失败，则该坏例子不是原全局定理的反例，而是一份合法闭合残差
\[
\operatorname{Res}^{\mathrm{cl}}_\nu(a)\in \mathrm{Obs}_{b'}(\nu^\#a),
\]
在具体模型中可表现为 reduced holonomy 类、\v{C}ech 二阶类、曲率供给类或 realization-kernel 类。
此外，若一份局部 Omega 证书族对所有允许基变均通过障碍门，则其纤维wise 障碍账本在基变范畴上函子化：对任意复合 \(b''\xrightarrow{\mu}b'\xrightarrow{\nu}b\)，有
\[
(\nu\mu)^\flat=\mu^\flat\nu^\flat,
\qquad
\delta((\nu\mu)^\flat c)=(\nu\mu)^\flat(\delta c),
\]
因而 cocycle 条件与已经定义的上同调类均按拉回传递。
\end{theorem}
\begin{proof}
任何全局闭合证明在对象层上都按固定顺序使用四类数据。首先必须有可接受对象与证书纤维，故若 \(\nu^\#a\notin\mathcal A_{b'}\) 或 \(\nu^*s\) 不属于目标证书纤维，则失败只能属于 \(\mathfrak R_{\mathrm{con}}\)。排除此情形后，对象已经存在；若原基上使用的证书等价、值保持或共同细化关系不被 \(\nu^*\) 保持，则同一证书类在拉回后分裂，失败属于 \(\mathfrak R_{\mathrm{rig}}\)。

再排除刚性失败后，局部对象及其等价关系均已合法。剩余局部到全局步骤必须先形成障碍账本。若拉回后的边数据不是 cocycle，或二阶障碍未满足 incidence 闭合，则相应上同调类尚未定义，失败属于 \(\mathfrak R_{\mathrm{obs}}\)，不能记为非零类。最后，若障碍对象已经合法而仍无顶点势、共同细化截面或 untwisting potential 将其杀死，则唯一可能的失败是闭合门失败，故得到 \(\mathfrak R_{\mathrm{cl}}\)。四类的互斥性来自这个先后依赖次序，穷尽性来自全局闭合证明无法绕过这四个阶段。

若前三门成立而闭合门失败，则障碍对象已经位于 \(\mathrm{Obs}_{b'}(\nu^\#a)\) 中；因此它是已分类的残差，而不是未成形的异常。最后，障碍门的定义给出 \(\nu^\flat\) 与 \(\delta\) 交换，复合相容性又给出 \((\nu\mu)^\flat=\mu^\flat\nu^\flat\)。于是 cocycle、coboundary 与合法上同调类均沿允许基变函子传递。证毕。
\end{proof}

先证明 \(W_N(c)\subseteq W_i(c)\)。若 \(x\in E_N\cap c_N^{-1}(c)\)，令 \(e:=\rho_{N,i}(x)\)。由 \(c_N=c_i\circ\rho_{N,i}\) 得 \(e\in E_i\cap c_i^{-1}(c)\)。又由 \(\Theta_N=\Theta_i\circ\rho_{N,i}\) 与 \(\rho_{N,i}(\widetilde e_c)=e_c\)，有
\[
\Theta_N(x)-\Theta_N(\widetilde e_c)
=\Theta_i(e)-\Theta_i(e_c),
\]
故每个生成元都属于 \(W_i(c)\)。

反向包含由满射性给出。取任意 \(e\in E_i\cap c_i^{-1}(c)\)。由于 \(\rho_{N,i}\) 是满射，存在 \(x\in E_N\) 使 \(\rho_{N,i}(x)=e\)。于是
\[
c_N(x)=c_i(\rho_{N,i}(x))=c_i(e)=c,
\]
且
\[
\Theta_i(e)-\Theta_i(e_c)
=\Theta_N(x)-\Theta_N(\widetilde e_c).
\]
因此 \(W_i(c)\subseteq W_N(c)\)。两边相等成立，取维数最大值得到 \(L_i\) 在构造尾段中不下降。

现在设终端寄存器与线性解码器满足题设等式。对固定 \(c\)，上述等式表明 \(W_N(c)\) 的每个生成元都属于 \(\operatorname{im}D_c\)。故
\[
W_i(c)=W_N(c)\subseteq \operatorname{im}D_c.
\]
于是
\[
\dim_{\mathbb F_p}W_i(c)\le \dim_{\mathbb F_p}\operatorname{im}D_c\le B.
\]
对所有 \(c\in c_i(E_i)\) 取最大值，得到 \(B\ge L_i\)。证毕。
\end{proof}

\begin{proof}
对固定尺度 \(j\)，将 \(\mathsf E_{j+1}\) 按父层元素分解，得
\[
\mathsf E_{j+1}
=\sum_{\sigma\in\Sigma_j\setminus X_j}
\sum_{\tau\in\operatorname{Ch}_j(\sigma)}\mathfrak N_{j+1,\tau}
+\sum_{\sigma\in X_j}
\sum_{\tau\in\operatorname{Ch}_j(\sigma)}\mathfrak N_{j+1,\tau}.
\]
非例外项由局部门估计控制，例外项由 spill 预算控制，因此
\[
\mathsf E_{j+1}
\le
\sum_{\sigma\in\Sigma_j\setminus X_j}
(q_{j,\sigma}\mathfrak N_{j,\sigma}+e_{j,\sigma})+r_j
\le Q_j\mathsf E_j+G_j+r_j.
\]
对该一阶递推从 \(0\) 到 \(J-1\) 迭代，即得到所示公式。

现在证明失败分类。所谓证书链被接受，按定义正是所有非例外 packet 的五类原子审计条件与所有例外 spill 预算同时成立。若链被拒绝，由有限性可取最小拒绝尺度 \(j\)。在该尺度上，若所有非例外 \(\sigma\) 的五类原子条件均成立且例外 spill 亦不超出 \(r_j\)，则尺度 \(j\) 不会被拒绝，矛盾。因此必有某个原子条件失败。逐项展开定义，失败原子只能是地址/相位投影、局部门估计、sector/slab 预算、sticky 子地址纤维、prefix/Frostman 预算或例外 spill 预算之一。每一项均给出相应的有限见证，例如违规子元素、违规标签纤维、违规地址纤维、违规测试集或违规例外总和。证毕。
\end{proof}
