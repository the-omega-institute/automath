\begin{proposition}[正则语言 dyadic polycube 的 Stokes--dyadic 通量 \texorpdfstring{$\zeta$}{zeta}：分式延拓与极点谱]\label{prop:spg-regular-language-stokes-dyadic-zeta-rationality}
固定 \(n\ge 1\)，令字母表 \(\Sigma:=\{0,1\}^n\)。
对长度 \(m\) 的词 \(w\in\Sigma^m\)，令 \(Q(w)\subset I^n\) 表示其对应的边长 \(2^{-m}\) 的 dyadic \(n\)-立方体（按 dyadic 地址的标准编码）。
给定正则语言 \(L\subset\Sigma^\ast\)，记
\[
L_m:=L\cap\Sigma^m,
\qquad
N_m(L):=\card{L_m},
\qquad
U_m(L):=\bigcup_{w\in L_m}Q(w)\subset I^n.
\]
令 \(\omega_0\) 为 \eqref{eq:spg-universal-stokes-probe-form} 中满足 \(\dd\omega_0=\dd x_1\cdots \dd x_n\) 的通用 Stokes 探针。
则 \(\operatorname{Vol}(U_m(L))=N_m(L)\,2^{-mn}\)，并且由 \eqref{eq:spg-probe-volume-identity} 有
\[
\int_{\partial U_m(L)}\omega_0=\operatorname{Vol}(U_m(L))=N_m(L)\,2^{-mn}.
\]
据此定义 Stokes--dyadic 通量 \(\zeta\) 函数
\[
\zeta_L(s):=\sum_{m\ge 0}2^{ms}\int_{\partial U_m(L)}\omega_0
=\sum_{m\ge 0}N_m(L)\,2^{m(s-n)}.
\]
设 \(L\) 的确定型有限自动机存在 \(r\) 维线性表示 \(N_m(L)=u^{\mathsf T}M^m v\)，其中
\(
M\in\ZZ_{\ge 0}^{r\times r}
\)
为转移计数矩阵，\(u,v\in\{0,1\}^r\) 分别为初态与终态指示向量。
则：
\begin{enumerate}
  \item \textbf{（全平面亚纯分式延拓）}
        在 \(\Re(s)\) 足够小使级数收敛的半平面内，有严格分式表达
        \[
        \boxed{\;
        \zeta_L(s)=u^{\mathsf T}\bigl(I-2^{s-n}M\bigr)^{-1}v\;}
        \]
        且右端作为 \(2^{s-n}\) 的有理函数给出 \(\zeta_L\) 在全平面上的亚纯延拓。
  \item \textbf{（极点谱）}
        其极点集包含于
        \[
        \mathcal P_L:=\left\{\,s\in\CC:\ \det\!\bigl(I-2^{s-n}M\bigr)=0\,\right\}.
        \]
        等价地，若 \(\lambda\neq 0\) 遍历 \(M\) 的特征值，则 \(\mathcal P_L\) 由方程 \(2^{s-n}=1/\lambda\) 描述。
  \item \textbf{（primitive 情形的主极点与 dyadic 维/内容）}
        若 \(M\) 为 primitive 矩阵，记 Perron 根 \(\rho:=\rho(M)>0\)，并令 \(l,r\) 为对应的左/右 Perron 本征向量，规范化为 \(l^{\mathsf T}r=1\)。
        则存在常数 \(C>0\) 与次谱模 \(|\lambda_2|<\rho\) 使对一切 \(m\ge 0\) 有
        \[
        N_m(L)=\bigl(u^{\mathsf T}r\bigr)\bigl(l^{\mathsf T}v\bigr)\rho^m+O\!\bigl(|\lambda_2|^m\bigr),
        \]
        从而令
        \[
        d_{\mathrm{dyad}}(L):=\log_2\rho
        \]
        则 dyadic Minkowski 维（沿 dyadic 壳层）满足 \(d_{\mathrm{dyad}}(L)=\log_2\rho\)，并且 dyadic Minkowski 内容极限存在且具有闭式
        \[
        \mathcal M_2(L):=\lim_{m\to\infty}2^{m(n-d_{\mathrm{dyad}}(L))}\operatorname{Vol}(U_m(L))
        =\bigl(u^{\mathsf T}r\bigr)\bigl(l^{\mathsf T}v\bigr).
        \]
        此时 \(\zeta_L(s)\) 在
        \(
        s_\ast:=n-d_{\mathrm{dyad}}(L)=n-\log_2\rho
        \)
        处具有单极点，并满足留数律
        \[
        \operatorname{Res}_{s=s_\ast}\zeta_L(s)=\frac{\mathcal M_2(L)}{\log 2}.
        \]
        进一步存在常数 \(C'>0\) 使指数收敛成立：
        \[
        \left|2^{m(n-d_{\mathrm{dyad}}(L))}\int_{\partial U_m(L)}\omega_0-\mathcal M_2(L)\right|
        \le C'\left(\frac{|\lambda_2|}{\rho}\right)^m.
        \]
\end{enumerate}
\end{proposition}
\begin{proof}
由 \(\int_{\partial U_m(L)}\omega_0=\operatorname{Vol}(U_m(L))\) 与 \(\operatorname{Vol}(U_m(L))=N_m(L)\,2^{-mn}\)，得
\[
\zeta_L(s)=\sum_{m\ge 0}u^{\mathsf T}M^m v\cdot 2^{m(s-n)}.
\]
当 \(\Re(s)\) 足够小使得谱半径条件 \(\rho(M)\,|2^{s-n}|<1\) 成立时，矩阵几何级数给出
\(
\sum_{m\ge 0}(2^{s-n}M)^m=(I-2^{s-n}M)^{-1}
\)，
从而得到 (1) 的分式表达；右端的亚纯延拓性与极点包含关系由分母行列式零点给出，得到 (2)。
\par
对 (3)，primitive 条件下 Perron--Frobenius 定理给出分解
\(
M^m=\rho^m\,r\,l^{\mathsf T}+O(|\lambda_2|^m)
\)。
左乘 \(u^{\mathsf T}\)、右乘 \(v\) 得到 \(N_m(L)\) 的主项展开；
再代回 \(\operatorname{Vol}(U_m(L))=N_m(L)\,2^{-mn}\) 与 \(d_{\mathrm{dyad}}(L)=\log_2\rho\) 得 dyadic 维与内容闭式，并推出指数收敛。
\par
最后，对极点与留数：由主项
\(
\operatorname{Vol}(U_m(L))=\mathcal M_2(L)\,2^{-m(n-d_{\mathrm{dyad}}(L))}+O((|\lambda_2|/\rho)^m\,2^{-m(n-d_{\mathrm{dyad}}(L))})
\)
可将 \(\zeta_L(s)\) 分解为一个几何级数主项与在 \(s=s_\ast\) 邻域解析的余项。
主项给出单极点与留数律，其形式与命题 \ref{prop:spg-dyadic-flux-zeta-minkowski-pole} 的证明同型。证毕。
\end{proof}

\begin{remark}[有限状态线性表示的统一口径]\label{rem:spg-fsa-linear-representation-zeta-rationality}
上述分式表达只依赖于 \(m\mapsto N_m\) 的线性表示 \(N_m=u^{\mathsf T}M^m v\)。
因此，对任何具有确定型有限状态表示的 dyadic 占据序列（包括 sofic/确定型表示下的占据计数），其 Stokes--dyadic 通量 \(\zeta\) 均自动满足全平面亚纯分式延拓与极点谱描述。
\end{remark}
