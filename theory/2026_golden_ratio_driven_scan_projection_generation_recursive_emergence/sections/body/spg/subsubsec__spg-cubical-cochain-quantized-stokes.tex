\paragraph{立方余链的量化 Stokes 分解：\texorpdfstring{$L^\infty$}{L-infty} 最优常数与离散--连续统一}
本段把定理 \ref{thm:spg-cube-homotopy-decomposition-and-stokes-defect} 的连续同伦分解搬运到标准立方分解上的余链复形，并给出与连续情形一致的最优常数。

\begin{theorem}[立方余链的量化 Stokes--Poincar\'e 引理与最优常数]\label{thm:spg-cubical-quantized-stokes-poincare-linf-optimal}
令 \(I^n=[0,1]^n\) 取标准定向与标准立方分解，并令 \((C^\bullet(I^n;\RR),d)\) 为其立方余链复形。
对 \(\omega\in C^k(I^n;\RR)\) 定义上确界范数
\[
\|\omega\|_\infty:=\max_{\sigma^{(k)}}|\omega(\sigma)|,
\]
其中最大值取遍全部定向 \(k\)-胞腔。
则对任意 \(0\le k\le n-1\) 与任意 \(\omega\in C^k(I^n;\RR)\)，存在
\(\eta\in C^{k-1}(I^n;\RR)\) 与 \(E\in C^{k}(I^n;\RR)\) 使得
\begin{equation}\label{eq:spg-cubical-stokes-decomposition}
\omega=d\eta+E,\qquad dE=d\omega,
\end{equation}
并且满足一致的最优控制
\begin{equation}\label{eq:spg-cubical-stokes-defect-optimal-constant}
\|E\|_\infty\le \frac{1}{2(k+1)}\,\|d\omega\|_\infty.
\end{equation}
此外，常数 \(\frac{1}{2(k+1)}\) 在“线性右逆”意义下不可改进：若 \(n\ge k+1\)，则对任意线性算子
\(\mathcal K:C^{k+1}(I^n;\RR)\to C^{k}(I^n;\RR)\)
只要满足 \(d\mathcal K=\mathrm{Id}\) 在 \(d(C^{k}(I^n;\RR))\) 上成立，必有
\[
\|\mathcal K\|_{\infty\to\infty}\ge \frac{1}{2(k+1)}.
\]
\leanverified{paper\_spg\_cubical\_quantized\_stokes\_poincare\_linf\_optimal}
\end{theorem}
\begin{proof}
取 \(\Omega^k_{\mathrm{sm}}(I^n)\) 为 \(I^n\) 邻域上的光滑 \(k\)-形式限制到 \(I^n\) 的空间，并沿用 \S\ref{sec:spg} 中的同伦算子 \(K_\ell\)（定义 \eqref{eq:spg-cube-homotopy-operator-Kk}）与其估计
\[
\|K_{\ell}\alpha\|_\infty\le \frac{1}{2\ell}\,\|\alpha\|_\infty
\qquad(\alpha\in\Omega^{\ell}_{\mathrm{sm}}(I^n)),
\]
以及同伦恒等式 \(\dd(K_{k+1}\beta)=\beta-K_{k+2}(\dd\beta)\)。
\par
在标准立方分解上取 de Rham--Whitney 型互逆对：
令 \(W:C^k(I^n;\RR)\to\Omega^k_{\mathrm{sm}}(I^n)\) 为立方 Whitney 映射，
令 \(\mathcal I:\Omega^k_{\mathrm{sm}}(I^n)\to C^k(I^n;\RR)\) 为胞腔积分映射
\(
(\mathcal I\alpha)(\sigma):=\int_{\sigma}\alpha
\)
（按标准体积归一化）。
它们满足链映射与互逆关系
\[
\mathcal I\circ W=\mathrm{Id},\qquad
\dd\circ W=W\circ d,\qquad
\mathcal I\circ \dd=d\circ \mathcal I,
\]
并且可取算子范数界 \(\|W\|_{\infty\to\infty}\le 1\)、\(\|\mathcal I\|_{\infty\to\infty}\le 1\)。
\par
定义线性算子
\[
\mathcal H_{k+1}:=\mathcal I\circ K_{k+1}\circ W:\ C^{k+1}(I^n;\RR)\to C^{k}(I^n;\RR).
\]
由链映射关系与连续同伦恒等式可得离散同伦恒等式
\[
d\mathcal H_{k+1}+\mathcal H_{k+2}d=\mathrm{Id}\qquad\text{在 }C^{k+1}(I^n;\RR)\text{ 上成立}.
\]
于是对给定 \(\omega\in C^k(I^n;\RR)\)，令
\[
\eta:=\mathcal H_{k}(\omega),\qquad E:=\mathcal H_{k+1}(d\omega),
\]
便得到 \eqref{eq:spg-cubical-stokes-decomposition}。
并且
\[
\|E\|_\infty
\le \|\mathcal I\|_{\infty\to\infty}\,\|K_{k+1}(W(d\omega))\|_\infty
\le \frac{1}{2(k+1)}\,\|W(d\omega)\|_\infty
\le \frac{1}{2(k+1)}\,\|d\omega\|_\infty,
\]
得到 \eqref{eq:spg-cubical-stokes-defect-optimal-constant}。
\par
最优性取标准嵌入的 \((k+1)\)-立方胞腔 \(\tau\subset I^n\) 并取 \(\theta\in C^{k+1}(I^n;\RR)\) 为仅在 \(\tau\) 上取值 \(1\) 的余链。
若 \(d\mathcal K(\theta)=\theta\) 在 \(d(C^k)\) 上成立，则
\[
1=\theta(\tau)=d(\mathcal K\theta)(\tau)=\sum_{F\subset\partial\tau}\pm(\mathcal K\theta)(F),
\]
其中 \(\partial\tau\) 含 \(2(k+1)\) 个 \(k\)-面。
由三角不等式得
\(
1\le 2(k+1)\|\mathcal K\theta\|_\infty
\le 2(k+1)\|\mathcal K\|_{\infty\to\infty}\|\theta\|_\infty
=2(k+1)\|\mathcal K\|_{\infty\to\infty}
\)，
从而 \(\|\mathcal K\|_{\infty\to\infty}\ge \frac{1}{2(k+1)}\)。证毕。
\end{proof}

\begin{remark}[中心立方与立方余链记号的等价改写]\label{rem:spg-cubical-centered-csquare-rewrite}
将立方体由 \(I^n=[0,1]^n\) 改写为中心立方 \(\widetilde I^n=[-\tfrac12,\tfrac12]^n\) 仅涉及平移同胚；
标准立方分解与定向在该平移下自然对应，从而余链复形与上确界范数 \(\|\cdot\|_\infty\) 等距同构。
因此，若以 \(C^k_{\square}(\widetilde I^n;\RR)\) 表示中心立方剖分的 \(k\)-余链空间，并以 \(\delta\) 表示余边界，则定理 \ref{thm:spg-cubical-quantized-stokes-poincare-linf-optimal} 可表述为存在线性算子族
\[
H_k:C^k_{\square}(\widetilde I^n;\RR)\to C^{k-1}_{\square}(\widetilde I^n;\RR),\qquad H_0=0,
\]
使得对任意 \(0\le k\le n-1\) 与任意 \(c\in C^k_{\square}(\widetilde I^n;\RR)\) 成立分解
\[
c=\delta(H_kc)+E(c),\qquad E(c):=H_{k+1}(\delta c),
\]
并满足最优估计
\[
\|E(c)\|_\infty\le \frac{1}{2(k+1)}\|\delta c\|_\infty.
\]
其中可取 \(H_k\) 为证明中由连续同伦算子与 Whitney--积分互逆对所诱导的离散同伦算子 \(\mathcal H_k\) 在上述等距同构下的对应物。
\end{remark}

\begin{corollary}[超立方体图上的势一致化与方形缺陷的最优线性控制]\label{cor:spg-hypercube-gradient-consistency-by-square-defect}
令 \(\Omega_m=\{0,1\}^m\) 视为标准 \(m\)-维立方复形 \(I^m\) 的 \(0\)-骨架，并以其 \(1\)-胞腔（边）与 \(2\)-胞腔（方形）诱导余链复形 \((C^\bullet(I^m;\RR),d)\)。
对任意边权 \(\omega\in C^1(I^m;\RR)\) 记
\(
\varepsilon:=\|d\omega\|_\infty
\)，其中 \(d\omega\in C^2(I^m;\RR)\) 逐方形给出环和缺陷。
则存在顶点势 \(f\in C^0(I^m;\RR)\) 与残差 \(r\in C^1(I^m;\RR)\) 使得
\[
\omega=df+r,\qquad \|r\|_\infty\le \frac14\,\varepsilon,
\qquad dr=d\omega,
\]
并且常数 \(\frac14\) 为最优。
\leanverified{stokes\_k1\_denominator}
\leanverified{stokes\_denominator\_strict\_mono}
\leanverified{gradient\_residual\_bound\_seed}
\leanverified{hypercube\_cell\_counts}
\leanverified{paper\_spg\_hypercube\_gradient\_consistency\_by\_square\_defect}
\end{corollary}
\begin{proof}
对定理 \ref{thm:spg-cubical-quantized-stokes-poincare-linf-optimal} 取 \(k=1\) 即得分解 \(\omega=d\eta+E\) 且 \(\|E\|_\infty\le \frac14\|d\omega\|_\infty\)。
取 \(f:=\eta\)、\(r:=E\) 即得所示结论；最优性为定理 \ref{thm:spg-cubical-quantized-stokes-poincare-linf-optimal} 的 \(k=1\) 特例。证毕。
\end{proof}

\begin{theorem}[超立方体连续时间链的近详细平衡：方形曲率的最优控制]\label{thm:spg-hypercube-near-detailed-balance-optimal}
令 \(\Omega_m=\{0,1\}^m\) 为标准 \(m\)-维立方复形 \(I^m\) 的 \(0\)-骨架，并令 \(q:\Omega_m\times\Omega_m\to(0,\infty)\) 为一族连续时间跳跃率，满足 \(q(u,v)>0\) 当且仅当 \(u\sim v\)。
定义反对称边余链
\[
\omega(u,v):=\log\frac{q(u,v)}{q(v,u)}\in C^1(I^m;\RR),
\qquad \omega(v,u)=-\omega(u,v),
\]
并令方形环和缺陷为 \(d\omega\in C^2(I^m;\RR)\)，记
\(
\varepsilon:=\|d\omega\|_\infty
\)。
则存在势函数 \(f\in C^0(I^m;\RR)\) 使得对一切边 \(u\sim v\) 有
\[
\bigl|\omega(u,v)-(f(v)-f(u))\bigr|\le \frac{\varepsilon}{4}.
\]
从而令 \(\pi_f(u):=\exp(f(u))\) 得到逐边乘法型近详细平衡：
\[
\exp\!\Bigl(-\frac{\varepsilon}{4}\Bigr)
\le
\frac{\pi_f(u)\,q(u,v)}{\pi_f(v)\,q(v,u)}
\le
\exp\!\Bigl(\frac{\varepsilon}{4}\Bigr),
\qquad \forall\,u\sim v.
\]
并且常数 \(\frac14\) 为最优：存在仅支撑在单个方形 \(2\)-胞腔边界上的 \(\omega\)（从而可由某 \(q\) 实现）使得对任意 \(f\) 皆有
\(
\|\omega-df\|_\infty\ge \varepsilon/4
\)。
\leanverified{near\_detailed\_balance\_log\_identity}
\leanverified{near\_detailed\_balance\_optimality\_seed}
\leanverified{paper\_spg\_hypercube\_near\_detailed\_balance\_optimal}
\end{theorem}
\begin{proof}
由推论 \ref{cor:spg-hypercube-gradient-consistency-by-square-defect}，存在 \(f\) 使得 \(\|\omega-df\|_\infty\le \varepsilon/4\)，即对每条边 \(u\sim v\) 有
\(
|\omega(u,v)-(f(v)-f(u))|\le \varepsilon/4
\)。
又
\[
\log\frac{\pi_f(u)\,q(u,v)}{\pi_f(v)\,q(v,u)}
=
f(u)+\log q(u,v)-f(v)-\log q(v,u)
=
\omega(u,v)-(f(v)-f(u)),
\]
故指数化得到所示双侧界。
\par
最优性由推论 \ref{cor:spg-hypercube-gradient-consistency-by-square-defect} 的最优常数断言直接推出；并且给定任意该推论中的极端例 \(\omega\)，取一对称正跳跃率实现它（例如对每条定向边设 \(q(u,v):=\exp(\omega(u,v))\)、\(q(v,u):=1\)），即可将该 \(\omega\) 解释为某连续时间链的对数非平衡一形式。证毕。
\end{proof}
