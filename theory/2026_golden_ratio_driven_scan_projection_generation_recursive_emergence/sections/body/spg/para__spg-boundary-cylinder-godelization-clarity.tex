% Gödel arithmetization for boundary cylinders and clarity exponent readout.

\begin{theorem}[边界柱 Gödelization 的双对数维与清晰度指数的算术刻画]\label{thm:spg-boundary-cylinder-godelization-clarity}
在 \S\ref{subsec:spg-scan-clarity} 的设定下，固定 \(T\)-不变概率测度 \(\mu\)，并令 \(\mathsf P\in\mathcal F_\infty\)。
对分辨率 \(m\) 的柱划分 \(\mathcal C_m\)（见命题 \ref{prop:spg-scan-error-cylinder}）定义边界柱族
\[
\partial_m\mathsf P
:=\left\{C\in\mathcal C_m:\ \mu(\mathsf P\cap C)>0,\ \mu(C\setminus \mathsf P)>0\right\},
\qquad
N_m(\partial\mathsf P):=\#\partial_m\mathsf P.
\]
固定素数序列 \((p_k)_{k\ge 1}\)。
对每个 \(m\) 取任意双射 \(\iota_m:\mathcal C_m\to \{1,\dots,\#\mathcal C_m\}\)，并定义边界柱 Gödel 整数
\[
G_m^{\partial}(\mathsf P):=\prod_{C\in\partial_m\mathsf P} p_{\iota_m(C)}\in\NN.
\]
若存在柱复杂度底 \(\lambda_{\mathrm{cyl}}>1\) 使 \(\#\{C\in\mathcal C_m:\mu(C)>0\}\asymp \lambda_{\mathrm{cyl}}^m\)，则有算术维等价
\[
\overline{\dim}_B(\partial\mathsf P)
\ =\
\limsup_{m\to\infty}\frac{\log\log G_m^{\partial}(\mathsf P)}{m\log\lambda_{\mathrm{cyl}}}.
\]
并且在定理 \ref{thm:spg-clarity-volume-boundary-scaling} 的厚度假设下，若存在指数 \(d\) 使 \(N_m(\partial\mathsf P)=\lambda_{\mathrm{cyl}}^{dm+o(m)}\)，则清晰度指数可被纯算术读出：
\[
\lim_{m\to\infty}\left(-\frac{1}{m}\log_{\lambda_{\mathrm{cyl}}}\varepsilon_m(\mathsf P;\mu)\right)=1-d
\quad\Longleftrightarrow\quad
d=\limsup_{m\to\infty}\frac{\log\log G_m^{\partial}(\mathsf P)}{m\log\lambda_{\mathrm{cyl}}}.
\]
\end{theorem}
\begin{proof}
记 \(N:=N_m(\partial\mathsf P)\) 且 \(M:=\#\mathcal C_m\)。
由于 \(\partial_m\mathsf P\) 由 \(N\) 个互异柱构成，\(G_m^{\partial}(\mathsf P)\) 为 \(N\) 个互异素数之积，并且其素数指标被 \(M\) 夹逼；因此有
\[
\prod_{k=1}^{N}p_k
\ \le\
G_m^{\partial}(\mathsf P)
\ \le\
p_M^{\,N}.
\]
由素数定理的等价形式
\(
\log\!\bigl(\prod_{k=1}^{N}p_k\bigr)=(1+o(1))\,N\log N
\)
与
\(
\log p_M=\log M+O(\log\log M)
\)
得到
\[
\log\log G_m^{\partial}(\mathsf P)
\ge
\log\!\bigl((1+o(1))\,N\log N\bigr)
=
\log N+O(\log\log N),
\]
\[
\log\log G_m^{\partial}(\mathsf P)
\le
\log\!\bigl(N(\log M+O(\log\log M))\bigr)
=
\log N+O(\log\log M).
\]
在柱复杂度底假设下，\(\log M\asymp m\log\lambda_{\mathrm{cyl}}\)，从而 \(\log\log M=O(\log m)\)。
于是存在常数 \(A,B>0\) 使对充分大 \(m\) 有
\[
\bigl|\log\log G_m^{\partial}(\mathsf P)-\log N_m(\partial\mathsf P)\bigr|
\le
A\log m+B.
\]
将上式除以 \(m\log\lambda_{\mathrm{cyl}}\) 并取 \(\limsup\) 即得算术维等价。
\par
清晰度指数部分由定理 \ref{thm:spg-clarity-volume-boundary-scaling} 的指数律
\(
\lim_{m\to\infty}-\frac1m\log_{\lambda_{\mathrm{cyl}}}\varepsilon_m(\mathsf P;\mu)=1-d
\)
与上述把 \(d\) 以 \(\log\log G_m^{\partial}(\mathsf P)\) 表达的等价直接推出。证毕。
\end{proof}
