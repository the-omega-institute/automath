% Second-order term of primorial double-log recovers dyadic Minkowski content.

\begin{theorem}[Gödel 双对数的二阶项恢复 dyadic Minkowski 含量]\label{thm:spg-godel-doublelog-second-order-minkowski-content}
设 \(A\subset I^n=[0,1]^n\)，并令 \(U_m(A)\) 为尺度 \(2^{-m}\) 的 dyadic 外逼近。
记
\[
N_m(A):=\card{\{Q\in\mathcal Q_m:\ Q\subset U_m(A)\}},
\]
并假设存在常数 \(d\in[0,n]\) 与 \(M\in(0,\infty)\) 使得
\[
N_m(A)=M\,2^{md}+o(2^{md})\qquad(m\to\infty),
\]
即 \(A\) 的 dyadic Minkowski 维为 \(d\)，且 dyadic Minkowski 含量为 \(M\)。
令 \((p_k)_{k\ge 1}\) 为素数序列，并定义含量归一化的 primorial Gödel 整数
\[
G_m^\ast(A):=\prod_{k=1}^{N_m(A)} p_k.
\]
则成立：
\begin{enumerate}
  \item 双对数斜率读出维度
  \[
  \lim_{m\to\infty}\frac{\log\log G_m^\ast(A)}{m\log 2}=d.
  \]
  \item 若 \(d>0\)，则双对数二阶项恢复含量
  \[
  M=\lim_{m\to\infty}\frac{\exp\!\bigl(\log\log G_m^\ast(A)-md\log 2\bigr)}{m\,d\,\log 2}.
  \]
\end{enumerate}
\leanverified{paper\_spg\_godel\_doublelog\_second\_order\_minkowski\_content}
\end{theorem}
\begin{proof}
记 \(P_N:=\prod_{k=1}^{N}p_k\)。
由素数定理的等价形式可得渐近
\[
\log\log P_N=\log N+\log\log N+o(1)\qquad(N\to\infty).
\]
取 \(N=N_m(A)\)，并由假设 \(N_m(A)=M2^{md}(1+o(1))\) 得
\[
\log N_m(A)=md\log 2+\log M+o(1),
\qquad
\log\log N_m(A)=\log m+\log(d\log 2)+o(1)\qquad(d>0).
\]
代入得到
\[
\log\log G_m^\ast(A)
=\log\log P_{N_m(A)}
=md\log 2+\log M+\log m+\log(d\log 2)+o(1).
\]
将其除以 \(m\log 2\) 并令 \(m\to\infty\) 得第 (1) 条。
对第 (2) 条，移项并指数化得到
\[
\exp\!\bigl(\log\log G_m^\ast(A)-md\log 2\bigr)
=M\cdot m\cdot d\log 2\cdot (1+o(1)),
\]
从而所示极限成立。证毕。
\end{proof}
