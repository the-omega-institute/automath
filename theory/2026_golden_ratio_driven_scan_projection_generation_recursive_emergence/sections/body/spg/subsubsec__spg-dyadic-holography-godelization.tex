\subsubsection{dyadic 立方复形的边界单射与 Gödel 字典}\label{subsubsec:spg-dyadic-holography-godelization}
定理 \ref{thm:spg-dyadic-outer-approx-stokes-gain-minkowski-readout} 已给出：dyadic 外近似 \(U_m(A)\) 的体积可由单一 Stokes 探针 \(\omega_0\) 的边界通量精确读出。
本小节进一步补齐两条与“可审计算术化”直接对齐的离散结论：
其一是固定 dyadic 立方复形下顶维边界算子的单射性（顶维无同调），从而体链由边界链唯一决定；
其二是将 dyadic 占据向量做平方自由 Gödel 编码后，其双对数增长率与占据数的指数率一致，从而把“边界 \(\to\) 体积 \(\to\) Gödel 整数 \(\to\) 指数”闭合为一条可复算链。

\begin{theorem}[dyadic 立方复形上的边界--体积强全息：顶维边界算子单射]\label{thm:spg-dyadic-cubical-boundary-injective}
令 \(I^n=[0,1]^n\)。
对任意 \(m\ge 1\)，记 \(\mathcal Q_m\) 为将 \(I^n\) 按边长 \(2^{-m}\) 的 dyadic \(n\)-立方体剖分得到的有限 \(n\)-维立方复形；
令 \(C_k(\mathcal Q_m;\ZZ)\) 为其取定取向的 \(k\)-维胞腔链群，\(\partial_k:C_k\to C_{k-1}\) 为胞腔边界算子。
则顶维边界算子
\[
\partial_n:C_n(\mathcal Q_m;\ZZ)\longrightarrow C_{n-1}(\mathcal Q_m;\ZZ)
\]
为单射。
因此，对任意 dyadic polycube（即由 \(\mathcal Q_m\) 的 \(n\)-胞腔以系数 \(0/1\) 取并所对应的 \(n\)-链）\(U\)，其边界链 \(\partial_n[U]\) 唯一决定 \([U]\)；
特别地，对外近似 \(U_m(A)\)（定义 \ref{thm:spg-dyadic-outer-approx-stokes-gain-minkowski-readout}）而言，\(\partial U_m(A)\) 在链意义上唯一确定 \(U_m(A)\)。
\end{theorem}
\begin{proof}
作为胞腔复形，\(|\mathcal Q_m|=I^n\) 可缩，故其胞腔同调满足 \(H_n(\mathcal Q_m;\ZZ)=0\)。
又由于 \(\mathcal Q_m\) 无 \((n+1)\)-胞腔，故 \(C_{n+1}(\mathcal Q_m;\ZZ)=0\) 并且 \(\mathrm{im}(\partial_{n+1})=0\)。
由定义
\[
H_n(\mathcal Q_m;\ZZ)=\ker(\partial_n)/\mathrm{im}(\partial_{n+1})=\ker(\partial_n),
\]
从而 \(\ker(\partial_n)=0\)，即 \(\partial_n\) 单射。
其余断言由单射性直接推出。证毕。
\end{proof}

\begin{corollary}[顶维全息精确性：体链与界循环的同构]\label{cor:spg-dyadic-topdim-holography-exactness}
在定理 \ref{thm:spg-dyadic-cubical-boundary-injective} 的设定下，令
\[
Z_{n-1}(\mathcal Q_m;\ZZ):=\ker\bigl(\partial_{n-1}:C_{n-1}(\mathcal Q_m;\ZZ)\to C_{n-2}(\mathcal Q_m;\ZZ)\bigr)
\]
为 \((n-1)\)-循环群。
则边界算子诱导群同构
\[
\partial_n:\ C_n(\mathcal Q_m;\ZZ)\ \xrightarrow{\ \cong\ }\ Z_{n-1}(\mathcal Q_m;\ZZ).
\]
此外，若以标准网格计数，则
\[
\rank C_n(\mathcal Q_m;\ZZ)=2^{nm},\qquad
\rank C_{n-1}(\mathcal Q_m;\ZZ)=n(2^m+1)\,2^{(n-1)m},
\]
从而 \(Z_{n-1}(\mathcal Q_m;\ZZ)\) 在 \(C_{n-1}(\mathcal Q_m;\ZZ)\) 中的余维为
\[
\rank C_{n-1}-\rank Z_{n-1}
=(n-1)2^{nm}+n2^{(n-1)m}.
\]
\end{corollary}
\begin{proof}
由于 \(|\mathcal Q_m|=I^n\) 可缩，故 \(H_{n-1}(\mathcal Q_m;\ZZ)=0\)，从而
\(
\ker(\partial_{n-1})=\operatorname{im}(\partial_n)
\)。
与定理 \ref{thm:spg-dyadic-cubical-boundary-injective} 给出的 \(\partial_n\) 单射性合并，即得
\(\partial_n:C_n(\mathcal Q_m;\ZZ)\to Z_{n-1}(\mathcal Q_m;\ZZ)\)
为双射，从而为同构。
\par
秩计数来自：dyadic \(n\)-立方体数为 \((2^m)^n=2^{nm}\)；
而 \((n-1)\)-面数为 \(n\) 个法向方向，每个方向包含 \((2^m+1)\) 个切片位置与 \(2^{(n-1)m}\) 个横截面网格单元，故总数为 \(n(2^m+1)2^{(n-1)m}\)。
余维公式由 \(\rank Z_{n-1}=\rank C_n\) 代入得到。
证毕。
\end{proof}

\begin{theorem}[Stokes--Gödel 双对数维：体积外逼近的 Gödel 整数化指数率仅由占据数决定]\label{thm:spg-stokes-godel-doublelog-dimension}
沿用 dyadic 外近似 \(U_m(A)\subset I^n\) 的记号，并令
\[
N_m(A):=\#\{Q\in\mathcal Q_m:\ Q\subset U_m(A)\}
=2^{mn}\operatorname{Vol}(U_m(A)),
\]
其中 \(\operatorname{Vol}\) 为 Lebesgue 体积。
固定素数序列 \((p_j)_{j\ge 1}\)。
将 \(U_m(A)\) 的 dyadic 占据向量 \(b^{(m)}\in\{0,1\}^{2^{mn}}\)（按任意固定枚举识别 \(\mathcal Q_m\)）作平方自由 Gödel 编码
\[
G_m(A):=\prod_{j=1}^{2^{mn}} p_j^{\,b^{(m)}_j}\in\NN.
\]
则当 \(m\to\infty\) 时成立渐近恒等式
\[
\log\log G_m(A)=\log N_m(A)+O(\log m).
\]
从而定义
\[
d_G(A):=\limsup_{m\to\infty}\frac{\log\log G_m(A)}{m\log 2},
\qquad
d_V(A):=\limsup_{m\to\infty}\frac{\log N_m(A)}{m\log 2},
\]
有 \(d_G(A)=d_V(A)\)。
\end{theorem}
\begin{proof}
记 \(S_m:=\{j:\ b^{(m)}_j=1\}\)，则 \(|S_m|=N_m(A)\) 且
\[
\log G_m(A)=\sum_{j\in S_m}\log p_j.
\]
由于 \(\log p_j\ge \log 2\) 对一切 \(j\) 成立，得下界
\[
\log G_m(A)\ge N_m(A)\log 2.
\]
另一方面，对 \(j\le 2^{mn}\) 有 \(\log p_j\le \log p_{2^{mn}}\)，故
\[
\log G_m(A)\le N_m(A)\log p_{2^{mn}}.
\]
取对数得到
\[
\log N_m(A)+\log\log 2
\le
\log\log G_m(A)
\le
\log N_m(A)+\log\!\Big(\frac{\log p_{2^{mn}}}{\log 2}\Big).
\]
由素数的粗上界（例如对充分大 \(k\) 有 \(p_k\le k^2\)），可得
\(
\log p_{2^{mn}}\le 2mn\log 2
\)，从而 \(\log(\log p_{2^{mn}})=O(\log m)\)。
于是得到
\(
\log\log G_m(A)=\log N_m(A)+O(\log m)
\)。
两边除以 \(m\log 2\) 并取 \(\limsup\) 得 \(d_G(A)=d_V(A)\)。证毕。
\end{proof}

\begin{theorem}[Gödel--素数编码的密度全息与中心极限定理]\label{thm:spg-godel-prime-density-holography-clt}
令 \((p_n)_{n\ge 1}\) 为素数序列。
对二元序列 \(x=(x_n)_{n\ge 1}\in\{0,1\}^{\NN}\)，定义其长度 \(n\) 的 Gödel--素数编码与权重和
\[
G_n(x):=\prod_{j=1}^{n}p_j^{\,x_j},
\qquad
\Theta_n:=\sum_{j=1}^{n}\log p_j.
\]
\begin{enumerate}
  \item \textbf{密度全息（加权极限等于通常密度）.}
        若通常密度极限 \(d(x):=\lim_{n\to\infty}\frac1n\sum_{j=1}^n x_j\) 存在，则
        \[
        \lim_{n\to\infty}\frac{\log G_n(x)}{\Theta_n}=d(x).
        \]
  \item \textbf{i.i.d. 情形的高斯涨落（对数编码的 CLT）.}
        若 \(X_1,X_2,\dots\) 独立同分布且 \(X_j\sim \mathrm{Bernoulli}(p)\)（\(0<p<1\)），令
        \[
        S_n:=\log G_n(X)=\sum_{j=1}^n X_j\log p_j,
        \qquad
        \sigma_n^2:=\Var(S_n)=p(1-p)\sum_{j=1}^n (\log p_j)^2,
        \]
        则
        \[
        \frac{S_n-p\Theta_n}{\sigma_n}\ \Longrightarrow\ \mathcal N(0,1).
        \]
        并且 \(\sigma_n\asymp \sqrt n\,\log n\)，故涨落规模为 \(\sqrt n\log n\)。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{（1）}写
\[
\log G_n(x)-d(x)\Theta_n=\sum_{j=1}^n (x_j-d(x))\log p_j.
\]
记 \(A_m:=\sum_{j=1}^m (x_j-d(x))\)，则由密度极限存在性得 \(A_m=o(m)\)。
令 \(b_j:=\log p_j\)（单调递增），对 \(\sum_{j=1}^n (x_j-d)b_j\) 作 Abel 变换得
\[
\sum_{j=1}^n (x_j-d)b_j=A_n b_n-\sum_{j=1}^{n-1}A_j(b_{j+1}-b_j).
\]
由素数定理 \(p_n\sim n\log n\) 推出 \(b_n=\log n+\log\log n+o(1)\)，并且 \(b_{j+1}-b_j=O(1/j)\)。
于是 \(A_n b_n=o(n)\cdot O(\log n)=o(n\log n)\)，且
\[
\sum_{j=1}^{n-1}A_j(b_{j+1}-b_j)
\;=\;
\sum_{j=1}^{n-1}o(j)\cdot O(1/j)
\;=\;
o(n).
\]
另一方面 \(\Theta_n\sim n\log n\)，故
\(
\bigl(\log G_n(x)-d(x)\Theta_n\bigr)/\Theta_n\to 0
\)，即所示极限成立。
\par
\textbf{（2）}令 \(Y_j:=(X_j-p)\log p_j\)，则 \(\EE[Y_j]=0\)，且 \(\sum_{j\le n}\Var(Y_j)=\sigma_n^2\)。
又 \(\max_{j\le n}|Y_j|\le \max_{j\le n}\log p_j\sim \log n\)。
并且
\[
\sigma_n^2=p(1-p)\sum_{j=1}^n(\log p_j)^2\asymp \sum_{j=1}^n(\log j)^2\asymp n(\log n)^2,
\]
从而 \(\max_{j\le n}|Y_j|/\sigma_n\to 0\)，Lindeberg 条件成立，故
\(
\sum_{j\le n}Y_j/\sigma_n\Rightarrow \mathcal N(0,1)
\)。
注意 \(\sum_{j\le n}Y_j=S_n-p\Theta_n\)，即得中心极限定理。证毕。
\end{proof}

\begin{corollary}[Gödel 比值的指数相干判别：密度差导出 \texorpdfstring{$n\log n$}{n log n} 级证书]\label{cor:spg-godel-prime-ratio-density}
设 \(x,y\in\{0,1\}^{\NN}\) 的通常密度 \(d(x),d(y)\) 均存在且 \(d(x)\neq d(y)\)。
则
\[
\log\frac{G_n(x)}{G_n(y)}
=
\bigl(d(x)-d(y)\bigr)\Theta_n+o(\Theta_n),
\]
从而
\[
\left|\frac{G_n(x)}{G_n(y)}\right|
=
\exp\Bigl(\bigl(d(x)-d(y)\bigr)\Theta_n+o(\Theta_n)\Bigr),
\qquad
\Theta_n\sim n\log n.
\]
\end{corollary}
\begin{proof}
对 \(x,y\) 分别应用定理 \ref{thm:spg-godel-prime-density-holography-clt}(1) 并相减即可。证毕。
\end{proof}

\begin{theorem}[边界 Gödelization 的可逆全息字典：由边界素因子分解重构 dyadic polycube]\label{thm:spg-boundary-godelization-holographic-dictionary}
固定 \(m,n\) 并令 \(\mathcal F_{m}^{n-1}\) 为 \(\mathcal Q_m\) 的全部 \((n-1)\)-胞腔（面）。
对每一条有向 \((n-1)\)-面 \(\vec f\) 赋予互异素数 \(q_{\vec f}\)。
对任意 dyadic polycube \(U\subset I^n\)，其边界链可写为
\[
\partial_n[U]=\sum_{\vec f\in \overrightarrow{\mathcal F_{m}^{n-1}}} \varepsilon_{\vec f}\,\vec f,
\qquad \varepsilon_{\vec f}\in\{0,1\},
\]
其中 \(\varepsilon_{\vec f}=1\) 表示该有向面在边界中以正系数出现（若以负系数出现，则等价地由其反向取向面承载）。
据此定义边界 Gödel 码
\[
H_m(U):=\prod_{\vec f} q_{\vec f}^{\,\varepsilon_{\vec f}}\in\NN.
\]
则映射 \(U\mapsto H_m(U)\) 在 dyadic polycube 全体上为单射。
并且 \(U\) 可由 \(H_m(U)\) 的素因子分解显式重构：分解给出 \(\partial_n[U]\) 的取向面支撑，再由 \(\partial_n\) 的单射性得到唯一的 \(n\)-链 \([U]\)。
\end{theorem}
\begin{proof}
由素因子分解唯一性，\(H_m(U)\) 唯一确定指数族 \((\varepsilon_{\vec f})\)，从而唯一确定边界链 \(\partial_n[U]\)。
由定理 \ref{thm:spg-dyadic-cubical-boundary-injective}，\(\partial_n\) 单射，故 \(\partial_n[U]\) 唯一确定 \([U]\)，进而确定 dyadic polycube \(U\)。
构造性重构过程即为：对 \(H_m(U)\) 作素因子分解得到 \(\partial_n[U]\) 的取向面支撑，随后以定理 \ref{thm:spg-linear-time-bulk-decode-from-boundary} 的线性时间译码显式求得唯一 \(x=[U]\)。证毕。
\end{proof}

\paragraph{连续边界通量的形参重构与 Gödel 轴译码}
离散 dyadic 链层的“边界单射”依赖于顶维同调的消失。
在连续几何模型中，对严格凸域的边界通量同样存在可逆译码机制：在若干可计算的形参族上，有限阶边界多项式通量即为形参的完备不变量。
下述结果把“Stokes 探针的体积读出”提升为“二阶通量的形状矩阵读出”，并给出对角 Gödel 轴编码的显式可逆译码与噪声阈值。

\begin{theorem}[椭球形状矩阵的边界通量重构]\label{thm:spg-ellipsoid-boundary-flux-reconstruction}
令 \(d\ge 2\)，并令 \(Q\in\mathrm{Sym}_d^{+}\) 为实对称正定矩阵。
定义椭球及其边界
\[
E_Q:=\{x\in\RR^d:\ x^{\mathsf T}Q^{-1}x\le 1\},
\qquad
S_Q:=\partial E_Q.
\]
记 \(n(x)\) 为 \(S_Q\) 上外法单位法向，\(dS\) 为 \((d-1)\)-维面积测度，并定义边界标量与边界二阶通量矩阵
\[
B_0:=\int_{S_Q}\langle x,n(x)\rangle\,dS,
\qquad
B:=(B_{ij})_{1\le i,j\le d},\qquad
B_{ij}:=\int_{S_Q}x_ix_j\langle x,n(x)\rangle\,dS.
\]
则成立完全重构关系
\[
\operatorname{Vol}(E_Q)=\frac{1}{d}B_0,
\qquad
B=\operatorname{Vol}(E_Q)\,Q,
\qquad
Q=\frac{d}{B_0}\,B.
\]
因此 \(Q\) 的谱决定椭球在正交基下的半轴长度平方。
\end{theorem}
\begin{proof}
取向量场 \(X(x):=x\)，则 \(\mathrm{div}\,X=d\)，由散度定理得
\[
B_0=\int_{S_Q}\langle x,n\rangle\,dS=\int_{E_Q}\mathrm{div}\,X\,dx=d\,\operatorname{Vol}(E_Q),
\]
从而 \(\operatorname{Vol}(E_Q)=B_0/d\)。
\par
固定 \(i,j\)，取向量场 \(X^{ij}(x):=x_ix_j\,x\)。
逐分量计算得 \(\mathrm{div}\,X^{ij}=(d+2)x_ix_j\)，故
\[
B_{ij}=\int_{S_Q}\langle X^{ij},n\rangle\,dS=\int_{E_Q}\mathrm{div}\,X^{ij}\,dx
=(d+2)\int_{E_Q}x_ix_j\,dx.
\]
取 \(A\in\mathrm{GL}_d(\RR)\) 使 \(Q=AA^{\mathsf T}\)，则 \(E_Q=A(B_d)\)，其中 \(B_d:=\{y\in\RR^d:\ |y|_2\le 1\}\)。
令 \(x=Ay\)，得
\[
\int_{E_Q}xx^{\mathsf T}\,dx
=\det(A)\,A\Bigl(\int_{B_d}yy^{\mathsf T}\,dy\Bigr)A^{\mathsf T}.
\]
由旋转对称性，\(\int_{B_d}yy^{\mathsf T}\,dy=c\,I_d\)，并由取迹得到
\(
c\,d=\int_{B_d}|y|_2^2\,dy=\frac{d}{d+2}\operatorname{Vol}(B_d)
\)，故 \(c=\operatorname{Vol}(B_d)/(d+2)\)。
因此
\[
\int_{E_Q}xx^{\mathsf T}\,dx
=\frac{\det(A)\operatorname{Vol}(B_d)}{d+2}\,AA^{\mathsf T}
=\frac{\operatorname{Vol}(E_Q)}{d+2}\,Q.
\]
代回 \(B_{ij}=(d+2)\int_{E_Q}x_ix_j\,dx\) 得 \(B=\operatorname{Vol}(E_Q)\,Q\)，从而 \(Q=(d/B_0)B\)。证毕。
\end{proof}

\begin{corollary}[对角 Gödel 轴编码的边界可逆译码]\label{cor:spg-godel-axis-boundary-invertibility}
取互异素数 \(p_1,\dots,p_d\) 与指数 \(b_1,\dots,b_d\in\ZZ_{\ge 0}\)，并令
\[
Q=\mathrm{diag}\bigl(p_1^{2b_1},\dots,p_d^{2b_d}\bigr),
\qquad
E:=E_Q.
\]
则仅由边界通量数据 \(\bigl(B_0,B_{11},\dots,B_{dd}\bigr)\) 可恢复
\[
N:=\prod_{i=1}^d p_i^{b_i}=\frac{B_0}{d\,\operatorname{Vol}(B_d)},
\qquad
b_i=\frac{1}{2\log p_i}\log\!\left(\frac{d\,B_{ii}}{B_0}\right)\qquad(1\le i\le d).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:spg-ellipsoid-boundary-flux-reconstruction}，有 \(\operatorname{Vol}(E)=B_0/d\) 且 \(Q=(d/B_0)B\)，故
\[
\frac{d\,B_{ii}}{B_0}=Q_{ii}=p_i^{2b_i},
\]
取对数即得 \(b_i\)。
又由 \(E=A(B_d)\) 且 \(\det(A)=\prod_i p_i^{b_i}\)，有
\(
\operatorname{Vol}(E)=\operatorname{Vol}(B_d)\prod_i p_i^{b_i}
\)，
从而
\(
N=\operatorname{Vol}(E)/\operatorname{Vol}(B_d)=B_0/(d\,\operatorname{Vol}(B_d))
\)。
证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\ell_p$}{lp}-超椭球族的边界--体积译码恒等式]\label{thm:spg-lp-superellipsoid-boundary-volume-decoding}
令 \(d\ge 2\)，\(p>1\)，并取 \(a=(a_1,\dots,a_d)\in(0,\infty)^d\)。
定义 \(\ell_p\)-超椭球与其边界
\[
\Omega_p(a):=\Bigl\{x\in\RR^d:\ \sum_{i=1}^d\Bigl|\frac{x_i}{a_i}\Bigr|^p\le 1\Bigr\},
\qquad
S_p(a):=\partial\Omega_p(a).
\]
记 \(n\) 为外法单位法向，并定义边界通量矩
\[
B_0(p):=\int_{S_p(a)}\langle x,n\rangle\,dS,
\qquad
B_i(p):=\int_{S_p(a)}|x_i|^p\langle x,n\rangle\,dS\qquad(1\le i\le d).
\]
则对每个 \(i\) 有
\[
\operatorname{Vol}(\Omega_p(a))=\frac{1}{d}B_0(p),
\qquad
B_i(p)=a_i^p\,\operatorname{Vol}(\Omega_p(a)),
\qquad
a_i^p=\frac{d\,B_i(p)}{B_0(p)}.
\]
\end{theorem}
\begin{proof}
第一式同定理 \ref{thm:spg-ellipsoid-boundary-flux-reconstruction} 的体积恒等式：对 \(X(x)=x\) 用散度定理即得。
\par
对第二式，取向量场 \(X_i(x):=|x_i|^p x\)。
由于 \(p>1\)，函数 \(t\mapsto |t|^p\) 为 \(C^1\)，从而 \(X_i\) 足够正则。
逐分量计算得 \(\mathrm{div}\,X_i=(d+p)|x_i|^p\)，故
\[
B_i(p)=\int_{S_p(a)}\langle X_i,n\rangle\,dS
=\int_{\Omega_p(a)}\mathrm{div}\,X_i\,dx
=(d+p)\int_{\Omega_p(a)}|x_i|^p\,dx.
\]
令 \(U_p:=\Omega_p(1,\dots,1)\)，并作缩放 \(x=\mathrm{diag}(a_1,\dots,a_d)y\)，得
\[
\int_{\Omega_p(a)}|x_i|^p\,dx
=\Bigl(\prod_{k=1}^d a_k\Bigr)a_i^p\int_{U_p}|y_i|^p\,dy,
\qquad
\operatorname{Vol}(\Omega_p(a))
=\Bigl(\prod_{k=1}^d a_k\Bigr)\operatorname{Vol}(U_p).
\]
令 \(f(y)=\sum_{k=1}^d|y_k|^p\)，则 \(U_p=\{f\le 1\}\)。
由层析积分公式 \(\int_{U_p} f\,dy=\int_0^1 \operatorname{Vol}(\{y\in U_p:\ f(y)>t\})\,dt\)。
注意 \(\{f\le t\}=t^{1/p}U_p\)（\(t\in[0,1]\)）且体积按 \(t^{d/p}\) 缩放，
故 \(\operatorname{Vol}(\{f>t\})=\operatorname{Vol}(U_p)(1-t^{d/p})\)，从而
\[
\int_{U_p} f\,dy
=\operatorname{Vol}(U_p)\int_0^1(1-t^{d/p})\,dt
=\operatorname{Vol}(U_p)\frac{d}{d+p}.
\]
由对称性 \(\int_{U_p}|y_1|^p=\cdots=\int_{U_p}|y_d|^p\)，且 \(\int f\,dy=\sum_{i=1}^d\int|y_i|^p\,dy\)，得
\(
\int_{U_p}|y_i|^p\,dy=\operatorname{Vol}(U_p)/(d+p)
\)。
代回即得
\(
\int_{\Omega_p(a)}|x_i|^p\,dx=\frac{a_i^p}{d+p}\operatorname{Vol}(\Omega_p(a))
\)，
再代回 \(B_i(p)=(d+p)\int |x_i|^p\) 得 \(B_i(p)=a_i^p\operatorname{Vol}(\Omega_p(a))\)。证毕。
\end{proof}

\begin{proposition}\label{prop:spg-lp-superellipsoid-decoding-multiplicative-noise-stability}
在定理 \ref{thm:spg-lp-superellipsoid-boundary-volume-decoding} 的设定下，
设观测到扰动通量 \(\widetilde B_0(p),\widetilde B_i(p)\) 满足相对误差界
\[
(1-\varepsilon)B_0(p)\le \widetilde B_0(p)\le (1+\varepsilon)B_0(p),
\qquad
(1-\varepsilon)B_i(p)\le \widetilde B_i(p)\le (1+\varepsilon)B_i(p),
\qquad (0<\varepsilon<1).
\]
并用同一译码规则定义
\[
\widetilde a_i^p:=\frac{d\,\widetilde B_i(p)}{\widetilde B_0(p)}.
\]
则对每个 \(i\) 有对数稳定性估计
\[
\left|\log\!\left(\frac{\widetilde a_i^p}{a_i^p}\right)\right|
\le
\log\!\left(\frac{1+\varepsilon}{1-\varepsilon}\right),
\qquad
\left|\log\!\left(\frac{\widetilde a_i}{a_i}\right)\right|
\le
\frac{1}{p}\log\!\left(\frac{1+\varepsilon}{1-\varepsilon}\right).
\]
进一步，在 Gödel 特化 \(a_i^p=p_i^{r_i}\)（互异素数 \(p_i\) 与指数 \(r_i\in\RR\)）下，
令 \(\widetilde r_i:=\log_{p_i}(\widetilde a_i^p)\)，则
\[
|\widetilde r_i-r_i|
\le
\frac{1}{\log p_i}\log\!\left(\frac{1+\varepsilon}{1-\varepsilon}\right).
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:spg-lp-superellipsoid-boundary-volume-decoding}，
\(
a_i^p=d\,B_i(p)/B_0(p)
\)。
因此
\[
\frac{\widetilde a_i^p}{a_i^p}
=\frac{\widetilde B_i(p)/B_i(p)}{\widetilde B_0(p)/B_0(p)}.
\]
由于 \(\widetilde B_i(p)/B_i(p)\) 与 \(\widetilde B_0(p)/B_0(p)\) 均落在区间 \([1-\varepsilon,1+\varepsilon]\)，有
\[
\frac{1-\varepsilon}{1+\varepsilon}
\le
\frac{\widetilde a_i^p}{a_i^p}
\le
\frac{1+\varepsilon}{1-\varepsilon}.
\]
取对数并取绝对值得第一条不等式；第二条由
\(
\log(\widetilde a_i/a_i)=\frac{1}{p}\log(\widetilde a_i^p/a_i^p)
\)
立得。
最后一条由
\(
\widetilde r_i-r_i=\log_{p_i}(\widetilde a_i^p/a_i^p)=\frac{1}{\log p_i}\log(\widetilde a_i^p/a_i^p)
\)
直接推出。证毕。
\end{proof}

\begin{proposition}[加权 \texorpdfstring{$\ell_1$}{l1} 环面上的稳定长度与 Gödel 对数]\label{prop:spg-weighted-l1-torus-stable-length-godel}
令 \(w_i=\log p_i>0\)，并在环面 \(T^d=\RR^d/\ZZ^d\) 上考虑平移不变的 Finsler 结构
\[
|v|_w:=\sum_{i=1}^d w_i|v_i|,
\qquad v\in T_xT^d\simeq\RR^d.
\]
对 \(m\in H_1(T^d;\ZZ)\simeq\ZZ^d\)，定义稳定长度
\[
L_w(m):=\inf\Bigl\{\int_0^1|\gamma'(t)|_w\,dt:\ \gamma\ \text{为代表 }m\text{ 的闭 Lipschitz 曲线}\Bigr\}.
\]
则
\[
L_w(m)=\sum_{i=1}^d w_i|m_i|.
\]
特别地，对任意 \(b\in\{0,1\}^d\)，有
\[
L_w(b)=\sum_{i:b_i=1}\log p_i
=\log\Bigl(\prod_{i=1}^d p_i^{b_i}\Bigr).
\]
\end{proposition}
\begin{proof}
取常系数闭 \(1\)-形式
\[
\alpha:=\sum_{i=1}^d \mathrm{sgn}(m_i)\,w_i\,d\theta_i,
\]
其中 \(\theta_i\) 为标准环面坐标。
对任意切向量 \(v\) 有
\[
\alpha(v)=\sum_i \mathrm{sgn}(m_i)w_iv_i\le \sum_i w_i|v_i|=|v|_w,
\]
故对任意代表 \(m\) 的 \(\gamma\) 有
\[
\int_0^1|\gamma'(t)|_w\,dt\ge \int_\gamma \alpha=\sum_i \mathrm{sgn}(m_i)w_im_i=\sum_i w_i|m_i|.
\]
另一方面，取分段沿坐标方向的闭路，分别绕第 \(i\) 个圆周 \(|m_i|\) 次，即取到等号，从而得到 \(L_w(m)=\sum_i w_i|m_i|\)。证毕。
\end{proof}

\begin{proposition}[6D 超立方体的 Gödel 平方类模型：算术化与度量保持]\label{prop:spg-hypercube6-godel-squareclass}
令 \(d=6\)，并以按位加法赋予 \(\{0,1\}^{6}\cong(\ZZ/2\ZZ)^{6}\) 群结构。
取前 \(6\) 个素数 \(p_1,\dots,p_6\)，并定义映射
\[
\Phi:\{0,1\}^6\longrightarrow \QQ^\times/(\QQ^\times)^2,
\qquad
\Phi(v):=\left[\prod_{j=1}^{6}p_j^{\,v_j}\right],
\]
其中 \([\cdot]\) 表示平方类。
则：
\begin{enumerate}
  \item \(\Phi\) 为群同构：\(\{0,1\}^6\) 同构到由 \([p_1],\dots,[p_6]\) 生成的平方类子群；
  \item 若在平方类上定义素支撑距离
        \[
        d_{\square}(a,b):=\card{\{p\in\{p_1,\dots,p_6\}:\ p\ \text{在 }ab^{-1}\text{ 的平方自由代表中出现}\}},
        \]
        则对任意 \(u,v\in\{0,1\}^6\) 有
        \[
        d_{\square}\bigl(\Phi(u),\Phi(v)\bigr)=d_H(u,v),
        \]
        其中 \(d_H\) 为 Hamming 距离。
\end{enumerate}
\end{proposition}
\begin{proof}
在平方类群中指数按模 \(2\) 约化，故
\[
\Phi(u)\Phi(v)=\left[\prod_{j}p_j^{u_j+v_j}\right]
=\left[\prod_{j}p_j^{u_j\oplus v_j}\right]
=\Phi(u\oplus v),
\]
从而 \(\Phi\) 为群同态；其核仅含零向量，故为单射，有限群单射即为同构，得第 (1) 条。
对第 (2) 条，\(ab^{-1}\) 的平方自由代表中出现 \(p_j\) 当且仅当 \(u_j\neq v_j\)，其计数即为 Hamming 距离。证毕。
\end{proof}

\begin{theorem}[紧边界连续加性译码的平凡性]\label{thm:spg-compact-boundary-continuous-decoder-trivial}
设 \(G\) 为紧拓扑群，\(\psi:G\to(\RR,+)\) 为连续群同态，则 \(\psi\equiv 0\)。
因此，若要求“边界 \(\to\) Gödel 寄存器”的译码器是紧边界对称群上的连续加性可观测量，则该译码必为平凡；
要实现对无界 Gödel 数据的精确抽取，必然迫使译码在边界上失去连续性，或经由非紧覆盖引入可承载非平凡 \(\RR\)-角色的圈因子结构（从而在圆维语义下出现正的可见通道）。
\end{theorem}
\begin{proof}
\(\psi(G)\subset\RR\) 为紧子群，而 \((\RR,+)\) 的紧子群仅为 \(\{0\}\)，故 \(\psi(G)=\{0\}\) 并得 \(\psi\equiv 0\)。
其余断言为直接推论。证毕。
\end{proof}

\begin{theorem}[Gödel 边界译码的相对误差阈值]\label{thm:spg-godel-boundary-decoding-relative-error-threshold}
在推论 \ref{cor:spg-godel-axis-boundary-invertibility} 的设定下，设可观测边界通量满足相对误差界 \(\varepsilon\in(0,1)\)：
\[
\widetilde B_0=B_0(1+\delta_0),
\qquad
\widetilde B_{ii}=B_{ii}(1+\delta_i),
\qquad
|\delta_0|,|\delta_i|\le \varepsilon.
\]
令 \(\lfloor \cdot\rceil\) 表示最近整数。
定义译码
\[
\widetilde b_i:=\left\lfloor \frac{1}{2\log p_i}\log\!\left(\frac{d\,\widetilde B_{ii}}{\widetilde B_0}\right)\right\rceil.
\]
若
\[
\varepsilon<\frac{p_{\min}-1}{p_{\min}+1},
\qquad
p_{\min}:=\min_{1\le i\le d}p_i,
\]
则必有 \(\widetilde b_i=b_i\) 对所有 \(i\) 成立。
\end{theorem}
\begin{proof}
记真实比值 \(r_i:=dB_{ii}/B_0=p_i^{2b_i}\) 与观测比值 \(\widetilde r_i:=d\widetilde B_{ii}/\widetilde B_0\)。
由误差界得
\[
\frac{1-\varepsilon}{1+\varepsilon}\,r_i
\le
\widetilde r_i
\le
\frac{1+\varepsilon}{1-\varepsilon}\,r_i.
\]
最近整数译码的充分条件为 \(\widetilde r_i\in(r_i/p_i,\ r_ip_i)\)，等价于
\(
\frac{1+\varepsilon}{1-\varepsilon}<p_i
\)。
因此只需 \(\frac{1+\varepsilon}{1-\varepsilon}<p_{\min}\)，即 \(\varepsilon<\frac{p_{\min}-1}{p_{\min}+1}\)，便得 \(\widetilde b_i=b_i\)。
证毕。
\end{proof}
