\paragraph{锯齿（fence）偏序与理想格}
\begin{definition}[锯齿偏序与理想格]\label{def:pom-fence-poset}
对 \(\ell\ge 0\)，定义 \textbf{锯齿偏序}（fence poset）\(Z_\ell\) 为底集 \(\{1,\dots,\ell\}\) 上的偏序，
其覆盖关系沿线性顺序交替定向：
\[
1\prec 2 \succ 3\prec 4 \succ \cdots.
\]
对有限偏序 \(P\)，记 \(J(P)\) 为其\textbf{下闭集}（order ideal）按包含关系组成的格（\(\wedge=\cap\)，\(\vee=\) 取下闭包后并）。
\end{definition}

\begin{theorem}[纤维的 Birkhoff 完备表示：纤维为锯齿理想格]\label{thm:pom-fiber-birkhoff-fence-ideal-lattice}
固定 \(x\in X_m\)，设可逆位点诱导图分解为路径并
\[
\Gamma_x\ \cong\ \bigsqcup_{i=1}^r P_{\ell_i}.
\]
则在自然的“单次逆代换开关可达性”偏序下，纤维 \(F_x\) 可被赋予有限分布式格结构 \(\mathcal L_x\)，并存在规范同构
\[
\mathcal L_x\ \cong\ J\!\Big(\bigsqcup_{i=1}^r Z_{\ell_i}\Big).
\]
特别地，\(\mathcal L_x\) 的 join-不可约元的偏序（Birkhoff 表示的基底）同构于
\(\bigsqcup_i Z_{\ell_i}\)，因此多重集 \(\{\ell_i\}\) 完全决定纤维内部的合取/析取结构与区间结构。
\end{theorem}
\leanverified{paper\_pom\_fiber\_birkhoff\_fence\_ideal\_lattice}
\begin{proof}
对单分量 \(P_\ell\)，注意：在任意有限偏序 \(P\) 上，映射
\[
\mathsf{Ant}(P)\ \longleftrightarrow\ J(P),
\qquad
A\mapsto \downarrow A,\ \ I\mapsto \max(I)
\]
给出反链（antichain）与下闭集之间的规范双射（\(\downarrow A\) 为下闭包，\(\max(I)\) 为极大元集合）。
而对锯齿偏序 \(Z_\ell\)，其可比性仅发生在相邻点，
故反链 \(\mathsf{Ant}(Z_\ell)\) 恰等价于路径图 \(P_\ell\) 上的独立集 \(\mathsf{Ind}(P_\ell)\)（“不可同时取相邻点”）。
于是得到规范双射 \(\mathsf{Ind}(P_\ell)\cong J(Z_\ell)\)。
\par
通过该双射把 \(J(Z_\ell)\) 的包含序与 \(\wedge,\vee\) 运算拉回到 \(\mathsf{Ind}(P_\ell)\)（再经定理 \ref{thm:pom-fiber-indset-factorization} 拉回到 \(F_x\)），
即可在纤维上定义出分布式格结构 \(\mathcal L_x\)。
\par
对不交并 \(\Gamma_x=\bigsqcup_i P_{\ell_i}\)，独立集系统分解为笛卡尔乘积，
而理想格满足 \(J(\bigsqcup_i P_i)\cong\prod_i J(P_i)\)。
合并两步并用定理 \ref{thm:pom-fiber-indset-factorization} 给出的纤维同构，即得所述分布式格表示。证毕。
\end{proof}

\begin{corollary}[rank 多项式的乘积因子化与 Fibonacci 归一化]\label{cor:pom-fiber-fence-rank-poly-factorization}
以理想大小 \(|I|\) 为秩函数，定义纤维格的 rank 多项式
\[
R_x(q):=\sum_{I\in \mathcal L_x} q^{|I|}.
\]
则
\[
R_x(q)=\prod_{i=1}^r R_{\ell_i}(q),
\qquad
R_\ell(q):=\sum_{I\in J(Z_\ell)} q^{|I|}.
\]
并且在 \(q=1\) 处恢复 Fibonacci 计数：\(|J(Z_\ell)|=R_\ell(1)=F_{\ell+2}\)，从而
\(

\)
（与定理 \ref{thm:pom-fiber-indset-factorization} 的基数闭式一致）。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-fiber-birkhoff-fence-ideal-lattice} 的直积分解，
\(\mathcal L_x\cong\prod_i J(Z_{\ell_i})\) 且秩函数可加，生成函数因子化成立。
最后，\(|J(Z_\ell)|=|\mathsf{Ind}(P_\ell)|=F_{\ell+2}\) 为路径独立集计数的标准恒等式。证毕。

\end{proof}

\paragraph{区间闭包与 Möbius 刚性}
\begin{theorem}[区间闭包：任意区间仍为锯齿理想格]\label{thm:pom-fence-interval-closure}
设 \(\mathcal L_x\cong J(P_x)\) 且 \(P_x=\bigsqcup_i Z_{\ell_i}\)。
对任意区间 \([I,J]\subseteq \mathcal L_x\)，令
\[
P_{I,J}:=J\setminus I\ \subseteq\ P_x
\]
并赋以 \(P_x\) 的诱导偏序。
则存在自然格同构
\[
[I,J]\ \cong\ J(P_{I,J}).
\]
并且由于 \(P_x\) 为锯齿并，诱导子偏序 \(P_{I,J}\) 必分解为若干锯齿偏序的离散并（即仍落回 fence 范畴）。
\end{theorem}
\leanverified{paper\_pom\_fence\_interval\_closure}
\begin{proof}
对一般 \(J(P)\)，映射 \(\Phi:[I,J]\to J(J\setminus I)\) 由 \(\Phi(K)=K\setminus I\) 给出，
其逆为 \(L\mapsto I\cup L\)。直接核验可得 \(\Phi\) 为格同构。
当 \(P_x\) 为锯齿并时，诱导子偏序的 Hasse 图为路径森林，从而其连通分量仍为若干锯齿。证毕。
\end{proof}

\begin{theorem}[Möbius 刚性：\texorpdfstring{\(\mu\in\{0,\pm1\}\)}{mu in {0,±1}}]\label{thm:pom-fence-mobius-rigidity}
沿用定理 \ref{thm:pom-fence-interval-closure} 的记号。
则区间 Möbius 数满足
\[
\mu_{\mathcal L_x}(I,J)=
\begin{cases}
(-1)^{|P_{I,J}|},& P_{I,J}\ \text{为反链（antichain）},\\
0,& \text{否则}.
\end{cases}
\]
因此对任意 \([I,J]\subseteq\mathcal L_x\)，有 \(\mu_{\mathcal L_x}(I,J)\in\{0,\pm 1\}\)。
该取值规律给出一条纯离散的区间证书：\(\mu(I,J)\neq 0\) 当且仅当该区间等价于一个布尔格（即区间内的 join-不可约元彼此不可比）。
\end{theorem}
\leanverified{paper\_pom\_fence\_mobius\_rigidity}
\begin{proof}
由定理 \ref{thm:pom-fence-interval-closure}，\(\mu_{\mathcal L_x}(I,J)=\mu_{J(P_{I,J})}(\hat0,\hat1)\)。
而对一般分布式格 \(J(P)\)，区间 \([\hat0,\hat1]\) 为布尔格当且仅当 \(P\) 为反链；
此时 \(J(P)\cong \{0,1\}^{|P|}\) 且 \(\mu(\hat0,\hat1)=(-1)^{|P|}\)。
若 \(P\) 含可比对，则 \(J(P)\) 含非平凡三元链，从而 \(\mu(\hat0,\hat1)=0\)。
合并即得。证毕。
\end{proof}

\begin{remark}[与纤维带符号指数的对比：两类 \texorpdfstring{\(0/\pm 1\)}{0/±1} 刚性]\label{rem:pom-fence-mobius-vs-signed-index}
定理 \ref{thm:pom-fence-mobius-rigidity} 给出的 Möbius 刚性是\emph{格论区间}层面的 \(\{0,\pm 1\}\) 取值规律，
其非零性由区间是否为布尔格（等价于差分子偏序是否为反链）完全判别。
另一方面，在 \(\mathsf{Ind}(P_\ell)\) 的“逆代换次数”统计上，纤维带符号指数不变量
\(\sigma_m(x)=Z_x(-1)\)（\S\ref{subsubsec:single-slice-reflector-completeness} 中的结论组，亦见命题 \ref{prop:pom-fiber-index-cgf} 一节）
同样满足 \(\{0,\pm 1\}\) 刚性，但其消失判别来自路径独立集多项式在 \(-1\) 处的 6 周期/模 \(3\) 阻碍。
因此两类 \(\{0,\pm 1\}\) 不变量分别刻画“区间布尔性”与“路径独立集相消相位”，互补地服务于纤维的离散审计。
\end{remark}

\paragraph{多链谱：zeta 多项式、序多项式与 \texorpdfstring{\(\pi\)}{pi} 常数}
\begin{definition}[zeta 多项式与序多项式]\label{def:pom-zeta-order-polynomial}
对有界有限格 \(L\)，定义其 zeta 多项式 \(Z_L(k)\) 为从 \(\hat0\) 到 \(\hat1\) 的 \(k\)-段多链数：
\[
Z_L(k):=\#\{\hat0=I_0\subseteq I_1\subseteq \cdots \subseteq I_k=\hat1\}.
\]
对有限偏序 \(P\)，记其序多项式（order polynomial）为
\[
\Omega_P(k):=\#\{f:P\to\{1,\dots,k\}:\ f \ \text{保序}\}.
\]
\end{definition}

\begin{theorem}[分布式格的 zeta/序多项式等价]\label{thm:pom-zeta-equals-order-poly}
对任意有限偏序 \(P\)，其理想格满足
\[
Z_{J(P)}(k)=\Omega_P(k)\qquad(k\in\NN).
\]
\end{theorem}
\leanverified{paper\_pom\_zeta\_equals\_order\_poly}
\begin{proof}
给定多链 \(\hat0=I_0\subseteq\cdots\subseteq I_k=\hat1\)，定义 \(f:P\to\{1,\dots,k\}\) 为
\(
f(p):=\min\{j:\ p\in I_j\}
\)。
由下闭性可核验 \(f\) 保序；反向由 \(f\) 定义 \(I_j:=\{p:\ f(p)\le j\}\) 得到多链。
两构造互逆，故计数相等。证毕。
\end{proof}

\end{proof}

\paragraph{最大链枚举、连续体积与 Ehrhart--Hilbert 细化谱}

\begin{definition}[线性扩张数与最大链数]\label{def:pom-fence-linext-maxchain}
对有限偏序 \(P\)，记 \(e(P):=\#\mathrm{LinExt}(P)\) 为其线性扩张数。
对有限格 \(L\)，记 \(\mathrm{MC}(L)\) 为其最大链（从 \(\hat0\) 到 \(\hat1\) 的饱和链）集合。
\end{definition}

\begin{theorem}[最大链 \(\Longleftrightarrow\) 线性扩张：分布式格的调度等价]\label{thm:pom-maxchains-equals-linext}
对任意有限偏序 \(P\)，其理想格 \(J(P)\) 的最大链与 \(P\) 的线性扩张存在规范双射，
从而
\[
|\mathrm{MC}(J(P))|=e(P).
\]
特别地，对纤维格 \(\mathcal L_x\cong J(P_x)\)（定理 \ref{thm:pom-fiber-birkhoff-fence-ideal-lattice}），
\[
|\mathrm{MC}(\mathcal L_x)|=e(P_x).
\]
\end{theorem}
\begin{proof}
给定最大链 \(\hat0=I_0\subset I_1\subset\cdots\subset I_N=\hat1\)（其中 \(N=|P|\)），
每一步必恰加入一个新元 \(p_t\in P\)，即 \(I_t=I_{t-1}\cup\{p_t\}\)。
下闭性保证加入顺序满足偏序约束，因此 \((p_1,\dots,p_N)\) 为 \(P\) 的线性扩张。
反向地，任意线性扩张按顺序取前缀下闭包即可得到一条最大链。
两构造互逆，得双射与计数等式。证毕。

\end{proof}

\begin{theorem}[锯齿分量的最大链数闭式：Euler 交错数控制]\label{thm:pom-fence-maxchains-euler}
设 \(P_x=\bigsqcup_{i=1}^r Z_{\ell_i}\)，并记 \(N:=\sum_{i=1}^r \ell_i\)。
令 \(E_n\) 为 Euler 交错数（up/down numbers），由指数型生成函数定义：
\[
\sec z+\tan z=\sum_{n\ge 0} E_n\,\frac{z^n}{n!}.
\]
则有严格恒等式
\[
|\mathrm{MC}(\mathcal L_x)|
\;=\;
e(P_x)
\;=\;
\binom{N}{\ell_1,\dots,\ell_r}\cdot \prod_{i=1}^r E_{\ell_i}.
\]
从而纤维的“时间调度复杂度”
\[
\mathsf H_{\mathrm{sched}}(x):=\log |\mathrm{MC}(\mathcal L_x)|
\]
具有可审计分解式
\[
\mathsf H_{\mathrm{sched}}(x)
=\log(N!) -\sum_{i=1}^r \log(\ell_i!) + \sum_{i=1}^r \log(E_{\ell_i}).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-maxchains-equals-linext}，\(|\mathrm{MC}(\mathcal L_x)|=e(P_x)\)。
对不交并偏序 \(P=\bigsqcup_i P_i\)，线性扩张等价于：先对每个分量选择一条线性扩张，再把这些序列按多项式系数 \(\binom{N}{|P_1|,\dots,|P_r|}\) 方式交织（shuffle）。
故 \(e(P)=\binom{N}{|P_1|,\dots,|P_r|}\prod_i e(P_i)\)。
\par
最后，锯齿偏序 \(Z_n\) 的线性扩张数满足 \(e(Z_n)=E_n\)，这是 Euler 交错数的标准等价刻画（由 \(\sec z+\tan z\) 的递推/EGF 或与交错置换的经典双射给出）。
代入 \(P_i=Z_{\ell_i}\) 即得结论。证毕。
\end{proof}
\leanverified{paper\_pom\_fence\_maxchains\_euler}

\begin{corollary}[最大链数的极值结构：全孤立分量最优]\label{cor:pom-fence-maxchains-all-isolated-optimal}
沿用定理 \ref{thm:pom-fence-maxchains-euler} 的记号，并记 \(N=\sum_i \ell_i\)。
则在固定 \(N\) 的约束下，最大链数满足严格上界
\[
|\mathrm{MC}(\mathcal L_x)|
=
N!\prod_{i=1}^r \frac{E_{\ell_i}}{\ell_i!}
\le
N!,
\]
且当且仅当 \(\ell_1=\cdots=\ell_r=1\)（即 \(P_x=\bigsqcup_{i=1}^{N} Z_1\)，等价于全部分量孤立）时取等号。
因此，在固定总长度 \(N\) 的口径下，纤维的“调度自由度”\(|\mathrm{MC}(\mathcal L_x)|\) 由\emph{最碎裂}的分量结构极大化。
\end{corollary}
\begin{proof}
由于 \(E_\ell\) 计数交错置换，故对任意 \(\ell\ge 1\) 有 \(E_\ell\le \ell!\)，且对 \(\ell\ge 2\) 严格小于（交错置换为全排列的真子集）。
代入定理 \ref{thm:pom-fence-maxchains-euler} 的分解式即得
\(
\prod_i(E_{\ell_i}/\ell_i!)\le 1
\)，从而 \(|\mathrm{MC}(\mathcal L_x)|\le N!\)。
等号成立当且仅当每个因子 \(E_{\ell_i}/\ell_i!=1\)，即 \(\ell_i=1\) 对一切 \(i\)。证毕。
\end{proof}

\begin{definition}[order polytope 与连续时间参数化密度]\label{def:pom-order-polytope-volume}
对有限偏序 \(P\)（\(|P|=N\)），定义其 \textbf{order polytope}
\[
\mathcal O(P):=\{y\in[0,1]^P:\ y_p\le y_q\ \text{当}\ p\le_P q\}.
\]
记 \(\mathrm{vol}(\mathcal O(P))\) 为其 \(N\)-维 Lebesgue 体积。
对纤维 \(x\)，定义连续时间参数化密度
\(
\mathsf V(x):=\mathrm{vol}(\mathcal O(P_x)).
\)
\end{definition}

\begin{theorem}[离散—连续同一性：线性扩张数 = 归一化体积]\label{thm:pom-order-polytope-volume-linext}
对任意有限偏序 \(P\)（\(|P|=N\)），有严格恒等式
\[
\mathrm{vol}(\mathcal O(P))=\frac{e(P)}{N!}.
\]
特别地，对 \(P_x=\bigsqcup_i Z_{\ell_i}\)：
\[
\mathsf V(x)=\prod_{i=1}^r \frac{E_{\ell_i}}{\ell_i!},
\qquad
|\mathrm{MC}(\mathcal L_x)|=e(P_x)=N!\,\mathsf V(x).
\]
\end{theorem}
\begin{proof}
对每条线性扩张 \(\pi=(p_1,\dots,p_N)\)，考虑单纯形
\[
\Delta_\pi:=\{y\in[0,1]^P:\ 0\le y_{p_1}\le y_{p_2}\le\cdots\le y_{p_N}\le 1\}.
\]
若 \(\pi\) 为线性扩张，则 \(\Delta_\pi\subseteq \mathcal O(P)\)；
反之，\(\mathcal O(P)\) 可按坐标全序关系分解为这些单纯形的不交并（忽略边界）。
每个 \(\Delta_\pi\) 体积均为 \(1/N!\)，故 \(\mathrm{vol}(\mathcal O(P))=e(P)/N!\)。
对 \(P_x\) 由定理 \ref{thm:pom-fence-maxchains-euler} 代入即得。证毕。
\end{proof}

\begin{theorem}[双不变量可达域的凸包定理：\texorpdfstring{\((\log d,\log\mathrm{MC})\)}{(log d, log MC)} 的 Pareto 几何由原子长度生成]\label{thm:pom-fiber-ab-convex-hull}
固定总长度 \(N\ge 1\)，对任意分量分解 \(N=\sum_{i=1}^r \ell_i\) 定义归一化不变量
\[
A(\{\ell_i\})
:=\frac{1}{N}\sum_{i=1}^r \log F_{\ell_i+2}
=\frac{1}{N}\log d_m(x),
\]
\[
B(\{\ell_i\})
:=\frac{1}{N}\sum_{i=1}^r \log\frac{E_{\ell_i}}{\ell_i!}
=\frac{1}{N}\log \mathsf V(x)
=\frac{1}{N}\bigl(\log|\mathrm{MC}(\mathcal L_x)|-\log(N!)\bigr),
\]
其中 \(d_m(x)=\prod_i F_{\ell_i+2}\) 由定理 \ref{thm:pom-fiber-indset-factorization} 给出，而 \(\mathsf V(x)=\prod_i (E_{\ell_i}/\ell_i!)\) 与 \(|\mathrm{MC}(\mathcal L_x)|=N!\mathsf V(x)\) 由定理 \ref{thm:pom-order-polytope-volume-linext} 给出。
令原子点集
\[
\mathcal P
:=
\left\{
\left(\frac{1}{\ell}\log F_{\ell+2},\ \frac{1}{\ell}\log\frac{E_\ell}{\ell!}\right):\ \ell\in\NN
\right\}
\subset\RR^2,
\]
并令 \(\mathcal R\subset\RR^2\) 为所有由分解 \(\{\ell_i\}\) 产生的点 \((A(\{\ell_i\}),B(\{\ell_i\}))\) 在 \(N\to\infty\) 的尺度极限中所形成的可达域（即取闭包的极限可达点集）。
则有
\[
\mathcal R=\overline{\mathrm{conv}}(\mathcal P).
\]
从而 \((A,B)\) 的 Pareto 前沿等价于 \(\overline{\mathrm{conv}}(\mathcal P)\) 的支撑曲线：对任意支撑方向 \((\alpha,\beta)\in\RR^2\)，极值 \(\sup_{(A,B)\in\mathcal R}(\alpha A+\beta B)\) 由有限多个原子长度 \(\ell\) 所张成的支撑超平面实现。
\end{theorem}
\begin{proof}
对任意分解 \(N=\sum_i \ell_i\)，记权重 \(w_i:=\ell_i/N\)，则 \(\sum_i w_i=1\)，并且
\[
(A(\{\ell_i\}),B(\{\ell_i\}))
=\sum_{i=1}^r w_i\left(\frac{1}{\ell_i}\log F_{\ell_i+2},\ \frac{1}{\ell_i}\log\frac{E_{\ell_i}}{\ell_i!}\right),
\]
故任一有限 \(N\) 的可达点均为 \(\mathcal P\) 的凸组合，从而 \(\mathcal R\subseteq \overline{\mathrm{conv}}(\mathcal P)\)。
反向地，对任意有限凸组合 \(\sum_{j=1}^s \theta_j p_{\ell_j}\)（\(\theta_j\ge 0,\sum_j\theta_j=1\) 且 \(p_{\ell_j}\in\mathcal P\)），取大 \(N\) 并令 \(\ell_j\) 重复出现约 \(\theta_j N/\ell_j\) 次即可构造一列分解使其对应的 \((A,B)\) 逼近该凸组合；再由有限凸组合在闭凸包中的稠密性得到 \(\overline{\mathrm{conv}}(\mathcal P)\subseteq \mathcal R\)。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{\(\log(2/\pi)\)}{log(2/pi)} 常数的极限密度]\label{thm:pom-fence-volume-log2pi}
Euler 交错数满足奇点主导渐近（\(\sec z+\tan z\) 的最近奇点位于 \(z=\pi/2\)），从而
\[
\lim_{n\to\infty}\frac{1}{n}\log\frac{E_n}{n!}=\log\frac{2}{\pi}.
\]
因此在“体相”尺度下（例如当所有 \(\ell_i\to\infty\) 或 \(\sum_{\ell_i\le L}\ell_i=o(N)\) 对任意定值 \(L\) 成立时），有密度收敛
\[
\frac{1}{N}\log\mathsf V(x)
=\sum_{i=1}^r \frac{\ell_i}{N}\,\frac{1}{\ell_i}\log\frac{E_{\ell_i}}{\ell_i!}
\longrightarrow
\log\frac{2}{\pi}.
\]
\end{theorem}
\begin{proof}
第一式为 Euler 交错数的经典渐近结论。由 \(\sec z+\tan z\) 在 \(z=\pm\pi/2\) 的简单极点与标准系数渐近定理得到。
第二式由加权平均与前述极限立得。证毕。
\end{proof}

\begin{remark}[“四分角”最近奇点律与 \texorpdfstring{\(\pi/2\)}{pi/2} 的普适折叠尺度]\label{rem:pom-quarter-turn-pi2-scale}
定理 \ref{thm:pom-fence-volume-log2pi} 中 \(\log(2/\pi)\) 常数来自解析组合的最近奇点支配：\(\sec z+\tan z\) 的最近实奇点位于 \(z=\pm\pi/2\)。
由于三角函数的基本周期为 \(2\pi\)，该奇点对应“全角的四分之一”，即 \(\pi/2=(2\pi)/4\)；因而 \(\log(2/\pi)=-\log(\pi/2)\) 可被理解为一个严格的“四分角奇点律”。
\par
另一方面，由推论 \ref{cor:pom-fence-multichain-entropy-pi} 的闭式
\(
h_{\mathrm{ref}}(k)=-\log\!\bigl(2\sin(\pi/(4k+2))\bigr)
\)，令 \(N:=2k+1\) 则存在完全显式的高阶渐近展开
\[
h_{\mathrm{ref}}(k)
=\log N-\log\pi+\frac{\pi^2}{24N^2}+\frac{\pi^4}{2880N^4}+O(N^{-6}).
\]
因此本文中出现的两类熵常数
\[
-\log\pi,\qquad \log\frac{2}{\pi}=-\log\frac{\pi}{2}
\]
并非彼此独立：它们共享唯一的角尺度 \(\pi/2\)（等价地共享比值 \(2\pi/(\pi/2)=4\)），并在谱细化侧与调度体积侧分别以“半角量子化”与“最近奇点支配”的形式实现。
从而系数 \(4\) 的再现可被统一归因于同一 \(\pi/2\) 机制：在谱侧，它由半角恒等式与半奇整数模态强制给出（注 \ref{rem:pom-Kk-factor4-spectral-rigidity}）；
在解析侧，它由全角周期与最近奇点位置的比值锁定。
\par
在纯谱几何层面，上述“端点正规性 \(\Rightarrow\) 半奇整数模态 \(\Rightarrow\) 半角表示”的链条提供了一个实现无关的结构原型，
从而当在视界端点/圆锥尖点口径下出现 \(1/4\) 型比例常数（例如黑洞熵公式中的 \(1/4\) 系数）时，可将其视为同一角尺度在“半角/四分角”两种等价表述下的严格投影，而不依赖额外物理输入。
\end{remark}

\paragraph{Ehrhart--Hilbert 细化谱：整数细化时间的可审计计数}

\begin{definition}[Ehrhart 多项式]\label{def:pom-ehrhart-order-polytope}
对格多面体 \(Q\subset\RR^N\)，记其 Ehrhart 多项式为
\(
L_Q(t):=\#(tQ\cap\ZZ^N)
\)（\(t\in\ZZ_{\ge 0}\)）。
\end{definition}

\begin{theorem}[order polytope 的 Ehrhart 计数 = 序多项式（带平移）]\label{thm:pom-ehrhart-equals-orderpoly-shift}
\leanverified{paper\_pom\_ehrhart\_equals\_orderpoly\_shift}
对任意有限偏序 \(P\)，有
\[
L_{\mathcal O(P)}(t)=\Omega_P(t+1)\qquad(t\in\ZZ_{\ge 0}),
\]
其中 \(\Omega_P(k)\) 为定义 \ref{def:pom-zeta-order-polynomial} 的序多项式（保序映射计数）。
特别地，
\[
L_{\mathcal O(P)}(1)=\Omega_P(2)=|J(P)|.
\]
\end{theorem}
\begin{proof}
\(t\mathcal O(P)\cap\ZZ^P\) 的点等价于取整数赋值 \(a:P\to\{0,1,\dots,t\}\) 且满足 \(a(p)\le a(q)\)（当 \(p\le_P q\)）。
将其整体平移 \(a\mapsto a+1\) 即得到保序映射 \(P\to\{1,\dots,t+1\}\)，故计数为 \(\Omega_P(t+1)\)。
当 \(t=1\) 时，\(\{0,1\}\)-值保序映射与 order ideal 的指示函数等价，故 \(\Omega_P(2)=|J(P)|\)。证毕。
\end{proof}

\begin{corollary}[一般纤维的 Ehrhart--Hilbert 细化谱乘积闭合]\label{cor:pom-fiber-ehrhart-product}
\leanverified{paper\_pom\_fiber\_ehrhart\_product}
对 \(P_x=\bigsqcup_i Z_{\ell_i}\)，定义纤维细化谱
\[
\mathsf{Ehr}_x(t):=L_{\mathcal O(P_x)}(t)\qquad(t\in\ZZ_{\ge 0}).
\]
则
\[
\mathsf{Ehr}_x(t)=\Omega_{P_x}(t+1)=\prod_{i=1}^r \Omega_{Z_{\ell_i}}(t+1)
=\prod_{i=1}^r L_{\mathcal O(Z_{\ell_i})}(t).
\]
并且在 \(t=1\) 处恢复纤维基数：\(\mathsf{Ehr}_x(1)=|J(P_x)|=|\mathcal L_x|=|F_x|\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-ehrhart-equals-orderpoly-shift}，\(\mathsf{Ehr}_x(t)=\Omega_{P_x}(t+1)\)。
而序多项式对不交并偏序严格乘法分解：\(\Omega_{\bigsqcup_i P_i}(k)=\prod_i\Omega_{P_i}(k)\)，
代入 \(P_i=Z_{\ell_i}\) 即得。最后一式由 \(t=1\) 的指示函数刻画立得。证毕。
\end{proof}

\begin{definition}[\texorpdfstring{\(h^*\)}{h*}-多项式：Hilbert 细化分子]\label{def:pom-hstar-order-polytope}
对维数 \(N\) 的格多面体 \(Q\)，其 Ehrhart 级数可写为
\[
\sum_{t\ge 0} L_Q(t)\,z^t=\frac{h_Q^*(z)}{(1-z)^{N+1}},
\]
其中 \(h_Q^*(z)\in\ZZ[z]\) 为 \(h^*\)-多项式。
对偏序 \(P\)，记 \(h_P^*(z):=h_{\mathcal O(P)}^*(z)\)。
\end{definition}

\begin{theorem}[下降分布表示：\texorpdfstring{\(h^*\)}{h*} 为线性扩张的下降母函数]\label{thm:pom-hstar-descent-linext}
设 \(P\) 为一个\emph{自然标号}偏序（即 \(p\le_P q \Rightarrow p\le q\) 作为整数标号）。
则其 order polytope 的 \(h^*\)-多项式满足
\[
h_P^*(z)=\sum_{\pi\in\mathrm{LinExt}(P)} z^{\mathrm{des}(\pi)},
\qquad
\mathrm{des}(\pi):=\#\{1\le j<N:\ \pi_j>\pi_{j+1}\}.
\]
特别地，\(h_P^*(1)=e(P)=N!\,\mathrm{vol}(\mathcal O(P))\)。
\end{theorem}
\begin{proof}
order polytope 的标准 unimodular 三角剖分可由线性扩张索引（见定理 \ref{thm:pom-order-polytope-volume-linext} 中的单纯形族 \(\{\Delta_\pi\}\)），
其 shelling 的 \(h\)-向量由下降统计给出；对 unimodular 三角剖分，\(h\)-多项式恰等于 \(h^*\)-多项式。
因此 \(h_P^*(z)\) 为线性扩张下降母函数。
代入 \(z=1\) 并用定理 \ref{thm:pom-order-polytope-volume-linext} 得 \(h_P^*(1)=e(P)=N!\mathrm{vol}(\mathcal O(P))\)。证毕。
\end{proof}

\begin{remark}[\texorpdfstring{\(Z_n\)}{Z_n} 的精细不变量：下降结构对 Euler 数的细化]\label{rem:pom-fence-hstar-refinement}
对锯齿偏序 \(Z_n\)，虽其标准标号 \(1\prec 2\succ 3\prec\cdots\) 非自然标号，但可通过选取任意线性扩张重标号得到同构的自然标号偏序；
由于 \(\mathcal O(P)\) 在坐标置换下保持格等价，\(h^*\)-多项式与体积不随该选择改变。
因此 \(h_{Z_n}^*(z)\) 给出 \(E_n=e(Z_n)\) 的一个可审计细化：其系数和为 \(E_n\)，而系数分布记录调度序列的下降结构。
\par
值得强调的是：Euler 交错数 \(E_n\) 同时计数交错置换（alternating permutations），并与 \(Z_n\) 的线性扩张数等势，
但两者集合并非逐项相同；\(h_{Z_n}^*(z)\) 所记录的下降结构是以“自然标号的线性扩张”为参照的细化统计，
从而避免了把交错置换的固定升降模式误读为 \(h^*\) 的（退化）下降分布。
\end{remark}
