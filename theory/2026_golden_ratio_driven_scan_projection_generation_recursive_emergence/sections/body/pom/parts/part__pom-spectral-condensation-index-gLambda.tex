\begin{definition}[谱凝聚指数]\label{def:pom-spectral-condensation-index-gLambda}
对每个整数 \(q\ge 2\)，沿用 \(r_q=\rho(A_q)\) 与
\(
\Lambda_q=\max_{\mu\in\sigma(A_q)\setminus\{r_q\}}|\mu|
\)
的记号（注 \ref{rem:pom-collision-kernel-rh-phase}）。
定义谱凝聚指数为
\[
\boxed{\;
g_q^{(\Lambda)}:=\frac{1}{q}\log\frac{r_q}{\Lambda_q}\;}
\qquad(\text{从而 }\ \Lambda_q/r_q=\exp(-qg_q^{(\Lambda)})).
\]
\end{definition}

\begin{corollary}[\texorpdfstring{$\delta_q$}{deltaq} 与 \texorpdfstring{$g_q^{(\Lambda)}$}{gLambda} 的严格恒等式]\label{cor:pom-deltaq-gLambda-identity}
\leanverified{paper\_pom\_deltaq\_glambda\_identity}
在定义 \ref{def:pom-spectral-condensation-index-gLambda} 的记号下，越线指数满足严格恒等式
\[
\boxed{\;
\delta_q=\log\!\left(\frac{\Lambda_q^2}{r_q}\right)=\log r_q-2q\,g_q^{(\Lambda)}\;}
\]
从而 \(\delta_q=0\) 等价于 \(g_q^{(\Lambda)}=\frac{1}{2q}\log r_q\)。
\end{corollary}
\begin{proof}
由 \(g_q^{(\Lambda)}=\frac{1}{q}(\log r_q-\log\Lambda_q)\) 得 \(\log\Lambda_q=\log r_q-qg_q^{(\Lambda)}\)。
代入 \(\delta_q=2\log\Lambda_q-\log r_q\) 即得所示恒等式。证毕。
\end{proof}
