定理 \ref{thm:pom-diagonal-rate-critical-laplacian-pseudodeterminant} 给出了 \(\operatorname{pdet}(L_w)\) 的闭式。
该闭式在张量积下诱导出一个完全显式的协同因子分解。

\begin{theorem}[伪行列式在张量积下的协同因子分解]\label{thm:pom-diagonal-rate-critical-laplacian-pdet-tensor-synergy-factorization}
\leanverified{paper\_pom\_diagonal\_rate\_critical\_laplacian\_pdet\_tensor\_synergy\_factorization}
取两有限集 \(X,Y\)，全支撑分布 \(w\in\Delta(X)\)、\(v\in\Delta(Y)\)，并令 \(\omega:=w\otimes v\in\Delta(X\times Y)\)。
记 \(n:=|X|\ge 2\)、\(m:=|Y|\ge 2\)。
令
\[
A(w):=\sum_{x\in X}\sqrt{w(x)},\quad C(w):=A(w)^2-1,
\qquad
A(v):=\sum_{y\in Y}\sqrt{v(y)},\quad C(v):=A(v)^2-1.
\]
则成立严格恒等式
\[
\operatorname{pdet}(L_{\omega})
=\operatorname{pdet}(L_w)^{\,m}\operatorname{pdet}(L_v)^{\,n}\cdot \mathsf S(w,v),
\]
其中协同因子 \(\mathsf S(w,v)\) 仅依赖于 \(n,m\) 与 \(C(w),C(v)\)，并具有闭式
\[
\mathsf S(w,v)
:=\frac{(1+C(w))^{m-1}(1+C(v))^{n-1}\,C(w)^{m(n-1)}\,C(v)^{n(m-1)}}
{\bigl(C(w)+C(v)+C(w)C(v)\bigr)^{nm-1}}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-diagonal-rate-critical-laplacian-pseudodeterminant}，
\[
\operatorname{pdet}(L_w)=\frac{A(w)^{n-2}}{C(w)^{n-1}}\cdot \frac{1}{\sqrt{\prod_{x\in X}w(x)}},
\qquad
\operatorname{pdet}(L_v)=\frac{A(v)^{m-2}}{C(v)^{m-1}}\cdot \frac{1}{\sqrt{\prod_{y\in Y}v(y)}}.
\]
同时
\[
A(\omega)=A(w)A(v),\qquad
C(\omega)=A(\omega)^2-1=C(w)+C(v)+C(w)C(v),
\]
以及
\[
\prod_{(x,y)\in X\times Y}\omega(x,y)
=\Bigl(\prod_{x\in X}w(x)\Bigr)^{m}\Bigl(\prod_{y\in Y}v(y)\Bigr)^{n}.
\]
将上述恒等式分别代入 \(\operatorname{pdet}(L_\omega)\) 与 \(\operatorname{pdet}(L_w)^m\operatorname{pdet}(L_v)^n\)，约去几何均值项并整理指数得到
\[
\frac{\operatorname{pdet}(L_{\omega})}{\operatorname{pdet}(L_w)^{\,m}\operatorname{pdet}(L_v)^{\,n}}
=\frac{A(w)^{2(m-1)}A(v)^{2(n-1)}\,C(w)^{m(n-1)}\,C(v)^{n(m-1)}}
{C(\omega)^{nm-1}}.
\]
再用 \(A(w)^2=1+C(w)\)、\(A(v)^2=1+C(v)\) 即得所示协同因子闭式。证毕。
\end{proof}

\begin{corollary}[几何均值剥离后的纯 \texorpdfstring{$H_{1/2}$}{H1/2} 结构]\label{cor:pom-diagonal-rate-critical-laplacian-pdet-tensor-synergy-geometricmean-stripped}
沿用定理 \ref{thm:pom-diagonal-rate-critical-laplacian-pseudodeterminant} 中的几何均值
\(
G(w):=\exp(\frac1n\sum_{x\in X}\log w(x))
\)，
定义重标定伪行列式
\[
\widetilde{\operatorname{pdet}}(L_w):=\operatorname{pdet}(L_w)\,G(w)^{n/2}.
\]
则对乘积分布 \(\omega=w\otimes v\) 成立
\[
\widetilde{\operatorname{pdet}}(L_{w\otimes v})
=\widetilde{\operatorname{pdet}}(L_w)^{\,m}\widetilde{\operatorname{pdet}}(L_v)^{\,n}\cdot \mathsf S(w,v),
\]
其中 \(\mathsf S(w,v)\) 为定理 \ref{thm:pom-diagonal-rate-critical-laplacian-pdet-tensor-synergy-factorization} 中的闭式，且可通过
\(
C(\cdot)=e^{H_{1/2}(\cdot)}-1
\)
仅用 \(H_{1/2}\) 表示。
\end{corollary}
\begin{proof}
由 \(\prod_{(x,y)}\omega(x,y)=(\prod_x w(x))^m(\prod_y v(y))^n\) 立得 \(G(w\otimes v)=G(w)\,G(v)\)。
因此将定理 \ref{thm:pom-diagonal-rate-critical-laplacian-pdet-tensor-synergy-factorization} 两端分别乘以 \(G(w\otimes v)^{nm/2}=G(w)^{mn/2}G(v)^{nm/2}\) 即得所示重标定恒等式。
最后一条由 \(C=e^{H_{1/2}}-1\) 直接改写。证毕。
\end{proof}

\begin{proposition}[概率单纯形的协方差拉普拉斯的伪行列式闭式]\label{prop:covariance-laplacian-pdet-closed-form}
取 \(k\ge 2\) 并令 \(q=(q_1,\dots,q_k)\in\RR_{>0}^k\) 满足 \(\sum_{i=1}^{k}q_i=1\)。
定义
\[
L(q):=\mathrm{diag}(q)-qq^{\mathsf T}.
\]
则 \(L(q)\) 半正定且 \(\ker L(q)=\Span\{\ind\}\)，并且其伪行列式满足闭式
\[
\boxed{\;\det{}'(L(q))=k\prod_{i=1}^{k}q_i.\;}
\]
\end{proposition}
\leanverified{paper\_covariance\_laplacian\_pdet\_closed\_form}
\begin{proof}
由 \(\sum_i q_i=1\) 得 \(L(q)\ind=0\)，故 \(\ind\in\ker L(q)\)。
又对任意 \(x\in\RR^k\) 有
\(
x^{\mathsf T}L(q)x=\sum_i q_i x_i^2-(\sum_i q_i x_i)^2\ge 0
\)，
因此 \(L(q)\) 半正定且 \(\rank L(q)=k-1\)。
\par
令 \(w_{ij}:=q_iq_j\ (i\neq j)\) 并考虑带权完全图 \(K_k\) 的 Laplacian \(L_w\)：
其对角元为 \( (L_w)_{ii}=\sum_{j\neq i}w_{ij}=q_i(1-q_i)\)，非对角元为 \((L_w)_{ij}=-w_{ij}=-q_iq_j\)。
因此 \(L_w=L(q)\)。
\par
由矩阵树定理，任一代数余子式等于带权生成树权重和 \(\tau\)，并且 \(\det{}'(L)=k\,\tau\)。
对边权具乘积分解 \(w_{ij}=a_ia_j\) 的完全图，Prüfer 编码给出
\(
\tau=(\sum_i a_i)^{k-2}\prod_i a_i
\)。
取 \(a_i=q_i\) 且 \(\sum_i q_i=1\)，得到 \(\tau=\prod_i q_i\)，从而 \(\det{}'(L(q))=k\prod_i q_i\)。证毕。
\end{proof}
