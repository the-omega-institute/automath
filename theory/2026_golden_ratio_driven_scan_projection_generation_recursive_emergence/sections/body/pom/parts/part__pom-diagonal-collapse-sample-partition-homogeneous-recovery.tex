\paragraph{对角坍缩：当矩阶 \(q\sim \tau m\) 时的二次自由能退化}
回忆：对任意分辨率 \(m\) 与任意实数 \(q\ge 0\)，幂和矩定义为
\(
S_q(m)=\sum_{x\in X_m} d_m(x)^q
\)
（见 \S\ref{subsubsec:pom-real-q-pressure-thermo-frontier} 的开头），并在推论 \ref{cor:pom-escort-zero-temperature-superexp-concentration} 中指出：
当 \(q\) 随 \(m\) 线性增长时，escort 分布对最大纤维层集呈 \(m^2\) 级别超指数集中。
下述结论表明：在同一对角标度 \(q\sim \tau m\) 下，
\(\log S_q(m)\) 的全部多重分形信息在 \(m^2\) 尺度上完全坍缩为顶端指数 \(\alpha_\ast\)。

\begin{theorem}[对角坍缩二次自由能律]\label{thm:pom-diagonal-collapse-quadratic-free-energy}
\leanverified{paper\_pom\_diagonal\_collapse\_quadratic\_free\_energy}
令
\(
M_m:=\max_{x\in X_m}d_m(x)
\)
与
\(
\alpha_m:=\frac{1}{m}\log M_m
\)
如推论 \ref{cor:pom-escort-zero-temperature-superexp-concentration}。
假设极限
\[
\alpha_\ast:=\lim_{m\to\infty}\alpha_m=\lim_{m\to\infty}\frac{1}{m}\log M_m
\]
存在且有限。
取任意固定常数 \(\tau>0\)，并令 \(q_m:=\lfloor\tau m\rfloor\)。
若满足
\[
\log|X_m|=O(m)\qquad(m\to\infty),
\]
则成立极限
\[
\boxed{\;
\lim_{m\to\infty}\frac{1}{m^2}\log S_{q_m}(m)=\tau\,\alpha_\ast\; }.
\]
\end{theorem}
\begin{proof}
由 \(S_{q_m}(m)\ge M_m^{q_m}\) 得
\[
\frac{1}{m^2}\log S_{q_m}(m)\ \ge\ \frac{q_m}{m}\cdot \frac{1}{m}\log M_m
\xrightarrow[m\to\infty]{}\tau\alpha_\ast.
\]
另一方面 \(S_{q_m}(m)\le |X_m|\,M_m^{q_m}\)，从而
\[
\frac{1}{m^2}\log S_{q_m}(m)
\ \le\
\frac{1}{m^2}\log|X_m|
\ +\
\frac{q_m}{m}\cdot\frac{1}{m}\log M_m.
\]
由假设 \(\log|X_m|=O(m)\) 得 \(\frac{1}{m^2}\log|X_m|\to 0\)；
并由 \(\frac{q_m}{m}\to\tau\) 与 \(\frac{1}{m}\log M_m\to\alpha_\ast\) 得上界亦趋于 \(\tau\alpha_\ast\)。
夹逼即得结论。证毕。
\end{proof}

\begin{corollary}[对角提取原理：由单一对角确定 \texorpdfstring{$\alpha_\ast$}{alpha*}]\label{cor:pom-diagonal-extract-alpha-star}
在定理 \ref{thm:pom-diagonal-collapse-quadratic-free-energy} 的设定下，对任意 \(\tau>0\) 均有
\[
\boxed{\;
\alpha_\ast=\lim_{m\to\infty}\frac{1}{\tau m^2}\log S_{\lfloor\tau m\rfloor}(m)\;}
\]
并且右端极限与 \(\tau\) 无关。
\end{corollary}

\begin{theorem}[顶端简并指数的对角次阶项（有间距情形）]\label{thm:pom-diagonal-subleading-top-degeneracy}
\leanverified{paper\_pom\_diagonal\_subleading\_top\_degeneracy}
在定理 \ref{thm:pom-diagonal-collapse-quadratic-free-energy} 的设定下，
令最大纤维达到者集合与其计数为
\[
\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
N_m:=|\mathcal{M}_m|.
\]
并令第二大纤维多重度
\[
M_m^{(2)}:=\max\{d_m(x):\ x\in X_m\setminus\mathcal{M}_m\},
\]
以及顶端间距指数（见命题 \ref{prop:pom-escort-freezing-gap-threshold}）
\[
\delta:=\liminf_{m\to\infty}\frac{1}{m}\log\frac{M_m}{M_m^{(2)}}\ \in[0,\infty].
\]
若满足 \(\delta>0\) 且极限
\[
f(\alpha_\ast):=\lim_{m\to\infty}\frac{1}{m}\log N_m
\]
存在（从而 \(f(\alpha_\ast)\) 可视为顶端简并指数；在零温制度下它与定义 \ref{def:pom-max-fiber-entropy-rate} 的 \(h_\ast\) 一致），
则对任意固定 \(\tau>0\) 与 \(q_m=\lfloor\tau m\rfloor\) 有
\[
\boxed{\;
\log S_{q_m}(m)=q_m\log M_m+\log N_m+o(1)\;}
\]
以及
\[
\boxed{\;
\lim_{m\to\infty}\frac{1}{m}\Bigl(\log S_{q_m}(m)-q_m\log M_m\Bigr)=f(\alpha_\ast)\; }.
\]
若进一步满足 \(\log M_m=\alpha_\ast m+o(1)\)，则上述次阶极限亦可改写为
\[
\lim_{m\to\infty}\frac{1}{m}\Bigl(\log S_{q_m}(m)-q_m\alpha_\ast m\Bigr)=f(\alpha_\ast).
\]
\end{theorem}
\begin{proof}
将求和按最大层集与其补集分解：
\[
S_{q_m}(m)=\sum_{x\in\mathcal{M}_m}d_m(x)^{q_m}+\sum_{x\notin\mathcal{M}_m}d_m(x)^{q_m}
=N_m M_m^{q_m}+\sum_{x\notin\mathcal{M}_m}d_m(x)^{q_m}.
\]
对补集部分，用 \(d_m(x)\le M_m^{(2)}\) 得
\[
0\le \sum_{x\notin\mathcal{M}_m}d_m(x)^{q_m}\le |X_m|\,(M_m^{(2)})^{q_m}.
\]
因此
\[
\frac{\sum_{x\notin\mathcal{M}_m}d_m(x)^{q_m}}{N_m M_m^{q_m}}
\le
\frac{|X_m|}{N_m}\left(\frac{M_m^{(2)}}{M_m}\right)^{q_m}.
\]
由 \(\delta>0\) 的定义，存在 \(\delta_0\in(0,\delta)\) 与 \(m_0\) 使对所有 \(m\ge m_0\) 有 \(M_m^{(2)}/M_m\le e^{-\delta_0 m}\)。
又因 \(q_m\asymp m\)，故
\[
\left(\frac{M_m^{(2)}}{M_m}\right)^{q_m}\le \exp(-\delta_0 q_m m)=\exp(-\Theta(m^2)).
\]
而 \(|X_m|/N_m=\exp(O(m))\)（在本文 Zeckendorf 稳定类型制度下 \(\log|X_m|=O(m)\)，且 \(N_m\ge 1\)），
从而上式右端趋于 \(0\)。
于是
\(
S_{q_m}(m)=N_m M_m^{q_m}\bigl(1+o(1)\bigr)
\)，取对数得到第一条渐近式，进而除以 \(m\) 并用 \(f(\alpha_\ast)\) 的定义得到第二条极限。
最后一条改写由 \(\log M_m=\alpha_\ast m+o(1)\) 与 \(q_m=O(m)\) 推出
\(
q_m(\log M_m-\alpha_\ast m)=o(m)
\)，故不改变 \((1/m)\) 尺度极限。证毕。
\end{proof}

\paragraph{等值划分的 Möbius 完备性：有限样本相等模式与幂和的不变量刻画}
固定有限集 \(X\) 与分布 \(w\in\Delta(X)\)。
令 \(X_1,\dots,X_t\stackrel{\mathrm{i.i.d.}}{\sim} w\)，并令 \(\Pi_t\) 为 \([t]\) 的随机划分：\(i,j\) 处于同一块当且仅当 \(X_i=X_j\)。
以“细化”为偏序：\(\sigma\preceq\pi\) 表示 \(\sigma\) 比 \(\pi\) 更细（即 \(\pi\) 是 \(\sigma\) 的粗化）。
记 \(\mathcal{P}_t\) 为划分格，\(\mu(\sigma,\pi)\) 为其 Möbius 函数。

\leanverified{paper\_pom\_partition\_mobius\_event\_inversion}
\begin{theorem}[事件--划分 Möbius 反演]\label{thm:pom-partition-mobius-event-inversion}
对任意 \(\sigma\in\mathcal{P}_t\) 定义事件
\[
E_\sigma:=\{X_i=X_j\ \forall i,j\in B,\ \forall B\in\sigma\},
\]
并定义幂和
\(
p_k(w):=\sum_{x\in X}w(x)^k
\)
与函数
\[
G(\sigma):=\prod_{B\in\sigma}p_{|B|}(w).
\]
则对任意 \(\sigma\in\mathcal{P}_t\) 有严格恒等式
\[
\boxed{\;
\mathbb{P}(E_\sigma)=G(\sigma),\qquad
\mathbb{P}(E_\sigma)=\sum_{\pi\succeq\sigma}\mathbb{P}(\Pi_t=\pi)\; }.
\]
从而在划分格上有精确 Möbius 反演
\[
\boxed{\;
\mathbb{P}(\Pi_t=\pi)=\sum_{\sigma\preceq\pi}\mu(\sigma,\pi)\,G(\sigma),\qquad \pi\in\mathcal{P}_t\; }.
\]
\end{theorem}
\begin{proof}
事件 \(E_\sigma\) 表示“每个块内样本全相等”。
对每个块 \(B\)（大小 \(|B|\)），独立性给出其概率为 \(\sum_{x}w(x)^{|B|}=p_{|B|}(w)\)；
块间不交，故 \(\mathbb{P}(E_\sigma)=\prod_{B\in\sigma}p_{|B|}(w)=G(\sigma)\)。
\par
另一方面，\(E_\sigma\) 当且仅当真实相等划分 \(\Pi_t\) 为 \(\sigma\) 的粗化（即 \(\Pi_t\succeq\sigma\)）；
故 \(\mathbb{P}(E_\sigma)=\sum_{\pi\succeq\sigma}\mathbb{P}(\Pi_t=\pi)\)。
对该关系在格 \((\mathcal{P}_t,\preceq)\) 上作 Möbius 反演即得结论。证毕。
\end{proof}

\begin{corollary}[由相等模式显式提取幂和]\label{cor:pom-partition-extract-power-sums}
\leanverified{paper\_pom\_partition\_extract\_power\_sums}
令 \((u)_k:=u(u-1)\cdots(u-k+1)\) 为下降阶乘。
则对任意整数 \(k\le t\) 有恒等式
\[
\boxed{\;
p_k(w)=\mathbb{E}\left[\frac{1}{(t)_k}\sum_{B\in\Pi_t}(|B|)_k\right]\; }.
\]
因此 \(\Pi_t\) 的分布唯一决定 \(\bigl(p_2(w),\dots,p_t(w)\bigr)\)。
\end{corollary}
\begin{proof}
从 \([t]\) 中等概率抽取 \(k\) 个互异指标，其全相等事件的概率为 \(p_k(w)\)。
而在给定划分 \(\Pi_t\) 条件下，该概率等于“抽到同一块内”的概率：
\(
\sum_{B\in\Pi_t}\binom{|B|}{k}/\binom{t}{k}
=\frac{1}{(t)_k}\sum_{B}(|B|)_k
\)。
对 \(\Pi_t\) 取期望即得。证毕。
\end{proof}

\begin{theorem}[有限样本不可区分性：全对称统计量的完备不变量]\label{thm:pom-finite-sample-indistinguishability-power-sums}
\leanverified{paper\_pom\_finite\_sample\_indistinguishability\_power\_sums}
令 \(\mathfrak{S}_t\) 作用于 \([t]\) 的指标置换，\(\mathfrak{S}_X\) 作用于 \(X\) 的标签置换。
称统计量 \(\Phi(X_1,\dots,X_t)\) 为\emph{全对称}，若对任意 \(\sigma\in\mathfrak{S}_t\)、\(\pi\in\mathfrak{S}_X\) 有
\[
\Phi(X_{\sigma(1)},\dots,X_{\sigma(t)})=\Phi(X_1,\dots,X_t),
\qquad
\Phi(\pi(X_1),\dots,\pi(X_t))=\Phi(X_1,\dots,X_t).
\]
则对任意两分布 \(w,w'\in\Delta(X)\)，以下命题等价：
\begin{enumerate}
  \item \(p_k(w)=p_k(w')\) 对所有 \(k=2,\dots,t\) 成立；
  \item \(\Pi_t\) 在 \(w\) 与 \(w'\) 下同分布；
  \item 对任意全对称统计量 \(\Phi\)，\(\Phi(X_1,\dots,X_t)\) 在 \(w\) 与 \(w'\) 下同分布。
\end{enumerate}
并且若存在某个 \(k\le t\) 使 \(p_k(w)\neq p_k(w')\)，则存在全对称检验以正优势区分二者，例如 \(\mathbf{1}_{\{X_1=\cdots=X_k\}}\)。
\end{theorem}
\begin{proof}
\textbf{(1)\(\Rightarrow\)(2).}
由定理 \ref{thm:pom-partition-mobius-event-inversion}，\(\Pi_t\) 的全部原子概率为 \(\{G(\sigma)\}_{\sigma\in\mathcal{P}_t}\) 的线性组合；
而 \(G(\sigma)\) 仅依赖 \(\{p_{|B|}(w)\}_{|B|\le t}\)。
故 (1) 蕴含 (2)。
\par
\textbf{(2)\(\Rightarrow\)(3).}
若 \(\Phi\) 同时对指标置换与标签置换不变，则 \(\Phi\) 只能依赖于样本间的相等关系，
而该相等关系恰由随机划分 \(\Pi_t\) 编码；因此 \(\Phi\) 为 \(\Pi_t\) 的可测函数。
由 \(\Pi_t\) 同分布得到 \(\Phi\) 同分布。
\par
\textbf{(3)\(\Rightarrow\)(1).}
取 \(\Phi=\mathbf 1_{\{X_1=\cdots=X_k\}}\)（其显然全对称），则
\(
\EE[\Phi]=p_k(w)
\)
与
\(
\EE[\Phi]=p_k(w')
\)
给出 \(p_k(w)=p_k(w')\)（对 \(2\le k\le t\)）。
证毕。
\end{proof}

\paragraph{完全齐次多项式序列：PF\texorpdfstring{$_\infty$}{∞} 性、递推与有限反演}
设 \(w\in\Delta(X)\) 且 \(|X|=n\)，将其坐标写为 \(w=(w_1,\dots,w_n)\)（含重数，允许重排）。
定义完全齐次对称多项式
\[
h_t(w):=\sum_{\substack{r_1,\dots,r_n\in\NN_0\\ \sum_i r_i=t}}\prod_{i=1}^n w_i^{r_i}\qquad(t\ge0),
\]
并记生成函数
\[
H_w(z):=\sum_{t\ge0}h_t(w)\,z^t=\prod_{i=1}^n\frac{1}{1-w_i z}.
\]

\begin{theorem}[PF\texorpdfstring{$_\infty$}{∞} 与全正性]\label{thm:pom-complete-homogeneous-pf-infty}
序列 \(\{h_t(w)\}_{t\ge0}\) 为 Pólya frequency 序列（PF\(_\infty\)）：
其 Toeplitz 矩阵 \(\bigl[h_{j-i}(w)\bigr]_{i,j\ge0}\)（约定 \(h_k=0\) 若 \(k<0\)）全正（任意有限子式非负）。
因此 \(\{h_t(w)\}\) 对数凸并满足比值单调性：
\[
h_t(w)^2\le h_{t-1}(w)\,h_{t+1}(w)\quad(t\ge1),
\qquad
\frac{h_{t+1}(w)}{h_t(w)}\ \ \text{随 \(t\) 单调不减}.
\]
\end{theorem}
\begin{proof}
每个几何因子 \((1-w_i z)^{-1}=\sum_{t\ge0} w_i^t z^t\) 的系数序列为 PF\(_\infty\)（其生成函数符合 PF\(_\infty\) 的标准结构表述）。
PF\(_\infty\) 在 Cauchy 乘积（卷积）下封闭，故有限乘积 \(H_w(z)=\prod_i(1-w_i z)^{-1}\) 的系数序列仍为 PF\(_\infty\)，即得全正性。
对数凸与比值单调性为 PF\(_\infty\) 的直接推论（例如由 \(2\times2\) 主子式非负推出 \(h_t^2\le h_{t-1}h_{t+1}\)）。证毕。
\leanverified{paper\_pom\_complete\_homogeneous\_pf\_infty}
\end{proof}

\begin{corollary}[留下 \(t\) 个未查询点的 Bayes 全恢复极限与危险率单调]\label{cor:pom-bayes-full-recovery-hazard-monotone}
在微正则折叠语义（定义 \ref{def:pom-microcanonical-fold-class}）下，
固定有限字母表 \(X\) 与分布 \(w\in\Delta(X)\)（全支撑，且取一列 \(N\to\infty\) 使 \(Nw(x)\in\NN\) 对一切 \(x\) 成立）。
令 \(d_N(x):=Nw(x)\)，并在先验 \(F\sim\mathrm{Unif}(\mathfrak{Fold}(d_N))\) 下查询 \(k=N-t\) 个点（仅留下固定的 \(t\) 个未查询点）。
则整体 Bayes 精确恢复成功率满足极限
\[
\boxed{\;
\lim_{N\to\infty}\mathrm{Succ}^{\mathrm{Bayes}}_t(N)=h_t(w)\; }.
\]
因此在该“固定剩余点数”的 Bayes 全恢复极限中，成功率曲线为 \(\{h_t(w)\}_{t\ge0}\)。
\par
并且若定义对数危险率
\[
\Delta_t:=-\log\frac{h_{t+1}(w)}{h_t(w)},
\]
则 \(\Delta_t\) 随 \(t\) 单调不增。
若记最大质量 \(w_\star:=\max_i w_i\)，则
\[
\lim_{t\to\infty}\frac{h_{t+1}(w)}{h_t(w)}=w_\star,
\]
且该极限由单调序列 \(h_{t+1}(w)/h_t(w)\) 从下逼近。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-microcanonical-count-sufficient-statistic-posterior-uniform} 的后验均匀性，
当 \(k=N-t\) 时给定轨迹的 Bayes 最优条件成功率等于后验模空间大小的倒数（参见推论 \ref{cor:pom-microcanonical-posterior-map-success-double-exponential} 的同型论证）；
其仅依赖于未查询部分的剩余计数向量 \(R=(R(x))_{x\in X}\)：
若 \(R(x)\) 表示标签 \(x\) 在 \(t\) 个未查询点中出现次数，则有
\[
\mathrm{Succ}^{\mathrm{Bayes}}\bigl(F\mid T_{1:N-t}\bigr)=\frac{\prod_{x\in X}R(x)!}{t!}.
\]
在微正则先验下，未查询点等价于从多重集 \(\{x^{\times d_N(x)}\}\) 中无放回抽取 \(t\) 次，
故 \(R\) 为多元超几何分布并满足
\[
\mathbb{P}(R=r)=\frac{\prod_{x\in X}\binom{d_N(x)}{r(x)}}{\binom{N}{t}}
\qquad\Bigl(\sum_x r(x)=t\Bigr).
\]
于是整体成功率为
\[
\mathrm{Succ}^{\mathrm{Bayes}}_t(N)
=\EE\!\left[\frac{\prod_x R(x)!}{t!}\right]
=\frac{1}{(N)_t}\sum_{\substack{r(x)\in\NN_0\\ \sum_x r(x)=t}}\ \prod_{x\in X}\bigl(d_N(x)\bigr)_{r(x)},
\]
其中 \((u)_r=u(u-1)\cdots(u-r+1)\) 且 \((N)_t=N(N-1)\cdots(N-t+1)\)。
对固定 \(t\) 与固定 \(x\)，有 \((d_N(x))_{r}=d_N(x)^{r}+O(N^{r-1})\)，故
\[
\prod_x (d_N(x))_{r(x)}
=N^{t}\prod_x w(x)^{r(x)}+O(N^{t-1}),
\qquad
(N)_t=N^{t}+O(N^{t-1}).
\]
将其代回并除以 \((N)_t\)，得到
\[
\mathrm{Succ}^{\mathrm{Bayes}}_t(N)
=\sum_{\substack{r(x)\in\NN_0\\ \sum_x r(x)=t}}\ \prod_{x\in X} w(x)^{r(x)}\ +\ O(N^{-1})
=h_t(w)+O(N^{-1}),
\]
从而 \(\mathrm{Succ}^{\mathrm{Bayes}}_t(N)\to h_t(w)\)。
\par
最后，由定理 \ref{thm:pom-complete-homogeneous-pf-infty}，比值 \(h_{t+1}/h_t\) 单调不减，故 \(\Delta_t\) 单调不增。
又 \(H_w(z)=\prod_i(1-w_i z)^{-1}\) 的收敛半径为 \(1/w_\star\)，由系数比值判别得 \(\lim_{t\to\infty}h_{t+1}/h_t=w_\star\)；
单调收敛由比值单调性给出。证毕。
\end{proof}

\begin{theorem}[有限阶递推：\texorpdfstring{$h_t(w)$}{h_t(w)} 的线性递推与 Newton 可计算性]\label{thm:pom-complete-homogeneous-linear-recurrence}
\leanverified{paper\_pom\_complete\_homogeneous\_linear\_recurrence}
令 \(e_j(w)\) 为 \(w\) 的基本对称多项式（\(e_0=1\)，且因 \(w\in\Delta\) 有 \(e_1=\sum_i w_i=1\)）。
则对所有 \(t\ge1\) 成立递推
\[
\boxed{\;
h_t(w)=\sum_{j=1}^{\min(n,t)}(-1)^{j-1}e_j(w)\,h_{t-j}(w),\qquad h_0(w)=1\; }.
\]
等价地，
\[
\left(\sum_{j=0}^{n}(-1)^j e_j(w)\,z^j\right)\left(\sum_{t\ge0}h_t(w)\,z^t\right)=1.
\]
此外，\(\{e_j(w)\}_{j\le n}\) 可由幂和 \(\{p_k(w)\}_{k\le n}\) 经 Newton 恒等式确定；
结合推论 \ref{cor:pom-partition-extract-power-sums}，\(\{h_t(w)\}\) 可由有限样本相等模式以 \(O(n)\) 状态常内存递推计算。
\end{theorem}
\begin{proof}
由生成函数恒等式
\(
H_w(z)=\prod_{i=1}^n(1-w_i z)^{-1}
\)
可得其分母
\(
\prod_{i=1}^n(1-w_i z)=\sum_{j=0}^n(-1)^j e_j(w)\,z^j
\)。
两边相乘为 \(1\)，比较 \(z^t\) 系数即得递推。
Newton 恒等式为对称多项式的标准关系：\(\{p_k\}\) 决定 \(\{e_j\}\)。
最后一条把 \(\{p_k\}\) 的可观测性（推论 \ref{cor:pom-partition-extract-power-sums}）与递推拼接即可。证毕。
\end{proof}

\begin{theorem}[由“全恢复曲线”反演概率谱]\label{thm:pom-invert-w-from-homogeneous-curve}
设 \(w\in\Delta(X)\) 满足全支撑（\(w_i>0\)）且 \(|X|=n\) 已知。
则已知有限段 \(\{h_t(w)\}_{t=0}^{n}\) 即可唯一确定多项式
\[
Q_w(z):=\prod_{i=1}^n(1-w_i z)=\sum_{j=0}^n(-1)^j e_j(w)\,z^j,
\]
从而唯一确定多重集 \(\{w_i\}_{i=1}^n\)（含重数）。
因此“全恢复曲线”在 \(t=0,\dots,n\) 上的取值构成 \(w\) 的一个完全不变量。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-complete-homogeneous-linear-recurrence}，对 \(t=1,\dots,n\) 的递推式给出关于未知系数 \(e_1,\dots,e_t\) 的上三角线性系统；

由全支撑性，\(Q_w\) 的根集为 \(\{1/w_i\}\)，故 \(\{w_i\}\)（含重数）由 \(Q_w\) 唯一确定。证毕。
\end{proof}

\end{proof}

\begin{theorem}[对角矩 \texorpdfstring{$\Rightarrow$}{=>} 顶端谱]\label{thm:pom-closure-diagonal-to-top-spectrum}

\[

\]
收敛到顶端简并指数 \(f(\alpha_\ast)\)。
\end{theorem}
