\paragraph{有限窗 Pareto 几何、零温精确塌缩与右端谱障碍}
在有限窗 escort 导数公式、\(\kappa_m\) 的 Gibbs--变分前沿、零温两项展开与预言机热相切割都已固定之后，还可把其中若干隐含的几何量继续压缩为四条直接可用的后果。前两条给出固定分辨率下的严格 Legendre 曲率与零温塌缩速率；后两条则把层集谱右端点与有限整数温度上的冻结障碍完全闭合。

\begin{theorem}[有限窗 Pareto 前沿的严格 Legendre 几何与曲率公式]\label{thm:pom-finite-pareto-legendre-curvature}
固定分辨率 \(m\ge 1\)。记
\[
d_m^{\min}:=\min_{x\in X_m}d_m(x),
\qquad
M_m:=\max_{x\in X_m}d_m(x).
\]
对任意 \(q\in\RR\) 定义扩展 escort 族
\[
w_{m,q}(x):=\frac{d_m(x)^q}{S_q(m)},
\qquad
S_q(m):=\sum_{x\in X_m}d_m(x)^q,
\qquad
F_m(q):=\log S_q(m).
\]
再定义可达熵--信息区域
\[
\mathcal D_m:=\{(\kappa_m(\pi),H(\pi)):\ \pi\in\Delta(X_m)\}\subset\RR^2,
\]
以及其上边界函数
\[
\mathcal H_m(\kappa):=
\sup\{H(\pi):\ \pi\in\Delta(X_m),\ \kappa_m(\pi)=\kappa\}.
\]
若多重度函数 \(d_m\) 并非常值，则成立：
\begin{enumerate}
  \item 对所有 \(q\in\RR\)，
  \[
  F_m'(q)=\kappa_m(w_{m,q}),
  \qquad
  F_m''(q)=\Var_{w_{m,q}}\!\bigl(\log d_m(X)\bigr)>0.
  \]
  \item 映射 \(q\mapsto F_m'(q)\) 为从 \(\RR\) 到
  \[
  (\log d_m^{\min},\log M_m)
  \]
  的严格递增实解析双射。
  \item 上边界函数 \(\mathcal H_m\) 在 \((\log d_m^{\min},\log M_m)\) 上实解析且严格凹，并由 escort 族参数化为
  \[
  \mathcal H_m\bigl(F_m'(q)\bigr)=F_m(q)-qF_m'(q).
  \]
  \item 若 \(\kappa=F_m'(q)\)，则
  \[
  \frac{\dd}{\dd\kappa}\mathcal H_m(\kappa)=-q,
  \qquad
  \frac{\dd^2}{\dd\kappa^2}\mathcal H_m(\kappa)
  =-\frac{1}{\Var_{w_{m,q}}(\log d_m(X))}<0.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
命题 \ref{prop:pom-escort-derivative-identities} 的证明只用到
\[
F_m(q)=\log\sum_{x\in X_m}e^{q\log d_m(x)},
\]
故对一切 \(q\in\RR\) 同样成立，从而
\[
F_m'(q)=\kappa_m(w_{m,q}),
\qquad
F_m''(q)=\Var_{w_{m,q}}(\log d_m(X))\ge 0.
\]
若 \(d_m\) 并非常值，则 \(\log d_m(X)\) 在任意 \(w_{m,q}\) 下都不是常数，故方差严格为正。这证明了第一条，并推出 \(F_m\) 在 \(\RR\) 上严格凸。

再看端点。由于 \(w_{m,q}\) 对 \(d_m(x)^q\) 作指数倾斜，当 \(q\to+\infty\) 时其质量集中到 \(\{d_m=M_m\}\)，故
\[
F_m'(q)=\kappa_m(w_{m,q})\longrightarrow \log M_m.
\]
同理，当 \(q\to-\infty\) 时其质量集中到 \(\{d_m=d_m^{\min}\}\)，故
\[
F_m'(q)\longrightarrow \log d_m^{\min}.
\]
由 \(F_m''>0\) 可知 \(F_m'\) 严格递增；再由实解析性即得第二条。

对第三条，固定 \(q\in\RR\)。定理 \ref{thm:pom-kappa-gibbs-variational-escort-frontier} 的 KL 展开对任意 \(\pi\in\Delta(X_m)\) 给出
\[
F_m(q)=q\,\kappa_m(\pi)+H(\pi)+D_{\mathrm{KL}}(\pi\|w_{m,q}).
\]
若再要求 \(\kappa_m(\pi)=F_m'(q)=\kappa_m(w_{m,q})\)，则
\[
H(\pi)\le F_m(q)-qF_m'(q),
\]
且等号当且仅当 \(\pi=w_{m,q}\)。因此
\[
\mathcal H_m\bigl(F_m'(q)\bigr)=F_m(q)-qF_m'(q).
\]
由第二条，\(q\mapsto F_m'(q)\) 可逆，故 \(\mathcal H_m\) 在整个开区间上由上述公式定义，且为实解析函数。

最后对恒等式
\[
\mathcal H_m(F_m'(q))=F_m(q)-qF_m'(q)
\]
对 \(q\) 求导，得到
\[
\mathcal H_m'(F_m'(q))\,F_m''(q)=-qF_m''(q),
\]
故
\[
\mathcal H_m'(\kappa)=-q.
\]
再对 \(\kappa\) 求导并用 \(\dd\kappa/\dd q=F_m''(q)\)，便得
\[
\mathcal H_m''(\kappa)=-\frac{1}{F_m''(q)}
=-\frac{1}{\Var_{w_{m,q}}(\log d_m(X))}<0.
\]
证毕。
\end{proof}

\begin{theorem}[固定分辨率零温极限：escort 族精确塌缩到最大纤维均匀分布]\label{thm:pom-finite-zero-temp-escort-uniform-maxfiber}
\leanverified{paper\_pom\_finite\_zero\_temp\_escort\_uniform\_maxfiber}
固定分辨率 \(m\ge 1\)。沿用
\[
M_m:=\max_{x\in X_m}d_m(x),
\qquad
\mathcal M_m:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
N_m:=|\mathcal M_m|.
\]
定义次大多重度
\[
M_m^{(2)}:=\max\{d_m(x):\ x\notin \mathcal M_m\},
\]
并在 \(\mathcal M_m=X_m\) 时约定 \(M_m^{(2)}:=0\)。记
\[
\rho_m:=\frac{M_m^{(2)}}{M_m}\in[0,1),
\qquad
U_m^{\max}:=\mathrm{Unif}(\mathcal M_m).
\]
则当 \(q\to+\infty\) 时，escort 分布 \(w_{m,q}\) 收敛到 \(U_m^{\max}\)，并且有显式界
\[
\|w_{m,q}-U_m^{\max}\|_{\mathrm{TV}}
=1-w_{m,q}(\mathcal M_m)
\le
\frac{|X_m|-N_m}{N_m}\,\rho_m^q.
\]
同时有
\[
\kappa_m(w_{m,q})=\log M_m+O(\rho_m^q),
\qquad
H(w_{m,q})=\log N_m+O(q\rho_m^q).
\]
\end{theorem}
\begin{proof}
总变差界正是命题 \ref{prop:pom-escort-maxfiber-tv-bound}，其中 \(D_m=M_m\)、\(A_{\max}=\mathcal M_m\)、\(K=N_m\)。

再证 \(\kappa_m\) 的估计。由定义
\[
\kappa_m(w_{m,q})
=\sum_{x\in X_m}w_{m,q}(x)\log d_m(x)
=w_{m,q}(\mathcal M_m)\log M_m+\sum_{x\notin\mathcal M_m}w_{m,q}(x)\log d_m(x),
\]
故
\[
0\le \log M_m-\kappa_m(w_{m,q})
=\sum_{x\notin\mathcal M_m}w_{m,q}(x)\bigl(\log M_m-\log d_m(x)\bigr)
\le (\log M_m)\bigl(1-w_{m,q}(\mathcal M_m)\bigr).
\]
代入总变差界即得
\(
\kappa_m(w_{m,q})=\log M_m+O(\rho_m^q)
\)。

对熵，由 \(w_{m,q}\) 在 \(\mathcal M_m\) 上为常密度，可写成精确分解
\[
w_{m,q}=(1-\varepsilon_q)U_m^{\max}+\varepsilon_q\nu_q,
\qquad
\varepsilon_q:=1-w_{m,q}(\mathcal M_m),
\]
其中 \(\nu_q\) 支撑于 \(X_m\setminus\mathcal M_m\)。于是
\[
H(w_{m,q})
=h_2(\varepsilon_q)+(1-\varepsilon_q)\log N_m+\varepsilon_q H(\nu_q),
\]
其中 \(h_2(t):=-t\log t-(1-t)\log(1-t)\)。又有
\[
H(\nu_q)\le \log|X_m|,
\qquad
\varepsilon_q=O(\rho_m^q).
\]
因此
\[
h_2(\varepsilon_q)=O(\varepsilon_q|\log\varepsilon_q|)=O(q\rho_m^q),
\qquad
\varepsilon_q H(\nu_q)=O(\rho_m^q),
\]
从而
\[
H(w_{m,q})=\log N_m+O(q\rho_m^q).
\]
证毕。
\end{proof}

\begin{theorem}[层集谱右端点与零温缺口的支撑函数表达]\label{thm:pom-spectrum-right-edge-support-function}
在命题 \ref{prop:pom-variational-unity-tau-from-f} 与定理 \ref{thm:pom-zero-temperature-two-term-expansion-maxfiber-entropy} 的设定下，记
\[
\Sigma_f:=\{\alpha\in[0,\log 2]:\ f(\alpha)>-\infty\}.
\]
则有
\[
\alpha_\ast=\sup\Sigma_f.
\]
并且对每个整数 \(a\ge 1\)，若记整数阶压力 \(P_a=\tau(a)\) 并定义零温缺口
\[
R(a):=P_a-a\alpha_\ast,
\]
则成立支撑函数公式
\[
R(a)=\sup_{\alpha\in\Sigma_f}\bigl(f(\alpha)-a(\alpha_\ast-\alpha)\bigr).
\]
\end{theorem}
\begin{proof}
第一式正是定理 \ref{thm:pom-zero-temperature-two-term-expansion-maxfiber-entropy} 中对 \(\alpha_\ast\) 的刻画。

另一方面，命题 \ref{prop:pom-variational-unity-tau-from-f} 给出
\[
\tau(a)=\sup_{\alpha\in[0,\log 2]}\bigl(f(\alpha)+a\alpha\bigr).
\]
由于此处 \(P_a=\tau(a)\) 是同一整数阶压力，故
\[
P_a=\sup_{\alpha\in[0,\log 2]}\bigl(f(\alpha)+a\alpha\bigr).
\]
由于 \(f(\alpha)=-\infty\) 在 \([0,\log 2]\setminus\Sigma_f\) 上不贡献上确界，故
\[
P_a=\sup_{\alpha\in\Sigma_f}\bigl(f(\alpha)+a\alpha\bigr).
\]
两边减去 \(a\alpha_\ast\) 即得
\[
R(a)
=\sup_{\alpha\in\Sigma_f}\bigl(f(\alpha)-a(\alpha_\ast-\alpha)\bigr).
\]
证毕。
\end{proof}

\begin{corollary}[右端谱高度给出的冻结障碍]\label{cor:pom-right-edge-height-freezing-obstruction}
在定理 \ref{thm:pom-spectrum-right-edge-support-function} 的设定下，对每个整数 \(a\ge 1\) 都有
\[
R(a)\ge f(\alpha_\ast)=h_\ast.
\]
等价地，
\[
P_a\ge a\alpha_\ast+h_\ast.
\]
特别地，若
\[
h_\ast>0,
\]
则对所有整数 \(a\ge 1\) 成立
\[
P_a-a\alpha_\ast>0,
\qquad
h_\infty^{\mathrm{esc}}(a)>0.
\]
因此，有限整数温度上的精确冻结
\[
h_\infty^{\mathrm{esc}}(a)=0
\]
不可能发生。
\end{corollary}
\begin{proof}
在定理 \ref{thm:pom-spectrum-right-edge-support-function} 的支撑函数公式中取 \(\alpha=\alpha_\ast\)，即可得到
\[
R(a)\ge f(\alpha_\ast).
\]
再由定理 \ref{thm:pom-zero-temperature-two-term-expansion-maxfiber-entropy}，
\[
f(\alpha_\ast)=h_\ast,
\]
故
\[
R(a)\ge h_\ast.
\]
另一方面，定理 \ref{thm:pom-escort-minentropy-zero-temperature-identity} 给出
\[
h_\infty^{\mathrm{esc}}(a)=P_a-a\alpha_\ast=R(a).
\]
若 \(h_\ast>0\)，则 \(R(a)\ge h_\ast>0\)，从而 \(h_\infty^{\mathrm{esc}}(a)>0\) 对所有整数 \(a\ge 1\) 成立；这便排除了精确冻结 \(h_\infty^{\mathrm{esc}}(a)=0\) 的可能性。证毕。
\end{proof}
