定理 \ref{thm:pom-diagonal-rate-critical-spectrum-secular-interlacing} 给出了临界非零谱的世俗方程。
在乘积分布下，该一维世俗函数在张量积意义下满足严格的乘法 Stieltjes 递推。

\begin{proposition}[乘积结构下的乘法 Stieltjes 递推]\label{prop:pom-diagonal-rate-critical-spectrum-stieltjes-tensor-recursion}
\leanverified{paper\_pom\_diagonal\_rate\_critical\_spectrum\_stieltjes\_tensor\_recursion}
对任意全支撑分布 \(w\in\Delta(X)\) 记
\[
A(w):=\sum_{x\in X}\sqrt{w(x)},\qquad
C(w):=A(w)^2-1,
\qquad
t_x(w):=\frac{A(w)}{\sqrt{w(x)}}.
\]
定义世俗函数
\[
\mathcal S_w(z):=\sum_{x\in X}\frac{1}{t_x(w)-z},
\qquad
\mathcal F_w(z):=1-\mathcal S_w(z),
\]
其零点刻画了定理 \ref{thm:pom-diagonal-rate-critical-spectrum-secular-interlacing} 中 \(CL_w\) 的非平凡谱。
\par
令 \(v\in\Delta(Y)\) 全支撑，\(\omega:=w\otimes v\in\Delta(X\times Y)\)。
则对任意 \((x,y)\in X\times Y\) 有
\(
t_{(x,y)}(\omega)=t_x(w)\,t_y(v)
\)，并且对任意复数 \(z\)（避开极点）成立严格恒等式
\[
\mathcal S_{w\otimes v}(z)
=\sum_{x\in X}\frac{1}{t_x(w)}\,\mathcal S_v\!\Bigl(\frac{z}{t_x(w)}\Bigr),
\qquad
\mathcal F_{w\otimes v}(z)
=1-\sum_{x\in X}\frac{1}{t_x(w)}\Bigl(1-\mathcal F_v\!\bigl(z/t_x(w)\bigr)\Bigr).
\]
\end{proposition}
\begin{proof}
由 \(A(w\otimes v)=A(w)A(v)\) 得
\[
t_{(x,y)}(\omega)
=\frac{A(w\otimes v)}{\sqrt{\omega(x,y)}}
=\frac{A(w)A(v)}{\sqrt{w(x)v(y)}}
=t_x(w)\,t_y(v).
\]
因此
\[
\mathcal S_{w\otimes v}(z)
=\sum_{x\in X}\sum_{y\in Y}\frac{1}{t_x(w)t_y(v)-z}
=\sum_{x\in X}\frac{1}{t_x(w)}\sum_{y\in Y}\frac{1}{t_y(v)-z/t_x(w)}
=\sum_{x\in X}\frac{1}{t_x(w)}\,\mathcal S_v\!\Bigl(\frac{z}{t_x(w)}\Bigr).
\]
\(\mathcal F\) 的恒等式由 \(\mathcal F=1-\mathcal S\) 代入整理即得。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$k$}{k} 重张量的迭代组合]\label{cor:pom-diagonal-rate-critical-spectrum-stieltjes-tensor-recursion-kfold}
\leanverified{paper\_pom\_diagonal\_rate\_critical\_spectrum\_stieltjes\_tensor\_recursion\_kfold}
取 \(k\ge 2\) 个全支撑分布 \(w^{(1)}\in\Delta(X_1),\dots,w^{(k)}\in\Delta(X_k)\)，并令 \(\omega:=\bigotimes_{i=1}^k w^{(i)}\)。
则对任意复数 \(z\)（避开极点）有迭代展开
\[
\mathcal S_{\omega}(z)
=\sum_{x_1\in X_1}\frac{1}{t_{x_1}(w^{(1)})}
\sum_{x_2\in X_2}\frac{1}{t_{x_2}(w^{(2)})}\cdots
\sum_{x_{k-1}\in X_{k-1}}\frac{1}{t_{x_{k-1}}(w^{(k-1)})}\,
\mathcal S_{w^{(k)}}\!\Bigl(\frac{z}{t_{x_1}(w^{(1)})\cdots t_{x_{k-1}}(w^{(k-1)})}\Bigr).
\]
\end{corollary}
\begin{proof}
对命题 \ref{prop:pom-diagonal-rate-critical-spectrum-stieltjes-tensor-recursion} 重复应用并递归展开即可。证毕。
\end{proof}
