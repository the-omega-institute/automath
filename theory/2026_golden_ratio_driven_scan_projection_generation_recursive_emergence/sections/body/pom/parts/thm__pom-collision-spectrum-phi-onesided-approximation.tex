\begin{theorem}[碰撞谱对 \texorpdfstring{$\varphi$}{phi} 的严格一侧逼近]\label{thm:pom-collision-spectrum-phi-onesided-approximation}
\leanverified{paper\_pom\_collision\_spectrum\_phi\_onesided\_approximation}
沿用命题 \ref{prop:pom-rq-qinfty-endpoint} 与注 \ref{rem:pom-recover-phi-from-collision-spectrum} 的记号。
对每个整数 \(q\ge 2\)，定义未经 \(q\)-方向加速的内部估计量
\[
\widehat\varphi_q^{\mathrm{raw}}
:=
4e^{-2h_q/q}
=
r_q^{2/q}.
\]
则成立严格的一侧夹逼
\[
\boxed{\;
\varphi\ \le\ \widehat\varphi_q^{\mathrm{raw}}\ \le\ \varphi^{\,1+2/q}
\qquad(q\ge 2).\;}
\]
等价地，
\[
0\le \log\!\left(\frac{\widehat\varphi_q^{\mathrm{raw}}}{\varphi}\right)\le \frac{2\log\varphi}{q},
\]
并且
\[
0\le \widehat\varphi_q^{\mathrm{raw}}-\varphi
\le
\varphi\bigl(\varphi^{2/q}-1\bigr).
\]
因此，仅凭已审计的碰撞谱根序列 \((r_q)\) 即可在无需 Fibonacci 先验输入的前提下，从上侧对 \(\varphi\) 给出一条完全内部的定量逼近链。
\end{theorem}
\begin{proof}
记
\[
L:=\log\!\left(\frac{2}{\sqrt{\varphi}}\right).
\]
由命题 \ref{prop:pom-rq-qinfty-endpoint}，对每个 \(q\ge 2\) 有
\[
0\le L-\frac{h_q}{q}\le \frac{\log\varphi}{q}.
\]
又由定义，
\[
\widehat\varphi_q^{\mathrm{raw}}
=
4e^{-2h_q/q}
=
4e^{-2L}\,e^{2(L-h_q/q)}
=
\varphi\,e^{2(L-h_q/q)}.
\]
由于 \(L-h_q/q\ge 0\)，立得
\[
\widehat\varphi_q^{\mathrm{raw}}\ge \varphi.
\]
另一方面，由 \(L-h_q/q\le (\log\varphi)/q\) 得
\[
\widehat\varphi_q^{\mathrm{raw}}
\le
\varphi\,e^{2\log\varphi/q}
=
\varphi^{\,1+2/q}.
\]
这给出第一组夹逼。

对两侧取对数，立即得到
\[
0\le \log\!\left(\frac{\widehat\varphi_q^{\mathrm{raw}}}{\varphi}\right)\le \frac{2\log\varphi}{q}.
\]
最后，
\[
\widehat\varphi_q^{\mathrm{raw}}-\varphi
=
\varphi\bigl(e^{2(L-h_q/q)}-1\bigr)
\le
\varphi\bigl(e^{2\log\varphi/q}-1\bigr)
=
\varphi\bigl(\varphi^{2/q}-1\bigr),
\]
即得误差上界。证毕。
\end{proof}
