\begin{lemma}[碰撞等价于模 \texorpdfstring{$F_{m+2}$}{F\_{m+2}} 同余]\label{lem:pom-fold-congruence}
设 \(\omega,\omega'\in\Omega_m=\{0,1\}^m\)，并定义窗口值
$$
N(\omega):=\sum_{k=1}^m \omega_k F_{k+1}.
$$
则有等价关系
$$
\Fold_m(\omega)=\Fold_m(\omega')
\quad\Longleftrightarrow\quad
N(\omega)\equiv N(\omega')\pmod{F_{m+2}}.
$$
\leanverified{Fold\_eq\_iff\_weight\_mod}
\end{lemma}
\begin{proof}
记 \(x=\Fold_m(\omega)\in X_m\)、\(x'=\Fold_m(\omega')\in X_m\)。由引理 \ref{lem:pom-one-bit}（单隐藏位分解）存在唯一 \(b(\omega),b(\omega')\in\{0,1\}\) 使
$$
N(\omega)=V_m(x)+b(\omega)F_{m+2},\qquad
N(\omega')=V_m(x')+b(\omega')F_{m+2}.
$$
若 \(x=x'\)，则两式相减得 \(N(\omega)-N(\omega')=(b(\omega)-b(\omega'))F_{m+2}\)，从而同余成立。
反之，若 \(N(\omega)\equiv N(\omega')\ (\mathrm{mod}\ F_{m+2})\)，则两者在模 \(F_{m+2}\) 下的余数相同。又由于 \(V_m(\Fold_m(\cdot))\) 正是 Zeckendorf 规范形在前 \(m\) 位对应的值（定义 \ref{def:pom-window-value}），该余数唯一确定折叠后的稳定类型 \(x\in X_m\)，因此 \(x=x'\)，即 \(\Fold_m(\omega)=\Fold_m(\omega')\)。证毕。
\end{proof}

\begin{proposition}[任意 \texorpdfstring{$q$}{q} 的傅里叶 moment-kernel 闭式]\label{prop:pom-moment-fourier-q}
\leanverified{paper\_pom\_moment\_fourier\_q}
令 $F:=F_{m+2}$，并在加法群 $G:=\ZZ/F\ZZ$ 上定义同余分布
$$
c_m(r):=\#\{\omega\in\Omega_m:\ N(\omega)\equiv r\ (\mathrm{mod}\ F)\},\qquad r\in G.
$$
取 $\omega_F:=e^{2\pi i/F}$，定义离散傅里叶变换
$$
\widehat c_m(t):=\sum_{r\in G} c_m(r)\,\omega_F^{tr},\qquad t\in G.
$$
则对任意整数 $q\ge 2$ 有显式公式
$$
\boxed{\;
S_q(m)=\frac{1}{F^{\,q-1}}
\sum_{t_1,\dots,t_{q-1}\in G}
\widehat c_m(t_1)\cdots \widehat c_m(t_{q-1})\,
\widehat c_m\!\Bigl(-\sum_{j=1}^{q-1}t_j\Bigr)\;}
$$
并且 $\widehat c_m$ 具有完全因子化形式
$$
\boxed{\;
\widehat c_m(t)=\sum_{\omega\in\Omega_m}\omega_F^{tN(\omega)}
=\prod_{k=1}^m\bigl(1+\omega_F^{tF_{k+1}}\bigr)\; }.
$$
\end{proposition}
\begin{proof}
因子化式由
$$
\widehat c_m(t)=\sum_{\omega_1,\dots,\omega_m\in\{0,1\}}\ \prod_{k=1}^m \omega_F^{t\omega_kF_{k+1}}
=\prod_{k=1}^m\sum_{\omega_k\in\{0,1\}} \omega_F^{t\omega_kF_{k+1}}
$$
直接得到。

为得一般 $q$ 的闭式，先用傅里叶反演
$$
c_m(r)=\frac{1}{F}\sum_{t\in G}\widehat c_m(t)\,\omega_F^{-tr}.
$$
代入 $S_q(m)=\sum_{r\in G}c_m(r)^q$ 并展开：
$$
S_q(m)
=\frac{1}{F^q}\sum_{t_1,\dots,t_q\in G}\widehat c_m(t_1)\cdots\widehat c_m(t_q)\sum_{r\in G}\omega_F^{-(t_1+\cdots+t_q)r}.
$$
由角色正交性，
$$
\sum_{r\in G}\omega_F^{-(t_1+\cdots+t_q)r}
=
\begin{cases}
F,& t_1+\cdots+t_q\equiv 0\ (\mathrm{mod}\ F),\\
0,& \text{否则}.
\end{cases}
$$
因此
$$
S_q(m)=\frac{1}{F^{q-1}}\sum_{\substack{t_1,\dots,t_q\in G\\ t_1+\cdots+t_q\equiv 0}}
\widehat c_m(t_1)\cdots\widehat c_m(t_q).
$$
再以 $t_q=-(t_1+\cdots+t_{q-1})$ 消去约束即得所述公式。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$S_2(m)$}{S2(m)} 的 Plancherel 表达与角色因子化]\label{prop:pom-s2-plancherel}
\leanverified{paper\_pom\_s2\_plancherel}
令 \(G_m:=\ZZ/F_{m+2}\ZZ\)，并取其标准角色
\(
\chi_j(r):=\exp(2\pi i jr/F_{m+2})
\)
（\(j\in G_m\)）。对同余计数 \(c_m(r)=\#\{\omega\in\Omega_m:\ N(\omega)\equiv r\}\) 定义离散傅里叶变换
\[
\widehat c_m(j):=\sum_{r\in G_m} c_m(r)\,\chi_j(r).
\]
则二阶碰撞矩满足严格恒等式
\[
\boxed{\;
S_2(m)=\sum_{r\in G_m}c_m(r)^2=\frac{1}{|G_m|}\sum_{j\in G_m}\bigl|\widehat c_m(j)\bigr|^2\;}
\]
并且 \(\widehat c_m(j)\) 具有完全因子化形式
\[
\boxed{\;
\widehat c_m(j)=\sum_{\omega\in\Omega_m}\chi_j\!\bigl(N(\omega)\bigr)
=\prod_{k=1}^m\Bigl(1+\chi_j(F_{k+1})\Bigr)
=\prod_{k=1}^m\left(1+\exp\!\frac{2\pi i j F_{k+1}}{F_{m+2}}\right).\;}
\]
从而
\[
S_2(m)
=\frac{1}{F_{m+2}}\sum_{j=0}^{F_{m+2}-1}\ \prod_{k=1}^m\left|1+\exp\!\frac{2\pi i j F_{k+1}}{F_{m+2}}\right|^2.
\]
\end{proposition}
\begin{proof}
第一式是有限阿贝尔群上的 Plancherel 恒等式。这里给出与本节记号一致的直接推导：由反演公式
\(
c_m(r)=\frac{1}{|G_m|}\sum_{j}\widehat c_m(j)\,\overline{\chi_j(r)}
\)
与角色正交性
\(
\sum_{r}\chi_j(r)\overline{\chi_{j'}(r)}=|G_m|\,\mathbf{1}_{j=j'}
\)
立得
\[
\sum_{r}c_m(r)^2
=\frac{1}{|G_m|}\sum_{j}|\widehat c_m(j)|^2.
\]

对因子化：由定义
\[
\widehat c_m(j)
=\sum_{r}\sum_{\substack{\omega\in\Omega_m\\N(\omega)\equiv r}}\chi_j(r)
=\sum_{\omega\in\Omega_m}\chi_j\!\bigl(N(\omega)\bigr).
\]
再用窗口值的线性分解 \(N(\omega)=\sum_{k=1}^m \omega_k F_{k+1}\)（定义 \ref{def:pom-window-value}）与角色的乘法性，
\[
\chi_j\!\bigl(N(\omega)\bigr)
=\prod_{k=1}^m \chi_j(F_{k+1})^{\omega_k},
\]
对每一位 \(\omega_k\in\{0,1\}\) 求和即可得到
\(
\widehat c_m(j)=\prod_{k=1}^m(1+\chi_j(F_{k+1}))
\)。
最后一式由 \(|G_m|=F_{m+2}\) 与 \(\chi_j(F_{k+1})=\exp(2\pi i jF_{k+1}/F_{m+2})\) 代入得到。证毕。
\end{proof}

\begin{remark}[最大非平凡傅里叶系数证书与临界共振的阻碍]\label{rem:pom-max-fourier-obstruction}
把同余计数归一化为概率分布 \(p_m(r):=c_m(r)/2^m\)，并记 \(\widehat p_m(j):=\widehat c_m(j)/2^m\)。
则命题 \ref{prop:pom-s2-plancherel} 等价于
\[
\sum_{r\in G_m}p_m(r)^2=\frac{1}{F_{m+2}}\sum_{j\in G_m}\bigl|\widehat p_m(j)\bigr|^2.
\]
若令 \(A_m:=\max_{j\in G_m,\ j\neq 0}|\widehat p_m(j)|\)，则立刻得到
\[
\sum_{r\in G_m}p_m(r)^2\le \frac{1}{F_{m+2}}\Bigl(1+(F_{m+2}-1)A_m^2\Bigr).
\]
然而附录 \ref{app:fold-multiplicity} 的临界共振定理 \ref{thm:fold-critical-resonance-constant} 证明沿共振频率 \(j=F_m,F_{m+1}\) 有
\(
|\widehat p_m(j)|\to C_\varphi>0
\)
（并由推论 \ref{cor:fold-critical-resonance-gap} 给出相应的碰撞下界）。
因此在 raw Fold 上，\(A_m\) 不可能随 \(m\) 指数衰减：若要在傅里叶侧导出“碰撞趋均匀/谱收缩”，需要控制的是频率\emph{平均能量}或对 edge/bulk 分解后的 bulk 大偏差，而非仅控制最大系数（另见注 \ref{rem:fold-edge-nondecay}）。
\end{remark}

\begin{remark}[严格交换的“Fold--Fourier--Parseval”桥包（交换图）]\label{rem:fold-fourier-parseval-bridge}
在命题 \ref{prop:pom-moment-fourier-q}--\ref{prop:pom-s2-plancherel} 的记号下，令 \(F:=F_{m+2}\)、\(G_m:=\ZZ/F\ZZ\)。

因此在 Fold 支路上，“空间/时间对偶”可被严格落成：空间侧的卷积（串行叠加）\(\Longleftrightarrow\) 频域侧的点乘（可分解的乘法链），并由 Parseval 把二者钉死到同一个可审计标尺上。
\end{remark}

\end{corollary}

\begin{proof}
第一不等式由定义 \(g_{\mathrm{col}}^2(m)=\mathsf{Col}_m-1/F_{m+2}\) 直接得到。
对第二不等式，由命题 \ref{prop:pom-s2-recurrence} 有
\(
S_2(m)=2\,\mathbf{1}^{\mathsf T}A_2^{\,m-2}\mathbf{1}
\)。
由于 \(\mathbf{1}\le v\) 且 \(A_2^{m-2}\) 非负，得
\[
\mathbf{1}^{\mathsf T}A_2^{\,m-2}\mathbf{1}\le \mathbf{1}^{\mathsf T}A_2^{\,m-2}v=\mathbf{1}^{\mathsf T}(r_2^{m-2}v)=r_2^{m-2}\,(\mathbf{1}^{\mathsf T}v).
\]
因此
\[
\mathsf{Col}_m=\frac{S_2(m)}{4^m}\le \frac{2(\mathbf{1}^{\mathsf T}v)}{4^m}\,r_2^{m-2}
=\frac{2(\mathbf{1}^{\mathsf T}v)}{r_2^2}\left(\frac{r_2}{4}\right)^m
=C_2\left(\frac{r_2}{4}\right)^m.
\]
证毕。
\end{proof}

\begin{lemma}[三阶递推的最小性]\label{lem:pom-s2-minimal-order}
\leanverified{paper\_pom\_s2\_minimal\_order}
序列 \(\{S_2(m)\}_{m\ge 2}\) 不满足任何二阶常系数线性递推。因而命题 \ref{prop:pom-s2-recurrence} 的三阶递推在“线性递推阶数”意义下是最小的。
\end{lemma}
\begin{proof}
反设存在常数 \(\alpha,\beta\) 使对所有 \(m\) 有 \(S_2(m)=\alpha S_2(m-1)+\beta S_2(m-2)\)。
由初值 \(S_2(2)=6,S_2(3)=14,S_2(4)=36,S_2(5)=88\) 解线性方程组
$$
36=\alpha\cdot 14+\beta\cdot 6,\qquad
88=\alpha\cdot 36+\beta\cdot 14
$$
得 \(\alpha=6/5,\ \beta=16/5\)。于是
$$
S_2(6)=\alpha S_2(5)+\beta S_2(4)=\frac65\cdot 88+\frac{16}{5}\cdot 36=\frac{1104}{5}=220.8,
$$
与 \(S_2(6)=220\)（命题 \ref{prop:pom-s2-recurrence} 或脚本 \path{scripts/exp_fold_collision_s2.py}）矛盾。故不存在二阶递推。证毕。
\leanverified{momentSum\_two\_minimal\_recurrence\_order}
\end{proof}

\begin{remark}[最小 realization 维数与非负核]\label{rem:pom-s2-minimal-realization}
由引理 \ref{lem:pom-s2-minimal-order}，\(\{S_2(m)\}\) 的最小线性递推阶数为 \(3\)，因此任何线性状态空间表示 \(S_2(m)=u^{\mathsf T}B^{m}v\) 的最小维数至少为 \(3\)。命题 \ref{prop:pom-s2-recurrence} 给出的三态矩阵 \(A_2\) 因而在维数意义下是最小的；并且它是一个\emph{非负整数} realization（可视为一张三态有限图的邻接矩阵），从而将 \(S_2\) 的指数尺度完全纳入 Perron--Frobenius/\(\zeta\) 的统一接口。
\end{remark}

\begin{remark}[Hankel 最小实现与 \texorpdfstring{$A_2$}{A2} 的非负相似标准形]\label{rem:pom-s2-hankel}
将序列重标为 \(a_n:=S_2(n+2)/2\)（\(n\ge 0\)）。由递推阶数为 \(3\) 可知 Hankel 矩阵族的秩为 \(3\)，因此存在一个三维的\textbf{最小线性实现}（加权自动机/线性表示的标准口径）：取
$$
H_0=(a_{i+j})_{0\le i,j\le 2},\qquad H_1=(a_{i+j+1})_{0\le i,j\le 2},
$$
则在 \(H_0\) 可逆时可定义“移位实现矩阵”
$$
A:=H_0^{-1}H_1,
$$
从而 \(a_n=u^{\mathsf T}A^{n}v\)（某些 \(u,v\)）给出最小 realization。该 Hankel 标准形 \(A\) 一般包含符号（例如出现负系数），但它与非负整数核 \(A_2\) 相似：存在行列式为 \(\pm 1\) 的整矩阵 \(P\) 使
$$
P^{-1}AP=A_2.
$$
上述 Hankel 构造与 unimodular 相似变换可由脚本 \path{scripts/exp_fold_collision_hankel_realization.py} 一键复现，从而把“最小 realization（可证明）”与“非负核（可解释为有限图）”严格对齐。
\end{remark}

\begin{theorem}[Hankel 秩定理（加权自动机的线性 Myhill--Nerode 口径）]\label{thm:hankel-rank-minimal}
\leanverified{paper\_hankel\_rank\_minimal}
设 \(k\) 为域（例如 \(\mathbb{Q}\) 或 \(\mathbb{R}\)）。给定一个序列 \(a=(a_n)_{n\ge 0}\subset k\)，其 Hankel 矩阵定义为
$$
H(a):=(a_{i+j})_{i,j\ge 0}.
$$
称 \((u,A,v)\) 为 \(a\) 的一个线性表示（linear representation），若 \(u,v\in k^d\)、\(A\in k^{d\times d}\)，且对一切 \(n\ge 0\) 有
$$
a_n=u^{\mathsf T}A^n v.
$$
则有
$$
\mathrm{rank}\bigl(H(a)\bigr)=\min\{d:\exists\ (u,A,v)\text{ 给出维数为 }d\text{ 的线性表示}\}.
$$
等价地，\(\mathrm{rank}(H(a))\) 等于生成该序列的最小加权自动机（有理级数）的状态数（见 \cite{BerstelReutenauer1988RationalSeries}）。
\end{theorem}

\begin{lemma}[Hankel 窗口行列式非零：秩至少为 \texorpdfstring{$3$}{3}]\label{lem:pom-s2-hankel-det}
令 \(a_n:=S_2(n+2)/2\)。取 \(3\times 3\) Hankel 窗口
$$
H_0:=
\begin{pmatrix}
a_0 & a_1 & a_2\\
a_1 & a_2 & a_3\\
a_2 & a_3 & a_4
\end{pmatrix}.
$$
则
$$
(a_0,a_1,a_2,a_3,a_4)=\Bigl(3,7,18,44,110\Bigr),
\qquad
\det(H_0)=-2\neq 0.
$$
因此 Hankel 秩至少为 \(3\)，从而任何线性 realization 的最小维数至少为 \(3\)。
\end{lemma}
\begin{proof}
由 \(S_2(2)=6,S_2(3)=14,S_2(4)=36,S_2(5)=88,S_2(6)=220\)（命题 \ref{prop:pom-s2-recurrence}），即得
\(
a_n=S_2(n+2)/2
\)
的前五项为 \((3,7,18,44,110)\)。直接代入计算 \(\det(H_0)\) 得 \(-2\)。证毕。
\leanverified{hankelS2\_det}
\end{proof}

\begin{theorem}[Hankel 秩恰为 \texorpdfstring{$3$}{3}，且三态为最小碰撞核]\label{thm:pom-s2-rank-exact}
\leanverified{paper\_pom\_s2\_rank\_exact}
令 \(a_n:=S_2(n+2)/2\) 并以 \(H=(a_{i+j})_{i,j\ge 0}\) 表示其 Hankel 矩阵。则：
\begin{enumerate}
  \item \(\mathrm{rank}(H)=3\)。
  \item \(a_n\) 的最小线性表示（等价于最小加权自动机）的状态维数为 \(3\)（由定理 \ref{thm:hankel-rank-minimal}）。
  \item 命题 \ref{prop:pom-s2-recurrence} 给出的三态非负整数核 \(A_2\) 在维数意义下是最小的碰撞核 realization。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)} 由引理 \ref{lem:pom-s2-hankel-det}，存在一个 \(3\times 3\) Hankel 主子式 \(\det(H_0)\neq 0\)，故 \(\mathrm{rank}(H)\ge 3\)。
另一方面，命题 \ref{prop:pom-s2-recurrence} 给出三维线性 realization
$$
a_n=\mathbf{1}^{\mathsf T}A_2^{\,n}\mathbf{1},
$$
因此 Hankel 矩阵的行/列空间包含于 \(\mathbb{R}^3\) 的一个 3 维线性表示所张成的子空间，从而 \(\mathrm{rank}(H)\le 3\)。合并得 \(\mathrm{rank}(H)=3\)。

\textbf{(2)} 由定理 \ref{thm:hankel-rank-minimal}，\(\mathrm{rank}(H)\) 等于最小线性表示维数；结合 (1) 的 \(\mathrm{rank}(H)=3\)，得最小维数为 \(3\)。

\textbf{(3)} 由 (2) 得任何实现的维数至少为 3，而 \(A_2\) 给出 3 维实现，故其在维数意义下最小。证毕。
\leanverified{hankelS2\_rank\_exact\_three}
\end{proof}

\begin{corollary}[碰撞指数与 \texorpdfstring{$h_2$}{h2}]\label{cor:pom-s2-asymptotic}
令 \(r_2\) 为多项式
$$
x^3-2x^2-2x+2=0
$$
的最大实根（\(r_2\approx 2.481194304092\)）。则
$$
S_2(m)=\Theta(r_2^m),
\qquad
\mathbb{P}\bigl[\Fold_m(\omega)=\Fold_m(\omega')\bigr]
=\Theta\!\left(\left(\frac{r_2}{4}\right)^m\right).
$$
进一步，令 \(w_m(x):=d_m(x)/2^m\) 为均匀微态经折叠后的输出分布，则其 \(\mathrm{R\acute{e}nyi}\)--2 投影熵率存在且为
$$
h_2:=\lim_{m\to\infty}\frac{1}{m}\Bigl(-\log\sum_{x\in X_m}w_m(x)^2\Bigr)
=\log\!\left(\frac{4}{r_2}\right)\approx 0.477554342632.
$$
\end{corollary}
\begin{proof}
由命题 \ref{prop:pom-s2-recurrence}，\(S_2(m)\) 的生成函数具有分母 \(1-2z-2z^2+2z^3\)，其收敛半径由该分母的最小模零点决定，等价于递推特征多项式的最大实根 \(r_2\)。因此 \(S_2(m)=\Theta(r_2^m)\)。又
$$
\sum_{x\in X_m}w_m(x)^2=\sum_x \frac{d_m(x)^2}{4^m}=\frac{S_2(m)}{4^m},
$$
代入渐近即可得 \(h_2=\log(4/r_2)\)。证毕。
\leanverified{momentSum\_two\_gt\_two\_fib\_strict}
\end{proof}

\begin{corollary}[二阶意义下的”近似均匀覆盖”：偏离常数的指数刻画]\label{cor:pom-renyi2-near-uniform}
\leanverified{paper\_pom\_renyi2\_near\_uniform}
令 \(w_m(x)=d_m(x)/2^m\) 为均匀微态的折叠推前，并令 \(u_m(x)\equiv 1/|X_m|=1/F_{m+2}\) 为 \(X_m\) 上的均匀分布。定义二阶碰撞比（亦即 Rényi--2 相对熵的指数化量）
\[
\begin{aligned}
R_m
&:=\frac{\sum_x w_m(x)^2}{\sum_x u_m(x)^2}\\
&=|X_m|\sum_x w_m(x)^2\\
&=\frac{F_{m+2}\,S_2(m)}{4^m}.
\end{aligned}
\]
则有指数率
\[
\boxed{
\begin{aligned}
\lim_{m\to\infty}\frac{1}{m}\log R_m
&=\log\!\left(\frac{\varphi\,r_2}{4}\right)\\
&\approx 3.657482\times 10^{-3}.
\end{aligned}
}.
\]
因此二阶意义下，\(w_m\) 偏离均匀分布的强度以基数 \(\varphi r_2/4\approx 1.003664\) 逐分辨率增长。
\end{corollary}
\begin{proof}
由定义，
\(
\sum_x u_m(x)^2=|X_m|\cdot |X_m|^{-2}=|X_m|^{-1}=F_{m+2}^{-1}
\)。
又由定义 \ref{def:pom-s2} 与推论 \ref{cor:pom-s2-asymptotic}，
\(
\sum_x w_m(x)^2=S_2(m)/4^m=\Theta((r_2/4)^m)
\)。
于是
\[
R_m=F_{m+2}\cdot \Theta\!\left(\left(\frac{r_2}{4}\right)^m\right)
=\Theta\!\left(\left(\frac{\varphi\,r_2}{4}\right)^m\right),
\]
其中使用 Fibonacci 渐近 \(F_{m+2}=\Theta(\varphi^m)\)。取对数除以 \(m\) 并令 \(m\to\infty\) 得结论。证毕。
\end{proof}

\begin{corollary}[熵谱夹逼：\texorpdfstring{$h_2\le h_1\le \log\varphi$}{h2<=h1<=log phi}]\label{cor:pom-shannon-entropy-squeeze}
\leanverified{paper\_pom\_shannon\_entropy\_squeeze}
令 \(w_m(x)=d_m(x)/2^m\) 为均匀微态的折叠推前，并定义其 Shannon 熵
\[
H_1(m):=-\sum_{x\in X_m}w_m(x)\log w_m(x).
\]
设其熵率上下极限为
\[
\underline h_1:=\liminf_{m\to\infty}\frac{1}{m}H_1(m),
\qquad
\overline h_1:=\limsup_{m\to\infty}\frac{1}{m}H_1(m).
\]
则有夹逼
\[
\boxed{\;
h_2\ \le\ \underline h_1\ \le\ \overline h_1\ \le\ \log\varphi\; }.
\]
从而 \(\overline h_1\) 与上界 \(\log\varphi\) 的间隙被二阶常数完全量化：
\[
0\ \le\ \log\varphi-\overline h_1\ \le\ \log\varphi-h_2
=\log\!\left(\frac{\varphi r_2}{4}\right),
\]
其中右端恰为推论 \ref{cor:pom-renyi2-near-uniform} 的偏离指数常数（约 \(3.657482\times 10^{-3}\)）。
\end{corollary}
\begin{proof}
对任意有限支持分布 \(\pi\)，Rényi 熵随阶数单调递减，故 \(H_2(\pi)\le H_1(\pi)\le H_0(\pi)\)；
这里 \(H_2(\pi)=-\log\sum_x\pi(x)^2\)、\(H_0(\pi)=\log|\mathrm{supp}(\pi)|\)。
应用于 \(\pi=w_m\) 得
\[
-\log\sum_x w_m(x)^2\ \le\ H_1(m)\ \le\ \log|X_m|=\log F_{m+2}.
\]
两端除以 \(m\) 并取 \(\liminf,\limsup\)，再用 \(\lim_{m\to\infty}\frac{1}{m}\log F_{m+2}=\log\varphi\) 与推论 \ref{cor:pom-s2-asymptotic} 的
\(
h_2=\lim_{m\to\infty}\frac{1}{m}\bigl(-\log\sum_x w_m(x)^2\bigr)
\)
即得第一组夹逼。不等式 \(\log\varphi-h_2=\log(\varphi r_2/4)\) 由 \(h_2=\log(4/r_2)\) 直接化简得到。证毕。
\end{proof}

\begin{corollary}[min-entropy 熵率的闭式端点]\label{cor:pom-min-entropy-rate}
\leanverified{paper\_pom\_min\_entropy\_rate}
令
\[
H_\infty(m):=-\log\max_{x\in X_m}w_m(x),
\qquad
h_\infty:=\lim_{m\to\infty}\frac{1}{m}H_\infty(m).
\]
则极限存在且满足闭式
\[
\boxed{\;
h_\infty=\log 2-\frac{1}{2}\log\varphi=\log\!\left(\frac{2}{\sqrt{\varphi}}\right).\;}
\]
\end{corollary}
\begin{proof}
由 \(\max_x w_m(x)=D_m/2^m\)（其中 \(D_m=\max_x d_m(x)\)），有

\(
\lim_{m\to\infty}\frac{1}{m}\log D_m=\frac{1}{2}\log\varphi
\)，
代入即得结论。证毕。
\end{proof}
