\begin{lemma}[移位 Fibonacci 的严格融合恒等式与正缺陷项]\label{lem:pom-shifted-fib-fusion-defect-positive}
取 Fibonacci 数列 \(F_0=0,F_1=1\)。
则对任意整数 \(a,b\ge 0\) 恒有恒等式
\begin{equation}\label{eq:pom-shifted-fib-fusion-defect}
\boxed{\;
F_{a+2}F_{b+2}=F_{a+b+2}+F_aF_b.\;}
\end{equation}
特别地，
\(
F_{a+2}F_{b+2}\ge F_{a+b+2}
\)，且严格性当且仅当 \(a,b\ge 1\)；
因此移位序列 \(G(n):=F_{n+2}\) 在加法半群 \((\NN_{\ge 1},+)\) 上严格超乘法，
其”融合缺陷”由正项 \(F_aF_b\) 精确给出。
\leanverified{fib\_shifted\_fusion\_defect}
\leanverified{paper\_pom\_shifted\_fib\_fusion\_defect\_positive}
\end{lemma}
\begin{proof}
由 Fibonacci 加法公式
\(
F_{m+n}=F_mF_{n+1}+F_{m-1}F_n
\)，取 \(m=a+2\)、\(n=b\) 得
\[
F_{a+b+2}=F_{a+2}F_{b+1}+F_{a+1}F_b.
\]
又由递推 \(F_{b+2}=F_{b+1}+F_b\) 与 \(F_{a+2}=F_{a+1}+F_a\)，有
\[
F_{a+2}F_{b+2}
=F_{a+2}(F_{b+1}+F_b)
=\underbrace{F_{a+2}F_{b+1}+F_{a+1}F_b}_{=\,F_{a+b+2}}
\;+\;F_aF_b,
\]
即得 \eqref{eq:pom-shifted-fib-fusion-defect}。
当且仅当 \(F_aF_b>0\)（即 \(a,b\ge 1\)）时为严格不等式。证毕。
\end{proof}

\begin{corollary}[二维行列式的 Fibonacci 体积律]\label{cor:pom-fib-determinant-volume-law}
对任意整数 \(a,b\ge 0\) 定义矩阵
\[
M(a,b):=\begin{pmatrix}
F_{a+2} & F_a\\
F_b & F_{b+2}
\end{pmatrix}.
\]
则其行列式满足严格恒等式
\[
\boxed{\;\det M(a,b)=F_{a+b+2}.\;}
\]
该恒等式把”移位乘积—融合缺陷”结构提升为一个确定性的面积（体积）律：二分量参数对 \((a,b)\) 在整数格上的面积像由单一 Fibonacci 数 \(F_{a+b+2}\) 完全刻画。
\leanverified{fib\_determinant\_volume\_law}
\end{corollary}
\begin{proof}
由引理 \ref{lem:pom-shifted-fib-fusion-defect-positive}，
\(
F_{a+2}F_{b+2}-F_aF_b=F_{a+b+2}
\)，而左端即 \(\det M(a,b)\)。证毕。
\end{proof}

\begin{theorem}[秩一融合缺陷的 Fibonacci 刚性与唯一正解]\label{thm:pom-rankone-fusion-defect-fibonacci-rigidity}
设实数序列 \((G_n)_{n\ge -2}\subset\RR\) 满足边界归一化
\[
G_{-2}=0,\qquad G_{-1}=1,\qquad G_0=1,
\]
并对任意整数 \(a,b\ge 0\) 满足融合缺陷恒等式
\begin{equation}\label{eq:pom-rankone-fusion-defect-star}
\boxed{\;
G_aG_b=G_{a+b}+G_{a-2}G_{b-2}.\;}
\end{equation}
若进一步假定 \(G_1>0\) 且 \(G_2>0\)，则对所有 \(n\ge -2\) 有
\[
G_n=F_{n+2},
\]
其中 \(F_0=0,F_1=1\) 为 Fibonacci 数列。
特别地 \(G_1=2,G_2=3\)，并且对所有 \(n\ge 0\) 成立二阶递推
\[
G_{n+2}=G_{n+1}+G_n.
\]
\end{theorem}
\leanverified{paper\_pom\_rankone\_fusion\_defect\_fibonacci\_rigidity}
\begin{proof}
记 \(x:=G_1\)。
对 \eqref{eq:pom-rankone-fusion-defect-star} 取 \((a,b)=(1,1)\) 得
\[
G_1^2=G_2+G_{-1}^2,
\qquad\Longrightarrow\qquad
G_2=x^2-1.
\]
取 \((a,b)=(2,1)\) 得
\[
G_2G_1=G_3+G_0G_{-1},
\qquad\Longrightarrow\qquad
G_3=xG_2-1.
\]
取 \((a,b)=(2,2)\) 得
\[
G_2^2=G_4+G_0^2,
\qquad\Longrightarrow\qquad
G_4=G_2^2-1.
\]
对一般 \(b=2\)，由 \eqref{eq:pom-rankone-fusion-defect-star} 得对任意 \(a\ge 0\),
\[
G_aG_2=G_{a+2}+G_{a-2}G_0=G_{a+2}+G_{a-2},
\]
即
\begin{equation}\label{eq:pom-rankone-fusion-defect-rec-b2}
G_{a+2}=G_2G_a-G_{a-2}.
\end{equation}
令 \(a=4\) 得
\[
G_6=G_2G_4-G_2=G_2^3-2G_2.
\]
再对 \eqref{eq:pom-rankone-fusion-defect-star} 取 \((a,b)=(3,3)\) 得
\[
G_3^2=G_6+G_1^2.
\]
将 \(G_2=x^2-1\)、\(G_3=x(x^2-1)-1\) 及 \(G_6=G_2^3-2G_2\) 代入并化简，得到一元约束
\[
x(x-2)(x-1)(x+1)=0.
\]
由 \(G_1=x>0\) 与 \(G_2=x^2-1>0\) 得 \(x>1\)，故唯一可行解为 \(x=2\)，于是 \(G_1=2,G_2=3\)。

最后，取 \eqref{eq:pom-rankone-fusion-defect-star} 的 \(b=1\) 与 \(b=2\) 两族，得到对所有 \(n\ge 0\),
\[
G_nG_1=G_{n+1}+G_{n-2}G_{-1}=G_{n+1}+G_{n-2},
\qquad
G_nG_2=G_{n+2}+G_{n-2}.
\]
两式相减并用 \(G_2-G_1=1\) 得
\[
G_{n+2}-G_{n+1}=G_n,
\]
即 \(G_{n+2}=G_{n+1}+G_n\)（\(n\ge 0\)）。
结合初值 \(G_{-2}=0,G_{-1}=1,G_0=1\)，得 \(G_n=F_{n+2}\) 对所有 \(n\ge -2\) 成立。证毕。
\end{proof}

\begin{theorem}[融合缺陷恒等式的有限检验基]\label{thm:pom-rankone-fusion-defect-finite-basis}
\leanverified{paper\_pom\_rankone\_fusion\_defect\_finite\_basis}
设 \((G_n)_{n\ge -2}\subset\RR\) 满足
\(
G_{-2}=0,G_{-1}=1,G_0=1
\)。
假定对所有 \(a\ge 0\) 成立两族一维恒等式
\begin{equation}\label{eq:pom-rankone-fusion-defect-A1}
G_aG_1=G_{a+1}+G_{a-2},
\end{equation}
\begin{equation}\label{eq:pom-rankone-fusion-defect-A2}
G_aG_2=G_{a+2}+G_{a-2},
\end{equation}
并满足单点约束
\begin{equation}\label{eq:pom-rankone-fusion-defect-A3}
G_3^2=G_6+G_1^2.
\end{equation}
若进一步假定 \(G_1>0\) 且 \(G_2>0\)，则 \((G_n)\) 必满足二维融合缺陷恒等式 \eqref{eq:pom-rankone-fusion-defect-star} 对所有 \(a,b\ge 0\) 成立。
\end{theorem}
\begin{proof}
由 \eqref{eq:pom-rankone-fusion-defect-A1} 取 \(a=1\) 得 \(G_2=G_1^2-1\)，故序列由参数 \(x:=G_1\) 控制。
由 \eqref{eq:pom-rankone-fusion-defect-A1} 与 \eqref{eq:pom-rankone-fusion-defect-A2} 可将 \(G_3,G_4,G_6\) 表达为 \(x\) 的多项式组合；再由 \eqref{eq:pom-rankone-fusion-defect-A3} 给出代数约束。
沿定理 \ref{thm:pom-rankone-fusion-defect-fibonacci-rigidity} 的低阶消元可得
\(
x(x-2)(x-1)(x+1)=0
\)。
在 \(G_1>0\) 且 \(G_2=G_1^2-1>0\) 下强制 \(x=2\)，从而 \(G_n=F_{n+2}\)（\(n\ge -2\)）。
由引理 \ref{lem:pom-shifted-fib-fusion-defect-positive}，移位 Fibonacci 满足
\(
F_{a+2}F_{b+2}=F_{a+b+2}+F_aF_b
\)；
将 \(G_n=F_{n+2}\) 代回即得 \eqref{eq:pom-rankone-fusion-defect-star}。证毕。
\end{proof}

\begin{proposition}[路径分量乘法多重度的细化单调性与极值界]\label{prop:pom-path-component-multiplicity-refinement-monotone-extrema}
\leanverified{paper\_pom\_path\_component\_multiplicity\_refinement\_monotone\_extrema}
定义路径分量长度 \(\ell_1,\dots,\ell_r\ge 1\) 的乘法多重度
\[
d(\ell_1,\dots,\ell_r):=\prod_{i=1}^r F_{\ell_i+2},
\qquad
L:=\sum_{i=1}^r \ell_i.
\]
若将两分量 \((a,b)\) 融合为单分量 \((a+b)\)（其中 \(a,b\ge 1\)），则多重度严格下降且下降量具有闭式：
\[
\boxed{\;
d(\ldots,a,b,\ldots)-d(\ldots,a+b,\ldots)
=\Big(\prod_{\text{其余 }i}F_{\ell_i+2}\Big)\,F_aF_b\ >\ 0.\;}
\]
因此 \(d\) 对“分量细化偏序”严格单调：任何细化操作（把某分量拆成两段正长度之和）都严格增大 \(d\)。
从而对任意分解 \(L=\sum_i\ell_i\) 恒有极值界
\[
\boxed{\;
F_{L+2}\ \le\ \prod_{i=1}^r F_{\ell_i+2}\ \le\ 2^L.\;}
\]
左端等号当且仅当 \(r=1\)（单分量）；右端等号当且仅当 \(\ell_i\equiv 1\)（全为长度 \(1\) 的最细分量）。
\leanverified{paper\_pom\_multiplicity\_lower\_eq\_iff}
\leanverified{paper\_pom\_multiplicity\_upper\_eq\_iff}
\end{proposition}
\begin{proof}
差值闭式由引理 \ref{lem:pom-shifted-fib-fusion-defect-positive} 取 \((a,b)\) 并乘以其余因子得到：
\[
F_{a+2}F_{b+2}-F_{a+b+2}=F_aF_b>0\qquad(a,b\ge 1).
\]
\par
\textbf{下界：}从任意分解 \((\ell_1,\dots,\ell_r)\) 出发，逐步融合两分量直至得到单分量 \((L)\)；
每一步融合都使 \(d\) 严格下降，故
\(
d(\ell_1,\dots,\ell_r)\ge d(L)=F_{L+2}
\)，且仅当 \(r=1\) 时取等号。
\par
\textbf{上界：}对每个 \(\ell\ge 1\) 用归纳可证 \(F_{\ell+2}\le 2^\ell\)：\(\ell=1\) 时 \(F_3=2\)；
若对 \(\ell-1,\ell-2\) 成立，则
\(
F_{\ell+2}=F_{\ell+1}+F_\ell\le 2^{\ell-1}+2^{\ell-2}\le 2^\ell
\)。
因此
\(
\prod_i F_{\ell_i+2}\le \prod_i 2^{\ell_i}=2^{\sum_i\ell_i}=2^L
\)。
若某 \(\ell_i\ge 2\)，则 \(F_{\ell_i+2}<2^{\ell_i}\)，从而整体严格小于 \(2^L\)；
故上界取等号当且仅当 \(\ell_i=1\) 对所有 \(i\) 成立。证毕。
\end{proof}

\begin{proposition}[固定部件数下的纤维乘法多重度极值律：极端碎裂与二元下包络]\label{prop:pom-multiplicity-fixed-r-extrema}
令 \(L\ge 1\) 且 \(1\le r\le L\)，并令
\[
\mathcal{C}_{L,r}:=\Bigl\{(\ell_1,\dots,\ell_r):\ \ell_i\in\NN_{\ge 1},\ \sum_{i=1}^r\ell_i=L\Bigr\}.
\]
对 \(\boldsymbol{\ell}=(\ell_1,\dots,\ell_r)\in\mathcal{C}_{L,r}\) 记
\(
d(\boldsymbol{\ell})=\prod_{i=1}^r F_{\ell_i+2}
\)。
则在 \(\mathcal{C}_{L,r}\) 上有如下严格极值结构（端点均唯一至置换）：
\begin{enumerate}
  \item \textbf{（极大值：极端碎裂）}有
  \begin{equation}\label{eq:pom-multiplicity-fixed-r-max}
  \boxed{\;\max_{\mathcal{C}_{L,r}} d(\boldsymbol{\ell})=2^{\,r-1}\,F_{L-r+3}\;}
  \end{equation}
  且取等号当且仅当 \(\boldsymbol{\ell}\) 为 \((1,\dots,1,L-r+1)\) 的排列。
  \item \textbf{（极小值：尽可能多的 \(2\)-部件）}令
  \(
  u:=L-r
  \)
  （相对全 \(1\)-基线的“可分配余长”）。
  若 \(u\le r\)（等价于 \(L\le 2r\)），则
  \begin{equation}\label{eq:pom-multiplicity-fixed-r-min-short}
  \boxed{\;\min_{\mathcal{C}_{L,r}} d(\boldsymbol{\ell})=2^{\,2r-L}\,3^{\,L-r}\;}
  \end{equation}
  且取等号当且仅当 \(\boldsymbol{\ell}\) 由 \(u\) 个 \(2\) 与 \(r-u\) 个 \(1\) 组成。
  若 \(u\ge r\)（等价于 \(L\ge 2r\)），则
  \begin{equation}\label{eq:pom-multiplicity-fixed-r-min-long}
  \boxed{\;\min_{\mathcal{C}_{L,r}} d(\boldsymbol{\ell})=3^{\,r-1}\,F_{L-2r+4}\;}
  \end{equation}
  且取等号当且仅当 \(\boldsymbol{\ell}\) 由 \(r-1\) 个 \(2\) 与一个长度 \(L-2r+2\) 的部件组成（注意此时 \(L-2r+2\ge 2\)）。
\end{enumerate}
\leanpartial{paper\_pom\_multiplicity\_fixed\_r\_extrema\_algebra\_core}{仅证明中的 3 个 Fibonacci 代数替换恒等式；极值定理本体（Finset 组合枚举）留后轮}
\leanverified{paper\_pom\_multiplicity\_fixed\_r\_extrema}
\leanverified{fib\_three\_mul}
\leanverified{two\_fib\_add\_four\_eq\_three\_fib\_add\_three\_plus\_fib}
\leanverified{fib\_add\_one\_eq\_mul\_add\_mul}
\leanverified{fib\_shifted\_fusion}
\leanverified{fib\_shifted\_fusion\_five}
\leanverified{two\_fib\_gt\_fib\_mul\_fib}
\leanverified{two\_fib\_gt\_three\_fib}
\end{proposition}
\begin{proof}
\textbf{极大值：}取任意 \(a,b\ge 2\)。由引理 \ref{lem:pom-shifted-fib-fusion-defect-positive} 与 Fibonacci 加法公式
\(
F_{m+n}=F_mF_{n+1}+F_{m-1}F_n
\)
得
\[
2F_{a+b+1}-F_{a+2}F_{b+2}
=\underbrace{(2F_{a+b+1}-F_{a+b+2})}_{=\,F_{a+b-1}}
-F_aF_b
=F_{a-1}F_{b-1}>0.
\]
因此对两分量替换
\(
(a,b)\mapsto(1,a+b-1)
\)
严格增大乘积 \(F_{a+2}F_{b+2}\mapsto 2F_{a+b+1}\)。
从任意 \(\boldsymbol{\ell}\in\mathcal{C}_{L,r}\) 出发，若存在至少两项 \(\ge 2\)，反复对其中两项应用上述替换，可严格增大 \(d\) 并使“长度 \(1\) 的部件数”每步增加 \(1\)。
过程必在有限步后终止于恰有 \(r-1\) 个 \(1\) 的分解，即为 \((1,\dots,1,L-r+1)\) 的排列；其值即为 \eqref{eq:pom-multiplicity-fixed-r-max}，且严格性给出端点唯一性。
\par
\textbf{极小值：第一步（消去可避免的 \(1\)）。}对任意 \(b\ge 3\)，有
\[
2F_{b+2}-3F_{b+1}
=2(F_{b+1}+F_b)-3F_{b+1}
=F_{b-2}>0,
\]
因此替换
\(
(1,b)\mapsto(2,b-1)
\)
严格减小乘积 \(2F_{b+2}\mapsto 3F_{b+1}\)。
故在极小化问题中，当 \(L>r\) 时一切仍出现的 \(1\) 都只能与所有其余部件长度 \(\le 2\) 相容；于是当 \(L\le 2r\) 时，极小解必由 \(1\) 与 \(2\) 组成，且由于 \(\sum_i\ell_i=L\) 约束其计数唯一，得到 \eqref{eq:pom-multiplicity-fixed-r-min-short} 与端点结构。
\par
\textbf{极小值：第二步（消去多余的 \(\ge 3\) 部件）。}在 \(L\ge 2r\) 情形下，已可令所有部件长度 \(\ge 2\)。
若存在两项 \(a,b\ge 3\)，则由引理 \ref{lem:pom-shifted-fib-fusion-defect-positive} 与恒等式 \(3F_n=F_{n+2}+F_{n-2}\)（\(n\ge 2\)）可得
\[
F_{a+2}F_{b+2}-3F_{a+b}
=\bigl(F_{a+b+2}+F_aF_b\bigr)-\bigl(F_{a+b+2}+F_{a+b-2}\bigr)
=F_aF_b-F_{a+b-2}
=F_{a-2}F_{b-2}>0,
\]
从而替换
\(
(a,b)\mapsto(2,a+b-2)
\)
严格减小乘积 \(F_{a+2}F_{b+2}\mapsto 3F_{a+b}\)。
反复应用该替换，可严格下降 \(d\) 并使“长度 \(\ge 3\) 的部件数”每步减少 \(1\)，最终得到至多一项 \(\ge 3\) 的分解；在 \(\sum_i\ell_i=L\) 与 \(r\) 固定下，该终态必为 \(r-1\) 个 \(2\) 与一个长度 \(L-2r+2\) 的部件，从而得到 \eqref{eq:pom-multiplicity-fixed-r-min-long} 与端点唯一性。证毕。
\end{proof}

\begin{corollary}[固定总长度下的“部件数—多重度”外包络：可达区间与单调分层]\label{cor:pom-multiplicity-fixed-L-r-envelope}
\leanverified{paper\_pom\_multiplicity\_fixed\_L\_r\_envelope}
对固定 \(L\ge 1\) 与 \(1\le r\le L\)，令
\[
\mathcal{D}_{L,r}:=\Bigl\{d(\boldsymbol{\ell}):\ \boldsymbol{\ell}\in\mathcal{C}_{L,r}\Bigr\}.
\]
则 \(\mathcal{D}_{L,r}\subseteq\bigl[d_{\min}(L,r),d_{\max}(L,r)\bigr]\)，其中端点由命题 \ref{prop:pom-multiplicity-fixed-r-extrema} 的闭式给出且均可达。
此外，对任意 \(1\le r<r'\le L\) 有严格单调分层
\[
\boxed{\ \min\mathcal{D}_{L,r}\ <\ \min\mathcal{D}_{L,r'},\qquad
\max\mathcal{D}_{L,r}\ <\ \max\mathcal{D}_{L,r'}\ } .
\]
因此部件数 \(r\) 在长度 \(L\) 下同时控制乘法多重度的上下边界，并将“可实现多重度”划分为按 \(r\) 严格上移的闭式外包络族。
\end{corollary}
\begin{proof}
区间外包络与端点可达由命题 \ref{prop:pom-multiplicity-fixed-r-extrema} 直接给出。
对单调性：由命题 \ref{prop:pom-path-component-multiplicity-refinement-monotone-extrema}，任取 \(\boldsymbol{\ell}\in\mathcal{C}_{L,r}\) 做一次细化（把某部件拆成两段正长度）得到 \(\boldsymbol{\ell}'\in\mathcal{C}_{L,r+1}\) 且 \(d(\boldsymbol{\ell}')>d(\boldsymbol{\ell})\)，故 \(\max\mathcal{D}_{L,r+1}>\max\mathcal{D}_{L,r}\)。
另一方面，任取 \(\boldsymbol{\ell}'\in\mathcal{C}_{L,r+1}\) 融合相邻两段得到 \(\boldsymbol{\ell}\in\mathcal{C}_{L,r}\) 且 \(d(\boldsymbol{\ell})<d(\boldsymbol{\ell}')\)，从而 \(\min\mathcal{D}_{L,r}<\min\mathcal{D}_{L,r+1}\)。
对 \(r<r'\) 迭代上述严格不等式即得结论。证毕。
\end{proof}

\begin{proposition}[多重度的二参数 Binet 正规形与全局误差界]\label{prop:pom-multiplicity-binet-two-parameter-normal-form}
\leanverified{paper\_pom\_multiplicity\_binet\_two\_parameter\_normal\_form}
令 \(\varphi=\frac{1+\sqrt5}{2}\)，并设 \(\ell_1,\dots,\ell_r\ge 1\)、\(L=\sum_i\ell_i\)。
则乘法多重度 \(d(\ell_1,\dots,\ell_r)=\prod_i F_{\ell_i+2}\) 具有精确分解
\[
\boxed{\;
d(\ell_1,\dots,\ell_r)
=\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr)^r\varphi^{L}\prod_{i=1}^r\bigl(1-(-\varphi^{-2})^{\ell_i+2}\bigr).\;}
\]
由于 \(\ell_i\ge 1\Rightarrow |(-\varphi^{-2})^{\ell_i+2}|\le \varphi^{-6}\)，从而得到统一双边界
\[
\boxed{\;
\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr)^r\varphi^{L}(1-\varphi^{-6})^r
\ \le\
d(\ell_1,\dots,\ell_r)
\ \le\
\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr)^r\varphi^{L}(1+\varphi^{-6})^r.\;}
\]
因此 \(d\) 在对数尺度上由两参数 \((L,r)\) 主导，而振荡项由显式指数 \(\varphi^{-6}\) 一致控制。
\end{proposition}
\begin{proof}
由 Binet 公式 \(F_n=(\varphi^n-\psi^n)/\sqrt5\) 且 \(\psi=-\varphi^{-1}\)，得
\[
F_{\ell+2}=\frac{\varphi^{\ell+2}}{\sqrt5}\Bigl(1-\Bigl(\frac{\psi}{\varphi}\Bigr)^{\ell+2}\Bigr)
=\frac{\varphi^{\ell+2}}{\sqrt5}\bigl(1-(-\varphi^{-2})^{\ell+2}\bigr).
\]
对 \(i\) 取乘积得到所示精确分解。
又因 \(\ell_i+2\ge 3\)，有
\(
|(-\varphi^{-2})^{\ell_i+2}|\le \varphi^{-6}
\)，
故每一因子落在区间 \([1-\varphi^{-6},\,1+\varphi^{-6}]\) 内；再取乘积即得双边界。证毕。
\end{proof}

\begin{proposition}[融合缺陷的结合律约束：Fibonacci \texorpdfstring{$2$}{2}-余因子恒等式]\label{prop:pom-fusion-defect-2cocycle-identity}
令 \(G(n):=F_{n+2}\)，并定义融合缺陷项
\(
\omega(a,b):=F_aF_b
\)（\(a,b\ge 0\)）。
由引理 \ref{lem:pom-shifted-fib-fusion-defect-positive} 的融合律
\(
G(a)G(b)=G(a+b)+\omega(a,b)
\)
与乘法结合律，推出对任意 \(a,b,c\ge 0\) 必有严格恒等式
\[
\boxed{\;
\omega(a,b)G(c)+\omega(a+b,c)
=\omega(b,c)G(a)+\omega(b+c,a)
=\omega(c,a)G(b)+\omega(c+a,b).\;}
\]
该恒等式表明 \(\omega\) 作为”融合缺陷项”满足非平凡的 \(2\)-余因子一致性：任意三分量的总缺陷与融合顺序无关，并由上述三种显式表达共同刻画。
\leanverified{fib\_fusion\_defect\_cocycle}
\leanverified{paper\_pom\_shifted\_fusion\_defect\_package}
\end{proposition}
\begin{proof}
对乘积 \(G(a)G(b)G(c)\) 分别按 \((ab)c\) 与 \(a(bc)\) 两种括号方式展开：
\[
\bigl(G(a)G(b)\bigr)G(c)=\bigl(G(a+b)+\omega(a,b)\bigr)G(c)=G(a+b+c)+\omega(a+b,c)+\omega(a,b)G(c),
\]
\[
G(a)\bigl(G(b)G(c)\bigr)=G(a)\bigl(G(b+c)+\omega(b,c)\bigr)=G(a+b+c)+\omega(a,b+c)+\omega(b,c)G(a),
\]
比较两式并消去公共项 \(G(a+b+c)\) 得
\[
\omega(a,b)G(c)+\omega(a+b,c)
=\omega(b,c)G(a)+\omega(a,b+c).
\]
由于 \(\omega\) 对称（\(\omega(a,b)=\omega(b,a)\)），有 \(\omega(a,b+c)=\omega(b+c,a)\)，从而得到第一条等式。
循环置换 \((a,b,c)\) 得三者相等。证毕。
\end{proof}
