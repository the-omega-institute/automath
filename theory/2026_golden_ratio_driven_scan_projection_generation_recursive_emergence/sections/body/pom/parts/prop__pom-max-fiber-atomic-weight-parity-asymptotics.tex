\begin{proposition}[最大原子权重的奇偶精细渐近与 min-entropy 常数项]\label{prop:pom-max-atomic-weight-parity-second-order}
\leanverified{paper\_pom\_max\_atomic\_weight\_parity\_second\_order}
沿用推论 \ref{cor:pom-max-fiber-rate-endpoint} 的记号，并设 \(\varphi=(1+\sqrt5)/2\)。
则存在仅依赖于奇偶的常数
\[
C_{\mathrm{par}}(2k)=\frac{\varphi^2}{\sqrt5},\qquad
C_{\mathrm{par}}(2k+1)=\frac{2\sqrt{\varphi}}{\sqrt5},
\]
使得当 \(m\to\infty\) 时
\[
p_{\max}(m)
:=C_{\mathrm{par}}(m)\left(\frac{\sqrt{\varphi}}{2}\right)^m\left(1+O(\varphi^{-m})\right).
\]
从而 min-entropy（自然对数）
\[
H_\infty(w_m):=-\log p_{\max}(m)
\]
具有奇偶可判别的两项展开
\[
H_\infty(w_m)
:=m\log\!\left(\frac{2}{\sqrt{\varphi}}\right)-\log C_{\mathrm{par}}(m)+O(\varphi^{-m}).
\]
\end{proposition}
\begin{proof}
由推论 \ref{cor:pom-max-fiber-rate-endpoint} 的精确式，
\[
p_{\max}(2k)=\frac{F_{k+2}}{2^{2k}}=\frac{F_{k+2}}{4^k},
\qquad
p_{\max}(2k+1)=\frac{F_{k+1}}{2^{2k}}=\frac{F_{k+1}}{4^k}.
\]
由 Binet 公式
\[
F_n=\frac{\varphi^n-(-\varphi^{-1})^n}{\sqrt5}
:=\frac{\varphi^n}{\sqrt5}\left(1+O(\varphi^{-2n})\right),
\]
得到
\[
p_{\max}(2k)
:=\frac{\varphi^{k+2}}{\sqrt5\,4^k}\left(1+O(\varphi^{-2k})\right)
:=\frac{\varphi^2}{\sqrt5}\left(\frac{\varphi}{4}\right)^k\left(1+O(\varphi^{-m})\right)
:=\frac{\varphi^2}{\sqrt5}\left(\frac{\sqrt\varphi}{2}\right)^{2k}\left(1+O(\varphi^{-m})\right),
\]
以及
\[
p_{\max}(2k+1)
:=\frac{\varphi^{k+1}}{\sqrt5\,4^k}\left(1+O(\varphi^{-2k})\right)
:=\frac{\varphi}{\sqrt5}\left(\frac{\varphi}{4}\right)^k\left(1+O(\varphi^{-m})\right)
:=\frac{2\sqrt{\varphi}}{\sqrt5}\left(\frac{\sqrt\varphi}{2}\right)^{2k+1}\left(1+O(\varphi^{-m})\right).
\]
这给出 \(p_{\max}(m)\) 的所示两分支渐近，并由定义 \(H_\infty(w_m)=-\log p_{\max}(m)\) 取负对数即可得到两项展开。证毕。
\end{proof}
