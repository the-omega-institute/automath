\begin{theorem}[稳定类型同余半环上的降阶加法缺陷与唯一进位元]\label{thm:pom-stable-addition-carry-defect-unique-element}
对每个 \(m\ge 1\) 记 \(M_m:=F_{m+2}\)，并沿用命题 \ref{prop:pom-fold-as-section} 的规范截面
\[
s_m:\ZZ/M_m\ZZ\to X_m,\qquad s_m(r)=\pi_m(Z(r)).
\]
令 \(V_m:X_m\to\{0,1,\dots,M_m-1\}\) 为值映射（定义 \ref{def:pom-window-value}），并用其把 \(X_m\) 与 \(\ZZ/M_m\ZZ\) 识别。
在该识别下，对 \(x,y\in X_m\) 定义由加法群诱导的稳定加法
\[
x\boxplus_m y
:=
s_m\bigl(V_m(x)+V_m(y)\ \mathrm{mod}\ M_m\bigr).
\]
再令 \(\pi_{m+1\to m}:X_{m+1}\to X_m\) 为前缀截断（定义 \ref{def:pi-m2-m1-golden}），并对 \(x,y\in X_{m+1}\) 定义二值进位指示
\[
\kappa^{\mathrm{car}}_{m+1}(x,y)
:=\ind_{\{V_{m+1}(x)+V_{m+1}(y)\ge M_{m+1}\}}\in\{0,1\}.
\]
定义唯一进位元
\[
\chi^{\mathrm{car}}_m:=s_m(F_m)\in X_m.
\]
则对任意 \(x,y\in X_{m+1}\) 有严格恒等式
\[
\boxed{\;
\pi_{m+1\to m}(x\boxplus_{m+1}y)
=
\pi_{m+1\to m}(x)\boxplus_m \pi_{m+1\to m}(y)\boxplus_m\Bigl(\kappa^{\mathrm{car}}_{m+1}(x,y)\cdot \chi^{\mathrm{car}}_m\Bigr)\;}
\]
其中 \(\kappa\cdot \chi^{\mathrm{car}}_m\) 表示：\(\kappa=0\) 时取 \(X_m\) 的幺元 \(s_m(0)\)，\(\kappa=1\) 时取 \(\chi^{\mathrm{car}}_m\)。

\leanverified{paper\_pom\_stable\_addition\_carry\_defect\_unique\_element}
\end{theorem}
\begin{proof}
令 \(a:=V_{m+1}(x)\)、\(b:=V_{m+1}(y)\)，并令 \(c\in\{0,1,\dots,M_{m+1}-1\}\) 满足
\[
c\equiv a+b\pmod{M_{m+1}}.
\]
由 \(\boxplus_{m+1}\) 的定义，\(x\boxplus_{m+1}y=s_{m+1}(c)\)，从而
\(
V_{m+1}(x\boxplus_{m+1}y)=c
\)。
对任意 \(z\in X_{m+1}\)，有
\(
V_{m+1}(z)=V_m(\pi_{m+1\to m}z)+z_{m+1}F_{m+2}
\)，故
\[
V_m(\pi_{m+1\to m}z)\equiv V_{m+1}(z)\pmod{M_m}.
\]
取 \(z=x\boxplus_{m+1}y\) 得
\[
V_m\bigl(\pi_{m+1\to m}(x\boxplus_{m+1}y)\bigr)\equiv c\pmod{M_m}.
\]
另一方面，由进位指示的定义，
\[
c=a+b-\kappa^{\mathrm{car}}_{m+1}(x,y)\,M_{m+1}.
\]
又由 Fibonacci 递推 \(M_{m+1}=F_{m+3}=F_{m+2}+F_{m+1}\equiv F_{m+1}\pmod{F_{m+2}}=\pmod{M_m}\)，得
\[
c\equiv a+b-\kappa^{\mathrm{car}}_{m+1}(x,y)\,F_{m+1}\pmod{M_m}.
\]
并且 \(-F_{m+1}\equiv F_m\pmod{F_{m+2}}=\pmod{M_m}\)，故
\[
c\equiv (a\bmod M_m)+(b\bmod M_m)+\kappa^{\mathrm{car}}_{m+1}(x,y)\,F_m\pmod{M_m}.
\]
将 \(a\bmod M_m\)、\(b\bmod M_m\) 分别解释为 \(V_m(\pi_{m+1\to m}x)\)、\(V_m(\pi_{m+1\to m}y)\)，并由 \(s_m\) 的定义把同余类拉回 \(X_m\)，即得所示恒等式。
其中 \(F_m\) 对应的稳定类型正是 \(\chi^{\mathrm{car}}_m=s_m(F_m)\)，故缺陷元唯一。证毕。
\end{proof}
