\paragraph{\texorpdfstring{$\bmod 3$}{mod 3} 标记更新的尖锐主项、坏分量统计与 \texorpdfstring{$\chi$}{chi} 异常自由能差}
在命题 \ref{prop:pom-multiplicity-composition-sharp-main-term-constant} 的简单极点主项、定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity} 的 \(\bmod 3\) 稀少事件指数界，以及定理 \ref{thm:pom-toggle-scan-sign-general-fiber}--\ref{thm:pom-toggle-time-reversal-sign-general-fiber} 的取向符号规则基础上，还可把“指数上界”进一步提升为带显式前因子的尖锐渐近与准幂型极限定理。

\begin{theorem}[两参数更新配分函数的简单极点常数在 \texorpdfstring{$u$}{u} 方向上的稳定延拓]\label{thm:pom-multiplicity-composition-mod3-simple-pole-u}
固定 \(q\ge 0\)。定义
\[
B_q(t):=\sum_{k\not\equiv 1\ (\mathrm{mod}\ 3)}F_{k+2}^{\,q}t^k,
\qquad
C_q(t):=\sum_{k\equiv 1\ (\mathrm{mod}\ 3)}F_{k+2}^{\,q}t^k,
\]
\[
A_q(t;u):=B_q(t)+u\,C_q(t)
=\sum_{k\ge 1}F_{k+2}^{\,q}u^{\mathbf 1_{\{k\equiv 1\ (\mathrm{mod}\ 3)\}}}t^k
\qquad (u\ge 0).
\]
再定义
\[
\mathcal Z^{(q)}(t;u):=\frac{1}{1-A_q(t;u)}
=\sum_{L\ge 0}Z_L^{(q)}(u)\,t^L.
\]
则对每个固定 \(u\ge 0\)，方程
\[
A_q(\rho_q(u);u)=1
\]
在区间 \((0,\varphi^{-q})\) 上有唯一解 \(\rho_q(u)\)，并记
\[
\lambda_q(u):=\rho_q(u)^{-1},
\qquad
\mu_q(u):=\rho_q(u)\,\partial_tA_q(\rho_q(u);u).
\]
则 \(t=\rho_q(u)\) 是 \(\mathcal Z^{(q)}(t;u)\) 的唯一支配简单极点，并存在 \(\vartheta_q(u)\in(0,1)\) 使
\begin{equation}\label{eq:pom-multiplicity-composition-mod3-simple-pole-u}
\boxed{\;
Z_L^{(q)}(u)
=
\frac{1}{\mu_q(u)}\,\lambda_q(u)^L
+ O\!\bigl(\lambda_q(u)^L\,\vartheta_q(u)^L\bigr)
\qquad (L\to\infty).}
\end{equation}
此外，对任意紧区间 \(I\Subset(0,\infty)\)，误差因子 \(\vartheta_q(u)\) 可在 \(u\in I\) 上一致选取。
\leanverified{paper\_pom\_multiplicity\_composition\_mod3\_simple\_pole\_u}
\end{theorem}
\begin{proof}
定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity} 已给出
\[
\mathcal Z^{(q)}(t;u)=\frac{1}{1-A_q(t;u)}.
\]
对每个固定 \(u\ge 0\)，函数 \(t\mapsto A_q(t;u)\) 在 \((0,\varphi^{-q})\) 上严格递增且系数非负，并且其支持集合始终包含互素步长 \(2,3\)。因此方程 \(A_q(t;u)=1\) 存在唯一正根 \(\rho_q(u)\in(0,\varphi^{-q})\)。又
\[
\partial_tA_q(\rho_q(u);u)>0,
\]
故 \(1-A_q(t;u)\) 在 \(t=\rho_q(u)\) 具有简单零点，从而 \(\mathcal Z^{(q)}(t;u)\) 在该点有简单极点。

由系数非负性与 Pringsheim 定理，\(\rho_q(u)\) 是收敛圆上的奇点。又因为支持集合含有互素步长 \(2,3\)，故 \(A_q(t;u)\) 非周期；于是若 \(|t|=\rho_q(u)\) 且 \(A_q(t;u)=1\)，由三角不等式取等条件只能有 \(t=\rho_q(u)\)。因此 \(\rho_q(u)\) 是唯一支配奇点。再由解析函数零点离散性，可取
\[
R_q(u)\in\bigl(\rho_q(u),\varphi^{-q}\bigr]
\]
使 \(\rho_q(u)<|t|<R_q(u)\) 内无其他奇点。对
\[
\vartheta_q(u):=\rho_q(u)/R_q(u)\in(0,1)
\]
应用标准留数传递，得到
\[
Z_L^{(q)}(u)
=
\frac{1}{\rho_q(u)\,\partial_tA_q(\rho_q(u);u)}\,\rho_q(u)^{-L}
+ O\!\bigl(R_q(u)^{-L}\bigr),
\]
即 \eqref{eq:pom-multiplicity-composition-mod3-simple-pole-u}。

若 \(I\Subset(0,\infty)\) 为紧区间，则 \(u\mapsto \rho_q(u)\) 由隐函数定理在 \(I\) 上解析，且上述“次近奇点模长”可在紧集上与支配根保持正距离；因此 \(\vartheta_q(u)\) 可在 \(I\) 上一致选取。证毕。
\end{proof}

\begin{theorem}[禁止坏分量与单缺陷层的尖锐主项]\label{thm:pom-multiplicity-composition-mod3-zero-one-defect-sharp}
在 \(P_L^{(q)}=\mathbb P_L^{(q)}\) 下，记
\[
A_L(\boldsymbol{\ell})=\#\{i:\ \ell_i\equiv 1\ (\mathrm{mod}\ 3)\}.
\]
并记
\[
\rho_q^{(0)}:=\rho_q(0),
\qquad
\lambda_q^{(0)}:=\lambda_q(0),
\qquad
\mu_q^{(0)}:=\mu_q(0).
\]
再沿用命题 \ref{prop:pom-multiplicity-composition-sharp-main-term-constant} 的记号
\[
\lambda(q)=\lambda_q(1),
\qquad
\mu_q=\mu_q(1).
\]
则有尖锐渐近：
\begin{enumerate}
  \item 禁止坏分量时，
  \[
  \boxed{\;
  \mathbb P_L^{(q)}(A_L=0)
  \sim
  \frac{\mu_q}{\mu_q^{(0)}}\Bigl(\frac{\lambda_q^{(0)}}{\lambda(q)}\Bigr)^L.}
  \]
  \item 恰有一个坏分量时，
  \[
  \boxed{\;
  \mathbb P_L^{(q)}(A_L=1)
  \sim
  \frac{\mu_q\,C_q(\rho_q^{(0)})}{(\mu_q^{(0)})^2}\,
  L\Bigl(\frac{\lambda_q^{(0)}}{\lambda(q)}\Bigr)^L.}
  \]
  \item 从而
  \[
  \boxed{\;
  \mathbb P_L^{(q)}(A_L\le 1)\sim \mathbb P_L^{(q)}(A_L=1),}
  \]
  即单缺陷层严格支配零缺陷层。
\end{enumerate}
\leanverified{paper\_pom\_multiplicity\_composition\_mod3\_zero\_one\_defect\_sharp}
\end{theorem}
\begin{proof}
由
\[
\mathcal Z^{(q)}(t;u)=\frac{1}{1-B_q(t)-uC_q(t)}
\]
直接取 \(u\)-系数可得
\[
[u^0]\mathcal Z^{(q)}(t;u)=\frac{1}{1-B_q(t)},
\qquad
[u^1]\mathcal Z^{(q)}(t;u)=\frac{C_q(t)}{(1-B_q(t))^2}.
\]
因此
\[
\sum_{L\ge 0}\mathbb P_L^{(q)}(A_L=0)\,Z_L^{(q)}\,t^L=\frac{1}{1-B_q(t)},
\]
\[
\sum_{L\ge 0}\mathbb P_L^{(q)}(A_L=1)\,Z_L^{(q)}\,t^L=\frac{C_q(t)}{(1-B_q(t))^2}.
\]

对第一式，定理 \ref{thm:pom-multiplicity-composition-mod3-simple-pole-u} 在 \(u=0\) 与命题 \ref{prop:pom-multiplicity-composition-sharp-main-term-constant} 在 \(u=1\) 给出
\[
[t^L]\frac{1}{1-B_q(t)}
=
\frac{1}{\mu_q^{(0)}}(\lambda_q^{(0)})^L
+ O\!\bigl((\lambda_q^{(0)}\vartheta)^L\bigr),
\]
\[
Z_L^{(q)}
=
\frac{1}{\mu_q}\lambda(q)^L+O\!\bigl(\lambda(q)^L\theta^L\bigr).
\]
相除即得第 \(1\) 条。

对第二式，注意 \(1-B_q(t)\) 在 \(t=\rho_q^{(0)}\) 处有简单零点，故
\[
1-B_q(t)=-(t-\rho_q^{(0)})\,B_q'(\rho_q^{(0)})+O\!\bigl((t-\rho_q^{(0)})^2\bigr).
\]
于是
\[
\frac{C_q(t)}{(1-B_q(t))^2}
=
\frac{C_q(\rho_q^{(0)})}{(\rho_q^{(0)})^2(B_q'(\rho_q^{(0)}))^2}\,
\frac{1}{(1-t/\rho_q^{(0)})^2}
+ O\!\left(\frac{1}{1-t/\rho_q^{(0)}}\right).
\]
由标准二阶极点传递得到
\[
[t^L]\frac{C_q(t)}{(1-B_q(t))^2}
\sim
\frac{C_q(\rho_q^{(0)})}{(\rho_q^{(0)})^2(B_q'(\rho_q^{(0)}))^2}\,
L\,(\lambda_q^{(0)})^L.
\]
而
\[
\mu_q^{(0)}=\rho_q^{(0)}B_q'(\rho_q^{(0)}),
\]
故上式可重写为
\[
[t^L]\frac{C_q(t)}{(1-B_q(t))^2}
\sim
\frac{C_q(\rho_q^{(0)})}{(\mu_q^{(0)})^2}\,L\,(\lambda_q^{(0)})^L.
\]
再除以 \(Z_L^{(q)}\sim \mu_q^{-1}\lambda(q)^L\) 即得第 \(2\) 条。

最后，由第 \(1\) 与第 \(2\) 条，
\[
\frac{\mathbb P_L^{(q)}(A_L=0)}{\mathbb P_L^{(q)}(A_L=1)}
\sim
\frac{\mu_q^{(0)}}{C_q(\rho_q^{(0)})}\,\frac{1}{L}\longrightarrow 0,
\]
故第 \(3\) 条成立。证毕。
\end{proof}

\begin{theorem}[坏分量计数的准幂型、线性密度与 Gaussian 波动]\label{thm:pom-multiplicity-composition-mod3-badcount-clt}
对固定 \(q\ge 0\)，定义
\[
\psi_q(s):=\log\lambda_q(e^s)-\log\lambda(q),
\qquad s\in\RR.
\]
则在 \(P_L^{(q)}\) 下，坏分量计数 \(A_L\) 满足：
\begin{enumerate}
  \item 对任意紧区间 \(K\Subset\RR\)，存在 \(\vartheta_{q,K}\in(0,1)\)，使得在 \(s\in K\) 上一致有
  \[
  \boxed{\;
  \mathbb E_{P_L^{(q)}}[e^{sA_L}]
  =
  \frac{Z_L^{(q)}(e^s)}{Z_L^{(q)}}
  =
  \frac{\mu_q}{\mu_q(e^s)}\,e^{L\psi_q(s)}
  \bigl(1+O(\vartheta_{q,K}^L)\bigr).}
  \]
  \item 因而存在
  \[
  m_q:=\psi_q'(0),
  \qquad
  \sigma_q^2:=\psi_q''(0)>0,
  \]
  使得
  \[
  \frac{A_L}{L}\xrightarrow{\mathbb P} m_q,
  \qquad
  \frac{A_L-Lm_q}{\sqrt L}\Longrightarrow \mathcal N(0,\sigma_q^2).
  \]
  \item 若记
  \[
  \rho_q:=\rho_q(1)=\rho(q),
  \qquad
  \mu_q=\mu_q(1),
  \]
  则一阶斜率具有闭式
  \[
  \boxed{\;
  m_q=\frac{C_q(\rho_q)}{\mu_q}.}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
对任意紧区间 \(K\Subset\RR\)，其像 \(e^K\Subset(0,\infty)\) 为紧集。由定理 \ref{thm:pom-multiplicity-composition-mod3-simple-pole-u} 的一致版本，
\[
Z_L^{(q)}(e^s)
=
\frac{1}{\mu_q(e^s)}\,\lambda_q(e^s)^L
\bigl(1+O(\vartheta_{q,K}^L)\bigr)
\]
在 \(s\in K\) 上一致成立，而
\[
Z_L^{(q)}
=
\frac{1}{\mu_q}\lambda(q)^L\bigl(1+O(\theta^L)\bigr).
\]
相除即得第 \(1\) 条。

于是 \(\mathbb E[e^{sA_L}]\) 具有标准 quasi-powers 形态，故由解析准幂定理得到第 \(2\) 条的大数律与中心极限定理。

为计算导数，对任意 \(s\in\RR\) 定义倾斜更新分布
\[
p_{q,s}(k):=F_{k+2}^{\,q}e^{s\mathbf 1_{\{k\equiv 1\ (\mathrm{mod}\ 3)\}}}\rho_q(e^s)^k,
\qquad k\ge 1.
\]
由根方程 \(A_q(\rho_q(e^s);e^s)=1\) 可知 \(\sum_{k\ge 1}p_{q,s}(k)=1\)。令
\[
K(k):=k,
\qquad
Y(k):=\mathbf 1_{\{k\equiv 1\ (\mathrm{mod}\ 3)\}},
\]
并以 \(\mathbb E_s,\operatorname{Var}_s\) 表示在 \(p_{q,s}\) 下的期望与方差。将 \(\sum_k p_{q,s}(k)=1\) 对 \(s\) 求导得
\[
\mathbb E_s[Y]-\psi_q'(s)\,\mathbb E_s[K]=0,
\]
故
\begin{equation}\label{eq:pom-multiplicity-composition-mod3-badcount-first-derivative}
\psi_q'(s)=\frac{\mathbb E_s[Y]}{\mathbb E_s[K]}.
\end{equation}
再利用
\[
\partial_s\log p_{q,s}(k)=Y(k)-\psi_q'(s)K(k)
\]
与 \(\mathbb E_s[\partial_s\log p_{q,s}(K)]=0\)，可得
\[
\psi_q''(s)=\frac{\operatorname{Var}_s\!\bigl(Y-\psi_q'(s)K\bigr)}{\mathbb E_s[K]}.
\]
由于 \(Y\) 不是 \(K\) 在 \(\{1,2,3,\dots\}\) 上的仿射函数，故方差严格为正，从而 \(\sigma_q^2=\psi_q''(0)>0\)。

最后在 \(s=0\) 时有
\[
\mathbb E_0[Y]=\sum_{k\equiv 1\ (3)}F_{k+2}^{\,q}\rho_q^k=C_q(\rho_q),
\qquad
\mathbb E_0[K]=\rho_q A_q'(\rho_q)=\mu_q.
\]
将其代入 \eqref{eq:pom-multiplicity-composition-mod3-badcount-first-derivative} 即得第 \(3\) 条。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\chi$}{chi} 异常速率猜想的严格证明]\label{thm:pom-multiplicity-composition-chi-anomaly-sharp-rate}
在推论 \ref{cor:pom-multiplicity-composition-witten-sign-exp-sparse} 的同一识别下取 \(q=1\)。记
\[
\rho_0:=\rho_1(0),
\qquad
\lambda_0:=\lambda_1(0),
\qquad
\mu_0:=\mu_1(0),
\]
\[
\lambda_1:=\lambda_1(1)=\lambda(1),
\qquad
\mu_1:=\mu_1(1).
\]
并定义两类“单坏分量负号”生成函数
\[
C_c^-(t):=\sum_{\ell\equiv 1\ (\mathrm{mod}\ 6)}F_{\ell+2}\,t^\ell,
\qquad
C_\iota^-(t):=\sum_{\ell\equiv 4,7\ (\mathrm{mod}\ 12)}F_{\ell+2}\,t^\ell.
\]
则当 \(L\to\infty\) 时成立尖锐渐近
\[
\boxed{\;
\mathbb P_L^{(1)}\!\bigl(\operatorname{sgn}(c_x)=-1\bigr)
\sim
\frac{\mu_1\,C_c^-(\rho_0)}{\mu_0^2}\,
L\Bigl(\frac{\lambda_0}{\lambda_1}\Bigr)^L,}
\]
\[
\boxed{\;
\mathbb P_L^{(1)}\!\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)
\sim
\frac{\mu_1\,C_\iota^-(\rho_0)}{\mu_0^2}\,
L\Bigl(\frac{\lambda_0}{\lambda_1}\Bigr)^L.}
\]
因而
\[
\boxed{\;
-\lim_{L\to\infty}\frac{1}{L}\log \mathbb P_L^{(1)}\!\bigl(\operatorname{sgn}(c_x)=-1\bigr)
=
-\lim_{L\to\infty}\frac{1}{L}\log \mathbb P_L^{(1)}\!\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)
=
\log\frac{\lambda_1}{\lambda_0}.}
\]
特别地，猜想 \ref{conj:pom-chi-anom-rate-free-energy-gap} 成立，且
\[
I_c=I_\iota=\log\frac{\lambda_1(1)}{\lambda_1(0)}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 与 \ref{thm:pom-toggle-time-reversal-sign-general-fiber}，一旦 \(A_L\ge 2\)，必有
\[
\operatorname{sgn}(c_x)=\operatorname{sgn}(\iota_x)=+1.
\]
故两类负号事件都只能来自 \(A_L=0\) 或 \(A_L=1\)。

先看扫描符号。若 \(A_L=1\)，由定理 \ref{thm:pom-toggle-scan-sign-general-fiber}，唯一坏分量长度 \(\ell\) 满足
\[
\operatorname{sgn}(c_x)=-1
\iff
(-1)^{e(\ell)}=-1.
\]
又在约束 \(\ell\equiv 1\ (\mathrm{mod}\ 3)\) 下，仅有 \(\ell\equiv 1,4\ (\mathrm{mod}\ 6)\) 两类，而 \(e(\ell)=1\) 当且仅当 \(\ell\equiv 1\ (\mathrm{mod}\ 6)\)。因此“恰有一个坏分量且扫描符号为负”的生成函数正是
\[
\frac{C_c^-(t)}{(1-B_1(t))^2}.
\]
与定理 \ref{thm:pom-multiplicity-composition-mod3-zero-one-defect-sharp} 第 \(2\) 条完全同样的二阶极点传递给出
\[
[t^L]\frac{C_c^-(t)}{(1-B_1(t))^2}
\sim
\frac{C_c^-(\rho_0)}{\mu_0^2}\,L\,\lambda_0^L.
\]
再除以
\[
Z_L^{(1)}\sim \mu_1^{-1}\lambda_1^L
\]
便得到第一条渐近。

对时间反演符号，由定理 \ref{thm:pom-toggle-time-reversal-sign-general-fiber}，当 \(A_L=1\) 时有
\[
\operatorname{sgn}(\iota_x)=\delta(\ell).
\]
而定理 \ref{thm:pom-toggle-time-reversal-sign-mod12} 给出
\[
\delta(\ell)=-1
\iff
\ell\equiv 2,3,4,6,7,8\ (\mathrm{mod}\ 12).
\]
在坏分量约束 \(\ell\equiv 1\ (\mathrm{mod}\ 3)\) 下，允许的同余类仅为 \(1,4,7,10\ (\mathrm{mod}\ 12)\)，故
\[
\delta(\ell)=-1
\iff
\ell\equiv 4,7\ (\mathrm{mod}\ 12).
\]
因此“恰有一个坏分量且时间反演符号为负”的生成函数正是
\[
\frac{C_\iota^-(t)}{(1-B_1(t))^2}.
\]
同样的二阶极点传递给出
\[
[t^L]\frac{C_\iota^-(t)}{(1-B_1(t))^2}
\sim
\frac{C_\iota^-(\rho_0)}{\mu_0^2}\,L\,\lambda_0^L,
\]
再除以 \(Z_L^{(1)}\sim \mu_1^{-1}\lambda_1^L\) 即得第二条渐近。

最后，由定理 \ref{thm:pom-multiplicity-composition-mod3-zero-one-defect-sharp} 第 \(1\) 条，
\[
\mathbb P_L^{(1)}(A_L=0)=O\!\left(\Bigl(\frac{\lambda_0}{\lambda_1}\Bigr)^L\right),
\]
少一个线性因子 \(L\)，故不会改变负号事件的主项。于是两类负号事件共享同一指数底 \(\lambda_0/\lambda_1\)，从而得到速率公式，并由定义识别为
\[
I_c=I_\iota=\log\frac{\lambda_1}{\lambda_0}
=
\log\frac{\lambda_1(1)}{\lambda_1(0)}.
\]
证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\chi$}{chi} 异常层相对于 supersymmetric 层的精确层级]\label{cor:pom-multiplicity-composition-chi-vs-supersym}
在 \(q=1\) 下，定义
\[
\mathcal S_L:=\{x:\ s_m(x)\neq 0\},
\qquad
\mathcal C_L^-:=\{x:\ \operatorname{sgn}(c_x)=-1\},
\qquad
\mathcal I_L^-:=\{x:\ \operatorname{sgn}(\iota_x)=-1\}.
\]
则存在正常数
\[
A_{\mathrm{sup}}:=\frac{\mu_1}{\mu_0},
\qquad
A_c:=\frac{\mu_1\,C_c^-(\rho_0)}{\mu_0^2},
\qquad
A_\iota:=\frac{\mu_1\,C_\iota^-(\rho_0)}{\mu_0^2},
\]
使
\[
\mathbb P_L^{(1)}(\mathcal S_L)\sim
A_{\mathrm{sup}}\Bigl(\frac{\lambda_0}{\lambda_1}\Bigr)^L,
\]
\[
\mathbb P_L^{(1)}(\mathcal C_L^-)\sim
A_c\,L\Bigl(\frac{\lambda_0}{\lambda_1}\Bigr)^L,
\qquad
\mathbb P_L^{(1)}(\mathcal I_L^-)\sim
A_\iota\,L\Bigl(\frac{\lambda_0}{\lambda_1}\Bigr)^L.
\]
因而
\[
\boxed{\;
\frac{\mathbb P_L^{(1)}(\mathcal C_L^-)}{\mathbb P_L^{(1)}(\mathcal S_L)}
\sim
\frac{C_c^-(\rho_0)}{\mu_0}\,L,
\qquad
\frac{\mathbb P_L^{(1)}(\mathcal I_L^-)}{\mathbb P_L^{(1)}(\mathcal S_L)}
\sim
\frac{C_\iota^-(\rho_0)}{\mu_0}\,L.}
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-fiber-rewrite-parity-exact-split-mod3-obstruction}，
\[
s_m(x)\neq 0
\iff
A_L=0.
\]
故 \(\mathcal S_L\) 的渐近正是定理 \ref{thm:pom-multiplicity-composition-mod3-zero-one-defect-sharp} 第 \(1\) 条在 \(q=1\) 的专门化。两类负号事件的渐近则由定理 \ref{thm:pom-multiplicity-composition-chi-anomaly-sharp-rate} 给出。逐项相除即得所示比值。证毕。
\end{proof}
