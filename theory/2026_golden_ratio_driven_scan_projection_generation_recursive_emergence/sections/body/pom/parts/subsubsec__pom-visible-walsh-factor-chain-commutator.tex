\subsubsection{折叠可见 Walsh 系统、Parseval 再生与交换子缺陷}\label{subsubsec:pom-visible-walsh-factor-chain-commutator}

令 \(\mu_m\) 为 \(\Omega_m=\{0,1\}^m\) 上的均匀分布，\(\pi_m:=P_m\mu_m\) 为其在 \(X_m\) 上的折叠推前。
定义拉回等距嵌入
\[
U_m:L^2(X_m,\pi_m)\to L^2(\Omega_m,\mu_m),
\qquad
(U_m f)(\omega):=f(\Fold_m(\omega)).
\]
其伴随算子满足
\[
(U_m^\ast h)(x)=\frac{1}{d_m(x)}\sum_{\omega\in\Fold_m^{-1}(x)}h(\omega),
\qquad h\in L^2(\Omega_m,\mu_m),\ x\in X_m.
\]
由命题 \ref{prop:fold-conditional-expectation} 与定理 \ref{prop:fold-pushforward-lift}，
\(U_m^\ast\) 正是函数口径下的纤维平均，而折叠条件期望 \(E_m\) 满足
\[
E_m=U_mU_m^\ast.
\]
仍记 \(P_m^\square\) 为超立方体简单随机游走在 \(L^2(\Omega_m,\mu_m)\) 上的 Markov 算子，并把定义 \ref{def:pom-fold-factor-chain} 的折叠因子链在 \(L^2(X_m,\pi_m)\) 上的作用写为
\[
\overline P_m^\square:=U_m^\ast P_m^\square U_m.
\]

\begin{theorem}[折叠可见 Walsh 系统的 Parseval 再生]\label{thm:pom-visible-walsh-parseval-frame}
对任意 \(I\subseteq[m]\) 定义 Walsh 字符
\[
\chi_I(\omega):=(-1)^{\sum_{i\in I}\omega_i}\qquad(\omega\in\Omega_m),
\]
以及其折叠可见像
\[
g_I:=U_m^\ast\chi_I\in L^2(X_m,\pi_m).
\]
则对任意 \(x\in X_m\) 有
\[
g_I(x)=\frac{1}{d_m(x)}\sum_{\omega\in\Fold_m^{-1}(x)}\chi_I(\omega)
=\frac{\mathcal H_x(\varepsilon^I)}{d_m(x)},
\]
其中 \(\varepsilon^I\in\{\pm1\}^m\) 与推论 \ref{cor:pom-fiber-walsh-hardcore-stokes-sign} 的记号一致。
此外，族 \(\{g_I\}_{I\subseteq[m]}\) 构成 \(L^2(X_m,\pi_m)\) 的 Parseval 框架：对任意 \(f\in L^2(X_m,\pi_m)\)，
\[
\|f\|_{L^2(\pi_m)}^2
=\sum_{I\subseteq[m]}\bigl|\langle f,g_I\rangle_{\pi_m}\bigr|^2,
\]
且
\[
f=\sum_{I\subseteq[m]}\langle f,g_I\rangle_{\pi_m}\,g_I.
\]
\end{theorem}
\begin{proof}
显式公式由推论 \ref{cor:pom-fiber-walsh-hardcore-stokes-sign} 立即给出。
由于 \(U_m\) 为等距嵌入，而 \(\{\chi_I\}_{I\subseteq[m]}\) 是 \(L^2(\Omega_m,\mu_m)\) 的正交标准基，
对任意 \(f\in L^2(X_m,\pi_m)\) 有
\[
\|f\|_{L^2(\pi_m)}^2
=\|U_m f\|_{L^2(\mu_m)}^2
=\sum_{I\subseteq[m]}\bigl|\langle U_m f,\chi_I\rangle_{\mu_m}\bigr|^2
=\sum_{I\subseteq[m]}\bigl|\langle f,U_m^\ast\chi_I\rangle_{\pi_m}\bigr|^2.
\]
即得 Parseval 恒等式。
再生公式由 Walsh 展开
\(
U_m f=\sum_{I\subseteq[m]}\langle U_m f,\chi_I\rangle_{\mu_m}\chi_I
\)
对两侧施加 \(U_m^\ast\)，并用 \(U_m^\ast U_m=\mathrm{Id}\) 得出。证毕。
\end{proof}

\leanverified{paper\_pom\_visible\_walsh\_parseval\_frame}

\begin{theorem}[超立方体 Walsh 谱继承的交换子判据]\label{thm:pom-visible-walsh-commutator-defect}
对任意 \(I\subseteq[m]\) 记
\[
\lambda_I:=1-\frac{2|I|}{m},
\qquad
[P_m^\square,E_m]:=P_m^\square E_m-E_mP_m^\square.
\]
则折叠可见 Walsh 模式满足精确缺陷公式
\[
(\overline P_m^\square-\lambda_I\mathrm{Id})g_I
=U_m^\ast[P_m^\square,E_m]\chi_I.
\]
因此，\(g_I\) 成为 \(\overline P_m^\square\) 在本征值 \(\lambda_I\) 处的本征函数，当且仅当
\[
U_m^\ast[P_m^\square,E_m]\chi_I=0.
\]
\end{theorem}
\begin{proof}
由 \(E_m=U_mU_m^\ast\) 与 \(g_I=U_m^\ast\chi_I\)，
\[
\overline P_m^\square g_I
=U_m^\ast P_m^\square U_mU_m^\ast\chi_I
=U_m^\ast P_m^\square E_m\chi_I.
\]
将右端拆为
\[
U_m^\ast E_mP_m^\square\chi_I
+U_m^\ast[P_m^\square,E_m]\chi_I.
\]
又由于 \(U_m^\ast E_m=U_m^\ast\)，且 Walsh 字符满足标准谱关系
\(
P_m^\square\chi_I=\lambda_I\chi_I
\)，
遂得
\[
\overline P_m^\square g_I
=\lambda_I U_m^\ast\chi_I+U_m^\ast[P_m^\square,E_m]\chi_I
=\lambda_I g_I+U_m^\ast[P_m^\square,E_m]\chi_I.
\]
移项即得所述公式；判据随之成立。证毕。
\end{proof}

\leanverified{paper\_pom\_visible\_walsh\_commutator\_defect}

\begin{corollary}[逐层精确继承与首谱隙饱和的充分条件]\label{cor:pom-visible-walsh-sectorwise-inheritance}
\leanverified{paper\_pom\_visible\_walsh\_sectorwise\_inheritance}
定义超立方体第 \(k\) Walsh 层
\[
\mathcal W_k:=\mathrm{span}\{\chi_I:\ |I|=k\},
\]
以及其折叠可见像
\[
\mathcal G_k:=\mathrm{span}\{g_I:\ |I|=k\}\subseteq L^2(X_m,\pi_m).
\]
若 \([P_m^\square,E_m]\) 在 \(\mathcal W_k\) 上恒为零，则
\[
\overline P_m^\square|_{\mathcal G_k}
=\left(1-\frac{2k}{m}\right)\mathrm{Id}.
\]
特别地，若在 \(k=1\) 时上述条件成立，且存在某个单点集 \(I\) 使 \(g_I\not\equiv0\)，则
\[
\mathrm{gap}(\overline P_m^\square)=\frac{2}{m}.
\]
\end{corollary}
\begin{proof}
第一式由定理 \ref{thm:pom-visible-walsh-commutator-defect} 对所有 \(|I|=k\) 的线性组合立即得到。
若 \(k=1\) 条件成立且某个 \(g_I\neq0\)，则 \(\lambda_I=1-2/m\) 确为 \(\overline P_m^\square\) 的非平凡本征值，
故
\(
\mathrm{gap}(\overline P_m^\square)\le 2/m
\)。
再与定理 \ref{thm:pom-fold-factor-chain-gap-lsi} 的下界
\(
\mathrm{gap}(\overline P_m^\square)\ge 2/m
\)
合并即得等式。证毕。
\end{proof}

\begin{proposition}[Gram 投影、传递矩阵与 trace/resolvent 的可见表示]\label{prop:pom-visible-walsh-gram-transfer}
令分析算子
\[
C_m:L^2(X_m,\pi_m)\to \ell^2(\mathcal P([m])),
\qquad
(C_m f)_I:=\langle f,g_I\rangle_{\pi_m},
\]
并定义
\[
\mathsf G_m:=C_mC_m^\ast,
\qquad
\mathsf M_m:=C_m\overline P_m^\square C_m^\ast,
\qquad
\Lambda_m:=\mathrm{diag}\!\left(1-\frac{2|I|}{m}\right)_{I\subseteq[m]}.
\]
则：
\begin{enumerate}
  \item \(\mathsf G_m\) 是 \(\ell^2(\mathcal P([m]))\) 上到
  \(
  \mathcal V_m:=C_m(L^2(X_m,\pi_m))
  \)
  的正交投影；特别地，
  \[
  \mathrm{rank}\,\mathsf G_m=|X_m|.
  \]
  \item 若记交换子缺陷矩阵
  \[
  (\mathsf \Delta_m)_{I,J}
  :=
  \langle E_m\chi_I,\,[P_m^\square,E_m]\chi_J\rangle_{\mu_m},
  \]
  则
  \[
  \mathsf M_m=\mathsf G_m\Lambda_m+\mathsf \Delta_m,
  \qquad
  \mathsf G_m\mathsf M_m=\mathsf M_m\mathsf G_m=\mathsf M_m.
  \]
  \item \(C_m\) 把 \(\overline P_m^\square\) 与 \(\mathsf M_m|_{\mathcal V_m}\) 酉等价。
  因而对任意整数 \(t\ge 1\) 及任意 \(|z|<1\)，
  \[
  \mathrm{Tr}\bigl((\overline P_m^\square)^t\bigr)=\mathrm{Tr}(\mathsf M_m^t),
  \]
  \[
  \bigl\langle f,(\mathrm{Id}-z\overline P_m^\square)^{-1}f\bigr\rangle_{\pi_m}
  =
  \bigl\langle C_m f,(\mathrm{Id}-z\mathsf M_m)^{-1}C_m f\bigr\rangle_{\ell^2},
  \]
  并且
  \[
  \bigl\langle f,(\mathrm{Id}-\overline P_m^\square)f\bigr\rangle_{\pi_m}
  =
  \bigl\langle C_m f,(\mathrm{Id}-\mathsf M_m)C_m f\bigr\rangle_{\ell^2}.
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
由定理 \ref{thm:pom-visible-walsh-parseval-frame}，\(C_m\) 为等距嵌入，故 \(C_mC_m^\ast\) 是到 \(\mathcal V_m\) 的正交投影，
且
\(
\mathrm{rank}\,\mathsf G_m=\dim L^2(X_m,\pi_m)=|X_m|
\)。
对矩阵元，
\[
(\mathsf M_m)_{I,J}
=\langle g_I,\overline P_m^\square g_J\rangle_{\pi_m}
=\left(1-\frac{2|J|}{m}\right)\langle g_I,g_J\rangle_{\pi_m}
+\langle E_m\chi_I,\,[P_m^\square,E_m]\chi_J\rangle_{\mu_m},
\]
其中最后一步由定理 \ref{thm:pom-visible-walsh-commutator-defect} 与 \(U_m g_I=E_m\chi_I\) 得到，
故 \(\mathsf M_m=\mathsf G_m\Lambda_m+\mathsf \Delta_m\)。
又由于 \(\mathsf M_m=C_m\overline P_m^\square C_m^\ast\) 的像与定义域都落在 \(\mathcal V_m\) 中，
遂有 \(\mathsf G_m\mathsf M_m=\mathsf M_m\mathsf G_m=\mathsf M_m\)。
最后，\(C_m:L^2(X_m,\pi_m)\to\mathcal V_m\) 为酉算子，且
\[
C_m\overline P_m^\square=\mathsf M_m C_m,
\]
故 \(\overline P_m^\square\) 与 \(\mathsf M_m|_{\mathcal V_m}\) 酉等价。
迹公式与 resolvent/Dirichlet 公式于是由函数演算立即推出；
由于 \(\mathsf M_m\) 在 \(\mathcal V_m^\perp\) 上恒为零，
\(\mathrm{Tr}(\mathsf M_m^t)\) 与 \(\mathrm{Tr}((\mathsf M_m|_{\mathcal V_m})^t)\) 相同。证毕。
\end{proof}
\leanverified{paper\_pom\_visible\_walsh\_gram\_transfer}

\begin{remark}[可见层级与谱继承需要分离讨论]\label{rem:pom-visible-walsh-gap-separation}
定义最小可见层级
\[
k_0(m):=\min\{\,|I|\ge 1:\ g_I\not\equiv 0\,\}.
\]
该量只记录 Walsh 模式在折叠后首次出现的层级，并不自动给出 \(\overline P_m^\square\) 的精确谱值。
由定理 \ref{thm:pom-visible-walsh-commutator-defect} 可见，
任何试图把
\(
1-\frac{2|I|}{m}
\)
直接识别为可见模态本征值，或把谱隙写成某个“最小可见层”的线性函数，均必须额外证明相应 Walsh 层上的交换子缺陷 \([P_m^\square,E_m]\) 消失。
在完全对易的极端情形 \(\mathsf \Delta_m=0\) 下，
上述可见表示才退化为逐层对角化，并恢复超立方体 Walsh 层的精确谱继承。
\end{remark}
