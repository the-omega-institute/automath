\begin{theorem}[有限极点情形的 Hankel 谱隙诊断]\label{thm:pom-hankel-finite-pole-spectral-gap}
\leanverified{paper\_pom\_hankel\_finite\_pole\_spectral\_gap}
设 \((a_m)_{m\ge 0}\subset\RR\) 满足分解
\[
a_m=\sum_{j=1}^{d}c_j x_j^{m}+e_m,
\qquad
x_1>x_2>\cdots>x_d>1,\quad
c_j>0,\quad
|e_m|\le C,
\]
其中 \(d\ge 1\) 与 \(C\ge 0\) 为常数。
对任意 \(N\ge d\) 定义 \(N\times N\) Hankel 矩阵
\[
H_N:=\bigl(a_{i+j}\bigr)_{0\le i,j\le N-1}.
\]
令 Vandermonde 矩阵 \(V\in\RR^{d\times d}\) 为
\[
V_{k+1,j}:=x_j^{k}\qquad(0\le k\le d-1,\ 1\le j\le d),
\]
并记其最小奇异值 \(\sigma_\ast:=\sigma_{\min}(V)>0\)。
再令
\[
c_\ast:=\min_{1\le j\le d}c_j,
\qquad
x_\ast:=\min_{1\le j\le d}x_j=x_d,
\qquad
\kappa:=c_\ast\,\sigma_\ast^2\,x_\ast^{-2d}.
\]
则对所有 \(N\ge d\) 有
\[
\sigma_{d+1}(H_N)\le CN,
\qquad
\sigma_{d}(H_N)\ge \kappa x_\ast^{2N}-CN.
\]
特别地，当 \(N\) 满足 \(\kappa x_\ast^{2N}\ge 2CN\) 时，\(H_N\) 存在可由有限数据核验的秩 \(d\) 谱隙：
\[
\sigma_d(H_N)\ge CN\ge \sigma_{d+1}(H_N).
\]
\end{theorem}
\begin{proof}
写
\[
H_N=S_N+E_N,\qquad
S_N:=\sum_{j=1}^{d}c_j\,u_j u_j^{\mathsf T},
\qquad
u_j:=(1,x_j,\dots,x_j^{N-1})^{\mathsf T},
\qquad
E_N:=(e_{i+j})_{0\le i,j\le N-1}.
\]
由 \(|e_{i+j}|\le C\) 得 \(\|E_N\|_{\mathrm{F}}\le CN\)，从而 \(\|E_N\|_{\mathrm{op}}\le CN\)。
因此 \(\sigma_{d+1}(H_N)\le \|E_N\|_{\mathrm{op}}\le CN\)。
\par
令 \(U\in\RR^{N\times d}\) 的第 \(j\) 列为 \(\sqrt{c_j}\,u_j\)，则 \(S_N=UU^{\mathsf T}\)，从而
\(
\sigma_d(S_N)=\sigma_{\min}(U)^2
\)。
取 \(U\) 的最后 \(d\) 行组成子矩阵 \(W_N\in\RR^{d\times d}\)，则
\[
W_N
=V\,\mathrm{diag}\!\bigl(\sqrt{c_1}\,x_1^{N-d},\dots,\sqrt{c_d}\,x_d^{N-d}\bigr),
\]
由列缩放对最小奇异值的下界得到
\[
\sigma_{\min}(W_N)\ \ge\ \sigma_{\min}(V)\cdot\min_{j}\bigl(\sqrt{c_j}\,x_j^{N-d}\bigr)
\ \ge\ \sigma_\ast\,\sqrt{c_\ast}\,x_\ast^{N-d}.
\]
又因为 \(U^{\mathsf T}U\succeq W_N^{\mathsf T}W_N\)，故 \(\sigma_{\min}(U)\ge \sigma_{\min}(W_N)\)，从而
\[
\sigma_d(S_N)=\sigma_{\min}(U)^2
\ \ge\
c_\ast\sigma_\ast^2 x_\ast^{2(N-d)}
=\kappa x_\ast^{2N}.
\]
最后由 Weyl 奇异值扰动不等式
\(
\sigma_d(H_N)\ge \sigma_d(S_N)-\|E_N\|_{\mathrm{op}}
\)
得到 \(\sigma_d(H_N)\ge \kappa x_\ast^{2N}-CN\)。证毕。
\end{proof}

\begin{corollary}[碰撞矩序列的 Perron 根可审计反演]\label{cor:pom-collision-perron-auditable-inversion}
\leanverified{paper\_pom\_collision\_perron\_auditable\_inversion}
固定 \(q\ge 2\)，设 \(m\mapsto S_q(m)\) 自某偏移 \(m_0\) 起为 C-finite，并且存在常数 \(d_q\ge 1\)、\(C_q\ge 0\) 与参数
\[
x_{q,1}>x_{q,2}>\cdots>x_{q,d_q}>1,
\qquad
c_{q,j}>0,
\]
使得对所有 \(m\ge m_0\) 有分解
\[
S_q(m)=\sum_{j=1}^{d_q}c_{q,j}\,x_{q,j}^{m}+e_{q,m},
\qquad
|e_{q,m}|\le C_q.
\]
对 \(N\ge d_q\) 定义 Hankel 块
\[
H_{N}^{(q)}:=\bigl(S_q(m_0+i+j)\bigr)_{0\le i,j\le N-1}.
\]
则当 \(N\) 足够大时，\(H_N^{(q)}\) 的奇异值出现可核验的秩 \(d_q\) 谱隙；并且在谱隙成立的区间内，标准的 Hankel pencil/Prony 反演机制可稳定恢复主尺度 \(x_{q,1}\)。
若进一步将 \(x_{q,1}\) 识别为该 C-finite 序列的 Perron 根 \(r_q\)（例如在非负最小 realization 口径下成立），则 \(r_q\) 可由有限 Hankel 块在离线环境中稳定读出，其误差由谱比 \(x_{q,2}/x_{q,1}\) 的指数衰减与余项界 \(C_q\) 共同控制。
\end{corollary}
\begin{proof}
将定理 \ref{thm:pom-hankel-finite-pole-spectral-gap} 应用于序列 \(a_m:=S_q(m_0+m)\) 即得谱隙存在性的断言。
在 \(H_N^{(q)}\) 存在秩 \(d_q\) 谱隙时，Hankel pencil/Prony 机制在条件数由谱隙控制的意义下稳定，并可恢复 \(\{x_{q,j}\}\)（特别是主尺度 \(x_{q,1}\)）；误差估计来自标准的特征值稳定性与截断误差传播界，主导项由 \(x_{q,2}/x_{q,1}\) 控制。证毕。
\end{proof}
