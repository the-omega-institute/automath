\begin{theorem}[二阶尺度的跃迁—碰撞同构与闭式恒等式]\label{thm:pom-hiddenbit-jump-collision-isomorphism}
设 \(m\ge 4\)。沿用引理 \ref{lem:pom-one-bit} 的单隐藏位分解
\[
N(\omega)=V_m(\Fold_m(\omega))+b(\omega)F_{m+2}\qquad(\omega\in\Omega_m),
\]
并沿用结论 \ref{con:pom-hidden-bit-mass-conservation-truncation} 的纤维分裂计数
\(
d_{m,b}(x)\ (b\in\{0,1\})
\) 与 \(d_m(x)=d_{m,0}(x)+d_{m,1}(x)\)。
令碰撞关系 \(R_m\subseteq\Omega_m^2\) 如命题 \ref{prop:pom-s2-recurrence}，
并定义“跨隐藏位跃迁碰撞集”
\[
J_m:=\{(\omega,\eta)\in\Omega_m\times\Omega_m:\ \Fold_m(\omega)=\Fold_m(\eta),\ b(\omega)=0,\ b(\eta)=1\}.
\]
则存在显式双射
\[
\Phi_m:\ R_{m-2}\ \xrightarrow{\ \cong\ }\ J_m,
\]
从而得到精确恒等式
\[
\boxed{\;
|J_m|
=\sum_{x\in X_m} d_{m,0}(x)\,d_{m,1}(x)
=|R_{m-2}|
=S_2(m-2).\;}
\]
\leanverified{crossWeightCorrelation\_eq\_momentSum\_two}
\leanverified{paper\_pom\_hiddenbit\_jump\_collision\_isomorphism}
\end{theorem}
\begin{proof}
为避免长度歧义，记
\[
N_{m-2}(\alpha):=\sum_{k=1}^{m-2}\alpha_kF_{k+1}\qquad(\alpha\in\Omega_{m-2}),
\qquad
N_m(\omega):=\sum_{k=1}^{m}\omega_kF_{k+1}\qquad(\omega\in\Omega_m).
\]
取任意 \((\alpha,\beta)\in R_{m-2}\)。由引理 \ref{lem:pom-fold-congruence}（对分辨率 \(m-2\)）可知
\(
N_{m-2}(\alpha)\equiv N_{m-2}(\beta)\ (\mathrm{mod}\ F_m)
\)。
另一方面，\(\max_{\gamma\in\Omega_{m-2}}N_{m-2}(\gamma)=F_{m+1}-2<2F_m\)（由 \(\max_{\omega\in\Omega_\ell}N_\ell(\omega)=F_{\ell+3}-2\) 与 \(F_{m+1}<2F_m\)），故
\[
\Delta:=N_{m-2}(\beta)-N_{m-2}(\alpha)\in\{0,\pm F_m\}.
\]
把长度 \(m-2\) 的二进制串视为前缀，并用并置表示拼接。定义 \(\Phi_m(\alpha,\beta)=(\omega,\eta)\) 如下：
\begin{itemize}
\item 若 \(\Delta=0\)，令 \(\omega:=\alpha 00\)，\(\eta:=\beta 11\)；
\item 若 \(\Delta=F_m\)，令 \(\omega:=\alpha 00\)，\(\eta:=\beta 01\)；
\item 若 \(\Delta=-F_m\)，令 \(\omega:=\beta 10\)，\(\eta:=\alpha 11\)。
\end{itemize}
三种情形均直接给出
\[
N_m(\eta)-N_m(\omega)=F_{m+1}+F_m=F_{m+2},
\]
从而 \(N_m(\eta)\equiv N_m(\omega)\ (\mathrm{mod}\ F_{m+2})\)，再由引理 \ref{lem:pom-fold-congruence}（对分辨率 \(m\)）得 \(\Fold_m(\eta)=\Fold_m(\omega)\)。
又因 \(N_m(\omega)<F_{m+2}\) 且 \(N_m(\eta)=N_m(\omega)+F_{m+2}\)，由单隐藏位分解的唯一性得到 \(b(\omega)=0\)、\(b(\eta)=1\)。
故 \(\Phi_m(\alpha,\beta)\in J_m\)。

反之，取任意 \((\omega,\eta)\in J_m\)。由定义有 \(\Fold_m(\omega)=\Fold_m(\eta)\) 且 \(b(\omega)=0,b(\eta)=1\)，故由引理 \ref{lem:pom-one-bit} 得
\[
N_m(\eta)=N_m(\omega)+F_{m+2}=N_m(\omega)+F_{m+1}+F_m.\tag{$\ast$}
\]
写 \(\omega=\alpha ab\)、\(\eta=\beta cd\)（\(\alpha,\beta\in\Omega_{m-2}\)，且 \(a,c\) 对应权重 \(F_m\)，\(b,d\) 对应权重 \(F_{m+1}\)）。
由 \(\max_{\gamma\in\Omega_{m-2}}N_{m-2}(\gamma)=F_{m+1}-2\) 得 \(|N_{m-2}(\beta)-N_{m-2}(\alpha)|<F_{m+1}\)，与 \((\ast)\) 比较迫使 \(d-b=1\)，即 \((b,d)=(0,1)\)。
将其代回 \((\ast)\) 并按 \(F_m,F_{m+1}\) 的系数分离，得到
\[
N_{m-2}(\beta)-N_{m-2}(\alpha) + (c-a)F_m = F_m,
\]
从而 \(\Delta:=N_{m-2}(\beta)-N_{m-2}(\alpha)\in\{0,\pm F_m\}\)。
并且三种取值分别对应于末两位模式
\[
(ab,cd)\in\{(00,11),(00,01),(10,11)\}.
\]
据此可唯一恢复 \((\alpha,\beta)\in R_{m-2}\) 并满足 \(\Phi_m(\alpha,\beta)=(\omega,\eta)\)。
因此 \(\Phi_m\) 为双射。

最后，按 \(x=\Fold_m(\omega)=\Fold_m(\eta)\) 分块计数可知
\(
|J_m|=\sum_{x\in X_m}d_{m,0}(x)d_{m,1}(x)
\)；
而由定义 \(|R_{m-2}|=S_2(m-2)\)（定义 \ref{def:pom-s2}）。证毕。
\end{proof}

\begin{corollary}[碰撞条件下的隐藏位跃迁概率闭式与代数极限]\label{cor:pom-hiddenbit-jump-probability-closed-form}
在碰撞关系 \(R_m\) 上取均匀分布（即从所有碰撞有序对 \((\omega,\omega')\in R_m\) 等概率抽取）。
则
\[
\boxed{\;
\mathbb{P}\bigl[b(\omega)\neq b(\omega')\ \big|\ (\omega,\omega')\in R_m\bigr]
=\frac{2|J_m|}{|R_m|}
=\frac{2S_2(m-2)}{S_2(m)}\;}
\]
并且令 \(r_2:=\rho(A_2)\) 为命题 \ref{prop:pom-s2-recurrence} 的三态碰撞核 \(A_2\) 的 Perron 根（等价地，为 \(\lambda^3-2\lambda^2-2\lambda+2=0\) 的最大实根），则极限存在且
\[
\lim_{m\to\infty}\mathbb{P}\bigl[b(\omega)\neq b(\omega')\ \big|\ (\omega,\omega')\in R_m\bigr]
=\frac{2}{r_2^2}.
\]
\end{corollary}
\leanverified{paper\_pom\_hiddenbit\_jump\_probability\_closed\_form}
\begin{proof}
由定理 \ref{thm:pom-hiddenbit-jump-collision-isomorphism}，\(|J_m|=S_2(m-2)\)，并且在碰撞有序对中事件 \(b(\omega)\neq b(\omega')\) 恰由 \((0,1)\) 与 \((1,0)\) 两个方向组成，故计数为 \(2|J_m|\)，从而得到闭式。
由命题 \ref{prop:pom-s2-recurrence} 的 Perron--Frobenius 渐近 \(S_2(m)\asymp r_2^m\)，得 \(S_2(m-2)/S_2(m)\to r_2^{-2}\)，代回即得极限。证毕。
\end{proof}

\begin{theorem}[隐藏位分裂偏差能量的二阶守恒律]\label{thm:pom-hiddenbit-bias-energy-identity}
定义隐藏位分裂偏差能量
\[
\mathcal{E}_m:=\sum_{x\in X_m}\bigl(d_{m,0}(x)-d_{m,1}(x)\bigr)^2.
\]
则对一切 \(m\ge 4\) 成立精确恒等式
\[
\boxed{\ \mathcal{E}_m=S_2(m)-4S_2(m-2)\ }.
\]
从而在碰撞加权测度
\[
\nu_m(x):=\frac{d_m(x)^2}{S_2(m)}\qquad(x\in X_m)
\]
下，令 \(p_m(x):=d_{m,1}(x)/d_m(x)\in[0,1]\)，则有
\[
\mathbb{E}_{x\sim\nu_m}\bigl[(1-2p_m(x))^2\bigr]
=\frac{\mathcal{E}_m}{S_2(m)}
=1-\frac{4S_2(m-2)}{S_2(m)}
\xrightarrow[m\to\infty]{}
1-\frac{4}{r_2^2}.
\]
\leanverified{paper\_pom\_hiddenbit\_bias\_energy\_identity}
\leanverified{momentSum\_two\_quadruple\_le}
\end{theorem}
\begin{proof}
由恒等式 \((a-b)^2=(a+b)^2-4ab\) 得
\[
\mathcal{E}_m=\sum_x d_m(x)^2-4\sum_x d_{m,0}(x)d_{m,1}(x)=S_2(m)-4|J_m|.
\]
再由定理 \ref{thm:pom-hiddenbit-jump-collision-isomorphism} 中 \(|J_m|=S_2(m-2)\) 即得所示等式。
第二部分由
\(
(d_{m,0}-d_{m,1})^2=d_m(x)^2(1-2p_m(x))^2
\)
两边对 \(x\) 求和并除以 \(S_2(m)\) 得出；极限由推论 \ref{cor:pom-hiddenbit-jump-probability-closed-form} 的比值收敛得到。证毕。
\end{proof}

\begin{proposition}[隐藏位分裂诱导的条件期望与 Watatani 指数元闭式]\label{prop:pom-hiddenbit-watatani-index-element}
令 \(\mathcal{R}_m=\Omega_m\times_{X_m}\Omega_m\) 为折叠等价关系群胚，并记其群胚代数
\[
A_{m,1}:=\CC[\mathcal{R}_m]\ \cong\ \bigoplus_{x\in X_m}M_{d_m(x)}(\CC)
\]
如命题 \ref{prop:fold-groupoid-wedderburn}。
对每个 \(x\in X_m\)，令 \(b\) 把矩阵块 \(M_{d_m(x)}(\CC)\) 分裂为两个对角子块
\(
M_{d_{m,0}(x)}(\CC)\oplus M_{d_{m,1}(x)}(\CC)
\)
（若 \(d_{m,1}(x)=0\) 则仅保留第一块），并令
\[
B_{m,1}^{(b)}:=\bigoplus_{x\in X_m}\Bigl(M_{d_{m,0}(x)}(\CC)\oplus M_{d_{m,1}(x)}(\CC)\Bigr)\ \subset\ A_{m,1}.
\]
令 \(E_{m,1}^{(b)}:A_{m,1}\to B_{m,1}^{(b)}\) 为逐块“消去非对角子块”的标准忠实条件期望。
则 \(E_{m,1}^{(b)}\) 具有有限 Watatani 指数，且其指数元位于 \(Z(B_{m,1}^{(b)})\) 并在每个 \(x\)-块上满足
\[
\mathrm{IndW}\!\left(E_{m,1}^{(b)}\right)\Big|_{x}
=
\frac{d_m(x)}{d_{m,0}(x)}\,I_{d_{m,0}(x)}
\ \oplus\
\frac{d_m(x)}{d_{m,1}(x)}\,I_{d_{m,1}(x)},
\]
其中约定当 \(d_{m,1}(x)=0\) 时第二直和项缺失。
\leanverified{paper\_pom\_hiddenbit\_watatani\_index\_element}
\end{proposition}
\begin{proof}
在每个 \(x\)-块上，\(A_x:=M_{n}(\CC)\)（\(n=d_m(x)\)）与
\(
B_x:=M_{a}(\CC)\oplus M_{b}(\CC)
\)
（\(a=d_{m,0}(x)\)，\(b=d_{m,1}(x)\)）为有限维半单 \(C^\ast\)-代数，且 \(E_x\) 为块对角投影。
Watatani 指数元在有限维情形可由准基的标准计算给出，并落在中心 \(Z(B_x)=\CC p_0\oplus\CC p_1\) 上；其在两块上的标量系数分别为 \(n/a\) 与 \(n/b\)。
将其写成对角块上的标量乘单位矩阵即得所示公式；对所有 \(x\) 直和合并得到结论。证毕。
\end{proof}

\begin{proposition}[补对偶不动点稳定型的隐藏位分裂二分律]\label{prop:pom-hiddenbit-complement-fixedpoint-bsplit-dichotomy}
设 \(F_{m+2}\) 为奇数（等价于 \(\mathrm{Fix}(\jmath_m)\neq\varnothing\)，见命题 \ref{prop:fold-complement-duality-fixedpoints-parity}），并令 \(x_\star\in X_m\) 为补对偶 \(\jmath_m\) 的唯一不动点。
记其余类坐标 \(r_\star:=V_m(x_\star)\in\{0,\dots,F_{m+2}-1\}\)，则 \(r_\star\) 满足
\[
2r_\star\equiv F_{m+1}-2\pmod{F_{m+2}}.
\]
在此情形下有如下二分律：
\begin{enumerate}
\normalfont
  \item 若 \(F_{m+1}\) 为偶数，则
  \[
  r_\star=\frac{F_{m+1}-2}{2}\in[0,F_{m+1}-2],
  \qquad
  d_{m,0}(x_\star)=d_{m,1}(x_\star).
  \]
  \item 若 \(F_{m+1}\) 为奇数，则
  \[
  r_\star=\frac{F_{m+3}-2}{2}>F_{m+1}-2,
  \qquad
  d_{m,1}(x_\star)=0.
  \]
\end{enumerate}
\leanverified{paper\_pom\_hiddenbit\_complement\_fixedpoint\_bsplit\_dichotomy}
\end{proposition}
\begin{proof}
由命题 \ref{prop:fold-complement-duality-fixedpoints-parity}(1)，不动点条件等价于所示线性同余；当 \(F_{m+2}\) 为奇数时解唯一。

若 \(F_{m+1}\) 偶，则 \(F_{m+1}-2\) 偶，因而整数解
\(
r_\star=(F_{m+1}-2)/2
\)
直接落在 \([0,F_{m+2})\) 并满足同余，故为唯一解。
又由补对偶的数值恒等式 \(N(C(\omega))=S_m-N(\omega)\)（见命题 \ref{prop:fold-complement-duality}）与 \(\tau_m(r_\star)=r_\star\)，补对偶在余类 \(r_\star\) 的纤维内把“低层”（\(N=r_\star\)）双射到“高层”（\(N=r_\star+F_{m+2}\)）；
按结论 \ref{con:pom-hidden-bit-mass-conservation-truncation} 的解释，这恰是 \(b\)-分裂两支之间的双射，故 \(d_{m,0}(x_\star)=d_{m,1}(x_\star)\)。

若 \(F_{m+1}\) 奇，则 \(F_{m+1}-2\) 奇，故同余解必须满足
\(
2r_\star=F_{m+1}-2+F_{m+2}
\)
（因为 \(0\le r_\star<F_{m+2}\) 且 \(F_{m+2}\) 奇），即
\(
r_\star=(F_{m+3}-2)/2
\)。
由
\(
r_\star-(F_{m+1}-2)=\frac{F_{m+2}-F_{m+1}+2}{2}=\frac{F_m+2}{2}>0
\)
得 \(r_\star>F_{m+1}-2\)。
而对任意余类 \(r\)，若 \(r>F_{m+1}-2\)，则 \(r+F_{m+2}>F_{m+3}-2=\max_{\omega\in\Omega_m}N(\omega)\)，从而不存在 \(\omega\) 使 \(N(\omega)=r+F_{m+2}\)，即 \(d_{m,1}(w(r))=0\)。
取 \(r=r_\star\) 即得 \(d_{m,1}(x_\star)=0\)。证毕。
\end{proof}

\begin{conjecture}[碰撞加权下跃迁概率的重整化极限律]\label{conj:pom-hiddenbit-jump-renormalization-limit-law}
令
\[
p_m(x):=\frac{d_{m,1}(x)}{d_m(x)}\in[0,1],
\qquad
\nu_m(x):=\frac{d_m(x)^2}{S_2(m)}\quad(x\in X_m).
\]
则存在唯一的 Borel 概率测度 \(\nu_\infty\) 于 \([0,1]\) 使得 \(p_m(X)\) 在 \(X\sim\nu_m\) 下的分布弱收敛到 \(\nu_\infty\)。
并且 \(\nu_\infty\) 的二次矩约束由三态碰撞核 \(A_2\) 的 Perron 根 \(r_2\) 锁定：
\[
\int_{0}^{1}2t(1-t)\,\dd\nu_\infty(t)=\frac{2}{r_2^2},
\qquad
\int_{0}^{1}(1-2t)^2\,\dd\nu_\infty(t)=1-\frac{4}{r_2^2}.
\]
\end{conjecture}
