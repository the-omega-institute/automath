\begin{theorem}[一般纤维扫描元的奇偶分解]\label{thm:pom-toggle-scan-sign-general-fiber}
\leanverified{paper\_pom\_toggle\_scan\_sign\_general\_fiber}
设 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)，从而 \(F_x\cong\prod_{j=1}^r\mathsf{Ind}(P_{\ell_j})\) 且 \(c_x=\prod_{j=1}^r c_{\ell_j}\)（定义 \ref{def:pom-toggle-coxeter-element}）。
定义
\[
e(\ell):=
\begin{cases}
1,& \ell\equiv 1,3\pmod 6,\\
0,& \text{其余},
\end{cases}
\qquad
A:=\{j:\ \ell_j\equiv 1\pmod 3\}.
\]
则 \(c_x\) 的置换符号满足：
\begin{enumerate}
  \item 若 \(|A|\ge 2\)，则 \(\operatorname{sgn}(c_x)=+1\)；
  \item 若 \(|A|=1\) 且 \(A=\{k\}\)，则 \(\operatorname{sgn}(c_x)=(-1)^{e(\ell_k)}\)；
  \item 若 \(|A|=0\)，则 \(\operatorname{sgn}(c_x)=(-1)^{\sum_{j=1}^r e(\ell_j)}\)。
\end{enumerate}
\end{theorem}

\begin{proof}
记 \(n_j:=|\mathsf{Ind}(P_{\ell_j})|=F_{\ell_j+2}\)。
对有限集 \(U,V\) 与 \(\sigma\in S_U\)，有 \(\operatorname{sgn}(\sigma\times\mathrm{id}_V)=\operatorname{sgn}(\sigma)^{|V|}\)，因 \(U\times V\) 分解为 \(|V|\) 个互不相交的 \(U\)-层。
由此得到
\[
\operatorname{sgn}(c_x)=\prod_{j=1}^r \operatorname{sgn}(c_{\ell_j})^{\prod_{i\ne j}n_i}.
\]
由定理 \ref{thm:pom-toggle-scan-sign-mod6}，\(\operatorname{sgn}(c_{\ell_j})=(-1)^{e(\ell_j)}\)。
而 \(n_i=F_{\ell_i+2}\) 为偶当且仅当 \(\ell_i\equiv 1\pmod 3\)，故 \(\prod_{i\ne j}n_i\) 为奇当且仅当除 \(j\) 外所有 \(i\) 不在 \(A\)。
据此分情形：\(|A|\ge 2\) 时所有指数为偶；\(|A|=1\) 时仅 \(j=k\) 给出奇指数；\(|A|=0\) 时所有指数为奇，得到所示三式。证毕。
\end{proof}

\begin{theorem}[扫描元符号的边数奇偶刻画]\label{thm:pom-toggle-scan-sign-edge-parity}
\leanverified{paper\_pom\_toggle\_scan\_sign\_edge\_parity}
设 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)，并令 \(\mathcal F_x\) 为定义 \ref{def:pom-fiber-reconstruction-graph} 的纤维重构图。
则成立严格恒等式
\[
\operatorname{sgn}(c_x)=(-1)^{|E(\mathcal F_x)|}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-reconstruction-cartesian-product}，
\(
\mathcal F_x\cong\prod_{j=1}^r \Gamma_{\ell_j}
\)。
对有限图的 Cartesian 乘积有一般恒等式
\[
|E(\textstyle\prod_{j=1}^r G_j)|
=\sum_{j=1}^r |E(G_j)|\prod_{i\ne j}|V(G_i)|.
\]
取 \(G_j=\Gamma_{\ell_j}\) 得
\[
|E(\mathcal F_x)|
=\sum_{j=1}^r |E(\Gamma_{\ell_j})|\prod_{i\ne j}|V(\Gamma_{\ell_i})|.
\]
另一方面，由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 的证明，
\[
\operatorname{sgn}(c_x)=\prod_{j=1}^r \operatorname{sgn}(c_{\ell_j})^{\prod_{i\ne j}|V(\Gamma_{\ell_i})|}.
\]
而对单分量，定理 \ref{thm:pom-toggle-scan-sign-mod6} 给出 \(\operatorname{sgn}(c_\ell)=(-1)^{e(\ell)}\)，并且推论 \ref{cor:pom-toggle-fibcube-edge-parity} 给出
\(
|E(\Gamma_\ell)|\equiv e(\ell)\pmod 2
\)；
故
\(
\operatorname{sgn}(c_\ell)=(-1)^{|E(\Gamma_\ell)|}
\)。
将其代入上式并合并指数，得到
\[
\operatorname{sgn}(c_x)
=\prod_{j=1}^r (-1)^{|E(\Gamma_{\ell_j})|\prod_{i\ne j}|V(\Gamma_{\ell_i})|}
=(-1)^{|E(\mathcal F_x)|},
\]
即得结论。证毕。
\end{proof}

\begin{corollary}[可缩纤维中的偶性刚性]\label{cor:pom-toggle-contractible-parity-rigidity}
\leanverified{paper\_pom\_toggle\_contractible\_parity\_rigidity}
设 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)，并沿定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 记 \(A:=\{j:\ell_j\equiv 1\pmod 3\}\)。

\begin{enumerate}
  \item 若 \(|A|\ge 2\)，则 \(\operatorname{sgn}(c_x)=+1\)；
  \item 若 \(|A|=1\) 且 \(A=\{k\}\)，则
  \[
  \operatorname{sgn}(c_x)=-1
  \quad\Longleftrightarrow\quad
  \ell_k\equiv 1\pmod 6.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}

若 \(|A|=1\) 且 \(A=\{k\}\)，则 \(\operatorname{sgn}(c_x)=(-1)^{e(\ell_k)}\)。
在约束 \(\ell_k\equiv 1\pmod 3\) 下，\(e(\ell_k)=1\) 当且仅当 \(\ell_k\equiv 1\pmod 6\)，从而得到所示等价。证毕。
\end{proof}

\begin{theorem}[时间反演的不动点乘积公式与符号闭式]\label{thm:pom-toggle-time-reversal-fixedpoints-sign-closed-form}
设 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)。
记
\[
N_x:=|F_x|=|V(\mathcal F_x)|=\prod_{j=1}^r |\mathsf{Ind}(P_{\ell_j})|.
\]
令
\[
\mathrm{Fix}(\iota_x):=\{\omega\in \Fold_m^{-1}(x):\ \iota_x(\omega)=\omega\}.
\]
则：
\begin{enumerate}
  \item 存在严格乘积公式
  \[
  |\mathrm{Fix}(\iota_x)|=\prod_{j=1}^r f(\ell_j),
  \qquad
  f(2k)=F_{k+1},\quad f(2k+1)=F_{k+3}.
  \]
  \item 有闭式
  \[
  \operatorname{sgn}(\iota_x)=(-1)^{\frac{N_x-|\mathrm{Fix}(\iota_x)|}{2}}.
  \]
\end{enumerate}
\leanverified{paper\_pom\_toggle\_time\_reversal\_fixedpoints\_sign\_closed\_form}
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-indset-factorization} 的分量分解与定理 \ref{thm:pom-toggle-time-reversal-sign-general-fiber} 中的定义，
\(
\iota_x=\prod_{j=1}^r \iota_{\ell_j}
\)
且各 \(\iota_{\ell_j}\) 仅作用于第 \(j\) 个分量并彼此交换。
因此 \(\omega=(\omega_1,\dots,\omega_r)\) 为 \(\iota_x\) 的不动点当且仅当每个 \(\omega_j\) 为 \(\iota_{\ell_j}\) 的不动点，从而
\[
|\mathrm{Fix}(\iota_x)|=\prod_{j=1}^r |\mathrm{Fix}(\iota_{\ell_j})|.
\]
而定理 \ref{thm:pom-toggle-time-reversal-sign-mod12} 的证明给出单路径情形的不动点计数：
若 \(\ell=2k\) 则 \(|\mathrm{Fix}(\iota_\ell)|=F_{k+1}\)，若 \(\ell=2k+1\) 则 \(|\mathrm{Fix}(\iota_\ell)|=F_{k+3}\)。
代入即得乘积公式。
\par
对符号闭式：\(\iota_x\) 为对合，其循环分解仅含不动点与互不相交的换位；设换位数为 \(t\)，则 \(\operatorname{sgn}(\iota_x)=(-1)^t\)。
同时被移动点数为 \(2t=N_x-|\mathrm{Fix}(\iota_x)|\)，故 \(t=(N_x-|\mathrm{Fix}(\iota_x)|)/2\)，从而得到所示闭式。证毕。
\end{proof}

\begin{corollary}[时间反演对合的轨道数闭式]\label{cor:pom-time-reversal-orbit-count}
\leanverified{paper\_pom\_time\_reversal\_orbit\_count}
在定理 \ref{thm:pom-toggle-time-reversal-fixedpoints-sign-closed-form} 的设定下，令 \(\langle \iota_x\rangle\) 作用于有限集合 \(F_x\)。
则其轨道数满足
\[
\card{F_x/\langle \iota_x\rangle}
=\frac{\card{F_x}+\card{\mathrm{Fix}(\iota_x)}}{2}
=\frac{d_m(x)+\card{\mathrm{Fix}(\iota_x)}}{2}.
\]
\end{corollary}

\begin{proof}
对合的轨道分解仅由不动点与 \(2\)-点轨道组成；设 \(2\)-点轨道数为 \(t_x\)，则
\(
\card{F_x}=\card{\mathrm{Fix}(\iota_x)}+2t_x
\)，而轨道总数为 \(\card{\mathrm{Fix}(\iota_x)}+t_x\)。
消去 \(t_x\) 得所示公式。证毕。
\end{proof}

\begin{proposition}[对称独立集多项式的闭式与分量乘法分解]\label{prop:pom-time-reversal-symmetric-indset-polynomial}
对路径 \(P_\ell\) 定义独立集多项式与对称独立集多项式
\[
I_\ell(t):=\sum_{S\in \mathsf{Ind}(P_\ell)}t^{|S|},
\qquad
J_\ell(t):=\sum_{S\in \mathrm{Fix}(\iota_\ell)}t^{|S|}.
\]
则对任意 \(n\ge 1\) 有
\[
J_{2n}(t)=I_{n-1}(t^2),
\qquad
J_{2n+1}(t)=I_{n}(t^2)+t\,I_{n-1}(t^2).
\]
并且对任意 \(x\in X_m\)，若 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)，则
\[
J_x(t):=\sum_{\omega\in \mathrm{Fix}(\iota_x)} t^{R_x(\omega)}
=\prod_{j=1}^r J_{\ell_j}(t),
\]
其中 \(R_x(\omega)\) 为定理 \ref{thm:pom-fiber-indset-factorization} 的独立集坐标下的选取数（亦即逆代换次数；定义 \ref{def:pom-fiber-rewrite-rv}）。
\end{proposition}

\begin{proof}
\textbf{偶数路径。}
令 \(\ell=2n\) 并记反演 \(r(i)=2n+1-i\)。
若 \(S\in\mathrm{Fix}(\iota_{2n})\)，则 \(i\in S\Rightarrow r(i)\in S\)。
由于 \(\{n,n+1\}\) 为边，故 \(S\cap\{n,n+1\}=\varnothing\)。
令 \(T:=S\cap\{1,\dots,n-1\}\)，则 \(T\in\mathsf{Ind}(P_{n-1})\)，并且 \(S\) 唯一由 \(T\) 给出为 \(S=T\sqcup r(T)\)。
因此 \(|S|=2|T|\)，得到 \(J_{2n}(t)=\sum_{T\in\mathsf{Ind}(P_{n-1})}(t^2)^{|T|}=I_{n-1}(t^2)\)。
\par
\textbf{奇数路径。}
令 \(\ell=2n+1\)，中点为 \(n+1\)，反演为 \(r(i)=2n+2-i\)。
对 \(S\in\mathrm{Fix}(\iota_{2n+1})\) 分两类：
若 \(n+1\notin S\)，则 \(S\) 由 \(T:=S\cap\{1,\dots,n\}\in\mathsf{Ind}(P_n)\) 唯一决定，且 \(|S|=2|T|\)，贡献为 \(I_n(t^2)\)；
若 \(n+1\in S\)，则邻点 \(n,n+2\) 不可取，从而 \(T:=S\cap\{1,\dots,n-1\}\in\mathsf{Ind}(P_{n-1})\) 唯一决定 \(S\)，且 \(|S|=2|T|+1\)，贡献为 \(tI_{n-1}(t^2)\)。
相加得到公式。
\par
\textbf{一般纤维。}
由分量分解 \(F_x\cong \prod_{j=1}^r \mathsf{Ind}(P_{\ell_j})\)，且 \(\iota_x=\prod_j \iota_{\ell_j}\) 逐分量作用，故不动点集合分解为直积；
同时 \(R_x\) 为各分量选取数之和，故权计数多项式严格乘法分解。证毕。
\end{proof}
\leanverified{paper\_pom\_time\_reversal\_symmetric\_indset\_polynomial}

\begin{proposition}[时间反演定点占比的乘法律与指数稀疏]\label{prop:pom-time-reversal-fixedpoint-ratio-decay}
对路径分量定义定点占比
\[
p(\ell):=\frac{\card{\mathrm{Fix}(\iota_\ell)}}{\card{\mathsf{Ind}(P_\ell)}}.
\]
则对任意 \(x\in X_m\) 若 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)，有严格乘法分解
\[
\frac{\card{\mathrm{Fix}(\iota_x)}}{d_m(x)}=\prod_{j=1}^r p(\ell_j).
\]
此外存在常数 \(C>0\) 使对所有 \(\ell\ge 1\) 有
\[
p(\ell)\le C\,\varphi^{-\ell/2},
\]
其中 \(\varphi=(1+\sqrt5)/2\)。
更精确地，对偶数 \(\ell=2n\) 有
\[
p(2n)=\frac{F_{n+1}}{F_{2n+2}}=\frac{1}{L_{n+1}},
\]
其中 Lucas 数列 \((L_k)\) 满足 \(L_0=2,L_1=1\) 且 \(L_{k+1}=L_k+L_{k-1}\)。
\end{proposition}

\begin{proof}
乘法分解由定理 \ref{thm:pom-toggle-time-reversal-fixedpoints-sign-closed-form} 的不动点乘积公式与 \(d_m(x)=\prod_j F_{\ell_j+2}\)（定理 \ref{thm:pom-fiber-indset-factorization}）直接推出。
\par
对偶数情形，定理 \ref{thm:pom-toggle-time-reversal-fixedpoints-sign-closed-form} 给出
\(
\card{\mathrm{Fix}(\iota_{2n})}=F_{n+1}
\)，且 \(\card{\mathsf{Ind}(P_{2n})}=F_{2n+2}\)。
利用恒等式 \(F_{2k}=F_k L_k\)（由 Binet 公式或标准递推恒等式推出），得 \(F_{2n+2}=F_{n+1}L_{n+1}\)，从而 \(p(2n)=1/L_{n+1}\)。
由于 \(L_{n}\asymp \varphi^{n}\)，得到 \(p(2n)\asymp \varphi^{-n}=\varphi^{-\ell/2}\)。
奇数情形由 \(\card{\mathrm{Fix}(\iota_{2n+1})}=F_{n+3}\) 与 \(F_{2n+3}\asymp \varphi^{2n+3}\) 同样给出指数率上界，故存在统一常数 \(C\)。证毕。
\end{proof}
\leanverified{paper\_pom\_time\_reversal\_fixedpoint\_ratio\_decay}

\begin{proposition}[时间反演定点数的两类极端情形]\label{prop:pom-time-reversal-fixedpoints-extremes}
设 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)。
则：
\begin{enumerate}
\normalfont
  \item \(\card{\mathrm{Fix}(\iota_x)}=d_m(x)\) 当且仅当 \(\ell_j=1\) 对所有 \(j\) 成立；此时 \(d_m(x)=2^{r}\) 且 \(\iota_x\) 为恒等置换。
  \item \(\card{\mathrm{Fix}(\iota_x)}=1\) 当且仅当 \(\ell_j=2\) 对所有 \(j\) 成立（或 \(r=0\) 的平凡情形）；此时 \(d_m(x)=3^{r}\) 且纤维中唯一时间反演对称点对应空独立集。
\end{enumerate}
\end{proposition}

\begin{proof}
由定理 \ref{thm:pom-toggle-time-reversal-fixedpoints-sign-closed-form}，
\(
\card{\mathrm{Fix}(\iota_x)}=\prod_j f(\ell_j)
\)
且
\(
d_m(x)=\prod_j F_{\ell_j+2}
\)。
对单路径可直接核对：当 \(\ell=1\) 时 \(f(1)=2=F_3\)；当 \(\ell=2\) 时 \(f(2)=1\) 且 \(F_4=3\)；并且当 \(\ell\ge 2\) 时恒有 \(1\le f(\ell)<F_{\ell+2}\)（例如由 \(p(\ell)<1\) 的严格性，见命题 \ref{prop:pom-time-reversal-fixedpoint-ratio-decay}）。
因此乘积等号 \(\prod_j f(\ell_j)=\prod_j F_{\ell_j+2}\) 当且仅当每个分量均取到唯一的等号情形 \(\ell_j=1\)；
而乘积为 \(1\) 当且仅当每个分量均满足 \(f(\ell_j)=1\)，即 \(\ell_j=2\)（或无分量）。
对应的 \(d_m(x)\) 由 \(F_3=2\) 与 \(F_4=3\) 立得。证毕。
\end{proof}
\leanverified{paper\_pom\_time\_reversal\_fixedpoints\_extremes}

\begin{theorem}[Pisano-\(1\) 分量计数的指数型大偏差与线性增殖]\label{thm:pom-pisano1-components-ldp-linear}
在 \(X_m\) 上取均匀分布 \(U_m=\mathrm{Unif}(X_m)\)。
则存在常数 \(\mu>0\) 与 \(c>0\) 使得对任意固定 \(\varepsilon\in(0,\mu)\)，当 \(m\) 充分大时有指数型偏差界
\[
\PP_{x\sim U_m}\!\bigl[N_m(x)\le \varepsilon m\bigr]\ \le\ e^{-c m}.
\]
\end{theorem}
\begin{proof}
由折叠核的有限状态在线归一化实现（有限延迟的逐位归一化与逆归一化同步），可在读取 \(x\in X_m\) 的同时以有限状态扩张跟踪 \(\Gamma_x\) 的分量边界与当前分量长度在 \(\bmod 3\) 下的余类。
因此存在有限状态有向图 \(\mathcal G\) 与边可观测 \(\phi:E(\mathcal G)\to\{0,1\}\)，使得对每个 \(x\in X_m\)，其在 \(\mathcal G\) 中诱导的状态路径 \(\gamma(x)\) 满足
\[
N_m(x)=\sum_{t=1}^{m} \phi\bigl(\gamma_t(x)\bigr)+O(1),
\]
其中 \(O(1)\) 仅来自端点初始化与截断的有限步修正。
\par
对 \(\theta\in\RR\) 令 \(B_\theta\) 为 \(\mathcal G\) 的加权邻接矩阵：若存在边 \(e:u\to v\)，则令 \((B_\theta)_{uv}\) 的对应贡献为 \(e^{\theta\phi(e)}\)，并按多重边求和。
由 Perron--Frobenius 理论，谱半径 \(\rho(B_\theta)\) 为正且随 \(\theta\) 实解析，并给出对数矩母函数
\[
\Lambda(\theta):=\lim_{m\to\infty}\frac{1}{m}\log \EE_{U_m}\!\bigl[e^{\theta N_m(X)}\bigr]
=\log\rho(B_\theta)-\log\rho(B_0).
\]
由于 \(\phi\) 非恒零（存在 \(x\) 使得 \(\Gamma_x\) 含 \(\ell\equiv 1\pmod 3\) 的分量，从而某些边上 \(\phi=1\)），并且 \(\mathcal G\) 的强连通性保证该事件在指数尺度上具有非零频率，故 \(\mu:=\Lambda'(0)>0\)。
由 \(\Lambda\) 的凸性与 Chernoff 界，对任意 \(\varepsilon\in(0,\mu)\) 可取某个 \(\theta<0\) 使得
\[
\PP_{U_m}\!\bigl[N_m\le \varepsilon m\bigr]\le
\exp\!\bigl(m(\Lambda(\theta)-\theta\varepsilon)+O(1)\bigr)
\le e^{-cm}
\]
对充分大 \(m\) 成立，其中 \(c=c(\varepsilon)>0\)。
证毕。
\end{proof}

\begin{theorem}[奇置换扇区的指数稀薄性]\label{thm:pom-toggle-odd-sector-exponential-thinness}
对每个 \(m\ge 1\) 在 \(X_m\) 上取均匀分布。
则存在常数 \(\kappa>0\) 使得当 \(m\) 充分大时有
\[
\PP\bigl(\operatorname{sgn}(c_x)=-1\bigr)\ \le\ e^{-\kappa m},
\qquad
\PP\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)\ \le\ e^{-\kappa m}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 与 \ref{thm:pom-toggle-time-reversal-sign-general-fiber}，若 \(\operatorname{sgn}(c_x)=-1\) 或 \(\operatorname{sgn}(\iota_x)=-1\)，则必有 \(|A|\le 1\)，亦即 \(N_m(x)\le 1\)。
取定理 \ref{thm:pom-pisano1-components-ldp-linear} 的任意 \(\varepsilon\in(0,\mu)\)，当 \(m\) 充分大时 \(1\le \varepsilon m\)，从而
\[
\PP\bigl(\operatorname{sgn}(c_x)=-1\bigr)\le \PP\bigl(N_m(x)\le 1\bigr)\le \PP\bigl(N_m(x)\le \varepsilon m\bigr)\le e^{-cm},
\]
对 \(\operatorname{sgn}(\iota_x)\) 同理。令 \(\kappa:=c\) 即得。
证毕。
\end{proof}

\begin{corollary}[纤维多重度的线性 \texorpdfstring{$2$}{2}-adic 可分性]\label{cor:pom-fiber-multiplicity-linear-2adic-divisibility}
令 \(d_m(x)=|\Fold_m^{-1}(x)|\) 为纤维多重度。
则存在常数 \(\alpha>0\) 与 \(c'>0\) 使得对充分大 \(m\) 有
\[
\PP_{x\sim\mathrm{Unif}(X_m)}\!\bigl[v_2\bigl(d_m(x)\bigr)\le \alpha m\bigr]\ \le\ e^{-c' m}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-fiber-indset-factorization} 的分量乘积表达式 \(d_m(x)=\prod_{j=1}^r F_{\ell_j(x)+2}\)，并用 Fibonacci 数列模 \(2\) 的 Pisano 周期为 \(3\)，知
\(
F_{\ell+2}\equiv 0\pmod 2
\iff
\ell\equiv 1\pmod 3
\)。
因此每个满足 \(\ell_j(x)\equiv 1\pmod 3\) 的分量至少贡献一个 \(2\)-因子，得到
\[
v_2\bigl(d_m(x)\bigr)\ \ge\ N_m(x).
\]
取定理 \ref{thm:pom-pisano1-components-ldp-linear} 的任意 \(\alpha\in(0,\mu)\)，则
\(
\{v_2(d_m(x))\le \alpha m\}\subseteq \{N_m(x)\le \alpha m\}
\)；
应用该定理即可得到所述指数界。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$2$}{2}-进估值的分量分解与整除门槛]\label{prop:pom-fiber-multiplicity-v2-thresholds}
设 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\)。
则纤维多重度满足乘积闭式
\[
d_m(x)=\prod_{j=1}^r F_{\ell_j+2},
\]
从而其 \(2\)-进估值满足严格恒等式
\[
v_2\bigl(d_m(x)\bigr)=\sum_{j=1}^r v_2\!\bigl(F_{\ell_j+2}\bigr).
\]
并且由 Fibonacci 数列的 \(2\)-进估值分类（例如命题 \ref{prop:fold-fiber-convolution-invertibility-divisibility} 中给出的分情形公式）可得以下门槛判别：
\begin{enumerate}
\normalfont
  \item \(d_m(x)\) 为奇数当且仅当对所有 \(j\) 都有 \(\ell_j\not\equiv 1\pmod 3\)。
  \item \(4\mid d_m(x)\) 当且仅当 \(\sum_{j=1}^r v_2(F_{\ell_j+2})\ge 2\)。
  等价地：存在某个 \(\ell_j\equiv 4\pmod 6\)（此时 \(v_2(F_{\ell_j+2})\ge 3\)），或存在至少两个 \(\ell_j\equiv 1\pmod 6\)（各贡献 \(v_2=1\)）。
  \item 更一般地，对任意 \(k\ge 1\)，有
  \[
  2^k\mid d_m(x)\qquad\Longleftrightarrow\qquad \sum_{j=1}^r v_2(F_{\ell_j+2})\ge k.
  \]
\end{enumerate}
\leanverified{paper\_pom\_fiber\_multiplicity\_v2\_thresholds}
\end{proposition}
\begin{proof}
乘积公式与 \(\mathsf{Ind}(P_\ell)\) 的基数 \(|\mathsf{Ind}(P_\ell)|=F_{\ell+2}\) 及分量独立性同义（定理 \ref{thm:pom-fiber-indset-factorization}）。
取 \(v_2\) 得到和式。
第一条由 \(F_n\) 奇偶三周期 \(F_n\equiv 0\ (\mathrm{mod}\ 2)\iff 3\mid n\) 推出。
其余各条由 \(v_2\) 的定义、和式与 Fibonacci 的 \(2\)-进估值分情形公式直接推出。
证毕。
\end{proof}

\begin{proposition}[奇纤维上的 Witten 指数--扫描符号耦合律]\label{prop:pom-witten-index-scan-sign-coupling-odd-fiber}
沿定理 \ref{thm:pom-fiber-rewrite-parity-exact-split-mod3-obstruction} 记
\(
s_m(x):=Z_x(-1)
\)。
定义奇纤维扇区
\[
X_m^{\mathrm{odd}}:=\{x\in X_m:\ d_m(x)\ \text{为奇}\},
\]
并对 \(x\in X_m^{\mathrm{odd}}\) 的分量分解 \(\Gamma_x\cong\bigsqcup_{j=1}^r P_{\ell_j}\) 定义
\[
N_2(x):=\#\{j:\ \ell_j\equiv 2\pmod 6\}.
\]
则在 \(X_m^{\mathrm{odd}}\) 上成立严格恒等式
\[
s_m(x)=\operatorname{sgn}(c_x)\cdot (-1)^{N_2(x)}.
\]
\end{proposition}
\begin{proof}
由 \(Z_x(t)\) 在分量分解下的乘法性（定理 \ref{thm:pom-fiber-indset-factorization}），有
\(
s_m(x)=\prod_{j=1}^r I_{\ell_j}(-1)
\)，其中 \(I_\ell(t)=\sum_{S\in\mathsf{Ind}(P_\ell)}t^{|S|}\)。
而 \(I_\ell(-1)\) 具有 \(6\) 周期取值规律：当 \(\ell\equiv 2,3\pmod 6\) 时取 \(-1\)，当 \(\ell\equiv 0,5\pmod 6\) 时取 \(+1\)，当 \(\ell\equiv 1,4\pmod 6\) 时取 \(0\)。
在 \(x\in X_m^{\mathrm{odd}}\) 上排除 \(\ell\equiv 1\ (\mathrm{mod}\ 3)\) 分量，故 \(I_{\ell_j}(-1)\in\{\pm 1\}\) 且
\[
s_m(x)=(-1)^{N_2(x)+N_3(x)},
\qquad
N_3(x):=\#\{j:\ \ell_j\equiv 3\pmod 6\}.
\]
另一方面，在 \(X_m^{\mathrm{odd}}\) 上有 \(A=\varnothing\)，故由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 的第三种情形得到
\[
\operatorname{sgn}(c_x)=(-1)^{N_3(x)}.
\]
相乘即得 \(s_m(x)=\operatorname{sgn}(c_x)\,(-1)^{N_2(x)}\)。证毕。
\end{proof}

\begin{proposition}[分量并结构下的二值函子性]\label{prop:pom-binary-invariants-functoriality}
\leanverified{paper\_pom\_binary\_invariants\_functoriality}

\begin{theorem}[最大纤维族的取向指纹刚性与显式周期分类]\label{thm:pom-maxfiber-orientation-charge-periodic}
令最大纤维达到者集合
\[
\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=D_m\}.
\]
在定理 \ref{thm:pom-max-achievers-phase-stabilization} 的稳定阈值区间内，存在唯一的 \(\chi_m\in\{\pm1\}^2\) 使得
\[
\chi(x)=\chi_m\qquad(\forall\,x\in \mathcal{M}_m),
\]
其中 \(\chi(x)=(\operatorname{sgn}(c_x),\operatorname{sgn}(\iota_x))\) 如命题 \ref{prop:pom-orientation-fingerprint-geometry-audit}。
并且 \(\chi_m\) 具有如下周期律：
\begin{enumerate}
\normalfont
  \item 若 \(m=2k\)，则 \(\chi_m\) 仅依赖于 \(k\bmod 12\)，并满足
  \[
  \chi_m=
  \begin{cases}
  (+,+), & k\equiv 0,5,10,11\pmod{12},\\
  (-,+), & k\equiv 1,9\pmod{12},\\
  (+,-), & k\equiv 2,4,6,8\pmod{12},\\
  (-,-), & k\equiv 3,7\pmod{12}.
  \end{cases}
  \]
  \item 若 \(m=2k+1\)，则第二分量恒为 \(+1\)，并且
  \[
  \chi_m=
  \begin{cases}
  (+,+), & m\equiv 5\pmod{6}\ \ (\Leftrightarrow\ k\equiv 2\pmod3),\\
  (-,+), & m\equiv 1,3\pmod{6}\ \ (\Leftrightarrow\ k\equiv 0,1\pmod3).
  \end{cases}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由注 \ref{rem:pom-max-achievers-graph-structure}，在稳定阈值区间内若 \(m=2k\) 则
\(\mathcal{M}_m\) 中的任意 \(x\) 满足 \(\Gamma_x\cong P_k\)，从而 \(\chi(x)=(\operatorname{sgn}(c_k),\operatorname{sgn}(\iota_k))\)；
若 \(m=2k+1\) 则 \(\Gamma_x\cong P_{k-1}\sqcup P_1\)。
因此 \(\chi\) 在 \(\mathcal{M}_m\) 上只依赖于该同构型，必为常值，从而存在唯一 \(\chi_m\)。
\par
对偶数 \(m=2k\)，由定理 \ref{thm:pom-toggle-scan-sign-mod6} 与定理 \ref{thm:pom-toggle-time-reversal-sign-mod12} 得
\(\operatorname{sgn}(c_k)\) 仅依赖于 \(k\bmod 6\)，而 \(\operatorname{sgn}(\iota_k)\) 仅依赖于 \(k\bmod 12\)；
将两者合并即可得到 \(k\bmod 12\) 的显式枚举表。
\par
对奇数 \(m=2k+1\)，写分量长度为 \((\ell_1,\ell_2)=(k-1,1)\)。
若 \(k\equiv 2\pmod3\)，则 \(\ell_1\equiv 1\pmod3\) 且 \(\ell_2\equiv 1\pmod3\)，于是 \(A\) 含两元；由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 与 \ref{thm:pom-toggle-time-reversal-sign-general-fiber} 的第一种情形可得
\(\operatorname{sgn}(c_x)=\operatorname{sgn}(\iota_x)=+1\)。
若 \(k\not\equiv 2\pmod3\)，则仅有 \(\ell_2=1\) 落入 \(\ell\equiv 1\pmod3\) 同余类，从而 \(|A|=1\)；
由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 得 \(\operatorname{sgn}(c_x)=(-1)^{e(1)}=-1\)，并由定理 \ref{thm:pom-toggle-time-reversal-sign-general-fiber} 得 \(\operatorname{sgn}(\iota_x)=\delta(1)=+1\)，从而得到所示分类。证毕。
\end{proof}

\begin{corollary}[最大纤维族上的带符号指数灭绝律与 \(6\) 周期表]\label{cor:pom-maxfiber-signed-index-extinction-periodic}
沿用定理 \ref{thm:pom-maxfiber-orientation-charge-periodic} 的记号，并令 \(\mathfrak{s}_m(x):=Z_x(-1)\)（定义 \ref{def:pom-fiber-signed-index}）。
则在稳定阈值区间内，对任意 \(x\in\mathcal{M}_m\) 有：
\begin{enumerate}
\normalfont
  \item 若 \(m\) 为奇数，则 \(\mathfrak{s}_m(x)=0\)。
  \item 若 \(m=2k\) 为偶数，则 \(\mathfrak{s}_m(x)=J_k\)，其中 \(J_\ell:=I_\ell(-1)\) 为定理 \ref{thm:pom-fiber-signed-index-periodicity} 的 \(6\) 周期序列；等价地
  \[
  \mathfrak{s}_m(x)=
  \begin{cases}
  +1, & k\equiv 0,5\pmod 6,\\
  -1, & k\equiv 2,3\pmod 6,\\
  0, & k\equiv 1,4\pmod 6.
  \end{cases}
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由注 \ref{rem:pom-max-achievers-graph-structure}：若 \(m=2k\)，则 \(\Gamma_x\cong P_k\)，故 \(Z_x(t)=I_k(t)\) 且 \(\mathfrak{s}_m(x)=I_k(-1)=J_k\)；
若 \(m=2k+1\)，则 \(\Gamma_x\cong P_{k-1}\sqcup P_1\)，从而
\(
\mathfrak{s}_m(x)=J_{k-1}\cdot J_1
\)；
而 \(J_1=0\)（定理 \ref{thm:pom-fiber-signed-index-periodicity}），故 \(\mathfrak{s}_m(x)=0\)。
最后一条分情形表即为 \(6\) 周期序列 \(J_k\) 的取值表。证毕。
\end{proof}
