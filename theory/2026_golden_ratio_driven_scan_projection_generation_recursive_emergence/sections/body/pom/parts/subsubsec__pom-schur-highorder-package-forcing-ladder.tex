\paragraph{高阶单包完备性、非平凡通道强迫与幂和谱阶梯}
在上述双向可逆编译桥之外，固定分辨率上的单阶 Schur 包还可进一步压缩为三条互补的对象层结论：首先，在高阶区间 \(q\ge |X_m|\) 上，单个 Schur 通道包已经足以恢复完整纤维多重度多重集及其群/代数分解；其次，非平凡通道的总能量不但由循环型方差精确表达，而且可由最原始的二阶碰撞率给出显式正下界；最后，完整多重度阶梯可沿幂和矩的 \(q\)-轴逐层剥离，并且每一级的比值收敛速度由相邻谱隙指数控制。

\begin{theorem}[单个固定阶 Schur 包在 \texorpdfstring{$q\ge |X_m|$}{q>=|X_m|} 时决定完整纤维多重集与对象层分解]\label{thm:pom-highorder-schur-package-determines-full-fiber-multiset}
\leanverified{paper\_pom\_highorder\_schur\_package\_determines\_full\_fiber\_multiset}
固定分辨率 \(m\)，记按重数计的完整纤维多重度多重集为
\[
\mathcal D_m^{\mathrm{full}}
:=
\bigl\{d_m(x):x\in X_m\bigr\}_{\mathrm{mult}},
\qquad
n_m:=|X_m|.
\]
对整数 \(q\ge 2\)，记全 Schur 通道迹向量
\[
\mathbf T_q(m):=\bigl(T_{q,\lambda}(m)\bigr)_{\lambda\vdash q}.
\]
若 \(q\ge n_m\)，则 \(\mathbf T_q(m)\) 唯一决定 \(\mathcal D_m^{\mathrm{full}}\)。因而它同时唯一决定：
\begin{enumerate}
  \item 全部碰撞矩
  \[
  S_r(m):=\sum_{x\in X_m}d_m(x)^r\qquad(r\ge 1);
  \]
  \item 平凡通道生成函数
  \[
  H_m(z)=\prod_{x\in X_m}(1-d_m(x)z)^{-1};
  \]
  \item primitive 纤维坐标
  \[
  b_n(m):=\frac{1}{n}\sum_{d\mid n}\mu(d)\,S_{n/d}(m)\qquad(n\ge 1);
  \]
  \item bin-fold gauge 群与纤维群胚代数的分解型
  \[
  G_m^{\mathrm{bin}}\cong\prod_{x\in X_m}\mathfrak S_{d_m(x)},
  \qquad
  \CC[\mathcal R_m^{\mathrm{bin}}]\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由推论 \ref{cor:pom-schur-traces-moments-bijection}，\(\mathbf T_q(m)\) 唯一决定低阶碰撞矩
\[
S_1(m),S_2(m),\dots,S_q(m).
\]
由于 \(q\ge n_m\)，特别已知前 \(n_m\) 个幂和
\[
p_k:=S_k(m)=\sum_{x\in X_m}d_m(x)^k
\qquad(1\le k\le n_m).
\]
记 \(e_k(\mathcal D_m^{\mathrm{full}})\) 为多重集 \(\mathcal D_m^{\mathrm{full}}\) 的第 \(k\) 个基本对称函数，并约定 \(e_0(\mathcal D_m^{\mathrm{full}})=1\)。Newton--Girard 恒等式给出递推
\[
k\,e_k(\mathcal D_m^{\mathrm{full}})
=
\sum_{j=1}^{k}(-1)^{j-1}e_{k-j}(\mathcal D_m^{\mathrm{full}})\,p_j
\qquad(1\le k\le n_m),
\]
从而 \(e_1,\dots,e_{n_m}\) 被唯一确定。
于是
\[
Q_m(z):=\prod_{x\in X_m}(1-d_m(x)z)
=
\sum_{k=0}^{n_m}(-1)^k e_k(\mathcal D_m^{\mathrm{full}})\,z^k
\]
被唯一恢复；这正是命题 \ref{prop:pom-trivial-channel-finite-decidability} 中的多项式。其根多重集为
\(
\{d_m(x)^{-1}:x\in X_m\}_{\mathrm{mult}}
\)，故 \(\mathcal D_m^{\mathrm{full}}\) 本身唯一确定。

一旦 \(\mathcal D_m^{\mathrm{full}}\) 已知，全部碰撞矩 \(S_r(m)\) 对任意 \(r\ge 1\) 都可直接计算，得到第 \(1\) 项；第 \(2\) 项由 \(H_m(z)=\prod_x(1-d_m(x)z)^{-1}\) 立得；第 \(3\) 项由命题 \ref{prop:pom-schur-trace-jacobi-trudi-euler-primitive-fiber} 中的 Möbius 反演公式得到。最后，命题 \ref{prop:fold-bin-gauge-decomposition} 与命题 \ref{prop:fold-bin-groupoid-algebra-wedderburn} 表明，bin-fold gauge 群与纤维群胚代数的分解型只依赖于各纤维多重度的取值及其重数，因此第 \(4\) 项亦被唯一决定。证毕。
\end{proof}

\begin{theorem}[非平凡 Schur 通道的不可消去性与显式能量下界]\label{thm:pom-schur-nontrivial-energy-explicit-lower-bound}
\leanverified{paper\_pom\_schur\_nontrivial\_energy\_explicit\_lower\_bound}
固定分辨率 \(m\) 与整数 \(q\ge 2\)。定义非平凡 Schur 通道总能量
\[
\mathcal E_q^{\mathrm{nt}}(m)
:=
\sum_{\lambda\vdash q,\ \lambda\neq(q)}
\left|\frac{T_{q,\lambda}(m)}{f^\lambda}\right|^2
=
\mathcal V_q(m),
\]
并记归一化二阶碰撞率
\[
\mathrm{Col}_m:=2^{-2m}S_2(m).
\]
则有显式严格下界
\[
\boxed{
\mathcal E_q^{\mathrm{nt}}(m)
\ge
\frac{\binom{q}{2}^2}{(q!)^3}\,2^{2mq}\bigl(1-\mathrm{Col}_m\bigr)^2.
}
\]
特别地，只要 \(|X_m|>1\)，便有
\[
\mathcal E_q^{\mathrm{nt}}(m)>0
\qquad(\forall\,q\ge 2),
\]
从而对每个 \(q\ge 2\) 都至少存在一个非平凡分拆 \(\lambda\neq(q)\) 使
\[
T_{q,\lambda}(m)\neq 0.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-schur-variance-decomposition}，
\[
\mathcal E_q^{\mathrm{nt}}(m)
=
\frac{1}{q!}\sum_{\sigma\in S_q}
\bigl|\mathcal C_\sigma(m)-\overline{\mathcal C}(m)\bigr|^2,
\qquad
\overline{\mathcal C}(m):=\frac{1}{q!}\sum_{\sigma\in S_q}\mathcal C_\sigma(m).
\]
令 \(e\in S_q\) 为恒等元，并令 \(\tau\in S_q\) 为任一换位。则
\[
\mathcal C_e(m)=S_1(m)^q=(2^m)^q=2^{mq},
\]
\[
\mathcal C_\tau(m)=S_2(m)\,S_1(m)^{q-2}=2^{mq}\,\mathrm{Col}_m.
\]
另一方面，若 \(\sigma\) 的循环型为 \(1^{c_1}2^{c_2}\cdots\)，则由 \(S_r(m)\le S_1(m)^r\) 得
\[
\mathcal C_\sigma(m)
=
\prod_{r\ge 1}S_r(m)^{c_r}
\le
\prod_{r\ge 1}S_1(m)^{r c_r}
=
S_1(m)^q
=
2^{mq}.
\]
因此，\(\mathcal C_e(m)\) 是全部 \(\mathcal C_\sigma(m)\) 的最大值。把 \(\binom{q}{2}\) 个换位项单独拿出，并把其余 \(q!-1-\binom{q}{2}\) 项都以上界 \(\mathcal C_e(m)\) 估计，可得
\[
\overline{\mathcal C}(m)
\le
\frac{1}{q!}
\Bigl(
\mathcal C_e(m)
+\binom{q}{2}\mathcal C_\tau(m)
+\bigl(q!-1-\binom{q}{2}\bigr)\mathcal C_e(m)
\Bigr).
\]
于是
\[
\mathcal C_e(m)-\overline{\mathcal C}(m)
\ge
\frac{\binom{q}{2}}{q!}
\bigl(\mathcal C_e(m)-\mathcal C_\tau(m)\bigr)
=
\frac{\binom{q}{2}}{q!}\,2^{mq}\bigl(1-\mathrm{Col}_m\bigr).
\]
再由
\[
\mathcal E_q^{\mathrm{nt}}(m)
=
\frac{1}{q!}\sum_{\sigma\in S_q}
\bigl|\mathcal C_\sigma(m)-\overline{\mathcal C}(m)\bigr|^2
\ge
\frac{1}{q!}\bigl(\mathcal C_e(m)-\overline{\mathcal C}(m)\bigr)^2
\]
立得所示下界。

若 \(|X_m|>1\)，则 \(\{d_m(x)\}_{x\in X_m}\) 至少含两个正项，故严格不等式
\[
S_2(m)=\sum_{x\in X_m}d_m(x)^2
<
\left(\sum_{x\in X_m}d_m(x)\right)^2
=
S_1(m)^2
=
2^{2m}
\]
成立，即 \(\mathrm{Col}_m<1\)。代回下界便得 \(\mathcal E_q^{\mathrm{nt}}(m)>0\)。由于 \(\mathcal E_q^{\mathrm{nt}}(m)\) 是非负项和，这又迫使至少一个非平凡通道满足 \(T_{q,\lambda}(m)\neq 0\)。证毕。
\end{proof}

\leanverified{paper\_pom\_power\_sum\_recursive\_spectrum\_peeling}
\begin{theorem}[碰撞矩轴的递归谱剥离：完整多重度阶梯与指数收敛率]\label{thm:pom-power-sum-recursive-spectrum-peeling}
固定分辨率 \(m\)，沿用定理 \ref{thm:pom-fiber-spectrum-prony-hankel-rank} 的有限指数和表示，并把不同纤维多重度按降序重写为
\[
\Delta_{1,m}>\Delta_{2,m}>\cdots>\Delta_{r_m,m}>0,
\qquad
N_{j,m}:=\#\{x\in X_m:\ d_m(x)=\Delta_{j,m}\}.
\]
则对所有整数 \(q\ge 0\) 有
\[
S_q(m)=\sum_{j=1}^{r_m}N_{j,m}\Delta_{j,m}^{\,q}.
\]
定义残差序列
\[
R_m^{(0)}(q):=S_q(m),
\qquad
R_m^{(j)}(q):=R_m^{(j-1)}(q)-N_{j,m}\Delta_{j,m}^{\,q}
\qquad(1\le j\le r_m).
\]
则对每个 \(1\le j\le r_m\) 都有精确表示
\[
R_m^{(j-1)}(q)=\sum_{s=j}^{r_m}N_{s,m}\Delta_{s,m}^{\,q},
\]
从而极限
\[
\Delta_{j,m}
=
\lim_{q\to\infty}\frac{R_m^{(j-1)}(q+1)}{R_m^{(j-1)}(q)},
\qquad
N_{j,m}
=
\lim_{q\to\infty}\frac{R_m^{(j-1)}(q)}{\Delta_{j,m}^{\,q}}
\]
全部存在，并可由 \(q\)-轴递归恢复完整多重度阶梯。
若 \(1\le j<r_m\) 并定义谱隙比
\[
\theta_{j,m}:=\frac{\Delta_{j+1,m}}{\Delta_{j,m}}\in(0,1),
\]
则进一步有指数收敛估计
\[
\boxed{
\frac{R_m^{(j-1)}(q+1)}{R_m^{(j-1)}(q)}
=
\Delta_{j,m}+O\!\bigl(\theta_{j,m}^{\,q}\bigr),
}
\]
\[
\boxed{
\frac{R_m^{(j-1)}(q)}{\Delta_{j,m}^{\,q}}
=
N_{j,m}+O\!\bigl(\theta_{j,m}^{\,q}\bigr),
}
\]
其中隐含常数仅依赖于固定的 \(m\)。最后一级 \(j=r_m\) 时，上述两式分别精确等于 \(\Delta_{r_m,m}\) 与 \(N_{r_m,m}\)。
\end{theorem}
\begin{proof}
定理 \ref{thm:pom-fiber-spectrum-prony-hankel-rank} 已给出
\[
S_q(m)=\sum_{j=1}^{r_m}N_{j,m}\Delta_{j,m}^{\,q}.
\]
按定义减去前 \(j-1\) 级主项，立即得到
\[
R_m^{(j-1)}(q)=\sum_{s=j}^{r_m}N_{s,m}\Delta_{s,m}^{\,q}
\qquad(1\le j\le r_m).
\]
当 \(j=r_m\) 时，
\[
R_m^{(r_m-1)}(q)=N_{r_m,m}\Delta_{r_m,m}^{\,q},
\]
故两条极限与最后一条精确陈述都显然成立。

现取 \(1\le j<r_m\)。从上式提取首项可写成
\[
R_m^{(j-1)}(q)
=
N_{j,m}\Delta_{j,m}^{\,q}
\left(
1+
\sum_{s=j+1}^{r_m}\frac{N_{s,m}}{N_{j,m}}
\left(\frac{\Delta_{s,m}}{\Delta_{j,m}}\right)^q
\right).
\]
由于对所有 \(s\ge j+1\) 都有
\(
\Delta_{s,m}/\Delta_{j,m}\le \theta_{j,m}<1
\)，括号中的修正项满足
\[
\sum_{s=j+1}^{r_m}\frac{N_{s,m}}{N_{j,m}}
\left(\frac{\Delta_{s,m}}{\Delta_{j,m}}\right)^q
=
O\!\bigl(\theta_{j,m}^{\,q}\bigr).
\]
因此
\[
R_m^{(j-1)}(q)
=
N_{j,m}\Delta_{j,m}^{\,q}
\Bigl(1+O\!\bigl(\theta_{j,m}^{\,q}\bigr)\Bigr),
\]
两边除以 \(\Delta_{j,m}^{\,q}\) 即得
\[
\frac{R_m^{(j-1)}(q)}{\Delta_{j,m}^{\,q}}
=
N_{j,m}+O\!\bigl(\theta_{j,m}^{\,q}\bigr),
\]
从而第二个极限成立。

再由同一展开式对 \(q+1\) 与 \(q\) 取比，
\[
\frac{R_m^{(j-1)}(q+1)}{R_m^{(j-1)}(q)}
=
\Delta_{j,m}
\frac{1+O\!\bigl(\theta_{j,m}^{\,q+1}\bigr)}
{1+O\!\bigl(\theta_{j,m}^{\,q}\bigr)}
=
\Delta_{j,m}+O\!\bigl(\theta_{j,m}^{\,q}\bigr),
\]
便得到第一条估计与相应极限。逐级递归即恢复全部 \(\Delta_{j,m}\) 与 \(N_{j,m}\)。证毕。
\end{proof}
