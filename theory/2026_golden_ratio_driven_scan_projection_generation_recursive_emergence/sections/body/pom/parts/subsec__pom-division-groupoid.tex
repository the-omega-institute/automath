\subsection{结论簇 II：商余作为平移群胚横截面}\label{subsec:pom-division-groupoid}
\begin{definition}[平移群胚与基本域]\label{def:pom-translation-groupoid}
固定整数 $b>0$，令 $\mathbb{Z}$ 作用于 $\mathbb{Z}$：
\[
k\cdot a:=a-kb.
\]
其轨道即同余类 $a\bmod b$。取基本域
\[
\mathcal F_b:=\{0,1,\dots,b-1\},
\]
并定义投影
\[
\Pi_b(a):=\text{唯一满足 }a\equiv r\pmod b\text{ 且 }r\in\mathcal F_b\text{ 的 }r。
\]
\end{definition}

\begin{theorem}[商余作为横截面投影]\label{thm:pom-division-section}
对任意 $a\in\mathbb{Z}$，存在唯一 $(q,r)\in\mathbb{Z}\times\mathcal F_b$ 使
\[
a=qb+r,\qquad r=\Pi_b(a).
\]
其中 $q$ 是轨道坐标，$r$ 是沿平移轨道投影到横截面的像。

\end{theorem}
