\begin{proposition}[四阶碰撞核的五态非负整数 realization]\label{prop:pom-s4-recurrence}

\begin{corollary}[四阶碰撞指数与 \texorpdfstring{$h_4$}{h4}]\label{cor:pom-s4-asymptotic}

$$
x^5-2x^4-7x^3-2x+2=0
$$
的最大实根（\(r_4\approx 3.846059041178\)）。则
$$
S_4(m)=\Theta(r_4^m),
\qquad
h_4:=\lim_{m\to\infty}\frac{1}{m}\Bigl(-\log\sum_{x\in X_m}\pi_m(x)^4\Bigr)
=\log\!\left(\frac{16}{r_4}\right).

\end{corollary}
\begin{proof}
由命题 \ref{prop:pom-s4-recurrence}，生成函数分母为 \(\det(I-zA_4)\)，其最小模零点为 \(z_\star=1/\rho(A_4)\)，从而 \(S_4(m)=\Theta(r_4^m)\)。并且
$$
\sum_{x\in X_m}\pi_m(x)^4=\sum_x\frac{d_m(x)^4}{16^m}=\frac{S_4(m)}{16^m},

代入渐近即得 \(h_4=\log(16/r_4)\)。证毕。
\leanverified{paper\_pom\_s4\_asymptotic}

\begin{remark}[四阶碰撞核的贴线 RH 破缺：谱分解数值证书（\texorpdfstring{$A_4$}{A4}）]\label{rem:pom-collision-rh-break-a4-spectrum}

\[
\mu_2\approx -1.963592213449,\qquad
\mu_3\approx 0.503913309522,\qquad
\mu_{4,5}\approx -0.193190068626\pm 0.698726683936\,\mathrm{i}.

\Lambda_4:=\max_{\mu\in\sigma(A_4)\setminus\{r_4\}}|\mu|
=|\mu_2|
\approx 1.963592213449,

\sqrt{r_4}\approx 1.961137180612

\[
\Lambda_4>\sqrt{r_4},\qquad
R_4:=\frac{\Lambda_4}{\sqrt{r_4}}\approx 1.001251841463.
\]
用“极点实部指数”口径可写为
\[
\beta_4:=\frac{\log \Lambda_4}{\log r_4}\approx 0.500928740205=\frac12+9.2874\times 10^{-4},

\delta_4:=\log\!\left(\frac{\Lambda_4^2}{r_4}\right)=2\log R_4\approx 0.002502117125.

 该超出由一个稳定的负实模态主导（\(\mu_2<0\)），从而对应一条不可消去的反相振荡慢模。
\end{remark}

\begin{remark}[四阶碰撞核的单参数 RH 临界点：边权阈值 \texorpdfstring{$t_\star$}{t*}（resultant 消元，可审计）]\label{rem:pom-collision-rh-break-a4-threshold}
把 \(A_4\) 中唯一的多重边权 \(7\) 视为连续参数 \(t>0\)，定义一参数族
\[
A_4(t):=
\begin{pmatrix}
0&1&0&0&0\\
0&0&t&0&1\\
0&1&2&0&0\\
1&0&1&0&0\\
0&0&0&1&0
\end{pmatrix},
\qquad
p_t(x):=\det(xI-A_4(t))=x^5-2x^4-tx^3-2x+2.
\]
记 \(r(t)=\rho(A_4(t))\)、\(\Lambda(t)=\max_{\mu\in\mathrm{spec}(A_4(t))\setminus\{r(t)\}}|\mu|\)，并令 Newman 型偏移
\(
\delta(t):=\log(\Lambda(t)^2/r(t))
\)
（与注 \ref{rem:pom-collision-rh-break-a4} 的口径一致）。
在该族上，\(\delta(t)\) 的达界点可由\textbf{负实根通道}刻画：临界点等价于存在 \(y<0\) 使得 \(p_t(y)=p_t(y^2)=0\)，从而（排除退化根 \(y=\pm 1\)）满足
\[
y^8-2y^6+2y^5-2y^4+2y^3-2=0.
\]
令 \(z:=-y>0\)，则等价地
\(
z^8-2z^6-2z^5-2z^4-2z^3-2=0
\)，并且由 \(p_t(y)=0\) 可显式解出
\[
t=\frac{y^5-2y^4-2y+2}{y^3}
=z^2+2z-\frac{2}{z^2}-\frac{2}{z^3}.
\]
进一步对 \(y\) 作结式消元可得 \(t\) 的 8 次整系数证书 \(P_4(t)=0\)；其唯一正实根记为 \(t_\star\)。
数值核验表明：当 \(t<t_\star\) 时 \(\Lambda(t)<\sqrt{r(t)}\)（严格 kernel RH），当 \(t=t_\star\) 时恰达界（RH-sharp，且 \(\mu_\star=-\sqrt{r_\star}\)），而当 \(t>t_\star\) 时出现 \(\Lambda(t)>\sqrt{r(t)}\) 的越线。
 因此原整数核 \(A_4=A_4(7)\) 位于临界右侧且仅有临界附近的超出 \(7-t_\star\ll 1\)，从而解释了 \(R_4-1\) 的 \(10^{-3}\) 量级。
\end{remark}

\begin{proposition}[\texorpdfstring{$h_q$}{hq} 的几何化：投影输出测度的 Rényi 维数谱]\label{prop:pom-renyi-dimension-spectrum}
令 \(w_m(x)=d_m(x)/2^m\) 为均匀微态的折叠推前。把分辨率 \(m\) 与几何尺度对应为
\[
\varepsilon_m:=\varphi^{-m}.
\]
定义输出测度在尺度 \(\varepsilon_m\) 下的 \(q\)-阶 Rényi 维数近似为
\[
D_q(m):=\frac{1}{q-1}\cdot \frac{\log\sum_{x\in X_m}w_m(x)^q}{\log \varepsilon_m}
\qquad(q\ge 2).
\]
若 \(h_q=\lim_{m\to\infty}\frac{1}{m}H_q(m)\) 存在（例如当 \(S_q(m)=\Theta(r_q^m)\) 时成立），则极限
\[
D_q:=\lim_{m\to\infty}D_q(m)
\]
存在且满足闭式
\[
\boxed{\;
D_q=\frac{h_q}{(q-1)\log\varphi}\; }.
\]
\end{proposition}
\begin{proof}
由定义 \(H_q(m)=-\log\sum_x w_m(x)^q\) 与 \(\log\varepsilon_m=-m\log\varphi\) 得
\[
D_q(m)=\frac{1}{q-1}\cdot \frac{-H_q(m)}{-m\log\varphi}
=\frac{1}{(q-1)\log\varphi}\cdot \frac{1}{m}H_q(m).
\]

\[
\boxed{\;
h_{q_1}\ \ge\ \frac{q_2-q_1}{q_2-q_0}\,h_{q_0}\ +\ \frac{q_1-q_0}{q_2-q_0}\,h_{q_2}\; }.
\]
\end{proposition}
\begin{proof}
固定 \(m\)。取任意 \(q_0,q_2\ge 0\) 与 \(\theta\in(0,1)\)，令 \(q_1=\theta q_0+(1-\theta)q_2\)。
由 Hölder 不等式（或 log-sum-exp 的凸性）得
\[
S_{q_1}(m)=\sum_{x\in X_m}d_m(x)^{q_1}
\le
\left(\sum_{x\in X_m}d_m(x)^{q_0}\right)^{\theta}
\left(\sum_{x\in X_m}d_m(x)^{q_2}\right)^{1-\theta}
=S_{q_0}(m)^{\theta}S_{q_2}(m)^{1-\theta}.
\]
取对数即得 \(q\mapsto \log S_q(m)\) 的凸性。
\par
令 \(q_0<q_1<q_2\) 为整数并取
\(
\theta=\frac{q_2-q_1}{q_2-q_0}
\)
使得 \(q_1=\theta q_0+(1-\theta)q_2\)。
对上式取 \(\frac{1}{m}\log\) 并令 \(m\to\infty\)，得到
\[
\log r_{q_1}\le \theta\log r_{q_0}+(1-\theta)\log r_{q_2},
\]
即 \(P\) 的离散凸性。最后 \(h_q=q\log 2-P(q)\) 为线性函数减去凸函数，从而在整数点上为凹。证毕。
\leanverified{paper\_pom\_pressure\_convexity}
\end{proof}

\begin{proposition}[跨 \texorpdfstring{$q$}{q} 不等式：\texorpdfstring{$g(q)$}{g(q)} 的单调性]\label{prop:pom-crossq-g-monotone}
对任意整数 \(m\ge 2\) 与 \(2\le p<q\)，令 \(d_m=(d_m(x))_{x\in X_m}\) 并记 \(S_k(m)=\sum_x d_m(x)^k=\|d_m\|_k^k\)。
则有限维范数不等式给出
\[
\|d_m\|_p\ \le\ |X_m|^{\frac{1}{p}-\frac{1}{q}}\ \|d_m\|_q,
\]
从而
\[
S_q(m)\ \ge\ S_p(m)^{q/p}\,|X_m|^{1-q/p}.
\]
若存在 \(r_k\) 使 \(S_k(m)=\Theta(r_k^m)\)，并且 \(|X_m|\sim \varphi^m\)，则有跨阶约束
\[
\boxed{\;
\log r_q \ \ge\ \frac{q}{p}\log r_p + \left(1-\frac{q}{p}\right)\log\varphi\; }.
\]
等价地，定义
\[
g(q):=\frac{\log r_q-\log\varphi}{q},
\]
则 \(g(q)\) 在 \(q\ge 2\) 上单调不减。
表 \ref{tab:fold_collision_crossq_g_profile_q2_23} 给出 \(q=2,\dots,23\) 的可复算读数，可作为后续扩展/重实现的回归一致性检查（脚本 \texttt{scripts/exp\_fold\_collision\_crossq\_g\_profile\_q2\_23.py} 一键生成）。
\leanverified{paper\_pom\_crossq\_g\_monotone}
\end{proposition}
\begin{proof}
对长度 \(n=|X_m|\) 的非负向量 \(d_m\)，当 \(2\le p<q\) 时有 \(\|d_m\|_p\le n^{1/p-1/q}\|d_m\|_q\)。
两边取 \(q\) 次幂得到
\(
\|d_m\|_q^q \ge \|d_m\|_p^q\,n^{1-q/p}
\)，
即
\(
S_q(m)\ge S_p(m)^{q/p}|X_m|^{1-q/p}
\)。
对两边取 \(\frac{1}{m}\log\) 并令 \(m\to\infty\)，利用
\(
\lim_{m\to\infty}\frac{1}{m}\log S_k(m)=\log r_k
\)
与
\(
\lim_{m\to\infty}\frac{1}{m}\log|X_m|=\log\varphi
\)
即得第一式；再代回定义得到 \(g(q)\) 的单调性。证毕。
\end{proof}

\begin{definition}[闭合缺陷 \texorpdfstring{$\Delta_q$}{Deltaq}：共振/选择定则的不变量化]\label{def:pom-closure-defect-deltaq}
令 \(h=\lfloor q/2\rfloor\)。
定义\textbf{闭合缺陷}（closure defect）
\[
\boxed{\;
\Delta_q\ :=\ (2h+1)-d_q\;}
\]
其中 \(2h+1\) 为“差分模态截断”下的名义模态数（注 \ref{rem:pom-closure-dim-mode-trunc}），而 \(d_q\) 为最小递推阶数/最小 realization 维数（推论 \ref{cor:pom-moment-dim-collapse}）。
\end{definition}
\begin{remark}[解释：\texorpdfstring{$\Delta_q>0$}{Deltaq>0} 即“选择定则”，并给出可量化降维增益]\label{rem:pom-closure-defect-selection-rule}
当 \(\Delta_q>0\) 时，定理 \ref{thm:pom-moment-resonance} 表明存在非平凡的模态约束使得名义 \(2h+1\) 维模态 realization 不是最小实现；因此 \(\Delta_q\) 可被视为“可激发模态空间的余维”，即共振/坍缩触发的\emph{选择定则强度}。
\par
工程上，这直接量化了对 \(q\)-阶谱后端的可压缩增益：若某实现以 \(2h+1\) 个差分模态作为状态闭包并进行递推，则在共振相区可进一步压到 \(d_q\) 维最小实现（由递推/伴随矩阵或 Hankel 最小化恢复），其名义压缩比约为
\[
\mathrm{gain}(q)\ \approx\ \frac{2h+1}{d_q}.
\]
例如 \(q=15\) 时，\(2h+1=15\) 而已核验 \(d_{15}=11\)，对应一个稳定的结构性降维缺口。
\end{remark}

\begin{remark}[闭合维数 \texorpdfstring{$2\lfloor q/2\rfloor+1$}{2 floor(q/2)+1} 的结构来源：同余角色模态的有限截断]\label{rem:pom-closure-dim-mode-trunc}
命题 \ref{prop:pom-s2-plancherel} 给出一个把“几何—同余—谱”焊接在一起的可审计模板：二阶碰撞矩
\(
S_2(m)=\sum_r c_m(r)^2
\)
既是同余切片上 residue 分布的 \(\ell^2\) 质量，也是其离散傅里叶系数的 Plancherel 能量平均；并且每个频率 \(j\) 上的傅里叶系数 \(\widehat c_m(j)\) 由局部因子 \(\prod_k(1+\chi_j(F_{k+1}))\) 完全决定。

在更高阶 \(q\) 的 moment-kernel 编译（命题 \ref{prop:pom-moment-kernel-exists}）中，本质上是在同步直积上施加“逐位输出一致”的约束；该约束把 \(q\) 份拷贝的相位信息压缩为一个\textbf{有限取值的差分模态索引}：它记录“相位在共轭配对下的净偏置”（可理解为 \(\widehat c\) 与 \(\overline{\widehat c}\) 的幂次差 \(a-b\) 的有效范围）。由配对/共轭不变性，净偏置的可取范围被强制截断为
\[
-(\lfloor q/2\rfloor)\ \le\ a-b\ \le\ \lfloor q/2\rfloor,
\]
因此在\emph{非共振的典型情形}下，只需跟踪 \(2\lfloor q/2\rfloor+1\) 个模态即可闭合地递推到下一步分辨率；这给出了“每两阶增加 2 个自由度”的结构原因。该模态截断与 Hankel 秩下界（定理 \ref{thm:pom-moment-hankel}）在已核验范围 \(q\le 8\) 内精确吻合，从而把经验指纹升级为可解释的有限闭包机制。
而表 \ref{tab:fold_collision_moment_recursions_9_17} 的 \(q\ge 9\) 现象表明：当 Fibonacci 同余结构诱发代数退化时，可见模态空间会出现额外约束，使得读出所能激发的子空间维数严格小于名义上界（推论 \ref{cor:pom-moment-dim-collapse}）。因此更关键的统一对象不再是“继续提高 \(q\)”，而是刻画何时、以及为何出现不可激发的模态（见猜想 \ref{conj:pom-moment-visible} 与定理 \ref{thm:pom-moment-resonance} 的模态约束口径）。
\end{remark}
