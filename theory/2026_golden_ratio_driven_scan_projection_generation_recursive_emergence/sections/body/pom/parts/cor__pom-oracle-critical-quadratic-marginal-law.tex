\begin{corollary}[外置预算临界点的二次边缘律与中等偏差尺度]\label{cor:pom-oracle-critical-quadratic-marginal-law}
\leanverified{paper\_pom\_oracle\_critical\_quadratic\_marginal\_law}
在定理 \ref{thm:pom-oracle-success-variational-laplace} 与定理 \ref{thm:pom-ldp-uniform-and-wm-exact-spectrum} 的设定下，
假设 \(\tau\) 在 \(q=1\) 的左邻域二次可微且 \(\tau''(1^-)>0\)。
记
\[
a_1:=\tau'(1^-),\qquad
\beta_{\mathrm{crit}}:=\frac{a_1}{\log 2}.
\]
取线性比特预算 \(B_m:=\lceil \beta m\rceil\) 并令 \(a_m:=B_m\log 2/m\)。
则当 \(\beta<\beta_{\mathrm{crit}}\) 时有指数律
\[
-\frac{1}{m}\log \mathrm{Succ}^{\mathrm{orc}}_m(B_m)
=\inf_{\alpha\le \beta\log 2}I_w(\alpha)+o(1)
=I_w(\beta\log 2)+o(1),
\]
并且当 \(\beta\uparrow\beta_{\mathrm{crit}}\)（仍满足 \(\beta<\beta_{\mathrm{crit}}\)）时具有局部二次展开
\[
I_w(\beta\log 2)
=\frac{(\log 2)^2}{2\,\tau''(1^-)}(\beta_{\mathrm{crit}}-\beta)^2
+o\!\bigl((\beta_{\mathrm{crit}}-\beta)^2\bigr).
\]
从而得到中等偏差尺度下的定量边缘律
\[
\log \mathrm{Succ}^{\mathrm{orc}}_m(B_m)
=-\frac{(\log 2)^2}{2\,\tau''(1^-)}(\beta_{\mathrm{crit}}-\beta)^2\,m
+o\!\bigl(m(\beta_{\mathrm{crit}}-\beta)^2\bigr).
\]
\end{corollary}

\begin{proof}
由定理 \ref{thm:pom-oracle-success-variational-laplace}，
\[
\mathrm{Succ}^{\mathrm{orc}}_m(B_m)
=\mathbb{E}_{w_m}\!\left[\min\bigl(1,e^{m(a_m-Y_m)}\bigr)\right],
\qquad
Y_m(x):=\frac{1}{m}\log d_m(x).
\]
分解为
\[
\mathrm{Succ}^{\mathrm{orc}}_m(B_m)=\mathbb{P}_{w_m}(Y_m\le a_m)+R_m,
\qquad
R_m:=\mathbb{E}_{w_m}\!\left[e^{m(a_m-Y_m)}\mathbf 1_{\{Y_m>a_m\}}\right].
\]
对任意 \(\eta>0\)，有
\[
R_m\le \mathbb{P}_{w_m}\!\left(a_m<Y_m\le a_m+\eta\right)+e^{-m\eta}.
\]
由 \(w_m\)-LDP（定理 \ref{thm:pom-ldp-uniform-and-wm-exact-spectrum}），当 \(\beta<\beta_{\mathrm{crit}}\)（即 \(\beta\log 2<a_1\)）时，
\[
\lim_{m\to\infty}-\frac{1}{m}\log \mathbb{P}_{w_m}(Y_m\le \beta\log 2)
=\inf_{\alpha\le \beta\log 2}I_w(\alpha)
=I_w(\beta\log 2),
\]
其中最后一步使用 \(I_w\) 的凸性与其唯一零点位于 \(a_1\)。
再令 \(\eta\downarrow 0\) 并注意 \(a_m=\beta\log 2+O(1/m)\)，得
\(-\frac{1}{m}\log \mathrm{Succ}^{\mathrm{orc}}_m(B_m)=I_w(\beta\log 2)+o(1)\)。
\par
为得二次展开，令 \(\Lambda_w(t)=\sup_{\alpha\in\RR}(t\alpha-I_w(\alpha))\) 为对偶对数矩母函数。
在上述热力学制度下，\(\Lambda_w\) 在 \(t=0^-\) 二次可微且
\(\Lambda_w'(0^-)=a_1\)、\(\Lambda_w''(0^-)=\tau''(1^-)\)。
由 Legendre 对偶的局部反演可得
\[
I_w(\alpha)=\frac{(\alpha-a_1)^2}{2\,\tau''(1^-)}+o\!\bigl((\alpha-a_1)^2\bigr)
\qquad(\alpha\to a_1,\ \alpha<a_1).
\]
取 \(\alpha=\beta\log 2\) 并注意 \(a_1-\beta\log 2=(\beta_{\mathrm{crit}}-\beta)\log 2\) 即得所示展开与中等偏差式。证毕。
\end{proof}
