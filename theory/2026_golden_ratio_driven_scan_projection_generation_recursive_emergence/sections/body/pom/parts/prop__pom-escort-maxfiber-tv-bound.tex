\begin{proposition}[escort 测度对极大纤维的指数聚束：有限层的全变差界]\label{prop:pom-escort-maxfiber-tv-bound}
\leanverified{paper\_pom\_escort\_maxfiber\_tv\_bound}
固定分辨率 \(m\ge 1\)。
对任意 \(q>0\) 定义 \(q\)-escort 分布
\[
w_{m,q}(x):=\frac{d_m(x)^q}{S_q(m)},
\qquad
S_q(m):=\sum_{y\in X_m}d_m(y)^q.
\]
记
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
A_{\max}:=\{x\in X_m:\ d_m(x)=D_m\},
\qquad
K:=|A_{\max}|,
\]
并记次大多重度
\[
D_m^{(2)}:=\max_{x\notin A_{\max}}d_m(x),
\]
其中约定 \(D_m^{(2)}=0\) 当 \(A_{\max}=X_m\)。
令 \(M:=|X_m|-K\)。
则成立显式全变差估计
\[
\left\|w_{m,q}-\mathrm{Unif}(A_{\max})\right\|_{\mathrm{TV}}
\le
\frac{M}{K}\left(\frac{D_m^{(2)}}{D_m}\right)^{q}.
\]
\end{proposition}

\begin{proof}
由于 \(d_m(x)=D_m\) 在 \(A_{\max}\) 上常值，\(w_{m,q}\) 在 \(A_{\max}\) 上亦为常密度，从而
\[
\left\|w_{m,q}-\mathrm{Unif}(A_{\max})\right\|_{\mathrm{TV}}
=1-w_{m,q}(A_{\max}).
\]
另一方面，
\[
S_q(m)
=KD_m^q+\sum_{x\notin A_{\max}}d_m(x)^q
\le KD_m^q+M\,(D_m^{(2)})^q,
\]
故
\[
w_{m,q}(A_{\max})
=\frac{KD_m^q}{S_q(m)}
\ge
\frac{KD_m^q}{KD_m^q+M(D_m^{(2)})^q}.
\]
将其代回并化简得到
\[
1-w_{m,q}(A_{\max})
\le
\frac{M(D_m^{(2)})^q}{KD_m^q}
=\frac{M}{K}\left(\frac{D_m^{(2)}}{D_m}\right)^q.
\]
证毕。
\end{proof}
