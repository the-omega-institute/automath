\begin{theorem}[任意活动度下的根单位滤波与指数级均匀化]\label{thm:pom-fiber-rewrite-root-of-unity-filter-modq-exp-uniformization}
\leanverified{paper\_pom\_fiber\_rewrite\_root\_of\_unity\_filter\_modq\_exp\_uniformization}
固定整数 \(q\ge 2\) 并令 \(\zeta_q:=e^{2\pi i/q}\)。
对任意 \(m\ge 1\)、\(x\in X_m\) 与活动度 \(t>0\)，在纤维 \(\Fold_m^{-1}(x)\) 上取 hard-core 权重分布
\[
\PP_t(\omega\mid x):=\frac{t^{r(\omega)}}{Z_x(t)},
\qquad
Z_x(t):=\sum_{\omega\in\Fold_m^{-1}(x)} t^{r(\omega)},
\]
并记
\[
R_x:=r(\omega)\in\ZZ_{\ge 0},
\qquad
R_x^{(q)}:=R_x\bmod q\in \ZZ/q\ZZ.
\]
则对任意 \(a\in\ZZ/q\ZZ\) 有精确公式
\[
\PP_t(R_x^{(q)}=a\mid x)
:=\frac{1}{q\,Z_x(t)}\sum_{k=0}^{q-1}\zeta_q^{-ak}\,Z_x(t\zeta_q^{k}).
\]
令传递矩阵与谱半径
\[
A(w):=
\begin{pmatrix}
1 & w\\
1 & 0
\end{pmatrix},
\qquad
\rho(w):=\rho\bigl(A(w)\bigr),
\]
并定义收缩率
\[
\vartheta(t,q):=\max_{1\le k\le q-1}\frac{\rho(t\zeta_q^k)}{\rho(t)}\in[0,1).
\]
则存在常数 \(C(t,q)\in(0,\infty)\)（仅依赖于 \(t,q\)）使得对所有 \(x\in X_m\) 与所有 \(a\in\ZZ/q\ZZ\) 成立
\[
\left|\PP_t(R_x^{(q)}=a\mid x)-\frac{1}{q}\right|
\le \frac{q-1}{q}\,C(t,q)^{c(x)}\,\vartheta(t,q)^{\,|V(\Gamma_x)|},
\qquad
\left\|\mathcal L_t(R_x^{(q)}\mid x)-\mathrm{Unif}(\ZZ/q\ZZ)\right\|_{\mathrm{TV}}
\le \frac{q-1}{2}\,C(t,q)^{c(x)}\,\vartheta(t,q)^{\,|V(\Gamma_x)|}.
\]
\end{theorem}

\begin{proof}
\textbf{根单位滤波.}
将 \(Z_x(t)\) 写为
\[
Z_x(t)=\sum_{j\ge 0}N_j(x)\,t^j,
\qquad
N_j(x):=\card{\{\omega\in\Fold_m^{-1}(x):\ r(\omega)=j\}}.
\]
则对任意 \(t>0\) 有
\[
\sum_{j\equiv a\ (q)}N_j(x)\,t^j
=\frac{1}{q}\sum_{k=0}^{q-1}\zeta_q^{-ak}\sum_{j\ge 0}N_j(x)\,(t\zeta_q^k)^{j}
=\frac{1}{q}\sum_{k=0}^{q-1}\zeta_q^{-ak}\,Z_x(t\zeta_q^k),
\]
除以 \(Z_x(t)=\sum_j N_j(x)t^j\) 即得概率公式。
\par
\textbf{指数级均匀化.}
由定理 \ref{thm:pom-fiber-indset-factorization}，若 \(\Gamma_x\simeq \bigsqcup_{i=1}^{c(x)} P_{\ell_i}\)，则
\[
Z_x(w)=\prod_{i=1}^{c(x)} I_{\ell_i}(w),
\qquad
|V(\Gamma_x)|=\sum_{i=1}^{c(x)} \ell_i.
\]
对固定 \(w\in\CC\)，独立集多项式 \(I_\ell(w)\) 满足二阶线性递推，因而可写成矩阵幂的矩阵元：
\[
\binom{I_\ell(w)}{I_{\ell-1}(w)}=A(w)^{\ell-1}\binom{1+w}{1}\qquad(\ell\ge 1).
\]
从而存在常数 \(C_w\in(0,\infty)\) 使得对所有 \(\ell\ge 1\) 有 \(|I_\ell(w)|\le C_w\,\rho(w)^\ell\)。
对给定的 \(t>0\) 与有限集合 \(\{t\zeta_q^k:0\le k\le q-1\}\) 取
\[
C(t,q):=\max_{0\le k\le q-1}\max(C_{t\zeta_q^k},1)
\]
即可使界对所有 \(k\) 同时成立。
\par
当 \(1\le k\le q-1\) 时，矩阵 \(A(t\zeta_q^k)\) 的非负模矩阵为 \(|A(t\zeta_q^k)|=A(t)\)。
由 Perron--Frobenius 型比较 \(\rho(A)\le \rho(|A|)\) 得 \(\rho(t\zeta_q^k)\le \rho(t)\)；
并且由于 \(t\zeta_q^k\) 含非平凡相位，该不等式严格成立，从而 \(\vartheta(t,q)<1\)。
于是对所有 \(\ell\ge 1\) 有
\[
\left|\frac{I_\ell(t\zeta_q^k)}{I_\ell(t)}\right|
\le
C(t,q)\left(\frac{\rho(t\zeta_q^k)}{\rho(t)}\right)^\ell
\le
C(t,q)\,\vartheta(t,q)^\ell.
\]
由乘积分解得到
\[
\left|\frac{Z_x(t\zeta_q^k)}{Z_x(t)}\right|
=\prod_{i=1}^{c(x)}\left|\frac{I_{\ell_i}(t\zeta_q^k)}{I_{\ell_i}(t)}\right|
\le C(t,q)^{c(x)}\,\vartheta(t,q)^{\sum_i\ell_i}
=C(t,q)^{c(x)}\,\vartheta(t,q)^{|V(\Gamma_x)|}.
\]
代回根单位滤波式并对 \(k=1,\dots,q-1\) 求和得到
\[
\left|\PP_t(R_x^{(q)}=a\mid x)-\frac{1}{q}\right|
\le \frac{1}{q}\sum_{k=1}^{q-1}\left|\frac{Z_x(t\zeta_q^k)}{Z_x(t)}\right|
\le \frac{q-1}{q}\,C(t,q)^{c(x)}\,\vartheta(t,q)^{|V(\Gamma_x)|}.
\]
总变差界由恒等式
\(
\|P-U\|_{\mathrm{TV}}\le \tfrac12\sum_a |P(a)-U(a)|
\)
立得。证毕。
\end{proof}

\begin{theorem}[根单位滤波收缩率 \texorpdfstring{$\vartheta(t,q)$}{vartheta(t,q)} 的二阶精确渐近]\label{thm:pom-root-unity-filter-vartheta-second-order}
\leanverified{paper\_pom\_root\_unity\_filter\_vartheta\_second\_order}
固定 \(t>0\)，并沿用定理 \ref{thm:pom-fiber-rewrite-root-of-unity-filter-modq-exp-uniformization} 的 \(A(w)\)、\(\rho(w)\) 与 \(\vartheta(t,q)\) 记号。
则当 \(q\to\infty\) 时有二阶展开
\[
\boxed{\;
\vartheta(t,q)
=1-\frac{4\pi^2}{q^2}\,\alpha(t)+O\!\left(\frac{1}{q^4}\right)\;},
\]
其中
\[
\alpha(t)
:=\frac{t\bigl(2t+\sqrt{4t+1}+1\bigr)}{(4t+1)^{3/2}\,(\sqrt{4t+1}+1)^2}\ >\ 0.
\]
特别地，当 \(t=1\) 时，
\[
1-\vartheta(1,q)=\frac{2\sqrt5\,\pi^2}{25}\cdot\frac{1}{q^2}+O\!\left(\frac{1}{q^4}\right).
\]
\end{theorem}
\begin{proof}
矩阵 \(A(w)\) 的特征根为
\[
\lambda_\pm(w)=\frac{1\pm\sqrt{1+4w}}{2}.
\]
当 \(w=t>0\) 时，\(\lambda_+(t)>0\) 且 \(|\lambda_-(t)|<\lambda_+(t)\)，从而 \(\rho(t)=\lambda_+(t)\)。
由连续性，存在邻域 \(U\ni t\) 使得对所有 \(w\in U\) 都有 \(|\lambda_+(w)|>|\lambda_-(w)|\)，故在 \(U\) 上
\(
\rho(w)=|\lambda_+(w)|
\)。
\par
令 \(w=t e^{i\theta}\) 且 \(|\theta|\) 足够小，使 \(w\in U\)。
定义
\[
g(\theta):=\lambda_+(t e^{i\theta})=\frac{1+\sqrt{1+4t e^{i\theta}}}{2}.
\]
由于 \(g\) 在 \(\theta=0\) 的邻域内解析且 \(g(0)>0\)，对 \(g\) 作二阶 Taylor 展开并计算模长可得（注意 \(|g(\theta)|^2\) 仅含偶次项）
\[
\frac{|g(\theta)|}{g(0)}=1-\alpha(t)\,\theta^2+O(\theta^4),
\]
其中 \(\alpha(t)\) 即题述显式表达式，且 \(\alpha(t)>0\) 由 \(t>0\) 立得。
\par
令 \(\theta_k:=2\pi k/q\)。
对固定 \(t\)，由 \(\rho(te^{i\theta})/\rho(t)\) 在紧集 \([\delta,2\pi-\delta]\) 上的连续性与严格小于 \(1\) 性，
可取 \(\delta>0\) 与 \(c_\delta>0\) 使得当 \(|\theta|\in[\delta,2\pi-\delta]\) 时
\(
\rho(te^{i\theta})/\rho(t)\le 1-c_\delta
\)。
另一方面，当 \(q\) 足够大时，\(|\theta_1|=2\pi/q<\delta\)，并由上式得到
\[
\frac{\rho(t\zeta_q)}{\rho(t)}
=\frac{|g(\theta_1)|}{g(0)}
=1-\alpha(t)\theta_1^2+O(\theta_1^4)
=1-\frac{4\pi^2}{q^2}\alpha(t)+O(q^{-4}).
\]
因此对充分大 \(q\)，\(\vartheta(t,q)\) 的最大值由最小非零角度 \(\theta_1\)（及对称的 \(\theta_{q-1}\)）取得，从而得到所示二阶展开。
取 \(t=1\) 并代入 \(\alpha(1)=\sqrt5/50\) 即得特例。证毕。
\end{proof}

\begin{corollary}[模 \texorpdfstring{$q$}{q} 均匀化的 \texorpdfstring{$q^2$}{q^2} 定律]\label{cor:pom-root-unity-filter-q2-law}
\leanverified{paper\_pom\_root\_unity\_filter\_q2\_law}
沿用定理 \ref{thm:pom-fiber-rewrite-root-of-unity-filter-modq-exp-uniformization} 的设定，并固定 \(t>0\)。
记
\[
K(t,q,x):=\frac{q-1}{2}\,C(t,q)^{c(x)}.
\]
则对任意 \(0<\varepsilon<1\)，当 \(q\to\infty\) 时，只要
\[
|V(\Gamma_x)|
\ \ge\
\frac{q^2}{4\pi^2\alpha(t)}
\left(\log\frac{K(t,q,x)}{\varepsilon}+O(1)\right),
\]
就有
\[
\left\|\mathcal L_t(R_x^{(q)}\mid x)-\mathrm{Unif}(\ZZ/q\ZZ)\right\|_{\mathrm{TV}}
\le \varepsilon,
\]
其中 \(\alpha(t)\) 见定理 \ref{thm:pom-root-unity-filter-vartheta-second-order}。
因此在大模数极限中，达到固定均匀精度所需的候选位点规模在主尺度上必为 \(\Theta(q^2)\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-fiber-rewrite-root-of-unity-filter-modq-exp-uniformization} 的总变差界，
\[
\left\|\mathcal L_t(R_x^{(q)}\mid x)-\mathrm{Unif}(\ZZ/q\ZZ)\right\|_{\mathrm{TV}}
\le K(t,q,x)\,\vartheta(t,q)^{|V(\Gamma_x)|}.
\]
取对数并使用定理 \ref{thm:pom-root-unity-filter-vartheta-second-order} 给出的
\(
\log \vartheta(t,q)=-4\pi^2\alpha(t)/q^2+O(q^{-4})
\)，
即可将条件 \(K\vartheta^{|V|}\le \varepsilon\) 等价转写为所示下界（将高阶项吸收到 \(O(1)\) 中）。证毕。
\end{proof}

\begin{corollary}[逆代换次数谱的无缺口区间性]\label{cor:pom-fiber-rewrite-spectrum-no-gaps}
\leanverified{paper\_pom\_fiber\_rewrite\_spectrum\_no\_gaps}
沿用定理 \ref{thm:pom-fiber-rewrite-root-of-unity-filter-modq-exp-uniformization} 的记号。
设候选位点诱导图分解为路径不交并
\[
\Gamma_x\ \cong\ \bigsqcup_{i=1}^{c(x)} P_{\ell_i}.
\]
定义
\[
R_{\max}(x):=\sum_{i=1}^{c(x)}\left\lceil \frac{\ell_i}{2}\right\rceil.
\]
则对系数 \(N_j(x)=\card{\{\omega\in\Fold_m^{-1}(x):\ r(\omega)=j\}}\) 有
\[
N_j(x)>0\qquad\Longleftrightarrow\qquad 0\le j\le R_{\max}(x).
\]
等价地，
\[
\{r(\omega):\ \omega\in\Fold_m^{-1}(x)\}\ =\ \{0,1,2,\dots,R_{\max}(x)\}.
\]
\end{corollary}

\begin{proof}
对单一路径 \(P_{\ell}\)，任取 \(0\le j\le \lceil \ell/2\rceil\)。
取顶点集
\(
S_j:=\{1,3,5,\dots,2j-1\}
\)
（若 \(j=0\) 则取空集），则 \(S_j\) 为 \(P_\ell\) 的独立集且 \(|S_j|=j\)。
因此路径独立集多项式 \(I_\ell(t)\) 的系数满足
\(
[t^j]I_\ell(t)>0
\)
当且仅当 \(0\le j\le \lceil \ell/2\rceil\)。
\par
对不交并 \(\Gamma_x=\bigsqcup_i P_{\ell_i}\)，由定理 \ref{thm:pom-fiber-indset-factorization} 的乘积分解
\(
Z_x(t)=\prod_i I_{\ell_i}(t)
\)，
其系数支持为各分量支持集的 Minkowski 和。
由于每个分量支持均为区间 \(\{0,1,\dots,\lceil \ell_i/2\rceil\}\)，其 Minkowski 和即为
\(
\{0,1,\dots,\sum_i\lceil \ell_i/2\rceil\}
\)，
从而得到所示无缺口结论。证毕。
\end{proof}
