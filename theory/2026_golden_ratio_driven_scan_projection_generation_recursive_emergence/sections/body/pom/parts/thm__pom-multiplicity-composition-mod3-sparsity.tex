\begin{theorem}[\texorpdfstring{$\bmod 3$}{mod 3} 稀少事件的指数衰减：权重分解下的 Witten 指数与取向签名]\label{thm:pom-multiplicity-composition-mod3-sparsity}
固定 \(q\ge 0\) 并考虑 \(q\)-倾斜的有序分解测度
\[
\mathbb{P}^{(q)}_L(\boldsymbol{\ell})
:=\frac{1}{Z_L^{(q)}}\prod_{i}F_{\ell_i+2}^{\,q},
\qquad
Z_L^{(q)}:=\sum_{\boldsymbol{\ell}\in\mathcal C_L}\prod_i F_{\ell_i+2}^{\,q},
\]
其中 \(\mathcal C_L\) 为长度和为 \(L\) 的有序分解集合（定义见本节上文）。
令
\[
A_L(\boldsymbol{\ell})
:=\#\{i:\ \ell_i\equiv 1\ (\mathrm{mod}\ 3)\}.
\]
则存在常数 \(C(q)>0\) 与 \(\eta(q)\in(0,1)\) 使得当 \(L\to\infty\) 时有指数界
\[
\mathbb{P}^{(q)}_L\!\bigl(A_L\le 1\bigr)\ \le\ C(q)\,\eta(q)^L.
\]
\leanverified{paper\_pom\_multiplicity\_composition\_mod3\_sparsity}
\end{theorem}
\begin{proof}
对 \(u\in[0,1]\) 定义一参扰动的更新母函数
\[
A_{q}(t;u):=\sum_{k\ge 1}F_{k+2}^{\,q}\,u^{\mathbf 1_{\{k\equiv 1\ (\mathrm{mod}\ 3)\}}}\,t^k,
\qquad |t|<\varphi^{-q},
\]
并定义二元生成函数
\[
\mathcal Z^{(q)}(t;u):=\sum_{L\ge 0}\left(\sum_{\boldsymbol{\ell}\in\mathcal C_L}\prod_i F_{\ell_i+2}^{\,q}\,u^{A_L(\boldsymbol{\ell})}\right)t^L.
\]
由有序分解的串接结构，
\[
\mathcal Z^{(q)}(t;u)=\sum_{r\ge 0}A_q(t;u)^r=\frac{1}{1-A_q(t;u)}.
\]
对每个固定 \(u\in[0,1]\)，函数 \(t\mapsto A_q(t;u)\) 在 \((0,\varphi^{-q})\) 上严格递增且系数非负；
因此存在唯一 \(\rho(q;u)\in(0,\varphi^{-q})\) 使 \(A_q(\rho(q;u);u)=1\)，并记 \(\lambda(q;u):=\rho(q;u)^{-1}\)。
\par
当 \(u<1\) 时，\(A_q(t;u)\) 由 \(A_q(t;1)\) 去掉一组严格正的同余类项并作缩放得到，故对每个 \(t>0\) 有 \(A_q(t;u)<A_q(t;1)\)；
于是 \(\rho(q;u)>\rho(q;1)\)，等价于 \(\lambda(q;u)<\lambda(q;1)\)。
\par
由命题 \ref{prop:pom-multiplicity-composition-sharp-main-term-constant} 的简单极点系数传递，对每个固定 \(u\in[0,1]\) 都有
\[
[t^L]\mathcal Z^{(q)}(t;u)=\Theta\!\bigl(\lambda(q;u)^L\bigr),
\qquad L\to\infty,
\]
并且常数可在 \(u\) 的紧子区间上一致控制。
因此
\[
\mathbb{E}_{\mathbb P^{(q)}_L}\!\bigl[u^{A_L}\bigr]
=\frac{[t^L]\mathcal Z^{(q)}(t;u)}{[t^L]\mathcal Z^{(q)}(t;1)}
=O\!\left(\left(\frac{\lambda(q;u)}{\lambda(q;1)}\right)^{\!L}\right).
\]
取任意固定 \(u\in(0,1)\)，由 Markov 不等式
\[
\mathbb{P}^{(q)}_L(A_L\le 1)
\le u^{-1}\,\mathbb{E}[u^{A_L}]
\le C(q)\,u^{-1}\left(\frac{\lambda(q;u)}{\lambda(q;1)}\right)^{\!L}.
\]
令 \(\eta(q):=\lambda(q;u)/\lambda(q;1)\in(0,1)\) 即得所示指数界。证毕。
\end{proof}

\begin{corollary}[Witten 指数非零与取向负号的指数稀疏性（分解模型）]\label{cor:pom-multiplicity-composition-witten-sign-exp-sparse}
在定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity} 的设定下取 \(q=1\)，并令 \(\boldsymbol{\ell}\sim\mathbb P^{(1)}_L\)。
则：
\begin{enumerate}
  \item 若以 \(\boldsymbol{\ell}\) 的路径分量视为某纤维分解 \(\Gamma_x\cong\bigsqcup_i P_{\ell_i}\)，则纤维带符号指数 \(s_m(x)=Z_x(-1)\) 的非零事件满足
  \[
  \mathbb P^{(1)}_L\!\bigl(s_m(x)\neq 0\bigr)=\mathbb P^{(1)}_L(A_L=0)\ \le\ C\,\eta^L.
  \]
  \item 同样在上述识别下，扫描元与时间反演对合的负号事件满足
  \[
  \mathbb P^{(1)}_L\!\bigl(\operatorname{sgn}(c_x)=-1\bigr)\ \le\ C\,\eta^L,
  \qquad
  \mathbb P^{(1)}_L\!\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)\ \le\ C\,\eta^L.
\]
\end{enumerate}
其中常数 \(C>0,\eta\in(0,1)\) 与 \(L\) 无关。
\leanverified{paper\_pom\_multiplicity\_composition\_witten\_sign\_exp\_sparse}
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-fiber-rewrite-parity-exact-split-mod3-obstruction}，
\(
s_m(x)\neq 0
\)
当且仅当不存在 \(\ell_i\equiv 1\ (\mathrm{mod}\ 3)\)，即 \(A_L=0\)。
由定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity} 得到指数上界。
\par
对符号事件：由定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 与定理 \ref{thm:pom-toggle-time-reversal-sign-general-fiber}，
一旦 \(|A|\ge 2\)（即 \(A_L\ge 2\)），必有 \(\operatorname{sgn}(c_x)=\operatorname{sgn}(\iota_x)=+1\)。
因此 \(\operatorname{sgn}(c_x)=-1\) 或 \(\operatorname{sgn}(\iota_x)=-1\) 均蕴含 \(A_L\le 1\)，再用定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity} 即得结论。证毕。
\end{proof}

\begin{conjecture}[\texorpdfstring{$\chi$}{chi}-异常的速率存在性与自由能差刻画]\label{conj:pom-chi-anom-rate-free-energy-gap}
令 \(\pi_m\) 表示均匀微态输入在稳定类型层的推前分布
\(
\pi_m(x)=d_m(x)/2^m
\)
（见定义 \ref{def:fold-pushforward-renyi-collision-moment} 的记号）。
记随机稳定类型 \(X\sim \pi_m\)。
猜想存在常数 \(I_c,I_\iota>0\) 使得极限
\[
\lim_{m\to\infty}\frac1m\log \mathbb P_{\pi_m}\bigl(\operatorname{sgn}(c_X)=-1\bigr)=-I_c,
\qquad
\lim_{m\to\infty}\frac1m\log \mathbb P_{\pi_m}\bigl(\operatorname{sgn}(\iota_X)=-1\bigr)=-I_\iota
\]
存在，并且二者由“禁止 \(\ell\equiv 1\ (\mathrm{mod}\ 3)\) 分量”的受限配分函数与全配分函数之间的压力差给出：
若记 \(u\mapsto \lambda(1;u)\) 为本节证明中由 \(A_1(t;u)=1\) 定义的主尺度，则
\[
I_c=I_\iota=\log\!\left(\frac{\lambda(1;1)}{\lambda(1;0)}\right).
\]
\end{conjecture}
