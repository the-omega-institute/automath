\subsubsection{黄金偏置的二阶刻画与散度谱：\texorpdfstring{$\mathrm{TV}$}{TV}/\texorpdfstring{$\chi^2$}{chi2}/Rényi 的刚性塌缩}

\begin{proposition}[黄金偏置的“方差=偏置”刻画与唯一性]\label{prop:pom-golden-bias-variance-equals-bias}
令 \(p\in(1/2,1)\)，\(q:=1-p\)。
则方程
\[
pq=p-q
\]
在区间 \((1/2,1)\) 内存在唯一解，且该解为
\[
\boxed{\ p=\varphi^{-1},\qquad q=\varphi^{-2}\ }.
\]
等价地，若 \(B\sim\mathrm{Bern}(p)\)，则
\[
\mathrm{Var}(B)=\mathbb{E}(2B-1)
\quad\Longleftrightarrow\quad
p=\varphi^{-1}.
\]
该刻画在不引入 Fibonacci 递推的前提下，把”黄金偏置”从二元分裂的二阶波动（方差）与一阶不平衡（偏置）之间的代数自洽性中唯一析出。
\leanverified{variance\_bias\_quadratic}
\leanverified{quadratic\_discriminant}
\leanverified{positive\_root\_in\_interval}
\leanverified{paper\_pom\_golden\_bias\_variance\_equals\_bias}
\end{proposition}

\begin{proof}
注意 \(\mathrm{Var}(B)=pq\) 且 \(\mathbb{E}(2B-1)=p-q=2p-1\)。
故条件 \(pq=p-q\) 等价于
\(
p(1-p)=2p-1
\)，即
\(
p^2+p-1=0
\)。
其两根为 \(p=\frac{-1\pm\sqrt5}{2}\)，其中落在 \((1/2,1)\) 的唯一根为 \(p=\frac{\sqrt5-1}{2}=\varphi^{-1}\)。
从而 \(q=1-p=\varphi^{-2}\)。证毕。
\end{proof}

\begin{proposition}[互补黄金 Bernoulli 对的散度塌缩与闭式线性化]\label{prop:pom-complementary-golden-bernoulli-divergence-collapse}
取互补对
\[
P:=\mathrm{Bern}(\varphi^{-1}),\qquad
Q:=\mathrm{Bern}(\varphi^{-2})\quad(\varphi^{-2}=1-\varphi^{-1}).
\]
则常用散度出现同值塌缩与闭式线性化：
\[
\boxed{\;
\mathrm{TV}(P,Q)=|\varphi^{-1}-\varphi^{-2}|=\varphi^{-3},\qquad
\chi^2(P\|Q)=\varphi^{-3}\;}
\]
\[
\boxed{\;
D(P\|Q)=\varphi^{-3}\log_2\varphi,\qquad
H^2(P,Q)=1-2\sqrt{\varphi^{-3}}=1-2\varphi^{-3/2}.\;}
\]
因此该互补对在”可分性尺度”上呈现单参数刚性：上述量均由 \(\varphi\) 的同一幂级控制。
\leanverified{paper\_pom\_complementary\_golden\_bernoulli\_divergence\_collapse}
\end{proposition}

\begin{proof}
对二点分布，\(\mathrm{TV}(P,Q)=\frac12\sum_x|P(x)-Q(x)|=|p-q|\)，取 \(p=\varphi^{-1},q=\varphi^{-2}\) 得 \(\mathrm{TV}=\varphi^{-3}\)。
\par
对 \(\chi^2(P\|Q)=\sum_x \frac{(P(x)-Q(x))^2}{Q(x)}\)，有
\[
\chi^2(P\|Q)=(p-q)^2\Bigl(\frac{1}{q}+\frac{1}{p}\Bigr)=\frac{(p-q)^2}{pq}.
\]
而 \(pq=\varphi^{-3}\)、\(p-q=\varphi^{-3}\)，故 \(\chi^2=\varphi^{-3}\)。
\par
KL 散度为
\[
D(P\|Q)=p\log_2\frac{p}{q}+q\log_2\frac{q}{p}=(p-q)\log_2\frac{p}{q}.
\]
此处 \(p/q=\varphi\)，且 \(p-q=\varphi^{-3}\)，故 \(D(P\|Q)=\varphi^{-3}\log_2\varphi\)。
\par
Hellinger 平方距离为
\(
H^2(P,Q)=1-\sum_x\sqrt{P(x)Q(x)}=1-(\sqrt{pq}+\sqrt{qp})=1-2\sqrt{pq}=1-2\varphi^{-3/2}
\)。
证毕。
\end{proof}

\begin{proposition}[Rényi 散度族的全阶闭式与端点谱]\label{prop:pom-complementary-golden-bernoulli-renyi-spectrum}
对任意 \(\alpha>0\)、\(\alpha\neq 1\)，仍令
\(
P=\mathrm{Bern}(\varphi^{-1}),\ Q=\mathrm{Bern}(\varphi^{-2})
\)。
则 Rényi 散度（bits）满足显式闭式
\[
\boxed{\;
D_\alpha(P\|Q)
=\frac{1}{\alpha-1}\log_2\!\Big(\varphi^{\alpha-2}+\varphi^{-\alpha-1}\Big)
=\frac{1}{\alpha-1}\log_2\!\Big(\varphi^{-\alpha-1}(1+\varphi^{2\alpha-1})\Big).\;}
\]
并且端点值均为闭式：
\[
\boxed{\;
D_{1/2}(P\|Q)=3\log_2\varphi-2,\quad
D_{2}(P\|Q)=\log_2(1+\varphi^{-3}),\quad
D_{\infty}(P\|Q)=\log_2\varphi.\;}
\]
从而互补黄金对的“全阶可分性谱”在解析上由单一代数数 \(\varphi\) 完全控制。
\leanverified{paper\_pom\_complementary\_golden\_bernoulli\_renyi\_spectrum}
\end{proposition}

\begin{proof}
Rényi 散度定义给出
\[
D_\alpha(P\|Q)=\frac{1}{\alpha-1}\log_2\!\Big(p^\alpha q^{1-\alpha}+q^\alpha p^{1-\alpha}\Big),
\qquad p=\varphi^{-1},\ q=\varphi^{-2}.
\]
直接化简指数得到
\(
p^\alpha q^{1-\alpha}=\varphi^{\alpha-2}
\)、
\(
q^\alpha p^{1-\alpha}=\varphi^{-\alpha-1}
\)，
即得全阶闭式。
端点代入 \(\alpha=\tfrac12,2\) 立得；而 \(D_\infty(P\|Q)=\log_2\max_x \frac{P(x)}{Q(x)}=\log_2(p/q)=\log_2\varphi\)。证毕。
\end{proof}

\subsubsection{后验离散化与有限窗审计：\texorpdfstring{$\varphi$}{phi}-logistic、显式误差与 hitting-time}

\begin{proposition}[后验在 \texorpdfstring{$\varphi$}{phi}-幂网格上的离散化：\texorpdfstring{$\varphi$}{phi}-logistic]\label{prop:pom-phi-logistic-posterior-grid}
\leanverified{paper\_pom\_phi\_logistic\_posterior\_grid}
考虑二相位假设 \(\Theta\in\{\theta_+,\theta_-\}\) 等先验，且在 \(\theta_+\) 下
\(
B_i\stackrel{\mathrm{iid}}{\sim}\mathrm{Bern}(\varphi^{-1})
\)，
在 \(\theta_-\) 下
\(
B_i\stackrel{\mathrm{iid}}{\sim}\mathrm{Bern}(\varphi^{-2})
\)。
令 \(S_n:=\sum_{i=1}^{n}B_i\)，并令整数
\[
K_n:=2S_n-n\in\{-n,-n+2,\dots,n\}.
\]
则后验赔率满足严格幂律
\[
\boxed{\;
\frac{\mathbb{P}(\theta_+\mid B_1^n)}{\mathbb{P}(\theta_-\mid B_1^n)}=\varphi^{K_n}\;}
\]
从而后验概率为离散的 \(\varphi\)-logistic 族：
\[
\boxed{\;
\mathbb{P}(\theta_+\mid B_1^n)=\frac{1}{1+\varphi^{-K_n}},\qquad
\mathbb{P}(\theta_-\mid B_1^n)=\frac{1}{1+\varphi^{K_n}}.\;}
\]
因此“可实现后验集合”精确等于
\[
\Bigl\{(1+\varphi^{-k})^{-1}:\ k\equiv n\ (\mathrm{mod}\ 2),\ |k|\le n\Bigr\},
\]
表现为 \(\varphi\)-幂网格上的可审计离散化，而非连续流形。
\end{proposition}

\begin{proof}
由 Bayes 公式与独立性，
\[
\frac{\mathbb{P}(\theta_+\mid B_1^n)}{\mathbb{P}(\theta_-\mid B_1^n)}
=\frac{\mathbb{P}(B_1^n\mid\theta_+)}{\mathbb{P}(B_1^n\mid\theta_-)}
=\left(\frac{p}{1-p}\right)^{2S_n-n},
\qquad p=\varphi^{-1}.
\]
而 \(\frac{p}{1-p}=\frac{\varphi^{-1}}{\varphi^{-2}}=\varphi\)，故赔率为 \(\varphi^{K_n}\)，再化为 logistic 即得。
可实现集合由 \(K_n\) 的取值集合直接给出。证毕。
\end{proof}

\begin{theorem}[有限样本最优误差的显式公式与锐渐近前因子]\label{thm:pom-bayes-error-explicit-sharp-asymptotic}
\leanverified{paper\_pom\_bayes\_error\_explicit\_sharp\_asymptotic}
在命题 \ref{prop:pom-phi-logistic-posterior-grid} 的二相位模型中，等先验下最优 Bayes 判别为多数表决（命题 \ref{prop:pom-max-fiber-even-majority-vote} 的 \(\varphi\)-极限口径）。
其最优误差概率满足完全显式求和式
\[
\boxed{\;
P_e(n)=\sum_{j=0}^{\lfloor n/2\rfloor}\binom{n}{j}\,\varphi^{-j}\,\varphi^{-2(n-j)}
=\sum_{j=0}^{\lfloor n/2\rfloor}\binom{n}{j}\,\varphi^{-2n+j}.\;}
\]
并且当 \(n\to\infty\) 时存在锐渐近（含显式前因子）
\[
\boxed{\;
P_e(n)=\varphi^{2}\sqrt{\frac{2}{\pi n}}\,(2\varphi^{-3/2})^{\,n}\,(1+o(1)).\;}
\]
该式把二相位识别的有限窗审计从指数率提升到可直接数值核验的精度层级。
\end{theorem}

\begin{proof}
在 \(\theta_+\) 下，误判事件为 \(S_n\le \lfloor n/2\rfloor\)；
由对称互补性（把 \(\theta_+\) 与 \(\theta_-\) 互换等价于把 \(B\) 取反），两相位下的误差概率相同，故 Bayes 误差即该二项尾和。
取 \(p=\varphi^{-1}\)、\(q=\varphi^{-2}\) 得到显式求和式。
\par
为得锐渐近，设 \(n\) 偶并令边界项
\[
T_0:=\binom{n}{n/2}(pq)^{n/2}=\binom{n}{n/2}\varphi^{-3n/2}.
\]
令 \(T_r:=\binom{n}{n/2-r}p^{n/2-r}q^{n/2+r}\)（\(r\ge 0\)），则
\[
\frac{T_{r+1}}{T_r}
=\frac{n/2-r}{n/2+r+1}\cdot\frac{q}{p}
\le \frac{q}{p}=\varphi^{-1},
\]
并且对每个固定 \(r\)，当 \(n\to\infty\) 时该比值的乘积修正趋于 \(1\)。
因此
\[
\frac{P_e(n)}{T_0}
=\sum_{r\ge 0}\frac{T_r}{T_0}
\xrightarrow[n\to\infty]{}
\sum_{r\ge 0}\left(\frac{q}{p}\right)^r
=\frac{1}{1-q/p}
=\frac{p}{p-q}
=\varphi^2,
\]
其中极限交换由支配收敛定理保证（\(\frac{T_r}{T_0}\le (\frac{q}{p})^r\)）。
\par
最后由中心二项式系数渐近
\(
\binom{n}{n/2}\sim 2^n\sqrt{\frac{2}{\pi n}}
\)
得到
\(
T_0\sim \sqrt{\frac{2}{\pi n}}(2\varphi^{-3/2})^{n}
\)，
合并即得所示前因子。
奇数 \(n\) 的情形仅需把边界项改为 \(j=\lfloor n/2\rfloor\)，同理可得同一渐近。证毕。
\end{proof}

\begin{proposition}[置信阈值达到时间的 \texorpdfstring{$\varphi$}{phi}-阈值律与期望样本量闭式]\label{prop:pom-confidence-hitting-time-phi-threshold}
在命题 \ref{prop:pom-phi-logistic-posterior-grid} 的设定下，定义后验误差
\[
\varepsilon_n:=\min\{\mathbb{P}(\theta_+\mid B_1^n),\ \mathbb{P}(\theta_-\mid B_1^n)\}
=\frac{1}{1+\varphi^{|K_n|}}.
\]
给定目标 \(\epsilon\in(0,1/2)\)，令停止时刻
\[
\tau_\epsilon:=\inf\{n\ge 1:\ \varepsilon_n\le \epsilon\}.
\]
则停止条件等价于整数阈值
\[
\boxed{\;
|K_n|\ge T_\epsilon,\qquad
T_\epsilon:=\left\lceil\log_\varphi\!\Big(\frac{1-\epsilon}{\epsilon}\Big)\right\rceil.\;}
\]
并且在真相位为 \(\theta_+\)（或 \(\theta_-\)）时，\(K_n=\sum_{i=1}^n(2B_i-1)\) 为具有漂移
\[
\mathbb{E}[K_n]=(\varphi^{-1}-\varphi^{-2})n=\varphi^{-3}n
\]
的 \(\pm 1\) 随机游走，从而当 \(\epsilon\downarrow 0\)（等价于 \(T_\epsilon\to\infty\)）时，
\[
\boxed{\;
\mathbb{E}[\tau_\epsilon]=\varphi^{3}\,T_\epsilon+O(1)
=\frac{\varphi^{3}}{\log\varphi}\log\!\Big(\frac{1-\epsilon}{\epsilon}\Big)+O(1).\;}
\]
\end{proposition}

\begin{proof}
由命题 \ref{prop:pom-phi-logistic-posterior-grid}，\(\varepsilon_n=\min\{(1+\varphi^{-K_n})^{-1},(1+\varphi^{K_n})^{-1}\}=(1+\varphi^{|K_n|})^{-1}\)。
因此 \(\varepsilon_n\le \epsilon\) 等价于 \(\varphi^{|K_n|}\ge (1-\epsilon)/\epsilon\)，即 \(|K_n|\ge T_\epsilon\)。
\par
在 \(\theta_+\) 下，增量 \(X_i:=2B_i-1\in\{\pm 1\}\) 独立同分布且
\(
\mathbb{E}[X_i]=p-q=\varphi^{-3}
\)。
令 \(\tau:=\tau_\epsilon\)。
由于 \(|K_\tau|\in[T_\epsilon,T_\epsilon+1]\)（步长为 \(1\)），并且 \(\mathbb{P}(K_\tau\le -T_\epsilon)\) 以 \((q/p)^{T_\epsilon}=\varphi^{-T_\epsilon}\) 指数衰减，
可得 \(\mathbb{E}[K_\tau]=T_\epsilon+O(1)\)。
对鞅 \(M_n:=K_n-\varphi^{-3}n\) 应用可选停并取期望，得到
\(
\mathbb{E}[K_\tau]=\varphi^{-3}\mathbb{E}[\tau]
\)，
从而 \(\mathbb{E}[\tau]=\varphi^{3}T_\epsilon+O(1)\)。
最后用
\(
T_\epsilon=\frac{1}{\log\varphi}\log\bigl(\frac{1-\epsilon}{\epsilon}\bigr)+O(1)
\)
改写即得。证毕。
\end{proof}

\subsubsection{几何--统计桥接与顶端谱刚性：\texorpdfstring{$C_\varphi$}{Cphi} 的同一化}

\begin{corollary}[最大/平均纤维浓缩指数与二相位可分性常数的同一化]\label{cor:pom-D-over-avg-exponent-equals-chernoff}
\leanverified{paper\_pom\_d\_over\_avg\_exponent\_equals\_chernoff}
令 \(D_m\) 为最大纤维多重度（定义 \ref{def:pom-top-fiber-spectrum}），并令平均纤维多重度为
\[
\overline d_m:=\frac{|\Omega_m|}{|X_m|}=\frac{2^m}{F_{m+2}}.
\]
记二相位识别的 Chernoff 常数（bits）为
\(
C_\varphi=\frac32\log_2\varphi-1
\)
（定理 \ref{thm:pom-max-fiber-even-chernoff-exponent}）。
则在稳定尺度上有严格同一化：
\[
\boxed{\;
\lim_{m\to\infty}\frac{1}{m}\log_2\frac{D_m}{\overline d_m}=C_\varphi.\;}
\]
因此“几何侧的最坏/平均浓缩指数”与“统计侧的最优可分性指数”共享同一代数常数。
\end{corollary}

\begin{proof}
由推论 \ref{cor:pom-D-rec} 的闭式与 \(F_{m+2}\sim \varphi^{m}/\sqrt5\)，可得
\(
\frac{D_m}{\overline d_m}
=\Theta((\varphi^{3/2}/2)^m)
\)
（亦见注 \ref{rem:pom-defect-gap} 的自然对数版本），从而
\[
\lim_{m\to\infty}\frac{1}{m}\log_2\frac{D_m}{\overline d_m}
=\log_2\!\left(\frac{\varphi^{3/2}}{2}\right)
=\frac32\log_2\varphi-1
=C_\varphi.
\]
证毕。
\end{proof}

\begin{corollary}[顶端谱缺口的奇偶极限常数]\label{cor:pom-top-gap-limit-constants}
设 \(D_m^{(2)}\) 为第二大纤维多重度（定义 \ref{def:pom-top-fiber-spectrum}），并定义顶端相对缺口
\[
G_m:=1-\frac{D_m^{(2)}}{D_m}.
\]
则有奇偶极限常数（推论 \ref{cor:pom-second-max-gap-constant}）：
\[
\boxed{\;
\lim_{k\to\infty}G_{2k}=\varphi^{-6},\qquad
\lim_{k\to\infty}G_{2k+1}=\frac{1}{2}\varphi^{-5}.\;}
\]
\leanverified{paper\_pom\_top\_gap\_limit\_constants}
\end{corollary}

\begin{proof}
由推论 \ref{cor:pom-second-max-gap-constant} 已得 \(D_m^{(2)}/D_m\) 的奇偶极限（并给出指数收敛），直接改写即得所示缺口极限常数。证毕。
\end{proof}

\begin{proposition}[奇数四相在 \(q\)-矩加权下的有效塌缩阈值：显式选择律]\label{prop:pom-odd-fourphase-q-collapse-threshold}
在奇数稳定相 \(m=2k+1\ge 13\) 的四相结构中（注 \ref{rem:pom-max-fiber-achievers-bsplit}），设偏置两相在 \(b\)-分裂口径下的最大分支为
\[
A_k:=F_{k+1}+F_{k-2},
\]
平衡两相的最大分支为
\[
B_k:=F_{k+1},
\]
并记比值 \(R_k:=A_k/B_k\to 1+\varphi^{-3}\)。
对任意容差 \(\delta\in(0,1)\)，若以 \(q\)-矩权重比较两类相位贡献，则当
\[
\boxed{\;
q\ge q_\delta:=\left\lceil\frac{\log(1/\delta)}{\log(1+\varphi^{-3})}\right\rceil\;}
\]
时，偏置两相对该 \(q\)-矩的总贡献占比至少为 \(1-\delta\)。
从而“在最大达到者族内部由四相有效塌缩到两相”的现象可由单一闭式阈值 \(q_\delta\) 精确刻画。
\end{proposition}

\begin{proof}
在最大相位族内部，平衡两相与偏置两相各有两条。
若将相位贡献按“最大分支的 \(q\)-次幂”粗化，则偏置类占比为
\[
\frac{2A_k^q}{2A_k^q+2B_k^q}=\frac{1}{1+R_k^{-q}}.
\]
当 \(q\ge \frac{\log(1/\delta)}{\log R_k}\) 时有 \(R_k^{-q}\le \delta\)，从而占比 \(\ge \frac{1}{1+\delta}\ge 1-\delta\)。
再用 \(R_k\to 1+\varphi^{-3}\) 得到给定阈值的渐近闭式，并取上整得到 \(q_\delta\)。证毕。
\end{proof}

\begin{proposition}[最大纤维序列的谱分解闭式：四周期次主振荡]\label{prop:pom-D-spectral-decomposition-fourperiod}
最大纤维序列 \((D_m)\) 满足二步递推 \(D_m=D_{m-2}+D_{m-4}\)（推论 \ref{cor:pom-D-rec}；并取约定 \(D_0:=1,D_1:=2\)）。
其特征方程为 \(\lambda^4-\lambda^2-1=0\)，根为 \(\pm\sqrt{\varphi}\) 与 \(\pm \mathrm{i}/\sqrt{\varphi}\)。
因此存在实常数 \(a,b,c,d\) 使得对所有 \(m\ge 0\) 都有谱分解闭式
\[
\boxed{\;
D_m=a(\sqrt{\varphi})^{m}+b(-\sqrt{\varphi})^{m}
\;+\;\varphi^{-m/2}\Bigl(c\cos\!\Bigl(\frac{\pi m}{2}\Bigr)+d\sin\!\Bigl(\frac{\pi m}{2}\Bigr)\Bigr).\;}
\]
其中第三项给出严格四周期的次主振荡，且振幅按 \(\varphi^{-m/2}\) 指数衰减。
\end{proposition}

\begin{proof}
由递推的通解结构，存在常数 \(A_+,A_-,B_+,B_-\in\CC\) 使
\[
D_m=A_+(\sqrt{\varphi})^m+A_-(-\sqrt{\varphi})^m
+B_+\left(\frac{\mathrm{i}}{\sqrt{\varphi}}\right)^m
+B_-\left(\frac{-\mathrm{i}}{\sqrt{\varphi}}\right)^m.
\]
由于 \(D_m\in\RR\)，复根项必须成共轭对；将其改写为余弦/正弦组合即得
\[
B_+\left(\frac{\mathrm{i}}{\sqrt{\varphi}}\right)^m
+\overline{B_+}\left(\frac{-\mathrm{i}}{\sqrt{\varphi}}\right)^m
=\varphi^{-m/2}\Bigl(c\cos(\tfrac{\pi m}{2})+d\sin(\tfrac{\pi m}{2})\Bigr),
\]
其中 \(c,d\in\RR\)。
再将 \(A_\pm\) 重命名为 \(a,b\) 即得结论。证毕。
\end{proof}
