\begin{corollary}[纤维 Walsh 模式的硬核表示与全奇偶系数]\label{cor:pom-fiber-walsh-hardcore-stokes-sign}
\leanverified{paper\_pom\_fiber\_walsh\_hardcore\_stokes\_sign}
沿用定理 \ref{thm:pom-fiber-multivariate-holographic-conservation} 的记号。
对任意 \(I\subseteq[m]\) 定义符号向量 \(\varepsilon^I\in\{\pm1\}^m\) 为
\(
\varepsilon^I_r=-1\iff r\in I
\)，并令 Walsh 字符
\[
\chi_I(\omega):=\prod_{r=1}^{m}(\varepsilon^I_r)^{\omega_r}=(-1)^{\sum_{r\in I}\omega_r}\qquad(\omega\in\Omega_m).
\]
则对任意 \(x\in X_m\) 有严格恒等式
\[
\sum_{\omega\in \Fold_m^{-1}(x)}\chi_I(\omega)=\mathcal H_x(\varepsilon^I)
=(\varepsilon^I)^x\sum_{S\in\mathsf{Ind}(\Gamma_x)}\ \prod_{v\in S}\ \prod_{r\in J_x(v)}\varepsilon^I_r,
\]
其中 \(J_x(v)\) 为定理 \ref{thm:pom-fiber-multivariate-holographic-conservation} 中的长度 \(3\) 局部区间。
\par
特别地，当 \(I=[m]\) 时，
\[
\sum_{\omega\in \Fold_m^{-1}(x)}(-1)^{|\omega|_1}=(-1)^{|x|_1}\,Z_x(-1)=(-1)^{|x|_1}\,s(x),
\]
其中 \(s(x):=Z_x(-1)\) 见推论 \ref{cor:pom-fiber-zx-minus-one-psi3-sign}。
并且对任意非空 \(I\)，上述 Walsh 求和作为 \(\Omega_m\) 上子集 \(\Fold_m^{-1}(x)\) 的 Walsh 系数，可由其有向边界通量表示（定理 \ref{thm:xi-hypercube-fourier-walsh-boundary-parseval}）。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-fiber-multivariate-holographic-conservation} 的证明，
\(
\mathcal H_x(\boldsymbol z)=\sum_{\omega\in\Fold_m^{-1}(x)}\prod_{r=1}^m z_r^{\omega_r}
\)。
取 \(\boldsymbol z=\varepsilon^I\) 立得
\[
\mathcal H_x(\varepsilon^I)=\sum_{\omega\in\Fold_m^{-1}(x)}\prod_{r=1}^{m}(\varepsilon^I_r)^{\omega_r}
=\sum_{\omega\in\Fold_m^{-1}(x)}\chi_I(\omega).
\]
另一方面，对 \(\varepsilon^I_r\in\{\pm 1\}\) 有 \((\varepsilon^I_r)^{-1}=\varepsilon^I_r\)，故对任意 \(v\in V(\Gamma_x)\),
\[
w_{x,v}(\varepsilon^I)
=\frac{\varepsilon^I_{j_x(v)+1}\varepsilon^I_{j_x(v)+2}}{\varepsilon^I_{j_x(v)}}
=\varepsilon^I_{j_x(v)}\varepsilon^I_{j_x(v)+1}\varepsilon^I_{j_x(v)+2}
=\prod_{r\in J_x(v)}\varepsilon^I_r,
\]
代回 \(\mathcal H_x\) 的 hard-core 展开即得第一组等式。
\par
当 \(I=[m]\) 时 \(w_{x,v}(\varepsilon^{[m]})=-1\)，从而
\(
\mathcal H_x(\varepsilon^{[m]})=(-1)^{|x|_1}\sum_{S\in\mathsf{Ind}(\Gamma_x)}(-1)^{|S|}
=(-1)^{|x|_1}Z_x(-1)
\)。
并且由定理 \ref{thm:pom-fiber-biased-bernoulli-pushforward-hardcore} 的证明中恒等式 \(|\omega|_1=|x|_1+r(\omega)\)，亦可直接得到
\(
(-1)^{|\omega|_1}=(-1)^{|x|_1}(-1)^{r(\omega)}
\)
并在纤维上求和得到同一结论。
最后一条由定理 \ref{thm:xi-hypercube-fourier-walsh-boundary-parseval}（取 \(B=\Fold_m^{-1}(x)\)）直接推出。证毕。
\end{proof}
