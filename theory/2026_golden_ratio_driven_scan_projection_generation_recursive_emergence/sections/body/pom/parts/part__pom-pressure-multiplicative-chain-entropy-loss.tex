\paragraph{乘法链式律与熵亏损：escort R\'enyi 熵率的可逆读出}
对整数 \(b\ge 2\)，对任意分布 \(\pi\in\Delta(X_m)\) 定义 R\'enyi-\(b\) 熵
\[
H_b(\pi):=\frac{1}{1-b}\log\!\left(\sum_{x\in X_m}\pi(x)^b\right).
\]
对整数 \(a\ge 1\)，记 \(\pi_{m,a}:=w_{m,a}\) 为整数 escort 分布。

\begin{theorem}[乘法链式恒等式：有限 \(m\) 的 escort--R\'enyi 分解]\label{thm:pom-multiplicative-chain-rule-finite-m}
对任意整数 \(a\ge 1\)、\(b\ge 2\) 与任意分辨率 \(m\)，成立严格恒等式
\[
\boxed{\;
\log S_{ab}(m)
=
b\,\log S_a(m)-(b-1)\,H_b(\pi_{m,a})\; }.
\]
等价地，
\[
\boxed{\;
S_{ab}(m)=S_a(m)^b\cdot \sum_{x\in X_m}\pi_{m,a}(x)^b\; }.
\]
\leanverified{paper\_pom\_multiplicative\_chain\_rule\_finite\_m\_seeds}
\leanverified{paper\_pom\_multiplicative\_chain\_rule\_finite\_m\_package}
\end{theorem}
\begin{proof}
由 \(\pi_{m,a}(x)=d_m(x)^a/S_a(m)\) 得
\[
\sum_{x\in X_m}\pi_{m,a}(x)^b
=\frac{1}{S_a(m)^b}\sum_{x\in X_m} d_m(x)^{ab}
=\frac{S_{ab}(m)}{S_a(m)^b}.
\]
取对数并用 \(H_b\) 的定义改写即得。证毕。
\end{proof}

\begin{theorem}[极限层的压力--熵分解：乘法尺度上的精确分解等式]\label{thm:pom-pressure-entropy-factorization-multiplicative}
\leanverified{paper\_pom\_pressure\_entropy\_factorization\_multiplicative}
假设对所有整数 \(q\ge 1\) 极限压力
\[
P_q:=\lim_{m\to\infty}\frac{1}{m}\log S_q(m)
\]
存在。
并对任意整数 \(a\ge 1\)、\(b\ge 2\) 假设极限熵率
\[
h^{\mathrm{esc}}_b(a):=\lim_{m\to\infty}\frac{1}{m}H_b(\pi_{m,a})
\]
存在。
则成立乘法尺度分解：
\[
\boxed{\;
P_{ab}=b\,P_a-(b-1)\,h^{\mathrm{esc}}_b(a)\; }.
\]
\end{theorem}
\begin{proof}
将定理 \ref{thm:pom-multiplicative-chain-rule-finite-m} 两边除以 \(m\) 并令 \(m\to\infty\) 即得。证毕。
\end{proof}

\begin{corollary}[可观测性：整数压力谱完全决定全部 escort R\'enyi 熵率]\label{cor:pom-escort-renyi-rate-inversion}
在定理 \ref{thm:pom-pressure-entropy-factorization-multiplicative} 的设定下，对任意整数 \(a\ge 1\)、\(b\ge 2\) 有显式反演公式
\[
\boxed{\;
h^{\mathrm{esc}}_b(a)=\frac{bP_a-P_{ab}}{b-1}\; }.
\]
特别地，由 \(S_1(m)=2^m\) 得 \(P_1=\log 2\)，因此基准分布 \(w_m=d_m/2^m\) 的 R\'enyi-\(b\) 熵率为
\[
\boxed{\;
h^{\mathrm{esc}}_b(1)=\frac{b\log 2-P_{b}}{b-1}\; }.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-pressure-entropy-factorization-multiplicative} 直接移项即得；第二式取 \(a=1\) 并用 \(P_1=\log 2\)。证毕。
\end{proof}

\begin{definition}[熵亏损函数（乘法半群上的 \(1\)-cocycle）]\label{def:pom-entropy-loss-cocycle}
在极限压力谱存在时，对整数 \(a\ge 1\)、\(b\ge 2\) 定义熵亏损
\[
\boxed{\;
\mathcal{H}_{\mathrm{loss}}(a,b):=bP_a-P_{ab}\; }.
\]
若同时 \(h^{\mathrm{esc}}_b(a)\) 存在，则
\(
\mathcal{H}_{\mathrm{loss}}(a,b)=(b-1)\,h^{\mathrm{esc}}_b(a)
\)。
\end{definition}

\begin{theorem}[严格 \(1\)-cocycle 方程与平直性：熵亏损的链式分解]\label{thm:pom-entropy-loss-1cocycle-flat}
在定义 \ref{def:pom-entropy-loss-cocycle} 的设定下，对任意整数 \(a\ge 1\)、\(b,c\ge 2\) 有
\[
\boxed{\;
\mathcal{H}_{\mathrm{loss}}(a,bc)=c\,\mathcal{H}_{\mathrm{loss}}(a,b)+\mathcal{H}_{\mathrm{loss}}(ab,c)\; }.
\]
等价地，若相应熵率存在，则
\[
(bc-1)\,h^{\mathrm{esc}}_{bc}(a)=c(b-1)\,h^{\mathrm{esc}}_{b}(a)+(c-1)\,h^{\mathrm{esc}}_{c}(ab).
\]
\end{theorem}
\begin{proof}
由定义，
\[
\mathcal{H}_{\mathrm{loss}}(a,bc)
=bc\,P_a-P_{abc}
=c(bP_a-P_{ab})+(cP_{ab}-P_{abc})
=c\,\mathcal{H}_{\mathrm{loss}}(a,b)+\mathcal{H}_{\mathrm{loss}}(ab,c).
\]
第二式由 \(\mathcal{H}_{\mathrm{loss}}=(\cdot-1)h^{\mathrm{esc}}\) 代入得到。证毕。
\end{proof}
\leanverified{paper\_pom\_entropy\_loss\_1cocycle\_flat}

\begin{corollary}[沿任意因子链的展开与交换诱导的平直性条件]\label{cor:pom-entropy-loss-factor-chain-expansion}
设 \(b=\prod_{j=1}^{k} r_j\) 为任意有序分解（\(r_j\ge 2\)）。
则
\[
\boxed{\;
\mathcal{H}_{\mathrm{loss}}(a,b)
=
\sum_{j=1}^{k}\left(\prod_{t=j+1}^{k} r_t\right)\,
\mathcal{H}_{\mathrm{loss}}\!\left(a\prod_{t=1}^{j-1}r_t,\ r_j\right)\; }.
\]
并且对任意 \(p,q\ge 2\)（由 \(pq=qp\)）必满足平直性恒等式
\[
\boxed{\;
q\,\mathcal{H}_{\mathrm{loss}}(a,p)+\mathcal{H}_{\mathrm{loss}}(ap,q)
=
p\,\mathcal{H}_{\mathrm{loss}}(a,q)+\mathcal{H}_{\mathrm{loss}}(aq,p)\; }.
\]
\end{corollary}
\begin{proof}
对分解 \(b=r_1\cdots r_k\) 迭代应用定理 \ref{thm:pom-entropy-loss-1cocycle-flat} 即得展开式。
对平直性恒等式，分别对 \(pq\) 与 \(qp\) 使用同一展开（两者相等）即得。证毕。
\end{proof}
\leanverified{paper\_pom\_entropy\_loss\_factor\_chain\_expansion}

\begin{theorem}[乘法细化单调性与几何望远镜：\texorpdfstring{$\alpha_\ast$}{alpha*} 的收敛与显式误差界]\label{thm:pom-multiplicative-refinement-telescope-alpha}
\leanverified{paper\_pom\_multiplicative\_refinement\_telescope\_alpha}
在极限压力谱存在并且最大纤维指数极限
\[
\alpha_\ast:=\lim_{m\to\infty}\frac{1}{m}\log M_m,
\qquad
M_m:=\max_{x\in X_m}d_m(x),
\]
存在时成立。并且在 \(\lim_{m\to\infty}\frac{1}{m}\log|X_m|<\infty\) 的指数生长口径下，由夹逼 \(M_m^q\le S_q(m)\le |X_m|M_m^q\) 可知亦有
\[
\alpha_\ast=\lim_{q\to\infty}\frac{P_q}{q}.
\]
固定整数 \(a\ge 1\)、\(b\ge 2\)，并定义乘法细化序列
\[
\alpha_{a,b}(n):=\frac{P_{ab^n}}{ab^n}\qquad(n\ge 0).
\]
则对每个 \(n\ge 0\) 有差分恒等式
\[
\boxed{\;
\alpha_{a,b}(n)-\alpha_{a,b}(n+1)
=\frac{\mathcal{H}_{\mathrm{loss}}(ab^n,b)}{ab^{n+1}}
=\frac{b-1}{ab^{n+1}}\,h^{\mathrm{esc}}_{b}(ab^n)\ \ge\ 0\; }.
\]
从而 \(\{\alpha_{a,b}(n)\}\) 单调不增。
并且对任意 \(N\ge 0\) 成立望远镜展开
\[
\boxed{\;
\alpha_{a,b}(N)
=
\alpha_\ast+\sum_{k=N}^{\infty}\frac{\mathcal{H}_{\mathrm{loss}}(ab^k,b)}{ab^{k+1}}\; }.
\]
若进一步 \(\lim_{m\to\infty}\frac{1}{m}\log|X_m|=\log\varphi\)（黄金均值稳定类型的指数率），则有可计算误差上界
\[
\boxed{\;
0\le \alpha_{a,b}(N)-\alpha_\ast\le \frac{\log\varphi}{ab^N}\; }.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-pressure-entropy-factorization-multiplicative}（取 \(a\leftarrow ab^n\)）得
\(
P_{ab^{n+1}}=bP_{ab^n}-\mathcal{H}_{\mathrm{loss}}(ab^n,b)
\)；
两边除以 \(ab^{n+1}\) 即得差分式与单调性。
对 \(n=N,N+1,\dots\) 求和并用 \(\alpha_{a,b}(n)\to \alpha_\ast\)（因为 \(ab^n\to\infty\)）即得望远镜展开。
\par
最后，\(0\le H_b(\pi)\le \log|X_m|\) 给出
\(
0\le h^{\mathrm{esc}}_b(\cdot)\le \log\varphi
\)，从而
\(
0\le \mathcal{H}_{\mathrm{loss}}(ab^k,b)\le (b-1)\log\varphi
\)。
代入望远镜展开并对几何级数求和得误差上界。证毕。
\end{proof}

\begin{theorem}[零温度恒等式：escort 的 min-entropy 率等于 \texorpdfstring{$P_a-a\alpha_\ast$}{Pa-a alpha*}]\label{thm:pom-escort-minentropy-zero-temperature-identity}
在极限压力谱存在且 \(\alpha_\ast\) 存在的前提下，令
\[
H_\infty(\pi):=-\log\!\left(\max_{x\in X_m}\pi(x)\right)
\]
为 min-entropy。
则对任意固定整数 \(a\ge 1\)，极限
\[
h_\infty^{\mathrm{esc}}(a):=\lim_{m\to\infty}\frac{1}{m}H_\infty(\pi_{m,a})
\]
存在且满足严格恒等式
\[
\boxed{\;
h_\infty^{\mathrm{esc}}(a)=P_a-a\alpha_\ast\; }.
\]
\end{theorem}
\begin{proof}
由 \(\pi_{m,a}(x)=d_m(x)^a/S_a(m)\) 得
\(
\max_x\pi_{m,a}(x)=M_m^a/S_a(m)
\)，因此
\[
H_\infty(\pi_{m,a})=\log S_a(m)-a\log M_m.
\]
两边除以 \(m\) 并令 \(m\to\infty\)（分别用 \(\lim_{m\to\infty}\frac{1}{m}\log S_a(m)=P_a\) 与 \(\lim_{m\to\infty}\frac{1}{m}\log M_m=\alpha_\ast\)）即得结论。证毕。
\end{proof}

\begin{corollary}[冻结判别：熵亏损消失 \texorpdfstring{$\Longleftrightarrow$}{<->} 乘法链式的纯幂等式]\label{cor:pom-escort-freezing-criterion}
在定理 \ref{thm:pom-escort-minentropy-zero-temperature-identity} 的设定下，对固定整数 \(a\ge 1\)，下列命题等价：
\begin{enumerate}
  \item \(h_\infty^{\mathrm{esc}}(a)=0\)（escort 在指数尺度上冻结）；
  \item \(P_a=a\alpha_\ast\)；
  \item 对所有整数 \(b\ge 2\)，有 \(P_{ab}=bP_a\)（等价 \(\mathcal{H}_{\mathrm{loss}}(a,b)\equiv 0\)）。
\end{enumerate}
\end{corollary}
\begin{proof}
\((1)\Longleftrightarrow(2)\) 由定理 \ref{thm:pom-escort-minentropy-zero-temperature-identity} 立即得到。
\((2)\Longleftrightarrow(3)\) 由不等式 \(S_{ab}(m)\le S_a(m)^b\)（从而 \(P_{ab}\le bP_a\)）与 \(S_{ab}(m)\ge M_m^{ab}\)（从而 \(P_{ab}\ge ab\alpha_\ast\)）夹逼得到：
当 \(P_a=a\alpha_\ast\) 时两端同值，故 \(P_{ab}=bP_a\)；
反向地，若对所有 \(b\ge 2\) 有 \(P_{ab}=bP_a\)，则
\(
\alpha_\ast=\lim_{b\to\infty}\frac{P_{ab}}{ab}=\frac{P_a}{a}
\)，从而 \(P_a=a\alpha_\ast\)。证毕。
\end{proof}

\begin{theorem}[熵亏损锥约束：界、平直性与冻结极限共同定义可行域]\label{thm:pom-entropy-loss-feasibility-cone}
\leanverified{paper\_pom\_entropy\_loss\_feasibility\_cone}
在极限压力谱存在且 \(\alpha_\ast\) 存在的前提下，熵亏损场 \(\mathcal{H}_{\mathrm{loss}}(a,b)=bP_a-P_{ab}\) 必同时满足：
\begin{enumerate}
  \item \textbf{（非负性与上界）}若 \(\lim_{m\to\infty}\frac{1}{m}\log|X_m|=\log\varphi\)，则对所有 \(a\ge 1,b\ge 2\) 有
  \[
  \boxed{\;
  0\ \le\ \mathcal{H}_{\mathrm{loss}}(a,b)\ \le\ (b-1)\log\varphi\; }.
  \]
  \item \textbf{（平直性/乘法一致性）}对所有 \(a\ge 1,b,c\ge 2\) 有
  \[
  \boxed{\;
  \mathcal{H}_{\mathrm{loss}}(a,bc)=c\,\mathcal{H}_{\mathrm{loss}}(a,b)+\mathcal{H}_{\mathrm{loss}}(ab,c)\; }.
  \]
  \item \textbf{（冻结极限兼容性）}记
  \[
  R(a):=P_a-a\alpha_\ast,
  \]
  则
  \[
  \boxed{\;
  R(a)=\lim_{b\to\infty}\frac{\mathcal{H}_{\mathrm{loss}}(a,b)}{b}\; }.
  \]
  且 \(0\le R(a)\le \log\varphi\)，并且 \(R(a)=0\) 当且仅当 \(\mathcal{H}_{\mathrm{loss}}(a,b)\equiv 0\)（对所有 \(b\ge 2\)）。
\end{enumerate}
上述约束共同定义了一个严格的可行域（锥体）\(\mathfrak{C}\)：任何声称来源于折叠纤维计数的压力谱都必须落在 \(\mathfrak{C}\) 内。
\end{theorem}
\begin{proof}
(1) 非负性由 \(S_{ab}(m)\le S_a(m)^b\) 推出 \(P_{ab}\le bP_a\) 得到。
上界由定理 \ref{thm:pom-multiplicative-chain-rule-finite-m} 与 \(0\le H_b(\pi)\le \log|X_m|\) 给出：
\(
\log S_{ab}(m)\ge b\log S_a(m)-(b-1)\log|X_m|
\)，
两边除以 \(m\) 并令 \(m\to\infty\) 得
\(
P_{ab}\ge bP_a-(b-1)\log\varphi
\)，移项即得上界。
\par
(2) 为定理 \ref{thm:pom-entropy-loss-1cocycle-flat}。
\par
(3) 由定义
\[
\frac{\mathcal{H}_{\mathrm{loss}}(a,b)}{b}
=
P_a-\frac{P_{ab}}{b}
=P_a-a\cdot\frac{P_{ab}}{ab},
\]
令 \(b\to\infty\) 且用 \(\alpha_\ast=\lim_{q\to\infty}P_q/q\)（取 \(q=ab\)）即得极限恒等式。
界 \(0\le R(a)\le\log\varphi\) 由 \(M_m^a\le S_a(m)\le |X_m|M_m^a\) 的 \(1/m\) 对数极限得到。
最后一条“当且仅当”由推论 \ref{cor:pom-escort-freezing-criterion} 得到。证毕。
\end{proof}
