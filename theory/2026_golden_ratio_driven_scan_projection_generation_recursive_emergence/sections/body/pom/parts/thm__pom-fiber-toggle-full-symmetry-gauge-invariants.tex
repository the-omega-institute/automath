\[
\mathcal T_{\ell}\ \cong\ \mathfrak S_{\mathsf{Ind}(P_\ell)}

\leanverified{paper\_pom\_toggle\_group\_path\_full\_symmetric}
\end{theorem}

\begin{theorem}[纤维开关群的不变算子代数：\texorpdfstring{$2^r$}{2^r} 扇区与 \(\CC^2\) 张量化]\label{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor}
在推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric} 的设定下，令
\[
\mathcal H_x:=\ell^2(F_x)\ \cong\ \bigotimes_{i=1}^r \ell^2\!\bigl(\mathsf{Ind}(P_{\ell_i})\bigr),
\]
并令 \(\mathcal T_x\) 以置换表象 \(U_g\)（标准基重排）作用于 \(\mathcal H_x\)。
定义共轭不变子代数
\[
\End(\mathcal H_x)^{\mathcal T_x}:=\{A\in \End(\mathcal H_x):\ U_gAU_g^{-1}=A\ \ \forall g\in\mathcal T_x\}.
\]
则 \(\End(\mathcal H_x)^{\mathcal T_x}\) 为交换代数，并存在正则同构
\[
\boxed{\;
\End(\mathcal H_x)^{\mathcal T_x}
\ \cong\
\bigotimes_{i=1}^r \mathrm{span}\{I_i,J_i\}
\ \cong\
\bigotimes_{i=1}^r \CC^{2}\;},
\]
其中对每个分量 \(i\) 记 \(n_i:=|\mathsf{Ind}(P_{\ell_i})|=F_{\ell_i+2}\)，\(I_i\) 为 \(\ell^2([n_i])\) 上恒等算子，\(J_i\) 为全 \(1\) 算子（矩阵元恒为 \(1\)）。
因此
\[
\dim_\CC \End(\mathcal H_x)^{\mathcal T_x}=2^r,
\]
且其一组自然线性基可取为
\[
\Bigl\{\ \bigotimes_{i=1}^r B_i\ :\ B_i\in \mathrm{span}\{I_i,J_i\}\ \Bigr\}.
\]
\leanverified{paper\_pom\_toggle\_gauge\_invariant\_endomorphisms\_c2\_tensor}
\end{theorem}

\begin{proof}
对单分量 \(i\)，由定理 \ref{thm:pom-toggle-group-path-full-symmetric} 得 \(\mathcal T_{\ell_i}\cong \mathfrak S_{n_i}\) 并以置换表象作用于 \(\ell^2([n_i])\)。
对 \(A\in\End(\ell^2([n_i]))\)，条件 \(U_gAU_g^{-1}=A\) 等价于矩阵元仅依赖于有序对 \((a,b)\in [n_i]\times[n_i]\) 在 \(\mathfrak S_{n_i}\) 作用下的轨道。
而 \(\mathfrak S_{n_i}\) 在 \([n_i]\times[n_i]\) 上恰有两条轨道（对角与非对角），故不变子空间为二维并由 \(\{I_i,J_i\}\) 张成。
\par
对直积群 \(\prod_i \mathfrak S_{n_i}\) 的张量积置换表象，不变子代数为各分量不变子代数的张量积，从而得到所示同构与维数公式。
\end{proof}

\begin{corollary}[显式规范投影与 \texorpdfstring{$2^r$}{2^r} 扇区同时对角化]\label{cor:pom-toggle-gauge-sector-decomposition-projections}
\leanverified{paper\_pom\_toggle\_gauge\_sector\_decomposition\_projections}
沿定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor} 的设定。
对每个分量 \(i\) 定义两正交投影
\[
P_i^{\mathrm{triv}}:=\frac{1}{n_i}J_i,
\qquad
P_i^{\mathrm{std}}:=I_i-\frac{1}{n_i}J_i.
\]
则 \(\mathcal H_x\) 具有正交直和分解
\[
\boxed{\;
\mathcal H_x
\ =\ \bigoplus_{S\subseteq [r]}
\Bigl(\ \bigotimes_{i\in S}\mathrm{Im}(P_i^{\mathrm{std}})\ \Bigr)\ \otimes
\Bigl(\ \bigotimes_{i\notin S}\mathrm{Im}(P_i^{\mathrm{triv}})\ \Bigr)\; }.
\]
并且对任意 \(A\in \End(\mathcal H_x)^{\mathcal T_x}\)，其在每个扇区 \(S\subseteq [r]\) 上均以一个标量作用。
此外，群平均算子
\[
\Pi_x(A):=\frac{1}{|\mathcal T_x|}\sum_{g\in\mathcal T_x}U_gAU_g^{-1}
\]
为到 \(\End(\mathcal H_x)^{\mathcal T_x}\) 的显式条件期望。
\end{corollary}

\begin{proof}
对单分量的置换表象有标准分解
\(
\ell^2([n_i])=\CC\mathbf 1\oplus \mathbf 1^\perp
\)，其中 \(\CC\mathbf 1=\mathrm{Im}(P_i^{\mathrm{triv}})\) 为平凡表示，\(\mathbf 1^\perp=\mathrm{Im}(P_i^{\mathrm{std}})\) 为标准表示。
直积群的表象为张量积表象，故分解按子集 \(S\) 张量化得到。
由定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor}，\(\End(\mathcal H_x)^{\mathcal T_x}\) 为交换代数且由各分量的 \(\mathrm{span}\{I_i,J_i\}\) 张量生成，与上述分解同时对角化一致，从而在每个扇区上为标量。
群平均算子是有限群作用下的不变部分投影，故为条件期望。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$q$}{q}-重轨道计数的分量 Bell 张量化]\label{prop:pom-toggle-orbit-count-bell-product}
对整数 \(q\ge 1\)，令 \(\mathcal T_x\) 在 \(F_x^q\) 上以对角方式作用，并记其轨道数为 \(\mathcal O_q(x)\)。
对每个分量记 \(n_i:=|\mathsf{Ind}(P_{\ell_i})|=F_{\ell_i+2}\)，并定义截断 Bell 数
\[
B_q(n):=\#\{\text{把 }[q]\text{ 划分为至多 }n\text{ 个块的集合划分}\}.
\]
则有严格恒等式
\[
\boxed{\;
\mathcal O_q(x)\ =\ \prod_{i=1}^r B_q(n_i)\; }.
\]
\leanverified{paper\_pom\_toggle\_orbit\_count\_bell\_product}
\end{proposition}

\begin{proof}
由推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric}，\(\mathcal T_x\cong \prod_i \mathfrak S_{n_i}\)，并在识别 \(F_x\cong \prod_i [n_i]\) 下按坐标独立作用。
因此 \(F_x^q\cong \prod_i [n_i]^q\)，且轨道分解为各坐标轨道的笛卡尔积，故轨道数相乘。
\par
对单分量 \(\mathfrak S_n\curvearrowright [n]^q\)，两个 \(q\)-元组同轨当且仅当它们诱导相同的“相等关系”：
\[
a_s=a_t\ \Longleftrightarrow\ b_s=b_t\qquad(1\le s,t\le q).
\]
该相等关系等价于 \([q]\) 的一个集合划分，且可实现的块数至多为 \(n\)；因此轨道数恰为 \(B_q(n)\)。
合并两步得到结论。证毕。
\end{proof}

\begin{corollary}[二重轨道与交换子代数维数的严格同一]\label{cor:pom-toggle-orbit-count-commutant-identity}
沿用命题 \ref{prop:pom-toggle-orbit-count-bell-product} 与定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor} 的记号，则
\[
\mathcal O_2(x)=2^r=\dim_{\CC}\End(\mathcal H_x)^{\mathcal T_x}.
\]
\leanverified{paper\_pom\_toggle\_orbit\_count\_commutant\_identity}
\end{corollary}

\begin{proof}
由命题 \ref{prop:pom-toggle-orbit-count-bell-product}，
\[
\mathcal O_2(x)=\prod_{i=1}^r B_2(n_i).
\]
而对任意 \(n_i\ge 2\)，都有
\[
B_2(n_i)=S(2,1)+S(2,2)=2.
\]
故 \(\mathcal O_2(x)=2^r\)。另一边，定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor} 已给出
\(
\dim_{\CC}\End(\mathcal H_x)^{\mathcal T_x}=2^r
\)。
两式合并即得结论。证毕。
\end{proof}

\begin{theorem}[高阶 Bell 轨道的体积主项与群体积归一化]\label{thm:pom-toggle-orbit-count-asymptotic-volume-group}
\leanverified{paper\_pom\_toggle\_orbit\_count\_asymptotic\_volume\_group}
记
\[
d_m(x)=\prod_{i=1}^r n_i,
\qquad
|\mathcal T_x|=\prod_{i=1}^r n_i!.
\]
则存在常数 \(0\le \rho_x<1\)，使得当 \(q\to\infty\) 时，
\[
\mathcal O_q(x)
=
\frac{d_m(x)^q}{|\mathcal T_x|}
\Bigl(1+O_x(\rho_x^q)\Bigr).
\]
特别地，
\[
\lim_{q\to\infty}\frac1q\log \mathcal O_q(x)=\log d_m(x),
\qquad
\lim_{q\to\infty} d_m(x)^{-q}\mathcal O_q(x)=|\mathcal T_x|^{-1}.
\]
\end{theorem}

\begin{proof}
由命题 \ref{prop:pom-toggle-orbit-count-bell-product} 及
\[
B_q(n)=\sum_{k=0}^{n}S(q,k),
\]
只需固定 \(n\) 估计 \(B_q(n)\)。包含--排斥给出
\[
n!\,S(q,n)=\sum_{j=0}^{n}(-1)^j\binom{n}{j}(n-j)^q
=n^q+O_n((n-1)^q).
\]
故
\[
S(q,n)=\frac{n^q}{n!}+O_n((n-1)^q).
\]
对 \(k\le n-1\) 有 \(S(q,k)=O_n((n-1)^q)\)，从而
\[
B_q(n)
=
\frac{n^q}{n!}+O_n((n-1)^q)
=
\frac{n^q}{n!}\Bigl(1+O_n((1-\tfrac1n)^q)\Bigr).
\]
逐因子相乘，取
\[
\rho_x:=\max_{1\le i\le r}\Bigl(1-\frac1{n_i}\Bigr)<1,
\]
便得
\[
\mathcal O_q(x)
=
\prod_{i=1}^r\frac{n_i^q}{n_i!}\Bigl(1+O_i(\rho_x^q)\Bigr)
=
\frac{d_m(x)^q}{|\mathcal T_x|}\Bigl(1+O_x(\rho_x^q)\Bigr).
\]
两条极限式随即成立。证毕。
\end{proof}

\begin{corollary}[Bell 轨道塔的双尺度层析]\label{cor:pom-toggle-orbit-count-two-scale-tomography}
设
\[
\Gamma_x\cong \bigsqcup_{i=1}^r P_{\ell_i},
\qquad
\dim_\square(\mathcal F_x):=\sum_{i=1}^r \left\lceil\frac{\ell_i}{2}\right\rceil.
\]
则同一轨道塔同时满足
\[
r=\log_2\mathcal O_2(x),
\qquad
\lim_{q\to\infty}\frac1q\log \mathcal O_q(x)=\log d_m(x).
\]
并且
\[
\left|
\lim_{q\to\infty}\frac1q\log \mathcal O_q(x)
-(\log\varphi)\sum_{i=1}^r\ell_i
\right|
\le r\log\varphi,
\]
\[
\left|
\lim_{q\to\infty}\frac1q\log \mathcal O_q(x)
-2(\log\varphi)\dim_\square(\mathcal F_x)
\right|
\le r\log\varphi.
\]
\leanverified{paper\_pom\_toggle\_orbit\_count\_two\_scale\_tomography}
\end{corollary}

\begin{proof}
第一式由推论 \ref{cor:pom-toggle-orbit-count-commutant-identity} 给出，第二式由定理 \ref{thm:pom-toggle-orbit-count-asymptotic-volume-group} 给出。再由
\[
d_m(x)=\prod_{i=1}^r F_{\ell_i+2}
\]
与 Binet 估计
\[
\varphi^{\ell_i}\le F_{\ell_i+2}\le \varphi^{\ell_i+1},
\]
得
\[
\left|\log d_m(x)-(\log\varphi)\sum_{i=1}^r\ell_i\right|\le r\log\varphi.
\]
最后利用
\[
2\left\lceil\frac{\ell_i}{2}\right\rceil-1\le \ell_i\le 2\left\lceil\frac{\ell_i}{2}\right\rceil
\]
逐项求和，即得第二条不等式。证毕。
\end{proof}

\begin{theorem}[有限 Bell 层析的前两级精确反演]\label{thm:pom-toggle-orbit-count-finite-lowlevel-inversion}
记
\[
m_1(x):=\#\{i:\ell_i=1\}=\#\{i:n_i=2\},
\qquad
m_2(x):=\#\{i:\ell_i=2\}=\#\{i:n_i=3\}.
\]
则有精确公式
\[
\mathcal O_3(x)=4^{m_1(x)}5^{\,r-m_1(x)},
\]
\[
\mathcal O_4(x)=8^{m_1(x)}14^{m_2(x)}15^{\,r-m_1(x)-m_2(x)}.
\]
从而
\[
m_1(x)=\frac{r\log 5-\log \mathcal O_3(x)}{\log(5/4)},
\]
\[
m_2(x)=
\frac{r\log 15+m_1(x)\log(15/8)-\log \mathcal O_4(x)}{\log(15/14)}.
\]
\leanverified{paper\_pom\_toggle\_orbit\_count\_finite\_lowlevel\_inversion}
\end{theorem}

\begin{proof}
由 \(n_i=F_{\ell_i+2}\)，可取值仅为 \(2,3,5,8,\dots\)。由
\[
B_3(2)=1+3=4,
\qquad
B_3(n)=1+3+1=5\quad(n\ge 3),
\]
结合命题 \ref{prop:pom-toggle-orbit-count-bell-product} 得
\[
\mathcal O_3(x)=4^{m_1(x)}5^{\,r-m_1(x)}.
\]
同理，
\[
B_4(2)=1+7=8,
\qquad
B_4(3)=1+7+6=14,
\qquad
B_4(n)=1+7+6+1=15\quad(n\ge 4).
\]
而 Fibonacci 数列中不存在 \(4\)，故其余情形统一落在 \(n_i\ge 5\)。于是
\[
\mathcal O_4(x)=8^{m_1(x)}14^{m_2(x)}15^{\,r-m_1(x)-m_2(x)}.
\]
两条反演公式由取对数直接解出。证毕。
\end{proof}

\begin{proposition}[轨道指数母函数的截断指数分解]\label{prop:pom-toggle-orbit-count-egf-factorization}
\leanverified{paper\_pom\_toggle\_orbit\_count\_egf\_factorization}
定义指数母函数
\[
\mathcal O_x^{\exp}(z):=\sum_{q\ge 0}\mathcal O_q(x)\frac{z^q}{q!},
\qquad
E_n(u):=\sum_{k=0}^{n}\frac{u^k}{k!}.
\]
则有闭式分解
\[
\mathcal O_x^{\exp}(z)=\prod_{i=1}^r E_{n_i}(e^z-1).
\]
\end{proposition}

\begin{proof}
由命题 \ref{prop:pom-toggle-orbit-count-bell-product}，
\[
\mathcal O_x^{\exp}(z)
=
\prod_{i=1}^r\left(\sum_{q\ge 0}B_q(n_i)\frac{z^q}{q!}\right).
\]
而 Stirling 数的标准指数母函数为
\[
\sum_{q\ge 0}S(q,k)\frac{z^q}{q!}=\frac{(e^z-1)^k}{k!}.
\]
于是
\[
\sum_{q\ge 0}B_q(n)\frac{z^q}{q!}
=
\sum_{k=0}^{n}\frac{(e^z-1)^k}{k!}
=
E_n(e^z-1),
\]
逐因子相乘即得结论。证毕。
\end{proof}

\begin{corollary}[同一 Bell 塔的三重质量读出]\label{cor:pom-toggle-orbit-count-three-masses}
\leanpartial{paper\_pom\_toggle\_orbit\_count\_three\_masses}{The requested Lean signature is false as stated: at \(r=0\), the right-hand side is fixed to \(0\), \(1\), and \(1\), so arbitrary \texttt{degree\_u}, \texttt{top\_coeff\_denom}, and \texttt{growth\_mass} cannot satisfy it.}
记
\[
A_x:=\sum_{i=1}^r n_i,
\qquad
M_x:=\prod_{i=1}^r n_i=d_m(x),
\qquad
G_x:=\prod_{i=1}^r n_i!=|\mathcal T_x|.
\]
则同一母函数同时满足：
\[
\deg_{u}\mathcal O_x^{\exp}(u)=A_x \qquad (u:=e^z-1),
\]
\[
[u^{A_x}]\,\mathcal O_x^{\exp}(u)=G_x^{-1},
\]
\[
\lim_{q\to\infty}\frac1q\log \mathcal O_q(x)=\log M_x.
\]
\end{corollary}

\begin{proof}

故总次数为 \(\sum_i n_i=A_x\)，最高次项系数为
\[
\prod_{i=1}^r \frac1{n_i!}=G_x^{-1}.
\]
最后一式正是定理 \ref{thm:pom-toggle-orbit-count-asymptotic-volume-group} 的极限式。证毕。
\end{proof}
