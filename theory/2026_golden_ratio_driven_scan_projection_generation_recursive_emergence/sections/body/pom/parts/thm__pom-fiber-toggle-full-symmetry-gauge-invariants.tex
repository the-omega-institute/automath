\begin{theorem}[路径独立集的开关群为全对称群]\label{thm:pom-toggle-group-path-full-symmetric}
对任意 \(\ell\ge 1\)，令 \(\mathcal T_{\ell}:=\langle \tau_1,\dots,\tau_\ell\rangle\) 为定义 \ref{def:pom-fiber-toggle-group} 的路径分量开关群，其作用对象为有限集合 \(\mathsf{Ind}(P_\ell)\)。
则存在自然同构
\[
\mathcal T_{\ell}\ \cong\ \mathfrak S_{\mathsf{Ind}(P_\ell)}
\ \cong\ \mathfrak S_{F_{\ell+2}}.
\]
\end{theorem}

\begin{corollary}[一般纤维开关群的全对称分量化]\label{cor:pom-toggle-group-general-fiber-full-symmetric}
设 \(x\in X_m\) 且 \(\Gamma_x\simeq \bigsqcup_{i=1}^r P_{\ell_i}\)。
令 \(\mathcal T_x\le \mathrm{Sym}(F_x)\) 为定义 \ref{def:pom-fiber-toggle-group} 的纤维开关群。
则存在自然同构
\[
\boxed{\;
\mathcal T_x\ \cong\ \prod_{i=1}^r \mathfrak S_{F_{\ell_i+2}}\; }.
\]
\end{corollary}

\begin{proof}
由推论 \ref{cor:pom-toggle-product-components} 有 \(\mathcal T_x\cong \prod_i \mathcal T_{\ell_i}\)。
对每个分量应用定理 \ref{thm:pom-toggle-group-path-full-symmetric} 并合并即得。
\end{proof}

\begin{theorem}[纤维开关群的不变算子代数：\texorpdfstring{$2^r$}{2^r} 扇区与 \(\CC^2\) 张量化]\label{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor}
在推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric} 的设定下，令
\[
\mathcal H_x:=\ell^2(F_x)\ \cong\ \bigotimes_{i=1}^r \ell^2\!\bigl(\mathsf{Ind}(P_{\ell_i})\bigr),
\]
并令 \(\mathcal T_x\) 以置换表象 \(U_g\)（标准基重排）作用于 \(\mathcal H_x\)。
定义共轭不变子代数
\[
\End(\mathcal H_x)^{\mathcal T_x}:=\{A\in \End(\mathcal H_x):\ U_gAU_g^{-1}=A\ \ \forall g\in\mathcal T_x\}.
\]
则 \(\End(\mathcal H_x)^{\mathcal T_x}\) 为交换代数，并存在正则同构
\[
\boxed{\;
\End(\mathcal H_x)^{\mathcal T_x}
\ \cong\
\bigotimes_{i=1}^r \mathrm{span}\{I_i,J_i\}
\ \cong\
\bigotimes_{i=1}^r \CC^{2}\;},
\]
其中对每个分量 \(i\) 记 \(n_i:=|\mathsf{Ind}(P_{\ell_i})|=F_{\ell_i+2}\)，\(I_i\) 为 \(\ell^2([n_i])\) 上恒等算子，\(J_i\) 为全 \(1\) 算子（矩阵元恒为 \(1\)）。
因此
\[
\dim_\CC \End(\mathcal H_x)^{\mathcal T_x}=2^r,
\]
且其一组自然线性基可取为
\[
\Bigl\{\ \bigotimes_{i=1}^r B_i\ :\ B_i\in \mathrm{span}\{I_i,J_i\}\ \Bigr\}.
\]
\end{theorem}

\begin{proof}
对单分量 \(i\)，由定理 \ref{thm:pom-toggle-group-path-full-symmetric} 得 \(\mathcal T_{\ell_i}\cong \mathfrak S_{n_i}\) 并以置换表象作用于 \(\ell^2([n_i])\)。
对 \(A\in\End(\ell^2([n_i]))\)，条件 \(U_gAU_g^{-1}=A\) 等价于矩阵元仅依赖于有序对 \((a,b)\in [n_i]\times[n_i]\) 在 \(\mathfrak S_{n_i}\) 作用下的轨道。
而 \(\mathfrak S_{n_i}\) 在 \([n_i]\times[n_i]\) 上恰有两条轨道（对角与非对角），故不变子空间为二维并由 \(\{I_i,J_i\}\) 张成。
\par
对直积群 \(\prod_i \mathfrak S_{n_i}\) 的张量积置换表象，不变子代数为各分量不变子代数的张量积，从而得到所示同构与维数公式。
\end{proof}

\begin{corollary}[显式规范投影与 \texorpdfstring{$2^r$}{2^r} 扇区同时对角化]\label{cor:pom-toggle-gauge-sector-decomposition-projections}
沿定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor} 的设定。
对每个分量 \(i\) 定义两正交投影
\[
P_i^{\mathrm{triv}}:=\frac{1}{n_i}J_i,
\qquad
P_i^{\mathrm{std}}:=I_i-\frac{1}{n_i}J_i.
\]
则 \(\mathcal H_x\) 具有正交直和分解
\[
\boxed{\;
\mathcal H_x
\ =\ \bigoplus_{S\subseteq [r]}
\Bigl(\ \bigotimes_{i\in S}\mathrm{Im}(P_i^{\mathrm{std}})\ \Bigr)\ \otimes
\Bigl(\ \bigotimes_{i\notin S}\mathrm{Im}(P_i^{\mathrm{triv}})\ \Bigr)\; }.
\]
并且对任意 \(A\in \End(\mathcal H_x)^{\mathcal T_x}\)，其在每个扇区 \(S\subseteq [r]\) 上均以一个标量作用。
此外，群平均算子
\[
\Pi_x(A):=\frac{1}{|\mathcal T_x|}\sum_{g\in\mathcal T_x}U_gAU_g^{-1}
\]
为到 \(\End(\mathcal H_x)^{\mathcal T_x}\) 的显式条件期望。
\end{corollary}

\begin{proof}
对单分量的置换表象有标准分解
\(
\ell^2([n_i])=\CC\mathbf 1\oplus \mathbf 1^\perp
\)，其中 \(\CC\mathbf 1=\mathrm{Im}(P_i^{\mathrm{triv}})\) 为平凡表示，\(\mathbf 1^\perp=\mathrm{Im}(P_i^{\mathrm{std}})\) 为标准表示。
直积群的表象为张量积表象，故分解按子集 \(S\) 张量化得到。
由定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor}，\(\End(\mathcal H_x)^{\mathcal T_x}\) 为交换代数且由各分量的 \(\mathrm{span}\{I_i,J_i\}\) 张量生成，与上述分解同时对角化一致，从而在每个扇区上为标量。
群平均算子是有限群作用下的不变部分投影，故为条件期望。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$q$}{q}-重轨道计数的分量 Bell 张量化]\label{prop:pom-toggle-orbit-count-bell-product}
对整数 \(q\ge 1\)，令 \(\mathcal T_x\) 在 \(F_x^q\) 上以对角方式作用，并记其轨道数为 \(\mathcal O_q(x)\)。
对每个分量记 \(n_i:=|\mathsf{Ind}(P_{\ell_i})|=F_{\ell_i+2}\)，并定义截断 Bell 数
\[
B_q(n):=\#\{\text{把 }[q]\text{ 划分为至多 }n\text{ 个块的集合划分}\}.
\]
则有严格恒等式
\[
\boxed{\;
\mathcal O_q(x)\ =\ \prod_{i=1}^r B_q(n_i)\; }.
\]
\end{proposition}

\begin{proof}
由推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric}，\(\mathcal T_x\cong \prod_i \mathfrak S_{n_i}\)，并在识别 \(F_x\cong \prod_i [n_i]\) 下按坐标独立作用。
因此 \(F_x^q\cong \prod_i [n_i]^q\)，且轨道分解为各坐标轨道的笛卡尔积，故轨道数相乘。
\par
对单分量 \(\mathfrak S_n\curvearrowright [n]^q\)，两个 \(q\)-元组同轨当且仅当它们诱导相同的“相等关系”：
\[
a_s=a_t\ \Longleftrightarrow\ b_s=b_t\qquad(1\le s,t\le q).
\]
该相等关系等价于 \([q]\) 的一个集合划分，且可实现的块数至多为 \(n\)；因此轨道数恰为 \(B_q(n)\)。
合并两步得到结论。证毕。
\end{proof}
