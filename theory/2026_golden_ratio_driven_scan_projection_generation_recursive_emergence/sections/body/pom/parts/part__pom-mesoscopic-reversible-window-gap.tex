\paragraph{严格谱隙下的介观可逆窗与充分恢复位率下沉}
沿用定理 \ref{thm:pom-oracle-renyi-pressure-failure-bound} 的
\(C_m(B),\mathrm{Succ}_m(B)\) 记号，并对每个分辨率 \(m\) 记
\[
M_m:=\max_{x\in X_m}d_m(x),
\qquad
\mathcal M_m:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
K_m:=|\mathcal M_m|,
\]
\[
M_m^{(2)}:=\max\{d_m(x):\ x\in X_m\setminus\mathcal M_m\},
\]
其中若次大纤维不存在则约定 \(M_m^{(2)}:=1\)。
再设
\[
\alpha_\ast:=\lim_{m\to\infty}\frac{1}{m}\log M_m,\qquad
g_\ast:=\lim_{m\to\infty}\frac{1}{m}\log K_m,
\qquad
\alpha_\ast^{(2)}:=\limsup_{m\to\infty}\frac{1}{m}\log M_m^{(2)},
\]
并假设严格谱隙
\[
\alpha_\ast^{(2)}<\alpha_\ast.
\]

\begin{theorem}[严格谱隙下存在一段严格低于最大纤维尺度的介观可逆窗]\label{thm:pom-mesoscopic-reversible-window-strict-gap}
设整数序列 \((B_m)_{m\ge 1}\subset\NN\) 满足
\[
\alpha_\ast^{(2)}
<
\liminf_{m\to\infty}\frac{B_m\log 2}{m}
\le
\limsup_{m\to\infty}\frac{B_m\log 2}{m}
<
\alpha_\ast.
\]
则对充分大的 \(m\)，有精确恒等式
\[
C_m(B_m)=2^m-K_m\bigl(M_m-2^{B_m}\bigr),
\]
从而
\[
1-\mathrm{Succ}_m(B_m)
=
\frac{K_m\bigl(M_m-2^{B_m}\bigr)}{2^m}.
\]
进一步，
\[
\frac{2^{B_m}}{M_m}\longrightarrow 0,
\]
并且
\[
1-\mathrm{Succ}_m(B_m)
=
\exp\!\bigl((\alpha_\ast+g_\ast-\log 2)m+o(m)\bigr).
\]
\end{theorem}
\leanverified{paper\_pom\_mesoscopic\_reversible\_window\_strict\_gap}
\begin{proof}
由
\[
\liminf_{m\to\infty}\frac{B_m\log 2}{m}>\alpha_\ast^{(2)}
\]
及 \(\alpha_\ast^{(2)}=\limsup_m \frac{1}{m}\log M_m^{(2)}\) 可知，对充分大的 \(m\) 有
\[
2^{B_m}>M_m^{(2)}.
\]
同理，由
\[
\limsup_{m\to\infty}\frac{B_m\log 2}{m}<\alpha_\ast
\]
及 \(\alpha_\ast=\lim_m \frac{1}{m}\log M_m\) 可知，对充分大的 \(m\) 亦有
\[
2^{B_m}<M_m.
\]
故当 \(m\) 充分大时，除最大层 \(\mathcal M_m\) 外，其余所有纤维都完整落在阈值 \(2^{B_m}\) 之下，而 \(\mathcal M_m\) 中的 \(K_m\) 条最大纤维都被截断到 \(2^{B_m}\)。
因此
\[
\begin{aligned}
C_m(B_m)
&=\sum_{x\in X_m\setminus\mathcal M_m} d_m(x)+\sum_{x\in\mathcal M_m}2^{B_m}\\
&=\Bigl(2^m-K_mM_m\Bigr)+K_m2^{B_m}\\
&=2^m-K_m\bigl(M_m-2^{B_m}\bigr),
\end{aligned}
\]
其中使用了
\[
\sum_{x\in X_m}d_m(x)=|\Omega_m|=2^m.
\]
两边除以 \(2^m\) 即得成功率公式。

再由 \(\limsup_m \frac{B_m\log 2}{m}<\alpha_\ast\) 与
\(\alpha_\ast=\lim_m \frac1m\log M_m\)，得
\[
\frac{2^{B_m}}{M_m}\longrightarrow 0.
\]
故
\[
1-\mathrm{Succ}_m(B_m)
=
\frac{K_mM_m}{2^m}\left(1-\frac{2^{B_m}}{M_m}\right)
=
\frac{K_mM_m}{2^m}(1+o(1)).
\]
最后，由
\[
\frac{1}{m}\log K_m\to g_\ast,
\qquad
\frac{1}{m}\log M_m\to \alpha_\ast,
\]
得到
\[
\frac{K_mM_m}{2^m}
=
\exp\!\bigl((\alpha_\ast+g_\ast-\log 2)m+o(m)\bigr),
\]
从而结论成立。证毕。
\end{proof}

\begin{corollary}[最大层总质量指数可忽略时的充分恢复位率下沉]\label{cor:pom-sufficient-recovery-rate-sinks-below-maxfiber}
在定理 \ref{thm:pom-mesoscopic-reversible-window-strict-gap} 的设定下，若进一步满足
\[
\alpha_\ast+g_\ast<\log 2,
\]
则对任意
\[
\beta\in\left(\frac{\alpha_\ast^{(2)}}{\log 2},\,\frac{\alpha_\ast}{\log 2}\right)
\]
并取
\[
B_m:=\lfloor \beta m\rfloor,
\]
都有
\[
\mathrm{Succ}_m(B_m)\longrightarrow 1.
\]
因此，充分恢复所需的位率不再受最大纤维指数 \(\alpha_\ast/\log 2\) 的直接支配，而已严格下沉到次大纤维指数 \(\alpha_\ast^{(2)}/\log 2\) 之上任意邻域。
\end{corollary}
\begin{proof}
对 \(B_m=\lfloor \beta m\rfloor\)，有
\[
\frac{B_m\log 2}{m}\longrightarrow \beta\log 2,
\]
故定理 \ref{thm:pom-mesoscopic-reversible-window-strict-gap} 适用，并给出
\[
1-\mathrm{Succ}_m(B_m)
=
\exp\!\bigl((\alpha_\ast+g_\ast-\log 2)m+o(m)\bigr).
\]
由假设 \(\alpha_\ast+g_\ast<\log 2\)，右端指数趋于 \(-\infty\)，故
\[
\mathrm{Succ}_m(B_m)\longrightarrow 1.
\]
证毕。
\end{proof}
