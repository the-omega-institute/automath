\begin{corollary}[折叠语义下的小失真节省律：\texorpdfstring{$q=\tfrac12$}{q=1/2} 配分的首阶斜率常数]\label{cor:pom-diagonal-rate-small-distortion-saving-half-moment-fold}
在折叠语义下对每个分辨率 \(m\) 令推前分布 \(w_m(x)=d_m(x)/2^m\)，并定义半阶配分与其 Hellinger 常数
\[
S_{1/2}(m):=\sum_{x\in X_m} d_m(x)^{1/2},\qquad
A_m:=A_{1/2}(w_m)=\sum_{x\in X_m}\sqrt{w_m(x)}=2^{-m/2}S_{1/2}(m).
\]
令信息节省量
\[
\Delta_m(\delta):=H(w_m)-R_{w_m}(\delta).
\]
则当 \(\delta\downarrow 0\) 时，对每个固定 \(m\) 有
\[
\Delta_m(\delta)=\delta\log\frac{A_m^2-1}{\delta}+O(\delta),
\]
其中 \(O(\delta)\) 的常数可取为仅依赖于 \(w_m\) 的有界量。
\par
进一步，取任意序列 \(\delta_m\downarrow 0\) 满足 \(\log(1/\delta_m)=o(m)\)。
若极限
\[
a_{1/2}:=\lim_{m\to\infty}\frac{1}{m}\log(A_m^2-1)
\]
存在（允许为 \(0\)），则成立比值极限
\[
\lim_{m\to\infty}\frac{\Delta_m(\delta_m)}{m\,\delta_m}=a_{1/2}.
\]
若再假设半阶压力极限
\[
\tau(1/2):=\lim_{m\to\infty}\frac{1}{m}\log S_{1/2}(m)
\]
存在且 \(a_{1/2}>0\)，则必有显式恒等式
\[
a_{1/2}=2\tau(1/2)-\log 2.
\]
\end{corollary}
\begin{proof}
对固定 \(m\) 将定理 \ref{thm:pom-diagonal-rate-small-distortion-hellinger-endpoint} 应用于 \(w=w_m\)，并取差
\(
\Delta_m(\delta)=H(w_m)-R_{w_m}(\delta)
\)
即得首式；其中 \(A_{1/2}(w_m)=A_m\)。
\par
将 \(\delta=\delta_m\) 代入并除以 \(m\delta_m\)，得到
\[
\frac{\Delta_m(\delta_m)}{m\delta_m}
=\frac{1}{m}\log(A_m^2-1)-\frac{1}{m}\log\delta_m+O\!\left(\frac{1}{m}\right).
\]
由 \(\log(1/\delta_m)=o(m)\) 得 \(\frac{1}{m}\log\delta_m\to 0\)，并结合 \(m\to\infty\) 下 \(O(1/m)\to 0\)，从而若 \(\frac{1}{m}\log(A_m^2-1)\to a_{1/2}\) 则比值极限成立。
\par
若 \(a_{1/2}>0\)，则 \(A_m^2\) 必指数发散，从而
\(
\log(A_m^2-1)=\log A_m^2+o(m)
\)。
又由 \(A_m=2^{-m/2}S_{1/2}(m)\) 得
\[
\frac{1}{m}\log A_m^2
=\frac{2}{m}\log S_{1/2}(m)-\log 2
\ \longrightarrow\ 2\tau(1/2)-\log 2,
\]
从而得到恒等式。证毕。
\end{proof}

\begin{theorem}[半阶压强的极端下界与平方根冻结判据]\label{thm:pom-half-order-pressure-extreme-lower-bound-sqrt-freezing}
\leanverified{paper\_pom\_half\_order\_pressure\_extreme\_lower\_bound\_sqrt\_freezing}
对每个 \(m\) 记最大纤维 \(M_m:=\max_{x\in X_m}d_m(x)\)，最大集合
\(
\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=M_m\}
\)
及其基数 \(K_m:=|\mathcal{M}_m|\)。
设指数率极限
\[
\alpha_*:=\lim_{m\to\infty}\frac{1}{m}\log M_m,\qquad
g_*:=\lim_{m\to\infty}\frac{1}{m}\log K_m
\]
存在（以 \(\liminf\) 版本亦成立）。
则总有极端下界
\[
\liminf_{m\to\infty}\frac{1}{m}\log S_{1/2}(m)\ \ge\ g_*+\frac{1}{2}\alpha_*,
\]
等价地（用 \(A_m=2^{-m/2}S_{1/2}(m)\)）
\[
\liminf_{m\to\infty}\frac{1}{m}\log A_m\ \ge\ g_*+\frac{1}{2}\bigl(\alpha_*-\log 2\bigr).
\]
\par
下列两命题等价：
\begin{enumerate}
  \item \(\displaystyle S_{1/2}(m)=K_m M_m^{1/2}\bigl(1+o(1)\bigr)\)（半阶配分由最大层主导）；
  \item \(\displaystyle \sum_{x\notin\mathcal{M}_m} d_m(x)^{1/2}=o\bigl(K_m M_m^{1/2}\bigr)\)。
\end{enumerate}
在上述等价成立时，若再满足 \(2g_*+\alpha_*>\log 2\)，则
\[
a_{1/2}:=\lim_{m\to\infty}\frac{1}{m}\log(A_m^2-1)
\]
存在且满足
\[
a_{1/2}=2g_*+\alpha_*-\log 2.
\]
\end{theorem}
\begin{proof}
由 \(S_{1/2}(m)\ge \sum_{x\in\mathcal{M}_m} M_m^{1/2}=K_m M_m^{1/2}\) 得
\[
\frac{1}{m}\log S_{1/2}(m)\ge \frac{1}{m}\log K_m+\frac{1}{2m}\log M_m,
\]
取 \(\liminf\) 即得极端下界；将 \(A_m=2^{-m/2}S_{1/2}(m)\) 代入得到等价形式。
\par
令
\(
T_m:=\sum_{x\notin\mathcal{M}_m} d_m(x)^{1/2}
\)，则 \(S_{1/2}(m)=K_m M_m^{1/2}+T_m\)。
于是
\[
\frac{S_{1/2}(m)}{K_m M_m^{1/2}}=1+\frac{T_m}{K_m M_m^{1/2}},
\]
从而 \(S_{1/2}(m)=K_m M_m^{1/2}(1+o(1))\) 当且仅当 \(T_m=o(K_m M_m^{1/2})\)。
\par
在上述等价成立时，
\[
A_m^2
=\frac{S_{1/2}(m)^2}{2^m}
=\frac{K_m^2 M_m}{2^m}\,(1+o(1)).
\]
若 \(2g_*+\alpha_*-\log 2>0\)，则 \(A_m^2\to\infty\)，且
\[
\frac{1}{m}\log(A_m^2-1)
=\frac{1}{m}\log A_m^2+o(1)
=2g_*+\alpha_*-\log 2+o(1),
\]
从而 \(a_{1/2}\) 存在并等于所示常数。证毕。
\end{proof}

\begin{corollary}[临界平面的端点斜率：由 \texorpdfstring{$q=\tfrac12$}{q=1/2} 压强反演]\label{cor:pom-query-distortion-critical-slope-inverts-half-pressure}
对每个分辨率 \(m\) 与 \(\beta\in[0,1)\)，定义有限窗临界比特密度
\[
b_{\mathrm{crit}}^{(m)}(\beta,\delta):=\frac{(1-\beta)\,R_{w_m}(\delta)}{\log 2},\qquad \delta\in[0,\delta_0(m)),\qquad \delta_0(m):=1-\sum_x w_m(x)^2.
\]
则当 \(\delta\downarrow 0\) 时有
\[
\frac{\partial}{\partial \delta}\,b_{\mathrm{crit}}^{(m)}(\beta,\delta)
=-\frac{1-\beta}{\log 2}\,\log\frac{A_m^2-1}{\delta}+O(1),
\]
其中 \(O(1)\) 的常数可取为仅依赖于 \(w_m\) 的有界量。
从而在任何满足 \(\delta_m\downarrow 0\)、\(\log(1/\delta_m)=o(m)\) 的对角尺度上，
\[
\lim_{m\to\infty}\frac{-\partial_\delta b_{\mathrm{crit}}^{(m)}(\beta,\delta_m)}{m}
=\frac{1-\beta}{\log 2}\,a_{1/2},
\]
其中 \(a_{1/2}\) 如推论 \ref{cor:pom-diagonal-rate-small-distortion-saving-half-moment-fold}。
若进一步 \(\tau(1/2)\) 存在且 \(a_{1/2}>0\)，则亦有
\[
\lim_{m\to\infty}\frac{-\partial_\delta b_{\mathrm{crit}}^{(m)}(\beta,\delta_m)}{m}
=\frac{1-\beta}{\log 2}\,\bigl(2\tau(1/2)-\log 2\bigr).
\]
\end{corollary}
\begin{proof}
由定义 \(b_{\mathrm{crit}}^{(m)}=(1-\beta)R_{w_m}(\delta)/\log 2\) 与
\(
R'_{w_m}(\delta)=-\lambda_m(\delta)
\)
（定理 \ref{thm:pom-diagonal-rate-explicit-param-and-derivative}），可得
\[
\partial_\delta b_{\mathrm{crit}}^{(m)}(\beta,\delta)
=-\frac{1-\beta}{\log 2}\,\lambda_m(\delta).
\]
由定理 \ref{thm:pom-diagonal-rate-small-distortion-hellinger-endpoint}（取 \(w=w_m\)）在 \(\delta\downarrow 0\) 的对偶展开
\(
\lambda_m(\delta)=\log\frac{A_m^2-1}{\delta}+O(1)
\)
即得首式。
\par
再将 \(\delta=\delta_m\) 代入并除以 \(m\)，并用 \(\log(1/\delta_m)=o(m)\) 消去 \(\frac{1}{m}\log(1/\delta_m)\) 项，得到第二式；
最后一式由推论 \ref{cor:pom-diagonal-rate-small-distortion-saving-half-moment-fold} 的恒等式 \(a_{1/2}=2\tau(1/2)-\log 2\)（在 \(a_{1/2}>0\) 下）代入即得。证毕。
\end{proof}
