在定理 \ref{thm:pom-diagonal-rate-critical-line-laplacian-first-order} 的首阶谱律之上，可以将小失真端点的最优核视为一族可逆连续时间链的 Euler 离散化。

\begin{corollary}[临界连续时间生成元与完全图电导]\label{cor:pom-diagonal-rate-critical-continuous-time-generator}
沿用定理 \ref{thm:pom-diagonal-rate-critical-line-laplacian-first-order} 的设定与记号。
定义矩阵
\[
Q_w:=\lim_{\delta\downarrow 0}\frac{K_\delta-I}{\delta},
\]
则该极限在矩阵元意义下存在，并对任意 \(x\ne y\) 成立
\[
(Q_w)_{xy}=\frac{1}{C}\sqrt{\frac{w(y)}{w(x)}},
\qquad
(Q_w)_{xx}=-\sum_{y\ne x}(Q_w)_{xy}
=-\frac{A-\sqrt{w(x)}}{C\sqrt{w(x)}}.
\]
从而 \(K_\delta=I+\delta Q_w+o(\delta)\)（\(\delta\downarrow 0\)）。
并且 \(Q_w\) 为以 \(w\) 为平稳分布的可逆连续时间生成元：
\[
\sum_{y\in X}(Q_w)_{xy}=0,\qquad
w(x)(Q_w)_{xy}=w(y)(Q_w)_{yx}\quad(\forall x,y).
\]
若定义电导
\[
c_{xy}:=w(x)(Q_w)_{xy}=\frac{1}{C}\sqrt{w(x)w(y)}=c_{yx},
\]
则 \(Q_w\) 在完全图上的可逆表示为 \((Q_w)_{xy}=c_{xy}/w(x)\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-diagonal-rate-critical-line-laplacian-first-order} 的证明可得逐元展开：
当 \(x\ne y\) 时
\(
K_\delta(y\mid x)=\delta\,\frac{\sqrt{w(y)}}{\sqrt{w(x)}\,C}+o(\delta)
\)，
而
\(
K_\delta(x\mid x)=1-\delta\,\frac{A-\sqrt{w(x)}}{C\sqrt{w(x)}}+o(\delta)
\)。
因此 \(\lim_{\delta\downarrow 0}(K_\delta-I)/\delta\) 逐元存在并给出所示 \(Q_w\) 的矩阵元。
由 \(\sum_y K_\delta(y\mid x)=1\) 得 \(\sum_y (Q_w)_{xy}=0\)；
由 \(P^\star_\delta(x,y)=w(x)K_\delta(y\mid x)=w(y)K_\delta(x\mid y)\) 得详细平衡。
电导表达式为定义的直接推论。证毕。
\end{proof}

\begin{corollary}[张量积下的电导核乘法闭合]\label{cor:pom-diagonal-rate-critical-conductance-tensor-multiplicativity}
取两全支撑分布 \(w\in\Delta(X)\)、\(v\in\Delta(Y)\)，并令 \(\omega:=w\otimes v\in\Delta(X\times Y)\)。
记
\[
A(w):=\sum_{x\in X}\sqrt{w(x)},\quad C(w):=A(w)^2-1,
\qquad
A(v):=\sum_{y\in Y}\sqrt{v(y)},\quad C(v):=A(v)^2-1,
\]
以及 \(A(\omega),C(\omega)\) 的对应定义。
则
\[
A(\omega)=A(w)A(v),\qquad
1+C(\omega)=(1+C(w))(1+C(v)),
\qquad
C(\omega)=C(w)+C(v)+C(w)C(v).
\]
令 \(Q_w,Q_v,Q_\omega\) 分别为推论 \ref{cor:pom-diagonal-rate-critical-continuous-time-generator} 对应的临界连续时间生成元，并记其电导为
\[
c^w_{x_1x_2}:=w(x_1)(Q_w)_{x_1x_2},\qquad
c^v_{y_1y_2}:=v(y_1)(Q_v)_{y_1y_2},\qquad
c^\omega_{(x_1,y_1),(x_2,y_2)}:=\omega(x_1,y_1)(Q_\omega)_{(x_1,y_1),(x_2,y_2)}.
\]
则对任意 \((x_1,y_1)\ne(x_2,y_2)\) 有严格恒等式
\[
c^\omega_{(x_1,y_1),(x_2,y_2)}
=\frac{1}{C(\omega)}\sqrt{w(x_1)w(x_2)}\sqrt{v(y_1)v(y_2)}
=\frac{C(w)C(v)}{C(\omega)}\,c^w_{x_1x_2}\,c^v_{y_1y_2}.
\]
\end{corollary}
\begin{proof}
由 \(A(\omega)=A(w)A(v)\) 立即得到 \(1+C(\omega)=A(\omega)^2=(1+C(w))(1+C(v))\)，展开即得 \(C(\omega)=C(w)+C(v)+C(w)C(v)\)。
电导恒等式由推论 \ref{cor:pom-diagonal-rate-critical-continuous-time-generator} 的闭式
\(
c_{xy}=\frac{1}{C}\sqrt{w(x)w(y)}
\)
对 \(w,v,\omega\) 分别代入并整理得到。证毕。
\end{proof}

\begin{theorem}[固定平均跳跃率约束下的熵极大可逆生成元]\label{thm:pom-diagonal-rate-critical-continuous-time-generator-maxent}
\leanverified{paper\_pom\_diagonal\_rate\_critical\_continuous\_time\_generator\_maxent}
令 \(X\) 为有限集，取全支撑分布 \(w\in\Delta(X)\)。
记 \(\mathcal Q_{\mathrm{rev}}(w)\) 为所有以 \(w\) 为平稳分布的可逆连续时间生成元集合。
给定归一化约束
\[
\sum_{x\in X}w(x)\sum_{y\ne x}Q_{xy}=1,
\]
在 \(\mathcal Q_{\mathrm{rev}}(w)\) 上最大化泛函
\[
\mathsf H(Q):=-\sum_{x\in X}w(x)\sum_{y\ne x}Q_{xy}\log Q_{xy}
\]
的唯一解恰为推论 \ref{cor:pom-diagonal-rate-critical-continuous-time-generator} 的 \(Q_w\)。
\end{theorem}
\begin{proof}
对任意 \(Q\in\mathcal Q_{\mathrm{rev}}(w)\) 令对称电导 \(c_{xy}:=w(x)Q_{xy}=w(y)Q_{yx}\)（\(x\ne y\)）。
则约束化为
\(
\sum_{x\ne y}c_{xy}=2\sum_{x<y}c_{xy}=1
\)。
并且
\[
\mathsf H(Q)
=-\sum_{x\ne y}c_{xy}\log\frac{c_{xy}}{w(x)}
=-\sum_{x<y}c_{xy}\Bigl(\log\frac{c_{xy}}{w(x)}+\log\frac{c_{xy}}{w(y)}\Bigr)
=-\sum_{x<y}2c_{xy}\log c_{xy}+\sum_{x<y}c_{xy}\bigl(\log w(x)+\log w(y)\bigr).
\]
右端关于 \((c_{xy})_{x<y}\) 为严格凹函数，且约束为线性，故最大化器唯一。
对拉格朗日函数求驻点，得到对每个 \(x<y\),
\[
-2(\log c_{xy}+1)+\log w(x)+\log w(y)-2\lambda=0,
\]
从而存在常数 \(\alpha>0\) 使
\(
c_{xy}=\alpha\sqrt{w(x)w(y)}
\)。
由 \(\sum_{x\ne y}c_{xy}=1\) 与恒等式
\(
\sum_{x<y}\sqrt{w(x)w(y)}=(A^2-1)/2=C/2
\)
可得 \(\alpha=1/C\)，因此
\(
Q_{xy}=c_{xy}/w(x)=\frac{1}{C}\sqrt{\frac{w(y)}{w(x)}}
\)
并与推论 \ref{cor:pom-diagonal-rate-critical-continuous-time-generator} 的 \(Q_w\) 一致。证毕。
\end{proof}

\begin{corollary}[割闭式与可审计的 Cheeger 上界]\label{cor:pom-diagonal-rate-critical-cheeger-auditable-upper-bound}
沿用推论 \ref{cor:pom-diagonal-rate-critical-continuous-time-generator}，令 \(c_{xy}:=w(x)(Q_w)_{xy}\)。
对任意 \(\varnothing\ne S\subsetneq X\) 定义
\[
\mathrm{vol}(S):=\sum_{x\in S}w(x),\qquad
\mathrm{cut}(S):=\sum_{x\in S,\,y\notin S}c_{xy}.
\]
则成立闭式
\[
\mathrm{cut}(S)=\frac{1}{C}\Bigl(\sum_{x\in S}\sqrt{w(x)}\Bigr)\Bigl(\sum_{y\notin S}\sqrt{w(y)}\Bigr).
\]
令（等价的）Cheeger 常数定义为
\[
\Phi(w):=\min_{\varnothing\ne S\subsetneq X}\frac{\mathrm{cut}(S)}{\min\{\mathrm{vol}(S),\mathrm{vol}(S^c)\}}.
\]
取任意排序 \(w_{(1)}\ge\cdots\ge w_{(n)}\) 并令 \(S_k:=\{(1),\dots,(k)\}\)。
定义扫掠证书
\[
\Phi_{\downarrow}(w):=\min_{1\le k\le n-1}\frac{\mathrm{cut}(S_k)}{\min\{\mathrm{vol}(S_k),\mathrm{vol}(S_k^c)\}},
\]
则有 \(\Phi(w)\le \Phi_{\downarrow}(w)\)。
记 \(\mu_2\) 为 \(L_w\) 的第二特征值（定理 \ref{thm:pom-diagonal-rate-critical-line-laplacian-first-order}），则由可逆链的标准 Cheeger 上界得到
\[
\mu_2\le 2\Phi(w)\le 2\Phi_{\downarrow}(w),
\]
其中 \(\Phi_{\downarrow}(w)\) 可在对 \(w\) 排序后以 \(O(n)\) 时间显式计算。
\end{corollary}
\begin{proof}
割闭式由
\(
c_{xy}=\frac{1}{C}\sqrt{w(x)w(y)}
\)
直接求和得到。
由定义，\(\Phi(w)\) 为对所有非平凡子集的最小值，故将候选集限制为 \(\{S_k\}_{k=1}^{n-1}\) 只能增大最小值，从而 \(\Phi(w)\le \Phi_{\downarrow}(w)\)。
最后一式为经典 Cheeger 不等式的上界方向（对可逆连续时间链的谱隙），与 \(\Phi(w)\le \Phi_{\downarrow}(w)\) 联立即得。
\(\Phi_{\downarrow}(w)\) 的计算只需一次排序并线性扫描前缀和。证毕。
\end{proof}
