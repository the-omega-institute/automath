\subsubsection{纤维奇偶—同伦等价与非可缩顶层相图}\label{subsubsec:pom-fiber-parity-homotopy-noncontractible-phase}

\begin{theorem}[纤维奇偶与同伦型的严格等价]\label{thm:pom-fiber-parity-homotopy-equivalence}
\leanverified{paper\_pom\_fiber\_parity\_homotopy\_equivalence}
对任意 \(m\ge 1\) 与 \(x\in X_m\)，设
\[
\Gamma_x\cong\bigsqcup_{i=1}^{r}P_{\ell_i},
\qquad
d_m(x)=\prod_{i=1}^{r}F_{\ell_i+2},
\qquad
K_x:=\mathrm{Ind}^{\Delta}(\Gamma_x),
\]
并记
\(
\tau(x):=\sum_{i=1}^{r}\lfloor(\ell_i+1)/3\rfloor
\)。
则有严格等价：
\begin{enumerate}
  \item \(|K_x|\) 可缩当且仅当 \(d_m(x)\) 为偶数；
  \item \(|K_x|\simeq S^{\tau(x)-1}\) 当且仅当 \(d_m(x)\) 为奇数。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-independence-complex-classification}，
\[
|K_x|\ \text{可缩}
\Longleftrightarrow
\exists i,\ \ell_i\equiv 1\pmod 3,
\]
且
\[
|K_x|\simeq S^{\tau(x)-1}
\Longleftrightarrow
\forall i,\ \ell_i\not\equiv 1\pmod 3.
\]
由推论 \ref{cor:pom-fiber-parity-mod3}，
\[
d_m(x)\equiv 0\pmod 2
\Longleftrightarrow
\exists i,\ \ell_i\equiv 1\pmod 3,
\]
以及
\[
d_m(x)\equiv 1\pmod 2
\Longleftrightarrow
\forall i,\ \ell_i\not\equiv 1\pmod 3.
\]
将两组等价直接拼合即得结论。
\end{proof}

\begin{definition}[非可缩最大纤维]\label{def:pom-max-noncontractible-fiber}
令
\[
\widetilde D_m
:=
\max\{d_m(x):x\in X_m,\ |K_x|\not\simeq\ast\}.
\]
由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，等价地
\[
\widetilde D_m
=
\max\{d_m(x):x\in X_m,\ d_m(x)\ \text{为奇数}\}.
\]
\end{definition}

\begin{theorem}[最大非可缩纤维的前三层谱封闭与模 \texorpdfstring{$6$}{6} 相图]\label{thm:pom-max-noncontractible-fiber-mod6-phase}
\leanverified{paper\_pom\_max\_noncontractible\_fiber\_mod6\_phase}
对所有 \(m\ge 17\)，记
\(
D_m=D_m^{(1)}>D_m^{(2)}>D_m^{(3)}>\cdots
\)
为定义 \ref{def:pom-top-fiber-spectrum} 的顶层纤维谱。
则
\[
\widetilde D_m=
\begin{cases}
D_m, & m\equiv 0,4\pmod 6,\\[2pt]
D_m^{(2)}, & m\equiv 1,5\pmod 6,\\[2pt]
D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]
\end{theorem}
\begin{proof}
由定义 \ref{def:pom-max-noncontractible-fiber}，\(\widetilde D_m\) 是“奇数纤维值”的最大元。
因此只需在顶层谱 \(D_m,D_m^{(2)},D_m^{(3)}\) 中判定奇偶。

\textbf{(i) 偶数支 \(m=2k\)。}
由定理 \ref{thm:pom-max-fiber}，
\(
D_{2k}=F_{k+2}
\)；
由定理 \ref{thm:pom-second-max-fiber-closed-form} 与定理 \ref{thm:pom-third-max-fiber-even-closed-form}，
\[
D_{2k}^{(2)}=F_{k+2}-F_{k-4},
\qquad
D_{2k}^{(3)}=F_{k+2}-F_{k-3}.
\]
由 Fibonacci 奇偶律（推论 \ref{cor:pom-fiber-parity-mod3} 的证明中 \(F_n\equiv 0\pmod 2\iff n\equiv 0\pmod 3\)）：
\begin{itemize}
  \item 若 \(k\equiv 0,2\pmod 3\)，则 \(k+2\not\equiv 0\pmod 3\)，故 \(D_{2k}\) 为奇数，遂 \(\widetilde D_{2k}=D_{2k}\)；
  \item 若 \(k\equiv 1\pmod 3\)，则 \(F_{k+2},F_{k-4}\) 皆偶，故 \(D_{2k}^{(2)}\) 为偶；而 \(k-3\equiv 1\pmod 3\)，\(F_{k-3}\) 为奇，故 \(D_{2k}^{(3)}\) 为奇。
\end{itemize}
结合 \(D_{2k}>D_{2k}^{(2)}>D_{2k}^{(3)}\) 得
\[
\widetilde D_{2k}=
\begin{cases}
D_{2k}, & k\equiv 0,2\pmod 3,\\
D_{2k}^{(3)}, & k\equiv 1\pmod 3,
\end{cases}
\]
等价于
\(
m\equiv 0,4\pmod 6
\)
对应 \(D_m\)，
\(
m\equiv 2\pmod 6
\)
对应 \(D_m^{(3)}\)。

\textbf{(ii) 奇数支 \(m=2k+1\)。}
由定理 \ref{thm:pom-max-fiber}，
\(
D_{2k+1}=2F_{k+1}
\)
恒为偶数。
由定理 \ref{thm:pom-second-max-fiber-closed-form}，
\[
D_{2k+1}^{(2)}=2F_{k+1}-F_{k-4},
\]
故 \(D_{2k+1}^{(2)}\) 为奇当且仅当 \(F_{k-4}\) 为奇，当且仅当 \(k\not\equiv 1\pmod 3\)。
若 \(k\equiv 1\pmod 3\)，则 \(D_{2k+1}^{(2)}\) 为偶。此时由注
\ref{rem:pom-third-max-fiber-odd-closed-form}（该分支首个指标 \(k=10\) 已满足其稳定阈值）
\[
D_{2k+1}^{(3)}=2F_{k+1}-(F_{k-4}+F_{k-8}),
\]
其中 \(F_{k-4}\) 为偶而 \(F_{k-8}\) 为奇，故 \(D_{2k+1}^{(3)}\) 为奇。
再由 \(D_{2k+1}>D_{2k+1}^{(2)}>D_{2k+1}^{(3)}\) 得
\[
\widetilde D_{2k+1}=
\begin{cases}
D_{2k+1}^{(2)}, & k\equiv 0,2\pmod 3,\\
D_{2k+1}^{(3)}, & k\equiv 1\pmod 3.
\end{cases}
\]
换算为 \(m=2k+1\) 的同余类即
\(
m\equiv 1,5\pmod 6
\)
对应 \(D_m^{(2)}\)，
\(
m\equiv 3\pmod 6
\)
对应 \(D_m^{(3)}\)。

合并偶、奇两支即得整体验证。
\end{proof}

\begin{theorem}[非可缩最大纤维的指数率守恒与模 \texorpdfstring{$6$}{6} 周期常数比]\label{thm:pom-max-noncontractible-fiber-exponent-mod6-ratio}
\leanpartial{paper\_pom\_max\_noncontractible\_fiber\_exponent\_mod6\_ratio}{the requested Lean signature does not assume enough to derive the global limit for \(\log \widetilde D_m / m\); exponential counterexamples with residue-class ratio limit \(0\) remain possible}
沿用定义 \ref{def:pom-max-noncontractible-fiber} 及定理
\ref{thm:pom-max-noncontractible-fiber-mod6-phase} 的记号。则
\[
\lim_{m\to\infty}\frac{1}{m}\log \widetilde D_m
=
\lim_{m\to\infty}\frac{1}{m}\log D_m
=
\frac{1}{2}\log\varphi.
\]
并且沿各模 \(6\) 剩余类有精确极限
\[
\lim_{\substack{m\to\infty\\ m\equiv 0,4\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1,
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 1,5\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1-\frac{1}{2\varphi^5},
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 2\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1-\varphi^{-5},
\qquad
\lim_{\substack{m\to\infty\\ m\equiv 3\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1-\frac{\varphi^{-5}+\varphi^{-9}}{2}.
\]
因而非可缩约束仅引入模 \(6\) 周期的常数折减，而不改变主指数率。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-max-fiber}，
\[
D_{2k}=F_{k+2},
\qquad
D_{2k+1}=2F_{k+1}.
\]
再由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，
\[
\widetilde D_m=
\begin{cases}
D_m, & m\equiv 0,4\pmod 6,\\[2pt]
D_m^{(2)}, & m\equiv 1,5\pmod 6,\\[2pt]
D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]

先看四个剩余类上的比值。若 \(m=2k\equiv 0,4\pmod 6\)，则
\(
\widetilde D_m=D_m=F_{k+2}
\)，故比值恒等于 \(1\)。
若 \(m=2k+1\equiv 1,5\pmod 6\)，则由定理 \ref{thm:pom-second-max-fiber-closed-form}
\[
\frac{\widetilde D_m}{D_m}
=
\frac{2F_{k+1}-F_{k-4}}{2F_{k+1}}.
\]
若 \(m=2k\equiv 2\pmod 6\)，则由定理 \ref{thm:pom-third-max-fiber-even-closed-form}
\[
\frac{\widetilde D_m}{D_m}
=
\frac{F_{k+2}-F_{k-3}}{F_{k+2}}.
\]
若 \(m=2k+1\equiv 3\pmod 6\)，则由注 \ref{rem:pom-third-max-fiber-odd-closed-form}
\[
\frac{\widetilde D_m}{D_m}
=
\frac{2F_{k+1}-F_{k-4}-F_{k-8}}{2F_{k+1}}.
\]
对上述三式应用 Binet 公式 \(F_n\sim \varphi^n/\sqrt5\)，分别得到
\[
\frac{F_{k-4}}{F_{k+1}}\to \varphi^{-5},
\qquad
\frac{F_{k-3}}{F_{k+2}}\to \varphi^{-5},
\qquad
\frac{F_{k-8}}{F_{k+1}}\to \varphi^{-9},
\]
于是四个剩余类上的极限常数恰如题述。

再看指数率。由定理 \ref{thm:pom-max-fiber} 的闭式，偶、奇两支都满足
\[
\frac{1}{m}\log D_m\longrightarrow \frac{1}{2}\log\varphi.
\]
另一方面，前述四个比值极限都严格位于 \((0,1]\) 中，故存在常数 \(c>0\) 与
\(m_0\)，使得对所有 \(m\ge m_0\) 都有
\[
c\le \frac{\widetilde D_m}{D_m}\le 1.
\]
因此
\[
\frac{1}{m}\log \widetilde D_m
=
\frac{1}{m}\log D_m
+\frac{1}{m}\log\!\left(\frac{\widetilde D_m}{D_m}\right)
\longrightarrow \frac{1}{2}\log\varphi.
\]
故 \(\widetilde D_m\) 与 \(D_m\) 共享同一主指数率，而差异完全体现在模 \(6\) 周期的常数层。证毕。
\end{proof}
