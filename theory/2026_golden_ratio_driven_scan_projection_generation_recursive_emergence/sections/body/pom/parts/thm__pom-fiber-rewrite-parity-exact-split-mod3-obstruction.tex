\begin{theorem}[纤维内逆代换次数的精确奇偶剖分与 \texorpdfstring{$\bmod 3$}{mod 3} 阻碍刻画]\label{thm:pom-fiber-rewrite-parity-exact-split-mod3-obstruction}
固定 \(m\ge 1\) 与 \(x\in X_m\)，并令纤维生成函数
\[
Z_x(t):=\sum_{\omega\in \Fold_m^{-1}(x)} t^{r(\omega)},
\qquad
d_m(x):=\card{\Fold_m^{-1}(x)}=Z_x(1),
\qquad
s_m(x):=Z_x(-1).
\]
定义纤维内的偶/奇计数
\[
A_x:=\card{\{\omega\in\Fold_m^{-1}(x):\ r(\omega)\equiv 0\ (\bmod\ 2)\}},
\qquad
B_x:=\card{\{\omega\in\Fold_m^{-1}(x):\ r(\omega)\equiv 1\ (\bmod\ 2)\}}.
\]
则恒有闭式
\[
A_x=\frac{d_m(x)+s_m(x)}{2},
\qquad
B_x=\frac{d_m(x)-s_m(x)}{2}.
\]
从而在纤维均匀测度下（定义 \ref{def:pom-fiber-rewrite-rv}），奇偶偏差满足精确恒等式
\[
\PP(R_x\equiv 0\ (\bmod\ 2)\mid x)-\PP(R_x\equiv 1\ (\bmod\ 2)\mid x)=\frac{s_m(x)}{d_m(x)}.
\]
特别地，
\[
R_x\ (\bmod\ 2)\ \text{严格均匀}
\quad\Longleftrightarrow\quad
s_m(x)=0
\quad\Longleftrightarrow\quad
\exists\,\text{连通分量 }P_{\ell}\subseteq \Gamma_x\ \text{满足}\ \ell\equiv 1\ (\bmod\ 3).
\]
\leanverified{parity\_split\_identity}
\leanverified{parity\_split\_identity\_odd}
\leanverified{paper\_pom\_fiber\_rewrite\_parity\_exact\_split\_mod3\_obstruction}
\end{theorem}

\begin{proof}
由定义，
\[
d_m(x)=\sum_{\omega\in\Fold_m^{-1}(x)}1=A_x+B_x,
\qquad
s_m(x)=\sum_{\omega\in\Fold_m^{-1}(x)}(-1)^{r(\omega)}=A_x-B_x.
\]
联立解得 \(A_x=(d_m(x)+s_m(x))/2\)、\(B_x=(d_m(x)-s_m(x))/2\)。
除以 \(d_m(x)\) 即得概率偏差恒等式。
\par
最后的等价性由推论 \ref{cor:pom-fiber-zx-minus-one-psi3-sign} 给出：
\(
s_m(x)=0
\Longleftrightarrow
m_3(x)\ge 1
\Longleftrightarrow
\exists i,\ \ell_i\equiv 1\ (\bmod\ 3)
\)。
证毕。
\end{proof}
