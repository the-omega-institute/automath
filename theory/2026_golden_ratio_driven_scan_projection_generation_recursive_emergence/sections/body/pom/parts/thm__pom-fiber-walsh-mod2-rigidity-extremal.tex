\begin{theorem}[纤维 Walsh 模式的分量乘法分解、模 \(2\) 刚性与极值判别]\label{thm:pom-fiber-walsh-mod2-rigidity-extremal}
\leanverified{paper\_pom\_fiber\_walsh\_mod2\_rigidity\_extremal}
固定 \(x\in X_m\) 并取定理 \ref{thm:pom-fiber-indset-factorization} 的分量分解
\(
\Gamma_x\simeq\bigsqcup_{i=1}^{r}P_{\ell_i}
\)。
对任意 \(I\subseteq[m]\)，沿用定理 \ref{thm:pom-fiber-walsh-signature-rademacher} 的记号，记
\(
W_x(I)=\sum_{\omega\in\Fold_m^{-1}(x)}\chi_I(\omega)
\)
与签名向量 \(\sigma_x(I)\in\{\pm1\}^{V(\Gamma_x)}\)。
对路径 \(P_\ell\) 及任意符号序列 \(s=(s_1,\dots,s_\ell)\in\{\pm1\}^{\ell}\) 定义带符号独立集和
\[
\mathcal Z_\ell(s):=\sum_{S\in\mathsf{Ind}(P_\ell)}\ \prod_{i\in S}s_i\ \in\ \ZZ,
\qquad(\ell\ge 0),
\]
并约定 \(\mathcal Z_0(\varnothing)=1\)。
则：
\begin{enumerate}
  \item \textbf{（分量乘法分解）}成立严格因子化
  \[
  W_x(I)=(\varepsilon^I)^x \prod_{i=1}^{r}\mathcal Z_{\ell_i}\!\bigl(\sigma_x(I)\big|_{P_{\ell_i}}\bigr).
  \]
  \item \textbf{（模 \(2\) 刚性）}对任意 \(\ell\ge 0\) 与任意 \(s\in\{\pm1\}^{\ell}\)，成立同余
  \[
  \mathcal Z_\ell(s)\equiv F_{\ell+2}\pmod 2.
  \]
  因而对任意 \(I\subseteq[m]\)，有
  \[
  W_x(I)\equiv d_m(x)\pmod 2.
  \]
  \item \textbf{（极值判别）}有充要条件
  \[
  |W_x(I)|=d_m(x)
  \quad\Longleftrightarrow\quad
  \sigma_x(I)_v=+1\ \ (\forall\,v\in V(\Gamma_x)),
  \]
  等价地
  \(
  |I\cap J_x(v)|\equiv 0\pmod 2
  \)
  对所有 \(v\) 成立。
  在此情形下有精确等式 \(W_x(I)=(\varepsilon^I)^x\,d_m(x)\)。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{分量乘法分解。}
由定理 \ref{thm:pom-fiber-walsh-signature-rademacher}(1)，
\[
W_x(I)=(\varepsilon^I)^x\sum_{S\in\mathsf{Ind}(\Gamma_x)}\ \prod_{v\in S}\sigma_x(I)_v.
\]
当 \(\Gamma_x=\bigsqcup_{i=1}^{r}P_{\ell_i}\) 时，独立集族分解为笛卡尔乘积
\(
\mathsf{Ind}(\Gamma_x)\cong\prod_{i=1}^{r}\mathsf{Ind}(P_{\ell_i})
\)，
并且权因子按分量相乘，故求和严格因子化为所示乘积。
\par
\textbf{模 \(2\) 刚性。}
固定 \(\ell\ge 0\) 与 \(s=(s_1,\dots,s_\ell)\)。
按末端顶点 \(\ell\) 是否被选入独立集分类，得到递推
\[
\mathcal Z_0(\varnothing)=1,\qquad
\mathcal Z_1(s_1)=1+s_1,\qquad
\mathcal Z_\ell(s)=\mathcal Z_{\ell-1}(s_1,\dots,s_{\ell-1})+s_\ell\,\mathcal Z_{\ell-2}(s_1,\dots,s_{\ell-2})\quad(\ell\ge 2).
\]
在模 \(2\) 下有 \(s_\ell\equiv 1\)，且 \(1+s_1\equiv 0\)，故
\(
\mathcal Z_0\equiv 1,\ \mathcal Z_1\equiv 0
\)
并满足 Fibonacci 递推，从而 \(\mathcal Z_\ell(s)\equiv F_{\ell+2}\pmod 2\)。
代入分量乘法分解并注意 \((\varepsilon^I)^x\equiv 1\pmod 2\)，得
\[
W_x(I)\equiv \prod_{i=1}^{r}F_{\ell_i+2}\equiv d_m(x)\pmod 2,
\]
其中最后一步用定理 \ref{thm:pom-fiber-indset-factorization} 的 \(d_m(x)=\prod_i F_{\ell_i+2}\)。
\par
\textbf{极值判别。}
记
\[
S(\sigma):=\sum_{S\in\mathsf{Ind}(\Gamma_x)}\ \prod_{v\in S}\sigma_v,
\qquad \sigma\in\{\pm1\}^{V(\Gamma_x)}.
\]
由三角不等式，
\(
|S(\sigma)|\le \sum_{S\in\mathsf{Ind}(\Gamma_x)}1=d_m(x)
\)，
且等号成立当且仅当所有项 \(\prod_{v\in S}\sigma_v\) 同号。
由于空集项恒为 \(+1\)，若存在 \(v_0\) 使 \(\sigma_{v_0}=-1\)，则单点独立集 \(\{v_0\}\in\mathsf{Ind}(\Gamma_x)\) 的项为 \(-1\)，从而出现异号项并使不等式严格。
反之若对所有 \(v\) 有 \(\sigma_v=+1\)，则每一项皆为 \(+1\)，从而 \(S(\sigma)=d_m(x)\)。
将 \(\sigma=\sigma_x(I)\) 与 \(W_x(I)=(\varepsilon^I)^x S(\sigma_x(I))\) 代回即得极值判别及精确等式。证毕。
\end{proof}
