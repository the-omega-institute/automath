\paragraph{对称可见层与次序可见层的分界}
前述结果已经给出同一纤维的两条内禀读出链：
一方面，开关群 \(\mathcal T_x\) 的不变算子代数
\(\End(\mathcal H_x)^{\mathcal T_x}\)
只保留置换对称压缩后的扇区信息；
另一方面，Birkhoff 表示下的纤维格
\(\mathcal L_x\cong J(\bigsqcup_i Z_{\ell_i})\)
则保留完整的分量长度谱。
下面把这两条可见层之间的严格分界写成若干独立命题。

\begin{theorem}[对称可见层与次序可见层的严格分裂]\label{thm:pom-fiber-symmetric-order-visible-layer-separation}
设 \(x,y\) 满足
\[
\Gamma_x\cong \bigsqcup_{i=1}^r P_{\ell_i},
\qquad
\Gamma_y\cong \bigsqcup_{i=1}^r P_{m_i},
\]
并假设多重集 \(\{\ell_i\}_{i=1}^r\) 与 \(\{m_i\}_{i=1}^r\) 不相等。
则有
\[
\End(\mathcal H_x)^{\mathcal T_x}
\cong
(\CC^2)^{\otimes r}
\cong
\End(\mathcal H_y)^{\mathcal T_y},
\]
但
\[
\mathcal L_x
\cong
J\!\Bigl(\bigsqcup_{i=1}^r Z_{\ell_i}\Bigr),
\qquad
\mathcal L_y
\cong
J\!\Bigl(\bigsqcup_{i=1}^r Z_{m_i}\Bigr),
\qquad
\mathcal L_x\ncong \mathcal L_y.
\]
因此，对称可见层只看见分量数 \(r\)，而次序可见层在格同构型意义下保留完整的长度多重集 \(\{\ell_i\}\)。
\end{theorem}
\leanverified{paper\_pom\_fiber\_symmetric\_order\_visible\_layer\_separation}

\begin{proof}
由定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor}，
\(\End(\mathcal H_x)^{\mathcal T_x}\) 与 \(\End(\mathcal H_y)^{\mathcal T_y}\) 都只依赖于分量数 \(r\)，故同构于 \((\CC^2)^{\otimes r}\)。
\par
另一方面，由定理 \ref{thm:pom-fiber-birkhoff-fence-ideal-lattice}，
\(\mathcal L_x\) 与 \(\mathcal L_y\) 分别是
\(P_x:=\bigsqcup_i Z_{\ell_i}\) 与 \(P_y:=\bigsqcup_i Z_{m_i}\) 的理想格。
有限分布式格的同构型由其 join-不可约元偏序唯一决定，而这里的 join-不可约元偏序恰为 \(P_x\) 与 \(P_y\)。
由于 \(P_x,P_y\) 的连通分量均为锯齿 \(Z_n\)，其分量大小恰恢复长度多重集；
既然 \(\{\ell_i\}\neq\{m_i\}\)，便有 \(P_x\ncong P_y\)，从而 \(\mathcal L_x\ncong\mathcal L_y\)。
\end{proof}

\begin{remark}[同一分量数下的最小分离样本]\label{rem:pom-fiber-symmetric-order-visible-layer-example}
取分量长度谱
\[
\{\ell_i\}=\{1,4\},
\qquad
\{m_i\}=\{2,3\}.
\]
此时 \(r=2\)，故由定理 \ref{thm:pom-fiber-symmetric-order-visible-layer-separation}，
两侧的对称可见层同为 \(\CC^2\otimes \CC^2\)。
然而纤维基数满足
\[
|F_x|=F_3F_6=16,
\qquad
|F_y|=F_4F_5=15.
\]
因而对应的 rank 多项式已在 \(q=1\) 处分离：
\(
R_x(1)\neq R_y(1)
\)；
等价地，格的 zeta 多项式也已在 \(k=2\) 处分离：
\(
Z_{\mathcal L_x}(2)\neq Z_{\mathcal L_y}(2)
\)。
\end{remark}

\begin{theorem}[体相极限下的调度熵普适律]\label{thm:pom-fiber-bulk-scheduling-entropy-universal}
设
\[
P_x=\bigsqcup_{i=1}^r Z_{\ell_i},
\qquad
N:=\sum_{i=1}^r \ell_i,
\]
并记
\[
\mathsf H_{\mathrm{sched}}(x):=\log |\mathrm{MC}(\mathcal L_x)|.
\]
若处于体相尺度，例如
\[
\ell_i\to\infty \ \text{整体增长},
\qquad\text{或更一般地}\qquad
\sum_{\ell_i\le L}\ell_i=o(N)\quad(\forall\,L<\infty),
\]
则有普适渐近律
\[
\mathsf H_{\mathrm{sched}}(x)
=
\log(N!)
+
N\log\frac{2}{\pi}
+
o(N).
\]
\leanverified{paper\_pom\_fiber\_bulk\_scheduling\_entropy\_universal}
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-fence-maxchains-euler}，
\[
\mathsf H_{\mathrm{sched}}(x)
=
\log(N!)
\;+\;
\sum_{i=1}^r \log\frac{E_{\ell_i}}{\ell_i!}.
\]
再由定理 \ref{thm:pom-fence-volume-log2pi}，
\[
\frac{1}{N}\sum_{i=1}^r \log\frac{E_{\ell_i}}{\ell_i!}
\longrightarrow
\log\frac{2}{\pi}.
\]
将其代回即得结论。
\end{proof}

\begin{corollary}[体相调度熵相对于全碎裂极值的线性赤字]\label{cor:pom-fiber-bulk-scheduling-deficit-linear}
在定理 \ref{thm:pom-fiber-bulk-scheduling-entropy-universal} 的设定下，
\[
\log(N!) - \mathsf H_{\mathrm{sched}}(x)
=
N\log\frac{\pi}{2}
+
o(N).
\]
\leanverified{paper\_pom\_fiber\_bulk\_scheduling\_deficit\_linear}
\end{corollary}

\begin{proof}
由定理 \ref{thm:pom-fiber-bulk-scheduling-entropy-universal} 直接移项即可。
\end{proof}

\begin{theorem}[布尔区间的单参数审计律]\label{thm:pom-fiber-boolean-interval-single-parameter-audit}
设 \([I,J]\subseteq \mathcal L_x\) 为任意区间，并令
\[
P_{I,J}:=J\setminus I
\]
带诱导偏序。
若 \(P_{I,J}\) 为一个含 \(q\) 个点的反链，则
\[
[I,J]\cong B_q,
\qquad
\mu_{\mathcal L_x}(I,J)=(-1)^q,
\qquad
|[I,J]|=2^q,
\qquad
Z_{[I,J]}(k)=k^q.
\]
反之，只要 \(\mu_{\mathcal L_x}(I,J)\neq 0\)，上述四条便同时成立，其中 \(q=|P_{I,J}|\)。
\leanverified{paper\_pom\_fiber\_boolean\_interval\_single\_parameter\_audit}
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-fence-interval-closure}，
\(
[I,J]\cong J(P_{I,J})
\)。
若 \(P_{I,J}\) 是 \(q\) 点反链，则
\(
J(P_{I,J})\cong B_q
\)，
故区间大小为 \(2^q\)，Möbius 数为 \((-1)^q\)。
再由定理 \ref{thm:pom-zeta-equals-order-poly}，
\[
Z_{[I,J]}(k)
=
Z_{J(P_{I,J})}(k)
=
\Omega_{P_{I,J}}(k).
\]
而 \(q\) 点反链的序多项式恰为 \(k^q\)。
\par
反向命题由定理 \ref{thm:pom-fence-mobius-rigidity} 立即给出：
\(\mu_{\mathcal L_x}(I,J)\neq 0\) 当且仅当 \(P_{I,J}\) 为反链。
\end{proof}

\begin{theorem}[固定对称可见层下的调度复杂度无界]\label{thm:pom-fiber-fixed-symmetric-layer-unbounded-scheduling}
\leanverified{paper\_pom\_fiber\_fixed\_symmetric\_layer\_unbounded\_scheduling}
固定 \(r\ge 1\)。
在分量数恰为 \(r\) 的谱类上，对称可见层恒退化为同一代数 \((\CC^2)^{\otimes r}\)；
然而次序可见层上的调度复杂度并无上界。
更具体地，对每个 \(N\ge r\) 定义
\[
P_{N,r}:=
Z_{N-r+1}\ \bigsqcup\ Z_1\ \bigsqcup\cdots\bigsqcup\ Z_1
\qquad
\text{（其中 \(Z_1\) 共出现 \(r-1\) 次）},
\]
并记
\(
\mathcal L_{N,r}:=J(P_{N,r})
\)。
则有渐近式
\[
\log |\mathrm{MC}(\mathcal L_{N,r})|
=
\log(N!)
+
N\log\frac{2}{\pi}
+
O_r(1).
\]
特别地，
\(
\log |\mathrm{MC}(\mathcal L_{N,r})|\to+\infty
\)；
因此固定 \(r\) 并不能控制全局调度复杂度。
\end{theorem}

\begin{proof}
分量数固定为 \(r\) 时，对称可见层只依赖于 \(r\)，这正是定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor} 的内容。
\par
对 \(\mathcal L_{N,r}\)，由定理 \ref{thm:pom-fence-maxchains-euler} 得
\[
|\mathrm{MC}(\mathcal L_{N,r})|
=
\binom{N}{N-r+1,1,\dots,1}E_{N-r+1}
=
\frac{N!}{(N-r+1)!}\,E_{N-r+1}.
\]
另一方面，\(\sec z+\tan z\) 在 \(z=\pm\pi/2\) 处的最近奇点支配给出
\[
\frac{E_n}{n!}=\Theta\!\left(\left(\frac{2}{\pi}\right)^n\right),
\]
从而
\[
\log\frac{E_n}{n!}
=
n\log\frac{2}{\pi}
+
O(1).
\]
取 \(n=N-r+1\) 并代回上式，得到
\[
\log |\mathrm{MC}(\mathcal L_{N,r})|
=
\log(N!)
+
(N-r+1)\log\frac{2}{\pi}
+
O(1)
=
\log(N!)
+
N\log\frac{2}{\pi}
+
O_r(1).
\]
证毕。
\end{proof}

\begin{corollary}[纯 switch-invariant operator sector 的审计不完备性]\label{cor:pom-fiber-toggle-invariant-audit-incomplete}
记
\[
\mathfrak A_x:=\End(\mathcal H_x)^{\mathcal T_x}.
\]
若某个纤维审计量 \(\mathcal O(x)\) 仅经由 \(\mathfrak A_x\) 因子化，即存在函数 \(F\) 使
\[
\mathcal O(x)=F(\mathfrak A_x),
\]
则在固定分量数的谱类上，\(\mathcal O\) 不可能完备恢复以下任一数据：
\[
\{\ell_i\},
\qquad
[\mathcal L_x],
\qquad
R_x(q),
\qquad
Z_{\mathcal L_x}(k),
\qquad
|\mathrm{MC}(\mathcal L_x)|,
\qquad
\mathsf H_{\mathrm{sched}}(x).
\]
换言之，若要获得完备的纤维审计，必须离开纯 switch-invariant operator sector，转向 Birkhoff 格与多链谱所定义的次序可见层。
\end{corollary}
\leanverified{paper\_pom\_fiber\_toggle\_invariant\_audit\_incomplete}

\begin{proof}
由定理 \ref{thm:pom-fiber-symmetric-order-visible-layer-separation}，
\(\mathfrak A_x\) 只依赖于分量数 \(r\)，因而不能恢复长度多重集 \(\{\ell_i\}\) 与格同构型 \([\mathcal L_x]\)。
由注 \ref{rem:pom-fiber-symmetric-order-visible-layer-example}，
同一 \(\mathfrak A_x\) 下既可出现不同的 \(|F_x|=|\mathcal L_x|=R_x(1)\)，也可出现不同的
\(
Z_{\mathcal L_x}(2)=|\mathcal L_x|
\)；
故 rank 多项式与 zeta 多项式都不可能仅由 \(\mathfrak A_x\) 决定。
\par
最后，定理 \ref{thm:pom-fiber-fixed-symmetric-layer-unbounded-scheduling} 表明：
在同一对称可见层 \((\CC^2)^{\otimes r}\) 下，
\(
|\mathrm{MC}(\mathcal L_x)|
\)
与
\(
\mathsf H_{\mathrm{sched}}(x)
\)
仍可沿谱序列无界增长。
因此任何仅依赖 \(\mathfrak A_x\) 的审计量都不可能对这些次序数据给出完备恢复。
\end{proof}
