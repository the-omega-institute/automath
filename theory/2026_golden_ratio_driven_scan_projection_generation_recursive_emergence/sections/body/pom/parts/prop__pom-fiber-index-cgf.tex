\begin{proposition}[碰撞谱的几何含义：纤维指数的指数矩母函数]\label{prop:pom-fiber-index-cgf}
沿用 \(\pi_m(x)=d_m(x)/2^m\) 的输出分布记号，并把 \(\gamma_m(X)\) 视为在 \(X_m\) 上以 \(\pi_m\) 抽样的随机变量。
则对任意整数 \(q\ge 2\) 有严格恒等式
\[
\boxed{\;
\mathbb{E}_{\pi_m}\!\left[\exp\!\bigl((q-1)m\,\gamma_m(X)\bigr)\right]
=\frac{S_q(m)}{2^m}\; }.
\]
从而
\[
\frac{1}{m}\log \mathbb{E}_{\pi_m}\!\left[\exp\!\bigl((q-1)m\,\gamma_m(X)\bigr)\right]
=\frac{1}{m}\log S_q(m)-\log 2.
\]
特别地，若存在 Perron 主尺度 \(S_q(m)=\Theta(r_q^m)\)（例如由 moment-kernel 实现给出），则极限累积母函数在整点 \(t=q-1\) 上满足
\[
\lim_{m\to\infty}\frac{1}{m}\log \mathbb{E}_{\pi_m}\!\left[e^{t m\,\gamma_m(X)}\right]
=\log r_{t+1}-\log 2,
\qquad
h_{t+1}=(t+1)\log 2-\log r_{t+1}.
\]
\leanverified{paper\_pom\_fiber\_index\_cgf\_q2\_specialized}
\leanverified{paper\_pom\_fiber\_index\_cgf\_q3\_specialized}
\leanverified{paper\_pom\_fiber\_index\_cgf\_q4\_specialized}
\leanverified{paper\_pom\_fiber\_index\_cgf\_general}
\leanverified{paper\_pom\_fiber\_index\_cgf\_expectation\_identity}
\leanverified{paper\_pom\_fiber\_index\_cgf\_package}
\end{proposition}
\begin{proof}
按定义展开并用 \(\pi_m(x)=d_m(x)/2^m\)：
\[
\begin{aligned}
\mathbb{E}_{\pi_m}\!\left[\exp\!\bigl((q-1)m\,\gamma_m(X)\bigr)\right]
&=\sum_{x\in X_m}\pi_m(x)\,e^{(q-1)\log d_m(x)}\\
&=\sum_{x\in X_m}\frac{d_m(x)}{2^m}\,d_m(x)^{q-1}\\
&=\frac{1}{2^m}\sum_{x\in X_m}d_m(x)^q\\
&=\frac{S_q(m)}{2^m}.
\end{aligned}
\]
取对数并除以 \(m\) 得第二式。最后一段由 \(S_q(m)=\Theta(r_q^m)\) 的指数尺度与 \(h_q=\log(2^q/r_q)\) 的定义直接推出。证毕。
\end{proof}

\begin{remark}[大偏差/多分形反演（接口 + 可执行构造）]\label{rem:pom-fiber-index-ldp}
命题 \ref{prop:pom-fiber-index-cgf} 表明：碰撞谱 \(q\mapsto h_q\) 等价于纤维指数 \(\gamma_m\) 的指数矩母函数在整点的增长率。
附录 \ref{app:pom-projective-pressure-operator} 给出一个具体的“整数 \(\to\) 实”提升：射影转移算子 \(\mathcal{L}_q\)（定义 \ref{def:pom-projective-transfer-operator}），并证明其在整数点 \(q\in\ZZ_{\ge 2}\) 上严格退化为对称幂 moment-kernel \(\widetilde A_q\)（定理 \ref{thm:pom-projective-operator-degenerates-to-moment-kernel}），从而把离散压力 \(\log r_q\) 对齐到一条连续压力曲线的可执行候选（注 \ref{rem:pom-projective-pressure-route}）。
在锥收缩/准紧性假设下，可取 \(\lambda(q)\) 为 \(\mathcal{L}_q\) 的 Perron 主特征值并定义凸函数
\(
\Lambda(t):=\log\lambda(t+1)-\log 2
\)
（对 Fold 主核即 \(|A_{\mathrm{out}}|=2\)），从而得到整点一致性
\(
\Lambda(q-1)=\log r_q-\log 2
\)。
若进一步满足可微/本质严格凸等标准条件，则 Gartner--Ellis 定理给出 \(\gamma_m\) 的大偏差率函数
\[
I(\gamma)=\sup_{t\ge 0}\bigl(t\gamma-\Lambda(t)\bigr).
\]
此时 \(I\)（从而纤维厚度谱）可由碰撞谱“反演”得到，构成“从 \(h_q\) 指纹回推纤维几何”的可证伪计算目标。
\end{remark}

\begin{remark}[碰撞谱 \texorpdfstring{$\leftrightarrow$}{<->} 典型损失：\texorpdfstring{$\Lambda(t)=\tau(t+1)-\log2$}{Lambda(t)=tau(t+1)-log2}]\label{rem:pom-lambda-tau-shift}

\[
S_q(m):=\sum_{x\in X_m} d_m(x)^q
\]
（当 \(q\in\mathbb{Z}_{\ge 2}\) 时与本文的碰撞矩一致）。
则对任意固定 \(m\) 与任意 \(t\in\mathbb{R}\)，由 \(\pi_m(x)=d_m(x)/2^m\) 直接展开得严格恒等式
\[
\mathbb{E}_{\pi_m}[e^{tL_m}]
=\sum_{x\in X_m}\frac{d_m(x)}{2^m}\,d_m(x)^t
=\frac{S_{t+1}(m)}{2^m}.
\]
因此若记
\[
\tau_m(q):=\frac1m\log S_q(m),
\qquad
\tau(q):=\lim_{m\to\infty}\tau_m(q)\quad(\text{在极限存在处}),

\[
\frac1m\log \mathbb{E}_{\pi_m}[e^{tL_m}]=\tau_m(t+1)-\log2,
\qquad
\Lambda(t)=\tau(t+1)-\log2.
\]
并且在 \(q=1\) 处（无需取极限）有精确导数恒等式
\[
\mu_m=\tau_m'(1),
\qquad
\sigma_m^2=\tau_m''(1).
\]
在可交换“取极限/求导”的假设下，上式取 \(m\to\infty\) 给出 \(\mu=\tau'(1)\) 与 \(\sigma^2=\tau''(1)\)。
\end{remark}

\begin{proposition}[Legendre 上界：谱由 moment 压力控制]\label{prop:pom-spectrum-legendre-upper}
\leanverified{paper\_pom\_spectrum\_legendre\_upper}
对任意 \(q\ge 2\) 与任意 \(\gamma\ge 0\)，有上界
$$
f(\gamma)\ \le\ \inf_{q\ge 2}\bigl(\tau(q)-q\gamma\bigr).
$$
特别地，在 \(\tau(q)=\log r_q\) 的情形下，
$$
f(\gamma)\ \le\ \inf_{q\ge 2}\bigl(\log r_q-q\gamma\bigr).
$$
\end{proposition}

\begin{proof}
固定 \(q\ge 2\) 与 \(\gamma,\varepsilon\)。对任意满足 \(|\gamma_m(x)-\gamma|\le\varepsilon\) 的 \(x\)，有
$$
d_m(x)^q=\exp(qm\gamma_m(x))\ \ge\ \exp\bigl(qm(\gamma-\varepsilon)\bigr).
$$
因此
$$
S_q(m)=\sum_{x\in X_m} d_m(x)^q
\ \ge\ N_m(\gamma,\varepsilon)\,\exp\bigl(qm(\gamma-\varepsilon)\bigr),
$$
即
$$
\frac{1}{m}\log N_m(\gamma,\varepsilon)\ \le\ \frac{1}{m}\log S_q(m)\ -\ q(\gamma-\varepsilon).
$$
取 \(m\to\infty\) 的 \(\limsup\) 并再令 \(\varepsilon\downarrow 0\)，得
$$
f(\gamma)\le \tau(q)-q\gamma.
$$
对所有 \(q\ge 2\) 取下确界即得结论。证毕。
\end{proof}

\begin{remark}[多重分形形式（作为可检验目标）]\label{rem:pom-spectrum-formalism}
命题 \ref{prop:pom-spectrum-legendre-upper} 给出严格的上界：moment 压力曲线 \(\tau(q)\) 的凸共轭控制了纤维多重度的（上）谱。
在标准热力学形式条件下（例如存在与 \(\tau\) 对偶的 Gibbs 结构并具有足够的准乘性），往往可得到相反方向的不等式从而实现等号，即“谱 \(f\) 与压力 \(\tau\) 互为 Legendre 变换”。
在本文框架中，这一等号可被视为一条可证伪的结构目标：它把“空间损失是一条谱”转化为“\(\{r_q\}\) 族决定谱形”的可计算命题。
\end{remark}

\begin{proposition}[Paley--Zygmund 下包络：仅由 \texorpdfstring{$(r_q,r_{2q})$}{(rq,r2q)} 给出“厚尾”指数下界]\label{prop:pom-spectrum-pz-lower}
\leanverified{paper\_pom\_spectrum\_pz\_lower}
取任意整数 \(q\ge 2\)。令 \(X\) 在 \(X_m\) 上均匀分布，并令非负随机变量
\[
Y:=d_m(X)^q.
\]
则对阈值 \(\theta=\tfrac12\) 的 Paley--Zygmund 不等式给出
\[
\#\Bigl\{x\in X_m:\ d_m(x)\ge \Bigl(\tfrac12\,\frac{S_q(m)}{|X_m|}\Bigr)^{1/q}\Bigr\}
\ \ge\ \frac14\cdot\frac{S_q(m)^2}{S_{2q}(m)}.
\]
从而若极限 \(r_k=\lim_{m\to\infty}S_k(m)^{1/m}\) 在 \(k=q,2q\) 处存在，并且 \(\lim_{m\to\infty}\frac1m\log|X_m|=\log\varphi\)，则定义
\[
\gamma_q:=\frac{\log r_q-\log\varphi}{q},
\qquad
\underline f(q):=2\log r_q-\log r_{2q},
\]
有严格下界
\[
\boxed{\;
\lim_{\varepsilon\downarrow 0}\ \liminf_{m\to\infty}\ \frac1m
\log \#\bigl\{x\in X_m:\ \gamma_m(x)\ge \gamma_q-\varepsilon\bigr\}
\ \ge\ \underline f(q)\; }.
\]
\end{proposition}
\begin{proof}
按均匀抽样 \(X\sim \mathrm{Unif}(X_m)\) 展开期望：
\[
\mathbb{E}[Y]=\frac{1}{|X_m|}\sum_{x\in X_m}d_m(x)^q=\frac{S_q(m)}{|X_m|},
\qquad
\mathbb{E}[Y^2]=\frac{S_{2q}(m)}{|X_m|}.
\]
对任意非负 \(Y\) 与 \(\theta\in(0,1)\)，Paley--Zygmund 不等式给出
\[
\mathbb{P}\!\left(Y\ge \theta\,\mathbb{E}Y\right)\ \ge\ (1-\theta)^2\frac{(\mathbb{E}Y)^2}{\mathbb{E}[Y^2]}.
\]
取 \(\theta=\tfrac12\) 并代入上式得到
\[
\mathbb{P}\!\left(Y\ge \tfrac12\,\mathbb{E}Y\right)\ \ge\ \frac14\cdot\frac{S_q(m)^2}{|X_m|\,S_{2q}(m)}.
\]
把概率乘以 \(|X_m|\) 即得到计数下界：
\[
\#\Bigl\{x\in X_m:\ d_m(x)^q\ge \tfrac12\,\frac{S_q(m)}{|X_m|}\Bigr\}
\ \ge\ \frac14\cdot\frac{S_q(m)^2}{S_{2q}(m)},
\]
等价改写即得命题第一条不等式。
\par
对第二条，取对数并除以 \(m\)，并令 \(m\to\infty\)：
右端给出
\[
\liminf_{m\to\infty}\frac1m\log\!\left(\frac14\cdot\frac{S_q(m)^2}{S_{2q}(m)}\right)
=2\log r_q-\log r_{2q}=\underline f(q),
\]
而左端的阈值满足
\[
\frac1m\log\Bigl(\Bigl(\tfrac12\,\frac{S_q(m)}{|X_m|}\Bigr)^{1/q}\Bigr)
=\frac{1}{q}\Bigl(\frac1m\log S_q(m)-\frac1m\log|X_m|+\frac1m\log\tfrac12\Bigr)
\to \gamma_q.
\]
因此对任意 \(\varepsilon>0\)，当 \(m\) 足够大时左端集合包含于 \(\{\gamma_m(x)\ge \gamma_q-\varepsilon\}\)，从而得到所述的 \(\varepsilon\downarrow 0\) 的 \(\liminf\) 下界。证毕。
\end{proof}

\begin{corollary}[间歇性缺口 \texorpdfstring{$\delta(q)$}{delta(q)}：从表格数值可直接审计的“尾部变薄”指标]\label{cor:pom-spectrum-intermittency-deficit}
在命题 \ref{prop:pom-spectrum-pz-lower} 的设定下，

\boxed{\;

其余陈述由命题 \ref{prop:pom-spectrum-pz-lower} 与 \(\gamma_q\) 的定义直接推出。证毕。
\end{proof}

\begin{proof}
由 \(\mathrm{Var}(\log d_m(X))=m\sigma_m^2\) 与 Chebyshev 得
\[
w_m\bigl(|\log d_m(X)-\mathbb{E}\log d_m(X)|>m\varepsilon\bigr)
\le \frac{m\sigma_m^2}{m^2\varepsilon^2}
=\frac{\sigma_m^2}{m\varepsilon^2},
\]
除以 \(m\) 即得到第一式。
\par
对第二式，注意对任意 \(x\in A_m(\varepsilon)\)，有
\(
\log d_m(x)\le m(\mu_m+\varepsilon)
\)，从而
\(
w_m(x)=d_m(x)/2^m\le \exp(m(\mu_m+\varepsilon))/2^m
\)。
于是
\[
w_m(A_m(\varepsilon))
\le |A_m(\varepsilon)|\cdot \frac{\exp(m(\mu_m+\varepsilon))}{2^m}.
\]
结合第一式的下界 \(w_m(A_m(\varepsilon))\ge 1-\sigma_m^2/(m\varepsilon^2)\) 并移项即得。证毕。
\end{proof}

\begin{remark}[同一个凸共轭的两种读出：\texorpdfstring{Legendre 谱上界 \(\leftrightarrow\) Chernoff 尾界}{Legendre<->Chernoff}]\label{rem:pom-legendre-chernoff-same-dual}
命题 \ref{prop:pom-spectrum-legendre-upper}（按计数测度的几何谱上界）与命题 \ref{prop:pom-quantile-chernoff}--推论 \ref{cor:pom-quantile-dual-optimizer}（按输出测度 \(\pi_m\) 的尾部控制）本质上由同一个凸对偶核给出。
令
\[
\Phi(q):=\tau(q)-\log 2,
\qquad
\Lambda(t):=\Phi(t+1)=\tau(t+1)-\log 2\quad(t\ge 1),
\]
则 Chernoff 率函数上界为
\[
I(\gamma)=\sup_{t\ge 1}\bigl(t\gamma-\Lambda(t)\bigr)
\]
（推论 \ref{cor:pom-quantile-dual-optimizer}）。
而 Legendre 上界可改写为
\[
\begin{aligned}
\inf_{q\ge 2}\bigl(\tau(q)-q\gamma\bigr)
&=\log 2-\gamma+\inf_{t\ge 1}\bigl(\Lambda(t)-t\gamma\bigr)\\
&=\log 2-\gamma-\sup_{t\ge 1}\bigl(t\gamma-\Lambda(t)\bigr)
=\log 2-\gamma-I(\gamma).
\end{aligned}
\]
因此二者共享同一个凸共轭 \(\sup_{t\ge 1}(t\gamma-\Lambda(t))\)：差别仅来自基准测度（计数 vs.\ \(\pi_m\)）引入的 \(-\log 2\) 归一化与 \(q\leftrightarrow t=q-1\) 的平移。
\par
在可微/严格凸的扩展口径下（例如把 \(q\) 视作实参数并延拓 \(\tau\)），最优矩阶 \(q^\star(\gamma)\)（无论用于谱上界还是用于尾界预算）满足同一条“支撑线条件”
\[
\gamma\ \in\ \partial \tau\bigl(q^\star(\gamma)\bigr),
\]
即 \(\tau\) 在 \(q^\star\) 处的次梯度命中 \(\gamma\)；这与压力的离散凸性（命题 \ref{prop:pom-pressure-convexity}）一起，把“选 \(q\)”从经验旋钮升级为由凸几何决定的对偶坐标。
\end{remark}

\subsubsection{在线延迟的不可约下界}\label{subsec:pom-delay-lb}
\begin{proposition}[单趟在线延迟 $3$ 的最小性]\label{prop:pom-delay-min}
任何实现 $\Fold_\infty:\{0,1,2\}^{\ZZ}\to X_\infty$ 的单趟在线转导器，其最小延迟满足
\[
\delta_{\min}(\Fold_\infty)\ge 3.
\]
结合定理 \ref{thm:online-delay-fold}，可得 $\delta_{\min}(\Fold_\infty)=3$。
\leanverified{paper\_pom\_delay\_min}
\end{proposition}
\begin{proof}
在小节 \ref{subsec:online-delay} 的在线核中，输出满足
\(
e=\lfloor \varphi R(s)+d\varphi^{-3}\rfloor
\)，其中 $s$ 为长度 $3$ 的同步状态、$d\in\{0,1,2\}$ 为当前输入。取状态 $s=002$（其属于状态集 $Q$），则
\[
R(s)=2\varphi^{-3},\qquad
e=\left\lfloor 2\varphi^{-2}+d\varphi^{-3}\right\rfloor.
\]
当 $d=0$ 时 $e=0$，当 $d=1$ 时 $e=1$，故该位输出确实依赖于“向前第 $3$ 位”输入。任何延迟 $2$ 的单趟在线机在输出此位时尚未读到该输入，必产生冲突，从而不可能实现 $\Fold_\infty$。证毕。
\end{proof}

\begin{remark}[空间缺陷与时间缺陷的正交性]\label{rem:pom-space-time-orth}
最大纤维多重度给出“值投影”的最坏空间缺陷（定理 \ref{thm:pom-max-fiber}），而单趟在线延迟给出“步骤投影”的不可约时间缺陷（命题 \ref{prop:pom-delay-min}）。两者来源于不同的投影门，因而互不替代：增大空间并不能把 $\delta_{\min}$ 压到 $2$，而压低延迟也无法消除 $d_m$ 的纤维多重度。
\end{remark}

\subsubsection{可复算 commuting package：真实输入 40 状态核}\label{subsec:real-input-40-commuting-package}
\paragraph{对象：真实输入的确定性 skew-product}
令黄金均值移位（禁词 $(11)$）的邻接矩阵
\[
A=\begin{pmatrix}1&1\\[2pt]1&0\end{pmatrix},\qquad \rho(A)=\varphi.
\]
把两条合法 Zeckendorf 输入轨道 $(x,y)\in X\times X$ 并入同步核状态，得到

（$|Q|=10,|\Sigma|=40$）。
每步读入 $(x_k,y_k)\in\{0,1\}^2$（受黄金均值守卫约束），令 $d_k=x_k+y_k\in\{0,1,2\}$，
并用 10 状态核的确定性转移 $s_{k+1}=\delta(s_k,d_k)$ 更新纤维坐标 $s\in Q$；

\begin{proposition}[确定性扩张不增熵：$\rho(M)=\varphi^2$]\label{prop:real-input-40-entropy}
真实输入 $40$ 状态核的拓扑熵满足
\[
h_{\mathrm{top}}(\Sigma)=h_{\mathrm{top}}(X\times X)=2\log\varphi,
\qquad
\rho(M)=e^{h_{\mathrm{top}}(\Sigma)}=\varphi^2.
\]
\end{proposition}
\begin{proof}
基系统为输入对语法 $X\times X$，其邻接矩阵为 $A\otimes A$，故 $h_{\mathrm{top}}(X\times X)=\log\rho(A\otimes A)=2\log\varphi$。
另一方面，给定任意基系统符号序列 $\{(x_k,y_k)\}$，则 $d_k=x_k+y_k$ 唯一确定，因而纤维更新 $s_{k+1}=\delta(s_k,d_k)$ 亦唯一确定。
因此对任意长度 $n$ 的基系统路径，从每个初始纤维态 $s_0$ 出发至多抬升出一条 $\Sigma$-路径，总路径数仅乘以常数 $|Q|$，指数增长率不变。
故 $h_{\mathrm{top}}(\Sigma)=h_{\mathrm{top}}(X\times X)$，从而 $\rho(M)=\varphi^2$。证毕。
\end{proof}
\leanverified{paper\_real\_input\_40\_entropy}
