\subsubsection{黄金互补二元信道的收缩常数刚性：TV 等距、最大相关与互信息 SDPI}\label{subsubsec:pom-golden-channel-contraction-sdpi}

本段把“隐藏位观测”视为一个固定二元信道，并把其收缩性质压缩为三类可审计常数：
(i) Dobrushin 系数（TV 收缩）；(ii) Hirschfeld--Gebelein--Rényi（HGR）最大相关；(iii) 互信息的最强数据处理不等式（SDPI）常数。
对黄金互补对 \(p=\varphi^{-1}\)、\(q=\varphi^{-2}\)（\(p-q=\varphi^{-3}\)）而言，这些常数发生刚性塌缩：TV/相关收缩因子为 \(\varphi^{-3}\)，互信息收缩因子为 \(\varphi^{-6}\)。

\begin{theorem}[黄金互补二元信道的收缩常数刚性：\texorpdfstring{$\delta=\rho_{\mathrm{m}}=\varphi^{-3}$}{delta=rho_m=phi^{-3}} 与 \texorpdfstring{$I$}{I}-收缩 \texorpdfstring{$\varphi^{-6}$}{phi^{-6}}]\label{thm:pom-golden-binary-channel-contraction}
令 \(\varphi=\frac{1+\sqrt5}{2}\)，并考虑二元输入 \(\Theta\in\{+,-\}\) 与二元输出 \(B\in\{0,1\}\) 的信道 \(W\)：
\[
W(B=1\mid \Theta=+)=\varphi^{-1},\qquad
W(B=1\mid \Theta=-)=\varphi^{-2}.
\]
则成立：
\begin{enumerate}
  \item \textbf{（TV 收缩/Dobrushin 系数）}
  \[
  \delta(W)
  :=\sup_{\theta\neq\theta'}\mathrm{TV}\bigl(W(\cdot\mid\theta),W(\cdot\mid\theta')\bigr)
  =|\varphi^{-1}-\varphi^{-2}|
  =\varphi^{-3}.
  \]
  \item \textbf{（HGR 最大相关）}Hirschfeld--Gebelein--Rényi 最大相关系数满足
  \[
  \rho_{\mathrm{m}}(W)=\varphi^{-3}.
  \]
  \item \textbf{（精确 TV 缩放）}对任意两输入先验 \(\mu,\nu\)（定义在 \(\{+,-\}\) 上）有严格恒等式
  \[
  \mathrm{TV}(\mu W,\nu W)=\varphi^{-3}\,\mathrm{TV}(\mu,\nu).
  \]
  \item \textbf{（互信息 SDPI，最优常数）}对任意马尔可夫链 \(U\to \Theta\to B\)（\(U\) 任意可测）有锐常数形式
  \[
  I(U;B)\le \varphi^{-6}\,I(U;\Theta),
  \]
  其中 \(\varphi^{-6}\) 为该信道的最优（不可改进）收缩因子，并满足
  \[
  \varphi^{-6}=\rho_{\mathrm{m}}(W)^2=\delta(W)^2.
  \]
\end{enumerate}
\leanverified{paper\_pom\_golden\_binary\_channel\_contraction}
\end{theorem}

\begin{proof}
设 \(p:=\varphi^{-1}\)、\(q:=\varphi^{-2}=1-p\)，并记 \(\Delta:=p-q=\varphi^{-3}\)。
\par
\textbf{(1)} 对二点分布，\(\mathrm{TV}(\mathrm{Bern}(p),\mathrm{Bern}(q))=|p-q|=|\Delta|\)，故 \(\delta(W)=|\Delta|=\varphi^{-3}\)。
\par
\textbf{(3)} 任取 \(\mu(+)=a\)、\(\nu(+)=b\)，则
\(
\mu W=\mathrm{Bern}(q+a\Delta)
\)、\(
\nu W=\mathrm{Bern}(q+b\Delta)
\)。
因此
\[
\mathrm{TV}(\mu W,\nu W)=|(q+a\Delta)-(q+b\Delta)|=|\Delta|\,|a-b|=\varphi^{-3}\,\mathrm{TV}(\mu,\nu).
\]
\par
\textbf{(2)} 令 \(X\in\{\pm 1\}\) 为 \(\Theta\) 的符号编码（\(X=+1\iff \Theta=+\)），并取 \(Y:=2B-1\in\{\pm 1\}\)。
则 \(\mathbb{E}[Y\mid X=+1]=2p-1=\Delta\)、\(\mathbb{E}[Y\mid X=-1]=2q-1=-\Delta\)，即
\(\mathbb{E}[Y\mid X]=\Delta X\)。
在均匀先验 \(\mathbb{P}(X=+1)=\mathbb{P}(X=-1)=\tfrac12\) 下，立得
\(\mathrm{Corr}(X,Y)=\Delta\)，从而按 HGR 系数对输入先验取上确界的定义有 \(\rho_{\mathrm{m}}(W)\ge|\Delta|\)。
另一方面，对二元对称信道可由 \(2\times 2\) 信道矩阵的奇异值计算得到 \(\rho_{\mathrm{m}}(W)=|2p-1|=|p-q|=|\Delta|\)。
故 \(\rho_{\mathrm{m}}(W)=|\Delta|=\varphi^{-3}\)。
\par
\textbf{(4)} 对任意马尔可夫链 \(U\to\Theta\to B\)，最强数据处理不等式给出
\(
I(U;B)\le \rho_{\mathrm{m}}(W)^2 I(U;\Theta)
\)。
代入 \(\rho_{\mathrm{m}}(W)=\varphi^{-3}\) 得
\(
I(U;B)\le \varphi^{-6} I(U;\Theta)
\)。
最优性可由二元对称信道的 SDPI 常数计算（等价地由小扰动极限下互信息二次型的收缩比）得到；该常数恰等于 \(\rho_{\mathrm{m}}(W)^2\)。证毕。
\end{proof}

\subsubsection{\texorpdfstring{$\varphi$}{phi}-logit 晶格：Bayes 更新在 \texorpdfstring{$\mathbb{Z}$}{Z} 上的平移动力学}\label{subsubsec:pom-phi-logit-lattice-dynamics}

\begin{corollary}[\texorpdfstring{$\varphi$}{phi}-logit 晶格：Bayes 更新在 \texorpdfstring{$\mathbb{Z}$}{Z} 上的平移动力学]\label{cor:pom-phi-logit-lattice-translation}
在命题 \ref{prop:pom-phi-logistic-posterior-grid} 的二相位模型中取等先验
\(\mathbb{P}(\Theta=\theta_+)=\mathbb{P}(\Theta=\theta_-)=\tfrac12\)。
定义后验 log-odds 过程
\[
Z_n:=\log_{\varphi}\frac{\mathbb{P}(\Theta=\theta_+\mid B_1^n)}{\mathbb{P}(\Theta=\theta_-\mid B_1^n)}.
\]
则 \(Z_n\in\mathbb{Z}\) 且存在整数值表示与平移递推：
\[
\boxed{\;Z_n=\sum_{i=1}^{n}(2B_i-1)\in\mathbb{Z},\qquad
Z_{n+1}=Z_n+(2B_{n+1}-1).\;}
\]
从而后验族为离散轨道
\[
\boxed{\;\frac{\mathbb{P}(\Theta=\theta_+\mid B_1^n)}{\mathbb{P}(\Theta=\theta_-\mid B_1^n)}=\varphi^{Z_n},\qquad
\mathbb{P}(\Theta=\theta_+\mid B_1^n)=\frac{1}{1+\varphi^{-Z_n}}.\;}
\]
因此推断过程在代数上等价于对基本单位 \(\varphi\) 的整数幂进行乘法随机游走；后验可达集合为 \(\varphi^{\mathbb{Z}}\) 的轨道像，呈现严格“算术可审计性”。
\end{corollary}

\begin{proof}
由命题 \ref{prop:pom-phi-logistic-posterior-grid}，
\(
\frac{\mathbb{P}(\theta_+\mid B_1^n)}{\mathbb{P}(\theta_-\mid B_1^n)}=\varphi^{K_n}
\)，其中
\(
K_n=\sum_{i=1}^n(2B_i-1)
\)。
故 \(Z_n=\log_\varphi(\text{odds})=K_n\in\mathbb{Z}\)，并立即推出递推与后验闭式。证毕。
\end{proof}

\subsubsection{对称双阈值停时的 Chebyshev 证书：母函数、独立性与有限谱分解}\label{subsubsec:pom-chebyshev-stopping-time-spectral}

\begin{theorem}[对称双阈值停时的概率母函数：Chebyshev 逆像闭式]\label{thm:pom-symmetric-double-threshold-pgf-chebyshev}
在定理 \ref{thm:pom-sprt-symmetric-threshold-error} 的设定下，令
\[
\tau_T:=\inf\{n\ge 0:\ |K_n|=T\}\qquad(T\in\mathbb{N}),
\]
并记 Chebyshev 第一类多项式 \(T_T(x)=\cos\!\bigl(T\arccos x\bigr)\)。
则在真相位为 \(\theta_+\) 时，停时 \(\tau_T\) 的概率母函数在 \(|u|\le 1\) 上满足严格闭式
\[
\boxed{\;
\mathbb{E}_{\theta_+}[u^{\tau_T}]
=\frac{\cosh\!\bigl(\tfrac{T}{2}\ln\varphi\bigr)}{T_T\!\bigl(\tfrac{\varphi^{3/2}}{2u}\bigr)}.\;}
\]
等价地，设
\[
\xi(u):=\frac{\varphi^{3/2}}{2u}
\;+\;\sqrt{\Bigl(\frac{\varphi^{3/2}}{2u}\Bigr)^2-1}\qquad(|u|\le 1),
\]
则
\[
\boxed{\;
\mathbb{E}_{\theta_+}[u^{\tau_T}]
=\frac{\varphi^{T/2}+\varphi^{-T/2}}{\xi(u)^T+\xi(u)^{-T}}.\;}
\]
\leanverified{paper\_pom\_symmetric\_double\_threshold\_pgf\_chebyshev}
\end{theorem}

\begin{proof}
令 \(p:=\mathbb{P}_{\theta_+}(Y_1=+1)=\varphi^{-1}\)、\(q:=\mathbb{P}_{\theta_+}(Y_1=-1)=\varphi^{-2}\)（其中 \(Y_i:=2B_i-1\)），并对 \(|k|<T\) 定义
\(
F_k(u):=\mathbb{E}_{\theta_+}[u^{\tau_T}\mid K_0=k]
\)。
则边界条件为 \(F_{\pm T}(u)=1\)，而对 \(|k|<T\) 由一步分解得到二阶线性差分方程
\[
F_k(u)=u\bigl(pF_{k+1}(u)+qF_{k-1}(u)\bigr).
\]
其通解可写为 \(F_k(u)=A r^k+B r^{-k}\)，其中 \(r=r(u)\) 满足特征方程
\(
pu r^2-r+qu=0
\)。
将该方程化为
\(
r+r^{-1}=\frac{1}{u\sqrt{pq}}=\frac{\varphi^{3/2}}{u}
\)
并取
\(
\xi(u):=\frac{\varphi^{3/2}}{2u}+\sqrt{\bigl(\frac{\varphi^{3/2}}{2u}\bigr)^2-1}
\)
可得两根为
\(
r_\pm=\sqrt{q/p}\,\xi(u)^{\pm 1}
\)。
代入边界条件并取 \(K_0=0\)（对称初始）解得
\[
F_0(u)=\frac{\cosh(\tfrac{T}{2}\ln(p/q))}{\frac12(\xi(u)^T+\xi(u)^{-T})}
=\frac{\varphi^{T/2}+\varphi^{-T/2}}{\xi(u)^T+\xi(u)^{-T}},
\]
其中用到 \(p/q=\varphi\)。再用 Chebyshev 恒等式
\(
T_T(x)=\frac12\bigl(\xi^T+\xi^{-T}\bigr)
\)
（当 \(x=\frac12(\xi+\xi^{-1})\)）并注意
\(
\frac12(\xi(u)+\xi(u)^{-1})=\frac{\varphi^{3/2}}{2u}
\)
即可得到第一种写法。证毕。
\end{proof}

\begin{proposition}[联合母函数的完全分离：\texorpdfstring{$(\tau_T,K_{\tau_T})$}{(tauT,KtauT)} 的独立性与常比率律]\label{prop:pom-tau-terminal-sign-independence}
\leanverified{paper\_pom\_tau\_terminal\_sign\_independence}
在定理 \ref{thm:pom-symmetric-double-threshold-pgf-chebyshev} 的设定下，
对任意满足 \(n\equiv T\pmod 2\) 的 \(n\ge T\)，有严格常比率恒等式
\[
\boxed{\;
\frac{\mathbb{P}_{\theta_+}(\tau_T=n,\ K_{\tau_T}=+T)}{\mathbb{P}_{\theta_+}(\tau_T=n,\ K_{\tau_T}=-T)}=\varphi^{T}\; }
\]
且该比率与 \(n\) 无关。
因此随机变量 \(\tau_T\) 与终端符号
\(
S_T:=\mathrm{sign}(K_{\tau_T})\in\{+,-\}
\)
\textbf{相互独立}，并且联合概率母函数严格分离为
\[
\boxed{\;
\mathbb{E}_{\theta_+}\!\left[u^{\tau_T}w^{K_{\tau_T}}\right]
=\mathbb{E}_{\theta_+}[u^{\tau_T}]\cdot\Bigl(\pi_T\,w^{T}+(1-\pi_T)\,w^{-T}\Bigr),\;}
\]
其中
\(
\pi_T:=\mathbb{P}_{\theta_+}(K_{\tau_T}=+T)=\frac{\varphi^{T}}{1+\varphi^{T}}
\)
（见定理 \ref{thm:pom-sprt-symmetric-threshold-error}）。
特别地，对任意可积函数 \(f\) 有
\(
\mathbb{E}_{\theta_+}[f(\tau_T)\mid S_T]=\mathbb{E}_{\theta_+}[f(\tau_T)]
\)。
\end{proposition}

\begin{proof}
固定 \(n\equiv T\ (\mathrm{mod}\ 2)\) 且 \(n\ge T\)。
考虑所有从 \(0\) 出发在时刻 \(n\) 首次到达 \(+T\) 的路径集合 \(\mathcal{P}_{n}^{+}\)，以及对应的首次到达 \(-T\) 的路径集合 \(\mathcal{P}_{n}^{-}\)。
将每条路径的步进序列逐步取反（\(+1\leftrightarrow -1\)）给出一个双射 \(\mathcal{P}_{n}^{+}\leftrightarrow \mathcal{P}_{n}^{-}\)。
对 \(\omega\in\mathcal{P}_n^{+}\) 设其正步数为 \(N_{+}\)、负步数为 \(N_{-}\)，则 \(N_{+}-N_{-}=T\)，且
\(
\mathbb{P}_{\theta_+}(\omega)=p^{N_{+}}q^{N_{-}}
\)。
映到 \(\omega'\in\mathcal{P}_n^{-}\) 后，步数交换，故
\[
\frac{\mathbb{P}_{\theta_+}(\omega)}{\mathbb{P}_{\theta_+}(\omega')}
=\Bigl(\frac{p}{q}\Bigr)^{N_{+}-N_{-}}
=\Bigl(\frac{p}{q}\Bigr)^{T}
=\varphi^{T}.
\]
对 \(\mathcal{P}_n^\pm\) 求和即得常比率律。
\par
由该比率与定理 \ref{thm:pom-sprt-symmetric-threshold-error} 给出的
\(
\pi_T/(1-\pi_T)=\varphi^T
\)
可知
\(
\mathbb{P}_{\theta_+}(\tau_T=n,S_T=+)/\mathbb{P}_{\theta_+}(\tau_T=n,S_T=-)
\)
对每个 \(n\) 都等于 \(\pi_T/(1-\pi_T)\)，从而
\(
\mathbb{P}_{\theta_+}(\tau_T=n,S_T=\pm)=\mathbb{P}_{\theta_+}(\tau_T=n)\mathbb{P}_{\theta_+}(S_T=\pm)
\)，即独立性成立。
联合母函数分离是独立性的直接等价改写。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\tau_T$}{tauT} 的精确概率质量函数：有限谱分解与显式系数]\label{prop:pom-tau-pmf-finite-spectral-decomposition}
令
\[
\theta_j:=\frac{(2j-1)\pi}{2T},\qquad
\lambda_j:=2\varphi^{-3/2}\cos\theta_j\quad (j=1,\dots,T),
\]
并定义系数
\[
c_j:=\frac{\cosh\!\bigl(\tfrac{T}{2}\ln\varphi\bigr)}{T}\,(-1)^{j-1}\tan\theta_j.
\]
则对所有 \(n\ge 0\) 有严格有限谱分解
\[
\boxed{\;\mathbb{P}_{\theta_+}(\tau_T=n)=\sum_{j=1}^{T} c_j\,\lambda_j^{n}.\;}
\]
并且谱对称性自动给出支撑的同余约束：当 \(n\not\equiv T\pmod 2\) 或 \(n<T\) 时，右端发生完全抵消，从而 \(\mathbb{P}_{\theta_+}(\tau_T=n)=0\)。
\end{proposition}

\begin{proof}
由定理 \ref{thm:pom-symmetric-double-threshold-pgf-chebyshev}，
\(
G(u):=\mathbb{E}_{\theta_+}[u^{\tau_T}]
\)
在 \(|u|\le 1\) 上为解析函数，并可延拓为有理函数
\[
G(u)=\frac{\cosh(\tfrac{T}{2}\ln\varphi)}{T_T(\varphi^{3/2}/(2u))}.
\]
Chebyshev 多项式的零点为
\(
\cos\theta_j\ (j=1,\dots,T)
\)，因此 \(G(u)\) 在
\(
u=1/\lambda_j=\frac{\varphi^{3/2}}{2\cos\theta_j}
\)
处有简单极点。
对 \(G(u)\) 作部分分式分解可得
\[
G(u)=\sum_{j=1}^{T}\frac{c_j}{1-\lambda_j u},
\]
其中 \(c_j\) 可由留数显式计算并化简为所给表达。
展开 \(\frac{1}{1-\lambda_j u}=\sum_{n\ge 0}\lambda_j^n u^n\) 并比较幂级数系数即得
\(
\mathbb{P}_{\theta_+}(\tau_T=n)=\sum_j c_j\lambda_j^n
\)。
同余/下界约束由 \(\lambda_{T+1-j}=-\lambda_j\) 的谱对称性导致的奇偶抵消得到。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\tau_T$}{tauT} 的二阶矩与方差闭式：\texorpdfstring{$\varphi$}{phi} 的幂级刚性]\label{prop:pom-tau-second-moment-variance-closed}
设 \(T\ge 1\)。
则停时 \(\tau_T\) 的方差满足严格闭式
\[
\boxed{\;
\mathrm{Var}_{\theta_+}(\tau_T)
=\frac{4T\,\varphi^{6}\bigl(\varphi^{2T}-1-T\varphi^{T}\bigr)}{(\varphi^{T}+1)^2}.\;}
\]
并且二阶矩可取闭式
\[
\boxed{\;
\mathbb{E}_{\theta_+}[\tau_T^2]
=\frac{\varphi^{6}}{(\varphi^{T}+1)^2}\Bigl(T^2(\varphi^{T}-1)^2+4T(\varphi^{2T}-1-T\varphi^{T})\Bigr).\;}
\]
特别地，当 \(T\to\infty\) 时有 \(\mathrm{Var}_{\theta_+}(\tau_T)\sim 4\varphi^{6}T\)。
\end{proposition}

\begin{proof}
由定理 \ref{thm:pom-symmetric-double-threshold-pgf-chebyshev}，\(\mathbb{E}_{\theta_+}[u^{\tau_T}]\) 在 \(u=1\) 的邻域内解析，且
\[
\mathbb{E}_{\theta_+}[\tau_T]=\left.\frac{d}{du}\mathbb{E}_{\theta_+}[u^{\tau_T}]\right|_{u=1},\qquad
\mathbb{E}_{\theta_+}[\tau_T(\tau_T-1)]=\left.\frac{d^2}{du^2}\mathbb{E}_{\theta_+}[u^{\tau_T}]\right|_{u=1}.
\]
对母函数在 \(u=1\) 处求导并化简（利用 \(T_T'(x)\)、\(T_T''(x)\) 的显式表达与 \(\varphi^{3/2}/2=\tfrac12(\sqrt\varphi+\varphi^{-1/2})\)）即可得到 \(\mathbb{E}_{\theta_+}[\tau_T^2]\) 与 \(\mathrm{Var}_{\theta_+}(\tau_T)\) 的闭式。
渐近 \(\mathrm{Var}_{\theta_+}(\tau_T)\sim 4\varphi^{6}T\) 由主项直接读出。证毕。
\end{proof}

\begin{theorem}[大阈值极限的中心极限定律：归一化波动尺度 \texorpdfstring{$2\varphi^{3}\sqrt{T}$}{2 phi^3 sqrt(T)}]\label{thm:pom-tau-clt-large-threshold}
当 \(T\to\infty\) 时，停时 \(\tau_T\)（在真相位 \(\theta_+\) 下）满足中心极限定律
\[
\boxed{\;
\frac{\tau_T-\mathbb{E}_{\theta_+}[\tau_T]}{2\varphi^{3}\sqrt{T}}\ \Longrightarrow\ \mathcal{N}(0,1).\;}
\]
并且在二阶矩意义下收敛
\[
\boxed{\;\frac{\tau_T}{T}\xrightarrow{L^2}\varphi^{3}.\;}
\]
因此阈值 \(T\) 的复杂度标度与随机波动标度分别为 \(T\) 与 \(\sqrt{T}\)，且常数由 \(\varphi\) 完全确定。
\end{theorem}

\begin{proof}
令 \(\mu:=\mathbb{E}_{\theta_+}[Y_1]=\varphi^{-3}\) 与 \(\sigma^2:=\mathrm{Var}_{\theta_+}(Y_1)=1-\mu^2=1-\varphi^{-6}\)。
由带漂移随机游走的功能型 CLT（Donsker 不变性原理）与首达时间的连续映射原理，可得 \(\tau_T\) 的主项尺度为 \(T/\mu=\varphi^{3}T\)，且归一化波动为
\(
\sigma\mu^{-3/2}\sqrt{T}=2\varphi^{3}\sqrt{T}
\)
（注意 \(\sigma^2\mu^{-3}=4\varphi^6\)）。
从而得到所示 CLT。
\par
\(L^2\) 收敛由 \(\mathbb{E}[\tau_T]=\varphi^{3}T+o(T)\)（定理 \ref{thm:pom-sprt-symmetric-threshold-mean-time}）与
\(\mathrm{Var}(\tau_T)=O(T)\)（命题 \ref{prop:pom-tau-second-moment-variance-closed}）联立推出。证毕。
\end{proof}

\begin{proposition}[指数尾与解析域：主特征根 \texorpdfstring{$2\varphi^{-3/2}\cos(\pi/2T)$}{2 phi^-3/2 cos(pi/2T)}]\label{prop:pom-tau-exponential-tail-analytic-disk}
令
\[
\rho_T:=\max_{1\le j\le T}|\lambda_j|
=2\varphi^{-3/2}\cos\!\Bigl(\frac{\pi}{2T}\Bigr)\in(0,1),
\]
其中 \(\lambda_j\) 如命题 \ref{prop:pom-tau-pmf-finite-spectral-decomposition} 所定义。
则存在常数 \(0<C_1(T)\le C_2(T)<\infty\) 使得对所有 \(n\ge 0\) 有双边指数界
\[
\boxed{\;
C_1(T)\,\rho_T^{n}\ \le\ \mathbb{P}_{\theta_+}(\tau_T>n)\ \le\ C_2(T)\,\rho_T^{n}.\;}
\]
等价地，概率母函数 \(\mathbb{E}_{\theta_+}[u^{\tau_T}]\) 在复平面内以 \(|u|<\rho_T^{-1}\) 为最大解析圆盘，其最近奇点由
\[
\boxed{\;
u_*=\rho_T^{-1}=\frac{\varphi^{3/2}}{2\cos(\pi/2T)}\;}
\]
给出。
\end{proposition}

\begin{proof}
由命题 \ref{prop:pom-tau-pmf-finite-spectral-decomposition} 的谱分解，
\(
\mathbb{P}(\tau_T=n)
\)
为有限个几何项 \( \lambda_j^n \) 的线性组合，其中最大模特征值为 \(\rho_T\)（对应 \(j=1\)）。
因此尾和 \(\mathbb{P}(\tau_T>n)=\sum_{m>n}\mathbb{P}(\tau_T=m)\) 同样由 \(\rho_T^{n}\) 主导，并可用有限谱分解给出上下界常数 \(C_1(T),C_2(T)\)。
\par
母函数的解析圆盘半径由最近极点决定；由证明 \ref{prop:pom-tau-pmf-finite-spectral-decomposition} 中的极点位置 \(u=1/\lambda_j\) 立得最近奇点为 \(u_*=1/\rho_T=\frac{\varphi^{3/2}}{2\cos(\pi/2T)}\)。证毕。
\end{proof}
