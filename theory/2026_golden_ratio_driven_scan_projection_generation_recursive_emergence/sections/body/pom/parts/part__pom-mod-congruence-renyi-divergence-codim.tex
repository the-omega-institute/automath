\begin{corollary}[模同余全息：几何维度等于“到均匀的 Rényi 散度”之余维]\label{cor:pom-mod-congruence-renyi-divergence-codim}
\leanverified{paper\_pom\_mod\_congruence\_renyi\_divergence\_codim}
沿命题 \ref{prop:pom-moment-congruence-q} 的设定，记 \(F:=F_{m+2}\)，并令同余计数
\[
c_m(r):=\#\{\omega\in\Omega_m:\ N(\omega)\equiv r\ (\mathrm{mod}\ F)\},
\qquad r\in\ZZ/F\ZZ.
\]
定义同余分布 \(p_m(r):=c_m(r)/2^m\) 以及均匀分布 \(u(r)\equiv 1/F\)。
对整数 \(q\ge 2\) 定义 Rényi 熵与 Rényi 散度（取自然对数）
\[
H_q(p):=\frac{1}{1-q}\log\sum_{r}p(r)^q,
\qquad
D_q(p\|u):=\frac{1}{q-1}\log\sum_{r}p(r)^q\,u(r)^{1-q}.
\]
则有严格恒等式
\[
\boxed{\;
D_q(p_m\|u)=\log F-H_q(p_m).\;}
\]
并且在指数主尺度口径 \(S_q(m)=\Theta(r_q^m)\)（从而 \(h_q=q\log 2-\log r_q\)）下，散度密度极限满足
\[
\boxed{\;
\lim_{m\to\infty}\frac{1}{m}D_q(p_m\|u)
=\log\varphi-\frac{h_q}{q-1}
=\log\varphi\bigl(1-D_q^{\mathrm{geom}}\bigr),\;}
\]
其中 \(D_q^{\mathrm{geom}}=\frac{h_q}{(q-1)\log\varphi}\) 为本文的 Rényi 维数谱（命题 \ref{prop:pom-renyi-dimension-spectrum}）。
因此在 Fold 的模同余接口下，几何维度可被解释为“到均匀的 Rényi 散度余维”：体积项 \(\log F\) 与熵项 \(H_q\) 的差即为边界亏损的可审计标量。
\end{corollary}
\begin{proof}
由 \(u(r)\equiv 1/F\)，有
\[
\sum_r p_m(r)^q\,u(r)^{1-q}=F^{q-1}\sum_r p_m(r)^q,
\]
代入 \(D_q\) 定义得到
\(
D_q(p_m\|u)=\log F+\frac{1}{q-1}\log\sum_r p_m(r)^q
\)。
而
\(
H_q(p_m)=\frac{1}{1-q}\log\sum_r p_m(r)^q
=-\frac{1}{q-1}\log\sum_r p_m(r)^q
\)，
合并即得第一式。
\par
另一方面，由命题 \ref{prop:pom-moment-congruence-q}，
\(
\sum_r p_m(r)^q=\sum_r c_m(r)^q/2^{mq}=S_q(m)/2^{mq}
\)，
故
\[
H_q(p_m)=\frac{1}{q-1}\bigl(qm\log 2-\log S_q(m)\bigr).
\]
两边除以 \(m\) 并令 \(m\to\infty\)，在 \(\tau(q)=\lim_{m\to\infty}\frac{1}{m}\log S_q(m)=\log r_q\) 的口径下得到
\(
\lim_{m\to\infty}\frac{1}{m}H_q(p_m)=\frac{1}{q-1}(q\log 2-\log r_q)=\frac{h_q}{q-1}
\)。
再用 \(\lim_{m\to\infty}\frac{1}{m}\log F_{m+2}=\log\varphi\) 得
\(
\lim_{m\to\infty}\frac{1}{m}D_q(p_m\|u)=\log\varphi-\frac{h_q}{q-1}
\)。
最后一式由 \(D_q^{\mathrm{geom}}=\frac{h_q}{(q-1)\log\varphi}\) 代入改写。证毕。
\end{proof}
