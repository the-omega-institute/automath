\begin{proposition}[相位残差证书的线性化常数：\texorpdfstring{$\rho\sim\sqrt{15}\,|\delta\theta|$}{rho~sqrt15|dtheta|} 且具统一二次误差界]\label{prop:pom-phase-residual-linearization-sqrt15}
\leanverified{paper\_pom\_phase\_residual\_linearization\_sqrt15}

\[

\qquad

\boxed{\;
\bigl|\rho(\zeta)-\sqrt{15}\,|\delta\theta|\bigr|\ \le\ 2\,\delta\theta^{2}\; }.
\]
因此在该小相位区间内，\(\rho(\zeta)\) 与 \(|\delta\theta|\) 一阶等价：
\(
\rho(\zeta)/|\delta\theta|=\sqrt{15}+O(|\delta\theta|)
\)。
\end{proposition}

记 \(x:=\delta\theta\)，则 \(\zeta=z_\ast e^{\mathrm{i}x}\)。

\[
\Re(\zeta)=\Re\bigl((a+\mathrm{i}b)(\cos x+\mathrm{i}\sin x)\bigr)=a\cos x-b\sin x
=-\frac14\cos x+\frac{\sqrt{15}}{4}\sin x.
\]
从而
\[
4\Re(\zeta)+1=1-\cos x+\sqrt{15}\sin x.
\]
\par
\textbf{若 \(x\ge 0\)}，则
\[
\rho(\zeta)=1-\cos x+\sqrt{15}\sin x,
\]
并有
\[
\bigl|\rho(\zeta)-\sqrt{15}\,x\bigr|
\le |1-\cos x|+\sqrt{15}\,|\sin x-x|
\le \frac{x^2}{2}+\sqrt{15}\cdot\frac{x^3}{6}
\le 2x^2,
\]
其中用到标准余项界 \(|1-\cos x|\le x^2/2\)、\(|\sin x-x|\le x^3/6\) 以及 \(x\le 1/10\)。
\par
\textbf{若 \(x<0\)}，令 \(u:=-x>0\)。此时
\[
4\Re(\zeta)+1=1-\cos u-\sqrt{15}\sin u<0\qquad(0<u\le\tfrac{1}{10}),
\]
故 \(\rho(\zeta)=-(1-\cos u-\sqrt{15}\sin u)=\sqrt{15}\sin u-(1-\cos u)\)。
于是
\[
\bigl|\rho(\zeta)-\sqrt{15}\,u\bigr|
\le |1-\cos u|+\sqrt{15}\,|\sin u-u|
\le \frac{u^2}{2}+\sqrt{15}\cdot\frac{u^3}{6}
\le 2u^2.
\]
合并两种情形并用 \(u=|x|\) 得结论。证毕。
\end{proof}

\[
\boxed{\ \mathcal L_{\mathrm{CM}}(T):=2T^2+T+2\ }
\]
可被视作一个“相位感知”的二阶消振子：在先行用二周期消去器压低由负实慢模态 \((\lambda_q^-)^m\) 触发的交替误差后（推论 \ref{cor:pom-aitken-cancels-two-cycle}），对剩余残差再施加 \(\mathcal L_{\mathrm{CM}}(T)\) 可进一步压低由复共轭对 \(\mu_q^m,\overline{\mu_q}^m\) 主导的准周期振荡项。
该接口提供了一个与 \(q\) 无关、仅由虚二次域根 \(z_\ast\) 决定的归一化湮灭子模板；其有效性可通过比较
\(
\bigl|\mathcal L_{\mathrm{CM}}(T)\,S_q\bigr|

在消振前后的衰减率来直接审计（不依赖额外的 DP 数据）。

进一步，令 \(\mu\ne 0\) 并写 \(\mu=|\mu|\,\zeta\)（\(\zeta\in\mathbb{S}^1\)），并定义归一化实振子
\[
u(m):=\frac{\mu^{m}}{|\mu|^{m}}+\frac{\overline{\mu}^{m}}{|\mu|^{m}}
=\zeta^{m}+\zeta^{-m}\in\mathbb{R}.
\]

\[
u(m+2)-(\zeta+\zeta^{-1})u(m+1)+u(m)=0.
\]

\[
2u(m+2)+u(m+1)+2u(m)=\bigl(2(\zeta+\zeta^{-1})+1\bigr)\,u(m+1),
\]
从而得到逐点恒等式
\[
\bigl|2u(m+2)+u(m+1)+2u(m)\bigr|
=\rho(\zeta)\,|u(m+1)|\qquad(\forall m\ge 0),
\]
其中 \(\rho(\zeta)\) 如 \eqref{eq:pom-phase-lock-residual-trace-equivalence}。
在锁定迹条件 \(\zeta+\zeta^{-1}=-\tfrac12\)（等价于 \(\rho(\zeta)=0\)）下，上式退化为固定小系数湮灭子
\(
2u(m+2)+u(m+1)+2u(m)\equiv 0
\)。
因此，对共振窗数据，若先用二周期消去器压低由负实慢模态 \((\lambda_q^-)^m\) 触发的交替项，再对剩余振荡残差按 \(|\mu_q|^m\) 归一化，则上述“二阶综合征”可直接在时域检验是否近零，从而把相位锁定转写为无需复求根的可审计接口；其偏离量由 \(\rho_q=\rho(\zeta_q)\) 精确量化。
\end{remark}

\begin{proposition}[共振锁定窗的二次 CM 刚性：\texorpdfstring{$\QQ(\sqrt{-15})$}{Q(sqrt(-15))} 的内禀出现]\label{prop:pom-cm-rigidity-qsqrtm15}

则二维子空间
\[
V_\zeta:=\mathrm{span}_{\QQ}\{f_\zeta,\ f_{\overline{\zeta}}\}
\]
在 \(T\) 下不变，且其上由 \(T\) 生成的交换 \(\QQ\)-代数同构于二次域
\[
\QQ[T]\big|_{V_\zeta}\ \cong\ \QQ(\zeta)\ \cong\ \QQ(\sqrt{-15}).
\]
更具体地，整数子环 \(\ZZ[\zeta]\subset \QQ(\sqrt{-15})\) 在该二维子模态上给出一个内禀的交换子代数作用；若写 \(\zeta=e^{\mathrm{i}\theta^\ast}\)，则
\[
\cos\theta^\ast=\Re(\zeta)=-\frac14.
\]
\end{proposition}
\begin{proof}
由于 \(Tf_\nu=\nu f_\nu\)，子空间 \(V_\zeta\) 在 \(T\) 下不变，且在基 \(\{f_\zeta,f_{\overline{\zeta}}\}\) 上 \(T\) 表为 \(\mathrm{diag}(\zeta,\overline{\zeta})\)。
当 \(\rho(\zeta)=0\) 时，\(\zeta\) 满足不可约二次多项式 \(2z^2+z+2=0\)，其判别式为 \(-15\)，故 \(\QQ(\zeta)\cong \QQ(\sqrt{-15})\)。
因此 \(\QQ[T]\big|_{V_\zeta}\) 与 \(\QQ(\zeta)\) 同构，并诱导出整数子环 \(\ZZ[\zeta]\) 的交换作用。
最后由 \(\zeta+\zeta^{-1}=-\tfrac12\)（即 \eqref{eq:pom-phase-lock-residual-trace-equivalence} 在 \(\rho(\zeta)=0\) 时的形式）得
\(
2\cos\theta^\ast=\zeta+\zeta^{-1}=-\tfrac12
\)，
从而 \(\cos\theta^\ast=-\tfrac14\)。证毕。
\end{proof}

\begin{remark}[近 CM 的稳定扰动模型（解释性语义；不作为推理前提）]\label{rem:pom-near-cm-stable-perturbation}

\begin{remark}[两步重标度点的 Poisson 峰 \texorpdfstring{$\Rightarrow$}{=>} Lorentz 极限：共振窗影子谱包的 Breit--Wigner 离散预极限]\label{rem:pom-resonance-poisson-lorentz-prelimit}
在共振窗 \(q=9,\dots,17\) 中，定义两步重标度模量与相位
\[
\eta_q:=\frac{|\mu_q|}{r_{q-2}}\in(0,1),\qquad \theta_q:=\arg(\mu_q),
\qquad a_q:=\eta_q e^{i\theta_q}=\frac{\mu_q}{r_{q-2}}\in\mathbb{D},
\]
其中 \(\eta_q\uparrow 1\) 的数值证书由表 \ref{tab:fold_collision_shadow_spectral_packet_q9_17} 给出。

B_q(z):=\frac{z-a_q}{1-\overline{a_q}z}.
\]
则其边界相位导数满足标准 Poisson 核恒等式
\[
\frac{d}{dt}\arg B_q(e^{it})
=\frac{1-\eta_q^{2}}{1-2\eta_q\cos(t-\theta_q)+\eta_q^{2}}
 \;=:\; P_{\eta_q}(t-\theta_q).
\]
令 \(\varepsilon_q:=1-\eta_q\downarrow 0\)。则在缩放窗口 \(t=\theta_q+\varepsilon_q u\) 上有严格极限
\[
\lim_{\varepsilon_q\downarrow 0}\ \varepsilon_q\,P_{1-\varepsilon_q}(\varepsilon_q u)=\frac{2}{1+u^{2}}
\qquad(\forall u\in\mathbb{R}),
\]
该极限由代入 \(\eta_q=1-\varepsilon_q\) 并对小角 \(\varepsilon_q u\) 作二阶展开得到：分子 \(1-\eta_q^2=2\varepsilon_q+O(\varepsilon_q^2)\)，分母 \(1-2\eta_q\cos(\varepsilon_q u)+\eta_q^2=\varepsilon_q^2(1+u^2)+O(\varepsilon_q^3)\)。
右端为标准 Lorentz（Cauchy）核；因此共振窗的两步影子谱包在严格分析意义上给出 Lorentz/Breit--Wigner 轮廓的离散预极限，其“共振宽度”自然等同于 \(\varepsilon_q=1-|\mu_q|/r_{q-2}\)，峰位为 \(\theta_q\)。

进一步，Poisson 核满足精确归一化
\(

\)
（任意 \(\eta\in(0,1)\)），从而当 \(\eta_q\uparrow 1\) 时峰形虽尖化但总强度（面积）保持不变；并且峰高满足显式公式
\[
P_{\eta_q}(0)=\frac{1+\eta_q}{1-\eta_q}=\frac{2}{\varepsilon_q}+O(1).
\]
于是得到“高—宽互易”：峰高量级为 \(\varepsilon_q^{-1}\)，峰宽量级为 \(\varepsilon_q\)，而面积恒为常数 \(2\pi\)；该归一化为把谱锁定的集聚与 Lorentz 峰在同一标尺下比较提供了自然尺度。

此外，在同一两步重标度口径下，复相位峰位的逐阶漂移可被宽度参数线性支配。
令
\(
\varepsilon_q:=1-\eta_q

\Delta\theta_q:=|\theta_{q+1}-\theta_q|

\[
\Delta\theta_q\ \le\ 0.143\,\varepsilon_q.

因此在自然缩放坐标 \(t=\theta_q+\varepsilon_q u\) 下，峰位随 \(q\) 的迁移为准同形的缓慢平移而非离散跳跃。
\end{remark}
