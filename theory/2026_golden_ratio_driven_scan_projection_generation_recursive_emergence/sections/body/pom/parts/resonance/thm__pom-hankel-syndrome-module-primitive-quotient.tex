\begin{theorem}[Hankel-综合征模的主理想稳定性：Hankel-null 等于最小递推的截断倍乘模]\label{thm:pom-hankel-syndrome-module-kernel-equals-multiples}
\leanverified{paper\_pom\_hankel\_syndrome\_module\_kernel\_equals\_multiples}
设 \((s_m)_{m\ge 0}\subset\ZZ\) 为整数序列。
假设存在首一且本原多项式
\[
P(\lambda)=\lambda^{d}+p_{d-1}\lambda^{d-1}+\cdots+p_0\in\ZZ[\lambda],
\]
以及整数 \(m_0\ge 0\)，使得对一切 \(m\ge m_0\) 有线性递推
\[
s_{m+d}+p_{d-1}s_{m+d-1}+\cdots+p_0s_{m}=0,
\]
并且在 \(\QQ[\lambda]\) 中，理想
\(
\{A(\lambda)\in\QQ[\lambda]:\ A(E)s\equiv 0\ \text{于所有}\ m\ge m_0\}
\)
由 \(P\) 生成（即 \(P\) 为从 \(m_0\) 起的最小湮灭多项式）。
\par
对任意 \(n\ge d+1\)，定义 Hankel 方块
\[
\mathsf H^{(m_0)}_{n}:=\bigl(s_{m_0+i+j}\bigr)_{0\le i,j\le n-1}\in\ZZ^{n\times n},
\]
并定义截断倍乘子模
\[
\mathcal{M}_n(P):=
\Bigl\{\operatorname{coeff}_{<n}\!\bigl(Q(\lambda)P(\lambda)\bigr)\in\ZZ^n:\ 
Q\in\ZZ[\lambda],\ \deg Q<n-d\Bigr\},
\]
其中 \(\operatorname{coeff}_{<n}\) 表示截断到 \((\lambda^0,\dots,\lambda^{n-1})\) 的系数向量。
则成立严格恒等式（作为 \(\ZZ\)-模）
\[
\boxed{\;
\ker_{\ZZ}\bigl((\mathsf H^{(m_0)}_{n})^{\mathsf T}\bigr)
\;=\;
\mathcal{M}_n(P)\; }.
\]
\end{theorem}
\begin{proof}
取任意 \(\alpha=(\alpha_0,\dots,\alpha_{n-1})\in\ker_{\ZZ}((\mathsf H^{(m_0)}_n)^{\mathsf T})\)。
则对所有 \(0\le k\le n-1\) 有
\[
0=\sum_{i=0}^{n-1}\alpha_i\,s_{m_0+k+i}.
\]
令残差序列
\[
r_k:=\sum_{i=0}^{n-1}\alpha_i\,s_{m_0+k+i}\qquad(k\ge 0),
\]
则 \(r_0=\cdots=r_{n-1}=0\)。
由于 \(s\) 在 \(m\ge m_0\) 上满足 \(P(E)s=0\)，并且 \(P(E)\) 与任意幂 \(E^i\) 交换，得
\[
P(E)r
=\sum_{i=0}^{n-1}\alpha_i\,P(E)E^i s
=\sum_{i=0}^{n-1}\alpha_i\,E^i P(E)s
=0,
\]
从而 \((r_k)\) 亦满足阶数 \(\le d\) 的同一递推。
又因 \(n\ge d+1\)，有 \(r_0=\cdots=r_{d}=0\)，递推解由前 \(d\) 项唯一确定，故 \(r_k\equiv 0\)（\(k\ge 0\)）。
\par
令
\(
A(\lambda):=\sum_{i=0}^{n-1}\alpha_i\lambda^i\in\ZZ[\lambda]
\)。
恒等式 \(r_k=0\)（\(k\ge 0\)）等价于 \(A(E)s\equiv 0\) 在所有 \(m\ge m_0\) 成立，
因此 \(A\) 属于上述在 \(\QQ[\lambda]\) 中的湮灭理想。
由最小性假设，该理想由 \(P\) 生成，故 \(P\mid A\) 于 \(\QQ[\lambda]\)。
又因 \(P\) 本原，Gauss 引理推出 \(P\mid A\) 于 \(\ZZ[\lambda]\)，从而 \(A=QP\) 且 \(Q\in\ZZ[\lambda]\)。
由 \(\deg A<n\) 与 \(\deg P=d\) 得 \(\deg Q<n-d\)，故
\(
\alpha=\operatorname{coeff}_{<n}(QP)\in\mathcal{M}_n(P)
\)，从而 \(\ker\subseteq\mathcal{M}_n(P)\)。
\par
反之，若 \(\alpha=\operatorname{coeff}_{<n}(QP)\) 且 \(\deg Q<n-d\)，则 \(QP(E)s=0\) 在 \(m\ge m_0\) 上成立；
对 \(k=0,\dots,n-1\) 取 \(m=m_0+k\) 即得 \((\mathsf H^{(m_0)}_n)^{\mathsf T}\alpha=0\)，故 \(\mathcal{M}_n(P)\subseteq\ker\)。
证毕。
\end{proof}

\begin{corollary}[综合征基的秩与自然生成元]\label{cor:pom-hankel-syndrome-module-rank-and-generators}
在定理 \ref{thm:pom-hankel-syndrome-module-kernel-equals-multiples} 的设定下，令 \(\Delta:=n-d\)。
则 \(\ker_{\ZZ}((\mathsf H^{(m_0)}_{n})^{\mathsf T})\) 为秩 \(\Delta\) 的自由阿贝尔群，并且一组自然生成元可取为
\[
\Bigl\{\operatorname{coeff}_{<n}\bigl(\lambda^j P(\lambda)\bigr):\ 0\le j\le \Delta-1\Bigr\}.
\]
因此任何综合征基（作为 \(\ZZ\)-基）必含且仅含 \(\Delta\) 个向量。
\end{corollary}
\begin{proof}
映射
\(
\ZZ[\lambda]_{<\Delta}\to\ZZ^n,\ Q\mapsto \operatorname{coeff}_{<n}(QP)
\)
为群同态且其像为 \(\mathcal{M}_n(P)\)；
由于 \(P\) 首一，\(\deg(QP)<n\) 时由 \(\operatorname{coeff}_{<n}(QP)=0\) 推出 \(QP=0\)，故该同态为单射。
于是 \(\mathcal{M}_n(P)\cong \ZZ^\Delta\)，再用定理 \ref{thm:pom-hankel-syndrome-module-kernel-equals-multiples} 得结论。
取 \(Q=\lambda^j\) 即得所示生成元。证毕。
\end{proof}

\begin{corollary}[综合征子格的唯一规范化：列 Hermite 正规形]\label{cor:pom-hankel-syndrome-hnf-unique}
在推论 \ref{cor:pom-hankel-syndrome-module-rank-and-generators} 的设定下，
存在唯一的列 Hermite 正规形矩阵 \(B\in\ZZ^{n\times\Delta}\) 使得
\[
\operatorname{colspan}_{\ZZ}(B)=\ker_{\ZZ}\bigl((\mathsf H^{(m_0)}_{n})^{\mathsf T}\bigr)\subset\ZZ^n.
\]
\end{corollary}
\begin{proof}
任意满秩子格均存在且仅存在列 Hermite 正规形表示。证毕。
\end{proof}

\begin{proposition}[多模数离线审计的确定性完备性]\label{prop:pom-hankel-syndrome-crt-offline-audit}
在推论 \ref{cor:pom-hankel-syndrome-module-rank-and-generators} 的设定下，取一组综合征基
\(\{\alpha^{(j)}\}_{j=1}^{\Delta}\subset\ZZ^n\)。
对 \(1\le j\le\Delta\)、\(0\le k\le n-1\) 定义
\[
e_{j,k}:=\sum_{i=0}^{n-1}\alpha^{(j)}_{i}\,s_{m_0+k+i}\in\ZZ.
\]
设存在有效上界 \(U\ge \max_{j,k}|e_{j,k}|\)。
取两两互素的模数 \(M_1,\dots,M_t\) 使 \(M:=\prod_{r=1}^{t}M_r>2U\)。
若对所有 \(r,j,k\) 均有
\[
e_{j,k}\equiv 0\pmod{M_r},
\]
则必有 \(e_{j,k}=0\)（整数意义）对所有 \(j,k\) 成立。
\end{proposition}
\begin{proof}
若某个 \(e_{j,k}\neq 0\) 且对所有 \(r\) 皆有 \(e_{j,k}\equiv 0\pmod{M_r}\)，则由中国剩余定理知 \(e_{j,k}\equiv 0\pmod{M}\)，从而 \(|e_{j,k}|\ge M>2U\)，与 \(|e_{j,k}|\le U\) 矛盾。证毕。
\end{proof}

\begin{theorem}[确定性同余审计阈值的最优性]\label{thm:pom-deterministic-congruence-audit-threshold-optimal}
\leanverified{paper\_pom\_deterministic\_congruence\_audit\_threshold\_optimal}
设 \(U\in\ZZ_{>0}\)、\(M\in\ZZ_{>0}\)，并令 \(A,B\in\Mat_{d\times d}(\ZZ)\) 满足逐项界
\[
|A_{ij}|\le U,\qquad |B_{ij}|\le U\qquad(1\le i,j\le d).
\]
若 \(A\equiv B\pmod M\)（逐项同余），则在 \(M>2U\) 时必有 \(A=B\)。
反之，当 \(M\le 2U\) 时，对任意 \(d\ge 1\) 存在 \(A\neq B\) 仍满足上述逐项界并且 \(A\equiv B\pmod M\)。
\end{theorem}
\begin{proof}
设 \(D:=A-B\)。
若 \(M>2U\)，则对任意 \(i,j\) 有 \(D_{ij}\in M\ZZ\) 且 \(|D_{ij}|\le 2U<M\)，故 \(D_{ij}=0\)，从而 \(A=B\)。
\par
反向：取 \(d=1\)。
若 \(M\) 为偶数，令 \(a=M/2\)、\(b=-M/2\)，则 \(|a|,|b|\le U\) 且 \(a\equiv b\pmod M\) 但 \(a\neq b\)。
若 \(M\) 为奇数，取 \(a=(M-1)/2\)、\(b=-(M+1)/2\)，在 \(M\le 2U-1\) 时同样满足 \(|a|,|b|\le U\) 且 \(a\equiv b\pmod M\)。
将该标量构造嵌入至任意维度 \(d\)（仅令某一条目取 \((a,b)\) 而其余条目相同）即得结论。证毕。
\end{proof}

\begin{remark}[零性审计与同余提升的阈值差异]\label{rem:pom-congruence-zero-vs-lift-threshold}
若仅需判定一个未知整数 \(e\) 是否为 \(0\)，并且已知 \(|e|\le U\)，则条件 \(M>U\) 已足以由 \(e\equiv 0\pmod M\) 推出 \(e=0\)。
但当要比较两未知对象 \(A,B\) 且仅知各自的逐项幅度上界时，其差的逐项幅度上界为 \(2U\)；
定理 \ref{thm:pom-deterministic-congruence-audit-threshold-optimal} 表明在该对称信息口径下阈值 \(M>2U\) 不可再降。
\end{remark}

\begin{theorem}[模同余的 \texorpdfstring{$2d$}{2d} 锁定：C-finite 序列的离线全恢复]\label{thm:pom-congruence-2d-locking-cfinite}
\leanverified{paper\_pom\_congruence\_2d\_locking\_cfinite}
设整数序列 \((a_m)_{m\ge 0}\subset\ZZ\) 自某一偏移 \(m_0\) 起满足阶数为 \(d\) 的整系数线性递推
\[
a_{m+d}=\sum_{j=0}^{d-1}\beta_j a_{m+j},\qquad \beta_j\in\ZZ,\quad m\ge m_0,
\]
并且存在先验界 \(U\in\ZZ_{\ge 0}\) 使
\[
|a_{m_0+i}|\le U\qquad(0\le i<2d).
\]
令 \(M>2U\)。
若仅给出窗口的模数据
\[
a_{m_0+i}\bmod M,\qquad i=0,1,\dots,2d-1,
\]
则可唯一确定 \((a_{m_0},\dots,a_{m_0+2d-1})\) 的整数取值，从而唯一确定全序列 \((a_m)_{m\ge m_0}\)。
此外，若 \((b_m)\) 亦满足同一递推且在该窗口上有 \(b_{m_0+i}\equiv a_{m_0+i}\pmod M\)，则对所有 \(m\ge m_0\) 有 \(b_m=a_m\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-deterministic-congruence-audit-threshold-optimal} 的标量特例（取 \(d=1\)）可知：当 \(M>2U\) 时，区间 \([-U,U]\) 内的整数由其模 \(M\) 余数唯一确定。
故可从窗口模数据唯一提升出 \((a_{m_0},\dots,a_{m_0+2d-1})\) 的真实整数值，随后递推关系唯一确定全部后续项。
最后一条由同样的提升唯一性与递推闭合性立得。证毕。
\end{proof}

\begin{theorem}[综合征子格对最小递推的反演唯一性]\label{thm:pom-hankel-syndrome-inversion-unique}
\leanverified{paper\_pom\_hankel\_syndrome\_inversion\_unique}
设 \(n\ge 2\)。
对任意首一且本原多项式 \(P(\lambda)\in\ZZ[\lambda]\)（\(\deg P=d<n\)）定义截断倍乘子模 \(\mathcal M_n(P)\subset\ZZ^n\) 如 \eqref{thm:pom-hankel-syndrome-module-kernel-equals-multiples} 所示。
若 \(P_1,P_2\in\ZZ[\lambda]\) 均首一且本原，且满足
\[
\mathcal M_n(P_1)=\mathcal M_n(P_2)\subset\ZZ^n,
\]
则必有 \(P_1=P_2\)。
\end{theorem}
\begin{proof}
记 \(d_i:=\deg P_i\)。
由于 \(d_i<n\)，有 \(\operatorname{coeff}_{<n}(P_i)\in \mathcal M_n(P_i)\)。
由 \(\mathcal M_n(P_1)=\mathcal M_n(P_2)\) 得 \(\operatorname{coeff}_{<n}(P_1)\in \mathcal M_n(P_2)\)，
故存在 \(Q\in\ZZ[\lambda]\) 且 \(\deg Q<n-d_2\) 使
\[
\operatorname{coeff}_{<n}(P_1)=\operatorname{coeff}_{<n}(Q(\lambda)P_2(\lambda)).
\]
由 \(\deg(QP_2)\le (n-d_2-1)+d_2=n-1\) 得 \(\deg(QP_2)<n\)，
因此截断系数向量唯一确定多项式本身，从而 \(P_1=QP_2\) 于 \(\ZZ[\lambda]\)。
同理存在 \(Q'\in\ZZ[\lambda]\) 使 \(P_2=Q'P_1\)，于是 \(P_1=(QQ')P_1\)，在整环 \(\ZZ[\lambda]\) 中推出 \(QQ'=1\)。
故 \(Q,Q'\in\{\pm 1\}\)；又因 \(P_1,P_2\) 首一，得 \(Q=Q'=1\)，从而 \(P_1=P_2\)。证毕。
\end{proof}

\begin{proposition}[整除与包含的反序对应；交为最小公倍子模]\label{prop:pom-hankel-syndrome-divisibility-reversal-lcm}
\leanverified{paper\_pom\_hankel\_syndrome\_divisibility\_reversal\_lcm}
设 \(P_1,P_2\in\ZZ[\lambda]\) 均为首一且本原多项式，且 \(\deg P_i<n\)。
则有反序对应
\[
P_1\mid P_2
\quad\Longleftrightarrow\quad
\mathcal M_n(P_2)\subseteq \mathcal M_n(P_1).
\]
并且若 \(\mathrm{lcm}(P_1,P_2)\) 表示二者在 \(\QQ[\lambda]\) 中的首一最小公倍多项式（从而亦属于 \(\ZZ[\lambda]\) 且本原），则
\[
\mathcal M_n(P_1)\cap \mathcal M_n(P_2)=\mathcal M_n\bigl(\mathrm{lcm}(P_1,P_2)\bigr).
\]
\end{proposition}
\begin{proof}
若 \(P_1\mid P_2\)，则任意 \(A(\lambda)=Q(\lambda)P_2(\lambda)\) 亦满足 \(P_1\mid A\)，
从而 \(\operatorname{coeff}_{<n}(A)\in\mathcal M_n(P_1)\)；故 \(\mathcal M_n(P_2)\subseteq\mathcal M_n(P_1)\)。
反之，若 \(\mathcal M_n(P_2)\subseteq\mathcal M_n(P_1)\)，
由于 \(\operatorname{coeff}_{<n}(P_2)\in\mathcal M_n(P_2)\)，得 \(\operatorname{coeff}_{<n}(P_2)\in\mathcal M_n(P_1)\)。
由 \(\deg P_2<n\) 与同样的“截断无损”论证（见定理 \ref{thm:pom-hankel-syndrome-inversion-unique} 的证明）推出 \(P_1\mid P_2\) 于 \(\ZZ[\lambda]\)。
\par
对交集：向量 \(v\in\mathcal M_n(P_1)\cap\mathcal M_n(P_2)\) 当且仅当存在 \(A(\lambda)\in\ZZ[\lambda]\) 使 \(\deg A<n\) 且
\[
v=\operatorname{coeff}_{<n}(A),\qquad P_1\mid A,\qquad P_2\mid A.
\]
由于 \(\ZZ[\lambda]\) 为 UFD，上述整除条件等价于 \(\mathrm{lcm}(P_1,P_2)\mid A\)，
从而等价于 \(v\in\mathcal M_n(\mathrm{lcm}(P_1,P_2))\)。
反向包含显然，故结论成立。证毕。
\end{proof}

\begin{corollary}[列 Hermite 正规形综合征证书的实现无关性]\label{cor:pom-hankel-syndrome-hnf-model-free}
在定理 \ref{thm:pom-hankel-syndrome-module-kernel-equals-multiples} 的设定下，令
\[
\mathcal B_{n}(P):=\mathrm{HNF}\!\bigl(\mathcal M_n(P)\bigr)\in\ZZ^{n\times(n-d)}
\]
表示子格 \(\mathcal M_n(P)\subset\ZZ^n\) 的唯一列 Hermite 正规形矩阵。
则对任意满足同一最小湮灭多项式 \(P\) 的序列 \((s_m)\)（从同一偏移 \(m_0\) 起），均有
\[
\mathrm{HNF}\!\Bigl(\ker_{\ZZ}\bigl((\mathsf H^{(m_0)}_{n})^{\mathsf T}\bigr)\Bigr)=\mathcal B_n(P),
\]
从而该证书仅依赖于 \((P,n)\)，与序列的初值数据无关。
特别地，在任何把初值视为随机变量的概率模型中，只要 \((P,n)\) 固定，则该证书与初值独立，因而互信息为零。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-hankel-syndrome-module-kernel-equals-multiples}，
\(
\ker_{\ZZ}((\mathsf H^{(m_0)}_{n})^{\mathsf T})=\mathcal M_n(P)
\)；
再由列 Hermite 正规形对满秩子格的唯一性（推论 \ref{cor:pom-hankel-syndrome-hnf-unique}）即得第一式。
第二部分：当 \((P,n)\) 固定时，\(\mathcal B_n(P)\) 为常量随机变量，故与任意初值随机变量独立，从而互信息为零。证毕。
\end{proof}

\begin{theorem}[原始缺口在非极小递推下的秩等于阶数缺陷]\label{thm:pom-hankel-syndrome-gap-rank-defect}
设整数序列 \((s_m)_{m\ge 0}\subset\ZZ\) 满足某首一且本原多项式 \(P(\lambda)\in\ZZ[\lambda]\)（\(\deg P=d\)）给出的递推 \(P(E)s=0\) 于所有 \(m\ge m_0\)。
设在 \(\QQ[\lambda]\) 中，从 \(m_0\) 起的最小湮灭多项式为首一且本原的 \(P_{\min}\in\ZZ[\lambda]\)，并写
\(
P=P_{\min}\cdot R
\)
于 \(\ZZ[\lambda]\)，其中 \(R\in\ZZ[\lambda]\) 首一且 \(\deg R=d-d_{\min}\)，\(d_{\min}:=\deg P_{\min}\)。
取 \(n\ge d+1\)，并定义商群
\[
\mathsf Q_n(P;s)
:=\ker_{\ZZ}\bigl((\mathsf H^{(m_0)}_{n})^{\mathsf T}\bigr)\big/\mathcal M_n(P).
\]
则 \(\mathsf Q_n(P;s)\) 为自由阿贝尔群且
\[
\operatorname{rank}\,\mathsf Q_n(P;s)=d-d_{\min}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-hankel-syndrome-module-kernel-equals-multiples}（对 \(P_{\min}\) 应用），
\(
\ker_{\ZZ}((\mathsf H^{(m_0)}_{n})^{\mathsf T})=\mathcal M_n(P_{\min})
\)。
又由命题 \ref{prop:pom-hankel-syndrome-divisibility-reversal-lcm} 的包含反序，对 \(P_{\min}\mid P\) 有
\(
\mathcal M_n(P)\subseteq \mathcal M_n(P_{\min})
\)。
\par
考虑同态
\[
\phi_{\min}:\ \ZZ[\lambda]_{<n-d_{\min}}\longrightarrow \ZZ^n,\qquad
Q\longmapsto \operatorname{coeff}_{<n}\!\bigl(QP_{\min}\bigr).
\]
由推论 \ref{cor:pom-hankel-syndrome-module-rank-and-generators}（对 \(P_{\min}\)），\(\phi_{\min}\) 为同构到 \(\mathcal M_n(P_{\min})\)。
在该识别下，
\[
\mathcal M_n(P)=\phi_{\min}\!\bigl(R\cdot \ZZ[\lambda]_{<n-d}\bigr),
\qquad n-d=(n-d_{\min})-\deg R.
\]
因此
\[
\mathsf Q_n(P;s)\ \cong\ \ZZ[\lambda]_{<n-d_{\min}}\big/ R\cdot\ZZ[\lambda]_{<n-d}.
\]
由于 \(R\) 首一，\(\ZZ[\lambda]\) 上的带余除法对任意 \(A\in\ZZ[\lambda]\) 给出唯一分解
\(
A=RQ+U
\)
且 \(\deg U<\deg R\)；
当 \(\deg A<n-d_{\min}\) 时必有 \(\deg Q<n-d\)。
因此每个商类都唯一由余式 \(U\)（\(\deg U<\deg R\)）代表，
从而该商群同构于 \(\ZZ^{\deg R}\)，为自由阿贝尔群且秩为 \(\deg R=d-d_{\min}\)。证毕。
\end{proof}

\begin{conjecture}[共振缺口的 \texorpdfstring{$(\lambda+2)$}{(lambda+2)}-链生成机制]\label{conj:pom-resonance-gap-lambda-plus-2-chain}

换言之，超出最小递推的全部额外 Hankel-null 方向由一个普适因子 \(R\) 对 \(P_{\min}\) 的截断倍乘闭包所强制生成。
\end{conjecture}

\begin{proposition}[递推约束的违约计数与汉明距离夹逼]\label{prop:pom-recurrence-code-violation-distance}
令 \(R\) 为交换环，取首一多项式
\(
P(\lambda)=\sum_{j=0}^{d}p_j\lambda^j\in R[\lambda]
\)
（\(p_d=1\)），并对整数 \(n\ge d+1\) 定义长度 \(n\) 的递推码
\[
\mathcal C_{P,n}(R):=\Bigl\{x\in R^n:\ \sum_{j=0}^{d} p_jx_{k+j}=0\ \text{对一切}\ 0\le k\le n-d-1\Bigr\}.
\]
对任意 \(x\in R^n\)，定义违约计数
\[
V_P(x):=\#\Bigl\{0\le k\le n-d-1:\ \sum_{j=0}^{d}p_jx_{k+j}\neq 0\Bigr\},
\]
并令 \(\mathrm{dist}(x,\mathcal C_{P,n}(R))\) 表示 \(x\) 到 \(\mathcal C_{P,n}(R)\) 的汉明距离。
则恒有
\[
\mathrm{dist}(x,\mathcal C_{P,n}(R))\ \le\ V_P(x)\ \le\ (d+1)\,\mathrm{dist}(x,\mathcal C_{P,n}(R)).
\]
\end{proposition}
\begin{proof}
右不等式：任取 \(c\in\mathcal C_{P,n}(R)\)，令 \(e:=x-c\)。
若第 \(k\) 条约束被违反，则 \(\sum_{j=0}^dp_je_{k+j}\neq 0\)，从而存在 \(j\in\{0,\dots,d\}\) 使 \(e_{k+j}\neq 0\)。
每个非零坐标 \(e_i\) 至多出现在 \(d+1\) 条约束之中（对应 \(k\in[i-d,i]\cap[0,n-d-1]\)），
故 \(V_P(x)\le(d+1)\cdot|\mathrm{supp}(e)|\)。
对所有 \(c\) 取最小即得 \(V_P(x)\le(d+1)\,\mathrm{dist}(x,\mathcal C_{P,n}(R))\)。
\par
左不等式：自左向右构造 \(x'\in R^n\)。
对 \(k=0,1,\dots,n-d-1\)，若第 \(k\) 条约束已满足则保持不变；
否则仅修改坐标 \(x'_{k+d}\) 使
\[
x'_{k+d}:=-\sum_{j=0}^{d-1}p_jx'_{k+j}.
\]
该修改不影响任何 \(<k\) 的约束（其均不含坐标 \(k+d\)），故最终得到 \(x'\in\mathcal C_{P,n}(R)\)；
修改次数不超过初始违约次数 \(V_P(x)\)，从而 \(\mathrm{dist}(x,\mathcal C_{P,n}(R))\le V_P(x)\)。证毕。
\end{proof}

\begin{conjecture}[饱和 shift-不变综合征子格的主理想分类]\label{conj:pom-hankel-shift-invariant-saturated-principal}
固定整数 \(n\ge 1\)，以系数同构识别
\[
\ZZ^n\ \cong\ \ZZ[\lambda]_{<n},\qquad
(x_0,\dots,x_{n-1})\longleftrightarrow \sum_{j=0}^{n-1}x_j\lambda^j.
\]
令移位算子 \(\sigma:\ZZ^n\to\ZZ^n\) 对应于多项式乘法 \(A(\lambda)\mapsto \lambda A(\lambda)\) 后再截断到次数 \(<n\)。
设 \(L\subset\ZZ^n\) 为秩 \(\Delta\) 的子格，并满足：
\begin{enumerate}
  \item \(\sigma(L)\subseteq L\)；
  \item \(L\) 为饱和子格（即 \(\ZZ^n/L\) 无挠）。
\end{enumerate}
则存在唯一的首一且本原多项式 \(P(\lambda)\in\ZZ[\lambda]\)（\(\deg P=n-\Delta\)）使
\[
L=\mathcal M_n(P).
\]
\end{conjecture}
