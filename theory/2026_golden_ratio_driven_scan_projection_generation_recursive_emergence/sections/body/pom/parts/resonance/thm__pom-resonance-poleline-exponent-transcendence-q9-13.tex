\begin{theorem}[共振窗可审计点的极线指数超越性：\texorpdfstring{$q\in\{9,10,11,13\}$}{q in \{9,10,11,13\}}]\label{thm:pom-resonance-poleline-exponent-transcendence-q9-13}
取 \(q\in\{9,10,11,13\}\)，令 \(P_q(\lambda)\in\ZZ[\lambda]\) 为共振窗最小整系数递推特征多项式（定理 \ref{thm:pom-moment-resonance}），令 \(r_q>1\) 为其 Perron 根，并令 \(\lambda_q^-<0\) 为最外负实根（表 \ref{tab:fold_collision_shadow_spectral_packet_q9_17}）。
记
\[
\Lambda_q:=|\lambda_q^-|=\max\{|\mu|:\ \mu\in\mathrm{Roots}(P_q),\ \mu\ne r_q\}
\qquad\text{（表 \ref{tab:fold_collision_kernel_rh_scan_q2_17}）}
\]
并定义极线指数
\[
\beta_q:=\frac{\log \Lambda_q}{\log r_q}.
\]
则 \(\beta_q\) 为超越数；等价地，\(\log r_q\) 与 \(\log \Lambda_q\) 在 \(\QQ\) 上线性无关。
\end{theorem}
\begin{proof}
记 \(d:=\deg P_q\) 并在分裂域 \(K_q/\QQ\) 中把 \(P_q\) 的根标号为 \(\alpha_1,\dots,\alpha_d\)，其中
\(\alpha_1=r_q\) 且 \(\alpha_2=\lambda_q^-\)。
由命题 \ref{prop:pom-observable-minpoly-galois-sd-q9-13}，有
\(\Gal(P_q/\QQ)\cong S_d\)，特别地其在根集上的作用 \(2\)-传递。
\par
\textbf{第一步：排除 \(\beta_q\in\QQ\).}
若 \(\beta_q=\frac{m}{n}\in\QQ\)（\(m,n\ge 1\)，互素），则
\[
\Lambda_q^n=r_q^m
\qquad\Longleftrightarrow\qquad
(-\alpha_2)^n=\alpha_1^m.
\]
由 \(2\)-传递性，对任意有序对 \((i,j)\)（\(i\ne j\)）存在 \(\sigma\in\Gal(K_q/\QQ)\) 使
\(\sigma(\alpha_1)=\alpha_i\)、\(\sigma(\alpha_2)=\alpha_j\)；
故对所有 \(i\ne j\) 均有
\begin{equation}\label{eq:pom-resonance-poleline-rational-constraint-allpairs}
(-\alpha_j)^n=\alpha_i^m.
\end{equation}
固定某个 \(i\)，取不同的 \(j,k\ne i\) 比较 \eqref{eq:pom-resonance-poleline-rational-constraint-allpairs} 得
\((-\alpha_j)^n=(-\alpha_k)^n\)，从而 \(\alpha_j/\alpha_k\) 为 \(n\) 次单位根，进而
\(|\alpha_j|=|\alpha_k|\)。
这推出 \(|\alpha_2|=\cdots=|\alpha_d|\)。
再取 \(i=2\)、\(j=3\) 与 \(i=1\)、\(j=3\) 的 \eqref{eq:pom-resonance-poleline-rational-constraint-allpairs}，得
\(|\alpha_3|^n=|\alpha_2|^m\) 与 \(|\alpha_3|^n=|\alpha_1|^m\)，从而 \(|\alpha_1|=|\alpha_2|\)。
于是所有根模长相等，与 Perron 根的严格模长主导性 \(|\alpha_1|>|\alpha_j|\ (j\ge 2)\) 矛盾（例如见表 \ref{tab:fold_collision_kernel_rh_scan_q2_17} 中 \(\Lambda_q<r_q\)）。
故 \(\beta_q\notin\QQ\)。
\par
\textbf{第二步：由 \(\beta_q\notin\QQ\) 推出 \(\beta_q\) 超越.}
注意到 \(r_q\) 与 \(\Lambda_q\) 皆为代数数且 \(r_q\neq 0,1\)。
若 \(\beta_q\) 为代数数且非有理，则由 Gelfond--Schneider 定理得 \(r_q^{\beta_q}\) 为超越数；
但按定义 \(r_q^{\beta_q}=\Lambda_q\) 为代数数，矛盾。
因此 \(\beta_q\) 只能是有理数或超越数；结合第一步，\(\beta_q\) 必为超越数。
\end{proof}

\begin{theorem}[Newman 比值对数的非零超越性与分裂域范数闭式：\texorpdfstring{$q\in\{9,10,11,13\}$}{q in \{9,10,11,13\}}]\label{thm:pom-resonance-newman-log-transcendence-norm-q9-13}
取 \(q\in\{9,10,11,13\}\) 并沿用定理 \ref{thm:pom-resonance-poleline-exponent-transcendence-q9-13} 的记号，令
\[
\upsilon_q:=\frac{\Lambda_q^{2}}{r_q}=\frac{(\lambda_q^-)^2}{r_q}\in\RR_{>0},
\qquad
\delta_q:=\log\upsilon_q.
\]
则 \(\upsilon_q\neq 1\)，并且 \(\delta_q\neq 0\) 且 \(\delta_q\) 为超越数。
更精确地，若 \(K_q\) 为 \(P_q\) 的分裂域，则有范数恒等式
\[
\mathrm{Norm}_{K_q/\QQ}(\upsilon_q)=\bigl((-1)^{d}P_q(0)\bigr)^{(d-1)!},
\qquad d:=\deg P_q,
\]
从而对上述 \(q\) 值（定理 \ref{thm:pom-moment-resonance} 给出 \(P_q(0)\notin\{\pm 1\}\)）可推出 \(\upsilon_q\neq 1\)。
\end{theorem}
\begin{proof}
记 \(d:=\deg P_q\)，在 \(K_q\) 中把根标号为 \(\alpha_1,\dots,\alpha_d\)，其中 \(\alpha_1=r_q\) 且 \(\alpha_2=\lambda_q^-\)。
由命题 \ref{prop:pom-observable-minpoly-galois-sd-q9-13}，\(\Gal(K_q/\QQ)\cong S_d\)。
于是
\[
\mathrm{Norm}_{K_q/\QQ}(\upsilon_q)
=\prod_{\sigma\in S_d}\sigma\!\left(\frac{\alpha_2^2}{\alpha_1}\right)
=\prod_{\sigma\in S_d}\frac{\alpha_{\sigma(2)}^2}{\alpha_{\sigma(1)}}.
\]
对固定的 \(k\in\{1,\dots,d\}\)，计数得：
\(\alpha_k\) 在分子中作为 \(\alpha_{\sigma(2)}\) 出现 \((d-1)!\) 次且每次带平方，贡献指数 \(2(d-1)!\)；
在分母中作为 \(\alpha_{\sigma(1)}\) 出现 \((d-1)!\) 次，贡献指数 \((d-1)!\)。
故总指数为 \((d-1)!\)，从而
\[
\mathrm{Norm}_{K_q/\QQ}(\upsilon_q)
=\prod_{k=1}^{d}\alpha_k^{(d-1)!}
=\Bigl(\prod_{k=1}^{d}\alpha_k\Bigr)^{(d-1)!}
=\bigl((-1)^{d}P_q(0)\bigr)^{(d-1)!}.
\]
另一方面，由定理 \ref{thm:pom-moment-resonance} 在 \(q\in\{9,10,11,13\}\) 上的显式多项式可直接读出 \(P_q(0)\notin\{\pm 1\}\)；
因此 \(|\mathrm{Norm}_{K_q/\QQ}(\upsilon_q)|>1\)，推出 \(\upsilon_q\neq 1\)（否则范数为 \(1\)）。
最后，\(\upsilon_q\in\overline{\QQ}\cap\RR_{>0}\) 且 \(\upsilon_q\neq 1\) 蕴含 \(\log\upsilon_q\neq 0\)，并由 Lindemann--Weierstrass 定理的直接推论得 \(\delta_q=\log\upsilon_q\) 为超越数。
\end{proof}

\begin{proposition}[Perron 场与最外负根场的线性互素性：\texorpdfstring{$q\in\{9,10,11,13\}$}{q in \{9,10,11,13\}}]\label{prop:pom-resonance-perron-negative-root-field-coprime-q9-13}
\leanverified{paper\_pom\_resonance\_perron\_negative\_root\_field\_coprime\_q9\_13}
取 \(q\in\{9,10,11,13\}\)，令
\[
F_q:=\QQ(r_q),\qquad E_q:=\QQ(\lambda_q^-),\qquad d:=\deg P_q.
\]
则
\[
F_q\cap E_q=\QQ,
\qquad
[\QQ(r_q,\lambda_q^-):\QQ]=d(d-1).
\]
\end{proposition}
\begin{proof}
令 \(K_q\) 为 \(P_q\) 的分裂域，记根为 \(\alpha_1,\dots,\alpha_d\)，其中 \(\alpha_1=r_q\) 且 \(\alpha_2=\lambda_q^-\)。
由命题 \ref{prop:pom-observable-minpoly-galois-sd-q9-13}，有 \(\Gal(K_q/\QQ)\cong S_d\)。
在该同构下，点稳定子 \(\mathrm{Stab}(i)\cong S_{d-1}\) 的不动域正是单根生成域：
\[
F_q=K_q^{\mathrm{Stab}(1)},
\qquad
E_q=K_q^{\mathrm{Stab}(2)}.
\]
又因两点稳定子生成整个对称群（例如 \((1,3)\in\mathrm{Stab}(2)\)、\((2,3)\in\mathrm{Stab}(1)\) 从而 \((1,2)=(1,3)(2,3)(1,3)\) 属于所生成子群，进而生成全体相邻换位），得
\[
F_q\cap E_q
=K_q^{\langle \mathrm{Stab}(1),\mathrm{Stab}(2)\rangle}
=K_q^{S_d}
=\QQ.
\]
进一步，
\[
[\QQ(r_q,\lambda_q^-):\QQ]
=[K_q^{\mathrm{Stab}(1)\cap \mathrm{Stab}(2)}:\QQ]
=[K_q^{S_{d-2}}:\QQ]
=[S_d:S_{d-2}]
=d(d-1).
\]
证毕。
\end{proof}

\begin{proposition}[Newman 比值场与判别式二次子域的互素性；二次子域不存在性]\label{prop:pom-resonance-newman-field-discriminant-quadratic-disjoint}
\leanverified{paper\_pom\_resonance\_newman\_field\_discriminant\_quadratic\_disjoint}
取 \(q\in\{9,10,11,13\}\)，令
\[
L_q:=\QQ(\upsilon_q)=\QQ\!\left(\frac{(\lambda_q^-)^2}{r_q}\right),
\qquad
k_q:=\QQ\!\left(\sqrt{\Disc(P_q)}\right)\subseteq K_q,
\]
其中 \(K_q\) 为 \(P_q\) 的分裂域。
则
\[
L_q\cap k_q=\QQ.
\]
特别地，\(L_q\) 不含任何二次子域。
\end{proposition}
\begin{proof}
由命题 \ref{prop:pom-observable-minpoly-galois-sd-q9-13}，\(\Gal(K_q/\QQ)\cong S_d\)（\(d=\deg P_q\)）。
设 \(A_d\triangleleft S_d\) 为交错群，则 \(K_q^{A_d}=\QQ(\sqrt{\Disc(P_q)})=k_q\) 为 \(K_q/\QQ\) 的唯一非平凡二次子扩张。
另一方面，令
\[
H:=\mathrm{Stab}(1,2)=\{\sigma\in S_d:\ \sigma(1)=1,\ \sigma(2)=2\}\cong S_{d-2}.
\]
则 \(H\) 固定 \(\alpha_1=r_q\) 与 \(\alpha_2=\lambda_q^-\)，从而固定 \(\upsilon_q=\alpha_2^2/\alpha_1\)，即 \(L_q\subseteq K_q^{H}\)。
注意 \(H\) 含奇置换（例如当 \(d\ge 4\) 时 \((3,4)\in H\)），故 \(\langle H,A_d\rangle=S_d\)。
于是
\[
K_q^{H}\cap K_q^{A_d}=K_q^{\langle H,A_d\rangle}=K_q^{S_d}=\QQ,
\]
从而 \(L_q\cap k_q=\QQ\)。
若 \(L_q\) 含二次子域 \(M\)，则 \(M\subseteq K_q\) 且 \([M:\QQ]=2\)，必有 \(M=k_q\)，与 \(L_q\cap k_q=\QQ\) 矛盾。证毕。
\end{proof}

\begin{proposition}[Newman 比值的非降阶性：不落入单根生成的次数 \texorpdfstring{$d$}{d} 子域]\label{prop:pom-resonance-newman-not-in-single-root-fields-q9-13}
\leanverified{paper\_pom\_resonance\_newman\_not\_in\_single\_root\_fields\_q9\_13}
取 \(q\in\{9,10,11,13\}\)，记 \(d:=\deg P_q\)（故 \(d\ge 7\)），并令
\(\upsilon_q=(\lambda_q^-)^2/r_q\)。
则
\[
\upsilon_q\notin\QQ(r_q),
\qquad
\upsilon_q\notin\QQ(\lambda_q^-).
\]
\end{proposition}
\begin{proof}
由命题 \ref{prop:pom-resonance-perron-negative-root-field-coprime-q9-13}，
\[
[\QQ(r_q,\lambda_q^-):\QQ(r_q)]=d-1\ge 6,
\qquad
[\QQ(r_q,\lambda_q^-):\QQ(\lambda_q^-)]=d-1\ge 6.
\]
若 \(\upsilon_q\in\QQ(r_q)\)，则 \((\lambda_q^-)^2=r_q\upsilon_q\in\QQ(r_q)\)；
从而 \(\lambda_q^-\) 在 \(\QQ(r_q)\) 上的次数至多为 \(2\)，与相对次数 \(d-1\ge 6\) 矛盾。
因此 \(\upsilon_q\notin\QQ(r_q)\)。
对 \(\QQ(\lambda_q^-)\) 的情形同理：若 \(\upsilon_q\in\QQ(\lambda_q^-)\)，则 \(r_q=(\lambda_q^-)^2/\upsilon_q\in\QQ(\lambda_q^-)\)，同样推出相对次数至多为 \(2\)，矛盾。证毕。
\end{proof}
