\paragraph{互补分歧、显式三挠 Kummer 生成与 \texorpdfstring{$A_2$}{A2} 型 Prym 刚性}
在命题 \ref{prop:cdim-resolvent-cubic-branchpoints-and-complementary-ramification}、结论 \ref{con:cdim-s4-v4-etale-c3-torsor}、命题 \ref{prop:cdim-leyang-sixth-norm-identity-and-3torsion}、命题 \ref{prop:cdim-s4-v4-kummer-model-and-resolvent-recovery}、结论 \ref{con:cdim-s4-v4-triple-bielliptic}、结论 \ref{con:cdim-s4-v4-prym-13-polarization} 与命题 \ref{prop:cdim-s4-v4-prym-a2-polarization-class} 的基础上，还可把 Lee--Yang 基曲线、三次预解式曲线及其 \(S_3\)-闭包之间的几何关系进一步压缩为一条完全闭合的代数链：三次预解式的 \((2,1)\)-型分歧纤维把双覆盖分歧点规范化为一个次数 \(6\) 的互补分歧除子；这一双覆盖与 genus-\(2\) 基曲线上的显式三挠 Kummer 元共同锁定循环三重 torsor；而 Prym 曲面的极化又被 \(C_3\)-对称唯一刚化为 \(A_2\)-Cartan 型。

\begin{theorem}[互补分歧六点在 \texorpdfstring{$E_{\mathrm{res}}$}{Eres} 上的线性等价]\label{thm:cdim-s4-v4-complementary-ramification-linear-equivalence}
\leanverified{paper\_cdim\_s4\_v4\_complementary\_ramification\_linear\_equivalence}
设
\[
\pi_{\mathrm{res}}:E_{\mathrm{res}}\longrightarrow \PP^1_y
\]
为三次预解式投影，\(R_{\pi}\) 为其 ramification divisor，并记 \(B_{\mathrm{comp}}\) 为命题 \ref{prop:cdim-resolvent-cubic-branchpoints-and-complementary-ramification} 中双覆盖
\[
C_{S_3}\longrightarrow E_{\mathrm{res}}
\]
的六个互补分歧点之和。若定义基底上的六点分歧除子
\[
D_{\mathrm{br}}
:=
[0]+[1]+[\infty]+\sum_{P(\alpha)=0}[\alpha]\in \Div(\PP^1_y),
\]
则在 \(\Div(E_{\mathrm{res}})\) 中有
\[
\pi_{\mathrm{res}}^*D_{\mathrm{br}}=2R_{\pi}+B_{\mathrm{comp}}.
\]
此外，
\[
R_{\pi}\sim 2\,\pi_{\mathrm{res}}^*[\infty],
\qquad
B_{\mathrm{comp}}\sim 2\,\pi_{\mathrm{res}}^*[\infty].
\]
\end{theorem}
\begin{proof}
由结论 \ref{con:cdim-s4-v4-six-branch-transposition} 与命题 \ref{prop:cdim-resolvent-cubic-branchpoints-and-complementary-ramification}，\(\pi_{\mathrm{res}}\) 的分歧值恰为
\[
\{0,1,\infty\}\cup\{P(y)=0\},
\]
并且每条分歧纤维都具有 \((2,1)\) 型：纤维上恰有一个简单分歧点与一个互补的非分歧点。于是把六条分歧纤维相加，立即得到
\[
\pi_{\mathrm{res}}^*D_{\mathrm{br}}=2R_{\pi}+B_{\mathrm{comp}}.
\]

由于 \(E_{\mathrm{res}}\) 为亏格 \(1\) 曲线，有 \(K_{E_{\mathrm{res}}}\sim 0\)。对三次覆盖 \(\pi_{\mathrm{res}}\) 应用 Riemann--Hurwitz，
\[
K_{E_{\mathrm{res}}}\sim \pi_{\mathrm{res}}^*K_{\PP^1}+R_{\pi}
\sim -2\,\pi_{\mathrm{res}}^*[\infty]+R_{\pi},
\]
故
\[
R_{\pi}\sim 2\,\pi_{\mathrm{res}}^*[\infty].
\]
另一方面，\(\PP^1\) 上任一次数 \(6\) 的除子都与 \(6[\infty]\) 线性等价，故
\[
D_{\mathrm{br}}\sim 6[\infty].
\]
将其代入上式并使用 \(R_{\pi}\sim 2\,\pi_{\mathrm{res}}^*[\infty]\)，得到
\[
B_{\mathrm{comp}}
\sim
\pi_{\mathrm{res}}^*D_{\mathrm{br}}-2R_{\pi}
\sim
6\,\pi_{\mathrm{res}}^*[\infty]-4\,\pi_{\mathrm{res}}^*[\infty]
=
2\,\pi_{\mathrm{res}}^*[\infty].
\]
证毕。
\end{proof}

\begin{corollary}[互补分歧双覆盖的次数 \texorpdfstring{$3$}{3} 根线束正规形]\label{cor:cdim-s4-v4-complementary-branch-square-root-line-bundle}
\leanverified{paper\_cdim\_s4\_v4\_complementary\_branch\_square\_root\_line\_bundle}
存在次数恰为 \(3\) 的线束
\[
L_{\mathrm{comp}}\in \operatorname{Pic}^3(E_{\mathrm{res}})
\]
使得
\[
L_{\mathrm{comp}}^{\otimes 2}\cong \mathcal O_{E_{\mathrm{res}}}(B_{\mathrm{comp}}),
\]
并且可取
\[
L_{\mathrm{comp}}\cong \mathcal O_{E_{\mathrm{res}}}\!\bigl(\pi_{\mathrm{res}}^*[\infty]\bigr).
\]
因此双覆盖 \(C_{S_3}\to E_{\mathrm{res}}\) 由一个次数 \(3\) 的平方根线束完全规范化。
\end{corollary}
\begin{proof}
由定理 \ref{thm:cdim-s4-v4-complementary-ramification-linear-equivalence}，
\[
B_{\mathrm{comp}}\sim 2\,\pi_{\mathrm{res}}^*[\infty].
\]
取
\[
L_{\mathrm{comp}}:=\mathcal O_{E_{\mathrm{res}}}\!\bigl(\pi_{\mathrm{res}}^*[\infty]\bigr),
\]
便有
\[
L_{\mathrm{comp}}^{\otimes 2}\cong \mathcal O_{E_{\mathrm{res}}}(B_{\mathrm{comp}}).
\]
又因 \(\deg\pi_{\mathrm{res}}=3\)，故
\[
\deg L_{\mathrm{comp}}=\deg \pi_{\mathrm{res}}^*[\infty]=3.
\]
对椭圆曲线上的分歧双覆盖，其数据等价于一个分歧除子 \(B\) 及其平方根线束 \(L\) 满足 \(L^{\otimes 2}\cong\mathcal O(B)\)，于是结论成立。证毕。
\end{proof}

\begin{theorem}[循环三重 torsor 的显式三挠 Kummer 生成]\label{thm:cdim-s4-v4-kummer-torsor-generated-by-explicit-3torsion}
设 \(K:=\QQ(\sqrt{-3})\)，并记
\[
\varpi_K:C_{S_3,K}\longrightarrow H_{\mathrm{LY},K}
\]
为结论 \ref{con:cdim-s4-v4-etale-c3-torsor} 的基变换。取
\[
Q_\pm:\ y=-\frac12,\qquad u=\pm \frac94\sqrt{-3},
\]
并记
\[
\eta:=[Q_+-Q_-]\in \mathrm{Jac}(H_{\mathrm{LY}})(K)[3].
\]
若定义
\[
\Theta=
\frac{A(y)+3\sqrt{-3}\,u}{16(2y+1)^3},
\qquad
A(y)=128y^3+219y^2+69y+16,
\]
则
\[
\operatorname{div}(\Theta)=3(Q_+-Q_-),
\]
且在函数域上有
\[
K(C_{S_3})\otimes_{\QQ}K
=
K(H_{\mathrm{LY}})(W),
\qquad
W^3=\Theta.
\]
因此 \(\varpi_K\) 正是由三挠子类 \(\eta\) 所定义的 Kummer \(C_3\)-torsor。
\end{theorem}
\begin{proof}
命题 \ref{prop:cdim-leyang-sixth-norm-identity-and-3torsion} 已给出
\[
\operatorname{div}(\Theta)=3(Q_+-Q_-),
\]
故 \(\eta=[Q_+-Q_-]\) 为 \(K\)-有理三挠子类。另一方面，命题 \ref{prop:cdim-s4-v4-kummer-model-and-resolvent-recovery} 表明，在 \(K\) 上有 Kummer 模型
\[
C_{S_3}\times_{\QQ}K \dashrightarrow \{(y,u,W):u^2=-y(y-1)P(y),\ W^3=\Theta\},
\]
从而相应函数域恰为
\[
K(H_{\mathrm{LY}})(W),\qquad W^3=\Theta.
\]
在含 \(\zeta_3\) 的基域上，étale 循环三重覆盖与三挠线束类的 Kummer 描述一一对应；而 \(\Theta\) 的除子恰为 \(3(Q_+-Q_-)\)，故该 torsor 对应的正是 \(\eta\)。证毕。
\end{proof}
\leanverified{paper\_cdim\_s4\_v4\_kummer\_torsor\_generated\_by\_explicit\_3torsion}

\begin{theorem}[pullback 核的精确 \texorpdfstring{$C_3$}{C3} 生成与 Jacobian 等变分裂]\label{thm:cdim-s4-v4-jacobian-pullback-kernel-and-prym-splitting}
\leanverified{paper\_cdim\_s4\_v4\_jacobian\_pullback\_kernel\_and\_prym\_splitting}
沿用定理 \ref{thm:cdim-s4-v4-kummer-torsor-generated-by-explicit-3torsion} 的记号，并置
\[
P_3:=\mathcal P_3=\ker\!\bigl(\mathrm{Nm}_{\varpi}:\mathrm{Jac}(C_{S_3})\to \mathrm{Jac}(H_{\mathrm{LY}})\bigr)^0.
\]
则 pullback 同态
\[
\varpi_K^*:\mathrm{Jac}(H_{\mathrm{LY},K})\longrightarrow \mathrm{Jac}(C_{S_3,K})
\]
满足
\[
\ker(\varpi_K^*)=\langle \eta\rangle\cong \ZZ/3\ZZ.
\]
此外，在 \(\QQ\)-同源范畴中有
\[
\mathrm{Jac}(C_{S_3})\sim_{\QQ}\mathrm{Jac}(H_{\mathrm{LY}})\times P_3,
\qquad
\dim P_3=2,
\qquad
P_3\sim_{\QQ}E_{\mathrm{res}}^2.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:cdim-s4-v4-kummer-torsor-generated-by-explicit-3torsion}，\(\varpi_K\) 是由三挠线束 \(\eta\) 所定义的 étale 循环三重覆盖，因此
\[
(\varpi_K)_*\mathcal O_{C_{S_3,K}}
\cong
\mathcal O_{H_{\mathrm{LY},K}}
\oplus \eta^{-1}\oplus \eta^{-2}.
\]
若 \(L\in \operatorname{Pic}^0(H_{\mathrm{LY},K})\) 满足 \(\varpi_K^*L\cong \mathcal O_{C_{S_3,K}}\)，则
\[
0\neq H^0\!\bigl(C_{S_3,K},\varpi_K^*L\bigr)
\cong
H^0(H_{\mathrm{LY},K},L)\oplus
H^0(H_{\mathrm{LY},K},L\otimes\eta^{-1})\oplus
H^0(H_{\mathrm{LY},K},L\otimes\eta^{-2}).
\]
对 genus-\(2\) 曲线上的次数 \(0\) 线束，存在非零截面当且仅当该线束平凡，故 \(L\) 只能取
\[
\mathcal O,\qquad \eta,\qquad \eta^2.
\]
于是
\[
\ker(\varpi_K^*)=\{0,\eta,\eta^2\}=\langle\eta\rangle.
\]

后半部分即结论 \ref{con:cdim-s4-v4-prym-13-polarization} 的内容：由范数--拉回分解与 Prym 的定义，
\[
\mathrm{Jac}(C_{S_3})\sim_{\QQ}\mathrm{Jac}(H_{\mathrm{LY}})\times P_3,
\qquad
\dim P_3=g(C_{S_3})-g(H_{\mathrm{LY}})=2,
\]
并且
\[
P_3\sim_{\QQ}E_{\mathrm{res}}^2.
\]
证毕。
\end{proof}

\begin{theorem}[Prym 曲面的 \texorpdfstring{$A_2$}{A2} 型极化刚性]\label{thm:cdim-s4-v4-prym-a2-polarized-isogeny-rigidity}
设 \(P_3\) 如定理 \ref{thm:cdim-s4-v4-jacobian-pullback-kernel-and-prym-splitting}。若进一步采用结论 \ref{con:xi-terminal-zm-resolvent-cubic-weierstrass-birational} 与命题 \ref{prop:xi-terminal-zm-resolvent-cubic-no-cm-max-end} 所给出的无复乘事实
\[
\End_{\overline{\QQ}}(E_{\mathrm{res}})=\ZZ,
\qquad
P_3\sim_{\QQ}E_{\mathrm{res}}^2,
\]
则：
\begin{enumerate}
  \item \(C_3\) 的生成元在 \(E_{\mathrm{res}}^2\) 的同源模型上可唯一取标准代表
  \[
  \tau=
  \begin{pmatrix}
  0&-1\\
  1&-1
  \end{pmatrix},
  \qquad \tau^3=I;
  \]
  \item 所有 \(\tau\)-不变的正定整极化类恰为
  \[
  S_k=
  k\begin{pmatrix}
  2&-1\\
  -1&2
  \end{pmatrix},
  \qquad k\in\ZZ_{>0};
  \]
  \item 对自然 Prym 极化 \(\lambda_{P_3}\) 而言，唯一可能的是 \(k=1\)，因而其极化同类在同源意义下唯一等价于 \(A_2\)-Cartan 形式
  \[
  \begin{pmatrix}
  2&-1\\
  -1&2
  \end{pmatrix}.
  \]
\end{enumerate}
\leanverified{paper\_cdim\_s4\_v4\_prym\_a2\_polarized\_isogeny\_rigidity}
\end{theorem}
\begin{proof}
这正是命题 \ref{prop:cdim-s4-v4-prym-a2-polarization-class} 的内容。其核心是：在 \(\End(E_{\mathrm{res}}^2)\cong M_2(\ZZ)\) 中，阶 \(3\) 元的 \(\GL_2(\ZZ)\)-共轭类唯一，故可取标准代表 \(\tau\)；而方程
\[
\tau^{\mathsf T}S\tau=S
\]
对对称整矩阵 \(S\) 的正定解恰为 \(S_k=k\begin{psmallmatrix}2&-1\\ -1&2\end{psmallmatrix}\)，其行列式为 \(3k^2\)。由于 \(P_3\) 的自然 Prym 极化型为 \((1,3)\)，其同类矩阵行列式只能为 \(3\)，故必有 \(k=1\)。证毕。
\end{proof}

\begin{corollary}[自然 Prym 极化绝非 genus-\texorpdfstring{$2$}{2} Jacobian 的主极化模型]\label{cor:cdim-s4-v4-prym-not-principally-polarized-genus2-jacobian}
带自然 Prym 极化的阿贝尔曲面 \((P_3,\lambda_{P_3})\) 不可能与任何 genus-\(2\) 曲线的 Jacobian 在极化意义下同构。
\end{corollary}
\begin{proof}
任意 genus-\(2\) Jacobian 都携带规范主极化，其极化型为 \((1,1)\)。而结论 \ref{con:cdim-s4-v4-prym-13-polarization} 与定理 \ref{thm:cdim-s4-v4-prym-a2-polarized-isogeny-rigidity} 已表明 \((P_3,\lambda_{P_3})\) 的自然极化型为 \((1,3)\)，并且在 \(C_3\)-对称约束下其极化同类唯一刚化为 \(A_2\)-Cartan 型。故它不可能作为主极化 genus-\(2\) Jacobian 出现。证毕。
\end{proof}
\leanverified{paper\_cdim\_s4\_v4\_prym\_not\_principally\_polarized\_genus2\_jacobian}

\begin{theorem}[兼容双椭圆 pencil 的精确数目与 \texorpdfstring{$C_3$}{C3} 传递性]\label{thm:cdim-s4-v4-compatible-bielliptic-pencils-exactly-three}
\leanverified{paper\_cdim\_s4\_v4\_compatible\_bielliptic\_pencils\_exactly\_three}
称次数 \(2\) 态射
\[
q:C_{S_3}\longrightarrow E
\]
到某条椭圆曲线 \(E\) 为与 \(y\)-投影兼容，若对应的函数域 \(\QQ(E)\) 是 \(S_3\)-Galois 扩张
\[
\QQ(C_{S_3})/\QQ(y)
\]
中的一个指数 \(2\) 中间域。则这样的兼容双椭圆 pencil 恰有三条。更具体地，存在三条映射
\[
q_i:C_{S_3}\longrightarrow E_{\mathrm{res}}
\qquad (i=1,2,3),
\]
使得每一条与 \(y\)-投影兼容的双椭圆 pencil 都与其中某个 \(q_i\) 同构，并且 deck 群 \(C_3\) 在这三条 pencil 上简单传递。
\end{theorem}
\begin{proof}
由结论 \ref{con:cdim-s4-v4-fixedfield-fiber-product}，
\[
\QQ(C_{S_3})/\QQ(y)
\]
的 Galois 群为 \(S_3\)。兼容的次数 \(2\) 中间商与指数 \(2\) 中间域一一对应，而指数 \(2\) 中间域又与 \(S_3\) 的阶 \(2\) 子群一一对应。由于 \(S_3\) 恰有三条阶 \(2\) 子群，故兼容双椭圆 pencil 恰有三条。

又由结论 \ref{con:cdim-s4-v4-triple-bielliptic}，每一条这样的商曲线都与 \(E_{\mathrm{res}}\) 在 \(\QQ\) 上同构。记其代表为
\[
q_i:C_{S_3}\to E_{\mathrm{res}},\qquad i=1,2,3.
\]
正规子群 \(C_3\triangleleft S_3\) 通过共轭在三条阶 \(2\) 子群上循环传递，因此在三条 pencil 上也传递。由于作用群与轨道都具有基数 \(3\)，该作用自动是简单传递。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$C_3$}{C3} deck 与双椭圆 involution 的极小闭包]\label{thm:cdim-s4-v4-deck-and-bielliptic-involution-generate-s3}
取结论 \ref{con:cdim-s4-v4-etale-c3-torsor} 中循环 étale 三重覆盖 \(\varpi:C_{S_3}\to H_{\mathrm{LY}}\) 的 deck 生成元
\[
\tau\in \Aut(C_{S_3}),
\qquad
\langle\tau\rangle\cong C_3,
\]
并取定理 \ref{thm:cdim-s4-v4-compatible-bielliptic-pencils-exactly-three} 中某条双椭圆 pencil 所对应的 involution
\[
\sigma\in \Aut(C_{S_3}),
\qquad
\langle\sigma\rangle\cong C_2.
\]
则
\[
\langle \sigma,\tau\rangle\cong S_3.
\]
并且该群在三条兼容双椭圆 pencil 上的作用，正是 \(S_3\) 在其三条换位子群上的自然共轭作用。
\end{theorem}
\begin{proof}
由结论 \ref{con:cdim-s4-v4-fixedfield-fiber-product}，
\[
\Aut_{\QQ(y)}(C_{S_3})\cong \Gal(\QQ(C_{S_3})/\QQ(y))\cong S_3.
\]
其中 \(\tau\) 对应一个 \(3\)-cycle，而 \(\sigma\) 对应一条阶 \(2\) 子群的生成元，即一个换位。任何 \(S_3\) 中的非平凡换位与 \(3\)-cycle 都共同生成全群，因此
\[
\langle \sigma,\tau\rangle=S_3
\]
作为 \(\Aut_{\QQ(y)}(C_{S_3})\) 的子群成立。

第二部分只需把定理 \ref{thm:cdim-s4-v4-compatible-bielliptic-pencils-exactly-three} 中“三条兼容双椭圆 pencil 与三条阶 \(2\) 子群对应”的描述，与 \(S_3\) 对其换位子群的共轭作用相比较即可。证毕。
\end{proof}
\leanverified{paper\_cdim\_s4\_v4\_deck\_and\_bielliptic\_involution\_generate\_s3}

\noindent
由此，三次预解式的简单分歧、Lee--Yang 基曲线上的显式三挠 Kummer 元、循环三重 Prym 的 \(A_2\)-型极化以及三条双椭圆 pencil 的 \(S_3\)-对称，不再是彼此平行的结构片段，而共同组成一条从互补分歧到极化刚性的闭链：双覆盖分歧点被 \((2,1)\)-型纤维唯一规范化，étale 三重层被单个三挠子类 \(\eta=[Q_+-Q_-]\) 完全生成，而 Prym 极化则在无复乘背景下被唯一锁定到 \(A_2\)-Cartan 型，最终与 deck 变换和双椭圆 involution 一起重新闭合为整个 \(S_3\) 自动同构群。
