\subsubsection{坏可逼近系统、审计稳定性与临界幂律常数}\label{subsubsec:cdim-badly-approximable-audit-stability}
定理 \ref{thm:cdim-dyadic-prefix-audit-exact} 的上界部分使用了坏可逼近系统的存在性以给出主指数的可达性。
下面将该存在性提升为一个等价刻画：对线性相位同态 \(\Phi_\Theta\)，
“在所有尺度上保持临界幂律分离”当且仅当 \(\Theta\) 坏可逼近。
该刻画把“审计深度增长率”与“几何数论中的刚性可逼近性类”完全对齐。

\begin{definition}[审计稳定性的分离度下界]\label{def:cdim-audit-stability-separation}
在定义 \ref{def:cdim-dyadic-prefix-audit-depth} 的记号下，
对 \(\Theta\in M_{d\times r}(\RR)\) 定义
\[
\Delta_Q(\Theta):=\min\Bigl\{\operatorname{dist}_{\TT^d,\infty}(\Theta k,0):\ 0\neq k\in\ZZ^r,\ |k|_\infty\le Q\Bigr\}.
\]
称 \(\Theta\) 在 \((r,d)\) 上\textbf{审计稳定}，若存在常数 \(c>0\) 使对一切 \(Q\ge 1\) 有
\[
\Delta_Q(\Theta)\ \ge\ c\,Q^{-r/d}.
\]
\end{definition}

\begin{proposition}[审计稳定性等价于坏可逼近]\label{prop:cdim-audit-stability-iff-badly-approximable}
对 \(\Theta\in M_{d\times r}(\RR)\)，下列条件等价：
\begin{enumerate}
  \item \(\Theta\) 满足定义 \ref{def:cdim-audit-stability-separation} 的审计稳定性；
  \item 存在常数 \(c'>0\) 使对一切 \(0\neq k\in\ZZ^r\) 有
  \[
  \operatorname{dist}_{\TT^d,\infty}(\Theta k,0)\ \ge\ c'\,|k|_\infty^{-r/d},
  \]
  即 \(\Theta\) 为坏可逼近系统（badly approximable system of linear forms）的 \(\ell^\infty\) 版本。
\end{enumerate}
\leanverified{paper\_cdim\_audit\_stability\_iff\_badly\_approximable}
\leanverified{audit\_stability\_iff\_badly\_approximable}
\end{proposition}
\begin{proof}
(2)\(\Rightarrow\)(1) 直接取 \(|k|_\infty\le Q\) 的下界即得。
\par
(1)\(\Rightarrow\)(2) 对任意 \(0\neq k\in\ZZ^r\)，取 \(Q:=|k|_\infty\)，则
\[
\operatorname{dist}_{\TT^d,\infty}(\Theta k,0)
\ge \Delta_Q(\Theta)
\ge c\,Q^{-r/d}
=c\,|k|_\infty^{-r/d}.
\]
取 \(c':=c\) 即得 (2)。证毕。
\end{proof}

\begin{corollary}[临界幂律下的深度极限]\label{cor:cdim-audit-stable-depth-limit}
若 \(\Theta\) 审计稳定，则最小审计深度满足
\[
\lim_{Q\to\infty}\frac{L_{\Theta}(Q)}{\log_2 Q}=\frac{r}{d}.
\]
\end{corollary}
\leanverified{paper\_cdim\_audit\_stable\_depth\_limit}
\begin{proof}
由定理 \ref{thm:cdim-dyadic-prefix-audit-exact} 的下界部分对任意 \(\Theta\) 有
\[
\liminf_{Q\to\infty}\frac{L_{\Theta}(Q)}{\log_2 Q}\ \ge\ \frac{r}{d}.
\]
若 \(\Theta\) 审计稳定，则由定义 \ref{def:cdim-dyadic-prefix-audit-depth} 与审计稳定性不等式
\(
\Delta_Q(\Theta)\ge cQ^{-r/d}
\)
可复刻定理 \ref{thm:cdim-dyadic-prefix-audit-exact} 上界证明中的分离度\(\Rightarrow\)dyadic 盒单射推理，从而得到
\(
L_{\Theta}(Q)\le \frac{r}{d}\log_2 Q+O(1)
\)，
从而
\(
\limsup_{Q\to\infty}\frac{L_{\Theta}(Q)}{\log_2 Q}\le r/d
\)。
两式合并得到极限。证毕。
\end{proof}

\begin{conjecture}[多维 Hurwitz 常数与临界行列式]\label{conj:cdim-hurwitz-constant-critical-determinant}
对给定 \(r>d\ge 1\)，令 \(\mathrm{BA}(d,r)\) 为坏可逼近矩阵集合，并定义
\[
H_{r,d}:=\sup_{\Theta\in\mathrm{BA}(d,r)}\ \inf_{0\neq k\in\ZZ^r}\ |k|_\infty^{r/d}\,\operatorname{dist}_{\TT^d,\infty}(\Theta k,0).
\]
猜想 \(H_{r,d}\) 为正且有限，并且由某个代数构造达到。
进一步猜想 \(H_{r,d}\) 等价于某个 \((r{+}d)\)-维星形体的临界行列式（critical determinant）或其对偶极值，从而与广义 Hermite 常数在刚性意义下同一。
\end{conjecture}
