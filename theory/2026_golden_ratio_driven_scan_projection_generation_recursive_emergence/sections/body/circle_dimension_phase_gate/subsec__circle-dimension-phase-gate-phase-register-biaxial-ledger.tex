\subsection{相位圆维与寄存圆维的双轴计量：连通圆因子与全不连通纤维的统一账本}\label{subsec:cdim-phase-register-biaxial-ledger}
定义 \ref{def:circle-dimension} 与定义 \ref{def:half-circle-dimension} 将可见相位通道编码为连通圆因子的计数。
本小节引入与其同量纲、但作用于全不连通紧纤维的寄存圆维 \(\pcdim\)（定义 \ref{def:cdim-pro-circle-dimension}），并给出两条可直接调用的结果链：其一是 \(\pcdim\) 的模素数刻画；其二是秩一/高秩有理型对象在 Pontryagin 对偶下的“环面相位 + \(\mathfrak n\)-adic 寄存器”分解。

\begin{theorem}[寄存圆维的等价刻画与模素数公式]\label{thm:cdim-register-circle-modp-formula}
设 \(K\) 为拓扑有限生成 profinite 交换群。
则以下量相等：
\[
\pcdim(K),
\quad
\min\bigl\{d:\exists\ \widehat{\ZZ}^{\,d}\twoheadrightarrow K\bigr\},
\quad
\sup_{p\ \mathrm{prime}}\dim_{\FF_p}(K/pK).
\]
此外，若 \(K\simeq\prod_p K_p\) 为 Sylow 分解，则
\[
\pcdim(K)=\sup_p \pcdim(K_p),
\qquad
\dim_{\FF_p}(K/pK)=\dim_{\FF_p}(K_p/pK_p).
\]
\end{theorem}
\leanverified{paper\_cdim\_register\_circle\_modp\_formula}
\begin{proof}
由备注 \ref{rem:cdim-pro-cdim-zhat-surjection}，\(\pcdim(K)\) 等于使 \(\widehat{\ZZ}^{\,d}\twoheadrightarrow K\) 成立的最小 \(d\)，且等于最小拓扑生成元个数。

对每个素数 \(p\)，\(K_p\) 为交换 pro-\(p\) 群，其 Frattini 子群为 \(pK_p\)；
故最小拓扑生成元个数满足
\[
\pcdim(K_p)=\dim_{\FF_p}(K_p/pK_p).
\]
再由命题 \ref{prop:cdim-pro-cdim-crt-reuse} 的 Sylow 乘积分解与生成元复用律，
\[
\pcdim(K)=\sup_p \pcdim(K_p).
\]
另一方面，\(K/pK\) 仅由 \(p\)-Sylow 分量贡献，故
\[
K/pK\cong K_p/pK_p,
\]
从而
\[
\sup_p\dim_{\FF_p}(K/pK)
=\sup_p\dim_{\FF_p}(K_p/pK_p)
=\sup_p\pcdim(K_p)
=\pcdim(K).
\]
证毕。
\end{proof}

\begin{definition}[秩一分母高度与超自然谱]\label{def:cdim-rankone-height-supernatural}
设 \(A\subseteq \QQ\) 为秩一无挠交换群且 \(1\in A\)。
对每个素数 \(p\)，定义
\[
h_p(A):=\sup\{k\in\NN:\ p^{-k}\in A\}\in\NN\cup\{0,\infty\},
\]
并记超自然数
\[
\mathfrak n(A):=\prod_p p^{h_p(A)}.
\]
定义
\[
\ZZ_{\mathfrak n(A)}:=\varprojlim_{m\mid\mathfrak n(A)}\ZZ/m\ZZ.
\]
\end{definition}

\begin{theorem}[秩一有理群对偶的相位--素数纤维分解]\label{thm:cdim-rankone-phase-prime-fiber}
设 \(A\subseteq \QQ\) 满足定义 \ref{def:cdim-rankone-height-supernatural}，令 \(X:=\widehat A\)，并令
\[
\pi_A:X\twoheadrightarrow \TT
\]
为包含 \(\ZZ\hookrightarrow A\) 的对偶满射。
则存在短正合列
\[
0\longrightarrow \ZZ_{\mathfrak n(A)}
\longrightarrow X
\xrightarrow{\ \pi_A\ }
\TT
\longrightarrow 0.
\]
进一步，取任意满足 \(m_0=1,\ m_k\mid m_{k+1},\ m_k\mid\mathfrak n(A)\) 且对每个 \(p\) 的指数趋近 \(h_p(A)\) 的链 \((m_k)\)，则有非典范同构
\[
X\cong
\varprojlim_{k}
\Bigl(
\TT\xleftarrow{z\mapsto z^{\,m_{k+1}/m_k}}\TT
\Bigr).
\]
\end{theorem}
\begin{proof}
由短正合列 \(0\to\ZZ\to A\to A/\ZZ\to 0\) 对偶化得
\[
0\to \widehat{A/\ZZ}\to \widehat A\to \widehat\ZZ\to 0,
\qquad
\widehat\ZZ\cong\TT.
\]
因此 \(\ker(\pi_A)\cong\widehat{A/\ZZ}\)。

对每个素数 \(p\)，\(A/\ZZ\) 的 \(p\)-部分由高度 \(h_p(A)\) 决定：
当 \(h_p(A)=m<\infty\) 时为 \(\ZZ/p^m\ZZ\)，当 \(h_p(A)=\infty\) 时为 Prüfer 群 \(\ZZ(p^\infty)\)。
故
\[
A/\ZZ\cong
\bigoplus_{p:\,h_p<\infty}\ZZ/p^{h_p(A)}\ZZ
\ \oplus\
\bigoplus_{p:\,h_p=\infty}\ZZ(p^\infty),
\]
对偶得到
\[
\widehat{A/\ZZ}\cong
\prod_{p:\,h_p<\infty}\ZZ/p^{h_p(A)}\ZZ
\times
\prod_{p:\,h_p=\infty}\ZZ_p
\cong
\ZZ_{\mathfrak n(A)}.
\]
于是得到所示短正合列。

另一方面，\(A=\varinjlim_k m_k^{-1}\ZZ\)，连接映射为乘 \(m_{k+1}/m_k\)。
对偶后即得所示圆周覆盖逆极限表达。证毕。
\end{proof}

\begin{remark}[规范纤维的商最小性]\label{rem:cdim-rankone-register-minimality}
在定理 \ref{thm:cdim-rankone-phase-prime-fiber} 的规范可见读出 \(\pi_A\) 下，纤维
\[
K_A:=\ker(\pi_A)\cong \ZZ_{\mathfrak n(A)}
\]
由分母高度 \(h_p(A)\) 唯一确定。
若保持同一可见相位轴并仅作纤维商化，即取任意闭子群 \(H\le K_A\) 并考虑商读出 \(X/H\twoheadrightarrow\TT\)，其新纤维必为 \(K_A/H\)，因此始终是 \(K_A\) 的连续商。
这给出“相位轴固定时，素数寄存器由规范纤维主导”的最小性口径。
\end{remark}

\begin{proposition}[常见对象的双轴维度赋值]\label{prop:cdim-phase-register-common-assignments}
记连续相位轴为 \(\cdim\)（或非有限生成情形下的 \(\cdim^\ast\)），寄存轴为 \(\pcdim\)。
则有：
\begin{enumerate}
  \item \(\cdim(\ZZ)=1,\ \cdim_+(\NN)=\tfrac12\)；
  \item 对有限交换群 \(F\)，\(\cdim(F)=0\)；
  \item \(\cdim(\ZZ^k)=k,\ \cdim_+(\NN^k)=\tfrac{k}{2}\)；
  \item 若 \(\mathcal O_K\) 为次数 \(d=[K:\QQ]\) 的代数整数环，则 \(\cdim(\mathcal O_K)=d\)；
  \item 对 \(A=\ZZ[1/n]\)，\(\cdim^\ast(A)=1\)，且
  \[
  0\to \ZZ_n\to \widehat A\to\TT\to 0;
  \]
  \item 对 \(\QQ\)，\(\cdim^\ast(\QQ)=1\)，且
  \[
  0\to \widehat\ZZ\to \widehat\QQ\to\TT\to 0;
  \]
  \item \(\pcdim(\ZZ_p)=1,\ \pcdim(\widehat\ZZ)=1\)。
\end{enumerate}
\leanverified{paper\_cdim\_phase\_register\_common\_assignments}
\end{proposition}
\begin{proof}
第 (1)--(4) 由定义 \ref{def:circle-dimension}、定义 \ref{def:half-circle-dimension} 与命题 \ref{prop:circle-dimension-laws} 直接得到。
第 (5)、(6) 是定理 \ref{thm:cdim-rankone-phase-prime-fiber} 在
\(\mathfrak n(\ZZ[1/n])=n^\infty\) 与 \(\mathfrak n(\QQ)=\prod_p p^\infty\) 的特例。
第 (7) 由定理 \ref{thm:cdim-register-circle-modp-formula}：
\[
\dim_{\FF_p}(\ZZ_p/p\ZZ_p)=1,\qquad
\widehat\ZZ/p\widehat\ZZ\cong \FF_p.
\]
故二者的 \(\pcdim\) 均为 \(1\)。证毕。
\end{proof}

\begin{definition}[高秩超自然局部化模]\label{def:cdim-highrank-supernatural-localization}
对超自然数 \(\mathfrak N=\prod_p p^{e_p}\)（\(e_p\in\NN\cup\{0,\infty\}\)），定义
\[
A_{\mathfrak N}
:=
\left\{\frac{a}{b}\in\QQ:\ v_p(b)\le e_p\ \forall p\right\},
\qquad
A_{r,\mathfrak N}:=A_{\mathfrak N}^{\,r}\ \ (r\ge 1).
\]
\end{definition}

\begin{theorem}[高秩局部化的环面相位与寄存轴最小秩]\label{thm:cdim-highrank-phase-register-minimality}
设 \(X_{r,\mathfrak N}:=\widehat{A_{r,\mathfrak N}}\)，并令 \(\pi_{r,\mathfrak N}:X_{r,\mathfrak N}\twoheadrightarrow\TT^r\) 为包含 \(\ZZ^r\hookrightarrow A_{r,\mathfrak N}\) 的对偶映射。
则：
\begin{enumerate}
  \item 存在短正合列
  \[
  0\to \ZZ_{\mathfrak N}^{\,r}\to X_{r,\mathfrak N}\xrightarrow{\ \pi_{r,\mathfrak N}\ }\TT^r\to 0;
  \]
  \item \(\cdim^\ast(A_{r,\mathfrak N})=r\)；
  \item 若 \(\mathfrak N\neq 1\)，则 \(\pcdim(\ZZ_{\mathfrak N}^{\,r})=r\)；
  \item 若某个素数 \(p\) 满足 \(e_p>0\)，则任何满足 \(L\twoheadrightarrow \ZZ_p^{\,r}\) 的拓扑有限生成 profinite 交换群 \(L\) 都有 \(\pcdim(L)\ge r\)。
\end{enumerate}
因而在“保持至少一个 \(p\)-adic 全秩读出”的寄存语义下，\(r\) 是不可压缩的独立寄存轴下界；规范核 \(\ZZ_{\mathfrak N}^{\,r}\) 达到该下界。
\end{theorem}
\begin{proof}
第 (1) 由定理 \ref{thm:cdim-rankone-phase-prime-fiber} 按坐标直积得到。
第 (2) 由 \(A_{r,\mathfrak N}\) 的 \(\QQ\)-秩为 \(r\) 与 \(\cdim^\ast\) 定义得到。
第 (3) 由定理 \ref{thm:cdim-register-circle-modp-formula}：
当 \(e_p>0\) 时 \(\ZZ_{\mathfrak N}/p\ZZ_{\mathfrak N}\cong\FF_p\)，故
\[
\dim_{\FF_p}\!\bigl(\ZZ_{\mathfrak N}^{\,r}/p\ZZ_{\mathfrak N}^{\,r}\bigr)=r,
\]
进而 \(\pcdim(\ZZ_{\mathfrak N}^{\,r})=r\)。
第 (4) 同理由模 \(p\) 商：
\[
r=\dim_{\FF_p}(\ZZ_p^{\,r}/p\ZZ_p^{\,r})
\le
\dim_{\FF_p}(L/pL)
\le
\pcdim(L).
\]
故 \(\pcdim(L)\ge r\)。证毕。
\end{proof}

\begin{conjecture}[有限状态折叠的一轴素数寄存普适性]\label{conj:cdim-one-axis-prime-register-universality}
设 \(f:\Omega\to X\) 为有限状态系统上的有限对一因子映射，并要求外置寄存器满足有限深度可读出、逆极限可复核的审计纪律。
猜想存在连续映射
\[
r:\Omega\to\widehat{\ZZ}
\]
使
\[
\Omega\to X\times\widehat{\ZZ},
\qquad
\omega\mapsto (f(\omega),r(\omega))
\]
为单射。
并且对任意满足该性质的实现，闭像 \(\overline{r(\Omega)}\) 满足
\[
\pcdim\!\bigl(\overline{r(\Omega)}\bigr)=1.
\]
该命题若成立，将给出“有限状态不可逆性可由单轴素数逆极限统一吸收”的结构化刻画。
\end{conjecture}
