\subsection{对称近似函数方程的后验实部约束：截断比率与收缩吸引子}\label{subsec:cdim-afe-posterior-sigma-contraction}
Riemann--Siegel--Gabcke 层级将临界线截面上的零点认证编译为可核验的区间证书（\S\ref{subsec:cdim-rs-gabcke-zero-recursion}）。
若希望在离临界线方向对零点实部给出可核验约束，则可从对称近似函数方程出发，在零点处建立“截断比率 \(\leftrightarrow\) 实部偏移”的后验不等式，并进一步把临界线的稳定性转写为一个收缩映射判据。

\paragraph{对称近似函数方程与余项证书}
令 \(s=\sigma+\mathrm{i}t\) 且 \(t\ge 2\)，并取 Riemann--Siegel 尺度
\[
N:=\Bigl\lfloor \sqrt{\frac{t}{2\pi}}\Bigr\rfloor,
\qquad
A_N(s):=\sum_{n\le N}n^{-s},
\qquad
B_N(s):=\sum_{n\le N}n^{-(1-s)}.
\]
定义
\begin{equation}\label{eq:cdim-afe-chi-def}
\chi(s):=\pi^{s-\frac12}\frac{\Gamma\!\left(\frac{1-s}{2}\right)}{\Gamma\!\left(\frac{s}{2}\right)}.
\end{equation}
对称近似函数方程在该尺度上给出分解
\begin{equation}\label{eq:cdim-afe-symmetric}
\zeta(s)=A_N(s)+\chi(s)B_N(s)+R_N(s),
\end{equation}
其中 \(R_N(s)\) 为余项；在高精度求值与余项控制文献中，\eqref{eq:cdim-afe-symmetric} 及其可计算余项界属于标准工具（例如 \cite{Titchmarsh1986RiemannZeta,Hiary2011FastZeta}）。

为把 \(\chi(s)\) 的模长与实部偏移关联，记
\begin{equation}\label{eq:cdim-afe-logchi-stirling}
\Delta_\chi(t,\sigma):=\log\bigl|\chi(\sigma+\mathrm{i}t)\bigr|-\Bigl(\tfrac12-\sigma\Bigr)\log\!\Bigl(\frac{t}{2\pi}\Bigr),
\end{equation}
则 Stirling 渐近给出 \(\Delta_\chi(t,\sigma)=O(1/t)\)，并可由显式余项形式提供有效上界（在 \(\sigma\) 落于紧集时一致）。

\begin{theorem}[对称近似函数方程诱导的后验实部约束]\label{thm:cdim-afe-posterior-sigma-constraint}
设 \(s=\sigma+\mathrm{i}t\) 为 \(\zeta\) 的非平凡零点，且 \(t\ge 2\)。
取 \eqref{eq:cdim-afe-symmetric} 的尺度 \(N\) 并沿用 \eqref{eq:cdim-afe-chi-def} 的 \(\chi(s)\) 与 \eqref{eq:cdim-afe-logchi-stirling} 的 \(\Delta_\chi(t,\sigma)\)。
假设存在可核验的余项上界
\[
|R_N(s)|\le \mathcal{E}(s),
\]
并且满足 \(\abs{A_N(s)}>\mathcal{E}(s)\) 与 \(\abs{B_N(s)}\neq 0\)。
令
\[
\eta(s):=\frac{\mathcal{E}(s)}{\abs{A_N(s)}}\in(0,1).
\]
则成立后验不等式
\begin{equation}\label{eq:cdim-afe-posterior-sigma-constraint}
\Biggl|
\Bigl(\sigma-\tfrac12\Bigr)
+
\frac{\log\!\Bigl(\abs{A_N(s)}/\abs{B_N(s)}\Bigr)}{\log\!\bigl(\tfrac{t}{2\pi}\bigr)}
\Biggr|
\le
\frac{1}{\log\!\bigl(\tfrac{t}{2\pi}\bigr)}
\log\!\Bigl(\frac{1+\eta(s)}{1-\eta(s)}\Bigr)
+
\frac{\abs{\Delta_\chi(t,\sigma)}}{\log\!\bigl(\tfrac{t}{2\pi}\bigr)}.
\end{equation}
\leanverified{paper\_cdim\_afe\_posterior\_sigma\_constraint}
\end{theorem}
\begin{proof}
由 \(\zeta(s)=0\) 与 \eqref{eq:cdim-afe-symmetric} 得
\[
A_N(s)+\chi(s)B_N(s)=-R_N(s).
\]
取模并用三角不等式得到
\[
\bigl||A_N(s)|-|\chi(s)|\,|B_N(s)|\bigr|
\le |A_N(s)+\chi(s)B_N(s)|
=|R_N(s)|
\le \mathcal{E}(s).
\]
由 \(\abs{A_N(s)}>\mathcal{E}(s)\) 推出
\[
\frac{\abs{A_N(s)}-\mathcal{E}(s)}{\abs{B_N(s)}}\le \abs{\chi(s)}\le \frac{\abs{A_N(s)}+\mathcal{E}(s)}{\abs{B_N(s)}}.
\]
对上式取对数，并以 \(\eta(s)=\mathcal{E}(s)/\abs{A_N(s)}\) 重写可得
\[
\Bigl|\log|\chi(s)|-\log\!\Bigl(\abs{A_N(s)}/\abs{B_N(s)}\Bigr)\Bigr|
\le
\log\!\Bigl(\frac{1+\eta(s)}{1-\eta(s)}\Bigr).
\]
由 \eqref{eq:cdim-afe-logchi-stirling}，
\(
\log|\chi(s)|=(\tfrac12-\sigma)\log(\tfrac{t}{2\pi})+\Delta_\chi(t,\sigma)
\)。
代回并整理后除以 \(\log(\tfrac{t}{2\pi})\) 得 \eqref{eq:cdim-afe-posterior-sigma-constraint}。证毕。
\end{proof}

\paragraph{余项控制下的单调隔离模板}
零点的实部约束需与高度方向的区间认证协同，方能被编译为离线可核验对象。
下述命题把“余项控制 \(\Rightarrow\) 单调隔离 \(\Rightarrow\) 唯一零点与误差包络”写成可直接调用的判据；在 \S\ref{subsec:cdim-rs-gabcke-zero-recursion} 的设定下，取 \(Z=Z_\zeta\)、\(S=Z_{\zeta,K}\) 与余项界 \eqref{eq:cdim-rs-gabcke-error0}--\eqref{eq:cdim-rs-gabcke-error1} 即可落入该模板。

\begin{theorem}[带余项控制的单调隔离与零点局部唯一性]\label{thm:cdim-monotone-isolation-unique-zero}
\leanpartial{paper\_cdim\_monotone\_isolation\_unique\_zero}{current Lean signature omits the continuity/differentiability hypotheses used by the proof text for strict monotonicity, IVT, and MVT}
设 \(Z:I\to\RR\) 为可微函数，\(I=[t_0-r,t_0+r]\)。
假设存在分解
\[
Z(t)=S(t)+\mathcal{R}(t),
\]
并存在可核验函数 \(\mathcal{E}_0,\mathcal{E}_1\ge 0\) 使得对一切 \(t\in I\) 成立
\[
\abs{\mathcal{R}(t)}\le \mathcal{E}_0(t),
\qquad
\abs{\mathcal{R}'(t)}\le \mathcal{E}_1(t).
\]
若满足
\[
m:=\inf_{t\in I}\Bigl(\abs{S'(t)}-\mathcal{E}_1(t)\Bigr)\ >\ 0,
\qquad
\abs{S(t_0)}+\mathcal{E}_0(t_0)\ <\ mr,
\]
则：
\begin{enumerate}
  \item \(Z\) 在 \(I\) 上严格单调，因而在 \(I\) 内至多存在一个零点；
  \item \(Z\) 在 \(I\) 内存在唯一零点 \(t_\ast\)；
  \item 该零点满足显式误差包络
  \begin{equation}\label{eq:cdim-monotone-isolation-envelope}
  \abs{t_\ast-t_0}\ \le\ \frac{\abs{S(t_0)}+\mathcal{E}_0(t_0)}{m}.
  \end{equation}
\end{enumerate}
\end{theorem}
\begin{proof}
由分解 \(Z'(t)=S'(t)+\mathcal{R}'(t)\) 与三角不等式，得对一切 \(t\in I\),
\[
\abs{Z'(t)}\ge \abs{S'(t)}-\abs{\mathcal{R}'(t)}\ge \abs{S'(t)}-\mathcal{E}_1(t)\ge m>0.
\]
因此 \(Z'\) 在 \(I\) 上处处不为零且连续，从而 \(Z'\) 符号恒定，\(Z\) 在 \(I\) 上严格单调，零点至多一个。
\par
另一方面，由 \(\abs{Z(t_0)}\le \abs{S(t_0)}+\mathcal{E}_0(t_0)<mr\) 且 \(Z'\) 在 \(I\) 上符号恒定并满足 \(\abs{Z'}\ge m\)，可得
\(
Z(t_0+r)
\)
与
\(
Z(t_0-r)
\)
必异号，从而由介值定理 \(Z\) 在 \(I\) 内存在零点；结合“至多一个零点”，零点唯一，记为 \(t_\ast\)。
\par
最后，对 \(Z\) 在 \(t_0\) 与 \(t_\ast\) 之间应用中值定理，存在 \(\xi\) 介于二者之间，使
\(
0=Z(t_\ast)=Z(t_0)+Z'(\xi)(t_\ast-t_0)
\)。
于是
\[
\abs{t_\ast-t_0}=\frac{\abs{Z(t_0)}}{\abs{Z'(\xi)}}
\le \frac{\abs{S(t_0)}+\mathcal{E}_0(t_0)}{m},
\]
即得 \eqref{eq:cdim-monotone-isolation-envelope}。证毕。
\end{proof}

\paragraph{临界线作为吸引子：收缩准则的可核验形式}
固定 \(t\ge 2\) 并取 \(N=\lfloor\sqrt{t/(2\pi)}\rfloor\)。
对 \(\sigma\in(0,1)\) 定义
\[
A_N(\sigma):=\sum_{n\le N}n^{-(\sigma+\mathrm{i}t)},
\qquad
B_N(\sigma):=\sum_{n\le N}n^{-(1-\sigma-\mathrm{i}t)},
\]
并在 \(\{A_N(\sigma)\neq 0,\ B_N(\sigma)\neq 0\}\) 上定义映射
\begin{equation}\label{eq:cdim-afe-Phi-def}
\Phi_t(\sigma):=\frac12-\frac{\log\!\bigl(\abs{A_N(\sigma)}/\abs{B_N(\sigma)}\bigr)}{\log\!\bigl(\tfrac{t}{2\pi}\bigr)}.
\end{equation}
当 \(\sigma=\tfrac12\) 时有 \(B_N(\tfrac12)=\overline{A_N(\tfrac12)}\)，故 \(\abs{A_N(\tfrac12)}=\abs{B_N(\tfrac12)}\) 且
\begin{equation}\label{eq:cdim-afe-Phi-fixedpoint}
\Phi_t\!\Bigl(\tfrac12\Bigr)=\tfrac12.
\end{equation}

\begin{theorem}[临界线吸引子的可检验收缩准则]\label{thm:cdim-afe-contraction-attractor}
设 \(s=\sigma+\mathrm{i}t\) 为 \(\zeta\) 的非平凡零点，并假设定理 \ref{thm:cdim-afe-posterior-sigma-constraint} 的余项证书条件成立。
令 \(\eta(s)=\mathcal{E}(s)/\abs{A_N(s)}\in(0,1)\)，并定义近似不动点误差
\begin{equation}\label{eq:cdim-afe-epsilon-def}
\varepsilon(t,\sigma):=
\frac{1}{\log\!\bigl(\tfrac{t}{2\pi}\bigr)}
\log\!\Bigl(\frac{1+\eta(s)}{1-\eta(s)}\Bigr)
+
\frac{\abs{\Delta_\chi(t,\sigma)}}{\log\!\bigl(\tfrac{t}{2\pi}\bigr)}.
\end{equation}
若存在区间 \(J\subset(0,1)\) 使得 \(\sigma\in J\)、\(\Phi_t\) 在 \(J\) 上可微，且满足收缩增益界
\[
\sup_{\xi\in J}\abs{\Phi_t'(\xi)}\le q<1,
\]
则成立吸引子型误差压制不等式
\begin{equation}\label{eq:cdim-afe-attractor-bound}
\abs{\sigma-\tfrac12}\ \le\ \frac{\varepsilon(t,\sigma)}{1-q}.
\end{equation}
\leanverified{paper\_cdim\_afe\_contraction\_attractor}
\end{theorem}
\begin{proof}
由定理 \ref{thm:cdim-afe-posterior-sigma-constraint} 与 \eqref{eq:cdim-afe-Phi-def} 得
\(
\abs{\sigma-\Phi_t(\sigma)}\le \varepsilon(t,\sigma)
\)。
又由 \eqref{eq:cdim-afe-Phi-fixedpoint} 与收缩性，
\[
\abs{\sigma-\tfrac12}
=\abs{\sigma-\Phi_t(\tfrac12)}
\le \abs{\sigma-\Phi_t(\sigma)}+\abs{\Phi_t(\sigma)-\Phi_t(\tfrac12)}
\le \varepsilon(t,\sigma)+q\,\abs{\sigma-\tfrac12}.
\]
移项即得 \eqref{eq:cdim-afe-attractor-bound}。证毕。
\end{proof}

\begin{conjecture}[统一收缩增益与黎曼猜想的动力学等价]\label{conj:cdim-afe-uniform-contraction-rh}
存在常数 \(t_0\ge 2\) 与 \(\kappa\in(0,1)\) 使得对所有 \(t\ge t_0\)，映射 \(\Phi_t\) 在某个包含 \(\sigma=\tfrac12\) 的邻域 \(J_t\subset(0,1)\) 上满足统一收缩
\[
\sup_{\xi\in J_t}\abs{\Phi_t'(\xi)}\le 1-\kappa,
\]
并且对任意非平凡零点 \(s=\sigma+\mathrm{i}t\)（\(t\to\infty\)），定理 \ref{thm:cdim-afe-posterior-sigma-constraint} 的余项比率 \(\eta(s)=\mathcal{E}(s)/\abs{A_N(s)}\) 满足 \(\eta(s)\to 0\)（或等价地，\(\varepsilon(t,\sigma)\to 0\)）。
\par
在该猜想下，定理 \ref{thm:cdim-afe-contraction-attractor} 推出充分高的全部零点满足 \(\sigma=\tfrac12\)；结合有限高度的离线核验，可得到一条以“统一收缩增益 + 相对余项衰减”为核心的 RH 证书路径。
\end{conjecture}

本小节给出一条与 \S\ref{subsec:cdim-symmetric-truncation-scale-flow} 互补的动力学接口：\eqref{eq:cdim-afe-posterior-sigma-constraint} 将零点实部偏移转写为“截断比率 + 余项证书 + \(\chi\)-项显式余差”的后验不等式，而 \eqref{eq:cdim-afe-attractor-bound} 将临界线稳定性转写为可核验的收缩增益判据。
若要把猜想 \ref{conj:cdim-afe-uniform-contraction-rh} 推进为可证定理，则关键困难集中为在可控邻域上建立 \(\Phi_t\) 的一致导数界，并把余项证书的相对衰减转化为可离线核验的显式不等式链。
