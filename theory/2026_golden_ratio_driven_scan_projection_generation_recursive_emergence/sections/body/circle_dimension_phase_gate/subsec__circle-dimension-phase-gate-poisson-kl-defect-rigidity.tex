\subsection{Poisson 相对熵的偶阶渐近展开与缺陷刚性}\label{subsec:cdim-poisson-kl-defect-rigidity}

沿用 \S\ref{subsec:cdim-poisson-cauchy} 的记号：\(\nu\) 为 \(\RR\) 上概率测度，均值 \(\bar\gamma\)，方差 \(\sigma^2\)，中心矩 \(\mu_n:=\int_{\RR}(\gamma-\bar\gamma)^n\dd\nu(\gamma)\)；\(h_t:=P_t*\nu\)，\(g_t(x):=P_t(x-\bar\gamma)\)。

\begin{theorem}[Poisson 相对熵仅含偶阶项]\label{thm:poisson-kl-even-orders}
\leanverified{paper\_poisson\_kl\_even\_orders}
设 \(N\ge 2\)，\(\nu\) 具有有限 \((2N{+}1)\) 阶绝对中心矩。则存在实系数 \(c_4,c_6,\dots,c_{2N}\) 使
\[
D_{\mathrm{KL}}(h_t\|g_t)
=\sum_{m=2}^{N}c_{2m}\,t^{-2m}+o(t^{-2N})
\qquad(t\to\infty).
\]
特别地，一切奇次幂 \(t^{-(2m+1)}\) 的系数为零。
\end{theorem}

\begin{proof}
以 \(x=\bar\gamma+ty\) 进行变量替换，并对 \(h_t/g_t-1\) 作到 \(2N\) 阶的中心矩 Taylor 展开；等价地，这正给出 \S\ref{subsec:cdim-poisson-cauchy} 中的 Cayley--Chebyshev 模态展开，可写成
\[
\delta_t(\bar\gamma+ty)
:=\frac{h_t(\bar\gamma+ty)}{g_t(\bar\gamma+ty)}-1
=\sum_{n=2}^{2N}\frac{\mu_n}{t^n}u_n(y)+O(t^{-(2N+1)}).
\]
一致余项允许在 \(\Phi(z)=(1+z)\log(1+z)-z\) 的幂级数中逐项积分：
\[
D_{\mathrm{KL}}(h_t\|g_t)
=\int_{\RR}\frac{\Phi(\delta_t(\bar\gamma+ty))}{\pi(1+y^2)}\dd y.
\]
权重 \((\pi(1+y^2))^{-1}\) 为偶函数，而 Cayley--Chebyshev 模态满足奇偶性 \(u_n(-y)=(-1)^n u_n(y)\)，故 \(\sum n_i\) 为奇时积分为零。因此展开中仅存偶次幂，即 \(c_{2m+1}=0\)。
\end{proof}

\begin{theorem}[八阶 Poisson 相对熵展开]\label{thm:poisson-kl-eighth}
\leanverified{paper\_poisson\_kl\_eighth}
设 \(\nu\) 具有有限 \(7\) 阶绝对中心矩且 \(\sigma^2>0\)。当 \(t\to\infty\) 时，
\begin{equation}\label{eq:poisson-kl-eighth}
\begin{split}
D_{\mathrm{KL}}(h_t\|g_t)
&=\frac{\sigma^4}{8t^4}
+\frac{\sigma^6+6\mu_3^2-8\sigma^2\mu_4}{64\,t^6}
\\
&\quad
+\frac{3\sigma^8-12\sigma^4\mu_4+9\sigma^2\mu_3^2}{256\,t^8}
+\frac{12\sigma^2\mu_6-30\mu_3\mu_5+20\mu_4^2}{256\,t^8}
+o(t^{-8}).
\end{split}
\end{equation}
\end{theorem}

\begin{proof}
以 \(x=\bar\gamma+ty\) 并用 \(N=6\) 阶的 Cayley 模态展开
\[
\delta_t(\bar\gamma+ty)
=\frac{\sigma^2}{t^2}u_2(y)+\frac{\mu_3}{t^3}u_3(y)+\cdots+\frac{\mu_6}{t^6}u_6(y)+O(t^{-7}).
\]
由于 \(\delta_t=O(t^{-2})\)，取 \(\Phi\) 的四阶截断
\[
\Phi(\delta_t)=\tfrac12\delta_t^2-\tfrac16\delta_t^3+\tfrac{1}{12}\delta_t^4+O(t^{-10}).
\]
逐项展开并对 Cauchy 权重 \((\pi(1+y^2))^{-1}\) 积分；各乘积仍是有限 Cayley 模态，其积分可由命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 证明中列出的精确加权积分恒等式等价地逐项读出零阶系数，遂得到式 \eqref{eq:poisson-kl-eighth} 的各系数。
\end{proof}

\begin{theorem}[双层缺陷阶梯]\label{thm:poisson-defect-ladder}
\leanverified{paper\_poisson\_defect\_ladder}
在定理 \ref{thm:poisson-kl-eighth} 的假设下，令
\[
C_6:=\frac{\sigma^6+6\mu_3^2-8\sigma^2\mu_4}{64},
\quad
C_8:=\frac{3\sigma^8-12\sigma^4\mu_4+9\sigma^2\mu_3^2+12\sigma^2\mu_6-30\mu_3\mu_5+20\mu_4^2}{256}.
\]
记归一化随机变量 \(X:=(\gamma-\bar\gamma)/\sigma\)，歪度 \(\beta_3:=\EE[X^3]\)，正交残差 \(Q_2:=X^2-1-\beta_3 X\) 及其范数 \(\kappa:=\|Q_2\|_{L^2(\nu)}^2\)。则
\[
\frac{64}{\sigma^6}C_6=-7-2\beta_3^2-8\kappa,
\]
且八阶系数满足
\[
\frac{256}{\sigma^8}C_8
=23+12\|Q_3\|_{L^2(\nu)}^2+32\kappa^2+\tfrac{61}{4}\beta_3^2\kappa+52\kappa+2\beta_3^4+13\beta_3^2,
\]
其中 \(Q_3:=XQ_2-\kappa X-\frac{\beta_3}{4}Q_2\)。特别地，
\[
C_8\ge\frac{23}{256}\sigma^8,
\]
等号成立当且仅当 \(\nu=\frac12\delta_{\bar\gamma+\sigma}+\frac12\delta_{\bar\gamma-\sigma}\)（对称二点法则）。
\end{theorem}

\begin{proof}
利用 \(X^2=1+\beta_3 X+Q_2\) 逐级代入，将高阶矩 \(\beta_4,\beta_5,\beta_6\) 展开为 \(\kappa,L:=\EE[XQ_2^2],M:=\EE[Q_2^3]\) 的函数并验证各系数。等号条件由右端各非负项同时为零给出 \(\beta_3=0\)、\(\kappa=0\)，从而 \(X^2=1\) a.s.，结合 \(\EE[X]=0\)、\(\EE[X^2]=1\) 得到对称二点分布。
\end{proof}

\begin{theorem}[缺陷放大]\label{thm:poisson-defect-amplification}
\leanverified{paper\_poisson\_defect\_amplification}
在定理 \ref{thm:poisson-kl-eighth} 的假设下，定义六阶与八阶缺陷
\[
\Delta_6(\nu):=-\frac{7}{64}\sigma^6-C_6=\frac{\sigma^6}{32}\beta_3^2+\frac{\sigma^6}{8}\kappa,
\]
\[
\Delta_8(\nu):=C_8-\frac{23}{256}\sigma^8.
\]
则
\begin{equation}\label{eq:defect-amplification}
\Delta_8(\nu)
=\frac{13\sigma^2}{8}\Delta_6(\nu)
+\frac{\sigma^8}{256}\Bigl(
12\|Q_3\|_{L^2(\nu)}^2
+32\kappa^2
+\tfrac{61}{4}\beta_3^2\kappa
+2\beta_3^4
\Bigr).
\end{equation}
特别地，\(\Delta_8(\nu)\ge\frac{13\sigma^2}{8}\Delta_6(\nu)\)，等号成立当且仅当 \(\nu\) 为对称二点法则。
\end{theorem}

\begin{proof}
将定理 \ref{thm:poisson-defect-ladder} 的 \(C_6,C_8\) 表达式代入 \(\Delta_6,\Delta_8\) 的定义，直接验证 \eqref{eq:defect-amplification}。下界由余项各非负项给出；等号条件的分析与定理 \ref{thm:poisson-defect-ladder} 相同。
\end{proof}

\begin{theorem}[对称二点法则的定量刚性]\label{thm:poisson-defect-stability}
在定理 \ref{thm:poisson-kl-eighth} 的假设下，令 \(\nu_c\) 为 \(\Delta=\gamma-\bar\gamma\) 的中心化法则，令 \(\rho_\sigma:=\frac12(\delta_{-\sigma}+\delta_\sigma)\)。则
\[
W_2(\nu_c,\rho_\sigma)
\le
4\,\sigma^{-1/2}\Delta_6(\nu)^{1/4}
+8\,\sigma^{-2}\Delta_6(\nu)^{1/2}.
\]
特别地，当 \(\sigma\) 固定时，\(W_2(\nu_c,\rho_\sigma)=O(\Delta_6(\nu)^{1/4})\) 当 \(\Delta_6(\nu)\downarrow 0\)。
\end{theorem}
\leanverified{paper\_poisson\_defect\_stability}

\begin{proof}
取符号配对 \(Y:=\mathbf{1}_{\{X\ge 0\}}-\mathbf{1}_{\{X<0\}}\in\{-1,1\}\)，由 \(|x-\operatorname{sgn}(x)|\le|x^2-1|\) 得
\(\EE[(X-Y)^2]\le \EE[(X^2-1)^2]\le 2\beta_3^2+2\kappa\)。
利用 \(\Delta_6(\nu)\) 的公式将 \(\beta_3^2,\kappa\) 翻译为 \(\Delta_6(\nu)\) 的表达式，并用三角不等式在 Wasserstein 距离下完成连接：
\[
W_2(\nu_c,\rho_\sigma)^2
\le \sigma^2\EE[(X-Y)^2]+W_2(Y\text{-law},\rho_\sigma/\sigma)^2\cdot\sigma^2.
\]
后项由 \(\EE[(X-Y)^2]\) 的尾部积分控制。取平方根并用 \((a+b)^{1/2}\le a^{1/2}+b^{1/2}\) 得到所述界。
\end{proof}
