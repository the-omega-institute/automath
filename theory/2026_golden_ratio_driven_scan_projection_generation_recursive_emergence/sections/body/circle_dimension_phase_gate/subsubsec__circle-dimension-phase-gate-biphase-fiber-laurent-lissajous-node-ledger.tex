\subsubsection{双相位平均门的纤维结构、Laurent 分支圆与非共振 Lissajous 节点账本}\label{subsubsec:cdim-biphase-fiber-laurent-lissajous-node-ledger}
在定理 \ref{thm:cdim-biphase-average-disk-critical-ring} 把双相位平均门的圆盘像与临界环完全外显，并由定理 \ref{thm:cdim-rhodonea-as-biphase-average-subflow} 与定理 \ref{thm:cdim-lissajous-chebyshev-elimination-square-critical-ring} 给出玫瑰轨道及 Lissajous 轨道的门投影之后，还可进一步把纤维计数、Laurent 双频干涉的分歧值圆以及非共振 Lissajous 的节点账本压缩为下列连续结论。

\begin{theorem}[双相位平均门的有序纤维计数与反对角退化]\label{thm:cdim-biphase-average-fiber-diagonal-antidiagonal}
沿用定理 \ref{thm:cdim-biphase-average-disk-critical-ring} 的双相位平均门
\[
\mathcal F:\TT^2\to\CC,
\qquad
\mathcal F(u,v):=\frac{e^{\mathrm{i}u}+e^{\mathrm{i}v}}{2}.
\]
并定义对角与反对角子环
\[
\Delta_{\mathcal F}:=
\{(u,v)\in\TT^2:\ u\equiv v\ (\mathrm{mod}\ 2\pi)\},
\]
\[
A_{\mathcal F}:=
\{(u,v)\in\TT^2:\ u\equiv v+\pi\ (\mathrm{mod}\ 2\pi)\}.
\]
则成立：
\begin{enumerate}
  \item \(\mathcal F(\TT^2)=\overline{\mathbb D}\)；
  \item \(\operatorname{Crit}(\mathcal F)=\Delta_{\mathcal F}\cup A_{\mathcal F}\)，并且
  \[
  \mathcal F(\Delta_{\mathcal F})=\partial\mathbb D,
  \qquad
  \mathcal F(A_{\mathcal F})=\{0\};
  \]
  \item 若 \(z\in\mathbb D\setminus\{0\}\)，则 \(\mathcal F^{-1}(z)\) 恰含两个有序原像；
  \item 若 \(z\in\partial\mathbb D\)，则 \(\mathcal F^{-1}(z)\) 为 \(\Delta_{\mathcal F}\) 上的单点；
  \item 若 \(z=0\)，则 \(\mathcal F^{-1}(0)=A_{\mathcal F}\) 为整条圆环纤维。
\end{enumerate}
\leanverified{paper\_cdim\_biphase\_fiber\_count\_seeds}
\end{theorem}
\begin{proof}
前两条正是定理 \ref{thm:cdim-biphase-average-disk-critical-ring} 的内容。只需补证纤维计数。

在 \(\RR^2\) 的局部角坐标中置
\[
w:=\frac{u+v}{2},
\qquad
\delta:=\frac{u-v}{2}.
\]
则
\[
\mathcal F(u,v)=e^{\mathrm{i}w}\cos\delta.
\]
设 \(z=re^{\mathrm{i}\theta}\in\overline{\mathbb D}\)。

若 \(0<r<1\)，记 \(\alpha:=\arccos r\in(0,\pi/2)\)。方程
\[
e^{\mathrm{i}w}\cos\delta=re^{\mathrm{i}\theta}
\]
强制 \(|\cos\delta|=r\)，故
\[
\delta\equiv \pm \alpha
\quad\text{或}\quad
\delta\equiv \pi\pm\alpha
\pmod{2\pi}.
\]
当 \(\delta\equiv \pi\pm\alpha\) 时，\(\cos\delta=-r\)，只能通过把 \(w\) 同时平移 \(\pi\) 来吸收负号；所得 \((u,v)\) 与 \(\delta\equiv\mp\alpha\)、\(w\equiv\theta\) 给出的两点在 \(\TT^2\) 上完全相同。因此仅有两组不同解：
\[
(u,v)\equiv (\theta+\alpha,\theta-\alpha),
\qquad
(u,v)\equiv (\theta-\alpha,\theta+\alpha)
\pmod{2\pi}.
\]
故 \(\mathcal F^{-1}(z)\) 恰有两个有序原像。

若 \(r=1\)，则 \(|\cos\delta|=1\)，故 \(\delta\in\pi\ZZ\)。与上同理，\(\delta\equiv \pi\) 只对应 \(\delta\equiv 0\) 的同一环面点，最终得到唯一原像
\[
(u,v)\equiv (\theta,\theta),
\]
它落在 \(\Delta_{\mathcal F}\) 上。

若 \(z=0\)，则 \(\cos\delta=0\)，即
\[
\delta\equiv \frac{\pi}{2}\pmod{\pi},
\]
等价于
\[
u-v\equiv \pi\pmod{2\pi}.
\]
故 \(\mathcal F^{-1}(0)=A_{\mathcal F}\)。
证毕。
\end{proof}
\leanverified{paper\_cdim\_biphase\_average\_fiber\_diagonal\_antidiagonal}

\leanverified{paper\_cdim\_rhodonea\_diagonal\_antidiagonal\_slice}
\begin{corollary}[有理玫瑰轨道作为双相位平均门的对角与反对角切片]\label{cor:cdim-rhodonea-diagonal-antidiagonal-slice}
对整数 \(d\ge 1\)、\(n\ge 0\)，定义复轨道
\[
\rho_{d,n}(t):=e^{\mathrm{i}dt}\cos(nt)
\qquad (t\in\RR).
\]
再定义一参数子流
\[
\widetilde\gamma_{d,n}(t):=((d+n)t,\,(d-n)t)\in\TT^2.
\]
则有严格恒等式
\[
\rho_{d,n}(t)=\mathcal F\bigl(\widetilde\gamma_{d,n}(t)\bigr).
\]
特别地，在代换 \(\theta=dt\) 后，
\[
\rho_{d,n}(t)=r(\theta)e^{\mathrm{i}\theta},
\qquad
r(\theta)=\cos\!\Bigl(\frac{n}{d}\theta\Bigr),
\]
故有理玫瑰极坐标轨道正是 \(\mathcal F\) 在有理子流上的切片。

若 \(n>0\)，则
\[
|\rho_{d,n}(t)|=1
\iff
\widetilde\gamma_{d,n}(t)\in\Delta_{\mathcal F}
\iff
t\in \frac{\pi}{n}\ZZ,
\]
并且
\[
\rho_{d,n}(t)=0
\iff
\widetilde\gamma_{d,n}(t)\in A_{\mathcal F}
\iff
t\in \frac{\pi}{2n}+\frac{\pi}{n}\ZZ.
\]
\leanverified{paper\_cdim\_rhodonea\_diagonal\_antidiagonal\_slice\_package}
\end{corollary}
\begin{proof}
直接计算得
\[
\mathcal F\bigl(\widetilde\gamma_{d,n}(t)\bigr)
=
\frac{e^{\mathrm{i}(d+n)t}+e^{\mathrm{i}(d-n)t}}{2}
=
e^{\mathrm{i}dt}\cos(nt)
=
\rho_{d,n}(t).
\]
若 \(n>0\)，则
\[
\widetilde\gamma_{d,n}(t)\in\Delta_{\mathcal F}
\iff
(d+n)t\equiv (d-n)t\pmod{2\pi}
\iff
2nt\in 2\pi\ZZ,
\]
即 \(t\in (\pi/n)\ZZ\)。这时 \(|\rho_{d,n}(t)|=|\cos(nt)|=1\)。

同理，
\[
\widetilde\gamma_{d,n}(t)\in A_{\mathcal F}
\iff
(d+n)t\equiv (d-n)t+\pi\pmod{2\pi}
\iff
2nt\in \pi+2\pi\ZZ,
\]
即
\[
t\in \frac{\pi}{2n}+\frac{\pi}{n}\ZZ,
\]
此时 \(\rho_{d,n}(t)=e^{\mathrm{i}dt}\cos(nt)=0\)。
证毕。
\end{proof}

\begin{theorem}[Laurent 双频干涉映射的临界环与分歧值圆]\label{thm:cdim-laurent-double-frequency-critical-value-circle}
对整数 \(p>q\ge 1\)，定义 Laurent 映射
\[
f_{p,q}:\CC^\times\to\CC,
\qquad
f_{p,q}(z):=\frac{z^p+z^q}{2}.
\]
则成立：
\begin{enumerate}
  \item 全部临界点满足
  \[
  f_{p,q}'(z)=0
  \iff
  z^{p-q}=-\frac{q}{p}.
  \]
  因而临界点恰位于圆
  \[
  \Sigma_{p,q}:=
  \left\{
  z\in\CC^\times:\ |z|=\left(\frac{q}{p}\right)^{1/(p-q)}
  \right\},
  \]
  且共有 \(p-q\) 个简单临界点；
  \item 对任一临界点 \(z_\ast\in\Sigma_{p,q}\)，有
  \[
  f_{p,q}(z_\ast)=\frac{p-q}{2p}\,z_\ast^q.
  \]
  从而全体临界值恰为同心圆
  \[
  f_{p,q}(\Sigma_{p,q})
  =
  \{w\in\CC:\ |w|=B_{p,q}\},
  \]
  其中
  \[
  B_{p,q}:=\frac{p-q}{2p}\left(\frac{q}{p}\right)^{q/(p-q)};
  \]
  \item 将 \(f_{p,q}\) 在 \(z=0\) 处作自然全纯延拓后，方程 \(w=f_{p,q}(z)\) 在 \(w=0\) 附近的局部反演，经选定 \(w^{1/q}\) 的单值支后可写成关于 \(w^{1/q}\) 的收敛幂级数；等价地，它作为 \(w\) 的 Puiseux 分支，其在 \(w\)-平面中的解析延拓半径不超过 \(B_{p,q}\)。对沿径向避开分歧值圆的主分支，该上界恰可达到。特别地，当 \(q=1\) 时，相应局部反函数关于 \(w\) 的通常 Taylor 展开收敛半径正好等于 \(B_{p,q}\)。
\end{enumerate}
\end{theorem}
\leanverified{paper\_cdim\_laurent\_double\_frequency\_critical\_value\_circle}
\begin{proof}
直接求导得
\[
f_{p,q}'(z)
=
\frac{1}{2}\bigl(pz^{p-1}+qz^{q-1}\bigr)
=
\frac{z^{q-1}}{2}\bigl(pz^{p-q}+q\bigr).
\]
故
\[
f_{p,q}'(z)=0
\iff
z^{p-q}=-\frac{q}{p}.
\]
取模即得全部临界点落在 \(\Sigma_{p,q}\) 上；角度由
\[
z^{p-q}=\frac{q}{p}e^{(2\ell+1)\pi\mathrm{i}}
\qquad
(\ell=0,\dots,p-q-1)
\]
给出，故共有 \(p-q\) 个点。再由
\[
f_{p,q}''(z)
=
\frac{1}{2}\bigl(p(p-1)z^{p-2}+q(q-1)z^{q-2}\bigr)
\]
与临界方程联立可知 \(f_{p,q}''(z_\ast)\neq 0\)，故这些临界点皆为简单临界点。

对临界点 \(z_\ast\)，由
\[
z_\ast^p=-\frac{q}{p}\,z_\ast^q
\]
得到
\[
f_{p,q}(z_\ast)
=
\frac{1}{2}\bigl(z_\ast^p+z_\ast^q\bigr)
=
\frac{1}{2}\left(1-\frac{q}{p}\right)z_\ast^q
=
\frac{p-q}{2p}\,z_\ast^q.
\]
于是
\[
|f_{p,q}(z_\ast)|
=
\frac{p-q}{2p}|z_\ast|^q
=
\frac{p-q}{2p}\left(\frac{q}{p}\right)^{q/(p-q)}
=
B_{p,q},
\]
从而全部临界值落在半径 \(B_{p,q}\) 的圆上；反之，随着 \(z_\ast\) 沿 \(\Sigma_{p,q}\) 取遍全部临界点，\(z_\ast^q\) 亦沿同心圆等角分布，故临界值集合正是该圆。

把 \(f_{p,q}\) 在 \(z=0\) 处按同一 Laurent 表达式自然延拓为全纯函数后，由
\[
2w=z^q(1+z^{p-q})
\]
可知 \((z,w)=(0,0)\) 附近由 Weierstrass 预备定理给出一个 \(q\)-值代数反演；选定 \(w^{1/q}\) 的单值支后，局部解可写成关于 \(w^{1/q}\) 的收敛幂级数。另一方面，代数函数的分支点恰为临界值，故其 \(w\)-平面解析延拓在首次碰到临界值之前不会失效，而最近临界值到原点的距离正是 \(B_{p,q}\)。因此延拓半径至多为 \(B_{p,q}\)；对沿径向避开分歧值圆的主分支，这一最近分支点控制机制给出精确半径 \(B_{p,q}\)。当 \(q=1\) 时，此即通常局部反函数的 Taylor 收敛半径。证毕。
\end{proof}

\leanverified{paper\_cdim\_nonresonant\_lissajous\_node\_count}
\begin{theorem}[非共振 Lissajous 轨道的节点计数与两类反射分解]\label{thm:cdim-nonresonant-lissajous-node-count}
设 \(a,b\in\NN\) 满足 \(\gcd(a,b)=1\)，并取相位 \(\delta\in\RR\) 使
\[
\frac{\delta}{\pi}\notin\QQ.
\]
考虑 Lissajous 轨道
\[
L_{a,b,\delta}(t):=(\cos(at+\delta),\cos(bt)),
\qquad
t\in\RR/2\pi\ZZ.
\]
则成立：
\begin{enumerate}
  \item 该轨道恰有
  \[
  N_{a,b}:=(a-1)b+a(b-1)=2ab-a-b
  \]
  个不同的自交点，且每个自交点都是普通横截节点；
  \item 全部节点分裂为两族互不相交的集合
  \[
  \mathcal N_{+,-},
  \qquad
  \mathcal N_{-,+},
  \]
  其中 \(|\mathcal N_{+,-}|=(a-1)b\)、\(|\mathcal N_{-,+}|=a(b-1)\)；
  \item 对任意
  \[
  m\in\{1,\dots,a-1\},
  \qquad
  n\in\{0,\dots,b-1\},
  \]
  定义
  \[
  t_{m,n}:=\pi\left(\frac{m}{a}+\frac{n}{b}\right),
  \qquad
  s_{m,n}:=\pi\left(-\frac{m}{a}+\frac{n}{b}\right).
  \]
  则
  \[
  t_{m,n}\not\equiv s_{m,n}\pmod{2\pi},
  \qquad
  L_{a,b,\delta}(t_{m,n})=L_{a,b,\delta}(s_{m,n}),
  \]
  且这些点恰给出 \(\mathcal N_{+,-}\) 的全部 \((a-1)b\) 个节点；
  \item 对任意
  \[
  m\in\{0,\dots,a-1\},
  \qquad
  n\in\{1,\dots,b-1\},
  \]
  定义
  \[
  t'_{m,n}:=\frac{\pi m-\delta}{a}+\frac{\pi n}{b},
  \qquad
  s'_{m,n}:=\frac{\pi m-\delta}{a}-\frac{\pi n}{b}.
  \]
  则
  \[
  t'_{m,n}\not\equiv s'_{m,n}\pmod{2\pi},
  \qquad
  L_{a,b,\delta}(t'_{m,n})=L_{a,b,\delta}(s'_{m,n}),
  \]
  且这些点恰给出 \(\mathcal N_{-,+}\) 的全部 \(a(b-1)\) 个节点。
\end{enumerate}
\leanverified{paper\_cdim\_nonresonant\_lissajous\_node\_count}
\end{theorem}
\begin{proof}
设 \(L_{a,b,\delta}(t)=L_{a,b,\delta}(s)\)。则存在符号
\[
\sigma_1,\sigma_2\in\{\pm1\}
\]
与整数 \(M,N\) 使
\[
at+\delta=\sigma_1(as+\delta)+2\pi M,
\qquad
bt=\sigma_2 bs+2\pi N.
\]
亦即
\[
a(t-\sigma_1 s)=(\sigma_1-1)\delta+2\pi M,
\qquad
b(t-\sigma_2 s)=2\pi N.
\]

若 \((\sigma_1,\sigma_2)=(+,+)\)，则
\[
a(t-s)=2\pi M,
\qquad
b(t-s)=2\pi N.
\]
由 \(\gcd(a,b)=1\) 可得 \(t-s\in 2\pi\ZZ\)，这只是平凡同参，不产生自交。

若 \((\sigma_1,\sigma_2)=(-,-)\)，则
\[
a(t+s)=2\pi M-2\delta,
\qquad
b(t+s)=2\pi N.
\]
消去 \(t+s\) 得
\[
\delta=\pi\left(M-\frac{a}{b}N\right),
\]
从而 \(\delta/\pi\in\QQ\)，与假设矛盾。因此此型无解。

于是全部非平凡自交只能来自 \((\sigma_1,\sigma_2)=(+,-)\) 或 \((-,+)\) 两种情形。

先看 \((+,-)\) 型。方程组化为
\[
a(t-s)=2\pi M,
\qquad
b(t+s)=2\pi N,
\]
故
\[
t=\pi\left(\frac{M}{a}+\frac{N}{b}\right),
\qquad
s=\pi\left(-\frac{M}{a}+\frac{N}{b}\right).
\]
若 \(M\equiv 0\pmod a\)，则 \(t\equiv s\pmod{2\pi}\)，不构成自交；故可取
\[
M\in\{1,\dots,a-1\}.
\]
又可按模 \(b\) 归一化取
\[
N\in\{0,\dots,b-1\}.
\]
由此得到恰好 \((a-1)b\) 组参数对。

记对应节点坐标为
\[
x_{M,N}=(-1)^M\cos\!\left(\delta+\pi a\frac{N}{b}\right),
\qquad
y_{M,N}=(-1)^N\cos\!\left(\pi b\frac{M}{a}\right).
\]
若
\[
(x_{M,N},y_{M,N})=(x_{M',N'},y_{M',N'}),
\]
则由
\[
\cos\!\left(\delta+\pi a\frac{N}{b}+M\pi\right)
=
\cos\!\left(\delta+\pi a\frac{N'}{b}+M'\pi\right)
\]
知，要么两角之差属于 \(2\pi\ZZ\)，要么两角之和属于 \(2\pi\ZZ\)。后一种情形将强制 \(2\delta/\pi\in\QQ\)，与假设矛盾。故只能有
\[
a\frac{N-N'}{b}+M-M'\in 2\ZZ.
\]
于是 \(a(N-N')/b\in\ZZ\)。由 \(\gcd(a,b)=1\) 且 \(0\le N,N'<b\)，必有 \(N=N'\)。再由
\[
\cos\!\left(\pi b\frac{M}{a}\right)=\cos\!\left(\pi b\frac{M'}{a}\right),
\qquad
1\le M,M'\le a-1,
\]
并且先前已知 \(M-M'\in 2\ZZ\)。余弦相等只可能由
\[
b\frac{M-M'}{a}\in 2\ZZ
\qquad\text{或}\qquad
b\frac{M+M'}{a}\in 2\ZZ
\]
触发。第一种情形由 \(\gcd(a,b)=1\) 与 \(|M-M'|<a\) 立即推出 \(M=M'\)。第二种情形则强制 \(a\mid(M+M')\)；由于
\[
2\le M+M'\le 2a-2,
\]
只能有 \(M+M'=a\)。这时 \(b(M+M')/a=b\) 为偶数，从而 \(b\) 必为偶数；由 \(\gcd(a,b)=1\) 可知 \(a\) 为奇数，于是 \(M\) 与 \(M'=a-M\) 奇偶性相反，这与 \(M-M'\in 2\ZZ\) 矛盾。故第二种情形不可能。综上 \(M=M'\)。故 \((M,N)\mapsto L_{a,b,\delta}(t_{M,N})\) 在所给范围内单射，从而 \(\mathcal N_{+,-}\) 恰有 \((a-1)b\) 个节点。

再看 \((-,+)\) 型。方程组变为
\[
a(t+s)=2\pi M-2\delta,
\qquad
b(t-s)=2\pi N,
\]
故
\[
t=\frac{\pi M-\delta}{a}+\frac{\pi N}{b},
\qquad
s=\frac{\pi M-\delta}{a}-\frac{\pi N}{b}.
\]
若 \(N\equiv 0\pmod b\)，则 \(t\equiv s\pmod{2\pi}\)，故可取
\[
N\in\{1,\dots,b-1\},
\qquad
M\in\{0,\dots,a-1\}.
\]
这给出 \(a(b-1)\) 组参数对。

记对应节点坐标为
\[
x'_{M,N}=(-1)^M\cos\!\left(\pi a\frac{N}{b}\right),
\qquad
y'_{M,N}=(-1)^N\cos\!\left(\pi b\frac{M}{a}-\frac{b\delta}{a}\right).
\]
若
\[
(x'_{M,N},y'_{M,N})=(x'_{M',N'},y'_{M',N'}),
\]
则由第二坐标的余弦相等，若取“和”为 \(2\pi\) 整倍便会强制 \(2\delta/\pi\in\QQ\)，仍与假设矛盾；因此只能有
\[
b\frac{M-M'}{a}+N-N'\in 2\ZZ.
\]
由 \(\gcd(a,b)=1\) 与 \(0\le M,M'<a\) 得 \(M=M'\)，并且 \(N-N'\in 2\ZZ\)。再由第一坐标相等，
\[
\cos\!\left(\pi a\frac{N}{b}\right)=\cos\!\left(\pi a\frac{N'}{b}\right).
\]
于是只可能有
\[
a\frac{N-N'}{b}\in 2\ZZ
\qquad\text{或}\qquad
a\frac{N+N'}{b}\in 2\ZZ.
\]
第一种情形由 \(\gcd(a,b)=1\) 与 \(1\le N,N'\le b-1\) 立即推出 \(N=N'\)。第二种情形则强制 \(b\mid(N+N')\)，而
\[
2\le N+N'\le 2b-2,
\]
故只能有 \(N+N'=b\)。于是 \(a(N+N')/b=a\) 必为偶数，从而 \(a\) 为偶数；由 \(\gcd(a,b)=1\) 可知 \(b\) 为奇数，于是 \(N\) 与 \(N'=b-N\) 奇偶性相反，这与 \(N-N'\in 2\ZZ\) 矛盾。故第二种情形不可能。综上 \(N=N'\)。故 \((M,N)\mapsto L_{a,b,\delta}(t'_{M,N})\) 在所给范围内亦为单射，从而 \(\mathcal N_{-,+}\) 恰有 \(a(b-1)\) 个节点。

若某节点同时属于两族，则存在 \((M,N)\)、\((M',N')\) 使
\[
(-1)^M\cos\!\left(\delta+\pi a\frac{N}{b}\right)
=
(-1)^{M'}\cos\!\left(\pi a\frac{N'}{b}\right).
\]
再次由余弦相等的两种可能性立刻得到 \(\delta/\pi\in\QQ\)，矛盾。故两族互不相交。

最后证明横截性。对 \((+,-)\) 型节点，有
\[
L'_{a,b,\delta}(t)=(-a\alpha,\,-b\beta),
\qquad
L'_{a,b,\delta}(s)=(-a\alpha,\,b\beta),
\]
其中
\[
\alpha:=\sin(at+\delta)=\sin(as+\delta),
\qquad
\beta:=\sin(bt)=-\sin(bs).
\]
两切向量平行当且仅当 \(\alpha\beta=0\)。若 \(\alpha=0\)，则 \(at+\delta\in\pi\ZZ\)，代入
\[
t=\pi\left(\frac{M}{a}+\frac{N}{b}\right)
\]
可得 \(\delta/\pi\in\QQ\)，矛盾。若 \(\beta=0\)，则 \(bt\in\pi\ZZ\)，从而
\[
b\frac{M}{a}+N\in\ZZ,
\]
于是 \(a\mid bM\)。由 \(\gcd(a,b)=1\) 得 \(a\mid M\)，这与 \(1\le M\le a-1\) 矛盾。故 \((+,-)\) 型节点皆为横截节点。

对 \((-,+)\) 型节点，有
\[
L'_{a,b,\delta}(t)=(-a\alpha,\,-b\beta),
\qquad
L'_{a,b,\delta}(s)=(a\alpha,\,-b\beta),
\]
其中
\[
\alpha:=\sin(at+\delta)=-\sin(as+\delta),
\qquad
\beta:=\sin(bt)=\sin(bs).
\]
同理，切向量平行当且仅当 \(\alpha\beta=0\)。若 \(\alpha=0\)，则
\[
M+a\frac{N}{b}\in\ZZ,
\]
从而 \(b\mid aN\)，由互素性得 \(b\mid N\)，与 \(1\le N\le b-1\) 矛盾。若 \(\beta=0\)，则 \(bt\in\pi\ZZ\)，进而 \(\delta/\pi\in\QQ\)，仍矛盾。故 \((-,+)\) 型节点也全为横截节点。

综上，全部自交点恰由两族参数给出，且总数为
\[
(a-1)b+a(b-1)=2ab-a-b,
\]
并且全部都是普通横截节点。证毕。
\end{proof}

\begin{proposition}[Lissajous 共振相位与余弦门奇异环交会的等价判据]\label{prop:cdim-lissajous-resonant-phase-critical-ring-equivalence}
\leanverified{paper\_cdim\_lissajous\_resonant\_phase\_critical\_ring\_equivalence}
沿用定理 \ref{thm:cdim-nonresonant-lissajous-node-count} 的设定，并定义相位偏移余弦门
\[
\Pi_\delta:\TT^2\to[-1,1]^2,
\qquad
\Pi_\delta(u,v):=(\cos(u+\delta),\cos v).
\]
其临界集为
\[
\operatorname{Crit}(\Pi_\delta)
=
\{(u,v)\in\TT^2:\ \sin(u+\delta)=0\}\cup \{(u,v)\in\TT^2:\ \sin v=0\}.
\]
再记 \(\lambda_{a,b}(t):=(at,bt)\in\TT^2\)。
则以下三条命题等价：
\begin{enumerate}
  \item 存在 \(t\not\equiv s\pmod{2\pi}\) 使得
  \[
  at+\delta\equiv -(as+\delta)\pmod{2\pi},
  \qquad
  bt\equiv -bs\pmod{2\pi};
  \]
  \item 存在整数 \(M,N\) 使
  \[
  \delta=\pi\left(M-\frac{a}{b}N\right),
  \]
  从而 \(\delta/\pi\in\QQ\)；
  \item 一参数子流 \(\lambda_{a,b}(\RR)\) 与 \(\Pi_\delta\) 的两条奇异环同时相交，即存在 \(t\) 满足
  \[
  \sin(at+\delta)=0,
  \qquad
  \sin(bt)=0.
  \]
\end{enumerate}
当这些条件成立时，定理 \ref{thm:cdim-nonresonant-lissajous-node-count} 中的非共振节点账本失效：部分双点退化为更高重接触或并入奇异相位点，因而不再服从该定理的一般位置节点计数公式。
\end{proposition}
\begin{proof}
命题 (1) 中的方程等价于
\[
a(t+s)=2\pi M-2\delta,
\qquad
b(t+s)=2\pi N
\]
对某些 \(M,N\in\ZZ\) 成立。消去 \(t+s\) 立即得到
\[
\delta=\pi\left(M-\frac{a}{b}N\right),
\]
故 \((1)\Leftrightarrow(2)\)。

另一方面，
\[
\sin(bt)=0
\iff
bt\in\pi\ZZ,
\]
\[
\sin(at+\delta)=0
\iff
at+\delta\in\pi\ZZ.
\]
联立两式并消去 \(t\)，便得到与 (2) 完全相同的条件，故 \((2)\Leftrightarrow(3)\)。

若上述条件成立，则 \(\lambda_{a,b}(\RR)\) 穿过 \(\Pi_\delta\) 的奇异环交叉点，因而投影轨道不再保持定理 \ref{thm:cdim-nonresonant-lissajous-node-count} 中的横截双点结构。于是非共振情形下的一般位置节点计数公式不再适用。证毕。
\end{proof}
