\subsubsection{闭轨道的可逆代数提升：Lissajous 轨道、Joukowsky 门与 Laurent 玫瑰}\label{subsubsec:cdim-lissajous-joukowsky-prime-register-lift}
定理 \ref{thm:cdim-arithmetic-singular-ring-one-parameter-subgroups} 已把频率向量 \((a,b)\) 的相位演化实现为算术奇异环中的一参数子群陪集。
在本文的记账语义下，这一事实可以进一步重写为“相位圆—素数账本核”二分结构：可见平面轨道来自相位圆在 Joukowsky 门下的投影，而全部共同素因子信息被压缩在有限覆叠核中，并由 prime-register 精确记录。
由此，平面上的闭合轨迹不再只是几何像，而获得一个可逆的代数提升。

\begin{theorem}[Lissajous 轨道的相位圆—素数账本核分解]\label{thm:cdim-lissajous-phase-circle-prime-ledger-kernel}
\leanverified{paper\_cdim\_lissajous\_phase\_circle\_prime\_ledger\_kernel}
设 \(a,b\in\NN_{>0}\)、\(\delta\in\RR\)，并记
\[
d:=\gcd(a,b),\qquad \alpha:=a/d,\qquad \beta:=b/d,\qquad \zeta:=e^{\mathrm{i}\delta}.
\]
定义
\[
q_d:\TT\to\TT,\qquad q_d(z)=z^d,
\]
\[
\widetilde\phi_{\alpha,\beta}:\TT\to\TT^2,\qquad \widetilde\phi_{\alpha,\beta}(w)=(w^\alpha,w^\beta),
\]
\[
\tau_\zeta:\TT^2\to\TT^2,\qquad \tau_\zeta(u,v)=(\zeta u,v),
\]
以及 Joukowsky 门
\[
J:\TT\to[-1,1],\qquad J(u)=\frac{u+u^{-1}}{2}.
\]
再置
\[
C_{\alpha,\beta}:=\widetilde\phi_{\alpha,\beta}(\TT)
=\{(u,v)\in\TT^2:\ u^\beta=v^\alpha\},
\]
\[
C_{\alpha,\beta,\zeta}:=\tau_\zeta(C_{\alpha,\beta})
=\{(u,v)\in\TT^2:\ u^\beta=\zeta^\beta v^\alpha\},
\]
以及
\[
\phi_{a,b,\delta}:=\tau_\zeta\circ \widetilde\phi_{\alpha,\beta}\circ q_d,
\qquad
\mathcal L_{a,b,\delta}:=(J\times J)\circ \phi_{a,b,\delta}.
\]
则：
\begin{enumerate}
  \item \(\widetilde\phi_{\alpha,\beta}\) 将 \(\TT\) 同构到 primitive 子环面 \(C_{\alpha,\beta}\)；
  \item \(q_d\) 为 \(d\)-对-\(1\) 覆盖，其核为 \(\mu_d=\{z\in\TT:\ z^d=1\}\)；
  \item \(\phi_{a,b,\delta}(\TT)=C_{\alpha,\beta,\zeta}\)，并且
  \[
  \mathcal L_{a,b,\delta}
  =(J\times J)\circ \tau_\zeta\circ \widetilde\phi_{\alpha,\beta}\circ q_d.
  \]
\end{enumerate}
特别地，Lissajous 轨道的可见轨形仅依赖于 primitive 频率对 \((\alpha,\beta)\) 与相位平移 \(\zeta\)，而全部共同因子信息则集中在有限核 \(\mu_d\) 中。
在定理 \ref{thm:cdim-arithmetic-singular-ring-one-parameter-subgroups} 的记号下，\(C_{\alpha,\beta,\zeta}\) 恰为陪集 \(g_\delta\cdot\gamma_{(a,b)}(\RR)\) 在 \(\TT^2\) 上的投影。
\end{theorem}
\begin{proof}
由定义立刻有
\[
\phi_{a,b,\delta}(z)
=
\tau_\zeta\bigl((z^d)^\alpha,(z^d)^\beta\bigr)
=
(\zeta z^{d\alpha},z^{d\beta})
=
(\zeta z^a,z^b).
\]
故最后一个分解式成立。
\par
先证第一条。若 \(\widetilde\phi_{\alpha,\beta}(w_1)=\widetilde\phi_{\alpha,\beta}(w_2)\)，则 \(\eta:=w_1w_2^{-1}\) 满足
\[
\eta^\alpha=1,\qquad \eta^\beta=1.
\]
由于 \(\gcd(\alpha,\beta)=1\)，必有 \(\eta=1\)，故 \(w_1=w_2\)，于是 \(\widetilde\phi_{\alpha,\beta}\) 单射。
其像满足关系 \(u^\beta=v^\alpha\)，故包含于 \(C_{\alpha,\beta}\)。
反过来，若 \((u,v)\in C_{\alpha,\beta}\)，写 \(u=e^{\mathrm{i}s}\)、\(v=e^{\mathrm{i}t}\)，则
\[
\beta s-\alpha t\in 2\pi\ZZ.
\]
取整数 \(m,n\) 使 \(m\alpha+n\beta=1\)，并令 \(\theta:=ms+nt\)。
则
\[
\alpha\theta-s=n(\alpha t-\beta s)\in 2\pi\ZZ,
\qquad
\beta\theta-t=m(\beta s-\alpha t)\in 2\pi\ZZ.
\]
故对 \(w=e^{\mathrm{i}\theta}\) 有 \(w^\alpha=u\)、\(w^\beta=v\)，从而 \((u,v)\in \widetilde\phi_{\alpha,\beta}(\TT)\)。
于是 \(\widetilde\phi_{\alpha,\beta}\) 是 \(\TT\) 到 \(C_{\alpha,\beta}\) 的双连续双射，给出紧群同构。
\par
第二条直接由 \(q_d(z)=1\iff z^d=1\) 得 \(\ker(q_d)=\mu_d\)，并且每个 \(w\in\TT\) 恰有 \(d\) 个 \(d\)-次根。
\par
第三条由 \(\tau_\zeta(C_{\alpha,\beta})=C_{\alpha,\beta,\zeta}\) 与
\[
\phi_{a,b,\delta}(\TT)
=
\tau_\zeta\bigl(\widetilde\phi_{\alpha,\beta}(q_d(\TT))\bigr)
=
\tau_\zeta(C_{\alpha,\beta})
=
C_{\alpha,\beta,\zeta}
\]
立即得到。
最后，定理 \ref{thm:cdim-arithmetic-singular-ring-one-parameter-subgroups} 已说明
\[
t\longmapsto (e^{\mathrm{i}(at+\delta)},e^{\mathrm{i}bt})
\]
正是陪集 \(g_\delta\cdot\gamma_{(a,b)}(\RR)\) 的 \(\TT^2\)-投影；令 \(z=e^{\mathrm{i}t}\) 即知其像集等于 \(C_{\alpha,\beta,\zeta}\)。证毕。
\end{proof}

\begin{corollary}[prime-register 交账本给出的核阶]\label{cor:cdim-lissajous-prime-register-meet-kernel}
\leanverified{paper\_cdim\_lissajous\_prime\_register\_meet\_kernel}
沿用定理 \ref{thm:cdim-lissajous-phase-circle-prime-ledger-kernel} 的记号。
对每个素数 \(p\) 定义 prime-register
\[
r_a(p):=v_p(a),\qquad r_b(p):=v_p(b),\qquad (r_a\wedge r_b)(p):=\min\{r_a(p),r_b(p)\}.
\]
则
\[
\mu_d\cong \prod_{p}\mu_{p^{(r_a\wedge r_b)(p)}},
\qquad
|\mu_d|
=
d
=
N(r_a\wedge r_b)
=
\prod_{p}p^{(r_a\wedge r_b)(p)}.
\]
因此 Lissajous 轨道中不可见的共同因子部分，恰由 \(r_a\wedge r_b\) 所记录。
\end{corollary}
\begin{proof}
算术恒等式
\[
d=\gcd(a,b)=\prod_{p}p^{\min\{v_p(a),v_p(b)\}}
\]
给出第二式。
又 \(\mu_d\) 为阶 \(d\) 的循环群，按中国剩余分解可写为其各 \(p\)-primary 部分的直积，故得第一式。证毕。
\end{proof}

\begin{theorem}[Lissajous 轨道的分支除子与边界签名]\label{thm:cdim-lissajous-branch-divisor-boundary-signature}
沿用定理 \ref{thm:cdim-lissajous-phase-circle-prime-ledger-kernel} 的记号。
令
\[
B:=\{u=\pm1\}\cup\{v=\pm1\}\subset\TT^2,
\qquad
\Sigma_{a,b,\delta}:=\phi_{a,b,\delta}^{-1}(B),
\qquad
\mathscr B_{\alpha,\beta,\zeta}:=C_{\alpha,\beta,\zeta}\cap B.
\]
再定义可见边界签名
\[
S_x^\pm:=\{y\in[-1,1]:(\pm1,y)\in \mathcal L_{a,b,\delta}(\TT)\},
\]
\[
S_y^\pm:=\{x\in[-1,1]:(x,\pm1)\in \mathcal L_{a,b,\delta}(\TT)\}.
\]
则：
\begin{enumerate}
  \item \(\#(C_{\alpha,\beta,\zeta}\cap\{u=\pm1\})=2\alpha\)，且 \(\#(C_{\alpha,\beta,\zeta}\cap\{v=\pm1\})=2\beta\)；
  \item 若
  \[
  c:=\#\bigl(C_{\alpha,\beta,\zeta}\cap\{u=\pm1\}\cap\{v=\pm1\}\bigr),
  \]
  则
  \[
  c=
  \begin{cases}
  0,& \zeta^\beta\notin\{\pm1\},\\
  2,& \zeta^\beta\in\{\pm1\},
  \end{cases}
  \]
  因而
  \[
  \#\mathscr B_{\alpha,\beta,\zeta}=2\alpha+2\beta-c;
  \]
  \item 若 \(\zeta^{2\beta}\neq 1\)，则
  \[
  |S_x^+|=|S_x^-|=\alpha,\qquad |S_y^+|=|S_y^-|=\beta.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
对第一条，固定 \(\varepsilon\in\{\pm1\}\)。
在 \(u=\varepsilon\) 上，关系 \(u^\beta=\zeta^\beta v^\alpha\) 化为
\[
v^\alpha=\zeta^{-\beta}\varepsilon^\beta.
\]
由于 \(v\mapsto v^\alpha\) 在 \(\TT\) 上恰有 \(\alpha\) 个原像，故每条边 \(u=\varepsilon\) 与 \(C_{\alpha,\beta,\zeta}\) 恰交于 \(\alpha\) 点；两种符号合计 \(2\alpha\) 点。
对 \(v=\varepsilon\) 完全同理，得到 \(2\beta\)。
\par
对第二条，角点满足 \((u,v)=(\varepsilon_1,\varepsilon_2)\in\{\pm1\}^2\)，其相容条件为
\[
\varepsilon_1^\beta=\zeta^\beta \varepsilon_2^\alpha.
\]
若 \(\zeta^\beta\notin\{\pm1\}\)，则右端不可能取值于 \(\{\pm1\}\)，故无角点。
若 \(\zeta^\beta\in\{\pm1\}\)，则因 \(\gcd(\alpha,\beta)=1\)，\(\alpha,\beta\) 不可能同时为偶数。
若 \(\beta\) 为奇数，则 \(\varepsilon_1\) 由 \(\varepsilon_2\) 唯一决定，故有 \(2\) 个角点；
若 \(\beta\) 为偶数，则 \(\alpha\) 必为奇数，此时 \(\varepsilon_2\) 由 \(\varepsilon_1\) 唯一决定，也有 \(2\) 个角点。
于是 \(c\) 的分类成立，再用容斥原理即得 \(\#\mathscr B_{\alpha,\beta,\zeta}=2\alpha+2\beta-c\)。
\par
对第三条，先看 \(S_x^\varepsilon\)。
边 \(x=\varepsilon\) 上的可见点对应于 \(C_{\alpha,\beta,\zeta}\cap\{u=\varepsilon\}\) 中各点的第二坐标经 \(J\) 的像。
上一步已知该交集中共有 \(\alpha\) 个点，其第二坐标 \(v_1,\dots,v_\alpha\) 满足
\[
v_j^\alpha=\zeta^{-\beta}\varepsilon^\beta.
\]
若存在 \(j\neq k\) 使 \(J(v_j)=J(v_k)\)，则必有 \(v_j=v_k^{-1}\)。
于是
\[
1=(v_jv_k)^\alpha=(\zeta^{-\beta}\varepsilon^\beta)^2=\zeta^{-2\beta},
\]
即 \(\zeta^{2\beta}=1\)，与假设矛盾。
故这些 \(J(v_j)\) 两两不同，从而 \(|S_x^\varepsilon|=\alpha\)。
\par
同理，边 \(y=\varepsilon\) 上共有 \(\beta\) 个第一坐标 \(u_1,\dots,u_\beta\) 满足
\[
u_j^\beta=\zeta^\beta \varepsilon^\alpha.
\]
若 \(J(u_j)=J(u_k)\) 且 \(j\neq k\)，则 \(u_j=u_k^{-1}\)，从而
\[
1=(u_ju_k)^\beta=(\zeta^\beta \varepsilon^\alpha)^2=\zeta^{2\beta},
\]
仍与假设矛盾。
故 \(|S_y^\varepsilon|=\beta\)。证毕。
\end{proof}

\begin{corollary}[边界签名与核账本的联合恢复]\label{cor:cdim-lissajous-boundary-kernel-recovery}
\leanverified{paper\_cdim\_lissajous\_boundary\_kernel\_recovery}
在定理 \ref{thm:cdim-lissajous-branch-divisor-boundary-signature} 的假设 \(\zeta^{2\beta}\neq1\) 下，有
\[
a=N(r_a\wedge r_b)\,|S_x^\pm|,
\qquad
b=N(r_a\wedge r_b)\,|S_y^\pm|.
\]
因此，可见边界签名恢复 primitive 频率对 \((\alpha,\beta)\)，prime-register 交账本恢复共同核 \(\mu_d\)；二者合并即可唯一恢复原始频率对 \((a,b)\) 及其全部素数指数。
\end{corollary}
\begin{proof}
由定理 \ref{thm:cdim-lissajous-branch-divisor-boundary-signature} 得 \(|S_x^\pm|=\alpha\)、\(|S_y^\pm|=\beta\)；
由推论 \ref{cor:cdim-lissajous-prime-register-meet-kernel} 得 \(d=N(r_a\wedge r_b)\)。
代入 \(a=d\alpha\)、\(b=d\beta\) 即得结论。证毕。
\end{proof}

\begin{theorem}[Laurent 角色玫瑰轨道的有限奇异环]\label{thm:cdim-rose-laurent-character-finite-singular-ring}
\leanverified{paper\_cdim\_rose\_laurent\_character\_finite\_singular\_ring}
设整数 \(d>n\ge 1\)，并定义
\[
F_{d,n}:\TT\to\CC,\qquad
F_{d,n}(z):=\frac{z^{d+n}+z^{d-n}}{2}.
\]
令
\[
\gamma(t):=F_{d,n}(e^{\mathrm{i}t})=e^{\mathrm{i}dt}\cos(nt),
\qquad
\theta:=dt.
\]
则 \(\gamma\) 正是玫瑰轨道
\[
r(\theta)=\cos\!\Bigl(\frac{n}{d}\theta\Bigr)
\]
的闭合参数化。
定义其相位圆奇异环为
\[
\Sigma^{\mathrm{rose}}_{d,n}:=\{z\in\TT:\ F_{d,n}(z)=0\}.
\]
则：
\begin{enumerate}
  \item
  \[
  \Sigma^{\mathrm{rose}}_{d,n}=\{z\in\TT:\ z^{2n}=-1\},
  \qquad
  \#\Sigma^{\mathrm{rose}}_{d,n}=2n;
  \]
  \item 原点 \(0\in\CC\) 是 \(\gamma\) 的 \(2n\)-重横截点：存在互异参数
  \[
  t_k=\frac{(2k+1)\pi}{2n}\qquad(0\le k\le 2n-1)
  \]
  使 \(\gamma(t_k)=0\) 且 \(\gamma'(t_k)\neq 0\)；
  \item 原点处的不同切线条数为
  \[
  \frac{n}{\gcd(n,d)}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由
\[
F_{d,n}(z)=\frac{z^{d-n}(z^{2n}+1)}{2}
\]
且 \(|z|=1\) 上 \(z^{d-n}\neq 0\)，立得
\[
F_{d,n}(z)=0 \iff z^{2n}=-1,
\]
从而第一条成立。
\par
对第二条，由
\[
\gamma(t)=e^{\mathrm{i}dt}\cos(nt)
\]
知 \(\gamma(t)=0\iff \cos(nt)=0\)，即 \(t=t_k\)。
又
\[
\gamma'(t)=e^{\mathrm{i}dt}\bigl(\mathrm{i}d\cos(nt)-n\sin(nt)\bigr).
\]
在 \(t=t_k\) 处有 \(\cos(nt_k)=0\)、\(\sin(nt_k)=\pm1\)，故
\[
\gamma'(t_k)=-n\sin(nt_k)e^{\mathrm{i}dt_k}\neq 0.
\]
因此原点处每一分支均为横截而非尖点。
\par
对第三条，切线方向由 \(\gamma'(t_k)\) 的方向决定，而相反向量给出同一条直线。
于是只需考察
\[
e^{2\mathrm{i}dt_k}
=
e^{\mathrm{i}d(2k+1)\pi/n}
=
e^{\mathrm{i}d\pi/n}\bigl(e^{2\pi\mathrm{i}d/n}\bigr)^k.
\]
前因子不影响基数，因此不同切线条数等于根单位 \(e^{2\pi\mathrm{i}d/n}\) 的阶，即
\[
\frac{n}{\gcd(n,d)}.
\]
证毕。
\end{proof}

\begin{remark}[闭轨道的可逆提升范式]\label{rem:cdim-closed-trajectory-reversible-lift}
Lissajous 情形中，可见图形由 \(C_{\alpha,\beta,\zeta}\) 经 \(J\times J\) 投影得到，而 prime-register 交账本 \(r_a\wedge r_b\) 记录全部共同因子核；
玫瑰情形中，可见图形由 Laurent 角色 \(F_{d,n}\) 给出，而其有限奇异环 \(\Sigma^{\mathrm{rose}}_{d,n}\) 记录原点处的多分支结构。
两者共同说明：闭合轨道的基本对象并非平面像本身，而是相位圆上的代数角色、有限核与分支除子的组合；
prime-register 因而提供了一条把几何闭轨道提升为可逆代数对象的统一接口。
\end{remark}

\paragraph{一致取根塔、有限分母可见性与有限素数见证}
为把上述有限核提升到一致取根塔，对整数 \(d\ge 2\) 记
\[
\Sigma_d:=\widehat{\ZZ[1/d]}
\cong
\varprojlim\bigl(\TT\xleftarrow{(\cdot)^d}\TT\xleftarrow{(\cdot)^d}\cdots\bigr).
\]
若 \(x=(z_0,z_1,\dots)\in \Sigma_d\) 满足 \(z_k=z_{k+1}^{\,d}\)，则记 \(\pi_k(x):=z_k\)。
对任意 \(q=m/d^K\in \ZZ[1/d]\)（\(m\in\ZZ,\ K\ge 0\)），定义
\[
\chi_q(x):=\pi_K(x)^m.
\]
由兼容关系 \(\pi_K=\pi_{K+1}^{\,d}\) 可知该定义与 \(q\) 的表示无关，并给出 \(\Sigma_d\) 的全部连续特征标。

\begin{theorem}[有理玫瑰轨道的 \texorpdfstring{$d$}{d}-adic solenoid 去奇点提升]\label{thm:cdim-rhodonea-solenoid-desingularized-lift}
\leanverified{paper\_cdim\_rhodonea\_solenoid\_desingularized\_lift}
取互素整数 \(d>n\ge 1\)。
定义连续映射
\[
\mathcal R_{n,d}:\Sigma_d\to\CC,
\qquad
\mathcal R_{n,d}(x):=\frac{\pi_1(x)^{d+n}+\pi_1(x)^{d-n}}{2}
=\frac{\chi_{(d+n)/d}(x)+\chi_{(d-n)/d}(x)}{2},
\]
以及连续群同态
\[
\iota_d:\RR/(2\pi d\ZZ)\to \Sigma_d,
\qquad
\iota_d([\theta]):=\bigl(e^{\mathrm{i}\theta},e^{\mathrm{i}\theta/d},e^{\mathrm{i}\theta/d^2},\dots\bigr).
\]
则 \(\iota_d\) 为拓扑嵌入，且
\[
\boxed{\;
(\mathcal R_{n,d}\circ \iota_d)([\theta])
=
e^{\mathrm{i}\theta}\cos\!\Bigl(\frac{n}{d}\theta\Bigr).\;}
\]
因此 \(\mathcal R_{n,d}\bigl(\iota_d(\RR/(2\pi d\ZZ))\bigr)\) 恰为极坐标曲线
\[
r(\theta)=\cos\!\Bigl(\frac{n}{d}\theta\Bigr)
\]
的全部闭合花瓣轨道；平面中的自交完全来自投影 \(\mathcal R_{n,d}\) 在该闭子群上的非单射性，而不是提升轨道本身的自交。
\end{theorem}
\begin{proof}
首先，\(\iota_d\) 的每个坐标都与模 \(2\pi d\ZZ\) 的商兼容，故其定义良好。
若 \(\iota_d([\theta])=\iota_d([\theta'])\)，则第二坐标相等：
\[
e^{\mathrm{i}\theta/d}=e^{\mathrm{i}\theta'/d},
\]
从而 \(\theta-\theta'\in 2\pi d\ZZ\)，即 \([\theta]=[\theta']\)。
因此 \(\iota_d\) 连续单射；又因其定义域紧而 \(\Sigma_d\) Hausdorff，故 \(\iota_d\) 为拓扑嵌入，且像为闭子群。

随后直接计算
\[
\begin{aligned}
(\mathcal R_{n,d}\circ \iota_d)([\theta])
&=
\frac{e^{\mathrm{i}(d+n)\theta/d}+e^{\mathrm{i}(d-n)\theta/d}}{2} \\
&=
e^{\mathrm{i}\theta}\frac{e^{\mathrm{i}n\theta/d}+e^{-\mathrm{i}n\theta/d}}{2}
=
e^{\mathrm{i}\theta}\cos\!\Bigl(\frac{n}{d}\theta\Bigr).
\end{aligned}
\]
这正是所述 Rhodonea 轨道的复参数式。证毕。
\end{proof}

\begin{proposition}[幂覆盖不承认全局连续开方截面]\label{prop:cdim-circle-power-no-continuous-section}
\leanverified{paper\_cdim\_circle\_power\_no\_continuous\_section}
对任意 \(d\ge 2\)，不存在连续映射
\[
s:\TT\to\TT
\]
满足 \(s(z)^d=z\) 对所有 \(z\in\TT\) 成立。
等价地，覆盖映射
\[
q_d:\TT\to\TT,
\qquad
q_d(z)=z^d
\]
不承认连续截面。
因而任何要求对分母 \(d\) 的取根过程保持全局连续的相位模型，都必须以一致取根塔为承载对象；在本文口径下，其最小紧相位系统正是 \(\Sigma_d\cong\widehat{\ZZ[1/d]}\)。
\end{proposition}
\begin{proof}
若存在这样的 \(s\)，则 \(q_d\circ s=\mathrm{id}_{\TT}\)。
对基本群 \(\pi_1(\TT)\cong\ZZ\) 取诱导同态，得到
\[
(q_d)_*\circ s_*=\mathrm{id}_{\ZZ}.
\]
但 \((q_d)_*\) 恰为乘 \(d\) 映射 \(\ZZ\xrightarrow{\times d}\ZZ\)，当 \(d\ge 2\) 时它在 \(\ZZ\) 上无右逆，矛盾。
故所述连续截面不存在。证毕。
\end{proof}

\begin{theorem}[Lissajous 频率对决定的最小相位 solenoid]\label{thm:cdim-lissajous-minimal-solenoid-factor}
\leanverified{paper\_cdim\_lissajous\_minimal\_solenoid\_factor}
给定正整数 \(a,b\)，令
\[
S:=\{p:\ p\mid ab\},
\qquad
G_S:=\ZZ[S^{-1}],
\qquad
\Sigma_S:=\widehat{G_S}.
\]
并对每个 \(q\in G_S\) 记 \(\chi_q\in\widehat{\Sigma_S}\cong G_S\) 为对应特征标。
则存在唯一连续一参数子群
\[
\eta:\RR\to\Sigma_S
\]
满足：对每个 \(q\in G_S\) 与 \(t\in\RR\)，均有
\[
\boxed{\;
\chi_q(\eta(t))=e^{\mathrm{i}qt}.\;}
\]
特别地，对任意 \(\delta\in\RR\)，
\[
\boxed{\;
t\longmapsto
\bigl(\Re(e^{\mathrm{i}\delta}\chi_a(\eta(t))),\Re(\chi_b(\eta(t)))\bigr)
=
\bigl(\cos(at+\delta),\cos(bt)\bigr).\;}
\]
此外，若 \(K\) 为紧交换群且其对偶群 \(\widehat K\) 含有 \(G_S\) 的一个子群副本，则存在连续满同态
\[
K\twoheadrightarrow \Sigma_S.
\]
因此 \(\Sigma_S\) 是由频率对 \((a,b)\) 的全部分母闭包所强迫出的最小紧相位因子。
\end{theorem}
\begin{proof}
Pontryagin 对偶给出自然同构
\[
\Hom_{\mathrm{cont}}(\RR,\Sigma_S)
\cong
\Hom(G_S,\widehat{\RR})
\cong
\Hom(G_S,\RR).
\]
包含映射 \(G_S\hookrightarrow\RR,\ q\mapsto q\) 因而唯一决定一参数子群 \(\eta\)，并满足
\[
\chi_q(\eta(t))=e^{\mathrm{i}qt}
\qquad(q\in G_S).
\]
对 \(q=a,b\) 分别代入并取实部，即得所示 Lissajous 参数式。

若 \(G_S\hookrightarrow\widehat K\)，则对偶化立即给出连续满同态
\[
K\twoheadrightarrow \widehat{G_S}=\Sigma_S.
\]
故最小性断言成立。证毕。
\end{proof}

\begin{corollary}[Rhodonea 与 Lissajous 的有理特征标统一表述]\label{cor:cdim-rational-character-projection-unifies-rose-lissajous}
\leanverified{paper\_cdim\_rational\_character\_projection\_unifies\_rose\_lissajous}
设 \(S\) 为有限素数集，\(\eta:\RR\to\Sigma_S\) 如定理 \ref{thm:cdim-lissajous-minimal-solenoid-factor}。
若既约分数 \(n/d\) 满足其分母的全部素因子都落在 \(S\) 中，并记
\[
q_{\pm}:=\frac{d\pm n}{d}\in G_S,
\]
则有
\[
\frac{\chi_{q_+}(\eta(\theta))+\chi_{q_-}(\eta(\theta))}{2}
=
e^{\mathrm{i}\theta}\cos\!\Bigl(\frac{n}{d}\theta\Bigr).
\]
另一方面，对任意 \(a,b\in\NN_{>0}\) 与 \(\delta\in\RR\)，
\[
t\longmapsto
\bigl(\Re(e^{\mathrm{i}\delta}\chi_a(\eta(t))),\Re(\chi_b(\eta(t)))\bigr)
=
\bigl(\cos(at+\delta),\cos(bt)\bigr).
\]
因此，Rhodonea 轨道与 Lissajous 轨道都可统一写成同类 solenoid 上有限个有理频率特征标的线性投影；两者的差别仅在于所选频率集合与观测算子的不同。
\end{corollary}
\begin{proof}
由定理 \ref{thm:cdim-lissajous-minimal-solenoid-factor}，
\[
\chi_{q_\pm}(\eta(\theta))=e^{\mathrm{i}(d\pm n)\theta/d}.
\]
于是
\[
\frac{\chi_{q_+}(\eta(\theta))+\chi_{q_-}(\eta(\theta))}{2}
=
\frac{e^{\mathrm{i}(d+n)\theta/d}+e^{\mathrm{i}(d-n)\theta/d}}{2}
=
e^{\mathrm{i}\theta}\cos\!\Bigl(\frac{n}{d}\theta\Bigr),
\]
而第二条同样直接由 \(\chi_q(\eta(t))=e^{\mathrm{i}qt}\) 取实部得到。证毕。
\end{proof}

\begin{theorem}[有限分母观测的有限层可见性与地址碰撞屏障]\label{thm:cdim-solenoid-finite-denominator-visibility-barrier}
\leanverified{paper\_cdim\_solenoid\_finite\_denominator\_visibility\_barrier}
在 \(\Sigma_d\) 上取有限集合
\[
\mathcal Q\subset d^{-K}\ZZ
\qquad (K\ge 0),
\]
并定义三角多项式观测
\[
F(x):=\sum_{q\in\mathcal Q}c_q\chi_q(x)
\qquad (c_q\in\CC).
\]
则存在 Laurent 多项式 \(P\in\CC[z,z^{-1}]\) 使得
\[
\boxed{\;
F=P\circ\pi_K.\;}
\]
因此，任何由分母至多为 \(d^K\) 的有限频率生成的观测都只能看见 \(\Sigma_d\) 的前 \(K\) 层取根地址；更深层坐标对该观测完全不可见。
特别地，任何由此类有限频率观测诱导的平面投影，其自交与多重覆盖都完全由 \(\pi_K\)-纤维的碰撞控制。
\end{theorem}
\begin{proof}
对每个 \(q\in\mathcal Q\)，可写成
\[
q=\frac{m_q}{d^{k_q}}
\qquad (m_q\in\ZZ,\ 0\le k_q\le K).
\]
由 \(\pi_{k_q}=\pi_K^{\,d^{K-k_q}}\) 得
\[
\chi_q(x)=\pi_{k_q}(x)^{m_q}
=
\pi_K(x)^{m_qd^{K-k_q}}.
\]
故若置
\[
P(z):=\sum_{q\in\mathcal Q}c_q\,z^{m_qd^{K-k_q}},
\]
则对一切 \(x\in\Sigma_d\) 都有 \(F(x)=P(\pi_K(x))\)。
于是 \(F\) 仅依赖于第 \(K\) 层投影 \(\pi_K\)。
因而凡是 \(\pi_K(x)=\pi_K(y)\) 的两点，都无法被任何此类有限分母观测区分；所述碰撞结论随即成立。证毕。
\end{proof}

\begin{theorem}[通用 solenoid 上的有限素数见证原理]\label{thm:cdim-qhat-finite-prime-witness}
记
\[
\Sigma_{\QQ}:=\widehat{\QQ},
\]
并对每个有限素数集 \(S\) 令
\[
\pi_S:\Sigma_{\QQ}\twoheadrightarrow \Sigma_S
\]
为包含 \(\ZZ[S^{-1}]\hookrightarrow \QQ\) 的对偶满射。
由于
\[
\QQ=\varinjlim_{S\Subset\mathbb P}\ZZ[S^{-1}],
\]
Pontryagin 对偶把该直极限送为逆极限，故有自然同构
\[
\Sigma_{\QQ}\cong \varprojlim_{S\Subset\mathbb P}\Sigma_S.
\]
若
\[
\kappa(x)=\sum_{q\in\mathcal Q}c_q\chi_q(x)
\qquad (\mathcal Q\subset\QQ\ \text{有限})
\]
是 \(\Sigma_{\QQ}\) 上的三角多项式（其中 \(\chi_q\) 表示 \(\QQ\) 中 \(q\) 所对应的特征标），则存在有限素数集 \(S\) 与 \(\Sigma_S\) 上的三角多项式 \(\kappa_S\)，使得
\[
\boxed{\;
\kappa=\kappa_S\circ\pi_S.\;}
\]
可取 \(S\) 为 \(\mathcal Q\) 中全部有理数分母素因子的并集。
因此，任何由有限个有理频率特征标构造的 Toeplitz、Gram 或正定性证书，都已在某个有限 prime-register 阶段 \(\Sigma_S\) 上封闭；若该证书失效，则其失效必可由某个有限素数子系统见证。
\end{theorem}
\begin{proof}
由于 \(\mathcal Q\) 有限，可取有限素数集 \(S\) 使 \(\mathcal Q\subset \ZZ[S^{-1}]\)。
对每个 \(q\in\mathcal Q\)，其作为 \(\QQ\) 的字符在 \(\Sigma_{\QQ}\) 上的评价仅依赖于 \(\pi_S(x)\)，故存在 \(\Sigma_S\) 上的字符 \(\widetilde\chi_q\) 满足
\[
\chi_q=\widetilde\chi_q\circ\pi_S.
\]
令
\[
\kappa_S(y):=\sum_{q\in\mathcal Q}c_q\widetilde\chi_q(y),
\]
即得 \(\kappa=\kappa_S\circ\pi_S\)。

任何有限 Toeplitz 或 Gram 证书只涉及有限个此类角色值及其有限次线性代数运算，因而同样经由同一 \(\pi_S\) 因子化。
于是其成立或失效都已在该有限阶段完全决定。证毕。
\end{proof}
