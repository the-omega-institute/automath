\begin{theorem}[局部化 solenoid 与算术奇异环的扭结核公式]\label{thm:cdim-solenoid-arithmetic-singular-ring-torsion-kernel-formula}
令 \(S\subset\mathbb{P}\) 为有限素数集，记 \(s:=|S|\ge 1\)，并令 \(\Sigma_S=\widehat{\ZZ[S^{-1}]}\) 与 \(\mathfrak{R}_S=\Sigma_S\times_{\TT}\TT^{S}\) 如上。
对整数 \(m\ge 1\)，按素因子分解写
\[
m=\prod_{p}p^{v_p(m)},
\qquad
m_{S}:=\prod_{p\in S}p^{v_p(m)},
\qquad
m_{S^c}:=\prod_{p\notin S}p^{v_p(m)}.
\]
记 \([m]\) 为紧交换群上的“乘以 \(m\)”自同态。
则：
\begin{enumerate}
  \item 在 \(\Sigma_S\) 上，
  \[
  \ker([m]:\Sigma_S\to \Sigma_S)\ \cong\ \mu_{m_{S^c}},
  \qquad
  \bigl|\ker([m]|_{\Sigma_S})\bigr|=m_{S^c}.
  \]
  \item 在 \(\mathfrak{R}_S\) 上，
  \[
  \bigl|\ker([m]:\mathfrak{R}_S\to \mathfrak{R}_S)\bigr|
  \ =\ m_{S^c}\cdot m^{\,s-1}.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(1)} 由 Pontryagin 对偶，\([m]\) 在 \(\Sigma_S\) 上的对偶同态为
\(\times m:\ZZ[S^{-1}]\to \ZZ[S^{-1}]\)。
因此
\[
\ker([m]|_{\Sigma_S})\ \widehat{\cong}\ \ZZ[S^{-1}]/m\,\ZZ[S^{-1}].
\]
在环 \(\ZZ[S^{-1}]\) 中，\(m_S\) 为单位元，故
\(
m\,\ZZ[S^{-1}]=m_{S^c}\,\ZZ[S^{-1}]
\)。
又 \(m_{S^c}\) 与 \(S\) 互素，局部化不改变模 \(m_{S^c}\) 的商，因此
\[
\ZZ[S^{-1}]/m\,\ZZ[S^{-1}]
\ \cong\
\ZZ[S^{-1}]/m_{S^c}\,\ZZ[S^{-1}]
\ \cong\
\ZZ/m_{S^c}\ZZ.
\]
对偶化得
\(
\ker([m]|_{\Sigma_S})\cong (\ZZ/m_{S^c}\ZZ)^\wedge\cong \mu_{m_{S^c}}
\)，核大小为 \(m_{S^c}\)。
\par
\textbf{(2)} 由推论 \ref{cor:cdim-arithmetic-singular-ring-splitting}，紧交换群扩张
\(
0\to \ker(\mu_S)\to \mathfrak{R}_S\to \Sigma_S\to 0
\)
在交换群范畴中分裂，从而存在（非典范）同构
\(
\mathfrak{R}_S\cong \Sigma_S\times \TT^{s-1}
\)。
在任意此类同构下，\([m]\) 变为分量上的乘法自同态，故
\[
\ker([m]|_{\mathfrak{R}_S})
\cong
\ker([m]|_{\Sigma_S})\times \ker([m]|_{\TT^{s-1}}).
\]
由 \(|\ker([m]|_{\TT^{s-1}})|=m^{s-1}\) 与 (1) 合并得到
\(
|\ker([m]|_{\mathfrak{R}_S})|=m_{S^c}\cdot m^{s-1}
\)。
证毕。
\end{proof}
