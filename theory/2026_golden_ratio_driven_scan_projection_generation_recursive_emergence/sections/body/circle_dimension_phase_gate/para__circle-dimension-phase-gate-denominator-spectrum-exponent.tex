\paragraph{分母自由度的谱指数：可逆素集合的高度增长律}\label{par:cdim-denominator-spectrum-exponent}
局部化群 \(\ZZ[S^{-1}]\) 的“结构”在连续相位通道意义下仍由圆因子与对偶连通部分控制（定义 \ref{def:circle-dimension}，命题 \ref{prop:circle-dimension-laws}）；
然而当协议把 \(\ZZ[S^{-1}]\) 作为实现层可逆算术对象使用时，\emph{分母}所允许的乘法半群规模决定了在有界高度窗口内需要外置多少离散证书轴。
本段以一个单标量指数把该“分母自由度”压缩为可审计的增长律。

\begin{definition}[分母自由度计数与谱指数]\label{def:cdim-denominator-spectrum-index}
固定素数集合 \(S\)。
定义由 \(S\) 生成的乘法半群
\[
\langle S\rangle
:=
\Bigl\{\prod_{p\in S}p^{e(p)}:\ e(p)\in\NN,\ e(p)=0\text{ 除有限多 }p\Bigr\}\subset\NN,
\]
并定义分母自由度计数函数
\[
N_S(B):=\card{\langle S\rangle\cap[1,B]}\qquad(B\ge 2),
\]
以及谱指数
\[
\theta(S):=\limsup_{B\to\infty}\frac{\log N_S(B)}{\log B}\in[0,1].
\]
\end{definition}

\begin{proposition}[有限可逆素集合 \texorpdfstring{$S$}{S} 的对数增长]\label{prop:cdim-denominator-growth-finite-S}
若 \(S\) 有限且 \(\card{S}=m\)，则存在常数 \(C(S)>0\) 使对一切 \(B\ge 2\) 有
\[
N_S(B)\ \le\ C(S)\,(\log B)^{m}.
\]
特别地，\(\theta(S)=0\)。
\leanverified{paper\_cdim\_denominator\_growth\_finite\_S}
\leanverified{paper\_cdim\_denominator\_growth\_finite\_s}
\leanverified{N\_S}
\leanverified{exponent\_le\_log}
\leanverified{N\_S\_le\_B}
\leanverified{N\_S\_zero}
\leanverified{N\_S\_mono}
\leanverified{N\_S\_le\_two\_pow\_log}
\end{proposition}
\begin{proof}
写 \(S=\{p_1,\dots,p_m\}\)。
任取 \(n\in\langle S\rangle\cap[1,B]\)，则存在唯一指数向量 \(e=(e_1,\dots,e_m)\in\NN^m\) 使
\(
n=\prod_{i=1}^{m}p_i^{e_i}
\)。
由 \(n\le B\) 得 \(e_i\log p_i\le \log B\)，从而 \(0\le e_i\le \frac{\log B}{\log p_i}\)。
因此
\[
N_S(B)
\le
\prod_{i=1}^{m}\left(1+\left\lfloor\frac{\log B}{\log p_i}\right\rfloor\right)
\le
C(S)\,(\log B)^m,
\]
其中 \(C(S)\) 吸收 \(\prod_i (\log p_i)^{-1}\) 与取整误差。
对 \(\log B\) 取对数并除以 \(\log B\) 得 \(\theta(S)=0\)。证毕。
\leanverified{N\_S\_mono\_set}
\leanverified{N\_S\_empty}
\leanverified{paper\_cdim\_denominator\_growth\_seeds}
\end{proof}

\begin{proposition}[禁素和收敛蕴含正密度]\label{prop:cdim-denominator-positive-density-thin-forbidden}
令 \(Q:=\mathbb{P}\setminus S\) 为禁素集合，并假设
\[
\sum_{q\in Q}\frac{1}{q}\ <\ \infty.
\]
则乘积
\[
c(S):=\prod_{q\in Q}\left(1-\frac{1}{q}\right)
\]
收敛且满足 \(c(S)>0\)。
并且集合 \(\langle S\rangle\) 在 \(\NN\) 中具有下渐近密度：
存在 \(B_0\) 使对一切 \(B\ge B_0\) 有
\[
N_S(B)\ \ge\ \frac{c(S)}{2}\,B.
\]
特别地 \(\theta(S)=1\)。
\leanverified{paper\_cdim\_denominator\_positive\_density\_thin\_forbidden}
\leanverified{multiplesUpTo}
\leanverified{card\_multiplesUpTo}
\leanverified{forbiddenMultiples}
\leanverified{forbiddenMultiples\_eq\_filter}
\leanverified{card\_forbiddenMultiples\_le\_sum}
\leanverified{card\_non\_multiples\_ge}
\leanverified{paper\_cdim\_denominator\_multiples\_seeds}
\end{proposition}
\begin{proof}
由 \(\log(1-x)=-x+O(x^2)\)（\(x\to 0\)）与 \(\sum_{q\in Q}q^{-1}<\infty\) 可知 \(\sum_{q\in Q}\abs{\log(1-1/q)}<\infty\)，故 \(c(S)\) 收敛且 \(c(S)>0\)。
\par
枚举 \(Q=\{q_1<q_2<\cdots\}\) 并令
\(
Q_m:=\{q_1,\dots,q_m\}
\)、
\(
A_m:=\{n\in\NN:\ q\nmid n\ \forall q\in Q_m\}
\)。
则 \(A_m\) 对模 \(L_m:=\prod_{i=1}^m q_i\) 为周期集合，且其精确密度为
\[
\lim_{B\to\infty}\frac{\card{A_m\cap[1,B]}}{B}
=\prod_{i=1}^m\left(1-\frac{1}{q_i}\right)
=:c_m,
\]
其中等式由中国剩余定理下的互素独立性给出。
记 \(A:=\{n\in\NN:\ q\nmid n\ \forall q\in Q\}\)，则 \(A=\bigcap_{m\ge 1}A_m\) 且 \(\card{A\cap[1,B]}=N_S(B)\)。
\par
由于 \(A\subseteq A_m\)，对任意 \(m\) 有
\(
\limsup_{B\to\infty}\card{A\cap[1,B]}/B\le c_m
\)。
令 \(U_{>m}:=\bigcup_{i>m}\{n:\ q_i\mid n\}\)。
则 \(A=A_m\setminus U_{>m}\)，从而
\[
\frac{\card{A\cap[1,B]}}{B}
\ge
\frac{\card{A_m\cap[1,B]}}{B}
\;-\;
\frac{\card{U_{>m}\cap[1,B]}}{B}.
\]
注意 \(\card{\{n\le B:\ q_i\mid n\}}/B\le 1/q_i\)，故
\(
\limsup_{B\to\infty}\card{U_{>m}\cap[1,B]}/B\le \sum_{i>m}1/q_i
\)。
取 \(\liminf_{B\to\infty}\) 得
\[
\liminf_{B\to\infty}\frac{\card{A\cap[1,B]}}{B}
\ge
c_m-\sum_{i>m}\frac{1}{q_i}.
\]
令 \(m\to\infty\) 并用 \(c_m\to c(S)\)、\(\sum_{i>m}1/q_i\to 0\) 得
\(
\liminf_{B\to\infty}\card{A\cap[1,B]}/B\ge c(S)
\)。
合并上下界可知密度存在且等于 \(c(S)\)，于是对充分大 \(B\) 有 \(N_S(B)\ge \frac{c(S)}{2}B\)。
指数结论 \(\theta(S)=1\) 由 \(N_S(B)\ge c(S)B/2\) 直接推出。证毕。
\end{proof}

\begin{proposition}[有界高度有理点的相位分离屏障]\label{prop:cdim-bounded-height-rationals-separation}
对 \(B\ge 2\) 定义既约分数集合
\[
\mathcal Q_B:=\left\{\frac{a}{b}\in\QQ:\ 1\le a\le B,\ 1\le b\le B,\ \gcd(a,b)=1\right\}.
\]
则 \(\card{\mathcal Q_B}\ge B^2/4\)。
并且对任意单射 \(\mathrm{enc}:\mathcal Q_B\hookrightarrow\TT^k\)，其最小分离度满足
\[
\Delta\bigl(\mathrm{enc}(\mathcal Q_B)\bigr)\ \le\ C_k\,B^{-2/k},
\]
其中 \(C_k>0\) 可取为命题 \ref{prop:cdim-torus-packing-bound} 的常数同阶。
特别地，若要求以绝对误差 \(\varepsilon\) 在 \(\mathcal Q_B\) 上唯一解码，则必须 \(\varepsilon\lesssim_k B^{-2/k}\)。
\leanpartial{paper\_cdim\_bounded\_height\_rationals\_separation\_lower\_B}{仅形式化弱下界 $|\mathcal Q_B| \ge B$（用 $b=1$ 列）；完整 $|\mathcal Q_B| \ge B^2/4$ 与 $\mathbb{Q}$ 双射待后轮}
\leanverified{coprimePairs}
\leanverified{firstColumn}
\leanverified{firstColumn\_subset\_coprimePairs}
\leanverified{firstColumn\_card}
\leanverified{coprimePairs\_card\_ge\_B}
\end{proposition}
\begin{proof}
考虑集合
\(
U:=\{(a,b)\in[1,B]^2:\ \gcd(a,b)>1\}
\)。
若 \(\gcd(a,b)>1\)，则存在素数 \(p\) 使 \(p\mid a\) 且 \(p\mid b\)，从而
\[
\card{U}
\le
\sum_{p}\left\lfloor\frac{B}{p}\right\rfloor^2
\le
B^2\sum_{n=2}^{\infty}\frac{1}{n^2}
\le
\frac{3}{4}B^2.
\]
因此互素对数满足
\(
\card{\{(a,b)\in[1,B]^2:\ \gcd(a,b)=1\}}\ge B^2/4
\)。
互素对 \((a,b)\mapsto a/b\) 给出 \(\mathcal Q_B\) 的一一对应（既约表示唯一），故 \(\card{\mathcal Q_B}\ge B^2/4\)。
\par
对有限集 \(E:=\mathrm{enc}(\mathcal Q_B)\subset\TT^k\) 应用命题 \ref{prop:cdim-torus-packing-bound} 得
\[
\Delta(E)\le C_k\,\card{E}^{-1/k}\le C_k\,(B^2/4)^{-1/k}\le C_k' B^{-2/k},
\]
并将常数并入 \(C_k\) 得结论。
若测量误差为 \(\varepsilon\) 且要求唯一解码，则必有 \(\varepsilon\le \Delta(E)/2\)，从而 \(\varepsilon\lesssim_k B^{-2/k}\)。证毕。
\end{proof}
