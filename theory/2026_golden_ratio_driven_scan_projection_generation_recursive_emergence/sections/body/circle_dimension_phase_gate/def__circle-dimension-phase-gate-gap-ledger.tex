\begin{definition}[后继--极限断层的三账本]\label{def:cdim-gap-ledger}

\begin{theorem}[亚临界相位门的有限剩余纤维正规形]\label{distill:gromov-dimension_waist_projection_families-cdim-subcritical-phase-gate-residual-ledger}
设 \(E\) 为非空有限 fold-aware 地址包，\(\mu:E\to\RR_{\ge0}\) 为概率权重，并令 \(A:=\{e\in E:\mu(e)>0\}\)。设
\[
q:E\longrightarrow \Theta
\]
为当前协议已授权的有限相位门投影，且所有可见谓词均通过 \(q\) 下降；这里不把任意连续圆坐标视为有限证书，只使用该 quotient-compatible 的有限读出。记
\[
A_\theta:=A\cap q^{-1}(\theta),\qquad
K:=|q(A)|,\qquad
M_q(A):=\max_{\theta\in q(A)}|A_\theta|.
\]
若一个有限残差寄存器 \(R\) 与映射 \(r:E\to R\) 使
\[
e\longmapsto (q(e),r(e))
\]
在 \(A\) 上单射，则
\[
|R|\ge M_q(A)\ge \left\lceil \frac{|A|}{K}\right\rceil .
\]
特别地，\(q\) 本身在 active 地址上无残差当且仅当 \(M_q(A)=1\)。若 \(K<|A|\)，则存在不可消去的 active 投影纤维，且任何相位门后继账本必须在定义 \ref{def:cdim-gap-ledger} 的补偿账中记录至少 \(M_q(A)\) 个同纤维可分支。若进一步存在 \(\alpha>0\) 使 \(\mu(e)\ge\alpha\) 对所有 \(e\in A\) 成立，则任意单代表相位 lift \(s:q(A)\to A\) 满足 \(q(s(\theta))=\theta\) 时都有
\[
\mu\bigl(A\setminus s(q(A))\bigr)
\ge \alpha\bigl(|A|-K\bigr).
\]
若 \(R=(\mathbb F_p)^B\)，则任意这样的 strictification 还满足
\[
B\ge \left\lceil \log_p M_q(A)\right\rceil .
\]
\end{theorem}

\begin{theorem}[后继相位门的 holonomy--leakage--外置宽度门控]\label{distill:cech_cohomology-holonomy_residue_classification-cdim-gap-holonomy-width-ledger}
固定素数 \(p\)。设
\[
\bigl(K_b,r_{b',b}\bigr)_{b'\ge b\ge b_0}
\]
是一族 fold-aware 的有限协议 nerve 及其允许细化映射。对每个 \(b\)，令 \(A_b=(\FF_p)^{d_b}\)，并设有线性嵌入 \(i_{b,b'}:A_b\hookrightarrow A_{b'}\)。设
\[
h_b\in Z^1(K_b;A_b)
\]
为第 \(b\) 层 reduced overlap 余循环，并令
\[
\mathcal S_b\subset H^1(K_b;A_b)
\]
为该层已登记的 slab/sector leakage 子空间，即由支撑在已标记 slab 或 sector 边上的余循环类张成。假设对 \(b'\ge b\) 有

[i_{b,b'}\,r_{b',b}^{\ast}h_b]-[h_{b'}]\in \mathcal S_{b'},
\qquad
 i_{b,b'}\,r_{b',b}^{\ast}\mathcal S_b\subseteq\mathcal S_{b'},
\]
并且诱导映射
\[
\overline r_{b',b}:
H^1(K_b;A_b)/\mathcal S_b
\longrightarrow
H^1(K_{b'};A_{b'})/\mathcal S_{b'}
\]
在所讨论的 reduced holonomy 类上保持零性。记
\[
\bar\eta_b:=[h_b]\bmod \mathcal S_b.

通过自然同构
\[
H^1(K_b;A_b)\cong H^1(K_b;\FF_p)\otimes_{\FF_p} A_b
\]
定义 reduced holonomy 宽度
\[
w_b:=\min\Bigl\{\dim_{\FF_p}P:\ P\subseteq A_b,
\ \bar\eta_b\in \operatorname{im}\bigl(H^1(K_b;\FF_p)\otimes P\to H^1(K_b;A_b)/\mathcal S_b\bigr)\Bigr\}.
\]
其中 \(w_b=0\) 当且仅当 \(\bar\eta_b=0\)。则在删去有限初始层后，后继相位门的闭路残差按下列优先规则落入恰好一种账本：

  B\ge \sup_b w_b;

\begin{definition}[有限三元 incidence 的平衡投影账]\label{def:distill-cdim-balanced-ternary-incidence-ledger}
设 \(A,B,C\) 为非空有限集合，\(R\subseteq A\times B\times C\) 为有限三元 incidence 关系，并设
\(Z\subseteq R\) 为已登记的例外层。记 \(R^\circ:=R\setminus Z\)。对 \((a,b)\in A\times B\)、
\((a,c)\in A\times C\)、\((b,c)\in B\times C\) 定义
\[
R^\circ_{a,b}:=\{c\in C:(a,b,c)\in R^\circ\},\quad
R^\circ_{a,c}:=\{b\in B:(a,b,c)\in R^\circ\},\quad
R^\circ_{b,c}:=\{a\in A:(a,b,c)\in R^\circ\}.
\]
若 \(R^\circ=\varnothing\)，以下最大值均约定为 \(0\)。置
\[
M_{AB}^{C}:=\max_{(a,b)}|R^\circ_{a,b}|,\qquad
M_{AC}^{B}:=\max_{(a,c)}|R^\circ_{a,c}|,\qquad
M_{BC}^{A}:=\max_{(b,c)}|R^\circ_{b,c}|.
\]
给定整数阈值
\(L_{AB}^{C},L_{AC}^{B},L_{BC}^{A}\ge0\)，称 \(R^\circ\) 通过该平衡投影测试，若
\[
M_{AB}^{C}\le L_{AB}^{C},\qquad
M_{AC}^{B}\le L_{AC}^{B},\qquad
M_{BC}^{A}\le L_{BC}^{A}.
\]
\end{definition}

\begin{proposition}[平衡投影测试的条件性 incidence 审计]\label{prop:distill-cdim-balanced-ternary-incidence-audit}
沿用定义 \ref{def:distill-cdim-balanced-ternary-incidence-ledger}。若 \(R^\circ\) 通过平衡投影测试，则
\[
|R^\circ|\le L_{AB}^{C}\,|\pi_{AB}(R^\circ)|,\qquad
|R^\circ|\le L_{AC}^{B}\,|\pi_{AC}(R^\circ)|,\qquad
|R^\circ|\le L_{BC}^{A}\,|\pi_{BC}(R^\circ)|.
\]
特别地，令
\[
P_A:=\pi_A(R^\circ),\qquad P_B:=\pi_B(R^\circ),\qquad P_C:=\pi_C(R^\circ).
\]
若 \(R^\circ\ne\varnothing\)，则其在最小坐标盒 \(P_A\times P_B\times P_C\) 中的密度满足
\[
\frac{|R^\circ|}{|P_A|\,|P_B|\,|P_C|}
\le
\min\left\{
\frac{L_{AB}^{C}}{|P_C|},
\frac{L_{AC}^{B}}{|P_B|},
\frac{L_{BC}^{A}}{|P_A|}
\right\}.
\]
因此，在给定例外层 \(Z\) 与三个 pair-fibre 阈值后，任一声称的有限 incidence surplus 若超过上述任一已授权投影上界，则必然落入下列可审计失败之一：
\[
M_{AB}^{C}>L_{AB}^{C},\qquad
M_{AC}^{B}>L_{AC}^{B},\qquad
M_{BC}^{A}>L_{BC}^{A},\qquad
Z\ne\varnothing.
\]
该判别只给出有限投影账的必要审计，不把 surplus 本身解释为群结构或算术结构。
\end{proposition}

\begin{theorem}[条件性三元 surplus 的 group-like--例外骨架分解]\label{distill:hrushovski_zilber_zariski_geometries_and_elekes_szabo_group_extraction-exceptional_stratum_dichotomy-balanced-ternary-group-exceptional-decomposition}
沿用定义 \ref{distill:hrushovski_zilber_zariski_geometries_and_elekes_szabo_group_extraction-exceptional_stratum_dichotomy-finite-ternary-exceptional-certificate}。若 \(R^\circ\) 相对于预登记骨架 \(\mathcal S^0\) 满足预算 \(\Lambda\) 的 incidence-to-configuration extraction 与 definable-envelope replacement 条件，则存在有限个两两不交的局部 group-like 包络 \((G_i)_{i\in I}\)、一个 slab 残差 \(S_{\mathrm{slab}}^+\) 与一个预算残差 \(S_{\mathrm{bud}}^+\)，使得
\[
R^\circ
=
S^0\sqcup\Bigl(\bigsqcup_{i\in I}G_i\Bigr)\sqcup S_{\mathrm{slab}}^+\sqcup S_{\mathrm{bud}}^+.
\]
并且满足以下性质：
\begin{enumerate}
  \item 对每个 \(i\in I\)，有
  \[
  M_{AB}^{C}(G_i)\le 1,\qquad M_{AC}^{B}(G_i)\le 1,\qquad M_{BC}^{A}(G_i)\le 1.
  \]
  因而每个 \(G_i\) 的 incidence surplus 已由一个有限局部群律解释，而不是由厚纤维解释。
  \item 存在递减子关系
  \[
  W_0\supseteq W_1\supseteq\cdots\supseteq W_N=S_{\mathrm{bud}}^+,
  \qquad
  W_0:=(R^\circ\setminus S^0)\setminus\bigsqcup_{i\in I}G_i,
  \]
  以及两两不交的纤维块 \(F_j:=W_{j-1}\setminus W_j\)，使得
  \(S_{\mathrm{slab}}^+=\bigsqcup_{j=1}^{N}F_j\)，且每个 \(F_j\) 是 \(W_{j-1}\) 中某个超阈值 pair-fiber：即它等于某个
  \(W_{j-1}\cap(\{a\}\times\{b\}\times C)\)、或
  \(W_{j-1}\cap(\{a\}\times B\times\{c\})\)、或
  \(W_{j-1}\cap(A\times\{b\}\times\{c\})\)，并且其基数分别严格大于
  \(L_{AB}^{C}\)、\(L_{AC}^{B}\)、\(L_{BC}^{A}\)。
  \item \(S_{\mathrm{bud}}^+\) 通过平衡投影测试，且
  \[
  |S_{\mathrm{bud}}^+|\le \Lambda.
  \]
  因而所有没有进入预登记 slab/sticky/Frostman/holonomy/budget 骨架、也没有进入 group-like 包络的剩余 incidence，至多形成有限预算骨架。
\end{enumerate}
此外，若 \(S_{\mathrm{bud}}^+\ne\varnothing\)，\(q:S_{\mathrm{bud}}^+\to\Theta\) 为任一有限相位读出，并记
\[
M_q(S_{\mathrm{bud}}^+):=\max_{\theta\in q(S_{\mathrm{bud}}^+)} |S_{\mathrm{bud}}^+\cap q^{-1}(\theta)|,
\]
则任何映射 \(\rho:S_{\mathrm{bud}}^+\to(\FF_p)^B\) 若能在每个 \(q\)-纤维内分离不同 incidence，即
\[
q(u)=q(v),\ u\ne v\quad\Longrightarrow\quad \rho(u)\ne\rho(v),
\]
必满足
\[
B\ge \left\lceil \log_p M_q(S_{\mathrm{bud}}^+)\right\rceil.
\]
\end{theorem}

\begin{proof}
先证明局部 group-like 包络的纤维单值性。设 \(G_i\) 由有限群 \(H_i\) 与单射
\(\alpha_i,\beta_i,\gamma_i\) 给出。若
\((a,b,c),(a,b,c')\in G_i\)，则
\(\gamma_i(c)=\alpha_i(a)\beta_i(b)=\gamma_i(c')\)，由 \(\gamma_i\) 的单射性得 \(c=c'\)。同理，若固定 \((a,c)\) 或 \((b,c)\)，分别由群中的左消去或右消去以及 \(\beta_i\)、\(\alpha_i\) 的单射性得到相应第三坐标唯一。因此
\(M_{AB}^{C}(G_i),M_{AC}^{B}(G_i),M_{BC}^{A}(G_i)\le 1\)。

由于 \(R^\circ\) 有限，所有包含于 \(R^\circ\setminus S^0\) 且基数大于 \(\Lambda\) 的局部 group-like 包络中，存在按两两不交条件极大的有限族 \((G_i)_{i\in I}\)。令
\[
W_0:=(R^\circ\setminus S^0)\setminus\bigsqcup_{i\in I}G_i.
\]
从 \(W_0\) 开始执行有限剔除过程：若当前 \(W_{j-1}\) 未通过平衡投影测试，则存在一个超阈值 pair-fiber。取其中任意一个作为 \(F_j\)，并令 \(W_j:=W_{j-1}\setminus F_j\)。每一步严格减少当前有限集合的基数，故过程必在有限步后停止。记停止时的关系为 \(W_N\)，并置
\[
S_{\mathrm{slab}}^+:=\bigsqcup_{j=1}^{N}F_j,\qquad S_{\mathrm{bud}}^+:=W_N.
\]
按构造，\(S_{\mathrm{slab}}^+\) 是超阈值 pair-fiber 的不交并，且 \(S_{\mathrm{bud}}^+\) 通过平衡投影测试。

若 \(|S_{\mathrm{bud}}^+|>\Lambda\)，则由 incidence-to-configuration extraction 与 definable-envelope replacement 条件，存在局部 group-like 包络
\(G_*\subseteq S_{\mathrm{bud}}^+\) 且 \(|G_*|>\Lambda\)。但
\(S_{\mathrm{bud}}^+\subseteq W_0\) 与所有 \(G_i\) 不交，又与 \(S^0\) 不交，故可将 \(G_*\) 加入族 \((G_i)_{i\in I}\)，这与该族的极大性矛盾。因此
\(|S_{\mathrm{bud}}^+|\le\Lambda\)。上述构造同时给出所需的不交分解。

最后证明寄存器下界。取 \(\theta\in q(S_{\mathrm{bud}}^+)\) 使
\(|S_{\mathrm{bud}}^+\cap q^{-1}(\theta)|=M_q(S_{\mathrm{bud}}^+)\)。由假设，\(\rho\) 在该有限集合上单射，因此
\[
M_q(S_{\mathrm{bud}}^+)\le |(\FF_p)^B|=p^B.
\]
取以 \(p\) 为底的对数并利用 \(B\in\NN\)，得到
\(B\ge\left\lceil\log_p M_q(S_{\mathrm{bud}}^+)\right\rceil\)。证毕。
\end{proof}

递归地，若在第 \(s\) 步已经选出互异的 envelope 构成集合

\(\Delta_L(R_s\cap E_s)=0\)，矛盾。因此 \(E_s\notin\mathcal S_s\)。取
\[
U_s:=R_s\cap E_s
\]
并令 \(\mathcal S_{s+1}:=\mathcal S_s\cup\{E_s\}\)。这样每一步都严格增加已选 envelope 的个数。

由于 \(\mathcal E\) 有限，递归至多进行 \(|\mathcal E|\) 次。若选尽所有 envelope 后仍有
\(\Delta_L(R_s)>0\)，抽取--替换条件会给出某个
\(E_s=\operatorname{Env}(K_s)\in\mathcal E\) 且
\(\Delta_L(R_s\cap E_s)>0\)。但此时每个 \(E_s\) 已被移除，故
\(R_s\cap E_s=\varnothing\)，矛盾。因此过程必在某一步停止，并得到
\(E_1,\ldots,E_t\) 与 \(U_1,
\ldots,U_t\)，且终止残差 \(W\) 满足
\(\Delta_L(W)=0\)。

最后，标签映射 \(\tau\) 已把每个候选 envelope 分入
\(\mathsf{grp}\) 或 \(\mathsf{exc}\)。选出的每个正 surplus 子账
\(U_j\) 都位于相应 \(E_j\) 内，故它的 algebraic explanation 已被登记为 group-like 或已分类例外。终止残差没有投影维数 surplus，已登记例外层已被先行移除，故不存在未登记的第三类 generic surplus 载体。应用于本文的 fold-aware phase-gate 账本时，\(A,B,C\) 是三种有限地址包，\(X_{\rm exc}\) 汇集 slab、sector、holonomy 与预算骨架，\(\mathcal E\) 是由有限配置账产生的候选 definable envelope；结论正是 repeated cross-window coincidences 必须先被 envelope 化，而不能直接被解释为算术读出。证毕。
\end{proof}

先说明宽度定义与寄存器下界。由于系数为域 \(\FF_p\) 上的有限维向量空间，逐坐标计算上链复形给出自然同构

H^1(K_b;A_b)\cong H^1(K_b;\FF_p)\otimes_{\FF_p}A_b.
\]
因此 \(\bar\eta_b\) 的系数支撑维数可以在 quotient

H^1(K_b;A_b)/\mathcal S_b
\]
中按最小系数子空间 \(P\subseteq A_b\) 定义，这正是 \(w_b\)。特别地，\(w_b=0\) 等价于 \(P=0\) 可行，亦即 \(\bar\eta_b=0\)。这说明 reduced 宽度为零时，所有非零残差若存在，只能已经落在登记的 slab/sector leakage 子空间中。

称秩为 \(B\) 的 \(p\)-寄存器 faithfully 记录第 \(b\) 层 reduced holonomy，是指存在线性映射
\[
\rho_b:A_b\to (\FF_p)^B,\qquad
\sigma_b:(\FF_p)^B\to A_b
\]
使得在 quotient 中
\[
(\mathrm{id}\otimes \sigma_b\rho_b)(\bar\eta_b)=\bar\eta_b.
\]
于是 \(\bar\eta_b\) 落入
\[
H^1(K_b;\FF_p)\otimes \operatorname{im}(\sigma_b)
\]
在 quotient 中的像，而 \(\dim\operatorname{im}(\sigma_b)\le B\)。由 \(w_b\) 的最小性得到
\[
B\ge w_b.
\]
对所有充分深的 \(b\) 取上确界，即得第三项中的统一下界；若 \(\sup_b w_b=\infty\)，则任意固定 \(B\) 都会在某个深层违反该不等式，故固定有限秩寄存器不可能 faithful。

现在证明分类。第一项是 coboundary gauge 情形：\([h_b]=0\) 最终恒成立时，所有足够深层的 reduced overlap cocycle 均可由顶点规范消去。若第一项不成立而 \(\bar\eta_b=0\) 最终恒成立，则足够深层的非零残差在 quotient 后消失，因而全部属于已登记 slab/sector leakage 子空间，这正是第二项。若 \(\bar\eta_b\) 最终非零，则 \(w_b>0\) 最终成立；此时 \(\sup_b w_b\) 或有限或无限。有限时得到第三项及其寄存器下界，无限时得到第四项及无限预算结论。由于这些判断按优先规则依次排除，四类互斥且覆盖所有可能。

最后，细化相容假设保证 \(\bar\eta_b\) 的零性不是某个孤立分辨率的伪不变量：被允许细化拉回后，除已登记的 slab/sector leakage 外，同一 reduced holonomy 类在后继层中保持其账本身份。因此该判别正是定义 \ref{def:cdim-gap-ledger} 中补偿账的 \v Cech 版本：trivial gauge 不进入补偿账，slab/sector leakage 进入已登记几何泄漏账，有限宽度 holonomy 需要相应有限 \(p\)-寄存器，而无界宽度则强制无限外置预算。证毕。
\end{proof}

\begin{proof}
固定 \(\theta\in q(A)\)。若 \((q,r)\) 在 \(A\) 上单射，则其限制在单个纤维 \(A_\theta\) 上仍为单射。由于 \(q\) 在该纤维上恒等于 \(\theta\)，单射性只能由 \(r|_{A_\theta}\) 提供。因此 \(|R|\ge |A_\theta|\)。对所有 \(\theta\in q(A)\) 取最大值得
\[
|R|\ge M_q(A).
\]
另一方面，\(A\) 是 \(K\) 个非空集合 \(A_\theta\) 的不交并，故抽屉原理给出
\[
M_q(A)\ge \left\lceil |A|/K\right\rceil .
\]
于是第一组不等式成立。

若 \(M_q(A)=1\)，每个 active 纤维至多含一个点，故 \(q|_A\) 单射，无需残差寄存器。反之若 \(q|_A\) 单射，则每个 \(A_\theta\) 至多一元，而 \(\theta\in q(A)\) 时非空，故 \(M_q(A)=1\)。因此 \(q\) 本身无残差等价于 \(M_q(A)=1\)。若 \(K<|A|\)，则 \(M_q(A)\ge2\)，所以至少有一个相位投影纤维含有两个 active 地址；这些地址在所有通过 \(q\) 下降的可见谓词下不可区分，只能进入补偿账。

设 \(s:q(A)\to A\) 为单代表 lift。它在每个 active 可见纤维中至多保留一个地址，因此在纤维 \(A_\theta\) 中遗漏 \(|A_\theta|-1\) 个 active 地址。由 \(\mu(e)\ge\alpha\) 得
\[
\mu\bigl(A\setminus s(q(A))\bigr)
=\sum_{\theta\in q(A)}\sum_{e\in A_\theta\setminus\{s(\theta)\}}\mu(e)
\ge \alpha\sum_{\theta\in q(A)}(|A_\theta|-1)
=\alpha(|A|-K).
\]
最后若 \(R=(\mathbb F_p)^B\)，则 \(|R|=p^B\)。结合 \(|R|\ge M_q(A)\) 得 \(p^B\ge M_q(A)\)，等价于所示对 \(B\) 的下界。证毕。
\end{proof}
