\paragraph{高阶校正项与结构刚性：全变差、相对熵与耗散的次主常数}
沿用命题 \ref{prop:cdim-poisson-l1-second-order-constant} 与命题 \ref{prop:cdim-poisson-kl-constant-one-eighth} 的单参设定：
令 \(\nu\) 为 \(\RR\) 上概率测度，\(\bar\gamma:=\int \gamma\,\dd\nu(\gamma)\)，\(\Delta:=\gamma-\bar\gamma\)，并记中心矩
\[
\mu_k:=\EE[\Delta^k]\qquad(k\ge 2),
\qquad
\sigma^2:=\mu_2>0.
\]
对 \(t>0\) 令 \(P_t(x)=\frac{1}{\pi}\frac{t}{x^2+t^2}\)，并定义粗粒化
\[
h_t(x):=\int_{\RR}P_t(x-\gamma)\,\dd\nu(\gamma),
\qquad
g_t(x):=P_t(x-\bar\gamma).
\]

\begin{proposition}[全变差的四次锐渐近：三阶中心矩线性项严格消隐]\label{prop:cdim-poisson-tv-fourth-order-correction}
\leanverified{paper\_cdim\_poisson\_tv\_fourth\_order\_correction}
在上述设定下，若进一步满足 \(\EE|\Delta|^5<\infty\)，则当 \(t\to\infty\) 有二项锐渐近
\begin{equation}\label{eq:cdim-poisson-tv-fourth-order-correction}
\boxed{\;
\|h_t-g_t\|_{L^{1}(\RR)}
=\frac{3\sqrt3}{4\pi}\frac{\sigma^2}{t^2}
\;+\;
\frac{3\sqrt3}{32\pi}\Bigl(\frac{2\mu_3^2}{\sigma^2}-3\mu_4\Bigr)\frac{1}{t^4}
\;+\;o(t^{-4})\; }.
\end{equation}
特别地，\(\|h_t-g_t\|_{L^{1}(\RR)}\) 的 \(t^{-3}\) 级系数恒为 \(0\)：三阶中心矩 \(\mu_3\) 不以线性项进入全变差渐近，首次形状校正出现在 \(t^{-4}\) 层并以 \(\mu_3^2\) 与 \(\mu_4\) 的组合出现。
\end{proposition}
\begin{proof}[证明要点]
对任意固定 \(x\) 以 \(\Delta\) 为变量在 \(\Delta=0\) 处作四阶 Taylor 展开并取期望，得到
\[
h_t(x)-g_t(x)
=\frac{\mu_2}{2}\partial_x^2 g_t(x)-\frac{\mu_3}{6}\partial_x^3 g_t(x)+\frac{\mu_4}{24}\partial_x^4 g_t(x)+R_t(x),
\qquad
\|R_t\|_{L^1(\RR)}=O(t^{-5}),
\]
其中余项由 \(\EE|\Delta|^5\) 与 \(\|\partial_x^5P_t\|_{L^1}\asymp t^{-5}\) 控制。
令 \(y=(x-\bar\gamma)/t\)，并把 \(\partial_x^k g_t\) 写成齐次形式，可将主项归结为
\[
\|h_t-g_t\|_{1}
=\frac{\sigma^2}{\pi}\frac{1}{t^2}\int_{\RR}\left|A(y)+\frac{\alpha}{t}B(y)+\frac{\beta}{t^2}C(y)\right|\dd y+O(t^{-5}),
\]
其中 \(\alpha:=4\mu_3/\sigma^2\)、\(\beta:=\mu_4/\sigma^2\)，并且
\[
A(y)=\frac{3y^2-1}{(1+y^2)^3}\ (\text{偶}),
\qquad
B(y)=\frac{y(y^2-1)}{(1+y^2)^4}\ (\text{奇}),
\qquad
C(y)=\frac{5y^4-10y^2+1}{(1+y^2)^5}\ (\text{偶}).
\]
由于 \(\mathrm{sign}(A)\) 为偶函数而 \(B\) 为奇函数，线性校正项满足
\(
\int_{\RR}B(y)\,\mathrm{sign}(A(y))\dd y=0
\)，故 \(t^{-3}\) 级贡献严格消失。
进一步利用 \(A\) 在 \(\pm 1/\sqrt3\) 处为简单零点，对零点随 \(t^{-1}\) 的位移作隐函数展开，可得
\[
\int_{\RR}\left|A+\frac{\alpha}{t}B+\frac{\beta}{t^2}C\right|
=\int_{\RR}|A|
\;+\;\frac{3\sqrt3}{256}\bigl(\alpha^2-24\beta\bigr)\frac{1}{t^2}
\;+\;o(t^{-2}),
\]
且 \(\int_{\RR}|A(y)|\dd y=\frac{3\sqrt3}{4}\)（命题 \ref{prop:cdim-poisson-kernel-derivative-l1-energy} 中 \eqref{eq:cdim-poisson-pt-second-l1} 的等价表述）。
代回并化简得到 \eqref{eq:cdim-poisson-tv-fourth-order-correction}。
\end{proof}

\begin{proposition}[相对熵的六次校正项：六阶矩组合的完全显式常数]\label{prop:cdim-poisson-kl-sixth-order-correction}
\leanverified{paper\_cdim\_poisson\_kl\_sixth\_order\_correction}
在上述设定下，若进一步满足 \(\EE|\Delta|^5<\infty\)，则当 \(t\to\infty\) 有二项锐渐近
\begin{equation}\label{eq:cdim-poisson-kl-sixth-order-correction}
\boxed{\;
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{\sigma^4}{8t^4}
\;+\;
\frac{\sigma^6+6\mu_3^2-8\sigma^2\mu_4}{64}\cdot\frac{1}{t^6}
\;+\;o(t^{-6})\; }.
\end{equation}
记六次校正系数
\[
C_6(\nu):=\frac{\sigma^6+6\mu_3^2-8\sigma^2\mu_4}{64},
\]
则 \eqref{eq:cdim-poisson-kl-sixth-order-correction} 等价于极限
\[
\lim_{t\to\infty}t^{6}\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{\sigma^4}{8t^4}\right)=C_6(\nu).
\]
特别地，上式为偶阶展开：不存在 \(t^{-5}\) 级项，亦即
\[
\lim_{t\to\infty}t^{5}\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{\sigma^4}{8t^4}\right)=0.
\]
\end{proposition}
\begin{proof}[证明要点]
记 \(r_t:=h_t/g_t=1+\delta_t\)。由四阶展开并除以 \(g_t\) 的齐次形式可得（于 \(g_t\dd x\) 的积分意义下）
\[
\delta_t(x)
=\frac{\sigma^2}{t^2}u_2\!\left(\frac{x-\bar\gamma}{t}\right)
\;+\;\frac{\mu_3}{t^3}u_3\!\left(\frac{x-\bar\gamma}{t}\right)
\;+\;\frac{\mu_4}{t^4}u_4\!\left(\frac{x-\bar\gamma}{t}\right)
\;+\;O(t^{-5}),
\]
其中
\[
u_2(y)=\frac{3y^2-1}{(1+y^2)^2},\qquad
u_3(y)=\frac{4y(y^2-1)}{(1+y^2)^3},\qquad
u_4(y)=\frac{5y^4-10y^2+1}{(1+y^2)^4}.
\]
利用展开
\(
(1+\delta)\log(1+\delta)=\delta+\frac12\delta^2-\frac16\delta^3+O(\delta^4)
\)
以及 \(\int g_t\delta_t=\int(h_t-g_t)=0\)，得到
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac12\int_{\RR} g_t\,\delta_t^2\,\dd x-\frac16\int_{\RR} g_t\,\delta_t^3\,\dd x+O(t^{-8}).
\]
令 \(y=(x-\bar\gamma)/t\)，则 \(g_t(x)\dd x=\frac{1}{\pi(1+y^2)}\dd y\)。
直接计算四个有理积分（均可由 \(y=\tan\theta\) 化为初等积分）：
\[
\int_{\RR}\frac{u_2(y)^2}{\pi(1+y^2)}\dd y=\frac14,\qquad
\int_{\RR}\frac{u_3(y)^2}{\pi(1+y^2)}\dd y=\frac{3}{16},
\]
\[
\int_{\RR}\frac{u_2(y)u_4(y)}{\pi(1+y^2)}\dd y=-\frac18,\qquad
\int_{\RR}\frac{u_2(y)^3}{\pi(1+y^2)}\dd y=-\frac{3}{32}.
\]
代入并收集 \(t^{-4}\)、\(t^{-6}\) 的贡献，得到 \eqref{eq:cdim-poisson-kl-sixth-order-correction}。
\end{proof}

\begin{proposition}[三阶可微 \texorpdfstring{$f$}{f}-散度的六次次主项：由 \texorpdfstring{$f'''(1)$}{f'''} 控制的结构常数]\label{prop:cdim-poisson-fdiv-sixth-order-correction}
\leanverified{paper\_cdim\_poisson\_fdiv\_sixth\_order\_correction}
令 \(f\) 在 \(1\) 的邻域 \(C^3\)，满足 \(f(1)=0\) 且 \(f''(1)>0\)。
在命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 的矩条件下，当 \(t\to\infty\) 有
\begin{equation}\label{eq:cdim-poisson-fdiv-sixth-order-correction}
\boxed{\;
D_f(h_t\mid g_t)
=\frac{f''(1)\sigma^4}{8t^4}
\;+\;
\Bigl[
f''(1)\Bigl(\frac{3\mu_3^2}{32}-\frac{\sigma^2\mu_4}{8}\Bigr)
-\frac{f'''(1)\sigma^6}{64}
\Bigr]\frac{1}{t^6}
\;+\;o(t^{-6})\; }.
\end{equation}
其中 \(\int_{\RR}g_t\bigl(\frac{h_t}{g_t}-1\bigr)=0\) 使得 \(f'(1)\) 的仿射正规化项在散度中严格消失，故六次次主项仅依赖 \(f''(1)\) 与 \(f'''(1)\)。
\end{proposition}
\begin{proof}[证明要点]
仍记 \(r_t=1+\delta_t\) 并沿用命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 中的 \(\delta_t\) 展开。
对 \(f(1+\delta)\) 在 \(\delta=0\) 处作三阶 Taylor 展开并积分，利用 \(\int g_t\delta_t=0\) 消去一次项，得到
\[
D_f(h_t\mid g_t)
=\frac{f''(1)}{2}\int_{\RR} g_t\,\delta_t^2\,\dd x
\;+\;\frac{f'''(1)}{6}\int_{\RR} g_t\,\delta_t^3\,\dd x
\;+\;O(t^{-8}).
\]
将 \(\delta_t\) 的展开代入并使用上一个命题证明中的四个有理积分值，即得 \eqref{eq:cdim-poisson-fdiv-sixth-order-correction}。
\end{proof}

\begin{corollary}[熵耗散率的七次校正项]\label{cor:cdim-poisson-kl-dissipation-seventh-order-correction}
\leanverified{paper\_cdim\_poisson\_kl\_dissipation\_seventh\_order\_correction}
在命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 的假设下，令分数 Fisher--Krein 耗散率由
\[
\frac{\dd}{\dd t}D_{\mathrm{KL}}(h_t\mid g_t)=-\mathcal I_1(h_t\mid g_t)
\]
定义（命题 \ref{prop:xi-poisson-kl-dissipation-fractional-fisher}）。
则当 \(t\to\infty\) 有
\begin{equation}\label{eq:cdim-poisson-dissipation-seventh-order-correction}
\boxed{\;
\mathcal I_1(h_t\mid g_t)
=\frac{\sigma^4}{2t^5}
\;+\;
\frac{3(\sigma^6+6\mu_3^2-8\sigma^2\mu_4)}{32}\cdot\frac{1}{t^7}
\;+\;o(t^{-7})\; }.
\end{equation}
等价地，若记 \(C_6(\nu)\) 如命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 所定义，则
\[
\lim_{t\to\infty}t^{7}\left(\mathcal I_1(h_t\mid g_t)-\frac{\sigma^4}{2t^5}\right)=6C_6(\nu).
\]
并且由命题 \ref{prop:cdim-poisson-kl-sixth-order-coefficient-negative} 得到普适上界
\[
\limsup_{t\to\infty}t^{7}\left(\mathcal I_1(h_t\mid g_t)-\frac{\sigma^4}{2t^5}\right)\le -\frac{21}{32}\sigma^6.
\]
上述不等式取等当且仅当 \(\mu_3=0\) 且 \(\nu\) 为关于均值 \(\bar\gamma\) 的对称二点分布，即
\[
\nu=\tfrac12\delta_{\bar\gamma+\sigma}+\tfrac12\delta_{\bar\gamma-\sigma}.
\]
\end{corollary}
\begin{proof}
由耗散恒等式
\(
\frac{\dd}{\dd t}D_{\mathrm{KL}}(h_t\mid g_t)=-\mathcal I_1(h_t\mid g_t)
\)
与 \eqref{eq:cdim-poisson-kl-sixth-order-correction} 逐项求导，得到 \eqref{eq:cdim-poisson-dissipation-seventh-order-correction}，并因此
\(
t^{7}\bigl(\mathcal I_1(h_t\mid g_t)-\frac{\sigma^4}{2t^5}\bigr)\to 6C_6(\nu)
\)。
由命题 \ref{prop:cdim-poisson-kl-sixth-order-coefficient-negative} 得 \(C_6(\nu)\le -7\sigma^6/64\)，从而给出 \(\limsup\) 上界。
若取等，则必须 \(\mu_3=0\) 且命题 \ref{prop:cdim-poisson-kl-sixth-order-coefficient-negative} 的 Cauchy--Schwarz 取等；
因此 \((\Delta/\sigma)^2=1\) 几乎处处成立，并由 \(\EE[\Delta]=0\) 推得权重相等，从而得到所述对称二点分布。证毕。
\end{proof}

\leanverified{paper\_cdim\_poisson\_kl\_sixth\_order\_coefficient\_negative}
\begin{proposition}[六次系数的严格负性：非退化混合的熵损失必低于主项]\label{prop:cdim-poisson-kl-sixth-order-coefficient-negative}
设 \(\sigma^2=\mu_2>0\) 且 \(\mu_4<\infty\)。则恒有严格不等式
\begin{equation}\label{eq:cdim-poisson-kl-sixth-order-coefficient-negative}
\sigma^6+6\mu_3^2-8\sigma^2\mu_4<0,
\end{equation}
并且更强地
\begin{equation}\label{eq:cdim-poisson-kl-sixth-order-coefficient-upperbound}
\sigma^6+6\mu_3^2-8\sigma^2\mu_4\le -7\sigma^6-2\mu_3^2<0.
\end{equation}
上述上界取等当且仅当 \(\nu\) 的支撑至多两点（等价地：随机变量 \(X:=\Delta\) 满足 \(X^2-\sigma^2\) 与 \(X\) 线性相关）。
从而在固定 \((\bar\gamma,\sigma^2)\) 的类中，命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 的六次系数 \(C_6(\nu)\) 的上确界为 \(-7\sigma^6/64\)，且仅当 \(\nu=\tfrac12\delta_{\bar\gamma+\sigma}+\tfrac12\delta_{\bar\gamma-\sigma}\) 时达到。
\end{proposition}
\begin{proof}
由 Cauchy--Schwarz，不等式
\[
\mu_3^2=\bigl(\EE[X(X^2-\sigma^2)]\bigr)^2
\le \EE[X^2]\cdot \EE[(X^2-\sigma^2)^2]
=\sigma^2(\mu_4-\sigma^4)
\]
成立，从而 \(\mu_4\ge \sigma^4+\mu_3^2/\sigma^2\)。
代入得
\[
\sigma^6+6\mu_3^2-8\sigma^2\mu_4
\le
\sigma^6+6\mu_3^2-8\sigma^2\left(\sigma^4+\frac{\mu_3^2}{\sigma^2}\right)
=-7\sigma^6-2\mu_3^2<0,
\]
从而得到 \eqref{eq:cdim-poisson-kl-sixth-order-coefficient-negative}--\eqref{eq:cdim-poisson-kl-sixth-order-coefficient-upperbound}。
Cauchy--Schwarz 取等当且仅当 \(X(X^2-\sigma^2)\) 与 \(X\) 线性相关，即存在常数 \(c\) 使 \(X^2-\sigma^2=cX\) 几乎处处成立；故 \(X\) 只能取该二次方程的至多两根，等价于 \(\nu\) 支撑至多两点。
特别地，由 \eqref{eq:cdim-poisson-kl-sixth-order-coefficient-upperbound} 可得
\(
C_6(\nu)\le -7\sigma^6/64-\mu_3^2/32\le -7\sigma^6/64
\)。
若在固定 \((\bar\gamma,\sigma^2)\) 的类中取等，则需 \(\mu_3=0\) 且 Cauchy--Schwarz 取等，从而 \(\Delta^2=\sigma^2\) 几乎处处成立并强制两点权重相等，故 \(\nu=\tfrac12\delta_{\bar\gamma+\sigma}+\tfrac12\delta_{\bar\gamma-\sigma}\)。证毕。
\end{proof}

\begin{proposition}[二散度矩层析：由全变差与相对熵的大尺度系数反演 \texorpdfstring{$(\sigma^2,\mu_3^2,\mu_4)$}{(sigma^2,mu3^2,mu4)}]\label{prop:cdim-poisson-two-divergence-moment-tomography}
在命题 \ref{prop:cdim-poisson-tv-fourth-order-correction} 与命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 的条件下，定义尺度归一化观测量
\[
\mathcal B(t):=t^2\|h_t-g_t\|_{L^1(\RR)},
\qquad
\mathcal A(t):=t^4D_{\mathrm{KL}}(h_t\mid g_t).
\]
则存在系数 \(b_2,a_2\) 使得
\[
\mathcal B(t)=\frac{3\sqrt3}{4\pi}\sigma^2+\frac{b_2}{t^2}+o(t^{-2}),
\qquad
\mathcal A(t)=\frac{\sigma^4}{8}+\frac{a_2}{t^2}+o(t^{-2}),
\]
其中
\[
b_2=\frac{3\sqrt3}{32\pi}\Bigl(\frac{2\mu_3^2}{\sigma^2}-3\mu_4\Bigr),
\qquad
a_2=\frac{\sigma^6+6\mu_3^2-8\sigma^2\mu_4}{64}.
\]
因此一旦从曲线 \(\mathcal A,\mathcal B\) 的大尺度二阶系数读出 \((\sigma^2,a_2,b_2)\)，则 \((\mu_3^2,\mu_4)\) 被唯一确定，且具有闭式反演：
令
\[
K:=\frac{32\pi}{3\sqrt3}\,b_2\ (=\,\frac{2\mu_3^2}{\sigma^2}-3\mu_4),
\qquad
L:=64a_2\ (=\,\sigma^6+6\mu_3^2-8\sigma^2\mu_4),
\]
则
\begin{equation}\label{eq:cdim-poisson-two-divergence-moment-tomography-inversion}
\boxed{\;
\mu_4=\frac{L}{\sigma^2}-\sigma^4-3K,
\qquad
\mu_3^2=\frac{\sigma^2}{2}(3\mu_4+K)\; }.
\end{equation}
该“二散度矩层析”仅依赖大尺度二阶系数即可抽取 \((\mu_3^2,\mu_4)\) 的闭式反演；其与命题 \ref{prop:cdim-poisson-central-slice-stieltjes-identifiability} 的中心剖面 Stieltjes 反演互补：后者在全尺度上恢复中心对称化，而此处给出低阶矩组合的快速读出接口。
\end{proposition}
\begin{proof}
前两式分别由 \eqref{eq:cdim-poisson-tv-fourth-order-correction} 与 \eqref{eq:cdim-poisson-kl-sixth-order-correction} 直接读出；反演为代数消元。
\end{proof}

\begin{proposition}[二阶正规形的全局剖面收敛：密度差与比值偏移的统一极限]\label{prop:cdim-poisson-second-order-profile-convergence}
\leanverified{paper\_cdim\_poisson\_second\_order\_profile\_convergence}
在本段的单参 Poisson--Cauchy 粗粒化设定下，设 \(\sigma^2=\mu_2>0\) 且 \(\EE[|\Delta|^4]<\infty\)。
定义两条剖面函数
\[
R(y):=\frac{1}{\pi}\frac{3y^2-1}{(1+y^2)^3},
\qquad
U(y):=\frac{3y^2-1}{(1+y^2)^2}.
\]
则当 \(t\to\infty\) 时成立两类全局一致收敛：
\begin{enumerate}
  \item \textbf{密度差剖面：}
  \[
  \sup_{y\in\RR}\left|
  \frac{t^3}{\sigma^2}\Bigl(h_t(\bar\gamma+ty)-g_t(\bar\gamma+ty)\Bigr)-R(y)
  \right|=O(t^{-1}),
  \]
  并且对任意 \(1\le p\le\infty\) 有
  \[
  \left\|
  \frac{t^3}{\sigma^2}\Bigl(h_t(\bar\gamma+t\cdot)-g_t(\bar\gamma+t\cdot)\Bigr)-R
  \right\|_{L^p(\RR)}=O(t^{-1}).
  \]
  \item \textbf{比值偏移剖面：}由于 \(g_t>0\) 且 \(h_t>0\)，密度比在全空间上良定，并且
  \[
  \sup_{y\in\RR}\left|
  \frac{t^2}{\sigma^2}\left(\frac{h_t}{g_t}(\bar\gamma+ty)-1\right)-U(y)
  \right|=O(t^{-1}).
  \]
  若进一步记 \(G(y):=\frac{1}{\pi(1+y^2)}\)（标准 Cauchy 密度），则对任意 \(1\le p\le\infty\) 亦有加权收敛
  \[
  \left\|
  \frac{t^2}{\sigma^2}\left(\frac{h_t}{g_t}(\bar\gamma+t\cdot)-1\right)-U
  \right\|_{L^p(\RR,G(y)\dd y)}=O(t^{-1}).
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
由定义 \(g_t(x)=P_t(x-\bar\gamma)\) 与 \(h_t(x)=\EE[P_t(x-\bar\gamma-\Delta)]\)。
令 \(y=(x-\bar\gamma)/t\)，则
\[
g_t(\bar\gamma+ty)=\frac{1}{\pi t}\frac{1}{1+y^2}.
\]
对任意 \(y\in\RR\) 与 \(\varepsilon\in\RR\) 记
\[
F(y,\varepsilon):=\frac{P_t(t(y-\varepsilon))}{P_t(ty)}=\frac{1+y^2}{1+(y-\varepsilon)^2}.
\]
则 \(\frac{h_t}{g_t}(\bar\gamma+ty)=\EE[F(y,\Delta/t)]\)。
注意下述有理恒等式与其统一余项界。
由直接代数化简得到
\[
F(y,\varepsilon)
=\frac{1+y^2}{1+(y-\varepsilon)^2}
=1+\frac{2y}{1+y^2}\varepsilon+U(y)\varepsilon^2
-\frac{4y(1-y^2)}{(1+y^2)^2\bigl(1+(y-\varepsilon)^2\bigr)}\,\varepsilon^3
-\frac{3y^2-1}{(1+y^2)^2\bigl(1+(y-\varepsilon)^2\bigr)}\,\varepsilon^4.
\]
由于 \(\bigl(1+(y-\varepsilon)^2\bigr)^{-1}\le 1\) 且
\(
\frac{|y||1-y^2|}{(1+y^2)^2},\frac{|3y^2-1|}{(1+y^2)^2}
\)
在 \(y\in\RR\) 上一致有界，故存在绝对常数 \(C>0\) 使对一切 \(y,\varepsilon\in\RR\) 有
\[
\left|F(y,\varepsilon)-1-\frac{2y}{1+y^2}\varepsilon-U(y)\varepsilon^2\right|
\le C\bigl(|\varepsilon|^3+|\varepsilon|^4\bigr).
\]
代入 \(\varepsilon=\Delta/t\) 并取期望，利用 \(\EE[\Delta]=0\) 得到一致展开
\[
\frac{h_t}{g_t}(\bar\gamma+ty)-1
=\frac{\sigma^2}{t^2}U(y)+O\!\left(\frac{\EE[|\Delta|^3]}{t^3}+\frac{\EE[|\Delta|^4]}{t^4}\right),
\]
其中常数仅依赖于上述 \(C\)。
亦可等价地写为
\[
\frac{h_t}{g_t}(\bar\gamma+ty)-1
=\frac{\EE[\Delta^2]}{2t^2}\,\partial_\varepsilon^2F(y,0)+O(t^{-3})
=\frac{\sigma^2}{t^2}U(y)+O(t^{-3}),
\]
从而第二条的 \(L^\infty\) 收敛成立；加权 \(L^p(\RR,G\dd y)\) 收敛由 \(G\dd y\) 为有限测度与一致界直接推出。
\par
第一条由恒等式
\[
h_t(\bar\gamma+ty)-g_t(\bar\gamma+ty)=g_t(\bar\gamma+ty)\left(\frac{h_t}{g_t}(\bar\gamma+ty)-1\right)
\]
与上式展开给出：
\[
\frac{t^3}{\sigma^2}\bigl(h_t(\bar\gamma+ty)-g_t(\bar\gamma+ty)\bigr)
=\frac{t^2}{\pi\sigma^2}\frac{1}{1+y^2}\left(\frac{h_t}{g_t}(\bar\gamma+ty)-1\right)
=\frac{1}{\pi}\frac{U(y)}{1+y^2}+O\!\left(\frac{1}{t(1+y^2)}\right),
\]
即 \(R(y)\) 的一致逼近。
由于余项被 \(Ct^{-1}(1+y^2)^{-1}\) 控制，且 \((1+y^2)^{-1}\in L^p(\RR)\)（\(1\le p\le\infty\)），故 \(L^p\) 收敛同阶成立。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$L^p$}{Lp} 范数的锐渐近常数：由二阶剖面极限直接读出]\label{cor:cdim-poisson-density-lp-sharp-constants}
在命题 \ref{prop:cdim-poisson-second-order-profile-convergence} 的设定下，对任意 \(1\le p<\infty\) 有极限
\begin{equation}\label{eq:cdim-poisson-density-lp-sharp-constants}
\boxed{\;
\lim_{t\to\infty}\frac{t^{3-1/p}}{\sigma^2}\,\|h_t-g_t\|_{L^p(\RR)}
\;=\;
\|R\|_{L^p(\RR)}
\;=\;
\frac{1}{\pi}\left(\int_{\RR}\frac{|3y^2-1|^p}{(1+y^2)^{3p}}\,\dd y\right)^{1/p}\; }.
\end{equation}
并且 \(p=\infty\) 时与命题 \ref{prop:cdim-poisson-density-linfty-universal-constant} 一致：
\[
\lim_{t\to\infty}\frac{t^3}{\sigma^2}\,\|h_t-g_t\|_{L^\infty(\RR)}=\|R\|_{L^\infty(\RR)}=\frac{1}{\pi}.
\]
特别地，当 \(p=2\) 时有闭式常数
\begin{equation}\label{eq:cdim-poisson-density-l2-sharp-constant}
\boxed{\;
\lim_{t\to\infty}\frac{t^{5/2}}{\sigma^2}\,\|h_t-g_t\|_{L^2(\RR)}
=\frac{\sqrt3}{4\sqrt\pi}\; }.
\end{equation}
\end{corollary}
\begin{proof}
由命题 \ref{prop:cdim-poisson-second-order-profile-convergence} 的 \(L^p(\RR)\) 收敛，
\[
\left\|\frac{t^3}{\sigma^2}\bigl(h_t(\bar\gamma+t\cdot)-g_t(\bar\gamma+t\cdot)\bigr)-R\right\|_{L^p(\RR)}\to 0,
\]
故
\(
\frac{t^3}{\sigma^2}\|h_t(\bar\gamma+t\cdot)-g_t(\bar\gamma+t\cdot)\|_{L^p(\RR)}\to \|R\|_{L^p(\RR)}
\)。
由缩放 \(x=\bar\gamma+ty\) 得
\(
\|h_t(\bar\gamma+t\cdot)-g_t(\bar\gamma+t\cdot)\|_{L^p(\RR)}=t^{-1/p}\|h_t-g_t\|_{L^p(\RR)}
\)，
代回即得 \eqref{eq:cdim-poisson-density-lp-sharp-constants}；\(p=\infty\) 情形同理。
\par
对 \(p=2\)，展开
\[
\|R\|_{L^2(\RR)}^2=\frac{1}{\pi^2}\int_{\RR}\frac{(3y^2-1)^2}{(1+y^2)^6}\,\dd y
=\frac{1}{\pi^2}\int_{\RR}\frac{9y^4-6y^2+1}{(1+y^2)^6}\,\dd y.
\]
利用恒等式（\(n>m+\tfrac12\)）
\(
\int_{\RR}\frac{y^{2m}}{(1+y^2)^{n}}\,\dd y=B\!\left(m+\tfrac12,n-m-\tfrac12\right)
\)
并取 \((n,m)=(6,0),(6,1),(6,2)\) 得
\[
\int_{\RR}\frac{1}{(1+y^2)^6}\,\dd y=\frac{63\pi}{256},\quad
\int_{\RR}\frac{y^2}{(1+y^2)^6}\,\dd y=\frac{7\pi}{256},\quad
\int_{\RR}\frac{y^4}{(1+y^2)^6}\,\dd y=\frac{3\pi}{256}.
\]
代回得
\(
\int_{\RR}\frac{(3y^2-1)^2}{(1+y^2)^6}\dd y=\frac{3\pi}{16}
\)，
从而 \(\|R\|_2=\frac{1}{\pi}\sqrt{\frac{3\pi}{16}}=\frac{\sqrt3}{4\sqrt\pi}\)。
将其代入 \eqref{eq:cdim-poisson-density-lp-sharp-constants} 的 \(p=2\) 情形即得 \eqref{eq:cdim-poisson-density-l2-sharp-constant}。证毕。
\end{proof}
\leanverified{paper\_cdim\_poisson\_density\_lp\_sharp\_constants}

\begin{theorem}[Poisson--Cauchy 粗粒化下 Kolmogorov 距离的普适极限]\label{thm:cdim-poisson-kolmogorov-universal-limit}
\leanverified{paper\_cdim\_poisson\_kolmogorov\_universal\_limit}
在本段的单参 Poisson--Cauchy 粗粒化设定下，令
\[
F_t(x):=\int_{-\infty}^{x}h_t(u)\,\dd u,
\qquad
G_t(x):=\int_{-\infty}^{x}g_t(u)\,\dd u,
\qquad
d_K(t):=\sup_{x\in\RR}\abs{F_t(x)-G_t(x)}.
\]
若 \(\EE[|\Delta|^{3}]<\infty\)，则当 \(t\to\infty\) 有极限
\[
\lim_{t\to\infty}\frac{t^2}{\sigma^2}\,d_K(t)=\frac{9}{16\pi\sqrt3}.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:cdim-poisson-l1-second-order-constant} 的三阶展开与余项控制，可写
\[
h_t(x)-g_t(x)=\frac{\sigma^2}{2}P_t''(x-\bar\gamma)+\mathcal R_t(x),
\qquad
\|\mathcal R_t\|_{L^1(\RR)}=O(t^{-3}).
\]
对 \(x\) 积分得到
\[
F_t(x)-G_t(x)=\frac{\sigma^2}{2}P_t'(x-\bar\gamma)+\int_{-\infty}^{x}\mathcal R_t(u)\,\dd u,
\]
从而
\(
\sup_x\left|\int_{-\infty}^{x}\mathcal R_t(u)\,\dd u\right|\le \|\mathcal R_t\|_{1}=O(t^{-3})
\)。
因此
\[
d_K(t)=\frac{\sigma^2}{2}\,\|P_t'\|_{L^\infty(\RR)}+O(t^{-3}).
\]
直接计算
\[
P_t'(x)=-\frac{2}{\pi}\frac{t\,x}{(x^2+t^2)^2}.
\]
令 \(x=ty\) 得
\(
\abs{P_t'(ty)}=\frac{2}{\pi t^2}\frac{\abs{y}}{(1+y^2)^2}
\)，其最大值在 \(y=1/\sqrt3\) 处取到，并且
\[
\sup_{y\in\RR}\frac{\abs{y}}{(1+y^2)^2}=\frac{9}{16\sqrt3}.
\]
故 \(\|P_t'\|_{\infty}=\frac{9}{8\pi\sqrt3}t^{-2}\)，代回得到
\[
d_K(t)=\frac{9}{16\pi\sqrt3}\frac{\sigma^2}{t^2}+o(t^{-2}),
\]
即所述极限。证毕。
\end{proof}

\begin{proposition}[Poisson--Cauchy 粗粒化下密度差的 \texorpdfstring{$L^\infty$}{L∞} 普适常数]\label{prop:cdim-poisson-density-linfty-universal-constant}
在本段设定下，若 \(\EE[|\Delta|^{3}]<\infty\)，则当 \(t\to\infty\) 有极限
\[
\lim_{t\to\infty}\frac{t^3}{\sigma^2}\,\|h_t-g_t\|_{L^\infty(\RR)}=\frac{1}{\pi}.
\]
\leanverified{paper\_cdim\_poisson\_density\_linfty\_universal\_constant}
\end{proposition}
\begin{proof}
仍由三阶 Taylor 展开（证明见命题 \ref{prop:cdim-poisson-l1-second-order-constant}）可得一致分解
\[
h_t(x)-g_t(x)=\frac{\sigma^2}{2}P_t''(x-\bar\gamma)+O(t^{-4})
\qquad\text{于 }L^\infty(\RR),
\]
其中余项由 \(\EE|\Delta|^3<\infty\) 与 \(\|P_t'''\|_{\infty}\asymp t^{-4}\) 控制。
对 \(P_t''\) 的闭式
\[
P_t''(x)=\frac{2}{\pi}\frac{t(3x^2-t^2)}{(x^2+t^2)^3}
\]
作缩放 \(x=\bar\gamma+ty\)，得到
\[
\frac{t^3}{2}P_t''(\bar\gamma+ty-\bar\gamma)=\frac{1}{\pi}\frac{3y^2-1}{(1+y^2)^3}=R(y),
\]
其中 \(R\) 为命题 \ref{prop:cdim-poisson-second-order-profile-convergence} 的密度差剖面函数。
由于 \(\sup_{y\in\RR}\abs{R(y)}=\abs{R(0)}=1/\pi\)，并且余项在 \(\|\cdot\|_\infty\) 下为 \(o(1)\)，即得所述极限。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\dot H^\alpha$}{Hdot} 频谱能量的锐极限：方差锁定的普适常数]\label{prop:cdim-poisson-sobolev-energy-sharp-limit}
在本段的单参 Poisson--Cauchy 粗粒化设定下，设 \(\sigma^2=\EE[\Delta^2]\in(0,\infty)\) 并令 \(G_t:=h_t-g_t\)。
对任意 \(\alpha\ge 0\) 定义频谱能量
\[
E_\alpha(t):=\int_{\RR}|\xi|^{2\alpha}\,\bigl|\widehat{G_t}(\xi)\bigr|^2\,\dd\xi.
\]
则当 \(t\to\infty\) 有锐极限
\begin{equation}\label{eq:cdim-poisson-sobolev-energy-sharp-limit}
\boxed{\;
\lim_{t\to\infty}\frac{t^{2\alpha+5}}{\sigma^4}\,E_\alpha(t)=\frac{\Gamma(2\alpha+5)}{2^{2\alpha+6}}\; }.
\end{equation}
并且 \(E_\alpha\in C^1((0,\infty))\) 且满足耗散律
\begin{equation}\label{eq:cdim-poisson-sobolev-energy-dissipation}
\boxed{\;
\frac{\dd}{\dd t}E_\alpha(t)=-2\int_{\RR}|\xi|^{2\alpha+1}\,\bigl|\widehat{G_t}(\xi)\bigr|^2\,\dd\xi\ \le\ 0\; }.
\end{equation}
从而进一步有
\[
\lim_{t\to\infty}\frac{t^{2\alpha+6}}{\sigma^4}\left(-\frac{\dd}{\dd t}E_\alpha(t)\right)
=\frac{\Gamma(2\alpha+6)}{2^{2\alpha+6}}.
\]
\end{proposition}
\begin{proof}
由 \(\widehat{P_t}(\xi)=e^{-t|\xi|}\) 得
\[
\widehat{h_t}(\xi)=e^{-t|\xi|}\,\widehat\nu(\xi),
\qquad
\widehat{g_t}(\xi)=e^{-t|\xi|}\,e^{-\mathrm{i}\bar\gamma\xi},
\]
故
\[
\widehat{G_t}(\xi)=e^{-t|\xi|}\,e^{-\mathrm{i}\bar\gamma\xi}\Bigl(\EE[e^{-\mathrm{i}\xi\Delta}]-1\Bigr).
\]
于是
\[
E_\alpha(t)=\int_{\RR}|\xi|^{2\alpha}e^{-2t|\xi|}\Bigl|\EE[e^{-\mathrm{i}\xi\Delta}]-1\Bigr|^2\,\dd\xi.
\]
令 \(\xi=u/t\)，得到
\[
E_\alpha(t)=t^{-(2\alpha+1)}\int_{\RR}|u|^{2\alpha}e^{-2|u|}\Bigl|\EE\!\left[e^{-\mathrm{i}(u/t)\Delta}\right]-1\Bigr|^2\,\dd u.
\]
注意 \(\EE[\Delta]=0\) 且对任意实数 \(a\) 有 \(|e^{-\mathrm{i}a}-1+\mathrm{i}a|\le a^2/2\)。
因此
\[
\left|\EE\!\left[e^{-\mathrm{i}(u/t)\Delta}\right]-1\right|
=\left|\EE\!\left[e^{-\mathrm{i}(u/t)\Delta}-1+\mathrm{i}(u/t)\Delta\right]\right|
\le \frac{u^2}{2t^2}\,\EE[\Delta^2]=\frac{\sigma^2u^2}{2t^2},
\]
从而 \(t^4\bigl|\EE[e^{-\mathrm{i}(u/t)\Delta}]-1\bigr|^2\le \frac{\sigma^4u^4}{4}\)。
另一方面，方差有限等价于特征函数在零点的二阶展开
\[
\EE\!\left[e^{-\mathrm{i}(u/t)\Delta}\right]-1=-\frac{\sigma^2}{2}\frac{u^2}{t^2}+o(t^{-2})\qquad(t\to\infty),
\]
从而 \(t^4\bigl|\EE[e^{-\mathrm{i}(u/t)\Delta}]-1\bigr|^2\to \frac{\sigma^4}{4}u^4\)。
由支配收敛定理得到
\[
E_\alpha(t)
=\frac{\sigma^4}{4}\,t^{-(2\alpha+5)}\int_{\RR}|u|^{2\alpha+4}e^{-2|u|}\,\dd u+o(t^{-(2\alpha+5)}).
\]
最后计算
\(
\int_{\RR}|u|^{2\alpha+4}e^{-2|u|}\dd u
=2\int_0^\infty u^{2\alpha+4}e^{-2u}\dd u
=\Gamma(2\alpha+5)/2^{2\alpha+4}
\)
并代回即得 \eqref{eq:cdim-poisson-sobolev-energy-sharp-limit}。
\par
对耗散律，直接对 \(E_\alpha(t)=\int |\xi|^{2\alpha}e^{-2t|\xi|}\abs{\widehat\nu(\xi)-e^{-\mathrm{i}\bar\gamma\xi}}^{2}\dd\xi\) 在积分号内求导即可得到 \eqref{eq:cdim-poisson-sobolev-energy-dissipation}；
最后一个极限由 \eqref{eq:cdim-poisson-sobolev-energy-sharp-limit} 逐项求导（或等价地用 \(\Gamma(2\alpha+6)=(2\alpha+5)\Gamma(2\alpha+5)\)）得到。证毕。
\end{proof}
