\paragraph{高阶误差校正与熵矩层析的闭式读出}
沿用 \S\ref{subsec:cdim-poisson-cauchy} 与命题 \ref{prop:cdim-poisson-l1-second-order-constant} 的记号：
\[
h_t:=P_t*\nu,\qquad
g_t(x):=P_t(x-\bar\gamma),\qquad
\bar\gamma:=\int \gamma\,\dd\nu(\gamma),\qquad
\sigma^2:=\Var(\nu)\in(0,\infty).
\]
并记中心矩
\[
\mu_3:=\int (\gamma-\bar\gamma)^3\,\dd\nu(\gamma),\qquad
\mu_4:=\int (\gamma-\bar\gamma)^4\,\dd\nu(\gamma).
\]

\begin{theorem}[\texorpdfstring{$L^p$}{Lp} 误差主项常数的全谱极限]\label{thm:cdim-poisson-cauchy-lp-sharp-constants-restated}
对任意 \(p\in(1,\infty)\)，定义
\[
C_p:=\frac1\pi\left(\int_{\RR}\frac{|3y^2-1|^p}{(1+y^2)^{3p}}\,\dd y\right)^{1/p}.
\]
则有
\[
\lim_{t\to\infty}\frac{t^{3-\frac1p}}{\sigma^2}\,\|h_t-g_t\|_{L^p(\RR)}=C_p.
\]
并且
\[
\lim_{t\to\infty}\frac{t^3}{\sigma^2}\,\|h_t-g_t\|_{L^\infty(\RR)}=\frac1\pi.
\]
特别地，
\[
C_2=\sqrt{\frac{3}{16\pi}}.
\]
\leanverified{paper\_cdim\_poisson\_cauchy\_lp\_sharp\_constants\_seeds}
\leanverified{paper\_cdim\_poisson\_cauchy\_lp\_sharp\_constants\_package}
\end{theorem}
\begin{proof}
由定理 \ref{thm:cdim-poisson-cauchy-lp-sharp-constants} 直接得到前两式。
对 \(p=2\)，同一定理已给出
\[
\lim_{t\to\infty}\frac{t^{5/2}}{\sigma^2}\,\|h_t-g_t\|_{L^2(\RR)}=\frac{\sqrt3}{4\sqrt\pi}
\]
故 \(C_2=\sqrt{3/(16\pi)}\)。证毕。
\end{proof}

\begin{theorem}[相对熵二项尖锐渐近与 \texorpdfstring{$t^{-6}$}{t^-6} 系数的矩多项式]\label{thm:cdim-poisson-kl-two-term-sharp-moment-polynomial}
若 \(\EE|\gamma-\bar\gamma|^5<\infty\)，则当 \(t\to\infty\) 有
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=
\frac{\sigma^4}{8t^4}
+\frac{\sigma^6-8\sigma^2\mu_4+6\mu_3^2}{64\,t^6}
+o(t^{-6}).
\]
因此二项系数由 \((\sigma^2,\mu_3,\mu_4)\) 唯一决定。
\leanverified{paper\_cdim\_poisson\_kl\_two\_term\_sharp\_seeds}
\leanverified{paper\_cdim\_poisson\_kl\_two\_term\_sharp\_moment\_polynomial}
\end{theorem}
\begin{proof}
这正是命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 的等价写法。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$t^{-6}$}{t^-6} 系数的普适负性与耗散二项展开]\label{prop:cdim-poisson-kl-sixth-term-negative-and-dissipation-restated}
在定理 \ref{thm:cdim-poisson-kl-two-term-sharp-moment-polynomial} 的条件下，有
\[
\sigma^6-8\sigma^2\mu_4+6\mu_3^2\le -\sigma^6<0.
\]
从而
\[
t^4D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{\sigma^4}{8}
-\frac{|\sigma^6-8\sigma^2\mu_4+6\mu_3^2|}{64\,t^2}
+o(t^{-2}),
\]
即 \(t^4D_{\mathrm{KL}}(h_t\mid g_t)\) 以 \(t^{-2}\) 速率自下方逼近 \(\sigma^4/8\)。

并且相对熵耗散率
\[
\mathcal I_1(h_t\mid g_t):=-\frac{\dd}{\dd t}D_{\mathrm{KL}}(h_t\mid g_t)
\]
满足
\[
\mathcal I_1(h_t\mid g_t)
=\frac{\sigma^4}{2t^5}
+\frac{3(\sigma^6-8\sigma^2\mu_4+6\mu_3^2)}{32\,t^7}
+o(t^{-7}),
\]
其二项同样严格为负。
\leanverified{cdim_poisson_kl_sixth_term_negative_and_dissipation}
\leanverified{paper_cdim_poisson_kl_sixth_term_negative_package}
\end{proposition}
\begin{proof}
由命题 \ref{prop:cdim-poisson-kl-sixth-order-coefficient-negative}，
\[
\sigma^6+6\mu_3^2-8\sigma^2\mu_4
\le -7\sigma^6-2\mu_3^2
\le -\sigma^6<0,
\]
从而得到所述负性与“从下方逼近”展开。

再由推论 \ref{cor:cdim-poisson-kl-dissipation-seventh-order-correction}，
\[
\mathcal I_1(h_t\mid g_t)
=\frac{\sigma^4}{2t^5}
+\frac{3(\sigma^6+6\mu_3^2-8\sigma^2\mu_4)}{32\,t^7}
+o(t^{-7}),
\]
与当前写法完全一致。证毕。
\end{proof}

\begin{theorem}[二阶可实现 Edgeworth 校正与 \texorpdfstring{$t^{-6}$}{t^-6} 级 KL 提升]\label{thm:cdim-poisson-second-order-edgeworth-benchmark}
定义二阶校正基准
\[
g_t^{(2)}(x):=g_t(x)+\sigma^2 r_t(x),
\qquad
r_t(x):=\frac12\,\partial_x^2 g_t(x).
\]
则成立：
\begin{enumerate}
  \item 对任意 \(t>\sigma\)，\(g_t^{(2)}\) 为概率密度（非负且积分为 \(1\)）；
  \item 若 \(\EE|\gamma-\bar\gamma|^4<\infty\)，则
  在 \(\mu_3\neq 0\) 时
  \[
  \lim_{t\to\infty}\frac{t^6}{\mu_3^2}\,D_{\mathrm{KL}}(h_t\mid g_t^{(2)})=\frac{3}{32}.
  \]
  特别地，当 \(\mu_3=0\) 时，
  \[
  D_{\mathrm{KL}}(h_t\mid g_t^{(2)})=O(t^{-8}).
  \]
\end{enumerate}
\leanverified{paper\_cdim\_poisson\_second\_order\_edgeworth\_benchmark}
\end{theorem}
\begin{proof}
由 \(r_t=\frac12\partial_x^2 g_t\) 与 \(g_t=P_t(\cdot-\bar\gamma)\) 的显式式，
令 \(y=(x-\bar\gamma)/t\) 得
\[
\frac{r_t(x)}{g_t(x)}
=\frac{1}{t^2}u(y),\qquad
u(y):=\frac{3y^2-1}{(1+y^2)^2}.
\]
直接求导可得 \(u_{\min}=-1\)（在 \(y=0\) 取到）与 \(u_{\max}=9/16\)。
因此
\[
g_t^{(2)}(x)
=g_t(x)\!\left(1+\frac{\sigma^2}{t^2}u(y)\right)
\ge g_t(x)\!\left(1-\frac{\sigma^2}{t^2}\right)>0
\quad(t>\sigma).
\]
又 \(\int r_t\,\dd x=\frac12\int \partial_x^2 g_t\,\dd x=0\)，故 \(\int g_t^{(2)}=1\)。

再由三阶 Taylor 展开与 \(\EE|\gamma-\bar\gamma|^4<\infty\) 的余项控制，
存在 \(\eta_t\) 使
\[
h_t-g_t^{(2)}
=-\frac{\mu_3}{6}\partial_x^3 g_t+\eta_t,
\qquad
\|\eta_t\|_{L^\infty}=O(t^{-5}),
\qquad
\|\eta_t\|_{L^1}=O(t^{-4}).
\]
于是
\[
\delta_t^{(2)}:=\frac{h_t}{g_t^{(2)}}-1
=\frac{\mu_3}{t^3}v\!\left(\frac{x-\bar\gamma}{t}\right)+O(t^{-4})
\]
一致成立，其中
\[
v(y)=-\frac{4y(1-y^2)}{(1+y^2)^3}.
\]
且 \(\int g_t^{(2)}\delta_t^{(2)}=\int(h_t-g_t^{(2)})=0\)。因此
\[
D_{\mathrm{KL}}(h_t\mid g_t^{(2)})
=\frac12\int g_t^{(2)}(\delta_t^{(2)})^2\,\dd x+o(t^{-6}).
\]
换元 \(y=(x-\bar\gamma)/t\) 并利用 \(g_t^{(2)}=g_t(1+O(t^{-2}))\)，得到
\[
D_{\mathrm{KL}}(h_t\mid g_t^{(2)})
=\frac{\mu_3^2}{2t^6}\int_{\RR}\frac{1}{\pi(1+y^2)}\,v(y)^2\,\dd y+o(t^{-6}).
\]
而该积分值为 \(3/16\)（与命题 \ref{prop:cdim-poisson-kl-sixth-order-correction} 中同型有理积分一致），故
\[
D_{\mathrm{KL}}(h_t\mid g_t^{(2)})
=\frac{3\mu_3^2}{32t^6}+o(t^{-6}).
\]
若 \(\mu_3=0\)，则 \(\delta_t^{(2)}=O(t^{-4})\)，故
\(
D_{\mathrm{KL}}(h_t\mid g_t^{(2)})=O(t^{-8})
\)。
证毕。
\end{proof}

\begin{theorem}[熵矩层析：由两条 KL 剖面重建至四阶中心矩]\label{thm:cdim-poisson-entropy-moment-tomography-up-to-fourth}
在定理 \ref{thm:cdim-poisson-kl-two-term-sharp-moment-polynomial} 与定理 \ref{thm:cdim-poisson-second-order-edgeworth-benchmark} 的条件下，定义
\[
A_4:=\lim_{t\to\infty}t^4D_{\mathrm{KL}}(h_t\mid g_t),
\]
\[
A_6:=\lim_{t\to\infty}t^6\!\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{A_4}{t^4}\right),
\]
\[
B_6:=\lim_{t\to\infty}t^6D_{\mathrm{KL}}(h_t\mid g_t^{(2)}).
\]
则 \((\sigma^2,\mu_3^2,\mu_4)\) 由 \((A_4,A_6,B_6)\) 唯一决定，并满足闭式反演
\[
\sigma^2=\sqrt{8A_4},
\qquad
\mu_3^2=\frac{32}{3}B_6,
\qquad
\mu_4=\frac{\sigma^6+6\mu_3^2-64A_6}{8\sigma^2}.
\]
\leanverified{paper\_cdim\_poisson\_entropy\_moment\_tomography\_seeds}
\leanverified{paper\_cdim\_poisson\_entropy\_moment\_tomography\_package}
\leanverified{paper\_cdim\_poisson\_entropy\_moment\_tomography}
\end{theorem}
\begin{proof}
由定理 \ref{thm:cdim-poisson-kl-two-term-sharp-moment-polynomial}，
\[
A_4=\frac{\sigma^4}{8},
\qquad
A_6=\frac{\sigma^6-8\sigma^2\mu_4+6\mu_3^2}{64}.
\]
由定理 \ref{thm:cdim-poisson-second-order-edgeworth-benchmark}，
\[
B_6=\frac{3\mu_3^2}{32}.
\]
先由 \(A_4\) 得 \(\sigma^2=\sqrt{8A_4}\)，再由 \(B_6\) 得 \(\mu_3^2=\frac{32}{3}B_6\)，最后解 \(A_6\) 的线性方程即得
\[
\mu_4=\frac{\sigma^6+6\mu_3^2-64A_6}{8\sigma^2}.
\]
唯一性随即成立。证毕。
\end{proof}
