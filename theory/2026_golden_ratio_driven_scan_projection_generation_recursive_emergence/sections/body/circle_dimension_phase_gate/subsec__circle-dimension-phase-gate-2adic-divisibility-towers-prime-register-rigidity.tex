\subsection{可除塔的全息分离、Fibonacci/椭圆二幂刚性与单位群圆维无界性}\label{subsec:cdim-2adic-divisibility-towers-prime-register-rigidity}
本小节给出一组与 prime-register 外置轴兼容的二幂可除塔结论。
其核心结构是：加法层保持原循环闭包，而单位层的奇素因子分叉可导致拓扑生成数无界，从而在允许 \(\infty\) 的口径下出现无限 pro-circle dimension。

\begin{definition}[允许 \(\infty\) 的 pro-circle 维数]\label{def:cdim-pcdim-infinity-extension}
对任意 profinite 交换群 \(P\)，定义
\[
\pcdim_{\infty}(P):=
\inf\Bigl\{r\in\NN\cup\{0\}:\ \exists\ \text{连续满射 }\widehat{\ZZ}^{\,r}\twoheadrightarrow P\Bigr\},
\]
并约定当上述集合为空时 \(\pcdim_{\infty}(P)=\infty\)。
\end{definition}

\begin{theorem}[奇数可除塔的全息分离与单位群非有限生成]\label{thm:cdim-odd-divisibility-tower-holographic-separation}
设 \((n_k)_{k\ge 1}\) 为正整数序列，满足 \(n_k\mid n_{k+1}\) 且 \(n_k\ge 2\)。定义
\[
R:=\varprojlim_{k}\ZZ/n_k\ZZ.
\]
则成立：
\begin{enumerate}
  \item 加法群 \(R^{+}\) 为原循环紧交换群；
  \item 若存在子列 \((n_{k_j})_{j\ge 1}\) 使每个 \(n_{k_j}\) 为奇数且 \(\omega(n_{k_j})\to\infty\)，则单位群 \(R^{\times}\) 不是拓扑有限生成；
  \item 在定义 \ref{def:cdim-pcdim-infinity-extension} 的口径下，
\[
  \pcdim_{\infty}(R^{\times})=\infty.
\]
\end{enumerate}
\end{theorem}
\leanverified{paper\_cdim\_odd\_divisibility\_tower\_holographic\_separation}
\begin{proof}
\textbf{(1)} 每个加法层 \(\ZZ/n_k\ZZ\) 为循环群，且由自然投影组成逆系统。元素
\[
\mathbf 1:=(1\bmod n_k)_{k\ge 1}\in R
\]
在每一层都生成 \(\ZZ/n_k\ZZ\)，故其闭生成子群在每一层投影均满，从而在逆极限中稠密；因此 \(R^{+}\) 原循环。
\par
\textbf{(2)} 固定任一奇数层 \(n:=n_{k_j}\)，写
\[
n=\prod_{i=1}^{\omega(n)} p_i^{a_i},
\qquad p_i\ \text{互异奇素数}.
\]
由中国剩余定理，
\[
(\ZZ/n\ZZ)^{\times}\cong\prod_{i=1}^{\omega(n)}(\ZZ/p_i^{a_i}\ZZ)^{\times}.
\]
在每个因子中 \(-1\) 都是阶 \(2\) 元，故
\[
(\ZZ/n\ZZ)^{\times}\supseteq (\ZZ/2\ZZ)^{\omega(n)}.
\]
记有限群最小生成元个数为 \(d(\cdot)\)，则
\[
d\bigl((\ZZ/n\ZZ)^{\times}\bigr)\ge d\bigl((\ZZ/2\ZZ)^{\omega(n)}\bigr)=\omega(n).
\]
若 \(R^{\times}\) 可由 \(r\) 个元素拓扑生成，则其任一有限商
\((\ZZ/n_k\ZZ)^{\times}\) 也至多由 \(r\) 个元素生成，故
\[
d\bigl((\ZZ/n_{k_j}\ZZ)^{\times}\bigr)\le r\quad(\forall j),
\]
与 \(\omega(n_{k_j})\to\infty\) 矛盾。
\par
\textbf{(3)} 由 \(\pcdim_{\infty}\) 的定义，\(\pcdim_{\infty}(R^\times)<\infty\) 蕴含 \(R^\times\) 拓扑有限生成，与第 (2) 条矛盾，故 \(\pcdim_{\infty}(R^\times)=\infty\)。
证毕。
\end{proof}

\begin{corollary}[Fibonacci 二幂可除塔的单位群无限圆维]\label{cor:cdim-fibonacci-2power-adic-unit-infinite}
\leanverified{paper\_cdim\_fibonacci\_2power\_adic\_unit\_infinite}
定义
\[
\ZZ_{\mathrm{Fib},2}:=\varprojlim_{k\to\infty}\ZZ/(F_{2^k}\ZZ).
\]
则 \(\ZZ_{\mathrm{Fib},2}^{+}\) 原循环，且 \(\ZZ_{\mathrm{Fib},2}^{\times}\) 非拓扑有限生成。
进一步，存在常数 \(k_0\) 使对所有 \(k\ge k_0\) 都有
\[
d\!\left((\ZZ/F_{2^k}\ZZ)^{\times}\right)\ge \omega(F_{2^k})\ge k-k_0,
\]
并且
\[
\pcdim_{\infty}\!\left(\ZZ_{\mathrm{Fib},2}^{\times}\right)=\infty.
\]
\end{corollary}
\begin{proof}
可除性 \(F_m\mid F_n\)（当 \(m\mid n\)）给出逆系统合法性，因此加法群原循环由定理
\ref{thm:cdim-odd-divisibility-tower-holographic-separation}(1) 直接得到。
\par
又 \(F_n\) 偶当且仅当 \(3\mid n\)，而 \(3\nmid 2^k\)，故 \(F_{2^k}\) 全为奇数。
由 Fibonacci 序列的原始素因子定理，存在 \(k_0\) 使每个 \(k\ge k_0\) 都有素数
\(q_k\mid F_{2^k}\) 且 \(q_k\nmid F_{2^j}\)（\(j<k\)）。
故这些 \(q_k\) 两两不同，并给出
\[
\omega(F_{2^k})\ge k-k_0.
\]
于是定理 \ref{thm:cdim-odd-divisibility-tower-holographic-separation}(2)(3) 可直接应用，结论成立。
\end{proof}

\begin{definition}[Fibonacci 首次触发指数]\label{def:cdim-fibonacci-rank-of-apparition}
对奇素数 \(p\neq 5\)，定义
\[
r_F(p):=\min\{n\ge 1:\ p\mid F_n\}.
\]
\end{definition}

\begin{theorem}[二幂 Fibonacci 原始素因子的 \(2\)-进同余刚性]\label{thm:cdim-fibonacci-primitive-prime-2adic-congruence}
设 \(p\neq 5\) 为奇素数，且 \(p\) 为 \(F_{2^k}\) 的原始素因子，即
\[
p\mid F_{2^k},
\qquad
p\nmid F_{2^j}\ \ (j<k).
\]
则
\[
r_F(p)=2^k,
\qquad
p\equiv\left(\frac{5}{p}\right)\pmod{2^k},
\]
其中 \(\bigl(\frac{5}{p}\bigr)\) 为 Legendre 符号。
\end{theorem}
\begin{proof}
由 \(p\mid F_{2^k}\) 得 \(r_F(p)\mid 2^k\)，故 \(r_F(p)=2^j\)（某 \(j\le k\)）。
若 \(j<k\)，则按定义 \(p\mid F_{2^j}\)，与原始性矛盾，故 \(r_F(p)=2^k\)。
\par
另一方面，Lucas 判别给出
\[
r_F(p)\mid p-\left(\frac{5}{p}\right).
\]
代入 \(r_F(p)=2^k\) 即得
\[
2^k\mid p-\left(\frac{5}{p}\right),
\]
即所需同余。
证毕。
\end{proof}
\leanverified{paper\_cdim\_fibonacci\_primitive\_prime\_2adic\_congruence}

\begin{corollary}[原始素因子的二进制前缀锁定与柱面归类]\label{cor:cdim-fibonacci-primitive-prime-binary-prefix-lock}
在定理 \ref{thm:cdim-fibonacci-primitive-prime-2adic-congruence} 的条件下，原始素因子 \(p\) 必满足
\[
p\equiv 1\pmod{2^k}
\Longleftrightarrow
\left(\frac{5}{p}\right)=1,
\]
\[
p\equiv -1\pmod{2^k}
\Longleftrightarrow
\left(\frac{5}{p}\right)=-1.
\]
等价地，\(p\) 在 \(\ZZ_2\) 中恰落入两条长度 \(k\) 的柱面之一：
\[
p\in 1+2^k\ZZ_2
\quad\text{或}\quad
p\in (-1)+2^k\ZZ_2.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:cdim-fibonacci-primitive-prime-2adic-congruence} 中
\(
p\equiv (\frac{5}{p})\pmod{2^k}
\)
直接得到。证毕。

\leanverified{binary\_prefix\_zmod\_form}
\end{proof}

\begin{theorem}[Lee--Yang 椭圆二幂可除塔的精确阶素数与单位群圆维无界]\label{thm:cdim-leyang-elliptic-2power-tower-infinite-unit-pcdim}
沿用结论 \ref{con:xi-terminal-zm-elliptic-rational-points-denominator-law} 与
\ref{con:xi-terminal-zm-elliptic-eds-strong-divisibility} 的记号：在积分模型
\[
\widetilde E:\ \widetilde Y^2=\widetilde X^3-16\widetilde X+16
\]
上取 \(\widetilde P=(8,20)\in \widetilde E(\QQ)\)，并写
\[
\widetilde X(n\widetilde P)=\frac{A_n}{B_n^2},
\qquad
A_n,B_n\in\ZZ,\ \gcd(A_n,B_n)=1,\ B_n>0.
\]
定义二幂塔椭圆-adic 环
\[
\ZZ_{\widetilde E,2}:=\varprojlim_k \ZZ/(B_{2^k}\ZZ).
\]
则成立：
\begin{enumerate}
  \item \(B_{2^j}\mid B_{2^k}\)（\(j<k\)）；
  \item 存在 \(k_0\) 使对每个 \(k\ge k_0\) 都存在素数 \(p_k\neq 2,37\) 满足
  \[
  \operatorname{ord}_{\widetilde E(\FF_{p_k})}(\widetilde P)=2^k,
  \]
  因而
  \[
  2^k\mid \#\widetilde E(\FF_{p_k});
  \]
  \item \(\ZZ_{\widetilde E,2}^{\times}\) 非拓扑有限生成，并且
  \[
  \pcdim_{\infty}\!\left(\ZZ_{\widetilde E,2}^{\times}\right)=\infty.
  \]
\end{enumerate}
\leanverified{paper\_cdim\_leyang\_elliptic\_2power\_tower\_infinite\_unit\_pcdim}
\end{theorem}
\begin{proof}
第 (1) 条即结论 \ref{con:xi-terminal-zm-elliptic-eds-strong-divisibility} 的强可除性在 \(n=2^k\) 方向的直接特化。
\par
由结论 \ref{con:xi-terminal-zm-elliptic-eds-primitive-primes-noncm}，存在 \(k_0\) 使得每个 \(k\ge k_0\) 都有原始素因子
\[
p_k\mid B_{2^k},\qquad p_k\nmid B_m\ (m<2^k),
\]
且可取 \(p_k\neq 2,37\)。
再用结论 \ref{con:xi-terminal-zm-elliptic-eds-modp-order-equivalence}：
\[
p_k\mid B_n
\Longleftrightarrow
\operatorname{ord}_{\widetilde E(\FF_{p_k})}(\widetilde P)\mid n.
\]
由 \(p_k\mid B_{2^k}\) 得该阶整除 \(2^k\)；若该阶为 \(2^j<2^k\)，则 \(p_k\mid B_{2^j}\)，与原始性矛盾。
故必有
\[
\operatorname{ord}_{\widetilde E(\FF_{p_k})}(\widetilde P)=2^k.
\]
Lagrange 定理遂给出 \(2^k\mid \#\widetilde E(\FF_{p_k})\)。
\par
最后，\(\{p_k\}_{k\ge k_0}\) 两两不同且均为奇素数，故 \(B_{2^k}\) 的奇素因子个数无界。
与定理 \ref{thm:cdim-odd-divisibility-tower-holographic-separation}(2) 同样的有限商生成元下界可得
\[
d\bigl((\ZZ/B_{2^k}\ZZ)^{\times}\bigr)\to\infty,
\]
从而 \(\ZZ_{\widetilde E,2}^{\times}\) 不可能拓扑有限生成。
按定义 \ref{def:cdim-pcdim-infinity-extension} 即有
\(
\pcdim_{\infty}(\ZZ_{\widetilde E,2}^{\times})=\infty
\)。
证毕。
\end{proof}

\begin{corollary}[二幂椭圆塔单位群在有限素数观测下不可辨识]\label{cor:cdim-elliptic-2power-tower-finite-prime-observation-indistinguishable}
令 \(K:=\ZZ_{\widetilde E,2}^{\times}\)。
则对任意有限素数集 \(S\) 与任意有限观测族
\[
\mathcal O\subseteq \bigcup_{p\in S}\mathrm{Hom}_{\mathrm{cont}}(K,\FF_p),
\]
其共同核 \(\bigcap_{\chi\in\mathcal O}\ker(\chi)\) 为无限闭子群。
特别地，任何仅依赖有限多个素数方向的有限观测字典都无法在 \(K\) 上实现有限歧义的单射辨识。
\end{corollary}
\leanverified{paper\_cdim\_elliptic\_2power\_tower\_finite\_prime\_observation\_indistinguishable}
\begin{proof}
由定理 \ref{thm:cdim-leyang-elliptic-2power-tower-infinite-unit-pcdim}(3) 得 \(\pcdim_{\infty}(K)=\infty\)。
将其代入推论 \ref{cor:conclusion-unbounded-prime-profile-finite-observation-nonidentifiable}（取该处的 \(K\)）即得。证毕。
\end{proof}

\begin{conjecture}[符号与二幂阶的相干极限定律]\label{conj:cdim-fibonacci-elliptic-2adic-coherence-law}
保持上述记号，并作如下两条极限预测：
\begin{enumerate}
  \item 若 \(p_k\) 为 \(F_{2^k}\) 的原始素因子，则二值序列
  \[
  \left(\frac{5}{p_k}\right)\in\{\pm1\}
  \]
  渐近等分布；等价地，推论 \ref{cor:cdim-fibonacci-primitive-prime-binary-prefix-lock}
  中两条 \(2\)-进柱面在原始素因子层面长期均匀占据。
  \item 若 \(p_k\) 为定理 \ref{thm:cdim-leyang-elliptic-2power-tower-infinite-unit-pcdim}
  给出的精确 \(2^k\) 阶素数，则随机变量
  \[
  \nu_2\!\bigl(\#\widetilde E(\FF_{p_k})\bigr)-k
  \]
  形成紧族，并在 \(k\to\infty\) 时趋于一个紧支撑极限分布。
\end{enumerate}
\end{conjecture}
