\subsubsection{循环 lift 的 primitive 轨道与 Dirichlet 梳齿：整除筛选、闭式与平方根放缩}\label{subsubsec:zeta-online-kernel-cyclic-lift-primitives}
本段把附录 \ref{app:real-input-40-finite-rh} 的循环相位提升 \(M^{\langle q\rangle}=M\otimes P_q\)（命题 \ref{prop:finite-rh-phase-lift}）与正文的
\(s\)-平面极点字典（命题 \ref{prop:s-plane-spectrum-poles}）、
以及 primitive 的 Möbius 反演模板（注 \ref{prop:finite-dimensional-explicit-formula}）对齐，
从而得到一组可直接引用的闭式推论：循环 lift 一方面在临界线极点侧实现“\(q\)-梳齿”细分，
另一方面在长度/迹侧引入严格的整除门控，并据此给出 primitive 轨道数与误差的封闭形。

\begin{corollary}[循环 lift 的 primitive 轨道数闭式]\label{cor:zeta-cyclic-lift-primitive-orbits}
设 \(M\) 为有限维矩阵，取整数 \(q\ge 2\)，令 \(P_q\) 为 \(q\)-循环置换矩阵并令 \(M^{\langle q\rangle}:=M\otimes P_q\)（命题 \ref{prop:finite-rh-phase-lift}）。
记迹序列 \(a_n:=\Tr(M^n)\)、\(a_n^{\langle q\rangle}:=\Tr((M^{\langle q\rangle})^n)\)，以及对应的 primitive 轨道数
\[
p_n:=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d},
\qquad
p_n^{\langle q\rangle}:=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a^{\langle q\rangle}_{n/d}
\]
（Witt/Möbius 反演，见注 \ref{prop:finite-dimensional-explicit-formula}）。
则：
\begin{enumerate}
  \item 若 \(q\nmid n\)，则 \(p_n^{\langle q\rangle}=0\)。
  \item 若 \(n=qm\)，则有封闭形
  \[
  \boxed{\;
  p_{qm}^{\langle q\rangle}
  =\frac{1}{m}\sum_{d\mid m}\mu(d)\,\Tr\!\bigl(M^{qm/d}\bigr)\; }.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由推论 \ref{cor:finite-rh-phase-lift-trace-filter} 的整除筛选，
\(
a_n^{\langle q\rangle}=q a_n
\)
当且仅当 \(q\mid n\)，否则 \(a_n^{\langle q\rangle}=0\)。
代入 Möbius 反演得到第一条；当 \(n=qm\) 时，仅有 \(n/d\) 仍被 \(q\) 整除（即 \(d\mid m\)）的项贡献，从而得到第二条闭式。证毕。

\begin{corollary}[Dirichlet 极点的 \(q\)-梳齿化与 RH 不变量]\label{cor:zeta-cyclic-lift-dirichlet-comb}
设 \(M\) 为有限维矩阵，\(\lambda=\rho(M)>1\)，并令
\(
Z_M(s)=\det(I-\lambda^{-s}M)^{-1}
\)
为 Dirichlet 化 \(\zeta\)-接口（命题 \ref{prop:s-plane-spectrum-poles}）。
对循环 lift \(M^{\langle q\rangle}=M\otimes P_q\)，其 Dirichlet 化
\(
Z_{M^{\langle q\rangle}}(s)=\det(I-\lambda^{-s}M^{\langle q\rangle})^{-1}
\)
满足：
\begin{enumerate}
  \item \textbf{（实部不变）} \(Z_{M^{\langle q\rangle}}\) 的全部极点实部集合与 \(Z_M\) 完全相同。
  \item \textbf{（虚部密度放大）} 若 \(M\) 的某个非零特征值为 \(\mu\neq 0\)，则对应的极点列在虚部方向的基频间隔由 \(2\pi/\log\lambda\) 缩小为
  \[
  \boxed{\;
  \frac{2\pi}{q\log\lambda}\; },
  \]
  等价地极点虚部密度严格放大 \(q\) 倍（命题 \ref{prop:finite-rh-phase-lift} 的极点梳齿公式）。
  \item \textbf{（kernel-RH 保持）} 核版 RH/Ramanujan 半径界
  \(
  \max_{\mu\neq\lambda}|\mu|\le \sqrt{\lambda}
  \)
  在循环 lift 下严格不变（命题 \ref{prop:finite-rh-phase-lift}）；其 \(s\)-平面表述即右半开带 \(\tfrac12<\Re(s)<1\) 无极点的性质保持不变（命题 \ref{prop:s-plane-spectrum-poles}）。
\end{enumerate}
\leanverified{paper\_zeta\_cyclic\_lift\_dirichlet\_comb}
\end{corollary}
\begin{proof}
命题 \ref{prop:finite-rh-phase-lift} 给出谱相位细分
\(
\mathrm{spec}(M^{\langle q\rangle})=\{\mu\,\omega_q^k\}
\)
且模长不变；再由命题 \ref{prop:s-plane-spectrum-poles} 的字典
\(
s=(\log\mu+2\pi i\mathbb{Z})/\log\lambda
\)
即可得到 (1)(2)。
(3) 仅涉及模长，故由 \(|\omega_q^k|=1\) 立得。证毕。
\end{proof}

\begin{corollary}[二次原子因子在循环 lift 下的 cyclotomic 分裂]\label{cor:zeta-cyclic-lift-atomic-witt-cyclotomic-splitting}
设 \(M(u)\) 为依赖参数 \(u\) 的有限维矩阵族，并假设在某个系数环中成立分解
\[
\det(I-zM(u))=(1-uz^2)\,Q(z,u).
\]
对任意整数 \(q\ge 2\)，令 \(M(u)^{\langle q\rangle}:=M(u)\otimes P_q\) 为循环 lift。
则 \(\det(I-zM(u)^{\langle q\rangle})\) 必含显式因子
\[
\prod_{k=0}^{q-1}\bigl(1-u(z\omega_q^k)^2\bigr)
=\prod_{k=0}^{q-1}\bigl(1-uz^2\omega_q^{2k}\bigr).
\]
并且该 cyclotomic 乘积可化简为
\[
\prod_{k=0}^{q-1}\bigl(1-uz^2\omega_q^{2k}\bigr)
=
\begin{cases}
1-u^q z^{2q}, & q\ \text{\rm 为奇数},\\[2pt]
\bigl(1-u^{q/2}z^{q}\bigr)^2, & q\ \text{\rm 为偶数}.
\end{cases}
\]
\end{corollary}
\begin{proof}
由 Kronecker 谱公式或命题 \ref{prop:finite-rh-phase-lift-artin}，对任意 \(z\in\CC\) 有
\[
\det\!\bigl(I-zM(u)^{\langle q\rangle}\bigr)
=\prod_{k=0}^{q-1}\det\!\bigl(I-z\omega_q^k M(u)\bigr).
\]
将 \(\det(I-tM(u))=(1-ut^2)Q(t,u)\) 代入 \(t=z\omega_q^k\) 即得首个因子结论。
\par
令 \(x:=uz^2\)。
若 \(q\) 为奇数，则 \(k\mapsto 2k\) 在 \(\ZZ/q\ZZ\) 上可逆，从而
\[
\prod_{k=0}^{q-1}(1-x\omega_q^{2k})
=\prod_{k=0}^{q-1}(1-x\omega_q^{k})
=1-x^q.
\]
若 \(q\) 为偶数，则集合 \(\{\omega_q^{2k}\}_{k=0}^{q-1}\) 枚举全部 \(q/2\) 次单位根且每个出现两次，故
\[
\prod_{k=0}^{q-1}(1-x\omega_q^{2k})
=\left(\prod_{j=0}^{q/2-1}(1-x\omega_{q/2}^{j})\right)^2
=(1-x^{q/2})^2.
\]
代回 \(x=uz^2\) 即得结论。证毕。
\end{proof}

\begin{proposition}[循环 lift 的“素数影子”主项与平方根误差]\label{prop:zeta-cyclic-lift-prime-shadow-asymptotic}
设 \(M\) 为 primitive 非负整数矩阵，\(\lambda=\rho(M)>1\)，并记
\(
\Lambda:=\max_{\mu\neq\lambda}|\mu|
\)。
令 \(p_n^{\langle q\rangle}\) 为推论 \ref{cor:zeta-cyclic-lift-primitive-orbits} 的 primitive 轨道数。
则对任意整数 \(m\ge 1\) 有渐近式
\[
\boxed{\;
p_{qm}^{\langle q\rangle}
=\frac{\lambda^{qm}}{m}
+O\!\left(\frac{\max\{\Lambda^{qm},\,\lambda^{qm/2}\}}{m}\right),\qquad
p_n^{\langle q\rangle}=0\ (q\nmid n)\; }.
\]
特别地，若 kernel-RH 成立（\(\Lambda\le\sqrt{\lambda}\)），则
\[
\boxed{\;
p_{qm}^{\langle q\rangle}
=\frac{\lambda^{qm}}{m}
+O\!\left(\frac{\lambda^{qm/2}}{m}\right).\;}
\]
并且当存在达界谱点 \(|\mu|=\sqrt{\lambda}\) 时，该平方根尺度一般不可再改进（详见附录中关于 RH-sharp 振荡与奇偶门控的命题 \ref{prop:finite-rh-parity-general}）。
\leanverified{paper\_zeta\_cyclic\_lift\_prime\_shadow\_asymptotic\_seeds}
\end{proposition}
\begin{proof}
将推论 \ref{cor:zeta-cyclic-lift-primitive-orbits} 的闭式代入
\(
\Tr(M^n)=\lambda^n+O(\Lambda^n)
\)：
主项来自 \(d=1\) 的 Perron 贡献 \(\lambda^{qm}/m\)。
误差项一部分来自 \(O(\Lambda^{qm/d})\)，其最大量级为 \(O(\Lambda^{qm})\)；
另一部分来自 \(d\ge 2\) 的 Perron 次级项 \(\lambda^{qm/d}\le \lambda^{qm/2}\)。
合并并除以 \(m\) 得到所示上界；RH 情形下 \(\Lambda^{qm}\le \lambda^{qm/2}\) 立得简化式。证毕。
\end{proof}

\begin{corollary}[符号翻转模态与临界线半格点梳齿]\label{cor:zeta-signflip-half-lattice}
设 \(M\) 为有限维矩阵，\(\lambda=\rho(M)>1\)，并令 \(Z_M(s)=\det(I-\lambda^{-s}M)^{-1}\)。
则以下两命题等价：
\begin{enumerate}
  \item \(M\) 存在特征值 \(\mu=-\sqrt{\lambda}\)；
  \item \(Z_M\) 在临界线 \(\Re(s)=\tfrac12\) 上存在半格点极点晶格
  \[
  \boxed{\;
  s=\frac12+\frac{(2k+1)\pi i}{\log\lambda}\qquad(k\in\ZZ)\; }.
  \]
\end{enumerate}
更一般地，若存在达界谱点 \(\mu=\sqrt{\lambda}e^{i\theta}\)，则临界线上的极点列为
\[
\boxed{\;
s=\frac12+\frac{(\theta+2\pi k)i}{\log\lambda}\qquad(k\in\ZZ)\; }.
\]
\leanverified{paper\_zeta\_signflip\_half\_lattice\_seeds}
\end{corollary}
\begin{proof}
由命题 \ref{prop:s-plane-spectrum-poles}，\(\lambda^{-s}\mu=1\) 当且仅当
\(
s=(\log\mu+2\pi i k)/\log\lambda
\)。
当 \(\mu=\sqrt{\lambda}e^{i\theta}\) 时可取 \(\log\mu=\tfrac12\log\lambda+i(\theta+2\pi\ell)\)，代入即得一般式；
取 \(\theta=\pi\) 得半格点情形。
反向推理同理：临界线极点给出 \(|\mu|=\sqrt{\lambda}\) 且辐角为 \(\theta\)（模 \(2\pi\)）。证毕。
\end{proof}

\paragraph{有限探针与分辨率：RH-sharp 梳齿的可寻址性与审计极限}
设 \(M\) 处于 RH-sharp：存在达界谱点 \(\mu=\sqrt{\lambda}e^{i\theta}\)（\(\lambda=\rho(M)>1\)）。
由推论 \ref{cor:zeta-cyclic-lift-dirichlet-comb} 的梳齿公式，对任意 \(q\ge 2\) 的循环 lift
\(
M^{\langle q\rangle}=M\otimes P_q
\)
在临界线 \(\Re(s)=\tfrac12\) 上生成一组 \(q\)-等分虚部梳齿。
将虚部按基本周期 \(2\pi/\log\lambda\) 取模，定义
\[
\mathcal T_q(\theta):=
\left\{\left.\frac{\theta+2\pi j}{q\log\lambda}\ \right|\ j=0,1,\dots,q-1\right\}
\ \subset\ \RR/\Bigl(\frac{2\pi}{\log\lambda}\Bigr)\ZZ .
\]

\begin{theorem}[有限探针规避：任何有限探针皆可被循环 lift 闭包规避]\label{thm:zeta-cyclic-lift-finite-probe-evasion}
给定任意有限集合
\(
S\subset \RR/\bigl(\tfrac{2\pi}{\log\lambda}\bigr)\ZZ
\)。
若非退化对齐情形（即 \(\theta\equiv 0\ (\mathrm{mod}\ 2\pi)\) 且 \(0\in S\)），则存在无穷多个整数 \(q\ge 2\) 使
\[
S\cap \mathcal T_q(\theta)=\varnothing.
\]
因此，任何仅在有限多个临界线采样点上核验 \(\zeta\)-型对象的”有限探针审计”，都不可能在循环 lift 闭包意义下证伪 RH-sharp：在存在达界模态时，总可选取某些 lift 阶 \(q\) 使全部采样点避开梳齿极点。
\leanverified{paper\_zeta\_cyclic\_lift\_finite\_probe\_evasion}
\end{theorem}
\begin{proof}
将商群 \(\RR/(\tfrac{2\pi}{\log\lambda})\ZZ\) 通过
\(
x:=t\log\lambda/(2\pi)\ (\mathrm{mod}\ 1)
\)
同构为圆周 \(\mathbb T:=\RR/\ZZ\)，并记 \(\alpha:=\theta/(2\pi)\in[0,1)\)。
则 \(\mathcal T_q(\theta)\) 对应的相位点集为
\[
\mathcal P_q(\alpha):=\left\{\left.\frac{\alpha+j}{q}\ \right|\ j=0,1,\dots,q-1\right\}\subset\mathbb T,
\]
并且对任意 \(x\in\mathbb T\) 有等价关系
\(
x\in\mathcal P_q(\alpha)\Longleftrightarrow qx-\alpha\in\ZZ
\)。
令 \(X\subset\mathbb T\) 为 \(S\) 的像。欲证结论等价于证明存在无穷多个 \(q\ge 2\) 使 \(X\cap\mathcal P_q(\alpha)=\varnothing\)。
\par
\emph{（1）\(\alpha\) 非有理：}若 \(\alpha\notin\QQ\)，则对任意有理 \(x\) 有 \(qx\in\QQ\) 从而 \(qx-\alpha\notin\ZZ\)（无解）；对任意无理 \(x\)，若 \(q_1x-\alpha,q_2x-\alpha\in\ZZ\) 则 \((q_1-q_2)x\in\ZZ\) 迫使 \(x\in\QQ\)，矛盾，故每个无理 \(x\) 至多排除一个 \(q\)。
于是有限集合 \(X\) 仅排除有限多个 \(q\)，从而存在无穷多 \(q\) 避开。
\par
\emph{（2）\(\alpha\) 有理：}设 \(\alpha=a/b\in\QQ\cap[0,1)\) 为既约分数。
若 \(b=1\)（即 \(\alpha=0\)），则 \(x\in\mathcal P_q(0)\) 等价于 \(qx\in\ZZ\)。
当 \(0\notin X\) 时，\(X\) 中所有可能被命中的点必为有理且分母 \(\ge 2\)，从而每个点只排除某个正整数的倍数集合；取无穷多大素数 \(q\) 即可同时避开。
而当 \(0\in X\) 时，\(0\in\mathcal P_q(0)\) 对所有 \(q\) 成立，故无法避开（对应定理陈述中的退化对齐情形）。
\par
现设 \(b\ge 2\)，并取任意素因子 \(p\mid b\)。
若 \(x=m/n\in X\cap\QQ\) 为既约分数且存在 \(q\) 使 \(qx-\alpha\in\ZZ\)，则必有 \(b\mid n\)；
写 \(n=bd\)（\(d\ge 1\)），由
\(
q\cdot \frac{m}{bd}-\frac{a}{b}\in\ZZ
\)
推出
\(
qm-ad\in bd\,\ZZ
\)，
从而（因 \(\gcd(m,bd)=1\)）得到 \(q\equiv ad\,m^{-1}\ (\mathrm{mod}\ bd)\)。
特别地 \(q=dq'\) 且 \(q'\equiv a m^{-1}\ (\mathrm{mod}\ b)\)，故 \(\gcd(q',b)=1\)。
因此对任意 \(p\mid b\) 都有
\[
v_p(q)=v_p(d)\qquad(x=m/(bd)\ \text{被命中时}).
\]
由于 \(X\) 有限，所有可命中的有理点所对应的 \(v_p(d)\) 取值集合有限。
取 \(N\) 大于该有限集合的最大值并令 \(q:=p^{N}\)；
则 \(v_p(q)=N\) 不可能等于任何可命中的 \(v_p(d)\)，从而 \(X\cap\mathcal P_q(\alpha)=\varnothing\)。
令 \(N\to\infty\) 得到无穷多个这样的 \(q\)。证毕。
\end{proof}

\begin{proposition}[可寻址网格的覆盖半径：一般相位为 \texorpdfstring{$\Theta(Q^{-1})$}{Theta(1/Q)}]\label{prop:zeta-cyclic-lift-addressable-grid-covering-radius}
沿用上文 \(\alpha\in[0,1)\) 与 \(\mathcal P_q(\alpha)\) 的记号，并定义到阶 \(Q\) 的网格并集
\[
\mathcal P_{\le Q}(\alpha):=\bigcup_{q=2}^{Q}\mathcal P_q(\alpha)\subset\mathbb T .
\]
令 \(d_{\mathbb T}\) 为圆周距离，则对任意 \(Q\ge 2\) 有
\[
\boxed{\;
\frac{\min\{\alpha,\,1-\alpha\}}{Q}
\ \le\
\sup_{x\in\mathbb T} d_{\mathbb T}\bigl(x,\mathcal P_{\le Q}(\alpha)\bigr)
\ \le\
\frac{1}{2Q}\; }.
\]
特别地，当 \(\alpha\in(0,1)\) 固定时，上式给出最坏覆盖半径的数量级 \(\Theta(Q^{-1})\)。
\leanverified{paper\_zeta\_addressable\_grid\_covering\_radius\_seeds}
\leanverified{paper\_zeta\_cyclic\_lift\_addressable\_grid\_covering\_radius}
\end{proposition}
\begin{proof}
上界由 \(q=Q\) 的单层梳齿给出：\(\mathcal P_Q(\alpha)\) 为步长 \(1/Q\) 的等距网格，故任意 \(x\) 到 \(\mathcal P_Q(\alpha)\) 的距离 \(\le 1/(2Q)\)，从而到并集 \(\mathcal P_{\le Q}(\alpha)\) 的距离亦 \(\le 1/(2Q)\)。
\par
下界取 \(x=0\) 即得：对每个 \(q\)，\(\mathcal P_q(\alpha)\) 中离 \(0\) 最近的点为 \(\alpha/q\) 或 \(1-(1-\alpha)/q\)，故
\(
d_{\mathbb T}(0,\mathcal P_q(\alpha))=\min\{\alpha,1-\alpha\}/q
\)；
再对 \(q\le Q\) 取最小值即为 \(\min\{\alpha,1-\alpha\}/Q\)。
从而最坏覆盖半径（对 \(x\) 取上确界）不小于该值。证毕。
\end{proof}

\begin{proposition}[点数计数下界：最坏覆盖半径至少为 \texorpdfstring{$\gg Q^{-2}$}{gg Q^-2}]\label{prop:zeta-cyclic-lift-covering-radius-counting-lb}
沿用 \(\mathcal P_{\le Q}(\alpha)\) 的记号。令
\(
N_Q:=\bigl|\mathcal P_{\le Q}(\alpha)\bigr|
\)
为其点数（注意可能有重复点，故仅有上界）。
则对任意 \(Q\ge 2\) 都存在 \(x\in\mathbb T\) 使
\[
\boxed{\;
d_{\mathbb T}\bigl(x,\mathcal P_{\le Q}(\alpha)\bigr)\ \ge\ \frac{1}{Q(Q+1)}\; }.
\]
\leanverified{paper\_zeta\_covering\_radius\_counting\_lb\_seeds}
\end{proposition}
\begin{proof}
显然有点数上界
\[
N_Q\le \sum_{q=2}^{Q} q=\frac{Q(Q+1)}{2}-1.
\]
若以半径 \(r\) 的圆弧覆盖 \(\mathcal P_{\le Q}(\alpha)\) 的全部点，则总覆盖长度至多 \(2r\,N_Q\)。
取 \(r<\frac{1}{2N_Q}\) 时，总长度 \(<1\)，故必存在未覆盖点 \(x\) 满足
\(
d_{\mathbb T}(x,\mathcal P_{\le Q}(\alpha))\ge r
\)。
代入 \(N_Q\le Q(Q+1)/2\) 并取 \(r=\frac{1}{Q(Q+1)}\) 即得结论。证毕。
\end{proof}

\begin{corollary}[预言机分辨率律：由 lift 阶上界 \(Q\) 导出的最坏分辨率尺度]\label{cor:zeta-cyclic-lift-resolution-law}
在 RH-sharp 设定下，定义（按基本周期取模的）最坏分辨率
\[
\varepsilon_Q:=\sup_{t\in \RR/(\tfrac{2\pi}{\log\lambda})\ZZ}\ \inf_{2\le q\le Q}\ \inf_{t_q\in\mathcal T_q(\theta)} d(t,t_q),
\]
其中 \(d\) 为圆周距离（与上文 \(d_{\mathbb T}\) 对齐）。
则有归一化恒等式
\[
\frac{\varepsilon_Q\log\lambda}{2\pi}
=\sup_{x\in\mathbb T} d_{\mathbb T}\bigl(x,\mathcal P_{\le Q}(\alpha)\bigr),
\qquad \alpha=\frac{\theta}{2\pi},
\]
并因此由命题 \ref{prop:zeta-cyclic-lift-addressable-grid-covering-radius} 得到两侧界
\[
\boxed{\;
\frac{\min\{\alpha,\,1-\alpha\}}{Q}
\ \le\
\frac{\varepsilon_Q\log\lambda}{2\pi}
\ \le\
\frac{1}{2Q}\; }.
\]
从而在一般相位 \(\alpha\in(0,1)\) 固定时，要把最坏不确定性压到 \(\varepsilon\) 的量级，任何依赖“循环 lift 到阶 \(Q\)”的机制在资源上都必须满足
\[
Q\ \gtrsim\ \frac{1}{\varepsilon\log\lambda},
\]
且该 \(Q^{-1}\) 标度在数量级上不可改进。
对齐情形 \(\alpha=0\) 的 \(Q^{-2}\) 改进见注 \ref{rem:zeta-cyclic-lift-aligned-farey-q2}。
\leanverified{paper\_zeta\_cyclic\_lift\_resolution\_law\_seeds}
\end{corollary}

\begin{remark}[对齐情形 \texorpdfstring{$\alpha=0$}{alpha=0} 与 \texorpdfstring{$Q^{-2}$}{Q^-2} 改进]\label{rem:zeta-cyclic-lift-aligned-farey-q2}
当 \(\alpha=0\)（等价于 \(\theta\equiv 0\ (\mathrm{mod}\ 2\pi)\)）时，\(\mathcal P_q(0)=\{j/q\}\) 为标准有理网格，其并集 \(\mathcal P_{\le Q}(0)\) 与分母 \(\le Q\) 的 Farey 网格同阶，最坏覆盖半径可改进到 \(\Theta(Q^{-2})\)（经典 Dirichlet/Farey 逼近阶）。
该情形下，资源标度从一般相位的 \(Q\asymp \varepsilon^{-1}\) 改进为 \(Q\asymp \varepsilon^{-1/2}\)（在归一化变量 \(x\) 上）。
\end{remark}
