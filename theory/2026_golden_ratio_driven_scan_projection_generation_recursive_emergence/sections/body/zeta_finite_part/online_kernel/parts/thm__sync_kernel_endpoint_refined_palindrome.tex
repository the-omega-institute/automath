\begin{theorem}[自对偶同步核的端点细化回文律]\label{thm:sync-kernel-endpoint-refined-palindrome}
沿用附录 \ref{app:kernel-self-dual} 的自对偶记号：设存在置换矩阵 \(\Pi\) 与状态 involution \(\iota\) 使得
\[
\Pi^{-1}B_0\Pi=B_1,\qquad \Pi^{-1}B_1\Pi=B_0,
\qquad\text{从而}\qquad
\Pi^{-1}B(u)\Pi=u\,B(1/u),
\]
其中 \(B(u)=B_0+uB_1\)。
对任意 \(n\ge 0\) 与任意状态 \(p,q\) 定义端点细化计数多项式
\[
A^{(n)}_{p,q}(u):=\bigl(B(u)^n\bigr)_{p,q}=\sum_{k=0}^{n}N^{(n)}_{p,q}(k)\,u^{k},
\]
其中 \(N^{(n)}_{p,q}(k)\) 表示长度为 \(n\)、从 \(p\) 到 \(q\) 的路径中输出 \(1\) 的出现次数为 \(k\) 的路径数（在自对偶核的计数口径下等价于对 \(B_1\) 边的计数）。
则成立端点细化回文律
\[
\boxed{\;
A^{(n)}_{p,q}(u)=u^{n}\,A^{(n)}_{\iota(p),\iota(q)}(1/u)\;}
\]
等价地，对一切 \(0\le k\le n\) 有系数对称
\[
\boxed{\;
N^{(n)}_{p,q}(k)=N^{(n)}_{\iota(p),\iota(q)}(n-k).\;}
\]

\leanverified{conj\_pow\_one}
\leanverified{conj\_pow}
\leanverified{pow\_conj}
\leanverified{paper\_sync\_kernel\_endpoint\_refined\_palindrome}
\leanverified{paper\_conclusion\_sync\_kernel\_palindrome\_seeds}
\end{theorem}
\begin{proof}
由相似变换恒等式 \(\Pi^{-1}B(u)\Pi=uB(1/u)\) 取 \(n\) 次幂得到
\[
\Pi^{-1}B(u)^n\Pi=u^n\,B(1/u)^n.
\]
取 \((p,q)\) 元素并用 \(\Pi\) 在状态集合上诱导的置换即 \(\iota\)，可写为
\[
\bigl(B(u)^n\bigr)_{p,q}=u^n\bigl(B(1/u)^n\bigr)_{\iota(p),\iota(q)}.
\]
将两侧按 \(u\) 的幂级数展开并逐项比较系数即得所示回文律与系数对称。证毕。
\end{proof}
