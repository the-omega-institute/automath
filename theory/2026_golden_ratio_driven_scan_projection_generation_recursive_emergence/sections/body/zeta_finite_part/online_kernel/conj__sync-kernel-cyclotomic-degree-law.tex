\subsubsection{cyclotomic elimination/resultant 的次数与整性输出}\label{subsubsec:sync-cyclotomic-elimination-degree-norm-integrality}
沿用命题 \ref{prop:sync-cyclotomic-elimination} 与注 \ref{rem:sync-hatdelta-explicit} 的记号：
对整数 \(m\ge 4\) 令
\[
s_m:=2\cos(\pi/m),\qquad
K_m:=\QQ(s_m),\qquad
R_m(w):=\operatorname{Res}_s\bigl(\widehat\Delta(w,s),\Phi_m^+(s)\bigr)\in\ZZ[w],
\]
其中 \(\Phi_m^+(s)\) 为 \(s_m\) 的最小多项式（整系数、首一）。
本小节给出 \(R_m\) 的次数、对称性、首项范数与根的局部整性约束；其中精确次数律与偶模数偶函数性对应独立子论文定理 1.2 的 cyclotomic 部分
\cite{Internal2026SelfDualSynchronisationKernelCyclotomicTwists}。

\begin{theorem}[cyclotomic elimination/resultant 的精确次数]\label{thm:sync-cyclotomic-elimination-degree-law}
\leanverified{paper\_sync\_cyclotomic\_elimination\_degree\_law}
对任意整数 \(m\ge 4\)，有
\[
\boxed{\;\deg_w R_m=3\varphi(2m)\;}
\]
从而等价地，
\[
\deg_w R_m=
\begin{cases}
6\varphi(m),& m\ \text{为偶数},\\
3\varphi(m),& m\ \text{为奇数}.
\end{cases}
\]
\end{theorem}
\begin{proof}
由于 \(\Phi_m^+\) 首一，结式的根分解公式给出
\[
R_m(w)=\prod_{\sigma:K_m\hookrightarrow\CC}\widehat\Delta\bigl(w,\sigma(s_m)\bigr).
\]
对任意嵌入 \(\sigma\)，\(\widehat\Delta(w,\sigma(s_m))\) 关于 \(w\) 的首项系数为 \(\sigma(s_m)^2-1\)（注 \ref{rem:sync-hatdelta-explicit}）。
若 \(\sigma(s_m)=\pm 1\)，则存在 \((2m)\)-次原始单位根 \(\zeta_{2m}\) 与互素指数 \(a\) 使
\(\sigma(s_m)=\zeta_{2m}^a+\zeta_{2m}^{-a}=\pm 1\)；
从而 \(\zeta_{2m}^a\) 满足 \(x^2\mp x+1=0\)，其阶只能为 \(3\) 或 \(6\)。
但 \(\gcd(a,2m)=1\) 强制 \(\zeta_{2m}^a\) 仍为 \(2m\)-次原始单位根，故仅当 \(2m=6\)（即 \(m=3\)）才可能发生。
因此当 \(m\ge 4\) 时必有 \(\sigma(s_m)\neq \pm 1\)，从而 \(\deg_w\widehat\Delta(w,\sigma(s_m))=6\) 对所有 \(\sigma\) 成立，得到
\[
\deg_w R_m=\sum_{\sigma}6=6\,[K_m:\QQ].
\]
另一方面，\(s_m=\zeta_{2m}+\zeta_{2m}^{-1}\) 生成 \(\QQ(\zeta_{2m})\) 的实子域，
且复共轭在 \(\QQ(\zeta_{2m})\) 上的固定子域即为 \(K_m\)（\(m\ge 4\) 时指数为 \(2\)）。
于是
\(
[K_m:\QQ]=\varphi(2m)/2
\)。
合并得 \(\deg_w R_m=6\cdot \varphi(2m)/2=3\varphi(2m)\)。
最后利用 \(\varphi(2m)=\varphi(m)\)（\(m\) 奇）与 \(\varphi(2m)=2\varphi(m)\)（\(m\) 偶）得到分段公式。证毕。
\end{proof}

\begin{proposition}[偶模数下的偶函数性与降维次数]\label{prop:sync-cyclotomic-elimination-evenness}
设 \(m\ge 4\) 为偶数，则
\[
\boxed{\;R_m(w)=R_m(-w)\;}
\qquad\text{从而}\qquad
R_m(w)\in\ZZ[w^2],
\]
并且视 \(R_m\) 为 \(w^2\) 的多项式时有
\[
\boxed{\;\deg_{w^2}R_m=3\varphi(m)\; }.
\]
\leanverified{paper\_sync\_cyclotomic\_elimination\_evenness}
\end{proposition}
\begin{proof}
由注 \ref{rem:sync-hatdelta-explicit} 的对称性 \(\widehat\Delta(w,-s)=\widehat\Delta(-w,s)\) 可得 \(\widehat\Delta(-w,-s)=\widehat\Delta(w,s)\)。
当 \(m\) 为偶数时，\(\gcd(m+1,2m)=1\)，故自同态 \(\zeta_{2m}\mapsto \zeta_{2m}^{m+1}=-\zeta_{2m}\) 诱导 \(K_m\) 的一个 \(\QQ\)-自同构并把 \(s_m=\zeta_{2m}+\zeta_{2m}^{-1}\) 送到 \(-s_m\)。
因此 \(\{\sigma(s_m)\}_{\sigma:K_m\hookrightarrow\CC}\) 在变换 \(s\mapsto -s\) 下封闭。
由结式的根分解公式，
\[
R_m(-w)=\prod_{\sigma}\widehat\Delta\bigl(-w,\sigma(s_m)\bigr)
=\prod_{\sigma}\widehat\Delta\bigl(w,-\sigma(s_m)\bigr)
=\prod_{\sigma}\widehat\Delta\bigl(w,\sigma(s_m)\bigr)=R_m(w),
\]
从而 \(R_m\) 为偶函数，故 \(R_m(w)\in\ZZ[w^2]\)。
由定理 \ref{thm:sync-cyclotomic-elimination-degree-law} 对偶数 \(m\) 有 \(\deg_w R_m=6\varphi(m)\)，故 \(\deg_{w^2}R_m=\deg_w R_m/2=3\varphi(m)\)。证毕。
\end{proof}

\begin{proposition}[首项系数的范数公式与唯一素数异常]\label{prop:sync-cyclotomic-elimination-leading-coefficient-norm}
设 \(m\ge 4\)，并令 \(\operatorname{LC}(R_m)\) 表示 \(R_m(w)\) 的首项系数。
则：
\begin{enumerate}
  \item \(R_m(0)=1\)；
  \item \(\operatorname{LC}(R_m)=N_{K_m/\QQ}(s_m^2-1)\)；
  \item 记 \(\varepsilon(m):=1\)（\(m\) 奇）或 \(\varepsilon(m):=2\)（\(m\) 偶），则
  \[
  \boxed{\;
  \bigl|\operatorname{LC}(R_m)\bigr|=
  \begin{cases}
  p^{\varepsilon(m)},& m=3p^k\ \text{对某个素数 }p\text{与 }k\ge 1,\\
  1,& m\neq 3p^k,
  \end{cases}\qquad (m\ge 4)\;}
  \]
  因而当 \(m\) 为偶数时，\(\bigl|\operatorname{LC}(R_m)\bigr|\neq 1\) 的唯一可能为 \(m=3\cdot 2^k\)。
\end{enumerate}
\end{proposition}
\begin{proof}
由根分解公式 \(R_m(w)=\prod_{\sigma}\widehat\Delta(w,\sigma(s_m))\)，代入 \(w=0\) 并注意 \(\widehat\Delta(0,s)=1\)，得到 \(R_m(0)=1\)。
\par
同一分解下，\(\widehat\Delta(w,s)\) 的 \(w^6\) 系数为 \(s^2-1\)，故
\[
\operatorname{LC}(R_m)=\prod_{\sigma}\bigl(\sigma(s_m)^2-1\bigr)=N_{K_m/\QQ}(s_m^2-1).
\]
\par
令 \(\zeta_{2m}:=e^{\pi i/m}\) 并置 \(\zeta_m:=\zeta_{2m}^2=e^{2\pi i/m}\)。
则
\(
s_m^2-1=\zeta_m+\zeta_m^{-1}+1
\)
落在 \(\QQ(\zeta_m)^+\) 中，并且
\[
\zeta_m+\zeta_m^{-1}+1=\zeta_m^{-1}(1+\zeta_m+\zeta_m^2)
=\zeta_m^{-1}\frac{1-\zeta_m^3}{1-\zeta_m}.
\]
因此在 \(\QQ(\zeta_m)\) 上的绝对范数满足
\[
\Bigl|N_{\QQ(\zeta_m)/\QQ}(\zeta_m+\zeta_m^{-1}+1)\Bigr|
=\frac{\bigl|N_{\QQ(\zeta_m)/\QQ}(1-\zeta_m^3)\bigr|}{\bigl|N_{\QQ(\zeta_m)/\QQ}(1-\zeta_m)\bigr|}.
\]
记 \(d:=m/\gcd(m,3)\)。则 \(\zeta_m^3\) 为原始 \(d\)-次单位根，且 \(1-\zeta_m^3\in\QQ(\zeta_d)\)。
由塔式范数与恒等式 \(\,N_{\QQ(\zeta_n)/\QQ}(1-\zeta_n)=\Phi_n(1)\,\) 得
\[
\Bigl|N_{\QQ(\zeta_m)/\QQ}(\zeta_m+\zeta_m^{-1}+1)\Bigr|
=\frac{\Phi_d(1)^{\varphi(m)/\varphi(d)}}{\Phi_m(1)}.
\]
由 Möbius 反演公式 \(\Phi_n(x)=\prod_{d\mid n}(x^d-1)^{\mu(n/d)}\)，令 \(x\to 1\) 并用 \(x^d-1\sim d(x-1)\) 得
\[
\Phi_n(1)=\prod_{d\mid n} d^{\mu(n/d)}.
\]
由该积式读取 \(p\)-进赋值可得：当且仅当 \(n\) 为素数幂 \(p^k\) 时 \(\Phi_n(1)=p\)，否则 \(\Phi_n(1)=1\)。
从而若 \(3\nmid m\) 则 \(d=m\) 且上式比值为 \(1\)；
若 \(3\mid m\) 则 \(d=m/3\)，上式非平凡当且仅当 \(m/3\) 为素数幂 \(p^k\)，并在该情形下得到
\(
\bigl|N_{\QQ(\zeta_m)/\QQ}(\zeta_m+\zeta_m^{-1}+1)\bigr|=p^2
\)。
\par
当 \(m\) 为奇数时，\(K_m=\QQ(\zeta_m)^+\) 且 \(\QQ(\zeta_m)\) 为其二次 CM 扩张，故对实子域元有
\(
\bigl|N_{K_m/\QQ}(s_m^2-1)\bigr|
=\sqrt{\bigl|N_{\QQ(\zeta_m)/\QQ}(s_m^2-1)\bigr|}
\)；
当 \(m\) 为偶数时，\(K_m\) 为 \(\QQ(\zeta_m)^+\) 的二次扩张而 \(s_m^2-1\in\QQ(\zeta_m)^+\)，故
\(
\bigl|N_{K_m/\QQ}(s_m^2-1)\bigr|
=\bigl|N_{\QQ(\zeta_m)/\QQ}(s_m^2-1)\bigr|
\)。
结合上一步 \(p^2/1\) 的判别，即得所述 \(p^{\varepsilon(m)}\) 分类。证毕。
\end{proof}
\leanverified{paper\_sync\_cyclotomic\_elimination\_leading\_coefficient\_norm}

\begin{corollary}[根的 \texorpdfstring{$S$}{S}-整性与分母素数唯一性]\label{cor:sync-cyclotomic-elimination-root-sintegrality}
设 \(m\ge 4\)，并令 \(S_m\) 为所有整除 \(\operatorname{LC}(R_m)\) 的有理素数集合。
则对任意代数数 \(w_0\) 满足 \(R_m(w_0)=0\)，\(w_0\) 在一切 \(\ell\notin S_m\) 处为 \(\ell\)-进整，从而为 \(S_m\)-整数。
特别地，由命题 \ref{prop:sync-cyclotomic-elimination-leading-coefficient-norm} 得：
\begin{enumerate}
  \item 若 \(m\neq 3p^k\)，则 \(S_m=\varnothing\)，从而 \(R_m\) 的每个根均为代数整数；
  \item 若 \(m=3p^k\)，则 \(S_m=\{p\}\)（且若 \(m\) 为偶数必有 \(p=2\)），从而根的分母素因子（若存在）只能来自唯一素数 \(p\)。
\end{enumerate}
\end{corollary}
\begin{proof}
取素数 \(\ell\notin S_m\)。则 \(\operatorname{LC}(R_m)\) 在局部化环 \(\ZZ_{(\ell)}\) 中可逆；
把方程 \(R_m(w_0)=0\) 以 \(\operatorname{LC}(R_m)\) 归一化，可得一首一多项式，其系数落在 \(\ZZ_{(\ell)}\)。
因此 \(w_0\) 在 \(\ell\)-进意义下整。
由任意 \(\ell\notin S_m\) 的局部整性即得 \(S_m\)-整性结论；其余断言由命题 \ref{prop:sync-cyclotomic-elimination-leading-coefficient-norm} 直接读出。证毕。
\end{proof}
\leanverified{paper\_sync\_cyclotomic\_elimination\_root\_sintegrality}
