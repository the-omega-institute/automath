\label{subsec:zeta-online-kernel-part2}

使得对所有可逆 $r$ 都有恒等式
\[
\widehat{\Delta}(w,r+r^{-1})=\Delta(w/r,r^2).
\]

R_m(w):=\operatorname{Res}_s\bigl(\widehat{\Delta}(w,s),\Phi_m^+(s)\bigr)\in\ZZ[w].

\begin{remark}[completion 的不变量子环：乘法扭曲 \texorpdfstring{$u$}{u} 压缩为加法坐标 \texorpdfstring{$s$}{s}]\label{rem:completion-invariant-ring}
在变量层面，完成化所用的替换
\(
u=r^2,\ s=r+r^{-1}
\)
可理解为对合 \(r\mapsto r^{-1}\) 的不变量化：Laurent 环的固定子环满足
\[
\ZZ[r,r^{-1}]^{\,r\mapsto r^{-1}}
=\ZZ\!\left[r+r^{-1}\right]
=\ZZ[s].
\]
因此在自对偶对称（\(u\leftrightarrow 1/u\)）下，root-of-unity 的乘法扭曲参数 \(u\in\CC^\times\) 经 completion 被压缩为一个实的“迹坐标” \(s=2\cos(\pi/m)\)，而 \(\widehat\Delta(w,s)\in\ZZ[s][w]\) 则提供了把“乘法侧扭曲谱”转写为“加法侧代数证书”的统一接口。
\end{remark}

\begin{remark}[同步核的完成化行列式显式式与 \texorpdfstring{$s\mapsto -s$}{s->-s} 对称]\label{rem:sync-hatdelta-explicit}
对同步核的输出位势加权，附录 \ref{app:pressure-analytic} 给出非零谱的特征多项式 $F(\lambda,u)\in\ZZ[u][\lambda]$，并且
\[
\Delta(z,u)=z^6F(1/z,u)
\]
（等价地，$B(u)$ 仅有 $6$ 个非零特征值）。取完成化变量 $u=r^2,\ w=zr,\ s=r+r^{-1}$，则命题 \ref{prop:sync-cyclotomic-elimination} 中的完成化行列式可写成以下\textbf{整系数闭式}：
\[
\boxed{\;
\widehat\Delta(w,s)=
1-sw-5w^2+3sw^3+(5-s^2)w^4+(s^3-6s)w^5+(s^2-1)w^6.
\;}
\]
并且直接代入可验其满足
\[
\widehat\Delta(w,-s)=\widehat\Delta(-w,s).
\]
因此当 $s$ 变号时，$w$-根集合在 $w\mapsto -w$ 下对应，零点模长不变；从而由 $\min\{|w|:\widehat\Delta(w,s)=0\}$ 所定义的“最小模零点”仅依赖于 $|s|$。
\end{remark}

\begin{corollary}[端点 \(u=1\)（\(s=2\)）的闭式因式分解与最小模根]\label{cor:sync-hatdelta-s2-factorization}
取 \(s=2\)，则
\[
\widehat\Delta(w,2)
=1-2w-5w^2+6w^3+w^4-4w^5+3w^6
=(w-1)(w+1)(3w-1)(w^3-w^2+w+1).
\]
因此方程 \(\widehat\Delta(w,2)=0\) 的根中，满足 \(|w|\) 最小者为
\[
\boxed{\ w_\star(2)=\frac13\ },\qquad \rho(2)=\frac{1}{|w_\star(2)|}=3.
\]
\leanverified{paper\_sync\_hatdelta\_s2\_factorization}
\end{corollary}
\begin{proof}
直接代入并因式分解得所示分解。
为说明最小模根确为 \(1/3\)，注意当 \(|w|<1/3\) 时
\[
\bigl|w^3-w^2+w\bigr|
\le |w|+|w|^2+|w|^3
<\frac13+\frac19+\frac1{27}
=\frac{13}{27}<1,
\]
从而 \(w^3-w^2+w+1\neq 0\)。又因 \((w\pm 1)=0\) 的零点模长为 \(1\)，故 \(|w|\) 的最小可能值只能来自因子 \((3w-1)\)，即 \(w_\star(2)=1/3\)。
于是 \(\rho(2)=1/|w_\star(2)|=3\)。证毕。
\end{proof}

\begin{proposition}[端点导数的扩散字典：\texorpdfstring{$\Lambda'(2)$}{Lambda'(2)} 锁定单位圆扭曲的二次耗散与 near-\texorpdfstring{$1$}{1} 主奇点轨迹]\label{prop:completion-endpoint-jet-diffusion}
假设某个完成化/单位圆扭曲通道满足严格相位--幅度分离：存在在 \(s=2\) 的某邻域解析的实函数 \(g(s)=\log\Lambda(s)\)（其中 \(\Lambda(2)>1\)）使得对 \(u=e^{it}\) 有
\[
P(it)=\log\lambda(e^{it})=\frac{it}{2}+g\!\left(S(t)\right),
\qquad
S(t):=2\cos(t/2).
\]
记 \(\lambda:=\Lambda(2)\)，并定义单位圆谱比
\[
\eta(t):=\frac{|\lambda(e^{it})|}{\lambda}
=\exp\!\bigl(\Re P(it)-\log\lambda\bigr)
=\frac{\Lambda(S(t))}{\Lambda(2)}.
\]
则当 \(t\to 0\) 时有端点 jet 展开
\[
\boxed{\;
\log\eta(t)
=-\frac{g'(2)}{4}\,t^{2}
+\left(\frac{g'(2)}{192}+\frac{g''(2)}{32}\right)t^{4}
+O(t^{6})\;}
\]
从而二次“扩散方差”与耗散常数由端点导数闭式给出
\[
\boxed{\;
\Re P(it)=\log\lambda-\frac{\sigma^{2}}{2}t^{2}+O(t^{4}),\qquad
\sigma^{2}=\frac{g'(2)}{2}=\frac{\Lambda'(2)}{2\Lambda(2)},\qquad
\kappa_\infty:=\frac{\sigma^{2}}{2}=\frac{\Lambda'(2)}{4\Lambda(2)}\; }.
\]
进一步定义把单位圆扭曲的主奇点拉回 \(s\)-平面的归一化轨迹
\[
s_{\mathrm{pole}}(t):=\frac{P(it)}{\log\lambda},
\]
则
\[
\boxed{\;
s_{\mathrm{pole}}(t)
=1+i\,\frac{t}{2\log\lambda}-\frac{\kappa_\infty}{\log\lambda}\,t^{2}+O(t^{4})\;}
\]
其中漂移项系数 \(1/2\) 为相位--幅度分离的普适输出，而耗散项完全由端点导数 \(\Lambda'(2)\) 决定。
\end{proposition}
\begin{proof}
由 \(S(t)=2\cos(t/2)=2-\frac{t^{2}}{4}+\frac{t^{4}}{192}+O(t^{6})\) 与 \(g\) 在 \(2\) 处 Taylor 展开即得。
由于 \(|e^{it/2}|=1\)，有 \(\Re P(it)=g(S(t))\)，从而 \(\eta(t)=\exp(g(S(t))-g(2))\)。
最后 \(s_{\mathrm{pole}}(t)=1+\frac{it}{2\log\lambda}+\frac{g(S(t))-g(2)}{\log\lambda}\) 并代入上式即可。证毕。
\end{proof}

\paragraph{\texorpdfstring{$s$}{s}-平面完成化函数方程的谱互反判据}\label{par:s-plane-functional-eq-criterion}
上一段的“函数方程”发生在参数侧（$u\leftrightarrow 1/u$ 或 $\theta\leftrightarrow-\theta$）。
若希望把对称进一步提升为 Dirichlet 坐标 $z=\lambda^{-s}$ 下的
\(
s\leftrightarrow 1-s
\)
型对称，则需要一个独立于参数自对偶的谱互反结构。

\leanverified{paper\_reciprocal\_spectrum\_functional\_equation}
\begin{proposition}[\texorpdfstring{$\lambda$}{lambda}-互反（谱互反闭包）\texorpdfstring{$\Longleftrightarrow$}{<->} 完成化函数方程]\label{prop:reciprocal-spectrum-functional-equation}
设 $M$ 为有限维矩阵，$\lambda:=\rho(M)>0$ 为其 Perron 根。令
\[
\Delta_M(z):=\det(I-zM),\qquad
Z_M(s):=\Delta_M(\lambda^{-s})^{-1},
\qquad
r:=\deg_z\Delta_M(z).
\]
并定义完成化函数
\[
\Xi_M(s):=\lambda^{-rs/2}\,Z_M(s).
\]
则下述三条陈述等价：
\begin{enumerate}
  \item \textbf{谱互反闭包}对任意 \(\mu\in\mathrm{spec}(M)\setminus\{0\}\)，都有
  \[
  \frac{\lambda}{\mu}\in\mathrm{spec}(M)\setminus\{0\}
  \qquad(\text{按代数重数计}).
  \]
  \item \textbf{\(\Delta_M\) 的 \(\lambda\)-互反}存在常数 \(C'\neq 0\) 使得对所有复数 \(z\) 成立
  \[
  \lambda^{r}z^{r}\Delta_M\!\left(\frac{1}{\lambda z}\right)=C'\Delta_M(z).
  \]
  \item \textbf{完成化函数方程}存在常数 \(C\neq 0\) 使得
  \[
  \Xi_M(s)=C\,\Xi_M(1-s),
  \qquad C=C'\lambda^{-r/2}.
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
首先证明 (1)\(\Rightarrow\)(2)。记 \(M\) 的全部非零特征值（按代数重数计）为 \(\mu_1,\dots,\mu_r\)，则
\[
\Delta_M(z)=\prod_{j=1}^{r}(1-\mu_j z).
\]
从而
\[
\begin{aligned}
\lambda^r z^r\Delta_M\!\left(\frac{1}{\lambda z}\right)
&=\lambda^r z^r\prod_{j=1}^r\left(1-\frac{\mu_j}{\lambda z}\right)
\\
&=\lambda^r z^r\prod_{j=1}^r\left(-\frac{\mu_j}{\lambda z}\right)\prod_{j=1}^r\left(1-\frac{\lambda}{\mu_j}z\right)
\\
&=(-1)^r\Bigl(\prod_{j=1}^r\mu_j\Bigr)\prod_{j=1}^r\left(1-\frac{\lambda}{\mu_j}z\right).
\end{aligned}
\]
由 (1) 的互反闭包，\(\{\lambda/\mu_j\}_{j=1}^r\) 与 \(\{\mu_j\}_{j=1}^r\) 作为多重集合相同，故
\[
\prod_{j=1}^r\left(1-\frac{\lambda}{\mu_j}z\right)=\prod_{j=1}^r(1-\mu_j z)=\Delta_M(z).
\]
因此 (2) 成立，且可取 \(C'=(-1)^r\prod_{j=1}^r\mu_j\neq 0\)。
\par
下面证明 (2)\(\Rightarrow\)(3)。把 (2) 中的 \(z\) 取为 \(z=\lambda^{-s}\)，注意到
\(
(\lambda z)^{-1}=\lambda^{-(1-s)}
\)，得到
\[
\lambda^{r(1-s)}\,\Delta_M\!\bigl(\lambda^{-(1-s)}\bigr)=C'\,\Delta_M(\lambda^{-s}).
\]
两边取倒数并整理可得
\[
Z_M(1-s)=\Delta_M\!\bigl(\lambda^{-(1-s)}\bigr)^{-1}=\frac{\lambda^{r(1-s)}}{C'}\,\Delta_M(\lambda^{-s})^{-1}=\frac{\lambda^{r(1-s)}}{C'}\,Z_M(s).
\]
于是
\[
\Xi_M(1-s)
=\lambda^{-r(1-s)/2}Z_M(1-s)
=\frac{\lambda^{r(1-s)/2}}{C'}\,Z_M(s),
\]
而
\(
\Xi_M(s)=\lambda^{-rs/2}Z_M(s)
\)。
故
\(
\Xi_M(s)=C'\lambda^{-r/2}\,\Xi_M(1-s)
\)，即 (3) 成立，且 \(C=C'\lambda^{-r/2}\)。
\par
最后证明 (3)\(\Rightarrow\)(1)。由于 \(\Xi_M(s)=\lambda^{-rs/2}Z_M(s)\)，两者具有相同的极点集合与极点阶数。
若 \(s_0\) 为 \(Z_M\) 的一个极点，则由 (3) 得 \(1-s_0\) 亦为极点，且阶数相同。
由命题 \ref{prop:s-plane-spectrum-poles} 的“谱 \(\leftrightarrow\) 极点”字典，
极点 \(s_0\) 对应某个 \(\mu\in\mathrm{spec}(M)\setminus\{0\}\) 使 \(\lambda^{-s_0}\mu=1\)，从而 \(\lambda^{-(1-s_0)}=\mu/\lambda\)。
而 \(1-s_0\) 为极点意味着存在 \(\mu'\in\mathrm{spec}(M)\setminus\{0\}\) 使 \(\lambda^{-(1-s_0)}\mu'=1\)，即 \((\mu/\lambda)\mu'=1\)，从而 \(\mu'=\lambda/\mu\)。
极点阶数相同给出代数重数一致性，故 (1) 成立。
\end{proof}

\begin{remark}[等价的 \texorpdfstring{$\Delta_M$}{Delta}-互反多项式判据]\label{rem:delta-reciprocal-polynomial}
命题 \ref{prop:reciprocal-spectrum-functional-equation} 的谱条件等价于 $z$-根集合在
\(
z\mapsto (\lambda z)^{-1}
\)
下按重数不变，亦等价于存在 $C'\neq 0$ 使得
\[
\lambda^{r}z^{r}\Delta_M\!\left(\frac{1}{\lambda z}\right)=C'\Delta_M(z).
\]
因此当 $\Delta_M$ 的系数可精确计算（例如整系数矩阵）时，上式提供一个逐项可审计的判别式。
\end{remark}

\begin{proposition}[互反闭包 \texorpdfstring{$+$}{+} kernel RH \texorpdfstring{$\Rightarrow$}{=>} toy-RH]\label{prop:reciprocal-plus-kernel-rh-implies-toy-rh}
\leanverified{paper\_reciprocal\_plus\_kernel\_rh\_implies\_toy\_rh}
设 \(M\) 为有限维矩阵，\(\lambda=\rho(M)>1\)，并令
\(
Z_M(s)=\det(I-\lambda^{-s}M)^{-1}
\)。
若 \(M\) 满足命题 \ref{prop:reciprocal-spectrum-functional-equation} 的互反闭包（等价地：\(\Delta_M\) 为 \(\lambda\)-互反多项式），并且满足 kernel RH（Ramanujan 半径界）
\[
\max_{\mu\in\mathrm{spec}(M),\,\mu\neq \lambda}|\mu|\le \sqrt{\lambda},
\]
则 \(Z_M\) 满足 toy-RH：开带 \(0<\Re(s)<1\) 内的全部非平凡极点都落在 \(\Re(s)=\tfrac12\)。
等价地，
\[
(\forall\,\mu\ \text{满足}\ 1<|\mu|<\lambda)\qquad |\mu|=\sqrt{\lambda}.
\]
从而在互反闭包成立时，kernel RH 与 toy-RH 等价。
\end{proposition}
\begin{proof}
设 \(\mu\in\mathrm{spec}(M)\) 满足 \(1<|\mu|<\lambda\)。由 kernel RH 得 \(|\mu|\le \sqrt{\lambda}\)。
若 \(|\mu|<\sqrt{\lambda}\)，则由互反闭包 \(\lambda/\mu\in\mathrm{spec}(M)\)，并且
\(
|\lambda/\mu|>\sqrt{\lambda}
\)，这与 kernel RH 矛盾。
故对所有满足 \(1<|\mu|<\lambda\) 的特征值必有 \(|\mu|=\sqrt{\lambda}\)，等价于 toy-RH（命题 \ref{prop:s-plane-spectrum-poles} 的字典给出 \(\Re(s)=\log|\mu|/\log\lambda\)）。
\par
反向地，toy-RH 意味着不存在 \(\sqrt{\lambda}<|\mu|<\lambda\) 的特征值，否则将产生 \(\tfrac12<\Re(s)<1\) 的离线极点；而 \(|\mu|\le \lambda\) 总成立（\(\lambda=\rho(M)\)），故 toy-RH \(\Rightarrow\) kernel RH。
因此在互反闭包条件下二者等价。
\end{proof}

\begin{proposition}[有限维行列式的竖向晶格障碍]\label{prop:finite-kernel-lattice-obstruction}

Z_M(s):=\det(I-\lambda^{-s}M)^{-1}.

则 \(F\) 的零点与极点集合是\textbf{有限条竖向等差列的有限并集}，从而其计数函数满足

N_F(T):=\#\{\,s\in\mathrm{Zeros}(F)\cup\mathrm{Poles}(F):\ 0<\Im(s)\le T\,\}=O(T).

因此不存在任何固定有限乘积 \(F\) 使其“临界带内的零点/极点集合”与黎曼 \(\zeta(s)\)（或 \(\xi(s)\)）的非平凡零点集合一致：后者由 Riemann--von Mangoldt 公式满足 \(N_\zeta(T)\asymp T\log T\)。

\begin{proof}
第一段是命题 \ref{prop:s-plane-spectrum-poles} 的直接展开：\(\det(I-\lambda^{-s}M)=0\) 当且仅当 \(\lambda^{-s}=1/\mu\)（某个 \(\mu\neq 0\)），取对数即得极点列公式；虚部步长来自 \(\log(\cdot)\) 的 \(2\pi i\)-多值性。

对有限乘积 \(F\)，其零点/极点只可能来自各因子 \(\det(I-\lambda_j^{-s}M_j)\) 的零点集合之并集；每个因子在 \((0,T]\) 内贡献 \(O(T)\) 个等差点（常数仅依赖 \(\log\lambda_j\) 与 \(\#(\mathrm{spec}(M_j)\setminus\{0\})\)），故并集仍为 \(O(T)\)。
\par
最后一段：\(\zeta(s)\) 的非平凡零点计数满足 \(N_\zeta(T)=\frac{T}{2\pi}\log\frac{T}{2\pi}-\frac{T}{2\pi}+O(\log T)\)，故 \(N_\zeta(T)\) 与 \(O(T)\) 的增长级别不相容，因而不可能与任何固定有限乘积 \(F\) 的零/极点集合一致。证毕。
\end{proof}

\mathcal{E}\ +\ \sum_{j=1}^{J}\Delta_j\ZZ,
\]
其中 \(\mathcal{E}\subset\RR\) 为某个有限集合。特别地，对每个在带内确有出现的步长 \(\Delta_j\)，差集包含一条严格算术子晶格 \(\Delta_j\ZZ\)。
\end{corollary}
\begin{proof}
命题 \ref{prop:finite-kernel-lattice-obstruction} 已给出零/极点集合为有限条竖向等差列的有限并；将每条竖线上的点写成 \(\sigma_\ell+i(t_\ell+k\Delta_{j(\ell)})\) 形式即得首式。
\par
差集方面，任取两点 \(t=t_\ell+k\Delta_{j(\ell)}\)、\(t'=t_{\ell'}+k'\Delta_{j(\ell')}\)，则
\[
t-t'=(t_\ell-t_{\ell'})+\bigl(k\Delta_{j(\ell)}-k'\Delta_{j(\ell')}\bigr)\in (t_\ell-t_{\ell'})+\sum_{j=1}^J\Delta_j\ZZ,
\]
并且有限并集只引入有限个偏移量集合
\(
\mathcal{E}:=\{t_\ell-t_{\ell'}\}_{\ell,\ell'}.
\)
最后一段：若某个步长 \(\Delta_j\) 的等差列在带内出现，则该列在虚轴方向上含无穷点，故其内部差集必包含 \(\Delta_j\ZZ\)。证毕。
\end{proof}

\begin{corollary}[统计学晶格障碍：二点差分必含狄拉克梳齿]\label{cor:finite-determinant-statistical-lattice}
在推论 \ref{cor:finite-determinant-arithmetic-lattice} 的条件下，若 \(\mathcal{T}_F(a,b)\) 中存在某条步长为 \(\Delta>0\) 的完整等差列 \(t_0+\Delta\ZZ\)（例如某个因子 \(\det(I-\lambda^{-s}M)\) 的特征值列未被完全相消），则定义二点差分测度
\[
\nu_T:=\frac{1}{T}\sum_{t,t'\in\mathcal{T}_F(a,b)\cap(0,T]}\delta_{t-t'}\qquad(T>0),
\]
则对每个 \(n\in\ZZ\) 都有
\[
\liminf_{T\to\infty}\ \nu_T\bigl(\{n\Delta\}\bigr)\ \ge\ \frac{1}{\Delta}\ >0.
\]
因此任意沿 \(T\to\infty\) 的极限二点差分统计必包含在 \(\Delta\ZZ\) 上的原子部分；这排除了所有“unfolding 后二点差分具有连续密度”的目标统计形态。
\end{corollary}
\begin{proof}
设 \(\mathcal{A}_T:=(t_0+\Delta\ZZ)\cap(0,T]\)，则 \(|\mathcal{A}_T|=T/\Delta+O(1)\)。
对任意整数 \(n\)，对偶差 \(t-t'=n\Delta\) 的有序对数至少包含 \(\{(t_0+(k+n)\Delta,\ t_0+k\Delta)\}\)（当两点均落在 \((0,T]\) 内），故
\[
\nu_T(\{n\Delta\})
\ge
\frac{1}{T}\#\bigl\{(t,t')\in\mathcal{A}_T^2:\ t-t'=n\Delta\bigr\}
=\frac{|\mathcal{A}_T|-O(1)}{T}
=\frac{1}{\Delta}+o(1).
\]
取 \(\liminf\) 即得结论。证毕。
\end{proof}
\leanverified{paper\_finite\_determinant\_statistical\_lattice}

\begin{proposition}[临界圆模态数 \texorpdfstring{$B$}{B} 与临界线零极点密度的线性标定]\label{prop:critical-circle-density-calibration}
设 \(M\) 为有限维矩阵，\(\lambda=\rho(M)>1\)，并令
\(
Z_M(s)=\det(I-\lambda^{-s}M)^{-1}
\)。
记临界圆上的全部达界特征值（按代数重数计）为
\[
\mu_1,\dots,\mu_B,\qquad |\mu_b|=\sqrt{\lambda}.
\]
则 \(Z_M\) 在临界线 \(\Re(s)=\tfrac12\) 上的全部极点由 \(B\) 条竖向等差列给出：
\[
s=\frac12+\frac{i(\arg\mu_b+2\pi k)}{\log\lambda},\qquad b=1,\dots,B,\ k\in\ZZ,
\]
其虚部步长恒为 \(\Delta=2\pi/\log\lambda\)。并且高度 \(T\) 内的临界线极点计数满足
\[
N_{\mathrm{crit}}(T)
:=\sum_{\substack{s_0\in\mathrm{Poles}(Z_M)\\ \Re(s_0)=\tfrac12,\ 0<\Im(s_0)\le T}}\mathrm{ord}_{s_0}Z_M
=\frac{B\log\lambda}{2\pi}\,T+O(1)\qquad(T\to\infty).
\]
\end{proposition}
\begin{proof}
由命题 \ref{prop:finite-kernel-lattice-obstruction} 的极点字典，\(\mu_b\) 产生极点列
\(
s=(\log\mu_b+2\pi ik)/\log\lambda
\)。
当 \(|\mu_b|=\sqrt{\lambda}\) 时 \(\Re(s)=\log|\mu_b|/\log\lambda=\tfrac12\)，并将 \(\log\mu_b\) 写成 \(\tfrac12\log\lambda+i\arg\mu_b\) 即得临界线参数式；步长由 \(2\pi i/\log\lambda\) 给出。
\par
计数方面，每条等差列在 \((0,T]\) 内贡献 \(T/\Delta+O(1)=(\log\lambda/(2\pi))T+O(1)\) 个点；对 \(b=1,\dots,B\) 求和得到所示主项与 \(O(1)\) 余项。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$T\log T$}{T log T} 密度匹配的必要条件：\texorpdfstring{$B(T)\asymp (\log T)/(\log\lambda)$}{B(T) ~ (log T)/(log lambda)}]\label{cor:critical-circle-mode-growth}
\leanverified{paper\_critical\_circle\_mode\_growth}
若希望在“核版 RH/RH-sharp 的临界线模态”框架内把高度 \(T\) 内的临界线零极点密度提升到 \(N(T)\asymp T\log T\) 量级（如 Riemann--von Mangoldt 主项），则对任意固定 \(\lambda>1\) 的有限维核，必要地必须引入一族随尺度增长的核 \(M_T\) 使其临界圆模态数 \(B=B(T)\) 满足
\[
B(T)\ \asymp\ \frac{\log T}{\log\lambda}.
\]
在真实输入 \(40\) 状态核上 \(\lambda=\varphi^2\)，故 \(B(T)\asymp (\log T)/(2\log\varphi)\)。
\end{corollary}
\begin{proof}
由命题 \ref{prop:critical-circle-density-calibration}，固定 \(\lambda\) 时临界线点数满足 \(N_{\mathrm{crit}}(T)=\frac{B\log\lambda}{2\pi}T+O(1)\)。
若要达到 \(N_{\mathrm{crit}}(T)\asymp T\log T\)，则必须有 \(B=B(T)\asymp (\log T)/(\log\lambda)\)。
对 \(\lambda=\varphi^2\) 代入 \(\log\lambda=2\log\varphi\) 即得。证毕。
\end{proof}

\begin{remark}[对 \texorpdfstring{$\lambda=\varphi^2$}{lambda=phi^2} 的显式 lift 生成器：用循环提升把 \texorpdfstring{$B$}{B} 线性做大]\label{rem:critical-circle-mode-growth-cyclic-lift}

M^{\langle q\rangle}:=M\otimes P_q

对真实输入 \(40\) 状态核，临界圆上唯一达界特征值为 \(-\sqrt{\lambda}=-\varphi\)（注 \ref{rem:finite-rh-40-ramanujan}），故循环提升把临界圆模态数放大为 \(B=q\) 并将临界线梳齿间距细化为

q(T)\ \asymp\ \frac{\log T}{\log\lambda}
=\frac{\log T}{2\log\varphi},
\]
即为“lift--scale 标定律”的一个完全显式实例。
\end{remark}

\end{remark}
