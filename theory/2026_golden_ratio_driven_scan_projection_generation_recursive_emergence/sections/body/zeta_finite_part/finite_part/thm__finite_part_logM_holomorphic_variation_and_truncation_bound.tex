\begin{theorem}[\texorpdfstring{$\log\mathfrak{M}(\theta)$}{logM(theta)} 的解析性与变分公式]\label{thm:finite-part-logM-holomorphic-variation}
在小节 \ref{subsec:logM-analyticity} 的设定下，设 \(\theta\mapsto M_\theta\) 在某复邻域内全纯，并假设 \(\lambda(\theta)=\rho(M_\theta)\) 在该邻域内为简单特征值；记 \(z_\star(\theta)=\lambda(\theta)^{-1}\) 且 \(\zeta_\theta(z)=\det(I-zM_\theta)^{-1}\)。
则 \(\theta\mapsto \log\mathfrak{M}(\theta)\) 在该邻域内全纯，并且满足变分恒等式
\[
\boxed{\;
\frac{d}{d\theta}\log\mathfrak{M}(\theta)
=\frac{d}{d\theta}\log C(\theta)
\;+\;
\sum_{k\ge 2}\frac{\mu(k)}{k}\,\frac{d}{d\theta}\log \zeta_\theta\!\bigl(z_\star(\theta)^k\bigr)\; }.
\]
此外，对每个固定 \(z\)（满足 \(|z|<|z_\star(\theta)|\)）有基本微分恒等式
\[
\boxed{\;
\partial_\theta \log \zeta_\theta(z)
=
\mathrm{Tr}\!\Bigl((I-zM_\theta)^{-1}\,z\,\partial_\theta M_\theta\Bigr)\; }.
\]
\end{theorem}

\begin{proof}
由 \(\log\mathfrak{M}(\theta)=\log C(\theta)+\sum_{k\ge 2}\frac{\mu(k)}{k}\log \zeta_\theta(z_\star(\theta)^k)\) 的接口式（见 \S\ref{app:real-input-40-logm-theta}），对 \(k\ge 2\) 有 \(|z_\star(\theta)^k|<|z_\star(\theta)|\)，从而各项在邻域内全纯且尾部一致收敛，故和函数全纯并可逐项求导，得到第一式。
\par
对第二式，记 \(A_\theta(z)=I-zM_\theta\)，则 \(\log\zeta_\theta(z)=-\log\det A_\theta(z)\)。
对 \(\theta\) 求导并用 \(\partial_\theta\log\det A=\mathrm{Tr}(A^{-1}\partial_\theta A)\) 与 \(\partial_\theta A_\theta(z)=-z\,\partial_\theta M_\theta\) 即得。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\log\mathfrak{M}(\theta)$}{logM(theta)} 的可审计截断误差界]\label{thm:finite-part-logM-truncation-explicit-bound}
沿用定理 \ref{thm:finite-part-logM-holomorphic-variation} 的设定，并记 \(d=\dim M_\theta\)、\(\lambda=\lambda(\theta)>1\)。
定义 \(K\) 阶截断
\[
\log\mathfrak{M}_K(\theta):=\log C(\theta)+\sum_{k=2}^{K}\frac{\mu(k)}{k}\log \zeta_\theta(\lambda^{-k}).
\]
则对所有整数 \(K\ge 2\) 有显式误差上界
\[
\boxed{\;
\bigl|\log\mathfrak{M}(\theta)-\log\mathfrak{M}_K(\theta)\bigr|
\le
\frac{d}{K+1}\cdot\frac{\lambda^{-K}}{(1-\lambda^{-1})^2}\; }.
\]
\end{theorem}

\begin{proof}
写出特征值分解（按代数重数计）\(\det(I-zM_\theta)=\prod_{j=1}^{d}(1-\lambda_j z)\)，其中 \(|\lambda_j|\le\lambda\)。
当 \(k\ge 2\) 且 \(z=\lambda^{-k}\) 时，有 \(|\lambda_j z|\le \lambda^{1-k}\le\lambda^{-1}<1\)。
由不等式 \(|\log(1-w)|\le |w|/(1-|w|)\)（\(|w|<1\)）得
\[
|\log \zeta_\theta(\lambda^{-k})|
=\left|\sum_{j=1}^{d}\log(1-\lambda_j\lambda^{-k})^{-1}\right|
\le \sum_{j=1}^{d}\frac{|\lambda_j|\lambda^{-k}}{1-|\lambda_j|\lambda^{-k}}
\le d\cdot \frac{\lambda^{1-k}}{1-\lambda^{-1}}.
\]
因此
\[
\bigl|\log\mathfrak{M}(\theta)-\log\mathfrak{M}_K(\theta)\bigr|
\le
\sum_{k>K}\frac{1}{k}\,|\log \zeta_\theta(\lambda^{-k})|
\le
\frac{d}{1-\lambda^{-1}}\sum_{k>K}\frac{\lambda^{1-k}}{k}.
\]
对 \(k>K\) 有 \(1/k\le 1/(K+1)\)，故
\[
\sum_{k>K}\frac{\lambda^{1-k}}{k}
\le
\frac{1}{K+1}\sum_{k>K}\lambda^{1-k}
=
\frac{1}{K+1}\cdot\frac{\lambda^{-K}}{1-\lambda^{-1}}.
\]
合并即得所示界。证毕。
\end{proof}

\begin{corollary}[截断层间的比例传递与穷竭误差]\label{cor:finite-part-logM-relative-scale-transfer}
沿用定理 \ref{thm:finite-part-logM-truncation-explicit-bound} 的记号。则对任意整数 \(L>K\ge 2\)，有
\[
\bigl|\log\mathfrak{M}_L(\theta)-\log\mathfrak{M}_K(\theta)\bigr|
\le
\frac{d}{K+1}\cdot\frac{\lambda^{-K}}{(1-\lambda^{-1})^2}.
\]
若进一步有 \(\log\mathfrak{M}(\theta)\neq 0\)，则
\[
\left|
\frac{\log\mathfrak{M}_K(\theta)}{\log\mathfrak{M}(\theta)}-1
\right|
\le
\frac{d}{(K+1)\,|\log\mathfrak{M}(\theta)|}\cdot\frac{\lambda^{-K}}{(1-\lambda^{-1})^2}.
\]
\end{corollary}

\begin{proof}
第一式中，\(\log\mathfrak{M}_L(\theta)-\log\mathfrak{M}_K(\theta)\) 正是从 \(k=K+1\) 到 \(k=L\) 的有限尾和，因此其绝对值不超过定理 \ref{thm:finite-part-logM-truncation-explicit-bound} 给出的从 \(k=K+1\) 到无穷的总尾和界。第二式则由
\[
\left|
\frac{\log\mathfrak{M}_K(\theta)}{\log\mathfrak{M}(\theta)}-1
\right|
=
\frac{\bigl|\log\mathfrak{M}_K(\theta)-\log\mathfrak{M}(\theta)\bigr|}{|\log\mathfrak{M}(\theta)|}
\]
与定理 \ref{thm:finite-part-logM-truncation-explicit-bound} 直接得到。证毕。
\end{proof}

\begin{theorem}[完备 Euler denominator ledger 与可公度--奇异择一性]\label{distill:euclid_elements-proportion_and_denominator_spectrum_families-zeta-completed-denominator-ledger-alternative}
\textbf{日期时间：2026-04-23T12:21:04+08:00。家族：Euclid--Elements / 比例与 denominator-spectrum 族。依赖状态：conditional audit。}
沿用定理 \ref{thm:finite-part-logM-holomorphic-variation} 的设定，固定参数值 \(\theta\)。令 \(\lambda=\lambda(\theta)>1\) 并记 \(M=M_\theta\)。定义完备 Euler denominator ledger
\[
\widehat{\mathcal L}:=\prod_{k\ge2}\prod_{r\ge1}\mathbb C\,\mathbf e_{k,r}
\]
及系数族
\[
A_{k,r}(\theta):=\frac{\mu(k)}{kr}\,\lambda^{-kr}\,\operatorname{Tr}(M^r)
\qquad(k\ge2,\ r\ge1).
\]
则
\[
\mathfrak D_\theta:=\bigl(A_{k,r}(\theta)\bigr)_{k,r}\in\widehat{\mathcal L}
\]
是良定义的完备 ledger，且其标量求值绝对收敛：
\[
\sum_{k\ge2}\sum_{r\ge1}|A_{k,r}(\theta)|<\infty.
\]
进一步，若
\[
\operatorname{Supp}^{\mu}(\mathfrak D_\theta):=\bigl\{(k,r):\mu(k)\operatorname{Tr}(M^r)\ne0\bigr\},
\]
则 denominator spectrum
\[
\mathsf{Den}(\theta):=\bigl\{k\ge2:\exists r\ge1\text{ with }(k,r)\in\operatorname{Supp}^{\mu}(\mathfrak D_\theta)\bigr\}
\]
满足如下精确择一性。
\begin{enumerate}
  \item \textbf{可公度周期情形。}若存在 \(N\in\mathbb N_{>0}\) 使得对每个 \(k\in\mathsf{Den}(\theta)\) 均有 \(k\mid N\)，则全部非零 Euler denominator support 穿过 \(N\) 的有限整除格。等价地，每个活跃 denominator 原子关于单一公共 denominator 环 \(\mathbb Z/N\mathbb Z\) 具有周期性。
  \item \textbf{奇异有限素数情形。}若 \(\mathsf{Den}(\theta)\) 的 support 集中在有限素数集 \(P\) 上但不被任何单一 \(N\) 的因子格所包含，则对每个高度界 \(H\)，存在活跃原子 \((k,r)\in\operatorname{Supp}^{\mu}(\mathfrak D_\theta)\) 满足 \(k\) 的全部素因子都在 \(P\) 中且
  \[
  \sum_{p\in P}v_p(k)>H.
  \]
  因此障碍并非来自新素数的出现，而是无界的 \(P\)-adic denominator 高度。
  \item \textbf{不可公度多素数情形。}若 \(\mathsf{Den}(\theta)\) 中元素之素因子的并集为无穷集，则没有有限素数寄存器能够容纳活跃的 Euler denominator support。
\end{enumerate}
对有限 audit window \(\mathcal W\subset\{(k,r):k\ge2,r\ge1\}\)，将 \(\operatorname{Supp}^{\mu}(\mathfrak D_\theta)\) 替换为
\[
\operatorname{Supp}^{\mu}_{\mathcal W}(\mathfrak D_\theta)
:=\bigl\{(k,r)\in\mathcal W:\mu(k)\operatorname{Tr}(M^r)\ne0\text{ 已被认证}\bigr\},
\]
上述三项陈述仍然成立。关于完整 ledger 的任何结论另需对认证非零 support 之外的每个原子提供零证书。具有 \(\mu(k)=0\) 的原子不构成该 Euler ledger 的 denominator 障碍。
\end{theorem}
\begin{proof}
设 \(\alpha_1,\ldots,\alpha_d\) 为 \(M\) 的特征值（按代数重数计）。由 \(\lambda=\rho(M)\) 知 \(|\alpha_j|\le\lambda\)，从而
\[
|\operatorname{Tr}(M^r)|\le\sum_{j=1}^{d}|\alpha_j|^r\le d\lambda^r.
\]
因此
\[
|A_{k,r}(\theta)|
\le \frac{d}{kr}\lambda^{r-kr}
=\frac{d}{kr}\lambda^{-(k-1)r}.
\]
先对 \(k\) 再对 \(r\) 求和，
\[
\sum_{k\ge2}\sum_{r\ge1}|A_{k,r}(\theta)|
\le d\sum_{k\ge2}\frac1k\sum_{r\ge1}\frac{\lambda^{-(k-1)r}}{r}
\le d\sum_{k\ge2}\frac1k\sum_{r\ge1}\lambda^{-(k-1)r}.
\]
内层几何级数为 \(\lambda^{-(k-1)}/(1-\lambda^{-(k-1)})\)，可被常数 \(C=C(\lambda)\) 倍的 \(\lambda^{-(k-1)}\) 所控制。因此末项级数受控于
\[
 dC\sum_{k\ge2}\frac{\lambda^{-(k-1)}}{k}<\infty.
\]
这证明了 \(\mathfrak D_\theta\) 是完备乘积 ledger 的合法元素，且其标量求值绝对收敛。\(\operatorname{Supp}^{\mu}\) 的使用是本质性的：若 \(\mu(k)=0\)，则该 denominator 上的每个系数无论 \(\operatorname{Tr}(M^r)\) 如何均为零，因而此类原子不能贡献 Euler denominator 障碍。

下证择一性。若公共 denominator \(N\) 存在，则每个活跃的 \(k\) 属于有限集 \(\{d:d\mid N\}\)。从而全部活跃 denominator 的比较在 \(N\) 的有限整除格内进行，等价地在有限公共 denominator 环 \(\mathbb Z/N\mathbb Z\) 内进行。这恰为可公度周期情形。

其次，假设活跃 denominator 仅使用有限素数集 \(P\)，但不被任何固定 \(N\) 的因子格所包含。若对所有活跃 denominator 上的高度 \(\sum_{p\in P}v_p(k)\) 被某 \(H\) 所界定，则仅有有限多个指数向量 \((v_p(k))_{p\in P}\) 可以出现。由相应有限多个 \(k\) 的最小公倍数即得单一整数 \(N\) 整除每个活跃 denominator，与假设矛盾。因此对每个 \(H\) 存在 support 集中在 \(P\) 上且高度大于 \(H\) 的活跃 \(k\)，这正是奇异有限素数情形。此处陈述显式限制于素因子在 \(P\) 中的 denominator；不包含 \(P\) 外素数的原子不在此投影之内。

最后，若活跃素数 support 为无穷集，则任何仅使用有限多素数坐标的寄存器都遗漏了某个素因子在寄存器之外的活跃 denominator。因此该 support 不能用有限素数寄存器表示。三种情形穷尽一切可能：活跃素数 support 或者无穷，或者有限；在有限情形下，指数高度或者有界（从而由最小公倍数构造给出公共 denominator），或者无界。三者按构造互斥。

有限窗口的断言是将同一论证施加于认证子集 \(\operatorname{Supp}^{\mu}_{\mathcal W}(\mathfrak D_\theta)\)。由于未认证原子可能在后续被认证为非零，完整 ledger 的结论仅当审计同时为认证非零 support 之外的原子提供零证书时才成立。这正是 Omega denominator-spectrum 规则：有限截断仅为其显式包含的内容提供合法证书，而全局可公度性则需全原子 ledger 审计，而非投影假象。证毕。
\end{proof}

\begin{theorem}[squarefree Euler 尾穷竭与稳定化 denominator 证书]\label{distill:euclid_elements-exhaustion_and_finite_budget_closure_families-zeta-squarefree-euler-tail-exhaustion}
\textbf{日期时间：2026-04-23T12:38:38+08:00。家族：Euclid--Elements / 穷竭与有限预算闭包族。依赖状态：rigidity。}
沿用定理 \ref{thm:finite-part-logM-holomorphic-variation} 的设定，固定 \(\theta\)。令 \(M=M_\theta\)，\(d=\dim M\)，\(\lambda=\rho(M)>1\)。对 \(k\ge2\) 及 \(r\ge1\)，定义 Euler 系数原子
\[
A_{k,r}(\theta):=\frac{\mu(k)}{kr}\,\lambda^{-kr}\operatorname{Tr}(M^r).
\]
对 \(K\ge2\)，定义绝对尾预算
\[
\mathcal E_K(\theta):=\sum_{k>K}\sum_{r\ge1}|A_{k,r}(\theta)|.
\]
则
\[
\boxed{\;
\mathcal E_K(\theta)\le
\frac{d}{K+1}\cdot\frac{\lambda^{-K}}{(1-\lambda^{-1})^2}.
\;}
\]
特别地，完备 Euler ledger 被有限 denominator 窗口穷竭，其显式残差预算趋于零。

令
\[
\mathsf{Den}(\theta):=\{k\ge2:\exists r\ge1\text{ with }\mu(k)\operatorname{Tr}(M^r)\ne0\},
\qquad
\mathsf{Den}_{\le K}(\theta):=\mathsf{Den}(\theta)\cap\{2,\ldots,K\}.
\]
则每个活跃 denominator 均为 squarefree。因此有限 denominator 证书 \(\mathsf{Den}_{\le K}(\theta)\) 在 \(K\to\infty\) 时稳定化，当且仅当活跃素数 support
\[
\mathsf P(\theta):=\bigcup_{k\in\mathsf{Den}(\theta)}\{p:p\mid k\}
\]
为有限集。在此情形下，全部活跃 denominator 落入单一 squarefree 整数
\[
N_\theta:=\prod_{p\in\mathsf P(\theta)}p
\]
的因子格中。若证书不稳定化，则 \(\mathsf P(\theta)\) 为无穷集；等价地，对每个有限素数寄存器 \(P_0\) 和每个高度界 \(K_0\)，存在活跃的 squarefree denominator \(k>K_0\) 且其某个素因子在 \(P_0\) 之外。因此，在当前 \(\lambda>1\) Euler 有限部分假设下，完备 denominator-ledger 择一性特化为 squarefree 二分法：有限可公度 support 对不可公度的无穷素数尾。更一般的 denominator-ledger 择一性中的奇异有限素数高度分支在此不可能出现，因为 \(\mu(k)\ne0\) 强制 \(k\) 为 squarefree。
\end{theorem}
\begin{proof}
设 \(\alpha_1,\ldots,\alpha_d\) 为 \(M\) 的特征值（按代数重数计）。由 \(\lambda=\rho(M)\) 知 \(|\alpha_i|\le\lambda\)，从而
\[
|\operatorname{Tr}(M^r)|\le\sum_{i=1}^d|\alpha_i|^r\le d\lambda^r.
\]
因此
\[
|A_{k,r}(\theta)|
\le \frac{d}{kr}\lambda^{r-kr}
=\frac{d}{kr}\lambda^{-(k-1)r}.
\]
对 \(k>K\) 及 \(r\ge1\) 求和，并利用 \(1/k\le1/(K+1)\)，得
\[
\mathcal E_K(\theta)\le
\frac{d}{K+1}\sum_{k>K}\sum_{r\ge1}\frac{\lambda^{-(k-1)r}}{r}.
\]
由于 \(r^{-1}\le1\) 且 \(k>K\ge2\)，
\[
\sum_{r\ge1}\frac{\lambda^{-(k-1)r}}{r}\le
\sum_{r\ge1}\lambda^{-(k-1)r}\le
\frac{\lambda^{-(k-1)}}{1-\lambda^{-1}}.
\]
从而
\[
\mathcal E_K(\theta)\le
\frac{d}{K+1}\cdot\frac1{1-\lambda^{-1}}
\sum_{k>K}\lambda^{-(k-1)}
=\frac{d}{K+1}\cdot\frac{\lambda^{-K}}{(1-\lambda^{-1})^2},
\]
此即所断言的残差预算。

下确定哪些有限证书稳定化。若 \(k\in\mathsf{Den}(\theta)\)，则 \(\mu(k)\ne0\)。Möbius 函数恰在可被某个完全平方数整除的整数上为零；因此每个活跃 denominator \(k\) 均为 squarefree。

首先假设 \(\mathsf P(\theta)\) 为有限集。由于每个活跃的 \(k\) 为 squarefree 且仅使用 \(\mathsf P(\theta)\) 中的素数，故 \(k\) 整除
\[
N_\theta=\prod_{p\in\mathsf P(\theta)}p.
\]
这样的因子仅有有限个，因此 \(\mathsf{Den}(\theta)\) 有限，且递增证书 \(\mathsf{Den}_{\le K}(\theta)\) 在 \(K\ge\max\mathsf{Den}(\theta)\) 后稳定化（空集情形按自然约定理解）。

反之，若 \(\mathsf{Den}_{\le K}(\theta)\) 稳定化，则 \(\mathsf{Den}(\theta)\) 有限，从而其素因子的并集亦有限。这证明了稳定化与有限活跃素数 support 的等价性。

若稳定化不成立，则 \(\mathsf{Den}(\theta)\) 为无穷集。若 \(\mathsf P(\theta)\) 有限，squarefree 性将迫使 \(\mathsf{Den}(\theta)\) 落入 \(\prod_{p\in\mathsf P(\theta)}p\) 的有限因子格中，矛盾。因此 \(\mathsf P(\theta)\) 为无穷集。给定有限素数寄存器 \(P_0\) 和界 \(K_0\)，仅有有限个活跃 denominator \(k\le K_0\) 存在，且它们仅涉及有限多个素数。由于 \(\mathsf P(\theta)\) 无穷，可选取一个素因子在那些有限多素数之并集和 \(P_0\) 之外的活跃 denominator。该 denominator 为 squarefree 且必然大于 \(K_0\)。

解析尾估计与 denominator 证书陈述因此分离了两种极限闭包概念：标量 Euler 尾由显式预算 \(\mathcal E_K(\theta)\) 控制，而有限 denominator 证书仅在有限素数可公度情形下稳定化。证毕。
\end{proof}

\begin{proposition}[有限 Euler 窗口归一化的 \v Cech 秩界与规范唯一性]\label{distill:cech_cohomology-finite_nerve_obstruction_bounds-zeta-euler-window-normalization-gluing}
沿用定理 \ref{distill:euclid_elements-proportion_and_denominator_spectrum_families-zeta-completed-denominator-ledger-alternative} 中的 completed Euler denominator ledger 记号。固定一个有限 audit window
\[
\mathcal W\subset\{(k,r):k\ge2,\ r\ge1\},
\]
并令 \(\{U_i\}_{i\in I}\) 为 \(\mathcal W\) 的有限覆盖，\(\mathcal N\) 为其 nerve。设 \(\mathcal N\) 的一骨架连通。令 \(A\) 为有限阿贝尔规范群，并假设该有限窗口的局部 Euler 归一化审计具有如下形式：每个顶点 \(i\) 上给出局部归一化选择，任意非空重叠边 \(ij\) 上的两种选择之差由元素
\[
g_{ij}\in A,\qquad g_{ji}=-g_{ij}
\]
记录，且三重重叠相容性正是
\[
g\in Z^1(\mathcal N;A).
\]
顶点规范 \(a=(a_i)\in C^0(\mathcal N;A)\) 的作用约定为把边账本改为
\[
g'_{ij}=g_{ij}-(a_j-a_i).
\]
并规定该有限窗口的归一化可粘接，当且仅当存在某个顶点规范使所有 \(g'_{ij}\) 同时为零。

则该有限窗口中所有可能的归一化粘接障碍都落在
\[
[g]\in\check H^1(\mathcal N;A),
\]
其可登记障碍秩不超过 \(\mathrm d(\check H^1(\mathcal N;A))\)，其中 \(\mathrm d\) 为有限阿贝尔群的最小生成元数。若 \(\check H^1(\mathcal N;A)=0\)，则有限窗口归一化总可粘接；并且在固定一个根顶点 \(o\in I\) 且要求 \(a_o=0\) 后，杀死 \(g\) 的顶点规范唯一。未固定根时，该规范唯一性只成立到全局 \(A\)-平移为止。

因此，在该有限 \((\mathcal W,\mathcal N,A)\) 审计接口内，\(\check H^1(\mathcal N;A)=0\) 排除了 Euler 窗口归一化的有限 \v Cech holonomy 失败；对完整 \(\mathfrak D_\theta\) 的任何剩余认证失败仍可能来自窗口外零证书缺失、Euler 尾预算或解析截断误差，而不是该有限窗口的粘接类。
\end{proposition}
\begin{proof}
由于三重重叠相容性被假设为 \(g\in Z^1(\mathcal N;A)\)，边账本定义了上同调类 \([g]\in\check H^1(\mathcal N;A)\)。若顶点规范 \(a\) 被施加，则
\[
g' = g-\delta a.
\]
因此 \([g']=[g]\)。换言之，\([g]\) 是有限窗口归一化审计在允许规范下的不变量。

按粘接规则，有限窗口可粘接当且仅当存在 \(a\in C^0(\mathcal N;A)\) 使 \(g-\delta a=0\)，也就是当且仅当 \([g]=0\)。所以所有可能的粘接障碍构成 \(\check H^1(\mathcal N;A)\) 的子集；它们生成的有限阿贝尔子群所需生成元数不超过 \(\check H^1(\mathcal N;A)\) 本身的最小生成元数。由于 \(\mathcal N\) 与 \(A\) 均有限，该上同调群可由有限 incidence 矩阵的 Smith 标准形计算。

若 \(\check H^1(\mathcal N;A)=0\)，则每个上循环 \(g\) 都是余边界，故存在 \(a\) 使 \(g=\delta a\)。施加该规范后 \(g'=0\)，从而有限窗口归一化可粘接。

最后证明唯一性陈述。设 \(a,b\in C^0(\mathcal N;A)\) 都满足 \(\delta a=g=\delta b\)。则 \(c:=a-b\) 满足 \(\delta c=0\)，即对每条边 \(ij\) 都有 \(c_j-c_i=0\)。由于 \(\mathcal N\) 的一骨架连通，沿任意路径传递可得所有 \(c_i\) 相等。因此两个规范相差一个全局常量 \(c\in A\)。若再规定根顶点 \(o\) 上 \(a_o=b_o=0\)，则该常量为零，故 \(a=b\)。这说明零上同调给出的是存在性与模全局规范的唯一性，而非未定规范下的绝对唯一性。

完整 Euler denominator ledger 还包含 \(\mathcal W\) 外的原子与无穷尾部。上述论证只处理固定有限窗口的 \v Cech 粘接类；因此即使该类消失，窗口外零证书、尾和预算或解析截断仍可独立失败。证毕。
\end{proof}
