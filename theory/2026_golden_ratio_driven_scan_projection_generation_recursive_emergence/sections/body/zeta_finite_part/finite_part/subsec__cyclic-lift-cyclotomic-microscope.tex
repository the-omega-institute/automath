\begin{proposition}[\(\gcd/\mathrm{lcm}\) 合成律：复合循环 lift 的多重覆盖分解]\label{prop:finite-part-cyclic-lift-gcd-lcm}
\leanverified{paper\_finite\_part\_cyclic\_lift\_gcd\_lcm\_seeds}
\leanverified{paper\_finite\_part\_cyclic\_lift\_gcd\_lcm\_package}
\leanverified{paper\_finite\_part\_cyclic\_lift\_gcd\_lcm}
对任意整数 \(m,n\ge 2\)，令 \(g:=\gcd(m,n)\)、\(\ell:=\mathrm{lcm}(m,n)\)。
则存在置换相似使得
\[
\boxed{\;
P_m\otimes P_n\ \cong\ \bigoplus_{t=1}^{g} P_{\ell}\; }.
\]
因此对任意方阵 \(M\) 有
\[
\boxed{\;
(M^{\langle m\rangle})^{\langle n\rangle}
\cong
\bigoplus_{t=1}^{g} M^{\langle \ell\rangle}\; }.
\]
从而在 \(\zeta\)-层面得到严格幂律
\[
\boxed{\;
\zeta_{(M^{\langle m\rangle})^{\langle n\rangle}}(z)
=\bigl(\zeta_{M^{\langle \ell\rangle}}(z)\bigr)^{g}\; }.
\]
特别地，当 \(\gcd(m,n)=1\) 时有 \((M^{\langle m\rangle})^{\langle n\rangle}\cong M^{\langle mn\rangle}\)。
\end{proposition}
\begin{proof}
矩阵 \(P_m\otimes P_n\) 在标准基上诱导置换
\(
(i,j)\mapsto(i+1,j+1)
\)
（指标分别按 \(m,n\) 取模），其轨道长度为满足 \(k\equiv 0\ (\mathrm{mod}\ m)\) 且 \(k\equiv 0\ (\mathrm{mod}\ n)\) 的最小正整数，即 \(\ell\)；
而轨道条数为 \(mn/\ell=g\)。
故 \(P_m\otimes P_n\) 的循环分解由 \(g\) 个长度为 \(\ell\) 的循环构成，从而与 \(\bigoplus_{t=1}^{g}P_\ell\) 置换相似。
第二式由张量积的结合律
\(
(M\otimes P_m)\otimes P_n\cong M\otimes(P_m\otimes P_n)
\)
与直和分解推出。
最后 \(\zeta\) 的幂律由 \(\det\) 在直和下的乘法性得到。证毕。
\end{proof}

\begin{theorem}[Gauss--cyclotomic 压缩：循环 lift 的约化行列式常数闭式]\label{thm:finite-part-cyclic-lift-reduced-constant-closed}
设 \(M\in\ZZ_{\ge 0}^{d\times d}\) 原始，Perron 根为 \(\rho=\rho(M)>0\)，其余特征值（按代数重数）为 \(\mu_2,\dots,\mu_d\)。
令
\[
\lambda_j:=\frac{\mu_j}{\rho}\qquad(j=2,\dots,d),
\]
并定义主极点留数常数（命题 \ref{prop:finite-part-residue-reduced-determinant}）
\[
C(M):=\lim_{z\to 1/\rho}(1-\rho z)\,\zeta_M(z).
\]
则对任意整数 \(q\ge 1\)，循环 lift \(M^{\langle q\rangle}=M\otimes P_q\) 的留数常数满足完全闭式
\[
\boxed{\;
C(M^{\langle q\rangle})
=\frac{1}{q}\prod_{j=2}^{d}\bigl(1-\lambda_j^{q}\bigr)^{-1}\; }.
\]
等价地，
\[
\boxed{\;
\log C(M^{\langle q\rangle})
=-\log q-\sum_{j=2}^{d}\log\!\bigl(1-\lambda_j^{q}\bigr)\; }.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:finite-rh-phase-lift} 的谱相位细分，\(M^{\langle q\rangle}\) 的特征值为 \(\mu_j\omega_q^k\)。
因此
\[
C(M^{\langle q\rangle})
=\prod_{(j,k)\neq(1,0)}\left(1-\frac{\mu_j\omega_q^k}{\rho}\right)^{-1}.
\]
将 \(j=1\) 与 \(j\ge 2\) 分离：
\[
\prod_{k=1}^{q-1}(1-\omega_q^k)^{-1}=\frac{1}{q},
\qquad
\prod_{k=0}^{q-1}\left(1-\lambda_j\omega_q^k\right)=1-\lambda_j^{q}\quad(|\lambda_j|<1),
\]
代回即得所示闭式。
对数式仅为取对数后的记号改写（由 \(|\lambda_j|<1\) 的幂级数定义可取一致分支）。证毕。
\end{proof}

\begin{definition}[常数层序列、谱幂和与扭点多项式]\label{def:finite-part-cyclic-lift-psi}
在定理 \ref{thm:finite-part-cyclic-lift-reduced-constant-closed} 的设定下，定义
\[
\Psi_q(M):=\log q+\log C(M^{\langle q\rangle})\qquad(q\ge 1).
\]
并进一步定义去平凡因子的常数层异常
\[
\mathcal{E}_q(M):=\Psi_q(M)-\log C(M)
=\log q+\log\frac{C(M^{\langle q\rangle})}{C(M)}\qquad(q\ge 1),
\]
以及扭点多项式
\[
F_M(t):=\frac{\det(I-tM/\rho)}{1-t}=\prod_{j=2}^{d}(1-\lambda_j t).
\]
最后记归一化谱幂和与相对谱半径比
\[
u_n(M):=\sum_{j=2}^{d}\lambda_j^{n}\qquad(n\ge 1),
\qquad
\Lambda_{\mathrm{rel}}(M):=\max_{2\le j\le d}|\lambda_j|\in(0,1).
\]
\end{definition}

\begin{proposition}[cyclotomic 分解：常数层异常的除数格反演]\label{prop:finite-part-cyclic-lift-cyclotomic-divisor-mobius}
令 \(\Phi_r\) 表示第 \(r\) 个 cyclotomic 多项式。
对每个整数 \(r\ge 2\) 定义其常数层分量
\[
\mathcal{C}_r(M):=-\sum_{j=2}^{d}\log \Phi_r(\lambda_j),
\]
其中 \(\log\Phi_r(z)\) 取 \(|z|<1\) 上由 \(\Phi_r(0)=1\) 所诱导的解析分支（因 \(|\lambda_j|<1\)，该分支在各 \(\lambda_j\) 处良定义）。
则对任意整数 \(q\ge 2\) 有严格恒等式
\[
\boxed{\;
\mathcal{E}_q(M)=\sum_{\substack{r\mid q\\ r\ge 2}}\mathcal{C}_r(M)\; }.
\]
并且在除数格上可由 Möbius 反演恢复
\[
\boxed{\;
\mathcal{C}_q(M)=\sum_{r\mid q}\mu(r)\,\mathcal{E}_{q/r}(M)\qquad(q\ge 2)\; }.
\]
\end{proposition}
\begin{proof}
由定义 \ref{def:finite-part-cyclic-lift-psi} 与定理 \ref{thm:finite-part-cyclic-lift-reduced-constant-closed}，
\[
\mathcal{E}_q(M)
=-\sum_{j=2}^{d}\log\!\left(\frac{1-\lambda_j^{q}}{1-\lambda_j}\right).
\]
由 cyclotomic 因子分解 \(1-x^{q}=\prod_{r\mid q}\Phi_r(x)\)（且 \(\Phi_1(x)=x-1\)）得到
\(
\frac{1-x^{q}}{1-x}=\prod_{r\mid q,\ r\ge 2}\Phi_r(x)
\)。
将 \(x=\lambda_j\) 代入并取对数求和即得第一式。
第二式为除数格上的 Möbius 反演（Dirichlet 卷积意义下）对 \(\mathcal{E}_q=\sum_{r\mid q,\ r\ge 2}\mathcal{C}_r\) 的唯一逆。证毕。
\end{proof}

\begin{proposition}[倍数求和展开：\(\Psi_q\) 是归一化幂和的折叠变换]\label{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}
在定义 \ref{def:finite-part-cyclic-lift-psi} 的记号下，对每个 \(q\ge 1\) 有绝对收敛展开
\[
\boxed{\;
\Psi_q(M)=\sum_{m\ge 1}\frac{1}{m}\,u_{qm}(M)\; }.
\]

\[
\boxed{\;
\Psi_q(M)=u_q(M)+O\!\left(\Lambda_{\mathrm{rel}}(M)^{2q}\right)\qquad(q\to\infty)\; }.

\end{proposition}
\begin{proof}
由定理 \ref{thm:finite-part-cyclic-lift-reduced-constant-closed}，
\(
\Psi_q(M)=-\sum_{j=2}^{d}\log(1-\lambda_j^q)

在 \(|x|<1\) 下用幂级数恒等式 \(-\log(1-x)=\sum_{m\ge 1}x^m/m\)，并逐项求和得到
\[
\Psi_q(M)=\sum_{j=2}^{d}\sum_{m\ge 1}\frac{\lambda_j^{qm}}{m}
=\sum_{m\ge 1}\frac{1}{m}\sum_{j=2}^{d}\lambda_j^{qm}

与
\(
\sum_{m\ge 2}\Lambda_{\mathrm{rel}}^{qm}=O(\Lambda_{\mathrm{rel}}^{2q})

\end{proof}

\begin{proposition}[root-of-unity Fourier sieve：扭点取值筛出谱幂和的同余类分量]\label{prop:finite-part-cyclic-lift-root-unity-fourier-sieve}
对任意整数 \(q\ge 2\) 与 \(0\le r\le q-1\)，定义离散 Fourier 系数
\[
\Gamma_{q,r}(M):=-\frac{1}{q}\sum_{j=0}^{q-1}\omega_q^{-rj}\,\log F_M(\omega_q^j),
\]
其中 \(\log F_M\) 取 \(|t|\le 1\) 上由 \(F_M(0)=1\) 所诱导的解析分支。
则：
\begin{enumerate}
  \item 对 \(1\le r\le q-1\) 有绝对收敛展开
  \[
  \boxed{\;
  \Gamma_{q,r}(M)=\sum_{m\ge 0}\frac{u_{r+mq}(M)}{r+mq}\; }.
  \]
  \item 对 \(r=0\) 有
  \[
  \boxed{\;
  \Gamma_{q,0}(M)=\sum_{m\ge 1}\frac{u_{mq}(M)}{mq}\; }.
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
由 \(F_M(t)=\prod_{j=2}^{d}(1-\lambda_j t)\) 与 \(|\lambda_j|<1\) 可知 \(F_M\) 在 \(|t|\le 1\) 内无零点，故 \(\log F_M\) 的解析分支存在且由 \(t=0\) 处的 Taylor 展开唯一确定。
由幂级数恒等式 \(-\log(1-x)=\sum_{n\ge 1}x^n/n\)（\(|x|<1\)）得对 \(|t|\le 1\) 成立绝对收敛展开
\[
-\log F_M(t)=\sum_{n\ge 1}\frac{u_n(M)}{n}\,t^n.
\]
代入 \(\Gamma_{q,r}\) 并交换求和（绝对收敛保证可交换）得
\[
\Gamma_{q,r}(M)=\sum_{n\ge 1}\frac{u_n(M)}{n}\cdot \frac{1}{q}\sum_{j=0}^{q-1}\omega_q^{j(n-r)}.

于是当 \(1\le r\le q-1\) 时 \(n=r+mq\,(m\ge 0)\)；当 \(r=0\) 时 \(n=mq\,(m\ge 1)\)，从而得到两式。证毕。
\end{proof}

\begin{theorem}[polylog--Dirichlet 母函数：常数层异常生成的谱 \texorpdfstring{$L$}{L} 通道]\label{thm:finite-part-cyclic-lift-polylog-dirichlet-xi}
\leanverified{paper\_etds\_finite\_part\_cyclic\_lift\_polylog\_dirichlet\_xi}
对复数 \(s\) 满足 \(\Re(s)>1\)，定义 Dirichlet 母函数
\[
\Xi_M(s):=\sum_{q\ge 1}\frac{\mathcal{E}_q(M)}{q^s}.
\]
则存在绝对收敛的 polylog 表示
\[
\boxed{\;
\Xi_M(s)=\sum_{j=2}^{d}\sum_{m\ge 1}\frac{1}{m}\Bigl(\mathrm{Li}_s(\lambda_j^{m})-\zeta(s)\lambda_j^{m}\Bigr)\; },
\]
其中 \(\mathrm{Li}_s(z)=\sum_{q\ge 1}z^{q}/q^{s}\)（\(|z|<1\)）为 polylog，\(\zeta(s)\) 为 Riemann \(\zeta\) 函数。
并且 \(\Xi_M(s)\) 作为 \(s\) 的函数在 \(s=1\) 处至多有一简单极点，其留数为
\[
\boxed{\;
\mathrm{Res}_{s=1}\,\Xi_M(s)
=\sum_{j=2}^{d}\log(1-\lambda_j)
=\log\det\nolimits'\!\left(I-\frac{M}{\rho}\right)
=-\log C(M)\; }.
\]
\end{theorem}
\begin{proof}
由定义 \ref{def:finite-part-cyclic-lift-psi} 与幂级数恒等式 \(-\log(1-x)=\sum_{m\ge 1}x^m/m\) 得
\[
\mathcal{E}_q(M)=-\sum_{j=2}^{d}\log\!\left(\frac{1-\lambda_j^{q}}{1-\lambda_j}\right)
=\sum_{j=2}^{d}\sum_{m\ge 1}\frac{1}{m}\bigl(\lambda_j^{qm}-\lambda_j^{m}\bigr),
\]
其中交换求和由 \(|\lambda_j|<1\) 给出的绝对收敛保证。
对 \(q\) 作 Dirichlet 加权求和并交换求和次序，
\[
\sum_{q\ge 1}\frac{\lambda^{qm}}{q^s}=\mathrm{Li}_s(\lambda^m),
\qquad
\sum_{q\ge 1}\frac{1}{q^s}=\zeta(s),
\]
即得所示 polylog 公式（\(\Re(s)>1\) 下绝对收敛）。
由于 \(|\lambda_j|<1\)，\(\mathrm{Li}_s(\lambda_j^m)\) 关于 \(s\) 全纯；唯一可能的极点来自 \(\zeta(s)\)，其在 \(s=1\) 的留数为 \(1\)。
因此
\[
\mathrm{Res}_{s=1}\,\Xi_M(s)
=-\sum_{j=2}^{d}\sum_{m\ge 1}\frac{\lambda_j^{m}}{m}
=\sum_{j=2}^{d}\log(1-\lambda_j)
=\log\det\nolimits'\!\left(I-\frac{M}{\rho}\right)
=-\log C(M).
\]
证毕。
\end{proof}

\begin{proposition}[倍数求和变换的 Möbius 反演]\label{prop:finite-part-cyclic-lift-mobius-inversion}
在定义 \ref{def:finite-part-cyclic-lift-psi} 的记号下，令
\[
B_q(M):=\frac{1}{q}\Psi_q(M)\qquad(q\ge 1).
\]
则对任意 \(n\ge 1\) 成立精确反演式
\[
\boxed{\;
\frac{u_n(M)}{n}=\sum_{t\ge 1}\mu(t)\,B_{tn}(M)\; }.
\]
\end{proposition}
\leanverified{paper\_etds\_finite\_part\_cyclic\_lift\_mobius\_inversion}
\leanverified{paper\_finite\_part\_cyclic\_lift\_mobius\_inversion}
\begin{proof}
由命题 \ref{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}，
\[
B_q(M)=\frac{1}{q}\sum_{m\ge 1}\frac{1}{m}u_{qm}(M)=\sum_{m\ge 1}\frac{u_{qm}(M)}{qm}.
\]
定义 \(g(n):=u_n(M)/n\)，则 \(B_q(M)=\sum_{m\ge 1}g(qm)\) 为按倍数求和的 Dirichlet 变换。
对该变换作标准 Möbius 反演得
\(
g(n)=\sum_{t\ge 1}\mu(t)\,B_{tn}(M)
\)。
由于 \(|g(n)|\le (d-1)\Lambda_{\mathrm{rel}}(M)^n/n\)，右端绝对收敛。证毕。
\end{proof}

\begin{theorem}[仅由循环 lift 常数确定非 Perron 谱：\(q\)-轴可反演的谱可识别性]\label{thm:finite-part-cyclic-lift-spectrum-identifiability}
在定理 \ref{thm:finite-part-cyclic-lift-reduced-constant-closed} 的设定下，给定全体常数序列 \(\{C(M^{\langle q\rangle})\}_{q\ge 1}\)，则：
\begin{enumerate}
  \item 由定义 \ref{def:finite-part-cyclic-lift-psi} 与命题 \ref{prop:finite-part-cyclic-lift-mobius-inversion}，可逐 \(n\) 精确恢复归一化幂和序列 \(\{u_n(M)\}_{n\ge 1}\)；
  \item 序列 \(u_n(M)=\sum_{j=2}^{d}\lambda_j^n\) 是有限指数和，满足有限阶线性递推；存在最小阶 \(r\le d-1\) 使得 \(\{u_n(M)\}\) 由前 \(2r\) 项唯一决定其全部参数 \(\{\lambda_j\}_{j=2}^{d}\)（含代数重数），该可识别性与 \eqref{eq:xi-scan-fingerprint-one-sided-exponential-sum} 的有限原子谱指纹恢复同型；
  \item 因而可唯一确定约化特征多项式 \(\prod_{j=2}^{d}(t-\lambda_j)\) 以及约化行列式 \(\det\nolimits'(I-tM/\rho)\)。
\end{enumerate}
\end{theorem}
\leanverified{paper\_finite\_part\_cyclic\_lift\_spectrum\_identifiability}
\begin{proof}
第一点已由命题 \ref{prop:finite-part-cyclic-lift-mobius-inversion} 给出。
第二点属于 Prony--Hankel 机制的标准可识别性：有限指数和等价于 Hankel 矩阵有限秩，从而存在最小线性递推；由递推系数（或等价的最小多项式）得到 \(\{\lambda_j\}\)（含重数），并由前 \(2r\) 项唯一确定递推。
第三点由 \(\lambda_j=\mu_j/\rho\) 与多项式系数由根集确定直接推出。证毕。
\end{proof}

\begin{theorem}[\(q\)-轴极限提取谱半径比]\label{thm:finite-part-cyclic-lift-limsup-spectral-gap}
在定义 \ref{def:finite-part-cyclic-lift-psi} 的记号下成立
\[
\boxed{\;
\limsup_{q\to\infty}\bigl|\Psi_q(M)\bigr|^{1/q}
=\Lambda_{\mathrm{rel}}(M)\; }.
\]
\end{theorem}
\leanverified{paper\_etds\_finite\_part\_cyclic\_lift\_limsup\_spectral\_gap}
\leanverified{paper\_finite\_part\_cyclic\_lift\_limsup\_spectral\_gap}
\begin{proof}
由命题 \ref{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}，
\(
\Psi_q(M)=u_q(M)+O(\Lambda_{\mathrm{rel}}(M)^{2q})
\)。
于是
\[
\limsup_{q\to\infty}|\Psi_q(M)|^{1/q}
=\limsup_{q\to\infty}|u_q(M)|^{1/q},
\]
因为误差项的指数率严格小于 \(\Lambda_{\mathrm{rel}}(M)\)。
\par
令 \(\Lambda:=\Lambda_{\mathrm{rel}}(M)\) 并将达到模长的归一化谱点写为 \(\eta_1,\dots,\eta_J\)（\(|\eta_j|=1\)，按重数计），则
\(
 u_q(M)=\Lambda^{q}\sum_{j=1}^{J}\eta_j^{q}+O((\Lambda')^q)
\)
其中 \(\Lambda'<\Lambda\)。
设 \(t_q:=\sum_{j=1}^{J}\eta_j^{q}\)。
用平方模的 Ces\`aro 平均可得
\[
\lim_{N\to\infty}\frac{1}{N}\sum_{q=1}^{N}|t_q|^2
=\sum_{1\le i,j\le J}\ \lim_{N\to\infty}\frac{1}{N}\sum_{q=1}^{N}(\eta_i\overline{\eta_j})^{q}
=\sum_{\substack{1\le i,j\le J\\ \eta_i=\eta_j}}1\ >\ 0,
\]
从而 \(\limsup_{q\to\infty}|t_q|>0\)。
因此存在无穷多个 \(q\) 使 \(|u_q(M)|\ge c\,\Lambda^{q}\)（某个常数 \(c>0\)），推出
\(\limsup_{q\to\infty}|u_q(M)|^{1/q}\ge \Lambda\)；
反向上界 \(\le \Lambda\) 由 \(|u_q(M)|\le(d-1)\Lambda^{q}\) 立得。
合并即得结论。证毕。
\end{proof}

\begin{remark}[循环 lift 的常数层谱通道]\label{rem:finite-part-cyclic-lift-spectral-microscope}
命题 \ref{prop:finite-part-cyclic-lift-cyclotomic-sector}、\ref{prop:finite-part-cyclic-lift-torsion-product} 与命题 \ref{prop:finite-part-cyclic-lift-gcd-lcm} 给出循环 lift 在 \(\zeta\)-层面的 cyclotomic 扇区分解、扭点取值公式与算术可组合性；
定理 \ref{thm:finite-part-cyclic-lift-reduced-constant-closed} 则把常数层压缩为归一化非 Perron 谱的 \(q\)-次幂乘积指纹，并由命题 \ref{prop:finite-part-cyclic-lift-cyclotomic-divisor-mobius} 给出其在除数格上的 cyclotomic 分量分解与反演。
另一方面，倍数迹恒等式（推论 \ref{cor:finite-part-cyclic-lift-trace-sieve}）与 root-of-unity Fourier sieve（命题 \ref{prop:finite-part-cyclic-lift-root-unity-fourier-sieve}）表明常数层异常可在不引入长度轴的情形下沿扭点直接筛出幂和同余类分量；当 \(q\ge d\) 时，单层扭点数据已足以重构约化谱多项式与全部非 Perron 归一化谱点（定理 \ref{thm:finite-part-cyclic-lift-single-q-torsion-reconstruction} 与推论 \ref{cor:finite-part-cyclic-lift-zeta-torsion-determines-spectrum}）。
最后，polylog--Dirichlet 母函数 \(\Xi_M\)（定理 \ref{thm:finite-part-cyclic-lift-polylog-dirichlet-xi}）把该离散常数层通道进一步封装为一条具有唯一简单极点的谱 \(L\) 型对象，其留数精确等于 \(\log\det\nolimits'(I-M/\rho)\)。
\end{remark}
