\paragraph{次主模态的内生相位与振荡周期（可审计）}
命题 \ref{prop:chi-slice-time-correlation} 给出相关的指数包络尺度 $\rho_{\mathrm{corr}}(t)=\Lambda(t)/\lambda_1(t)$。在有限维情形中，该指数衰减由一个或一对次主特征值主导：若 $\lambda_2(t)$ 不为正实数，则相关函数除衰减包络外还会携带一个由 $\arg(\lambda_2(t))$ 控制的振荡因子。

\begin{remark}[次主相位与内生振荡周期]
\label{prop:chi-slice-subleading-phase}
在命题 \ref{prop:chi-slice-time-correlation} 的设定下，取
\[
\lambda_1(t):=\rho\bigl(M_{\theta(t)}\bigr)>0,
\qquad
\lambda_2(t)\in\mathrm{spec}(M_{\theta(t)})\setminus\{\lambda_1(t)\}
\]
为满足 $|\lambda_2(t)|=\Lambda(t)$ 的次主特征值（若出现共轭成对则任选其一）。定义
\[
\rho_2(t):=\frac{|\lambda_2(t)|}{\lambda_1(t)}\in[0,1),
\qquad
\omega(t):=\arg\bigl(\lambda_2(t)\bigr)\in(-\pi,\pi],
\]
并在 $\omega(t)\ne 0$ 时定义内生振荡周期
\[
T_{\mathrm{osc}}(t):=\frac{2\pi}{|\omega(t)|}.
\]
则对任意中心化的有限状态可观测量 $f$，其相关函数的谱表示中存在次主项
\[
\left(\frac{\lambda_2(t)}{\lambda_1(t)}\right)^n
=\rho_2(t)^n\,e^{i n\omega(t)},
\]
从而当 $\omega(t)\ne 0$ 时，相关衰减在指数包络 $\rho_2(t)^n$ 上叠加一个内生相位频率 $\omega(t)$；当 $\omega(t)=\pi$（即 $\lambda_2(t)<0$）时出现二步交替振荡。
\end{remark}

\begin{remark}[留数与有限部的分层：黄金幂量子与 Möbius 尾的可分离注入]\label{rem:chi-ldp-residue-finite-part-layering}
由恒等式 \(\varphi^{8}=F_{8}\varphi+F_{7}=21\varphi+13\) 立得
\[
\boxed{\; C=\frac{47+21\sqrt5}{100}=\frac{13+21\varphi}{50}=\frac{\varphi^{8}}{50}\; }.
\]
另一方面，真实输入核在主极点处具有精确 \(\zeta\)-分解
\(
\zeta_M=\zeta_{\mathrm{in}}\zeta_{\mathrm{twist}}\zeta_{\mathrm{triv}}
\)
（表 \ref{tab:real_input_40_finite_part_split}），其中主极点仅属于 \(\zeta_{\mathrm{in}}\)，并且
\[
C_{\mathrm{tot}}=C_{\mathrm{in}}\zeta_{\mathrm{twist}}(z_\star)\zeta_{\mathrm{triv}}(z_\star),
\qquad
\log\mathfrak{M}_{\mathrm{tot}}=\log\mathfrak{M}_{\mathrm{in}}+\log\mathfrak{M}_{\mathrm{twist}}+\log\mathfrak{M}_{\mathrm{triv}}.
\]
对解析因子 \(f\in\{\zeta_{\mathrm{twist}},\zeta_{\mathrm{triv}}\}\)（在 \(z=z_\star\) 解析且非零），其对有限部常数的贡献可按 Möbius--Abel 口径写为
\[
\boxed{\;
\log\mathfrak{M}_f
=
\log f(z_\star)+\sum_{k\ge 2}\frac{\mu(k)}{k}\log f(z_\star^{\,k})\; }.
\]
因此，留数常数 \(C\)（可完全代数化，且在本核上被锁定为 \(\varphi^{8}\) 的有理缩放）提供“主尺度/体积”的离散证书，
而有限部常数的其余信息则以 Möbius 尾的采样 \(\{\log f(z_\star^{\,k})\}_{k\ge 2}\) 形式被分层注入；这使“体积（留数）—熵（有限部）”的关系可在同一可审计接口下被无损拆分与对比。
此外，表 \ref{tab:real_input_40_finite_part_split} 中 \(\zeta_{\mathrm{twist}}(z_\star)=(1+\sqrt5)/4=\varphi/2\) 的精确值在主极点处携带一个不可约的 \(1/4\) 规一化因子，从而把 twist 通道的分数化归一化与主留数的有理规一化并置为同一分解口径下的硬常数接口。
\end{remark}

\paragraph{平凡层坍缩 \texorpdfstring{$+$}{+} 机制纯度常数（真机制不变量）}
在该核上
\[
\Delta(z)=\det(I-zM)=(1-z)^2(1+z)^3(1-3z+z^2)(1+z-z^2)
\]
给出一个天然的“平凡层/机制层”分离：\((1-z)^2(1+z)^3\) 对应边界固定点与符号平凡模态，而两条二次因子承载真正的机制块。
据此可定义一个剔除平凡漂移后的“机制纯度常数”，用于跨核/跨扭曲的可比对齐。

\begin{definition}[机制纯度常数：剥离平凡因子]\label{def:real-input-40-mechanism-purity-constant}
令 \(z_\star=\lambda^{-1}=\varphi^{-2}\)，并记
\[
\Delta_{\mathrm{triv}}(z):=(1-z)^2(1+z)^3,\qquad
\Delta_{\mathrm{mech}}(z):=(1-3z+z^2)(1+z-z^2).
\]
定义\textbf{真机制贡献（机制纯度常数）}为
\[
\boxed{\;
\log\mathfrak{M}_{\mathrm{mech}}
:=
-\sum_{m\ge 2}\frac{\mu(m)}{m}\log \Delta_{\mathrm{mech}}(z_\star^{\,m})
\; }.
\]
并定义\textbf{平凡层贡献}为
\[
\boxed{\;
\log\mathfrak{M}_{\mathrm{triv}}
:=\log\mathfrak{M}-\log\mathfrak{M}_{\mathrm{mech}}.\;}
\]
\end{definition}

\begin{proposition}[平凡层显式坍缩与二核机制常数]\label{prop:real-input-40-triv-mech-split}
在定义 \ref{def:real-input-40-mechanism-purity-constant} 的记号下，有严格分解
\[
\boxed{\ \log\mathfrak{M}=\log\mathfrak{M}_{\mathrm{triv}}+\log\mathfrak{M}_{\mathrm{mech}}\ }.
\]
其中平凡层贡献完全闭式可写为
\[
\boxed{\;
\log\mathfrak{M}_{\mathrm{triv}}
=\log C-z_\star+3z_\star^2+2\log(1-z_\star)+3\log(1+z_\star)\; }.
\]
等价地，\(\mathfrak{M}_{\mathrm{mech}}:=\exp(\log\mathfrak{M}_{\mathrm{mech}})\) 与附录中的二次 carry-常数一致（定理 \ref{thm:real-input-40-carry-constant}），并且上述分解正是“二核压缩”恒等式（命题 \ref{prop:real-input-40-mertens-two-series}）。

\paragraph{可计算流程（与“离散化—正则化—有限部”同构）}
给定一个有限图移位的邻接矩阵 $M$（或加权矩阵族 $M_\theta$），并固定截断阶 $K\ge 2$，可按下述步骤计算有限部常数：
\begin{enumerate}
  \item \textbf{离散化：}由图/转导器得到矩阵 $M$（或 $M_\theta$），并定义 $\zeta(z)=1/\det(I-zM)$（或 $\zeta_\theta(z)=1/\det(I-zM_\theta)$）。
  \item \textbf{主尺度：}计算 Perron--Frobenius 主特征值 $\lambda=\rho(M)$（或 $\lambda(\theta)=\rho(M_\theta)$），并令 $z_\star=1/\lambda$（或 $z_\star(\theta)=1/\lambda(\theta)$）。
  \item \textbf{留数常数：}计算主极点留数常数
  \[
  C:=\lim_{z\to z_\star}(1-\lambda z)\zeta(z)
  \qquad
  (\text{或 }C(\theta):=\lim_{z\to z_\star(\theta)}(1-\lambda(\theta)z)\zeta_\theta(z)).
  \]
  在“简单极点”情形，$C$ 亦可由行列式导数或特征值乘积表达为闭式（见附录 \ref{app:real-input-40-finite-part}）。
  \item \textbf{正则化（Abel 采样）：}对 $k=2,\dots,K$，在收敛圆盘内采样 $\log\zeta(z_\star^k)$（或 $\log\zeta_\theta(z_\star(\theta)^k)$）。
  \item \textbf{有限部：}用 Möbius 尾的截断和近似
  \[
  \log\mathfrak{M}\approx \log C+\sum_{k=2}^{K}\frac{\mu(k)}{k}\log\zeta(z_\star^k)
  \quad
  (\text{或 }\log\mathfrak{M}(\theta)\approx \log C(\theta)+\sum_{k=2}^{K}\frac{\mu(k)}{k}\log\zeta_\theta(z_\star(\theta)^k)).
  \]
  \item \textbf{Mertens 常数输出：}定义
  \[
  \mathsf{M}:=\gamma+\log\mathfrak{M},
  \qquad
  \mathsf{M}(\theta):=\gamma+\log\mathfrak{M}(\theta).
  \]
\end{enumerate}
由于 $|z_\star|<1$，故 $z_\star^k$ 指数衰减，且 $\log\zeta(z)=O(z)$（$z\to 0$）使得 $\log\zeta(z_\star^k)$ 在 $k$ 上快速趋零，从而该流程在数值上具有很强的稳定性与快速收敛性。
并且当盘内采样存在加性误差时，有限部常数的漂移仅以 \(1/k\) 的 Möbius 权重线性聚合并被指数尾和统一吸收（命题 \ref{prop:real-input-40-logM-stability-functional}）。

\paragraph{orbit--Mertens 乘积（把 $\mathsf{M}$ 变成可对比标量）}
定义长度截断的 prime-orbit Mertens 乘积
\[
\mathcal{P}(N):=\prod_{\substack{\tau\ \mathrm{primitive}\\|\tau|\le N}}\left(1-\lambda^{-|\tau|}\right)^{-1}.
\]
取对数并用 $-\log(1-x)=x+O(x^2)$ 得
\[
\log\mathcal{P}(N)=\sum_{n\le N}\frac{p_n}{\lambda^n}+O(1)
=\log N+\mathsf{M}+o(1),
\]
因此
\[
\mathcal{P}(N)\sim e^{\mathsf{M}}\,N=e^{\gamma}\mathfrak{M}\,N.
\]
若用“轨道范数” $N(\tau):=\lambda^{|\tau|}$ 改写，并令 $x=\lambda^N$（等价于 $N=\log x/\log\lambda$），则得到等价的 $\log x$ 型主项
\[
\prod_{N(\tau)\le x}\left(1-\frac{1}{N(\tau)}\right)^{-1}
\sim \frac{e^{\gamma}\mathfrak{M}}{\log\lambda}\,\log x.
\]

\begin{remark}[三通道二阶耦合的“中介结构”]
由 Hessian 的非对角项可见二阶耦合具有明显层级：
\[
H_{-2}>H_{e-}>H_{e2}.
\]
这意味着“负携带--碰撞”耦合最强，而“输出--碰撞”最弱；输出与负携带的相关性可被理解为在碰撞通道上被放大并重分配所致。该层级关系是数值可审计的（见附录 \ref{app:real-input-40-logm-theta} 的复现脚本输出）。
\end{remark}

\paragraph{arity-charge 的同一接口并入}
定义 arity-charge
\[
\chi:=\varphi_e-\varphi_2\in\{-1,0,1\}.
\]
更一般地，若把 $\chi$ 作为一条额外势并入加权，则它只引入线性重参数化（无需扩展状态空间）：
\[
P(\theta_e,\theta_-,\theta_2,\theta_\chi)=P(\theta_e+\theta_\chi,\theta_-,\theta_2-\theta_\chi).
\]
在三参数压力 $P(\theta_e,\theta_-,\theta_2)=\log\rho(M_\theta)$ 下，对 $\chi$ 的线性加权对应一维切线
\[
P_\chi(t):=P(t,0,-t).
\]
因此 $\chi$ 的典型密度与方差密度由
\(
P_\chi'(0),P_\chi''(0)
\)
读出，并可与同一族 $\log\mathfrak{M}(\theta)$ 在该方向的切片一起作为“素数影子”的分叉维度；具体数值与可复现计算见附录 \ref{app:real-input-40-logm-theta} 与
\texttt{scripts/exp\_sync\_kernel\_real\_input\_40\_logM\_chi.py}。

\paragraph{\texorpdfstring{$\chi$}{chi} 的一维切片相图（可计算流程）}
固定
\(
\theta(t)=(t,0,-t)
\)
并令
\[
P_\chi(t):=P(\theta(t))=\log\rho(M_{\theta(t)}),
\qquad
\log\mathfrak{M}_\chi(t):=\log\mathfrak{M}(\theta(t)).
\]
对每个 $t$，按上文“可计算流程”的步骤计算 $\lambda(t)=e^{P_\chi(t)}$、$z_\star(t)=\lambda(t)^{-1}$、$C(t)$，并用截断 Möbius 尾近似
\[
\log\mathfrak{M}_\chi(t)\approx \log C(t)+\sum_{k=2}^{K}\frac{\mu(k)}{k}\log\zeta_{\theta(t)}(z_\star(t)^k).
\]
该一维切片在 $t=0$ 的统计量由压力导数给出：
\[
\mathbb{E}[\chi]=P_\chi'(0),\qquad \mathrm{Var}_\infty(\chi)=P_\chi''(0),
\]
并且在真实输入核上（附录 \ref{app:real-input-40-logm-theta}）有闭式与数值
\[
\mathbb{E}[\chi]=\frac{17-5\sqrt5}{20}\approx 0.2909830056,
\qquad
\mathrm{Var}_\infty(\chi)\approx 0.0841438530.
\]
若记三维 Hessian（协方差密度）为 $H=\nabla^2P(0)$，则
\[
\mathrm{Var}_\infty(\chi)=H_{ee}+H_{22}-2H_{e2},
\qquad
\mathrm{Cov}_\infty(\chi,\varphi_-)=H_{e-}-H_{-2}\approx -0.0831619.
\]
