\subsubsection{主极点留数与约化行列式：去 Perron 因子的常数层谱指纹}\label{subsec:finite-part-reduced-determinant-residue}
本小节把 Abel/Hadamard 有限部常数分解中出现的主极点留数项 \(C(\theta)\)（见 \ref{def:abel-finite-part} 与命题 \ref{prop:abel-finite-part-mobius-zeta}）
化为一个纯谱对象：\emph{去 Perron 因子后的约化行列式}。
该表述在常数层上提供一个与 moment 阶数兼容的“乘法偏差”指纹，从而可将常数漂移的差异直接还原为有限谱的块间不一致。

\begin{proposition}[主极点留数的谱乘积公式]\label{prop:finite-part-residue-reduced-determinant}
设 \(A\) 为原始（primitive）非负矩阵，Perron 根为 \(\rho=\rho(A)>0\)，并定义
\(
\zeta_A(z):=\det(I-zA)^{-1}
\)。
记主极点 \(z_\star:=1/\rho\)，并令
\[
C(A):=\lim_{z\to z_\star}(1-\rho z)\,\zeta_A(z)
\]
为主极点留数常数（简单极点情形下存在且 \(C(A)>0\)）。
设 \(A\) 的全部特征值按代数重数为 \(\rho=\mu_1,\mu_2,\dots,\mu_d\)。
则有显式闭式
\[
\boxed{\;
C(A)=\prod_{j=2}^{d}\left(1-\frac{\mu_j}{\rho}\right)^{-1}\; }.
\]
等价地，若定义\textbf{约化行列式}
\[
\det\nolimits'\!\left(I-\frac{A}{\rho}\right)
\;:=\;
\prod_{j=2}^{d}\left(1-\frac{\mu_j}{\rho}\right),
\]
则
\[
\boxed{\;
C(A)=\det\nolimits'\!\left(I-\frac{A}{\rho}\right)^{-1}\; }.
\]
\leanverified{paper\_reduced\_det\_golden\_mean}
\leanverified{goldenMean\_trace\_eq\_one}
\leanverified{goldenMean\_det\_eq\_neg\_one}
\leanverified{goldenMean\_det\_pow\_general}
\end{proposition}
\begin{proof}
由特征值分解
\[
\det(I-zA)=\prod_{j=1}^{d}(1-z\mu_j)=(1-\rho z)\prod_{j=2}^{d}(1-z\mu_j),
\]
从而
\[
\zeta_A(z)=\frac{1}{1-\rho z}\cdot \prod_{j=2}^{d}(1-z\mu_j)^{-1}.
\]
令 \(z\to z_\star=1/\rho\) 得
\(
C(A)=\prod_{j=2}^{d}(1-\mu_j/\rho)^{-1}
\)。
约化行列式的表述仅为记号改写。证毕。
\end{proof}

\begin{corollary}[moment 阶的约化行列式比值指纹]\label{cor:finite-part-moment-anomaly-reduced-determinant-ratio}
对任意整数 \(q\ge 2\)，令 \(A^{(q)}_\theta\) 表示某个 \(q\)-重 moment-kernel 的原始 realization（参数 \(\theta\) 固定），并记
\(
\rho_q=\rho(A^{(q)}_\theta)
\)、
\(
z_{\star,q}=\rho_q^{-1}
\)，以及其主极点留数常数
\[
C_q(\theta):=\lim_{z\to z_{\star,q}}(1-\rho_q z)\,\det(I-zA^{(q)}_\theta)^{-1}.
\]
定义常数层的 moment 偏差（去指数主尺度后的乘法偏差）为
\[
\mathsf{Anom}^{(q)}_{\mathrm{res}}(\theta):=\log C_q(\theta)-q\log C_1(\theta).
\]
则
\[
\boxed{\;
\mathsf{Anom}^{(q)}_{\mathrm{res}}(\theta)
=
-\log\det\nolimits'\!\left(I-\frac{A^{(q)}_\theta}{\rho_q}\right)
\;+\;
q\,\log\det\nolimits'\!\left(I-\frac{A^{(1)}_\theta}{\rho_1}\right)\; }.
\]
从而其指数化满足约化行列式比值闭式
\[
\boxed{\;
\exp\bigl(\mathsf{Anom}^{(q)}_{\mathrm{res}}(\theta)\bigr)
\;=\;
\frac{\det\nolimits'\!\left(I-\frac{A^{(1)}_\theta}{\rho_1}\right)^{q}}
{\det\nolimits'\!\left(I-\frac{A^{(q)}_\theta}{\rho_q}\right)}\; }.
\]
\leanverified{moment\_anomaly\_ratio\_proxy}
\end{corollary}
\begin{proof}
对 \(A^{(q)}_\theta\) 与 \(A^{(1)}_\theta\) 分别应用命题 \ref{prop:finite-part-residue-reduced-determinant}，
并取对数相减即得。证毕。
\end{proof}

当 moment-kernel realization 具有由 \(q\)-重 fiber-product 语义诱导的自然置换对称（因子置换 \(\mathfrak{S}_q\) 作用）时，
约化行列式在表示论意义上可进一步分解为一组不可约通道的乘积，从而把常数层漂移提升为逐通道的可审计坐标。

\begin{proposition}[\texorpdfstring{$(\mathfrak{S}_q)$}{(Sq)}-等变约化行列式的通道因子化]\label{prop:finite-part-reduced-determinant-sq-channel-factorization}
固定整数 \(q\ge 2\) 与参数 \(\theta\)。
设 \(A^{(q)}_\theta\) 的状态空间携带一个线性 \(\mathfrak{S}_q\)-作用，使其与 \(A^{(q)}_\theta\) 对易；等价地，存在 \(\mathfrak{S}_q\)-等变分解
\[
A^{(q)}_\theta\ \cong\ \bigoplus_{\lambda\vdash q}\bigl(B_{q,\lambda}\otimes I_{f^\lambda}\bigr),
\qquad f^\lambda:=\dim S^\lambda,
\]
其中 \(\lambda\) 遍历 \(q\) 的分拆、\(S^\lambda\) 为相应 Specht 模，\(B_{q,\lambda}\) 作用于 \(\lambda\)-等变多重度空间。
记 \(\rho_q:=\rho(A^{(q)}_\theta)\)。
则有严格恒等式
\[
\det\!\left(I-\frac{A^{(q)}_\theta}{\rho_q}\right)
\;=\;
\prod_{\lambda\vdash q}\det\!\left(I-\frac{B_{q,\lambda}}{\rho_q}\right)^{f^\lambda}.
\]
进一步，由原始性可知 Perron 特征值 \(\rho_q\) 简单且其特征向量 \(\mathfrak{S}_q\)-不变，
从而归一化谱点 \(1\) 仅落在平凡分拆 \(\lambda=(q)\) 的通道中；因此约化行列式满足
\[
\det\nolimits'\!\left(I-\frac{A^{(q)}_\theta}{\rho_q}\right)
=
\det\nolimits'\!\left(I-\frac{B_{q,(q)}}{\rho_q}\right)\cdot
\prod_{\lambda\vdash q,\ \lambda\neq(q)}
\det\!\left(I-\frac{B_{q,\lambda}}{\rho_q}\right)^{f^\lambda}.
\]
相应地，主极点留数常数分解为
\[
\log C_q(\theta)
=
\log C_{q,(q)}(\theta)
\;+\;
\sum_{\lambda\vdash q,\ \lambda\neq(q)}
f^\lambda\,\log\det\!\left(I-\frac{B_{q,\lambda}}{\rho_q}\right)^{-1},
\]
其中
\(
C_{q,(q)}(\theta):=\det\nolimits'\!\left(I-\frac{B_{q,(q)}}{\rho_q}\right)^{-1}
\)。
\leanverified{symmetric\_group\_orders}
\end{proposition}
\begin{proof}
由等变分解与行列式在直和/张量下的乘法性，
\[
\det(I-zA^{(q)}_\theta)=\prod_{\lambda\vdash q}\det(I-zB_{q,\lambda})^{f^\lambda},
\]
取 \(z=\rho_q^{-1}\) 得第一式。
又因 \(A^{(q)}_\theta\) 原始，Perron 特征值简单且对应特征向量处处严格正，从而在置换作用下仍不变；
故 \(1\)（在 \(A^{(q)}_\theta/\rho_q\) 的谱中）仅出现于平凡通道，非平凡通道均满足 \(\rho(B_{q,\lambda})<\rho_q\)，从而 \(\det(I-B_{q,\lambda}/\rho_q)\neq 0\)。
于是约化仅发生在 \(\lambda=(q)\) 的因子上，得到约化行列式的分解式；取对数即得对 \(\log C_q(\theta)\) 的表达。证毕。
\end{proof}

\begin{corollary}[常数层 moment 偏差的不可约通道加性分解]\label{cor:finite-part-moment-anomaly-channel-additivity}
在命题 \ref{prop:finite-part-reduced-determinant-sq-channel-factorization} 的设定下，
对每个非平凡通道 \(\lambda\vdash q\)、\(\lambda\neq(q)\) 定义
\[
\Phi_{q,\lambda}(\theta)
:=
\log\det\!\left(I-\frac{B_{q,\lambda}}{\rho_q}\right)^{-1}.
\]
则常数层 moment 偏差（定义 \ref{cor:finite-part-moment-anomaly-reduced-determinant-ratio}）满足
\[
\mathsf{Anom}^{(q)}_{\mathrm{res}}(\theta)
=
\Bigl(\log C_{q,(q)}(\theta)-q\log C_1(\theta)\Bigr)
\;+\;
\sum_{\lambda\vdash q,\ \lambda\neq(q)} f^\lambda\,\Phi_{q,\lambda}(\theta).
\]
因此常数层漂移可被严格分解为平凡通道的尺度差项与各非平凡 \(\mathfrak{S}_q\) 通道的可审计贡献之和。
\leanverified{anomaly\_channel\_count}
\end{corollary}
\begin{proof}
由命题 \ref{prop:finite-part-reduced-determinant-sq-channel-factorization} 的 \(\log C_q(\theta)\) 分解式代入
\(
\mathsf{Anom}^{(q)}_{\mathrm{res}}(\theta)=\log C_q(\theta)-q\log C_1(\theta)
\)
即得。证毕。
\end{proof}

\begin{theorem}[约化 resolvent（群逆）给出的常数层梯度审计公式]\label{thm:finite-part-reduced-determinant-group-inverse-gradient}
令 \(A(\theta)\) 为一族原始非负矩阵，\(\theta\in\RR^k\) 光滑，Perron 根记为 \(\rho(\theta)\)。
取左右 Perron 向量 \(r(\theta),\ell(\theta)\) 满足
\(
A(\theta)r(\theta)=\rho(\theta)r(\theta)
\)、
\(
\ell(\theta)^\top A(\theta)=\rho(\theta)\ell(\theta)^\top
\)
并归一化为 \(\ell(\theta)^\top r(\theta)=1\)。
记谱投影 \(P(\theta):=r(\theta)\ell(\theta)^\top\)，并定义约化 resolvent（群逆/ Drazin 逆）
\[
R(\theta):=\Bigl(I-\frac{A(\theta)}{\rho(\theta)}\Bigr)^{\#},
\qquad
\Bigl(I-\frac{A}{\rho}\Bigr)R=R\Bigl(I-\frac{A}{\rho}\Bigr)=I-P,\quad RP=PR=0.
\]
定义常数层函数
\[
\Phi(\theta):=\log C(A(\theta))
=
-\log\det\nolimits'\!\Bigl(I-\frac{A(\theta)}{\rho(\theta)}\Bigr).
\]
则对任意方向导数 \(\partial_\theta\) 成立
\[
\boxed{\;
\partial_\theta \Phi(\theta)
=
\mathrm{Tr}\!\left(
R(\theta)\left(
\frac{\partial_\theta A(\theta)}{\rho(\theta)}
-\frac{\partial_\theta\rho(\theta)}{\rho(\theta)^2}A(\theta)
\right)\right),\qquad
\partial_\theta\rho(\theta)=\ell(\theta)^\top(\partial_\theta A(\theta))\,r(\theta)\; }.
\]
若进一步 \(A(\theta)=A_0+\sum_{j=1}^{k}\theta_jE_j\) 为线性边权参数化，则
\[
\boxed{\;
\partial_{\theta_j}\Phi(\theta)
=
\frac{1}{\rho(\theta)}\,\mathrm{Tr}\!\bigl(R(\theta)E_j\bigr)
-\frac{\ell(\theta)^\top E_j r(\theta)}{\rho(\theta)^2}\,
\mathrm{Tr}\!\bigl(R(\theta)A(\theta)\bigr)\; }.
\]
\leanverified{group\_inverse\_vieta\_proxy}
\end{theorem}
\begin{proof}
记 \(B(\theta):=I-A(\theta)/\rho(\theta)\)。
由于 \(A(\theta)\) 原始，\(B(\theta)\) 在 Perron 方向上具有单零特征值、其余谱点均非零，故群逆 \(B(\theta)^{\#}=R(\theta)\) 存在并满足所列恒等式。
在约化子空间 \((I-P)\RR^d\) 上，\(\det\nolimits'(B(\theta))\) 等于 \(B(\theta)\) 的非零谱乘积；其对数微分满足
\(
\partial_\theta\log\det\nolimits'(B)=\mathrm{Tr}(B^{\#}\partial_\theta B)
\)。
对 \(\Phi(\theta)=-\log\det\nolimits'(B(\theta))\) 取导，并代入
\(
\partial_\theta B=-(\partial_\theta A)/\rho+(\partial_\theta\rho)A/\rho^2
\)
即得第一式；第二式仅为将 \(\partial_{\theta_j}A=E_j\) 代回后的展开。证毕。
\end{proof}

\begin{corollary}[通道化常数项的逐分量消去与最小消去阶]\label{cor:finite-part-channel-elimination-lcm}
\leanverified{paper\_finite\_part\_channel\_elimination\_lcm}
在命题 \ref{prop:finite-part-reduced-determinant-sq-channel-factorization} 的设定下，
可将常数层漂移提升为逐通道向量
\[
\boldsymbol{\Phi}_q(\theta):=\bigl(\Phi_{q,\lambda}(\theta)\bigr)_{\lambda\vdash q,\ \lambda\neq(q)}.
\]
于是 \(\boldsymbol{\Phi}_q(\theta)=0\) 当且仅当对所有 \(\lambda\neq(q)\) 都有 \(\Phi_{q,\lambda}(\theta)=0\)。
若这些通道量取值于某一环面型交换群（例如 \((\RR/2\pi\ZZ)^r\) 的坐标分量），并对每个通道定义其阶 \(\mathrm{ord}(\Phi_{q,\lambda}(\theta))\)（若存在），则使全部通道同时消去的最小正整数为
\[
\boxed{\;
q_\star=\mathrm{lcm}_{\lambda\vdash q,\ \lambda\neq(q)}\ \mathrm{ord}\bigl(\Phi_{q,\lambda}(\theta)\bigr)\; }.
\]
\end{corollary}
\begin{proof}
第一条为直积群的逐分量判别。
第二条为交换群中有限多个 torsion 元的共同消去阶的标准最小公倍数律。证毕。
\end{proof}

\begin{theorem}[指数缺陷与常数缺陷的局部可分离性：同一核上的双审计轴]\label{thm:finite-part-exp-fin-double-threshold-separation}
令 \(A(\theta)\) 为一族原始非负矩阵，\(\rho(\theta)\) 为 Perron 根，并记
\[
\Lambda(\theta):=\max\{|\mu|:\ \mu\in\mathrm{spec}(A(\theta)),\ \mu\neq \rho(\theta)\}.
\]
定义两类审计量：
\[
\Delta(\theta):=\log\!\Bigl(\frac{\Lambda(\theta)^2}{\rho(\theta)}\Bigr),
\qquad
\Phi(\theta):=\log C(A(\theta)).
\]
假设在某点 \(\theta_0\) 附近，\(\Lambda(\theta)\) 由一个简单非 Perron 特征值（或一对简单共轭特征值）实现，从而 \(\Delta\) 与 \(\Phi\) 在 \(\theta_0\) 处可微。
则以下两命题等价：
\begin{enumerate}
  \item 存在方向 \(v_{\mathrm{exp}},v_{\mathrm{fin}}\in T_{\theta_0}\RR^k\) 使
  \[
  D\Delta(\theta_0)[v_{\mathrm{fin}}]=0,\ \ D\Phi(\theta_0)[v_{\mathrm{fin}}]\neq 0;
  \qquad
  D\Phi(\theta_0)[v_{\mathrm{exp}}]=0,\ \ D\Delta(\theta_0)[v_{\mathrm{exp}}]\neq 0.
  \]
  \item 线性泛函 \(D\Delta(\theta_0)\) 与 \(D\Phi(\theta_0)\) 在对偶空间 \((T_{\theta_0}\RR^k)^\ast\) 中不共线。
\end{enumerate}
因此在一般参数化下，“平方根界/近-RH 指数门槛”（由 \(\Delta\) 度量）与“有限部常数漂移”（由 \(\Phi\) 度量）为同一核上的两条可独立操控审计轴；
其中 \(D\Phi\) 可由定理 \ref{thm:finite-part-reduced-determinant-group-inverse-gradient} 的约化 resolvent 迹型公式完全有限维化。
\leanverified{paper\_finite\_part\_exp\_fin\_double\_threshold\_separation}
\end{theorem}
\begin{proof}
若 \(D\Delta(\theta_0)\) 与 \(D\Phi(\theta_0)\) 不共线，则 \(\ker D\Delta(\theta_0)\not\subseteq \ker D\Phi(\theta_0)\) 且 \(\ker D\Phi(\theta_0)\not\subseteq \ker D\Delta(\theta_0)\)；
分别取 \(v_{\mathrm{fin}}\in \ker D\Delta(\theta_0)\setminus \ker D\Phi(\theta_0)\) 与 \(v_{\mathrm{exp}}\in \ker D\Phi(\theta_0)\setminus \ker D\Delta(\theta_0)\) 即得 (1)。
反之若 (1) 成立，则两核互不包含，从而两线性泛函不共线。证毕。
\end{proof}

\begin{corollary}[设计原则：指数门槛与常数漂移需由两套谱数据同时控制]\label{cor:finite-part-double-threshold-design-principle}
在定理 \ref{thm:finite-part-exp-fin-double-threshold-separation} 的设定下，
若目标是同时实现 \(\Delta(\theta)\le 0\)（平方根界）与 \(\Phi(\theta)=0\)（常数层无漂移），
则必须同时控制两套互不等价的谱灵敏度数据：一套由达界非 Perron 模态的微扰决定（控制 \(\Delta\) 的一阶项），另一套由约化 resolvent 迹型泛函决定（控制 \(\Phi\) 的一阶项）。
当 \(\mathfrak{S}_q\) 通道化可用时，后者进一步还原为逐通道 \(\Phi_{q,\lambda}\) 的消去（推论 \ref{cor:finite-part-channel-elimination-lcm}）。
\leanverified{paper\_finite\_part\_double\_threshold\_design\_principle}
\end{corollary}

\begin{proposition}[核版 RH 给出的留数常数压扁界：维数 \texorpdfstring{$\times$}{x} 平方根缺口]\label{prop:finite-part-residue-constant-rh-squeeze}
设 \(A\) 为原始非负矩阵，\(\rho=\rho(A)>1\)，维数 \(d:=\dim A\)，并以命题 \ref{prop:finite-part-residue-reduced-determinant} 的记号定义主极点留数常数 \(C(A)\)。
若核版 RH（平方根界）成立，即对一切非 Perron 特征值 \(\mu\neq\rho\) 有 \(|\mu|\le \rho^{1/2}\)，则有双侧界
\[
(1+\rho^{-1/2})^{-(d-1)}\ \le\ C(A)\ \le\ (1-\rho^{-1/2})^{-(d-1)}.
\]
特别地，当 \(\rho\ge 4\) 时（从而 \(\rho^{-1/2}\le \tfrac12\)），有粗界
\[
|\log C(A)|\ \le\ 2(d-1)\rho^{-1/2}.
\]
\leanverified{reduced\_det\_golden\_mean\_squeeze}
\end{proposition}

\begin{proof}
由命题 \ref{prop:finite-part-residue-reduced-determinant}，
\(
C(A)=\prod_{j=2}^{d}(1-\mu_j/\rho)^{-1}
\)（\(\mu_1=\rho\)）。
原始非负矩阵的谱关于共轭闭合，故 \(C(A)>0\) 且
\(
C(A)=\prod_{j=2}^{d}|1-\mu_j/\rho|^{-1}
\)。
在 \(|\mu_j|/\rho\le \rho^{-1/2}\) 下有
\(
1-\rho^{-1/2}\le |1-\mu_j/\rho|\le 1+\rho^{-1/2}
\)，逐项相乘即得双侧界。
当 \(\rho^{-1/2}\le\tfrac12\) 时，用 \(\log(1-x)^{-1}\le 2x\)（\(0<x\le\tfrac12\)）与 \(\log(1+x)\le x\le 2x\) 得
\[
-(d-1)\log(1+\rho^{-1/2})
\ \le\ \log C(A)\ \le\ (d-1)\log(1-\rho^{-1/2})^{-1},
\]
从而 \(|\log C(A)|\le 2(d-1)\rho^{-1/2}\)。证毕。
\end{proof}

\begin{theorem}[异常—RH—维数三岔：线性留数漂移迫使 RH 破裂或维数指数爆炸]\label{thm:finite-part-residue-drift-trichotomy}
\leanverified{paper\_finite\_part\_residue\_drift\_trichotomy}
对某个 moment-kernel 族 \(\{A^{(q)}_\theta\}_{q\ge 2}\)（\(\theta\) 固定），记
\(\rho_q:=\rho(A^{(q)}_\theta)\)、\(d_q:=\dim A^{(q)}_\theta\) 与 \(C_q(\theta):=C(A^{(q)}_\theta)\)。
假设存在常数 \(c>1\) 使 \(\rho_q\ge c^q\) 对所有充分大 \(q\) 成立，并假设维数次指数增长
\[
\log d_q=o(q)\qquad(q\to\infty).
\]
若核版 RH 对所有充分大 \(q\) 成立，则必有
\[
\log C_q(\theta)=o(q)\qquad(q\to\infty).
\]
因此，任一满足
\[
\liminf_{q\to\infty}\frac{1}{q}\,|\log C_q(\theta)|>0
\]
的“线性留数漂移”现象，都必然迫使下述二者之一发生：
\begin{enumerate}
  \item 存在某个有限 \(q\) 使核版 RH 失效；
  \item \(d_q\) 不能保持次指数增长，必须进入指数增长区间。
\end{enumerate}
\end{theorem}

\begin{proof}
由命题 \ref{prop:finite-part-residue-constant-rh-squeeze}，当 \(q\) 充分大使 \(\rho_q\ge 4\) 时有
\(
|\log C_q(\theta)|\le 2(d_q-1)\rho_q^{-1/2}
\le 2d_q\,c^{-q/2}
\)。
在 \(\log d_q=o(q)\) 下，右端为 \(o(q)\)，故 \(\log C_q(\theta)=o(q)\)。
若 \(\liminf_{q\to\infty}\frac{1}{q}|\log C_q(\theta)|>0\) 仍成立，则与上述 \(o(q)\) 结论矛盾；因此只能落入 (1) 或 (2)。证毕。
\end{proof}

\begin{corollary}[预言机—异常等价律：非零异常必以“有限见证”或“维数爆炸”出现]\label{cor:finite-part-oracle-anomaly-equivalence}
\leanverified{paper\_finite\_part\_oracle\_anomaly\_equivalence}
在本文 \(\mathbf{PW}\) 语义中，设某个“异常量”在 \(q\)-重 moment 放大下呈线性规模，并且其谱侧见证可归约为留数常数漂移的线性下界，即存在常数 \(\alpha>0\) 使
\[
|\mathsf{Anom}(q)|\not\equiv 0
\quad\Longrightarrow\quad
\liminf_{q\to\infty}\frac{1}{q}\,|\log C_q(\theta)|\ge \alpha.
\]
则若该异常量非零，必有且仅有如下二者之一发生：存在有限阶 \(q\) 的核版 RH 反例见证；或 \(d_q\) 至少指数增长，从而无限核性质只能以预言机式的无限族检查（或外置模式轴）表达。
\end{corollary}

\begin{proof}
由假设，异常非零迫使 \(|\log C_q(\theta)|\) 出现线性漂移。
在 \(\rho_q\ge c^q\) 与 \(d_q\) 次指数增长的常规尺度假设下，定理 \ref{thm:finite-part-residue-drift-trichotomy} 立即给出二择一结局。证毕。
\end{proof}
