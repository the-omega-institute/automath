% Explicit bound for log M under Kernel-RH.

\begin{proposition}[Kernel--RH 假设下 Abel/Mertens 有限部常数的显式上界]\label{prop:finite-part-logM-kernel-rh-explicit-bound}
\leanverified{paper\_finite\_part\_logM\_kernel\_rh\_explicit\_bound}
设 \(A\) 为维数 \(d\) 的本原非负矩阵，Perron 根为 \(\rho>1\)，并令
\[
\zeta(z):=\det(I-zA)^{-1},\qquad z_\star:=\rho^{-1}.
\]
记主极点留数常数
\[
C:=\lim_{z\to z_\star}(1-\rho z)\,\zeta(z)>0,
\]
并设 Abel 有限部常数满足 Möbius--Abel 表达式
\[
\log\mathfrak{M}
=
\log C
\;+\;
\sum_{k\ge 2}\frac{\mu(k)}{k}\log\zeta(z_\star^{k}),
\]
其中 \(\mu\) 为 Möbius 函数。
假设满足 Kernel--RH 型平方根界
\[
\max_{j\ge 2}|\lambda_j|\le \sqrt{\rho},
\]
其中 \(\{\lambda_j\}_{j=1}^d\) 为 \(A\) 的特征值（按代数重数，\(\lambda_1=\rho\)）。
则当 \(\rho\ge 4\) 时有显式估计
\[
|\log\mathfrak{M}|
\ \le\
2(d-1)\rho^{-1/2}
\ +\ \frac{2}{\rho-1}
\ +\ \frac{2(d-1)\rho^{-1/2}}{\rho-1}.
\]
\end{proposition}
\begin{proof}
由命题 \ref{prop:finite-part-residue-constant-rh-squeeze}（在 \(\rho\ge 4\) 下的粗界部分）得
\[
|\log C|\le 2(d-1)\rho^{-1/2}.
\]
对任意 \(k\ge 2\)，由特征值分解
\(
\zeta(z)=\prod_{j=1}^{d}(1-\lambda_j z)^{-1}
\)
得到
\[
\log\zeta(\rho^{-k})
=-\log(1-\rho^{1-k})-\sum_{j=2}^{d}\log(1-\lambda_j\rho^{-k}).
\]
当 \(\rho\ge 4\) 且 \(k\ge 2\) 时有 \(0<\rho^{1-k}\le 1/\rho\le 1/4\)，从而
\[
\bigl|\log(1-\rho^{1-k})^{-1}\bigr|\le 2\rho^{1-k}.
\]
又由 \(|\lambda_j\rho^{-k}|\le \rho^{-(k-1/2)}\le \rho^{-3/2}\le 1/8\)，同样得到
\[
\bigl|\log(1-\lambda_j\rho^{-k})^{-1}\bigr|\le 2|\lambda_j|\rho^{-k}\le 2\rho^{-(k-1/2)}.
\]
因此
\[
\bigl|\log\zeta(\rho^{-k})\bigr|
\le 2\rho^{1-k}+2(d-1)\rho^{-(k-1/2)}.
\]
代回 Möbius--Abel 表达式并用 \(|\mu(k)|\le 1\)、\(1/k\le 1\)，得到
\[
|\log\mathfrak{M}|
\le |\log C|
\;+\;
\sum_{k\ge 2}2\rho^{1-k}
\;+\;
\sum_{k\ge 2}2(d-1)\rho^{-(k-1/2)}.
\]
两项几何级数分别为
\[
\sum_{k\ge 2}2\rho^{1-k}=2\sum_{m\ge 1}\rho^{-m}=\frac{2}{\rho-1},
\]
\[
\sum_{k\ge 2}2(d-1)\rho^{-(k-1/2)}
=2(d-1)\rho^{-3/2}\sum_{m\ge 0}\rho^{-m}
=\frac{2(d-1)\rho^{-1/2}}{\rho-1}.
\]
合并即得所示不等式。证毕。
\end{proof}
