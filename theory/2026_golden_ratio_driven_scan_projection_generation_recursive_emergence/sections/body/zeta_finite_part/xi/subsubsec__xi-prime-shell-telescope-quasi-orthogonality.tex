\subsubsection{显式公式的素数壳层望远镜与跨尺度准正交：由增量控制编译的审计接口}\label{subsubsec:xi-prime-shell-telescope-quasi-orthogonality}
本节给出一条与 \S\ref{subsec:discrete-abel-weil-polar-rigidity} 的离散 Abel--Weil 口径互补的“连续测试函数”显式公式接口：以对数尺度 \(H\) 为分辨率，构造在频域集中于 \(\gamma\approx T\) 的 Mellin 窗函数；随后对 \(H\) 施行 dyadic 细化得到严格的望远镜分解，并把素数端贡献拆解为对数壳层 Dirichlet 多项式的增量。
该增量在时间平均下满足显式的跨尺度准正交界，从而把“尺度间耦合”制度化为可审计、可压缩的误差项。

\paragraph{对数窗与 Mellin 频窗}
取 \(g\in C_c^\infty(\RR)\) 为实值偶函数，并为简化壳层分离的常数，假设 \(\operatorname{supp}(g)\subseteq[-1,1]\)。
对 \(T\in\RR\) 与 \(H>0\) 定义测试函数
\begin{equation}\label{eq:xi-prime-shell-test-function}
\varphi_{T,H}(x):=x^{-1/2-\mathrm{i}T}\,g\!\Bigl(\frac{\log x}{H}\Bigr)\qquad(x>0),
\end{equation}
并记其 Mellin 变换
\[
\widehat{\varphi}_{T,H}(s):=\int_{0}^{\infty}\varphi_{T,H}(x)\,x^{s-1}\,\dd x.
\]
代换 \(x=e^{u}\) 得
\begin{equation}\label{eq:xi-prime-shell-mellin-transform-window}
\widehat{\varphi}_{T,H}(s)=\int_{\RR} g(u/H)\,e^{u(s-\frac12-\mathrm{i}T)}\,\dd u.
\end{equation}
因此对任意非平凡零点 \(\rho=\beta+\mathrm{i}\gamma\) 有
\begin{equation}\label{eq:xi-prime-shell-mellin-at-zero}
\widehat{\varphi}_{T,H}(\rho)
=\int_{\RR} g(u/H)\,e^{u(\beta-\frac12)}\,e^{\mathrm{i}u(\gamma-T)}\,\dd u,
\end{equation}
特别地，当 \(\beta=\tfrac12\) 时 \(\widehat{\varphi}_{T,H}(\tfrac12+\mathrm{i}\gamma)\) 为 \(g(u/H)\) 的 Fourier 变换在频率 \(\gamma-T\) 处的取值，从而在尺度 \(H^{-1}\) 上集中于 \(\gamma\approx T\)。

\begin{theorem}[Weil 型显式公式在对数窗下的壳层接口]\label{thm:xi-prime-shell-explicit-formula-window}
在 \eqref{eq:xi-prime-shell-test-function} 的设定下，存在由 \(\Gamma\)-因子给出的显式项 \(\mathcal G_g(T,H)\) 使得成立恒等式
\begin{equation}\label{eq:xi-prime-shell-explicit-formula}
\boxed{\;
\sum_{\rho}\widehat{\varphi}_{T,H}(\rho)
=
\int_{0}^{\infty}\varphi_{T,H}(x)\,\dd x
-\sum_{p}\sum_{k\ge 1}(\log p)\Bigl(\varphi_{T,H}(p^k)+p^{-k}\varphi_{T,H}(p^{-k})\Bigr)
\;+\;\mathcal G_g(T,H)\;}
\end{equation}
其中 \(\rho\) 遍历全部非平凡零点（含重数），\(p\) 遍历素数。
并且对任意 \(N\ge 1\) 有超多项式衰减
\begin{equation}\label{eq:xi-prime-shell-gamma-term-rapid-decay}
\boxed{\;
\abs{\mathcal G_g(T,H)}\ll_{g,H,N}(1+\abs{T})^{-N}\;}
\end{equation}
其本质来源于 \(\widehat{\varphi}_{T,H}(s)\) 对竖直方向的 Schwartz 衰减与 \(\Gamma'/\Gamma\) 的多项式增长相匹配（参见 \cite{Titchmarsh1986RiemannZeta} 中关于显式公式与 \(\Gamma\)-项控制的标准讨论）。
\end{theorem}

\begin{proof}[证明要点]
将 \(\varphi_{T,H}\) 视为对数变量上的紧支撑平滑测试函数，并对 \(\zeta\) 的完成化函数应用 Weil 型显式公式，即得到 \eqref{eq:xi-prime-shell-explicit-formula} 的结构形式（Prime-power 项可由 \(\Lambda(p^k)=\log p\) 的分解得到）。
\eqref{eq:xi-prime-shell-gamma-term-rapid-decay} 来自：\(g\in C_c^\infty\) 使得 \(\widehat{\varphi}_{T,H}\) 在 \(\Im(s)\) 方向快速衰减；而 \(\Gamma'/\Gamma\) 的增长可由 Stirling 估计为多项式，从而任意阶分部积分给出任意次幂衰减。
\end{proof}

\paragraph{dyadic 尺度望远镜与素数壳层增量}
固定 \(H_0>0\)，令 \(H_j:=2^jH_0\)（\(j\ge 0\)）。
定义壳层增量窗
\[
\Delta_j(x):=g\!\Bigl(\frac{\log x}{H_j}\Bigr)-g\!\Bigl(\frac{\log x}{H_{j-1}}\Bigr),
\qquad
\varphi_T^{(j)}(x):=x^{-1/2-\mathrm{i}T}\Delta_j(x)\qquad(j\ge 1).
\]
由望远镜恒等式 \(g(\log x/H_J)=g(\log x/H_0)+\sum_{j=1}^{J}\Delta_j(x)\) 得
\[
\varphi_{T,H_J}(x)=\varphi_{T,H_0}(x)+\sum_{j=1}^{J}\varphi_T^{(j)}(x).
\]
将其代入定理 \ref{thm:xi-prime-shell-explicit-formula-window} 并逐项作差，得到壳层望远镜分解；其中由于 \(\operatorname{supp}(g)\subseteq[-1,1]\)，当 \(x>1\) 时 \(\Delta_j(x)\neq 0\) 蕴含
\begin{equation}\label{eq:xi-prime-shell-support-localization}
H_{j-1}<\log x\le H_j,
\end{equation}
因而每个 \(j\) 的 prime-power 项仅涉及满足 \(\log(p^k)\) 落在壳层 \((H_{j-1},H_j]\) 的有限集合。

\paragraph{壳层 Dirichlet 多项式与跨尺度准正交}
定义壳层 Dirichlet 多项式
\begin{equation}\label{eq:xi-prime-shell-dirichlet-polynomial}
D_j(T):=\sum_{p}\sum_{k\ge 1}a_j(p^k)\,e^{-\mathrm{i}T\log(p^k)},
\qquad
a_j(p^k):=(\log p)\,p^{-k/2}\,\Delta_j(p^k).
\end{equation}
在 \eqref{eq:xi-prime-shell-support-localization} 的支撑约束下，对每个固定 \(j\) 上式为有限和，从而 \(D_j(T)\) 为有限 Dirichlet 多项式。
并且 prime-power 项可写为
\(
\sum_{p,k}(\log p)\varphi_T^{(j)}(p^k)=D_j(T)
\)，
而对偶项 \(p^{-k}\varphi_T^{(j)}(p^{-k})\) 在 \(g\) 偶性下等于 \(\overline{D_j(T)}\)。
因此显式公式的素数端壳层增量可以被视作 \(2\Re D_j(T)\) 的可审计读出。

\begin{proposition}[壳层增量在时间平均下的准正交性]\label{prop:xi-prime-shell-quasi-orthogonality}
\leanverified{paper\_xi\_prime\_shell\_quasi\_orthogonality}
沿用定义 \eqref{eq:xi-prime-shell-dirichlet-polynomial}，并记
\[
A_j:=\sum_{p}\sum_{k\ge 1}\abs{a_j(p^k)}\qquad(j\ge 1),
\]
则对任意 \(j<k\) 与任意区间 \([T_0,T_0+L]\)（\(L>0\)）有交叉项界
\begin{equation}\label{eq:xi-prime-shell-cross-term-bound}
\boxed{\;
\abs{\int_{T_0}^{T_0+L}D_j(T)\,\overline{D_k(T)}\,\dd T}
\le
\frac{2}{H_0(2^{k-1}-2^{j})}\,A_jA_k\;}
\end{equation}
并从而得到能量分解
\begin{equation}\label{eq:xi-prime-shell-energy-decomposition}
\boxed{\;
\int_{T_0}^{T_0+L}\abs{\sum_{j=1}^{J}D_j(T)}^2\,\dd T
=
\sum_{j=1}^{J}\int_{T_0}^{T_0+L}\abs{D_j(T)}^2\,\dd T
+O\!\Bigl(\sum_{1\le j<k\le J}\frac{A_jA_k}{H_0(2^{k-1}-2^j)}\Bigr)\;}
\end{equation}
即跨尺度相互作用随尺度间距呈指数级衰减（分母近似 \(2^k\)）。
\end{proposition}

\begin{proof}
展开并交换求和积分，得到
\[
\int_{T_0}^{T_0+L}D_j(T)\overline{D_k(T)}\,\dd T
=\sum_{n}\sum_{m}a_j(n)\overline{a_k(m)}\int_{T_0}^{T_0+L}e^{-\mathrm{i}T(\log n-\log m)}\,\dd T,
\]
其中 \(n=p^r,m=q^\ell\) 遍历各自壳层内出现的素数幂。
对内层积分有闭式
\(
\frac{e^{-\mathrm{i}(T_0+L)(\log n-\log m)}-e^{-\mathrm{i}T_0(\log n-\log m)}}{-\mathrm{i}(\log n-\log m)}
\)
从而
\(
\abs{\int_{T_0}^{T_0+L}e^{-\mathrm{i}T(\log n-\log m)}\,\dd T}\le \frac{2}{\abs{\log(n/m)}}.
\)
若 \(n\) 属于壳层 \(j\) 且 \(m\) 属于壳层 \(k>j\)，由 \eqref{eq:xi-prime-shell-support-localization} 得
\(
\log m\ge H_{k-1},\ \log n\le H_j
\),
因而
\(
\abs{\log(n/m)}\ge H_{k-1}-H_j=H_0(2^{k-1}-2^j)
\)。
代回并对系数取绝对值求和即得 \eqref{eq:xi-prime-shell-cross-term-bound}。
\eqref{eq:xi-prime-shell-energy-decomposition} 由展开 \(\abs{\sum D_j}^2\) 并对交叉项应用 \eqref{eq:xi-prime-shell-cross-term-bound} 得到。
\end{proof}

\begin{conjecture}[壳层负反馈收缩准则]\label{conj:xi-prime-shell-negative-feedback}
在上述壳层分解中，定义对任意 \(T\in\RR\) 的壳层负反馈范数
\begin{equation}\label{eq:xi-prime-shell-negative-feedback-norm}
\mathfrak F_g(T):=\sum_{j\ge 1}2^{-j/2}\,\sup_{u\in[T,T+1]}\abs{D_j(u)}.
\end{equation}
猜想存在某个实值偶函数 \(g\in C_c^\infty(\RR)\)（可取 \(\operatorname{supp}(g)\subseteq[-1,1]\)）使得
\begin{equation}\label{eq:xi-prime-shell-negative-feedback-uniform-bound}
\boxed{\;
\sup_{T\in\RR}\mathfrak F_g(T)<\infty.\;}
\end{equation}
该一致有界性可被理解为：在 dyadic 壳层望远镜下，素数端增量在所有尺度上均被限制在平方根权重可吸收的量级，从而形成一条仅以增量控制为输入的跨尺度收缩律；其若能与显式公式的零点端读出闭合对齐，将为 RH 提供一条完全可审计的充分条件路线。
\end{conjecture}
