\paragraph{全局积分缺陷与端点通量：闭式核、总和律与黄金尺度渐近}\label{par:xi-comoving-integrated-jensen-defect-sumrule}
在共动平移族 \(x_0\mapsto F_{\zeta,x_0}\) 的圆盘化完成化口径下，Jensen 缺陷 \(D_{\zeta,x_0}(\varrho)\)（\eqref{eq:xi-comoving-shifted-jensen-defect}）对零点逐项可加；其在 \(x_0\) 上的全局积分因此分解为一组仅依赖离线偏移 \(\delta\) 的闭式核的总和。
该分解完全消去高度变量 \(\gamma\)，并导出 \(L^1\)-型端点判据、可见阈值处的非整数幂尖点律，以及黄金分辨率链上的显式端点饱和渐近。

\paragraph{单零点截断缺陷核的 \texorpdfstring{$L^1$}{L1} 闭式}
固定 \(0<\delta<1\)，定义 Cayley 像
\[
w_\delta(x):=\mathcal{C}(x+\mathrm{i}\delta)\qquad(x\in\RR),
\]
并令 \([u]_+:=\max\{0,u\}\)。
对 \(0<\varrho<1\) 定义单零点的截断缺陷核
\begin{equation}\label{eq:xi-single-defect-kernel}
\mathcal{D}_{\varrho,\delta}(x):=\Bigl[\log\frac{\varrho}{|w_\delta(x)|}\Bigr]_+.
\end{equation}

\begin{theorem}[平移 Jensen 缺陷的显式核分解]\label{thm:xi-comoving-jensen-defect-kernel-decomposition}
\leanverified{paper\_xi\_comoving\_jensen\_defect\_kernel\_decomposition}
令 \(\mathcal{Z}^-\) 表示 \(\Xi_{\zeta}\) 的全部离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\) 中满足 \(\beta<\tfrac12\) 的部分（按重数计），并记
\(
\delta(\rho):=\tfrac12-\beta>0
\)。
则对任意 \(x_0\in\RR\) 与 \(0<\varrho<1\) 有
\begin{equation}\label{eq:xi-comoving-jensen-defect-kernel-decomposition}
\boxed{\;
D_{\zeta,x_0}(\varrho)
=\sum_{\rho\in\mathcal{Z}^-}\mathcal{D}_{\varrho,\delta(\rho)}(\gamma-x_0)\;}
\end{equation}
其中核 \(\mathcal{D}_{\varrho,\delta}\) 由 \eqref{eq:xi-single-defect-kernel} 定义。
并且对每个固定的 \((x_0,\varrho)\)，上式右端仅有有限个非零项（因而绝对收敛）。
\end{theorem}

\begin{proof}
由 Jensen 求和式（命题 \ref{prop:app-jensen-formula-defect}），对解析函数 \(F_{\zeta,x_0}\) 有
\[
D_{\zeta,x_0}(\varrho)=\sum_{|a|<\varrho}\log\!\Bigl(\frac{\varrho}{|a|}\Bigr),
\]
其中求和遍历 \(F_{\zeta,x_0}\) 在 \(|w|<\varrho\) 内的零点（按重数计）。
另一方面，由零点像公式 \eqref{eq:xi-comoving-disk-image} 可知：任一满足 \(\beta<\tfrac12\) 的零点 \(\rho=\beta+\mathrm{i}\gamma\) 在平移纤维 \(x_0\) 上诱导盘内零点
\(
w_{\rho,x_0}=\mathcal{C}((\gamma-x_0)+\mathrm{i}\delta(\rho))
\)；
而临界线上的零点像落在 \(|w|=1\) 上，不进入 \(\mathbb{D}\) 内部。
因此 \(D_{\zeta,x_0}(\varrho)\) 可写为对 \(\rho\in\mathcal{Z}^-\) 的逐项贡献之和，并用截断 \([\,\cdot\,]_+\) 把条件 \(|w_{\rho,x_0}|<\varrho\) 编码为
\(
\bigl[\log(\varrho/|w_{\rho,x_0}|)\bigr]_+
\)；
这正是 \eqref{eq:xi-comoving-jensen-defect-kernel-decomposition}。
\par
为证“局部有限性”，由 \eqref{eq:xi-cayley-modulus-delta} 可知 \(\mathcal{D}_{\varrho,\delta}(x)=0\) 除非 \(|x|<X(\varrho,\delta)\)。
故对固定 \(x_0\) 与 \(\varrho\)，只有满足 \(|\gamma-x_0|<X(\varrho,\delta(\rho))\) 的零点可能贡献非零项；
而零点集在有界高度区间内有限，从而非零项有限。证毕。
\end{proof}

\begin{proposition}[单零点积分缺陷的闭式表达]\label{prop:xi-single-defect-integrated-closed-form}
\leanverified{paper\_xi\_single\_defect\_integrated\_closed\_form}
对任意 \(0<\varrho<1\) 与 \(0<\delta<1\)，函数 \(\mathcal{D}_{\varrho,\delta}\in L^1(\RR)\) 且
\begin{equation}\label{eq:xi-single-defect-integrated-closed-form}
\boxed{\;
J_{\varrho}(\delta):=\int_{\RR}\mathcal{D}_{\varrho,\delta}(x)\,\dd x
=2(1+\delta)\arctan\!\Bigl(\frac{X(\varrho,\delta)}{1+\delta}\Bigr)
-2(1-\delta)\arctan\!\Bigl(\frac{X(\varrho,\delta)}{1-\delta}\Bigr)\;}
\end{equation}
其中
\begin{equation}\label{eq:xi-single-defect-X}
\boxed{\;
X(\varrho,\delta)
:=\sqrt{\max\Bigl\{0,\ \frac{\varrho^2(1+\delta)^2-(1-\delta)^2}{1-\varrho^2}\Bigr\}}\ \ge\ 0.\;}
\end{equation}
\end{proposition}

\begin{proof}
由 Cayley 变换 \eqref{eq:app-cayley-disk}，对 \(t=x+\mathrm{i}\delta\) 直接计算
\begin{equation}\label{eq:xi-cayley-modulus-delta}
|w_\delta(x)|^2
=\left|\frac{1+\mathrm{i}(x+\mathrm{i}\delta)}{1-\mathrm{i}(x+\mathrm{i}\delta)}\right|^2
=\frac{x^2+(1-\delta)^2}{x^2+(1+\delta)^2}.
\end{equation}
因此条件 \(|w_\delta(x)|<\varrho\) 等价于
\[
x^2+(1-\delta)^2<\varrho^2\bigl(x^2+(1+\delta)^2\bigr),
\]
即 \(|x|<X(\varrho,\delta)\)（其中 \(X\) 由 \eqref{eq:xi-single-defect-X} 给出；若右端为负则区间为空并且 \(X=0\)）。
若 \(X=0\)，则 \(\mathcal{D}_{\varrho,\delta}\equiv 0\) 从而积分为 \(0\)，而 \eqref{eq:xi-single-defect-integrated-closed-form} 右端亦为 \(0\)；下设 \(X>0\)。
从而
\[
\int_{\RR}\mathcal{D}_{\varrho,\delta}(x)\,\dd x
=\int_{-X}^{X}\log\frac{\varrho}{|w_\delta(x)|}\,\dd x
=\int_{-X}^{X}\left(\log\varrho-\frac12\log\frac{x^2+(1-\delta)^2}{x^2+(1+\delta)^2}\right)\dd x.
\]
记 \(A:=1+\delta\)、\(B:=1-\delta\)。
利用原函数恒等式
\[
\int \log(x^2+a^2)\,\dd x
=x\log(x^2+a^2)-2x+2a\arctan(x/a),
\]
并在 \([-X,X]\) 上代入整理，可化为
\begin{equation}\label{eq:xi-single-defect-integrated-intermediate}
\int_{\RR}\mathcal{D}_{\varrho,\delta}
=2X\log\varrho+X\log\frac{X^2+A^2}{X^2+B^2}
+2A\arctan(X/A)-2B\arctan(X/B).
\end{equation}
另一方面，边界点满足 \(|w_\delta(X)|=\varrho\)，由 \eqref{eq:xi-cayley-modulus-delta} 得
\(
\frac{X^2+B^2}{X^2+A^2}=\varrho^2
\)，从而
\(
\log\frac{X^2+A^2}{X^2+B^2}=-2\log\varrho
\)。
代回 \eqref{eq:xi-single-defect-integrated-intermediate} 可见前两项严格抵消，得到 \eqref{eq:xi-single-defect-integrated-closed-form}。证毕。
\end{proof}

\begin{corollary}[阈值不可见性与端点饱和]\label{cor:xi-single-defect-threshold-saturation}
\leanverified{paper\_xi\_single\_defect\_threshold\_saturation}
对任意 \(0<\delta<1\) 与 \(0<\varrho<1\)，积分质量 \(J_{\varrho}(\delta)\) 满足：
\begin{enumerate}
  \item 若 \(\varrho\le \frac{1-\delta}{1+\delta}\)，则 \(J_{\varrho}(\delta)=0\)；
  \item 若 \(\varrho> \frac{1-\delta}{1+\delta}\)，则 \(J_{\varrho}(\delta)>0\) 且由 \eqref{eq:xi-single-defect-integrated-closed-form} 给出；
  \item 极限存在并且
  \begin{equation}\label{eq:xi-single-defect-saturation}
  \boxed{\;\lim_{\varrho\uparrow 1}J_{\varrho}(\delta)=2\pi\delta.\;}
  \end{equation}
\end{enumerate}
\end{corollary}

\begin{proof}
由 \eqref{eq:xi-single-defect-X}，条件 \(X(\varrho,\delta)=0\) 等价于
\(
\varrho^2(1+\delta)^2-(1-\delta)^2\le 0
\)，即 \(\varrho\le (1-\delta)/(1+\delta)\)。
此时 \(\mathcal{D}_{\varrho,\delta}\equiv 0\) 从而 \(J_{\varrho}(\delta)=0\)。
当 \(\varrho>(1-\delta)/(1+\delta)\) 时有 \(X(\varrho,\delta)>0\)，由定义得 \(J_{\varrho}(\delta)>0\)，并由命题 \ref{prop:xi-single-defect-integrated-closed-form} 的闭式 \eqref{eq:xi-single-defect-integrated-closed-form} 给出。
最后令 \(\varrho\uparrow 1\) 则 \(X(\varrho,\delta)\to\infty\)，从 \eqref{eq:xi-single-defect-integrated-closed-form} 得
\(
J_{\varrho}(\delta)\to \pi(1+\delta)-\pi(1-\delta)=2\pi\delta
\)，得到 \eqref{eq:xi-single-defect-saturation}。证毕。
\end{proof}

\begin{proposition}[可见区长度与对数半径导数]\label{prop:xi-single-defect-mass-log-derivative}
\leanverified{paper\_xi\_single\_defect\_mass\_log\_derivative}
固定 \(0<\delta<1\)。
则映射 \(\varrho\mapsto J_{\varrho}(\delta)\) 在 \((0,1)\) 上局部绝对连续，并且对几乎处处 \(0<\varrho<1\) 有
\begin{equation}\label{eq:xi-single-defect-log-derivative}
\boxed{\;
\frac{\dd}{\dd(\log\varrho)}J_{\varrho}(\delta)
=\int_{\RR}\mathbf 1_{\{|\mathcal{C}(x+\mathrm{i}\delta)|<\varrho\}}\,\dd x
=2X(\varrho,\delta)\;}
\end{equation}
其中 \(X(\varrho,\delta)\) 由 \eqref{eq:xi-single-defect-X} 给出。
\end{proposition}

\begin{proof}
对固定 \(x\) 定义
\(
f_x(\varrho):=\mathcal{D}_{\varrho,\delta}(x)
\)。
当 \(\varrho\le |\mathcal{C}(x+\mathrm{i}\delta)|\) 时 \(f_x(\varrho)=0\)，当 \(\varrho>|\mathcal{C}(x+\mathrm{i}\delta)|\) 时
\(
f_x(\varrho)=\log(\varrho/|\mathcal{C}(x+\mathrm{i}\delta)|)
\)。
因此对几乎处处 \(\varrho\)（除去 \(\varrho=|\mathcal{C}(x+\mathrm{i}\delta)|\) 的点）成立
\[
\frac{\dd}{\dd(\log\varrho)}f_x(\varrho)=\mathbf 1_{\{|\mathcal{C}(x+\mathrm{i}\delta)|<\varrho\}}.
\]
并且对任意紧区间 \([\varrho_1,\varrho_2]\subset(0,1)\)，有一致支配
\(
0\le f_x(\varrho)\le \bigl[\log(\varrho_2/|\mathcal{C}(x+\mathrm{i}\delta)|)\bigr]_+
\)，其关于 \(x\) 的积分有限（由命题 \ref{prop:xi-single-defect-integrated-closed-form}，它等于 \(J_{\varrho_2}(\delta)\)）。
故可用支配收敛在积分号内对 \(\log\varrho\) 求导，得到
\[
\frac{\dd}{\dd(\log\varrho)}J_{\varrho}(\delta)
=\int_{\RR}\frac{\dd}{\dd(\log\varrho)}f_x(\varrho)\,\dd x
=\int_{\RR}\mathbf 1_{\{|\mathcal{C}(x+\mathrm{i}\delta)|<\varrho\}}\,\dd x.
\]
最后由 \eqref{eq:xi-cayley-modulus-delta}，集合 \(\{x:\ |\mathcal{C}(x+\mathrm{i}\delta)|<\varrho\}\) 恰为区间 \((-X(\varrho,\delta),X(\varrho,\delta))\)，其长度为 \(2X(\varrho,\delta)\)。证毕。
\end{proof}

\paragraph{Jensen 缺陷的全局 \texorpdfstring{$L^1$}{L1} 总和律与端点通量恒等式}
定义平移族 Jensen 缺陷的全局积分
\begin{equation}\label{eq:xi-integrated-defect-I}
\boxed{\;
\mathcal{I}_D(\varrho):=\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0\qquad(0<\varrho<1).\;}
\end{equation}

\begin{theorem}[平移 coarea 恒等式：全局积分缺陷的高度消去分解与端点饱和]\label{thm:xi-integrated-defect-sumrule-endpoint-flux}
 令 \(\mathcal{Z}^-\) 与 \(\delta(\rho)\) 如定理 \ref{thm:xi-comoving-jensen-defect-kernel-decomposition} 所定义（按重数计）。
 则对任意 \(0<\varrho<1\)，总平均缺陷 \(\mathcal{I}_D(\varrho)\) 满足精确恒等式（允许取扩展实数 \(+\infty\)）
\begin{equation}\label{eq:xi-integrated-defect-sumrule}
\boxed{\;
\mathcal{I}_D(\varrho)
=\sum_{\rho\in\mathcal{Z}^-}J_{\varrho}(\delta(\rho))\ \in\ [0,\infty].\;}
\end{equation}
其中 \(J_{\varrho}(\delta):=\int_{\RR}\mathcal{D}_{\varrho,\delta}(x)\,\dd x\) 由命题 \ref{prop:xi-single-defect-integrated-closed-form} 的闭式 \eqref{eq:xi-single-defect-integrated-closed-form} 给出。
并且极限存在且满足严格恒等式
\begin{equation}\label{eq:xi-integrated-defect-limit-endpoint-flux}
\boxed{\;
\lim_{\varrho\uparrow 1}\mathcal{I}_D(\varrho)
=2\pi\sum_{\rho\in\mathcal{Z}^-}\delta(\rho)
=2\pi\,\Phi_{-1}(B).\;}
\end{equation}
\end{theorem}

\begin{proof}
 由定理 \ref{thm:xi-comoving-jensen-defect-kernel-decomposition}，
 \(
 D_{\zeta,x_0}(\varrho)=\sum_{\rho\in\mathcal{Z}^-}\mathcal{D}_{\varrho,\delta(\rho)}(\gamma-x_0)
 \)
 为非负项之和。
故由 Tonelli 定理（允许取 \(+\infty\)）有
 \[
 \mathcal{I}_D(\varrho)
 =\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0
 =\sum_{\rho\in\mathcal{Z}^-}\int_{\RR}\mathcal{D}_{\varrho,\delta(\rho)}(\gamma-x_0)\,\dd x_0.
 \]
 对每一项作代换 \(x=\gamma-x_0\) 得
 \(
 \int_{\RR}\mathcal{D}_{\varrho,\delta(\rho)}(\gamma-x_0)\dd x_0
 =\int_{\RR}\mathcal{D}_{\varrho,\delta(\rho)}(x)\dd x
 =J_{\varrho}(\delta(\rho))
 \)，从而得到 \eqref{eq:xi-integrated-defect-sumrule}。
 \par
 由定义可知 \(\mathcal{I}_D(\varrho)\) 随 \(\varrho\uparrow 1\) 单调不减，因此极限（取扩展实数）存在。
并且对每个固定 \(\rho\in\mathcal{Z}^-\)，由推论 \ref{cor:xi-single-defect-threshold-saturation} 有
 \(
 J_{\varrho}(\delta(\rho))\uparrow 2\pi\delta(\rho)
 \)。
再对计数测度应用单调收敛定理（或等价地：对非负级数逐项取极限）得到
 \[
 \lim_{\varrho\uparrow 1}\mathcal{I}_D(\varrho)
 =\sum_{\rho\in\mathcal{Z}^-}\lim_{\varrho\uparrow 1}J_{\varrho}(\delta(\rho))
 =2\pi\sum_{\rho\in\mathcal{Z}^-}\delta(\rho).
 \]
 最后由端点通量的加性表示 \(\Phi_{-1}(B)=\sum_{\rho\in\mathcal{Z}^-}\delta(\rho)\)（见 \eqref{eq:xi-endpoint-absorption-kernel-limit}）得到 \eqref{eq:xi-integrated-defect-limit-endpoint-flux}。证毕。
\end{proof}

\paragraph{极限缺陷势：对数核的 Poisson 原函数与频域可逆性}
上一段给出 \(x_0\)-积分版的标量证书。
若进一步把 \(\varrho\uparrow 1\) 的极限在每个共动地址 \(x_0\) 上保留，则得到一个随 \(x_0\) 变化的\emph{极限缺陷势}，其核函数具有 Poisson 原函数结构，并在有限缺陷口径下携带完整的频域指纹。

\begin{proposition}[对数缺陷势的 Poisson 尺度原函数表示]\label{prop:xi-log-defect-potential-poisson-primitive}
\leanverified{paper\_xi\_log\_defect\_potential\_poisson\_primitive}
对任意 \(\delta\in(0,1)\) 与 \(\gamma\in\RR\)，定义对数缺陷势
\begin{equation}\label{eq:xi-log-defect-potential}
\boxed{\;
\mathcal{K}_{\gamma,\delta}(x):=-\log\bigl|\mathcal{C}((x-\gamma)+\mathrm{i}\delta)\bigr|\qquad(x\in\RR).\;}
\end{equation}
并定义 Poisson 核
\[
P_s(u):=\frac{1}{\pi}\frac{s}{u^2+s^2}\qquad(s>0).
\]
则：
\begin{enumerate}
  \item \textbf{（Poisson 原函数表示）}对所有 \(x\in\RR\) 有
  \begin{equation}\label{eq:xi-log-defect-poisson-primitive}
  \boxed{\;
  \mathcal{K}_{\gamma,\delta}(x)
  =\pi\int_{1-\delta}^{1+\delta}P_s(x-\gamma)\,\dd s\; }.
  \end{equation}
  \item \textbf{（尺度平移律）}对任意 \(t>0\)，Poisson 卷积满足
  \begin{equation}\label{eq:xi-log-defect-poisson-convolution}
  \boxed{\;
  (P_t*\mathcal{K}_{\gamma,\delta})(x)
  =\frac12\log\frac{(x-\gamma)^2+(t+1+\delta)^2}{(x-\gamma)^2+(t+1-\delta)^2}\; }.
  \end{equation}
  \item \textbf{（尺度导数）}\(\partial_t(P_t*\mathcal{K}_{\gamma,\delta})(x)\) 具有显式差分形式
  \begin{equation}\label{eq:xi-log-defect-poisson-scale-derivative}
  \boxed{\;
  \partial_t(P_t*\mathcal{K}_{\gamma,\delta})(x)
  =\pi\Bigl(P_{t+1+\delta}(x-\gamma)-P_{t+1-\delta}(x-\gamma)\Bigr)\; }.
  \end{equation}
\end{enumerate}
\end{proposition}
\begin{proof}
由 \eqref{eq:app-cayley-disk} 与 \(|\mathcal{C}(t)|^2=\frac{|1+\mathrm{i}t|^2}{|1-\mathrm{i}t|^2}\)，对 \(t=u+\mathrm{i}\delta\) 得
\[
-\log|\mathcal{C}(u+\mathrm{i}\delta)|
=\frac12\log\frac{u^2+(1+\delta)^2}{u^2+(1-\delta)^2}.
\]
用恒等式 \(\int \frac{s}{u^2+s^2}\,\dd s=\frac12\log(u^2+s^2)\) 立即得到 \eqref{eq:xi-log-defect-poisson-primitive}（再作平移 \(u=x-\gamma\)）。
\par
为证 \eqref{eq:xi-log-defect-poisson-convolution}，记 \(f_a(u):=\log(u^2+a^2)\) 并考虑调和延拓
\(
U_a(u,t):=\log\bigl(u^2+(t+a)^2\bigr)
\)（\(\Im(u+\mathrm{i}t)>0\)）。
其边界值满足 \(U_a(u,0)=f_a(u)\)。
由 Poisson 积分的唯一性，\((P_t*f_a)(u)=U_a(u,t)\)。
将 \(\mathcal{K}_{\gamma,\delta}\) 写为 \(f_{1+\delta}-f_{1-\delta}\) 的一半并平移回 \(x-\gamma\)，即得 \eqref{eq:xi-log-defect-poisson-convolution}。
\par
最后对 \eqref{eq:xi-log-defect-poisson-convolution} 关于 \(t\) 求导并整理为 Poisson 核形式即得 \eqref{eq:xi-log-defect-poisson-scale-derivative}。证毕。
\end{proof}

\begin{theorem}[单零 co-moving Jensen 权的基点积分守恒律]\label{thm:xi-comoving-jensen-weight-integral-conservation}
在命题 \ref{prop:xi-log-defect-potential-poisson-primitive} 的设定下，取任意 \((\gamma,\delta)\in\RR\times(0,1)\)，并令
\[
w_{x_0}:=\mathcal{C}\bigl((\gamma-x_0)+\mathrm{i}\delta\bigr)\in\mathbb{D},
\qquad
J_{\gamma,\delta}(x_0):=\log\frac{1}{|w_{x_0}|}=\mathcal{K}_{\gamma,\delta}(x_0).
\]
则有严格恒等式
\[
\int_{\RR}J_{\gamma,\delta}(x_0)\,\dd x_0=2\pi\delta.
\]
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-log-defect-poisson-primitive}，
\[
\mathcal{K}_{\gamma,\delta}(x)=\pi\int_{1-\delta}^{1+\delta}P_s(x-\gamma)\,\dd s.
\]
两端对 \(x\in\RR\) 积分并交换积分次序，利用 \(\int_{\RR}P_s(x-\gamma)\,\dd x=1\) 得
\[
\int_{\RR}\mathcal{K}_{\gamma,\delta}(x)\,\dd x
=\pi\int_{1-\delta}^{1+\delta}\left(\int_{\RR}P_s(x-\gamma)\,\dd x\right)\dd s
=\pi\int_{1-\delta}^{1+\delta}1\,\dd s
=2\pi\delta.
\]
证毕。
\end{proof}
\leanverified{paper\_xi\_comoving\_jensen\_weight\_integral\_conservation}

\begin{proposition}[Dirichlet--to--Neumann 算子作用于 Jensen 核：闭式表达与 Poisson 互易]\label{prop:xi-dtn-acts-on-jensen-kernel}
令 \(|D|\) 为 \(\RR\) 上 Dirichlet--to--Neumann 算子（Fourier 乘子 \(|\xi|\)），并令 \(P_t*f=\exp(-t|D|)f\) 表示 Poisson 半群（\(t>0\)）。
在命题 \ref{prop:xi-log-defect-potential-poisson-primitive} 的设定下，对任意 \((\gamma,\delta)\in\RR\times(0,1)\) 有分布恒等式
\[
\boxed{\;
|D|\,\mathcal{K}_{\gamma,\delta}
=\pi\Bigl(P_{1-\delta}(\cdot-\gamma)-P_{1+\delta}(\cdot-\gamma)\Bigr)\; }.
\]
特别地，取 \(j_\delta(x):=\mathcal{K}_{0,\delta}(x)\) 得
\[
\boxed{\;
|D|\,j_\delta
=\pi\bigl(P_{1-\delta}-P_{1+\delta}\bigr)\; }.
\]
并且 Poisson 平滑满足显式参数平移公式 \eqref{eq:xi-log-defect-poisson-convolution}。
\end{proposition}
\begin{proof}
由 Dirichlet--to--Neumann 恒等式，\(P_t*f=e^{-t|D|}f\) 且在分布意义下
\[
\partial_t(P_t*f)=-P_t*(|D|f)\qquad(t>0).
\]
将 \(f=\mathcal{K}_{\gamma,\delta}\) 代入，并用 \eqref{eq:xi-log-defect-poisson-scale-derivative} 得
\[
P_t*(|D|\mathcal{K}_{\gamma,\delta})(x)
=-\partial_t(P_t*\mathcal{K}_{\gamma,\delta})(x)
=\pi\Bigl(P_{t+1-\delta}(x-\gamma)-P_{t+1+\delta}(x-\gamma)\Bigr).
\]
令 \(t\downarrow 0\)，由 \(P_t*\) 对 \(\mathcal{S}'(\RR)\) 的强连续性得到所述恒等式。
平移 \(\gamma=0\) 的特例即为 \(j_\delta\) 的公式。证毕。
\end{proof}

\begin{theorem}[极限缺陷势的频域谱分解与可辨识性（有限缺陷口径）]\label{thm:xi-limit-defect-potential-fourier-injective}
设离临界线零点多重集为有限族 \(\{\rho_j\}_{j=1}^{M}\)（按重数计），其中
\(
\rho_j=\beta_j+\mathrm{i}\gamma_j
\)
且 \(\beta_j<\tfrac12\)。
令 \(\delta_j:=\tfrac12-\beta_j\in(0,1)\)，重数为 \(m_j\in\NN\)。
定义极限缺陷势
\begin{equation}\label{eq:xi-limit-defect-potential}
\boxed{\;
\mathcal{D}_\infty(x_0)
:=\lim_{\varrho\uparrow 1}D_{\zeta,x_0}(\varrho)
=\sum_{j=1}^{M} m_j\,\mathcal{K}_{\gamma_j,\delta_j}(x_0)\qquad(x_0\in\RR)\;}
\end{equation}
其中 \(\mathcal{K}_{\gamma,\delta}\) 如 \eqref{eq:xi-log-defect-potential}。
则：
\begin{enumerate}
  \item \(\mathcal{D}_\infty'\in L^1(\RR)\cap L^2(\RR)\) 且
  \begin{equation}\label{eq:xi-log-defect-potential-derivative}
  \boxed{\;
  \frac{\dd}{\dd x}\mathcal{K}_{\gamma,\delta}(x)
  =\frac{x-\gamma}{(x-\gamma)^2+(1+\delta)^2}-\frac{x-\gamma}{(x-\gamma)^2+(1-\delta)^2}\; }.
  \end{equation}
  \item 取 Fourier 约定 \(\widehat f(\xi)=\int_{\RR}e^{-\mathrm{i}\xi x}f(x)\,\dd x\)，则对 \(\xi\neq 0\) 有显式谱分解
  \begin{equation}\label{eq:xi-limit-defect-potential-fourier}
  \boxed{\;
  \widehat{\mathcal{D}_\infty'}(\xi)
  =-\mathrm{i}\pi\,\operatorname{sgn}(\xi)\sum_{j=1}^{M} m_j\Bigl(e^{-(1+\delta_j)|\xi|}-e^{-(1-\delta_j)|\xi|}\Bigr)e^{-\mathrm{i}\gamma_j\xi}\; }.
  \end{equation}
  \item \textbf{（零化单射性）}若 \(\mathcal{D}_\infty\equiv 0\)，则 \(M=0\)，即不存在离临界线零点。
\end{enumerate}
\leanverified{paper\_xi\_limit\_defect\_potential\_fourier\_injective}
\end{theorem}

\begin{proof}
由 Jensen 求和式（命题 \ref{prop:app-jensen-formula-defect}）与 \(|w_{\rho_j,x_0}|<1\)（\(\delta_j>0\)）可知：对固定 \(x_0\)，
\(
D_{\zeta,x_0}(\varrho)
=\sum_{j=1}^{M}m_j\log\frac{\varrho}{|w_{\rho_j,x_0}|}
\)
当 \(\varrho\) 充分接近 \(1\) 时成立，并随 \(\varrho\uparrow 1\) 单调收敛到
\(
\sum_j m_j\log\frac{1}{|w_{\rho_j,x_0}|}
=\sum_j m_j\,\mathcal{K}_{\gamma_j,\delta_j}(x_0)
\)；
从而得到 \eqref{eq:xi-limit-defect-potential}。
\par
由 \eqref{eq:xi-log-defect-potential} 的显式对数比表达，直接求导得到 \eqref{eq:xi-log-defect-potential-derivative}；
每一项在无穷远处为 \(O(|x|^{-3})\)，故属于 \(L^1\cap L^2\)，有限求和得 (1)。
\par
由经典 Fourier 变换恒等式（在上述约定下）：
\[
\widehat{\frac{x}{x^2+a^2}}(\xi)
=-\mathrm{i}\pi\,\operatorname{sgn}(\xi)\,e^{-a|\xi|}\qquad(a>0,\ \xi\neq 0),
\]
并结合平移因子 \(e^{-\mathrm{i}\gamma\xi}\)，对 \eqref{eq:xi-log-defect-potential-derivative} 逐项 Fourier 变换并求和即得 \eqref{eq:xi-limit-defect-potential-fourier}。
\par
最后证 (3)。由 \(\delta_j>0\) 得 \(|\mathcal{C}((x-\gamma_j)+\mathrm{i}\delta_j)|<1\)（\(\forall x\)），故 \(\mathcal{K}_{\gamma_j,\delta_j}(x)>0\)（\(\forall x\)）。
因此 \(\mathcal{D}_\infty(x)=\sum_j m_j\mathcal{K}_{\gamma_j,\delta_j}(x)\) 为处处非负且在 \(M\ge 1\) 时处处严格正的函数；若 \(\mathcal{D}_\infty\equiv 0\)，只能是 \(M=0\)。证毕。
\end{proof}

\begin{corollary}[对数缺陷势锥的极端射线与原子分解唯一性（有限缺陷口径）]\label{cor:xi-log-defect-cone-extreme}
令
\[
\mathcal{C}_{\log}:=\left\{\sum_{j=1}^{M} m_j\,\mathcal{K}_{\gamma_j,\delta_j}\ :\ M<\infty,\ m_j\in\NN,\ (\gamma_j,\delta_j)\in\RR\times(0,1)\right\}\subset L^1(\RR).
\]
则 \(\overline{\mathcal{C}_{\log}}\) 在 \(L^1(\RR)\) 中构成闭凸锥，并且其极端射线恰由单核 \(\RR_{\ge 0}\cdot\mathcal{K}_{\gamma,\delta}\) 给出。
此外，若
\(
\sum_j m_j\mathcal{K}_{\gamma_j,\delta_j}=\sum_\ell n_\ell\mathcal{K}_{\eta_\ell,\varepsilon_\ell}
\)
几乎处处成立（两侧皆为有限和），则两侧参数多重集按重数一致。
\end{corollary}

\begin{proof}
闭凸锥性为生成元集合的标准性质。
对有限和的分解唯一性由定理 \ref{thm:xi-limit-defect-potential-fourier-injective} 的频域谱分解推出：两侧相减并对导数取 Fourier 变换得到一条有限指数和恒等式，从而各指数项参数与系数按重数一致。
极端射线结论随之成立：若单核可作非平凡分解，则差给出一条非零的零表示，与分解唯一性矛盾。证毕。
\end{proof}

I(\varrho,\delta):=\int_{\RR}\mathcal{D}_{\varrho,\delta}(x)\,\dd x

\begin{proposition}[黄金尺度渐近：端点通量极限的显式速率常数]\label{prop:xi-integrated-defect-golden-asymptotics}
固定 \(0<\delta<1\)，并令 \(\varepsilon:=1-\varrho^2\downarrow 0\)。
则当 \(\varrho\uparrow 1\) 时成立渐近展开
\begin{equation}\label{eq:xi-integrated-defect-asymptotics-eps}
\boxed{\;
I(\varrho,\delta)
=2\pi\delta
-4\sqrt{\delta}\,\varepsilon^{1/2}
+\left(\frac{1+\delta^2}{6\sqrt{\delta}}-\sqrt{\delta}\right)\varepsilon^{3/2}
+O_\delta(\varepsilon^{5/2}).\;}
\end{equation}
特别地，在黄金分辨率链
\begin{equation}\label{eq:xi-golden-resolution-chain-rho}
\boxed{\;\varrho_m:=\tanh\!\Bigl(\frac12 m\log\varphi\Bigr)\qquad(m\in\NN)\;}
\end{equation}
上成立精确闭式
\begin{equation}\label{eq:xi-golden-resolution-chain-rho-closed}
\boxed{\;
\varrho_m=\frac{\varphi^m-1}{\varphi^m+1},
\qquad
\frac{1+\varrho_m}{1-\varrho_m}=\varphi^m,
\qquad
\frac{1-\varrho_m}{1+\varrho_m}=\varphi^{-m},
\qquad
1-\varrho_m^2=\frac{4\varphi^m}{(\varphi^m+1)^2}
=4\varphi^{-m}+O(\varphi^{-2m}).\;}
\end{equation}
上有
\begin{equation}\label{eq:xi-integrated-defect-asymptotics-golden-chain}
\boxed{\;
I(\varrho_m,\delta)
=2\pi\delta
-8\sqrt{\delta}\,\varphi^{-m/2}
+O_\delta(\varphi^{-3m/2}).\;}
\end{equation}
\end{proposition}
\leanverified{paper\_xi\_integrated\_defect\_golden\_asymptotics}

\begin{proof}
由 \eqref{eq:xi-single-defect-X}，对 \(A:=1+\delta\) 有
\(
X(\varrho,\delta)^2=\frac{4\delta}{\varepsilon}-A^2
\)，从而 \(X\to\infty\) 且
\[
\frac{1}{X}
=\frac{\varepsilon^{1/2}}{2\delta^{1/2}}
\left(1+\frac{A^2}{8\delta}\varepsilon+O(\varepsilon^2)\right),
\qquad
\frac{1}{X^3}
=\frac{\varepsilon^{3/2}}{8\delta^{3/2}}
\left(1+O(\varepsilon)\right).
\]
对 \(\arctan(X/a)\)（\(a>0\)）使用标准展开
\[
\arctan(X/a)=\frac{\pi}{2}-\frac{a}{X}+\frac{a^3}{3X^3}+O(X^{-5}),
\]
代入 \eqref{eq:xi-single-defect-integrated-closed-form} 并按 \(\varepsilon^{1/2}\) 收集项即可得到 \eqref{eq:xi-integrated-defect-asymptotics-eps}。
\par
对黄金链，使用恒等式 \(\tanh(\tfrac12\log t)=\frac{t-1}{t+1}\)（\(t>0\)）并取 \(t=\varphi^m\)，得到 \eqref{eq:xi-golden-resolution-chain-rho-closed}；
代入 \eqref{eq:xi-integrated-defect-asymptotics-eps} 即得 \eqref{eq:xi-integrated-defect-asymptotics-golden-chain}。证毕。
\end{proof}

\begin{corollary}[黄金链上的可见窗口与饱和上界]\label{cor:xi-golden-window-saturation-upperbound}
\leanpartial{paper\_xi\_golden\_window\_saturation\_upperbound}{The current Lean bound is false at the threshold case \texttt{X = 0}, so a positivity hypothesis on \texttt{X} is required.}
对任意 \(0<\delta<1\) 与 \(0<\varrho<1\)，令
\(
q(\varrho):=\frac{1-\varrho}{1+\varrho}
\)。
则由 \eqref{eq:xi-single-defect-X} 定义的可见窗口半长度 \(X(\varrho,\delta)\) 满足恒等式
\begin{equation}\label{eq:xi-visible-window-q-identity}
\boxed{\;
X(\varrho,\delta)^2
=\max\Bigl\{0,\ \delta\bigl(q(\varrho)^{-1}+q(\varrho)\bigr)-(1+\delta^2)\Bigr\}. \;}
\end{equation}
并且当 \(X(\varrho,\delta)>0\) 时有定量上界
\begin{equation}\label{eq:xi-saturation-upperbound-by-window}
\boxed{\;
0\le 2\pi\delta-J_{\varrho}(\delta)\le \frac{8\delta}{X(\varrho,\delta)}\; }.
\end{equation}
特别地在黄金分辨率链 \(\{\varrho_m\}\) 上，
\(
q(\varrho_m)=\varphi^{-m}
\)
从而
\(
X(\varrho_m,\delta)^2=\delta(\varphi^m+\varphi^{-m})-(1+\delta^2)=\delta\,\varphi^m+O(1)
\)，并由 \eqref{eq:xi-saturation-upperbound-by-window} 得
\(
2\pi\delta-J_{\varrho_m}(\delta)=O_\delta(\varphi^{-m/2})
\)。
\end{corollary}

\begin{proof}
由 \(q=q(\varrho)=\frac{1-\varrho}{1+\varrho}\) 得 \(\varrho=\frac{1-q}{1+q}\)。代入 \eqref{eq:xi-single-defect-X} 并化简得 \eqref{eq:xi-visible-window-q-identity}。
当 \(X(\varrho,\delta)>0\) 时，写 \(X:=X(\varrho,\delta)\)。由 \(\arctan u=\frac{\pi}{2}-\arctan(u^{-1})\) 与 \eqref{eq:xi-single-defect-integrated-closed-form} 得
\[
2\pi\delta-J_{\varrho}(\delta)
=2\Bigl[(1+\delta)\arctan\!\Bigl(\frac{1+\delta}{X}\Bigr)
-(1-\delta)\arctan\!\Bigl(\frac{1-\delta}{X}\Bigr)\Bigr].
\]
用 \(\arctan u\le u\)（\(u\ge 0\)）估计即得 \eqref{eq:xi-saturation-upperbound-by-window}。
最后对黄金链，\eqref{eq:xi-golden-resolution-chain-rho-closed} 给出 \(q(\varrho_m)=\varphi^{-m}\)，代回 \eqref{eq:xi-visible-window-q-identity} 得所述 \(X(\varrho_m,\delta)\) 的闭式与增长率，从而得到结论。证毕。
\end{proof}
