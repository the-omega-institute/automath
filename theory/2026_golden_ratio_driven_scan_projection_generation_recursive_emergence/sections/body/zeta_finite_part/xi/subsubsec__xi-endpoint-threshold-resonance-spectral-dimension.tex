\paragraph{阈值强度与端点谱维数：零阶原子 \texorpdfstring{$\mathfrak m_{-1}$}{m(-1)} 与幂律衰减}
将表示测度在端点处拆分为原子与非原子部分：
\[
\mu=\mathfrak m_{-1}\delta_{-1}+\mu_{\neq -1},
\qquad
\mathfrak m_{-1}:=\mu(\{-1\})\ge 0,
\qquad
\mu_{\neq -1}:=\mu\!\big|_{\TT\setminus\{-1\}}.
\]

\begin{corollary}[原子部分的严格分离]\label{cor:xi-endpoint-atom-separation}
\leanverified{paper\_xi\_endpoint\_atom\_separation}
在定理 \ref{thm:xi-endpoint-heat-kernel-probe} 的设定下，对每个 \(N\ge 0\) 有
\[
\boxed{\;
a_N
=\mathfrak m_{-1}
\;+\;
\int_{\TT\setminus\{-1\}}\left|\frac{1-\xi}{2}\right|^{2N}\,d\mu_{\neq -1}(\xi)\;}
\]
并且 \(a_N-\mathfrak m_{-1}\downarrow 0\)。
因此 \(\mathfrak m_{-1}\) 为端点阈值承载的唯一“零阶强度参数”，而端点的其余可见结构全部体现在余项 \(a_N-\mathfrak m_{-1}\) 的衰减律中。
\end{corollary}

\begin{proof}
由测度分解与 \eqref{eq:xi-endpoint-heat-kernel-identity} 立得第一式；再由 \eqref{eq:xi-endpoint-heat-kernel-limit} 得 \(a_N\downarrow \mathfrak m_{-1}\)，从而余项单调下降至 \(0\)。证毕。
\end{proof}

\begin{theorem}[端点谱维数的 Laplace 分型]\label{thm:xi-endpoint-spectral-dimension-laplace}
设 \(\mu_{\neq -1}\) 在 \(\xi=-1\) 的邻域满足局部幂律：存在 \(\alpha>-1\) 与常数 \(K_\pm\ge 0\)，使得当 \(\varepsilon\downarrow 0\) 时
\[
\mu_{\neq -1}\bigl(\{\pi<\arg\xi<\pi+\varepsilon\}\bigr)
=K_+\,\varepsilon^{\alpha+1}+o(\varepsilon^{\alpha+1}),
\qquad
\mu_{\neq -1}\bigl(\{\pi-\varepsilon<\arg\xi<\pi\}\bigr)
=K_-\,\varepsilon^{\alpha+1}+o(\varepsilon^{\alpha+1}).
\]
则端点探针余项满足幂律渐近
\begin{equation}\label{eq:xi-endpoint-laplace-asymptotic}
\boxed{\;
a_N-\mathfrak m_{-1}
\sim
(\alpha+1)\,2^{\alpha}\,
\Gamma\!\left(\frac{\alpha+1}{2}\right)\,
(K_++K_-)\,N^{-(\alpha+1)/2}\qquad(N\to\infty).}
\end{equation}
特别地，若 \(\mu\) 在 \(\xi=-1\) 邻域绝对连续，且角密度 \(f(\theta)\) 在 \(\theta=\pi\) 处连续且 \(f(\pi)>0\)，则 \(\alpha=0\)、\(K_+=K_-=f(\pi)\)，从而
\[
\boxed{\;
a_N-\mathfrak m_{-1}\sim 2f(\pi)\sqrt{\frac{\pi}{N}}\qquad(N\to\infty).}
\]
\end{theorem}

\begin{proof}
写 \(\xi=e^{\mathrm{i}\theta}=-e^{\mathrm{i}\varphi}\)（\(\theta=\pi+\varphi\)、\(|\varphi|\ll 1\)），则
\[
\left|\frac{1-\xi}{2}\right|
=\left|\sin\frac{\theta}{2}\right|
=\left|\cos\frac{\varphi}{2}\right|
=1-\frac{\varphi^2}{8}+O(\varphi^4),
\]
从而 \(\left|\frac{1-\xi}{2}\right|^{2N}=\exp\!\bigl(-\frac{N\varphi^2}{4}+O(N\varphi^4)\bigr)\)，主贡献来自 \(|\varphi|\lesssim N^{-1/2}\)。
将 \(a_N-\mathfrak m_{-1}=\int_{\TT\setminus\{-1\}}\left|\frac{1-\xi}{2}\right|^{2N}\,d\mu_{\neq-1}(\xi)\) 分解到 \(\varphi>0\) 与 \(\varphi<0\) 两侧，并用假设把 \(\mu_{\neq-1}\) 的局部增量近似为 \((\alpha+1)K_\pm|\varphi|^{\alpha}\,d\varphi\)。
对每侧作代换 \(t=\frac{N}{4}\varphi^2\) 得
\[
\int_{0}^{\infty}e^{-N\varphi^2/4}\,(\alpha+1)K_+\,\varphi^{\alpha}\,d\varphi
=(\alpha+1)2^{\alpha}K_+\,N^{-(\alpha+1)/2}\,
\Gamma\!\left(\frac{\alpha+1}{2}\right),
\]
负侧同理并将 \(K_+\) 替换为 \(K_-\)，合并即得 \eqref{eq:xi-endpoint-laplace-asymptotic}；绝对连续情形取 \(\alpha=0\) 即得特例。证毕。
\end{proof}

\begin{theorem}[对数端点窗的 Laplace--Stieltjes 渐近]\label{thm:xi-endpoint-logwindow-laplace-stieltjes}
设 \(\mu_{\neq -1}\) 在端点 \(\xi=-1\) 的对称角窗
\[
I_{\varepsilon}:=\Bigl\{\xi=e^{\mathrm{i}\theta}\in\TT:\ |\theta-\pi|<\varepsilon\Bigr\}\qquad(0<\varepsilon<\pi)
\]
满足对数型边界层展开：存在常数 \(A_0,B_0\in\RR\) 使得当 \(\varepsilon\downarrow 0\) 时
\begin{equation}\label{eq:xi-endpoint-logwindow-mass-expansion}
\mu_{\neq -1}(I_{\varepsilon})
=A_0\,\varepsilon\log\frac{1}{\varepsilon}+B_0\,\varepsilon+o(\varepsilon).
\end{equation}
则端点探针余项满足精确两项渐近
\begin{equation}\label{eq:xi-endpoint-logwindow-probe-asymptotic}
\boxed{\;
a_N-\mathfrak m_{-1}
=\frac{\sqrt{\pi}}{\sqrt N}\left[\frac{A_0}{2}\bigl(\log N+\gamma_{\mathrm E}-2\bigr)+B_0\right]
+o\!\left(\frac{1}{\sqrt N}\right)\qquad(N\to\infty),\;}
\end{equation}
其中 \(\gamma_{\mathrm E}\) 为 Euler--Mascheroni 常数。
\end{theorem}
\begin{proof}
令
\(
M(\varepsilon):=\mu_{\neq -1}(I_{\varepsilon})
\)
（则 \(M(\varepsilon)=o(1)\) 且 \(M(0)=0\)）。
由 \(\left|\frac{1-e^{\mathrm{i}\theta}}{2}\right|=\left|\sin(\theta/2)\right|
=\left|\cos\bigl((\theta-\pi)/2\bigr)\right|\)，
将 \(\mu_{\neq -1}\) 推前到端点角距变量 \(\varepsilon=|\theta-\pi|\) 得到 Stieltjes 表示
\[
a_N-\mathfrak m_{-1}
=\int_{0}^{\pi}\cos^{2N}\!\left(\frac{\varepsilon}{2}\right)\,\dd M(\varepsilon).
\]
对该 Stieltjes 积分分部积分（注意 \(\cos(\pi/2)=0\) 且 \(M(0)=0\)）得到
\[
a_N-\mathfrak m_{-1}
=-\int_{0}^{\pi}M(\varepsilon)\,\dd\!\left(\cos^{2N}\!\left(\frac{\varepsilon}{2}\right)\right)
=N\int_{0}^{\pi}M(\varepsilon)\tan\!\left(\frac{\varepsilon}{2}\right)\cos^{2N}\!\left(\frac{\varepsilon}{2}\right)\,\dd\varepsilon.
\]
由于 \(\cos^{2N}(\varepsilon/2)\) 在 \(\varepsilon\gg N^{-1/2}\) 上指数衰减，主贡献来自 \(\varepsilon=O(N^{-1/2})\)。
作缩放 \(\varepsilon=2x/\sqrt N\) 并用渐近
\[
\tan\!\left(\frac{x}{\sqrt N}\right)=\frac{x}{\sqrt N}+O\!\left(\frac{x^{3}}{N^{3/2}}\right),
\qquad
\cos^{2N}\!\left(\frac{x}{\sqrt N}\right)=e^{-x^{2}}\left(1+O\!\left(\frac{x^{4}}{N}\right)\right),
\]
可得
\begin{equation}\label{eq:xi-endpoint-logwindow-scaled-integral}
a_N-\mathfrak m_{-1}
=2\int_{0}^{\infty}x\,M\!\left(\frac{2x}{\sqrt N}\right)e^{-x^{2}}\,\dd x
+o\!\left(\frac{1}{\sqrt N}\right).
\end{equation}
将 \eqref{eq:xi-endpoint-logwindow-mass-expansion} 代入：
\[
M\!\left(\frac{2x}{\sqrt N}\right)
=\frac{2x}{\sqrt N}\left(A_0\log\frac{\sqrt N}{2x}+B_0+o(1)\right),
\]
并代回 \eqref{eq:xi-endpoint-logwindow-scaled-integral}，得到
\[
a_N-\mathfrak m_{-1}
=\frac{4}{\sqrt N}\left[
A_0\int_{0}^{\infty}x^{2}e^{-x^{2}}\log\frac{\sqrt N}{2x}\,\dd x
+B_0\int_{0}^{\infty}x^{2}e^{-x^{2}}\,\dd x
\right]
+o\!\left(\frac{1}{\sqrt N}\right).
\]
其中
\(
\int_{0}^{\infty}x^{2}e^{-x^{2}}\,\dd x=\frac{\sqrt{\pi}}{4}
\)；
并且令
\(
J:=\int_{0}^{\infty}x^{2}e^{-x^{2}}\log x\,\dd x
\)，代换 \(t=x^{2}\) 得
\[
J=\frac14\int_{0}^{\infty}t^{1/2}e^{-t}\log t\,\dd t=\frac14\Gamma'(3/2)
=\frac{\sqrt{\pi}}{8}\,\psi(3/2),
\]
其中 \(\psi=\Gamma'/\Gamma\) 为 digamma 函数；由 \(\psi(3/2)=2-\gamma_{\mathrm E}-2\log 2\) 得
\[
\int_{0}^{\infty}x^{2}e^{-x^{2}}\log\frac{\sqrt N}{2x}\,\dd x
=\frac{\sqrt{\pi}}{8}\bigl(\log N+\gamma_{\mathrm E}-2\bigr).
\]
代回即得 \eqref{eq:xi-endpoint-logwindow-probe-asymptotic}。证毕。
\end{proof}

\begin{corollary}[视界谱测度的端点探针常数项]\label{cor:xi-horizon-mu-endpoint-probe-constant}
\leanverified{paper\_xi\_horizon\_mu\_endpoint\_probe\_constant}
假设 RH 成立并令 \(\mu_{\zeta}\) 为附录定理 \ref{thm:app-horizon-spectral-measure-atomic} 的视界谱测度。
则 \(\mu_{\zeta}(\{-1\})=0\)，并且端点探针余项满足
\begin{equation}\label{eq:xi-horizon-mu-endpoint-probe-logwindow}
\boxed{\;
a_N
=\frac{1}{4\sqrt{\pi N}}\bigl(\log N+\gamma_{\mathrm E}-2\log\pi\bigr)
+o\!\left(\frac{1}{\sqrt N}\right)\qquad(N\to\infty).\;}
\end{equation}
\end{corollary}
\begin{proof}
由 \eqref{eq:app-horizon-mu-atomic}，\(\mu_{\zeta}\) 的支撑为 \(\{\xi_{\gamma},\xi_{\gamma}^{-1}:\gamma\in\Gamma_+\}\subset\TT\setminus\{-1\}\)，故 \(\mu_{\zeta}(\{-1\})=0\)。
又由附录定理 \ref{thm:app-horizon-endpoint-window-mass}，
\[
\mu_{\zeta}(I_{\varepsilon})
=\frac{\varepsilon}{2\pi}\left(\log\frac{1}{\pi\varepsilon}+1\right)+o(\varepsilon)
=\frac{1}{2\pi}\varepsilon\log\frac{1}{\varepsilon}+\frac{1-\log\pi}{2\pi}\varepsilon+o(\varepsilon).
\]
将 \(A_0=\frac{1}{2\pi}\)、\(B_0=\frac{1-\log\pi}{2\pi}\)、\(\mathfrak m_{-1}=0\) 代入定理 \ref{thm:xi-endpoint-logwindow-laplace-stieltjes} 的渐近式并化简，即得 \eqref{eq:xi-horizon-mu-endpoint-probe-logwindow}。证毕。
\end{proof}

\begin{proposition}[对数端点窗诱导的深度下界]\label{prop:xi-endpoint-logwindow-depth-lower-bound}
在定理 \ref{thm:xi-endpoint-logwindow-laplace-stieltjes} 的设定下，进一步假设 \(A_0>0\)。
则当 \(\varepsilon\downarrow 0\) 时，若某个 \(N\) 满足
\(
|a_N-\mathfrak m_{-1}|\le \varepsilon
\)，
则必有
\begin{equation}\label{eq:xi-endpoint-logwindow-depth-lb}
N\ \ge\ \frac{\pi A_0^{2}}{4\varepsilon^{2}}(\log N)^{2}\,(1+o(1)).
\end{equation}
特别地，对 \(\mu=\mu_{\zeta}\)（从而 \(A_0=\frac{1}{2\pi}\)）得到
\[
N\ \ge\ \frac{1}{16\pi\,\varepsilon^{2}}(\log N)^{2}\,(1+o(1)).
\]
\end{proposition}
\begin{proof}
由 \eqref{eq:xi-endpoint-logwindow-probe-asymptotic}，
\[
|a_N-\mathfrak m_{-1}|
=\left(\frac{A_0\sqrt{\pi}}{2}+o(1)\right)\frac{\log N}{\sqrt N},
\]
从而条件 \(|a_N-\mathfrak m_{-1}|\le \varepsilon\) 蕴含
\(
\sqrt N\ge \left(\frac{A_0\sqrt{\pi}}{2}+o(1)\right)\frac{\log N}{\varepsilon}
\)。
平方并整理即得 \eqref{eq:xi-endpoint-logwindow-depth-lb}；\(\mu_{\zeta}\) 的特例代入 \(A_0=\frac{1}{2\pi}\) 得到。证毕。
\end{proof}

\begin{theorem}[红移权重尾项的主项可积化]\label{thm:xi-redshift-tail-main-term}
令 \(N_{\zeta}(T)\) 表示黎曼 \(\zeta\) 的非平凡零点计数函数（取 \(0<\gamma\le T\)，按重数计），并定义红移权重尾和
\[
\mathcal{R}(T):=\sum_{\gamma>T}\frac{\mathrm{mult}(\gamma)}{1+\gamma^2}\qquad(T>0).
\]
则当 \(T\to\infty\) 时有显式主项
\[
\boxed{\;
\mathcal{R}(T)=\frac{1}{2\pi}\frac{\log\!\bigl(T/(2\pi)\bigr)+1}{T}
+O\!\left(\frac{\log T}{T^{2}}\right).\;}
\]
\end{theorem}
\begin{proof}
记 Riemann--von Mangoldt 主项
\[
M(t):=\frac{t}{2\pi}\log\frac{t}{2\pi}-\frac{t}{2\pi},
\qquad
M'(t)=\frac{1}{2\pi}\log\frac{t}{2\pi}.
\]
令误差项 \(E(t):=N_{\zeta}(t)-M(t)\)，则 \(E(t)=O(\log t)\)。
因此
\[
\mathcal{R}(T)
=\int_{T}^{\infty}\frac{1}{1+t^{2}}\,\dd M(t)+\int_{T}^{\infty}\frac{1}{1+t^{2}}\,\dd E(t).
\]
对 \(\int \frac{1}{1+t^2}\dd E(t)\) 分部积分并用 \(E(t)=O(\log t)\) 与 \(\frac{\dd}{\dd t}(1+t^2)^{-1}=O(t^{-3})\) 得
\(
\int_{T}^{\infty}\frac{1}{1+t^{2}}\,\dd E(t)=O((\log T)/T^{2})
\)。
对主项积分分部积分两次并利用 \(\frac{\dd}{\dd t}(1+t^{2})^{-1}=-2t(1+t^{2})^{-2}\)，得到边界项抵消后的恒等式
\[
\int_{T}^{\infty}\frac{1}{1+t^{2}}\,\dd M(t)
=\int_{T}^{\infty}\frac{M'(t)}{1+t^{2}}\,\dd t
=\frac{1}{2\pi}\int_{T}^{\infty}\frac{\log(t/(2\pi))}{1+t^{2}}\,\dd t.
\]
最后用 \((1+t^{2})^{-1}=t^{-2}+O(t^{-4})\) 得
\[
\int_{T}^{\infty}\frac{\log(t/(2\pi))}{1+t^{2}}\,\dd t
=\int_{T}^{\infty}\frac{\log(t/(2\pi))}{t^{2}}\,\dd t
+O\!\left(\int_{T}^{\infty}\frac{\log t}{t^{4}}\,\dd t\right)
=\frac{\log(T/(2\pi))+1}{T}
+O\!\left(\frac{\log T}{T^{3}}\right).
\]
合并即得所示渐近。证毕。
\end{proof}

\paragraph{视界重整化的本征指数：端点原子模态与连续模态的互补收缩率}
附录 \eqref{eq:scale-psi-m} 的 Möbius 半群 \(\{\psi_m\}_{m>0}\subset\mathrm{Aut}(\mathbb{D})\) 在端点坐标中严格线性化，并满足
\(B^{-1}(\psi_m(w))=m\,B^{-1}(w)\)（定理 \ref{thm:xi-endpoint-poisson-weight-scaling}）。
另一方面，Carath\'eodory 表示给出 \(\Re \mathscr{C}_{\zeta}(w)=\int_{\TT}P_w(\xi)\,d\mu(\xi)\)（Poisson 积分）。
将二者合并可把端点谱维数 \(d_{-1}\) 提升为重整化轨道上的本征收缩指数。

\begin{theorem}[端点连续谱的重整化指数]\label{thm:xi-endpoint-renormalization-index}
设 RH 成立，使 \(\mathscr{C}_{\zeta}\) 为 Carath\'eodory 并由测度 \(\mu\) 表示。
令 \(m>1\)，并记 \(\psi_{m^n}=\psi_m^{\circ n}\)。
在端点谱维数口径（定义 \ref{def:xi-endpoint-spectral-dimension}）下，对任意紧子集 \(K\Subset\mathbb{D}\) 存在常数 \(C_K>0\)，使得对所有 \(n\ge 0\) 有
\[
\sup_{w\in K}\left|
m^{-n}\Re \mathscr{C}_{\zeta}\!\bigl(\psi_{m^n}(w)\bigr)
-\mathfrak m_{-1}\,B^{-1}(w)
\right|
\le C_K\,m^{-n d_{-1}/2}.
\]
特别地：若 \(\mathfrak m_{-1}>0\)，则端点原子产生不衰减阈值模态
\(
m^{-n}\Re \mathscr{C}_{\zeta}(\psi_{m^n}(w))\to \mathfrak m_{-1}B^{-1}(w)
\)；
若 \(\mathfrak m_{-1}=0\)，则 \(m^{-n}\Re \mathscr{C}_{\zeta}(\psi_{m^n}(w))=O(m^{-n d_{-1}/2})\)。
\end{theorem}

\begin{proof}
由 Poisson 表示，
\(
\Re \mathscr{C}_{\zeta}(\psi_{m^n}(w))
=\int_{\TT}P_{\psi_{m^n}(w)}(\xi)\,d\mu(\xi)
\)。
对端点原子项 \(\xi=-1\)，由定理 \ref{thm:xi-endpoint-poisson-weight-scaling} 得
\(
P_{\psi_{m^n}(w)}(-1)=m^n P_w(-1)=m^n B^{-1}(w)
\)，故其在 \(m^{-n}\) 归一化后恰为 \(\mathfrak m_{-1}B^{-1}(w)\)。
对余项，\(w\in K\) 时 \(B^{-1}(w)\) 有界，从而 \(\psi_{m^n}(w)\) 的端点层深度与 \(m^n\) 同阶；
Poisson 核在端点邻域的有效角宽度为 \(m^{-n/2}\)，其对 \(\mu_{\neq-1}\) 的作用等价于一类端点热核平均。
由定理 \ref{thm:xi-endpoint-spectral-dimension-laplace} 的 Laplace 幂律分型即可得到 \(m^{-n d_{-1}/2}\) 的一致衰减界。证毕。
\end{proof}

\paragraph{多通道 strictification：算子值端点原子与完全正谱维数}
在 \(\mathcal{A}=\CC[G^{\mathrm{BC}}_m]\) 的 strictification 可见商上，完全正 Carath\'eodory 族由算子值 Herglotz 表示与块 Toeplitz--PSD 完备性刻画（定理 \ref{thm:xi-operator-herglotz-toeplitz-completeness} 与定理 \ref{thm:xi-gbc-maximal-cp-visible-realization}）。
端点探针可按同一二项式模板在块 Toeplitz 层级上定义，从而把“端点阈值承载”提升为完全正、逐态可审计的不变量族。

\begin{theorem}[算子值端点原子与谱维数]\label{thm:xi-operator-endpoint-heat-kernel-dimension}
令 \(\mathcal{A}\) 为有限维 \(C^\ast\)-代数，并令 \(F:\mathbb{D}\to\mathcal{A}\) 为算子值 Carath\'eodory 函数。
写 \(F(w)=c_0+2\sum_{n\ge 1}c_n w^n\) 且令块 Toeplitz 截断
\(
\mathbb{T}_N(F):=(c_{j-k})_{0\le j,k\le N}\in M_{N+1}(\mathcal{A})
\)。
对二项式向量 \(b^{(N)}\) 定义算子值端点探针
\[
\boxed{\;
\mathbf{a}_N
:=\frac{(b^{(N)}\otimes I)^\ast\,\mathbb{T}_N(F)\,(b^{(N)}\otimes I)}{4^N}\ \in\ \mathcal{A}_+.\;}
\]
则存在 \(\mathcal{A}\)-值正测度 \(\boldsymbol\mu\) 使 \(F\) 具有 Herglotz 表示 \eqref{eq:xi-operator-herglotz}，并存在端点原子
\(\mathbf{m}_{-1}:=\boldsymbol\mu(\{-1\})\in\mathcal{A}_+\) 使得
\[
\boxed{\;
\mathbf{a}_N\downarrow \mathbf{m}_{-1}\qquad(N\to\infty)\;}
\]
（强算子拓扑；等价地对任意态 \(\omega\) 有 \(\omega(\mathbf{a}_N)\downarrow \omega(\mathbf{m}_{-1})\)）。
若进一步对每个态 \(\omega\) 诱导的标量测度 \(\mu_{\omega}:=\omega\circ\boldsymbol\mu\) 在端点具有谱维数 \(d_{-1}(\omega)\)，则
\[
\omega(\mathbf{a}_N-\mathbf{m}_{-1})\asymp N^{-d_{-1}(\omega)/2},
\]
从而 strictification 后的端点谱维数成为完全正且逐态可核验的不变量族。
\end{theorem}

\begin{proof}
由定理 \ref{thm:xi-operator-herglotz-toeplitz-completeness}，\(F\) 存在 \(\mathcal{A}\)-值正测度表示。
对任意态 \(\omega\)，复合得到标量 Carath\'eodory 函数 \(\omega\circ F\) 及其正测度 \(\mu_{\omega}\)；
将定理 \ref{thm:xi-endpoint-heat-kernel-probe} 与定理 \ref{thm:xi-endpoint-spectral-dimension-laplace} 施加于 \(\mu_{\omega}\)，得 \(\omega(\mathbf{a}_N)\downarrow \omega(\mathbf{m}_{-1})\) 及幂律估计。
有限维 \(C^\ast\)-代数上态分离正元，故得到 \(\mathbf{a}_N\downarrow \mathbf{m}_{-1}\) 的强算子收敛与逐态谱维数结论。证毕。
\end{proof}

\paragraph{非 RH 相的端点聚焦否证：二项式模板与真共振极点的检出门槛}
当 RH 不成立时，\(\mathscr{C}_{\zeta}\) 在 \(\mathbb{D}\) 内出现极点并强制某一级 Toeplitz 截断出现负惯性（定理 \ref{thm:xi-finite-window-true-resonance-index}）。
下述结论说明：在“最近极点主导”的有限型口径下，同一端点二项式模板将不可避免地产生负值，从而形成一份可复制的端点聚焦否证器；其出现门槛由最近极点 \(r_{\min}\)（等价地由多极矩非温和指数 \(\Lambda_{\mathrm{mom}}\)）控制。

\begin{theorem}[端点二项式否证器（最近极点主导口径）]\label{thm:xi-endpoint-binomial-falsifier}
假设 RH 不成立，并设 \(\mathscr{C}_{\zeta}\) 在 \(\mathbb{D}\) 内存在极点 \(w_\ast\) 实现
\(
r_{\min}:=\min\{|w|:\ \mathscr{C}_{\zeta}\ \text{在}\ w\ \text{处有极点}\}\in(0,1)
\)
（见 \eqref{eq:xi-horizon-rmin-pole}）。
若该最深极点在模长上为孤立主导（例如模长最小者唯一），则存在无穷序列 \(N_j\to\infty\) 使
\[
\boxed{\;
(b^{(N_j)})^{\mathsf T}\mathbf{T}_{N_j}\,b^{(N_j)}<0\;}
\]
从而端点二项式模板在某些深度层级上给出严格负值的有限维否证器。
并且其出现门槛满足
\[
N_j\ \gtrsim\ \Lambda_{\mathrm{mom}}^{-1},
\qquad
\Lambda_{\mathrm{mom}}:=-\log r_{\min}>0,
\]
其中 \(\Lambda_{\mathrm{mom}}\) 可由端点 horocycle 深度在高高度区间的二次压缩闭式估计（定理 \ref{thm:xi-horizon-multipole-temperedness-dichotomy}，式 \eqref{eq:xi-horizon-Lambda-closed}--\eqref{eq:xi-horizon-Lambda-asymptotic}）。
\end{theorem}

\begin{proof}
由定理 \ref{thm:xi-horizon-forced-geometric-tail}，最近极点 \(w_\ast\) 在系数中强迫几何尾项
\(
c_n=-\frac{R_\ast}{2w_\ast^{n+1}}+h_n
\)
（\(n\ge 1\)、\(R_\ast\neq 0\)），其中 \(\{h_n\}\) 来自在较大圆盘内解析的余项。
另一方面，二项式向量对应多项式 \(p_N(\xi)=(1-\xi)^N\)，其自相关 Laurent 多项式满足
\(
|1-\xi|^{2N}=\sum_{n=-N}^{N}(-1)^n\binom{2N}{N+n}\xi^n
\)；
因此对任意 Toeplitz 数据均有组合恒等式
\[
(b^{(N)})^{\mathsf T}\mathbf{T}_N b^{(N)}
=\sum_{n=-N}^{N}(-1)^n\binom{2N}{N+n}\,c_n.
\]
将 \(c_n\) 分解为极点主导项与余项后，主导项的幅度为 \(\asymp 4^N r_{\min}^{-(N+1)}\)，而余项至多为 \(\asymp 4^N\)；
故当 \(N\) 大到使 \(r_{\min}^{-N}\) 主导时，符号由主导项的相位振荡决定，从而必在无穷多个 \(N\) 上取负值。
最后门槛由 \(r_{\min}^{-N}\) 压过 \(O(1)\) 余项的条件推出，即 \(N\gtrsim (-\log r_{\min})^{-1}=\Lambda_{\mathrm{mom}}^{-1}\)。证毕。
\end{proof}
