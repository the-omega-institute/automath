\paragraph{端点热核序列的 Abel--Stokes 表示与端点原子的解析恢复}\label{subsubsec:xi-endpoint-heat-kernel-probe-abel-stokes-atom-recovery}
在 \S\ref{subsubsec:xi-endpoint-heat-kernel-probe-toeplitz} 的积分表示、单调极限与谱隙误差界基础上，端点热核序列还可被统一提升为一个解析外推公式。

\begin{theorem}[端点热核序列的 Abel--Stokes 表示与 \texorpdfstring{$(-1)$}{(-1)} 原子质量的解析恢复]\label{thm:xi-endpoint-heat-probe-abel-stokes-atom-recovery}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_abel\_stokes\_atom\_recovery}
沿用 \S\ref{subsubsec:xi-endpoint-heat-kernel-probe-toeplitz} 的记号，令
\[
r(\xi):=\left|\frac{1-\xi}{2}\right|^2\in[0,1],
\qquad
a_N=\int_{\TT}r(\xi)^N\,\dd\mu(\xi),
\qquad
m_{-1}:=\mu(\{-1\}).
\]
定义生成函数
\[
A(z):=\sum_{N\ge 0}a_N z^N,\qquad |z|<1.
\]
则成立严格表示
\begin{equation}\label{eq:xi-endpoint-heat-probe-abel-stokes-genfun}
\boxed{\;
A(z)=\int_{\TT}\frac{1}{1-z\,r(\xi)}\,\dd\mu(\xi).\;}
\end{equation}
并且端点原子质量满足两种完全显式的恢复公式
\begin{equation}\label{eq:xi-endpoint-heat-probe-abel-stokes-recovery}
\boxed{\;
m_{-1}
=
\lim_{z\uparrow 1}(1-z)A(z)
=
\lim_{N\to\infty}\frac1N\sum_{n=0}^{N-1}a_n.\;}
\end{equation}
若进一步存在常数 \(\rho<1\) 使
\[
r(\xi)\le \rho
\qquad
\bigl(\xi\in\operatorname{supp}\mu\setminus\{-1\}\bigr),
\]
则对所有 \(N\ge 0\) 有指数型误差界
\begin{equation}\label{eq:xi-endpoint-heat-probe-abel-stokes-exp-bound}
\boxed{\;
0\le a_N-m_{-1}\le \mu(\TT\setminus\{-1\})\,\rho^N.\;}
\end{equation}
\end{theorem}

\begin{proof}
由命题 \ref{prop:xi-endpoint-heat-probe-toeplitz-herglotz-identity}，
\(
a_N=\int_{\TT}r(\xi)^N\,\dd\mu(\xi)
\)。
当 \(|z|<1\) 时，
\[
\sum_{N\ge 0}|z\,r(\xi)|^N\le \frac{1}{1-|z|},
\]
故可用支配收敛交换求和与积分：
\[
A(z)
=
\sum_{N\ge 0}\int_{\TT}r(\xi)^N z^N\,\dd\mu(\xi)
=
\int_{\TT}\sum_{N\ge 0}\bigl(z\,r(\xi)\bigr)^N\,\dd\mu(\xi)
=
\int_{\TT}\frac{1}{1-z\,r(\xi)}\,\dd\mu(\xi),
\]
即得 \eqref{eq:xi-endpoint-heat-probe-abel-stokes-genfun}。

又因 \(r(\xi)=1\) 当且仅当 \(\xi=-1\)，可把测度分解为
\[
\mu=m_{-1}\delta_{-1}+\mu_{\neq -1}.
\]
于是
\[
A(z)
=
\frac{m_{-1}}{1-z}
+
\int_{\TT\setminus\{-1\}}\frac{1}{1-z\,r(\xi)}\,\dd\mu(\xi).
\]
对第二项，\(\TT\setminus\{-1\}\) 上处处有 \(r(\xi)<1\)，故当 \(z\uparrow 1\) 时它保持有界。
因此
\[
\lim_{z\uparrow 1}(1-z)A(z)=m_{-1}.
\]
另一方面，推论 \ref{cor:xi-endpoint-heat-probe-monotone-limit-gap} 已给出 \(a_N\downarrow m_{-1}\)。
单调收敛序列的 Ces\`aro 平均与原序列具有相同极限，故
\[
\lim_{N\to\infty}\frac1N\sum_{n=0}^{N-1}a_n=m_{-1}.
\]
这便得到 \eqref{eq:xi-endpoint-heat-probe-abel-stokes-recovery}。

若进一步满足 \(r(\xi)\le \rho<1\) 于 \(\operatorname{supp}\mu\setminus\{-1\}\) 上成立，则由推论
\ref{cor:xi-endpoint-heat-probe-monotone-limit-gap} 的同一估计可得
\[
a_N-m_{-1}
=
\int_{\TT\setminus\{-1\}}r(\xi)^N\,\dd\mu(\xi)
\le
\rho^N\,\mu(\TT\setminus\{-1\}),
\]
即得 \eqref{eq:xi-endpoint-heat-probe-abel-stokes-exp-bound}。证毕。
\end{proof}
