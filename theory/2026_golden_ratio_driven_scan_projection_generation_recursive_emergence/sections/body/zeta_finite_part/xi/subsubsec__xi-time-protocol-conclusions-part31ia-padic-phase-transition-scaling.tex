\paragraph{\texorpdfstring{$\pi_\varphi^t$}{pi^t} / \texorpdfstring{$5^t$}{5^t} 层上的连续尺度相图与阈值减半}
在模 \(\pi_\varphi^t\) 与模 \(5^t\) 的精确核计数闭式已经建立之后，还可沿比例尺度 \(t\asymp \lambda q\) 提取出连续指数曲线：\(\mathcal O\)-层在 \(\lambda=1\) 处冻结，而整数层的冻结阈值精确前移到 \(\lambda=\frac12\)。

\begin{theorem}[模 \texorpdfstring{$\pi_\varphi^t$}{pi^t} 层上的核—像指数存在阈值 \texorpdfstring{$\lambda=1$}{lambda=1} 的二次相变]\label{thm:xi-symq-golden-cq-modpi-quadratic-phase-transition}
\leanverified{paper\_xi\_symq\_golden\_cq\_modpi\_quadratic\_phase\_transition}
沿用结论 \ref{con:xi-symq-golden-cq-mod-pi-kernel-image-count} 的记号。固定 \(\lambda>0\)，并令
\[
t_q:=\lfloor \lambda q\rfloor.
\]
则极限
\[
\mathfrak k_{\mathcal O}(\lambda)
:=
\lim_{q\to\infty}\frac{1}{q^2}\log_5\#\ker(C_{q,t_q}),
\qquad
\mathfrak i_{\mathcal O}(\lambda)
:=
\lim_{q\to\infty}\frac{1}{q^2}\log_5\#\mathrm{Im}(C_{q,t_q})
\]
存在，并满足
\[
\boxed{\;
\mathfrak k_{\mathcal O}(\lambda)=
\begin{cases}
\dfrac{\lambda(2-\lambda)}{2},& 0\le \lambda\le 1,\\[6pt]
\dfrac12,& \lambda\ge 1,
\end{cases}
\qquad
\mathfrak i_{\mathcal O}(\lambda)=
\begin{cases}
\dfrac{\lambda^2}{2},& 0\le \lambda\le 1,\\[6pt]
\lambda-\dfrac12,& \lambda\ge 1.
\end{cases}
\;}
\]
其中右端按连续性延拓到 \(\lambda=0\)。

进一步，当 \(q\) 充分大使 \(t_q\ge 1\) 时，若 \(b\) 在 \(V_{t_q}\) 上均匀分布，则
\[
\mathbb P\!\bigl(C_{q,t_q}x=b\ \text{有解}\bigr)
=
\frac{\#\mathrm{Im}(C_{q,t_q})}{\#V_{t_q}}
=
5^{-\Sigma(q,t_q)},
\]
从而
\[
\lim_{q\to\infty}\frac{-1}{q^2}\log_5
\mathbb P\!\bigl(C_{q,t_q}x=b\ \text{有解}\bigr)
=
\mathfrak k_{\mathcal O}(\lambda).
\]
特别地，\(\lambda=1\) 是严格阈值：当 \(0\le \lambda<1\) 时，核与像的指数都呈二次曲线；当 \(\lambda>1\) 时，核指数冻结为 \(1/2\)，而像指数转为线性增长。
\end{theorem}
\begin{proof}
当 \(0<\lambda\le 1\) 时，对充分大的 \(q\) 有 \(1\le t_q\le q\)。由结论 \ref{con:xi-symq-golden-cq-mod-pi-kernel-image-count}，
\[
\Sigma(q,t_q)=\frac{t_q(2q-t_q+1)}{2}
=
\frac{\lambda(2-\lambda)}{2}q^2+o(q^2),
\]
故
\[
\frac{1}{q^2}\log_5\#\ker(C_{q,t_q})
=
\frac{\Sigma(q,t_q)}{q^2}
\longrightarrow
\frac{\lambda(2-\lambda)}{2}.
\]
又因
\[
\#\mathrm{Im}(C_{q,t_q})=5^{\,t_q(q+1)-\Sigma(q,t_q)},
\]
于是
\[
\frac{1}{q^2}\log_5\#\mathrm{Im}(C_{q,t_q})
=
\frac{t_q(q+1)-\Sigma(q,t_q)}{q^2}
\longrightarrow
\lambda-\frac{\lambda(2-\lambda)}{2}
=
\frac{\lambda^2}{2}.
\]

当 \(\lambda\ge 1\) 时，对充分大的 \(q\) 有 \(t_q\ge q\)，故
\[
\Sigma(q,t_q)=\frac{q(q+1)}{2}
=
\frac12 q^2+o(q^2).
\]
因此
\[
\frac{1}{q^2}\log_5\#\ker(C_{q,t_q})\longrightarrow \frac12.
\]
并且
\[
\frac{1}{q^2}\log_5\#\mathrm{Im}(C_{q,t_q})
=
\frac{t_q(q+1)-\Sigma(q,t_q)}{q^2}
\longrightarrow
\lambda-\frac12.
\]
这就证明了两条极限公式；\(\lambda=0\) 的取值由连续延拓给出。

对可解概率，由结论 \ref{con:xi-symq-golden-cq-mod-pi-kernel-image-count} 的二值纤维律，像内每条纤维大小恒为 \(\#\ker(C_{q,t_q})\)，像外无解，因此
\[
\mathbb P\!\bigl(C_{q,t_q}x=b\ \text{有解}\bigr)
=
\frac{\#\mathrm{Im}(C_{q,t_q})}{\#V_{t_q}}
=
5^{-\Sigma(q,t_q)}.
\]
再除以 \(q^2\) 取负对数并取极限，即得所示速率。证毕。
\end{proof}

\begin{theorem}[限制标量到 \texorpdfstring{$\ZZ$}{Z} 后冻结阈值精确减半]\label{thm:xi-symq-golden-cq-mod5-half-threshold-phase-transition}
\leanverified{paper\_xi\_symq\_golden\_cq\_mod5\_half\_threshold\_phase\_transition}
沿用结论 \ref{con:xi-symq-golden-cq-mod-5-kernel-size} 的记号。固定 \(\lambda>0\)，并令
\[
t_q:=\lfloor \lambda q\rfloor.
\]
则极限
\[
\mathfrak k_{\ZZ}(\lambda)
:=
\lim_{q\to\infty}\frac{1}{q^2}\log_5\#\ker(\widetilde C_{q,t_q}),
\qquad
\mathfrak i_{\ZZ}(\lambda)
:=
\lim_{q\to\infty}\frac{1}{q^2}\log_5\#\mathrm{Im}(\widetilde C_{q,t_q})
\]
存在，并满足
\[
\boxed{\;
\mathfrak k_{\ZZ}(\lambda)=
\begin{cases}
2\lambda(1-\lambda),& 0\le \lambda\le \dfrac12,\\[6pt]
\dfrac12,& \lambda\ge \dfrac12,
\end{cases}
\qquad
\mathfrak i_{\ZZ}(\lambda)=
\begin{cases}
2\lambda^2,& 0\le \lambda\le \dfrac12,\\[6pt]
2\lambda-\dfrac12,& \lambda\ge \dfrac12.
\end{cases}
\;}
\]
其中右端同样按连续性延拓到 \(\lambda=0\)。

特别地，\(\mathcal O/(\pi_\varphi^t)\) 层的临界点位于 \(\lambda=1\)，而 \(\ZZ/5^t\ZZ\) 层的冻结阈值恰位于
\[
\lambda=\frac12.
\]
换言之，限制标量 \(\mathcal O\to \ZZ\) 不仅把秩翻倍，而且把 \(5\)-进时间尺度精确压缩了一个因子 \(2\)。
\end{theorem}
\begin{proof}
先设 \(q=2m\) 为偶数。若 \(0<\lambda\le \frac12\)，则对充分大的 \(q\) 有 \(t_q\le m\)，由结论 \ref{con:xi-symq-golden-cq-mod-5-kernel-size} 得
\[
\log_5\#\ker(\widetilde C_{q,t_q})
=
t_q(4m-2t_q+1)
=
2\lambda(1-\lambda)q^2+o(q^2),
\]
故
\[
\frac{1}{q^2}\log_5\#\ker(\widetilde C_{q,t_q})
\longrightarrow
2\lambda(1-\lambda).
\]
若 \(\lambda\ge \frac12\)，则对充分大的 \(q\) 有 \(t_q\ge m\)，于是
\[
\log_5\#\ker(\widetilde C_{q,t_q})
=
2m^2+m
=
\frac12 q^2+o(q^2),
\]
故极限为 \(1/2\)。

若 \(q=2m+1\) 为奇数，则同理由结论 \ref{con:xi-symq-golden-cq-mod-5-kernel-size} 得
\[
t_q(4m-2t_q+3)=2\lambda(1-\lambda)q^2+o(q^2)
\qquad
\left(0<\lambda\le \frac12\right),
\]
以及
\[
2m^2+3m+1=\frac12 q^2+o(q^2)
\qquad
\left(\lambda\ge \frac12\right).
\]
因此两种奇偶情形都给出同一极限函数，这就证明了 \(\mathfrak k_{\ZZ}\) 的分段公式；\(\lambda=0\) 再由连续延拓处理。

对像指数，只需注意
\[
\#W_{t_q}=5^{\,2t_q(q+1)},
\qquad
\#\mathrm{Im}(\widetilde C_{q,t_q})
=
\frac{\#W_{t_q}}{\#\ker(\widetilde C_{q,t_q})}.
\]
于是
\[
\frac{1}{q^2}\log_5\#\mathrm{Im}(\widetilde C_{q,t_q})
\longrightarrow
2\lambda-\mathfrak k_{\ZZ}(\lambda),
\]
代入上一段的分段公式即得
\[
\mathfrak i_{\ZZ}(\lambda)=
\begin{cases}
2\lambda^2,& 0\le \lambda\le \dfrac12,\\[6pt]
2\lambda-\dfrac12,& \lambda\ge \dfrac12.
\end{cases}
\]
证毕。
\end{proof}
