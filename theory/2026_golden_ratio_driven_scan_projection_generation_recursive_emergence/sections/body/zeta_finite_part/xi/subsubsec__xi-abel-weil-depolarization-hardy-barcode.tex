\paragraph{去极化的 Hardy 能量倍率与旋转相关条形码}
本段把去极化算子 \(\Delta_m\) 的“根单位旋转推送”机制具体化为一个完全可计算的 \(H^2\) 能量律，并将通道能量偏差与旋转相关谱在 Hilbert 空间中统一为一条 Parseval 恒等式。

\begin{theorem}[去极化差分的 Hardy 能量主发散律]\label{thm:abel-depolarization-hardy-energy-blowup}
\leanverified{paper\_abel\_depolarization\_hardy\_energy\_blowup}
取整数 \(m\ge 2\) 并假设 \(\mathcal A_{b,\delta}\in H^2(\mathbb D)\)（例如在定理 \ref{thm:abel-harddamped-universal-renorm-constant} 的假设下，对任意 \(\delta>0\) 皆成立）。
定义阻尼版去极化差分
\[
\Delta_m\mathcal A_{b,\delta}(r):=\mathcal A_{b,\delta}(r)-m\,\mathcal A_{b^m,\delta}(r^m).
\]
则有精确能量恒等式
\begin{equation}\label{eq:abel-depolarization-energy-identity}
\boxed{\;
\|\Delta_m\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2
=\sum_{j=1}^{m-1}\|\mathcal A_{b,\delta}^{(m,j)}\|_{H^2(\mathbb D)}^2
+(m-1)^2\|\mathcal A_{b^m,\delta}\|_{H^2(\mathbb D)}^2.\;}
\end{equation}
并且在定理 \ref{thm:abel-harddamped-universal-renorm-constant} 的同一假设下，当 \(\delta\downarrow 0\) 有主发散渐近
\[
\boxed{\;
\|\Delta_m\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2
=\frac{(m-1)\bigl(2+\gamma-\log(4\pi)\bigr)}{2\,\delta\,\log b}+O_{b,m}(1).\;}
\]
从而倍率极限存在且为
\[
\boxed{\;
\lim_{\delta\downarrow 0}\frac{\|\Delta_m\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2}{\|\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2}=m-1.\;}
\]
\end{theorem}

\begin{proof}
由命题 \ref{prop:abel-polyphase-reconstruction} 与 \(\mathcal A_{b^m,\delta}=\mathcal A_{b,\delta}^{(m,0)}\)（注 \ref{rem:abel-hardy-energy-decimation-applied}），有分解
\[
\mathcal A_{b,\delta}(r)=\sum_{j=0}^{m-1}r^j\,\mathcal A_{b,\delta}^{(m,j)}(r^m),
\qquad
\mathcal A_{b^m,\delta}(r^m)=\mathcal A_{b,\delta}^{(m,0)}(r^m).
\]
因此
\[
\Delta_m\mathcal A_{b,\delta}(r)
=\sum_{j=1}^{m-1}r^j\,\mathcal A_{b,\delta}^{(m,j)}(r^m)+(1-m)\,\mathcal A_{b,\delta}^{(m,0)}(r^m).
\]
由命题 \ref{prop:abel-hardy-energy-decimation-orthogonal} 的通道正交性与 \(\|r^j f(r^m)\|_{H^2}=\|f\|_{H^2}\)，得到能量恒等式 \eqref{eq:abel-depolarization-energy-identity}。
\par
在定理 \ref{thm:abel-harddamped-universal-renorm-constant} 的假设下，
\[
\|\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2
=\frac{2+\gamma-\log(4\pi)}{2\,\delta\,\log b}+O_b(1),
\qquad
\|\mathcal A_{b^m,\delta}\|_{H^2(\mathbb D)}^2
=\frac{2+\gamma-\log(4\pi)}{2\,\delta\,\log(b^m)}+O_{b,m}(1).
\]
而 \(\log(b^m)=m\log b\)，故
\(
\|\mathcal A_{b^m,\delta}\|_{H^2}^2=\frac{2+\gamma-\log(4\pi)}{2m\,\delta\,\log b}+O_{b,m}(1)
\)。
将其代入 \eqref{eq:abel-depolarization-energy-identity} 并用
\(
\sum_{j=1}^{m-1}\|\mathcal A_{b,\delta}^{(m,j)}\|_{H^2}^2
=\|\mathcal A_{b,\delta}\|_{H^2}^2-\|\mathcal A_{b^m,\delta}\|_{H^2}^2
\)
（命题 \ref{prop:abel-detail-energy-budget}），化简即得主发散系数为
\(
\frac{(m-1)(2+\gamma-\log(4\pi))}{2\,\delta\,\log b}
\)
并得到倍率极限。证毕。
\end{proof}

\begin{proposition}[加性角色谱的旋转相关实现]\label{prop:abel-character-spectrum-as-rotation-correlation}
\leanverified{paper\_abel\_character\_spectrum\_as\_rotation\_correlation}
令 \(\omega\in\TT\) 并定义 Hardy 空间上的旋转算子
\[
(U_\omega f)(r):=f(\omega r).
\]
则 \(U_\omega\) 为 \(H^2(\mathbb D)\) 上的酉算子。
并且对任意 \(\delta>0\) 有
\[
\boxed{\;
\langle \mathcal A_{b,\delta},U_\omega\mathcal A_{b,\delta}\rangle_{H^2(\mathbb D)}
=\sum_{n\ge 0}|E_b(n)|^2\,b^{-2\delta n}\,\omega^n.\;}
\]
若取整数 \(m\ge 2\) 且 \(\omega=e^{2\pi i/m}\)，则对任意加性角色 \(\chi_a(n):=\omega^{a n}\)（\(a=0,1,\dots,m-1\)）有
\[
\boxed{\;
\mathcal F_{\chi_a}(\delta)
=\frac{\langle \mathcal A_{b,\delta},U_{\omega^{a}}\mathcal A_{b,\delta}\rangle_{H^2(\mathbb D)}}
{\|\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2},\;}
\]
其中 \(\mathcal F_\chi(\delta)\) 由命题 \ref{prop:abel-channel-equipartition-character} 定义。
\end{proposition}

\begin{proof}
设 \(\mathcal A_{b,\delta}(r)=\sum_{n\ge 0}a_nr^n\)，其中 \(a_n:=E_b(n)b^{-\delta n}\)。
则
\(
U_\omega \mathcal A_{b,\delta}(r)=\sum_{n\ge 0}a_n\omega^n r^n
\)。
由 Hardy 内积的系数刻画 \(\langle f,g\rangle=\sum_{n\ge 0}a_n\overline{b_n}\) 得
\[
\langle \mathcal A_{b,\delta},U_\omega\mathcal A_{b,\delta}\rangle
=\sum_{n\ge 0}a_n\overline{a_n\omega^n}
=\sum_{n\ge 0}|a_n|^2\omega^n,
\]
即所述恒等式。
取 \(\omega=e^{2\pi i/m}\) 并注意 \(\chi_a(n)=\omega^{a n}\) 即得归一化表达。证毕。
\end{proof}

\begin{corollary}[同余通道能量偏差的 Parseval 恒等式]\label{cor:abel-congruence-energy-parseval-barcode}
\leanverified{paper\_abel\_congruence\_energy\_parseval\_barcode}
固定整数 \(m\ge 2\)，并定义同余通道能量占比向量
\[
p_j(\delta):=\frac{\|\mathcal A_{b,\delta}^{(m,j)}\|_{H^2(\mathbb D)}^2}{\|\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2},
\qquad j=0,1,\dots,m-1.
\]
则 \(p(\delta)\) 为概率向量，并且对 \(\omega=e^{2\pi i/m}\) 有精确 Parseval 恒等式
\[
\boxed{\;
m\sum_{j=0}^{m-1}\left|p_j(\delta)-\frac1m\right|^2
=\sum_{a=1}^{m-1}\left|
\frac{\langle \mathcal A_{b,\delta},U_{\omega^{a}}\mathcal A_{b,\delta}\rangle_{H^2(\mathbb D)}}
{\|\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}^2}
\right|^2.\;}
\]
\end{corollary}

\begin{proof}
由命题 \ref{prop:abel-channel-equipartition-character} 的证明，\(\mathcal F_{\chi_a}(\delta)=\sum_{j=0}^{m-1}p_j(\delta)\omega^{aj}\) 为 \(p(\delta)\) 的离散 Fourier 系数。
对 \(a=0,1,\dots,m-1\) 施用离散 Parseval 得
\[
\sum_{a=0}^{m-1}\left|\sum_{j=0}^{m-1}p_j(\delta)\omega^{aj}\right|^2
=m\sum_{j=0}^{m-1}|p_j(\delta)|^2.
\]
注意 \(a=0\) 项为 \(|\sum_j p_j(\delta)|^2=1\)，并用
\(
\sum_{j=0}^{m-1}\left|p_j(\delta)-\frac1m\right|^2=\sum_j|p_j(\delta)|^2-\frac1m
\)
整理得到
\[
m\sum_{j=0}^{m-1}\left|p_j(\delta)-\frac1m\right|^2
=\sum_{a=1}^{m-1}|\mathcal F_{\chi_a}(\delta)|^2.
\]
最后由命题 \ref{prop:abel-character-spectrum-as-rotation-correlation} 把 \(\mathcal F_{\chi_a}(\delta)\) 识别为归一化旋转相关系数即得结论。证毕。
\end{proof}
