\subsubsection{轴交换的 Cayley 互译与 Stieltjes--Hankel 分层正性：视界 Weyl 剖面的单变量证书}\label{subsubsec:xi-axis-swap-stieltjes-hankel-positivity}
附录 \ref{subsubsec:app-horizon-weyl-carath-toeplitz} 已给出视界 Weyl 函数 \(m_{\zeta}\) 的 Herglotz 正性与圆盘 Carath\'eodory 正性之间的 Cayley 精确互译（式 \eqref{eq:app-horizon-m-c-relation}）。
该互译把上半平面 \(\mathbb{H}\) 上的虚部正性推送为单位圆盘 \(\mathbb{D}\) 上的实部正性，从而为“实/虚轴交换”提供一个严格且可核验的数学对应物。
本段在同一互译框架下进一步对 \(\Im m_{\zeta}\) 引入平方根剖面变量，将 RH 归约为一族单变量 Stieltjes 正性及其 Hankel 分层，并记录离切片零点在该剖面上的极点签名。

\paragraph{Herglotz--Carath\'eodory 的 Cayley 互译（轴交换）}
沿用附录中的记号
\[
E_{\zeta}(z):=\Xi_{\zeta}\!\left(\tfrac12+\mathrm{i}z\right),\qquad
m_{\zeta}(z):=-\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}\qquad(z\in\mathbb{H}),
\]
并令 \(\mathcal{C}:\mathbb{H}\to\mathbb{D}\) 为 Cayley 紧化（\eqref{eq:app-cayley-disk}）。
视界 Carath\'eodory 对数导数函数 \(\mathscr{C}_{\zeta}\) 由 \eqref{eq:app-horizon-carath-def} 定义，且满足严格恒等式
\[
m_{\zeta}(z)=\mathrm{i}\,\mathscr{C}_{\zeta}\!\bigl(\mathcal{C}(z)\bigr),
\qquad
\Re \mathscr{C}_{\zeta}(\mathcal{C}(z))=\Im m_{\zeta}(z)\qquad(z\in\mathbb{H})
\]
（见 \eqref{eq:app-horizon-m-c-relation}）。
因此 \(\Im m_{\zeta}\ge 0\) 与 \(\Re \mathscr{C}_{\zeta}\ge 0\) 完全等价，并给出“轴交换”下正性条件的可操作表述。

\begin{proposition}[Carath\'eodory 接口处的退化障碍]\label{prop:xi-carath-degeneracy-pure-imag-constant}
设 \(C\) 为单位圆盘 \(\mathbb{D}\) 上的解析函数。
若 \(\Re C(w)\equiv 0\) 于 \(\mathbb{D}\)，则 \(C\) 为纯虚常数。
\end{proposition}
\begin{proof}
写 \(C=u+\mathrm{i}v\)，其中 \(u,v\) 为实调和函数。
解析性给出 Cauchy--Riemann 方程 \(u_x=v_y\)、\(u_y=-v_x\)。
若 \(u\equiv 0\)，则 \(u_x=u_y\equiv 0\)，从而 \(v_x=v_y\equiv 0\)，故 \(v\) 为常数，\(C\) 为纯虚常数。证毕。
\end{proof}

\leanverified{paper\_xi\_carath\_degeneracy\_pure\_imag\_constant}

\begin{remark}\label{rem:xi-carath-degeneracy-no-cancellation}
因此，在圆盘 Carath\'eodory/Toeplitz 证书接口下，若强制 \(\Re \mathscr{C}_{\zeta}\equiv 0\)，则 \(\mathscr{C}_{\zeta}\) 必退化为纯虚常数，从而 Toeplitz 层级失去非平凡内容。
相应的正性语义应保持半平面几何 \(\Re \mathscr{C}_{\zeta}\ge 0\) 而非消去其实部。
\end{remark}

\begin{theorem}[Stieltjes--Hankel 分层正性与 RH 的等价]\label{thm:xi-axis-swap-stieltjes-hankel-criterion}
\leanverified{paper\_xi\_axis\_swap\_stieltjes\_hankel\_criterion}
对每个 \(x\in\RR\) 定义单变量剖面函数
\begin{equation}\label{eq:xi-gx-stieltjes-profile}
g_x(u):=\frac{\Im m_{\zeta}\!\left(x+\mathrm{i}\sqrt{u}\right)}{\sqrt{u}},\qquad u>0.
\end{equation}
则下列命题等价：
\begin{enumerate}
  \item RH 成立；
  \item 对每个 \(x\in\RR\)，函数 \(g_x\) 为 Stieltjes 函数：存在常数 \(a_x\ge 0\) 与支撑于 \([0,\infty)\) 的正 Borel 测度 \(\nu_x\)，使得对每个 \(u>0\) 积分 \(\int_{0}^{\infty}\frac{1}{u+t}\,\dd\nu_x(t)\) 收敛并满足
  \[
  g_x(u)=a_x+\int_{0}^{\infty}\frac{1}{u+t}\,\dd\nu_x(t);
  \]
  \item 对每个 \(x\in\RR\)，函数 \(g_x\) 在 \((0,\infty)\) 上完全单调：对任意整数 \(n\ge 0\)、任意 \(u>0\) 有
  \[
  (-1)^n\,\frac{\dd^n}{\dd u^n}g_x(u)\ge 0;
  \]
  \item 对每个 \(x\in\RR\) 与每个 \(u_0>0\)，令
  \[
  M_n(x;u_0):=(-1)^n\left.\frac{\dd^n}{\dd u^n}g_x(u)\right|_{u=u_0}\qquad(n\ge 0),
  \]
  并定义 Hankel 矩阵族与其一次移位族
  \[
  H_N(x;u_0):=\bigl(M_{j+k}(x;u_0)\bigr)_{0\le j,k\le N},\qquad
  \widetilde H_N(x;u_0):=\bigl(M_{j+k+1}(x;u_0)\bigr)_{0\le j,k\le N}.
  \]
  则对所有 \(N\ge 0\) 有 \(H_N(x;u_0)\succeq 0\) 且 \(\widetilde H_N(x;u_0)\succeq 0\)。
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{（RH \(\Rightarrow\) Stieltjes）}
由附录定理 \ref{thm:app-horizon-weyl-herglotz}，RH 成立当且仅当 \(m_{\zeta}\) 为 Herglotz 函数。
因此在 RH 前提下，存在常数 \(a\ge 0\)、\(b\in\RR\) 与满足 \(\int_{\RR}\frac{1}{1+t^2}\,\dd\mu(t)<\infty\) 的正 Borel 测度 \(\mu\) 使得 Nevanlinna 表示
\[
m_{\zeta}(z)=az+b+\int_{\RR}\left(\frac{1}{t-z}-\frac{t}{1+t^2}\right)\dd\mu(t),
\qquad \Im z>0.
\]
取 \(z=x+\mathrm{i}\sqrt{u}\)（\(x\in\RR,u>0\)），注意积分中 \(-\frac{t}{1+t^2}\) 为实数项，得
\[
\Im m_{\zeta}(x+\mathrm{i}\sqrt{u})
=a\sqrt{u}+\int_{\RR}\frac{\sqrt{u}}{(t-x)^2+u}\,\dd\mu(t).
\]
两边除以 \(\sqrt{u}\) 并用 \eqref{eq:xi-gx-stieltjes-profile} 得
\[
g_x(u)=a+\int_{\RR}\frac{1}{(t-x)^2+u}\,\dd\mu(t).
\]
令 \(\tau_x(t):=(t-x)^2\) 并取推前测度 \(\nu_x:=(\tau_x)_\#\mu\)，则 \(\nu_x\) 支撑于 \([0,\infty)\) 且
\[
g_x(u)=a+\int_{0}^{\infty}\frac{1}{u+t}\,\dd\nu_x(t),
\]
从而 \(g_x\) 为 Stieltjes 函数（取 \(a_x=a\)）。\par

\textbf{（Stieltjes \(\Rightarrow\) 完全单调）}
由 Stieltjes 表示逐次求导（可由单调收敛/支配收敛逐次合法化）得
\[
\frac{\dd^n}{\dd u^n}g_x(u)=(-1)^n n!\int_{0}^{\infty}(u+t)^{-(n+1)}\,\dd\nu_x(t),
\]
故 \((-1)^n g_x^{(n)}(u)\ge 0\)（\(\forall n\ge 0,u>0\)），即 \(g_x\) 完全单调。 \par

\textbf{（完全单调族 \(\Rightarrow\) RH）}
若对每个 \(x\in\RR\)，\(g_x\) 完全单调，则特别有 \(g_x(u)\ge 0\)（\(\forall u>0\)）。
令 \(y>0\) 且取 \(u=y^2\)，由 \eqref{eq:xi-gx-stieltjes-profile} 得
\[
\Im m_{\zeta}(x+\mathrm{i}y)=y\,g_x(y^2)\ge 0\qquad(\forall x\in\RR,y>0).
\]
另一方面，\(m_{\zeta}=-E_{\zeta}'/E_{\zeta}\) 在 \(\mathbb{H}\) 上至多亚纯；其极点对应 \(E_{\zeta}\) 在 \(\mathbb{H}\) 内的零点，且留数为负整数。
若存在极点 \(z_0\in\mathbb{H}\)，则在其邻域可写
\(
m_{\zeta}(z)=-\frac{m}{z-z_0}+O(1)
\)
（\(m\in\NN\)）。
取 \(z=z_0+\mathrm{i}t\)（\(t\in\RR\) 且 \(|t|\) 足够小并保持 \(\Im z>0\)），则主部的虚部为 \(m/t\)，在 \(t>0\) 与 \(t<0\) 两侧取相反符号，从而 \(\Im m_{\zeta}\) 不可能在穿孔邻域内保持非负。
与 \(\Im m_{\zeta}\ge 0\) 矛盾，故 \(\mathbb{H}\) 内无极点，\(m_{\zeta}\) 在 \(\mathbb{H}\) 上解析且 \(\Im m_{\zeta}\ge 0\)。
由附录定理 \ref{thm:app-horizon-weyl-herglotz} 得 RH。 \par

\textbf{（完全单调 \(\Rightarrow\) Hankel--PSD）}
由 Bernstein 定理，完全单调函数是某个正测度的 Laplace 变换：存在 \([0,\infty)\) 上的正 Borel 测度 \(\sigma_x\) 使对一切 \(u>0\) 有
\[
g_x(u)=\int_{0}^{\infty}e^{-us}\,\dd\sigma_x(s).
\]
因此对任意 \(u_0>0\) 与 \(n\ge 0\)，
\[
M_n(x;u_0)=(-1)^n g_x^{(n)}(u_0)=\int_{0}^{\infty}s^n e^{-u_0 s}\,\dd\sigma_x(s),
\]
从而 \(\{M_n(x;u_0)\}_{n\ge 0}\) 为 \([0,\infty)\) 上正测度 \(e^{-u_0 s}\dd\sigma_x(s)\) 的矩序列。
由此对任意向量 \(c=(c_0,\dots,c_N)\in\CC^{N+1}\) 有
\[
c^\ast H_N(x;u_0)c
=\int_0^\infty\left|\sum_{k=0}^{N}c_k s^k\right|^2 e^{-u_0 s}\,\dd\sigma_x(s)\ge 0,
\]
故 \(H_N(x;u_0)\succeq 0\)。
同理，
\(
c^\ast \widetilde H_N(x;u_0)c
=\int_0^\infty s\left|\sum_{k=0}^{N}c_k s^k\right|^2 e^{-u_0 s}\,\dd\sigma_x(s)\ge 0
\)，
故 \(\widetilde H_N(x;u_0)\succeq 0\)。\par

\textbf{（Hankel--PSD \(\Rightarrow\) 完全单调）}
固定 \(x\in\RR\) 与 \(u_0>0\)。
由 Stieltjes 矩问题的判别定理，条件 \(H_N(x;u_0)\succeq 0\) 与 \(\widetilde H_N(x;u_0)\succeq 0\)（\(\forall N\)）等价于存在支撑于 \([0,\infty)\) 的正测度 \(\eta_{x,u_0}\) 使
\(
M_n(x;u_0)=\int_{0}^{\infty}s^n\,\dd\eta_{x,u_0}(s)
\)。
故 \(M_n(x;u_0)\ge 0\)（\(\forall n\)），即
\(
(-1)^n g_x^{(n)}(u_0)\ge 0
\)。
由于 \(u_0\) 任意，\(g_x\) 在 \((0,\infty)\) 上完全单调。
\end{proof}

\begin{proposition}[离切片零点的共动极点签名：高度消去后的离线量读出]\label{prop:xi-gx-offcritical-pole-residue}
若 RH 不成立，则存在 \(\Xi_{\zeta}\) 的零点 \(\rho=\beta+\mathrm{i}\gamma\) 满足 \(\beta\ne\tfrac12\)。
由函数方程对称性可取 \(\beta<\tfrac12\)，并记
\[
\delta:=\tfrac12-\beta>0,\qquad z_0:=\gamma+\mathrm{i}\delta\in\mathbb{H}.
\]
则 \(z_0\) 为 \(E_{\zeta}\) 在 \(\mathbb{H}\) 内的零点。
若其在 \(z_0\) 处的重数为 \(m\ge 1\)，则 \eqref{eq:xi-gx-stieltjes-profile} 中的剖面函数 \(g_{\gamma}\) 在 \(u=\delta^2\) 处具有简单极点，且留数为
\[
\Res_{u=\delta^2} g_{\gamma}(u)=2m.
\]
\end{proposition}

\begin{proof}
在 \(z_0\) 的邻域写
\(
E_{\zeta}(z)=(z-z_0)^m h(z)
\)
且 \(h(z_0)\ne 0\)。
则
\[
m_{\zeta}(z)=-\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}
=-\frac{m}{z-z_0}-\frac{h'(z)}{h(z)}.
\]
后项在邻域解析有界。
取 \(z=\gamma+\mathrm{i}\sqrt{u}\)（\(u>0\)）得
\[
-\frac{m}{z-z_0}=-\frac{m}{\mathrm{i}(\sqrt{u}-\delta)}=\mathrm{i}\,\frac{m}{\sqrt{u}-\delta},
\]
其虚部为 \(\frac{m}{\sqrt{u}-\delta}\)。
从而
\[
g_{\gamma}(u)
=\frac{1}{\sqrt{u}}\left(\frac{m}{\sqrt{u}-\delta}+O(1)\right)
=\frac{m}{\sqrt{u}(\sqrt{u}-\delta)}+O(1).
\]
又
\[
\sqrt{u}(\sqrt{u}-\delta)=\frac{u-\delta^2}{\sqrt{u}+\delta},
\qquad
\frac{m}{\sqrt{u}(\sqrt{u}-\delta)}
=\frac{m(\sqrt{u}+\delta)}{\sqrt{u}}\cdot\frac{1}{u-\delta^2}.
\]
当 \(u\to\delta^2\) 时 \(\sqrt{u}\to\delta\)，括号趋于 \(2m\)，故 \(g_{\gamma}\) 在 \(u=\delta^2\) 处为简单极点且留数为 \(2m\)。证毕。
\end{proof}

\leanverified{paper\_xi\_gx\_offcritical\_pole\_residue}

\begin{corollary}[符号翻转的局部否证：单剖面的有限证据]\label{cor:xi-gx-local-sign-flip-falsifier}
在命题 \ref{prop:xi-gx-offcritical-pole-residue} 的设定下，存在 \(\varepsilon>0\) 使得
\[
g_{\gamma}(u)<0\quad(\delta^2-\varepsilon<u<\delta^2),
\qquad
g_{\gamma}(u)>0\quad(\delta^2<u<\delta^2+\varepsilon).
\]
因此，仅由实变量剖面 \(u\mapsto g_{\gamma}(u)\) 在某点处的符号翻转即可给出 RH 失败的局部否证；翻转点 \(u=\delta^2\) 同时编码离线偏移 \(\delta\)。
\end{corollary}

\begin{proof}
由命题 \ref{prop:xi-gx-offcritical-pole-residue}，
\(
g_{\gamma}(u)=\frac{2m}{u-\delta^2}+O(1)
\)
（\(u\to\delta^2\)）。
主部在 \(\delta^2\) 两侧符号相反且发散，取 \(\varepsilon\) 充分小使误差项不改变符号，即得结论。证毕。
\end{proof}

\begin{proposition}[端点 Poisson 权重与径向对数深度的双曲夹逼]\label{prop:xi-endpoint-poisson-hyperbolic-scale-bounds}
对任意 \(a\in\mathbb{D}\) 定义径向对数深度与端点 \(-1\) Poisson 权重
\[
d(a):=\log\frac1{|a|},\qquad
p(a):=P_a(-1)=\frac{1-|a|^2}{|1+a|^2}.
\]
则成立双侧估计
\[
\tanh\!\Bigl(\frac{d(a)}{2}\Bigr)\ \le\ p(a)\ \le\ \coth\!\Bigl(\frac{d(a)}{2}\Bigr),
\]
且下界取等当且仅当 \(a\) 在正实轴上；上界取等当且仅当 \(a\) 在负实轴上。
\end{proposition}

\begin{proof}
令 \(r:=|a|\in(0,1)\)。
由三角不等式得
\(
1-r\le |1+a|\le 1+r
\)。
代入 \(p(a)=\frac{1-r^2}{|1+a|^2}\) 得
\[
\frac{1-r^2}{(1+r)^2}\le p(a)\le \frac{1-r^2}{(1-r)^2},
\]
即
\[
\frac{1-r}{1+r}\le p(a)\le \frac{1+r}{1-r}.
\]
另一方面，\(d(a)=\log(1/r)\) 满足
\[
\tanh\!\Bigl(\frac{d(a)}{2}\Bigr)=\frac{1-r}{1+r},\qquad
\coth\!\Bigl(\frac{d(a)}{2}\Bigr)=\frac{1+r}{1-r},
\]
故得所示不等式。
取等号分别对应 \(|1+a|=1+r\) 与 \(|1+a|=1-r\)，即 \(\arg a=0\) 与 \(\arg a=\pi\)。证毕。
\end{proof}

\paragraph{正性刚性与离线否证的对齐}
本段的剖面化判据把附录中的“视界正性”进一步压缩为单变量 Stieltjes 结构：在 RH 前提下，\(\Im m_{\zeta}\) 的平方根剖面 \eqref{eq:xi-gx-stieltjes-profile} 必须在每条垂直线 \(x=\mathrm{const}\) 上形成 Stieltjes--Hankel 全分层正性（定理 \ref{thm:xi-axis-swap-stieltjes-hankel-criterion}）。
另一方面，一旦存在离切片零点，则该剖面在 \(u=\delta^2\) 处留下不可去极点与强制符号翻转（命题 \ref{prop:xi-gx-offcritical-pole-residue} 与推论 \ref{cor:xi-gx-local-sign-flip-falsifier}），与“全分层正性”不相容。
该结论与内因子相位电流的逐项正性一致：盘内点谱缺陷在相位通道上只能以 Poisson 核的正贡献出现并不可抵消（定理 \ref{thm:app-inner-phase-current-poisson}），而端点权重 \(p(a)=P_a(-1)\) 与径向对数深度 \(d(a)\) 之间的双曲夹逼（命题 \ref{prop:xi-endpoint-poisson-hyperbolic-scale-bounds}）给出二者在端点读出上的可比尺度。
