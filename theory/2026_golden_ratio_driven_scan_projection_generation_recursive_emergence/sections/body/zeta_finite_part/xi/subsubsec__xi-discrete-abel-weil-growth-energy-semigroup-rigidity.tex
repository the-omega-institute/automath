\paragraph{增长指数、临界能量与阻尼半群的统一刻画}
在定义
\[
E_b(n)=b^{-n/2}\bigl(\psi_0(b^n)-b^n\bigr),\qquad
\mathcal A_b(r)=\sum_{n\ge 0}E_b(n)r^n
\]
与
\[
\sigma_*:=\sup_{\rho}\Re\rho
\]
（\(\rho\) 遍历 \(\zeta\) 的全部非平凡零点）不变的前提下，可得到如下严格判别与结构结果。

\begin{theorem}[右端零点实部的增长指数判别式]\label{thm:abel-growth-exponent-criterion}
\leanverified{paper\_abel\_growth\_exponent\_criterion}
固定 \(b>1\)，定义离散增长指数
\[
\kappa_b:=\limsup_{n\to\infty}\frac{1}{n}\log|E_b(n)|.
\]
则有恒等式
\[
\sigma_*=\frac12+\frac{\kappa_b}{\log b}.
\]
因此对任意固定 \(b>1\)，成立
\[
\mathrm{RH}\Longleftrightarrow \kappa_b=0.
\]
\end{theorem}
\begin{proof}
由 Cauchy--Hadamard 定理，
\[
\mathrm{rad}(\mathcal A_b)
=\frac{1}{\limsup_{n\to\infty}|E_b(n)|^{1/n}}
=e^{-\kappa_b}.
\]
另一方面，定理 \ref{thm:radius-captures-sigma-star} 给出
\[
\mathrm{rad}(\mathcal A_b)=b^{-(\sigma_*-\frac12)}.
\]
两式相等并取对数即得
\[
\kappa_b=(\sigma_*-\tfrac12)\log b,\qquad
\sigma_*=\tfrac12+\kappa_b/\log b.
\]
最后一条等价关系由 \(\sigma_*=\tfrac12\iff \mathrm{RH}\) 直接推出。
\end{proof}

\begin{theorem}[\texorpdfstring{$H^2$}{H2} 能量临界指数与右偏距一致]\label{thm:abel-h2-critical-index}
\leanverified{paper\_abel\_h2\_critical\_index}
固定 \(b>1\)，对 \(\delta\ge 0\) 定义阻尼 Abel 生成函数
\[
\mathcal A_{b,\delta}(r):=\mathcal A_b(b^{-\delta}r)
=\sum_{n\ge 0}E_b(n)b^{-\delta n}r^n,
\]
并定义二次能量
\[
\mathcal E_b(\delta):=\sum_{n\ge 0}|E_b(n)|^2b^{-2\delta n}.
\]
记临界指数
\[
\delta_2(b):=\inf\{\delta\ge 0:\ \mathcal E_b(\delta)<\infty\}.
\]
则有精确恒等式
\[
\delta_2(b)=\sigma_*-\frac12.
\]
特别地，
\[
\mathrm{RH}\Longleftrightarrow \delta_2(b)=0.
\]
\end{theorem}
\begin{proof}
对幂级数 \(f(r)=\sum_{n\ge 0}a_nr^n\)，Hardy 空间 Parseval 恒等式给出
\[
\|f\|_{H^2(\mathbb D)}^2=\sum_{n\ge 0}|a_n|^2.
\]
代入 \(a_n=E_b(n)b^{-\delta n}\) 得
\[
\mathcal A_{b,\delta}\in H^2(\mathbb D)
\Longleftrightarrow
\mathcal E_b(\delta)<\infty.
\]
再由定理 \ref{thm:hardy-criterion-delta}，
\[
\mathcal A_{b,\delta}\in H^2(\mathbb D)
\Longleftrightarrow
\sigma_*<\frac12+\delta
\Longleftrightarrow
\delta>\sigma_*-\frac12.
\]
故
\[
\{\delta\ge 0:\mathcal E_b(\delta)<\infty\}
=(\sigma_*-\tfrac12,\infty),
\]
其下确界即 \(\delta_2(b)=\sigma_*-\frac12\)。
\end{proof}

\begin{theorem}[阻尼参数的重整化半群与能量耗散恒等式]\label{thm:abel-damping-semigroup-dissipation}
\leanverified{paper\_abel\_damping\_semigroup\_dissipation}
固定 \(b>1\)。在 \(H^2(\mathbb D)\) 上定义
\[
(T_\delta f)(z):=f(b^{-\delta}z),\qquad \delta\ge 0.
\]
则成立：
\begin{enumerate}
\item \(\{T_\delta\}_{\delta\ge 0}\) 构成强连续收缩半群，即
\[
T_{\delta_1+\delta_2}=T_{\delta_1}T_{\delta_2},\qquad
\|T_\delta f\|_{H^2}\le \|f\|_{H^2}.
\]
\item 其生成元可写为
\[
Gf=-(\log b)\,z f'(z),
\]
定义域可取
\[
D(G)=\Bigl\{f(z)=\sum_{n\ge 0}a_n z^n:\ \sum_{n\ge 0}n^2|a_n|^2<\infty\Bigr\},
\]
并满足 \(\frac{\mathrm d}{\mathrm d\delta}T_\delta f=GT_\delta f\)（\(f\in D(G)\)）。
\item 若 \(\delta_0>\delta_2(b)\)，则轨道
\[
\delta\mapsto \mathcal A_{b,\delta}
=T_{\delta-\delta_0}\mathcal A_{b,\delta_0},
\qquad \delta\ge \delta_0
\]
在每个 \(\delta>\delta_0\) 处可微，且满足
\[
\frac{\mathrm d}{\mathrm d\delta}\|\mathcal A_{b,\delta}\|_{H^2}^2
=-2(\log b)\sum_{n\ge 0}n\,|E_b(n)|^2b^{-2\delta n}.
\]
上式在 \(\delta=\delta_0\) 处按右导数仍成立。
\end{enumerate}
\end{theorem}
\begin{proof}
若 \(f(z)=\sum_{n\ge 0}a_n z^n\)，则
\[
T_\delta f(z)=\sum_{n\ge 0}a_n b^{-\delta n}z^n,
\]
从而
\[
\|T_\delta f\|_{H^2}^2=\sum_{n\ge 0}|a_n|^2b^{-2\delta n}\le \sum_{n\ge 0}|a_n|^2=\|f\|_{H^2}^2,
\]
并且 \(T_{\delta_1+\delta_2}=T_{\delta_1}T_{\delta_2}\) 由复合直接给出。多项式在 \(H^2\) 稠密，且在多项式上 \(\delta\mapsto T_\delta f\) 显然连续，故得强连续收缩半群。

对 \(f\in D(G)\)，
\[
\frac{T_h f(z)-f(z)}{h}
=\sum_{n\ge 1}a_n\frac{b^{-hn}-1}{h}z^n
\xrightarrow[h\downarrow 0]{H^2}
-(\log b)\sum_{n\ge 1}n a_n z^n
=-(\log b)\,z f'(z),
\]
故生成元为所述 \(G\)。

再由 Parseval，
\[
\|\mathcal A_{b,\delta}\|_{H^2}^2
=\sum_{n\ge 0}|E_b(n)|^2b^{-2\delta n}.
\]
当 \(\delta>\delta_0\) 时，
\[
n\,b^{-2\delta n}
=\bigl(n b^{-2(\delta-\delta_0)n}\bigr)b^{-2\delta_0 n},
\]
其中 \(n b^{-2(\delta-\delta_0)n}\) 有界，且
\(\sum_{n\ge 0}|E_b(n)|^2b^{-2\delta_0 n}<\infty\)。
故可逐项求导并得到
\[
\frac{\mathrm d}{\mathrm d\delta}\|\mathcal A_{b,\delta}\|_{H^2}^2
=\sum_{n\ge 0}|E_b(n)|^2\frac{\mathrm d}{\mathrm d\delta}(b^{-2\delta n})
=-2(\log b)\sum_{n\ge 0}n|E_b(n)|^2b^{-2\delta n}.
\]
令 \(\delta\downarrow \delta_0\) 即得右导数形式。
\end{proof}

\begin{proposition}[\texorpdfstring{$H^2$}{H2} 能量控制下的递归截断误差上界]\label{prop:abel-h2-tail-optimal-bound}
\leanverified{paper\_abel\_h2\_tail\_optimal\_bound}
设 \(\delta>\delta_2(b)\)，并记
\[
\mathcal A_{b,\delta}(r)=\sum_{n\ge 0}a_n r^n,\qquad
a_n:=E_b(n)b^{-\delta n}.
\]
对任意整数 \(N\ge 0\) 与任意 \(|r|<1\)，定义
\[
S_N(r):=\sum_{n=0}^{N-1}a_n r^n,\qquad
R_N(r):=\sum_{n\ge N}a_n r^n.
\]
则有
\[
|R_N(r)|
\le
\|\mathcal A_{b,\delta}\|_{H^2(\mathbb D)}\,
\frac{|r|^N}{\sqrt{1-|r|^2}}.
\]
该估计在仅依赖 \(\|\mathcal A_{b,\delta}\|_{H^2}\) 的范数级信息下为同阶尖锐。
\end{proposition}
\begin{proof}
由 Cauchy--Schwarz 不等式，
\[
|R_N(r)|
=\left|\sum_{n\ge N}a_n r^n\right|
\le
\left(\sum_{n\ge N}|a_n|^2\right)^{1/2}
\left(\sum_{n\ge N}|r|^{2n}\right)^{1/2}
\le
\|\mathcal A_{b,\delta}\|_{H^2}
\left(\frac{|r|^{2N}}{1-|r|^2}\right)^{1/2}.
\]
若把 \(H^2\) 球面上的尾部系数方向取为 \(\bar r^n\) 的比例向量，则上述 Cauchy--Schwarz 可逼近取等，故阶数最优。
\end{proof}

\begin{conjecture}[小阻尼极限的谱主导项]\label{conj:abel-h2-small-damping-spectral-leading}
在 RH 与全部非平凡零点单纯性的假设下，固定 \(b>1\)。定义
\[
\mathcal E_b(\delta)=\sum_{n\ge 0}|E_b(n)|^2b^{-2\delta n},\qquad \delta>0.
\]
当 \(\delta\downarrow 0\) 时，期望存在主导渐近
\[
\mathcal E_b(\delta)\sim
\frac{1}{2\delta\log b}\sum_{\rho}\frac{1}{|\rho|^2},
\]
其中求和遍历全部非平凡零点，并按共轭与函数方程对称配对计重。
\end{conjecture}

\begin{conjecture}[Pad\'e--Prony 递归的有限前缀谱恢复]\label{conj:abel-pade-prony-finite-prefix-spectral-recovery}
固定 \(b>1\)。在 RH 假设下，\(\mathcal A_b\) 的边界奇性集中在 \(|r|=1\)，其角频率集合为
\[
\Theta_b:=\{-\gamma\log b\ (\mathrm{mod}\ 2\pi):\ \rho=\tfrac12+i\gamma\}.
\]
期望存在一族近对角 Pad\'e 逼近（等价地，Hankel--Prony 递归系统），仅由有限前缀
\[
\{E_b(n)\}_{0\le n\le N}
\]
构造，使其极点角分布以可量化误差收敛到 \(\Theta_b\)；并且在极点稳定分离条件下，残差系数同时收敛到
\[
\left\{-\frac{1}{\rho}\right\},
\]
与离散 Abel--Weil 分解中的部分分式系数一致。
\end{conjecture}
