\paragraph{显式 \(3\)-挠识别、局部分子平方化与 Kummer 纤维的三次剩余判据}
沿用定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion}、结论 \ref{con:xi-leyang-kummer-cardano-covering-isomorphism} 与定理 \ref{thm:xi-terminal-zm-s3-endoscopic-homology-a2-identification} 的记号。记
\[
C:=H_{\mathrm{LY}},
\qquad
C_6:=C_{S_3},
\qquad
E:=\overline{\mathcal R},
\qquad
K:=\QQ(\sqrt{-3}),
\qquad
s:=\sqrt{-3}.
\]
再定义
\[
\Theta:=\frac{A(y)+3su}{16(2y+1)^3}=\eta^{-1}\in K(C)^\times.
\]
则 \(\Theta\) 与 \(C_6\to C\) 的无分歧循环三次覆盖在 Jacobian、局部因子与有限域纤维三侧可进一步压缩为下列五条后果。

\begin{theorem}[显式 \(3\)-挠子群与无分歧三重扭量的同一性]\label{thm:xi-leyang-explicit-3torsor-generator-identification}
在上述记号下，成立
\[
\operatorname{div}(\Theta)=3(Q_+-Q_-),
\qquad
[Q_+-Q_-]\in J(C)(K)[3]\setminus\{0\}.
\]
并且由无分歧循环三重覆盖
\[
\pi:C_6\longrightarrow C
\]
所确定的 \(3\)-挠循环子群恰为
\[
\boxed{\;
\langle[Q_+-Q_-]\rangle
=
\langle[Q_--Q_+]\rangle
\subset J(C)[3].\;}
\]
等价地，基变换到 \(K\) 后，覆盖 \(\pi_K:C_6\otimes_{\QQ}K\to C\otimes_{\QQ}K\) 双有理同构于 Kummer \(\mu_3\)-扭量
\[
W^3=\Theta,
\]
因而其扭量类正由线丛类 \([Q_+-Q_-]\) 给出。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion}，
\[
\eta=\frac{A(y)-3su}{16(2y+1)^3},
\qquad
\operatorname{div}(\eta)=3(Q_--Q_+).
\]
同一定理又给出 \(\iota(\eta)=\eta^{-1}\)，其中 \(\iota:(y,u)\mapsto(y,-u)\) 为超椭圆对合。于是
\[
\Theta=\eta^{-1}=\iota(\eta)=\frac{A(y)+3su}{16(2y+1)^3},
\]
并且
\[
\operatorname{div}(\Theta)=-\operatorname{div}(\eta)=3(Q_+-Q_-).
\]
故 \([Q_+-Q_-]\) 为 \(K\)-有理非平凡 \(3\)-挠元。

另一方面，结论 \ref{con:xi-leyang-kummer-cardano-covering-isomorphism} 说明
\[
K(C_6)=K(C)(\xi),
\qquad
\xi^3=\eta,
\]
并给出 \(C_6\otimes_{\QQ}K\cong C_\eta\)。令 \(W:=\xi^{-1}\)，则
\[
W^3=\eta^{-1}=\Theta.
\]
因此 \(\pi_K\) 正是由方程 \(W^3=\Theta\) 给出的 Kummer \(\mu_3\)-扭量。Kummer 理论表明：若 \(\operatorname{div}(f)=3D\)，则方程 \(W^3=f\) 所定义的 \(\mu_3\)-扭量由除子类 \([D]\) 给出。此处 \(D=Q_+-Q_-\)，故对应的循环子群正是
\(
\langle[Q_+-Q_-]\rangle
\)；
将生成元取逆不改变该子群，故亦等于 \(\langle[Q_--Q_+]\rangle\)。证毕。
\end{proof}

\begin{theorem}[良约化处局部 Hasse--Weil 因子的平方分解]\label{thm:xi-leyang-local-factor-square-decomposition}
设 \(p\) 为使 \(C_6,C,E\) 同时良约化的素数。记
\[
P_{C_6,p}(T),\qquad P_{C,p}(T),\qquad P_{E,p}(T)
\]
为各自局部 Hasse--Weil \(P\)-因子。则成立精确分解
\[
\boxed{\;
P_{C_6,p}(T)=P_{C,p}(T)\,P_{E,p}(T)^2.\;}
\]
等价地，对任意 \(r\ge 1\)，都有
\[
\boxed{\;
\#C_6(\FF_{p^r})-(p^r+1)
=
\bigl(\#C(\FF_{p^r})-(p^r+1)\bigr)
+2\bigl(\#E(\FF_{p^r})-(p^r+1)\bigr).\;}
\]
\end{theorem}
\begin{proof}
这正是推论 \ref{cor:xi-terminal-zm-s3-endoscopic-pointcount-localzeta-square} 在记号 \(C=H_{\mathrm{LY}}\)、\(C_6=X_V\)、\(E=\overline{\mathcal R}\) 下的直接重述。证毕。
\end{proof}

\begin{theorem}[非平凡 \texorpdfstring{$C_3$}{C3}-Artin 因子与椭圆局部因子的同一性]\label{thm:xi-leyang-c3-artin-factor-equals-elliptic}
设 \(p\) 为使 \(C_6,C,E\) 同时良约化的素数，并取 \(\ell\neq p\)。记
\[
\Gal(C_6/C)=\langle\sigma\rangle\cong C_3,
\]
其两个非平凡复特征记为 \(\chi,\chi^2\)。令
\[
L_p(\chi,T),\qquad L_p(\chi^2,T)
\]
分别表示 Frobenius 在
\[
H^1_{\acute et}(C_{6,\overline{\FF}_p},\QQ_\ell)\otimes_{\QQ}K
\]
的 \(\chi\)-与 \(\chi^2\)-等型部分上的局部 Artin 因子。则有
\[
\boxed{\;
L_p(\chi,T)=L_p(\chi^2,T)=P_{E,p}(T).\;}
\]
因此去掉有限个坏素数 Euler 因子后，成立
\[
\boxed{\;
L(\chi,s)=L(E,s),
\qquad
L(\chi^2,s)=L(E,s).\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-terminal-zm-s3-endoscopic-chow-motive-h1-splitting}，
\[
h^1(C_6)\simeq h^1(C)\oplus h^1(E)^{\oplus 2}.
\]
其中 \(\sigma\) 在第一项 \(h^1(C)\) 上作用为平凡，在 Prym 块 \(h^1(E)^{\oplus 2}\) 上的作用由命题 \ref{prop:xi-terminal-zm-s3-endoscopic-prym-a2-coxeter} 的 Coxeter 矩阵
\[
U=
\begin{pmatrix}
0 & -1\\
1 & -1
\end{pmatrix}
\]
给出，其特征多项式为
\[
T^2+T+1=(T-\zeta_3)(T-\zeta_3^2).
\]
故在 \(K=\QQ(\zeta_3)=\QQ(\sqrt{-3})\) 上，\(U\) 可对角化为两个特征值 \(\zeta_3,\zeta_3^2\) 的直和，从而
\[
h^1(C_6)\otimes_{\QQ}K
\simeq
\bigl(h^1(C)\otimes_{\QQ}K\bigr)\oplus
\bigl(h^1(E)\otimes\chi\bigr)\oplus
\bigl(h^1(E)\otimes\chi^2\bigr).
\]
在良约化处，deck 变换与 Frobenius 对易，因此上述分解与 Frobenius 作用相容。于是 \(\chi\)-与 \(\chi^2\)-等型部分上的 Frobenius 特征多项式都等于 \(P_{E,p}(T)\)，即
\[
L_p(\chi,T)=P_{E,p}(T),
\qquad
L_p(\chi^2,T)=P_{E,p}(T).
\]
对几乎所有良素数逐项相乘，即得到不完全 Euler 乘积的全局恒等式。证毕。
\end{proof}

\begin{theorem}[Kummer 规范形中纤维可解性的三次剩余判据]\label{thm:xi-leyang-kummer-fiber-cubic-residue-criterion}
令
\[
U:=C\setminus\{Q_+,Q_-\},
\qquad
V:=\pi^{-1}(U)\subset C_6.
\]
则在 \(K\) 上，\(V\) 双有理同构于 \(C\times\mathbb A^1_W\) 中由方程
\[
W^3=\Theta
\]
所切出的仿射曲线。设 \(L/K\) 为任意特征不等于 \(2,3\) 的域扩张，并假设 \(\mu_3\subset L\)。则对任意 \(P\in U(L)\)，成立
\[
\boxed{\;
\pi^{-1}(P)(L)\neq\varnothing
\iff
\Theta(P)\in L^{\times 3}.\;}
\]
并且更精确地，
\[
\boxed{\;
\#\pi^{-1}(P)(L)=
\begin{cases}
3,& \Theta(P)\in L^{\times 3},\\
0,& \Theta(P)\notin L^{\times 3}.
\end{cases}}
\]
若进一步 \(L=\FF_q\) 且 \(q\equiv 1\pmod 3\)，并取三次特征
\[
\chi_3:\FF_q^\times\to\{1,\zeta_3,\zeta_3^2\},
\]
则对每个 \(P\in U(\FF_q)\) 都有显式计数
\[
\boxed{\;
\#\pi^{-1}(P)(\FF_q)
=
1+\chi_3(\Theta(P))+\chi_3(\Theta(P))^2.\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-leyang-explicit-3torsor-generator-identification}，在 \(K\) 上有双有理模型
\[
V:\quad W^3=\Theta.
\]
由于 \(P\in U(L)\)，\(\Theta(P)\) 既非零也非极点，因此纤维 \(\pi^{-1}(P)\) 的 \(L\)-点恰对应于方程
\[
W^3=\Theta(P)
\]
在 \(L\) 中的解。

若 \(\Theta(P)\in L^{\times 3}\)，取 \(W_0\in L^\times\) 使 \(W_0^3=\Theta(P)\)。因 \(\mu_3\subset L\) 且 \(\operatorname{char}(L)\neq 3\)，三根
\[
W_0,\qquad \zeta_3W_0,\qquad \zeta_3^2W_0
\]
两两不同，故纤维恰有三个 \(L\)-点。反之，若方程在 \(L\) 中有解，则 \(\Theta(P)\) 必是立方。于是前两条结论成立。

在有限域情形 \(L=\FF_q\) 且 \(q\equiv 1\pmod 3\) 时，\(\FF_q^\times/(\FF_q^\times)^3\) 恰由 \(\chi_3\) 检测，并满足标准恒等式
\[
1+\chi_3(a)+\chi_3(a)^2=
\begin{cases}
3,& a\in (\FF_q^\times)^3,\\
0,& a\notin (\FF_q^\times)^3.
\end{cases}
\]
将 \(a=\Theta(P)\) 代入即得最后一式。证毕。
\end{proof}

\begin{corollary}[分裂素数上的 cubic 角色和与椭圆迹的边界修正公式]\label{cor:xi-leyang-split-prime-cubic-character-sum-trace}
设 \(p\) 为使 \(C_6,C,E\) 同时良约化的素数，并满足
\[
p\equiv 1\pmod 3.
\]
则 \(-3\) 在 \(\FF_p\) 中为平方，故 \(Q_\pm\in C(\FF_p)\)。仍记
\[
U:=C\setminus\{Q_+,Q_-\},
\qquad
V:=\pi^{-1}(U),
\]
并取三次特征 \(\chi_3:\FF_p^\times\to\{1,\zeta_3,\zeta_3^2\}\)。则
\[
\boxed{\;
\#V(\FF_p)
=
\sum_{P\in U(\FF_p)}
\Bigl(1+\chi_3(\Theta(P))+\chi_3(\Theta(P))^2\Bigr).\;}
\]
从而定义 character sum
\[
\Sigma_p(\Theta):=
\sum_{P\in U(\FF_p)}
\Bigl(\chi_3(\Theta(P))+\chi_3(\Theta(P))^2\Bigr),
\]
则有
\[
\boxed{\;
\Sigma_p(\Theta)=\#V(\FF_p)-\#U(\FF_p).\;}
\]
再令
\[
a_p(E):=p+1-\#E(\FF_p),
\qquad
\varepsilon_p:=2-\#\pi^{-1}(Q_+)(\FF_p)-\#\pi^{-1}(Q_-)(\FF_p).
\]
则成立精确公式
\[
\boxed{\;
\Sigma_p(\Theta)=-2a_p(E)+\varepsilon_p.\;}
\]
特别地，
\[
\boxed{\;
a_p(E)=-\frac12\bigl(\Sigma_p(\Theta)-\varepsilon_p\bigr),\qquad
|\varepsilon_p|\le 4.\;}
\]
\end{corollary}
\begin{proof}
第一式由定理 \ref{thm:xi-leyang-kummer-fiber-cubic-residue-criterion} 对 \(P\in U(\FF_p)\) 逐点求和得到；第二式只是移项。

又由于 \(Q_\pm\in C(\FF_p)\)，有
\[
\#U(\FF_p)=\#C(\FF_p)-2,
\]
并且
\[
\#V(\FF_p)=\#C_6(\FF_p)-\#\pi^{-1}(Q_+)(\FF_p)-\#\pi^{-1}(Q_-)(\FF_p).
\]
因此
\[
\Sigma_p(\Theta)
=\#V(\FF_p)-\#U(\FF_p)
=\#C_6(\FF_p)-\#C(\FF_p)+\varepsilon_p.
\]
再由定理 \ref{thm:xi-leyang-local-factor-square-decomposition} 在 \(r=1\) 时的点数恒等式，
\[
\#C_6(\FF_p)-\#C(\FF_p)=2\#E(\FF_p)-2(p+1)=-2a_p(E).
\]
代回即得
\[
\Sigma_p(\Theta)=-2a_p(E)+\varepsilon_p.
\]
最后，由 \(\pi\) 为次数 \(3\) 的有限覆盖，\(\#\pi^{-1}(Q_\pm)(\FF_p)\le 3\)，故
\[
|\varepsilon_p|\le 4.
\]
证毕。
\end{proof}
