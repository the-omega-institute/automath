\paragraph{Poisson--Cauchy 粗粒化与圆盘紧致化框架下的独立投稿化结论链}
以下结论固定沿用本文既定的 Poisson--Cauchy 粗粒化与圆盘紧致化设定，并以命题 \(11.667\)、命题 \(11.670\)、定义 \(11.7.4\)、定理 \(11.885\)、推论 \(B.1\)、命题 \(B.54\) 的记号体系为共同基底。

\begin{conclusion}[Poisson--Cauchy 粗粒化相对熵的六阶锐展开]\label{con:xi-poisson-cauchy-kl-sixth-order-independent-submission}
设 \(\tilde\nu\in\mathcal P(\RR)\)，均值为 \(\bar\gamma\)，中心矩记为
\[
\mu_k:=\int_{\RR}(\gamma-\bar\gamma)^k\,\dd\tilde\nu(\gamma),\qquad
\sigma^2:=\mu_2>0.
\]
令
\[
\mathsf P_t(x):=\frac{1}{\pi}\frac{t}{x^2+t^2},\qquad
\tilde h_t:=\mathsf P_t*\tilde\nu,\qquad
\tilde g_t(x):=\mathsf P_t(x-\bar\gamma).
\]
若 \(\int_{\RR}|\gamma-\bar\gamma|^6\,\dd\tilde\nu(\gamma)<\infty\)，则当 \(t\to\infty\) 时
\[
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)
=\frac{\sigma^4}{8t^4}
+\frac{1}{t^6}\left(
\frac{\sigma^6}{64}
-\frac{\sigma^2\mu_4}{8}
+\frac{3\mu_3^2}{32}
\right)
+o(t^{-6}).
\]
\end{conclusion}
\begin{proof}
记 \(u:=\gamma-\bar\gamma\)、\(z:=x-\bar\gamma\)。由卷积表示
\[
\tilde h_t(x)=\int_{\RR}\mathsf P_t(z-u)\,\dd\tilde\nu(\gamma).
\]
对 \(u\) 在 \(0\) 处作四阶 Taylor 展开并对 \(\tilde\nu\) 取期望，得
\[
\tilde h_t
=\mathsf P_t(z)
+\frac{\mu_2}{2}\mathsf P_t''(z)
-\frac{\mu_3}{6}\mathsf P_t^{(3)}(z)
+\frac{\mu_4}{24}\mathsf P_t^{(4)}(z)
+R_t,
\]
其中 \(\|R_t\|_\infty=O(t^{-6})\)、\(\|R_t\|_{L^1}=O(t^{-5})\)。
令
\[
\delta_t(x):=\frac{\tilde h_t(x)-\tilde g_t(x)}{\tilde g_t(x)}.
\]
取缩放 \(y:=z/t\)，并写
\[
\tilde g_t(\bar\gamma+ty)\,t=G(y):=\frac{1}{\pi}\frac{1}{1+y^2}.
\]
可得一致展开
\[
\delta_t(\bar\gamma+ty)
=\frac{\mu_2}{t^2}U_2(y)+\frac{\mu_3}{t^3}U_3(y)+\frac{\mu_4}{t^4}U_4(y)+O(t^{-5}),
\]
其中
\[
U_2(y)=\frac{3y^2-1}{(1+y^2)^2},\quad
U_3(y)=-\frac{4y(1-y^2)}{(1+y^2)^3},\quad
U_4(y)=\frac{5y^4-10y^2+1}{(1+y^2)^4}.
\]
并且 \(\sup_y|\delta_t(\bar\gamma+ty)|=O(t^{-2})\)。
于是
\[
(1+\delta)\log(1+\delta)=\delta+\frac{\delta^2}{2}-\frac{\delta^3}{6}+O(\delta^4),
\]
且
\[
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)
=\int_{\RR}\tilde g_t(1+\delta_t)\log(1+\delta_t)\,\dd x,\qquad
\int_{\RR}\tilde g_t\delta_t\,\dd x=0.
\]
代入并换元 \(x=\bar\gamma+ty\)，得
\[
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)
=\frac12\int_{\RR}G\,\delta_t^2\,\dd y
-\frac16\int_{\RR}G\,\delta_t^3\,\dd y
+O(t^{-8}).
\]
由奇偶性（\(G,U_2,U_4\) 为偶函数，\(U_3\) 为奇函数）消去 \(t^{-5}\) 项，进而
\[
D_{\mathrm{KL}}
=\frac{\mu_2^2}{2t^4}\int_{\RR}GU_2^2\,\dd y
+\frac{1}{t^6}\left(
\mu_2\mu_4\int_{\RR}GU_2U_4\,\dd y
+\frac{\mu_3^2}{2}\int_{\RR}GU_3^2\,\dd y
-\frac{\mu_2^3}{6}\int_{\RR}GU_2^3\,\dd y
\right)
+o(t^{-6}).
\]
标准有理积分给出
\[
\int_{\RR}GU_2^2\,\dd y=\frac14,\qquad
\int_{\RR}GU_3^2\,\dd y=\frac{3}{16},
\]
\[
\int_{\RR}GU_2U_4\,\dd y=-\frac18,\qquad
\int_{\RR}GU_2^3\,\dd y=-\frac{3}{32}.
\]
代回并以 \(\mu_2=\sigma^2\) 化简即得结论。证毕。
\end{proof}

\begin{conclusion}[一般 \(C^3\) \texorpdfstring{$f$}{f}-散度的六阶普适展开]\label{con:xi-poisson-cauchy-fdiv-sixth-order-independent}
在结论 \ref{con:xi-poisson-cauchy-kl-sixth-order-independent-submission} 的设定下，令 \(f:(0,\infty)\to\RR\) 为凸函数，满足 \(f(1)=0\)，并在 \(1\) 邻域内 \(C^3\)；记 \(f''(1),f^{(3)}(1)\in\RR\)。
定义
\[
D_f(\tilde h_t\mid \tilde g_t)
:=\int_{\RR}\tilde g_t(x)\,
f\!\left(\frac{\tilde h_t(x)}{\tilde g_t(x)}\right)\,\dd x.
\]
则当 \(t\to\infty\) 时
\[
D_f(\tilde h_t\mid \tilde g_t)
=\frac{f''(1)\sigma^4}{8t^4}
+\frac{1}{t^6}\left(
f''(1)\left[-\frac{\sigma^2\mu_4}{8}+\frac{3\mu_3^2}{32}\right]
-\frac{f^{(3)}(1)\sigma^6}{64}
\right)
+o(t^{-6}).
\]
特别地，取 \(f(u)=u\log u\) 可回收结论
\ref{con:xi-poisson-cauchy-kl-sixth-order-independent-submission}。
\end{conclusion}
\begin{proof}
沿用上一结论中的 \(\delta_t:=\tilde h_t/\tilde g_t-1\)，其满足 \(\delta_t=O(t^{-2})\) 一致成立。
由 \(f\in C^3\)，有
\[
f(1+\delta)
=f(1)+f'(1)\delta+\frac{f''(1)}{2}\delta^2+\frac{f^{(3)}(1)}{6}\delta^3+O(\delta^4).
\]
积分后一次项因
\(
\int \tilde g_t\delta_t\,\dd x=0
\)
消失。其余项代入结论
\ref{con:xi-poisson-cauchy-kl-sixth-order-independent-submission}
证明中已得的 \(\delta_t\) 展开与同一组积分常数，直接化简即得所示系数。证毕。
\end{proof}

\begin{conclusion}[二阶正规形的函数型极限定律与全局一致收敛]\label{con:xi-poisson-cauchy-second-order-profile-uniform-limit}
设 \(\int_{\RR}|\gamma-\bar\gamma|^3\,\dd\tilde\nu(\gamma)<\infty\) 且 \(\sigma^2>0\)。定义重标度轮廓
\[
\Delta_t(y):=\frac{t^3}{\sigma^2}\Bigl(\tilde h_t(\bar\gamma+ty)-\tilde g_t(\bar\gamma+ty)\Bigr).
\]
令
\[
R(y):=\frac1\pi\frac{3y^2-1}{(1+y^2)^3}.
\]
则
\[
\lim_{t\to\infty}\sup_{y\in\RR}|\Delta_t(y)-R(y)|=0,
\]
并且对任意 \(1\le p<\infty\) 有
\[
\lim_{t\to\infty}\|\Delta_t-R\|_{L^p(\RR,\dd y)}=0.
\]
\end{conclusion}
\begin{proof}
记 \(u:=\gamma-\bar\gamma\)、\(z:=ty\)。对 \(\mathsf P_t(z-u)\) 作二阶 Taylor 展开并取 \(\tilde\nu\)-期望：
\[
\tilde h_t(\bar\gamma+ty)
=\mathsf P_t(ty)+\frac{\sigma^2}{2}\mathsf P_t''(ty)+\mathcal R_t(y),
\]
且三阶余项满足
\[
|\mathcal R_t(y)|
\le \frac{1}{6}\!\left(\int_{\RR}|u|^3\,\dd\tilde\nu\right)\!
\sup_{\xi\in\RR}|\mathsf P_t^{(3)}(\xi)|
=O(t^{-4}).
\]
故
\[
\Delta_t(y)
=\frac{t^3}{2}\mathsf P_t''(ty)+O(t^{-1}).
\]
由尺度恒等式
\(
\mathsf P_t''(ty)=t^{-3}\mathsf P_1''(y)
\)
与
\[
\mathsf P_1''(y)=\frac{2}{\pi}\frac{3y^2-1}{(1+y^2)^3},
\]
得
\[
\Delta_t(y)=\frac12\mathsf P_1''(y)+O(t^{-1})=R(y)+O(t^{-1}),
\]
且误差在 \(y\) 上一致，故得到全局一致收敛。

进一步，\(|R(y)|\lesssim (1+y^2)^{-2}\)，并可取同阶可积控制 \(\sup_{t\ge t_0}|\Delta_t(y)|\lesssim (1+y^2)^{-2}\)；
于是由支配收敛定理可得任意 \(1\le p<\infty\) 的 \(L^p\) 收敛。证毕。
\end{proof}

\begin{conclusion}[\texorpdfstring{$L^2$}{L2} 距离的锐渐近常数]\label{con:xi-poisson-cauchy-l2-sharp-independent}
在结论 \ref{con:xi-poisson-cauchy-second-order-profile-uniform-limit} 的假设下，
\[
\lim_{t\to\infty}\frac{t^{5/2}}{\sigma^2}\,
\|\tilde h_t-\tilde g_t\|_{L^2(\RR,\dd x)}
=\frac{\sqrt3}{4\sqrt\pi}.
\]
\end{conclusion}
\begin{proof}
由二阶正规形可写
\[
\tilde h_t-\tilde g_t=\frac{\sigma^2}{2}\mathsf P_t''+\eta_t,
\]
且 \(\|\eta_t\|_\infty=O(t^{-4})\)、\(\|\eta_t\|_{L^1}=O(t^{-3})\)，从而
\[
\|\eta_t\|_{L^2}^2\le \|\eta_t\|_\infty\|\eta_t\|_{L^1}=O(t^{-7})=o(t^{-5}).
\]
故
\[
\|\tilde h_t-\tilde g_t\|_2^2
=\frac{\sigma^4}{4}\|\mathsf P_t''\|_2^2+o(t^{-5}).
\]
又
\[
\mathsf P_t''(x)=t^{-3}\mathsf P_1''(x/t)
\quad\Longrightarrow\quad
\|\mathsf P_t''\|_2^2=t^{-5}\|\mathsf P_1''\|_2^2.
\]
直接计算
\[
\|\mathsf P_1''\|_2^2
=\int_{\RR}\left(\frac{2}{\pi}\frac{3y^2-1}{(1+y^2)^3}\right)^2\dd y
=\frac{3}{4\pi}.
\]
因此
\[
\|\tilde h_t-\tilde g_t\|_2^2
=\frac{3\sigma^4}{16\pi}\,t^{-5}+o(t^{-5}),
\]
开方即得所述常数。证毕。
\end{proof}

\begin{conclusion}[\texorpdfstring{$L^\infty$}{Linf} 距离的锐渐近常数]\label{con:xi-poisson-cauchy-linf-sharp-independent}
在结论 \ref{con:xi-poisson-cauchy-second-order-profile-uniform-limit} 的假设下，
\[
\lim_{t\to\infty}\frac{t^{3}}{\sigma^2}\,
\|\tilde h_t-\tilde g_t\|_{L^\infty(\RR,\dd x)}
=\frac{1}{\pi}.
\]
\end{conclusion}
\begin{proof}
仍由
\(
\tilde h_t-\tilde g_t=\frac{\sigma^2}{2}\mathsf P_t''+\eta_t
\)
与 \(\|\eta_t\|_\infty=o(t^{-3})\) 得
\[
\|\tilde h_t-\tilde g_t\|_\infty
=\frac{\sigma^2}{2}\|\mathsf P_t''\|_\infty+o(t^{-3}).
\]
再由尺度关系
\[
\|\mathsf P_t''\|_\infty=t^{-3}\|\mathsf P_1''\|_\infty,\qquad
\mathsf P_1''(y)=\frac{2}{\pi}\frac{3y^2-1}{(1+y^2)^3},
\]
可知 \(|\mathsf P_1''|\) 在 \(y=0\) 取最大值 \(2/\pi\)，故
\[
\|\mathsf P_t''\|_\infty=\frac{2}{\pi t^3}.
\]
代回得到
\[
\|\tilde h_t-\tilde g_t\|_\infty
=\frac{\sigma^2}{\pi t^3}+o(t^{-3}),
\]
即结论成立。证毕。
\end{proof}

\begin{conclusion}[共动 Jensen 缺陷的 Cauchy--时间纤维单变量表示]\label{con:xi-comoving-jensen-cauchy-timefiber-onevariable}
取 \(x_0\in\RR\)，定义平移嵌入
\[
s_{x_0}(w):=\frac12+\mathrm{i}\bigl(x_0+\mathcal C^{-1}(w)\bigr),\qquad
F_{\zeta,x_0}(w):=\Xi_\zeta\!\bigl(s_{x_0}(w)\bigr),
\]
并按定义 \(11.7.4\) 记共动 Jensen 缺陷
\[
D_{\zeta,x_0}(\varrho)
:=\frac{1}{2\pi}\int_0^{2\pi}\log\bigl|F_{\zeta,x_0}(\varrho e^{\mathrm{i}\theta})\bigr|\,\dd\theta
-\log|F_{\zeta,x_0}(0)|.
\]
则对任意 \(\varrho\in(0,1)\) 有一维表示
\[
D_{\zeta,x_0}(\varrho)
=\int_{\RR}
\log\left|
\Xi_\zeta\!\left(
\frac12-y_\varrho(x)+\mathrm{i}\bigl(x_0+x_\varrho(x)\bigr)
\right)\right|
\frac{\dd x}{\pi(1+x^2)}
-\log|\Xi_\zeta(-\tfrac12+\mathrm{i}x_0)|,
\]
其中
\[
x_\varrho(x)=\frac{4\varrho x}{(1+\varrho)^2+(1-\varrho)^2x^2},
\qquad
y_\varrho(x)=\frac{(1-\varrho^2)(1+x^2)}{(1+\varrho)^2+(1-\varrho)^2x^2}.
\]
\end{conclusion}
\begin{proof}
由 \(\mathcal C^{-1}(0)=\mathrm{i}\) 得
\(
s_{x_0}(0)=-\tfrac12+\mathrm{i}x_0
\)。
记
\(
t(\varrho,\theta):=\mathcal C^{-1}(\varrho e^{\mathrm{i}\theta})
\)，则
\[
D_{\zeta,x_0}(\varrho)
=\frac{1}{2\pi}\int_0^{2\pi}
\log\left|\Xi_\zeta\!\left(\frac12+\mathrm{i}\bigl(x_0+t(\varrho,\theta)\bigr)\right)\right|
\dd\theta
-\log|\Xi_\zeta(-\tfrac12+\mathrm{i}x_0)|.
\]
取半角替换 \(x=\tan(\theta/2)\)，由推论 \(B.1\) 得
\[
\frac{\dd\theta}{2\pi}=\frac{\dd x}{\pi(1+x^2)}.
\]
并由定理 \(11.885\) 的显式参数化（等价于
\eqref{eq:app-t-rho-theta-explicit}）可得
\[
t(\varrho,\theta(x))=x_\varrho(x)+\mathrm{i}\,y_\varrho(x),
\]
其中 \(x_\varrho,y_\varrho\) 如上。
故
\[
\frac12+\mathrm{i}\bigl(x_0+t\bigr)
=\Bigl(\frac12-y_\varrho(x)\Bigr)+\mathrm{i}\bigl(x_0+x_\varrho(x)\bigr),
\]
代回即得结论。证毕。
\end{proof}

\begin{conclusion}[端点单零点 Jensen 核的 Poisson--尺度分解与时间切片表示]\label{con:xi-endpoint-singlezero-jensen-poisson-scale-slicing}
令 \(\mathcal C:\mathbb H\to\mathbb D\) 为 Cayley 变换。对任意 \(\gamma\in\RR\)、\(\delta>0\)，由命题 \(B.54\) 的模长恒等式定义
\[
\kappa_\delta(\gamma)
:=\log\frac{1}{|\mathcal C(\gamma+\mathrm{i}\delta)|}
=\frac12\log\frac{\gamma^2+(1+\delta)^2}{\gamma^2+(1-\delta)^2}.
\]
则有精确 Poisson--尺度积分分解
\[
\kappa_\delta(\gamma)
=\pi\int_{|1-\delta|}^{1+\delta}\mathsf P_t(\gamma)\,\dd t,
\qquad
\mathsf P_t(\gamma)=\frac1\pi\frac{t}{\gamma^2+t^2}.
\]
进一步，给定有限原子族 \(\{(\gamma_j,\delta_j,\omega_j)\}_{j=1}^N\)（\(\omega_j\ge0\)），定义
\[
\nu_t:=\sum_{j=1}^N
\omega_j\,
\mathbf 1_{\{|1-\delta_j|\le t\le 1+\delta_j\}}
\delta_{\gamma_j},
\]
以及端点 Jensen 势
\[
\mathcal D(x_0):=\sum_{j=1}^N\omega_j\,\kappa_{\delta_j}(x_0-\gamma_j).
\]
则成立时间切片表示
\[
\mathcal D(x_0)=\pi\int_0^\infty(\mathsf P_t*\nu_t)(x_0)\,\dd t.
\]
\end{conclusion}
\begin{proof}
由命题 \(B.54\) 得
\[
|\mathcal C(\gamma+\mathrm{i}\delta)|^2
=\frac{\gamma^2+(\delta-1)^2}{\gamma^2+(\delta+1)^2},
\]
取对数即得 \(\kappa_\delta\) 的闭式表达。
另一方面，对任意 \(0\le a<b\) 有
\[
\int_a^b\mathsf P_t(\gamma)\,\dd t
=\frac{1}{\pi}\int_a^b\frac{t}{\gamma^2+t^2}\,\dd t
=\frac{1}{2\pi}\log\frac{\gamma^2+b^2}{\gamma^2+a^2}.
\]
取 \(a=|1-\delta|\)、\(b=1+\delta\) 即得
\(
\kappa_\delta(\gamma)=\pi\int_a^b\mathsf P_t(\gamma)\,\dd t
\)。

对有限原子族求和并交换有限求和与积分（Fubini 直接适用）：
\[
\mathcal D(x_0)
=\sum_{j=1}^N\omega_j\pi\int_{|1-\delta_j|}^{1+\delta_j}
\mathsf P_t(x_0-\gamma_j)\,\dd t
\]
\[
=\pi\int_0^\infty
\sum_{j=1}^N\omega_j\mathbf 1_{\{|1-\delta_j|\le t\le 1+\delta_j\}}
\mathsf P_t(x_0-\gamma_j)\,\dd t
=\pi\int_0^\infty(\mathsf P_t*\nu_t)(x_0)\,\dd t.
\]
证毕。
\end{proof}
