\begin{theorem}[例外块的整系数逆矩阵闭式]\label{thm:xi-replica-softcore-exceptional-block-inverse-integral-closed}
\leanverified{paper\_xi\_replica\_softcore\_exceptional\_block\_inverse\_integral\_closed}
矩阵 \(A^{(q)}\) 可逆，且
\[
\bigl(A^{(q)}\bigr)^{-1}=2\bigl(C^{(q)}\bigr)^{-1}-E_{00},
\qquad
E_{00}:=e_0e_0^{\mathsf T}.
\]
等价地，对任意 \(0\le k,\ell\le q\)，
\[
\bigl(A^{(q)}\bigr)^{-1}_{k,\ell}
=2(-1)^{k+\ell-q}\binom{k}{q-\ell}\mathbf 1_{\{k+\ell\ge q\}}
-\mathbf 1_{\{k=\ell=0\}}.
\]
特别地，\(\bigl(A^{(q)}\bigr)^{-1}\in\ZZ^{(q+1)\times(q+1)}\)。
\end{theorem}
\begin{proof}
设 \(R\) 为反序置换矩阵（\(R_{k,\ell}=\mathbf 1_{\{k+\ell=q\}}\)），\(P\) 为 Pascal 下三角矩阵
\[
P_{i,j}:=\binom{i}{j}\mathbf 1_{\{i\ge j\}}.
\]
则
\[
(RP)_{k,\ell}=P_{q-k,\ell}=\binom{q-k}{\ell}=C^{(q)}_{k,\ell},
\]
故 \(C^{(q)}=RP\) 可逆，且
\[
\bigl(C^{(q)}\bigr)^{-1}=P^{-1}R,\qquad
(P^{-1})_{i,j}=(-1)^{i-j}\binom{i}{j}\mathbf 1_{\{i\ge j\}}.
\]
从而
\[
\bigl(C^{(q)}\bigr)^{-1}_{k,\ell}
=(P^{-1})_{k,q-\ell}
=(-1)^{k+\ell-q}\binom{k}{q-\ell}\mathbf 1_{\{k+\ell\ge q\}}.
\]
另一方面，\(C^{(q)}e_0=\mathbf 1\) 与 \((C^{(q)})_{0,\ell}=\binom q\ell\) 给出
\[
\bigl(C^{(q)}\bigr)^{-1}\mathbf 1=e_0,\qquad
(b^{(q)})^{\mathsf T}\bigl(C^{(q)}\bigr)^{-1}=e_0^{\mathsf T},
\]
故
\[
1+(b^{(q)})^{\mathsf T}\bigl(C^{(q)}\bigr)^{-1}\mathbf 1=2.
\]
对秩一扰动 \(C^{(q)}+\mathbf 1(b^{(q)})^{\mathsf T}\) 应用 Sherman--Morrison 公式：
\[
\bigl(C^{(q)}+\mathbf 1(b^{(q)})^{\mathsf T}\bigr)^{-1}
=\bigl(C^{(q)}\bigr)^{-1}
-\frac{\bigl(C^{(q)}\bigr)^{-1}\mathbf 1(b^{(q)})^{\mathsf T}\bigl(C^{(q)}\bigr)^{-1}}
{1+(b^{(q)})^{\mathsf T}\bigl(C^{(q)}\bigr)^{-1}\mathbf 1}
=\bigl(C^{(q)}\bigr)^{-1}-\frac12E_{00}.
\]
再由
\[
A^{(q)}=\frac12\bigl(C^{(q)}+\mathbf 1(b^{(q)})^{\mathsf T}\bigr)
\]
即得
\[
\bigl(A^{(q)}\bigr)^{-1}
=2\bigl(C^{(q)}+\mathbf 1(b^{(q)})^{\mathsf T}\bigr)^{-1}
=2\bigl(C^{(q)}\bigr)^{-1}-E_{00}.
\]
代入 \((C^{(q)})^{-1}\) 的逐项公式即得结论。证毕。
\end{proof}

\begin{corollary}[例外谱倒数一次幂和与线性 Vieta 系数]\label{cor:xi-replica-softcore-exceptional-reciprocal-first-power-vieta-linear}
\leanverified{paper\_xi\_replica\_softcore\_exceptional\_reciprocal\_first\_power\_vieta\_linear}
记 \(\chi_{\mathrm{exc}}^{(q)}(x):=\det(xI-A^{(q)})=\prod_{i=0}^{q}(x-\lambda_i^{(q)})\)。则
\[
\sum_{i=0}^{q}\frac1{\lambda_i^{(q)}}
=\Tr\!\bigl((A^{(q)})^{-1}\bigr)
=-1+2(-1)^qF_{q+1}.
\]
并且其一次项系数满足
\[
[x]\chi_{\mathrm{exc}}^{(q)}(x)
=(-1)^{q+\frac{q(q+1)}2}2^{-q}\Bigl(-1+2(-1)^qF_{q+1}\Bigr).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-replica-softcore-exceptional-block-inverse-integral-closed}，
\[
\Tr\!\bigl((A^{(q)})^{-1}\bigr)=2\Tr\!\bigl((C^{(q)})^{-1}\bigr)-1.
\]
而
\[
\bigl(C^{(q)}\bigr)^{-1}_{k,k}
=(-1)^q\binom{k}{q-k}\mathbf 1_{\{2k\ge q\}},
\]
故
\[
\Tr\!\bigl((C^{(q)})^{-1}\bigr)
=(-1)^q\sum_{k=\lceil q/2\rceil}^{q}\binom{k}{q-k}
=(-1)^q\sum_{r=0}^{\lfloor q/2\rfloor}\binom{q-r}{r}
=(-1)^qF_{q+1}.
\]
从而
\[
\sum_{i=0}^{q}\lambda_i^{-1}
=\Tr\!\bigl((A^{(q)})^{-1}\bigr)
=-1+2(-1)^qF_{q+1}.
\]
一次项系数由 Vieta 公式给出：
\[
[x]\chi_{\mathrm{exc}}^{(q)}(x)
=(-1)^q\!\left(\prod_{i=0}^{q}\lambda_i^{(q)}\right)
\!\left(\sum_{i=0}^{q}\frac1{\lambda_i^{(q)}}\right).
\]
再代入
\(
\prod_{i=0}^{q}\lambda_i^{(q)}
=\det(A^{(q)})
=(-1)^{\frac{q(q+1)}2}2^{-q}
\)
（定理 \ref{thm:pom-replica-softcore-exceptional-spectrum-product}）即得所示闭式。证毕。
\end{proof}

\begin{proposition}[例外谱倒数二次幂和的 Fibonacci 闭式]\label{prop:xi-replica-softcore-exceptional-reciprocal-second-power-fibonacci}
\leanverified{paper\_xi\_replica\_softcore\_exceptional\_reciprocal\_second\_power\_fibonacci}
对任意 \(q\ge 1\)，有
\[
\sum_{i=0}^{q}\frac1{(\lambda_i^{(q)})^2}
=\Tr\!\bigl((A^{(q)})^{-2}\bigr)
=1+4F_{2q+2}.
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:xi-replica-softcore-exceptional-block-inverse-integral-closed}，
\[
(A^{(q)})^{-2}
=\bigl(2(C^{(q)})^{-1}-E_{00}\bigr)^2
=4(C^{(q)})^{-2}
-4\bigl((C^{(q)})^{-1}E_{00}+E_{00}(C^{(q)})^{-1}\bigr)
+E_{00}.
\]
由 \((C^{(q)})^{-1}e_0=e_q\) 与 \(e_0^{\mathsf T}(C^{(q)})^{-1}=e_q^{\mathsf T}\) 得
\[
(C^{(q)})^{-1}E_{00}=e_qe_0^{\mathsf T},\qquad
E_{00}(C^{(q)})^{-1}=e_0e_q^{\mathsf T},
\]
故（\(q\ge1\)）两者迹均为 \(0\)。于是
\[
\Tr\!\bigl((A^{(q)})^{-2}\bigr)=4\Tr\!\bigl((C^{(q)})^{-2}\bigr)+1.
\]
又 \(C^{(q)}=B_q^{\mathsf T}\) 与 \(B_q=\mathrm{Sym}^q(M)\)（\(M=\begin{psmallmatrix}1&1\\1&0\end{psmallmatrix}\)）给出
\[
\mathrm{spec}(C^{(q)})=\{(-1)^j\varphi^{q-2j}\}_{j=0}^{q},
\]
故
\[
\Tr\!\bigl((C^{(q)})^{-2}\bigr)
=\sum_{j=0}^{q}\varphi^{-2(q-2j)}
=\sum_{j=0}^{q}\varphi^{-2q+4j}
=\frac{\varphi^{2q+4}-\varphi^{-2q}}{\varphi^4-1}.
\]
利用 \(\varphi^4-1=\varphi^2\sqrt5\)，化简为
\[
\Tr\!\bigl((C^{(q)})^{-2}\bigr)
=\frac{\varphi^{2q+2}-\varphi^{-2q-2}}{\sqrt5}
=F_{2q+2}.
\]
代回即得结论。证毕。
\end{proof}

\begin{theorem}[秩一耦合参数下的惯性刚性]\label{thm:xi-replica-softcore-exceptional-rankone-coupling-inertia-rigidity}
定义参数族
\[
A^{(q)}(t):=\frac12\Bigl(C^{(q)}+t\,\mathbf 1(b^{(q)})^{\mathsf T}\Bigr),\qquad t\ge 0.
\]
则有：
\begin{enumerate}
  \item
  \[
  \det A^{(q)}(t)=2^{-(q+1)}\det(C^{(q)})(1+t).
  \]
  \item 对 \(t\in[0,1]\)，\(\det A^{(q)}(t)\neq0\)，故谱在该区间内无零点穿越，惯性保持常值。
  \item 特别地，
  \[
  \#\{\lambda_i^{(q)}>0\}=\left\lceil\frac{q+1}{2}\right\rceil,\qquad
  \#\{\lambda_i^{(q)}<0\}=\left\lfloor\frac{q+1}{2}\right\rfloor.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由矩阵行列式引理，
\[
\det\!\bigl(C^{(q)}+t\,\mathbf 1(b^{(q)})^{\mathsf T}\bigr)
=\det(C^{(q)})\Bigl(1+t\,(b^{(q)})^{\mathsf T}(C^{(q)})^{-1}\mathbf 1\Bigr).
\]
由定理 \ref{thm:xi-replica-softcore-exceptional-block-inverse-integral-closed} 的证明，
\(
(b^{(q)})^{\mathsf T}(C^{(q)})^{-1}\mathbf 1=1
\)，故
\[
\det\!\bigl(C^{(q)}+t\,\mathbf 1(b^{(q)})^{\mathsf T}\bigr)=\det(C^{(q)})(1+t).
\]
再乘上整体缩放因子 \(2^{-(q+1)}\) 即得第 1 点。
因此对 \(t\ge0\) 有 \(\det A^{(q)}(t)\neq0\)，第 2 点随即成立。
在 \(t=0\) 时
\[
A^{(q)}(0)=\frac12C^{(q)},\qquad
\mathrm{spec}(A^{(q)}(0))=\left\{\frac{(-1)^j\varphi^{q-2j}}{2}\right\}_{j=0}^{q},

其符号仅由 \(j\) 奇偶决定；偶指标对应正值，奇指标对应负值。
计数得正值 \(\lceil(q+1)/2\rceil\) 个、负值 \(\lfloor(q+1)/2\rfloor\) 个。
由惯性常值传递到 \(t=1\)（即 \(A^{(q)}\)），得到第 3 点。证毕。
\end{proof}

\(
\frac12(-1)^i\varphi^{q-2i}\ (0\le i\le q)
\)，故
\[
\mathrm{neg}(A_{0,q})=\left\lfloor\frac{q+1}{2}\right\rfloor.
\]
由秩一正半定扰动的惯性界，
\[
\mathrm{neg}(A^{(q)})\in
\left\{
\left\lfloor\frac{q+1}{2}\right\rfloor,\ 
\left\lfloor\frac{q+1}{2}\right\rfloor-1
\right\}.
\]
另一方面，定理 \ref{thm:pom-replica-softcore-exceptional-spectrum-product} 给出
\[
\det(A^{(q)})=(-1)^{\frac{q(q+1)}2}2^{-q}\neq 0.
\]
故
\[
(-1)^{\mathrm{neg}(A^{(q)})}
=\mathrm{sgn}\det(A^{(q)})
=(-1)^{\frac{q(q+1)}2}
=(-1)^{\left\lfloor\frac{q+1}{2}\right\rfloor},
\]
其中最后一步按 \(q\bmod 4\) 逐类即得。
若 \(\mathrm{neg}(A^{(q)})=\lfloor\frac{q+1}{2}\rfloor-1\)，奇偶性将与上式矛盾。
故必有
\(
\mathrm{neg}(A^{(q)})=\lfloor\frac{q+1}{2}\rfloor
\)。
证毕。
\end{proof}

\begin{corollary}[固定 \texorpdfstring{$m$}{m} 时 \texorpdfstring{$q\mapsto S_m(q)$}{q->S\_m(q)} 的统一线性递推]\label{cor:xi-replica-softcore-fixed-m-q-recurrence-characteristic-roots}
设 \(m\ge 3\)，定义
\[
\mathcal A_m:=\{\,I(\Gamma_E):E\subsetneq E(C_m)\,\}\subset\ZZ_{>0},
\]
并对每个 \(a\in\mathcal A_m\) 记
\[
\mu_m(a):=\#\{\,E\subsetneq E(C_m):I(\Gamma_E)=a\,\}.
\]
则对任意 \(q\ge 0\)，
\[
2^mS_m(q)-\Omega_m(q)=\sum_{a\in\mathcal A_m}\mu_m(a)\,a^q.
\]
从而 \(q\mapsto S_m(q)\) 满足常系数线性递推，其可取特征多项式为
\[
\boxed{\;
\Bigl(\prod_{a\in\mathcal A_m}(t-a)\Bigr)\cdot\bigl(t^2-L_m t+(-1)^m\bigr).\;}
\]
相应地，\(S_m\) 的 \(q\)-方向结构由两类特征根共同决定：
\(\mathcal A_m\)（真子图独立集底数）与
\(\varphi^m,(-1)^m\varphi^{-m}\)（\(\Omega_m\)-通道）。
\end{corollary}
\begin{proof}
第一式由定理
\ref{thm:xi-replica-softcore-exceptional-moment-cycle-independent-set-master}
按同值 \(I(\Gamma_E)=a\) 分组即得。
右端是有限指数和 \(\sum_a\mu_m(a)a^q\)。
记移位算子 \(\mathscr E\) 作用为 \((\mathscr E f)(q)=f(q+1)\)，
则该有限指数和被
\(
\prod_{a\in\mathcal A_m}(\mathscr E-a)
\)
湮灭；而 \(\Omega_m\) 满足二阶递推
\[
\Omega_m(q+2)-L_m\Omega_m(q+1)+(-1)^m\Omega_m(q)=0.
\]
故两部分相加后被所示乘积多项式湮灭，即得结论。
该结论与命题 \ref{prop:xi-terminal-replica-softcore-fixed-m-q-recurrence-PAQ}
同构。证毕。
\end{proof}
