\paragraph{Lee--Yang 奇异环上的双支权重平均}
定理 \ref{thm:dephys-timefiber-two-ended} 给出时间纤维轨道 \(t(\varrho,\theta)\) 上的角平均表达。
在无平移情形下，该角变量可经 Lee--Yang 逆平方门压缩为负实参数，并由 Haar 推前密度给出一个显式 Jacobian，从而把圆周平均重写为奇异环上的双支权重积分。

\begin{theorem}[时间纤维角平均的 Lee--Yang 双支表示]\label{thm:xi-timefiber-leyang-two-branch-jensen-average}
取 \(0<\varrho<1\)，并令
\[
t(\varrho,\theta)
=\frac{2\varrho\sin\theta}{1+\varrho^2+2\varrho\cos\theta}
\mathrm{i}\,\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta}
\in\mathbb H,
\qquad \theta\in[0,2\pi),
\]
如定理 \ref{thm:dephys-timefiber-two-ended}。
定义 Lee--Yang 变量
\[
\Lambda(\theta):=-\frac{1}{4\cos^2\theta}\in\left(-\infty,-\frac14\right]
\]
并令 \(\mu_{+}:=\Lambda_{\ast}\bigl(\frac{\dd\theta}{2\pi}\bigr)\)。
则 \(\mu_{+}\) 在 \(\lambda<-\tfrac14\) 上具有密度
\[
\frac{\dd\mu_{+}}{\dd\lambda}(\lambda)=\frac{1}{\pi(-\lambda)\sqrt{-4\lambda-1}}
\qquad\Bigl(\lambda<-\frac14\Bigr)
\]
（见附录推论 \ref{cor:leyang-interference-intensity-density}）。
对 \(\lambda\le -\tfrac14\) 定义
\[
c(\lambda):=\frac{1}{2\sqrt{-\lambda}},\qquad
s(\lambda):=\sqrt{1-c(\lambda)^2}
=\frac{\sqrt{-4\lambda-1}}{2\sqrt{-\lambda}},
\]
并定义两支上半平面点
\[
t_{\pm}(\varrho,\lambda)
:=\frac{2\varrho\,s(\lambda)}{1+\varrho^2\pm 2\varrho\,c(\lambda)}
\mathrm{i}\,\frac{1-\varrho^2}{1+\varrho^2\pm 2\varrho\,c(\lambda)}\in\mathbb H.
\]
则 Jensen 缺陷满足严格恒等式
\begin{equation}\label{eq:xi-timefiber-leyang-two-branch-jensen-average}
\boxed{\;
D_{\zeta}(\varrho)
=
\int_{-\infty}^{-1/4}
\frac12\Bigl[
\log\bigl|\Xi_{\zeta}\!\bigl(\tfrac12+\mathrm{i}\,t_{+}(\varrho,\lambda)\bigr)\bigr|
\;+\;
\log\bigl|\Xi_{\zeta}\!\bigl(\tfrac12+\mathrm{i}\,t_{-}(\varrho,\lambda)\bigr)\bigr|
\Bigr]\,\mu_{+}(\dd\lambda)
-\log\bigl|\Xi_{\zeta}(-\tfrac12)\bigr|\; }.
\end{equation}
\leanverified{paper\_xi\_timefiber\_leyang\_two\_branch\_jensen\_average}
\end{theorem}
\begin{proof}
由定理 \ref{thm:dephys-timefiber-two-ended}，
\[
D_{\zeta}(\varrho)
=\frac{1}{2\pi}\int_{0}^{2\pi}\log\Bigl|\Xi_{\zeta}\!\Bigl(\tfrac12+\mathrm{i}\,t(\varrho,\theta)\Bigr)\Bigr|\,\dd\theta
-\log\bigl|\Xi_{\zeta}(-\tfrac12)\bigr|.
\]
对固定 \(\theta\) 写 \(t(\varrho,\theta)=x(\theta)+\mathrm{i}y(\theta)\)。
注意到 \(\Xi_{\zeta}\) 由实系数完成化数据构造，故满足
\(
\overline{\Xi_{\zeta}(\overline{s})}=\Xi_{\zeta}(s)
\)，从而 \(|\Xi_{\zeta}(s)|=|\Xi_{\zeta}(\overline{s})|\)。
当 \(\theta\mapsto-\theta\) 时，\(\cos\theta\) 不变而 \(\sin\theta\) 变号，因而 \(y(\theta)\) 不变而 \(x(\theta)\) 变号；
于是
\[
\tfrac12+\mathrm{i}\,t(\varrho,-\theta)
=\overline{\tfrac12+\mathrm{i}\,t(\varrho,\theta)},
\qquad
\log\Bigl|\Xi_{\zeta}\!\Bigl(\tfrac12+\mathrm{i}\,t(\varrho,-\theta)\Bigr)\Bigr|
=\log\Bigl|\Xi_{\zeta}\!\Bigl(\tfrac12+\mathrm{i}\,t(\varrho,\theta)\Bigr)\Bigr|.
\]
因此该对数模长仅依赖于 \(\cos\theta\) 与 \(|\sin\theta|\)。
\par
对给定 \(\lambda\le -\tfrac14\)，令 \(c=c(\lambda)\)、\(s=s(\lambda)\)。
方程 \(\Lambda(\theta)=\lambda\) 等价于 \(|\cos\theta|=c\)，从而在 \([0,2\pi)\) 上的预像由四点 \((\cos\theta,\sin\theta)=(\pm c,\pm s)\) 给出；
在上述对称性下，\(\sin\theta\) 的符号不改变对数模长，只剩下 \(\cos\theta=\pm c\) 的两支。
将 \((\cos\theta,\sin\theta)=(\pm c,s)\) 代入 \(t(\varrho,\theta)\) 的显式式子，即得到两支点 \(t_{\pm}(\varrho,\lambda)\)。
\par
最后，对角平均应用 \(\mu_{+}\) 的推前定义（等价地使用 \(\mu_{+}\) 的密度形式）即可把 \(\theta\)-积分改写为 \(\lambda\)-积分，并在每条 \(\lambda\)-纤维上取两支平均，得到 \eqref{eq:xi-timefiber-leyang-two-branch-jensen-average}。证毕。
\end{proof}

\begin{proposition}[端点支的 Lee--Yang 标度律]\label{prop:xi-timefiber-leyang-endpoint-branch-scaling}
沿用定理 \ref{thm:xi-timefiber-leyang-two-branch-jensen-average} 的记号，并写
\[
\lambda=-\frac14-\xi,\qquad \xi\ge 0.
\]
则端点支 \(t_{-}(\varrho,\lambda)\) 的虚部满足精确恒等式
\begin{equation}\label{eq:xi-timefiber-leyang-endpoint-branch-im-exact}
\Im\,t_{-}(\varrho,\lambda)
=\frac{1-\varrho^2}{(1-\varrho)^2+2\varrho\bigl(1-c(\lambda)\bigr)},
\qquad
c(\lambda)=\frac{1}{\sqrt{1+4\xi}}.
\end{equation}
并且当 \(\varrho\uparrow 1\) 且 \(\xi=o(1)\) 时成立可控渐近
\begin{equation}\label{eq:xi-timefiber-leyang-endpoint-branch-im-asymptotic}
\boxed{\;
\Im\,t_{-}\!\left(\varrho,-\tfrac14-\xi\right)
=\frac{2(1-\varrho)}{(1-\varrho)^2+4\xi}\,\bigl(1+o(1)\bigr)\; }.
\end{equation}
\leanverified{paper\_xi\_timefiber\_leyang\_endpoint\_branch\_scaling}
\end{proposition}
\begin{proof}
由 \(t_{-}(\varrho,\lambda)\) 的定义可得
\[
\Im\,t_{-}(\varrho,\lambda)
=\frac{1-\varrho^2}{1+\varrho^2-2\varrho c(\lambda)}
=\frac{1-\varrho^2}{(1-\varrho)^2+2\varrho(1-c(\lambda))},
\]
得到 \eqref{eq:xi-timefiber-leyang-endpoint-branch-im-exact} 的第一式。
当 \(\lambda=-\tfrac14-\xi\) 时有 \(-\lambda=\tfrac14+\xi\)，故
\(
c(\lambda)=\frac{1}{2\sqrt{-\lambda}}=\frac{1}{\sqrt{1+4\xi}}
\)。
\par
在 \(\xi\to 0\) 下展开 \(c(\lambda)=(1+4\xi)^{-1/2}=1-2\xi+O(\xi^2)\)，从而
\(
1-c(\lambda)=2\xi+O(\xi^2)
\)。
并且当 \(\varrho\uparrow 1\) 时 \(1-\varrho^2=2(1-\varrho)+O((1-\varrho)^2)\)。
代入 \eqref{eq:xi-timefiber-leyang-endpoint-branch-im-exact} 并在假设 \(\xi=o(1)\) 下统一整理误差项，即得 \eqref{eq:xi-timefiber-leyang-endpoint-branch-im-asymptotic}。证毕。
\end{proof}
