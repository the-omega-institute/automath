\begin{proposition}[\texorpdfstring{$\infty$}{infty}-纤维的二分点与切线整系数表达]\label{prop:xi-ed-infty-fiber-halving-tangent-integral}
\leanverified{paper\_xi\_ed\_infty\_fiber\_halving\_tangent\_integral}
在段落 \ref{thm:xi-ed-rational-branchpoints-group-law-rigidity} 的记号下，令
\[
R_1:=(-4,-1)\in E_{\mathrm D}(\QQ).
\]
则存在有理点
\[
Q_1:=(-8,-31)\in E_{\mathrm D}(\QQ)
\]
满足 \(2Q_1=R_1\)。
并且 \(E_{\mathrm D}\) 在 \(Q_1\) 处的切线为
\[
u=8m+33.
\]
\end{proposition}
\begin{proof}
记
\(
f(m):=-4m^3-16m^2+16m+65
\)。
直接代入得 \(f(-8)=961=31^2\)，故 \(Q_1=(-8,-31)\in E_{\mathrm D}(\QQ)\)。
由 \(u^2=f(m)\) 的隐式微分
\[
\frac{\dd u}{\dd m}=\frac{f'(m)}{2u},
\qquad
f'(m)=-12m^2-32m+16,
\]
代入 \((m,u)=(-8,-31)\) 得 \(\dd u/\dd m=8\)，故切线为 \(u+31=8(m+8)\)，即 \(u=8m+33\)。

将 \(u=8m+33\) 代回 \(u^2=f(m)\) 并化简得到
\[
f(m)-(8m+33)^2=-4(m+4)(m+8)^2,
\]
故该直线与 \(E_{\mathrm D}\) 的交点（计交数）为 \(Q_1\)（二重）与 \((-4,1)\)（一重）。
按椭圆曲线的割线--切线法则，
\[
2Q_1+(-4,1)=O,
\]
从而 \(2Q_1=-(-4,1)=(-4,-1)=R_1\)。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$X_H\to E_{\mathrm D}$}{XH to ED} 的显式双覆盖与分支仅在 \texorpdfstring{$\{P_-,P_+\}$}{P-,P+}]\label{thm:xi-ed-xh-explicit-double-cover-branch-only-at-infty-fiber}
沿用段落 \ref{thm:xi-ed-y-divisor-and-group-law-identity} 的点
\[
P_-:=(-2,1),\qquad P_+:=(2,1),
\]
并取命题 \ref{prop:xi-ed-infty-fiber-halving-tangent-integral} 的 \(Q_1=(-8,-31)\)。
定义
\[
t:=\frac{u-1}{u-8m-33}\in \QQ(E_{\mathrm D}).
\]
则在 \(E_{\mathrm D}\) 上有主除子恒等式
\[
\operatorname{div}(t)=P_-+P_+-2Q_1.
\]
令
\[
\QQ(X_H):=\QQ(E_{\mathrm D})\bigl(\sqrt{t}\bigr),
\]
并以 \(X_H\) 表示该函数域的光滑射影模型。
则 \(X_H\to E_{\mathrm D}\) 为二次覆盖，且恰在 \(P_-\) 与 \(P_+\) 两点处分歧；并且 \(g(X_H)=2\)。
\end{theorem}
\begin{proof}
记两条线性函数
\[
\ell_1:=u-1,\qquad \ell_2:=u-8m-33.
\]
由 \(u=1\) 与 \(u^2=-4m^3-16m^2+16m+65\) 得
\[
-4m^3-16m^2+16m+65-1=-4(m-2)(m+2)(m+4),
\]
故
\[
\operatorname{div}(\ell_1)=P_-+P_++(-4,1)-3O.
\]
另一方面，\(\ell_2=0\) 为 \(Q_1\) 处切线；由命题 \ref{prop:xi-ed-infty-fiber-halving-tangent-integral}，其与 \(E_{\mathrm D}\) 的交点（计交数）为 \(2Q_1+(-4,1)\)，故
\[
\operatorname{div}(\ell_2)=2Q_1+(-4,1)-3O.
\]
因此
\[
\operatorname{div}(t)=\operatorname{div}(\ell_1)-\operatorname{div}(\ell_2)=P_-+P_+-2Q_1.
\]

由上述主除子可见，\(t\) 在 \(P_-\) 与 \(P_+\) 处具有奇阶零点，而在其余点的赋值皆为偶数（特别地在 \(Q_1\) 处为二阶极点）。
因此二次扩张 \(\QQ(E_{\mathrm D})(\sqrt{t})/\QQ(E_{\mathrm D})\) 的分支点恰为 \(P_-,P_+\)。
由 Hurwitz 公式
\[
2g(X_H)-2=2\bigl(2g(E_{\mathrm D})-2\bigr)+\deg\operatorname{Branch}(X_H/E_{\mathrm D})=0+2
\]
得到 \(g(X_H)=2\)。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$t$}{t}-坐标的二次模型与 \texorpdfstring{$E_{\mathrm D}$}{ED} 的生成元表示]\label{thm:xi-ed-t-quadratic-model-primitive-generator}
\leanverified{paper\_xi\_ed\_t\_quadratic\_model\_primitive\_generator}
沿用定理 \ref{thm:xi-ed-xh-explicit-double-cover-branch-only-at-infty-fiber} 的椭圆曲线
\[
E_{\mathrm D}:\ u^2=f(m),\qquad f(m)=-4m^3-16m^2+16m+65,
\]
与有理函数
\(
t=\frac{u-1}{u-8m-33}\in\QQ(E_{\mathrm D})
\)。
则成立：
\begin{enumerate}
  \item 在 \(\QQ(E_{\mathrm D})\) 中有恒等式
  \begin{equation}\label{eq:xi-ed-t-quadratic-in-m}
  (t-1)^2m^2+16t^2m+64t^2+4t-4=0.
  \end{equation}
  \item 令
  \begin{equation}\label{eq:xi-ed-t-primitive-generator-W}
  W:=\frac{(t-1)^2m+8t^2}{2}\in\QQ(E_{\mathrm D}),
  \end{equation}
  则有显式 Weierstrass 关系
  \begin{equation}\label{eq:xi-ed-edt-weierstrass}
  W^2=31t^3-13t^2-3t+1.
  \end{equation}
  \item 因而
  \[
  \QQ(E_{\mathrm D})=\QQ(t,W),
  \]
  并且由 \(T:=t\) 诱导的次数 \(2\) 态射
  \(
  E_{\mathrm D}\to\PP^1_T
  \)
  的分支数据由三次式 \(31T^3-13T^2-3T+1\) 的根与无穷远点给出。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(t(u-8m-33)=u-1\) 得
\[
u=\frac{8mt+33t-1}{t-1}.
\]
将其代回 \(u^2=f(m)\)，并清分母 \((t-1)^2\) 后化简，得到关于 \(m\) 的多项式关系
\[
4(m+4)\Bigl((t-1)^2m^2+16t^2m+64t^2+4t-4\Bigr)=0.
\]
对一般点 \(m\neq -4\)，即得 \eqref{eq:xi-ed-t-quadratic-in-m}。
由 \eqref{eq:xi-ed-t-quadratic-in-m} 的判别式计算得到
\[
\bigl(2(t-1)^2m+16t^2\bigr)^2
=16\bigl(31t^3-13t^2-3t+1\bigr),
\]
两边同除以 \(16\) 并用 \eqref{eq:xi-ed-t-primitive-generator-W} 的定义，即得 \eqref{eq:xi-ed-edt-weierstrass}。
最后，由 \eqref{eq:xi-ed-t-quadratic-in-m} 与 \eqref{eq:xi-ed-t-primitive-generator-W} 可解出
\[
m=\frac{2W-8t^2}{(t-1)^2}\in\QQ(t,W),
\qquad
u=\frac{8mt+33t-1}{t-1}\in\QQ(t,W),
\]
从而 \(\QQ(E_{\mathrm D})\subseteq \QQ(t,W)\)；反向包含显然，故 \(\QQ(E_{\mathrm D})=\QQ(t,W)\)。
分支描述由椭圆曲线的二次表示 \(W^2=\text{三次式}(T)\) 的标准分支结构给出。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$X_H$}{XH} 的双椭圆化与 Lee--Yang 椭圆曲线的显式出现]\label{thm:xi-ed-xh-bielliptic-leyang-edt}
在定理 \ref{thm:xi-ed-xh-hyperelliptic-model-explicit} 的模型 \(\mathcal C\) 上考虑对合
\[
\iota_1:(S,W)\longmapsto(-S,W),
\qquad
\iota_2:(S,W)\longmapsto(-S,-W).
\]
则：
\begin{enumerate}
  \item 商曲线 \(\mathcal C/\langle\iota_1\rangle\) 为椭圆曲线
  \[
  E_{\mathrm D}^{(t)}:\quad W^2=31T^3-13T^2-3T+1,\qquad T=S^2,
  \]
  且自然映射 \(\pi_{\mathrm D}:\mathcal C\to E_{\mathrm D}^{(t)}\) 为次数 \(2\) 的态射。
  \item 商曲线 \(\mathcal C/\langle\iota_2\rangle\) 在 \(\QQ\) 上同构于 Lee--Yang 椭圆曲线
  \[
  E:\quad Y^2=X^3-X+\frac14,
  \]
  并且可取显式次数 \(2\) 态射 \(\pi_E:\mathcal C\to E\) 为
  \begin{equation}\label{eq:xi-ed-xh-to-leyang-map}
  X=\frac{1-S^2}{4S^2},
  \qquad
  Y=\frac{W}{8S^3}.
  \end{equation}
  特别地，在 \(\QQ(X_H)\) 内成立刚性代数关系
  \begin{equation}\label{eq:xi-ed-xh-TX-dictionary}
  T=S^2=\frac{1}{4X+1}.
  \end{equation}
\end{enumerate}
\end{theorem}
\begin{proof}
对 \(\iota_1\)，不变量域由 \(T=S^2\) 与 \(W\) 生成，故商曲线满足 \(W^2=31T^3-13T^2-3T+1\)，并且商映射显然为次数 \(2\)。
\par
对 \(\iota_2\)，取不变量 \(X=\frac{1-S^2}{4S^2}\) 与 \(Y=\frac{W}{8S^3}\)。
由 \(\mathcal C\) 的方程直接计算：
\[
Y^2=\frac{W^2}{64S^6}
=\frac{31S^6-13S^4-3S^2+1}{64S^6}
=X^3-X+\frac14,
\]
故得到到 \(E\) 的态射 \(\pi_E\)。
最后由 \(X=\frac{1-T}{4T}\) 立即解得 \(T=\frac{1}{4X+1}\)，即 \eqref{eq:xi-ed-xh-TX-dictionary}。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\Jac(X_H)$}{Jac(XH)} 的 \texorpdfstring{$(2,2)$}{(2,2)}-同源分解与图形核]\label{thm:xi-ed-xh-jacobian-22-isogeny-graph-kernel}

\[
\Phi:\ E\times E_{\mathrm D}^{(t)}\longrightarrow \Jac(\mathcal C)\cong \Jac(X_H),
\qquad
(P,Q)\longmapsto \pi_E^*(P-\mathcal O_E)+\pi_{\mathrm D}^*(Q-\mathcal O_{\mathrm D})
\]
为一条 \((2,2)\)-同源（特别地 \(\deg\Phi=4\)）。
其核为阶 \(4\) 的图形子群
\[
\ker(\Phi)=\{(R,\psi(R)):\ R\in E[2]\},
\]
其中 \(\psi:E[2]\xrightarrow{\sim}E_{\mathrm D}^{(t)}[2]\) 为命题 \ref{prop:xi-ed-xh-2torsion-dictionary-mod2} 所确定的 \(2\)-挠同余。
\end{theorem}
\begin{proof}
由 \(\iota_1,\iota_2\) 与超椭圆对合 \((S,W)\mapsto(S,-W)\) 的可交换性可知，其生成一个 Klein 四元群作用于 \(\mathcal C\)。
由标准的 Kani--Rosen 分解（亦可由范数与拉回伴随性直接检验像的维数与满性）可知由两条非平凡二次商映射 \(\pi_E,\pi_{\mathrm D}\) 诱导的 \(\Phi\) 为同源，且同源型为 \((2,2)\)，从而 \(\deg\Phi=4\)。
\par
另一方面，\(\psi\) 把 \(E\) 与 \(E_{\mathrm D}^{(t)}\) 的三条有限 \(2\)-挠分支点按 \eqref{eq:xi-ed-xh-branch-dictionary} 的分式线性对应识别。
对任意 \(R\in E[2]\)，对应的分支除子拉回在 \(\Jac(\mathcal C)\) 中给出相同的 \(2\)-挠类，故
\(\pi_E^*(R-\mathcal O_E)+\pi_{\mathrm D}^*(\psi(R)-\mathcal O_{\mathrm D})=0\)，从而图形子群包含于核。
该子群阶为 \(\#E[2]=4\)，与 \(\deg\Phi\) 一致，故必为 \(\ker(\Phi)\)。证毕。
\end{proof}

\begin{proposition}[两条椭圆商不互同源与 \texorpdfstring{$\End^0(\Jac(X_H))$}{End0(Jac(XH))} 的刚性]\label{prop:xi-ed-xh-nonisogenous-and-endomorphism-rigidity}

\End^0_{\QQ}\!\bigl(\Jac(X_H)\bigr)\cong \QQ\times\QQ,
\]
并且定理 \ref{thm:xi-ed-xh-jacobian-22-isogeny-graph-kernel} 的 \((2,2)\)-分解在 \(\QQ\)-同源意义下唯一。
\end{proposition}
\begin{proof}
取良素数 \(p=7\)。
在 \(\FF_7\) 上直接计数得
\[
\#E(\FF_7)=9,\qquad \#E_{\mathrm D}^{(t)}(\FF_7)=13,
\]
从而
\(
a_7(E)=7+1-9=-1
\)
而
\(
a_7(E_{\mathrm D}^{(t)})=7+1-13=-5
\)，两者不等。
由 Tate 同源判别可知两曲线不在 \(\QQ\) 上同源。
再由 \(\Jac(X_H)\sim_{\QQ}E\times E_{\mathrm D}^{(t)}\) 与 \(\Hom_{\QQ}(E,E_{\mathrm D}^{(t)})=0\) 得
\(
\End^0_{\QQ}(\Jac(X_H))\cong \End^0_{\QQ}(E)\times \End^0_{\QQ}(E_{\mathrm D}^{(t)})\cong \QQ\times\QQ
\)；
唯一性亦随之成立。证毕。
\end{proof}

\begin{corollary}[端oscopic 情形下的 \texorpdfstring{$L$}{L}-函数因子分解与 \texorpdfstring{$\ell$}{l}-进表示直和]\label{cor:xi-ed-xh-lfunction-tatemodule-endoscopic-splitting}
在上述设定下，\(\Jac(X_H)\) 的 Hasse--Weil \(L\)-函数满足
\[
L\!\left(\Jac(X_H),s\right)=L(E,s)\,L\!\left(E_{\mathrm D}^{(t)},s\right),
\]
等价地其 \(\ell\)-进 Tate 模表示分裂为
\[
V_\ell\!\left(\Jac(X_H)\right)\ \simeq\ V_\ell(E)\ \oplus\ V_\ell\!\left(E_{\mathrm D}^{(t)}\right)
\]
作为 \(G_{\QQ}\)-表示。
从自守表示的端oscopic 视角，该分裂对应于一个典型的 Yoshida 型提升情形：与 \(\Jac(X_H)\) 关联的四维表示为两条二维 \(\GL_2\) 型表示的直和。
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-ed-xh-jacobian-22-isogeny-graph-kernel}，\(\Jac(X_H)\) 与 \(E\times E_{\mathrm D}^{(t)}\) 在 \(\QQ\) 上同源。
同源阿贝尔簇在良素处具有相同局部 Euler 因子，因而 \(L\)-函数分解成立；对应的 Tate 模直和分解与端oscopic 解释属于标准对应。证毕。
\end{proof}

\begin{theorem}[三条二次子覆盖的分支除子之显式识别]\label{thm:xi-ed-three-quadratic-subcovers-explicit-branch-divisors}
沿用段落 \ref{thm:xi-ed-y-branchpoints-leyang-via-hm} 的三次态射 \(y:E_{\mathrm D}\to\PP^1_y\) 与点的记号：
\[
U_0:=Q_0=\left(\frac74,-\frac{19}{4}\right),\quad
U_1:=(2,-1),\quad
U_{g,1},U_{g,2},U_{g,3}\in E_{\mathrm D}(\overline\QQ),
\quad
P_-:=(-2,1),\quad P_+:=(2,1),
\]
其中 \(U_{g,i}\) 为每条 Lee--Yang 分支值纤维中与分歧点互补的未分歧点。
记
\[
\Delta(y):=-y(y-1)g(y),\qquad g(y)=256y^3+411y^2+165y+32,
\qquad
t:=\frac{u-1}{u-8m-33}.
\]
定义三条二次扩张
\[
\QQ(X_V):=\QQ(E_{\mathrm D})\bigl(\sqrt{\Delta(y)}\bigr),\qquad
\QQ(X_H):=\QQ(E_{\mathrm D})\bigl(\sqrt{t}\bigr),\qquad
\QQ(Y):=\QQ(E_{\mathrm D})\bigl(\sqrt{\Delta(y)\,t}\bigr).
\]
则三条二次覆盖的分支除子满足
\[
\operatorname{Branch}(X_V/E_{\mathrm D})
=U_0+U_1+U_{g,1}+U_{g,2}+U_{g,3}+P_+,
\]
\[
\operatorname{Branch}(Y/E_{\mathrm D})
=U_0+U_1+U_{g,1}+U_{g,2}+U_{g,3}+P_-,
\]
\[
\operatorname{Branch}(X_H/E_{\mathrm D})=P_-+P_+.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-s3-fiberproduct-branch-parity-xa-xv-ed}，
\(
\QQ(X_V)=\QQ(E_{\mathrm D})(\sqrt{\Delta(y)})
\)
且分支点由 \(\operatorname{ord}_Q(\Delta(y))\) 的奇偶性刻画；
其约化分支支撑恰为六个分歧值 \(y\in\{0,1,\infty\}\cup\{g(y)=0\}\) 的未分歧原像点。
按段落 \ref{thm:xi-ed-y-branchpoints-leyang-via-hm}，这些点正是
\(
U_0,U_1,U_{g,1},U_{g,2},U_{g,3},P_+
\)，
从而得到第一式。

由定理 \ref{thm:xi-ed-xh-explicit-double-cover-branch-only-at-infty-fiber}，
\(\operatorname{Branch}(X_H/E_{\mathrm D})=P_-+P_+\)。

最后，二次扩张中“乘积取对称差”的分支规律表明：
\(\sqrt{\Delta(y)t}\) 所定义的二次覆盖之分支集合为
\(\operatorname{Branch}(X_V/E_{\mathrm D})\ \Delta\ \operatorname{Branch}(X_H/E_{\mathrm D})\)，
故正为
\(
U_0+U_1+U_{g,1}+U_{g,2}+U_{g,3}+P_-,
\)
得到第二式。证毕。
\end{proof}

\begin{proposition}[双二次子域格与三条二次子扩张]\label{prop:xi-ed-v4-biquadratic-subfield-lattice}
令
\[
\QQ(Z):=\QQ(E_{\mathrm D})\bigl(\sqrt{\Delta(y)},\sqrt{t}\bigr).
\]
则 \(\QQ(Z)/\QQ(E_{\mathrm D})\) 为双二次（\(V_4\)）扩张，其三条二次子扩张分别为
\[
\QQ(E_{\mathrm D})\bigl(\sqrt{\Delta(y)}\bigr)=\QQ(X_V),\qquad
\QQ(E_{\mathrm D})\bigl(\sqrt{t}\bigr)=\QQ(X_H),\qquad
\QQ(E_{\mathrm D})\bigl(\sqrt{\Delta(y)t}\bigr)=\QQ(Y),
\]
并且相应分支除子由定理 \ref{thm:xi-ed-three-quadratic-subcovers-explicit-branch-divisors} 显式给出。
\end{proposition}
\begin{proof}
由于 \(\Delta(y)\notin (\QQ(E_{\mathrm D})^\times)^2\) 且 \(t\notin(\QQ(E_{\mathrm D})^\times)^2\)，并且二者之分支集合不相同（前者在六个未分歧伴随点处分歧，后者仅在 \(P_\pm\) 处分歧），
故二者在 \(\QQ(E_{\mathrm D})^\times/(\QQ(E_{\mathrm D})^\times)^2\) 中线性无关，从而生成双二次扩张。
三条二次子扩张由标准子域格给出；分支除子由二次扩张的赋值奇偶判据与“对称差”规律得到。证毕。
\end{proof}

\begin{theorem}[函数 \texorpdfstring{$t$}{t} 的相对范数闭式与结果式完全因式分解]\label{thm:xi-ed-norm-t-closed-form-resultant-factorization}
沿用段落 \ref{thm:xi-ed-plane-quadratic-birational-equivalence} 的二元形式
\[
F(m,y)=(m-2)(m+2)^2y^2+(4m^2-17)y+(4m-7)=0,
\]
并记
\[
U(m,y):=2(m-2)(m+2)^2y+(4m^2-17)
\]
使得在 \(E_{\mathrm D}\) 上有 \(u=U(m,y)\)。
令
\[
t=\frac{U(m,y)-1}{U(m,y)-8m-33}\in\QQ(E_{\mathrm D}).
\]
则在三次扩张 \(\QQ(E_{\mathrm D})/\QQ(y)\) 中成立范数恒等式
\[
\operatorname{Norm}_{\QQ(E_{\mathrm D})/\QQ(y)}(t)=\frac{1}{(8y-3)^2}.
\]
更精确地，
\[
\operatorname{Res}_m\bigl(U-1,\ F\bigr)=8y^3(24y-23),\qquad
\operatorname{Res}_m\bigl(U-8m-33,\ F\bigr)=8y^3(8y-3)^2(24y-23),
\]
从而上述范数恒等式由两者相除得到。
因此 \(t\) 在 \(E_{\mathrm D}\) 上唯一的极点位于 \(y=3/8\) 的纤维中且为二重极点；特别地 \(y(Q_1)=3/8\)。
\end{theorem}
\begin{proof}
因 \([\QQ(E_{\mathrm D}):\QQ(y)]=3\)，对任意 \(P(m)\in\QQ(y)[m]\) 有
\[
\operatorname{Norm}_{\QQ(E_{\mathrm D})/\QQ(y)}(P(m))\doteq \operatorname{Res}_m(P,F),
\]
其中 \(\doteq\) 表示相差 \(\QQ(y)^\times\) 中单位因子。
当分子分母关于 \(m\) 的次数相同，单位因子在比值中抵消。
对 \(P=U-1\) 与 \(Q=U-8m-33\) 直接计算结果式并因式分解即得所示完全分解，从而
\[
\operatorname{Norm}(t)=\operatorname{Res}_m(U-1,F)\Big/\operatorname{Res}_m(U-8m-33,F)=\frac{1}{(8y-3)^2}.
\]
最后由 \(Q_1=(-8,-31)\) 代入 \(y=\frac{u-(4m^2-17)}{2(m-2)(m+2)^2}\) 直接得 \(y(Q_1)=3/8\)，
并与范数所揭示的唯一二重极点纤维一致。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$t$}{t} 作为原始元的三次模型与判别式重构]\label{thm:xi-ed-t-primitive-element-cubic-model-discriminant}
沿用定理 \ref{thm:xi-ed-norm-t-closed-form-resultant-factorization} 的 \(t\)。
则：
\begin{enumerate}
  \item \(t\) 为扩张 \(\QQ(E_{\mathrm D})/\QQ(y)\) 的原始元，即 \(\QQ(E_{\mathrm D})=\QQ(y)(t)\)；
  \item \(t\) 满足三次方程
  \[
  (8y-3)^2t^3-(32y+19)t^2+(16y+11)t-1=0;
  \]
  \item 该三次多项式关于 \(t\) 的判别式满足
  \[
  \Disc_t\Bigl((8y-3)^2t^3-(32y+19)t^2+(16y+11)t-1\Bigr)
  =2^{12}\Delta(y),
  \qquad
  \Delta(y)=-y(y-1)g(y).
  \]
\end{enumerate}
从而 Lee--Yang 三次式 \(g(y)\) 在去除显然因子 \(y(y-1)\) 后，作为判别式的本原不可约因子被内禀地恢复。
\end{theorem}
\begin{proof}
由定义 \(t=\frac{U-1}{U-8m-33}\)，与约束 \(F(m,y)=0\) 联立并对变量 \(m\) 取结果式，可得到一个关于 \((t,y)\) 的一元消元多项式。
去除与退化纤维相关的可约因子后，得到所示唯一三次关系。
该三次式首项系数为 \((8y-3)^2\)，常数项为 \(-1\)，与定理 \ref{thm:xi-ed-norm-t-closed-form-resultant-factorization} 的范数恒等式相容。
其判别式为直接代数计算，并化简为 \(2^{12}\Delta(y)\)。

此外，若 \(t\in\QQ(y)\)，则其极点除子必须为某个 \(y\)-函数的拉回；但定理 \ref{thm:xi-ed-xh-explicit-double-cover-branch-only-at-infty-fiber} 表明 \(t\) 在 \(E_{\mathrm D}\) 上的极点为 \(2Q_1\)，其支撑不可能由 \(y\) 的纤维拉回给出，矛盾。
故 \(t\notin\QQ(y)\)。
结合三次关系，得到 \([\QQ(y)(t):\QQ(y)]=3\)，从而 \(\QQ(E_{\mathrm D})=\QQ(y)(t)\)。证毕。
\end{proof}
