\paragraph{径向自由度的强制性：越界根 \texorpdfstring{$S_0>2$}{S0>2} 只能由离线点实现}

|y|\neq 1

\Re(s)\neq\tfrac12
\quad\Longleftrightarrow\quad

\begin{proposition}[自对偶不排除离线零点；可寻址性才排除]\label{prop:xi-offcritical-constructible-but-null-or-extra-register}

  \item \textbf{（离线零点机制显式可构造）}
  因子 \((S_L(s)-S_0)\) 所诱导的零点必满足 \(|u_L(s)|\neq 1\)，即必落在 unitary slice 之外的径向通道。
  \item \textbf{（unitary slice 语义下的结构性退化）}

第 (2) 条即本段对方程 \(S_L(s)=S_0>2\) 的分析所给出的 \(|u_L(s)|\neq 1\)。

\end{proof}

\paragraph{类型检查崩塌：常数层 strictification 的边界——径向依赖不可被 \texorpdfstring{$\mathrm{CT}_\eta$}{CT} 吸收}
在本文的中心扩张/strictification 口径中，\(\mathrm{CT}_\eta\) 被定义为\emph{纯常数}门：它保持所有扭曲通道 pole \(\rho_\chi(\theta)\) 不变，仅对有限部漂移 \(\log\mathfrak{M}_\chi(\theta)\) 做加性平移（定义 \ref{def:pom-finite-part-counterterm-gate}）。
因此（命题 \ref{prop:pom-counterterm-anom-cancel}）所有可 strictify 的残差，必须在“有限部/常数层”闭合，且 strictification 不改变任何指数级主尺度。
\begin{remark}[常数级异常的可规约性：有限部漂移向量的 \texorpdfstring{$\mathcal{A}_G$}{AG}-torsor 结构]\label{rem:xi-finite-part-drift-torsor}

令
\[
\mathbf{m}(\theta):=\bigl(\log\mathfrak{M}_\chi(\theta)\bigr)_{\chi\in\widehat G\setminus\{\chi_0\}}
\ \in\ \RR^{\widehat G\setminus\{\chi_0\}}\ \cong\ \mathcal{A}_G.
\]
则纯常数 counterterm 门 \(\mathrm{CT}_\eta\)（\(\eta\in\mathcal{A}_G\)）保持所有 pole 数据 \(\rho_\chi(\theta)\) 不变，仅作平移
\[
\mathbf{m}(\theta)\longmapsto \mathbf{m}(\theta)+\eta.
\]
因此在固定 pole 数据的前提下，允许的漂移向量族自然构成一个以 \(\mathcal{A}_G\) 为结构群的仿射纤维（torsor）。
换言之，“异常”并非新的指数级障碍，而是常数层的可规约自由度；其全部信息等价于在该 torsor 上选取一条规范截面（亦即选择一种 strictification 规范）。
\end{remark}

\begin{theorem}[局域可逆表示的维数阻碍：径向自由度不可压缩为单一相位]\label{thm:radial-dimension-obstruction}
\leanverified{paper\_xi\_radial\_dimension\_obstruction}
设 \(M=S^{1}\times\RR\) 为相位--径向参数空间，并赋予幂半群作用
\[
d\cdot(\theta,\varrho):=(d\theta,d\varrho)\qquad(d\ge 1).
\]
若存在连续映射 \(\Phi:M\to S^{1}\) 同时满足：
\begin{enumerate}
  \item \textbf{局域可逆读出：}对每个 \(x\in M\) 存在邻域 \(U\ni x\) 与连续读出 \(\Psi:\Phi(U)\to U\) 使 \(\Psi\circ(\Phi|_{U})=\mathrm{id}_{U}\)；
  \item \textbf{幂相容：}\(\Phi(d\cdot x)=\Phi(x)^{d}\) 对一切 \(d\ge 1\) 与 \(x\in M\) 成立；
\end{enumerate}
则导出矛盾。
因此，在保持连续性与局域可逆读出的审计口径下，\((\theta,\varrho)\) 的相位--径向状态不可能被压缩到单一圆周寄存器；任意闭合接口必须显式保留额外寄存器以承载径向自由度。
\end{theorem}
\begin{proof}
由 (1)，映射 \(\Phi|_{U}:U\to \Phi(U)\) 具有连续右逆 \(\Psi\)，从而 \(\Phi|_{U}\) 为同胚（其逆为 \(\Psi|_{\Phi(U)}\)）。
但 \(M=S^{1}\times\RR\) 为二维流形，故 \(U\) 同胚于平面开圆盘；而 \(S^{1}\) 的任一开子集同胚于开区间。
开圆盘与开区间不可能同胚（例如去掉一点后前者仍连通而后者分裂为两段），矛盾。
因此 (1) 已不可能成立，从而 (1)--(2) 不可同时满足。证毕。
\end{proof}

\paragraph{最小 \texorpdfstring{$(2\times2)$}{(2x2)} 自对偶桥接核的显式轴外零点与完全分类}
前述反例通过“在完成化端追加越界根”展示了径向模式的强制性；
作为互补，下面给出一个不依赖伪因子拼接、而是在\emph{最小可控自对偶核}上直接导出轴外零点的显式原型。
该模型是附录 \S\ref{app:kernel-self-dual} 的自对偶正常形（命题 \ref{prop:self-dual-normal-form-1pmu}）在 \(\dim V_+=\dim V_-=1\) 的最小实例。

\begin{theorem}[最小自对偶桥接核的零点全分类：显式非临界线零点塔]\label{thm:xi-2x2-selfdual-offcritical-template}
\leanverified{paper\_xi\_2x2\_selfdual\_offcritical\_template}
取整数 \(a,b\in\ZZ\)，并令
\[
  B_0=\begin{pmatrix}a&-b\\-b&a\end{pmatrix},\qquad
  \Pi=\begin{pmatrix}1&0\\0&-1\end{pmatrix},\qquad
  B_1=\Pi^{-1}B_0\Pi=\begin{pmatrix}a&b\\b&a\end{pmatrix},
\]
定义
\(
B(u):=B_0+uB_1
\)
与
\(
\Delta(z,u):=\det(I-zB(u))
\)。
对任意 \(L>1\)，令完成化切片
\[
  \Xi_{\Delta,L}(s):=\Delta\!\bigl(L^{-s},\,L^{2s-1}\bigr).
\]
则：
\begin{enumerate}
  \item \textbf{（两通道因子分解）}
  \[
  \Delta(z,u)
  =\Bigl(1-z\bigl((a-b)+(a+b)u\bigr)\Bigr)\,
   \Bigl(1-z\bigl((a+b)+(a-b)u\bigr)\Bigr).
  \]
  \item \textbf{（二次方程化与对数晶格）}
  令 \(r:=L^{s-\tfrac12}\)（从而 \(u=L^{2s-1}=r^{2}\)、\(L^{-s}=L^{-1/2}r^{-1}\)）。
  则零点条件 \(\Xi_{\Delta,L}(s)=0\) 等价于以下两条二次方程之一成立：
  \[
  (a+b)r^2-\sqrt L\,r+(a-b)=0,
  \qquad
  (a-b)r^2-\sqrt L\,r+(a+b)=0.
  \]
  每个二次方程给出两个代数根 \(r\)，从而得到四族零点塔
  \[
  s=\frac12+\frac{\log r}{\log L},
  \qquad
  \log r=\log|r|+\mathrm{i}\bigl(\arg r+2\pi k\bigr),\quad k\in\ZZ,
  \]
  即虚部以周期 \(2\pi/\log L\) 作等差平移。
  \item \textbf{（显式轴外零点判别）}
  若 \(b\neq 0\)，则上述四族中至少存在一族满足 \(|r|\neq 1\)，从而对应零点满足 \(\Re(s)\neq\tfrac12\)。
  更进一步，除去可能的对齐情形 \(L=4a^2\)（此时仅可能出现 \(r=\pm 1\) 的 slice 零点）外，所有代数根均满足 \(|r|\neq 1\)，故全部零点塔都离开临界线。
\end{enumerate}
\end{theorem}

\begin{proof}
直接计算得
\[
B(u)=\begin{pmatrix}a(1+u)&b(u-1)\\ b(u-1)&a(1+u)\end{pmatrix},
\qquad
I-zB(u)=
\begin{pmatrix}1-za(1+u)&-zb(u-1)\\ -zb(u-1)&1-za(1+u)\end{pmatrix}.
\]
故
\[
\Delta(z,u)
=\bigl(1-za(1+u)\bigr)^2-\bigl(zb(u-1)\bigr)^2
=\Bigl(1-z\bigl((a-b)+(a+b)u\bigr)\Bigr)\Bigl(1-z\bigl((a+b)+(a-b)u\bigr)\Bigr),
\]
得到 (1)。
对 (2)，代入 \(u=r^2\)、\(L^{-s}=L^{-1/2}r^{-1}\)，并分别令每个线性因子为零即可得到两条二次方程；
零点塔的“对数晶格”表达来自 \(\log r\) 的多值性。
\par
对 (3)，先注意两条二次方程互为倒数变量变换：若 \(r\) 满足第一式，则 \(r^{-1}\) 满足第二式。
若某个代数根满足 \(|r|=1\)，则 \(r^{-1}=\overline r\)，并由实系数性可知 \(r\) 同时满足两条二次方程。
两式相减得 \(2b(r^2-1)=0\)，在 \(b\neq 0\) 下推出 \(r=\pm 1\)。
将 \(r=\pm 1\) 代回任一二次方程得到 \(2a\mp\sqrt L=0\)，从而 \(L=4a^2\)；
因此若 \(L\neq 4a^2\) 则不可能出现 \(|r|=1\) 的根，所有根均满足 \(|r|\neq 1\)。
即便在对齐情形 \(L=4a^2\) 下出现 \(r=\pm 1\) 的 slice 根，另一同方程的伴随根满足
\(
|r|=\bigl|\frac{a-b}{a+b}\bigr|
\)
或其倒数（由根乘积公式 \(r_1r_2=(a-b)/(a+b)\) 等），在 \(b\neq 0\) 下不为 \(1\)，因而仍给出轴外零点塔。
证毕。
\end{proof}

\begin{corollary}[最小自对偶桥接核的切片纯度充要条件]\label{cor:xi-2x2-selfdual-slice-purity-iff}

\end{corollary}

\begin{proof}
若 \(b\neq 0\)，则由定理 \ref{thm:xi-2x2-selfdual-offcritical-template} 第 (3) 条，必存在代数根满足 \(|r|\neq 1\)，从而对应零点塔满足 \(\Re(s)\neq\tfrac12\)，故不可能切片纯。
\par
现设 \(b=0\)。
此时定理 \ref{thm:xi-2x2-selfdual-offcritical-template} 的两条二次方程同化为同一方程
\[
a r^2-\sqrt L\,r+a=0,
\]
其两根 \(r_1,r_2\) 满足 \(r_1r_2=1\)。
因此“全部根满足 \(|r|=1\)”等价于判别式不正：
\(
L-4a^2\le 0
\)。
当 \(L<4a^2\) 时两根为共轭复根且乘积为 \(1\)，故 \(|r_1|=|r_2|=1\)；当 \(L=4a^2\) 时出现重根 \(r=\sqrt L/(2a)=\pm 1\)，仍满足 \(|r|=1\)。
当 \(L>4a^2\) 时两根为互为倒数的实根且不为 \(\pm 1\)，故必有 \(|r|\neq 1\) 的根，从而产生 \(\Re(s)\neq\tfrac12\) 的零点塔。
综上得到充要条件。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$(\rho,\theta)$}{(rho,theta)} 分裂与 unitary slice 投影遗忘]\label{cor:xi-2x2-selfdual-orthogonal-axis-projection-forget}
在定理 \ref{thm:xi-2x2-selfdual-offcritical-template} 的显式模型中，参数
\(
u=L^{2s-1}
\)
的极分解 \(u=e^{\rho+\mathrm{i}\theta}\) 满足
\[
\rho=(2\Re(s)-1)\log L,\qquad \theta=2\Im(s)\log L\ (\mathrm{mod}\ 2\pi).
\]
因此 \(\Re(s)=\tfrac12\) 等价于 \(\rho=0\)（unitary slice）。
对任意一条由 \(|r|\neq 1\) 生成的零点塔，\(\rho\) 是其不可消去的法向参数；将读出限制为纯相位轴（只保留 \(\theta\)）必然遗忘 \(\rho\)，从而在可核验寻址意义下需要模式轴外延或退化为 \(\mathsf{NULL}\)（与定理 \ref{thm:xi-offset-null-type-safety}、定理 \ref{thm:radial-dimension-obstruction} 的一般结论一致）。
\end{corollary}
