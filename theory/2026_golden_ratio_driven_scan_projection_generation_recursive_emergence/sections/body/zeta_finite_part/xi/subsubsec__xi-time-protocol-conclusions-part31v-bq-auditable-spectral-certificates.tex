\paragraph{反向二项核的可审计谱证书：Perron 投影、primitive 轨道与 resolvent 主极点}
在结论 \ref{con:xi-bq-golden-spectrum-orthogonality-norm} 与结论 \ref{con:xi-replica-softcore-exceptional-symq-evaluation-coupling} 已经给出
\(B_q=\Sym^q(M)\)、二项权自伴随性以及 Lucas 型迹公式之后，还可把这组输入压缩成一条直接服务于谱审计的闭式证书链。其核心在于：主特征模态、全矩阵大幂、primitive 轨道计数与 resolvent 主极点均可由同一组显式常数统一控制。

\begin{theorem}[反向二项核 Perron 投影的显式公式与统一谱隙]\label{thm:xi-bq-auditable-perron-projector-gap}
\leanverified{paper\_xi\_bq\_auditable\_perron\_projector\_gap}
固定整数 \(q\ge 1\)，记
\[
B_q(i,j):=\binom{q-j}{i}\qquad(0\le i,j\le q),
\]
\[
W_q:=\diag\!\Bigl(\binom{q}{0}^{-1},\binom{q}{1}^{-1},\dots,\binom{q}{q}^{-1}\Bigr),
\qquad
\|u\|_{W_q}^2:=u^{\mathsf T}W_qu.
\]
再记
\[
\mu_k:=(-1)^k\varphi^{\,q-2k}\qquad(0\le k\le q),
\]
以及主特征值 \(\mu_0=\varphi^q\) 对应的特征向量
\[
v^{(0)}:=(v^{(0)}_0,\dots,v^{(0)}_q)^{\mathsf T},
\qquad
v^{(0)}_j:=\binom{q}{j}\varphi^{\,q-j}.
\]
定义 Perron 投影
\[
\Pi_q:=\frac{v^{(0)}v^{(0)\mathsf T}W_q}{\|v^{(0)}\|_{W_q}^2},
\qquad
T_q:=\varphi^{-q}B_q.
\]
则成立：
\[
\boxed{\;
\|v^{(0)}\|_{W_q}^2=(\varphi+2)^q=(\sqrt5)^q\varphi^q\;}
\]
\[
\boxed{\;
(\Pi_q)_{ij}
=
\frac{\binom{q}{i}\varphi^{\,q-i-j}}{(\sqrt5)^q}
\qquad(0\le i,j\le q)\;}
\]
以及对任意 \(n\ge 0\)，
\[
\boxed{\;
\|T_q^n-\Pi_q\|_{W_q\to W_q}=\varphi^{-2n}\;}
\]
从而第二特征值模长与主特征值之比满足
\[
\boxed{\;
\frac{|\mu_1|}{\mu_0}=\varphi^{-2},\;}
\]
该谱隙与 \(q\) 无关。
\end{theorem}
\begin{proof}
由结论 \ref{con:xi-bq-golden-spectrum-orthogonality-norm}，向量 \(v^{(0)}\) 正是多项式
\((\varphi x+y)^q\) 在基 \(\{x^{q-j}y^j\}_{j=0}^{q}\) 下的系数向量，且对应本征值 \(\mu_0=\varphi^q\)。
同一结论同时给出
\[
\|v^{(0)}\|_{W_q}^2=(\varphi+2)^q.
\]
又由恒等式 \(\varphi+2=\sqrt5\,\varphi\)，得到
\[
\|v^{(0)}\|_{W_q}^2=(\sqrt5)^q\varphi^q.
\]

对投影条目式，注意
\[
(W_qv^{(0)})_j=\binom{q}{j}^{-1}\binom{q}{j}\varphi^{\,q-j}=\varphi^{\,q-j},
\]
故
\[
(\Pi_q)_{ij}
=
\frac{v_i^{(0)}(W_qv^{(0)})_j}{\|v^{(0)}\|_{W_q}^2}
=
\frac{\binom{q}{i}\varphi^{\,q-i}\varphi^{\,q-j}}{(\sqrt5)^q\varphi^q}
=
\frac{\binom{q}{i}\varphi^{\,q-i-j}}{(\sqrt5)^q}.
\]

再由结论 \ref{con:xi-bq-golden-spectrum-orthogonality-norm}，\(B_q\) 在 \(\langle\cdot,\cdot\rangle_{W_q}\) 下可正交对角化，且
\[
\mathrm{spec}(T_q)=\{(-1)^k\varphi^{-2k}:0\le k\le q\}.
\]
减去 \(\Pi_q\) 等于删去主特征值 \(1\) 所在的一维谱投影，因此
\[
\mathrm{spec}(T_q^n-\Pi_q)=\{0\}\cup\{(-1)^{kn}\varphi^{-2kn}:1\le k\le q\}.
\]
由于这是 \(W_q\)-正规算子的谱分解，故其 \(W_q\)-算子范数等于谱半径，即
\[
\|T_q^n-\Pi_q\|_{W_q\to W_q}
=
\max_{1\le k\le q}\varphi^{-2kn}
=
\varphi^{-2n}.
\]
最后
\(
|\mu_1|/\mu_0=\varphi^{q-2}/\varphi^q=\varphi^{-2}
\)
立即成立。证毕。
\end{proof}

\begin{corollary}[反向二项核幂的逐项 sharp 渐近与最终严格正性]\label{cor:xi-bq-auditable-entrywise-asymptotic-positivity}
\leanverified{paper\_xi\_bq\_auditable\_entrywise\_asymptotic\_positivity}
在定理 \ref{thm:xi-bq-auditable-perron-projector-gap} 的记号下，对任意固定 \(q\ge 1\) 与任意 \(0\le i,j\le q\)，当 \(n\to\infty\) 时有
\[
\boxed{\;
\bigl(B_q^n\bigr)_{ij}
=
\frac{\binom{q}{i}}{(\sqrt5)^q}\,
\varphi^{\,q(n+1)-i-j}
+O_q\!\bigl(\varphi^{(q-2)n}\bigr)\;}
\]
并且存在 \(N_q\)，使得当 \(n\ge N_q\) 时，
\[
\boxed{\;
\bigl(B_q^n\bigr)_{ij}>0\qquad(0\le i,j\le q).\;}
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-bq-auditable-perron-projector-gap}，
\[
B_q^n=\varphi^{qn}\Pi_q+\varphi^{qn}(T_q^n-\Pi_q).
\]
对固定 \(q\)，有限维空间上一切矩阵范数彼此等价，故存在常数 \(C_q>0\) 使
\[
\max_{0\le i,j\le q}|A_{ij}|\le C_q\|A\|_{W_q\to W_q}
\]
对任意 \((q+1)\times(q+1)\) 矩阵 \(A\) 都成立。于是
\[
\bigl(B_q^n\bigr)_{ij}
=
\varphi^{qn}(\Pi_q)_{ij}
+O_q\!\bigl(\varphi^{qn}\|T_q^n-\Pi_q\|_{W_q\to W_q}\bigr).
\]
代入定理 \ref{thm:xi-bq-auditable-perron-projector-gap} 的两条显式式得到
\[
\bigl(B_q^n\bigr)_{ij}
=
\varphi^{qn}\frac{\binom{q}{i}\varphi^{\,q-i-j}}{(\sqrt5)^q}
+O_q\!\bigl(\varphi^{qn}\varphi^{-2n}\bigr),
\]
即
\[
\bigl(B_q^n\bigr)_{ij}
=
\frac{\binom{q}{i}}{(\sqrt5)^q}\varphi^{\,q(n+1)-i-j}
+O_q\!\bigl(\varphi^{(q-2)n}\bigr).
\]
主项系数严格为正，而余项与主项之比为 \(O_q(\varphi^{-2n})\to 0\)，故每个固定 \(q\) 存在 \(N_q\) 使全部矩阵元在 \(n\ge N_q\) 时同时严格为正。证毕。
\end{proof}

\begin{theorem}[primitive 轨道数的 Lucas--M\"obius 闭式与最终严格正性]\label{thm:xi-bq-auditable-primitive-orbit-lucas-mobius}
\leanverified{paper\_xi\_bq\_auditable\_primitive\_orbit\_lucas\_mobius}
固定 \(q\ge 1\)。定义
\[
p_n(B_q):=\frac1n\sum_{d\mid n}\mu_{\mathrm{Mob}}(d)\,
\Tr\!\bigl(B_q^{\,n/d}\bigr)\qquad(n\ge 1).
\]
则对任意 \(n\ge 1\) 有完全显式公式
\[
\boxed{\;
p_n(B_q)
=
\frac1n\sum_{d\mid n}\mu_{\mathrm{Mob}}(d)
\left(
\sum_{k=0}^{\lfloor (q-1)/2\rfloor}
(-1)^{k n/d}L_{(q-2k)n/d}
+\mathbf 1_{\{2\mid q\}}(-1)^{qn/(2d)}
\right)\;}
\]
其中 \(L_m\) 为 Lucas 数。并且若记
\[
\gamma_q:=\max\!\left\{q-2,\frac q2\right\},
\]
则有渐近式
\[
\boxed{\;
p_n(B_q)=\frac{\varphi^{qn}}{n}
+O_q\!\left(\frac{\varphi^{\gamma_q n}}{n}\right)\;}
\]
特别地，存在 \(N_q\) 使得
\[
\boxed{\;
p_n(B_q)>0\qquad(n\ge N_q).\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-bq-auditable-perron-projector-gap} 的谱公式，
\[
\Tr(B_q^m)=\sum_{r=0}^{q}(-1)^{rm}\varphi^{\,m(q-2r)}.
\]
对 \(0\le r<q/2\) 将 \(r\) 与 \(q-r\) 配对，得到
\[
(-1)^{rm}\varphi^{\,m(q-2r)}
+(-1)^{(q-r)m}\varphi^{-m(q-2r)}
=
(-1)^{rm}L_{(q-2r)m},
\]
因为 \((q-2r)m\equiv qm\pmod 2\)。若 \(q\) 为偶数，还需补上中间项
\(
(-1)^{qm/2}
\)。
故
\[
\Tr(B_q^m)
=
\sum_{k=0}^{\lfloor (q-1)/2\rfloor}(-1)^{km}L_{(q-2k)m}
+\mathbf 1_{\{2\mid q\}}(-1)^{qm/2}.
\]
将 \(m=n/d\) 代入 Möbius 反演定义即得显式公式。

再由 Lucas 闭式
\(
L_t=\varphi^t+(-\varphi^{-1})^t
\)
可知
\[
\Tr(B_q^m)=\varphi^{qm}+O_q\!\bigl(\varphi^{(q-2)m}\bigr)
\qquad(m\ge 1).
\]
于是
\[
\frac1n\Tr(B_q^n)
=
\frac{\varphi^{qn}}{n}
+O_q\!\left(\frac{\varphi^{(q-2)n}}{n}\right).
\]
对其余除数 \(d>1\)，有 \(n/d\le n/2\)，故
\[
\Tr\!\bigl(B_q^{\,n/d}\bigr)\ll_q \varphi^{q n/2}.
\]
因此
\[
\frac1n\sum_{\substack{d\mid n\\ d>1}}
\mu_{\mathrm{Mob}}(d)\Tr\!\bigl(B_q^{\,n/d}\bigr)
=
O_q\!\left(\frac{\varphi^{qn/2}}{n}\right).
\]
合并两部分即得
\[
p_n(B_q)=\frac{\varphi^{qn}}{n}
+O_q\!\left(\frac{\varphi^{\max\{q-2,q/2\}n}}{n}\right).
\]
由于 \(\gamma_q<q\)，主项最终严格支配误差，故 \(p_n(B_q)\) 对充分大的 \(n\) 必为正。证毕。
\end{proof}

\begin{theorem}[resolvent 主极点与解析余项分解]\label{thm:xi-bq-auditable-resolvent-pole-remainder}
\leanverified{paper\_xi\_bq\_auditable\_resolvent\_pole\_remainder}
固定 \(q\ge 1\)，定义
\[
R_q(z):=(I-zB_q)^{-1}.
\]
则在圆盘
\[
|z|<\varphi^{2-q}
\]
内成立严格分解
\[
\boxed{\;
R_q(z)=\frac{\Pi_q}{1-\varphi^q z}+H_q(z)\;}
\]
其中 \(\Pi_q\) 为定理 \ref{thm:xi-bq-auditable-perron-projector-gap} 的 Perron 投影，而 \(H_q(z)\) 在 \(|z|<\varphi^{2-q}\) 内解析。并且对同一圆盘中的全部 \(z\) 都有
\[
\boxed{\;
\|H_q(z)\|_{W_q\to W_q}
\le
\frac{1}{1-\varphi^{q-2}|z|}\;}
\]
从而 \(z=\varphi^{-q}\) 是 \(R_q(z)\) 离原点最近的唯一极点；其主部的 rank-one 系数为 \(\Pi_q\)，而关于变量 \(z\) 的留数为 \(-\varphi^{-q}\Pi_q\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-bq-auditable-perron-projector-gap}，\(B_q\) 在 \(\langle\cdot,\cdot\rangle_{W_q}\) 下可正交对角化，故存在两两 \(W_q\)-正交的谱投影
\(\Pi_q^{(0)},\Pi_q^{(1)},\dots,\Pi_q^{(q)}\) 使
\[
B_q=\sum_{k=0}^{q}\mu_k\Pi_q^{(k)},
\qquad
\Pi_q^{(0)}=\Pi_q.
\]
因此在谱点之外
\[
R_q(z)=\sum_{k=0}^{q}\frac{\Pi_q^{(k)}}{1-\mu_k z}
=
\frac{\Pi_q}{1-\varphi^q z}
+\sum_{k=1}^{q}\frac{\Pi_q^{(k)}}{1-\mu_k z}.
\]
定义
\[
H_q(z):=\sum_{k=1}^{q}\frac{\Pi_q^{(k)}}{1-\mu_k z}.
\]
由于对 \(k\ge 1\) 有 \(|\mu_k|\le \varphi^{q-2}\)，故当 \(|z|<\varphi^{2-q}\) 时一切分母均不为零，从而 \(H_q\) 在该圆盘内解析。

再由 \(W_q\)-正交谱分解，经 \(W_q^{1/2}\) 共轭后可视为通常 Euclidean 意义下的正交投影分解，因此
\[
\|H_q(z)\|_{W_q\to W_q}
=
\max_{1\le k\le q}\left|\frac{1}{1-\mu_k z}\right|
\le
\frac{1}{1-\varphi^{q-2}|z|}.
\]
最后，主特征值对应的极点位于 \(z=\varphi^{-q}\)，其余极点模长均至少为 \(\varphi^{2-q}\)。由于
\(
\varphi^{-q}<\varphi^{2-q}
\)，故 \(z=\varphi^{-q}\) 是离原点最近的唯一极点。并且
\[
\Res_{z=\varphi^{-q}}R_q(z)
=
\lim_{z\to\varphi^{-q}}(z-\varphi^{-q})\frac{\Pi_q}{1-\varphi^q z}
=
-\varphi^{-q}\Pi_q.
\]
若以标准变量 \(s=1-\varphi^q z\) 书写主部，则主极点系数正是 \(\Pi_q\)；等价地，
\[
R_q(z)=\frac{\Pi_q}{1-\varphi^q z}+H_q(z)
\]
已给出所需的可审计主极点分解。证毕。
\end{proof}
