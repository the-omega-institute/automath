\paragraph{Jensen 缺陷的能量--熵分解与变分证书接口}\label{par:xi-jensen-defect-entropy-energy-decomposition}
Jensen 缺陷 \(D_{\zeta,x_0}(\varrho)\) 的定义是以对数模长作圆周平均（\eqref{eq:xi-comoving-shifted-jensen-defect}）。
在同一半径 \(|w|=\varrho\) 上，引入平方模的归一化密度可把该平均精确分解为一个“振幅项”（二次能量）与一个“逆向熵项”（\(\mu\) 对密度的相对熵），从而把 RH 的一致小化判据转写为对能量--熵差的统一控制。

\begin{proposition}[Jensen 缺陷的熵--能量分解恒等式]\label{prop:xi-jensen-defect-entropy-energy-identity}
\leanpartial{paper\_xi\_jensen\_defect\_entropy\_energy\_identity}{the requested Lean signature is false as stated in this repository: taking \(Z=0\), \(f_0=2\), and \(KL=0\) makes the hypothesis force \(D=-\log 2\), while the conclusion collapses to \(D=0\)}
设 \(F\) 为单位圆盘 \(\mathbb D\) 上解析函数且 \(F(0)\neq 0\)。
对 \(0<\varrho<1\) 定义
\[
w_\varrho(\theta):=\bigl|F(\varrho e^{\mathrm{i}\theta})\bigr|^2,\qquad
Z_\varrho:=\int_{0}^{2\pi} w_\varrho(\theta)\,\frac{\dd\theta}{2\pi},
\qquad
p_\varrho(\theta):=\frac{w_\varrho(\theta)}{Z_\varrho}.
\]
并令 \(\mu(\dd\theta):=\frac{\dd\theta}{2\pi}\)。
若 \(\log p_\varrho\in L^1(\mu)\)，则 Jensen 缺陷
\[
D_F(\varrho):=\int_{0}^{2\pi}\log\bigl|F(\varrho e^{\mathrm{i}\theta})\bigr|\,\mu(\dd\theta)-\log|F(0)|
\]
满足精确恒等式
\begin{equation}\label{eq:xi-jensen-defect-entropy-energy-identity}
\boxed{\;
D_F(\varrho)
=\frac12\log\!\Bigl(\frac{Z_\varrho}{|F(0)|^2}\Bigr)
-\frac12\,D_{\mathrm{KL}}\!\bigl(\mu\mid p_\varrho\mu\bigr)\;}
\end{equation}
其中
\[
D_{\mathrm{KL}}\!\bigl(\mu\mid p_\varrho\mu\bigr)
:=\int_{0}^{2\pi}\log\!\Bigl(\frac{1}{p_\varrho(\theta)}\Bigr)\,\mu(\dd\theta)
\in[0,\infty].
\]
\end{proposition}

\begin{proof}
由定义有
\(
2D_F(\varrho)=\int_0^{2\pi}\log w_\varrho(\theta)\,\mu(\dd\theta)-2\log|F(0)|
\)。
另一方面，
\[
D_{\mathrm{KL}}\!\bigl(\mu\mid p_\varrho\mu\bigr)
=\int \log\frac{1}{p_\varrho}\,\dd\mu
=\int (\log Z_\varrho-\log w_\varrho)\,\dd\mu
=\log Z_\varrho-\int\log w_\varrho\,\dd\mu.
\]
即 \(\int\log w_\varrho\,\dd\mu=\log Z_\varrho-D_{\mathrm{KL}}(\mu\mid p_\varrho\mu)\)。
代回 \(2D_F\) 即得 \eqref{eq:xi-jensen-defect-entropy-energy-identity}。证毕。
\end{proof}

\begin{proposition}[幂映射拉回下的能量项与熵项不变]\label{prop:xi-jensen-defect-power-pullback-invariance}
\leanverified{paper\_xi\_jensen\_defect\_power\_pullback\_invariance}
取整数 \(m\ge 1\)，令 \(G(w):=F(w^m)\)。
对任意 \(0<\varrho<1\)，令 \(p^F_{\varrho^m}\) 与 \(p^G_\varrho\) 由命题 \ref{prop:xi-jensen-defect-entropy-energy-identity} 的规则分别从 \(F\) 与 \(G\) 构造。
则：
\begin{enumerate}
  \item \textbf{（平方模均值不变）}
  \begin{equation}\label{eq:xi-power-pullback-energy-invariant}
  \int_{0}^{2\pi}\bigl|G(\varrho e^{\mathrm{i}\theta})\bigr|^2\,\mu(\dd\theta)
  =\int_{0}^{2\pi}\bigl|F(\varrho^m e^{\mathrm{i}\phi})\bigr|^2\,\mu(\dd\phi).
  \end{equation}
  \item \textbf{（逆向相对熵不变）}
  \begin{equation}\label{eq:xi-power-pullback-entropy-invariant}
  D_{\mathrm{KL}}\!\bigl(\mu\mid p^G_\varrho\mu\bigr)=D_{\mathrm{KL}}\!\bigl(\mu\mid p^F_{\varrho^m}\mu\bigr).
  \end{equation}
  \item \textbf{（Jensen 缺陷精确重整）}
  \begin{equation}\label{eq:xi-power-pullback-jensen-renormalization}
  D_G(\varrho)=D_F(\varrho^m).
  \end{equation}
\end{enumerate}
\end{proposition}

\begin{proof}
\eqref{eq:xi-power-pullback-energy-invariant} 由换元 \(\phi=m\theta\)（利用 \(2\pi\)-周期性）直接得到：
\[
\int_0^{2\pi}|F(\varrho^m e^{\mathrm{i}m\theta})|^2\,\mu(\dd\theta)
=\int_0^{2\pi}|F(\varrho^m e^{\mathrm{i}\phi})|^2\,\mu(\dd\phi).
\]
由定义可见 \(p^G_\varrho(\theta)=p^F_{\varrho^m}(m\theta)\)（分子与归一化常数同按 \eqref{eq:xi-power-pullback-energy-invariant} 变换），
而映射 \(\theta\mapsto m\theta\) 保持 \(\mu\) 不变，故 \eqref{eq:xi-power-pullback-entropy-invariant} 成立。
将 \eqref{eq:xi-power-pullback-energy-invariant}--\eqref{eq:xi-power-pullback-entropy-invariant} 代入 \eqref{eq:xi-jensen-defect-entropy-energy-identity} 即得 \eqref{eq:xi-power-pullback-jensen-renormalization}。证毕。
\end{proof}

\begin{theorem}[Jensen 平均的熵正则化变分原理]\label{thm:xi-jensen-entropy-regularized-variational}
\leanverified{paper\_xi\_jensen\_entropy\_regularized\_variational}
令 \(w:\TT\to(0,\infty)\) 可测，且
\[
\int_{0}^{2\pi} w(\theta)\,\mu(\dd\theta)<\infty,
\qquad
\int_{0}^{2\pi}\log w(\theta)\,\mu(\dd\theta)>-\infty.
\]
则成立精确变分表述
\begin{equation}\label{eq:xi-jensen-entropy-regularized-variational}
\boxed{\;
\int_{0}^{2\pi}\log w\,\dd\mu
=\inf_{\substack{q>0\\ \int q\,\dd\mu=1}}
\left\{\log\!\int_{0}^{2\pi}\frac{w}{q}\,\dd\mu
-D_{\mathrm{KL}}\!\bigl(\mu\mid q\mu\bigr)\right\}\;}
\end{equation}
并且极小子在 \(\mu\)-几乎处处意义下唯一，且为
\begin{equation}\label{eq:xi-jensen-entropy-regularized-variational-minimizer}
\boxed{\;q^\star(\theta)=\frac{w(\theta)}{\int_{0}^{2\pi}w(\phi)\,\mu(\dd\phi)}\; }.
\end{equation}
\end{theorem}

\begin{proof}
对任意密度 \(q\)（\(q>0\)、\(\int q\,\dd\mu=1\)），有恒等分解
\[
\int\log w\,\dd\mu
=\int\log\frac{w}{q}\,\dd\mu+\int\log q\,\dd\mu
=\int\log\frac{w}{q}\,\dd\mu-D_{\mathrm{KL}}(\mu\mid q\mu).
\]
由 Jensen 不等式
\(
\int\log\frac{w}{q}\,\dd\mu\le \log\int\frac{w}{q}\,\dd\mu
\)
得
\[
\int\log w\,\dd\mu\le
\log\!\int\frac{w}{q}\,\dd\mu-D_{\mathrm{KL}}(\mu\mid q\mu),
\]
对 \(q\) 取下确界得到 \eqref{eq:xi-jensen-entropy-regularized-variational} 的“\(\le\)”。
取 \(q=q^\star=w/\int w\) 时，\(\frac{w}{q}\equiv \int w\) 为常数，Jensen 取等，从而下确界可达并给出等号。
若某个 \(q\) 达到下确界，则必满足 Jensen 取等条件，迫使 \(\frac{w}{q}\) 为常数；
结合 \(\int q\,\dd\mu=1\) 得 \(q=q^\star\)（\(\mu\)-a.e.），从而唯一性成立。证毕。
\end{proof}

\begin{proposition}[候选密度的上界证书：熵项、振幅项与振幅松弛]\label{prop:xi-jensen-variational-upperbound-certificate}
在定理 \ref{thm:xi-jensen-entropy-regularized-variational} 的设定下，给定任意密度 \(q\)（\(q>0\)、\(\int q\,\dd\mu=1\)），定义振幅松弛
\begin{equation}\label{eq:xi-jensen-oscillation-slack}
\Delta_{\mathrm{osc}}(w,q)
:=\log\!\int_{0}^{2\pi}\frac{w}{q}\,\dd\mu-\int_{0}^{2\pi}\log\!\Bigl(\frac{w}{q}\Bigr)\dd\mu\ \ge\ 0.
\end{equation}
则恒等分解成立：
\begin{equation}\label{eq:xi-jensen-variational-slack-identity}
\boxed{\;
\int_{0}^{2\pi}\log w\,\dd\mu
=\log\!\int_{0}^{2\pi}\frac{w}{q}\,\dd\mu
-D_{\mathrm{KL}}\!\bigl(\mu\mid q\mu\bigr)
-\Delta_{\mathrm{osc}}(w,q)\;}
\end{equation}
从而得到可审计上界
\begin{equation}\label{eq:xi-jensen-variational-upperbound}
\boxed{\;
\int_{0}^{2\pi}\log w\,\dd\mu
\le
\log\!\int_{0}^{2\pi}\frac{w}{q}\,\dd\mu
-D_{\mathrm{KL}}\!\bigl(\mu\mid q\mu\bigr)\;}
\end{equation}
并且若存在 \(\eta\ge 0\) 使得
\begin{equation}\label{eq:xi-jensen-multiplicative-bracket}
e^{-\eta}\le \frac{w(\theta)}{\bigl(\int w\,\dd\mu\bigr)\,q(\theta)}\le e^{\eta}\qquad(\mu\text{-a.e.}),
\end{equation}
则有定量控制
\begin{equation}\label{eq:xi-jensen-oscillation-slack-bound}
\boxed{\;0\le \Delta_{\mathrm{osc}}(w,q)\le 2\eta.\;}
\end{equation}
\end{proposition}

\begin{proof}
由
\(
\int\log w=\int\log\frac{w}{q}+\int\log q
\)
与 \(\int\log q=-D_{\mathrm{KL}}(\mu\mid q\mu)\) 得
\[
\int\log w\,\dd\mu
=\int\log\frac{w}{q}\,\dd\mu-D_{\mathrm{KL}}(\mu\mid q\mu).
\]
再用 \eqref{eq:xi-jensen-oscillation-slack} 的定义改写为 \eqref{eq:xi-jensen-variational-slack-identity}；
由 \(\Delta_{\mathrm{osc}}\ge 0\) 得 \eqref{eq:xi-jensen-variational-upperbound}。
若 \eqref{eq:xi-jensen-multiplicative-bracket} 成立，则
\[
\frac{w(\theta)}{q(\theta)}\le \Bigl(\int w\,\dd\mu\Bigr)e^{\eta}
\quad\Longrightarrow\quad
\log\!\int\frac{w}{q}\,\dd\mu\le \log\!\int w\,\dd\mu+\eta,
\]
并且
\[
\log\!\Bigl(\frac{w(\theta)}{q(\theta)}\Bigr)\ge \log\!\int w\,\dd\mu-\eta
\quad\Longrightarrow\quad
\int\log\!\Bigl(\frac{w}{q}\Bigr)\dd\mu\ge \log\!\int w\,\dd\mu-\eta.
\]
相减即得 \eqref{eq:xi-jensen-oscillation-slack-bound}。证毕。
\leanverified{paper\_xi\_jensen\_variational\_upperbound\_certificate}
\end{proof}

\begin{proposition}[Poisson--KL 对齐假设下黄金链的一致证书速率]\label{prop:xi-golden-chain-poisson-kl-alignment-rate}
\leanverified{paper\_xi\_golden\_chain\_poisson\_kl\_alignment\_rate}
令 \(\varrho_m\) 为黄金分辨率链（\eqref{eq:xi-golden-resolution-chain-rho}），并令
\(
t_m:=\varphi^m=\frac{1+\varrho_m}{1-\varrho_m}
\)
（\eqref{eq:xi-golden-resolution-chain-rho-closed}）。
由 \(\varrho_m=\frac{\varphi^m-1}{\varphi^m+1}\) 还可得
\(
(1-\varrho_m^2)^{-1}=\frac14(\varphi^m+2+\varphi^{-m})\asymp t_m
\)。
假设存在 \(\RR\) 上概率测度族 \(\{\nu^{(x_0)}\}_{x_0\in\RR}\) 与一列非负误差 \(\{\varepsilon_m\}_{m\ge 1}\)，使得对所有 \(x_0\in\RR\) 与 \(m\ge 1\) 成立对齐不等式
\begin{equation}\label{eq:xi-golden-chain-alignment-inequality}
\boxed{\;
0\le D_{\zeta,x_0}(\varrho_m)
\le D_{\mathrm{KL}}\!\bigl(h^{(x_0)}_{t_m}\mid g_{t_m,m(x_0)}\bigr)+\varepsilon_m\;}
\end{equation}
其中
\(
h^{(x_0)}_{t}:=P_t*\nu^{(x_0)}
\)，并令 \(m(x_0):=\int_{\RR}\gamma\,\dd\nu^{(x_0)}(\gamma)\)；
且 \(g_{t,a}(x):=\frac{1}{\pi}\frac{t}{(x-a)^2+t^2}\)。
若满足一致矩控制
\[
\sup_{x_0\in\RR}\mathrm{Var}\bigl(\nu^{(x_0)}\bigr)\le V<\infty,
\qquad
\sup_{x_0\in\RR}\int_{\RR}|\gamma-m(x_0)|^4\,\dd\nu^{(x_0)}(\gamma)<\infty,
\qquad
\varepsilon_m=O(t_m^{-5}),
\]
则存在常数 \(C>0\) 使得
\begin{equation}\label{eq:xi-golden-chain-uniform-defect-rate}
\boxed{\;
\sup_{x_0\in\RR}D_{\zeta,x_0}(\varrho_m)
\le \frac{V^2}{8}\,\varphi^{-4m}+C\,\varphi^{-5m}\qquad(m\ge 1).\;}
\end{equation}
\end{proposition}

\begin{proof}
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的首项与四阶矩一致有界性，存在常数 \(C_0>0\) 使对所有 \(x_0\) 与 \(m\) 有
\[
D_{\mathrm{KL}}\!\bigl(h^{(x_0)}_{t_m}\mid g_{t_m,m(x_0)}\bigr)
=\frac{\mathrm{Var}(\nu^{(x_0)})^2}{8t_m^4}+O(t_m^{-6})
\le \frac{V^2}{8t_m^4}+C_0 t_m^{-6}.
\]
将其代入 \eqref{eq:xi-golden-chain-alignment-inequality} 并对 \(x_0\) 取上确界得
\[
\sup_{x_0}D_{\zeta,x_0}(\varrho_m)
\le \frac{V^2}{8t_m^4}+C_0 t_m^{-6}+\varepsilon_m.
\]
由 \(t_m=\varphi^m\) 与 \(\varepsilon_m=O(t_m^{-5})\)，得到 \eqref{eq:xi-golden-chain-uniform-defect-rate}。证毕。
\end{proof}

\begin{remark}[接口边界：对齐不等式与一致矩控制]\label{rem:xi-golden-chain-alignment-interface-boundary}
命题 \ref{prop:xi-golden-chain-poisson-kl-alignment-rate} 把黄金链上的一致缺陷上界归约为两类可审计接口：对齐不等式 \eqref{eq:xi-golden-chain-alignment-inequality} 以及对 \(\{\nu^{(x_0)}\}\) 的一致矩控制。
在当前文本中，上述两类接口的可验证来源仍需由 \(\zeta/\Xi\) 的端点模型、谱移/Carath\'eodory 数据与相应的可逆性实现给出；其典型闭合口径包括：
\begin{itemize}
  \item \textbf{对齐口径：}将 \(D_{\zeta,x_0}(\varrho)\) 以严格不等式或等式对齐到某个 Poisson 粗粒化 KL 量（或其双边夹逼），并给出随 \(m\) 衰减且对 \(x_0\) 一致的误差界；
  \item \textbf{矩界口径：}从端点可核验数据推出 \(\sup_{x_0}\mathrm{Var}(\nu^{(x_0)})\) 及必要的高阶绝对中心矩上界（以保证 Poisson--KL 的渐近常数与余项可在族内一致选取）。
\end{itemize}
一旦上述接口闭合，Poisson 粗粒化的显式首项常数（例如 \(t^{-4}\) 主项系数 \(1/8\) 及一般阶系数 \(\binom{2n-2}{n-1}/2^{2n}\)）即可把熵侧的证书速率完全定量化，并与 \(\varphi\)-分辨率的端点标度直接对接。
\end{remark}

\liminf_{m\to\infty} t_m^4D_{\zeta,x_0}(\varrho_m)\ \ge\ \frac{\mathrm{Var}(\nu^{(x_0)})^2}{8}\ >\ 0.
\]
从而任何能够推出 \(t_m^4\sup_{x_0\in\RR}D_{\zeta,x_0}(\varrho_m)\to 0\) 的可审计证书链，将自动排除 RH 失败。
\end{conjecture}

\begin{conjecture}[黄金尺度下的熵刚性与端点基准流]\label{conj:xi-horizon-entropy-rigidity-golden-reference}
令 \(\varrho_m\) 为黄金分辨率链（\eqref{eq:xi-golden-resolution-chain-rho}），并令 \(t_m:=\varphi^m=\frac{1+\varrho_m}{1-\varrho_m}\)（\eqref{eq:xi-golden-resolution-chain-rho-closed}）。
