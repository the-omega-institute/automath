\begin{corollary}[统一伸缩下的显影复杂度“异常”协变律：\texorpdfstring{$\mathscr{C}_{\partial}(m)=\mathscr{C}_{\partial}(1)+\log m$}{C(m)=C(1)+log m}]\label{cor:xi-toeplitz-visibility-complexity-anomaly}
在推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 的点状缺陷设定下，令 \(\mathsf P\) 为相应 Pick--Poisson Gram 矩阵，并定义边界熵势与边界体积势
\[
S_{\partial}:=-\log\det\mathsf P,
\qquad
V_{\partial}:=\log\mathrm{tr}(\mathsf P)=\log\Bigl(\sum_{j=1}^{\kappa}p_j\Bigr),
\]
以及显影复杂度功能
\[
\mathscr{C}_{\partial}:=S_{\partial}+(\kappa-1)V_{\partial}.
\]
则推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 的门槛括号可写为
\[
\log(1+C_0)+\mathscr{C}_{\partial}.
\]
并且在统一伸缩 \(\phi_m\)（命题 \ref{prop:xi-pick-poisson-det-scaling-law}）下，若令 \(\mathsf P(m)\) 为伸缩后的 Gram 矩阵并相应定义
\(
S_{\partial}(m):=-\log\det\mathsf P(m)
\)、\(V_{\partial}(m):=\log\mathrm{tr}(\mathsf P(m))\)、\(\mathscr{C}_{\partial}(m):=S_{\partial}(m)+(\kappa-1)V_{\partial}(m)\)，
则对一切 \(m>0\) 成立严格协变律
\[
\boxed{\;
\mathscr{C}_{\partial}(m)=\mathscr{C}_{\partial}(1)+\log m\; }.
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:xi-pick-poisson-det-factorization} 的分解
\(
\det\mathsf P=(\prod_j p_j)(\prod_{j<k}\rho_{jk}^2)
\)
得
\(
S_{\partial}=-\sum_j\log p_j-2\sum_{j<k}\log\rho_{jk}
\)，
从而推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 中括号内的对数势恰为
\(
\log(1+C_0)+S_{\partial}+(\kappa-1)V_{\partial}
\)。
\par
另一方面，由命题 \ref{prop:xi-pick-poisson-det-scaling-law}，
\(
\det\mathsf P(m)=m^{-\kappa}\det\mathsf P
\)
且 \(p_j(m)=m^{-1}p_j\)，故
\[
S_{\partial}(m)=S_{\partial}+\kappa\log m,
\qquad
V_{\partial}(m)=\log\Bigl(\sum_j p_j(m)\Bigr)=\log\Bigl(m^{-1}\sum_j p_j\Bigr)=V_{\partial}-\log m.
\]
代入 \(\mathscr{C}_{\partial}(m)=S_{\partial}(m)+(\kappa-1)V_{\partial}(m)\) 即得
\(
\mathscr{C}_{\partial}(m)=\mathscr{C}_{\partial}(1)+\log m
\)。
证毕。
\end{proof}
