\paragraph{Poisson--KL 耗散链的闭式封口与全阶矩阶跃}
以下条目仅依赖 Poisson 核 \(P_t\)、卷积流与二阶正规形接口，不把 RH 的真伪作为前提。
目标是把证书链中可直接审计的主常数与阶跃指数整理为统一闭式。

\begin{conclusion}[二阶正规形中的两处闭式常数]\label{con:xi-poisson-second-order-two-closed-constants}
设 \(\tilde\nu\) 为 \(\RR\) 上概率测度，
\[
\bar\gamma:=\int_{\RR}\gamma\,\dd\tilde\nu(\gamma),\qquad
\sigma^2:=\int_{\RR}(\gamma-\bar\gamma)^2\,\dd\tilde\nu(\gamma)\in(0,\infty),
\]
并假设
\(
\int_{\RR}|\gamma-\bar\gamma|^3\,\dd\tilde\nu(\gamma)<\infty
\)。
令
\[
h_t:=P_t*\tilde\nu,\qquad
g_t(x):=\frac{1}{\pi}\frac{t}{(x-\bar\gamma)^2+t^2}.
\]
则当 \(t\to\infty\) 时，
\[
\|h_t-g_t\|_{L^1(\RR)}
=\frac{9}{4\pi\sqrt3}\,\sigma^2\,t^{-2}+o(t^{-2}),
\]
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac18\,\sigma^4\,t^{-4}+o(t^{-4}).
\]
\end{conclusion}
\begin{proof}
第一式由定理 \ref{thm:xi-cauchy-poisson-tv-sharp-constant} 直接得到，
并用恒等式
\(
\frac{3\sqrt3}{4\pi}=\frac{9}{4\pi\sqrt3}
\)
改写常数。
第二式由命题 \ref{prop:xi-poisson-tv-kl-second-order}
或定理 \ref{thm:xi-poisson-kl-sharp-second-moment-only} 直接给出。证毕。
\end{proof}

\begin{conclusion}[Cauchy 核导数与 \texorpdfstring{\(\chi^2\)}{chi2} 能量的全阶闭式律]\label{con:xi-poisson-cauchy-derivative-chi2-all-orders}
令
\[
g_t(x):=\frac{1}{\pi}\frac{t}{x^2+t^2}\qquad(t>0),
\]
并记 \(g_t^{(n)}:=\partial_x^n g_t\)（\(n\ge1\)）。
则对任意整数 \(n\ge1\) 有
\[
\int_{\RR}\frac{\bigl(g_t^{(n)}(x)\bigr)^2}{g_t(x)}\,\dd x
=\frac{n^2(2n-2)!}{2^{2n-1}}\,t^{-2n}.
\]
\end{conclusion}
\begin{proof}
该式即命题 \ref{prop:xi-cauchy-poisson-derivative-energy-closed-form}
的等价重述。证毕。
\end{proof}

\begin{conclusion}[矩消失触发的熵衰减阶跃与主常数闭式]\label{con:xi-poisson-moment-vanishing-kl-step-closed-form}
设 \(k\ge2\)，并令
\[
\mu_j:=\int_{\RR}(\gamma-\bar\gamma)^j\,\dd\tilde\nu(\gamma),\qquad
\mu_1=\cdots=\mu_{k-1}=0,\qquad
\mu_k\neq0,
\]
且
\(
\int_{\RR}|\gamma-\bar\gamma|^{k+1}\,\dd\tilde\nu(\gamma)<\infty
\)。
在 \(h_t,g_t\) 同结论 \ref{con:xi-poisson-second-order-two-closed-constants} 的记号下，有
\[
h_t-g_t
=\frac{(-1)^k\mu_k}{k!}\,\partial_x^k g_t
+o(t^{-k-1})
\quad\text{于 }\RR\text{ 上一致成立},
\]
并且
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{\mu_k^2}{2(k!)^2}
\int_{\RR}\frac{\bigl(\partial_x^k g_t(x)\bigr)^2}{g_t(x)}\,\dd x
+o(t^{-2k}),
\]
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{(2k-2)!}{2^{2k}\,((k-1)!)^2}\,\mu_k^2\,t^{-2k}
+o(t^{-2k}).
\]
\end{conclusion}
\begin{proof}
首式由定理 \ref{thm:xi-cauchy-poisson-moment-gap-kl-leading}
证明中的 Taylor 主项识别直接得到。
第二式由同一定理的 KL 二次主项展开给出；
再代入结论 \ref{con:xi-poisson-cauchy-derivative-chi2-all-orders} 的闭式积分常数即得。证毕。
\end{proof}

\begin{conclusion}[Poisson 混合族的渐近反 Pinsker 常数]\label{con:xi-poisson-asymptotic-reverse-pinsker-constant}
在结论 \ref{con:xi-poisson-second-order-two-closed-constants} 的设定下，
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{2\pi^2}{27}\,\|h_t-g_t\|_{L^1(\RR)}^2+o(t^{-4})
\qquad(t\to\infty).
\]
\end{conclusion}
\begin{proof}
由结论 \ref{con:xi-poisson-second-order-two-closed-constants}
中两条主项式直接取比值即可；
等价表述见推论 \ref{cor:xi-cauchy-poisson-pinsker-sharp-ratio}。证毕。
\end{proof}

\begin{conclusion}[Poisson 熵耗散率的首项常数与幂律指数]\label{con:xi-poisson-fisher-leading-constant-half}
在结论 \ref{con:xi-poisson-second-order-two-closed-constants} 的设定下，记
\[
\mathcal I_1(h_t\mid g_t):=-\frac{\dd}{\dd t}D_{\mathrm{KL}}(h_t\mid g_t),
\]
则
\[
\mathcal I_1(h_t\mid g_t)
=\frac12\,\sigma^4\,t^{-5}+o(t^{-5})
\qquad(t\to\infty).
\]
\end{conclusion}
\begin{proof}
由推论 \ref{cor:xi-cauchy-poisson-fisher-sharp-asymptotics}
可得该展开；
亦可由结论 \ref{con:xi-poisson-second-order-two-closed-constants}
中的 \(D_{\mathrm{KL}}\) 主项逐项求导得到。证毕。
\end{proof}

\begin{conjecture}[共动端点谱混合的全阶矩熄灭与超幂律证书]\label{conj:xi-poisson-comoving-endpoint-all-moments-extinguish}
设每个端点参数 \(x_0\) 与分辨率参数 \(\varrho\in(0,1)\) 对应一族概率测度
\(
\nu_{x_0,\varrho}
\)。
记其中心矩
\[
\mu_j(x_0;\varrho)
:=\int_{\RR}(\gamma-\bar\gamma_{x_0,\varrho})^j\,\dd\nu_{x_0,\varrho}(\gamma),
\qquad
\bar\gamma_{x_0,\varrho}:=\int_{\RR}\gamma\,\dd\nu_{x_0,\varrho}(\gamma).
\]
若对每个 \(j\ge2\) 都有一致衰减
\[
\sup_{x_0}\,|\mu_j(x_0;\varrho)|
=o\!\bigl((1-\varrho)^{j/2}\bigr)
\qquad(\varrho\uparrow1),
\]
则对任意固定 \(k\ge2\)，由结论
\ref{con:xi-poisson-moment-vanishing-kl-step-closed-form}
可推出
\[
\sup_{x_0}
D_{\mathrm{KL}}\!\bigl(h_t^{(x_0,\varrho)}\mid g_t^{(x_0,\varrho)}\bigr)
=o(t^{-2k})\qquad(t\to\infty),
\]
并沿任意 \(t=t(\varrho)\uparrow\infty\) 获得超幂律衰减。
该机制与猜想 \ref{conj:xi-jensen-poisson-defect-alignment-refined}
兼容，可作为 Jensen 缺陷证书链极限闭合的充分结构条件。
\end{conjecture}
