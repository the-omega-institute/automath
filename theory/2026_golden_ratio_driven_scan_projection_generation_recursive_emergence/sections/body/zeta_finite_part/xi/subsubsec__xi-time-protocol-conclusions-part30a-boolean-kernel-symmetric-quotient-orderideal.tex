\paragraph{\texorpdfstring{$S_q$}{Sq}-分层商矩阵与下闭主子式的闭式结构}
在全空间对角化与整数不变量之外，还可在按集合大小分层的 \((q+1)\)-维商空间与任意下闭主子块上给出逐项可审计的闭式公式。

\begin{theorem}[\texorpdfstring{$S_q$}{Sq}-分层商矩阵的显式逆：反三角二项式核与 \texorpdfstring{$(0,0)$}{(0,0)} 单点补偿]\label{thm:xi-boolean-two-layer-symmetric-quotient-inverse}
\leanverified{paper\_xi\_boolean\_two\_layer\_symmetric\_quotient\_inverse}
令 \(q\ge 1\)，并在 \(V_q=2^{[q]}\) 上定义
\[
e_k:=\sum_{\card{U}=k}\mathbf 1_U\qquad(0\le k\le q).
\]
记 \(B^{(q)}(a,b)\) 在 \(\Span\{e_0,\dots,e_q\}\) 上的商矩阵为
\[
M^{(q)}(a,b):=(M_{k\ell})_{0\le k,\ell\le q},
\qquad
M_{k\ell}:=b\binom{q}{\ell}+d\binom{q-k}{\ell},
\qquad d:=a-b.
\]
若 \(a\neq 0\) 且 \(d\neq 0\)，则 \(M^{(q)}(a,b)\) 可逆，且对任意 \(0\le k,\ell\le q\) 有
\[
\bigl(M^{(q)}(a,b)^{-1}\bigr)_{k\ell}
=\frac{(-1)^{k+\ell-q}}{d}\binom{k}{k+\ell-q}\mathbf 1_{\{k+\ell\ge q\}}
-\frac{b}{ad}\,\delta_{k0}\delta_{\ell0}.
\]
因此 \(M^{(q)}(a,b)^{-1}\) 的非零支撑恰为反三角区域
\(\{(k,\ell):k+\ell\ge q\}\) 与单点 \((0,0)\) 的并。
\end{theorem}
\begin{proof}
设
\[
C^{(q)}:=(C_{k\ell})_{0\le k,\ell\le q},
\qquad
C_{k\ell}:=\binom{q-k}{\ell},
\]
并记 \(\mathbf 1:=(1,\dots,1)^{\mathsf T}\)、\(v:=(\binom{q}{0},\dots,\binom{q}{q})^{\mathsf T}\)。
则由定义直接得到
\[
M^{(q)}(a,b)=d\,C^{(q)}+b\,\mathbf 1\,v^{\mathsf T}.
\]
矩阵 \(C^{(q)}\) 是 Pascal 矩阵的行反序形式，故为整可逆矩阵，且其逆矩阵满足二项式反演闭式
\[
\bigl(C^{(q)}{}^{-1}\bigr)_{k\ell}
=(-1)^{k+\ell-q}\binom{k}{k+\ell-q}\mathbf 1_{\{k+\ell\ge q\}}.
\]
对秩一扰动应用 Sherman--Morrison 公式：
\[
M^{-1}
=\frac{1}{d}C^{-1}
-\frac{b}{d^2}\cdot
\frac{C^{-1}\mathbf 1\,v^{\mathsf T}C^{-1}}
{1+\frac{b}{d}v^{\mathsf T}C^{-1}\mathbf 1}.
\]
由二项式反演有 \(C^{-1}\mathbf 1=e_0\)；
又因 \(v^{\mathsf T}\) 正是 \(C\) 的第 \(0\) 行，故 \(v^{\mathsf T}C^{-1}=e_0^{\mathsf T}\)，于是
\[
v^{\mathsf T}C^{-1}\mathbf 1=1,\qquad
C^{-1}\mathbf 1\,v^{\mathsf T}C^{-1}=e_0e_0^{\mathsf T}=E_{00}.
\]
并且分母化为 \(1+b/d=a/d\)。
代回即得
\[
M^{-1}=\frac{1}{d}C^{-1}-\frac{b}{ad}E_{00},
\]
逐项展开即为所述闭式。证毕。
\end{proof}

\begin{theorem}[分层商矩阵的 Smith 正规形：端点由 \texorpdfstring{$\gcd(a,b)$}{gcd(a,b)} 决定，中间层恒为 \texorpdfstring{$|a-b|$}{|a-b|}]\label{thm:xi-boolean-two-layer-symmetric-quotient-smith}
令 \(q\ge 1\)、\(a,b\in\ZZ\)，记 \(d:=a-b\)、\(g:=\gcd(a,b)=\gcd(a,d)\)。
则 \(M^{(q)}(a,b)\in\Mat_{q+1}(\ZZ)\) 的 Smith 正规形可取为
\[
\mathrm{SNF}\!\bigl(M^{(q)}(a,b)\bigr)
=\operatorname{diag}\!\Bigl(
g,\ \underbrace{|d|,\dots,|d|}_{q-1\text{ 次}},\ \frac{|ad|}{g}
\Bigr).
\]
\leanverified{paper\_xi\_boolean\_two\_layer\_symmetric\_quotient\_smith}
\end{theorem}
\begin{proof}
沿用定理 \ref{thm:xi-boolean-two-layer-symmetric-quotient-inverse} 的记号，左乘幺模矩阵 \(C^{(q)}{}^{-1}\) 得
\[
N:=C^{(q)}{}^{-1}M^{(q)}(a,b)=dI_{q+1}+b\,e_0v^{\mathsf T}.
\]
故 \(N\) 的第 \(0\) 行为 \((a,\ b\binom{q}{1},\dots,b\binom{q}{q})\)，其余部分在主对角上为 \(d\)。
对每个 \(1\le j\le q-1\)，依次施行整幺模变换
\[
\mathrm{col}_j\leftarrow \mathrm{col}_j-\binom{q}{j}\,\mathrm{col}_q,\qquad
\mathrm{row}_q\leftarrow \mathrm{row}_q+\binom{q}{j}\,\mathrm{row}_j.
\]
可同时消去第 \(0\) 行第 \(j\) 列与第 \(q\) 行第 \(j\) 列而不破坏其余 \(d\)-对角结构。
经行列重排后，\(N\) 与
\[
\operatorname{diag}\!\Bigl(
dI_{q-1},\ \begin{pmatrix}a&b\\[2pt]0&d\end{pmatrix}
\Bigr)
\]
幺模等价。
因此问题化归为 \(2\times 2\) 块
\(
A_2=\begin{psmallmatrix}a&b\\ 0&d\end{psmallmatrix}
\)
的 Smith 形。
有
\[
\gcd\text{(全部条目)}=\gcd(a,b,d)=g,\qquad
|\det A_2|=|ad|,
\]
故
\[
\mathrm{SNF}(A_2)=\operatorname{diag}\!\Bigl(g,\frac{|ad|}{g}\Bigr).
\]
与前面的 \(q-1\) 个 \(|d|\) 因子合并即得结论。证毕。
\end{proof}

\begin{corollary}[余核分解与模 \texorpdfstring{$m$}{m} 核计数]\label{cor:xi-boolean-two-layer-symmetric-quotient-coker-modm}
\leanverified{paper\_xi\_boolean\_two\_layer\_symmetric\_quotient\_coker\_modm}
在定理 \ref{thm:xi-boolean-two-layer-symmetric-quotient-smith} 设定下，
\[
\operatorname{coker}\!\bigl(M^{(q)}(a,b)\bigr)
\cong
\ZZ/g\ZZ
\ \oplus\
(\ZZ/|d|\ZZ)^{\oplus(q-1)}
\ \oplus\
\ZZ/(|ad|/g)\ZZ.
\]
并且对任意整数 \(m\ge 1\)，同余核基数满足
\[
\#\ker\bigl(M^{(q)}(a,b)\bmod m\bigr)
=\gcd(g,m)\cdot \gcd(|d|,m)^{q-1}\cdot \gcd(|ad|/g,m).
\]
\end{corollary}
\begin{proof}
由 Smith 结构定理直接代入不变量因子即可。证毕。
\end{proof}

\begin{theorem}[全空间实惯性指数的显式计数]\label{thm:xi-boolean-two-layer-inertia-count}
\leanverified{paper\_xi\_boolean\_two\_layer\_inertia\_count}
令 \(q\ge 2\)、\(a,b\in\RR\)、\(d:=a-b\)、\(N:=2^q\)。
则 \(B^{(q)}(a,b)\) 的惯性指数 \((n_+,n_-,n_0)\) 满足：
\begin{enumerate}
  \item 若 \(a\neq 0\) 且 \(d\neq 0\)，则 \(n_0=0\)，并且
  \[
  (n_+,n_-)=
  \begin{cases}
  \bigl(2^{q-1}-1+\mathbf 1_{\{a>0\}},\ 2^{q-1}+\mathbf 1_{\{a<0\}}\bigr),&d>0,\\[2mm]
  \bigl(2^{q-1}+\mathbf 1_{\{a>0\}},\ 2^{q-1}-1+\mathbf 1_{\{a<0\}}\bigr),&d<0.
  \end{cases}
  \]
  特别地，签名 \(\sigma:=n_+-n_-\) 满足
  \[
  \sigma=\operatorname{sgn}(a)-\operatorname{sgn}(d).
  \]
  \item 若 \(a=0\) 且 \(d\neq 0\)，则 \(n_0=1\)，且其余 \(n_\pm\) 由上式取
  \(\mathbf 1_{\{a>0\}}=\mathbf 1_{\{a<0\}}=0\) 得到。
  \item 若 \(d=0\)（即 \(a=b\)），则 \(B^{(q)}(a,b)=bJ_N\)，故
  \[
  (n_+,n_-,n_0)=
  \begin{cases}
  (1,0,N-1),&b>0,\\
  (0,1,N-1),&b<0,\\
  (0,0,N),&b=0.
  \end{cases}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-boolean-two-layer-zeta-mu-diagonalization}，
\[
\mu_q^{\mathsf T}B^{(q)}(a,b)\mu_q
=\operatorname{diag}\bigl(a,\ d(-1)^{\card{S}}\ (S\neq\varnothing)\bigr).
\]
由 Sylvester 惯性定理，惯性指数由该对角元的符号计数给出。
对 \(q\ge 2\)，非空子集中奇基数有 \(2^{q-1}\) 个，偶基数有 \(2^{q-1}-1\) 个；
将 \(S=\varnothing\) 对应条目 \(a\) 并入计数即可得到前两种情形。
当 \(d=0\) 时矩阵退化为 \(bJ_N\)，其秩为 \(1\)（\(b\neq 0\)）或 \(0\)（\(b=0\)），结论随即成立。证毕。
\end{proof}

\begin{theorem}[任意下闭主子式的精确行列式]\label{thm:xi-boolean-two-layer-order-ideal-principal-minor}
设 \(q\ge 1\)，\(R\) 为含单位交换环，\(a,b\in R\)，\(d:=a-b\)。
对任意下闭集（order ideal）\(I\subseteq V_q\)，主子式满足
\[
\det B^{(q)}(a,b)[I,I]
=a^{\mathbf 1_{\{\varnothing\in I\}}}
d^{\,|I|-\mathbf 1_{\{\varnothing\in I\}}}
(-1)^{\sum_{S\in I\setminus\{\varnothing\}}\card{S}}.
\]
特别地，若 \(\varnothing\notin I\)，则
\[
\det B^{(q)}(a,b)[I,I]
=d^{|I|}(-1)^{\sum_{S\in I}\card{S}}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-boolean-two-layer-zeta-mu-diagonalization}，有
\[
B^{(q)}(a,b)=Z_q^{\mathsf T}\Delta_q(a,b)Z_q.
\]
限制到指标集 \(I\) 得
\[
B^{(q)}(a,b)[I,I]=Z_I^{\mathsf T}\Delta_I Z_I.
\]
由于 \(I\) 下闭，取按包含关系的线性扩张后，\(Z_I\) 为单位三角矩阵，故 \(\det Z_I=1\)。
于是
\[
\det B^{(q)}(a,b)[I,I]
=(\det Z_I)^2\det\Delta_I
=\det\Delta_I.
\]
\(\Delta_I\) 的对角元恰为：当 \(\varnothing\in I\) 时给出一项 \(a\)，其余 \(S\neq\varnothing\) 给出 \(d(-1)^{\card{S}}\)。
逐项相乘即得结论。证毕。
\end{proof}

\begin{theorem}[下闭主子矩阵的 Smith 形形状不变性]\label{thm:xi-boolean-two-layer-order-ideal-smith-shape-invariance}
\leanverified{paper\_xi\_boolean\_two\_layer\_order\_ideal\_smith\_shape\_invariance}
设 \(q\ge 1\)、\(a,b\in\ZZ\)、\(d:=a-b\)、\(g:=\gcd(a,b)\)。
令 \(I\subseteq V_q\) 为非空下闭集，记 \(m:=|I|\)。
则 \(B^{(q)}(a,b)[I,I]\) 的 Smith 正规形由以下三类给出：
\begin{enumerate}
  \item 若 \(\varnothing\notin I\)，则
  \[
  \mathrm{SNF}\!\bigl(B^{(q)}(a,b)[I,I]\bigr)
  =\operatorname{diag}\!\bigl(\underbrace{|d|,\dots,|d|}_{m\text{ 次}}\bigr).
  \]
  \item 若 \(\varnothing\in I\) 且 \(m=1\)，则
  \[
  \mathrm{SNF}\!\bigl(B^{(q)}(a,b)[I,I]\bigr)=\operatorname{diag}(|a|).
  \]
  \item 若 \(\varnothing\in I\) 且 \(m\ge 2\)，则
  \[
  \mathrm{SNF}\!\bigl(B^{(q)}(a,b)[I,I]\bigr)
  =\operatorname{diag}\!\Bigl(
  g,\ \underbrace{|d|,\dots,|d|}_{m-2\text{ 次}},\ \frac{|ad|}{g}
  \Bigr).
  \]
\end{enumerate}
因此在固定 \((a,b)\) 与 \(m\) 时，含 \(\varnothing\) 的所有 \(m\) 元下闭集对应主子矩阵两两幺模等价；不含 \(\varnothing\) 的情形同样仅由 \(m\) 决定。
\end{theorem}
\begin{proof}
沿用定理 \ref{thm:xi-boolean-two-layer-order-ideal-principal-minor} 的分解
\(
B[I,I]=Z_I^{\mathsf T}\Delta_I Z_I
\)
且 \(Z_I\in\mathrm{GL}_m(\ZZ)\)，故
\[
\mathrm{SNF}\!\bigl(B[I,I]\bigr)=\mathrm{SNF}(\Delta_I).
\]
若 \(\varnothing\notin I\)，\(\Delta_I\) 对角元全为 \(\pm d\)，与 \(\operatorname{diag}(d,\dots,d)\) 幺模等价，得第一类。
若 \(\varnothing\in I\)，则 \(\Delta_I\) 与 \(\operatorname{diag}(a,d,\dots,d)\) 幺模等价（符号 \(\pm\) 可由单位吸收）。
当 \(m=1\) 得第二类；当 \(m\ge 2\) 时，仅需将 \(2\times 2\) 块 \(\operatorname{diag}(a,d)\) 化为
\(
\operatorname{diag}(g,|ad|/g)
\)，其余 \(d\) 因子保持不变，即得第三类。证毕。
\end{proof}
