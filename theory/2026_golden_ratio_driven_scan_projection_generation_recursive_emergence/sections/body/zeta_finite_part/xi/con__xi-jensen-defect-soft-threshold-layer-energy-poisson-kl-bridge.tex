\paragraph{Jensen 缺陷的软阈值下界链：深度核、层析二阶量与 Poisson--KL 桥接接口}
以下统一采用 Jensen 缺陷
\[
D_{\zeta,x_0}(\varrho)
:=\frac1{2\pi}\int_0^{2\pi}\log\bigl|F_{\zeta,x_0}(\varrho e^{\mathrm{i}\theta})\bigr|\,\dd\theta-\log|F_{\zeta,x_0}(0)|
\]
及 Cayley 深度恒等式
\[
h_{x_0}(\rho)=1-|w_{\rho,x_0}|^2
=\frac{4\delta}{(\gamma-x_0)^2+(1+\delta)^2},
\qquad
w_{\rho,x_0}:=\mathcal C\!\bigl((\gamma-x_0)+\mathrm{i}\delta\bigr).
\]

\begin{conclusion}[Jensen 缺陷的软阈值深度下界]\label{con:xi-jensen-soft-threshold-depth-lower-bound}
设 \(f\) 在单位圆盘 \(\mathbb D\) 解析且 \(f(0)\neq 0\)，零点（按重数）为 \(\{a_k\}\subset\mathbb D\)。
在 Jensen 公式可用的常规条件下（例如有限零点，或 Blaschke 条件），对任意 \(0<\varrho<1\) 有
\[
D_f(\varrho)
\ge
\frac1{2\varrho^2}\sum_k\bigl(\varrho^2-|a_k|^2\bigr)_+
=
\frac1{2\varrho^2}\sum_k\bigl((1-|a_k|^2)-(1-\varrho^2)\bigr)_+.
\]
\end{conclusion}
\begin{proof}[证明要点]
由 Jensen 公式
\[
D_f(\varrho)=\sum_{|a_k|<\varrho}\log\frac{\varrho}{|a_k|}.
\]
对任意 \(|a|<\varrho\)，令 \(y=\varrho^2/|a|^2>1\)。由初等不等式
\(
\log y\ge (y-1)/y
\)
得
\[
\log\frac{\varrho}{|a|}
=\frac12\log\frac{\varrho^2}{|a|^2}
\ge
\frac12\cdot\frac{\varrho^2-|a|^2}{\varrho^2}.
\]
对 \(|a_k|<\varrho\) 求和即得结论。证毕。
\end{proof}

\begin{conclusion}[对 \texorpdfstring{$F_{\zeta,x_0}$}{F(zeta,x0)} 的逐零点正项下界]\label{con:xi-jensen-soft-threshold-defect-kernel-explicit-sum}
在有限缺陷口径下，设
\[
\rho_j=\frac12-\delta_j+\mathrm{i}\gamma_j,\qquad
\delta_j>0,\qquad
j=1,\dots,J
\]
给出 \(F_{\zeta,x_0}\) 在 \(\mathbb D\) 内的全部零点像 \(w_{\rho_j,x_0}\)（按重数 \(m_j\)）。
则对任意 \(\varrho\in(0,1)\) 与 \(x_0\in\RR\) 有
\[
D_{\zeta,x_0}(\varrho)
\ge
\frac1{2\varrho^2}\sum_{j=1}^J
m_j\bigl(h_{x_0}(\rho_j)-(1-\varrho^2)\bigr)_+.
\]
\end{conclusion}
\begin{proof}[证明要点]
将结论 \ref{con:xi-jensen-soft-threshold-depth-lower-bound} 应用于 \(f=F_{\zeta,x_0}\)，并以
\(
a_k=w_{\rho_j,x_0}
\)
（按重数）代入：
\[
D_{\zeta,x_0}(\varrho)\ge
\frac1{2\varrho^2}\sum_j m_j\bigl(\varrho^2-|w_{\rho_j,x_0}|^2\bigr)_+.
\]
再用
\(
1-|w_{\rho_j,x_0}|^2=h_{x_0}(\rho_j)
\)
改写即得。证毕。
\end{proof}

\begin{conclusion}[\texorpdfstring{$\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0$}{integrated defect} 的闭式强制下界]\label{con:xi-jensen-soft-threshold-integrated-arctan-closed-lower-bound}
沿用结论 \ref{con:xi-jensen-soft-threshold-defect-kernel-explicit-sum} 的记号，令
\[
\tau:=1-\varrho^2\in(0,1).
\]
则有
\[
\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0
\ge
\frac1{2\varrho^2}\sum_{j=1}^{J}m_j\,\mathcal I_{\mathrm{soft}}(\delta_j;\tau),
\]
其中
\[
\mathcal I_{\mathrm{soft}}(\delta;\tau):=
\begin{cases}
0, & \tau\ge \dfrac{4\delta}{(1+\delta)^2},\\[2mm]
\dfrac{8\delta}{1+\delta}\arctan\!\Bigl(\dfrac{A(\delta;\tau)}{1+\delta}\Bigr)-2\tau A(\delta;\tau),
& \tau<\dfrac{4\delta}{(1+\delta)^2},
\end{cases}
\]
\[
A(\delta;\tau):=\sqrt{\frac{4\delta}{\tau}-(1+\delta)^2}.
\]
\end{conclusion}
\begin{proof}[证明要点]
由结论 \ref{con:xi-jensen-soft-threshold-defect-kernel-explicit-sum}：
\[
\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0
\ge
\frac1{2\varrho^2}\sum_j m_j
\int_{\RR}\left(\frac{4\delta_j}{(\gamma_j-x_0)^2+(1+\delta_j)^2}-\tau\right)_+\dd x_0.
\]
对每项作平移 \(u=x_0-\gamma_j\) 后，仅依赖 \(\delta_j\)。
当 \(\tau\ge4\delta/(1+\delta)^2\) 时支集为空，积分为 \(0\)。
否则支集为 \(|u|<A(\delta;\tau)\)，并且
\[
\int_{-A}^{A}\frac{4\delta}{u^2+(1+\delta)^2}\,\dd u
=\frac{8\delta}{1+\delta}\arctan\!\Bigl(\frac{A}{1+\delta}\Bigr),\qquad
\int_{-A}^{A}\tau\,\dd u=2\tau A.
\]
相减即得闭式核 \(\mathcal I_{\mathrm{soft}}\)。证毕。
\end{proof}

\begin{conclusion}[\texorpdfstring{$\varrho\uparrow 1$}{rho->1} 时的极限刚性下界]\label{con:xi-jensen-soft-threshold-integrated-limit-rigidity}
在结论 \ref{con:xi-jensen-soft-threshold-integrated-arctan-closed-lower-bound} 的设定下，
\[
\lim_{\tau\downarrow 0}\mathcal I_{\mathrm{soft}}(\delta;\tau)
=\frac{4\pi\delta}{1+\delta}\qquad(\delta>0).
\]
从而
\[
\liminf_{\varrho\uparrow 1}\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0
\ge
2\pi\sum_{j=1}^{J}m_j\frac{\delta_j}{1+\delta_j}.
\]
\end{conclusion}
\begin{proof}[证明要点]
当 \(\tau\downarrow 0\) 时，\(A(\delta;\tau)\to\infty\)，故
\[
\arctan\!\Bigl(\frac{A}{1+\delta}\Bigr)\to\frac\pi2.
\]
同时
\[
2\tau A
=2\tau\sqrt{\frac{4\delta}{\tau}-(1+\delta)^2}
=4\sqrt{\delta\tau}+o(1)\to 0.
\]
代入 \(\mathcal I_{\mathrm{soft}}\) 即得极限。
再结合
\(
\frac1{2\varrho^2}\to\frac12
\)
得到整体下界。证毕。
\end{proof}

\begin{conclusion}[高度窗内离线计数导致 \texorpdfstring{$\sup D_{\zeta,x_0}$}{sup defect} 的强制下界]\label{con:xi-jensen-soft-threshold-window-sup-lower-bound}
给定 \(T>0\)、\(\delta_\star>0\)、\(\varrho\in(0,1)\)，令 \(\tau=1-\varrho^2\)。
若
\[
\tau<\frac{4\delta_\star}{(1+\delta_\star)^2},
\qquad
A_\star:=A(\delta_\star;\tau)>0,
\]
定义内窗口计数
\[
M_T^{\mathrm{in}}(\delta_\star;\varrho)
:=\sum_{\substack{1\le j\le J\\ |\gamma_j|\le T-A_\star\\ \delta_j\ge \delta_\star}}m_j.
\]
则有显式下界
\[
\sup_{|x_0|\le T}D_{\zeta,x_0}(\varrho)
\ge
\frac{1}{4T\,\varrho^2}\,
M_T^{\mathrm{in}}(\delta_\star;\varrho)\,
\mathcal I_{\mathrm{soft}}(\delta_\star;\tau).
\]
\end{conclusion}
\begin{proof}[证明要点]
由结论 \ref{con:xi-jensen-soft-threshold-defect-kernel-explicit-sum}，
\[
D_{\zeta,x_0}(\varrho)\ge
\frac1{2\varrho^2}\sum_j m_j
\left(\frac{4\delta_j}{(\gamma_j-x_0)^2+(1+\delta_j)^2}-\tau\right)_+.
\]
在 \(\delta\in(0,1)\) 上，函数
\(
\delta\mapsto \frac{4\delta}{u^2+(1+\delta)^2}
\)
单调不减，因此对 \(\delta_j\ge\delta_\star\) 可下界为 \(\delta_\star\) 对应核。
若 \(|\gamma_j|\le T-A_\star\)，则该核支集区间 \([\gamma_j-A_\star,\gamma_j+A_\star]\subset[-T,T]\)，故
\[
\int_{-T}^{T}\left(\frac{4\delta_\star}{(\gamma_j-x_0)^2+(1+\delta_\star)^2}-\tau\right)_+\dd x_0
=\mathcal I_{\mathrm{soft}}(\delta_\star;\tau).
\]
对满足条件的 \(j\) 求和并积分于 \(x_0\in[-T,T]\)，得
\[
\int_{-T}^{T}D_{\zeta,x_0}(\varrho)\,\dd x_0
\ge
\frac{M_T^{\mathrm{in}}(\delta_\star;\varrho)}{2\varrho^2}\,
\mathcal I_{\mathrm{soft}}(\delta_\star;\tau).
\]
再用
\(
\int_{-T}^{T}D_{\zeta,x_0}\le 2T\sup_{|x_0|\le T}D_{\zeta,x_0}
\)
即得结论。证毕。
\end{proof}

\begin{conclusion}[Jensen 缺陷的一维加权积分控制平方深度能量]\label{con:xi-jensen-defect-weighted-integral-controls-square-depth-energy}
设 \(f\) 满足结论 \ref{con:xi-jensen-soft-threshold-depth-lower-bound} 的条件，零点为 \(\{a_k\}\)。
定义
\[
\mathcal E_2(f):=\sum_k(1-|a_k|^2)^2.
\]
则有确定性下界
\[
\int_0^1\varrho^3D_f(\varrho)\,\dd\varrho\ \ge\ \frac18\,\mathcal E_2(f).
\]
\end{conclusion}
\begin{proof}[证明要点]
由结论 \ref{con:xi-jensen-soft-threshold-depth-lower-bound}，
\[
D_f(\varrho)\ge \frac1{2\varrho^2}\sum_k\bigl(h_k-(1-\varrho^2)\bigr)_+,\qquad h_k:=1-|a_k|^2.
\]
乘以 \(\varrho^3\) 并积分：
\[
\int_0^1\varrho^3D_f(\varrho)\,\dd\varrho
\ge
\frac12\sum_k\int_0^1\varrho\bigl(h_k-(1-\varrho^2)\bigr)_+\,\dd\varrho.
\]
令 \(\tau=1-\varrho^2\)（\(\dd\tau=-2\varrho\,\dd\varrho\)），得
\[
\int_0^1\varrho^3D_f(\varrho)\,\dd\varrho
\ge
\frac14\sum_k\int_0^1(h_k-\tau)_+\,\dd\tau.
\]
而 \(\int_0^1(h-\tau)_+\,\dd\tau=\frac12h^2\)（\(0<h\le 1\)），故
\[
\int_0^1\varrho^3D_f(\varrho)\,\dd\varrho
\ge
\frac18\sum_k h_k^2
=\frac18\mathcal E_2(f).
\]
证毕。
\end{proof}

\begin{conclusion}[对 \texorpdfstring{$F_{\zeta,x_0}$}{F(zeta,x0)} 的二阶层析质量下界]\label{con:xi-jensen-defect-second-order-tomography-closed-lower-bound}
在结论 \ref{con:xi-jensen-soft-threshold-defect-kernel-explicit-sum} 的有限缺陷设定下，对每个固定 \(x_0\) 有
\[
\int_0^1\varrho^3D_{\zeta,x_0}(\varrho)\,\dd\varrho
\ge
\frac18\sum_{j=1}^{J}m_j\,h_{x_0}(\rho_j)^2.
\]
进一步在 \(x_0\) 上积分，得到闭式下界
\[
\int_{\RR}\!\int_0^1\varrho^3D_{\zeta,x_0}(\varrho)\,\dd\varrho\,\dd x_0
\ge
\pi\sum_{j=1}^{J}m_j\,\frac{\delta_j^2}{(1+\delta_j)^3}.
\]
\end{conclusion}
\begin{proof}[证明要点]
第一式直接由结论 \ref{con:xi-jensen-defect-weighted-integral-controls-square-depth-energy}
与 \(1-|w_{\rho_j,x_0}|^2=h_{x_0}(\rho_j)\) 得到。
第二式中
\[
\int_{\RR}h_{x_0}(\rho_j)^2\,\dd x_0
=
\int_{\RR}
\frac{16\delta_j^2}{\bigl((\gamma_j-x_0)^2+(1+\delta_j)^2\bigr)^2}\,\dd x_0
=
\frac{8\pi\delta_j^2}{(1+\delta_j)^3},
\]
其中用到标准积分公式
\(
\int_{\RR}\frac{\dd u}{(u^2+a^2)^2}=\frac{\pi}{2a^3}
\)。
再乘以前系数 \(1/8\) 即得结论。证毕。
\end{proof}

\begin{conclusion}[有限 \texorpdfstring{$\varrho$}{rho} 采样的二阶量下界证书]\label{con:xi-jensen-defect-finite-rho-sampling-second-order-certificate}
在有限缺陷设定下，固定 \(x_0\)。
取采样节点
\[
0=\varrho_0<\varrho_1<\cdots<\varrho_L<1,\qquad
w_\ell:=\int_{\varrho_{\ell-1}}^{\varrho_\ell}\varrho^3\,\dd\varrho
=\frac{\varrho_\ell^4-\varrho_{\ell-1}^4}{4},
\]
并记 \(\tau_L:=1-\varrho_L^2\)。
则有离散证书下界
\[
\sum_{\ell=1}^{L}w_\ell\,D_{\zeta,x_0}(\varrho_\ell)
\ge
\frac18\sum_{j=1}^{J}m_j\bigl(h_{x_0}(\rho_j)-\tau_L\bigr)_+^2.
\]
等价地，可写成
\[
\sum_{\ell=1}^{L}w_\ell\,D_{\zeta,x_0}(\varrho_\ell)
\ge
\frac18\sum_{j=1}^{J}m_j\,h_{x_0}(\rho_j)^2-R_L(x_0),
\]
其中
\[
R_L(x_0):=
\frac18\sum_{j=1}^{J}m_j
\Bigl(h_{x_0}(\rho_j)^2-\bigl(h_{x_0}(\rho_j)-\tau_L\bigr)_+^2\Bigr)\ge 0
\]
为显式可计算余项。
\end{conclusion}
\begin{proof}[证明要点]
由 Jensen 公式，\(\varrho\mapsto D_{\zeta,x_0}(\varrho)\) 单调不减。
故右端点求积给出
\[
\int_0^{\varrho_L}\varrho^3D_{\zeta,x_0}(\varrho)\,\dd\varrho
\le
\sum_{\ell=1}^{L}w_\ell D_{\zeta,x_0}(\varrho_\ell).
\]
另一方面，将结论 \ref{con:xi-jensen-soft-threshold-defect-kernel-explicit-sum} 的下界在区间 \((0,\varrho_L)\) 积分并作 \(\tau=1-\varrho^2\) 换元，得到
\[
\int_0^{\varrho_L}\varrho^3D_{\zeta,x_0}(\varrho)\,\dd\varrho
\ge
\frac18\sum_{j=1}^{J}m_j\bigl(h_{x_0}(\rho_j)-\tau_L\bigr)_+^2.
\]
两式合并即得第一式；第二式由恒等变形得到。证毕。
\end{proof}

\begin{conclusion}[端点化缺陷与 Poisson--KL 耗散的二阶能量主导对齐（结构性猜想）]\label{con:xi-endpoint-defect-poisson-kl-second-order-energy-dominant-alignment}
设 \(t=t(\varrho)\uparrow\infty\) 为 Cayley--Poisson 尺度对应，且存在由边界数据函子化得到的密度族 \(\{h^{(x_0)}\}_{x_0\in\RR}\)。
猜想存在绝对常数 \(C_1,C_2>0\)，使对 \(\varrho\) 充分接近 \(1\) 与所有 \(x_0\in\RR\) 成立
\[
D_{\zeta,x_0}(\varrho)
\le
C_1\,D_{\mathrm{KL}}\!\bigl(P_t*h^{(x_0)}\,\big\|\,P_t*c_0\bigr)
+C_2\!\int_0^1 r^3D_{\zeta,x_0}(r)\,\dd r.
\]
若该弱对齐形式成立，则由结论
\ref{con:xi-jensen-defect-second-order-tomography-closed-lower-bound}
与 \ref{con:xi-jensen-defect-finite-rho-sampling-second-order-certificate}
可把误差项转化为显式可审计二阶能量预算，从而为端点化证书链提供闭合接口。
\end{conclusion}
