\begin{corollary}[排斥半径的零点计数驱动律]\label{cor:dephys-repulsion-radius-zero-count-driving}

\begin{corollary}[Poisson--Busemann 恒等式：时间纤维坐标即端点 Poisson 权重]\label{cor:dephys-poisson-busemann-timefiber}
\leanverified{paper\_dephys\_poisson\_busemann\_timefiber}
对任意 \(w\in\mathbb{D}\)，令 \(t=\mathcal{C}^{-1}(w)=x+\mathrm{i}y\in\mathbb{H}\)。
则成立严格恒等式
\[
\boxed{\;y=B^{-1}(w)=\frac{1-|w|^2}{|1+w|^2}\;}
\]
其中 \(B^{-1}\) 为以端点 \(-1\) 为基点的指数化 Busemann 坐标（命题 \ref{prop:app-busemann-poisson-minus1}）。
因此，在端点紧化口径下，“时间纤维”并非额外引入的连续坐标轴，而是端点 \(-1\) 的谐测度密度（Poisson 核）本身。
\end{corollary}

\begin{proof}
令 \(t=x+\mathrm{i}y\in\mathbb{H}\) 且 \(w=\mathcal{C}(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\)。
则有
\[
1+w
=1+\frac{1+\mathrm{i}t}{1-\mathrm{i}t}
=\frac{2}{1-\mathrm{i}t},
\qquad
|1+w|^2=\frac{4}{|1-\mathrm{i}t|^2}=\frac{4}{x^2+(1+y)^2}.
\]
并且由
\(
|w|^2=\frac{|1+\mathrm{i}t|^2}{|1-\mathrm{i}t|^2}
\)
得
\[
1-|w|^2
=\frac{|1-\mathrm{i}t|^2-|1+\mathrm{i}t|^2}{|1-\mathrm{i}t|^2}
=\frac{4y}{x^2+(1+y)^2}.
\]
两式相除即得 \(\frac{1-|w|^2}{|1+w|^2}=y\)，从而 \(B^{-1}(w)=y\)。
特别地，出现在 \(1-|w|^2\) 与 \(|1+w|^2\) 中的常数 \(4\) 仅为 Cayley 规一化下共形因子 \(2\) 的平方，在该不变量比值中严格消去。证毕。
\end{proof}

\begin{remark}[Cayley 标尺与常数 \(4\) 的可规一化性：不变量只依赖比值与对数差]\label{rem:dephys-cayley-scale-gauge-four}
上述推导中的“\(4\)”属于坐标规一化常数，而非新的动力学不变量。
为说明这一点，考虑带标尺参数 \(\lambda>0\) 的 Cayley 族
\[
\mathcal{C}_{\lambda}(t):=\frac{\lambda+\mathrm{i}t}{\lambda-\mathrm{i}t}\qquad(t\in\mathbb{H}),
\]
它保持同一端点基准（\(t=\infty\) 仍映到 \(-1\)）但改变线性标尺。
直接计算得到
\[
|1+\mathcal{C}_{\lambda}(x+\mathrm{i}y)|^2
=\frac{4\lambda^2}{x^2+(\lambda+y)^2},
\qquad
1-|\mathcal{C}_{\lambda}(x+\mathrm{i}y)|^2
=\frac{4\lambda y}{x^2+(\lambda+y)^2}.
\]
因此指数化 Busemann/Poisson 比值仍严格消去共形平方常数，并给出
\[
B^{-1}\!\bigl(\mathcal{C}_{\lambda}(x+\mathrm{i}y)\bigr)=\frac{y}{\lambda}.
\]
取对数得到 \(b_{-1}:=\log B^{-1}=\log y-\log\lambda\)，表明标尺变换只改变 \(b_{-1}\) 的加性零点；
因此任何可观测几何量只能出现在对重标定不变的组合中（如 \(B^{-1}\) 的比值、或 \(b_{-1}\) 的对数差），而不可能依赖孤立的“\(4\)”常数。
在账本语义中亦可把 \(b_{-1}\) 视作一类“熵势/对数势”，此时规一化常数只决定加性零点而不改变差分观测；其形式角色与几何熵公式中的固定规一化常数（例如黑洞熵中的 \(1/4\)）同型。
\end{remark}

\begin{remark}[共形紧化中的 \(2^2\) 因子与“面积律”形式类比]\label{rem:dephys-conformal-factor-2square-area-law}
结论 \ref{cor:dephys-poisson-busemann-timefiber} 中的共形平方常数并非一维特例。
在更高维的紧化/投影坐标中，同样出现共形因子 \(\Omega(x)=\frac{2}{1+|x|^2}\) 及其平方（或更一般幂次），从而球面度量与面积元的拉回密度带有“\(4\)”型常数。
例如立体投影把二维球面度量写成
\[
\dd s^2=\frac{4}{(1+|z|^2)^2}|\dd z|^2,
\qquad
\dd A=\frac{4}{(1+|z|^2)^2}\,\dd A_{\RR^2}.
\]
这进一步表明：本文端点视界接口中出现的“\(4\)”应被理解为紧化共形因子的通用平方常数，而非特定算术对象所独有的不变量。
\end{remark}

\begin{proposition}[曲率规一化定律：圆盘双曲度量中的系数 \(4\) 等价于常曲率 \(-1\) 的唯一标尺]\label{prop:dephys-hyperbolic-curvature-normalization-four}
\leanverified{paper\_dephys\_hyperbolic\_curvature\_normalization\_four}
上半平面模型取双曲度量
\[
\dd s_{\mathbb{H}}^{2}:=\frac{\dd x^{2}+\dd y^{2}}{y^{2}}
\]
则其高斯曲率为 \(-1\)。
任取把 \(\mathbb{H}\) 共形等价到单位圆盘 \(\mathbb{D}\) 的 Möbius 变换 \(w=w(t)\)，圆盘上的双曲度量必可写为
\[
\boxed{\;
\dd s_{\mathbb{D}}^{2}=\frac{4\,|\dd w|^{2}}{(1-|w|^{2})^{2}}\;}
\]
并且该系数 \(4\) 由“曲率规一化”为 \(-1\) 唯一锁定：
若改写为
\(
\dd s_{\mathbb{D}}^{2}=\frac{\alpha\,|\dd w|^{2}}{(1-|w|^{2})^{2}}
\),
则曲率变为
\[
\boxed{\;K_{\alpha}=-\frac{4}{\alpha}.\;}
\]
\end{proposition}

\begin{proof}
对共形度量 \(\dd s^{2}=\rho(w)^{2}(\dd x^{2}+\dd y^{2})=\rho(w)^{2}|\dd w|^{2}\)，高斯曲率满足
\(
K=-\rho^{-2}\Delta(\log\rho)
\)
（\(\Delta\) 为欧氏拉普拉斯）。
取 \(\rho_{\alpha}(w)=\frac{\sqrt{\alpha}}{1-|w|^{2}}\)，则
\(
\log\rho_{\alpha}=\frac12\log\alpha-\log(1-|w|^{2})
\)，常数项拉普拉斯为零，且直接计算得
\(
\Delta\bigl(-\log(1-|w|^{2})\bigr)=\frac{4}{(1-|w|^{2})^{2}}
\)。
故
\[
K_{\alpha}
=-\frac{(1-|w|^{2})^{2}}{\alpha}\cdot \frac{4}{(1-|w|^{2})^{2}}
=-\frac{4}{\alpha}.
\]
因此 \(K_{\alpha}=-1\) 当且仅当 \(\alpha=4\)。
而任意 Möbius 等价 \(w=w(t)\) 仅改变坐标，不改变“曲率 \(-1\) 的唯一共形完备度量”这一规范化结论，故圆盘模型必落在 \(\alpha=4\) 的标尺上。证毕。
\end{proof}

\begin{remark}[视界测度规一化：Poisson 核的单位质量与曲率 \(-1\) 标尺的相容性]\label{rem:dephys-poisson-kernel-unit-mass-curvature-gauge}
在单位圆盘上，Poisson 核
\[
P_{w}(e^{\mathrm{i}\theta})=\frac{1-|w|^{2}}{|e^{\mathrm{i}\theta}-w|^{2}}

满足严格归一化

\boxed{\;\frac{1}{2\pi}\int_{0}^{2\pi}P_{w}(e^{\mathrm{i}\theta})\,\dd\theta=1.\;}

\paragraph{端点吸引半群与“时间”定义：唯一 Koenigs 深度}
时间纤维在本文中并非外加坐标，而是端点紧化后由伸缩半群强迫出现的一维深度账本。
下述定理把这一点严格化为端点线性化的刚性与唯一性：任何与尺度伸缩兼容的可逆审计，其连续累积只能沿这一维 Koenigs 深度发生。

\begin{theorem}[端点线性化的刚性：Koenigs 深度的唯一性]\label{thm:dephys-endpoint-koenigs-rigidity}
令 \(\Phi=\{\phi_m\}_{m>0}\subset \mathrm{Aut}(\mathbb{D})\) 为一族 Möbius 自同构，满足 \(\phi_{mn}=\phi_m\circ\phi_n\) 且对所有 \(m>0\) 有 \(\phi_m(\pm 1)=\pm 1\)。
设存在在 \(u=-1\) 的某个邻域内解析的坐标 \(h\) 满足 \(h(-1)=0\)、\(h'(-1)\neq 0\) 且
\begin{equation}\label{eq:dephys-koenigs-functional-equation}
h\bigl(\phi_m(u)\bigr)=\frac{1}{m}\,h(u)\qquad(\forall\,m>0).
\end{equation}
则在该邻域内存在常数 \(c\neq 0\) 使 \(h(u)=c\,\mathfrak{w}(u)\)，其中 \(\mathfrak{w}(u)=\frac{1+u}{1-u}\) 为端点半平面坐标（式 \eqref{eq:scale-linearization-halfplane-coordinate}）；并且对所有 \(m>0\) 与所有 \(u\in\mathbb{D}\) 有 \(\phi_m=\psi_m\)，其中 \(\psi_m\) 由式 \eqref{eq:scale-psi-m} 给出。
因此端点深度（Busemann/Koenigs 深度）
\[
Y(u):=-\log\bigl|\mathfrak{w}(u)\bigr|
\]
满足严格平移律
\[
Y\!\bigl(\psi_m(u)\bigr)=Y(u)+\log m
\]
（即式 \eqref{eq:endpoint-busemann-addition-law}）。
\end{theorem}

\begin{proof}
由 \eqref{eq:dephys-koenigs-functional-equation} 在 \(u=-1\) 处求导并用 \(h'(-1)\neq 0\) 得 \(\phi_m'(-1)=1/m\)。
另一方面，端点坐标 \(\mathfrak{w}\) 将 \(\mathbb{D}\) 共轭到右半平面 \(\mathbb{H}_{\mathrm{R}}\)（式 \eqref{eq:scale-linearization-halfplane-coordinate}），并把 \(\phi_m\) 共轭为 \(\mathbb{H}_{\mathrm{R}}\) 的自同构且保持 \(0\) 与 \(\infty\) 不动；故存在 \(a_m>0\) 使
\(
\mathfrak{w}(\phi_m(u))=a_m\,\mathfrak{w}(u)
\)，并由链式法则得 \(a_m=\phi_m'(-1)=1/m\)。
于是 \(\mathfrak{w}(\phi_m(u))=\mathfrak{w}(u)/m\)，与命题 \ref{prop:scale-linearization-identity-multipliers} 的 \(\mathfrak{w}(\psi_m(u))=\mathfrak{w}(u)/m\) 一致；因 \(\mathfrak{w}\) 单射，得 \(\phi_m=\psi_m\)。
再令 \(g(u):=\frac{h(u)}{\mathfrak{w}(u)}\)（在 \(u=-1\) 邻域内解析），由 \eqref{eq:dephys-koenigs-functional-equation} 与 \(\mathfrak{w}(\psi_m(u))=\mathfrak{w}(u)/m\) 得 \(g(\psi_m(u))=g(u)\)；
取任意 \(m>1\)，迭代 \(\psi_m^{\circ n}(u)\to-1\)（端点吸引），由连续性得 \(g(u)=g(-1)\) 在邻域内恒定，从而 \(h=c\,\mathfrak{w}\)。
最后的平移律即为式 \eqref{eq:endpoint-busemann-addition-law}。证毕。
\end{proof}

\begin{remark}[端点伸缩半群中的另一处“\(4\)”：Möbius 判别式锁定时间参数]\label{rem:dephys-psi-discriminant-locks-time}
对尺度伸缩诱导的端点自同构 \(\psi_m\)（式 \eqref{eq:scale-psi-m}），将其写作 Möbius 形式 \(\psi_m(u)=(au+b)/(cu+d)\)，其中
\[
a=d=m+1,\qquad b=c=1-m,
\]
则判别式（行列式）满足
\[
\boxed{\;ad-bc=(m+1)^2-(1-m)^2=4m.\;}
\]
该常数并不产生新的不变量：它只是端点规一化的代数影像，并通过导数公式
\(
\psi_m'(u)=\frac{ad-bc}{(cu+d)^2}
\)
唯一锁定端点乘子与时间加法律 \(Y(\psi_m(u))=Y(u)+\log m\)（见附录命题 \ref{prop:scale-linearization-identity-multipliers} 与式 \eqref{eq:endpoint-busemann-addition-law}）。
与注 \ref{rem:dephys-cayley-scale-gauge-four} 同理，该类常数只影响势函数的加性零点，而不改变任何可审计差分。
\end{remark}

\paragraph{周期折叠的最小离散外置：绕数证书（winding certificate）}
在 Dirichlet 参数化 \(z=\lambda^{-s}\) 中，虚向角变量天然折叠；若坚持用主值辐角 \(\mathrm{Arg}\) 作为可见角信息，则可逆化必然强迫外置一个至少可数的离散纤维。

\begin{theorem}[Dirichlet 虚向折叠的最小可逆化：绕数证书的不可省略性]\label{thm:dephys-dirichlet-winding-minimal}
\leanverified{paper\_dephys\_dirichlet\_winding\_minimal}

则对任意整数 \(k\in\ZZ\) 有
\[
z(\sigma,t+kT_{\lambda})=z(\sigma,t),
\qquad\Longrightarrow\qquad
\bigl(|z|,\mathrm{Arg}(z)\bigr)(\sigma,t+kT_{\lambda})=\bigl(|z|,\mathrm{Arg}(z)\bigr)(\sigma,t).
\]
因此对任意集合 \(S\)，若存在映射
\[
(\sigma,t)\longmapsto \bigl(|z|,\mathrm{Arg}(z),\eta(\sigma,t)\bigr)\in (0,1)\times(-\pi,\pi]\times S
\]
为单射，则存在单射 \(\iota:\ZZ\hookrightarrow S\)（即外置轴至少可数无穷）。
此外，绕数证书 \(\kappa_{\lambda}(s)\in\ZZ\)（式 \eqref{eq:dirichlet-winding-certificate}）满足一致性约束
\(
t\log\lambda+\mathrm{Arg}(z)=2\pi\kappa_{\lambda}(s)\in 2\pi\ZZ
\)
（式 \eqref{eq:dirichlet-arg-consistency}），从而给出一条可审计离散外置；与非周期相位轴 \(u_{\mathrm C}(t)\)（式 \eqref{eq:dirichlet-cayley-phase-axis}）合并即得到三纤维注入坐标 \(\mathcal{E}_{\lambda}\)（定义 \ref{def:dirichlet-trifiber-injection}，命题 \ref{prop:dirichlet-trifiber-injection-audit}）。
\end{theorem}

\begin{proof}
固定任意 \(\sigma>0\) 与 \(t_0\in\RR\)。对每个 \(k\in\ZZ\)，点 \((\sigma,t_0+kT_{\lambda})\) 在 \((|z|,\mathrm{Arg}(z))\) 下具有相同像。
若扩展映射为单射，则 \(k\mapsto \eta(\sigma,t_0+kT_{\lambda})\) 必为单射，故 \(S\) 含有一份可数无穷子集，等价于存在 \(\ZZ\hookrightarrow S\) 的单射。
后半部分由定义 \eqref{eq:dirichlet-winding-certificate} 与命题 \ref{prop:dirichlet-trifiber-injection-audit} 直接给出。证毕。
\end{proof}

  \item 任意两条实现若在上述 \(2d\) 点一致，则共享同一 \(\mathrm{HANKELNF}_q\)，从而对所有 \(m\ge 2\) 完全一致（命题 \ref{prop:pom-moment-finite-sampling-decidable}）。
\end{enumerate}
因此窗口偏移 \(\ell_q\) 与相应 Hankel 块即构成“信息不泄漏”的最强有限证书：在固定类型（固定 \(q\)）下，动态指纹被一个有限窗口完全锁定。
\end{theorem}

\begin{proof}
由定义 \ref{def:pom-hankel-normal-form} 与命题 \ref{prop:pom-moment-finite-sampling-decidable} 直接推出。证毕。
\end{proof}

\paragraph{闭合链：缺陷证书—端点压缩—类型拒绝}
定理 \ref{thm:dephys-defect-repulsion-radius} 把“盘内零点是否存在”压缩为单参数标量缺陷 \(D_{\zeta}(\varrho)\) 的可审计半径证书；
定理 \ref{thm:dephys-timefiber-two-ended} 表明 \(\varrho\uparrow 1\) 的边界层必然把高高度信息压缩到端点邻域的窗口贡献；
定理 \ref{thm:dephys-pw-nohair} 则在 \(\mathbf{PW}\) 闭包与 unitary slice 可见域下排除了与相位并列的额外连续自由度。
在此基础上，定理 \ref{thm:dephys-endpoint-koenigs-rigidity} 给出“时间”作为端点吸引半群的唯一 Koenigs 深度（不可外加与相位并列的连续坐标寄存器），定理 \ref{thm:dephys-dirichlet-winding-minimal} 进一步表明虚向折叠的可逆化必然强迫外置可数离散证书；而定理 \ref{thm:dephys-ptypical-primitive-tower}--\ref{thm:dephys-hankel-finite-audit} 则把该离散外置具体化为可审计毛发族：绕数证书 \(\kappa_{\lambda}\)、\((p)\)-典型 primitive 差分塔 \(\{p_{p^k}\}\)、稳定残差谱 \(\{\alpha_{p,r}\}\)（以及 \(p=2\) 的 \(\{\varepsilon_r\}\)）、与 Hankel 偏移证书 \(\ell_q\)（及其正规形 \(\mathrm{HANKELNF}_q\)）。
此外，幂半群 \(w\mapsto w^m\) 在圆盘化接口上对 Jensen 缺陷满足严格共变 \(D_{F^{(m)}}(\varrho)=D_F(\varrho^m)\)（命题 \ref{prop:app-jensen-defect-power-covariance}），并在端点邻域分解为离散残差与精度势账本（命题 \ref{prop:endpoint-orthogonal-decomposition-z4-Rplus} 及命题 \ref{prop:chebyshev-endpoint-derivative-splitting}）；在 \(p\)-典型方向，Dwork 同余的模 \(p^\ell\) 迭代下降进一步表明高层对象的低精度投影仅依赖低层的迭代 Frobenius 推前（命题 \ref{prop:witt-frobenius-iterated-descent}）。

\begin{theorem}[可审计闭环的 RH 证书工程：解析—演算—算术/谱三重证书]\label{thm:dephys-rh-certificate-audit-loop}
在本文固定的“圆盘完成化 \(F_\zeta\) + unitary slice 类型化核验 + 有界切片语义 \(\mathsf{Sem}\)”协议下，RH 的核验任务可被分解为三类互相正交、且可独立离线复核的证书：
\begin{enumerate}
  \item \textbf{解析证书（缺陷链与正性—矩层级链）}：
  在圆盘完成化接口下，“盘内零点禁入（RH）”可被压缩为两条等价但互补的可数证书链：
  \begin{itemize}
    \item \textbf{缺陷/排斥半径链：}
    构造一列 \(\varrho_n\uparrow 1\) 并给出可核验上界 \(D_\zeta(\varrho_n)\le \delta_n\)（\(\delta_n\downarrow 0\)），或等价地证书化
    \(
    r_{\ast}(\varrho_n)=\varrho_n e^{-D_\zeta(\varrho_n)}\uparrow 1
    \)；
    其零点事件在半径方向的“折角/跳变”由对数半径导数律 \( \frac{\dd}{\dd(\log\varrho)}D_\zeta(\varrho)=N_{F_\zeta}(\varrho) \) 精确承载（命题 \ref{prop:dephys-jensen-defect-log-derivative}），并由排斥半径驱动律 \( \frac{\dd}{\dd(\log\varrho)}\log r_\ast(\varrho)=1-N_{F_\zeta}(\varrho) \) 给出可定位的停滞见证（推论 \ref{cor:dephys-repulsion-radius-zero-count-driving}）。
    \item \textbf{正性—Toeplitz--PSD 层级链：}
    令视界 Carath\'eodory 对数导数函数 \(\mathscr{C}_\zeta\) 如附录 \eqref{eq:app-horizon-carath-def} 定义。
    则 RH 等价于 \(\mathscr{C}_\zeta\) 属于 Carath\'eodory 类，并等价于 Toeplitz--PSD 层级
    \(
    \mathbf{T}_N=(c_{j-k})_{0\le j,k\le N}\succeq 0
    \)
    对所有 \(N\ge 0\) 成立（定理 \ref{thm:app-horizon-toeplitz-lmi}）。
    每一级失败都给出有限维否证器（负特征值），并在实结构下可进一步定位到最小 \(N_\star\) 与扇区符号 \(\epsilon\in\{+,-\}\)（推论 \ref{cor:app-horizon-toeplitz-z2-cert-split}）。
    该层级在任一固定 \(N\) 上仅依赖 \(\Xi_\zeta\) 在 \(s=-\tfrac12\) 的有限阶局部导数数据（命题 \ref{prop:app-horizon-coeff-local}），并在 \(|w|<1/3\) 内可借欧拉积补丁把算术项置入一致绝对收敛域以逐项审计（定理 \ref{thm:app-horizon-euler-patch}）。
  \end{itemize}
  由此，两条链在数学上等价，但失败见证互补：前者以缺陷下界/排斥半径停滞为见证，后者以有限维负特征值为见证。
  \item \textbf{演算证书（实现中立性）}：
  在有界切片 \(\mathbf{PW}^{\ast}_{\le}\) 上，以保守语义函子 \(\mathsf{Sem}\)（定义 \ref{def:pom-sem-functor}）固化实现等价。
  当缺陷上界链以 \(\mathsf{Sem}\) 的签名/谱输出为公共接口给出时，定理 \ref{thm:pom-sem-conservative} 保证该链仅依赖自然同构类，从而与实现选择无关（详见定理 \ref{thm:dephys-impl-independent-defect-certificate}）。
  \item \textbf{算术/谱证书（有限指纹与整数护栏）}：
  Hankel-正规形的 \(2d\) 点锁定（定理 \ref{thm:dephys-hankel-finite-audit}）把谱侧核心指纹降为有限窗口复算；
  而 \(p\)-进同余链与端点刚性指纹（命题 \ref{prop:dephys-padic-guardrails-endpoint-rigidity}）提供独立的整数层一致性护栏。
\end{enumerate}
其结果是：在该协议口径下，任何“信息泄漏/实现漂移/端点异常”若要与可见核验结论共存，必须表现为至少一类证书的失败；
从而 RH 的困难可被等价集中为两类可数化的解析任务之一：构造可审计的缺陷上界链（或排斥半径趋一链），或证书化视界对数导数的正性（等价地：证明 Toeplitz--PSD 层级全体成立）；其余证书仅用于隔离实现自由度与整数层一致性护栏，使核验难点不被实现细节稀释。
\end{theorem}

\begin{proof}
解析证书的 soundness：缺陷/排斥半径链由定理 \ref{thm:dephys-defect-repulsion-radius} 给出；而正性—Toeplitz--PSD 层级链由定理 \ref{thm:app-horizon-toeplitz-lmi}（等价于推论 \ref{cor:app-horizon-carath-rh} 与定理 \ref{thm:app-horizon-weyl-herglotz}）给出。
演算证书由定义 \ref{def:pom-sem-functor} 与定理 \ref{thm:pom-sem-conservative} 给出，并在定理 \ref{thm:dephys-impl-independent-defect-certificate} 中形式化；
算术/谱证书分别由定理 \ref{thm:dephys-hankel-finite-audit} 与命题 \ref{prop:dephys-padic-guardrails-endpoint-rigidity} 给出。
三类证书的正交性来自其约束对象分别落在缺陷标量、语义自然同构类与整数/有限窗口指纹三个互不混淆的输出域上（参见 \S\ref{subsec:xi-certificate-coherence-orthogonal-axis-capacity} 的地址三轴口径）。证毕。
\end{proof}

\begin{remark}[端点阻碍的二元正交分解：\texorpdfstring{$\zeta$}{zeta} 型对数主尺度与 \texorpdfstring{$L(\chi_4)$}{L(chi4)} 型反变本征通道]\label{rem:endpoint-obstruction-dichotomy}
在端点 \(w\to-1\) 的符号分解口径下，端点载荷在偶/反变两类观测上呈现为一对严格正交的分量，其来源可归因于投影系统自同构在端点邻域所诱导的有限维不变子表示。
其一为 \(\zeta\)-型对数主尺度：端点三通道权核的 Dirichlet 正则化系数具有闭式 \(\zeta(1+\sigma)\) 符号表示，且其发散结构由 \(\zeta\) 在 \(1\) 处的极点完全控制（命题 \ref{prop:endpoint-dirichlet-symbol-weights}）。
其对应的重整化有限部在 \(\sigma\downarrow 0\) 下具有显式闭式（定理 \ref{thm:endpoint-even-finite-part-renormalization}）。
其二为 \(\chi_4\)-型反变本征通道：在最小反变纤维坐标 \(\delta\) 上，端点反变观测抽出一维不变子表示，其符号为 \(L(1+\sigma,\chi_4)\)，并与上述 \(\zeta\)-型偶通道权核严格解耦（定理 \ref{thm:endpoint-zeta-Lchi4-symbol-pair}）；并且该通道在 \(\sigma\downarrow 0\) 处解析，极限输出为常数 \(L(1,\chi_4)=\pi/4\)（推论 \ref{cor:endpoint-odd-analytic-at-zero}）。
此外，在 primitive 奇长扇区中，端点分支差分证书满足选择律消失，从而端点主刚性退化为纯 \(\zeta\)-型对数账本（注 \ref{rem:endpoint-logscale-parity-decouple}）。
\end{remark}
与有界切片上的语义保守性（定理 \ref{thm:pom-sem-conservative}）合并后，上述共变/下降律把“隐藏分量”约束为可审计的不变量输出域，从而将判等与实现自审计收束为有限可复核指纹。

\begin{theorem}[实现无关的缺陷证书链：由 Jensen 缺陷到 RH]\label{thm:dephys-impl-independent-defect-certificate}
固定一列半径 \(\varrho_n\uparrow 1\)。
在有界切片 \(\mathbf{PW}^{\ast}_{\le}\) 上，设存在一列投影词 \(w_n\) 及一列可离线复核的有限证书对象 \(C_n\)，使得证书输出一个上界 \(\widehat D_n\in\RR_{\ge 0}\) 与误差预算 \(\varepsilon_n\ge 0\) 并满足
\begin{equation}\label{eq:dephys-defect-certificate-soundness}
D_{\zeta}(\varrho_n)\le \widehat D_n+\varepsilon_n\qquad(\forall n).
\end{equation}
若 \(\widehat D_n+\varepsilon_n\to 0\)，则 RH 成立。
此外，若每个 \((\widehat D_n,\varepsilon_n)\) 仅依赖于 \(\mathsf{Sem}(w_n)\) 的有限输出（定义 \ref{def:pom-sem-functor}），则该缺陷证书链在切片内与实现选择无关：对任意 \(w_n'\in\mathbf{PW}^{\ast}_{\le}\) 若 \(\mathsf{Sem}(w_n')=\mathsf{Sem}(w_n)\)，则存在自然同构 \(w_n'\simeq w_n\)（定理 \ref{thm:pom-sem-conservative}），并且 \((\widehat D_n,\varepsilon_n)\) 在该语义意义下保持不变。
\end{theorem}

\begin{proof}
由 \eqref{eq:dephys-defect-certificate-soundness} 与排斥半径定义 \(r_{\ast}(\varrho)=\varrho e^{-D_{\zeta}(\varrho)}\) 得
\[
r_{\ast}(\varrho_n)\ge \varrho_n\exp\!\bigl(-(\widehat D_n+\varepsilon_n)\bigr)\xrightarrow[n\to\infty]{}1,
\]
故由定理 \ref{thm:dephys-defect-repulsion-radius} 推出 RH。
实现中立性部分：若 \((\widehat D_n,\varepsilon_n)\) 仅依赖 \(\mathsf{Sem}(w_n)\)，则 \(\mathsf{Sem}(w_n')=\mathsf{Sem}(w_n)\) 蕴含两者输出一致；并由定理 \ref{thm:pom-sem-conservative} 得 \(w_n'\simeq w_n\)。
在“地址先于取值”的约定下（\S\ref{sec:recursive-addressing}），证书对象的存在性即其可寻址性；从而该上界链不携带实现侧的隐式自由度，
亦不可能通过递归地址化引入新的可见事件（命题 \ref{prop:recursive-addressing-refinement} 与推论 \ref{cor:recursive-addressing-visible-sigma-nonexpansion}）。证毕。
\end{proof}

由此得到一个体系内闭合的可证伪重述：任何需要径向通道解释的偏离，若不经由 residue/逆极限轴或上述离散证书的非局部外置，就必在可见审计中触发 \(\mathsf{NULL}\) 并被协议拒绝；因而，零点审计的可见证书只能沿“缺陷—端点—相位”链路出现，而不会被额外连续坐标寄存器所稀释。相反，一切可逆化的新增自由度只能以离散毛发进入证书域，而不可能表现为与相位解耦的额外连续坐标寄存器。

在此基础上，\S\ref{subsec:xi-certificate-coherence-orthogonal-axis-capacity} 进一步把“可见证书”的一致性提升为证书对象的逆极限同一性，并将离临界线信息的端点投影严格化为二次径向通道与由精度势诱导的容量下界，从而给出“构造离线零点”的可证伪接口表述。
