\begin{corollary}[零点云 Poisson 扫描的二阶协方差张量可观测定律]\label{cor:xi-comoving-horizon-scan-poisson-kl-covariance-tensor}
在定理 \ref{thm:xi-comoving-horizon-scan-fourier-inversion} 的有限缺陷设定下，记
\[
H(x):=\sum_{j=1}^{M} m_j\,h_x(\rho_j),
\qquad
h_x(\rho_j)=\frac{4\delta_j}{(\gamma_j-x)^2+(1+\delta_j)^2},
\]
并定义权重
\[
\omega_j:=\frac{\delta_j}{1+\delta_j},\qquad
s_j:=1+\delta_j,\qquad
M_0:=\sum_{j=1}^{M} m_j\omega_j\in(0,\infty),
\qquad
p_j:=\frac{m_j\omega_j}{M_0}.
\]
令归一化扫描密度
\[
W(x):=\frac{H(x)}{4\pi M_0}.
\]
则：
\begin{enumerate}
  \item \(W\) 具有严格的“异尺度 Cauchy 混合”表示
  \[
  W(x)=\sum_{j=1}^{M} p_j\,\mathsf P_{s_j}(x-\gamma_j),
  \qquad
  \mathsf P_t(x)=\frac{1}{\pi}\frac{t}{x^2+t^2}.
  \]
  \item 对任意 \(t>0\)，其 Poisson--Cauchy 粗粒化 \(W_t:=\mathsf P_t*W\) 满足
  \[
  W_t(x)=\sum_{j=1}^{M} p_j\,\mathsf P_{t+s_j}(x-\gamma_j).
  \]
  \item 令随机向量 \((\Gamma,S)\) 取值于 \(\RR\times(0,\infty)\)，以
  \(
  \PP[(\Gamma,S)=(\gamma_j,s_j)]=p_j
  \)
  给定分布，并记
  \(
  \mu:=\EE[\Gamma]
\)、\(
  \bar s:=\EE[S]
  \)。
  则
  \[
  t^4\,D_{\mathrm{KL}}\!\Bigl(W_t\ \Bigm|\ \mathsf P_{t+\bar s}(\,\cdot-\mu)\Bigr)
  \longrightarrow
  \frac18\Bigl(\bigl(\Var(\Gamma)-\Var(S)\bigr)^2+4\,\Cov(\Gamma,S)^2\Bigr),
  \]
  并且在允许位置--尺度投影修正后（定理 \ref{thm:xi-poisson-cauchy-location-scale-information-projection}）有
  \[
  t^4\,\inf_{\alpha,\beta}
  D_{\mathrm{KL}}\!\Bigl(W_t\ \Bigm|\ \mathsf P_{t+\bar s+\alpha/(t+\bar s)}(\,\cdot-\mu-\beta/(t+\bar s))\Bigr)
  \longrightarrow
  \frac1{16}\Bigl(\bigl(\Var(\Gamma)-\Var(S)\bigr)^2+4\,\Cov(\Gamma,S)^2\Bigr).
  \]
\end{enumerate}
\leanverified{paper\_xi\_comoving\_horizon\_scan\_poisson\_kl\_covariance\_tensor}
\end{corollary}
\begin{proof}
由 \(h_x(\rho_j)=4\delta_j\bigl((\gamma_j-x)^2+(1+\delta_j)^2\bigr)^{-1}\) 与
\(
\mathsf P_s(x-\gamma)=\pi^{-1}s\bigl((x-\gamma)^2+s^2\bigr)^{-1}
\)
得恒等式
\[
h_x(\rho_j)=4\pi\,\frac{\delta_j}{1+\delta_j}\,\mathsf P_{1+\delta_j}(x-\gamma_j)
=4\pi\,\omega_j\,\mathsf P_{s_j}(x-\gamma_j).
\]
对 \(j\) 求和并除以 \(4\pi M_0\) 即得 (1)。
(2) 由半群卷积 \(\mathsf P_t*\mathsf P_{s_j}=\mathsf P_{t+s_j}\) 直接推出。
最后把 \(W_t\) 写为
\(
W_t(x)=\EE[\mathsf P_{t+S}(x-\Gamma)]
\)
并应用定理 \ref{thm:xi-poisson-cauchy-location-scale-kl-sharp-m2} 与
\ref{thm:xi-poisson-cauchy-location-scale-information-projection} 即得 (3)。
证毕。
\end{proof}
