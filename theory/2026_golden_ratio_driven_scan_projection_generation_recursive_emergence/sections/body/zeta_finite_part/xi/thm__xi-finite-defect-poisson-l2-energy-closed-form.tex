\begin{theorem}[жңүйҷҗзјәйҷ·жү«жҸҸеҠҝзҡ„ Poisson--\texorpdfstring{$L^2$}{L2} дә’дҪңз”ЁиғҪй—ӯејҸ]\label{thm:xi-finite-defect-poisson-l2-energy-closed-form}
д»Ө
\[
\nu=\sum_{j=1}^{\kappa}m_j\,\delta_{(\gamma_j,\delta_j)},
\qquad
m_j\in\NN,\ \gamma_j\in\RR,\ \delta_j\in(0,1),
\]
е№¶д»Ө \(u_\delta\)гҖҒ\(J_\nu\) еҸҠ Poisson ж ё \(P_s\) еҰӮе®ҡзҗҶ
\ref{thm:xi-cayley-modulus-poisson-fourier-fingerprint-layered} дёҺжҺЁи®ә
\ref{cor:xi-cayley-depth-scan-area-lowfreq}гҖӮ
д»Ҙ Fourier д№ҳеӯҗ \(|\xi|\) е®ҡд№үз®—еӯҗ \(|D|\)пјҢе№¶еҜ№ \(t\ge 0\) е®ҡд№үиғҪйҮҸ
\[
\mathcal E_\nu(t):=\bigl\|P_t*(|D|J_\nu)\bigr\|_{L^2(\RR)}^2.
\]
еҲҷжңүе®Ңе…ЁжҳҫејҸиЎЁзӨә
\begin{equation}\label{eq:xi-finite-defect-poisson-l2-energy-closed-form}
\boxed{\;
\mathcal E_\nu(t)
=
\pi\sum_{j,k=1}^{\kappa}m_jm_k
\sum_{\sigma,\tau\in\{-1,+1\}} s_\sigma s_\tau\,
\frac{A_{jk}^{\sigma\tau}(t)}{\bigl(A_{jk}^{\sigma\tau}(t)\bigr)^2+(\gamma_j-\gamma_k)^2}\; }
\end{equation}
е…¶дёӯ \(s_{-1}:=+1,\ s_{+1}:=-1\)пјҢе№¶дё”
\[
A_{jk}^{\sigma\tau}(t):=(t+1+\sigma\delta_j)+(t+1+\tau\delta_k)=2t+2+\sigma\delta_j+\tau\delta_k.
\]
\end{theorem}
\begin{proof}
з”ұе‘Ҫйўҳ \ref{prop:xi-jensen-poisson-intertwiner-two-scale}пјҢ
\[
P_t*(|D|J_\nu)
=\pi\sum_{j=1}^{\kappa}m_j\Bigl(P_{t+1-\delta_j}-P_{t+1+\delta_j}\Bigr)(\cdot-\gamma_j).
\]
еӣ жӯӨ
\[
\mathcal E_\nu(t)
=\pi^2\sum_{j,k}m_jm_k
\sum_{\sigma,\tau\in\{-1,+1\}} s_\sigma s_\tau\,
\Bigl\langle P_{t+1+\sigma\delta_j}(\cdot-\gamma_j),\,P_{t+1+\tau\delta_k}(\cdot-\gamma_k)\Bigr\rangle,
\]
е…¶дёӯзәҰе®ҡ \(t+1+(-1)\delta=t+1-\delta\)пјҢ\(t+1+(+1)\delta=t+1+\delta\)гҖӮ
\par
з”ұ \(\widehat{P_s}(\xi)=e^{-s|\xi|}\) дёҺ PlancherelпјҢ
\begin{align*}
\Bigl\langle P_a(\cdot-\gamma_j),P_b(\cdot-\gamma_k)\Bigr\rangle
&=\frac{1}{2\pi}\int_{\RR}e^{-(a+b)|\xi|}e^{-\mathrm{i}(\gamma_j-\gamma_k)\xi}\,\dd\xi\\
&=\frac{1}{\pi}\frac{a+b}{(a+b)^2+(\gamma_j-\gamma_k)^2}.
\end{align*}
еҸ– \(a=t+1+\sigma\delta_j\)гҖҒ\(b=t+1+\tau\delta_k\) е№¶д»ЈеӣһпјҢеҚіеҫ—
\eqref{eq:xi-finite-defect-poisson-l2-energy-closed-form}гҖӮиҜҒжҜ•гҖӮ
\end{proof}

\begin{corollary}[еҚ•зјәйҷ·жғ…еҪўзҡ„й—ӯејҸиғҪйҮҸдёҺ \texorpdfstring{$(t+1)^{-3}$}{(t+1)^-3} иЎ°еҮҸ]\label{cor:xi-finite-defect-poisson-l2-energy-single-defect}
еңЁе®ҡзҗҶ \ref{thm:xi-finite-defect-poisson-l2-energy-closed-form} дёӯеҸ– \(\kappa=1\)пјҢи®°еҸӮж•°дёә \((\gamma,\delta,m)\)гҖӮ
еҲҷеҜ№д»»ж„Ҹ \(t\ge 0\) жңү
\[
\mathcal E_\nu(t)
=\pi m^2\bigl\|P_{t+1-\delta}-P_{t+1+\delta}\bigr\|_{L^2(\RR)}^2
=\pi m^2\frac{\delta^2}{(t+1)\bigl((t+1)^2-\delta^2\bigr)}.
\]
зү№еҲ«ең°
\[
\mathcal E_\nu(0)=\pi m^2\frac{\delta^2}{1-\delta^2},
\qquad
\lim_{t\to\infty}(t+1)^3\mathcal E_\nu(t)=\pi(m\delta)^2.
\]
\leanverified{singleDefectEnergy\_rational\_identity}
\leanverified{singleDefectEnergy\_zero}
\end{corollary}
\begin{proof}
з”ұе‘Ҫйўҳ \ref{prop:xi-jensen-poisson-intertwiner-two-scale}пјҢ
\(
P_t*(|D|J_\nu)=\pi m(P_{t+1-\delta}-P_{t+1+\delta})(\cdot-\gamma)
\)пјӣ
е№із§»дёҚж”№еҸҳ \(L^2\) иҢғж•°пјҢж•…
\(\mathcal E_\nu(t)=\pi^2m^2\|P_{t+1-\delta}-P_{t+1+\delta}\|_2^2\)гҖӮ
еҶҚз”ұеҶ…з§Ҝе…¬ејҸ
\[
\langle P_a,P_b\rangle=\frac{1}{\pi}\frac{a+b}{(a+b)^2}
=\frac{1}{\pi}\frac{1}{a+b}\qquad(a,b>0)
\]
пјҲеҸ– \(\gamma_j=\gamma_k\) зҡ„зү№дҫӢпјүеұ•ејҖеҫ—
\[
\|P_{t+1-\delta}-P_{t+1+\delta}\|_2^2
=\frac{1}{\pi}\left(\frac{1}{2(t+1-\delta)}+\frac{1}{2(t+1+\delta)}-\frac{1}{t+1}\right)
=\frac{1}{\pi}\frac{\delta^2}{(t+1)\bigl((t+1)^2-\delta^2\bigr)}.
\]
д»ЈеӣһеҚіеҫ—й—ӯејҸпјӣжһҒйҷҗз”ұиҜҘй—ӯејҸзӣҙжҺҘиҜ»еҮәгҖӮиҜҒжҜ•гҖӮ
\end{proof}

\begin{corollary}[иғҪе°ҫзҡ„еҠ жқғжҖ»еҒҸз§»иҙЁйҮҸ]\label{cor:xi-finite-defect-poisson-l2-energy-tail-mass}
еңЁе®ҡзҗҶ \ref{thm:xi-finite-defect-poisson-l2-energy-closed-form} зҡ„и®ҫе®ҡдёӢпјҢе®ҡд№ү
\[
\Phi(\nu):=\sum_{j=1}^{\kappa}m_j\delta_j.
\]
еҲҷеӯҳеңЁеёёж•° \(C(\nu)\) дҪҝеҫ—
\[
\mathcal E_\nu(t)=\frac{\pi\,\Phi(\nu)^2}{(t+1)^3}+O_\nu\bigl((t+1)^{-4}\bigr)
\qquad(t\to\infty),
\]
д»ҺиҖҢ
\[
\lim_{t\to\infty}(t+1)^3\mathcal E_\nu(t)=\pi\,\Phi(\nu)^2.
\]
е№¶дё” \(\Phi(\nu)\) еҸҜз”ұжү«жҸҸеҠҝйқўз§ҜеҫӢиҜ»еҮәпјҡ
\[
\int_{\RR}J_\nu(x)\,\dd x=2\pi\,\Phi(\nu).
\]
\leanverified{paper\_xi\_finite\_defect\_poisson\_l2\_energy\_tail\_mass\_seeds\_wrap}
\end{corollary}
\begin{proof}
з”ұ \eqref{eq:xi-finite-defect-poisson-l2-energy-closed-form}пјҢ
д»Ө \(A:=2t+2\) е№¶еӣәе®ҡ \(j,k\)пјҢи®°
\(
f_{jk}(x)=\frac{x}{x^2+(\gamma_j-\gamma_k)^2}
\)гҖӮ
еҪ“ \(A\to\infty\) ж—¶жңүдёҖиҮҙеұ•ејҖ \(f_{jk}(A+\varepsilon)=\frac{1}{A+\varepsilon}+O(A^{-3})\)пјҲеҜ№еӣәе®ҡ \(\varepsilon\) дёҖиҮҙпјүгҖӮ
еҜ№еӣӣйЎ№дәӨжӣҝз»„еҗҲ
\[
f_{jk}(A-\delta_j-\delta_k)-f_{jk}(A-\delta_j+\delta_k)-f_{jk}(A+\delta_j-\delta_k)+f_{jk}(A+\delta_j+\delta_k)
\]
дҪңдәҢйҳ¶е·®еҲҶеұ•ејҖеҸҜеҫ—
\(
\frac{8\delta_j\delta_k}{A^3}+O(A^{-4})
\)пјӣ
д»Јеӣһе№¶еҜ№ \(j,k\) жұӮе’ҢпјҢеҫ—еҲ°
\[
\mathcal E_\nu(t)=\pi\sum_{j,k}m_jm_k\left(\frac{8\delta_j\delta_k}{(2t+2)^3}+O((t+1)^{-4})\right)
=\frac{\pi(\sum_j m_j\delta_j)^2}{(t+1)^3}+O_\nu((t+1)^{-4}).
\]
йқўз§ҜеҫӢз”ұжҺЁи®ә \ref{cor:xi-cayley-depth-scan-area-lowfreq} з»ҷеҮәгҖӮиҜҒжҜ•гҖӮ
\end{proof}

\begin{proposition}[\texorpdfstring{$L^2$}{L2} иғҪйҮҸйӣ¶еҲҡжҖ§]\label{prop:xi-finite-defect-poisson-l2-energy-zero-rigidity}
еңЁе®ҡзҗҶ \ref{thm:xi-finite-defect-poisson-l2-energy-closed-form} зҡ„жңүйҷҗзјәйҷ·и®ҫе®ҡдёӢпјҢ
\[
\mathcal E_\nu(0)=0
\quad\Longleftrightarrow\quad
\nu\equiv 0
\quad\Longleftrightarrow\quad
J_\nu\equiv 0.
\]
\leanverified{singleDefectEnergy\_nonneg\_num}
\leanverified{singleDefectEnergy\_zero\_rigidity\_single}
\leanverified{singleDefectEnergy\_zero\_eq\_zero\_iff}
\end{proposition}
\begin{proof}
иӢҘ \(\nu\equiv 0\) еҲҷ \(J_\nu\equiv 0\) дё” \(\mathcal E_\nu(0)=0\) жҳҫз„¶гҖӮ
\par
еҸҚд№ӢиӢҘ \(\mathcal E_\nu(0)=\||D|J_\nu\|_{L^2(\RR)}^2=0\)пјҢеҲҷ \(|D|J_\nu=0\)пјҲ\(L^2\) ж„Ҹд№үдёӢпјүпјҢд»ҺиҖҢ
\(|\xi|\,\widehat{J_\nu}(\xi)=0\) еҮ д№ҺеӨ„еӨ„жҲҗз«ӢгҖӮ
ж•… \(\widehat{J_\nu}\) ж”Ҝж’‘дәҺ \(\{\xi=0\}\)пјҢеҚі \(J_\nu\) дёәеёёж•°еҲҶеёғгҖӮ
иҖҢз”ұжҺЁи®ә \ref{cor:xi-cayley-depth-scan-area-lowfreq} жңү \(J_\nu\in L^1(\RR)\)пјҢеӣ жӯӨиҜҘеёёж•°еҸӘиғҪдёә \(0\)пјҢеҚі \(J_\nu\equiv 0\)гҖӮ
\par
жңҖеҗҺпјҢ\(J_\nu\) зҡ„ Fourier--Laplace жҢҮзә№дёәжңүйҷҗжҢҮж•°е’ҢпјҲжҺЁи®ә \ref{cor:xi-cayley-depth-scan-area-lowfreq} зҡ„жҳҫејҸ \(\widehat{J_\nu}\) е…¬ејҸпјүпјҢ
е…¶жҢҮж•°еӨҡйҮҚйӣҶдёҺзі»ж•°е”ҜдёҖзЎ®е®ҡ \(\{(\gamma_j,\delta_j,m_j)\}\)пјӣ
ж•… \(J_\nu\equiv 0\) ејәиҝ« \(\nu\equiv 0\)гҖӮиҜҒжҜ•гҖӮ
\leanverified{paper\_singleDefectEnergy\_zero\_rigidity\_package}
\end{proof}
