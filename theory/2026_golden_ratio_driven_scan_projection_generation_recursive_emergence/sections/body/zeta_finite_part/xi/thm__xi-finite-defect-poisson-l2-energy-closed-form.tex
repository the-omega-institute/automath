\begin{theorem}[有限缺陷扫描势的 Poisson--\texorpdfstring{$L^2$}{L2} 互作用能闭式]\label{thm:xi-finite-defect-poisson-l2-energy-closed-form}
令
\[
\nu=\sum_{j=1}^{\kappa}m_j\,\delta_{(\gamma_j,\delta_j)},
\qquad
m_j\in\NN,\ \gamma_j\in\RR,\ \delta_j\in(0,1),
\]
并令 \(u_\delta\)、\(J_\nu\) 及 Poisson 核 \(P_s\) 如定理
\ref{thm:xi-cayley-modulus-poisson-fourier-fingerprint-layered} 与推论
\ref{cor:xi-cayley-depth-scan-area-lowfreq}。
以 Fourier 乘子 \(|\xi|\) 定义算子 \(|D|\)，并对 \(t\ge 0\) 定义能量
\[
\mathcal E_\nu(t):=\bigl\|P_t*(|D|J_\nu)\bigr\|_{L^2(\RR)}^2.
\]
则有完全显式表示
\begin{equation}\label{eq:xi-finite-defect-poisson-l2-energy-closed-form}
\boxed{\;
\mathcal E_\nu(t)
=
\pi\sum_{j,k=1}^{\kappa}m_jm_k
\sum_{\sigma,\tau\in\{-1,+1\}} s_\sigma s_\tau\,
\frac{A_{jk}^{\sigma\tau}(t)}{\bigl(A_{jk}^{\sigma\tau}(t)\bigr)^2+(\gamma_j-\gamma_k)^2}\; }
\end{equation}
其中 \(s_{-1}:=+1,\ s_{+1}:=-1\)，并且
\[
A_{jk}^{\sigma\tau}(t):=(t+1+\sigma\delta_j)+(t+1+\tau\delta_k)=2t+2+\sigma\delta_j+\tau\delta_k.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:xi-jensen-poisson-intertwiner-two-scale}，
\[
P_t*(|D|J_\nu)
=\pi\sum_{j=1}^{\kappa}m_j\Bigl(P_{t+1-\delta_j}-P_{t+1+\delta_j}\Bigr)(\cdot-\gamma_j).
\]
因此
\[
\mathcal E_\nu(t)
=\pi^2\sum_{j,k}m_jm_k
\sum_{\sigma,\tau\in\{-1,+1\}} s_\sigma s_\tau\,
\Bigl\langle P_{t+1+\sigma\delta_j}(\cdot-\gamma_j),\,P_{t+1+\tau\delta_k}(\cdot-\gamma_k)\Bigr\rangle,
\]
其中约定 \(t+1+(-1)\delta=t+1-\delta\)，\(t+1+(+1)\delta=t+1+\delta\)。
\par
由 \(\widehat{P_s}(\xi)=e^{-s|\xi|}\) 与 Plancherel，
\begin{align*}
\Bigl\langle P_a(\cdot-\gamma_j),P_b(\cdot-\gamma_k)\Bigr\rangle
&=\frac{1}{2\pi}\int_{\RR}e^{-(a+b)|\xi|}e^{-\mathrm{i}(\gamma_j-\gamma_k)\xi}\,\dd\xi\\
&=\frac{1}{\pi}\frac{a+b}{(a+b)^2+(\gamma_j-\gamma_k)^2}.
\end{align*}
取 \(a=t+1+\sigma\delta_j\)、\(b=t+1+\tau\delta_k\) 并代回，即得
\eqref{eq:xi-finite-defect-poisson-l2-energy-closed-form}。证毕。
\end{proof}
\leanverified{paper\_xi\_finite\_defect\_poisson\_l2\_energy\_closed\_form}

\begin{corollary}[单缺陷情形的闭式能量与 \texorpdfstring{$(t+1)^{-3}$}{(t+1)^-3} 衰减]\label{cor:xi-finite-defect-poisson-l2-energy-single-defect}
在定理 \ref{thm:xi-finite-defect-poisson-l2-energy-closed-form} 中取 \(\kappa=1\)，记参数为 \((\gamma,\delta,m)\)。
则对任意 \(t\ge 0\) 有
\[
\mathcal E_\nu(t)
=\pi m^2\bigl\|P_{t+1-\delta}-P_{t+1+\delta}\bigr\|_{L^2(\RR)}^2
=\pi m^2\frac{\delta^2}{(t+1)\bigl((t+1)^2-\delta^2\bigr)}.
\]
特别地
\[
\mathcal E_\nu(0)=\pi m^2\frac{\delta^2}{1-\delta^2},
\qquad
\lim_{t\to\infty}(t+1)^3\mathcal E_\nu(t)=\pi(m\delta)^2.
\]
\leanverified{singleDefectEnergy\_rational\_identity}
\leanverified{singleDefectEnergy\_zero}
\leanverified{paper\_xi\_finite\_defect\_poisson\_l2\_energy\_single\_defect}
\end{corollary}
\begin{proof}
由命题 \ref{prop:xi-jensen-poisson-intertwiner-two-scale}，
\(
P_t*(|D|J_\nu)=\pi m(P_{t+1-\delta}-P_{t+1+\delta})(\cdot-\gamma)
\)；
平移不改变 \(L^2\) 范数，故
\(\mathcal E_\nu(t)=\pi^2m^2\|P_{t+1-\delta}-P_{t+1+\delta}\|_2^2\)。
再由内积公式
\[
\langle P_a,P_b\rangle=\frac{1}{\pi}\frac{a+b}{(a+b)^2}
=\frac{1}{\pi}\frac{1}{a+b}\qquad(a,b>0)
\]
（取 \(\gamma_j=\gamma_k\) 的特例）展开得
\[
\|P_{t+1-\delta}-P_{t+1+\delta}\|_2^2
=\frac{1}{\pi}\left(\frac{1}{2(t+1-\delta)}+\frac{1}{2(t+1+\delta)}-\frac{1}{t+1}\right)
=\frac{1}{\pi}\frac{\delta^2}{(t+1)\bigl((t+1)^2-\delta^2\bigr)}.
\]
代回即得闭式；极限由该闭式直接读出。证毕。
\end{proof}

\begin{corollary}[能尾的加权总偏移质量]\label{cor:xi-finite-defect-poisson-l2-energy-tail-mass}
在定理 \ref{thm:xi-finite-defect-poisson-l2-energy-closed-form} 的设定下，定义
\[
\Phi(\nu):=\sum_{j=1}^{\kappa}m_j\delta_j.
\]
则存在常数 \(C(\nu)\) 使得
\[
\mathcal E_\nu(t)=\frac{\pi\,\Phi(\nu)^2}{(t+1)^3}+O_\nu\bigl((t+1)^{-4}\bigr)
\qquad(t\to\infty),
\]
从而
\[
\lim_{t\to\infty}(t+1)^3\mathcal E_\nu(t)=\pi\,\Phi(\nu)^2.
\]
并且 \(\Phi(\nu)\) 可由扫描势面积律读出：
\[
\int_{\RR}J_\nu(x)\,\dd x=2\pi\,\Phi(\nu).
\]
\leanverified{paper\_xi\_finite\_defect\_poisson\_l2\_energy\_tail\_mass}
\leanverified{paper\_xi\_finite\_defect\_poisson\_l2\_energy\_tail\_mass\_seeds\_wrap}
\end{corollary}
\begin{proof}
由 \eqref{eq:xi-finite-defect-poisson-l2-energy-closed-form}，
令 \(A:=2t+2\) 并固定 \(j,k\)，记
\(
f_{jk}(x)=\frac{x}{x^2+(\gamma_j-\gamma_k)^2}
\)。
当 \(A\to\infty\) 时有一致展开 \(f_{jk}(A+\varepsilon)=\frac{1}{A+\varepsilon}+O(A^{-3})\)（对固定 \(\varepsilon\) 一致）。
对四项交替组合
\[
f_{jk}(A-\delta_j-\delta_k)-f_{jk}(A-\delta_j+\delta_k)-f_{jk}(A+\delta_j-\delta_k)+f_{jk}(A+\delta_j+\delta_k)
\]
作二阶差分展开可得
\(
\frac{8\delta_j\delta_k}{A^3}+O(A^{-4})
\)；
代回并对 \(j,k\) 求和，得到
\[
\mathcal E_\nu(t)=\pi\sum_{j,k}m_jm_k\left(\frac{8\delta_j\delta_k}{(2t+2)^3}+O((t+1)^{-4})\right)
=\frac{\pi(\sum_j m_j\delta_j)^2}{(t+1)^3}+O_\nu((t+1)^{-4}).
\]
面积律由推论 \ref{cor:xi-cayley-depth-scan-area-lowfreq} 给出。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$L^2$}{L2} 能量零刚性]\label{prop:xi-finite-defect-poisson-l2-energy-zero-rigidity}
在定理 \ref{thm:xi-finite-defect-poisson-l2-energy-closed-form} 的有限缺陷设定下，
\[
\mathcal E_\nu(0)=0
\quad\Longleftrightarrow\quad
\nu\equiv 0
\quad\Longleftrightarrow\quad
J_\nu\equiv 0.
\]
\leanverified{singleDefectEnergy\_nonneg\_num}
\leanverified{singleDefectEnergy\_zero\_rigidity\_single}
\leanverified{singleDefectEnergy\_zero\_eq\_zero\_iff}
\leanverified{paper\_xi\_finite\_defect\_poisson\_l2\_energy\_zero\_rigidity}
\end{proposition}
\begin{proof}
若 \(\nu\equiv 0\) 则 \(J_\nu\equiv 0\) 且 \(\mathcal E_\nu(0)=0\) 显然。
\par
反之若 \(\mathcal E_\nu(0)=\||D|J_\nu\|_{L^2(\RR)}^2=0\)，则 \(|D|J_\nu=0\)（\(L^2\) 意义下），从而
\(|\xi|\,\widehat{J_\nu}(\xi)=0\) 几乎处处成立。
故 \(\widehat{J_\nu}\) 支撑于 \(\{\xi=0\}\)，即 \(J_\nu\) 为常数分布。
而由推论 \ref{cor:xi-cayley-depth-scan-area-lowfreq} 有 \(J_\nu\in L^1(\RR)\)，因此该常数只能为 \(0\)，即 \(J_\nu\equiv 0\)。
\par
最后，\(J_\nu\) 的 Fourier--Laplace 指纹为有限指数和（推论 \ref{cor:xi-cayley-depth-scan-area-lowfreq} 的显式 \(\widehat{J_\nu}\) 公式），
其指数多重集与系数唯一确定 \(\{(\gamma_j,\delta_j,m_j)\}\)；
故 \(J_\nu\equiv 0\) 强迫 \(\nu\equiv 0\)。证毕。
\leanverified{paper\_singleDefectEnergy\_zero\_rigidity\_package}
\end{proof}
