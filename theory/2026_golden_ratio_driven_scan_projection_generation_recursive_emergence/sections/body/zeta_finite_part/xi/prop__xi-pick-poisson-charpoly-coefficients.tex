\begin{proposition}[Pick--Poisson 主子式的伪双曲配分函数表达]\label{prop:xi-pick-poisson-principal-minors-partition}
\leanverified{paper\_xi\_pick\_poisson\_principal\_minors\_partition}
沿用命题 \ref{prop:xi-toeplitz-negative-spectrum-pick-poisson-gram} 的记号，并记端点 Poisson 权重
\[
p_j:=\frac{1-|a_j|^2}{|1+a_j|^2},
\qquad
\rho(a,b):=\left|\frac{a-b}{1-\overline b a}\right|\qquad(a,b\in\mathbb D).
\]
对任意指标子集 \(I\subset[\kappa]:=\{1,\dots,\kappa\}\)，记 \(\mathsf P_I\) 为 \(\mathsf P\) 的主子矩阵，并定义权重
\[
W(I):=\Bigl(\prod_{j\in I}p_j\Bigr)\Bigl(\prod_{j<k\in I}\rho(a_j,a_k)^2\Bigr),
\]
约定空积为 \(1\)（从而 \(W(\varnothing)=1\)）。则对每个 \(I\) 成立
\[
\boxed{\;\det(\mathsf P_I)=W(I).\;}
\]
\end{proposition}
\begin{proof}
命题 \ref{prop:xi-pick-poisson-det-factorization} 的证明把 \(\mathsf P\) 写作 \(DGD^\ast\)，其中 \(D\) 为对角矩阵且 \(G_{jk}=(1-\overline a_j a_k)^{-1}\)。
对任意子集 \(I\)，主子矩阵同样满足 \(\mathsf P_I=D_I G_I D_I^\ast\)，并可对 \(G_I\) 应用同一 Cauchy 行列式公式得到
\(
\det G_I=\bigl(\prod_{j\in I}(1-|a_j|^2)^{-1}\bigr)\prod_{j<k\in I}\rho(a_j,a_k)^2
\)。
与 \(\det D_I=\prod_{j\in I}\frac{1-|a_j|^2}{1+\overline a_j}\) 合并即可得到所示主子式闭式。证毕。
\end{proof}

\begin{proposition}[逆矩阵主子式的互补闭式与显式对角公式]\label{prop:xi-pick-poisson-inverse-principal-minors-jacobi}
\leanverified{paper\_xi\_pick\_poisson\_inverse\_principal\_minors\_jacobi}
在命题 \ref{prop:xi-pick-poisson-principal-minors-partition} 的记号下，进一步假设 \(\{a_j\}\) 两两不同，从而 \(\mathsf P\succ 0\) 且可逆。
对任意 \(I\subset[\kappa]\) 记补集 \(I^c:=[\kappa]\setminus I\)，则 Jacobi 互补主子式恒等式给出
\[
\boxed{\;
\det\!\bigl((\mathsf P^{-1})_I\bigr)=\frac{\det(\mathsf P_{I^c})}{\det(\mathsf P)}
=\frac{W(I^c)}{W([\kappa])}\; }.
\]
特别地，对任意 \(i\in[\kappa]\) 有显式对角公式
\[
\boxed{\;
(\mathsf P^{-1})_{ii}
=\frac{\det(\mathsf P_{[\kappa]\setminus\{i\}})}{\det(\mathsf P)}
=\frac{1}{p_i\prod_{j\ne i}\rho(a_i,a_j)^2}\; } ,
\]
从而
\[
\boxed{\;
\mathrm{tr}(\mathsf P^{-1})
=\sum_{i=1}^{\kappa}\frac{1}{p_i\prod_{j\ne i}\rho(a_i,a_j)^2}\; } .
\]
\end{proposition}
\begin{proof}
由于 \(\mathsf P\) 可逆，对主子矩阵应用 Jacobi 互补主子式恒等式得到
\(
\det((\mathsf P^{-1})_I)=\det(\mathsf P_{I^c})/\det(\mathsf P)
\)。
再由命题 \ref{prop:xi-pick-poisson-principal-minors-partition} 的闭式 \(\det(\mathsf P_J)=W(J)\) 代入即得第一式。
取 \(I=\{i\}\) 得对角公式；对 \(i\) 求和得到迹公式。证毕。
\end{proof}

\begin{proposition}[主子式气体的配分函数行列式表示]\label{prop:xi-pick-poisson-partition-function-determinant}
沿用命题 \ref{prop:xi-pick-poisson-principal-minors-partition} 的记号。
对任意 \(z\in\CC\) 定义
\[
\mathcal{Z}(z):=\det(I_\kappa+z\mathsf P).
\]
则有精确的子集展开
\[
\boxed{\;
\mathcal{Z}(z)=\sum_{I\subseteq[\kappa]} z^{|I|}\,W(I)
=\sum_{r=0}^{\kappa} z^{r}Z_r\; } ,
\]
其中 \(Z_r\) 如命题 \ref{prop:xi-pick-poisson-charpoly-coefficients} 所定义。
\end{proposition}
\begin{proof}
行列式的主子式展开给出
\(
\det(I_\kappa+z\mathsf P)=\sum_{I\subseteq[\kappa]} z^{|I|}\det(\mathsf P_I)
\)。
由命题 \ref{prop:xi-pick-poisson-principal-minors-partition} 代入 \(\det(\mathsf P_I)=W(I)\) 得第一式。
按子集大小分组，且 \(Z_r=\sum_{|I|=r}W(I)\)，得到第二式。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\det\mathsf P$}{det P} 的双曲合流 Vandermonde 律]\label{prop:xi-pick-poisson-confluent-vandermonde}
取 \(\kappa\ge2\) 与 \(a_\ast\in\mathbb{D}\setminus\{-1\}\)。
设存在互异复数 \(t_1,\dots,t_\kappa\) 与小参数 \(\varepsilon\downarrow0\)，使得
\[
a_j(\varepsilon)=a_\ast+\varepsilon t_j+O(\varepsilon^2)\qquad (1\le j\le\kappa),
\]
并令 \(\mathsf P(\varepsilon)\) 为由 \(\{a_j(\varepsilon)\}\) 构造的 Pick--Poisson Gram 矩阵。
则当 \(\varepsilon\to0\) 时有合流渐近
\[
\boxed{\;
\det\mathsf P(\varepsilon)
=\varepsilon^{\kappa(\kappa-1)}\,
\frac{(1-|a_\ast|^2)^{\kappa(2-\kappa)}}{|1+a_\ast|^{2\kappa}}\,
\Biggl(\prod_{1\le j<k\le\kappa}|t_j-t_k|^2\Biggr)\,
\bigl(1+O(\varepsilon)\bigr)\; } .
\]
\end{proposition}
\begin{proof}
由命题 \ref{prop:xi-pick-poisson-det-factorization}，
\(
\det\mathsf P=\prod_{j=1}^{\kappa}p_j\prod_{1\le j<k\le\kappa}\rho(a_j,a_k)^2
\)。
记 \(p(a)=\frac{1-|a|^2}{|1+a|^2}\)，则 \(p(a_j(\varepsilon))=p(a_\ast)+O(\varepsilon)\)，从而
\[
\prod_{j=1}^{\kappa}p(a_j(\varepsilon))=p(a_\ast)^\kappa\,(1+O(\varepsilon)).
\]
对 \(j\ne k\)，由 \(a_j-a_k=\varepsilon(t_j-t_k)+O(\varepsilon^2)\) 得
\[
|a_j(\varepsilon)-a_k(\varepsilon)|=\varepsilon|t_j-t_k|\,(1+O(\varepsilon)),
\]
并且
\(
1-\overline{a_k(\varepsilon)}a_j(\varepsilon)=1-|a_\ast|^2+O(\varepsilon)
\)，
故
\[
\rho(a_j(\varepsilon),a_k(\varepsilon))
=\frac{\varepsilon|t_j-t_k|}{1-|a_\ast|^2}\,(1+O(\varepsilon)).
\]
将上述代回乘积结构并整理 \(\varepsilon\) 的幂次与常数项，即得所示渐近。证毕。
\end{proof}

\begin{proposition}[Pick--Poisson 特征多项式系数的显式化]\label{prop:xi-pick-poisson-charpoly-coefficients}
\leanverified{paper\_xi\_pick\_poisson\_charpoly\_coefficients}
沿用上式记号，并对 \(0\le r\le\kappa\) 定义
\[
Z_r:=\sum_{\substack{I\subset[\kappa]\\|I|=r}}W(I),\qquad Z_0:=1.
\]
则 \(\mathsf P\) 的特征多项式满足
\[
\boxed{\;
\chi_{\mathsf P}(\lambda):=\det(\lambda I_\kappa-\mathsf P)
=\sum_{r=0}^{\kappa}(-1)^r Z_r\,\lambda^{\kappa-r}.\;}
\]
\end{proposition}
\begin{proof}
线性代数恒等式表明：Hermitian 矩阵的特征多项式中 \(\lambda^{\kappa-r}\) 的系数等于所有 \(r\times r\) 主子式之和，即
\(
e_r(\mathsf P)=\sum_{|I|=r}\det(\mathsf P_I)
\)。
由命题 \ref{prop:xi-pick-poisson-principal-minors-partition}，\(\det(\mathsf P_I)=W(I)\)，故 \(e_r(\mathsf P)=Z_r\)，从而得到所述系数展开。证毕。
\end{proof}

\begin{proposition}[Toeplitz 负谱的全系数极限：基本对称多项式的收敛]\label{prop:xi-toeplitz-negative-spectrum-coefficient-limit}
沿用命题 \ref{prop:xi-toeplitz-negative-spectrum-pick-poisson-gram} 的设定，并令
\(\lambda_1^-(\mathbf{T}_N),\dots,\lambda_\kappa^-(\mathbf{T}_N)<0\)
为 \(\mathbf{T}_N\) 的 \(\kappa\) 个负特征值（按重数计，\(N\gg 1\) 时存在）。
则对每个 \(0\le r\le\kappa\) 成立极限
\[
\boxed{\;
\lim_{N\to\infty}e_r\!\bigl(-\lambda_1^-(\mathbf{T}_N),\dots,-\lambda_\kappa^-(\mathbf{T}_N)\bigr)=Z_r.\;}
\]
等价地，负谱特征多项式在系数意义下收敛到 Pick--Poisson 特征多项式：
\[
\boxed{\;
\prod_{j=1}^{\kappa}\bigl(\lambda+\lambda_j^-(\mathbf{T}_N)\bigr)\ \longrightarrow\ \chi_{\mathsf P}(\lambda)\qquad(N\to\infty).\;}
\]
\end{proposition}
\begin{proof}
命题 \ref{prop:xi-toeplitz-negative-spectrum-pick-poisson-gram} 给出算子范数收敛
\(
A_N:=R_N^\ast(\mathbf{T}_N|_{\mathcal K_N})R_N\to-\mathsf P
\)。
由 Weyl 不等式，\(A_N\) 的特征值（按重数计）逐点收敛到 \(-\mathsf P\) 的特征值。
另一方面，\(\mathbf{T}_N|_{\mathcal K_N}\) 与 \(A_N\) 合同，故其谱与 \(A_N\) 相同；而对 \(N\gg 1\) 时 \(\mathcal K_N\) 捕获了全部负方向（定理 \ref{thm:xi-toeplitz-negative-squares-equals-blaschke-defect}），从而 \(-\lambda_j^-(\mathbf{T}_N)\) 与 \(\mathsf P\) 的特征值对应并收敛。
基本对称多项式 \(e_r\) 为根的连续对称函数，故可将极限逐项传递并得到
\(
e_r(-\lambda_1^-,\dots,-\lambda_\kappa^-)\to e_r(\mathsf P)=Z_r
\)，其中最后一步由命题 \ref{prop:xi-pick-poisson-charpoly-coefficients}。证毕。
\end{proof}

\begin{corollary}[Pick--Poisson 最小特征值的比值下界]\label{cor:xi-pick-poisson-lambda-min-ratio-bound}
\leanverified{paper\_xi\_pick\_poisson\_lambda\_min\_ratio\_bound}
在 \(\{a_j\}\) 两两不同从而 \(\mathsf P\) 正定的情形下，有
\[
\boxed{\;\lambda_{\min}(\mathsf P)\ \ge\ \frac{Z_\kappa}{Z_{\kappa-1}}.\;}
\]
并且对所有充分大的 \(N\)，有扰动下界
\[
-\lambda_\kappa^{-}(\mathbf{T}_N)\ \ge\ \frac{Z_\kappa}{Z_{\kappa-1}}\ -\ \|A_N+\mathsf P\|_{\mathrm{op}},
\qquad
A_N:=R_N^\ast(\mathbf{T}_N|_{\mathcal K_N})R_N.
\]
\end{corollary}
\begin{proof}
设 \(\mathsf P\) 的特征值为 \(0<\lambda_1\le\cdots\le\lambda_\kappa\)。
则
\(
Z_\kappa=e_\kappa(\mathsf P)=\prod_{j=1}^\kappa \lambda_j
\)
且
\(
Z_{\kappa-1}=e_{\kappa-1}(\mathsf P)=\sum_{j=1}^\kappa\prod_{i\ne j}\lambda_i
=(\prod_{j=1}^\kappa\lambda_j)\sum_{j=1}^\kappa\lambda_j^{-1}
\)。
因此
\(
Z_{\kappa-1}/Z_\kappa=\sum_j\lambda_j^{-1}\ge \lambda_1^{-1}
\)，即 \(\lambda_{\min}(\mathsf P)=\lambda_1\ge Z_\kappa/Z_{\kappa-1}\)。
第二式由 Weyl 特征值扰动界 \(|\lambda_{\min}(A)-\lambda_{\min}(B)|\le\|A-B\|_{\mathrm{op}}\) 应用到 \(A=-A_N\)、\(B=\mathsf P\) 得到。证毕。
\end{proof}

\begin{proposition}[负谱向 Pick--Poisson 极限的指数收敛预算]\label{prop:xi-toeplitz-negative-spectrum-exponential-rate}
沿用命题 \ref{prop:xi-toeplitz-negative-spectrum-pick-poisson-gram} 的设定，并假设
\[
r_\ast:=\max_{1\le j\le\kappa}|a_j|\ <\ 1.
\]
则存在常数 \(C>0\) 使对充分大的 \(N\) 有
\[
\boxed{\;\|A_N+\mathsf P\|_{\mathrm{op}}\ \le\ C\,r_\ast^{\,N}.\;}
\]
从而对每个 \(1\le j\le \kappa\) 有特征值误差界
\[
\bigl|-\lambda_j^-(\mathbf{T}_N)-\lambda_j(\mathsf P)\bigr|\ \le\ C\,r_\ast^{\,N},
\]
并且对每个 \(0\le r\le\kappa\) 存在常数 \(C_r\) 使系数误差满足
\[
\Bigl|e_r\!\bigl(-\lambda_1^-(\mathbf{T}_N),\dots,-\lambda_\kappa^-(\mathbf{T}_N)\bigr)-Z_r\Bigr|
\ \le\ C_r\,r_\ast^{\,N}.
\]
\end{proposition}
\begin{proof}
命题 \ref{prop:xi-toeplitz-negative-spectrum-pick-poisson-gram} 的证明结构表明：当 \(N(1-|a_j|)\to\infty\) 时，截断误差在算子范数下可由 \(|a_j|^N\) 的线性组合控制。
在 \(r_\ast<1\) 的情形下，\(|a_j|^N\le r_\ast^N\) 一致成立，吸收常数即可得到第一式。
特征值误差由 Weyl 不等式直接推出。
系数误差由 \(e_r\) 对根的 Lipschitz 连续性（在由 \(\mathsf P\) 的谱与其邻域组成的紧集上）与特征值误差界合并得到。证毕。
\end{proof}

\begin{corollary}[负谱倒数和的指数收敛与闭式极限]\label{cor:xi-toeplitz-negative-reciprocal-sum-limit}
在命题 \ref{prop:xi-toeplitz-negative-spectrum-exponential-rate} 的设定下，进一步假设 \(\{a_j\}\) 两两不同从而 \(\mathsf P\succ0\)。
将 \(\mathbf{T}_N\) 的负特征值按重数写作
\[
\lambda_1^-(\mathbf{T}_N)\le\cdots\le\lambda_\kappa^-(\mathbf{T}_N)<0,
\qquad
\mu_j(N):=-\lambda_j^-(\mathbf{T}_N)>0.
\]
则存在常数 \(C>0\) 使对充分大的 \(N\) 有
\[
\boxed{\;
\sum_{j=1}^{\kappa}\frac{1}{-\lambda_j^-(\mathbf{T}_N)}
=\sum_{j=1}^{\kappa}\frac{1}{\mu_j(N)}
=\mathrm{tr}(\mathsf P^{-1})+O(r_\ast^{\,N})\; } .
\]
并且由命题 \ref{prop:xi-pick-poisson-inverse-principal-minors-jacobi} 得到完全显式极限表达
\[
\boxed{\;
\lim_{N\to\infty}\sum_{j=1}^{\kappa}\frac{1}{-\lambda_j^-(\mathbf{T}_N)}
=\sum_{i=1}^{\kappa}\frac{1}{p_i\prod_{j\ne i}\rho(a_i,a_j)^2}\; } .
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:xi-toeplitz-negative-spectrum-exponential-rate}，\(\mu_j(N)\) 以指数速率收敛到 \(\mathsf P\) 的特征值 \(\lambda_j(\mathsf P)>0\)，并且当 \(N\) 足够大时有一致下界 \(\mu_j(N)\ge \lambda_{\min}(\mathsf P)/2\)。
因此函数 \(x\mapsto 1/x\) 在区间 \([\lambda_{\min}(\mathsf P)/2,\infty)\) 上 Lipschitz，从而
\[
\sum_{j=1}^{\kappa}\left|\frac{1}{\mu_j(N)}-\frac{1}{\lambda_j(\mathsf P)}\right|
=O(r_\ast^{\,N}).
\]
而 \(\sum_j \lambda_j(\mathsf P)^{-1}=\mathrm{tr}(\mathsf P^{-1})\)，得到第一式。
第二式由命题 \ref{prop:xi-pick-poisson-inverse-principal-minors-jacobi} 的迹闭式直接代入。证毕。
\end{proof}
