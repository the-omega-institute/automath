\begin{conclusion}[椭圆迹幂和的整性、次数界与 Lee--Yang 判别式理想]\label{con:xi-terminal-elliptic-trace-integrality-degree-discriminant-ideal}

\begin{enumerate}
  \item \textbf{（整性与无有限极点）}
        \(T_m(y)\in\ZZ[y]\)，从而 \(T_m\) 在有限 \(y\)-点处不存在极点（仅可能在 \(y=\infty\) 处有极）。

        \deg_y T_m\ \le\ \left\lceil\frac{m}{2}\right\rceil.
        \]
  \item \textbf{（判别式理想与奇异集合）}
        \(\mathcal{O}\) 的判别式理想由

        \Disc_{\lambda}\bigl(\Pi(\lambda,y)\bigr)=-\,y(y-1)P_{\mathrm{LY}}(y)

        \bigl(P_{\mathrm{LY}}(y)=256y^{3}+411y^{2}+165y+32\bigr)

\end{conclusion}
\begin{proof}[证明要点]
\textbf{(1)} 由于 \(\Pi(\lambda,y)\in\ZZ[y][\lambda]\) 关于 \(\lambda\) 首一，\(\lambda\) 对 \(\ZZ[y]\) 整，从而 \(\lambda^{m}\) 亦对 \(\ZZ[y]\) 整并属于有限自由 \(\ZZ[y]\)-代数 \(\mathcal{O}\)。
