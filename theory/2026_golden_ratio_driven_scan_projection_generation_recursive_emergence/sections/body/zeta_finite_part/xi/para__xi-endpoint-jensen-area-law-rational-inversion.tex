\paragraph{端点 Jensen 场的面积律与有理因子反演}\label{par:xi-endpoint-jensen-area-law-rational-inversion}
在共动参数 \(x_0\) 下，离临界线零点的单点端点贡献可写成一维对数势核。
该核在 \(x_0\)-方向可积，且积分仅依赖偏移 \(\delta=\tfrac12-\beta\)。
在有限缺陷口径下，指数化后的端点场具有显式有理延拓，零极点集合与缺陷参数一一对应。

\begin{theorem}[单缺陷端点 Jensen 贡献的显式面积律]\label{thm:xi-endpoint-jensen-single-defect-area-law}
设 \(\rho=\beta+\mathrm{i}\gamma\) 满足 \(\beta<\tfrac12\)，并记
\(
\delta:=\tfrac12-\beta>0
\)。
定义单缺陷端点贡献
\begin{equation}\label{eq:xi-endpoint-jensen-single-defect}
J_{\rho}^{\mathrm{end}}(x_0)
:=\log\frac{1}{|w_{\rho,x_0}|},
\qquad
w_{\rho,x_0}:=\mathcal C\!\bigl((\gamma-x_0)+\mathrm{i}\delta\bigr).
\end{equation}
则有显式表达
\begin{equation}\label{eq:xi-endpoint-jensen-single-defect-explicit}
J_{\rho}^{\mathrm{end}}(x_0)
=\frac12\log
\frac{(\gamma-x_0)^2+(1+\delta)^2}{(\gamma-x_0)^2+(1-\delta)^2},
\end{equation}
并且
\begin{equation}\label{eq:xi-endpoint-jensen-single-defect-area-law}
\int_{\RR}J_{\rho}^{\mathrm{end}}(x_0)\,\dd x_0
=\pi\bigl((1+\delta)-|1-\delta|\bigr)
=2\pi\min\{\delta,1\}.
\end{equation}
特别地，在 \(\zeta\) 的临界条带口径 \(0<\delta<\tfrac12\) 下，
\begin{equation}\label{eq:xi-endpoint-jensen-single-defect-area-law-zeta}
\int_{\RR}J_{\rho}^{\mathrm{end}}(x_0)\,\dd x_0=2\pi\delta.
\end{equation}
\end{theorem}

\begin{proof}
由附录定理 \ref{thm:app-comoving-horizon-localization}，
\[
h_{x_0}(\rho)=1-|w_{\rho,x_0}|^2
=\frac{4\delta}{(\gamma-x_0)^2+(1+\delta)^2}.
\]
故
\[
|w_{\rho,x_0}|^2
=1-\frac{4\delta}{(\gamma-x_0)^2+(1+\delta)^2}
=\frac{(\gamma-x_0)^2+(1-\delta)^2}{(\gamma-x_0)^2+(1+\delta)^2},
\]
即得 \eqref{eq:xi-endpoint-jensen-single-defect-explicit}。

令 \(x=\gamma-x_0\)，则
\[
\int_{\RR}J_{\rho}^{\mathrm{end}}(x_0)\,\dd x_0
=\frac12\int_{\RR}\log\frac{x^2+(1+\delta)^2}{x^2+(1-\delta)^2}\,\dd x.
\]
对任意 \(a,b>0\)，记
\(
I(a,b):=\int_{\RR}\log\frac{x^2+a^2}{x^2+b^2}\,\dd x
\)。
对 \(a\) 求导并用
\(
\int_{\RR}\frac{\dd x}{x^2+a^2}=\pi/a
\)
得
\[
\partial_a I(a,b)=\int_{\RR}\frac{2a}{x^2+a^2}\,\dd x=2\pi.
\]
由 \(I(b,b)=0\) 可知 \(I(a,b)=2\pi(a-b)\)。
取 \(a=1+\delta,\ b=|1-\delta|\) 得
\[
\int_{\RR}J_{\rho}^{\mathrm{end}}(x_0)\,\dd x_0
=\frac12I(1+\delta,|1-\delta|)
=\pi\bigl((1+\delta)-|1-\delta|\bigr)
=2\pi\min\{\delta,1\}.
\]
当 \(0<\delta<\tfrac12\) 时即化为 \eqref{eq:xi-endpoint-jensen-single-defect-area-law-zeta}。证毕。
\end{proof}

\begin{corollary}[有限缺陷系综的可加面积律与质量读出]\label{cor:xi-endpoint-jensen-finite-ensemble-area-law}
设有限缺陷多重集
\(
\{\rho_j=\beta_j+\mathrm{i}\gamma_j\}_{j=1}^{N}
\)
带重数 \(m_j\in\NN\)，并记 \(\delta_j:=\tfrac12-\beta_j>0\)。
定义端点场
\begin{equation}\label{eq:xi-endpoint-jensen-finite-ensemble-field}
D_{\zeta}^{\mathrm{end}}(x_0)
:=\sum_{j=1}^{N}m_j\,J_{\rho_j}^{\mathrm{end}}(x_0).
\end{equation}
则
\begin{equation}\label{eq:xi-endpoint-jensen-finite-ensemble-area-law}
\int_{\RR}D_{\zeta}^{\mathrm{end}}(x_0)\,\dd x_0
=2\pi\sum_{j=1}^{N}m_j\min\{\delta_j,1\}.
\end{equation}
若进一步 \(0<\delta_j<1\)（尤其是 \(\zeta\) 临界条带情形），记
\(
M_0:=\sum_{j=1}^{N}m_j\delta_j
\)，则
\begin{equation}\label{eq:xi-endpoint-jensen-finite-ensemble-m0-readout}
M_0=\frac{1}{2\pi}\int_{\RR}D_{\zeta}^{\mathrm{end}}(x_0)\,\dd x_0.
\end{equation}
\end{corollary}

\begin{proof}
由定理 \ref{thm:xi-endpoint-jensen-single-defect-area-law} 对每个 \(j\) 成立
\(
\int J_{\rho_j}^{\mathrm{end}}=2\pi\min\{\delta_j,1\}
\)。
有限和可交换积分，得
\eqref{eq:xi-endpoint-jensen-finite-ensemble-area-law}。
当 \(\delta_j<1\) 时 \(\min\{\delta_j,1\}=\delta_j\)，即得
\eqref{eq:xi-endpoint-jensen-finite-ensemble-m0-readout}。证毕。
\end{proof}

\begin{theorem}[端点场指数化函数的有理因子化与唯一反演]\label{thm:xi-endpoint-jensen-rational-factorization-inversion}
在推论 \ref{cor:xi-endpoint-jensen-finite-ensemble-area-law} 设定下，额外假设
\(
0<\delta_j<1
\)（临界条带口径）。
定义
\begin{equation}\label{eq:xi-endpoint-jensen-rationalized-field}
R_{\zeta}^{\mathrm{end}}(x_0):=\exp\!\bigl(2D_{\zeta}^{\mathrm{end}}(x_0)\bigr).
\end{equation}
则存在有理函数 \(R_{\zeta}^{\mathrm{end}}(z)\in\RR(z)\) 使其在实轴上的限制为 \eqref{eq:xi-endpoint-jensen-rationalized-field}，并满足精确因子分解
\begin{equation}\label{eq:xi-endpoint-jensen-rational-factorization}
R_{\zeta}^{\mathrm{end}}(z)
=\prod_{j=1}^{N}
\left(
\frac{(z-\gamma_j)^2+(1+\delta_j)^2}
{(z-\gamma_j)^2+(1-\delta_j)^2}
\right)^{m_j}.
\end{equation}
因而其零极点为：
\[
\text{零点 }z=\gamma_j\pm \mathrm{i}(1+\delta_j),\ \text{重数 }m_j;
\]
\[
\text{极点 }z=\gamma_j\pm \mathrm{i}(1-\delta_j),\ \text{重数 }m_j.
\]
并且，若已知 \(D_{\zeta}^{\mathrm{end}}\) 在某非平凡开区间 \(I\subset\RR\) 上的限制，
则多重集 \(\{(\gamma_j,\delta_j,m_j)\}_{j=1}^{N}\) 被唯一确定。
\end{theorem}

\begin{proof}
由 \eqref{eq:xi-endpoint-jensen-single-defect-explicit}，
\[
\exp\!\bigl(2J_{\rho_j}^{\mathrm{end}}(x_0)\bigr)
=\frac{(x_0-\gamma_j)^2+(1+\delta_j)^2}{(x_0-\gamma_j)^2+(1-\delta_j)^2}.
\]
对 \(j\) 取乘积并计入重数 \(m_j\)，即得实轴上的乘积表达；将 \(x_0\) 替换为复变量 \(z\) 即得
\eqref{eq:xi-endpoint-jensen-rational-factorization} 与有理延拓。
由 \(0<\delta_j<1\)，零点高度 \(1+\delta_j>1\)，极点高度 \(1-\delta_j\in(0,1)\)，故零极点层分离且不会跨层抵消。

唯一性方面，若两组有限参数给出同一 \(D_{\zeta}^{\mathrm{end}}|_I\)，则对应
\(
R_1|_I=R_2|_I
\)。
于是有理函数 \(R_1-R_2\) 在具有聚点的集合 \(I\) 上为零，只能恒等为零，故 \(R_1\equiv R_2\)。
再由 \eqref{eq:xi-endpoint-jensen-rational-factorization} 的零极点分解唯一性，得到参数多重集唯一。证毕。
\end{proof}
