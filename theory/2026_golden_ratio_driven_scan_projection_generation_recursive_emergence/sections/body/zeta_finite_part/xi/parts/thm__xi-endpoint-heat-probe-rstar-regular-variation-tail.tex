\begin{theorem}[谱隙半径端点的正则变差决定尾项渐近]\label{thm:xi-endpoint-heat-probe-rstar-regular-variation-tail}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_rstar\_regular\_variation\_tail}
沿用 \eqref{eq:xi-endpoint-heat-probe-rstar-def} 的谱隙半径 \(r_\ast\) 与尾项
\[
T_N:=a_N-m_{-1}=\int_{\TT\setminus\{-1\}} r(\xi)^N\,\dd\mu(\xi),
\qquad r(\xi)=\frac{|1-\xi|^2}{4}\in[0,1).
\]
令 \(\nu:=r_\#(\mu|_{\TT\setminus\{-1\}})\) 为 \([0,r_\ast]\) 上的推前测度，则
\[
T_N=\int_{[0,r_\ast]} s^{N}\,\dd\nu(s),
\qquad
r_\ast=\sup\supp(\nu).
\]
\begin{enumerate}
  \item 若 \(\nu(\{r_\ast\})>0\)，则
  \[
  T_N\sim \nu(\{r_\ast\})\,r_\ast^{N}.
  \]
  \item 若 \(\nu(\{r_\ast\})=0\)，并且存在 \(\alpha>-1\) 与常数 \(C>0\) 使得当 \(\varepsilon\downarrow 0\) 时
  \[
  \nu([r_\ast-\varepsilon,r_\ast])\sim C\,\varepsilon^{\alpha+1},
  \]
  则
  \[
  T_N\sim C\,\Gamma(\alpha+2)\,r_\ast^{N+\alpha+1}\,N^{-(\alpha+1)}.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
由推前定义，
\(
T_N=\int_{[0,r_\ast]} s^N\,\dd\nu(s)
\)。

\textbf{(1) 原子情形。}
将原子项分离：
\[
T_N=r_\ast^N\,\nu(\{r_\ast\})+\int_{[0,r_\ast)} s^{N}\,\dd\nu(s).
\]
对余项作归一化并应用主导收敛定理：
\[
0\le r_\ast^{-N}\int_{[0,r_\ast)} s^{N}\,\dd\nu(s)
=\int_{[0,r_\ast)}\Bigl(\frac{s}{r_\ast}\Bigr)^{N}\,\dd\nu(s)\xrightarrow[N\to\infty]{}0,
\]
因为对每个 \(s<r_\ast\) 有 \((s/r_\ast)^N\to 0\)，且 \((s/r_\ast)^N\le 1\) 并且 \(\nu([0,r_\ast))<\infty\)。
故 \(T_N\sim \nu(\{r_\ast\})r_\ast^N\)。

\textbf{(2) 无原子且正则变差。}
由 Stieltjes 分部积分（注意 \(\nu(\{r_\ast\})=0\)）可得
\[
T_N
:=N\int_{0}^{r_\ast} s^{N-1}\,\nu([s,r_\ast])\,\dd s.
\]
\par
令 \(\varepsilon_N:=r_\ast N^{-2/3}\)。
将积分拆为 \([0,r_\ast-\varepsilon_N]\) 与 \([r_\ast-\varepsilon_N,r_\ast]\) 两段。
第一段满足
\[
0\le N\int_{0}^{r_\ast-\varepsilon_N} s^{N-1}\nu([s,r_\ast])\,\dd s
\le N\,\nu([0,r_\ast])\,r_\ast\,(r_\ast-\varepsilon_N)^{N-1}
=o\!\left(r_\ast^{N+\alpha+1}N^{-(\alpha+1)}\right),
\]
因为 \((r_\ast-\varepsilon_N)^{N-1}=r_\ast^{N-1}(1-N^{-2/3})^{N-1}\le r_\ast^{N-1}e^{-N^{1/3}/2}\)（对 \(N\) 足够大）。
\par
对第二段作尺度变换 \(s=r_\ast(1-u/N)\)，此时 \(u\in[0,N^{1/3}]\)，从而
\[
T_N
=r_\ast^{N}\int_{0}^{N^{1/3}}(1-u/N)^{N-1}\,
\nu\!\left([r_\ast-r_\ast u/N,\ r_\ast]\right)\,\dd u
+o\!\left(r_\ast^{N+\alpha+1}N^{-(\alpha+1)}\right).
\]
当 \(0\le u\le N^{1/3}\) 时，\(\varepsilon=r_\ast u/N\le \varepsilon_N\downarrow 0\)。
由正则变差假设，存在 \(o(1)\to 0\) 使
\[
\nu([r_\ast-r_\ast u/N,r_\ast])=C\,(r_\ast u/N)^{\alpha+1}(1+o(1)),
\]
且该 \(o(1)\) 可取为一致于 \(u\in[0,N^{1/3}]\)（因为其只依赖于 \(\sup_{\varepsilon\le \varepsilon_N}\) 的小量）。
并且由 \(\log(1-u/N)=-u/N+O(u^2/N^2)\) 与 \(u\le N^{1/3}\) 得
\[
(1-u/N)^{N-1}=e^{-u}(1+o(1))
\]
一致成立。
因此
\[
T_N
\sim r_\ast^{N}\cdot C\,r_\ast^{\alpha+1}\,N^{-(\alpha+1)}
\int_{0}^{N^{1/3}}e^{-u}u^{\alpha+1}\,\dd u.
\]
令 \(N\to\infty\) 并用 \(\int_{0}^{N^{1/3}}e^{-u}u^{\alpha+1}\,\dd u\to \Gamma(\alpha+2)\) 即得所示渐近。证毕。
\end{proof}
