\begin{theorem}[Adams 扭曲与圆分方靶点的端点热核探针]\label{thm:xi-endpoint-heat-probe-adams-twist-roots}\label{thm:xi-endpoint-heat-probe-cyclotomic-target-extraction}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_adams\_twist\_roots}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_cyclotomic\_target\_extraction}
令 \(\mu\) 为 \(\TT\) 上有限正 Borel 测度，并令 \(c_h:=\int_{\TT}\xi^{-h}\,\dd\mu(\xi)\)（\(h\in\ZZ\)）。
对任意整数 \(d\ge 1\) 与任意 \(\omega\in\TT\)，定义 \(d\)-扭曲矩
\[
c^{(d)}_h:=c_{dh}\qquad(h\in\ZZ),
\]
以及相应 Toeplitz 主子矩阵
\(
\mathbf{T}^{(d)}_N:=(c^{(d)}_{j-k})_{0\le j,k\le N}
\)。
定义圆分方探针向量
\[
b^{(N)}(\omega):=\bigl(\binom Nk(-\omega)^k\bigr)_{k=0}^{N}\in\CC^{N+1},
\]
并定义扭曲端点热核探针
\[
a_N^{(d)}(\omega):=4^{-N}\,(b^{(N)}(\omega))^\ast\,\mathbf{T}^{(d)}_N\,b^{(N)}(\omega).
\]
则对所有 \(N\ge 0\) 成立严格恒等式
\[
\boxed{\;
a_N^{(d)}(\omega)=\int_{\TT}\left(\frac{|1-\omega\xi^{d}|^2}{4}\right)^{N}\,\dd\mu(\xi)\ \in\ [0,\mu(\TT)].\;}
\]
并且序列 \((a_N^{(d)}(\omega))_{N\ge 0}\) 单调下降且满足极限抽取公式
\[
\boxed{\;
a_N^{(d)}(\omega)\ \downarrow\ \mu\bigl(\Sigma_{d,\omega}\bigr)\qquad(N\to\infty),\;}
\]
其中
\[
\Sigma_{d,\omega}:=\{\xi\in\TT:\ \omega\,\xi^{d}=-1\}
=\{\xi\in\TT:\ \xi^{d}=-\overline{\omega}\}.
\]
特别地，取 \(\omega=1\) 时恢复 Adams 扭曲读出集合 \(\{\xi^d=-1\}\) 的原子质量。
\end{theorem}

\begin{proof}
由二项式展开
\[
(1-\omega\xi^{d})^N=\sum_{k=0}^{N}\binom Nk(-\omega)^k\xi^{dk},
\]
并按命题 \ref{prop:xi-endpoint-heat-probe-toeplitz-herglotz-identity} 的同型计算，
\[
(b^{(N)}(\omega))^\ast \mathbf{T}^{(d)}_N b^{(N)}(\omega)
=\int_{\TT}\left|\sum_{k=0}^{N}\binom Nk(-\omega)^k\xi^{dk}\right|^2\dd\mu(\xi)
=\int_{\TT}|1-\omega\xi^{d}|^{2N}\dd\mu(\xi),
\]
两边同除以 \(4^N\) 得恒等式。
\par
对任意 \(\xi\in\TT\)，有 \(0\le |1-\omega\xi^{d}|\le 2\)，故
\(
0\le |1-\omega\xi^{d}|^2/4\le 1
\)。
并且 \(|1-\omega\xi^{d}|=2\) 当且仅当 \(\omega\xi^{d}=-1\)，即 \(\xi\in\Sigma_{d,\omega}\)。
因此核函数 \(\left(|1-\omega\xi^{d}|^2/4\right)^N\) 随 \(N\) 单调下降到 \(\ind_{\Sigma_{d,\omega}}\)。
由单调收敛定理得到极限等式。证毕。
\end{proof}

\begin{theorem}[Adams--二项探针在旋相参数上的 Fourier 对角化与矩反演]\label{thm:xi-adams-binomial-probe-fourier-diagonalization}
在定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots} 的设定下，固定整数 \(d\ge 1\) 与阶数 \(N\ge 0\)。
则函数 \(\omega\mapsto a_N^{(d)}(\omega)\) 在 \(\TT\) 上是次数 \(\le N\) 的 Laurent 多项式，并且具有显式 Fourier 展开
\begin{equation}\label{eq:xi-adams-binomial-probe-fourier-expansion}
\boxed{\;
a_N^{(d)}(\omega)=\sum_{m=-N}^{N}\Bigl((-1)^m\,4^{-N}\binom{2N}{N+m}\,c_{-dm}\Bigr)\,\omega^{m}\qquad(\omega\in\TT).\;}
\end{equation}
从而对每个 \(|m|\le N\)，其 Fourier 系数满足
\begin{equation}\label{eq:xi-adams-binomial-probe-fourier-coefficient}
\boxed{\;
\widehat a_{N}^{(d)}(m)
\,:=\,
\frac{1}{2\pi}\int_{0}^{2\pi}a_N^{(d)}(e^{\mathrm{i}\theta})\,e^{-\mathrm{i}m\theta}\,\dd\theta
\;=\;
(-1)^m\,4^{-N}\binom{2N}{N+m}\,c_{-dm}\;}
\end{equation}
并可精确反演得到
\begin{equation}\label{eq:xi-adams-binomial-probe-moment-inversion}
\boxed{\;
c_{-dm}
\,=\,
(-1)^m\,\frac{4^{N}}{\binom{2N}{N+m}}\cdot \widehat a_{N}^{(d)}(m)\qquad(|m|\le N).\;}
\end{equation}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots}，
\[
a_N^{(d)}(\omega)
=4^{-N}\int_{\TT}(1-\omega\xi^{d})^{N}(1-\overline{\omega}\xi^{-d})^{N}\,\dd\mu(\xi).
\]
对两因子作二项展开：
\[
(1-\omega\xi^{d})^{N}=\sum_{k=0}^{N}\binom Nk(-\omega)^{k}\xi^{dk},
\qquad
(1-\overline{\omega}\xi^{-d})^{N}=\sum_{\ell=0}^{N}\binom N\ell(-\overline{\omega})^{\ell}\xi^{-d\ell}.
\]
相乘并积分得到
\[
a_N^{(d)}(\omega)
=4^{-N}\sum_{k,\ell=0}^{N}\binom Nk\binom N\ell (-\omega)^{k}(-\overline{\omega})^{\ell}
\int_{\TT}\xi^{d(k-\ell)}\,\dd\mu(\xi).
\]
由 \(c_h=\int_{\TT}\xi^{-h}\,\dd\mu(\xi)\) 可知
\(
\int_{\TT}\xi^{d(k-\ell)}\,\dd\mu(\xi)=c_{-d(k-\ell)}
\)。
令 \(m:=k-\ell\)。在 \(\omega\in\TT\) 上有 \(\overline{\omega}=\omega^{-1}\)，因此
\[
(-\omega)^{k}(-\overline{\omega})^{\ell}=(-1)^{k+\ell}\omega^{k-\ell}=(-1)^{m}\omega^{m}.
\]
从而
\[
a_N^{(d)}(\omega)
=\sum_{m=-N}^{N}(-1)^{m}\omega^{m}\,4^{-N}\,c_{-dm}\sum_{\ell}\binom N{\ell+m}\binom N\ell,
\]
其中 \(\ell\) 在使指标合法的范围内求和。
当 \(|m|\le N\) 时，Vandermonde 型恒等式给出
\(
\sum_{\ell}\binom N{\ell+m}\binom N\ell=\binom{2N}{N+m}
\)，代回即得 \eqref{eq:xi-adams-binomial-probe-fourier-expansion}。
\par
再对 \(\omega=e^{\mathrm{i}\theta}\) 取 Fourier 正交提取：对 \(|m|\le N\) 有
\[
\frac{1}{2\pi}\int_{0}^{2\pi}e^{\mathrm{i}(m'-m)\theta}\,\dd\theta=\mathbf 1_{\{m'=m\}},
\]
故 \eqref{eq:xi-adams-binomial-probe-fourier-coefficient} 由 \eqref{eq:xi-adams-binomial-probe-fourier-expansion} 逐项给出，进而反解得到 \eqref{eq:xi-adams-binomial-probe-moment-inversion}。证毕。
\end{proof}

\begin{theorem}[二项式探针核的正定性与 Toeplitz--PSD 的等价]\label{thm:xi-adams-binomial-probe-kernel-toeplitz-psd-equivalence}
在定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots} 的设定下，固定 \(d\ge 1\) 与 \(N\ge 0\)，并定义二参数核
\[
K_N^{(d)}(\omega,\omega')
:=4^{-N}\,(b^{(N)}(\omega))^\ast\,\mathbf{T}^{(d)}_N\,b^{(N)}(\omega')
\qquad(\omega,\omega'\in\TT).
\]
则以下两条严格等价：
\begin{enumerate}
  \item \(\mathbf{T}^{(d)}_N\succeq 0\)（Hermitian--PSD）；
  \item \(K_N^{(d)}\) 是 \(\TT\) 上的正定核：对任意有限点列 \(\omega_1,\dots,\omega_M\in\TT\)，矩阵
  \(
  \bigl[K_N^{(d)}(\omega_i,\omega_j)\bigr]_{i,j=1}^{M}
  \)
  半正定。
\end{enumerate}
并且更强：取任意 \(N+1\) 个两两不同的点 \(\omega_0,\dots,\omega_N\in\TT\)，令
\[
\mathbf K:=\bigl[K_N^{(d)}(\omega_i,\omega_j)\bigr]_{0\le i,j\le N},
\]
则有
\[
\boxed{\;
\mathbf{T}^{(d)}_N\succeq 0\quad\Longleftrightarrow\quad \mathbf K\succeq 0.\;}
\]
\end{theorem}
\begin{proof}
\textbf{(1)\(\Rightarrow\)(2).}
对任意 \(\omega_1,\dots,\omega_M\)，令 \(\mathbf B:=[b^{(N)}(\omega_1)\ \cdots\ b^{(N)}(\omega_M)]\in\CC^{(N+1)\times M}\)。
则由定义
\[
\bigl[K_N^{(d)}(\omega_i,\omega_j)\bigr]_{i,j=1}^{M}
=4^{-N}\,\mathbf B^\ast\,\mathbf{T}^{(d)}_N\,\mathbf B,
\]
为 \(\mathbf{T}^{(d)}_N\) 的合同像，故若 \(\mathbf{T}^{(d)}_N\succeq 0\) 则该矩阵半正定。
\par
\textbf{(2)\(\Rightarrow\)(1).}
取 \(N+1\) 个两两不同的点 \(\omega_0,\dots,\omega_N\in\TT\)，并令 \(\mathbf B:=[b^{(N)}(\omega_0)\ \cdots\ b^{(N)}(\omega_N)]\in\CC^{(N+1)\times(N+1)}\)。
由
\(
b^{(N)}(\omega)=(\binom Nk(-1)^k\omega^k)_{k=0}^{N}
\)
可得分解
\[
\mathbf B
=\mathrm{diag}\bigl(\binom Nk(-1)^k\bigr)_{k=0}^{N}\cdot \mathbf V,
\qquad
\mathbf V_{k,j}:=\omega_j^{\,k}\quad(0\le k,j\le N),
\]
其中 \(\mathbf V\) 为 Vandermonde 矩阵。
由于 \(\omega_j\) 两两不同，\(\det\mathbf V=\prod_{0\le i<j\le N}(\omega_j-\omega_i)\neq 0\)，且对角因子显然可逆，故 \(\mathbf B\) 可逆。
又由定义 \(\mathbf K=4^{-N}\mathbf B^\ast\mathbf{T}^{(d)}_N\mathbf B\)，从而
\[
\mathbf{T}^{(d)}_N=(\mathbf B^{-1})^\ast\,(4^N\mathbf K)\,\mathbf B^{-1}.
\]
若 \(\mathbf K\succeq 0\)，则右端为半正定合同像，故 \(\mathbf{T}^{(d)}_N\succeq 0\)。
这也同时给出所述“单一网格”判别式。证毕。
\end{proof}

\begin{theorem}[strictification 语义下的探针完备性]\label{thm:xi-adams-binomial-probe-completeness-strictification}
令 \(C_1,C_2\) 为 \(\mathbb D\) 上 Carath\'eodory 函数，并令 \(\mu_1,\mu_2\) 为其 Herglotz 测度，使得
\[
c_{j,h}:=\int_{\TT}\xi^{-h}\,\dd\mu_j(\xi)\qquad(h\in\ZZ,\ j\in\{1,2\}).
\]
对每个 \(N\ge 0\) 与 \(\omega\in\TT\) 定义旋相端点探针
\[
a_{N,j}(\omega):=\int_{\TT}\left(\frac{|1-\omega\xi|^2}{4}\right)^{N}\dd\mu_j(\xi)\qquad(j\in\{1,2\}).
\]
若对一切 \(N\ge 0\) 与一切 \(\omega\in\TT\) 都有 \(a_{N,1}(\omega)=a_{N,2}(\omega)\)，则 \(\mu_1=\mu_2\)。
从而 \(C_1-C_2\) 必为纯虚常数，且两者诱导的 Pick 核一致：\(K_{C_1}\equiv K_{C_2}\)（见定理 \ref{thm:xi-pick-kernel-gauge-rigidity}）。
\end{theorem}
\begin{proof}
固定任意 \(m\in\ZZ\)，取任意 \(N\ge |m|\)。
由定理 \ref{thm:xi-adams-binomial-probe-fourier-diagonalization}（取 \(d=1\)）与假设 \(a_{N,1}\equiv a_{N,2}\)，得其 Fourier 系数相等，从而
\[
(-1)^m4^{-N}\binom{2N}{N+m}\,c_{1,-m}
=\widehat a_{N,1}(m)
=\widehat a_{N,2}(m)
=(-1)^m4^{-N}\binom{2N}{N+m}\,c_{2,-m}.
\]
故 \(c_{1,-m}=c_{2,-m}\)。由于 \(m\in\ZZ\) 任意，得到两测度的全部 Fourier 系数一致，从而 \(\mu_1=\mu_2\)。
Herglotz 表示的唯一性给出 \(C_1-C_2\) 为纯虚常数；最后一条由定理 \ref{thm:xi-pick-kernel-gauge-rigidity} 直接推出。证毕。
\end{proof}

\begin{corollary}[惰性随机游走热核表象]\label{cor:xi-adams-binomial-probe-lazy-walk-heat-kernel}
\leanverified{paper\_xi\_adams\_binomial\_probe\_lazy\_walk\_heat\_kernel}
在定理 \ref{thm:xi-adams-binomial-probe-fourier-diagonalization} 的设定下，
令 \((X_N)_{N\ge 0}\) 为 \(\ZZ\) 上惰性随机游走：\(X_0=0\) 且增量 \(\Delta_N:=X_N-X_{N-1}\) 满足
\[
\PP(\Delta_N=1)=\PP(\Delta_N=-1)=\tfrac14,\qquad \PP(\Delta_N=0)=\tfrac12,
\]
并定义交替矩序列 \(\widetilde c_{dm}:=(-1)^m c_{-dm}\)。
则对任意 \(N\ge 0\) 与 \(\theta\in\RR\) 有
\[
\boxed{\;
\PP(X_N=m)=4^{-N}\binom{2N}{N+m}\qquad(-N\le m\le N),\;}
\]
并且
\[
\boxed{\;
a_N^{(d)}(e^{\mathrm{i}\theta})
=\EE\Bigl[\widetilde c_{dX_N}\,e^{\mathrm{i}\theta X_N}\Bigr].\;}
\]
\end{corollary}
\begin{proof}
惰性步进的特征多项式为
\[
\EE[z^{\Delta_1}]=\frac14(z+2+z^{-1})=\frac{(1+z)^2}{4z},
\]
故
\(
\EE[z^{X_N}]=\left(\frac{(1+z)^2}{4z}\right)^N
=4^{-N}z^{-N}(1+z)^{2N}
\)，比较系数得 \(\PP(X_N=m)=4^{-N}\binom{2N}{N+m}\)。
第二式由 \eqref{eq:xi-adams-binomial-probe-fourier-expansion} 直接把权重识别为 \(\PP(X_N=m)\) 并改写为期望即得。证毕。
\end{proof}

\begin{theorem}[断层反演的放大因子相变：中心区可控而边缘区指数病态]\label{thm:xi-adams-binomial-probe-inversion-conditioning-phase-transition}
沿用定理 \ref{thm:xi-adams-binomial-probe-fourier-diagonalization} 的反演式 \eqref{eq:xi-adams-binomial-probe-moment-inversion}。
设对某固定 \(d\ge 1\)、某固定阶 \(N\)，以数值方式获得 Fourier 系数近似
\(
\widehat a_N^{(d)}(m)+\delta_{N,m}
\)。
则反演得到的矩误差满足精确放大律
\[
\boxed{\;
\Delta c_{-dm}
=(-1)^m\,\frac{4^N}{\binom{2N}{N+m}}\ \delta_{N,m}\qquad(|m|\le N).\;}
\]
并且存在绝对常数 \(C_1,C_2>0\) 使得在中心区 \(|m|\le N^{2/3}\) 内有二侧估计
\[
\boxed{\;
C_1\sqrt N\,e^{m^2/N}\ \le\ \frac{4^N}{\binom{2N}{N+m}}\ \le\ C_2\sqrt N\,e^{m^2/N}\; }.
\]
另一方面，对任意固定比例 \(\alpha\in(0,1)\)，当 \(|m|\ge \alpha N\) 时存在 \(c(\alpha)>0\) 使
\[
\boxed{\;
\frac{4^N}{\binom{2N}{N+m}}\ \ge\ \exp\bigl(c(\alpha)\,N\bigr),\;}
\]
从而高频区反演不可避免地呈指数病态。
\end{theorem}
\begin{proof}
第一式由 \eqref{eq:xi-adams-binomial-probe-moment-inversion} 的线性性直接得到。
\par
令
\(
w_N(m):=4^{-N}\binom{2N}{N+m}
\)
（推论 \ref{cor:xi-adams-binomial-probe-lazy-walk-heat-kernel} 中即 \(\PP(X_N=m)\)）。
对中心区 \(|m|\le N^{2/3}\)，局部中心极限定理（或 Stirling--de Moivre 近似）给出存在绝对常数 \(A_1,A_2>0\) 使
\[
\frac{A_1}{\sqrt N}\,e^{-m^2/N}\ \le\ w_N(m)\ \le\ \frac{A_2}{\sqrt N}\,e^{-m^2/N},
\]
取倒数并令 \(C_1=1/A_2\)、\(C_2=1/A_1\) 即得二侧估计。
\par
对 \(|m|\ge \alpha N\) 的大偏差区，由 Chernoff--Cram\'er 大偏差界可得
\(
w_N(m)\le \exp(-c(\alpha)N)
\)
（其中 \(c(\alpha)>0\) 仅依赖于 \(\alpha\)），取倒数即得最后一式。证毕。
\end{proof}

\begin{proposition}[等分网格的离散反演：有限点采样精确重建 Toeplitz 截断]\label{prop:xi-adams-binomial-probe-discrete-inversion-equispaced-grid}
在定理 \ref{thm:xi-adams-binomial-probe-fourier-diagonalization} 的设定下，固定 \(d\ge 1\) 与 \(N\ge 0\)。
取等分网格
\[
\omega_j:=\exp\!\left(\frac{2\pi \mathrm{i}j}{2N+1}\right)\qquad(0\le j\le 2N).
\]
则仅由采样值 \(\{a_N^{(d)}(\omega_j)\}_{j=0}^{2N}\) 可精确重建全部矩 \(\{c_{-dm}\}_{|m|\le N}\)：
\[
\boxed{\;
c_{-dm}
=(-1)^m\,\frac{4^N}{\binom{2N}{N+m}}\cdot
\frac{1}{2N+1}\sum_{j=0}^{2N} a_N^{(d)}(\omega_j)\,\omega_j^{-m}\qquad(|m|\le N).\;}
\]
从而 Toeplitz 截断 \(\mathbf{T}^{(d)}_N=(c_{d(j-k)})_{0\le j,k\le N}\) 亦由上述 \(2N+1\) 个标量探针值完全确定。
\end{proposition}
\begin{proof}
由定理 \ref{thm:xi-adams-binomial-probe-fourier-diagonalization}，\(\omega\mapsto a_N^{(d)}(\omega)\) 为次数 \(\le N\) 的 Laurent 多项式
\(
a_N^{(d)}(\omega)=\sum_{m=-N}^{N}\widehat a_N^{(d)}(m)\,\omega^{m}
\)。
而对等分根单位，正交性给出：对任意整数 \(k\) 满足 \(|k|\le N\)，有
\[
\frac{1}{2N+1}\sum_{j=0}^{2N}\omega_j^{\,k}
=\mathbf 1_{\{k=0\}}.
\]
因此离散 Fourier 提取无混叠：
\[
\widehat a_N^{(d)}(m)
=\frac{1}{2N+1}\sum_{j=0}^{2N} a_N^{(d)}(\omega_j)\,\omega_j^{-m}\qquad(|m|\le N).
\]
将其代入反演式 \eqref{eq:xi-adams-binomial-probe-moment-inversion} 即得所示闭式。证毕。
\end{proof}

\begin{theorem}[由单标量探针序列识别补支撑谱半径]\label{thm:xi-endpoint-heat-probe-cyclotomic-complement-radius}
\leanpartial{paper\_xi\_endpoint\_heat\_probe\_cyclotomic\_complement\_radius}{Requested signature depends on a new `...Data` paper theorem in a brand-new Lean file, which this round forbids as a shell-style abstraction.}
沿用定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots} 的记号并固定 \((d,\omega)\)。
记
\[
m_\infty:=\mu(\Sigma_{d,\omega}),
\qquad
t_N:=a_N^{(d)}(\omega)-m_\infty
=\int_{\TT\setminus\Sigma_{d,\omega}}\left(\frac{|1-\omega\xi^{d}|^2}{4}\right)^{N}\dd\mu(\xi).
\]
并定义补支撑的端点能量谱半径
\[
r_\ast:=\sup_{\xi\in\supp(\mu)\setminus\Sigma_{d,\omega}}\frac{|1-\omega\xi^{d}|^2}{4}\in[0,1].
\]
则有极限上确界公式
\[
\boxed{\;\limsup_{N\to\infty} t_N^{1/N}=r_\ast.\;}
\]
从而仅由标量流 \(\{a_N^{(d)}(\omega)\}_{N\ge 0}\) 即可同时识别目标原子总质量 \(m_\infty\) 与补支撑谱半径 \(r_\ast\)（决定指数型或亚指数型收敛）。
\end{theorem}

\begin{proof}
令
\(
\varphi(\xi):=\frac{|1-\omega\xi^{d}|^2}{4}\in[0,1]
\)，并令 \(\nu\) 为 \(\TT\setminus\Sigma_{d,\omega}\) 在 \(\varphi\) 下的推前测度：
\[
\nu(E):=\mu\bigl(\varphi^{-1}(E)\cap(\TT\setminus\Sigma_{d,\omega})\bigr)\qquad(E\subseteq[0,1]\ \text{Borel}).
\]
则
\[
t_N=\int_{[0,1]} s^{N}\,\dd\nu(s).
\]
由 \(\supp(\nu)\subseteq[0,r_\ast]\) 得 \(t_N\le \nu([0,1])\,r_\ast^{\,N}\)，从而 \(\limsup_{N\to\infty}t_N^{1/N}\le r_\ast\)。
\par
反之，任取 \(\varepsilon>0\)。由 \(r_\ast=\sup\supp(\nu)\) 可知 \(\nu((r_\ast-\varepsilon,r_\ast])>0\)，于是
\[
t_N\ge (r_\ast-\varepsilon)^N\,\nu\bigl((r_\ast-\varepsilon,r_\ast]\bigr),
\]
从而 \(\liminf_{N\to\infty}t_N^{1/N}\ge r_\ast-\varepsilon\)。
令 \(\varepsilon\downarrow 0\) 即得 \(\limsup_{N\to\infty}t_N^{1/N}=r_\ast\)。证毕。
\end{proof}

\begin{corollary}[补支撑谱隙下的指数收敛与可审计误差界]\label{cor:xi-endpoint-heat-probe-adams-twist-spectral-gap-exp-bound}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_adams\_twist\_spectral\_gap\_exp\_bound}
在定理 \ref{thm:xi-endpoint-heat-probe-cyclotomic-complement-radius} 的设定下，若 \(r_\ast<1\)，则对所有 \(N\ge 0\) 有
\[
0\le a_N^{(d)}(\omega)-\mu(\Sigma_{d,\omega})
=\int_{\TT\setminus\Sigma_{d,\omega}}\left(\frac{|1-\omega\xi^{d}|^2}{4}\right)^{N}\dd\mu(\xi)
\le r_\ast^{\,N}\,\mu(\TT\setminus\Sigma_{d,\omega}).
\]
\end{corollary}
\begin{proof}
由 \(0\le |1-\omega\xi^{d}|^2/4\le r_\ast\) 在 \(\operatorname{supp}(\mu)\setminus\Sigma_{d,\omega}\) 上成立，直接估计积分即得。证毕。
\end{proof}

\begin{theorem}[偶阶单位根原子质量的 \(2\)-进 Möbius 反演]\label{thm:xi-adams-fiber-mobius-inversion}
令 \(\mu\) 为 \(\TT\) 上有限正 Borel 测度。
对每个 \(d\ge 1\) 记
\[
m_d:=\mu\bigl(\{\xi\in\TT:\ \xi^{d}=-1\}\bigr).
\]
由定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots}（取 \(\omega=1\)）知 \(m_d=\lim_{N\to\infty}a_N^{(d)}(1)\)。
\par
对每个 \(n\ge 1\) 定义“最小半阶原子质量”
\[
f(n):=\mu\bigl(\{\xi\in\TT:\ \xi^{n}=-1\ \text{且对一切}\ 0<k<n\ \text{有}\ \xi^{k}\neq-1\}\bigr),
\]
等价地，\(f(n)\) 为精确阶 \(2n\) 的原子总质量。
则对任意分解 \(n=2^{a}u\)（\(a\ge 0\)、\(u\) 为奇数），成立反演公式
\[
\boxed{\quad
f(2^{a}u)=\sum_{v\mid u}\mu_{\mathrm{Mob}}(u/v)\,m_{2^{a}v},\qquad u\ \text{奇},\quad}
\]
其中 \(\mu_{\mathrm{Mob}}\) 为经典 Möbius 函数。
\end{theorem}
\begin{proof}
对任意 \(\xi\in\TT\)，若存在 \(n_0\ge 1\) 使 \(\xi^{n_0}=-1\)，令 \(n_0\) 为满足该条件的最小正整数；
否则 \(\xi\) 不对任何 \(m_d\) 产生贡献。
由定义可知：对几乎处处 \(\xi\)（按原子分解逐点讨论亦成立），其质量被唯一归入某一层 \(f(n_0)\)。
\par
对任意 \(d\ge 1\) 有
\[
\xi^{d}=-1
\iff n_0\mid d\ \text{且}\ \frac{d}{n_0}\ \text{为奇数}.
\]
因此
\[
m_d=\sum_{\substack{n_0\mid d\\(d/n_0)\ \mathrm{odd}}} f(n_0).
\]
将 \(d\) 写为 \(d=2^{a}u\)（\(u\) 奇），条件“\(d/n_0\) 为奇数”等价于 \(n_0\) 与 \(d\) 具有相同的 \(2\)-进指数，
即 \(n_0=2^{a}v\) 且 \(v\mid u\)。
于是得到对每个固定 \(a\) 的奇除子累加式
\[
m_{2^{a}u}=\sum_{v\mid u} f(2^{a}v),\qquad u\ \text{奇}.
\]
在奇整数的除法格上对函数 \(u\mapsto m_{2^{a}u}\) 作 Möbius 反演即得所示公式。证毕。
\end{proof}

\begin{corollary}[偶阶单位根处原子谱的唯一可恢复性]\label{cor:xi-adams-even-order-atomic-spectrum-determined}
在定理 \ref{thm:xi-adams-fiber-mobius-inversion} 的设定下，序列 \((m_d)_{d\ge 1}\) 唯一决定：
\begin{enumerate}
  \item 全部分层质量 \((f(n))_{n\ge 1}\)；
  \item \(\mu\) 在所有偶阶单位根处的原子质量分布（按精确阶 \(2n\) 分层）。
\end{enumerate}
并且
\[
\bigl(\forall d\ge 1,\ m_d=0\bigr)\quad\Longleftrightarrow\quad
\mu\ \text{在所有偶阶单位根处无原子}.
\]
\end{corollary}
\begin{proof}
前两条由定理 \ref{thm:xi-adams-fiber-mobius-inversion} 的显式反演直接给出。
最后的等价式：若 \(m_d\equiv 0\)，则反演给出 \(f(n)\equiv 0\)，从而偶阶单位根处无原子；
反向若 \(f(n)\equiv 0\)，则由分解式 \(m_{2^{a}u}=\sum_{v\mid u}f(2^{a}v)\) 得 \(m_d\equiv 0\)。证毕。
\end{proof}

\begin{theorem}[自对偶对称下端点探针序列确定 Toeplitz 矩]\label{thm:xi-selfdual-single-probe-determines-toeplitz-moments}
令 \(\mu\) 为 \(\TT\) 上有限正 Borel 测度，并假设其关于共轭不变：\(\mu(A)=\mu(\overline A)\)（\(A\) Borel）。
记
\[
a_N:=a_N^{(1)}(1)=\int_{\TT}\left(\frac{|1-\xi|^2}{4}\right)^{N}\dd\mu(\xi),\qquad N\ge 0.
\]
令 \(X:=\Re(\xi)\in[-1,1]\)，并记实矩
\[
m_n:=\int_{\TT}(\Re\xi)^n\,\dd\mu(\xi)=\EE[X^n],\qquad n\ge 0.
\]
则对每个 \(n\ge 0\) 有显式二项反演
\[
\boxed{\quad
m_n=\sum_{k=0}^{n}(-1)^k\binom{n}{k}2^k\,a_k.\quad}
\]
并且对任意 \(n\ge 0\)，Toeplitz 系数满足
\[
\boxed{\quad
c_n:=\int_{\TT}\xi^{-n}\,\dd\mu(\xi)=\int_{\TT}\cos(n\theta)\,\dd\mu(e^{i\theta})
=\EE[T_n(X)],\quad}
\]
其中 \(T_n\) 为第一类 Chebyshev 多项式（\(T_n(\cos\theta)=\cos(n\theta)\)）。
因此序列 \((a_N)_{N\ge 0}\) 唯一决定全部 \((c_n)_{n\ge 0}\)，从而决定所有 Toeplitz 主子矩阵 \(\mathbf{T}_N=(c_{j-k})_{0\le j,k\le N}\)。
\end{theorem}
\begin{proof}
由 \(|1-e^{i\theta}|^2=2-2\cos\theta\) 得
\[
\frac{|1-\xi|^2}{4}=\frac{1-\Re(\xi)}{2}=\frac{1-X}{2}.
\]
因此
\[
a_N=\EE\!\left[\left(\frac{1-X}{2}\right)^{\!N}\right]
=2^{-N}\sum_{k=0}^{N}\binom{N}{k}(-1)^k\,\EE[X^k]
=2^{-N}\sum_{k=0}^{N}\binom{N}{k}(-1)^k m_k.
\]
令 \(b_N:=2^{N}a_N\)，则 \(b_N=\sum_{k\le N}\binom{N}{k}(-1)^k m_k\) 为标准二项变换；
二项反演给出
\(
m_n=\sum_{k=0}^{n}(-1)^k\binom{n}{k}b_k
=\sum_{k=0}^{n}(-1)^k\binom{n}{k}2^k a_k
\)，得第一式。
\par
由共轭不变性，\(\int_{\TT}\sin(n\theta)\,\dd\mu(e^{i\theta})=0\)，故
\(
c_n=\int_{\TT}e^{-in\theta}\dd\mu=\int_{\TT}\cos(n\theta)\dd\mu
\)。
再由 \(T_n(\cos\theta)=\cos(n\theta)\) 与 \(X=\cos\theta\) 得到第二式。证毕。
\end{proof}

\begin{theorem}[Adams 纤维端点的谱维数最小律]\label{thm:xi-adams-fiber-spectral-dimension-minimum}
令 \(\mu\) 为 \(\TT\) 上有限正 Borel 测度，并固定整数 \(d\ge 1\)。
记
\[
S_d:=\{\xi\in\TT:\ \xi^{d}=-1\},\qquad
m_d:=\mu(S_d),\qquad
a_N^{(d)}:=a_N^{(d)}(1)=\int_{\TT}\left(\frac{|1-\xi^{d}|^2}{4}\right)^{N}\dd\mu(\xi).
\]
假设对每个 \(\zeta\in S_d\)，存在指数 \(\alpha(\zeta)>0\) 与常数 \(0<c_-(\zeta)\le c_+(\zeta)<\infty\)，使得当 \(\varepsilon\downarrow 0\) 时
\[
c_-(\zeta)\,\varepsilon^{\alpha(\zeta)}
\le
\mu\bigl(\{\xi\in\TT:\ |\arg(\xi)-\arg(\zeta)|\le \varepsilon\}\bigr)
\le
c_+(\zeta)\,\varepsilon^{\alpha(\zeta)}.
\]
令 \(\alpha_{\min}:=\min_{\zeta\in S_d}\alpha(\zeta)\)。
则存在常数 \(0<C_-\le C_+<\infty\)（依赖于 \(\{c_\pm(\zeta)\}\) 与 \(d\)）使得
\[
\boxed{\quad
C_-\,N^{-\alpha_{\min}/2}\ \le\ a_N^{(d)}-m_d\ \le\ C_+\,N^{-\alpha_{\min}/2}\qquad(N\to\infty).\quad}
\]
\end{theorem}
\begin{proof}
将积分分解为 \(S_d\) 的邻域并集与其补集。
补集上 \(|1-\xi^{d}|<2\) 且由紧性可取 \(\delta>0\) 使 \(|1-\xi^{d}|^2/4\le 1-\delta\)，其贡献为 \(O((1-\delta)^N)\)。
\par
固定 \(\zeta\in S_d\)，写 \(\xi=e^{i(\theta_\zeta+h)}\)（\(\theta_\zeta:=\arg\zeta\)）。
则 \(\zeta^{d}=-1\) 给出
\[
\xi^{d}=e^{id\theta_\zeta}e^{idh}=(-1)e^{idh},
\qquad
\frac{|1-\xi^{d}|^2}{4}
=\frac{|1+e^{idh}|^2}{4}
=\cos^2\!\left(\frac{dh}{2}\right).
\]
因此存在 \(h_0>0\) 与常数 \(A_\pm>0\) 使当 \(|h|\le h_0\) 时
\[
\exp(-A_+N d^2 h^2)\ \le\ \left(\frac{|1-\xi^{d}|^2}{4}\right)^{N}\ \le\ \exp(-A_-N d^2 h^2).
\]
由两侧幂律假设，取 \(\varepsilon\asymp N^{-1/2}\) 得到下界
\[
a_N^{(d)}-m_d
\ge
\exp(-A_+d^2)\,\mu\bigl(\{|\arg(\xi)-\theta_\zeta|\le c/\sqrt{N}\}\bigr)
\gtrsim N^{-\alpha(\zeta)/2}.
\]
对满足 \(\alpha(\zeta)=\alpha_{\min}\) 的 \(\zeta\) 取上述下界并吸收常数即得左侧不等式。
\par
对上界，将 \(S_d\) 的 \(h_0\)-邻域分解为 \(|h|\in(2^{-(j+1)},2^{-j}]\) 的 dyadic 环带并求和，
结合
\(
\mu(|h|\le 2^{-j})\lesssim 2^{-j\alpha(\zeta)}
\)
与
\(
\exp(-A_-N d^2 h^2)\le \exp(-A_-Nd^2 2^{-2(j+1)})
\)
可得每个 \(\zeta\) 的邻域贡献被 \(O(N^{-\alpha(\zeta)/2})\) 控制；
再对有限集合 \(S_d\) 求和并用 \(\alpha(\zeta)\ge \alpha_{\min}\) 即得右侧不等式。证毕。
\end{proof}

\begin{corollary}[有限阶探针下的 \(2\)-进 Möbius 反演误差界]\label{cor:xi-adams-fiber-mobius-inversion-finiteN-error}
在定理 \ref{thm:xi-adams-fiber-mobius-inversion} 的设定下，固定 \(N\ge 0\) 并记
\[
\widehat m_d:=a_N^{(d)}(1),\qquad e_d:=\widehat m_d-m_d\ge 0.
\]
对 \(n=2^{a}u\)（\(u\) 奇）定义有限阶估计
\[
\widehat f(2^{a}u):=\sum_{v\mid u}\mu_{\mathrm{Mob}}(u/v)\,\widehat m_{2^{a}v}.
\]
则确定性误差界成立：
\[
\bigl|\widehat f(2^{a}u)-f(2^{a}u)\bigr|
\le
\sum_{v\mid u} e_{2^{a}v}.
\]
若进一步对每个 \(v\mid u\) 的补支撑谱半径满足
\[
r_\ast(2^{a}v,1)
:=\sup_{\xi\in\operatorname{supp}(\mu)\setminus\{\xi^{2^{a}v}=-1\}}\frac{|1-\xi^{2^{a}v}|^2}{4}\ <\ 1,
\]
则由推论 \ref{cor:xi-endpoint-heat-probe-adams-twist-spectral-gap-exp-bound} 得指数误差预算
\[
e_{2^{a}v}\le r_\ast(2^{a}v,1)^{\,N}\,\mu\bigl(\TT\setminus\{\xi^{2^{a}v}=-1\}\bigr).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-adams-fiber-mobius-inversion} 的线性反演，
\(
\widehat f-f=\sum_{v\mid u}\mu_{\mathrm{Mob}}(u/v)\,(\widehat m_{2^{a}v}-m_{2^{a}v})
\)。
取绝对值并用 \(|\mu_{\mathrm{Mob}}|\le 1\) 即得第一式。
第二式为推论 \ref{cor:xi-endpoint-heat-probe-adams-twist-spectral-gap-exp-bound}（取 \(\omega=1\)）逐项代入。证毕。
\end{proof}
