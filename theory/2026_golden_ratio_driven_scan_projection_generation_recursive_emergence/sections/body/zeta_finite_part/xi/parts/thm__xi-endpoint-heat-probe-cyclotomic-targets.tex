\paragraph{圆分方靶点的端点热核探针：任意 \texorpdfstring{$\omega\in\TT$}{omega in T} 的原子读出与补支撑谱半径}

\begin{theorem}[圆分方靶点热核探针的 Toeplitz 二次型实现与原子质量抽取]\label{thm:xi-endpoint-heat-probe-cyclotomic-target-extraction}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_cyclotomic\_target\_extraction}
沿用 \S\ref{subsubsec:xi-endpoint-heat-kernel-probe-toeplitz} 的 Fourier 系数 \(c_h\) 与 \(d\)-扭曲 Toeplitz 主子矩阵 \(\mathbf{T}_N^{(d)}=(c_{d(j-k)})_{0\le j,k\le N}\) 的记号（定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots}）。
对任意 \(\omega\in\TT\) 定义二项式探针向量
\[
b^{(N)}(\omega):=\bigl(b^{(N)}_k(\omega)\bigr)_{k=0}^{N}\in\CC^{N+1},
\qquad
b^{(N)}_k(\omega):=\binom Nk\,\omega^{k},
\]
并定义圆分方靶点的扭曲端点热核探针
\[
a_N^{(d)}(\omega):=4^{-N}\,\bigl(b^{(N)}(\omega)\bigr)^{\ast}\mathbf{T}_N^{(d)}\,b^{(N)}(\omega).
\]
则对所有 \(N\ge 0\) 成立严格恒等式
\[
\boxed{\;
a_N^{(d)}(\omega)
=\int_{\TT}\left(\frac{|1+\omega\,\xi^{d}|^{2}}{4}\right)^{\!N}\dd\mu(\xi)\ \in\ [0,\mu(\TT)]\;}
\]
并且序列 \(N\mapsto a_N^{(d)}(\omega)\) 单调不增。
记圆分方根集合
\[
\Sigma_{d,\omega}:=\{\xi\in\TT:\ \xi^{d}=\overline{\omega}\},
\qquad
m_\infty(d,\omega):=\mu(\Sigma_{d,\omega}),
\]
则有单调极限抽取公式
\[
\boxed{\;
a_N^{(d)}(\omega)\ \downarrow\ m_\infty(d,\omega)\qquad(N\to\infty).\;}
\]
特别地，当 \(\omega=-1\) 时，\(b^{(N)}(\omega)=b^{(N)}\) 且 \(a_N^{(d)}(\omega)=a_N^{(d)}\)，回到定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots} 的端点 \((-1)\) 情形。
\end{theorem}
\begin{proof}
由定义与 Fubini 交换求和积分，
\[
\bigl(b^{(N)}(\omega)\bigr)^{\ast}\mathbf{T}_N^{(d)}b^{(N)}(\omega)
=\sum_{j,k=0}^{N}\overline{b_j^{(N)}(\omega)}\,b_k^{(N)}(\omega)\,c_{d(j-k)}
=\int_{\TT}\left|\sum_{k=0}^{N}b_k^{(N)}(\omega)\,\xi^{dk}\right|^{2}\dd\mu(\xi).
\]
而
\(
\sum_{k=0}^{N}\binom Nk\,\omega^{k}\,\xi^{dk}=(1+\omega\,\xi^{d})^{N}
\)，
故上式等于 \(\int_{\TT}|1+\omega\xi^{d}|^{2N}\dd\mu(\xi)\)；除以 \(4^{N}\) 即得恒等式。
又因 \(|1+\omega\xi^{d}|\le 2\)，可知 integrand \(r_{d,\omega}(\xi)^N\)（其中 \(r_{d,\omega}(\xi):=|1+\omega\xi^{d}|^{2}/4\in[0,1]\)）随 \(N\) 单调不增，从而 \(a_N^{(d)}(\omega)\) 单调不增。
注意到 \(r_{d,\omega}(\xi)=1\) 当且仅当 \(\omega\xi^{d}=1\)，亦即 \(\xi^{d}=\overline{\omega}\)。
由单调收敛定理，
\(
\lim_{N\to\infty}a_N^{(d)}(\omega)=\mu(\{r_{d,\omega}=1\})=\mu(\Sigma_{d,\omega})
\)，
得到极限抽取公式。证毕。
\end{proof}

\begin{theorem}[由单标量探针序列识别补支撑谱半径]\label{thm:xi-endpoint-heat-probe-complement-radius-from-scalar-flow}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_complement\_radius\_from\_scalar\_flow}
在定理 \ref{thm:xi-endpoint-heat-probe-cyclotomic-target-extraction} 的设定下，定义尾矩序列
\[
t_N^{(d)}(\omega):=a_N^{(d)}(\omega)-m_\infty(d,\omega)
=\int_{\TT\setminus\Sigma_{d,\omega}}\left(\frac{|1+\omega\,\xi^{d}|^{2}}{4}\right)^{\!N}\dd\mu(\xi),
\]
以及补支撑谱半径
\[
r_{\ast}(d,\omega)
:=\sup_{\xi\in\operatorname{supp}(\mu)\setminus\Sigma_{d,\omega}}\frac{|1+\omega\,\xi^{d}|^{2}}{4}\in[0,1].
\]
则成立根极限公式
\[
\boxed{\;
\lim_{N\to\infty}\bigl(t_N^{(d)}(\omega)\bigr)^{1/N}=r_{\ast}(d,\omega).\;}
\]
因此，仅由标量流 \(N\mapsto a_N^{(d)}(\omega)\) 即可同时识别：
(i) 目标圆分方原子总质量 \(m_\infty(d,\omega)\)；
(ii) 补支撑的端点能量谱半径 \(r_\ast(d,\omega)\)（从而判别 \(r_\ast<1\) 的指数型收敛与 \(r_\ast=1\) 的非指数型情形）。
\end{theorem}
\begin{proof}
令 \(\varphi(\xi):=\frac{|1+\omega\xi^{d}|^{2}}{4}\in[0,1]\)，并令 \(\nu\) 为 \(\TT\setminus\Sigma_{d,\omega}\) 在 \(\varphi\) 下的推前测度：
\[
\nu(E):=\mu\bigl(\varphi^{-1}(E)\cap(\TT\setminus\Sigma_{d,\omega})\bigr)\qquad(E\subset[0,1]\ \text{Borel}).
\]
则
\(
t_N^{(d)}(\omega)=\int_{[0,1]} s^{N}\dd\nu(s)
\)。
由定义 \(\sup\supp(\nu)=r_{\ast}(d,\omega)\)。
上界：对 \(\supp(\nu)\subset[0,r_\ast]\) 有
\(
t_N^{(d)}(\omega)\le r_\ast(d,\omega)^N\,\nu([0,1])
\)，故 \(\limsup (t_N^{(d)}(\omega))^{1/N}\le r_\ast(d,\omega)\)。
\par
下界：任取 \(\varepsilon>0\)。由 \(r_\ast(d,\omega)\in\supp(\nu)\) 得 \(\nu((r_\ast(d,\omega)-\varepsilon,r_\ast(d,\omega)])>0\)。
因此
\[
t_N^{(d)}(\omega)\ge \int_{(r_\ast-\varepsilon,r_\ast]} s^N\dd\nu(s)
\ge (r_\ast(d,\omega)-\varepsilon)^N\,\nu((r_\ast-\varepsilon,r_\ast]),
\]
从而 \(\liminf (t_N^{(d)}(\omega))^{1/N}\ge r_\ast(d,\omega)-\varepsilon\)。
令 \(\varepsilon\downarrow 0\) 得 \(\liminf\ge r_\ast\)，与上界合并即得所示根极限。证毕。
\end{proof}
