\begin{theorem}[端点尾序列的 Hausdorff 矩闭包]\label{thm:xi-endpoint-heat-probe-hausdorff-hankel-closure}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_hausdorff\_hankel\_closure}
沿用定理 \ref{thm:xi-endpoint-heat-probe-rstar-regular-variation-tail} 的记号，记
\[
T_N:=a_N-m_{-1}=\int_{[0,r_\ast]} s^N\,\dd\nu(s),
\qquad
r_\ast=\sup\supp(\nu)\subset[0,1].
\]
则成立：
\begin{enumerate}
  \item \((T_N)_{N\ge 0}\) 是一个 Hausdorff 矩序列；
  \item 对任意整数 \(r\ge 0\)，Hankel 矩阵
  \[
  H^{(r)}:=(T_{i+j})_{0\le i,j\le r}
  \]
  半正定；
  \item \((T_N)\) 完全单调：
  \begin{equation}\label{eq:xi-endpoint-heat-probe-complete-monotone}
  \boxed{\;(-1)^k\,\Delta^k T_N\ge 0\qquad(N,k\ge 0);\;}
  \end{equation}
  \item \((T_N)\) 对数凸：
  \begin{equation}\label{eq:xi-endpoint-heat-probe-logconvex}
  \boxed{\;T_{N+1}^2\le T_NT_{N+2}\qquad(N\ge 0).\;}
  \end{equation}
\end{enumerate}
若进一步 \(T_N>0\) 对所有 \(N\ge 0\) 成立，并记
\[
\rho_N:=\frac{T_{N+1}}{T_N},
\]
则 \((\rho_N)\) 单调不减；再结合定理
\ref{thm:xi-endpoint-heat-probe-rstar-root-ratio}，得到
\[
\rho_N\uparrow r_\ast.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-endpoint-heat-probe-rstar-regular-variation-tail}，
\(
T_N=\int_{[0,r_\ast]} s^N\,\dd\nu(s)
\)
且 \(r_\ast\le 1\)。
因此 \((T_N)\) 是 \([0,1]\) 上有限正测度的矩序列，从而为 Hausdorff 矩序列。

对任意 \(c=(c_0,\dots,c_r)\in\CC^{r+1}\)，有
\[
\sum_{i,j=0}^{r}c_i\overline{c_j}T_{i+j}
=
\int_{[0,r_\ast]}
\left|\sum_{i=0}^{r}c_i s^i\right|^2\dd\nu(s)\ge 0,
\]
故 \(H^{(r)}\succeq 0\)。

再由差分算子表示，
\[
\Delta^k T_N
=
\int_{[0,r_\ast]} s^N(s-1)^k\,\dd\nu(s),
\]
于是
\[
(-1)^k\Delta^k T_N
=
\int_{[0,r_\ast]} s^N(1-s)^k\,\dd\nu(s)\ge 0,
\]
得到 \eqref{eq:xi-endpoint-heat-probe-complete-monotone}。

对数凸性既可由 \(2\times 2\) Hankel 主子式非负得到，也可直接用 Cauchy--Schwarz：
\[
T_{N+1}^2
=
\left(\int s^{N+1}\,\dd\nu\right)^2
\le
\left(\int s^N\,\dd\nu\right)\left(\int s^{N+2}\,\dd\nu\right)
=
T_NT_{N+2}.
\]
于是 \eqref{eq:xi-endpoint-heat-probe-logconvex} 成立。

若 \(T_N>0\) 对所有 \(N\) 成立，则
\[
\rho_{N+1}-\rho_N
=
\frac{T_{N+2}}{T_{N+1}}-\frac{T_{N+1}}{T_N}
=
\frac{T_NT_{N+2}-T_{N+1}^2}{T_NT_{N+1}}\ge 0,
\]
故 \((\rho_N)\) 单调不减。
最后由定理 \ref{thm:xi-endpoint-heat-probe-rstar-root-ratio} 的比值极限结论得
\(
\rho_N\uparrow r_\ast
\)。
证毕。
\end{proof}

\begin{theorem}[比值序列的一阶偏移恢复端点局部谱维数]\label{thm:xi-endpoint-heat-probe-ratio-recovers-local-spectral-dimension}
沿用定理 \ref{thm:xi-endpoint-heat-probe-rstar-regular-variation-tail} 的设定，并假设处于其第 \(2\) 种正则变差情形：
\[
\nu([r_\ast-\varepsilon,r_\ast])\sim C\,\varepsilon^{\alpha+1}
\qquad(\varepsilon\downarrow 0),
\qquad C>0,\ \alpha>-1.
\]
记
\[
\rho_N:=\frac{T_{N+1}}{T_N}.
\]
则有精确一阶展开
\[
\boxed{\;
\frac{\rho_N}{r_\ast}
=
1-\frac{\alpha+1}{N}+o(N^{-1})
\qquad(N\to\infty).\;}
\]
从而
\[
\boxed{\;
\alpha+1
=
\lim_{N\to\infty}N\left(1-\frac{\rho_N}{r_\ast}\right).\;}
\]
因此端点局部谱维数指数 \(\alpha\) 可由单一比值序列 \((\rho_N)\) 的一阶偏移完全恢复。
\end{theorem}
\leanverified{paper\_xi\_endpoint\_heat\_probe\_ratio\_recovers\_local\_spectral\_dimension}
\begin{proof}
由定理 \ref{thm:xi-endpoint-heat-probe-rstar-regular-variation-tail}，
\[
T_N\sim A\,r_\ast^N\,N^{-(\alpha+1)},
\qquad
A:=C\,\Gamma(\alpha+2)\,r_\ast^{\alpha+1}.
\]
于是
\[
\rho_N=\frac{T_{N+1}}{T_N}
\sim
r_\ast\left(\frac{N}{N+1}\right)^{\alpha+1}
=
r_\ast\left(1+\frac{1}{N}\right)^{-(\alpha+1)}.
\]
对右端作一阶展开，
\[
\left(1+\frac{1}{N}\right)^{-(\alpha+1)}
=
1-\frac{\alpha+1}{N}+o(N^{-1}),
\]
便得到所示公式。再移项并取极限，即得第二式。证毕。
\end{proof}

\begin{theorem}[前 \texorpdfstring{$N{+}1$}{N+1} 个端点探针对 Toeplitz 主子矩阵的有限充分性与最优性]\label{thm:xi-endpoint-first-n-plus-one-probes-sharp-toeplitz-recovery}
\leanverified{paper\_xi\_endpoint\_first\_n\_plus\_one\_probes\_sharp\_toeplitz\_recovery}
在定理 \ref{thm:xi-selfdual-single-probe-determines-toeplitz-moments} 的共轭不变设定下，固定 \(N\ge 0\)。
则前 \(N+1\) 个端点探针值
\[
(a_0,a_1,\dots,a_N)
\]
已唯一决定 Toeplitz 主子矩阵
\[
\mathbf{T}_N:=(c_{j-k})_{0\le j,k\le N}.
\]
更具体地，存在两级上三角可逆变换
\[
(a_0,\dots,a_N)
\Longleftrightarrow
(m_0,\dots,m_N)
\Longleftrightarrow
(c_0,\dots,c_N),
\]
其中
\[
m_n:=\int_{\TT}(\Re\xi)^n\,\dd\mu(\xi),
\qquad
c_n:=\int_{\TT}\xi^{-n}\,\dd\mu(\xi).
\]
并且该采样数是最优的：对任意 \(r<N+1\)，存在两组共轭不变有限正测度 \(\mu_+\neq\mu_-\)，使得它们的前 \(r\) 个端点探针值相同，
\[
a_k(\mu_+)=a_k(\mu_-)\qquad(0\le k\le r-1),
\]
但对应的 Toeplitz 主子矩阵满足
\[
\mathbf{T}_N(\mu_+)\neq \mathbf{T}_N(\mu_-).
\]
因此，少于 \(N+1\) 个端点探针值在一般情形下不足以唯一恢复 \(\mathbf{T}_N\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-selfdual-single-probe-determines-toeplitz-moments}，对每个 \(0\le n\le N\) 都有二项反演
\[
m_n=\sum_{k=0}^{n}(-1)^k\binom{n}{k}2^k a_k.
\]
故映射
\[
(a_0,\dots,a_N)\longmapsto (m_0,\dots,m_N)
\]
由一个上三角矩阵给出，其对角元为
\(
(-2)^n\neq 0
\)，因而可逆。

另一方面，第一类 Chebyshev 多项式满足
\[
T_n(x)=2^{n-1}x^n+\sum_{k=0}^{n-1}\tau_{n,k}x^k
\qquad(n\ge 1),
\]
且 \(T_0(x)=1\)。
由
\[
c_n=\EE[T_n(X)],
\qquad X:=\Re\xi,
\]
可得
\[
c_n=
\begin{cases}
m_0, & n=0,\\[4pt]
2^{n-1}m_n+\sum_{k=0}^{n-1}\tau_{n,k}m_k, & 1\le n\le N.
\end{cases}
\]
因此映射
\[
(m_0,\dots,m_N)\longmapsto (c_0,\dots,c_N)
\]
同样是上三角可逆变换。由于共轭不变性给出 \(c_{-n}=c_n\)，故 \(\mathbf{T}_N\) 完全由 \((c_0,\dots,c_N)\) 唯一决定，从而前 \(N+1\) 个探针值已足以恢复 \(\mathbf{T}_N\)。

再证最优性。任取 \(r<N+1\)，并选取互异点
\[
x_0,\dots,x_N\in(-1,1).
\]
定义线性映射 \(\mathcal P_r:\RR^{N+1}\to\RR^{r}\) 为
\[
(\mathcal P_r w)_k:=\sum_{j=0}^{N}w_j\left(\frac{1-x_j}{2}\right)^k,
\qquad 0\le k\le r-1.
\]
由于 \(r<N+1\)，\(\ker(\mathcal P_r)\neq\{0\}\)；取非零向量 \(u\in\ker(\mathcal P_r)\)。
再取任意严格正权重 \(w=(w_0,\dots,w_N)\in(0,\infty)^{N+1}\)，并选取 \(\varepsilon>0\) 足够小，使 \(w\pm\varepsilon u\) 逐项仍为正。

令 \(\theta_j:=\arccos x_j\)，并定义两组共轭不变有限正测度
\[
\mu_\pm:=
\sum_{j=0}^{N}(w_j\pm\varepsilon u_j)\,
\frac{\delta_{e^{i\theta_j}}+\delta_{e^{-i\theta_j}}}{2}.
\]
则对 \(0\le k\le r-1\)，有
\[
a_k(\mu_+)-a_k(\mu_-)
=
2\varepsilon\sum_{j=0}^{N}u_j\left(\frac{1-x_j}{2}\right)^k
=
2\varepsilon\,(\mathcal P_r u)_k
=
0,
\]
故前 \(r\) 个探针值完全一致。

若再有 \(\mathbf T_N(\mu_+)=\mathbf T_N(\mu_-)\)，则 \((c_0,\dots,c_N)\) 相同，从而由上面两级可逆三角变换可知 \((m_0,\dots,m_N)\) 亦相同。
但
\[
m_n(\mu_+)-m_n(\mu_-)=2\varepsilon\sum_{j=0}^{N}u_jx_j^n
\qquad(0\le n\le N).
\]
由于 Vandermonde 矩阵 \((x_j^n)_{0\le n,j\le N}\) 可逆，存在某个 \(0\le n\le N\) 使上式非零，矛盾。
因此 \(\mathbf T_N(\mu_+)\neq \mathbf T_N(\mu_-)\)。
这就证明了少于 \(N+1\) 个探针值时的一般不可唯一恢复性。证毕。
\end{proof}
