\begin{proposition}[误差预算支配读出分布的全变差]\label{prop:dephys-eps-sound-dominates-tv}
\leanverified{paper\_dephys\_eps\_sound\_dominates\_tv}
固定接口对象 \(\mathcal I\) 及其参考抽样分布 \(\mathbb{P}_{\mathcal I}\)（定义 \ref{def:pom-eps-sound-rewrite}）。
对任意同源同靶的投影词 \(w,w'\)，记其诱导可见读出为随机变量 \(Y_w,Y_{w'}\)，并令
\[
\mu_w:=(Y_w)_*\mathbb{P}_{\mathcal I},
\qquad
\mu_{w'}:=(Y_{w'})_*\mathbb{P}_{\mathcal I}.
\]
若存在 \(\varepsilon\)-sound 证书 \(w\Longrightarrow_{\varepsilon}w'\)，则
\[
\boxed{\;\|\mu_w-\mu_{w'}\|_{\mathrm{TV}}\le \varepsilon.\;}
\]
\end{proposition}
\begin{proof}
由定义 \ref{def:pom-eps-sound-rewrite}，存在坏事件 \(E\) 使 \(\mathbb{P}_{\mathcal I}(E)\le \varepsilon\)，且在 \(E^c\) 上有 \(Y_w=Y_{w'}\) 几乎处处成立。
对任意可测集合 \(A\) 于读出空间，有
\[
\mu_w(A)-\mu_{w'}(A)=\mathbb{P}_{\mathcal I}(Y_w\in A)-\mathbb{P}_{\mathcal I}(Y_{w'}\in A)
=\mathbb{E}_{\mathbb{P}_{\mathcal I}}\!\bigl[\ind_{\{Y_w\in A\}}-\ind_{\{Y_{w'}\in A\}}\bigr].
\]
括号内差异仅可能在 \(E\) 上非零，故
\[
\bigl|\mu_w(A)-\mu_{w'}(A)\bigr|
\le \mathbb{P}_{\mathcal I}(E)\le \varepsilon.
\]
对 \(A\) 取上确界即得 \(\|\mu_w-\mu_{w'}\|_{\mathrm{TV}}\le \varepsilon\)。证毕。
\end{proof}

\begin{corollary}[最小预算距离支配全变差，且给出总预算的不可改进下界]\label{cor:dephys-budget-distance-dominates-tv}
\leanverified{paper\_dephys\_budget\_distance\_dominates\_tv}
沿用命题 \ref{prop:dephys-eps-sound-dominates-tv} 的设定，并令 \(d_{\mathcal I}\) 为最小预算距离（定义 \ref{def:pom-eps-distance}）。
则对任意同源同靶的 \(w,w'\) 恒有
\[
\boxed{\;\|\mu_w-\mu_{w'}\|_{\mathrm{TV}}\le d_{\mathcal I}(w,w').\;}
\]
并且对任意有限证书链
\(
w=w_0\Longrightarrow_{\varepsilon_1}w_1\Longrightarrow_{\varepsilon_2}\cdots\Longrightarrow_{\varepsilon_r}w_r=w'
\)
总预算满足信息论下界
\[
\|\mu_w-\mu_{w'}\|_{\mathrm{TV}}\le \sum_{i=1}^r\varepsilon_i.
\]
因而连接 \(w\) 与 \(w'\) 的所有有限证书链的总预算下确界不小于 \(\|\mu_w-\mu_{w'}\|_{\mathrm{TV}}\)。
\end{corollary}
\begin{proof}
第一式由命题 \ref{prop:dephys-eps-sound-dominates-tv} 对任意可行 \(\varepsilon\) 成立并对 \(\varepsilon\) 取下确界得到。
对证书链，先由命题 \ref{prop:pom-eps-sound-compose} 得到 \(w\Longrightarrow_{\sum_i\varepsilon_i}w'\)，再应用命题 \ref{prop:dephys-eps-sound-dominates-tv} 即得第二式。
最后对所有链取下确界得到总预算下界。证毕。
\end{proof}

\begin{proposition}[最小预算距离的逐点最大性（泛性质）]\label{prop:dephys-budget-distance-universal-property}
\leanverified{paper\_dephys\_budget\_distance\_universal\_property}
固定接口 \(\mathcal I\)，并令 \(\mathsf D\) 为所有满足下述条件的函数 \(\delta:\mathbf{PW}(A,B)\times\mathbf{PW}(A,B)\to\overline{\RR}_{\ge 0}\) 的集合：
\begin{enumerate}
  \item \(\delta(w,w)=0\)；
  \item 若存在 \(\varepsilon\)-sound 证书 \(w\Longrightarrow_\varepsilon w'\)，则 \(\delta(w,w')\le \varepsilon\)。
\end{enumerate}
则对任意 \(\delta\in\mathsf D\) 与任意 \(w,w'\) 恒有
\[
\boxed{\;\delta(w,w')\le d_{\mathcal I}(w,w').\;}
\]
因此 \(d_{\mathcal I}\) 是满足上述两条约束的函数族中的逐点最大元。
\end{proposition}
\begin{proof}
由条件 (2)，\(\delta(w,w')\) 不超过所有可行预算 \(\varepsilon\)；对 \(\varepsilon\) 取下确界即得
\(
\delta(w,w')\le d_{\mathcal I}(w,w')
\)。
证毕。
\end{proof}
