\paragraph{三次余因子判别式核、分歧纤维剥离与倍点轨道刚性}\label{par:xi-terminal-zm-cubic-cofactor-discriminant-kernel-s3-rigidity-height}
沿用定理 \ref{thm:xi-terminal-zm-leyang-elliptic-structure} 的谱四次与椭圆化记号：
\[
\Pi(\lambda,y)=\lambda^4-\lambda^3-(2y+1)\lambda^2+\lambda+y(y+1),
\]
\[
\mathrm{PLY}(y):=256y^3+411y^2+165y+32=P_{\mathrm{LY}}(y),
\qquad
B(y):=y(y-1)\mathrm{PLY}(y),
\]
\[
\mathcal E:\ w^2=4x^3-4x+1,\qquad
y=x^2+\frac{w-1}{2}.
\]
并记 \(Y:=w/2\)、\(\omega:=\dd x/(2Y)\)。于是
\[
\dd y=(4xY+3x^2-1)\omega,\qquad
\Disc_{\lambda}(\Pi)=-B(y).
\]

\begin{theorem}[三次余因子判别式的导数平方消去公式]\label{thm:xi-terminal-zm-cubic-cofactor-disc-derivative-square-elimination}
设 \(P\in \mathcal E(\QQ)\) 满足
\[
y(P)\notin \{0,1\}\cup\{\mathrm{PLY}=0\},
\]
且 \(P\) 不是 \(y\) 的分歧点（等价于 \(\dd y(P)\neq 0\)）。记
\[
x(P)=x_0,\qquad w(P)=w_0,\qquad y(P)=y_0.
\]
则 \(\Pi(\lambda,y_0)\) 在 \(\lambda=x_0\) 处为单根，故存在唯一首一三次多项式 \(C_P(\lambda)\in\QQ[\lambda]\) 使
\[
\Pi(\lambda,y_0)=(\lambda-x_0)\,C_P(\lambda).
\]
并且有恒等式
\[
\Disc\!\bigl(C_P\bigr)
=\frac{-B(y_0)}{(2x_0w_0+3x_0^2-1)^2}
=\frac{-y_0(y_0-1)\mathrm{PLY}(y_0)}{(2x_0w_0+3x_0^2-1)^2}.
\]
特别地，
\[
\QQ\!\left(\sqrt{\Disc(C_P)}\right)=\QQ\!\left(\sqrt{-B(y_0)}\right).
\]
\leanverified{paper\_xi\_terminal\_zm\_cubic\_cofactor\_disc\_derivative\_square\_elimination}
\end{theorem}
\begin{proof}
由 \(\Pi(\lambda,y)=0\) 与椭圆同构 \((\lambda,y)=(x,\,x^2+\frac{w-1}{2})\)，可知 \(\lambda=x_0\) 为 \(\Pi(\lambda,y_0)\) 的有理根。

由
\[
\dd y=(4xY+3x^2-1)\omega,\qquad Y=\frac{w}{2},
\]
且 \(\dd y(P)\neq 0\)，得
\[
2x_0w_0+3x_0^2-1\neq 0.
\]
另一方面，将 \(w=2y+1-2x^2\) 代入
\[
4\Pi(x,y)=w^2-(4x^3-4x+1),
\]
在“保持 \(y\) 不变”的偏导口径下计算得
\[
\partial_{\lambda}\Pi(x,y)=-2xw-3x^2+1.
\]
故
\[
\partial_{\lambda}\Pi(x_0,y_0)=-(2x_0w_0+3x_0^2-1)\neq 0,
\]
从而 \(x_0\) 为单根，唯一首一三次余因子 \(C_P\) 存在。

设 \(f(\lambda):=\Pi(\lambda,y_0)=(\lambda-x_0)C_P(\lambda)\)。
由于 \((\lambda-x_0)\) 与 \(C_P\) 互素，判别式乘法公式给出
\[
\Disc(f)=\Disc(\lambda-x_0)\Disc(C_P)\operatorname{Res}(\lambda-x_0,C_P)^2
=\Disc(C_P)\,C_P(x_0)^2.
\]
并且 \(f' (x_0)=C_P(x_0)\)，故
\[
\Disc(C_P)=\frac{\Disc(f)}{(f'(x_0))^2}
=\frac{\Disc_{\lambda}(\Pi(\cdot,y_0))}{\bigl(\partial_{\lambda}\Pi(x_0,y_0)\bigr)^2}.
\]
代入 \(\Disc_{\lambda}(\Pi)=-B(y)\) 及上式导数表达即得
\[
\Disc(C_P)=\frac{-B(y_0)}{(2x_0w_0+3x_0^2-1)^2}.
\]
最后一条二次域恒等式由平方类不变性立得。
\end{proof}

\begin{theorem}[判别式核函数的除子结构与分歧纤维剥离]\label{thm:xi-terminal-zm-discriminant-kernel-divisor-branch-fiber-peeling}
在函数域 \(\QQ(\mathcal E)\) 中定义
\[
\Delta:=\frac{-B(y)}{(\dd y/\omega)^2}
=\frac{-y(y-1)\mathrm{PLY}(y)}{(4xY+3x^2-1)^2}.
\]
则 \(\Delta\) 为 \(\mathcal E\) 上亚纯函数，并满足
\[
\operatorname{div}(\Delta)=\sum_{Q\in U}[Q]-10[O],
\]
其中 \(O\) 为无穷远点，且
\[
U:=y^{-1}\!\bigl(\{0,1\}\cup\{\mathrm{PLY}=0\}\bigr)\setminus \operatorname{Ram}(y).
\]
并且每个有限分歧值纤维恰有两点落在 \(U\) 中，故 \(|U|=10\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-terminal-zm-leyang-elliptic-structure}，\(y:\mathcal E\to\PP^1\) 的有限分歧值集合为
\[
\{0,1\}\cup\{\mathrm{PLY}=0\},
\]
且每个有限分歧值纤维均为 \((2,1,1)\) 型：一处简单分歧点加两处非分歧点。

设 \(y_*\) 为任一有限分歧值，\(R\in y^{-1}(y_*)\) 为简单分歧点，\(Q_1,Q_2\) 为非分歧点。
在 \(Q_i\) 处取局部参数 \(t\)，有 \(y-y_*\sim t\)。
由于 \(B(y)\) 在 \(y_*\) 处为单零点，得 \(\operatorname{ord}_{Q_i}(B(y))=1\)；
又 \((\dd y/\omega)(Q_i)\neq 0\)，故
\[
\operatorname{ord}_{Q_i}(\Delta)=1.
\]
在分歧点 \(R\) 处有 \(y-y_*\sim t^2\)，故 \(\operatorname{ord}_{R}(B(y))=2\)；
同时 \(\dd y/\omega\sim t\)，故 \(\operatorname{ord}_{R}((\dd y/\omega)^2)=2\)。
二者相消，\(\operatorname{ord}_{R}(\Delta)=0\)。

再看无穷远点 \(O\)。在短 Weierstrass 坐标中
\[
\operatorname{ord}_{O}(x)=-2,\qquad \operatorname{ord}_{O}(Y)=-3.
\]
因此
\[
\operatorname{ord}_{O}(y)=-4,\qquad
\operatorname{ord}_{O}(B(y))=-20,
\]
且
\[
\frac{\dd y}{\omega}=4xY+3x^2-1
\]
主导项为 \(xY\)，故 \(\operatorname{ord}_{O}(\dd y/\omega)=-5\)，从而
\[
\operatorname{ord}_{O}\!\bigl((\dd y/\omega)^2\bigr)=-10,\qquad
\operatorname{ord}_{O}(\Delta)=-10.
\]
除上述点外，分子分母皆非零有限，故无其他零极点。于是
\[
\operatorname{div}(\Delta)=\sum_{Q\in U}[Q]-10[O].
\]
有限分歧值共有 \(5\) 个，每个提供 \(2\) 个非分歧点，故 \(|U|=10\)。
\end{proof}

\begin{theorem}[椭圆倍点轨道上的三次余因子 \texorpdfstring{$S_3$}{S3} 刚性与有限例外]\label{thm:xi-terminal-zm-cubic-cofactor-s3-rigidity-finite-exception}
设 \(P_0\in \mathcal E(\QQ)\) 为无限阶点，\(P_n:=[n]P_0\)。
对满足 \(\dd y(P_n)\neq 0\) 的整数 \(n\)，定义
\[
C_n(\lambda):=\frac{\Pi(\lambda,y(P_n))}{\lambda-x(P_n)}\in\QQ[\lambda].
\]
则存在有限集合 \(S\subset\ZZ\)，使得对一切 \(n\notin S\) 有：
\begin{enumerate}
\item \(C_n(\lambda)\) 在 \(\QQ\) 上不可约；
\item \(\operatorname{Gal}(C_n/\QQ)\cong S_3\)。
\end{enumerate}
\end{theorem}
\begin{proof}
先处理“不可约但伽罗瓦群为循环三次”的例外。
对不可约三次多项式，伽罗瓦群为 \(A_3\cong C_3\) 当且仅当判别式为平方。
由定理 \ref{thm:xi-terminal-zm-cubic-cofactor-disc-derivative-square-elimination}，
\[
\Disc(C_n)=\Delta(P_n)\qquad(\dd y(P_n)\neq 0).
\]
考虑二重覆盖
\[
\widetilde{\mathcal C}:\ z^2=\Delta
\]
（底为 \(\mathcal E\)）。
由定理 \ref{thm:xi-terminal-zm-discriminant-kernel-divisor-branch-fiber-peeling}，
\(\Delta\) 的奇阶零点恰有 \(10\) 个，且在 \(O\) 处极点阶为 \(10\)（偶数，不分歧）。
故 \(\widetilde{\mathcal C}\to \mathcal E\) 仅在这 \(10\) 点分歧。
Riemann--Hurwitz 公式给出
\[
g(\widetilde{\mathcal C})=2g(\mathcal E)-1+\frac{10}{2}=6.
\]
由 Faltings 定理，\(\widetilde{\mathcal C}(\QQ)\) 有限，因此满足 \(\Disc(C_n)\in\QQ^{\times2}\) 的 \(n\) 仅有有限个。

再处理“可约”的例外。
若 \(C_n\) 可约，则 \(\Pi(\lambda,y(P_n))\) 至少有两个不同有理根，即存在
\[
Q_n\in \mathcal E(\QQ),\qquad Q_n\neq P_n,\qquad y(Q_n)=y(P_n).
\]
令
\[
\mathcal Z:=\{(P,Q)\in \mathcal E\times \mathcal E:\ y(P)=y(Q)\},
\qquad
\Gamma:=\langle (P_0,O),(O,P_0)\rangle\subset (\mathcal E\times\mathcal E)(\QQ).
\]
若可约情形无穷多，则 \(\mathcal Z\cap\Gamma\) 无穷。
由 Mordell--Lang，\(\mathcal Z\) 含有某个代数子群平移分支；
等价地，存在同态 \(\varphi:\mathcal E\to\mathcal E\) 与点 \(T\) 使
\[
y(P)\equiv y(\varphi(P)+T)
\]
在 \(\QQ(\mathcal E)\) 中恒等。
比较次数得
\[
\deg(y)=\deg(y\circ(\varphi+T))=\deg(y)\deg(\varphi),
\]
故 \(\deg(\varphi)=1\)，即 \(\varphi\) 为自同构。
于是 \(u:=\varphi+T\) 满足 \(y\circ u=y\)，故 \(u\in \operatorname{Aut}_{\QQ(y)}(\QQ(\mathcal E))\)。
而定理 \ref{thm:xi-terminal-zm-leyang-elliptic-structure} 给出四次覆盖的单值群为 \(S_4\)；
对对应稳定子 \(H\simeq S_3\)，有
\[
\operatorname{Aut}_{\QQ(y)}(\QQ(\mathcal E))
\cong N_{S_4}(H)/H=1.
\]
故仅有 \(u=\mathrm{id}\)，与 \(Q_n\neq P_n\) 矛盾。
因此可约 \(n\) 也仅有限多。

合并两部分并并入 \(\dd y(P_n)=0\) 的有限例外（分歧点集有限），得到所述有限集合 \(S\)。
对 \(n\notin S\)，\(C_n\) 不可约且判别式非平方，故伽罗瓦群只能为 \(S_3\)。
\end{proof}

\begin{corollary}[二次脊柱在一次剥离下的不变性]\label{cor:xi-terminal-zm-cubic-cofactor-quadratic-spine-invariance}
在定理 \ref{thm:xi-terminal-zm-cubic-cofactor-s3-rigidity-finite-exception} 的设定下，对任意满足 \(\dd y(P_n)\neq 0\) 且 \(C_n\) 可分离的 \(n\)，其分裂域的二次分解子域恒为
\[
\QQ\!\left(\sqrt{-B(y(P_n))}\right)
=\QQ\!\left(\sqrt{-y(P_n)\bigl(y(P_n)-1\bigr)\mathrm{PLY}(y(P_n))}\right),
\]
与被剥离的一次因子 \(\lambda-x(P_n)\) 无关。
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-terminal-zm-cubic-cofactor-disc-derivative-square-elimination}，
\[
\Disc(C_n)=\frac{-B(y(P_n))}{\square},
\]
故 \(\Disc(C_n)\) 与 \(-B(y(P_n))\) 在 \(\QQ^\times/\QQ^{\times2}\) 中同类。
而可分离三次扩张的唯一二次分解子域由判别式平方类决定，结论即得。
\end{proof}

\begin{theorem}[判别式核函数沿倍点序列的二次型高度增长]\label{thm:xi-terminal-zm-discriminant-kernel-height-quadratic-growth}
设 \(P_0\in \mathcal E(\QQ)\) 为无限阶点，\(P_n=[n]P_0\)。
记 \(\hat h\) 为 \(\mathcal E\) 上 N\'eron--Tate 高度，\(h\) 为绝对对数高度。
则对函数 \(\Delta\) 有
\[
h\!\bigl(\Delta(P_n)\bigr)=10\,\hat h(P_n)+O(1)
=10\,n^2\hat h(P_0)+O(1)\qquad (n\to\infty).
\]
因而在对数高度尺度下，\(\Delta(P_n)\) 呈严格二次指数增长。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-terminal-zm-discriminant-kernel-divisor-branch-fiber-peeling},
\[
\operatorname{div}(\Delta)=\sum_{Q\in U}[Q]-10[O],
\]
故 \(\Delta:\mathcal E\to\PP^1\) 的次数为 \(10\)（即在 \(O\) 处极点阶 \(10\)）。
对椭圆曲线上的有理函数应用高度比较
\[
h(f(P))=\deg(f)\,\hat h(P)+O(1)
\]
（参见结论 \ref{con:xi-terminal-zm-elliptic-height-tunneling-degree-rigidity} 的同型用法），取 \(f=\Delta\) 即得
\[
h(\Delta(P_n))=10\,\hat h(P_n)+O(1).
\]
再用 \(\hat h([n]P_0)=n^2\hat h(P_0)\) 得第二式。
\end{proof}

\begin{conjecture}[判别式平方类注入性与二次脊柱的几乎处处互异]\label{conj:xi-terminal-zm-cubic-cofactor-disc-squareclass-injectivity}
在定理 \ref{thm:xi-terminal-zm-cubic-cofactor-s3-rigidity-finite-exception} 的设定下，记
\[
D_n:=\Disc(C_n)\in\QQ^\times.
\]
猜想除去有限多个 \(n\) 外，平方类 \(D_n\in \QQ^\times/\QQ^{\times2}\) 两两不同；
等价地，二次域
\[
\QQ(\sqrt{D_n})
\]
随 \(n\) 变化时几乎处处互异。

该猜想可重述为：对 \(m\neq n\)，关系
\[
\frac{D_m}{D_n}\in\QQ^{\times2}
\]
仅在有限多对 \((m,n)\) 上成立；几何上对应于 \(\mathcal E\times\mathcal E\) 上由判别式核函数比值定义的二重覆盖的有理点有限性问题，预期受 Faltings/Mordell--Lang 机制约束。
\end{conjecture}
