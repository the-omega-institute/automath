\paragraph{外幂层的 Schubert 精细化：\texorpdfstring{$\Lambda^k C_q$}{Lambda^k Cq} 的 \texorpdfstring{$\pi_\varphi$}{pi}-指数谱与对偶结构}
沿用本段记号：\(\mathcal O=\ZZ[\varphi]\)、\(\pi_\varphi=\varphi+2\)、
\(
C_q\in\Mat_{q+1}(\mathcal O)
\)
以及其阶梯 Smith 正规形（结论 \ref{con:xi-symq-golden-cq-smith-staircase}）。
固定 \(1\le k\le q+1\)，并令 \(L:=\mathcal O^{q+1}\)。
考虑外幂诱导同态
\[
\Lambda^k C_q:\ \Lambda^k L\longrightarrow \Lambda^k L,
\]
其矩阵尺寸为 \(N\times N\)（\(N=\binom{q+1}{k}\)）。

\begin{theorem}[外幂 Smith--Schubert 参数化]\label{thm:xi-exterior-power-smith-schubert}
\leanverified{paper\_xi\_exterior\_power\_smith\_schubert}
设 \(q\ge 2\) 且 \(1\le k\le q+1\)，令 \(N=\binom{q+1}{k}\)。
则存在非降指数序列 \(0\le e_1\le\cdots\le e_N\) 使得
\[
\mathrm{SNF}_{\mathcal O}(\Lambda^k C_q)=\mathrm{diag}\bigl(\pi_\varphi^{e_1},\dots,\pi_\varphi^{e_N}\bigr),
\]
并且 \(\{e_1,\dots,e_N\}\)（按多重度计）恰为 \(k\)-子集和多重集
\[
\Bigl\{\ \sum_{i\in S} i:\ S\subseteq\{0,1,\dots,q\},\ |S|=k\ \Bigr\}.
\]
等价地，令 \(a:=q+1-k\) 并对每个 \(k\)-子集 \(S=\{i_1<\cdots<i_k\}\) 定义分拆
\[
\lambda(S)=(\lambda_1\ge\cdots\ge \lambda_k\ge 0),
\qquad
\lambda_j:=i_{k+1-j}-(k-j),
\]
则 \(\lambda(S)\subseteq a^k\) 且
\[
\sum_{i\in S} i=\frac{k(k-1)}{2}+|\lambda(S)|.
\]
因此对余核 \(M_{q,k}:=\mathrm{coker}_{\mathcal O}(\Lambda^k C_q)\) 有 \(\mathcal O\)-模同构
\[
\boxed{\;
M_{q,k}\ \cong\ \bigoplus_{\lambda\subseteq a^k}\ \mathcal O/(\pi_\varphi^{\,\frac{k(k-1)}{2}+|\lambda|})\; }.
\]
\end{theorem}
\begin{proof}
由结论 \ref{con:xi-symq-golden-cq-smith-staircase}，存在幺模矩阵 \(Q_q,U_q\in\mathrm{GL}_{q+1}(\mathcal O)\) 使
\(
C_q=Q_qD_qU_q
\)，其中 \(D_q=\mathrm{diag}(\pi_\varphi^0,\pi_\varphi^1,\dots,\pi_\varphi^q)\)。
取外幂得
\[
(\Lambda^k Q_q)(\Lambda^k C_q)(\Lambda^k U_q)=\Lambda^k D_q.
\]
由于 \(\Lambda^k Q_q,\Lambda^k U_q\) 在 \(\Lambda^k L\) 上可逆且其行列式为单位元的幂，故仍为幺模矩阵，从而 Smith 类仅由 \(\Lambda^k D_q\) 决定。
\par
在标准基 \(e_0,\dots,e_q\) 下 \(D_qe_i=\pi_\varphi^{i}e_i\)，故在外幂基
\(
e_{i_1}\wedge\cdots\wedge e_{i_k}
\)
上有
\[
(\Lambda^k D_q)(e_{i_1}\wedge\cdots\wedge e_{i_k})
=\pi_\varphi^{i_1+\cdots+i_k}(e_{i_1}\wedge\cdots\wedge e_{i_k}),
\]
于是对角元的指数多重集恰为 \(k\)-子集和多重集；将指数按非降顺序重排即得到 Smith 正规形。
\par
对分拆参数化，映射 \(S=\{i_1<\cdots<i_k\}\mapsto\lambda(S)\) 与 \(i_j=\lambda_{k+1-j}+j-1\) 互为逆，
并且
\(
|\lambda(S)|=\sum_{j=1}^{k}(\lambda_j)=\sum_{j=1}^{k}(i_j-(j-1))=\sum_{i\in S}i-\frac{k(k-1)}{2}
\)。
矩形约束 \(\lambda(S)\subseteq a^k\) 等价于 \(i_k\le q\)。
最后，Smith 正规形直接给出余核分解。证毕。
\end{proof}

\begin{corollary}[Poincar\'e 对偶下的镜像配对]\label{cor:xi-exterior-power-poincare-duality}
\leanverified{paper\_xi\_exterior\_power\_poincare\_duality}
设 \(a=q+1-k\) 并对 \(\lambda\subseteq a^k\) 定义矩形补分拆
\[
\lambda^\vee_i:=a-\lambda_{k+1-i}\qquad(1\le i\le k).
\]
令 \(e(\lambda):=\frac{k(k-1)}{2}+|\lambda|\)。
则对一切 \(\lambda\subseteq a^k\) 成立
\[
e(\lambda)+e(\lambda^\vee)=kq.
\]
并且若将定理 \ref{thm:xi-exterior-power-smith-schubert} 的指数序列写为
\(
e_1\le\cdots\le e_N
\)，则满足严格镜像配对
\[
e_i+e_{N+1-i}=kq\qquad(1\le i\le N).
\]
\end{corollary}
\begin{proof}
矩形补分拆满足面积守恒 \( |\lambda|+|\lambda^\vee|=ka=k(q+1-k)\)。
代入 \(e(\lambda)\) 得
\[
e(\lambda)+e(\lambda^\vee)
=k(k-1)+ka
=k(k-1)+k(q+1-k)
=kq.
\]
由于 \(\lambda\mapsto\lambda^\vee\) 为集合上的自反对合并将 \(e(\lambda)\) 送到 \(kq-e(\lambda)\)，指数多重集在 \(r\mapsto kq-r\) 下封闭；
将其按非降序排列即得到镜像配对恒等式。证毕。
\end{proof}

\begin{proposition}[余核的 \texorpdfstring{$\mathcal O/(\pi_\varphi^{kq})$}{O/(pi^{kq})}-值完美配对]\label{prop:xi-exterior-power-perfect-pairing}
\leanverified{paper\_xi\_exterior\_power\_perfect\_pairing}
令 \(M_{q,k}=\mathrm{coker}_{\mathcal O}(\Lambda^k C_q)\)。
则存在 \(\mathcal O\)-双线性配对
\[
\langle\cdot,\cdot\rangle:\ M_{q,k}\times M_{q,k}\to \mathcal O/(\pi_\varphi^{kq})
\]
使得在定理 \ref{thm:xi-exterior-power-smith-schubert} 的直和分解下，
\(\langle\cdot,\cdot\rangle\) 将 \(\mathcal O/(\pi_\varphi^{e(\lambda)})\) 与 \(\mathcal O/(\pi_\varphi^{e(\lambda^\vee)})\) 配对为完美对偶，且对非互补因子取零。
特别地，\(\langle\cdot,\cdot\rangle\) 非退化，并诱导（非自然但严格存在的）同构
\[
M_{q,k}\ \cong\ \mathrm{Hom}_{\mathcal O}\!\bigl(M_{q,k},\ \mathcal O/(\pi_\varphi^{kq})\bigr).
\]
\end{proposition}
\begin{proof}
对任意整数 \(0\le e\le kq\)，定义标准配对
\[
\mathcal O/(\pi_\varphi^{e})\times \mathcal O/(\pi_\varphi^{kq-e})
\longrightarrow \mathcal O/(\pi_\varphi^{kq}),
\qquad
\langle \bar x,\bar y\rangle:=\overline{\pi_\varphi^{kq-e}xy}.
\]
若 \(x\) 改变一个 \(\pi_\varphi^{e}\)-倍数，则 \(\pi_\varphi^{kq-e}xy\) 改变一个 \(\pi_\varphi^{kq}\)-倍数，故该配对良定；
对第二变量同理。
该配对在每个因子上给出同构
\(
\mathcal O/(\pi_\varphi^{kq-e})\cong \mathrm{Hom}_{\mathcal O}(\mathcal O/(\pi_\varphi^{e}),\mathcal O/(\pi_\varphi^{kq}))
\)，因而为完美对偶。
\par
由推论 \ref{cor:xi-exterior-power-poincare-duality}，对每个 \(\lambda\subseteq a^k\) 有 \(e(\lambda^\vee)=kq-e(\lambda)\)。
将上述标准配对逐个施加到互补分拆对应的两条循环因子上，并对其余直和因子之间取零，即得到所述配对与非退化性；
伴随同构由完美对偶的直和保持性推出。证毕。
\end{proof}

\begin{proposition}[外幂行列式指数与余核基数]\label{prop:xi-exterior-power-det-coker-size}
令 \(N=\binom{q+1}{k}\)。
则外幂行列式的 \(\pi_\varphi\)-阶满足
\[
\det(\Lambda^k C_q)\ \sim\ \pi_\varphi^{\,\sum_{i=1}^{N}e_i},
\qquad
\sum_{i=1}^{N}e_i=\binom{q}{k-1}\frac{q(q+1)}{2}
=\binom{q+1}{k}\frac{kq}{2}.
\]
并且
\[
\#\,\mathrm{coker}_{\mathcal O}(\Lambda^k C_q)=5^{\,\binom{q+1}{k}\frac{kq}{2}}.
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:xi-exterior-power-smith-schubert} 的 Smith 正规形，行列式在单位元意义下等于对角不变因子之积，
故 \(\pi_\varphi\)-阶为 \(\sum_i e_i\)。
另一方面，\(k\)-子集和总和为
\[
\sum_{S:\ |S|=k}\ \sum_{i\in S} i
=\sum_{i=0}^{q} i\cdot \#\{S:\ |S|=k,\ i\in S\}
=\binom{q}{k-1}\sum_{i=0}^{q} i
=\binom{q}{k-1}\frac{q(q+1)}{2},
\]
并由恒等式 \(\binom{q+1}{k}k=(q+1)\binom{q}{k-1}\) 得到等价形式
\(\sum_i e_i=\binom{q+1}{k}\frac{kq}{2}\)。
\par
对余核基数，Smith 分解给出
\(
\mathrm{coker}_{\mathcal O}(\Lambda^k C_q)\cong \bigoplus_i \mathcal O/(\pi_\varphi^{e_i})
\)；
又由 \(\mathrm N(\pi_\varphi)=5\) 知 \(\#(\mathcal O/(\pi_\varphi^{e}))=5^{e}\)，故
\(
\#\,\mathrm{coker}=\prod_i 5^{e_i}=5^{\sum_i e_i}
\)。
证毕。
\end{proof}

\begin{proposition}[指数生成函数与 \texorpdfstring{$\pi_\varphi$}{pi}-分级 Hilbert 级数]\label{prop:xi-exterior-power-gaussian-binom-hilbert}
\leanverified{paper\_xi\_exterior\_power\_gaussian\_binom\_hilbert}
定义指数生成多项式
\[
E_{q,k}(t):=\sum_{\substack{S\subseteq\{0,\dots,q\}\\ |S|=k}} t^{\sum_{i\in S} i}.
\]
则有闭式
\[
\boxed{\;
E_{q,k}(t)=t^{\frac{k(k-1)}{2}}\prod_{i=1}^{k}\frac{1-t^{q+2-i}}{1-t^{i}}\; }.
\]
令 \(M_{q,k}=\mathrm{coker}_{\mathcal O}(\Lambda^k C_q)\)，并记剩余域 \(\mathcal O/(\pi_\varphi)\cong\FF_5\)。
则由 \(\pi_\varphi\)-幂过滤诱导的 Hilbert 级数
\[
H_{q,k}(t):=\sum_{s\ge 0}\dim_{\FF_5}\bigl(\pi_\varphi^{s}M_{q,k}/\pi_\varphi^{s+1}M_{q,k}\bigr)\,t^{s}
\]
满足
\[
\boxed{\;
H_{q,k}(t)=\frac{\binom{q+1}{k}-E_{q,k}(t)}{1-t}\; }.
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:xi-exterior-power-smith-schubert} 的分拆参数化，
\[
E_{q,k}(t)
=\sum_{\lambda\subseteq a^k} t^{e(\lambda)}
=t^{\frac{k(k-1)}{2}}\sum_{\lambda\subseteq a^k} t^{|\lambda|}.
\]
而矩形内分拆面积的生成函数为高斯二项系数：
\(
\sum_{\lambda\subseteq a^k} t^{|\lambda|}
=\prod_{i=1}^{k}\frac{1-t^{a+i}}{1-t^{i}}
\)；
将 \(a=q+1-k\) 代入并重排因子即得所示乘积式。
\par
对 Hilbert 级数，由同一分解 \(M_{q,k}\cong\bigoplus_{\lambda}\mathcal O/(\pi_\varphi^{e(\lambda)})\)，
每个循环因子贡献级数 \(1+t+\cdots+t^{e(\lambda)-1}=(1-t^{e(\lambda)})/(1-t)\)。
求和得到
\[
H_{q,k}(t)=\frac{1}{1-t}\sum_{\lambda\subseteq a^k}\bigl(1-t^{e(\lambda)}\bigr)
=\frac{\binom{q+1}{k}-E_{q,k}(t)}{1-t}.
\]
证毕。
\end{proof}

\begin{corollary}[Schubert 长度谱与 Jordan 块]\label{cor:xi-exterior-power-jordan-schubert}
将 \(M_{q,k}\) 视为 \(\FF_5\)-向量空间，并令幂零算子 \(N\) 为“乘以 \(\pi_\varphi\)”所诱导的线性变换
\(
N:M_{q,k}\to M_{q,k}
\)。
则 \(N\) 的 Jordan 块长度多重集恰为
\[
\Bigl\{\,e(\lambda)=\frac{k(k-1)}{2}+|\lambda|:\ \lambda\subseteq (q+1-k)^k\,\Bigr\},
\]
并且在推论 \ref{cor:xi-exterior-power-poincare-duality} 的补分拆对合下满足块长互补
\[
e(\lambda)+e(\lambda^\vee)=kq.
\]
\end{corollary}
\begin{proof}
对任意 \(e\ge 1\)，\(\mathcal O/(\pi_\varphi^{e})\) 作为 \(\FF_5\)-向量空间的维数为 \(e\)，且乘以 \(\pi_\varphi\) 在基 \(1,\pi_\varphi,\dots,\pi_\varphi^{e-1}\) 下给出单 Jordan 链，
故其 Jordan 块长度恰为 \(e\)。
将此结论逐项应用于定理 \ref{thm:xi-exterior-power-smith-schubert} 的直和分解得到长度谱；
互补关系由推论 \ref{cor:xi-exterior-power-poincare-duality} 立即推出。证毕。
\end{proof}

\begin{theorem}[Jordan 长度谱的仿射 Poincar\'e 对偶、显式方差与高斯极限]\label{thm:xi-exterior-power-jordan-affine-poincare-clt}
设 \(q\ge 2\) 且 \(1\le k\le q\)，并沿用
\[
M_{q,k}:=\mathrm{coker}_{\mathcal O}(\Lambda^k C_q)
\]
的记号。对 \(0\le s\le kq-1\)，定义
\[
h_{q,k}(s):=\dim_{\FF_5}\bigl(\pi_\varphi^{s}M_{q,k}/\pi_\varphi^{s+1}M_{q,k}\bigr).
\]
则成立：
\begin{enumerate}
  \item 仿射 Poincar\'e 对偶
  \[
  \boxed{\;
  h_{q,k}(s)+h_{q,k}(kq-1-s)=\binom{q+1}{k}
  \qquad(0\le s\le kq-1).\;}
  \]
  等价地，Hilbert 级数满足函数方程
  \[
  \boxed{\;
  H_{q,k}(t)+t^{kq-1}H_{q,k}(t^{-1})
  =
  \binom{q+1}{k}\,\frac{1-t^{kq}}{1-t}.\;}
  \]

  \item 若 \(S\) 在全部 \(k\)-子集
  \(
  \{S\subseteq\{0,\dots,q\}:|S|=k\}
  \)
  上均匀分布，并定义
  \[
  L_{q,k}:=\sum_{i\in S} i,
  \]
  则 \(L_{q,k}\) 与“均匀随机选取一个 Jordan 块长度”同分布，并满足
  \[
  \boxed{\;
  \mathbb E[L_{q,k}]=\frac{kq}{2},
  \qquad
  \mathrm{Var}(L_{q,k})=\frac{k(q+1-k)(q+2)}{12}.\;}
  \]

  \item 若 \(k=k(q)\) 满足
  \[
  \frac{k(q)}{q+1}\longrightarrow \tau\in(0,1),
  \]
  则标准化长度变量满足中心极限定理
  \[
  \boxed{\;
  \frac{L_{q,k(q)}-\frac{k(q)q}{2}}
  {\sqrt{\frac{k(q)(q+1-k(q))(q+2)}{12}}}
  \Longrightarrow \mathcal N(0,1).\;}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由推论 \ref{cor:xi-exterior-power-jordan-schubert}，Jordan 块长度多重集恰为
\[
\{\,e(\lambda):\ \lambda\subseteq (q+1-k)^k\,\}.
\]
因此每个长度为 \(e(\lambda)\) 的 Jordan 块对分级层
\(
\pi_\varphi^{s}M_{q,k}/\pi_\varphi^{s+1}M_{q,k}
\)
在且仅在 \(0\le s\le e(\lambda)-1\) 时贡献一维，于是
\[
h_{q,k}(s)=\#\{\lambda\subseteq (q+1-k)^k:\ e(\lambda)>s\}.
\]
再由推论 \ref{cor:xi-exterior-power-poincare-duality} 的互补关系
\[
e(\lambda^\vee)=kq-e(\lambda),
\]
可知补分拆对合将集合
\(
\{e(\lambda)\le s\}
\)
双射到
\(
\{e(\lambda)>kq-1-s\}
\)。
因此
\[
h_{q,k}(kq-1-s)
=
\#\{\lambda:\ e(\lambda)\le s\},
\]
与上一式相加即得
\[
h_{q,k}(s)+h_{q,k}(kq-1-s)=\binom{q+1}{k}.
\]
第二个函数方程只需将上述系数恒等式乘以 \(t^s\) 后对 \(0\le s\le kq-1\) 求和即可。

对第 \(2\) 条，由定理 \ref{thm:xi-exterior-power-smith-schubert} 的 \(k\)-子集参数化，
\[
e(\lambda(S))=\sum_{i\in S} i.
\]
故均匀随机 Jordan 块长度与简单随机抽样和
\(
L_{q,k}
\)
同分布。
其均值为
\[
\mathbb E[L_{q,k}]
=
\sum_{i=0}^{q} i\ \mathbb P(i\in S)
=
\frac{k}{q+1}\sum_{i=0}^{q} i
=
\frac{kq}{2}.
\]
将总体
\(
\{0,1,\dots,q\}
\)
视为大小 \(q+1\) 的有限总体，其总体方差为
\[
\sigma_{\mathrm{pop}}^2
=
\frac{1}{q+1}\sum_{i=0}^{q}\left(i-\frac{q}{2}\right)^2
=
\frac{q(q+2)}{12}.
\]
简单随机抽样和的经典方差公式给出
\[
\mathrm{Var}(L_{q,k})
=
k\,\frac{q+1-k}{q}\,\sigma_{\mathrm{pop}}^2
=
\frac{k(q+1-k)(q+2)}{12}.
\]

对第 \(3\) 条，同一识别说明 \(L_{q,k(q)}\) 正是从有限总体 \(\{0,\dots,q\}\) 中不放回均匀抽取 \(k(q)\) 个点之和。
当
\(
k(q)/(q+1)\to\tau\in(0,1)
\)
时，经典 Hájek 中心极限定理适用，从而上述标准化变量依分布收敛到标准正态分布。证毕。
\end{proof}
