为把上述乘子刚性与挠群结构直接对接，记
\[
\pi:E_0\longrightarrow E_0/\{\pm1\}\cong \PP^1_t,\qquad \pi(P):=t(P),
\]
则 \(\mathcal L\circ \pi=\pi\circ [2]\)。

\begin{theorem}[迭代不动点的奇阶挠点刻画]\label{thm:xi-j-sextic-elliptic-lattes-fixedpoints-odd-torsion}
对任意整数 \(n\ge 1\)，有精确集合恒等式
\[
\boxed{\ \mathrm{Fix}(\mathcal L^{\circ n})
=\pi\bigl(E_0[2^n-1]\cup E_0[2^n+1]\bigr)\ }.
\]
特别地，
\[
\boxed{\ \card{\mathrm{Fix}(\mathcal L^{\circ n})}=4^n+1\ }.
\]
\end{theorem}
\begin{proof}
取 \(x\in \PP^1_t\) 并任选 \(P\in E_0(\overline{\QQ})\) 使 \(\pi(P)=x\)。
则
\[
\mathcal L^{\circ n}(x)=x
\iff
\pi([2^n]P)=\pi(P)
\iff
[2^n]P=\pm P,
\]
因为 \(\pi\) 正是 \(E_0\) 对对合 \(P\mapsto -P\) 的商映射。
于是
\[
[2^n]P=P \iff (2^n-1)P=O,\qquad
[2^n]P=-P \iff (2^n+1)P=O,
\]
从而得到第一式。
\par
现令 \(m\) 为奇数。
由于 \(P=-P\) 等价于 \(2P=O\)，而奇数阶挠子群 \(E_0[m]\) 中仅有 \(O\) 满足该条件，故
\[
\card{\pi(E_0[m])}
=\frac{\card{E_0[m]}-1}{2}+1
=\frac{m^2+1}{2}.
\]
取 \(m_-:=2^n-1\)、\(m_+:=2^n+1\)。
二者均为奇数且互素，因而
\[
E_0[m_-]\cap E_0[m_+]=E_0[\gcd(m_-,m_+)]=E_0[1]=\{O\},
\]
于是其像仅在 \(\infty=\pi(O)\) 处相交。
因此
\[
\card{\mathrm{Fix}(\mathcal L^{\circ n})}
=\frac{m_-^2+1}{2}+\frac{m_+^2+1}{2}-1
=4^n+1.
\]
证毕。
\end{proof}

\begin{corollary}[精确周期点的算术分类与 Jordan 计数公式]\label{cor:xi-j-sextic-elliptic-lattes-exact-periodic-jordan}
对奇整数 \(m\ge 1\)，定义
\[
\operatorname{ord}_m^{\pm}(2):=\min\{r\ge 1:\ 2^r\equiv \pm1\pmod m\},
\]
并记
\[
E_0[m]_{\mathrm{prim}}
:=\{P\in E_0(\overline{\QQ}):\ \operatorname{ord}(P)=m\}.
\]
再令
\[
J_2(m):=m^2\prod_{\ell\mid m}\bigl(1-\ell^{-2}\bigr),
\]
则对任意 \(n\ge 1\)，\(\mathcal L\) 的精确周期 \(n\) 点集合满足
\[
\boxed{\ 
\mathrm{Per}_n(\mathcal L)
=\bigsqcup_{\substack{m\ge 1\ \mathrm{odd}\\ \operatorname{ord}_m^{\pm}(2)=n}}
\pi\bigl(E_0[m]_{\mathrm{prim}}\bigr)\ }.
\]
相应地，
\[
\boxed{\ 
\card{\mathrm{Per}_n(\mathcal L)}
=\mathbf 1_{n=1}
+\frac12\sum_{\substack{m>1\ \mathrm{odd}\\ \operatorname{ord}_m^{\pm}(2)=n}}J_2(m)\ },
\]
并且精确周期长度为 \(n\) 的闭轨条数满足
\[
\boxed{\ 
\card{\mathrm{Cycles}_n(\mathcal L)}
=\mathbf 1_{n=1}
+\frac1{2n}\sum_{\substack{m>1\ \mathrm{odd}\\ \operatorname{ord}_m^{\pm}(2)=n}}J_2(m)\ }.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-j-sextic-elliptic-lattes-fixedpoints-odd-torsion}，点 \(x=\pi(P)\) 为周期点当且仅当存在 \(r\ge 1\) 使
\[
\pi([2^r]P)=\pi(P)
\iff
[2^r]P=\pm P,
\]
亦即 \(P\) 为奇阶挠点，或者 \(P=O\)。
若 \(P\) 的精确阶为奇数 \(m\)，则
\[
\mathcal L^{\circ r}(\pi(P))=\pi(P)
\iff
[2^r]P=\pm P
\iff
2^r\equiv \pm1\pmod m.
\]
因此 \(\pi(P)\) 的精确周期恰为 \(\operatorname{ord}_m^{\pm}(2)\)，于是得到集合分解。
若 \(\pi(P)=\pi(Q)\)，则 \(Q=\pm P\)，故 \(\operatorname{ord}(Q)=\operatorname{ord}(P)\)；
因此不同的奇整数 \(m\) 给出彼此不交的部分。
\par
对 \(m=1\) 有 \(E_0[1]_{\mathrm{prim}}=\{O\}\)，其像为 \(\{\infty\}\)，且 \(\operatorname{ord}_1^{\pm}(2)=1\)。
对奇数 \(m>1\)，对合 \(P\mapsto -P\) 在 \(E_0[m]_{\mathrm{prim}}\) 上无不动点，故
\[
\card{\pi(E_0[m]_{\mathrm{prim}})}=\frac12\,\card{E_0[m]_{\mathrm{prim}}}
=\frac12\,J_2(m).
\]
对周期轨条数，只需再除以轨长 \(n\) 即可。证毕。
\end{proof}

\begin{theorem}[预周期点的挠点刻画与尾长公式]\label{thm:xi-j-sextic-elliptic-lattes-preperiodic-torsion-tail}
记
\[
\mathrm{PrePer}(\mathcal L)
:=\{x\in \PP^1_t:\ x\ \text{在}\ \mathcal L\ \text{下预周期}\}.
\]
则有严格刻画
\[
\boxed{\ \mathrm{PrePer}(\mathcal L)=\pi\bigl(E_0(\overline{\QQ})_{\mathrm{tors}}\bigr)\ }.
\]
更精确地，设 \(P\in E_0(\overline{\QQ})\) 的精确阶分解为
\[
\operatorname{ord}(P)=2^s m,\qquad s\ge 0,\quad m\ \text{为奇数},
\]
并令 \(x:=\pi(P)\)。
若以 \(\operatorname{tail}(x)\) 表示进入周期部分前的最短迭代步数，以 \(\operatorname{per}(x)\) 表示最终周期长度，则
\[
\boxed{\ \operatorname{tail}(x)=s,\qquad
\operatorname{per}(x)=\operatorname{ord}_m^{\pm}(2)\ }.
\]
\end{theorem}
\begin{proof}
若 \(x=\pi(P)\) 为预周期点，则存在整数 \(r>s\ge 0\) 使
\[
\mathcal L^{\circ r}(x)=\mathcal L^{\circ s}(x).
\]
等价地，
\[
\pi([2^r]P)=\pi([2^s]P),
\]
于是
\[
[2^r]P=\pm [2^s]P,
\qquad\text{即}\qquad
(2^{r-s}\mp1)\,[2^s]P=O.
\]
由于 \(2^{r-s}\mp1\neq 0\)，可得 \(\bigl(2^s(2^{r-s}\mp1)\bigr)P=O\)，从而 \(P\) 为挠点。
反过来，挠点在倍映射 \([2]\) 下的轨道有限，故其像必为预周期点。
于是得到
\(
\mathrm{PrePer}(\mathcal L)=\pi(E_0(\overline{\QQ})_{\mathrm{tors}})
\)。
\par
若 \(\operatorname{ord}(P)=2^s m\) 且 \(m\) 为奇数，则 \([2^s]P\) 的精确阶恰为 \(m\)。
因此 \(\mathcal L^{\circ s}(x)=\pi([2^s]P)\) 进入奇阶挠点层，并由推论
\ref{cor:xi-j-sextic-elliptic-lattes-exact-periodic-jordan}
立刻得到其最终周期长度为 \(\operatorname{ord}_m^{\pm}(2)\)。
\par
还需证明尾长恰为 \(s\)。
若存在 \(0\le r<s\) 使 \(\mathcal L^{\circ r}(x)\) 已经周期，则对某个 \(n\ge 1\) 有
\[
\pi([2^{r+n}]P)=\pi([2^r]P)
\iff
[2^n]([2^r]P)=\pm [2^r]P,
\]
即
\[
(2^n\mp1)\,[2^r]P=O.
\]
但 \(2^n\mp1\) 始终为奇数，而 \([2^r]P\) 的精确阶为 \(2^{s-r}m\)，其 \(2\)-部分非平凡；
故它不可能被奇数杀死，矛盾。
因此 \(\operatorname{tail}(x)=s\)。证毕。
\end{proof}

\begin{theorem}[后临界树的精确层级公式]\label{thm:xi-j-sextic-elliptic-lattes-postcritical-tree-levels}
\leanverified{paper\_xi\_j\_sextic\_elliptic\_lattes\_postcritical\_tree\_levels}
设
\[
\mathcal P:=\{\infty,1728,\alpha,\beta\}=\pi(E_0[2]),
\]
并记三个非零 \(2\)-挠点为
\[
T_1:=(1728,0),\qquad T_2:=(\alpha,0),\qquad T_3:=(\beta,0).
\]
则对任意 \(n\ge 1\) 都有
\[
\boxed{\ \mathcal L^{-n}(\infty)=\pi(E_0[2^n])\ },
\]
\[
\boxed{\ \mathcal L^{-n}(1728)=\pi(T_1+E_0[2^n]),\quad
\mathcal L^{-n}(\alpha)=\pi(T_2+E_0[2^n]),\quad
\mathcal L^{-n}(\beta)=\pi(T_3+E_0[2^n])\ }.
\]
并且这些集合的基数分别为
\[
\boxed{\ \card{\mathcal L^{-n}(\infty)}=2^{2n-1}+2\ },
\]
\[
\boxed{\ \card{\mathcal L^{-n}(1728)}
=\card{\mathcal L^{-n}(\alpha)}
=\card{\mathcal L^{-n}(\beta)}
=2^{2n-1}\ }.
\]
因此
\[
\boxed{\ \card{\mathcal L^{-n}(\mathcal P)}=2^{2n+1}+2\ }.
\]
\end{theorem}
\begin{proof}
由 \(\mathcal L\circ \pi=\pi\circ [2]\)，有
\[
x\in \mathcal L^{-n}(\infty)
\iff
\exists P\in E_0,\ \pi(P)=x,\ \pi([2^n]P)=\infty=\pi(O)
\iff
[2^n]P=O,
\]
故 \(\mathcal L^{-n}(\infty)=\pi(E_0[2^n])\)。
同理，以 \(1728=\pi(T_1)\) 为例，
\[
x\in \mathcal L^{-n}(1728)
\iff
\exists P,\ \pi(P)=x,\ \pi([2^n]P)=\pi(T_1)
\iff
[2^n]P=\pm T_1.
\]
由于 \(T_1=-T_1\)，上式等价于 \([2^n]P=T_1\)，即 \(P\in T_1+E_0[2^n]\)。
其余两式完全同理。
\par
现作计数。
首先 \(\card{E_0[2^n]}=4^n\)，而对合 \(P\mapsto -P\) 在 \(E_0[2^n]\) 上的不动点恰为 \(E_0[2]\)，共有 \(4\) 个。
因此
\[
\card{\pi(E_0[2^n])}
=\frac{4^n-4}{2}+4
=2^{2n-1}+2.
\]
另一方面，在任意非零 \(2\)-挠点陪集 \(T_i+E_0[2^n]\) 上，对合 \(P\mapsto -P\) 无不动点；
否则 \(P=-P\) 将推出 \(2P=O\)，从而 \([2^n]P=O\neq T_i\)。
故
\[
\card{\pi(T_i+E_0[2^n])}
=\frac{\card{T_i+E_0[2^n]}}{2}
=\frac{4^n}{2}
=2^{2n-1}.
\]
最后，四个 \(n\)-级逆像层对应于 \(\mathcal P\) 的四个不同点，故彼此不交，求和即得
\[
\card{\mathcal L^{-n}(\mathcal P)}
=(2^{2n-1}+2)+3\cdot 2^{2n-1}
=2^{2n+1}+2.
\]
证毕。
\end{proof}

\begin{theorem}[迭代不动点方程的射影除法多项式分解]\label{thm:xi-j-sextic-elliptic-lattes-projective-division-factorization}
对奇整数 \(m\ge 1\)，定义 \(m\)-阶射影除法多项式
\[
\Phi_m(t):=
\prod_{\substack{P\in E_0[m]\setminus\{O\}\\ P\sim -P}}
\bigl(t-\pi(P)\bigr)\in \QQ[t],
\]
其中 \(P\sim -P\) 表示对 \(\{\pm1\}\)-轨道取代表；
当 \(m=1\) 时约定空积 \(\Phi_1(t)=1\)。
则 \(\Phi_m\) 为首一多项式，且
\[
\deg \Phi_m=\frac{m^2-1}{2}.
\]
若把 \(\mathcal L^{\circ n}(t)-t\) 写成既约分式
\[
\mathcal L^{\circ n}(t)-t=\frac{N_n(t)}{D_n(t)},
\qquad N_n,D_n\in\QQ[t],\ \gcd(N_n,D_n)=1,
\]
则对任意 \(n\ge 1\)，存在常数 \(c_n\in\QQ^\times\) 使
\[
\boxed{\ N_n(t)=c_n\,\Phi_{2^n-1}(t)\,\Phi_{2^n+1}(t)\ }.
\]
因此 \(\mathcal L^{\circ n}(t)-t\) 的全部有限零点，恰由两层奇挠点
\[
E_0[2^n-1],\qquad E_0[2^n+1]
\]
在商映射 \(\pi\) 下的像给出。
\end{theorem}
\begin{proof}
由定义，\(\Phi_m\) 的根集正是 \(\pi(E_0[m]\setminus\{O\})\) 的不同有限像点。
Galois 群作用保持挠点阶与 \(\pm\)-轨道，故 \(\Phi_m\) 的系数属于 \(\QQ\)。
由定理 \ref{thm:xi-j-sextic-elliptic-lattes-fixedpoints-odd-torsion}，
\[
\mathrm{Fix}(\mathcal L^{\circ n})
=\pi\bigl(E_0[2^n-1]\cup E_0[2^n+1]\bigr),
\]
且 \(E_0[2^n-1]\cap E_0[2^n+1]=\{O\}\)。
因此 \(\mathcal L^{\circ n}(t)-t\) 的有限零点集合恰为
\[
\pi(E_0[2^n-1]\setminus\{O\})\ \sqcup\ \pi(E_0[2^n+1]\setminus\{O\}),
\]
从而 \(N_n(t)\) 与 \(\Phi_{2^n-1}(t)\Phi_{2^n+1}(t)\) 有完全相同的零点集合。
\par
这些零点全为单根。
事实上，若 \(x\) 为 \(\mathcal L^{\circ n}\) 的有限不动点，则由推论
\ref{cor:xi-j-sextic-elliptic-lattes-multiplier-rigidity}
有
\[
(\mathcal L^{\circ n})'(x)=\pm 2^n\neq 1,
\]
故
\[
\bigl(\mathcal L^{\circ n}(t)-t\bigr)'|_{t=x}
=(\mathcal L^{\circ n})'(x)-1\neq 0.
\]
于是 \(N_n\) 在每个有限零点处的重数都等于 \(1\)。
\par
再作次数比较。
对奇数 \(m\)，由
\[
\card{\pi(E_0[m])}=\frac{m^2+1}{2}
\]
并去掉无穷远点 \(\pi(O)\)，得到
\[
\deg \Phi_m=\frac{m^2-1}{2}.
\]
另一方面，\(\mathcal L^{\circ n}\) 为次数 \(4^n\) 的有理映射，而其全部 \(4^n+1\) 个不动点中恰有一个是 \(\infty\)；
故 \(N_n\) 的有限零点总数为 \(4^n\)，从而
\[
\deg N_n=4^n.
\]
同时
\[
\deg \Phi_{2^n-1}+\deg \Phi_{2^n+1}
=\frac{(2^n-1)^2-1}{2}+\frac{(2^n+1)^2-1}{2}
=4^n.
\]
因此二者仅差一个非零常数因子 \(c_n\in\QQ^\times\)。证毕。
\end{proof}
