\begin{conclusion}[Lee--Yang 立方与“立方--二次”对偶分解]\label{con:xi-leyang-cubic-quadratic-dual-decomposition}
在 \(\QQ[y]\) 中有刚性加性分解
\[
P(y)=A(y)+16(2y+1)^3,\qquad
27y(y-1)=A(y)-16(2y+1)^3.
\]
从而在 \(K(H_{\mathrm{LY}})\) 中成立乘性分解
\[
\bigl(A(y)-16(2y+1)^3\bigr)\bigl(A(y)+16(2y+1)^3\bigr)=-27u^2,
\]
以及等价范数恒等式
\[
\bigl(A(y)-3su\bigr)\bigl(A(y)+3su\bigr)=\bigl(16(2y+1)^3\bigr)^2,
\qquad s:=\sqrt{-3}.
\]
\end{conclusion}
\begin{proof}
直接展开
\(
16(2y+1)^3=128y^3+192y^2+96y+16
\)
并与 \(A(y)\) 比较系数，即得前两式。
相乘后得到
\[
A(y)^2-256(2y+1)^6=27y(y-1)P(y)=-27u^2.
\]
再由 \(s^2=-3\) 与定理 \ref{thm:xi-leyang-sextic-perfect-power-collapse} 的
\(
A(y)^2+27u^2=256(2y+1)^6
\)
即得范数形式。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$t$}{t}-参数化三次方程的判别式与分歧纤维精确分解]\label{thm:xi-leyang-gt-discriminant-and-special-fiber-factorization}\leanverified{paper\_xi\_leyang\_gt\_discriminant\_and\_special\_fiber\_factorization}
令
\[
G_t(y):=8t(2y+1)^3-A(y)
=(64t-128)y^3+(96t-219)y^2+(48t-69)y+(8t-16).
\]
则关于 \(y\) 的判别式为
\[
\operatorname{Disc}_y(G_t)=-3^9\bigl(2304t^2-8896t+8549\bigr).
\]
特别地，唯一有限分歧值为
\[
t_\pm=\frac{139}{72}\pm\frac{7}{144}\sqrt7.
\]
并且在 \(\QQ(\sqrt7)\) 上存在 \(c_\pm\in\QQ(\sqrt7)^\times\) 使
\[
G_{t_+}(y)=c_+\Bigl(y-\frac{3+\sqrt7}{2}\Bigr)^2\Bigl(y+2-\frac{3\sqrt7}{4}\Bigr),
\]
\[
G_{t_-}(y)=c_-\Bigl(y-\frac{3-\sqrt7}{2}\Bigr)^2\Bigl(y+2+\frac{3\sqrt7}{4}\Bigr).
\]
\end{theorem}
\begin{proof}
将
\(
(a,b,c,d)=(64t-128,\,96t-219,\,48t-69,\,8t-16)
\)
代入三次判别式公式
\[
\operatorname{Disc}(ay^3+by^2+cy+d)
=b^2c^2-4ac^3-4b^3d-27a^2d^2+18abcd,
\]
直接化简即得判别式表达式。
由其零点可得唯一有限分歧值 \(t_\pm\)。
将 \(t=t_\pm\) 代回 \(G_t(y)\) 并在 \(\QQ(\sqrt7)\) 上因式分解，即得所示“平方因子 \(\times\) 一次因子”分解。证毕。
\end{proof}

\begin{conclusion}[双极塌缩：\texorpdfstring{$H_{\mathrm{LY}}$}{HLY} 的 Weierstrass 分歧在 \texorpdfstring{$t$}{t}-坐标下压缩为 \texorpdfstring{$\pm2$}{+/-2}]\label{con:xi-leyang-weierstrass-branch-binary-collapse-in-t}
记
\[
t(y)=\frac{A(y)}{8(2y+1)^3},\qquad
w:=\frac{u}{(2y+1)^3}\in \QQ(H_{\mathrm{LY}}).
\]
则恒有
\[
(t-2)(t+2)=-\frac{27}{64}\,w^2.
\]
并且
\[
t(0)=t(1)=t(\infty)=2,\qquad
t=-2\ \text{于全部 }P(y)=0\text{ 的三点上}.
\]
因此 \(\pi:H_{\mathrm{LY}}\to\PP^1_y\) 的全部 Weierstrass 分歧点（即 \(u=0\)）在 \(t\)-坐标下恰映为 \(\{\pm2\}\)。
\end{conclusion}
\begin{proof}
由结论 \ref{con:xi-leyang-cubic-quadratic-dual-decomposition}，
\[
t+2=\frac{A+16(2y+1)^3}{8(2y+1)^3}=\frac{P(y)}{8(2y+1)^3},
\]
\[
t-2=\frac{A-16(2y+1)^3}{8(2y+1)^3}=\frac{27y(y-1)}{8(2y+1)^3}.
\]
相乘得
\[
(t-2)(t+2)
=\frac{27y(y-1)P(y)}{64(2y+1)^6}
=-\frac{27u^2}{64(2y+1)^6}
=-\frac{27}{64}\,w^2.
\]
代入 \(y=0,1\) 与无穷远主项比较可得 \(t=2\)；
而 \(P(y)=0\) 时由 \(P=A+16(2y+1)^3\) 得 \(t=-2\)。
故全部分歧点在 \(t\)-坐标下仅取 \(\pm2\)。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\eta$}{eta} 所定义三重映射的全分歧结构]\label{thm:xi-leyang-eta-degree3-complete-ramification}
在 \(K=\QQ(\sqrt{-3})\) 上记
\[
\eta:=\frac{A(y)-3su}{16(2y+1)^3}\in K(H_{\mathrm{LY}})^\times,\qquad s:=\sqrt{-3},
\]
并令
\[
\varphi_\eta:H_{\mathrm{LY}}\longrightarrow \PP^1_K
\]
为对应有理函数态射。则：
\begin{enumerate}
  \item \(\deg(\varphi_\eta)=3\)；
  \item \(Q_-\) 与 \(Q_+\) 为全分歧点，且
  \[
  e(Q_-)=e(Q_+)=3;
  \]
  \item 除 \(Q_\pm\) 外，其余全部分歧点恰为四点
  \[
  P_{\varepsilon,\sigma}:\ 
  y(P_{\varepsilon,\sigma})=y_\varepsilon:=\frac{3+\varepsilon\sqrt7}{2},\quad
  u(P_{\varepsilon,\sigma})=\sigma\sqrt{f(y_\varepsilon)},
  \]
  其中 \(\varepsilon,\sigma\in\{\pm1\}\)、\(f(y):=-y(y-1)P(y)\)，并且均为简单分歧：
  \[
  e(P_{\varepsilon,\sigma})=2.
  \]
\end{enumerate}
因而分歧除子为
\[
\operatorname{Ram}(\varphi_\eta)
=2Q_-+2Q_++\sum_{\varepsilon=\pm1}\sum_{\sigma=\pm1}P_{\varepsilon,\sigma},
\]
且不存在其他分歧点。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion}，
\[
\operatorname{div}(\eta)=3(Q_--Q_+),
\]
故 \(\deg(\varphi_\eta)=3\)，且 \(Q_\pm\) 分别是零/极点阶 \(3\) 的全分歧点，故 \(e(Q_\pm)=3\)。

对 \(\eta\) 微分并用 \(u^2=f(y)\)、\(du=\frac{f'(y)}{2u}\,dy\) 化简得
\[
d\eta
=-\frac{3\,(2y^2-6y+1)\,\bigl(9u+sA(y)\bigr)}{16\,u\,(2y+1)^4}\,dy.
\]
于是有限值处的临界条件为
\(
2y^2-6y+1=0
\)
或
\(
9u+sA(y)=0
\)。
后者代入结论 \ref{con:xi-leyang-cubic-quadratic-dual-decomposition} 的
\(
A^2+27u^2=256(2y+1)^6
\)
可得 \((2y+1)^6=0\)，即 \(y=-\frac12\)，并对应 \(Q_-\)。
前者给出
\[
y=y_\pm=\frac{3\pm\sqrt7}{2}.
\]
这两点处 \(u\neq 0\)（等价于 \(t(y_\pm)=t_\pm\neq\pm2\)），故各自对应双点
\(
u=\pm\sqrt{f(y_\pm)}
\)，共四点 \(P_{\varepsilon,\sigma}\)。
由于 \(2y^2-6y+1\) 在 \(y_\pm\) 为单根，故这些点均为简单分歧，\(e=2\)。

最后由 Hurwitz 公式（\(\deg=3,\ g(H_{\mathrm{LY}})=2\)）得总分歧量
\[
\sum_{P}(e(P)-1)=2g-2+2\deg=2+6=8.
\]
已列分歧贡献为
\[
(3-1)+(3-1)+4\cdot(2-1)=2+2+4=8,
\]
故列表完备。证毕。
\end{proof}

\begin{conclusion}[\texorpdfstring{$\eta$}{eta} 的对数微分刚性可积化公式]\label{con:xi-leyang-log-eta-rigid-integrable-identity}
在 \(\Omega^1_{H_{\mathrm{LY}}/K}\) 中有恒等式
\[
d\log\eta=\frac{d\eta}{\eta}
=\frac{9}{\sqrt{-3}}\cdot\frac{2y^2-6y+1}{(2y+1)u}\,dy
=-3\sqrt{-3}\cdot\frac{2y^2-6y+1}{(2y+1)u}\,dy.
\]
并且若记
\[
t:=\eta+\eta^{-1},
\]
则等价地有
\[
d\log\eta=\frac{dt}{\eta-\eta^{-1}}.
\]
在约定
\(
\sqrt{t^2-4}:=\eta-\eta^{-1}
\)
下，可写为
\[
d\log\eta=\frac{dt}{\sqrt{t^2-4}}.
\]
\end{conclusion}
\begin{proof}
由定义
\[
\eta^{-1}=\frac{A(y)+3su}{16(2y+1)^3},
\]
故
\[
\eta+\eta^{-1}=\frac{A(y)}{8(2y+1)^3}=t(y),\qquad
\eta-\eta^{-1}=-\frac{3su}{8(2y+1)^3}.
\]
对 \(t=\eta+\eta^{-1}\) 求微分：
\[
dt=(\eta-\eta^{-1})\frac{d\eta}{\eta},
\]
即得
\(
d\log\eta=\frac{dt}{\eta-\eta^{-1}}
\)。
再由定理 \ref{thm:xi-leyang-ty-three-branch-profile}
\[
\frac{dt}{dy}=-\frac{27}{8}\cdot\frac{2y^2-6y+1}{(2y+1)^4},
\]
代入 \(\eta-\eta^{-1}\) 即得
\[
d\log\eta
=\frac{-\frac{27}{8}\frac{2y^2-6y+1}{(2y+1)^4}dy}
{-\frac{3su}{8(2y+1)^3}}
=\frac{9}{s}\frac{2y^2-6y+1}{(2y+1)u}\,dy.
\]
由 \(s=\sqrt{-3}\) 得等价写法。证毕。
\end{proof}

\begin{conclusion}[Lee--Yang 立方判别式的 \texorpdfstring{$-3\Delta_E$}{-3 DeltaE} 结构]\label{con:xi-leyang-cubic-discriminant-minus3-deltae}
设
\[
P(y)=256y^3+411y^2+165y+32.
\]
则
\[
\operatorname{Disc}(P)=-3^9\cdot 31^2\cdot 37.
\]
从而
\[
\QQ\!\bigl(\sqrt{\operatorname{Disc}(P)}\bigr)=\QQ(\sqrt{-111})=\QQ(\sqrt{-3\cdot 37}).
\]
另一方面，对椭圆曲线
\[
E:\ Y^2=X^3-X+\frac14
\]
有
\[
\Delta_E=-16\!\left(4(-1)^3+27\left(\frac14\right)^2\right)=37.
\]
因此
\[
\operatorname{Disc}(P)=-3^9\cdot 31^2\cdot \Delta_E,
\qquad
\QQ\!\bigl(\sqrt{\operatorname{Disc}(P)}\bigr)=\QQ\!\bigl(\sqrt{-3\Delta_E}\bigr).
\]
\end{conclusion}
\begin{proof}
第一式由三次多项式判别式闭式直接代入系数化简得到；
\(\Delta_E\) 由短 Weierstrass 判别式公式
\(
\Delta=-16(4a^3+27b^2)
\)
（\(a=-1,b=\tfrac14\)）即得。两式合并推出结论。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$J(H_{\mathrm{LY}})$}{J(HLY)} 的有理 \texorpdfstring{$2$}{2}-挠子完全刻画]\label{thm:xi-leyang-jacobian-rational-2torsion-complete}
\leanverified{paper\_xi\_leyang\_jacobian\_rational\_2torsion\_complete}
记
\[
J:=\mathrm{Jac}(H_{\mathrm{LY}}),
\qquad
e_0:=\bigl[(0,0)-\infty\bigr],\qquad
e_1:=\bigl[(1,0)-\infty\bigr].
\]
则
\[
J(\QQ)[2]\cong (\ZZ/2\ZZ)^2,
\]
并由 \(e_0,e_1\) 生成。
\end{theorem}
\begin{proof}
在模型 \(u^2=-y(y-1)P(y)\) 上，
\[
\operatorname{div}(y)=2(0,0)-2\infty,\qquad
\operatorname{div}(y-1)=2(1,0)-2\infty,
\]
故 \(2e_0=2e_1=0\)，且 \(e_0,e_1\) 非零且线性独立，从而
\(
(\ZZ/2\ZZ)^2\subseteq J(\QQ)[2]
\)。

再看反向包含。\(H_{\mathrm{LY}}\) 的 Weierstrass 点为
\[
\infty,\ (0,0),\ (1,0),\ \text{以及 }P(y)=0\text{ 的三点}.
\]
由于 \(P\) 在 \(\QQ\) 上无有理根，三点构成大小 \(3\) 的单一 Galois 轨道。
属 \(2\) Jacobian 的 \(2\)-挠点可由 Weierstrass 点偶子集（模补集）描述；
\(\QQ\)-有理 \(2\)-挠点对应 Galois 稳定类。
在上述轨道分解下，新增稳定偶子集仅给出已由
\(\{\infty,(0,0)\}\)、\(\{\infty,(1,0)\}\) 生成的同一三非零类，
不产生新的独立 \(2\)-挠点。
故 \(J(\QQ)[2]\) 恰为由 \(e_0,e_1\) 生成的 \((\ZZ/2\ZZ)^2\)。证毕。
\end{proof}
