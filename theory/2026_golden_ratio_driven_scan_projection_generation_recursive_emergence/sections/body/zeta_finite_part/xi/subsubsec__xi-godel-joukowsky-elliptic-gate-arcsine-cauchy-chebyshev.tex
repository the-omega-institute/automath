\paragraph{Gödel--Joukowsky 椭圆化门的径向统计：反正弦律与定宽窗口}
定理 \ref{thm:xi-cayley-joukowsky-harmonic-measure-ellipse} 把“奇异环”的圆周角测度推前为椭圆外域的调和测度。
在该制度下，径向信息 \(|S|\) 的全部统计可由 \(\pm2\) 分支支集的刚性与恒等式 \(a(L)^2-b(L)^2=4\) 显式读出。

\begin{theorem}[模平方的反正弦律与“定宽窗口”]\label{thm:xi-joukowsky-modsquare-arcsine-fixedwidth}
\leanverified{paper\_xi\_joukowsky\_modsquare\_arcsine\_fixedwidth}
令 \(W\sim \nu_L\)（定理 \ref{thm:xi-cayley-joukowsky-harmonic-measure-ellipse}），并令 \(\Theta\sim\mathrm{Unif}[0,2\pi)\)。
则
\[
|W|^2\ \stackrel{d}{=}\ L+L^{-1}+2\cos(2\Theta),
\]
从而 \(|W|^2\) 的分布为区间
\[
\bigl[b(L)^2,\ a(L)^2\bigr]
=\bigl[(\sqrt{L}-L^{-1/2})^2,\ (\sqrt{L}+L^{-1/2})^2\bigr]
\]
上的反正弦分布，其密度为
\[
f_L(t)=\frac{1}{\pi\sqrt{\bigl(t-b(L)^2\bigr)\bigl(a(L)^2-t\bigr)}},
\qquad
b(L)^2<t<a(L)^2.
\]
特别地，
\[
\mathrm{length}\bigl(\mathrm{supp}(|W|^2)\bigr)=a(L)^2-b(L)^2=4,
\]
即 \(|W|^2\) 的支撑区间长度与 \(L\) 无关。
\end{theorem}

\begin{proof}
由 \(W=J_L(e^{\mathrm{i}\Theta})=\sqrt{L}e^{\mathrm{i}\Theta}+L^{-1/2}e^{-\mathrm{i}\Theta}\) 得
\[
|W|^2=(\sqrt{L}e^{\mathrm{i}\Theta}+L^{-1/2}e^{-\mathrm{i}\Theta})(\sqrt{L}e^{-\mathrm{i}\Theta}+L^{-1/2}e^{\mathrm{i}\Theta})
:=L+L^{-1}+e^{2\mathrm{i}\Theta}+e^{-2\mathrm{i}\Theta}.
\]
故 \(|W|^2=L+L^{-1}+2\cos(2\Theta)\)。
又 \(\cos(2\Theta)\) 在 \([-1,1]\) 上服从反正弦律，做仿射变换得到所示密度；
端点差 \(a(L)^2-b(L)^2=(\sqrt{L}+L^{-1/2})^2-(\sqrt{L}-L^{-1/2})^2=4\) 给出“定宽窗口”。证毕。
\end{proof}

\begin{corollary}[偶次径向矩与零阶 Bessel 因子分解]\label{cor:xi-joukowsky-modsquare-moments-bessel}
\leanverified{paper\_xi\_joukowsky\_modsquare\_moments\_bessel}
沿用定理 \ref{thm:xi-joukowsky-modsquare-arcsine-fixedwidth} 的记号。
令 \(B(L):=\sqrt{L}-L^{-1/2}\)。
则对任意整数 \(m\ge 0\)，成立闭式矩公式
\[
\EE_{\nu_L}\bigl[|W|^{2m}\bigr]
=\sum_{j=0}^m \binom{m}{j}\binom{2j}{j}\,B(L)^{2(m-j)}.
\]
并且对任意复数 \(s\)（在绝对收敛域内，或作解析延拓）有指数矩分解
\[
\EE_{\nu_L}\bigl[e^{s|W|^2}\bigr]
=e^{s(L+L^{-1})}\,I_0(2s),
\qquad
I_0(2s):=\frac{1}{2\pi}\int_{0}^{2\pi}e^{2s\cos\theta}\,\dd\theta.
\]
从而中心化后的指数矩完全与 \(L\) 无关：
\[
\EE_{\nu_L}\bigl[e^{s(|W|^2-\EE|W|^2)}\bigr]=I_0(2s),
\qquad
\EE|W|^2=L+L^{-1}.
\]
\end{corollary}

\begin{proof}
由 \(|W|^2=(\sqrt{L}-L^{-1/2})^2+4\cos^2\Theta=B(L)^2+4\cos^2\Theta\) 及二项式展开，
\[
|W|^{2m}=\sum_{j=0}^{m}\binom{m}{j}B(L)^{2(m-j)}4^j\cos^{2j}\Theta.
\]
用积分恒等式 \(\EE[\cos^{2j}\Theta]=\binom{2j}{j}/4^j\) 得矩公式。
指数矩由
\[
\EE[e^{s|W|^2}]
=e^{s(L+L^{-1})}\cdot \frac{1}{2\pi}\int_{0}^{2\pi}e^{2s\cos(2\theta)}\,\dd\theta
=e^{s(L+L^{-1})}I_0(2s)
\]
得到；中心化版本剥离前因子即得。证毕。
\end{proof}

\begin{proposition}[任一分位数的反演：径向信息的单观测可逆性]\label{prop:xi-joukowsky-modsquare-quantile-inversion}
\leanverified{paper\_xi\_joukowsky\_modsquare\_quantile\_inversion}
取任意 \(\alpha\in(0,1)\)，令 \(q_\alpha(L)\) 表示 \(|W|^2\) 在 \(\nu_L\) 下的 \(\alpha\)-分位数。
则有显式公式
\[
q_\alpha(L)=L+L^{-1}-2\cos(\pi\alpha).
\]
特别地，对任意固定 \(\alpha\in(0,1)\)，映射 \(L\mapsto q_\alpha(L)\) 在 \(L\ge 1\) 上严格单调并可逆；
令 \(c:=q_\alpha(L)+2\cos(\pi\alpha)\)，则反演为
\[
\sqrt{L}=\sqrt{\frac{c+\sqrt{c^2-4}}{2}},
\qquad
L=\frac{c+\sqrt{c^2-4}}{2}.
\]
\end{proposition}

\begin{proof}
由定理 \ref{thm:xi-joukowsky-modsquare-arcsine-fixedwidth}，\(|W|^2=c_0+2X\) 且 \(c_0=L+L^{-1}\)，其中 \(X=\cos(2\Theta)\) 为 \([-1,1]\) 上反正弦分布并满足
\(
\PP(X\le x)=\frac12+\frac{1}{\pi}\arcsin(x)
\)。
解 \(\alpha=\frac12+\frac{1}{\pi}\arcsin(x)\) 得 \(x=\sin(\pi(\alpha-\tfrac12))=-\cos(\pi\alpha)\)，代回即得分位数公式。
反演由方程 \(L+L^{-1}=c\) 的显式解得出；在 \(L\ge 1\) 上取 \(L=\frac{c+\sqrt{c^2-4}}{2}\) 给出单值反演。证毕。
\end{proof}

\begin{theorem}[容量与平衡测度：椭圆满体的外域规一化]\label{thm:xi-joukowsky-capacity-equilibrium-measure}
令 \(K_L\) 为椭圆 \(E_L\) 所围成的紧致满椭圆（其多项式凸包）。
则其对数容量满足
\[
\mathrm{cap}(K_L)=\sqrt{L},
\]
并且 \(\nu_L\) 恰为 \(K_L\) 的平衡测度（等价地：无穷远处的调和测度），即在所有支撑于 \(K_L\) 的概率测度中唯一极小化对数能量。
\end{theorem}

\begin{proof}
考虑外域共形映射 \(f_L(z)=\sqrt{L}\,z+L^{-1/2}z^{-1}\)，它把 \(\{|z|>1\}\) 双全纯映到 \(\CC\setminus K_L\)，并在无穷远处展开为 \(f_L(z)=\sqrt{L}\,z+O(z^{-1})\)。
经典势论表明：该外域规一化的首项系数即为容量，故 \(\mathrm{cap}(K_L)=\sqrt{L}\)；同时外域在无穷远处的调和测度沿边界参数 \(z=e^{\mathrm{i}\theta}\) 的推前即为 \(\nu_L\)，并与平衡测度一致。证毕。
\end{proof}

\begin{theorem}[Joukowsky 椭圆的 Dirichlet--to--Neumann 共轭公式：外域能量的严格边界字典]\label{thm:xi-joukowsky-dtn-conjugacy}
令 \(L>1\)，并令
\[
f_L(z):=\sqrt{L}\,z+L^{-1/2}z^{-1}\qquad(|z|>1)
\]
为把外域 \(\{|z|>1\}\) 双全纯映到 \(\CC\setminus K_L\) 的共形映射（见定理 \ref{thm:xi-joukowsky-capacity-equilibrium-measure} 的证明）。
取 \(f\in H^{1/2}(\TT)\)（实值或复值皆可），并令 \(V\) 为外圆盘上的调和函数，满足边界迹 \(V(e^{\mathrm{i}\theta})=f(e^{\mathrm{i}\theta})\) 且 \(V(z)\to 0\)（\(|z|\to\infty\)）。
定义椭圆边界数据 \(\tilde f\) 为
\[
\tilde f\bigl(f_L(e^{\mathrm{i}\theta})\bigr):=f(e^{\mathrm{i}\theta}),
\]
并令 \(U\) 为 \(\CC\setminus K_L\) 上满足 \(U(w)\to 0\)（\(|w|\to\infty\)）且边界迹 \(U|_{E_L}=\tilde f\) 的调和函数。
记外域 Dirichlet--to--Neumann 算子
\[
\Lambda_{\TT}f:=-\partial_{n}V\big|_{\TT},
\qquad
\Lambda_{E_L}\tilde f:=-\partial_{n}U\big|_{E_L},
\]
其中 \(n\) 均取外法单位法向（指向外域）。
则对 \(w=f_L(e^{\mathrm{i}\theta})\in E_L\) 有严格共轭恒等式
\[
\boxed{\;
(\Lambda_{E_L}\tilde f)(w)=\frac{1}{|f_L'(e^{\mathrm{i}\theta})|}\,(\Lambda_{\TT}f)(e^{\mathrm{i}\theta}).\;}
\]
并且 Dirichlet 能量在共形输运下保持：
\[
\int_{\CC\setminus K_L}|\nabla U|^2\,\dd A
=\int_{|z|>1}|\nabla V|^2\,\dd A
=\langle f,\Lambda_{\TT}f\rangle_{L^2(\dd\theta)}
=\langle \tilde f,\Lambda_{E_L}\tilde f\rangle_{L^2(\dd s)}.
\]
此外在边界参数化上成立几何因子恒等式
\[
|f_L'(e^{\mathrm{i}\theta})|
=\left|\sqrt{f_L(e^{\mathrm{i}\theta})^{2}-4}\right|,
\]
其绝对值正是后文椭圆侧双分支因子 \(\sqrt{w^2-4}\) 的边界模长。
\end{theorem}
\begin{proof}
由共形性，\(U\circ f_L\) 在 \(|z|>1\) 上调和且其边界迹为 \(f\)，并满足无穷远衰减；由解的唯一性知 \(V=U\circ f_L\)。
\par
对二维共形映射，Dirichlet 积分不变：以 \(w=f_L(z)\) 代换并用 \(\det Df_L(z)=|f_L'(z)|^2\) 与 \(\nabla_z(V)=Df_L(z)^{\mathsf T}\nabla_w(U)\) 的链式关系，可得
\(
|\nabla_w U(w)|^2\,\dd A_w=|\nabla_z V(z)|^2\,\dd A_z
\)，从而 \(\int_{\CC\setminus K_L}|\nabla U|^2=\int_{|z|>1}|\nabla V|^2\)。
\par
法向导数在边界上的变换由共形尺度给出：在 \(z=e^{\mathrm{i}\theta}\) 处有
\[
\partial_n U\bigl(f_L(e^{\mathrm{i}\theta})\bigr)=\frac{1}{|f_L'(e^{\mathrm{i}\theta})|}\,\partial_n V(e^{\mathrm{i}\theta}),
\]
故 \(\Lambda_{E_L}\tilde f=|f_L'|^{-1}\Lambda_{\TT}f\)。
最后，Green 恒等式（或能量--边界配对公式）给出
\(
\int |\nabla V|^2=\langle f,\Lambda_{\TT}f\rangle
\) 与
\(
\int |\nabla U|^2=\langle \tilde f,\Lambda_{E_L}\tilde f\rangle
\)。
\par
几何因子恒等式由直接计算得到：
\(
f_L'(z)=\sqrt{L}-L^{-1/2}z^{-2}
\)，
在 \(|z|=1\) 上有
\(
|f_L'(e^{\mathrm{i}\theta})|
=|\sqrt{L}e^{\mathrm{i}\theta}-L^{-1/2}e^{-\mathrm{i}\theta}|
=|\sqrt{f_L(e^{\mathrm{i}\theta})^2-4}|
\)。
证毕。
\end{proof}

\begin{remark}[外域作用量与 \texorpdfstring{$H^{1/2}$}{H^{1/2}} 边界规范的严格同一]\label{rem:xi-joukowsky-dtn-ads2cft1-dictionary}
定理 \ref{thm:xi-joukowsky-dtn-conjugacy} 给出两条严格同一：
（i）外域 Dirichlet 能量与边界 Dirichlet--to--Neumann 配对相等，从而把外域作用量等价转写为边界 \(H^{1/2}\) 半范数；
（ii）椭圆侧的几何权重由 \(|f_L'|\) 给出，而在边界参数化上 \(|f_L'|=|\sqrt{w^2-4}|\)。
因此该共轭关系在本文语义中实现了一个无余项的“AdS\(_2\)/CFT\(_1\)”型能量字典，并与后续 Cauchy 变换的双分支因子 \(\sqrt{w^2-4}\) 完全对齐。
\end{remark}

\paragraph{Cauchy 变换的双分支分解与奇异环输运}
Joukowsky 二重覆盖以 \([-2,2]\) 为唯一分歧支集（参见命题 \ref{prop:xi-joukowsky-double-cover-uniformization}）；
因此任何由单位圆测度推前得到的椭圆侧解析对象，都携带一个由 \(\sqrt{w^2-4}\) 控制的双分支结构。

\begin{theorem}[Joukowsky 输运的 Cauchy 变换双分支分解]\label{thm:xi-joukowsky-cauchy-transform-two-branch}
设 \(\sigma\) 为 \(\TT\) 上有限复 Borel 测度，并令 \(\sigma_L:=(J_L)_\#\sigma\) 为其在 \(E_L\) 上的推前。
对 \(w\in\CC\setminus\mathrm{supp}(\sigma_L)\)，取分支 \(\sqrt{w^2-4}\sim w\)（\(w\to\infty\)），并定义
\[
z_\pm(w):=\frac{w\pm\sqrt{w^2-4}}{2\sqrt{L}}.
\]
记 \(\TT\) 上的 Cauchy 变换
\[
G_\sigma(z):=\int_{\TT}\frac{1}{\zeta-z}\,\dd\sigma(\zeta),
\]
以及 \(E_L\) 上的 Cauchy 变换
\[
G_{\sigma_L}(w):=\int_{E_L}\frac{1}{w-\xi}\,\dd\sigma_L(\xi).
\]
则成立严格恒等式
\[
\boxed{\;
G_{\sigma_L}(w)=\frac{1}{\sqrt{w^2-4}}
\Bigl(
z_-(w)\,G_\sigma\!\bigl(z_-(w)\bigr)\;-\;z_+(w)\,G_\sigma\!\bigl(z_+(w)\bigr)
\Bigr).\;}
\]
\end{theorem}

\begin{proof}
由推前定义，
\(
G_{\sigma_L}(w)=\int_{\TT}\frac{1}{w-J_L(\zeta)}\,\dd\sigma(\zeta)
\)。
注意
\[
w-J_L(\zeta)=w-\sqrt{L}\,\zeta-\frac{1}{\sqrt{L}}\zeta^{-1}
=-\frac{\sqrt{L}(\zeta-z_+)(\zeta-z_-)}{\zeta},
\qquad
z_\pm=z_\pm(w),
\]
其中 \(z_\pm\) 为二次方程 \(\sqrt{L}\,\zeta^2-w\zeta+L^{-1/2}=0\) 的两根。
于是
\[
\frac{1}{w-J_L(\zeta)}
=\frac{z_-}{\sqrt{w^2-4}}\cdot\frac{1}{\zeta-z_-}-\frac{z_+}{\sqrt{w^2-4}}\cdot\frac{1}{\zeta-z_+},
\]
其中使用了 \(\sqrt{L}(z_+-z_-)=\sqrt{w^2-4}\)。
对 \(\dd\sigma(\zeta)\) 积分并按定义识别 \(G_\sigma\) 得结论。证毕。
\end{proof}

\begin{corollary}[Haar 输运的外域 Cauchy 刚性：椭圆尺度的解析不可见性]\label{cor:xi-joukowsky-haar-cauchy-rigidity}
取 \(\sigma=m_{\TT}\) 为单位圆 \(\TT\) 上的归一化 Haar 测度，并令 \(\nu_L:=(J_L)_\# m_{\TT}\) 为其在椭圆 \(E_L\) 上的推前（定理 \ref{thm:xi-cayley-joukowsky-harmonic-measure-ellipse}）。
则对任意 \(L>0\) 与任意 \(w\in\CC\setminus K_L\)（\(K_L\) 为 \(E_L\) 围成的满椭圆，见定理 \ref{thm:xi-joukowsky-capacity-equilibrium-measure}），其 Cauchy 变换满足严格恒等式
\[
\boxed{\;
G_{\nu_L}(w)=\int_{E_L}\frac{1}{w-\xi}\,\dd\nu_L(\xi)\ =\ \frac{1}{\sqrt{w^2-4}}\; }.
\]
特别地，右端与 \(L\) 无关；因此任何仅由外域解析泛函（等价地由 Cauchy 变换/外域 resolvent）定义的观测量，都无法区分 \(E_L\) 的尺度参数。
\end{corollary}
\begin{proof}
对 Haar 测度 \(m_{\TT}\)，其 Cauchy 变换满足
\[
G_{m_{\TT}}(z)=\int_{\TT}\frac{1}{\zeta-z}\,\dd m_{\TT}(\zeta)=
\begin{cases}
0,& |z|<1,\\
-\dfrac{1}{z},& |z|>1,
\end{cases}
\]
这可由留数定理直接计算得到。
当 \(w\in\CC\setminus K_L\) 时，二次根 \(z_-(w)\) 落在单位圆内而 \(z_+(w)\) 落在单位圆外。
将上述两值代入定理 \ref{thm:xi-joukowsky-cauchy-transform-two-branch} 得
\[
G_{\nu_L}(w)=\frac{1}{\sqrt{w^2-4}}
\Bigl(z_-(w)\cdot 0-z_+(w)\cdot\Bigl(-\frac{1}{z_+(w)}\Bigr)\Bigr)
=\frac{1}{\sqrt{w^2-4}}.
\]
证毕。
\end{proof}

\begin{corollary}[奇异环输运原理：椭圆侧奇异性由圆周像完全控制]\label{cor:xi-joukowsky-singularity-transport}
在定理 \ref{thm:xi-joukowsky-cauchy-transform-two-branch} 的设定下，设 \(G_\sigma\) 的不可解析延拓障碍支撑于闭集 \(S\subset\TT\)。
则 \(G_{\sigma_L}\) 的非解析性除去固有分支点 \(\{\pm2\}\) 外，仅可能出现在集合 \(J_L(S)\subset E_L\) 上。
更精确地，若 \(z_\pm(w)\in S\)，则 \(w\) 为 \(G_{\sigma_L}\) 的候选奇异点；若 \(z_\pm(w)\notin S\) 且 \(w\neq\pm2\)，则 \(G_{\sigma_L}\) 在 \(w\) 的邻域解析。
\end{corollary}

\begin{proof}
由定理 \ref{thm:xi-joukowsky-cauchy-transform-two-branch} 的表示式，奇异性来源仅有两类：\(\sqrt{w^2-4}\) 的分支点 \(\pm2\)，以及 \(G_\sigma(z_\pm(w))\) 的奇异性。
后者的候选奇异点集合为 \(\{w:z_+(w)\in S\}\cup\{w:z_-(w)\in S\}\)，其在边界参数化下的像即为 \(J_L(S)\)。证毕。
\end{proof}

\paragraph{Chebyshev–根式正常形与 \texorpdfstring{$r$}{r}-无关的正交分解}
椭圆上由倍角诱导的自映射在一般 \(L\neq 1\) 情形下必然引入根式项；
而在 \(\nu_L\) 的双线性配对下，Chebyshev–根式基给出与 \(L\) 无关的对角化正交分解。

\begin{proposition}[倍角自映射的 Chebyshev–根式正常形]\label{prop:xi-joukowsky-multiple-angle-chebyshev-radical-normalform}
\leanverified{paper\_xi\_joukowsky\_multiple\_angle\_chebyshev\_radical\_normalform}
令 \(L>1\) 并定义椭圆上的 \(n\) 倍角自映射
\[
\Phi_{n,L}\bigl(J_L(e^{\mathrm{i}\theta})\bigr):=J_L(e^{\mathrm{i}n\theta})
\qquad(n\in\NN).
\]
取归一化 Chebyshev 多项式 \(C_n(w):=2T_n(w/2)\) 与第二类多项式 \(U_{n-1}\)。
则对任意 \(w\in E_L\)（在命题 \ref{prop:xi-joukowsky-double-cover-uniformization} 的两层单值化上取 \(\sqrt{w^2-4}\)），有代数恒等式
\[
\Phi_{n,L}(w)=a_n(L)\,C_n(w)+b_n(L)\,\sqrt{w^2-4}\;U_{n-1}\!\Bigl(\frac{w}{2}\Bigr),
\]
其中
\[
a_n(L)=\frac{L^{\frac{n-1}{2}}+L^{-\frac{n-1}{2}}}{2},
\qquad
b_n(L)=\frac{L^{-\frac{n-1}{2}}-L^{\frac{n-1}{2}}}{2}.
\]
并且 \(b_n(L)=0\) 当且仅当 \(L=1\)，此时 \(\Phi_{n,1}(w)=C_n(w)\) 退化为纯多项式动力学。
\end{proposition}

\begin{proof}
令 \(r=\sqrt{L}\) 且 \(u:=r e^{\mathrm{i}\theta}\)，则在 \(E_L\) 上 \(w=J_L(e^{\mathrm{i}\theta})=u+u^{-1}\)。
于是
\[
\Phi_{n,L}(w)=r e^{\mathrm{i}n\theta}+r^{-1}e^{-\mathrm{i}n\theta}=r^{1-n}u^n+r^{n-1}u^{-n}.
\]
将右端分解为对称与反对称组合，并用恒等式 \(u^n+u^{-n}=C_n(u+u^{-1})=C_n(w)\) 与
\(
u^n-u^{-n}=\sqrt{w^2-4}\,U_{n-1}(w/2)
\)
得到所示正常形；系数由 \(r^{\pm(n-1)}=L^{\pm(n-1)/2}\) 代入即得。
最后，\(b_n(L)=0\iff r^{n-1}=r^{1-n}\iff r=1\iff L=1\)。证毕。
\end{proof}

\begin{theorem}[Chebyshev 正交性：\texorpdfstring{$L$}{L}-无关的双线性谱分解]\label{thm:xi-joukowsky-chebyshev-orthogonality-L-invariant}
\leanverified{paper\_xi\_joukowsky\_chebyshev\_orthogonality\_l\_invariant}
在 \(E_L\) 上定义双线性矩配对
\[
\langle f,g\rangle_L:=\int_{E_L} f(w)\,g(w)\,\dd\nu_L(w).
\]
对 \(n,m\ge 1\) 令 \(C_n(w)=2T_n(w/2)\)，并令
\[
D_n(w):=\sqrt{w^2-4}\;U_{n-1}\!\Bigl(\frac{w}{2}\Bigr)
\]
（同命题 \ref{prop:xi-joukowsky-multiple-angle-chebyshev-radical-normalform} 的分支约定）。
则成立严格正交关系
\[
\langle C_n,C_m\rangle_L=2\,\delta_{nm},
\qquad
\langle D_n,D_m\rangle_L=-2\,\delta_{nm},
\qquad
\langle C_n,D_m\rangle_L=0,
\]
并且正交常数完全与 \(L\) 无关。
\end{theorem}

\begin{proof}
由 \(\nu_L=\dd\theta/(2\pi)\) 且 \(u=\sqrt{L}e^{\mathrm{i}\theta}\)（定理 \ref{thm:xi-cayley-joukowsky-harmonic-measure-ellipse}），有 \(C_n(w)=u^n+u^{-n}\) 与 \(D_n(w)=u^n-u^{-n}\) 作为 \(\theta\)-函数恒等式。
展开后均为有限个 \(e^{\mathrm{i}k\theta}\) 的线性组合；除 \(k=0\) 外积分为零，而 \(k=0\) 仅在 \(n=m\) 时由 \(u^n u^{-m}\) 与 \(u^{-n}u^m\) 两项贡献。
因此 \(\langle C_n,C_m\rangle_L=2\delta_{nm}\)。
同理 \((u^n-u^{-n})^2=u^{2n}-2+u^{-2n}\) 给出 \(\langle D_n,D_n\rangle_L=-2\)，并由奇偶对称消去得到 \(\langle C_n,D_m\rangle_L=0\)。
所有结论不依赖于 \(\sqrt{L}\) 的尺度因子，因为只在 \(k=0\) 时出现 \(u^k\) 的常数项。证毕。
\end{proof}

\begin{conjecture}[奇异环--Langlands--Riemann 的可审计函子性]\label{conj:xi-singular-ring-langlands-riemann-auditable-functor}
预期存在一类以“自对偶有界型函数与奇异环测度”为对象的函子 \(\mathcal F\)，把适当范畴中的自对偶自守表示 \(\pi\)（或其完成化 \(L\)-函数 \(\Lambda(s,\pi)\)）关联到一对数据
\[
\mathcal F(\pi)=\bigl(F_\pi,\sigma_\pi\bigr),
\]
其中 \(F_\pi\) 为单位圆盘上的自对偶有界型函数，\(\sigma_\pi\) 为 \(\TT\) 上有限正测度，并满足：
\begin{enumerate}
  \item \(F_\pi\) 的边界奇异内因子由 \(\sigma_\pi\) 的奇异分量完全刻画（以 Herglotz--Nevanlinna 表示为桥梁），并且所有“离临界线偏移信息”在 Cayley 模型中可等价转写为 \(\sigma_\pi\) 的奇异部分；
  \item 算术导体的改变在 \((F_\pi,\sigma_\pi)\) 侧对应于 Joukowsky--椭圆化门的输运：\(\sigma_\pi\mapsto (J_L)_\#\sigma_\pi\)，其中 \(L\) 为由导体指数诱导的椭圆化尺度，并与局部--整体函子性相容；
  \item 广义 Riemann 假设（对 \(\pi\)）等价于 \(\sigma_\pi\) 的奇异分量为零（“奇异环无质量”），从而可转化为单位圆上的正性/Toeplitz 型可审计判据。
\end{enumerate}
该猜想的“可计算入口”由定理 \ref{thm:xi-joukowsky-cauchy-transform-two-branch} 与推论 \ref{cor:xi-joukowsky-singularity-transport} 支撑：一旦 \(\sigma_\pi\) 被识别，椭圆化后的解析观测由双分支 Cauchy 表示式完全控制，从而把导体变形下的不变量与奇异环上的缺陷在同一框架内对齐。
\end{conjecture}
