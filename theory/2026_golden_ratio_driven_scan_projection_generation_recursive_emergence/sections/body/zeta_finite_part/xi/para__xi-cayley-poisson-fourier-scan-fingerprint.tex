\paragraph{Cayley 模长扫描律的等价重述与频域可逆层析}\label{par:xi-cayley-poisson-fourier-scan-fingerprint}
本段把段落 \ref{par:xi-comoving-integrated-jensen-defect-sumrule} 与段落
\ref{par:xi-endpoint-jensen-area-law-rational-inversion} 的核恒等式改写为一组统一接口：
由单核 \(u_\delta\) 的 Poisson 尺度表示出发，给出有限缺陷势 \(J_\zeta\) 的 Fourier--Laplace 指纹、\(|D|\)-互织以及扫描迹判据。

\begin{theorem}[Cayley 模长的 Poisson 尺度积分表示与 Fourier 指纹]\label{thm:xi-cayley-modulus-poisson-fourier-fingerprint-layered}
设
\[
\mathcal C(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t},
\qquad
\mathcal C^{-1}(w)=-\mathrm{i}\,\frac{w-1}{w+1},
\qquad
P_s(x)=\frac{1}{\pi}\frac{s}{x^2+s^2}\ (s>0).
\]
对任意 \(\delta\in(0,1)\) 与 \(x\in\RR\)，定义
\[
u_\delta(x):=-\log\bigl|\mathcal C(x+\mathrm{i}\delta)\bigr|.
\]
则有：
\begin{enumerate}
  \item \textbf{闭式对数比表示}
  \[
  u_\delta(x)=\frac12\log\frac{x^2+(1+\delta)^2}{x^2+(1-\delta)^2};
  \]
  \item \textbf{Poisson 尺度积分表示}
  \[
  u_\delta(x)=\pi\int_{1-\delta}^{1+\delta}P_s(x)\,\dd s;
  \]
  \item \textbf{Fourier--Laplace 指纹}
  （约定 \(\widehat f(\xi)=\int_{\RR}e^{-\mathrm{i}\xi x}f(x)\dd x\)）：
  对 \(\xi\neq 0\)，
  \[
  \widehat{u_\delta}(\xi)
  =\pi\,\frac{e^{-(1-\delta)|\xi|}-e^{-(1+\delta)|\xi|}}{|\xi|},
  \]
  且连续延拓到 \(\xi=0\) 时
  \[
  \widehat{u_\delta}(0)=\int_{\RR}u_\delta(x)\,\dd x=2\pi\delta.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-cayley-modulus-delta} 立得第 1 条。
第 2 条由
\(
\int_a^b P_s(x)\dd s=\frac{1}{2\pi}\log\frac{x^2+b^2}{x^2+a^2}
\)
取 \(a=1-\delta,b=1+\delta\) 即得。
第 3 条由第 2 条与
\(
\widehat{P_s}(\xi)=e^{-s|\xi|}
\)
逐点 Fourier 变换得到；
当 \(\xi\to 0\) 时分子
\(
e^{-(1-\delta)|\xi|}-e^{-(1+\delta)|\xi|}
=2\delta|\xi|+o(|\xi|)
\)，故极限为 \(2\pi\delta\)。
\end{proof}

\begin{corollary}[深度参数的扫描面积律与低频迹公式]\label{cor:xi-cayley-depth-scan-area-lowfreq}
在定理 \ref{thm:xi-cayley-modulus-poisson-fourier-fingerprint-layered} 的记号下，
对任意有限离散缺陷谱
\[
\nu=\sum_{j=1}^{\kappa}m_j\,\delta_{(\gamma_j,\delta_j)},
\qquad
m_j\in\NN,\ \delta_j\in(0,1),
\]
定义
\[
J_\nu(x):=\sum_{j=1}^{\kappa}m_j\,u_{\delta_j}(x-\gamma_j).
\]
则
\[
\int_{\RR}J_\nu(x)\,\dd x=2\pi\sum_{j=1}^{\kappa}m_j\delta_j,
\qquad
\widehat{J_\nu}(0)=2\pi\sum_{j=1}^{\kappa}m_j\delta_j.
\]
并且 \(\widehat{J_\nu}\) 在 \(\xi=0\) 连续，满足
\[
\widehat{J_\nu}(\xi)
=\sum_{j=1}^{\kappa}
m_j\pi\frac{e^{-(1-\delta_j)|\xi|}-e^{-(1+\delta_j)|\xi|}}{|\xi|}e^{-\mathrm{i}\gamma_j\xi}
\quad(\xi\neq 0).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-cayley-modulus-poisson-fourier-fingerprint-layered} 逐项平移与线性叠加即得。
\end{proof}

\begin{theorem}[有限缺陷口径下端点 Jensen 潜势的显式分解与尾部主项]\label{thm:xi-finite-defect-endpoint-jensen-explicit-tail}
设离临界线零点有限，记 \(\rho_j=\beta_j+\mathrm{i}\gamma_j\)（重数 \(m_j\)）、
\(\delta_j=\frac12-\beta_j\in(0,1)\)。
定义
\[
J_\zeta(x_0):=\lim_{\varrho\uparrow 1}D_{\zeta,x_0}(\varrho).
\]
则
\[
J_\zeta(x_0)=\sum_{j=1}^{\kappa}m_j\,u_{\delta_j}(x_0-\gamma_j),
\]
并且 \(J_\zeta\in L^1(\RR)\cap C^\omega(\RR)\)。
此外，当 \(|x_0|\to\infty\) 时，
\[
J_\zeta(x_0)=\frac{2}{x_0^2}\sum_{j=1}^{\kappa}m_j\delta_j+O(|x_0|^{-3}).
\]
若附加中心化条件
\(
\sum_j m_j\delta_j\gamma_j=0
\)，则提高为
\[
J_\zeta(x_0)=\frac{2}{x_0^2}\sum_{j=1}^{\kappa}m_j\delta_j+O(|x_0|^{-4}).
\]
\end{theorem}
\begin{proof}
分解式由 \eqref{eq:xi-limit-defect-potential} 与
\eqref{eq:xi-log-defect-potential} 直接得到。
\(L^1\) 与解析性由定理 \ref{thm:xi-limit-defect-potential-fourier-injective} 与
推论 \ref{cor:xi-limit-defect-potential-L1-sumrule}。
尾部展开由
\[
u_\delta(x)=\frac12\log\!\left(1+\frac{4\delta}{x^2+(1-\delta)^2}\right)
=\frac{2\delta}{x^2}+O(|x|^{-4})
\]
代入有限和并展开 \(x=x_0-\gamma_j\) 得到。
\end{proof}

\begin{theorem}[Jensen 潜势的 Fourier--Laplace 指纹与缺陷谱注入性]\label{thm:xi-jensen-potential-fourier-laplace-injective}
\leanverified{paper\_xi\_jensen\_potential\_fourier\_laplace\_injective}
在定理 \ref{thm:xi-finite-defect-endpoint-jensen-explicit-tail} 的设定下，对 \(\xi\neq 0\) 有
\[
\widehat{J_\zeta}(\xi)
=\sum_{j=1}^{\kappa}m_j\,
\pi\frac{e^{-(1-\delta_j)|\xi|}-e^{-(1+\delta_j)|\xi|}}{|\xi|}\,e^{-\mathrm{i}\gamma_j\xi}.
\]
令 \(s>0\) 并定义
\[
G(s):=\frac{s}{\pi}\widehat{J_\zeta}(s)
=\sum_{j=1}^{\kappa}m_j\Bigl(e^{-(1-\delta_j+\mathrm{i}\gamma_j)s}-e^{-(1+\delta_j+\mathrm{i}\gamma_j)s}\Bigr).
\]
则 \(G\) 为有限指数和；其指数集合与系数唯一决定多重集
\(
\{(\gamma_j,\delta_j,m_j)\}_{j=1}^{\kappa}
\)。
换言之，映射
\[
\{(\gamma_j,\delta_j,m_j)\}_{j=1}^{\kappa}\longmapsto J_\zeta(\cdot)
\]
在有限缺陷类别上为注入映射。
\end{theorem}
\begin{proof}
Fourier 公式由上一推论逐项应用得到。
若两组有限数据给出同一 \(J_\zeta\)，则差函数对应 \(G\equiv 0\)。
于是得到
\(
\sum_{\ell=1}^{N}c_\ell e^{-\lambda_\ell s}=0\ (s>0)
\)
（\(\Re\lambda_\ell>0\)，\(\lambda_\ell\) 两两不同）。
该函数解析延拓后恒零，故在 \(s=0\) 处各阶导均为零：
\(
\sum_\ell c_\ell(-\lambda_\ell)^n=0
\)。
由 Vandermonde 可逆性得 \(c_\ell=0\) 全部成立，故指数与系数按重数一致，进而恢复全部 \((\gamma,\delta,m)\)。
\end{proof}

\begin{proposition}[Jensen--Poisson 互织：\texorpdfstring{$|D|$}{|D|} 产生双尺度 Poisson 剖面]\label{prop:xi-jensen-poisson-intertwiner-two-scale}
在有限缺陷设定下，令 \(|D|\) 为 Fourier 乘子 \(|\xi|\)。

|D|\,J_\zeta

P_t*(|D|J_\zeta)

\end{proposition}
\begin{proof}
由命题 \ref{prop:xi-dtn-acts-on-jensen-kernel} 对单核成立的恒等式逐项求和即得。
\end{proof}

\begin{theorem}[有限缺陷口径下 RH 的扫描迹等价判据]\label{thm:xi-finite-defect-rh-scan-trace-equivalence}
\leanverified{paper\_xi\_finite\_defect\_rh\_scan\_trace\_equivalence}
定义
\[
\mathcal J(\varrho):=\frac{1}{2\pi}\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0.
\]
在有限缺陷设定下有
\[
\lim_{\varrho\uparrow 1}\mathcal J(\varrho)
=\sum_{j=1}^{\kappa}m_j\delta_j.
\]
并且以下命题等价：
\begin{enumerate}
  \item RH 成立；
  \item \(\displaystyle \lim_{\varrho\uparrow 1}\mathcal J(\varrho)=0\)；
  \item \(\widehat{J_\zeta}(0)=0\)（等价于 \(\int_{\RR}J_\zeta(x)\dd x=0\)）。
\end{enumerate}
\end{theorem}
\begin{proof}
第一式由定理 \ref{thm:xi-integrated-defect-sumrule-endpoint-flux} 立得。
等价性 (1)\(\Leftrightarrow\)(2) 即推论 \ref{cor:xi-scalar-certificate-integrated-defect}。
而
\[
\widehat{J_\zeta}(0)=\int_{\RR}J_\zeta(x)\,\dd x
=\lim_{\varrho\uparrow 1}\int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0
=2\pi\lim_{\varrho\uparrow 1}\mathcal J(\varrho),
\]
故 (2)\(\Leftrightarrow\)(3)。
\end{proof}

\begin{proposition}[有限半径扫描迹闭式与高度变量消去]\label{prop:xi-finite-rho-scan-trace-closed-form}
对 \(\delta\in(0,1)\)、\(\varrho\in(0,1)\) 定义
\[
K_{\varrho,\delta}(x):=\left[\log\frac{\varrho}{|\mathcal C(x+\mathrm{i}\delta)|}\right]_+.
\]
则：
\begin{enumerate}
  \item 若 \(\varrho\le\frac{1-\delta}{1+\delta}\)，则 \(K_{\varrho,\delta}\equiv 0\)；
  \item 若 \(\varrho>\frac{1-\delta}{1+\delta}\)，则其支集为 \([-X_*,X_*]\)，其中
  \[
  X_*^2=\frac{\varrho^2(1+\delta)^2-(1-\delta)^2}{1-\varrho^2},
  \]
  并有闭式
  \[
  \int_{\RR}K_{\varrho,\delta}(x)\,\dd x
  =2X_*\log\varrho
  +X_*\log\frac{X_*^2+(1+\delta)^2}{X_*^2+(1-\delta)^2}
  \]
  \[
  \qquad\qquad
  +2\Bigl((1+\delta)\arctan\frac{X_*}{1+\delta}
  -(1-\delta)\arctan\frac{X_*}{1-\delta}\Bigr);
  \]
  \item 在有限缺陷下，
  \[
  \int_{\RR}D_{\zeta,x_0}(\varrho)\,\dd x_0
  =\sum_{j=1}^{\kappa}m_j\int_{\RR}K_{\varrho,\delta_j}(x)\,\dd x,
  \]
  因而该扫描迹只依赖 \(\{(\delta_j,m_j)\}\)，与 \(\{\gamma_j\}\) 无关。
\end{enumerate}
\end{proposition}
\begin{proof}
第 1、2 条由命题 \ref{prop:xi-single-defect-integrated-closed-form} 的推导过程与
\eqref{eq:xi-single-defect-X}、\eqref{eq:xi-single-defect-integrated-intermediate} 直接得到（其中 \(\mathcal D_{\varrho,\delta}\) 与 \(K_{\varrho,\delta}\) 同义）。
第 3 条由定理 \ref{thm:xi-comoving-jensen-defect-kernel-decomposition} 与变量平移 \(x=\gamma_j-x_0\) 逐项积分即得。
\end{proof}
