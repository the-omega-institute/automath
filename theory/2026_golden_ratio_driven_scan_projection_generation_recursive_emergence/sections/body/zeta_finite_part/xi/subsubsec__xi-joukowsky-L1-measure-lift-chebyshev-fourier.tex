\begin{proposition}[\texorpdfstring{$L=1$}{L=1} 退化下的测度提升与 Chebyshev--Fourier 传输]\label{prop:xi-joukowsky-L1-measure-lift-chebyshev-fourier}
\leanverified{paper\_xi\_joukowsky\_l1\_measure\_lift\_chebyshev\_fourier}
令
\[
J_1:\TT\to[-2,2],\qquad J_1(\zeta)=\zeta+\zeta^{-1}.
\]
记 \(\mathscr{M}_{\mathrm{inv}}(\TT)\) 为满足反演不变性 \(\sigma(A)=\sigma(A^{-1})\) 的有限 Borel 测度全体，并记 \(\mathscr{M}([-2,2])\) 为 \([-2,2]\) 上的有限 Borel 测度全体。
则推前映射
\[
\mathcal{J}:\mathscr{M}_{\mathrm{inv}}(\TT)\to\mathscr{M}([-2,2]),
\qquad
\mathcal{J}(\sigma):=(J_1)_{\#}\sigma
\]
为双射，其逆可由如下提升给出：对任意 \(\mu\in\mathscr{M}([-2,2])\)，令
\[
s_{\pm}(x):=\exp\!\bigl(\pm\mathrm{i}\arccos(x/2)\bigr)\in\TT,
\qquad
\sigma:=\tfrac12(s_{+})_{\#}\mu+\tfrac12(s_{-})_{\#}\mu,
\]
则 \(\sigma\in\mathscr{M}_{\mathrm{inv}}(\TT)\) 且 \((J_1)_{\#}\sigma=\mu\)。
\par
并且对任意 \(n\ge 0\)，其 Chebyshev--Fourier 矩满足严格恒等式
\[
\int_{-2}^{2} T_n(x/2)\,\dd\mu(x)=\int_{\TT}\zeta^n\,\dd\sigma(\zeta),
\]
其中 \(T_n\) 为第一类 Chebyshev 多项式。
\end{proposition}
\begin{proof}
若 \(\sigma\in\mathscr{M}_{\mathrm{inv}}(\TT)\) 且 \((J_1)_{\#}\sigma=\mu\)，则对任意 Borel 集 \(B\subset\TT\) 有
\[
\sigma(B)=\tfrac12\sigma(B)+\tfrac12\sigma(B^{-1})
=\tfrac12\sigma(B\cup B^{-1})
=\tfrac12\mu(J_1(B)),
\]
其中用到 \(J_1^{-1}(J_1(B))=B\cup B^{-1}\)。
故 \(\sigma\) 由 \(\mu\) 唯一确定。
\par
反之，对给定 \(\mu\) 构造的 \(\sigma\) 显然满足 \(\sigma(A)=\sigma(A^{-1})\)，且由 \(J_1\circ s_{\pm}=\mathrm{id}_{[-2,2]}\) 得 \((J_1)_{\#}\sigma=\mu\)，从而 \(\mathcal{J}\) 为双射。
\par
矩恒等式由恒等式
\[
T_n\!\Bigl(\frac{\zeta+\zeta^{-1}}{2}\Bigr)=\frac{\zeta^n+\zeta^{-n}}{2}
\]
与反演不变性给出：\(\int_{\TT}\zeta^{-n}\,\dd\sigma=\int_{\TT}\zeta^{n}\,\dd\sigma\)，从而
\[
\int_{-2}^{2}T_n(x/2)\,\dd\mu(x)
=\int_{\TT}\frac{\zeta^n+\zeta^{-n}}{2}\,\dd\sigma(\zeta)
=\int_{\TT}\zeta^n\,\dd\sigma(\zeta).
\]
证毕。
\end{proof}
