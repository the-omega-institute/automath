
\begin{corollary}[密度比的二阶剖面极限与一致展开]\label{cor:xi-cauchy-poisson-density-ratio-second-order-profile}
\leanverified{paper\_xi\_cauchy\_poisson\_density\_ratio\_second\_order\_profile}
在命题 \ref{prop:xi-poisson-tv-kl-second-order} 的设定下，若进一步满足四阶矩有限
\[
\int_{\RR}|\gamma-\bar\gamma|^4\,\dd\tilde\nu(\gamma)<\infty,
\]
令 \(y:=(x-\bar\gamma)/t\)，并定义二阶归一化剖面
\[
\mathcal R_t(y):=
t^2\left(\frac{h_t(\bar\gamma+ty)}{g_t(\bar\gamma+ty)}-1\right).
\]
则当 \(t\to\infty\) 时成立一致展开
\[
\mathcal R_t(y)=\frac{\mathrm{Var}}{2}\,\mathcal H(y)+O(t^{-1})
\qquad\text{一致于 }y\in\RR,
\]
其中
\[
\mathcal H(y)=\frac{2(3y^2-1)}{(1+y^2)^2}.
\]
\end{corollary}
\begin{proof}[证明要点]
相对熵与密度比在共同平移下不变，故可取 \(\bar\gamma=0\) 并令 \(\Delta:=\gamma-\bar\gamma\)。
由 \(g_t(ty)=\frac{1}{\pi t}\frac{1}{1+y^2}\) 与
\(
h_t(ty)=\frac{1}{\pi t}\EE[(1+(y-\Delta/t)^2)^{-1}]
\)，得到密度比恒等式
\[
\frac{h_t}{g_t}(ty)=\EE\!\left[F\!\left(y,\frac{\Delta}{t}\right)\right],
\qquad
F(y,\varepsilon):=\frac{1+y^2}{1+(y-\varepsilon)^2}.
\]
由直接代数化简得到恒等式
\[
F(y,\varepsilon)
=1+\frac{2y}{1+y^2}\varepsilon+\frac{3y^2-1}{(1+y^2)^2}\varepsilon^2
-\frac{4y(1-y^2)}{(1+y^2)^2(1+(y-\varepsilon)^2)}\,\varepsilon^3
-\frac{3y^2-1}{(1+y^2)^2(1+(y-\varepsilon)^2)}\,\varepsilon^4.
\]
由于 \((1+(y-\varepsilon)^2)^{-1}\le 1\) 且系数在 \(y\in\RR\) 上一致有界，
结合 \(\EE[\Delta]=0\) 与矩条件可得
\[
\frac{h_t}{g_t}(ty)-1
=\frac{\mathrm{Var}}{t^2}\frac{3y^2-1}{(1+y^2)^2}+O(t^{-3})
\qquad\text{一致于 }y\in\RR.
\]
两边乘以 \(t^2\)，并注意 \(\mathcal H(y)=2(3y^2-1)/(1+y^2)^2\)，即得结论。证毕。
\end{proof}

\paragraph{二阶剖面极限、全 \texorpdfstring{$L^p$}{Lp} 尺度与矩投影反演}

\begin{theorem}[Poisson--Cauchy 二阶校正项的全局形状极限与双结点刚性]\label{thm:xi-cauchy-poisson-second-order-shape-limit-node-rigidity}
\leanverified{paper\_xi\_cauchy\_poisson\_second\_order\_shape\_limit\_node\_rigidity}
在命题 \ref{prop:xi-poisson-tv-kl-second-order} 的设定下，设
\[
\sigma^2:=\int_{\RR}(\gamma-\bar\gamma)^2\,\dd\tilde\nu(\gamma)\in(0,\infty),
\]
并假设四阶中心绝对矩有限：
\[
\int_{\RR}|\gamma-\bar\gamma|^4\,\dd\tilde\nu(\gamma)<\infty,
\]
并沿用其二阶分解
\[
h_t(x)=g_t(x)+\sigma^2 r_t(x)+\varepsilon_t(x),
\]
其中
\[
r_t(x)=\frac1\pi\frac{t\bigl(3(x-\bar\gamma)^2-t^2\bigr)}{\bigl((x-\bar\gamma)^2+t^2\bigr)^3},
\qquad
\sup_{x\in\RR}|\varepsilon_t(x)|=O(t^{-4}).
\]
令 \(y:=(x-\bar\gamma)/t\)，并定义
\[
\Phi(y):=\frac1\pi\frac{3y^2-1}{(1+y^2)^3},
\qquad
U(y):=\frac{3y^2-1}{(1+y^2)^2}.
\]
则当 \(t\to\infty\) 时有一致极限
\[
\sup_{y\in\RR}\left|
\frac{t^3}{\sigma^2}\bigl(h_t(\bar\gamma+ty)-g_t(\bar\gamma+ty)\bigr)-\Phi(y)
\right|\longrightarrow 0,
\]
\[
\sup_{y\in\RR}\left|
\frac{t^2}{\sigma^2}\left(\frac{h_t}{g_t}-1\right)(\bar\gamma+ty)-U(y)
\right|\longrightarrow 0.
\]
进一步，存在 \(t_0>0\) 与两条连续结点轨道 \(x_\pm(t)\)（\(t\ge t_0\)）使
\[
h_t(x_\pm(t))=g_t(x_\pm(t)),\qquad
x_\pm(t)=\bar\gamma\pm \frac{t}{\sqrt3}+o(t),
\]
并且对所有 \(t\ge t_0\)：
\[
h_t-g_t<0\ \text{于 }(x_-(t),x_+(t)),
\qquad
h_t-g_t>0\ \text{于 }\RR\setminus[x_-(t),x_+(t)].
\]
\end{theorem}
\begin{proof}
由 \(x=\bar\gamma+ty\) 直接计算
\[
t^3r_t(\bar\gamma+ty)=\frac1\pi\frac{3y^2-1}{(1+y^2)^3}=\Phi(y).
\]
故
\[
\frac{t^3}{\sigma^2}\bigl(h_t-g_t\bigr)(\bar\gamma+ty)
=\Phi(y)+\frac{t^3}{\sigma^2}\varepsilon_t(\bar\gamma+ty),
\]
而 \(\sup_x|\varepsilon_t(x)|=O(t^{-4})\) 给出误差为 \(O(t^{-1})\) 一致趋零，得第一式。

由推论 \ref{cor:xi-cauchy-poisson-density-ratio-second-order-profile}，
\[
t^2\left(\frac{h_t}{g_t}-1\right)(\bar\gamma+ty)
=\frac{\sigma^2}{2}\,\mathcal H(y)+O(t^{-1})
\qquad\text{一致于 }y\in\RR,
\]
其中 \(\mathcal H(y)=2U(y)\)，故第二式成立。

记
\[
F_t(y):=\frac{t^3}{\sigma^2}\bigl(h_t-g_t\bigr)(\bar\gamma+ty).
\]
\(\Phi\) 在 \(y_\pm:=\pm 1/\sqrt3\) 为简单零点，且在 \((y_-,y_+)\) 为负、外侧为正。
由 \(F_t\to\Phi\) 的一致收敛与简单零点稳定性，存在 \(t_0\) 使 \(t\ge t_0\) 时，
\(F_t\) 在 \(y_\pm\) 邻域各有且仅有一根 \(y_\pm(t)\)，且
\(
y_\pm(t)\to y_\pm
\)；
并在其余区域保持与 \(\Phi\) 同号。
令 \(x_\pm(t):=\bar\gamma+t\,y_\pm(t)\)，即得
\[
x_\pm(t)=\bar\gamma\pm \frac{t}{\sqrt3}+o(t)
\]
及所述符号分布。证毕。
\end{proof}

\begin{theorem}[全 \texorpdfstring{$L^p$}{Lp} 尺度律的锐渐近与可积常数]\label{thm:xi-cauchy-poisson-all-lp-sharp-scaling}
\leanverified{paper\_xi\_cauchy\_poisson\_all\_lp\_sharp\_scaling}
在定理 \ref{thm:xi-cauchy-poisson-second-order-shape-limit-node-rigidity} 的条件下，对任意 \(p\in[1,\infty]\) 存在显式常数 \(C_p\in(0,\infty)\) 使
\[
\|h_t-g_t\|_{L^p(\RR)}
\sim
\sigma^2\,C_p\,t^{-(3-1/p)}
\qquad(t\to\infty),
\]
其中
\[
C_p=
\begin{cases}
\displaystyle
\frac1\pi\left(\int_{\RR}\frac{|3y^2-1|^p}{(1+y^2)^{3p}}\,\dd y\right)^{1/p},
&1\le p<\infty,\\[1.2em]
\displaystyle
\frac1\pi\sup_{y\in\RR}\frac{|3y^2-1|}{(1+y^2)^3}=\frac1\pi,
&p=\infty.
\end{cases}
\]
并且
\[
C_2=\frac{\sqrt3}{4\sqrt\pi}.
\]
\end{theorem}
\begin{proof}
由 \(h_t-g_t=\sigma^2r_t+\varepsilon_t\) 且
\(
r_t(\bar\gamma+ty)=\pi^{-1}t^{-3}(3y^2-1)(1+y^2)^{-3}
\)，
对 \(1\le p<\infty\) 有
\[
\|r_t\|_{L^p}^p
=\frac{t^{-(3p-1)}}{\pi^p}\int_{\RR}\frac{|3y^2-1|^p}{(1+y^2)^{3p}}\,\dd y,
\]
故 \(\|r_t\|_{L^p}=C_p t^{-(3-1/p)}\)。

余项估计由插值给出：
\[
\|\varepsilon_t\|_{L^p}
\le
\|\varepsilon_t\|_{L^\infty}^{1-1/p}\|\varepsilon_t\|_{L^1}^{1/p}
=O(t^{-4(1-1/p)})\,O(t^{-3/p})
=O(t^{-(4-1/p)})
=o\!\bigl(t^{-(3-1/p)}\bigr).
\]
故主项由 \(\sigma^2r_t\) 决定，得到 \(1\le p<\infty\) 的结论。

当 \(p=\infty\) 时，
\[
\|r_t\|_{L^\infty}
=\frac1{\pi t^3}\sup_{y\in\RR}\frac{|3y^2-1|}{(1+y^2)^3}
=\frac1{\pi t^3},
\]
且 \(\|\varepsilon_t\|_{L^\infty}=O(t^{-4})=o(t^{-3})\)，故结论同样成立。
最后 \(C_2\) 由
\(
\int_{\RR}(3y^2-1)^2(1+y^2)^{-6}\dd y=3\pi/16
\)
直接化简即得。证毕。
\end{proof}

\begin{corollary}[一般光滑 \texorpdfstring{$f$}{f}-散度的六阶展开（矩记号重写）]\label{cor:xi-cauchy-poisson-fdiv-sixth-order-moment-rewrite}
在定理 \ref{thm:xi-cauchy-poisson-second-order-shape-limit-node-rigidity} 的条件下，若再有五阶中心绝对矩有限，
记
\[
m_2:=\sigma^2,\qquad
m_3:=\int_{\RR}(\gamma-\bar\gamma)^3\,\dd\tilde\nu(\gamma),\qquad
m_4:=\int_{\RR}(\gamma-\bar\gamma)^4\,\dd\tilde\nu(\gamma).
\]
若 \(f:(0,\infty)\to\RR\) 凸、\(f(1)=0\)，且在 \(1\) 邻域三阶可微并满足 \(f''(1),f'''(1)<\infty\)，则
\[
D_f(h_t\mid g_t)
=\frac{f''(1)}{8}\frac{m_2^2}{t^4}
+\frac1{t^6}\left[
f''(1)\left(\frac{3}{32}m_3^2-\frac18 m_2m_4\right)
-\frac{f'''(1)}{64}m_2^3
\right]
+o(t^{-6}).
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:xi-cauchy-poisson-fdiv-sixth-order-expansion} 直接改写 \((\sigma^2,\mu_3,\mu_4)\mapsto(m_2,m_3,m_4)\) 得到；
其余项 \(O(t^{-8})\) 自动属于 \(o(t^{-6})\)。证毕。
\end{proof}

\begin{theorem}[单尺度视界剖面的中心矩投影反演公式]\label{thm:xi-cauchy-poisson-single-scale-moment-inversion}
在推论 \ref{cor:xi-cauchy-poisson-fdiv-sixth-order-moment-rewrite} 的条件下，定义
\[
\delta_t(x):=\frac{h_t(x)}{g_t(x)}-1,
\qquad
\langle \varphi,\psi\rangle_t:=\int_{\RR} g_t(x)\,\varphi(x)\psi(x)\,\dd x.
\]
令
\[
U(y):=\frac{3y^2-1}{(1+y^2)^2},\qquad
V(y):=\frac{4y(y^2-1)}{(1+y^2)^3},\qquad
W(y):=\frac{5y^4-10y^2+1}{(1+y^2)^4},
\]
\[
U_t(x):=U\!\left(\frac{x-\bar\gamma}{t}\right),\quad
V_t(x):=V\!\left(\frac{x-\bar\gamma}{t}\right),\quad
W_t(x):=W\!\left(\frac{x-\bar\gamma}{t}\right).
\]
则有极限反演
\[
m_2=\lim_{t\to\infty}4t^2\,\langle U_t,\delta_t\rangle_t,
\qquad
m_3=\lim_{t\to\infty}\frac{16}{3}t^3\,\langle V_t,\delta_t\rangle_t,
\]
\[
m_4=\lim_{t\to\infty}\frac{32}{5}\,t^4
\left(\langle W_t,\delta_t\rangle_t+\frac{m_2}{8t^2}\right).
\]
等价地，第三式可写为
\[
m_4=
\lim_{t\to\infty}\frac{32}{5}\,t^4\left(
\langle W_t,\delta_t\rangle_t+\frac{1}{8t^2}\,\lim_{s\to\infty}4s^2\langle U_s,\delta_s\rangle_s
\right).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的比值展开，
\[
\delta_t(\bar\gamma+ty)=\frac{m_2}{t^2}U(y)+\frac{m_3}{t^3}V(y)+\frac{m_4}{t^4}W(y)+O(t^{-5}),
\]
且
\(
g_t(\bar\gamma+ty)\,t\,\dd y=G(y)\dd y
\),
\(
G(y)=\pi^{-1}(1+y^2)^{-1}
\)。
故
\[
\langle U_t,\delta_t\rangle_t
=\frac{m_2}{t^2}\!\int_{\RR}GU^2
+\frac{m_4}{t^4}\!\int_{\RR}GUW
+O(t^{-5}),
\]
其中 \(\int GUV=0\)（奇偶性）。
又
\[
\int_{\RR}GU^2=\frac14,\qquad \int_{\RR}GUW=-\frac18,
\]
从而
\[
4t^2\langle U_t,\delta_t\rangle_t=m_2+O(t^{-2}),
\]
得到第一式。

同理
\[
\langle V_t,\delta_t\rangle_t
=\frac{m_3}{t^3}\!\int_{\RR}GV^2+O(t^{-4})
=\frac{3m_3}{16t^3}+O(t^{-4}),
\]
故第二式成立。

再者
\[
\langle W_t,\delta_t\rangle_t
=\frac{m_2}{t^2}\!\int_{\RR}GUW+\frac{m_4}{t^4}\!\int_{\RR}GW^2+O(t^{-5})
=-\frac{m_2}{8t^2}+\frac{5m_4}{32t^4}+O(t^{-5}),
\]
其中 \(\int GW^2=5/32\)。
整理即得第三式，并由第一式替换 \(m_2\) 得等价写法。证毕。
\end{proof}
