\paragraph{整数层的 \texorpdfstring{$2\times 2$}{2x2} 块化与模 \texorpdfstring{$\pi_\varphi^t$}{pi^t}/\texorpdfstring{$5^t$}{5^t} 计数闭式：从 \texorpdfstring{$\mathcal O$}{O}-Smith 形到可审计有限群指纹}
沿用本段记号：\(\mathcal O=\ZZ[\varphi]\)、\(\pi_\varphi=\varphi+2\)、以及
\(
C_q\in\Mat_{q+1}(\mathcal O)
\)
（结论 \ref{con:xi-symq-golden-cq-smith-staircase}）。
为把 \(\mathcal O\)-模层面的 Smith 结构进一步压到整数层与有限模计数上，固定 \(\mathcal O\) 的 \(\ZZ\)-基 \((1,\varphi)\) 并识别 \(\mathcal O\simeq \ZZ^2\)。

\begin{definition}[限制标量矩阵 \texorpdfstring{$\widetilde C_q$}{Ctilde}]\label{def:xi-symq-golden-cq-scalar-restriction}
将 \(\mathcal O^{q+1}\) 视为秩 \(2(q+1)\) 的自由 \(\ZZ\)-模：
\[
\mathcal O^{q+1}\ \cong\ \ZZ^{2(q+1)}
\qquad\text{（按基 }(1,\varphi)\text{ 逐坐标展开）}.
\]
令 \(\widetilde C_q\in\Mat_{2(q+1)}(\ZZ)\) 表示由 \(C_q\) 所诱导的 \(\ZZ\)-线性自同态在该基下的矩阵（即限制标量 \(\mathrm{Res}_{\mathcal O/\ZZ}\) 的整数表示）。
\end{definition}

\begin{definition}[乘以 \texorpdfstring{$\pi_\varphi$}{pi} 的整数矩阵 \texorpdfstring{$A$}{A}]\label{def:xi-symq-golden-pi-integer-matrix}
令 \(A\in\Mat_2(\ZZ)\) 表示乘法映射
\(
[\,\pi_\varphi\,]_{\mathcal O}:\mathcal O\to\mathcal O
\)
在基 \((1,\varphi)\) 下的矩阵。
则
\[
\boxed{\;
A=\begin{pmatrix}2&1\\[2pt]1&3\end{pmatrix}\;}
\qquad\text{且}\qquad
\det(A)=\mathrm N(\pi_\varphi)=5.
\]
\end{definition}
\begin{proof}
\(\pi_\varphi\cdot 1=2+\varphi\) 给出第一列 \((2,1)^{\top}\)，并且
\[
\pi_\varphi\cdot \varphi=(\varphi+2)\varphi
=\varphi^2+2\varphi=(\varphi+1)+2\varphi=1+3\varphi
\]
给出第二列 \((1,3)^{\top}\)。
行列式等于该 \(\ZZ\)-线性映射的指数（亦即代数范数），并由结论 \ref{con:xi-symq-golden-trace-det-norm} 的 \(\mathrm N(\pi_\varphi)=5\) 得到。证毕。
\end{proof}

\begin{conclusion}[整数层面的完全 \texorpdfstring{$(2\times2)$}{2x2} 块化：\texorpdfstring{$\widetilde C_q\sim \bigoplus_{r=0}^{q}A^r$}{Ctilde ~ ⊕ A^r}]\label{con:xi-symq-golden-cq-integer-2x2-blockization}
令
\[
\mathcal A_q:=\mathrm{diag}\bigl(A^{0},A^{1},\dots,A^{q}\bigr)\in\Mat_{2(q+1)}(\ZZ).
\]
则存在 \(U,V\in\mathrm{GL}_{2(q+1)}(\ZZ)\) 使得
\[
\boxed{\;
U\,\widetilde C_q\,V=\mathcal A_q\; }.
\]
\end{conclusion}
\begin{proof}
由结论 \ref{con:xi-symq-golden-cq-smith-staircase}，存在幺模矩阵 \(Q_q,U_q\in\mathrm{GL}_{q+1}(\mathcal O)\) 使
\[
C_q=Q_q\cdot \mathrm{diag}(1,\pi_\varphi,\pi_\varphi^2,\dots,\pi_\varphi^{q})\cdot U_q.
\]
在基 \((1,\varphi)\) 下限制标量，\(\mathcal O\)-幺模矩阵诱导 \(\ZZ\)-幺模矩阵：
其整数表示的行列式等于 \(\mathcal O\)-行列式的代数范数，而单位元的范数为 \(\pm 1\)，故属于 \(\mathrm{GL}_{2(q+1)}(\ZZ)\)。
另一方面，\(\pi_\varphi^{r}\) 在 \(\mathcal O\simeq \ZZ^2\) 上的整数表示为 \(A^r\)。
因此
\[
\widetilde C_q=\widetilde Q_q\cdot \mathrm{diag}(A^0,A^1,\dots,A^q)\cdot \widetilde U_q,
\]
其中 \(\widetilde Q_q,\widetilde U_q\in\mathrm{GL}_{2(q+1)}(\ZZ)\)。
取 \(U=\widetilde Q_q^{-1}\)、\(V=\widetilde U_q^{-1}\) 即得。证毕。
\end{proof}

\begin{proposition}[幂次 \texorpdfstring{$A^r$}{A^r} 的 Smith 形闭式]\label{prop:xi-symq-golden-Ar-smith-closed-form}
\leanverified{paper\_xi\_symq\_golden\_ar\_smith\_closed\_form}
对任意整数 \(r\ge 0\)，成立
\[
\boxed{\;
\mathrm{SNF}_{\ZZ}(A^{r})=\mathrm{diag}\bigl(5^{\lfloor r/2\rfloor},\,5^{\lceil r/2\rceil}\bigr)\; }.
\]
\end{proposition}
\begin{proof}
直接计算得
\[
A^2=
\begin{pmatrix}5&5\\[2pt]5&10\end{pmatrix}
=5M,
\qquad
M:=\begin{pmatrix}1&1\\[2pt]1&2\end{pmatrix}\in\mathrm{SL}_2(\ZZ).
\]
于是 \(A^{2k}=5^{k}M^{k}\)，且因 \(M^{k}\in\mathrm{SL}_2(\ZZ)\)，其条目不可能同被某个素数整除；
故 \(A^{2k}\) 的全体条目公因子恰为 \(5^{k}\)。
另一方面，\(\det(A^{2k})=5^{2k}\)，在 \(2\times 2\) 情形 Smith 不变因子满足
\(
d_1=\gcd(\text{全部 }1\times 1\text{ 子式})
\)、
\(
d_1d_2=\det
\)；
从而 \(\mathrm{SNF}_{\ZZ}(A^{2k})=\mathrm{diag}(5^{k},5^{k})\)。
\par
对奇次幂，\(A^{2k+1}=5^{k}M^{k}A\)。
同理，若 \(M^{k}A\) 的全部条目可被 \(5\) 整除，则 \(\det(M^{k}A)=\det(A)=5\) 将可被 \(25\) 整除，矛盾。
故 \(\gcd(\text{条目})=1\)，于是 \(\gcd(\text{条目 of }A^{2k+1})=5^{k}\)，并由 \(\det(A^{2k+1})=5^{2k+1}\) 得
\(\mathrm{SNF}_{\ZZ}(A^{2k+1})=\mathrm{diag}(5^{k},5^{k+1})\)。
合并即得所示闭式。证毕。
\end{proof}

因此作为加法群有同构

\mathcal O/(\pi_\varphi^{e})\ \cong\ \ZZ^2/A^{e}\ZZ^2.

由命题 \ref{prop:xi-symq-golden-Ar-smith-closed-form}，

\ZZ^2/A^{e}\ZZ^2\cong \ZZ/5^{\lfloor e/2\rfloor}\ZZ\oplus \ZZ/5^{\lceil e/2\rceil}\ZZ

证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\widetilde C_q$}{Ctilde} 的 Smith 指数重数谱]\label{cor:xi-symq-golden-cq-integer-smith-multiplicity-spectrum}

\begin{conclusion}[模 \texorpdfstring{$\pi_\varphi^{t}$}{pi^t} 的核—像计数与纤维二值律]\label{con:xi-symq-golden-cq-mod-pi-kernel-image-count}
令 \(t\ge 1\) 并设
\[
V_t:=(\mathcal O/\pi_\varphi^{t})^{q+1},
\qquad
C_{q,t}:V_t\to V_t
\]
为 \(C_q\) 在 \(V_t\) 上诱导的同态。
定义
\[
\Sigma(q,t):=\sum_{r=0}^{q}\min(r,t)
=
\begin{cases}
\dfrac{t(2q-t+1)}{2},& 1\le t\le q,\\[6pt]
\dfrac{q(q+1)}{2},& t\ge q.
\end{cases}
\]
则有精确计数
\[
\boxed{\;
\#\ker(C_{q,t})=5^{\Sigma(q,t)},\qquad
\#\mathrm{Im}(C_{q,t})=5^{t(q+1)-\Sigma(q,t)}\; }.
\]
并且对任意 \(b\in V_t\)，同余方程 \(C_{q,t}x=b\) 的解数满足二值律
\[
\boxed{\;
\#\{x\in V_t:\ C_{q,t}x=b\}=
\begin{cases}
5^{\Sigma(q,t)},& b\in\mathrm{Im}(C_{q,t}),\\
0,& b\notin\mathrm{Im}(C_{q,t}).
\end{cases}}
\]
\end{conclusion}
\begin{proof}
由结论 \ref{con:xi-symq-golden-cq-smith-staircase} 的 Smith 形
\(
\mathrm{SNF}_{\mathcal O}(C_q)=\mathrm{diag}(1,\pi_\varphi,\dots,\pi_\varphi^q)
\)，
在模 \(\pi_\varphi^t\) 下 \(C_{q,t}\) 分解为坐标乘法 \(x\mapsto \pi_\varphi^{r}x\) 的直和。
单坐标核大小为 \(5^{\min(r,t)}\)，乘法性给出 \(\#\ker(C_{q,t})=5^{\Sigma(q,t)}\)。
又 \(\#V_t=5^{t(q+1)}\)，故 \(\#\mathrm{Im}=\#V_t/\#\ker\)。
最后一条由有限群同态的纤维大小恒等式给出：像内每条纤维大小恒为 \(\#\ker\)，像外纤维为空。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\pi_\varphi$}{pi}-进核规模的严格差分律]\label{cor:xi-symq-golden-cq-pi-adic-kernel-growth}

\Sigma(q,t+1)-\Sigma(q,t)
=\#\{r\in\{0,1,\dots,q\}:\ r\ge t+1\}

代回结论 \ref{con:xi-symq-golden-cq-mod-pi-kernel-image-count} 即得。证毕。
\end{proof}

\begin{conclusion}[模 \texorpdfstring{$5^{t}$}{5^t} 的整数核大小：分段二次闭式]\label{con:xi-symq-golden-cq-mod-5-kernel-size}

W_t:=(\ZZ/5^{t}\ZZ)^{2(q+1)},

\widetilde C_{q,t}:W_t\to W_t

\begin{enumerate}
  \item 若 \(q=2m\)，则
  \[
  \#\ker(\widetilde C_{q,t})
  =
  \begin{cases}
  5^{\,t(4m-2t+1)},& 1\le t\le m,\\[4pt]
  5^{\,2m^{2}+m},& t\ge m;
  \end{cases}
  \]
  \item 若 \(q=2m+1\)，则
  \[
  \#\ker(\widetilde C_{q,t})
  =
  \begin{cases}
  5^{\,t(4m-2t+3)},& 1\le t\le m+1,\\[4pt]

故其在 \(W_t\) 上的核大小相同。
而 \(\mathcal A_q\) 的核大小是各块核大小之乘积。
由命题 \ref{prop:xi-symq-golden-Ar-smith-closed-form}，
\(
\mathrm{SNF}_{\ZZ}(A^r)=\mathrm{diag}(5^{\lfloor r/2\rfloor},5^{\lceil r/2\rceil})

因此模 \(5^{t}\) 的单块核指数为

对 \(r=0,1,\dots,q\) 求和并按 \(q\) 奇偶分段化简，即得所示二次闭式。证毕。
\end{proof}

\begin{theorem}[模 \texorpdfstring{$5^t$}{5^t} 核增长的二阶差分恢复整数 Smith 指数]\label{thm:xi-symq-golden-cq-mod5-second-difference-smith}
沿用定义 \ref{def:xi-symq-golden-cq-scalar-restriction}、推论 \ref{cor:xi-symq-golden-cq-integer-smith-multiplicity-spectrum} 与结论 \ref{con:xi-symq-golden-cq-mod-5-kernel-size} 的记号。
对 \(t\ge 0\) 定义
\[
\kappa_q(t):=\log_5\#\ker(\widetilde C_{q,t}),
\qquad
\kappa_q(0):=0,
\]
其中 \(\widetilde C_{q,t}:W_t\to W_t\) 为 \(\widetilde C_q\) 在 \(W_t=(\ZZ/5^t\ZZ)^{2(q+1)}\) 上诱导的同态。
再对 \(t\ge 1\) 定义离散二阶差分
\[
m_q(t):=2\kappa_q(t)-\kappa_q(t-1)-\kappa_q(t+1).
\]
若
\[
\mathrm{SNF}_{\ZZ}(\widetilde C_q)
=
\operatorname{diag}(5^{e_1},5^{e_2},\dots,5^{e_{2(q+1)}}),
\qquad
e_1\le e_2\le \cdots\le e_{2(q+1)},
\]
则对每个 \(t\ge 1\) 都有
\[
\boxed{\;
m_q(t)=\#\{j:\ e_j=t\}.\;}
\]
特别地，写 \(q=2m\) 或 \(q=2m+1\)（\(m\ge 1\)），则：
\begin{enumerate}
  \item 若 \(q=2m\)，则
  \[
  m_q(t)=
  \begin{cases}
  4,&1\le t\le m-1,\\
  3,&t=m,\\
  0,&t\ge m+1;
  \end{cases}
  \]
  \item 若 \(q=2m+1\)，则
  \[
  m_q(t)=
  \begin{cases}
  4,&1\le t\le m,\\
  1,&t=m+1,\\
  0,&t\ge m+2.
  \end{cases}
  \]
\end{enumerate}
因而整数层的 Smith 正规形完全由核增长序列
\(
\{\#\ker(\widetilde C_{q,t})\}_{t\ge 1}
\)
唯一恢复。
\end{theorem}
\begin{proof}
这也是命题 \ref{prop:xi-hankel-normalform-hp-discrete-concave-invert} 在 \((H_0,p)=(\widetilde C_q,5)\) 下的具体化；为保持本段局部记号闭合，仍给出直接计算。

设
\[
\mathrm{SNF}_{\ZZ}(\widetilde C_q)
=
\operatorname{diag}(5^{e_1},\dots,5^{e_{2(q+1)}}).
\]
则由 Smith 正规形的标准计数律，
\[
\#\ker(\widetilde C_{q,t})
=
\prod_{j=1}^{2(q+1)}5^{\min(e_j,t)},
\]
从而
\[
\kappa_q(t)=\sum_{j=1}^{2(q+1)}\min(e_j,t).
\]
因此
\[
\begin{aligned}
m_q(t)
&=
\sum_{j=1}^{2(q+1)}\Bigl(2\min(e_j,t)-\min(e_j,t-1)-\min(e_j,t+1)\Bigr).
\end{aligned}
\]
对固定 \(e:=e_j\) 考察括号内的值：
\[
2\min(e,t)-\min(e,t-1)-\min(e,t+1)
=
\begin{cases}
0,& e<t,\\
1,& e=t,\\
0,& e>t.
\end{cases}
\]
故每个满足 \(e_j=t\) 的对角指数恰贡献 \(1\)，其余贡献均为 \(0\)，于是
\[
m_q(t)=\#\{j:\ e_j=t\}.
\]
再将推论 \ref{cor:xi-symq-golden-cq-integer-smith-multiplicity-spectrum} 的显式重数谱直接代入，便得到所列分段公式。证毕。
\end{proof}

\begin{corollary}[模 \texorpdfstring{$5$}{5} 单层核大小与余核群型的刚性]\label{cor:xi-symq-golden-cq-mod5-single-layer-rigidity}
沿用定理 \ref{thm:xi-symq-golden-cq-mod5-second-difference-smith} 的记号，并记
\[
G_q:=\mathrm{coker}_{\ZZ}(\widetilde C_q).
\]
则有：
\begin{enumerate}
  \item
  \[
  \#\ker(\widetilde C_{q,1})=5^{2q-1}.
  \]
  因而仅由模 \(5\) 的单层核大小即可恢复
  \[
  q=\frac{1}{2}\Bigl(\log_5\#\ker(\widetilde C_{q,1})+1\Bigr).
  \]
  \item 整条序列
  \(
  \{\#\ker(\widetilde C_{q,t})\}_{t\ge 1}
  \)
  唯一决定有限阿贝尔 \(5\)-群 \(G_q\) 的同构类型。
  \item 写 \(q=2m\) 或 \(q=2m+1\)（\(m\ge 1\)），则
  \[
  G_{2m}\cong (\ZZ/5^m\ZZ)^3\oplus \bigoplus_{t=1}^{m-1}(\ZZ/5^t\ZZ)^4,
  \]
  \[
  G_{2m+1}\cong \ZZ/5^{m+1}\ZZ\oplus \bigoplus_{t=1}^{m}(\ZZ/5^t\ZZ)^4.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}

\[
\kappa_q(1)=2q-1,
\]
故
\[
\#\ker(\widetilde C_{q,1})=5^{\kappa_q(1)}=5^{2q-1},

第 (1) 条成立。

第 (2) 条由定理 \ref{thm:xi-symq-golden-cq-mod5-second-difference-smith} 立即得到：核增长序列唯一恢复每个正整数 \(t\) 作为 Smith 指数的重数，而有限阿贝尔 \(5\)-群的同构类型正由这些初等因子的多重集唯一决定。

对第 (3) 条，只需把定理 \ref{thm:xi-symq-golden-cq-mod5-second-difference-smith} 中恢复出的 Smith 指数重数直接改写为余核的直和分解即可。证毕。
\end{proof}
