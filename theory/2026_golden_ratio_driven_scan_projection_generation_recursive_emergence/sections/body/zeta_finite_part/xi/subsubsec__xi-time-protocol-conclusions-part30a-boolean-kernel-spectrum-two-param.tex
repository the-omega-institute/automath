\paragraph{两参数不交矩阵的全谱分解与秩一例外谱}
在幂集标号 \(V_q=2^{[q]}\) 上，给定参数 \(a,b\in\RR\)，定义对称矩阵
\begin{equation}\label{eq:xi-two-param-disjointness-definition}
\bigl(B^{(q)}(a,b)\bigr)_{U,V}:=
\begin{cases}
a,&U\cap V=\varnothing,\\
b,&U\cap V\neq\varnothing,
\end{cases}
\qquad(U,V\in V_q).
\end{equation}
记 \(d:=a-b\)。注意到 \(A_q=B^{(q)}(2,1)\)，而 \(2T^{(q)}=A_q\)。

\begin{theorem}[两参数矩阵 \(B^{(q)}(a,b)\) 的特征多项式分解与世俗方程]\label{thm:xi-two-param-disjointness-full-spectrum-secular}
\leanverified{paper\_xi\_two\_param\_disjointness\_full\_spectrum\_secular}
设 \(q\ge 2\)，并令
\[
K:=\begin{pmatrix}1&1\\[2pt]1&0\end{pmatrix},\qquad
J_2:=\begin{pmatrix}1&1\\[2pt]1&1\end{pmatrix},\qquad
D_q:=K^{\otimes q},\qquad
J_q:=J_2^{\otimes q}=\mathbf 1\mathbf 1^{\mathsf T}\in\Mat_{2^q}(\RR).
\]
则有
\begin{equation}\label{eq:xi-two-param-disjointness-kronecker-form}
B^{(q)}(a,b)=d\,D_q+b\,J_q.
\end{equation}
设
\[
\varphi:=\frac{1+\sqrt5}{2},\qquad \psi:=\frac{1-\sqrt5}{2},\qquad
\mu_i^{(q)}:=\varphi^{\,q-i}\psi^{\,i}\qquad(0\le i\le q),
\]
并置
\[
\alpha:=1+\frac{2}{\sqrt5},\qquad \beta:=1-\frac{2}{\sqrt5},\qquad
w_i:=\binom{q}{i}\alpha^{q-i}\beta^{i}\ (>0),\qquad \sum_{i=0}^q w_i=2^q.
\]
则 \(B^{(q)}(a,b)\) 的特征多项式在 \(\RR[\lambda]\) 中分解为
\begin{equation}\label{eq:xi-two-param-disjointness-charpoly-factorization}
\chi_{B^{(q)}(a,b)}(\lambda)=
P_{q,a,b}(\lambda)\cdot \prod_{i=0}^{q}\bigl(\lambda-d\mu_i^{(q)}\bigr)^{\binom{q}{i}-1},
\end{equation}
其中 \(P_{q,a,b}(\lambda)\) 为次数 \(q+1\) 的首一多项式，其根（按代数重数）恰为秩一扰动产生的例外谱 \(\{\lambda_0,\dots,\lambda_q\}\)。
当 \(b\neq 0\) 时，例外谱满足世俗方程
\begin{equation}\label{eq:xi-two-param-disjointness-secular}
1-b\sum_{i=0}^q \frac{w_i}{\lambda-d\mu_i^{(q)}}=0
\qquad\Longleftrightarrow\qquad
\sum_{i=0}^q \frac{w_i}{\lambda-d\mu_i^{(q)}}=\frac1b.
\end{equation}
更进一步，将实数点集 \(\{d\mu_i^{(q)}\}_{i=0}^q\) 递增重排为 \(\delta_0<\cdots<\delta_q\)。
若 \(b\neq 0\)，则 \(P_{q,a,b}\) 在每个连通分支
\[
(-\infty,\delta_0),\ (\delta_0,\delta_1),\ \dots,\ (\delta_{q-1},\delta_q),\ (\delta_q,\infty)
\]
中恰有一个简单实根；该根落在端区间的方向由 \(\mathrm{sign}(b)\) 决定。
\end{theorem}
\begin{proof}
矩阵 \(K\) 对称且特征值为 \(\varphi,\psi\)；因此 \(D_q=K^{\otimes q}\) 可正交对角化，其特征值为 \(\mu_i^{(q)}=\varphi^{q-i}\psi^{i}\)，重数为 \(\binom{q}{i}\)。
另一方面 \(J_q=\mathbf 1\mathbf 1^{\mathsf T}\) 为秩一算子：在 \(\mathbf 1^\perp\) 上为零，在 \(\Span\{\mathbf 1\}\) 上为标量 \(2^q\)。
故对每个 \(i\)，在 \(D_q\) 的 \(\mu_i^{(q)}\)-特征空间 \(E_i\) 内，子空间 \(E_i\cap \mathbf 1^\perp\)（维数 \(\binom{q}{i}-1\)）上
\(
B^{(q)}(a,b)=d\,D_q+b\,J_q
\)
退化为 \(d\mu_i^{(q)}\) 倍恒等算子，从而得到 \eqref{eq:xi-two-param-disjointness-charpoly-factorization} 中的主体谱因子。
\par
剩余的 \((q+1)\)-维不变子空间由各 \(E_i\) 中 \(\mathbf 1\) 的正交投影所张成。取其正交归一基 \(\{s_i\}_{i=0}^q\)，并写
\(
\mathbf 1=\sum_{i=0}^q \sqrt{w_i}\,s_i
\)。
则在该基下，
\[
\left.B^{(q)}(a,b)\right|_{\Span\{s_i\}}
\cong
d\,\mathrm{diag}\bigl(\mu_0^{(q)},\dots,\mu_q^{(q)}\bigr)
b\,gg^{\mathsf T},
\qquad
g:=(\sqrt{w_0},\dots,\sqrt{w_q})^{\mathsf T}.
\]
由秩一扰动的行列式引理得到世俗方程 \eqref{eq:xi-two-param-disjointness-secular}，并定义次数 \(q+1\) 的首一多项式 \(P_{q,a,b}\)。
\par
关于根的分支唯一性：由于 \(w_i>0\)，函数
\(
\lambda\mapsto \sum_{i=0}^q w_i/(\lambda-\delta_i)
\)
在每个分支上严格单调，且在端点处发散到 \(\pm\infty\)，故每个分支存在唯一简单实根；端区间的落向由右端常数 \(1/b\) 的符号决定。证毕。
\end{proof}

\begin{theorem}[例外多项式的拉格朗日型闭式与节点差积的斐波那契乘积]\label{thm:xi-two-param-disjointness-exceptional-poly-nodes}
\leanverified{paper\_xi\_two\_param\_disjointness\_exceptional\_poly\_nodes}
沿用定理 \ref{thm:xi-two-param-disjointness-full-spectrum-secular} 的记号，令
\[
G_{q,d}(\lambda):=\prod_{i=0}^q\bigl(\lambda-d\mu_i^{(q)}\bigr).
\]
则例外多项式 \(P_{q,a,b}\) 具有显式表示
\begin{equation}\label{eq:xi-two-param-disjointness-exceptional-poly-lagrange}
P_{q,a,b}(\lambda)=G_{q,d}(\lambda)-b\sum_{i=0}^q w_i\prod_{j\ne i}\bigl(\lambda-d\mu_j^{(q)}\bigr).
\end{equation}
特别地，对每个 \(0\le i\le q\)，成立节点取值公式
\begin{equation}\label{eq:xi-two-param-disjointness-exceptional-poly-nodes}
P_{q,a,b}\bigl(d\mu_i^{(q)}\bigr)
=-b\,w_i\prod_{j\ne i}\bigl(d\mu_i^{(q)}-d\mu_j^{(q)}\bigr)
=-b\,d^{q}\,w_i\prod_{j\ne i}\bigl(\mu_i^{(q)}-\mu_j^{(q)}\bigr).
\end{equation}
此外，令 \((F_k)_{k\ge 1}\) 为斐波那契数列（\(F_1=F_2=1\)），则对每个 \(0\le i\le q\) 有
\begin{equation}\label{eq:xi-two-param-disjointness-mu-vandermonde-fibonacci}
\prod_{j\ne i}\bigl(\mu_i^{(q)}-\mu_j^{(q)}\bigr)
=(-1)^{\,i(q+1)-\binom{i+1}{2}}\,
(\sqrt5)^{\,q}\,
\Bigl(\prod_{k=1}^{i}F_k\Bigr)\Bigl(\prod_{k=1}^{q-i}F_k\Bigr)\,
\varphi^{\frac{q(q-1)}{2}+i(1-q)}.
\end{equation}
\end{theorem}
\begin{proof}
由秩一扰动行列式引理，
\[
P_{q,a,b}(\lambda)
=\det\!\Bigl(\lambda I-d\,\mathrm{diag}(\mu_0^{(q)},\dots,\mu_q^{(q)})-b\,gg^{\mathsf T}\Bigr)
=\prod_{i=0}^q(\lambda-d\mu_i^{(q)})\Bigl(1-b\sum_{i=0}^q\frac{w_i}{\lambda-d\mu_i^{(q)}}\Bigr),
\]
整理即得 \eqref{eq:xi-two-param-disjointness-exceptional-poly-lagrange}；代入 \(\lambda=d\mu_i^{(q)}\) 并取极限（或以余子式展开）得到 \eqref{eq:xi-two-param-disjointness-exceptional-poly-nodes}。
\par
最后对差积 \(\prod_{j\ne i}(\mu_i^{(q)}-\mu_j^{(q)})\) 作几何参数化。写
\(
\mu_i^{(q)}=\varphi^q r^i
\)
其中
\(
r:=\psi/\varphi=-\varphi^{-2}.
\)
则
\[
\prod_{j\ne i}\bigl(\mu_i^{(q)}-\mu_j^{(q)}\bigr)
=\varphi^{q^2}\prod_{j\ne i}(r^i-r^j).
\]
对几何点列 \(\{r^0,\dots,r^q\}\) 的离散差积按 \(j<i\) 与 \(j>i\) 分解，可将 \(\prod_{j\ne i}(r^i-r^j)\) 写成符号因子与
\(
\prod_{k=1}^{i}(1-r^k)\prod_{k=1}^{q-i}(1-r^k)
\)
的乘积，并伴随一项 \(r\)-幂。
再利用恒等式
\[
1-r^k
=1-(-\varphi^{-2})^k
=\frac{\varphi^{2k}-(-1)^k}{\varphi^{2k}}
=\frac{\varphi^k-\psi^k}{\varphi^k}
=\frac{\sqrt5\,F_k}{\varphi^k},
\]
即可化简到 \eqref{eq:xi-two-param-disjointness-mu-vandermonde-fibonacci} 所示闭式。证毕。
\end{proof}

\begin{theorem}[例外多项式与节点多项式的结果式：纯整数斐波那契积公式]\label{thm:xi-two-param-disjointness-resultant-integer-fibonacci}
\leanverified{paper\_xi\_two\_param\_disjointness\_resultant\_integer\_fibonacci}
设 \(q\ge 2\) 且 \(b\neq 0\)。沿用 \(G_{q,d}\) 的记号，则
\begin{equation}\label{eq:xi-two-param-disjointness-resultant}
\Res\bigl(P_{q,a,b},G_{q,d}\bigr)
=\prod_{i=0}^q P_{q,a,b}\bigl(d\mu_i^{(q)}\bigr)
=(-1)^{\binom{q+2}{2}}\,
b^{q+1}\,d^{q(q+1)}\,
\Bigl(\prod_{i=0}^q\binom{q}{i}\Bigr)\,
\Bigl(\prod_{k=1}^q F_k^{\,2(q+1-k)}\Bigr).
\end{equation}
尤其当 \(b,d\neq 0\) 时，\(\Res(P_{q,a,b},G_{q,d})\) 为非零整数，从而 \(P_{q,a,b}\) 与 \(G_{q,d}\) 在 \(\ZZ[\lambda]\) 中互素。
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-two-param-disjointness-exceptional-poly-nodes}，
\[
\prod_{i=0}^q P_{q,a,b}\bigl(d\mu_i^{(q)}\bigr)
=(-b)^{q+1}d^{q(q+1)}\Bigl(\prod_{i=0}^q w_i\Bigr)\cdot
\prod_{i=0}^q\prod_{j\ne i}\bigl(\mu_i^{(q)}-\mu_j^{(q)}\bigr).
\]
后者等于
\(
(-1)^{\binom{q+1}{2}}\prod_{0\le i<j\le q}(\mu_i^{(q)}-\mu_j^{(q)})^2
\)。
另一方面，
\[
\prod_{i=0}^q w_i
=\Bigl(\prod_{i=0}^q\binom{q}{i}\Bigr)\cdot(\alpha\beta)^{\frac{q(q+1)}{2}}
=\Bigl(\prod_{i=0}^q\binom{q}{i}\Bigr)\cdot 5^{-\frac{q(q+1)}{2}},
\]
而对范德蒙德平方作同样分解并用 \(1-r^k=\sqrt5 F_k/\varphi^k\) 化简可得
\[
\prod_{0\le i<j\le q}(\mu_i^{(q)}-\mu_j^{(q)})^2
=5^{\frac{q(q+1)}{2}}\prod_{k=1}^q F_k^{\,2(q+1-k)}.
\]
代回并合并符号指数：
\(
(-b)^{q+1}(-1)^{\binom{q+1}{2}}=(-1)^{\binom{q+2}{2}}b^{q+1}
\)，即得 \eqref{eq:xi-two-param-disjointness-resultant}。互素性由结果式非零直接推出。证毕。
\end{proof}

\paragraph{按集合大小轨道的商矩阵：闭式行列式与迹}
令 \(\mathcal O_k:=\{U\subseteq[q]:\card{U}=k\}\) 为 \(V_q\) 的轨道分层。
记 \(M^{(q)}(a,b)\in\Mat_{q+1}(\RR)\) 为 \(B^{(q)}(a,b)\) 在按集合大小分层的商矩阵，即
\begin{equation}\label{eq:xi-two-param-disjointness-quotient-matrix}
M^{(q)}(a,b)_{k,\ell}=b\binom{q}{\ell}+d\binom{q-k}{\ell}\qquad(0\le k,\ell\le q).
\end{equation}

\begin{theorem}[商矩阵 \(M^{(q)}(a,b)\) 的行列式与迹]\label{thm:xi-two-param-disjointness-quotient-det-trace}
令 \(d=a-b\)。则
\begin{equation}\label{eq:xi-two-param-disjointness-quotient-det}
\det M^{(q)}(a,b)=a\,d^{q}\,(-1)^{\frac{q(q+1)}{2}},
\end{equation}
并且
\begin{equation}\label{eq:xi-two-param-disjointness-quotient-trace}
\tr M^{(q)}(a,b)=b\,2^q+d\,F_{q+1}.
\end{equation}
\end{theorem}
\begin{proof}
写 \(M=d\mathsf C+b\mathsf B\)，其中
\(
\mathsf C_{k,\ell}=\binom{q-k}{\ell}
\)
且
\(
\mathsf B=\mathbf 1\,\mathbf b^{\mathsf T}
\)
（\(\mathbf 1=(1,\dots,1)^{\mathsf T}\)，\(\mathbf b_\ell=\binom{q}{\ell}\)）。
对 \(\mathsf C\) 将列反序置换得上三角且对角元全为 \(1\)，故
\(
\det\mathsf C=(-1)^{q(q+1)/2}.
\)
\par
又二项反演给出 \(\mathsf C^{-1}\) 的显式公式，从而
\[
(\mathsf C^{-1}\mathbf 1)_k=\sum_{\ell=0}^{q-k}(-1)^{q-k+\ell}\binom{q-k}{\ell}=
\begin{cases}
0,&k<q,\\
1,&k=q,
\end{cases}
\]
即 \(\mathsf C^{-1}\mathbf 1=e_q\)。于是 \(\mathbf b^{\mathsf T}\mathsf C^{-1}\mathbf 1=\mathbf b^{\mathsf T}e_q=\binom{q}{q}=1\)。
由秩一行列式引理，
\[
\det(d\mathsf C+b\mathsf B)
=d^{q+1}\det(\mathsf C)\cdot\det\bigl(I+(b/d)\mathsf C^{-1}\mathsf B\bigr)
=d^{q+1}\det(\mathsf C)\cdot\bigl(1+(b/d)\mathbf b^{\mathsf T}\mathsf C^{-1}\mathbf 1\bigr)
=d^q\det(\mathsf C)\cdot(d+b),
\]
而 \(d+b=a\)，得 \eqref{eq:xi-two-param-disjointness-quotient-det}。
\par
迹公式由
\(
\tr(b\mathsf B)=b\sum_{\ell=0}^q\binom{q}{\ell}=b\,2^q
\)
及
\[
\tr(d\mathsf C)=d\sum_{k=0}^q\binom{q-k}{k}=d\,F_{q+1}
\]
给出；后一恒等式满足斐波那契递推且初值一致，故成立。证毕。
\end{proof}

\paragraph{幂和的项链展开与参数分离}
\begin{theorem}[例外谱幂和的项链展开：\(q\)-依赖的完全分离]\label{thm:xi-two-param-disjointness-exceptional-power-sums-necklace}
设 \(m\ge 1\)。令 \(\{\lambda_0,\dots,\lambda_q\}\) 为定理 \ref{thm:xi-two-param-disjointness-full-spectrum-secular} 中例外谱，即 \(P_{q,a,b}\) 的根（按代数重数计）。
定义
\[
S^{\mathrm{exc}}_{m}(q;a,b):=\sum_{j=0}^q \lambda_j^m=\tr\bigl(M^{(q)}(a,b)^m\bigr).
\]
令 \(\mathcal N_m\) 为长度 \(m\) 的二进制项链集合，即 \(\{0,1\}^m\) 在循环移位作用下的轨道集合。
对 \([w]\in\mathcal N_m\) 取任一代表 \(w\)，记 \(|[w]|\) 为轨道大小，\(t(w)\) 为 \(1\) 的个数。
若 \(w\) 含至少一个 \(1\)，沿循环从某个 \(1\) 出发，\(1\) 与下一 \(1\) 之间的 \(0\)-块长度构成循环序列 \((\ell_1,\dots,\ell_{t(w)})\)，并置
\[
\Phi(w):=\prod_{r=1}^{t(w)} F_{\ell_r+3}.
\]
则成立恒等式
\begin{equation}\label{eq:xi-two-param-disjointness-exceptional-power-sums-necklace}
S^{\mathrm{exc}}_{m}(q;a,b)=
d^m\sum_{i=0}^q \bigl(\mu_i^{(q)}\bigr)^m
+\sum_{\substack{[w]\in\mathcal N_m\\ [w]\ne [0^m]}}
|[w]|\, b^{t(w)}d^{m-t(w)}\,\Phi(w)^{\,q}.
\end{equation}
其中第一项对应纯 \(\mathsf C\)-字，而所有含 \(\mathsf B\) 的循环字贡献均以 \(\Phi(w)^q\) 的形式出现。
\end{theorem}
\begin{proof}
对商矩阵写 \(M=d\mathsf C+b\mathsf B\)。
展开 \((d\mathsf C+b\mathsf B)^m\) 得到长度 \(m\) 的 \(\{\mathsf C,\mathsf B\}\)-字之和；迹在循环置换下不变，故可按项链轨道 \([w]\) 分组，每一轨道贡献系数为 \(|[w]|\)。
\par
对含至少一个 \(\mathsf B\) 的字 \(w\)，在循环意义下可写为
\[
\mathsf C^{\ell_1}\mathsf B\mathsf C^{\ell_2}\mathsf B\cdots \mathsf C^{\ell_t}\mathsf B,
\qquad t=t(w).
\]
由于 \(\mathsf B=\mathbf 1\mathbf b^{\mathsf T}\) 为秩一，可将每段 \(\mathsf B\mathsf C^{\ell}\mathsf B\) 的贡献归结为标量 \(\mathbf b^{\mathsf T}\mathsf C^{\ell}\mathbf 1\)。
注意到 \(\mathsf C\) 是 \(D_q\) 在按集合大小分层之不变子空间上的商表示，因而对任意 \(\ell\ge 0\) 有
\[
\mathbf b^{\mathsf T}\mathsf C^{\ell}\mathbf 1
=\mathbf 1^{\mathsf T}D_q^{\ell}\mathbf 1
=\bigl(\mathbf 1_2^{\mathsf T}K^{\ell}\mathbf 1_2\bigr)^q,
\qquad
\mathbf 1_2:=(1,1)^{\mathsf T}.
\]
而 \(\mathbf 1_2^{\mathsf T}K^{\ell}\mathbf 1_2=F_{\ell+3}\)（由 \(K\) 的 Fibonacci 幂公式直接计算），故 \(\mathbf b^{\mathsf T}\mathsf C^{\ell}\mathbf 1=(F_{\ell+3})^q\)。
从而
\[
\tr\bigl(\mathsf C^{\ell_1}\mathsf B\mathsf C^{\ell_2}\mathsf B\cdots \mathsf C^{\ell_t}\mathsf B\bigr)
=\prod_{r=1}^{t}\bigl(\mathbf b^{\mathsf T}\mathsf C^{\ell_r}\mathbf 1\bigr)
=\prod_{r=1}^{t}F_{\ell_r+3}^{\,q}
=\Phi(w)^q.
\]
将系数 \(b^t d^{m-t}\) 与轨道大小并入即得第二项。
纯字对应 \(d^m\tr(\mathsf C^m)=d^m\sum_i(\mu_i^{(q)})^m\)。
综上得 \eqref{eq:xi-two-param-disjointness-exceptional-power-sums-necklace}。证毕。
\end{proof}

\leanverified{paper\_xi\_two\_param\_disjointness\_full\_trace\_necklace}
\begin{theorem}[全空间迹矩的项链展开：卢卡斯主项与斐波那契项链项]\label{thm:xi-two-param-disjointness-full-trace-necklace}
设 \(m\ge 1\)，并令 \(L_m:=\varphi^m+\psi^m\) 为第 \(m\) 个卢卡斯数。
则全矩阵满足
\begin{equation}\label{eq:xi-two-param-disjointness-full-trace-necklace}
\tr\bigl(B^{(q)}(a,b)^m\bigr)
=d^m\,L_m^{\,q}
+\sum_{\substack{[w]\in\mathcal N_m\\ [w]\ne [0^m]}}
|[w]|\, b^{t(w)}d^{m-t(w)}\,\Phi(w)^{\,q},
\end{equation}
其中 \(\mathcal N_m,t(w),\Phi(w)\) 同定理 \ref{thm:xi-two-param-disjointness-exceptional-power-sums-necklace}。
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-two-param-disjointness-kronecker-form}，在全空间上 \(B^{(q)}(a,b)=dD_q+bJ_q\)，其中 \(D_q=K^{\otimes q}\)，\(J_q=\mathbf 1\mathbf 1^{\mathsf T}\) 为秩一。
对 \(\{D_q,J_q\}\)-字同理作项链分组。
纯字给出
\[
\tr\bigl((dD_q)^m\bigr)=d^m\tr\bigl((K^m)^{\otimes q}\bigr)
=d^m\bigl(\tr(K^m)\bigr)^q
=d^mL_m^q.
\]
含 \(J_q\) 的字同样因秩一而塌缩为 \(\Phi(w)^q\) 结构，从而得到 \eqref{eq:xi-two-param-disjointness-full-trace-necklace}。证毕。
\end{proof}

\begin{corollary}[谱碰撞的算术排除：例外谱与主体谱严格分离]\label{cor:xi-two-param-disjointness-spectrum-disjointness}
在 \(b,d\neq 0\) 下，例外多项式 \(P_{q,a,b}\) 与节点多项式 \(G_{q,d}\) 在 \(\ZZ[\lambda]\) 中互素，且
\[
\Res\bigl(P_{q,a,b},G_{q,d}\bigr)=
(-1)^{\binom{q+2}{2}}\,
b^{q+1}\,d^{q(q+1)}\,
\Bigl(\prod_{i=0}^q\binom{q}{i}\Bigr)\,
\Bigl(\prod_{k=1}^q F_k^{\,2(q+1-k)}\Bigr)\neq 0.
\]
因此对全矩阵 \(B^{(q)}(a,b)\)，主体谱的每个特征值 \(d\mu_i^{(q)}\) 的代数重数恰为 \(\binom{q}{i}-1\)，且不存在任何例外特征值与之相等；例外谱各根亦必为单根。
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-two-param-disjointness-resultant-integer-fibonacci}，结果式为非零整数，故互素；互素性等价于两谱不相交。
单根性亦可由定理 \ref{thm:xi-two-param-disjointness-full-spectrum-secular} 中世俗方程的单调性推出。证毕。
\end{proof}

\end{remark}
