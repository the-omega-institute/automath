\noindent
定义 \ref{def:xi-leyang-zero-divisor-functor} 的 Lee--Yang 零因子张量函子以 \(\CC^{\times}\) 为支撑空间。
当半单参数处于紧实共轭类时（定理 \ref{thm:xi-leyang-compactness-by-fundamental-circle}），其零点支撑事实上落在单位圆。
在协议侧施加“可逆素集合 \(S\)”的相位门纪律时，单位圆上的相干零点可以被提升为 \(S\)-solenoid 上的除子对象，并保持与张量结构严格兼容。
\par

\begin{definition}[紧交换群上的有效除子与乘法推前卷积]\label{def:xi-solenoid-divisor-convolution}
令 \(\Sigma\) 为紧阿贝尔群，并记其群乘法为
\(
\mu_{\Sigma}:\Sigma\times\Sigma\to \Sigma
\)。
定义 \(\Sigma\) 上的有效除子幺半群为
\[
\mathrm{Div}^{+}(\Sigma)
:=\left\{\sum_{i=1}^{n}m_i[\xi_i]:\ n<\infty,\ m_i\in\ZZ_{\ge 1},\ \xi_i\in\Sigma\right\},
\]
其加法为除子相加。
定义乘法卷积
\[
\boxtimes_{\Sigma}:\ \mathrm{Div}^{+}(\Sigma)\times \mathrm{Div}^{+}(\Sigma)\to \mathrm{Div}^{+}(\Sigma),
\qquad
D_1\boxtimes_{\Sigma} D_2
:=(\mu_{\Sigma})_{\ast}(D_1\times D_2).
\]
\end{definition}

\begin{definition}[\texorpdfstring{$S$}{S}-solenoid 上的 Lee--Yang 有效除子]\label{def:xi-leyang-solenoidal-divisor}
固定素数集合 \(S\)，并令 \(\Sigma_S=\widehat{\ZZ[S^{-1}]}\) 为 \S\ref{subsec:cdim-solenoid-terminal-phase-system} 的 \(S\)-solenoid，\(\pi_S:\Sigma_S\twoheadrightarrow \TT\) 为标准满射。
取复约化群 \({}^{L}G\)、其极大环面 \(T\subset {}^{L}G\)，并取半单元 \(g\in {}^{L}G(\CC)\)。
假设 \(g\) 共轭于某个 \(t\in T\cap K\)（其中 \(K\subset {}^{L}G(\CC)\) 为极大紧子群）。
\par
记 \(X^{\ast}(T)\) 为 \(T\) 的角色群，则 \(t\) 给出评价同态
\(
\mathrm{ev}_t:X^{\ast}(T)\to \TT
\)，
\(
\mathrm{ev}_t(\chi)=\chi(t)
\)。
定义 \(\mathrm{ev}_t\) 的一个 \(S\)-相位提升为一个群同态
\[
\widetilde{\mathrm{ev}}_t:\ X^{\ast}(T)\longrightarrow \Sigma_S
\qquad\text{满足}\qquad
\pi_S\circ \widetilde{\mathrm{ev}}_t=\mathrm{ev}_t.
\]
对任意有限维表示 \(V\in \mathrm{Rep}({}^{L}G)\)，将 \(V|_{T}\) 按权分解
\[
V=\bigoplus_{\chi\in X^{\ast}(T)} V_{\chi},
\qquad
t\cdot v=\chi(t)\,v\ \ (v\in V_{\chi}),
\]
并定义 \(S\)-solenoid 上的 Lee--Yang 有效除子
\[
\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V)
:=
\sum_{\chi\in X^{\ast}(T)} \dim(V_{\chi})\,[\widetilde{\mathrm{ev}}_t(\chi)^{-1}]
\ \in\ \mathrm{Div}^{+}(\Sigma_S),
\]
其中逆元取于 \(\Sigma_S\) 的群结构。
\end{definition}

\begin{theorem}[solenoidal Lee--Yang 提升的对称幺半张量函子性]\label{thm:xi-leyang-solenoidal-lift-monoidal}
\leanverified{paper\_xi\_leyang\_solenoidal\_lift\_monoidal}
沿用定义 \ref{def:xi-leyang-solenoidal-divisor} 的记号。
则赋值
\[
\mathrm{Rep}({}^{L}G)\longrightarrow \mathrm{Div}^{+}(\Sigma_S),
\qquad
V\longmapsto \mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V)
\]
满足：
\begin{enumerate}
  \item 对任意 \(V,W\in \mathrm{Rep}({}^{L}G)\)，有
  \[
  \mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V\oplus W)
  =\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V)+\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(W);
  \]
  \item 对任意 \(V,W\in \mathrm{Rep}({}^{L}G)\)，有
  \[
  \mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V\otimes W)
  =\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V)\boxtimes_{\Sigma_S}\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(W),
  \]
  其中 \(\boxtimes_{\Sigma_S}\) 由定义 \ref{def:xi-solenoid-divisor-convolution} 给出。
\end{enumerate}
\end{theorem}
\begin{proof}
直和情形下，权空间分解按加法分裂：
\(
(V\oplus W)_{\chi}=V_{\chi}\oplus W_{\chi}
\)，
从而重数 \(\dim((V\oplus W)_{\chi})=\dim(V_{\chi})+\dim(W_{\chi})\) 逐项相加，即得第一式。
\par
张量积情形下，权满足乘法法则：
\(
(V\otimes W)_{\chi\psi}\cong V_{\chi}\otimes W_{\psi}
\)，
从而
\(
\dim((V\otimes W)_{\chi\psi})=\dim(V_{\chi})\dim(W_{\psi})
\)。
另一方面，由提升同态的乘法性，
\(
\widetilde{\mathrm{ev}}_t(\chi\psi)=\widetilde{\mathrm{ev}}_t(\chi)\cdot \widetilde{\mathrm{ev}}_t(\psi)
\)，
故零点（按重数）在 \(\Sigma_S\) 上按群乘法进行卷积推前。
这正是 \(\boxtimes_{\Sigma_S}\) 的定义。证毕。
\end{proof}

\begin{corollary}[单位圆投影的严格复原]\label{cor:xi-leyang-solenoidal-projection-recovers-circle}
\leanverified{paper\_xi\_leyang\_solenoidal\_projection\_recovers\_circle}
沿用定义 \ref{def:xi-leyang-solenoidal-divisor} 的设定，并将 \(\TT\) 视为 \(\CC^{\times}\) 的子群。
记 \((\pi_S)_{\ast}:\mathrm{Div}^{+}(\Sigma_S)\to \mathrm{Div}^{+}(\TT)\) 为 \(\pi_S\) 的推前。
则对任意 \(V\in\mathrm{Rep}({}^{L}G)\) 有严格等式
\[
(\pi_S)_{\ast}\bigl(\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V)\bigr)
=
\mathrm{LY}_t(V),
\]
其中 \(\mathrm{LY}_t(V)\) 由 \eqref{eq:xi-leyang-zero-divisor} 定义，并且其支撑落在 \(\TT\)（由 \(t\in K\)）。
\end{corollary}
\begin{proof}
对任意权 \(\chi\in X^{\ast}(T)\)，由 \(\pi_S\circ \widetilde{\mathrm{ev}}_t=\mathrm{ev}_t\) 得
\[
\pi_S\bigl(\widetilde{\mathrm{ev}}_t(\chi)\bigr)=\chi(t)\in\TT.
\]
在权分解下 \(V(t)\) 的特征值（按重数）为 \(\chi(t)\)（重数为 \(\dim(V_{\chi})\)），故多项式
\(
P_t(V;z)=\det(I-zV(t))
\)
的零点有效除子为
\(
\sum_{\chi}\dim(V_{\chi})[\chi(t)^{-1}]
\)。
将 \(\mathrm{LY}^{(S)}_{\widetilde{\mathrm{ev}}_t}(V)\) 推前即把
\([\widetilde{\mathrm{ev}}_t(\chi)^{-1}]\) 送到 \([\chi(t)^{-1}]\)，于是得到所示恒等式。证毕。
\end{proof}
