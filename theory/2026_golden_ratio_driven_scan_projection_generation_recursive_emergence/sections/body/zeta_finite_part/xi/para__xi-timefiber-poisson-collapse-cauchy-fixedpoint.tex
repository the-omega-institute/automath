\paragraph{时间纤维的环平均塌缩与 Cauchy 不动点}\label{par:xi-timefiber-poisson-collapse-cauchy-fixedpoint}
本段把“等半径时间纤维”在上半平面的逆像轨道写成显式参数，并给出一个完全精确的环平均塌缩律：对任意实点 \(u\)，沿时间纤维的 Cauchy 核（等价地 Poisson 核）平均与半径 \(\varrho\) 无关，且塌缩为基点 \(\mathrm{i}\) 的值。

\begin{proposition}[Cayley 时间纤维的 Poisson 核塌缩：\texorpdfstring{$\varrho$}{rho}-无关的环平均]\label{prop:xi-cayley-timefiber-poisson-collapse}
\leanverified{paper\_xi\_cayley\_timefiber\_poisson\_collapse}
取 \(0<\varrho<1\)，并令时间纤维的上半平面逆像轨道
\[
t_{\varrho}(\theta):=\mathcal{C}^{-1}(\varrho e^{\mathrm{i}\theta})
=x_{\varrho}(\theta)+\mathrm{i}y_{\varrho}(\theta),
\qquad \theta\in[0,2\pi),
\]
其中 \(\mathcal{C}\) 为本文固定的 Cayley 紧化（式 \eqref{eq:app-cayley-disk}），故
\[
t_{\varrho}(\theta)=\mathrm{i}\,\frac{1-\varrho e^{\mathrm{i}\theta}}{1+\varrho e^{\mathrm{i}\theta}},
\qquad
x_{\varrho}(\theta)=\frac{2\varrho\sin\theta}{1+\varrho^2+2\varrho\cos\theta},\quad
y_{\varrho}(\theta)=\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta}.
\]
定义上半平面 Poisson（亦即 Cauchy）核与基准 Cauchy 密度
\[
P_y(u):=\frac{1}{\pi}\frac{y}{u^2+y^2}\quad(y>0),\qquad
c_0(u):=P_1(u)=\frac{1}{\pi}\frac{1}{1+u^2}.
\]
则对任意 \(u\in\RR\) 同时成立如下两条等价塌缩恒等式：
\begin{align}
&\boxed{\;
\frac{1}{2\pi}\int_{0}^{2\pi}\frac{\dd\theta}{u-t_{\varrho}(\theta)}
=\frac{1}{u-\mathrm{i}}\;}\label{eq:xi-timefiber-stieltjes-cauchy-collapse}\\[2mm]
&\boxed{\;
\frac{1}{2\pi}\int_{0}^{2\pi}P_{y_{\varrho}(\theta)}\!\bigl(u-x_{\varrho}(\theta)\bigr)\,\dd\theta
=c_0(u)\;}\label{eq:xi-timefiber-poisson-ring-collapse}
\end{align}
从而上述环平均严格与 \(\varrho\) 无关。
\end{proposition}

\begin{proof}
记左端为 \(I(u)\)。
令 \(z=e^{\mathrm{i}\theta}\)（\(|z|=1\)），则 \(\dd\theta=\dd z/(\mathrm{i}z)\)，并且
\(
t_{\varrho}(\theta)=\mathrm{i}\frac{1-\varrho z}{1+\varrho z}
\)。
因此
\[
I(u)
=\frac{1}{2\pi\mathrm{i}}\oint_{|z|=1}
\frac{1+\varrho z}{z\bigl((u-\mathrm{i})+\varrho(u+\mathrm{i})z\bigr)}\,\dd z.
\]
被积函数的极点为 \(z=0\) 与 \(z_\ast=-\frac{u-\mathrm{i}}{\varrho(u+\mathrm{i})}\)。
对实 \(u\)，有 \(|u-\mathrm{i}|=|u+\mathrm{i}|\)，故 \(|z_\ast|=\varrho^{-1}>1\)，单位圆内仅含极点 \(z=0\)。
其留数为
\[
\operatorname{Res}_{z=0}\frac{1+\varrho z}{z\bigl((u-\mathrm{i})+\varrho(u+\mathrm{i})z\bigr)}
=\frac{1}{u-\mathrm{i}},
\]
由留数定理得到 \eqref{eq:xi-timefiber-stieltjes-cauchy-collapse}。
\par
另一方面，对任意 \(x\in\RR,y>0\) 有恒等式
\(
\Im\!\left(\frac{1}{u-(x+\mathrm{i}y)}\right)=\frac{y}{(u-x)^2+y^2}
\)。
故由 \eqref{eq:xi-timefiber-stieltjes-cauchy-collapse} 取虚部并乘以 \(\pi^{-1}\) 得
\[
\frac{1}{2\pi}\int_{0}^{2\pi}P_{y_{\varrho}(\theta)}\!\bigl(u-x_{\varrho}(\theta)\bigr)\,\dd\theta
=\frac{1}{\pi}\Im\!\left(\frac{1}{u-\mathrm{i}}\right)
=\frac{1}{\pi}\frac{1}{1+u^2}
=c_0(u),
\]
即 \eqref{eq:xi-timefiber-poisson-ring-collapse}。证毕。
\end{proof}

\begin{corollary}[时间纤维诱导的随机仿射不变性：Cauchy 密度的精确不动点]\label{cor:xi-timefiber-cauchy-fixedpoint}
令 \(\Theta\sim\mathrm{Unif}[0,2\pi)\)，并置
\[
X_\varrho:=x_\varrho(\Theta),\qquad Y_\varrho:=y_\varrho(\Theta)>0.
\]
令随机变量 \(U\) 与 \((X_\varrho,Y_\varrho)\) 独立，且 \(U\) 服从密度 \(c_0(u)=\frac{1}{\pi}\frac{1}{1+u^2}\)。
则
\[
U':=X_\varrho+Y_\varrho\,U
\quad\Longrightarrow\quad
U'\ \stackrel{d}{=}\ U,
\qquad \forall\,\varrho\in(0,1).
\]
\end{corollary}

\begin{proof}
条件于 \((X_\varrho,Y_\varrho)=(x,y)\) 时，随机变量 \(x+yU\) 的条件密度为 \(P_y(u-x)\)。
因此
\[
\mathrm{dens}(U')=\mathbb{E}\bigl[P_{Y_\varrho}(u-X_\varrho)\bigr].
\]
由命题 \ref{prop:xi-cayley-timefiber-poisson-collapse} 的 \eqref{eq:xi-timefiber-poisson-ring-collapse} 得
\(
\mathbb{E}[P_{Y_\varrho}(u-X_\varrho)]=c_0(u)
\)，从而 \(U'\) 仍服从 \(c_0\)。证毕。
\end{proof}

\begin{theorem}[时间纤维随机仿射系统的收缩率、平稳律唯一性与 Cauchy 识别]\label{thm:xi-timefiber-random-affine-lyapunov-cauchy-unique}
\leanverified{paper\_xi\_timefiber\_random\_affine\_lyapunov\_cauchy\_unique}
固定 \(0<\varrho<1\)，并令 \((X_n,Y_n)_{n\ge 0}\) 为独立同分布样本，分布同 \((X_\varrho,Y_\varrho)\)。
对任意初值 \(U_0\in\RR\)，定义随机递推
\[
U_{n+1}=X_n+Y_nU_n\qquad(n\ge 0).
\]
则成立：
\begin{enumerate}
  \item \(Y_n>0\) 几乎必然，且对任意两条在同一噪声下耦合的轨道 \((U_n^{(u)},U_n^{(v)})\)（\(u\neq v\)）有顶 Lyapunov 指数极限
  \[
  \lambda(\varrho)
  :=\lim_{n\to\infty}\frac1n\log|U_n^{(u)}-U_n^{(v)}|
  =\mathbb{E}[\log Y_\varrho]
  =\log(1-\varrho^2)<0.
  \]
  \item 对任意初值 \(U_0\in\RR\)，极限 \(U_\infty:=\lim_{n\to\infty}U_n\) 几乎必然存在，并且其分布与 \(U_0\) 无关；相应地，\((U_n)\) 在 \(\RR\) 上存在唯一不变分布。
  \item 该唯一不变分布恰为标准 Cauchy 分布 \(c_0(u)=\frac{1}{\pi(1+u^2)}\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(U_{n+1}^{(u)}-U_{n+1}^{(v)}=Y_n(U_n^{(u)}-U_n^{(v)})\) 得
\[
U_n^{(u)}-U_n^{(v)}=\Bigl(\prod_{k=0}^{n-1}Y_k\Bigr)(u-v),
\]
从而
\(
\frac1n\log|U_n^{(u)}-U_n^{(v)}|
=\frac1n\sum_{k=0}^{n-1}\log Y_k+\frac1n\log|u-v|
\)。
由强大数定律得第一条中的
\(
\lim_{n\to\infty}\frac1n\log|U_n^{(u)}-U_n^{(v)}|
=\mathbb{E}[\log Y_\varrho]
\)。
\par
为计算 \(\mathbb{E}[\log Y_\varrho]\)，由
\(
Y_\varrho(\theta)=\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta}
=\frac{1-\varrho^2}{|1+\varrho e^{\mathrm{i}\theta}|^2}
\)
得到
\[
\mathbb{E}[\log Y_\varrho]
=\log(1-\varrho^2)-\mathbb{E}\bigl[\log|1+\varrho e^{\mathrm{i}\Theta}|^2\bigr]
=\log(1-\varrho^2)-2\mathbb{E}\bigl[\log|1+\varrho e^{\mathrm{i}\Theta}|\bigr].
\]
对解析函数 \(f(z)=1+\varrho z\)（\(|\varrho|<1\)）应用 Jensen 公式可得
\(
\frac1{2\pi}\int_{0}^{2\pi}\log|f(e^{\mathrm{i}\theta})|\,\dd\theta=\log|f(0)|=0
\)，
故 \(\mathbb{E}[\log|1+\varrho e^{\mathrm{i}\Theta}|]=0\)，于是
\(
\mathbb{E}[\log Y_\varrho]=\log(1-\varrho^2)<0
\)。
\par
由 \(\mathbb{E}[\log Y_\varrho]<0\) 得 \(\prod_{k=0}^{n-1}Y_k\to 0\) 几乎必然，因此对任意两初值 \(u,v\) 有 \(|U_n^{(u)}-U_n^{(v)}|\to 0\) 几乎必然；取 \(v=0\) 即知轨道差异被指数收缩。
同时，由 \(X_\varrho\) 与 \(Y_\varrho\) 的显式有界性可知 \(\mathbb{E}[\log^+|X_\varrho|]<\infty\)，从而递推的随机级数展开
\(
U_n=\sum_{j=0}^{n-1}X_{n-1-j}\prod_{\ell=0}^{j-1}Y_{n-1-\ell}+\bigl(\prod_{k=0}^{n-1}Y_k\bigr)U_0
\)
在 \(n\to\infty\) 下几乎必然收敛，得到极限 \(U_\infty\) 的存在性；由耦合收缩性，极限分布与初值无关并给出平稳律唯一性。
\par
最后，由推论 \ref{cor:xi-timefiber-cauchy-fixedpoint}，标准 Cauchy 分布在随机仿射更新 \(U\mapsto X_\varrho+Y_\varrho U\) 下保持不变，因而是一个不变分布。
结合唯一性，得到该不变分布必为 \(c_0\)。证毕。
\end{proof}

\begin{proposition}[时间纤维平均的投影原则：Poisson 延拓在环平均下的 \texorpdfstring{$\varrho$}{rho}-独立性]\label{prop:xi-timefiber-poisson-projection-principle}
\leanverified{paper\_xi\_timefiber\_poisson\_projection\_principle}
令 \(f\in L^1(\RR,c_0(u)\,\dd u)\)，并定义其上半平面 Poisson 延拓
\[
(\mathcal P f)(x+\mathrm{i}y):=\int_{\RR} f(u)\,P_y(x-u)\,\dd u
\qquad (y>0).
\]
则对任意 \(\varrho\in(0,1)\) 成立精确恒等式
\[
\boxed{\;
\frac{1}{2\pi}\int_{0}^{2\pi}(\mathcal P f)\!\bigl(t_\varrho(\theta)\bigr)\,\dd\theta
=\int_{\RR}f(u)\,c_0(u)\,\dd u
=(\mathcal P f)(\mathrm{i})\; }.
\]
\end{proposition}

\begin{proof}
由 Fubini 定理（\(f\in L^1(\RR,c_0)\) 足以保证交换积分合法）得
\[
\frac{1}{2\pi}\int_{0}^{2\pi}(\mathcal P f)(t_\varrho(\theta))\,\dd\theta
=\int_{\RR} f(u)\left[\frac{1}{2\pi}\int_{0}^{2\pi}P_{y_\varrho(\theta)}\!\bigl(x_\varrho(\theta)-u\bigr)\,\dd\theta\right]\dd u.
\]
括号内由 \eqref{eq:xi-timefiber-poisson-ring-collapse} 等于 \(c_0(u)\)，故得所示恒等式。证毕。
\end{proof}

\begin{definition}[熵型审计泛函]\label{def:xi-cauchy-poisson-entropy-audit-functional}
对任意满足 \(h>0\)、\(\log h\in L^1(\RR,c_0(u)\,\dd u)\) 且 \(\int_{\RR}h(u)\,c_0(u)\,\dd u=1\) 的密度比，定义
\[
\mathcal J(h):=-\int_{\RR}\log h(u)\,c_0(u)\,\dd u.
\]
\end{definition}

\begin{proposition}[共动圆审计表达式]\label{prop:xi-cauchy-poisson-entropy-audit-circle-average}
在定义 \ref{def:xi-cauchy-poisson-entropy-audit-functional} 的设定下，
\(\mathcal J(h)\ge 0\)，并且对任意 \(\varrho\in(0,1)\) 有精确恒等式
\[
\boxed{\;
\mathcal J(h)
=-\frac{1}{2\pi}\int_{0}^{2\pi}\bigl(\mathcal P\log h\bigr)\!\bigl(t_\varrho(\theta)\bigr)\,\dd\theta\; }.
\]
\end{proposition}
\begin{proof}
由命题 \ref{prop:xi-timefiber-poisson-projection-principle} 取 \(f=\log h\) 得
\[
\frac{1}{2\pi}\int_{0}^{2\pi}(\mathcal P\log h)(t_\varrho(\theta))\,\dd\theta
=\int_{\RR}\log h(u)\,c_0(u)\,\dd u,
\]
移项即得所示恒等式。
另一方面，由 Jensen 不等式与归一化 \(\int h\,c_0=1\) 有
\[
\int_{\RR}\log h\,c_0
\le \log\int_{\RR}h\,c_0=\log 1=0,
\]
故 \(\mathcal J(h)=-\int \log h\,c_0\ge 0\)。证毕。
\end{proof}
