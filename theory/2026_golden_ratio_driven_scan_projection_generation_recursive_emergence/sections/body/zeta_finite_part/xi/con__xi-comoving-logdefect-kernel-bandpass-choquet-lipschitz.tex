\paragraph{共动圆盘化下的对数缺陷核：二次载荷比较、尺度带通分解与稳定常数}
在共动圆盘化记号下，对任意 \(\gamma\in\RR\)、\(\delta\in(0,\tfrac12]\)，定义
\[
\ell_{\gamma,\delta}(x)
:=
-\log\Bigl|\mathcal{C}\bigl((x-\gamma)+\mathrm{i}\delta\bigr)\Bigr|
=
\frac12\log\!\frac{(x-\gamma)^2+(1+\delta)^2}{(x-\gamma)^2+(1-\delta)^2}
\ge 0.
\]
并记对应二次载荷核
\[
h_{\gamma,\delta}(x)
:=
1-\Bigl|\mathcal{C}\bigl((x-\gamma)+\mathrm{i}\delta\bigr)\Bigr|^2
=
\frac{4\delta}{(x-\gamma)^2+(1+\delta)^2}.
\]
当 \(x=x_0\) 且 \((\gamma,\delta)\) 对应离线零点 \(\rho\) 时，\(h_{\gamma,\delta}(x_0)\) 与
\eqref{eq:xi-comoving-prefix-endpoint-margin} 的 \(h_{x_0}(\rho)\) 一致。

\begin{theorem}[临界带内对数载荷与二次载荷的全局双边比较]\label{thm:xi-logdefect-vs-quadratic-load-two-sided}
\leanverified{paper\_xi\_logdefect\_vs\_quadratic\_load\_two\_sided}
对任意 \(\gamma\in\RR\)、\(\delta\in(0,\tfrac12]\) 与 \(x\in\RR\)，有
\[
\frac12\,h_{\gamma,\delta}(x)
\le
\ell_{\gamma,\delta}(x)
\le
\frac92\,h_{\gamma,\delta}(x).
\]
因而对任意有限族 \(\{(\gamma_j,\delta_j,m_j)\}_{j=1}^{M}\subset \RR\times(0,\tfrac12]\times\NN\)，定义
\[
\Phi(x):=\sum_{j=1}^{M}m_j\,\ell_{\gamma_j,\delta_j}(x),
\qquad
H(x):=\sum_{j=1}^{M}m_j\,h_{\gamma_j,\delta_j}(x),
\]
则逐点比较仍成立：
\[
\frac12\,H(x)\le \Phi(x)\le \frac92\,H(x)\qquad(x\in\RR).
\]
\end{theorem}
\begin{proof}
设
\[
r(x):=\Bigl|\mathcal{C}\bigl((x-\gamma)+\mathrm{i}\delta\bigr)\Bigr|\in(0,1),
\qquad
h_{\gamma,\delta}(x)=1-r(x)^2.
\]
由基本不等式
\[
1-u\le -\log u\le \frac{1-u}{u}\qquad(u\in(0,1)),
\]
取 \(u=r(x)^2\)，得
\[
\ell_{\gamma,\delta}(x)
=-\frac12\log r(x)^2
\ge \frac12\bigl(1-r(x)^2\bigr)
=\frac12\,h_{\gamma,\delta}(x).
\]
另一方面
\[
\ell_{\gamma,\delta}(x)
\le \frac12\,\frac{1-r(x)^2}{r(x)^2}
=\frac12\,\frac{h_{\gamma,\delta}(x)}{r(x)^2}.
\]
由
\[
r(x)^2=\frac{(x-\gamma)^2+(1-\delta)^2}{(x-\gamma)^2+(1+\delta)^2}
\ge \frac{(1-\delta)^2}{(1+\delta)^2}
\ge \frac{(1/2)^2}{(3/2)^2}
=\frac19
\]
（因 \(\delta\le\tfrac12\)），故
\[
\ell_{\gamma,\delta}(x)\le \frac12\cdot 9\,h_{\gamma,\delta}(x)=\frac92\,h_{\gamma,\delta}(x).
\]
对有限族逐项求和即得 \(\Phi\) 与 \(H\) 的比较。证毕。
\end{proof}

\begin{theorem}[对数缺陷核的尺度带通 Poisson 表示与正算子分解]\label{thm:xi-logdefect-bandpass-poisson-representation}
\leanverified{paper\_xi\_logdefect\_bandpass\_poisson\_representation}
对任意 \(\gamma\in\RR\)、\(\delta\in(0,\tfrac12]\)，有精确恒等式
\[
\ell_{\gamma,\delta}(x)
=
\int_{1-\delta}^{1+\delta}\frac{y}{(x-\gamma)^2+y^2}\,\dd y
=
\pi\int_{1-\delta}^{1+\delta}P_y(x-\gamma)\,\dd y,
\]
其中
\[
P_y(u):=\frac{1}{\pi}\frac{y}{u^2+y^2}\qquad(y>0)
\]
为实线 Poisson 核。

进一步，对任意有限正 Borel 测度 \(\nu\)（定义在 \(\RR\times(0,\tfrac12]\)）记
\[
L_\nu(x):=\int_{\RR\times(0,1/2]} \ell_{\gamma,\delta}(x)\,\dd\nu(\gamma,\delta).
\]
则有尺度带通正分解
\[
L_\nu(x)
=
\pi\int_{1/2}^{3/2}\bigl(P_y*\sigma_\nu(y,\cdot)\bigr)(x)\,\dd y,
\]
其中对 \(y\in[1/2,3/2]\) 定义关于 \(\gamma\) 的有限正测度
\[
\sigma_\nu(y,B)
:=
\nu\!\bigl(\{(\gamma,\delta):\ \gamma\in B,\ y\in[1-\delta,1+\delta]\}\bigr),
\qquad
B\in\mathcal{B}(\RR).
\]
\end{theorem}
\begin{proof}
直接积分
\[
\int \frac{y}{(x-\gamma)^2+y^2}\,\dd y
=
\frac12\log\!\bigl((x-\gamma)^2+y^2\bigr)
\]
并代入上下限 \(y=1\pm\delta\)，即得第一组恒等式。
再用 \(P_y(u)=(\pi)^{-1}y/(u^2+y^2)\) 得第二组表示。

对 \(L_\nu\)，由被积函数非负可用 Tonelli--Fubini 换序：
\[
\begin{aligned}
L_\nu(x)
&=
\pi\int_{\RR\times(0,1/2]}\int_{1-\delta}^{1+\delta}P_y(x-\gamma)\,\dd y\,\dd\nu(\gamma,\delta)\\
&=
\pi\int_{1/2}^{3/2}\int_{\RR}P_y(x-\gamma)\,\dd\sigma_\nu(y,\gamma)\,\dd y.
\end{aligned}
\]
即得所述分解。证毕。
\end{proof}

\begin{theorem}[对数缺陷锥的 Choquet--Bauer 型完备性]\label{thm:xi-logdefect-cone-choquet-bauer-completeness}
定义凸锥
\[
\mathfrak{L}
:=
\left\{
L:\ 
L(x)=\int_{\RR\times(0,1/2]}\ell_{\gamma,\delta}(x)\,\dd\nu(\gamma,\delta),\
\nu\in\mathcal{M}_+\bigl(\RR\times(0,1/2]\bigr)
\right\}.
\]
记线性映射 \(T:\nu\mapsto L_\nu\)。则：
\begin{enumerate}
  \item \(T\) 是从 \(\mathcal{M}_+(\RR\times(0,1/2])\) 到 \(\mathfrak{L}\) 的线性双射；
  \item 若 \(L_{\nu_1}\equiv L_{\nu_2}\)，则 \(\nu_1=\nu_2\)；
  \item \(\mathfrak{L}\) 的极射线恰为单核射线
  \[
  \mathrm{ExtRays}(\mathfrak{L})
  =
  \{\lambda\,\ell_{\gamma,\delta}:\ \lambda>0,\ (\gamma,\delta)\in\RR\times(0,1/2]\}.
  \]
\end{enumerate}
因此 \(\mathfrak{L}\) 可视作 Choquet--Bauer 型单纯锥。
\leanverified{paper\_xi\_logdefect\_cone\_choquet\_bauer\_completeness}
\end{theorem}
\begin{proof}[证明要点]
线性与满射由定义直接成立。

记 \(\mu:=\nu_1-\nu_2\) 为有限符号测度，若 \(L_{\nu_1}\equiv L_{\nu_2}\) 则 \(L_\mu\equiv0\)。
由定理 \ref{thm:xi-logdefect-bandpass-poisson-representation} 与
\(\widehat{P_y}(\xi)=e^{-y|\xi|}\)，对 \(\xi\neq0\) 有
\[
\widehat{L_\mu}(\xi)
=
\frac{\pi}{|\xi|}
\int_{\RR\times(0,1/2]}
e^{-\mathrm{i}\gamma\xi}
\bigl(e^{-(1-\delta)|\xi|}-e^{-(1+\delta)|\xi|}\bigr)\,\dd\mu(\gamma,\delta).
\]
故 \(\widehat{L_\mu}\equiv0\) 等价于上式右端恒为零。
由有限符号测度的 Fourier--Laplace 变换唯一性（参数区间 \(\delta\in(0,\tfrac12]\) 紧，核在 \(\xi\neq0\) 上解析且非退化）可推出 \(\mu=0\)，故 \(\nu_1=\nu_2\)。

极射线部分：\(\mathcal{M}_+\) 的极射线是 Dirac 射线 \(\lambda\delta_{(\gamma,\delta)}\)。
由 \(T\) 的线性双射性，极射线在锥同构下保持，故 \(\mathfrak{L}\) 的极射线正是其像 \(\lambda\ell_{\gamma,\delta}\)。证毕。
\end{proof}

\begin{theorem}[对数缺陷核的零阶矩与一阶仿射再现恒等式]\label{thm:xi-logdefect-affine-reproducing-moment-identities}
\leanverified{paper\_xi\_log\_defect\_affine\_reproducing\_moment\_identities}
对任意 \(\gamma\in\RR\)、\(\delta\in(0,\tfrac12]\)，有
\[
\int_{\RR}\ell_{\gamma,\delta}(x)\,\dd x=2\pi\delta.
\]
并且一阶矩在主值意义下满足
\[
\mathrm{p.v.}\!\int_{\RR}x\,\ell_{\gamma,\delta}(x)\,\dd x
=
2\pi\delta\,\gamma.
\]
因此对任意仿射函数 \(p(x)=ax+b\)，
\[
\mathrm{p.v.}\!\int_{\RR}p(x)\,\ell_{\gamma,\delta}(x)\,\dd x
=
2\pi\delta\,p(\gamma).
\]
进而若 \(\nu\in\mathcal{M}_+(\RR\times(0,1/2])\) 且
\(\int |\delta\gamma|\,\dd\nu<\infty\)，则
\[
\int_{\RR}L_\nu(x)\,\dd x
=
2\pi\int \delta\,\dd\nu,
\]
\[
\mathrm{p.v.}\!\int_{\RR}x\,L_\nu(x)\,\dd x
=
2\pi\int \delta\gamma\,\dd\nu.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-logdefect-bandpass-poisson-representation}，
\[
\int_{\RR}\ell_{\gamma,\delta}(x)\,\dd x
=
\int_{1-\delta}^{1+\delta}\left(\int_{\RR}\frac{y}{(x-\gamma)^2+y^2}\,\dd x\right)\dd y
=
\int_{1-\delta}^{1+\delta}\pi\,\dd y
=
2\pi\delta.
\]

写 \(x=\gamma+u\)，则
\[
\mathrm{p.v.}\!\int_{\RR}x\,\ell_{\gamma,\delta}(x)\,\dd x
=
\gamma\int_{\RR}\ell_{0,\delta}(u)\,\dd u
\mathrm{p.v.}\!\int_{\RR}u\,\ell_{0,\delta}(u)\,\dd u.
\]
\(\ell_{0,\delta}\) 为偶函数，故第二项主值为零；第一项由零阶矩给出 \(2\pi\delta\gamma\)。
仿射函数情形由线性性得到。

对 \(L_\nu\) 的两式由 Tonelli--Fubini 与上式逐项积分得到。证毕。
\end{proof}

\begin{theorem}[参数 \(\gamma\) 平移在 \texorpdfstring{$L^1$}{L1} 中的精确 Lipschitz 常数]\label{thm:xi-logdefect-gamma-translation-l1-lipschitz}
\leanverified{paper\_xi\_log\_defect\_gamma\_translation\_l1\_lipschitz}
\leanverified{paper\_xi\_logdefect\_gamma\_translation\_l1\_lipschitz}
固定 \(\delta\in(0,\tfrac12]\)，映射
\[
\Gamma_\delta:\gamma\longmapsto \ell_{\gamma,\delta}\in L^1(\RR)
\]
全局 Lipschitz，且
\[
\|\ell_{\gamma,\delta}-\ell_{\gamma',\delta}\|_{L^1(\RR)}
\le
2\,|\gamma-\gamma'|\,\log\!\frac{1+\delta}{1-\delta}.
\]
特别地，
\[
2\log\!\frac{1+\delta}{1-\delta}\le 2\log 3
\qquad(\delta\le\tfrac12).
\]
\end{theorem}
\begin{proof}
由平移结构 \(\ell_{\gamma,\delta}(x)=\ell_{0,\delta}(x-\gamma)\)，令 \(\Delta:=\gamma-\gamma'\)：
\[
\|\ell_{\gamma,\delta}-\ell_{\gamma',\delta}\|_1
=
\int_{\RR}\bigl|\ell_{0,\delta}(u)-\ell_{0,\delta}(u+\Delta)\bigr|\,\dd u
\le
|\Delta|\,\|\ell_{0,\delta}'\|_1.
\]
其中用到绝对连续函数估计
\(
\|f(\cdot)-f(\cdot+\Delta)\|_1\le |\Delta|\,\|f'\|_1
\)。

设 \(a:=1+\delta,\ b:=1-\delta\)，则
\[
\ell_{0,\delta}'(u)
=
\frac{u}{u^2+a^2}-\frac{u}{u^2+b^2}
=
-\frac{4\delta\,u}{(u^2+a^2)(u^2+b^2)}.
\]
故
\[
\|\ell_{0,\delta}'\|_1
=
8\delta\int_0^\infty \frac{u}{(u^2+a^2)(u^2+b^2)}\,\dd u.
\]
令 \(t=u^2\)，得
\[
\int_0^\infty \frac{u}{(u^2+a^2)(u^2+b^2)}\,\dd u
=
\frac12\int_0^\infty\frac{\dd t}{(t+a^2)(t+b^2)}
=
\frac{1}{2(a^2-b^2)}\log\frac{a^2}{b^2}.
\]
又 \(a^2-b^2=4\delta\)，故
\[
\|\ell_{0,\delta}'\|_1
=
2\log\frac{a}{b}
=
2\log\frac{1+\delta}{1-\delta}.
\]
代回即得 Lipschitz 常数闭式。
对 \(\delta\le\tfrac12\)，有 \((1+\delta)/(1-\delta)\le 3\)，遂得统一上界 \(2\log3\)。证毕。
\end{proof}

\paragraph{证书接口含义}
上述链条在形式上建立了二次载荷 \(H\) 与对数载荷 \(\Phi\) 的绝对常数桥接，并将对数载荷重写为尺度带通的 Poisson 正分解；
由此，\(\ell\)-核锥的极射线结构与仿射再现不变量可直接并入“纤维账本—证书链”接口，
而 \(\gamma\)-方向的 \(L^1\)-Lipschitz 常数给出离线评估中的显式稳定阈值。
