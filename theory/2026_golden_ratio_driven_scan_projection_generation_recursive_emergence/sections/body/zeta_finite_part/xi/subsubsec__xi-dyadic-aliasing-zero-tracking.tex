\subsubsection{Poisson 混叠、二进制细化与零点跟踪：\texorpdfstring{$\Xi$}{Xi} 的离散余弦采样证书}\label{subsubsec:xi-dyadic-aliasing-zero-tracking}
本节把 unitary slice 的“固定网格、二进制细化、离线可复核”制度落实为一条可逐行核验的解析链条。
核心输入是 \(\Xi\) 的余弦变换表示与 Poisson 求和：固定步长的梯形采样并非近似技巧，而严格等价于对真信号的平移别名叠加；别名尾和在临界线指数衰减下给出显式的超快误差界；在单零点邻域内则由 Rouch\'e 定理与共轭对称性推出唯一零点跟踪，从而把“分辨率加倍 \(\Rightarrow\) 证书区间指数收缩”编译为可审计预算律。

\paragraph{完成化与余弦核}
记完成的 Riemann \(\xi\)-函数
\[
\xi(s):=\frac12\,s(s-1)\pi^{-s/2}\Gamma\!\Bigl(\frac{s}{2}\Bigr)\zeta(s),
\qquad
\Xi(t):=\xi\!\Bigl(\frac12+\mathrm{i}t\Bigr)\qquad(t\in\RR).
\]
则 \(\Xi\) 为实偶整函数。
存在快速衰减的实偶核 \(\Phi\) 使得 \(\Xi\) 为其余弦变换
\begin{equation}\label{eq:xi-cosine-kernel-representation}
\boxed{\;
\Xi(t)=\int_{0}^{\infty}\Phi(x)\cos(tx)\,\dd x\;}
\end{equation}
（参见 \cite{Shi2015RealRootedPolyaXi} 中对核 \(\Phi\) 的显式构造）。

\paragraph{固定步长离散化与别名叠加}
对任意步长 \(h>0\) 定义离散余弦采样
\begin{equation}\label{eq:xi-dyadic-discrete-cosine-sampling}
\Xi_h(t):=
h\sum_{k\in\ZZ}\Phi(kh)\cos(tkh)\qquad(t\in\RR).
\end{equation}
由于 \(\Phi\) 快速衰减，上式绝对收敛并在复平面紧集上一致收敛，因此 \(\Xi_h\) 延拓为整函数，且在实轴上仍为实偶函数。

\begin{theorem}[Poisson 混叠恒等式、二进制细化递推与别名尾项界]\label{thm:xi-dyadic-aliasing-poisson-and-error}
在表示 \eqref{eq:xi-cosine-kernel-representation} 与采样 \eqref{eq:xi-dyadic-discrete-cosine-sampling} 的设定下，成立如下结论。
\begin{enumerate}
  \item \textbf{（Poisson 混叠）}
  对所有 \(t\in\RR\) 有严格恒等式
  \begin{equation}\label{eq:xi-poisson-aliasing-identity}
  \boxed{\;
  \Xi_h(t)=\sum_{m\in\ZZ}\Xi\!\Bigl(t+\frac{2\pi m}{h}\Bigr)\;}
  \end{equation}
  从而别名误差具有闭式表示
  \begin{equation}\label{eq:xi-poisson-aliasing-error-closed-form}
  \boxed{\;
  \Xi_h(t)-\Xi(t)=\sum_{m\neq 0}\Xi\!\Bigl(t+\frac{2\pi m}{h}\Bigr)\; }.
  \end{equation}
  \item \textbf{（二进制细化递推）}
  将指标分解为偶/奇两类，得到严格递推
  \begin{equation}\label{eq:xi-dyadic-refinement-recursion}
  \boxed{\;
  \Xi_{h/2}(t)
  =\frac12\,\Xi_h(t)
  +\frac{h}{2}\sum_{k\in\ZZ}\Phi\!\Bigl(\bigl(k+\tfrac12\bigr)h\Bigr)\,
  \cos\!\Bigl(\bigl(k+\tfrac12\bigr)ht\Bigr)\;}
  \end{equation}
  其中第二项把“分辨率加倍所引入的增量”隔离为半格点采样项。
  \item \textbf{（固定紧区间上一致的指数级别名尾界）}
  对任意 \(\varepsilon>0\) 存在常数 \(A_\varepsilon,C_\varepsilon>0\) 使得
  \begin{equation}\label{eq:xi-critical-line-exponential-decay}
  \abs{\Xi(u)}\le C_\varepsilon(1+\abs{u})^{A_\varepsilon}\exp\!\Bigl(-\frac{\pi}{4}\abs{u}\Bigr)\qquad(u\in\RR),
  \end{equation}
  从而对任意 \(T>0\) 存在常数 \(C_{T,\varepsilon}>0\) 使得
  \begin{equation}\label{eq:xi-aliasing-sup-error-exp}
  \boxed{\;
  \sup_{\abs{t}\le T}\abs{\Xi_h(t)-\Xi(t)}
  \le C_{T,\varepsilon}\exp\!\Bigl(-\frac{\pi^{2}}{2h}\Bigr)\;}
  \end{equation}
  并且 \eqref{eq:xi-aliasing-sup-error-exp} 的主尺度随 \(h\mapsto h/2\) 以 \(\exp(-c/h)\) 型超快方式收缩（参见 \cite{TrefethenWeideman2014TrapezoidalRule} 关于 Poisson 求和/梯形法与指数收敛的系统讨论）。
\end{enumerate}
\end{theorem}
\begin{proof}[证明要点]
\begin{enumerate}
  \item 令 Fourier 变换 \(F(t):=\int_{\RR}\Phi(x)e^{\mathrm{i}tx}\,\dd x\)。由 \(\Phi\) 的偶性与 \eqref{eq:xi-cosine-kernel-representation} 得 \(F(t)=2\Xi(t)\)。
  Poisson 求和公式给出
  \(
  h\sum_{k\in\ZZ}\Phi(kh)e^{\mathrm{i}tkh}
  =\sum_{m\in\ZZ}F(t+2\pi m/h)
  \)；
  取实部即得 \eqref{eq:xi-poisson-aliasing-identity}，并推出 \eqref{eq:xi-poisson-aliasing-error-closed-form}。
  \item 由定义 \(\Xi_{h/2}(t)=\frac{h}{2}\sum_{n\in\ZZ}\Phi(nh/2)\cos(tnh/2)\)，将 \(n=2k\) 与 \(n=2k+1\) 分离并整理即得 \eqref{eq:xi-dyadic-refinement-recursion}。
  \item 由 \(\zeta(\tfrac12+\mathrm{i}u)\) 的凸性界与 \(\Gamma\)-因子的 Stirling 型衰减可得 \eqref{eq:xi-critical-line-exponential-decay}（可参见 \cite{Titchmarsh1986RiemannZeta}）。
  将 \eqref{eq:xi-poisson-aliasing-error-closed-form} 的尾和按 \(|t+2\pi m/h|\ge 2\pi|m|/h-T\) 估计，并用 \eqref{eq:xi-critical-line-exponential-decay} 控制几何级数，即得 \eqref{eq:xi-aliasing-sup-error-exp}。
\end{enumerate}
\end{proof}

\begin{theorem}[单零点的唯一跟踪与二进制分辨率指数收敛]\label{thm:xi-dyadic-zero-tracking}
\leanverified{paper\_xi\_dyadic\_zero\_tracking}
设 \(\gamma\in\RR\) 为 \(\Xi\) 的单零点：\(\Xi(\gamma)=0\)、\(\Xi'(\gamma)\neq 0\)。
则存在 \(r=r(\gamma)>0\) 与 \(h_0=h_0(\gamma)>0\) 使得对所有 \(0<h\le h_0\)：
\begin{enumerate}
  \item \(\Xi_h\) 在圆盘 \(D(\gamma,r)=\{z\in\CC:\ \abs{z-\gamma}<r\}\) 内恰有一个零点（按重数计），记为 \(\gamma_h\)；
  \item 该零点必为实数：\(\gamma_h\in\RR\)；
  \item 其漂移满足显式控制
  \begin{equation}\label{eq:xi-zero-drift-lipschitz}
  \boxed{\;
  \abs{\gamma_h-\gamma}
  \le
  \frac{2}{\abs{\Xi'(\gamma)}}\,
  \sup_{\abs{t-\gamma}\le r}\abs{\Xi_h(t)-\Xi(t)}\;}
  \end{equation}
  并且取 dyadic 网格 \(h_m:=2^{-m}\) 时存在常数 \(C_{\gamma,\varepsilon}>0\) 使
  \begin{equation}\label{eq:xi-zero-drift-dyadic-exp}
  \boxed{\;
  \abs{\gamma_{h_m}-\gamma}
  \le
  C_{\gamma,\varepsilon}\exp\!\bigl(-(\pi^2/2)\,2^{m}\bigr)\;}
  \end{equation}
  其中指数常数 \(\pi^2/2\) 来自定理 \ref{thm:xi-dyadic-aliasing-poisson-and-error} 的别名尾界 \eqref{eq:xi-aliasing-sup-error-exp}。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(\Xi'(\gamma)\neq 0\) 可取 \(r>0\) 使得
\(
\inf_{\abs{z-\gamma}=r}\abs{\Xi(z)}\ge \frac12\abs{\Xi'(\gamma)}r
\)。
再取 \(h_0\) 使得对 \(0<h\le h_0\),
\(
\sup_{\abs{z-\gamma}=r}\abs{\Xi_h(z)-\Xi(z)}<\frac12\abs{\Xi'(\gamma)}r
\)；
则由 Rouch\'e 定理知 \(\Xi_h\) 与 \(\Xi\) 在 \(D(\gamma,r)\) 内零点数相同，故存在且仅存在一个零点 \(\gamma_h\)。
\par
由于 \(\Xi_h\) 在实轴上取实值且其零点关于实轴成共轭对称，而 \(D(\gamma,r)\) 关于实轴对称且仅含一个零点，故该零点必落在实轴上，得到 \(\gamma_h\in\RR\)。
\par
由 \(\Xi_h(\gamma_h)=0\) 得 \(\abs{\Xi(\gamma_h)}=\abs{\Xi(\gamma_h)-\Xi_h(\gamma_h)}\le \sup_{\abs{t-\gamma}\le r}\abs{\Xi_h(t)-\Xi(t)}\)。
同时，由均值定理与 \(\Xi(\gamma)=0\) 有 \(\abs{\Xi(\gamma_h)}\ge \frac12\abs{\Xi'(\gamma)}\abs{\gamma_h-\gamma}\)（取 \(r\) 更小可保证 \(\inf_{\abs{t-\gamma}\le r}\abs{\Xi'(t)}\ge \frac12\abs{\Xi'(\gamma)}\)）。
合并即得 \eqref{eq:xi-zero-drift-lipschitz}。
将 \eqref{eq:xi-aliasing-sup-error-exp} 代入并令 \(h=h_m\) 即得 \eqref{eq:xi-zero-drift-dyadic-exp}。
\end{proof}

\begin{theorem}[离散化误差与截断误差的分离控制：可核验零点包含区间]\label{thm:xi-dyadic-truncation-separation-and-interval}
在定理 \ref{thm:xi-dyadic-aliasing-poisson-and-error} 的设定下，对 \(K\ge 0\) 定义有限截断
\[
\Xi_{h}^{[K]}(t):=
h\sum_{\abs{k}\le K}\Phi(kh)\cos(tkh).
\]
则对任意 \(T>0\) 有完全分离的误差分解
\begin{equation}\label{eq:xi-truncation-error-separation}
\boxed{\;
\sup_{\abs{t}\le T}\abs{\Xi(t)-\Xi_{h}^{[K]}(t)}
\le
\underbrace{\sup_{\abs{t}\le T}\sum_{m\ne 0}\abs{\Xi\!\Bigl(t+\frac{2\pi m}{h}\Bigr)}}_{\text{别名误差}}
\;+\;
\underbrace{2h\sum_{k>K}\abs{\Phi(kh)}}_{\text{截断误差}}\;}
\end{equation}
其中别名误差在固定 \(T\) 下满足 \(\ll \exp(-\pi^2/(2h))\)（定理 \ref{thm:xi-dyadic-aliasing-poisson-and-error}），而截断误差由 \(\Phi\) 的快速衰减控制。
\par
进一步，设 \(t_0\in\RR\) 满足 \(\Xi_h^{[K]}(t_0)=0\)，并取 \(r>0\) 使得
\[
m_0:=\inf_{\abs{t-t_0}\le r}\abs{\Xi'(t)}>0.
\]
若
\[
\delta:=\sup_{\abs{t-t_0}\le r}\abs{\Xi(t)-\Xi_h^{[K]}(t)}<m_0r,
\]
则 \(\Xi\) 在区间 \((t_0-r,t_0+r)\) 内存在且仅存在一个实零点 \(\gamma\)，并且满足可核验包含
\begin{equation}\label{eq:xi-verified-root-inclusion-interval}
\boxed{\;
\gamma\in\bigl[t_0-\delta/m_0,\ t_0+\delta/m_0\bigr]\;}
\end{equation}
该包含区间长度在 dyadic 细化 \(h\mapsto h/2\) 与同步增大截断 \(K\mapsto 2K\) 下按 \(\exp(-\pi^2/(2h))\) 的主尺度收缩（当截断误差被压到同阶或更低时）。
\end{theorem}
\begin{proof}
由三角不等式与 \eqref{eq:xi-poisson-aliasing-error-closed-form} 得
\[
\sup_{\abs{t}\le T}\abs{\Xi(t)-\Xi_h^{[K]}(t)}
\le
\sup_{\abs{t}\le T}\abs{\Xi(t)-\Xi_h(t)}
\;+\;
\sup_{\abs{t}\le T}\abs{\Xi_h(t)-\Xi_h^{[K]}(t)}.
\]
第一项由 \eqref{eq:xi-poisson-aliasing-error-closed-form} 直接得到别名尾和；第二项由尾部截断
\(
\Xi_h(t)-\Xi_h^{[K]}(t)=h\sum_{\abs{k}>K}\Phi(kh)\cos(tkh)
\)
并用 \(\abs{\cos(\cdot)}\le 1\) 估计得 \(\le 2h\sum_{k>K}\abs{\Phi(kh)}\)，得到 \eqref{eq:xi-truncation-error-separation}。
\par
在包含区间部分，由 \(\Xi_h^{[K]}(t_0)=0\) 得 \(\abs{\Xi(t_0)}\le \delta\)。
对任意 \(\abs{t-t_0}\le r\)，由均值定理有 \(\abs{\Xi(t)-\Xi(t_0)}\ge m_0\abs{t-t_0}\)；
若 \(\abs{t-t_0}>\delta/m_0\)，则 \(\abs{\Xi(t)}\ge m_0\abs{t-t_0}-\abs{\Xi(t_0)}>\delta-\delta=0\)，从而零点只能落在区间 \eqref{eq:xi-verified-root-inclusion-interval} 内。
唯一性由 \(m_0>0\)（即 \(\Xi'\) 在该区间无零）推出。证毕。
\end{proof}
