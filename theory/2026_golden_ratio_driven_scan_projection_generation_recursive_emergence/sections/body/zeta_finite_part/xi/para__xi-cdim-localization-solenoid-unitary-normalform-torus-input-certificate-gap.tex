\paragraph{单角色连续可见性、环面输入阻断与联合有限证书的不完备性}
在有限维环面值观测的单相位因子化与受限分母支撑角色族的精确商核已经建立之后，还可进一步把有限维连续可见性压缩为一组更强的终端结论：所有有限维连续酉表示都只是单一连续角色的整数幂直和，而非空局部化 solenoid 不接受任何非平凡环面输入；与截断 Jensen 剖面合并后，任何有限证书都仍然保留无限纤维。

\begin{theorem}[有限维连续酉表示的单角色整幂正规形]\label{thm:xi-cdim-localization-solenoid-finite-dimensional-unitary-single-character-normal-form}
\leanverified{paper\_xi\_cdim\_localization\_solenoid\_finite\_dimensional\_unitary\_single\_character\_normal\_form}
设 \(S\subset\mathbb P\) 为有限素数集，并记
\[
G_S:=\ZZ[S^{-1}],
\qquad
\Sigma_S:=\widehat{G_S}.
\]
任取连续有限维酉表示
\[
\rho:\Sigma_S\longrightarrow \mathrm{U}(d).
\]
则存在连续角色
\[
\chi:\Sigma_S\to \TT
\]
与整数 \(n_1,\dots,n_d\in \ZZ\)，使得在某个酉基下
\[
\boxed{\;
\rho(\xi)
=
\operatorname{diag}\bigl(\chi(\xi)^{n_1},\dots,\chi(\xi)^{n_d}\bigr)
\qquad(\forall \xi\in \Sigma_S).\;}
\]
特别地，\(\rho(\Sigma_S)\) 的连通 Lie 部分至多是一条圆周，不存在 Lie 秩至少为 \(2\) 的连续酉像。
\end{theorem}
\begin{proof}
由推论 \ref{cor:xi-cdim-localization-solenoid-finite-dimensional-unitary-representations-from-circle}，存在连续满射
\[
q:\Sigma_S\twoheadrightarrow \TT
\]
与连续酉表示
\[
\beta:\TT\to \mathrm{U}(d)
\]
使
\[
\rho=\beta\circ q.
\]
把 \(\chi:=q\) 记作对应的连续角色。

由于 \(\TT\) 是紧交换群，\(\beta(\TT)\) 由一族彼此交换的酉矩阵组成，因此可同时酉对角化。故存在某个酉基与整数 \(n_1,\dots,n_d\in\ZZ\)，使对一切 \(z\in\TT\) 都有
\[
\beta(z)=\operatorname{diag}(z^{n_1},\dots,z^{n_d}).
\]
代回 \(z=\chi(\xi)\) 即得
\[
\rho(\xi)
=
\operatorname{diag}\bigl(\chi(\xi)^{n_1},\dots,\chi(\xi)^{n_d}\bigr).
\]
最后，\(\rho(\Sigma_S)\) 的连通部分包含于对角环面中由单参数 \(z\mapsto \operatorname{diag}(z^{n_1},\dots,z^{n_d})\) 所生成的闭子群，故其 Lie 秩至多为 \(1\)。证毕。
\end{proof}

\begin{theorem}[非空局部化 solenoid 不接受任何非平凡环面输入]\label{thm:xi-cdim-localization-solenoid-no-nontrivial-torus-input}
\leanverified{paper\_xi\_cdim\_localization\_solenoid\_no\_nontrivial\_torus\_input}
设 \(S\subset\mathbb P\) 为非空有限素数集，并记
\[
\Sigma_S:=\widehat{\ZZ[S^{-1}]}.
\]
则对任意 \(d\ge 1\)，都有
\[
\boxed{\;
\operatorname{Hom}_{\mathrm{cts}}(\TT^d,\Sigma_S)=0.\;}
\]
等价地，任意连续群同态
\[
\Phi:\TT^d\to \Sigma_S
\]
都必为零映射。
\end{theorem}
\begin{proof}
Pontryagin 对偶给出自然同构
\[
\operatorname{Hom}_{\mathrm{cts}}(\TT^d,\Sigma_S)
\cong
\operatorname{Hom}_{\ZZ}(G_S,\ZZ^d),
\qquad
G_S:=\ZZ[S^{-1}].
\]
因此只需证明任意加法同态
\[
\phi:G_S\to \ZZ^d
\]
都为零。

取任意 \(p\in S\)。由于 \(p^{-n}\in G_S\) 对每个 \(n\ge 1\) 都成立，故
\[
p^n\phi(p^{-n})=\phi(1)
\qquad(n\ge 1).
\]
于是
\[
\phi(1)\in \bigcap_{n\ge 1}p^n\ZZ^d=\{0\},
\]
从而 \(\phi(1)=0\)。

再取任意
\[
x=\frac{a}{b}\in G_S,
\qquad
a\in \ZZ,
\qquad
b\in \langle S\rangle.
\]
则
\[
b\,\phi(x)=\phi(a)=a\,\phi(1)=0.
\]
由于 \(\ZZ^d\) 无挠，只能有 \(\phi(x)=0\)。这对所有 \(x\in G_S\) 成立，故 \(\phi=0\)。证毕。
\end{proof}

\begin{theorem}[有限维连续可见性与有限半径 Jensen 证书的联合不完备性]\label{thm:xi-cdim-localization-solenoid-finite-visibility-jensen-certificate-incomplete}
设 \(S\subset\mathbb P\) 为非空有限素数集，固定 \(0<\rho_\star<1\)。任取有限族连续酉表示
\[
\rho_j:\Sigma_S\to \mathrm{U}(d_j)
\qquad
(1\le j\le r),
\]
并沿用定理 \ref{thm:xi-time-part60e-truncated-jensen-profile-completeness-blindness} 的记号，把
\[
\mathcal J_{\rho_\star}(F)
\]
记为单位圆盘上非零全纯函数 \(F\) 的截断 Jensen 剖面。定义联合有限证书
\[
\mathcal C_{\rho_\star}(\xi,F)
:=
\Bigl(
\rho_1(\xi),\dots,\rho_r(\xi),\mathcal J_{\rho_\star}(F)
\Bigr),
\qquad
\xi\in \Sigma_S.
\]
则对任意基对象对 \((\xi_0,F_0)\)，都存在一列两两不同的对象对
\[
\{(\xi_n,F_n)\}_{n\ge 1}
\]
满足
\[
\mathcal C_{\rho_\star}(\xi_n,F_n)
=
\mathcal C_{\rho_\star}(\xi_0,F_0)
\qquad(n\ge 1).
\]
特别地，每个非空纤维
\[
\mathcal C_{\rho_\star}^{-1}(c)
\]
都至少包含可数无限多个元素。
\end{theorem}
\begin{proof}
先处理 arithmetic 侧。令
\[
D:=\sum_{j=1}^r d_j,
\]
并把 \(\rho_1,\dots,\rho_r\) 组成块对角表示
\[
\rho:=\rho_1\oplus\cdots\oplus \rho_r:\Sigma_S\to \mathrm{U}(D).
\]
由定理 \ref{thm:xi-cdim-localization-solenoid-finite-dimensional-unitary-single-character-normal-form}，存在连续角色
\[
\chi:\Sigma_S\to \TT
\]
与整数 \(n_1,\dots,n_D\in\ZZ\)，使
\[
\rho(\xi)=\operatorname{diag}\bigl(\chi(\xi)^{n_1},\dots,\chi(\xi)^{n_D}\bigr).
\]

若全部 \(n_i=0\)，则 \(\rho\) 恒等于单位矩阵，因此所有 \(\rho_j\) 都是常值。另一方面，\(\Sigma_S\) 由短正合列
\[
0\to \prod_{p\in S}\ZZ_p\to \Sigma_S\to \TT\to 0
\]
映到圆周，因而是无限紧群。于是可直接取一列两两不同的 \(\xi_n\in \Sigma_S\)，并有
\[
\rho_j(\xi_n)=\rho_j(\xi_0)
\qquad
(\forall j,n).
\]

下设并非全部 \(n_i\) 为零。记
\[
g:=\gcd\{|n_i|:\ n_i\neq 0\}\ge 1.
\]
则
\[
\ker(\rho)=\ker(\chi^g).
\]
设 \(\chi=\chi_h\) 对应于某个非零 \(h\in G_S\)。则 \(\chi^g=\chi_{gh}\) 对偶于单射
\[
\mu_{gh}:\ZZ\to G_S,
\qquad
k\mapsto kgh.
\]
由于 \(S\neq \varnothing\)，群 \(G_S=\ZZ[S^{-1}]\) 不是有限生成阿贝尔群：任取 \(p\in S\)，任何有限生成子群都包含于某个 \(N^{-1}\ZZ\)，而 \(p^{-m}\notin N^{-1}\ZZ\) 对 \(m\) 充分大时成立。故商群
\[
G_S/\operatorname{im}(\mu_{gh})
\]
不可能有限；否则 \(G_S\) 将成为循环群与有限群的扩张，从而有限生成，矛盾。对偶化得到
\[
\ker(\chi^g)\cong \widehat{G_S/\operatorname{im}(\mu_{gh})},
\]
因而 \(\ker(\rho)\) 是无限紧群。于是可取一列两两不同的
\[
\eta_n\in \ker(\rho),
\qquad
\xi_n:=\xi_0+\eta_n.
\]
对所有 \(j,n\) 都有
\[
\rho_j(\xi_n)=\rho_j(\xi_0).
\]

再处理解析侧。取一列两两不同的实数
\[
a_n\in (\rho_\star,1)
\qquad(n\ge 1),
\]
并定义有限 Blaschke 因子
\[
B_{a_n}(z):=\frac{z-a_n}{1-a_n z},
\qquad
F_n:=F_0\,B_{a_n}.
\]
每个 \(B_{a_n}\) 的新增零点都满足 \(\rho_\star<|a_n|<1\)。由定理 \ref{thm:xi-time-part60e-truncated-jensen-profile-completeness-blindness} 第 \(3\) 条，
\[
\mathcal J_{\rho_\star}(F_n)=\mathcal J_{\rho_\star}(F_0)
\qquad(n\ge 1).
\]
又因 \(a_n\) 两两不同，\(F_n\) 两两不同。

综上，对每个 \(n\ge 1\) 都有
\[
\mathcal C_{\rho_\star}(\xi_n,F_n)
=
\mathcal C_{\rho_\star}(\xi_0,F_0),
\]
且对象对 \((\xi_n,F_n)\) 两两不同。于是每个非空纤维都至少包含一列可数无限的不同对象。证毕。
\end{proof}
