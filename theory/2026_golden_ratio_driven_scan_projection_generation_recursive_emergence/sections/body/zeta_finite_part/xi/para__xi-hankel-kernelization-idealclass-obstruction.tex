\begin{proposition}[整性在 \texorpdfstring{$\GL_d(\ZZ)$}{GLd(Z)} 共轭下保持]\label{prop:xi-hankel-integrality-preserved-under-unimodular-conjugacy}
\leanverified{paper\_xi\_hankel\_integrality\_preserved\_under\_unimodular\_conjugacy}
令 \(A\in M_d(\QQ)\) 且 \(P\in \GL_d(\ZZ)\)。
若 \(P^{-1}AP\in M_d(\ZZ)\)，则 \(A\in M_d(\ZZ)\)。
\end{proposition}
\begin{proof}
由 \(P\in\GL_d(\ZZ)\) 得 \(P^{-1}\in M_d(\ZZ)\)。
若 \(B:=P^{-1}AP\in M_d(\ZZ)\)，则 \(A=PBP^{-1}\) 为整数矩阵之积，故 \(A\in M_d(\ZZ)\)。证毕。
\end{proof}

\begin{theorem}[平方因子自由特征多项式的格参数化与非负实现判据]\label{thm:xi-hankel-nonnegative-kernelization-idealclass-obstruction}
取首一多项式 \(f(x)\in\ZZ[x]\) 满足 \(\deg f=d\) 且在 \(\QQ[x]\) 上不可约并可分（等价于判别式非零）。
令
\[
K:=\QQ[x]/(f(x)),
\qquad
\mathcal O_f:=\ZZ[x]/(f(x))\subset K,
\qquad
\bar x\in \mathcal O_f
\]
为 \(x\) 的像。
记 \(\mathscr L(\mathcal O_f)\) 为所有满足 \(I\otimes_{\ZZ}\QQ=K\) 的有限生成 \(\mathcal O_f\)-子模 \(I\subset K\) 的集合（称为 full \(\mathcal O_f\)-格），并在其上取等价关系
\[
I\sim J\quad:\Longleftrightarrow\quad \exists\,\alpha\in K^\times,\ J=\alpha I.
\]
则存在自然双射
\[
\{A\in M_d(\ZZ):\ \chi_A=f\}/\sim_{\GL_d(\ZZ)}
\ \stackrel{\sim}{\longrightarrow}\
\mathscr L(\mathcal O_f)/\sim,
\]
其将 \(\GL_d(\ZZ)\)-共轭类 \([A]\) 送到由 \(\bar x\) 在 \(\ZZ^d\) 上诱导的 \(\mathcal O_f\)-结构所对应的 full \(\mathcal O_f\)-格类 \([I]\)。
\par
并且在上述对应下，以下命题等价：
\begin{enumerate}
  \item 存在 \(P\in \GL_d(\ZZ)\) 使 \(P^{-1}AP\in M_d(\ZZ_{\ge 0})\)；
  \item 在对应的格类中存在某个代表 \(I\in\mathscr L(\mathcal O_f)\) 与某组 \(\ZZ\)-基 \(\mathcal B\) 使得
  \[
  [m_{\bar x}]_{\mathcal B}\in M_d(\ZZ_{\ge 0}),
  \]
  其中 \(m_{\bar x}:I\to I\) 为乘以 \(\bar x\) 的 \(\ZZ\)-线性自同态。
\end{enumerate}
\end{theorem}
\begin{proof}
由于 \(f\) 在 \(\QQ[x]\) 上不可约且 \(\chi_A=f\)，最小多项式亦为 \(f\)，故 \(\QQ^d\) 作为 \(\QQ[x]\)-模（令 \(x\) 作用为 \(A\)）为一个 \(d\) 维简单模，从而成为一维 \(K\)-向量空间。
选取一个 \(K\)-线性同构
\[
\iota:K\stackrel{\sim}{\longrightarrow}\QQ^d.
\]
令 \(I:=\iota^{-1}(\ZZ^d)\subset K\)。
则 \(I\) 为 \(\ZZ\)-秩 \(d\) 的格，且 \(\ZZ^d\) 在 \(A\) 作用下不变等价于 \(I\) 在 \(\bar x\) 作用下不变，故 \(\mathcal O_f I\subset I\)。
又因 \(\iota\) 为同构，\(I\otimes_{\ZZ}\QQ=K\)，故 \(I\in\mathscr L(\mathcal O_f)\)。
若更换 \(\iota\) 为 \(\iota\circ m_\alpha\)（\(\alpha\in K^\times\)），则得到的格变为 \(\alpha^{-1}I\)，故对应于同一等价类 \([I]\in\mathscr L(\mathcal O_f)/\sim\)。
\par
反向构造：给定 \(I\in\mathscr L(\mathcal O_f)\) 与一组 \(\ZZ\)-基 \(\mathcal B\)，由 \(\mathcal O_f I\subset I\) 得 \(m_{\bar x}(I)\subset I\)，从而 \(m_{\bar x}:I\to I\) 的矩阵
\(
A:=[m_{\bar x}]_{\mathcal B}
\)
属于 \(M_d(\ZZ)\)。
该 \(\QQ\)-线性算子在 \(K\otimes_{\ZZ}\QQ\cong K\) 上与乘以 \(\bar x\) 相同，故其特征多项式为 \(f\)，即 \(\chi_A=f\)。
更换 \(\ZZ\)-基即为 \(\GL_d(\ZZ)\)-共轭。
若以 \(\alpha I\) 代替 \(I\) 并用基 \(\alpha\mathcal B\)，则矩阵不变，因此仅依赖于格等价类 \([I]\)。
由上述两方向互为逆构造得到双射。
\par
最后，(1) 断言存在某组 \(\ZZ\)-基使 \(\bar x\) 的矩阵条目全为非负整数；
而在格模型中这恰等价于存在某个代表格 \(I\) 的某组 \(\ZZ\)-基 \(\mathcal B\) 满足 \([m_{\bar x}]_{\mathcal B}\in M_d(\ZZ_{\ge 0})\)，即 (2)。
证毕。
\end{proof}

\begin{corollary}[可逆理想子类与 \texorpdfstring{$\mathrm{Pic}(\mathcal O_f)$}{Pic(Of)}]\label{cor:xi-hankel-invertible-ideal-subclass-pic}
\leanverified{paper\_xi\_hankel\_invertible\_ideal\_subclass\_pic}
令 \(\mathscr L^\times(\mathcal O_f)\subset \mathscr L(\mathcal O_f)\) 为全体可逆分式理想（即在 \(K\) 中可逆的 \(\mathcal O_f\)-子模）。
则按缩放等价 \(\sim\) 的商满足自然同构
\[
\mathscr L^\times(\mathcal O_f)/\sim\ \cong\ \mathrm{Pic}(\mathcal O_f).
\]
因此，若定理 \ref{thm:xi-hankel-nonnegative-kernelization-idealclass-obstruction} 所得格类 \([I]\) 中含可逆理想代表，则该类可由 \(\mathrm{Pic}(\mathcal O_f)\) 内禀记录；并且“非负整数共轭可实现性”在该子类上等价于在相应 Picard 类中寻找使 \([m_{\bar x}]_{\mathcal B}\) 条目非负的 \(\ZZ\)-基。
\end{corollary}
\begin{proof}
可逆分式理想在乘法下构成交换群，按主理想 \(K^\times\) 的作用取商即得 \(\mathrm{Pic}(\mathcal O_f)\)。
而 \(\sim\) 恰为对 \(\alpha\in K^\times\) 的缩放 \(I\mapsto \alpha I\)。
故 \(\mathscr L^\times(\mathcal O_f)/\sim\cong \mathrm{Pic}(\mathcal O_f)\)。证毕。
\end{proof}
