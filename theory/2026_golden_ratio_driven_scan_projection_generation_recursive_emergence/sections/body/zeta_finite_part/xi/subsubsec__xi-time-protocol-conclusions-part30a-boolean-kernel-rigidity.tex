\subsubsection{Boolean 互斥核的行列式刚性、逆核闭式与整数结构}
固定整数 \(q\ge 2\)，记
\[
U_q:=\{0,1\}^q,\qquad N:=2^q.
\]
定义对称核矩阵 \(T^{(q)}\in\RR^{U_q\times U_q}\)：
\[
T^{(q)}_{u,v}=
\begin{cases}
1,&u\wedge v=\mathbf 0,\\[2pt]
\frac12,&u\wedge v\neq \mathbf 0,
\end{cases}
\qquad (u,v\in U_q),
\]
其中 \(\wedge\) 与 \(\vee\) 分别表示逐位与、逐位或，\(\mathbf 0,\mathbf 1\) 分别为全零与全一向量。

为避免记号歧义，可通过支撑同构将 \(U_q\) 与幂集 \(V_q:=2^{[q]}\) 识别：对 \(u\in U_q\) 令
\(
\mathrm{supp}(u):=\{i\in[q]:u_i=1\}=:U
\)。
则 \(u\wedge v=\mathbf 0\) 等价于 \(U\cap V=\varnothing\)。
在该识别下，记
\[
A_q:=2T^{(q)}\in\Mat_{2^q}(\ZZ),
\qquad
(A_q)_{U,V}=
\begin{cases}
2,&U\cap V=\varnothing,\\
1,&U\cap V\neq \varnothing,
\end{cases}
\qquad (U,V\subseteq[q]).
\]

\begin{proposition}\label{prop:xi-disjointness-boolean-zeta-ldlt}
取任意与包含关系相容的线性扩张对 \(V_q=2^{[q]}\) 排序。
令 \(Z_q\) 为 Boolean 格 \((V_q,\subseteq)\) 的 \(\zeta\)-矩阵：
\[
(Z_q)_{A,B}:=\ind_{\{A\subseteq B\}}\qquad (A,B\subseteq[q]),
\]
并令对角矩阵 \(\Delta_q\) 由
\[
(\Delta_q)_{\varnothing,\varnothing}:=2,
\qquad
(\Delta_q)_{S,S}:=(-1)^{\card{S}}\quad(S\neq\varnothing)
\]
给定。
则成立严格因子分解
\[
A_q=Z_q^{\mathsf T}\Delta_q Z_q.
\]
等价地，令 \(\mu_q:=Z_q^{-1}\) 为 Möbius 矩阵（即
\(
(\mu_q)_{A,B}=(-1)^{\card{B\setminus A}}\ind_{\{A\subseteq B\}}
\)），则有整同余对角化
\[
\mu_q^{\mathsf T}A_q\mu_q=\Delta_q.
\]
\leanverified{paper\_xi\_disjointness\_boolean\_zeta\_ldlt}
\end{proposition}
\begin{proof}
由定义，
\[
\bigl(Z_q^{\mathsf T}\Delta_q Z_q\bigr)_{U,V}
=\sum_{S\subseteq[q]}(Z_q)_{S,U}(\Delta_q)_{S,S}(Z_q)_{S,V}
=\sum_{S\subseteq U\cap V}(\Delta_q)_{S,S}.
\]
若 \(U\cap V=\varnothing\)，则仅 \(S=\varnothing\) 贡献，取值 \(2\)。
若 \(U\cap V\neq\varnothing\)，则
\[
\sum_{S\subseteq U\cap V}(\Delta_q)_{S,S}
=2+\sum_{\varnothing\neq S\subseteq U\cap V}(-1)^{\card{S}}
=2+\bigl((1-1)^{\card{U\cap V}}-1\bigr)=1.
\]
故逐项有 \(\bigl(Z_q^{\mathsf T}\Delta_q Z_q\bigr)_{U,V}=(A_q)_{U,V}\)，从而分解成立。
\end{proof}

\begin{corollary}\label{cor:xi-disjointness-boolean-order-ideal-principal-minors}
\leanverified{paper\_xi\_disjointness\_boolean\_order\_ideal\_principal\_minors}
设 \(\mathcal I\subseteq V_q\) 为任意下闭集（order ideal），即 \(S\in\mathcal I\) 且 \(R\subseteq S\Rightarrow R\in\mathcal I\)。
记 \(A_q[\mathcal I]\) 为按 \(\mathcal I\) 取行列得到的主子矩阵。
则
\[
\det A_q[\mathcal I]
=\prod_{S\in\mathcal I}(\Delta_q)_{S,S}
=2^{\ind_{\{\varnothing\in\mathcal I\}}}\cdot (-1)^{\#\{S\in\mathcal I\setminus\{\varnothing\}:\ \card{S}\ \mathrm{odd}\}}.
\]
从而对任意下闭集 \(\mathcal I\)，主子矩阵 \(A_q[\mathcal I]\) 必为可逆整矩阵，且 \(\abs{\det A_q[\mathcal I]}\in\{1,2\}\)。
\end{corollary}
\begin{proof}
在与包含关系相容的排序下，\(Z_q\) 为单位上三角整矩阵，故 \(\det(Z_q)=1\)。
对下闭集 \(\mathcal I\)，若 \(B\in\mathcal I\) 则任意 \(A\subseteq B\) 亦在 \(\mathcal I\)，因此 \(Z_q\) 的 \(\mathcal I\)-列在 \(\mathcal I\) 之外无非零项，遂
\[
A_q[\mathcal I]=Z_{\mathcal I}^{\mathsf T}\Delta_{\mathcal I}Z_{\mathcal I},
\qquad
Z_{\mathcal I}:=Z_q[\mathcal I].
\]
其中 \(Z_{\mathcal I}\) 仍为单位三角整矩阵，\(\det(Z_{\mathcal I})=1\)，故
\(\det(A_q[\mathcal I])=\det(\Delta_{\mathcal I})\)
即为所示乘积公式。
\end{proof}

不交指示矩阵 \(\bigl(\ind_{\{U\cap V=\varnothing\}}\bigr)_{U,V\subseteq[q]}\) 及其块递推常作为 Kneser--Sierpiński 矩阵出现；相关背景与递推表述可参见 Jukna 的专著笔记（例如对不交矩阵与 Möbius 反演结构的讨论）\cite{Jukna2011ExtremalCombinatorics}。
\end{remark}

\begin{conclusion}[行列式刚性：\texorpdfstring{$\det(2T^{(q)})=2$}{det(2T)=2}]\label{con:xi-terminal-disjointness-boolean-det-rigidity}
设
\[
K:=\begin{pmatrix}1&1\\[2pt]1&0\end{pmatrix},
\qquad
C:=\begin{pmatrix}1&1\\[2pt]1&1\end{pmatrix},
\qquad
D_q:=K^{\otimes q},
\qquad
J_q:=C^{\otimes q}=\mathbf 1\mathbf 1^{\mathsf T}.
\]
则有
\[
2T^{(q)}=D_q+J_q,
\]
并且
\[
\boxed{\;\det\!\bigl(2T^{(q)}\bigr)=2,\qquad
\det\!\bigl(T^{(q)}\bigr)=2^{-(2^q-1)}\; }.
\]
\end{conclusion}
\begin{proof}
由张量积逐坐标分解，
\(
(D_q)_{u,v}=\mathbf 1_{\{u\wedge v=\mathbf 0\}}
\)，故 \(2T^{(q)}=D_q+J_q\)。
又 \(\det(K)=-1\)，由 Kronecker 行列式公式
\[
\det(D_q)=\det(K)^{q2^{q-1}}=(-1)^{q2^{q-1}}=1
\qquad(q\ge 2).
\]
记
\[
K^{-1}=\begin{pmatrix}0&1\\[2pt]1&-1\end{pmatrix},
\qquad
D_q^{-1}=(K^{-1})^{\otimes q}.
\]
并有
\[
\mathbf 1^{\mathsf T}D_q^{-1}\mathbf 1
=\bigl(\mathbf 1_2^{\mathsf T}K^{-1}\mathbf 1_2\bigr)^q
=1^q=1.
\]
由矩阵行列式引理
\[
\det(D_q+\mathbf 1\mathbf 1^{\mathsf T})
=\det(D_q)\Bigl(1+\mathbf 1^{\mathsf T}D_q^{-1}\mathbf 1\Bigr)
=2.
\]
即 \(\det(2T^{(q)})=2\)。
再由 \(\det(2T^{(q)})=2^{N}\det(T^{(q)})\) 得
\(
\det(T^{(q)})=2^{-(N-1)}
\)。证毕。
\end{proof}

\begin{conclusion}[逆核的矩阵级闭式与支撑稀疏结构]\label{con:xi-terminal-disjointness-boolean-inverse-closed-form}
记
\[
E_{00}:=\begin{pmatrix}1&0\\[2pt]0&0\end{pmatrix}.
\]
则有矩阵级恒等式
\[
\boxed{\;
\bigl(T^{(q)}\bigr)^{-1}
=2\,(K^{-1})^{\otimes q}-E_{00}^{\otimes q}\; }.
\]
等价地，对任意 \(u,v\in U_q\)：
\[
\boxed{\;
\bigl(T^{(q)}\bigr)^{-1}_{u,v}
=
\begin{cases}
-1,&(u,v)=(\mathbf 0,\mathbf 0),\\[2pt]
2(-1)^{|u\wedge v|},&u\vee v=\mathbf 1,\ (u,v)\neq(\mathbf 0,\mathbf 0),\\[2pt]
0,&\text{其余情形},
\end{cases}\;}
\]
并且
\[
\#\{(u,v):\ (T^{(q)})^{-1}_{u,v}\neq 0\}=3^q+1,
\]
\[
\#\{(u,v):\ (T^{(q)})^{-1}_{u,v}=2\}=\frac{3^q+1}{2},
\qquad
\#\{(u,v):\ (T^{(q)})^{-1}_{u,v}=-2\}=\frac{3^q-1}{2}.
\]
并且对任意 \(u\in U_q\)，若记 \(|u|=r\)，则第 \(u\)-行（等价地第 \(u\)-列）的非零元个数满足
\[
\#\{v\in U_q:\ (T^{(q)})^{-1}_{u,v}\neq 0\}
=
\begin{cases}
2,&u=\mathbf 0,\\[2pt]
2^r,&u\neq \mathbf 0.
\end{cases}
\]
此外，
\[
\sum_{u,v\in U_q}\bigl|(T^{(q)})^{-1}_{u,v}\bigr|=2\cdot 3^q+1,
\qquad
\bigl\|(T^{(q)})^{-1}\bigr\|_F^2=4\cdot 3^q+1.
\]
此外，
\[
\operatorname{adj}\!\bigl(2T^{(q)}\bigr)=\bigl(T^{(q)}\bigr)^{-1}\in\Mat_{2^q}(\ZZ).
\]
\end{conclusion}
\begin{proof}
由 Sherman--Morrison 公式与上一结论中的
\(
\mathbf 1^{\mathsf T}D_q^{-1}\mathbf 1=1
\)，
\[
(D_q+\mathbf 1\mathbf 1^{\mathsf T})^{-1}
=D_q^{-1}-\frac12D_q^{-1}\mathbf 1\mathbf 1^{\mathsf T}D_q^{-1}.
\]
又
\[
D_q^{-1}\mathbf 1
=(K^{-1}\mathbf 1_2)^{\otimes q}
=e_0^{\otimes q},
\]
故
\[
\bigl(T^{(q)}\bigr)^{-1}
=2(D_q+\mathbf 1\mathbf 1^{\mathsf T})^{-1}
=2D_q^{-1}-e_0^{\otimes q}(e_0^{\otimes q})^{\mathsf T}
=2(K^{-1})^{\otimes q}-E_{00}^{\otimes q}.
\]
其中 \((K^{-1})_{00}=0\)、\((K^{-1})_{01}=(K^{-1})_{10}=1\)、\((K^{-1})_{11}=-1\)，
故 \((K^{-1})^{\otimes q}_{u,v}\neq 0\iff u\vee v=\mathbf 1\)，且符号为 \((-1)^{|u\wedge v|}\)；
再扣除 \(E_{00}^{\otimes q}\) 即得逐项分段式。
\par
计数方面，约束 \(u\vee v=\mathbf 1\) 每坐标有三种局部型 \((0,1),(1,0),(1,1)\)，故总数为 \(3^q\)；再加上 \((\mathbf 0,\mathbf 0)\) 给出 \(3^q+1\) 个非零项。
若 \(k\) 为 \((1,1)\) 坐标数，则对应有序对数为 \(\binom{q}{k}2^{q-k}\)；
偶 \(k\) 与奇 \(k\) 求和分别为
\[
\frac{(2+1)^q+(2-1)^q}{2}=\frac{3^q+1}{2},
\qquad
\frac{(2+1)^q-(2-1)^q}{2}=\frac{3^q-1}{2},
\]
分别对应 \(2\) 与 \(-2\)。
固定 \(u\in U_q\)，记 \(|u|=r\)。
当 \(u\neq \mathbf 0\) 时，条件 \(u\vee v=\mathbf 1\) 等价于：在 \(u_i=0\) 的坐标上 \(v_i=1\) 必取，在 \(u_i=1\) 的坐标上 \(v_i\in\{0,1\}\) 自由，故对应 \(v\) 的个数为 \(2^r\)。
当 \(u=\mathbf 0\) 时，满足 \(u\vee v=\mathbf 1\) 的 \(v\) 唯一为 \(\mathbf 1\)，并叠加 \((\mathbf 0,\mathbf 0)\) 项，故该行恰有 \(2\) 个非零元；由对称性列计数相同。
\par
绝对值和与 Frobenius 范数平方由同一支撑计数直接得到：除 \((\mathbf 0,\mathbf 0)\) 外共有 \(3^q\) 个非零项且绝对值皆为 \(2\)，从而
\[
\sum_{u,v}\bigl|(T^{(q)})^{-1}_{u,v}\bigr|=1+2\cdot 3^q,
\qquad
\sum_{u,v}\bigl((T^{(q)})^{-1}_{u,v}\bigr)^2=1+4\cdot 3^q.
\]
最后由上一结论 \(\det(2T^{(q)})=2\) 得
\(
\operatorname{adj}(2T^{(q)})=2(2T^{(q)})^{-1}=(T^{(q)})^{-1}
\)，整性随即成立。证毕。
\end{proof}

\begin{conclusion}[Smith 形、余核与整可解性判别]\label{con:xi-terminal-disjointness-boolean-smith-mod2-rank-kernel}
沿用 \(A_q:=2T^{(q)}\in\Mat_{2^q}(\ZZ)\) 的记号。
则
\[
\mathrm{SNF}\!\bigl(A_q\bigr)=\operatorname{diag}(\underbrace{1,\dots,1}_{2^q-1\text{ 次}},2),
\]
从而
\[
\operatorname{coker}(A_q)\cong \ZZ/2\ZZ.
\]
进一步，对任意 \(b\in\ZZ^{U_q}\)，方程
\[
A_qx=b
\]
在 \(\ZZ^{U_q}\) 中可解当且仅当
\[
b_{\mathbf 0}\equiv 0\pmod 2,
\]
其中 \(b_{\mathbf 0}\) 表示 \(\mathbf 0\in U_q\) 对应坐标。
并且在 \(\FF_2\) 上有
\[
\operatorname{rank}_{\FF_2}\!\bigl(A_q\bmod 2\bigr)=2^q-1,
\qquad
\ker_{\FF_2}\!\bigl(A_q\bmod 2\bigr)=\langle \delta_{\mathbf 0}\rangle.
\]
\end{conclusion}
\begin{proof}
令 \(V_q=2^{[q]}\) 并按包含关系相容的线性扩张对 \(V_q\) 排序。
由命题 \ref{prop:xi-disjointness-boolean-zeta-ldlt}，记 \(\mu_q:=Z_q^{-1}\)，则
\[
\mu_q^{\mathsf T}A_q\mu_q=\Delta_q.
\]
由于 \(\mu_q^{\mathsf T},\mu_q\in\mathrm{GL}_{2^q}(\ZZ)\)，Smith 正规形在左右乘幺模矩阵下不变，故
\(
\mathrm{SNF}(A_q)=\mathrm{SNF}(\Delta_q)
\)。
而 \(\Delta_q\) 为对角矩阵，其对角元为 \(2\) 及若干 \(\pm1\)；对角上的 \(-1\) 可由行/列乘以 \(-1\) 消去，得到与 \(\mathrm{diag}(2,1,\dots,1)\) 等价之对角形，
从而
\[
\mathrm{SNF}\!\bigl(A_q\bigr)=\operatorname{diag}(\underbrace{1,\dots,1}_{2^q-1\text{ 次}},2),
\qquad
\operatorname{coker}(A_q)\cong\ZZ/2\ZZ.
\]
\par
对整数可解性，写 \(A_q=Z_q^{\mathsf T}\Delta_q Z_q\) 并令 \(y:=Z_qx\in\ZZ^{V_q}\)。
左乘 \((Z_q^{\mathsf T})^{-1}=\mu_q^{\mathsf T}\) 得同解等价变形
\[
\Delta_q\,y=\mu_q^{\mathsf T}b.
\]
由于 \(\Delta_q\) 仅在 \(\varnothing\) 分量的对角元为 \(2\)，其余对角元均为 \(\pm1\)，该方程存在整数解 \(y\) 当且仅当 \((\mu_q^{\mathsf T}b)_{\varnothing}\) 为偶数。
又 Möbius 矩阵满足 \((\mu_q)_{A,\varnothing}=\ind_{\{A=\varnothing\}}\)，从而
\(
(\mu_q^{\mathsf T}b)_{\varnothing}=b_{\varnothing}=b_{\mathbf 0}
\)，
因此 \(A_qx=b\) 在 \(\ZZ^{U_q}\) 中可解当且仅当 \(b_{\mathbf 0}\equiv 0\pmod 2\)。
\par
在 \(\FF_2\) 上，Smith 形给出 \(\operatorname{rank}_{\FF_2}(A_q\bmod 2)=2^q-1\) 且核为一维。
又 \(A_q\) 的 \(\mathbf 0\)-列全为 \(2\)，模 \(2\) 后成为零列，故 \(\delta_{\mathbf 0}\in\ker\)；
由一维性得 \(\ker=\langle \delta_{\mathbf 0}\rangle\)。
证毕。
\end{proof}

\begin{conclusion}[全谱倒一阶与倒二阶矩闭式]\label{con:xi-terminal-disjointness-boolean-inverse-spectral-moments}
记 \(T^{(q)}\) 的全体特征值（按代数重数）为
\[
\{\theta_{\alpha}^{(q)}\}_{\alpha=1}^{2^q}.
\]
则对任意 \(q\ge 1\) 有
\[
\sum_{\alpha=1}^{2^q}\frac{1}{\theta_{\alpha}^{(q)}}
=\operatorname{tr}\!\bigl((T^{(q)})^{-1}\bigr)
=2(-1)^q-1,
\]
\[
\sum_{\alpha=1}^{2^q}\frac{1}{(\theta_{\alpha}^{(q)})^{2}}
=\operatorname{tr}\!\bigl((T^{(q)})^{-2}\bigr)
=4\cdot 3^q+1.
\]
\end{conclusion}
\begin{proof}
由谱映射与迹恒等式，
\[
\sum_{\alpha}\theta_{\alpha}^{-1}=\operatorname{tr}(T^{-1}),
\qquad
\sum_{\alpha}\theta_{\alpha}^{-2}=\operatorname{tr}(T^{-2}).
\]
第一式由结论 \ref{con:xi-terminal-disjointness-boolean-inverse-closed-form} 的逐项逆核直接得到：\((T^{(q)})^{-1}_{\mathbf 0,\mathbf 0}=-1\)、\((T^{(q)})^{-1}_{\mathbf 1,\mathbf 1}=2(-1)^q\)，其余对角元皆为 \(0\)，故
\[
\operatorname{tr}\!\bigl((T^{(q)})^{-1}\bigr)=2(-1)^q-1.
\]
第二式由同一结论中的范数闭式
\[
\bigl\|(T^{(q)})^{-1}\bigr\|_F^2=4\cdot 3^q+1
\]
及 \(T^{(q)}\) 对称性给出的
\[
\operatorname{tr}\!\bigl((T^{(q)})^{-2}\bigr)=\bigl\|(T^{(q)})^{-1}\bigr\|_F^2
\]
立即推出。证毕。
\end{proof}

\begin{conclusion}[\texorpdfstring{$(T^{(q)})^{-1}$}{T^{-1}} 的 Smith 标准形]\label{con:xi-terminal-disjointness-boolean-inverse-smith}
令 \(N=2^q\)。
则整数矩阵 \((T^{(q)})^{-1}\in\Mat_{N}(\ZZ)\) 的 Smith 标准形为
\[
\mathrm{SNF}\!\bigl((T^{(q)})^{-1}\bigr)=\mathrm{diag}(1,\underbrace{2,\dots,2}_{N-1\text{ 次}}).
\]
\end{conclusion}
\begin{proof}
由结论 \ref{con:xi-terminal-disjointness-boolean-smith-mod2-rank-kernel}，
存在幺模矩阵 \(U,V\in\mathrm{GL}_{N}(\ZZ)\) 使得
\[
UA_qV=\mathrm{diag}(\underbrace{1,\dots,1}_{N-1\text{ 次}},2)=:\Delta.
\]
取逆并乘以行列式（\(\det(A_q)=2=\det(\Delta)\)），得到
\[
(T^{(q)})^{-1}=\operatorname{adj}(A_q)=\det(A_q)A_q^{-1}
=V\,\det(\Delta)\Delta^{-1}\,U
=V\,\mathrm{diag}(\underbrace{2,\dots,2}_{N-1\text{ 次}},1)\,U.
\]
右端与 \(\mathrm{diag}(2,\dots,2,1)\) 在 \(\ZZ\) 上左右幺模等价，从而其 Smith 标准形为
\(\mathrm{diag}(1,2,\dots,2)\)。
证毕。
\end{proof}

\begin{conclusion}[例外谱负一次幂和的显式闭式与递推]\label{con:xi-terminal-replica-softcore-exceptional-neg1-closed-recurrence}
令 \(\{\lambda_i^{(q)}\}_{i=0}^{q}\) 为例外谱（即 \(\mathrm{spec}(A^{(q)})\)），并定义
\[
S_{-1}(q):=\sum_{i=0}^{q}\bigl(\lambda_i^{(q)}\bigr)^{-1}.
\]
则
\[
S_{-1}(q)=\frac{2\bigl((-1)^q\varphi^{q+2}+\varphi^{-q}\bigr)}{\varphi^2+1}-1\in\ZZ,
\]
并满足
\[
S_{-1}(q)=-S_{-1}(q-1)+S_{-1}(q-2)-1\qquad(q\ge 4),
\]
且
\[
S_{-1}(2)=3,\qquad S_{-1}(3)=-7.
\]
\end{conclusion}
\begin{proof}
由推论 \ref{cor:xi-terminal-replica-softcore-exceptional-reciprocal-sum-fibonacci}，
\(
S_{-1}(q)=-1+2(-1)^qF_{q+1}
\)。
将 Binet 公式与 \(\varphi^2+1=\sqrt5\,\varphi\) 代入即得所示闭式。
设 \(U_q:=S_{-1}(q)+1\)，则 \(U_q\) 为 \((-\varphi)^q,\varphi^{-q}\) 的线性组合，
两者同满足 \(x^2+x-1=0\) 的递推核，故
\(
U_q=-U_{q-1}+U_{q-2}
\)，即
\(
S_{-1}(q)=-S_{-1}(q-1)+S_{-1}(q-2)-1
\)。
初值由 \(q=2,3\) 直接代入。证毕。
\end{proof}

\begin{conclusion}[全逆平方迹与例外谱负二次幂和]\label{con:xi-terminal-replica-softcore-exceptional-neg2-trace-closed-recurrence}
定义

S_{-2}(q):=\sum_{i=0}^{q}\bigl(\lambda_i^{(q)}\bigr)^{-2}.

并且

\operatorname{tr}\bigl((T^{(q)})^{-2}\bigr)

=1+3^q\cdot 4
=4\cdot 3^q+1.

\begin{conclusion}[互补 Möbius 反演：线性方程 \texorpdfstring{$T^{(q)}f=g$}{Tq f=g} 的显式求解]\label{con:xi-terminal-disjointness-complementary-mobius-inversion-linear-solve}
把 \(T^{(q)}\) 视作 \(U_q\) 上函数空间的算子
\[
(T^{(q)}f)(u):=\sum_{v\in U_q}T^{(q)}_{u,v}f(v).
\]
则对任意 \(g:U_q\to\RR\)，方程 \(T^{(q)}f=g\) 存在唯一解，且
\[
f(\mathbf 0)=-g(\mathbf 0)+2g(\mathbf 1),
\]
\[
f(u)=2\sum_{\substack{v\in U_q\\ u\vee v=\mathbf 1}}(-1)^{|u\wedge v|}g(v),
\qquad u\neq \mathbf 0.
\]
\end{conclusion}
\begin{proof}
