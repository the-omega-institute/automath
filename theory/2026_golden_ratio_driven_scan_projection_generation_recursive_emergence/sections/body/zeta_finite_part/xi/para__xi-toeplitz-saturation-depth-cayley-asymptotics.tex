\begin{proposition}[盘内极点的模长闭式与边界层深度的高高度渐近]\label{prop:xi-toeplitz-diskpole-modulus-closed-form-high-height}
\leanverified{paper\_xi\_toeplitz\_diskpole\_modulus\_closed\_form\_high\_height}
取 \(\gamma\in\RR\)、\(\delta>0\)，并令
\[
t:=\gamma+\mathrm{i}\delta\in\mathbb{H},
\qquad
a:=\mathcal{C}(t)=\frac{t-\mathrm{i}}{t+\mathrm{i}}\in\mathbb{D}.
\]
则有精确恒等式
\begin{equation}\label{eq:xi-cayley-modulus-square-closed}
|a|^2=\frac{\gamma^2+(\delta-1)^2}{\gamma^2+(\delta+1)^2}.
\end{equation}
从而
\begin{equation}\label{eq:xi-cayley-one-minus-modsq-closed}
1-|a|^2=\frac{4\delta}{\gamma^2+(1+\delta)^2}.
\end{equation}
并且 \(1-|a|\) 具有双边估计
\begin{equation}\label{eq:xi-cayley-one-minus-mod-bounds}
\frac{2\delta}{\gamma^2+(1+\delta)^2}
\ \le\
1-|a|
\ \le\
\frac{4\delta}{\gamma^2+(1+\delta)^2}.
\end{equation}
在 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，成立渐近展开
\begin{equation}\label{eq:xi-cayley-one-minus-mod-high-height-asymptotic}
1-|a|
=\frac{2\delta}{\gamma^2}
\ +\ O\!\left(\frac{\delta(1+\delta)}{\gamma^4}\right).
\end{equation}
\end{proposition}
\begin{proof}
由 \(a=(t-\mathrm{i})/(t+\mathrm{i})\) 得
\[
|a|^2=\frac{|t-\mathrm{i}|^2}{|t+\mathrm{i}|^2}
=\frac{\gamma^2+(\delta-1)^2}{\gamma^2+(\delta+1)^2},
\]
即得 \eqref{eq:xi-cayley-modulus-square-closed}。
进而
\[
1-|a|^2
=\frac{|t+\mathrm{i}|^2-|t-\mathrm{i}|^2}{|t+\mathrm{i}|^2}
=\frac{(\gamma^2+(\delta+1)^2)-(\gamma^2+(\delta-1)^2)}{\gamma^2+(\delta+1)^2}
=\frac{4\delta}{\gamma^2+(1+\delta)^2},
\]
得到 \eqref{eq:xi-cayley-one-minus-modsq-closed}。
再由恒等式
\(
1-|a|=(1-|a|^2)/(1+|a|)
\)
与 \(1<1+|a|<2\) 得到 \eqref{eq:xi-cayley-one-minus-mod-bounds}。
\par
最后在 \(|\gamma|\to\infty\) 时，\eqref{eq:xi-cayley-one-minus-modsq-closed} 给出
\(
1-|a|^2=\frac{4\delta}{\gamma^2}+O(\delta(1+\delta)\gamma^{-4})
\)；
并由
\(
1-|a|=(1-|a|^2)/(1+|a|)=\tfrac12(1-|a|^2)+O\bigl((1-|a|^2)^2\bigr)
\)
与 \(1-|a|^2=O(\delta\gamma^{-2})\) 得 \eqref{eq:xi-cayley-one-minus-mod-high-height-asymptotic}。证毕。
\end{proof}

\begin{corollary}[显影饱和阶与最浅深度的同阶性]\label{cor:xi-toeplitz-saturation-index-comparable-to-shallowest-depth}
在有限缺陷口径下，设盘内极点多重集为 \(\mathcal A=\{a_1,\dots,a_\kappa\}\subset\mathbb{D}\)，并记
\[
h_{\min}:=\min_{1\le j\le \kappa}(1-|a_j|).
\]
在 Toeplitz 负惯性最终稳定为 \(\kappa\) 的稳定区间内，定义显影饱和阶
\[
N_{\mathrm{sat}}:=\min\{N\ge 0:\ \nu_-(\mathbf{T}_N)=\kappa\}.
\]
若进一步满足推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 的误差衰减假设，并假设端点 Poisson 权重与伪双曲分离度在该缺陷族上具有粗下界（例如 \(\inf_j p_j>0\) 与 \(\inf_{j<k}\rho_{jk}>0\)），则存在常数 \(0<c<C\)（仅依赖于 \(\kappa\) 与上述粗界及误差衰减常数）使
\[
\frac{c}{N_{\mathrm{sat}}}\ \le\ h_{\min}\ \le\ \frac{C}{N_{\mathrm{sat}}}.
\]
特别地，若 \(a_\ast=\mathcal{C}(\gamma_\ast+\mathrm{i}\delta_\ast)\) 实现最浅深度且 \(|\gamma_\ast|\to\infty\)、\(\delta_\ast\) 有界，则
\[
N_{\mathrm{sat}}\ \asymp\ \frac{\gamma_\ast^2}{\delta_\ast}.
\]
\end{corollary}
\begin{proof}
由 \(\nu_-(\mathbf{T}_N)\) 对 \(N\) 的单调性与稳定性假设，\(N_{\mathrm{sat}}\) 存在。
取 \(a_\ast\) 实现 \(h_{\min}=1-|a_\ast|\)。
由于 \(\nu_-(\mathbf{T}_{N_{\mathrm{sat}}})=\kappa\) 蕴含 \(\mathbf{T}_{N_{\mathrm{sat}}}\nsucceq 0\)，故 Toeplitz--PSD 在阶 \(N_{\mathrm{sat}}\) 给出失败见证。
对极点 \(a_\ast\) 应用定理 \ref{thm:xi-toeplitz-length-vs-boundary-depth} 的粗尺度必要下界 \(N\gtrsim h^{-1}\)，得到
\(
N_{\mathrm{sat}}\gtrsim h_{\min}^{-1}
\)，等价于 \(h_{\min}\gtrsim N_{\mathrm{sat}}^{-1}\)，从而得到左侧不等式（吸收常数）。
\par
另一方面，在推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 的假设下，由其充分条件可知存在常数 \(C'>0\) 使当 \(N\ge C'h_{\min}^{-1}\) 时成立 \(\nu_-(\mathbf{T}_N)=\kappa\)；
由 \(N_{\mathrm{sat}}\) 的最小性得到 \(N_{\mathrm{sat}}\le C'h_{\min}^{-1}\)，等价于 \(h_{\min}\le C'/N_{\mathrm{sat}}\)，从而得到右侧不等式（重命名常数）。
\par
最后一条由命题 \ref{prop:xi-toeplitz-diskpole-modulus-closed-form-high-height} 的 \eqref{eq:xi-cayley-one-minus-mod-high-height-asymptotic} 将 \(h_{\min}=1-|a_\ast|\) 转写为 \(\asymp \delta_\ast/\gamma_\ast^2\)，并代回 \(N_{\mathrm{sat}}\asymp h_{\min}^{-1}\) 得到。证毕。
\end{proof}
