\subparagraph{最优传输与度量侧锐常数}
在二阶一致展开 \eqref{eq:xi-poisson-cauchy-second-order-splitting-rt} 的口径下，\(t\to\infty\) 的主导项不仅决定 \(L^p\) 强范数与 Kolmogorov 距离的锐常数，也在一维情形中确定最优传输映射的普适剖面，并给出 Wasserstein 距离的显式极限常数。

\begin{theorem}[Wasserstein 距离的锐渐近常数与最优映射普适剖面]\label{thm:xi-poisson-cauchy-wasserstein-sharp-constant-map-profile}
沿用 \(\tilde h_t=\mathsf P_t*\tilde\nu\)、\(\tilde g_t(x)=\mathsf P_t(x-\bar\gamma)\) 的设定，并假设 \(\int_{\RR}|\gamma-\bar\gamma|^3\,\dd\tilde\nu(\gamma)<\infty\) 且 \(\sigma^2>0\)。
对任意 \(p\in[1,\infty)\) 记 \(\mathsf W_p(\tilde h_t,\tilde g_t)\) 为 \(\RR\) 上代价 \(|x-y|^p\) 所诱导的 \(p\)-Wasserstein 距离。
则存在极限
\[
\lim_{t\to\infty}\frac{t}{\sigma^2}\,\mathsf W_p(\tilde h_t,\tilde g_t)
=c_p,
\qquad
c_p:=\left(\frac{\Gamma\!\bigl(\frac{p+1}{2}\bigr)^2}{\pi\,\Gamma(p+1)}\right)^{1/p}.
\]
并且若 \(\tilde H_t,\tilde G_t\) 分别为 \(\tilde h_t,\tilde g_t\) 的分布函数，令单调最优传输映射
\[
T_t:=\tilde G_t^{-1}\circ \tilde H_t,
\]
则有一致剖面极限
\[
\lim_{t\to\infty}\ \sup_{x\in\RR}\left|
\frac{t}{\sigma^2}\bigl(T_t(x)-x\bigr)
+\frac{\frac{x-\bar\gamma}{t}}{1+\bigl(\frac{x-\bar\gamma}{t}\bigr)^2}
\right|=0.
\]
特别地
\[
\lim_{t\to\infty}\frac{t}{\sigma^2}\,\mathsf W_1(\tilde h_t,\tilde g_t)=\frac{1}{\pi},
\qquad
\lim_{t\to\infty}\frac{t}{\sigma^2}\,\mathsf W_2(\tilde h_t,\tilde g_t)=\frac{1}{2\sqrt2}.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:xi-poisson-cauchy-kolmogorov-second-order-constant} 的证明，存在一致展开
\[
\tilde H_t(x)-\tilde G_t(x)=\sigma^2\int_{-\infty}^{x}r_t(u)\,\dd u+O(t^{-3}),
\]
其中 \(\int_{-\infty}^{x}r_t(u)\,\dd u=-\frac{1}{\pi t^2}\frac{y}{(1+y^2)^2}\)、\(y=(x-\bar\gamma)/t\)。
记
\(
B_t(x):=\pi(\tilde H_t(x)-\tilde G_t(x))
\)，则 \(\sup_x|B_t(x)|=O(t^{-2})\)。
\par
对 Cauchy 分布 \(\tilde g_t\) 有显式
\[
\tilde G_t(x)=\frac12+\frac1\pi\arctan\!\left(\frac{x-\bar\gamma}{t}\right),
\qquad
\tilde G_t^{-1}(u)=\bar\gamma+t\tan\!\bigl(\pi(u-\tfrac12)\bigr).
\]
因此
\[
T_t(x)=\bar\gamma+t\tan\!\left(\arctan\!\left(\frac{x-\bar\gamma}{t}\right)+B_t(x)\right).
\]
令 \(y=(x-\bar\gamma)/t\)，并用 \(\tan(A+B)\) 恒等式与 \(\tan B=B+O(B^3)\)（一致适用于 \(B=B_t(x)\)），得到
\[
T_t(x)-x
=t(1+y^2)\,B_t(x)+O(t^{-3})
=-\frac{\sigma^2}{t}\frac{y}{1+y^2}+o(t^{-1})
\]
在全体 \(x\in\RR\) 上一致成立，从而得到映射剖面极限。
\par
一维情形下最优传输代价由单调映射给出：
\[
\mathsf W_p(\tilde h_t,\tilde g_t)^p=\int_{\RR}|T_t(x)-x|^p\,\tilde h_t(x)\,\dd x.
\]
由上一段一致展开与 \(\|\tilde h_t-\tilde g_t\|_{L^1}=O(t^{-2})\)（命题 \(11.670\)），可用 \(\tilde g_t\) 替代 \(\tilde h_t\) 而仅引入 \(o(t^{-p})\) 误差。
再作尺度代换 \(x=\bar\gamma+ty\)，并用 \(\tilde g_t(\bar\gamma+ty)=\frac{1}{\pi t(1+y^2)}\)，得到
\[
\mathsf W_p(\tilde h_t,\tilde g_t)^p
=\frac{\sigma^{2p}}{\pi t^p}\int_{\RR}\frac{|y|^p}{(1+y^2)^{p+1}}\,\dd y+o(t^{-p}).
\]
最后
\[
\int_{\RR}\frac{|y|^p}{(1+y^2)^{p+1}}\,\dd y
=2\int_0^\infty\frac{y^p}{(1+y^2)^{p+1}}\,\dd y
=\mathrm B\!\left(\frac{p+1}{2},\frac{p+1}{2}\right)
=\frac{\Gamma\!\bigl(\frac{p+1}{2}\bigr)^2}{\Gamma(p+1)}.
\]
取 \(p\) 次方根即得常数 \(c_p\)。
\(p=1,2\) 的特例由 \(\Gamma\) 的显式值化简得到。证毕。
\end{proof}

\begin{corollary}[相对熵与传输的锐标定（极限意义）]\label{cor:xi-poisson-cauchy-kl-wasserstein-calibration}
在定理 \ref{thm:xi-poisson-cauchy-wasserstein-sharp-constant-map-profile} 的假设下，对任意 \(p\in[1,\infty)\) 有
\[
\lim_{t\to\infty}\frac{t^2\,D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)}{\mathsf W_p(\tilde h_t,\tilde g_t)^2}
=\frac{1}{8c_p^2}
=\frac18\left(\frac{\pi\,\Gamma(p+1)}{\Gamma\!\bigl(\frac{p+1}{2}\bigr)^2}\right)^{2/p}.
\]
特别地，
\[
\lim_{t\to\infty}\frac{t^2\,D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)}{\mathsf W_2(\tilde h_t,\tilde g_t)^2}=1,
\qquad
\lim_{t\to\infty}\frac{t^2\,D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)}{\mathsf W_1(\tilde h_t,\tilde g_t)^2}=\frac{\pi^2}{8}.
\]
\end{corollary}

\begin{proof}
由结论 \ref{con:xi-poisson-cauchy-kl-sixth-order-independent-submission} 的首项，
\(
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)=\frac{\sigma^4}{8t^4}+o(t^{-4})
\)。
又由定理 \ref{thm:xi-poisson-cauchy-wasserstein-sharp-constant-map-profile}，
\(
\mathsf W_p(\tilde h_t,\tilde g_t)=c_p\frac{\sigma^2}{t}+o(t^{-1})
\)。
代入并化简即得。证毕。
\end{proof}

\begin{proposition}[Poisson--Cauchy 核导数的 \texorpdfstring{$L^2$}{L2} 完全闭式]\label{prop:xi-poisson-kernel-derivative-l2-closed-form}
\leanverified{paper\_xi\_poisson\_kernel\_derivative\_l2\_closed\_form}
令 \(\mathsf P_t(x)=\frac1\pi\frac{t}{x^2+t^2}\)。
则对任意整数 \(k\ge 0\) 有
\[
\int_{\RR}\bigl(\partial_x^k \mathsf P_t(x)\bigr)^2\,\dd x
=\frac{(2k)!}{\pi\,2^{2k+1}}\,t^{-(2k+1)}.
\]
\end{proposition}

\begin{proof}
由命题 \(11.668\) 的 Fourier 乘子表达式 \(\widehat{\mathsf P_t}(\xi)=e^{-t|\xi|}\)，有
\(
\widehat{\partial_x^k\mathsf P_t}(\xi)=(i\xi)^k e^{-t|\xi|}
\)。
由本文一致的 Parseval 规范
\(
\int_{\RR}|f(x)|^2\,\dd x=\frac{1}{2\pi}\int_{\RR}|\widehat f(\xi)|^2\,\dd\xi
\)，得到
\[
\int_{\RR}\bigl(\partial_x^k \mathsf P_t(x)\bigr)^2\,\dd x
=\frac{1}{2\pi}\int_{\RR}|\xi|^{2k}e^{-2t|\xi|}\,\dd\xi
=\frac{1}{\pi}\int_0^\infty \xi^{2k}e^{-2t\xi}\,\dd\xi.
\]
最后由 \(\int_0^\infty \xi^{2k}e^{-2t\xi}\,\dd\xi=(2k)!/(2t)^{2k+1}\) 得结论。证毕。
\end{proof}

\begin{theorem}[多极阶主导下的 \texorpdfstring{$L^2$}{L2} 锐主项]\label{thm:xi-poisson-cauchy-multipole-l2-leading-term}
设 \(\tilde\nu\in\mathcal P(\RR)\) 的中心矩 \(\mu_j:=\int_{\RR}(\gamma-\bar\gamma)^j\,\dd\tilde\nu(\gamma)\) 存在至 \(j=k+1\)。
若满足
\[
\mu_1=\mu_2=\cdots=\mu_{k-1}=0,
\qquad
\mu_k\neq 0,
\]
则当 \(t\to\infty\) 时有
\[
\|\tilde h_t-\tilde g_t\|_{L^2(\RR)}
\sim
\frac{|\mu_k|}{k!}\sqrt{\frac{(2k)!}{\pi\,2^{2k+1}}}\,t^{-(k+\frac12)}.
\]
\end{theorem}

\begin{proof}
对 \(u=\gamma-\bar\gamma\) 在 \(0\) 处作 \(k\) 阶 Taylor 展开并对 \(\tilde\nu\) 取期望，可得
\[
\tilde h_t(x)-\tilde g_t(x)
=\sum_{j=1}^{k}\frac{(-1)^j\mu_j}{j!}\,\partial_x^j\mathsf P_t(x-\bar\gamma)+R_{k+1,t}(x),
\]
其中余项由 \(\mu_{k+1}\) 与 \(\partial_x^{k+1}\mathsf P_t\) 的一致界控制。
在假设下主项为 \(\frac{(-1)^k\mu_k}{k!}\partial_x^k\mathsf P_t(x-\bar\gamma)\)。
由命题 \ref{prop:xi-poisson-kernel-derivative-l2-closed-form} 得
\(
\|\partial_x^k\mathsf P_t\|_2=\sqrt{\frac{(2k)!}{\pi\,2^{2k+1}}}\,t^{-(k+\frac12)}
\)。
并且余项满足 \(\|R_{k+1,t}\|_2=o(t^{-(k+\frac12)})\)（由 \(t\)-标度与 \(\mu_{k+1}\) 的有限性得到），故得到所示锐主项。证毕。
\end{proof}

\begin{proposition}[度量侧刚性：超阶衰减强制原子化]\label{prop:xi-poisson-cauchy-metric-rigidity-atomic-only}
固定 \(p\in[1,\infty)\)。
若满足下述任一超阶衰减条件：
\[
\mathsf W_p(\tilde h_t,\tilde g_t)=o(t^{-1}),
\quad\text{或}\quad
\sup_{x\in\RR}|\tilde H_t(x)-\tilde G_t(x)|=o(t^{-2}),
\quad\text{或}\quad
\|\tilde h_t-\tilde g_t\|_{L^p(\RR)}=o\!\left(t^{\frac1p-3}\right),
\]
则必有 \(\sigma^2=0\)，从而 \(\tilde\nu=\delta_{\bar\gamma}\)，并且对一切 \(t>0\) 有 \(\tilde h_t\equiv \tilde g_t\)。
\end{proposition}

\begin{proof}
若 \(\sigma^2>0\)，则由定理 \ref{thm:xi-poisson-cauchy-wasserstein-sharp-constant-map-profile}、定理 \ref{thm:xi-poisson-cauchy-kolmogorov-second-order-constant} 与定理 \ref{thm:xi-poisson-cauchy-lp-sharp-spectrum} 分别得到相应的重标度极限为严格正的显式常数，与任一超阶衰减矛盾。
故 \(\sigma^2=0\)。
方差为零强制 \(\tilde\nu\) 退化为 \(\bar\gamma\) 处原子测度，代入定义即 \(\tilde h_t=\mathsf P_t*\delta_{\bar\gamma}=\tilde g_t\)。证毕。
\end{proof}
