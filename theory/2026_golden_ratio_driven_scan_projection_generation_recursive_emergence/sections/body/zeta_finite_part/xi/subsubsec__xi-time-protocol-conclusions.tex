\Longrightarrow
\mathbb P(\mathrm{Fail}_{m,n,N})\to 1.

\begin{proof}

\end{proof}

\end{theorem}

\begin{proof}

\begin{theorem}[局部指数即局部时间纤维维数]\label{thm:xi-local-index-equals-local-time-fiber}

\dim(\mathcal T_x)_{\min}=\mathrm{Ind}(E)(x).

\mathcal A=L^\infty(\Omega),\ \mathcal B=L^\infty(X)

\mathrm{Ind}(E)(x)=|\Fold^{-1}(x)|\quad\text{(a.e.)}.

T_{\min}(x)\ge \frac{\log \mathrm{Ind}(E)(x)}{\log |\mathfrak A_0|},

并在有限型编码条件下渐近可达。
\end{theorem}

\begin{proof}

\end{proof}

\begin{theorem}[时间谱三元组：长度算子重建距离与体积]\label{thm:xi-time-spectral-triple-distance-volume}

\begin{theorem}[时间 \(\zeta\)-函数主奇性由增长常数决定]\label{thm:xi-time-zeta-singularity-from-growth}
定义时间谱 \(\zeta\)-函数
\[
\zeta_{\mathrm{time}}(s):=\mathrm{Tr}(|D|^{-s}).
\]
其收敛半平面由谱维数 \(p\) 决定：\(\Re(s)>p\) 收敛，\(\Re(s)\le p\) 发散。
若时间球计数满足
\[
\#\{g:\widetilde\ell(g)\le t\}\sim e^{\kappa t},
\]
则 \(p\) 与 \(\kappa\) 存在确定性换算（同一增长标尺的倒数量纲等价）。
若热核满足
\[
\mathrm{Tr}(e^{-t|D|})
\sim
\sum_{j=0}^{J}a_j\,t^{\,j-p}
\qquad(t\downarrow 0),
\]
则 \(\zeta_{\mathrm{time}}\) 在 \(s=p-j\) 处出现可控极点，残数由 \(a_j\) 唯一确定。
\end{theorem}

\begin{proof}
由 Mellin 变换关系
\(
\zeta_{\mathrm{time}}(s)=\Gamma(s)^{-1}\int_0^\infty t^{s-1}\mathrm{Tr}(e^{-t|D|})dt
\)
与 Tauber 型增长定理得到收敛域与极点结构。
时间球增长率与临界指数对应同一 abscissa，给出换算关系。
\end{proof}

\begin{theorem}[纤维时间的多重分形律由压强函数唯一确定]\label{thm:xi-fiber-time-multifractal-legendre}
令局部指数为 \(\mathrm{Ind}_m(x)\)，定义消歧时间
\[
T_m(x):=\log \mathrm{Ind}_m(x).
\]
设极限压强
\[
P(s):=\lim_{m\to\infty}\frac1m\log\sum_{x\in X_m}e^{\,sT_m(x)}
\]
存在并在一区间可微。
则 \(\frac1mT_m(X_m)\) 满足大偏差原理，其速率函数
\[
I(a)=\sup_s\{sa-P(s)\}.
\]
并且等值集
\(
E(a)=\{x:\lim_{m\to\infty}m^{-1}T_m(x)=a\}
\)
在时间度量 \(d_C\) 下的维数由 \(P\) 的 Legendre 数据显式给出。
\end{theorem}

\begin{proof}
由 G\"artner--Ellis 条件（压强可微）得到速率函数。
多重分形维数公式由大偏差与时间度量覆盖估计合并得到，参数化由同一 \(P\) 完成。
\end{proof}

\begin{theorem}[时间诱导 Dirichlet 形式：扩散与拉普拉斯的一元生成]\label{thm:xi-time-induced-dirichlet-form}
