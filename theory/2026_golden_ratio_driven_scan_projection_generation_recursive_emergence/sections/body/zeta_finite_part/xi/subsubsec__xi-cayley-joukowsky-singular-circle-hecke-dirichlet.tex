\begin{theorem}[Cayley--Joukowsky 奇异环测度律与椭圆外域调和测度的闭式密度]\label{thm:xi-cayley-joukowsky-harmonic-measure-ellipse}
\leanverified{paper\_xi\_cayley\_joukowsky\_harmonic\_measure\_ellipse}
令 Cayley 紧化 \(\mathcal C(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\) 如定理
\ref{thm:xi-cayley-modulus-poisson-fourier-fingerprint-layered}，并以其将扩展实线
\(\widehat{\RR}:=\RR\cup\{\infty\}\) 送入单位圆 \(\mathbb T\)。
定义 Cauchy 权测度
\[
\dd\mu_{\mathrm C}(x):=\frac{\dd x}{\pi(1+x^2)}.
\]
则 \(\mathcal C_\#\mu_{\mathrm C}=\frac{\dd\theta}{2\pi}\)（在 \(\mathcal C(x)=e^{\mathrm{i}\theta}\) 的角坐标下）。
\par
对任意 \(L>0\) 定义 Joukowsky--椭圆化映射
\[
J_L:\mathbb T\to\CC,\qquad
J_L(e^{\mathrm{i}\theta})
=\sqrt{L}\,e^{\mathrm{i}\theta}+\frac{1}{\sqrt{L}}e^{-\mathrm{i}\theta}
=a(L)\cos\theta+\mathrm{i}\,b(L)\sin\theta,
\]
其中
\[
a(L)=\sqrt{L}+\frac{1}{\sqrt{L}},\qquad
b(L)=\sqrt{L}-\frac{1}{\sqrt{L}}.
\]
记椭圆像 \(E_L:=J_L(\mathbb T)\)。
令 \(\nu_L:=(J_L\circ \mathcal C)_\#\mu_{\mathrm C}\)。
则：
\begin{enumerate}
  \item \(\nu_L\) 为 \(E_L\) 外域在无穷远处的调和测度；并且在参数 \(\theta\) 下严格均匀：
  \[
  \nu_L=\frac{\dd\theta}{2\pi}\qquad(\text{经由 }S=J_L(e^{\mathrm{i}\theta})\text{ 识别}).
  \]
  \item 若 \(L\neq 1\)，以椭圆弧长元 \(\dd s\) 表示，则 \(\nu_L\) 具有闭式密度
  \[
  \dd\nu_L(S)
  =\frac{1}{2\pi}\frac{1}{|J_L'(e^{\mathrm{i}\theta})|}\,\dd s
  =\frac{1}{2\pi}\frac{1}{\sqrt{a(L)^2\sin^2\theta+b(L)^2\cos^2\theta}}\,\dd s,
  \qquad S=J_L(e^{\mathrm{i}\theta}).
  \]
  \item 在奇异极限 \(L\downarrow 1\) 下，椭圆退化为区间 \(E_1=[-2,2]\)，且 \(\nu_L\) 弱收敛到弧正弦测度
  \[
  \dd\nu_1(x)=\frac{\dd x}{\pi\sqrt{4-x^2}},\qquad x\in(-2,2).
  \]
  并且 \(\mathcal C(\infty)=-1\) 在 \(J_1\) 下的像点为端点 \(-2\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(\mathcal C(x)=e^{\mathrm{i}\theta}\) 且 \(\theta=2\arctan x\) 得
\(
\dd\theta=2\,\dd x/(1+x^2)
\)，故
\(
\dd\theta/(2\pi)=\dd x/(\pi(1+x^2))
\)，从而 \(\mathcal C_\#\mu_{\mathrm C}=\dd\theta/(2\pi)\)。
又由
\(
S(\theta)=a(L)\cos\theta+\mathrm{i}b(L)\sin\theta
\)
得
\[
\frac{\dd S}{\dd\theta}=-a(L)\sin\theta+\mathrm{i}b(L)\cos\theta,\qquad
\dd s=\Bigl|\frac{\dd S}{\dd\theta}\Bigr|\,\dd\theta
=\sqrt{a(L)^2\sin^2\theta+b(L)^2\cos^2\theta}\,\dd\theta,
\]
故第二条成立。
当 \(L=1\) 时 \(S(\theta)=2\cos\theta\)，以 \(\theta\) 在 \([0,2\pi)\) 上均匀并按两支覆盖换元得到弧正弦密度；端点陈述由 \(\mathcal C(\infty)=-1\) 与 \(J_1(-1)=-2\) 直接给出。
\end{proof}

\begin{proposition}[椭圆环平均的离散残数公式与壁穿越跳跃]\label{prop:xi-elliptic-ring-average-residue-wallcrossing}
\leanverified{paper\_xi\_elliptic\_ring\_average\_residue\_wallcrossing}
令 \(f\) 为有理函数。对 \(r>0\) 定义椭圆环平均
\[
\mathcal L_{r^2}(f):=\frac{1}{2\pi}\int_{0}^{2\pi} f\!\left(re^{\mathrm{i}\theta}+r^{-1}e^{-\mathrm{i}\theta}\right)\,\dd\theta,
\]
并定义置换
\[
g(z):=f(z+z^{-1}),\qquad h(z):=\frac{g(z)}{z}.
\]
若 \(r\) 不等于 \(h\) 的任何非零极点的模长，则有离散残数公式
\[
\boxed{\;
\mathcal L_{r^2}(f)=\Res_{z=0}h(z)+\sum_{0<|a|<r}\Res_{z=a}h(z)\;}
\]
（求和取遍满足 \(0<|a|<r\) 的全部极点并按重数计）。
此外，当 \(r\) 穿越某个模长为 \(r_\star\) 的极点簇时，\(\mathcal L_{r^2}(f)\) 的跳量恰等于该模长圆周上全部残数之和
\[
\mathcal L_{r_+^2}(f)-\mathcal L_{r_-^2}(f)=\sum_{|a|=r_\star}\Res_{z=a}h(z),
\]
其中 \(r_-<r_\star<r_+\) 且 \((r_-,r_+)\) 内无其它非零极点模长。
\end{proposition}
\begin{proof}
令 \(z=re^{\mathrm{i}\theta}\)。则 \(\dd\theta=\dd z/(\mathrm{i}z)\)，并且
\[
\mathcal L_{r^2}(f)=\frac{1}{2\pi \mathrm{i}}\oint_{|z|=r} f(z+z^{-1})\,\frac{\dd z}{z}
=\frac{1}{2\pi\mathrm{i}}\oint_{|z|=r}h(z)\,\dd z.
\]
当 \(r\) 避开非零极点模长时，按留数定理，该积分等于圆盘 \(|z|<r\) 内全部留数之和，即得到所示公式。
跳跃恒等式由积分路径从 \(|z|=r_-\) 连续变形到 \(|z|=r_+\) 并跨过 \(|z|=r_\star\) 的极点圆周时的留数补偿项给出。证毕。
\end{proof}

\begin{theorem}[椭圆环平均的两值壁穿越判别]\label{thm:xi-elliptic-ring-average-two-valued-wallcrossing}
\leanverified{paper\_xi\_elliptic\_ring\_average\_two\_valued\_wallcrossing}
在命题 \ref{prop:xi-elliptic-ring-average-residue-wallcrossing} 的设定下，取开区间 \(I\subset(0,\infty)\)，并令
\(\{r_1<\cdots<r_N\}\) 为 \(h\) 在 \(I\) 内全部 \emph{非零} 极点模长的互异集合。
对每条壁定义总跳量
\[
S_j:=\sum_{|a|=r_j}\Res_{z=a}h(z)\qquad(j=1,\dots,N),
\]
并令部分和 \(T_j:=\sum_{k\le j}S_k\)（约定 \(T_0:=0\)）。
则函数 \(r\mapsto\mathcal L_{r^2}(f)\) 在 \(I\setminus\{r_1,\dots,r_N\}\) 上至多取两点值，当且仅当存在 \(\Delta\in\CC\) 使得对一切 \(j=1,\dots,N\) 有
\[
\boxed{\;T_j\in\{0,\Delta\}\;}
\]
在此情形下，忽略 \(S_j=0\) 的壁，则每个非零跳量必满足 \(S_j\in\{+\Delta,-\Delta\}\)，且其符号沿 \(j\) 严格交替；反之该交替条件亦充分。
\end{theorem}
\begin{proof}
由命题 \ref{prop:xi-elliptic-ring-average-residue-wallcrossing}，在每个连通分支 \((r_j,r_{j+1})\subset I\) 上有
\[
\mathcal L_{r^2}(f)=C_0+\sum_{k\le j}S_k=C_0+T_j,
\]
其中 \(C_0\) 为最左分支上的常值。
因此 \(\mathcal L_{r^2}(f)\) 在 \(I\setminus\{r_j\}\) 上至多取两值当且仅当 \(T_j\) 在 \(\{0,\Delta\}\) 中取值（取 \(\Delta\) 为两值之差）。
在该条件下，若 \(S_j\neq 0\) 则 \(S_j=T_j-T_{j-1}\) 强制 \(S_j=\Delta\) 当且仅当 \(T_{j-1}=0\)，且 \(S_j=-\Delta\) 当且仅当 \(T_{j-1}=\Delta\)，故符号交替；反向蕴含同理由 \(T_j\) 的二值性推出。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$s\mapsto 1-s$}{s->1-s} 对合下的 Cayley 共轭与对数微分恒等式]\label{prop:xi-cayley-tau-conjugacy-logdiff}
\leanverified{paper\_xi\_cayley\_tau\_conjugacy\_logdiff}
令 \(\tau:\PP^1(\CC)\to\PP^1(\CC)\) 为对合 \(\tau(s)=1-s\)，并定义 Möbius 变换
\[
\mathcal C_\tau(s):=\frac{s}{1-s}\qquad (s\neq 1).
\]
则成立：
\begin{enumerate}
  \item \(\mathcal C_\tau\circ\tau=\iota\circ\mathcal C_\tau\)，其中 \(\iota(w)=1/w\)。等价地，
  \[
  \mathcal C_\tau(1-s)=\frac{1}{\mathcal C_\tau(s)}\qquad (s\neq 0,1).
  \]
  \item 临界线 \(\Re(s)=\tfrac12\) 在 \(\mathcal C_\tau\) 下的像为单位圆 \(|w|=1\)，并且
  \[
  \Re(s)>\tfrac12\iff |\mathcal C_\tau(s)|>1,\qquad
  \Re(s)<\tfrac12\iff |\mathcal C_\tau(s)|<1.
  \]
  \item 在三点穿孔球面 \(\PP^1\setminus\{0,1,\infty\}\) 上有精确的 \(1\)-形式恒等式
  \[
  \frac{\dd s}{s(1-s)}=\frac{\dd w}{w},\qquad w=\mathcal C_\tau(s).
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
直接计算得
\(
\mathcal C_\tau(1-s)=\frac{1-s}{s}=1/\mathcal C_\tau(s)
\)，从而 (1) 成立。
写 \(s=\sigma+\mathrm{i}t\)，则
\[
|\mathcal C_\tau(s)|^2=\frac{\sigma^2+t^2}{(1-\sigma)^2+t^2},
\]
故 \(|\mathcal C_\tau(s)|=1\) 当且仅当 \(\sigma=\tfrac12\)，并且 \(\sigma\gtrless \tfrac12\) 等价于 \(|\mathcal C_\tau(s)|\gtrless 1\)，得到 (2)。
对 (3)，由反解 \(s=\frac{w}{1+w}\) 得
\[
\dd s=\frac{\dd w}{(1+w)^2},\qquad s(1-s)=\frac{w}{(1+w)^2},
\]
从而 \(\dd s/(s(1-s))=\dd w/w\)。证毕。
\end{proof}

\begin{corollary}[临界线的圆周参数化与 Cauchy 权重的角测度规一化]\label{cor:xi-critical-line-circle-cauchy-uniformization}
\leanverified{paper\_xi\_critical\_line\_circle\_cauchy\_uniformization}
在命题 \ref{prop:xi-cayley-tau-conjugacy-logdiff} 的记号下，取 \(w=e^{\mathrm{i}\theta}\)（\(\theta\in(-\pi,\pi)\)）并令
\[
s(\theta):=\frac{w}{1+w}.
\]
则有显式参数化
\[
s(\theta)=\frac12+\frac{\mathrm{i}}{2}\tan\frac{\theta}{2}.
\]
因此若写 \(s=\tfrac12+\mathrm{i}t\)，则
\begin{equation}\label{eq:xi-critical-line-t-theta}
t=\frac12\tan\frac{\theta}{2},\qquad
\dd\theta=\frac{\dd t}{t^2+\tfrac14}.
\end{equation}
并且在缩放变量 \(x:=2t\) 下，
\begin{equation}\label{eq:xi-critical-line-cauchy-scale2}
\frac{\dd\theta}{2\pi}=\frac{\dd x}{\pi(1+x^2)}.
\end{equation}
\end{corollary}
\begin{proof}
由
\(
1+e^{\mathrm{i}\theta}=2\cos(\theta/2)e^{\mathrm{i}\theta/2}
\)
得
\[
s(\theta)=\frac{e^{\mathrm{i}\theta}}{1+e^{\mathrm{i}\theta}}
=\frac{e^{\mathrm{i}\theta/2}}{2\cos(\theta/2)}
=\frac12+\frac{\mathrm{i}}{2}\tan(\theta/2).
\]
进而
\(
\dd t=\frac14\sec^2(\theta/2)\dd\theta
\)
且
\(
t^2+\tfrac14=\tfrac14\sec^2(\theta/2)
\)，从而得到 \eqref{eq:xi-critical-line-t-theta}。
令 \(x=2t\) 则 \(\dd\theta=\frac{2\,\dd x}{1+x^2}\)，即 \eqref{eq:xi-critical-line-cauchy-scale2}。证毕。
\end{proof}

\begin{theorem}[奇异环 Jensen 恒等式与临界线缺陷的标量化]\label{thm:xi-singular-ring-jensen-identity-defect}
\leanverified{paper\_xi\_singular\_ring\_jensen\_identity\_defect}
设 \(f\) 为整函数并满足对称关系 \(f(s)=f(1-s)\) 以及 \(f(0)\neq 0\)。
假设对数平均
\begin{equation}\label{eq:xi-singular-ring-jensen-Jf-def}
\mathfrak J(f):=\frac{1}{2\pi}\int_{-\infty}^{\infty}\log\bigl|f(\tfrac12+\mathrm{i}t)\bigr|\,
\frac{\dd t}{t^2+\tfrac14}
\end{equation}
存在且有限。
则有恒等式
\begin{equation}\label{eq:xi-singular-ring-jensen-Jf-zeros}
\mathfrak J(f)=\log|f(0)|+\sum_{\rho:\ f(\rho)=0,\ \Re(\rho)<\tfrac12}\log\left|\frac{1-\rho}{\rho}\right|,
\end{equation}
其中求和按零点重数计，并且右端为非负级数。
特别地，定义缺陷
\begin{equation}\label{eq:xi-singular-ring-jensen-defect}
\Delta(f):=\mathfrak J(f)-\log|f(0)|\ \ge\ 0,
\end{equation}
则 \(\Delta(f)=0\) 当且仅当 \(f\) 的全部零点满足 \(\Re(\rho)=\tfrac12\)。
\end{theorem}
\begin{proof}
定义
\[
s(w):=\frac{w}{1+w},\qquad F(w):=f(s(w))\qquad(|w|<1),
\]
则 \(F\) 在 \(\mathbb D\) 内解析且 \(F(0)=f(0)\neq 0\)。
由推论 \ref{cor:xi-critical-line-circle-cauchy-uniformization}，
\(
s(e^{\mathrm{i}\theta})=\tfrac12+\frac{\mathrm{i}}{2}\tan(\theta/2)
\)
且 \(\dd\theta=\dd t/(t^2+\tfrac14)\)，从而
\[
\mathfrak J(f)=\frac{1}{2\pi}\int_{-\pi}^{\pi}\log|F(e^{\mathrm{i}\theta})|\,\dd\theta
=\frac{1}{2\pi}\int_{0}^{2\pi}\log|F(e^{\mathrm{i}\theta})|\,\dd\theta.
\]
由 Jensen 公式，
\[
\frac{1}{2\pi}\int_{0}^{2\pi}\log|F(e^{\mathrm{i}\theta})|\,\dd\theta
=\log|F(0)|+\sum_{a:\ F(a)=0,\ |a|<1}\log\frac{1}{|a|}.
\]
另一方面，\(F(a)=0\) 等价于 \(f(\rho)=0\) 且 \(\rho=\frac{a}{1+a}\)，亦即 \(a=\frac{\rho}{1-\rho}=\mathcal C_\tau(\rho)\)。
并且由命题 \ref{prop:xi-cayley-tau-conjugacy-logdiff}(2)，\(|a|<1\) 当且仅当 \(\Re(\rho)<\tfrac12\)。
最后注意
\(
\log\frac1{|a|}=\log\left|\frac{1-\rho}{\rho}\right|
\)，代回即得 \eqref{eq:xi-singular-ring-jensen-Jf-zeros}，并推出 \eqref{eq:xi-singular-ring-jensen-defect}。
由于 \(f(s)=f(1-s)\) 强制零点在 \(\rho\mapsto 1-\rho\) 下成对出现，故不存在 \(\Re(\rho)<\tfrac12\) 的零点等价于不存在 \(\Re(\rho)>\tfrac12\) 的零点，从而 \(\Delta(f)=0\) 当且仅当全部零点落在 \(\Re(\rho)=\tfrac12\)。证毕。
\end{proof}

\begin{proposition}[离临界线偏移的缺陷下界]\label{prop:xi-singular-ring-defect-delta-lowerbound}
在定理 \ref{thm:xi-singular-ring-jensen-identity-defect} 的设定下，设 \(\rho=\beta+\mathrm{i}\gamma\) 为零点且 \(\beta<\tfrac12\)，并记 \(\delta:=\tfrac12-\beta>0\)。
则
\begin{equation}\label{eq:xi-singular-ring-defect-single-lowerbound}
\log\left|\frac{1-\rho}{\rho}\right|
=\frac12\log\!\left(1+\frac{2\delta}{|\rho|^2}\right)
\ge \frac{\delta}{|\rho|^2+2\delta}.
\end{equation}
从而有下界
\begin{equation}\label{eq:xi-singular-ring-defect-sum-lowerbound}
\Delta(f)\ge \sum_{\rho:\ \Re(\rho)<\tfrac12}\frac{\tfrac12-\Re(\rho)}{|\rho|^2+2(\tfrac12-\Re(\rho))}.
\end{equation}
若进一步满足 \(\Re(\rho)\in[0,1]\)（对所有零点），则
\begin{equation}\label{eq:xi-singular-ring-defect-sum-lowerbound-simplified}
\Delta(f)\ge \sum_{\rho:\ \Re(\rho)<\tfrac12}\frac{\tfrac12-\Re(\rho)}{|\rho|^2+1}.
\end{equation}
\end{proposition}
\leanverified{paper\_xi\_singular\_ring\_defect\_delta\_lowerbound}
\begin{proof}
计算得
\[
\left|\frac{1-\rho}{\rho}\right|^2
=\frac{(1-\beta)^2+\gamma^2}{\beta^2+\gamma^2}
=1+\frac{1-2\beta}{|\rho|^2}
=1+\frac{2\delta}{|\rho|^2},
\]
从而得到 \eqref{eq:xi-singular-ring-defect-single-lowerbound} 的第一等式。
对 \(x>0\) 使用不等式 \(\log(1+x)\ge \frac{x}{1+x}\) 并取 \(x=2\delta/|\rho|^2\) 得到第二个不等式。
对所有满足 \(\Re(\rho)<\tfrac12\) 的零点求和并与 \eqref{eq:xi-singular-ring-jensen-defect} 对照即可得到 \eqref{eq:xi-singular-ring-defect-sum-lowerbound}。
若 \(\delta\le \tfrac12\)，则 \(|\rho|^2+2\delta\le |\rho|^2+1\)，故得到 \eqref{eq:xi-singular-ring-defect-sum-lowerbound-simplified}。证毕。
\end{proof}

\begin{proposition}[缺陷预算导出的离线零点计数约束]\label{prop:xi-singular-ring-defect-counting-bound}
\leanverified{paper\_xi\_singular\_ring\_defect\_counting\_bound}
在命题 \ref{prop:xi-singular-ring-defect-delta-lowerbound} 的设定下，给定 \(R>0\)、\(\eta>0\)。
则满足
\[
|\rho|\le R,\qquad \Re(\rho)\le \tfrac12-\eta
\]
的零点个数（按重数计）满足上界
\begin{equation}\label{eq:xi-singular-ring-defect-counting-bound}
\#\Bigl\{\rho:\ |\rho|\le R,\ \Re(\rho)\le \tfrac12-\eta\Bigr\}
\le \Delta(f)\cdot\frac{R^2+2\eta}{\eta}.
\end{equation}
\end{proposition}
\begin{proof}
对满足条件的每个零点有 \(\delta\ge \eta\) 且 \(|\rho|^2+2\delta\le R^2+2\eta\)，故由
\eqref{eq:xi-singular-ring-defect-sum-lowerbound} 得单项贡献不少于 \(\eta/(R^2+2\eta)\)。
将该下界乘以零点个数并与 \(\Delta(f)\) 比较即得 \eqref{eq:xi-singular-ring-defect-counting-bound}。证毕。
\end{proof}

\begin{proposition}[缺陷的乘法性]\label{prop:xi-singular-ring-defect-multiplicative}
设 \(f,g\) 满足定理 \ref{thm:xi-singular-ring-jensen-identity-defect} 的全部条件，且 \((fg)(0)\neq 0\)。
则
\begin{equation}\label{eq:xi-singular-ring-defect-additive}
\Delta(fg)=\Delta(f)+\Delta(g),
\qquad
\Delta(f^n)=n\,\Delta(f)\ \ (n\in\NN).
\end{equation}
\end{proposition}
\begin{proof}

代入 \eqref{eq:xi-singular-ring-jensen-defect} 即得 \eqref{eq:xi-singular-ring-defect-additive}。证毕。
\end{proof}

\begin{theorem}[伸缩半群在 Cayley 圆上的显式共轭与端点固定性]\label{thm:xi-cayley-dilation-mobius-conjugacy-fixedpoints}
\leanverified{paper\_xi\_cayley\_dilation\_mobius\_conjugacy\_fixedpoints}

\[
\varphi_m:=\mathcal C\circ D_m\circ\mathcal C^{-1}:\mathbb T\to\mathbb T.

则 \(\varphi_m\) 为 Möbius 变换并具有闭式
\[
\varphi_m(w)=\frac{(1-m)+(1+m)w}{(1+m)+(1-m)w}.

  \item \(\varphi_m\) 保持单位圆且构成交换半群：\(\varphi_{mn}=\varphi_m\circ\varphi_n\)；
  \item \(\varphi_m\) 的不动点为 \(w=\pm 1\)，其中 \(w=-1=\mathcal C(\infty)\) 为无穷端点；
  \item 对任意 \(L>0\)，在椭圆坐标 \(S=J_L(w)\) 下，\(\varphi_m\) 固定 \(S=\pm a(L)\)；在奇异退化 \(L=1\) 时固定端点 \(\pm 2\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(\mathcal C^{-1}(w)=-\mathrm{i}\frac{w-1}{w+1}\)（\(w\neq -1\)）代入 \(\varphi_m(w)=\mathcal C(m\mathcal C^{-1}(w))\) 并化简即得所示闭式。
半群关系由 \(D_{mn}=D_m\circ D_n\) 的共轭函子性推出。
直接代入得 \(\varphi_m(\pm 1)=\pm 1\)，且 \(\mathcal C(\infty)=-1\) 为伸缩不变点。
最后由 \(J_L(\pm 1)=\pm(\sqrt{L}+1/\sqrt{L})=\pm a(L)\) 与 \(a(1)=2\) 得到端点固定性。
\end{proof}

\begin{theorem}[Cauchy--cosh 规范化与伸缩作用的严格线性化]\label{thm:xi-cauchy-cosh-unitary-linearization}
\leanverified{paper\_xi\_cauchy\_cosh\_unitary\_linearization}
在 Hilbert 空间
\[
\mathcal H:=L^2\!\Bigl(\RR,\dd\mu_{\mathrm C}(x)=\frac{\dd x}{\pi(1+x^2)}\Bigr)
\]
上，对 \(m>0\) 定义加权伸缩算子
\[
(U_m f)(x):=\sqrt{\frac{m(1+x^2)}{m^2+x^2}}\;f\!\left(\frac{x}{m}\right).
\]
则 \(\{U_m\}_{m>0}\) 为一族酉算子并满足表示律 \(U_{mn}=U_mU_n\)。
\par
进一步，定义酉同构
\[
(Wf)_\pm(y):=f(\pm e^{y})\sqrt{\frac{e^{y}}{\pi(1+e^{2y})}},\qquad y\in\RR,
\]
则
\(
W:\mathcal H\to L^2(\RR,\dd y)\oplus L^2(\RR,\dd y)
\)
为酉同构，并且对任意 \(m>0\) 有严格平移线性化
\[
(WU_m f)_\pm(y)=(Wf)_\pm\bigl(y-\log m\bigr).
\]
\end{theorem}
\begin{proof}
直接计算范数并换元 \(u=x/m\)：
\[
\|U_m f\|_{\mathcal H}^2
=\int_{\RR}\frac{m(1+x^2)}{m^2+x^2}\left|f\!\left(\frac{x}{m}\right)\right|^2\frac{\dd x}{\pi(1+x^2)}
=\int_{\RR}\left|f(u)\right|^2\frac{\dd u}{\pi(1+u^2)}
=\|f\|_{\mathcal H}^2,
\]
故 \(U_m\) 酉；表示律由代入核对即可。
\par
对 \(W\) 的酉性，把 \(\RR\) 分解为正负半轴并令 \(t=e^{y}\) 即得
\(
\|f\|_{\mathcal H}^2=\sum_\pm\|(Wf)_\pm\|_{L^2(\dd y)}^2
\)。
而对 \(x=\pm e^{y}\) 有
\[
(U_m f)(\pm e^{y})
=\sqrt{\frac{m(1+e^{2y})}{m^2+e^{2y}}}\;f(\pm e^{y-\log m}),
\]
乘以权因子并化简得到 \((WU_m f)_\pm(y)=(Wf)_\pm(y-\log m)\)。
\end{proof}

\begin{theorem}[奇异环 Hecke--Dirichlet 算子族的完全谱对角化与 \texorpdfstring{$\zeta$}{zeta} 乘子]\label{thm:xi-singular-circle-hecke-dirichlet-zeta-multiplier}
\leanverified{paper\_xi\_singular\_circle\_hecke\_dirichlet\_zeta\_multiplier}
取 \(s\in\CC\) 满足 \(\Re(s)>1\)。
在 \(\mathcal H\) 上定义算子级 Dirichlet 级数
\[
\mathcal H_s:=\sum_{m\ge 1}m^{-s}U_m,
\]
则该级数在算子范数意义下绝对收敛并给出有界算子。
令 \(\mathcal F\) 为 \(y\)-变量上的 Fourier 变换（对两分量分别作用，约定
\((\mathcal Fg)(t)=\int_{\RR}e^{-\mathrm{i}ty}g(y)\dd y\)），则有酉等价
\[
(\mathcal F W)\,\mathcal H_s\,(\mathcal F W)^{-1}
=\mathrm{M}_{\zeta(s+\mathrm{i}t)}\oplus \mathrm{M}_{\zeta(s+\mathrm{i}t)},
\]
其中 \(\mathrm{M}_{\phi}\) 表示乘子算子 \((\mathrm{M}_{\phi}h)(t)=\phi(t)h(t)\)。
因此
\[
\|\mathcal H_s\|
=\operatorname*{ess\,sup}_{t\in\RR}|\zeta(s+\mathrm{i}t)|.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-cauchy-cosh-unitary-linearization}，\(WU_mW^{-1}\) 在每个分量上为平移算子
\(
T_{\log m}:g(y)\mapsto g(y-\log m)
\)。
由 \(\Re(s)>1\) 得 \(\sum_{m\ge 1}|m^{-s}|<\infty\)，且 \(\|T_{\log m}\|=1\)，故 \(\mathcal H_s\) 在算子范数下绝对收敛并
\[
W\mathcal H_sW^{-1}=\sum_{m\ge 1}m^{-s}T_{\log m}.
\]
Fourier 变换满足 \(\mathcal F(T_a g)(t)=e^{-\mathrm{i}ta}\widehat g(t)\)；
取 \(a=\log m\) 得 \(\mathcal F(T_{\log m}g)(t)=m^{-\mathrm{i}t}\widehat g(t)\)。
于是
\[
\mathcal F\left(\sum_{m\ge 1}m^{-s}T_{\log m}g\right)(t)
=\left(\sum_{m\ge 1}m^{-(s+\mathrm{i}t)}\right)\widehat g(t)
=\zeta(s+\mathrm{i}t)\widehat g(t),
\]
从而得到乘子对角化。
范数恒等式由乘子算子在 \(L^2\) 上的标准刻画给出。
\end{proof}

\begin{conjecture}[\texorpdfstring{$\xi$}{xi} 的奇异环散射行列式实现与临界线的酉谱闭合判据]\label{conj:xi-singular-circle-xi-scattering-determinant}
设 \(\mathcal H_s\) 如定理 \ref{thm:xi-singular-circle-hecke-dirichlet-zeta-multiplier}。
预期存在一族可构造的正规化算子 \(\widetilde{\mathcal H}_s\)（例如对 \(\mathcal H_s\) 在 \(s=1\) 的极点作剥离并对高频作可控截断后取极限），满足：
\begin{enumerate}
  \item \(s\mapsto \widetilde{\mathcal H}_s\) 在 \(\Re(s)\ge \tfrac12\) 上给出 Fredholm 算子族，并存在由 \(x\mapsto -1/x\) 在 Cayley 圆上诱导的对合算子 \(\mathcal J\) 使得
  \[
  \widetilde{\mathcal H}_s^{\,*}\sim \mathcal J\,\widetilde{\mathcal H}_{1-\overline{s}}\,\mathcal J^{-1}
  \]
  （酉共轭意义下）；
  \item 其 Fredholm 行列式满足
  \[
  \det\nolimits_{\mathrm F}\!\bigl(I-\widetilde{\mathcal H}_s\bigr)=C\cdot \xi(s)
  \qquad(C\neq 0),
  \]
  其中 \(\xi\) 为完成的 Riemann \(\xi\)-函数。
\end{enumerate}
在上述正规化成立时，Riemann 假设等价于：在 \(\Re(s)=\tfrac12\) 上，\(\widetilde{\mathcal H}_s\) 的散射谱为纯酉（无非酉谱泄漏）。
\end{conjecture}
