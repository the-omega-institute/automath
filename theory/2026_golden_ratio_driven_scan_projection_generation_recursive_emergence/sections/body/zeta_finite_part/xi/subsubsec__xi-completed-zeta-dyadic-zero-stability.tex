\subsubsection{完成 \texorpdfstring{$\zeta$}{zeta} 核的二进递归、余项相位律与零点稳定}\label{subsubsec:xi-completed-zeta-dyadic-zero-stability}
本小节给出一条与 \S\ref{subsubsec:xi-theta-kernel-dyadic-scale-decomposition} 互补的闭环：在完成 \(\zeta\) 核上直接施行二进层级截断，以显式余项预算驱动零点稳定性与后验证书，并以 Hurwitz 极限机制给出一个可审计的 RH 归约条件。

\paragraph{完成核的二进层级与余项预算}
定义
\begin{equation}\label{eq:xi-completed-zeta-lambda-def}
\Lambda(s):=\pi^{-s/2}\Gamma\!\left(\frac{s}{2}\right)\zeta(s),
\qquad
\vartheta_{\mathrm J}(\mathrm{i}x):=\sum_{n\in\ZZ}e^{-\pi n^2x}\quad(x>0).
\end{equation}

\begin{theorem}[二进尺度递归的完成核分解与超指数余项界]\label{thm:xi-completed-zeta-dyadic-remainder-budget}
在解析延拓意义下，\(\Lambda\) 满足
\begin{equation}\label{eq:xi-completed-zeta-lambda-symmetric-integral}
\Lambda(s)
=-\frac1s-\frac1{1-s}
+\frac12\int_{1}^{\infty}
\Bigl(x^{s/2}+x^{(1-s)/2}\Bigr)\bigl(\vartheta_{\mathrm J}(\mathrm{i}x)-1\bigr)\,\frac{\dd x}{x}.
\end{equation}
对 \(K\in\NN\) 定义二进层级截断
\begin{equation}\label{eq:xi-completed-zeta-lambdaK-def}
\Lambda_K(s):=
-\frac1s-\frac1{1-s}
+\frac12\sum_{k=0}^{K-1}\int_{2^k}^{2^{k+1}}
\Bigl(x^{s/2}+x^{(1-s)/2}\Bigr)\bigl(\vartheta_{\mathrm J}(\mathrm{i}x)-1\bigr)\,\frac{\dd x}{x},
\end{equation}
并记增量
\begin{equation}\label{eq:xi-completed-zeta-lambdaK-increment}
\Delta_K(s):=
\frac12\int_{2^K}^{2^{K+1}}
\Bigl(x^{s/2}+x^{(1-s)/2}\Bigr)\bigl(\vartheta_{\mathrm J}(\mathrm{i}x)-1\bigr)\,\frac{\dd x}{x},
\end{equation}
则 \(\Lambda_{K+1}(s)=\Lambda_K(s)+\Delta_K(s)\)。
余项为
\begin{equation}\label{eq:xi-completed-zeta-lambdaK-remainder}
R_K(s):=\Lambda(s)-\Lambda_K(s)
=\frac12\int_{2^K}^{\infty}
\Bigl(x^{s/2}+x^{(1-s)/2}\Bigr)\bigl(\vartheta_{\mathrm J}(\mathrm{i}x)-1\bigr)\,\frac{\dd x}{x}.
\end{equation}
并且在临界线 \(s=\frac12+\mathrm{i}t\) 上，对任意 \(t\in\RR\) 与 \(K\ge 0\) 有
\begin{equation}\label{eq:xi-completed-zeta-critical-remainder-bound}
\left|R_K\!\left(\frac12+\mathrm{i}t\right)\right|
\le \frac{4}{\pi}\,2^{-3K/4}e^{-\pi 2^K}.
\end{equation}
令
\begin{equation}\label{eq:xi-completed-zeta-xiK-def}
\xi(s):=\frac12\,s(s-1)\Lambda(s),
\qquad
\xi_K(s):=\frac12\,s(s-1)\Lambda_K(s),
\end{equation}
则
\begin{equation}\label{eq:xi-completed-zeta-xiK-error-budget}
\left|\xi\!\left(\frac12+\mathrm{i}t\right)-\xi_K\!\left(\frac12+\mathrm{i}t\right)\right|
\le
\frac{\left|\left(\frac12+\mathrm{i}t\right)\left(-\frac12+\mathrm{i}t\right)\right|}{2}
\cdot
\frac{4}{\pi}\,2^{-3K/4}e^{-\pi 2^K}.
\end{equation}
\end{theorem}
\begin{proof}
表示 \eqref{eq:xi-completed-zeta-lambda-symmetric-integral} 为 Jacobi \(\vartheta\) 变换与 Mellin 表示的标准对称化形式（参见 \cite{Titchmarsh1986RiemannZeta} 的等价写法）。
\eqref{eq:xi-completed-zeta-lambdaK-def}--\eqref{eq:xi-completed-zeta-lambdaK-remainder} 由积分区间按二进壳分割直接得到。
\par
当 \(x\ge 1\) 时，
\[
\vartheta_{\mathrm J}(\mathrm{i}x)-1
=2\sum_{n\ge1}e^{-\pi n^2x}
\le 2\sum_{n\ge1}e^{-\pi nx}
=\frac{2e^{-\pi x}}{1-e^{-\pi x}}
\le 4e^{-\pi x}.
\]
在 \(s=\frac12+\mathrm{i}t\) 上有
\[
\left|x^{s/2}+x^{(1-s)/2}\right|
\le 2x^{1/4}.
\]
代入 \eqref{eq:xi-completed-zeta-lambdaK-remainder} 得
\[
\left|R_K\!\left(\frac12+\mathrm{i}t\right)\right|
\le 4\int_{2^K}^{\infty}x^{-3/4}e^{-\pi x}\,\dd x
\le
4\cdot 2^{-3K/4}\int_{2^K}^{\infty}e^{-\pi x}\,\dd x
=\frac{4}{\pi}\,2^{-3K/4}e^{-\pi 2^K},
\]
即 \eqref{eq:xi-completed-zeta-critical-remainder-bound}。
\eqref{eq:xi-completed-zeta-xiK-error-budget} 由 \eqref{eq:xi-completed-zeta-xiK-def} 与
\(
\xi-\xi_K=\frac12 s(s-1)R_K
\)
直接推出。
\end{proof}

\begin{theorem}[余项的相位律与首项振荡主导]\label{thm:xi-completed-zeta-remainder-phase-law}
对任意 \(s\in\CC\)，余项 \(R_K\) 具有不完全伽马函数展开
\begin{equation}\label{eq:xi-completed-zeta-remainder-incomplete-gamma}
R_K(s)=\sum_{n=1}^{\infty}\Biggl[
(\pi n^2)^{-s/2}\Gamma\!\left(\frac{s}{2},\pi n^2 2^K\right)
+
(\pi n^2)^{-(1-s)/2}\Gamma\!\left(\frac{1-s}{2},\pi n^2 2^K\right)
\Biggr].
\end{equation}
并且对任意固定 \(T>0\)，当 \(|t|\le T\) 且 \(K\to\infty\) 时，一致成立
\begin{equation}\label{eq:xi-completed-zeta-remainder-phase-asymptotics}
\begin{aligned}
R_K\!\left(\frac12+\mathrm{i}t\right)
=\;&
\frac{2}{\pi}\,2^{-3K/4}e^{-\pi 2^K}
\cos\!\left(\frac{tK\log2}{2}\right)\\
&+O_T\!\left(2^{-7K/4}e^{-\pi2^K}\right)
+O\!\left(2^{-3K/4}e^{-4\pi2^K}\right).
\end{aligned}
\end{equation}
因此在层级 \(K\) 上，余项主项具有确定角频率 \(\frac{t\log2}{2}\)，振幅以 \(e^{-\pi2^K}\) 超指数衰减。
\end{theorem}
\begin{proof}
将
\(
\vartheta_{\mathrm J}(\mathrm{i}x)-1
=2\sum_{n\ge1}e^{-\pi n^2x}
\)
代入 \eqref{eq:xi-completed-zeta-lambdaK-remainder} 并逐项积分，得到
\[
\int_{2^K}^{\infty}x^{s/2-1}e^{-\pi n^2x}\,\dd x
=(\pi n^2)^{-s/2}\Gamma\!\left(\frac{s}{2},\pi n^2 2^K\right),
\]
及其对称项，从而得到 \eqref{eq:xi-completed-zeta-remainder-incomplete-gamma}。
\par
取 \(s=\frac12+\mathrm{i}t\)。对
\(
\Gamma(a,x)=x^{a-1}e^{-x}(1+O_a(x^{-1}))
\)
应用于
\(
a=\frac14\pm\frac{\mathrm{i}t}{2},\ x=\pi n^2 2^K
\)，
\(n=1\) 项给出主导贡献
\[
\frac{1}{\pi}\,2^{-3K/4}e^{-\pi2^K}
\left(e^{\mathrm{i}\frac{tK\log2}{2}}+e^{-\mathrm{i}\frac{tK\log2}{2}}\right),
\]
即主项
\(
\frac{2}{\pi}2^{-3K/4}e^{-\pi2^K}\cos(\frac{tK\log2}{2})
\)。
\(n=1\) 的下一阶 \(x^{-1}\) 项给出
\(
O_T(2^{-7K/4}e^{-\pi2^K})
\)，
而 \(n\ge2\) 项整体由
\(
O(2^{-3K/4}e^{-4\pi2^K})
\)
控制。综上得到 \eqref{eq:xi-completed-zeta-remainder-phase-asymptotics}。
\end{proof}

\paragraph{误差预算到零点几何的稳定传输}
\begin{theorem}[二进递归下简单零点的超指数稳定性]\label{thm:xi-completed-zeta-simple-zero-stability}
设 \(\rho\) 为 \(\xi\) 的单零点，即 \(\xi(\rho)=0\)、\(\xi'(\rho)\neq0\)。
取 \(r>0\) 使闭圆盘 \(\overline{D}(\rho,r)\) 内除 \(\rho\) 外无其他 \(\xi\)-零点。
则存在 \(K_0(\rho,r)\) 使得对所有 \(K\ge K_0\)：
\begin{enumerate}
  \item \(\xi_K\) 在 \(D(\rho,r)\) 内恰有一个零点 \(\rho_K\)（计重数）；
  \item \(\rho_K\to\rho\) 且满足
  \begin{equation}\label{eq:xi-completed-zeta-simple-zero-stability-rate}
  |\rho_K-\rho|
  =O_{\rho,r}\!\left(2^{-3K/4}e^{-\pi2^K}\right).
  \end{equation}
\end{enumerate}
\end{theorem}
\begin{proof}
记
\[
m_{\rho,r}:=\min_{|s-\rho|=r}|\xi(s)|>0.
\]
由 \eqref{eq:xi-completed-zeta-lambdaK-remainder} 在紧集上的一致收敛与定理 \ref{thm:xi-completed-zeta-dyadic-remainder-budget} 给出的显式衰减，存在 \(K_0\) 使
\[
\max_{|s-\rho|=r}|\xi(s)-\xi_K(s)|<m_{\rho,r}\qquad(K\ge K_0).
\]
据 Rouch\'e 定理，\(\xi\) 与 \(\xi_K\) 在 \(D(\rho,r)\) 内零点数相同，故 \(\xi_K\) 在该圆盘内恰有一个零点 \(\rho_K\)。
再由 Hurwitz 定理与局部反函数估计，可得 \(\rho_K\to\rho\)；
并且位移由边界误差 \(\max_{|s-\rho|=r}|\xi-\xi_K|\) 线性控制，结合上述余项衰减即得 \eqref{eq:xi-completed-zeta-simple-zero-stability-rate}。
\end{proof}

\begin{proposition}[带误差预算的临界线后验零点证书]\label{prop:xi-completed-zeta-posterior-zero-certificate}
记
\[
\Xi(t):=\xi\!\left(\frac12+\mathrm{i}t\right),\qquad
\Xi_K(t):=\xi_K\!\left(\frac12+\mathrm{i}t\right).
\]
给定 \(t_0\in\RR\)、\(r>0\)，令 \(I=[t_0-r,t_0+r]\)。
若存在 \(M>0\) 与 \(\varepsilon>0\) 满足
\begin{equation}\label{eq:xi-completed-zeta-posterior-derivative-lower}
\inf_{t\in I}\left|\Xi_K'(t)\right|\ge M,
\end{equation}
\begin{equation}\label{eq:xi-completed-zeta-posterior-error-budget}
\sup_{t\in I}\left|\Xi(t)-\Xi_K(t)\right|\le \varepsilon,
\qquad
\left|\Xi_K(t_0)\right|\le Mr-\varepsilon,
\end{equation}
则 \(\Xi\) 在 \(I\) 内存在且仅存在一个零点 \(t_\ast\)，并满足
\begin{equation}\label{eq:xi-completed-zeta-posterior-zero-envelope}
\left|t_\ast-t_0\right|
\le
\frac{|\Xi_K(t_0)|+\varepsilon}{M}.
\end{equation}
其中 \(\varepsilon\) 可由 \eqref{eq:xi-completed-zeta-xiK-error-budget} 作为层级 \(K\) 的显式函数给出，从而得到 \(K\mapsto\) 零点包络半径的闭式预算。
\end{proposition}
\begin{proof}
由 \eqref{eq:xi-completed-zeta-posterior-derivative-lower}，\(\Xi_K\) 在 \(I\) 上严格单调，且
\[
|\Xi_K(t)-\Xi_K(t_0)|\ge M|t-t_0|.
\]
当 \(|t-t_0|=r\) 时，
\[
|\Xi(t)|\ge |\Xi_K(t)|-\varepsilon
\ge |\Xi_K(t_0)|+Mr-\varepsilon>0.
\]
因此端点附近不存在额外过零；再由 \(\Xi=\Xi_K+(\Xi-\Xi_K)\) 与误差预算 \eqref{eq:xi-completed-zeta-posterior-error-budget}，可得 \(I\) 内恰有一个零点 \(t_\ast\)。
并且
\[
0=|\Xi(t_\ast)|
\ge |\Xi_K(t_\ast)|-\varepsilon
\ge |\Xi_K(t_0)|-M|t_\ast-t_0|-\varepsilon,
\]
整理即得 \eqref{eq:xi-completed-zeta-posterior-zero-envelope}。
\end{proof}

\paragraph{Dyadic--LP 闭包与 RH 归约}
\begin{conjecture}[Dyadic--LP 闭包猜想]\label{conj:xi-completed-zeta-dyadic-lp}
对每个 \(K\ge 1\)，整函数 \(t\mapsto \Xi_K(t)\) 的全部零点均为实数；
等价地，\(\xi_K(s)\) 的全部非平凡零点均位于直线 \(\Re(s)=\frac12\)。
\end{conjecture}

\begin{theorem}[Dyadic--LP 闭包蕴含 RH]\label{thm:xi-completed-zeta-dyadic-lp-implies-rh}
\leanverified{paper\_xi\_completed\_zeta\_dyadic\_lp\_implies\_rh}
若猜想 \ref{conj:xi-completed-zeta-dyadic-lp} 对所有 \(K\) 成立，则黎曼猜想成立。
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-completed-zeta-lambdaK-remainder} 在任意紧集上的一致收敛，\(\xi_K\to\xi\) 在紧集上一致。
取 \(\xi\) 的任一非平凡零点 \(\rho\)。
由 Hurwitz 定理，存在 \(K\to\infty\) 的零点序列 \(\rho_K\) 使 \(\rho_K\to\rho\) 且 \(\xi_K(\rho_K)=0\)。
若猜想 \ref{conj:xi-completed-zeta-dyadic-lp} 成立，则每个 \(\rho_K\) 满足 \(\Re(\rho_K)=\frac12\)，极限后得到 \(\Re(\rho)=\frac12\)。
因此 \(\xi\) 的全部非平凡零点均在临界线上，即 RH 成立。
\end{proof}
