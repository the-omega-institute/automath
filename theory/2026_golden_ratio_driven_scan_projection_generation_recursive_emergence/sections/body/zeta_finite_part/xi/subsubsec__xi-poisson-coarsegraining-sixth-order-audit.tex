\paragraph{六阶信息签名与双测度对比的普适律}\label{para:xi-poisson-sixth-order-audit-laws}
定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 已将 Cauchy--Poisson 粗粒化下的相对熵尾部推进到 \(t^{-6}\) 并给出显式系数组合。
下述结果把该六阶系数的符号与取等情形完全刻画为中心矩的代数约束，并给出两测度比较时的主导信息源层级，从而形成可离线核验的普适判别式。

\begin{corollary}[相对熵二项锐渐近的极限等价式]\label{cor:xi-cauchy-poisson-kl-two-term-limit}
\leanverified{paper\_xi\_cauchy\_poisson\_kl\_two\_term\_limit}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，记
\[
C_6:=\frac{3\mu_3^2}{32}-\frac{\sigma^2\mu_4}{8}+\frac{\sigma^6}{64}.
\]
则当 \(t\to\infty\) 时
\[
D_{\mathrm{KL}}(h_t\mid g_t)=\frac{\sigma^4}{8t^4}+\frac{C_6}{t^6}+O(t^{-8}),
\]
并且
\[
\lim_{t\to\infty}t^6\!\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{\sigma^4}{8t^4}\right)=C_6.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的展开式直接得到。
\end{proof}

\begin{corollary}[Poisson--Cauchy 粗粒化的熵曲率不变量]\label{cor:xi-cauchy-poisson-entropy-curvature-invariant}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，定义
\[
\mathfrak K(\tilde\nu)
:=\lim_{t\to\infty}
t^6\!\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{\sigma^4}{8t^4}\right).
\]
则该极限存在且
\[
\boxed{\;
\mathfrak K(\tilde\nu)
=\frac{3\mu_3^2}{32}-\frac{\sigma^2\mu_4}{8}+\frac{\sigma^6}{64}=C_6.\;}
\]
因此，“无毛”二阶主项 \(\sigma^4/(8t^4)\) 之后的首个信息论可见层级精确位于 \(t^{-6}\)，并由 \((\mu_3,\mu_4)\) 显式编码。
进一步，若 \(\tilde\nu\) 关于 \(\bar\gamma\) 对称（等价于 \(\mu_3=0\)），则
\[
\mathfrak K(\tilde\nu)=\frac{\sigma^6}{64}-\frac{\sigma^2\mu_4}{8},
\qquad
\mu_4=\frac{\sigma^4}{8}-\frac{8}{\sigma^2}\,\mathfrak K(\tilde\nu).
\]
\end{corollary}
\begin{proof}
由推论 \ref{cor:xi-cauchy-poisson-kl-two-term-limit}，
\[
\lim_{t\to\infty}
t^6\!\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{\sigma^4}{8t^4}\right)
=C_6
=\frac{3\mu_3^2}{32}-\frac{\sigma^2\mu_4}{8}+\frac{\sigma^6}{64},
\]
从而得到第一式。
若 \(\mu_3=0\)，则上式退化为
\(
\mathfrak K=\sigma^6/64-\sigma^2\mu_4/8
\)，
对 \(\mu_4\) 代数消元即得反演公式。证毕。
\end{proof}

\begin{proposition}[六阶信息签名的普适负号与二点刚性]\label{prop:xi-cauchy-poisson-sixth-signature-negativity-rigidity}
\leanverified{paper\_xi\_cauchy\_poisson\_sixth\_signature\_negativity\_rigidity}
沿用定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的设定，并记标准化偏度与峰度
\[
\gamma_1:=\frac{\mu_3}{\sigma^3},
\qquad
\beta_2:=\frac{\mu_4}{\sigma^4},
\]
以及六阶信息签名
\[
\mathcal J:=1-8\beta_2+6\gamma_1^2.
\]
则 \(\mathcal J\) 满足普适上界
\begin{equation}\label{eq:xi-cauchy-poisson-sixth-signature-upper-bound}
\boxed{\;
\mathcal J\le -7-2\gamma_1^2\le -7.\;}
\end{equation}
因此，定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 中 \(t^{-6}\) 项系数
\[
C_6
:=\frac{1}{64}\Bigl(\sigma^6-8\sigma^2\mu_4+6\mu_3^2\Bigr)
=\frac{\sigma^6}{64}\,\mathcal J
\]
严格为负并满足
\begin{equation}\label{eq:xi-cauchy-poisson-sixth-coefficient-negative}
\boxed{\;
C_6\le -\frac{\mu_3^2}{32}-\frac{7\sigma^6}{64}<0.\;}
\end{equation}
特别地，\(C_6\le -\frac{7\sigma^6}{64}\)。
并且该后者上界取等当且仅当 \(\tilde\nu\) 为方差固定类中的对称二点分布，即在平移回均值 \(\bar\gamma\) 后
\begin{equation}\label{eq:xi-cauchy-poisson-sixth-rigidity-two-point}
\boxed{\;
\tilde\nu=\frac12\delta_{\bar\gamma-a}+\frac12\delta_{\bar\gamma+a}
\quad\text{对某个 }a>0\text{成立}. \;}
\end{equation}
等价地，在给定 \((\bar\gamma,\sigma^2)\) 的类中，六阶系数 \(C_6\) 的上确界为 \(-7\sigma^6/64\)，且仅当 \(\tilde\nu=\frac12\delta_{\bar\gamma-\sigma}+\frac12\delta_{\bar\gamma+\sigma}\) 时达到。
\end{proposition}
\begin{proof}
令中心化随机变量 \(X:=\gamma-\bar\gamma\)，则 \(\mathbb E[X]=0\)、\(\mathbb E[X^2]=\sigma^2\)、\(\mathbb E[X^3]=\mu_3\)、\(\mathbb E[X^4]=\mu_4\)。
考虑 \((1,X,X^2)\) 的 Gram 矩阵
\[
M:=\mathbb E
\begin{pmatrix}
1\\ X\\ X^2
\end{pmatrix}
\begin{pmatrix}
1&X&X^2
\end{pmatrix}
=
\begin{pmatrix}
1&0&\sigma^2\\
0&\sigma^2&\mu_3\\
\sigma^2&\mu_3&\mu_4
\end{pmatrix}\succeq 0.
\]
故 \(\det M\ge 0\)，即
\[
\sigma^2\mu_4-\mu_3^2-\sigma^6\ge 0
\quad\Longleftrightarrow\quad
\beta_2\ge \gamma_1^2+1.
\]
代入 \(\mathcal J=1-8\beta_2+6\gamma_1^2\) 得
\[
\mathcal J\le 1-8(\gamma_1^2+1)+6\gamma_1^2=-7-2\gamma_1^2,
\]
两边乘以 \(\sigma^6/64\) 并代入 \(\gamma_1=\mu_3/\sigma^3\)，即
\[
C_6\le -\frac{\mu_3^2}{32}-\frac{7\sigma^6}{64},
\]
从而得到 \eqref{eq:xi-cauchy-poisson-sixth-signature-upper-bound} 与 \eqref{eq:xi-cauchy-poisson-sixth-coefficient-negative}。
\par
若 \(C_6=-\frac{7\sigma^6}{64}\)，则由 \(\mathcal J=-7\) 可知上式链中必须处处取等，故 \(\gamma_1=0\) 且 \(\beta_2=\gamma_1^2+1=1\)。
由 \(\beta_2=1\) 得 \(\mu_4=\sigma^4\)，于是 \(\mathbb E[(X^2-\sigma^2)^2]=\mu_4-\sigma^4=0\)，从而 \(X^2=\sigma^2\) a.s.，即 \(X\in\{\pm\sigma\}\) a.s.。
再由 \(\mathbb E[X]=0\) 得两点权重相等，故 \(X=\pm\sigma\) 等概率。
平移回 \(\bar\gamma\) 并记 \(a:=\sigma>0\)，得到 \eqref{eq:xi-cauchy-poisson-sixth-rigidity-two-point}。证毕。
\end{proof}

\begin{corollary}[Pearson \texorpdfstring{$\chi^2$}{chi2} 与平方 Hellinger 的六阶显式系数]\label{cor:xi-cauchy-poisson-chi2-hellinger-sixth-order}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下：
\begin{enumerate}
  \item 取 \(f(z)=(z-1)^2\)（即 Pearson \(\chi^2\) 散度，\(f''(1)=2,f'''(1)=0\)），则
  \[
  \chi^2(h_t\mid g_t)
  =\frac{\sigma^4}{4t^4}
  +\frac{1}{t^6}\left(\frac{3\mu_3^2}{16}-\frac{\sigma^2\mu_4}{4}\right)
  +O(t^{-8}).
  \]
  \item 取 \(f(z)=(\sqrt z-1)^2\)（平方 Hellinger 散度，\(f''(1)=\frac12,f'''(1)=-\frac34\)），则
  \[
  H^2(h_t,g_t)
  =\frac{\sigma^4}{16t^4}
  +\frac{1}{t^6}\left(
  \frac{3\mu_3^2}{64}
  -\frac{\sigma^2\mu_4}{16}
  +\frac{3\sigma^6}{256}\right)
  +O(t^{-8}).
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
把上述两组 \(\bigl(f''(1),f'''(1)\bigr)\) 直接代入命题
\ref{prop:xi-cauchy-poisson-fdiv-sixth-order-expansion}
的通式 \eqref{eq:xi-cauchy-poisson-fdiv-sixth-order-expansion} 即得。证毕。
\end{proof}

\begin{corollary}[四次重整化相对熵的最终单调趋近]\label{cor:xi-cauchy-poisson-kl-renormalized-monotone-approach}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，若 \(\sigma^2>0\)，则
\[
t^4D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{\sigma^4}{8}+\frac{C_6}{t^2}+O(t^{-4}),
\qquad
C_6<0.
\]
其中 \(C_6\) 如命题 \ref{prop:xi-cauchy-poisson-sixth-signature-negativity-rigidity} 所定义。
从而存在 \(t_0>0\)，使得对一切 \(t\ge t_0\) 成立
\[
\frac{\dd}{\dd t}\!\Bigl(t^4D_{\mathrm{KL}}(h_t\mid g_t)\Bigr)>0,
\qquad
t^4D_{\mathrm{KL}}(h_t\mid g_t)<\frac{\sigma^4}{8}.
\]
即四次重整化相对熵从下方趋近其极限值 \(\sigma^4/8\)，并在充分大尺度上严格单调递增。
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 与命题 \ref{prop:xi-cauchy-poisson-sixth-signature-negativity-rigidity} 可得
\(
t^4D_{\mathrm{KL}}=\sigma^4/8+C_6t^{-2}+O(t^{-4})
\)
且 \(C_6<0\)。
于是
\[
\frac{\dd}{\dd t}\!\Bigl(t^4D_{\mathrm{KL}}\Bigr)
=-2C_6t^{-3}+O(t^{-5}),
\]
右端在 \(t\) 充分大时严格为正，得到最终单调性。
同理由 \(C_6t^{-2}+O(t^{-4})<0\) 得到从下趋近结论。证毕。
\end{proof}

\begin{proposition}[三阶光滑 \texorpdfstring{$f$}{f}-散度的六阶普适展开]\label{prop:xi-cauchy-poisson-fdiv-sixth-order-expansion}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，
令 \(f:(0,\infty)\to\RR\) 为凸函数，满足 \(f(1)=0\)，并且在 \(1\) 的邻域三阶可微且 \(f''(1),f'''(1)\in\RR\) 有限。
定义 Csisz\'ar \texorpdfstring{$f$}{f}-散度
\[
D_f(h\mid g):=\int_{\RR} g(x)\,f\!\left(\frac{h(x)}{g(x)}\right)\,\dd x.
\]
则当 \(t\to\infty\) 时成立仅含偶次幂的六阶展开（特别地，\(t^{-5}\) 项恒等消失）
\begin{equation}\label{eq:xi-cauchy-poisson-fdiv-sixth-order-expansion}
\boxed{\;
D_f(h_t\mid g_t)
=\frac{f''(1)\sigma^4}{8t^4}
\;+\;\frac{1}{t^6}\left[
f''(1)\left(\frac{3\mu_3^2}{32}-\frac{\sigma^2\mu_4}{8}\right)
-\frac{f'''(1)\sigma^6}{64}
\right]
\;+\;O(t^{-8}).\;}
\end{equation}
\end{proposition}
\begin{proof}[证明要点]
沿用定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 证明中的尺度置换 \(x=ty\) 与密度比记号 \(r_t=h_t/g_t=1+\delta_t\)。
由 \(\int_{\RR}g_t\,\delta_t=\int(h_t-g_t)=0\) 消去一次项，并对 \(f(1+u)\) 在 \(u=0\) 处作三阶 Taylor 展开，得到
\[
D_f(h_t\mid g_t)
=\frac{f''(1)}{2}\int_{\RR} g_t\,\delta_t^2\,\dd x
\;+\;\frac{f'''(1)}{6}\int_{\RR} g_t\,\delta_t^3\,\dd x
\;+\;O\!\left(\int_{\RR} g_t\,|\delta_t|^4\right).
\]
定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的推导给出 \(\delta_t=O(t^{-2})\) 且
\[
\int_{\RR} g_t\,\delta_t^2\,\dd x=\frac{\sigma^4}{4t^4}
\;+\;\frac{1}{t^6}\left(\frac{3\mu_3^2}{16}-\frac{\sigma^2\mu_4}{4}\right)
\;+\;O(t^{-8}),
\qquad
\int_{\RR} g_t\,\delta_t^3\,\dd x=-\frac{3\sigma^6}{32t^6}+O(t^{-8}),
\]
并且 \(\int g_t|\delta_t|^4=O(t^{-8})\)。
代回并化简即得 \eqref{eq:xi-cauchy-poisson-fdiv-sixth-order-expansion}。证毕。
\end{proof}

\begin{corollary}[六阶系数负号导致的主项上界与从下趋近]\label{cor:xi-cauchy-poisson-kl-approach-below}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，若 \(\sigma^2>0\)，则
\begin{equation}\label{eq:xi-cauchy-poisson-kl-approach-below}
\boxed{\;
\lim_{t\to\infty} t^6\left(\frac{\sigma^4}{8t^4}-D_{\mathrm{KL}}(h_t\mid g_t)\right)
=-C_6\ \ge\ \frac{7\sigma^6}{64}\; }
\end{equation}
其中 \(C_6\) 如命题 \ref{prop:xi-cauchy-poisson-sixth-signature-negativity-rigidity} 所定义。
特别地，存在 \(t_0\) 使对一切 \(t\ge t_0\) 成立
\[
D_{\mathrm{KL}}(h_t\mid g_t)<\frac{\sigma^4}{8t^4}.
\]
因此相对熵在大尺度处从下方趋近其 \(t^{-4}\) 主导项上界。
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics}，
\[
D_{\mathrm{KL}}(h_t\mid g_t)=\frac{\sigma^4}{8t^4}+\frac{C_6}{t^6}+O(t^{-8}),
\]
从而
\(
t^6\bigl(\frac{\sigma^4}{8t^4}-D_{\mathrm{KL}}(h_t\mid g_t)\bigr)=-C_6+O(t^{-2})
\)
并收敛到 \(-C_6\)。
由命题 \ref{prop:xi-cauchy-poisson-sixth-signature-negativity-rigidity} 得 \(C_6\le -7\sigma^6/64<0\)，从而 \eqref{eq:xi-cauchy-poisson-kl-approach-below} 成立。
最后由于误差项为 \(O(t^{-8})\)，存在 \(t_0\) 使 \(t\ge t_0\) 时
\(
\frac{C_6}{t^6}+O(t^{-8})<0
\)，
代回即得 \(D_{\mathrm{KL}}(h_t\mid g_t)<\sigma^4/(8t^4)\)。证毕。
\end{proof}

\begin{corollary}[耗散--熵比值的自相似极限与二阶缺陷]\label{cor:xi-cauchy-poisson-fisher-kl-selfsimilar-defect}
\leanverified{paper\_xi\_cauchy\_poisson\_fisher\_kl\_selfsimilar\_defect}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，若 \(\sigma^2>0\)，则
\[
\mathcal I_1(h_t\mid g_t)
=\frac{\sigma^4}{2t^5}+\frac{6C_6}{t^7}+O(t^{-9}),
\]
并且成立极限律
\[
\lim_{t\to\infty}\frac{t\,\mathcal I_1(h_t\mid g_t)}{D_{\mathrm{KL}}(h_t\mid g_t)}=4,
\]
\[
\lim_{t\to\infty}t^7\!\left(
\mathcal I_1(h_t\mid g_t)-\frac{4}{t}D_{\mathrm{KL}}(h_t\mid g_t)
\right)=2C_6.
\]
\end{corollary}
\begin{proof}
第一式由推论 \ref{cor:xi-cauchy-poisson-fisher-sharp-asymptotics} 与
\(
C_6=\frac{1}{64}(\sigma^6-8\sigma^2\mu_4+6\mu_3^2)
\)
同一化得到。
再与
\(
D_{\mathrm{KL}}=\frac{\sigma^4}{8t^4}+\frac{C_6}{t^6}+O(t^{-8})
\)
联立，直接计算
\[
\frac{t\,\mathcal I_1}{D_{\mathrm{KL}}}
=\frac{\sigma^4/2+6C_6t^{-2}+O(t^{-4})}{\sigma^4/8+C_6t^{-2}+O(t^{-4})}\to 4,
\]
以及
\[
\mathcal I_1-\frac{4}{t}D_{\mathrm{KL}}
=\frac{2C_6}{t^7}+O(t^{-9}),
\]
即得两条极限式。证毕。
\end{proof}

\begin{theorem}[两测度比较的方差差主导律：\texorpdfstring{$t^{-4}$}{t^-4} 的唯一信息源]\label{thm:xi-poisson-two-measure-variance-gap}
设 \(\tilde\nu,\hat\nu\) 为 \(\RR\) 上概率测度，具有相同均值
\[
\bar\gamma:=\int_{\RR}\gamma\,\dd\tilde\nu(\gamma)=\int_{\RR}\gamma\,\dd\hat\nu(\gamma),
\]
并满足三阶中心绝对矩有限：
\[
\int_{\RR}|\gamma-\bar\gamma|^3\,\dd\tilde\nu(\gamma)<\infty,
\qquad
\int_{\RR}|\gamma-\bar\gamma|^3\,\dd\hat\nu(\gamma)<\infty.
\]
令 Poisson 粗粒化密度
\[
\tilde h_t(x):=\int_{\RR}\frac{1}{\pi}\frac{t}{(x-\gamma)^2+t^2}\,\dd\tilde\nu(\gamma),
\qquad
\hat h_t(x):=\int_{\RR}\frac{1}{\pi}\frac{t}{(x-\gamma)^2+t^2}\,\dd\hat\nu(\gamma),
\]
并记
\(
\Delta\sigma^2:=\Var(\tilde\nu)-\Var(\hat\nu)
\)。
对任意凸函数 \(f:(0,\infty)\to\RR\) 满足 \(f(1)=0\) 且 \(f''(1)\in(0,\infty)\)，有两测度 \texorpdfstring{$f$}{f}-散度的普适锐渐近
\begin{equation}\label{eq:xi-poisson-two-measure-variance-gap}
\boxed{\;
D_f(\tilde h_t\mid \hat h_t)\sim \frac{f''(1)}{8}\,\frac{(\Delta\sigma^2)^2}{t^4},
\qquad t\to\infty.\;}
\end{equation}
特别地，取 \(f(u)=u\log u\) 得
\(
D_{\mathrm{KL}}(\tilde h_t\mid \hat h_t)\sim \frac{(\Delta\sigma^2)^2}{8t^4}
\)。
\end{theorem}
\begin{proof}[证明要点]
对 \(\tilde\nu\) 与 \(\hat\nu\) 分别应用命题 \ref{con:xi-poisson-coarsegraining-second-order-normal-form}，得到
\[
\tilde h_t=g_t+\tilde\sigma^2 r_t+\tilde\varepsilon_t,
\qquad
\hat h_t=g_t+\hat\sigma^2 r_t+\hat\varepsilon_t,
\qquad
\|\tilde\varepsilon_t\|_{L^1}+\|\hat\varepsilon_t\|_{L^1}=O(t^{-3}),
\]
其中 \(g_t(x)=\frac{1}{\pi}\frac{t}{(x-\bar\gamma)^2+t^2}\)，且 \(r_t=\tfrac12\partial_x^2 g_t\)（见命题 \ref{prop:xi-poisson-tv-kl-second-order} 的记号）。
于是
\[
\frac{\tilde h_t}{\hat h_t}
=1+\Delta\sigma^2\,\frac{r_t}{g_t}+o(t^{-2})
\quad\text{于 }L^2(g_t\,\dd x)\text{ 意义下}.
\]
令 \(\delta_t:=\tilde h_t/\hat h_t-1\)，则 \(\delta_t=\frac{\Delta\sigma^2}{t^2}A\!\left(\frac{x-\bar\gamma}{t}\right)+o(t^{-2})\)，其中
\(
A(y)=\frac{3y^2-1}{(1+y^2)^2}
\)
与定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 一致，且 \(\int \hat h_t\delta_t=\int(\tilde h_t-\hat h_t)=0\)。
对 \(f(1+u)\) 在 \(u=0\) 处作二阶 Taylor 展开并积分，得到
\[
D_f(\tilde h_t\mid \hat h_t)
=\frac{f''(1)}{2}\int_{\RR}\hat h_t\,\delta_t^2\,\dd x+o(t^{-4}).
\]
将 \(\hat h_t\) 替换为 \(g_t\) 不影响 \(t^{-4}\) 主项，并作代换 \(y=(x-\bar\gamma)/t\)，得到
\[
\int_{\RR}\hat h_t\,\delta_t^2\,\dd x
=\frac{(\Delta\sigma^2)^2}{t^4}\cdot \frac{1}{\pi}\int_{\RR}\frac{A(y)^2}{(1+y^2)}\,\dd y
\;+\;o(t^{-4}).
\]
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的常数计算 \(\frac{1}{\pi}\int gA^2=\frac14\)，代回即得 \eqref{eq:xi-poisson-two-measure-variance-gap}。证毕。
\end{proof}

\begin{corollary}[方差对齐后的下一层信息：三阶中心矩差的 \texorpdfstring{$t^{-6}$}{t^-6} 主导律]\label{cor:xi-poisson-two-measure-third-moment-gap}
在定理 \ref{thm:xi-poisson-two-measure-variance-gap} 的设定下，
进一步假设两测度不仅均值相同，而且方差相同且四阶矩有限：
\[
\Var(\tilde\nu)=\Var(\hat\nu)=\sigma^2>0,
\qquad
\int_{\RR}|\gamma-\bar\gamma|^4\,\dd\tilde\nu(\gamma)<\infty,\quad
\int_{\RR}|\gamma-\bar\gamma|^4\,\dd\hat\nu(\gamma)<\infty.
\]
记三阶中心矩
\[
\mu_3:=\int_{\RR}(\gamma-\bar\gamma)^3\,\dd\tilde\nu(\gamma),
\qquad
\hat\mu_3:=\int_{\RR}(\gamma-\bar\gamma)^3\,\dd\hat\nu(\gamma).
\]
则对任意凸 \(f\) 满足 \(f(1)=0\) 且 \(f''(1)\in(0,\infty)\)，有更快的普适锐渐近
\begin{equation}\label{eq:xi-poisson-two-measure-third-moment-gap}
\boxed{\;
D_f(\tilde h_t\mid \hat h_t)\sim \frac{3f''(1)}{32}\,\frac{(\mu_3-\hat\mu_3)^2}{t^6},
\qquad t\to\infty.\;}
\end{equation}
特别地，
\(
D_{\mathrm{KL}}(\tilde h_t\mid \hat h_t)\sim \frac{3}{32}\frac{(\mu_3-\hat\mu_3)^2}{t^6}
\)。
\end{corollary}
\begin{proof}[证明要点]
在均值与方差对齐时，密度比的 \(t^{-2}\) 主项相消。
沿用定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 中对尺度变量 \(\varepsilon=X/t\) 的 Taylor 展开，
对 \(\tilde\nu\) 与 \(\hat\nu\) 分别得到（于 \(L^2(g_t\,\dd x)\) 意义下）
\[
\frac{\tilde h_t}{g_t}=1+\frac{\sigma^2}{t^2}A\!\left(\frac{x-\bar\gamma}{t}\right)
\;+\;\frac{\mu_3}{t^3}B\!\left(\frac{x-\bar\gamma}{t}\right)+O(t^{-4}),
\]
\[
\frac{\hat h_t}{g_t}=1+\frac{\sigma^2}{t^2}A\!\left(\frac{x-\bar\gamma}{t}\right)
\;+\;\frac{\hat\mu_3}{t^3}B\!\left(\frac{x-\bar\gamma}{t}\right)+O(t^{-4}),
\]
其中
\(
B(y)=\frac{4y(y^2-1)}{(1+y^2)^3}
\)。
因此
\[
\delta_t:=\frac{\tilde h_t}{\hat h_t}-1
=\frac{\mu_3-\hat\mu_3}{t^3}\,B\!\left(\frac{x-\bar\gamma}{t}\right)+o(t^{-3})
\quad\text{于 }L^2(g_t\,\dd x)\text{ 意义下},
\]
并且 \(\int \hat h_t\delta_t=\int(\tilde h_t-\hat h_t)=0\)。
对 \(f(1+u)\) 作二阶 Taylor 展开并积分，得到
\[
D_f(\tilde h_t\mid \hat h_t)
=\frac{f''(1)}{2}\int_{\RR}\hat h_t\,\delta_t^2\,\dd x+o(t^{-6}).
\]
将 \(\hat h_t\) 替换为 \(g_t\) 不影响 \(t^{-6}\) 主项，并代换 \(y=(x-\bar\gamma)/t\) 得
\[
\int_{\RR}\hat h_t\,\delta_t^2\,\dd x
=\frac{(\mu_3-\hat\mu_3)^2}{t^6}\cdot \frac{1}{\pi}\int_{\RR}\frac{B(y)^2}{(1+y^2)}\,\dd y
\;+\;o(t^{-6}).
\]
由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的常数计算 \(\frac{1}{\pi}\int gB^2=\frac{3}{16}\)，代回即得 \eqref{eq:xi-poisson-two-measure-third-moment-gap}。证毕。
\end{proof}

\begin{theorem}[Poisson--Cauchy 粗粒化的 \texorpdfstring{$L^2$}{L2} 锐渐近常数与 KL 标度等价]\label{thm:xi-cauchy-poisson-l2-sharp-kl-scaling}
\leanverified{paper\_xi\_cauchy\_poisson\_l2\_sharp\_kl\_scaling}
沿用命题 \ref{con:xi-poisson-coarsegraining-second-order-normal-form} 的设定，记
\[
\sigma^2:=\int_{\RR}(\gamma-\bar\gamma)^2\,\dd\tilde\nu(\gamma)\in(0,\infty).
\]
若三阶中心绝对矩有限，则
\[
\lim_{t\to\infty}\frac{t^{5/2}}{\sigma^2}\,\|h_t-g_t\|_{L^2(\RR)}
=\sqrt{\frac{3}{16\pi}}.
\]
若进一步满足五阶中心绝对矩有限（即定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件），则
\[
D_{\mathrm{KL}}(h_t\mid g_t)\sim \frac{2\pi}{3}\,t\,\|h_t-g_t\|_{L^2(\RR)}^2
\qquad(t\to\infty).
\]
\end{theorem}
\begin{proof}
由命题 \ref{con:xi-poisson-coarsegraining-second-order-normal-form}，
\[
h_t-g_t=\sigma^2 r_t+\varepsilon_t,
\qquad
r_t(x)=\frac{1}{\pi}\frac{t\bigl(3(x-\bar\gamma)^2-t^2\bigr)}{\bigl((x-\bar\gamma)^2+t^2\bigr)^3},
\]
并且 \(\|\varepsilon_t\|_{L^\infty}=O(t^{-4})\)、\(\|\varepsilon_t\|_{L^1}=O(t^{-3})\)。
由插值不等式
\[
\|\varepsilon_t\|_{L^2}^2\le \|\varepsilon_t\|_{L^\infty}\|\varepsilon_t\|_{L^1}=O(t^{-7})=o(t^{-5}),
\]
故 \(\|\varepsilon_t\|_{L^2}=o(t^{-5/2})\)。
于是
\[
\|h_t-g_t\|_{L^2}
=\sigma^2\|r_t\|_{L^2}+o(t^{-5/2}).
\]
对 \(\|r_t\|_{L^2}\) 作代换 \(y=(x-\bar\gamma)/t\) 得
\[
\|r_t\|_{L^2}^2
=\frac{1}{\pi^2 t^5}
\int_{\RR}\frac{(3y^2-1)^2}{(1+y^2)^6}\,\dd y
=\frac{3}{16\pi}\,t^{-5},
\]
其中最后一步由 \(y=\tan\theta\) 与 Beta 积分恒等式给出。
故
\[
\|r_t\|_{L^2}=\sqrt{\frac{3}{16\pi}}\,t^{-5/2},
\]
从而第一式成立。
\par
再由定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics}，
\[
D_{\mathrm{KL}}(h_t\mid g_t)=\frac{\sigma^4}{8t^4}+O(t^{-6}),
\]
而第一部分给出
\[
\|h_t-g_t\|_{L^2}^2
=\sigma^4\frac{3}{16\pi}\,t^{-5}+o(t^{-5}).
\]
因此
\[
\frac{D_{\mathrm{KL}}(h_t\mid g_t)}{t\,\|h_t-g_t\|_{L^2}^2}
\to
\frac{1/8}{3/(16\pi)}
=\frac{2\pi}{3},
\]
即得所述标度等价。证毕。
\end{proof}
