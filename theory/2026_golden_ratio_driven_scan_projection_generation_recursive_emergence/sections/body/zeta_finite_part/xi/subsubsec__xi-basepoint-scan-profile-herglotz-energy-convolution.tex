\begin{theorem}[扫描剖面的有理 Herglotz 表示与极点层析]\label{thm:xi-basepoint-scan-rational-herglotz-pole-tomography}
沿用定义 \ref{def:xi-basepoint-scan-profile} 与命题 \ref{prop:xi-basepoint-depth-closed-form} 的记号，并令 \(a_k:=1+\delta_k\)。
则存在唯一函数 \(F\) 满足：
\begin{enumerate}
  \item \(F\) 在上半平面 \(\mathbb{C}_{+}\) 解析且 \(\Im F(z)>0\)（Herglotz）；
  \item \(F(z)=O(1/|z|)\) 当 \(|z|\to\infty\) 且 \(z\in\mathbb{C}_{+}\)；
  \item 边界值满足 \(\Im F(x)=H(x)\) 对所有 \(x\in\mathbb{R}\) 成立。
\end{enumerate}
并且 \(F\) 具有显式部分分式表示
\[
\boxed{\;
F(z)=
-\sum_{k=1}^{\kappa}\frac{4\delta_k}{a_k}\cdot\frac{1}{z-(\gamma_k-\mathrm{i}a_k)}\; }.
\]
因此轴外参数 \(\{(\gamma_k,\delta_k)\}\) 可由 \(F\) 的极点与留数唯一恢复：极点位于 \(\gamma_k-\mathrm{i}(1+\delta_k)\)，相应留数为 \(-4\delta_k/(1+\delta_k)\)。
\leanverified{paper\_xi\_basepoint\_scan\_rational\_herglotz\_pole\_tomography}
\end{theorem}
\begin{proof}
由命题 \ref{prop:xi-basepoint-depth-closed-form}，有
\(
H(x)=\sum_{k=1}^{\kappa}\frac{4\delta_k}{(x-\gamma_k)^2+a_k^2}
\)。
对任意 \(z=x+\mathrm{i}y\in\mathbb{C}_{+}\)，注意
\[
-\frac{4\delta_k}{a_k}\cdot\frac{1}{z-(\gamma_k-\mathrm{i}a_k)}
=-\frac{4\delta_k}{a_k}\cdot\frac{1}{(x-\gamma_k)+\mathrm{i}(y+a_k)},
\]
其虚部为
\[
\Im\!\left(-\frac{4\delta_k}{a_k}\cdot\frac{1}{(x-\gamma_k)+\mathrm{i}(y+a_k)}\right)
=\frac{4\delta_k}{a_k}\cdot\frac{y+a_k}{(x-\gamma_k)^2+(y+a_k)^2}\ >\ 0.
\]
故 \(F\) 在 \(\mathbb{C}_{+}\) 上解析且 \(\Im F>0\)，并且显然 \(F(z)=O(1/|z|)\)。
令 \(y\downarrow 0\)，得
\[
\Im F(x)
=\sum_{k=1}^{\kappa}\frac{4\delta_k}{a_k}\cdot\frac{a_k}{(x-\gamma_k)^2+a_k^2}
=H(x),
\]
从而 \(F\) 满足 (1)--(3)。
\par
唯一性：若 \(F_1,F_2\) 亦满足 (1)--(3)，则 \(G:=F_1-F_2\) 在 \(\mathbb{C}_{+}\) 解析、且 \(\Im G(x)=0\)（\(x\in\mathbb{R}\)），并满足 \(G(z)=O(1/|z|)\) 于无穷远。
由 Schwarz 反射可将 \(G\) 延拓为整个函数并在无穷远趋于 \(0\)，故由最大模原理得 \(G\equiv 0\)，从而 \(F_1\equiv F_2\)。
最后，部分分式公式直接给出极点与留数；由 \(a_k=1+\delta_k\) 即可恢复 \(\delta_k\)，且留数一致性提供自洽核验。证毕。
\end{proof}

\begin{proposition}[扫描剖面的 \texorpdfstring{$L^{2}$}{L2} 能量闭式与正定相互作用核]\label{prop:xi-basepoint-scan-l2-energy-pd-kernel}
\leanverified{paper\_xi\_basepoint\_scan\_l2\_energy\_pd\_kernel}
沿用定理 \ref{thm:xi-basepoint-scan-inversion-uniqueness} 的设定，并令 \(a_k:=1+\delta_k\)。
则 \(H\in L^{2}(\mathbb{R})\) 且具有完全闭式分解
\[
\boxed{\;
\int_{\mathbb{R}}H(x)^{2}\,\dd x
=16\pi\sum_{j,k=1}^{\kappa}\delta_j\delta_k\,
\frac{a_j+a_k}{a_ja_k\bigl((\gamma_j-\gamma_k)^{2}+(a_j+a_k)^{2}\bigr)}\; }.
\]
并且核
\[
K_{jk}:=\frac{a_j+a_k}{a_ja_k\bigl((\gamma_j-\gamma_k)^{2}+(a_j+a_k)^{2}\bigr)}
\]
为正定核（\((K_{jk})\succ 0\)）。
\end{proposition}
\begin{proof}
写 \(\psi_k(x):=\frac{1}{(x-\gamma_k)^2+a_k^2}\)，则 \(H(x)=\sum_k 4\delta_k\,\psi_k(x)\)，从而
\[
\int_{\mathbb{R}}H(x)^2\,\dd x
=16\sum_{j,k}\delta_j\delta_k\int_{\mathbb{R}}\psi_j(x)\psi_k(x)\,\dd x.
\]
对任意 \(A,B>0\) 与 \(u,v\in\mathbb{R}\)，有恒等式
\[
\int_{\mathbb{R}}\frac{\dd x}{((x-u)^{2}+A^{2})((x-v)^{2}+B^{2})}
=\frac{\pi(A+B)}{AB((u-v)^{2}+(A+B)^{2})},
\]
可由留数计算得到。取 \(A=a_j,B=a_k,u=\gamma_j,v=\gamma_k\) 即得所示闭式。
\par
由同一恒等式反推
\(
K_{jk}=\frac{1}{\pi}\int_{\mathbb{R}}\psi_j(x)\psi_k(x)\,\dd x
\)，
故 \((K_{jk})\) 为 \(L^2(\mathbb{R})\) 中向量族 \(\{\psi_k\}\) 的 Gram 矩阵，必正定。证毕。
\end{proof}

\begin{corollary}[对角下界：\texorpdfstring{$\|H\|_{2}^{2}$}{||H||2^2} 的显式严格下界]\label{cor:xi-basepoint-scan-l2-energy-diagonal-lower-bound}
在命题 \ref{prop:xi-basepoint-scan-l2-energy-pd-kernel} 的设定下，有
\[
\int_{\mathbb{R}}H(x)^{2}\,\dd x
\ \ge\ 8\pi\sum_{k=1}^{\kappa}\frac{\delta_k^{2}}{(1+\delta_k)^{3}},
\]
并且当 \(\kappa\ge 2\) 时上述不等式严格成立。
\end{corollary}
\begin{proof}
由命题 \ref{prop:xi-basepoint-scan-l2-energy-pd-kernel} 的闭式分解与 \(K_{jk}>0\)（从而交叉项非负）可得
\[
\int_{\mathbb{R}}H(x)^{2}\,\dd x\ge 16\pi\sum_{k=1}^{\kappa}\delta_k^2 K_{kk}.
\]
而
\(
K_{kk}=\frac{2a_k}{a_k^2(0+(2a_k)^2)}=\frac{1}{2a_k^3}
\)，代入 \(a_k=1+\delta_k\) 即得下界。
若 \(\kappa\ge 2\)，则至少存在一项 \(j\neq k\) 使 \(\delta_j\delta_k K_{jk}>0\)，故严格。证毕。
\end{proof}

\begin{proposition}[Cauchy 稳定卷积律：扫描剖面的自卷积生成“和集谱”]\label{prop:xi-basepoint-scan-cauchy-stable-convolution}
令 \(P_a(x):=\frac{a}{\pi(x^2+a^2)}\) 为标准 Cauchy--Poisson 核（\(a>0\)），并令 \(a_k:=1+\delta_k\)。
则
\[
\frac{4\delta_k}{(x-\gamma_k)^2+a_k^2}
=4\pi\frac{\delta_k}{a_k}\,P_{a_k}(x-\gamma_k),
\qquad
H(x)=\sum_{k=1}^{\kappa}4\pi\frac{\delta_k}{a_k}\,P_{a_k}(x-\gamma_k).
\]
由于 \(P_a*P_b=P_{a+b}\)，得到扫描剖面的自卷积闭式：
\[
\boxed{\;
(H*H)(x)=
\sum_{j,k=1}^{\kappa}\frac{16\pi^{2}\delta_j\delta_k}{a_ja_k}\,
P_{a_j+a_k}\bigl(x-(\gamma_j+\gamma_k)\bigr)\; }.
\]
\leanverified{paper\_xi\_basepoint\_scan\_cauchy\_stable\_convolution}
更一般地，对任意 \(n\ge 2\)，\(H^{(*n)}\) 仍为有限 Cauchy 核混合：其中心落在 \(\{\gamma_{i_1}+\cdots+\gamma_{i_n}\}\) 的 \(n\)-重和集上，尺度参数落在 \(\{a_{i_1}+\cdots+a_{i_n}\}\) 上。
\end{proposition}
\begin{proof}
由 Fourier 变换 \(\widehat{P_a}(\omega)=e^{-a|\omega|}\) 得
\(
\widehat{P_a*P_b}=\widehat{P_a}\cdot\widehat{P_b}=e^{-(a+b)|\omega|}=\widehat{P_{a+b}}
\)，故 \(P_a*P_b=P_{a+b}\)。
对移位核 \(P_a(\cdot-\gamma)\) 卷积即给出中心相加、尺度相加；将 \(H\) 展开为上述有限和并两两卷积即可得闭式。\(n\)-次卷积由归纳法同理。证毕。
\end{proof}
