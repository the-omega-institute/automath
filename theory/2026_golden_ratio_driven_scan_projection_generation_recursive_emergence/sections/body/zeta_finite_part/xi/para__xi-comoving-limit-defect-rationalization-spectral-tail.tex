\begin{proposition}[极限缺陷势的指数化有理化与零极点读出（有限缺陷口径）]\label{prop:xi-limit-defect-potential-rationalization}
在定理 \ref{thm:xi-limit-defect-potential-fourier-injective} 的有限缺陷设定下，记
\(\mathcal{D}_\infty(x)=\sum_{j=1}^{M} m_j\,\mathcal{K}_{\gamma_j,\delta_j}(x)\)（\eqref{eq:xi-limit-defect-potential}），并定义
\[
R(x):=\exp\!\bigl(-2\mathcal{D}_\infty(x)\bigr)\qquad(x\in\RR).
\]
则 \(R:\RR\to(0,1]\) 具有完全显式的乘积表示
\begin{equation}\label{eq:xi-limit-defect-rationalization}
\boxed{\;
R(x)
=\prod_{j=1}^{M}
\left(\frac{(x-\gamma_j)^2+(1-\delta_j)^2}{(x-\gamma_j)^2+(1+\delta_j)^2}\right)^{m_j}\; } ,
\end{equation}
并且由于 \(m_j\in\NN\)，右端由同一表达式在 \(x\) 上延拓为 \(\CC\) 上的有理函数。
更进一步，\(R|_{\RR}\)（等价地 \(\mathcal{D}_\infty|_{\RR}\)）唯一决定多重集 \(\{(\gamma_j,\delta_j,m_j)\}_{j=1}^{M}\)。
\leanverified{paper\_xi\_limit\_defect\_potential\_rationalization}
\end{proposition}
\begin{proof}
由 \(\mathcal{K}_{\gamma,\delta}(x)=-\log|\mathcal{C}((x-\gamma)+\mathrm{i}\delta)|\)（\eqref{eq:xi-log-defect-potential}）与 Cayley 模长闭式 \eqref{eq:xi-cayley-modulus-delta} 可得
\[
-2\mathcal{K}_{\gamma,\delta}(x)
=\log\frac{(x-\gamma)^2+(1-\delta)^2}{(x-\gamma)^2+(1+\delta)^2}.
\]
代入 \(\mathcal{D}_\infty=\sum_j m_j\mathcal{K}_{\gamma_j,\delta_j}\) 并指数化得到 \eqref{eq:xi-limit-defect-rationalization}。
\par
若两组有限缺陷数据给出同一 \(R|_{\RR}\)，则两侧由 \eqref{eq:xi-limit-defect-rationalization} 给出的有理函数在 \(\RR\) 上一致，因而在 \(\CC\) 上一致；
于是两侧的零点--极点多重集相同。
而 \eqref{eq:xi-limit-defect-rationalization} 的零点仅可能位于
\(
x=\gamma_j\pm \mathrm{i}(1-\delta_j)
\)，极点仅可能位于
\(
x=\gamma_j\pm \mathrm{i}(1+\delta_j)
\)，其重数均为 \(m_j\)。
故 \((\gamma_j,\delta_j,m_j)\) 由零极点位置与重数唯一恢复。证毕。
\end{proof}

\begin{corollary}[极限缺陷势的 \texorpdfstring{$L^1$}{L1} 求和律：总离线偏移的精确可观测量]\label{cor:xi-limit-defect-potential-L1-sumrule}
\leanverified{paper\_xi\_limit\_defect\_potential\_l1\_sumrule}
在定理 \ref{thm:xi-limit-defect-potential-fourier-injective} 的有限缺陷设定下，\(\mathcal{D}_\infty\in L^1(\RR)\)，并且成立严格恒等式
\begin{equation}\label{eq:xi-limit-defect-L1-sumrule}
\boxed{\;
\int_{\RR}\mathcal{D}_\infty(x)\,\dd x
=2\pi\sum_{j=1}^{M} m_j\,\delta_j\; } .
\end{equation}
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-comoving-jensen-weight-integral-conservation}，
\(
\int_{\RR}\mathcal{K}_{\gamma_j,\delta_j}(x)\dd x=2\pi\delta_j
\)。
对有限和 \(\mathcal{D}_\infty=\sum_j m_j\mathcal{K}_{\gamma_j,\delta_j}\) 逐项积分即得 \eqref{eq:xi-limit-defect-L1-sumrule}；同时给出 \(\mathcal{D}_\infty\in L^1(\RR)\)。证毕。
\end{proof}

\begin{corollary}[谱尾指数读取最大离线偏移]\label{cor:xi-limit-defect-potential-spectral-tail-reads-deltamax}
在定理 \ref{thm:xi-limit-defect-potential-fourier-injective} 的有限缺陷设定下，若 \(M\ge 1\) 并记
\[
\delta_{\max}:=\max_{1\le j\le M}\delta_j\in(0,1),
\]
则对 \(\widehat{\mathcal{D}_\infty'}\)（\eqref{eq:xi-limit-defect-potential-fourier}）有严格指数率
\begin{equation}\label{eq:xi-limit-defect-spectral-tail-deltamax}
\boxed{\;
\limsup_{|\xi|\to\infty}\frac{1}{|\xi|}\log\Bigl|\widehat{\mathcal{D}_\infty'}(\xi)\Bigr|
=-(1-\delta_{\max})\; } .
\end{equation}
\end{corollary}
\leanverified{paper\_xi\_limit\_defect\_potential\_spectral\_tail\_reads\_deltamax}
\begin{proof}[证明要点]
由 \eqref{eq:xi-limit-defect-potential-fourier} 与三角不等式，存在常数 \(C<\infty\) 使对所有 \(\xi\neq 0\) 有
\[
\bigl|\widehat{\mathcal{D}_\infty'}(\xi)\bigr|
\le C\,e^{-(1-\delta_{\max})|\xi|},
\]
从而得到 \(\limsup\le -(1-\delta_{\max})\)。
\par
为得反向不等式，令 \(I_{\max}:=\{j:\ \delta_j=\delta_{\max}\}\) 且记
\(
G(\xi):=\sum_{j\in I_{\max}} m_j e^{-\mathrm{i}\gamma_j\xi}
\)。
由 Dirichlet 同时逼近定理，可取一列 \(\xi_n\to\infty\) 使得 \(e^{-\mathrm{i}\gamma_j\xi_n}\to 1\) 对所有 \(j\in I_{\max}\) 成立，
从而 \(|G(\xi_n)|\to \sum_{j\in I_{\max}}m_j>0\)。
将 \eqref{eq:xi-limit-defect-potential-fourier} 中 \(\delta<\delta_{\max}\) 的指数项并入余项可得
\[
\widehat{\mathcal{D}_\infty'}(\xi)
=\mathrm{i}\pi\,\operatorname{sgn}(\xi)\,e^{-(1-\delta_{\max})|\xi|}\,G(\xi)
\;+\;O\!\bigl(e^{-(1-\delta_{\max}+\eta)|\xi|}\bigr),
\]
其中 \(\eta>0\) 仅依赖于偏移层之间的间隔；若除 \(\delta_{\max}\) 外无其它偏移层，则可取 \(\eta=2\delta_{\max}\)。
沿 \(\xi=\xi_n\) 取 \(n\to\infty\) 得
\(
|\widehat{\mathcal{D}_\infty'}(\xi_n)|\ge c\,e^{-(1-\delta_{\max})|\xi_n|}
\)
对某 \(c>0\) 成立，从而
\(
\limsup_{|\xi|\to\infty}\frac{1}{|\xi|}\log|\widehat{\mathcal{D}_\infty'}(\xi)|
\ge -(1-\delta_{\max})
\)。
合并两侧不等式即得 \eqref{eq:xi-limit-defect-spectral-tail-deltamax}。证毕。
\end{proof}

\begin{theorem}[极限缺陷势的 \(L^2\) 能量闭式与正定相互作用核]\label{thm:xi-limit-defect-potential-L2-energy-kernel}
在定理 \ref{thm:xi-limit-defect-potential-fourier-injective} 的有限缺陷设定下，\(\mathcal{D}_\infty\in L^2(\RR)\)，并且
\begin{equation}\label{eq:xi-limit-defect-L2-energy-closed}
\boxed{\;
\int_{\RR}\mathcal{D}_\infty(x)^2\,\dd x
=\frac{\pi}{2}\sum_{j,k=1}^{M}m_jm_k\,\mathcal{Q}\!\bigl(\gamma_j-\gamma_k;\ A_j,B_j,A_k,B_k\bigr)\;}
\end{equation}
其中
\(
A_j:=1-\delta_j
\)、\(
B_j:=1+\delta_j
\)，并定义对 \(\Delta\in\RR\)、\(u>0\) 的原函数
\[
\Phi(\Delta,u):=
\begin{cases}
u\log(\Delta^2+u^2)-2u+2|\Delta|\arctan\!\bigl(u/|\Delta|\bigr),&\Delta\ne 0,\\[2mm]
u\log(u^2)-2u,&\Delta=0,
\end{cases}
\]
以及四端点组合核
\begin{equation}\label{eq:xi-limit-defect-L2-energy-kernel-Q}
\boxed{\;
\mathcal{Q}(\Delta;A,B,C,D)
:=\Phi(\Delta,B+D)-\Phi(\Delta,A+D)-\Phi(\Delta,B+C)+\Phi(\Delta,A+C).}
\end{equation}
此外，该核诱导的双线性型为半正定：对任意实系数 \((c_j)_{j=1}^{M}\) 有
\begin{equation}\label{eq:xi-limit-defect-L2-energy-kernel-psd}
\boxed{\;
\sum_{j,k=1}^{M}c_jc_k\,\mathcal{Q}\!\bigl(\gamma_j-\gamma_k;\ A_j,B_j,A_k,B_k\bigr)\ \ge\ 0.}
\end{equation}
\end{theorem}
\begin{proof}
由命题 \ref{prop:xi-log-defect-potential-poisson-primitive} 的 Poisson 原函数表示，对任意 \((\gamma,\delta)\in\RR\times(0,1)\) 有
\[
\mathcal{K}_{\gamma,\delta}(x)=\pi\int_{1-\delta}^{1+\delta}P_t(x-\gamma)\,\dd t,
\qquad
P_t(u):=\frac{1}{\pi}\frac{t}{u^2+t^2}.
\]
将其代入 \(\mathcal{D}_\infty(x)=\sum_{j=1}^{M}m_j\mathcal{K}_{\gamma_j,\delta_j}(x)\) 得
\[
\mathcal{D}_\infty(x)
=\pi\sum_{j=1}^{M}m_j\int_{A_j}^{B_j}P_t(x-\gamma_j)\,\dd t.
\]
于是由 Tonelli 定理与 Poisson 核卷积闭性 \(P_t*P_s=P_{t+s}\) 得
\[
\int_{\RR}\mathcal{D}_\infty(x)^2\,\dd x
=\pi^2\sum_{j,k=1}^{M}m_jm_k\int_{A_j}^{B_j}\!\int_{A_k}^{B_k}
\left(\int_{\RR}P_t(x-\gamma_j)P_s(x-\gamma_k)\,\dd x\right)\dd t\,\dd s
\]
\[
=\pi^2\sum_{j,k=1}^{M}m_jm_k\int_{A_j}^{B_j}\!\int_{A_k}^{B_k}
P_{t+s}(\gamma_j-\gamma_k)\,\dd t\,\dd s
=\pi\sum_{j,k=1}^{M}m_jm_k\int_{A_j}^{B_j}\!\int_{A_k}^{B_k}
\frac{t+s}{(\gamma_j-\gamma_k)^2+(t+s)^2}\,\dd t\,\dd s.
\]
对内层积分使用恒等式
\(
\int \frac{u}{\Delta^2+u^2}\,\dd u=\frac12\log(\Delta^2+u^2)
\)
并作变量代换 \(u=t+s\)，可化为
\[
\int_{A_k}^{B_k}\frac{t+s}{\Delta^2+(t+s)^2}\,\dd s
=\frac12\Bigl[\log\bigl(\Delta^2+(t+B_k)^2\bigr)-\log\bigl(\Delta^2+(t+A_k)^2\bigr)\Bigr],
\quad \Delta:=\gamma_j-\gamma_k.
\]
再对 \(t\in[A_j,B_j]\) 积分，并利用 \(\frac{\dd}{\dd u}\Phi(\Delta,u)=\log(\Delta^2+u^2)\)，即得到
\eqref{eq:xi-limit-defect-L2-energy-closed}--\eqref{eq:xi-limit-defect-L2-energy-kernel-Q}。
\par
半正定性由 Gram 表示立即推出：令
\(
f_j(x):=\pi\int_{A_j}^{B_j}P_t(x-\gamma_j)\,\dd t
\)，则
\[
\int_{\RR}f_j(x)f_k(x)\,\dd x=\frac{\pi}{2}\,\mathcal{Q}(\gamma_j-\gamma_k;A_j,B_j,A_k,B_k),
\]
从而对任意 \((c_j)\) 有
\[
\sum_{j,k}c_jc_k\,\mathcal{Q}(\gamma_j-\gamma_k;A_j,B_j,A_k,B_k)
=\frac{2}{\pi}\int_{\RR}\left(\sum_{j=1}^{M}c_j f_j(x)\right)^{2}\dd x\ \ge\ 0.
\]
证毕。
\end{proof}
