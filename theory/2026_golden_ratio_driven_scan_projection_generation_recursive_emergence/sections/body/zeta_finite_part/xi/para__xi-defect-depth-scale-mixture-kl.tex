\paragraph{深度谱诱导的尺度混合：相对熵的 \texorpdfstring{$(t+1)^{-2}$}{(t+1)^{-2}} 主导律与刚性}\label{par:xi-defect-depth-scale-mixture-kl}
尺度粗粒化 \(t\mapsto S_{\mathrm{def}}(t)\)（上一命题）把有限深度谱 \(\{(\delta_j,m_j)\}\) 压缩为一条单变量 Stieltjes 指纹曲线。
本段给出同一深度谱在信息论侧的一条配套定量：以缺陷权重诱导的尺度混合 Cauchy 密度相对单一基准的相对熵具有精确的 \((t+1)^{-2}\) 主导常数，并由此推出一条刚性判别。

\begin{proposition}[缺陷权重的尺度混合 Cauchy 相对熵：\texorpdfstring{$(t+1)^{-2}$}{(t+1)^{-2}} 主导常数]\label{prop:xi-defect-scale-mixture-kl-leading}
\leanverified{paper\_xi\_defect\_scale\_mixture\_kl\_leading}
令 \(g_a(x):=\frac{1}{\pi}\frac{a}{x^2+a^2}\)（\(a>0\)）为尺度 Cauchy 密度，并令
\[
c_t(x):=g_{t+1}(x)=\frac{1}{\pi}\frac{t+1}{x^2+(t+1)^2}\qquad(t\ge 0)
\]
为 Poisson 粗粒化基准（见命题 \ref{prop:xi-poisson-relative-entropy-contraction} 的记号）。
给定有限深度谱 \(\{(\delta_j,m_j)\}_{j=1}^N\)（\(\delta_j\ge 0\)、\(m_j>0\)），令 \(a=t+1\)、\(a_j=a+\delta_j\)，并定义缺陷权重
\[
w_j(t):=\frac{m_j\delta_j}{a_j},\qquad
S(t):=\sum_{j=1}^N w_j(t),\qquad
p_j(t):=\frac{w_j(t)}{S(t)}.
\]
构造尺度混合密度
\[
h_t(x):=\sum_{j=1}^N p_j(t)\,g_{a_j}(x).
\]
若 \(\sum_{j=1}^N m_j\delta_j>0\)，则当 \(t\to\infty\) 有
\begin{equation}\label{eq:xi-defect-scale-mixture-kl-leading}
\boxed{\;
D_{\mathrm{KL}}(h_t\mid c_t)
=\frac{\mu^2}{4(t+1)^2}+O\!\bigl((t+1)^{-3}\bigr),\qquad
\mu:=\frac{\sum_{j=1}^N m_j\delta_j^2}{\sum_{j=1}^N m_j\delta_j}\;}
\end{equation}
从而极限常数存在且可计算：
\[
\boxed{\;
\lim_{t\to\infty}(t+1)^2D_{\mathrm{KL}}(h_t\mid c_t)
=\frac14\left(\frac{\sum_{j=1}^N m_j\delta_j^2}{\sum_{j=1}^N m_j\delta_j}\right)^2\; }.
\]
\end{proposition}

\begin{proof}
记 \(\Phi:=\sum_j m_j\delta_j\)、\(\Phi_2:=\sum_j m_j\delta_j^2\)，并令 \(a=t+1\to\infty\)。
由
\[
w_j(t)=\frac{m_j\delta_j}{a+\delta_j}
=\frac{m_j\delta_j}{a}-\frac{m_j\delta_j^2}{a^2}+O(a^{-3})
\]
得到
\[
S(t)=\frac{\Phi}{a}-\frac{\Phi_2}{a^2}+O(a^{-3}),
\qquad
p_j(t)=\frac{m_j\delta_j}{\Phi}+O(a^{-1}),
\qquad
\sum_j p_j(t)\delta_j=\frac{\Phi_2}{\Phi}+O(a^{-1})=\mu+O(a^{-1}).
\]
另一方面，
\(
\partial_a g_a(x)=\frac{1}{\pi}\frac{x^2-a^2}{(x^2+a^2)^2}
\)
且
\[
\frac{\partial_a g_a(x)}{g_a(x)}=\frac{x^2-a^2}{a(x^2+a^2)},
\qquad
\left|\frac{\partial_a g_a}{g_a}\right|\le \frac{1}{a}.
\]
对每个固定 \(\delta\ge 0\) 在 \(a\to\infty\) 下作 Taylor 展开得
\[
g_{a+\delta}(x)=g_a(x)+\delta\,\partial_a g_a(x)+O(\delta^2 a^{-3})
\qquad\text{一致于 }x\in\RR,
\]
从而
\[
h_t(x)=g_a(x)+\mu_t\,\partial_a g_a(x)+R_t(x),
\qquad
\mu_t:=\sum_j p_j(t)\delta_j=\mu+O(a^{-1}),
\qquad
\|R_t\|_\infty=O(a^{-3}).
\]
令 \(g:=g_a\)、\(\varepsilon:=h_t-g\)、\(r:=\varepsilon/g\)，则 \(\int_{\RR}\varepsilon\,\dd x=0\) 且 \(r=O(a^{-1})\)。
由 \((1+r)\log(1+r)=\tfrac12 r^2+O(|r|^3)\) 与 \(\int g r=0\)，得到
\[
D_{\mathrm{KL}}(h_t\mid g)
=\frac12\int_{\RR}g\,r^2\,\dd x+O\!\left(\int_{\RR}g|r|^3\,\dd x\right)
=\frac12\int_{\RR}\frac{\varepsilon^2}{g}\,\dd x+O(a^{-3}).
\]
主项取 \(\varepsilon=\mu_t\,\partial_a g+O(a^{-3})\) 得
\[
D_{\mathrm{KL}}(h_t\mid g_a)
=\frac12\mu_t^2\int_{\RR}\frac{(\partial_a g_a)^2}{g_a}\,\dd x+O(a^{-3}).
\]
最后直接计算（代换 \(x=ay\)）：
\[
\int_{\RR}\frac{(\partial_a g_a)^2}{g_a}\,\dd x
=\frac{1}{a^2}\cdot\frac{1}{\pi}\int_{\RR}\frac{(y^2-1)^2}{(1+y^2)^3}\,\dd y
=\frac{1}{a^2}\left(1-4\cdot\frac12+4\cdot\frac{3}{8}\right)
=\frac{1}{2a^2}.
\]
代回并用 \(\mu_t=\mu+O(a^{-1})\) 即得 \eqref{eq:xi-defect-scale-mixture-kl-leading}。证毕。
\end{proof}

\begin{corollary}[尺度缺陷的刚性判别：若快于 \texorpdfstring{$(t+1)^{-2}$}{(t+1)^{-2}} 衰减则深度谱退化]\label{cor:xi-defect-scale-mixture-kl-rigidity}
\leanverified{paper\_xi\_defect\_scale\_mixture\_kl\_rigidity}
在命题 \ref{prop:xi-defect-scale-mixture-kl-leading} 的设定下，若
\[
D_{\mathrm{KL}}(h_t\mid c_t)=o\!\bigl((t+1)^{-2}\bigr)\qquad(t\to\infty),
\]
则必有 \(\sum_{j=1}^N m_j\delta_j^2=0\)，从而 \(\delta_j=0\) 对所有 \(j\) 成立。
\end{corollary}

\begin{proof}
由 \eqref{eq:xi-defect-scale-mixture-kl-leading}，
\(
(t+1)^2D_{\mathrm{KL}}(h_t\mid c_t)\to \frac14(\Phi_2/\Phi)^2
\)。
若左端趋于 \(0\)，则 \(\Phi_2=0\)；再由 \(m_j>0\) 与 \(\delta_j\ge 0\) 得 \(\delta_j=0\) 全体。证毕。
\end{proof}
