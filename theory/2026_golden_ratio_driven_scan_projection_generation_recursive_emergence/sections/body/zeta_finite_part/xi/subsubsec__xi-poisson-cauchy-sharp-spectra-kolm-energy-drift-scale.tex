\paragraph{二阶可见项驱动的普适锐常数：\texorpdfstring{$L^p$}{Lp} 谱、Kolmogorov 距离、谱能量与交点漂移}

本段在命题 \(11.667\)、命题 \(11.670\) 的二阶一致展开口径下，抽取一组与高阶形状无关的极限常数。
其共同结构是：在 \(t\to\infty\) 的粗粒化极限中，\(\tilde h_t-\tilde g_t\) 的主导项完全由方差 \(\sigma^2\) 乘以显式二阶可见项 \(r_t\) 决定，而余项由更高阶中心矩在更快阶上控制。

\subparagraph{二阶可见项}
沿用结论 \ref{con:xi-poisson-cauchy-kl-sixth-order-independent-submission} 的记号：
\(\tilde h_t=\mathsf P_t*\tilde\nu\)、\(\tilde g_t(x)=\mathsf P_t(x-\bar\gamma)\)。
定义二阶可见项
\begin{equation}\label{eq:xi-poisson-cauchy-visible-r2-rt}
r_t(x):=\frac12\,\partial_x^2\tilde g_t(x)
=\frac{1}{\pi}\,t\,\frac{3(x-\bar\gamma)^2-t^2}{\bigl((x-\bar\gamma)^2+t^2\bigr)^3}.
\end{equation}
在命题 \(11.667\) 的假设下有分解
\begin{equation}\label{eq:xi-poisson-cauchy-second-order-splitting-rt}
\tilde h_t=\tilde g_t+\sigma^2 r_t+\varepsilon_t,
\end{equation}
其中 \(\varepsilon_t\) 的一致界与可积界由命题 \(11.667\)、命题 \(11.670\) 给出（余项受三阶绝对中心矩控制）。

\begin{theorem}[\texorpdfstring{$L^p$}{Lp} 锐渐近谱]\label{thm:xi-poisson-cauchy-lp-sharp-spectrum}
\leanverified{paper\_xi\_poisson\_cauchy\_lp\_sharp\_spectrum}
假设 \(\int_{\RR}|\gamma-\bar\gamma|^3\,\dd\tilde\nu(\gamma)<\infty\) 且 \(\sigma^2>0\)。
则对任意 \(p\in[2,\infty)\) 存在极限
\[
\lim_{t\to\infty}\frac{t^{3-\frac1p}}{\sigma^2}\,
\|\tilde h_t-\tilde g_t\|_{L^p(\RR,\dd x)}
=\frac{1}{\pi}\left(\int_{\RR}\frac{|3y^2-1|^{p}}{(1+y^2)^{3p}}\,\dd y\right)^{1/p}.
\]
并且在 \(p=\infty\) 处有
\[
\lim_{t\to\infty}\frac{t^{3}}{\sigma^2}\,
\|\tilde h_t-\tilde g_t\|_{L^\infty(\RR,\dd x)}
=\frac{1}{\pi}.
\]
\end{theorem}
\begin{proof}
由分解 \eqref{eq:xi-poisson-cauchy-second-order-splitting-rt}，
\[
\tilde h_t-\tilde g_t=\sigma^2 r_t+\varepsilon_t.
\]
命题 \(11.667\)、命题 \(11.670\) 给出 \(\|\varepsilon_t\|_\infty=O(t^{-4})\)，并且以三阶 Taylor 余项的积分型表达结合 Minkowski 不等式可得
\(
\|\varepsilon_t\|_{L^p}=O(t^{\frac1p-4})
\)
对任意 \(p\ge 1\) 成立。

另一方面，令 \(y=(x-\bar\gamma)/t\)，由 \eqref{eq:xi-poisson-cauchy-visible-r2-rt} 得
\[
r_t(\bar\gamma+ty)=\frac{1}{\pi t^3}\frac{3y^2-1}{(1+y^2)^3},
\qquad \dd x=t\,\dd y.
\]
因此对 \(p<\infty\)，
\[
\|r_t\|_{L^p(\RR,\dd x)}
=t^{\frac1p-3}\cdot \frac{1}{\pi}\left(\int_{\RR}\frac{|3y^2-1|^{p}}{(1+y^2)^{3p}}\,\dd y\right)^{1/p}.
\]
故
\(
\|\varepsilon_t\|_{L^p}/\|r_t\|_{L^p}=O(t^{-1})\to 0
\)，从而
\[
\|\tilde h_t-\tilde g_t\|_{L^p}
=\sigma^2\|r_t\|_{L^p}+o\!\left(t^{\frac1p-3}\right),
\]
并得到所述极限。

对 \(p=\infty\) 同理：\(\|\varepsilon_t\|_\infty=O(t^{-4})\)，并且
\[
\|r_t\|_\infty=\frac{1}{\pi t^3}\sup_{y\in\RR}\frac{|3y^2-1|}{(1+y^2)^3}
=\frac{1}{\pi t^3},
\]
其中最大值在 \(y=0\) 处取到。证毕。
\end{proof}

\begin{theorem}[Kolmogorov 距离的二阶锐常数]\label{thm:xi-poisson-cauchy-kolmogorov-second-order-constant}
\leanverified{paper\_xi\_poisson\_cauchy\_kolmogorov\_second\_order\_constant}
在定理 \ref{thm:xi-poisson-cauchy-lp-sharp-spectrum} 的假设下，定义分布函数
\[
\tilde H_t(x):=\int_{-\infty}^{x}\tilde h_t(u)\,\dd u,
\qquad
\tilde G_t(x):=\int_{-\infty}^{x}\tilde g_t(u)\,\dd u.
\]
则存在极限
\[
\lim_{t\to\infty}\frac{t^{2}}{\sigma^2}\sup_{x\in\RR}\bigl|\tilde H_t(x)-\tilde G_t(x)\bigr|
=\frac{9}{16\sqrt3\,\pi}.
\]
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-poisson-cauchy-second-order-splitting-rt}，
\[
\tilde H_t(x)-\tilde G_t(x)
=\sigma^2\int_{-\infty}^{x}r_t(u)\,\dd u+\int_{-\infty}^{x}\varepsilon_t(u)\,\dd u.
\]
并且 \(\sup_x\bigl|\int_{-\infty}^x\varepsilon_t\bigr|\le \|\varepsilon_t\|_{L^1}=O(t^{-3})\)（命题 \(11.670\)），故该项在乘以 \(t^2/\sigma^2\) 后趋于 \(0\)。

对主项作尺度代换 \(y=(u-\bar\gamma)/t\) 并直接积分得
\[
\int_{-\infty}^{x}r_t(u)\,\dd u
=-\frac{1}{\pi t^2}\frac{y}{(1+y^2)^2},
\qquad y=\frac{x-\bar\gamma}{t}.
\]
因此
\[
\sup_{x\in\RR}|\tilde H_t(x)-\tilde G_t(x)|
=\frac{\sigma^2}{\pi t^2}\sup_{y\in\RR}\frac{|y|}{(1+y^2)^2}+o(t^{-2}).
\]
函数 \(f(y)=|y|/(1+y^2)^2\) 在 \(y>0\) 上的临界点满足 \(f'(y)=0\iff 1-3y^2=0\)，故峰值点为 \(y=1/\sqrt3\)，且
\(
f(1/\sqrt3)=9/(16\sqrt3)
\)。
代回即得极限常数。证毕。
\end{proof}

\begin{theorem}[任意阶谱能量的普适极限常数]\label{thm:xi-poisson-cauchy-spectral-energy-universal-constant}
\leanverified{paper\_xi\_poisson\_cauchy\_spectral\_energy\_universal\_constant}
假设 \(\sigma^2<\infty\)。
对任意 \(\alpha\ge 0\)，定义谱能量泛函
\[
\mathcal E_\alpha(t):=\frac{1}{2\pi}\int_{\RR}|\xi|^{2\alpha}\,
\bigl|\widehat{\tilde h_t}(\xi)-\widehat{\tilde g_t}(\xi)\bigr|^2\,\dd \xi.
\]
则存在极限
\[
\lim_{t\to\infty}\frac{t^{2\alpha+5}}{\sigma^4}\,\mathcal E_\alpha(t)
=\frac{\Gamma(2\alpha+5)}{2^{2\alpha+7}\pi}.
\]
\end{theorem}
\begin{proof}
由 Poisson 半群的 Fourier 乘子（命题 \(11.668\)）可写
\[
\widehat{\tilde h_t}(\xi)=e^{-t|\xi|}\varphi(\xi),
\qquad
\widehat{\tilde g_t}(\xi)=e^{-t|\xi|}e^{-i\bar\gamma\xi},
\qquad
\varphi(\xi):=\int_{\RR}e^{-i\xi\gamma}\,\dd\tilde\nu(\gamma).
\]
故
\[
\mathcal E_\alpha(t)=\frac{1}{2\pi}\int_{\RR}|\xi|^{2\alpha}e^{-2t|\xi|}
\bigl|\varphi(\xi)-e^{-i\bar\gamma\xi}\bigr|^2\,\dd\xi.
\]
二阶矩有限蕴含 \(\varphi(\xi)=e^{-i\bar\gamma\xi}\bigl(1-\frac{\sigma^2\xi^2}{2}+o(\xi^2)\bigr)\) 于 \(\xi\to 0\)，从而
\(
|\varphi(\xi)-e^{-i\bar\gamma\xi}|^2=\frac{\sigma^4\xi^4}{4}+o(\xi^4)
\)。
令 \(\xi=s/t\)，得
\[
\frac{t^{2\alpha+5}}{\sigma^4}\mathcal E_\alpha(t)
=\frac{1}{2\pi}\int_{\RR}|s|^{2\alpha}e^{-2|s|}
\frac{t^4}{\sigma^4}\left|\varphi\!\left(\frac{s}{t}\right)-e^{-i\bar\gamma s/t}\right|^2\,\dd s.
\]
括号内逐点收敛到 \(s^4/4\)，并可由 \(|\varphi|\le 1\) 与指数衰减构造统一可积上界，故可用支配收敛定理交换极限与积分。
于是
\[
\lim_{t\to\infty}\frac{t^{2\alpha+5}}{\sigma^4}\mathcal E_\alpha(t)
=\frac{1}{2\pi}\int_{\RR}|s|^{2\alpha}e^{-2|s|}\cdot\frac{s^4}{4}\,\dd s
=\frac{1}{4\pi}\cdot 2\int_0^\infty e^{-2s}s^{2\alpha+4}\,\dd s.
\]
最后由 Gamma 积分 \(\int_0^\infty e^{-2s}s^{2\alpha+4}\,\dd s=\Gamma(2\alpha+5)/2^{2\alpha+5}\) 得到所示常数。证毕。
\end{proof}

\begin{theorem}[密度交点的常数级漂移由三阶中心矩完全决定]\label{thm:xi-poisson-cauchy-density-intersection-drift-m3}
假设 \(\int_{\RR}|\gamma-\bar\gamma|^{4}\,\dd\tilde\nu(\gamma)<\infty\) 且 \(\sigma^2>0\)。
令
\(
\Delta_t(x):=\tilde h_t(x)-\tilde g_t(x)
\)。
则存在 \(t_0\) 使对所有 \(t\ge t_0\)，方程 \(\Delta_t(x)=0\) 在 \(\RR\) 上恰有两个实根 \(a_t<b_t\)，并且有漂移展开
\[
a_t=\bar\gamma-\frac{t}{\sqrt3}+\frac{\mu_3}{3\sigma^2}+o(1),
\qquad
b_t=\bar\gamma+\frac{t}{\sqrt3}+\frac{\mu_3}{3\sigma^2}+o(1),
\]
其中 \(\mu_3=\int_{\RR}(\gamma-\bar\gamma)^3\,\dd\tilde\nu(\gamma)\)。
并且符号结构稳定：
\[
\Delta_t(x)>0\ \text{于 }(-\infty,a_t)\cup(b_t,\infty),
\qquad
\Delta_t(x)<0\ \text{于 }(a_t,b_t).
\]
\end{theorem}
\begin{proof}
记 \(u=\gamma-\bar\gamma\)、\(z=x-\bar\gamma\)。
对 \(u\mapsto \mathsf P_t(z-u)\) 在 \(0\) 处作三阶 Taylor 展开并对 \(\tilde\nu\) 取期望，利用 \(\int u\,\dd\tilde\nu=0\) 得
\[
\Delta_t(x)=\frac{\sigma^2}{2}\partial_x^2\tilde g_t(x)-\frac{\mu_3}{6}\partial_x^3\tilde g_t(x)+R_t(x),
\]
其中余项 \(R_t\) 由四阶绝对中心矩与 \(\partial_x^4\mathsf P_t\) 的一致界控制，从而 \(\|R_t\|_\infty=O(t^{-5})\)。
令 \(y=z/t\)，直接计算得到
\[
\Delta_t(\bar\gamma+ty)=\frac{\sigma^2}{\pi t^3}k_2(y)+\frac{\mu_3}{\pi t^4}k_3(y)+O(t^{-5}),
\]
其中
\[
k_2(y):=\frac{3y^2-1}{(1+y^2)^3},
\qquad
k_3(y):=\frac{4y(y^2-1)}{(1+y^2)^4}.
\]
函数 \(k_2\) 为偶函数并在 \(y=\pm y_0\) 处恰有单零点，\(y_0=1/\sqrt3\)，且 \(k_2'(y_0)\neq 0\)。
设
\(
F_t(y):=\pi t^3\Delta_t(\bar\gamma+ty)
\)，则
\[
F_t(y)=\sigma^2 k_2(y)+\frac{\mu_3}{t}k_3(y)+O(t^{-2})
\]
在紧集上一致成立。
由隐函数机制可知对充分大 \(t\)，在 \(y_0\) 与 \(-y_0\) 的邻域中各存在唯一零点 \(y_t^+\to y_0\)、\(y_t^-\to -y_0\)，并且由一致收敛与 \(k_2\) 的符号结构知无其他零点。

取 \(y=y_0+\delta/t\) 并展开得
\[
0=F_t(y)=\sigma^2 k_2'(y_0)\frac{\delta}{t}+\frac{\mu_3}{t}k_3(y_0)+O(t^{-2}),
\]
从而 \(\delta=-\mu_3 k_3(y_0)/(\sigma^2 k_2'(y_0))\)。
直接代入 \(y_0=1/\sqrt3\) 化简得
\(
-k_3(y_0)/k_2'(y_0)=1/3
\)，故 \(\delta=\mu_3/(3\sigma^2)\)。
于是
\[
b_t=\bar\gamma+t y_t^+=\bar\gamma+\frac{t}{\sqrt3}+\frac{\mu_3}{3\sigma^2}+o(1),
\qquad
a_t=\bar\gamma+t y_t^-=\bar\gamma-\frac{t}{\sqrt3}+\frac{\mu_3}{3\sigma^2}+o(1).
\]
符号结构由 \(k_2\) 的符号与一致收敛直接推出。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$L^2$}{L2} 最优尺度修正的唯一性与常数减半]\label{thm:xi-poisson-cauchy-l2-optimal-scale-halving}\leanverified{paper\_xi\_poisson\_cauchy\_l2\_optimal\_scale\_halving}
在定理 \ref{thm:xi-poisson-cauchy-lp-sharp-spectrum} 的假设下，考虑同中心 Cauchy 族
\[
\tilde g_s(x):=\mathsf P_s(x-\bar\gamma),\qquad s>0.
\]
则存在 \(t_0\) 使对所有 \(t\ge t_0\) 存在唯一的极小化尺度 \(s_t>0\) 满足
\[
\|\tilde h_t-\tilde g_{s_t}\|_{L^2(\RR)}=\inf_{s>0}\|\tilde h_t-\tilde g_s\|_{L^2(\RR)}.
\]
并且该最优尺度具有一阶修正
\[
s_t=t+\frac{3\sigma^2}{4t}+o(t^{-1}),
\]
且最小误差满足锐渐近
\[
\inf_{s>0}\|\tilde h_t-\tilde g_s\|_{L^2(\RR)}
=\frac{\sqrt3}{8\sqrt\pi}\,\frac{\sigma^2}{t^{5/2}}+o(t^{-5/2}).
\]
\end{theorem}
\begin{proof}
由 \eqref{eq:xi-poisson-cauchy-second-order-splitting-rt} 与 \(\|\varepsilon_t\|_{L^2}=O(t^{-7/2})\)（由命题 \(11.667\)、命题 \(11.670\) 的一致界与插值界得到），可写
\[
\tilde h_t=\tilde g_t+\sigma^2 r_t+o_{L^2}(t^{-5/2}).
\]
设 \(s=t+\Delta\) 且 \(|\Delta|\le t^{1/2}\)。
由 \(\tilde g_s\) 对 \(s\) 的可微性与 Taylor 展开得
\[
\tilde g_s=\tilde g_t+\Delta\,\partial_t\tilde g_t+O\!\left(\Delta^2\|\partial_{tt}\tilde g_{\theta}\|\right),
\]
其中 \(\theta\) 介于 \(t\) 与 \(s\) 之间。
直接由显式表达式可得 \(\|\partial_t\tilde g_t\|_{L^2}\asymp t^{-3/2}\)、\(\|\partial_{tt}\tilde g_t\|_{L^2}\asymp t^{-5/2}\)，故当 \(\Delta=O(t^{-1})\) 时二阶余项为 \(o(t^{-5/2})\)。
因此在 \(L^2\) 意义下
\[
\tilde h_t-\tilde g_s
=\sigma^2 r_t-\Delta\,\partial_t\tilde g_t+o_{L^2}(t^{-5/2}).
\]
令 \(\Delta=a\sigma^2/t\) 并用尺度代换 \(y=(x-\bar\gamma)/t\) 得
\[
r_t(\bar\gamma+ty)=\frac{1}{\pi t^3}\frac{3y^2-1}{(1+y^2)^3},
\qquad
\partial_t\tilde g_t(\bar\gamma+ty)=\frac{1}{\pi t^2}\frac{y^2-1}{(1+y^2)^2}.
\]
从而
\[
\|\sigma^2 r_t-\Delta\,\partial_t\tilde g_t\|_{L^2}^2
=\frac{\sigma^4}{\pi^2 t^5}\int_{\RR}\left(\frac{3y^2-1}{(1+y^2)^3}-a\frac{y^2-1}{(1+y^2)^2}\right)^2\dd y+o(t^{-5}).
\]
右端为关于 \(a\) 的严格凸二次型，其唯一极小点由显式积分常数
\[
\int_{\RR}\frac{(3y^2-1)^2}{(1+y^2)^6}\dd y=\frac{3\pi}{16},
\quad
\int_{\RR}\frac{(3y^2-1)(y^2-1)}{(1+y^2)^5}\dd y=\frac{3\pi}{16},
\quad
\int_{\RR}\frac{(y^2-1)^2}{(1+y^2)^4}\dd y=\frac{\pi}{4}
\]
给出，解得 \(a=3/4\)，并且最小值对应的主导常数为 \(\frac{3\pi}{64}\)。
因此
\[
s_t=t+\frac{3\sigma^2}{4t}+o(t^{-1}),
\qquad
\inf_{s>0}\|\tilde h_t-\tilde g_s\|_{2}
=\frac{\sigma^2}{\pi t^{5/2}}\sqrt{\frac{3\pi}{64}}+o(t^{-5/2})
=\frac{\sqrt3}{8\sqrt\pi}\frac{\sigma^2}{t^{5/2}}+o(t^{-5/2}).
\]
唯一性由严格凸性得到。证毕。
\end{proof}
