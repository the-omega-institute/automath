\paragraph{Cayley 相位门、曲线奇异环与端点 Cauchy 账本}
沿用定理 \ref{thm:xi-cayley-joukowsky-harmonic-measure-ellipse} 的 Cayley 紧化
\[
\mathcal C(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t},
\qquad t\in\widehat{\RR},
\]
其中 \(\mathcal C(\widehat{\RR})=\TT\) 且 \(\mathcal C(\infty)=-1\)。
在这一相位门下，双频干涉曲线、单位圆触环、Lissajous 相位退化、端点 Poisson 权以及 Toeplitz 正性视界都可写成同一 \(\mathrm{U}(1)\) 轨道在不同坐标中的投影与账本公式。

\begin{theorem}[Cayley 相位门下的双旋转干涉闭式与奇偶对称]\label{thm:xi-cayley-phase-double-rotation-rational-parity}
\leanverified{paper\_xi\_cayley\_phase\_double\_rotation\_rational\_parity}
对每个整数 \(m\ge 0\)，定义整系数多项式
\[
P_m(t):=\sum_{j=0}^{m}(-1)^j\binom{2m}{2j}t^{2j},
\qquad
Q_m(t):=\sum_{j=0}^{m-1}(-1)^j\binom{2m}{2j+1}t^{2j+1}.
\]
其中按空和约定 \(Q_0=0\)。
则对一切 \(t\in\RR\) 都有
\[
\boxed{\;
\mathcal C(t)^m=\frac{P_m(t)+\mathrm{i}Q_m(t)}{(1+t^2)^m}.\;}
\]
其中 \(P_m\) 为偶多项式，\(Q_m\) 为奇多项式；特别地，
\[
\mathcal C(-t)^m=\overline{\mathcal C(t)^m}.
\]
进一步，对整数 \(p>q\ge 0\) 定义双旋转干涉轨道
\[
\Gamma_{p,q}(t):=\frac12\bigl(\mathcal C(t)^p+\mathcal C(t)^q\bigr).
\]
则存在 \(A_{p,q},B_{p,q}\in\ZZ[t]\) 使
\[
\boxed{\;
\Gamma_{p,q}(t)=\frac{A_{p,q}(t)+\mathrm{i}B_{p,q}(t)}{(1+t^2)^p},\;}
\]
其中 \(A_{p,q}\) 为偶多项式，\(B_{p,q}\) 为奇多项式，并满足
\[
\deg A_{p,q}\le 2p,
\qquad
\deg B_{p,q}\le 2p-1.
\]
\end{theorem}
\begin{proof}
由
\[
\mathcal C(t)^m
=
\left(\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\right)^m
=
\frac{(1+\mathrm{i}t)^{2m}}{(1+t^2)^m},
\]
展开 \((1+\mathrm{i}t)^{2m}\) 并分离偶次与奇次项，便得到所示 \(P_m,Q_m\) 的闭式；奇偶性由幂次直接给出。
\par
对第二部分，记 \(s:=p-q\ge 1\)，并定义
\[
E_s(t):=\frac{(1+\mathrm{i}t)^s+(1-\mathrm{i}t)^s}{2}\in\ZZ[t].
\]
则 \(E_s\) 为偶多项式，且
\[
\Gamma_{p,q}(t)
=
\frac{(1+\mathrm{i}t)^q\bigl((1+\mathrm{i}t)^s+(1-\mathrm{i}t)^s\bigr)}{2(1-\mathrm{i}t)^p}
=
\frac{(1+\mathrm{i}t)^{p+q}E_s(t)}{(1+t^2)^p}.
\]
再把
\[
(1+\mathrm{i}t)^{p+q}=U_{p,q}(t)+\mathrm{i}V_{p,q}(t)
\]
写成实部与虚部，其中 \(U_{p,q},V_{p,q}\in\ZZ[t]\)，且 \(U_{p,q}\) 为偶、\(V_{p,q}\) 为奇。
于是
\[
A_{p,q}:=E_s\,U_{p,q},
\qquad
B_{p,q}:=E_s\,V_{p,q}
\]
满足全部结论。
证毕。
\end{proof}

\begin{proposition}[玫瑰轨道的单位圆触环与奇偶分岔]\label{prop:xi-rose-unit-circle-touch-ring-parity-bifurcation}
\leanverified{paper\_xi\_rose\_unit\_circle\_touch\_ring\_parity\_bifurcation}
对互素整数 \(d>n\ge 1\)，定义
\[
\mathcal R_{n,d}
:=
\left\{
\frac12\bigl(w^{d+n}+w^{d-n}\bigr):\ w\in\TT
\right\}\subset\CC.
\]
记其单位圆触环为
\[
\Sigma_{n,d}:=\mathcal R_{n,d}\cap\TT.
\]
则有
\[
\boxed{\;
\Sigma_{n,d}
=
\{\zeta^{\,d-n}:\ \zeta^{2n}=1\}\subset\mu_{2n},\;}
\]
并且
\[
\boxed{\;
|\Sigma_{n,d}|
=
\frac{2n}{\gcd(2n,d)}
=
\begin{cases}
2n,& d\ \text{为奇},\\[2pt]
n,& d\ \text{为偶}.
\end{cases}\;}
\]
因此 \(\Sigma_{n,d}\) 作为 \(\TT\) 的有限子群，恰为阶 \(2n/\gcd(2n,d)\) 的根单位点群。
\end{proposition}
\begin{proof}
设
\[
z=\frac12\bigl(w^{d+n}+w^{d-n}\bigr)
=
w^{d-n}\frac{1+w^{2n}}{2}.
\]
由于 \(|w^{d-n}|=1\)，故 \(|z|=1\) 当且仅当 \(|1+w^{2n}|=2\)。
而 \(w^{2n}\in\TT\)，因此后一条件等价于 \(w^{2n}=1\)。
从而
\[
\Sigma_{n,d}
=
\{w^{d-n}:\ w^{2n}=1\}
=
\{\zeta^{\,d-n}:\ \zeta^{2n}=1\}.
\]
于是 \(|\Sigma_{n,d}|\) 等于幂映射
\(\mu_{2n}\to\mu_{2n}\), \(\zeta\mapsto \zeta^{d-n}\)
的像大小，即
\[
|\Sigma_{n,d}|=\frac{2n}{\gcd(2n,d-n)}.
\]
又因 \(\gcd(d,n)=1\)，有
\[
\gcd(2n,d-n)=\gcd(2n,d),
\]
故
\[
|\Sigma_{n,d}|=\frac{2n}{\gcd(2n,d)}.
\]
当 \(d\) 为奇数时 \(\gcd(2n,d)=1\)；当 \(d\) 为偶数时，由 \(\gcd(d,n)=1\) 知 \(n\) 必为奇数，因而 \(\gcd(2n,d)=2\)。
于是得到奇偶分岔公式。
证毕。
\end{proof}

\begin{theorem}[Lissajous 相位退化环的等距有限轨道与浸入判别]\label{thm:xi-lissajous-phase-singular-ring-immersion}
\leanverified{paper\_xi\_lissajous\_phase\_singular\_ring\_immersion}
对互素整数 \(a,b\ge 1\) 与相位 \(\delta\in\RR\)，定义
\[
\mathcal L_{a,b,\delta}(u):=
\bigl(\cos(au+\delta),\ \cos(bu)\bigr)\subset\RR^2.
\]
其相位退化环定义为
\[
\mathfrak S_{a,b}
:=
\left\{
\delta\ (\mathrm{mod}\ \pi):
\exists u\in\RR\ \text{使}\ \mathcal L_{a,b,\delta}'(u)=0
\right\}.
\]
则有严格等式
\[
\boxed{\;
\mathfrak S_{a,b}
=
\frac{\pi}{b}\ZZ\Big/\pi\ZZ,\;}
\]
因而
\[
|\mathfrak S_{a,b}|=b.
\]
并且：
\begin{enumerate}
  \item 若 \(\delta\notin\mathfrak S_{a,b}\)，则 \(\mathcal L_{a,b,\delta}\) 为浸入；
  \item 若 \(\delta\in\mathfrak S_{a,b}\)，则 \(\mathcal L_{a,b,\delta}\) 存在驻点，从而出现相位退化。
\end{enumerate}
\end{theorem}
\begin{proof}
直接计算得
\[
\mathcal L_{a,b,\delta}'(u)
=
\bigl(-a\sin(au+\delta),\ -b\sin(bu)\bigr).
\]
故 \(\mathcal L_{a,b,\delta}'(u)=0\) 当且仅当
\[
\sin(bu)=0,
\qquad
\sin(au+\delta)=0.
\]
前者给出
\[
u=\frac{k\pi}{b}\qquad(k\in\ZZ).
\]
代入后者得到
\[
\delta\equiv \ell\pi-\frac{ak\pi}{b}\pmod\pi
\qquad(\ell\in\ZZ).
\]
由于 \(\gcd(a,b)=1\)，当 \(k\) 在 \(0,1,\dots,b-1\) 变动时，
\(ak\bmod b\) 遍历全部剩余类，因此
\[
\mathfrak S_{a,b}=\frac{\pi}{b}\ZZ/\pi\ZZ.
\]
其余结论由驻点与浸入的定义立即推出。
证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\mathrm{U}(1)$}{U(1)} 干涉投影下的曲线统一]\label{cor:xi-cayley-u1-interference-projection-unification}
对整数 \(m\ge 1\) 与相位 \(\delta\in\RR\)，定义

X_{m,\delta}(t):=\Re\!\bigl(e^{\mathrm{i}\delta}\mathcal C(t)^m\bigr),
\qquad
Y_m(t):=\Re\!\bigl(\mathcal C(t)^m\bigr).

则成立：
\begin{enumerate}

  u=2\arctan t,

  =
  \bigl(\cos(au+\delta),\cos(bu)\bigr),

  \item 对任意互素整数 \(d>n\ge 1\)，有

\end{enumerate}

\begin{proof}

故第一条成立。
又若记 \(w=\mathcal C(t)\in\TT\)，则

这正是 \(\mathcal R_{n,d}\) 的复表示。
最后一条由命题 \ref{prop:xi-rose-unit-circle-touch-ring-parity-bifurcation} 与定理 \ref{thm:xi-lissajous-phase-singular-ring-immersion} 直接给出。
证毕。
\end{proof}

\begin{proposition}[有限循环轨道上的 Lissajous 采样与别名格分解]\label{prop:xi-lissajous-cyclic-orbit-alias-lattice}
令 \(N\ge 1\)，取互素整数 \(a,b\)，并定义
\[
\bigl(w_1(j),w_2(j)\bigr)
:=
\bigl(e^{2\pi \mathrm{i}aj/N},\ e^{2\pi \mathrm{i}bj/N}\bigr)
\in\TT^2,
\qquad j=0,1,\dots,N-1.
\]
则 \(\{(w_1(j),w_2(j))\}_{j=0}^{N-1}\) 构成 \(\TT^2\) 上一个阶 \(N\) 的循环子群轨道。
对频带有限的三角多项式
\[
H(w_1,w_2)
=
\sum_{|m|,|n|\le M}c_{m,n}\,w_1^m w_2^n,
\]
定义其消去格
\[
\Lambda_N(a,b):=
\{(m,n)\in\ZZ^2:\ am+bn\equiv 0\pmod N\}.
\]
再对 \(\ell\in\ZZ/N\ZZ\) 定义离散 Fourier 模
\[
\widehat H(\ell)
:=
\frac1N\sum_{j=0}^{N-1}
H\!\bigl(w_1(j),w_2(j)\bigr)\,
e^{-2\pi \mathrm{i}\ell j/N}.
\]
则有精确别名分解
\[
\boxed{\;
\widehat H(\ell)
=
\sum_{\substack{|m|,|n|\le M\\ am+bn\equiv \ell\;(\mathrm{mod}\,N)}}
c_{m,n}.\;}
\]
特别地，零频项满足
\[
\boxed{\;
\frac1N\sum_{j=0}^{N-1}H\!\bigl(w_1(j),w_2(j)\bigr)
=
\sum_{(m,n)\in\Lambda_N(a,b)}c_{m,n}.\;}
\]
因此：
\begin{enumerate}
  \item 若
  \[
  \Lambda_N(a,b)\cap\bigl([-M,M]^2\setminus\{(0,0)\}\bigr)=\varnothing,
  \]
  则零频平均无别名并且恰等于 \(c_{0,0}\)；
  \item 若模 \(N\) 线性形式
  \[
  \kappa_{a,b,N}:\ (m,n)\longmapsto am+bn\pmod N
  \]
  在频带 \([-M,M]^2\) 上单射，等价地
  \[
  \Lambda_N(a,b)\cap\bigl([-2M,2M]^2\setminus\{(0,0)\}\bigr)=\varnothing,
  \]
  则全部系数 \(c_{m,n}\) 都可由样本序列
  \(\{H(w_1(j),w_2(j))\}_{j=0}^{N-1}\)
  经有限离散 Fourier 反演唯一恢复。
\end{enumerate}
\end{proposition}
\leanverified{paper\_xi\_lissajous\_cyclic\_orbit\_alias\_lattice}
\begin{proof}
若 \((w_1(j_0),w_2(j_0))=(1,1)\)，则
\[
e^{2\pi \mathrm{i}a j_0/N}=e^{2\pi \mathrm{i}b j_0/N}=1.
\]
从而 \(N\mid a j_0\) 且 \(N\mid b j_0\)。
取整数 \(u,v\) 使 \(ua+vb=1\)，便得
\[
N\mid j_0(ua+vb)=j_0,
\]
故该轨道确有阶 \(N\)。
\par
对任意 \(\ell\in\ZZ/N\ZZ\)，交换求和次序得
\[
\begin{aligned}
\widehat H(\ell)
&=
\frac1N\sum_{j=0}^{N-1}
\sum_{|m|,|n|\le M}
c_{m,n}\,
e^{2\pi \mathrm{i}(am+bn)j/N}\,
e^{-2\pi \mathrm{i}\ell j/N}\\
&=
\sum_{|m|,|n|\le M}c_{m,n}\,
\frac1N\sum_{j=0}^{N-1}
e^{2\pi \mathrm{i}(am+bn-\ell)j/N}.
\end{aligned}
\]
括号内的有限几何和等于 \(1\) 当且仅当 \(am+bn\equiv \ell\pmod N\)，否则等于 \(0\)。
这就得到盒装恒等式；零频项为其特例。
\par
若 \(\Lambda_N(a,b)\cap([-M,M]^2\setminus\{(0,0)\})=\varnothing\)，则在零频壳内唯一留下的频点是 \((m,n)=(0,0)\)，故平均值等于 \(c_{0,0}\)。
更一般地，样本序列的离散 Fourier 变换把每个 \(\ell\) 上的系数压缩为同一同余壳的和，因此全部系数可唯一恢复，当且仅当 \(\kappa_{a,b,N}\) 在频带上单射。
而单射等价于不存在两点差落入 \(\Lambda_N(a,b)\setminus\{0\}\)，亦即
\[
\Lambda_N(a,b)\cap\bigl([-2M,2M]^2\setminus\{(0,0)\}\bigr)=\varnothing.
\]
证毕。
\end{proof}

\begin{theorem}[端点 Poisson 权的 Cayley 反推与 Cauchy 闭式]\label{thm:xi-endpoint-poisson-cauchy-cayley-pullback}
\leanverified{paper\_xi\_endpoint\_poisson\_cauchy\_cayley\_pullback}
固定 \(0<r<1\)，令 \(z_r:=-r\in\mathbb D\)。
其 Poisson 核记为
\[
P_{z_r}(\theta):=\frac{1-r^2}{|e^{\mathrm{i}\theta}+r|^2}.
\]
若以 Cayley 参数
\[
e^{\mathrm{i}\theta}=\mathcal C(t)
\qquad\Longleftrightarrow\qquad
\theta=2\arctan t
\]
回拉到实轴，则调和测度满足严格恒等式
\[
\boxed{\;
P_{z_r}(\theta)\,\frac{\dd\theta}{2\pi}
=
\frac{1}{\pi}\frac{\Gamma_r}{t^2+\Gamma_r^2}\,\dd t,\qquad
\Gamma_r:=\frac{1+r}{1-r}.\;}
\]
换言之，端点 \(-1\) 的内层 Poisson 权在 Cayley 反推下恰为尺度参数 \(\Gamma_r\) 的 Cauchy 分布。
\end{theorem}
\begin{proof}
由 \(\theta=2\arctan t\) 可得
\[
\cos\theta=\frac{1-t^2}{1+t^2},
\qquad
\dd\theta=\frac{2\,\dd t}{1+t^2}.
\]
于是
\[
|e^{\mathrm{i}\theta}+r|^2
=
1+r^2+2r\cos\theta
=
\frac{(1+r)^2+(1-r)^2t^2}{1+t^2}.
\]
代回 Poisson 核并整理得
\[
\begin{aligned}
P_{z_r}(\theta)\frac{\dd\theta}{2\pi}
&=
\frac{1-r^2}{|e^{\mathrm{i}\theta}+r|^2}\cdot \frac{\dd\theta}{2\pi}\\
&=
\frac{1-r^2}{\pi\bigl((1+r)^2+(1-r)^2t^2\bigr)}\,\dd t\\
&=
\frac{1}{\pi}\frac{(1+r)/(1-r)}{t^2+\bigl((1+r)/(1-r)\bigr)^2}\,\dd t,
\end{aligned}
\]
即得所示公式。
证毕。
\end{proof}

\begin{theorem}[Poisson--Jensen 的端点 Cauchy 账本形式]\label{thm:xi-endpoint-poisson-jensen-cauchy-ledger}
设 \(F\) 在 \(\overline{\mathbb D}\) 的邻域内全纯，且 \(F(0)\neq 0\)。
记其位于 \(\mathbb D\) 内的零点（按重数计）为 \(\{a_j\}\subset\mathbb D\)。
对 \(0<r<1\) 定义 \(z_r:=-r\) 与
\[
\Gamma_r:=\frac{1+r}{1-r}.
\]
则有完全等价的端点局部化 Poisson--Jensen 公式
\[
\boxed{\;
\log|F(-r)|
=
\int_{\RR}
\log|F(\mathcal C(t))|\,
\frac{1}{\pi}\frac{\Gamma_r}{t^2+\Gamma_r^2}\,\dd t
-\sum_j
\log\left|
\frac{-r-a_j}{1+r\overline{a_j}}
\right|.\;}
\]
因此，端点层的边界平均严格转写为一条 Cauchy 加权的实轴积分，而盘内零点贡献则以显式 Green 账本项逐个扣除。
\end{theorem}
\begin{proof}
经典 Poisson--Jensen 公式在点 \(z_r=-r\) 处给出
\[
\log|F(-r)|
=
\frac{1}{2\pi}\int_{0}^{2\pi}
\log|F(e^{\mathrm{i}\theta})|\,P_{z_r}(\theta)\,\dd\theta
-\sum_j
\log\left|
\frac{-r-a_j}{1+r\overline{a_j}}
\right|.
\]
再将边界积分中的测度替换为定理
\ref{thm:xi-endpoint-poisson-cauchy-cayley-pullback}
所给的 Cauchy 权表示，即得结论。
证毕。
\end{proof}

\begin{theorem}[Toeplitz 截断正性、Carath\'eodory 补全与子圆盘零点禁入]\label{thm:xi-toeplitz-caratheodory-completion-tail-exclusion}
设
\[
f(z)=a_0+2\sum_{n\ge 1}a_n z^n
\]
在 \(\mathbb D\) 内解析，并令 \(a_{-n}:=\overline{a_n}\)。
对 \(N\ge 0\) 定义 Toeplitz 截断
\[
\mathbf T_N:=(a_{j-k})_{0\le j,k\le N}.
\]
若 \(\mathbf T_N\succeq 0\)，则特别有 \(a_0\in\RR_{\ge 0}\)，并且存在一个 Carath\'eodory 补全
\[
g(z)=a_0+2\sum_{n=1}^{N}a_n z^n+2\sum_{n>N}b_n z^n
\]
满足 \(\Re g(z)\ge 0\) 于整个 \(\mathbb D\)。
进一步，若对某个 \(0<r<1\) 与 \(\varepsilon\ge 0\) 有
\[
\sum_{n>N}|a_n-b_n|\,r^n\le \varepsilon,
\]
则对所有 \(|z|\le r\) 都成立定量下界
\[
\boxed{\;
\Re f(z)\ge
\frac{1-r}{1+r}\,a_0-2\varepsilon.\;}
\]
特别地，若
\[
2\varepsilon<\frac{1-r}{1+r}\,a_0,
\]
则 \(f\) 在闭圆盘 \(|z|\le r\) 内无零点。
\end{theorem}
\begin{proof}
由 Carath\'eodory--Fej\'er 截断矩问题的标准完备性，\(\mathbf T_N\succeq 0\) 保证存在 Carath\'eodory 函数 \(g\) 与给定前 \(N\) 个 Fourier 矩一致；这与附录定理 \ref{thm:app-horizon-toeplitz-lmi} 的全阶等价属于同一矩表示机制。
\par
对该 \(g\) 取 Herglotz 表示
\[
g(z)=\mathrm{i}\beta+\int_{\TT}\frac{\xi+z}{\xi-z}\,\dd\mu(\xi),
\]
其中 \(\mu\) 为有限正测度。
于是
\[
\Re g(z)=\int_{\TT}\frac{1-|z|^2}{|\xi-z|^2}\,\dd\mu(\xi)
=\int_{\TT}P_z(\xi)\,\dd\mu(\xi).
\]
将 \(z=0\) 代入 Herglotz 表示得
\[
g(0)=\mathrm{i}\beta+\mu(\TT).
\]
而按构造 \(g(0)=a_0\in\RR\)，故 \(\beta=0\) 且 \(\mu(\TT)=a_0\)。
对 \(|z|\le r\)，Poisson 核满足
\[
P_z(\xi)\ge \frac{1-|z|}{1+|z|}\ge \frac{1-r}{1+r},
\]
从而
\[
\Re g(z)\ge \frac{1-r}{1+r}\,a_0.
\]
\par
另一方面，对 \(|z|\le r\) 有
\[
\begin{aligned}
|f(z)-g(z)|
&=
2\left|
\sum_{n>N}(a_n-b_n)z^n
\right|\\
&\le
2\sum_{n>N}|a_n-b_n|\,|z|^n\\
&\le
2\sum_{n>N}|a_n-b_n|\,r^n\\
&\le 2\varepsilon.
\end{aligned}
\]
因此
\[
\Re f(z)\ge \Re g(z)-|f(z)-g(z)|
\ge
\frac{1-r}{1+r}\,a_0-2\varepsilon.
\]
若右端严格为正，则 \(f\) 在 \(|z|\le r\) 上实部处处正，故不可能取零值，于是得到零点禁入结论。
证毕。
\end{proof}

\begin{conjecture}[端点共动 Toeplitz 截断的有限闭包]\label{conj:xi-endpoint-comoving-toeplitz-finite-closure}
沿用附录 \ref{subsubsec:app-horizon-weyl-carath-toeplitz} 的视界 Carath\'eodory 接口
\(\mathscr C_{\zeta}\)。
对 \(0<r<1\)，定义将 \(-r\) 送到原点的圆盘自同构
\[
\phi_r(z):=\frac{z+r}{1+r z},
\qquad
\mathscr C_{\zeta,r}^{\mathrm{mov}}(z):=\mathscr C_{\zeta}\!\bigl(\phi_r^{-1}(z)\bigr).
\]
记 \(\mathbf T_N(\mathscr C_{\zeta,r}^{\mathrm{mov}})\) 为其 Maclaurin 系数生成的 Toeplitz 主子矩阵。
由于 Carath\'eodory 类在 \(\mathrm{Aut}(\mathbb D)\) 下不变，附录定理 \ref{thm:app-horizon-toeplitz-lmi} 立即给出：RH 成立当且仅当
\[
\mathbf T_N(\mathscr C_{\zeta,r}^{\mathrm{mov}})\succeq 0
\qquad
(\forall\,r\in(0,1),\ \forall\,N\ge 0).
\]
预期更强地存在显式函数
\[
r_\star=r_\star(T,\eta)\uparrow 1,
\qquad
N_\star=N_\star(T,\eta),
\qquad
\varepsilon_\star=\varepsilon_\star(T,\eta)\downarrow 0,
\]
使得以下有限截断判据成立：
若
\[
\mathbf T_n(\mathscr C_{\zeta,r_\star}^{\mathrm{mov}})\succeq 0
\qquad (0\le n\le N_\star)
\]
并且与该前缀矩数据相容的 Carath\'eodory 补全在 \(|z|\le r_\star\) 上的尾项误差不超过 \(\varepsilon_\star\)（按定理 \ref{thm:xi-toeplitz-caratheodory-completion-tail-exclusion} 的口径），则 \(\Xi_{\zeta}\) 在高度窗
\[
|\Im \rho|\le T
\]
内不存在满足
\[
\left|\Re \rho-\frac12\right|\ge \eta
\]
的零点。
换言之，端点共动图表下的有限 Toeplitz--PSD 证书族应足以把有限高度范围内的圆盘零点禁入压缩为显式的有限阶半正定核验问题。
\end{conjecture}
