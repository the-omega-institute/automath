\paragraph{Fibonacci 幂化矩阵与例外谱幂和的词--项链闭式}
沿用前述记号。对 \(q\ge 2\)，令
\[
b^{(q)}:=\bigl(\tbinom q0,\tbinom q1,\dots,\tbinom qq\bigr)^{\mathsf T},
\qquad
\mathbf 1:=(1,\dots,1)^{\mathsf T}\in\RR^{q+1},
\]
\[
C^{(q)}_{i,j}:=\binom{q-i}{j}\qquad(0\le i,j\le q),
\qquad
B^{(q)}:=\mathbf 1\,(b^{(q)})^{\mathsf T},
\qquad
A^{(q)}:=\frac12\bigl(B^{(q)}+C^{(q)}\bigr).
\]
记例外谱幂和
\[
S_m(q):=\sum_{i=0}^{q}\bigl(\lambda_i^{(q)}\bigr)^m=\Tr\!\bigl((A^{(q)})^m\bigr)\qquad(m\ge 1).
\]

\begin{conclusion}[Fibonacci 幂化矩阵系数恒等式]\label{con:xi-replica-softcore-fibonacci-powered-matrix-coefficient-identity}
对任意整数 \(q\ge 2\) 与 \(r\ge 0\)，成立严格恒等式
\[
\boxed{\;
b^{(q)\mathsf T}(C^{(q)})^{r}\mathbf 1
=F_{r+3}^{\,q}\; }.
\]
\end{conclusion}
\begin{proof}
按矩阵乘积展开：
\[
b^{(q)\mathsf T}(C^{(q)})^r\mathbf 1
=\sum_{k_0,\dots,k_r=0}^{q}\binom{q}{k_0}\prod_{t=1}^{r}\binom{q-k_{t-1}}{k_t}.
\]
对固定 \((k_0,\dots,k_r)\)，其组合意义为：先从 \([q]\) 选 \(S_0\subseteq[q]\) 且 \(|S_0|=k_0\)，再对每个 \(t\ge 1\) 从 \([q]\setminus S_{t-1}\) 选 \(S_t\) 且 \(|S_t|=k_t\)。故等价计数对象为
\[
(S_0,\dots,S_r),\qquad
S_t\subseteq[q],\ \ S_t\cap S_{t-1}=\varnothing\ (t\ge 1).
\]
对每个坐标 \(u\in[q]\)，定义
\[
x_t(u):=\mathbf 1_{\{u\in S_t\}}\in\{0,1\}\qquad(0\le t\le r).
\]
条件 \(S_t\cap S_{t-1}=\varnothing\) 等价于
\[
x_{t-1}(u)\,x_t(u)=0\qquad(1\le t\le r),
\]
即长度 \(r+1\) 的二元序列中不出现相邻两个 \(1\)。
设 \(a_n\) 为长度 \(n\) 的该类序列个数，则
\[
a_n=a_{n-1}+a_{n-2},\qquad a_1=2,\ a_2=3,
\]
故 \(a_{r+1}=F_{r+3}\)。
不同坐标 \(u\) 的选择相互独立，因而总数为 \((F_{r+3})^q\)。证毕。
\end{proof}

\begin{conclusion}[例外谱全阶幂和的词--项链 Fibonacci 乘积展开]\label{con:xi-replica-softcore-exceptional-power-sum-word-necklace-fibonacci-product}
定义
\[
\Omega_m(q):=\Tr\!\bigl((C^{(q)})^m\bigr)
=
\begin{cases}
\dfrac{\varphi^{m(q+2)}+(-1)^q\varphi^{-mq}}{\varphi^{2m}+1},& m\ \text{为奇数},\\[10pt]
\dfrac{\varphi^{m(q+2)}-\varphi^{-mq}}{\varphi^{2m}-1},& m\ \text{为偶数},
\end{cases}
\]
其中 \(\varphi=\dfrac{1+\sqrt5}{2}\)。
对 \(\mathcal W_m:=\{0,1\}^m\)，任取 \(w\in\mathcal W_m\setminus\{0^m\}\)，记 \(t(w)\) 为 \(w\) 中 \(1\) 的个数。
按圆周顺序将相邻两个 \(1\) 之间的 \(0\)-段长记为
\[
\ell_1(w),\dots,\ell_{t(w)}(w)\in\ZZ_{\ge 0},
\qquad
\sum_{j=1}^{t(w)}\ell_j(w)=m-t(w),
\]
并定义
\[
\Phi(w):=\prod_{j=1}^{t(w)}F_{\ell_j(w)+3}\in\ZZ_{\ge 2}.
\]
则对一切 \(m\ge 1\)、\(q\ge 2\) 有
\[
\boxed{\;
2^m S_m(q)=\Omega_m(q)+\sum_{w\in\mathcal W_m\setminus\{0^m\}}\bigl(\Phi(w)\bigr)^q\; }.
\]
等价地，若 \([w]\) 表示循环移位轨道（即二元项链），\(|\mathrm{Orb}(w)|\) 为其轨道大小，则
\[
\boxed{\;
2^m S_m(q)=\Omega_m(q)+\sum_{[w]\subset\mathcal W_m\setminus\{0^m\}}|\mathrm{Orb}(w)|\cdot\bigl(\Phi(w)\bigr)^q\; }.
\]
\end{conclusion}
\begin{proof}
由 \(A^{(q)}=\frac12(B^{(q)}+C^{(q)})\)，得
\[
2^mS_m(q)=\Tr\!\bigl((B^{(q)}+C^{(q)})^m\bigr)
=\sum_{\omega\in\{B,C\}^m}\Tr(\omega),
\]
其中 \(\omega\) 遍历长度 \(m\) 的线性词并按矩阵乘积解释。

若 \(\omega=C^m\)，贡献为 \(\Tr((C^{(q)})^m)=\Omega_m(q)\)。
又 \(C^{(q)}=B_q^{\mathsf T}\)、\(B_q=\Sym^q(K)\)（\(K=\begin{psmallmatrix}1&1\\1&0\end{psmallmatrix}\)），故
\[
\mathrm{spec}\!\bigl((C^{(q)})^m\bigr)
=\bigl\{\varphi^{m(q-j)}(-\varphi^{-1})^{mj}:0\le j\le q\bigr\},
\]
从而
\[
\Omega_m(q)=\sum_{j=0}^{q}(-1)^{mj}\varphi^{m(q-2j)},
\]
这是公比 \(((-1)^m\varphi^{-2m})\) 的几何级数，直接求和即得上式闭式。

若 \(\omega\) 含至少一个 \(B\)，利用迹的循环不变性，可写成
\[
\omega\sim C^{\ell_1}BC^{\ell_2}B\cdots C^{\ell_t}B,
\qquad
\ell_j\ge 0,\ \sum_{j=1}^{t}\ell_j=m-t.
\]
由 \(B^{(q)}=\mathbf1(b^{(q)})^{\mathsf T}\) 为秩一矩阵，
\[
\Tr(\omega)
=\prod_{j=1}^{t}\Bigl(b^{(q)\mathsf T}(C^{(q)})^{\ell_j}\mathbf1\Bigr).
\]
应用结论 \ref{con:xi-replica-softcore-fibonacci-powered-matrix-coefficient-identity}，
\[
\Tr(\omega)=\prod_{j=1}^{t}F_{\ell_j+3}^{\,q}
=\Bigl(\prod_{j=1}^{t}F_{\ell_j+3}\Bigr)^q
=\Phi(w)^q,
\]
其中 \(w\) 为对应二元词（\(1\leftrightarrow B,\ 0\leftrightarrow C\)）。
对全部线性词求和即得第一式；按循环轨道聚合即得项链式。证毕。
\end{proof}

\begin{corollary}[例外谱二次幂和的闭式]\label{cor:xi-replica-softcore-exceptional-power-sum-square-closed-form-via-fib-necklace}
对一切 \(q\ge 2\)，有
\[
\boxed{\;
S_2(q)=\frac14\Bigl(4^q+2\cdot 3^q+F_{2q+2}\Bigr)\; }.
\]
\leanverified{paper\_xi\_replica\_softcore\_exceptional\_power\_sum\_square\_closed\_form\_via\_fib\_necklace}
\end{corollary}
\begin{proof}
在结论 \ref{con:xi-replica-softcore-exceptional-power-sum-word-necklace-fibonacci-product} 中取 \(m=2\)。
此时 \(\Omega_2(q)=F_{2q+2}\)。
非零词为 \(11,10,01\)：其中
\[
\Phi(11)=F_3F_3=4,\qquad
\Phi(10)=\Phi(01)=F_4=3.
\]
故
\[
4S_2(q)=F_{2q+2}+4^q+2\cdot 3^q,
\]
整理即得结论。证毕。
\end{proof}

\begin{conclusion}[六阶与七阶例外幂和的完全指数分解]\label{con:xi-replica-softcore-exceptional-power-sum-m6m7-full-exponential-decomposition}
对一切 \(q\ge 2\)，同时成立
\[
\boxed{
S_6(q)=\frac1{64}\Big(
64^q+6\cdot48^q+6\cdot40^q+9\cdot36^q+6\cdot32^q+12\cdot30^q+2\cdot27^q
+6\cdot26^q+3\cdot25^q+6\cdot24^q+6\cdot21^q+\Omega_6(q)
\Big)},
\]
\[
\Omega_6(q)=\frac{\varphi^{6(q+2)}-\varphi^{-6q}}{\varphi^{12}-1}\in\ZZ,
\]
\[
\boxed{
\begin{aligned}
S_7(q)=\frac1{128}\Big(&
128^q+7\cdot96^q+7\cdot80^q+14\cdot72^q+7\cdot64^q+21\cdot60^q
+7\cdot54^q+7\cdot52^q+7\cdot50^q\\
&+14\cdot48^q+7\cdot45^q+7\cdot42^q+7\cdot40^q+7\cdot39^q+7\cdot34^q
+\Omega_7(q)\Big),
\end{aligned}}
\]
\[
\Omega_7(q)=\frac{\varphi^{7(q+2)}+(-1)^q\varphi^{-7q}}{\varphi^{14}+1}\in\ZZ.
\]
\end{conclusion}
\begin{proof}
由结论 \ref{con:xi-replica-softcore-exceptional-power-sum-word-necklace-fibonacci-product}，
对固定 \(m\) 有
\[
2^mS_m(q)=\Omega_m(q)+\sum_{w\in\{0,1\}^m\setminus\{0^m\}}\Phi(w)^q.
\]
故只需对长度 \(m\) 的循环二元词按 \(\Phi(w)\) 取值聚合计数。

当 \(m=6\) 时，按 \(t(w)\) 与圆周间隔分拆统计得到：
\[
\begin{array}{c|c}
\Phi\ \text{底数} & \text{出现次数}\\\hline
64,48,40,36,32,30,27,26,25,24,21
&1,6,6,9,6,12,2,6,3,6,6
\end{array}
\]
并与 \(\Omega_6\) 合并即得第一式。

当 \(m=7\) 时，同理可得：
\[
\begin{array}{c|c}
\Phi\ \text{底数} & \text{出现次数}\\\hline
128,96,80,72,64,60,54,52,50,48,45,42,40,39,34
&1,7,7,14,7,21,7,7,7,14,7,7,7,7,7
\end{array}
\]
与 \(\Omega_7\) 合并即得第二式。证毕。
\end{proof}

\begin{conclusion}[固定阶 \(m\) 时 \(q\)-方向最小线性递推]\label{con:xi-replica-softcore-fixed-m-q-minimal-recurrence}
固定整数 \(m\ge 1\)。定义
\[
\mathcal B_m:=\bigl\{\Phi(w):w\in\{0,1\}^m\setminus\{0^m\}\bigr\}\subset\ZZ_{\ge 2},
\]
并将结论 \ref{con:xi-replica-softcore-exceptional-power-sum-word-necklace-fibonacci-product} 按相同底数聚合：
\[
2^mS_m(q)=\Omega_m(q)+\sum_{\alpha\in\mathcal B_m}c_m(\alpha)\,\alpha^q,
\qquad c_m(\alpha)\in\ZZ_{>0}.
\]
则 \(q\mapsto 2^mS_m(q)\) 的最小特征多项式为
\[
\boxed{
P_m(X)=\bigl(X^2-L_mX+(-1)^m\bigr)\prod_{\alpha\in\mathcal B_m}(X-\alpha),
}
\]
其中 \(L_m\) 为 Lucas 数。
因此最小递推阶恰为 \(|\mathcal B_m|+2\)，其全部特征根恰由两类组成：
\[
\{\alpha:\alpha\in\mathcal B_m\}\ \cup\ \{\varphi^m,\ (-\varphi^{-1})^m\}.
\]
\end{conclusion}
\begin{proof}
由上式分解，\(\alpha^q\) 被算子 \(E-\alpha\) 湮灭（\(E\) 为移位 \(q\mapsto q+1\)）。
而 \(\Omega_m(q)\) 满足
\[
\Omega_m(q+2)-L_m\Omega_m(q+1)+(-1)^m\Omega_m(q)=0,
\]
故被 \(E^2-L_mE+(-1)^m\) 湮灭。
于是 \(P_m(E)\) 湮灭 \(2^mS_m(q)\)。

下面证最小性。
将 \(\Omega_m\) 写成
\[
\Omega_m(q)=u_m(\varphi^m)^q+v_m\bigl((-\varphi^{-1})^m\bigr)^q,
\qquad
u_m,v_m\neq 0.
\]
故 \(2^mS_m(q)\) 是以下互异底数指数序列的线性组合：
\[
\{\alpha^q\}_{\alpha\in\mathcal B_m}\ \cup
\{(\varphi^m)^q,\ ((-\varphi^{-1})^m)^q\}.
\]
其中 \(\mathcal B_m\subset\ZZ_{\ge 2}\)，而 \(\varphi^m,(-\varphi^{-1})^m\notin\ZZ\)，且彼此不同。
在特征 \(0\) 的域上，不同底数指数序列线性无关（Vandermonde 判别）。
因此任何湮灭 \(2^mS_m\) 的多项式算子都必须包含全部一次因子
\((E-\alpha)\)（\(\alpha\in\mathcal B_m\)）以及二次因子
\(E^2-L_mE+(-1)^m\)。
故最小特征多项式恰为 \(P_m\)。证毕。
\end{proof}
