\paragraph{右端谱隙、有限阶递归与自然边界阻断}\label{par:xi-discrete-abel-weil-right-edge-gap-recursion-natural-boundary}
固定 \(b>1\)，并记
\[
E_b(n):=b^{-n/2}\bigl(\psi_0(b^n)-b^n\bigr),\qquad
\mathcal A_b(r):=\sum_{n\ge 0}E_b(n)r^n.
\]
由离散 Abel--Weil 展开（定理 \ref{thm:discrete-abel-weil-expansion}，对应文内定理 11.860 的离散化口径）可得
\[
\mathcal A_b(r)
=-\sum_{\rho}\frac{1}{\rho}\cdot\frac{1}{1-r\,b^{\rho-\frac12}}+H_b(r),
\]
其中 \(\rho\) 遍历 \(\zeta(s)\) 的非平凡零点，且 \(H_b\) 在 \(|r|<b^{1/2}\) 解析。
因此零点 \(\rho\) 与 \(r\)-平面极点
\(
r_\rho=b^{-(\rho-\frac12)}
\)
一一对应于同一部分分式结构。

\begin{theorem}[右端谱隙诱导的二阶递归与指数误差界]\label{thm:xi-abel-rightedge-gap-ar2-exp-error}
\leanverified{paper\_xi\_abel\_rightedge\_gap\_ar2\_exp\_error}
固定 \(b>1\)。设存在一对共轭非平凡零点
\[
\rho_\star=\beta_\star+\mathrm{i}\gamma_\star,\qquad
\overline{\rho_\star}=\beta_\star-\mathrm{i}\gamma_\star,
\]
满足 \(\beta_\star>\frac12\)，并存在 \(\eta>0\) 使得除该对之外的全部非平凡零点 \(\rho\) 均满足
\[
\Re(\rho)\le \beta_\star-\eta.
\]
记
\[
\alpha:=b^{\rho_\star-\frac12}
=b^{\beta_\star-\frac12}e^{\mathrm{i}\gamma_\star\log b}.
\]
则
\[
E_b(n+2)-2\Re(\alpha)\,E_b(n+1)+|\alpha|^2E_b(n)
=O\!\left(\left(b^{\beta_\star-\eta-\frac12}\right)^n\right)
\qquad(n\to\infty).
\]
且
\[
2\Re(\alpha)=2b^{\beta_\star-\frac12}\cos(\gamma_\star\log b),\qquad
|\alpha|^2=b^{2(\beta_\star-\frac12)}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:discrete-abel-weil-expansion}，极点位于 \(r_\rho=b^{-(\rho-\frac12)}\)。
在给定谱隙下，
\[
r_\star=b^{-(\rho_\star-\frac12)}=\alpha^{-1},\qquad
\overline{r_\star}=\overline{\alpha}^{-1}
\]
是离原点最近的一对极点，且其余极点满足
\[
|r_\rho|\ge b^\eta|r_\star|.
\]
定义剥离函数
\[
R(r):=\mathcal A_b(r)+\frac{1/\rho_\star}{1-\alpha r}
+\frac{1/\overline{\rho_\star}}{1-\overline{\alpha}r}.
\]
则 \(R\) 在任意 \(\varrho_0<b^\eta|r_\star|=b^{-(\beta_\star-\eta-\frac12)}\) 的圆盘内解析。
若 \(R(r)=\sum_{n\ge 0}r_n r^n\)，Cauchy 估计给出
\[
r_n=O\!\left(\left(b^{\beta_\star-\eta-\frac12}\right)^n\right).
\]
从而
\[
E_b(n)=-\frac{\alpha^n}{\rho_\star}-\frac{\overline{\alpha}^n}{\overline{\rho_\star}}+r_n.
\]
由于 \(\alpha^n,\overline{\alpha}^n\) 都被特征多项式
\[
(\lambda-\alpha)(\lambda-\overline{\alpha})
=\lambda^2-2\Re(\alpha)\lambda+|\alpha|^2
\]
湮没，二阶算子作用后仅剩余项贡献，得到所述指数误差界。证毕。
\end{proof}

\begin{theorem}[指数精度二阶递归强制右端零点出现]\label{thm:xi-abel-ar2-exp-error-forces-rightedge-zero}
\leanverified{paper\_xi\_abel\_ar2\_exp\_error\_forces\_rightedge\_zero}
固定 \(b>1\)。设存在 \(A\in\RR\)、\(B>0\)、\(0<\theta<1\) 与常数 \(C>0\)，使得
\[
\bigl|E_b(n+2)-AE_b(n+1)+BE_b(n)\bigr|\le C\theta^n
\qquad(\forall n\ge 0).
\]
令
\[
Q(r):=1-Ar+Br^2,\qquad
\lambda_{1,2}:=\frac{A\pm\sqrt{A^2-4B}}{2}.
\]
若进一步满足
\[
\limsup_{n\to\infty}|E_b(n)|^{1/n}
=\max\{|\lambda_1|,|\lambda_2|\}
>\theta,
\]
则 \(\mathcal A_b(r)\) 在 \(|r|<1/\theta\) 内存在不可去极点 \(r_0\)，并满足
\[
|r_0|=\frac{1}{\max\{|\lambda_1|,|\lambda_2|\}}<1.
\]
因此存在非平凡零点 \(\rho\) 使得
\[
r_0=b^{-(\rho-\frac12)}
\quad\Longleftrightarrow\quad
\rho=\frac12-\frac{\log r_0}{\log b},
\]
从而 \(\Re(\rho)>\frac12\)。
\end{theorem}
\begin{proof}
定义残差
\[
\varepsilon_n:=E_b(n+2)-AE_b(n+1)+BE_b(n),\qquad
\mathcal E(r):=\sum_{n\ge 0}\varepsilon_n r^n.
\]
由 \(|\varepsilon_n|\le C\theta^n\) 得 \(\mathcal E\) 在 \(|r|<1/\theta\) 内解析。
直接计算得
\[
Q(r)\mathcal A_b(r)
=E_b(0)+\bigl(E_b(1)-AE_b(0)\bigr)r+r^2\mathcal E(r),
\]
故 \(\mathcal A_b\) 在 \(|r|<1/\theta\) 内至多在 \(Q(r)=0\) 处有极点。

若 \(Q\) 的两根在该圆盘内均可去，则 \(\mathcal A_b\) 在 \(|r|<1/\theta\) 全纯，从而
\(
\limsup |E_b(n)|^{1/n}\le \theta
\)，与假设矛盾。
故至少存在一根 \(r_0\) 为不可去极点。

再由 Cauchy--Hadamard 公式，
\[
\frac{1}{R}=\limsup_{n\to\infty}|E_b(n)|^{1/n}
=\max\{|\lambda_1|,|\lambda_2|\},
\]
其中 \(R\) 为 \(\mathcal A_b\) 在 \(0\) 处收敛半径，因此
\(
R=1/\max\{|\lambda_1|,|\lambda_2|\}
\)。
而在 \(|r|<1/\theta\) 内的奇性只能来自 \(Q(r)=0\)，故至少存在不可去根 \(r_0\) 满足 \(|r_0|=R\)，即所示模长公式。

又 \(b>1\Rightarrow b^{1/2}>1\)，而 \(|r_0|<1\)，故 \(r_0\) 位于 \(H_b\) 的解析域内。
据定理 \ref{thm:discrete-abel-weil-expansion}，该极点必为某个
\(
r_\rho=b^{-(\rho-\frac12)}
\)。
由 \(|r_0|<1\) 得 \(\Re(\rho)>\frac12\)。证毕。
\end{proof}

\begin{theorem}[临界线极点稠密性导出单位圆自然边界]\label{thm:xi-abel-rh-dense-poles-natural-boundary}
\leanverified{paper\_xi\_abel\_rh\_dense\_poles\_natural\_boundary}
固定 \(b>1\)。假设 RH 成立，则每个非平凡零点写作
\(
\rho=\frac12+\mathrm{i}\gamma
\)。
定义
\[
\mathcal R_b:=\left\{e^{-\mathrm{i}\gamma\log b}:\ \zeta\!\left(\frac12+\mathrm{i}\gamma\right)=0\right\}\subset\{|r|=1\}.
\]
若 \(\mathcal R_b\) 在单位圆上稠密，且不存在沿单位圆弧的无限主部完全抵消机制（例如
\(\gamma\mapsto e^{-\mathrm{i}\gamma\log b}\) 在零点集合上满足非共振注入性），
则单位圆 \(|r|=1\) 是 \(\mathcal A_b(r)\) 的自然边界。
\end{theorem}
\begin{proof}
在 RH 下，每个零点项对应
\[
\frac{1}{1-r\,b^{\rho-\frac12}}
=\frac{1}{1-r\,e^{\mathrm{i}\gamma\log b}},
\]
其边界奇点位于 \(r=e^{-\mathrm{i}\gamma\log b}\in\{|r|=1\}\)。
余项 \(H_b\) 在 \(|r|<b^{1/2}\) 解析，不能在单位圆上制造或消去上述不可去奇性。
若 \(\mathcal R_b\) 稠密且不存在无限抵消，则单位圆任意弧段均含不可去奇点，故不存在跨越单位圆任一点的解析延拓；即单位圆为自然边界。证毕。
\end{proof}

\begin{corollary}[自然边界下不存在有限阶指数误差锁定]\label{cor:xi-abel-natural-boundary-no-finite-order-exp-lock}
在定理 \ref{thm:xi-abel-rh-dense-poles-natural-boundary} 的假设下，
对任意固定阶 \(d\ge 1\) 与常系数 \((c_0,\dots,c_d)\neq 0\)，不存在 \(0<\theta<1\) 使
\[
\sum_{j=0}^{d}c_jE_b(n+j)=O(\theta^n)\qquad(n\to\infty).
\]
\end{corollary}
\begin{proof}
反设存在上述关系。令
\[
Q_d(r):=\sum_{j=0}^{d}c_jr^j,\qquad
\mathcal E_d(r):=\sum_{n\ge 0}\left(\sum_{j=0}^{d}c_jE_b(n+j)\right)r^n.
\]
则 \(\mathcal E_d\) 在 \(|r|<1/\theta\) 解析，且
\[
Q_d(r)\mathcal A_b(r)=P_{d-1}(r)+r^d\mathcal E_d(r),
\]
其中 \(P_{d-1}\) 为至多 \(d-1\) 次多项式。
故 \(\mathcal A_b\) 在 \(|r|<1/\theta\) 亚纯，且因 \(1/\theta>1\) 可跨越单位圆局部延拓。
这与定理 \ref{thm:xi-abel-rh-dense-poles-natural-boundary} 的自然边界矛盾。证毕。
\end{proof}

\begin{conjecture}[对数采样 Abel--Weil 边界极点的普适稠密性]\label{conj:xi-abel-universal-boundary-pole-density}
固定 \(b>1\)。在 RH 前提下，
\[
\mathcal R_b=\left\{e^{-\mathrm{i}\gamma\log b}:\ \zeta\!\left(\frac12+\mathrm{i}\gamma\right)=0\right\}
\]
在单位圆上稠密。若该猜想成立，则 \(\mathcal A_b(r)\) 以单位圆为自然边界，且由推论
\ref{cor:xi-abel-natural-boundary-no-finite-order-exp-lock} 可知
\(\{E_b(n)\}\) 不存在任何“有限阶常系数 + 严格指数误差”的递归闭合。
\end{conjecture}
