\begin{theorem}[圆维约束下 Gödel 素数账本的 Lambert--\(W\) 精细下界]\label{thm:xi-terminal-zm-cdim-prime-ledger-lambertw-lower-bound}
\leanverified{paper\_xi\_terminal\_zm\_cdim\_prime\_ledger\_lambertw\_lower\_bound}
设有限支撑指数向量 \(r:\mathbb P\to\NN\) 的 Gödel 账本编码为
\[
N(r):=\prod_{p}p^{r(p)}.
\]
令 \(\cdim(G):=\dim(\widehat G^\circ)\)，并记 \(r_0:=\cdim(G)\in\NN\)。
若系统需记录 \(N\) 个可区分状态且保持 \(r_0\) 个独立相位通道可审计，则所需素数轴数 \(k\) 满足
\[
k\ge \frac{r_0\log N}{\mathrm W(r_0\log N)}\,(1-o(1)),
\]
其中 \(\mathrm W\) 为 Lambert--\(W\) 函数主支。等价地，
\[
k\ge
\frac{r_0\log N}
{\log(r_0\log N)-\log\log(r_0\log N)}\,(1+o(1)).
\]
\end{theorem}
\begin{proof}
记 primorial
\[
P_k^\#:=\prod_{i=1}^k p_i,\qquad \log P_k^\#=\vartheta(p_k),
\]
其中 \(\vartheta(x)=\sum_{p\le x}\log p\)。由素数定理，
\[
\vartheta(p_k)\sim p_k\sim k\log k.
\]
可审计容量要求给出 \(\log P_k^\#\gtrsim r_0\log N\)。设
\(
A:=r_0\log N
\)，则必须有
\[
k\log k\gtrsim A.
\]
方程 \(k\log k=A\) 的解为
\[
k=\frac{A}{\mathrm W(A)}.
\]
故得到
\[
k\ge \frac{A}{\mathrm W(A)}(1-o(1))
=\frac{r_0\log N}{\mathrm W(r_0\log N)}(1-o(1)).
\]
再由
\(
\mathrm W(A)=\log A-\log\log A+o(1)
\)
即得第二式。证毕。
\end{proof}

\begin{theorem}[临界值平方性与偶符号亏损]\label{thm:xi-terminal-zm-jg-critical-square-even-sign-defect}
\leanverified{paper\_xi\_terminal\_zm\_jg\_critical\_square\_even\_sign\_defect}
设 \(K\) 为特征不为 \(2\) 的域，\(P(z)=a_n z^n+\cdots+a_0\in K[z]\) 可分离且 \(a_n a_0\neq 0\)。令 \(F:=K(r)\)，并定义
\[
Q_r(w):=\operatorname{Res}_z\!\bigl(P(z),\,r z^2-w z+r^{-1}\bigr)\in F[w].
\]
记 \(w_1,\dots,w_n\) 为 \(Q_r\) 在某一分裂域中的根（按重数），并置 \(\Delta_i:=w_i^2-4\)。则：
\begin{enumerate}
  \item
  \[
  Q_r(2)=r^n P(1/r)^2,\qquad Q_r(-2)=r^n P(-1/r)^2.
  \]
  \item
  \[
  \prod_{i=1}^n (w_i^2-4)=
  \left(\frac{r^n P(1/r)P(-1/r)}{a_n a_0}\right)^2\in F.
  \]
  特别地，若 \(P(1/r)P(-1/r)\neq 0\)，则该乘积在 \(F^\times/(F^\times)^2\) 中为平凡类。
  \item 若 \(P(1/r)P(-1/r)\neq 0\)，设 \(M/F\) 为 \(Q_r\) 的分裂域，并令
  \[
  N:=M\bigl(\sqrt{\Delta_1},\dots,\sqrt{\Delta_n}\bigr),
  \]
  则
  \[
  [N:M]\mid 2^{n-1},
  \]
  且在 Kummer 理论同构下有
  \[
  \operatorname{Gal}(N/M)\subseteq
  \left\{
  (\varepsilon_1,\dots,\varepsilon_n)\in (\ZZ/2\ZZ)^n:\ \varepsilon_1+\cdots+\varepsilon_n=0
  \right\}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
设 \(z_1,\dots,z_n\) 为 \(P\) 的根。由结式根表示，
\[
Q_r(w)=a_n^2\prod_{j=1}^n \bigl(r z_j^2-w z_j+r^{-1}\bigr).
\]
令 \(w=2\)：
\[
r z_j^2-2z_j+r^{-1}=r^{-1}(r z_j-1)^2,
\]
从而
\[
Q_r(2)=a_n^2 r^{-n}\Bigl(\prod_{j=1}^n (r z_j-1)\Bigr)^2.
\]
又
\[
\prod_{j=1}^n (r z_j-1)
=(-1)^n r^n \frac{P(1/r)}{a_n},
\]
故 \(Q_r(2)=r^n P(1/r)^2\)。同理得 \(Q_r(-2)=r^n P(-1/r)^2\)。

再由
\[
Q_r(w)=(a_n a_0)\prod_{i=1}^n (w-w_i)
\]
可得
\[
Q_r(2)Q_r(-2)
=(a_n a_0)^2\prod_{i=1}^n (2-w_i)(-2-w_i)
=(a_n a_0)^2\prod_{i=1}^n (w_i^2-4).
\]
代入前述两式即得第二条。

在第三条条件下，\(\Delta_i\in M^\times\)。设
\[
V:=\operatorname{span}_{\FF_2}\{[\Delta_1],\dots,[\Delta_n]\}\subset M^\times/(M^\times)^2.
\]
Kummer 理论给出
\(
\operatorname{Gal}(N/M)\cong V^\vee
\)
与
\(
[N:M]=2^{\dim_{\FF_2}V}
\)。
第二条说明
\(
\sum_i [\Delta_i]=0
\)，故 \(\dim_{\FF_2}V\le n-1\)，于是 \([N:M]\mid 2^{n-1}\)；
并且对应符号翻转向量满足偶校验约束。证毕。
\end{proof}

\begin{proposition}[二次输运模板的全局偶校验]\label{prop:xi-quadratic-transport-global-even-parity}
\leanverified{paper\_xi\_quadratic\_transport\_global\_even\_parity}
设 \(K\) 为特征不为 \(2\) 的域，取 \(u,v\in K^\times\)。
令 \(P(z)\in K[z]\) 为次数 \(n\ge 1\) 的可分多项式，并定义二次输运结式
\[
Q_{u,v}(w):=\operatorname{Res}_z\!\bigl(P(z),\,u z^2-w z+v\bigr)\in K[w].
\]
记 \(w_1,\dots,w_n\) 为 \(Q_{u,v}\) 在某一分裂域 \(M\) 中的根（按重数），并置
\(
D(w):=w^2-4uv
\)。
则在 \(M^\times/(M^\times)^2\) 中恒有
\[
\Bigl[\prod_{i=1}^{n}D(w_i)\Bigr]=1.
\]
\end{proposition}
\begin{proof}
在二次扩张 \(K(\sqrt{uv})\) 中，\(D\) 的根为 \(\alpha_\pm=\pm 2\sqrt{uv}\)。
对任意符号 \(\pm\)，有
\[
u z^2-\alpha_\pm z+v
=u\Bigl(z\mp \sqrt{v/u}\Bigr)^2
\qquad\text{于 }K(\sqrt{uv})[z],
\]
从而
\[
Q_{u,v}(\alpha_\pm)
=\operatorname{Res}_z\!\Bigl(P(z),\,u\bigl(z\mp \sqrt{v/u}\bigr)^2\Bigr)
=u^{n}P\!\bigl(\pm \sqrt{v/u}\bigr)^2
\]
为 \(K(\sqrt{uv})\) 中的平方。
另一方面，由结式的根表示且 \(\mathrm{lc}(D)=1\)，有严格恒等式
\[
\operatorname{Res}_w(Q_{u,v}(w),D(w))
=\prod_{D(\alpha)=0}Q_{u,v}(\alpha),
\]
故该值在 \(K(\sqrt{uv})\) 中为平方。
而在分裂域 \(M\) 中同一结式亦可写为
\[
\operatorname{Res}_w(Q_{u,v}(w),D(w))
=(\mathrm{lc}(Q_{u,v}))^{2}\prod_{i=1}^{n}D(w_i),
\]
其中 \(\mathrm{lc}(Q_{u,v})\in K^\times\)，故 \((\mathrm{lc}(Q_{u,v}))^{2}\) 在平方类群中平凡。
于是 \(\prod_i D(w_i)\) 在 \(M^\times/(M^\times)^2\) 中为平凡类，结论成立。证毕。
\end{proof}

\begin{theorem}[代数尺度的有限例外相干性]\label{thm:xi-terminal-zm-jg-algebraic-scale-finite-exception-coherence}
\leanverified{paper\_xi\_terminal\_zm\_jg\_algebraic\_scale\_finite\_exception\_coherence}
设 \(K\) 为特征不为 \(2\) 的域，\(P\in K[z]\) 可分离且 \(a_n a_0\neq 0\)，\(r\in K^\times\)。
记 \(L/K\) 与 \(M/K\) 分别为 \(P\) 与 \(Q_r\) 的分裂域。若对任意根对 \((z_i,z_j)\) 均有
\[
r^2\neq (z_i z_j)^{-1},
\]
则
\[
M=L,\qquad \operatorname{Gal}(Q_r/K)\cong \operatorname{Gal}(P/K).
\]
并且对固定 \(P\)，使上述条件失效的 \(r\in K^\times\) 仅落在有限集合
\[
\mathcal E_P(K):=
\left\{
r\in K^\times:\ \exists\,i,j,\ r^2=(z_i z_j)^{-1}
\right\}
\]
中。
\end{theorem}
\begin{proof}
由 \(w_i=r z_i+r^{-1}z_i^{-1}\in L\) 得 \(M\subseteq L\)。

反向包含：固定 \(i\)，令 \(w_i:=r z_i+r^{-1}z_i^{-1}\in M\)，并设
\[
g_i(z):=r z^2-w_i z+r^{-1}\in M[z].
\]
其两根为 \(z_i\) 与 \(r^{-2}z_i^{-1}\)。条件保证 \(r^{-2}z_i^{-1}\) 不是 \(P\) 的根，故 \(\gcd(P,g_i)\) 在 \(M[z]\) 中为一次多项式，且唯一根为 \(z_i\)。因此 \(z_i\in M\)。对全部 \(i\) 成立，得 \(L\subseteq M\)，从而 \(L=M\)。

有限性由集合 \(\{(z_i z_j)^{-1}\}_{i,j}\) 有限立即推出：每个给定值在 \(K\) 中至多产生两个平方根，故 \(\mathcal E_P(K)\) 有限。证毕。
\end{proof}

\begin{theorem}[Lee--Yang 单尺度解析反演与吸引分支唯一性]\label{thm:xi-terminal-zm-leyang-single-scale-analytic-inversion}
\leanverified{paper\_xi\_terminal\_zm\_leyang\_single\_scale\_analytic\_inversion}
设 \(P\in\CC[z]\) 可分离且全部根满足 \(|z_i|=1\)，取实尺度 \(r>1\)。定义
\[
E_r:=\{\,w=r z+r^{-1}z^{-1}:|z|=1\,\},
\qquad
Q_r(w):=\operatorname{Res}_z\!\bigl(P(z),\,r z^2-w z+r^{-1}\bigr).
\]
则对 \(Q_r\) 的任一根 \(w\)，二次方程
\[
r z^2-w z+r^{-1}=0
\]
有两解
\[
z_\pm=\frac{w\pm\sqrt{w^2-4}}{2r},
\]
并满足
\[
|z_+|=1,\qquad |z_-|=r^{-2}
\]
（按适当重标记）。因此存在唯一分支选择
\[
s_r:\{Q_r\text{ 的根}\}\to \{P\text{ 的根}\},
\]
其中 \(s_r(w)\) 定义为唯一满足 \(|z|=1\) 的解。该选择恢复 \(P\) 的根多重集，进而由 Vieta 对称多项式唯一恢复 \(P\) 的系数。
\end{theorem}
\begin{proof}
若 \(w\) 是 \(Q_r\) 的根，则 \(w=r z_i+r^{-1}z_i^{-1}\) 对某根 \(z_i\) 成立，且对应二次方程的两根正是 \(z_i\) 与 \(r^{-2}z_i^{-1}\)。
由 \(|z_i|=1\) 得另一根模长为 \(r^{-2}\)。因 \(r>1\)，二者模长严格分离，故单位圆分支唯一。对全部根实施该选择得到 \(P\) 的根多重集，再由 Vieta 公式恢复系数。证毕。
\end{proof}

\begin{theorem}[Gödel 编码下的整数平方审计证书]\label{thm:xi-terminal-zm-godel-integer-transport-square-audit}
设 \(P\in\ZZ[z]\)，取互素整数 \(a,b\ge 1\)，令 \(r=a/b\)。定义清分母后的输运结式
\[
Q_{a,b}(w):=\operatorname{Res}_z\!\bigl(P(z),\,a^2 z^2-a b\,w z+b^2\bigr)\in\ZZ[w].
\]
则
\[
Q_{a,b}(2),\ Q_{a,b}(-2)\in\ZZ
\]
均为完全平方；并且
\[
\operatorname{Res}_w\!\bigl(Q_{a,b}(w),\,w^2-4\bigr)=Q_{a,b}(2)\,Q_{a,b}(-2)
\]
同样为完全平方。
\end{theorem}
\begin{proof}
代入 \(w=2\) 与 \(w=-2\) 分别得
\[
a^2 z^2-2ab z+b^2=(a z-b)^2,\qquad
a^2 z^2+2ab z+b^2=(a z+b)^2.
\]
故
\[
Q_{a,b}(2)=\operatorname{Res}_z\!\bigl(P(z),(a z-b)^2\bigr)
=\operatorname{Res}_z\!\bigl(P(z),a z-b\bigr)^2,
\]
\[
Q_{a,b}(-2)=\operatorname{Res}_z\!\bigl(P(z),(a z+b)^2\bigr)
=\operatorname{Res}_z\!\bigl(P(z),a z+b\bigr)^2.
\]
两者皆为整数平方。再由结式对因子分解的乘性
\[
\operatorname{Res}_w\!\bigl(Q_{a,b}(w),w^2-4\bigr)
=\operatorname{Res}_w\!\bigl(Q_{a,b}(w),w-2\bigr)\,
\operatorname{Res}_w\!\bigl(Q_{a,b}(w),w+2\bigr)
=Q_{a,b}(2)Q_{a,b}(-2),
\]
即得结论。证毕。
\end{proof}
\leanverified{paper\_xi\_terminal\_zm\_godel\_integer\_transport\_square\_audit}
