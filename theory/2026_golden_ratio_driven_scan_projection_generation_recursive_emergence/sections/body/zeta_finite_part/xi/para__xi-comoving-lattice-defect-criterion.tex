\paragraph{离散共动地址与晶格缺陷判据}\label{par:xi-comoving-lattice-defect-criterion}
统一判据定理 \ref{thm:xi-rh-iff-uniform-comoving-defect} 的量词对 \(x_0\in\RR\) 取上确界。
在共动图表中，离线零点的可见性由 \(\gamma-x_0\) 的相对位移决定（\eqref{eq:xi-comoving-disk-image}），从而允许把连续地址压缩为任意固定晶格上的离散采样而不改变可证性的逻辑强度。

\begin{theorem}[晶格共动缺陷判据]\label{thm:xi-rh-iff-lattice-comoving-defect}
固定任意步长 \(h>0\)。
下列命题等价：
\begin{enumerate}
  \item[(i)] RH 成立；
  \item[(ii)] 存在 \(\varrho_m\uparrow 1\)、\(\varepsilon_m\downarrow 0\)，使得对所有 \(m\) 有
  \begin{equation}\label{eq:xi-lattice-comoving-defect-criterion}
  \boxed{\;
  \sup_{n\in\ZZ}D_{\zeta,nh}(\varrho_m)\le \varepsilon_m\; }.
  \end{equation}
\end{enumerate}
\leanverified{paper\_xi\_rh\_iff\_lattice\_comoving\_defect}
\end{theorem}

\begin{proof}
\textbf{(i)\(\Rightarrow\)(ii).}
若 RH 成立，则对任意 \(x_0\) 圆盘化完成化函数 \(F_{\zeta,x_0}\) 在 \(\mathbb{D}\) 内无零点，故由 Jensen 求和式（命题 \ref{prop:app-jensen-formula-defect}）得 \(D_{\zeta,x_0}(\varrho)\equiv 0\)；
特别地对所有 \(n\in\ZZ\) 与 \(0<\varrho<1\) 有 \(D_{\zeta,nh}(\varrho)=0\)。
取任意 \(\varrho_m\uparrow 1\) 与 \(\varepsilon_m\downarrow 0\) 即得 \eqref{eq:xi-lattice-comoving-defect-criterion}。
\par
\textbf{(ii)\(\Rightarrow\)(i).}
反设 RH 不成立，则存在离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\) 且 \(\beta\neq\tfrac12\)。
由函数方程对称性可取 \(\beta<\tfrac12\)，并记 \(\delta:=\tfrac12-\beta>0\)。
取 \(n\in\ZZ\) 使得 \(|\gamma-nh|\le h/2\)，并令 \(x_0:=nh\)。
由 \eqref{eq:xi-comoving-disk-image}，点
\[
a:=w_{\rho,x_0}=\mathcal{C}\bigl((\gamma-x_0)+\mathrm{i}\delta\bigr)\in\mathbb{D}
\]
为 \(F_{\zeta,x_0}\) 的零点。
并且由 Cayley 模长公式（由 \eqref{eq:app-cayley-disk} 直接代数化简；亦见 \eqref{eq:xi-cayley-modulus-delta}）得
\[
|a|^2=\frac{(\gamma-x_0)^2+(1-\delta)^2}{(\gamma-x_0)^2+(1+\delta)^2}
\le
\frac{(h/2)^2+(1-\delta)^2}{(h/2)^2+(1+\delta)^2}
<1.
\]
记 \(r:=|a|<1\)。
由于 \(\varrho_m\uparrow 1\)，存在 \(m_0\) 使得对所有 \(m\ge m_0\) 有 \(\varrho_m>r\)。
此时由命题 \ref{prop:app-jensen-single-zero-lower-bound}（单零点 Jensen 下界）得到
\[
D_{\zeta,x_0}(\varrho_m)\ \ge\ \log\!\Bigl(\frac{\varrho_m}{r}\Bigr).
\]
结合假设 \eqref{eq:xi-lattice-comoving-defect-criterion} 得
\(
\log(\varrho_m/r)\le \varepsilon_m
\)，即
\(
r\ge \varrho_m e^{-\varepsilon_m}
\)。
令 \(m\to\infty\) 得 \(r\ge 1\)，与 \(r<1\) 矛盾。
故离临界线零点不存在，从而 RH 成立。证毕。
\end{proof}

\begin{corollary}[有限深度排斥的晶格证书]\label{cor:xi-lattice-local-depth-repulsion}
固定 \(h>0\) 与 \(\delta_0>0\)。
定义
\begin{equation}\label{eq:xi-lattice-depth-repulsion-Rh}
\boxed{\;
R_h(\delta_0)
:=\sqrt{\frac{(h/2)^2+(1-\delta_0)^2}{(h/2)^2+(1+\delta_0)^2}}\ \in(0,1).\;}
\end{equation}
取任意 \(\varrho\in(0,1)\) 满足 \(\varrho>R_h(\delta_0)\)。
若存在某个晶格点 \(x_0\in h\ZZ\) 使得
\begin{equation}\label{eq:xi-lattice-depth-repulsion-test}
\boxed{\;
D_{\zeta,x_0}(\varrho)\ <\ \log\!\Bigl(\frac{\varrho}{R_h(\delta_0)}\Bigr),\;}
\end{equation}
则不存在离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\) 同时满足 \(\tfrac12-\beta\ge \delta_0\) 且 \(|\gamma-x_0|\le h/2\)。
\leanverified{paper\_xi\_lattice\_local\_depth\_repulsion}
\end{corollary}

\begin{proof}
反设存在上述离线零点 \(\rho\)，记 \(\delta=\tfrac12-\beta\ge\delta_0\)。
由 \eqref{eq:xi-comoving-disk-image}，\(a=w_{\rho,x_0}\) 为 \(F_{\zeta,x_0}\) 的零点，且
\[
|a|^2=\frac{(\gamma-x_0)^2+(1-\delta)^2}{(\gamma-x_0)^2+(1+\delta)^2}
\le
\frac{(h/2)^2+(1-\delta_0)^2}{(h/2)^2+(1+\delta_0)^2}
=R_h(\delta_0)^2.
\]
故 \(|a|\le R_h(\delta_0)<\varrho\)。
由命题 \ref{prop:app-jensen-single-zero-lower-bound} 得
\[
D_{\zeta,x_0}(\varrho)\ \ge\ \log\!\Bigl(\frac{\varrho}{|a|}\Bigr)\ \ge\ \log\!\Bigl(\frac{\varrho}{R_h(\delta_0)}\Bigr),
\]
与 \eqref{eq:xi-lattice-depth-repulsion-test} 矛盾。证毕。
\end{proof}

\begin{proposition}[晶格共动的去高度化可见性]\label{prop:xi-lattice-deheightized-visibility}
\leanverified{paper\_xi\_lattice\_deheightized\_visibility}
固定步长 \(h>0\)。
设 \(\rho=\beta+\mathrm{i}\gamma\) 为离临界线零点，\(\delta=\tfrac12-\beta>0\)。
取晶格点 \(x_0\in h\ZZ\) 使 \(|\gamma-x_0|\le h/2\)。
则其移位盘内像 \(w_{\rho,x_0}\)（\eqref{eq:xi-comoving-disk-image}）满足
\begin{equation}\label{eq:xi-lattice-visibility-lowerbound}
\boxed{\;
1-|w_{\rho,x_0}|^2
=\frac{4\delta}{(\gamma-x_0)^2+(1+\delta)^2}
\ \ge\
\frac{4\delta}{(h/2)^2+(1+\delta)^2}\; }.
\end{equation}
因此对任意 \(\varrho\in(0,1)\)，若
\begin{equation}\label{eq:xi-lattice-visibility-threshold}
1-\varrho^2\ <\ \frac{4\delta}{(h/2)^2+(1+\delta)^2},
\end{equation}
则必有 \(|w_{\rho,x_0}|<\varrho\)，并由单零点 Jensen 下界推出
\[
D_{\zeta,x_0}(\varrho)\ \ge\ \log\!\Bigl(\frac{\varrho}{|w_{\rho,x_0}|}\Bigr)\ >0.
\]
上述阈值仅依赖于 \((\delta,h)\)，不含高度量级 \(|\gamma|\)。
\end{proposition}

\begin{proof}
由附录定理 \ref{thm:app-comoving-horizon-localization} 的闭式表示
\(
1-|w_{\rho,x_0}|^2=\frac{4\delta}{(\gamma-x_0)^2+(1+\delta)^2}
\)
得 \eqref{eq:xi-lattice-visibility-lowerbound}，再用 \((\gamma-x_0)^2\le (h/2)^2\) 得下界。
若 \eqref{eq:xi-lattice-visibility-threshold} 成立，则 \(1-|w_{\rho,x_0}|^2>1-\varrho^2\)。
由于 \(r\mapsto 1-r^2\) 在 \([0,1)\) 上严格递减，得到 \(|w_{\rho,x_0}|<\varrho\)。
最后应用命题 \ref{prop:app-jensen-single-zero-lower-bound} 即得所示下界。证毕。
\end{proof}
