\paragraph{Kummer--Cardano 覆盖同构、商曲线识别与全分歧谱闭式}
沿用定理 \ref{thm:xi-leyang-kummer-c3-etale-normal-form}、
\ref{thm:xi-leyang-cardano-chebyshev-linearization-z}、
\ref{thm:xi-leyang-eta-degree3-complete-ramification} 的记号。
记 \(K:=\QQ(\sqrt{-3})\)，并令 \(C_\eta\) 为 \(H_{\mathrm{LY}}\) 上 Kummer 方程
\[
\xi^3=\eta
\]
对应曲线的规范化。
再记 \(C_{S_3}\) 为既有 \(S_3\)-闭包三次曲线（即文中 \(C_6\)），并令 \(E_{\mathrm{res}}\) 为分解三次模型 \(R(z,y)=0\) 的光滑射影模型。

\begin{conclusion}[Kummer--Cardano 同一性与覆盖同构]\label{con:xi-leyang-kummer-cardano-covering-isomorphism}
存在 \(K\)-曲线同构
\[
C_{S_3}\otimes_{\QQ}K\cong C_\eta.
\]
并且在函数域层面成立显式生成元对换
\[
K(C_{S_3})=K(H_{\mathrm{LY}})(z)=K(H_{\mathrm{LY}})(\xi),
\]
其中
\[
z=\frac{2y+1}{3}\Bigl(2(\xi+\xi^{-1})-1\Bigr).
\]
\end{conclusion}
\begin{proof}
由定理 \ref{thm:xi-leyang-cardano-chebyshev-linearization-z}，记
\[
t:=\eta+\eta^{-1}=\frac{A(y)}{8(2y+1)^3},
\qquad
s:=\xi+\xi^{-1},
\]
则 \(s\) 满足 Chebyshev 方程
\[
s^3-3s=t,
\]
并且在变换
\[
z=\frac{2y+1}{3}(2s-1)
\]
下与 \(R(z,y)=0\) 等价。
故 \(z\in K(H_{\mathrm{LY}})(\xi)\)，且 \(K(H_{\mathrm{LY}})(z)\subseteq K(H_{\mathrm{LY}})(\xi)\)。

另一方面，由
\[
z=\frac{2y+1}{3}(2s-1)
\]
可反解
\[
s=\frac12\left(1+\frac{3z}{2y+1}\right)\in K(H_{\mathrm{LY}})(z).
\]
由定理 \ref{thm:xi-leyang-kummer-c3-etale-normal-form}，
\[
[K(H_{\mathrm{LY}})(\xi):K(H_{\mathrm{LY}})]=3,
\]
而 \(C_{S_3}\to H_{\mathrm{LY}}\) 亦为既定三次覆盖，故
\[
[K(H_{\mathrm{LY}})(z):K(H_{\mathrm{LY}})]=3.
\]
于是二者必相等：
\[
K(H_{\mathrm{LY}})(z)=K(H_{\mathrm{LY}})(\xi).
\]
函数域同一即推出规范化曲线同构
\(
C_{S_3}\otimes_{\QQ}K\cong C_\eta
\)。
证毕。
\end{proof}

\begin{conclusion}[显式 \(S_3\) 作用与商曲线识别]\label{con:xi-leyang-explicit-s3-action-and-quotients}
在结论 \ref{con:xi-leyang-kummer-cardano-covering-isomorphism} 的识别下，于 \(C_\eta\) 上定义
\[
\rho:(y,u,\xi)\longmapsto (y,u,\zeta_3\xi),
\qquad
\sigma:(y,u,\xi)\longmapsto (y,-u,\xi^{-1}),
\]
其中 \(\zeta_3\in K\) 为原三次单位根。则
\[
\rho^3=\sigma^2=\mathrm{id},
\qquad
\sigma\rho\sigma^{-1}=\rho^{-1},
\]
从而 \(\langle\rho,\sigma\rangle\cong S_3\)。
并且有商曲线同构
\[
C_\eta/\langle\rho\rangle\cong H_{\mathrm{LY}},
\qquad
C_\eta/\langle\sigma\rangle\cong E_{\mathrm{res}}.
\]
其中第二个同构由
\[
s:=\xi+\xi^{-1},
\qquad
z:=\frac{2y+1}{3}(2s-1)
\]
给出，且 \(K(C_\eta)^{\langle\sigma\rangle}=K(y,z)\)。
\end{conclusion}
\begin{proof}
\(\rho\) 保持 \(\xi^3=\eta\) 显然成立。
由定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion} 的
\(
\iota(\eta)=\eta^{-1}
\)
及 \(\iota:(y,u)\mapsto(y,-u)\)，可得
\[
(\xi^{-1})^3=\eta^{-1}=\eta(y,-u),
\]
故 \(\sigma\) 亦保持方程。
直接计算得 \(\rho^3=\sigma^2=\mathrm{id}\)。
且
\[
\sigma\rho\sigma^{-1}(\xi)
=\sigma\rho(\xi^{-1})
=\sigma(\zeta_3^{-1}\xi^{-1})
=\zeta_3^{-1}\xi
=\rho^{-1}(\xi),
\]
从而得到 \(S_3\) 关系。

对 \(\langle\rho\rangle\) 的商，固定域即 \(K(H_{\mathrm{LY}})\)，故
\(
C_\eta/\langle\rho\rangle\cong H_{\mathrm{LY}}
\)。
对 \(\sigma\) 而言，\(y\) 与 \(s=\xi+\xi^{-1}\) 为不变量。
由定理 \ref{thm:xi-leyang-cardano-chebyshev-linearization-z}，
\[
s^3-3s=t(y)
\quad\Longleftrightarrow\quad
R\!\left(\frac{2y+1}{3}(2s-1),y\right)=0,
\]
故 \(z=\frac{2y+1}{3}(2s-1)\) 满足 \(R(z,y)=0\)，并有 \(K(y,s)=K(y,z)\)。
又 \(\xi\) 满足 \(X^2-sX+1=0\)，故
\(
[K(C_\eta):K(y,s)]\le 2
\)。
而 \(\sigma\) 为对合，故
\(
[K(C_\eta):K(C_\eta)^{\langle\sigma\rangle}]=2
\)。
于是 \(K(C_\eta)^{\langle\sigma\rangle}=K(y,s)=K(y,z)\)，即
\(
C_\eta/\langle\sigma\rangle\cong E_{\mathrm{res}}
\)。
证毕。
\end{proof}

\begin{conclusion}[双覆盖 \(C_\eta\to E_{\mathrm{res}}\) 的分歧除子]\label{con:xi-leyang-double-cover-branch-as-two-lines}
令
\[
\pi_\sigma:C_\eta\to E_{\mathrm{res}}
\]
为结论 \ref{con:xi-leyang-explicit-s3-action-and-quotients} 的次数 \(2\) 商映射。
其分歧点恰为 \(\sigma\) 的不动点，且分歧像点满足显式刻画
\[
\operatorname{Br}(\pi_\sigma)
=\bigl(E_{\mathrm{res}}\cap L_+\bigr)\cup\bigl(E_{\mathrm{res}}\cap L_-\bigr),
\]
其中
\[
L_+:\ z=2y+1,
\qquad
L_-:\ 3z+5(2y+1)=0.
\]
并且 \(E_{\mathrm{res}}\cap L_+\) 由 \((y,z)=(0,1),(1,3)\) 与无穷远点构成；
\(E_{\mathrm{res}}\cap L_-\) 对应 \(P(y)=0\) 的三根所给出的三点。
\end{conclusion}
\begin{proof}
\(\sigma\)-不动条件为
\[
u=0,\qquad \xi=\xi^{-1},
\]
故 \(\xi=\pm 1\)。

当 \(\xi=1\) 时，\(s=\xi+\xi^{-1}=2\)，于是
\[
z=\frac{2y+1}{3}(2s-1)=2y+1.
\]
将其代入
\[
R(z,y)=z^3+(2y+1)z^2-(4y^2+4y+1)z-(8y^3+13y^2+5y+1)
\]
得到
\[
R(2y+1,y)=-y(y-1),
\]
故交点对应 \(y=0,1,\infty\)。

当 \(\xi=-1\) 时，\(s=-2\)，故
\[
z=-\frac53(2y+1),
\]
并计算得
\[
R\!\left(-\frac53(2y+1),y\right)=-\frac1{27}P(y),
\]
故交点恰对应 \(P(y)=0\) 的三根。

由结论 \ref{con:xi-leyang-kummer-cardano-covering-isomorphism}，
\(
g(C_\eta)=g(C_{S_3})=4
\)；
又 \(E_{\mathrm{res}}\) 为椭圆曲线，故 \(g(E_{\mathrm{res}})=1\)。
对次数 \(2\) 覆盖应用 Riemann--Hurwitz：
\[
2g(C_\eta)-2=2\bigl(2g(E_{\mathrm{res}})-2\bigr)+\deg\operatorname{Ram}(\pi_\sigma),
\]
得 \(\deg\operatorname{Ram}(\pi_\sigma)=6\)。
上述两条直线各给出三点，合计六个简单分歧点，刻画完备。证毕。
\end{proof}

\begin{conclusion}[三个互为共轭的双椭圆商与显式公式]\label{con:xi-leyang-three-conjugate-bielliptic-quotients}
在 \(C_\eta\) 上，对 \(i=0,1,2\) 定义
\[
s_i:=\zeta_3^i\xi+\zeta_3^{-i}\xi^{-1},
\qquad
z_i:=\frac{2y+1}{3}(2s_i-1).
\]
则对每个 \(i\) 都有
\[
R(z_i,y)=0,
\]
从而给出三条次数 \(2\) 映射
\[
\pi_i:C_\eta\longrightarrow E_{\mathrm{res}},
\qquad
(y,u,\xi)\longmapsto (y,z_i).
\]
并且 \(\rho\) 在 \(\{\pi_0,\pi_1,\pi_2\}\) 上作用为三循环置换。
\end{conclusion}
\begin{proof}
由恒等式
\[
(\alpha+\alpha^{-1})^3-3(\alpha+\alpha^{-1})=\alpha^3+\alpha^{-3},
\]
取 \(\alpha=\zeta_3^i\xi\)，得
\[
s_i^3-3s_i=\xi^3+\xi^{-3}=\eta+\eta^{-1}=t(y).
\]
再由定理 \ref{thm:xi-leyang-cardano-chebyshev-linearization-z} 的等价变换
\[
z=\frac{2y+1}{3}(2s-1)
\]
可知每个 \(z_i\) 都满足 \(R(z_i,y)=0\)，故得到映射 \(\pi_i\)。

令
\[
\sigma_i:=\rho^i\sigma\rho^{-i}.
\]
由定义可得 \(\sigma_i(s_i)=s_i\)，故 \(\sigma_i(z_i)=z_i\)，从而 \(z_i\) 落在固定域
\(
K(C_\eta)^{\langle\sigma_i\rangle}
\)。
而 \(\sigma_i\) 为对合，故 \(\pi_i\) 为相应商映射，次数为 \(2\)。

最后，
\[
\rho(s_i)=s_{i+1},\qquad \rho(z_i)=z_{i+1}\quad (\mathrm{mod}\ 3),
\]
故 \(\rho\) 在三条映射上诱导三循环。证毕。
\end{proof}

\begin{conclusion}[映射 \(t:H_{\mathrm{LY}}\to\PP^1\) 的五分歧值全谱]\label{con:xi-leyang-t-map-degree6-five-branch-values}
令
\[
t=\eta+\eta^{-1}=\frac{A(y)}{8(2y+1)^3}\in \QQ(H_{\mathrm{LY}}),
\]
并视为态射
\[
t:H_{\mathrm{LY}}\to\PP^1_t.
\]
则 \(\deg(t)=6\)，其分歧值集合恰为
\[
\{\infty,\,2,\,-2,\,t_+,\,t_-\},
\qquad
t_\pm=\frac{139}{72}\pm\frac{7\sqrt7}{144}.
\]
局部分歧指数如下：
\begin{enumerate}
  \item \(t=\infty\)：恰有两点（\(y=-\frac12\) 上方两点），各自 \(e=3\)；
  \item \(t=2\)：恰有三点（\(y=0,1,\infty\) 的三 Weierstrass 点），各自 \(e=2\)；
  \item \(t=-2\)：恰有三点（\(P(y)=0\) 的三 Weierstrass 点），各自 \(e=2\)；
  \item \(t=t_+\)：恰有两点（\(y=\frac{3+\sqrt7}{2}\) 上方两点），各自 \(e=2\)；
  \item \(t=t_-\)：恰有两点（\(y=\frac{3-\sqrt7}{2}\) 上方两点），各自 \(e=2\)。
\end{enumerate}
并且总分歧度满足
\[
\sum_{P\in H_{\mathrm{LY}}}(e_P-1)=14.
\]
\end{conclusion}
\begin{proof}
将 \(t\) 分解为
\[
H_{\mathrm{LY}}\xrightarrow{\,y\,}\PP^1_y\xrightarrow{\,t(y)\,}\PP^1_t.
\]
其中 \(y\)-映射为次数 \(2\) 的超椭圆覆盖，分歧点为 \(u=0\) 的六个 Weierstrass 点；
由定理 \ref{thm:xi-leyang-ty-three-branch-profile}，\(t(y)\) 为次数 \(3\) 映射，且在
\(
y=-\frac12
\)
处全分歧 \(e=3\)，在
\(
y=\frac{3\pm\sqrt7}{2}
\)
处简单分歧 \(e=2\)。

首先，\(y=-\frac12\) 上方两点满足 \(u\neq0\)，故在 \(y\)-层不分歧；复合后局部次数均为 \(3\)，对应分歧值 \(t=\infty\)。

其次，结论 \ref{con:xi-leyang-weierstrass-branch-binary-collapse-in-t} 给出
\[
t(0)=t(1)=t(\infty)=2,\qquad
t=-2\ \Longleftrightarrow\ P(y)=0.
\]
并且在上述 \(y\)-点处，\(t(y)\) 的导数不为零，故复合分歧仅来自超椭圆层，局部指数均为 \(2\)。
于是分别得到 \(t=2\) 与 \(t=-2\) 上的三点分歧。

再者，在 \(y=\frac{3\pm\sqrt7}{2}\) 上方各有两点，且该处超椭圆层不分歧、\(t(y)\)-层分歧指数为 \(2\)，故对应 \(t=t_\pm\) 各有两点简单分歧。

综上分歧值恰为五点，且总分歧度
\[
2\cdot(3-1)+3\cdot(2-1)+3\cdot(2-1)+2\cdot(2-1)+2\cdot(2-1)=14.
\]
与 Riemann--Hurwitz
\[
2g(H_{\mathrm{LY}})-2
=\deg(t)\cdot\bigl(2g(\PP^1)-2\bigr)+\sum_P(e_P-1)
\]
（即 \(2=-12+\sum_P(e_P-1)\)）完全一致。证毕。
\end{proof}

\begin{conclusion}[映射 \(\xi:C_\eta\to\PP^1\) 的三戈纳分歧全谱]\label{con:xi-leyang-xi-map-trigonal-complete-branch-spectrum}
函数 \(\xi\) 诱导态射
\[
\xi:C_\eta\to\PP^1_\xi,
\]
其次数为 \(3\)。
该映射全部分歧均为简单分歧，且分歧点集恰为定理
\ref{thm:xi-leyang-eta-degree3-complete-ramification}
中 \(\eta:H_{\mathrm{LY}}\to\PP^1_\eta\) 的四个简单分歧点在
\[
\pi_\eta:C_\eta\to H_{\mathrm{LY}}
\]
下的全逆像。
因此分歧点总数为 \(4\times 3=12\)，每点分歧指数均为 \(2\)。
等价地，
\[
\sum_{P\in C_\eta}(e_P-1)=12.
\]
\end{conclusion}
\begin{proof}
由定理 \ref{thm:xi-leyang-kummer-c3-etale-normal-form}，\(\pi_\eta\) 是次数 \(3\) 的无分歧覆盖；
由定理 \ref{thm:xi-leyang-eta-degree3-complete-ramification}，
\(\eta\) 在 \(Q_\pm\) 处全分歧 \(e=3\)，并在四个点 \(P_{\varepsilon,\sigma}\) 处简单分歧 \(e=2\)。

考虑交换分解
\[
C_\eta \xrightarrow{\ \xi\ } \PP^1_\xi \xrightarrow{(\cdot)^3} \PP^1_\eta,
\qquad
\eta\circ\pi_\eta=\xi^3.
\]
其次数满足
\[
\deg(\eta\circ\pi_\eta)=\deg(\eta)\deg(\pi_\eta)=3\cdot 3=9
=3\deg(\xi),
\]
故 \(\deg(\xi)=3\)。

在 \(Q_\pm\) 的逆像上，\(\eta\circ\pi_\eta\) 的局部指数为 \(3\)；
而立方映射在 \(\xi=0,\infty\) 处局部指数同为 \(3\)，故 \(\xi\) 在这些点不再分歧。

对四个简单分歧点 \(P_{\varepsilon,\sigma}\)，其 \(\eta\)-值不在 \(\{0,\infty\}\)，故立方映射在对应 \(\xi\)-值处不分歧。
又 \(\pi_\eta\) 无分歧，故每个 \(P_{\varepsilon,\sigma}\) 的三点逆像均保持指数 \(2\)。
于是得到 \(4\cdot 3=12\) 个简单分歧点。

最后由 Riemann--Hurwitz 校验：
\[
2g(C_\eta)-2
=\deg(\xi)\cdot(-2)+\sum_P(e_P-1),
\]
且 \(g(C_\eta)=4,\ \deg(\xi)=3\)，故
\[
6=-6+\sum_P(e_P-1),
\]
从而 \(\sum_P(e_P-1)=12\)，与上述计数一致。证毕。
\end{proof}
