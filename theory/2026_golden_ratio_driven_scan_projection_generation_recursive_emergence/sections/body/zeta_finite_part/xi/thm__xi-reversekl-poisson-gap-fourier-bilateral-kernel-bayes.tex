\begin{theorem}[Poisson--反向 KL 极值缺口与 Fourier 缺口的双边强凸夹逼]\label{thm:xi-reversekl-poisson-gap-fourier-bilateral}
\leanverified{paper\_xi\_reversekl\_poisson\_gap\_fourier\_bilateral}
沿用定理 \ref{thm:xi-reversekl-poisson-extremal-gap-fourier-stability} 的记号。
定义 Fourier 缺口
\[
S_r(\mu):=\sum_{n\ge 1} r^{2n}\bigl(1-|\widehat\mu(n)|^2\bigr)\ \in[0,\infty).
\]
令
\[
a_r:=\frac{1-r}{1+r},\qquad b_r:=\frac{1+r}{1-r}.
\]
则对一切 \(\TT\) 上概率测度 \(\mu\) 成立双边夹逼
\begin{equation}\label{eq:xi-reversekl-poisson-gap-fourier-bilateral}
\boxed{\;
a_r^{2}\,S_r(\mu)\ \le\ \Delta_r(\mu)\ \le\ b_r^{2}\,S_r(\mu).\;}
\end{equation}
\end{theorem}
\begin{proof}
令 \(\Phi\sim\mu\)，并对每个 \(\theta\in\TT\) 定义随机变量
\(
X_\theta:=P_r(\theta-\Phi)
\)。
由 Poisson 核的显式公式得到一致夹逼 \(X_\theta\in[a_r,b_r]\)（a.s.）。
置 \(\varphi(x):=-\log x\)，则 \(\varphi\in C^2((0,\infty))\) 且 \(\varphi''(x)=x^{-2}\)。
因此在区间 \([a_r,b_r]\) 上有
\[
b_r^{-2}\ \le\ \varphi''(x)\ \le\ a_r^{-2}.
\]
对取值于 \([a_r,b_r]\) 的任意随机变量 \(X\)，由二阶 Taylor 余项的积分表示（或等价的强凸/强光滑 Jensen 余项估计）得到
\[
\frac{b_r^{-2}}{2}\Var(X)
\ \le\
\EE[\varphi(X)]-\varphi(\EE[X])
\ \le\
\frac{a_r^{-2}}{2}\Var(X).
\]
应用于 \(X=X_\theta\) 并注意 \(\EE[X_\theta]=p_r(\theta)\)，得
\[
\frac{1}{2b_r^{2}}\Var(X_\theta)
\ \le\
\int_{\TT}-\log P_r(\theta-\phi)\,\dd\mu(\phi)+\log p_r(\theta)
\ \le\
\frac{1}{2a_r^{2}}\Var(X_\theta).
\]
对 \(\theta\) 按 \(m\) 积分，并用平移不变性与
\(
\int_{\TT}-\log P_r\,\dd m=-\log(1-r^2)
\)
（见定理 \ref{thm:xi-reversekl-poisson-extremal-gap-fourier-stability} 证明中的 Jensen 平均计算）得到
\[
\frac{1}{2b_r^2}\int_{\TT}\Var(X_\theta)\,\dd m(\theta)
\ \le\ \Delta_r(\mu)\ \le\
\frac{1}{2a_r^2}\int_{\TT}\Var(X_\theta)\,\dd m(\theta).
\]
最后由 Parseval 恒等式与 \(\widehat{P_r}(n)=r^{|n|}\)、\(\widehat p_r(n)=r^{|n|}\widehat\mu(n)\) 得
\[
\int_{\TT}\Var(X_\theta)\,\dd m(\theta)
=\|P_r\|_{L^2(m)}^2-\|p_r\|_{L^2(m)}^2
=2\sum_{n\ge 1}r^{2n}\bigl(1-|\widehat\mu(n)|^2\bigr)
=2S_r(\mu).
\]
代回并用 \(b_r^{-2}=a_r^2\)、\(a_r^{-2}=b_r^2\) 即得 \eqref{eq:xi-reversekl-poisson-gap-fourier-bilateral}。证毕。
\end{proof}

\begin{proposition}[Fourier 缺口的显式正核能量表示]\label{prop:xi-reversekl-fourier-gap-positive-kernel-energy}
\leanverified{paper\_xi\_reversekl\_fourier\_gap\_positive\_kernel\_energy}
在定理 \ref{thm:xi-reversekl-poisson-gap-fourier-bilateral} 的记号下，定义
\[
K_r(\vartheta):=\sum_{n\ge 1} r^{2n}\bigl(1-\cos(n\vartheta)\bigr),\qquad \vartheta\in\TT.
\]
则对任意概率测度 \(\mu\) 有精确能量表示
\begin{equation}\label{eq:xi-reversekl-fourier-gap-energy}
\boxed{\;
S_r(\mu)=\iint_{\TT^2}K_r(\theta-\phi)\,\dd\mu(\theta)\,\dd\mu(\phi).\;}
\end{equation}
并且 \(K_r\) 具有闭式
\begin{equation}\label{eq:xi-reversekl-fourier-gap-kernel-closed}
\boxed{\;
K_r(\vartheta)=\frac{r^2(1+r^2)(1-\cos\vartheta)}{(1-r^2)\,(1-2r^2\cos\vartheta+r^4)}\ \ge\ 0.\;}
\end{equation}
特别地，\(K_r(\vartheta)=0\) 当且仅当 \(\vartheta\equiv 0\ (\mathrm{mod}\ 2\pi)\)。
\end{proposition}
\begin{proof}
由 Fubini 与
\[
|\widehat\mu(n)|^2
=\left(\int_{\TT}e^{-in\theta}\,\dd\mu(\theta)\right)\left(\int_{\TT}e^{in\phi}\,\dd\mu(\phi)\right)
=\iint_{\TT^2}e^{-in(\theta-\phi)}\,\dd\mu(\theta)\,\dd\mu(\phi),
\]
取实部得
\(
|\widehat\mu(n)|^2=\iint\cos(n(\theta-\phi))\,\dd\mu(\theta)\,\dd\mu(\phi)
\)。
代入 \(S_r(\mu)\) 并交换求和与积分得到 \eqref{eq:xi-reversekl-fourier-gap-energy}。
\par
对闭式，先由几何级数计算
\[
\sum_{n\ge 1}r^{2n}e^{in\vartheta}
=\frac{r^2e^{i\vartheta}}{1-r^2e^{i\vartheta}},
\]
取实部得到
\[
\sum_{n\ge 1}r^{2n}\cos(n\vartheta)
=\frac{r^2\cos\vartheta-r^4}{1-2r^2\cos\vartheta+r^4}.
\]
于是
\[
K_r(\vartheta)
=\sum_{n\ge 1}r^{2n}-\sum_{n\ge 1}r^{2n}\cos(n\vartheta)
=\frac{r^2}{1-r^2}-\frac{r^2\cos\vartheta-r^4}{1-2r^2\cos\vartheta+r^4},
\]
化简即得 \eqref{eq:xi-reversekl-fourier-gap-kernel-closed}。
非负性与零点刻画由分子分母的显然性得到。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\Delta_r$}{Delta-r} 控制到最近点质量的 \texorpdfstring{$W_2$}{W2} 上界]\label{prop:xi-reversekl-gap-w2-to-delta}
\leanverified{paper\_xi\_reversekl\_gap\_w2\_to\_delta}
令 \(\mathrm{dist}_{\TT}\) 为 \(\TT\) 上测地距离（取值于 \([0,\pi]\)），并令 \(W_2\) 为以 \(\mathrm{dist}_{\TT}\) 定义的二阶 Wasserstein 距离。
取 \(\phi_\star=\arg\widehat\mu(1)\)（任取一辐角）。
则对任意 \(r\in(0,1)\) 有
\begin{equation}\label{eq:xi-reversekl-gap-w2-to-delta}
\boxed{\;
W_2\bigl(\mu,\delta_{\phi_\star}\bigr)^2
\ \le\
\frac{\pi^2}{2a_r^{2}r^2}\,\Delta_r(\mu).\;}
\end{equation}
\end{proposition}
\begin{proof}
令 \(\Phi\sim\mu\)，并置 \(U:=\mathrm{dist}_{\TT}(\Phi,\phi_\star)\in[0,\pi]\)。
则 \(W_2(\mu,\delta_{\phi_\star})^2=\EE[U^2]\)。
由恒等式 \(1-\cos U=2\sin^2(U/2)\) 及 \(\sin x\ge \frac{2}{\pi}x\)（\(x\in[0,\pi/2]\)）得
\[
1-\cos U\ \ge\ \frac{2}{\pi^2}U^2,
\qquad\text{即}\qquad
U^2\ \le\ \frac{\pi^2}{2}\,(1-\cos U).
\]
取期望并注意
\[
\EE[\cos(\Phi-\phi_\star)]
=\Re\!\left(e^{-i\phi_\star}\widehat\mu(1)\right)
=|\widehat\mu(1)|,
\]
得到
\[
\EE[U^2]\le \frac{\pi^2}{2}\bigl(1-|\widehat\mu(1)|\bigr)
\le \frac{\pi^2}{2}\bigl(1-|\widehat\mu(1)|^2\bigr).
\]
另一方面由 \(S_r(\mu)\ge r^2(1-|\widehat\mu(1)|^2)\) 得
\(
1-|\widehat\mu(1)|^2\le S_r(\mu)/r^2
\)。
再由定理 \ref{thm:xi-reversekl-poisson-gap-fourier-bilateral} 的下界 \(\Delta_r(\mu)\ge a_r^2S_r(\mu)\)，得到
\[
W_2(\mu,\delta_{\phi_\star})^2
\le \frac{\pi^2}{2}\cdot \frac{S_r(\mu)}{r^2}
\le \frac{\pi^2}{2a_r^2r^2}\,\Delta_r(\mu).
\]
证毕。
\end{proof}

\begin{proposition}[互素频率的相位放大：两条近字符信息推出一阶近字符]\label{prop:xi-reversekl-coprime-frequency-phase-amplification}
\leanverified{paper\_xi\_reversekl\_coprime\_frequency\_phase\_amplification}
令 \(p,q\in\NN\) 互素，\(\mu\) 为 \(\TT\) 上概率测度，并设
\[
\varepsilon_p:=1-|\widehat\mu(p)|,\qquad \varepsilon_q:=1-|\widehat\mu(q)|.
\]
则存在整数 \(u,v\) 满足 \(up+vq=1\) 且 \(|u|\le q,\ |v|\le p\)，并且成立
\begin{equation}\label{eq:xi-reversekl-coprime-frequency-phase-amplification}
\boxed{\;
1-|\widehat\mu(1)|
\ \le\
\bigl(|u|\sqrt{\varepsilon_p}+|v|\sqrt{\varepsilon_q}\bigr)^2
\ \le\
(p+q)^2\max(\varepsilon_p,\varepsilon_q).\;}
\end{equation}
\end{proposition}
\begin{proof}
令 \(\Phi\sim\mu\)，并对任意单位模随机变量 \(Z\) 使用恒等式
\[
\inf_{|\omega|=1}\EE|Z-\omega|^2
=2\bigl(1-|\EE Z|\bigr).
\]
取 \(Z=e^{-ip\Phi}\) 与 \(Z=e^{-iq\Phi}\)，分别得到相位 \(\omega_p,\omega_q\in\TT\) 使
\[
\EE\bigl|e^{-ip\Phi}-\omega_p\bigr|^2\le 2\varepsilon_p,\qquad
\EE\bigl|e^{-iq\Phi}-\omega_q\bigr|^2\le 2\varepsilon_q.
\]
取 Bézout 系数 \(u,v\)（满足 \(up+vq=1\) 且 \(|u|\le q,|v|\le p\)）。
对 \(|z|=|w|=1\) 与整数 \(k\) 有
\(
|z^k-w^k|\le |k|\cdot |z-w|
\)。
故
\[
\bigl|e^{-i\Phi}-\omega_p^{u}\omega_q^{v}\bigr|
=\bigl|(e^{-ip\Phi})^{u}(e^{-iq\Phi})^{v}-\omega_p^{u}\omega_q^{v}\bigr|
\le |(e^{-ip\Phi})^{u}-\omega_p^{u}|+|(e^{-iq\Phi})^{v}-\omega_q^{v}|.
\]
取 \(L^2\) 范数并用 Minkowski 不等式得
\[
\Bigl(\EE\bigl|e^{-i\Phi}-\omega_p^{u}\omega_q^{v}\bigr|^2\Bigr)^{1/2}
\le |u|\sqrt{2\varepsilon_p}+|v|\sqrt{2\varepsilon_q}.
\]
两边平方并用上述最小二乘相位恒等式（对 \(Z=e^{-i\Phi}\)）得到
\[
2\bigl(1-|\widehat\mu(1)|\bigr)
\le 2\bigl(|u|\sqrt{\varepsilon_p}+|v|\sqrt{\varepsilon_q}\bigr)^2,
\]
即 \eqref{eq:xi-reversekl-coprime-frequency-phase-amplification} 的第一条不等式。
第二条由 \(|u|\le q,|v|\le p\) 及
\(
\sqrt{\varepsilon_p}+\sqrt{\varepsilon_q}\le 2\sqrt{\max(\varepsilon_p,\varepsilon_q)}
\)
得到。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\Delta_r$}{Delta-r} 的贝叶斯分解：缺口作为先验--后验平均 KL]\label{thm:xi-reversekl-gap-bayes-decomposition}
\leanverified{paper\_xi\_reversekl\_gap\_bayes\_decomposition}
沿用定理 \ref{thm:xi-reversekl-poisson-extremal-gap-fourier-stability} 的记号。
定义 \(\TT^2\) 上联合概率测度 \(\mathbf B_r\)：
\[
\dd\mathbf B_r(\theta,\phi)=P_r(\theta-\phi)\,\dd m(\theta)\,\dd\mu(\phi),
\]
并记其 \(\theta\)-边缘为 \(p_r(\theta)\dd m(\theta)\)。
对每个 \(\theta\) 定义后验（条件分布）\(\mu_\theta^{(r)}\)：
\[
\dd\mu_\theta^{(r)}(\phi)=\frac{P_r(\theta-\phi)}{p_r(\theta)}\,\dd\mu(\phi).
\]
则成立精确恒等式
\begin{equation}\label{eq:xi-reversekl-gap-bayes-decomposition}
\boxed{\;
\Delta_r(\mu)
=\int_{\TT}D_{\mathrm{KL}}\!\bigl(\mu\mid \mu_\theta^{(r)}\bigr)\,\dd m(\theta).\;}
\end{equation}
特别地，\(\Delta_r(\mu)\ge 0\)，且 \(\Delta_r(\mu)=0\) 当且仅当 \(\mu\) 为点质量。
\end{theorem}
\begin{proof}
注意到
\[
\frac{\dd(m\times\mu)}{\dd\mathbf B_r}(\theta,\phi)=\frac{1}{P_r(\theta-\phi)},
\]
从而
\[
D_{\mathrm{KL}}(m\times\mu\mid \mathbf B_r)
=\iint_{\TT^2}-\log P_r(\theta-\phi)\,\dd m(\theta)\,\dd\mu(\phi)
=\int_{\TT}-\log P_r(\theta)\,\dd m(\theta)
=-\log(1-r^2),
\]
其中最后一步同定理 \ref{thm:xi-reversekl-poisson-extremal-gap-fourier-stability} 证明中的 Jensen 平均计算。
\par
另一方面，对 \(\mathbf B_r\) 的边缘--条件分解应用 KL 的链式法则，得
\[
D_{\mathrm{KL}}(m\times\mu\mid \mathbf B_r)
=D_{\mathrm{KL}}(m\mid p_r m)+\int_{\TT}D_{\mathrm{KL}}\!\bigl(\mu\mid \mu_\theta^{(r)}\bigr)\,\dd m(\theta).
\]
两式相减即得 \eqref{eq:xi-reversekl-gap-bayes-decomposition}。
非负性由相对熵非负性给出。
若 \(\Delta_r(\mu)=0\)，则 \(D_{\mathrm{KL}}(\mu\mid \mu_\theta^{(r)})=0\) 对 \(m\)-几乎处处 \(\theta\) 成立，因而 \(\mu_\theta^{(r)}=\mu\)（\(\mu\)-a.e.）。
这迫使 \(\phi\mapsto P_r(\theta-\phi)\) 在 \(\mu\) 的支撑上为常值，从而 \(\mu\) 只能为点质量。
反向蕴含由直接代入得到。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\Delta_r$}{Delta-r} 的平均 Pinsker：后验偏移的平方可积]\label{cor:xi-reversekl-gap-mean-pinsker}
\leanverified{paper\_xi\_reversekl\_gap\_mean\_pinsker}
沿用定理 \ref{thm:xi-reversekl-gap-bayes-decomposition} 的记号，记全变差距离为 \(\|\cdot\|_{\mathrm{TV}}\)。
则
\begin{equation}\label{eq:xi-reversekl-gap-mean-pinsker}
\boxed{\;
\Delta_r(\mu)\ \ge\ 2\int_{\TT}\bigl\|\mu-\mu_\theta^{(r)}\bigr\|_{\mathrm{TV}}^{2}\,\dd m(\theta).\;}
\end{equation}
\end{corollary}
\begin{proof}
对每个 \(\theta\) 由 Pinsker 不等式
\(
\|\mu-\mu_\theta^{(r)}\|_{\mathrm{TV}}^2\le \tfrac12 D_{\mathrm{KL}}(\mu\mid \mu_\theta^{(r)})
\)
得
\[
D_{\mathrm{KL}}(\mu\mid \mu_\theta^{(r)})\ge 2\|\mu-\mu_\theta^{(r)}\|_{\mathrm{TV}}^2.
\]
对 \(\theta\) 按 \(m\) 积分并用 \eqref{eq:xi-reversekl-gap-bayes-decomposition} 即得 \eqref{eq:xi-reversekl-gap-mean-pinsker}。证毕。
\end{proof}
