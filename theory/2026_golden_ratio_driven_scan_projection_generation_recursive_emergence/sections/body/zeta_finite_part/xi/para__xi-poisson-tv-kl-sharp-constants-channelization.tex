\paragraph{常数闭式化与相位二通道：Poisson--KL 子链的可核验接口}\label{par:xi-poisson-tv-kl-sharp-constants-channelization}
命题 \ref{prop:xi-poisson-tv-kl-second-order} 给出二阶正规形分解
\(
h_t=g_t+\sigma^2 r_t+\varepsilon_t
\)
并把全变差与相对熵的主项常数化归为若干有理积分。
下述命题将这些常数以闭式记录为可独立引用的证书接口；随后给出二阶偏差在相位坐标中的两余弦通道分解（这解释了常数可闭式化的结构来源），并提出一条与高阶矩塔相容的通道化猜想。

\begin{proposition}[TV/KL 的锐常数与六阶系数：从正规形积分到闭式不变量]\label{prop:xi-cauchy-poisson-tv-kl-sharp-interface}
令 \(\tilde\nu\) 为 \(\RR\) 上概率测度，重心为
\(
\bar\gamma:=\int_{\RR}\gamma\,\dd\tilde\nu(\gamma)
\)。
对 \(t>0\) 定义 Poisson--Cauchy 粗粒化密度与同重心基准密度
\[
h_t(x):=\int_{\RR}\frac{1}{\pi}\frac{t}{(x-\gamma)^2+t^2}\,\dd\tilde\nu(\gamma),\qquad
g_t(x):=\frac{1}{\pi}\frac{t}{(x-\bar\gamma)^2+t^2}.
\]
记方差 \(\sigma^2:=\int_{\RR}(\gamma-\bar\gamma)^2\,\dd\tilde\nu(\gamma)\)，并假设三阶中心绝对矩有限。
则当 \(t\to\infty\) 有
\begin{align}
&\boxed{\;
\|h_t-g_t\|_{L^1(\RR)}
=\frac{3\sqrt3}{4\pi}\,\frac{\sigma^2}{t^2}+O(t^{-3})\;}\label{eq:xi-cauchy-poisson-tv-sharp-mainterm}\\[1mm]
&\boxed{\;
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{1}{8}\,\frac{\sigma^4}{t^4}+o(t^{-4})\;}
\label{eq:xi-cauchy-poisson-kl-mainterm}
\end{align}
并且若进一步假设四阶中心矩 \(\mu_4:=\int_{\RR}(\gamma-\bar\gamma)^4\,\dd\tilde\nu(\gamma)<\infty\)，则存在三阶中心矩 \(\mu_3:=\int_{\RR}(\gamma-\bar\gamma)^3\,\dd\tilde\nu(\gamma)\) 使六阶项可由闭式系数组合给出（见定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics}）：
\begin{equation}\label{eq:xi-cauchy-poisson-kl-sixth-interface}
\boxed{\;
D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{\sigma^4}{8t^4}
\;+\;\frac{1}{t^6}\left(
\frac{3\mu_3^2}{32}
-\frac{\sigma^2\mu_4}{8}
+\frac{\sigma^6}{64}
\right)
\;+\;O(t^{-8})\; }.
\end{equation}
此外，若 \(\sigma^2>0\)，则 Pinsker 比值的二次极限常数存在且为
\begin{equation}\label{eq:xi-cauchy-poisson-pinsker-interface}
\boxed{\;
\lim_{t\to\infty}\frac{D_{\mathrm{KL}}(h_t\mid g_t)}{\|h_t-g_t\|_{L^1(\RR)}^2}
=\frac{2\pi^2}{27}\; }.
\end{equation}
\end{proposition}

\begin{proof}
\eqref{eq:xi-cauchy-poisson-tv-sharp-mainterm} 由命题 \ref{prop:xi-poisson-tv-kl-second-order} 的正规形分解与 \(\|\varepsilon_t\|_{L^1}=O(t^{-3})\) 得
\(
\|h_t-g_t\|_{L^1}=\sigma^2\|r_t\|_{L^1}+O(t^{-3})
\)，再与定理 \ref{thm:xi-cauchy-poisson-tv-sharp-constant} 中 \(\|r_t\|_{L^1}=\frac{3\sqrt3}{4\pi}\,t^{-2}\) 合并即得。
\eqref{eq:xi-cauchy-poisson-kl-mainterm} 即命题 \ref{prop:xi-poisson-tv-kl-second-order} 的 \(t^{-4}\) 主项。
六阶接口 \eqref{eq:xi-cauchy-poisson-kl-sixth-interface} 为定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的闭式展开重述。
Pinsker 接口 \eqref{eq:xi-cauchy-poisson-pinsker-interface} 为推论 \ref{cor:xi-cauchy-poisson-pinsker-sharp-ratio} 的等价表述。证毕。
\end{proof}

\begin{corollary}[四次重整化极限的方差可识别性]\label{cor:xi-cauchy-poisson-kl-variance-readout}
在命题 \ref{prop:xi-cauchy-poisson-tv-kl-sharp-interface} 的条件下，极限
\[
\lim_{t\to\infty} t^4 D_{\mathrm{KL}}(h_t\mid g_t)=\frac{\sigma^4}{8}
\]
存在。
因此方差可由单标量观测曲线 \(t\mapsto D_{\mathrm{KL}}(h_t\mid g_t)\) 在无穷远处的四次重整化极限唯一确定：
\[
\sigma^2=\Bigl(8\lim_{t\to\infty} t^4 D_{\mathrm{KL}}(h_t\mid g_t)\Bigr)^{1/2}.
\]
\end{corollary}

\begin{corollary}[黄金比尺度链上的四次重整化展开]\label{cor:xi-cauchy-poisson-kl-golden-scale}
\leanverified{paper\_xi\_cauchy\_poisson\_kl\_golden\_scale}
令 \(t_m:=\varphi^m\)（\(m\in\mathbb N\)），其中 \(\varphi=\frac{1+\sqrt5}{2}\)。
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，当 \(m\to\infty\) 有
\begin{equation}\label{eq:xi-cauchy-poisson-kl-golden-scale}
\boxed{\;
\varphi^{4m}D_{\mathrm{KL}}(h_{t_m}\mid g_{t_m})
=\frac{\sigma^4}{8}
+\varphi^{-2m}\left(
\frac{3\mu_3^2}{32}-\frac{\sigma^2\mu_4}{8}+\frac{\sigma^6}{64}
\right)
+O(\varphi^{-4m})\; }.
\end{equation}
\end{corollary}
\begin{proof}
对 \eqref{eq:xi-cauchy-poisson-kl-sharp-asymptotics} 代入 \(t=\varphi^m\)，并乘以 \(\varphi^{4m}\) 即得。
\end{proof}

\begin{corollary}[耗散尾积分的 \texorpdfstring{$t^{-4}$}{t^-4} 常数]\label{cor:xi-cauchy-poisson-fisher-tail-integral}
在定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的条件下，
\[
\int_t^\infty \mathcal I_1(h_s\mid g_s)\,\dd s
=D_{\mathrm{KL}}(h_t\mid g_t)
=\frac{\sigma^4}{8t^4}+O(t^{-6}),
\qquad t\to\infty.
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:xi-poisson-kl-dissipation-fractional-fisher}，
\(
\frac{\dd}{\dd t}D_{\mathrm{KL}}(h_t\mid g_t)=-\mathcal I_1(h_t\mid g_t)
\)。
又由 \eqref{eq:xi-cauchy-poisson-kl-sharp-asymptotics} 得 \(D_{\mathrm{KL}}(h_t\mid g_t)\to 0\)（\(t\to\infty\)），
故对 \([t,\infty)\) 积分得到
\(
D_{\mathrm{KL}}(h_t\mid g_t)=\int_t^\infty \mathcal I_1(h_s\mid g_s)\,\dd s
\)。
再由 \eqref{eq:xi-cauchy-poisson-kl-sharp-asymptotics} 取出 \(t^{-4}\) 与 \(t^{-6}\) 阶即可。
\end{proof}

\begin{proposition}[相位--Fourier 正规形：二阶偏差恰落在两条低频余弦通道]\label{prop:xi-poisson-second-order-two-cosine-channels}
沿用命题 \ref{prop:xi-poisson-tv-kl-second-order} 的正规形分解，并记
\[
F_2(y):=\frac{3y^2-1}{(1+y^2)^2}.
\]
令相位变量
\[
\theta:=\arctan\!\Bigl(\frac{x-\bar\gamma}{t}\Bigr)\in\Bigl(-\frac{\pi}{2},\frac{\pi}{2}\Bigr).
\]
则二阶偏差满足严格恒等式
\begin{equation}\label{eq:xi-poisson-second-order-phase-two-channels}
\boxed{\;
\frac{r_t(x)}{g_t(x)}
=\frac{1}{t^2}F_2\!\Bigl(\frac{x-\bar\gamma}{t}\Bigr)
=-\frac{1}{2t^2}\Bigl(\cos(2\theta)+\cos(4\theta)\Bigr)\; }.
\end{equation}
特别地，\(t^{-2}\) 级别的密度比偏差完全由两条 Fourier 余弦通道 \(\cos(2\theta)\)、\(\cos(4\theta)\) 承载。
\end{proposition}

\begin{proof}
令 \(y=\tan\theta=(x-\bar\gamma)/t\)，则 \(1+y^2=\sec^2\theta\)。
直接化简得到
\[
F_2(\tan\theta)=\frac{3\tan^2\theta-1}{\sec^4\theta}
=3\sin^2\theta\cos^2\theta-\cos^4\theta
=-\frac12\bigl(\cos(2\theta)+\cos(4\theta)\bigr),
\]
从而 \eqref{eq:xi-poisson-second-order-phase-two-channels} 成立。证毕。
\end{proof}

\begin{remark}[二通道正交与二次常数的闭式读出]\label{rem:xi-poisson-second-order-orthogonal-constants}
在 \(y=\tan\theta\) 变换下，权测度满足
\(
\frac{1}{\pi}\frac{1}{1+y^2}\,\dd y=\frac{1}{\pi}\,\dd\theta
\) 于 \(\theta\in(-\tfrac{\pi}{2},\tfrac{\pi}{2})\)。
因此二次积分（从而 \(t^{-4}\) 级别的 \(\chi^2\)/KL 主项常数）可写为两通道的 \(L^2\) 能量：
\[
\frac{1}{\pi}\int_{\RR}\frac{F_2(y)^2}{1+y^2}\,\dd y
=\frac{1}{\pi}\int_{-\pi/2}^{\pi/2}F_2(\tan\theta)^2\,\dd\theta
=\frac{1}{4\pi}\int_{-\pi/2}^{\pi/2}\bigl(\cos(2\theta)+\cos(4\theta)\bigr)^2\,\dd\theta.
\]
由 \(\cos(2\theta)\perp \cos(4\theta)\)（在 \((-\pi/2,\pi/2)\) 上的均匀测度下）得到该积分等于 \(1/4\)，与命题 \ref{prop:xi-poisson-tv-kl-second-order} 中的闭式常数一致。
\end{remark}

\begin{conjecture}[全阶偶次展开与通道化普适性：KL 作为 cumulant 的解析二次型塔]\label{conj:xi-cauchy-poisson-even-tower-channelization}
在 \(\tilde\nu\) 具有全阶中心矩且 \(\mathbb{E}(\gamma-\bar\gamma)=0\) 的情形，存在一族仅依赖阶数的显式通道函数 \(\{\Psi_k\}_{k\ge 1}\subset L^2\!\bigl((-\tfrac{\pi}{2},\tfrac{\pi}{2}),\dd\theta\bigr)\)，每个 \(\Psi_k\) 为有限 Fourier 模的线性组合，使得
\[
\log\frac{h_t}{g_t}\bigl(\bar\gamma+t\tan\theta\bigr)
\sim
\sum_{k\ge 1}\frac{1}{t^k}\,A_k(\tilde\nu)\,\Psi_k(\theta),
\qquad t\to\infty,
\]
其中 \(A_k(\tilde\nu)\) 可由中心 cumulant 的显式多项式给出。
并且相对熵具有偶次塔结构
\[
D_{\mathrm{KL}}(h_t\mid g_t)\sim \sum_{m\ge 2}\frac{1}{t^{2m}}\,Q_m(\kappa_2,\dots,\kappa_{m+1}),
\qquad t\to\infty,
\]
其中 \(Q_m\) 为显式多项式二次型，且其首两层与定理 \ref{thm:xi-cauchy-poisson-kl-sharp-asymptotics} 的 \(t^{-4}\)、\(t^{-6}\) 系数一致。
\end{conjecture}
