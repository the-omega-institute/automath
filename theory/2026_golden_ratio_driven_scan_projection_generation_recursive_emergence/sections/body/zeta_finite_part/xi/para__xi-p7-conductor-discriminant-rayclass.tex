\begin{theorem}[三条基本不可约 Artin 表示的全局导子]\label{thm:xi-p7-s5-global-artin-conductors-rho456}
\leanverified{paper\_xi\_p7\_s5\_global\_artin\_conductors\_rho456}
在定理 \ref{thm:xi-p7-s5-minimal-artin-dedekind-zeta-factorization} 的记号下，三条不可约表示 \(\rho_4,\rho_5,\rho_6\) 的 Artin 导子满足
\[
\mathfrak f(\rho_4)=2^{2}q,\qquad
\mathfrak f(\rho_5)=2^{4}q^{2},\qquad
\mathfrak f(\rho_6)=2^{4}q^{3}.
\]
\end{theorem}
\begin{proof}
对温和素数 \(\ell\)，Artin 导子指数满足
\(
a_\ell(\rho)=\dim\rho-\dim\rho^{I_\ell}
\)。
由定理 \ref{thm:xi-p7-s5-dyadic-inertia-and-discriminant-exponents} 得 \(I_2\cong C_3\)，由定理 \ref{thm:xi-p7-s5-odd-ramification-discriminant-and-conductor} 得 \(I_q\cong C_2\)。
因此
\[
\dim \rho^{C_m}=\frac1m\sum_{g\in C_m}\chi_\rho(g).
\]
\par
对 \(C_3\) 的生成元可取为一个 \(3\)-循环。
由 \(5\) 点置换表示 \(\Ind_{S_4}^{S_5}\mathbf 1\cong \mathbf 1\oplus\rho_4\) 与不动点计数（\(3\)-循环固定 \(2\) 点），得
\(\chi_{\rho_4}(3\text{-cycle})=2-1=1\)，故
\(
\dim\rho_4^{I_2}=(4+2\cdot 1)/3=2
\)，从而 \(a_2(\rho_4)=2\)。
同理由 \(10\) 点无序二元组置换表示 \(\Ind_{S_2\times S_3}^{S_5}\mathbf 1\cong \mathbf 1\oplus\rho_4\oplus\rho_5\) 与不动点数 \(1\)，得
\(\chi_{\rho_5}(3\text{-cycle})=1-1-1=-1\)，从而
\(
\dim\rho_5^{I_2}=(5+2\cdot(-1))/3=1
\) 与 \(a_2(\rho_5)=4\)。
再由 \(20\) 点有序对置换表示 \(\Ind_{S_3}^{S_5}\mathbf 1\cong \mathbf 1\oplus 2\rho_4\oplus\rho_5\oplus\rho_6\) 与不动点数 \(2\)，得
\(\chi_{\rho_6}(3\text{-cycle})=2-1-2\cdot 1-(-1)=0\)，从而
\(
\dim\rho_6^{I_2}=(6+0)/3=2
\) 与 \(a_2(\rho_6)=4\)。
\par
对 \(q\) 的导子指数已由定理 \ref{thm:xi-p7-s5-odd-ramification-discriminant-and-conductor} 给出：
\(
a_q(\rho_4)=1,\ a_q(\rho_5)=2,\ a_q(\rho_6)=3
\)。
将各素处指数合并即得三条全局导子。证毕。
\end{proof}

\begin{theorem}[奇分歧换位惰性下的共同三维不变子空间与 Schur 函子导子闭式]\label{thm:xi-p7-s5-q-inertia-vacuum3-schur-functor-conductors}
\leanverified{paper\_xi\_p7\_s5\_q\_inertia\_vacuum3\_schur\_functor\_conductors}
在定理 \ref{thm:xi-p7-s5-odd-ramification-discriminant-and-conductor} 的设定下，令 \(I_q\simeq C_2\) 为 \(q\) 处惰性群并取其非平凡特征 \(\varepsilon\)。
则对 \(i\in\{4,5,6\}\) 有：
\begin{enumerate}
  \item 存在分解
  \[
  \rho_i\big|_{I_q}\ \cong\ \mathbf 1^{\oplus 3}\oplus \varepsilon^{\oplus a_q(\rho_i)}.
  \]
  等价地，
  \(
  \dim(\rho_i^{I_q})=3
  \)
  对 \(i=4,5,6\) 同时成立。
  \item 设 \(V=\rho_i\) 且记 \(a:=a_q(V)\in\{1,2,3\}\)。
  则对任意整数 \(r\ge 1\)，
  \begin{equation}\label{eq:xi-p7-q-schur-symr-conductor-closed}
  a_q\!\bigl(\operatorname{Sym}^{r}V\bigr)
  =\sum_{\substack{0\le k\le r\\ k\ \mathrm{odd}}}
  \binom{r-k+2}{2}\binom{a+k-1}{k},
  \end{equation}
  且对 \(1\le r\le \dim V\)，
  \begin{equation}\label{eq:xi-p7-q-schur-wedger-conductor-closed}
  a_q\!\bigl(\bigwedge^{r}V\bigr)
  =\sum_{\substack{0\le k\le r\\ k\ \mathrm{odd}}}
  \binom{3}{r-k}\binom{a}{k}.
  \end{equation}
  特别地在 \(r=2\) 时有线性律
  \[
  a_q\!\bigl(\operatorname{Sym}^{2}\rho_i\bigr)
  =a_q\!\bigl(\bigwedge^{2}\rho_i\bigr)
  =3\,a_q(\rho_i),
  \]
  从而
  \[
  a_q\!\bigl(\operatorname{Sym}^{2}\rho_4\bigr)=a_q\!\bigl(\bigwedge^{2}\rho_4\bigr)=3,\qquad
  a_q\!\bigl(\operatorname{Sym}^{2}\rho_5\bigr)=a_q\!\bigl(\bigwedge^{2}\rho_5\bigr)=6,\qquad
  a_q\!\bigl(\operatorname{Sym}^{2}\rho_6\bigr)=a_q\!\bigl(\bigwedge^{2}\rho_6\bigr)=9.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
温和情形的导子公式给出 \(a_q(V)=\dim V-\dim(V^{I_q})\)。
由定理 \ref{thm:xi-p7-s5-odd-ramification-discriminant-and-conductor} 的数值
\(
a_q(\rho_4)=1,\ a_q(\rho_5)=2,\ a_q(\rho_6)=3
\)
以及 \(\dim\rho_i\in\{4,5,6\}\)，立得
\(
\dim(\rho_i^{I_q})=\dim\rho_i-a_q(\rho_i)=3
\)。
由于 \(I_q\simeq C_2\) 的不可约表示均为 \(\mathbf 1\) 与 \(\varepsilon\)，从而得到
\(
\rho_i|_{I_q}\cong \mathbf 1^{\oplus 3}\oplus \varepsilon^{\oplus a_q(\rho_i)}
\)。
\par
令 \(V=\rho_i=U\oplus W\)，其中 \(I_q\) 在 \(U\) 上作用为 \(+1\)（\(\dim U=3\)），在 \(W\) 上作用为 \(-1\)（\(\dim W=a\)）。
则
\[
\operatorname{Sym}^{r}(V)=\bigoplus_{k=0}^{r}\operatorname{Sym}^{r-k}(U)\otimes \operatorname{Sym}^{k}(W),
\]
且 \(I_q\) 在第 \(k\) 项上的作用为 \((-1)^k\)。
因此 \(I_q\)-不变子空间来自偶 \(k\) 项，而导子指数等于奇 \(k\) 项的维数和：
\[
a_q(\operatorname{Sym}^r V)=\sum_{\substack{0\le k\le r\\ k\ \mathrm{odd}}}
\dim\operatorname{Sym}^{r-k}(U)\cdot \dim\operatorname{Sym}^{k}(W).
\]
由 \(\dim U=3\) 与 \(\dim W=a\) 的对称幂维数公式
\(
\dim\operatorname{Sym}^{d}(U)=\binom{d+2}{2}
\),
\(
\dim\operatorname{Sym}^{k}(W)=\binom{a+k-1}{k}
\)，
即得 \eqref{eq:xi-p7-q-schur-symr-conductor-closed}。
\par
外幂同理：写
\[
\bigwedge^{r}(V)=\bigoplus_{k=0}^{r}\bigwedge^{r-k}(U)\otimes \bigwedge^{k}(W),
\]
奇 \(k\) 项贡献导子，且
\(
\dim\bigwedge^{r-k}(U)=\binom{3}{r-k}
\),
\(
\dim\bigwedge^{k}(W)=\binom{a}{k}
\)，
得到 \eqref{eq:xi-p7-q-schur-wedger-conductor-closed}。
在 \(r=2\) 时仅有 \(k=1\) 贡献，从而
\(
a_q(\operatorname{Sym}^{2}V)=a_q(\bigwedge^{2}V)=\binom{3}{2}\binom{a}{1}=3a
\)，代入 \(a=a_q(\rho_i)\) 得所列数值。证毕。
\end{proof}

\begin{theorem}[素数 \(2\) 处三循环惰性分解与二次函子导子显式化]\label{thm:xi-p7-s5-dyadic-c3-inertia-quadratic-functors-conductors}
\leanverified{paper\_xi\_p7\_s5\_dyadic\_c3\_inertia\_quadratic\_functors\_conductors}
在定理 \ref{thm:xi-p7-s5-dyadic-inertia-and-discriminant-exponents} 的设定下，\(I_2\simeq C_3\)。
取生成元 \(g\in I_2\) 并令 \(\zeta:=e^{2\pi i/3}\)，\(\chi(g)=\zeta\)。
则对 \(i\in\{4,5,6\}\) 存在唯一整数对 \((a_i,b_i)\) 使得
\[
\rho_i\big|_{I_2}\ \cong\ \mathbf 1^{\oplus a_i}\oplus \chi^{\oplus b_i}\oplus (\chi^2)^{\oplus b_i},
\]
并且由 \(\chi_{\rho_i}(g)=\operatorname{Tr}(\rho_i(g))\) 给出闭式
\[
a_i=\frac{\dim\rho_i+2\chi_{\rho_i}(g)}{3},\qquad
b_i=\frac{\dim\rho_i-\chi_{\rho_i}(g)}{3}.
\]
在定理 \ref{thm:xi-p7-s5-global-artin-conductors-rho456} 的特征值数据
\(\chi_{\rho_4}(g)=1,\ \chi_{\rho_5}(g)=-1,\ \chi_{\rho_6}(g)=0\)
下，谱型具体为
\[
\rho_4|_{I_2}\cong \mathbf 1^{\oplus 2}\oplus \chi\oplus \chi^2,\qquad
\rho_5|_{I_2}\cong \mathbf 1\oplus \chi^{\oplus 2}\oplus (\chi^2)^{\oplus 2},\qquad
\rho_6|_{I_2}\cong \mathbf 1^{\oplus 2}\oplus \chi^{\oplus 2}\oplus (\chi^2)^{\oplus 2}.
\]
并且二次函子与张量积在素数 \(2\) 处的温和导子指数满足
\[
\begin{aligned}
&a_2\!\bigl(\operatorname{Sym}^{2}\rho_4\bigr)=6,\qquad &&a_2\!\bigl(\bigwedge^{2}\rho_4\bigr)=4,\\
&a_2\!\bigl(\operatorname{Sym}^{2}\rho_5\bigr)=10,\qquad &&a_2\!\bigl(\bigwedge^{2}\rho_5\bigr)=6,\\
&a_2\!\bigl(\operatorname{Sym}^{2}\rho_6\bigr)=14,\qquad &&a_2\!\bigl(\bigwedge^{2}\rho_6\bigr)=10,
\end{aligned}
\qquad
\begin{aligned}
&a_2(\rho_4\otimes\rho_5)=14,\qquad a_2(\rho_4\otimes\rho_6)=16,\qquad a_2(\rho_5\otimes\rho_6)=20.
\end{aligned}
\]
\end{theorem}

\begin{proof}
\(C_3\) 的不可约复表示均为一维角色 \(\mathbf 1,\chi,\chi^2\)。
由于 \(\rho_i\) 的特征标为实值（定理 \ref{thm:xi-p7-s5-global-artin-conductors-rho456} 的特征标计算），\(\chi\) 与 \(\chi^2=\overline{\chi}\) 的重数必相等，
故可写
\(
\rho_i|_{I_2}\cong \mathbf 1^{\oplus a_i}\oplus \chi^{\oplus b_i}\oplus(\chi^2)^{\oplus b_i}
\)。
由
\(
\dim\rho_i=a_i+2b_i
\)
与
\(
\chi_{\rho_i}(g)=a_i+b_i(\zeta+\zeta^2)=a_i-b_i
\)
解得所示闭式，并据此得到三条谱型。
\par
温和导子指数满足 \(a_2(V)=\dim V-\dim(V^{I_2})\)。
对 \(V\) 的 \(g\)-特征值重数记为 \((a,b,b)\)（对应 \((1,\zeta,\zeta^2)\)）。
则在 \(\operatorname{Sym}^2V\) 中，\(g\)-不变向量来自 \((1\cdot 1)\) 与 \((\zeta\cdot \zeta^2)\) 两类：
\[
\dim(\operatorname{Sym}^2V)^{I_2}=\binom{a+1}{2}+b^2,
\qquad
a_2(\operatorname{Sym}^2V)=\binom{\dim V+1}{2}-\binom{a+1}{2}-b^2.
\]
同理在 \(\bigwedge^2V\) 中，
\[
\dim(\bigwedge^2V)^{I_2}=\binom{a}{2}+b^2,
\qquad
a_2(\bigwedge^2V)=\binom{\dim V}{2}-\binom{a}{2}-b^2.
\]
将 \((a,b)=(2,1),(1,2),(2,2)\) 代入即得 \(\rho_4,\rho_5,\rho_6\) 的二次函子导子指数。
\par
对张量积，若 \(V,W\) 的特征值重数分别为 \((a,b,b)\)、\((a',b',b')\)，
则
\(
(V\otimes W)^{I_2}
\)
来自 \(1\otimes 1\) 与 \(\zeta\otimes \zeta^2,\ \zeta^2\otimes \zeta\)：
\[
\dim(V\otimes W)^{I_2}=aa'+2bb',
\qquad
a_2(V\otimes W)=\dim(V)\dim(W)-aa'-2bb'.
\]
代入三组 \((a,b)\) 即得所列张量积指数。证毕。
\end{proof}

\begin{corollary}[两素温和分歧下的全局导子分解：对称/外平方与张量积]\label{cor:xi-p7-s5-global-conductors-schur-quadratic-tensor}
\leanverified{paper\_xi\_p7\_s5\_global\_conductors\_schur\_quadratic\_tensor}
在上述记号下，分裂域 \(L_{p_7}/\QQ\) 的有限分歧素数集合包含于 \(\{2,q\}\)，且两处皆为温和分歧。
因此对任意由 \(\rho_4,\rho_5,\rho_6\) 经有限次直和、张量积、对称幂与外幂所生成的 Artin 表示 \(V\)，其全局导子满足
\[
\mathfrak f(V)=2^{a_2(V)}q^{a_q(V)}.
\]
特别地，
\[
\mathfrak f(\operatorname{Sym}^2\rho_4)=2^{6}q^{3},\qquad
\mathfrak f(\bigwedge^2\rho_4)=2^{4}q^{3},\qquad
\mathfrak f(\operatorname{Sym}^2\rho_5)=2^{10}q^{6},\qquad
\mathfrak f(\bigwedge^2\rho_5)=2^{6}q^{6},
\]
\[
\mathfrak f(\operatorname{Sym}^2\rho_6)=2^{14}q^{9},\qquad
\mathfrak f(\bigwedge^2\rho_6)=2^{10}q^{9},
\]
以及
\[
\mathfrak f(\rho_4\otimes\rho_5)=2^{14}q^{9},\qquad
\mathfrak f(\rho_4\otimes\rho_6)=2^{16}q^{12},\qquad
\mathfrak f(\rho_5\otimes\rho_6)=2^{20}q^{15}.
\]
\end{corollary}

\begin{proof}
温和分歧下全局导子按素数处指数分解：
\(
\mathfrak f(V)=\prod_p p^{a_p(V)}
\)。
由定理 \ref{thm:xi-p7-s5-dyadic-inertia-and-discriminant-exponents} 与定理 \ref{thm:xi-p7-s5-odd-ramification-discriminant-and-conductor}，有限分歧仅可能发生在 \(\{2,q\}\)，且 Swan 部分为零，
故 \(\mathfrak f(V)=2^{a_2(V)}q^{a_q(V)}\)。
数值由定理 \ref{thm:xi-p7-s5-q-inertia-vacuum3-schur-functor-conductors}（给出 \(q\) 处指数）与定理 \ref{thm:xi-p7-s5-dyadic-c3-inertia-quadratic-functors-conductors}（给出 \(2\) 处指数）代入得到。证毕。
\end{proof}

\begin{theorem}[判别式—导子完全相容与二次塔的相对判别式范数]\label{thm:xi-p7-s5-discriminant-conductor-compatibility-and-relative-discriminant}
三类中间域的绝对判别式满足导子乘积分解
\[
\Disc(K_{5})=-\mathfrak f(\rho_4),\qquad
\Disc(K_{10})=-\mathfrak f(\rho_4)\mathfrak f(\rho_5),\qquad
\Disc(K_{20})=-\mathfrak f(\rho_4)^2\mathfrak f(\rho_5)\mathfrak f(\rho_6),
\]
从而
\[
\Disc(K_{10})=-2^{6}q^{3},\qquad
\Disc(K_{20})=-2^{12}q^{7}.
\]
并且二次扩张 \(K_{20}/K_{10}\) 的相对判别式 \(\mathfrak D_{K_{20}/K_{10}}\) 满足
\[
\mathrm N_{K_{10}/\QQ}\!\bigl(\mathfrak D_{K_{20}/K_{10}}\bigr)=q.
\]
因而 \(\mathfrak D_{K_{20}/K_{10}}\) 必为 \(K_{10}\) 中某个素理想，且 \(K_{20}/K_{10}\) 在一切有限素处除该单一素理想外皆不分歧；同时该扩张在无限素处恰一处实嵌入复化（定理 \ref{thm:xi-p7-s5-signatures-and-unique-archimedean-ramification}）。
\end{theorem}
\begin{proof}
对任意 \(K=L^H\)，Dedekind \(\zeta_K\) 对应的 \(S_5\)-置换表示为 \(\Ind_H^{S_5}\mathbf 1\)，其 Artin 导子等于 \(\Disc(K)\)。
将 \(\Ind_H^{S_5}\mathbf 1\) 分解到 \(\mathbf 1,\rho_4,\rho_5,\rho_6\) 并利用导子对直和的乘法性，即得三条判别式分解式；
符号为负由定理 \ref{thm:xi-p7-s5-signatures-and-unique-archimedean-ramification} 的签名确定。
代入定理 \ref{thm:xi-p7-s5-global-artin-conductors-rho456} 的 \(\mathfrak f(\rho_i)\) 即得到 \(\Disc(K_{10})\) 与 \(\Disc(K_{20})\) 的显式指数。
\par
相对判别式范数恒等式
\[
|\Disc(K_{20})|
\;=\;
|\Disc(K_{10})|^{2}\cdot \mathrm N_{K_{10}/\QQ}\!\bigl(\mathfrak D_{K_{20}/K_{10}}\bigr)
\]
给出 \(\mathrm N(\mathfrak D_{K_{20}/K_{10}})=q\)。
由于 \(q\) 为有理素数，\(\mathfrak D_{K_{20}/K_{10}}\) 必为某个素理想；否则其范数将可分解为非平凡乘积。
有限处分歧支集与无限处分歧结论随之成立。证毕。
\end{proof}

\begin{conclusion}[二次扩张的窄射线类特征与分歧模数的极简性]\label{con:xi-p7-s5-narrow-ray-class-character}
令 \(\mathfrak q\subset\mathcal O_{K_{10}}\) 为定理 \ref{thm:xi-p7-s5-discriminant-conductor-compatibility-and-relative-discriminant} 中满足 \(\mathfrak D_{K_{20}/K_{10}}=\mathfrak q\) 的素理想，并令 \(\infty_{0}\) 为定理 \ref{thm:xi-p7-s5-signatures-and-unique-archimedean-ramification} 所刻画的唯一复化实位。
则二次扩张 \(K_{20}/K_{10}\) 定义了一个非平凡二次窄射线类特征
\[
\chi:\ \mathrm{Cl}^{+}_{K_{10}}(\mathfrak q\cdot \infty_{0})\longrightarrow \{\pm 1\},
\]
其有限部分的导子恰为 \(\mathfrak q\)，并且该特征在除 \(\mathfrak q\) 外的一切有限素处处处无分歧、在 \(\infty_{0}\) 处处处分歧。
\end{conclusion}
\begin{proof}
类域论将任意有限阿贝尔扩张与相应射线类群的有限指数子群对应；对二次扩张得到二次特征。
定理 \ref{thm:xi-p7-s5-discriminant-conductor-compatibility-and-relative-discriminant} 给出有限分歧恰集中于 \(\mathfrak q\)，
定理 \ref{thm:xi-p7-s5-signatures-and-unique-archimedean-ramification} 给出无限分歧恰集中于 \(\infty_0\)。
因此该扩张对应的窄射线类特征的模数可取为 \(\mathfrak q\cdot \infty_{0}\)，并且其有限导子必为 \(\mathfrak q\)。证毕。
\end{proof}
