\begin{proposition}[对数缺陷势的 \texorpdfstring{$L^\infty$}{Linfty} 线性控制]\label{prop:xi-log-defect-potential-linfty-bound-by-total-delta}
沿用对数缺陷势 \(\mathcal{K}_{\gamma,\delta}\) 的定义 \eqref{eq:xi-log-defect-potential}，并假设 \(0<\delta\le \tfrac12\)。
令
\[
h_{\gamma,\delta}(x):=1-\bigl|\mathcal{C}((x-\gamma)+\mathrm{i}\delta)\bigr|^2
=\frac{4\delta}{(x-\gamma)^2+(1+\delta)^2}\qquad(x\in\RR).
\]
则对所有 \(x\in\RR\) 成立双侧比较不等式
\begin{equation}\label{eq:xi-logdefect-compare-h}
\boxed{\;
\frac12\,h_{\gamma,\delta}(x)\ \le\ \mathcal{K}_{\gamma,\delta}(x)\ \le\ \frac92\,h_{\gamma,\delta}(x)\; }.
\end{equation}
因此对任意有限点状缺陷族 \(\{(\gamma_j,\delta_j,m_j)\}_{j=1}^{M}\)（\(m_j\in\NN\)、\(\delta_j\in(0,\tfrac12]\)）与其总对数势
\[
\Phi(x):=\sum_{j=1}^{M}m_j\,\mathcal{K}_{\gamma_j,\delta_j}(x),
\]
有一致上界
\begin{equation}\label{eq:xi-logdefect-linfty-linear-delta}
\boxed{\;
\|\Phi\|_{L^\infty(\RR)}\ \le\ 18\sum_{j=1}^{M}m_j\,\delta_j\; }.
\end{equation}
\end{proposition}
\begin{proof}
由 \eqref{eq:xi-cayley-modulus-delta}（平移不影响模长），对任意 \(x\in\RR\) 有
\[
\bigl|\mathcal{C}((x-\gamma)+\mathrm{i}\delta)\bigr|^2
=\frac{(x-\gamma)^2+(1-\delta)^2}{(x-\gamma)^2+(1+\delta)^2}
=1-\frac{4\delta}{(x-\gamma)^2+(1+\delta)^2}
=1-h_{\gamma,\delta}(x).
\]
从而
\[
\mathcal{K}_{\gamma,\delta}(x)
=-\log\bigl|\mathcal{C}((x-\gamma)+\mathrm{i}\delta)\bigr|
=-\frac12\log\!\bigl(1-h_{\gamma,\delta}(x)\bigr).
\]
由于 \(0<\delta\le \tfrac12\) 蕴含 \(0\le h_{\gamma,\delta}(x)\le h_{\gamma,\delta}(\gamma)=\frac{4\delta}{(1+\delta)^2}\le \frac89\)，
对 \(u\in[0,\frac89]\) 应用基本不等式
\[
u\ \le\ -\log(1-u)\ \le\ \frac{u}{1-u},
\]
得到
\[
\frac12\,h_{\gamma,\delta}(x)\ \le\ \mathcal{K}_{\gamma,\delta}(x)\ \le\ \frac{1}{2}\cdot\frac{h_{\gamma,\delta}(x)}{1-h_{\gamma,\delta}(x)}
\le \frac{1}{2}\cdot 9\,h_{\gamma,\delta}(x)=\frac92\,h_{\gamma,\delta}(x),
\]
即 \eqref{eq:xi-logdefect-compare-h}。
进一步由 \(h_{\gamma,\delta}(x)\le \frac{4\delta}{(x-\gamma)^2+1}\le 4\delta\) 得
\(
\mathcal{K}_{\gamma,\delta}(x)\le \frac92\cdot 4\delta=18\delta
\)。
对 \(\Phi\) 逐项求和并取上确界即得 \eqref{eq:xi-logdefect-linfty-linear-delta}。证毕。
\end{proof}
