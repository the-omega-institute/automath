\subsubsection{复权原子 Euler 乘积的迹类实现：从 Euler 因子到循环块 Fredholm 行列式的构造性反向截面}\label{subsec:cyclic-traceclass-lift-complex}
上一小节 \ref{subsec:cyclic-traceclass-lift} 以 \(\beta_j\in(0,1)\) 的正性口径给出了“原子谱数据 \((\beta_j,n_j,m_j)\mapsto L_{\mathcal D}\)”的循环块迹类提升。
该构造并不依赖 \(\beta_j\) 的实正性：在复权情形，只需在迹类范数意义下控制可数直和即可得到同样严格的反向截面，从而把原子 Euler 乘积的参数族提升为可直接编译的迹类算子证书。

\begin{corollary}[Euler 乘积到迹类 Fredholm \texorpdfstring{$\zeta$}{zeta} 的反向截面：任意原子 Euler 乘积的构造性实现（复权版）]\label{cor:cyclic-euler-product-complex-section}
\leanverified{paper\_cyclic\_euler\_product\_complex\_section}
设给定可数数据族 \(\mathcal D=\{(\beta_j,n_j,m_j)\}_{j\in J}\subset \CC\times \NN_{\ge 1}\times \NN\)，并假设
\begin{equation}\label{eq:tc-executability-complex}
\sum_{j\in J} m_j\,n_j\,|\beta_j|^{1/n_j}<\infty,
\qquad
|\beta_j|<1\ \ (j\in J).
\end{equation}
对每个 \(j\in J\) 选取一个 \(n_j\)-次根 \(\alpha_j\in\CC\) 使 \(\alpha_j^{n_j}=\beta_j\)（从而 \(|\alpha_j|=|\beta_j|^{1/n_j}\)）。
令
\[
H_{\mathcal D}:=\bigoplus_{j\in J}\bigl(\CC^{n_j}\otimes \CC^{m_j}\bigr),
\qquad
L_{\mathcal D}:=\bigoplus_{j\in J}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr)\in \mathcal{S}_1(H_{\mathcal D}),
\]
其中 \(C(n,\alpha)=\alpha\Pi_n\) 为定义 \ref{def:cyclic-block} 的循环块。
定义
\[
\zeta_{\mathcal D}(r):=\det(I-rL_{\mathcal D})^{-1}.
\]
则在 \(|r|<1\) 上有严格恒等式
\[
\boxed{\;
\zeta_{\mathcal D}(r)=\prod_{j\in J}\bigl(1-\beta_j r^{n_j}\bigr)^{-m_j}\; }.
\]
\end{corollary}

\begin{proof}
由 \eqref{eq:tc-executability-complex} 与 \(|\alpha_j|=|\beta_j|^{1/n_j}\) 得
\(
\sum_{j\in J}m_jn_j|\alpha_j|<\infty
\)，故 \(L_{\mathcal D}\in\mathcal{S}_1(H_{\mathcal D})\)。
并且由 \(|\alpha_j|<1\) 与 \(\alpha_j\to 0\)（由可和性）可知 \(\|L_{\mathcal D}\|=\sup_j|\alpha_j|<1\)，从而 \((I-rL_{\mathcal D})\) 在 \(|r|<1\) 内可逆。
\par
对每个循环块，由命题 \ref{prop:cycle-permutation-determinant} 有
\[
\det\bigl(I-rC(n_j,\alpha_j)\bigr)=1-(\alpha_j r)^{n_j}=1-\beta_j r^{n_j}.
\]
再由直和/张量的 Fredholm 行列式乘法性得到
\[
\det(I-rL_{\mathcal D})
=\prod_{j\in J}\det\bigl(I-r(C(n_j,\alpha_j)\otimes I_{m_j})\bigr)
=\prod_{j\in J}\bigl(1-\beta_j r^{n_j}\bigr)^{m_j}.
\]
两边取倒数即得结论。证毕。
\end{proof}

\begin{theorem}[任意复权原子 Euler 乘积的正规迹类实现与分类]\label{thm:cyclic-fredholm-realisation}
设
\[
\mathcal D=\{(\beta_j,n_j,m_j)\}_{j\in J}
\subset (\CC\setminus\{0\})\times \NN_{\ge 1}\times \NN
\]
满足
\[
\sum_{j\in J}m_j\,n_j\,|\beta_j|^{1/n_j}<\infty.
\]
对每个 \(j\in J\) 选取 \(\alpha_j\in\CC\) 使 \(\alpha_j^{n_j}=\beta_j\)，并定义
\[
H_{\mathcal D}:=\bigoplus_{j\in J}\bigl(\CC^{n_j}\otimes \CC^{m_j}\bigr),
\qquad
L_{\mathcal D}:=\bigoplus_{j\in J}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
\]
则 \(L_{\mathcal D}\in\mathcal{S}_1(H_{\mathcal D})\)，并且
\[
\det(I-rL_{\mathcal D})^{-1}
=\prod_{j\in J}\bigl(1-\beta_j r^{n_j}\bigr)^{-m_j}
\]
作为 \(r=0\) 附近的全纯 germ 成立。若记
\[
r_{\mathcal D}:=\sup_{j\in J}|\beta_j|^{1/n_j},
\]
则 \(r_{\mathcal D}<\infty\)，且上式在 \(|r|<r_{\mathcal D}^{-1}\) 上成立。

进一步地，对任意正规迹类算子 \(L\in\mathcal{S}_1(H)\)，下列两条等价：
\begin{enumerate}
  \item 在 \(r=0\) 的某个邻域内有
  \[
  \det(I-rL)^{-1}
  =\prod_{j\in J}\bigl(1-\beta_j r^{n_j}\bigr)^{-m_j}.
  \]
  \item 存在 Hilbert 空间 \(H_0\) 与酉算子 \(U:H\to H_0\oplus H_{\mathcal D}\)，使
  \[
  ULU^{-1}=0_{H_0}\oplus L_{\mathcal D}.
  \]
\end{enumerate}
特别地，任一正规实现的非零谱多重集都恰为
\[
\biguplus_{j\in J}\bigl\{\alpha_j\omega:\omega^{n_j}=1\bigr\}^{m_j},
\]
从而其非零谱总代数重数与迹范数分别为
\[
\sum_{j\in J}m_jn_j,
\qquad
\sum_{j\in J}m_jn_j|\beta_j|^{1/n_j}.
\]
\end{theorem}

\begin{proof}
每个直和块 \(C(n_j,\alpha_j)\otimes I_{m_j}\) 的奇异值都等于 \(|\alpha_j|\)，其重数为 \(m_jn_j\)。因此
\[
\|L_{\mathcal D}\|_1
=\sum_{j\in J}m_jn_j|\alpha_j|
=\sum_{j\in J}m_jn_j|\beta_j|^{1/n_j}
<\infty,
\]
故 \(L_{\mathcal D}\in\mathcal{S}_1(H_{\mathcal D})\)。又因 \(m_jn_j\ge 1\) 对所有 \(j\) 都成立，上述可和性也推出 \(r_{\mathcal D}<\infty\)。

对每个 \(j\)，由命题 \ref{prop:cycle-permutation-determinant} 有
\[
\det\!\bigl(I-r(C(n_j,\alpha_j)\otimes I_{m_j})\bigr)
=\det(I-rC(n_j,\alpha_j))^{m_j}
=\bigl(1-\beta_j r^{n_j}\bigr)^{m_j}.
\]
令
\[
L_{\mathcal D}^{(N)}
:=\bigoplus_{j\le N}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
\]
则
\[
\|L_{\mathcal D}-L_{\mathcal D}^{(N)}\|_1
=\sum_{j>N}m_jn_j|\beta_j|^{1/n_j}\to 0.
\]
由 Fredholm 行列式在迹范数下的连续性，
\[
\det(I-rL_{\mathcal D})
=\lim_{N\to\infty}\det(I-rL_{\mathcal D}^{(N)})
\]
在 \(r\) 的紧子集上一致成立，从而得到
\[
\det(I-rL_{\mathcal D})^{-1}
=\prod_{j\in J}\bigl(1-\beta_j r^{n_j}\bigr)^{-m_j}
\]
作为 \(r=0\) 附近的全纯 germ 成立。并且
\[
\|L_{\mathcal D}\|
=\sup_{j\in J}|\alpha_j|
=\sup_{j\in J}|\beta_j|^{1/n_j}
=r_{\mathcal D},
\]
故该恒等式在 \(|r|<r_{\mathcal D}^{-1}\) 上成立。

\textup{(ii)}\(\Rightarrow\)\textup{(i)} 显然。反过来假设 \textup{(i)} 成立。由定理前半部分，\(L_{\mathcal D}\) 是一个具有同一 Fredholm 行列式 germ 的正规迹类模型，因此
\[
\det(I-rL)=\det(I-rL_{\mathcal D})
\]
作为整函数恒等。由 \cite[Theorem~3.7]{Simon2005TraceIdeals}，\(\det(I-rT)\) 的零点恰为 \(T\) 的非零特征值倒数，且零点重数等于对应的代数重数。故 \(L\) 与 \(L_{\mathcal D}\) 具有完全相同的非零谱多重集。

由于 \(L\) 是紧正规算子，可作正交分解
\[
H=\ker L\oplus\bigoplus_{\lambda\in\mathrm{Spec}(L)\setminus\{0\}}E_\lambda,
\qquad
L|_{E_\lambda}=\lambda I.
\]
对每个非零特征值 \(\lambda\)，令
\[
J_\lambda
:=\bigl\{j\in J:\lambda=\alpha_j\omega
\text{ 对某个 }\omega^{n_j}=1\bigr\}.
\]
因为 \(\lambda\) 的代数重数有限，故 \(J_\lambda\) 为有限集，且
\[
\dim E_\lambda=\sum_{j\in J_\lambda}m_j.
\]
从而可选取正交分解
\[
E_\lambda=\bigoplus_{j\in J_\lambda}E_{\lambda,j},
\qquad
\dim E_{\lambda,j}=m_j.
\]
对每个 \(j\)，设
\[
H_j:=\bigoplus_{\omega^{n_j}=1}E_{\alpha_j\omega,j}.
\]
则各 \(H_j\) 两两正交、均为 \(L\) 的约化子空间，且 \(L|_{H_j}\) 酉等价于
\[
\bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}.
\]
另一方面，离散 Fourier 变换将 \(\Pi_{n_j}\) 对角化，因此
\[
\bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}
\simeq
C(n_j,\alpha_j)\otimes I_{m_j}.
\]
于是
\[
L\simeq 0_{\ker L}\oplus L_{\mathcal D},
\]
即得 \textup{(ii)}。最后一段关于非零谱总代数重数与迹范数的公式立即随之得到。证毕。
\end{proof}
