\begin{theorem}[\texorpdfstring{$\chi$}{chi}-标记在扫描—反演兼容互换下的刚性]\label{thm:fold-groupoid-chi-rigidity}
沿命题 \ref{prop:fold-groupoid-z2x2-sector-by-orientation} 的记号，在 Wedderburn 分解
\[
\CC[\mathcal R_m]\ \cong\ \bigoplus_{x\in X_m} M_{d_m(x)}(\CC)
\]
中，令 \(P(c_x),P(\iota_x)\in M_{d_m(x)}(\CC)\) 分别为扫描元与时间反演对合在块上的置换矩阵表示。
设 \(\Phi\in\mathrm{Aut}^{\ast}(\CC[\mathcal R_m])\) 满足：存在置换 \(\sigma:X_m\to X_m\) 与一族酉元
\(
U_x\in M_{d_m(\sigma(x))}(\CC)
\)
使得对每个 \(x\in X_m\) 有
\[
\Phi\!\left(M_{d_m(x)}(\CC)\right)=M_{d_m(\sigma(x))}(\CC),
\]
并且
\[
\Phi\!\bigl(P(c_x)\bigr)=U_x\,P(c_{\sigma(x)})\,U_x^{-1},
\qquad
\Phi\!\bigl(P(\iota_x)\bigr)=U_x\,P(\iota_{\sigma(x)})\,U_x^{-1}.
\]
则对所有 \(x\in X_m\) 都有
\[
\chi(\sigma(x))=\chi(x),
\qquad
\chi(x)=\bigl(\operatorname{sgn}(c_x),\operatorname{sgn}(\iota_x)\bigr)\in\{\pm1\}^2.
\]
\leanverified{sign\_conj\_eq\_sign}
\leanverified{sign\_eq\_of\_conj}
\leanverified{sign\_square\_eq\_one}
\leanverified{sign\_inv\_eq\_sign}
\leanverified{sign\_conj\_pow\_eq\_sign\_pow}
\leanverified{paper\_chi\_rigidity\_extended\_package}
\leanverified{sign\_double\_conj\_eq\_sign}
\leanverified{sign\_conj\_pow\_inv\_eq}
\leanverified{paper\_chi\_rigidity\_full\_homomorphism\_package}
\leanverified{paper\_fold\_groupoid\_chi\_rigidity}
\end{theorem}
\begin{proof}
对任意置换 \(\pi\) 的置换矩阵 \(P(\pi)\)，恒有 \(\det P(\pi)=\operatorname{sgn}(\pi)\in\{\pm1\}\)。
又因行列式对共轭不变，
\[
\det\!\bigl(\Phi(P(c_x))\bigr)
=\det\!\bigl(U_xP(c_{\sigma(x)})U_x^{-1}\bigr)
=\det P(c_{\sigma(x)}).
\]
另一方面，\(\Phi\) 在对应矩阵块上为 \(\ast\)-同构，故保持行列式（作为多项式函数）不变，从而
\(
\det(\Phi(P(c_x)))=\det(P(c_x))
\)；
于是 \(\operatorname{sgn}(c_{\sigma(x)})=\operatorname{sgn}(c_x)\)。
对 \(\iota_x\) 同理，得到 \(\operatorname{sgn}(\iota_{\sigma(x)})=\operatorname{sgn}(\iota_x)\)。
合并即得 \(\chi(\sigma(x))=\chi(x)\)。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\chi$}{chi}-保持的 \texorpdfstring{$\ast$}{*}-自同构群的半直积分解与分类]\label{prop:fold-groupoid-aut-chi-semidirect-classification}
对 \((\alpha,\beta)\in\{\pm1\}^2\) 记扇区两侧理想
\(
\CC[\mathcal{R}_m]^{\alpha,\beta}
\)
如命题 \ref{prop:fold-groupoid-z2x2-sector-by-orientation}。
定义
\[
\mathrm{Aut}^{\ast}_{\chi}(\CC[\mathcal R_m])
:=\left\{\Phi\in\mathrm{Aut}^{\ast}(\CC[\mathcal R_m]):\
\Phi\bigl(\CC[\mathcal{R}_m]^{\alpha,\beta}\bigr)=\CC[\mathcal{R}_m]^{\alpha,\beta}\ \ \forall(\alpha,\beta)\right\},
\]
以及允许的块置换群
\[
\Pi_{\chi}
:=\left\{\sigma\in\mathrm{Sym}(X_m):\ d_m(\sigma(x))=d_m(x),\ \chi(\sigma(x))=\chi(x)\ \ \forall x\in X_m\right\}.
\]
则存在自然群同构
\[
\mathrm{Aut}^{\ast}_{\chi}(\CC[\mathcal R_m])
\ \cong\
\Bigl(\prod_{x\in X_m}\mathrm{PU}\bigl(d_m(x)\bigr)\Bigr)\rtimes \Pi_{\chi}.
\]
\leanverified{paper\_fold\_groupoid\_aut\_chi\_semidirect\_classification}
\end{proposition}
\begin{proof}
由定理 \ref{thm:fold-groupoid-aut-semidirect-pi0}，
\(\mathrm{Aut}^{\ast}(\CC[\mathcal R_m])\)
分解为“逐块内自同构 \(\prod_x\mathrm{PU}(d_m(x))\)”与“同阶块的置换”之半直积。
\par
若 \(\Phi\in\mathrm{Aut}^{\ast}_{\chi}(\CC[\mathcal R_m])\)，则其诱导的块置换 \(\sigma\) 必满足：它不改变块阶数（否则不可能置换），并且保持扇区理想。
而扇区理想恰由 \(\chi(x)\) 分组的块直和给出（命题 \ref{prop:fold-groupoid-z2x2-sector-by-orientation}），
因此 \(\sigma\) 必满足 \(\chi(\sigma(x))=\chi(x)\)，即 \(\sigma\in\Pi_{\chi}\)。
反之，对任意 \(\sigma\in\Pi_{\chi}\) 与任意逐块共轭的内自同构，显然保持每个扇区两侧理想不变，从而给出 \(\mathrm{Aut}^{\ast}_{\chi}(\CC[\mathcal R_m])\) 的一个元素。
因此限制到 \(\chi\)-保持子群后，半直积结构与所述分类成立。证毕。
\end{proof}
