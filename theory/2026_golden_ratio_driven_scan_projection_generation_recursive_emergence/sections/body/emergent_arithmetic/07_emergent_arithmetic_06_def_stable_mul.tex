\subsection{乘法作为二层归一化：卷积与规范形}\label{subsec:emergent-arithmetic-stable-mul}

\begin{definition}[稳定乘法]\label{def:stable-mul}
对 $c,d\in X_\infty$，定义
\[
c\otimes d:=\mathrm{Val}^{-1}\bigl(\mathrm{Val}(c)\cdot\mathrm{Val}(d)\bigr)\in X_\infty.
\]
\leanverified{stableMul}
\end{definition}

\begin{theorem}[乘法的定义性同构（记号固定）]\label{thm:mul-definitional}
$(X_\infty,\oplus,\otimes)$ 为交换半环，且 $\mathrm{Val}$ 给出与 $(\NN,+,\times)$ 的半环同构。
\leanverified{paper\_stable\_commutative\_ring}
\leanverified{paper\_stableValue\_pow}
\leanverified{paper\_totient\_fib\_9\_10}
\leanverified{paper\_seven\_mul\_three\_zero\_X6}
\end{theorem}

\textbf{证明：}
由定义 \ref{def:stable-add} 与 \ref{def:stable-mul}，$\mathrm{Val}$ 同时保持加法与乘法。由于 $\mathrm{Val}$ 为双射，交换律、结合律与分配律均从 $(\NN,+,\times)$ 传递，因而得到交换半环同构。

由此，自然数、加法、乘法在稳定地址空间中完全由\textbf{语法约束 + 规范形归一化}定义，而不需将 $(+,\times)$ 作为原语预置。

\subsubsection{乘法由加法迭代内生}\label{subsec:mul-as-iterated-add}
\begin{theorem}[乘法可由加法与迭代内生]\label{thm:mul-by-iterated-add}
在 $X_\infty$ 上定义
\[
c\odot d:=\underbrace{c\oplus c\oplus\cdots\oplus c}_{\mathrm{Val}(d)\text{ 次}},
\qquad
\mathrm{Val}(d)=0\text{ 时取 }0_X,
\]
则对任意 $c,d\in X_\infty$，$c\odot d=c\otimes d$。
\leanverified{iteratedStableAdd}
\leanverified{stableValue\_iteratedStableAdd}
\leanverified{iteratedStableAdd\_eq\_stableMul}
\leanverified{paper\_mul\_by\_iterated\_add}
\end{theorem}

\textbf{证明：}
由定理 \ref{thm:add-definitional}，$\mathrm{Val}$ 保持 $\oplus$ 且双射，故
\[
\mathrm{Val}(c\odot d)=\mathrm{Val}(c)\mathrm{Val}(d)=\mathrm{Val}(c\otimes d),
\]
从而 $c\odot d=c\otimes d$。证毕。

\begin{corollary}[“合并”的第一种严格版本]\label{cor:mul-no-new-primitive}
$\otimes$ 由 $\oplus$ 与可迭代计数唯一确定，无需作为新的原语引入。
\end{corollary}

S:X_\infty\to X_\infty,\qquad S(c):=\Fold_\infty(c+e_1).

\begin{theorem}[后继结构同构]\label{thm:successor-structure}
$\mathrm{Val}$ 给出结构同构

\mathrm{Val}(0_X)=0,\qquad \mathrm{Val}(S(c))=\mathrm{Val}(c)+1,
\]
并且 $\mathrm{Val}$ 为双射。

$\mathrm{Val}(0_X)=0$。又

\mathrm{Val}(S(c))=\mathrm{Val}(c+e_1)=\mathrm{Val}(c)+\mathrm{Val}(e_1)=\mathrm{Val}(c)+1,

\begin{corollary}[加法由后继迭代给出]\label{cor:add-from-successor}
令
\[
c\oplus_S d:=S^{\circ\mathrm{Val}(d)}(c),
\]
则 $c\oplus_S d=c\oplus d$，且 $c\oplus d=\Fold_\infty(c+d)$。
\leanverified{stableValue\_stableSucc\_iterate}
\leanverified{stableSucc\_iterate\_eq\_stableAdd}
\leanverified{paper\_add\_from\_successor}
\end{corollary}

\textbf{证明：}
由定理 \ref{thm:successor-structure}，$\mathrm{Val}(c\oplus_S d)=\mathrm{Val}(c)+\mathrm{Val}(d)$，由双射性得 $c\oplus_S d=c\oplus d$；折叠公式由推论 \ref{cor:add-as-fold}。证毕。

\begin{theorem}[乘法由后继迭代内生]\label{thm:mul-from-successor}
定义
\[
c\otimes_S d:=\underbrace{c\oplus_S c\oplus_S\cdots\oplus_S c}_{\mathrm{Val}(d)\text{ 次}},
\qquad \mathrm{Val}(d)=0\text{ 时取 }0_X,
\]
则 $c\otimes_S d=c\otimes d$。
\leanverified{stableMul\_from\_successor}
\leanverified{paper\_mul\_from\_successor}
\end{theorem}

\textbf{证明：}
由推论 \ref{cor:add-from-successor}，$\oplus_S=\oplus$，故
\[
\mathrm{Val}(c\otimes_S d)=\mathrm{Val}(c)\mathrm{Val}(d)=\mathrm{Val}(c\otimes d),
\]
由双射性得结论。证毕。

\begin{corollary}[极简原语集]\label{cor:primitive-minimal-successor}
\leanverified{paper\_ea\_primitive\_minimal\_successor}
稳定地址层的算术由后继 $S$ 与折叠 $\Fold_\infty$ 生成：$\oplus$ 为 $S$ 的一层迭代读出，$\otimes$ 为该迭代的二层叠加读出。
\end{corollary}

\subsubsection{单一原语 \texorpdfstring{$\circ$}{o} 的两层读出}\label{subsec:arith-composition}
本小节给出一个与“扫描迭代 $\Rightarrow$ 稳定编码 $\Rightarrow$ 归一化读出”链条对齐的定理化版本：不再将加法与乘法作为原语预置，仅保留\textbf{复合}这一最底层原语，并在稳定地址层通过折叠回到规范形。
取任意非空集合 $A$ 与自映射 $g:A\to A$。记
\[
S_0:=\langle g\rangle=\{g^{\circ n}:n\in\NN\}\subseteq \mathrm{End}(A),\qquad
S_1:=\{I_n:n\in\NN\}\subseteq \mathrm{End}(\mathrm{End}(A)),
\]
其中 $I_n(f):=f^{\circ n}$。两层对象的唯一运算均为复合 $\circ$。

\begin{theorem}[单一原语 \texorpdfstring{$\circ$}{o} 的两层读出同时生成 $(+,\times)$]\label{thm:composition-two-layer}
定义嵌入 $\iota_0:\NN\to S_0$ 为
\[
\iota_0(n):=g^{\circ n},\qquad g^{\circ 0}:=\mathrm{id}_A.
\]
并令 $\iota_1:\NN\to S_1$ 为 $\iota_1(n):=I_n$。则对任意 $m,n\in\NN$，成立
\[
\iota_0(m+n)=\iota_0(m)\circ \iota_0(n),
\qquad
\iota_1(mn)=\iota_1(m)\circ \iota_1(n).
\]
\leanverified{paper\_ea\_composition\_two\_layer\_seeds}
\leanverified{paper\_ea\_composition\_two\_layer\_package}
\leanverified{paper\_composition\_two\_layer}
\end{theorem}

\textbf{证明：}
由迭代定义与复合结合律，
\[
g^{\circ(m+n)}=g^{\circ m}\circ g^{\circ n},
\]
故 $\iota_0(m+n)=\iota_0(m)\circ \iota_0(n)$。另一方面，对任意 $f\in\mathrm{End}(A)$，
\[
(\iota_1(m)\circ \iota_1(n))(f)=I_m(I_n(f))=I_m(f^{\circ n})=(f^{\circ n})^{\circ m}
=f^{\circ (nm)}=I_{mn}(f).
\]
由于对一切 $f$ 相等，故 $\iota_1(m)\circ \iota_1(n)=\iota_1(mn)$。证毕。

\begin{theorem}[稳定地址半环的“去原语化”实现]\label{thm:arith-composition}
沿用定义 \ref{def:stable-add} 与 \ref{def:stable-mul}，并记 $\mathrm{Val}:X_\infty\to\NN$ 为值映射。对任意定理 \ref{thm:composition-two-layer} 中的 $(A,g)$，定义两条“过程语义”映射
\[
\Phi_0:X_\infty\to \mathrm{End}(A),\quad \Phi_0(c):=g^{\circ\mathrm{Val}(c)},
\]
\[
\Phi_1:X_\infty\to \mathrm{End}(\mathrm{End}(A)),\quad \Phi_1(c):=I_{\mathrm{Val}(c)}.
\]
则对任意 $c,d\in X_\infty$，
\[
\Phi_0(c\oplus d)=\Phi_0(c)\circ\Phi_0(d),\qquad
\Phi_1(c\otimes d)=\Phi_1(c)\circ\Phi_1(d).
\]
\leanverified{paper\_arith\_composition}
也即：在稳定地址层的 $\oplus,\otimes$ 同时被同一原语 $\circ$ 在两层对象上实现。
\end{theorem}

\textbf{证明：}
由 $\oplus$ 定义与定理 \ref{thm:composition-two-layer} 的第一条，
\[
\Phi_0(c\oplus d)=g^{\circ\mathrm{Val}(c\oplus d)}
=g^{\circ(\mathrm{Val}(c)+\mathrm{Val}(d))}
 =g^{\circ\mathrm{Val}(c)}\circ g^{\circ\mathrm{Val}(d)}
=\Phi_0(c)\circ\Phi_0(d).
\]
同理由 $\otimes$ 定义与定理 \ref{thm:composition-two-layer} 的第二条，
\[
\Phi_1(c\otimes d)=I_{\mathrm{Val}(c\otimes d)}
=I_{\mathrm{Val}(c)\mathrm{Val}(d)}
=I_{\mathrm{Val}(c)}\circ I_{\mathrm{Val}(d)}
=\Phi_1(c)\circ\Phi_1(d).
\]
证毕。

\begin{corollary}[仅复合的读出落回稳定地址]\label{cor:composition-pullback}
记 $Z:=\mathrm{Val}^{-1}$，并定义
\[
\rho_0:S_0\to X_\infty,\quad \rho_0(g^{\circ n}):=Z(n),\qquad
\rho_1:S_1\to X_\infty,\quad \rho_1(I_n):=Z(n).
\]
则对任意 $c=Z(m),d=Z(n)\in X_\infty$，
\[
c\oplus d=\rho_0\bigl(\rho_0^{-1}(c)\circ \rho_0^{-1}(d)\bigr),
\qquad
c\otimes d=\rho_1\bigl(\rho_1^{-1}(c)\circ \rho_1^{-1}(d)\bigr).
\]
\end{corollary}

\textbf{证明：}
由定理 \ref{thm:composition-two-layer} 得 $\rho_0(g^{\circ m}\circ g^{\circ n})=Z(m+n)$ 与 $\rho_1(I_m\circ I_n)=Z(mn)$，再与 $\oplus,\otimes$ 的定义对齐即可。证毕。
\leanverified{paper\_composition\_pullback}

\begin{corollary}[“合并掉 $(+,\times)$”的严格表述]\label{cor:primitive-free-arith}
在配置层，算术原语可压缩为以下最小生成集：
\begin{itemize}
  \item 计数配置的逐点叠加（自由交换幺半群结构）；
  \item Fibonacci 关系诱导的局部可逆重写（群胚生成元）；
  \item 选取规范形截面 $\Fold_\infty$（或等价的 $\mathrm{Val}$ 与 Zeckendorf 规范形 $Z$）。
\end{itemize}
在过程层，等价地可用复合运算 $\circ$ 与折叠 $\Fold_\infty$ 来表达：$\oplus$ 与 $\otimes$ 分别是复合在一层与二层对象上的读出，并经折叠归一化后落回 $X_\infty$。
等价地，运算层只需 $\circ$，可见层只需编码 $Z$（或 $\mathrm{Val}/\Fold_\infty$）。
\end{corollary}

\textbf{证明：}
配置层的三类原语由定理 \ref{thm:monoid-quotient-is-N}、定理 \ref{thm:zeckendorf-transversal} 与推论 \ref{cor:fold-as-section} 直接给出；过程层的两层复合读出由定理 \ref{thm:composition-two-layer} 与 \ref{thm:arith-composition} 给出。加法的折叠接口由推论 \ref{cor:add-as-fold} 保证，因而结论成立。
\leanverified{paper\_primitive\_free\_arith}

\begin{remark}[商余作为规范形选择]\label{rem:division-as-selection}
给定 $a,b\in X_\infty$ 且 $\mathrm{Val}(b)\ge 1$，整数层的商余关系
\[
\mathrm{Val}(a)=\mathrm{Val}(q)\,\mathrm{Val}(b)+\mathrm{Val}(r),\qquad 0\le \mathrm{Val}(r)<\mathrm{Val}(b)
\]
可等价重写为稳定地址层的方程
\[
a=q\otimes b\ \oplus\ r,
\]
再配合不等式约束选取唯一的规范代表 $(q,r)$。因此“除法/商余”并非新的原语，而是同一重写群胚中的\textbf{解方程 + 规范形选择}问题；其可计算性完全由 $\Fold_\infty$ 的可计算性继承。
\end{remark}

\begin{proposition}[可计算实现]\label{prop:zeckendorf-arith-computable}
\leanverified{paper\_zeckendorf\_arith\_computable}
对任意 $c,d\in X_\infty$，$c\oplus d$ 与 $c\otimes d$ 可由有限步算法计算：先在值层执行整数加法/乘法得到 $\mathrm{Val}(c)\pm\mathrm{Val}(d)$ 或 $\mathrm{Val}(c)\mathrm{Val}(d)$，再以 Zeckendorf 贪心算法取唯一规范形（附录 C）。
\end{proposition}

\begin{theorem}[有限乘法表的细化相容性判别]\label{distill:cech_cohomology-refinement_stability_separation-ea-finite-multiplication-shadow-criterion}
令
\[
r_m:X_\infty\longrightarrow X_m
\]
为取前 \(m\) 个稳定 Fibonacci 数字的有限分辨率投影，并在 \(X_\infty\) 上使用定义 \ref{def:stable-mul} 的稳定乘法 \(\otimes\)。给定任意有限二元运算
\[
\star_m:X_m\times X_m\longrightarrow X_m,
\]
称 \(\star_m\) 为 \(\otimes\) 的深度 \(m\) 可采纳阴影，若对所有 \(c,d\in X_\infty\) 均有
\[
r_m(c\otimes d)=r_m(c)\star_m r_m(d).
\]
则 \(\star_m\) 可采纳当且仅当下列两项同时成立：
\begin{enumerate}
  \item 乘法对深度 \(m\) 等价关系相容：若
  \[
  r_m(c)=r_m(c'),\qquad r_m(d)=r_m(d'),
  \]
  则
  \[
  r_m(c\otimes d)=r_m(c'\otimes d');
  \]
  \item 对任意 \(x,y\in X_m\)，若取任意提升 \(c,d\in X_\infty\) 满足
  \(r_m(c)=x\)、\(r_m(d)=y\)，则
  \[
  x\star_m y=r_m(c\otimes d).
  \]
\end{enumerate}
若第一项失败，则存在四元缺口账本
\[
(c,c',d,d')
\]
具有相同的有限输入读出而给出不同的有限乘积读出。若第一项成立而第二项失败，则存在表格缺口账本
\[
(x,y,c,d)
\]
使 \(x=r_m(c)\)、\(y=r_m(d)\)，但 \(x\star_m y\ne r_m(c\otimes d)\)。在任一失败情形中，\(\star_m\) 都只能被记录为有限分辨率算术 artifact，不能作为第 \ref{sec:emergent-arithmetic} 节中值保持重写或稳定乘法的对象层替代品。
\end{theorem}
\begin{definition}[有限乘法阴影的 incidence-coordinate 见证]\label{def:distill-ea-finite-mul-shadow-incidence-witness}
设 \(U,V\subseteq X_\infty\) 为有限集合，并设
\(\Gamma\subseteq U\times V\times X_\infty\) 为一个有限三元关系。称 \(\Gamma\) 在 \(U\times V\) 上由加法链见证，若满足：
\begin{enumerate}
  \item 对每个 \((c,d)\in U\times V\)，存在 \(e\in X_\infty\) 使 \((c,d,e)\in\Gamma\)；
  \item 对每个 \((c,d,e)\in\Gamma\)，令 \(N:=\mathrm{Val}(d)\)，存在序列
  \[
  u_0,u_1,\ldots,u_N\in X_\infty
  \]
  使
  \[
  u_0=0_X,\qquad u_{j+1}=u_j\oplus c\quad(0\le j<N),\qquad u_N=e.
  \]
\end{enumerate}
\end{definition}

\begin{proposition}[有限乘法阴影的 incidence 唯一性]\label{prop:distill-ea-finite-mul-shadow-incidence-uniqueness}
若 \(\Gamma\subseteq U\times V\times X_\infty\) 在 \(U\times V\) 上由加法链见证，则
\[
\Gamma=\{(c,d,c\otimes d):c\in U,\ d\in V\}.
\]
反之，上式右端给出的稳定乘法图在 \(U\times V\) 上由加法链见证。因而任何有限乘法阴影只要通过该 incidence-coordinate 测试，便没有额外的非函数性分支，也不能偏离定义 \ref{def:stable-mul} 的稳定乘法。
\end{proposition}
\begin{proof}
先设 \((c,d,e)\in\Gamma\)，并取定义 \ref{def:distill-ea-finite-mul-shadow-incidence-witness} 中的见证链。由定理 \ref{thm:add-definitional}，
\(\mathrm{Val}\) 保持 \(\oplus\)，故归纳得
\[
\mathrm{Val}(u_j)=j\,\mathrm{Val}(c)\qquad(0\le j\le N).
\]
在 \(j=0\) 时这是 \(u_0=0_X\)；若已知 \(j\) 阶成立，则
\[
\mathrm{Val}(u_{j+1})=\mathrm{Val}(u_j\oplus c)
=\mathrm{Val}(u_j)+\mathrm{Val}(c)
=(j+1)\mathrm{Val}(c).
\]
取 \(j=N=\mathrm{Val}(d)\)，得
\[
\mathrm{Val}(e)=\mathrm{Val}(c)\mathrm{Val}(d)=\mathrm{Val}(c\otimes d),
\]
其中最后一步用定义 \ref{def:stable-mul}。由于 \(\mathrm{Val}\) 为双射，故 \(e=c\otimes d\)。这说明
\(\Gamma\subseteq\{(c,d,c\otimes d):c\in U,d\in V\}\)。另一方面，见证定义的第一项保证每个 \((c,d)\) 至少出现一个 \(e\)，而刚才已证明该 \(e\) 必为 \(c\otimes d\)，故反向包含也成立。

反过来，给定 \((c,d)\in U\times V\)，令 \(N=\mathrm{Val}(d)\)，并设
\[
u_j:=\mathrm{Val}^{-1}(j\,\mathrm{Val}(c))\qquad(0\le j\le N).
\]
则 \(\nu_0=0_X\)，且由 \(\mathrm{Val}\) 对加法的定义性同构可知
\[
\mathrm{Val}(\nu_j\oplus c)=(j+1)\mathrm{Val}(c)=\mathrm{Val}(\nu_{j+1}),
\]
故 \(\nu_{j+1}=\nu_j\oplus c\)。同时
\[
\nu_N=\mathrm{Val}^{-1}(\mathrm{Val}(c)\mathrm{Val}(d))=c\otimes d.
\]
这给出稳定乘法图的加法链见证。证毕。
\end{proof}

先设 \(\star_m\) 可采纳。若

\end{proof}
