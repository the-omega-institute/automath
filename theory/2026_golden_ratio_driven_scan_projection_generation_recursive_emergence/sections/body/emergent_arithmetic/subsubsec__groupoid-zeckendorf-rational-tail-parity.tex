\begin{theorem}[纤维尾分布对连通连续对称的奇数有理同伦阶梯具有完全决定性]\label{thm:fold-groupoid-aut0-rational-tail-spheres}
沿用定理 \ref{thm:fold-groupoid-aut0-rational-cohomology} 的记号，记
\[
K_m:=\mathrm{Aut}^{\ast}(\CC[\mathcal{R}_m])_0
\cong
\prod_{x\in X_m}\mathrm{PU}\bigl(d_m(x)\bigr).
\]
对每个整数 \(r\ge 2\)，定义纤维尾计数
\[
N_{\ge r}(m):=\#\{x\in X_m:\ d_m(x)\ge r\}.
\]
则 \(K_m\) 的有理同伦型完全由尾分布 \(\{N_{\ge r}(m)\}_{r\ge 2}\) 决定；更精确地，
\[
K_m\simeq_{\QQ}\prod_{r\ge 2}\bigl(S^{2r-1}\bigr)^{N_{\ge r}(m)}.
\]
等价地，其有理上同调环满足
\[
H^\ast(K_m;\QQ)
\cong
\Lambda_{\QQ}\!\Bigl(
\xi_{r,j}:\ 2\le r\le D_m,\ 1\le j\le N_{\ge r}(m),\ \deg \xi_{r,j}=2r-1
\Bigr),
\]
并且有理同伦群满足
\[
\operatorname{rank}_{\QQ}\bigl(\pi_{2r-1}(K_m)\otimes\QQ\bigr)=N_{\ge r}(m)
\qquad (r\ge 2),
\]
\[
\pi_q(K_m)\otimes\QQ=0
\qquad
\bigl(q\ \text{为偶数，或 }q>2D_m-1\bigr).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-groupoid-aut0-rational-cohomology}，
\[
K_m\cong \prod_{x\in X_m}\mathrm{PU}\bigl(d_m(x)\bigr).
\]
对每个 \(d\ge 2\)，\(\mathrm{PU}(d)=\mathrm{SU}(d)/\mu_d\) 仅差有限中心商，故与 \(\mathrm{SU}(d)\) 具有相同的有理同伦型，而经典结果给出
\[
\mathrm{SU}(d)\simeq_{\QQ} S^3\times S^5\times\cdots\times S^{2d-1}.
\]
因此
\[
\mathrm{PU}(d)\simeq_{\QQ}\prod_{r=2}^{d}S^{2r-1}.
\]
把这一分解对全部 \(x\in X_m\) 取直积，可得
\[
K_m
\simeq_{\QQ}
\prod_{x\in X_m}\prod_{r=2}^{d_m(x)}S^{2r-1}
\simeq_{\QQ}
\prod_{r\ge 2}\bigl(S^{2r-1}\bigr)^{\#\{x:\ d_m(x)\ge r\}}
\]
\[
=
\prod_{r\ge 2}\bigl(S^{2r-1}\bigr)^{N_{\ge r}(m)}.
\]
这就得到有理同伦型的乘积分解。

奇维球有限乘积的有理上同调是相应奇次数生成元上的外代数，故得到所示上同调环结构。再由有理 Hurewicz--Sullivan 理论，每个 \(S^{2r-1}\) 因子在次数 \(2r-1\) 上贡献一个一维有理同伦群，而在偶数次数上无贡献；将这些贡献相加即得同伦群秩公式。证毕。
\end{proof}

\begin{theorem}[连通连续对称的全部有理拓扑复杂度恰等于微态--类型缺口]\label{thm:fold-groupoid-aut0-rational-generator-gap}
沿用定理 \ref{thm:fold-groupoid-aut0-rational-tail-spheres} 的记号，定义
\[
g_{\mathrm{rat}}(m):=\sum_{r\ge 2}N_{\ge r}(m).
\]
则有完全闭式
\[
g_{\mathrm{rat}}(m)=\sum_{x\in X_m}\bigl(d_m(x)-1\bigr)=2^m-|X_m|.
\]
进一步，\(K_m\) 的有理 cup-length 与 rational LS-category 满足
\[
\operatorname{cup}_{\QQ}(K_m)=\operatorname{cat}_0(K_m)=2^m-|X_m|.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-groupoid-aut0-rational-tail-spheres}，若某个纤维满足 \(d_m(x)=d\)，则对应因子 \(\mathrm{PU}(d)\) 在有理同伦型中贡献
\[
S^3\times S^5\times\cdots\times S^{2d-1},
\]
故恰有 \(d-1\) 个奇维球因子。于是
\[
g_{\mathrm{rat}}(m)=\sum_{x\in X_m}\bigl(d_m(x)-1\bigr).
\]
另一方面，
\[
\sum_{x\in X_m}d_m(x)=|\Omega_m|=2^m,
\]
故
\[
g_{\mathrm{rat}}(m)=\sum_x d_m(x)-\sum_x1=2^m-|X_m|.
\]

再由定理 \ref{thm:fold-groupoid-aut0-rational-tail-spheres}，
\[
H^\ast(K_m;\QQ)\cong \Lambda_{\QQ}(V_m),
\qquad
\dim_{\QQ}V_m=g_{\mathrm{rat}}(m).
\]
外代数中全部奇数生成元的乘积非零，而再乘任何新的奇生成元便为零，因此
\[
\operatorname{cup}_{\QQ}(K_m)=\dim_{\QQ}V_m=g_{\mathrm{rat}}(m).
\]
同时，\(K_m\) 的极小 Sullivan 模型同伦等价于 \((\Lambda V_m,0)\)。对零微分且生成元全为奇数的极小模型，rational LS-category 等于生成元数，故
\[
\operatorname{cat}_0(K_m)=\dim_{\QQ}V_m=g_{\mathrm{rat}}(m).
\]
综上，
\[
\operatorname{cup}_{\QQ}(K_m)=\operatorname{cat}_0(K_m)=2^m-|X_m|.
\]
证毕。

\end{proof}

\begin{corollary}[window-\(6\) 的退化谱尾与有理同伦阶梯完全重合]\label{cor:fold-groupoid-window6-rational-tail-ladder}
\leanverified{paper\_fold\_groupoid\_window6\_rational\_tail\_ladder}
在 \(m=6\) 时，由推论 \ref{cor:fold-bin-m6-fiber-hist}，
\[
|X_6|=21,
\qquad
2:8,\ 3:4,\ 4:9.
\]
因此
\[
N_{\ge 2}(6)=21,\qquad
N_{\ge 3}(6)=13,\qquad
N_{\ge 4}(6)=9.
\]
于是
\[
K_6\simeq_{\QQ}(S^3)^{21}\times (S^5)^{13}\times (S^7)^9,
\]
\[
H^\ast(K_6;\QQ)\cong \Lambda(V_3\oplus V_5\oplus V_7),
\qquad
\dim V_3=21,\ \dim V_5=13,\ \dim V_7=9,
\]
并且其 Poincar\'e 多项式满足
\[
P_{K_6}(t)=(1+t^3)^{21}(1+t^5)^{13}(1+t^7)^9.
\]
相应地，
\[
\operatorname{rank}_{\QQ}\bigl(\pi_3(K_6)\otimes\QQ\bigr)=|X_6|=21,
\]
\[
\operatorname{rank}_{\QQ}\bigl(\pi_5(K_6)\otimes\QQ\bigr)
=
|X_6|-\#\{x\in X_6:\ d_6(x)=2\}
=13,
\]
\[
\operatorname{rank}_{\QQ}\bigl(\pi_7(K_6)\otimes\QQ\bigr)
=
\#\{x\in X_6:\ d_6(x)=4\}
=9.
\]
\end{corollary}
\begin{proof}
由直方图
\[
2:8,\qquad 3:4,\qquad 4:9
\]
直接得到
\[
N_{\ge 2}(6)=8+4+9=21,
\qquad
N_{\ge 3}(6)=4+9=13,
\qquad
N_{\ge 4}(6)=9.
\]
代入定理 \ref{thm:fold-groupoid-aut0-rational-tail-spheres} 即得有理同伦型、上同调环与有理同伦群秩。Poincar\'e 多项式由奇维球乘积对应的外代数公式直接得到。最后三条秩公式只是把尾计数按直方图重写。证毕。
\end{proof}

\begin{proposition}[window-\(6\) 的 boundary sheet parity 在有理连续包络中完全不可见]\label{prop:fold-groupoid-window6-boundary-parity-rationally-invisible}
\leanverified{paper\_fold\_groupoid\_window6\_boundary\_parity\_rationally\_invisible}
设
\[
C_{\mathrm{bdry}}
:=
\langle \tau_w:\ w\in X_6^{\mathrm{bdry}}\rangle
\cong (\ZZ/2\ZZ)^3
\]
为命题 \ref{prop:xi-time-part60ab3-window6-boundary-center-residual} 中的 boundary 中心残余子群。由命题 \ref{prop:fold-discrete-to-continuous-gauge-embedding}，其经离散规范群到连通连续对称群的自然嵌入可视为 \(K_6\) 的一个有限子群。记该包含映射仍为
\[
\iota:C_{\mathrm{bdry}}\hookrightarrow K_6.
\]
则对每个 \(q>0\)，诱导上同调拉回
\[
\iota^\ast:H^q(K_6;\QQ)\longrightarrow H^q(C_{\mathrm{bdry}};\QQ)
\]
恒为零。等价地，任一有理特征类、任一有理同伦不变量，以及任一通过 \(\QQ\)-系数 de Rham 或 Sullivan 机制构造的连续证书，都无法探测 window-\(6\) 的 boundary sheet parity；该 parity 只能在积分或模 \(2\) 层面被看见。
\end{proposition}
\begin{proof}
有限群 \(C_{\mathrm{bdry}}\cong(\ZZ/2\ZZ)^3\) 的有理上同调在正次数上恒为零：
\[
H^q(C_{\mathrm{bdry}};\QQ)=0
\qquad (q>0),
\]
因为特征 \(0\) 系数下有限群代数是半单的。故对任意 \(q>0\)，包含映射 \(\iota\) 诱导的拉回
\[
\iota^\ast:H^q(K_6;\QQ)\to H^q(C_{\mathrm{bdry}};\QQ)
\]
只能是零映射。

另一方面，由推论 \ref{cor:fold-groupoid-window6-rational-tail-ladder}，\(K_6\) 的全部有理同伦理论已完全由奇维球乘积
\[
(S^3)^{21}\times (S^5)^{13}\times (S^7)^9
\]
所决定；而有限 \(2\)-群 \(C_{\mathrm{bdry}}\) 的 rationalization 同伦等价于一点。因此 boundary sheet parity 在有理连续包络中彻底消失，只能保留为 \(2\)-primary 的离散阴影。证毕。
\end{proof}
