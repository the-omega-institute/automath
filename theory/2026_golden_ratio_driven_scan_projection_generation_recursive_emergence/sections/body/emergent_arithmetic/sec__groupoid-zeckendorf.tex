\subsection{群胚视角：值保持重写的轨道与 Zeckendorf 横截面}\label{sec:groupoid-zeckendorf}

上一小节已经把稳定算术表述为一个商结构：所有同值配置在 Fibonacci 关系下落入同一同余类，而 Zeckendorf 规范形给出每个同余类的唯一代表。
本小节给出与之等价但更操作性的表述：把这些关系实现为一族可逆局部变换，并把规范化解释为沿值保持轨道投影到唯一横截面。
其意义有二：一方面，它把“同值”从静态商关系改写为可执行的群胚轨道；另一方面，它为后续定向化归一化重写提供直接起点。

\begin{definition}[Fibonacci 局部变换（可逆重写）]\label{def:fib-moves}
定义以下作用于 $\mathcal{D}$ 的部分可逆变换（在各自可行域上）：
\begin{itemize}
  \item \textbf{基准变换：}
  \[
  E_0(a)=a+2e_1-e_2\quad\text{当 }a_{2}\ge 1,
  \qquad
  E_0^{-1}(a)=a-2e_1+e_2\quad\text{当 }a_{1}\ge 2.
  \]
  \item \textbf{递推变换（$k\ge 1$）：}
  \[
  E_k(a)=a+e_k+e_{k+1}-e_{k+2}\quad\text{当 }a_{k+2}\ge 1,
  \]
  其逆变换 $E_k^{-1}$ 在 $a_k,a_{k+1}\ge 1$ 时定义为
  \[
  E_k^{-1}(a)=a-e_k-e_{k+1}+e_{k+2}.
  \]
\end{itemize}
令 $\mathcal{G}$ 为由 $\{E_k,E_k^{-1}\}_{k\ge 0}$ 生成的群胚（对象为 $\mathcal{D}$，态射为这些部分双射的复合）。
\end{definition}

\begin{proposition}[轨道不变量]\label{prop:orbits-preserve-val}
对任意群胚态射（由 $E_k$ 与 $E_k^{-1}$ 复合得到的可逆变换）$a\mapsto a'$，都有 $\mathrm{Val}(a)=\mathrm{Val}(a')$。
\end{proposition}

\textbf{证明：}
对生成元 $E_k$，
\[
\mathrm{Val}(E_k(a))-\mathrm{Val}(a)
=F_{k+1}+F_{k+2}-F_{k+3}=0,
\]
逆变换同理。复合保持不变量。
对基准生成元 $E_0$ 亦有 $\mathrm{Val}(E_0(a))-\mathrm{Val}(a)=2F_2-F_3=0$，故结论成立。
\leanverified{paper\_orbits\_preserve\_val}

\begin{theorem}[轨道横截面与折叠的群论解释]\label{thm:zeckendorf-transversal}
对任意 $a\in\mathcal{D}$，其 $\mathcal{G}$-轨道与其值纤维一致：
\[
\mathcal{G}\cdot a \ =\ \{b\in\mathcal{D}:\mathrm{Val}(b)=\mathrm{Val}(a)\}.
\]
并且该轨道与 $X_\infty$ 恰好在一点相交，交点为 $Z(\mathrm{Val}(a))$。因此定义
\[
\Fold_\infty(a):=Z(\mathrm{Val}(a))\in X_\infty
\]
给出“沿轨道投影到横截面”的折叠算子；它把每个不可观测的值保持对称轨道压缩为唯一稳定代表。
\end{theorem}

\textbf{证明：}
由命题 \ref{prop:orbits-preserve-val} 得 $\mathcal{G}\cdot a$ 包含于值纤维。反向包含由定理 \ref{thm:monoid-quotient-is-N} 给出：若 $\mathrm{Val}(b)=\mathrm{Val}(a)$，则 $[b]=[a]$，即 $b\equiv a$。而 $\equiv$ 正是由基准同余 $e_2\equiv 2e_1$ 与递推同余 $e_{k+2}\equiv e_{k+1}+e_k$ 在配置空间上的闭包；相应的局部可逆重写由 $E_0$ 与 $E_k$（$k\ge 1$）实现，故存在群胚态射把 $a$ 送到 $b$。最后，值纤维中落在 $X_\infty$ 的元素按 Zeckendorf 唯一性至多一个，而由 $Z$ 的定义知 $\mathrm{Val}(Z(\mathrm{Val}(a)))=\mathrm{Val}(a)$，即 $Z(\mathrm{Val}(a))$ 属于该值纤维，故交点唯一且为所述。
\leanverified{paper\_zeckendorf\_transversal}

\begin{remark}[两套自由交换幺半群坐标与规范截面]\label{rem:two-coordinate-monoids}
配置空间 $\mathcal{D}$ 是由生成元 $\{e_k\}_{k\ge 1}$ 生成的自由交换幺半群。值映射 $\mathrm{Val}$ 按 Fibonacci 关系取商并与 $(\NN,+)$ 同构（定理 \ref{thm:monoid-quotient-is-N}）；折叠 $\Fold_\infty$ 给出该商映射的一个\textbf{规范截面}：每个同余类唯一代表在 $X_\infty$。
另一方面，乘法半群 $(\NN_{>0},\times)$ 是由素数集合生成的自由交换幺半群：整数 $n$ 对应指数向量 $(v_p(n))_p$。因此同一个“整数对象”拥有两套自由坐标系：Fibonacci/Zeckendorf 坐标适配加法归一化，素数指数坐标适配乘法的线性化。二者之间的坐标变换携带全局算术信息；在扫描—投影读出框架中，这些非局部信息必须由“携带/碰撞/多重度”等缺陷维度承载。
\end{remark}

\subsubsection{定向化归一化重写（\texorpdfstring{$\Fold_\infty$}{Fold\_infty} 的编译表述）}\label{subsec:directed-rewrite}
定理 \ref{thm:zeckendorf-transversal} 已经把折叠解释为“沿值保持轨道投影到唯一稳定代表”。
为获得直接可执行的归一化过程，只需把群胚生成元定向为一套值保持重写：
\begin{itemize}
  \item \textbf{相邻合并规则 (\ref{rule:fold:adj})：}若 $a_k\ge 1$ 且 $a_{k+1}\ge 1$，则
  \[
  a\ \to\ a-e_k-e_{k+1}+e_{k+2}.
  \]
  \item \textbf{底位去重规则 (\ref{rule:fold:dedup1})：}若 $a_1\ge 2$，则
  \[
  a\ \to\ a-2e_1+e_2.
  \]
  \item \textbf{第二位去重规则 (\ref{rule:fold:dedup2})：}若 $a_2\ge 2$，则
  \[
  a\ \to\ a-2e_2+e_1+e_3.
  \]
  \item \textbf{一般位去重规则 (\ref{rule:fold:dedupk})（$k\ge 3$）：}若 $a_k\ge 2$，则
  \[
  a\ \to\ a-2e_k+e_{k-2}+e_{k+1}.
  \]
\end{itemize}
它们分别对应
\(
F_{k+1}+F_{k+2}=F_{k+3},
2F_2=F_3,
2F_3=F_2+F_4,
2F_{k+1}=F_{k-1}+F_{k+2}
\),
因此均保持 $\mathrm{Val}$ 不变。

\begin{proposition}[终止性与唯一正规形]\label{prop:directed-rewrite-termination}
上述重写系统对任意 $a\in\mathcal{D}$ 强终止，并且不可约项恰为 $X_\infty$；因此任意归约序列终止于唯一正规形，且该正规形等于 $\Fold_\infty(a)$。
\end{proposition}

\textbf{证明：}
设
\[
A(a)=\sum_{k\ge 1} a_k,\qquad
B(a)=\sum_{k\ge 1} k\,a_k,\qquad
C(a)=a_2,
\qquad
\mu(a)=(A(a),B(a),C(a))
\]
按字典序比较。则规则 (\ref{rule:fold:adj}) 与 (\ref{rule:fold:dedup1}) 使 $A$ 减少 1；规则 (\ref{rule:fold:dedupk})（$k\ge 3$）使 $B$ 减少 1；规则 (\ref{rule:fold:dedup2}) 保持 $A,B$ 不变但使 $C$ 减少 2。故每步重写都严格降低 $\mu$，系统终止。不可约项必满足 $a_k\le 1$ 且无相邻 $11$，即属于 $X_\infty$；反之，当 $a\in X_\infty$ 时，四条规则 (\ref{rule:fold:adj})--(\ref{rule:fold:dedupk}) 的触发条件均不满足，因此不可约。又因每步重写保持 $\mathrm{Val}$，终止形与初态同值；由定理 \ref{thm:zeckendorf-transversal} 的横截面唯一性知正规形唯一且为 $\Fold_\infty(a)$。证毕。

\leanverified{step\_confluent}
\leanverified{paper\_directed\_rewrite\_termination}

\begin{remark}[与 forcing 语义的接口]\label{rem:directed-rewrite-forcing-interface}
定向化归一化重写并不创造新的对象层事实。
按推论 \ref{cor:observer-indexed-reading-of-emergent-arithmetic} 与命题 \ref{prop:observer-indexed-value-preserving-no-creation}，它只能作用于那些已经完成地址对象化、已经通过局部截面判据并已获得非 \(\mathsf{NULL}\) 读出的值类。
因此，\(\Fold_\infty\) 的编译表述不是把未定义字符串强行送入算术层，而是在既有对象层值类内部选取唯一规范代表。
\end{remark}

\begin{remark}[残差的两种自然落点]\label{rem:residual}
折叠 $\Fold_\infty$ 是值纤维到横截面的投影；因此“残差”可视为从 $a$ 到 $\Fold_\infty(a)$ 的群胚态射（重写路径/轨道坐标）。在有限分辨率 $m$ 下，则对应附录 D 中的折叠多重度 $d_m$：大量微态压缩到同一稳定类型，丢失的信息即是这些预像的区分标签。
\end{remark}

\subsubsection{折叠等价关系群胚与碰撞矩的规范对称}\label{subsubsec:groupoid-fold-collision-gauge}
上一节将 Zeckendorf 规范形解释为“值保持群胚的轨道横截面”。在固定分辨率 \(m\) 下，折叠投影
\(\Fold_m:\Omega_m=\{0,1\}^m\twoheadrightarrow X_m\)
（定义 \ref{def:fold-word}）
还诱导一个更直接的等价关系群胚：两微态等价当且仅当它们在稳定类型层不可区分。该群胚的有限维 \(\ast\)-代数实现把碰撞矩 \(S_q(m)=\sum_x d_m(x)^q\) 与“离散/连续规范对称”的维数与挠性严格对齐。

\begin{definition}[折叠等价关系群胚与群胚代数]\label{def:fold-equivalence-groupoid-algebra}
固定分辨率 \(m\ge 1\)。定义等价关系群胚
\[
\mathcal{R}_m:=\Omega_m\times_{X_m}\Omega_m
:=\{(\omega,\omega')\in\Omega_m\times\Omega_m:\ \Fold_m(\omega)=\Fold_m(\omega')\}.
\]
其复群胚代数 \(\CC[\mathcal{R}_m]\) 取为由 \(\mathcal{R}_m\) 的箭头生成的有限维 \(\ast\)-代数（卷积乘法与反向取逆为 \(\ast\)）。
\end{definition}

\begin{definition}[推前分布、碰撞矩与 \texorpdfstring{$q$}{q}-Rényi 偏离]\label{def:fold-pushforward-renyi-collision-moment}
在均匀微态基线 \(A\sim\mathrm{Unif}(\Omega_m)\) 下，记折叠推前分布
\[
\pi_m(x):=\PP\bigl(\Fold_m(A)=x\bigr)=\frac{d_m(x)}{2^m}\qquad(x\in X_m),
\]
并令 \(u_m\) 为 \(X_m\) 上的均匀分布 \(u_m(x)=|X_m|^{-1}\)。
对任意实数 \(q>0\)，定义 \(q\)-碰撞矩（\(q\)-阶多重度矩）
\[
S_q(m):=\sum_{x\in X_m} d_m(x)^q.
\]
当 \(q>1\) 时，记 \(q\)-Rényi 散度
\(
D_q(\pi_m\Vert u_m)
\)
为标准定义
\(
D_q(\pi\Vert u)=\frac{1}{q-1}\log\sum_x \pi(x)^q u(x)^{1-q}
\)。
\end{definition}

\begin{proposition}[碰撞矩与 Rényi 偏离的严格恒等式]\label{prop:fold-renyi-collision-identity}
在定义 \ref{def:fold-pushforward-renyi-collision-moment} 的记号下，对任意 \(q>1\) 有严格恒等式
\begin{equation}\label{eq:fold-renyi-collision-identity}
D_q(\pi_m\Vert u_m)
=\frac{1}{q-1}\Bigl(\log S_q(m)-q\,m\log 2+(q-1)\log|X_m|\Bigr).
\end{equation}
\leanverified{paper\_pom\_collision\_moment\_package}
\end{proposition}
\begin{proof}
由 \(\pi_m(x)=d_m(x)/2^m\)、\(u_m(x)=|X_m|^{-1}\)，有
\[
\sum_{x\in X_m}\pi_m(x)^q\,u_m(x)^{1-q}
=\sum_{x\in X_m}\Bigl(\frac{d_m(x)}{2^m}\Bigr)^q\,|X_m|^{q-1}
=|X_m|^{q-1}2^{-mq}S_q(m).
\]
代入 Rényi 散度定义即得 \eqref{eq:fold-renyi-collision-identity}。证毕。
\end{proof}

\begin{proposition}[大纤维计数的纯矩上界]\label{prop:fold-large-fiber-moment-upperbounds}
沿定义 \ref{def:fold-pushforward-renyi-collision-moment}，对阈值 \(D\ge 1\) 定义大纤维集合
\[
\mathcal L_m(D):=\{x\in X_m:\ d_m(x)\ge D\}.
\]
则对任意 \(q>0\) 有确定性不等式
\begin{equation}\label{eq:fold-large-fiber-count-upperbound}
|\mathcal L_m(D)|\ \le\ \frac{S_q(m)}{D^q}.
\end{equation}

\pi_m(\mathcal L_m(D))
=\sum_{x\in\mathcal L_m(D)}\frac{d_m(x)}{2^m}
\ \le\ \frac{S_q(m)}{2^m\,D^{q-1}}.
\end{equation}
\end{proposition}

\[

\le 2^{-m}\sum_{x\in\mathcal L_m(D)}\frac{d_m(x)^q}{D^{q-1}}
\le \frac{S_q(m)}{2^mD^{q-1}},
\]
即 \eqref{eq:fold-large-fiber-mass-upperbound}。证毕。
