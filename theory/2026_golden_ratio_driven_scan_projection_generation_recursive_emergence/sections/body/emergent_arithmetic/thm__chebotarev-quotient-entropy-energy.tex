\begin{theorem}[Chebotarev 分层在群商下的相对熵链式分解]\label{thm:kernel-chebotarev-quotient-relative-entropy-chain}
设 \(\pi:G_2\twoheadrightarrow G_1\) 为有限阿贝尔群满同态，并记 \(K:=\ker\pi\)。
固定温度参数 \(u\in(0,1]\) 与长度 \(n\ge 1\)，并假设总 primitive 计数
\[
B_n(\mathbf{1};u):=
\sum_{c_2\in G_2}B^{(2)}_{n,c_2}(u)
=
\sum_{c_1\in G_1}B^{(1)}_{n,c_1}(u)
\]
严格为正，其中粗层计数由细层推前给出
\[
B^{(1)}_{n,c_1}(u):=
\sum_{c_2\in\pi^{-1}(c_1)}B^{(2)}_{n,c_2}(u)
\qquad(c_1\in G_1).
\]
定义概率分布
\[
\mu_n^{(i)}(c):=\frac{B^{(i)}_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(i=1,2),
\]
并记 \(G_i\) 上的均匀分布为 \(\mathrm{unif}_{G_i}\)。
对任意满足 \(\mu_n^{(1)}(c_1)>0\) 的 \(c_1\in G_1\)，定义纤维条件分布
\[
\mu_{n,\mathrm{fib}}^{(c_1)}(c_2):=
\frac{\mu_n^{(2)}(c_2)}{\mu_n^{(1)}(c_1)}
\qquad(c_2\in\pi^{-1}(c_1)),
\]
并记该纤维上的均匀分布为 \(\mathrm{unif}_{\pi^{-1}(c_1)}\)。
则有严格的相对熵链式恒等式
\[
\boxed{\;
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr)\; }.
\]
其中按约定，\(\mu_n^{(1)}(c_1)=0\) 的纤维项贡献 \(0\)。
\end{theorem}

\begin{proof}
由定义可知 \(\mu_n^{(1)}=\pi_\ast\mu_n^{(2)}\)。
又由于 \(\pi\) 为有限群满同态，每个纤维 \(\pi^{-1}(c_1)\) 的大小都等于 \(|K|\)，故对任意 \(c_2\in G_2\) 有
\[
\mathrm{unif}_{G_2}(c_2)
=
\frac{1}{|G_2|}
=
\frac{1}{|G_1|\,|K|}
=
\mathrm{unif}_{G_1}(\pi(c_2))\cdot \mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2).
\]
当 \(\mu_n^{(1)}(\pi(c_2))>0\) 时，进一步有分解
\[
\mu_n^{(2)}(c_2)
=
\mu_n^{(1)}(\pi(c_2))\cdot
\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2).
\]
因此
\[
\log\frac{\mu_n^{(2)}(c_2)}{\mathrm{unif}_{G_2}(c_2)}
=
\log\frac{\mu_n^{(1)}(\pi(c_2))}{\mathrm{unif}_{G_1}(\pi(c_2))}
+
\log\frac{\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2)}{\mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2)}.
\]
对 \(c_2\in G_2\) 以 \(\mu_n^{(2)}(c_2)\) 加权求和，并按纤维分组，即得
\[
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr).
\]
证毕。
\end{proof}

\begin{theorem}[可见角色能量与隐藏角色能量的正交分解]\label{thm:kernel-chebotarev-visible-hidden-character-energy}
\leanverified{paper\_kernel\_chebotarev\_visible\_hidden\_character\_energy}
在定理 \ref{thm:kernel-chebotarev-quotient-relative-entropy-chain} 的设定下，定义相对均匀密度
\[
r_n^{(i)}(c):=|G_i|\,\mu_n^{(i)}(c)
=
\frac{|G_i|\,B^{(i)}_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(i=1,2).
\]
则有精确正交分解
\[
\boxed{\;
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
+
\frac{1}{|G_2|}\sum_{c_1\in G_1}\sum_{c_2\in\pi^{-1}(c_1)}
\bigl(r_n^{(2)}(c_2)-r_n^{(1)}(c_1)\bigr)^2\; }.
\]
进一步，令
\[
K^\perp:=\{\chi\in\widehat G_2:\ \chi|_K\equiv 1\}\cong \widehat G_1,
\]
并定义细层扭曲 primitive 谱
\[
B_n^{(2)}(\chi;u):=\sum_{c_2\in G_2}\chi(c_2)\,B^{(2)}_{n,c_2}(u)
\qquad(\chi\in\widehat G_2),
\]
则有 Fourier/Plancherel 形式
\[
\boxed{\;
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2\; },
\]
\[
\boxed{\;
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in K^\perp\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2\; }.
\]
因此，商映射所删除的恰是 kernel 上非平凡的隐藏角色能量：
\[
\boxed{\;
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus K^\perp}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2\; }.
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:kernel-chebotarev-L2-monotone} 的证明，
\[
r_n^{(1)}(c_1)
=
\frac{1}{|K|}\sum_{c_2\in\pi^{-1}(c_1)}r_n^{(2)}(c_2),
\]
因此 \(r_n^{(1)}\circ\pi\) 正是 \(r_n^{(2)}\) 在 \(L^2(\mathrm{unif}_{G_2})\) 中相对于纤维常函数子空间的条件期望，也即正交投影。由 Hilbert 空间的 Pythagoras 恒等式，
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}\circ\pi-1\|_{L^2(\mathrm{unif}_{G_2})}^2
+
\|r_n^{(2)}-r_n^{(1)}\circ\pi\|_{L^2(\mathrm{unif}_{G_2})}^2.
\]
第一项与 \(\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2\) 相同；第二项按纤维展开即得所示正交分解。
\par
再证 Fourier 形式。对 \(G_2\) 上角色基应用 Parseval 恒等式，并取 Fourier 系数
\[
\widehat r_n^{(2)}(\chi)
:=
\frac{1}{|G_2|}\sum_{c_2\in G_2}r_n^{(2)}(c_2)\,\overline{\chi(c_2)}
=
\frac{B_n^{(2)}(\overline\chi;u)}{B_n(\mathbf{1};u)}.
\]
由于取绝对值平方时 \(\chi\) 与 \(\overline\chi\) 无区别，故
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}|\widehat r_n^{(2)}(\chi)|^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
另一方面，\(G_1\) 的角色经拉回正好识别为 \(K^\perp\subseteq \widehat G_2\)，故对粗层密度同理得到
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in K^\perp\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
二式相减即得隐藏角色能量的显式表达。证毕。
\end{proof}
