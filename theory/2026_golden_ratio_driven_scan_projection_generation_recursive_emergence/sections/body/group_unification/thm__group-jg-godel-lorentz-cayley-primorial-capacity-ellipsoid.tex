给出（见定理 \ref{thm:group-jg-elliptic-two-scale-stokes-mean} 的记号）；

\begin{theorem}[Gödel--Lorentz 代数化：Joukowsky 椭圆的 Minkowski 不变量与合成律]\label{thm:group-jg-godel-lorentz-algebraization}
\leanverified{paper\_group\_jg\_godel\_lorentz\_algebraization}
沿用定理 \ref{thm:group-jg-ellipse-prime-homomorphism} 的素数寄存器 \(\mathsf P=\NN^{(\mathbb P)}\) 与范数 \(N:\mathsf P\to\NN_{>0}\)。
对 \(r\in\mathsf P\) 且 \(N(r)>1\)，令
\[
L:=N(r),\qquad r_{\mathrm{geo}}:=\sqrt{L}>1,\qquad
\Delta(r):=\log r_{\mathrm{geo}}=\frac12\log N(r).
\]
记 Joukowsky 椭圆
\[
E_J(r):=\{\,w=r_{\mathrm{geo}} z+r_{\mathrm{geo}}^{-1}z^{-1}:|z|=1\,\}\subset\CC,
\]
并记其长短半轴
\[
a(r):=r_{\mathrm{geo}}+r_{\mathrm{geo}}^{-1},\qquad b(r):=r_{\mathrm{geo}}-r_{\mathrm{geo}}^{-1}.
\]
定义
\[
\Lambda(r):=
\begin{pmatrix}
\frac{a(r)}{2} & \frac{b(r)}{2}\\[2pt]
\frac{b(r)}{2} & \frac{a(r)}{2}
\end{pmatrix}\in\GL_2(\RR).
\]
则：
\begin{enumerate}
  \item \(a(r)^2-b(r)^2=4\)，且 \(\det\Lambda(r)=1\)。因此 \((a(r),b(r))\) 落在双曲线 \(x^2-y^2=4\) 上，等价地，\(\{E_J(r)\}\) 为一族共焦椭圆（焦距恒为 \(2\)）。
  \item 令 \(\Delta=\Delta(r)\)，则
  \[
  a(r)=2\cosh\Delta,\qquad b(r)=2\sinh\Delta,
  \]
  从而 \(\Lambda(r)\in\mathrm{SO}^+(1,1)\subset \mathrm{SL}_2(\RR)\)，并且 \(\Lambda(r)\) 的特征值为 \(e^{\pm\Delta}\)。
  \item 映射 \(\Lambda:\mathsf P\to \mathrm{SO}^+(1,1)\) 为交换幺半群同态：
  \[
  \Lambda(r_1+r_2)=\Lambda(r_1)\Lambda(r_2).
  \]
  等价地，若 \((a_i,b_i)=(a(r_i),b(r_i))\)，则
  \[
  a(r_1+r_2)=\frac{a_1a_2+b_1b_2}{2},\qquad
  b(r_1+r_2)=\frac{a_1b_2+b_1a_2}{2}.
  \]
  \item 谱半径满足
  \[
  \log \rho\bigl(\Lambda(r)\bigr)=\Delta(r)=\frac12\log N(r).
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
由 \(a=r_{\mathrm{geo}}+r_{\mathrm{geo}}^{-1}\)、\(b=r_{\mathrm{geo}}-r_{\mathrm{geo}}^{-1}\) 直接计算得
\(
a^2-b^2=4
\)；
并令 \(\Delta=\log r_{\mathrm{geo}}\) 得 \(r_{\mathrm{geo}}=e^{\Delta}\)，故
\(
a=2\cosh\Delta,\ b=2\sinh\Delta
\)。
于是
\(
\det\Lambda=\cosh^2\Delta-\sinh^2\Delta=1
\)，且 \(\Lambda\) 的特征值为 \(\cosh\Delta\pm\sinh\Delta=e^{\pm\Delta}\)。
\par
由 \(N(r_1+r_2)=N(r_1)N(r_2)\) 得 \(\Delta(r_1+r_2)=\Delta(r_1)+\Delta(r_2)\)，故
\[
\Lambda(r_1+r_2)
=
\begin{pmatrix}\cosh(\Delta_1+\Delta_2)&\sinh(\Delta_1+\Delta_2)\\ \sinh(\Delta_1+\Delta_2)&\cosh(\Delta_1+\Delta_2)\end{pmatrix}
=\Lambda(r_1)\Lambda(r_2),
\]
并由双曲函数加法公式得到半轴的双线性合成式。
最后 \(\rho(\Lambda)=e^{\Delta}\) 给出谱半径恒等式。证毕。
\end{proof}

\begin{theorem}[面积守恒规范化的 Cayley 共轭与 Einstein 加法]\label{thm:group-jg-joukowsky-area-preserving-cayley}
\leanverified{paper\_group\_jg\_joukowsky\_area\_preserving\_cayley}
沿用定理 \ref{thm:group-jg-godel-lorentz-algebraization} 的记号，并记
\[
E(a):=\{(x,y)\in\RR^2:\ x^2/a^2+y^2a^2=1\}\qquad(a>0)
\]
为面积守恒椭圆族（定理 \ref{thm:group-jg-ellipse-prime-homomorphism}）。
对 \(r\in\mathsf P\) 定义面积守恒规范化
\[
\widehat E_J(r):=(a(r)b(r))^{-1/2}\cdot E_J(r),
\]
即在平面坐标上施以相似缩放因子 \((a(r)b(r))^{-1/2}\)。
则：
\begin{enumerate}
  \item \(\widehat E_J(r)\) 属于 \(E(a)\) 族，且其参数可写为
  \[
  \alpha(r):=\sqrt{\frac{a(r)}{b(r)}}
  =\sqrt{\frac{L+1}{L-1}}
  =\sqrt{\coth(\Delta(r))}\in(1,\infty).
  \]
  \item 令 \(u(r):=\alpha(r)^2=\coth(\Delta(r))\)，则存在交换结合律
  \[
  u(r_1+r_2)=u(r_1)\odot u(r_2),
  \qquad
  u\odot v:=\frac{uv+1}{u+v}.
  \]
  等价地，令 \(t(r):=u(r)^{-1}=\tanh(\Delta(r))\in(0,1)\)，则满足 Einstein 加法
  \[
  t(r_1+r_2)=\frac{t(r_1)+t(r_2)}{1+t(r_1)t(r_2)}.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
由定义 \(\widehat E_J\) 的半轴为
\[
\hat a=\frac{a}{\sqrt{ab}}=\sqrt{\frac{a}{b}},\qquad
\hat b=\frac{b}{\sqrt{ab}}=\sqrt{\frac{b}{a}}=\hat a^{-1},
\]
故 \(\widehat E_J(r)\) 具有方程 \(x^2/\hat a^2+y^2\hat a^2=1\)，即属于 \(E(a)\) 族且参数为 \(\alpha=\hat a=\sqrt{a/b}\)。
代入 \(a=r_{\mathrm{geo}}+r_{\mathrm{geo}}^{-1}\)、\(b=r_{\mathrm{geo}}-r_{\mathrm{geo}}^{-1}\) 得
\(
a/b=(L+1)/(L-1)
\)，再由 \(r_{\mathrm{geo}}=e^{\Delta}\) 得 \((L+1)/(L-1)=\coth\Delta\)，从而 \(\alpha^2=\coth\Delta\)。
最后由双曲恒等式
\(
\coth(\Delta_1+\Delta_2)=\frac{\coth\Delta_1\coth\Delta_2+1}{\coth\Delta_1+\coth\Delta_2}
\)
与
\(
\tanh(\Delta_1+\Delta_2)=\frac{\tanh\Delta_1+\tanh\Delta_2}{1+\tanh\Delta_1\tanh\Delta_2}
\)
并结合 \(\Delta(r_1+r_2)=\Delta(r_1)+\Delta(r_2)\) 得结论。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$p$}{p}-元 Gödel 外置的 primorial 下界与椭圆资源下界]\label{prop:group-jg-boundary-primorial-ellipse-bound}
\leanverified{paper\_group\_jg\_boundary\_primorial\_ellipse\_bound}
设 \(p\) 为素数且 \(b\ge 1\)。
取互异素数 \(q_1,\dots,q_b\)，并定义 \((\ZZ/p\ZZ)^b\) 的标量 Gödel 外置
\[
g(\varepsilon_1,\dots,\varepsilon_b):=\prod_{i=1}^b q_i^{\varepsilon_i}\qquad(\varepsilon_i\in\{0,1,\dots,p-1\}).
\]
记最坏情形范数
\[
L_{\max}:=\max_{\varepsilon} g(\varepsilon)=\prod_{i=1}^b q_i^{p-1},
\]
并记 primorial \(p_b\#:=\prod_{i=1}^b p_i\)（\(p_i\) 为第 \(i\) 个素数）。
则：
\begin{enumerate}
  \item \(L_{\max}\ge (p_b\#)^{p-1}\)，且当且仅当 \(\{q_i\}_{i=1}^b=\{p_1,\dots,p_b\}\)（忽略排列）时取等号。
  \item 对应的 Koenigs 位移 \(\Delta_{\max}:=\frac12\log L_{\max}\) 满足
  \[
  \Delta_{\max}\ge \frac{p-1}{2}\log(p_b\#).
  \]
  \item 相应的 Joukowsky 椭圆 \(E_J(L_{\max})\) 的半轴与面积满足
  \[
  a_{\max}\ge (p_b\#)^{\frac{p-1}{2}}+(p_b\#)^{\frac{1-p}{2}},\qquad
  b_{\max}\ge (p_b\#)^{\frac{p-1}{2}}-(p_b\#)^{\frac{1-p}{2}},
  \]
  \[
  \mathrm{Area}\bigl(E_J(L_{\max})\bigr)=\pi\,\bigl(L_{\max}-L_{\max}^{-1}\bigr)\ge \pi\,\bigl((p_b\#)^{p-1}-(p_b\#)^{1-p}\bigr).
  \]
  并且偏心率 \(e=2/a\) 满足
  \[
  e_{\max}\le \frac{2\,(p_b\#)^{\frac{p-1}{2}}}{(p_b\#)^{p-1}+1}.
  \]
  \item 面积守恒规范化参数 \(\alpha_{\max}=\sqrt{(L_{\max}+1)/(L_{\max}-1)}\) 满足
  \[
  \alpha_{\max}\le \sqrt{\frac{(p_b\#)^{p-1}+1}{(p_b\#)^{p-1}-1}},
  \]
  因而当 \(b\to\infty\) 时其上界单调趋于 \(1\)。
\end{enumerate}
\end{proposition}

\begin{proof}
将 \(\{q_i\}\) 递增排列为 \(q_{(1)}\le\cdots\le q_{(b)}\)。由素数序性质 \(q_{(i)}\ge p_i\)，故
\[
L_{\max}=\prod_i q_{(i)}^{p-1}\ge \prod_i p_i^{p-1}=(p_b\#)^{p-1},
\]
等号当且仅当逐项相等。
其余结论由 \(\Delta=\tfrac12\log L\)、\(a(L)=\sqrt L+1/\sqrt L\)、\(b(L)=\sqrt L-1/\sqrt L\) 以及 \(\mathrm{Area}=\pi a b=\pi(L-L^{-1})\) 的单调性直接推出。
偏心率恒为 \(e=2/a\)（焦距为 \(2\)），故 \(L_{\max}\ge (p_b\#)^{p-1}\Rightarrow e_{\max}\le e\bigl((p_b\#)^{p-1}\bigr)\)。
对 \(\alpha^2=(L+1)/(L-1)\) 的递减性同理。证毕。
\end{proof}

\begin{theorem}[高维椭球函子：Gödel 范数的体积对数坐标]\label{thm:group-jg-godel-ellipsoid-volume}
设 \(r\in\mathsf P\) 的支撑为互异素数集合 \(\{p_1,\dots,p_d\}\)，并记 \(r_i:=r(p_i)\)。
定义对角椭球
\[
\mathcal E_d(r):=\left\{x\in\RR^d:\ \sum_{i=1}^d \frac{x_i^2}{p_i^{r_i}}\le 1\right\}.
\]
则体积满足显式等式
\[
\mathrm{Vol}\bigl(\mathcal E_d(r)\bigr)
=\mathrm{Vol}(B_d)\prod_{i=1}^d p_i^{r_i/2}
=\mathrm{Vol}(B_d)\,\sqrt{N(r)}.
\]
从而
\[
\log\mathrm{Vol}\bigl(\mathcal E_d(r)\bigr)-\log\mathrm{Vol}(B_d)=\frac12\log N(r)=\Delta(r).
\]
\leanverified{paper\_group\_jg\_godel\_ellipsoid\_volume}
\end{theorem}

\begin{proof}
\(\mathcal E_d(r)\) 为单位球 \(B_d\) 在对角线性变换 \(x_i\mapsto p_i^{r_i/2}x_i\) 下的像，体积按行列式缩放：
\(
\mathrm{Vol}(T(B_d))=|\det T|\mathrm{Vol}(B_d)
\)。
此处 \(|\det T|=\prod_i p_i^{r_i/2}=\sqrt{N(r)}\)。证毕。
\end{proof}

\begin{corollary}[体积差与椭圆面积的解析同一化]\label{cor:group-jg-ellipse-area-ellipsoid-gap}

并在 \(\Delta(r)\to\infty\) 时有渐近 \(\log\mathrm{Area}(E_J(r))=2\Delta(r)+\log\pi+o(1)\)。
\leanverified{paper\_group\_jg\_ellipse\_area\_ellipsoid\_gap}

\mathrm{Area}\bigl(E_J(r)\bigr)=\pi a(r)b(r)=\pi(r_{\mathrm{geo}}^2-r_{\mathrm{geo}}^{-2})

\begin{theorem}[Gödel 可分解码本的 primorial 极小化与椭球--椭圆三重最优性]\label{prop:group-jg-min-total-godel-cost-primorial}
\leanverified{paper\_group\_jg\_min\_total\_godel\_cost\_primorial}
令 \(X\) 为有限集，\(|X|=n\ge 1\)。
称注入
\[
\iota:X\hookrightarrow \NN_{\mathrm{sf},>1}
\]
为\emph{Gödel 可分解}，若其像满足两两互素条件
\[
\gcd(\iota(x),\iota(y))=1\qquad (x\neq y).
\]
定义总 Gödel 成本
\[
C(\iota):=\sum_{x\in X}\log \iota(x)
=\log\!\Bigl(\prod_{x\in X}\iota(x)\Bigr).
\]
则成立：
\begin{enumerate}
  \item \textbf{总 Gödel 成本的极小性.}
        在全体 Gödel 可分解码本中，
        \[
        \min_{\iota} C(\iota)=\log(p_n\#),
        \]
        且极小码本恰由 \(\iota(X)=\{p_1,\dots,p_n\}\)（忽略排列）给出。
  \item \textbf{椭球体积乘积的极小性.}
        取有限素数集 \(P_0\) 包含 \(\iota(X)\) 中出现的全部素因子，并令 \(D:=|P_0|\)。
        对每个 \(x\in X\)，定义寄存器 \(r_x\in\NN^{(P_0)}\) 使 \(N(r_x)=\iota(x)\)。
        则
        \[
        \prod_{x\in X}\mathrm{Vol}\bigl(\mathcal E_D(r_x)\bigr)
        =
        \mathrm{Vol}(B_D)^n\sqrt{\prod_{x\in X}\iota(x)},
        \]
        因而其极小化与 \(\prod_{x\in X}\iota(x)\) 的极小化等价，并由 \(\{p_1,\dots,p_n\}\) 同时达到。
  \item \textbf{Joukowsky 椭圆面积乘积的极小性.}
        在同一设定下，
        \[
        \prod_{x\in X}\mathrm{Area}\bigl(E_J(r_x)\bigr)
        =
        \pi^n\prod_{x\in X}\bigl(\iota(x)-\iota(x)^{-1}\bigr),
        \]
        且该乘积在 Gödel 可分解约束下同样由 \(\iota(X)=\{p_1,\dots,p_n\}\) 取得极小值。
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(1)}
对每个 \(x\in X\) 任选一个素因子 \(q_x\mid \iota(x)\)。
由于 \(\iota\) 的像两两互素，\(\{q_x\}_{x\in X}\) 必为互异素数族。
从而
\[
\prod_{x\in X}\iota(x)\ \ge\ \prod_{x\in X}q_x.
\]
将 \(\{q_x\}\) 递增排列为 \(q_{(1)}<\cdots<q_{(n)}\)。
由素数序的极小性 \(q_{(i)}\ge p_i\) 得
\[
\prod_{x\in X}\iota(x)\ \ge\ \prod_{i=1}^{n}q_{(i)}
\ \ge\ \prod_{i=1}^{n}p_i
=p_n\#.
\]
两边取对数得到 \(C(\iota)\ge \log(p_n\#)\)。
当 \(\iota(X)=\{p_1,\dots,p_n\}\) 时取等。
若出现等号，则上式中每一步都必须取等，故 \(q_{(i)}=p_i\) 且 \(\iota(x)=q_x\) 对每个 \(x\) 成立，于是极小像集恰为 \(\{p_1,\dots,p_n\}\)。

\textbf{(2)}
由定理 \ref{thm:group-jg-godel-ellipsoid-volume}，
\[
\mathrm{Vol}\bigl(\mathcal E_D(r_x)\bigr)=\mathrm{Vol}(B_D)\sqrt{N(r_x)}
=\mathrm{Vol}(B_D)\sqrt{\iota(x)}.
\]
对 \(x\in X\) 取乘积即得所示闭式。
因此体积乘积的极小化与 \(\prod_x\iota(x)\) 的极小化等价，故由第 (1) 条得到同一极小码本。

\textbf{(3)}
由推论 \ref{cor:group-jg-ellipse-area-ellipsoid-gap}，
\[
\mathrm{Area}\bigl(E_J(r_x)\bigr)=\pi\bigl(\iota(x)-\iota(x)^{-1}\bigr).
\]
令 \(h(t):=t-t^{-1}\)（\(t>1\)），则 \(h'(t)=1+t^{-2}>0\)，故 \(h\) 严格递增。
沿第 (1) 条的记号，对每个 \(x\) 有 \(\iota(x)\ge q_x\)，从而
\[
\prod_{x\in X}\bigl(\iota(x)-\iota(x)^{-1}\bigr)
\ge
\prod_{x\in X}\bigl(q_x-q_x^{-1}\bigr)
\ge
\prod_{i=1}^{n}\bigl(p_i-p_i^{-1}\bigr).
\]
故面积乘积的极小值同样由 \(\iota(X)=\{p_1,\dots,p_n\}\) 取得。证毕。
\end{proof}

\begin{remark}[互素约束的必要性]\label{rem:group-jg-min-total-godel-cost-coprime-necessity}
primorial 极小化是 Gödel 可分解类的结构性结论，而不是全体平方自由注入类的结论。
若只要求平方自由且互异，则可出现共享素因子的像集；例如当 \(n=4\) 时，
\[
\{2,3,5,6\}\subset \NN_{\mathrm{sf},>1}
\]
满足互异与平方自由，但其乘积 \(180\) 严格小于 \(p_4\#=210\)。
因此“两两互素”正是乘法可分解语义所需的约束。
\end{remark}
