\subsubsection{Fibonacci 边界计数的平方恒等式与 NAP 的解析极小性}\label{subsubsec:bdry-fib-square-identities-nap-minimality}

令 \(X_m^{\mathrm{bdry}}:=\{u\in X_m:\ u_1=u_m=1\}\) 且 \(b(m):=|X_m^{\mathrm{bdry}}|\)。
在本文的黄金均值计数律下有 \(b(m)=F_{m-2}\)（见 \S\ref{subsec:bdry-tower-zeck-gut-part1}），其中 \((F_n)\) 为 Fibonacci 数列。

\begin{proposition}[平方恒等式：奇指标边界层的二平方分解]\label{prop:bdry-fib-square-identity}
对任意整数 \(m\ge 2\) 恒有
\[
b(2m-1)=b(m)^2+b(m+1)^2.
\]
\leanverified{cBoundaryCount\_nine}
\leanverified{cBoundaryCount\_ten}
\leanverified{cBoundaryCount\_eq\_fib\_extended}
\leanverified{cBoundaryCount\_square\_identity}
\leanverified{cBoundaryCount\_eleven}
\leanverified{cBoundaryCount\_twelve}
\leanverified{cBoundaryCount\_square\_identity\_general}
\leanverified{cBoundaryCount\_eq\_fib\_general}
\leanverified{cBoundaryCount\_square\_identity\_general'}
\leanverified{cBoundaryCount\_eq\_fib\_extended\_twelve}
\leanverified{cBoundaryCount\_square\_identity\_m4}
\leanverified{cBoundaryCount\_square\_identity\_m5}
\leanverified{cBoundaryCount\_square\_identity\_m6}
\leanverified{paper\_bdry\_fib\_square\_identity}
\leanverified{paper\_cBoundaryCount\_extended}
\end{proposition}
\begin{proof}
由 \(b(m)=F_{m-2}\)，令 \(n:=m-2\)。
则 \(b(2m-1)=F_{2m-3}=F_{2n+1}\)、\(b(m)=F_n\)、\(b(m+1)=F_{n+1}\)。
而 Fibonacci 的标准恒等式给出
\(
F_{2n+1}=F_n^2+F_{n+1}^2
\)，代回即得结论。
证毕。
\leanverified{cBoundaryCount\_recurrence}
\leanverified{cBoundaryCount\_square\_identity\_seeds}
\leanverified{paper\_boundary\_layer\_extended}
\end{proof}

\begin{corollary}[数值实例：\(m=6\) 的边界层]\label{cor:bdry-m6-square-instance}

|X_{11}^{\mathrm{bdry}}|=|X_{6}^{\mathrm{bdry}}|^2+|X_{7}^{\mathrm{bdry}}|^2=3^2+5^2=34.

\begin{proposition}[替换层差的 Cassini 型因式分解]\label{prop:bdry-gap-33-cassini-factorization}
令 \(\Sigma_{\mathrm{SM}}=\{4,6,8\}\) 且 \(\Sigma_{SO(10)}=\{6,8,11\}\) 为本文的 Zeckendorf--GUT 分辨率签名（\S\ref{subsec:bdry-tower-zeck-gut-part1}）。
则有严格恒等式
\[
|X_{11}^{\mathrm{bdry}}|-|X_{4}^{\mathrm{bdry}}|
=
|X_{6}^{\mathrm{bdry}}|\Bigl(|X_{8}^{\mathrm{bdry}}|+|X_{6}^{\mathrm{bdry}}|\Bigr)
=33.
\]
\leanverified{cassini\_gap\_factorization\_instances}
\leanverified{paper\_cassini\_gap\_factorization\_extended}
\leanverified{paper\_bdry\_gap\_33\_cassini\_factorization}
\end{proposition}
\begin{proof}
由 \(b(m)=F_{m-2}\) 得 \(|X_{11}^{\mathrm{bdry}}|-|X_{4}^{\mathrm{bdry}}|=F_9-F_2=34-1=33\)。
另一方面，将 Cassini 恒等式 \(F_{n+1}^2-F_nF_{n+2}=(-1)^n\) 与平方恒等式 \(F_{2n+1}=F_n^2+F_{n+1}^2\) 联立可得：当 \(n\) 为偶数时，
\[
F_{2n+1}-1=F_n(F_{n+2}+F_n).
\]
取 \(n=4\) 得 \(F_9-1=F_4(F_6+F_4)=3(8+3)=33\)，再以 \(F_4=b(6)\)、\(F_6=b(8)\) 改写即得所示分解。
证毕。
\leanverified{cassini\_gap\_33\_factorization}
\end{proof}

\begin{theorem}[NAP 约束下的解析极小性]\label{thm:nap-so10-analytic-minimality}
令 \(\Sigma(\mathfrak g)\) 为本文定义的 Zeckendorf--GUT 分辨率签名（\S\ref{subsec:bdry-tower-zeck-gut-part1}），并施加约束
\[
\{6,8\}\subseteq \Sigma(\mathfrak g).
\]
则对任意复简单李代数 \(\mathfrak g\) 有 \(\dim\mathfrak g\ge 45\)，且等号成立当且仅当 \(\mathfrak g\cong \mathfrak{so}(10)\)（类型 \(D_5\)）。
在等号情形下其签名必为 \(\Sigma(\mathfrak g)=\{6,8,11\}\)。
\end{theorem}
\begin{proof}
\(\{6,8\}\subseteq\Sigma(\mathfrak g)\) 等价于 \(\dim\mathfrak g\) 的 Zeckendorf 分解同时包含 \(F_4=3\) 与 \(F_6=8\) 两项。
若 \(\dim\mathfrak g<45\)，由复简单李代数分类与维数公式可知可取的维数仅为
\[
3,\ 8,\ 10,\ 14,\ 15,\ 21,\ 24,\ 28,\ 35,\ 36,
\]
分别对应 \(A_1,A_2,B_2/C_2,G_2,A_3,B_3/C_3,A_4,D_4,A_5,B_4/C_4\)。
对上述各数作 Zeckendorf 分解可逐一核对：无一同时含 \(3\) 与 \(8\) 两项。
因此约束不可能在 \(\dim<45\) 内成立，从而 \(\dim\mathfrak g\ge 45\)。
当 \(\dim\mathfrak g=45\) 时，分类表中唯一的复简单李代数为 \(D_5\)，即 \(\mathfrak g\cong\mathfrak{so}(10)\)。
并且 \(45=34+8+3=F_9+F_6+F_4\)，故 \(\Sigma(\mathfrak g)=\{11,8,6\}\)。
证毕。
\leanverified{nap\_so10\_minimality}
\leanverified{paper\_gut\_nap\_so10\_analytic\_minimality\_seeds}
\leanverified{paper\_gut\_nap\_so10\_analytic\_minimality\_package}
\leanverified{paper\_nap\_so10\_analytic\_minimality}
\end{proof}
