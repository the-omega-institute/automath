语义—算术层面上，以规范化交换幺半群外置范畴 \(\mathrm{Extcmon}(\pi,s)\) 及其在可数情形下由素数寄存器实现的 Gödel 嵌入为背景；几何层面上，以 Joukowsky--Gödel 输运多项式 \(Q_r\) 与其诱导的椭圆边界族为背景。

\begin{theorem}[Joukowsky--Gödel 输运的回拉分解恒等式]\label{thm:group-jg-pullback-factorization}
设 \(K\) 为特征 \(0\) 的域，\(P(z)\in K[z]\) 为次数 \(n\ge 1\) 的多项式，
\[
P(z)=a_n\prod_{j=1}^{n}(z-z_j),\qquad a_n\neq 0,\quad z_j\neq 0.
\]
对任意 \(r\in K^\times\)，定义二次核
\[
R_{r,w}(z):=r z^2-w z+r^{-1},
\]
并定义 Joukowsky--Gödel 输运多项式
\[
Q_r(w):=\operatorname{Res}_{z}\!\bigl(P(z),R_{r,w}(z)\bigr)\in K[r,r^{-1}][w].
\]
再定义倒易多项式
\[
P^{\vee}(u):=u^n P(1/u)\in K[u].
\]
则在 \(K[r,r^{-1}][z,z^{-1}]\) 中成立恒等式
\[
\boxed{\;
r^n z^n\,Q_r\!\left(r z+r^{-1}z^{-1}\right)=P(z)\,P^{\vee}(r^2 z).
\;}
\]
\leanverified{paper\_group\_jg\_pullback\_factorization}
\end{theorem}
\begin{proof}
由结式乘积公式（在代数闭包中计算）可得
\[
Q_r(w)=a_n^2\prod_{j=1}^{n}\bigl(r z_j^2-w z_j+r^{-1}\bigr).
\]
代入 \(w=r z+r^{-1}z^{-1}\) 并逐项乘以 \(z\)，有
\[
\begin{aligned}
z\bigl(r z_j^2-(r z+r^{-1}z^{-1})z_j+r^{-1}\bigr)
&=r z z_j(z_j-z)+r^{-1}(z-z_j)\\
&=(z-z_j)\bigl(r^{-1}-r z z_j\bigr)\\
&=r^{-1}(z-z_j)(1-r^2 z z_j).
\end{aligned}
\]
故
\[
\begin{aligned}
z^n Q_r\!\left(r z+r^{-1}z^{-1}\right)
&=a_n^2\prod_{j=1}^{n}\Bigl[r^{-1}(z-z_j)(1-r^2 z z_j)\Bigr]\\
&=a_n^2 r^{-n}\Bigl(\prod_{j=1}^{n}(z-z_j)\Bigr)\Bigl(\prod_{j=1}^{n}(1-r^2 z z_j)\Bigr).
\end{aligned}
\]
另一方面
\[
\prod_{j=1}^{n}(z-z_j)=\frac{P(z)}{a_n},\qquad
\prod_{j=1}^{n}(1-r^2 z z_j)=\frac{P^{\vee}(r^2 z)}{a_n}.
\]
代回即得
\[
z^n Q_r\!\left(r z+r^{-1}z^{-1}\right)=r^{-n}P(z)\,P^{\vee}(r^2 z),
\]
两边同乘 \(r^n\) 即证。
\end{proof}

\begin{theorem}[输运结式首项系数刚性]\label{thm:group-jg-lc-rigidity}
\leanverified{paper\_group\_jg\_lc\_rigidity}
在上述记号下，
\[
\deg_w Q_r=n,\qquad \operatorname{LC}_w(Q_r)=a_n a_0.
\]
特别地，\(\operatorname{LC}_w(Q_r)\) 与尺度参数 \(r\) 无关。
\end{theorem}
\begin{proof}
设 \(z_1,\dots,z_n\) 为 \(P\) 在代数闭包中的根（计重数）。结式的根表示给出
\[
Q_r(w)=a_n^2\prod_{j=1}^{n}\bigl(r z_j^2-w z_j+r^{-1}\bigr).
\]

\[
\operatorname{LC}_w(Q_r)
=a_n^2(-1)^n\prod_{j=1}^{n}z_j
=a_n^2(-1)^n\cdot(-1)^n\frac{a_0}{a_n}

\]
又因 \(a_0\neq 0\) 从而 \(\prod_j z_j\neq 0\)，故 \(\deg_w Q_r=n\)。
\end{proof}

\begin{theorem}[判别式平方类守恒]\label{thm:group-jg-discriminant-squareclass}
设 \(K\) 的特征不等于 \(2\)，且 \(P\) 可分离。记 \(\Delta(P)\)、\(\Delta(Q_r)\) 为对应主变量的判别式。则在平方类群
\[
K(r)^\times/(K(r)^\times)^2
\]
中有
\[
\Delta(Q_r)\equiv \Delta(P).
\]
等价地，
\[
K(r)\bigl(\sqrt{\Delta(Q_r)}\bigr)=K(r)\bigl(\sqrt{\Delta(P)}\bigr).
\]
\end{theorem}
\begin{proof}
设 \(z_1,\dots,z_n\) 为 \(P\) 的根，并记
\[
w_j:=r z_j+r^{-1}z_j^{-1}.
\]
由定理 \ref{thm:group-jg-lc-rigidity}，\(Q_r\) 的首项系数为 \(a_n a_0\)，故
\[
Q_r(w)=(a_n a_0)\prod_{j=1}^{n}(w-w_j).
\]
于是
\[
\Delta(Q_r)=(a_n a_0)^{2n-2}\prod_{i<j}(w_i-w_j)^2.
\]
另一方面
\[
w_i-w_j
=r(z_i-z_j)+r^{-1}(z_i^{-1}-z_j^{-1})
=(z_i-z_j)\Bigl(r-\frac{r^{-1}}{z_i z_j}\Bigr),
\]
从而
\[
\prod_{i<j}(w_i-w_j)^2
=\prod_{i<j}(z_i-z_j)^2
\cdot
\prod_{i<j}\Bigl(r-\frac{r^{-1}}{z_i z_j}\Bigr)^2.
\]
再用
\[
\Delta(P)=a_n^{2n-2}\prod_{i<j}(z_i-z_j)^2
\]
可得
\[
\Delta(Q_r)=a_0^{2n-2}\Delta(P)\cdot
\prod_{i<j}\Bigl(r-\frac{r^{-1}}{z_i z_j}\Bigr)^2.
\]
右侧除 \(\Delta(P)\) 外均为平方，故平方类相同。
\end{proof}

\begin{theorem}[全息系数抽取与原多项式反演]\label{thm:group-jg-holographic-coefficient-recovery}
设
\[
Q_r(r t)\in K[r,r^{-1}][t].
\]
记 \([r^k]\) 为关于 \(r\) 的 Laurent 系数抽取算子，则
\[
[r^n]\,Q_r(r t)=a_0 P(t),\qquad
[r^{-n}]\,Q_r(r t)=a_n^2.
\]
特别地，配合 \(\operatorname{LC}_w(Q_r)=a_n a_0\)，可由
\(
[r^{-n}]Q_r(r t),\ \operatorname{LC}_w(Q_r),\ [r^n]Q_r(r t)
\)
代数重构 \(P\) 的全部系数（在固定与 \(\operatorname{LC}_w(Q_r)\) 相容的平方根分支后）。
\end{theorem}
\begin{proof}
仍记 \(z_1,\dots,z_n\) 为 \(P\) 的根。由结式根表示，
\[
\begin{aligned}
Q_r(r t)
&=a_n^2\prod_{j=1}^{n}\bigl(r z_j^2-r t z_j+r^{-1}\bigr)\\
&=a_n^2 r^n\prod_{j=1}^{n}\bigl(z_j^2-t z_j+r^{-2}\bigr).
\end{aligned}
\]
将括号内乘积视为 \(r^{-2}\) 的多项式，其常数项为
\[
\prod_{j=1}^{n}(z_j^2-t z_j)
=\Bigl(\prod_{j=1}^{n}z_j\Bigr)\prod_{j=1}^{n}(z_j-t)
=\frac{a_0}{a_n^2}P(t).
\]
故
\[
[r^n]\,Q_r(r t)=a_n^2\cdot \frac{a_0}{a_n^2}P(t)=a_0 P(t).
\]
另一方面，每个因子取 \(r^{-1}\) 项即得最低次端项，因而
\[
[r^{-n}]\,Q_r(r t)=a_n^2.
\]
结论成立。
\end{proof}

\begin{theorem}[超越尺度下的伽罗瓦全信息保持]\label{thm:group-jg-transcendental-galois-lossless}
设 \(K\) 为特征 \(0\) 的域，\(P\in K[z]\) 可分离且 \(a_0 a_n\neq 0\)。取 \(r\) 超越于 \(K\)，记 \(F:=K(r)\)。设 \(L\) 为 \(P\) 在 \(K\) 上的分裂域，\(M\) 为 \(Q_r\) 在 \(F\) 上的分裂域，则
\[
M=L(r),
\qquad
\operatorname{Gal}(Q_r/F)\cong \operatorname{Gal}(P/K).
\]
\leanverified{paper\_group\_jg\_transcendental\_galois\_lossless}
\end{theorem}
\begin{proof}

\[
g_i(z):=r z^2-w_i z+r^{-1}\in M[z].
\]
其一根为 \(z_i\)，另一根为 \(r^{-2}z_i^{-1}\)。若后一根也是 \(P\) 的根，则存在 \(j\) 使
\(
z_j=r^{-2}z_i^{-1}
\),
即 \(r^2=(z_i z_j)^{-1}\in \overline{K}\)，与 \(r\) 超越矛盾。故 \(\gcd(P,g_i)\) 在 \(M[z]\) 中为一次多项式，且等于 \(z-z_i\)（差一个单位）。由欧几里得算法系数闭合性得 \(z_i\in M\)。对所有 \(i\) 同理，故 \(L\subseteq M\)，进而 \(L(r)\subseteq M\)。

综上 \(M=L(r)\)。由于 \(r\) 为超越元且在两侧均固定，限制映射给出
\(
\operatorname{Gal}(L(r)/K(r))\cong \operatorname{Gal}(L/K)

\end{proof}

首项系数为
\[
L=(-1)^n a_n^2\prod_{j=1}^{n} z_j.
\]
判别式公式给出
\[
\operatorname{Disc}_{w}(Q_r)=L^{2n-2}\prod_{1\le i<j\le n}(w_i-w_j)^2.
\]
而
\[
\begin{aligned}
w_i-w_j
&=r(z_i-z_j)+r^{-1}(z_i^{-1}-z_j^{-1})\\
&=(z_i-z_j)\Bigl(r-r^{-1}(z_i z_j)^{-1}\Bigr)\\
&=(z_i-z_j)\frac{r^2 z_i z_j-1}{r z_i z_j}.
\end{aligned}
\]
故
\[
\prod_{i<j}(w_i-w_j)^2
=\Bigl(\prod_{i<j}(z_i-z_j)^2\Bigr)\,
r^{-n(n-1)}
\Bigl(\prod_{i<j}(r^2 z_i z_j-1)^2\Bigr)\,
\Bigl(\prod_{i<j}(z_i z_j)^{-2}\Bigr),
\]
且
\[
\prod_{i<j}(z_i z_j)^{-2}
=\prod_{k=1}^{n} z_k^{-2(n-1)}
=\Bigl(\prod_{k=1}^{n} z_k\Bigr)^{-2(n-1)}.
\]
与
\[
L^{2n-2}=a_n^{4n-4}\Bigl(\prod_{k=1}^{n} z_k\Bigr)^{2n-2}
\]
相乘后，\(\prod z_k\) 指数抵消，得到
\[
\operatorname{Disc}_{w}(Q_r)
=a_n^{4n-4}\Bigl(\prod_{i<j}(z_i-z_j)^2\Bigr)\,
r^{-n(n-1)}
\prod_{i<j}(r^2 z_i z_j-1)^2.
\]
再用
\[
\operatorname{Disc}_{z}(P)=a_n^{2n-2}\prod_{i<j}(z_i-z_j)^2
\]
代换即证。重根判据由判别式零点条件直接推出。
\end{proof}

\begin{theorem}[复时间相干半径的显式公式]\label{thm:group-complex-time-coherence-radius}
\leanverified{paper\_group\_complex\_time\_coherence\_radius}
在定理 \ref{thm:group-jg-discriminant-factorization} 的条件下，令
\[
Q_t(w):=Q_{e^t}(w),\qquad t\in\CC,
\]
并定义碰撞谱
\[
\Sigma(P):=\{\,t\in\CC:\ e^{2t}z_i z_j=1\ \text{对某些}\ i<j\,\}.
\]
则对任意 \(t_0\notin\Sigma(P)\)，存在 \(\varepsilon>0\) 及根标号
\((w_1(t),\dots,w_n(t))\)，使得在 \(|t-t_0|<\varepsilon\) 内：
\begin{enumerate}
  \item \(Q_t(w)=0\) 的全部根均为单根；
  \item 每个 \(w_j(t)\) 都是 \(t\) 的全纯函数；
  \item 任一由 \(\{w_j(t)\}\) 构成的对称全纯函数在该圆盘内全纯。
\end{enumerate}
进一步地，若
\[
\rho(t_0):=\operatorname{dist}\bigl(t_0,\Sigma(P)\bigr),
\]
则上述解析延拓的最大半径恰为 \(\rho(t_0)\)。

在特例 \(|z_j|=1\) 下，写 \(z_j=e^{i\theta_j}\)（\(\theta_j\in\RR/2\pi\ZZ\)），则
\[
\boxed{\;
\rho(0)=\frac12\min_{i<j}\operatorname{dist}_{2\pi\ZZ}(\theta_i+\theta_j,0).
\;}
\]
若 \(\{z_j\}\) 为 \(m\) 次单位根的一个满轨道（即 \(\theta_j=2\pi j/m\) 的平移），则
\[
\boxed{\;\rho(0)=\frac{\pi}{m}.\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:group-jg-discriminant-factorization}，
\[
\operatorname{Disc}_{w}(Q_t)
=C_P\,e^{-n(n-1)t}\prod_{i<j}(e^{2t}z_i z_j-1)^2,
\qquad
C_P:=a_n^{2n-2}\operatorname{Disc}_{z}(P)\neq 0.
\]
故 \(\operatorname{Disc}_{w}(Q_t)=0\) 当且仅当 \(t\in\Sigma(P)\)。对 \(t_0\notin\Sigma(P)\)，
\(Q_t(w)\) 的系数关于 \(t\) 全纯且在 \(t_0\) 处无重根，由全纯隐函数定理可在邻域内选取全纯根分支 \(w_j(t)\)。对称全纯函数作为这些分支的对称组合仍全纯。

最大半径由最近判别式零点决定：当半径超过 \(\operatorname{dist}(t_0,\Sigma(P))\) 时必触及碰撞点，无法继续保持单根标号；而在该距离以内判别式保持非零，标号可持续延拓，故最大半径即 \(\rho(t_0)\)。

若 \(|z_j|=1\)，则 \(z_i z_j=e^{i(\theta_i+\theta_j)}\)，碰撞方程
\(
e^{2t}z_i z_j=1
\)
等价于
\[
2t+i(\theta_i+\theta_j)\in 2\pi i\ZZ
\quad\Longleftrightarrow\quad
t\in i\Bigl(\pi\ZZ-\frac{\theta_i+\theta_j}{2}\Bigr).
\]
故 \(t_0=0\) 到碰撞谱的最小距离为
\(
\frac12\min_{i<j}\operatorname{dist}_{2\pi\ZZ}(\theta_i+\theta_j,0)
\)。
当 \(\{z_j\}\) 为等间距 \(m\)-轨道时，\(\theta_i+\theta_j\) 的最小非零模 \(2\pi\) 距离为 \(2\pi/m\)，于是 \(\rho(0)=\pi/m\)。
\end{proof}

\begin{theorem}[两尺度椭圆投影的代数闭包与显式反演]\label{thm:group-two-scale-conic-embedding}
\leanverified{paper\_group\_two\_scale\_conic\_embedding}
取 \(r_1,r_2\in K^\times\) 且 \(r_1^2\neq r_2^2\)。定义两尺度 Joukowsky 映射
\[
J_r(z)=r z+r^{-1}z^{-1},
\]
以及
\[
f_{r_1,r_2}:\mathbb{G}_m\to\mathbb{A}^2,\qquad
f_{r_1,r_2}(z)=(w_1,w_2):=\bigl(J_{r_1}(z),J_{r_2}(z)\bigr).
\]
则：
\begin{enumerate}
  \item \(f_{r_1,r_2}\) 为代数嵌入（在代数闭包上单射，且给出双有理参数化）；
  \item 其像为光滑仿射锥曲线 \(C_{r_1,r_2}\subset\mathbb{A}^2\)，由单个二次方程刻画
  \[
  \boxed{\;
  r_1 r_2\,(r_1 w_1-r_2 w_2)(r_1 w_2-r_2 w_1)=(r_1^2-r_2^2)^2;
  \;}
  \]
  \item 对任意满足该方程的点 \((w_1,w_2)\)，其原像由显式有理式给出
  \[
  \boxed{\;
  z=\frac{r_1 w_1-r_2 w_2}{r_1^2-r_2^2},
  \qquad
  z^{-1}=\frac{r_1 r_2\,(r_1 w_2-r_2 w_1)}{r_1^2-r_2^2}.
  \;}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定义
\[
w_1=r_1 z+r_1^{-1}z^{-1},\qquad
w_2=r_2 z+r_2^{-1}z^{-1}.
\]
把 \((z,z^{-1})\) 视为未知量，则系数矩阵为
\[
\begin{pmatrix}
r_1 & r_1^{-1}\\
r_2 & r_2^{-1}
\end{pmatrix},
\]
其行列式
\(
\frac{r_1^2-r_2^2}{r_1 r_2}\neq 0
\)。
故可唯一解得
\[
z=\frac{r_1 w_1-r_2 w_2}{r_1^2-r_2^2},\qquad
z^{-1}=\frac{r_1 r_2\,(r_1 w_2-r_2 w_1)}{r_1^2-r_2^2}.
\]
代入约束 \(z\cdot z^{-1}=1\) 即得二次方程
\[
r_1 r_2\,(r_1 w_1-r_2 w_2)(r_1 w_2-r_2 w_1)=(r_1^2-r_2^2)^2.
\]
反之，任意满足该方程的 \((w_1,w_2)\) 代入上式都给出唯一 \(z\in\mathbb{G}_m\)，故 \(f_{r_1,r_2}\) 单射且双有理可逆，其像即该仿射锥曲线。

最后验证光滑性。设
\[
F(w_1,w_2):=r_1 r_2\,(r_1 w_1-r_2 w_2)(r_1 w_2-r_2 w_1)-(r_1^2-r_2^2)^2.
\]
记 \(A=r_1 w_1-r_2 w_2,\ B=r_1 w_2-r_2 w_1\)，则
\[
\partial_{w_1}F=r_1 r_2(r_1 B-r_2 A),\qquad
\partial_{w_2}F=r_1 r_2(r_1 A-r_2 B).
\]
若两偏导同零，则线性方程组推出 \(A=B=0\)（因 \(r_1^2\neq r_2^2\)），与曲线上恒有
\(
AB=(r_1^2-r_2^2)^2/(r_1 r_2)\neq 0
\)
矛盾。故曲线无奇点，即为光滑锥曲线。
\end{proof}
