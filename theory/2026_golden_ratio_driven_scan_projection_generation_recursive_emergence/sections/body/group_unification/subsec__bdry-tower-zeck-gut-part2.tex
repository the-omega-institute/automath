\begin{proposition}[可逆性：规范 \texorpdfstring{$\ZZ_2$}{Z2} 反演的严格共轭]\label{prop:fib-tail-reversibility}
令
\[

\]

\]
\end{proposition}
\begin{proof}

\[

\]
当 \(m\) 偶数时 \(\det(G_m)=1\)，从而 \(G_m\in\mathrm{SL}_2(\ZZ)\)。
又由于 \(\det(G_m)=1\)，有
\[
G_m^{-1}=\begin{pmatrix}F_{m+3}&-F_{m+2}\\[2pt]-F_{m+4}&F_{m+3}\end{pmatrix},
\]
并且直接计算
\[
R\,G_m\,R
=
\begin{pmatrix}-1&0\\[2pt]0&1\end{pmatrix}
\begin{pmatrix}F_{m+3}&F_{m+2}\\[2pt]F_{m+4}&F_{m+3}\end{pmatrix}
\begin{pmatrix}-1&0\\[2pt]0&1\end{pmatrix}
=
\begin{pmatrix}F_{m+3}&-F_{m+2}\\[2pt]-F_{m+4}&F_{m+3}\end{pmatrix}
=G_m^{-1}.
\]
\leanverified{paper\_fib\_tail\_reversibility}
证毕。
\end{proof}
