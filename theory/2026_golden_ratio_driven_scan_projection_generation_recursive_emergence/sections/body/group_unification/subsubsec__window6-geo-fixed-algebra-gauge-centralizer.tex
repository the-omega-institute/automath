\begin{proposition}[对合不变集合的不动点计数与 Stokes 边界读出]\label{prop:terminal-hypercube-affine-involution-fixedpoint-gauss-stokes}
设 \(m\ge 2\)，并令 \(\sigma=\sigma_{i,j}\) 为定理 \ref{thm:window6-chiral-compression-hypercube-adjacency} 中的仿射对合（\(1\le i<j\le m\)）。
对任意满足 \(\sigma(B)=B\) 的子集 \(B\subseteq\Omega_m\)，定义不动点计数
\[
f(B):=\#\{\omega\in B:\ \sigma(\omega)=\omega\}.
\]
则有严格恒等式
\[
f(B)=\frac{|B|-\sum_{\omega\in B}\chi_{\{i,j\}}(\omega)}{2}.
\]
并且当取权向量 \(\mathbf 1=(1,\dots,1)\) 时，\(\sum_{\omega\in B}\chi_{\{i,j\}}(\omega)\) 可由 \(B\) 的有向边界 \(\partial\overrightarrow{B}\) 上 \(1\)-形式通量唯一确定（定理 \ref{thm:xi-hypercube-fourier-walsh-boundary-parseval}），从而 \(f(B)\) 为边界可观测量。
\par
在 window-\(6\) 中取 \((i,j)=(1,5)\) 并令 \(\sigma=\sigma_{\mathrm{geo}}\)，则 \(\sigma_{\mathrm{geo}}\) 的不动点恰为 \(\{\omega\in\Omega_6:\ \omega_1\neq \omega_5\}\)，并且对任意纤维 \(F_x=F_6^{-1}(x)\)（推论 \ref{cor:terminal-foldbin6-geo-orbit-charge} 的记号）有
\[
f_x=\frac{|F_x|-\sum_{\omega\in F_x}\chi_{\{1,5\}}(\omega)}{2}.
\]
\leanverified{paper\_terminal\_hypercube\_affine\_involution\_fixedpoint\_gauss\_stokes}
\end{proposition}
\begin{proof}
方程 \(\sigma(\omega)=\omega\) 在 \((i,j)\) 两坐标上等价于 \(\omega_i=1-\omega_j\)，即 \(\omega_i\neq \omega_j\)。
因此不动点指标满足
\[
\mathbf 1_{\{\omega_i\neq \omega_j\}}
=\frac{1-(-1)^{\omega_i+\omega_j}}{2}
=\frac{1-\chi_{\{i,j\}}(\omega)}{2},
\]
对 \(\omega\in B\) 求和即得
\(
f(B)=\frac{|B|-\sum_{\omega\in B}\chi_{\{i,j\}}(\omega)}{2}
\)。
当取 \(\mathbf w=\mathbf 1\) 与 \(I=\{i,j\}\) 时，定理 \ref{thm:xi-hypercube-fourier-walsh-boundary-parseval} 将
\(
\sum_{\omega\in B}\chi_{\{i,j\}}(\omega)
\)
表示为 \(\partial\overrightarrow{B}\) 上的通量求和，故 \(f(B)\) 边界可见。
window-\(6\) 的特例由 \(\sigma_{\mathrm{geo}}=\sigma_{1,5}\) 直接专门化得到。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\sigma_{\mathrm{geo}}$}{sigma_geo}-不变子代数的 Wedderburn 精细分解与连续对称压缩（window-\(6\)）]\label{thm:terminal-window6-geo-fixed-subalgebra-wedderburn}
沿用推论 \ref{cor:terminal-foldbin6-geo-orbit-charge} 的记号。
记 window-\(6\) 的二进制纤维群胚代数为
\[
A_6:=\CC[\mathcal{R}_6^{\mathrm{bin}}]\ \cong\ \bigoplus_{x\in X_6}M_{d_x}(\CC),
\qquad d_x:=|F_6^{-1}(x)|.
\]
对每个 \(x\in X_6\)，令 \(\pi_x\) 为 \(\sigma_{\mathrm{geo}}\) 在纤维 \(F_6^{-1}(x)\) 上诱导的置换，并取任意标号以得到对应的置换矩阵 \(P_x\in U(d_x)\)。
定义 \(\ast\)-自同构
\[
\alpha:=\bigoplus_{x\in X_6}\alpha_x,\qquad
\alpha_x(T):=P_xTP_x^{-1}\qquad (T\in M_{d_x}(\CC)).
\]
则其不变子代数
\(
A_6^\alpha:=\{a\in A_6:\ \alpha(a)=a\}
\)
与标号选择无关，并满足完全显式的 Wedderburn 精细分解
\[
\boxed{\;
A_6^\alpha\ \cong\
\CC^{\oplus 14}\ \oplus\ M_2(\CC)^{\oplus 12}\ \oplus\ M_3(\CC)^{\oplus 6}\ \oplus\ M_4(\CC)^{\oplus 2}\; }.
\]
其恒等连通 \(\ast\)-自同构群（等价地，内自同构群的连通分量）满足
\[
\boxed{\;
\Aut(A_6^\alpha)^0\ \cong\
\mathrm{PU}(2)^{12}\times \mathrm{PU}(3)^{6}\times \mathrm{PU}(4)^{2}\; }.
\]
\end{theorem}
\begin{proof}
固定 \(x\in X_6\)。由推论 \ref{cor:terminal-foldbin6-geo-orbit-charge}，\(\pi_x\) 为对合置换，其循环分解仅包含 \(1\)-循环与 \(2\)-循环。
令 \(q(x)=(f_x,p_x)\) 为轨道电荷，其中 \(f_x\) 为不动点数、\(p_x\) 为 \(2\)-循环数，且 \(d_x=f_x+2p_x\)。
存在酉矩阵 \(U_x\) 使
\[
U_xP_xU_x^{-1}=\mathrm{diag}\!\bigl(I_{f_x+p_x},\,-I_{p_x}\bigr).
\]
因此
\[
M_{d_x}(\CC)^{\alpha_x}
=\{T\in M_{d_x}(\CC):\ P_xT=TP_x\}
\cong M_{f_x+p_x}(\CC)\oplus M_{p_x}(\CC),
\]
该同构仅依赖于 \((f_x,p_x)\) 而与纤维标号无关。
\par
将表 \ref{tab:foldbin6_geo_orbit_charge} 给出的 \((f_x,p_x)\) 直方图逐类代入并汇总即可得到全局 Wedderburn 分解。
再用有限维半单 \(\mathrm{C}^\ast\)-代数的标准结构：每个 \(M_n(\CC)\) 的内自同构连通分量为 \(\mathrm{PU}(n)\)，而 \(\CC\) 不产生非平凡内自同构，即得所述连通群分解。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\sigma_{\mathrm{geo}}$}{sigma_geo} 在离散规范群中的中心化子与共轭类基数（window-\(6\)）]\label{thm:terminal-window6-geo-gauge-centralizer-class}
令离散规范群
\[
G^{\mathrm{bin}}_6:=\prod_{x\in X_6}\mathfrak{S}_{d_x},
\qquad d_x:=|F_6^{-1}(x)|
\]
如命题 \ref{prop:conclusion-discrete-gauge-double-exp}（取 \(m=6\)）。
将 \(\sigma_{\mathrm{geo}}\) 在各纤维上诱导的置换族 \((\pi_x)_{x\in X_6}\) 视为 \(G^{\mathrm{bin}}_6\) 的元素 \(g\)（逐分量拼接）。
则中心化子规模与共轭类基数具有闭式
\[
|C_{G^{\mathrm{bin}}_6}(g)|=2{,}783{,}138{,}807{,}808,
\qquad
|G^{\mathrm{bin}}_6:C_{G^{\mathrm{bin}}_6}(g)|=314{,}928,
\]
并由生成片段 \eqref{eq:foldbin6_geo_gauge_centralizer} 给出可复核的数值证书。
\end{theorem}
\begin{proof}
由 \(G^{\mathrm{bin}}_6\) 的直积分解，中心化子按分量分解为直积：
\(
C_{G^{\mathrm{bin}}_6}(g)=\prod_{x\in X_6}C_{\mathfrak S_{d_x}}(\pi_x)
\)，从而
\(
|C_{G^{\mathrm{bin}}_6}(g)|=\prod_{x\in X_6}|C_{\mathfrak S_{d_x}}(\pi_x)|
\)。
而 \(\pi_x\) 的循环型由轨道电荷 \(q(x)=(f_x,p_x)\) 唯一确定：其为 \(1^{f_x}2^{p_x}\)。
对称群中给定循环型 \(1^{f}2^{p}\) 的中心化子基数为 \(f!\,2^{p}p!\)，故 \(|C_{\mathfrak S_{d_x}}(\pi_x)|=f_x!\,2^{p_x}p_x!\)。
将表 \ref{tab:foldbin6_geo_orbit_charge} 的七类 \((f_x,p_x)\) 与出现次数代入并取乘积，即得到 \(|C_{G^{\mathrm{bin}}_6}(g)|\) 的闭式。
同时 \(|G^{\mathrm{bin}}_6|=\prod_x (d_x!)=(2!)^8(3!)^4(4!)^9\) 由 \(d_x\) 的直方图给出，故共轭类基数为商 \(|G^{\mathrm{bin}}_6|/|C_{G^{\mathrm{bin}}_6}(g)|\)。
上述两项的数值化与一致性核验由生成片段 \eqref{eq:foldbin6_geo_gauge_centralizer} 固定。证毕。
\end{proof}

\begin{corollary}[轨道电荷的局部分块与半单对称压缩的内生样式]\label{cor:terminal-window6-geo-orbit-charge-symmetry-compression}
对任意 \(x\in X_6\)，若 \(q(x)=(f_x,p_x)\) 且 \(d_x=f_x+2p_x\)，则由定理 \ref{thm:terminal-window6-geo-fixed-subalgebra-wedderburn} 的局部分解可得
\[
M_{d_x}(\CC)^{\alpha_x}\cong M_{f_x+p_x}(\CC)\oplus M_{p_x}(\CC),
\]
从而在酉对称的 Lie 代数层面存在自然压缩型同态
\[
\mathfrak u(d_x)\ \longrightarrow\ \mathfrak u(f_x+p_x)\oplus \mathfrak u(p_x),
\]
其半单部分对应于
\[
\mathfrak{su}(d_x)\ \longrightarrow\ \mathfrak{su}(f_x+p_x)\oplus \mathfrak{su}(p_x)\oplus \mathfrak u(1)^{\oplus \varepsilon_x},
\]
其中 \(\varepsilon_x\in\{0,1\}\) 由直和分块是否同时非平凡决定。
特别地，window-\(6\) 的轨道电荷直方图（表 \ref{tab:foldbin6_geo_orbit_charge}）在 \(d_x=4\) 的纤维上强制出现三种压缩型：
\[
(f_x,p_x)=(2,1):\ \ \mathfrak{su}(4)\to \mathfrak{su}(3)\oplus \mathfrak u(1),\qquad
(f_x,p_x)=(0,2):\ \ \mathfrak{su}(4)\to \mathfrak{su}(2)\oplus \mathfrak{su}(2)\oplus \mathfrak u(1),
\]
并在 \(d_x=3\) 的纤维上强制出现
\[
(f_x,p_x)=(1,1):\ \ \mathfrak{su}(3)\to \mathfrak{su}(2)\oplus \mathfrak u(1).
\]
\end{corollary}
