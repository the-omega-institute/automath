\begin{corollary}[Lee--Yang 单位圆假设下的全息恒等式]\label{cor:group-jg-leyang-holographic-identity}
在定理 \ref{thm:group-jg-pullback-factorization} 的设定下，进一步假设 \(K\subseteq\CC\) 且 \(P\) 的全部零点满足
\[
|z_j|=1\qquad(1\le j\le n),
\]
并取 \(r\in\RR_{>0}\)。
记 \(J_r(z)=r z+r^{-1}z^{-1}\)。
则对任意 \(z\in\CC^\times\) 有严格恒等式
\[
\boxed{\;
Q_r\!\bigl(J_r(z)\bigr)=r^{n}\,P(z)\,P\!\bigl(r^{-2}z^{-1}\bigr).
\;}
\]
\leanverified{paper\_group\_jg\_leyang\_holographic\_identity}
\leanverified{J\_r}
\leanverified{P}
\leanverified{Q\_r\_eval\_at\_J}
\leanverified{J\_r\_eq}
\leanverified{P\_at\_reciprocal}
\leanverified{r\_sq\_P\_P\_reciprocal}
\leanverified{Q\_r\_eval\_at\_J\_eq}
\leanverified{leyang\_holographic\_n2}
\end{corollary}

\begin{proof}
由定理 \ref{thm:group-jg-pullback-factorization}，
\[
r^n z^n\,Q_r\!\bigl(J_r(z)\bigr)=P(z)\,P^{\vee}(r^2 z).
\]
由倒易多项式的定义 \(P^{\vee}(u)=u^n P(1/u)\)，取 \(u=r^2 z\) 得
\[
P^{\vee}(r^2 z)=r^{2n}z^n P(r^{-2}z^{-1}).
\]
代回并消去公共因子 \(z^n\) 即得结论。证毕。
\end{proof}

\begin{corollary}[双结式消元与同心圆双谱]\label{cor:group-jg-leyang-double-resultant}
\leanverified{paper\_group\_jg\_leyang\_double\_resultant}
沿用推论 \ref{cor:group-jg-leyang-holographic-identity} 的设定，并定义
\[
S_r(z):=\operatorname{Res}_w\!\bigl(Q_r(w),\,r z^2-w z+r^{-1}\bigr)\in K(r)[z].
\]
则在 \(K(r)[z]\) 中有严格恒等式
\[
\boxed{\;
S_r(z)=(-1)^n r^{n}\,P(z)\,\Bigl[z^{n}P\!\bigl(r^{-2}z^{-1}\bigr)\Bigr].
\;}
\]
从而在 \(\CC\) 上其零点集合恰为
\[
\mathrm{Zero}(S_r)
=\{z_j\}_{j=1}^n\ \cup\ \{r^{-2}z_j^{-1}\}_{j=1}^n,
\]
其中两簇零点分别落在圆 \(|z|=1\) 与 \(|z|=r^{-2}\) 上。
\end{corollary}

\begin{proof}
因 \(r z^2-w z+r^{-1}\) 关于 \(w\) 为一次多项式，结式的线性特化公式给出
\[
\operatorname{Res}_w\!\bigl(Q_r(w),\,r z^2-w z+r^{-1}\bigr)
=(-z)^n\,Q_r\!\left(\frac{r z^2+r^{-1}}{z}\right)
=(-z)^n Q_r\!\bigl(J_r(z)\bigr).
\]
再代入推论 \ref{cor:group-jg-leyang-holographic-identity} 得
\[
S_r(z)=(-z)^n r^{n}P(z)P(r^{-2}z^{-1})
=(-1)^n r^{n}P(z)\,\bigl[z^nP(r^{-2}z^{-1})\bigr],
\]
即所示恒等式。零点位置由两因子分别给出。证毕。
\end{proof}

\begin{corollary}[单尺度椭圆输运的可逆性与全息重构]\label{cor:group-jg-leyang-single-scale-reconstruction}
\leanverified{paper\_group\_jg\_leyang\_single\_scale\_reconstruction}
在推论 \ref{cor:group-jg-leyang-double-resultant} 的设定下，进一步假设 \(r>1\)，且 \(P\) 可分离并首一。
则映射
\[
\mathcal{T}_r:\ P\ \longmapsto\ Q_r
\]
为单射，并且存在由 \((r,Q_r)\) 唯一确定的重构方式：由 \(Q_r\) 计算 \(S_r(z)\)，则 \(P\) 等于 \(S_r\) 的唯一首一次数 \(n\) 因子，其全部零点落在单位圆 \(|z|=1\) 上。
\end{corollary}

\begin{proof}
由推论 \ref{cor:group-jg-leyang-double-resultant}，
\[
S_r(z)=(-1)^n r^n\,P(z)\cdot \bigl[z^nP(r^{-2}z^{-1})\bigr].
\]
当 \(r>1\) 时，第二因子的零点全部落在圆 \(|z|=r^{-2}<1\) 上，而 \(P\) 的零点全部落在 \(|z|=1\) 上，故两簇零点不相交。
因此 \(S_r\) 在 \(\CC[z]\) 的分解中存在且仅存在一个次数 \(n\) 的因子，其零点全在单位圆上；取其首一规范即得 \(P\)。
若 \(\mathcal{T}_r(P_1)=\mathcal{T}_r(P_2)\)，则相应的 \(S_r\) 相同，从而上述单位圆根簇因子相同，故 \(P_1=P_2\)。证毕。
\end{proof}
