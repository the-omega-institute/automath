    \item \textbf{确定性“面积律”分解：\texorpdfstring{\(\det(L_k+tI)\)}{det(L_k+tI)} 的 \(q\)-形式与指数级有限尺寸余项。}
    取 \(t>0\)，令
    \(
    \eta(t)=\mathrm{arsinh}\sqrt{t/4}
    \)
    与 \(q(t)=e^{-2\eta(t)}\in(0,1)\)。
    则有严格恒等式
    \[
    \det(L_k+tI)=\frac{\cosh((2k+1)\eta(t))}{\cosh(\eta(t))}
    =q(t)^{-k}\frac{1+q(t)^{2k+1}}{1+q(t)},
    \]
    从而
    \(
    \log\det(L_k+tI)=2k\eta(t)-\log(1+q(t))+\log(1+q(t)^{2k+1})
    \)
    ，且余项满足 \(0<\log(1+q^{2k+1})<q^{2k+1}\)（推论 \ref{cor:pom-Lk-area-law-qform}）。

    \item \textbf{强 Szeg\H{o} 型 bulk--surface 分解与可积闭式：符号 \texorpdfstring{\(a_t(\theta)\)}{a_t(theta)} 的积分密度恰为 \texorpdfstring{\(2\eta(t)\)}{2 eta(t)}.}
    记
    \(a_t(\theta)=1+\frac{t}{4\sin^2(\theta/2)}\)（\(\theta\in(0,\pi)\)）。
    由 \(\det(K_k)=1\) 得 \(\det(I+tK_k)=\det(L_k+tI)\)，并有严格分解
    \[
    \log\det(I+tK_k)
    =k\cdot \frac{1}{\pi}\int_{0}^{\pi}\log a_t(\theta)\,\dd\theta
    +\log\frac{1+q(t)^{2k+1}}{1+q(t)},
    \]
    其中体相积分项可显式化为
    \(
    \frac{1}{\pi}\int_{0}^{\pi}\log\!\bigl(1+\frac{t}{4\sin^2(\theta/2)}\bigr)\,\dd\theta
    =2\eta(t)
    \)
    （命题 \ref{prop:pom-Kk-strong-szego-bulk-surface}）。

    \item \textbf{解析可审计的 resolvent 迹：\texorpdfstring{\(\tr(L_k+tI)^{-1}\)}{tr((L_k+tI)^{-1})} 的双曲闭式。}
    resolvent 迹满足严格恒等式
    \[
    \tr(L_k+tI)^{-1}
    =\frac{2k+1}{2\sqrt{t(4+t)}}\,\tanh\bigl((2k+1)\eta(t)\bigr)-\frac{1}{2(4+t)},
    \]
    因而迹密度极限为
    \(
    \lim_{k\to\infty}\frac{1}{2k+1}\tr(L_k+tI)^{-1}=\frac{1}{2\sqrt{t(4+t)}}
    \)
    （推论 \ref{cor:pom-Lk-resolvent-trace-hyperbolic}）。

    \item \textbf{Green 核矩阵元的完全闭式与两点传播压缩：\texorpdfstring{\((L_k+tI)^{-1}_{ij}\)}{(L_k+tI)^{-1}_{ij}} 的双曲乘积公式。}
    对 \(1\le i\le j\le k\) 有严格闭式
    \[
    (L_k+tI)^{-1}_{ij}
    =\frac{\sinh(2i\eta(t))\,\cosh((2(k-j)+1)\eta(t))}
    {\sinh(2\eta(t))\,\cosh((2k+1)\eta(t))},
    \qquad (L_k+tI)^{-1}_{ji}=(L_k+tI)^{-1}_{ij}.
    \]
    特别地，
    \(
    (L_k+tI)^{-1}_{11}=\cosh((2k-1)\eta)/\cosh((2k+1)\eta)
    \)
    与
    \(
    (L_k+tI)^{-1}_{1k}=\cosh(\eta)/\cosh((2k+1)\eta)
    \)
    完全显式（命题 \ref{prop:pom-Lk-green-kernel-entries}）。

    \item \textbf{指数聚类与相关长度：质量参数 \texorpdfstring{\(t\)}{t} 的严格“间隙律”。}
    对任意 \(1\le i\le j\le k\) 有一致上界
    \[
    (L_k+tI)^{-1}_{ij}\le \frac{1}{\sinh(2\eta(t))}\,e^{-2\eta(t)(j-i)},
    \]
    从而相关长度可取
    \(
    \xi(t)=1/(2\eta(t))=1/(2\,\mathrm{arsinh}\sqrt{t/4})
    \)
    （推论 \ref{cor:pom-Lk-exponential-clustering-gaplaw}）。

    \item \textbf{半无限极限的边界固定点：反射系数 \texorpdfstring{\(q(t)\)}{q(t)} 的代数闭式与 Riccati 方程。}
    边界 Green 值满足
    \(
    \lim_{k\to\infty}(L_k+tI)^{-1}_{11}=q(t)=e^{-2\eta(t)}
    \)
    ，且
    \[
    q(t)=\Bigl(\frac{\sqrt{4+t}-\sqrt{t}}{2}\Bigr)^2
    =1+\frac{t}{2}-\frac12\sqrt{t(4+t)},
    \qquad
    q^2-(2+t)q+1=0,
    \]
    并由此形成半无限 continued-fraction/Riccati 的代数证书（推论 \ref{cor:pom-Lk-boundary-fixed-point-riccati}）。

    \item \textbf{表面自由能与边界响应：常数项闭式与可观测导数。}
    设
    \(f(t)=\lim_{k\to\infty}\frac1k\log\det(L_k+tI)=2\eta(t)\)
    与
    \(g(t)=\lim_{k\to\infty}(\log\det-k f(t))=-\log(1+q(t))\)，则有精确分解
    \[
    \log\det(L_k+tI)=k f(t)+g(t)+\log(1+q(t)^{2k+1}),
    \]
    且
    \(
    g'(t)=\frac{q(t)}{1+q(t)}\cdot \frac{1}{\sqrt{t(4+t)}}
    \)
    （推论 \ref{cor:pom-Lk-surface-free-energy}；并见注 \ref{rem:pom-fence-green-kernel-area-law-audit} 的有限窗审计表 \ref{tab:fence_green_kernel_area_law_resolvent_audit}）。
