\begin{theorem}[交换--取补对合诱导的手征压缩与邻接算子降维交织]\label{thm:window6-chiral-compression-hypercube-adjacency}
设 \(m\ge 2\)，令 \(\Omega_m:=\{0,1\}^m\) 并记 \(V_m:=\CC^{\Omega_m}\) 为其复值函数空间。
取两坐标 \(i\neq j\)，定义超立方体图 \(Q_m\) 的一个仿射对合
\[
\sigma_{i,j}(\omega)_i:=1-\omega_j,\qquad
\sigma_{i,j}(\omega)_j:=1-\omega_i,\qquad
\sigma_{i,j}(\omega)_k:=\omega_k\ (k\ne i,j).
\]
令 \((\sigma_{i,j}\cdot f)(\omega):=f(\sigma_{i,j}^{-1}\omega)\) 为诱导置换表示，并定义符号等变分量
\[
V_m^-:=\{f\in V_m:\ \sigma_{i,j}\cdot f=-f\}.
\]
记 \(A_m\) 为 \(Q_m\) 的邻接算子：
\(
(A_m f)(\omega):=\sum_{\omega'\sim\omega}f(\omega')
\)。
则成立：
\begin{enumerate}
  \item \(\dim_{\CC}V_m^-=2^{m-2}\)。
  \item 存在一个由 \((i,j)\) 与坐标顺序唯一确定的线性同构
  \(
  \Phi_{i,j}:\CC^{\Omega_{m-2}}\xrightarrow{\sim}V_m^-
  \)
  使得对每个 \(u\in\Omega_{m-2}\) 有
  \[
  \Phi_{i,j}(\mathbf 1_u)=\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)},
  \]
  其中 \(\omega^{00}(u)\in\Omega_m\) 表示在第 \(i,j\) 位取 \((0,0)\)、其余位按既定顺序填入 \(u\) 的顶点，\(\omega^{11}(u)\) 类似。
  \item 邻接算子在 \(V_m^-\) 上与 \(A_{m-2}\) 交织：
  \[
  A_m\circ \Phi_{i,j}=\Phi_{i,j}\circ A_{m-2},
  \qquad\text{从而}\qquad
  A_m|_{V_m^-}\ \cong\ A_{m-2}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
对合 \(\sigma_{i,j}\) 在 \(\Omega_m\) 上的轨道大小只能为 \(1\) 或 \(2\)。
当 \((\omega_i,\omega_j)\in\{(0,1),(1,0)\}\) 时有 \(\sigma_{i,j}(\omega)=\omega\)；当 \((\omega_i,\omega_j)\in\{(0,0),(1,1)\}\) 时两点互换。
因此每个 \(u\in\Omega_{m-2}\) 给出唯一的 \(2\)-循环轨道 \(\{\omega^{00}(u),\omega^{11}(u)\}\)，总数为 \(2^{m-2}\)。
每条 \(2\)-循环在 \(V_m\) 上贡献一维反对称方向 \(\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}\)，从而 \(\dim V_m^-=2^{m-2}\)，并且所示 \(\Phi_{i,j}\) 给出同构。
\par
对交织恒等式，只需计算对基向量的作用。
固定 \(u\in\Omega_{m-2}\) 并考虑差向量 \(\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}\)。
在坐标 \(i\) 与 \(j\) 方向的一步翻转把 \(\omega^{00}(u)\) 送到 \((1,0,u)\) 与 \((0,1,u)\)，把 \(\omega^{11}(u)\) 送到 \((0,1,u)\) 与 \((1,0,u)\)，两者对差向量的贡献相消。
其余 \(m-2\) 个坐标方向仅在 \(u\) 上翻转某一位 \(t\)，并把两点同时送到 \(\omega^{00}(u\oplus e_t)\) 与 \(\omega^{11}(u\oplus e_t)\)，故
\[
A_m\bigl(\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}\bigr)
=\sum_{t=1}^{m-2}\bigl(\mathbf 1_{\omega^{00}(u\oplus e_t)}-\mathbf 1_{\omega^{11}(u\oplus e_t)}\bigr)
=\Phi_{i,j}(A_{m-2}\mathbf 1_u),
\]
即 \(A_m\circ \Phi_{i,j}=\Phi_{i,j}\circ A_{m-2}\)。证毕。
\end{proof}

\begin{theorem}[仿射对合的 Walsh--谱手征分解]\label{thm:window6-affine-chiral-walsh-decomposition}
沿用定理 \ref{thm:window6-chiral-compression-hypercube-adjacency} 的设定，并记
\(
V_m^+:=\{f\in V_m:\ \sigma_{i,j}\cdot f=f\}
\)。
对任意 \(S\subseteq[m]\) 定义 Walsh 字符
\[
\chi_S(\omega):=(-1)^{\sum_{k\in S}\omega_k}\qquad(\omega\in\Omega_m).
\]
令 \(R:=[m]\setminus\{i,j\}\)。则对任意 \(J\subseteq R\)：
\begin{enumerate}
  \item \(\chi_J\in V_m^+\) 且 \(\chi_{J\cup\{i,j\}}\in V_m^+\)；
  \item 定义
  \[
  \psi_J^+:=\chi_{J\cup\{i\}}-\chi_{J\cup\{j\}},
  \qquad
  \psi_J^-:=\chi_{J\cup\{i\}}+\chi_{J\cup\{j\}},
  \]
  则 \(\psi_J^+\in V_m^+\) 且 \(\psi_J^-\in V_m^-\)。
\end{enumerate}
并且集合
\[
\mathcal B^+:=\Bigl\{\chi_J,\ \chi_{J\cup\{i,j\}},\ \psi_J^+:\ J\subseteq R\Bigr\}
\]
构成 \(V_m^+\) 的一组 Walsh--正交基，而
\[
\mathcal B^-:=\Bigl\{\psi_J^-:\ J\subseteq R\Bigr\}
\]
构成 \(V_m^-\) 的一组 Walsh--正交基。
特别地，
\[
\dim V_m^+=3\cdot 2^{m-2},
\qquad
\dim V_m^-=2^{m-2},
\qquad
V_m=V_m^+\oplus V_m^-.
\]
\end{theorem}
\begin{proof}
对任意 \(J\subseteq R\)，由于 \(\sigma_{i,j}\) 保持 \(R\) 上坐标不变且在 \((i,j)\) 两位的“交换并取补”只引入偶数次取补贡献，直接计算得
\(
\chi_J\circ\sigma_{i,j}=\chi_J
\)
与
\(
\chi_{J\cup\{i,j\}}\circ\sigma_{i,j}=\chi_{J\cup\{i,j\}}
\)，从而二者属于 \(V_m^+\)。
进一步，
\(
\chi_{J\cup\{i\}}\circ\sigma_{i,j}=-\chi_{J\cup\{j\}}
\)
与
\(
\chi_{J\cup\{j\}}\circ\sigma_{i,j}=-\chi_{J\cup\{i\}}
\)，
故 \(\psi_J^\pm\) 分别落在 \(V_m^\pm\) 中。
\par
Walsh 字符族 \(\{\chi_S\}_{S\subseteq[m]}\) 在 \(\ell^2(\Omega_m)\) 中两两正交，
且 \(\psi_J^\pm\) 为同一维数层上的线性组合并在不同 \(J\) 上仍保持正交。
基数计数给出 \(|\mathcal B^-|=2^{m-2}\)、\(|\mathcal B^+|=3\cdot 2^{m-2}\)，并与 \(\dim V_m=2^m\) 相容，从而两组正交集分别为 \(V_m^\pm\) 的基，且 \(V_m=V_m^+\oplus V_m^-\)。证毕。
\end{proof}

\begin{theorem}[手征子空间上的热迹闭式与 \texorpdfstring{$\Phi_3$}{Phi3} 因子]\label{thm:window6-chiral-heat-trace-phi3}
在 \(\Omega_m\) 上定义无权超立方体拉普拉斯算子
\[
(\Delta f)(\omega):=\sum_{r=1}^{m}\bigl(f(\omega)-f(\omega^{(r)})\bigr),
\]
其中 \(\omega^{(r)}\) 表示翻转第 \(r\) 位后的顶点。
由于 \(\sigma_{i,j}\) 为 \(Q_m\) 的自同构，\(\Delta\) 与 \(\sigma_{i,j}\) 可交换，从而可在 \(V_m^\pm\) 上定义手征热迹
\[
Z^\pm(t):=\Tr_{V_m^\pm}\!\bigl(e^{-t\Delta}\bigr)\qquad(t>0).
\]
则有闭式
\[
Z^-(t)=e^{-2t}(1+e^{-2t})^{m-2},
\qquad
Z^+(t)=(1+e^{-2t})^{m-2}\bigl(1+e^{-2t}+e^{-4t}\bigr).
\]
令 \(q:=e^{-2t}\)，则
\[
Z^-(q)=q(1+q)^{m-2},
\qquad
Z^+(q)=(1+q)^{m-2}(1+q+q^2),
\]
其中 \(1+q+q^2=\Phi_3(q)\) 为三次分圆多项式。
\end{theorem}
\begin{proof}
Walsh 字符 \(\chi_S\)（\(S\subseteq[m]\)）对角化 \(\Delta\)，并满足
\(
\Delta\chi_S=2|S|\cdot \chi_S
\)。
由定理 \ref{thm:window6-affine-chiral-walsh-decomposition}，在 \(V_m^-\) 上的正交基可取为 \(\{\psi_J^-:\ J\subseteq R\}\)，且每个 \(\psi_J^-\) 由 \(|J|+1\) 阶 Walsh 字符线性生成，故其特征值为 \(2(|J|+1)\)。
因此
\[
Z^-(t)=\sum_{J\subseteq R}e^{-2t(|J|+1)}=e^{-2t}\sum_{J\subseteq R}e^{-2t|J|}
=e^{-2t}(1+e^{-2t})^{m-2}.
\]
同理，\(V_m^+\) 的正交基可取为 \(\{\chi_J,\psi_J^+,\chi_{J\cup\{i,j\}}:\ J\subseteq R\}\)，其特征值分别为 \(2|J|\)、\(2(|J|+1)\)、\(2(|J|+2)\)，从而
\[
Z^+(t)=\sum_{J\subseteq R}\Bigl(e^{-2t|J|}+e^{-2t(|J|+1)}+e^{-2t(|J|+2)}\Bigr)
=(1+e^{-2t})^{m-2}(1+e^{-2t}+e^{-4t}).
\]
令 \(q=e^{-2t}\) 即得 \(q\)-形式与 \(\Phi_3\) 因子。证毕。
\end{proof}

\begin{theorem}[手征窗口证书的 definable-envelope replacement 与 stabilizer extraction]\label{distill:hrushovski_zilber_zariski_geometries_and_elekes_szabo_group_extraction-definable_envelope_and_stabilizer_theorem_schema-window6-chiral-coset-envelope-stabilizer}
沿用定理 \ref{thm:window6-chiral-compression-hypercube-adjacency} 的记号，并把
\(G_R:=\Omega_{m-2}\) 视为 \(\mathbb F_2\)-向量群，群律记为 \(\oplus\)。
对 \(a\in G_R\)，令 \(\tau_a(u):=u\oplus a\)，并令 \(\widetilde\tau_a\) 为 \(\Omega_m\) 上固定第 \(i,j\) 位且在其余坐标上按 \(\tau_a\) 平移的自同构。
对任意 \(C\subseteq G_R\) 定义手征证书
\[
\eta_C:=\sum_{u\in C}\bigl(\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}\bigr)\in V_m^- .
\]
设 \(S\subseteq G_R\) 为有限对称集且 \(0\in S\)，令 \(H:=\langle S\rangle\le G_R\)。
假设 \(H\) 关于 \(S\) 的词直径至多为 \(L\)，并且存在 \(\varepsilon\ge0\) 使得对每个 \(s\in S\) 有
\[
\bigl\|\widetilde\tau_s\cdot\eta_C-\eta_C\bigr\|_2^2\le 2\varepsilon\card{G_R}.
\]
对每个陪集 \(Q\in G_R/H\) 记
\[
p_Q:=\frac{\card{C\cap Q}}{\card{H}},
\qquad
\operatorname{Env}_H(C):=\bigcup_{\substack{Q\in G_R/H\\ p_Q>1/2}}Q.
\]
则 \(\operatorname{Env}_H(C)=q_H^{-1}\{Q:p_Q>1/2\}\) 是由商映射 \(q_H:G_R\to G_R/H\) 定义的 \(H\)-饱和包络，并且成立：
\begin{enumerate}
  \item \(H\le \operatorname{Stab}(\operatorname{Env}_H(C))\)，其中
  \[
  \operatorname{Stab}(E):=\{a\in G_R:E\oplus a=E\}.
  \]
  因而 \(\widetilde\tau_h\cdot\eta_{\operatorname{Env}_H(C)}=\eta_{\operatorname{Env}_H(C)}\) 对所有 \(h\in H\) 成立。
  \item 近不变证书与其精确稳定子包络满足
  \[
  \card{C\triangle \operatorname{Env}_H(C)}\le L\varepsilon\card{G_R},
  \qquad
  \bigl\|\eta_C-\eta_{\operatorname{Env}_H(C)}\bigr\|_2^2\le 2L\varepsilon\card{G_R}.
  \]
  \item 对任意 \(0<\alpha\le1/2\)，混合陪集满足
  \[
  \card{\{Q\in G_R/H:\alpha\le p_Q\le1-\alpha\}}
  \le
  \frac{L\varepsilon\card{G_R}}{2\alpha(1-\alpha)\card{H}}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
若 \(C,D\subseteq G_R\)，则不同 \(u\) 对应的差向量
\(\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}\) 两两正交且平方范数为 \(2\)，故
\[
\|\eta_C-\eta_D\|_2^2=2\card{C\triangle D}.
\]
因此假设等价于
\[
\card{(C\oplus s)\triangle C}\le \varepsilon\card{G_R}\qquad(s\in S).
\]
任取 \(h\in H\)。由词直径假设，可写
\(h=s_1\oplus\cdots\oplus s_r\)，其中 \(s_k\in S\) 且 \(r\le L\)。
利用 \(G_R\) 的平移保持基数，逐步作对称差估计得
\[
\card{(C\oplus h)\triangle C}
\le \sum_{k=1}^r \card{(C\oplus s_k)\triangle C}
\le L\varepsilon\card{G_R}.
\]
于是
\[
\frac1{\card{H}}\sum_{h\in H}\card{(C\oplus h)\triangle C}
\le L\varepsilon\card{G_R}.
\]
固定陪集 \(Q\in G_R/H\)，令 \(A_Q:=C\cap Q\) 且 \(N:=\card{H}\)。
\(H\) 在 \(Q\) 上自由传递，因此
\[
\frac1{\card{H}}\sum_{h\in H}\card{A_Q\cap(A_Q\oplus h)}
=\frac{\card{A_Q}^2}{N}=Np_Q^2.
\]
又 \(\card{A_Q\triangle(A_Q\oplus h)}=2\card{A_Q}-2\card{A_Q\cap(A_Q\oplus h)}\)，故
\[
\frac1{\card{H}}\sum_{h\in H}\card{A_Q\triangle(A_Q\oplus h)}
=2N p_Q(1-p_Q).
\]
对所有陪集求和得到
\[
2\card{H}\sum_{Q\in G_R/H}p_Q(1-p_Q)\le L\varepsilon\card{G_R}.
\]
另一方面，按 \(\operatorname{Env}_H(C)\) 的定义，
\[
\card{C\triangle\operatorname{Env}_H(C)}
=\card{H}\sum_{Q\in G_R/H}\min(p_Q,1-p_Q).
\]
由于 \(\min(p,1-p)\le2p(1-p)\) 对 \(0\le p\le1\) 成立，便得
\[
\card{C\triangle\operatorname{Env}_H(C)}
\le 2\card{H}\sum_Qp_Q(1-p_Q)\le L\varepsilon\card{G_R}.
\]
平方范数估计再由第一式给出。
最后，若 \(\alpha\le p_Q\le1-\alpha\)，则 \(p_Q(1-p_Q)\ge\alpha(1-\alpha)\)。
将这些陪集的贡献代入上式即得混合陪集计数界。
由于 \(\operatorname{Env}_H(C)\) 是若干 \(H\)-陪集的并，任意 \(h\in H\) 均稳定它；相应手征向量的精确不变性随即由 \(\widetilde\tau_h\cdot\eta_E=\eta_{E\oplus h}\) 得到。证毕。
\end{proof}
