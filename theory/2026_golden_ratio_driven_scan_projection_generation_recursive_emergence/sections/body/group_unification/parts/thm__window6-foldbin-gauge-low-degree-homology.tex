\begin{theorem}[\(\{2,4\}\)-阈值完全决定低维同调：\texorpdfstring{$H_1,H_2,H^2(\cdot;\ZZ_2)$}{H1,H2,H^2} 的闭式]\label{thm:window6-foldbin-gauge-low-degree-homology}
\leanverified{paper\_window6\_foldbin\_gauge\_low\_degree\_homology}
对分辨率 \(m\ge 1\)，令
\(
\mathcal{G}^{\mathrm{bin}}_m:=\Aut(\Fold^{\mathrm{bin}}_m)
\)
为二进制区间折叠诱导的离散 fold-gauge 群，并记纤维多重度
\(
d^{\mathrm{bin}}_m(w):=\card{(\Fold^{\mathrm{bin}}_m)^{-1}(w)}
\)（\(w\in X_m\)）。
定义阈值计数
\[
N_2:=\#\{w\in X_m:\ d^{\mathrm{bin}}_m(w)\ge 2\},
\qquad
N_4:=\#\{w\in X_m:\ d^{\mathrm{bin}}_m(w)\ge 4\}.
\]
则存在自然同构
\[
H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\ \cong\ (\ZZ_2)^{N_2},
\qquad
H_2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\ \cong\ (\ZZ_2)^{\,N_4+\binom{N_2}{2}},
\]
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)\ \cong\ (\ZZ_2)^{\,N_4+\binom{N_2+1}{2}}.
\]
此外，令容量曲线
\[
C(t):=\sum_{w\in X_m}\min\{d^{\mathrm{bin}}_m(w),t\}\qquad(t\ge 0),
\]
则整数点跃迁满足 \(N_2=C(2)-C(1)\)、\(N_4=C(4)-C(3)\)，从而上述三者由 \(C(1),\dots,C(4)\) 完全恢复。
\end{theorem}
\begin{proof}
由纤维直积分解
\(
\mathcal{G}^{\mathrm{bin}}_m\cong \prod_{w\in X_m}\mathfrak{S}_{d^{\mathrm{bin}}_m(w)}
\)。
对 \(H_1\)，有
\(
H_1(G;\ZZ)\cong G^{\mathrm{ab}}
\)
且 Abel 化与有限直积交换，故
\[
H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)
\cong
\bigoplus_{w\in X_m}\mathfrak{S}_{d^{\mathrm{bin}}_m(w)}^{\mathrm{ab}}.
\]
当 \(d\ge 2\) 时 \(\mathfrak{S}_d^{\mathrm{ab}}\cong\ZZ_2\)（符号同态），当 \(d\le 1\) 时 Abel 化平凡，故得到
\(
H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\cong (\ZZ_2)^{N_2}
\)。
\par
对 \(H_2\)，对有限直积应用群同调的 K\"unneth 分解（整数系数），得到自然同构
\[
H_2\!\Bigl(\prod_{w}G_w;\ZZ\Bigr)
\cong
\Bigl(\bigoplus_{w}H_2(G_w;\ZZ)\Bigr)\ \oplus\
\Bigl(\bigoplus_{w<w'}H_1(G_w;\ZZ)\otimes H_1(G_{w'};\ZZ)\Bigr),
\]
其中 \(G_w=\mathfrak{S}_{d^{\mathrm{bin}}_m(w)}\)。
而对对称群，\(H_2(\mathfrak{S}_d;\ZZ)=0\)（\(d\le 3\)）且 \(H_2(\mathfrak{S}_d;\ZZ)\cong\ZZ_2\)（\(d\ge 4\)），并且 \(H_1(\mathfrak{S}_d;\ZZ)\cong\ZZ_2\) 当且仅当 \(d\ge 2\)。
因此第一类直和项贡献 \((\ZZ_2)^{N_4}\)，而第二类张量项对每对 \((w,w')\)（二者均满足 \(d^{\mathrm{bin}}_m(\cdot)\ge 2\)）贡献一份
\(
\ZZ_2\otimes\ZZ_2\cong\ZZ_2
\)，其对数为 \(\binom{N_2}{2}\)。
从而
\(
H_2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\cong (\ZZ_2)^{N_4+\binom{N_2}{2}}
\)。
\par
最后由泛系数定理（\(\ZZ_2\) 系数）得
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)
\cong
\mathrm{Hom}\!\bigl(H_2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ),\ZZ_2\bigr)\ \oplus\
\mathrm{Ext}\!\bigl(H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ),\ZZ_2\bigr).
\]
由于 \(H_1,H_2\) 均为 \(\ZZ_2\)-向量空间，\(\mathrm{Hom}(\ZZ_2,\ZZ_2)\cong\ZZ_2\) 且 \(\mathrm{Ext}(\ZZ_2,\ZZ_2)\cong\ZZ_2\)，故
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)
\cong
(\ZZ_2)^{\,\dim_{\ZZ_2}H_2+\dim_{\ZZ_2}H_1}
=
(\ZZ_2)^{\,N_4+\binom{N_2}{2}+N_2}
=
(\ZZ_2)^{\,N_4+\binom{N_2+1}{2}}.
\]
\par
对容量曲线：对任意整数 \(k\ge 1\) 与任意 \(d\in\NN\)，有
\(
\min(d,k)-\min(d,k-1)=\mathbf 1_{\{d\ge k\}}
\)。
对 \(w\in X_m\) 求和即得
\[
C(k)-C(k-1)=\#\{w\in X_m:\ d^{\mathrm{bin}}_m(w)\ge k\}.
\]
取 \(k=2,4\) 即得 \(N_2=C(2)-C(1)\)、\(N_4=C(4)-C(3)\)。
证毕。
\end{proof}

\begin{remark}[\texorpdfstring{$H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)$}{H^2} 的三类自然生成元]\label{rem:window6-foldbin-gauge-h2-generators}
在上述证明的分解下，二次上同调具有自然直和分解
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)
\cong
(\ZZ_2)^{N_4}\ \oplus\ (\ZZ_2)^{\binom{N_2}{2}}\ \oplus\ (\ZZ_2)^{N_2},
\]
三类直和分量分别对应：单纤维 Schur 乘子类（来自 \(d^{\mathrm{bin}}_m(w)\ge 4\) 的对称群因子）、跨纤维配对类（来自 \(H_1\otimes H_1\) 的 K\"unneth 项），以及由
\(\mathrm{Ext}(H_1,\ZZ_2)\)
给出的 Bockstein 型类。
\end{remark}
