\begin{theorem}[Joukowsky--Gödel 椭圆边界的双尺度对数平均恒等式]\label{thm:group-jg-elliptic-two-scale-stokes-mean}
\leanverified{paper\_group\_jg\_elliptic\_two\_scale\_stokes\_mean}
在定理 \ref{thm:group-jg-pullback-factorization} 的记号下，取 \(K=\CC\) 且令 \(r>1\)。
设 \(P(z)\in\CC[z]\) 次数为 \(n\ge 1\)，并满足 \(P(0)\neq 0\)。
令
\[
J_r(e^{i\theta})=r e^{i\theta}+r^{-1}e^{-i\theta},
\qquad
E_r:=J_r(\{e^{i\theta}:\theta\in[0,2\pi)\}),
\]
并令 \(\mu_r\) 为 Haar 测度 \(\frac{\dd\theta}{2\pi}\) 在 \(J_r\) 下的推前测度。
定义半径 \(\rho>0\) 的圆周对数平均泛函
\[
\mathcal M_\rho(P):=\frac1{2\pi}\int_{0}^{2\pi}\log|P(\rho e^{i\theta})|\,\dd\theta.
\]
则成立严格恒等式
\[
\boxed{\;
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)
=
\mathcal M_{1}(P)+\mathcal M_{r^{-2}}(P)+n\log r.
\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:group-jg-pullback-factorization}，对 \(|z|=1\) 有
\[
r^n z^n Q_r\!\left(r z+r^{-1}z^{-1}\right)=P(z)\,P^{\vee}(r^2 z).
\]
取 \(z=e^{i\theta}\) 并取对数绝对值，注意 \(\log|z^n|=0\)，得
\[
\log\bigl|Q_r(J_r(e^{i\theta}))\bigr|
=
\log|P(e^{i\theta})|+\log|P^\vee(r^2 e^{i\theta})|-n\log r.
\]
对 \(\theta\in[0,2\pi)\) 取平均，并以 \(\mu_r\) 的定义将左侧识别为 \(\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)\)，得到
\[
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)
=\mathcal M_1(P)+\frac1{2\pi}\int_{0}^{2\pi}\log|P^\vee(r^2 e^{i\theta})|\,\dd\theta-n\log r.
\]
另一方面，由 \(P^\vee(u)=u^nP(1/u)\) 得
\[
P^\vee(r^2 e^{i\theta})=r^{2n}e^{in\theta}P(r^{-2}e^{-i\theta}),
\qquad
\bigl|P^\vee(r^2 e^{i\theta})\bigr|=r^{2n}\bigl|P(r^{-2}e^{-i\theta})\bigr|.
\]
于是
\[
\frac1{2\pi}\int_{0}^{2\pi}\log|P^\vee(r^2 e^{i\theta})|\,\dd\theta
=2n\log r+\mathcal M_{r^{-2}}(P),
\]
代回即得结论。
\end{proof}

\begin{theorem}[Joukowsky 椭圆的双变量指数矩母函数闭式]\label{thm:group-jg-joukowsky-bivariate-mgf-bessel}
在定理 \ref{thm:group-jg-elliptic-two-scale-stokes-mean} 的记号下，令 \(r>1\)，并记
\[
J_r(e^{i\theta})=r e^{i\theta}+r^{-1}e^{-i\theta},
\qquad
E_r:=J_r(\TT),
\qquad
\mu_r:=J_{r\ast}\!\left(\frac{\dd\theta}{2\pi}\right).
\]
对 \(u,v\in\CC\) 定义双变量指数矩母函数
\[
M_r(u,v):=\int_{E_r}\exp\bigl(u w+v\overline w\bigr)\,\dd\mu_r(w).
\]
则成立闭式表达
\[
\boxed{\;
M_r(u,v)=I_0\!\Bigl(2\sqrt{\,u^2+v^2+(r^2+r^{-2})uv\,}\Bigr),
\;}
\]
其中 \(I_0\) 为零阶第一类修正 Bessel 函数。
特别地：
\begin{enumerate}
  \item 取 \(v=0\) 得 \(M_r(u,0)=I_0(2u)\)，与 \(r\) 无关；
  \item \(r\) 的影响仅通过标量 \(t=r^2+r^{-2}\) 进入，故全部混合矩
  \(\int_{E_r} w^a\overline w^{\,b}\,\dd\mu_r(w)\)
  都是 \(t\) 的多项式（系数仅依赖于 \((a,b)\)）。
\end{enumerate}
\end{theorem}
\begin{proof}
取 \(z=e^{i\theta}\)。则
\[
w=r z+r^{-1}z^{-1},\qquad \overline w=r z^{-1}+r^{-1}z,
\]
从而
\[
u w+v\overline w
=(ur+vr^{-1})z+(ur^{-1}+vr)z^{-1}.
\]
记 \(A:=ur+vr^{-1}\)、\(B:=ur^{-1}+vr\)，则
\[
M_r(u,v)=\frac1{2\pi}\int_0^{2\pi}\exp\!\bigl(Ae^{i\theta}+Be^{-i\theta}\bigr)\,\dd\theta.
\]
标准恒等式给出
\[
\frac1{2\pi}\int_0^{2\pi}\exp\!\bigl(Ae^{i\theta}+Be^{-i\theta}\bigr)\,\dd\theta
=I_0\!\bigl(2\sqrt{AB}\bigr),
\]
而
\[
AB=(ur+vr^{-1})(ur^{-1}+vr)=u^2+v^2+(r^2+r^{-2})uv,
\]
代回即得主公式。
当 \(v=0\) 时 \(AB=u^2\)，得到 \(M_r(u,0)=I_0(2u)\)。
最后由 \(M_r\) 在 \((u,v)\) 的 Taylor 展开系数即为混合矩，且主公式中 \(r\) 仅通过 \(t=r^2+r^{-2}\) 出现，故混合矩为 \(t\) 的多项式。证毕。
\end{proof}

\begin{corollary}[椭圆边界对数平均的乘法分解]\label{cor:group-jg-elliptic-two-scale-mahler-multiplicative}
在定理 \ref{thm:group-jg-elliptic-two-scale-stokes-mean} 的条件下，
\[
\exp\!\left(\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)\right)
=
r^n\,\exp(\mathcal M_1(P))\,\exp(\mathcal M_{r^{-2}}(P)).
\]
\end{corollary}
\begin{proof}
对定理 \ref{thm:group-jg-elliptic-two-scale-stokes-mean} 两侧取指数即可。
\end{proof}

\begin{theorem}[Lee--Yang 椭圆输运的 Mahler--自由能恒等式]\label{thm:group-jg-leyang-mahler-free-energy}
在定理 \ref{thm:group-jg-pullback-factorization} 的记号下，取 \(K=\CC\) 且令 \(r>1\)。
设 \(P(z)\in\CC[z]\) 次数为 \(n\ge 1\)，并满足 Lee--Yang 条件：其全部零点满足 \(|z_j|=1\)。
令
\[
J_r(e^{i\theta})=r e^{i\theta}+r^{-1}e^{-i\theta},
\qquad
E_r:=J_r(\{e^{i\theta}:\theta\in[0,2\pi)\}),
\]
并令 \(\mu_r\) 为 Haar 测度 \(\frac{\dd\theta}{2\pi}\) 在 \(J_r\) 下的推前测度。
则成立严格恒等式
\[
\boxed{\;
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)=\log|a_n a_0|+n\log r\; }.
\]
特别地，若 \(P\) 首一，则
\[
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)=n\log r.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:group-jg-pullback-factorization}，对 \(|z|=1\) 有
\[
r^n z^n Q_r\!\left(r z+r^{-1}z^{-1}\right)=P(z)\,P^{\vee}(r^2 z).
\]
取对数绝对值并对 \(|z|=1\) 以 \(\frac{\dd\theta}{2\pi}\) 积分，注意 \(\log|z|=0\)，得
\[
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)
=\int_{0}^{2\pi}\log|P(e^{i\theta})|\,\frac{\dd\theta}{2\pi}
\ +\ \int_{0}^{2\pi}\log|P^{\vee}(r^2 e^{i\theta})|\,\frac{\dd\theta}{2\pi}
\ -\ n\log r.
\]
由于 \(P\) 的全部零点位于单位圆，Jensen 公式给出
\[
\int_{0}^{2\pi}\log|P(e^{i\theta})|\,\frac{\dd\theta}{2\pi}=\log|a_n|.
\]
另一方面，\(P^\vee(u)=u^n P(1/u)\) 的首项系数为 \(a_0\)，且 \(P^\vee(r^2 z)\) 的零点为 \((r^2 z_j)^{-1}\)，其模长皆为 \(r^{-2}<1\)。
再由 Jensen 公式得
\[
\int_{0}^{2\pi}\log|P^{\vee}(r^2 e^{i\theta})|\,\frac{\dd\theta}{2\pi}
=\log|a_0 r^{2n}|=\log|a_0|+2n\log r.
\]
代回即得所述恒等式。证毕。
\end{proof}

\leanverified{paper\_group\_jg\_leyang\_mahler\_free\_energy}

\begin{corollary}[椭圆输运自由能的线性响应]\label{cor:group-jg-leyang-free-energy-linear-flux}
在定理 \ref{thm:group-jg-leyang-mahler-free-energy} 的条件下，定义
\[
\mathcal F(r):=\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w).
\]
则 \(\mathcal F(r)=\log|a_n a_0|+n\log r\)，从而
\[
\frac{\dd \mathcal F}{\dd(\log r)}\equiv n.
\]
\leanverified{paper\_group\_jg\_leyang\_free\_energy\_linear\_flux}
\end{corollary}

\begin{proof}
由定理 \ref{thm:group-jg-leyang-mahler-free-energy} 直接得到表达式并对 \(\log r\) 求导。证毕。
\end{proof}

\begin{theorem}[Lee--Yang 椭圆上的倍角动力学与二进制符号化]\label{thm:group-jg-leyang-ellipse-doubling-shift}
取 \(r>1\)，令 \(E_r\) 与参数化 \(w(\theta)=r e^{i\theta}+r^{-1}e^{-i\theta}\) 如上。
定义映射
\[
T_r:E_r\to E_r,\qquad T_r(w(\theta)):=w(2\theta)\qquad(\theta\in\RR/2\pi\ZZ).
\]
则：
\begin{enumerate}
  \item \(T_r\) 与圆周倍角映射 \(D:\theta\mapsto 2\theta\ (\mathrm{mod}\ 2\pi)\) 拓扑共轭，故 \(h_{\mathrm{top}}(T_r)=\log 2\)；
  \item \(\mu_r\) 为 \(T_r\) 的不变概率测度；
  \item 记 \(h(\theta)=w(\theta)\)，则对 \(\mu_r=h_\ast(\frac{\dd\theta}{2\pi})\) 的 Lyapunov 指数满足
  \[
  \chi(T_r,\mu_r):=\int_{E_r}\log|DT_r|\,\dd\mu_r=\log 2.
  \]
  \item 除去 dyadic 角所对应的零测集后，存在可测同构把 \(D\) 识别为二进制左移，因此 \(T_r\) 在 \((E_r,\mu_r)\) 上与 \(\{0,1\}^{\NN}\) 的左移系统测度同构。
\end{enumerate}
\end{theorem}

\begin{proof}
由于 \(\theta\mapsto w(\theta)\) 为 \(\RR/2\pi\ZZ\) 到 \(E_r\) 的连续双射且 \(E_r\) 紧，故为同胚 \(h:\TT\to E_r\)。
由定义 \(T_r\circ h=h\circ D\)，故拓扑共轭成立，特别地 \(h_{\mathrm{top}}(T_r)=h_{\mathrm{top}}(D)=\log 2\)。
\par
\(\mu_r=h_\ast(\frac{\dd\theta}{2\pi})\) 的不变性由
\(
D_\ast(\frac{\dd\theta}{2\pi})=\frac{\dd\theta}{2\pi}
\)
推出。
\par
在 \(\theta\) 坐标下，链式法则给出
\[
DT_r\bigl(h(\theta)\bigr)=\frac{h'(2\theta)\cdot 2}{h'(\theta)}.
\]
因此
\[
\int_{E_r}\log|DT_r|\,\dd\mu_r
=\int_{0}^{2\pi}\left(\log 2+\log|h'(2\theta)|-\log|h'(\theta)|\right)\frac{\dd\theta}{2\pi}
=\log 2,
\]
其中最后一步使用 \(\theta\mapsto 2\theta\) 对 Haar 测度的保持性。
最后一条为倍角映射与二进制左移的经典符号化（去除二进制展开二义性的 dyadic 集）。证毕。
\end{proof}
