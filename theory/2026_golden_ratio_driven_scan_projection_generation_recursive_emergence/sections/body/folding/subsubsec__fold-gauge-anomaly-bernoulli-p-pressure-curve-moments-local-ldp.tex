\paragraph{周期曲线的三重覆盖、二阶矩闭式与局部大偏差预因子}
本段在定理 \ref{thm:fold-bernoulli-p-pressure-cycle-equation} 与定理 \ref{thm:fold-bernoulli-p-ldp-compressed-cubic-parametrization} 的符号体系下，
把“周期方程 \(\Phi_p(u,z)=1\)”抽象为一条三次代数曲线，并给出再生块二阶矩与局部大偏差预因子的闭式输入。

\begin{proposition}[压力周期曲线的三重覆盖模型与分支除子]\label{prop:fold-bernoulli-p-pressure-cycle-trigonal-curve-branch}
\leanverified{paper\_fold\_bernoulli\_p\_pressure\_cycle\_trigonal\_curve\_branch}
取 \(p\in(0,1)\)，记 \(\kappa:=\frac{p}{1-p}\) 并令
\[
C(z):=1-(1-p)z-p(1-p)z^{2}.
\]
考虑仿射代数曲线 \(\mathcal X_p\subset \mathbb A^2_{z,w}\)：
\begin{equation}\label{eq:fold-bernoulli-p-pressure-cycle-trigonal-curve}
w^{3}+\kappa\,z\,C(z)\,w-\kappa\,C(z)=0.
\end{equation}
则成立：
\begin{enumerate}
  \item 在开集 \(z\neq 0\) 上，曲线 \(\mathcal X_p\) 与周期方程 \(\Phi_p(u,z)=1\) 的解集双有理等价；
  显式变换为
  \[
  w=puz,\qquad u=\frac{w}{pz}.
  \]
  \item 令 \(\widetilde{\mathcal X}_p\) 为 \(\mathcal X_p\) 的规范化，投影
  \(
  \pi:\widetilde{\mathcal X}_p\to\PP^1_z
  \)
  为次数 \(3\) 覆盖，其 \(w\)-判别式为
  \begin{equation}\label{eq:fold-bernoulli-p-pressure-cycle-trigonal-discriminant}
  \Delta(z)=-\kappa^{2}\,C(z)^{2}\bigl(4\kappa z^{3}C(z)+27\bigr).
  \end{equation}
  因而有限分支值由两类点组成：\(C(z)=0\) 的两点（在其上为全分歧，分歧指数 \(e=3\)），以及
  \(4\kappa z^{3}C(z)+27=0\) 的五个点（按重数计，在其上为简单分歧，\(e=2\)）。
  \item 在无穷远处，\(\pi^{-1}(\infty)\) 含两点，其中一处不分歧（\(e=1\)），另一处为简单分歧（\(e=2\)）。
  于是总分歧量满足
  \[
  \sum_{P\in\widetilde{\mathcal X}_p}(e_P-1)=2\cdot(3-1)+5\cdot(2-1)+1\cdot(2-1)=10,
  \]
  从而由 Riemann--Hurwitz 公式得到
  \begin{equation}\label{eq:fold-bernoulli-p-pressure-cycle-trigonal-genus}
  g(\widetilde{\mathcal X}_p)=3.
  \end{equation}
\end{enumerate}
\end{proposition}
\begin{proof}
由定理 \ref{thm:fold-bernoulli-p-pressure-cycle-equation}，周期方程 \(\Phi_p(u,z)=1\) 等价于
\[
C(z)\bigl(1-puz^{2}\bigr)=p^{2}(1-p)u^{3}z^{3}.
\]
令 \(v:=puz^{2}\) 并设 \(w:=v/z=puz\)，即得到
\[
(1-p)w^{3}+p z C(z)\,w-p C(z)=0,
\]
两边除以 \(1-p\) 即为 \eqref{eq:fold-bernoulli-p-pressure-cycle-trigonal-curve}，并给出 \(z\neq 0\) 上的双有理变换。
\par
把 \eqref{eq:fold-bernoulli-p-pressure-cycle-trigonal-curve} 视作关于 \(w\) 的三次方程 \(w^{3}+a(z)w+b(z)=0\)，其中
\(
a(z)=\kappa z C(z)
\)、\(
b(z)=-\kappa C(z)
\)。
其 \(w\)-判别式为 \(\Delta(z)=-4a(z)^{3}-27b(z)^{2}\)，代入即得 \eqref{eq:fold-bernoulli-p-pressure-cycle-trigonal-discriminant}。
当 \(C(z)=0\) 时，方程退化为 \(w^{3}=0\)，故为三重根并对应全分歧点（\(e=3\)）。
当 \(4\kappa z^{3}C(z)+27=0\) 且 \(C(z)\neq 0\) 时，判别式为单零点，三次方程出现一对并合根，故为简单分歧（\(e=2\)）。
\par
对 \(z=\infty\)，令 \(t=1/z\) 并把方程乘以 \(t^{3}\) 写成
\[
t^{3}w^{3}-p^{2}w+p^{2}t+O(t^{2})=0\qquad(t\to 0),
\]
其中用到 \(C(1/t)=-p(1-p)t^{-2}-(1-p)t^{-1}+1\) 与 \(\kappa=\frac{p}{1-p}\)。
该形式给出一条解析分支 \(w=t+O(t^{2})\)（故 \(e=1\)），以及一条 Puiseux 分支 \(w\sim \pm p\,t^{-3/2}\)，其自然局部参数可取 \(s=\sqrt t\)，从而 \(z=t^{-1}=s^{-2}\) 给出 \(e=2\) 的简单分歧。
于是无穷远处的分歧贡献为 \(1\)。
\par
将全部分歧贡献代入 Riemann--Hurwitz 公式
\(
2g-2=3\cdot(-2)+\sum(e_P-1)
\)
即可得到 \eqref{eq:fold-bernoulli-p-pressure-cycle-trigonal-genus}。证毕。
\end{proof}

\begin{proposition}[周期倾斜下再生块二阶矩的闭式输入]\label{prop:fold-bernoulli-p-cycle-tilt-second-moments-closed}
\leanverified{paper\_fold\_bernoulli\_p\_cycle\_tilt\_second\_moments\_closed}
沿用定理 \ref{thm:fold-bernoulli-p-pressure-cycle-equation} 的周期倾斜测度 \(\PP_\theta\) 与定理 \ref{thm:fold-bernoulli-p-ldp-compressed-cubic-parametrization} 的记号。
对 \(\theta\in\RR\) 令 \(u=e^\theta\)、\(z=e^{-P_{G,p}(\theta)}\)，并设
\[
v:=puz^{2},\qquad
A:=1-(1-p)z,\qquad
B:=1-v,\qquad
C:=C(z).
\]
则在 \(\PP_\theta\) 下，随机变量族 \((M,(K_j)_{j\ge 0},(N_i)_{i\ge 1})\) 仍相互独立且分别为几何分布，并且在周期约束 \(\Phi_p(u,z)=1\) 下有简化
\begin{equation}\label{eq:fold-bernoulli-p-cycle-tilt-qM-CoverA}
\PP_\theta(M=m)\ \propto\ \left(\frac{C}{A}\right)^{m}\qquad(m\ge 0).
\end{equation}
进一步有
\[
\EE_\theta[M]=\frac{C}{p(1-p)z^{2}},\qquad
\Var_\theta(M)=\frac{C\,A}{p^{2}(1-p)^{2}z^{4}},
\]
\[
\EE_\theta[K]=\frac{(1-p)z}{A},\quad
\Var_\theta(K)=\frac{(1-p)z}{A^{2}},\qquad
\EE_\theta[N]=\frac{v}{B},\quad
\Var_\theta(N)=\frac{v}{B^{2}}.
\]
令再生块奖励与长度为 \((X_0,L_0)\)（定理 \ref{thm:fold-bernoulli-p-regeneration-bivariate-pgf}），并记
\(
s(\theta)=P'_{G,p}(\theta)=\EE_\theta[X_0]/\EE_\theta[L_0]
\)。
则一、二阶矩与协方差满足闭式
\[
\EE_\theta[X_0]=\EE_\theta[M]\Bigl(3+\EE_\theta[N]\Bigr),
\qquad
\EE_\theta[L_0]=2+\bigl(\EE_\theta[M]+1\bigr)\EE_\theta[K]+\EE_\theta[M]\Bigl(3+2\EE_\theta[N]\Bigr),
\]
\[
\Var_\theta(X_0)=\EE_\theta[M]\Var_\theta(N)+\Var_\theta(M)\bigl(3+\EE_\theta[N]\bigr)^{2},
\]
\[
\Var_\theta(L_0)=\bigl(\EE_\theta[M]+1\bigr)\Var_\theta(K)+4\,\EE_\theta[M]\Var_\theta(N)
\;+\;\Var_\theta(M)\bigl(\EE_\theta[K]+2\EE_\theta[N]+3\bigr)^{2},
\]
\[
\Cov_\theta(X_0,L_0)=2\,\EE_\theta[M]\Var_\theta(N)
\;+\;\Var_\theta(M)\bigl(3+\EE_\theta[N]\bigr)\bigl(\EE_\theta[K]+2\EE_\theta[N]+3\bigr).
\]
从而压力二阶导数具有完全闭式输入
\begin{equation}\label{eq:fold-bernoulli-p-cycle-tilt-second-derivative-variance-expanded}
P''_{G,p}(\theta)
=\frac{\Var_\theta\!\left(X_0-s(\theta)L_0\right)}{\EE_\theta[L_0]}
=\frac{\Var_\theta(X_0)-2s(\theta)\Cov_\theta(X_0,L_0)+s(\theta)^{2}\Var_\theta(L_0)}{\EE_\theta[L_0]}.
\end{equation}
\end{proposition}
\begin{proof}
周期倾斜下的几何闭包性与 \(K,N,M\) 的倾斜参数见命题 \ref{prop:fold-bernoulli-p-cycle-tilt-geometric-closure}。
当 \(\Phi_p(u,z)=1\) 时，由定理 \ref{thm:fold-bernoulli-p-ldp-compressed-cubic-parametrization} 的推导可将 \(q_M\) 化简为 \(q_M=C/A\)，从而得到 \eqref{eq:fold-bernoulli-p-cycle-tilt-qM-CoverA}。
各几何分布的均值与方差为标准公式。
\par
再由分解式 \eqref{eq:fold-bernoulli-p-regeneration-decomposition}，
\[
X_0=\sum_{i=1}^{M}(3+N_i),\qquad
L_0=2+\sum_{j=0}^{M}K_j+\sum_{i=1}^{M}(3+2N_i),
\]
对固定 \(M\) 条件下利用独立性逐项求矩；
并对随机和使用恒等式
\(
\Var\!\left(\sum_{i=1}^{M}Y_i\right)=\EE[M]\Var(Y)+\Var(M)\EE[Y]^{2}
\)
以及
\(
\Cov\!\left(\sum_{i=1}^{M}Y_i,\sum_{i=1}^{M}Z_i\right)=\EE[M]\Cov(Y,Z)+\Var(M)\EE[Y]\EE[Z]
\)
即可得到所列闭式。
最后式 \eqref{eq:fold-bernoulli-p-cycle-tilt-second-derivative-variance-expanded} 由定理 \ref{thm:fold-bernoulli-p-pressure-cycle-equation} 的
\eqref{eq:fold-bernoulli-p-pressure-second-derivative-cycle-tilt}
代入并展开得出。证毕。
\end{proof}

\begin{theorem}[有限时刻矩母函数的谱主项与闭式预因子]\label{thm:fold-bernoulli-p-finite-time-mgf-pf-prefactor}
\leanverified{paper\_fold\_bernoulli\_p\_finite\_time\_mgf\_pf\_prefactor}
取 \(p\in(0,1)\) 与 \(u>0\)。
令 \(Q_p(u)\) 为倾斜核，并记其 Perron 特征值为 \(\rho(u;p)>0\)。
从再生态 \(d\) 出发，令
\(
M_m(u)=\EE[u^{G_m}\mid S_0=d]
\)。
则存在 \(\gamma(u;p)\in(0,\rho(u;p))\) 与常数 \(A_p(u)\) 使得
\begin{equation}\label{eq:fold-bernoulli-p-finite-time-mgf-pf-asymptotic}
M_m(u)=A_p(u)\,\rho(u;p)^{m}+O\!\left(\gamma(u;p)^{m}\right)\qquad(m\to\infty).
\end{equation}
并且 \(A_p(u)\) 可写成 \(\rho=\rho(u;p)\) 与 \(u\) 的完全闭式有理函数：
\begin{equation}\label{eq:fold-bernoulli-p-finite-time-mgf-pf-prefactor-closed}
\boxed{\;
A_p(u)=
\frac{(\rho^2-pu)\Bigl[\rho^3+\bigl(p^2(u-1)+p\bigr)\rho^2+p\,u(pu-1)\rho-p^2(1-p)u\Bigr]}
{\rho^6+p(1-p-2u)\rho^4+2p^2(1-p)u^3\rho^3+p^2\bigl(u^2-2u(1-p)\bigr)\rho^2+p^3(1-p)u^2}\; }.
\end{equation}
特别地，\(A_p(1)=1\)。
\end{theorem}
\begin{proof}
由于 \(Q_p(u)\) 为不可约非负矩阵，Perron--Frobenius 定理给出主特征值 \(\rho(u;p)\) 的单性与严格谱隙。
取对应左右 Perron 特征向量 \(r,\ell\) 并规范化使 \(\ell^{\mathsf T}r=1\)，则有谱投影分解
\[
Q_p(u)^{m}=\rho(u;p)^{m}\,r\ell^{\mathsf T}+O\!\left(\gamma(u;p)^{m}\right),
\qquad \gamma(u;p)<\rho(u;p).
\]
左乘 \(e_d^{\mathsf T}\)、右乘 \(\mathbf 1\) 得 \eqref{eq:fold-bernoulli-p-finite-time-mgf-pf-asymptotic}，其中
\(
A_p(u)=(e_d^{\mathsf T}r)(\ell^{\mathsf T}\mathbf 1)
\)。
将定理 \ref{thm:fold-bernoulli-p-doob-transform-closed} 给出的 \(r,\ell\) 的闭式代入并作代数化简，即得到 \eqref{eq:fold-bernoulli-p-finite-time-mgf-pf-prefactor-closed}。
最后令 \(u=1\) 时 \(M_m(1)\equiv 1\) 与 \(\rho(1;p)=1\)，故 \(A_p(1)=1\)。证毕。
\end{proof}

\begin{theorem}[精确局部大偏差：点质量的 \texorpdfstring{$m^{-1/2}$}{m^{-1/2}} 预因子闭式]\label{thm:fold-bernoulli-p-local-ldp-prefactor}
\leanverified{paper\_fold\_bernoulli\_p\_local\_ldp\_prefactor}
固定 \(p\in(0,1)\)。
取任意 \(s\in(0,1)\)，令 \(\theta_s\) 为方程
\(
P'_{G,p}(\theta_s)=s
\)
的唯一实解，并记 \(u_s=e^{\theta_s}\)、\(\rho_s=\rho(u_s;p)\)。
令整数序列 \(k_m\) 满足 \(k_m/m\to s\)。
则当 \(m\to\infty\) 时成立局部大偏差的精确渐近
\begin{equation}\label{eq:fold-bernoulli-p-local-ldp-asymptotic}
\PP(G_m=k_m\mid S_0=d)
=
\frac{A_p(u_s)}{u_s^{k_m}}\,
\frac{\rho_s^{\,m}}{\sqrt{2\pi m\,P''_{G,p}(\theta_s)}}\,
\bigl(1+o(1)\bigr),
\end{equation}
其中 \(A_p(u_s)\) 由定理 \ref{thm:fold-bernoulli-p-finite-time-mgf-pf-prefactor} 给出闭式，
而 \(P''_{G,p}(\theta_s)\) 可由命题 \ref{prop:fold-bernoulli-p-cycle-tilt-second-moments-closed} 的二阶矩闭式输入得到。
\end{theorem}
\begin{proof}
由 Cauchy 系数公式，
\[
\PP(G_m=k\mid S_0=d)=[u^k]M_m(u)=\frac{1}{2\pi i}\oint M_m(u)\,u^{-k-1}\,\dd u,
\]
积分环绕正实轴附近的解析邻域。
将定理 \ref{thm:fold-bernoulli-p-finite-time-mgf-pf-prefactor} 的谱主项
\(
M_m(u)=A_p(u)\rho(u;p)^m+O(\gamma(u;p)^m)
\)
代入，主导相位函数为
\(
\Psi(u)=m\log\rho(u;p)-k\log u
\)。
其临界点由 \(\Psi'(u)=0\) 给出，等价于在 \(\theta=\log u\) 下满足 \(P'_{G,p}(\theta)=k/m\)；
当 \(k_m/m\to s\) 时临界点收敛到 \(u_s\)。
在该点处二阶项满足 \(\Psi''(u_s)=mP''_{G,p}(\theta_s)/u_s^2>0\)，故鞍点法给出高斯型 \(\sqrt{m^{-1}}\) 预因子；
而误差项因 \(\gamma(u;p)<\rho(u;p)\) 具有指数衰减，对系数贡献可忽略。
于是得到 \eqref{eq:fold-bernoulli-p-local-ldp-asymptotic}。证毕。
\end{proof}
