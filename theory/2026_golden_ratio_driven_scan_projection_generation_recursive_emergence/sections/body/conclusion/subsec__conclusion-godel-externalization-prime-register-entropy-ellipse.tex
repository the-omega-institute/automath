\subsection{哥德尔化外置的范畴模型、素数寄存器极值律与椭圆同构}\label{subsec:conclusion-godel-externalization-prime-register-entropy-ellipse}
在扩展保存的语义中，不可逆读出通过外置记录轴提升为扩展空间上的单射。本小节在不复述既有结构定理的前提下，给出交换幺半群口径下的哥德尔化外置初对象、复合投影分解、纤维最小账本代价、\(\zeta\)-正则化素数独立律，以及单位圆--椭圆--素数寄存器之间的可加几何同构。

\begin{definition}[规范化交换幺半群外置]\label{def:conclusion-cmon-externalization}
设 \(\pi:\Omega\twoheadrightarrow X\) 为满射，\(s:X\to\Omega\) 为截面（\(\pi\circ s=\mathrm{id}_X\)）。
称 \((R,r)\) 为 \((\pi,s)\) 的规范化交换幺半群外置，若
\begin{enumerate}
  \item \(R\) 为交换幺半群，单位元记为 \(1_R\)；
  \item \(r:\Omega\to R\) 满足 \(r(s(x))=1_R\)（\(\forall x\in X\)）；
  \item 扩展读出
  \[
  \pi^e:=(\pi,r):\Omega\to X\times R
  \]
  为单射。
\end{enumerate}
记所有此类对象构成的范畴为
\[
\mathsf{Ext}^{\mathrm{cmon}}(\pi,s),
\]
其态射 \((R,r)\to(R',r')\) 为交换幺半群同态 \(\varphi:R\to R'\) 且 \(r'=\varphi\circ r\)。
\end{definition}

\begin{theorem}[哥德尔化外置的普适素数寄存器]\label{thm:conclusion-cmon-externalization-initial-prime-register}
\leanverified{paper\_conclusion\_cmon\_externalization\_initial\_prime\_register}
沿用定义 \ref{def:conclusion-cmon-externalization} 的记号。则：
\begin{enumerate}
  \item 范畴 \(\mathsf{Ext}^{\mathrm{cmon}}(\pi,s)\) 存在初对象 \((\mathcal P_0,r_0)\)；
  \item 若 \(S:=\Omega\setminus s(X)\) 可数，则 \(\mathcal P_0\) 可单射嵌入 \((\NN_{\ge 1},\times)\)，从而得到严格意义上的素数寄存器外置实现。
\end{enumerate}
\end{theorem}
\begin{proof}
令
\[
S:=\Omega\setminus s(X).
\]
取 \(\mathcal P_0:=\NN^{(S)}\)（\(S\) 上有限支撑指数向量），并采用乘法记号：
对 \(\alpha,\beta\in\NN^{(S)}\) 定义
\[
(\alpha\cdot\beta)(\omega):=\alpha(\omega)+\beta(\omega),\qquad 1_{\mathcal P_0}:=0.
\]
记 \(e_\omega\in\NN^{(S)}\) 为 \(\omega\in S\) 的基向量。定义
\[
r_0(s(x)):=1_{\mathcal P_0},\qquad
r_0(\omega):=e_\omega\ \ (\omega\in S).
\]

先证 \((\pi,r_0)\) 单射。设 \(\omega,\omega'\in\Omega\) 满足
\[
\pi(\omega)=\pi(\omega'),\qquad r_0(\omega)=r_0(\omega').
\]
若 \(r_0(\omega)=1_{\mathcal P_0}\)，则 \(\omega=s(x),\ \omega'=s(x')\)，且
\[
x=\pi(\omega)=\pi(\omega')=x',
\]
故 \(\omega=\omega'\)。
若 \(r_0(\omega)=e_u\)（\(u\in S\)），则同理 \(r_0(\omega')=e_u\) 推得 \(\omega=\omega'=u\)。
故 \((\pi,r_0)\) 单射，且 \(r_0\circ s=1_{\mathcal P_0}\)，从而 \((\mathcal P_0,r_0)\) 是对象。

再证初对象性质。任取 \((R,r)\in\mathsf{Ext}^{\mathrm{cmon}}(\pi,s)\)。
对生成元定义
\[
\varphi(e_\omega):=r(\omega)\quad(\omega\in S),\qquad \varphi(1_{\mathcal P_0})=1_R.
\]
由自由交换幺半群泛性质，\(\varphi\) 唯一延拓为幺半群同态 \(\varphi:\mathcal P_0\to R\)。
并且对 \(s(x)\) 与 \(\omega\in S\) 分别有
\[
r(s(x))=1_R=\varphi(1_{\mathcal P_0})=\varphi(r_0(s(x))),
\]
\[
r(\omega)=\varphi(e_\omega)=\varphi(r_0(\omega)).
\]
故 \(r=\varphi\circ r_0\)，且该 \(\varphi\) 唯一，\((\mathcal P_0,r_0)\) 为初对象。

若 \(S\) 可数，取单射 \(\iota:S\hookrightarrow\mathbb P\)（\(\mathbb P\) 为素数集），定义
\[
\Gamma:\mathcal P_0\to\NN_{\ge 1},\qquad
\Gamma(\alpha):=\prod_{\omega\in S}\iota(\omega)^{\alpha(\omega)}.
\]
由算术基本定理，\(\Gamma\) 为单射幺半群同态。
令 \(r_G:=\Gamma\circ r_0\)，则
\[
(\pi,r_G):\Omega\hookrightarrow X\times \NN_{\ge 1}
\]
为单射，得到素数幂坐标形式的哥德尔化外置。证毕。
\end{proof}

\begin{corollary}[复合投影的外置链式分解与素数轴可分配性]\label{cor:conclusion-composite-projection-externalization-factorization}
设
\[
\Omega\xrightarrow{\pi}X\xrightarrow{\tau}Y
\]
为满射复合，\(s_X:Y\to X\) 为 \(\tau\) 的截面，\(s_\Omega:X\to\Omega\) 为 \(\pi\) 的截面，并记
\[
s:=s_\Omega\circ s_X:Y\to\Omega.
\]
则定理 \ref{thm:conclusion-cmon-externalization-initial-prime-register} 的初对象满足自然同构
\[
\mathcal P_0(\tau\circ\pi,s)
\cong
\mathcal P_0(\pi,s_\Omega)\times \mathcal P_0(\tau,s_X).
\]
若两侧生成集可数，可选互不相交素数子集实现素数轴分配，使复合外置对应的哥德尔整数分解为两个互素因子。
\end{corollary}
\begin{proof}
记
\[
S_\pi:=\Omega\setminus s_\Omega(X),\qquad
S_\tau:=X\setminus s_X(Y),
\]
\[
S:=\Omega\setminus s(Y)=\Omega\setminus s_\Omega(s_X(Y)).
\]
则有不交并分解
\[
S=S_\pi\sqcup s_\Omega(S_\tau).
\]
证明如下：若 \(\omega\notin s_\Omega(X)\)，则 \(\omega\in S_\pi\subset S\)；
若 \(\omega=s_\Omega(x)\in s_\Omega(X)\)，则 \(\omega\in S\iff x\notin s_X(Y)\iff x\in S_\tau\iff \omega\in s_\Omega(S_\tau)\)。
故分解成立且不交。

对自由交换幺半群有标准同构
\[
\NN^{(A\sqcup B)}\cong \NN^{(A)}\times \NN^{(B)}.
\]
代入 \(A=S_\pi,\ B=s_\Omega(S_\tau)\)，并用 \(s_\Omega:S_\tau\to s_\Omega(S_\tau)\) 的双射得结论。

若 \(S_\pi,S_\tau\) 可数，取互不相交素数集 \(\mathbb P_\pi,\mathbb P_\tau\subset\mathbb P\) 及单射
\[
\iota_\pi:S_\pi\hookrightarrow\mathbb P_\pi,\qquad
\iota_\tau:S_\tau\hookrightarrow\mathbb P_\tau.
\]
对应哥德尔整数因子分别仅含 \(\mathbb P_\pi\) 与 \(\mathbb P_\tau\) 的素因子，故互素。证毕。
\end{proof}

\begin{theorem}[纤维内最小哥德尔代价与调和 KL 分解]\label{thm:conclusion-fiber-minimal-godel-cost-harmonic-kl}
\leanverified{paper\_conclusion\_fiber\_minimal\_godel\_cost\_harmonic\_kl}
设 \(A\) 为有限集，\(|A|=m\ge 1\)，\(p\) 为 \(A\) 上概率分布。
对任意单射编码 \(c:A\hookrightarrow \NN_{\ge 1}\)，定义
\[
J(c;p):=\EE_{a\sim p}\!\bigl[\log c(a)\bigr].
\]
记
\[
J^\star(p):=\inf_{c\ \mathrm{单射}}J(c;p).
\]
将 \(p\) 的坐标按降序重排为 \(p_{(1)}\ge\cdots\ge p_{(m)}\)。则：
\begin{enumerate}
  \item 精确极值公式
  \[
  J^\star(p)=\sum_{i=1}^{m}p_{(i)}\log i;
  \]
  \item 设调和数 \(H_m:=\sum_{i=1}^{m}1/i\)，并定义调和先验
  \[
  h^{(m)}_i:=\frac{1}{iH_m}\quad(1\le i\le m),
  \]
  则有恒等式
  \[
  H(p)=J^\star(p)+\log H_m-D_{\mathrm{KL}}\!\bigl(p_{(\cdot)}\,\|\,h^{(m)}\bigr);
  \]
  \item 因而
  \[
  J^\star(p)\le H(p)\le J^\star(p)+\log H_m,
  \]
  且右端取等当且仅当 \(p_{(i)}=1/(iH_m)\)（\(1\le i\le m\)）。
\end{enumerate}
\end{theorem}
\begin{proof}
先证第一式。固定降序标号使 \(p_1\ge\cdots\ge p_m\)。
任意单射 \(c\) 产生 \(m\) 个互异正整数 \(n_i:=c(i)\)。
设其升序排列为 \(n_{[1]}<\cdots<n_{[m]}\)，则 \(n_{[i]}\ge i\)。
存在置换 \(\sigma\) 使 \(n_i=n_{[\sigma(i)]}\)，故
\[
J(c;p)=\sum_{i=1}^{m}p_i\log n_{[\sigma(i)]}.
\]
由重排不等式（降序权重配升序代价的最小化）得
\[
\sum_{i=1}^{m}p_i\log n_{[\sigma(i)]}\ge \sum_{i=1}^{m}p_i\log n_{[i]}
\ge \sum_{i=1}^{m}p_i\log i.
\]
取 \(c(i)=i\) 达到下界，故
\[
J^\star(p)=\sum_{i=1}^{m}p_i\log i=\sum_{i=1}^{m}p_{(i)}\log i.
\]

再证恒等式：
\[
D_{\mathrm{KL}}\!\bigl(p_{(\cdot)}\|h^{(m)}\bigr)
=\sum_{i=1}^{m}p_{(i)}\log\frac{p_{(i)}}{1/(iH_m)}
=\sum_{i=1}^{m}p_{(i)}\log p_{(i)}+\sum_{i=1}^{m}p_{(i)}\log i+\log H_m.
\]
注意 \(\sum p_{(i)}\log p_{(i)}=-H(p)\)，并代入 \(J^\star(p)\) 即得第二式。

由 \(D_{\mathrm{KL}}\ge 0\) 得右侧不等式及其取等条件。
左侧不等式由 \(p_{(i)}\le 1/i\)（降序向量的经典界）推出
\[
-\log p_{(i)}\ge \log i.
\]
乘以 \(p_{(i)}\) 并求和得 \(H(p)\ge J^\star(p)\)。证毕。
\end{proof}

\begin{corollary}[投影条件熵与最小素数寄存器账本的可实现界]\label{cor:conclusion-conditional-entropy-min-prime-register-ledger}
设 \(\pi:\Omega\twoheadrightarrow X\) 为有限纤维投影，\(\mu\) 为 \(\Omega\) 上概率分布。
记纤维 \(F_x:=\pi^{-1}(x)\)、\(m_x:=|F_x|\)、\(\mu_x\) 为条件分布（\(x\sim \pi_\#\mu\)）。
允许不同纤维复用同一整数编码，定义最小外置代价
\[
\mathcal J^\star(\mu\mid\pi):=
\inf_{\{c_x:F_x\hookrightarrow\NN_{\ge 1}\}}
\EE_{\omega\sim\mu}\!\left[\log c_{\pi(\omega)}(\omega)\right].
\]
则：
\begin{enumerate}
  \item
  \[
  \mathcal J^\star(\mu\mid\pi)=\EE_{x\sim\pi_\#\mu}\!\left[J^\star(\mu_x)\right];
  \]
  \item 条件熵分解
  \[
  H(\Omega\mid X)=
  \mathcal J^\star(\mu\mid\pi)
  +\EE_x[\log H_{m_x}]
  -\EE_x\!\left[D_{\mathrm{KL}}\!\bigl(\mu_x^\downarrow\|h^{(m_x)}\bigr)\right];
  \]
  \item 可实现界
  \[
  \mathcal J^\star(\mu\mid\pi)\le H(\Omega\mid X)\le
  \mathcal J^\star(\mu\mid\pi)+\EE_x[\log H_{m_x}],
  \]
  且
  \[
  \EE_x[\log H_{m_x}]
  \le \log H_{m_{\max}}
  \le \log\!\bigl(1+\log m_{\max}\bigr),\qquad
  m_{\max}:=\max_x m_x.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
目标函数按纤维分解为
\[
\sum_{x\in X}\pi_\#\mu(x)\,\EE_{\mu_x}[\log c_x].
\]
各 \(x\) 独立最小化，得第一式。
第二式由定理 \ref{thm:conclusion-fiber-minimal-godel-cost-harmonic-kl} 对每个 \(\mu_x\) 应用后取期望得到。
第三式由 \(D_{\mathrm{KL}}\ge 0\) 直接得到。
最后一行由 \(H_m\le 1+\log m\) 与 Jensen 单调性得证。证毕。
\end{proof}

\begin{theorem}[相对熵率刻画无误差纤维补码的最小比特率]\label{thm:conclusion-relative-entropy-minimal-sidebit-rate}
设 \((X,\mathcal B,\mu,T)\) 为遍历可逆测度保持系统，\((Y,\mathcal C,\nu,S)\) 为其因子，并以可测满射 \(\pi:X\to Y\) 满足 \(\pi\circ T=S\circ \pi\) 与 \(\nu=\mu\circ\pi^{-1}\)。
取一有限生成分割 \(\alpha\)（使得 \(\bigvee_{n\in\ZZ}T^{-n}\alpha\) 生成 \(\mathcal B\)），并记相对熵率
\[
h:=h_\mu(T,\alpha\mid \pi^{-1}\mathcal C).
\]
对 \(n\ge 1\) 令
\(
\alpha^{n-1}_0:=\bigvee_{j=0}^{n-1}T^{-j}\alpha
\)
为长度 \(n\) 的名称分割。
若存在可测编码
\[
\mathrm{Enc}_n:X\to\{0,1\}^{b_n}
\]
与可测解码
\[
\mathrm{Dec}_n:Y\times\{0,1\}^{b_n}\to X
\]
使得对 \(\mu\)-几乎处处 \(x\in X\)，点 \(\mathrm{Dec}_n(\pi(x),\mathrm{Enc}_n(x))\) 与 \(x\) 落在同一 \(\alpha^{n-1}_0\)-原子中（等价于：给定因子观测与 \(b_n\) 比特附加信息可无误差恢复长度 \(n\) 名称），则必有
\[
\liminf_{n\to\infty}\frac{b_n}{n}\ \ge\ \frac{h}{\log 2}.
\]
反之，存在一组 \((\mathrm{Enc}_n,\mathrm{Dec}_n)\) 使得
\[
\limsup_{n\to\infty}\frac{b_n}{n}\ \le\ \frac{h}{\log 2}.
\]
因此，相对熵率 \(h\) 精确刻画了“因子压缩在无误差可逆意义下所需的最小纤维比特率”。
\end{theorem}
\begin{proof}
令随机变量 \(W_n\) 为 \(\alpha^{n-1}_0\) 的原子索引（长度 \(n\) 名称），并将 \(Y\) 视为 \(\pi\) 诱导的可见 \(\sigma\)-代数。
标准一致性给出
\[
\lim_{n\to\infty}\frac{1}{n}H(W_n\mid Y)=h,
\]
其中 \(H(\cdot\mid\cdot)\) 为 Shannon 条件熵（自然对数）。
若 \(\mathrm{Enc}_n\) 与 \(\mathrm{Dec}_n\) 实现无误差名称恢复，则在每条纤维上 \(\mathrm{Enc}_n\) 给出对 \(W_n\) 的无误差描述，故由条件无误差源编码下界得
\(
b_n\log 2\ge H(W_n\mid Y)
\)；
两边除以 \(n\) 并取 \(\liminf\) 得下界。
上界由条件 Shannon--Fano/算术编码在每个 \(y\) 上对 \(W_n\) 的典型集合实施无误差编码给出，使 \(b_n\log 2\le H(W_n\mid Y)+o(n)\)；取 \(\limsup\) 即得。证毕。
\end{proof}

\begin{theorem}[\(\zeta\)-正则化素数独立律与熵极值原理]\label{thm:conclusion-zeta-regularized-prime-independence-maxent}
记 \(v_p(n)\) 为 \(p\)-进赋值。对 \(\beta>1\)，定义
\[
q_\beta(n):=\frac{n^{-\beta}}{\zeta(\beta)},\qquad n\in\NN_{\ge 1}.
\]
则：
\begin{enumerate}
  \item \textbf{素数独立律：}
  在 \(n\sim q_\beta\) 下，\(\{v_p(n)\}_{p\in\mathbb P}\) 相互独立，且
  \[
  \PP\!\bigl(v_p=k\bigr)=(1-p^{-\beta})\,p^{-\beta k},\qquad k\in\NN.
  \]
  \item \textbf{\(\zeta\)-正则化熵恒等式：}
  对任意分布 \(r\)（满足 \(H(r)<\infty,\ \EE_r[\log n]<\infty\)），
  \[
  H(r)=\beta\,\EE_r[\log n]+\log\zeta(\beta)-D_{\mathrm{KL}}(r\|q_\beta).
  \]
  特别地
  \[
  H(r)\le \beta\,\EE_r[\log n]+\log\zeta(\beta),
  \]
  且取等当且仅当 \(r=q_\beta\)。
  \item \textbf{素数耦合熵损失分解：}
  令 \(V:=(V_p)_{p\in\mathbb P}\)（\(V_p:=v_p(n)\)）在 \(r\) 下的联合分布。
  则
  \[
  D_{\mathrm{KL}}(r\|q_\beta)
  =\mathrm{TC}(V)+\sum_{p\in\mathbb P}
  D_{\mathrm{KL}}\!\bigl(\mathcal L(V_p)\,\|\,\mathrm{Geom}(p^{-\beta})\bigr),
  \]
  其中 \(\mathrm{TC}(V)\) 为总相关，\(\mathrm{TC}(V)\ge 0\) 且当且仅当素数轴独立时取零。
  \item \textbf{独立约束下最大熵：}
  在约束 \(\EE[\log n]=L\in(0,\infty)\) 且要求 \(\{v_p\}\) 独立的分布类中，Shannon 熵最大值由唯一的 \(q_{\beta^\star}\) 取得，其中 \(\beta^\star\) 由
  \[
  L=\EE_{q_{\beta^\star}}[\log n]
  \]
  唯一确定。
\end{enumerate}
\end{theorem}
\begin{proof}
由 Euler 乘积
\[
\zeta(\beta)=\prod_{p}(1-p^{-\beta})^{-1},
\]
与分解 \(n=\prod_p p^{v_p(n)}\) 得
\[
q_\beta(n)
=\prod_p (1-p^{-\beta})\,p^{-\beta v_p(n)}.
\]
故 \(\{v_p\}\) 独立，边缘为几何分布，得第 1 条。

第 2 条由直接计算：
\[
D_{\mathrm{KL}}(r\|q_\beta)
=\sum_n r(n)\log\frac{r(n)}{n^{-\beta}/\zeta(\beta)}
=-H(r)+\beta\EE_r[\log n]+\log\zeta(\beta).
\]
移项即得。

第 3 条使用 KL 的链式分解。把 \(r\) 在唯一分解坐标上视作 \(V\) 的联合分布。
在有限素数子集 \(F\subset\mathbb P\) 上有
\[
D_{\mathrm{KL}}\!\Bigl(\mathcal L(V_F)\,\Big\|\,\prod_{p\in F}\mathrm{Geom}(p^{-\beta})\Bigr)
=D_{\mathrm{KL}}\!\Bigl(\mathcal L(V_F)\,\Big\|\,\prod_{p\in F}\mathcal L(V_p)\Bigr)
+\sum_{p\in F}D_{\mathrm{KL}}\!\bigl(\mathcal L(V_p)\,\|\,\mathrm{Geom}(p^{-\beta})\bigr).
\]
令 \(F\uparrow\mathbb P\) 即得所述分解。

第 4 条中，独立性给出
\[
H(n)=\sum_p H(V_p),\qquad
\EE[\log n]=\sum_p (\log p)\,\EE[V_p].
\]
在约束下以拉格朗日乘子 \(\lambda\) 最大化
\[
\sum_p\Big(H(V_p)-\lambda(\log p)\EE[V_p]\Big)
\]
可分解为各 \(p\) 的一维问题；其最优解为几何分布，参数满足
\[
\alpha_p=e^{-\lambda\log p}=p^{-\beta},\qquad \beta:=\lambda.
\]
故整体分布为 \(q_\beta\)。
唯一性来自函数
\[
\beta\mapsto \EE_{q_\beta}[\log n]=-\frac{\zeta'(\beta)}{\zeta(\beta)}
\]
在 \((1,\infty)\) 上连续严格递减，且值域为 \((0,\infty)\)。证毕。
\end{proof}

\begin{theorem}[截断素数寄存器的碰撞概率闭式]\label{thm:conclusion-truncated-prime-register-collision-product}
\leanverified{paper\_conclusion\_truncated\_prime\_register\_collision\_product}
在定理 \ref{thm:conclusion-zeta-regularized-prime-independence-maxent} 的记号下，取 \(N,N'\overset{\mathrm{i.i.d.}}{\sim}q_\beta\)（\(\beta>1\)）。
对任意阈值 \(P\) 定义截断素数寄存器
\[
\mathrm{TR}_P(n):=\bigl(v_p(n)\bigr)_{p\le P}.
\]
则其碰撞概率具有完全乘积闭式
\[
\boxed{\;
\PP\!\bigl(\mathrm{TR}_P(N)=\mathrm{TR}_P(N')\bigr)
=\prod_{p\le P}\frac{1-p^{-\beta}}{1+p^{-\beta}}\; }.
\]
\end{theorem}
\begin{proof}
由素数独立律，
\[
\PP\!\bigl(\mathrm{TR}_P(N)=\mathrm{TR}_P(N')\bigr)
=\prod_{p\le P}\PP\!\bigl(v_p(N)=v_p(N')\bigr).
\]
对固定 \(p\)，由几何分布 \(\PP(v_p=k)=(1-p^{-\beta})p^{-\beta k}\) 得
\[
\PP\!\bigl(v_p(N)=v_p(N')\bigr)
=\sum_{k\ge 0}(1-p^{-\beta})^2p^{-2\beta k}
=\frac{(1-p^{-\beta})^2}{1-p^{-2\beta}}
=\frac{1-p^{-\beta}}{1+p^{-\beta}}.
\]
相乘即得。证毕。
\end{proof}

\begin{theorem}[单位圆--椭圆--素数寄存器的可加几何同构]\label{thm:conclusion-unit-circle-ellipse-prime-register-isomorphism}
对每个 \(n\in\NN_{\ge 1}\) 定义
\[
A_n:=\mathrm{diag}\!\bigl(\sqrt n,\ 1/\sqrt n\bigr)\in \mathrm{SL}(2,\RR),
\qquad
E_n:=A_n(S^1)\subset\RR^2.
\]
则：
\begin{enumerate}
  \item 映射 \(n\mapsto A_n\) 给出交换幺半群同构
  \[
  (\NN_{\ge 1},\times)\cong (\{A_n\}_{n\ge 1},\cdot),\qquad A_{nm}=A_nA_m.
  \]
  因而素数分解 \(n=\prod p^{e_p}\) 等价于矩阵分解 \(A_n=\prod A_p^{e_p}\)。
  \item 在
  \[
  \mathcal P_1:=\{B\in\mathrm{Sym}^+(2,\RR):\det B=1\}
  \]
  的标准仿射不变度量下，
  \[
  d(I,A_n)=\frac{|\log n|}{\sqrt2}.
  \]
  \item 对任意有限集 \(A\) 与分布 \(p\)，在所有单射
  \(c:A\hookrightarrow\NN_{\ge 1}\)（等价于所有 \(\Phi:A\hookrightarrow\{E_n:n\ge 1\}\)）中，
  \[
  \inf_c\ \EE_{a\sim p}\!\bigl[d(I,A_{c(a)})\bigr]
  =\frac{1}{\sqrt2}J^\star(p).
  \]
  因而
  \[
  \frac{1}{\sqrt2}J^\star(p)
  \le \frac{1}{\sqrt2}H(p)
  \le \frac{1}{\sqrt2}J^\star(p)+\frac{1}{\sqrt2}\log H_m,\qquad m:=|A|.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
第一条由对角矩阵乘法直接得到；唯一分解给出不可约生成元对应于素数。

第二条中，\(\mathcal P_1\) 的 geodesic 距离满足
\[
d(I,B)=\|\log B\|_F.
\]
对 \(A_n\) 有
\[
\log A_n=\mathrm{diag}\!\left(\frac12\log n,-\frac12\log n\right),
\]
故
\[
\|\log A_n\|_F
=\sqrt{\left(\frac12\log n\right)^2+\left(\frac12\log n\right)^2}
=\frac{|\log n|}{\sqrt2}.
\]

第三条中，任意 \(\Phi\) 等价于单射编码 \(c_\Phi:A\hookrightarrow\NN_{\ge 1}\)。
记 \(\Phi(a)=E_{c_\Phi(a)}\)，并约定 \(A_{\Phi(a)}:=A_{c_\Phi(a)}\)。
由第二条
\[
\EE[d(I,A_{\Phi(a)})]
=\frac{1}{\sqrt2}\EE[\log c_\Phi(a)].
\]
故对 \(\Phi\) 取下确界即为 \(\frac{1}{\sqrt2}J^\star(p)\)。
最后一行由定理 \ref{thm:conclusion-fiber-minimal-godel-cost-harmonic-kl} 的不等式直接得到。证毕。
\end{proof}

\begin{proposition}[投影词的哥德尔半直积表示]\label{prop:conclusion-projection-word-godel-semidirect}
\leanverified{paper\_conclusion\_projection\_word\_godel\_semidirect}
设 \(\Sigma\) 为有限字母表，\(\kappa:\Sigma\hookrightarrow\NN_{\ge 1}\) 为单射。
对词 \(w=\sigma_1\cdots\sigma_\ell\in\Sigma^\ast\)，定义
\[
\mathbf e(w)\in \NN^{(\NN)},\qquad
\mathbf e(w)_i=
\begin{cases}
\kappa(\sigma_i),&1\le i\le \ell,\\
0,&i>\ell.
\end{cases}
\]
定义移位
\[
(\mathrm{Sh}^k a)_i=
\begin{cases}
a_{i-k},&i>k,\\
0,&i\le k.
\end{cases}
\]
在 \(\NN^{(\NN)}\rtimes\NN\) 上定义乘法
\[
(a,m)\cdot(b,n):=(a+\mathrm{Sh}^m b,\ m+n).
\]
令
\[
\Psi:\Sigma^\ast\to\NN^{(\NN)}\rtimes\NN,\qquad
\Psi(w):=\bigl(\mathbf e(w),|w|\bigr).
\]
则 \(\Psi\) 为单射幺半群同态。
若 \(p_i\) 为第 \(i\) 个素数，定义
\[
G(w):=\prod_{i=1}^{|w|}p_i^{\kappa(\sigma_i)},
\]
则 \(w\mapsto G(w)\) 为单射，且 \(w\) 可由赋值向量 \((v_{p_i}(G(w)))_{i\ge 1}\) 唯一恢复。
\end{proposition}
\begin{proof}
对任意 \(u,v\in\Sigma^\ast\) 有
\[
\mathbf e(uv)=\mathbf e(u)+\mathrm{Sh}^{|u|}\mathbf e(v),\qquad |uv|=|u|+|v|,
\]
故
\[
\Psi(uv)=\Psi(u)\cdot\Psi(v),
\]
即 \(\Psi\) 为同态。
若 \(\Psi(u)=\Psi(v)\)，则长度相同且各位置编码相同，由 \(\kappa\) 单射得 \(u=v\)，故 \(\Psi\) 单射。

对 \(G(w)\)，由算术基本定理可唯一恢复每个 \(v_{p_i}(G(w))=\kappa(\sigma_i)\)（\(1\le i\le |w|\)）及尾部零赋值，故 \(w\) 唯一恢复。证毕。
\end{proof}
