\subsection{属 \texorpdfstring{$49$}{49} 的 \texorpdfstring{$S_4$}{S4}--Hurwitz 塔中 Hodge 表示对固定点、商亏格与分支的恢复}\label{subsec:conclusion-s4-hodge-fixedpoint-branch-euler-recovery}
沿用定理 \ref{thm:conclusion-cf-s4-holomorphic-differential-decomposition}、
结论 \ref{con:xi-time-part9n-burnside-kani-rosen-local-zeta-pointcount} 与结论 \ref{con:xi-time-part9n-c2-c3-quotient-jacobian-complete-splitting} 的记号。
记
\[
X
\]
为属 \(49\) 的 \(S_4\)-Hurwitz 曲线，
\[
V:=H^0(X,\Omega_X^1)
\cong
5\cdot \mathrm{sgn}\ \oplus\ 4\cdot V_2\ \oplus\ 3\cdot V_3\ \oplus\ 9\cdot V_3',
\]
并置
\[
H:=X/A_4,\qquad
C_6:=X/V_4,\qquad
Z:=X/D_8,\qquad
C_F:=X/S_3,\qquad
Y:=X/C_4.
\]
在这一口径下，Hodge 表示不仅决定各中间商的维数分层，而且足以反向锁定全部非平凡共轭类固定点数、主要商曲线亏格表、换位商局部 Euler 因子的平方块，以及整条 \(S_4\)-正则覆盖的分支型。

\leanverified{paper\_conclusion\_s4\_hodge\_determines\_fixedpoint\_counts}
\begin{theorem}[\texorpdfstring{$S_4$}{S4} 各非平凡共轭类在属 \texorpdfstring{$49$}{49} 曲线上的固定点数由 Hodge 表示唯一决定]\label{thm:conclusion-s4-hodge-determines-fixedpoint-counts}
设 \(\tau,\delta,\sigma_3,\sigma_4\in S_4\) 分别表示换位、双换位、\(3\)-循环与 \(4\)-循环。
则其在 \(X\) 上的固定点数满足
\[
\boxed{\;
\#\operatorname{Fix}_X(\tau)=24,\qquad
\#\operatorname{Fix}_X(\delta)=0,\qquad
\#\operatorname{Fix}_X(\sigma_3)=0,\qquad
\#\operatorname{Fix}_X(\sigma_4)=0.\;}
\]
特别地，类函数
\[
g\longmapsto \#\operatorname{Fix}_X(g)
\]
在全部非平凡共轭类上的取值已由 \(V\) 的表示分解唯一锁定。
\end{theorem}
\begin{proof}
由 \(S_4\) 的标准特征表，
\[
\chi_{\mathrm{sgn}}=(1,-1,1,1,-1),\qquad
\chi_{V_2}=(2,0,2,-1,0),
\]
\[
\chi_{V_3}=(3,1,-1,0,-1),\qquad
\chi_{V_3'}=(3,-1,-1,0,1),
\]
故 \(V\) 的特征标 \(\chi_V\) 在四类非平凡共轭类上的取值为
\[
\chi_V(\tau)=5(-1)+4\cdot 0+3\cdot 1+9(-1)=-11,
\]
\[
\chi_V(\delta)=5\cdot 1+4\cdot 2+3(-1)+9(-1)=1,
\]
\[
\chi_V(\sigma_3)=5\cdot 1+4(-1)+3\cdot 0+9\cdot 0=1,
\]
\[
\chi_V(\sigma_4)=5(-1)+4\cdot 0+3(-1)+9\cdot 1=1.
\]

对任意非平凡有限阶全纯自同构 \(g\) 作用于紧 Riemann 曲线 \(X\)，其固定点都是孤立的，且局部 Lefschetz 指数恒为 \(1\)。
于是拓扑 Lefschetz 公式给出
\[
\#\operatorname{Fix}_X(g)
=
\sum_{i=0}^{2}(-1)^i\operatorname{Tr}\!\bigl(g\mid H^i(X,\CC)\bigr).
\]
又
\[
H^0(X,\CC)\cong \CC,\qquad
H^2(X,\CC)\cong \CC,
\]
且
\[
H^1(X,\CC)\cong V\oplus \overline V
\]
作为 \(g\)-模成立，因此
\[
\#\operatorname{Fix}_X(g)
=
2-\chi_V(g)-\overline{\chi_V(g)}
=
2-2\Re\chi_V(g).
\]
由于上述四个角色值都为实数，遂有
\[
\#\operatorname{Fix}_X(g)=2-2\chi_V(g).
\]
将 \(\chi_V(\tau)=-11\) 与
\(
\chi_V(\delta)=\chi_V(\sigma_3)=\chi_V(\sigma_4)=1
\)
代入，即得
\[
\#\operatorname{Fix}_X(\tau)=24,
\qquad
\#\operatorname{Fix}_X(\delta)=\#\operatorname{Fix}_X(\sigma_3)=\#\operatorname{Fix}_X(\sigma_4)=0.
\]
证毕。
\end{proof}

\begin{theorem}[全部标准商曲线的亏格表可由 Hodge 表示单独恢复]\label{thm:conclusion-s4-hodge-recovers-standard-quotient-genera}
\leanverified{paper\_conclusion\_s4\_hodge\_recovers\_standard\_quotient\_genera}
沿用定理 \ref{thm:conclusion-s4-hodge-determines-fixedpoint-counts} 的记号。
则下列代表性子群商曲线亏格满足
\[
\boxed{
\begin{aligned}
g(X/\langle\tau\rangle)&=19,&
g(X/\langle\delta\rangle)&=25,&
g(X/\langle\sigma_3\rangle)&=17,&
g(X/\langle\sigma_4\rangle)&=13,\\
g(X/V_4)&=13,&
g(X/S_3)&=3,&
g(X/D_8)&=4,&
g(X/A_4)&=5,&
g(X/S_4)&=0.
\end{aligned}}
\]
特别地，这重新恢复了既有记号中的
\[
g(C_6)=13,\qquad
g(C_F)=3,\qquad
g(Z)=4,\qquad
g(H)=5,
\qquad
g(X/S_4)=0.
\]
\end{theorem}
\begin{proof}
对任意子群 \(U\le S_4\)，商曲线亏格满足
\[
g(X/U)=\dim V^U=\frac{1}{|U|}\sum_{u\in U}\chi_V(u).
\]
由定理 \ref{thm:conclusion-s4-hodge-determines-fixedpoint-counts} 的计算，
\[
\chi_V(1)=g(X)=49,\qquad
\chi_V(\tau)=-11,\qquad
\chi_V(\delta)=\chi_V(\sigma_3)=\chi_V(\sigma_4)=1.
\]

于是四类循环子群直接给出
\[
g(X/\langle\tau\rangle)=\frac{49+(-11)}{2}=19,
\]
\[
g(X/\langle\delta\rangle)=\frac{49+1}{2}=25,
\]
\[
g(X/\langle\sigma_3\rangle)=\frac{49+1+1}{3}=17,
\]
\[
g(X/\langle\sigma_4\rangle)=\frac{49+1+1+1}{4}=13.
\]

再对其余四类标准子群作平均。
对 Klein 四元群 \(V_4\)，类分布为
\(
1+3\delta
\)，故
\[
g(X/V_4)=\frac{49+3\cdot 1}{4}=13.
\]
对 \(S_3\)，类分布为
\(
1+3\tau+2\sigma_3
\)，故
\[
g(X/S_3)=\frac{49+3(-11)+2\cdot 1}{6}=3.
\]
取标准 dihedral 子群 \(D_8\le S_4\)，其类分布为
\(
1+2\sigma_4+3\delta+2\tau
\)，故
\[
g(X/D_8)=\frac{49+2\cdot 1+3\cdot 1+2(-11)}{8}=4.
\]
对 \(A_4\)，类分布为
\(
1+3\delta+8\sigma_3
\)，故
\[
g(X/A_4)=\frac{49+3\cdot 1+8\cdot 1}{12}=5.
\]

最后，由 \(V\) 中平凡表示不出现，
\[
g(X/S_4)=\dim V^{S_4}=0.
\]
证毕。
\end{proof}

\begin{theorem}[任一换位商曲线的局部 Euler 因子带有严格平方块]\label{thm:conclusion-s4-transposition-quotient-square-euler-factor}
任取换位 \(\tau\in S_4\)，令 \(C_2:=\langle\tau\rangle\)。
设 \(p\) 为一处使
\[
X/C_2,\qquad C_F=X/S_3,\qquad Y=X/C_4
\]
共同良约化的素数，并记局部 Hasse--Weil 分子为 \(P_{-,p}(T)\)。
则有严格分解
\[
\boxed{\;
P_{X/C_2,p}(T)=P_{C_F,p}(T)^2\,P_{Y,p}(T).\;}
\]
等价地，点数满足
\[
\boxed{\;
\#(X/C_2)(\FF_p)
=
2\,\#C_F(\FF_p)+\#Y(\FF_p)-2(p+1).\;}
\]
因此，任一换位商曲线的局部 Euler 因子都带有一个不可消去的完美平方因子
\[
P_{C_F,p}(T)^2.
\]
\end{theorem}
\begin{proof}
由 \(S_4\) 中换位子群彼此共轭，只需固定一个代表 \(C_2=\langle\tau\rangle\)。
结论 \ref{con:xi-time-part9n-burnside-kani-rosen-local-zeta-pointcount} 已给出
\[
P_{X/C_2}(v,T)=P_{C_F}(v,T)^2\,P_Y(v,T)
\]
以及
\[
a_{X/C_2}(v,n)=2a_{C_F}(v,n)+a_Y(v,n)
\]
对共同良约化素位 \(v\) 与一切 \(n\ge 1\) 成立。
取 \(n=1\) 并把 \(v\) 记为素数 \(p\)，即得
\[
P_{X/C_2,p}(T)=P_{C_F,p}(T)^2\,P_{Y,p}(T).
\]

再记
\[
a_p(C):=p+1-\#C(\FF_p).
\]
由上式的一次项比较，
\[
a_p(X/C_2)=2a_p(C_F)+a_p(Y).
\]
移项即得
\[
\#(X/C_2)(\FF_p)
=
2\,\#C_F(\FF_p)+\#Y(\FF_p)-2(p+1).
\]
证毕。
\end{proof}

\begin{theorem}[属 \texorpdfstring{$49$}{49} 的 \texorpdfstring{$S_4$}{S4} 正则覆盖之全部分支类型可由 Hodge 表示反向恢复]\label{thm:conclusion-s4-hodge-recovers-full-branch-type}
\leanverified{paper\_conclusion\_s4\_hodge\_recovers\_full\_branch\_type}
设
\[
f:X\longrightarrow \PP^1_u
\]
为文稿既定的 \(S_4\)-正则覆盖。
则仅由 \(V=H^0(X,\Omega_X^1)\) 的表示分解及定理 \ref{thm:conclusion-s4-hodge-determines-fixedpoint-counts}--\ref{thm:conclusion-s4-hodge-recovers-standard-quotient-genera}，即可推出：
\begin{enumerate}
  \item 任一非平凡 inertia 生成元都必须是换位；
  \item 不存在双换位、\(3\)-循环或 \(4\)-循环分支；
  \item 总分支值个数恰为 \(12\)；
  \item 每个分支值都是简单换位分支。
\end{enumerate}
\end{theorem}
\begin{proof}
设 \(x\in X\) 位于某个分支点之上，则其稳定子
\[
I_x\le S_4
\]
非平凡。
在特征 \(0\) 的曲线覆盖中，惯性群必为循环群；取其生成元 \(g_x\neq 1\)。
由于 \(g_x\) 固定点 \(x\)，故
\[
x\in \operatorname{Fix}_X(g_x).
\]
由定理 \ref{thm:conclusion-s4-hodge-determines-fixedpoint-counts}，双换位、\(3\)-循环与 \(4\)-循环在 \(X\) 上都无固定点，因此 \(g_x\) 只能是换位。
这就证明了全部非平凡 inertia 生成元都属于换位类，从而分支只能来自换位。

另一方面，由定理 \ref{thm:conclusion-s4-hodge-recovers-standard-quotient-genera}，
\[
g(X)=49,\qquad g(X/S_4)=0.
\]
故对 \(f:X\to \PP^1_u\) 应用 Riemann--Hurwitz 公式，
\[
2g(X)-2
=
|S_4|\bigl(2g(\PP^1)-2\bigr)+R_f
=
24(-2)+R_f,
\]
从而
\[
R_f=96+48=144.
\]

现取任一分支值 \(u_0\in \PP^1_u\)。
既然其 inertia 群由换位生成，则该 inertia 群阶恰为 \(2\)。
由于 \(f\) 是正则 \(S_4\)-覆盖，\(u_0\) 上方的点数为
\[
\frac{|S_4|}{2}=12,
\]
且每一点的 ramification index 都等于 \(2\)，故该分支值对总分歧度的贡献为
\[
12\cdot (2-1)=12.
\]
因此分支值总数恰为
\[
\frac{R_f}{12}=\frac{144}{12}=12.
\]
并且每个分支值都对应于惯性阶 \(2\) 的简单换位分支。证毕。
\end{proof}

上述四条结果把 genus \(49\) 的 \(S_4\)-Hurwitz 塔进一步压缩为一条完全可逆的表示论链：Hodge 分解先唯一决定四类非平凡共轭元的固定点数，再唯一决定全部标准商曲线的亏格表；由 Burnside--Kani--Rosen 关系，换位商曲线在每个良素处必带严格平方 Euler 因子；最后，这些数据反过来又锁定了整条正则覆盖只能由 \(12\) 个简单换位分支值组成。于是，固定点几何、商曲线亏格、局部 \(L\)-因子与分支型不再是彼此独立的输入，而是 \(H^0(X,\Omega_X^1)\) 的 \(S_4\)-表示在几何与算术两侧的统一投影。
