若再把 Zeckendorf canonical section 视为有限循环商上的概率核，则前一条 Euler--Poisson 边界链还可继续闭合为另一组更精细的有限分辨率热核命题。其核心不再是“截断 Poisson 与零速率编码”本身，而是：canonical Euler 包在 \(G_m\) 上实际上给出一个以超指数精度逼近 wrapped Poisson 的近热半群，并且这一近热半群在 \(t=0\) 处与精确热核共享完整的 \((F_{m+2}-1)\)-阶 jet；由此，正实 Euler 温度轴与单位根 Dirichlet 相位轴便首次出现严格的谱范畴分离。

\leanverified{paper\_conclusion\_zeckendorf\_euler\_wrapped\_poisson\_fourier}
\begin{theorem}[canonical Euler 包的 wrapped Poisson Fourier 符号]\label{thm:conclusion-zeckendorf-euler-wrapped-poisson-fourier}
固定分辨率 \(m\)，记
\[
F:=F_{m+2},
\qquad
G_m:=\ZZ/(F\ZZ),
\qquad
E_F(z):=\sum_{n=0}^{F-1}\frac{z^n}{n!}.
\]
在 \(G_m\) 上定义 canonical Euler 包
\[
\mu_{m,t}(r):=\frac{t^r}{r!\,E_F(t)}
\qquad (0\le r\le F-1,\ t\ge 0),
\]
并定义 wrapped Poisson 核
\[
\wp_{F,t}(r):=e^{-t}\sum_{\ell\ge 0}\frac{t^{r+\ell F}}{(r+\ell F)!},
\qquad
r\in G_m.
\]
若对 \(k\in G_m\) 记角色
\[
\chi_k(r):=e^{-2\pi i kr/F},
\]
则有精确 Fourier 公式
\[
\widehat\mu_{m,t}(k)
:=
\sum_{r\in G_m}\mu_{m,t}(r)\chi_k(r)
=
\frac{E_F\!\bigl(te^{-2\pi i k/F}\bigr)}{E_F(t)},
\]
以及
\[
\widehat\wp_{F,t}(k)
:=
\sum_{r\in G_m}\wp_{F,t}(r)\chi_k(r)
=
e^{\,t(e^{-2\pi i k/F}-1)}.
\]
因此，canonical Euler 包在有限分辨率商环上的谱符号恰由截断指数函数在 \(F\)-次单位根处的取样给出，而 wrapped Poisson 核则保留完整的连续热符号。
\end{theorem}
\begin{proof}
由 Zeckendorf 范围双射，\(V_m\) 把 canonical Euler 权无损传送到
\[
\{0,1,\dots,F-1\}\cong G_m,
\]
故 \(\mu_{m,t}\) 的质量函数正是
\[
r\longmapsto \frac{t^r}{r!\,E_F(t)}.
\]
于是
\[
\widehat\mu_{m,t}(k)
=
\frac{1}{E_F(t)}\sum_{r=0}^{F-1}\frac{(te^{-2\pi i k/F})^r}{r!}
=
\frac{E_F(te^{-2\pi i k/F})}{E_F(t)}.
\]

另一方面，
\[
\widehat\wp_{F,t}(k)
=
e^{-t}\sum_{r=0}^{F-1}\sum_{\ell\ge 0}
\frac{t^{r+\ell F}}{(r+\ell F)!}\,e^{-2\pi i k(r+\ell F)/F}.
\]
因为
\[
e^{-2\pi i k\ell F/F}=1,
\]
故双重和并为
\[
e^{-t}\sum_{n\ge 0}\frac{(te^{-2\pi i k/F})^n}{n!}
=
e^{\,t(e^{-2\pi i k/F}-1)}.
\]
证毕。
\end{proof}

\begin{theorem}[canonical Euler 包对 wrapped Poisson 热核的超指数逼近]\label{thm:conclusion-zeckendorf-euler-wrapped-poisson-superexp-approx}
\leanverified{paper\_conclusion\_zeckendorf\_euler\_wrapped\_poisson\_superexp\_approx}
记
\[
T_F(t):=\sum_{n\ge F}\frac{t^n}{n!}.
\]
则对每个 \(t\ge 0\)，都有
\[
d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t})\le e^{-t}T_F(t).
\]
于是对任意固定 \(T>0\)，
\[
\sup_{0\le t\le T}d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t})
=
O_T\!\left(\frac{T^F}{F!}\right),
\]
并且
\[
\sup_{0\le t\le T}\sup_{k\in G_m}
\left|
\widehat\mu_{m,t}(k)-e^{\,t(e^{-2\pi i k/F}-1)}
\right|
=
O_T\!\left(\frac{T^F}{F!}\right).
\]
特别地，因 \(F=F_{m+2}\asymp \varphi^m\)，故
\[
\log \sup_{0\le t\le T}d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t})
=
-\Theta(m\varphi^m).
\]
\end{theorem}
\begin{proof}
令 \(N_t\sim \operatorname{Poisson}(t)\)。则
\[
\mu_{m,t}=\mathcal L(N_t\mid N_t<F)
\]
经自然嵌入 \(\{0,\dots,F-1\}\hookrightarrow G_m\) 视为 \(G_m\) 上的分布，而
\[
\wp_{F,t}=\mathcal L(N_t\bmod F).
\]
把 \(N_t\mapsto N_t\bmod F\) 视为测度推前，并用总变差在可测映射下不增，得到
\[
d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t})
\le
d_{\mathrm{TV}}\!\bigl(\mathcal L(N_t\mid N_t<F),\mathcal L(N_t)\bigr)
=
\PP(N_t\ge F)
=
e^{-t}T_F(t).
\]

再由 \(|\chi_k|\le 1\) 及 Fourier 系数对总变差的标准控制，
\[
\left|\widehat\mu_{m,t}(k)-\widehat\wp_{F,t}(k)\right|
\le 2\,d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t})
\le 2e^{-t}T_F(t),
\]
故在 \(0\le t\le T\) 上也有同阶 \(O_T(T^F/F!)\) 上界。

最后，由 Stirling 公式
\[
\log(F!)=F\log F-F+O(\log F)
\]
与 \(F\asymp \varphi^m\)，便得到
\[
\log\!\left(\frac{T^F}{F!}\right)=-\Theta(m\varphi^m).
\]
证毕。
\end{proof}

\begin{corollary}[有限分辨率 Euler 近热半群的超指数缺陷]\label{cor:conclusion-zeckendorf-euler-approx-semigroup-defect}
\leanverified{paper\_conclusion\_zeckendorf\_euler\_approx\_semigroup\_defect}
对任意 \(s,t\ge 0\)，都有
\[
d_{\mathrm{TV}}\!\bigl(\mu_{m,s}*\mu_{m,t},\mu_{m,s+t}\bigr)
\le
e^{-s}T_F(s)+e^{-t}T_F(t)+e^{-(s+t)}T_F(s+t),
\]
其中卷积取于循环群 \(G_m\)。因此 \((\mu_{m,t})_{t\ge 0}\) 在每个固定分辨率 \(m\) 上构成一个 semigroup defect 为
\[
O_T\!\left(\frac{T^F}{F!}\right)
\]
的近似热半群。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-zeckendorf-euler-wrapped-poisson-fourier}，
\[
\widehat\wp_{F,s}(k)\widehat\wp_{F,t}(k)
=
e^{\,s(e^{-2\pi i k/F}-1)}e^{\,t(e^{-2\pi i k/F}-1)}
=
e^{\,(s+t)(e^{-2\pi i k/F}-1)}
=
\widehat\wp_{F,s+t}(k),
\]
故
\[
\wp_{F,s}*\wp_{F,t}=\wp_{F,s+t}.
\]
再由总变差对卷积的收缩性，
\[
d_{\mathrm{TV}}(\mu_{m,s}*\mu_{m,t},\wp_{F,s}*\wp_{F,t})
\le
d_{\mathrm{TV}}(\mu_{m,s},\wp_{F,s})
+
d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t}).
\]
因此
\[
d_{\mathrm{TV}}(\mu_{m,s}*\mu_{m,t},\mu_{m,s+t})
\le
d_{\mathrm{TV}}(\mu_{m,s},\wp_{F,s})
+
d_{\mathrm{TV}}(\mu_{m,t},\wp_{F,t})
+
d_{\mathrm{TV}}(\wp_{F,s+t},\mu_{m,s+t}).
\]
最后代入定理 \ref{thm:conclusion-zeckendorf-euler-wrapped-poisson-superexp-approx} 即得。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$(F-1)$}{(F-1)} 阶 jet 刚性]\label{prop:conclusion-zeckendorf-euler-fminusone-jet-rigidity}
\leanverified{paper\_conclusion\_zeckendorf\_euler\_fminusone\_jet\_rigidity}
对每个固定 \(k\in G_m\)，有
\[
\widehat\mu_{m,t}(k)
=
e^{\,t(e^{-2\pi i k/F}-1)}+O(t^F),
\qquad
t\to 0.
\]
等价地，
\[
\frac{d^n}{dt^n}\widehat\mu_{m,t}(k)\Big|_{t=0}
=
\bigl(e^{-2\pi i k/F}-1\bigr)^n,
\qquad
0\le n\le F-1.
\]
因此，canonical Euler 包与精确 wrapped Poisson 热核在 \(t=0\) 处共享完整的 \((F-1)\)-阶 Taylor germ；有限分辨率造成的非半群性直到次数 \(F\) 才首次显影。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-zeckendorf-euler-wrapped-poisson-fourier}，
\[
\widehat\mu_{m,t}(k)=\frac{E_F(te^{-2\pi i k/F})}{E_F(t)}.
\]
而对固定 \(F\)，截断指数函数满足
\[
E_F(z)=e^z+O(z^F)
\qquad (z\to 0).
\]
故
\[
\widehat\mu_{m,t}(k)
=
\frac{e^{\,te^{-2\pi i k/F}}+O(t^F)}{e^t+O(t^F)}
=
e^{\,t(e^{-2\pi i k/F}-1)}+O(t^F).
\]
于是两边在 \(t=0\) 处的前 \(F-1\) 阶导数完全一致。证毕。
\end{proof}

\begin{theorem}[Euler 温度轴与 Dirichlet 相位轴的严格谱分离]\label{thm:conclusion-zeckendorf-euler-temperature-phase-spectral-separation}
在当前项目框架内，\(e\)-相关变形至少分裂为三种互不相同的谱机制：
\begin{enumerate}
  \item 在有限分辨率商环 \(G_m\) 上，canonical Euler 包以超指数精度逼近 wrapped Poisson 热核，其非平凡模的谱幅值满足
  \[
  |\widehat\wp_{F,t}(k)|
  =
  \exp\!\bigl(-t(1-\cos(2\pi k/F))\bigr),
  \]
  因而在低频区具有扩散型律
  \[
  |\widehat\wp_{F,t}(k)|
  =
  \exp\!\left(
  -\frac{2\pi^2tk^2}{F^2}
  +O\!\left(\frac{k^4}{F^4}\right)
  \right).
  \]
  \item 在真实输入 \(40\) 状态核的输出位势侧，正实 Euler 点 \(u=e=e^1\) 严格位于解析圆盘与 RH/Ramanujan window 的内部；即正文给出的 branch radius
  \[
  R_\theta\approx 2.76230289346087
  \]
  满足 \(1<R_\theta\)，而临界参数
  \[
  u_R\approx 3.59262181344
  \]
  满足 \(e<u_R\)。
  \item 然而在同一核的单位根 Dirichlet 扭曲轴上，正文又已证明存在模数 \(m\) 与某个非平凡扭曲通道，使其误差指数越过平方根门槛，即
  \[
  \theta_m>\frac12.
  \]
\end{enumerate}
因此，正实 Euler 温度轴与单位根 Dirichlet 相位轴并不属于同一谱范畴：前者是扩散型且处于 Ramanujan 可允许区的，后者则携带真正算术性的非扩散阻碍。障碍由相位生成，而非由正实 Euler 温度生成。
\end{theorem}
\begin{proof}
第一个断言由定理 \ref{thm:conclusion-zeckendorf-euler-wrapped-poisson-fourier} 直接给出：
\[
\widehat\wp_{F,t}(k)=e^{\,t(e^{-2\pi i k/F}-1)}.
\]
取模便得
\[
|\widehat\wp_{F,t}(k)|=e^{-t(1-\cos(2\pi k/F))}.
\]
对 \(\cos\) 在零点处展开，即得低频扩散型律。

第二个断言完全来自正文已建立的输出位势分歧分析：branch radius \(R_\theta\) 与核版 RH 临界点 \(u_R\) 都已有闭式判别与数值，而
\[
1<R_\theta,
\qquad
e<u_R.
\]
因此 Euler 点 \(u=e\) 同时严格位于解析圆盘与 RH/Ramanujan window 的内部。

第三个断言则是正文关于 Dirichlet 扭曲谱的现成结论：存在某个非平凡模通道，使对应误差指数严格大于 \(1/2\)。把这三条性质并读，便得到正实 Euler 温度轴与单位根 Dirichlet 相位轴的严格谱分离。证毕。
\end{proof}

\begin{conjecture}[有限分辨率 Euler 刚性]\label{conj:conclusion-zeckendorf-euler-finite-resolution-rigidity}
固定 \(m\)，令 \(F=F_{m+2}\)。设 \((\nu_{m,t})_{t\ge 0}\) 是 \(G_m\) 上一族解析概率测度，满足：
\begin{enumerate}
  \item \(\nu_{m,0}=\delta_0\)；
  \item 对每个 \(k\in G_m\)，其 Fourier 变换在 \(t=0\) 处满足
  \[
  \frac{d^n}{dt^n}\widehat\nu_{m,t}(k)\Big|_{t=0}
  =
  \bigl(e^{-2\pi i k/F}-1\bigr)^n,
  \qquad
  0\le n\le F-1;
  \]
  \item 在任意紧区间 \(0\le t\le T\) 上，其半群缺陷满足
  \[
  d_{\mathrm{TV}}(\nu_{m,s}*\nu_{m,t},\nu_{m,s+t})
  =
  O_T\!\left(\frac{T^F}{F!}\right).
  \]
\end{enumerate}
则必有
\[
\nu_{m,t}(r)=\frac{t^r}{r!\,E_F(t)}
\qquad (0\le r\le F-1),
\]
亦即 \(\nu_{m,t}=\mu_{m,t}\)。换言之，factorial canonical section 应当是有限分辨率中唯一同时实现
\[
\text{\((F-1)\) 阶 Poisson germ},
\qquad
\text{超指数半群缺陷},
\qquad
\text{Zeckendorf canonical section 支撑}
\]
的解析正测度族。
\end{conjecture}
