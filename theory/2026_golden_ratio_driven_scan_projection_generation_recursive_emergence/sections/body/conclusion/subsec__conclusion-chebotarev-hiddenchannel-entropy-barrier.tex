\subsection{primitive Chebotarev 群商熵的 hidden-channel 二阶律与主导障碍}\label{subsec:conclusion-chebotarev-hiddenchannel-entropy-barrier}

沿用定理 \ref{thm:xi-time-part64ac-abelian-quotient-kl-chain}、定理 \ref{thm:xi-time-part64ac-visible-hidden-character-energy-splitting} 与定理 \ref{thm:kernel-chebotarev-second-main-term-witness} 的记号。前文已经把 primitive Chebotarev 分布在阿贝尔群商下的 KL 链式律、可见/隐藏角色能量的严格正交分解，以及主导非平凡通道的次主振荡模态完全显式化。将这三条接口并读后，还可继续抽出如下三条结论：其一，群商所抹去的全部相对熵在二阶上恰等于隐藏角色能量的一半；其二，若隐藏谱中存在唯一主导通道，则该商熵损失沿无穷子序列呈现精确的指数障碍，而该障碍完全由被删去的最慢隐藏字符控制；其三，隐藏主导字符本身可以被选择成一条在细层主量级振荡、而在粗层推前后完全湮灭的幽灵统计通道。

\begin{theorem}[群商 Chebotarev 熵损失的二阶精确展开]\label{thm:conclusion-chebotarev-hiddenchannel-kl-second-order}
设
\[
\pi:G_2\twoheadrightarrow G_1
\]
为有限阿贝尔群满同态，\(K:=\ker\pi\)，并固定 \(u\in(0,1]\)。沿用 primitive Chebotarev 分布
\[
\mu_{n,u}^{(i)}\qquad (i=1,2)
\]
及其相对均匀基线密度
\[
r_n^{(i)}(c_i;u):=|G_i|\,\mu_{n,u}^{(i)}(c_i).
\]
定义隐藏角色能量
\[
\mathcal E_{n,u}^{\mathrm{hid}}(\pi)
:=
\frac{1}{B_n(u)^2}
\sum_{\chi\in \widehat G_2\setminus K^\perp}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
若
\[
\eta(u)<\lambda_1(u),
\]
则有精确二阶展开
\[
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(2)}\Vert u_{G_2}\bigr)
-
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(1)}\Vert u_{G_1}\bigr)
=
\frac12\,\mathcal E_{n,u}^{\mathrm{hid}}(\pi)
+
O\!\left(\Bigl(\frac{\eta(u)}{\lambda_1(u)}\Bigr)^{3n}\right).
\]
因此，群商所删去的全部 primitive Chebotarev 相对熵，在主阶上恰等于隐藏角色通道二次能量之一半。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part64ac-abelian-quotient-kl-chain}，
\[
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(2)}\Vert u_{G_2}\bigr)
-
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(1)}\Vert u_{G_1}\bigr)
=
\sum_{c_1\in G_1}
\mu_{n,u}^{(1)}(c_1)\,
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(c_1),\mathrm{fib}}\Vert u_{\pi^{-1}(c_1)}\bigr).
\]
记
\[
a(c_1):=r_n^{(1)}(c_1;u),
\qquad
b(c_2):=r_n^{(2)}(c_2;u),
\qquad
\delta(c_2):=b(c_2)-a(\pi(c_2)).
\]
则在纤维 \(\pi^{-1}(c_1)\) 上，
\[
\frac{\mu_{n,u}^{(c_1),\mathrm{fib}}(c_2)}{u_{\pi^{-1}(c_1)}(c_2)}
=
\frac{b(c_2)}{a(c_1)}.
\]
因此熵损失可改写为
\[
\frac{1}{|G_2|}\sum_{c_2\in G_2}
b(c_2)\log\frac{b(c_2)}{a(\pi(c_2))}.
\]

由定理 \ref{thm:xi-time-part64ab-primitive-chebotarev-haar-tv-exponent}，在有限群上 primitive Chebotarev 分布以指数率趋于 Haar 分布，故存在
\[
\varepsilon_n:=\left(\frac{\eta(u)}{\lambda_1(u)}\right)^n
\]
使
\[
a(c_1)=1+O(\varepsilon_n),
\qquad
b(c_2)=1+O(\varepsilon_n)
\]
在 \(c_1\in G_1\)、\(c_2\in G_2\) 上一致成立，从而 \(\delta(c_2)=O(\varepsilon_n)\) 一致。于是
\[
\log\frac{b}{a}
=
\log\Bigl(1+\frac{\delta}{a}\Bigr)
=
\frac{\delta}{a}-\frac{\delta^2}{2a^2}+O(\varepsilon_n^3),
\]
并且
\[
b\log\frac{b}{a}
=
\delta+\frac{\delta^2}{2a}+O(\varepsilon_n^3).
\]
对 \(G_2\) 求和时，线性项消失，因为对每个 \(c_1\in G_1\),
\[
\sum_{c_2\in\pi^{-1}(c_1)}\delta(c_2)=0
\]
恰表达了 \(r_n^{(1)}\) 是 \(r_n^{(2)}\) 的纤维平均。故
\[
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(2)}\Vert u_{G_2}\bigr)
-
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(1)}\Vert u_{G_1}\bigr)
=
\frac{1}{2|G_2|}\sum_{c_2\in G_2}\delta(c_2)^2
+
O(\varepsilon_n^3).
\]
再由定理 \ref{thm:xi-time-part64ac-visible-hidden-character-energy-splitting} 的第三式，
\[
\frac{1}{|G_2|}\sum_{c_2\in G_2}\delta(c_2)^2
=
\mathcal E_{n,u}^{\mathrm{hid}}(\pi).
\]
代回即得结论。证毕。
\end{proof}

\begin{theorem}[隐藏主导通道的熵障碍]\label{thm:conclusion-chebotarev-hiddenchannel-dominant-entropy-barrier}
沿用定理 \ref{thm:conclusion-chebotarev-hiddenchannel-kl-second-order} 的设定，并定义隐藏谱半径
\[
\eta_{\mathrm{hid}}(u,\pi)
:=
\max_{\chi\in \widehat G_2\setminus K^\perp}
\rho\!\bigl(M_{K,\chi}(u)\bigr).
\]
假设存在唯一字符
\[
\chi_\ast\in \widehat G_2\setminus K^\perp
\]
与某个简单特征值 \(\mu_\ast\in\Spec(M_{K,\chi_\ast}(u))\) 满足
\[
|\mu_\ast|=\eta_{\mathrm{hid}}(u,\pi),
\]
且该通道在 \(\mu_\ast\) 处存在谱隙。则存在常数 \(c_-,c_+>0\) 与无穷子序列 \(n_j\to\infty\)，使
\[
c_-\left(\frac{\eta_{\mathrm{hid}}(u,\pi)}{\lambda_1(u)}\right)^{2n_j}
\le
D_{\mathrm{KL}}\!\bigl(\mu_{n_j,u}^{(2)}\Vert u_{G_2}\bigr)
-
D_{\mathrm{KL}}\!\bigl(\mu_{n_j,u}^{(1)}\Vert u_{G_1}\bigr)
\le
c_+\left(\frac{\eta_{\mathrm{hid}}(u,\pi)}{\lambda_1(u)}\right)^{2n_j}.
\]
若进一步 \(\mu_\ast/\eta_{\mathrm{hid}}(u,\pi)\) 为单位根，则上式可在某条正密度算术级数上成立。

因此，群商是否真正抹去最慢的算术障碍，完全由该商是否删去隐藏主导字符通道决定。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part64ac-visible-hidden-character-energy-splitting}，
\[
\mathcal E_{n,u}^{\mathrm{hid}}(\pi)
=
\frac{1}{B_n(u)^2}
\sum_{\chi\in \widehat G_2\setminus K^\perp}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
对唯一隐藏主导字符 \(\chi_\ast\)，由有限维谱分解与简单特征值的谱隙，
\[
B_n^{(2)}(\chi_\ast;u)
=
c_\ast\,\frac{\mu_\ast^n}{n}
+
O\!\left(\frac{\eta_-^n}{n}\right)
\]
对某个 \(c_\ast\neq 0\) 与某个 \(\eta_-<\eta_{\mathrm{hid}}(u,\pi)\) 成立。于是沿无穷多 \(n\) 有
\[
\bigl|B_n^{(2)}(\chi_\ast;u)\bigr|
\ge
c\,\frac{\eta_{\mathrm{hid}}(u,\pi)^n}{n}
\]
对某个 \(c>0\) 成立；若 \(\mu_\ast/\eta_{\mathrm{hid}}(u,\pi)\) 为单位根，则这一相位回归在某条算术级数上稳定发生。

另一方面，对所有隐藏字符 \(\chi\notin K^\perp\)，都有统一上界
\[
\bigl|B_n^{(2)}(\chi;u)\bigr|
=
O\!\left(\frac{\eta_{\mathrm{hid}}(u,\pi)^n}{n}\right).
\]
再由 primitive Perron 渐近
\[
B_n(u)\sim c_0\,\frac{\lambda_1(u)^n}{n}
\]
（其中 \(c_0>0\)），即得沿上述无穷子序列
\[
c_-'\left(\frac{\eta_{\mathrm{hid}}(u,\pi)}{\lambda_1(u)}\right)^{2n_j}
\le
\mathcal E_{n_j,u}^{\mathrm{hid}}(\pi)
\le
c_+'\left(\frac{\eta_{\mathrm{hid}}(u,\pi)}{\lambda_1(u)}\right)^{2n_j}
\]
对某些 \(c_-',c_+'>0\) 成立。

最后由定理 \ref{thm:conclusion-chebotarev-hiddenchannel-kl-second-order}，
\[
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(2)}\Vert u_{G_2}\bigr)
-
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}^{(1)}\Vert u_{G_1}\bigr)
=
\frac12\,\mathcal E_{n,u}^{\mathrm{hid}}(\pi)
+
O\!\left(\Bigl(\frac{\eta(u)}{\lambda_1(u)}\Bigr)^{3n}\right).
\]
由于
\[
\eta_{\mathrm{hid}}(u,\pi)\le \eta(u)<\lambda_1(u),
\]
三阶项严格低于上述二阶主项，故可吸收入常数，得到所示双边界。证毕。
\end{proof}

\begin{theorem}[隐藏 Artin 通道的幽灵振荡]\label{thm:conclusion-hidden-artin-ghost-oscillation}
沿用定理 \ref{thm:conclusion-chebotarev-hiddenchannel-dominant-entropy-barrier} 的设定。若唯一隐藏主导字符
\[
\chi_\ast\in \widehat G_2\setminus K^\perp
\]
及其简单特征值 \(\mu_\ast\) 满足
\[
|\mu_\ast|=\eta_{\mathrm{hid}}(u,\pi)
\]
且在 \(\mu_\ast\) 处具有谱隙，则存在一个零均值类函数
\[
\varepsilon:G_2\to \CC,
\qquad
\pi_\ast\varepsilon=0,
\]
使得相应 primitive 统计量
\[
\Pi_n(\varepsilon;u)
:=
\sum_{c_2\in G_2}\varepsilon(c_2)\,B_{n,c_2}^{(2)}(u)
\]
满足渐近展开
\[
\Pi_n(\varepsilon;u)
=
c_\varepsilon\,\frac{\mu_\ast^n}{n}
+
O\!\left(\frac{\eta_-^n}{n}\right),
\qquad
c_\varepsilon\neq 0,
\]
其中 \(\eta_-<\eta_{\mathrm{hid}}(u,\pi)\)。特别地，存在常数 \(c>0\) 与无穷多个 \(n\) 使
\[
|\Pi_n(\varepsilon;u)|
\ge
c\,\frac{\eta_{\mathrm{hid}}(u,\pi)^n}{n},
\]
但其在商层上的全部推前统计恒等为零。
\end{theorem}
\begin{proof}
取
\[
\varepsilon:=\overline{\chi_\ast}.
\]
由于 \(\chi_\ast\neq \mathbf 1\) 是有限阿贝尔群上的非平凡角色，故
\[
\sum_{c_2\in G_2}\varepsilon(c_2)=0,
\]
从而 \(\varepsilon\) 为零均值类函数。又因 \(\chi_\ast\notin K^\perp\)，其在 \(K\) 上非平凡；于是对任意 \(c_1\in G_1\) 与任意提升 \(\widetilde c_1\in \pi^{-1}(c_1)\)，有
\[
(\pi_\ast\varepsilon)(c_1)
=
\sum_{k\in K}\varepsilon(\widetilde c_1k)
=
\varepsilon(\widetilde c_1)\sum_{k\in K}\varepsilon(k)
=
0,
\]
故 \(\pi_\ast\varepsilon=0\)。

另一方面，由定义
\[
\Pi_n(\varepsilon;u)
=
\sum_{c_2\in G_2}\overline{\chi_\ast(c_2)}\,B_{n,c_2}^{(2)}(u)
=
B_n^{(2)}(\chi_\ast;u),
\]
而定理 \ref{thm:xi-time-part64ac-dominant-channel-pole-order-witness-oscillation} 对唯一主导隐藏通道正给出
\[
B_n^{(2)}(\chi_\ast;u)
=
c_\varepsilon\,\frac{\mu_\ast^n}{n}
+
O\!\left(\frac{\eta_-^n}{n}\right),
\qquad
c_\varepsilon\neq 0,
\]
并沿无穷多个 \(n\) 有对应的 \(\Omega(\eta_{\mathrm{hid}}(u,\pi)^n/n)\) 下界。由于 \(\pi_\ast\varepsilon\equiv 0\)，粗层上由 \(\varepsilon\) 诱导的任何推前统计都恒为零，故该通道在细层主量级振荡而在商层完全不可见。证毕。
\end{proof}

\paragraph{统一读法}
这三条结果与 Artin-visible hull 的最小商原理并读后，可见 primitive Chebotarev 群商下的熵损失并不是抽象的粗化量，而是被删去通道在二阶能量中的精确账本；一旦隐藏谱中出现唯一主导字符，该账本便沿无穷子序列呈现不可压低的指数障碍，而其尺度完全由最慢隐藏通道控制。更进一步，determinant-活跃通道与零均值类函数活跃通道共同切出一个唯一最小的可见壳；最优二主项见证 \(\psi_{\ast,u}\) 亦已在其自身的最小承载商上闭合；多个证书族的融合则对应于这些可见壳在直积中的对角像。于是，商熵损失、幽灵振荡、完整 determinant 与 extremal class-function witness 不再彼此分离，而被压缩为同一个 Artin-visible hull 的不同投影。
