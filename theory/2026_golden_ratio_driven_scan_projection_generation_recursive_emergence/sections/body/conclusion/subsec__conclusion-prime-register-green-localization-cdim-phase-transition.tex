\begin{theorem}[素数寄存器算法半群的 Green--秩几何完全化]\label{thm:conclusion-prime-register-green-rank-geometry-complete}
对
\[
N\in S_n
\]
定义 Green 秩
\[
\rho(N):=\operatorname{rk}(N):=\bigl|\operatorname{Im}(f_N)\bigr|.
\]
则 \((S_n,\star)\) 的 Green 两侧理想结构完全由 \(\rho\) 控制：
\[
N\mathrel{\mathcal J}M
\iff
\rho(N)=\rho(M),
\]
\[
N\le_{\mathcal J}M
\iff
\rho(N)\le \rho(M).
\]
特别地，对每个 \(1\le r\le n\)，秩至多 \(r\) 的双侧理想恰为
\[
I_{\le r}:=\{N\in S_n:\rho(N)\le r\},
\]
并满足显式基数公式
\[
|I_{\le r}|
=
\sum_{k=1}^{r}\binom{n}{k}\,k!\,S(n,k),
\]
其中 \(S(n,k)\) 为第二类 Stirling 数。最小双侧理想则为
\[
K_n:=I_{\le 1}=\{N\in S_n:\rho(N)=1\},
\qquad
|K_n|=n,
\]
且 \(K_n\) 作为子半群同构于一个 \(n\) 元左零带。
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-prime-register-green-rank-stratification} 已经给出
\[
S_n\star N\star S_n
=
\{M\in S_n:\operatorname{rk}(M)\le \operatorname{rk}(N)\},
\]
并据此说明 \(\mathcal J\)-类恰按秩分层。将 \(\operatorname{rk}(N)\) 改记为 \(\rho(N)\)，即得前两条等价关系。

同一定理还给出秩恰为 \(k\) 的元素个数
\[
\#\{N\in S_n:\rho(N)=k\}
=
\binom{n}{k}\,k!\,S(n,k).
\]
对 \(1\le k\le r\) 求和便得 \(I_{\le r}\) 的基数公式。

最后，由定理 \ref{thm:xi-prime-register-green-rank-stratification} 的第 \(3\) 条，
\[
K_n
=
\{N\in S_n:\rho(N)=1\},
\qquad
|K_n|=n,
\]
并且该层的乘法满足左零律。故 \(K_n\) 是唯一极小双侧理想，且同构于 \(n\) 元左零带。证毕。
\end{proof}

\begin{theorem}[素数寄存器幂等元的有限 retraction 正规形]\label{thm:conclusion-prime-register-idempotent-retraction-normal-form}
\leanverified{paper\_conclusion\_prime\_register\_idempotent\_retraction\_normal\_form}
对
\[
N=\prod_{i=0}^{n-1}q_i^{a_i}\in S_n,
\qquad
f_N(i)=a_i,
\]
下列命题等价：
\begin{enumerate}
\item \(N\star N=N\)；
\item \(f_N\) 为幂等映射；
\item 指数向量满足
\[
a_{a_i}=a_i
\qquad (\forall i\in[n]).
\]
\end{enumerate}
若记
\[
F(N):=\{j\in[n]:a_j=j\},
\]
则上述条件又等价于
\[
a_i\in F(N)\qquad (\forall i\in[n]),
\]
并且此时 \(f_N\) 正是从 \([n]\) 到固定点集 \(F(N)\) 的一个 retraction。于是幂等元总数恰为
\[
\#\operatorname{Idem}(S_n)
=
\sum_{k=1}^{n}\binom{n}{k}k^{\,n-k}.
\]
更进一步，若 \(N\) 为秩 \(k\) 的幂等元，则其 \(\mathcal H\)-类上的最大子群同构于
\[
\mathfrak S_k.
\]
\end{theorem}
\begin{proof}
前三个命题的等价性正是推论 \ref{cor:killo-projection-idempotent-prime-register} 的内容。该推论同时给出
\[
\operatorname{Im}(f_N)=\operatorname{Fix}(f_N)=\{j\in[n]:a_j=j\}=F(N),
\]
故 \(f_N\) 的像等于其固定点集，于是 \(f_N\) 正是到 \(F(N)\) 的 retraction。反过来，任一到某个子集 \(F\subseteq[n]\) 的 retraction 都满足 \(a_{a_i}=a_i\)，从而对应于幂等元。

计数时，先选固定点集 \(F\subseteq[n]\) 且 \(|F|=k\)，共有 \(\binom{n}{k}\) 种；固定点上取值被强制为恒等，而其余 \(n-k\) 个点可任意映入 \(F\)，共有 \(k^{n-k}\) 种。故
\[
\#\operatorname{Idem}(S_n)=\sum_{k=1}^{n}\binom{n}{k}k^{n-k}.
\]

最后设 \(N\) 是秩 \(k\) 的幂等元，并记 \(F:=\operatorname{Im}(f_N)\)，\(|F|=k\)。经同构
\[
\Phi:(S_n,\star)\xrightarrow{\sim}([n]^{[n]},\circ)
\]
转到全变换半群
\[
T_n:=([n]^{[n]},\circ)
\]
后，\(f_N\) 成为对 \(F\) 的 retraction。记 \(e:=f_N\)。考虑幂等元 \(e\) 处的局部单子
\[
eT_ne:=\{\,e\circ g\circ e:\ g\in T_n\,\}.
\]
限制到 \(F\) 上给出半群同构
\[
\operatorname{res}_F:eT_ne\xrightarrow{\sim}T_F,
\qquad
h\longmapsto h|_F,
\]
其逆由
\[
u\longmapsto \iota_F\circ u\circ e
\]
给出，其中 \(\iota_F:F\hookrightarrow [n]\) 为包含映射。于是 \(eT_ne\) 的可逆元恰对应于 \(T_F\) 的可逆元，也就是 \(F\) 上的全置换群 \(\mathfrak S(F)\)。而幂等元 \(e\) 的最大子群正是 \(eT_ne\) 的单位群，故该最大子群同构于
\[
\mathfrak S(F)\cong \mathfrak S_k.
\]
证毕。
\end{proof}

\begin{theorem}[可比链型幂等子族的 \texorpdfstring{$\gcd$}{gcd}--带坍缩]\label{thm:conclusion-prime-register-comparable-idempotent-gcd-band}
\leanverified{paper\_conclusion\_prime\_register\_comparable\_idempotent\_gcd\_band}
对每个满足
\[
0\in F\subseteq[n]
\]
的子集，记链型规范幂等元为
\[
N_F:=\Phi^{-1}(I_F),
\qquad
\kappa(N_F):=\prod_{i\in F}q_i.
\]
若 \(F\) 与 \(G\) 在包含关系下可比，则成立精确交换律
\[
N_F\star N_G=N_G\star N_F=N_{F\cap G},
\]
以及
\[
\kappa(N_F\star N_G)=\gcd(\kappa(N_F),\kappa(N_G)).
\]
并且
\[
\operatorname{Im}(f_{N_F\star N_G})
=
\operatorname{Fix}(f_{N_F\star N_G})
=
F\cap G.
\]
因此，任一可比链所张成的链型幂等子族，在自指合成 \(\star\) 下都塌缩为一个交换幂等的 \(\gcd\)-带。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:xi-chain-idempotent-star-saturation-comparable-gcd} 与其后续像集公式的直接重述。证毕。
\end{proof}

\begin{theorem}[有限秩加法账本的无限幂塔碰撞]\label{thm:conclusion-finite-rank-additive-ledger-power-tower-collision}
设 \(A\) 为任意有限生成交换可消去幺半群，并设
\[
\Psi:\NN_{\times}\longrightarrow A
\]
为幺半群同态，其中 \(A\) 以加法记。则必存在互异整数
\[
u\neq v
\]
使得
\[
\Psi(u^t)=\Psi(v^t)\qquad (\forall t\ge 1).
\]
因此，任何有限秩加法账本对乘法结构的同态化，一旦存在，便必然携带一条无限幂塔碰撞族。

更特别地，若
\[
\Phi:\NN_{\times}\longrightarrow \ZZ\oplus L,
\qquad
\Phi(ab)=\Phi(a)+\Phi(b),
\]
其中 \(L\) 为有限生成交换群，则 \(\Phi\) 不可能单射；一旦 \(\Phi\) 非单射，上述幂塔碰撞便自动出现。
\end{theorem}
\begin{proof}
由定理 \ref{thm:killo-prime-freedom-non-finitizability}，不存在幺半群单射
\[
\NN_{\times}\hookrightarrow A
\]
到任何有限生成交换可消去幺半群 \(A\) 中。因此 \(\Psi\) 必为非单射。取 \(u\neq v\) 使
\[
\Psi(u)=\Psi(v).
\]
由于 \(\Psi\) 为从乘法到加法的幺半群同态，对任意 \(t\ge 1\) 都有
\[
\Psi(u^t)=t\,\Psi(u)=t\,\Psi(v)=\Psi(v^t).
\]
故同一对碰撞自动沿幂塔无限延伸。

当 \(A=\ZZ\oplus L\) 时，第一句同样适用；此外，定理 \ref{thm:killo-godel-compression-not-finite-rank-homologizable} 进一步排除了所有这类有限生成 \(L\) 上的单射同态化账本。证毕。
\end{proof}

\begin{theorem}[局部化对偶的素数支持函子性]\label{thm:conclusion-localized-dual-primesupport-functoriality}
\leanverified{paper\_conclusion\_localized\_dual\_primesupport\_functoriality}
对有限素数集 \(S,T\subset\mathbb P\)，记
\[
G_S:=\ZZ[S^{-1}],
\qquad
\Sigma_S:=\widehat{G_S}.
\]
则下列命题等价：
\[
\exists\ \text{连续满射}\ \Sigma_S\twoheadrightarrow \Sigma_T
\iff
T\subseteq S.
\]
特别地，
\[
\Sigma_S\cong \Sigma_T
\iff
S=T.
\]
换言之，有限素数局部化对偶的同构型本身已是素数支持集的完备编码。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:xi-localized-solenoid-terminal-object-poset} 的第一部分与定理 \ref{thm:xi-localized-integers-divisibility-topological-rigidity} 的直接组合。更具体地，连续满射
\[
\Sigma_S\twoheadrightarrow \Sigma_T
\]
经 Pontryagin 对偶等价于离散端的单射
\[
G_T\hookrightarrow G_S,
\]
而这等价于 \(T\subseteq S\)。双向应用该判别便得到
\[
\Sigma_S\cong \Sigma_T\iff S=T.
\]
证毕。
\end{proof}

\begin{theorem}[全局乘法到有限支撑局部化的圆维相变]\label{thm:conclusion-global-multiplication-to-localized-cdim-phase-transition}
记实现层半圆维与结构层同态半圆维分别为
\[
\cdim^{\mathrm{impl}}(X)
:=
\inf\left\{\frac{m}{2}:\ \exists\,\text{单射 }X\hookrightarrow \NN^m\right\},
\]
\[
\cdim_{+}^{\mathrm{hom}}(M)
:=
\inf\left\{\frac{k}{2}:\ \exists\,\text{幺半群单射同态 }M\hookrightarrow (\NN^k,+)\right\}.
\]
则存在严格相变：
\[
\cdim^{\mathrm{impl}}(\NN_{\times})=\frac12,
\qquad
\cdim_{+}^{\mathrm{hom}}(\NN_{\times})=+\infty.
\]
另一方面，对任意有限素数集 \(S\subset\mathbb P\)，若
\[
G_S:=\ZZ[S^{-1}],
\]
则在有限秩扩展口径下
\[
\cdim_{\mathrm{fr}}(G_S)=1,
\]
且 Pontryagin 对偶满足短正合列
\[
0\longrightarrow \prod_{p\in S}\ZZ_p
\longrightarrow \widehat{G_S}
\longrightarrow \TT
\longrightarrow 0.
\]
因此，从全局乘法对象 \(\NN_{\times}\) 过渡到任意固定有限支撑局部化 \(G_S\) 时，结构层的连通圆维成本从 \(+\infty\) 降到 \(1\)；但全部素数支持信息并未消失，而是完整转移到可恢复的 profinite 核
\[
\prod_{p\in S}\ZZ_p
\]
之中。
\end{theorem}
\begin{proof}
前两式正是命题 \ref{prop:killo-implementation-vs-structural-half-circle-dimension} 的结论。至于局部化族，定理 \ref{thm:killo-localization-circle-dimension-prime-ledger-splitting} 已给出
\[
\cdim_{\mathrm{fr}}(G_S)=1
\]
以及短正合列
\[
0\longrightarrow \prod_{p\in S}\ZZ_p
\longrightarrow \widehat{G_S}
\longrightarrow \TT
\longrightarrow 0.
\]
同一定理还说明核的 \(p\)-主因子非平凡当且仅当 \(p\in S\)，故 \(S\) 可由该 profinite 核完全恢复。于是所谓“维数骤降”只发生在连通圆因子方向，而不发生在素数支持方向。证毕。
\end{proof}

\begin{conjecture}[有限算法半群塔与局部化账本的极限兼容]\label{conj:conclusion-prime-register-semigroup-localization-limit-compatibility}
设 \((S_n,\star)_{n\ge 1}\) 为素数寄存器有限算法半群塔，并以
\[
E_n^{\mathrm{ch}}=\{N_F:\ 0\in F\subseteq[n]\}
\]
记其链型幂等子带。则应存在一个严格的极限兼容机制，使得：
\begin{enumerate}
\item \(E_n^{\mathrm{ch}}\) 在 \(n\to\infty\) 时收敛为“有限素数支持投影”的初等骨架；
\item 其可比链上的坍缩律
\[
N_F\star N_G=N_{F\cap G},
\qquad
\kappa(N_F\star N_G)=\gcd(\kappa(N_F),\kappa(N_G))
\]
在极限上提升为局部化账本中的素数支持交；
\item 与此同时，任何试图以有限秩加法账本同态化整个 \(\NN_{\times}\) 的方案都必须失败；唯有在显式选定有限支持 \(S\) 之后，才会出现
\[
\cdim_{\mathrm{fr}}(G_S)=1
\]
这样的 rank-\(1\) 连通修复。
\end{enumerate}
\end{conjecture}

\paragraph{统一读法}
定理 \ref{thm:conclusion-prime-register-green-rank-geometry-complete}、\ref{thm:conclusion-prime-register-idempotent-retraction-normal-form} 与 \ref{thm:conclusion-prime-register-comparable-idempotent-gcd-band} 表明，素数寄存器有限算法半群并不只是“把函数编码进素数指数”的一次静态翻译；它在内部同时携带完整的 Green--秩几何、retraction 型幂等锥，以及在可比链上塌缩为 \(\gcd\)-带的交换极相。定理 \ref{thm:conclusion-finite-rank-additive-ledger-power-tower-collision} 则说明，任何有限秩加法账本一旦试图同态承载全局乘法结构，失败都不会停留在单点，而会沿整个幂塔无限复制。再由定理 \ref{thm:conclusion-zero-dimensional-ledger-positive-support-orthogonality} 与推论 \ref{cor:conclusion-localized-additive-systems-maximal-onecircle-class}，零维核虽可无限复杂，却不能替代任何正圆维支撑；有限素数局部化恰构成“单可见圆因子 + 可恢复 prime-ledger 核”仍能同时成立的极大类。更进一步，持续分支 natural extension 的最优支持链必然严格增长到 \(\omega\)-层，因此其实现层与结构层继续保持 \(\frac12\) 对 \(+\infty\) 的刚性分裂；而对任意有限局部化 \(G_S\)，所有连通 Lie 可见量都只能看到公共圆商 \(\TT\)，新增 slice growth 只能在 profinite 核中显影。最后，命题 \ref{prop:conclusion-implementation-injection-never-structural-dim-drop} 说明，实现层的 Gödel 注入或集合论高效编码并不会自动产生结构层的圆维缩减。于是，有限算法半群、有限秩账本障碍与局部化对偶的圆维重分配并非三条彼此无关的链，而是同一 prime-ledger 结构在有限支持、全局乘法与对偶完成化三个尺度上的不同显影。
