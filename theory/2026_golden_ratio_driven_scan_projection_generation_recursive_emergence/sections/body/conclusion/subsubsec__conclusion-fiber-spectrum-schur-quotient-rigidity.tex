\paragraph{纤维谱的初充分性、Schur 平坦化与商结构刚性}
沿用结论 \ref{con:xi-terminal-fiber-geometry-complete-spectrum}、定理 \ref{thm:xi-time-part10-fiber-topology-rankone-law}、定理 \ref{thm:xi-time-part62dg-fiber-reduced-euler-witten-parity}、定理 \ref{thm:conclusion-window6-ordinary-binary-strict-majorization} 与推论 \ref{cor:conclusion-window6-majorization-entropy-capacity-order} 的记号。记
\[
\Gamma_x\cong\bigsqcup_{i=1}^{r_x}P_{\ell_i(x)},
\qquad
\mathcal F_x\cong \prod_{i=1}^{r_x}\Gamma_{\ell_i(x)},
\qquad
K_x:=\operatorname{Ind}^{\Delta}(\Gamma_x),
\]
\[
\mathsf{Spec}_{\square}(x):=\{\ell_1(x),\dots,\ell_{r_x}(x)\},
\qquad
m_\ell(x):=\#\{\,i:\ell_i(x)=\ell\,\},
\qquad
\tau(x):=\sum_i\left\lfloor\frac{\ell_i(x)+1}{3}\right\rfloor,
\]
并定义可实现纤维谱集合
\[
\mathscr S_{\square}:=
\Bigl\{
\mathsf{Spec}_{\square}(x):\ x\in\bigsqcup_{m\ge 1}X_m
\Bigr\}.
\]
前述链条已经分别把纤维重构图的 Cartesian--Fibonacci 分解、独立集复形的球面/可缩二分、以及 window-\(6\) 上 ordinary fold 与 binary uplift 的严格主序固定下来。把这些接口并置之后，还可进一步压出如下几条后继后果。

\begin{theorem}[纤维谱作为图—拓扑—对称函子的初充分统计量]\label{thm:conclusion-fiber-spectrum-initial-sufficient-statistic}
\leanverified{paper\_conclusion\_fiber\_spectrum\_initial\_sufficient\_statistic}
设 \(\mathcal V\) 为任意集合，且
\[
\mathcal I:\{\mathcal F_x:\ x\in\bigsqcup_{m\ge 1}X_m\}\longrightarrow \mathcal V
\]
为任一仅依赖于图同构类的量。则存在唯一映射
\[
\widetilde{\mathcal I}:\mathscr S_{\square}\longrightarrow \mathcal V
\]
使得对所有稳定类型 \(x\) 都有
\[
\mathcal I(\mathcal F_x)=\widetilde{\mathcal I}\!\bigl(\mathsf{Spec}_{\square}(x)\bigr).
\]
特别地，下列对象全部唯一经由 \(\mathsf{Spec}_{\square}(x)\) 因子化：
\[
\mathcal F_x\ \text{的图同构类},
\qquad
|\operatorname{Aut}(\mathcal F_x)|,
\qquad
|K_x|\ \text{的同伦型},
\qquad
\tau(x).
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:pom-fiber-reconstruction-spectrum-complete}，
\(
\mathsf{Spec}_{\square}(x)
\)
是 \(\mathcal F_x\) 的完备图同构不变量。故任一只依赖图同构类的量都在 \(\mathscr S_{\square}\) 上存在唯一的下降映射 \(\widetilde{\mathcal I}\)。

其中，
\(
|\operatorname{Aut}(\mathcal F_x)|
\)
由定理 \ref{thm:pom-fiber-reconstruction-aut-group} 的显式群阶公式完全决定；
\(
|K_x|
\)
的同伦型由定理 \ref{thm:pom-fiber-independence-complex-classification} 与定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 完全决定；
而
\(
\tau(x)
\)
本身就是 \(\mathsf{Spec}_{\square}(x)\) 的显式函数。证毕。
\end{proof}

\begin{proposition}[Fibonacci 乘积统计对图谱分类的先天失真]\label{prop:conclusion-fiber-multiplicity-product-noninjective}
\leanverified{paper\_conclusion\_fiber\_multiplicity\_product\_noninjective}
定义有限多重集上的 Fibonacci 乘积映射
\[
\mathfrak m(\Lambda):=\prod_{\ell\in\Lambda}F_{\ell+2}.
\]
则 \(\mathfrak m\) 不是单射；事实上
\[
\mathfrak m(\{4\})=F_6=8=F_3^3=\mathfrak m(\{1,1,1\}).
\]
因此，任何仅依赖纤维多重度
\(
d_m(x)=|\Fold_m^{-1}(x)|
\)
的统计量，在一般情形下都不可能恢复纤维重构图 \(\mathcal F_x\) 的图同构型。更具体地，只要某个折叠子族同时实现谱
\(
\{4\}
\)
与
\(
\{1,1,1\}
\)，
便存在相同多重度对应不同内部几何的现象。
\end{proposition}
\begin{proof}
由定理 \ref{thm:pom-fiber-indset-factorization}，
\[
d_m(x)=\prod_i F_{\ell_i(x)+2}
=
\mathfrak m\!\bigl(\mathsf{Spec}_{\square}(x)\bigr).
\]
而上式中的具体例子已经表明 \(\mathfrak m\) 在有限多重集上并非单射。
另一方面，由推论 \ref{cor:pom-fiber-reconstruction-spectrum-complete}，
若 \(\mathcal F_x\cong \mathcal F_y\)，则
\(
\mathsf{Spec}_{\square}(x)=\mathsf{Spec}_{\square}(y)
\)。
故多重度统计在原则上先天地丢失图谱信息。证毕。
\end{proof}

\begin{theorem}[固定分量数下的对称熵极值定律]\label{thm:conclusion-fiber-symmetry-entropy-extremals}
\leanverified{paper\_conclusion\_fiber\_symmetry\_entropy\_extremals}
记
\[
r_x=\sum_{\ell}m_\ell(x).
\]
则有精确公式
\[
|\operatorname{Aut}(\mathcal F_x)|=2^{r_x}\prod_{\ell}m_\ell(x)!,
\]
从而成立双边界
\[
2^{r_x}\le |\operatorname{Aut}(\mathcal F_x)|\le 2^{r_x}r_x!.
\]
并且：
\begin{enumerate}
\item 左等号成立，当且仅当全部分量长度两两不同；
\item 右等号成立，当且仅当全部分量长度完全相等。
\end{enumerate}
若定义对称熵
\[
S_{\mathrm{sym}}(x):=\log |\operatorname{Aut}(\mathcal F_x)|,
\]
则
\[
r_x\log 2\le S_{\mathrm{sym}}(x)\le r_x\log 2+\log(r_x!).
\]
\end{theorem}
\begin{proof}
第一条公式正是定理 \ref{thm:pom-fiber-reconstruction-aut-group} 的群阶表达式。
又由于
\(
\sum_{\ell}m_\ell(x)=r_x
\)，
有
\[
1\le \prod_\ell m_\ell(x)!\le r_x!.
\]
当且仅当所有 \(m_\ell(x)\in\{0,1\}\) 时左端取等，也即全部分量长度两两不同；当且仅当全部质量集中在单个长度桶中时右端取等，也即全部分量长度完全相等。取对数即得对称熵的双边界。证毕。
\end{proof}

\begin{theorem}[图同构诱导的同伦刚性与奇偶守恒]\label{thm:conclusion-fiber-isomorphic-homotopy-rigidity}
设 \(x\in X_m\)、\(y\in X_n\) 且
\[
\mathcal F_x\cong \mathcal F_y.
\]
则
\[
|K_x|\simeq |K_y|.
\]
更精确地：
\begin{enumerate}
\item \(|K_x|\) 可缩当且仅当 \(|K_y|\) 可缩；
\item 若 \(|K_x|\) 非可缩，则
\[
|K_x|\simeq S^{\tau(x)-1},
\qquad
|K_y|\simeq S^{\tau(y)-1},
\qquad
\tau(x)=\tau(y);
\]
\item 同时有
\[
d_m(x)\equiv d_n(y)\pmod 2.
\]
\end{enumerate}
\end{theorem}
\begin{proof}
由推论 \ref{cor:pom-fiber-reconstruction-spectrum-complete}，
\[
\mathcal F_x\cong \mathcal F_y
\quad\Longrightarrow\quad
\mathsf{Spec}_{\square}(x)=\mathsf{Spec}_{\square}(y).
\]
因此 \(\tau(x)=\tau(y)\)。
再由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 与定理 \ref{thm:pom-fiber-independence-complex-classification}，
\(
\mathsf{Spec}_{\square}(x)
\)
一方面决定 \(|K_x|\) 是可缩还是球面，
另一方面在球面相中决定其维数
\(
\tau(x)-1
\)。
故 \(|K_x|\simeq |K_y|\)，并得到第 \(1\)、\(2\) 条。

最后，由同一定理，
\[
|K_x|\not\simeq *
\iff
d_m(x)\ \text{为奇},
\]
而可缩情形则等价于 \(d_m(x)\) 为偶。
由于 \(|K_x|\simeq |K_y|\)，二者奇偶性必一致，于是第 \(3\) 条成立。证毕。
\end{proof}

\begin{corollary}[拓扑极大化与乘法极大化的三层闭包及常数量级税]\label{cor:conclusion-fiber-topological-maximization-threelevel-tax}
定义
\[
\widetilde D_m:=\max\{d_m(x):x\in X_m,\ |K_x|\not\simeq *\},
\qquad
\overline d_m:=\frac{2^m}{F_{m+2}}.
\]
则对所有 \(m\ge 17\)，非可缩极大值始终只会在前三层纤维谱中切换：
\[
\widetilde D_m\in\{D_m,D_m^{(2)},D_m^{(3)}\}.
\]
并且
\[
\frac{1}{m}\log \widetilde D_m-\frac{1}{m}\log D_m\longrightarrow 0.
\]
更强地，
\[
\liminf_{m\to\infty}\frac{\widetilde D_m}{\overline d_m}
\ge
(1-\varphi^{-5})(1+C_\varphi)
\approx
1.1140121408.
\]
因此，要求纤维独立集复形非可缩并不会改变顶层纤维的指数率，但会留下严格可见的常数级拓扑税；与此同时，最大非可缩纤维仍在极限上严格高于平均纤维规模。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，
\[
\widetilde D_m\in\{D_m,D_m^{(2)},D_m^{(3)}\}
\qquad (m\ge 17).
\]
再由定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios}，
\(
\widetilde D_m/D_m
\)
沿每个模 \(6\) 子列都收敛到正有限常数，
故
\[
\frac{1}{m}\log \widetilde D_m-\frac{1}{m}\log D_m
=
\frac{1}{m}\log\!\left(\frac{\widetilde D_m}{D_m}\right)\to 0.
\]

另一方面，定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios} 给出
\[
\liminf_{m\to\infty}\frac{\widetilde D_m}{D_m}\ge 1-\varphi^{-5},
\]
而推论 \ref{cor:fold-max-fiber-constant-bias} 给出
\[
\liminf_{m\to\infty}\frac{D_m}{\overline d_m}\ge 1+C_\varphi.
\]
两式相乘即得所示下界。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的 Schur 平坦化与凸集中量同步下降]\label{thm:conclusion-window6-schur-flattening-convex-collapse}
记
\[
\mathbf p_6^{\mathrm{ord}}
:=
\frac{1}{64}\mathbf d_6^{\mathrm{ord},\downarrow},
\qquad
\mathbf p_6^{\mathrm{bin}}
:=
\frac{1}{64}\mathbf d_6^{\mathrm{bin},\downarrow},
\]
并定义
\[
G_6^{\mathrm{ord}}:=\Aut(\Fold_6),
\qquad
G_6^{\mathrm{bin}}:=\Aut(\Fold_6^{\mathrm{bin}}).
\]
则由严格主序
\[
\mathbf p_6^{\mathrm{ord}}\succ \mathbf p_6^{\mathrm{bin}}
\]
可推出：
\begin{enumerate}
\item 对任意定义在 \(21\)-维概率单纯形上的对称凸泛函 \(\Phi\)，有
\[
\Phi(\mathbf p_6^{\mathrm{bin}})\le \Phi(\mathbf p_6^{\mathrm{ord}});
\]
\item 对任意严格 Schur--凹的对称泛函 \(H\)，有
\[
H(\mathbf p_6^{\mathrm{bin}})>H(\mathbf p_6^{\mathrm{ord}});
\]
\item 特别地，
\[
\max_i p_{6,i}^{\mathrm{bin}}=\frac{4}{64}
<
\frac{5}{64}=\max_i p_{6,i}^{\mathrm{ord}},
\]
\[
\sum_i\bigl(p_{6,i}^{\mathrm{bin}}\bigr)^2=\frac{212}{64^2}
<
\frac{220}{64^2}=
\sum_i\bigl(p_{6,i}^{\mathrm{ord}}\bigr)^2;
\]
\item 并且全局 gauge 体积满足严格不等式
\[
\log |G_6^{\mathrm{bin}}|<\log |G_6^{\mathrm{ord}}|.
\]
\end{enumerate}
因此，binary uplift 在 window-\(6\) 上不仅提升了 Shannon 熵与全部 Rényi 熵，而且同步压低了最大原子质量、二阶碰撞能量与规范群体积。
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-window6-ordinary-binary-strict-majorization} 已给出
\(
\mathbf d_6^{\mathrm{ord},\downarrow}\succ \mathbf d_6^{\mathrm{bin},\downarrow}
\)。
归一化不改变主序关系，故
\(
\mathbf p_6^{\mathrm{ord}}\succ \mathbf p_6^{\mathrm{bin}}
\)。
第 \(1\) 条由 Hardy--Littlewood--Pólya 定理立即推出；第 \(2\) 条则由严格 Schur--凹性的定义得到。

第 \(3\) 条可直接由直方图
\[
1:2,\ 2:4,\ 3:8,\ 4:5,\ 5:2,
\qquad
2:8,\ 3:4,\ 4:9
\]
计算得到。
其中 ordinary 侧的平方和为
\[
2\cdot 1^2+4\cdot 2^2+8\cdot 3^2+5\cdot 4^2+2\cdot 5^2=220,
\]
binary 侧为
\[
8\cdot 2^2+4\cdot 3^2+9\cdot 4^2=212.
\]

对第 \(4\) 条，注意
\[
\log |G_6^{\mathrm{ord}}|
=
\sum_{w\in X_6}\log\bigl(d_6^{\mathrm{ord}}(w)!\bigr),
\qquad
\log |G_6^{\mathrm{bin}}|
=
\sum_{w\in X_6}\log\bigl(d_6^{\mathrm{bin}}(w)!\bigr).
\]
令
\[
g(t):=\log\Gamma(t+1)\qquad (t\ge 1).
\]
由于 trigamma 函数恒正，\(g\) 在 \([1,\infty)\) 上严格凸。再由
\(
\mathbf d_6^{\mathrm{ord},\downarrow}\succ \mathbf d_6^{\mathrm{bin},\downarrow}
\)
与 Karamata 不等式，得到
\[
\sum_i g\!\bigl(\mathbf d_{6,i}^{\mathrm{ord},\downarrow}\bigr)
>
\sum_i g\!\bigl(\mathbf d_{6,i}^{\mathrm{bin},\downarrow}\bigr).
\]
而对正整数 \(n\) 有 \(g(n)=\log(n!)\)，于是
\[
\log |G_6^{\mathrm{ord}}|>\log |G_6^{\mathrm{bin}}|.
\]
证毕。
\end{proof}

\begin{corollary}[Schur 平坦化并不逼近齐轨道商]\label{cor:conclusion-window6-schur-flattening-not-quotient}
window-\(6\) 的 binary uplift 同时具有如下三种彼此张力极强的性质：
\begin{enumerate}
\item 相对于 ordinary fold 实现严格 Schur 平坦化；
\item 全部熵型量严格增大，而最大纤维、碰撞能量与 gauge 体积同时下降；
\item 其纤维大小仍呈三值分布
\[
2:8,\qquad 3:4,\qquad 4:9,
\]
因而既不可能来自任何有限群的自由作用轨道划分，也不可能来自任何有限阿贝尔群的线性陪集划分。
\end{enumerate}
因此，在该窗口上，“更高熵、更低集中”与“更接近齐轨道商”是严格分离的两件事；Schur 平坦化并不会把多值纤维划分线性化为自由商结构。
\end{corollary}
\begin{proof}
第 \(1\) 条与第 \(2\) 条由定理 \ref{thm:conclusion-window6-schur-flattening-convex-collapse} 给出。
而第 \(3\) 条则由定理 \ref{thm:xi-time-part60acb-constant-fiber-free-group-action-criterion} 立即推出：任何来自自由有限群作用或有限阿贝尔群线性陪集的划分都必须是常纤维，而 window-\(6\) 的 binary 纤维直方图并非常纤维。证毕。
\end{proof}

\begin{conjecture}[可实现纤维谱的并半群闭包]\label{conj:conclusion-realizable-fiber-spectrum-semigroup-closure}
若
\[
\Lambda,\Lambda'\in \mathscr S_{\square},
\]
则其多重集并
\[
\Lambda\sqcup \Lambda'
\]
仍属于 \(\mathscr S_{\square}\)，并且可在某个更高分辨率层上实现。
\end{conjecture}

\noindent
若该猜想成立，则 \(\mathscr S_{\square}\) 将成为由连通生成元 \(\{\ell\}\) 所生成的交换并半群，而定理 \ref{thm:conclusion-fiber-spectrum-initial-sufficient-statistic} 中的图同构类、同伦理型、球面维数与对称熵都将提升为该并半群上的显式函子。与此同时，命题 \ref{prop:conclusion-fiber-multiplicity-product-noninjective} 的 Fibonacci 乘积非单射将不再只是抽象参数层上的现象，而会在真实折叠族内部系统性地实现。

\noindent
综上，纤维多重度并非主不变量；真正主导图几何、同伦理型与对称熵的是分量长度多重集 \(\mathsf{Spec}_{\square}(x)\)。另一方面，window-\(6\) 的局部谱 Schur 平坦化虽然同时压低了最大原子、碰撞能量与 gauge 体积，却并不能抹去非自由群胚性，也不能把拓扑极大化从其模 \(6\) 周期常数税中释放出来。由此形成的两条闭链分别是
\[
\mathsf{Spec}_{\square}(x)
\Longrightarrow
\text{图同构初充分性}
\Longrightarrow
\text{同伦/对称/球面维数统一因子化}
\Longrightarrow
\text{拓扑极大化仅在前三层内切换},
\]
以及
\[
\text{window-}6\text{ 严格主序}
\Longrightarrow
\text{Schur 平坦化}
\Longrightarrow
\text{凸集中量同步下降}
\Longrightarrow
\text{仍严格排斥自由商与线性陪集化}.
\]
