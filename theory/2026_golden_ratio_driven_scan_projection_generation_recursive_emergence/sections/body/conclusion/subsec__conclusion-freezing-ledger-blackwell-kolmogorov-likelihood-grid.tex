\begin{theorem}[冻结 escort 制度的双极账本完备性]\label{thm:conclusion-freezing-dual-ledger-completeness}
设 \(\{F_m\}\) 与 \(\{\widetilde F_m\}\) 为两列折叠系统；在同一固定 \(q>0\) 下，它们都满足冻结条件。记各自最大纤维尺度与最大层简并度为
\[
M_m:=\max_x d_m(x),
\qquad
\nu_m:=\#\{x:\ d_m(x)=M_m\},
\]
\[
\widetilde M_m:=\max_x \widetilde d_m(x),
\qquad
\widetilde \nu_m:=\#\{x:\ \widetilde d_m(x)=\widetilde M_m\}.
\]
若
\[
\frac{M_m}{\widetilde M_m}\to 1,
\qquad
\frac{\nu_m}{\widetilde \nu_m}\to 1,
\]
则对任意序列 \(L_m\ge 1\)、\(r_m\ge 2\)，只要
\[
\frac{L_m}{M_m}\to \lambda\in[0,\infty),
\qquad
\frac{r_m-1}{\nu_m}\to \Lambda\in[0,\infty),
\]
便有
\[
\mathrm{Succ}^{\mathrm{rev}}_{m,q}(L_m)
-
\widetilde{\mathrm{Succ}}^{\mathrm{rev}}_{m,q}(L_m)
\longrightarrow 0,
\]
且两边同趋于
\[
\min(1,\lambda).
\]
同时，对应的 Bayes--InfoNCE 最优值满足
\[
\mathcal L^\star_{m,q}(r_m)
-
\widetilde{\mathcal L}^\star_{m,q}(r_m)
\longrightarrow 0,
\]
且两边同趋于
\[
\EE\bigl[\log(1+\mathrm{Poisson}(\Lambda))\bigr].
\]
因此，在冻结 regime 中，可逆恢复与对比学习的极限行为分别只由两本彼此独立的账本 \(M_m\) 与 \(\nu_m\) 决定。
\end{theorem}

\begin{proof}
冻结区中的可逆成功率极限只依赖于比值 \(L_m/M_m\)，并收敛到 \(\min(1,\lambda)\)。由于
\[
\frac{L_m}{\widetilde M_m}
=
\frac{L_m}{M_m}\cdot \frac{M_m}{\widetilde M_m}
\longrightarrow \lambda,
\]
故两系统的可逆极限一致，从而差值趋于零。

另一方面，冻结区中的 Bayes--InfoNCE 极限只依赖于
\[
\frac{r_m-1}{\nu_m}.
\]
由
\[
\frac{r_m-1}{\widetilde \nu_m}
=
\frac{r_m-1}{\nu_m}\cdot \frac{\nu_m}{\widetilde \nu_m}
\longrightarrow \Lambda,
\]
可知两系统诱导出同一 Poisson 参量 \(\Lambda\)，于是最优值都收敛到
\[
\EE\bigl[\log(1+\mathrm{Poisson}(\Lambda))\bigr].
\]
两边相减即得结论。证毕。
\end{proof}

\begin{corollary}[尺度账本与简并账本的正交不可替代性]\label{cor:conclusion-scale-ledger-degeneracy-ledger-orthogonality}
\leanverified{paper\_conclusion\_scale\_ledger\_degeneracy\_ledger\_orthogonality}
在定理 \ref{thm:conclusion-freezing-dual-ledger-completeness} 的设定下：
\begin{enumerate}
  \item 若
  \[
  \frac{M_m}{\widetilde M_m}\to 1,
  \qquad
  \frac{\nu_m}{\widetilde \nu_m}\to c\neq 1,
  \]
  则存在序列 \(r_m\) 使两系统的可逆成功率极限相同，而 Bayes--InfoNCE 极限不同。
  \item 若
  \[
  \frac{\nu_m}{\widetilde \nu_m}\to 1,
  \qquad
  \frac{M_m}{\widetilde M_m}\to c\neq 1,
  \]
  则存在序列 \(L_m\) 使两系统的 Bayes--InfoNCE 极限相同，而可逆成功率极限不同。
\end{enumerate}
\end{corollary}

\begin{proof}
先证第 \(1\) 条。任取满足
\[
\frac{L_m}{M_m}\to \lambda
\]
的预算序列，则由定理 \ref{thm:conclusion-freezing-dual-ledger-completeness}，两系统的可逆极限都等于 \(\min(1,\lambda)\)。
再取 \(r_m\) 使
\[
\frac{r_m-1}{\nu_m}\to \Lambda>0.
\]
则另一系统对应的参量为
\[
\frac{r_m-1}{\widetilde \nu_m}\to \Lambda c.
\]
由于 \(\Lambda\mapsto \EE[\log(1+\mathrm{Poisson}(\Lambda))]\) 随 \(\Lambda\) 严格递增，二者的 Bayes--InfoNCE 极限不同。

再证第 \(2\) 条。不妨经重标记设 \(c>1\)。取 \(\lambda\in(1/c,1)\)，并令
\[
L_m:=\lfloor \lambda M_m\rfloor.
\]
则
\[
\frac{L_m}{M_m}\to \lambda,
\qquad
\frac{L_m}{\widetilde M_m}\to c\lambda>1.
\]
故两系统的可逆极限分别为 \(\lambda\) 与 \(1\)，从而不同。另一方面，只要取 \(r_m\) 满足
\[
\frac{r_m-1}{\nu_m}\to \Lambda,
\]
由于 \(\nu_m/\widetilde \nu_m\to 1\)，两系统的 Bayes--InfoNCE 极限仍一致。证毕。
\end{proof}

\begin{theorem}[单阶 Rényi 刚性判据]\label{thm:conclusion-single-renyi-rigidity-for-coarsening}
设
\[
f:\Omega\to X,
\qquad
g:X\to Y,
\qquad
h:=g\circ f.
\]
并记
\[
d_f(x):=\#f^{-1}(x),
\qquad
d_h(y):=\#h^{-1}(y)=\sum_{x\in g^{-1}(y)}d_f(x).
\]
对 \(q>1\) 定义
\[
S_q(f):=\sum_{x\in X} d_f(x)^q,
\qquad
S_q(h):=\sum_{y\in Y} d_h(y)^q.
\]
若某一严格 Rényi 阶上有
\[
S_q(h)=S_q(f),
\]
则 \(g\) 在集合 \(\{x\in X:\ d_f(x)>0\}\) 上必为单射。特别地，对一切 \(s\ge 1\) 与 \(T\ge 0\)，有
\[
S_s(h)=S_s(f),
\qquad
C_h(T)=C_f(T),
\]
其中
\[
C_f(T):=\sum_{x\in X}\min(d_f(x),T),
\qquad
C_h(T):=\sum_{y\in Y}\min(d_h(y),T).
\]
\end{theorem}

\begin{proof}
对每个 \(y\in Y\) 令
\[
A_y:=g^{-1}(y),
\qquad
a_x:=d_f(x)\quad(x\in A_y).
\]
则
\[
S_q(h)-S_q(f)
=
\sum_{y\in Y}
\left[
\left(\sum_{x\in A_y}a_x\right)^q
-
\sum_{x\in A_y}a_x^q
\right].
\]
因 \(t\mapsto t^q\) 在 \(q>1\) 时严格凸，每个方括号都非负。若总和为零，则每个方括号都只能为零。严格凸性又推出：这只可能在每个块 \(A_y\) 中至多有一个 \(a_x\) 为正时发生。于是 \(g\) 在 \(\{d_f>0\}\) 上必为单射。

因此，\(\{d_h(y):d_h(y)>0\}\) 与 \(\{d_f(x):d_f(x)>0\}\) 作为多重集完全一致，于是对任意 \(s\ge 1\) 都有
\[
S_s(h)=S_s(f),
\]
并且对任意阈值 \(T\ge 0\) 都有
\[
C_h(T)=C_f(T).
\]
证毕。
\leanverified{paper\_conclusion\_single\_renyi\_rigidity\_for\_coarsening}
\end{proof}

\begin{corollary}[严格粗化的精确位预算损失]\label{cor:conclusion-strict-coarsening-exact-bit-budget-loss}
沿用定理 \ref{thm:conclusion-single-renyi-rigidity-for-coarsening} 的记号。固定整数 \(B\ge 0\)，令 \(T:=2^B\)。若存在 \(y\in Y\) 与两个不同的 \(x_1,x_2\in g^{-1}(y)\) 使
\[
d_f(x_1)>0,
\qquad
d_f(x_2)>0,
\qquad
d_f(x_1)+d_f(x_2)>T,
\]
则
\[
C_h(T)<C_f(T).
\]
定义达到成功率 \(s\in(0,1]\) 的最小比特预算
\[
B_f^\star(s):=
\min\{b\ge 0:\ C_f(2^b)\ge s|\Omega|\},
\]
\[
B_h^\star(s):=
\min\{b\ge 0:\ C_h(2^b)\ge s|\Omega|\}.
\]
则对任意
\[
s\in\left(\frac{C_h(T)}{|\Omega|},\,\frac{C_f(T)}{|\Omega|}\right]
\]
都有
\[
B_h^\star(s)\ge B+1>B\ge B_f^\star(s).
\]
\end{corollary}

\begin{proof}
在给定块 \(g^{-1}(y)\) 上，严格合并两个正纤维且其和大于阈值 \(T\)，会使
\[
\min(d_f(x_1),T)+\min(d_f(x_2),T)
>
\min(d_f(x_1)+d_f(x_2),T).
\]
其余各块只会保持不变或继续降低，因此
\[
C_h(T)<C_f(T).
\]
于是对任意满足
\[
\frac{C_h(T)}{|\Omega|}< s\le \frac{C_f(T)}{|\Omega|}
\]
的成功率 \(s\)，系统 \(f\) 在 \(B\) 比特时已达到该阈值，而系统 \(h\) 在 \(B\) 比特时仍未达到。故
\[
B_f^\star(s)\le B,
\qquad
B_h^\star(s)\ge B+1.
\]
证毕。
\end{proof}

\begin{theorem}[有限纤维容量的算法会计定理]\label{thm:conclusion-finite-fiber-capacity-algorithmic-accounting}
\leanverified{paper\_conclusion\_finite\_fiber\_capacity\_algorithmic\_accounting}
设 \(f:\Omega\to X\) 为有限可计算映射，\(U\) 为固定通用前缀机。对每个 \(x\in X\) 与 \(B\ge 0\)，令
\[
K_x(B):=\{\omega\in f^{-1}(x):\ K_U(\omega\mid x)\le B\}.
\]
则存在仅依赖于 \(U\) 与 \(f\) 的常数 \(c\)，使得对所有 \(B\ge 0\) 都有
\[
\sum_{x\in X}|K_x(B)|
\le
C_f(2^B)
\le
\sum_{x\in X}|K_x(B+c)|
\le
2^c\,C_f(2^B),
\]
其中
\[
C_f(T):=\sum_{x\in X}\min(d_f(x),T),
\qquad
d_f(x):=\#f^{-1}(x).
\]
因而有限映射的最优纤维保持容量与条件 Kolmogorov 描述谱在聚合意义下只差一个实现常数。
\end{theorem}

\begin{proof}
对任意固定 \(x\)，显然
\[
|K_x(B)|\le d_f(x),
\qquad
|K_x(B)|\le 2^B,
\]
因此
\[
|K_x(B)|\le \min(d_f(x),2^B).
\]
对 \(x\) 求和即得
\[
\sum_{x\in X}|K_x(B)|\le C_f(2^B).
\]

另一方面，由容量与条件复杂度的聚合等价，存在常数 \(c\) 使得对每个纤维都有
\[
|K_x(B+c)|\ge \min(d_f(x),2^B),
\]
于是
\[
C_f(2^B)
=
\sum_{x\in X}\min(d_f(x),2^B)
\le
\sum_{x\in X}|K_x(B+c)|.
\]
同一聚合等价的另一侧上界又给出
\[
\sum_{x\in X}|K_x(B+c)|
\le
2^c\sum_{x\in X}\min(d_f(x),2^B)
=
2^c\,C_f(2^B).
\]
证毕。
\end{proof}

\begin{theorem}[全局严格化不可能与旁带位数的不可近似性]\label{thm:conclusion-global-strictification-sideband-inapproximability}
设 \(\mathcal A\) 为任一语义上图灵完备的投影--外置算法类，其语义等价关系记为 \(\equiv\)。则：
\begin{enumerate}
  \item 不存在总可计算的规范化映射
  \[
  \mathrm{NF}:\mathcal A\to \mathbf{PW}^\ast
  \]
  同时满足语义正确性、规范唯一性以及与有限语义层正规形的一致兼容。
  \item 不存在总可计算的有限审计切片选择器，能对任意 \(A,B\in\mathcal A\) 选出一个有限切片并仅靠该切片上的可比数据判定 \(A\equiv B\)。
  \item 对全定义可计算映射 \(\pi\) 记 \(b(\pi)\) 为最小外置预算，并令
  \[
  a(\pi):=\left\lceil \log_2 b(\pi)\right\rceil.
  \]
  则集合 \(\{\pi:\ a(\pi)=0\}\) 不是递归可枚举的；并且：
  \begin{enumerate}
    \item 不存在总可计算函数 \(A\) 满足
    \[
    |A(\pi)-a(\pi)|<1
    \qquad (\forall\,\pi);
    \]
    \item 不存在总可计算函数 \(A\) 与常数 \(\varepsilon>0\) 使得对所有满足 \(a(\pi)\ge 1\) 的 \(\pi\) 都有
    \[
    a(\pi)\le A(\pi)<(2-\varepsilon)a(\pi).
    \]
  \end{enumerate}
\end{enumerate}
\end{theorem}

\begin{proof}
若第 \(1\) 条成立，则可通过比较 \(\mathrm{NF}(A)\) 与 \(\mathrm{NF}(B)\) 的字符串相等性来判定语义等价 \(A\equiv B\)。这与图灵完备语义类上的等价不可判定性矛盾。

若第 \(2\) 条成立，则给定 \(A,B\) 后可先计算相应有限切片，再比较该切片上的有限语义指纹，从而同样得到 \(A\equiv B\) 的判定算法，亦与语义不可判定性矛盾。

对第 \(3\) 条，注意到
\[
a(\pi)=0 \iff b(\pi)=1.
\]
而定理 \ref{thm:conclusion-minimal-external-budget-undecidable} 已证明集合
\[
\{\pi:\ b(\pi)=1\}
\]
不是递归可枚举的，故 \(\{\pi:\ a(\pi)=0\}\) 也不是递归可枚举的。

若存在加性误差 \(<1\) 的近似算法 \(\widetilde a\)，则在二值族 \(a(\pi)\in\{0,1\}\) 上，
\[
|\widetilde a(\pi)-a(\pi)|<1
\]
足以严格区分 \(a(\pi)=0\) 与 \(a(\pi)=1\)，从而判定 \(\{\pi:\ a(\pi)=0\}\)，矛盾。

对乘性近似，先从定理 \ref{thm:conclusion-minimal-external-budget-undecidable} 的二值族
\[
b(\pi)\in\{1,2\}
\]
出发，定义纤维倍增映射
\[
\pi^\sharp(x,\epsilon):=\pi(x),
\qquad
\epsilon\in\{0,1\}.
\]
则每条纤维大小恰被乘以 \(2\)，从而
\[
b(\pi^\sharp)=2\,b(\pi)\in\{2,4\},
\qquad
a(\pi^\sharp)\in\{1,2\}.
\]
若存在乘性因子 \(<2\) 的近似算法，则在该新二值族上，
\[
a(\pi^\sharp)=1\Rightarrow \widetilde a(\pi^\sharp)<2,
\qquad
a(\pi^\sharp)=2\Rightarrow \widetilde a(\pi^\sharp)\ge 2
\]
从而同样可区分两种情形，并再次导出停机可判定，矛盾。证毕。
\end{proof}

\begin{theorem}[bin-fold 的单常数全息律]\label{thm:conclusion-binfold-single-constant-holography}
设 \(u_m\) 为稳定层均匀基线，\(\mu_m\) 为原始 bin-fold 分布，并令
\[
D_{\chi^2}:=\lim_{m\to\infty}\chi^2(\mu_m\Vert u_m),
\qquad
P_e^\ast:=\lim_{m\to\infty}P_e^\ast(m),
\qquad
D_{TV}^\infty:=\lim_{m\to\infty}D_{TV}(\mu_m,u_m).
\]
则三者满足精确代数闭合：
\[
\varphi=\frac{5D_{\chi^2}+3}{2},
\]
\[
P_e^\ast=\frac{1}{2+5D_{\chi^2}},
\]
\[
D_{TV}^\infty=1-\frac{2}{2+5D_{\chi^2}}.
\]
因此，单个 \(\chi^2\) 极限常数已同时决定黄金比例、最优 Bayes 错误极限与总变差极限。
\end{theorem}

\begin{proof}
既有极限闭式给出
\[
D_{\chi^2}=\frac{2\varphi-3}{5},
\qquad
P_e^\ast=\frac1{\sqrt5},
\qquad
D_{TV}^\infty=1-\frac2{\sqrt5}.
\]
由第一式立即得到
\[
\varphi=\frac{5D_{\chi^2}+3}{2}.
\]
再由恒等式
\[
2\varphi-1=\sqrt5
\]
可得
\[
\sqrt5=2+5D_{\chi^2}.
\]
把它代回其余两式，便得到
\[
P_e^\ast=\frac{1}{2+5D_{\chi^2}},
\qquad
D_{TV}^\infty=1-\frac{2}{2+5D_{\chi^2}}.
\]
证毕。
\end{proof}

\begin{theorem}[固定样本 escort 族的计数壳完备性]\label{thm:conclusion-fixed-sample-escort-count-shell-sufficiency}
\leanverified{paper\_conclusion\_fixed\_sample\_escort\_count\_shell\_sufficiency}
固定 \(Q<\infty\) 与样本数 \(n\)。设
\[
\mathcal E_m^{(n,Q)}:=\{\pi_{m,q}^{\otimes n}:0\le q\le Q\},
\]
并记末位计数统计量
\[
N_n(w^{(1)},\dots,w^{(n)}):=\sum_{j=1}^{n}w_m^{(j)}\in\{0,\dots,n\}.
\]
则对任意标准 Borel 动作空间、任意取值于 \([0,1]\) 的有界损失函数 \(\ell\) 与任意先验 \(\Pi\) 而言，使用全样本的最优 Bayes 风险 \(R_m^\ast(\Pi,\ell)\) 与只使用 \(N_n\) 的最优 Bayes 风险 \(R_{m,N_n}^\ast(\Pi,\ell)\) 满足
\[
\bigl|R_m^\ast(\Pi,\ell)-R_{m,N_n}^\ast(\Pi,\ell)\bigr|
=
O_{n,Q}\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
特别地，固定样本 escort 半轴在决策论意义下的全部参数依赖，渐近上完全压缩到单一整数统计量 \(N_n\) 上。
\end{theorem}

\begin{proof}
由 escort 族的 fixed-sample Le Cam 塌缩，实验族
\[
\mathcal E_m^{(n,Q)}
\]
与二项试验族
\[
\mathcal G^{(n,Q)}:=\{\mathrm{Binomial}(n,\theta(q)):0\le q\le Q\},
\qquad
\theta(q)=\frac{1}{1+\varphi^{q+1}},
\]
之间的 Le Cam 距离满足
\[
\Delta_{\mathrm{LeCam}}(\mathcal E_m^{(n,Q)},\mathcal G^{(n,Q)})
=
O_{n,Q}\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
而二项试验族对末位计数 \(N_n\) 天然充分，因此“先取全样本、再压到 \(N_n\)”与“直接观察 \(\mathcal G^{(n,Q)}\)”在决策意义下等价。于是由 Blackwell--Le Cam 风险比较不等式，对任意有界损失与任意先验都有
\[
\bigl|R_m^\ast(\Pi,\ell)-R_{m,N_n}^\ast(\Pi,\ell)\bigr|
\ll
\Delta_{\mathrm{LeCam}}(\mathcal E_m^{(n,Q)},\mathcal G^{(n,Q)}),
\]
从而得到所示估计。证毕。
\end{proof}

\begin{proposition}[对数似然的格量子化]\label{prop:conclusion-loglikelihood-affine-lattice-quantization}
\leanverified{paper\_conclusion\_loglikelihood\_affine\_lattice\_quantization}
固定 \(p\neq q\)。令
\[
\theta(s)=\frac{1}{1+\varphi^{s+1}},
\]
\[
\Delta_{p,q}:=
\log\frac{\theta(q)(1-\theta(p))}{\theta(p)(1-\theta(q))},
\qquad
b_{p,q}:=
\log\frac{1-\theta(q)}{1-\theta(p)}.
\]
则 \(n\)-样本对数似然比满足
\[
\Lambda_n(p,q)=n\,b_{p,q}+N_n\,\Delta_{p,q},
\]
从而其取值集恰为仿射格
\[
n\,b_{p,q}+\Delta_{p,q}\ZZ.
\]
特别地，任一固定 \(n\) 的似然阈值判别规则都严格等价于 \(N_n\) 上的一个区间判别；并且只要阈值扰动不足以跨越相邻格点，其判别结果完全不变。
\end{proposition}

\begin{proof}
在 Bernoulli 骨架上，一步对数似然增量只取两值：
\[
\log\frac{\theta(q)}{\theta(p)}
\qquad\text{或}\qquad
\log\frac{1-\theta(q)}{1-\theta(p)}.
\]
对 \(n\) 个样本求和，并把其中取第一值的次数记为 \(N_n\)，便得到
\[
\Lambda_n(p,q)
=
N_n\log\frac{\theta(q)}{\theta(p)}
+
(n-N_n)\log\frac{1-\theta(q)}{1-\theta(p)}.
\]
整理后即为
\[
\Lambda_n(p,q)
=
n\log\frac{1-\theta(q)}{1-\theta(p)}
+
N_n\log\frac{\theta(q)(1-\theta(p))}{\theta(p)(1-\theta(q))}
=
n\,b_{p,q}+N_n\,\Delta_{p,q}.
\]
因此其取值集恰是步长为 \(\Delta_{p,q}\) 的仿射整数格。

于是，对任意阈值 \(A\)，判定条件
\[
\Lambda_n(p,q)\ge A
\]
严格等价于
\[
N_n\ge \frac{A-nb_{p,q}}{\Delta_{p,q}}
\]
或其反向不等式，从而判别区域在 \(N_n\) 空间中必为区间。若阈值扰动不足以跨越相邻格点，则被截取的 \(N_n\) 子集不变，判别结果亦不变。证毕。
\end{proof}

\paragraph{统一读法}
至此，冻结 escort 制度在结论层面被压成三本彼此正交而不可互代的账本。其一，\(\mathrm{Succ}^{\mathrm{rev}}_{m,q}\) 的极限只记住最大纤维尺度 \(M_m\)；其二，Bayes--InfoNCE 的冻结极限只记住顶层简并度 \(\nu_m\)；其三，固定样本下的一切决策又进一步塌缩为末位计数 \(N_n\) 所张成的一维格几何。与此同时，任一严格粗化在单个严格 Rényi 阶上已经暴露为真实容量损失，而有限纤维容量与条件 Kolmogorov 描述谱之间仅差一个实现常数；更进一步，全定义可计算投影的最小旁带预算既不可全局严格化，也不能在优于因子 \(2\) 的尺度上近似。于是，在本文的冻结 regime 中，统计可见性、可逆恢复与算法语义不再受同一数量统摄，而是严格分裂为尺度账本、简并账本与计数格三条资源轴。
