\subsection{斯托克斯延拓与有限相位可见性：可延拓子晶格的半维界与曲面可实现性}\label{subsec:conclusion-finite-phase-visibility-lagrangian-realizability}
本小节将边界上的中间度交叉配对与体内延拓性联系起来，并将其编译为有限相位采样计数的可核验增长约束。
圆维与相位采样计数沿用 \S\ref{sec:circle-dimension-phase-gate} 的口径（定义 \ref{def:cdim-phase-spectrum}）。

\begin{proposition}\label{prop:conclusion-stokes-extension-isotropic-half-rank}
\leanverified{paper\_conclusion\_stokes\_extension\_isotropic\_half\_rank}
设 \(\Sigma\) 为闭、定向、光滑 \(2k\) 维流形，且 \(k\) 为奇数，并假设中间度交叉配对在自由部分上为非退化辛形式：
\[
\langle a,b\rangle_\Sigma:=\int_\Sigma a\wedge b,
\qquad
a,b\in H^k(\Sigma;\RR)\ \text{来自}\ H^k(\Sigma;\ZZ)_{\mathrm{fr}}.
\]
设 \(M\) 为紧、定向、光滑 \((2k+1)\) 维流形，\(\partial M=\Sigma\)。
记限制同态在自由部分上的像为
\[
r:\ H^k(M;\ZZ)_{\mathrm{fr}}\longrightarrow H^k(\Sigma;\ZZ)_{\mathrm{fr}},
\qquad
L:=\operatorname{im}(r).
\]
令 \(b:=\cdim(H^k(\Sigma;\ZZ)_{\mathrm{fr}})\)。
则 \(L\otimes_\ZZ \RR\subset H^k(\Sigma;\RR)\) 为各向同性子空间，并且
\[
\cdim(L)\ \le\ \frac{b}{2}.
\]
\end{proposition}

\begin{proof}
取 \(\alpha,\beta\in H^k(M;\RR)\) 为闭形式代表。
由广义斯托克斯公式，
\[
\int_\Sigma r\alpha\wedge r\beta
=
\int_{\partial M}\alpha\wedge\beta
=
\int_{M}\dd(\alpha\wedge\beta)
=
\int_{M}(\dd\alpha\wedge\beta+(-1)^k\alpha\wedge\dd\beta)
=0.
\]
因此 \(\operatorname{im}(H^k(M;\RR)\to H^k(\Sigma;\RR))\) 在辛配对下为各向同性子空间。
由于该辛形式在自由部分上非退化，任一各向同性子空间维数至多为半维，故
\(
\dim_\RR(L\otimes \RR)\le b/2
\)。
最后 \(\cdim(L)=\dim_\QQ(L\otimes_\ZZ \QQ)=\dim_\RR(L\otimes_\ZZ \RR)\)，得到结论。证毕。
\end{proof}

\begin{corollary}\label{cor:conclusion-stokes-extension-phase-sampling-bound}
在命题 \ref{prop:conclusion-stokes-extension-isotropic-half-rank} 的设定下，对任意 \(N\ge 1\) 有
\[
\boxed{\;
\Sigma_{L}^{\mathrm{ph}}(N)=\bigl|\operatorname{Hom}(L,\mu_N)\bigr|
\le
N^{\,b/2}.
\;}
\]
\end{corollary}

\begin{proof}
由于 \(L\) 为自由阿贝尔群，存在 \(\ell=\cdim(L)\) 使 \(L\cong \ZZ^\ell\)。
于是
\[
\Sigma_{L}^{\mathrm{ph}}(N)=\bigl|\operatorname{Hom}(\ZZ^\ell,\mu_N)\bigr|
=N^\ell.
\]
由命题 \ref{prop:conclusion-stokes-extension-isotropic-half-rank} 有 \(\ell\le b/2\)，从而得不等式。证毕。
\end{proof}

\begin{theorem}[相位计数的增长指数严格等于晶格秩]\label{thm:conclusion-phase-sampling-growth-exponent-rank}
\leanverified{paper\_conclusion\_phase\_sampling\_growth\_exponent\_rank}
沿用命题 \ref{prop:conclusion-stokes-extension-isotropic-half-rank} 的设定。则对任意 \(N\ge 1\) 都有精确公式
\[
\Sigma_{L}^{\mathrm{ph}}(N)=N^{\cdim(L)}.
\]
从而极限
\[
\beta_{\mathrm{ph}}(L)
:=
\lim_{N\to\infty}\frac{\log \Sigma_{L}^{\mathrm{ph}}(N)}{\log N}
\]
存在，且满足
\[
\beta_{\mathrm{ph}}(L)=\cdim(L).
\]
特别地，若 \(L\) 来自某个 \((2k+1)\) 维填充的限制像，则
\[
\beta_{\mathrm{ph}}(L)\le \frac{b}{2}.
\]
\end{theorem}
\begin{proof}
由于 \(L\) 为自由阿贝尔群，存在 \(\ell=\cdim(L)\) 使 \(L\cong \ZZ^\ell\)。于是对任意 \(N\ge 1\),
\[
\Sigma_{L}^{\mathrm{ph}}(N)
=
\bigl|\operatorname{Hom}(\ZZ^\ell,\mu_N)\bigr|
=
N^\ell
=
N^{\cdim(L)}.
\]
两边取对数并除以 \(\log N\)，便得
\[
\frac{\log \Sigma_{L}^{\mathrm{ph}}(N)}{\log N}=\cdim(L),
\]
故极限存在且 \(\beta_{\mathrm{ph}}(L)=\cdim(L)\)。

若 \(L\) 来自命题 \ref{prop:conclusion-stokes-extension-isotropic-half-rank} 的填充限制像，则该命题已给出
\[
\cdim(L)\le \frac{b}{2}.
\]
与上式合并即可得到
\[
\beta_{\mathrm{ph}}(L)\le \frac{b}{2}.
\]
证毕。
\end{proof}

\begin{corollary}[一次相位饱和强制达到拉格朗日上限]\label{cor:conclusion-phase-sampling-single-saturation-lagrangian}
沿用命题 \ref{prop:conclusion-stokes-extension-isotropic-half-rank} 的设定。若存在某个 \(N_0\ge 2\) 使
\[
\Sigma_{L}^{\mathrm{ph}}(N_0)=N_0^{\,b/2},
\]
则
\[
\cdim(L)=\frac{b}{2}.
\]
因而 \(L\otimes_\ZZ\RR\subset H^k(\Sigma;\RR)\) 是一个极大各向同性子空间；特别地，它是一个 Lagrangian 子空间。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-phase-sampling-growth-exponent-rank}，
\[
\Sigma_{L}^{\mathrm{ph}}(N_0)=N_0^{\cdim(L)}.
\]
与假设合并得到
\[
N_0^{\cdim(L)}=N_0^{\,b/2}.
\]
由于 \(N_0\ge 2\)，故
\[
\cdim(L)=\frac{b}{2}.
\]
另一方面，命题 \ref{prop:conclusion-stokes-extension-isotropic-half-rank} 已知 \(L\otimes_\ZZ\RR\) 为各向同性子空间。既然其维数已达到半维上界 \(b/2\)，便必为极大各向同性，也即 Lagrangian 子空间。证毕。
\end{proof}

\begin{corollary}\label{cor:conclusion-stokes-extension-obstruction-certificate}
沿用上述设定。若存在某个 \(N\ge 1\) 使
\[
\Sigma_{L}^{\mathrm{ph}}(N)\ >\ N^{\,b/2},
\]
则不存在任何 \((2k+1)\) 维填充 \(M\) 使得 \(L\) 作为限制像 \(\operatorname{im}(H^k(M;\ZZ)_{\mathrm{fr}}\to H^k(\Sigma;\ZZ)_{\mathrm{fr}})\) 出现。
\end{corollary}

\begin{proof}
这与推论 \ref{cor:conclusion-stokes-extension-phase-sampling-bound} 矛盾。证毕。
\end{proof}

\begin{conjecture}\label{conj:conclusion-lagrangian-realizability}
设 \(\Sigma\) 为闭、定向、光滑 \(2k\) 维流形，\(k\) 奇，且 \(H^k(\Sigma;\ZZ)\) 无挠。
记 \(Q\) 为自由部分 \(H^k(\Sigma;\ZZ)\) 上的辛交叉配对。
对任意本原拉格朗日子晶格 \(L\subset H^k(\Sigma;\ZZ)\)（即 \(L\) 各向同性、\(\cdim(L)=\cdim(H^k(\Sigma;\ZZ))/2\)，且商 \(H^k(\Sigma;\ZZ)/L\) 无挠），存在紧、定向 \((2k+1)\) 维流形 \(M\) 与边界同胚 \(\partial M\cong \Sigma\)，使得
\[
\operatorname{im}\bigl(H^k(M;\ZZ)_{\mathrm{fr}}\to H^k(\Sigma;\ZZ)_{\mathrm{fr}}\bigr)=L.
\]
\end{conjecture}

\begin{theorem}\label{thm:conclusion-surface-lagrangian-realizability}
对任意闭、定向曲面 \(\Sigma_g\)（\(g\ge 1\)）及任意本原拉格朗日子晶格 \(L\subset H^1(\Sigma_g;\ZZ)\)，存在 \(g\) 把手体 \(H_g\) 与同胚 \(\phi:\partial H_g\to \Sigma_g\)，使
\[
\operatorname{im}\bigl(H^1(H_g;\ZZ)\xrightarrow{\ \phi^\ast\circ r\ } H^1(\Sigma_g;\ZZ)\bigr)=L,
\]
其中 \(r\) 为边界限制映射。
\end{theorem}

\begin{proof}
标准把手体 \(H_g\) 的边界为 \(\Sigma_g\)，其限制像 \(L_0\subset H^1(\Sigma_g;\ZZ)\) 为某个本原拉格朗日子晶格。
对给定 \(L\)，辛格子线性代数给出存在 \(\Psi\in \mathrm{Sp}_{2g}(\ZZ)\) 使 \(\Psi(L_0)=L\)。
而映射类群到 \(\mathrm{Sp}_{2g}(\ZZ)\) 的自然表示为满射，故存在曲面自同胚 \(\varphi:\Sigma_g\to\Sigma_g\) 诱导 \(\varphi^\ast=\Psi\)。
取 \(\phi:=\varphi\circ(\partial H_g\cong \Sigma_g)\)，则 \(\phi^\ast(L_0)=L\)，从而 \(H_g\) 在边界标识 \(\phi\) 下实现 \(L\)。证毕。
\end{proof}
