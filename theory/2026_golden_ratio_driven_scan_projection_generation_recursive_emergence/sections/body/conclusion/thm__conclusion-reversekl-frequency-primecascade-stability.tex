\paragraph{频率排他性、素因子级联与固定相位切片的稳定阱}
沿用定理 \ref{thm:xi-reversekl-single-frequency-exact-minimizer}、定理 \ref{thm:xi-reversekl-single-frequency-rigidity-equivalences}、定理 \ref{thm:xi-reversekl-padic-entropy-cascade}、命题 \ref{prop:xi-reversekl-coprime-frequency-phase-amplification}、定理 \ref{thm:xi-reversekl-affine-second-variation-strong-convex} 与推论 \ref{cor:xi-reversekl-strict-convex-unique-minimizer} 的记号。把单频封口、\(p\)-进熵级联、互素频率放大与仿射线强凸性并置之后，还可继续抽取出一条频率--素因子--熵几何的刚性链。其核心现象是：精确单频极值既具有严格的频率排他性，又在 prime-cascade 上留下完整坐标，并在固定相位切片内形成二次稳定势阱。

\begin{theorem}[单频精确封口的频率排他性]\label{thm:conclusion-reversekl-single-frequency-exclusivity}
固定 \(0<r<1\)。
设 \(\mu\in \mathrm{Prob}(\TT)\)，且存在两个不同的正整数 \(m\neq n\) 使
\[
\mathcal H_r(\mu)
= -\log\!\bigl(1-r^{2n}|\widehat\mu(n)|^2\bigr)
= -\log\!\bigl(1-r^{2m}|\widehat\mu(m)|^2\bigr).
\]
则 \(\mu\) 必为 \(\TT\) 上的 Haar 概率测度 \(m_{\TT}\)。
\end{theorem}
\leanverified{paper\_conclusion\_reversekl\_single\_frequency\_exclusivity}
\begin{proof}
若 \(|\widehat\mu(n)|=0\)，则第一条等式给出 \(\mathcal H_r(\mu)=0\)。由
\[
\mathcal H_r(\mu)=\int_{\TT}-\log(P_r*\mu)\,\dd m_{\TT}\ge -\log\int_{\TT}(P_r*\mu)\,\dd m_{\TT}=0
\]
可知必须处处取等，从而 \(P_r*\mu\equiv 1\)。比较 Fourier 系数即得 \(\widehat\mu(k)=0\) 对一切 \(k\neq 0\) 成立，故 \(\mu=m_{\TT}\)。同理若 \(|\widehat\mu(m)|=0\) 也只能得到 Haar 情形。

以下设 \(|\widehat\mu(n)|,|\widehat\mu(m)|>0\)。不妨设 \(n<m\)。
若 \(|\widehat\mu(n)|=1\)，则第一条等式化为
\[
\mathcal H_r(\mu)=-\log(1-r^{2n}).
\]
再由第二条等式得
\[
1-r^{2m}|\widehat\mu(m)|^2=1-r^{2n},
\qquad
|\widehat\mu(m)|=r^{\,n-m}>1,
\]
矛盾。故必有 \(0<|\widehat\mu(n)|<1\)。

于是可应用定理 \ref{thm:xi-reversekl-single-frequency-rigidity-equivalences}，得到
\[
\mu=\nu_{c,n,\theta_0},
\qquad
c:=|\widehat\mu(n)|\in(0,1),
\]
从而
\[
\widehat\mu(k)=
\begin{cases}
c^{|j|}e^{-inj\theta_0},& k=nj,\ j\in\ZZ,\\
0,& n\nmid k.
\end{cases}
\]
若 \(n\nmid m\)，则 \(\widehat\mu(m)=0\)，这已在前面排除。故只能有 \(m=\ell n\) 且 \(\ell\ge 2\)。此时
\[
|\widehat\mu(m)|=|\widehat\mu(\ell n)|=c^\ell.
\]
把两条等式写成
\[
-\log(1-x)=-\log(1-x^\ell),
\qquad
x:=r^{2n}c^2\in(0,1),
\]
便得到 \(x=x^\ell\)。但 \(\ell\ge 2\) 与 \(x\in(0,1)\) 不可能同时满足该恒等式，矛盾。故非 Haar 情形不存在。证毕。
\end{proof}

\begin{corollary}[精确极值层的频率花束闭包]\label{cor:conclusion-reversekl-extremal-frequency-bouquet}
对每个整数 \(n\ge 1\) 与固定 \(r\in(0,1)\)，定义
\[
\mathcal E_n(r):=\{\nu_{c,n,\theta_0}:0<c<1,\ \theta_0\in\TT\}.
\]
则对任意 \(m\neq n\)，有
\[
\mathcal E_n(r)\cap \mathcal E_m(r)=\varnothing.
\]
并且其弱-\(*\) 闭包满足
\[
\overline{\mathcal E_n(r)}^{\,w^\ast}
=
\{\nu_{c,n,\theta_0}:0\le c\le 1,\ \theta_0\in\TT\},
\]
从而
\[
\overline{\mathcal E_n(r)}^{\,w^\ast}\cap \overline{\mathcal E_m(r)}^{\,w^\ast}
=\{m_{\TT}\}.
\]
因此精确极值层构成一族仅在 Haar 点相接的频率花束。
\end{corollary}
\begin{proof}
若 \(\mu\in\mathcal E_n(r)\cap \mathcal E_m(r)\)，则 \(\mu\) 同时在频率 \(n\) 与频率 \(m\) 上达到单频封口等号，与定理 \ref{thm:conclusion-reversekl-single-frequency-exclusivity} 矛盾。故 \(\mathcal E_n(r)\cap \mathcal E_m(r)=\varnothing\)。

再证闭包描述。任取序列
\[
\mu_j=\nu_{c_j,n,\theta_j}\in\mathcal E_n(r).
\]
令
\[
z_j:=c_j e^{-in\theta_j}\in\overline{\mathbb D}.
\]
取一子列仍记为 \(z_j\)，使其收敛到某个 \(z\in\overline{\mathbb D}\)。把 \(z\) 写成
\[
z=ce^{-in\theta_0},\qquad 0\le c\le 1,\ \theta_0\in\TT.
\]
则对任意 \(k\in\ZZ\)，由定理 \ref{thm:xi-reversekl-single-frequency-exact-minimizer} 中的 Fourier 闭式，
\[
\widehat{\mu_j}(k)=
\begin{cases}
z_j^{\,m},& k=nm,\ m\ge 0,\\
\overline{z_j}^{\,|m|},& k=nm,\ m<0,\\
0,& n\nmid k.
\end{cases}
\]
故
\[
\widehat{\mu_j}(k)\longrightarrow
\begin{cases}
z^{\,m},& k=nm,\ m\ge 0,\\
\overline{z}^{\,|m|},& k=nm,\ m<0,\\
0,& n\nmid k,
\end{cases}
\]
这正是 \(\nu_{c,n,\theta_0}\) 的 Fourier--Stieltjes 系数。由 Fourier 唯一性，
\[
\mu_j\stackrel{w^\ast}{\longrightarrow}\nu_{c,n,\theta_0}.
\]
反向包含显然，因为 \(c\downarrow 0\) 时 \(\nu_{c,n,\theta_0}\to m_{\TT}\)，而 \(c\uparrow 1\) 时 \(\nu_{c,n,\theta_0}\to \nu_{1,n,\theta_0}\)。

最后，若某个 \(\mu\neq m_{\TT}\) 同时属于两条闭包，则由上式可写成
\[
\mu=\nu_{c,n,\theta_0}=\nu_{c',m,\phi_0}
\]
其中 \(c,c'>0\)。这再次与定理 \ref{thm:conclusion-reversekl-single-frequency-exclusivity} 矛盾。故两个闭包只能在 Haar 点相交。证毕。
\end{proof}

\begin{theorem}[精确极值族的 \(p\)-进熵级联闭式]\label{thm:conclusion-reversekl-extremal-prime-cascade-closed}
\leanverified{paper\_conclusion\_reversekl\_extremal\_prime\_cascade\_closed}
固定 \(0<r<1\)，并取
\[
\mu=\nu_{c,n,\theta_0},\qquad 0<c<1.
\]
对任意素数 \(p\)，写
\[
n=p^{v_p(n)}n_{(p)},
\qquad
p\nmid n_{(p)},
\qquad
x:=r^{2n}c^2\in(0,1).
\]
沿用定理 \ref{thm:xi-reversekl-padic-entropy-cascade} 的记号
\[
\mathcal H_{p,j}:=\mathcal H_{r^{p^j}}\!\bigl(\mu^{\langle p^j\rangle}\bigr),
\qquad
\mathcal G_{p,j}:=\mathcal H_{p,j}-\mathcal H_{p,j+1}.
\]
则有显式公式
\[
\mathcal H_{p,j}=
\begin{cases}
-\log(1-x),& 0\le j\le v_p(n),\\[4pt]
-\log\!\bigl(1-x^{p^{\,j-v_p(n)}}\bigr),& j\ge v_p(n),
\end{cases}
\]
以及
\[
\mathcal G_{p,j}=0
\qquad (0\le j<v_p(n)),
\]
\[
\mathcal G_{p,v_p(n)+\ell}
=
\log\!\bigl(1+x^{p^\ell}+x^{2p^\ell}+\cdots+x^{(p-1)p^\ell}\bigr)
\qquad (\ell\ge 0).
\]
因此，精确单频极值在每个素数方向上都呈现一段长度恰为 \(v_p(n)\) 的零平顶，随后进入 \(p\)-幂尾流。
\end{theorem}
\begin{proof}
先取 \(0\le j<v_p(n)\)。由定理 \ref{thm:xi-reversekl-single-frequency-exact-minimizer} 中的 Fourier 闭式，
\[
\widehat{\mu^{\langle p^j\rangle}}(k)
=\widehat\mu(p^jk)
=
\begin{cases}
c^{|m|}e^{-inm\theta_0},& k=(n/p^j)m,\ m\in\ZZ,\\
0,& (n/p^j)\nmid k.
\end{cases}
\]
故
\[
\mu^{\langle p^j\rangle}=\nu_{c,\;n/p^j,\;p^j\theta_0}.
\]
再由定理 \ref{thm:xi-reversekl-single-frequency-exact-minimizer} 的精确值公式，
\[
\mathcal H_{p,j}
=\mathcal H_{r^{p^j}}\!\bigl(\nu_{c,\;n/p^j,\;p^j\theta_0}\bigr)
=-\log\!\Bigl(1-(r^{p^j})^{2n/p^j}c^2\Bigr)
=-\log(1-r^{2n}c^2)
=-\log(1-x).
\]

若 \(j=v_p(n)+\ell\ge v_p(n)\)，则
\[
\widehat{\mu^{\langle p^j\rangle}}(k)=\widehat\mu(p^{v_p(n)+\ell}k).
\]
非零条件变为 \(k=n_{(p)}m\)，且对应系数为
\[
\widehat{\mu^{\langle p^j\rangle}}(n_{(p)}m)
=c^{p^\ell|m|}e^{-ip^jn_{(p)}m\theta_0}.
\]
因此
\[
\mu^{\langle p^j\rangle}
=\nu_{c^{p^\ell},\,n_{(p)},\,p^j\theta_0}.
\]
再次代入精确值公式得到
\[
\mathcal H_{p,j}
=-\log\!\Bigl(1-(r^{p^j})^{2n_{(p)}}c^{2p^\ell}\Bigr)
=-\log\!\bigl(1-x^{p^\ell}\bigr).
\]
于是 \(0\le j<v_p(n)\) 时 \(\mathcal H_{p,j}\) 常值，从而 \(\mathcal G_{p,j}=0\)。
对 \(j=v_p(n)+\ell\) 有
\[
\mathcal G_{p,v_p(n)+\ell}
=-\log(1-x^{p^\ell})+\log(1-x^{p^{\ell+1}})
=\log\frac{1-x^{p^{\ell+1}}}{1-x^{p^\ell}}.
\]
再利用
\[
\frac{1-y^p}{1-y}=1+y+\cdots+y^{p-1}
\]
并取 \(y=x^{p^\ell}\) 即得第二个闭式。证毕。
\end{proof}

\begin{corollary}[零平顶长度恢复频率的素因子指数]\label{cor:conclusion-reversekl-zero-plateau-recovers-frequency}
在结论 \ref{thm:conclusion-reversekl-extremal-prime-cascade-closed} 的设定下，对每个素数 \(p\) 定义
\[
z_p(\mu):=
\sup\{J\ge 0:\ \mathcal G_{p,0}=\mathcal G_{p,1}=\cdots=\mathcal G_{p,J-1}=0\}.
\]
则有
\[
z_p(\mu)=v_p(n).
\]
从而频率标签可由零平顶长度完全恢复：
\[
n=\prod_{p}p^{\,z_p(\mu)}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-reversekl-extremal-prime-cascade-closed}，
\[
\mathcal G_{p,j}=0\iff j<v_p(n).
\]
故零平顶长度恰为 \(v_p(n)\)。对全部素数取乘积便恢复 \(n\) 的素因子分解。证毕。
\end{proof}

\begin{theorem}[完整 prime-cascade 在模旋转后完全分类精确极值族]\label{thm:conclusion-reversekl-full-prime-cascade-classification}
\leanverified{paper\_conclusion\_reversekl\_full\_prime\_cascade\_classification}
固定 \(0<r<1\)，并定义
\[
\mathcal E(r):=\{\nu_{c,n,\theta_0}:n\ge 1,\ 0<c<1,\ \theta_0\in\TT\}.
\]
对旋转商空间 \(\mathcal E(r)/\TT\)，映射
\[
[\nu_{c,n,\theta_0}]
\longmapsto
\bigl(\mathcal G_{p,j}(\nu_{c,n,\theta_0})\bigr)_{p,\;j\ge 0}
\]
是单射。更强地，完整级联数组唯一恢复参数对 \((n,c)\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-reversekl-extremal-prime-cascade-closed} 可知，数组 \((\mathcal G_{p,j})_{p,j}\) 与旋转参数 \(\theta_0\) 无关，而只依赖于
\[
n,\qquad x:=r^{2n}c^2\in(0,1).
\]
先由推论 \ref{cor:conclusion-reversekl-zero-plateau-recovers-frequency}，可从各素数方向的零平顶长度恢复全部 \(v_p(n)\)，从而唯一恢复 \(n\)。

在 \(n\) 已知后，任取素数 \(p\)，记 \(v:=v_p(n)\)。则由定理 \ref{thm:conclusion-reversekl-extremal-prime-cascade-closed}，
\[
\mathcal G_{p,v}=\psi_p(x),
\qquad
\psi_p(x):=\log\!\bigl(1+x+x^2+\cdots+x^{p-1}\bigr).
\]
由于
\[
1+x+\cdots+x^{p-1}
\]
在 \((0,1)\) 上严格递增，故 \(\psi_p\) 亦在 \((0,1)\) 上严格递增，从而 \(\mathcal G_{p,v}\) 唯一决定 \(x\)。
于是
\[
c=r^{-n}\sqrt{x}
\]
亦被唯一恢复。故完整 prime-cascade 数组唯一确定 \((n,c)\)，从而在模掉旋转参数后给出单射分类。证毕。
\end{proof}

\begin{theorem}[互素双近字符的一阶熵势垒]\label{thm:conclusion-reversekl-coprime-double-nearchar-entropy-barrier}
固定 \(0<r<1\)，令 \(p,q\in\NN\) 互素，并对 \(\mu\in\mathrm{Prob}(\TT)\) 记
\[
\varepsilon_p:=1-|\widehat\mu(p)|,\qquad
\varepsilon_q:=1-|\widehat\mu(q)|,
\qquad
\eta_{p,q}(\mu):=(p+q)^2\max\{\varepsilon_p,\varepsilon_q\}.
\]
则有统一下界
\[
\mathcal H_r(\mu)\ge -\log\!\Bigl(1-r^2\bigl(1-\eta_{p,q}(\mu)\bigr)_+^2\Bigr),
\qquad
(t)_+:=\max\{t,0\}.
\]
进一步，若
\[
\mathcal H_r(\mu)\le h<-\log(1-r^2),
\]
则必有
\[
\eta_{p,q}(\mu)\ge 1-r^{-1}\sqrt{1-e^{-h}}.
\]
因此，两条互素频率上的近字符锁定会被算术放大成一阶 Fourier 近字符，并立即触发统一熵势垒。
\end{theorem}
\begin{proof}
由命题 \ref{prop:xi-reversekl-coprime-frequency-phase-amplification}，
\[
1-|\widehat\mu(1)|\le \eta_{p,q}(\mu).
\]
故
\[
|\widehat\mu(1)|\ge \bigl(1-\eta_{p,q}(\mu)\bigr)_+.
\]
再由推论 \ref{cor:xi-reversekl-single-frequency-closed-lower} 在 \(n=1\) 的情形，
\[
\mathcal H_r(\mu)\ge -\log\bigl(1-r^2|\widehat\mu(1)|^2\bigr),
\]
代入上式即得第一条不等式。

若进一步有 \(\mathcal H_r(\mu)\le h<-\log(1-r^2)\)，则由定理 \ref{thm:xi-reversekl-single-frequency-entropy-budget-sup} 在 \(n=1\) 的上确界公式，
\[
|\widehat\mu(1)|\le r^{-1}\sqrt{1-e^{-h}}<1.
\]
与前面的下界合并，得到
\[
\bigl(1-\eta_{p,q}(\mu)\bigr)_+\le r^{-1}\sqrt{1-e^{-h}}.
\]
由于右端小于 \(1\)，可直接化为
\[
\eta_{p,q}(\mu)\ge 1-r^{-1}\sqrt{1-e^{-h}}.
\]
证毕。
\end{proof}

\begin{theorem}[固定相位切片上的二次稳定性]\label{thm:conclusion-reversekl-affine-phase-slice-quadratic-stability}
固定整数 \(n\ge 1\)、参数 \(0<c<1\)、相位 \(\theta_0\in\TT\)，并定义仿射切片
\[
\mathcal A_{n,c,\theta_0}
:=
\{\mu\in\mathrm{Prob}(\TT):\ \widehat\mu(n)=ce^{-in\theta_0}\}.
\]
则 \(\mathcal A_{n,c,\theta_0}\) 为凸集，且 \(\nu_{c,n,\theta_0}\) 是其上 \(\mathcal H_r\) 的唯一极小点。更强地，对任意
\[
\mu\in\mathcal A_{n,c,\theta_0}
\]
都成立定量二次稳定性估计
\[
\mathcal H_r(\mu)-\mathcal H_r(\nu_{c,n,\theta_0})
\ge
\frac{1}{4b_r^2}\,\|\mu-\nu_{c,n,\theta_0}\|_r^2,
\qquad
b_r:=\frac{1+r}{1-r},
\]
其中 \(\|\cdot\|_r\) 取自定理 \ref{thm:xi-reversekl-affine-second-variation-strong-convex}。
\end{theorem}
\begin{proof}
由于 \(\mu\mapsto \widehat\mu(n)\) 是仿射映射，\(\mathcal A_{n,c,\theta_0}\) 显然为凸集。由定理 \ref{thm:xi-reversekl-single-frequency-exact-minimizer}，
\[
\mathcal H_r(\nu_{c,n,\theta_0})=-\log(1-r^{2n}c^2),
\]
且 \(\nu_{c,n,\theta_0}\) 在更大的约束集 \(\{|\widehat\mu(n)|=c\}\) 上已是极小测度，故在切片 \(\mathcal A_{n,c,\theta_0}\) 上也是极小点。再由推论 \ref{cor:xi-reversekl-strict-convex-unique-minimizer}，凸集上的极小点必唯一，故唯一性成立。

设
\[
\nu:=\nu_{c,n,\theta_0},
\qquad
\mu_{1/2}:=\frac{\mu+\nu}{2}\in\mathcal A_{n,c,\theta_0}.
\]
由于 \(\nu\) 为 \(\mathcal A_{n,c,\theta_0}\) 上极小点，故
\[
\mathcal H_r(\mu_{1/2})\ge \mathcal H_r(\nu).
\]
将定理 \ref{thm:xi-reversekl-affine-second-variation-strong-convex} 的弦差下界用于 \((\nu,\mu)\) 与 \(\lambda=\frac12\)，得到
\[
\frac12\mathcal H_r(\nu)+\frac12\mathcal H_r(\mu)-\mathcal H_r(\mu_{1/2})
\ge
\frac{1}{8b_r^2}\,\|\mu-\nu\|_r^2.
\]
再代入 \(\mathcal H_r(\mu_{1/2})\ge \mathcal H_r(\nu)\)，便有
\[
\frac12\bigl(\mathcal H_r(\mu)-\mathcal H_r(\nu)\bigr)
\ge
\frac{1}{8b_r^2}\,\|\mu-\nu\|_r^2.
\]
整理即得所求二次稳定性不等式。证毕。
\end{proof}

\begin{corollary}[固定相位切片上的熵极小序列强收敛到 Poisson 极值]\label{cor:conclusion-reversekl-affine-phase-slice-strong-convergence}
沿用定理 \ref{thm:conclusion-reversekl-affine-phase-slice-quadratic-stability} 的设定。
若一列测度 \(\{\mu_j\}\subset \mathcal A_{n,c,\theta_0}\) 满足
\[
\mathcal H_r(\mu_j)\downarrow -\log(1-r^{2n}c^2),
\]
则
\[
\|\mu_j-\nu_{c,n,\theta_0}\|_r\longrightarrow 0.
\]
特别地，
\[
\mu_j\stackrel{w^\ast}{\longrightarrow}\nu_{c,n,\theta_0}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-reversekl-affine-phase-slice-quadratic-stability}，
\[
0\le \frac{1}{4b_r^2}\,\|\mu_j-\nu_{c,n,\theta_0}\|_r^2
\le
\mathcal H_r(\mu_j)-\mathcal H_r(\nu_{c,n,\theta_0}).
\]
右端按假设趋于 \(0\)，故
\[
\|\mu_j-\nu_{c,n,\theta_0}\|_r\to 0.
\]
又由 \(\|\cdot\|_r\) 的定义，
\[
\|\mu_j-\nu_{c,n,\theta_0}\|_r^2
=
\sum_{k\in\ZZ}r^{2|k|}\bigl|\widehat\mu_j(k)-\widehat{\nu_{c,n,\theta_0}}(k)\bigr|^2.
\]
由于 \(r^{|k|}>0\) 对每个固定 \(k\) 都严格正，故
\[
\widehat\mu_j(k)\longrightarrow \widehat{\nu_{c,n,\theta_0}}(k)
\qquad (k\in\ZZ).
\]
由 Fourier--Stieltjes 系数逐点收敛推出弱-\(*\) 收敛，得到
\[
\mu_j\stackrel{w^\ast}{\longrightarrow}\nu_{c,n,\theta_0}.
\]
证毕。
\end{proof}
