若进一步把 bare exponential source 直接置于 Zeckendorf 规范截面上，则同一结论层还可抽出另一条与零温端点相接的封口链。其出发点只需主体中已经固定的三项接口：合法词语言
\[
X_m:=\{w\in\{0,1\}^m:\ w_iw_{i+1}=0\},
\]
Zeckendorf 取值
\[
V_m(w):=\sum_{k=1}^{m}w_kF_{k+1},
\]
以及 \(V_m:X_m\to\{0,1,\dots,F_{m+2}-1\}\) 的范围双射。沿此接口，指数级语言可被阶乘权完全重标架为 \(e\) 的初段级数，而由此诱导的热力学、边界分布与编码代价都会塌缩到常数或零速率层级。

\begin{theorem}[Zeckendorf 规范截面的 Euler 重标架]\label{thm:conclusion-zeckendorf-euler-reindexing}
定义
\[
\mathcal E_m:=\sum_{w\in X_m}\frac{1}{V_m(w)!}.
\]
则对每个 \(m\ge 1\)，都有精确恒等式
\[
\mathcal E_m=\sum_{n=0}^{F_{m+2}-1}\frac{1}{n!}.
\]
进一步，定义约化词集
\[
\mathcal X_{\mathrm{red}}
:=
\{\varnothing\}
\cup
\bigcup_{m\ge 1}\{w\in X_m:\ w_m=1\},
\]
并约定 \(V(\varnothing)=0\)、对非空 \(w\) 取 \(V(w):=V_m(w)\)。则有
\[
e=\sum_{w\in\mathcal X_{\mathrm{red}}}\frac{1}{V(w)!}.
\]
\leanverified{paper\_conclusion\_zeckendorf\_euler\_reindexing\_seeds}
\leanverified{paper\_conclusion\_zeckendorf\_euler\_reindexing\_package}
\leanverified{paper\_conclusion\_zeckendorf\_euler\_reindexing}
\end{theorem}
\begin{proof}
由主体中已经建立的 Zeckendorf 范围双射，
\[
V_m:X_m\longrightarrow\{0,1,\dots,F_{m+2}-1\}
\]
为双射，故有限和换元立即给出
\[
\mathcal E_m=\sum_{n=0}^{F_{m+2}-1}\frac{1}{n!}.
\]

对第二式，Zeckendorf 唯一性表明：每个正整数 \(n\ge 1\) 恰对应唯一一个末位为 \(1\) 的有限合法词，而 \(n=0\) 对应空词。于是
\[
\sum_{w\in\mathcal X_{\mathrm{red}}}\frac{1}{V(w)!}
=
\sum_{n=0}^{\infty}\frac{1}{n!}
=
e.
\]
证毕。
\end{proof}

\begin{corollary}[Zeckendorf--Euler 重标架的阶乘型超指数误差]\label{cor:conclusion-zeckendorf-euler-superexponential-tail}
记
\[
R_m:=e-\mathcal E_m.
\]
则对每个 \(m\ge 1\)，有
\[
0<R_m<\frac{2}{F_{m+2}!}.
\]
从而
\[
\log R_m=-F_{m+2}\log F_{m+2}+O(F_{m+2})=-\Theta(m\varphi^m).
\]
因此，在深度变量 \(m\) 上，Euler 重标架误差呈阶乘尺度的超指数衰减。
\leanverified{paper\_conclusion\_zeckendorf\_euler\_superexponential\_tail}
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-zeckendorf-euler-reindexing}，
\[
R_m=\sum_{n=F_{m+2}}^\infty \frac{1}{n!}.
\]
令 \(N:=F_{m+2}\)。则
\[
\sum_{n=N}^{\infty}\frac{1}{n!}
\le
\frac{1}{N!}\sum_{j\ge 0}\frac{1}{(N+1)^j}
<
\frac{2}{N!},
\]
从而得到所示上界。
再由 Stirling 公式
\[
\log(N!)=N\log N-N+O(\log N)
\]
与 \(F_{m+2}\asymp \varphi^m\) 即得
\[
\log R_m=-F_{m+2}\log F_{m+2}+O(F_{m+2})=-\Theta(m\varphi^m).
\]
证毕。
\end{proof}

\begin{theorem}[factorial 势的正温度压力塌缩]\label{thm:conclusion-factorial-potential-positive-temperature-pressure-collapse}
\leanverified{paper\_conclusion\_factorial\_potential\_positive\_temperature\_pressure\_collapse}
对 \(\beta\ge 0\) 定义配分函数
\[
Z_m(\beta):=\sum_{w\in X_m}(V_m(w)!)^{-\beta}.
\]
则对每个 \(\beta>0\)，有
\[
Z_m(\beta)=\sum_{n=0}^{F_{m+2}-1}(n!)^{-\beta}
\longrightarrow
E(\beta):=\sum_{n=0}^{\infty}(n!)^{-\beta}<\infty.
\]
因此深度压力
\[
P(\beta):=\lim_{m\to\infty}\frac{1}{m}\log Z_m(\beta)
\]
满足
\[
P(\beta)=0\qquad(\beta>0),
\]
而
\[
P(0)=\lim_{m\to\infty}\frac{1}{m}\log|X_m|=\log\varphi.
\]
故 \(\beta=0\) 处出现严格跃迁：
\[
\lim_{\beta\downarrow 0}P(\beta)=0\neq \log\varphi=P(0).
\]
\end{theorem}
\begin{proof}
第一恒等式由定理 \ref{thm:conclusion-zeckendorf-euler-reindexing} 的同一换元直接得到。
当 \(\beta>0\) 时，
\[
\frac{(n+1)!^{-\beta}}{(n!)^{-\beta}}=(n+1)^{-\beta}\to 0,
\]
故级数 \(E(\beta)\) 绝对收敛，从而 \(Z_m(\beta)\) 收敛到有限正常数 \(E(\beta)\)，于是
\[
\frac{1}{m}\log Z_m(\beta)\to 0.
\]
当 \(\beta=0\) 时，
\[
Z_m(0)=|X_m|=F_{m+2},
\]
而 \(F_{m+2}\sim \varphi^{m+2}/\sqrt5\)，故
\[
P(0)=\lim_{m\to\infty}\frac{1}{m}\log F_{m+2}=\log\varphi.
\]
证毕。
\end{proof}

\begin{theorem}[Zeckendorf--Euler 测度的 Poisson 边界律]\label{thm:conclusion-zeckendorf-euler-poisson-boundary}
\leanverified{paper\_conclusion\_zeckendorf\_euler\_poisson\_boundary}
在 \(X_m\) 上定义概率测度
\[
\mu_m(w):=\frac{1}{\mathcal E_m\,V_m(w)!}.
\]
若 \(W_m\sim \mu_m\)，并记
\[
N_m:=V_m(W_m),
\]
则
\[
\PP(N_m=n)=\frac{1}{\mathcal E_m\,n!}\mathbf 1_{\{0\le n<F_{m+2}\}}.
\]
亦即，\(N_m\) 恰为参数 \(1\) 的 Poisson 分布在区间 \([0,F_{m+2}-1]\) 上的重整化截断。更强地，
\[
d_{\mathrm{TV}}\!\left(N_m,\operatorname{Poisson}(1)\right)=\frac{e-\mathcal E_m}{e}=\frac{R_m}{e},
\]
从而
\[
d_{\mathrm{TV}}\!\left(N_m,\operatorname{Poisson}(1)\right)<\frac{2}{e\,F_{m+2}!}.
\]
\end{theorem}
\begin{proof}
第一式由 \(\mu_m\) 的定义和定理 \ref{thm:conclusion-zeckendorf-euler-reindexing} 直接得到。
设 \(p_n:=e^{-1}/n!\) 为 \(\operatorname{Poisson}(1)\) 的质量函数，并令
\[
q_n:=\frac{1}{\mathcal E_m n!}\mathbf 1_{\{0\le n<F_{m+2}\}}.
\]
则当 \(n<F_{m+2}\) 时，
\[
q_n-p_n=\left(\frac{e}{\mathcal E_m}-1\right)p_n>0,
\]
而当 \(n\ge F_{m+2}\) 时，\(q_n=0\)。于是
\[
2d_{\mathrm{TV}}(q,p)
=
\sum_{n<F_{m+2}}(q_n-p_n)+\sum_{n\ge F_{m+2}}p_n
=
\left(1-\frac{\mathcal E_m}{e}\right)+\frac{R_m}{e}
=
\frac{2R_m}{e},
\]
故
\[
d_{\mathrm{TV}}(q,p)=\frac{R_m}{e}.
\]
再结合推论 \ref{cor:conclusion-zeckendorf-euler-superexponential-tail} 即得上界。证毕。
\end{proof}

\begin{theorem}[Rényi--Shannon 熵梯的有限极限]\label{thm:conclusion-zeckendorf-euler-renyi-shannon-finite-limit}
设 \(\alpha>0\)、\(\alpha\neq 1\)，并定义
\[
E(\alpha):=\sum_{n=0}^{\infty}(n!)^{-\alpha}.
\]
则 \(\mu_m\) 的 Rényi 熵满足
\[
H_\alpha(\mu_m)
:=
\frac{1}{1-\alpha}\log\sum_{w\in X_m}\mu_m(w)^\alpha
\longrightarrow
\frac{1}{1-\alpha}\bigl(\log E(\alpha)-\alpha\bigr).
\]
同时 Shannon 熵亦收敛，并且
\[
H(\mu_m)\longrightarrow
1+\frac{1}{e}\sum_{n=0}^{\infty}\frac{\log(n!)}{n!}
\approx 1.3048422423\ \text{nats}
\approx 1.8824894320\ \text{bits}.
\]
因此，尽管 \(|X_m|\sim \varphi^m\) 指数增长，Euler 源诱导的整条 Rényi--Shannon 熵梯仍全部稳定在常数级。
\end{theorem}
\begin{proof}
由定义与定理 \ref{thm:conclusion-zeckendorf-euler-reindexing}，
\[
\sum_{w\in X_m}\mu_m(w)^\alpha
=
\mathcal E_m^{-\alpha}\sum_{w\in X_m}(V_m(w)!)^{-\alpha}
=
\mathcal E_m^{-\alpha}Z_m(\alpha).
\]
由定理 \ref{thm:conclusion-factorial-potential-positive-temperature-pressure-collapse}，\(Z_m(\alpha)\to E(\alpha)\)，且 \(\mathcal E_m\to e\)，故
\[
H_\alpha(\mu_m)\to \frac{1}{1-\alpha}\bigl(\log E(\alpha)-\alpha\bigr).
\]

对 Shannon 熵，直接计算得
\[
H(\mu_m)
=
-\sum_{w\in X_m}\mu_m(w)\log\mu_m(w)
=
\log\mathcal E_m+\frac{1}{\mathcal E_m}\sum_{n=0}^{F_{m+2}-1}\frac{\log(n!)}{n!}.
\]
由于
\[
\sum_{n=0}^{\infty}\frac{\log(n!)}{n!}<\infty,
\]
可由支配收敛定理得到
\[
H(\mu_m)\longrightarrow
\log e+\frac{1}{e}\sum_{n=0}^{\infty}\frac{\log(n!)}{n!}
=
1+\frac{1}{e}\sum_{n=0}^{\infty}\frac{\log(n!)}{n!}.
\]
数值近似由该闭式直接求值得到。证毕。
\end{proof}

\begin{proposition}[factorial canonical section 的零速率可编码性]\label{prop:conclusion-factorial-canonical-section-zero-rate-code}
存在一族前缀码 \(\mathcal C_m\)，使得对 \(W_m\sim \mu_m\)，有
\[
\sup_{m\ge 1}\EE\,|\mathcal C_m(W_m)|<\infty.
\]
特别地，
\[
\lim_{m\to\infty}\frac{1}{m}\EE\,|\mathcal C_m(W_m)|=0.
\]
\end{proposition}
\begin{proof}
先对整数随机变量 \(N_m=V_m(W_m)\) 编码。取任意通用整数前缀码，其码长满足
\[
\ell(n)\le c_1+c_2\log_2(n+2)
\]
对某些绝对常数 \(c_1,c_2>0\) 成立。于是
\[
\EE\,\ell(N_m)
\le
c_1+\frac{c_2}{\mathcal E_m}\sum_{n=0}^{F_{m+2}-1}\frac{\log_2(n+2)}{n!}.
\]
由于 \(\inf_m \mathcal E_m\ge 2\)，且
\[
\sum_{n=0}^{\infty}\frac{\log_2(n+2)}{n!}<\infty,
\]
故右端在 \(m\) 上一致有界。

另一方面，\(V_m:X_m\to\{0,\dots,F_{m+2}-1\}\) 为双射，故可先用该整数前缀码编码 \(N_m\)，再经 \(V_m^{-1}\) 无损译回 \(W_m\)。于是存在前缀码族 \(\mathcal C_m\) 使
\[
\sup_{m\ge 1}\EE\,|\mathcal C_m(W_m)|<\infty.
\]
两边同除以 \(m\) 即得零速率结论。证毕。
\end{proof}

\begin{remark}[均匀微态位率门槛与 canonical section 的分离]\label{rem:conclusion-factorial-canonical-section-vs-uniform-capacity}
上述零速率可编码性并不与均匀微态基线下的正位率恢复门槛相冲突。后者约束的是在纤维反演与均匀源口径下的完整微态恢复，而这里的 \(\mu_m\) 已把全部质量推入由 \(1/n!\) 权重支配的 canonical section；因此塌缩为零速率的并非 \(\Fold_m\) 的绝对几何复杂度，而是该几何在 factorial 倾斜下的有效统计维度。
\end{remark}
