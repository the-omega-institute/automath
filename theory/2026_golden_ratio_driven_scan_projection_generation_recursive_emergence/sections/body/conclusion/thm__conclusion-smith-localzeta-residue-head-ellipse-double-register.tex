\begin{theorem}[局部 \texorpdfstring{$\mathcal Z$}{Z} 因子的极点--留数--head 三元组]\label{thm:conclusion-smith-local-zeta-pole-residue-head-triple}
\leanverified{paper\_conclusion\_smith\_local\_zeta\_pole\_residue\_head\_triple}
设 \(A\in M_{m\times n}(\ZZ)\) 满足 \(\operatorname{rank}_{\QQ}(A)=m\)，并固定素数 \(p\)。沿用定理 \ref{thm:conclusion-smith-mobius-zeta-singlepole-platform} 的记号
\[
h_p:=\max_i e_i(p),
\qquad
E_p:=\sum_{i=1}^{m}e_i(p),
\qquad
s_0:=n-m.
\]
则局部因子
\[
\mathcal Z_{A,p}(s)
=
\sum_{k\ge 0}p^{-k(s-n+m)+f_p(k)}
\]
同时唯一决定如下三项数据：
\begin{enumerate}
  \item 极点位置 \(s_0=n-m\)，它只依赖自由亏损 \(n-m\)；
  \item 留数
  \[
  \operatorname*{Res}_{s=s_0}\mathcal Z_{A,p}(s)=\frac{p^{E_p}}{\log p},
  \]
  它只依赖 \(p\)-主总亏损质量 \(E_p\)；
  \item head 长度
  \[
  h_p=\min\{k\ge 0:\ f_p(k)=E_p\}
  =\min\{k\ge 0:\ \Delta_p(k+1)=0\},
  \]
  它只依赖最大局部指数 \(h_p=\max_i e_i(p)\)。
\end{enumerate}
因此，局部 \(\mathcal Z\)-因子在解析层首先分裂为“自由亏损位置、总质量留数、最大高度 head 长度”三个彼此正交的局部坐标；而与推论 \ref{cor:conclusion-smith-local-zeta-complete-invariant} 合并后，它们又共同嵌入对完整 \(p\)-主 Smith 谱的完备描述。
\end{theorem}
\begin{proof}
极点位置与留数公式正是定理 \ref{thm:conclusion-smith-mobius-zeta-singlepole-platform} 的直接内容。

再由同一定理中的有理尾分解，
\[
\mathcal Z_{A,p}(s)
=
\sum_{k=0}^{h_p-1}p^{-k(s-n+m)+f_p(k)}
\;+\;
\frac{p^{-h_p(s-n+m)+E_p}}{1-p^{-(s-n+m)}},
\]
可见当且仅当 \(k\ge h_p\) 时，\(\min(k,e_i(p))=e_i(p)\) 对所有 \(i\) 同时成立，也即
\[
f_p(k)=E_p.
\]
于是
\[
h_p=\min\{k\ge 0:\ f_p(k)=E_p\}.
\]
又因为
\[
\Delta_p(k+1)=f_p(k+1)-f_p(k)
\]
是一个最终归零的非负整数列，上式等价于
\[
h_p=\min\{k\ge 0:\ \Delta_p(k+1)=0\}.
\]
证毕。
\end{proof}

\begin{corollary}[unimodular 判别的留数刚性]\label{cor:conclusion-smith-unimodular-residue-rigidity}
\leanverified{paper\_conclusion\_smith\_unimodular\_residue\_rigidity}
在上述设定下，下列条件等价：
\[
d_1=\cdots=d_m=1;
\]
\[
E_p=0
\qquad(\forall p);
\]
\[
\operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)=\frac{1}{\log p}
\qquad(\forall p).
\]
\end{corollary}
\begin{proof}
Smith 不变量全为 \(1\) 当且仅当每个素数方向都没有 \(p\)-主亏损质量，即
\[
E_p=\sum_{i=1}^{m}e_i(p)=0
\qquad(\forall p).
\]
再由定理 \ref{thm:conclusion-smith-local-zeta-pole-residue-head-triple} 中的留数公式，
\[
\operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)=\frac{p^{E_p}}{\log p},
\]
便得到所述等价性。证毕。
\end{proof}

\begin{proposition}[局部留数层的分类盲性]\label{prop:conclusion-smith-local-residue-classification-blindness}
\leanverified{paper\_conclusion\_smith\_local\_residue\_classification\_blindness}
局部留数
\[
\operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)
\]
只能恢复总质量 \(E_p\)，却不足以恢复完整的 \(p\)-主 Smith 谱。更精确地，当 \(m\ge 3\) 时，局部谱型
\[
(e_i(p))=(2,1,0,\dots,0)
\qquad\text{与}\qquad
(e_i(p))=(1,1,1,0,\dots,0)
\]
具有相同留数
\[
\frac{p^3}{\log p},
\]
但对应不同的局部 \(\mathcal Z\)-因子。
\end{proposition}
\begin{proof}
这两种谱型都满足
\[
E_p=\sum_{i=1}^{m}e_i(p)=3,
\]
故由定理 \ref{thm:conclusion-smith-local-zeta-pole-residue-head-triple} 有相同留数
\[
\operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)=\frac{p^3}{\log p}.
\]
然而第一种谱型具有最大高度 \(h_p=2\)，第二种谱型具有 \(h_p=1\)。由同一定理，\(h_p\) 正是局部因子的首次平台长度；因此两者的有理尾平台起点不同，局部 \(\mathcal Z\)-因子不可能相同。证毕。
\end{proof}

\begin{theorem}[局部留数乘积恢复全局 Smith 扭量体积]\label{thm:conclusion-smith-torsion-volume-from-local-residues}
\leanverified{paper\_conclusion\_smith\_torsion\_volume\_from\_local\_residues}
定义 Smith 扭量体积
\[
\tau(A):=\prod_{i=1}^{m}d_i.
\]
则有完全乘积公式
\[
\tau(A)
=
\prod_{p}\Bigl(\log p\cdot \operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)\Bigr).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-smith-local-zeta-pole-residue-head-triple}，
\[
\log p\cdot \operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)=p^{E_p}.
\]
因此
\[
\prod_{p}\Bigl(\log p\cdot \operatorname*{Res}_{s=n-m}\mathcal Z_{A,p}(s)\Bigr)
=
\prod_{p}p^{E_p}
=
\prod_{p}p^{\sum_{i=1}^{m}e_i(p)}
=
\prod_{i=1}^{m}\prod_{p}p^{e_i(p)}
=
\prod_{i=1}^{m}d_i
=
\tau(A).
\]
证毕。
\end{proof}

\begin{corollary}[有限 prime-stage 的局部 \texorpdfstring{$\mathcal Z$}{Z} 审计先天地不完备]\label{cor:conclusion-smith-finite-prime-stage-local-zeta-incomplete}
任取有限素数集 \(S\subset \mathbb P\)。若审计器只读取
\[
\{\mathcal Z_{A,p}(s)\}_{p\in S}
\]
或等价地只读取
\[
\{K_p(k)\}_{p\in S,\ k\ge 1},
\]
则它不可能唯一恢复完整 Smith 谱；更强地，这样的有限视界审计类中始终含有无穷多个两两不同的 Smith completion。
\end{corollary}
\begin{proof}
由推论 \ref{cor:conclusion-smith-local-zeta-complete-invariant}，单个局部因子与单个 \(p\)-主 Smith 谱等价，因此固定有限 \(S\) 的局部因子族只决定 \(S\)-主部分。
另一方面，推论 \ref{cor:conclusion-finite-prime-stage-solenoid-smith-incompleteness} 已经表明：任何固定有限 prime-stage 都保留无穷多个两两不同的 Smith completion。故有限 prime-stage 的局部 \(\mathcal Z\) 审计先天地不完备。证毕。
\end{proof}

\begin{theorem}[整数轴对齐椭圆半群的双寄存器自由实现与线性化障碍]\label{thm:conclusion-integer-ellipse-double-register-free-realization}
\leanverified{paper\_conclusion\_integer\_ellipse\_double\_register\_free\_realization}
记
\[
\mathsf E_{\NN}:=\{E_{a,b}:a,b\in\NN_{>0}\},
\]
其乘法由逐坐标半轴相乘给出。则：
\begin{enumerate}
  \item \((\mathsf E_{\NN},\odot)\) 与两份 prime-register 自由交换幺半群的直积规范同构：
  \[
  (\mathsf E_{\NN},\odot)
  \cong
  \bigl(\NN^{(\mathbb P)}\times \NN^{(\mathbb P)},+\bigr).
  \]
  因而它可被理解为一个纯双寄存器初对象：一份寄存器记录横轴素数估值，另一份寄存器记录纵轴素数估值。
  \item 不存在任何单射幺半群同态
  \[
  \mathsf E_{\NN}\hookrightarrow M,
  \]
  其中 \(M\) 为有限生成交换可消去幺半群；特别地，对任意有限 \(k\)，都不存在忠实线性化
  \[
  \mathsf E_{\NN}\hookrightarrow (\NN^k,+).
  \]
  \item 若以集合嵌入衡量实现层复杂度，而以幺半群同态嵌入衡量结构层复杂度，则
  \[
  \mathrm{cdim}^{\mathrm{impl}}(\mathsf E_{\NN})=\frac12,
  \qquad
  \mathrm{cdim}^{\mathrm{hom}}_{+}(\mathsf E_{\NN})=+\infty.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
第一条是定理 \ref{thm:xi-integer-ellipse-free-commutative-monoid} 与定理 \ref{thm:killo-ellipse-diagonal-prime-register-equivalence} 的直接合并表述：每个 \(E_{a,b}\) 唯一对应于两族有限支撑素数估值向量 \((v_p(a))_p\) 与 \((v_p(b))_p\)，从而得到所示同构。

第二条由定理 \ref{thm:conclusion-integer-ellipse-grothendieck-double-countable-rank} 的第二、第三条直接给出。

对第三条，\(\mathsf E_{\NN}\) 是可数无限集，故由命题 \ref{prop:killo-implementation-vs-structural-half-circle-dimension} 的实现层定义，
\[
\mathrm{cdim}^{\mathrm{impl}}(\mathsf E_{\NN})=\frac12.
\]
而第二条已排除一切有限秩加性线性化；再由同一命题的结构层定义，得到
\[
\mathrm{cdim}^{\mathrm{hom}}_{+}(\mathsf E_{\NN})=+\infty.
\]
证毕。
\end{proof}

\begin{conjecture}[有界总亏损下的局部 \texorpdfstring{$\mathcal Z$}{Z} 喷射有限决定性]\label{conj:conclusion-smith-bounded-total-loss-local-zeta-jet-determinacy}
固定素数 \(p\) 与整数 \(M\ge 1\)。考虑全体满足
\[
E_p=\sum_{i=1}^{m}e_i(p)\le M
\]
的满行秩整数矩阵。则存在一个只依赖于 \(M\) 的整数 \(J(M)\)，使得正规化局部因子
\[
\bigl(1-p^{-(s-n+m)}\bigr)\mathcal Z_{A,p}(s)
\]
在 \(s=n-m\) 处的 \(J(M)\)-阶 Taylor 喷射，已经唯一决定多重集 \(\{e_i(p)\}_{i=1}^{m}\)。
\end{conjecture}
\begin{proof}[论据]
定理 \ref{thm:conclusion-smith-local-zeta-pole-residue-head-triple} 表明：零阶留数已经恢复总质量 \(E_p\)，而同一局部因子首次进入平台的时刻恢复最大指数 \(h_p\)。另一方面，推论 \ref{cor:conclusion-smith-local-zeta-complete-invariant} 说明完整局部因子已经是完备不变量。因而在 \(E_p\le M\) 的有界族上，局部因子的有限阶 Taylor 喷射应当足以恢复全部有限分拆数据；缺失的一步只是把“完整函数完备性”压缩为“有界质量族上的有限喷射完备性”。证毕。
\end{proof}
