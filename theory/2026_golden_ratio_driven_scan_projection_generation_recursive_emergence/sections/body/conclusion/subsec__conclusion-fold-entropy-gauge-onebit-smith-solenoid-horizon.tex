\subsection{输出熵、规范体积、末位充分统计与 Smith--solenoid 的有限视界歧义}\label{subsec:conclusion-fold-entropy-gauge-onebit-smith-solenoid-horizon}
沿用小节 \ref{subsec:conclusion-maxfiber-average-scale-rigidity}、小节 \ref{subsec:conclusion-smith-kernel-audit-modularity-blind-direction}、定理 \ref{thm:xi-foldbin-gauge-entropy-one-nat-law}、推论 \ref{cor:xi-foldbin-shannon-gauge-constant-mirror}、结论 \ref{con:xi-terminal-foldbin-lastbit-sufficient-statistic}、命题 \ref{prop:xi-foldbin-renyi-deficit-spectrum-closed-form}、定理 \ref{thm:xi-time-part9od-escort-escort-fdiv-binary-closure}、定理 \ref{thm:xi-smith-finite-pwindow-complete-recovery-determinant-index} 与定理 \ref{thm:killo-localization-circle-dimension-prime-ledger-splitting} 的记号。前述链条已经分别把 bin-fold 的输出熵、规范体积、最大纤维尺度、末位充分统计，以及 Smith 的局部 \(p\)-窗、prime-register 核与有限阶段 arithmetic solenoid 的可见层压缩为显式闭式。把这些接口并置，可进一步抽出如下后果。

\begin{theorem}[输出熵与规范体积密度的仿射恒等式]\label{thm:conclusion-fold-output-entropy-gauge-affine-identity}
记
\[
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
H(\mu_m):=-\sum_{w\in X_m}\mu_m(w)\log\mu_m(w),
\]
\[
\kappa_m:=2^{-m}\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w),
\qquad
g_m:=2^{-m}\log|G_m^{\mathrm{bin}}|.
\]
则对每个 \(m\ge 1\) 都有精确恒等式
\[
\kappa_m=m\log 2-H(\mu_m).
\]
进一步，
\[
g_m
=
m\log 2-H(\mu_m)-1
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
\leanverified{paper\_conclusion\_fold\_output\_entropy\_gauge\_affine\_identity}
\end{theorem}
\begin{proof}
由定义
\[
\mu_m(w)=\frac{d_m^{\mathrm{bin}}(w)}{2^m}
\]
可得
\[
\log d_m^{\mathrm{bin}}(w)=m\log 2+\log\mu_m(w).
\]
于是
\[
\kappa_m
=
\sum_{w\in X_m}\mu_m(w)\log d_m^{\mathrm{bin}}(w)
=
m\log 2+\sum_{w\in X_m}\mu_m(w)\log\mu_m(w)
=
m\log 2-H(\mu_m).
\]
这就得到第一式。

另一方面，定理 \ref{thm:xi-foldbin-gauge-entropy-one-nat-law} 已给出
\[
g_m
=
m\log 2-H(\mu_m)-1
+\frac{(\varphi/2)^m}{2\sqrt5}
\Bigl(
\varphi^2 m\log\frac{2}{\varphi}
+\varphi^2\log(2\pi)-\log\varphi
\Bigr)
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
由于括号内仅为 \(O(m)\)，故第二式随之成立。证毕。
\end{proof}

\begin{theorem}[隐藏位预算的一纳特税律]\label{thm:conclusion-hidden-visible-budget-one-nat-tax}
\leanverified{paper\_conclusion\_hidden\_visible\_budget\_one\_nat\_tax}
在 bin-fold 口径下，定义隐藏位预算
\[
h_m^{\mathrm{hid}}
:=
\sum_{w\in X_m}\mu_m(w)\log_2 d_m^{\mathrm{bin}}(w),
\qquad
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\]
以及可见位预算
\[
h_m^{\mathrm{vis}}:=m-h_m^{\mathrm{hid}}.
\]
则规范体积密度满足
\[
g_m
=
(\log 2)\,h_m^{\mathrm{hid}}-1
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right),
\]
等价地，
\[
g_m
=
m\log 2-(\log 2)\,h_m^{\mathrm{vis}}-1
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
特别地，
\[
g_m=(\log 2)\,h_m^{\mathrm{hid}}-1+o(1).
\]
因此，隐藏位预算一旦换算到自然对数单位，规范体积密度恰好比它少去一纳特；这一常数税与 \(m\) 无关，而只来自 Stirling 首项中的全局 \(-d\) 余迹。
\end{theorem}
\begin{proof}
由定义，
\[
(\log 2)\,h_m^{\mathrm{hid}}
=
\sum_{w\in X_m}\mu_m(w)\log d_m^{\mathrm{bin}}(w)
=
\kappa_m.
\]
再由定理 \ref{thm:conclusion-fold-output-entropy-gauge-affine-identity}，
\[
g_m=\kappa_m-1+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
两式合并即得
\[
g_m=(\log 2)\,h_m^{\mathrm{hid}}-1+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
另一方面，由定义 \(h_m^{\mathrm{vis}}=m-h_m^{\mathrm{hid}}\)，故
\[
(\log 2)\,h_m^{\mathrm{hid}}=m\log 2-(\log 2)\,h_m^{\mathrm{vis}},
\]
代回便得第二式。最后利用 \(\varphi/2<1\)，误差项即为 \(o(1)\)。证毕。
\end{proof}

\noindent
因此，在 bin-fold 口径下，规范体积并不是独立于输出统计的第二账本；它只是输出 Shannon 熵经由一条斜率为 \(-1\)、常数位移为 \(1\) 的仿射重标后，再附加指数小校正项所得的同一量。

\begin{corollary}[规范体积的超指数律]\label{cor:conclusion-gauge-volume-superexponential-law}\leanverified{paper\_conclusion\_gauge\_volume\_superexponential\_law}
记
\[
c_\varphi:=-\frac{\log\varphi}{\varphi\sqrt5}.
\]
则有
\[
\kappa_m=m\log\frac{2}{\varphi}+c_\varphi+o(1),
\qquad
g_m=m\log\frac{2}{\varphi}+c_\varphi-1+o(1).
\]
从而
\[
\log |G_m^{\mathrm{bin}}|
=
2^m\left(
m\log\frac{2}{\varphi}+c_\varphi-1+o(1)
\right),
\]
特别地，
\[
|G_m^{\mathrm{bin}}|
=
\exp\!\bigl(\Theta(m2^m)\bigr),
\]
故其关于分辨率 \(m\) 的增长是严格超指数的。
\end{corollary}
\begin{proof}
推论 \ref{cor:xi-foldbin-shannon-gauge-constant-mirror} 已给出
\[
\kappa_m
=
m\log\frac{2}{\varphi}
-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right),
\]
以及
\[
g_m
=
m\log\frac{2}{\varphi}
-1-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
把
\(
c_\varphi:=-\frac{\log\varphi}{\varphi\sqrt5}
\)
代入即得前两式，其中误差项当然是 \(o(1)\)。

再由
\[
\log |G_m^{\mathrm{bin}}|=2^m g_m
\]
便得所示展开。因为 \(g_m\) 的主项为 \(m\log(2/\varphi)\)，故 \(\log |G_m^{\mathrm{bin}}|=\Theta(m2^m)\)，于是
\(
|G_m^{\mathrm{bin}}|=\exp(\Theta(m2^m))
\)
并严格超指数增长。证毕。
\end{proof}

\noindent
这说明输出统计中仅为 \(O(1)\) 量级的常数漂移，经由规范体积的 \(2^m\)-倍放大后，会在账本体量上显影为 \(\Theta(m2^m)\) 级的指数成本。

\begin{corollary}[熵常数漂移在规范体积上的 \texorpdfstring{$2^m$}{2^m} 倍放大]\label{cor:conclusion-entropy-constant-gauge-superexponential-amplification}
\leanverified{paper\_conclusion\_entropy\_constant\_gauge\_superexponential\_amplification}
若存在有界实数列 \((\delta_m)_{m\ge 1}\) 使
\[
H(\mu_m)=m\log\varphi+\delta_m,
\qquad
\delta_m=O(1),
\]
则
\[
\log|G_m^{\mathrm{bin}}|
=
2^m\left(
m\log\frac{2}{\varphi}-\delta_m-1
\right)
+o(2^m).
\]
更一般地，设 \(\{(\mu_m^{(i)},G_m^{(i)})\}_{m\ge 1}\)（\(i=1,2\)）为两列候选机制，满足
\[
H(\mu_m^{(i)})=m\log\varphi+\delta_m^{(i)},
\qquad
\delta_m^{(i)}=O(1),
\]
且
\[
\log|G_m^{(i)}|
=
2^m\left(
m\log\frac{2}{\varphi}-\delta_m^{(i)}-1
\right)
+o(2^m).
\]
若
\[
\delta_m^{(1)}-\delta_m^{(2)}\to c,
\]
则
\[
\log\frac{|G_m^{(1)}|}{|G_m^{(2)}|}
=
-c\,2^m+o(2^m).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-fold-output-entropy-gauge-affine-identity}，
\[
g_m
=
m\log 2-H(\mu_m)-1
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
把
\(
H(\mu_m)=m\log\varphi+\delta_m
\)
代入，得到
\[
g_m
=
m\log\frac{2}{\varphi}-\delta_m-1
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
再乘以 \(2^m\)，并注意
\[
2^m\cdot O\!\left(m\left(\frac{\varphi}{2}\right)^m\right)
=
O(m\varphi^m)
=
o(2^m),
\]
便得第一式。

对两列候选机制，上述展开逐项相减即可得到
\[
\log|G_m^{(1)}|-\log|G_m^{(2)}|
=
-\bigl(\delta_m^{(1)}-\delta_m^{(2)}\bigr)2^m+o(2^m).
\]
令 \(m\to\infty\) 便得到最后一式。证毕。
\end{proof}

\begin{theorem}[最大纤维的非凝聚律]\label{thm:conclusion-maxfiber-noncondensation-law}
\leanverified{paper\_conclusion\_maxfiber\_noncondensation\_law}
记
\[
M_m:=\max_{w\in X_m}d_m^{\mathrm{bin}}(w),
\qquad
\overline d_m:=\frac{2^m}{|X_m|}=\frac{2^m}{F_{m+2}}.
\]
则
\[
\lim_{m\to\infty}\frac1m\log\frac{M_m}{\overline d_m}=0,
\qquad
\frac{M_m}{\overline d_m}\longrightarrow \frac{\varphi^2}{\sqrt5}.
\]
若记
\[
C_\varphi^{\mathrm{mf}}:=\frac{\varphi^2}{\sqrt5}-1\approx 0.2244178274,
\]
则特别地
\[
\liminf_{m\to\infty}\frac{M_m}{\overline d_m}
\ge
1+C_\varphi^{\mathrm{mf}}>1.
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-maxfiber-average-scale-rigidity} 已经给出更强结论
\[
\frac{M_m}{\overline d_m}\longrightarrow \frac{\varphi^2}{\sqrt5}>1.
\]
两边取对数并除以 \(m\)，立得
\[
\lim_{m\to\infty}\frac1m\log\frac{M_m}{\overline d_m}=0.
\]
再把
\(
C_\varphi^{\mathrm{mf}}:=\varphi^2/\sqrt5-1
\)
代回，即得所示的严格常数级偏置下界。证毕。
\end{proof}

\noindent
于是，冻结、碰撞与大纤维现象并不是由极少数额外指数级的“怪物纤维”单独主导；顶端纤维确有永久偏置，但这一偏置只停留在常数倍层，而不再产生新的指数尺度。

\begin{theorem}[固定温度信息几何的一位塌缩]\label{thm:conclusion-fold-onebit-information-geometry-collapse}
\leanverified{paper\_conclusion\_fold\_onebit\_information\_geometry\_collapse}
记
\[
u_m(w):=\frac1{|X_m|},
\qquad
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
\pi_{m,q}(w):=\frac{d_m^{\mathrm{bin}}(w)^q}{\sum_{v\in X_m}d_m^{\mathrm{bin}}(v)^q},
\]
并令末位指标
\[
\ell(w):=w_m\in\{0,1\},
\qquad
\theta(q):=\frac{1}{1+\varphi^{q+1}},
\qquad
\beta_q:=(1-\theta(q),\theta(q)).
\]
把 \(\beta_q\) 视为定义在 \(\{0,1\}\) 上的 Bernoulli 律。则有：
\begin{enumerate}
  \item 存在实数列 \((\beta_m)_{m\ge 1}\) 满足 \(\beta_m\to\log\varphi\)。若定义
  \[
  \widetilde\mu_m(w):=Z_m^{-1}u_m(w)e^{-\beta_m\ell(w)},
  \]
  则存在常数 \(C>0\) 使
  \[
  D_{\mathrm{TV}}(\mu_m,\widetilde\mu_m)
  \le
  C\left(\frac{\varphi}{2}\right)^m.
  \]
  \item 对任意固定 \(p,q\ge 0\)，
  \[
  \lim_{m\to\infty}D_{\mathrm{KL}}(\pi_{m,q}\Vert \pi_{m,p})
  =
  D_{\mathrm{KL}}(\beta_q\Vert \beta_p).
  \]
  \item 对任意固定 \(\alpha\in(0,\infty)\setminus\{1\}\) 与任意固定 \(p,q\ge 0\)，
  \[
  \lim_{m\to\infty}\frac1m D_\alpha(\mu_m\Vert u_m)=0,
  \qquad
  \lim_{m\to\infty}\frac1m D_\alpha(\pi_{m,q}\Vert \pi_{m,p})=0,
  \]
  并特别地，
  \[
  \lim_{m\to\infty}\frac1m D_{\mathrm{KL}}(\pi_{m,q}\Vert \pi_{m,p})=0.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
第 \(1\) 条正是结论 \ref{con:xi-terminal-foldbin-lastbit-sufficient-statistic} 的内容：其中已经证明 \(\mu_m\) 与按末位分块后的指数倾斜模型在全变差意义下以
\(
O((\varphi/2)^m)
\)
逼近，且倾斜参数 \(\beta_m\to\log\varphi\)。

第 \(2\) 条由定理 \ref{thm:xi-time-part9od-escort-escort-fdiv-binary-closure} 的 KL 极限闭式立即得到。

对第 \(3\) 条，命题 \ref{prop:xi-foldbin-renyi-deficit-spectrum-closed-form} 表明：对每个固定
\(
\alpha\in(0,\infty)\setminus\{1\}
\)，
\(
D_\alpha(\mu_m\Vert u_m)
\)
收敛到有限常数；故除以 \(m\) 后极限为 \(0\)。另一方面，定理 \ref{thm:xi-time-part9od-escort-escort-fdiv-binary-closure} 同时给出
\(
D_\alpha(\pi_{m,q}\Vert \pi_{m,p})
\)
与
\(
D_{\mathrm{KL}}(\pi_{m,q}\Vert \pi_{m,p})
\)
对任意固定 \(p,q\) 都收敛到有限极限，因此相应比率也都趋于 \(0\)。证毕。
\end{proof}

\noindent
由此，bin-fold 的固定温度信息几何在指数主阶上完全退化为末位一位 Bernoulli 影子：全部固定阶散度只保留 \(O(1)\) 量级，而不再携带正的信息率。

\begin{theorem}[截断 Smith 账本的无穷不可辨识性]\label{thm:conclusion-truncated-smith-ledger-infinite-indistinguishability}
\leanverified{paper\_conclusion\_truncated\_smith\_ledger\_infinite\_indistinguishability}
固定有限素数集 \(S\subset\PP\)、精度上界 \(K\in\NN\)，并令 \(A\in M_{m\times n}(\ZZ)\) 满足 \(\operatorname{rank}_{\QQ}(A)=m\)。记其 Smith 不变量为
\[
d_1\mid d_2\mid\cdots\mid d_m.
\]
定义截断可见账本
\[
\mathcal L_{S,K}(A)
:=
\bigl(K_p(k;A)\bigr)_{p\in S,\ 1\le k\le K},
\qquad
K_p(k;A):=\#\ker(A\bmod p^k).
\]
则对任意一个可实现的账本值 \(\mathcal L_{S,K}\)，都存在无穷多个两两不同的 Smith 不变量系
\[
\bigl(d_1^{(r)},\dots,d_m^{(r)}\bigr)\qquad(r\ge 1)
\]
使得每个不变量系都可由某个 \(m\times n\) 满行秩整数矩阵实现，且它们诱导出完全相同的截断账本 \(\mathcal L_{S,K}\)。
\end{theorem}
\begin{proof}
取任一实现该账本值的 Smith 数据
\[
d_1\mid d_2\mid\cdots\mid d_m.
\]
任选一枚不属于 \(S\) 的素数 \(\ell\notin S\)。对每个 \(r\ge 1\)，定义
\[
d_i^{(r)}:=d_i\quad(1\le i<m),
\qquad
d_m^{(r)}:=\ell^r d_m.
\]
由于原本就有 \(d_{m-1}\mid d_m\)，故仍有
\(
d_{m-1}^{(r)}\mid d_m^{(r)}
\)，于是这确实给出一族合法的 Smith 不变量序列。对每个 \(r\ge 1\)，取 \(A^{(r)}\in M_{m\times n}(\ZZ)\) 为实现该 Smith 数据的任意满行秩整数矩阵。

对任意 \(p\in S\)，记
\[
e_i(p):=v_p(d_i),
\qquad
e_i^{(r)}(p):=v_p(d_i^{(r)}).
\]
因为 \(\ell\neq p\)，故
\[
e_i^{(r)}(p)=e_i(p)\qquad(1\le i\le m).
\]
另一方面，由定理 \ref{thm:conclusion-smith-primepower-hessian-inversion} 中
\[
\xi_{A,p}(k)=\sum_{i=1}^{m}\min(k,e_i(p))
\]
的表示可知，对 \(A\) 与 \(A^{(r)}\) 分别有
\[
\log_p K_p(k;A)=k(n-m)+\sum_{i=1}^{m}\min(k,e_i(p)).
\]
\[
 \log_p K_p\bigl(k;A^{(r)}\bigr)
 =
 k(n-m)+\sum_{i=1}^{m}\min(k,e_i^{(r)}(p))
 =
 k(n-m)+\sum_{i=1}^{m}\min(k,e_i(p)).
 \]
因此对所有 \(p\in S\) 与 \(1\le k\le K\)，都有
\[
K_p\bigl(k;A^{(r)}\bigr)=K_p(k;A),
\]
也就是
\[
\mathcal L_{S,K}\bigl(A^{(r)}\bigr)=\mathcal L_{S,K}(A)
\qquad(\forall\,r\ge 1).
\]
然而不同的 \(r\) 给出不同的末项不变量 \(d_m^{(r)}=\ell^r d_m\)，故这些 Smith 不变量系两两不同。证毕。
\end{proof}

\begin{corollary}[有限 prime-stage solenoid 对完整 Smith 谱的先天不完备性]\label{cor:conclusion-finite-prime-stage-solenoid-smith-incompleteness}\leanverified{paper\_conclusion\_finite\_prime\_stage\_solenoid\_smith\_incompleteness}
固定有限素数集 \(S\subset\PP\)，并考虑 arithmetic solenoid 的短正合列
\[
0\longrightarrow \prod_{p\in S}\ZZ_p\longrightarrow \widehat{G_S}\longrightarrow \TT\longrightarrow 0.
\]
若一种审计方案只允许访问
\begin{enumerate}
  \item 固定的有限 prime-stage \(\widehat{G_S}\)；
  \item 某个固定精度上界 \(K\) 的模核大小数据
  \[
  \bigl\{K_p(k;A)\bigr\}_{p\in S,\ 1\le k\le K},
  \]
\end{enumerate}
则它不可能唯一恢复满行秩整数矩阵的完整 Smith 不变量；等价地，它也不可能唯一恢复全局线性亏损谱。更强地说，每一个这样的有限视界审计类都包含无穷多个两两不同的 Smith 不变量系。
\end{corollary}
\begin{proof}
由定理 \ref{thm:killo-localization-circle-dimension-prime-ledger-splitting}，固定有限阶段 \(\widehat{G_S}\) 的 prime-register 核恰为
\[
\prod_{p\in S}\ZZ_p,
\]
并且该阶段所能恢复的素数支撑恰为 \(S\) 本身。于是，在这一固定阶段内，所有模核读出都只能来自同一有限素数集 \(S\)。

但定理 \ref{thm:conclusion-truncated-smith-ledger-infinite-indistinguishability} 已经说明：对任意固定的 \(S\) 与 \(K\)，截断账本
\(
\{K_p(k;A)\}_{p\in S,1\le k\le K}
\)
仍然保留无穷多个两两不同的 Smith 不变量系。故任何仅停留在固定有限 prime-stage 且只读取有限精度模核账本的审计，都不可能唯一恢复完整 Smith 谱。证毕。
\end{proof}

\begin{theorem}[固定温度一位热场与有限 prime-stage completion 歧义的正交非忠实性]\label{thm:conclusion-joint-thermal-smith-nonfaithfulness}
固定有限温度集 \(Q\subset[0,\infty)\)、有限素数集 \(S\subset\PP\) 与精度上界 \(K\in\NN\)。定义联合可见证书
\[
\mathcal O_{Q,S,K}(m,A)
:=
\Bigl(
\{D_{\mathrm{KL}}(\pi_{m,q}\Vert \pi_{m,0})\}_{q\in Q},
\ \mathcal L_{S,K}(A)
\Bigr),
\]
其中 \(\pi_{m,q}\) 取自定理 \ref{thm:conclusion-fold-onebit-information-geometry-collapse}，\(\mathcal L_{S,K}(A)\) 取自定理 \ref{thm:conclusion-truncated-smith-ledger-infinite-indistinguishability}。则有：
\begin{enumerate}
  \item 对任意固定 \(q\in Q\)，热侧坐标都只在 \(O(1)\) 层收敛到末位 Bernoulli 影子给出的极限常数；
  \item 对任意一个可实现的账本值 \(\mathcal L_{S,K}\)，都存在无穷多个两两不同的 Smith 不变量系与之对应；
  \item 因而 \(\mathcal O_{Q,S,K}\) 不可能成为忠实证书；更强地说，在保持全部热侧坐标不变的同时，其算术侧仍保留无穷多个 completion 纤维。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fold-onebit-information-geometry-collapse}，对每个固定 \(q\in Q\)，散度
\[
D_{\mathrm{KL}}(\pi_{m,q}\Vert \pi_{m,0})
\]
都有有限极限，且该极限完全由末位两点 Bernoulli 影子 \(\beta_q\) 决定；故热侧在指数主阶上只保留一位统计影子。

另一方面，定理 \ref{thm:conclusion-truncated-smith-ledger-infinite-indistinguishability} 已经说明：对任意固定的 \(S\) 与 \(K\)，每个可实现的截断账本值都对应无穷多个两两不同的 Smith 不变量系。

现固定任意一个分辨率 \(m\)，并固定 bin-fold 热侧数据
\[
\{D_{\mathrm{KL}}(\pi_{m,q}\Vert \pi_{m,0})\}_{q\in Q}.
\]
再取一族满行秩整数矩阵 \(\{A^{(r)}\}_{r\ge 1}\)，使它们具有相同的 \(\mathcal L_{S,K}\) 而 Smith 不变量系两两不同。于是对所有 \(r\ge 1\) 都有
\[
\mathcal O_{Q,S,K}(m,A^{(r)})
=
\mathcal O_{Q,S,K}(m,A^{(1)}),
\]
但对应的完整 Smith completion 彼此不同。故该联合可见证书不可能忠实。证毕。
\end{proof}

\begin{conjecture}[Smith--solenoid 共尾正规化猜想]\label{conj:conclusion-smith-solenoid-cofinal-normalization}
任一试图通过“模核大小--prime-register--solenoidal completion”来闭合 classical RH 的正规化方案，若其可见证书在每一阶段仍只允许
\[
\text{有限 prime-support }S,
\qquad
\text{有限 exponent-depth }K,
\qquad
\text{以及固定阶 divergence/escort 族},
\]
则该正规化必同时落入两类终端塌缩：一方面，由定理 \ref{thm:conclusion-fold-onebit-information-geometry-collapse}，固定温度信息几何在指数主阶上退化为末位一位 Bernoulli 影子；另一方面，由定理 \ref{thm:conclusion-truncated-smith-ledger-infinite-indistinguishability} 与推论 \ref{cor:conclusion-finite-prime-stage-solenoid-smith-incompleteness}，有限阶段的 Smith--solenoid 账本始终保留无穷多 completion 歧义。因而，任何真正完备的此类正规化都必须在 \((p,k)\) 双坐标上共尾化；等价地，prime-support 与 exponent-depth 不能被任何固定视界同时截断。
\end{conjecture}

\begin{conjecture}[faithful \texorpdfstring{$\pi$}{pi} 公式的 entropy--bit--solenoid 三寄存器下界]\label{conj:conclusion-pi-formula-entropy-bit-solenoid-three-register}
任一若要 faithful 地承载 projective \(\pi\)-通道的公式、证书或正规化方案，都不能只停留于单一标量层，而至少须同时外置三类彼此正交的寄存器
\[
\mathfrak R_{\mathrm{const}},
\qquad
\mathfrak R_{\mathrm{bit}},
\qquad
\mathfrak R_{\mathrm{sol}}.
\]
其中
\begin{enumerate}
  \item \(\mathfrak R_{\mathrm{const}}\) 负责记录输出熵的 \(O(1)\) 常数项及其在规范体积上的 \(2^m\)-倍放大；
  \item \(\mathfrak R_{\mathrm{bit}}\) 负责记录末位两点热场、固定温度信息几何的一位塌缩、物理温度 \(q=1\) 的唯一临界选择，以及热轴与位预算轴的最小角点；
  \item \(\mathfrak R_{\mathrm{sol}}\) 负责记录随 prime-support 与 exponent-depth 共尾外推的 Smith--solenoid completion 账本。
\end{enumerate}
若缺失 \(\mathfrak R_{\mathrm{const}}\)，则无法区分由熵常数漂移诱导的 \(e^{\Theta(2^m)}\) 级规范体积裂口；若缺失 \(\mathfrak R_{\mathrm{bit}}\)，则无法命中唯一热临界点及其对应的最小反演角点；若缺失 \(\mathfrak R_{\mathrm{sol}}\)，则任何固定 prime-stage 审计都仍保留无穷 Smith completion 歧义。因而，这三条轴在 faithful \(\pi\)-公式中应当是不可互相压缩的。
\end{conjecture}

\noindent
因此，这条主线呈现出三重相互咬合的刚性：输出熵在一级仿射意义上唯一决定规范体积；固定温度信息几何在指数主阶上只剩末位一位；有限 prime-stage 的 Smith 账本则始终保留无穷 completion 歧义。若 classical RH 的某条 completion 路线最终必须经过这一接口，那么真正待突破的就不再是一个笼统的“无限维困难”，而是这三重刚性的叠加。
