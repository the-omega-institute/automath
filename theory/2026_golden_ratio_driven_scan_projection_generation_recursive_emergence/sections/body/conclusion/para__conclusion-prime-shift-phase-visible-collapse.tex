\begin{theorem}[prime-shift 半直积账本的相位可见部分为二生成]\label{thm:conclusion-prime-shift-phase-visible-two-generator}
\leanverified{paper\_conclusion\_prime\_shift\_phase\_visible\_two\_generator}
沿用定理 \ref{thm:conclusion-godel-semidirect-law} 的半直积幺半群
\(
\mathcal H=\mathcal P\rtimes_{\Sigma}\NN
\)
与其乘法 \((r,k)\cdot(s,\ell)=(r+\Sigma^k s,k+\ell)\)。
定义总次数同态
\[
\Omega:\mathcal P\to\NN,\qquad \Omega(r):=\sum_{p\in\mathbb P}r(p),
\]
以及投影幺半群同态
\[
\pi:\mathcal H\to\NN^2,\qquad \pi(r,k):=(k,\Omega(r)).
\]
则对任意交换群 \(H\)（写作乘法），任意幺半群同态 \(f:\mathcal H\to H\) 都唯一分解为
\[
f=\overline f\circ \pi,
\]
其中 \(\overline f:\NN^2\to H\) 为幺半群同态。
等价地，有自然同构
\[
\mathrm{Hom}_{\mathrm{Mon}}(\mathcal H,H)\ \cong\ \mathrm{Hom}_{\mathrm{Mon}}(\NN^2,H)\ \cong\ H^2,
\]
并且每个 \(f\) 由一对参数 \((a,\zeta)\in H^2\) 唯一确定，显式为
\[
f_{a,\zeta}(r,k)=a^k\,\zeta^{\Omega(r)}.
\]
\end{theorem}
\begin{proof}
记 \(a:=f(0,1)\in H\)（其中 \(0\) 表示 \(\mathcal P\) 的零元）。
对每个 \(t\ge 1\) 令 \(e_t\in\mathcal P\) 为单点基向量：\(e_t(p_t)=1\)，其余为 \(0\)，并记 \(\zeta_t:=f(e_t,0)\)。
由半直积乘法，
\[
(0,1)\cdot(e_t,0)=(\Sigma^1 e_t,1)=(e_{t+1},1),
\qquad
(e_{t+1},0)\cdot(0,1)=(e_{t+1},1).
\]
对两式施行 \(f\) 并用同态性得
\(
f(e_{t+1},1)=a\,\zeta_t=\zeta_{t+1}\,a
\)。
由于 \(H\) 交换，得 \(\zeta_{t+1}=\zeta_t\)，从而 \(\zeta_t\) 与 \(t\) 无关；记其公共值为 \(\zeta\)。
\par
任取 \(r\in\mathcal P\)（有限支撑），可写 \(r=\sum_{t\ge 1}r(p_t)e_t\)，而 \(\mathcal P\) 为交换幺半群，故
\[
f(r,0)=\prod_{t\ge 1}f(e_t,0)^{\,r(p_t)}=\prod_{t\ge 1}\zeta^{\,r(p_t)}=\zeta^{\Omega(r)}.
\]
又对任意 \((r,k)\in\mathcal H\)，有 \((r,k)=(r,0)\cdot(0,1)^k\)，故
\[
f(r,k)=f(r,0)\,f(0,1)^k=\zeta^{\Omega(r)}a^k,
\]
仅依赖 \((k,\Omega(r))\)，即 \(f=\overline f\circ\pi\)。
唯一性由 \(\pi\) 的像生成 \(\NN^2\) 且 \(\overline f\) 在生成元上的值由 \((a,\zeta)\) 决定立即得到。证毕。
\end{proof}

\begin{corollary}[相位字符空间与圆维解释]\label{cor:conclusion-prime-shift-phase-character-torus-cdim}
对任意 \(k\ge 1\)，有自然同构
\[
\mathrm{Hom}_{\mathrm{Mon}}(\mathcal H,\TT^k)\ \cong\ (\TT^k)^2\ \cong\ \TT^{2k}.
\]
从而 \(\mathcal H\) 的单位圆相位可见部分由两组独立的 \(\TT^k\)-参数控制。
若只记录相位可见的交换商 \(\NN^2\)，则其群完成满足 \(G(\NN^2)\cong\ZZ^2\)，并由定义 \ref{def:half-circle-dimension} 得
\(
\cdim_+(\NN^2)=\tfrac12\cdim(\ZZ^2)=1
\)。
\end{corollary}
\begin{proof}
取定理 \ref{thm:conclusion-prime-shift-phase-visible-two-generator} 中 \(H=\TT^k\) 即得第一式；
其余为定义 \ref{def:half-circle-dimension} 的直接代入。证毕。
\end{proof}

\begin{proposition}[相位不可辨识关系的完全刻画]\label{prop:conclusion-prime-shift-phase-indistinguishability}
对任意 \((r,k),(s,\ell)\in\mathcal H\)，下列命题等价：
\begin{enumerate}
  \item 对任意 \(q\ge 1\) 与任意幺半群同态 \(\Phi:\mathcal H\to\TT^q\)，恒有 \(\Phi(r,k)=\Phi(s,\ell)\)；
  \item \(k=\ell\) 且 \(\Omega(r)=\Omega(s)\)。
\end{enumerate}
换言之，\(\mathcal H\) 在全体单位圆相位读出下的分离极限恰由 \((k,\Omega)\) 给出。
\end{proposition}
\begin{proof}
\((2)\Rightarrow(1)\) 由推论 \ref{cor:conclusion-prime-shift-phase-character-torus-cdim}：任意 \(\Phi\) 仅依赖 \((k,\Omega)\)。
\par
\((1)\Rightarrow(2)\)：
若 \(k\neq \ell\)，取 \(q=1\)，选 \(a\in\TT\) 使 \(a^{k-\ell}\neq 1\)，并令 \(\Phi(r',k'):=a^{k'}\)（对应 \(\zeta=1\)），则 \(\Phi(r,k)\neq \Phi(s,\ell)\)，矛盾。
若 \(k=\ell\) 但 \(\Omega(r)\neq \Omega(s)\)，取 \(q=1\)，选 \(\zeta\in\TT\) 使 \(\zeta^{\Omega(r)-\Omega(s)}\neq 1\)，并令 \(\Phi(r',k'):=\zeta^{\Omega(r')}\)（对应 \(a=1\)），同得矛盾。
故必有 \(k=\ell\) 且 \(\Omega(r)=\Omega(s)\)。证毕。
\end{proof}

\begin{corollary}[Gödel 化在相位读出下的坍缩律]\label{cor:conclusion-godel-phase-collapse-length-weight}
设 \(\mathcal E\) 为字母表并沿用定理 \ref{thm:conclusion-godel-semidirect-law} 的嵌入 \(h:\mathcal E^\ast\hookrightarrow\mathcal H\)。
对词 \(w=e_1\cdots e_T\) 定义总码重
\[
W(w):=\sum_{t=1}^{T}\mathrm{code}(e_t)=\Omega(r_w).
\]
则对任意 \(q\ge 1\) 与任意幺半群同态 \(\Phi:\mathcal H\to\TT^q\)，复合 \(\Phi\circ h\) 仅依赖二元组 \((|w|,W(w))\)：
\[
|w|=|w'|\ \text{且}\ W(w)=W(w')
\quad\Longrightarrow\quad
(\Phi\circ h)(w)=(\Phi\circ h)(w').
\]
\end{corollary}
\begin{proof}
由定义 \(h(w)=(r_w,|w|)\) 且 \(\Omega(r_w)=W(w)\)，再由定理 \ref{thm:conclusion-prime-shift-phase-visible-two-generator} 得 \(\Phi\) 仅依赖 \((k,\Omega)\)，代入即得。证毕。
\end{proof}

\begin{theorem}[轨道生成对称下的有限相位原则]\label{thm:conclusion-orbit-generator-finite-phase-principle}
\leanverified{paper\_conclusion\_orbit\_generator\_finite\_phase\_principle}
令 \(I\) 为可数集合，\(\mathcal P_I:=\NN^{(I)}\) 为由 \(\{\mathbf e_i:i\in I\}\) 生成的自由交换幺半群。
设给定一族端同态 \(\Sigma_I^n:\mathcal P_I\to\mathcal P_I\) 来自对指标集的作用 \(i\mapsto \sigma_n(i)\)，并假设存在 \(i_0\in I\) 使得 \(\{\sigma_n(i_0):n\in\NN\}=I\)。
定义半直积幺半群
\[
\mathcal H_I:=\mathcal P_I\rtimes_{\Sigma_I}\NN,
\qquad
(r,k)\cdot(s,\ell):=(r+\Sigma_I^k s,k+\ell),
\]
以及总次数 \(\Omega_I(r):=\sum_{i\in I}r(i)\) 与投影 \(\pi_I(r,k):=(k,\Omega_I(r))\)。
则对任意交换群 \(H\)，任意幺半群同态 \(f:\mathcal H_I\to H\) 都唯一分解为 \(f=\overline f\circ \pi_I\)，并满足
\[
f(r,k)=a^k\,\zeta^{\Omega_I(r)}
\]
对某个唯一的 \((a,\zeta)\in H^2\)。
特别地，
\(
\mathrm{Hom}_{\mathrm{Mon}}(\mathcal H_I,\TT)\cong \TT^2
\)，与 \(|I|\) 无关。
\end{theorem}
\begin{proof}
令 \(a:=f(0,1)\) 且对 \(i\in I\) 置 \(\zeta_i:=f(\mathbf e_i,0)\)。
对任意 \(n\) 有
\(
(0,1)\cdot(\mathbf e_i,0)=(\Sigma_I^1\mathbf e_i,1)=(\mathbf e_{\sigma_1(i)},1)
\)
与
\(
(\mathbf e_{\sigma_1(i)},0)\cdot(0,1)=(\mathbf e_{\sigma_1(i)},1)
\)。
同态性给出
\(
a\zeta_i=\zeta_{\sigma_1(i)}a
\)，由交换性得 \(\zeta_{\sigma_1(i)}=\zeta_i\)。
沿轨道生成假设，任意 \(j\in I\) 皆可写为 \(j=\sigma_n(i_0)\)，故 \(\zeta_j=\zeta_{i_0}\) 对一切 \(j\) 成立；记公共值为 \(\zeta\)。
随后按自由交换性得 \(f(r,0)=\zeta^{\Omega_I(r)}\)，并由分解 \((r,k)=(r,0)\cdot(0,1)^k\) 得 \(f(r,k)=a^k\zeta^{\Omega_I(r)}\)。
其因子化与唯一性与定理 \ref{thm:conclusion-prime-shift-phase-visible-two-generator} 同理。证毕。
\end{proof}
