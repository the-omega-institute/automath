\subsection{Smith 局部 zeta 的 Hadamard 线性化、Newton 阶梯与最小 solenoidal 宿主}\label{subsec:conclusion-smith-hadamard-newton-solenoid-minimality}
沿用小节 \ref{subsec:conclusion-smith-bounded-profile-local-zeta}、
小节 \ref{subsec:conclusion-smith-concavity-localzeta-ellipse-semigroup} 与
小节 \ref{subsec:conclusion-phase-ledger-branch-trisplitting} 的记号。此前的 Smith 结论链已经分别给出：局部亏损轮廓的完全分类、Young 对偶与离散 Hessian 原子反演、局部 \(\mathfrak Z\)-因子的有理尾平台，以及有限素数局部化的唯一最小 solenoidal 宿主。把这些接口进一步并读后，还可把完整 Smith 谱的局部级数结构压缩为如下单链：每个坏素数方向上的亏损谱在直和下不是由通常乘法线性化，而是由 Hadamard 乘法线性化；其归一化分子系数的 \(p\)-进阶梯恰恢复整条 \(p\)-主 Young 边界；坏素集合则给出唯一最小 prime-support 与 solenoidal 宿主；而任何固定 prime-stage 或固定 jet-depth 都必留下不可消去的无限盲纤维。

\begin{theorem}[Smith 局部 zeta 的 Hadamard 线性化与原子自由分解]\label{thm:conclusion-smith-local-zeta-hadamard-linearization}
\leanverified{paper\_conclusion\_smith\_local\_zeta\_hadamard\_linearization}
设 \(A,B\) 为满行秩整数矩阵。对每个素数 \(p\)，记
\[
f_{A,p}(k):=\sum_i \min\bigl(k,e_i^{(A)}(p)\bigr),
\qquad
\mathfrak Z_{A,p}(X):=\sum_{k\ge 0}p^{f_{A,p}(k)}X^k,
\]
并定义 Hadamard 乘积
\[
\Bigl(\sum_{k\ge 0}a_kX^k\Bigr)\odot
\Bigl(\sum_{k\ge 0}b_kX^k\Bigr)
:=
\sum_{k\ge 0}a_kb_kX^k.
\]
则对任意素数 \(p\) 都有
\[
\mathfrak Z_{A\oplus B,p}
=
\mathfrak Z_{A,p}\odot \mathfrak Z_{B,p}.
\]
更进一步，若对每个 \(e\ge 1\) 定义 \(p\)-原子级数
\[
\mathfrak Z_{p,e}(X):=\sum_{k\ge 0}p^{\min(k,e)}X^k,
\]
并记 Hadamard 单位元
\[
\mathbf 1_{\odot}(X):=\sum_{k\ge 0}X^k,
\]
则
\[
\mathfrak Z_{A,p}
=
\bigodot_{i:\ e_i^{(A)}(p)>0} \mathfrak Z_{p,e_i^{(A)}(p)}.
\]
若 \(e_i^{(A)}(p)=0\) 对所有 \(i\) 都成立，则上式按约定解释为 \(\mathfrak Z_{A,p}=\mathbf 1_{\odot}(X)\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-kernel-loss-divisibility-valuation-rational-euler}，
\[
f_{A,p}(k)=\sum_i \min\bigl(k,e_i^{(A)}(p)\bigr),
\qquad
f_{B,p}(k)=\sum_j \min\bigl(k,e_j^{(B)}(p)\bigr).
\]
块直和 \(A\oplus B\) 的 Smith 估值多重集正是两组多重集的并，因此
\[
f_{A\oplus B,p}(k)=f_{A,p}(k)+f_{B,p}(k)
\qquad(k\ge 0).
\]
于是
\[
\mathfrak Z_{A\oplus B,p}(X)
=
\sum_{k\ge 0}p^{f_{A,p}(k)+f_{B,p}(k)}X^k
=
\sum_{k\ge 0}p^{f_{A,p}(k)}p^{f_{B,p}(k)}X^k
=
\mathfrak Z_{A,p}\odot \mathfrak Z_{B,p}.
\]
第二式则由
\[
f_{A,p}(k)=\sum_i \min\bigl(k,e_i^{(A)}(p)\bigr)
\]
逐项指数化后直接得到。证毕。
\end{proof}

\begin{corollary}[局部 Hadamard 因子的直和可消去性与原子唯一分解]\label{cor:conclusion-smith-local-zeta-directsum-cancellation}
\leanverified{paper\_conclusion\_smith\_local\_zeta\_directsum\_cancellation}
在固定行秩口径下，设 \(A,B,C\) 为具有相同行数的满行秩整数矩阵。若对每个素数 \(p\) 都有
\[
\mathfrak Z_{A\oplus C,p}=\mathfrak Z_{B\oplus C,p},
\]
则 \(A\) 与 \(B\) 具有完全相同的 Smith 不变量。等价地，在每个固定素数 \(p\) 上，\(\mathfrak Z_{A,p}\) 在去除 Hadamard 单位元 \(\mathbf 1_{\odot}(X)\) 后，具有唯一的正高度原子分解，且不存在非平凡的消去失败。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-smith-local-zeta-hadamard-linearization}，
\[
\mathfrak Z_{A\oplus C,p}
=
\mathfrak Z_{A,p}\odot \mathfrak Z_{C,p},
\qquad
\mathfrak Z_{B\oplus C,p}
=
\mathfrak Z_{B,p}\odot \mathfrak Z_{C,p}.
\]
比较 \(X^k\) 的系数可得
\[
p^{f_{A,p}(k)}p^{f_{C,p}(k)}
=
p^{f_{B,p}(k)}p^{f_{C,p}(k)}
\qquad(k\ge 0),
\]
故
\[
f_{A,p}(k)=f_{B,p}(k)
\qquad(\forall p,\ \forall k\ge 0).
\]
再由推论 \ref{cor:conclusion-smith-local-zeta-complete-invariant}，在行秩背景固定时，全部局部轮廓唯一恢复完整 Smith 谱，于是 \(A\) 与 \(B\) 的 Smith 不变量完全相同。固定素数 \(p\) 的原子唯一分解正是同一恢复定理在局部 Hadamard 口径下的重述。证毕。
\end{proof}

\begin{theorem}[归一化局部分子的 \(p\)-进 Newton 阶梯与 Young 共轭边界]\label{thm:conclusion-smith-normalized-local-numerator-newton-staircase}
设 \(A\) 为满行秩整数矩阵，并固定素数 \(p\)。记
\[
E_p^{\max}:=\max_i e_i(p),
\qquad
E_p^{\Sigma}:=\sum_i e_i(p),
\qquad
\Delta_p(k):=f_{A,p}(k)-f_{A,p}(k-1).
\]
定义归一化局部分子
\[
Q_{A,p}(X):=(1-X)\mathfrak Z_{A,p}(X).
\]
则 \(Q_{A,p}(X)\) 是次数恰为 \(E_p^{\max}\) 的多项式，并且唯一可写成
\[
Q_{A,p}(X)=1+\sum_{k=1}^{E_p^{\max}}q_{p,k}X^k,
\]
其中
\[
q_{p,k}=p^{f_{A,p}(k-1)}\bigl(p^{\Delta_p(k)}-1\bigr).
\]
进一步成立以下四条恢复律：
\[
v_p(q_{p,k})=f_{A,p}(k-1)
\qquad(1\le k\le E_p^{\max}),
\]
\[
1+\sum_{j=1}^{k}q_{p,j}=p^{f_{A,p}(k)}
\qquad(1\le k\le E_p^{\max}),
\]
\[
\Delta_p(k)=
\log_p\!\left(
\frac{1+\sum_{j=1}^{k}q_{p,j}}
{1+\sum_{j=1}^{k-1}q_{p,j}}
\right)
\qquad(1\le k\le E_p^{\max}),
\]
\[
E_p^{\Sigma}=\log_p Q_{A,p}(1),
\qquad
E_p^{\max}=\deg Q_{A,p}.
\]
特别地，\(Q_{A,p}(X)\) 单独即可恢复整个 \(p\)-主 Young 图。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-smith-local-zeta-complete-invariant}，
\[
\mathfrak Z_{A,p}(X)=\sum_{k\ge 0}p^{f_{A,p}(k)}X^k.
\]
因此
\[
(1-X)\mathfrak Z_{A,p}(X)
=
1+\sum_{k\ge 1}\bigl(p^{f_{A,p}(k)}-p^{f_{A,p}(k-1)}\bigr)X^k.
\]
又由定理 \ref{thm:conclusion-smith-kernel-dirichlet-euler-rational-local-factors}，当 \(k\ge E_p^{\max}\) 时有
\[
f_{A,p}(k)=E_p^{\Sigma},
\]
故当 \(k>E_p^{\max}\) 时对应系数为零，而当 \(k=E_p^{\max}\) 时有 \(\Delta_p(k)\ge 1\)，故最高次项非零，因而 \(\deg Q_{A,p}=E_p^{\max}\)。对 \(1\le k\le E_p^{\max}\)，
\[
q_{p,k}
=
p^{f_{A,p}(k)}-p^{f_{A,p}(k-1)}
=
p^{f_{A,p}(k-1)}\bigl(p^{\Delta_p(k)}-1\bigr).
\]
由于 \(\Delta_p(k)\ge 1\) 时 \(p\nmid (p^{\Delta_p(k)}-1)\)，遂得
\[
v_p(q_{p,k})=f_{A,p}(k-1).
\]
再对系数作望远镜求和，
\[
1+\sum_{j=1}^{k}q_{p,j}
=
1+\sum_{j=1}^{k}\bigl(p^{f_{A,p}(j)}-p^{f_{A,p}(j-1)}\bigr)
=
p^{f_{A,p}(k)},
\]
从而第三式立即成立。取 \(k=E_p^{\max}\) 得
\[
Q_{A,p}(1)=p^{f_{A,p}(E_p^{\max})}=p^{E_p^{\Sigma}}.
\]
最后，由推论 \ref{cor:conclusion-smith-loss-profile-young-duality}，
\[
e_j^\downarrow(p)=\#\{k\ge 1:\ \Delta_p(k)\ge j\},
\]
故 \(Q_{A,p}(X)\) 经由 \((\Delta_p(k))\) 即可恢复整条 \(p\)-主 Young 边界。证毕。
\end{proof}

\begin{theorem}[坏素集合决定的最小 solenoidal 宿主]\label{thm:conclusion-smith-minimal-solenoidal-host-by-bad-primes}
设 \(A\in M_{m\times n}(\ZZ)\) 为满行秩整数矩阵，并定义其坏素集合
\[
S_A:=\{p\in\mathbb P:\exists i,\ e_i(p)>0\}.
\]
则在固定行秩 \(m\) 的口径下，\(S_A\) 是唯一最小的有限素数集，使得局部级数族
\[
\{\mathfrak Z_{A,p}(X)\}_{p\in S_A}
\]
已经足以恢复 \(A\) 的全部 Smith 不变量。等价地，若记
\[
G_{S_A}:=\ZZ[S_A^{-1}],
\]
则其 Pontryagin 对偶短正合列
\[
0\longrightarrow \prod_{p\in S_A}\ZZ_p
\longrightarrow \widehat{G_{S_A}}
\longrightarrow \TT
\longrightarrow 0
\]
给出了完整 Smith 谱的唯一最小 solenoidal 宿主。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-smith-local-zeta-complete-invariant}，在固定行秩 \(m\) 的背景下，坏素数上的全部局部级数已足够恢复全部 \(p\)-主 Smith 估值多重集；而当 \(p\notin S_A\) 时，所有 \(e_i(p)=0\)，故对应局部级数是平凡因子。因此 \(\{\mathfrak Z_{A,p}\}_{p\in S_A}\) 已经充分。

为证最小性，取任意真子集 \(T\subsetneq S_A\)，并选 \(p_0\in S_A\setminus T\)。
记 \(A\) 的 Smith 不变量为 \(d_1\mid\cdots\mid d_m\)，并令
\[
b_i:=d_i\,p_0^{-e_i(p_0)}
\qquad(1\le i\le m).
\]
则 \((b_i)_{i=1}^{m}\) 仍满足整除链
\[
b_1\mid\cdots\mid b_m,
\]
故对角矩阵
\[
B:=\operatorname{diag}(b_1,\dots,b_m)
\]
是一个满行秩整数矩阵，并满足
\[
f_{B,q}=f_{A,q}\qquad(q\neq p_0),
\qquad
f_{B,p_0}\equiv 0.
\]
于是
\[
\mathfrak Z_{A,q}(X)=\mathfrak Z_{B,q}(X)
\qquad(q\in T),
\]
但 \(A\) 与 \(B\) 的 Smith 谱不同。因此任何完整分类器都必须包含全部 \(S_A\)。

唯一性则由定理 \ref{thm:conclusion-finite-prime-localization-pontryagin-rigidity} 给出：有限素数集 \(S\) 可由对偶核
\[
\prod_{p\in S}\ZZ_p
\]
的 \(p\)-主因子支撑唯一恢复，而定理 \ref{thm:conclusion-finite-prime-solenoid-terminal-object} 又给出对应的唯一单圆连续商 \(\TT\)。故与 \(S_A\) 对应的 solenoidal 宿主既最小又唯一。证毕。
\end{proof}

\begin{theorem}[固定 prime-stage 上不可见原子的自由作用]\label{thm:conclusion-smith-offstage-atom-free-action}
固定有限素数集 \(T\subset\mathbb P\)。对每个 \(\ell\notin T\) 与 \(e\ge 1\)，定义离台原子轮廓
\[
a_{\ell,e}
:=
\bigl(a_{\ell,e;p}\bigr)_{p\in\mathbb P},
\qquad
a_{\ell,e;p}(k):=
\begin{cases}
\min(k,e),&p=\ell,\\
0,&p\neq \ell.
\end{cases}
\]
令 \(\mathfrak M_{T^c}\) 为由全部 \(a_{\ell,e}\) 自由生成的交换幺半群，并记
\[
\mathbf f_A:=\bigl(f_{A,p}\bigr)_{p\in\mathbb P},
\qquad
\tau_T(\mathbf f_A):=\bigl(f_{A,p}\bigr)_{p\in T}.
\]
则 \(\mathfrak M_{T^c}\) 在全体满行秩整数矩阵的 Smith 亏损谱族上按
\[
\mathbf f_A\longmapsto \mathbf f_A+u
\qquad(u\in \mathfrak M_{T^c})
\]
作用，并满足：
\begin{enumerate}
  \item 该作用保持固定 prime-stage 限制：
  \[
  \tau_T(\mathbf f_A+u)=\tau_T(\mathbf f_A);
  \]
  \item 该作用是自由的；
  \item 每个 \(\tau_T\)-纤维都包含可数无穷多个两两不同的 Smith 谱。
\end{enumerate}
\end{theorem}
\begin{proof}
对任意 \(\ell\notin T\)，原子 \(a_{\ell,e}\) 仅改动素数 \(\ell\) 的亏损轮廓，而对每个 \(p\in T\) 都给出零增量，故
\[
\tau_T(\mathbf f_A+u)=\tau_T(\mathbf f_A)
\qquad(u\in \mathfrak M_{T^c}).
\]
这证明了第一条。

若
\[
\mathbf f_A+u=\mathbf f_A+v,
\]
则逐素数比较可得：对每个 \(\ell\notin T\)，两边在 \(\ell\)-方向上的局部亏损轮廓相同。再由定理 \ref{thm:conclusion-smith-local-profile-young-hessian-unification}，该局部轮廓的离散 Hessian 唯一恢复每个高度 \(e\) 的原子重数，故 \(u\) 与 \(v\) 在所有 \((\ell,e)\)-生成元上的重数完全一致，从而 \(u=v\)。因此作用是自由的。

最后，\(\mathfrak M_{T^c}\) 本身是可数无穷的自由交换幺半群，故从任一 \(\mathbf f_A\) 出发可得到可数无穷多个两两不同的谱族。并且每个原子 \(a_{\ell,e}\) 都可由直和一个 \(1\times 1\) 对角块 \([\ell^e]\) 显式实现，因此任意 \(u\in\mathfrak M_{T^c}\) 都对应某个由有限个此类对角块组成的块直和矩阵。于是每个 \(\tau_T\)-纤维都包含可数无穷多个两两不同、且可由满行秩整数矩阵实现的 Smith 谱。证毕。
\end{proof}

\begin{proposition}[固定坏素上的有限 jet 只看见 Young 图的截头]\label{prop:conclusion-smith-finite-jet-sees-only-truncated-young-diagram}
\leanverified{paper\_conclusion\_smith\_finite\_jet\_sees\_only\_truncated\_young\_diagram}
固定素数 \(p\) 与整数 \(K\ge 1\)。对任意满行秩整数矩阵 \(A\)，定义有限 jet
\[
J_{p,K}(A):=\bigl(f_{A,p}(0),f_{A,p}(1),\dots,f_{A,p}(K)\bigr).
\]
若 \(A\) 的 \(p\)-主估值降序列中存在某项满足
\[
e_j^\downarrow(p)>K,
\]
则对每个 \(r\ge 0\)，把该项替换为 \(e_j^\downarrow(p)+r\) 所得的新估值序列都对应某个满行秩整数矩阵 \(A^{(r)}\)，并满足
\[
J_{p,K}(A^{(r)})=J_{p,K}(A).
\]
因此，有限 jet \(J_{p,K}\) 只能识别 \(p\)-主 Young 图在高度 \(K\) 以下的截头，而对其上方尾部完全失明。
\end{proposition}
\begin{proof}
对任意 \(0\le k\le K\)，由 \(e_j^\downarrow(p)>K\) 可得
\[
\min\bigl(k,e_j^\downarrow(p)+r\bigr)=k=\min\bigl(k,e_j^\downarrow(p)\bigr).
\]
其余估值保持不变，故
\[
f_{A^{(r)},p}(k)=f_{A,p}(k)
\qquad(0\le k\le K).
\]
于是
\[
J_{p,K}(A^{(r)})=J_{p,K}(A).
\]
另一方面，不同的 \(r\) 给出不同的 \(p\)-主估值多重集，因此对应 Smith 谱彼此不同；而这些估值多重集都由定理 \ref{thm:conclusion-smith-loss-profile-family-classification} 可实现为满行秩整数矩阵。证毕。
\end{proof}

\begin{theorem}[有限 prime-stage 与有限 jet 的联合证书纤维仍含可数无穷 completion]\label{thm:conclusion-smith-finite-stage-finite-jet-joint-fiber-countably-infinite}
固定有限素数集 \(T\subset\PP\) 与整数 \(K\ge 1\)。设
\[
\mathcal O_{T,K}
\]
为任意一个只依赖于固定 prime-stage 数据
\[
\tau_T(\mathbf f_A)=\bigl(f_{A,p}\bigr)_{p\in T}
\]
与有限 jet 数据
\[
\bigl(J_{p,K}(A)\bigr)_{p\in T}
\]
的 Smith 观测器。则对每个非空观测纤维
\[
\mathcal O_{T,K}^{-1}(y)
\]
都存在可数无穷多个两两不同的 Smith completion。等价地，任何有限 prime-stage 加有限 jet 的联合证书，都不可能构成 Smith 结构的完备不变量。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-smith-offstage-atom-free-action}，对任意固定的 prime-stage 数据
\[
\tau_T(\mathbf f_A),
\]
其纤维内都存在可数无穷多个两两不同的 Smith 谱，它们由离台原子幺半群 \(\mathfrak M_{T^c}\) 的自由作用产生。

另一方面，这一自由作用只改动 \(T^c\) 上的局部亏损轮廓，而对每个 \(p\in T\) 都保持
\[
f_{A,p}(k)
\qquad (k\ge 0)
\]
完全不变。因而对每个 \(p\in T\) 与每个 \(K\ge 1\)，有限 jet
\[
J_{p,K}(A)=\bigl(f_{A,p}(0),f_{A,p}(1),\dots,f_{A,p}(K)\bigr)
\]
也保持不变。

于是，任何只经由
\[
\tau_T(\mathbf f_A)
\quad\text{与}\quad
\bigl(J_{p,K}(A)\bigr)_{p\in T}
\]
因子化的观测器 \(\mathcal O_{T,K}\)，在上述整条可数无穷自由轨道上都取同一数值。故其每个非空观测纤维都至少包含可数无穷多个两两不同的 Smith completion。证毕。
\end{proof}

\begin{conjecture}[Smith--solenoid 初对象]\label{conj:conclusion-smith-solenoid-initial-object}
对任意满行秩整数矩阵 \(A\)，记
\[
\mathfrak S(A):=
\left(
\widehat{G_{S_A}},
\ \{Q_{A,p}(X)\}_{p\in S_A}
\right).
\]
则 \(\mathfrak S(A)\) 应当构成“固定行秩背景下的完整 Smith 谱分类器”范畴中的初对象。更精确地，任何同时满足下列三条的分类器：
\begin{enumerate}
  \item 对块直和函子化；
  \item 能唯一恢复全部 Smith 不变量；
  \item 其素数支撑可由宿主核的 \(p\)-主因子读出；
\end{enumerate}
都应唯一因子化经过 \(\mathfrak S(A)\)。
\end{conjecture}

\noindent
综上，完整 Smith 谱在结论层面已不再表现为一串静态整数，而裂解为“宿主 + 内容”两层：宿主由坏素集合 \(S_A\) 唯一决定，其最小连续--全不连通承载正是 \(\widehat{\ZZ[S_A^{-1}]}\)；内容则由每个坏素数方向上的 \(p\)-进 Young 阶梯承载，而这一阶梯既可通过 Hadamard 线性化的局部级数 \(\mathfrak Z_{A,p}\) 读出，也可通过归一化分子 \(Q_{A,p}\) 的 Newton 阶梯完全恢复。与之相对，任何固定 prime-stage 或固定 jet-depth 的有限审计，都只能看到截断后的局部影子，并不可避免地留下可数无穷的盲纤维。于是，Hadamard 原子化、Young 阶梯恢复、坏素最小宿主与有限视界不完备性，已被压缩为同一条闭合的 Smith--solenoid 结构链。
