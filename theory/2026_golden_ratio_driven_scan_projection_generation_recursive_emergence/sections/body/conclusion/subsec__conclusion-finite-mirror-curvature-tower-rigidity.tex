\subsection{有限长度镜像律、KL 曲率账本与 tower 缺陷的刚性闭合}\label{subsec:conclusion-finite-mirror-curvature-tower-rigidity}
沿用定理 \ref{thm:sync-kernel-endpoint-refined-palindrome}、定理 \ref{thm:sync-kernel-gallavotti-cohen}、定理 \ref{thm:sync-kernel-even-edgeworth-c4}、命题 \ref{prop:sync-kernel-moderate-deviation-cauchy-certified}、命题 \ref{prop:einstein-legendre}、定理 \ref{thm:pom-bc-information-stokes-coboundary}、命题 \ref{prop:pom-bc-bianchi}、定理 \ref{thm:pom-tower-defect-covariance-law}、推论 \ref{cor:pom-bc-functoriality-criterion} 与推论 \ref{cor:pom-bc-zero-defect-rigidity-parenthesization} 的记号。前述链条已经分别把自对偶同步核的端点细化回文律、完成化偶性与中偏差偶次喷射，以及 POM 侧 Beck--Chevalley 曲率、离散 Bianchi 恒等式与 tower defect 的协方差表达固定为显式公式。把这些有限层恒等式并置，可进一步抽出如下四条刚性后果。

\begin{theorem}[自对偶同步核的有限长度镜像律与奇累积量湮灭]\label{thm:conclusion-selfdual-sync-kernel-finite-mirror-odd-cumulant}
设同步核在输出位势 \(e\in\{0,1\}\) 下的长度 \(n\) 配分多项式为
\[
Z_n(u):=\mathbf 1^\top B(u)^n\mathbf 1=\sum_{k=0}^n A_{n,k}u^k,
\]
其中 \(A_{n,k}\) 计数长度 \(n\) 路径中输出 \(1\) 的总数恰为 \(k\) 的可行路径。再令 \(S_n\) 为满足
\[
\mathbb P(S_n=k)=\frac{A_{n,k}}{Z_n(1)}
\qquad (0\le k\le n)
\]
的随机变量，并记 \(\kappa_r(S_n)\) 为其第 \(r\) 阶累积量。则对每个 \(n\ge 1\) 都有精确回文律
\[
A_{n,k}=A_{n,n-k}
\qquad (0\le k\le n),
\]
等价地，
\[
Z_n(u)=u^n Z_n(1/u).
\]
从而
\[
\mathbb E[S_n]=\frac{n}{2},
\qquad
\kappa_{2j+1}(S_n)=0
\quad (j\ge 1).
\]
换言之，除去均值项外，一切奇阶累积量在每一个有限长度上均严格消失。
\end{theorem}
\begin{proof}
由定理 \ref{thm:sync-kernel-endpoint-refined-palindrome}，
\[
N^{(n)}_{p,q}(k)=N^{(n)}_{\iota(p),\iota(q)}(n-k).
\]
对全部起终点求和，并利用 \(\iota\) 是状态集合上的置换，得
\[
A_{n,k}
=
\sum_{p,q}N^{(n)}_{p,q}(k)
=
\sum_{p,q}N^{(n)}_{\iota(p),\iota(q)}(n-k)
=
A_{n,n-k}.
\]
于是
\[
Z_n(u)
=
\sum_{k=0}^n A_{n,k}u^k
=
u^n\sum_{k=0}^n A_{n,n-k}u^{-k}
=
u^n Z_n(1/u).
\]

再定义累积母函数
\[
K_n(t):=\log\frac{Z_n(e^t)}{Z_n(1)}.
\]
由回文关系立得
\[
K_n(t)=nt+K_n(-t).
\]
因此
\[
\widetilde K_n(t):=K_n(t)-\frac{n}{2}t
\]
为偶函数，故其在 \(t=0\) 处的一切奇阶导数全部为零。又
\[
K_n'(0)=\mathbb E[S_n],
\qquad
K_n^{(r)}(0)=\kappa_r(S_n)\quad (r\ge 1),
\]
于是
\[
\mathbb E[S_n]=\frac{n}{2},
\qquad
\kappa_{2j+1}(S_n)=0\quad (j\ge 1).
\]
证毕。
\end{proof}

\begin{corollary}[Edgeworth 首阶项严格消失与中偏差级数的纯偶性]\label{cor:conclusion-selfdual-sync-kernel-edgeworth-evenness}
沿用定理 \ref{thm:conclusion-selfdual-sync-kernel-finite-mirror-odd-cumulant} 的记号，并定义
\[
Y_n:=\frac{S_n-\frac{n}{2}}{\sqrt{\operatorname{Var}(S_n)}}.
\]
则 \(Y_n\) 的一切奇阶标准化累积量严格为零。因而：
\begin{enumerate}
  \item \(Y_n\) 的 Edgeworth 展开中 \(n^{-1/2}\) 级首项恒等消失；
  \item 第一非高斯修正只能从 \(n^{-1}\) 级开始出现；
  \item 若 \(I\) 记同步核输出密度的率函数
  \[
  I(\alpha):=\sup_{\theta\in\RR}\bigl(\theta\alpha-P(\theta)+P(0)\bigr),
  \]
  则在中心点 \(\alpha=\tfrac12\) 的邻域内仅含偶次幂：
  \[
  I\!\left(\frac12+\delta\right)=\sum_{j\ge 1}c_{2j}\delta^{2j},
  \]
  且二次系数满足
  \[
  c_2=\frac{51}{11}.
  \]
\end{enumerate}
\leanverified{paper\_conclusion\_edgeworth\_evenness\_rate\_seeds}
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-selfdual-sync-kernel-finite-mirror-odd-cumulant} 已给出
\(
\kappa_{2j+1}(S_n)=0
\)
（\(j\ge 1\)）。标准化不会改变“奇阶为零”这一事实，故 \(Y_n\) 的全部奇阶标准化累积量仍严格为零。于是 Edgeworth 展开中的 \(n^{-1/2}\) 首项（其系数正比于标准化三阶累积量）必然消失，而首个非高斯修正只能落在偶 Hermite 扇区；这也与定理 \ref{thm:sync-kernel-even-edgeworth-c4} 中只出现 \(H_4\) 的显式公式完全一致。

另一方面，定理 \ref{thm:sync-kernel-gallavotti-cohen} 与命题 \ref{prop:sync-kernel-moderate-deviation-cauchy-certified} 已给出
\[
I(\alpha)=I(1-\alpha),
\]
故 \(I(\tfrac12+\delta)\) 作为 \(\delta\) 的函数必为偶函数，其 Taylor 展开只含偶次幂。再由命题 \ref{prop:einstein-legendre}，
\[
I''\!\left(\frac12\right)=\frac{1}{P''(0)},
\qquad
\frac{1}{2P''(0)}=\frac{51}{11},
\]
于是
\[
I\!\left(\frac12+\delta\right)=\frac{51}{11}\delta^2+O(\delta^4),
\]
即 \(c_2=51/11\)。证毕。
\end{proof}

\begin{theorem}[POM 曲率账本的括号无关性]\label{thm:conclusion-pom-curvature-ledger-parenthesization-invariance}
设
\[
X_0\xrightarrow{f_1}X_1\xrightarrow{f_2}\cdots\xrightarrow{f_r}X_r
\]
是一条有限满射链，并在终点 \(X_r\) 上给定概率测度 \(\nu\)。对任意二叉括号化方式 \(T\)，沿其内部结点自底向上逐次把相邻复合块
\[
X_{i-1}\xrightarrow{g}X_j\xrightarrow{h}X_k
\]
合并为 \(h\circ g\)，并在该结点记入曲率代价 \(\Omega_{\mathrm{KL}}(g,h;\mu)\)，其中 \(\mu\) 为由终点测度 \(\nu\) 经剩余后缀均匀提升到 \(X_k\) 的测度。记全部内部结点代价之和为 \(\mathcal C_T(\nu)\)。则 \(\mathcal C_T(\nu)\) 与括号化 \(T\) 无关，并且有显式闭式
\[
\mathcal C_T(\nu)
=
\kappa(f_r\circ\cdots\circ f_1;\nu)
-\sum_{i=1}^r
\kappa\!\bigl(f_i;L_{f_r\circ\cdots\circ f_{i+1}}\nu\bigr),
\]
其中约定 \(L_{f_r\circ\cdots\circ f_{r+1}}\nu=\nu\)。

特别地，任意 associahedron 上的局部五边形回路都具有零总 holonomy；亦即，KL 曲率在该离散组合层上完全可积。
\leanverified{curvature\_parenthesization}
\end{theorem}
\begin{proof}
对 \(r\) 作归纳。\(r=2\) 时，结论正是定理 \ref{thm:pom-bc-information-stokes-coboundary} 的恒等式
\[
\Omega_{\mathrm{KL}}(f_1,f_2;\nu)
=
\kappa(f_2\circ f_1;\nu)-\kappa(f_2;\nu)-\kappa(f_1;L_{f_2}\nu).
\]

现设结论对长度小于 \(r\) 的链成立。任取一棵非平凡括号树 \(T\)，设其顶层把链切成两段：
\[
g:=f_j\circ\cdots\circ f_1:X_0\twoheadrightarrow X_j,
\qquad
h:=f_r\circ\cdots\circ f_{j+1}:X_j\twoheadrightarrow X_r,
\]
并记
\[
\nu_j:=L_h\nu\in\Delta(X_j).
\]
由归纳假设，左右两棵子树的账本分别为
\[
\mathcal C_{T_{\mathrm{left}}}(\nu_j)
=
\kappa(g;\nu_j)
-\sum_{i=1}^{j}\kappa\!\bigl(f_i;L_{f_j\circ\cdots\circ f_{i+1}}\nu_j\bigr),
\]
\[
\mathcal C_{T_{\mathrm{right}}}(\nu)
=
\kappa(h;\nu)
-\sum_{i=j+1}^{r}\kappa\!\bigl(f_i;L_{f_r\circ\cdots\circ f_{i+1}}\nu\bigr).
\]
注意
\[
L_{f_j\circ\cdots\circ f_{i+1}}\nu_j
=
L_{f_r\circ\cdots\circ f_{i+1}}\nu
\qquad (1\le i\le j).
\]
顶层合并代价由定理 \ref{thm:pom-bc-information-stokes-coboundary} 给出：
\[
\Omega_{\mathrm{KL}}(g,h;\nu)
=
\kappa(h\circ g;\nu)-\kappa(h;\nu)-\kappa(g;L_h\nu)
=
\kappa(f_r\circ\cdots\circ f_1;\nu)-\kappa(h;\nu)-\kappa(g;\nu_j).
\]
将左右子树账本与顶层代价相加，\(\kappa(h;\nu)\) 与 \(\kappa(g;\nu_j)\) 完全望远镜消去，得到
\[
\mathcal C_T(\nu)
=
\kappa(f_r\circ\cdots\circ f_1;\nu)
-\sum_{i=1}^r
\kappa\!\bigl(f_i;L_{f_r\circ\cdots\circ f_{i+1}}\nu\bigr).
\]
右端显然与 \(T\) 无关，故 \(\mathcal C_T(\nu)\) 对括号化不变。

最后，任意两个括号化可由有限次局部重括号化连通；由于每一步总账本都不变，故任何 associahedron 回路的总 holonomy 必为零。证毕。
\end{proof}

\begin{theorem}[tower 缺陷全消失的充要判据]\label{thm:conclusion-pom-tower-defect-vanishing-criterion}
设
\[
Y\xrightarrow{g} Z\xrightarrow{h} W
\]
为满射塔，并令手稿中的 tower defect
\[
\mathfrak T_{g,h}(\psi)(w):=(\mathsf A_h\mathsf A_g\psi)(w)-(\mathsf A_{h\circ g}\psi)(w).
\]
则以下两条件等价：
\begin{enumerate}
  \item 对每个 \(w\in W\) 及每个函数 \(\psi:Y\to\RR\)，均有
  \[
  \mathfrak T_{g,h}(\psi)(w)=0;
  \]
  \item 对每个 \(w\in W\)，函数
  \[
  z\longmapsto d_g(z):=|g^{-1}(z)|
  \]
  在纤维 \(h^{-1}(w)\) 上恒为常数。
\end{enumerate}
亦即，tower averaging 的严格函子性当且仅当中间层的纤维大小在每条外层纤维上完全均匀。
\leanverified{paper\_conclusion\_tower\_defect\_vanishing\_seeds}
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-tower-defect-covariance-law}，对固定 \(w\in W\)，若令 \(Z\) 在 \(h^{-1}(w)\) 上均匀分布，并记
\[
N:=d_g(Z),
\qquad
\Psi:=(\mathsf A_g\psi)(Z),
\]
则
\[
\mathfrak T_{g,h}(\psi)(w)=-\frac{\Cov(N,\Psi)}{\EE[N]}.
\]

若 (2) 成立，则 \(N\) 为常数随机变量，从而对任意 \(\Psi\) 都有 \(\Cov(N,\Psi)=0\)，故 (1) 成立。

反之，设 (1) 成立。固定 \(w\in W\)。对任意函数 \(\Phi:h^{-1}(w)\to\RR\)，可取 \(\psi\) 在每条 \(g\)-纤维上取常值 \(\Phi(z)\)，于是
\[
(\mathsf A_g\psi)(z)=\Phi(z)
\qquad (z\in h^{-1}(w)).
\]
因此
\[
0=\mathfrak T_{g,h}(\psi)(w)=-\frac{\Cov(N,\Phi(Z))}{\EE[N]}
\]
对一切 \(\Phi\) 都成立。特别地，取 \(\Phi=N\)，得到
\[
\operatorname{Var}(N)=0.
\]
由于 \(Z\) 在有限集合 \(h^{-1}(w)\) 上均匀分布，\(\operatorname{Var}(N)=0\) 迫使 \(N\) 在该纤维上处处为常数；亦即 \(d_g\) 在 \(h^{-1}(w)\) 上恒定。故 (2) 成立。证毕。
\end{proof}
