\subsection{window-\texorpdfstring{$6$}{6} 的 cyclic \texorpdfstring{$B_3$}{B3} 根云 multiplicity 张量层析、整格反演与唯一各向同性温点}\label{subsec:conclusion-window6-cyclic-b3-rootcloud-multiplicity-tomography}

在定理 \ref{thm:conclusion-window6-multiplicity-pinning} 已把 cyclic 根云的 multiplicity pinning 固定到 \(B_3\) Cartan 标架、定理 \ref{thm:conclusion-window6-arbitrary-multiplicity-first-moment-axis-collapse} 与定理 \ref{thm:conclusion-window6-arbitrary-multiplicity-second-moment-rigidity} 已把任意 multiplicity 权重的一阶与二阶矩写成显式闭式、而定理 \ref{thm:conclusion-window6-cyclic-escort-first-moment} 与定理 \ref{thm:conclusion-window6-cyclic-escort-second-moment-tensor} 已把 cyclic escort 状态方程完全闭合之后，还可进一步把同一根云的低阶张量几何压缩为一条更细的层析链。其核心在于：任意权函数 \(f:\{2,3,4\}\to\RR\) 的前两阶统计并不保留三个彼此独立的非各向同性方向，而只保留三个标量参数；去掉迹后只剩两条不可约通道，其中一条在一阶矩上完全隐身，却在二阶矩上留下双重简并的中层缺陷。更强地，第二矩本身已经足以完整反演权函数，而整数权对应的全部对角张量恰填满一个指数为 \(120\) 的算术子格。将 \(f(t)=t^q\) 代回之后，固定 window-\(6\) 的 cyclic escort 热力学便化为一条全显式实解析状态方程，其一阶矩与二阶矩同时在且仅在 \(q=0\) 处恢复各向同性；两端极限则分别冻结到 \(d=4\) 与 \(d=2\) 壳层的各向异性协方差。

\begin{theorem}[cyclic multiplicity 权重的三参数正规形]\label{thm:conclusion-window6-cyclic-multiplicity-threeparameter-normalform}
\leanverified{paper\_conclusion\_window6\_cyclic\_multiplicity\_threeparameter\_normalform}
在 \(B_3\) 字典下记
\[
\iota_B:X_6^{\mathrm{cyc}}\to \Phi(B_3)\subset\RR^3.
\]
对任意函数
\[
f:\{2,3,4\}\to\RR
\]
定义一阶与二阶 multiplicity 观测
\[
m_f:=
\sum_{w\in X_6^{\mathrm{cyc}}}
f\!\bigl(d_6^{\mathrm{bin}}(w)\bigr)\,\iota_B(w),
\]
\[
M_f:=
\sum_{w\in X_6^{\mathrm{cyc}}}
f\!\bigl(d_6^{\mathrm{bin}}(w)\bigr)\,
\iota_B(w)\iota_B(w)^{\mathsf T}.
\]
再记
\[
c:=f(2),\qquad
u:=f(4)-f(2),\qquad
v:=f(3)-f(2).
\]
则有精确正规形
\[
m_f=u\,e_1,
\]
\[
M_f
=
10c\,I_3
+u\,\operatorname{diag}(5,6,2)
+v\,\operatorname{diag}(4,0,4).
\]
因此，window-\(6\) 的 cyclic multiplicity 权重在前两阶统计上只保留三个标量参数 \((c,u,v)\)；其中一阶矩仅看见 \(u=f(4)-f(2)\)，而完全看不见 \(v=f(3)-f(2)\)。
\end{theorem}

\begin{proof}
第一条由定理 \ref{thm:conclusion-window6-arbitrary-multiplicity-first-moment-axis-collapse} 立即给出：
\[
m_f=\bigl(f(4)-f(2)\bigr)e_1=u\,e_1.
\]
第二条由定理 \ref{thm:conclusion-window6-arbitrary-multiplicity-second-moment-rigidity}，
\[
M_f
=
\operatorname{diag}\!\Bigl(
f(2)+4f(3)+5f(4),\
4f(2)+6f(4),\
4f(2)+4f(3)+2f(4)
\Bigr).
\]
将
\[
f(3)=c+v,\qquad
f(4)=c+u
\]
代入后得到
\[
M_f
=
\operatorname{diag}(10c+4v+5u,\ 10c+6u,\ 10c+4v+2u),
\]
即
\[
M_f
=
10c\,I_3
+u\,\operatorname{diag}(5,6,2)
+v\,\operatorname{diag}(4,0,4).
\]
证毕。
\end{proof}

\begin{theorem}[去迹二阶矩的双通道分裂]\label{thm:conclusion-window6-cyclic-multiplicity-traceless-twochannel-splitting}
\leanverified{paper\_conclusion\_window6\_cyclic\_multiplicity\_traceless\_twochannel\_splitting}
沿用定理 \ref{thm:conclusion-window6-cyclic-multiplicity-threeparameter-normalform} 的记号，并定义去迹张量
\[
\widetilde M_f:=M_f-\frac{\operatorname{tr}M_f}{3}I_3.
\]
则有严格分裂
\[
\widetilde M_f
=
\frac{u}{3}\,\operatorname{diag}(2,5,-7)
+\frac{4v}{3}\,\operatorname{diag}(1,-2,1).
\]
特别地，去掉各向同性基底以后，window-\(6\) 的 cyclic multiplicity 几何只剩两条且仅两条不可约各向异性通道：其一是由 \(u=f(4)-f(2)\) 控制的轴向通道，其二是由 \(v=f(3)-f(2)\) 控制的中层盲通道。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-window6-cyclic-multiplicity-threeparameter-normalform}，
\[
M_f
=
10c\,I_3
+uQ_4
+vQ_3,
\qquad
Q_4:=\operatorname{diag}(5,6,2),
\qquad
Q_3:=\operatorname{diag}(4,0,4).
\]
故
\[
\operatorname{tr}M_f=30c+13u+8v.
\]
于是
\[
\widetilde M_f
=
u\!\left(Q_4-\frac{13}{3}I_3\right)
+v\!\left(Q_3-\frac{8}{3}I_3\right),
\]
从而
\[
\widetilde M_f
=
\frac{u}{3}\,\operatorname{diag}(2,5,-7)
+\frac{4v}{3}\,\operatorname{diag}(1,-2,1).
\]
证毕。
\end{proof}

\begin{corollary}[中层盲通道的双重简并]\label{cor:conclusion-window6-cyclic-multiplicity-middleblind-double-degeneracy}
\leanverified{paper\_conclusion\_window6\_cyclic\_multiplicity\_middleblind\_double\_degeneracy}
若
\[
f(4)=f(2),\qquad f(3)\neq f(2),
\]
则
\[
m_f=0,
\qquad
\widetilde M_f
=
\frac{4}{3}\bigl(f(3)-f(2)\bigr)\operatorname{diag}(1,-2,1).
\]
因此 \(\widetilde M_f\) 的谱为
\[
\left\{
\frac{4}{3}\bigl(f(3)-f(2)\bigr),\
\frac{4}{3}\bigl(f(3)-f(2)\bigr),\
-\frac{8}{3}\bigl(f(3)-f(2)\bigr)
\right\}.
\]
换言之，存在一类严格非平凡的 multiplicity 变形：它们在一阶矩上完全不可见，却在二阶矩上留下一个具有双重简并的纯中层张量缺陷；其简单特征轴恰为 \(e_2\)，而 \(e_1\)-\(e_3\) 平面保持二重退化。
\end{corollary}

\begin{proof}
在定理 \ref{thm:conclusion-window6-cyclic-multiplicity-traceless-twochannel-splitting} 中取
\[
u=f(4)-f(2)=0,
\qquad
v=f(3)-f(2)\neq 0,
\]
即可得到
\[
m_f=0,
\qquad
\widetilde M_f
=
\frac{4v}{3}\operatorname{diag}(1,-2,1).
\]
其谱直接由对角式读出。证毕。
\end{proof}

\begin{theorem}[二阶矩的完全反演与 index-\texorpdfstring{$120$}{120} 整格]\label{thm:conclusion-window6-cyclic-multiplicity-secondmoment-lattice-inversion}
设
\[
M_f=\operatorname{diag}(A,B,C).
\]
则 \(f\) 可由 \(M_f\) 唯一反演为
\[
f(2)=\frac{-2A+B+2C}{10},
\]
\[
f(3)=\frac{8A-9B+7C}{60},
\]
\[
f(4)=\frac{4A+3B-4C}{30}.
\]
并且
\[
m_f=\frac{A-C}{3}\,e_1.
\]

若进一步要求 \(f(2),f(3),f(4)\in\ZZ\)，则 \((A,B,C)\in\ZZ^3\) 必且只必满足
\[
\Lambda_{\mathrm{cyc}}
:=
\left\{
(A,B,C)\in\ZZ^3:
A-C\equiv 0 \!\!\!\pmod 3,\
4A-3B-C\equiv 0 \!\!\!\pmod{12},\
-2A+B+2C\equiv 0 \!\!\!\pmod{10}
\right\}.
\]
此外，\(\Lambda_{\mathrm{cyc}}\) 在 \(\ZZ^3\) 中的指数恰为
\[
[\ZZ^3:\Lambda_{\mathrm{cyc}}]=120.
\]
因此，window-\(6\) 的第二矩已经完整记住了权函数 \(f\) 本身；整数权的全部对角二阶矩恰好填满对角整格中的一个 index-\(120\) 算术子格。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-window6-cyclic-multiplicity-threeparameter-normalform}，
\[
A=f(2)+4f(3)+5f(4),\qquad
B=4f(2)+6f(4),\qquad
C=4f(2)+4f(3)+2f(4).
\]
解该线性系统便得到
\[
f(2)=\frac{-2A+B+2C}{10},\qquad
f(3)=\frac{8A-9B+7C}{60},\qquad
f(4)=\frac{4A+3B-4C}{30}.
\]
同时
\[
f(4)-f(2)=\frac{A-C}{3},
\]
故由定理 \ref{thm:conclusion-window6-cyclic-multiplicity-threeparameter-normalform} 的一阶矩公式，
\[
m_f=\bigl(f(4)-f(2)\bigr)e_1=\frac{A-C}{3}\,e_1.
\]

现假设 \(f(2),f(3),f(4)\in\ZZ\)。则
\[
-2A+B+2C=10f(2),
\]
\[
A-C=3\bigl(f(4)-f(2)\bigr),
\]
\[
4A-3B-C=12\bigl(f(3)-f(2)\bigr),
\]
从而 \((A,B,C)\in\Lambda_{\mathrm{cyc}}\)。

反之，若 \((A,B,C)\in\Lambda_{\mathrm{cyc}}\)，则
\[
f(2)=\frac{-2A+B+2C}{10}\in\ZZ,
\]
并由
\[
f(4)=f(2)+\frac{A-C}{3},
\qquad
f(3)=f(2)+\frac{4A-3B-C}{12},
\]
立得 \(f(3),f(4)\in\ZZ\)。故整数可实现像恰为 \(\Lambda_{\mathrm{cyc}}\)。

最后，\((f(2),f(3),f(4))\mapsto(A,B,C)\) 的系数矩阵为
\[
\begin{pmatrix}
1&4&5\\
4&0&6\\
4&4&2
\end{pmatrix},
\]
其行列式为
\[
120.
\]
因此其像格在 \(\ZZ^3\) 中的指数恰为 \(120\)。证毕。
\end{proof}

\begin{theorem}[cyclic escort 的显式状态方程与唯一各向同性温点]\label{thm:conclusion-window6-cyclic-multiplicity-escort-unique-isotropy}
\leanverified{paper\_conclusion\_window6\_cyclic\_multiplicity\_escort\_unique\_isotropy}
对任意实数 \(q\in\RR\)，定义 cyclic escort 配分函数、归一化一阶矩与归一化二阶矩
\[
Z_{\mathrm{cyc}}(q)
:=
\sum_{w\in X_6^{\mathrm{cyc}}}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q,
\]
\[
\mu(q)
:=
\frac{1}{Z_{\mathrm{cyc}}(q)}
\sum_{w\in X_6^{\mathrm{cyc}}}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q\,\iota_B(w),
\]
\[
\Sigma(q)
:=
\frac{1}{Z_{\mathrm{cyc}}(q)}
\sum_{w\in X_6^{\mathrm{cyc}}}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q\,
\iota_B(w)\iota_B(w)^{\mathsf T}.
\]
则有完全显式的状态方程
\[
Z_{\mathrm{cyc}}(q)=5\cdot 2^q+4\cdot 3^q+9\cdot 4^q,
\]
\[
\mu(q)
=
\frac{4^q-2^q}{5\cdot 2^q+4\cdot 3^q+9\cdot 4^q}\,e_1,
\]
\[
\Sigma(q)
=
\frac{1}{5\cdot 2^q+4\cdot 3^q+9\cdot 4^q}
\operatorname{diag}\!\bigl(
2^q+4\cdot 3^q+5\cdot 4^q,\
4\cdot 2^q+6\cdot 4^q,\
4\cdot 2^q+4\cdot 3^q+2\cdot 4^q
\bigr).
\]
并且
\[
\mu(q)=0 \iff q=0,
\]
\[
\Sigma(q)\ \text{为标量矩阵} \iff q=0.
\]
故 \(q=0\) 是 window-\(6\) 的 cyclic multiplicity escort 的唯一各向同性温点；在此点
\[
\mu(0)=0,
\qquad
\Sigma(0)=\frac{5}{9}I_3.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-window6-cyclic-escort-first-moment} 的闭式，
\[
Z_{\mathrm{cyc}}(q)=5\cdot 2^q+4\cdot 3^q+9\cdot 4^q,
\qquad
\sum_{w\in X_6^{\mathrm{cyc}}}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q\,\iota_B(w)
=(4^q-2^q)e_1.
\]
因此
\[
\mu(q)
=
\frac{4^q-2^q}{5\cdot 2^q+4\cdot 3^q+9\cdot 4^q}\,e_1.
\]
同理，由定理 \ref{thm:conclusion-window6-cyclic-escort-second-moment-tensor} 的显式对角公式，
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q\,
\iota_B(w)\iota_B(w)^{\mathsf T}
=
\operatorname{diag}\!\bigl(
2^q+4\cdot 3^q+5\cdot 4^q,\
4\cdot 2^q+6\cdot 4^q,\
4\cdot 2^q+4\cdot 3^q+2\cdot 4^q
\bigr),
\]
从而得到 \(\Sigma(q)\) 的闭式。上述推导只涉及有限项指数和，故对一切 \(q\in\RR\) 均成立。

由于
\[
4^q-2^q=2^q(2^q-1),
\]
故 \(\mu(q)=0\) 当且仅当 \(q=0\)。

另一方面，把
\[
f_q(t):=t^q\qquad (t\in\{2,3,4\})
\]
代入定理 \ref{thm:conclusion-window6-arbitrary-multiplicity-second-moment-rigidity}，可知 \(\Sigma(q)\) 为标量矩阵当且仅当 \(f_q\) 在 \(\{2,3,4\}\) 上常值。由于 \(2,3,4\) 两两不同，这恰当且仅当 \(q=0\)。

最后在 \(q=0\) 处，
\[
Z_{\mathrm{cyc}}(0)=5+4+9=18,
\]
而由定理 \ref{thm:conclusion-window6-cyclic-escort-second-moment-tensor}，
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_B(w)\iota_B(w)^{\mathsf T}=10I_3.
\]
故
\[
\Sigma(0)=\frac{10}{18}I_3=\frac59 I_3.
\]
证毕。
\end{proof}

\begin{corollary}[escort 温标两端的精确冻结]\label{cor:conclusion-window6-cyclic-multiplicity-escort-endpoint-freezing}
\leanverified{paper\_conclusion\_window6\_cyclic\_multiplicity\_escort\_endpoint\_freezing}
沿用定理 \ref{thm:conclusion-window6-cyclic-multiplicity-escort-unique-isotropy} 的记号，则 \(\mu(q)\) 与 \(\Sigma(q)\) 在整个实轴上都是实解析函数，并满足
\[
\mu(q)\xrightarrow[q\to+\infty]{}\frac19\,e_1,
\qquad
\Sigma(q)\xrightarrow[q\to+\infty]{}\frac19\,\operatorname{diag}(5,6,2),
\]
\[
\mu(q)\xrightarrow[q\to-\infty]{}-\frac15\,e_1,
\qquad
\Sigma(q)\xrightarrow[q\to-\infty]{}\frac15\,\operatorname{diag}(1,4,4).
\]
因此，固定 window-\(6\) 的 cyclic multiplicity escort 热力学在内部没有临界奇点，只在两端分别冻结到 \(d=4\) 与 \(d=2\) 壳层的各向异性极限。
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-window6-cyclic-multiplicity-escort-unique-isotropy} 的闭式，\(Z_{\mathrm{cyc}}(q)\) 是正系数指数和，故在 \(\RR\) 上处处正且实解析；\(\mu(q)\) 与 \(\Sigma(q)\) 作为其有理组合，亦在 \(\RR\) 上实解析。

当 \(q\to+\infty\) 时，\(4^q\) 为唯一主导项，故
\[
\mu(q)\longrightarrow \frac19\,e_1,
\qquad
\Sigma(q)\longrightarrow \frac19\,\operatorname{diag}(5,6,2).
\]
当 \(q\to-\infty\) 时，\(2^q\) 为唯一主导项，故
\[
\mu(q)\longrightarrow -\frac15\,e_1,
\qquad
\Sigma(q)\longrightarrow \frac15\,\operatorname{diag}(1,4,4).
\]
证毕。
\end{proof}

\begin{conjecture}[偶数窗口 cyclic escort 的唯一 isotropy 原理]\label{conj:conclusion-window6-evenwindow-cyclic-escort-unique-isotropy}
设 \(m\) 为充分大的偶数窗口，并假定其 cyclic 扇区同时满足 pinned root datum、完整中心分裂、Fibonacci 双频锁定与 codimension-one 根截面结构。对任意 \(q\in\RR\)，定义 multiplicity escort
\[
\nu_{m,q}(w)\propto d_m^{\mathrm{bin}}(w)^q
\qquad
(w\in X_m^{\mathrm{cyc}}).
\]
则预期以下三项同时成立：
\begin{enumerate}
  \item 存在唯一参数 \(q=0\)，使其一阶矩消失且二阶矩恢复各向同性；
  \item 对任意 \(q\neq 0\)，其中心化 Gaussian 长程部分至多沿 pinned 根截面中的一条实轴存活；
  \item 其余自由度只能退化为有限值中心余量，而不能形成第二个线性无关的无质量各向同性场。
\end{enumerate}
对 \(m=6\)，定理 \ref{thm:conclusion-window6-cyclic-multiplicity-escort-unique-isotropy} 与推论 \ref{cor:conclusion-window6-cyclic-multiplicity-escort-endpoint-freezing} 已给出最小非平凡实例：唯一各向同性温点恰为 \(q=0\)，而两端冻结极限严格各向异性。
\end{conjecture}

\paragraph{统一读法}
上述链条把 window-\(6\) 的 cyclic \(B_3\) 根云进一步压缩为一条低阶张量层析律：任意 multiplicity 权函数的前两阶统计只保留三个标量参数，而去迹后只剩两条不可约通道，其中中层盲通道在一阶矩上完全隐身，却在二阶矩上留下双重简并的张量缺陷。另一方面，第二矩本身已经足以完整反演权函数，并把整数像锁定到一个 index-\(120\) 的对角整格。将 \(f(t)=t^q\) 代回之后，固定窗 cyclic escort 热力学便不再只是“可解析而无相变”的粗陈述，而是变成一条完全显式的状态方程：一阶矩与二阶矩在且仅在 \(q=0\) 处同时回到 isotropy，离开该点后则必定沿单轴通道极化，并在 \(q\to\pm\infty\) 两端分别冻结到 \(d=4\) 与 \(d=2\) 壳层的各向异性协方差。因而，window-\(6\) 中 multiplicity 标签的作用不再只是给根云加权，而是把 cyclic \(B_3\) 根云的协方差几何、整性算术与 escort 温标共同压缩为一个唯一 isotropy 温点现象。
