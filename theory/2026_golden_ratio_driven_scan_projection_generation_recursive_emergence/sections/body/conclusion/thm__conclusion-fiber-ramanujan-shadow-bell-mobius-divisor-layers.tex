\paragraph{Ramanujan 纤维阴影对 Bell--Möbius 商格与复制轨道的完备坐标}
在定理 \ref{thm:conclusion-fiber-intermediate-system-lattice-mobius}、推论 \ref{cor:conclusion-intermediate-system-characteristic-polynomial-sign-rigidity} 与定理 \ref{thm:conclusion-oracle-capacity-determines-bell-mobius-intermediate-system} 已经表明容量曲线足以恢复纤维多重度谱之后，还可引入另一组完全不同但同样完备的除数型坐标。为避免与秩生成函数 \(R_f(u)\) 混淆，下文记有限满射 \(f:\Omega\twoheadrightarrow X\) 的纤维 Ramanujan 阴影为
\[
\mathscr R_f(q):=\sum_{x\in X}c_q(d(x))
\qquad(q\ge 1),
\]
其中 \(d(x):=\card{f^{-1}(x)}\)，并记
\[
N_f(r):=\#\{x\in X:\ r\mid d(x)\},
\qquad
h_f(s):=\#\{x\in X:\ d(x)=s\},
\qquad
D:=\max_{x\in X}d(x).
\]

\begin{theorem}[纤维多重度的 Ramanujan--divisor 层析公式]\label{thm:conclusion-fiber-ramanujan-shadow-divisor-tomography}
对每个整数 \(r\ge 1\)，有严格恒等式
\[
N_f(r)=\frac1r\sum_{q\mid r}\mathscr R_f(q).
\]
\leanverified{paper\_conclusion\_fiber\_ramanujan\_shadow\_divisor\_tomography}
\end{theorem}
\begin{proof}
经典 Ramanujan 恒等式
\[
\sum_{q\mid r}c_q(n)=r\,\mathbf 1_{r\mid n}
\]
对每个 \(n\in \NN\) 都成立。于是
\[
\frac1r\sum_{q\mid r}\mathscr R_f(q)
=
\frac1r\sum_{q\mid r}\sum_{x\in X}c_q(d(x))
=
\sum_{x\in X}\frac1r\sum_{q\mid r}c_q(d(x))
=
\sum_{x\in X}\mathbf 1_{r\mid d(x)}
=
N_f(r).
\]
证毕。
\end{proof}

\begin{theorem}[纤维直方图的 Ramanujan 完全反演]\label{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion}
对每个 \(1\le s\le D\)，有显式反演公式
\[
h_f(s)=\sum_{t=1}^{\lfloor D/s\rfloor}\mu(t)\,N_f(st)
=
\sum_{t=1}^{\lfloor D/s\rfloor}\frac{\mu(t)}{st}\sum_{q\mid st}\mathscr R_f(q).
\]
因此，有限阴影族 \(\{\mathscr R_f(q)\}_{1\le q\le D}\) 唯一决定完整纤维多重度谱 \(\{h_f(s)\}_{1\le s\le D}\)。
\leanverified{paper\_conclusion\_fiber\_ramanujan\_shadow\_histogram\_inversion}
\end{theorem}
\begin{proof}
由定义，
\[
N_f(r)=\sum_{u=1}^{\lfloor D/r\rfloor}h_f(ru),
\]
这正是正整数除数偏序上的上闭包和。对其作 Möbius 反演，得到
\[
h_f(s)=\sum_{t=1}^{\lfloor D/s\rfloor}\mu(t)\,N_f(st).
\]
再代入定理 \ref{thm:conclusion-fiber-ramanujan-shadow-divisor-tomography} 即得第二式。反演公式唯一，故 \(\{\mathscr R_f(q)\}_{1\le q\le D}\) 唯一决定 \(\{h_f(s)\}_{1\le s\le D}\)。证毕。
\end{proof}

\begin{theorem}[Bell--Möbius 中间系统不变量的 Ramanujan 完备性]\label{thm:conclusion-fiber-ramanujan-shadow-bell-mobius-completeness}
有限阴影族 \(\{\mathscr R_f(q)\}_{1\le q\le D}\) 唯一决定下列全部中间系统不变量：
\begin{enumerate}
\normalfont
  \item 中间系统格的逐直方图分解
  \[
  \mathcal Q(f)\cong \prod_{s=1}^{D}\Pi_s^{\,h_f(s)};
  \]
  \item 中间系统总数
  \[
  \card{\mathcal Q(f)}=\prod_{s=1}^{D}\mathrm{Bell}(s)^{h_f(s)};
  \]
  \item 顶端 Möbius 数
  \[
  \mu_{\mathcal Q(f)}(\hat 0,\hat 1)
  =
  (-1)^{|\Omega|-|X|}
  \prod_{s=1}^{D}(s-1)!^{\,h_f(s)};
  \]
  \item 对任意素数 \(p\)，其 \((p)\)-进赋值
  \[
  v_p\!\bigl(\mu_{\mathcal Q(f)}(\hat 0,\hat 1)\bigr)
  =
  \sum_{s=1}^{D}h_f(s)\sum_{j\ge 1}\left\lfloor\frac{s-1}{p^j}\right\rfloor;
  \]
  \item Bell--Boolean 刚性缺口
  \[
  \Xi_f:=
  \log_2\card{\mathcal Q(f)}-(|\Omega|-|X|)
  =
  \sum_{s=1}^{D}h_f(s)\log_2\!\frac{\mathrm{Bell}(s)}{2^{s-1}}
  \ge 0,
  \]
  且
  \[
  \Xi_f=0
  \iff
  h_f(s)=0\qquad(\forall s\ge 3).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion}，Ramanujan 阴影唯一决定直方图 \(\{h_f(s)\}\)。将定理 \ref{thm:conclusion-fiber-intermediate-system-lattice-mobius} 中的逐纤维分解
\[
\mathcal Q(f)\cong \prod_{x\in X}\Pi_{d(x)}
\]
按纤维大小 regroup，便得到
\[
\mathcal Q(f)\cong \prod_{s=1}^{D}\Pi_s^{\,h_f(s)}.
\]
Bell 数乘积公式与顶端 Möbius 数公式同理 regroup，即得前两式与
\[
\mu_{\mathcal Q(f)}(\hat 0,\hat 1)
=
\prod_{s=1}^{D}\bigl((-1)^{s-1}(s-1)!\bigr)^{h_f(s)}.
\]
又
\[
\sum_{s=1}^{D}(s-1)h_f(s)
=
\sum_{x\in X}(d(x)-1)
=
|\Omega|-|X|,
\]
故其符号指数化简为所示形式。

对任意素数 \(p\)，再结合 Legendre 公式
\[
v_p((s-1)!)=\sum_{j\ge 1}\left\lfloor\frac{s-1}{p^j}\right\rfloor,
\]
便得到 \((p)\)-进赋值公式。

最后，由基本不等式
\[
\mathrm{Bell}(s)\ge 2^{s-1},
\]
且等号当且仅当 \(s=1,2\)，可得
\[
\log_2\card{\mathcal Q(f)}
=
\sum_{s=1}^{D}h_f(s)\log_2\mathrm{Bell}(s)
\ge
\sum_{s=1}^{D}h_f(s)(s-1)
=
|\Omega|-|X|.
\]
将差值写开即得 \(\Xi_f\) 的表达式与非负性；等号成立当且仅当所有 \(s\ge 3\) 的纤维层都消失。证毕。
\end{proof}

\begin{theorem}[复制轨道塔的指数母函数由 Ramanujan 阴影完全决定]\label{thm:conclusion-fiber-ramanujan-shadow-orbit-egf-completeness}
令
\[
G_f:=\prod_{x\in X}\mathfrak S_{d(x)}
\]
按纤维独立置换作用于 \(\Omega^q\)，并记
\[
O_f(q):=\card{\Omega^q/G_f}.
\]
再定义受限 Bell 数
\[
B_n^{(\le s)}:=\sum_{k=0}^{\min(n,s)}S(n,k).
\]
则其指数母函数满足闭式
\[
\mathcal O_f(t):=\sum_{q\ge 0}\frac{O_f(q)}{q!}t^q
=
\prod_{s=1}^{D}
\left(
\sum_{n\ge 0}B_n^{(\le s)}\frac{t^n}{n!}
\right)^{h_f(s)}
=
\prod_{s=1}^{D}
\left(
\sum_{k=0}^{s}\frac{(e^t-1)^k}{k!}
\right)^{h_f(s)}.
\]
因此，整个复制轨道塔 \(\{O_f(q)\}_{q\ge 0}\) 都由有限阴影族 \(\{\mathscr R_f(q)\}_{1\le q\le D}\) 唯一决定。
\end{theorem}
\begin{proof}
与定理 \ref{thm:op-algebra-fold-gauge-orbit-classification-qtuple} 完全同理，\(G_f\) 对 \(\Omega^q\) 的对角作用轨道由“颜色分布 \((n_x)_{x\in X}\) 加每条纤维内的受限集合划分”完备刻画，因此
\[
O_f(q)
=
\sum_{\substack{(n_x)_{x\in X}\in\ZZ_{\ge 0}^{X}\\ \sum_x n_x=q}}
\frac{q!}{\prod_{x\in X}n_x!}
\prod_{x\in X}B_{n_x}^{(\le d(x))}.
\]
除以 \(q!\) 后对 \(q\) 求和，即得
\[
\mathcal O_f(t)
=
\prod_{x\in X}
\left(
\sum_{n\ge 0}B_n^{(\le d(x))}\frac{t^n}{n!}
\right).
\]
按纤维大小 regroup 便得第一重乘积式。再利用受限 Bell 数的标准指数母函数
\[
\sum_{n\ge 0}B_n^{(\le s)}\frac{t^n}{n!}
=
\sum_{k=0}^{s}\frac{(e^t-1)^k}{k!},
\]
得到第二重乘积式。最后由定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion}，\(\{h_f(s)\}\) 已被有限阴影族唯一恢复，故整条复制轨道塔亦被唯一决定。证毕。
\end{proof}

\begin{corollary}[低阶复制轨道的阈值尾完全显影]\label{cor:conclusion-fiber-ramanujan-shadow-low-order-thresholds}
\leanverified{paper\_conclusion\_fiber\_ramanujan\_shadow\_low\_order\_thresholds}
定义阈值尾
\[
T_r(f):=\#\{x\in X:\ d(x)\ge r\}=\sum_{s=r}^{D}h_f(s).
\]
则有显式闭式
\[
O_f(2)=|X|^2+T_2(f),
\]
\[
O_f(3)=|X|^3+3|X|\,T_2(f)+T_3(f).
\]
\end{corollary}
\begin{proof}
对 \(q=2\)，轨道型分为“两位置颜色不同”与“两位置同色”两类。前者贡献 \(|X|(|X|-1)\)；后者贡献
\[
\sum_{x\in X}B_2^{(\le d(x))}
=
\sum_{x\in X}\bigl(1+\mathbf 1_{d(x)\ge 2}\bigr)
=
|X|+T_2(f).
\]
相加得 \(O_f(2)=|X|^2+T_2(f)\)。

对 \(q=3\)，轨道型分为 \((1+1+1)\)、\((2+1)\)、\((3)\) 三类。第一类贡献 \(|X|(|X|-1)(|X|-2)\)。第二类贡献
\[
3(|X|-1)\sum_{x\in X}B_2^{(\le d(x))}
=
3(|X|-1)\bigl(|X|+T_2(f)\bigr).
\]
第三类贡献
\[
\sum_{x\in X}B_3^{(\le d(x))}
=
\sum_{x\in X}\bigl(1+3\mathbf 1_{d(x)\ge 2}+\mathbf 1_{d(x)\ge 3}\bigr)
=
|X|+3T_2(f)+T_3(f).
\]
三项相加并整理，即得
\[
O_f(3)=|X|^3+3|X|\,T_2(f)+T_3(f).
\]
证毕。
\end{proof}

\begin{theorem}[折叠粗化的 divisor--partition 双层完备坐标]\label{thm:conclusion-fold-ramanujan-shadow-divisor-partition-two-layer-rigidity}
对折叠满射
\[
\Fold_m:\Omega_m\twoheadrightarrow X_m,
\qquad
n_m:=F_{m+2},
\qquad
\mathscr R_m(q):=\sum_{x\in X_m}c_q(d_m(x)),
\]
其中 \(d_m(x):=\card{\Fold_m^{-1}(x)}\)，记
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
\mathfrak T_m:=\Bigl(n_m,\bigl(\mathscr R_m(q)\bigr)_{1\le q\le D_m}\Bigr).
\]
则二元数据 \(\mathfrak T_m\) 唯一决定折叠粗化的两层全部结构：
\begin{enumerate}
\normalfont
  \item 以可见同余商 \(R_m:=\Omega_m/\!\equiv_m\) 为目标的全部半环兼容中间商
  \[
  \Omega_m\twoheadrightarrow Q\twoheadrightarrow R_m
  \]
  的同构类格，恰由整数 \(n_m=F_{m+2}\) 的约数格唯一决定；
  \item 全部集合论中间商
  \[
  \Omega_m\twoheadrightarrow Y\twoheadrightarrow X_m
  \]
  的同构类格，以及其复制轨道塔 \(\{O_{\Fold_m}(q)\}_{q\ge 0}\)，都由有限阴影族 \(\{\mathscr R_m(q)\}_{1\le q\le D_m}\) 唯一决定。
\end{enumerate}
\end{theorem}
\begin{proof}
第一条正是定理 \ref{thm:xi-fold-congruence-quotient-divisor-lattice} 的内容：可见同余商 \(R_m\) 的全部半环合同与 \(F_{m+2}\) 的约数格反同构，从而半环兼容的中间商完全由 \(n_m\) 决定。

第二条中，把 \(f\) 取为 \(\Fold_m\)。由定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion}，\(\{\mathscr R_m(q)\}_{q\le D_m}\) 唯一恢复纤维直方图 \(\{h_m(s)\}\)；再由定理 \ref{thm:conclusion-fiber-ramanujan-shadow-bell-mobius-completeness}，全部集合论中间商格与其 Bell--Möbius 不变量被唯一恢复；最后由定理 \ref{thm:conclusion-fiber-ramanujan-shadow-orbit-egf-completeness}，复制轨道塔亦被唯一决定。证毕。
\end{proof}

\begin{conjecture}[纤维 Ramanujan 阴影的极小完备性]\label{conj:conclusion-fiber-ramanujan-shadow-minimal-completeness}
固定最大纤维界 \(D\)。在所有仅依赖于纤维多重度多重集 \(\{d(x)\}_{x\in X}\) 的 divisor-linear 统计族中，有限阴影族
\[
\{\mathscr R_f(q)\}_{1\le q\le D}
\]
应当是恢复
\[
\mathcal Q(f),
\qquad
\mu_{\mathcal Q(f)}(\hat 0,\hat 1),
\qquad
\mathcal O_f(t)
\]
的极小完备坐标族。更具体地，任意真子族
\[
\{\mathscr R_f(q)\}_{q\in S},
\qquad
S\subsetneq \{1,\dots,D\},
\]
都不应能在全体满足 \(\max_x d(x)\le D\) 的有限满射类上统一地区分这些不变量。
\end{conjecture}
