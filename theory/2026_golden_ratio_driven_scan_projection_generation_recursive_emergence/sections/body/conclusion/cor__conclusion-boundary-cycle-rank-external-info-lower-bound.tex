\begin{corollary}[散度完成化的最小外置寄存预算]\label{cor:conclusion-boundary-cycle-rank-external-info-lower-bound}
设 \(f:\Omega_n=\{0,1\}^n\twoheadrightarrow X\) 为任意满射粗粒化，并令 \(\mathsf B_f\) 为其边界邻接多图（定义 \ref{def:hypercube-boundary-multigraph}）。
记
\[
r:=\operatorname{rank}_{\ZZ}H_1(\mathsf B_f;\ZZ)=A(f)-|X|+c(\mathsf B_f)
\]
（命题 \ref{prop:hypercube-boundary-multigraph-h1-cdim}）。
对任意素数 \(p\)，考虑模 \(p\) 的边界算子 \(\partial_p:C_1(\mathsf B_f;\FF_p)\to C_0(\mathsf B_f;\FF_p)\)。
则对任意 \(\delta\in\mathrm{im}(\partial_p)\)，解集
\[
\mathcal F_p(\delta):=\{\,\varphi\in C_1(\mathsf B_f;\FF_p):\ \partial_p\varphi=\delta\,\}
\]
要么为空，要么为一个以 \(\ker(\partial_p)\cong \FF_p^{\,r}\) 为平移空间的仿射空间；因此若 \(\mathcal F_p(\delta)\neq\varnothing\)，则
\[
\boxed{\ |\mathcal F_p(\delta)|=p^{r}\ }.
\]
称一个有限集合 \(R\) 与映射 \(\ell:C_1(\mathsf B_f;\FF_p)\to R\) 为\emph{散度完成化寄存器}，若映射
\[
\Phi:\ C_1(\mathsf B_f;\FF_p)\ \to\ \operatorname{im}(\partial_p)\times R,
\qquad
\Phi(\varphi):=\bigl(\partial_p\varphi,\ell(\varphi)\bigr)
\]
为单射。
该口径与内部审计文档 \cite{Internal2026BoundaryDivergenceCompletionization} 的散度完成化定义一致。
则任何散度完成化寄存器都满足
\[
\boxed{\ |R|\ \ge\ p^{r}\ },
\]
并且存在 \(|R|=p^r\) 的散度完成化寄存器；因此最小外置寄存预算恰为 \(|R|_{\min}=p^r\)，等价地最小信息量为 \(r\log_2 p\) 比特。
\end{corollary}
\begin{proof}
若 \(\mathcal F_p(\delta)\) 非空，取一特解 \(\varphi_0\)，则任一解可写为 \(\varphi_0+\psi\)，其中 \(\psi\in\ker(\partial_p)\)；
反之对任意 \(\psi\in\ker(\partial_p)\) 有 \(\partial_p(\varphi_0+\psi)=\delta\)，故 \(\mathcal F_p(\delta)=\varphi_0+\ker(\partial_p)\)。
而 \(\ker(\partial_p)\cong H_1(\mathsf B_f;\FF_p)\) 且其 \(\FF_p\)-维数等于 \(\operatorname{rank}_{\ZZ}H_1(\mathsf B_f;\ZZ)=r\)，由命题 \ref{prop:hypercube-boundary-multigraph-h1-cdim} 及其链复形表述可得。
因此 \(|\ker(\partial_p)|=p^{r}\)，从而 \(|\mathcal F_p(\delta)|=p^{r}\)。
\par
若 \(\Phi\) 为单射，则其限制到 \(\ker(\partial_p)=\mathcal F_p(0)\) 上同样单射，故 \(\ell\) 在 \(\ker(\partial_p)\) 上必须单射，从而 \(|R|\ge |\ker(\partial_p)|=p^r\)。
\par
反过来，由有限维 \(\FF_p\)-向量空间分解可取直和
\(
C_1(\mathsf B_f;\FF_p)=\ker(\partial_p)\oplus W
\)
并令 \(\pi:C_1(\mathsf B_f;\FF_p)\to \ker(\partial_p)\) 为沿 \(W\) 的线性投影。
取 \(R:=\ker(\partial_p)\) 并定义 \(\ell:=\pi\)，则若 \(\partial_p\varphi=\partial_p\psi\) 且 \(\ell(\varphi)=\ell(\psi)\)，有 \(\varphi-\psi\in\ker(\partial_p)\) 且 \(\pi(\varphi-\psi)=0\)；
但 \(\pi|_{\ker(\partial_p)}=\mathrm{id}\)，故 \(\varphi=\psi\)，从而 \(\Phi\) 单射。
因此存在 \(|R|=p^r\) 的完成化寄存器。
证毕。
\end{proof}

\begin{corollary}[pro-\(p\) 外置的最小 \(p\)-圆维由边界回路秩锁定]\label{cor:conclusion-pcdim-prop-cycle-rank-minimal}
沿用推论 \ref{cor:conclusion-boundary-cycle-rank-external-info-lower-bound} 的记号，并固定素数 \(p\)。
令 \(K\) 为交换 pro-\(p\) 群，并记
\[
\pcdim(K):=\dim_{\FF_p}(K/pK).
\]
若存在映射 \(\Phi:\mathcal F_p(\delta)\to K\) 满足：对任意 \(\varphi,\psi\in\mathcal F_p(\delta)\)，差值 \(\Phi(\varphi)-\Phi(\psi)\) 仅依赖于 \(\varphi-\psi\in\ker(\partial_p)\)，并诱导 \(\FF_p\)-线性单射
\[
H_1(\mathsf B_f;\FF_p)\ \cong\ \ker(\partial_p)/\mathrm{im}(d_p)\ \hookrightarrow\ K/pK,
\]
则必有
\[
\boxed{\;\pcdim(K)\ \ge\ r\;}
\qquad\left(r=\operatorname{rank}_{\ZZ}H_1(\mathsf B_f;\ZZ)\right).
\]
并且该下界可达：取 \(K=\ZZ_p^{\,r}\) 时存在满足上述性质的 \(\Phi\)，从而 \(\pcdim(K)=r\)。
\end{corollary}
\begin{proof}
假设给出的线性单射蕴含
\(
r=\dim_{\FF_p}H_1(\mathsf B_f;\FF_p)\le \dim_{\FF_p}(K/pK)=\pcdim(K)
\)。
可达性取 \(K=\ZZ_p^{\,r}\) 并选取同构 \(K/pK\cong \FF_p^{\,r}\)，再由一特解 \(\varphi_0\in\mathcal F_p(\delta)\) 将任意 \(\varphi\) 映到其相对回路坐标的 \(p\)-进提升即可。证毕。
\end{proof}

\begin{corollary}[常数内存的规范 lift 选择器不可能]\label{cor:conclusion-canonical-lift-selector-memory-lb}
沿用推论 \ref{cor:conclusion-boundary-cycle-rank-external-info-lower-bound} 的记号，并固定一素数 \(p\)。
设存在一类确定性信息处理过程：对每个 \(\delta\in\mathrm{im}(\partial_p)\)（且 \(\mathcal F_p(\delta)\neq\varnothing\)）输出一个选定的解
\[
\mathrm{Sel}(\delta)\in \mathcal F_p(\delta),
\]
并要求该输出对同一 \(\delta\) 单值且可审计（即其内部有限状态足以区分不同的候选解）。
则其内部状态数必满足
\[
\#\mathrm{States}\ \ge\ p^{r},
\qquad\text{等价地}\qquad
\log_2\#\mathrm{States}\ \ge\ r\log_2 p.
\]
在推论 \ref{cor:conclusion-boundary-cycle-rank-density-explosion} 的假设下，任何对所有 \(n\) 统一有界的状态数都不可能实现上述选择器。
\end{corollary}
\begin{proof}
若 \(\mathrm{Sel}\) 对每个 \(\delta\) 选取唯一 \(\mathcal F_p(\delta)\) 中元素，则在最坏情形（例如 \(\delta=0\)）必须能区分 \(|\mathcal F_p(0)|=p^r\) 个候选。
由推论 \ref{cor:conclusion-boundary-cycle-rank-external-info-lower-bound} 的最小寄存预算结论，任何实现该单值化的有限状态外置寄存器都满足 \(|R|\ge p^r\)，将 \(R\) 视为内部状态集合即得所示不等式。
最后一条由 \(r_n=\Theta(n2^{n-1})\) 推出 \(p^{r_n}\to\infty\)，从而不存在统一有界的状态数。证毕。
\end{proof}

\begin{corollary}[边界回路秩诱导的 Löwner 体量素数下界]\label{cor:conclusion-boundary-cycle-rank-loewner-prime-lower-bound}
设 \(f:\Omega_n\twoheadrightarrow X\) 为任意满射粗粒化，并令
\(
r:=\operatorname{rank}_{\ZZ}H_1(\mathsf B_f;\ZZ)
\)
（命题 \ref{prop:hypercube-boundary-multigraph-h1-cdim}）。
取互异素数 \(q_1<\cdots<q_k\)，置 \(L_i:=\log q_i\)，并记
\(
V_{k,t}(L):=\operatorname{vol}(E_{k,t}(L))
\)
为定理 \ref{thm:xi-godel-loewner-exterior-box-loewner-holographic-weight} 的 \(t\)-阶外幂盒体 Löwner 体量。
若存在 Gödel 外置单射
\[
\iota:\ H_1(\mathsf B_f;\ZZ)\hookrightarrow \langle q_1,\dots,q_k\rangle\subset \QQ_{>0},
\]
则必有 \(k\ge r\)。
并且对任意 \(1\le t\le k\)，令 \(p_i\) 为第 \(i\) 个素数，则成立
\[
\boxed{\;
\log V_{k,t}(\log q_1,\dots,\log q_k)\ \ge\ \log V_{k,t}(\log p_1,\dots,\log p_k).\;}
\]
当且仅当 \(q_i=p_i\)（\(1\le i\le k\)）时取等号。
特别地，在最小通道预算 \(k=r\) 下，
\[
\frac{1}{\binom{r}{t}}\log V_{r,t}(\log q_1,\dots,\log q_r)
\ge
\frac{1}{\binom{r}{t}}\log V_{r,t}(\log p_1,\dots,\log p_r),
\]
且当 \(t\) 固定、\(r\to\infty\) 时右端满足渐近
\[
\frac{1}{\binom{r}{t}}\log V_{r,t}(\log p_1,\dots,\log p_r)
=t\log\log r+O(1).
\]
\end{corollary}
\begin{proof}
由于 \(H_1(\mathsf B_f;\ZZ)\cong \ZZ^{\,r}\) 且 \(\langle q_1,\dots,q_k\rangle\cong \ZZ^{\,k}\)（唯一分解定理），
任一单射 \(\iota:\ZZ^{r}\hookrightarrow \ZZ^{k}\) 在 \(\QQ\)-张量下诱导线性单射 \(\QQ^{r}\hookrightarrow \QQ^{k}\)，从而必有 \(k\ge r\)。
\par
对体量不等式，由定理 \ref{thm:xi-godel-loewner-exterior-box-loewner-holographic-weight}(3) 得
\[
\frac{1}{\binom{k}{t}}\log V_{k,t}(L)
=\frac{1}{\binom{k}{t}}\log\kappa_{\binom{k}{t}}+\frac12\log\binom{k}{t}-\log 2+\frac{t}{k}\sum_{i=1}^{k}\log L_i.
\]
其中前四项仅依赖 \((k,t)\)。
因此在固定 \((k,t)\) 下，最小化 \(\log V_{k,t}(L)\) 等价于最小化 \(\sum_{i=1}^{k}\log L_i=\sum_{i=1}^{k}\log\log q_i\)。
由素数序性质 \(q_i\ge p_i\)（逐项最小性）与单调性 \(x\mapsto \log\log x\)（\(x>1\)）递增，得
\(
\log\log q_i\ge \log\log p_i
\)
对每个 \(i\) 成立，求和即得所示不等式；逐项取等给出等号刻画。
\par
最后一条渐近式由定理 \ref{thm:xi-godel-prime-register-exterior-volume-quadratic-law} 取 \(k=r\) 直接得到。证毕。
\end{proof}

\begin{corollary}[面积密度非退化时回路秩的指数尺度]\label{cor:conclusion-boundary-cycle-rank-density-explosion}
取一列满射粗粒化 \(f_n:\Omega_n\twoheadrightarrow X_n\)，并记面积密度
\[
a_n:=\frac{A(f_n)}{n2^{n-1}}\in[0,1].
\]
若 \(\liminf_{n\to\infty}a_n>0\) 且 \(|X_n|/2^n\to 0\)，则回路秩
\[
r_n:=\operatorname{rank}_{\ZZ}H_1(\mathsf B_{f_n};\ZZ)=A(f_n)-|X_n|+c(\mathsf B_{f_n})
\]
满足
\[
\boxed{\ r_n=\Theta(n2^{n-1})\ }.
\]
因此在维度极限中，“边界可读出体积偏差”并不强制边界自由度消失；相反，在面积密度非退化时，残留自由度至少以指数规模增长。
从而不存在任何常数 \(M\) 能够对所有 \(n\) 同时给出上述模 \(p\) 自由度的统一单射外置编码。
\end{corollary}
\begin{proof}
由 \(c(\mathsf B_{f_n})\ge 1\) 与命题 \ref{prop:hypercube-boundary-multigraph-h1-cdim} 的秩公式得
\(
r_n\ge A(f_n)-|X_n|
=n2^{n-1}a_n-|X_n|
\)。
在假设下右端为 \(\Theta(n2^{n-1})\)。
另一方面 \(r_n\le |E(\mathsf B_{f_n})|=A(f_n)\le n2^{n-1}\)，故 \(r_n=\Theta(n2^{n-1})\)。证毕。
\end{proof}

\begin{corollary}[面积密度非退化下的回路秩饱和]\label{cor:conclusion-boundary-cycle-rank-saturation}
在推论 \ref{cor:conclusion-boundary-cycle-rank-density-explosion} 的假设下，记 \(A_n:=A(f_n)\)，则
\[
\frac{r_n}{A_n}\longrightarrow 1\qquad(n\to\infty).
\]
\end{corollary}
\begin{proof}
由 \(r_n=A_n-|X_n|+c(\mathsf B_{f_n})\) 且 \(c(\mathsf B_{f_n})\le |X_n|\) 得
\[
1-\frac{|X_n|}{A_n}\ \le\ \frac{r_n}{A_n}\ \le\ 1.
\]
又 \(A_n=n2^{n-1}a_n\) 且 \(\liminf a_n>0\)、\(|X_n|/2^n\to 0\) 蕴含
\[
\frac{|X_n|}{A_n}
=\frac{|X_n|/2^n}{a_n}\cdot\frac{2}{n}\longrightarrow 0,
\]
故夹逼得结论。证毕。
\end{proof}

\begin{theorem}[精确混合审计的秩守恒律]\label{thm:conclusion-boundary-mixed-audit-rank-conservation}
沿用推论 \ref{cor:conclusion-boundary-cycle-rank-external-info-lower-bound} 的记号，并固定素数 \(\wp\)。
考虑任意确定性混合审计协议：输入
\[
\omega\in C^1(\mathsf B_f;\FF_\wp)
\]
后，协议至多进行 \(q\) 次自适应回路查询，并输出一个外置寄存值
\[
R(\omega)\in \mathcal R.
\]
记该协议的完整查询转录为 \(\operatorname{Transcript}_q(\omega)\)。若二元数据
\[
\bigl(\operatorname{Transcript}_q(\omega),R(\omega)\bigr)
\]
能够唯一确定上同调类
\[
[\omega]\in H^1(\mathsf B_f;\FF_\wp),
\]
则必有
\[
q+\left\lceil \log_\wp |\mathcal R| \right\rceil \ge r,
\qquad
r=\dim_{\FF_\wp}H^1(\mathsf B_f;\FF_\wp).
\]
换言之，回路查询深度与外置寄存容量之间满足严格的加性秩守恒律。
\end{theorem}
\begin{proof}
把协议施用于零 \(1\)-上链 \(\omega=0\)。由于全部查询答案都为 \(0\)，协议沿一条确定的“全零转录”分支依次选出若干回路
\[
z_1,\dots,z_s\in Z_1(\mathsf B_f;\FF_\wp),
\qquad
s\le q.
\]
令 \(W\subset H_1(\mathsf B_f;\FF_\wp)\) 为这些回路类所张成的子空间，则
\[
\dim_{\FF_\wp}W\le q.
\]
考虑其消去子
\[
W^\perp
:=
\left\{
[\eta]\in H^1(\mathsf B_f;\FF_\wp):
\langle[\eta],z\rangle=0\ \ \forall z\in W
\right\}.
\]
由有限维线性代数，
\[
\dim_{\FF_\wp}W^\perp=r-\dim_{\FF_\wp}W\ge r-q.
\]

对任意 \([\eta]\in W^\perp\)，取其任一代表 \(\eta\in C^1(\mathsf B_f;\FF_\wp)\)。由于
\[
\langle[\eta],z_j\rangle=0
\qquad(1\le j\le s),
\]
协议在输入 \(\eta\) 时会沿与输入 \(0\) 完全相同的查询分支运行，因此
\[
\operatorname{Transcript}_q(\eta)=\operatorname{Transcript}_q(0).
\]
若
\[
\bigl(\operatorname{Transcript}_q(\omega),R(\omega)\bigr)
\]
要唯一确定上同调类，则映射
\[
[\eta]\longmapsto R(\eta)
\]
在 \(W^\perp\) 上必须为单射。于是
\[
|\mathcal R|\ge |W^\perp|=\wp^{\dim_{\FF_\wp}W^\perp}\ge \wp^{r-q}.
\]
取 \(\log_\wp\) 并上取整即得
\[
q+\left\lceil \log_\wp |\mathcal R| \right\rceil \ge r.
\]
证毕。
\end{proof}

\begin{corollary}[查询—寄存预算的指数守恒律]\label{cor:conclusion-boundary-query-register-budget-exponential-law}
在定理 \ref{thm:conclusion-boundary-mixed-audit-rank-conservation} 的设定下，若外置寄存器取值于任一满足
\[
|\mathcal R|=(E+1)^k
\]
的离散状态空间；特别地，可取正文中的截断 prime-register 空间 \(\mathcal P_{k,E}\)；则任意精确混合审计协议都满足
\[
q+k\log_\wp(E+1)\ge r.
\]
进一步，若存在一列满射粗粒化
\[
f_n:\Omega_n\twoheadrightarrow X_n
\]
满足推前熵非退化
\[
H(p_n)\ge h_0>0,
\qquad
|X_n|=o(2^n),
\]
则任意对应的精确混合审计族都满足
\[
q_n+k_n\log_\wp(E_n+1)=\Omega(2^n).
\]
\end{corollary}
\begin{proof}
第一式由定理 \ref{thm:conclusion-boundary-mixed-audit-rank-conservation} 与
\[
\log_\wp |\mathcal R|=\log_\wp (E+1)^k=k\log_\wp(E+1)
\]
直接得到。

对第二式，由推论 \ref{cor:spg-boundary-cycle-rank-from-entropy}，
\[
r(f_n)\ge \frac{2^n}{2\log 2}\,H(p_n)-|X_n|+1.
\]
在
\[
H(p_n)\ge h_0>0,
\qquad
|X_n|=o(2^n)
\]
下，右端为 \(\Omega(2^n)\)。再把
\[
q_n+k_n\log_\wp(E_n+1)\ge r(f_n)
\]
代入，即得所示指数下界。证毕。
\end{proof}
