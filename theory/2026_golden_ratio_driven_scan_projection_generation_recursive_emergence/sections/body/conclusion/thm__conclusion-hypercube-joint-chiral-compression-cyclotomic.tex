\begin{theorem}[互不相交仿射对合族的联合手征张量分解]\label{thm:conclusion-hypercube-joint-chiral-tensor-factorization}
\leanverified{paper\_conclusion\_hypercube\_joint\_chiral\_tensor\_factorization}
设
\[
\Omega_m=\{0,1\}^m,
\qquad
A_m=\text{超立方体 }Q_m\text{ 的邻接算子},
\]
并取两两不交的坐标对
\[
(i_1,j_1),\dots,(i_r,j_r)\subset [m],
\qquad
0\le r\le \left\lfloor\frac m2\right\rfloor.
\]
记相应仿射对合为 \(\sigma_a:=\sigma_{i_a,j_a}\)，并对符号向量
\[
\varepsilon=(\varepsilon_1,\dots,\varepsilon_r)\in\{\pm1\}^r
\]
定义联合特征子空间
\[
V_m^\varepsilon
:=
\bigcap_{a=1}^r
\{f\in \CC^{\Omega_m}:\ \sigma_a\cdot f=\varepsilon_a f\}.
\]
若 \(E_\pm\subset \CC^{\Omega_2}\) 表示单个二坐标仿射对合在局部二立方体上的 \(\pm1\)-特征子空间，则在适当重排坐标后存在规范同构
\[
\CC^{\Omega_m}
\cong
\CC^{\Omega_{m-2r}}\otimes (\CC^{\Omega_2})^{\otimes r},
\]
使得
\[
V_m^\varepsilon
\cong
\CC^{\Omega_{m-2r}}
\otimes
\bigotimes_{a=1}^r E_{\varepsilon_a}.
\]
特别地，若
\[
n_+(\varepsilon):=\#\{a:\varepsilon_a=+1\},
\qquad
n_-(\varepsilon):=\#\{a:\varepsilon_a=-1\},
\]
则
\[
\dim V_m^\varepsilon
=
2^{m-2r}3^{n_+(\varepsilon)}.
\]
\end{theorem}
\begin{proof}
由于各坐标对两两不交，诸 \(\sigma_a\) 两两对易，并且每个 \(\sigma_a\) 仅作用于对应的二坐标块。经重排坐标后，
\[
\Omega_m\cong \Omega_{m-2r}\times (\Omega_2)^r,
\]
故函数空间张量分解为
\[
\CC^{\Omega_m}
\cong
\CC^{\Omega_{m-2r}}\otimes (\CC^{\Omega_2})^{\otimes r}.
\]
联合特征条件逐块分离，于是
\[
V_m^\varepsilon
\cong
\CC^{\Omega_{m-2r}}
\otimes
\bigotimes_{a=1}^r E_{\varepsilon_a}.
\]
再由定理 \ref{thm:window6-affine-chiral-walsh-decomposition} 的单对合情形，
\[
\dim E_+=3,
\qquad
\dim E_-=1.
\]
故
\[
\dim V_m^\varepsilon
=
2^{m-2r}\cdot 3^{n_+(\varepsilon)}\cdot 1^{n_-(\varepsilon)}
=
2^{m-2r}3^{n_+(\varepsilon)}.
\]
证毕。
\end{proof}

\begin{theorem}[联合手征热迹的完全乘积分解]\label{thm:conclusion-hypercube-joint-chiral-heattrace-product}
\leanverified{paper\_conclusion\_hypercube\_joint\_chiral\_heattrace\_product}
在定理 \ref{thm:conclusion-hypercube-joint-chiral-tensor-factorization} 的设定下，定义
\[
Z_{m,\varepsilon}(t)
:=
\Tr_{V_m^\varepsilon}(e^{-t\Delta_m}),
\qquad
q:=e^{-2t}.
\]
则有闭式
\[
Z_{m,\varepsilon}(q)
=
q^{n_-(\varepsilon)}(1+q+q^2)^{n_+(\varepsilon)}(1+q)^{m-2r}.
\]
\end{theorem}
\begin{proof}
由
\[
Q_m\cong Q_{m-2r}\square Q_2^{\square r},
\]
拉普拉斯算子在张量分解上分裂为各因子的和，故热半群与热迹均乘法分离。自由坐标块 \(Q_1\) 的热迹因子为 \((1+q)\)。而由定理 \ref{thm:window6-chiral-heat-trace-phi3} 的单对合情形，每个局部负块贡献
\[
Z_-(q)=q,
\]
每个局部正块贡献
\[
Z_+(q)=1+q+q^2.
\]
于是把 \(m-2r\) 个自由因子、\(n_-(\varepsilon)\) 个负块与 \(n_+(\varepsilon)\) 个正块相乘，即得所示公式。证毕。
\end{proof}

\begin{theorem}[联合手征块的邻接谱生成函数]\label{thm:conclusion-hypercube-joint-chiral-spectrum-generator}
定义 Laurent 生成函数
\[
\Xi_{m,\varepsilon}(x)
:=
\sum_{\lambda\in\Spec(A_m|_{V_m^\varepsilon})}
\operatorname{mult}_{m,\varepsilon}(\lambda)x^\lambda.
\]
则有显式公式
\[
\Xi_{m,\varepsilon}(x)
=
(x+x^{-1})^{m-2r}(x^2+1+x^{-2})^{n_+(\varepsilon)}.
\]
特别地，
\[
\Spec(A_m|_{V_m^\varepsilon})
\subset
\{-m+2n_-(\varepsilon),-m+2n_-(\varepsilon)+2,\dots,m-2n_-(\varepsilon)\}.
\]
\end{theorem}
\begin{proof}
自由坐标块 \(Q_1\) 的邻接谱为 \(\{1,-1\}\)，故每个自由因子贡献 \((x+x^{-1})\)。对单个正块，由定理 \ref{thm:window6-chiral-heat-trace-phi3}
\[
Z_+(q)=1+q+q^2,
\]
可读出局部 Laplacian 谱为 \(\{0,2,4\}\)，故邻接谱为 \(\{2,0,-2\}\)，贡献因子 \((x^2+1+x^{-2})\)。单个负块由
\[
Z_-(q)=q
\]
可读出 Laplacian 谱只有 \(\{2\}\)，故邻接谱仅有 \(\{0\}\)，贡献因子 \(1\)。把各局部谱生成函数相乘，即得公式。

最后，正块只贡献 \(\{-2,0,2\}\) 的局部修正，而负块贡献零位移，自由块贡献 \(\{\pm1\}\)。于是全局谱支撑区间由这些局部极值相加得到所示包含关系。证毕。
\end{proof}

\begin{theorem}[全负手征塔的精确超立方体压缩]\label{thm:conclusion-hypercube-allnegative-exact-compression}
取全负符号
\[
\varepsilon_-:=(-1,\dots,-1)\in\{\pm1\}^r.
\]
则存在规范同构
\[
V_m^{\varepsilon_-}\cong \CC^{\Omega_{m-2r}},
\qquad
A_m|_{V_m^{\varepsilon_-}}\cong A_{m-2r},
\qquad
\Delta_m|_{V_m^{\varepsilon_-}}\cong \Delta_{m-2r}+2rI.
\]
从而
\[
\Spec(A_m|_{V_m^{\varepsilon_-}})
=
\{m-2r-2k:\ 0\le k\le m-2r\},
\]
其重数为 \(\binom{m-2r}{k}\)；并且当 \(r\ge 1\) 时，
\[
\det(\Delta_m|_{V_m^{\varepsilon_-}})
=
\prod_{k=0}^{m-2r}(2r+2k)^{\binom{m-2r}{k}}.
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-hypercube-joint-chiral-tensor-factorization} 在全负情形给出
\[
V_m^{\varepsilon_-}
\cong
\CC^{\Omega_{m-2r}}\otimes E_-^{\otimes r}
\cong
\CC^{\Omega_{m-2r}},
\]
因为每个 \(E_-\) 都是一维。另一方面，对互不相交的每一对坐标，可以逐次应用定理 \ref{thm:window6-chiral-compression-hypercube-adjacency}，每次都把邻接算子从 \(A_{m-2a}\) 压缩到 \(A_{m-2a-2}\)。迭代 \(r\) 次即得
\[
A_m|_{V_m^{\varepsilon_-}}\cong A_{m-2r}.
\]
再由 \(\Delta_m=mI-A_m\) 得
\[
\Delta_m|_{V_m^{\varepsilon_-}}
\cong
mI-A_{m-2r}
=
\Delta_{m-2r}+2rI.
\]
于是 Laplacian 谱为
\[
\{2r+2k:\ 0\le k\le m-2r\},
\]
重数为 \(\binom{m-2r}{k}\)；邻接谱与行列式公式随即成立。证毕。
\end{proof}

\begin{theorem}[轨道可见维与手征亏维的精确平衡]\label{thm:conclusion-involution-visible-chiral-dimension-balance}
设 \(\sigma\) 为任意有限集合 \(X\) 上的对合，其循环分解含 \(a\) 个不动点与 \(b\) 个二循环。则置换表示 \(\CC^X\) 的 \(\pm1\)-特征维数满足
\[
\dim V_\sigma^+=a+b,
\qquad
\dim V_\sigma^-=b.
\]
特别地，
\[
\dim V_\sigma^+=|X/\langle \sigma\rangle|.
\]
在 window-\(6\) 的 \(\sigma_{\mathrm{geo}}\) 情形，
\[
a=32,
\qquad
b=16,
\]
故
\[
\dim V^+=48,
\qquad
\dim V^-=16,
\qquad
64=48+16.
\]
\end{theorem}
\begin{proof}
每个不动点贡献一条 \(+1\)-特征直线；每个二循环在其二维张成空间中分解为一条对称线与一条反对称线，分别对应 \(+1\) 与 \(-1\) 特征。故总维数公式成立，而 \(a+b\) 正是轨道个数 \(|X/\langle\sigma\rangle|\)。

对 window-\(6\)，由推论 \ref{cor:terminal-foldbin6-geo-cycle-quotient-48}，\(\sigma_{\mathrm{geo}}\) 的循环型为 \(1^{32}2^{16}\)。代入即得最后三式。证毕。
\end{proof}

\begin{theorem}[stable-type 可观测量对整个手征函数演算轨道的严格失明]\label{thm:conclusion-window6-stabletype-blind-to-chiral-functional-calculus}
在 window-\(6\) 中，设
\[
g=\bar g\circ F_6
\]
仅依赖 stable type。则对任意 \(f\in V^-\) 与任意复系数多项式 \(P\)，都有
\[
\langle g,P(A_6)f\rangle=0,
\qquad
\langle g,P(\Delta_6)f\rangle=0.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:window6-chiral-orthogonality-to-type-observables}，任意 stable-type 可观测量 \(g\) 都与 \(V^-\) 正交。另一方面，由推论 \ref{cor:window6-chiral-sector-q4-spectrum}，

A_6|_{V^-}\cong A_4,

P(A_6)f\in V^-.
\]
于是
\[
\langle g,P(A_6)f\rangle=0.
\]
再由 \(\Delta_6=6I-A_6\)，\(P(\Delta_6)\) 同样保持 \(V^-\)，故第二式同理成立。证毕。
\end{proof}

\begin{theorem}[分圆缺口的 pair-by-pair 量子化]\label{thm:conclusion-hypercube-pairwise-cyclotomic-quantization}
对任意符号向量 \(\varepsilon\)，记全负向量为 \(\varepsilon_-\)。则有
\[
\frac{Z_{m,\varepsilon}(q)}{Z_{m,\varepsilon_-}(q)}
=
\left(\frac{1+q+q^2}{q}\right)^{n_+(\varepsilon)}
=
\left(\frac{\Phi_3(q)}{q}\right)^{n_+(\varepsilon)}.
\]
因此，在互不相交仿射对合族的联合手征分解中，每一个正块严格且仅严格贡献一个三次分圆单位 \(\Phi_3(q)/q\)；不存在其他 cyclotomic 因子的额外生成。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-hypercube-joint-chiral-heattrace-product}，
\[
Z_{m,\varepsilon}(q)
=
q^{n_-(\varepsilon)}(1+q+q^2)^{n_+(\varepsilon)}(1+q)^{m-2r},
\]
而
\[
Z_{m,\varepsilon_-}(q)
=
q^r(1+q)^{m-2r}.
\]
两式相除即得。证毕。
\end{proof}
