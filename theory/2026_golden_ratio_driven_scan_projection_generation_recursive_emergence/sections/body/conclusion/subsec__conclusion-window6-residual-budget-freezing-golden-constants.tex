\subsection{window-\texorpdfstring{$6$}{6} 残余异常预算、冻结谱离散曲率与黄金常数反演}\label{subsec:conclusion-window6-residual-budget-freezing-golden-constants}
在前述 window-\(6\) 的 boundary/cyclic 刚性分裂、冻结相高阶矩的仿射锁定以及黄金分支 Fibonacci 审计双相常数已经确立之后，还可把三条链条进一步压缩为一组相互啮合的终端后果：有限窗的 boundary sheet parity 零模在交换化中可被精确剥离，并留下恰与 cyclic 扇区同维的残余异常预算；冻结相的整数压力谱在阈值之后不仅进入严格仿射段、离散曲率完全消失，而且形成带截距补偿的精确半群律并把最小熵率钉扎到 \(g_*\)；而黄金传输误差的两条交错极限常数本身已经足以闭式反演 \(\varphi\)。

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的规范体积密度与折叠缺陷之间的精确对数缺口]\label{thm:conclusion-window6-gauge-defect-exact-log-gap}
沿用表 \ref{tab:fold_bin_gauge_m6_invariants} 与定理 \ref{thm:xi-foldbin-gauge-representation-distribution-law-m6-spectrum} 的记号。记
\[
G_6^{\mathrm{bin}}:=\mathcal G_6^{\mathrm{bin}}=\Aut(\Fold_6^{\mathrm{bin}}),
\qquad
g_6:=2^{-6}\log|G_6^{\mathrm{bin}}|,
\]
\[
\kappa_6:=
2^{-6}\sum_{w\in X_6}d_6^{\mathrm{bin}}(w)\log d_6^{\mathrm{bin}}(w),
\qquad
d_6^{\mathrm{bin}}(w):=\bigl|(\Fold_6^{\mathrm{bin}})^{-1}(w)\bigr|.
\]
则有严格闭式
\[
\boxed{\;
\kappa_6-g_6=\frac{49\log 2-\log 3}{64}.\;}
\]
\leanverified{window6\_gauge\_group\_factorial\_factors}
\leanverified{window6\_gauge\_group\_order}
\leanverified{paper\_window6\_gauge\_defect\_coefficient\_extraction}
\end{theorem}
\begin{proof}
由表 \ref{tab:fold_bin_gauge_m6_invariants}，window-\(6\) 的纤维直方图为
\[
2:8,\qquad 3:4,\qquad 4:9.
\]
另一方面，定理 \ref{thm:xi-foldbin-gauge-representation-distribution-law-m6-spectrum} 给出
\[
G_6^{\mathrm{bin}}\cong \mathfrak S_2^{\,8}\times \mathfrak S_3^{\,4}\times \mathfrak S_4^{\,9},
\]
故
\[
64\,g_6
=
\log|G_6^{\mathrm{bin}}|
=
8\log(2!)+4\log(3!)+9\log(4!).
\]
再由 \(\kappa_6\) 的定义，
\[
64\,\kappa_6
=
8(2\log 2)+4(3\log 3)+9(4\log 4).
\]
因此
\[
64(\kappa_6-g_6)
=
8(2\log 2-\log 2)
+4(3\log 3-\log 6)
+9(4\log 4-\log 24).
\]
逐项化简得到
\[
2\log 2-\log 2=\log 2,
\]
\[
3\log 3-\log 6=2\log 3-\log 2,
\]
\[
4\log 4-\log 24=5\log 2-\log 3.
\]
于是
\[
64(\kappa_6-g_6)
=
8\log 2+4(2\log 3-\log 2)+9(5\log 2-\log 3)
=
49\log 2-\log 3.
\]
两边同除以 \(64\) 即得结论。证毕。
\end{proof}

\begin{theorem}[冻结相后的矩谱半群线性化与最小熵钉扎]\label{thm:conclusion-frozen-moment-spectrum-semigroup-linearization}
\leanverified{paper\_conclusion\_frozen\_moment\_spectrum\_semigroup\_linearization}
在严格基态间隙
\[
\alpha_*^{(2)}<\alpha_*
\]
下，沿用定理 \ref{thm:fold-pressure-freezing-threshold} 与定理 \ref{thm:xi-fixed-freezing-escort-renyi-spectrum-collapse} 的记号，并记
\[
P_a:=\lim_{m\to\infty}\frac1m\log S_a(m),
\qquad
S_a(m):=\sum_{x\in X_m}d_m(x)^a,
\]
\[
a_{\mathrm c}:=\frac{\log\varphi-g_*}{\alpha_*-\alpha_*^{(2)}}.
\]
再定义 escort 分布
\[
\pi_{m,a}(x):=\frac{d_m(x)^a}{S_a(m)}.
\]
则对任意整数 \(a,b>a_{\mathrm c}\) 都有精确半群律
\[
\boxed{\;
P_{a+b}+g_*=P_a+P_b.\;}
\]
并且对任意整数 \(a>a_{\mathrm c}\) 有
\[
\boxed{\;
P_{a+1}-P_a=\alpha_*.\;}
\]
若最小熵率
\[
h_\infty(a):=\lim_{m\to\infty}\frac1m\Bigl(-\log\max_{x\in X_m}\pi_{m,a}(x)\Bigr)
\]
存在，则
\[
\boxed{\;
h_\infty(a)=g_*
\qquad (a>a_{\mathrm c}).\;}
\]
从而对任意整数 \(\ell\ge 1\) 与任意整数 \(a>a_{\mathrm c}\)，都有
\[
\boxed{\;
P_{a+\ell}-P_a=\ell\alpha_*,
\qquad
\lim_{m\to\infty}\frac1m\log\frac{S_{a+\ell}(m)}{S_a(m)}=\ell\alpha_*.\;}
\]
特别地，当 \(a>a_{\mathrm c}+1\) 时，
\[
P_{a+1}-2P_a+P_{a-1}=0.
\]
\leanverified{affine\_semigroup\_law}
\leanverified{paper\_conclusion\_frozen\_moment\_semigroup\_seeds}
\leanverified{paper\_conclusion\_frozen\_moment\_semigroup\_package}
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-pressure-freezing-threshold}，对任意实数 \(t>a_{\mathrm c}\) 都有
\[
P_t=t\alpha_*+g_*.
\]
因此对整数 \(a,b>a_{\mathrm c}\)，
\[
P_{a+b}+g_*
=
(a+b)\alpha_*+2g_*
=
(a\alpha_*+g_*)+(b\alpha_*+g_*)
=
P_a+P_b.
\]
同理，对整数 \(a>a_{\mathrm c}\)，
\[
P_{a+1}-P_a
=
\bigl((a+1)\alpha_*+g_*\bigr)-\bigl(a\alpha_*+g_*\bigr)
=
\alpha_*.
\]
再由定理 \ref{thm:xi-fixed-freezing-escort-renyi-spectrum-collapse} 在 \(q=\infty\) 时的结论，若最小熵率存在，则
\[
h_\infty(a)=g_*
\qquad (a>a_{\mathrm c}).
\]
把线性公式用于 \(a+\ell\) 与 \(a\) 即得
\[
P_{a+\ell}-P_a=\ell\alpha_*,
\]
而后式等价于
\[
\lim_{m\to\infty}\frac1m\log\frac{S_{a+\ell}(m)}{S_a(m)}
=
P_{a+\ell}-P_a
=
\ell\alpha_*.
\]
最后取 \(\ell=1\) 于 \(a-1,a,a+1\) 三点即可得到
\[
P_{a+1}-2P_a+P_{a-1}=0.
\]
证毕。
\end{proof}

G_6^{\mathrm{bin}}:=\Aut(\Fold_6^{\mathrm{bin}})=\mathcal G_6^{\mathrm{bin}},
\qquad
C_{\partial}:=(\ZZ/2\ZZ)^{X_6^{\mathrm{bdry}}}\subset (G_6^{\mathrm{bin}})^{\mathrm{ab}}.

\boxed{\;

\begin{corollary}[剥离 boundary sheet parity 零模后的最小忠实残余预算]\label{cor:conclusion-window6-minimal-faithful-residual-budget-18}
沿用定理 \ref{thm:conclusion-window6-boundary-quotient-cyclic-cardinality} 的记号。设
\[
\Lambda:(G_6^{\mathrm{bin}})^{\mathrm{ab}}\longrightarrow (\ZZ/2\ZZ)^r
\]
为任意 \(\FF_2\)-线性读出，且满足
\[
C_{\partial}\subseteq \ker\Lambda.
\]
若诱导映射
\[
\overline{\Lambda}:(G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}
\longrightarrow
(\ZZ/2\ZZ)^r
\]
仍为单射，则必有
\[
\boxed{\;
r\ge 18.\;}
\]
并且该下界可达，因而是最优的。
\end{corollary}
\leanverified{paper\_conclusion\_window6\_minimal\_faithful\_residual\_budget\_18}
\begin{proof}
由定理 \ref{thm:conclusion-window6-boundary-quotient-cyclic-cardinality}，
\[
(G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\cong (\ZZ/2\ZZ)^{18}
\]
作为 \(\FF_2\)-向量空间具有维数 \(18\)。
若 \(\overline{\Lambda}\) 为单射，则目标空间维数至少不小于定义域维数，从而
\[
r\ge 18.
\]
另一方面，取任意同构
\[
(G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\cong (\ZZ/2\ZZ)^{18}
\]
并与恒等映射合成，即可在 \(r=18\) 时实现单射。故该下界可达且最优。证毕。
\end{proof}

\begin{theorem}[黄金传输误差的交错极限常数显式反演 \texorpdfstring{$\varphi$}{phi}]\label{thm:conclusion-golden-alternating-constants-recover-phi}
沿用推论 \ref{cor:kronecker-w1-fibonacci-limits} 的记号。记
\[
c_-:=
\lim_{\substack{n\to\infty\\ \delta_n<0}}
F_n\,\mathsf T_{F_n}(\varphi^{-1}),
\qquad
c_+:=
\lim_{\substack{n\to\infty\\ \delta_n>0}}
F_n\,\mathsf T_{F_n}(\varphi^{-1}).
\]
则有显式反演公式
\[
\boxed{\;
\sqrt5
=
\frac{1}{c_- - c_+ + \frac1{15}},
\qquad
\varphi
=
\frac12\left(1+\frac{1}{c_- - c_+ + \frac1{15}}\right).\;}
\]
\leanverified{golden\_alternating\_recovery}
\leanverified{golden\_ratio\_from\_sqrt5}
\leanverified{paper\_conclusion\_golden\_alternating\_constants\_recover\_phi}
\end{theorem}
\begin{proof}
由推论 \ref{cor:kronecker-w1-fibonacci-limits}，
\[
c_-=\frac12+\frac{1}{2\sqrt5},
\qquad
c_+=\frac12-\frac{1}{2\sqrt5}+\frac1{15}.
\]
相减得到
\[
c_- - c_+
=
\left(\frac12+\frac{1}{2\sqrt5}\right)
-
\left(\frac12-\frac{1}{2\sqrt5}+\frac1{15}\right)
=
\frac1{\sqrt5}-\frac1{15}.
\]
因此
\[
\frac1{\sqrt5}=c_- - c_+ + \frac1{15},
\qquad
\sqrt5=\frac{1}{c_- - c_+ + \frac1{15}}.
\]
再代入
\[
\varphi=\frac{1+\sqrt5}{2}
\]
即得所述公式。证毕。
\end{proof}

\begin{corollary}[黄金分支 Fibonacci 审计常数的二相重整化刚性]\label{cor:conclusion-golden-fibonacci-two-phase-renormalization}
沿用定理 \ref{thm:conclusion-golden-alternating-constants-recover-phi} 的记号。则序列
\[
F_n\,\mathsf T_{F_n}(\varphi^{-1})
\]
不收敛，而恰有两个聚点
\[
\boxed{\;
c_-=\frac12+\frac{1}{2\sqrt5},
\qquad
c_+=\frac12-\frac{1}{2\sqrt5}+\frac1{15}.\;}
\]
其算术中心与围绕该中心的振幅分别为
\[
\boxed{\;
\frac{c_-+c_+}{2}=\frac{8}{15},
\qquad
\frac{c_--c_+}{2}=\frac{1}{2\sqrt5}-\frac1{30}.\;}
\]
因此，黄金分支的 Fibonacci 审计常数不能由单一标量重整化吸收；最小闭合的重整化对象是一条奇偶二相 cocycle。
\leanverified{golden\_alternating\_center}
\leanverified{golden\_alternating\_amplitude}
\leanverified{two\_phase\_distinct}
\leanverified{paper\_conclusion\_golden\_fibonacci\_two\_phase\_renormalization}
\end{corollary}
\begin{proof}
推论 \ref{cor:kronecker-w1-fibonacci-limits} 已给出
\[
c_-=\frac12+\frac{1}{2\sqrt5},
\qquad
c_+=\frac12-\frac{1}{2\sqrt5}+\frac1{15},
\]
从而全序列恰有这两个聚点。直接相加与相减得到
\[
\frac{c_-+c_+}{2}
=
\frac12\left(
\frac12+\frac{1}{2\sqrt5}
+\frac12-\frac{1}{2\sqrt5}+\frac1{15}
\right)
=
\frac{8}{15},
\]
以及
\[
\frac{c_--c_+}{2}
=
\frac12\left(
\frac1{\sqrt5}-\frac1{15}
\right)
=
\frac{1}{2\sqrt5}-\frac1{30}.
\]
证毕。
\end{proof}

以上结果把 window-\(6\) 的 boundary sheet parity、冻结相压力谱与黄金 Fibonacci 审计三条接口收束为同一闭合结构：boundary 零模的规范剥离把 \(21\) 维奇偶荷格精确还原为一个 \(18\) 维残余预算，并与 cyclic 扇区同维；冻结相在整数矩轴上不仅进入零离散曲率的严格仿射段，而且满足带截距补偿的精确半群律，并把最小熵率严格钉扎到同一常数 \(g_*\)；而黄金分支的交错极限常数则把 \(\varphi\) 本身编码为一条显式可反演的标量差值。于是，离散预算、冻结谱与黄金常数三端之间不再只是并列输出，而是构成一条可审计的互锁闭链。
