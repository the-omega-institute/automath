\begin{theorem}[特征 \texorpdfstring{$p$}{p} 下 Lucas 幂秩塌缩后的 shifted Hankel 行列式几何比率]\label{thm:conclusion-lucas-charp-shifted-hankel-geometric-ratio}
\leanverified{paper\_conclusion\_lucas\_charp\_shifted\_hankel\_geometric\_ratio}
在定理 \ref{thm:conclusion-charp-hankel-rank-classification} 的设定下，设
\[
\operatorname{char}(K)=p>q,\qquad
s:=\beta/\alpha,\qquad
m:=\operatorname{ord}(s)\le q,
\]
并假设所有同余类和
\[
C_a:=\sum_{\substack{0\le k\le q\\ k\equiv a\ (\mathrm{mod}\ m)}}\binom{q}{k}
\qquad(a=0,\dots,m-1)
\]
都非零。
则：
\begin{enumerate}
  \item 由定理 \ref{thm:conclusion-charp-hankel-rank-classification}，序列 \(a_n=L_n^q\) 的 Hankel 秩恰为 \(m\)；
  \item 若记
  \[
  \lambda_a:=\alpha^q s^a\qquad(a=0,\dots,m-1),
  \]
  则可取一个 \(m\)-维最小实现矩阵 \(\mathcal A_m\)，其谱恰为
  \[
  \mathrm{Spec}(\mathcal A_m)=\{\lambda_0,\lambda_1,\dots,\lambda_{m-1}\};
  \]
  \item 对任意 \(r\ge 0\)，定义 shifted Hankel 主块
  \[
  \widetilde H_m^{(r)}:=\bigl(a_{i+j+r}\bigr)_{0\le i,j\le m-1},
  \]
  则有精确几何比率
  \[
  \boxed{\;
  \det\!\bigl(\widetilde H_m^{(r)}\bigr)
  =
  \det\!\bigl(\widetilde H_m^{(0)}\bigr)\,
  \bigl(\alpha^{qm}s^{m(m-1)/2}\bigr)^r.\;}
  \]
\end{enumerate}
特别地：
\begin{enumerate}
  \item 若 \(m\) 为奇数，则
  \[
  s^{m(m-1)/2}=1,
  \qquad
  \det\!\bigl(\widetilde H_m^{(r)}\bigr)
  =
  \det\!\bigl(\widetilde H_m^{(0)}\bigr)\,\alpha^{qmr}.
  \]
  \item 若 \(m\) 为偶数且 \(s\) 为 primitive \(m\)-次根，则
  \[
  s^{m(m-1)/2}=(s^{m/2})^{m-1}=(-1)^{m-1}=-1,
  \]
  从而
  \[
  \det\!\bigl(\widetilde H_m^{(r)}\bigr)
  =
  \det\!\bigl(\widetilde H_m^{(0)}\bigr)\,(-\alpha^{qm})^r.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-charp-hankel-rank-classification}，
\[
a_n=\sum_{a=0}^{m-1}C_a(\alpha^q s^a)^n
=\sum_{a=0}^{m-1}C_a\lambda_a^n,
\]
且 \(\lambda_0,\dots,\lambda_{m-1}\) 两两不同、\(C_a\neq 0\)。
因此序列 \(a_n\) 是由 \(m\) 个互异指数节点生成的指数和，Hankel 秩等于 \(m\)，并可取最小实现矩阵 \(\mathcal A_m\) 与对角矩阵
\[
\mathrm{diag}(\lambda_0,\dots,\lambda_{m-1})
\]
相似，故其谱恰为 \(\{\lambda_a\}_{a=0}^{m-1}\)。

令
\[
V_m:=(\lambda_a^i)_{0\le i,a\le m-1}\in M_m(K),
\qquad
D_r:=\mathrm{diag}(C_0\lambda_0^r,\dots,C_{m-1}\lambda_{m-1}^r).
\]
则对任意 \(0\le i,j\le m-1\)，有
\[
(V_mD_rV_m^{\mathsf T})_{ij}
=
\sum_{a=0}^{m-1}\lambda_a^i(C_a\lambda_a^r)\lambda_a^j
=
\sum_{a=0}^{m-1}C_a\lambda_a^{i+j+r}
=
a_{i+j+r}.
\]
故
\[
\widetilde H_m^{(r)}=V_mD_rV_m^{\mathsf T}.
\]
取行列式得
\[
\det(\widetilde H_m^{(r)})
=
\det(V_m)^2\prod_{a=0}^{m-1}(C_a\lambda_a^r)
=
\det(\widetilde H_m^{(0)})\left(\prod_{a=0}^{m-1}\lambda_a\right)^r.
\]
而
\[
\prod_{a=0}^{m-1}\lambda_a
=
\prod_{a=0}^{m-1}\alpha^q s^a
=
\alpha^{qm}s^{\sum_{a=0}^{m-1}a}
=
\alpha^{qm}s^{m(m-1)/2}.
\]
代回即得所示精确几何比率。

若 \(m\) 为奇数，则 \(m\mid m(m-1)/2\)，故 \(s^{m(m-1)/2}=1\)。
若 \(m\) 为偶数且 \(s\) 为 primitive \(m\)-次根，则 \(s^{m/2}=-1\)，从而
\[
s^{m(m-1)/2}=(s^{m/2})^{m-1}=(-1)^{m-1}=-1.
\]
两种特例随即成立。证毕。
\end{proof}
