\begin{theorem}[window-\texorpdfstring{$6$}{6} 固定窗冻结律的完全闭式]\label{thm:conclusion-window6-fixed-window-freezing-closed}
\leanverified{paper\_conclusion\_window6\_fixed\_window\_freezing\_closed}
对 \(q\ge 0\) 定义
\[
S_q^{(6)}
:=
\sum_{w\in X_6}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q,
\qquad
P_q^{(6)}:=\log S_q^{(6)}.
\]
则有精确闭式
\[
\boxed{\;
S_q^{(6)}=8\cdot 2^q+4\cdot 3^q+9\cdot 4^q,\;}
\]
以及
\[
\boxed{\;
P_q^{(6)}
=
q\log 4+\log 9
+\log\!\left(
1+\frac49\left(\frac34\right)^q+\frac89\,2^{-q}
\right).\;}
\]
特别地，
\[
\boxed{\;
P_q^{(6)}-q\log 4-\log 9>0\qquad(q\ge 0),\;}
\]
并且当 \(q\to+\infty\) 时，
\[
\boxed{\;
P_q^{(6)}-q\log 4-\log 9
=
\frac49\left(\frac34\right)^q
+O\!\left(\left(\frac{9}{16}\right)^q\right).\;}
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:fold-bin-m6-fiber-hist}，window-\(6\) 的纤维直方图为
\[
2:8,\qquad 3:4,\qquad 4:9.
\]
于是直接得到
\[
S_q^{(6)}=8\cdot 2^q+4\cdot 3^q+9\cdot 4^q.
\]
把 \(9\cdot 4^q\) 提出便得
\[
P_q^{(6)}
=
\log\!\bigl(9\cdot 4^q\bigr)
+\log\!\left(
1+\frac49\left(\frac34\right)^q+\frac89\,2^{-q}
\right),
\]
即第二条盒装公式。

再记
\[
x_q:=\frac49\left(\frac34\right)^q,
\qquad
y_q:=\frac89\,2^{-q}.
\]
则 \(x_q,y_q>0\)，从而
\[
P_q^{(6)}-q\log 4-\log 9=\log(1+x_q+y_q)>0.
\]

最后，由 \(\log(1+t)=t+O(t^2)\)（\(t\to 0\)）得
\[
\log(1+x_q+y_q)=x_q+y_q+O\bigl((x_q+y_q)^2\bigr).
\]
而
\[
y_q=O\!\left(\left(\frac{9}{16}\right)^q\right),
\qquad
x_q^2=O\!\left(\left(\frac{9}{16}\right)^q\right),
\]
故
\[
\log(1+x_q+y_q)
=
x_q+O\!\left(\left(\frac{9}{16}\right)^q\right).
\]
代回即得所示渐近式。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} escort 质量的长短根反转临界指数]\label{thm:conclusion-window6-short-long-escort-crossover}
\leanverified{paper\_conclusion\_window6\_short\_long\_escort\_crossover}
沿用定理 \ref{thm:conclusion-window6-fixed-window-freezing-closed} 与定理 \ref{thm:conclusion-window6-short-long-degeneracy-splitting} 的记号，并记
\[
\mathcal B:=X_6^{\mathrm{bdry}}.
\]
定义 escort 分布
\[
\pi_q^{(6)}(w):=
\frac{\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q}{S_q^{(6)}}
\qquad(q\ge 0,\ w\in X_6),
\]
以及对任意 \(A\subseteq X_6\) 的聚合质量
\[
\pi_q^{(6)}(A):=\sum_{w\in A}\pi_q^{(6)}(w).
\]
则有精确公式
\[
\boxed{\;
\pi_q^{(6)}(\mathcal S)
=
\frac{5\cdot 4^q+2^q}{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q},\;}
\]
\[
\boxed{\;
\pi_q^{(6)}(\mathcal L)
=
\frac{4\cdot 4^q+4\cdot 3^q+4\cdot 2^q}{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q},\;}
\]
\[
\boxed{\;
\pi_q^{(6)}(\mathcal B)
=
\frac{3\cdot 2^q}{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q}.\;}
\]
若定义
\[
h(q):=4^q-4\cdot 3^q-3\cdot 2^q,
\]
则
\[
\pi_q^{(6)}(\mathcal S)-\pi_q^{(6)}(\mathcal L)
=
\frac{h(q)}{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q}.
\]
并且存在唯一实数 \(q_\ast\in(0,6)\) 使 \(h(q_\ast)=0\)；数值上
\[
\boxed{\;
q_\ast\approx 5.130144372342649.\;}
\]
相应地，
\[
\boxed{\;
\pi_q^{(6)}(\mathcal S)<\pi_q^{(6)}(\mathcal L)\quad(0\le q<q_\ast),\qquad
\pi_q^{(6)}(\mathcal S)>\pi_q^{(6)}(\mathcal L)\quad(q>q_\ast).\;}
\]
此外，
\[
\boxed{\;
\lim_{q\to\infty}\pi_q^{(6)}(\mathcal S)=\frac59,\qquad
\lim_{q\to\infty}\pi_q^{(6)}(\mathcal L)=\frac49,\qquad
\lim_{q\to\infty}\pi_q^{(6)}(\mathcal B)=0.\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-short-long-degeneracy-splitting}，
\[
\mathcal S:\ 1\times d=2,\ 5\times d=4,
\]
\[
\mathcal L:\ 4\times d=2,\ 4\times d=3,\ 4\times d=4,
\]
而 \(\mathcal B\) 中三条词全部落在 \(d=2\) 层。再由定理 \ref{thm:conclusion-window6-fixed-window-freezing-closed}，
\[
S_q^{(6)}=9\cdot 4^q+4\cdot 3^q+8\cdot 2^q.
\]
于是三条 escort 质量公式逐项代入即得。

两者之差则直接化简为
\[
\pi_q^{(6)}(\mathcal S)-\pi_q^{(6)}(\mathcal L)
=
\frac{(5\cdot 4^q+2^q)-(4\cdot 4^q+4\cdot 3^q+4\cdot 2^q)}
{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q}
=
\frac{h(q)}{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q}.
\]

为证明唯一性，定义归一化函数
\[
\widetilde h(q):=\frac{h(q)}{3^q}
=
\left(\frac43\right)^q-4-3\left(\frac23\right)^q.
\]
则
\[
\widetilde h'(q)
=
\left(\frac43\right)^q\log\frac43
+3\left(\frac23\right)^q\log\frac32
>0
\qquad(q\ge 0),
\]
故 \(\widetilde h\) 在 \([0,\infty)\) 上严格递增。又有
\[
\widetilde h(0)=-6,
\qquad
\widetilde h(6)=\frac{988}{729}>0.
\]
由于 \(h(q)\) 与 \(\widetilde h(q)\) 同号，故存在唯一 \(q_\ast\in(0,6)\) 使 \(h(q_\ast)=0\)，并且其符号分布正如定理所述。数值近似由该唯一根直接求得。

最后，把各分子分母同时除以 \(4^q\)，得到
\[
\pi_q^{(6)}(\mathcal S)
=
\frac{5+4^{-q}2^q}{9+4\left(\frac34\right)^q+8\cdot 2^{-q}}
\longrightarrow \frac59,
\]
\[
\pi_q^{(6)}(\mathcal L)
=
\frac{4+4\left(\frac34\right)^q+4\cdot 2^{-q}}{9+4\left(\frac34\right)^q+8\cdot 2^{-q}}
\longrightarrow \frac49,
\]
\[
\pi_q^{(6)}(\mathcal B)
=
\frac{3\cdot 2^{-q}}{9+4\left(\frac34\right)^q+8\cdot 2^{-q}}
\longrightarrow 0.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的 cyclic escort 一阶矩与轴向极化]\label{thm:conclusion-window6-cyclic-escort-first-moment}

则
\[
\overline{\mathfrak m}_{6,\mathrm{cyc}}(q)
=
\frac{(4^q-2^q)e_1}{5\cdot 2^q+4\cdot 3^q+9\cdot 4^q},

d_6=2:\quad \{-e_1\}\sqcup\{(0,\pm1,\pm1)\}.

\qquad
\sum_{\alpha\in\{-e_1\}}\alpha=-e_1.

归一化公式由定义立即得到。最后把分子分母同除以 \(4^q\)，便有
\[
\overline{\mathfrak m}_{6,\mathrm{cyc}}(q)
=
\frac{(1-2^{-q})e_1}{9+4\left(\frac34\right)^q+5\cdot 2^{-q}}
\longrightarrow \frac19\,e_1.
\]
证毕。
\end{proof}

\frac{2+5\cdot 4}{6}
=
\frac{11}{3},
\qquad
\mathbb E_{\mathcal L}[d_6^{\mathrm{bin}}]
=
\frac{4(2+3+4)}{12}
=3.
\]
同理，
\[
\mathbb E_{\mathcal S}[\log d_6^{\mathrm{bin}}]
=
\frac{\log 2+5\log 4}{6},
\]
\[
\mathbb E_{\mathcal L}[\log d_6^{\mathrm{bin}}]
=
\frac{4(\log 2+\log 3+\log 4)}{12}
=
\frac13(\log 2+\log 3+\log 4).
\]
两式相减即得
\[
\mathbb E_{\mathcal S}[\log d_6^{\mathrm{bin}}]
-\mathbb E_{\mathcal L}[\log d_6^{\mathrm{bin}}]
=
\frac16\log\frac{32}{9}>0.
\]
证毕。
\end{proof}
