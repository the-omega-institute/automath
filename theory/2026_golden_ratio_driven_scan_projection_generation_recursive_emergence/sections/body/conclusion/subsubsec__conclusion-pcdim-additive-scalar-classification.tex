\paragraph{加性标量化的刚性障碍与可和权重分类}
命题 \ref{prop:conclusion-pcdim-max-law} 表明 \(\operatorname{pcdim}\) 服从“最大值法则”而非加法法则。
若进一步要求“单一标量 + 直积可加”，则可得到如下结构结论。

\begin{theorem}[等权加性标量的无穷爆炸障碍]\label{thm:conclusion-additive-scalar-procircle-obstruction}
\leanverified{paper\_conclusion\_additive\_scalar\_procircle\_obstruction}
设
\[
\mathcal P_{\mathrm{fin}}
:=\left\{\prod_{p\in S}\ZZ_p^{\,m_p}\;:\;S\subseteq\mathcal P_{\mathrm{pr}}\ \text{有限},\ m_p\in\NN\right\},
\qquad
\widehat{\ZZ}:=\prod_{p}\ZZ_p.
\]
令
\[
D:\mathcal P_{\mathrm{fin}}\cup\{\widehat{\ZZ}\}\to[0,+\infty]
\]
满足加性
\[
D(G\times H)=D(G)+D(H).
\]
若存在常数 \(c_0>0\) 与无限素数集合 \(S_\infty\subseteq\mathcal P_{\mathrm{pr}}\)，使
\[
D(\ZZ_p)\ge c_0\qquad(\forall\,p\in S_\infty),
\]
则
\[
D(\widehat{\ZZ})=+\infty.
\]
因此，在“无限多素数轴上等权非退化”的前提下，有限标量加性方案不可能成立。
\end{theorem}
\begin{proof}
取互异素数序列 \((p_k)_{k\ge 1}\subseteq S_\infty\)，并记
\[
G_n:=\prod_{k=1}^{n}\ZZ_{p_k}.
\]
由加性得
\[
D(G_n)=\sum_{k=1}^{n}D(\ZZ_{p_k})\ge n\,c_0.
\]
另一方面，\(\widehat{\ZZ}\cong G_n\times H_n\)（\(H_n:=\prod_{p\notin\{p_1,\dots,p_n\}}\ZZ_p\)），故
\[
D(\widehat{\ZZ})=D(G_n)+D(H_n)\ge D(G_n)\ge n\,c_0.
\]
令 \(n\to\infty\) 即得 \(D(\widehat{\ZZ})=+\infty\)。证毕。
\end{proof}

\leanverified{paper\_conclusion\_additive\_scalar\_procircle\_l1\_classification}
\begin{theorem}[有限标量加性 pro-圆维的 \texorpdfstring{$\ell^1$}{l1} 分类]\label{thm:conclusion-additive-scalar-procircle-l1-classification}
定义
\[
\mathcal P_{\mathrm{bnd}}
:=\left\{K=\prod_{p}\ZZ_p^{\,m_p}\;:\;m_p\in\NN,\ \sup_p m_p<\infty\right\}.
\]
设
\[
D:\mathcal P_{\mathrm{bnd}}\to[0,+\infty)
\]
满足：
\begin{enumerate}
  \item \textbf{加性：}\(D(K\times L)=D(K)+D(L)\)；
  \item \textbf{有限扭结归零：}对有限群因子 \(F\) 有 \(D(F)=0\)；
  \item \textbf{有限截断连续性：}若 \(K^{(\le n)}:=\prod_{k=1}^{n}\ZZ_{p_k}^{\,m_{p_k}}\)（\(p_k\) 为第 \(k\) 个素数），则
  \[
  D(K)=\lim_{n\to\infty}D\!\bigl(K^{(\le n)}\bigr).
  \]
\end{enumerate}
则存在唯一序列 \(w=(w_p)_{p}\in\ell^1(\mathcal P_{\mathrm{pr}})\) 使
\[
w_p=D(\ZZ_p),\qquad
D\!\Bigl(\prod_p\ZZ_p^{\,m_p}\Bigr)=\sum_{p}m_p\,w_p
\quad(\forall\,K\in\mathcal P_{\mathrm{bnd}}).
\]
\end{theorem}
\begin{proof}
令 \(w_p:=D(\ZZ_p)\)。
对任意有限素数集 \(S\) 与有限积
\(
K_S=\prod_{p\in S}\ZZ_p^{\,m_p}
\)，由加性直接得到
\[
D(K_S)=\sum_{p\in S}m_p\,w_p.
\]
取常数剖面 \(K^{(M)}:=\prod_p\ZZ_p^{\,M}\in\mathcal P_{\mathrm{bnd}}\)（\(M\in\NN\)），由截断连续性
\[
D(K^{(M)})
=\lim_{n\to\infty}D\!\Bigl(\prod_{k=1}^{n}\ZZ_{p_k}^{\,M}\Bigr)
=\lim_{n\to\infty}M\sum_{k=1}^{n}w_{p_k}
<\infty.
\]
故 \(\sum_p w_p<\infty\)，即 \(w\in\ell^1\)。

再取一般 \(K=\prod_p\ZZ_p^{\,m_p}\in\mathcal P_{\mathrm{bnd}}\)，设 \(m_p\le M\)。
由有限截断连续性与有限积公式，
\[
D(K)
=\lim_{n\to\infty}\sum_{k=1}^{n}m_{p_k}w_{p_k}
=\sum_{p}m_p\,w_p,
\]
且该级数因 \(|m_p w_p|\le M|w_p|\) 与 \(w\in\ell^1\) 绝对收敛。
唯一性由 \(w_p=D(\ZZ_p)\) 立即给出。证毕。
\end{proof}

\begin{corollary}[可和权重的有限标量替代与可组合性]\label{cor:conclusion-additive-scalar-procircle-weighted-model}
\leanverified{paper\_conclusion\_additive\_scalar\_procircle\_weighted\_model}
任取 \(w=(w_p)_p\in\ell^1(\mathcal P_{\mathrm{pr}})\) 且 \(w_p>0\)。
对
\(
K=\prod_p\ZZ_p^{\,m_p}\in\mathcal P_{\mathrm{bnd}}
\)
定义
\[
\operatorname{pcdim}_w(K):=\sum_p m_p\,w_p.
\]
则：
\begin{enumerate}
  \item \(\operatorname{pcdim}_w\) 对直积严格可加，且 \(\operatorname{pcdim}_w(\ZZ_p)=w_p>0\)；
  \item \(\operatorname{pcdim}_w(K)\) 对所有 \(K\in\mathcal P_{\mathrm{bnd}}\) 都有限；
  \item 在定理 \ref{thm:conclusion-additive-scalar-procircle-l1-classification} 的公理下，任意有限标量加性方案都与某个 \(\operatorname{pcdim}_w\) 等价。
\end{enumerate}
该构造与定理 \ref{thm:conclusion-additive-scalar-procircle-obstruction} 相容：
有限性来自 \(\ell^1\) 可和权重，而非“无限多素数轴上的统一正下界”。
\end{corollary}
\begin{proof}
由定义直接得 (1) 与 (2)。
第 (3) 即定理 \ref{thm:conclusion-additive-scalar-procircle-l1-classification} 的唯一性结论。证毕。
\end{proof}
