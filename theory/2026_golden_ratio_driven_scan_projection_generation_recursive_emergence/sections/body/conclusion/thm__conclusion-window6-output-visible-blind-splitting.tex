\begin{theorem}[window-\texorpdfstring{$6$}{6} 输出律在 exact-gcd 分层上的双标量畸变律]\label{thm:conclusion-window6-output-exact-gcd-two-scalar-distortion}
沿用定理 \ref{thm:conclusion-window6-visible-crt-arithmetic-phase-space} 与定理 \ref{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer} 的记号，记
\[
R_6:=\ZZ/(21\ZZ),
\qquad
p_6(r):=\frac{d(r)}{64},
\qquad
u_6(r):=\frac1{21},
\]
并对每个 \(\delta\mid 21\) 记
\[
S_\delta:=\{r\in R_6:(r,21)=\delta\}.
\]
则 \(p_6\) 在 exact-gcd 分层上的推前质量满足
\[
p_6(S_1)=\frac{9}{16},\qquad
p_6(S_3)=\frac{9}{32},\qquad
p_6(S_7)=\frac{3}{32},\qquad
p_6(S_{21})=\frac{1}{16},
\]
而均匀基线满足
\[
u_6(S_1)=\frac{4}{7},\qquad
u_6(S_3)=\frac{2}{7},\qquad
u_6(S_7)=\frac{2}{21},\qquad
u_6(S_{21})=\frac{1}{21}.
\]
因此
\[
\boxed{\;
\frac{p_6(S_\delta)}{u_6(S_\delta)}
=
\begin{cases}
\frac{21}{16},& \delta=21,\\[4pt]
\frac{63}{64},& \delta\in\{1,3,7\},
\end{cases}\;}
\]
等价地，\(p_6\) 到 Ramanujan-visible 代数 \(\mathcal A_{21}\) 的正交投影恰为
\[
\boxed{\;
p_6^{\mathrm{vis}}(r)=\frac{3+\mathbf 1_{\{0\}}(r)}{64}.\;}
\]
\leanverified{paper\_conclusion\_window6\_output\_exact\_gcd\_two\_scalar\_distortion}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-hidden-multiplicity-neutrality-on-nonzero-strata}，
\[
d(r)=
\begin{cases}
4,&0\le r\le 8,\\
3,&9\le r\le 12,\\
2,&13\le r\le 20.
\end{cases}
\]
在 \(R_6\cong \ZZ/(21\ZZ)\) 中，
\[
S_{21}=\{0\},\qquad
S_7=\{7,14\},\qquad
S_3=\{3,6,9,12,15,18\},
\]
其余 \(12\) 个元素构成 \(S_1\)。于是
\[
\sum_{r\in S_1}d(r)=5\cdot 4+2\cdot 3+5\cdot 2=36,
\]
\[
\sum_{r\in S_3}d(r)=2\cdot 4+2\cdot 3+2\cdot 2=18,
\]
\[
\sum_{r\in S_7}d(r)=4+2=6,
\qquad
\sum_{r\in S_{21}}d(r)=4.
\]
两边除以 \(64\) 即得 \(p_6\) 的四个推前质量。对均匀分布 \(u_6\)，四个扇区质量仅由基数给出，故
\[
u_6(S_1)=\frac{12}{21}=\frac47,\quad
u_6(S_3)=\frac{6}{21}=\frac27,\quad
u_6(S_7)=\frac{2}{21},\quad
u_6(S_{21})=\frac1{21}.
\]
逐项相除即得双标量畸变律。

再由定理 \ref{thm:conclusion-window6-hidden-rank-ramanujan-visible-orthogonal-projection} 中对 \(d\) 的投影公式
\[
P_{21}d=3+\mathbf 1_{S_{21}},
\]
而 \(p_6=d/64\)，故
\[
p_6^{\mathrm{vis}}:=P_{21}p_6=\frac{3+\mathbf 1_{\{0\}}}{64}.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 二次非均匀性的 Ramanujan-visible / blind 正交分裂]\label{thm:conclusion-window6-output-chi2-visible-blind-orthogonal-splitting}
沿用上一定理的记号，并定义 blind 残差
\[
\eta:=p_6-p_6^{\mathrm{vis}}.
\]
则
\[
\eta(0)=0,\qquad
\eta(r)=\frac1{64}\ (1\le r\le 8),\qquad
\eta(r)=0\ (9\le r\le 12),\qquad
\eta(r)=-\frac1{64}\ (13\le r\le 20),
\]
且
\[
\eta\perp \mathcal A_{21}
\]
相对于计数内积
\[
\langle f,g\rangle:=\sum_{r\in R_6}f(r)\overline{g(r)}.
\]
因此成立精确毕达哥拉斯分裂
\[
\boxed{\;
\|p_6-u_6\|_2^2
=
\|p_6^{\mathrm{vis}}-u_6\|_2^2+\|\eta\|_2^2
=
\frac{5}{21504}+\frac{84}{21504}
=
\frac{89}{21504}.\;}
\]
等价地，
\[
\boxed{\;
\chi^2(p_6\mid u_6)
=
\frac{5}{1024}+\frac{84}{1024}
=
\frac{89}{1024}.\;}
\]
故在 window-\(6\) 中，Ramanujan-visible 部分只占全部 \(\chi^2\) 缺口的 \(5/89\)，而 blind 部分精确占 \(84/89\)。
\leanverified{paper\_conclusion\_window6\_output\_chi2\_visible\_blind\_orthogonal\_splitting}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-output-exact-gcd-two-scalar-distortion}，
\[
p_6^{\mathrm{vis}}(r)=\frac{3+\mathbf 1_{\{0\}}(r)}{64}.
\]
而 \(p_6(r)=d(r)/64\) 且 \(d(r)\) 只取 \(4,3,2\)，故直接相减即可得到 \(\eta\) 的显式分段公式。

由定理 \ref{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer}，\(\mathcal A_{21}\) 恰为按 \(S_1,S_3,S_7,S_{21}\) 分层常值的函数代数。于是只需在每个扇区上验证 \(\eta\) 的和为零：
\[
\sum_{r\in S_1}\eta(r)=5\cdot\frac1{64}+2\cdot 0+5\cdot\Bigl(-\frac1{64}\Bigr)=0,
\]
\[
\sum_{r\in S_3}\eta(r)=2\cdot\frac1{64}+2\cdot 0+2\cdot\Bigl(-\frac1{64}\Bigr)=0,
\]
\[
\sum_{r\in S_7}\eta(r)=\frac1{64}-\frac1{64}=0,
\qquad
\sum_{r\in S_{21}}\eta(r)=0.
\]
故 \(\eta\perp \mathcal A_{21}\)。又 \(u_6\) 为常值函数，故 \(u_6\in \mathcal A_{21}\)，从而
\[
p_6-u_6=(p_6^{\mathrm{vis}}-u_6)+\eta
\]
是正交分解。

接着计算范数。因为
\[
p_6^{\mathrm{vis}}(0)-u_6(0)=\frac1{16}-\frac1{21}=\frac{5}{336},
\qquad
p_6^{\mathrm{vis}}(r)-u_6(r)=\frac3{64}-\frac1{21}=-\frac1{1344}\quad(r\neq 0),
\]
故
\[
\|p_6^{\mathrm{vis}}-u_6\|_2^2
=
\Bigl(\frac{5}{336}\Bigr)^2+20\Bigl(\frac{1}{1344}\Bigr)^2
=
\frac{5}{21504}.
\]
另一方面，
\[
\|\eta\|_2^2
=
8\Bigl(\frac1{64}\Bigr)^2+8\Bigl(\frac1{64}\Bigr)^2
=
\frac1{256}
=
\frac{84}{21504}.
\]
正交毕达哥拉斯恒等式给出
\[
\|p_6-u_6\|_2^2=\frac{89}{21504}.
\]
再由
\[
\chi^2(p_6\mid u_6)=21\,\|p_6-u_6\|_2^2
\]
便得
\[
\chi^2(p_6\mid u_6)=\frac{89}{1024},
\]
并同步得到可见/盲分量的占比。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 相对熵的 exact-gcd 分层精确分裂]\label{thm:conclusion-window6-output-kl-exact-gcd-chain-splitting}
对分层
\[
\mathscr P:=\{S_1,S_3,S_7,S_{21}\},
\]
记 \(p_{6,\ast},u_{6,\ast}\) 为 \(p_6,u_6\) 向 \(\mathscr P\) 的推前分布。则
\[
\boxed{\;
D_{\mathrm{KL}}(p_6\mid u_6)
=
D_{\mathrm{KL}}(p_{6,\ast}\mid u_{6,\ast})
\sum_{\delta\mid 21}
p_{6,\ast}(S_\delta)\,
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_\delta)\,\big\|\,u_6(\cdot\mid S_\delta)\bigr).\;}
\]
并且这两部分分别可化为
\[
\boxed{\;
D_{\mathrm{vis}}
:=
D_{\mathrm{KL}}(p_{6,\ast}\mid u_{6,\ast})
=
\frac{15}{16}\log\frac{63}{64}
\frac{1}{16}\log\frac{21}{16},\;}
\]
\[
\boxed{\;
D_{\mathrm{blind}}
:=
\sum_{\delta\mid 21}
p_{6,\ast}(S_\delta)\,
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_\delta)\,\big\|\,u_6(\cdot\mid S_\delta)\bigr)
=
\frac14\log\frac{32}{27}.\;}
\]
从而
\[
\boxed{\;
D_{\mathrm{KL}}(p_6\mid u_6)
=
\frac{15}{16}\log\frac{63}{64}
\frac{1}{16}\log\frac{21}{16}
\frac14\log\frac{32}{27}.\;}
\]
\end{theorem}
\begin{proof}
相对熵关于有限分层满足标准链式法则，这给出第一式。

再由定理 \ref{thm:conclusion-window6-output-exact-gcd-two-scalar-distortion}，在三个非零扇区上都有统一粗粒化畸变因子
\[
\frac{p_6(S_\delta)}{u_6(S_\delta)}=\frac{63}{64}
\qquad(\delta\in\{1,3,7\}),
\]
而在零扇区上
\[
\frac{p_6(S_{21})}{u_6(S_{21})}=\frac{21}{16}.
\]
因此
\[
D_{\mathrm{vis}}
=
\frac{9}{16}\log\frac{63}{64}
+\frac{9}{32}\log\frac{63}{64}
+\frac{3}{32}\log\frac{63}{64}
+\frac{1}{16}\log\frac{21}{16},
\]
即
\[
D_{\mathrm{vis}}
=
\frac{15}{16}\log\frac{63}{64}
\frac{1}{16}\log\frac{21}{16}.
\]

现在计算扇区内部的条件项。在 \(S_1\) 上，\(d=4,3,2\) 的出现次数分别为 \(5,2,5\)，故
\[
p_6(\cdot\mid S_1)=
\Bigl(\underbrace{\tfrac19,\dots,\tfrac19}_{5},
\underbrace{\tfrac1{12},\tfrac1{12}}_{2},
\underbrace{\tfrac1{18},\dots,\tfrac1{18}}_{5}\Bigr),
\qquad
u_6(\cdot\mid S_1)\equiv \frac1{12}.
\]
于是
\[
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_1)\,\big\|\,u_6(\cdot\mid S_1)\bigr)
=
5\Bigl(\frac19\log\frac43+\frac1{18}\log\frac23\Bigr)
=
\frac{5}{18}\log\frac{32}{27}.
\]
同理，在 \(S_3\) 上三种重数各出现 \(2\) 次，从而
\[
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_3)\,\big\|\,u_6(\cdot\mid S_3)\bigr)
=
2\Bigl(\frac29\log\frac43+\frac19\log\frac23\Bigr)
=
\frac29\log\frac{32}{27}.
\]
而在 \(S_7\) 上条件分布为 \((2/3,1/3)\)，均匀基线为 \((1/2,1/2)\)，故
\[
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_7)\,\big\|\,u_6(\cdot\mid S_7)\bigr)
=
\frac23\log\frac43+\frac13\log\frac23
=
\frac13\log\frac{32}{27}.
\]
扇区 \(S_{21}\) 只含一点，其条件项为 \(0\)。

最后乘以各扇区质量
\[
p_6(S_1)=\frac9{16},\qquad
p_6(S_3)=\frac9{32},\qquad
p_6(S_7)=\frac3{32},
\]
得
\[
D_{\mathrm{blind}}
=
\frac9{16}\cdot \frac{5}{18}\log\frac{32}{27}
\frac9{32}\cdot \frac29\log\frac{32}{27}
\frac3{32}\cdot \frac13\log\frac{32}{27}
=
\frac14\log\frac{32}{27}.
\]
于是得到总公式。证毕。
\end{proof}
\leanverified{paper\_conclusion\_window6\_output\_kl\_exact\_gcd\_chain\_splitting}

\begin{corollary}[非零 conductor 扇区的统一盲熵量子]\label{cor:conclusion-window6-output-universal-blind-entropy-quantum}
\leanverified{paper\_conclusion\_window6\_output\_universal\_blind\_entropy\_quantum}
记
\[
Q:=\log\frac{32}{27}.
\]
则三个非零 conductor 扇区的条件相对熵满足
\[
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_1)\,\big\|\,u_6(\cdot\mid S_1)\bigr)=\frac{5}{18}Q,
\]
\[
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_3)\,\big\|\,u_6(\cdot\mid S_3)\bigr)=\frac{2}{9}Q,
\]
\[
D_{\mathrm{KL}}\!\bigl(p_6(\cdot\mid S_7)\,\big\|\,u_6(\cdot\mid S_7)\bigr)=\frac{1}{3}Q.
\]
因此，window-\(6\) 非零 conductor 扇区的全部局部 blind 熵缺口都由同一个量子 \(Q\) 支配，其差异仅体现为有理系数。
\end{corollary}
\begin{proof}
三条公式已在定理 \ref{thm:conclusion-window6-output-kl-exact-gcd-chain-splitting} 的证明中逐项算出。共同因子 \(Q\) 的出现，源于每个非零扇区内相对于局部均匀基线的相对权重都只取
\[
\frac43,\qquad 1,\qquad \frac23
\]
三值。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} 的主非均匀性严格驻留于 Ramanujan-blind 通道]\label{cor:conclusion-window6-output-blind-channel-dominates}
\leanverified{paper\_conclusion\_window6\_output\_blind\_channel\_dominates}
在 window-\(6\) 上，二次偏离与相对熵都满足严格的 blind 主导：
\[
\frac{\chi^2_{\mathrm{blind}}}{\chi^2_{\mathrm{total}}}
=
\frac{84}{89},
\]
\[
\frac{D_{\mathrm{blind}}}{D_{\mathrm{KL}}(p_6\mid u_6)}
=
\frac{\frac14\log(32/27)}
{\frac{15}{16}\log(63/64)+\frac{1}{16}\log(21/16)+\frac14\log(32/27)}.
\]
特别地，
\[
\frac{\chi^2_{\mathrm{blind}}}{\chi^2_{\mathrm{total}}}\approx 0.9438,
\qquad
\frac{D_{\mathrm{blind}}}{D_{\mathrm{KL}}(p_6\mid u_6)}\approx 0.9501.
\]
故 window-\(6\) 中超过 \(94\%\) 的二次非均匀性与超过 \(95\%\) 的熵缺口都不在 Ramanujan-visible 层，而驻留于其正交 blind 残差。
\end{corollary}
\begin{proof}
第一式由定理 \ref{thm:conclusion-window6-output-chi2-visible-blind-orthogonal-splitting} 的精确分裂立即得到。第二式则由定理 \ref{thm:conclusion-window6-output-kl-exact-gcd-chain-splitting} 的闭式直接相除。证毕。
\end{proof}

\begin{conjecture}[平方自由可见模上的 Ramanujan-visible / blind 主导原理]\label{conj:conclusion-window6-output-ramanujan-visible-blind-dominance}
设某一分辨率 \(m\) 的可见商半环满足
\[
\Omega_m/\!\equiv_m \cong \ZZ/(F_{m+2}\ZZ),
\]
并且 \(F_{m+2}\) 平方自由。记 \(p_m\) 为 bin-fold 输出律，\(u_m\) 为可见空间上的均匀分布，\(\mathcal A_m\) 为由全部 \(c_q\)（\(q\mid F_{m+2}\)）张成的 Ramanujan 代数。则应存在规范正交分裂
\[
p_m-u_m=(p_m-u_m)_{\mathrm{vis}}+(p_m-u_m)_{\mathrm{blind}},
\qquad
(p_m-u_m)_{\mathrm{vis}}\in \mathcal A_m,
\quad
(p_m-u_m)_{\mathrm{blind}}\perp \mathcal A_m,
\]
并且对全部充分大的已审计平方自由窗口，应有严格主导
\[
\chi^2_{\mathrm{blind}}>\chi^2_{\mathrm{vis}},
\qquad
D_{\mathrm{blind}}>D_{\mathrm{vis}}.
\]
更强地，\((p_m-u_m)_{\mathrm{vis}}\) 应只取决于 exact-gcd 分层，而绝大部分统计粗糙度由与全部 Ramanujan 可见扇区正交的 Boolean-tail 残差承载。
\end{conjecture}
