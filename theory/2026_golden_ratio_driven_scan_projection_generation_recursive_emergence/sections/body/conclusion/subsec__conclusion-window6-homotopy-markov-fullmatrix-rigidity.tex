\begin{theorem}[window-\texorpdfstring{$6$}{6} 的奇维有理同伦秩完全反演尾部三分层]\label{thm:conclusion-window6-rational-homotopy-inverts-tail-trichotomy}
记
\[
r_3:=\operatorname{rank}_{\QQ}\bigl(\pi_3(K_6)\otimes\QQ\bigr),
\qquad
r_5:=\operatorname{rank}_{\QQ}\bigl(\pi_5(K_6)\otimes\QQ\bigr),
\qquad
r_7:=\operatorname{rank}_{\QQ}\bigl(\pi_7(K_6)\otimes\QQ\bigr).
\]
则 window-\(6\) 的三类尾部机制可由三元组 \((r_3,r_5,r_7)\) 唯一恢复：
\[
\#\{w\in X_6:\ w_6=1\}=r_3-r_5,
\]
\[
\#\{w\in X_6:\ w_6=0,\ V_6(w)>8\}=r_5-r_7,
\]
\[
\#\{w\in X_6:\ w_6=0,\ V_6(w)\le 8\}=r_7.
\]
特别地，由
\[
(r_3,r_5,r_7)=(21,13,9)
\]
立即推出
\[
\#\{w\in X_6:\ w_6=1\}=8,
\qquad
\#\{w\in X_6:\ w_6=0,\ V_6(w)>8\}=4,
\qquad
\#\{w\in X_6:\ w_6=0,\ V_6(w)\le 8\}=9.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-foldbin-window6-rational-homotopy-poincare}，
\[
K_6\cong PU(2)^8\times PU(3)^4\times PU(4)^9.
\]
由于 \(PU(2)\) 的有理同伦只出现在次数 \(3\)，\(PU(3)\) 只出现在次数 \(3,5\)，而 \(PU(4)\) 只出现在次数 \(3,5,7\)，故若记
\[
n_2:=\#\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=2\},
\qquad
n_3:=\#\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=3\},
\qquad
n_4:=\#\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=4\},
\]
则必有
\[
r_3=n_2+n_3+n_4,
\qquad
r_5=n_3+n_4,
\qquad
r_7=n_4.
\]
因此
\[
n_2=r_3-r_5,
\qquad
n_3=r_5-r_7,
\qquad
n_4=r_7.
\]

另一方面，由推论 \ref{cor:conclusion-window6-twofiber-boundary-internal-splitting}，
\[
\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=2\}=\{w\in X_6:\ w_6=1\}.
\]
再由定理 \ref{thm:conclusion-window6-fibonacci-staircase}，
\[
\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=4\}
=
\{w\in X_6:\ w_6=0,\ V_6(w)\le 8\},
\]
\[
\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=3\}
=
\{w\in X_6:\ w_6=0,\ V_6(w)>8\}.
\]
代回 \(n_2,n_3,n_4\) 的表达式即得所示公式。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} 的有理同伦理型严格决定离散 gauge 体积]\label{cor:conclusion-window6-rational-homotopy-determines-gauge-volume}
仅由定理 \ref{thm:conclusion-window6-rational-homotopy-inverts-tail-trichotomy} 所恢复的三元组
\[
(n_2,n_3,n_4)=(8,4,9)
\]
即可唯一重建下列离散对象：
\[
A_6^{\mathrm{grpd}}
\cong
M_2(\CC)^{\oplus 8}\oplus M_3(\CC)^{\oplus 4}\oplus M_4(\CC)^{\oplus 9},
\]
\[
G_6^{\mathrm{bin}}
\cong
\mathfrak S_2^{\,8}\times \mathfrak S_3^{\,4}\times \mathfrak S_4^{\,9},
\qquad
\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}
\cong
(\ZZ/2\ZZ)^{21},
\]
并且
\[
|G_6^{\mathrm{bin}}|=(2!)^8(3!)^4(4!)^9,
\qquad
\log |G_6^{\mathrm{bin}}|=39\log 2+13\log 3,
\]
\[
g_6=\frac{39}{64}\log 2+\frac{13}{64}\log 3\approx 0.645542,
\]
\[
\kappa_6=\frac{11}{8}\log 2+\frac{3}{16}\log 3\approx 1.159067.
\]
因此，window-\(6\) 的奇维有理同伦包络不仅决定其连通自同构群，而且决定精确的离散 gauge 体积与可见熵密度。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-window6-rational-homotopy-inverts-tail-trichotomy}，纤维直方图唯一恢复为
\[
2:8,\qquad 3:4,\qquad 4:9.
\]
Wedderburn 分解于是直接给出 \(A_6^{\mathrm{grpd}}\) 的矩阵块直和分解。由小节
\ref{subsec:conclusion-foldbin-groupoid-morita-zeta-pi}
开头对 gauge 群的定义，
\[
G_6^{\mathrm{bin}}
\cong
\prod_{d\ge 1}\mathfrak S_d^{\,n_d(6)}
\cong
\mathfrak S_2^{\,8}\times \mathfrak S_3^{\,4}\times \mathfrak S_4^{\,9}.
\]
又由于
\[
\mathfrak S_2^{\mathrm{ab}}\cong \mathfrak S_3^{\mathrm{ab}}\cong \mathfrak S_4^{\mathrm{ab}}\cong \ZZ/2\ZZ,
\]
其交换化即为 \(21\) 个二元奇偶荷的直积。

群阶公式
\[
|G_6^{\mathrm{bin}}|=(2!)^8(3!)^4(4!)^9
\]
直接来自直积分解。将
\[
2!=2,\qquad 3!=2\cdot 3,\qquad 4!=2^3\cdot 3
\]
代入，得到
\[
\log |G_6^{\mathrm{bin}}|
=
8\log 2+4(\log 2+\log 3)+9(3\log 2+\log 3)
=
39\log 2+13\log 3.
\]
于是
\[
g_6=\frac{1}{64}\log|G_6^{\mathrm{bin}}|
=
\frac{39}{64}\log 2+\frac{13}{64}\log 3.
\]
另一方面，
\[
\kappa_6
=
\frac{1}{64}\sum_{w\in X_6}d_6^{\mathrm{bin}}(w)\log d_6^{\mathrm{bin}}(w)
=
\frac{1}{64}\bigl(8\cdot 2\log 2+4\cdot 3\log 3+9\cdot 4\log 4\bigr),
\]
化简即得
\[
\kappa_6=\frac{11}{8}\log 2+\frac{3}{16}\log 3.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 纤维群胚代数的 Morita 塌缩]\label{thm:conclusion-window6-groupoid-algebra-morita-collapse}
设
\[
A_6^{\mathrm{grpd}}:=\CC[\mathcal R_6^{\mathrm{bin}}].
\]
则对任意可加 Morita 不变量 \(E\)，都有规范同构
\[
E(A_6^{\mathrm{grpd}})\cong E(\CC)^{\oplus 21}.
\]
特别地，
\[
K_0(A_6^{\mathrm{grpd}})\cong \ZZ^{21},
\qquad
K_1(A_6^{\mathrm{grpd}})=0,
\]
\[
HH_0(A_6^{\mathrm{grpd}})\cong \CC^{21},
\qquad
HH_n(A_6^{\mathrm{grpd}})=0
\qquad (n\ge 1),
\]
\[
HP_0(A_6^{\mathrm{grpd}})\cong \CC^{21},
\qquad
HP_1(A_6^{\mathrm{grpd}})=0.
\]
因此，一切 Morita 稳定的非交换不变量在 window-\(6\) 上都完全遗忘矩阵尺寸 \(2,3,4\) 的分布，只保留稳定类型总数 \(21\)。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-window6-rational-homotopy-determines-gauge-volume}，
\[
A_6^{\mathrm{grpd}}\cong M_2(\CC)^{\oplus 8}\oplus M_3(\CC)^{\oplus 4}\oplus M_4(\CC)^{\oplus 9}.
\]
对任意 Morita 不变量 \(E\)，每个全矩阵代数都满足
\[
E(M_n(\CC))\cong E(\CC),
\qquad n\ge 1,
\]
而 \(E\) 对有限直和可加，因此
\[
E(A_6^{\mathrm{grpd}})
\cong
E(\CC)^{\oplus(8+4+9)}
=
E(\CC)^{\oplus 21}.
\]

关于 \(K\)-理论与 Hochschild 同调，定理
\ref{thm:conclusion-foldbin-groupoid-morita-k-hochschild}
已给出
\[
K_0(A_6^{\mathrm{grpd}})\cong \ZZ^{21},
\qquad
K_1(A_6^{\mathrm{grpd}})=0,
\qquad
HH_0(A_6^{\mathrm{grpd}})\cong \CC^{21},
\qquad
HH_n(A_6^{\mathrm{grpd}})=0\ (n\ge 1).
\]
而对复半单代数，periodic cyclic homology 在矩阵块上满足
\[
HP_0(M_n(\CC))\cong \CC,
\qquad
HP_1(M_n(\CC))=0,
\]
故对 \(A_6^{\mathrm{grpd}}\) 逐块求和便得
\[
HP_0(A_6^{\mathrm{grpd}})\cong \CC^{21},
\qquad
HP_1(A_6^{\mathrm{grpd}})=0.
\]
证毕。
\end{proof}

\begin{theorem}[粗粒可逆数据 \texorpdfstring{$(P_6,\pi_6)$}{(P6,pi6)} 对隐藏多重度谱的完全刚性]\label{thm:conclusion-window6-markov-stationary-data-rigidly-recovers-multiplicity}
设 \((X_6,P_6)\) 为定理 \ref{thm:terminal-foldbin6-pushforward-markov} 的 stable-type Markov 核，\(\pi_6\) 为其可逆平稳分布。则对任意 \(w,w'\in X_6\) 有
\[
d_6^{\mathrm{bin}}(w)=64\,\pi_6(w),
\]
\[
A_6(w,w')=384\,\pi_6(w)P_6(w,w').
\]
从而 \((P_6,\pi_6)\) 唯一确定整个多重度多重集
\[
\{d_6^{\mathrm{bin}}(w):\ w\in X_6\},
\]
进而唯一确定
\[
\CC[\mathcal R_6^{\mathrm{bin}}],
\qquad
G_6^{\mathrm{bin}},
\qquad
|G_6^{\mathrm{bin}}|.
\]
更强地，若一个置换 \(\Pi\in \mathrm{Sym}(X_6)\) 同时满足
\[
P_6(\Pi w,\Pi w')=P_6(w,w'),
\qquad
\pi_6(\Pi w)=\pi_6(w),
\]
则 \(\Pi=\mathrm{id}\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:terminal-foldbin6-pushforward-markov}，
\[
\pi_6(w)=\frac{d_6^{\mathrm{bin}}(w)}{64},
\qquad
P_6(w,w')=\frac{A_6(w,w')}{6\,d_6^{\mathrm{bin}}(w)}.
\]
第一式立即给出
\[
d_6^{\mathrm{bin}}(w)=64\,\pi_6(w).
\]
再将其代回第二式，得到
\[
A_6(w,w')=6\,d_6^{\mathrm{bin}}(w)P_6(w,w')
=
384\,\pi_6(w)P_6(w,w').
\]
因此 \((P_6,\pi_6)\) 唯一重建加权邻接多重度矩阵 \(A_6=(A_6(w,w'))\)，并且仅由 \(\pi_6\) 就已唯一重建全部纤维多重度。由 \(A_6\) 与多重度直方图，群胚代数、gauge 群及其基数便都唯一确定。

若 \(\Pi\) 同时保持 \(P_6\) 与 \(\pi_6\)，则上式表明它也保持 \(A_6\)：
\[
A_6(\Pi w,\Pi w')=A_6(w,w').
\]
于是可应用推论 \ref{cor:terminal-gamma6-multiplicity-rigidity}，推出 \(\Pi=\mathrm{id}\)。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 类型层的张量不可分性]\label{thm:conclusion-window6-typelayer-tensor-indecomposability}
令
\[
V:=\RR^{X_6},
\qquad
\dim V=21,
\]
并记局部推送生成元为 \(L_1,\dots,L_6\in \End(V)\)。则不存在任何非平凡张量分解
\[
V\cong U\otimes W,
\qquad
1<\dim U,\dim W<21,
\]
使得全部生成元同时落在局域和式代数
\[
\End(U)\otimes I_W\;+\;I_U\otimes \End(W)
\]
之内。
\end{theorem}
\begin{proof}
由定理 \ref{thm:window6-lie-envelope-sl21}，
\[
\mathfrak g_6:=\Lie\langle L_1,\dots,L_6\rangle=\mathfrak{sl}(V),
\qquad
\dim_{\RR}\mathfrak g_6=21^2-1=440.
\]
若存在所述张量分解，则由 \(21\) 的非平凡因子分解唯一性，只可能有
\[
(\dim U,\dim W)=(3,7)
\qquad\text{或}\qquad
(7,3).
\]
相应局域和式空间的维数至多为
\[
\dim \End(U)+\dim \End(W)-1
=
3^2+7^2-1
=
57,
\]
其中减去的 \(1\) 来自公共标量 \(I_U\otimes I_W\)。若所有 \(L_i\) 均落在该局域和式空间中，则其 Lie 闭包 \(\mathfrak g_6\) 也必须包含于其中，于是
\[
\dim \mathfrak g_6\le 57,
\]
与 \(\dim\mathfrak g_6=440\) 矛盾。证毕。
\end{proof}

\begin{corollary}[局部推送生成元的关联包络已充满整个 \texorpdfstring{$\End(V)$}{End(V)}]\label{cor:conclusion-window6-local-pushforward-envelope-full-matrix}
设
\[
\mathcal A_6:=\RR\langle I,L_1,\dots,L_6\rangle\subseteq \End(V).
\]
则
\[
\mathcal A_6=\End(V)\cong M_{21}(\RR).
\]
因此，window-\(6\) 类型层上的任意线性可观测量，都可由局部推送生成元在有限关联代数闭包内精确生成。结合定理 \ref{thm:conclusion-window6-typelayer-tensor-indecomposability}，任何试图以更小寄存器张量结构对该类型层作精确无辅助模拟的方案都必失败。
\end{corollary}
\begin{proof}
等式
\[
\mathcal A_6=\End(V)
\]
正是定理 \ref{thm:window6-edge-flux-full-matrix-saturation} 的内容。后一句则由此定理与定理 \ref{thm:conclusion-window6-typelayer-tensor-indecomposability} 直接推出：若存在更小张量寄存器的无辅助精确模拟，则全部生成元必落在某个局域和式代数中，这已被前一定理排除。证毕。
\end{proof}
