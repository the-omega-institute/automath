\begin{theorem}[氢型壳层地址的严格二级规范塔]\label{thm:conclusion-hydrogenic-strict-twolevel-gauge-tower}
沿用命题 \ref{prop:conclusion-hydrogenic-address-grammar} 与定理 \ref{thm:conclusion-hydrogenic-residual-audit-capacity} 的记号。定义
\[
\pi_n:\{(l,\mu):\ 0\le l\le n-1,\ 0\le \mu\le l\}\to \{0,\dots,n-1\},
\qquad
\pi_n(l,\mu):=l.
\]
则有严格因子分解
\[
\Pi_n^{\mathrm{cf,sp}}
=
\pi_n\circ \Pi_n^{\mathrm{pb,sp}}.
\]
并且对每个固定 \(0\le l\le n-1\)，成立不交并分解
\[
\bigl(\Pi_n^{\mathrm{cf,sp}}\bigr)^{-1}(l)
=
\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,0)
\sqcup
\bigsqcup_{\mu=1}^{l}
\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,\mu),
\]
其中
\[
\#\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,0)=2,
\qquad
\#\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,\mu)=4
\qquad
(\mu>0).
\]
因此，每个 \(l\)-块都唯一裂解为一枚轴向自旋双态与 \(l\) 个符号--自旋四态。
\end{theorem}
\begin{proof}
\(\Pi_n^{\mathrm{pb,sp}}\) 先记录 \((l,|m|)\)，\(\pi_n\) 再忘掉 \(|m|\)，故
\[
\Pi_n^{\mathrm{cf,sp}}=\pi_n\circ \Pi_n^{\mathrm{pb,sp}}
\]
立即成立。

固定 \(l\) 后，由命题 \ref{prop:conclusion-hydrogenic-address-grammar} 有唯一的
\[
n_r=n-l-1.
\]
于是
\[
\bigl(\Pi_n^{\mathrm{cf,sp}}\bigr)^{-1}(l)
=
\bigsqcup_{\mu=0}^{l}\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,\mu).
\]
若 \(\mu=0\)，则 \(m=0\) 固定，仅剩 \(\sigma=\pm \tfrac12\) 两种可能，故纤维大小为 \(2\)。若 \(\mu>0\)，则 \(m=\pm\mu\) 与 \(\sigma=\pm\tfrac12\) 独立，故纤维大小为 \(4\)。证毕。
\end{proof}

\begin{corollary}[两级规范塌缩的可见类数与比例分裂]\label{cor:conclusion-hydrogenic-visible-class-ratio-splitting}
对每个 \(n\ge 1\)，都有精确计数
\[
\#\mathcal A_n=2n^2,
\qquad
\#\operatorname{im}\Pi_n^{\mathrm{cf,sp}}=n,
\qquad
\#\operatorname{im}\Pi_n^{\mathrm{pb,sp}}=\frac{n(n+1)}{2}.
\]
从而可见类比例满足
\[
\frac{\#\operatorname{im}\Pi_n^{\mathrm{cf,sp}}}{\#\mathcal A_n}
=
\frac{1}{2n}\longrightarrow 0,
\qquad
\frac{\#\operatorname{im}\Pi_n^{\mathrm{pb,sp}}}{\#\mathcal A_n}
=
\frac{n+1}{4n}\longrightarrow \frac14.
\]
换言之，类函数规范把整个壳层压到一个渐近消失的可见比例，而相位盲规范仍保留一条正极限比例。
\end{corollary}
\begin{proof}
由命题 \ref{prop:conclusion-hydrogenic-address-grammar}，
\[
\#\mathcal A_n=2n^2.
\]
又由定理 \ref{thm:conclusion-hydrogenic-classfunction-block-collapse}，\(\Pi_n^{\mathrm{cf,sp}}\) 只记录 \(l\)，故其像基数为 \(n\)。由定理 \ref{thm:conclusion-hydrogenic-phaseblind-pair-collapse}，
\[
\#\operatorname{im}\Pi_n^{\mathrm{pb,sp}}
=
\sum_{l=0}^{n-1}(l+1)
=
\frac{n(n+1)}{2}.
\]
两边分别除以 \(2n^2\) 即得比例公式与极限。证毕。
\end{proof}

\begin{theorem}[相位盲规范投影的平均残余预算]\label{thm:conclusion-hydrogenic-phaseblind-average-residual-bits}
令
\[
U_n\sim \mathrm{Unif}(\mathcal A_n)
\]
为壳层 \(\mathcal A_n\) 上的均匀分布，并约定全部熵以 \(\log_2\) 计。则
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{pb,sp}}(U_n)\right)
=
2-\frac1n
\qquad\text{bits}.
\]
等价地，
\[
H\!\left(\Pi_n^{\mathrm{pb,sp}}(U_n)\right)
=
\log_2(2n^2)-\Bigl(2-\frac1n\Bigr)
=
2\log_2 n-1+\frac1n.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-hydrogenic-residual-audit-capacity}，\(\Pi_n^{\mathrm{pb,sp}}\) 的纤维只有两种大小：
\[
\#\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,0)=2,
\qquad
\#\bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,\mu)=4
\quad
(\mu>0).
\]
在均匀分布下，每个 \((l,0)\)-类的概率为
\[
\frac{2}{2n^2}=\frac1{n^2},
\]
这样的类共有 \(n\) 个；每个 \((l,\mu)\)（\(\mu>0\)）类的概率为
\[
\frac{4}{2n^2}=\frac{2}{n^2},
\]
这样的类共有
\[
\sum_{l=0}^{n-1}l=\frac{n(n-1)}2
\]
个。故条件熵为
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{pb,sp}}(U_n)\right)
=
n\cdot \frac1{n^2}\cdot \log_2 2
+
\frac{n(n-1)}2\cdot \frac{2}{n^2}\cdot \log_2 4
=
\frac1n+2\frac{n-1}{n}
=
2-\frac1n.
\]
再由链式法则与
\[
H(U_n)=\log_2\#\mathcal A_n=\log_2(2n^2)
\]
得到
\[
H\!\left(\Pi_n^{\mathrm{pb,sp}}(U_n)\right)
=
H(U_n)-H\!\left(U_n\middle|\Pi_n^{\mathrm{pb,sp}}(U_n)\right),
\]
即所示公式。证毕。
\end{proof}
\leanverified{paper\_conclusion\_hydrogenic\_phaseblind\_average\_residual\_bits}

\begin{theorem}[类函数规范投影的平均残余预算具有精确对数主项]\label{thm:conclusion-hydrogenic-classfunction-average-residual-logarithm}
在定理 \ref{thm:conclusion-hydrogenic-phaseblind-average-residual-bits} 的均匀壳层分布 \(U_n\) 下，
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\frac1{n^2}\sum_{l=0}^{n-1}(2l+1)\log_2(4l+2).
\]
并且当 \(n\to\infty\) 时，
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\log_2 n+2-\frac{1}{2\ln 2}+o(1).
\]
相应地，
\[
H\!\left(\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\log_2 n-1+\frac{1}{2\ln 2}+o(1).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-hydrogenic-residual-audit-capacity}，对固定 \(l\) 有
\[
\#\bigl(\Pi_n^{\mathrm{cf,sp}}\bigr)^{-1}(l)=2(2l+1)=4l+2.
\]
该纤维在均匀壳层分布下的概率为
\[
\frac{4l+2}{2n^2}=\frac{2l+1}{n^2}.
\]
故条件熵等于
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\frac1{n^2}\sum_{l=0}^{n-1}(2l+1)\log_2(4l+2).
\]

现在令
\[
x_{l,n}:=\frac{l+\frac12}{n}
\qquad
(0\le l\le n-1).
\]
则 \(2l+1=2nx_{l,n}\)、\(4l+2=4nx_{l,n}\)，从而上式可精确改写为
\[
\frac1{n^2}\sum_{l=0}^{n-1}(2l+1)\log_2(4l+2)
=
\log_2 n
+
\frac1n\sum_{l=0}^{n-1}2x_{l,n}\log_2(4x_{l,n}).
\]
记
\[
f(x):=2x\log_2(4x)
\qquad (0<x\le 1),
\]
并约定 \(f(0):=0\)。则 \(f\) 在 \([0,1]\) 上连续，而上式第二项正是 \(f\) 在区间 \([0,1]\) 上的中点 Riemann 和，因此
\[
\frac1n\sum_{l=0}^{n-1}2x_{l,n}\log_2(4x_{l,n})
\longrightarrow
\int_0^1 2x\log_2(4x)\,dx.
\]
直接计算该积分得
\[
\int_0^1 2x\log_2(4x)\,dx
=
\frac{1}{\ln 2}\int_0^1 2x(\ln 4+\ln x)\,dx
=
2-\frac{1}{2\ln 2}.
\]
于是
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\log_2 n+2-\frac{1}{2\ln 2}+o(1).
\]
最后再用
\[
H(U_n)=\log_2(2n^2)=1+2\log_2 n
\]
与链式法则得到
\[
H\!\left(\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
H(U_n)-H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\log_2 n-1+\frac{1}{2\ln 2}+o(1).
\]
证毕。
\end{proof}
\leanverified{paper\_conclusion\_hydrogenic\_classfunction\_average\_residual\_logarithm}

\begin{corollary}[发散隐藏成本完全由非阿贝尔块塌缩承担]\label{cor:conclusion-hydrogenic-divergent-hidden-cost-from-nonabelian-blocks}
在上述均匀壳层分布下，
\[
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
-
H\!\left(U_n\middle|\Pi_n^{\mathrm{pb,sp}}(U_n)\right)
=
\log_2 n-\frac{1}{2\ln 2}+o(1).
\]
因此，随壳层 \(n\) 增长而发散的隐藏预算完全来自
\[
(l,m,\sigma)\longmapsto l
\]
这一 \(\mathrm{SO}(3)\)-类函数块塌缩；而
\[
(l,m,\sigma)\longmapsto (l,|m|)
\]
只留下一个有界极限残差
\[
2-\frac1n\longrightarrow 2.
\]
\end{corollary}
\begin{proof}
将定理 \ref{thm:conclusion-hydrogenic-phaseblind-average-residual-bits} 与定理 \ref{thm:conclusion-hydrogenic-classfunction-average-residual-logarithm} 相减即可。证毕。
\end{proof}

\begin{theorem}[相位盲层的 generic 残余是严格的 dyadic torsor]\label{thm:conclusion-hydrogenic-generic-phaseblind-dyadic-torsor}
\leanverified{paper\_conclusion\_hydrogenic\_generic\_phaseblind\_dyadic\_torsor}
在定理 \ref{thm:conclusion-hydrogenic-phaseblind-average-residual-bits} 的均匀壳层分布下，定义
\[
\mu_n:\mathcal A_n\to \{0,\dots,n-1\},
\qquad
\mu_n(n_r,l,m,\sigma):=|m|,
\]
并记事件
\[
E_n:=\{\mu_n(U_n)=0\}.
\]
则有
\[
\PP(E_n)=\frac1n,
\qquad
\PP(E_n^c)=1-\frac1n.
\]
更具体地：
\begin{enumerate}
  \item 若 \(\Pi_n^{\mathrm{pb,sp}}(U_n)=(l,0)\)，则残余纤维
  \[
  \bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,0)
  \]
  可规范识别为
  \[
  \Bigl\{\pm \frac12\Bigr\},
  \]
  从而是一条 \(\ZZ/2\ZZ\)-torsor；
  \item 若 \(\Pi_n^{\mathrm{pb,sp}}(U_n)=(l,\mu)\) 且 \(\mu>0\)，则残余纤维
  \[
  \bigl(\Pi_n^{\mathrm{pb,sp}}\bigr)^{-1}(l,\mu)
  \]
  可规范识别为
  \[
  \{\pm1\}\times \Bigl\{\pm\frac12\Bigr\},
  \]
  从而是一条 \((\ZZ/2\ZZ)^2\)-torsor。
\end{enumerate}
因此，相位盲层的 generic 残余并非连续相位模糊，而是一份严格的二维 dyadic 地址；一维退化轴层只占 \(1/n\) 的概率质量。
\end{theorem}
\begin{proof}
满足 \(|m|=0\) 的态恰为
\[
(n-l-1,l,0,\sigma),
\qquad
0\le l\le n-1,\ \sigma\in\Bigl\{\pm\frac12\Bigr\},
\]
其总数为 \(2n\)。因此
\[
\PP(E_n)=\frac{2n}{\#\mathcal A_n}=\frac{2n}{2n^2}=\frac1n,
\qquad
\PP(E_n^c)=1-\frac1n.
\]

纤维的规范识别则直接来自定理 \ref{thm:conclusion-hydrogenic-strict-twolevel-gauge-tower} 与定理 \ref{thm:conclusion-hydrogenic-residual-audit-capacity}：当 \(\mu=0\) 时，\(m=0\) 固定，仅剩 \(\sigma=\pm\tfrac12\)；当 \(\mu>0\) 时，\(\operatorname{sgn}(m)\in\{\pm1\}\) 与 \(\sigma\in\{\pm\tfrac12\}\) 独立，故得到所述 \((\ZZ/2\ZZ)^2\)-torsor。证毕。
\end{proof}

\begin{proposition}[最坏情形预算与平均预算的匹配差异]\label{prop:conclusion-hydrogenic-worst-average-budget-gap}
沿用定理 \ref{thm:conclusion-hydrogenic-residual-audit-capacity} 中的最坏情形预算
\[
B_n^{\mathrm{cf}}=\Bigl\lceil \log_2(4n-2)\Bigr\rceil,
\qquad
B_n^{\mathrm{pb}}=2.
\]
则有精确公式
\[
B_n^{\mathrm{pb}}
-
H\!\left(U_n\middle|\Pi_n^{\mathrm{pb,sp}}(U_n)\right)
=
\frac1n,
\]
以及非取整轮廓上的渐近公式
\[
\log_2(4n-2)
-
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\frac{1}{2\ln2}+o(1).
\]
若再记
\[
\eta_n:=B_n^{\mathrm{cf}}-\log_2(4n-2)\in[0,1),
\]
则
\[
B_n^{\mathrm{cf}}
-
H\!\left(U_n\middle|\Pi_n^{\mathrm{cf,sp}}(U_n)\right)
=
\frac{1}{2\ln2}+\eta_n+o(1).
\]
因此：
\begin{enumerate}
  \item 相位盲层的最坏情形预算与平均预算渐近重合；
  \item 类函数层的非取整最坏情形轮廓稳定高出平均预算一个普适常数
  \[
  \frac{1}{2\ln2}\approx 0.7213\ \text{bits},
  \]
  而取整后只额外留下一个有界的亚比特振荡。
\end{enumerate}
\end{proposition}
\begin{proof}
第一式由定理 \ref{thm:conclusion-hydrogenic-phaseblind-average-residual-bits} 直接给出。第二式由
\[
\log_2(4n-2)=\log_2 n+2+o(1)
\]
与定理 \ref{thm:conclusion-hydrogenic-classfunction-average-residual-logarithm} 相减得到。第三式只是把
\[
B_n^{\mathrm{cf}}=\log_2(4n-2)+\eta_n
\]
代回第二式。证毕。
\end{proof}

\begin{conjecture}[中心势束缚态的二比特 phase-blind 普适律]\label{conj:conclusion-central-potential-two-bit-phaseblind-law}
设某一中心势束缚系统满足以下三项：
\begin{enumerate}
  \item 径向波函数可取实值规范；
  \item 角向子空间分解为 \(\mathrm{SO}(3)\) 的不可约块；
  \item 非零磁量子数始终以 \(\pm m\) 共轭对出现。
\end{enumerate}
则在固定主壳层上，经由
\[
(l,m,\sigma)\longmapsto (l,|m|)
\]
型 phase-blind 规范后，generic 残余预算应收敛到常数 \(2\) 比特，而全部发散隐藏成本都只能来自 \(l\)-块塌缩，而不可能来自 \(\pm m\) 配对本身。
\end{conjecture}

\noindent
由此，氢型定态在本文语义下不再只是标准教材中的谱分解，而被重写为一个可审计的二级规范塔：\(\Pi_n^{\mathrm{pb,sp}}\) 与 \(\Pi_n^{\mathrm{cf,sp}}\) 之间存在严格因子化，且每个 \(l\)-块精确裂解为一枚轴向自旋双态与若干符号--自旋四态。进一步，在均匀壳层分布下，这条塔在信息论层出现刚性的双轴分裂：phase-blind 层保留 \((n+1)/(4n)\to 1/4\) 的正可见比例，只留下 \(2-\tfrac1n\to 2\) 比特的有限平均残余，且其 generic 纤维是一条 \((\ZZ/2\ZZ)^2\)-torsor；相反，类函数层的可见比例坍缩为 \(1/(2n)\to 0\)，平均隐藏预算则增长为 \(\log_2 n+2-\frac{1}{2\ln2}+o(1)\)。因此，氢型 hidden-address geometry 在结论层被 rigidly 分裂为一条发散的非阿贝尔块轴与一条有限的 dyadic phase--spin 轴：前者承担全部对数发散隐藏成本，后者只保留一个两比特极限残差。
