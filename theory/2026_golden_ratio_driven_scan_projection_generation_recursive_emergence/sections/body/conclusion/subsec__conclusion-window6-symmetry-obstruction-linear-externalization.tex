\begin{theorem}[规范差外置的线性不可压缩下界]\label{thm:conclusion-linear-externalization-liminf}
\leanverified{paper\_conclusion\_linear\_externalization\_liminf}
设 \(G_m(\omega)\) 为长度 \(m\) 的规范差总和，并假设
\[
g_\star:=\lim_{m\to\infty}\frac{\EE[G_m]}{m}=\frac49.
\]
考虑任意局部重写正规化算法 \(S\)。若存在常数 \(R\in\NN\)，使得算法每一步至多消去 \(R\) 个单位规范差，且其步数 \(T_m(S)\) 满足
\[
G_m(\omega)\le R\,T_m(S)\qquad(\forall\,\omega,m),
\]
则几乎处处有
\[
\liminf_{m\to\infty}\frac{T_m(S)}{m}\ge \frac{g_\star}{R}=\frac{4}{9R}.
\]
并且对任意 \(a<4/(9R)\)，成立指数下尾界
\[
\limsup_{m\to\infty}\frac1m\log
\PP\!\left(\frac{T_m(S)}{m}\le a\right)<0.
\]
\end{theorem}
\begin{proof}
由 \(G_m\le R\,T_m(S)\) 立得事件包含
\[
\left\{\frac{T_m(S)}{m}\le a\right\}
\subseteq
\left\{\frac{G_m}{m}\le Ra\right\}.
\]
当 \(a<g_\star/R\) 时有 \(Ra<g_\star\)。由正文对 \(G_m/m\) 的大偏差下尾估计，存在 \(c(a)>0\) 使
\[
\PP\!\left(\frac{G_m}{m}\le Ra\right)\le e^{-c(a)m}
\]
对充分大 \(m\) 成立，故同样
\[
\PP\!\left(\frac{T_m(S)}{m}\le a\right)\le e^{-c(a)m},
\]
从而得到指数下尾结论。
再由 Borel--Cantelli，引理给出
\[
\PP\!\left(\frac{T_m(S)}{m}\le a\ \text{i.o.}\right)=0
\]
对每个有理 \(a<g_\star/R\) 成立。对可数有理稠密子集取交，得到几乎处处
\[
\liminf_{m\to\infty}\frac{T_m(S)}{m}\ge \frac{g_\star}{R}.
\]
证毕。
\end{proof}

\leanverified{paper\_conclusion\_prime\_register\_residual\_support\_linear\_incompressibility}

\begin{theorem}[可逆侧信息的长度下界：有限字母表无法压缩折叠纤维的信息缺口]\label{thm:conclusion-side-info-length-lower-bound}
\leanverified{paper\_conclusion\_side\_info\_length\_lower\_bound}

令 \(\Sigma\) 为有限字母表，\(|\Sigma|=M\ge 2\)。
考虑任意确定性可逆提升
\[
\Psi_m:\Omega_m\longrightarrow X_m\times \Sigma^{\le L_m}
\]
满足 \(\mathrm{pr}_{X_m}\circ \Psi_m=\Fold_m\)，且 \(\Psi_m\) 为单射。
则必有
\[
\boxed{\;
L_m\ \ge\ \left\lceil \log_M\!\Bigl(\frac{2^m}{F_{m+2}}\Bigr)\right\rceil-1\; }.
\]
并且
\[
\liminf_{m\to\infty}\frac{L_m}{m}\ \ge\ \frac{\log(2/\varphi)}{\log M},
\]
其中 \(\varphi\) 为黄金比例。
\leanverified{sideInfo\_length\_lower\_bound\_binary\_specialized}
\end{theorem}
\begin{proof}
由满射性与计数，存在 \(x\in X_m\) 使纤维大小满足
\[
|\Fold_m^{-1}(x)|\ \ge\ \frac{|\Omega_m|}{|X_m|}
=\frac{2^m}{F_{m+2}}.
\]
在该纤维上，\(\Psi_m\) 限制为单射
\(
\Fold_m^{-1}(x)\hookrightarrow \{x\}\times \Sigma^{\le L_m}
\)，故
\[
\frac{2^m}{F_{m+2}}
\le
|\Sigma^{\le L_m}|
=\sum_{\ell=0}^{L_m}M^\ell
=\frac{M^{L_m+1}-1}{M-1}
<M^{L_m+1}.
\]
取 \(\log_M\) 得
\(
L_m\ge \lceil \log_M(2^m/F_{m+2})\rceil-1
\)。
最后由 \(F_{m+2}=\Theta(\varphi^m)\) 得
\(
\lim_{m\to\infty}\frac1m\log(2^m/F_{m+2})=\log(2/\varphi)
\)，从而得到极限下界。证毕。
\end{proof}

\(
(\Fold_6^{\mathrm{bin}})^{-1}(w)\subset\{0,1\}^6
\)
中的最小 Hamming 距离。则
\[
\delta_{\min}(w)\in\{2,3,5\},\qquad
\delta_{\min}(w)=5
\Longleftrightarrow
w\in\{\texttt{010001},\texttt{100001}\},
\]
并且在 \(\delta_{\min}(w)=5\) 时必有
\[
d_6^{\mathrm{bin}}(w)=2.
\]
\end{proposition}
\begin{proof}
\(\delta_{\min}(w)\) 的取值集合与对应类型由 window-\(6\) 全分类直接给出。
对 \(\delta_{\min}(w)=5\) 的情形，纤维中任意两点距离至少 \(5\)。
在 \(\{0,1\}^6\) 中，半径 \(2\) 的 Hamming 球大小为
\[
1+\binom61+\binom62=22.
\]
若存在三点两两距离 \(\ge 5\)，则其半径 \(2\) 球两两不交，总点数至少 \(66>64\)，矛盾。
故该情形下纤维至多含两点；由 \(\delta_{\min}(w)=5\) 的定义又知纤维至少含两点，故恰为两点，即 \(d_6^{\mathrm{bin}}(w)=2\)。证毕。
\end{proof}

整除。

\paragraph{纤维平均与非可逆对称接口}
附录 \ref{app:op-algebra-fold-subalgebra} 将分辨率折叠的粗粒化实现为交换代数包含 \(\mathcal{B}_m\subset \mathcal{A}_m\) 与纤维平均条件期望 \(E_m:\mathcal{A}_m\to\mathcal{B}_m\)。

\begin{theorem}\label{thm:conclusion-fold-condexp-quantum-operation-noninvertible}
沿用命题 \ref{prop:fold-conditional-expectation} 的记号，并取计数迹 \(\tau_m\) 使得 \(\tau_m(P_a)=1\)（或等价的微观均匀迹 \(2^{-m}\tau_m\)）。
则 \(E_m\) 为保持单位元的完全正映射且满足 \(\tau_m\circ E_m=\tau_m\)，从而在 Heisenberg 图像下定义一量子信道。
此外，若存在 \(x\in X_m\) 使 \(d_m(x)>1\)，则 \(E_m\) 非单射，因而不可能为 \(\ast\)-自同构；该不可逆性由纤维凝聚所强制，并与非可逆对称以量子操作局域作用于可观测量的范式相容 \cite{OkadaTachikawa2024NonInvertibleQuantumOperations,SchaferNameki2023ICTPLecturesNonInvertibleGeneralizedSymmetries}。
与“条件期望/指数”在非可逆对称纲领中的角色对齐的综述性讨论可参见 \href{https://arxiv.org/abs/2509.18072}{arXiv:2509.18072}。
\leanverified{paper\_conclusion\_fold\_condexp\_quantum\_operation\_noninvertible}
\end{theorem}
\begin{proof}
命题 \ref{prop:fold-conditional-expectation} 已给出 \(E_m\) 的正性与保持单位元；在有限维交换 \(C^\ast\)-代数中，正性蕴含完全正。
对迹保持性，只需在基 \(\{P_a\}_{a\in\Omega_m}\) 上验证：若 \(x=\Fold_m(a)\)，则
\[
\tau_m(E_m(P_a))=\tau_m\!\left(\frac{1}{d_m(x)}\widetilde P_x\right)
=\frac{1}{d_m(x)}\sum_{\Fold_m(b)=x}\tau_m(P_b)=1=\tau_m(P_a),
\]
线性延拓得到 \(\tau_m\circ E_m=\tau_m\)。
若某纤维满足 \(d_m(x)>1\)，取不同 \(a\neq b\) 使 \(\Fold_m(a)=\Fold_m(b)=x\)，则
\(
E_m(P_a-P_b)=0
\)
而 \(P_a-P_b\neq 0\)，故 \(E_m\) 非单射，从而不可逆。证毕。
\end{proof}
