\subsection{循环扇区角色分解、扫描谱算术可编程性与 Kirchhoff--Green 坏素数刚性}\label{subsec:conclusion-cyclic-sector-scan-spectrum-kirchhoff-green-bad-prime-rigidity}
本小节在 \(\Fold_m\) 诱导的循环合法词扇区上，给出角色分解闭式及其在素长、素幂层上的刚性特化，极大扫描轨道长度的算术可编程性，以及临界群扭结与 Green 核坏素数之间的等价判据。

\begin{theorem}[循环 Zeckendorf 扇区的角色分解闭式]\label{thm:conclusion-cyclic-zeckendorf-sector-character-closed-form}
设
\[
X_m:=\{w\in\{0,1\}^m:\ w\ \text{不含子词}\ 11\},
\qquad
X_m^{\mathrm{cyc}}:=\{w\in X_m:\ w_1w_m=0\}.
\]
令 \(C_m=\langle\rho\rangle\) 为长度 \(m\) 的循环移位群在 \(X_m^{\mathrm{cyc}}\) 上的作用，记
\[
V_m:=\CC[X_m^{\mathrm{cyc}}],
\qquad
\chi_k(\rho):=e^{2\pi i k/m}\ \ (k=0,\dots,m-1).
\]
则 \(\chi_k\) 在 \(V_m\) 中的重数为
\[
\mathrm{mult}_m(k)
=\frac1m\sum_{d\mid m}L_d\,c_{m/d}(k),
\]
其中
\[
L_d:=F_{d-1}+F_{d+1},
\qquad
c_n(k):=\sum_{\substack{1\le a\le n\\(a,n)=1}}e^{-2\pi i ka/n}.
\]
特别地，\(\mathrm{mult}_m(k)\in\ZZ_{\ge 0}\)。
\leanverified{paper\_conclusion\_cyclic\_zeckendorf\_sector\_character\_closed\_form}
\end{theorem}
\begin{proof}
先记 \(\vartheta_m:=|X_m^{\mathrm{cyc}}|\)。对长度 \(m\) 的循环约束“无相邻 \(11\)”可用转移矩阵
\(
M=\begin{psmallmatrix}1&1\\ 1&0\end{psmallmatrix}
\)
编码，故
\[
\vartheta_m=\mathrm{tr}(M^m)=L_m.
\]
因此 \(|X_d^{\mathrm{cyc}}|=L_d\)。

对任意 \(j\in\{0,\dots,m-1\}\)，令 \(d=\gcd(m,j)\)。
若 \(w\in X_m^{\mathrm{cyc}}\) 被 \(\rho^j\) 固定，则 \(w\) 具有周期 \(d\)，可写为某个长度 \(d\) 词 \(u\) 的 \(m/d\) 次重复；循环拼接处约束与内部约束一致，因此 \(u\in X_d^{\mathrm{cyc}}\)。
反之任取 \(u\in X_d^{\mathrm{cyc}}\) 重复 \(m/d\) 次即得到 \(\rho^j\)-不动点。
故
\[
\#\mathrm{Fix}(\rho^j)=|X_d^{\mathrm{cyc}}|=L_d,\qquad d=\gcd(m,j).
\]

由有限交换群角色分解公式，
\[
\mathrm{mult}_m(k)
=\frac1m\sum_{j=0}^{m-1}\overline{\chi_k(\rho^j)}\,\#\mathrm{Fix}(\rho^j)
=\frac1m\sum_{j=0}^{m-1}e^{-2\pi i kj/m}L_{\gcd(m,j)}.
\]
按 \(d=\gcd(m,j)\) 分组，令 \(n=m/d\)。
满足 \(\gcd(m,j)=d\) 的 \(j\) 与模 \(n\) 的单位类一一对应（可写为 \(j\equiv da\pmod m\), \((a,n)=1\)；当 \(n=1\) 时对应 \(j=0\)），
于是对应指数和为 Ramanujan 和 \(c_n(k)=c_{m/d}(k)\)。从而得到闭式
\[
\mathrm{mult}_m(k)=\frac1m\sum_{d\mid m}L_d\,c_{m/d}(k).
\]
非负整数性由 \(\mathrm{mult}_m(k)=\langle\chi_{V_m},\chi_k\rangle_{C_m}\) 立即成立。证毕。
\end{proof}

\begin{theorem}[素长度循环 Zeckendorf 扇区的正则表示一阶偏置分解]\label{thm:conclusion-cyclic-zeckendorf-prime-regular-bias}
设 \(p\ge 2\) 为素数，并沿用定理 \ref{thm:conclusion-cyclic-zeckendorf-sector-character-closed-form} 的记号。定义
\[
a_p:=\frac{L_p-1}{p}.
\]
则 \(a_p\in\ZZ_{\ge 0}\)，并且循环扇区表示 \(V_p=\CC[X_p^{\mathrm{cyc}}]\) 具有严格分解
\[
V_p\cong \mathbf 1\ \oplus\ a_p\,\CC[C_p].
\]
等价地，平凡特征与非平凡特征的重数满足
\[
\mathrm{mult}_p(0)=a_p+1,
\qquad
\mathrm{mult}_p(k)=a_p\quad (1\le k\le p-1).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-cyclic-zeckendorf-sector-character-closed-form}，
\[
\mathrm{mult}_p(k)=\frac1p\bigl(L_1\,c_p(k)+L_p\,c_1(k)\bigr).
\]
由于 \(L_1=1\) 且 \(c_1(k)=1\)，故
\[
\mathrm{mult}_p(k)=\frac{c_p(k)+L_p}{p}.
\]
对素数模 \(p\)，Ramanujan 和满足
\[
c_p(0)=p-1,
\qquad
c_p(k)=-1\quad (1\le k\le p-1).
\]
于是
\[
\mathrm{mult}_p(0)=\frac{L_p+p-1}{p},
\qquad
\mathrm{mult}_p(k)=\frac{L_p-1}{p}\quad (1\le k\le p-1).
\]

只需证明 \(a_p=(L_p-1)/p\in\ZZ\)。令 \(\bar\varphi:=1-\varphi\)。由 Binet 公式
\[
L_p=\varphi^p+\bar\varphi^{\,p}.
\]
在商环 \(\ZZ[\varphi]/p\ZZ[\varphi]\) 中，Frobenius 同态给出
\[
\varphi^p+\bar\varphi^{\,p}\equiv (\varphi+\bar\varphi)^p=1^p=1\pmod p.
\]
故 \(p\mid(L_p-1)\)，从而 \(a_p\in\ZZ_{\ge 0}\)。

最后，\(\CC[C_p]\) 的角色分解中每个特征都恰出现一次，而 \(\mathbf 1\) 仅增加平凡特征一次；这与上式给出的重数结构完全一致，因此
\[
V_p\cong \mathbf 1\oplus a_p\,\CC[C_p].
\]
证毕。
\end{proof}

\begin{theorem}[素幂长度循环扇区的 \texorpdfstring{$p$}{p}-进壳层刚性]\label{thm:conclusion-cyclic-zeckendorf-primepower-shell-rigidity}
\leanverified{paper\_conclusion\_cyclic\_zeckendorf\_primepower\_shell\_rigidity}
设 \(m=p^\nu\)，其中 \(p\) 为素数、\(\nu\ge 1\)。记
\[
V_{p^\nu}:=\CC[X_{p^\nu}^{\mathrm{cyc}}],
\qquad
\widehat{C_{p^\nu}}=\{\chi_k:0\le k<p^\nu\}.
\]
对 \(0\le t\le \nu-1\)，定义第 \(t\) 个 \(p\)-进壳层
\[
\mathcal R_t:=\bigoplus_{\substack{0\le k<p^\nu\\ v_p(k)=t}}\chi_k,
\qquad
\mathcal R_\nu:=\chi_0.
\]
则存在整数 \(m_0,\dots,m_\nu\in\ZZ_{\ge 0}\)，使得
\[
V_{p^\nu}\cong \bigoplus_{t=0}^{\nu} m_t\,\mathcal R_t,
\]
并且这些重数具有显式闭式
\[
m_\nu=
\frac1{p^\nu}
\left(
L_{p^\nu}+\sum_{a=1}^{\nu}p^{a-1}(p-1)L_{p^{\nu-a}}
\right),
\]
以及对 \(0\le t\le \nu-1\),
\[
m_t=
\frac1{p^\nu}
\left(
L_{p^\nu}
+\sum_{a=1}^{t}p^{a-1}(p-1)L_{p^{\nu-a}}
-p^t L_{p^{\nu-t-1}}
\right).
\]
特别地，\(\chi_k\) 在 \(V_{p^\nu}\) 中的重数只依赖于 \(v_p(k)\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-cyclic-zeckendorf-sector-character-closed-form}，
\[
\mathrm{mult}_{p^\nu}(k)=\frac1{p^\nu}\sum_{d\mid p^\nu}L_d\,c_{p^\nu/d}(k).
\]
将因子写作 \(d=p^{\nu-a}\)（\(0\le a\le \nu\)），得到
\[
\mathrm{mult}_{p^\nu}(k)=\frac1{p^\nu}\sum_{a=0}^{\nu}L_{p^{\nu-a}}\,c_{p^a}(k).
\]

先设 \(k\neq 0\)，并记 \(t:=v_p(k)\in\{0,\dots,\nu-1\}\)。对 Ramanujan 和 \(c_{p^a}(k)\) 有标准公式
\[
c_{p^a}(k)=
\begin{cases}
1, & a=0,\\[1mm]
p^{a-1}(p-1), & 1\le a\le t,\\[1mm]
-p^t, & a=t+1,\\[1mm]
0, & a\ge t+2.
\end{cases}
\]
代回即得
\[
\mathrm{mult}_{p^\nu}(k)
=
\frac1{p^\nu}
\left(
L_{p^\nu}
+\sum_{a=1}^{t}p^{a-1}(p-1)L_{p^{\nu-a}}
-p^t L_{p^{\nu-t-1}}
\right),
\]
故同一 \(p\)-进壳层中全部角色具有相同重数，记为 \(m_t\)。

对 \(k=0\)，有
\[
c_{p^a}(0)=\varphi(p^a)=
\begin{cases}
1, & a=0,\\
p^{a-1}(p-1), & 1\le a\le \nu,
\end{cases}
\]
于是
\[
\mathrm{mult}_{p^\nu}(0)
=
\frac1{p^\nu}
\left(
L_{p^\nu}+\sum_{a=1}^{\nu}p^{a-1}(p-1)L_{p^{\nu-a}}
\right),
\]
这正是 \(m_\nu\)。

因此 \(V_{p^\nu}\) 按角色分解时，所有 \(v_p(k)=t\) 的角色均以同一重数 \(m_t\) 出现，从而
\[
V_{p^\nu}\cong \bigoplus_{t=0}^{\nu} m_t\,\mathcal R_t.
\]
证毕。
\end{proof}

\begin{corollary}[window-\(6\) 的 Fourier 签名由 Lucas--Ramanujan 闭式唯一确定]\label{cor:conclusion-window6-fourier-signature-lucas-ramanujan}
当 \(m=6\) 时，
\[
\bigl(\mathrm{mult}_6(0),\dots,\mathrm{mult}_6(5)\bigr)=(5,2,3,3,3,2).
\]
该向量与正文在 \(X_6^{\mathrm{cyc}}\) 上给出的 \(C_6\)-轨道分解及角色签名一致（见 \S\ref{subsec:bdry-tower-zeck-gut} 与表 \ref{tab:window6_c6_orbit_decomposition}）。
\leanverified{paper\_conclusion\_window6\_fourier\_signature\_lucas\_ramanujan}
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-cyclic-zeckendorf-sector-character-closed-form}，
\[
\mathrm{mult}_6(k)=\frac16\!\Bigl(L_1c_6(k)+L_2c_3(k)+L_3c_2(k)+L_6c_1(k)\Bigr).
\]
代入
\(
(L_1,L_2,L_3,L_6)=(1,3,4,18)
\)
与
\(
c_1(k)=1,\ c_2(k)=(-1)^k
\)
即可逐点得到
\[
\mathrm{mult}_6(0)=5,\ \mathrm{mult}_6(1)=2,\ \mathrm{mult}_6(2)=3,\ \mathrm{mult}_6(3)=3,\ \mathrm{mult}_6(4)=3,\ \mathrm{mult}_6(5)=2.
\]
故结论成立。证毕。
\end{proof}

\begin{theorem}[极大扫描轨道长度对素数集的算术可编程性]\label{thm:conclusion-lmax-prime-programmability}
\leanverified{paper\_conclusion\_lmax\_prime\_programmability}
设 \(\mathcal P\) 为有限素数集，满足
\[
\mathcal P\subseteq\{p:\ p\equiv 2\ (\mathrm{mod}\ 3),\ p\ge 5\}.
\]
定义
\[
\ell(p):=\frac{p+7}{3}\in\ZZ_{\ge 4},
\qquad
x:=\bigsqcup_{p\in\mathcal P}P_{\ell(p)}.
\]
则扫描元 \(c_x\) 的极大轨道长度满足
\[
L_{\max}(x)=\prod_{p\in\mathcal P}p.
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:pom-toggle-scan-max-orbit-arithmetic} 与定理 \ref{thm:pom-toggle-scan-linear-max-orbit}，
\[
L_{\max}(x)=\mathrm{lcm}_{p\in\mathcal P}\!\bigl(3\ell(p)-7\bigr).
\]
对每个 \(p\in\mathcal P\)，由 \(p\equiv 2\pmod 3\) 得 \(\ell(p)\in\ZZ\)，且
\[
3\ell(p)-7=p.
\]
故
\[
L_{\max}(x)=\mathrm{lcm}_{p\in\mathcal P}(p)=\prod_{p\in\mathcal P}p.
\]
证毕。
\end{proof}

\begin{corollary}[极大时间谱不可有限素数化]\label{cor:conclusion-lmax-not-finite-primeizable}
对任意有限素数集 \(S\)，存在纤维型 \(x\) 使得 \(L_{\max}(x)\) 含有 \(S\) 之外的素因子。
更强地，存在单分量 \(P_\ell\) 使 \(L_{\max}(P_\ell)=3\ell-7\) 为素数且不在 \(S\) 中。
\end{corollary}
\begin{proof}
由 Dirichlet 算术级数定理，形如 \(3n+2\) 的素数无穷多。
取 \(p\equiv 2\pmod 3\) 且 \(p\notin S\)，令 \(\ell=(p+7)/3\)。
由定理 \ref{thm:conclusion-lmax-prime-programmability}（取单分量）得
\[
L_{\max}(P_\ell)=3\ell-7=p\notin S.
\]
结论成立。证毕。
\end{proof}

\begin{proposition}[模 \(p\) 奇异性强制 Green 核出现不可避免坏素数]\label{prop:conclusion-modp-singularity-forces-green-bad-prime}
令 \(T\) 为不可约有限马尔可夫链转移矩阵，\(\pi\) 为其平稳分布，并记
\[
A:=I-T+\mathbf 1\pi^{\!\top},
\qquad
Z:=A^{-1}.
\]
设素数 \(p\) 满足：
\begin{enumerate}
  \item \(T,\pi\) 的全部分母与 \(p\) 互素；
  \item \(A\) 在 \(\FF_p\) 上奇异（等价于清分母后行列式模 \(p\) 为零）。
\end{enumerate}
则 \(Z\) 至少有一个条目在既约分数表示中的分母含因子 \(p\)。
等价地，若 \(p\notin S\)，则 \(Z\notin M_n(\ZZ[1/S])\)。
\leanverified{paper\_conclusion\_modp\_singularity\_forces\_green\_bad\_prime}
\end{proposition}
\begin{proof}
由 (1) 可取整数 \(N\) 满足 \((N,p)=1\) 且
\(
M:=NA\in M_n(\ZZ)
\)。
假设结论不成立，即 \(Z\) 的所有条目分母都与 \(p\) 互素，则
\(
Z\in M_n(\ZZ_{(p)})
\)
（\(\ZZ_{(p)}\) 为 \(p\)-局部化）。
于是 \(A\) 在 \(\ZZ_{(p)}\) 上可逆，从而
\(
M=NA
\)
也在 \(\ZZ_{(p)}\) 上可逆（因 \(N\) 为单位），故 \(\det(M)\in \ZZ_{(p)}^\times\)，即 \(p\nmid \det(M)\)。

这与 (2) 矛盾：\(A\) 在 \(\FF_p\) 上奇异等价于 \(M\) 在 \(\FF_p\) 上奇异，进而 \(\det(M)\equiv 0\pmod p\)。
矛盾说明 \(Z\) 至少一项分母含 \(p\)。证毕。
\end{proof}

\begin{theorem}[Kirchhoff--Green 行列式恒等式]\label{thm:conclusion-kirchhoff-green-determinant-identity}
\leanverified{paper\_conclusion\_kirchhoff\_green\_determinant\_identity}
设 \(G=(V,E)\) 为有限连通无向多重图，度矩阵 \(D=\mathrm{diag}(d_v)\)，邻接矩阵 \(A_G\)，Laplacian \(L:=D-A_G\)。
定义
\[
P:=D^{-1}A_G,\qquad
\pi_v:=\frac{d_v}{\mathrm{Vol}(G)},\qquad
\mathrm{Vol}(G):=\sum_{v\in V}d_v,
\]
并记
\[
A:=I-P+\mathbf 1\pi^{\!\top}.
\]
则有精确恒等式
\[
\det(A)=\frac{\tau(G)\,\mathrm{Vol}(G)}{\prod_{v\in V}d_v},
\]
其中 \(\tau(G)\) 为生成树数。
\end{theorem}
\begin{proof}
令 \(d=(d_v)_{v\in V}^{\!\top}\)。有
\[
DA=D(I-P)+D\mathbf 1\pi^{\!\top}
=L+\frac{dd^{\!\top}}{\mathrm{Vol}(G)}.
\]
利用恒等式
\(
\det(M+uv^{\!\top})=\det(M)+v^{\!\top}\!\operatorname{adj}(M)u
\)
（取 \(M=L\), \(u=d\), \(v=d/\mathrm{Vol}(G)\)），得
\[
\det(DA)=\det(L)+\frac1{\mathrm{Vol}(G)}\,d^{\!\top}\!\operatorname{adj}(L)d.
\]
连通图满足 \(\det(L)=0\) 且由矩阵树定理
\(
\operatorname{adj}(L)=\tau(G)\mathbf 1\mathbf 1^{\!\top}
\)。
于是
\[
\det(DA)
=\frac{\tau(G)}{\mathrm{Vol}(G)}(d^{\!\top}\mathbf 1)^2
=\tau(G)\,\mathrm{Vol}(G).
\]
再除以 \(\det(D)=\prod_v d_v\) 得
\[
\det(A)=\frac{\det(DA)}{\det(D)}
=\frac{\tau(G)\,\mathrm{Vol}(G)}{\prod_v d_v}.
\]
证毕。
\end{proof}

\begin{corollary}[临界群扭结与 Green 坏素数的等价判据]\label{cor:conclusion-critical-group-torsion-green-bad-prime-equivalence}
在定理 \ref{thm:conclusion-kirchhoff-green-determinant-identity} 的设定下，取素数 \(p\) 满足
\[
p\nmid \mathrm{Vol}(G)\prod_{v\in V}d_v.
\]
则以下命题等价：
\begin{enumerate}
  \item \(p\mid \tau(G)\)；
  \item \(\det(I-P+\mathbf 1\pi^{\!\top})\equiv 0\pmod p\)；
  \item \(I-P+\mathbf 1\pi^{\!\top}\) 在 \(\FF_p\) 上奇异；
  \item \(Z:=(I-P+\mathbf 1\pi^{\!\top})^{-1}\notin M_n(\ZZ_{(p)})\)（即 \(p\) 为 Green 核的不可消去坏素数）。
\end{enumerate}
特别地，在 window-\(6\) 四块骨架中，\eqref{eq:window6_edge_flux_skeleton_arithmetic} 与 \eqref{eq:window6_green_kernel_audit_prime_571} 给出
\[
\tau(\mathcal G_E)=123336=2^3\cdot 3^3\cdot 571,\qquad
\det(I-T+\mathbf 1\pi^{\!\top})=\frac{2^4\cdot 571}{3^6\cdot 11}.
\]
结合 \eqref{eq:window6_edge_flux_skeleton} 的 \((|U_2|,|U_1|,|U_L|,|U_R|)=(27,22,9,6)\)（故局部归一化素数仅来自 \(\{2,3,11\}\)），可知剔除局部素数后唯一新坏素数为
\[
p_\star=571.
\]
\end{corollary}
\begin{proof}
\((1)\Leftrightarrow(2)\)：由定理 \ref{thm:conclusion-kirchhoff-green-determinant-identity}，且分母在模 \(p\) 下可逆。

\((2)\Leftrightarrow(3)\)：矩阵在域上奇异当且仅当行列式为零。

\((3)\Rightarrow(4)\)：由命题 \ref{prop:conclusion-modp-singularity-forces-green-bad-prime}（取 \(T=P\)）立即得到。

\((4)\Rightarrow(3)\)：若 (3) 不成立，则 \(A:=I-P+\mathbf 1\pi^{\!\top}\) 在 \(\FF_p\) 可逆，故 \(\det(A)\in \ZZ_{(p)}^\times\)。
于是
\(
A^{-1}=\operatorname{adj}(A)/\det(A)\in M_n(\ZZ_{(p)})
\)，与 (4) 矛盾。
故四者等价。window-\(6\) 结论由前述三条证书直接代入即得。证毕。
\end{proof}

\begin{conjecture}[“生成树扭结--谱碰撞--分母坏素数”在一般窗口的刚性同一性]\label{conj:conclusion-tree-torsion-spectral-collision-green-bad-prime-rigidity}
设 \(m\) 为偶窗口，\(G_{E,m}\) 为对应粗粒化骨架图，粗链核与平稳分布分别为 \(T_m,\pi_m\)，并记

A_m:=I-T_m+\mathbf 1\pi_m^{\!\top}.
