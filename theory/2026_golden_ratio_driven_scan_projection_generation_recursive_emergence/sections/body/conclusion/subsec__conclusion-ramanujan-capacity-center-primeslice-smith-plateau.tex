沿用定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion}、
定理 \ref{thm:conclusion-integer-budget-curve-complete-profile}、
定理 \ref{thm:xi-foldbin-minimal-center-valued-index-spectrum-top}、

定理 \ref{thm:xi-layered-primeslice-local-alphabet-fibermax}、

定理 \ref{thm:conclusion-local-alphabet-budget-smith-prefix-budget-nonexchangeability}、

\begin{theorem}[纤维 Ramanujan 阴影对连续容量曲线的线性重建]\label{thm:conclusion-fiber-ramanujan-shadow-capacity-reconstruction}
\leanverified{paper\_conclusion\_fiber\_ramanujan\_shadow\_capacity\_reconstruction}
设 \(f:\Omega\twoheadrightarrow X\) 为有限满射，记
\[
d(x):=\card{f^{-1}(x)},
\qquad
D:=\max_{x\in X}d(x),
\qquad
\mathscr R_f(q):=\sum_{x\in X}c_q(d(x))
\qquad(q\ge 1),
\]
并定义连续阈值容量
\[
C_f(T):=\sum_{x\in X}\min(d(x),T)
\qquad(T\ge 0).
\]
则对任意 \(T\ge 0\)，成立精确重建公式
\[
C_f(T)
=
\sum_{s=1}^{D}\min(s,T)
\sum_{t=1}^{\lfloor D/s\rfloor}
\frac{\mu(t)}{st}\sum_{q\mid st}\mathscr R_f(q).
\]
因此，有限阴影族
\[
\{\mathscr R_f(q)\}_{1\le q\le D}
\]
已经唯一决定整条连续容量曲线 \(T\mapsto C_f(T)\)。
\end{theorem}

\begin{proof}
令
\[
h_f(s):=\#\{x\in X:\ d(x)=s\}
\qquad(1\le s\le D).
\]
则由定义
\[
C_f(T)=\sum_{s=1}^{D}\min(s,T)\,h_f(s).
\]
另一方面，定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion} 给出
\[
h_f(s)
=
\sum_{t=1}^{\lfloor D/s\rfloor}\frac{\mu(t)}{st}\sum_{q\mid st}\mathscr R_f(q).
\]
代回容量展开即得所示公式。由反演唯一性，\(\{\mathscr R_f(q)\}_{1\le q\le D}\) 遂唯一决定整条 \(C_f(\cdot)\)。证毕。
\end{proof}

\begin{theorem}[bin-fold 的容量、成功率与中心谱截断的 Ramanujan 完备性]\label{thm:conclusion-binfold-ramanujan-capacity-success-central-trace}
在 bin-fold 口径下，沿用定理 \ref{thm:xi-foldbin-minimal-center-valued-index-spectrum-top} 的记号，记
\[
d_w:=d_m^{\mathrm{bin}}(w),
\qquad
\mathscr R_m(q):=\sum_{w\in X_m}c_q(d_w),
\qquad
D_m:=\max_{w\in X_m}d_w,
\]
\[
\Lambda_m:=\sum_{w\in X_m}d_w z_w\in Z_m.
\]
则对任意整数预算 \(B\ge 0\)，离散预言机容量、最优成功率与中心谱截断迹满足
\[
C_m(B)=\operatorname{Tr}_{Z_m}\!\bigl(\Lambda_m\wedge 2^B\bigr),
\qquad
\mathrm{Succ}_m(B)=2^{-m}C_m(B),
\]
并且
\[
\operatorname{Tr}_{Z_m}\!\bigl(\Lambda_m\wedge 2^B\bigr)
=
\sum_{s=1}^{D_m}\min(s,2^B)
\sum_{t=1}^{\lfloor D_m/s\rfloor}
\frac{\mu(t)}{st}\sum_{q\mid st}\mathscr R_m(q).
\]
因此，在 bin-fold 的中心值指数元口径下，容量曲线、成功率曲线与谱截断迹并不是三套彼此并列的数据，而是同一有限 Ramanujan 阴影族 \(\{\mathscr R_m(q)\}_{1\le q\le D_m}\) 的不同函子像。
\end{theorem}

\begin{proof}
由定理 \ref{thm:xi-foldbin-oracle-capacity-central-spectral-truncation}，
\[
C_m(B)=\operatorname{Tr}_{Z_m}\!\bigl(\Lambda_m\wedge 2^B\bigr),
\qquad
\mathrm{Succ}_m(B)=2^{-m}C_m(B).
\]
又由 \(\Lambda_m\) 的谱值正是 \(\{d_w\}_{w\in X_m}\)，可把
\[
C_m(B)=\sum_{w\in X_m}\min(d_w,2^B)
\]
视作定理 \ref{thm:conclusion-fiber-ramanujan-shadow-capacity-reconstruction} 在 \(f=\Fold_m^{\mathrm{bin}}\)、\(T=2^B\) 下的专门化。代入即得第三式。证毕。
\end{proof}

\begin{theorem}[中心谱顶端与 Fuglede--Kadison 体积的 Ramanujan 重建]\label{thm:conclusion-center-top-fkdet-ramanujan-reconstruction}
在定理 \ref{thm:conclusion-binfold-ramanujan-capacity-success-central-trace} 的设定下，记
\[
h_m(s):=\#\{w\in X_m:\ d_w=s\}
\qquad(1\le s\le D_m),
\]
\[
M_m:=\max_{w\in X_m}d_w,
\qquad
K_m:=\#\{w\in X_m:\ d_w=M_m\}.
\]
则有：
\begin{enumerate}
  \item 纤维直方图由 Ramanujan 阴影显式恢复：
  \[
  h_m(s)=
  \sum_{t=1}^{\lfloor D_m/s\rfloor}
  \frac{\mu(t)}{st}\sum_{q\mid st}\mathscr R_m(q).
  \]
  \item 顶端谱值满足
  \[
  M_m=\max\{s\ge 1:\ h_m(s)\neq 0\}.
  \]
  \item 顶端几何重数满足
  \[
  K_m=h_m(M_m)=\frac1{M_m}\sum_{q\mid M_m}\mathscr R_m(q).
  \]
  \item 若 \(L_{\Lambda_m}:Z_m\to Z_m\) 为中心左乘算子，则
  \[
  \dim\ker\!\bigl(M_mI_{Z_m}-L_{\Lambda_m}\bigr)=K_m.
  \]
  \item 其 Fuglede--Kadison 对数体积满足
  \[
  \log\Delta_{\tau_m}(\Lambda_m)
  =
  2^{-m}\sum_{s=1}^{M_m}s\log s\,h_m(s).
  \]
\end{enumerate}
因此，中心谱顶端 \((M_m,K_m)\) 与 \(\log\Delta_{\tau_m}(\Lambda_m)\) 也都是有限 Ramanujan 阴影族的显式函子像。
\end{theorem}

\begin{proof}
第一式只是定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion} 在 \(f=\Fold_m^{\mathrm{bin}}\) 下的专门化。第二式由 \(M_m\) 的定义与直方图支撑上界等价得到。

对第三式，由定理 \ref{thm:conclusion-fiber-ramanujan-shadow-divisor-tomography}，
\[
N_m(r):=\#\{w\in X_m:\ r\mid d_w\}
=
\frac1r\sum_{q\mid r}\mathscr R_m(q).
\]
取 \(r=M_m\)。由于 \(M_m\) 已是最大纤维值，所有被 \(M_m\) 整除的 \(d_w\) 必恰等于 \(M_m\)，故
\[
K_m=h_m(M_m)=N_m(M_m)=\frac1{M_m}\sum_{q\mid M_m}\mathscr R_m(q).
\]
第四式正是定理 \ref{thm:xi-foldbin-minimal-center-valued-index-spectrum-top} 第 \(3\) 条。

最后，由定理 \ref{thm:xi-foldbin-fkdet-logvolume-kappa}，
\[
\log\Delta_{\tau_m}(\Lambda_m)
=
2^{-m}\sum_{w\in X_m}d_w\log d_w.
\]
按纤维值 regroup 即得
\[
2^{-m}\sum_{w\in X_m}d_w\log d_w
=
2^{-m}\sum_{s=1}^{M_m}s\log s\,h_m(s).
\]
证毕。
\end{proof}
\leanverified{paper\_conclusion\_center\_top\_fkdet\_ramanujan\_reconstruction}

\begin{proposition}[容量曲线的 Ramanujan 平台律]\label{prop:conclusion-capacity-ramanujan-plateau-law}
设 \(f:\Omega\twoheadrightarrow X\) 为有限满射，记
\[
D:=\max_{x\in X}d(x),
\qquad
\Delta C_f(s):=C_f(s)-C_f(s-1)
\qquad(s\ge 1).
\]
则成立
\[
\Delta C_f(s)=\#\{x\in X:\ d(x)\ge s\}.
\]
从而
\[
\Delta C_f(s)>0 \iff \exists x\in X,\ d(x)\ge s,
\qquad
\Delta C_f(s)=0 \iff s>D.
\]
特别地，
\[
D=\max\{s\ge 1:\ \Delta C_f(s)>0\},
\qquad
h_f(D)=\Delta C_f(D).
\]
并且把定理 \ref{thm:conclusion-fiber-ramanujan-shadow-capacity-reconstruction} 代入后，可得
\[
\Delta C_f(s)
=
\sum_{u=s}^{D}
\sum_{t=1}^{\lfloor D/u\rfloor}
\frac{\mu(t)}{ut}\sum_{q\mid ut}\mathscr R_f(q).
\]
\leanverified{paper\_conclusion\_capacity\_ramanujan\_plateau\_law\_seeds}
\end{proposition}

\begin{proof}
由恒等式
\[
\min(d,s)-\min(d,s-1)=\mathbf 1_{\{d\ge s\}},
\]
逐纤维求和即得
\[
\Delta C_f(s)=\sum_{x\in X}\mathbf 1_{\{d(x)\ge s\}}.
\]
后续各式均为直接推论。最后，由
\[
\Delta C_f(s)=\sum_{u=s}^{D}h_f(u)
\]
并把定理 \ref{thm:conclusion-fiber-ramanujan-shadow-histogram-inversion} 中的 \(h_f(u)\) 公式代入，便得到最后一式。证毕。
\end{proof}

\begin{theorem}[分层 prime-slice 极小字母表的 Ramanujan 平台判别]\label{thm:conclusion-primeslice-minimal-alphabet-ramanujan-plateau}
设
\[
X_0\xrightarrow{\pi_0}X_1\xrightarrow{\pi_1}\cdots\xrightarrow{\pi_{k-1}}X_k
\]
为有限纤维满射塔。对每层 \(j\) 记
\[
d_j(y):=\#\pi_j^{-1}(y)
\qquad(y\in X_{j+1}),
\]
\[
\mathscr R_j(q):=\sum_{y\in X_{j+1}}c_q(d_j(y)),
\qquad
C_j(s):=\sum_{y\in X_{j+1}}\min(d_j(y),s),
\]
\[
B_j:=\max_{y\in X_{j+1}}d_j(y),
\qquad
\Delta C_j(s):=C_j(s)-C_j(s-1).
\]
则有
\[
B_j=\max\{s\ge 1:\ \Delta C_j(s)>0\}.
\]
因此，层 \(j\) 上实现完全可逆所需的最小 prime-slice 标签字母表大小满足
\[
|\Lambda_j|_{\min}=B_j=\max\{s\ge 1:\ \Delta C_j(s)>0\}.
\]
若进一步存在统一上界 \(B_j\le B\)，则最小最大素数与总 distinct prime-slice 库存分别为
\[
\max_{0\le j\le k-1}\max \Lambda_j^{\min}=p_{(2B-1)2^{k-1}},
\qquad
\sum_{j=0}^{k-1}|\Lambda_j|_{\min}
=
\sum_{j=0}^{k-1}\max\{s\ge 1:\ \Delta C_j(s)>0\}.
\]
故 prime-slice 极小可逆标签数与其 prime 库存，都可被解释为 Ramanujan 所决定的容量平台几何。
\end{theorem}

\begin{proof}
对每一层 \(\pi_j\)，把命题 \ref{prop:conclusion-capacity-ramanujan-plateau-law} 应用于有限满射 \(f=\pi_j\)，即得
\[
B_j=\max\{s\ge 1:\ \Delta C_j(s)>0\}.
\]
而定理 \ref{thm:xi-layered-primeslice-local-alphabet-fibermax} 断言
\[
|\Lambda_j|_{\min}=B_j.
\]
合并即得中间公式。

若 \(B_j\le B\)，则关于最小最大素数与总库存的显式公式正是推论 \ref{cor:xi-layered-primeslice-inventory-prime-index-formula} 的内容。把 \(|\Lambda_j|_{\min}=B_j\) 代回即可得到最后一式。证毕。
\end{proof}

\begin{theorem}[Smith \(p\)-主谱的 Ramanujan partial-sum 平台判别]\label{thm:conclusion-smith-ramanujan-partialsum-plateau}
\leanverified{paper\_conclusion\_smith\_ramanujan\_partialsum\_plateau}
设 \(A\in M_{m\times n}(\ZZ)\) 且 \(\operatorname{rank}_{\QQ}(A)=m\)。固定素数 \(p\)，沿用定理 \ref{thm:conclusion-smith-pprimary-ramanujan-shadow-formula} 的记号，定义
\[
\mathcal R_{A,p}(k):=\sum_{i=1}^{m}c_{p^k}(d_i)
\qquad(k\ge 1),
\]
\[
S_p(k):=m+\sum_{j=1}^{k}\mathcal R_{A,p}(j),
\qquad
\Delta_p(k):=\#\{i:\ e_i(p)\ge k\}.
\]
则成立：
\begin{enumerate}
  \item 对每个 \(k\ge 1\)，有
  \[
  \Delta_p(k)=p^{-k}S_p(k).
  \]
  \item 若记
  \[
  E_p:=\max_i e_i(p),
  \]
  则
  \[
  E_p=\max\{k\ge 1:\ S_p(k)>0\},
  \qquad
  S_p(E_p+1)=0.
  \]
  \item 若再记正指数重数
  \[
  m_{p,\ell}:=\#\{i:\ e_i(p)=\ell\},
  \]
  则对 \(1\le \ell<E_p\) 有
  \[
  m_{p,\ell}
  =
  p^{-\ell-1}\Bigl((p-1)S_p(\ell)-\mathcal R_{A,p}(\ell+1)\Bigr),
  \]
  而顶层满足
  \[
  m_{p,E_p}=p^{-E_p}S_p(E_p).
  \]
\end{enumerate}
因此，在未知 \(E_p\) 的情形下，恢复 \(p\)-主 Smith 光谱所需的最短充分 Ramanujan 前缀恰为
\[
\bigl(\mathcal R_{A,p}(1),\dots,\mathcal R_{A,p}(E_p+1)\bigr),
\]
而额外一步的作用正是检测平台消失式
\[
S_p(E_p+1)=0.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-smith-pprimary-ramanujan-shadow-inversion}，
\[
\Delta_p(k)=p^{-k}\left(m+\sum_{j=1}^{k}\mathcal R_{A,p}(j)\right)=p^{-k}S_p(k),
\]
这证明了第一条。

又因为 \(\Delta_p(k)>0\) 当且仅当存在某个 \(i\) 使 \(e_i(p)\ge k\)，故 \(\Delta_p(k)\) 的最后非零位置正是 \(E_p\)。由第一条，\(\Delta_p(k)>0\iff S_p(k)>0\)，从而
\[
E_p=\max\{k\ge 1:\ S_p(k)>0\},
\qquad
S_p(E_p+1)=0.
\]

对第三条，由
\[
m_{p,\ell}=\Delta_p(\ell)-\Delta_p(\ell+1)
\]
与第一条可得
\[
m_{p,\ell}
=
p^{-\ell}S_p(\ell)-p^{-\ell-1}S_p(\ell+1)
=
p^{-\ell-1}\bigl(pS_p(\ell)-S_p(\ell+1)\bigr).
\]
再用
\[
S_p(\ell+1)=S_p(\ell)+\mathcal R_{A,p}(\ell+1)
\]
化简，即得
\[
m_{p,\ell}
=
p^{-\ell-1}\Bigl((p-1)S_p(\ell)-\mathcal R_{A,p}(\ell+1)\Bigr)
\qquad(1\le \ell<E_p).
\]
而在顶层 \(\ell=E_p\) 时，由 \(\Delta_p(E_p+1)=0\) 或等价地 \(S_p(E_p+1)=0\)，得到
\[
m_{p,E_p}=\Delta_p(E_p)=p^{-E_p}S_p(E_p).
\]

最后，定理 \ref{thm:conclusion-local-alphabet-budget-smith-prefix-budget-nonexchangeability} 与定理 \ref{thm:conclusion-smith-local-zeta-shortest-prefix} 已证明：若 \(E_p\) 未知，则最短充分前缀恰需多读取一步平台检测值。上式表明，在 Ramanujan 坐标中，这额外一步正对应于检验
\[
S_p(E_p+1)=0.
\]
证毕。
\end{proof}

\begin{conjecture}[Ramanujan 资源二重性]\label{conj:conclusion-ramanujan-resource-duality}
在本文的有限纤维塔与相关 Smith \(p\)-核族中，一切 visible 侧的局部可逆化资源与 hidden \(p\)-adic 侧的局部恢复资源，应分别由两类平台序列完全统辖：
\[
\Delta C_f(s)=C_f(s)-C_f(s-1),
\qquad
S_p(k)=m+\sum_{j=1}^{k}\mathcal R_{A,p}(j).
\]
更具体地说：
\begin{enumerate}
  \item 任一仅依赖于纤维多重度谱的资源量，包括连续容量、整数预算成功率、中心谱截断迹、顶端谱值/顶端重数以及 prime-slice 极小字母表库存，都应当可由有限纤维 Ramanujan 阴影 \(\{\mathscr R_f(q)\}_{q\le D}\) 函子化得到；
  \item 任一仅依赖于 Smith \(p\)-主赋值谱的资源量，包括最短充分前缀、平台检测门槛与正指数重数，都应当可由 prime-power Ramanujan 影 \(\{\mathcal R_{A,p}(k)\}_{k\ge 1}\) 函子化得到；
  \item 并且这两类 Ramanujan 坐标都应在“删去任一项即失去统一完备性”的意义下达到极小。
\end{enumerate}
若该猜想成立，则本文中的容量、谱截断、字母表大小与最短前缀这几类资源量，可被统一压缩为一套最小的 Ramanujan 资源语法。
\end{conjecture}
