\begin{theorem}[primitive 语义轴上规范 Gödel 吸引子的轨道饱和与 Hutchinson 唯一性]\label{thm:conclusion-tate-rankone-canonical-chart-orbit-hutchinson-rigidity}
\leanverified{paper\_conclusion\_tate\_rankone\_canonical\_chart\_orbit\_hutchinson\_rigidity}
记
\[
T:=T_2(E_{\min})\cong \ZZ_2^2.
\]
固定 primitive 向量
\[
v\in T\setminus 2T,
\qquad
T_v:=\ZZ_2v\subset T,
\qquad
B:=2^L>M,
\]
并定义
\[
\Xi_{B,M}(a_1,a_2,\dots):=\sum_{t\ge 1}a_tB^{t-1}v,
\qquad
K_{B,M}(v):=\Xi_{B,M}\bigl(\{0,\dots,M\}^{\NN}\bigr).
\]
则：
\begin{enumerate}
  \item 规范前缀轨道闭包满足
  \[
  \overline{\{\Psi(r,k)(0):\ (r,k)\in H_{\mathrm{can}}\}}
  =
  K_{B,M}(v);
  \]
  \item \(K_{B,M}(v)\) 是 self-affine 方程
  \[
  Y=\bigsqcup_{a=0}^{M}\bigl(av+BY\bigr)
  \]
  的唯一非空紧解。
\end{enumerate}
\end{theorem}
\begin{proof}
第 \(1\) 条正是定理 \ref{thm:xi-time-part9gc-canonical-godel-cantor-attractor-orbit-closure} 的第 \(3\) 条。

对第 \(2\) 条，记
\[
\tau_a(x):=av+Bx
\qquad (0\le a\le M).
\]
由于 \(T_v\cong \ZZ_2\) 是完备超度量紧群，且
\[
|\tau_a(x)-\tau_a(y)|_2=|B|_2\,|x-y|_2=2^{-L}|x-y|_2,
\]
每个 \(\tau_a\) 都是 Lipschitz 常数 \(2^{-L}<1\) 的严格压缩。故由 Hutchinson 不动点定理，
\[
Y=\bigcup_{a=0}^{M}\tau_a(Y)
\]
至多存在一个非空紧解。另一方面，定理 \ref{thm:xi-time-part9gc-canonical-godel-cantor-attractor-orbit-closure} 已给出
\[
K_{B,M}(v)=\bigsqcup_{a=0}^{M}\bigl(av+BK_{B,M}(v)\bigr),
\]
故该唯一解就是 \(K_{B,M}(v)\)。证毕。
\end{proof}

\begin{theorem}[分辨率秩与字值阈值的二预算正交]\label{thm:conclusion-tate-rank-digit-threshold-orthogonality}
\leanverified{paper\_conclusion\_tate\_rank\_digit\_threshold\_orthogonality}
设一个编码资源向量由
\[
\mathfrak R=(r,C)
\]
给出，其中 \(r\) 表示承载整条 dyadic 分辨率塔所需的自由 \(\ZZ_2\)-模秩，而 \(C=2^s\) 表示仿射字值
\[
\Xi_C(a_1,\dots,a_k)(x):=
C^k x+\sum_{t=1}^{k}a_tC^{t-1}v
\]
的基数阈值。则：
\begin{enumerate}
  \item 若某自由 \(\ZZ_2\)-模 \(\mathcal M\cong \ZZ_2^r\) 对全部 \(n\ge 1\) 都满足
  \[
  \mathcal M/2^n\mathcal M\cong E_{\min}[2^n],
  \]
  则必有
  \[
  r=2;
  \]
  \item 仿射字值映射
  \[
  \Xi_C:\bigsqcup_{k\ge 0}\{0,\dots,M\}^k\longrightarrow \operatorname{Aff}(T_v)
  \]
  忠实当且仅当
  \[
  C>M.
  \]
\end{enumerate}
特别地，ambient 秩与基数字阈是两条不可互偿的资源轴：增大 \(C\) 无法替代 \(r=2\)，提高 \(r\) 也无法修复 \(C\le M\) 的字值碰撞。
\end{theorem}
\begin{proof}
第 \(1\) 条正是推论 \ref{cor:xi-time-part9gc-resolution-tower-rank-two-rigidity}。第 \(2\) 条正是定理 \ref{thm:xi-time-part9gc-canonical-affine-word-faithful-threshold}。

若两类资源可以互偿，则应存在 \(r<2\) 而仅靠提高 \(C\) 仍忠实承载整条分辨率塔，或存在 \(C\le M\) 而仅靠提高 \(r\) 仍消除字值碰撞。前者与第 \(1\) 条矛盾，后者与第 \(2\) 条矛盾。故两条预算严格正交。证毕。
\end{proof}

\begin{theorem}[primitive 轴上自仿射 thickening 的 Haar 刚性]\label{thm:conclusion-tate-rankone-selfaffine-thickening-haar-rigidity}
\leanverified{paper\_conclusion\_tate\_rankone\_selfaffine\_thickening\_haar\_rigidity}
设 \(Y\subset T_v\) 为非空紧集，并满足
\[
Y=\bigcup_{a=0}^{M}\bigl(av+BY\bigr).
\]
则必有
\[
Y=K_{B,M}(v).
\]
特别地：
\begin{enumerate}
  \item 若 \(M=B-1\)，则
  \[
  Y=T_v;
  \]
  \item 若 \(M\le B-2\)，则
  \[
  \mu_{T_v}(Y)=0,
  \qquad
  \operatorname{Int}_{T_v}(Y)=\varnothing.
  \]
\end{enumerate}
因而，在 primitive 语义轴上，不存在位于 \(K_{B,M}(v)\) 与 \(T_v\) 之间的正 Haar 测度 self-affine thickening。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-tate-rankone-canonical-chart-orbit-hutchinson-rigidity}，上述 self-affine 方程的非空紧解唯一，故 \(Y=K_{B,M}(v)\)。再由定理 \ref{thm:xi-time-part9gc-canonical-godel-attractor-dimension-haar-dichotomy}，若 \(M=B-1\) 则
\[
K_{B,M}(v)=T_v,
\]
若 \(M\le B-2\) 则
\[
\mu_{T_v}\bigl(K_{B,M}(v)\bigr)=0,
\qquad
\operatorname{Int}_{T_v}\bigl(K_{B,M}(v)\bigr)=\varnothing.
\]
代入 \(Y=K_{B,M}(v)\) 即得。证毕。
\end{proof}

\begin{corollary}[codimension defect 的五重等价]\label{cor:conclusion-tate-rankone-codimension-defect-fivefold-equivalence}
\leanverified{paper\_conclusion\_tate\_rankone\_codimension\_defect\_fivefold\_equivalence}
定义
\[
\delta(B,M):=
1-\frac{\log(M+1)}{\log B}
=
\frac{L-\log_2(M+1)}{L},
\qquad
B=2^L.
\]
则下列五项彼此等价：
\[
\delta(B,M)=0;
\qquad
M=B-1;
\qquad
K_{B,M}(v)=T_v;
\qquad
\mu_{T_v}\bigl(K_{B,M}(v)\bigr)>0;
\qquad
\operatorname{Int}_{T_v}\bigl(K_{B,M}(v)\bigr)\neq\varnothing.
\]
因此，\(\delta(B,M)\) 不是附加统计量，而是 primitive 轴语义系统是否填满整条 \(2\)-进地址线的精确判据。
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-time-part9gc-canonical-godel-attractor-dimension-haar-dichotomy}，
\[
\dim_{\mathrm H}K_{B,M}(v)=\frac{\log(M+1)}{\log B},
\]
故
\[
\delta(B,M)=0
\iff
\log(M+1)=\log B
\iff
M+1=B
\iff
M=B-1.
\]
再由定理 \ref{thm:conclusion-tate-rankone-selfaffine-thickening-haar-rigidity}，\(M=B-1\) 等价于
\[
K_{B,M}(v)=T_v,
\]
并等价于正 Haar 测度与非空内部；而当 \(M\le B-2\) 时，后二者同时失败。证毕。
\end{proof}

\begin{theorem}[全塔 Haar 几何与 rank-one 地址影子的层兼容不可共轭性]\label{thm:conclusion-tate-fulltower-rankone-shadow-nonconjugacy}
记
\[
T:=T_2(E_{\min})\cong \ZZ_2^2,
\]
并以 \(\mu_{\mathrm{res}}\) 表示其归一化 Haar 概率测度。则不存在同胚
\[
\Theta:T\xrightarrow{\sim} K_{B,M}(v)
\]
同时满足下列两项：
\begin{enumerate}
  \item \(\Theta\) 把分辨率塔上的规范有限层仿射作用与 \(K_{B,M}(v)\) 上的规范前缀作用相互交织；
  \item \(\Theta\) 在每个有限层都与自然截断结构相容。
\end{enumerate}
换言之，全秩 Haar 分辨率塔与 rank-one 规范地址影子在层兼容范畴中绝不共轭。
\end{theorem}
\begin{proof}
反设存在这样的 \(\Theta\)。由于 \(\Theta\) 与全部有限层截断相容，\(K_{B,M}(v)\) 的第 \(n\) 层可见 residue 类数必须与
\[
E_{\min}[2^n]\cong (\ZZ/2^n\ZZ)^2
\]
一致，因而对每个 \(n\ge 1\) 都应等于 \(4^n\)。另一方面，
\[
K_{B,M}(v)\subseteq T_v\cong \ZZ_2
\]
始终被困在单一 rank-one 语义轴内，而
\[
\#(T_v/2^nT_v)=2^n.
\]
故任何来自 \(K_{B,M}(v)\) 的第 \(n\) 层截断至多只有 \(2^n\) 个 residue 类，不可能与 full-rank 分辨率塔的 \(4^n\) 个类逐层相容。故此类层兼容共轭不可能存在。证毕。
\end{proof}

\begin{conjecture}[rank-one 语义影子的完备性]\label{conj:conclusion-tate-rankone-shadow-completeness}
设 \(Y\subset T\) 为紧集，并满足：
\begin{enumerate}
  \item \(Y\) 是某个 finitely generated 规范仿射子半群对基点 \(0\) 的轨道闭包；
  \item \(Y\) 包含于某条 primitive 轴 \(T_v\subset T\) 中；
  \item \(Y\) 与全部有限层截断保持规范地址兼容性。
\end{enumerate}
则应存在唯一参数对
\[
(B,M),
\qquad
B=2^L>M,
\]
使得 \(Y\) 与 \(K_{B,M}(v)\) 在 \(T_v\) 上仿射共轭。

该猜想断言：在 rank-one 语义影子范畴中，所有层兼容的紧轨道闭包都已由规范 Gödel 吸引子 \(K_{B,M}(v)\) 完全穷尽。
\end{conjecture}

\begin{theorem}[分辨率塔与 rank-one 地址影子的二分刚性]\label{thm:conclusion-tate-resolution-shadow-bifurcation-rigidity}
在 Tate 模块框架中，同时存在如下两条不可混同的刚性链：
\[
\text{full resolution tower}
\Longrightarrow
T\cong \ZZ_2^2,\quad
\mu_{\mathrm{res}}=\text{Haar},\quad
\operatorname{rank}_{\ZZ_2}=2;
\]
\[
\text{canonical rank-one address shadow}
\Longrightarrow
K_{B,M}(v)\subset T_v\cong \ZZ_2,\quad
\dim_{\mathrm H}K_{B,M}(v)=\frac{\log(M+1)}{\log B},\quad
\Xi_C\ \text{faithful}\iff C>M.
\]
更具体地：
\begin{enumerate}
  \item 全塔一致性强制 ambient \(\ZZ_2\)-rank 恰为 \(2\)；
  \item primitive 轴上的规范地址几何由唯一的 self-affine 吸引子 \(K_{B,M}(v)\) 承载；
  \item 零 codimension defect、满地址覆盖、正 Haar 测度与非空内部彼此等价；
  \item 当 \(M\le B-2\) 时，rank-one 影子必为 Haar-null 且 nowhere dense 的真紧子系统；若进一步 \(M\ge 1\)，则它是 Cantor 子系统；
  \item full-rank Haar 分辨率塔与 rank-one 语义影子之间不存在层兼容共轭。
\end{enumerate}
因此，full-rank Haar 几何与 rank-one 地址影子不是同一对象的两个坐标，而是由不同资源约束与不同刚性原则锁定的两个范畴层。
\end{theorem}
\begin{proof}
第 \(1\) 条由定理 \ref{thm:conclusion-tate-rank-digit-threshold-orthogonality} 的第 \(1\) 条给出。第 \(2\) 条由定理 \ref{thm:conclusion-tate-rankone-canonical-chart-orbit-hutchinson-rigidity} 给出。第 \(3\) 条由推论 \ref{cor:conclusion-tate-rankone-codimension-defect-fivefold-equivalence} 给出。对第 \(4\) 条，定理 \ref{thm:conclusion-tate-rankone-selfaffine-thickening-haar-rigidity} 已给出 Haar 零测与空内部；而紧集空内部即 nowhere dense。若 \(M\ge 1\)，则由定理 \ref{thm:xi-time-part9gc-canonical-godel-cantor-attractor-orbit-closure}，\(K_{B,M}(v)\) 与 \(\{0,\dots,M\}^{\NN}\) 同胚，故它是 Cantor 子系统。第 \(5\) 条由定理 \ref{thm:conclusion-tate-fulltower-rankone-shadow-nonconjugacy} 给出。合并即得。证毕。
\end{proof}

\begin{theorem}[有限阴影完备而逆极限空洞的可数 rank-one 花束]\label{thm:conclusion-tate-rankone-finite-shadow-full-inverse-limit-hollow}
\leanverified{paper\_conclusion\_tate\_rankone\_finite\_shadow\_full\_inverse\_limit\_hollow}
仍记
\[
T:=T_2(E_{\min})\cong \ZZ_2^2,
\qquad
\rho_n:T\to T/2^nT,
\]
并以 \(\mu_{\mathrm{res}}\) 表示 \(T\) 的归一化 Haar 概率测度。则存在一个可数集合
\[
\mathcal V\subset T\setminus 2T,
\]
使得
\[
Y:=\bigcup_{v\in\mathcal V}T_v,
\qquad
T_v:=\ZZ_2v,
\]
满足对一切 \(n\ge 1\) 都有
\[
\rho_n(Y)=T/2^nT,
\]
但同时
\[
\mu_{\mathrm{res}}(Y)=0,
\qquad
Y\ \text{为第一纲集}.
\]
因此，可数个 primitive rank-one 语义轴已经能够在每一个有限 dyadic 阴影上实现满覆盖，但在逆极限 full-rank 几何中仍只是一个 Haar-null 且 meagre 的花束。
\end{theorem}
\begin{proof}
对每个 \(m\ge 1\)，记
\[
\mathcal P_m:=
\left((\ZZ/2^m\ZZ)^2\right)_{\mathrm{prim}}
=
\left\{
\bar u\in (\ZZ/2^m\ZZ)^2:\ \bar u\notin 2(\ZZ/2^m\ZZ)^2
\right\}.
\]
对每个 \(\bar u\in \mathcal P_m\)，任取一个 lift
\[
v_{m,\bar u}\in T\setminus 2T,
\qquad
\rho_m(v_{m,\bar u})=\bar u,
\]
并令
\[
\mathcal V:=\{v_{m,\bar u}:\ m\ge 1,\ \bar u\in \mathcal P_m\}.
\]
由于每个 \(\mathcal P_m\) 有限，故 \(\mathcal V\) 可数。

现证 \(\rho_n(Y)=T/2^nT\)。任取 \(\bar x\in T/2^nT\)。若 \(\bar x=0\)，则显然 \(\bar x\in \rho_n(T_v)\) 对任意 \(v\in \mathcal V\) 都成立。设 \(\bar x\neq 0\)。取最大整数
\[
m\in \{0,\dots,n-1\}
\]
使
\[
\bar x\in 2^m(T/2^nT)\setminus 2^{m+1}(T/2^nT).
\]
则存在 primitive 类
\[
\bar y\in \left(\ZZ/2^{n-m}\ZZ\right)^2_{\mathrm{prim}}
\]
满足
\[
\bar x=2^m\bar y.
\]
按构造可取
\[
v:=v_{n-m,\bar y}\in \mathcal V,
\qquad
\rho_{n-m}(v)=\bar y.
\]
于是
\[
\rho_n(2^mv)=2^m\bar y=\bar x,
\]
而 \(2^mv\in T_v\subset Y\)。故任意 \(\bar x\) 都属于 \(\rho_n(Y)\)，遂得
\[
\rho_n(Y)=T/2^nT.
\]

再证测度与 Baire 断言。由定理 \ref{thm:xi-time-part9gc-haar-cantor-splitting} 的证明，对任意 primitive \(v\) 与任意 \(n\ge 1\)，都有
\[
\#\rho_n(T_v)\le 2^n,
\qquad
\mu_{\mathrm{res}}(T_v)\le 2^n\cdot 4^{-n}=2^{-n}.
\]
令 \(n\to\infty\) 得
\[
\mu_{\mathrm{res}}(T_v)=0.
\]
在紧群 \(T\) 中，任一非空开集都具有正 Haar 测度，故 \(T_v\) 不可能含有内点；又 \(T_v\) 为闭子模，因此它是 nowhere dense。于是
\[
\mu_{\mathrm{res}}(Y)
\le
\sum_{v\in \mathcal V}\mu_{\mathrm{res}}(T_v)
=0,
\]
并且 \(Y\) 是可数个 nowhere dense 闭集的并，从而为第一纲集。证毕。
\end{proof}

\begin{corollary}[有限层审计完备性不推出 residual 完备性]\label{cor:conclusion-tate-finite-shadow-complete-not-residual-complete}
\leanverified{paper\_conclusion\_tate\_finite\_shadow\_complete\_not\_residual\_complete}
存在集合 \(Y\subset T\) 使得对每个 \(n\ge 1\),
\[
\rho_n(Y)=\rho_n(T),
\]
但其补集
\[
T\setminus Y
\]
同时是稠密的并且具有满 Haar 测度。因而，任何只依赖有限层 residue-shadow 的审计规则，都无法区分 \(Y\) 与整个 \(T\)；然而在逆极限层面，二者相差一个稠密满测集。
\end{corollary}
\begin{proof}
取定理 \ref{thm:conclusion-tate-rankone-finite-shadow-full-inverse-limit-hollow} 中的 \(Y\)。则 \(\rho_n(Y)=T/2^nT=\rho_n(T)\) 对一切 \(n\) 成立。另一方面，
\[
\mu_{\mathrm{res}}(Y)=0
\]
给出
\[
\mu_{\mathrm{res}}(T\setminus Y)=1.
\]
又因为 \(Y\) 为第一纲集而 \(T\) 为紧度量群，故 \(T\) 是 Baire 空间，从而 \(T\setminus Y\) 为 comeagre，特别稠密。证毕。
\end{proof}

\begin{theorem}[可数仿射语义图册的 Haar--Baire 消失]\label{thm:conclusion-tate-countable-affine-semantic-atlas-null-meagre}
\leanverified{paper\_conclusion\_tate\_countable\_affine\_semantic\_atlas\_null\_meagre}
设
\[
\{(B_i,M_i,v_i,A_i)\}_{i\in I}
\]
为至多可数族，其中
\[
B_i=2^{L_i},
\qquad
0\le M_i<B_i,
\qquad
v_i\in T\setminus 2T,
\]
且
\[
A_i:T\to T
\]
为连续仿射自映射。令
\[
U:=
\bigcup_{i\in I}A_i\bigl(K_{B_i,M_i}(v_i)\bigr).
\]
则
\[
\mu_{\mathrm{res}}(U)=0,
\qquad
U\ \text{为第一纲集}.
\]
特别地，任何可数个 canonical semantic charts 的仿射后处理，都不可能恢复整个 residual Tate 模块 \(T\)。
\end{theorem}
\begin{proof}
对每个 \(i\in I\)，由推论 \ref{cor:xi-time-part9gc-affine-nonrecoverability-fullrank-space}，
\[
A_i\bigl(K_{B_i,M_i}(v_i)\bigr)
\]
在 \(T\) 中始终 Haar 零测且内部为空。又因 \(K_{B_i,M_i}(v_i)\) 为紧集，连续像
\[
A_i\bigl(K_{B_i,M_i}(v_i)\bigr)
\]
仍为紧集，从而是闭集。闭且内部为空即意味着它是 nowhere dense。于是每个图册块都是 Haar-null 的 nowhere dense 紧集，故其可数并 \(U\) 仍然满足
\[
\mu_{\mathrm{res}}(U)=0,
\qquad
U\ \text{为第一纲集}.
\]
证毕。
\end{proof}

\begin{corollary}[residual 完备语义图册的不可数阈值]\label{cor:conclusion-tate-semantic-atlas-uncountable-threshold}
\leanverified{paper\_conclusion\_tate\_semantic\_atlas\_uncountable\_threshold}

\end{proof}

\begin{theorem}[单轴熵饱和与满秩 residual 几何的严格脱耦]\label{thm:conclusion-tate-uniaxial-entropy-saturation-rank-decoupling}
取任意 primitive 向量 \(v\in T\setminus 2T\)，并令
\[
M=B-1.
\]
则 canonical chart 达到其单轴允许的最大字增长：
\[
K_{B,B-1}(v)=T_v,
\qquad
\mathcal N_k=B^k,
\qquad
h_{\mathrm{grow}}=\log B.
\]
然而即使在这一“饱和”情形，仍有
\[
\mu_{\mathrm{res}}\bigl(K_{B,B-1}(v)\bigr)=0,
\qquad
\operatorname{Int}_T K_{B,B-1}(v)=\varnothing,
\]
并且对任意连续仿射映射 \(A:T\to T\),
\[
A\bigl(K_{B,B-1}(v)\bigr)\neq T.
\]
因此，单一语义轴上的熵饱和并不逼近 residual 空间的秩饱和；它只能填满一条 \(\ZZ_2\)-线，而永远不能填满整个 rank-two Tate 模块。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part9gc-canonical-godel-attractor-dimension-haar-dichotomy}，
\[
K_{B,B-1}(v)=T_v.
\]
再由定理 \ref{thm:xi-time-part9gc-canonical-affine-semigroup-word-growth}，
\[
\mathcal N_k=(M+1)^k=B^k,
\qquad
h_{\mathrm{grow}}=\log B.
\]
这表明单轴图册在自身轴内已达到最大字增长。

另一方面，\(T_v\) 是 \(T\cong \ZZ_2^2\) 的 rank-one 闭子模。由定理 \ref{thm:xi-time-part9gc-haar-cantor-splitting} 的证明，
\[
\mu_{\mathrm{res}}(T_v)=0.
\]
在紧群 \(T\) 中，Haar 零测集不可能含有内点，故
\[
\operatorname{Int}_T K_{B,B-1}(v)
=
\operatorname{Int}_T T_v
=
\varnothing.
\]
最后，由推论 \ref{cor:xi-time-part9gc-affine-nonrecoverability-fullrank-space}，
\[
A\bigl(K_{B,B-1}(v)\bigr)
\]
始终 Haar 零测且内部为空，特别不可能等于整个 \(T\)。证毕。
\end{proof}

\begin{proposition}[primitive-line 图册的连续统实现]\label{prop:conclusion-tate-primitive-line-atlas-continuum-upper-bound}
\leanverified{paper\_conclusion\_tate\_primitive\_line\_atlas\_continuum\_upper\_bound}
定义
\[
\mathcal L:=\{T_v:\ v\in T\setminus 2T\}.
\]
则有覆盖
\[
T=\bigcup_{L\in \mathcal L}L.
\]
并且 \(\mathcal L\) 具有连续统基数。特别地，存在一个基数为 \(2^{\aleph_0}\) 的 rank-one primitive-line 图册覆盖整个 residual 空间。
\end{proposition}
\begin{proof}
任取 \(x\in T\)。若 \(x=0\)，则显然 \(x\in T_v\) 对任意 primitive \(v\) 成立。设 \(x\neq 0\)。由于 \(T\cong \ZZ_2^2\) 无挠，存在唯一整数
\[
\nu_2(x):=\max\{m\ge 0:\ x\in 2^mT\}.
\]
令
\[
v:=2^{-\nu_2(x)}x.
\]
则 \(v\notin 2T\)，即 \(v\) primitive，且
\[
x=2^{\nu_2(x)}v\in T_v.
\]
故
\[
T=\bigcup_{v\in T\setminus 2T}T_v
=
\bigcup_{L\in \mathcal L}L.
\]

现证基数断言。对每个 \(a\in \ZZ_2\)，向量 \((1,a)\) 都是 primitive，并生成直线
\[
T_{(1,a)}=\{(x,ax):\ x\in \ZZ_2\}.
\]
若 \(a\neq b\)，则
\[
T_{(1,a)}\neq T_{(1,b)},
\]
因为一旦两者相等，就有 \((1,a)\in T_{(1,b)}\)，从而存在 \(u\in \ZZ_2\) 满足 \((1,a)=u(1,b)\)，第一坐标迫使 \(u=1\)，遂得 \(a=b\)，矛盾。由于 \(\ZZ_2\) 具有连续统基数，故 \(\mathcal L\) 至少有连续统多个不同元素；另一方面 \(\mathcal L\subseteq \mathcal P(T)\) 也至多具有连续统基数，因而
\[
|\mathcal L|=2^{\aleph_0}.
\]
证毕。
\end{proof}

\begin{conjecture}[residual 完备语义图册的连续统极小性]\label{conj:conclusion-tate-rankone-semantic-atlas-minimal-continuum}
任取一个由 canonical semantic charts 与连续仿射后处理构成的覆盖
\[
T=\bigcup_{j\in J}A_j\bigl(K_{B_j,M_j}(v_j)\bigr).
\]
则其基数应当满足
\[
|J|=2^{\aleph_0}.
\]
换言之，一旦坚持 rank-one semantic charts，residual 完备语义图册的极小阈值不只是“不可数”，而应当恰好等于连续统。
\end{conjecture}

\noindent
定理 \ref{thm:conclusion-tate-rankone-finite-shadow-full-inverse-limit-hollow} 至命题 \ref{prop:conclusion-tate-primitive-line-atlas-continuum-upper-bound} 说明，在 \(T_2(E_{\min})\cong \ZZ_2^2\) 上，finite-shadow 完备性、单轴字增长饱和与 inverse-limit residual 饱和是三种彼此不同的完备性。可数个 primitive rank-one 图册已经能在每一个有限层上实现全覆盖，甚至单条坐标轴在 \(M=B-1\) 时还达到最大字增长 \(\log B\)；然而这些对象在 full-rank Haar 几何中依旧只是 Haar-null 且 meagre 的 rank-one 花束。于是，真正控制语义图册基数的不是有限阴影 universality，也不是单轴熵的饱和，而是 inverse-limit 层面对 residual 秩二结构的全局饱和。
