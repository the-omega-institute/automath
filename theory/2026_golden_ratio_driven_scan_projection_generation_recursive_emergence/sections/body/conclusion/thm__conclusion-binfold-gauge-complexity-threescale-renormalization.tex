\begin{theorem}[bin-fold 规范群复杂度的三尺度不可约分裂]\label{thm:conclusion-binfold-gauge-complexity-three-scale-separation}
记
\[
\Xi_m:=\log k(G_m),
\qquad
\Gamma_m:=-\log \mathrm{cp}(G_m),
\qquad
\Sigma_m:=\log_2|H_2(G_m,\ZZ)|.
\]
则存在显式常数
\[
C_{\mathrm{conj}}
:=
\frac{\pi}{\sqrt5}\sqrt{\frac23}\left(\varphi+\varphi^{-1/2}\right),
\qquad
C_{\mathrm{vol}}
:=
1+\frac{\log\varphi}{\varphi\sqrt5},
\]
使得
\[
\Xi_m
=
C_{\mathrm{conj}}(2\varphi)^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr),
\]
\[
\Gamma_m
=
2^m\left(m\log\frac{2}{\varphi}-C_{\mathrm{vol}}\right)
-C_{\mathrm{conj}}(2\varphi)^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr),
\]
并且当 \(m\) 充分大时
\[
H_2(G_m,\ZZ)\cong (\ZZ/2\ZZ)^{\,n_m(n_m+1)/2},
\qquad
\Sigma_m=\frac12\,n_m(n_m+1),
\qquad
n_m=F_{m+2}.
\]
从而
\[
\lim_{m\to\infty}\frac1m\log \Xi_m=\frac12\log(2\varphi),
\qquad
\lim_{m\to\infty}\frac1m\log \Gamma_m=\log 2,
\qquad
\lim_{m\to\infty}\frac1m\log \Sigma_m=2\log\varphi,
\]
并且三者满足严格尺度分离
\[
\Xi_m=o(\Gamma_m),
\qquad
\Gamma_m=o(\Sigma_m).
\]
\end{theorem}
\begin{proof}
前两条渐近分别就是定理 \ref{thm:conclusion-binfold-gauge-conjugacy-midscale-asymptotic} 与定理 \ref{thm:conclusion-binfold-gauge-commuting-probability-superexponential}。
第三条则由定理 \ref{thm:conclusion-binfold-gauge-schur-multiplier-quadratic-explosion} 直接给出：当 \(m\) 足够大时，
\[
\Sigma_m=\log_2|H_2(G_m,\ZZ)|=\frac12\,n_m(n_m+1).
\]
又由 Binet 公式，
\[
n_m=F_{m+2}=\frac{\varphi^{m+2}}{\sqrt5}+O(\varphi^{-m}),
\]
故
\[
\Sigma_m=\Theta(\varphi^{2m}).
\]
类似地，
\[
\Xi_m=\Theta\!\bigl((2\varphi)^{m/2}\bigr),
\qquad
\Gamma_m=\Theta(m2^m).
\]
于是
\[
\lim_{m\to\infty}\frac1m\log \Xi_m=\frac12\log(2\varphi),
\qquad
\lim_{m\to\infty}\frac1m\log \Gamma_m=\log 2,
\qquad
\lim_{m\to\infty}\frac1m\log \Sigma_m=2\log\varphi.
\]
最后，由
\[
\sqrt{2\varphi}<2<\varphi^2
\]
立得
\[
\Xi_m=o(\Gamma_m),
\qquad
\Gamma_m=o(\Sigma_m).
\]
证毕。
\end{proof}

\begin{proposition}[bin-fold 规范群复杂度不存在单参数重整化]\label{prop:conclusion-binfold-gauge-complexity-no-single-parameter-renormalization}
\leanverified{paper\_conclusion\_binfold\_gauge\_complexity\_no\_single\_parameter\_renormalization}
不存在正实数列 \(\{a_m\}_{m\ge 1}\) 与三个非零有限常数 \(c_1,c_2,c_3\)，使得同时成立
\[
\frac{\Xi_m}{a_m}\to c_1,
\qquad
\frac{\Gamma_m}{a_m}\to c_2,
\qquad
\frac{\Sigma_m}{a_m}\to c_3.
\]
\end{proposition}
\begin{proof}
若存在这样的 \(a_m,c_1,c_2,c_3\)，则
\[
\Xi_m\asymp a_m,
\qquad
\Gamma_m\asymp a_m,
\qquad
\Sigma_m\asymp a_m,
\]
从而
\[
\Xi_m\asymp \Gamma_m\asymp \Sigma_m.
\]
但定理 \ref{thm:conclusion-binfold-gauge-complexity-three-scale-separation} 已给出
\[
\Xi_m=o(\Gamma_m),
\qquad
\Gamma_m=o(\Sigma_m),
\]
矛盾。证毕。
\end{proof}
