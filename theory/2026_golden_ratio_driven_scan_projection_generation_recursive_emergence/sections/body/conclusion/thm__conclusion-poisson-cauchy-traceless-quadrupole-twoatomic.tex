\begin{theorem}[Poisson--Cauchy 二阶信息指纹的 traceless quadrupole 因子化]\label{thm:conclusion-poisson-cauchy-traceless-quadrupole-factorization}
\leanverified{paper\_conclusion\_poisson\_cauchy\_traceless\_quadrupole\_factorization}
沿用本小节记号，定义
\[
\Sigma_\mu:=
\begin{pmatrix}
\Var_\mu(\Gamma) & \Cov_\mu(\Gamma,\Delta)\\
\Cov_\mu(\Gamma,\Delta) & \Var_\mu(\Delta)
\end{pmatrix},
\qquad
Q_\mu:=\Sigma_\mu-\frac{\operatorname{tr}\Sigma_\mu}{2}I_2
=
\begin{pmatrix}
A/2 & B\\
B & -A/2
\end{pmatrix},
\]
其中
\[
A=\Var_\mu(\Gamma)-\Var_\mu(\Delta),
\qquad
B=\Cov_\mu(\Gamma,\Delta),
\qquad
s:=t+\bar\Delta.
\]
则
\[
|Q_\mu|_{HS}^2=\frac{A^2}{2}+2B^2,
\]
并且全部二阶信息常数都只通过 \(Q_\mu\) 出现：
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t)=\frac14|Q_\mu|_{HS}^2,
\]
\[
\lim_{t\to\infty}s^5I_1(h_t\mid g_t)=|Q_\mu|_{HS}^2,
\]
\[
\lim_{t\to\infty}s^4\inf_{m,\sigma}D_{\mathrm{KL}}\!\bigl(h_t\mid P_\sigma(\cdot-m)\bigr)=\frac18|Q_\mu|_{HS}^2.
\]
\end{theorem}

\begin{proof}
由定义
\[
Q_\mu=
\begin{pmatrix}
A/2 & B\\
B & -A/2
\end{pmatrix},
\]
故
\[
|Q_\mu|_{HS}^2
=
2\left(\frac{A}{2}\right)^2+2B^2
=
\frac{A^2}{2}+2B^2.
\]
由定理 \ref{thm:conclusion-poisson-bivariate-kl-sharp}，
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t)
=
\frac{A^2}{8}+\frac{B^2}{2}
=
\frac14\left(\frac{A^2}{2}+2B^2\right)
=
\frac14|Q_\mu|_{HS}^2.
\]
由定理 \ref{thm:conclusion-poisson-bivariate-fisher-sharp}，
\[
\lim_{t\to\infty}s^5I_1(h_t\mid g_t)
=
\frac{A^2}{2}+2B^2
=
|Q_\mu|_{HS}^2.
\]
再由定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving}，
\[
\lim_{t\to\infty}s^4\inf_{m,\sigma}D_{\mathrm{KL}}\!\bigl(h_t\mid P_\sigma(\cdot-m)\bigr)
=
\frac{A^2}{16}+\frac{B^2}{4}
=
\frac18|Q_\mu|_{HS}^2.
\]
证毕。
\end{proof}

\begin{corollary}[二阶信息常数的 exact \texorpdfstring{$(1\!:\!2\!:\!4)$}{(1:2:4)} 律与渐近勾股分裂]\label{cor:conclusion-poisson-cauchy-exact-124-pythagoras}
\leanverified{paper\_conclusion\_poisson\_cauchy\_exact\_124\_pythagoras}
沿用定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving} 的最优参数 \((m_t,s_t)\)，并记
\[
g_t^\star(x):=P_{s_t}(x-m_t).
\]
则有精确比例律
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t)
=
2\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t^\star)
=
\frac14\lim_{t\to\infty}s^5I_1(h_t\mid g_t).
\]
更强地，
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(g_t^\star\mid g_t)
=
\frac18|Q_\mu|_{HS}^2
=
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t^\star),
\]
从而成立渐近勾股分裂
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t)
=
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t^\star)
+
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(g_t^\star\mid g_t).
\]
\end{corollary}

\begin{proof}
前三个比例等式直接由定理 \ref{thm:conclusion-poisson-cauchy-traceless-quadrupole-factorization} 中三个常数
\[
\frac14|Q_\mu|_{HS}^2,\qquad
\frac18|Q_\mu|_{HS}^2,\qquad
|Q_\mu|_{HS}^2
\]
的显式比例得到。

现证关于 \(D_{\mathrm{KL}}(g_t^\star\mid g_t)\) 的公式。沿用定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving} 证明中的重标定
\[
m=\bar\Gamma+\frac{\alpha}{s},
\qquad
\sigma=s+\frac{\beta}{s},
\]
并记
\[
a(y):=\frac{2y}{1+y^2},
\qquad
w(y):=\frac{y^2-1}{1+y^2},
\qquad
G(y):=\frac{1}{\pi(1+y^2)}.
\]
则最优点满足
\[
\alpha^\star=-B,\qquad \beta^\star=\frac{A}{2},
\]
且
\[
\frac{g_t^\star}{g_t}-1
=
\frac{1}{s^2}\bigl(\alpha^\star a(y)+\beta^\star w(y)\bigr)+o_{L^2(G\dd y)}(s^{-2}).
\]
由
\[
(1+u)\log(1+u)=u+\frac12u^2+o(u^2)
\]
与 \(\int G a=\int Gw=0\) 得
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(g_t^\star\mid g_t)
=
\frac12\int_{\RR}G(y)\bigl(\alpha^\star a(y)+\beta^\star w(y)\bigr)^2\dd y.
\]
又由定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving} 证明中的积分常数，
\[
\int_{\RR}G\,a^2\,\dd y=\frac12,\qquad
\int_{\RR}G\,w^2\,\dd y=\frac12,\qquad
\int_{\RR}G\,aw\,\dd y=0.
\]
因此
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(g_t^\star\mid g_t)
=
\frac{(\alpha^\star)^2}{4}+\frac{(\beta^\star)^2}{4}
=
\frac{B^2}{4}+\frac{A^2}{16}
=
\frac18|Q_\mu|_{HS}^2.
\]
而定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving} 已给出
\[
\lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t^\star)=\frac{A^2}{16}+\frac{B^2}{4},
\]
故二者相等。再与原始 KL 常数
\[
\frac14|Q_\mu|_{HS}^2
=
\frac18|Q_\mu|_{HS}^2+\frac18|Q_\mu|_{HS}^2
\]
相加关系合并，即得渐近勾股分裂。证毕。
\end{proof}

\begin{theorem}[最优参数漂移对 traceless quadrupole 的完备反演]\label{thm:conclusion-poisson-cauchy-optimal-drift-complete-inversion}
沿用定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving} 的最优参数 \((m_t,s_t)\)。则极限
\[
u_\mu:=
\left(
\lim_{t\to\infty}s(\bar\Gamma-m_t),\
\lim_{t\to\infty}2s(s_t-s)
\right)
\]
存在，且满足
\[
u_\mu=(B,A).
\]
特别地，
\[
Q_\mu=
\begin{pmatrix}
u_{\mu,2}/2 & u_{\mu,1}\\
u_{\mu,1} & -u_{\mu,2}/2
\end{pmatrix},
\]
故最优参数漂移向量 \(u_\mu\) 与 traceless quadrupole \(Q_\mu\) 线性同构。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving}，
\[
m_t=\bar\Gamma-\frac{B}{s}+o(s^{-1}),
\qquad
s_t=s+\frac{A}{2s}+o(s^{-1}).
\]
故
\[
s(\bar\Gamma-m_t)=B+o(1),
\qquad
2s(s_t-s)=A+o(1).
\]
取极限即得
\[
u_\mu=(B,A).
\]
再代回
\[
Q_\mu=
\begin{pmatrix}
A/2 & B\\
B & -A/2
\end{pmatrix}
\]
即可得到所示表达式。证毕。
\end{proof}

\begin{theorem}[二阶指纹的 metric-vanishing 六重等价]\label{thm:conclusion-poisson-cauchy-second-order-fingerprint-metric-vanishing}
\leanverified{paper\_conclusion\_poisson\_cauchy\_second\_order\_fingerprint\_metric\_vanishing}
对任意满足定理 \ref{thm:conclusion-poisson-bivariate-kl-sharp} 假设的 \(\mu\)，以下六个条件彼此等价：
\begin{enumerate}
  \item \(Q_\mu=0\)；
  \item \((A,B)=(0,0)\)；
  \item \(\displaystyle \lim_{t\to\infty}s^4D_{\mathrm{KL}}(h_t\mid g_t)=0\)；
  \item \(\displaystyle \lim_{t\to\infty}s^5I_1(h_t\mid g_t)=0\)；
  \item \(\displaystyle \lim_{t\to\infty}s^4\inf_{m,\sigma}D_{\mathrm{KL}}(h_t\mid P_\sigma(\cdot-m))=0\)；
  \item \(\displaystyle \lim_{t\to\infty}s^2\|h_t-g_t\|_{L^1}=0\)。
\end{enumerate}
并且它们还等价于最优参数漂移消失：
\[
\lim_{t\to\infty}s(\bar\Gamma-m_t)=0,
\qquad
\lim_{t\to\infty}2s(s_t-s)=0.
\]
\end{theorem}

\begin{proof}
\((1)\iff(2)\) 由 \(Q_\mu\) 的定义显然。
\((2)\iff(3)\)、\((2)\iff(4)\)、\((2)\iff(5)\) 由定理 \ref{thm:conclusion-poisson-cauchy-traceless-quadrupole-factorization} 立即得到。
最优参数漂移消失与 \((A,B)=(0,0)\) 的等价性由定理 \ref{thm:conclusion-poisson-cauchy-optimal-drift-complete-inversion} 立即给出。

现证 \((2)\iff(6)\)。由定理 \ref{thm:conclusion-poisson-bivariate-tv-sharp}，
\[
\lim_{t\to\infty}s^2\|h_t-g_t\|_{L^1}
=
\frac{1}{\pi}\int_{\RR}
\frac{|A(3y^2-1)-2By(3-y^2)|}{(1+y^2)^3}\,\dd y.
\]
右端为零当且仅当分子多项式
\[
A(3y^2-1)-2By(3-y^2)
=
2By^3+3Ay^2-6By-A
\]
恒等于零。比较 \(y^3,y^2,y,1\) 四项系数，即得 \(A=B=0\)。反向显然。证毕。
\end{proof}

\begin{theorem}[二阶信息谱只依赖于 covariance 的 traceless 类]\label{thm:conclusion-poisson-cauchy-traceless-class-sufficiency}
\leanverified{paper\_conclusion\_poisson\_cauchy\_traceless\_class\_sufficiency}
设
\[
\Sigma_\mu=
\begin{pmatrix}
\Var(\Gamma) & \Cov(\Gamma,\Delta)\\
\Cov(\Gamma,\Delta) & \Var(\Delta)
\end{pmatrix}.
\]
则定理 \ref{thm:conclusion-poisson-bivariate-kl-sharp}、推论 \ref{cor:conclusion-poisson-bivariate-fdiv-sharp}、定理 \ref{thm:conclusion-poisson-bivariate-fisher-sharp}、定理 \ref{thm:conclusion-poisson-bivariate-tv-sharp}、定理 \ref{thm:conclusion-poisson-bivariate-information-projection-halving} 与定理 \ref{thm:conclusion-poisson-cauchy-optimal-drift-complete-inversion} 中出现的全部二阶信息观测量——包括 raw KL 常数、任意平滑 \(f\)-散度常数、分数 Fisher 常数、最优投影 KL 常数、全变差常数以及最优参数漂移——都只通过
\[
[\Sigma_\mu]:=\Sigma_\mu \bmod \RR I_2
\]
出现；等价地，只通过 \(Q_\mu\) 出现。

特别地，若
\[
\Sigma_{\mu'}=\Sigma_\mu+\sigma^2 I_2
\qquad(\sigma^2\ge 0),
\]
则两模型具有完全相同的全部二阶信息指纹。
\end{theorem}

\begin{proof}
由定义
\[
Q_\mu=\Sigma_\mu-\frac{\operatorname{tr}\Sigma_\mu}{2}I_2,
\]
它正是 \(\Sigma_\mu\) 在商空间 \(\mathrm{Sym}_2(\RR)/\RR I_2\) 中的代表。定理 \ref{thm:conclusion-poisson-cauchy-traceless-quadrupole-factorization}、推论 \ref{cor:conclusion-poisson-cauchy-exact-124-pythagoras}、定理 \ref{thm:conclusion-poisson-cauchy-optimal-drift-complete-inversion} 与定理 \ref{thm:conclusion-poisson-cauchy-second-order-fingerprint-metric-vanishing} 已表明所有所列二阶观测量都可由 \(Q_\mu\) 唯一写出，故只依赖于该 traceless 类。

若
\[
\Sigma_{\mu'}=\Sigma_\mu+\sigma^2 I_2,
\]
则
\[
Q_{\mu'}
=
\Sigma_{\mu'}-\frac{\operatorname{tr}\Sigma_{\mu'}}{2}I_2
=
\Sigma_\mu+\sigma^2 I_2-\frac{\operatorname{tr}\Sigma_\mu+2\sigma^2}{2}I_2
=
Q_\mu.
\]
故全部二阶信息指纹一致。命题 \ref{prop:conclusion-poisson-bivariate-isotropic-noise-invariance} 正是这一事实在“加入独立中心等方差噪声”口径下的具体实例。证毕。
\end{proof}

\begin{theorem}[traceless quadrupole 模空间的双原子满射]\label{thm:conclusion-poisson-cauchy-traceless-plane-twoatomic-surjection}
\leanverified{paper\_conclusion\_poisson\_cauchy\_traceless\_plane\_twoatomic\_surjection}
给定任意
\[
(A,B)\in\RR^2.
\]
令
\[
R:=\sqrt{A^2+4B^2},
\qquad
u:=\sqrt{\frac{R+A}{2}},
\qquad
v:=\operatorname{sgn}(B)\sqrt{\frac{R-A}{2}},
\]
其中 \(B=0\) 时任取 \(\operatorname{sgn}(0)\in\{\pm1\}\)。则
\[
u^2-v^2=A,
\qquad
uv=B.
\]
因此，对任意 \(\bar\Gamma\in\RR\) 与任意 \(\bar\Delta>|v|\)，两原子测度
\[
\mu_{A,B}:=
\frac12\delta_{(\bar\Gamma+u,\ \bar\Delta+v)}
+
\frac12\delta_{(\bar\Gamma-u,\ \bar\Delta-v)}
\]
满足
\[
\EE\Gamma=\bar\Gamma,\qquad \EE\Delta=\bar\Delta,
\]
且其二阶不变量恰为给定的 \((A,B)\)。
换言之，\((A,B)\)-模空间可由正的双原子模型完全充满。
\end{theorem}

\begin{proof}
由构造，
\[
u^2-v^2
=
\frac{R+A}{2}-\frac{R-A}{2}
=
A.
\]
又
\[
u|v|
=
\sqrt{\frac{R+A}{2}\frac{R-A}{2}}
=
\sqrt{\frac{R^2-A^2}{4}}
=
|B|,
\]
故按 \(v\) 的符号选择得 \(uv=B\)。

对上述双原子测度，中心化变量分别为
\[
(\Gamma-\bar\Gamma,\Delta-\bar\Delta)\in\{(u,v),(-u,-v)\}
\]
且各以概率 \(1/2\) 取到。于是
\[
\Var(\Gamma)=u^2,\qquad
\Var(\Delta)=v^2,\qquad
\Cov(\Gamma,\Delta)=uv=B.
\]
因此
\[
\Var(\Gamma)-\Var(\Delta)=u^2-v^2=A.
\]
再由 \(\bar\Delta>|v|\)，两原子的第二坐标都严格正，故 \(\mu_{A,B}\) 的确支撑于 \(\RR\times(0,\infty)\)。证毕。
\end{proof}

\begin{corollary}[任意有限点缺陷 Poisson 场的二阶双原子规范形]\label{cor:conclusion-point-defect-poisson-twoatomic-normal-form}
\leanverified{paper\_conclusion\_point\_defect\_poisson\_twoatomic\_normal\_form}
沿用定理 \ref{thm:conclusion-point-defect-poisson-entropy-fingerprint} 的点缺陷 Poisson 场记号
\[
G_t(x)=\sum_{j=1}^{\kappa}m_j\delta_j\,P_{t+\delta_j}(x-\gamma_j),
\qquad
\mu=\frac1{M_0}\sum_{j=1}^{\kappa}m_j\delta_j\,\delta_{(\gamma_j,\delta_j)}.
\]
则存在一条仅含两个原子的测度
\[
\mu^\sharp=
\frac12\delta_{(\bar\Gamma+u,\bar\Delta+v)}
+
\frac12\delta_{(\bar\Gamma-u,\bar\Delta-v)}
\]
使得：
\begin{enumerate}
  \item \(\mu^\sharp\) 与 \(\mu\) 具有相同的一阶矩 \((\bar\Gamma,\bar\Delta)\)；
  \item \(\mu^\sharp\) 与 \(\mu\) 具有相同的 \(Q\)-矩阵，故也具有相同的 \((A,B)\)；
  \item 相应 Poisson 混合密度 \(h_t^\sharp\) 与 \(h_t\) 在定理 \ref{thm:conclusion-poisson-cauchy-traceless-quadrupole-factorization}、推论 \ref{cor:conclusion-poisson-cauchy-exact-124-pythagoras}、定理 \ref{thm:conclusion-poisson-cauchy-second-order-fingerprint-metric-vanishing} 与定理 \ref{thm:conclusion-poisson-cauchy-traceless-class-sufficiency} 所列全部二阶信息观测量上完全一致。
\end{enumerate}
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-point-defect-poisson-entropy-fingerprint}，点缺陷 Poisson 场的 raw KL 二阶常数只通过由 \(\mu\) 定义的 \((\bar\Gamma,\bar\Delta,A,B)\) 进入。对这一对 \((A,B)\)，定理 \ref{thm:conclusion-poisson-cauchy-traceless-plane-twoatomic-surjection} 构造出一条双原子测度 \(\mu^\sharp\)，它与 \(\mu\) 具有相同的一阶矩且满足
\[
Q_{\mu^\sharp}=Q_\mu.
\]
再由定理 \ref{thm:conclusion-poisson-cauchy-traceless-class-sufficiency}，所有二阶信息观测量都只依赖于 \(Q\)-类，故 \(\mu^\sharp\) 与 \(\mu\) 对应的全部二阶观测完全一致。证毕。
\end{proof}

\begin{theorem}[全变差二阶常数在 traceless quadrupole 平面上定义真范数]\label{thm:conclusion-poisson-cauchy-tv-second-order-norm}
\leanverified{paper\_conclusion\_poisson\_cauchy\_tv\_second\_order\_norm}
在 \(\mathrm{Sym}_2^0(\RR)\) 上定义
\[
|Q|_{TV}
:=
\frac{1}{\pi}
\int_{\RR}
\frac{|A(3y^2-1)-2By(3-y^2)|}{(1+y^2)^3}\,\dd y,
\qquad
Q=
\begin{pmatrix}
A/2 & B\\
B & -A/2
\end{pmatrix}.
\]
则 \(|\cdot|_{TV}\) 是 \(\mathrm{Sym}_2^0(\RR)\) 上的真范数，并且
\[
\lim_{t\to\infty}s^2\|h_t-g_t\|_{L^1}=|Q_\mu|_{TV}.
\]
其坐标轴标定常数精确为
\[
\left|
\begin{pmatrix}
A/2&0\\[1mm]0&-A/2
\end{pmatrix}
\right|_{TV}
=
\frac{3\sqrt3}{4\pi}|A|,
\]
\[
\left|
\begin{pmatrix}
0&B\\[1mm]B&0
\end{pmatrix}
\right|_{TV}
=
\frac{5}{2\pi}|B|.
\]
\end{theorem}

\begin{proof}
映射
\[
Q\longmapsto
\frac{A(3y^2-1)-2By(3-y^2)}{(1+y^2)^3}
\]
是从 \(\mathrm{Sym}_2^0(\RR)\) 到 \(L^1(\RR)\) 的线性映射。因此 \(|\cdot|_{TV}\) 的正齐次性与三角不等式都直接来自 \(L^1\)-范数。
定正性由定理 \ref{thm:conclusion-poisson-cauchy-second-order-fingerprint-metric-vanishing} 中
\[
\lim_{t\to\infty}s^2\|h_t-g_t\|_{L^1}=0
\iff
A=B=0
\iff
Q=0
\]
给出，故 \(|\cdot|_{TV}\) 是真范数。

等式
\[
\lim_{t\to\infty}s^2\|h_t-g_t\|_{L^1}=|Q_\mu|_{TV}
\]
只是定理 \ref{thm:conclusion-poisson-bivariate-tv-sharp} 用 \(Q_\mu\) 的坐标重写。
最后两条轴向常数正是该定理中 \(B=0\) 与 \(A=0\) 两个特例。证毕。
\end{proof}
