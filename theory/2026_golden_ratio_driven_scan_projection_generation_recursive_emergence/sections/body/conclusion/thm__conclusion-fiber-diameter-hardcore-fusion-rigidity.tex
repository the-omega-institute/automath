\begin{theorem}[纤维多重度的直径熵刚性与立方维二倍黄金斜率]\label{thm:conclusion-fiber-multiplicity-diameter-cubedim-rigidity}
\leanverified{paper\_conclusion\_fiber\_multiplicity\_diameter\_cubedim\_rigidity}
对任意 \(x\in X_m\)，设
\[
\Gamma_x\cong\bigsqcup_{i=1}^{r(x)}P_{\ell_i(x)},
\qquad
\mathcal F_x\cong \prod_{i=1}^{r(x)}\Gamma_{\ell_i(x)}
\]
为其纤维重构图的路径分量分解与 Fibonacci cube 的 Cartesian 分解。则有严格估计
\[
\varphi^{\mathrm{diam}(\mathcal F_x)}
\le d_m(x)\le
\varphi^{\mathrm{diam}(\mathcal F_x)+r(x)},
\]
\[
\varphi^{2\dim_\square(\mathcal F_x)-r(x)}
\le d_m(x)\le
\varphi^{2\dim_\square(\mathcal F_x)+r(x)}.
\]
等价地，
\[
0\le
\log d_m(x)-(\log\varphi)\,\mathrm{diam}(\mathcal F_x)
\le r(x)\log\varphi,
\]
\[
\bigl|
\log d_m(x)-2(\log\varphi)\dim_\square(\mathcal F_x)
\bigr|
\le r(x)\log\varphi.
\]
特别地，沿任意满足 \(r(x)=o(\mathrm{diam}(\mathcal F_x))\) 的族，有
\[
\frac{\log d_m(x)}{\mathrm{diam}(\mathcal F_x)}\longrightarrow \log\varphi,
\]
而若 \(r(x)=o(\dim_\square(\mathcal F_x))\)，则
\[
\frac{\log d_m(x)}{\dim_\square(\mathcal F_x)}\longrightarrow 2\log\varphi.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-reconstruction-cartesian-product}，
\[
\mathrm{diam}(\mathcal F_x)=\sum_{i=1}^{r(x)}\ell_i(x),
\qquad
\dim_\square(\mathcal F_x)=\sum_{i=1}^{r(x)}\left\lceil\frac{\ell_i(x)}{2}\right\rceil.
\]
另一方面，
\[
d_m(x)=|\mathcal F_x|=\prod_{i=1}^{r(x)}|\Gamma_{\ell_i(x)}|
=\prod_{i=1}^{r(x)}F_{\ell_i(x)+2}.
\]
由 Binet 公式可知对每个 \(\ell\ge 0\) 都有
\[
\varphi^\ell\le F_{\ell+2}\le \varphi^{\ell+1}.
\]
逐项相乘即得
\[
\varphi^{\sum_i\ell_i}\le d_m(x)\le \varphi^{\sum_i(\ell_i+1)}
=\varphi^{\mathrm{diam}(\mathcal F_x)+r(x)},
\]
从而得到第一组估计。

再由
\[
2\left\lceil\frac{\ell}{2}\right\rceil-1\le \ell\le 2\left\lceil\frac{\ell}{2}\right\rceil
\]
对每个 \(\ell\ge 0\) 成立，得到
\[
\sum_i \ell_i(x)\ge 2\dim_\square(\mathcal F_x)-r(x),
\qquad
\sum_i (\ell_i(x)+1)\le 2\dim_\square(\mathcal F_x)+r(x).
\]
与前述 \(F_{\ell+2}\) 的上下界合并，即得第二组估计。最后两条极限式由两侧同除以相应尺度并令误差比趋于零即得。证毕。
\end{proof}

\begin{theorem}[纤维后验配分函数的路径闭式与单根支配]\label{thm:conclusion-fiber-hardcore-partition-pressure-single-root}
\leanverified{paper\_conclusion\_fiber\_hardcore\_partition\_pressure\_single\_root}
固定 \(p\in(0,1)\)，记
\[
t:=\frac{p}{1-p},
\qquad
\lambda_\pm(t):=\frac{1\pm \sqrt{1+4t}}{2}.
\]
对任意 \(x\in X_m\)，令
\[
Z_x(t):=\sum_{S\in \mathsf{Ind}(\Gamma_x)} t^{|S|}.
\]
则有严格乘积分解
\[
Z_x(t)=\prod_{i=1}^{r(x)} I_{\ell_i(x)}(t),
\]
其中路径独立集多项式 \(I_\ell(t)\) 满足
\[
I_0(t)=1,\qquad
I_1(t)=1+t,\qquad
I_\ell(t)=I_{\ell-1}(t)+t\,I_{\ell-2}(t)\qquad (\ell\ge 2),
\]
并具有闭式
\[
I_\ell(t)=
\frac{\lambda_+(t)^{\ell+2}-\lambda_-(t)^{\ell+2}}
{\lambda_+(t)-\lambda_-(t)}.
\]
进一步，对每个固定 \(t>0\)，存在仅依赖于 \(t\) 的常数 \(c_t,C_t>0\)，使得对一切 \(x\in X_m\) 都有
\[
c_t^{\,r(x)}\lambda_+(t)^{\mathrm{diam}(\mathcal F_x)}
\le
Z_x(t)
\le
C_t^{\,r(x)}\lambda_+(t)^{\mathrm{diam}(\mathcal F_x)}.
\]
从而存在常数 \(A_t>0\) 使
\[
\bigl|
\log Z_x(t)-(\log\lambda_+(t))\,\mathrm{diam}(\mathcal F_x)
\bigr|
\le A_t\,r(x).
\]
\end{theorem}
\begin{proof}
由 \(\Gamma_x\cong\bigsqcup_i P_{\ell_i(x)}\) 可知独立集系统按分量分解为笛卡尔乘积，故配分函数满足
\[
Z_x(t)=\prod_{i=1}^{r(x)}\left(\sum_{S_i\in \mathsf{Ind}(P_{\ell_i(x)})} t^{|S_i|}\right)
=\prod_{i=1}^{r(x)} I_{\ell_i(x)}(t).
\]
对单条路径 \(P_\ell\)，按末端顶点是否被占据分类即得递推
\[
I_\ell(t)=I_{\ell-1}(t)+t\,I_{\ell-2}(t).
\]
其特征方程 \(\lambda^2=\lambda+t\) 的两根为 \(\lambda_\pm(t)\)，从而得到所示闭式。

再由
\[
I_\ell(t)
=
\frac{\lambda_+(t)^{\ell+2}}{\lambda_+(t)-\lambda_-(t)}
\left(
1-\left(\frac{\lambda_-(t)}{\lambda_+(t)}\right)^{\ell+2}
\right)
\]
以及 \(|\lambda_-(t)|<\lambda_+(t)\)，可知括号中的因子在 \(\ell\ge 0\) 上一致夹在两个只依赖于 \(t\) 的正常数之间。故存在 \(c_t,C_t>0\) 使
\[
c_t\,\lambda_+(t)^\ell\le I_\ell(t)\le C_t\,\lambda_+(t)^\ell.
\]
对全部分量相乘并用
\[
\mathrm{diam}(\mathcal F_x)=\sum_i\ell_i(x)
\]
即得所述乘法估计；取对数便得最后一式。证毕。
\end{proof}

\begin{theorem}[纤维后验占据数的显式热力学极限]\label{thm:conclusion-fiber-hardcore-occupancy-thermodynamic-limit}
\leanverified{paper\_conclusion\_fiber\_hardcore\_occupancy\_thermodynamic\_limit}
沿用定理 \ref{thm:conclusion-fold-fiber-hardcore-annealed-conservation} 的记号。对任意 \(x\in X_m\)，在活动度 \(t>0\) 的纤维后验下令
\[
S\sim P_{x,t},
\qquad
R_x:=|S|.
\]
则对任意实数 \(s\)，沿任意满足 \(r(x)=o(\mathrm{diam}(\mathcal F_x))\) 的族都有
\[
\frac{1}{\mathrm{diam}(\mathcal F_x)}
\log \mathbb E_{x,t}\!\left[e^{sR_x}\right]
\longrightarrow
\Psi_t(s):=
\log\frac{\lambda_+(te^s)}{\lambda_+(t)}.
\]
因此
\[
\frac{\mathbb E_{x,t}[R_x]}{\mathrm{diam}(\mathcal F_x)}
\longrightarrow
\Psi_t'(0)
=
\frac12\left(1-\frac{1}{\sqrt{1+4t}}\right),
\]
\[
\frac{\operatorname{Var}_{x,t}(R_x)}{\mathrm{diam}(\mathcal F_x)}
\longrightarrow
\Psi_t''(0)
=
\frac{t}{(1+4t)^{3/2}}.
\]
特别地，\(\{R_x/\mathrm{diam}(\mathcal F_x)\}\) 满足 G\"artner--Ellis 型大偏差原理，其速率函数为 \(\Psi_t^\ast\)。
\end{theorem}
\begin{proof}
由 hard-core 后验定义，
\[
\mathbb E_{x,t}[e^{sR_x}]
=
\frac{1}{Z_x(t)}\sum_{S\in\mathsf{Ind}(\Gamma_x)} (te^s)^{|S|}
=
\frac{Z_x(te^s)}{Z_x(t)}.
\]
将定理 \ref{thm:conclusion-fiber-hardcore-partition-pressure-single-root} 同时应用于分子与分母，得到
\[
\frac{1}{\mathrm{diam}(\mathcal F_x)}
\log \mathbb E_{x,t}[e^{sR_x}]
=
\log\frac{\lambda_+(te^s)}{\lambda_+(t)}
+
O\!\left(\frac{r(x)}{\mathrm{diam}(\mathcal F_x)}\right).
\]
在 \(r(x)=o(\mathrm{diam}(\mathcal F_x))\) 下，误差趋于零，从而得到极限 \(\Psi_t(s)\)。

对 \(\Psi_t(s)=\log\lambda_+(te^s)-\log\lambda_+(t)\) 求导，注意
\[
\lambda_+'(u)=\frac{1}{\sqrt{1+4u}},
\]
便得
\[
\Psi_t'(0)=\frac{t}{\lambda_+(t)\sqrt{1+4t}}
=\frac12\left(1-\frac{1}{\sqrt{1+4t}}\right),
\]
\[
\Psi_t''(0)=\frac{t}{(1+4t)^{3/2}}.
\]
由于 \(\Psi_t\) 在全实轴解析，G\"artner--Ellis 定理给出所述大偏差原理。证毕。
\end{proof}

\begin{corollary}[Bernoulli 观测下的纤维统计同构判别]\label{cor:conclusion-fiber-bernoulli-posterior-statistical-isomorphism}
\leanverified{paper\_conclusion\_fiber\_bernoulli\_posterior\_statistical\_isomorphism}
对 \(x,y\in X_m\)，定义谱多重集
\[
\operatorname{Spec}(x):=\{\ell_1(x),\dots,\ell_{r(x)}(x)\},
\qquad
\operatorname{Spec}(y):=\{\ell_1(y),\dots,\ell_{r(y)}(y)\}.
\]
则：
\begin{enumerate}
  \item 若 \(\operatorname{Spec}(x)=\operatorname{Spec}(y)\)，则对任意 \(p\in(0,1)\)，两条纤维后验
  \[
  \mu_{m,p}(\,\cdot\mid x),
  \qquad
  \mu_{m,p}(\,\cdot\mid y)
  \]
  在各自纤维与独立集空间的规范同构下彼此测度同构；
  \item 若再有 \(|x|_1=|y|_1\)，则推前概率满足
  \[
  \nu_{m,p}(x)=\nu_{m,p}(y).
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
若 \(\operatorname{Spec}(x)=\operatorname{Spec}(y)\)，则 \(\Gamma_x,\Gamma_y\) 同构为同一组路径的不交并，故由定理 \ref{thm:conclusion-fold-fiber-hardcore-annealed-conservation}，两条后验都被规范识别为该路径并族上活动度 \(t=p/(1-p)\) 的同一个 hard-core 模型，从而测度同构。

进一步，由定理 \ref{thm:conclusion-fold-fiber-hardcore-annealed-conservation}，
\[
\nu_{m,p}(x)=p^{|x|_1}(1-p)^{m-|x|_1}Z_x(t),
\qquad
\nu_{m,p}(y)=p^{|y|_1}(1-p)^{m-|y|_1}Z_y(t).
\]
而 \(\operatorname{Spec}(x)=\operatorname{Spec}(y)\) 蕴含 \(Z_x(t)=Z_y(t)\)，若再有 \(|x|_1=|y|_1\)，则上式相等。证毕。
\end{proof}

\begin{theorem}[单偏置后验的刚性完备性]\label{thm:conclusion-single-bias-posterior-complete-fiber-statistic}
\leanverified{paper\_conclusion\_single\_bias\_posterior\_complete\_fiber\_statistic}
固定 \(m\ge 2\) 与 \(x\in X_m\)。取任意 \(p\in(0,1)\)，并记
\[
t:=\frac{p}{1-p},
\qquad
Z_x(z):=\sum_{\omega\in \Fold_m^{-1}(x)} z^{r(\omega)}.
\]
令 \(R_{x,p}\) 表示纤维后验 \(\mu_{m,p}(\,\cdot\mid x)\) 下的逆代换次数随机变量。则单个边缘分布 \(\mathcal L(R_{x,p})\) 已唯一决定以下全部对象：
\begin{enumerate}
  \item 纤维生成多项式 \(Z_x(z)\)；
  \item \(Z_x\) 的负实根多重集 \(\{\alpha_{x,j}\}_j\)；
  \item 路径长度多重集 \(\{\ell_i(x)\}_{i=1}^{r(x)}\)；
  \item 纤维格
  \[
  \mathcal L_x\cong \prod_{i=1}^{r(x)}J(Z_{\ell_i(x)}).
  \]
\end{enumerate}
换言之，对固定 \(x\) 而言，任一单一非平凡偏置的后验逆代换次数分布，已是该纤维全部离散几何的完备统计量。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-biased-posterior-poisson-binomial}，
\[
\EE[s^{R_{x,p}}\mid x]=\frac{Z_x(ts)}{Z_x(t)}.
\]
因此 \(\mathcal L(R_{x,p})\) 唯一决定概率母函数
\[
G_{x,p}(s):=\EE[s^{R_{x,p}}\mid x].
\]
又由 \(Z_x(0)=1\) 可知
\[
G_{x,p}(0)=\frac{1}{Z_x(t)},
\]
从而
\[
Z_x(z)=\frac{G_{x,p}(z/t)}{G_{x,p}(0)}.
\]
故 \(Z_x\) 被唯一恢复。

再由定理 \ref{thm:pom-fiber-rewrite-realroot-ulc}，\(Z_x\) 的全部零点皆为负实根，且可写为
\[
Z_x(z)=\prod_{j}\left(1+\frac{z}{\alpha_{x,j}}\right),
\qquad
\alpha_{x,j}>0.
\]
于是 \(G_{x,p}(s)\) 的零点恰为 \(-\alpha_{x,j}/t\)，故 \(\mathcal L(R_{x,p})\) 亦唯一决定根多重集 \(\{\alpha_{x,j}\}_j\)。

最后，由定理 \ref{thm:pom-fiber-psi-exponent-mobius-reconstruct}，\(Z_x\) 的 cyclotomic 指数向量唯一恢复路径长度多重集 \(\{\ell_i(x)\}\)；再由定理 \ref{thm:pom-fiber-birkhoff-fence-ideal-lattice}，该长度多重集唯一决定纤维格
\[
\mathcal L_x\cong \prod_{i=1}^{r(x)}J(Z_{\ell_i(x)}).
\]
故四项对象均由 \(\mathcal L(R_{x,p})\) 唯一决定。证毕。
\end{proof}

\begin{theorem}[纤维 canonical rowmotion 的精确周期公式]\label{thm:conclusion-fiber-canonical-rowmotion-order-law}\leanverified{paper\_conclusion\_fiber\_canonical\_rowmotion\_order\_law}
对任意 \(x\in X_m\)，设
\[
\Gamma_x\cong \bigsqcup_{i=1}^{r(x)}P_{\ell_i(x)},
\qquad
\mathcal L_x\cong \prod_{i=1}^{r(x)}J(Z_{\ell_i(x)}).
\]
把每个因子 \(J(Z_{\ell_i(x)})\) 通过规范同构识别为 \(\mathsf{Ind}(P_{\ell_i(x)})\)，并定义纤维上的 canonical rowmotion
\[
\operatorname{Row}_x:=\prod_{i=1}^{r(x)}\operatorname{Row}_{P_{\ell_i(x)}}.
\]
则其精确周期满足
\[
\operatorname{ord}(\operatorname{Row}_x)=\mathrm{lcm}_{1\le i\le r(x)}\bigl(\ell_i(x)+2\bigr).
\]
若记
\[
m_d(x):=\#\{\,i:\ d\mid (\ell_i(x)+2)\,\}
\qquad (d\ge 3),
\]
则等价地
\[
\operatorname{ord}(\operatorname{Row}_x)
=
\mathrm{lcm}\{\,d\ge 3:\ m_d(x)>0\,\}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-birkhoff-fence-ideal-lattice}，\(\mathcal L_x\) 分解为路径锯齿理想格的 Cartesian 乘积；再经 Joseph--Roby 关于路径独立集 rowmotion 的周期公式 \cite{JosephRoby2018TogglingIndependentSetsPath}，每个路径因子上的 canonical rowmotion 满足
\[
\operatorname{ord}(\operatorname{Row}_{P_{\ell_i(x)}})=\ell_i(x)+2.
\]
直积动力系统的周期即各因子周期的最小公倍数，故得
\[
\operatorname{ord}(\operatorname{Row}_x)=\mathrm{lcm}_{1\le i\le r(x)}(\ell_i(x)+2).
\]
而一个整数 \(d\ge 3\) 出现在该最小公倍数的约束中，当且仅当存在某个 \(i\) 使 \(d\mid(\ell_i(x)+2)\)，亦即 \(m_d(x)>0\)。故第二式只是第一式的整除指纹改写。证毕。
\end{proof}

\begin{theorem}[cyclotomic 原始除子层的全局局部化律]\label{thm:conclusion-fiber-cyclotomic-primitive-divisor-global-localization}

\[
\sum_{x:\,m_3(x)=0}(-1)^{|x|_1}Z_x(-1)=0.
\]
再由推论 \ref{cor:conclusion-fiber-parity-balance-rowmotion-three-torsion}，该式可重写为
\[
\sum_{x:\,3\nmid \operatorname{ord}(\operatorname{Row}_x)}(-1)^{|x|_1}Z_x(-1)=0.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-biased-weighted-partition-identity}，
\[
\sum_{x\in X_m} t^{|x|_1}Z_x(t)=(1+t)^m.
\]
在 \(t=\tau\) 处，若 \(m_D(x)>0\)，则因 \(\Psi_D(\tau)=0\) 有
\[
Z_x(\tau)=0.
\]
故全和自动局部化到 \(m_D(x)=0\) 的层上，从而得到
\[
\sum_{x:\,m_D(x)=0}\tau^{|x|_1}Z_x(\tau)=(1+\tau)^m.
\]
当 \(D=3\) 时，\(\Psi_3(t)=1+t\)，故可取 \(\tau=-1\)，右端即为 \(0\)。这便给出
\[
\sum_{x:\,m_3(x)=0}(-1)^{|x|_1}Z_x(-1)=0.
\]
最后，由推论 \ref{cor:conclusion-fiber-parity-balance-rowmotion-three-torsion}，
\[
m_3(x)>0
\iff
3\mid \operatorname{ord}(\operatorname{Row}_x),
\]
故最终的 rowmotion 版本随之成立。证毕。
\end{proof}
