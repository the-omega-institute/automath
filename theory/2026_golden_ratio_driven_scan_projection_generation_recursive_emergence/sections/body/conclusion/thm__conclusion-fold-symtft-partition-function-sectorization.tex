\begin{theorem}\label{thm:conclusion-fold-symtft-partition-function-collision-moments}
在 \(B_m=\bigoplus_{x\in X_m}\CC\,\widetilde P_x\) 上定义线性泛函
\[
\varepsilon_m(\widetilde P_x):=d_m(x)^2.
\]
则 \((B_m,\mu,\eta,\varepsilon_m)\) 诱导一个交换半单 Frobenius 代数结构，从而给出二维定向 TQFT（亦可视为 SymTFT 的一个可计算模型 \cite{Bottini2025CategoricalSymmetriesQFT}）。
其在闭连通 genus-\(g\) 曲面 \(\Sigma_g\) 上的分区函数满足
\[
Z_m(\Sigma_g)=\sum_{x\in X_m} d_m(x)^{2-2g}\qquad(g\ge 0),
\]
并且按极小幂等元分解具有直和加法性。
\leanverified{tqft\_sphere\_eq\_momentSum\_two}
\end{theorem}
\begin{proof}
由于 \(\varepsilon_m(\widetilde P_x)\neq 0\)，双线性型 \(\langle a,b\rangle:=\varepsilon_m(ab)\) 在基 \(\{\widetilde P_x\}\) 下为非退化对角型，从而给出 Frobenius 结构。
在幂等基下余乘法满足
\[
\Delta(\widetilde P_x)=\varepsilon_m(\widetilde P_x)^{-1}\,\widetilde P_x\otimes \widetilde P_x,
\]
从而把手元为
\[
h=\mu\circ\Delta(1)=\sum_{x\in X_m}\varepsilon_m(\widetilde P_x)^{-1}\,\widetilde P_x=\sum_{x\in X_m}d_m(x)^{-2}\,\widetilde P_x.
\]
标准二维 TQFT 公式给出
\[
Z_m(\Sigma_g)=\varepsilon_m(h^{g-1})
=\sum_{x\in X_m}\varepsilon_m(\widetilde P_x)\,\bigl(d_m(x)^{-2}\bigr)^{g-1}
=\sum_{x\in X_m}d_m(x)^{2-2g}.
\]
最后的“直和加法性”由 \(B_m\cong \CC^{X_m}\) 的极小幂等元分解与上式按 \(\widetilde P_x\) 求和的形式直接推出。证毕。
\end{proof}

\begin{proposition}[Watatani 指标元--把手元同一与配分函数的迹矩化公式]\label{prop:conclusion-watatani-handle-identity-trace-moment}
在定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 的设定下，令 \(E_m\) 为折叠条件期望，并记其 Watatani 指标元
\[
\mathrm{Ind}_{\mathrm W}(E_m)=\sum_{x\in X_m}d_m(x)\,\widetilde P_x
\]
（定理 \ref{thm:op-algebra-fold-watatani-index-equals-multiplicity-field}）。
则该 TQFT 的把手元 \(h\) 满足严格恒等式
\[
h=\mathrm{Ind}_{\mathrm W}(E_m)^{-2}.
\]
进一步，定义归一化迹 \(\tau_{m,1}:B_m\to\CC\) 为
\[
\tau_{m,1}(\widetilde P_x):=\frac{d_m(x)}{2^m}\qquad(x\in X_m),
\]
则对任意整数 \(g\ge 0\) 有迹矩公式
\[
Z_m(\Sigma_g)=2^m\,\tau_{m,1}\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^{1-2g}\right).
\]
\leanverified{total\_free\_generators\_eq\_hidden\_dim}
\leanverified{jones\_index\_lower\_six}
\end{proposition}
\begin{proof}
在幂等基下 \(\mathrm{Ind}_{\mathrm W}(E_m)\) 在块 \(x\) 上取标量 \(d_m(x)\)，故其逆平方在块 \(x\) 上取标量 \(d_m(x)^{-2}\)，从而
\(
\mathrm{Ind}_{\mathrm W}(E_m)^{-2}=\sum_x d_m(x)^{-2}\widetilde P_x
\)；
与定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 证明中的把手元表达式一致，得第一式。
\par
同理，
\(
\mathrm{Ind}_{\mathrm W}(E_m)^{1-2g}=\sum_x d_m(x)^{1-2g}\widetilde P_x
\)，
因此
\[
2^m\,\tau_{m,1}\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^{1-2g}\right)
=2^m\sum_x d_m(x)^{1-2g}\tau_{m,1}(\widetilde P_x)
=\sum_x d_m(x)^{2-2g}
=Z_m(\Sigma_g),
\]
其中最后一步为定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 的闭式。证毕。
\end{proof}

\begin{corollary}[高属分区函数的双侧压制：负阶幂和由正阶压力控制]\label{cor:conclusion-tqft-genus-bound-by-positive-pressure}
对任意整数 \(g\ge 2\)，记 \(s:=2g-2>0\)，则
\[
Z_m(\Sigma_g)=\sum_{x\in X_m}d_m(x)^{-s}=S_{-s}(m),
\qquad
S_s(m):=\sum_{x\in X_m}d_m(x)^{s}.
\]
从而对每个 \(m\) 有下界
\[
\boxed{\;
Z_m(\Sigma_g)\ \ge\ \frac{|X_m|^2}{S_s(m)}\;}
\]
并且取对数后得到渐近形式
\[
\liminf_{m\to\infty}\frac{1}{m}\log Z_m(\Sigma_g)\ \ge\ 2\log\varphi\ -\ \limsup_{m\to\infty}\frac{1}{m}\log S_s(m),
\]
其中 \(\varphi=(1+\sqrt5)/2\)。
\end{corollary}
\begin{proof}
恒等式 \(Z_m(\Sigma_g)=S_{-s}(m)\) 由定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 取 \(s=2g-2\) 直接改写得到。
对有限 \(m\) 的下界由 Cauchy--Schwarz 应用于 \(a_x=d_m(x)^{s/2}\)、\(b_x=d_m(x)^{-s/2}\) 得
\(
S_s(m)S_{-s}(m)\ge |X_m|^2
\)，移项即得。
最后用 \(|X_m|=F_{m+2}\) 与 \(\lim_{m\to\infty}\frac{1}{m}\log|X_m|=\log\varphi\) 得到对数渐近不等式。证毕。
\end{proof}

\begin{proposition}[属配分函数的离散对数凸性]\label{prop:conclusion-tqft-genus-logconvexity}
记 \(Z_g:=Z_m(\Sigma_g)\)（\(g\in\ZZ_{\ge 0}\)）。
则对一切整数 \(g\ge 1\) 成立不等式
\[
Z_g^2\ \le\ Z_{g-1}\,Z_{g+1}.
\]
并且等号成立当且仅当 \(\{d_m(x):x\in X_m\}\) 为常数。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments}，
\(
Z_g=\sum_x d_m(x)^{2-2g}
\)。
令 \(a_x:=d_m(x)^{2-g}\)、\(b_x:=d_m(x)^{-g}\)，则 \(\sum_x a_xb_x=Z_g\) 且
\[
\sum_x a_x^2=\sum_x d_m(x)^{4-2g}=Z_{g-1},\qquad
\sum_x b_x^2=\sum_x d_m(x)^{-2g}=Z_{g+1}.
\]
对 \((a_x)_x\) 与 \((b_x)_x\) 应用 Cauchy--Schwarz 得 \(Z_g^2\le Z_{g-1}Z_{g+1}\)。
等号成立当且仅当存在常数 \(c\) 使对一切 \(x\) 有 \(a_x=c\,b_x\)，即 \(d_m(x)^2\) 为常数，等价于 \(d_m(x)\) 为常数。证毕。
\end{proof}

\begin{proposition}[属生成函数的有理性与极点谱]\label{prop:conclusion-tqft-genus-generating-function-rational}

\mathcal{G}_m(z):=\sum_{g=0}^{\infty}Z_m(\Sigma_g)\,z^g\qquad(|z|<1).

\mathcal{G}_m(z)
=\sum_{x\in X_m}\frac{d_m(x)^2}{1-z/d_m(x)^2}
=\sum_{x\in X_m}\frac{d_m(x)^4}{d_m(x)^2-z}.
\]
从而其极点集合（按重数计）由多重集 \(\{d_m(x)^2:x\in X_m\}\) 完全确定。

\begin{proposition}[属序列的有限线性递推与纤维谱的可恢复性]\label{prop:conclusion-tqft-genus-linear-recurrence-recovery}
令 \(D_1,\dots,D_r\) 为集合 \(\{d_m(x):x\in X_m\}\) 的不同取值，并记
\[
n_i:=\#\{x\in X_m:\ d_m(x)=D_i\}\qquad(1\le i\le r).
\]
则对一切 \(g\ge 0\) 有指数和表示
\[
Z_m(\Sigma_g)=\sum_{i=1}^r n_iD_i^{2-2g}
=\sum_{i=1}^r a_i\,\lambda_i^{\,g},
\qquad
a_i:=n_iD_i^2,\ \lambda_i:=D_i^{-2}.
\]
因此序列 \(g\mapsto Z_m(\Sigma_g)\) 满足阶数为 \(r\) 的常系数线性递推，其特征多项式可取为
\[
P(t)=\prod_{i=1}^r(1-\lambda_i t)=\prod_{i=1}^r\left(1-\frac{t}{D_i^{2}}\right).
\]
并且给定有限前缀 \(\{Z_m(\Sigma_g)\}_{g=0}^{2r-1}\) 可在代数意义下唯一恢复 \(\{(D_i,n_i)\}_{i=1}^r\)，从而恢复整条属序列。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 的闭式
\(
Z_m(\Sigma_g)=\sum_x d_m(x)^{2-2g}
\)，
按 \(d_m\) 的取值分组得到第一恒等式；令 \(a_i=n_iD_i^2\)、\(\lambda_i=D_i^{-2}\) 得到有限指数和表示。
有限指数和序列满足以 \(\{\lambda_i\}\) 为根的线性递推，递推特征多项式可取为 \(\prod_i(1-\lambda_i t)\)。
最后的可恢复性为 Prony 型反演的标准结论：由 \(2r\) 个连续值可唯一确定最小递推（等价于 Hankel 秩为 \(r\)），进而确定 \(\{\lambda_i\}\) 与 \(\{a_i\}\)，再由 \(\lambda_i=D_i^{-2}\) 与 \(a_i=n_iD_i^2\) 解出 \((D_i,n_i)\)。证毕。
\leanverified{genus\_recurrence\_order\_six}
\leanverified{distinct\_fiber\_sq\_six}
\end{proof}

\begin{theorem}[Witten 指数筛选的子 TQFT]\label{thm:conclusion-tqft-witten-index-susy-subtqft}
对每个 \(x\in X_m\)，令 \(Z_x(t)\) 为纤维生成函数（定义 \ref{def:pom-fiber-adm-path}），并定义指示函数
\[
s(x):=\mathbf{1}_{\{Z_x(-1)\neq 0\}}.
\]
则：
\begin{enumerate}
  \item \(s(x)=0\) 当且仅当候选位点诱导图分解 \(\Gamma_x\simeq\bigsqcup_i P_{\ell_i}\) 中存在分量满足 \(\ell_i\equiv 1\pmod 3\)；
  \item 元
  \[
  e_{\mathrm{susy}}:=\sum_{x\in X_m:\ s(x)=1}\widetilde P_x\ \in\ B_m
  \]
  为中心幂等元，且 \(B_m^{\mathrm{susy}}:=e_{\mathrm{susy}}B_m\) 为交换半单 Frobenius 子代数；
  \item 存在唯一二维定向 TQFT \(Z_m^{\mathrm{susy}}\) 使 \(Z_m^{\mathrm{susy}}(S^1)=B_m^{\mathrm{susy}}\)，并且对任意闭连通曲面 \(\Sigma_g\) 有显式闭式
  \[
  Z_m^{\mathrm{susy}}(\Sigma_g)=\sum_{x\in X_m}s(x)\,d_m(x)^{2-2g}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)} 由定理 \ref{thm:pom-fiber-signed-index-periodicity}，对任意 \(x\) 有 \(Z_x(-1)=\prod_i J_{\ell_i}\)，并且 \(J_{\ell}=0\) 当且仅当 \(\ell\equiv 1\pmod 3\)，故结论成立。
\par
\textbf{(2)} 在 \(B_m=\bigoplus_{x\in X_m}\CC\,\widetilde P_x\) 的幂等直和基下，\(\widetilde P_x\widetilde P_y=\delta_{xy}\widetilde P_x\)，因此任意子集求和给出中心幂等元；故 \(e_{\mathrm{susy}}\) 为中心幂等元，并且 \(B_m^{\mathrm{susy}}=e_{\mathrm{susy}}B_m\) 为交换半单 Frobenius 子代数。
\par
\textbf{(3)} 对交换 Frobenius 代数与二维定向 TQFT 的等价分类给出唯一性。配分函数闭式由定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 的推导逐项限制到幂等元子基上得到：将 \(X_m\) 的求和域限制到 \(s(x)=1\) 的指标子集，即得所示公式。证毕。
\end{proof}
