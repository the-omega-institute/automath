\begin{theorem}\label{thm:conclusion-fold-symtft-partition-function-collision-moments}
在 \(B_m=\bigoplus_{x\in X_m}\CC\,\widetilde P_x\) 上定义线性泛函

\[
Z_m(\Sigma_g)=\sum_{x\in X_m} d_m(x)^{2-2g}\qquad(g\ge 0),

\leanverified{tqft\_sphere\_eq\_momentSum\_two}
\leanverified{paper\_master\_audit\_certificate}
\end{theorem}
\begin{proof}

在幂等基下余乘法满足
\[
\Delta(\widetilde P_x)=\varepsilon_m(\widetilde P_x)^{-1}\,\widetilde P_x\otimes \widetilde P_x,
\]
从而把手元为
\[
h=\mu\circ\Delta(1)=\sum_{x\in X_m}\varepsilon_m(\widetilde P_x)^{-1}\,\widetilde P_x=\sum_{x\in X_m}d_m(x)^{-2}\,\widetilde P_x.
\]
标准二维 TQFT 公式给出
\[
Z_m(\Sigma_g)=\varepsilon_m(h^{g-1})
=\sum_{x\in X_m}\varepsilon_m(\widetilde P_x)\,\bigl(d_m(x)^{-2}\bigr)^{g-1}
=\sum_{x\in X_m}d_m(x)^{2-2g}.
\]
最后的“直和加法性”由 \(B_m\cong \CC^{X_m}\) 的极小幂等元分解与上式按 \(\widetilde P_x\) 求和的形式直接推出。证毕。
\end{proof}

\begin{proposition}[属配分函数的离散对数凸性]\label{prop:conclusion-tqft-genus-logconvexity}

\begin{theorem}[属配分函数塔的 Hausdorff 矩序列结构]\label{thm:conclusion-tqft-genus-hausdorff-moment-sequence}

\qquad\text{支撑于 }(0,1].
\]

\[
\frac{Z_m(\Sigma_{g+1})}{|X_m|}
=
\int_{[0,1]} t^g\,d\nu_m(t).
\]
因此，序列
\[

\]
是一个 Hausdorff 矩序列。特别地，对任意整数 \(g,k\ge 0\) 都有完全单调性
\[
(-1)^k\Delta^k Z_m(\Sigma_{g+1})\ge 0,
\]

=
|X_m|\cdot \frac{1}{|X_m|}\sum_{x\in X_m}\bigl(d_m(x)^{-2}\bigr)^g,
\]
即得矩表示。

\[
(-1)^k\Delta^k Z_m(\Sigma_{g+1})

|X_m|\int_{[0,1]} t^g(1-t)^k\,d\nu_m(t)\ge 0,

\begin{theorem}[属配分函数塔的平凡化判据]\label{thm:conclusion-tqft-genus-tower-trivialization-criterion}

  Z_m(\Sigma_{g_0})=Z_m(\Sigma_{g_0+1});

\textup{(1)}\(\Leftrightarrow\)\textup{(2)} 显然，因为 \(\Fold_m\) 已知满射。

\begin{theorem}[高属极限的底端谱逐层剥离]\label{thm:conclusion-tqft-high-genus-bottom-spectrum-peeling}
\leanverified{paper\_conclusion\_tqft\_high\_genus\_bottom\_spectrum\_peeling}
设纤维多重度的不同取值按升序写为
\[
1\le \delta_1<\delta_2<\cdots<\delta_s,
\]
其重数分别为
\[
\mu_j:=\#\{x\in X_m:\ d_m(x)=\delta_j\}.
\]
则对每个 \(g\ge 0\) 都有
\[
Z_m(\Sigma_g)=\sum_{j=1}^{s}\mu_j\,\delta_j^{\,2-2g}.
\]
并且对任意 \(1\le r\le s\) 成立精确剥离公式
\[
\lim_{g\to\infty}
\delta_r^{\,2g-2}
\left(
Z_m(\Sigma_g)-\sum_{j=1}^{r-1}\mu_j\delta_j^{\,2-2g}
\right)
=
\mu_r.
\]
特别地，
\[
\lim_{g\to\infty}Z_m(\Sigma_g)=\mu_1,
\]
以及当 \(s\ge 2\) 时，
\[
\lim_{g\to\infty}\delta_2^{\,2g-2}\bigl(Z_m(\Sigma_g)-\mu_1\bigr)=\mu_2.
\]
\end{theorem}
\begin{proof}
首式只是按多重度值分组。对任意固定的 \(r\)，有
\[
Z_m(\Sigma_g)-\sum_{j=1}^{r-1}\mu_j\delta_j^{\,2-2g}
=
\sum_{j=r}^{s}\mu_j\delta_j^{\,2-2g}
=
\delta_r^{\,2-2g}
\left(
\mu_r+\sum_{j=r+1}^{s}\mu_j\left(\frac{\delta_j}{\delta_r}\right)^{2-2g}
\right).
\]
当 \(j>r\) 时 \(\delta_j/\delta_r>1\)，而 \(2-2g\to-\infty\)，故括号中各 \(j>r\) 项趋于 \(0\)。于是极限即为 \(\mu_r\)。证毕。
\end{proof}

\begin{theorem}[Witten 指数筛选的子 TQFT]\label{thm:conclusion-tqft-witten-index-susy-subtqft}
对每个 \(x\in X_m\)，令 \(Z_x(t)\) 为纤维生成函数（定义 \ref{def:pom-fiber-adm-path}），并定义指示函数
\[
s(x):=\mathbf{1}_{\{Z_x(-1)\neq 0\}}.
\]
则：
\begin{enumerate}
  \item \(s(x)=0\) 当且仅当候选位点诱导图分解 \(\Gamma_x\simeq\bigsqcup_i P_{\ell_i}\) 中存在分量满足 \(\ell_i\equiv 1\pmod 3\)；
  \item 元
  \[
  e_{\mathrm{susy}}:=\sum_{x\in X_m:\ s(x)=1}\widetilde P_x\ \in\ B_m
  \]
  为中心幂等元，且 \(B_m^{\mathrm{susy}}:=e_{\mathrm{susy}}B_m\) 为交换半单 Frobenius 子代数；
  \item 存在唯一二维定向 TQFT \(Z_m^{\mathrm{susy}}\) 使 \(Z_m^{\mathrm{susy}}(S^1)=B_m^{\mathrm{susy}}\)，并且对任意闭连通曲面 \(\Sigma_g\) 有显式闭式
  \[
  Z_m^{\mathrm{susy}}(\Sigma_g)=\sum_{x\in X_m}s(x)\,d_m(x)^{2-2g}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)} 由定理 \ref{thm:pom-fiber-signed-index-periodicity}，对任意 \(x\) 有 \(Z_x(-1)=\prod_i J_{\ell_i}\)，并且 \(J_{\ell}=0\) 当且仅当 \(\ell\equiv 1\pmod 3\)，故结论成立。
\par
\textbf{(2)} 在 \(B_m=\bigoplus_{x\in X_m}\CC\,\widetilde P_x\) 的幂等直和基下，\(\widetilde P_x\widetilde P_y=\delta_{xy}\widetilde P_x\)，因此任意子集求和给出中心幂等元；故 \(e_{\mathrm{susy}}\) 为中心幂等元，并且 \(B_m^{\mathrm{susy}}=e_{\mathrm{susy}}B_m\) 为交换半单 Frobenius 子代数。
\par
\textbf{(3)} 对交换 Frobenius 代数与二维定向 TQFT 的等价分类给出唯一性。配分函数闭式由定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 的推导逐项限制到幂等元子基上得到：将 \(X_m\) 的求和域限制到 \(s(x)=1\) 的指标子集，即得所示公式。证毕。
\end{proof}

\begin{theorem}[非可缩 genus 子理论精确等于奇数多重度子谱]\label{thm:conclusion-tqft-noncontractible-genus-equals-odd-spectrum}
定义非可缩扇区的 genus 配分函数
\[
Z_m^{\mathrm{nc}}(\Sigma_g):=
\sum_{\substack{x\in X_m\\ |K_x|\not\simeq *}}d_m(x)^{2-2g}.
\]
则有严格恒等式
\[
Z_m^{\mathrm{nc}}(\Sigma_g)
=
\sum_{\substack{x\in X_m\\ d_m(x)\ \mathrm{odd}}}d_m(x)^{2-2g}.
\]
因此，定理 \ref{thm:conclusion-tqft-genus-hausdorff-moment-sequence}、定理 \ref{thm:conclusion-tqft-genus-tower-trivialization-criterion}、推论 \ref{cor:conclusion-tqft-genus-logconvexity-rigidity-equivalences}、定理 \ref{thm:conclusion-tqft-genus-tower-complete-multiplicity-invariant} 与定理 \ref{thm:conclusion-tqft-high-genus-bottom-spectrum-peeling} 在把求和域限制到奇数多重度层后逐字平行成立。特别地，非可缩 genus 子理论唯一决定奇数多重度子谱，并在高 genus 极限下从小到大逐层剥离全部奇数纤维层级。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-noncontractible-restricted-dpi-exact} 的证明末段，\(|K_x|\not\simeq *\) 当且仅当 \(d_m(x)\) 为奇数。将此代入定义即得第一式。
其余结论只需把前述各定理中的求和域从全体 \(x\in X_m\) 替换为奇数多重度子集；所有证明步骤保持不变。证毕。
\end{proof}

\begin{theorem}[正矩冻结与高属剥离是纤维谱两端的对偶显微镜]\label{thm:conclusion-positive-moment-freezing-high-genus-peeling-duality}
定义正矩
\[
S_a(m):=\sum_{x\in X_m}d_m(x)^a
\qquad(a\ge 1),
\]
并仍记
\[
D_m:=\max_{x\in X_m}d_m(x)=\delta_s.
\]
则有双端极限
\[
\lim_{a\to\infty}S_a(m)^{1/a}=D_m,
\qquad
\lim_{g\to\infty}Z_m(\Sigma_g)^{-1/(2g-2)}=\delta_1.
\]
更一般地，若 \(s\ge 2\)，则
\[
\lim_{g\to\infty}
\left(
Z_m(\Sigma_g)-\mu_1\delta_1^{\,2-2g}
\right)^{-1/(2g-2)}
=
\delta_2.
\]
换言之，大正矩冻结读出纤维谱的顶端，而高 genus 极限读出纤维谱的底端。
\end{theorem}
\begin{proof}
第一式是有限正数集的 \(\ell^a\)-范数极限：
\[
S_a(m)^{1/a}
=
\left(\sum_{j=1}^{s}\mu_j\delta_j^a\right)^{1/a}\to \delta_s=D_m.
\]
第二式由定理 \ref{thm:conclusion-tqft-high-genus-bottom-spectrum-peeling} 在 \(r=1\) 时给出
\[
Z_m(\Sigma_g)=\mu_1\delta_1^{\,2-2g}(1+o(1)),
\]
从而
\[
Z_m(\Sigma_g)^{-1/(2g-2)}
=
\delta_1\cdot \mu_1^{-1/(2g-2)}(1+o(1))^{-1/(2g-2)}
\to \delta_1.
\]
第三式同理由定理 \ref{thm:conclusion-tqft-high-genus-bottom-spectrum-peeling} 的 \(r=2\) 情形得到。证毕。
\end{proof}

\begin{conjecture}[genuine fold 的 genus 塔严格凸性]\label{conj:conclusion-genuine-fold-genus-strict-convexity}
对每个 genuinely nontrivial 的折叠层 \(m\ge 2\) 与每个 \(g\ge 1\)，应有严格不等式
\[
Z_m(\Sigma_g)^2<Z_m(\Sigma_{g-1})Z_m(\Sigma_{g+1}).
\]
等价地，真实折叠的纤维多重度谱不应在任一分辨率上呈现统一常值：
\[
\not\exists\, d\in\NN\ \text{使}\ d_m(x)=d\quad(\forall x\in X_m).
\]
\end{conjecture}
