\paragraph{导子 \texorpdfstring{$3$}{3} 的受限自由能与导子 \texorpdfstring{$2$}{2} 的顶层选择深度}
在定理 \ref{thm:pom-multiplicity-composition-mod3-simple-pole-u}--\ref{thm:pom-multiplicity-composition-mod3-badcount-clt} 所给出的两参数更新配分函数与坏分量统计基础上，还可把 \(\ell\equiv 1\pmod 3\) 的阻碍进一步重写成导子 \(3\) 的 Ramanujan--Dirichlet 投影；另一方面，定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 与定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase} 又表明，非可缩极大纤维并不由新的指数尺度决定，而只是由导子 \(2\) 在顶层 order statistics 上首次取负的位置选出。

\begin{theorem}[坏分量更新核的 Ramanujan--Dirichlet 精确分裂]\label{thm:conclusion-bad-component-ramanujan-dirichlet-splitting}
\leanverified{paper\_conclusion\_bad\_component\_ramanujan\_dirichlet\_splitting}
固定 \(q\ge 0\)。定义
\[
\Phi_q(t):=\sum_{k\ge 1}F_{k+2}^{\,q}t^k,
\qquad
R_q(t):=\sum_{k\ge 1}c_3(k)F_{k+2}^{\,q}t^k,
\qquad
X_q(t):=\sum_{k\ge 1}\chi_3(k)F_{k+2}^{\,q}t^k,
\]
其中
\[
c_3(k)=
\begin{cases}
2,& 3\mid k,\\
-1,& 3\nmid k,
\end{cases}
\qquad
\chi_3(k)=
\begin{cases}
0,& 3\mid k,\\
1,& k\equiv 1\pmod 3,\\
-1,& k\equiv 2\pmod 3.
\end{cases}
\]
再记
\[
C_q(t):=\sum_{k\equiv 1\ (\mathrm{mod}\ 3)}F_{k+2}^{\,q}t^k,
\qquad
B_q(t):=\sum_{k\not\equiv 1\ (\mathrm{mod}\ 3)}F_{k+2}^{\,q}t^k.
\]
则成立严格恒等式
\[
C_q(t)=\frac13\,\Phi_q(t)-\frac16\,R_q(t)+\frac12\,X_q(t),
\]
\[
B_q(t)=\frac23\,\Phi_q(t)+\frac16\,R_q(t)-\frac12\,X_q(t),
\]
并且两参数更新核
\[
A_q(t;u):=B_q(t)+u\,C_q(t)
\]
可重写为
\[
A_q(t;u)
=
\frac{2+u}{3}\,\Phi_q(t)
+\frac{1-u}{6}\,R_q(t)
+\frac{u-1}{2}\,X_q(t).
\]
\end{theorem}
\begin{proof}
在模 \(3\) 的三个剩余类上直接计算得
\[
\mathbf 1_{\{k\equiv 1\ (\mathrm{mod}\ 3)\}}
=
\frac13-\frac16\,c_3(k)+\frac12\,\chi_3(k).
\]
将上式逐项乘以 \(F_{k+2}^{\,q}t^k\) 并对 \(k\ge 1\) 求和，即得
\[
C_q(t)=\frac13\,\Phi_q(t)-\frac16\,R_q(t)+\frac12\,X_q(t).
\]
再由 \(B_q(t)=\Phi_q(t)-C_q(t)\) 得第二式。最后把 \(A_q(t;u)=B_q(t)+uC_q(t)\) 代入并整理系数，即得所示分裂。证毕。
\end{proof}

\begin{theorem}[固定坏分量层的全阶生成函数与统一尖锐主项]\label{thm:conclusion-fixed-bad-component-layer-sharp-main-term}
\leanverified{paper\_conclusion\_fixed\_bad\_component\_layer\_sharp\_main\_term}
对 \(L\ge 0\) 与 \(r\ge 0\)，定义
\[
Z_{L,r}^{(q)}
:=
\sum_{\substack{\boldsymbol{\ell}\in\mathcal C_L\\ A_L(\boldsymbol{\ell})=r}}
\prod_i F_{\ell_i+2}^{\,q},
\qquad
\mathcal Z_{q,r}(t):=\sum_{L\ge 0}Z_{L,r}^{(q)}t^L.
\]
再记
\[
\rho_q^{(0)}:=\rho_q(0),
\qquad
\lambda_q^{(0)}:=(\rho_q^{(0)})^{-1},
\qquad
\mu_q^{(0)}:=\rho_q^{(0)}\,B_q'(\rho_q^{(0)}).
\]
则对每个固定 \(r\ge 0\) 有闭式
\[
\mathcal Z_{q,r}(t)=\frac{C_q(t)^r}{(1-B_q(t))^{r+1}},
\]
并且当 \(L\to\infty\) 时成立尖锐渐近
\[
Z_{L,r}^{(q)}
\sim
\frac{C_q(\rho_q^{(0)})^{\,r}}{r!\,(\mu_q^{(0)})^{r+1}}
\,L^r\,(\lambda_q^{(0)})^L.
\]
因此，全部固定坏分量层共享同一指数尺度 \(\lambda_q^{(0)}\)，层间差异仅表现为多项式壳 \(L^r\) 与显式常数。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity}，
\[
\mathcal Z^{(q)}(t;u)=\frac{1}{1-B_q(t)-uC_q(t)}.
\]
在 \(|uC_q(t)|<|1-B_q(t)|\) 的区域内作几何展开，得到
\[
\mathcal Z^{(q)}(t;u)
=
\frac{1}{1-B_q(t)}
\sum_{r\ge 0}\left(\frac{uC_q(t)}{1-B_q(t)}\right)^r
=
\sum_{r\ge 0}u^r\frac{C_q(t)^r}{(1-B_q(t))^{r+1}}.
\]
比较 \(u^r\) 系数即得
\[
\mathcal Z_{q,r}(t)=\frac{C_q(t)^r}{(1-B_q(t))^{r+1}}.
\]

再由定理 \ref{thm:pom-multiplicity-composition-mod3-simple-pole-u} 在 \(u=0\) 的情形，\(t=\rho_q^{(0)}\) 是 \(1-B_q(t)\) 的唯一支配简单零点，故
\[
1-B_q(t)
=
\mu_q^{(0)}\left(1-\frac{t}{\rho_q^{(0)}}\right)
+ O\!\left(\left(1-\frac{t}{\rho_q^{(0)}}\right)^2\right).
\]
于是
\[
\mathcal Z_{q,r}(t)
=
\frac{C_q(\rho_q^{(0)})^{\,r}}{(\mu_q^{(0)})^{r+1}}
\left(1-\frac{t}{\rho_q^{(0)}}\right)^{-(r+1)}
+ O\!\left(\left(1-\frac{t}{\rho_q^{(0)}}\right)^{-r}\right).
\]
对上式应用标准奇点传递
\[
[t^L]\left(1-\frac{t}{\rho}\right)^{-(r+1)}
=
\binom{L+r}{r}\rho^{-L}
\sim
\frac{L^r}{r!}\rho^{-L},
\]
即得所示渐近。证毕。
\end{proof}

\begin{corollary}[受限拓扑扇区的统一自由能税]\label{cor:conclusion-bad-component-free-energy-tax}
\leanverified{paper\_conclusion\_bad\_component\_free\_energy\_tax}
记
\[
\lambda_q:=\lambda(q),
\qquad
\mu_q:=\mu_q(1),
\qquad
\Gamma_q:=\log\frac{\lambda_q}{\lambda_q^{(0)}}.
\]
则 \(\Gamma_q>0\)，并且对每个固定 \(r\ge 0\) 有
\[
\frac{Z_{L,r}^{(q)}}{Z_L^{(q)}}
\sim
\frac{\mu_q\,C_q(\rho_q^{(0)})^{\,r}}{r!\,(\mu_q^{(0)})^{r+1}}
\,L^r\,e^{-\Gamma_q L}.
\]
特别地，
\[
\frac{Z_{L,0}^{(q)}}{Z_L^{(q)}}
\sim
\frac{\mu_q}{\mu_q^{(0)}}\,e^{-\Gamma_q L},
\qquad
\frac{Z_{L,1}^{(q)}}{Z_{L,0}^{(q)}}
\sim
\frac{C_q(\rho_q^{(0)})}{\mu_q^{(0)}}\,L.
\]
由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，零坏分量层正对应于奇纤维亦即非可缩扇区；因此 \(\Gamma_q\) 正是该受限拓扑扇区相对于全系统的统一自由能税。
\end{corollary}
\begin{proof}
由命题 \ref{prop:pom-multiplicity-composition-sharp-main-term-constant}，
\[
Z_L^{(q)}\sim \frac{1}{\mu_q}\lambda_q^L.
\]
再将定理 \ref{thm:conclusion-fixed-bad-component-layer-sharp-main-term} 的渐近式除以上式，即得第一条。
由定理 \ref{thm:pom-multiplicity-composition-mod3-sparsity} 的证明，当 \(u=0<1\) 时有 \(\lambda_q^{(0)}<\lambda_q\)，故 \(\Gamma_q>0\)。
其余两式分别取 \(r=0\) 与 \((r+1)\)-层相邻比值得到。最后一句仅是把定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 的奇偶—同伦等价与 \(A_L=0\) 的定义合并。证毕。
\end{proof}

\begin{theorem}[非可缩顶层的导子 \texorpdfstring{$2$}{2} 选择深度]\label{thm:conclusion-conductor2-order-statistic-selection-depth}
\leanverified{paper\_conclusion\_conductor2\_order\_statistic\_selection\_depth}
对 \(m\ge 17\)，记
\[
D_m^{(1)}:=D_m>D_m^{(2)}>D_m^{(3)}>\cdots
\]
为顶层纤维谱，并定义
\[
j_\ast(m):=\min\{j\ge 1:\ c_2(D_m^{(j)})=-1\},
\qquad
c_2(n):=(-1)^n.
\]
则
\[
\widetilde D_m=D_m^{(j_\ast(m))}
\]
并且选择深度具有显式模 \(6\) 相图
\[
j_\ast(m)=
\begin{cases}
1,& m\equiv 0,4\pmod 6,\\
2,& m\equiv 1,5\pmod 6,\\
3,& m\equiv 2,3\pmod 6.
\end{cases}
\]
换言之，最大非可缩纤维恰是顶层 order statistics 中第一个满足导子 \(2\) 符号为负的层。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，
\[
|K_x|\not\simeq *
\Longleftrightarrow
d_m(x)\ \text{为奇数}.
\]
而按定义
\[
c_2(n)=(-1)^n=
\begin{cases}
-1,& n\ \text{为奇数},\\
1,& n\ \text{为偶数}.
\end{cases}
\]
因此 \(\widetilde D_m\) 恰是顶层纤维谱中第一个满足 \(c_2=-1\) 的元素，即
\[
\widetilde D_m=D_m^{(j_\ast(m))}.
\]
再由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，
\[
\widetilde D_m=
\begin{cases}
D_m,& m\equiv 0,4\pmod 6,\\
D_m^{(2)},& m\equiv 1,5\pmod 6,\\
D_m^{(3)},& m\equiv 2,3\pmod 6,
\end{cases}
\]
遂得所示 \(j_\ast(m)\) 的分段公式。证毕。
\end{proof}

\begin{corollary}[导子 \texorpdfstring{$2$}{2} 只改顶层前因子而不改主指数]\label{cor:conclusion-conductor2-prefactor-not-exponent}
\leanverified{paper\_conclusion\_conductor2\_prefactor\_not\_exponent}
非可缩顶层的导子 \(2\) 选择只发生在有界深度 \(j_\ast(m)\in\{1,2,3\}\) 内，因此不会改变主压力指数。更精确地，
\[
\lim_{m\to\infty}\frac{1}{m}\log \widetilde D_m
=
\frac12\log\varphi,
\]
而 \(\widetilde D_m/D_m\) 的全部剩余类依赖常数正是定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios} 中给出的四个极限值。
\end{corollary}
\begin{proof}
第一条正是推论 \ref{cor:conclusion-max-noncontractible-fiber-exponential-rate}。
由定理 \ref{thm:conclusion-conductor2-order-statistic-selection-depth}，\(\widetilde D_m\) 只是从 \(D_m^{(1)},D_m^{(2)},D_m^{(3)}\) 中选取第一个奇数层，因此该约束仅作用于有限层的 order statistics。相应的精确剩余类常数已由定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios} 完整给出。证毕。
\end{proof}

\begin{conjecture}[导子 \texorpdfstring{$3$}{3} 的指数税与导子 \texorpdfstring{$2$}{2} 的前因子税分离]\label{conj:conclusion-conductor3-exponent-conductor2-prefactor-separation}
对所有由乘积权
\[
d(x)=\prod_i F_{\ell_i+2}
\]
驱动的窗口族及其 \(q\)-倾斜 escort 配分函数，只要拓扑约束完全通过排除 \(\ell_i\equiv 1\pmod 3\) 的坏分量实现，则应当存在如下算术分工：
\begin{enumerate}
  \item 全部指数级压制都由导子 \(3\) 的受限更新核
  \[
  A_q(t;0)
  =
  \frac23\,\Phi_q(t)+\frac16\,R_q(t)-\frac12\,X_q(t)
  \]
  决定，亦即只取决于 Ramanujan 通道 \(R_q\) 与本原模 \(3\) 角色通道 \(X_q\) 所确定的受限自由能 \(\lambda_q^{(0)}\)；
  \item 全部顶层筛选则只由导子 \(2\) 的奇偶符号 \(c_2\) 在有限阶 order statistics 上的位置决定，因此只能改写前因子，而不能改写主指数。
\end{enumerate}
若该分离成立，则坏分量约束所产生的自由能税与非可缩极大纤维的模 \(6\) 周期税并非两类彼此平行的现象，而是同一算术分工在指数层与前因子层上的两种投影。
\end{conjecture}
