\begin{theorem}[\texorpdfstring{$S_4$}{S4}-Hurwitz 商塔 Jacobian 的四生成元完备性]\label{thm:conclusion-s4-hurwitz-jacobian-four-generator-semiring}
\leanverified{paper\_conclusion\_s4\_hurwitz\_jacobian\_four\_generator\_semiring}
沿用定理 \ref{thm:conclusion-cf-jacobian-four-factor-isogeny} 的记号，设
\[
H:=X/A_4,\qquad
Z:=X/D_8,\qquad
CF:=X/S_3,\qquad
Y:=X/C_4,\qquad
P:=\operatorname{Prym}(Y/Z).
\]
则对 \(S_4\) 的每一个共轭类子群 \(U\le S_4\)，商曲线 Jacobian \(J(X/U)\) 在 \(\QQ\)-同源意义下都属于
\[
\mathcal S
:=
\NN[J(H)]\oplus \NN[J(Z)]\oplus \NN[J(CF)]\oplus \NN[P]
\]
生成的交换半环。更具体地，
\[
\begin{aligned}
J(X/S_4)&\sim 0,\\
J(X/A_4)&\sim J(H),\\
J(X/D_8)&\sim J(Z),\\
J(X/S_3)&\sim J(CF),\\
J(X/C_4)&\sim J(Z)\times P,\\
J(X/V_4)&\sim J(H)\times J(Z)^2,\\
J(X/C_3)&\sim J(H)\times J(CF)\times P,\\
J(X/C_2)&\sim J(Z)\times J(CF)^2\times P,\\
J(X)&\sim J(H)\times J(Z)^2\times J(CF)^3\times P^3.
\end{aligned}
\]
\end{theorem}
\begin{proof}
前四项
\[
J(X/S_4)\sim 0,\qquad
J(X/A_4)\sim J(H),\qquad
J(X/D_8)\sim J(Z),\qquad
J(X/S_3)\sim J(CF)
\]
只是商定义的直接重述。由命题 \ref{prop:conclusion-cf-c4d8-antiinvariant-prym9}，二重覆盖 \(Y\to Z\) 的 Jacobian--Prym 分解给出
\[
J(X/C_4)=J(Y)\sim J(Z)\times P.
\]

对 \(V_4\)-商，主稿中的 Burnside--Kani--Rosen 关系给出
\[
J(X/V_4)\sim J(X/A_4)\times J(X/D_8)^2
=
J(H)\times J(Z)^2.
\]
同一 Hurwitz 商塔的 Jacobian 分解公式进一步给出
\[
J(X/C_3)\sim J(H)\times J(CF)\times P,
\qquad
J(X/C_2)\sim J(Z)\times J(CF)^2\times P.
\]
最后，
\[
J(X)\sim J(H)\times J(Z)^2\times J(CF)^3\times P^3
\]
正是定理 \ref{thm:conclusion-cf-jacobian-four-factor-isogeny}。于是全部共轭类子群商的 Jacobian 都落在四生成元半环 \(\mathcal S\) 中。证毕。
\end{proof}

\begin{theorem}[四生成元基的极小性与等变刚性]\label{thm:conclusion-s4-hurwitz-jacobian-generator-minimality}
结论 \ref{thm:conclusion-s4-hurwitz-jacobian-four-generator-semiring} 中的四元组
\[
\mathcal B:=\{J(H),\,J(Z),\,J(CF),\,P\}
\]
是极小的：不存在其任何真子集能够生成全部 \(J(X/U)\) 的有理同源类型。并且，若记 \(J(X)\) 中对应于
\[
\mathrm{sgn},\qquad V_2,\qquad V_3,\qquad V_3'
\]
四个不可约 \(S_4\)-型的等型阿贝尔子簇分别为
\[
A_{\mathrm{sgn}},\qquad A_{V_2},\qquad A_{V_3},\qquad A_{V_3'},
\]
则有
\[
A_{\mathrm{sgn}}\sim J(H),\qquad
A_{V_2}\sim J(Z)^2,\qquad
A_{V_3}\sim J(CF)^3,\qquad
A_{V_3'}\sim P^3,
\]
且当 \(\rho\neq \rho'\) 时，
\[
\operatorname{Hom}_{S_4}^0(A_\rho,A_{\rho'})=0.
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-cf-jacobian-four-factor-isogeny} 已把 \(J(X)\) 分解为
\[
J(X)\sim J(H)\times J(Z)^2\times J(CF)^3\times P^3,
\]
并把四个因子分别识别为 \(\mathrm{sgn},V_2,V_3,V_3'\) 四个 \(S_4\)-等型块。主稿的等变分解唯一性说明：不同不可约型之间不存在非零 \(S_4\)-等变有理同态，因此四个等型块彼此严格正交，即
\[
\operatorname{Hom}_{S_4}^0(A_\rho,A_{\rho'})=0
\qquad(\rho\neq\rho').
\]

若删去某个生成元，例如删去 \(P\)，则由该真子半环生成的任意对象都不再含有 \(V_3'\)-等型；但
\[
J(X/C_4),\qquad J(X/C_3),\qquad J(X/C_2),\qquad J(X)
\]
都含有正的 \(V_3'\)-重数。其余三个生成元的删除同理：删去 \(J(H)\) 会丢失 \(\mathrm{sgn}\)-等型，删去 \(J(Z)\) 会丢失 \(V_2\)-等型，删去 \(J(CF)\) 会丢失 \(V_3\)-等型。故 \(\mathcal B\) 为极小基。证毕。
\end{proof}

\begin{corollary}[商 Jacobian 的指数向量分类]\label{cor:conclusion-s4-hurwitz-jacobian-exponent-vectors}
对每个共轭类子群 \(U\le S_4\)，记
\[
\mathbf v(U)=(a_U,b_U,c_U,d_U)\in\NN^4
\]
满足
\[
J(X/U)\sim J(H)^{a_U}\times J(Z)^{b_U}\times J(CF)^{c_U}\times P^{d_U}.
\]
则 \(\mathbf v(U)\) 唯一确定，且显式为
\[
\begin{array}{c|c}
U & \mathbf v(U)\\\hline
S_4&(0,0,0,0)\\
A_4&(1,0,0,0)\\
D_8&(0,1,0,0)\\
S_3&(0,0,1,0)\\
C_4&(0,1,0,1)\\
V_4&(1,2,0,0)\\
C_3&(1,0,1,1)\\
C_2&(0,1,2,1)\\
\{1\}&(1,2,3,3)
\end{array}
\]
从而这些商 Jacobian 在 \(\QQ\)-同源意义下两两不同；更强地，任何 \(S_4\)-等变有理自同态在该分类下都严格块对角。
\end{corollary}
\begin{proof}
存在性由定理 \ref{thm:conclusion-s4-hurwitz-jacobian-four-generator-semiring} 给出。唯一性则由定理 \ref{thm:conclusion-s4-hurwitz-jacobian-generator-minimality} 的等型正交性得到：若两组指数向量不同，则某个不可约 \(S_4\)-型的重数不同，故对应商 Jacobian 不可能 \(S_4\)-等变同源。于是上表同时给出全部商 Jacobian 的唯一指数向量。块对角性也只是这一等型正交性的另一种表述。证毕。
\end{proof}

\begin{corollary}[不同等型块之间的 \texorpdfstring{$S_4$}{S4}-等变 Hom 在整系数下仍严格消失]\label{cor:conclusion-s4-hurwitz-jacobian-integral-hom-orthogonality}
沿用定理 \ref{thm:conclusion-s4-hurwitz-jacobian-generator-minimality} 的记号。若
\[
\rho\neq \rho'
\qquad
\text{属于 }
\{\mathrm{sgn},V_2,V_3,V_3'\},
\]
则有
\[
\Hom_{S_4}(A_\rho,A_{\rho'})=0.
\]
因此任意 \(S_4\)-等变端同态
\[
u\in \End_{S_4}(J(X))
\]
在四块分解
\[
J(X)\sim J(H)\times J(Z)^2\times J(CF)^3\times P^3
\]
下都必为严格块对角形式。
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-s4-hurwitz-jacobian-generator-minimality} 已给出
\[
\Hom_{S_4}^0(A_\rho,A_{\rho'})=0
\qquad(\rho\neq \rho').
\]
而对阿贝尔簇，\(\Hom(A_\rho,A_{\rho'})\) 是无挠有限生成阿贝尔群；若其非零，则张量化到 \(\QQ\) 后仍非零，从而与上式矛盾。故整系数 Hom 群也必为零。块对角性只是把这一结论对四块逐一应用到端同态矩阵的所有非对角块上。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$S_4$}{S4}-等变微分端同态代数的精确矩阵分解]\label{thm:conclusion-s4-hurwitz-differential-endomorphism-algebra-dimension}
记
\[
V:=H^0(X,\Omega_X^1).
\]
则由等型重数
\[
V\cong 5\cdot \mathrm{sgn}\oplus 4\cdot V_2\oplus 3\cdot V_3\oplus 9\cdot V_3'
\]
可得
\[
\End_{S_4}(V)
\cong
M_5(\CC)\oplus M_4(\CC)\oplus M_3(\CC)\oplus M_9(\CC).
\]
因而
\[
\dim_{\CC}\End_{S_4}(V)=5^2+4^2+3^2+9^2=131,
\]
并且其中心维数恰为
\[
4.
\]
\end{theorem}
\begin{proof}
标准双交换子理论表明：若一个复表示分解为
\[
\bigoplus_i m_i\rho_i
\]
且 \(\rho_i\) 两两非同构不可约，则其等变端同态代数同构于
\[
\bigoplus_i M_{m_i}(\CC).
\]
将
\[
(m_i)=(5,4,3,9)
\]
代入即可得到所示矩阵分解。复维数与中心维数分别由各矩阵块维数 \(m_i^2\) 与块数直接给出。证毕。
\end{proof}

\begin{theorem}[Prym 多数块与大维 \texorpdfstring{$S_4$}{S4}-等变商的强制 \texorpdfstring{$V_3'$}{V3'} 支撑]\label{thm:conclusion-s4-hurwitz-prym-majority-block-large-quotient-forcing}
设
\[
J_{\mathrm{vis}}:=J(H)\times J(Z)^2\times J(CF)^3.
\]
则
\[
J(X)\sim J_{\mathrm{vis}}\times P^3,
\qquad
\dim J_{\mathrm{vis}}=5+8+9=22,
\qquad
\dim P^3=27.
\]
特别地，
\[
27>22.
\]
因此，对任意 \(S_4\)-等变商阿贝尔簇 \(A\)：
\[
\dim A>22
\quad\Longrightarrow\quad
H^0(A,\Omega_A^1)\ \text{中必含 }V_3'\text{-等型},
\]
等价地，\(A\) 必含有 \(P\) 的非零 \(S_4\)-等变商。
\end{theorem}
\begin{proof}
第一组分解与维数恒等式正是定理 \ref{thm:conclusion-cf-a4free-visible-part-only-twenty-two-over-forty-nine} 的内容。于是去掉 \(P^3\) 之后，剩余三块总维数恰为 \(22\)。

若某个 \(S_4\)-等变商 \(A\) 不含 \(V_3'\)-等型，则其微分表示只能由
\[
\mathrm{sgn},\qquad V_2,\qquad V_3
\]
三类等型组成，从而其像完全落在 \(J_{\mathrm{vis}}\) 所张成的可见部分中，故
\[
\dim A\le 22.
\]
取逆否命题即得：凡 \(\dim A>22\)，其微分表示中必含 \(V_3'\)-等型。再由定理 \ref{thm:conclusion-s4-hurwitz-jacobian-generator-minimality} 中
\[
A_{V_3'}\sim P^3
\]
的识别，便知 \(A\) 必有非零 \(P\)-型等变商。证毕。
\end{proof}

\begin{conjecture}[四块分解的中心极小性]\label{conj:conclusion-s4-hurwitz-jacobian-central-minimality}
若主稿关于两类 Prym 额外自同态极小性的猜想成立，即
\[
\End_{\QQ}(P)=\ZZ,
\qquad
\End^0_{\QQ}\!\bigl(\Prym(C_6/H)\bigr)=\QQ(\zeta_3),
\]
则分解
\[
J(X)\sim J(H)\times J(Z)^2\times J(CF)^3\times P^3
\]
应当在 \(S_4\)-等变有理同源范畴中提升为中心极小分解：四块恰对应于
\[
\End^0(J(X))^{S_4}
\]
的四个不可再细化中心因子。

换言之，在等型正交性、\(2/3\)-主极化缺陷分离以及 Prym 内部额外自同态极小性的共同作用下，不应再存在把这四块进一步中心融合的空间。
\end{conjecture}
