\begin{theorem}[有限截断混合隐藏态的完全同态计数公式]\label{thm:conclusion-mixed-hiddenstate-finite-truncation-hom-count}
沿用小节 \ref{subsec:conclusion-mixed-hiddenstate-finite-abelian-observability} 的记号，并记
\[
H_S:=\mathcal H_S=K_S^{\mathrm{col}}\times \widehat{\ZZ}\cong (\ZZ/2\ZZ)^\beta\times \widehat{\ZZ},
\]
\[
H_{S,N}:=\mathcal H_{S,N}=K_S^{\mathrm{col}}\times \ZZ/(N\ZZ)
\cong
(\ZZ/2\ZZ)^\beta\times \ZZ/(N\ZZ).
\]
则对任意有限阿贝尔群 \(G\) 与任意正整数 \(N\)，都有自然群同构
\[
\operatorname{Hom}(H_{S,N},G)\cong G[2]^{\,\beta}\oplus G[N].
\]
特别地，
\[
\bigl|\operatorname{Hom}(H_{S,N},G)\bigr|
=
|G[2]|^{\beta}\,|G[N]|.
\]
\end{theorem}
\begin{proof}
由直积的泛性质，
\[
\operatorname{Hom}(H_{S,N},G)
\cong
\operatorname{Hom}\bigl((\ZZ/2\ZZ)^\beta,G\bigr)\oplus \operatorname{Hom}(\ZZ/(N\ZZ),G).
\]
第一项中，每个基向量的像必须满足 \(2g=0\)，故
\[
\operatorname{Hom}\bigl((\ZZ/2\ZZ)^\beta,G\bigr)\cong G[2]^\beta.
\]
第二项则是有限循环群的标准同构：
\[
\operatorname{Hom}(\ZZ/(N\ZZ),G)\cong G[N].
\]
两式合并即得自然同构。基数公式随即成立。证毕。
\end{proof}

\begin{theorem}[混合隐藏态有限观测的素数支撑完全分解]\label{thm:conclusion-mixed-hiddenstate-prime-support-complete-splitting}
若
\[
G=\bigoplus_p G_{(p)},
\qquad
N=\prod_p p^{a_p},
\]
则有自然分解
\[
\operatorname{Hom}(H_{S,N},G)
\cong
G_{(2)}[2]^{\,\beta}\oplus \bigoplus_{p^{a_p}\parallel N} G_{(p)}[p^{a_p}].
\]
从而
\[
\bigl|\operatorname{Hom}(H_{S,N},G)\bigr|
=
|G_{(2)}[2]|^{\beta}\prod_{p^{a_p}\parallel N}|G_{(p)}[p^{a_p}]|.
\]
因此，有限阿贝尔观测的全部素数支撑严格局限于
\[
\{2\}\cup \{p:p\mid N\}.
\]
凡 \(p\nmid 2N\) 的 \(p\)-主分量，对 \(H_{S,N}\) 完全不可见。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-mixed-hiddenstate-finite-truncation-hom-count}，
\[
\operatorname{Hom}(H_{S,N},G)\cong G[2]^\beta\oplus G[N].
\]
对有限阿贝尔群的主分量分解有
\[
G[2]=G_{(2)}[2],
\]
并且
\[
G[N]=\bigoplus_{p^{a_p}\parallel N}G_{(p)}[p^{a_p}],
\]
因为 \(N\)-挠元只能出现在 \(N\) 所含素数的主分量上。代回即得同构与基数分解。最后一条只是该分解的直接语言化表述。证毕。
\end{proof}

\begin{theorem}[有限截断混合隐藏态的字符导体 \texorpdfstring{\(\zeta\)}{zeta} 函数]\label{thm:conclusion-mixed-hiddenstate-character-conductor-zeta}
\leanverified{paper\_conclusion\_mixed\_hiddenstate\_character\_conductor\_zeta}
把 Pontryagin 对偶
\[
\widehat H_{S,N}\cong (\ZZ/2\ZZ)^\beta\times \ZZ/(N\ZZ)
\]
中的角色写成
\[
\chi=\varepsilon\otimes\psi,
\qquad
\varepsilon\in \widehat{(\ZZ/2\ZZ)^\beta},\ \psi\in \widehat{\ZZ/(N\ZZ)}.
\]
定义
\[
\operatorname{cond}(\chi):=\operatorname{cond}(\psi).
\]
则对每个 \(d\mid N\)，恰有
\[
\#\{\chi\in \widehat H_{S,N}:\ \operatorname{cond}(\chi)=d\}=2^\beta\varphi(d).
\]
因而字符导体 \(\zeta\)-函数
\[
\mathcal Z_{S,N}(s):=\sum_{\chi\in \widehat H_{S,N}}\operatorname{cond}(\chi)^{-s}
\]
具有精确闭式
\[
\mathcal Z_{S,N}(s)=2^\beta\sum_{d\mid N}\varphi(d)d^{-s}
=
2^\beta\prod_{p^a\parallel N}
\left(
1+\sum_{j=1}^{a}\varphi(p^j)p^{-js}
\right).
\]
\end{theorem}
\begin{proof}
\((\ZZ/2\ZZ)^\beta\) 的角色部分与导体无关，只给出 \(2^\beta\) 重简并。另一方面，\(\widehat{\ZZ/(N\ZZ)}\cong \ZZ/(N\ZZ)\) 的标准角色中，精确导体为 \(d\mid N\) 的角色个数恰为 \(\varphi(d)\)。于是精确导体计数为
\[
2^\beta\varphi(d).
\]
对 \(d\mid N\) 求和即得
\[
\mathcal Z_{S,N}(s)=2^\beta\sum_{d\mid N}\varphi(d)d^{-s}.
\]
再由 \(\varphi\) 的乘法性和约数分解，
\[
\sum_{d\mid N}\varphi(d)d^{-s}
=
\prod_{p^a\parallel N}
\left(
1+\sum_{j=1}^{a}\varphi(p^j)p^{-js}
\right),
\]
即得 Euler 乘积形式。证毕。
\end{proof}

\begin{corollary}[随机字符导体分布的 \texorpdfstring{$\beta$}{beta}-独立性]\label{cor:conclusion-mixed-hiddenstate-random-character-conductor-law-beta-independence}
在 \(\widehat H_{S,N}\) 上取均匀随机角色 \(\chi\)。则对每个 \(d\mid N\)，都有
\[
\PP\bigl(\operatorname{cond}(\chi)=d\bigr)=\frac{\varphi(d)}{N},
\]
从而导体的整个分布完全独立于 \(\beta\)。等价地，碰撞扭量参数 \(\beta\) 只带来一个均匀的 \(2^\beta\)-重简并，而不扭曲任何导体统计。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-mixed-hiddenstate-character-conductor-zeta}，
\[
\#\{\chi\in \widehat H_{S,N}:\ \operatorname{cond}(\chi)=d\}=2^\beta\varphi(d),
\]
而总角色数为
\[
|\widehat H_{S,N}|=2^\beta N.
\]
两式相除即得
\[
\PP\bigl(\operatorname{cond}(\chi)=d\bigr)
=
\frac{2^\beta\varphi(d)}{2^\beta N}
=
\frac{\varphi(d)}{N}.
\]
证毕。
\end{proof}

\begin{corollary}[随机字符导体的精确对数期望]\label{cor:conclusion-mixed-hiddenstate-random-character-logconductor}
在 \(\widehat H_{S,N}\) 上取均匀随机角色 \(\chi\)。则
\[
\EE[\log \operatorname{cond}(\chi)]
=
\frac1N\sum_{d\mid N}\varphi(d)\log d
=
\sum_{p^a\parallel N}
\left(
a-\frac{1-p^{-a}}{p-1}
\right)\log p.
\]
特别地，这个期望完全独立于 \(\beta\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-mixed-hiddenstate-character-conductor-zeta}，角色总数为
\[
|\widehat H_{S,N}|=2^\beta N,
\]
而导体为 \(d\) 的角色个数为 \(2^\beta\varphi(d)\)。因此
\[
\EE[\log \operatorname{cond}(\chi)]
=
\frac{1}{2^\beta N}\sum_{d\mid N}2^\beta\varphi(d)\log d
=
\frac1N\sum_{d\mid N}\varphi(d)\log d.
\]
从而 \(\beta\) 自动消去。

再把 \(N=\prod_{p^a\parallel N}p^a\) 分解。若记
\[
J_p:=v_p(\operatorname{cond}(\chi)),
\]
则
\[
\log\operatorname{cond}(\chi)=\sum_{p^a\parallel N}J_p\log p.
\]
对固定的 \(p^a\parallel N\)，有
\[
\PP(J_p=0)=\frac1{p^a},
\qquad
\PP(J_p=j)=\frac{\varphi(p^j)}{p^a}\quad(1\le j\le a).
\]
于是
\[
\EE[J_p]
=
\frac1{p^a}\sum_{j=1}^{a}j\varphi(p^j)
=
\frac1{p^a}\sum_{j=1}^{a}j(p^j-p^{j-1})
=
a-\frac{1-p^{-a}}{p-1}.
\]
对全部素数求和即得闭式。证毕。
\end{proof}

\begin{theorem}[循环受体的忠实可观测性判据与最小循环规模]\label{thm:conclusion-mixed-hiddenstate-cyclic-receiver-criterion}
存在单射群同态
\[
H_{S,N}\hookrightarrow \ZZ/(m\ZZ)
\]
当且仅当以下两种情形之一成立：
\begin{enumerate}
  \item \(\beta=0\) 且 \(N\mid m\)；
  \item \(\beta=1\)、\(N\) 为奇数，且 \(2N\mid m\)。
\end{enumerate}
等价地，
\[
H_{S,N}\ \text{为循环群}
\iff
\beta=0\ \text{或}\ (\beta=1\ \text{且}\ N\ \text{奇}).
\]
相应的最小循环受体规模为
\[
m_{\min}^{\mathrm{cyc}}=
\begin{cases}
N,& \beta=0,\\[1mm]
2N,& \beta=1,\ N\ \text{奇},\\[1mm]
+\infty,& \text{其余情形}.
\end{cases}
\]
\end{theorem}
\begin{proof}
任何循环群的子群仍然循环。因此若存在
\[
H_{S,N}\hookrightarrow \ZZ/(m\ZZ),
\]
则 \(H_{S,N}\) 自身必须循环。

现在
\[
H_{S,N}\cong (\ZZ/2\ZZ)^\beta\times \ZZ/(N\ZZ).
\]
若 \(\beta=0\)，显然它就是 \(\ZZ/(N\ZZ)\)，故循环。若 \(\beta=1\) 且 \(N\) 奇，则由互素分解
\[
\ZZ/2\ZZ\times \ZZ/(N\ZZ)\cong \ZZ/(2N\ZZ),
\]
仍然循环。反之，若 \(\beta\ge 2\)，则其 \(2\)-主分量至少含有 \((\ZZ/2\ZZ)^2\)，不循环。若 \(\beta=1\) 且 \(2\mid N\)，则 \(2\)-主分量为
\[
\ZZ/2\ZZ\times \ZZ/2^{v_2(N)}\ZZ,
\]
亦不循环。故循环性的充要条件得证。

若 \(H_{S,N}\) 循环，则它嵌入 \(\ZZ/(m\ZZ)\) 的充要条件是其阶整除 \(m\)。于是 \(\beta=0\) 时需要 \(N\mid m\)，\(\beta=1\) 且 \(N\) 奇时需要 \(2N\mid m\)。最小 \(m\) 遂如上。证毕。
\end{proof}

\begin{theorem}[忠实有限阿贝尔受体的最小生成元数]\label{thm:conclusion-mixed-hiddenstate-minimal-generator-number}
\leanverified{paper\_conclusion\_mixed\_hiddenstate\_minimal\_generator\_number}
记 \(d(G)\) 为有限阿贝尔群 \(G\) 的最小生成元数。则一切忠实受体
\[
H_{S,N}\hookrightarrow G
\]
都必须满足
\[
d(G)\ge d(H_{S,N}),
\]
且
\[
d(H_{S,N})=
\begin{cases}
0,& \beta=0,\ N=1,\\[1mm]
1,& \beta=0,\ N>1,\\[1mm]
\beta,& \beta\ge 1,\ N\ \text{奇},\\[1mm]
\beta+1,& 2\mid N.
\end{cases}
\]
该下界是可达的：取 \(G=H_{S,N}\) 即达到。
\end{theorem}
\begin{proof}
若 \(H_{S,N}\hookrightarrow G\)，则 \(H_{S,N}\) 是 \(G\) 的子群。有限阿贝尔群的最小生成元数对取子群单调不增，因此
\[
d(G)\ge d(H_{S,N}).
\]

现计算 \(d(H_{S,N})\)。把
\[
H_{S,N}\cong (\ZZ/2\ZZ)^\beta\times \ZZ/(N\ZZ)
\]
按素数分解为主分量直积。对奇素数 \(p\mid N\)，\(p\)-主分量是循环的，故需要 \(1\) 个生成元。对 \(p=2\)：
\begin{itemize}
  \item 若 \(2\nmid N\)，则 \(2\)-主分量为 \((\ZZ/2\ZZ)^\beta\)，需要 \(\beta\) 个生成元；
  \item 若 \(2\mid N\)，则 \(2\)-主分量为
  \[
  (\ZZ/2\ZZ)^\beta\times \ZZ/2^{v_2(N)}\ZZ,
  \]
  需要 \(\beta+1\) 个生成元。
\end{itemize}
有限阿贝尔群的最小生成元数等于各主分量生成元数的最大值，于是得到所示分段公式。取 \(G=H_{S,N}\) 即知下界可达。证毕。
\end{proof}

\begin{theorem}[带指数上界的忠实受体存在性与极小规模]\label{thm:conclusion-mixed-hiddenstate-exponent-constrained-receiver}
固定正整数 \(E\)。存在有限阿贝尔群 \(G\) 满足
\[
H_{S,N}\hookrightarrow G,
\qquad
\exp(G)\mid E
\]
当且仅当
\[
\exp(H_{S,N})\mid E.
\]
等价地，
\[
\begin{cases}
N\mid E,& \beta=0,\\[1mm]
\operatorname{lcm}(2,N)\mid E,& \beta>0.
\end{cases}
\]
在此条件下，所有满足 \(\exp(G)\mid E\) 的忠实受体中，最小可能基数仍恰为
\[
|G|_{\min}=2^\beta N.
\]
\end{theorem}
\begin{proof}
必要性显然：若 \(H_{S,N}\) 嵌入 \(G\)，则子群的指数整除宿主群指数，故
\[
\exp(H_{S,N})\mid \exp(G)\mid E.
\]
而
\[
\exp(H_{S,N})=
\begin{cases}
N,& \beta=0,\\[1mm]
\operatorname{lcm}(2,N),& \beta>0.
\end{cases}
\]

反之，若该整除条件成立，则取
\[
G=H_{S,N}
\]
自身即可，其指数显然整除 \(E\)，且嵌入由恒等映射给出。

关于最小基数，定理 \ref{thm:conclusion-mixed-hidden-state-minimal-finite-receiver} 已给出在全部有限阿贝尔受体中
\[
|G|_{\min}=2^\beta N.
\]
加入额外约束 \(\exp(G)\mid E\) 只会缩小可行类，不会降低该最小值；而 \(G=H_{S,N}\) 已在当前约束下可行，故最小值仍为 \(2^\beta N\)。证毕。
\end{proof}

\begin{theorem}[忠实有限受体的逐素数最小支撑律]\label{thm:conclusion-mixed-hiddenstate-prime-by-prime-minimal-support}
设
\[
N=\prod_p p^{a_p}.
\]
则有限阿贝尔群 \(G\) 存在一条忠实嵌入
\[
H_{S,N}\hookrightarrow G
\]
当且仅当对每个素数 \(p\) 都有：
\begin{enumerate}
  \item 若 \(p\neq 2\) 且 \(a_p>0\)，则 \(G_{(p)}\) 含有一个阶为 \(p^{a_p}\) 的循环子群；
  \item 对 \(p=2\)，\(G_{(2)}\) 含有子群
  \[
  (\ZZ/2\ZZ)^\beta\times \ZZ/2^{a_2}\ZZ,
  \]
  其中约定 \(a_2=0\) 时第二因子平凡。
\end{enumerate}
因而必有逐素数下界
\[
|G_{(p)}|\ge p^{a_p}\qquad (p\neq 2,\ a_p>0),
\]
\[
|G_{(2)}|\ge 2^{\beta+a_2},
\]
从而
\[
|G|\ge 2^\beta N.
\]
并且当且仅当每个 \(p\)-主分量都恰好取到上述最小值时，
\[
G\cong H_{S,N}.
\]
\end{theorem}
\begin{proof}
若 \(H_{S,N}\hookrightarrow G\)，则对每个 \(p\)，其 \(p\)-主分量
\[
(H_{S,N})_{(p)}
\]
必须嵌入 \(G_{(p)}\)。而
\[
(H_{S,N})_{(p)}\cong
\begin{cases}
\ZZ/p^{a_p}\ZZ,& p\neq 2,\ a_p>0,\\[1mm]
(\ZZ/2\ZZ)^\beta\times \ZZ/2^{a_2}\ZZ,& p=2.
\end{cases}
\]
故所述子群包含条件必要。

反之，若每个 \(G_{(p)}\) 都包含对应的 \((H_{S,N})_{(p)}\)，则这些嵌入可在各 \(p\)-主分量上独立给出，直和拼接后得到
\[
\bigoplus_p(H_{S,N})_{(p)}\hookrightarrow \bigoplus_p G_{(p)}=G,
\]
即 \(H_{S,N}\hookrightarrow G\)。

逐素数的基数下界由上述包含立即得出，相乘得到
\[
|G|\ge \prod_{p\neq 2,\ a_p>0}p^{a_p}\cdot 2^{\beta+a_2}=2^\beta N.
\]
若每个主分量恰取最小基数，则
\[
|G_{(p)}|=|(H_{S,N})_{(p)}|
\]
对所有 \(p\) 成立。有限同阶 \(p\)-群之间一旦存在单射便为同构，故每个主分量都与 \((H_{S,N})_{(p)}\) 同构，进而
\[
G\cong H_{S,N}.
\]
证毕。
\end{proof}
