\begin{theorem}[Pisano-\texorpdfstring{$1$}{1} 分量计数诱导的球面相、界面相与正号海三相律]\label{thm:conclusion-pisano1-threephase-sign-sea-law}
对任意 \(x\in X_m\)，设
\[
\Gamma_x\cong \bigsqcup_{j=1}^{r}P_{\ell_j(x)},
\qquad
N_m(x):=\#\{j:\ell_j(x)\equiv 1\pmod 3\}.
\]
则纤维同伦型与扫描/时间反演两类符号异常满足严格三相分裂：
\begin{enumerate}
  \item 若 \(N_m(x)=0\)，则
  \[
  |K_x|\simeq S^{\tau(x)-1},
  \qquad
  \operatorname{sgn}(c_x)=(-1)^{\sum_{j=1}^{r}e(\ell_j)},
  \qquad
  \operatorname{sgn}(\iota_x)=\prod_{j=1}^{r}\delta(\ell_j).
  \]
  \item 若 \(N_m(x)=1\) 且唯一坏分量为 \(A=\{k\}\)，则
  \[
  |K_x|\ \text{可缩},
  \qquad
  \operatorname{sgn}(c_x)=-1 \iff \ell_k\equiv 1\pmod 6,
  \qquad
  \operatorname{sgn}(\iota_x)=\delta(\ell_k).
  \]
  \item 若 \(N_m(x)\ge 2\)，则
  \[
  |K_x|\ \text{可缩},
  \qquad
  \operatorname{sgn}(c_x)=\operatorname{sgn}(\iota_x)=+1.
  \]
\end{enumerate}
因此，纤维对象层严格分裂为三类互不混同的相：球面相 \(N_m=0\)、余维一异常界面 \(N_m=1\)、以及 \(N_m\ge 2\) 的正号海。
\leanverified{paper\_conclusion\_pisano1\_threephase\_sign\_sea\_law}
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，
\[
|K_x|\simeq S^{\tau(x)-1}
\iff
N_m(x)=0,
\qquad
|K_x|\ \text{可缩}
\iff
N_m(x)\ge 1.
\]
另一方面，定理 \ref{thm:pom-toggle-scan-sign-general-fiber} 给出：
\begin{itemize}
  \item 若 \(N_m(x)=0\)，则
  \[
  \operatorname{sgn}(c_x)=(-1)^{\sum_{j=1}^{r}e(\ell_j)};
  \]
  \item 若 \(N_m(x)=1\) 且 \(A=\{k\}\)，则
  \[
  \operatorname{sgn}(c_x)=(-1)^{e(\ell_k)},
  \]
  并在约束 \(\ell_k\equiv 1\pmod 3\) 下等价于
  \(
  \operatorname{sgn}(c_x)=-1 \iff \ell_k\equiv 1\pmod 6
  \)；
  \item 若 \(N_m(x)\ge 2\)，则 \(\operatorname{sgn}(c_x)=+1\)。
\end{itemize}
同样，定理 \ref{thm:pom-toggle-time-reversal-sign-general-fiber} 给出：
\begin{itemize}
  \item 若 \(N_m(x)=0\)，则
  \[
  \operatorname{sgn}(\iota_x)=\prod_{j=1}^{r}\delta(\ell_j);
  \]
  \item 若 \(N_m(x)=1\) 且 \(A=\{k\}\)，则
  \[
  \operatorname{sgn}(\iota_x)=\delta(\ell_k);
  \]
  \item 若 \(N_m(x)\ge 2\)，则 \(\operatorname{sgn}(\iota_x)=+1\)。
\end{itemize}
把同伦判别与两条符号判别合并，即得三相律。证毕。
\end{proof}

\begin{corollary}[正号海的指数统治与球面相的测度消失]\label{cor:conclusion-pisano1-positive-sea-exponential-dominance}
在 \(X_m\) 上取均匀分布 \(U_m=\mathrm{Unif}(X_m)\)。
则存在常数 \(\mu>0,c>0\)，使对任意固定 \(\varepsilon\in(0,\mu)\)，充分大 \(m\) 时有
\[
U_m\bigl(N_m(x)\le \varepsilon m\bigr)\le e^{-cm}.
\]
从而特别存在常数 \(c_0>0\) 使得充分大 \(m\) 时
\[
U_m\bigl(N_m(x)\le 1\bigr)\le e^{-c_0m}.
\]
于是由结论 \ref{thm:conclusion-pisano1-threephase-sign-sea-law}，
\[
U_m\bigl(|K_x|\not\simeq *\bigr)\le e^{-c_0m},
\qquad
U_m\bigl(\operatorname{sgn}(c_x)=-1\bigr)\le e^{-c_0m},
\qquad
U_m\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)\le e^{-c_0m}.
\]
因此，在稳定类型的均匀抽样下，球面相与全部负号异常都是指数稀疏的；绝大多数纤维落在 \(N_m(x)\ge 2\) 的正号海内。
\leanverified{paper\_conclusion\_pisano1\_positive\_sea\_exponential\_dominance}
\end{corollary}

\begin{proof}
前一条指数下尾估计正是定理 \ref{thm:pom-pisano1-components-ldp-linear}。
固定任意 \(\varepsilon\in(0,\mu)\)，当 \(m\) 充分大时有 \(1\le \varepsilon m\)，故
\[
\{N_m(x)\le 1\}\subseteq \{N_m(x)\le \varepsilon m\},
\]
于是
\[
U_m\bigl(N_m(x)\le 1\bigr)\le e^{-cm}.
\]
再由结论 \ref{thm:conclusion-pisano1-threephase-sign-sea-law}，
\[
|K_x|\not\simeq * \iff N_m(x)=0,
\]
而
\[
\operatorname{sgn}(c_x)=-1
\quad\text{或}\quad
\operatorname{sgn}(\iota_x)=-1
\Longrightarrow
N_m(x)\le 1.
\]
将前述指数界代入即可。证毕。
\leanverified{paper\_conclusion\_pisano1\_positive\_sea\_seeds}
\leanverified{paper\_conclusion\_pisano1\_positive\_sea\_package}
\end{proof}

\begin{theorem}[双负号事件的同势垒异残量]\label{thm:conclusion-negative-sign-same-barrier-different-residue}
\leanverified{paper\_conclusion\_negative\_sign\_same\_barrier\_different\_residue}
在文稿的分解模型 \(P_L^{(1)}\) 下，存在显式正常数
\[
A_c:=\frac{\mu_1\,C_c^{-}(\rho_0)}{\mu_0^2},
\qquad
A_\iota:=\frac{\mu_1\,C_\iota^{-}(\rho_0)}{\mu_0^2},
\]
使得
\[
P_L^{(1)}\!\bigl(\operatorname{sgn}(c_x)=-1\bigr)\sim A_c\,L\left(\frac{\lambda_0}{\lambda_1}\right)^L,
\]
\[
P_L^{(1)}\!\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)\sim A_\iota\,L\left(\frac{\lambda_0}{\lambda_1}\right)^L.
\]
因此
\[
I_c=I_\iota=\log\frac{\lambda_1}{\lambda_0},
\]
并且
\[
\frac{P_L^{(1)}(\operatorname{sgn}(c_x)=-1)}
{P_L^{(1)}(\operatorname{sgn}(\iota_x)=-1)}
\longrightarrow
\frac{C_c^{-}(\rho_0)}{C_\iota^{-}(\rho_0)}.
\]
这表明扫描负号与时间反演负号共享同一自由能势垒，但其首残量常数由不同的算术同余类生成函数严格区分。
\end{theorem}

\begin{proof}
这正是定理 \ref{thm:pom-multiplicity-composition-chi-anomaly-sharp-rate} 的两条尖锐渐近式。
对两式分别取对数并除以 \(L\)，即得
\[
-\lim_{L\to\infty}\frac{1}{L}\log P_L^{(1)}\!\bigl(\operatorname{sgn}(c_x)=-1\bigr)
=
-\lim_{L\to\infty}\frac{1}{L}\log P_L^{(1)}\!\bigl(\operatorname{sgn}(\iota_x)=-1\bigr)
=
\log\frac{\lambda_1}{\lambda_0},
\]
即 \(I_c=I_\iota\)。
再把两条渐近式相除，线性因子 \(L\) 与公共指数底 \((\lambda_0/\lambda_1)^L\) 消去，即得常数比极限。证毕。
\end{proof}

\begin{theorem}[时间反演轨道压缩缺口的奇偶律]\label{thm:conclusion-time-reversal-orbit-compression-gap-parity}
定义时间反演商轨道数
\[
Q_x:=|F_x/\langle \iota_x\rangle|
\]
以及压缩缺口
\[
\Delta_x:=d_m(x)-Q_x.
\]
则有精确闭式
\[
Q_x=\frac{d_m(x)+|\mathrm{Fix}(\iota_x)|}{2},
\qquad
\Delta_x=\frac{d_m(x)-|\mathrm{Fix}(\iota_x)|}{2},
\]
并且
\[
\operatorname{sgn}(\iota_x)=(-1)^{\Delta_x}.
\]
等价地，
\[
\operatorname{sgn}(\iota_x)=+1
\iff
Q_x\equiv d_m(x)\pmod 2,
\]
\[
\operatorname{sgn}(\iota_x)=-1
\iff
Q_x\equiv d_m(x)+1\pmod 2.
\]
因此，时间反演符号并不是附着在纤维上的独立标记；它恰是轨道压缩缺口的奇偶性。
\leanverified{paper\_conclusion\_time\_reversal\_orbit\_compression\_gap\_parity}
\end{theorem}

\begin{proof}
由推论 \ref{cor:pom-time-reversal-orbit-count}，
\[
Q_x=\frac{d_m(x)+|\mathrm{Fix}(\iota_x)|}{2}.
\]
故
\[
\Delta_x=d_m(x)-Q_x=\frac{d_m(x)-|\mathrm{Fix}(\iota_x)|}{2}.
\]
另一方面，定理 \ref{thm:pom-toggle-time-reversal-fixedpoints-sign-closed-form} 给出
\[
\operatorname{sgn}(\iota_x)=(-1)^{(d_m(x)-|\mathrm{Fix}(\iota_x)|)/2}.
\]
与上式合并，即得
\[
\operatorname{sgn}(\iota_x)=(-1)^{\Delta_x}.
\]
其后的模 \(2\) 形式只是该恒等式的直接改写。证毕。
\end{proof}

\begin{corollary}[非可缩扇区上的时间反演符号完全由轨道商奇偶决定]\label{cor:conclusion-noncontractible-time-reversal-sign-orbit-quotient-parity}
若 \(|K_x|\not\simeq *\)，则 \(d_m(x)\) 必为奇数，并且
\[
\operatorname{sgn}(\iota_x)=(-1)^{Q_x+1}.
\]
等价地，
\[
|K_x|\not\simeq *
\Longrightarrow
\begin{cases}
\operatorname{sgn}(\iota_x)=+1 &\Longleftrightarrow\ Q_x\equiv 1\pmod 2,\\[1mm]
\operatorname{sgn}(\iota_x)=-1 &\Longleftrightarrow\ Q_x\equiv 0\pmod 2.
\end{cases}
\]
因此，在非可缩扇区内，时间反演符号不再是独立自由度，而被轨道商大小的奇偶性完全锁定。
\leanverified{paper\_conclusion\_noncontractible\_time\_reversal\_sign\_parity}
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-tqft-noncontractible-genus-equals-odd-spectrum}，
\[
|K_x|\not\simeq * \Longrightarrow d_m(x)\equiv 1\pmod 2.
\]
另一方面，定理 \ref{thm:conclusion-time-reversal-orbit-compression-gap-parity} 给出
\[
\operatorname{sgn}(\iota_x)=(-1)^{\Delta_x},
\qquad
\Delta_x=d_m(x)-Q_x.
\]
于是模 \(2\) 下
\[
\Delta_x\equiv 1-Q_x\equiv Q_x+1\pmod 2,
\]
从而
\[
\operatorname{sgn}(\iota_x)=(-1)^{Q_x+1}.
\]
其后的奇偶判据只是直接展开。证毕。
\end{proof}

\begin{theorem}[测度稀疏性与极值刚性的严格分离]\label{thm:conclusion-spherical-phase-measure-sparsity-extremal-rigidity-separation}
定义球面相
\[
\mathcal S_m:=\{x\in X_m:\ |K_x|\not\simeq *\}.
\]
则一方面，存在常数 \(c>0\) 使得
\[
U_m(\mathcal S_m)\le e^{-cm}
\]
对充分大 \(m\) 成立；另一方面，
\[
\sup_{x\in\mathcal S_m}d_m(x)=\widetilde D_m=\Theta(D_m),
\]
并且
\[
\liminf_{m\to\infty}\frac{\widetilde D_m}{D_m}>0.
\]
故球面相在计数测度下是指数罕见事件，但在极值谱上只付常数税。这说明拓扑复杂性与概率稀疏性并非同一个量纲，而在当前体系中形成两条彼此横截的尺度。
\end{theorem}

\begin{proof}
由结论 \ref{cor:conclusion-pisano1-positive-sea-exponential-dominance}，
\[
U_m(\mathcal S_m)\le e^{-cm}
\]
对某个 \(c>0\) 成立。
另一方面，由推论 \ref{cor:conclusion-max-noncontractible-fiber-exponential-rate}，
\[
\widetilde D_m=\Theta(D_m).
\]
特别地，存在常数 \(c'>0\) 使充分大 \(m\) 时
\[
\frac{\widetilde D_m}{D_m}\ge c',
\]
从而
\[
\liminf_{m\to\infty}\frac{\widetilde D_m}{D_m}\ge c'>0.
\]
两者并置即得“测度稀疏—极值刚性”分离。证毕。
\end{proof}

\begin{conjecture}[极大球面纤维的时间反演商比收敛]\label{conj:conclusion-maxsphere-fiber-time-reversal-orbit-ratio}
对每个剩余类 \(r\in\ZZ/6\ZZ\)，存在常数
\[
\varrho_r\in(0,1]
\]
使得对任意满足
\[
m\equiv r\pmod 6,
\qquad
d_m(x_m)=\widetilde D_m,
\qquad
|K_{x_m}|\not\simeq *,
\]
的极大球面纤维序列 \(\{x_m\}\)，都有
\[
\frac{|\mathrm{Fix}(\iota_{x_m})|}{\widetilde D_m}\longrightarrow \varrho_r,
\qquad
\frac{|F_{x_m}/\langle \iota_{x_m}\rangle|}{\widetilde D_m}
\longrightarrow
\frac{1+\varrho_r}{2}.
\]
该猜想的结构依据在于：一方面，定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios} 已表明极大球面纤维只在模 \(6\) 周期地从前三层谱中选取；另一方面，时间反演不动点数具有严格乘积公式
\[
|\mathrm{Fix}(\iota_x)|=\prod_j f(\ell_j),
\]
故一旦极大球面纤维的路径分解模板在每个模 \(6\) 类中最终稳定到有限候选，其固定点密度便应冻结为单一常数。

结合推论 \ref{cor:conclusion-noncontractible-time-reversal-sign-orbit-quotient-parity}，每个模 \(6\) 扇区的极大球面纤维应最终只需两类数据即可刻画：其一是连续振幅参数 \(\varrho_r\)，其二是离散奇偶位
\[
\epsilon_r(m):=\frac{\widetilde D_m-|\mathrm{Fix}(\iota_{x_m})|}{2}\pmod 2,
\]
后者完全决定时间反演符号。若该冻结成立，则模 \(6\) 扇区上的极大非可缩纤维将收缩为“一个连续密度参数加一个离散奇偶位”的最小数据包。
\end{conjecture}
