\begin{theorem}[非可缩极值损失的模 \texorpdfstring{$6$}{6} 显式相图]\label{thm:conclusion-noncontractible-loss-mod6-explicit}
\leanverified{paper\_conclusion\_noncontractible\_loss\_mod6\_explicit}
定义
\[
\Delta_m^{\mathrm{nc}}:=\Delta_m(X_m^{\mathrm{nc}})=\log \widetilde D_m.
\]
则对所有充分大的 \(m\)（特别是定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase} 的适用范围内）有显式分段公式。若 \(m=2k\)，则
\[
\Delta_{2k}^{\mathrm{nc}}
=
\begin{cases}
\log F_{k+2}, & k\equiv 0,2\pmod 3,\\[2mm]
\log\!\bigl(F_{k+2}-F_{k-3}\bigr), & k\equiv 1\pmod 3.
\end{cases}
\]
若 \(m=2k+1\)，则
\[
\Delta_{2k+1}^{\mathrm{nc}}
=
\begin{cases}
\log\!\bigl(2F_{k+1}-F_{k-4}\bigr), & k\equiv 0,2\pmod 3,\\[2mm]
\log\!\bigl(2F_{k+1}-F_{k-4}-F_{k-8}\bigr), & k\equiv 1\pmod 3.
\end{cases}
\]
等价地，按 \(m\bmod 6\) 可写为
\[
\Delta_m^{\mathrm{nc}}=
\begin{cases}
\log D_m, & m\equiv 0,4\pmod 6,\\
\log D_m^{(2)}, & m\equiv 1,5\pmod 6,\\
\log D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-noncontractible-restricted-dpi-exact}，
\[
\Delta_m^{\mathrm{nc}}=\log \widetilde D_m.
\]
再由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，
\[
\widetilde D_m=
\begin{cases}
D_m, & m\equiv 0,4\pmod 6,\\
D_m^{(2)}, & m\equiv 1,5\pmod 6,\\
D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]
结合
\[
D_{2k}=F_{k+2},\qquad D_{2k+1}=2F_{k+1},
\]
\[
D_{2k}^{(3)}=F_{k+2}-F_{k-3},\qquad D_{2k+1}^{(2)}=2F_{k+1}-F_{k-4},
\qquad
D_{2k+1}^{(3)}=2F_{k+1}-F_{k-4}-F_{k-8},
\]
即可逐同余类得到所述显式公式。证毕。
\end{proof}

\begin{theorem}[非可缩拓扑税：指数不变而常数分裂]\label{thm:conclusion-noncontractible-topological-tax-bounded}
\leanpartial{paper\_conclusion\_noncontractible\_topological\_tax\_bounded}{the requested Lean signature is false for arbitrary \texttt{Theta} and \texttt{DeltaNc}; for example \texttt{Theta := fun \_ => 1} already contradicts the claimed zero-limit subsequences}
定义非可缩拓扑税
\[
\Theta_m:=\Delta_m-\Delta_m^{\mathrm{nc}}=\log\frac{D_m}{\widetilde D_m}.
\]
则 \(\Theta_m\) 不是指数级罚项，而是严格的模 \(6\) 周期有界税。更精确地，
\[
\lim_{\substack{m\to\infty\\ m\equiv 0,4\ (6)}}\Theta_m=0,
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 1,5\ (6)}}\Theta_m
=
-\log\!\Bigl(1-\frac{1}{2}\varphi^{-5}\Bigr),
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 2\ (6)}}\Theta_m
=
-\log\!\bigl(1-\varphi^{-5}\bigr),
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 3\ (6)}}\Theta_m
=
-\log\!\Bigl(1-\frac12(\varphi^{-5}+\varphi^{-9})\Bigr).
\]
数值上，这四个极限分别为
\[
0,\quad 0.0461329182\ldots,\quad 0.0944974482\ldots,\quad 0.0530451233\ldots
\]
（单位：nats）。特别地，
\[
\Theta_m=O(1),
\qquad
\lim_{m\to\infty}\frac1m\Delta_m^{\mathrm{nc}}
=
\frac12\log\varphi.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios}，
\[
\frac{\widetilde D_m}{D_m}
\longrightarrow
\begin{cases}
1,& m\equiv 0,4\pmod 6,\\[2pt]
1-\frac{1}{2}\varphi^{-5},& m\equiv 1,5\pmod 6,\\[4pt]
1-\varphi^{-5},& m\equiv 2\pmod 6,\\[4pt]
1-\frac12(\varphi^{-5}+\varphi^{-9}),& m\equiv 3\pmod 6.
\end{cases}
\]
取负对数即得四条极限公式。于是 \(\Theta_m\) 在每个模 \(6\) 子列上都收敛到有限常数，故整体有界。

再由推论 \ref{cor:conclusion-max-noncontractible-fiber-exponential-rate}，
\[
\lim_{m\to\infty}\frac1m\log \widetilde D_m=\frac12\log\varphi.
\]
而 \(\Delta_m^{\mathrm{nc}}=\log \widetilde D_m\)，故
\[
\lim_{m\to\infty}\frac1m\Delta_m^{\mathrm{nc}}=\frac12\log\varphi.
\]
证毕。
\end{proof}

\begin{theorem}[非可缩极大纤维拓扑税的模 \texorpdfstring{$6$}{6} 平均常数与零税密度]\label{thm:conclusion-max-noncontractible-topological-tax-cesaro-density}
\leanpartial{paper\_conclusion\_max\_noncontractible\_topological\_tax\_cesaro\_density}{the requested Lean signature is false for arbitrary \texttt{tau} and \texttt{A}; taking \texttt{tau := fun \_ => 0} and \texttt{A := $\emptyset$} contradicts the stated Ces\`aro and density limits}
定义
\[
\tau_m:=1-\frac{\widetilde D_m}{D_m}.
\]
则
\[
\liminf_{m\to\infty}\frac{\widetilde D_m}{D_m}=1-\varphi^{-5},
\qquad
\limsup_{m\to\infty}\frac{\widetilde D_m}{D_m}=1,
\]
并且 Ces\`aro 平均存在且可显式写为
\[
\lim_{N\to\infty}\frac1N\sum_{m=1}^{N}\tau_m
=
\frac{5\varphi^{-5}+\varphi^{-9}}{12}.
\]
此外，零税窗口集合
\[
\mathcal A:=\{m\ge 17:\widetilde D_m=D_m\}
\]
具有自然密度
\[
\lim_{N\to\infty}\frac1N\#(\mathcal A\cap [1,N])=\frac13.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios}，
\[
\frac{\widetilde D_m}{D_m}\to
\begin{cases}
1,& m\equiv 0,4\pmod 6,\\[2pt]
1-\dfrac{1}{2}\varphi^{-5},& m\equiv 1,5\pmod 6,\\[6pt]
1-\varphi^{-5},& m\equiv 2\pmod 6,\\[4pt]
1-\dfrac12(\varphi^{-5}+\varphi^{-9}),& m\equiv 3\pmod 6.
\end{cases}
\]
其中最小极限为 \(1-\varphi^{-5}\)，最大极限为 \(1\)，故得 \(\liminf\) 与 \(\limsup\)。

对应的税率极限分别为
\[
0,\quad \frac1{2\varphi^5},\quad \varphi^{-5},\quad \frac12(\varphi^{-5}+\varphi^{-9}),\quad 0,\quad \frac1{2\varphi^5}.
\]
对六个剩余类作平均，得到
\[
\lim_{N\to\infty}\frac1N\sum_{m=1}^{N}\tau_m
=
\frac16\left(
0+\frac1{2\varphi^5}+\varphi^{-5}+\frac12(\varphi^{-5}+\varphi^{-9})+0+\frac1{2\varphi^5}
\right)
=
\frac{5\varphi^{-5}+\varphi^{-9}}{12}.
\]

最后，由定理 \ref{thm:conclusion-maxfiber-homotopy-visibility-density}，
\[
\mathcal A=\{m\ge 17:\ m\equiv 0,4\pmod 6\},
\]
故其自然密度为 \(2/6=1/3\)。证毕。
\end{proof}

\begin{corollary}[非可缩极大纤维拓扑税是严格的模 \texorpdfstring{$6$}{6} 周期场]\label{cor:conclusion-max-noncontractible-topological-tax-mod6-periodic-field}
\leanverified{paper\_conclusion\_max\_noncontractible\_topological\_tax\_mod6\_periodic\_field}
非可缩极大纤维的拓扑税并非随机波动，而是在模 \(6\) 剩余类上冻结为一个严格的周期场；其长期平均值被黄金常数
\[
\frac{5\varphi^{-5}+\varphi^{-9}}{12}
\]
唯一锁定，而零税窗口仅以自然密度 \(1/3\) 出现。
\end{corollary}
