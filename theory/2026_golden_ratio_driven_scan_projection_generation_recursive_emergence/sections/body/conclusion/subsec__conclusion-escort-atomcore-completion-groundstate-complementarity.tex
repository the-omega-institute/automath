\subsection{escort 常数层的 \texorpdfstring{$\log\varphi$}{log phi} 差额、平衡原子盲性与 atom/core 相图的互补结构}\label{subsec:conclusion-escort-atomcore-completion-groundstate-complementarity}
在定理 \ref{thm:xi-time-part65h-escort-renyi-o1-temperature-profile} 与定理 \ref{thm:xi-time-part65h-escort-shannon-debruijn-fisher} 已把 bin-fold escort 温度族的常数层、Fisher 信息与乘法射线压力缺口完全显式化，而小节 \ref{subsec:conclusion-completed-slice-atomblind-groundstate-escort-current}、小节 \ref{subsec:conclusion-realinput40-uv-atomcore-phase-shell-trace} 与小节 \ref{subsec:conclusion-primitive-weightedzeta-groundstate-hankel-adelic-splitting} 又已分别固定平衡原子的完成化失明、零温地基的正熵不可逆性以及二维 atom/core 分相之后，还可把这些接口进一步压缩为一条终端互补链。其核心事实是：escort 常数层沿温度轴损失的全部 \(O(1)\) 质量恰等于 \(\log\varphi\)；与此同时，完成化归一化会系统抹去 balanced atom，而有限部、残差壳层与二维压力相图则正好把这一被抹去的结构重新显影。

\begin{theorem}[escort 常数层的一 \texorpdfstring{$\log\varphi$}{log phi} 普适差额]\label{thm:conclusion-escort-constant-layer-universal-logphi-gap}
\leanverified{paper\_conclusion\_escort\_constant\_layer\_universal\_logphi\_gap}
设 \(\pi_{m,q}\) 为 bin-fold 的 escort 分布。对每个 \(\alpha>0\)，定义
\[
C_\alpha(q):=
\lim_{m\to\infty}\bigl(H_\alpha(\pi_{m,q})-m\log\varphi\bigr).
\]
则极限
\[
\lim_{q\to\infty}C_\alpha(q)
\]
存在，并且对一切 \(\alpha>0\) 都有严格恒等式
\[
\boxed{\;
C_\alpha(0)-\lim_{q\to\infty}C_\alpha(q)=\log\varphi.\;}
\]
特别地，
\[
C_\alpha(0)=-\frac12\log 5+2\log\varphi,
\qquad
\lim_{q\to\infty}C_\alpha(q)=-\frac12\log 5+\log\varphi.
\]
因此，escort 温度从 \(q=0\) 冷却到 \(q=\infty\) 时，全部 Rényi 常数层无论阶数如何改变，都只损失同一个常数包 \(\log\varphi\)。
\end{theorem}
\begin{proof}
当 \(\alpha\neq 1\) 时，定理 \ref{thm:xi-time-part65h-escort-renyi-o1-temperature-profile} 给出
\[
C_\alpha(q)
=
-\frac12\log 5
+
\frac{\alpha\log(\varphi+\varphi^{-q})-\log(\varphi+\varphi^{-\alpha q})}{\alpha-1}.
\]
令 \(q=0\) 并利用 \(\varphi+1=\varphi^2\)，得到
\[
C_\alpha(0)
=
-\frac12\log 5
+
\frac{2\alpha\log\varphi-2\log\varphi}{\alpha-1}
=
-\frac12\log 5+2\log\varphi.
\]
令 \(q\to\infty\) 则
\[
\lim_{q\to\infty}C_\alpha(q)
=
-\frac12\log 5
+
\frac{\alpha\log\varphi-\log\varphi}{\alpha-1}
=
-\frac12\log 5+\log\varphi.
\]
两式相减即得
\[
C_\alpha(0)-\lim_{q\to\infty}C_\alpha(q)=\log\varphi.
\]

当 \(\alpha=1\) 时，令
\[
\mathfrak h(q):=
\lim_{m\to\infty}\bigl(H(\pi_{m,q})-m\log\varphi\bigr).
\]
由定理 \ref{thm:xi-time-part65h-escort-shannon-debruijn-fisher}，
\[
\mathfrak h(q)
=
-\frac12\log 5+\log(\varphi+\varphi^{-q})
+\frac{q\log\varphi}{1+\varphi^{q+1}}.
\]
于是
\[
\mathfrak h(0)=-\frac12\log 5+\log(\varphi+1)=-\frac12\log 5+2\log\varphi,
\]
且
\[
\lim_{q\to\infty}\mathfrak h(q)=-\frac12\log 5+\log\varphi.
\]
故同样有
\[
\mathfrak h(0)-\lim_{q\to\infty}\mathfrak h(q)=\log\varphi.
\]
证毕。
\end{proof}

\begin{corollary}[escort Fisher 作用量的总质量恒等式]\label{cor:conclusion-escort-fisher-total-mass-logphi}
沿用定理 \ref{thm:xi-time-part65h-escort-shannon-debruijn-fisher} 的记号，定义
\[
\theta(q):=\frac{1}{1+\varphi^{q+1}},
\qquad
\mathcal I(q):=(\log\varphi)^2\theta(q)\bigl(1-\theta(q)\bigr),
\]
\[
\mathfrak h(q):=\lim_{m\to\infty}\bigl(H(\pi_{m,q})-m\log\varphi\bigr).
\]
则有精确导数关系
\[
\mathfrak h'(q)=-q\,\mathcal I(q),
\]
以及积分恒等式
\[
\boxed{\;
\int_0^\infty q\,\mathcal I(q)\,\dd q=\log\varphi.\;}
\]
\end{corollary}
\begin{proof}
导数公式正是定理 \ref{thm:xi-time-part65h-escort-shannon-debruijn-fisher}。再由定理 \ref{thm:conclusion-escort-constant-layer-universal-logphi-gap} 在 \(\alpha=1\) 的情形，
\[
\mathfrak h(0)-\lim_{q\to\infty}\mathfrak h(q)=\log\varphi.
\]
因此
\[
\int_0^\infty q\,\mathcal I(q)\,\dd q
=
-\int_0^\infty \mathfrak h'(q)\,\dd q
=
\mathfrak h(0)-\lim_{q\to\infty}\mathfrak h(q)
=
\log\varphi.
\]
证毕。
\end{proof}

\begin{theorem}[乘法射线上的有限平台刚性]\label{thm:conclusion-multiplicative-ray-finite-plateau-rigidity}
\leanverified{paper\_conclusion\_multiplicative\_ray\_finite\_plateau\_rigidity}
设
\[
\delta(a):=\frac{P_a}{a}-\alpha_*,
\qquad
H_{\mathrm{loss}}(a,b):=bP_a-P_{ab},
\]
其中 \(a\ge 1\)、\(b\ge 2\)，\(P_a\) 为整数压力，\(\alpha_*\) 为冻结斜率。若对某个 \(N\ge 1\) 有
\[
\delta(a)=\delta(ab^N),
\]
则必有
\[
\boxed{\;
H_{\mathrm{loss}}(ab^k,b)=0
\qquad
(k=0,1,\dots,N-1).\;}
\]
因此，同一乘法射线上的任何有限平台都不是平均抵消现象，而是逐层熵亏损逐项消失的刚性事件。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part65h-pressure-gap-ray-potential} 的有限望远镜展开，
\[
\delta(a)-\delta(ab^N)
=
\sum_{k=0}^{N-1}\frac{H_{\mathrm{loss}}(ab^k,b)}{ab^{k+1}}.
\]
同一定理又给出每一项都非负。若 \(\delta(a)=\delta(ab^N)\)，则右端为有限个非负项之和且总和为零，因此每一项都必须为零，即
\[
H_{\mathrm{loss}}(ab^k,b)=0
\qquad
(k=0,1,\dots,N-1).
\]
证毕。
\end{proof}

\begin{theorem}[乘法射线全冻结的四重等价]\label{thm:conclusion-multiplicative-ray-total-freezing-equivalences}
沿用定理 \ref{thm:conclusion-multiplicative-ray-finite-plateau-rigidity} 的记号。对任意整数 \(a\ge 1\)、\(b\ge 2\)，以下四条等价：
\begin{enumerate}
  \item \(\delta(a)=0\)；
  \item 对所有 \(k\ge 0\)，
  \[
  H_{\mathrm{loss}}(ab^k,b)=0;
  \]
  \item 对所有 \(k\ge 0\)，
  \[
  P_{ab^{k+1}}=b\,P_{ab^k};
  \]
  \item 对所有 \(k\ge 0\)，相应的 \(b\)-escort 熵率
  \[
  h_b^{\mathrm{esc}}(ab^k)=0.
  \]
\end{enumerate}
因此，单点的全冻结并非局部现象，而是沿整条乘法射线传播的严格代数锁定。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part65h-pressure-gap-ray-potential}，
\[
\delta(a)=\sum_{k=0}^{\infty}\frac{H_{\mathrm{loss}}(ab^k,b)}{ab^{k+1}},
\qquad
H_{\mathrm{loss}}(u,b)=(b-1)h_b^{\mathrm{esc}}(u),
\]
且各项均非负。

若 \(\delta(a)=0\)，则上式右侧为非负级数且和为零，故对所有 \(k\ge 0\) 必有
\[
H_{\mathrm{loss}}(ab^k,b)=0.
\]
这给出 \((1)\Rightarrow(2)\)。反向蕴含 \((2)\Rightarrow(1)\) 由同一展开式立即得到。

\((2)\) 与 \((3)\) 只是定义
\[
H_{\mathrm{loss}}(u,b)=bP_u-P_{ub}
\]
的重写；\((2)\) 与 \((4)\) 则由
\[
H_{\mathrm{loss}}(u,b)=(b-1)h_b^{\mathrm{esc}}(u)
\]
直接等价。证毕。
\end{proof}

\begin{theorem}[平衡原子的完成化失明与 finite-part 横坐标]\label{thm:conclusion-realinput40-balanced-atom-completion-finitepart-complementarity}
对真实输入 \(40\) 状态核，设
\[
\Delta(z;u,v)=(1-vz^2)P_8(z;u,v).
\]
则平衡原子因子 \(1-vz^2\) 在归一化完成化切片中完全不可见：完成化后的标准化对象与去除此原子后的 core 完成化对象具有相同的 \(s\)-依赖。另一方面，在 primitive 与 Abel 有限部层却存在精确残差
\[
P_{\mathrm{full}}(z;u,v)-P_{\mathrm{core}}(z;u,v)=vz^2,
\]
\[
\log\mathfrak M_{\mathrm{full}}(u,v)-\log\mathfrak M_{\mathrm{core}}(u,v)
=
v\,z_\star(u,v)^2,
\]
其中 \(z_\star(u,v)\) 为主极点位置。故完成化几何与 finite-part 常数形成严格互补：前者对平衡原子失明，而后者为同一原子提供唯一的横向坐标。
\end{theorem}
\begin{proof}
完成化归一化对 balanced atom 的失明，正是定理 \ref{thm:conclusion-completed-slice-balanced-atoms-become-constants} 与定理 \ref{thm:conclusion-realinput40-completed-slice-atomblind-finitepart-visible} 的直接内容。对同一原子而言，primitive 差额与 Abel 有限部位移由定理 \ref{thm:conclusion-realinput40-uv-primitive-surgery-abel-shift} 给出。将两部分并置，即得所述严格互补性。证毕。
\end{proof}

\begin{theorem}[零温半熵冻结与完成化自对偶都不蕴含 detailed balance]\label{thm:conclusion-realinput40-halfentropy-selfduality-not-detailed-balance}
在真实输入 \(40\) 状态核的零温缩放极限中，设
\[
\widehat F(w)=\det(I-wB_\infty),
\]
其中 \(B_\infty\) 为四状态地基矩阵。则一方面有
\[
h_{\mathrm{ground}}=\frac12\log\rho(B_\infty)=\frac12\,h_{\mathrm{top}}(\Sigma_{B_\infty})>0,
\]
另一方面，相应 Parry 极大熵链并不满足 detailed balance，而带有严格正的稳态环流。因而
\[
\boxed{\;
\text{completed self-duality}
\not\Rightarrow
\text{detailed balance},\qquad
\text{half-entropy freezing}
\not\Rightarrow
\text{detailed balance}. \;}
\]
\end{theorem}
\begin{proof}
半熵恒等式由定理 \ref{thm:conclusion-realinput40-ground-entropy-half-topological-entropy} 给出。另一方面，定理 \ref{thm:conclusion-realinput40-zero-temp-parry-infinite-entropy-production} 已证明对应 Parry 极限严格不可逆，并具有正的定向输运；推论 \ref{cor:conclusion-realinput40-selfdual-completion-not-detailed-balance} 则进一步表明完成化自对偶并不推出 detailed balance。故二者与详细平衡在逻辑上处于正交层。证毕。
\end{proof}

\begin{theorem}[二维压力面的 atom/core 最大值律]\label{thm:conclusion-realinput40-pressure-maxlaw-two-phase}
定义
\[
\rho_{\mathrm{full}}(u,v):=\rho(M_{u,v}),
\qquad
\rho_{\mathrm{core}}(u,v):=\max\{|\lambda|:P_8(\lambda^{-1};u,v)=0\}.
\]
则有精确最大值律
\[
\boxed{\;
\rho_{\mathrm{full}}(u,v)=\max\{\sqrt v,\rho_{\mathrm{core}}(u,v)\}.\;}
\]
因此二维压力
\[
P(u,v):=\log\rho_{\mathrm{full}}(u,v)
\]
被唯一分解为两相：
\[
P(u,v)=
\begin{cases}
\log\rho_{\mathrm{core}}(u,v),& \rho_{\mathrm{core}}(u,v)\ge \sqrt v,\\[4pt]
\frac12\log v,& \rho_{\mathrm{core}}(u,v)\le \sqrt v.
\end{cases}
\]
特别地，在 atom-dominant 区域内，压力关于 \(\log v\) 的斜率恒等于 \(1/2\)，与 core 结构及碰撞参数 \(u\) 无关。
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-realinput40-uv-atomcore-spectral-radius-splitting} 已给出
\[
\rho_{\mathrm{full}}(u,v)=\max\{\sqrt v,\rho_{\mathrm{core}}(u,v)\}.
\]
对数化后立即得到所述分相公式。若 \(\rho_{\mathrm{core}}(u,v)\le\sqrt v\)，则
\[
P(u,v)=\frac12\log v,
\]
故在变量 \(\log v\) 下其斜率恒为 \(1/2\)。证毕。
\end{proof}

\begin{theorem}[残差壳层层析与完成化归一化的严格互补性]\label{thm:conclusion-realinput40-residual-shell-completion-complementarity}
对同一二维系统，按周期长度 \(n\)、碰撞次数 \(c\) 与输出残差 \(r\pmod m\) 组织的 primitive 与 periodic 计数满足
\[
P^{\mathrm{full}}_{m,r}(n;c)-P^{\mathrm{core}}_{m,r}(n;c)
=
\mathbf 1_{\{n=2\}}\mathbf 1_{\{c=0\}}\mathbf 1_{\{r\equiv 1\;(\mathrm{mod}\,m)\}},
\]
\[
A^{\mathrm{full}}_{m,r}(n;c)-A^{\mathrm{core}}_{m,r}(n;c)
=
2\,\mathbf 1_{\{2\mid n\}}\mathbf 1_{\{c=0\}}\mathbf 1_{\{r\equiv n/2\;(\mathrm{mod}\,m)\}}.
\]
于是对任意残差角色 \(\chi:\ZZ/m\ZZ\to\CC^\times\)，都有
\[
\sum_r \chi(r)\bigl(P^{\mathrm{full}}_{m,r}(n;c)-P^{\mathrm{core}}_{m,r}(n;c)\bigr)
=
\chi(1)\mathbf 1_{\{n=2\}}\mathbf 1_{\{c=0\}},
\]
\[
\sum_r \chi(r)\bigl(A^{\mathrm{full}}_{m,r}(n;c)-A^{\mathrm{core}}_{m,r}(n;c)\bigr)
=
2\,\chi(n/2)\mathbf 1_{\{2\mid n\}}\mathbf 1_{\{c=0\}}.
\]
因此，残差 Fourier 层析恰好探测到完成化归一化所完全抹去的平衡原子；二者构成严格互补的一对观测层。
\end{theorem}
\begin{proof}
壳层差分公式正是定理 \ref{thm:conclusion-realinput40-uv-collision-output-delta-shell}。把这些差分分别与任意残差角色 \(\chi\) 配对并对 \(r\) 求和，便只留下所示单点或单壳项。另一方面，定理 \ref{thm:conclusion-realinput40-balanced-atom-completion-finitepart-complementarity} 已表明：归一化完成化对同一平衡原子完全失明。故残差层析与完成化归一化正好形成严格互补的两种观测。证毕。
\end{proof}

\begin{conjecture}[balanced atom 的三接口刚性]\label{conj:conclusion-balanced-atom-three-interface-rigidity}
固定分辨率 \(m\)，并记 \(F:=F_{m+2}\)。设 \((\nu_t)_{t\ge 0}\) 是来自同一有限分辨率 Euler 包范畴的一族解析概率测度，并假定其关联的 completed/primitive/weighted-\(\zeta\) 数据同时满足：
\begin{enumerate}
  \item \textbf{Poisson jet 条件：}对所有 \(k\in G_m\) 与所有 \(0\le n\le F-1\)，都有
  \[
  \frac{\dd^n}{\dd t^n}\widehat\nu_t(k)\Big|_{t=0}
  =
  \bigl(e^{-2\pi ik/F}-1\bigr)^n;
  \]
  \item \textbf{balanced completion 条件：}其标准 normalized completion 与去除 balanced atom 后的 core completion 不可区分；
  \item \textbf{atom-dominant 三签名条件：}其二维加权 \(\zeta\) 数据满足定理 \ref{thm:conclusion-realinput40-uv-atom-dominant-three-signature-synchrony} 的主谱、奇偶迹与 residue 壳同步律。
\end{enumerate}
则应有
\[
\nu_t=\mu_{m,t}
\qquad
(\forall t\ge 0).
\]
也就是说，一旦某个有限分辨率对象同时呈现 Poisson germ、balanced completion blindness 与 atom-dominant parity-shell law，它就不再只是与 canonical Euler 包相似，而应被这三接口刚性唯一锁定。
\end{conjecture}

综上，escort 常数层、乘法射线冻结、balanced atom 的完成化失明、零温地基的半熵不可逆性以及二维 atom/core 相图并不是彼此平行的几条技术支线。它们共同刻画的是同一条分裂机制：一方面，温度轴上的全部 \(O(1)\) 常数损失被 \(\log\varphi\) 精确计量；另一方面，二维加权 \(\zeta\) 在完成化层会主动压平 balanced atom，而 primitive、finite-part 与 residue-shell 三层则把同一原子重新显影为可精确审计的横向位移、壳层质量与相图切换。新增的三接口刚性猜想则进一步表明：在本文框架中，真正的终端对象不应再是单一 completion 证书或单一 residue 证书，而应是由低阶 Poisson germ、balanced completion blindness 与 atom-dominant 三签名共同拼接出的最小完备审计体。
