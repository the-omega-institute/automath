\subsection{Hankel 坏素数分层、hidden-bit 相位商阻尼与 Toeplitz 缺陷的 Nyquist 阈值}\label{subsec:conclusion-hankel-hiddenbit-toeplitz-alias-badprime-splitting}
本小节把三条此前分散的接口并置：uniform-baseline gauge anomaly 的 Hankel 刚性、奇数稳定相极大纤维上的 hidden-bit 统计商，以及有限型 Toeplitz 缺陷的 Hankel--Prony/Gram 证书。其共同点在于：坏素数、相位可见性与负方向采样门槛都不是连续漂移量，而是由有限算术结构精确切出的分层对象。

\begin{theorem}[二阶 Hankel 观测的坏素数分层与纯振幅素数]\label{thm:conclusion-secondorder-hankel-badprime-amplitude-splitting}
记
\[
B_m^{\mathrm{int}}:=2^{m+1}\overline G_m,
\qquad
V_m:=2^{2m+2}\operatorname{Var}(G_m).
\]
则它们的坏素数支撑满足严格分裂
\[
\mathfrak P_{\mathrm{mean}}=\{2,5\},
\qquad
\mathfrak P_{\mathrm{var}}=\{2,3,5,59\},
\]
并且
\[
\mathfrak P_{\mathrm{var}}
=
\mathfrak P_{\mathrm{mean}}
\sqcup
\{3,59\}.
\]
更强地，素数 \(59\) 是纯振幅坏素数：它不来自方差序列特征根
\[
\{4,2,1,-1,-2\}
\]
之间的任何差值，而仅来自 Hankel 主子式振幅因子的算术内容。
\end{theorem}

\begin{proof}
定理 \ref{thm:fold-gauge-anomaly-mean-hankel-rigidity-rank5-badprimes} 已证明 \(B_m^{\mathrm{int}}\) 的坏素数集恰为 \(\{2,5\}\)；定理 \ref{thm:fold-gauge-anomaly-variance-hankel-rigidity-rank13-prime59} 已证明 \(V_m\) 的坏素数集恰为 \(\{2,3,5,59\}\)。再由推论 \ref{cor:fold-gauge-anomaly-spectral-vs-amplitude-primes}，素数 \(59\) 不在谱节点差的素因子集合中，而只出现于振幅因子，故其纯振幅性成立。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$p$}{p}-进刚性沿偏移方向的仿射输运]\label{thm:conclusion-padic-hankel-stiffness-affine-transport}
设 \((a_n)_{n\ge 0}\subset \ZZ\) 为有限 Hankel 秩序列，取一个可逆见证块
\[
H_{k_0}\in M_d(\ZZ),
\qquad
\det(H_{k_0})\neq 0,
\]
并定义偏移传递矩阵
\[
A:=H_{k_0}^{-1}H_{k_0+1}.
\]
对任意素数 \(p\)，记
\[
\operatorname{Stiff}_p(k):=v_p(\det H_k).
\]
则对每个 \(r\ge 0\) 都有严格仿射律
\[
\operatorname{Stiff}_p(k_0+r)
=
\operatorname{Stiff}_p(k_0)+r\,v_p(\det A).
\]
因此，\(p\)-进坏化沿分辨率偏移并非任意漂移，而是一条斜率唯一等于 \(v_p(\det A)\) 的算术测地线。
\leanverified{paper\_conclusion\_padic\_hankel\_stiffness\_affine\_transport}
\end{theorem}

\begin{proof}
把定理 \ref{thm:conclusion-hankel-shift-padic-precision-slope} 应用于 \(K=\QQ_p\) 与均匀元 \(\pi=p\) 即得
\[
v_p(\det H_{k_0+r})
=
v_p(\det H_{k_0})+r\,v_p(\det A).
\]
这正是所述公式。证毕。
\end{proof}

\begin{corollary}[良素数的永久稳定性]\label{cor:conclusion-goodprime-permanent-hankel-stability}
沿用定理 \ref{thm:conclusion-padic-hankel-stiffness-affine-transport} 的记号。若某素数 \(p\) 满足
\[
\operatorname{Stiff}_p(k_0)=0,
\qquad
v_p(\det A)=0,
\]
则对所有 \(r\ge 0\) 都有
\[
\operatorname{Stiff}_p(k_0+r)=0.
\]
因而全部后续见证块 \(H_{k_0+r}\) 在 \(\mathbf F_p\) 上保持可逆，并且模 \(p\) 的 Hankel 未来轨道始终处于有限模单值化状态。
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-padic-hankel-stiffness-affine-transport}，
\[
\operatorname{Stiff}_p(k_0+r)=0
\qquad (r\ge 0).
\]
故每个 \(H_{k_0+r}\) 模 \(p\) 约化后都可逆。再由定理 \ref{thm:conclusion-hankel-finite-mod-singlevaluedization-shortest-locking-threshold}，良模数层上的未来 Hankel 轨道由相邻见证块唯一锁定，并保持纯周期传播。证毕。
\end{proof}

\begin{theorem}[奇数稳定相 hidden-bit 观测的精确三类商]\label{thm:conclusion-hiddenbit-triquotient-balanced-biased-splitting}
设 \(m=2k+1\ge 13\)，并记
\[
M_m:=\{x\in X_m:d_m(x)=D_m\}.
\]
以 hidden-bit 样本 \(B_1^n\) 所诱导的统计等价关系
\[
x\sim_{\mathrm{stat}}y
\iff
\operatorname{Law}(B_1^n\mid X=x)=\operatorname{Law}(B_1^n\mid X=y)
\quad(\forall n\ge 1)
\]
定义极大相位的可识别商。则
\[
M_m/\!\sim_{\mathrm{stat}}
\cong
\{\mathsf{bal},\mathsf{bias}_+,\mathsf{bias}_-\},
\]
其中
\[
\mathsf{bal}
\quad\text{对应}\quad
\operatorname{Bern}\!\left(\frac12\right)^{\otimes n},
\]
而另外两类的单样本参数分别为
\[
p_k^{\pm}
=
\frac12\pm\frac{F_{k-2}}{2F_{k+1}}.
\]
特别地，奇数稳定相中的 hidden-bit 可见性并不是对四个极大相位作模糊区分，而是先精确塌缩为一个平衡双点类与一对互补偏置类。
\end{theorem}

\begin{proof}
这正是定理 \ref{thm:conclusion-maxfiber-hiddenbit-quotient-even2-odd3} 在奇数稳定相上的内容。证毕。
\end{proof}

\begin{corollary}[偏置二面体的黄金极限与互信息阻尼]\label{cor:conclusion-hiddenbit-biased-face-golden-damping}
沿用定理 \ref{thm:conclusion-hiddenbit-triquotient-balanced-biased-splitting} 的记号。则
\[
p_k^+\to \varphi^{-1},
\qquad
p_k^-\to \varphi^{-2}
\qquad (k\to\infty),
\]
并且收敛速度为 \(O(\varphi^{-2k})\)。因此，奇数稳定相 hidden-bit 三类商的非平凡二元面在极限上收敛到黄金互补二元信道 \(W\)。其极限收缩常数满足
\[
\delta(W)=\rho_{\mathrm m}(W)=\varphi^{-3},
\qquad
\eta_I(W)=\varphi^{-6}.
\]
另一方面，平衡双点类在任意样本长度上仍保持不可学习：
\[
\inf_{\delta_n}\mathcal R_n=\frac12.
\]
故 hidden-bit 观测的刚性分裂是：一条精确三类商，加上一条收敛到 \(\varphi^{-6}\) 信息阻尼的非平凡二元面。
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-odd-maxfiber-hiddenbit-tristate-crystal}，
\[
\frac12+\frac{F_{k-2}}{2F_{k+1}}
=
\varphi^{-1}+O(\varphi^{-2k}),
\qquad
\frac12-\frac{F_{k-2}}{2F_{k+1}}
=
\varphi^{-2}+O(\varphi^{-2k}),
\]
故 \((p_k^+,p_k^-)\) 收敛到黄金互补偏置 \((\varphi^{-1},\varphi^{-2})\)。再由定理 \ref{thm:conclusion-golden-sprt-delta-closure}，对应极限二元信道的 TV/相关收缩与互信息 SDPI 常数分别为 \(\varphi^{-3}\) 与 \(\varphi^{-6}\)。平衡双点类上的 Bayes 墙则由推论 \ref{cor:conclusion-maxfiber-hiddenbit-balanced-bayes-wall} 给出。证毕。
\end{proof}

\begin{theorem}[Toeplitz 缺陷判定的精确 Nyquist 阈值]\label{thm:conclusion-toeplitz-defect-exact-nyquist-threshold}
在有限缺陷且无奇异内因子的口径下，设 Toeplitz 负惯性在稳定区间内最终稳定为 \(\kappa\)。记对应扫描样本为 \((u_n)_{n\ge 0}\)。则成立两条严格最小性结论：
\begin{enumerate}
\normalfont
  \item 判定阈值：前缀
  \[
  (u_0,\dots,u_{2\kappa-2})
  \]
  已唯一决定 Hankel 指纹块 \(H_{\kappa-1}\)，从而唯一决定负 Gram 图表
  \[
  -\,H_{\kappa-1}^{*}H_{\kappa-1}
  \]
  以及稳定区间中的负惯性 \(\kappa\)；该长度 \(2\kappa-1\) 在一般位置下是严格最小的。
  \item 重建阈值：若要求把缺陷节点与权重完整恢复，则一般还需再加一项样本，即完整窗口
  \[
  (u_0,\dots,u_{2\kappa-1})
  \]
  的 \(2\kappa\) 点数据；该阈值同样是严格最小的。
\end{enumerate}
因此，Toeplitz 缺陷存在一条精确的 Nyquist 型门槛：检测 \(\kappa\) 个负方向恰需 \(2\kappa-1\) 个样本，而完整重建恰需 \(2\kappa\) 个样本。
\leanverified{paper\_conclusion\_toeplitz\_defect\_exact\_nyquist\_threshold}
\end{theorem}

\begin{proof}
由定理 \ref{thm:typed-address-biaxial-completion-comoving-prony-threshold}，长度 \(2\kappa-1\) 的样本前缀正是判定 \(\operatorname{rank}(H_{\kappa-1})=\kappa\) 的最短门槛，而完整参数重建的一般位置阈值正是 \(2\kappa\) 个样本。另一方面，定理 \ref{thm:conclusion-visible-toeplitz-negative-gram-chart} 给出 Toeplitz 负方向的 exact Gram 图表，而定理 \ref{thm:xi-toeplitz-inertia-stabilizes-to-kappa} 说明稳定区间中的负惯性恰等于 \(\kappa\)。三者合并即得所述判定阈值与重建阈值分离。证毕。
\end{proof}

\begin{corollary}[有限阶相位闭合的 Hankel 别名解释]\label{cor:conclusion-charp-hankel-alias-interpretation}
沿用定理 \ref{thm:conclusion-charp-hankel-rank-classification} 的记号，设
\[
a_n=L_n^{\,q},
\qquad
s:=\beta/\alpha,
\qquad
m:=\operatorname{ord}(s).
\]
若 \(m\le q\)，则有精确重写
\[
a_n
=
\sum_{a=0}^{m-1}C_a(\alpha^q s^a)^n,
\qquad
C_a=
\sum_{\substack{0\le k\le q\\ k\equiv a\;(\mathrm{mod}\,m)}}\binom{q}{k},
\]
并且
\[
\operatorname{rank}H(a)
=
\#\{a\in\{0,\dots,m-1\}:C_a\neq 0\}.
\]
因此，Hankel 秩从 \(q+1\) 向下坍缩并不是系数偶然抵消，而是 \(q+1\) 个指数节点在有限阶相位闭合下按同余类发生别名合并。
\end{corollary}

\begin{proof}
这正是定理 \ref{thm:conclusion-charp-hankel-rank-classification} 在有限阶情形 \(m\le q\) 的重写。证毕。
\end{proof}

\begin{conjecture}[坏素数的分母--别名--振幅三分]\label{conj:conclusion-hankel-badprime-denominator-alias-amplitude-splitting}
对由整数 Hankel 主子式导出的坏素数集 \(\mathfrak P_{\mathrm{bad}}\)，应存在一个规范分解
\[
\mathfrak P_{\mathrm{bad}}
=
\mathfrak P_{\mathrm{den}}
\cup
\mathfrak P_{\mathrm{alias}}
\cup
\mathfrak P_{\mathrm{amp}},
\]
其中
\begin{enumerate}
\normalfont
  \item \(\mathfrak P_{\mathrm{den}}\) 由最小 realization 的分母与 \(p\)-整性障碍控制；
  \item \(\mathfrak P_{\mathrm{alias}}\) 由定理 \ref{thm:conclusion-charp-hankel-rank-classification} 所描述的有限阶相位闭合与节点别名控制；
  \item \(\mathfrak P_{\mathrm{amp}}\) 由不改变谱节点集合的纯振幅因子控制。
\end{enumerate}
并且这三类障碍应当彼此可分。低阶证据已由定理 \ref{thm:conclusion-secondorder-hankel-badprime-amplitude-splitting} 与推论 \ref{cor:conclusion-charp-hankel-alias-interpretation} 给出：\(\{2,5\}\) 已在均值层形成可审计坏素数集，\(59\) 已被严格证明为纯振幅坏素数，而有限阶节点闭合则提供了一条与振幅机制独立的 Hankel 秩坍缩通道。
\end{conjecture}
