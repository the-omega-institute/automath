\begin{theorem}[纤维后验的 hard-core 精确化与退火自由能守恒]\label{thm:conclusion-fold-fiber-hardcore-annealed-conservation}
\leanverified{paper\_conclusion\_fold\_fiber\_hardcore\_annealed\_conservation}
设输入分布为
\[
\mu_{m,p}:=\mathrm{Bernoulli}(p)^{\otimes m},
\qquad
t:=\frac{p}{1-p},
\]
并记
\[
\nu_{m,p}:=(\Fold_m)_\#\mu_{m,p}.
\]
对每个 \(x\in X_m\)，记
\[
F_x:=\Fold_m^{-1}(x),\qquad
\Theta_x:F_x\to \mathsf{Ind}(\Gamma_x),
\qquad
S(\omega):=\Theta_x(\omega),
\]
并令 \(Z_x(t)\) 为对应纤维生成函数。则：
\begin{enumerate}
  \item 推前分布满足显式闭式
  \[
  \nu_{m,p}(x)=p^{|x|_1}(1-p)^{m-|x|_1}Z_x(t);
  \]
  \item 纤维后验满足显式闭式
  \[
  \mu_{m,p}(\omega\mid x)=\frac{t^{|S(\omega)|}}{Z_x(t)},
  \]
  从而在正文给出的规范同构下，它恰是一族活动度为 \(t\) 的 hard-core 模型；
  \item 全部纤维满足精确守恒律
  \[
  \boxed{\;
  \sum_{x\in X_m} t^{|x|_1}Z_x(t)=(1+t)^m.\;}
  \]
\end{enumerate}
因此，折叠不会改变总体退火自由能；全部非平凡复杂性都被推入纤维内部的淬火 partition ratios。
\end{theorem}
\begin{proof}
由正文关于 Bernoulli 输入后验的显式公式，对任意 \(\omega\in\Fold_m^{-1}(x)\) 都有
\[
|\omega|_1=|x|_1+r(\omega)=|x|_1+|S(\omega)|,
\]
故
\[
\mu_{m,p}(\omega)
=
p^{|\omega|_1}(1-p)^{m-|\omega|_1}
=
p^{|x|_1}(1-p)^{m-|x|_1}
\left(\frac{p}{1-p}\right)^{|S(\omega)|}
=
t^{|S(\omega)|}p^{|x|_1}(1-p)^{m-|x|_1}.
\]
在纤维 \(F_x\) 上求和即得
\[
\nu_{m,p}(x)=p^{|x|_1}(1-p)^{m-|x|_1}\sum_{\omega\in F_x}t^{|S(\omega)|}
=p^{|x|_1}(1-p)^{m-|x|_1}Z_x(t).
\]
再以该边缘分布归一化即得
\[
\mu_{m,p}(\omega\mid x)=\frac{t^{|S(\omega)|}}{Z_x(t)}.
\]
在正文的规范同构下，这正是活动度为 \(t\) 的 hard-core 后验。

另一方面，对全体纤维求和有
\[
\sum_{x\in X_m} t^{|x|_1}Z_x(t)
=
\sum_{x\in X_m}\sum_{\omega\in \Fold_m^{-1}(x)} t^{|x|_1+r(\omega)}
=
\sum_{\omega\in\Omega_m} t^{|\omega|_1}.
\]
最后一个和式正是独立二项分布的配分函数，
\[
\sum_{\omega\in\Omega_m} t^{|\omega|_1}=(1+t)^m.
\]
证毕。
\end{proof}

\begin{theorem}[hard-core 纤维分解的精确二项恢复]\label{thm:conclusion-fold-fiber-hardcore-exact-binomial-recovery}
\leanverified{paper\_conclusion\_fold\_fiber\_hardcore\_exact\_binomial\_recovery}
取 \(p\in(0,1)\)，记
\[
t:=\frac{p}{1-p},
\qquad
X\sim \nu_{m,p}=(\Fold_m)_\#\mathrm{Bernoulli}(p)^{\otimes m}.
\]
在条件 \(X=x\) 下，于 \(\mathsf{Ind}(\Gamma_x)\) 上取活动度为 \(t\) 的 hard-core 测度，并令
\[
R:=|S|,
\qquad
S\sim P_{X,p}.
\]
再定义总占据量
\[
N:=|X|_1+R.
\]
则严格有
\[
N\sim \mathrm{Binomial}(m,p).
\]
等价地，其概率母函数满足
\[
\mathbb E[s^N]=(1-p+ps)^m.
\]
因此，折叠后的两阶段结构
\[
\text{pushforward }X \quad+\quad \text{fiber hard-core excitation }R
\]
在总占据量上无损恢复原始 Bernoulli 独立和。
\end{theorem}

\begin{proof}
在条件 \(X=x\) 下，\(R\) 的概率母函数为
\[
\mathbb E[s^R\mid X=x]=\frac{Z_x(ts)}{Z_x(t)}.
\]
故
\[
\mathbb E[s^N]
=\sum_{x\in X_m}\nu_{m,p}(x)s^{|x|_1}\frac{Z_x(ts)}{Z_x(t)}.
\]
由定理 \ref{thm:conclusion-fold-fiber-hardcore-annealed-conservation}，
\[
\nu_{m,p}(x)=p^{|x|_1}(1-p)^{m-|x|_1}Z_x(t),
\]
于是
\[
\mathbb E[s^N]
=
\sum_{x\in X_m}p^{|x|_1}(1-p)^{m-|x|_1}s^{|x|_1}Z_x(ts)
=(1-p)^m\sum_{x\in X_m}(ts)^{|x|_1}Z_x(ts).
\]
再用同一定理中的守恒律
\[
\sum_{x\in X_m}u^{|x|_1}Z_x(u)=(1+u)^m
\]
并取 \(u=ts\)，便得
\[
\mathbb E[s^N]=(1-p)^m(1+ts)^m=(1-p+ps)^m.
\]
这正是 \(\mathrm{Binomial}(m,p)\) 的母函数。证毕。
\end{proof}

\begin{theorem}[Bernoulli 波动预算的纤维内/纤维间精确分解]\label{thm:conclusion-fold-fiber-hardcore-variance-budget-splitting}
\leanverified{paper\_conclusion\_fold\_fiber\_hardcore\_variance\_budget\_splitting}
沿用定理 \ref{thm:conclusion-fold-fiber-hardcore-exact-binomial-recovery} 的记号，记
\[
m_{x,p}:=\mathbb E_{P_{x,p}}[R],
\qquad
\sigma^2_{x,p}:=\Var_{P_{x,p}}(R).
\]
则有严格方差预算恒等式
\[
mp(1-p)
=
\mathbb E_{X\sim \nu_{m,p}}\bigl[\sigma^2_{X,p}\bigr]
\Var_{X\sim \nu_{m,p}}\bigl(|X|_1+m_{X,p}\bigr).
\]
亦即，原始独立 Bernoulli 和的全部二阶波动，恰被分解为纤维内部 hard-core 激发的热涨落与折叠层之间的几何漂移涨落，两者之外不存在第三项残差。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-fold-fiber-hardcore-exact-binomial-recovery}，随机变量
\[
N:=|X|_1+R
\]
服从 \(\mathrm{Binomial}(m,p)\)，故
\[
\Var(N)=mp(1-p).
\]
对 \(N\) 相对于 \(\sigma(X)\) 应用全方差公式，
\[
\Var(N)=\mathbb E[\Var(N\mid X)]+\Var(\mathbb E[N\mid X]).
\]
而在条件 \(X=x\) 下，\(|X|_1\) 为常数，因此
\[
\Var(N\mid X=x)=\Var(R\mid X=x)=\sigma^2_{x,p},
\]
并且
\[
\mathbb E[N\mid X=x]=|x|_1+\mathbb E[R\mid X=x]=|x|_1+m_{x,p}.
\]
代回即得所示恒等式。证毕。
\end{proof}

\begin{theorem}[全阶累积量的普适传输]\label{thm:conclusion-fold-fiber-hardcore-universal-cumulant-transport}
沿用定理 \ref{thm:conclusion-fold-fiber-hardcore-exact-binomial-recovery} 的记号。对一切 \(n\ge 1\)，都有
\[
\kappa_n(|X|_1+R)=m\,\kappa_n(\mathrm{Bernoulli}(p)).
\]
特别地，
\[
\kappa_1=mp,\qquad
\kappa_2=mp(1-p),\qquad
\kappa_3=mp(1-p)(1-2p),
\]
\[
\kappa_4=mp(1-p)(1-6p+6p^2).
\]
因此，折叠—纤维—hard-core 三层结构虽然微观上高度非平凡，但其退火总占据量的整个累积量塔完全退化为一条普适的 binomial cumulant line。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-fold-fiber-hardcore-exact-binomial-recovery}，
\[
\mathbb E[e^{\theta(|X|_1+R)}]=(1-p+pe^\theta)^m.
\]
故其累积母函数为
\[
K(\theta):=\log\mathbb E[e^{\theta(|X|_1+R)}]
=m\log(1-p+pe^\theta).
\]
右端正是 \(m\) 个独立 \(\mathrm{Bernoulli}(p)\) 变量之和的累积母函数。对 \(\theta\) 在 \(0\) 处逐阶求导，即得全部累积量公式。证毕。
\end{proof}
