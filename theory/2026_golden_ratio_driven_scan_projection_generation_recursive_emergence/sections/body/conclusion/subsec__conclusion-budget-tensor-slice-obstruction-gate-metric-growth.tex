\begin{proposition}[外置预算的张量乘法律]\label{prop:conclusion-external-budget-tensor-multiplicativity}
\leanverified{paper\_conclusion\_external\_budget\_tensor\_multiplicativity}
设 \(D_i,Y_i\) 为可数集，\(\pi_i:D_i\to Y_i\) 为全定义可计算映射，且
\[
D(\pi_i):=\sup_{y\in Y_i}\#\pi_i^{-1}(y)<\infty\qquad(i=1,2).
\]
定义直积映射
\[
\pi_1\times\pi_2:\ D_1\times D_2\to Y_1\times Y_2,\qquad
(x_1,x_2)\mapsto(\pi_1(x_1),\pi_2(x_2)).
\]
则
\[
b(\pi_1\times\pi_2)=b(\pi_1)\,b(\pi_2).
\]
\end{proposition}
\begin{proof}
对任意 \((y_1,y_2)\in Y_1\times Y_2\)，有
\[
(\pi_1\times\pi_2)^{-1}(y_1,y_2)
=\pi_1^{-1}(y_1)\times\pi_2^{-1}(y_2),
\]
故
\[
\#(\pi_1\times\pi_2)^{-1}(y_1,y_2)
=\#\pi_1^{-1}(y_1)\cdot\#\pi_2^{-1}(y_2).
\]
取上确界得
\[
D(\pi_1\times\pi_2)=D(\pi_1)D(\pi_2).
\]
由定理 \ref{thm:conclusion-minimal-external-budget-undecidable} 的恒等式 \(b(\pi)=D(\pi)\)（有限纤维上界类）即得
\[
b(\pi_1\times\pi_2)=b(\pi_1)b(\pi_2).
\]
证毕。
\end{proof}

\begin{theorem}[二进制可逆化宽度不存在统一加性常数近似]\label{thm:conclusion-reversible-width-no-uniform-additive-approx}
\leanpartial{paper\_conclusion\_reversible\_width\_no\_uniform\_additive\_approx}{The stated proposition is false as written, since taking \texttt{A := reversibleWidth} and \texttt{c := 0} already gives a witness.}
对满足 \(D(\pi)<\infty\) 的全定义可计算映射，定义
\[
r(\pi):=\left\lceil\log_2 b(\pi)\right\rceil.
\]
则对任意常数 \(c\in\NN\)，不存在可计算函数 \(A\) 使
\[
|A(\pi)-r(\pi)|\le c\qquad(\forall\,\pi).
\]
\end{theorem}
\begin{proof}
反设存在此类 \(A\)。
由定理 \ref{thm:conclusion-minimal-external-budget-undecidable}，存在可计算族 \(\pi_{M,x}\) 满足
\[
b(\pi_{M,x})=
\begin{cases}
1,&M(x)\ \text{不停机},\\
2,&M(x)\ \text{停机}.
\end{cases}
\]
对任意 \(m\ge 1\) 取 \(m\)-重直积
\[
\Pi_{M,x}^{(m)}:=\pi_{M,x}^{\times m}.
\]
由命题 \ref{prop:conclusion-external-budget-tensor-multiplicativity}，
\[
b(\Pi_{M,x}^{(m)})=b(\pi_{M,x})^m\in\{1,2^m\},
\]
故
\[
r(\Pi_{M,x}^{(m)})\in\{0,m\},
\]
且 \(r(\Pi_{M,x}^{(m)})=m\) 当且仅当 \(M(x)\) 停机。

取 \(m>2c\)。由 \(|A-r|\le c\)：
\[
r=0\ \Longrightarrow\ A\le c,\qquad
r=m\ \Longrightarrow\ A\ge m-c>c.
\]
因此以阈值 \(c\) 比较 \(A(\Pi_{M,x}^{(m)})\) 可判定 \(M(x)\) 是否停机，矛盾。
故不存在统一加性常数近似。证毕。
\end{proof}

\begin{theorem}[有界切片可计算最优与全域最优综合器不可延拓]\label{thm:conclusion-slice-computable-global-optimal-synthesizer-impossible}
\leanverified{paper\_conclusion\_slice\_computable\_global\_optimal\_synthesizer\_impossible}

\begin{theorem}[同步逻辑门模型的最小触发器数与强不可计算性]\label{thm:conclusion-minimal-flipflop-count-strong-uncomputability}
\leanverified{paper\_conclusion\_minimal\_flipflop\_count\_strong\_uncomputability}
在同步 NAND 门系模型中，设 \(\tau\) 为确定性流式转导，记最小状态数为 \(s(\tau)\)，并定义最小触发器数
\[
D^\ast(\tau):=\min\{D:\ \exists\ \text{含 }D\text{ 个触发器的同步 NAND 电路实现 }\tau\}.
\]
则
\[
D^\ast(\tau)=\left\lceil\log_2 s(\tau)\right\rceil.
\]
并且存在可计算族 \(\tau_{M,x}\)，使得 \(D^\ast(\tau_{M,x})\) 在任意统一加性常数误差意义下均不可计算近似。
\end{theorem}
\begin{proof}
若某实现含 \(D\) 个触发器，则内部状态数至多 \(2^D\)，故 \(s(\tau)\le 2^D\) 并得
\[
D^\ast(\tau)\ge \left\lceil\log_2 s(\tau)\right\rceil.
\]
反向地，任取实现 \(\tau\) 的最小确定性转导器（\(s(\tau)\) 个状态），用 \(\lceil\log_2 s(\tau)\rceil\) 位编码状态并以组合逻辑实现转移/输出，即得
\[
D^\ast(\tau)\le \left\lceil\log_2 s(\tau)\right\rceil.
\]
二者合并得到恒等式。

再由定理 \ref{thm:conclusion-minimal-state-complexity-halting}，存在可计算族 \(\pi_{M,x}\)（每纤维至多 2）满足
\[
s(\pi_{M,x})=
\begin{cases}
T(M,x)+1,&T(M,x)<\infty,\\
O(1),&T(M,x)=\infty.
\end{cases}
\]
令 \(\tau_{M,x}:=\pi_{M,x}\)。对任意 \(N\) 构造 \(M^{(N)}\)：先空转 \(N\) 步再模拟 \(M\)。
则
\[
T(M,x)<\infty\ \Longrightarrow\ D^\ast(\tau_{M^{(N)},x})\ge \left\lceil\log_2(N+1)\right\rceil,
\]
而
\[
T(M,x)=\infty\ \Longrightarrow\ D^\ast(\tau_{M^{(N)},x})=O(1)
\]
（上界与 \(N\) 无关）。
给定任意加性常数 \(c\)，取 \(N\) 足够大使两类数值间隙 \(>2c\)，即可由 \(c\)-近似值区分停机与不停机，矛盾。
故不存在统一加性常数近似算法。证毕。
\end{proof}

\begin{theorem}[固定温度软序门加入后的概率有限态性与分布等价可判定性]\label{thm:conclusion-fixed-temp-soft-order-probabilistic-finite-state-decidable}

  \item 若所有门参数与转移概率属于可判定数域（例如代数数域 \(\overline{\QQ}\)），则两电路的分布语义等价是可判定问题。
\end{enumerate}
因此固定温度随机化不会跨越“语义图灵完备”的可判定性边界；跨越边界的结构地位仍由硬序门 \(P_{\le}\)（或等价的零温极限）所占据。
\end{theorem}

在串联、并联与扇出下，内部状态空间仅作有限直积并作有限商，因而仍为有限状态随机系统。
\par
对 (2)，对任一有限状态概率转导器，可为每个输入符号 \(a\) 写出对应的随机转移矩阵 \(P_a\)，并以线性泛函 \(L\) 表示所关心的输出分布坐标；对任意输入字 \(w=a_1\cdots a_m\)，输出分布坐标可写为
\[
L\bigl(\pi_0 P_{a_1}\cdots P_{a_m}\bigr),
\]
其中 \(\pi_0\) 为初始分布行向量。

当矩阵元素属于可判定数域时，有限个线性表达式的零判定可由有限算法完成，故分布等价可判定。
\par
最后一段结论由：若固定温度片段已语义图灵完备，则语义等价应不可判定；与 (2) 矛盾。
与本文中硬 first-fit 依赖 \(P_{\le}\) 并引入不可并行增长的结构性对照一致。证毕。
\end{proof}

\begin{theorem}[有限检验深度的 Busy--Beaver 下界]\label{thm:conclusion-finite-test-depth-busybeaver}
\leanverified{paper\_conclusion\_finite\_test\_depth\_busybeaver}
固定一种有效编码，使每个程序 \(p\) 计算一个全定义函数 \(f_p:\NN\to\{0,1\}\)，并以 \(|p|\) 表示其描述长度。
定义两程序的首个分歧位置
\[
\Delta(p,q):=\min\{n\in\NN:\ f_p(n)\neq f_q(n)\},
\]
若 \(f_p\equiv f_q\) 则令 \(\Delta(p,q)=\infty\)。
\par
若存在函数 \(\Phi:\NN\to\NN\) 使对所有 \(|p|,|q|\le L\) 成立
\[
f_p\not\equiv f_q\quad\Longrightarrow\quad \Delta(p,q)\le \Phi(L),
\]
则存在常数 \(c\)（仅依赖于编码）使对所有 \(L\) 有
\[
\Phi(L)\ \ge\ \mathrm{BB}(L-c),
\]
其中 \(\mathrm{BB}(m)\) 为 Busy--Beaver 函数（长度/状态数 \(\le m\) 的停机机在其停机输入上的最大停机时间）。
特别地，\(\Phi\) 不能为可计算函数。
\end{theorem}
\begin{proof}
对任意图灵机 \(M\) 与输入 \(x\)，定义全定义函数
\[
f_{M,x}(n):=
\begin{cases}
0,& M(x)\ \text{在 }n\text{ 步内停机},\\
1,& \text{否则},
\end{cases}
\qquad n\in\NN.
\]
令 \(g(n)\equiv 1\) 为常值函数，并取一个定长程序 \(q\) 计算 \(g\)。
若 \(M(x)\) 于时刻 \(T\) 停机，则对所有 \(n<T\) 有 \(f_{M,x}(n)=1\) 且 \(f_{M,x}(T)=0\neq 1=g(T)\)，从而 \(\Delta(p_{M,x},q)=T\)，其中 \(p_{M,x}\) 为计算 \(f_{M,x}\) 的程序。
并且存在常数 \(c\) 使得 \(|p_{M,x}|\le |(M,x)|+c\) 且 \(|q|\le c\)。
\par
若题设 \(\Phi\) 存在，则对任意停机的 \((M,x)\) 有
\[
T(M,x)=\Delta(p_{M,x},q)\ \le\ \Phi(|(M,x)|+c).
\]
对所有 \(|(M,x)|\le L-c\) 的停机实例取停机时间上确界即得 \(\mathrm{BB}(L-c)\le \Phi(L)\)。
由于 \(\mathrm{BB}\) 超越任意可计算函数，故 \(\Phi\) 不可计算。证毕。
\end{proof}

\begin{conjecture}[审计预算主尺度的 Busy--Beaver 同阶律]\label{conj:conclusion-audit-budget-busy-beaver-scale}
沿定理 \ref{thm:conclusion-audit-budget-escapes-all-computable-growth} 的记号，记
\[
B_{\max}(n):=\sup\{\beta(p):\ p\in\mathcal F,\ |p|\le n\}.
\]
猜想存在可计算单调函数 \(u,v\) 使对充分大 \(n\) 有
\[
u(\mathrm{BB}(n))\ \le\ B_{\max}(n)\ \le\ v(\mathrm{BB}(n)).
\]
该猜想断言：把图灵完备语义压入可审计有限门系切片所需的最小预算，其主增长尺度与 Busy--Beaver 级别一致。
\end{conjecture}
