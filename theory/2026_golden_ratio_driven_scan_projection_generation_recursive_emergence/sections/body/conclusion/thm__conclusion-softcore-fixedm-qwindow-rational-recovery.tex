\begin{theorem}[固定 \(m\) 的 \(q\)-向碰撞序列之有理母函数]\label{thm:conclusion-softcore-fixedm-qseries-rational-ogf}
\leanverified{paper\_conclusion\_softcore\_fixedm\_qseries\_rational\_ogf}
固定 \(m\ge 1\)。记
\[
\mathcal T_m^{\mathrm{dist}}=\{\tau_1,\dots,\tau_r\}\subset \ZZ_{>0}
\]
为命题 \ref{prop:conclusion-softcore-fixedm-qrecurrence-annihilator} 中的去重词迹集合，并记
\[
L_m:=\Tr(K^m),
\qquad
\omega_{m,\pm}:=\frac{L_m\pm\sqrt{L_m^2-4(-1)^m}}{2}
=\varphi^m,\ (-1)^m\varphi^{-m}.
\]
定义 ordinary generating function
\[
\mathcal S_m(z):=\sum_{q\ge 0}S_m(q)\,z^q.
\]
则 \(\mathcal S_m(z)\) 为有理函数，并可写成
\[
\mathcal S_m(z)
=
\frac{R_m(z)}
{(1-\omega_{m,+}z)(1-\omega_{m,-}z)\prod_{j=1}^{r}(1-\tau_j z)},
\]
其中 \(R_m(z)\in\QQ[z]\) 且
\[
\deg R_m\le r+1.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:conclusion-softcore-fixedm-qrecurrence-annihilator}，序列 \(q\mapsto S_m(q)\) 满足常系数线性递推，其一条湮灭多项式为
\[
\mathcal P_m(x)=
\bigl(x^2-L_mx+(-1)^m\bigr)\prod_{j=1}^{r}(x-\tau_j).
\]
因此 \(\mathcal S_m(z)\) 必为有理函数，其分母可取为
\[
z^{r+2}\mathcal P_m(1/z)
=
(1-\omega_{m,+}z)(1-\omega_{m,-}z)\prod_{j=1}^{r}(1-\tau_j z).
\]
由于递推阶数至多为 \(r+2\)，标准 C-finite 生成函数理论给出分子次数至多比分母低一阶，即 \(\deg R_m\le r+1\)。证毕。
\end{proof}

\begin{theorem}[generic 情形下的最小递推阶数与极点最小性]\label{thm:conclusion-softcore-fixedm-qrecurrence-generic-minimality}
\leanverified{paper\_conclusion\_softcore\_fixedm\_qrecurrence\_generic\_minimality}
在定理 \ref{thm:conclusion-softcore-fixedm-qseries-rational-ogf} 的记号下，若
\[
\{\tau_1,\dots,\tau_r,\omega_{m,+},\omega_{m,-}\}
\]
两两不同，则：
\begin{enumerate}
  \item 定理 \ref{thm:conclusion-softcore-fixedm-qseries-rational-ogf} 中给出的分母已是 \(\mathcal S_m(z)\) 的最小分母；
  \item 序列 \(q\mapsto S_m(q)\) 的最小递推阶数恰为
  \[
  r+2;
  \]
  \item 任意长度至少 \(2(r+2)\) 的连续初值段都已唯一决定整个序列。
\end{enumerate}
\end{theorem}
\begin{proof}
由命题 \ref{prop:conclusion-softcore-fixedm-qrecurrence-annihilator} 的证明，
\[
S_m(q)
=
2^{-m}\left(
\Omega_m(q)+\sum_{j=1}^{r}\mu_m(\tau_j)\tau_j^q
\right),
\]
其中每个 \(\mu_m(\tau_j)\in\ZZ_{>0}\)，而 \(\Omega_m(q)\) 是 \(\omega_{m,+}^q\) 与 \(\omega_{m,-}^q\) 的非平凡线性组合。于是 \(S_m(q)\) 是 \(r+2\) 个互异几何序列的非平凡线性组合。

在特征零域上，互异指数列 \(\{\lambda^q\}_{q\ge0}\) 线性无关，因此任一湮灭 \(S_m(q)\) 的最小多项式必须包含全部根
\[
\omega_{m,+},\ \omega_{m,-},\ \tau_1,\dots,\tau_r.
\]
故最小递推阶数恰为 \(r+2\)，且相应分母不能再约去，得到 \(\textup{(1)}\) 与 \(\textup{(2)}\)。

最后，阶数为 \(R\) 的 C-finite 序列由任意 \(2R\) 个连续样本唯一恢复，这可由 Hankel 秩或 Prony 反演标准理论得到。此处 \(R=r+2\)，故 \(\textup{(3)}\) 成立。证毕。
\end{proof}

\begin{theorem}[固定 \(m\) 的完整词迹谱可由有限 \(q\)-窗口完全恢复]\label{thm:conclusion-softcore-fixedm-wordtrace-spectrum-window-recovery}
\leanverified{paper\_conclusion\_softcore\_fixedm\_wordtrace\_spectrum\_window\_recovery}
在定理 \ref{thm:conclusion-softcore-fixedm-qrecurrence-generic-minimality} 的 generic 假设下，给定连续窗口
\[
S_m(0),S_m(1),\dots,S_m(2r+3),
\]
即可唯一恢复：
\begin{enumerate}
  \item 全部 distinct 非 \(K^m\) 词迹
  \[
  \mathcal T_m^{\mathrm{dist}}=\{\tau_1,\dots,\tau_r\};
  \]
  \item 各词迹重数
  \[
  \mu_m(\tau_j)\in\ZZ_{>0};
  \]
  \item Lucas 支的两条背景节点
  \[
  \omega_{m,+}=\varphi^m,\qquad \omega_{m,-}=(-1)^m\varphi^{-m}.
  \]
\end{enumerate}
因此，固定 \(m\) 的完整非交换词迹谱并不需要枚举词语言本身；一个长度 \(2r+4\) 的 \(q\)-窗口已经足以把它完全反演。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-softcore-fixedm-qrecurrence-generic-minimality}，给定该长度的窗口即可唯一恢复 \(\mathcal S_m(z)\) 的最小分母，从而恢复全部极点
\[
\{\omega_{m,+},\omega_{m,-},\tau_1,\dots,\tau_r\}.
\]
其中 \(\omega_{m,\pm}\) 是唯一满足
\[
x^2-L_mx+(-1)^m=0
\]
的二根，而 \(\tau_j\in\ZZ_{>0}\)，故这两类节点可从极点集中唯一分离。

一旦底数恢复，命题 \ref{prop:conclusion-softcore-fixedm-qrecurrence-annihilator} 中的展开
\[
S_m(q)
=
2^{-m}\Bigl(c_+\omega_{m,+}^q+c_-\omega_{m,-}^q+\sum_{j=1}^{r}\mu_m(\tau_j)\tau_j^q\Bigr)
\]
在前 \(r+2\) 个 \(q\)-值上形成一个 Vandermonde 线性系统。由于全部底数互异，该系统可逆，故所有系数唯一恢复；其中正整数系数 \(\mu_m(\tau_j)\) 即为所求词迹重数。证毕。
\end{proof}

\begin{theorem}[高矩阶软核的单模主导与指数型余谱压缩]\label{thm:conclusion-softcore-highq-one-mode-dominance}
固定 \(m\ge 1\)。记
\[
\lambda_0^{(q)}\ge \lambda_1^{(q)}\ge \cdots \ge \lambda_q^{(q)}
\]
为 \(T^{(q)}|_{\operatorname{Sym}^q(\CC^2)}\) 的特征值。则当 \(q\to\infty\) 时，有
\[
\frac{\max_{1\le i\le q}|\lambda_i^{(q)}|}{2^{q-1}}
\le
\left(\frac{\varphi}{2}\right)^q
\longrightarrow 0,
\]
并且
\[
\Tr\!\Bigl(\bigl(T^{(q)}|_{\operatorname{Sym}^q(\CC^2)}\bigr)^m\Bigr)
=
\bigl(\lambda_0^{(q)}\bigr)^m
\left(
1+O\!\left(q\left(\frac{\varphi}{2}\right)^{qm}\right)
\right).
\]
因此，当矩阶 \(q\) 增大时，软核对称幂谱进入真正的一模主导 regime，而全部余谱被压入 \((\varphi/2)^q\) 级的次指数壳。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-softcore-spectral-scale-separation}，
\[
\lambda_0^{(q)}\le 2^{q-1}+\frac12\varphi^q,
\qquad
|\lambda_i^{(q)}|\le \frac12\varphi^q\quad(1\le i\le q).
\]
同时，由同一分解 \(T^{(q)}|_{\operatorname{Sym}^q}=A_q+B_q\) 及 Weyl 下界，
\[
\lambda_0^{(q)}\ge 2^{q-1}-\frac12\varphi^q.
\]
故对充分大 \(q\)，\(\lambda_0^{(q)}>0\) 且
\[
\left|\frac{\lambda_i^{(q)}}{\lambda_0^{(q)}}\right|
\le
C\left(\frac{\varphi}{2}\right)^q
\qquad(1\le i\le q)
\]
对某个与 \(m\) 无关的常数 \(C>0\) 成立。

于是
\[
\Tr\!\Bigl(\bigl(T^{(q)}|_{\operatorname{Sym}^q}\bigr)^m\Bigr)
=
\sum_{i=0}^{q}\bigl(\lambda_i^{(q)}\bigr)^m
=
\bigl(\lambda_0^{(q)}\bigr)^m
\left(
1+\sum_{i=1}^{q}\Bigl(\frac{\lambda_i^{(q)}}{\lambda_0^{(q)}}\Bigr)^m
\right),
\]
而余项满足
\[
\sum_{i=1}^{q}\left|\frac{\lambda_i^{(q)}}{\lambda_0^{(q)}}\right|^m
\le
q\,C^m\left(\frac{\varphi}{2}\right)^{qm}
=
O\!\left(q\left(\frac{\varphi}{2}\right)^{qm}\right).
\]
这就得到所示单模主导展开。证毕。
\end{proof}
