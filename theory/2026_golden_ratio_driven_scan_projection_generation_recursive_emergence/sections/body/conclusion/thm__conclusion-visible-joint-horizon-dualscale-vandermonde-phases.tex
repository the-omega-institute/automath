\begin{theorem}[Toeplitz--扫描共同协议的 sharp 视界]\label{thm:conclusion-visible-joint-horizon-sharp-2d}
固定整数 \(D\ge 1\)，并设 \(\mathcal R_D^{\mathrm{gen}}\) 为最小 realization 维数不超过 \(D\) 且处于一般位置的对象类。考虑一类共同视界协议：给定整数 \(N\)，协议同时可访问 Toeplitz 截断族
\[
T_0(C),T_1(C),\dots,T_N(C)
\]
与扫描前缀
\[
u_0,u_1,\dots,u_{N-1}.
\]
若要求同一协议同时完成以下两项任务：
\[
\text{其一，判定 }T_n(C)\succeq 0\ \text{对一切 }n\ge 0\ \text{是否成立},
\]
\[
\text{其二，唯一恢复底层谱表示},
\]
则其最小共同视界恰为
\[
N_{\mathrm{joint}}(D)=2D.
\]
\leanverified{paper\_conclusion\_visible\_joint\_horizon\_sharp\_2d}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-toeplitz-psd-coherence-horizon-threshold} 与定理 \ref{thm:xi-oracle-collapse-toeplitz-psd-finite-truncation}，在 realization 维数不超过 \(D\) 的有界类上，存在统一阈值 \(N_0\le 2D\) 使
\[
T_n(C)\succeq 0\ (\forall n\ge 0)
\iff
T_n(C)\succeq 0\ (0\le n\le N_0).
\]
故当 \(N=2D\) 时，协议已足以完成全阶 Toeplitz--PSD 的判定。

另一方面，同一定理 \ref{thm:xi-toeplitz-psd-coherence-horizon-threshold} 的第二部分指出：在一般位置下，长度 \(2D\) 的样本前缀
\[
(u_0,\dots,u_{2D-1})
\]
足以唯一恢复谱参数，而任何长度不超过 \(2D-1\) 的前缀都存在非唯一性见证。故 \(N=2D\) 足以完成唯一恢复，而 \(N\le 2D-1\) 时任务二必不能普适完成。

于是若同一协议要求同时完成判定与重构，则必要且充分的共同视界正是 \(2D\)。证毕。
\end{proof}

\begin{theorem}[全角度 Toeplitz 正性的有限一元区间化]\label{thm:conclusion-visible-all-angle-interval-psd-collapse}
\leanverified{paper\_conclusion\_visible\_all\_angle\_interval\_psd\_collapse}
在定理 \ref{thm:conclusion-visible-joint-horizon-sharp-2d} 的 realization 维数不超过 \(D\) 口径下，设每个截断阶 \(n\) 所对应的 Schur 递归预测多项式记为 \(p_n\)，并进一步假设其系数都落在不变量子环 \(\QQ[u+u^{-1}]\) 中。则存在某个统一阈值 \(N_0\le 2D\)，使得
\[
T_n(C)\succeq 0\ \text{对一切 }n\ge 0\ \text{成立}
\]
当且仅当
\[
H(p_n)(x)\succeq 0
\qquad
(\forall x\in[-2,2],\ 0\le n\le N_0),
\]
其中 \(H(p_n)\) 为相应的 Schur--Cohn Hermitian 形式矩阵。
因此，在该有界类上，全角度核验与无穷 Toeplitz--PSD 层级严格塌缩为有限多个一元区间矩阵半正定约束。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-toeplitz-schur-cohn-lift}，对每个 \(n\ge 0\) 存在可逆矩阵 \(R_n\) 使
\[
R_n^\ast T_n(C)R_n=H(p_n),
\]
故
\[
T_n(C)\succeq 0
\iff
H(p_n)\succeq 0.
\]
若进一步 \(p_n\) 的系数属于 \(\QQ[u+u^{-1}]\)，则命题 \ref{prop:xi-toeplitz-interval-audit-invariant-ring} 把 Schur--Cohn 证书进一步压缩为区间约束
\[
H(p_n)(x)\succeq 0
\qquad(x\in[-2,2]).
\]

另一方面，定理 \ref{thm:xi-oracle-collapse-toeplitz-psd-finite-truncation} 保证：在 realization 维数不超过 \(D\) 的有界类上，存在统一阈值 \(N_0\le 2D\) 使得全阶 Toeplitz--PSD 条件等价于有限层级 \(0\le n\le N_0\) 的联合核验。将上述逐级合同等价与区间化接口代回，即得所述结论。证毕。
\end{proof}

\begin{theorem}[完整缺陷审计的双尺度阈值]\label{thm:conclusion-visible-complete-defect-audit-dual-scale-threshold}
\leanverified{paper\_conclusion\_visible\_complete\_defect\_audit\_dual\_scale\_threshold}
在一般位置的有限缺陷口径下，设扫描谱指纹满足
\[
u_n=4\pi\sum_{j=1}^{\kappa}w_j z_j^n,
\qquad
w_j>0,
\qquad
z_j=e^{-(1+\delta_j)}e^{-\mathrm{i}\gamma_j}\in\DD,
\]
并记最浅边界深度
\[
h_{\min}:=\min_{1\le j\le \kappa}(1-|a_j|),
\qquad
a_j=\mathcal C(\gamma_j+\mathrm{i}\delta_j).
\]
设共同视界协议以同一整数 \(N\) 同时访问扫描前缀
\[
u_0,\dots,u_{N-1}
\]
与 Toeplitz 截断 \(T_N(C)\)。若该协议要求同时完成：
\[
\text{其一，唯一恢复全部 } \{(\gamma_j,\delta_j,w_j)\}_{j=1}^{\kappa},
\]
\[
\text{其二，在 }T_N(C)\text{ 中完整显影全部 }\kappa\text{ 个负方向},
\]
则在推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 与推论 \ref{cor:xi-toeplitz-saturation-index-comparable-to-shallowest-depth} 的粗下界假设下，存在仅依赖于 \(\kappa\) 及端点 Poisson 权重、伪双曲分离度粗界的常数 \(c,C>0\)，使得
\[
N<2\kappa
\quad\Longrightarrow\quad
\text{任务一在一般位置下不可能普适完成},
\]
\[
N\le c\,h_{\min}^{-1}
\quad\Longrightarrow\quad
\nu_-\!\bigl(T_N(C)\bigr)\le \kappa-1,
\]
而当
\[
N\ge \max\{2\kappa,\ Ch_{\min}^{-1}\}
\]
时，两项任务可同时完成。
因此，完整缺陷审计存在一条真正的双尺度门槛
\[
\max\{2\kappa,\ h_{\min}^{-1}\},
\]
其中代数窗长与几何深度缺一不可，且互不替代。
\end{theorem}
\begin{proof}
对任务一，定理 \ref{thm:xi-node-spectrum-minimal-window-2kappa} 与推论 \ref{cor:xi-prony-decision-reconstruction-one-sample-gap} 表明：长度 \(2\kappa\) 的前缀足以在一般位置下唯一恢复节点与权重，而任何长度不超过 \(2\kappa-1\) 的前缀都存在非唯一性见证。故 \(N<2\kappa\) 时任务一不可能普适完成。

对任务二，命题 \ref{con:discussion-toeplitz-negative-count-threshold} 给出显影负方向所需的粗尺度必要下界：若 \(N\le c\,h_{\min}^{-1}\)，则
\[
\nu_-\!\bigl(T_N(C)\bigr)\le \kappa-1.
\]
另一方面，推论 \ref{cor:xi-toeplitz-visibility-complexity-threshold} 与推论 \ref{cor:xi-toeplitz-saturation-index-comparable-to-shallowest-depth} 共同给出充分性：在相同粗下界假设下，存在常数 \(C>0\) 使
\[
N\ge Ch_{\min}^{-1}
\quad\Longrightarrow\quad
\nu_-\!\bigl(T_N(C)\bigr)=\kappa.
\]

因此，当
\[
N\ge \max\{2\kappa,\ Ch_{\min}^{-1}\}
\]
时，任务一由最短采样窗定理完成，任务二由显影阈值完成；而任何一侧不满足时，相应任务都不能普适闭合。证毕。
\end{proof}

\leanverified{paper\_conclusion\_visible\_negative\_spectrum\_vandermonde\_energy}
\begin{theorem}[负广义谱乘积的 Vandermonde 分离能恒等式]\label{thm:conclusion-visible-negative-spectrum-vandermonde-energy}
在有限 Blaschke 扫描谱指纹分解
\[
u_n=4\pi\sum_{j=1}^{\kappa}w_j z_j^n,
\qquad
w_j>0,
\qquad
z_i\neq z_j\ (i\neq j)
\]
之下，记 Toeplitz 度量谱正规形中的 \(\kappa\) 个负广义特征值为
\[
\lambda_1^-,\dots,\lambda_\kappa^-.
\]
则成立完全显式闭式
\[
\prod_{i=1}^{\kappa}|\lambda_i^-|
=(4\pi)^{2\kappa}
\Bigl(\prod_{j=1}^{\kappa}w_j^2\Bigr)
\Bigl(\prod_{1\le i<j\le\kappa}|z_j-z_i|^4\Bigr).
\]
等价地，
\[
\sum_{1\le i<j\le\kappa}\log|z_j-z_i|
=
\frac14\log\!\Bigl(\prod_{i=1}^{\kappa}|\lambda_i^-|\Bigr)
-\frac{\kappa}{2}\log(4\pi)
-\frac12\sum_{j=1}^{\kappa}\log w_j.
\]
因此，负谱乘积在扣除权重项之后，精确等于缺陷点配置的 Vandermonde 分离能。
\end{theorem}
\begin{proof}
推论 \ref{cor:xi-negative-spectrum-product-dethankel-square} 给出
\[
\prod_{i=1}^{\kappa}|\lambda_i^-|
=
\bigl|\det(\mathsf H_{\kappa-1})\bigr|^2.
\]
而定理 \ref{thm:xi-dethankel-vandermonde-square-finite-blaschke} 进一步给出
\[
\det(\mathsf H_{\kappa-1})
=(4\pi)^{\kappa}
\Bigl(\prod_{j=1}^{\kappa}w_j\Bigr)
\Bigl(\prod_{1\le i<j\le\kappa}(z_j-z_i)^2\Bigr).
\]
对后式取模平方即得第一条恒等式。再取对数并整理，即得第二条。证毕。
\end{proof}

\begin{corollary}[负谱退化的精确二歧]\label{cor:conclusion-visible-negative-spectrum-degeneration-dichotomy}
设 \(\{(w_{j,n},z_{j,n})\}_{j=1}^{\kappa}\) 为一列固定缺陷数 \(\kappa\) 的有限缺陷数据，并记相应的负广义谱乘积为
\[
\Pi_n:=\prod_{i=1}^{\kappa}|\lambda_{i,n}^-|.
\]
则有如下精确判据：
\begin{enumerate}
  \item 若存在常数 \(w_->0\) 使 \(w_{j,n}\ge w_-\) 对一切 \(j,n\) 成立，则
  \[
  \Pi_n\to 0
  \quad\Longleftrightarrow\quad
  \min_{1\le i<j\le \kappa}|z_{i,n}-z_{j,n}|\to 0.
  \]
  \item 在一般情形下，
  \[
  \Pi_n\to 0
  \]
  当且仅当以下两类退化中至少发生一种：
  \[
  \min_{1\le j\le \kappa}w_{j,n}\to 0,
  \qquad\text{或}\qquad
  \min_{1\le i<j\le \kappa}|z_{i,n}-z_{j,n}|\to 0.
  \]
\end{enumerate}
因此，负谱乘积的塌缩只可能来自两种机制：权重熄灭或缺陷碰撞。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-visible-negative-spectrum-vandermonde-energy}，
\[
\Pi_n
=(4\pi)^{2\kappa}
\Bigl(\prod_{j=1}^{\kappa}w_{j,n}^2\Bigr)
\Bigl(\prod_{1\le i<j\le\kappa}|z_{j,n}-z_{i,n}|^4\Bigr).
\]
若 \(w_{j,n}\ge w_->0\) 一致成立，则第一乘积有统一正下界，故 \(\Pi_n\to 0\) 当且仅当 Vandermonde 乘积趋于零。又由于所有 \(z_{j,n}\in\DD\)，各距离统一有上界 \(2\)，于是该乘积趋于零当且仅当至少一个距离因子趋于零，即
\[
\min_{i<j}|z_{i,n}-z_{j,n}|\to 0.
\]
这便得到第一条。

不加权重下界时，同一乘积分解立即给出第二条析取判据。证毕。
\end{proof}

\begin{conjecture}[Toeplitz 审计视界的三相律]\label{conj:conclusion-visible-toeplitz-audit-three-phase-law}
设 \(N_{\mathrm{joint}}\) 记某对象类在 Toeplitz--扫描接口下的最小否证/重构共同视界。则该视界应当服从如下三相分裂：
\begin{enumerate}
  \item \textnormal{有界实现相：}若最小 realization 维数存在统一上界 \(D\)，则
  \[
  N_{\mathrm{joint}}\asymp 2D.
  \]
  \item \textnormal{有限缺陷几何相：}若对象类由一般位置的有限缺陷数据给出，并满足端点 Poisson 权重与伪双曲分离度的粗下界，则
  \[
  N_{\mathrm{joint}}\asymp \max\{2\kappa,\ h_{\min}^{-1}\}.
  \]
  \item \textnormal{纤维内 NP 相：}若对象类允许把纤维内 NP 完备谓词编译到 Toeplitz--PSD 视界，则
  \[
  N_{\mathrm{joint}}\ge \exp(\Omega(n)).
  \]
\end{enumerate}
这里第一相由定理 \ref{thm:conclusion-visible-joint-horizon-sharp-2d} 支撑，第二相由定理 \ref{thm:conclusion-visible-complete-defect-audit-dual-scale-threshold} 支撑，而第三相则与猜想 \ref{conj:xi-toeplitz-psd-horizon-block-foldsat} 所预言的指数不可压缩门槛一致。
\end{conjecture}
