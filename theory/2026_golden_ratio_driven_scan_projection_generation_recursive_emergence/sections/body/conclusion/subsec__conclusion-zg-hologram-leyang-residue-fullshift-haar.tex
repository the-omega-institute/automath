\subsection{ZG 的 Tauberian 边界、全息满移位共轭与 Lee--Yang 五分支 Haar 极限}\label{subsec:conclusion-zg-hologram-leyang-residue-fullshift-haar}
沿用定理 \ref{thm:xi-zg-abel-residue-log-density}、推论 \ref{cor:xi-zg-absolute-convergence-critical-line}、定理 \ref{thm:xi-time-part62deb-hologram-affine-fullshift-conjugacy}、推论 \ref{cor:xi-time-part62deb-hologram-fullshift-entropy-primitive-zeta}、推论 \ref{cor:xi-terminal-zm-leyang-ramification-pullback-2n}、推论 \ref{cor:xi-terminal-zm-leyang-branchpoint-tate-torsor-decomposition} 与定理 \ref{thm:xi-time-part9gc-resolution-tower-haar-limit} 的记号。前述链条已经分别给出：\(\mathrm{ZG}\) 的自然密度与 Dirichlet 级数的 Abel 边界律、全息事件像与单侧满移位的严格仿射共轭，以及 Lee--Yang 五有限分歧族在 dyadic 逆极限中的 Tate-torsor 分解与 Haar 极限律。由此还可进一步抽出如下五条后果：\(\mathrm{ZG}\) 的简单极点常数与自然密度的一致性、其对数加权 Mertens--Ces\`aro 律、全息事件系统的满移位动力学闭式、primitive 周期轨道的 Möbius 公式，以及 Lee--Yang 五分支有限层均匀测度向分量平均 Haar 测度的边界收敛。

\begin{theorem}[ZG 的 Dirichlet 级数在 \texorpdfstring{$s=1^+$}{s=1+} 的简单极点律，其主部系数等于自然密度]\label{thm:conclusion-zg-dirichlet-simple-pole-density}
定义
\[
A_{\mathrm{ZG}}(x):=\#\{n\le x:\ n\in \mathrm{ZG}\},
\qquad
D_{\mathrm{ZG}}(s):=\sum_{n\in \mathrm{ZG}}n^{-s}
\qquad(\Re(s)>1),
\]
以及
\[
\delta_{\mathrm{ZG}}:=\lim_{x\to\infty}\frac{A_{\mathrm{ZG}}(x)}{x}.
\]
则 \(\delta_{\mathrm{ZG}}\in(0,1)\)，并且成立
\[
\boxed{\;
\lim_{\sigma\downarrow 1}(\sigma-1)D_{\mathrm{ZG}}(\sigma)=\delta_{\mathrm{ZG}}.\;}
\]
特别地，
\[
\boxed{\;
D_{\mathrm{ZG}}(\sigma)\sim \frac{\delta_{\mathrm{ZG}}}{\sigma-1}
\qquad(\sigma\downarrow 1),\;}
\]
且其收敛横坐标满足
\[
\boxed{\;
\sigma_c(D_{\mathrm{ZG}})=1.\;}
\]
换言之，\(D_{\mathrm{ZG}}\) 在实轴边界 \(s=1^+\) 的主导奇性是一条简单极点律，其主部系数恰为自然密度 \(\delta_{\mathrm{ZG}}\)。
\end{theorem}
\begin{proof}
第一条极限正是定理 \ref{thm:xi-zg-abel-residue-log-density} 的第一条。由推论 \ref{cor:xi-zg-absolute-convergence-critical-line}，进一步有
\[
D_{\mathrm{ZG}}(\sigma)\sim \frac{\delta_{\mathrm{ZG}}}{\sigma-1}
\qquad(\sigma\downarrow 1),
\]
从而得到所述简单极点律。最后，定理 \ref{thm:xi-zg-abel-residue-log-density} 的第三条已给出
\[
\sigma_c(D_{\mathrm{ZG}})=1.
\]
证毕。
\end{proof}

\begin{corollary}[ZG 上的对数加权 Mertens--Ces\`aro 律]\label{cor:conclusion-zg-weighted-mertens-cesaro}
沿用定理 \ref{thm:conclusion-zg-dirichlet-simple-pole-density} 的记号。对任意固定整数 \(k\ge 0\)，成立
\[
\boxed{\;
\sum_{\substack{n\le X\\ n\in \mathrm{ZG}}}\frac{(\log n)^k}{n}
=
\frac{\delta_{\mathrm{ZG}}}{k+1}(\log X)^{k+1}
+o\!\bigl((\log X)^{k+1}\bigr)
\qquad(X\to\infty).\;}
\]
特别地，
\[
\boxed{\;
\sum_{\substack{n\le X\\ n\in \mathrm{ZG}}}\frac1n
=
\delta_{\mathrm{ZG}}\log X+o(\log X).\;}
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:xi-zg-abel-residue-log-density} 的证明，记
\[
A_{\mathrm{ZG}}(x)=\delta_{\mathrm{ZG}}x+r(x),
\qquad
r(x)=o(x).
\]
对
\[
f_k(x):=\frac{(\log x)^k}{x}
\]
作 Abel 分部求和，得到
\[
\sum_{\substack{n\le X\\ n\in \mathrm{ZG}}}f_k(n)
=
A_{\mathrm{ZG}}(X)f_k(X)-\int_1^X A_{\mathrm{ZG}}(t)f_k'(t)\,\dd t.
\]
代入 \(A_{\mathrm{ZG}}(t)=\delta_{\mathrm{ZG}}t+r(t)\) 后，
\[
\sum_{\substack{n\le X\\ n\in \mathrm{ZG}}}f_k(n)
=
\delta_{\mathrm{ZG}}
\left(
Xf_k(X)-\int_1^X tf_k'(t)\,\dd t
\right)
+
r(X)f_k(X)-\int_1^X r(t)f_k'(t)\,\dd t.
\]
由恒等式
\[
\frac{\dd}{\dd t}\bigl(tf_k(t)\bigr)=f_k(t)+tf_k'(t)
\]
可得
\[
Xf_k(X)-\int_1^X tf_k'(t)\,\dd t
=
\int_1^X f_k(t)\,\dd t+f_k(1),
\]
故
\[
\delta_{\mathrm{ZG}}
\left(
Xf_k(X)-\int_1^X tf_k'(t)\,\dd t
\right)
=
\delta_{\mathrm{ZG}}\int_1^X \frac{(\log t)^k}{t}\,\dd t+O(1)
=
\frac{\delta_{\mathrm{ZG}}}{k+1}(\log X)^{k+1}+O(1).
\]

另一方面，
\[
r(X)f_k(X)=o\!\bigl((\log X)^k\bigr)
=
o\!\bigl((\log X)^{k+1}\bigr).
\]
又因
\[
f_k'(t)=\frac{k(\log t)^{k-1}-(\log t)^k}{t^2},
\]
故有估计
\[
|f_k'(t)|\ll_k \frac{(\log t)^k}{t^2}.
\]
任取 \(\varepsilon>0\)，取 \(T_0\) 足够大，使 \(|r(t)|\le \varepsilon t\) 对 \(t\ge T_0\) 成立，则
\[
\left|\int_{T_0}^X r(t)f_k'(t)\,\dd t\right|
\ll_k
\varepsilon\int_{T_0}^X \frac{(\log t)^k}{t}\,\dd t
\ll_k
\varepsilon(\log X)^{k+1},
\]
而有限区间 \([1,T_0]\) 上的积分只是常数。令 \(\varepsilon\downarrow 0\)，得到
\[
\int_1^X r(t)f_k'(t)\,\dd t
=
o\!\bigl((\log X)^{k+1}\bigr).
\]
综上，
\[
\sum_{\substack{n\le X\\ n\in \mathrm{ZG}}}\frac{(\log n)^k}{n}
=
\frac{\delta_{\mathrm{ZG}}}{k+1}(\log X)^{k+1}
+o\!\bigl((\log X)^{k+1}\bigr).
\]
当 \(k=0\) 时便恢复定理 \ref{thm:xi-zg-abel-residue-log-density} 的调和主项。证毕。
\end{proof}

\begin{theorem}[全息事件像与单侧满移位的严格共轭]\label{thm:conclusion-hologram-fullshift-topological-conjugacy}
设 \(E\) 为有限事件集，取单射
\[
\mathrm{code}:E\hookrightarrow \{0,\dots,M\}\subset \ZZ,
\qquad
B:=2^L>M.
\]
沿用定理 \ref{thm:xi-time-part62deb-hologram-affine-fullshift-conjugacy} 的地址映射
\[
X:E^{\NN}\longrightarrow T_2(E_{\min}),
\qquad
X((e_t)_{t\ge 1})
:=
\sum_{t\ge 1}\mathrm{code}(e_t)B^{t-1}v,
\]
并记其像为
\[
\mathcal X:=X(E^{\NN})\subset T_2(E_{\min}),
\qquad
\sigma(e_1,e_2,\dots):=(e_2,e_3,\dots).
\]
则 \(X\) 是从 \(E^{\NN}\) 到 \(\mathcal X\) 的拓扑同胚，从而存在唯一连续映射
\[
\Sigma_{\mathcal X}:\mathcal X\to \mathcal X
\]
使
\[
\boxed{\;
X\circ \sigma=\Sigma_{\mathcal X}\circ X.\;}
\]
特别地，
\[
\boxed{\;
h_{\mathrm{top}}(\Sigma_{\mathcal X})=\log|E|,\qquad
\#\operatorname{Fix}(\Sigma_{\mathcal X}^n)=|E|^n\quad(n\ge 1),\;}
\]
并且 Artin--Mazur 动力学 \(\zeta\) 函数满足
\[
\boxed{\;
\zeta_{\Sigma_{\mathcal X}}(z)=\frac{1}{1-|E|z}.\;}
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-time-part62deb-hologram-affine-fullshift-conjugacy} 已证明：\(\mathcal X\) 上存在连续映射 \(\mathcal T\) 使
\[
\mathcal T\circ X=X\circ \sigma,
\]
且 \(X:E^{\NN}\to \mathcal X\) 为拓扑同胚。于是唯一可取
\[
\Sigma_{\mathcal X}:=\mathcal T=X\circ \sigma\circ X^{-1},
\]
从而得到严格共轭关系。

再由推论 \ref{cor:xi-time-part62deb-hologram-fullshift-entropy-primitive-zeta}，
\[
h_{\mathrm{top}}(\mathcal T)=\log|E|,
\qquad
\#\operatorname{Fix}(\mathcal T^n)=|E|^n,
\qquad
\zeta_{\mathcal T}(z)=\frac{1}{1-|E|z}.
\]
将 \(\mathcal T\) 改记为 \(\Sigma_{\mathcal X}\) 即得结论。证毕。
\end{proof}

\begin{corollary}[全息事件系统的 primitive 周期轨道公式]\label{cor:conclusion-hologram-fullshift-primitive-orbit-prime-law}
沿用定理 \ref{thm:conclusion-hologram-fullshift-topological-conjugacy} 的记号。记 \(\pi_{\mathcal X}(n)\) 为 \(\Sigma_{\mathcal X}\) 的 primitive \(n\)-周期轨道条数。则
\[
\boxed{\;
\pi_{\mathcal X}(n)
=
\frac1n\sum_{d\mid n}\mu(d)\,|E|^{n/d}.\;}
\]
特别地，
\[
\boxed{\;
\pi_{\mathcal X}(n)
=
\frac{|E|^n}{n}
+O\!\left(\frac{|E|^{n/2}}{n}\right),\;}
\]
从而
\[
\boxed{\;
\lim_{n\to\infty}\frac1n\log\bigl(n\,\pi_{\mathcal X}(n)\bigr)=\log|E|.\;}
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-hologram-fullshift-topological-conjugacy}，
\[
\#\operatorname{Fix}(\Sigma_{\mathcal X}^n)=|E|^n.
\]
另一方面，固定点数与 primitive 周期轨道数满足标准关系
\[
\#\operatorname{Fix}(\Sigma_{\mathcal X}^n)
=
\sum_{d\mid n} d\,\pi_{\mathcal X}(d).
\]
对该卷积作 Möbius 反演，即得
\[
\pi_{\mathcal X}(n)=\frac1n\sum_{d\mid n}\mu(d)\,|E|^{n/d}.
\]
主项对应 \(d=1\)。若 \(d>1\)，则 \(n/d\le n/2\)，故其余项总和满足
\[
\frac1n\sum_{\substack{d\mid n\\ d>1}}|E|^{n/d}
=
O\!\left(\frac{|E|^{n/2}}{n}\right).
\]
于是
\[
\pi_{\mathcal X}(n)
=
\frac{|E|^n}{n}
+O\!\left(\frac{|E|^{n/2}}{n}\right).
\]
最后，
\[
n\,\pi_{\mathcal X}(n)=|E|^n\bigl(1+O(|E|^{-n/2})\bigr),
\]
取对数并除以 \(n\) 即得
\[
\lim_{n\to\infty}\frac1n\log\bigl(n\,\pi_{\mathcal X}(n)\bigr)=\log|E|.
\]
证毕。
\end{proof}

\begin{theorem}[Lee--Yang 五有限分歧族的 Haar 边界极限]\label{thm:conclusion-leyang-five-branch-haar-boundary-limit}
设
\[
T:=T_2(E_{\min})\cong \varprojlim_n E_{\min}[2^n]\cong \ZZ_2^2.
\]
对 \(i\in\{0,\dots,4\}\) 与 \(n\ge 0\)，沿用推论 \ref{cor:xi-terminal-zm-leyang-branchpoint-tate-torsor-decomposition} 的记号
\[
\Pi_{i,n}:=[2^n]^{-1}(P_i),
\qquad
\Pi_{i,\infty}:=\varprojlim_n(\Pi_{i,n},[2]).
\]
固定每个分支分量中的一条相容提升
\[
\mathbf P_i\in \Pi_{i,\infty},
\]
并通过
\[
\Pi_{i,\infty}\cong \mathbf P_i+T
\]
把第 \(i\) 个分支 sheet 识别为 \(T\) 的一个仿射平移。再固定一个坐标识别 \(T\cong \ZZ_2^2\)，并以标准代表嵌入
\[
\iota_n:(\ZZ/2^n\ZZ)^2\hookrightarrow \ZZ_2^2.
\]
定义离散并空间
\[
\mathfrak R:=\{0,\dots,4\}\times T,
\]
以及支撑在深度 \(n\) 五分支有限层上的均匀原子测度
\[
\nu_n:=
\frac{1}{5\cdot 4^n}
\sum_{i=0}^{4}\ \sum_{u\in(\ZZ/2^n\ZZ)^2}
\delta_{(i,\ \mathbf P_i+\iota_n(u))}.
\]
则在弱收敛意义下，
\[
\boxed{\;
\nu_n\Longrightarrow
\frac15\sum_{i=0}^{4}\delta_i\otimes m_{\mathrm{Haar}}
\qquad(n\to\infty),\;}
\]
其中 \(m_{\mathrm{Haar}}\) 为 \(T\cong \ZZ_2^2\) 上的 Haar 概率测度。等价地，对任意 \(i\in\{0,\dots,4\}\) 与任意 dyadic 圆柱
\[
U(a,b;k):=
\{(x,y)\in \ZZ_2^2:\ x\equiv a,\ y\equiv b\pmod{2^k}\},
\]
都有
\[
\boxed{\;
\lim_{n\to\infty}\nu_n\bigl(\{i\}\times U(a,b;k)\bigr)=\frac{1}{5\cdot 4^k}.\;}
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:xi-terminal-zm-leyang-ramification-pullback-2n}，每个有限分歧族 \(\Pi_{i,n}\) 都是 \(E_{\min}[2^n]\) 的平移余类，因而
\[
\#\Pi_{i,n}=4^n,
\qquad
\sum_{i=0}^{4}\#\Pi_{i,n}=5\cdot 4^n.
\]
这正对应于 \(\nu_n\) 中每个原子权重为 \((5\cdot 4^n)^{-1}\)。

现在固定 \(i\) 与一个深度 \(k\) 的 dyadic 圆柱 \(U(a,b;k)\)。当 \(n\ge k\) 时，在标准代表集
\[
\iota_n\bigl((\ZZ/2^n\ZZ)^2\bigr)\subset \ZZ_2^2
\]
中，落入圆柱 \(U(a,b;k)\) 的点恰有
\[
4^{n-k}
\]
个，因为前 \(k\) 位二进制坐标被固定，而后 \(n-k\) 层每层仍各有 \(4\) 种选择。于是
\[
\nu_n\bigl(\{i\}\times U(a,b;k)\bigr)
=
\frac{4^{n-k}}{5\cdot 4^n}
=
\frac{1}{5\cdot 4^k}.
\]
该值对一切 \(n\ge k\) 都恒定，因此极限存在并等于 \(1/(5\cdot4^k)\)。

另一方面，由定理 \ref{thm:xi-time-part9gc-resolution-tower-haar-limit}，
\[
m_{\mathrm{Haar}}(U(a,b;k))=4^{-k}.
\]
故
\[
\left(\frac15\sum_{i=0}^{4}\delta_i\otimes m_{\mathrm{Haar}}\right)
\bigl(\{i\}\times U(a,b;k)\bigr)
=
\frac15\cdot 4^{-k}
=
\frac{1}{5\cdot 4^k}.
\]
而矩形集
\[
\{i\}\times U(a,b;k)
\]
生成 \(\mathfrak R\) 的 Borel \(\sigma\)-代数，故上述 cylinder 质量公式唯一确定弱极限。于是
\[
\nu_n\Longrightarrow \frac15\sum_{i=0}^{4}\delta_i\otimes m_{\mathrm{Haar}}.
\]
证毕。
\end{proof}
