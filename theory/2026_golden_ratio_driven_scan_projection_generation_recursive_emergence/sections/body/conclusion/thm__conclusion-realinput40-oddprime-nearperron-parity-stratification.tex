\begin{theorem}[奇素数角色的二次近 Perron 律]\label{thm:conclusion-realinput40-oddprime-quadratic-near-perron}
设
\[
\lambda:=\rho(1)=\varphi^2.
\]
对奇素数 \(p\)，定义
\[
t_p:=\frac{2\pi}{p},
\qquad
\beta_p:=\frac{\log \rho(e^{it_p})}{\log \lambda}.
\]
则当 \(p\to\infty\) 时，有渐近展开
\[
\beta_p
=
1-\frac{\kappa}{p^2}+O(p^{-4}),
\qquad
\kappa:=
\frac{(6\sqrt5-5)(2\pi)^2}{250\log(\varphi^2)}>0.
\]
特别地，
\[
\beta_p\to 1,
\]
从而对所有充分大的奇素数 \(p\) 都有
\[
\beta_p>\frac12.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-real-input-40-collision-prime-near-perron-rigidity}
\[
\log\frac{\rho(e^{it})}{\lambda}
=
-\frac{6\sqrt5-5}{250}\,t^2
+\frac{7+24\sqrt5}{75000}\,t^4
+O(t^6)
\qquad (t\to 0).
\]
代入
\[
t=t_p=\frac{2\pi}{p},
\]
得到
\[
\log\frac{\rho(e^{it_p})}{\lambda}
=
-\frac{(6\sqrt5-5)(2\pi)^2}{250}\,p^{-2}
+O(p^{-4}).
\]
又
\[
\beta_p
=
\frac{\log \rho(e^{it_p})}{\log \lambda}
=
1+\frac{\log(\rho(e^{it_p})/\lambda)}{\log \lambda},
\]
故
\[
\beta_p
=
1-\frac{\kappa}{p^2}+O(p^{-4}),
\]
其中
\[
\kappa=
\frac{(6\sqrt5-5)(2\pi)^2}{250\log(\varphi^2)}>0.
\]
于是 \(\beta_p\to 1\)，从而 \(\beta_p>\frac12\) 对所有充分大的奇素数成立。证毕。
\end{proof}

\begin{corollary}[奇偶角色与奇素数角色的谱分层]\label{cor:conclusion-realinput40-parity-vs-oddprime-spectral-stratification}
记
\[
\beta_{\mathrm{par}}
:=
\frac{\log \rho(-1)}{\log \lambda},
\qquad
\lambda=\varphi^2.
\]
则
\[
\beta_{\mathrm{par}}\approx 0.82186854047.
\]
并且存在奇素数阈值 \(p_0\)，使得对一切 \(p\ge p_0\) 的奇素数都有
\[
\beta_p>\beta_{\mathrm{par}}.
\]
因此，最小非平凡 torsion 点 \(u=-1\) 给出一个孤立的低指数 parity 层，而奇素数角色 \(u=e^{2\pi i/p}\) 则构成一条向 Perron 指数 \(1\) 二次逼近的高指数层。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-conductor-two-ramanujan-obstruction-nonrepairable}，
\[
\rho(-1)
\]
是
\[
x^3-2x^2-1=0
\]
的最大实根，且
\[
\beta_{\mathrm{par}}\approx 0.82186854047.
\]
另一方面，定理 \ref{thm:conclusion-realinput40-oddprime-quadratic-near-perron} 已证
\[
\beta_p\to 1
\qquad (p\to\infty,\ p\text{ 为奇素数}).
\]
故存在 \(p_0\) 使得对所有 \(p\ge p_0\) 的奇素数都满足
\[
\beta_p>\beta_{\mathrm{par}}.
\]
这正给出了所述谱分层。证毕。
\end{proof}
