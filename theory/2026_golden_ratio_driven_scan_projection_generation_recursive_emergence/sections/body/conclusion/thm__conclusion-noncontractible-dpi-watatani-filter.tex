\begin{theorem}[任意纤维族上的精确数据处理损失]\label{thm:conclusion-noncontractible-restricted-dpi-exact}
\leanverified{paper\_conclusion\_noncontractible\_restricted\_dpi\_exact}
固定分辨率 \(m\)。对任意非空子集 \(A\subseteq X_m\)，定义受限粗粒化损失
\[
\Delta_m(A):=
\sup_{\substack{\rho,\sigma\in \Delta(\Omega_m)\\
\supp((\Fold_m)_\ast\rho)\cup \supp((\Fold_m)_\ast\sigma)\subseteq A}}
\Bigl[
D(\rho\Vert \sigma)-D((\Fold_m)_\ast\rho\Vert (\Fold_m)_\ast\sigma)
\Bigr].
\]
再记
\[
D_A:=\max_{x\in A}d_m(x).
\]
则成立严格闭式
\[
\Delta_m(A)=\log D_A.
\]
特别地，若记
\[
X_m^{\mathrm{nc}}:=\{x\in X_m:\ |K_x|\not\simeq *\},
\qquad
\widetilde D_m:=\max_{x\in X_m^{\mathrm{nc}}}d_m(x),
\]
则
\[
\Delta_m(X_m^{\mathrm{nc}})=\log \widetilde D_m.
\]
\end{theorem}
\begin{proof}
对任意 \(\rho,\sigma\in\Delta(\Omega_m)\)，记
\[
\rho_X:=(\Fold_m)_\ast\rho,\qquad \sigma_X:=(\Fold_m)_\ast\sigma.
\]
与定理 \ref{thm:fold-max-data-processing-saturation-chi} 的证明同理，粗粒化相对熵满足分解
\[
D(\rho\Vert \sigma)-D(\rho_X\Vert \sigma_X)
=
\sum_{x\in X_m}\rho_X(x)\,D\!\bigl(\rho(\cdot\mid x)\Vert \sigma(\cdot\mid x)\bigr).
\]
而对每个 \(x\) 都有
\[
D\!\bigl(\rho(\cdot\mid x)\Vert \sigma(\cdot\mid x)\bigr)\le \log d_m(x),
\]
因为条件分布均支撑于大小为 \(d_m(x)\) 的有限集。若 \(\rho_X,\sigma_X\) 的支撑并于 \(A\) 内，则
\[
D(\rho\Vert \sigma)-D(\rho_X\Vert \sigma_X)
\le
\sum_{x\in A}\rho_X(x)\log d_m(x)
\le \log D_A.
\]

反向取等由单纤维实现。取 \(x^\star\in A\) 使 \(d_m(x^\star)=D_A\)，并任选 \(a^\star\in \Fold_m^{-1}(x^\star)\)。令
\[
\rho:=\delta_{a^\star},
\qquad
\sigma:=\mathrm{Unif}(\Fold_m^{-1}(x^\star)).
\]
则
\[
(\Fold_m)_\ast\rho=(\Fold_m)_\ast\sigma=\delta_{x^\star},
\]
故粗粒化后的相对熵为零，而
\[
D(\rho\Vert \sigma)=\log d_m(x^\star)=\log D_A.
\]
于是 \(\Delta_m(A)=\log D_A\)。

最后，由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，\(|K_x|\not\simeq *\) 当且仅当 \(d_m(x)\) 为奇数，故 \(X_m^{\mathrm{nc}}\) 正是奇纤维层。代入 \(A=X_m^{\mathrm{nc}}\) 即得末式。证毕。
\end{proof}

\begin{theorem}[非可缩筛子的单中心元实现]\label{thm:conclusion-noncontractible-central-projector-indw}
沿用
\[
A_{m,1}\cong \bigoplus_{x\in X_m}M_{d_m(x)}(\CC),
\qquad
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m}d_m(x)\,\widetilde P_x\in Z(A_{m,1})
\]
的记号，其中 \(\widetilde P_x\) 为块 \(x\) 的中心极小投影。定义中心投影
\[
p_m^{\mathrm{nc}}
:=
\frac{1-(-1)^{\mathrm{IndW}(E_{m,1})}}{2}
\quad\text{于 }Z(A_{m,1}),
\]
其中幂由中心函数演算给出。则
\[
p_m^{\mathrm{nc}}
=
\sum_{x\in X_m^{\mathrm{nc}}}\widetilde P_x
=
\sum_{d_m(x)\ \mathrm{odd}}\widetilde P_x.
\]
并且对每个整数 \(q\ge 0\),
\[
2^m\,\tau_{m,1}\!\bigl(p_m^{\mathrm{nc}}\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m^{\mathrm{nc}}}d_m(x)^{q+1}.
\]
从而
\[
\widetilde D_m
=
\lim_{q\to\infty}
\Bigl(
2^m\,\tau_{m,1}\!\bigl(p_m^{\mathrm{nc}}\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
\Bigr)^{1/(q+1)},
\]
以及
\[
\Delta_m(X_m^{\mathrm{nc}})
=
\lim_{q\to\infty}
\frac{1}{q+1}
\log
\Bigl(
2^m\,\tau_{m,1}\!\bigl(p_m^{\mathrm{nc}}\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
\Bigr).
\]
\end{theorem}
\begin{proof}
由中心函数演算，
\[
(-1)^{\mathrm{IndW}(E_{m,1})}
=
\sum_{x\in X_m}(-1)^{d_m(x)}\,\widetilde P_x,
\]
故
\[
p_m^{\mathrm{nc}}
=
\sum_{d_m(x)\ \mathrm{odd}}\widetilde P_x.
\]
再由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，奇纤维层正是非可缩层，故
\[
p_m^{\mathrm{nc}}
=
\sum_{x\in X_m^{\mathrm{nc}}}\widetilde P_x.
\]

另一方面，
\[
\mathrm{IndW}(E_{m,1})^{\,q}=\sum_{x\in X_m}d_m(x)^q\,\widetilde P_x,
\]
从而
\[
p_m^{\mathrm{nc}}\mathrm{IndW}(E_{m,1})^{\,q}
=
\sum_{x\in X_m^{\mathrm{nc}}}d_m(x)^q\,\widetilde P_x.
\]
由推论 \ref{cor:fold-indw-high-q-extract-maxfiber-and-chi} 的证明，
\[
\tau_{m,1}(\widetilde P_x)=\frac{d_m(x)}{2^m}.
\]
于是
\[
\tau_{m,1}\!\bigl(p_m^{\mathrm{nc}}\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\frac1{2^m}\sum_{x\in X_m^{\mathrm{nc}}}d_m(x)^{q+1},
\]
即得矩恒等式。

最后，对有限和取 \(\ell^\infty\) 极限，
\[
\lim_{q\to\infty}
\Bigl(\sum_{x\in X_m^{\mathrm{nc}}}d_m(x)^{q+1}\Bigr)^{1/(q+1)}
=
\max_{x\in X_m^{\mathrm{nc}}}d_m(x)
=
\widetilde D_m.
\]
再与定理 \ref{thm:conclusion-noncontractible-restricted-dpi-exact} 的
\(
\Delta_m(X_m^{\mathrm{nc}})=\log \widetilde D_m
\)
合并，即得最后一式。证毕。
\end{proof}

\begin{theorem}[极大块理想在非可缩筛子下的二值律]\label{thm:conclusion-noncontractible-maxideal-binary-law}
记
\[
I_m:=\bigoplus_{x:\ d_m(x)=D_m}M_{D_m}(\CC)\subset A_{m,1}
\]
为极大块理想。则非可缩中心投影 \(p_m^{\mathrm{nc}}\) 对 \(I_m\) 的作用满足严格二值律
\[
p_m^{\mathrm{nc}}I_m=
\begin{cases}
I_m,& m\equiv 0,4\pmod 6,\\
0,& m\equiv 1,2,3,5\pmod 6.
\end{cases}
\]
并且在保留情形 \(m\equiv 0,4\pmod 6\) 下，仍有
\[
p_m^{\mathrm{nc}}I_m=I_m\subseteq e_{\chi_m}A_{m,1}.
\]
\end{theorem}
\begin{proof}
在每个 \(D_m\)-块上，\(\mathrm{IndW}(E_{m,1})\) 取常值 \(D_m\)，故
\[
p_m^{\mathrm{nc}}\big|_{I_m}
=
\frac{1-(-1)^{D_m}}{2}\,\mathrm{id}_{I_m}.
\]
于是只需判定 \(D_m\) 的奇偶。
由定理 \ref{thm:pom-max-fiber}，
\[
D_{2k}=F_{k+2},\qquad D_{2k+1}=2F_{k+1}.
\]
故奇数层 \(m=2k+1\) 上 \(D_m\) 恒为偶数。偶数层上，\(F_{k+2}\) 为奇当且仅当 \(k+2\not\equiv 0\pmod 3\)，亦即 \(m=2k\equiv 0,4\pmod 6\)。这就给出了上述二值律。

最后，保留情形下的扇区包含关系正是命题 \ref{prop:fold-groupoid-maxblock-chi-homogeneity}。证毕。
\end{proof}

\begin{corollary}[奇数层的绝对极值证书必然同伦平凡]\label{cor:conclusion-noncontractible-odd-global-extremizers-contractible}
对所有充分大的奇数 \(m\)，任一达到全局数据处理损失 \(\Delta_m\) 的极值对，其宏观支撑必落在满足下列两条性质的纤维上：
\begin{enumerate}
  \item \(|K_x|\simeq *\)；
  \item \(\operatorname{sgn}(\iota_x)=+1\)。
\end{enumerate}
并且在这些奇数层上，最佳非可缩证书只能退居次顶层或第三层：
\[
\Delta_m(X_m^{\mathrm{nc}})
=
\begin{cases}
\log D_m^{(2)},& m\equiv 1,5\pmod 6,\\
\log D_m^{(3)},& m\equiv 3\pmod 6.
\end{cases}
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:fold-max-data-processing-saturation-chi}，任何达到 \(\Delta_m\) 的极值对，其宏观支撑只能落在
\[
\mathcal M_m:=\{x\in X_m:\ d_m(x)=D_m\}
\]
上。对奇数层 \(m=2k+1\)，有 \(D_m=2F_{k+1}\) 为偶数，故由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，\(\mathcal M_m\) 中所有纤维都可缩。另一方面，定理 \ref{thm:fold-max-data-processing-saturation-chi} 同时给出在稳定奇数层上 \(\operatorname{sgn}(\iota_x)=+1\)。

最后，由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，奇数层上的最大非可缩纤维分别为 \(D_m^{(2)}\) 或 \(D_m^{(3)}\)，再与定理 \ref{thm:conclusion-noncontractible-restricted-dpi-exact} 合并，即得末式。证毕。
\end{proof}

\begin{conjecture}[次极值非可缩块的扇区刚性]\label{conj:conclusion-noncontractible-top-sector-rigidity}
定义非可缩顶层理想
\[
\widetilde I_m
:=
\bigoplus_{x:\ d_m(x)=\widetilde D_m}M_{\widetilde D_m}(\CC)
\subset A_{m,1}.
\]
则应存在一个最终周期的符号指纹
\[
\widetilde\chi_m\in\{\pm1\}^2
\]
使得对所有充分大的 \(m\) 都有
\[
\widetilde I_m\subseteq e_{\widetilde\chi_m}A_{m,1}.
\]
更强地，\(\widetilde\chi_m\) 应只最终依赖于 \(m\bmod 12\)。
\end{conjecture}
