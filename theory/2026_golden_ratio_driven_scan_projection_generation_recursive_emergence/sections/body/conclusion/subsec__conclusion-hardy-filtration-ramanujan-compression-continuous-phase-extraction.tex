沿用小节 \ref{subsec:conclusion-phase-gram-mellin-hardy-radial-rigidity}、\S\ref{subsec:cdim-rational-major-arc-gram}--\S\ref{subsec:cdim-farey-operator-limit} 以及相位 Fourier 化段落 \ref{para:cdim-poisson-cauchy-phase-fourierization} 的记号。前述链条已经把有限模态 Hardy 投影的显式有理核、Poisson--Cauchy 导数比的有限 Laurent 截断、主弧 Gram 核的 Ramanujan 局域化、单尺度谱降的全尺度传播以及 Farey 采样连续化的算子范数极限整理成一条可审计闭环。把这些接口进一步向内压缩后，还可抽出一条此前未被单独封装的谱刚性链：有限 Hardy 旗标本身携带 exact filtration algebra 与完全平坦的 Christoffel 函数；Poisson--Cauchy 导数比在相位变量下严格落入有限 Laurent 壳，并因此允许 sharp 采样与 half-Hardy 无损压缩；在分母层常量子空间上，离散 Gram 算子又严格压缩为有限维 Ramanujan 相关矩阵；而 continuous phase 核在 Fourier 基下完全对角化，从而把低温秩一塌缩、高温固定频带平坦化、strict spectral drop 的单尺度离散证书以及 near-resonant phase 的连续提取统一到同一条相变链中。

\begin{theorem}[有限 Hardy 投影塔的 exact filtration algebra]\label{thm:conclusion-hardy-filtration-exact-projection-algebra}
\leanverified{paper\_conclusion\_hardy\_filtration\_exact\_projection\_algebra}
对任意整数 \(M>N\ge 1\)，令
\[
\mathcal H_N:=\operatorname{span}\{w(\cdot)^k:0\le k\le N-1\}\subset L^2(\RR,\dd\mu),
\qquad
\Pi_N:L^2(\RR,\dd\mu)\to \mathcal H_N
\]
为相应正交投影。则成立
\[
\Pi_M\Pi_N=\Pi_N\Pi_M=\Pi_N.
\]
并且差算子 \(\Pi_M-\Pi_N\) 仍为正交投影，其像空间恰为
\[
\mathcal H_M\ominus \mathcal H_N
=
\operatorname{span}\{w(\cdot)^k:N\le k\le M-1\},
\]
故
\[
\operatorname{rank}(\Pi_M-\Pi_N)=M-N.
\]
相应核差满足显式公式
\[
K_M(x,y)-K_N(x,y)
=
\bigl(w(x)\overline{w(y)}\bigr)^N
\frac{1-\bigl(w(x)\overline{w(y)}\bigr)^{M-N}}{1-w(x)\overline{w(y)}}.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:unit-circle-phase-hardy-kernel-rational}，\(\Pi_N\) 是到 \(\mathcal H_N\) 的正交投影，而 \(\mathcal H_N\subset\mathcal H_M\)。嵌套闭子空间上的正交投影代数立即给出
\[
\Pi_M\Pi_N=\Pi_N\Pi_M=\Pi_N.
\]
于是
\[
(\Pi_M-\Pi_N)^2=\Pi_M-\Pi_N,
\qquad
(\Pi_M-\Pi_N)^\ast=\Pi_M-\Pi_N,
\]
故 \(\Pi_M-\Pi_N\) 仍为正交投影。其像空间等于 \(\mathcal H_M\ominus\mathcal H_N\)，而 \(\{w^k\}_{k\ge 0}\) 构成该 Hardy 旗标的标准正交模态，故差空间恰由 \(N\le k\le M-1\) 的模态张成，维数为 \(M-N\)。核公式只需把几何级数闭式
\[
K_M(x,y)-K_N(x,y)=\sum_{k=N}^{M-1}\bigl(w(x)\overline{w(y)}\bigr)^k
\]
重新求和即可。证毕。
\end{proof}

\begin{theorem}[有限模态 Christoffel 函数的完全平坦律]\label{thm:conclusion-hardy-christoffel-complete-flatness}
对每个 \(x_0\in\RR\) 与 \(N\ge 1\)，在约束
\[
f\in\mathcal H_N,
\qquad
f(x_0)=1
\]
下，\(L^2(\RR,\dd\mu)\)-范数的最小值恰为
\[
\min\|f\|_{L^2(\dd\mu)}^2=\frac{1}{K_N(x_0,x_0)}=\frac1N.
\]
唯一极小元为
\[
f^\ast_{x_0,N}(x)=\frac{K_N(x,x_0)}{K_N(x_0,x_0)}=\frac1N\,K_N(x,x_0).
\]
因此，有限模态 Hardy 子空间的点值 Christoffel 函数在实轴上完全平坦，并恒等于 \(N\)。
\end{theorem}
\begin{proof}
\(\mathcal H_N\) 是 reproducing kernel Hilbert space，reproducing kernel 正是命题 \ref{prop:unit-circle-phase-hardy-kernel-rational} 给出的 \(K_N\)。故对任意 \(f\in\mathcal H_N\) 有
\[
f(x_0)=\langle f,K_N(\cdot,x_0)\rangle_{L^2(\dd\mu)}.
\]
若 \(f(x_0)=1\)，则由 Cauchy--Schwarz 不等式，
\[
1\le \|f\|_{L^2(\dd\mu)}^2\,\|K_N(\cdot,x_0)\|_{L^2(\dd\mu)}^2
=
\|f\|_{L^2(\dd\mu)}^2\,K_N(x_0,x_0).
\]
于是
\[
\|f\|_{L^2(\dd\mu)}^2\ge \frac1{K_N(x_0,x_0)}.
\]
当 \(f=K_N(\cdot,x_0)/K_N(x_0,x_0)\) 时等号达到；再由 Cauchy--Schwarz 等号条件知该极小元唯一。命题 \ref{prop:unit-circle-phase-hardy-kernel-rational} 还给出 \(K_N(x_0,x_0)=N\)，故得到所述完全平坦律。证毕。
\end{proof}

\begin{theorem}[Poisson--Cauchy 导数比的有限 Laurent 带宽与最高模态刚性]\label{thm:conclusion-poisson-cauchy-derivative-ratio-finite-laurent-rigidity}
记
\[
P_t(x):=\frac{1}{\pi}\frac{t}{x^2+t^2},
\qquad
W(y):=\frac{1+\mathrm{i}y}{1-\mathrm{i}y}\in\TT,
\]
并对整数 \(m\ge 0\) 定义归一化导数比
\[
Q_m(y):=t^m\frac{\partial_x^m P_t(ty)}{P_t(ty)}.
\]
则 \(Q_m\) 与尺度参数 \(t\) 无关，并存在唯一系数族 \(\{c_{m,k}\}_{|k|\le m}\subset\CC\) 使
\[
Q_m(y)=\sum_{k=-m}^{m}c_{m,k}W(y)^k.
\]
其最高模态系数满足刚性闭式
\[
c_{m,m}=\frac{\mathrm{i}^m m!}{2^m},
\qquad
c_{m,-m}=\frac{(-\mathrm{i})^m m!}{2^m}.
\]
特别地，\(Q_m\) 严格落在 Laurent 带宽 \(m\) 的有限模态壳中，而最高正、负模态永不退化。
\end{theorem}
\begin{proof}
这正是命题 \ref{prop:cdim-poisson-cauchy-finite-laurent-modes} 的内容。其证明给出：\(Q_m\) 对 \(t\) 不变，且在 Cayley 相位变量 \(W\) 下具有唯一的有限 Laurent 展开；最高模态系数则由分式分解与 \(W=\iota(y)\) 的显式代换直接计算得到。证毕。
\end{proof}

\begin{corollary}[Poisson--Cauchy 证书的最小 Laurent 阶可识别性]\label{cor:conclusion-poisson-cauchy-minimal-laurent-order-identifiability}
\leanverified{paper\_conclusion\_poisson\_cauchy\_minimal\_laurent\_order\_identifiability}
沿用定理 \ref{thm:conclusion-poisson-cauchy-derivative-ratio-finite-laurent-rigidity} 的记号。则 \(Q_m\) 不可能落在更小的 Laurent 子空间
\[
\operatorname{span}\{W^k:\ |k|\le m-1\}.
\]
因此，导数阶 \(m\) 可由 \(Q_m\) 的 Laurent 支撑唯一恢复。
\end{corollary}
\begin{proof}
若 \(Q_m\in \operatorname{span}\{W^k:|k|\le m-1\}\)，则其 \(W^{\pm m}\) 系数必为零。但定理 \ref{thm:conclusion-poisson-cauchy-derivative-ratio-finite-laurent-rigidity} 已给出
\[
c_{m,m}=\frac{\mathrm{i}^m m!}{2^m}\neq 0,
\qquad
c_{m,-m}=\frac{(-\mathrm{i})^m m!}{2^m}\neq 0,
\]
矛盾。故所述带宽 \(m\) 是最小可见阶。证毕。
\end{proof}

\begin{theorem}[有限 Laurent 证书的 sharp 采样数]\label{thm:conclusion-finite-laurent-certificate-sharp-sampling}
\leanverified{paper\_conclusion\_finite\_laurent\_certificate\_sharp\_sampling}
固定 \(m\ge 0\)。若给定 \(2m+1\) 个两两不同的相位点
\[
\zeta_0,\dots,\zeta_{2m}\in\TT,
\]
则任一 Laurent 多项式
\[
F(W)=\sum_{k=-m}^{m}a_kW^k
\]
都由采样值
\[
F(\zeta_0),\dots,F(\zeta_{2m})
\]
唯一确定。特别地，恢复 \(Q_m\) 的全部 Laurent 系数最少需要 \(2m+1\) 个复采样值，并且这一数目是 sharp 的。
\end{theorem}
\begin{proof}
若两个带宽不超过 \(m\) 的 Laurent 多项式在 \(\zeta_0,\dots,\zeta_{2m}\) 上取值相同，则其差 \(H(W)\) 在这 \(2m+1\) 个不同点上全为零。乘以 \(W^m\) 得到一个次数不超过 \(2m\) 的普通多项式
\[
P(W):=W^mH(W),
\]
它拥有 \(2m+1\) 个不同零点，只能恒等于零。故 \(H\equiv 0\)，唯一性成立。

另一方面，\(\operatorname{span}\{W^k:|k|\le m\}\) 的复维数正是 \(2m+1\)。少于此数的复采样不可能统一决定全部系数，故该数目是 sharp 的。证毕。
\end{proof}

\begin{corollary}[实值 Poisson--Cauchy 证书的 half-Hardy 无损压缩]\label{cor:conclusion-poisson-cauchy-halfhardy-lossless-compression}
沿用定理 \ref{thm:conclusion-poisson-cauchy-derivative-ratio-finite-laurent-rigidity} 的记号。由于 \(Q_m\) 在实轴上取实值，故其 Laurent 系数满足共轭对称
\[
c_{m,-k}=\overline{c_{m,k}}
\qquad (0\le k\le m).
\]
因此，有限 Hardy 投影
\[
\Pi_{m+1}Q_m
\]
已足以唯一恢复整个 \(Q_m\)。等价地，原本 \(2m+1\) 维的 Laurent 证书可被压缩为 \(m+1\) 个 Hardy 正模态而不丢失信息。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-poisson-cauchy-derivative-ratio-finite-laurent-rigidity}，
\[
Q_m(y)=\sum_{k=-m}^{m}c_{m,k}W(y)^k.
\]
因 \(Q_m(y)\in\RR\)（\(y\in\RR\)）且 \(W(y)\in\TT\)，故 Laurent 系数满足
\(
c_{m,-k}=\overline{c_{m,k}}
\)。
另一方面，\(\Pi_{m+1}\) 恰保留 Hardy 正模态
\[
1,W,\dots,W^m.
\]
故一旦知道
\[
c_{m,0},c_{m,1},\dots,c_{m,m},
\]
负模态便由共轭对称自动恢复，从而 \(\Pi_{m+1}Q_m\) 唯一决定 \(Q_m\)。证毕。
\end{proof}

\begin{theorem}[分母层常量子空间的 Ramanujan 压缩正规形]\label{thm:conclusion-majorarc-denominator-constant-ramanujan-normal-form}
定义
\[
\mathcal C_M
:=
\left\{
g\in \ell^2(\mathcal Q_M):
\exists (c_q)_{1\le q\le M}\ \text{s.t.}\ 
g(a/q)=c_q\ \forall (a,q)=1
\right\}.
\]
则在自然加权同构
\[
\mathcal C_M\cong \ell^2_\varphi(\{1,\dots,M\}),
\qquad
\|(c_q)\|_\varphi^2:=\sum_{q\le M}\varphi(q)\,|c_q|^2
\]
下，主弧 Gram 算子 \(K_{M,L}\) 的限制恰由一个 \(M\times M\) 的正半定 Ramanujan 矩阵表示：
\[
\mathcal R_{M,L}(q,q')
:=
\sum_{n\in\ZZ}\widehat\psi(n/L)\,c_q(n)c_{q'}(n),
\]
其中 \(c_q(n)=\sum_{(a,q)=1}e^{2\pi\mathrm{i}an/q}\) 为 Ramanujan 和。更精确地，若 \(g_c(a/q)=c_q\)，则
\[
\langle g_c,K_{M,L}g_c\rangle
=
\sum_{q,q'\le M}\overline{c_q}\,\mathcal R_{M,L}(q,q')\,c_{q'}.
\]
因此，主弧 Gram 核在“分母层常量”口径下严格压缩为一条有限维 Ramanujan 相关矩阵。
\end{theorem}
\begin{proof}
若 \(g\in\mathcal C_M\) 写作 \(g(a/q)=c_q\)，则
\[
\|g\|_{\ell^2(\mathcal Q_M)}^2=\sum_{q\le M}\varphi(q)\,|c_q|^2,
\]
故得到自然加权同构。再由命题 \ref{prop:conclusion66-gram-ramanujan-decomposition} 的幅度平方恒等式，
\[
\langle g,K_{M,L}g\rangle
=
\sum_{n\in\ZZ}\widehat\psi(n/L)
\left|
\sum_{q\le M}\ \sum_{(a,q)=1} c_q e^{2\pi\mathrm{i}an/q}
\right|^2.
\]
对每个固定 \(q\)，内层和恰等于 Ramanujan 和 \(c_q(n)\)，故
\[
\langle g,K_{M,L}g\rangle
=
\sum_{n\in\ZZ}\widehat\psi(n/L)
\left|
\sum_{q\le M} c_q\,c_q(n)
\right|^2.
\]
展开平方并交换求和，即得所述矩阵表示。由于右端本身是带非负权 \(\widehat\psi(n/L)\) 的平方和，\(\mathcal R_{M,L}\) 自动正半定。证毕。
\end{proof}

\begin{theorem}[continuous phase 卷积核的 Fourier 精确对角化]\label{thm:conclusion-farey-limit-kernel-fourier-diagonalization}
\leanverified{paper\_conclusion\_farey\_limit\_kernel\_fourier\_diagonalization}
设
\[
\psi_\tau(t):=\sum_{n\in\ZZ}\widehat\psi(n/\tau)e^{2\pi\mathrm{i}nt},
\qquad
(\mathcal K_\tau h)(x):=\int_0^1 \psi_\tau(x-y)\,h(y)\,\dd y.
\]
则圆 Fourier 基
\[
e_n(x):=e^{2\pi\mathrm{i}n x}
\qquad (n\in\ZZ)
\]
是 \(\mathcal K_\tau\) 的完整正交特征基，且对应特征值恰为
\[
\mathcal K_\tau e_n=\widehat\psi(n/\tau)\,e_n.
\]
从而若 \(\widehat\psi\ge 0\)，则
\[
\|\mathcal K_\tau\|_{\mathrm{op}}
=
\sup_{n\in\ZZ}\widehat\psi(n/\tau).
\]
\end{theorem}
\begin{proof}
由定义，
\[
(\mathcal K_\tau e_n)(x)
=
\int_0^1\psi_\tau(x-y)e^{2\pi\mathrm{i}ny}\,\dd y.
\]
将 \(\psi_\tau\) 展开为 Fourier 级数，
\[
\psi_\tau(x-y)
=
\sum_{m\in\ZZ}\widehat\psi(m/\tau)e^{2\pi\mathrm{i}m(x-y)},
\]
便得到
\[
(\mathcal K_\tau e_n)(x)
=
\sum_{m\in\ZZ}\widehat\psi(m/\tau)e^{2\pi\mathrm{i}mx}
\int_0^1 e^{-2\pi\mathrm{i}my}e^{2\pi\mathrm{i}ny}\,\dd y.
\]
积分项因 Fourier 正交性等于 \(\delta_{mn}\)，故
\[
(\mathcal K_\tau e_n)(x)=\widehat\psi(n/\tau)e^{2\pi\mathrm{i}nx}.
\]
由 \(\{e_n\}_{n\in\ZZ}\) 在 \(L^2([0,1])\) 中构成完整正交基，结论随之成立。若 \(\widehat\psi\ge 0\)，则 \(\mathcal K_\tau\) 为正自伴算子，其算子范数等于最大特征值。证毕。
\end{proof}

\begin{corollary}[低温端的 rank-one 常量相塌缩]\label{cor:conclusion-farey-lowtemperature-rankone-constant-collapse}
若 \(\widehat\psi\) 连续且
\[
\widehat\psi(\xi)\to 0
\qquad (|\xi|\to\infty),
\]
则当 \(\tau\downarrow 0\) 时，
\[
\mathcal K_\tau\xrightarrow[\mathrm{op}]{}\widehat\psi(0)\,\Pi_{\mathbf 1},
\]
其中 \(\Pi_{\mathbf 1}\) 为到常量函数子空间的正交投影。于是 continuous phase 的低温端严格退化为秩一常量相。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-farey-limit-kernel-fourier-diagonalization}，\(\mathcal K_\tau\) 在 Fourier 基下的特征值为
\[
\lambda_n(\tau)=\widehat\psi(n/\tau).
\]
而 \(\widehat\psi(0)\Pi_{\mathbf 1}\) 的特征值为 \(\widehat\psi(0)\)（当 \(n=0\)）与 \(0\)（当 \(n\neq 0\)）。故两者之差的算子范数为
\[
\|\mathcal K_\tau-\widehat\psi(0)\Pi_{\mathbf 1}\|_{\mathrm{op}}
=
\sup_{n\neq 0}|\widehat\psi(n/\tau)|.
\]
对任意 \(n\neq 0\)，有 \(|n/\tau|\ge 1/\tau\to\infty\)，故
\[
\sup_{n\neq 0}|\widehat\psi(n/\tau)|
\le
\sup_{|\xi|\ge 1/\tau}|\widehat\psi(\xi)|
\longrightarrow 0.
\]
这就给出了算子范数收敛。证毕。
\end{proof}

\begin{theorem}[高温端在固定频带上的完全平坦化]\label{thm:conclusion-farey-hightemperature-fixedband-flattening}
对任意固定整数 \(H\ge 0\)，记有限频带
\[
\mathcal P_H:=\operatorname{span}\{e_n:|n|\le H\}\subset L^2([0,1]).
\]
若 \(\widehat\psi\) 在 \(0\) 连续，则
\[
\bigl\|(\mathcal K_\tau-\widehat\psi(0)I)|_{\mathcal P_H}\bigr\|_{\mathrm{op}}
=
\max_{|n|\le H}\bigl|\widehat\psi(n/\tau)-\widehat\psi(0)\bigr|
\longrightarrow 0
\qquad (\tau\to\infty).
\]
因此，continuous phase 的高温端在任何固定 Fourier 频带上都退化为标量算子，低频模态趋于完全平坦相。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-farey-limit-kernel-fourier-diagonalization}，\(\mathcal P_H\) 上的特征值恰为
\[
\widehat\psi(n/\tau),
\qquad |n|\le H.
\]
而 \(\widehat\psi(0)I\) 在同一子空间上的特征值恒为 \(\widehat\psi(0)\)。因此限制算子之差的范数就是有限个特征值差的最大值。对每个固定 \(|n|\le H\)，当 \(\tau\to\infty\) 时有 \(n/\tau\to 0\)，由 \(\widehat\psi\) 在 \(0\) 的连续性得结论。证毕。
\end{proof}

\begin{theorem}[continuous strict drop 诱导单尺度离散证书并传播到全尺度]\label{thm:conclusion-continuous-strict-drop-implies-discrete-finite-certificate}
\leanverified{paper\_conclusion\_continuous\_strict\_drop\_implies\_discrete\_finite\_certificate}
设 \(\widehat\psi\ge 0\)。若存在某个 \(\tau_0>0\) 与 \(\varepsilon>0\) 使
\[
\|\mathcal K_{\tau_0}\|_{\mathrm{op}}\le 1-\varepsilon,
\]
则存在 \(M_0\) 使对一切充分大的 \(M\)，只要
\[
L/M^2\to\tau_0,
\]
便有
\[
\|\widetilde K_{M,L}\|_{\mathrm{op}}\le 1-\varepsilon/2.
\]
因此，在固定 primitive sofic--权系统口径下，一个连续 phase 上的 strict spectral drop 会自动生成一个有限、离散、可审计的单尺度证书，并由尺度自举传播为全尺度统一谱隙。
\end{theorem}
\begin{proof}
由定理 \ref{thm:gm-farey-sampling-operator-limit}，在 \(L/M^2\to\tau_0\) 时有

\|\widetilde K_{M,L}\|_{\mathrm{op}}\longrightarrow \|\mathcal K_{\tau_0}\|_{\mathrm{op}}.
\]
于是由假设 \(\|\mathcal K_{\tau_0}\|_{\mathrm{op}}\le 1-\varepsilon\)，存在 \(M_0\) 使对 \(M\ge M_0\) 有
\[
\|\widetilde K_{M,L}\|_{\mathrm{op}}\le 1-\varepsilon/2.
\]
这给出单一尺度上的离散谱降证书。再由定理 \ref{thm:conclusion67-scale-bootstrap}，该单尺度谱降会传播到同型标度 \(L\asymp M^2\) 下的全尺度幂次衰减；而定理 \ref{thm:conclusion68-finite-verification} 又把统一扭曲谱隙归约为有限检查集上的核验。故 continuous phase 上的 strict drop 与离散全尺度证书之间被双向锁定。证毕。
\end{proof}

\begin{theorem}[离散 near-resonant 簇提取 continuous resonant phase]\label{thm:conclusion-discrete-nearresonance-extracts-continuous-resonant-phase}
设存在一列尺度对
\[
(M_j,L_j)\to\infty,
\qquad
\tau_j:=L_j/M_j^2\to\tau_\ast\in(0,\infty),
\]
使得
\[
\|\widetilde K_{M_j,L_j}\|_{\mathrm{op}}\longrightarrow 1.
\]
则必有
\[
\|\mathcal K_{\tau_\ast}\|_{\mathrm{op}}=1.
\]
更一般地，若一条离散反例序列阻止任何统一谱降成立，则其任意 accumulation phase \(\tau_\ast\) 都属于连续 resonant set
\[
\mathfrak R:=\{\tau>0:\ \|\mathcal K_\tau\|_{\mathrm{op}}\ge 1\}.
\]
因此，离散 near-resonance 并不会消散在高分辨率尾部，而必在某个 continuous phase 上沉淀为真正的 resonant witness。
\end{theorem}
\begin{proof}
由定理 \ref{thm:gm-farey-sampling-operator-limit} 的算子范数收敛，
\[
\|\widetilde K_{M_j,L_j}\|_{\mathrm{op}}
\longrightarrow
\|\mathcal K_{\tau_\ast}\|_{\mathrm{op}}
\qquad (j\to\infty).
\]
故由假设立即得到 \(\|\mathcal K_{\tau_\ast}\|_{\mathrm{op}}=1\)。若离散统一谱降失败，则可选取一列离散反例使其算子范数不趋于某个严格小于 \(1\) 的常数。对任一收敛子列 \(\tau_j\to \tau_\ast\)，同样的极限传递给出
\[
\|\mathcal K_{\tau_\ast}\|_{\mathrm{op}}\ge 1,
\]
从而 \(\tau_\ast\in\mathfrak R\)。证毕。
\end{proof}

\begin{theorem}[Hardy 旗标、Ramanujan 压缩与 continuous phase 提取的谱刚性链]\label{thm:conclusion-hardy-filtration-ramanujan-continuous-phase-rigidity-chain}
第 \(12.24\) 至 \(12.27\) 节中的 Hardy 截断、有理主弧 Gram 核、Farey 连续化与尺度自举，可被进一步压缩为如下严格闭链：
\[
\boxed{
\text{有限 Hardy 旗标}
\Longrightarrow
\text{exact filtration algebra}
\Longrightarrow
\text{Christoffel 完全平坦}
}
\]
\[
\Longrightarrow
\boxed{
\text{分母层常量子空间的 Ramanujan 压缩}
\Longrightarrow
\text{continuous phase 核的 Fourier 对角化}
\Longrightarrow
\tau\downarrow 0\ \text{时 rank-one 常量相塌缩}
\Longrightarrow
\tau\uparrow\infty\ \text{时固定频带完全平坦化}
}
\]
\[
\Longrightarrow
\boxed{
\text{continuous strict drop}
\Longrightarrow
\text{单尺度离散证书}
\Longrightarrow
\text{全尺度自举}
\Longleftrightarrow
\text{离散 near-resonance 提取 continuous resonant phase}.
}
\]
更准确地说：
\begin{enumerate}
  \item 有限模态 Hardy 塔不仅可由显式有理核实现，而且其嵌套投影之间已经携带 exact filtration algebra 与完全平坦的点值 Christoffel 函数；
  \item 主弧 Gram 二次型在分母层常量口径下严格压缩为有限维 Ramanujan 相关矩阵，从而把高维离散谱问题规约为可审计的有限矩阵问题；
  \item continuous phase 核在 Fourier 基下完全对角化，因此两端相态不再需要额外近似：低温端秩一塌缩与高温端固定频带平坦化都由特征值 \(\widehat\psi(n/\tau)\) 的极限直接读出；
  \item 一旦某个 continuous phase 出现 strict spectral drop，Farey 采样极限便将其转写为有限离散证书，并由尺度自举传播到全尺度；反过来，任何离散 near-resonant 反例簇都必沉淀为某个 continuous resonant phase。
\end{enumerate}
因此，这里真正被闭合的对象并不是某个单尺度大筛矩阵，而是一条由有限 Hardy 旗标、Ramanujan 分母层压缩、卷积型 continuous phase 与 bootstrap 机制共同组成的谱相变链。
\end{theorem}
