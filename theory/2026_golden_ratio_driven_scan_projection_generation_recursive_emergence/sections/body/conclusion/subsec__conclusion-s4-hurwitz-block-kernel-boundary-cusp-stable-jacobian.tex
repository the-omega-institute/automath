\begin{theorem}[块核 Sylow--\texorpdfstring{$3$}{3} 的规范分裂]\label{thm:conclusion-s4-block-kernel-sylow3-canonical-splitting}
\leanverified{paper\_conclusion\_s4\_block\_kernel\_sylow3\_canonical\_splitting}
在 \(160\) 点分解
\[
\bigsqcup_{j=1}^{40}B_j,\qquad |B_j|=4
\]
下，块核 \(K\) 的 Sylow--\(3\) 子群 \(P_3\) 满足
\[
P_3\cong (\ZZ/3\ZZ)^{40},
\]
并且存在唯一分解
\[
P_3=\prod_{j=1}^{40}P_{3,j},
\]
其中每个 \(P_{3,j}\cong \ZZ/3\ZZ\) 仅支撑在块 \(B_j\) 上（在 \(B_j\) 内为一个三循环，固定一元素），在其余块上恒等。
\end{theorem}
\begin{proof}
由结论 \ref{con:xi-terminal-zm-braid-monodromy-psp4f3-identification}，
\[
|K|=2^{96}\cdot 3^{40},
\]
故任一 Sylow--\(3\) 子群 \(P_3\) 的阶为 \(3^{40}\)。
又因 \(K\) 在块集合上作用平凡，存在自然忠实嵌入
\[
K\hookrightarrow \prod_{j=1}^{40}\mathfrak S(B_j)\cong \mathfrak S_4^{40}.
\]
于是对每个 \(j\)，投影 \(\pi_j(P_3)\) 是 \(\mathfrak S_4\) 的 \(3\)-子群，故
\(
|\pi_j(P_3)|\le 3
\)。
由忠实性，
\[
|P_3|
\le \prod_{j=1}^{40}|\pi_j(P_3)|
\le 3^{40}.
\]
与 \(|P_3|=3^{40}\) 比较可知上式处处取等，故每个 \(\pi_j(P_3)\) 阶恰为 \(3\)，并且
\[
P_3\cong \prod_{j=1}^{40}\pi_j(P_3)\cong (\ZZ/3\ZZ)^{40}.
\]
定义
\[
P_{3,j}:=P_3\cap \ker\!\Bigl(\prod_{i\neq j}\pi_i\Bigr),
\]
则 \(P_{3,j}\cong \ZZ/3\ZZ\)，且 \(P_3=\prod_j P_{3,j}\)。
该分解由块坐标投影唯一确定，故具有规范性。证毕。
\end{proof}

\begin{theorem}[两换位碰撞边界的三分型分解与分歧阶]\label{thm:conclusion-s4-two-transposition-collision-three-type-boundary}
设 \(\Delta\) 为“相邻两换位分支点碰撞”的一般边界分量，并令
\[
\tau:=\sigma_i^2
\]
为对应局部 braid 单值。则 \(S_4\)-Hurwitz 映射
\[
f:\mathcal H_{S_4}\to \mathcal M
\]
在 \(\Delta\) 一般点上满足：
\begin{enumerate}
  \item 归一化拉回 \(f^{-1}(\Delta)^\nu\) 恰分解为 \(88\) 个几何不可约余维 \(1\) 分量；
  \item 分量分为三类：
  \[
  \text{A 型 }28\text{ 个},\quad
  \text{B 型 }24\text{ 个},\quad
  \text{C 型 }36\text{ 个};
  \]
  \item 分歧指数分别为
  \[
  e_{\mathrm A}=1,\qquad e_{\mathrm B}=1,\qquad e_{\mathrm C}=3.
  \]
\end{enumerate}
其中 A/B/C 三型可按 \(p:=g_ig_{i+1}\) 的循环型区分：A 对应 \(p=1\)，B 对应 \(p\) 为双换位，C 对应 \(p\) 为三循环。
\end{theorem}
\begin{proof}
命题 \ref{prop:xi-terminal-zm-hurwitz-generator-cycle-types-on-160-and-40} 给出
\[
\sigma_i\sim 3^{36}2^{12}1^{28}\quad\text{（在 \(160\) 点 Nielsen 纤维上）}.
\]
对 \(\tau=\sigma_i^2\)：
\begin{itemize}
  \item \(1\)-循环保持为 \(1\)-循环，贡献 \(28\)；
  \item 每个 \(2\)-循环平方后分裂为两个 \(1\)-循环，共贡献 \(2\cdot 12=24\)；
  \item 每个 \(3\)-循环平方后仍为 \(3\)-循环，贡献 \(36\)。
\end{itemize}
故
\[
\tau\sim 1^{52}3^{36},
\]
其循环总数为 \(52+36=88\)，即对应 \(88\) 个不可约边界分量。

局部 Hurwitz 理论中，\(\tau\) 的循环长度即映射到 \(\Delta\) 的局部分歧指数：
固定点对应 \(e=1\)，三循环对应 \(e=3\)。
因此 \(36\) 个 C 型分量为三次分歧，其余 \(28+24\) 个分量无分歧。
再由相邻换位对 \((g_i,g_{i+1})\) 的局部分类：
\(g_i=g_{i+1}\) 给出 \(p=1\)（A 型），
互不相交换位给出双换位积（B 型），
相交换位给出三循环积（C 型），得到三型标记。证毕。
\end{proof}

证毕。
\end{proof}

\begin{theorem}[三 cusp 型半稳定分裂模型的归一化组合型、节点数与主分量属数]\label{thm:conclusion-s4-cusp-stable-limit-normalization-node-genus}
设 \(X\) 为本节 \(S_4\)-正则一般纤维，\(g(X)=16\)（见结论 \ref{con:xi-terminal-zm-s4-galois-closure-genus-tower-holomorphic-differentials}）。
在定理 \ref{thm:conclusion-s4-two-transposition-collision-three-type-boundary} 的三类边界点处，令
\[
X_0^{\mathrm{ss}}
\]
为由 admissible cover 给出的半稳定分裂模型，则其归一化组合型满足：
\begin{enumerate}
  \item A 型（\(p=1,\ H\cong C_2\)）：
  \[
  \#\text{尾分量}=12,\quad
  \#\text{节点}=24,\quad
  g(X_{\mathrm{main}})=4;
  \]
  \item B 型（\(|p|=2,\ H\cong V_4\)）：
  \[
  \#\text{尾分量}=6,\quad
  \#\text{节点}=12,\quad
  g(X_{\mathrm{main}})=10;
  \]
  \item C 型（\(|p|=3,\ H\cong S_3\)）：
  \[
  \#\text{尾分量}=4,\quad
  \#\text{节点}=8,\quad
  g(X_{\mathrm{main}})=12.
  \]
\end{enumerate}
并且三型中每个尾分量均与主分量通过恰好 \(2\) 个节点连接。
若 \(\Gamma_{X_0^{\mathrm{ss}}}\) 为对偶图，则
\[
b_1(\Gamma_{X_0^{\mathrm{ss}}})=
\begin{cases}
12,&\text{A 型},\\
6,&\text{B 型},\\
4,&\text{C 型},
\end{cases}
\qquad
g(X_0^{\mathrm{ss}})=g(X_{\mathrm{main}})+b_1(\Gamma_{X_0^{\mathrm{ss}}})=16.
\]
\end{theorem}
\begin{proof}
设 \(G=S_4\)，令 \(p=g_ig_{i+1}\)，\(r:=|p|\in\{1,2,3\}\)，
并记尾部局部单值群 \(H:=\langle g_i,g_{i+1}\rangle\)。
对两换位生成子群，三种可能恰为
\[
(H,r)=(C_2,1),\ (V_4,2),\ (S_3,3).
\]
可容许覆盖的局部群论给出：
\begin{itemize}
  \item 尾分量数为指数 \([G:H]=24/|H|\)，故分别为 \(12,6,4\)；
  \item 节点惯性群阶为 \(r\)，主分量上节点预像数为 \(|G|/r=24/r\)，故分别为 \(24,12,8\)；
  \item 每个尾分量上的节点预像数为 \(|H|/r\)，三型均等于 \(2\)。
\end{itemize}
于是每个尾分量与主分量恰连两条边。

对偶图边数 \(E\) 等于节点数，顶点数 \(V=1+[G:H]\)（一个主分量加尾分量），
故
\[
b_1=E-V+1=
\begin{cases}
24-13+1=12,\\
12-7+1=6,\\
8-5+1=4.
\end{cases}
\]
三型尾分量均为三点分歧 \(H\)-覆盖的正规化，亏格为 \(0\)；
故稳定曲线算术亏格满足
\[
g(X_0^{\mathrm{ss}})=g(X_{\mathrm{main}})+b_1(\Gamma_{X_0^{\mathrm{ss}}}).
\]
又 \(g(X_0^{\mathrm{ss}})=g(X)=16\)（平坦退化下算术亏格守恒），遂得主分量属数分别为 \(4,10,12\)。证毕。
\end{proof}

\begin{proposition}[三 cusp 型尾部覆盖的三点分歧刚性与边界模量来源]\label{prop:conclusion-s4-cusp-tail-three-point-rigidity-and-boundary-moduli}

  \item A 型：\(H=C_2\)，分歧型 \((2,2,1)\)；
  \item B 型：\(H=V_4\)，分歧型 \((2,2,2)\)（对应三个非平凡元）；
  \item C 型：\(H=S_3\)，分歧型 \((2,2,3)\)。
\end{enumerate}
因此边界不可约分量的连续参数完全来自主分量上的分支与节点标记数据；尾部仅贡献离散多重度。
\end{proposition}

\((h_1,h_2,h_3)\in H^3\)（满足 \(h_1h_2h_3=1\) 且生成 \(H\)）在同时共轭与三分支点重标记下分类。

  \item \(S_3\) 的 \((2,2,3)\) 生成三元组在同时共轭下亦为唯一轨道。
\end{itemize}
故三型尾部皆刚性。
可容许覆盖边界局部是“主部数据 \(\times\) 尾部离散选择”的纤维化拼接，
既然尾部无连续模量，边界维数仅由主部分支与节点相对位置给出。证毕。
\end{proof}

\begin{theorem}[转置碰撞边界稳定化的不可约结点模型与尾群指数公式]\label{thm:conclusion-s4-cusp-stable-contraction-irreducible-nodal-index}
设 \(G=S_4\)，\(f:X\to \PP^1\) 为本节固定的 \(G\)-正则分歧覆盖，\(g(X)=16\)。
对任一相邻转置碰撞的一般边界分量，令
\[
H:=\langle g_i,g_{i+1}\rangle,\qquad p:=g_ig_{i+1},\qquad r:=|p|.
\]
记 \(\overline X_0\) 为对应稳定模型，则：
\begin{enumerate}
  \item \(\overline X_0\) 不可约，且全部结点均为非分离结点；
  \item 结点数满足
  \[
  \delta=[S_4:H];
  \]
  \item 归一化 \(\nu:\widetilde X_0\to \overline X_0\) 为光滑曲线，且
  \[
  g(\widetilde X_0)=16-\delta.
  \]
  因而三型对应
  \[
  (H,\delta,g(\widetilde X_0))
  \in\{(C_2,12,4),\ (V_4,6,10),\ (S_3,4,12)\}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-s4-cusp-stable-limit-normalization-node-genus}，
半稳定模型 \(X_0^{\mathrm{ss}}\) 由一条主分量与 \([S_4:H]\) 条尾有理分量组成，
每条尾分量恰以两个结点附着于主分量。
两点附着的有理分量在稳定化中均被收缩，且每收缩一条尾分量就在主分量上识别一对点，
产生一个非分离结点。
故稳定模型 \(\overline X_0\) 仅剩主分量并带
\(
\delta=[S_4:H]
\)
个非分离结点，特别地其不可约。

收缩尾分量不改变主分量的归一化，故
\(
\widetilde X_0\cong X_{\mathrm{main}}
\)。
又算术亏格守恒且 \(g(\overline X_0)=16\)，对不可约 \(\delta\)-结点曲线有
\[
g(\overline X_0)=g(\widetilde X_0)+\delta,
\]
从而 \(g(\widetilde X_0)=16-\delta\)。
代入三型 \(\delta=12,6,4\) 即得所列三元组。证毕。
\end{proof}

稳定模型有 \(\delta\) 个非分离结点，对应 \(\delta\) 个线性无关消失圈，

分歧指数由定理
\ref{thm:conclusion-s4-two-transposition-collision-three-type-boundary}
直接给出：A/B 型 \(e=1\)，C 型 \(e=3\)。

故有限部分阶为 \(3\)，并且 \(T^3\) 幂单。证毕。
\end{proof}

\begin{theorem}[40 块秩 \(3\) 作用诱导的强正则图参数与谱唯一性]\label{thm:conclusion-s4-40blocks-rank3-strongly-regular-parameters}
\leanverified{paper\_conclusion\_s4\_40blocks\_rank3\_strongly\_regular\_parameters}

故任一非平凡轨道关系图均为强正则图；
此处 \(v=40,k=12\)。
强正则参数满足
\[
(v-k-1)\mu=k(k-\lambda-1),
\]
代入得
\[
27\mu=12(11-\lambda)\iff 4\lambda+9\mu=44.
\]
由整数条件与 \(0\le \lambda<k\) 可得候选对
\((\lambda,\mu)=(11,0)\) 或 \((2,4)\)。
其中 \(\mu=0\) 时图必为若干 \(K_{13}\) 的不交并，与 \(v=40\) 不相容，
故唯一可行解为 \((\lambda,\mu)=(2,4)\)。

非平凡特征值由方程
\[
x^2+(\mu-\lambda)x+(\mu-k)=0
\]
得到 \(x=2,-4\)。
再由迹与平方和关系
\[
1+m_2+m_{-4}=40,\qquad
12+2m_2-4m_{-4}=0
\]
解得 \(m_2=24,\ m_{-4}=15\)。证毕。
\end{proof}

\begin{theorem}[邻接算子生成的交换子代数与显式中心幂等元]\label{thm:conclusion-s4-40blocks-commutant-idempotents-explicit}
沿定理 \ref{thm:conclusion-s4-40blocks-rank3-strongly-regular-parameters}，
令 \(V=\QQ^\Omega\)，\(A\in \operatorname{End}_\QQ(V)\) 为邻接算子。
则：
\begin{enumerate}
  \item 作为 \(\QQ[G]\)-模，
  \[
  V\cong \QQ\oplus V_{24}\oplus V_{15},
  \]
  其中 \(\dim V_{24}=24,\ \dim V_{15}=15\)；
  \item 交换子代数满足
  \[
  \operatorname{End}_G(V)=\QQ[A];
  \]
  \item 对应特征值 \(12,2,-4\) 的中心幂等元可取为
  \[
  e_{\mathrm{triv}}=\frac{(A-2I)(A+4I)}{160},\quad
  e_{24}=-\frac{(A-12I)(A+4I)}{60},\quad
  e_{15}=\frac{(A-12I)(A-2I)}{96},
  \]
  且
  \[
  e_{\mathrm{triv}}+e_{24}+e_{15}=I,\qquad
  e_\bullet^2=e_\bullet,\qquad
  e_\bullet e_{\bullet'}=0\ (\bullet\neq \bullet').
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
秩 \(3\) 传递作用的轨道代数维数为 \(3\)，故
\(
\dim \operatorname{End}_G(V)=3
\)。
由定理
\ref{thm:conclusion-s4-40blocks-rank3-strongly-regular-parameters}
可知 \(A\) 有三个互异有理特征值 \(12,2,-4\)，从而最小多项式为
\[
m_A(x)=(x-12)(x-2)(x+4),
\]
故 \(\dim \QQ[A]=3\)。于是 \(\operatorname{End}_G(V)=\QQ[A]\)。

特征值互异给出谱分解
\(
V=V_{12}\oplus V_2\oplus V_{-4}
\)，其维数分别为 \(1,24,15\)。
将拉格朗日插值投影公式应用于 \(A\) 即得上述三幂等元，且像分别为
\(
V_{12},V_2,V_{-4}
\)。
记
\(
V_{24}:=V_2,\ V_{15}:=V_{-4}
\)，结论成立。证毕。
\end{proof}

\begin{theorem}[40 重有限 étale 覆盖的推前局部系统分解与对应代数生成]\label{thm:conclusion-s4-etalecover-pushforward-local-system-decomposition-correspondence}
设
\[
f:\mathcal H_{S_3}\longrightarrow \mathcal M_2
\]
为选择属 \(2\) 曲线 Jacobian 上循环阶 \(3\) 子群所给的 \(40\) 重有限 étale 覆盖
（在特征 \(0\) 或避开 \(2,3\) 的驯良情形）。
其几何单值群同构于 \(G\cong PSp_4(\mathbf F_3)\)。
则推前局部系统分解为
\[
f_*\QQ\cong \QQ\oplus \mathcal V_{24}\oplus \mathcal V_{15},
\]
其中 \(\operatorname{rank}\mathcal V_{24}=24,\ \operatorname{rank}\mathcal V_{15}=15\)。

令
\[
\mathcal C\subset \mathcal H_{S_3}\times_{\mathcal M_2}\mathcal H_{S_3}
\]
为由 \(12\)-邻接子轨道定义的代数对应，则
\[
\operatorname{End}_{\mathrm{Loc}(\mathcal M_2,\QQ)}(f_*\QQ)=\QQ[\mathcal C].
\]
并且定理 \ref{thm:conclusion-s4-40blocks-commutant-idempotents-explicit}
中的 \(e_{24},e_{15}\) 给出两条非平凡投影对应，
分别切出 \(\mathcal V_{24}\) 与 \(\mathcal V_{15}\)。
\end{theorem}
\begin{proof}
局部系统 \(f_*\QQ\) 的几何纤维是 \(40\) 点集合上的置换表示，
其单值表示即为定理
\ref{thm:conclusion-s4-40blocks-commutant-idempotents-explicit}
中的 \(V\)。
因此纤维上的 \(\QQ[G]\)-分解
\(
V\cong \QQ\oplus V_{24}\oplus V_{15}
\)
提升为局部系统直和分解
\(
f_*\QQ\cong \QQ\oplus \mathcal V_{24}\oplus \mathcal V_{15}
\)。

对应 \(\mathcal C\) 在每个几何纤维上的作用即邻接算子 \(A\)，
故其生成的端同态代数纤维上为 \(\QQ[A]\)。
由定理
\ref{thm:conclusion-s4-40blocks-commutant-idempotents-explicit}
知 \(\QQ[A]=\operatorname{End}_G(V)\)，遂得到全局
\[
\operatorname{End}_{\mathrm{Loc}(\mathcal M_2,\QQ)}(f_*\QQ)=\QQ[\mathcal C].
\]
将 \(e_{24},e_{15}\) 视为 \(\mathcal C\) 的有理多项式组合，
即得两条投影对应并切出相应局部系统因子。证毕。
\end{proof}
