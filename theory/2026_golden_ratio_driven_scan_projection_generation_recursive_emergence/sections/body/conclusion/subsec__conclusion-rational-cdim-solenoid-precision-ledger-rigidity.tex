\subsection{有理圆维延拓、Lie--solenoid 分岔与素账本最小支撑}\label{subsec:conclusion-rational-cdim-solenoid-precision-ledger-rigidity}
本小节以定义 \ref{def:circle-dimension} 与定义 \ref{def:half-circle-dimension} 为结构层基准，
并沿用相位门 Jacobian 诱导的精度势账本解释。
文中“编码”均指实现层可逆化机制，不作为结构同构或圆维塌缩的替代。

\begin{definition}[有限有理秩阿贝尔类与有理圆维]\label{def:conclusion-abfr-cdimq}
记
\[
\mathrm{Ab}_{\mathrm{fr}}
:=
\left\{
A\ \text{abelian}:\ \dim_{\QQ}(A\otimes_{\ZZ}\QQ)<\infty
\right\}.
\]
对 \(A\in\mathrm{Ab}_{\mathrm{fr}}\) 定义
\[
\cdim_{\QQ}(A):=\dim_{\QQ}(A\otimes_{\ZZ}\QQ).
\]
\end{definition}

\begin{proposition}[有理圆维的唯一短正合可加延拓]\label{prop:conclusion-cdimq-unique-exact-extension}
沿用定义 \ref{def:conclusion-abfr-cdimq}。则：
\begin{enumerate}
  \item 若 \(T\) 为挠群，则 \(\cdim_{\QQ}(T)=0\)；
  \item 对任意短正合列
  \[
  0\to A'\to A\to A''\to 0
  \]
  （对象均在 \(\mathrm{Ab}_{\mathrm{fr}}\) 中），有
  \[
  \cdim_{\QQ}(A)=\cdim_{\QQ}(A')+\cdim_{\QQ}(A'');
  \]
  \item 若 \(G\) 为有限生成阿贝尔群，则
  \[
  \cdim_{\QQ}(G)=\operatorname{rank}(G)=\cdim(G).
  \]
\end{enumerate}
并且 \(\cdim_{\QQ}\) 具有唯一性：若
\[
\delta:\mathrm{Ab}_{\mathrm{fr}}\to \NN\cup\{0\}
\]
满足
\begin{enumerate}
  \item 对挠群 \(T\) 有 \(\delta(T)=0\)；
  \item 对短正合列可加；
  \item \(\delta(\ZZ)=1\)；
\end{enumerate}
则 \(\delta(A)=\cdim_{\QQ}(A)\) 对所有 \(A\in\mathrm{Ab}_{\mathrm{fr}}\) 成立。
\end{proposition}
\begin{proof}
若 \(T\) 挠，则 \(T\otimes_{\ZZ}\QQ=0\)，故第 (1) 条成立。
又因 \(\QQ\) 为 \(\ZZ\)-平坦模，张量保持短正合，故
\(\dim_{\QQ}\) 对短正合列可加，得第 (2) 条。
第 (3) 条由有限生成分解
\(
G\simeq \ZZ^r\oplus F
\)
（\(F\) 有限）立即得到：
\(
G\otimes\QQ\simeq \QQ^r
\)，且 \(\cdim(G)=r\)（定义 \ref{def:circle-dimension}）。

下证唯一性。取 \(A\in\mathrm{Ab}_{\mathrm{fr}}\)，记 \(T:=\mathrm{Tor}(A)\)。
由短正合
\[
0\to T\to A\to A/T\to 0
\]
与条件 (1)(2) 得 \(\delta(A)=\delta(A/T)\)。
故仅需考虑无挠有限秩群 \(A\)。
令
\[
r:=\dim_{\QQ}(A\otimes_{\ZZ}\QQ).
\]
取 \(a_1,\dots,a_r\in A\) 使 \((a_i\otimes 1)\) 为 \(A\otimes\QQ\) 的一组 \(\QQ\)-基，
设
\[
B:=\sum_{i=1}^r \ZZ a_i\cong \ZZ^r.
\]
则 \((A/B)\otimes\QQ=0\)，故 \(A/B\) 挠。
再由
\[
0\to B\to A\to A/B\to 0
\]
与条件 (1)(2)：
\[
\delta(A)=\delta(B)+\delta(A/B)=\delta(\ZZ^r)=r.
\]
其中 \(\delta(\ZZ^r)=r\delta(\ZZ)=r\)。
故 \(\delta(A)=r=\cdim_{\QQ}(A)\)。证毕。
\end{proof}

\begin{theorem}[素账本的 Lie--solenoid 二分律]\label{thm:conclusion-lie-solenoid-prime-ledger-dichotomy}
设 \(p\) 为素数。
\begin{enumerate}
  \item 不存在连续单射群同态
  \[
  \iota:\ZZ_p\hookrightarrow \TT^n\qquad(n<\infty).
  \]
  因而在“有限维紧连通 Lie 阿贝尔相位寄存器”约束下，无限 \(p\)-进账本不可内化为有限个相位圆维。
  \item 存在紧连通阿贝尔群 \(S_p\) 与短正合列
  \[
  0\to \ZZ_p\to S_p\to \TT\to 0.
  \]
  可取
  \[
  S_p:=\widehat{\ZZ[1/p]}.
  \]
  该 \(S_p\) 连通但非 Lie。
  \item 若以 Pontryagin 对偶定义紧群的有理圆维
  \[
  \cdim_{\QQ}^{\vee}(K):=\dim_{\QQ}\!\bigl(\widehat K\otimes_{\ZZ}\QQ\bigr),
  \]
  则
  \[
  \cdim_{\QQ}^{\vee}(S_p)=1.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
若存在连续单射 \(\iota:\ZZ_p\hookrightarrow \TT^n\)，则其像闭且 \(\iota\) 为紧群嵌入。
对偶化得满射
\[
\widehat\iota:\widehat{\TT^n}\simeq \ZZ^n\twoheadrightarrow \widehat{\ZZ_p}.
\]
而
\[
\widehat{\ZZ_p}\simeq C_{p^\infty}
\]
（Pr\"ufer \(p\)-群），非有限生成。
但 \(\ZZ^n\) 的任何商群均有限生成，矛盾，得第 (1) 条。

对第 (2) 条，先取离散侧短正合
\[
0\to \ZZ\to \ZZ[1/p]\to \ZZ[1/p]/\ZZ\to 0,
\]
且 \(\ZZ[1/p]/\ZZ\simeq C_{p^\infty}\)。
对偶化得到
\[
0\to \widehat{C_{p^\infty}}\to \widehat{\ZZ[1/p]}\to \widehat{\ZZ}\to 0,
\]
即
\[
0\to \ZZ_p\to S_p\to \TT\to 0.
\]
由于 \(\ZZ[1/p]\) 无挠，\(S_p\) 连通；又 \(\widehat{S_p}=\ZZ[1/p]\) 非有限生成，
故 \(S_p\) 非紧连通 Lie 阿贝尔群（其对偶应为有限生成自由阿贝尔群）。

第 (3) 条由
\[
\widehat{S_p}=\ZZ[1/p],\qquad
\ZZ[1/p]\otimes_{\ZZ}\QQ\simeq \QQ
\]
直接得到。证毕。
\end{proof}

\begin{definition}[秩一无挠群的高度签名]\label{def:conclusion-rank1-height-signature}
称可数阿贝尔群 \(A\) 为\textbf{秩一无挠}，若 \(A\) 无挠且
\[
\cdim_{\QQ}(A)=\dim_{\QQ}(A\otimes_{\ZZ}\QQ)=1.
\]
在同构意义下可将 \(A\) 缩放嵌入为
\(
\ZZ\subseteq A\subseteq \QQ
\)。
在该归一化下，对每个素数 \(p\) 定义其\textbf{高度}
\[
h_p(A):=\sup\bigl(\{k\in\NN:\ p^{-k}\in A\}\cup\{0\}\bigr)\in(\NN\cup\{0\})\cup\{\infty\},
\]
并称 \(\Sigma(A):=(h_p(A))_{p}\) 为 \(A\) 的高度签名。
\end{definition}

\begin{theorem}[秩一无挠群的高度刚性与一维 solenoid 对偶]\label{thm:conclusion-rank1-height-solenoid-dual}
\leanverified{paper\_conclusion\_rank1\_height\_solenoid\_dual}
令 \(A\) 为秩一无挠可数阿贝尔群。
则存在整数链 \(d_1\mid d_2\mid\cdots\) 使
\[
A=\bigcup_{j\ge 1}\frac{1}{d_j}\ZZ,
\qquad
v_p(d_j)\uparrow h_p(A)\ \ (p\ \text{任意素数}),
\]
并且其 Pontryagin 对偶具有一维 solenoid 的逆极限表示
\[
\widehat{A}\ \cong\ \varprojlim_j\Bigl(\TT \xleftarrow{\,z\mapsto z^{d_{j+1}/d_j}\,}\TT\Bigr).
\]
此外，在归一化 \(\ZZ\subseteq A\subseteq\QQ\) 下，\(\Sigma(A)\) 完全决定 \(A\) 的同构类：若 \(A,B\) 均为秩一无挠且 \(\Sigma(A)=\Sigma(B)\)，则 \(A\cong B\)。
\end{theorem}
\begin{proof}
取 \(A\) 的可数生成集合并将每个元素写为既约分数，取其分母集合的递增闭包即可构造整数链 \(d_1\mid d_2\mid\cdots\) 使
\(
A=\bigcup_j d_j^{-1}\ZZ
\)。
按定义 \(p^{-k}\in A\) 当且仅当存在 \(j\) 使 \(p^k\mid d_j\)，故 \(v_p(d_j)\) 的上确界恰为 \(h_p(A)\)，并可取为单调趋近该上确界的链。
\par
对偶表示：对任意 \(d\ge 1\)，离散群 \(d^{-1}\ZZ\) 的 Pontryagin 对偶同构于 \(\TT\)（字符由 \(t\mapsto e^{2\pi i t x}\) 给出）。
包含 \(d_j^{-1}\ZZ\hookrightarrow d_{j+1}^{-1}\ZZ\) 在对偶侧对应到覆盖映射
\(
\TT\to\TT,\ z\mapsto z^{d_{j+1}/d_j}
\)。
由于 Pontryagin 对偶把直极限送到逆极限，得到
\(
\widehat{A}\cong \varprojlim_j(\TT\xleftarrow{z\mapsto z^{d_{j+1}/d_j}}\TT)
\)。
\par
最后证高度刚性。在归一化 \(\ZZ\subseteq A\subseteq\QQ\) 下，\(A\) 等于
\[
\{x\in\QQ:\ v_p(x)\ge -h_p(A)\ \text{对所有 }p\},
\]
其中约定 \(h_p(A)=\infty\) 表示对该素数方向不施加下界约束。
因此 \(\Sigma(A)\) 逐素数刻画了允许出现的分母幂次上界，从而完全决定该 \(\QQ\) 的子群；若 \(\Sigma(A)=\Sigma(B)\) 则 \(A=B\)，从而 \(A\cong B\)。证毕。
\end{proof}

\begin{corollary}[Fibonacci 覆盖塔的逆极限溶子与 \texorpdfstring{$\widehat{\QQ}$}{Q-hat} 同构]\label{cor:conclusion-fibonacci-solenoid-dual-q}
\leanverified{paper\_conclusion\_fibonacci\_solenoid\_dual\_q}
令 \(\TT=\RR/\ZZ\)，并令 \(d_n:=F_{n+2}\)（\(F_n\) 为 Fibonacci 数列）。
考虑覆盖同态
\[
\varphi_n:\TT\to\TT,\qquad x\longmapsto d_n x\pmod 1,
\]
并定义逆极限紧交换群
\[
S_F:=\varprojlim(\TT,\varphi_n).
\]
则其 Pontryagin 对偶为直极限
\[
\widehat{S_F}\ \cong\ \varinjlim(\ZZ,\times d_n)
=\bigcup_{N\ge 1}\frac1{D_N}\ZZ,
\qquad D_N:=\prod_{n=0}^{N-1}d_n,
\]
并且 \(\widehat{S_F}\cong\QQ\)。
从而 \(S_F\cong \widehat{\QQ}\)；在局部紧交换群的对偶范畴中亦可识别为
\(
S_F\cong \mathbb A_\QQ/\QQ
\)。
\end{corollary}

\begin{proof}
逆极限与 Pontryagin 对偶把乘法覆盖 \(\varphi_n\) 送到 \(\ZZ\xrightarrow{\times d_n}\ZZ\) 的直极限，得到第一式。
下证该直极限等于 \(\QQ\)。
对任意素数 \(p\)，Fibonacci 数列在模 \(p\) 的状态对 \((F_n,F_{n+1})\in(\ZZ/p\ZZ)^2\) 上由递推确定且可逆，故从初态 \((0,1)\) 出发必回到 \((0,1)\)，从而存在 \(m\ge 1\) 使 \(F_m\equiv 0\pmod p\)。
因此 \(p\) 整除无穷多个 \(d_n\)，从而在乘积 \(D_N\) 中的 \(p\)-进赋值随 \(N\to\infty\) 发散。
于是对任意 \(k\ge 1\) 存在 \(N\) 使 \(p^k\mid D_N\)，即 \(p^{-k}\in \bigcup_N D_N^{-1}\ZZ\)。
由于对一切素数 \(p\) 均如此，该并集包含所有有限分母的有理数，故等于 \(\QQ\)。
最后的识别由 \(\widehat{\QQ}\cong \mathbb A_\QQ/\QQ\) 的标准 Pontryagin 对偶结论得到。证毕。
\end{proof}

\begin{theorem}[乘法自同构在 \texorpdfstring{$\widehat{\QQ}$}{Q-hat} 上的遍历性与强混合]\label{thm:conclusion-qhat-multiplication-ergodic-mixing}
令 \(S_F\) 如推论 \ref{cor:conclusion-fibonacci-solenoid-dual-q}，并以 \(\mu\) 记其 Haar 概率测度。
对任意整数 \(n\ge 2\)，令
\[
T_n:S_F\to S_F
\]
为与 \(\QQ\) 上的乘法自同构 \(q\mapsto nq\) 对偶的紧群自同构。
则 \(T_n\) 关于 \(\mu\) 遍历且强混合。
因此对任意 \(f\in L^1(S_F,\mu)\)，Birkhoff 平均几乎处处收敛：
\[
\lim_{N\to\infty}\frac1N\sum_{k=0}^{N-1} f\circ T_n^k
=\int_{S_F} f\,d\mu.
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:conclusion-fibonacci-solenoid-dual-q}，字符群 \(\widehat{S_F}\cong\QQ\)，并且 \(\{\chi_q:q\in\QQ\}\) 在 \(L^2(S_F,\mu)\) 中构成正交基。
对偶作用满足 \(\chi_q\circ T_n=\chi_{nq}\)。
\par
遍历性：若 \(g\in L^2\) 满足 \(g\circ T_n=g\)，写 Fourier 展开 \(g=\sum_q \widehat g(q)\chi_q\)。
则 \(\widehat g(q)=\widehat g(nq)\) 对所有 \(q\) 成立。
对任意 \(q\neq 0\)，轨道 \(\{n^k q:k\ge 0\}\) 无穷且互异；
而 \(\sum_q|\widehat g(q)|^2<\infty\)，故只能有 \(\widehat g(q)=0\)（否则在无穷互异频率上出现相同的非零系数，违反平方可和性）。
因此 \(g\) 仅余常数项，遍历性成立。
\par
混合性：对任意 \(q,q'\in\QQ\)，有
\[
\langle \chi_q\circ T_n^k,\chi_{q'}\rangle
=\langle \chi_{n^k q},\chi_{q'}\rangle
=\ind_{\{n^k q=q'\}}.
\]
当 \(q\neq 0\) 或 \(q'\neq 0\) 时，等式 \(n^k q=q'\) 最多在至多一个 \(k\) 上成立，故内积在 \(k\to\infty\) 时趋于 \(0\)；
再由三角多项式在 \(L^2\) 的稠密性得到对一般 \(L^2\) 函数的强混合。
最后的时间平均收敛由遍历性与 Birkhoff 定理推出。证毕。
\end{proof}

\begin{remark}[圆维含义：无限素数方向信息不提升相位圆维]\label{rem:conclusion-rank1-solenoid-circle-dimension}
对紧阿贝尔群 \(K\) 记 \(\cdim_{\QQ}^{\vee}(K)=\dim_{\QQ}(\widehat K\otimes_{\ZZ}\QQ)\)（见定理 \ref{thm:conclusion-lie-solenoid-prime-ledger-dichotomy}(3)）。
由 \(\widehat{\widehat{A}}=A\) 与 \(\cdim_{\QQ}(A)=1\) 得
\[
\cdim_{\QQ}^{\vee}(\widehat A)=1,
\]
即便高度签名 \(\Sigma(A)\) 在素数方向携带无限信息，其结构圆维仍保持为一维；无限性被严格容纳在逆极限纤维的 profinite/solenoidal 结构中，而非表现为无穷多个相位圆因子。
\end{remark}
\paragraph{高度序列与归一化周期模}\label{para:conclusion-solenoid-height-normalized-period-module}
定理 \ref{thm:conclusion-rank1-height-solenoid-dual} 表明，一维 solenoid 的离散侧分类信息可编码为高度序列 \(\Sigma(A)\)。
在有限覆盖逆塔的层无关积分口径（定理 \ref{thm:app-normalized-stokes-finite-cover-inverse-tower}）下，该分类参数亦可由单一“归一化斯托克斯周期模”完全恢复。

\begin{theorem}[归一化斯托克斯周期模恢复高度签名]\label{thm:conclusion-rank1-solenoid-normalized-stokes-period-module-recovers-height}
\leanverified{paper\_conclusion\_rank1\_solenoid\_normalized\_stokes\_period\_module\_recovers\_height}
令 \(A\) 为秩一无挠可数阿贝尔群，并取定理 \ref{thm:conclusion-rank1-height-solenoid-dual} 的整数链 \(d_1\mid d_2\mid\cdots\) 与表示
\[
\widehat{A}\ \cong\ \varprojlim_j\Bigl(\TT \xleftarrow{\,z\mapsto z^{d_{j+1}/d_j}\,}\TT\Bigr).
\]
在 \(\TT\simeq\RR/\ZZ\) 上取标准平移不变闭 \(1\)-形式 \(\mathrm d\theta\)（其周期群为 \(\ZZ\)）。
定义 \(\widehat{A}\) 的\emph{归一化斯托克斯周期模}
\[
\Pi_{\mathrm{St}}(\widehat{A})
:=
\bigcup_{j\ge 1}\left\{\frac1{d_j}\int_{\gamma}\mathrm d\theta:\ \gamma\ \text{为 }\TT\text{ 上的分段光滑闭曲线}\right\}
\subseteq \QQ\subseteq\RR.
\]
则 \(\Pi_{\mathrm{St}}(\widehat{A})=A\)（在归一化 \(\ZZ\subseteq A\subseteq\QQ\) 下）。
并且对每个素数 \(p\)，高度由周期模完全恢复：
\[
h_p(A)=\sup\left(\{k\in\NN:\ p^{-k}\in \Pi_{\mathrm{St}}(\widehat{A})\}\cup\{0\}\right)\quad(\text{允许 }=\infty).
\]
\end{theorem}

\begin{proof}
任意闭曲线 \(\gamma\) 在 \(\TT\) 上的绕数 \(\deg(\gamma)\in\ZZ\) 满足
\(
\int_\gamma \mathrm d\theta=\deg(\gamma)
\)。
故对每个 \(j\)，集合
\(
\left\{\frac1{d_j}\int_\gamma \mathrm d\theta\right\}
=d_j^{-1}\ZZ
\)。
取并得
\[
\Pi_{\mathrm{St}}(\widehat{A})=\bigcup_{j\ge 1}d_j^{-1}\ZZ=A,
\]
其中最后一步使用定理 \ref{thm:conclusion-rank1-height-solenoid-dual} 的离散侧刻画。
最后，\(p^{-k}\in A\) 当且仅当 \(k\le h_p(A)\)，由此即得恢复式。证毕。
\end{proof}

\begin{theorem}[profinite 纤维的斯托克斯不可见性]\label{thm:conclusion-profinite-fiber-stokes-invisibility}
\leanverified{paper\_conclusion\_profinite\_fiber\_stokes\_invisibility}
设有紧交换群短正合列
\[
0\longrightarrow P\longrightarrow E\stackrel{\pi}{\longrightarrow}\TT^r\longrightarrow 0,
\]
其中 \(P\) 为 profinite（紧、全不连通）交换群。
定义 \(E\) 上的斯托克斯可见形式子空间
\[
\Omega^k_{\mathrm{St}}(E):=\pi^\ast\Omega^k(\TT^r).
\]
若 \(\alpha\in\Omega^{r-1}_{\mathrm{St}}(E)\)，且 \(C\) 为 \(E\) 中一条 \(r\)-维分段光滑奇异链，满足 \(C\) 在 \(P\) 方向局部常值（从而 \(\partial C\) 亦局部常值），则成立
\[
\int_C d\alpha=\int_{\partial C}\alpha.
\]
并且该等式仅由 \(\pi(C)\subseteq\TT^r\) 与 \(\pi^\ast\) 的函子性决定，与 \(P\) 的内部结构无关。
\end{theorem}

\begin{proof}
由定义 \(\alpha=\pi^\ast\beta\)（某 \(\beta\in\Omega^{r-1}(\TT^r)\)），且 \(d\alpha=\pi^\ast(d\beta)\)。
当 \(C\) 在 \(P\) 方向局部常值时，\(\pi\) 在链意义下给出 \(\pi_\# C\)，并满足 \(\pi_\#(\partial C)=\partial(\pi_\# C)\)。
于是
\[
\int_C d\alpha
=\int_C \pi^\ast(d\beta)
=\int_{\pi_\# C} d\beta
=\int_{\partial(\pi_\# C)}\beta
=\int_{\partial C}\pi^\ast\beta
=\int_{\partial C}\alpha,
\]
其中第三个等号为 \(\TT^r\) 上经典广义斯托克斯定理。证毕。
\end{proof}

\begin{theorem}[有限精度势约束下的圆维单调性]\label{thm:conclusion-jacobian-potential-dimension-monotonicity}
令 \(U:=(-\tfrac12,\tfrac12)\)。
设 \(f:U^k\to U^\ell\) 为局部 Lipschitz 映射。
在 Rademacher 可微点 \(x\)，定义
\[
J_k f(x):=\sqrt{\det\!\bigl(Df(x)^{\mathsf T}Df(x)\bigr)},
\qquad
\mathcal P_f(x):=-\log J_k f(x),
\]
并约定 \(-\log 0:=+\infty\)。
若 \(\ell<k\)，则
\[
J_k f(x)=0\ \ \text{a.e.},\qquad
\mathcal P_f(x)=+\infty\ \ \text{a.e.}
\]
特别地，若随机向量 \(X\) 在 \(U^k\) 上有绝对连续分布，则
\[
\EE[\mathcal P_f(X)]=+\infty.
\]
\end{theorem}
\begin{proof}
由 Rademacher 定理，\(f\) 在几乎处处可微。
在可微点 \(x\)，\(Df(x)\) 为 \(\ell\times k\) 矩阵，故
\[
\operatorname{rank}Df(x)\le \ell<k.
\]
因此 \(Df(x)^{\mathsf T}Df(x)\) 是秩小于 \(k\) 的 \(k\times k\) 半正定矩阵，其行列式为 \(0\)，即 \(J_k f(x)=0\)。
故 \(\mathcal P_f(x)=+\infty\) 几乎处处成立。
若 \(X\) 绝对连续，则 \(X\) 落在该“几乎处处”集合上的概率为 \(1\)，于是 \(\EE[\mathcal P_f(X)]=+\infty\)。证毕。
\end{proof}

\begin{theorem}[线性相位覆盖的度--精度势恒等式]\label{thm:conclusion-torus-cover-degree-potential-identity}
\leanverified{paper\_conclusion\_torus\_cover\_degree\_potential\_identity}
取 \(A\in \mathrm{GL}_n(\ZZ)\)，定义
\[
F_A:\TT^n\to \TT^n,\qquad F_A(x)=Ax\ \ (\mathrm{mod}\ \ZZ^n).
\]
记覆盖度 \(d:=|\det A|\)。
对任意分段仿射的实现层右逆选择器 \(s_b:\TT^n\to\TT^n\)（满足 \(F_A\circ s_b=\mathrm{id}_{\TT^n}\) 几乎处处），有
\[
Ds_b(y)=A^{-1}\quad\text{a.e.},
\]
从而其精度势
\[
\mathcal P_{s_b}(y):=-\log|\det Ds_b(y)|
\]
满足
\[
\mathcal P_{s_b}(y)=\log d\quad\text{a.e.}
\]
并且对 \(Y\sim\mathrm{Haar}(\TT^n)\),
\[
\EE[\mathcal P_{s_b}(Y)]=\log d.
\]
该期望与分支选择器 \(b\) 无关。
\end{theorem}
\begin{proof}
\(F_A\) 是 \(\TT^n\) 的有限覆盖，覆盖度为 \(d=|\det A|\)。
对任意分段仿射右逆分支，在每个仿射片上都可写成
\[
y\longmapsto A^{-1}y+c
\]
（\(c\) 为常向量，取模 \(\ZZ^n\)）。
故在可微点 \(y\) 有 \(Ds_b(y)=A^{-1}\)，即
\[
|\det Ds_b(y)|=|\det A|^{-1}=d^{-1}.
\]
于是
\[
\mathcal P_{s_b}(y)=-\log(d^{-1})=\log d
\]
几乎处处成立。
对 Haar 分布取期望即得 \(\EE[\mathcal P_{s_b}(Y)]=\log d\)。
右端仅依赖 \(\det A\)，故与选择器无关。证毕。
\end{proof}

\begin{theorem}[覆盖核的最小素账本签名]\label{thm:conclusion-cover-kernel-min-prime-ledger-signature}
沿用定理 \ref{thm:conclusion-torus-cover-degree-potential-identity} 的记号。
记
\[
K_A:=\ker(F_A)\cong \ZZ^n/A\ZZ^n,\qquad |K_A|=d.
\]
设存在紧阿贝尔账本群 \(L\) 与连续群单射
\[
\iota:\TT^n\hookrightarrow \TT^n\times L,\qquad
\operatorname{pr}_1\circ \iota=F_A.
\]
则：
\begin{enumerate}
  \item \(K_A\hookrightarrow L_{\mathrm{tors}}\)；
  \item 若 \(L\) 有限，则 \(|L|\ge d\)，且取 \(L\cong K_A\) 可达下界；
  \item 若
  \[
  d=\prod_{p}p^{v_p(d)},
  \]
  则对每个 \(p\mid d\)，\(L\) 的 \(p\)-Sylow 子群非平凡。
  等价地，\(\{p:\ p\mid d\}\) 为任何此类实现不可消去的素账本支撑。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(\operatorname{pr}_1\circ\iota=F_A\)，存在连续同态 \(\psi:\TT^n\to L\) 使
\[
\iota(x)=\bigl(F_A(x),\psi(x)\bigr).
\]
若 \(x\in K_A\)，则 \(\iota(x)=(0,\psi(x))\)。
且 \(\iota\) 单射，故 \(\psi|_{K_A}\) 必单射，于是
\[
K_A\hookrightarrow L.
\]
因 \(K_A\) 有限，像落在 \(L_{\mathrm{tors}}\)，得第 (1) 条。
第 (2) 条中若 \(L\) 有限，则 \(|L|\ge |K_A|=d\)。
取 \(L=K_A\) 且 \(\psi\) 为商映射对应的核嵌入即可达到等号。
第 (3) 条由有限群嵌入保持各素数初等分量非平凡性直接得到。证毕。
\end{proof}

\begin{conjecture}[判别式支撑的最小素账本原则]\label{conj:conclusion-discriminant-supported-min-prime-ledger}
设 \(K/\QQ\) 为数域，\(\mathcal O_K\) 为整数环，\(\Delta_K\) 为判别式。
在“相位门 + 账本轴”的实现层可逆化框架下，定义
\[
\mathcal P_{\min}(K)
\]
为实现 \(\mathcal O_K\) 上精确四则运算且保持扩张一致性的最小素账本支撑集合。
则猜想
\[
\mathcal P_{\min}(K)=\{p:\ p\mid \Delta_K\}.
\]
等价地，所有不可消去的进位/分歧/纤维缺陷均由 ramified primes 强制，而在 \(\mathcal O_K[\Delta_K^{-1}]\) 上不需引入新素账本轴。
\end{conjecture}

\begin{conjecture}[Hecke--Stokes 互易接口：solenoid 周期模的本征谱恢复]\label{conj:conclusion-hecke-stokes-reciprocity-solenoid-period-spectrum}
