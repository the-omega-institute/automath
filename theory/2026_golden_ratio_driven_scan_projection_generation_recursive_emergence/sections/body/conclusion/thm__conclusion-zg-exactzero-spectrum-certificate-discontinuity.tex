\begin{theorem}[零精确维与全谱稠密性的严格并存]\label{thm:conclusion-zg-exact-zero-dense-spectrum-coexistence}
沿用
\[
\Xi_B:\Sigma_{\mathrm{ZG}}\to K_{B,v},
\qquad
\mu_B:=(\Xi_B)_\#\nu_{\mathrm{ZG}},
\qquad
B=2^L
\]
的记号。对每个 \(\alpha\in[0,\infty]\)，定义
\[
E_\alpha:=\{x\in K_{B,v}: d_{\mu_B}(x)=\alpha\}.
\]
则成立：
\[
\mu_B(E_0)=1,
\qquad
\mu_B(E_\alpha)=0
\quad (\alpha>0),
\]
并且对每个 \(\alpha\in[0,\infty]\)，集合 \(E_\alpha\) 都在 \(K_{B,v}\) 中稠密；特别地，
\[
E_\infty
\]
还是一个稠密 \(G_\delta\) 集。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zg-pushforward-exact-dimension-zero}，
\[
\mu_B(E_0)=1,
\qquad
\mu_B(E_\alpha)=0
\quad (\alpha>0).
\]

另一方面，定理 \ref{thm:conclusion-zg-full-local-dimension-spectrum-every-cylinder} 表明：任意非空地址 cylinder 的像 \(\Xi_B(C)\) 内都能实现任意给定的局部维数 \(\alpha\in[0,\infty]\)。由于这些像构成 \(K_{B,v}\) 的拓扑基，故每个 \(E_\alpha\) 都稠密。

最后，\(\alpha=\infty\) 的 \(G_\delta\) 性已由推论 \ref{cor:conclusion-zg-zero-infinite-local-dimension-topological-coexistence} 给出。证毕。
\end{proof}

\begin{theorem}[有限地址--有限 shadow 证书对局部维数完全失明]\label{thm:conclusion-zg-finite-prefix-shadow-local-dimension-blindness}
\leanverified{paper\_conclusion\_zg\_finite\_prefix\_shadow\_local\_dimension\_blindness}
设 \(S\subset\mathbb P\) 为可数无穷素数集，固定有限前缀长度 \(m\ge 1\) 与有限子集 \(S_0\Subset S\)。对
\[
(x,\xi)\in K_{B,v}\times \Sigma_S
\]
定义有限证书映射
\[
\mathcal C_{m,S_0}(x,\xi)
:=
\bigl(\operatorname{pref}_m(x),\pi_{S\to S_0}(\xi)\bigr),
\]
其中 \(\operatorname{pref}_m(x)\) 表示 \(x=\Xi_B(z)\) 的前 \(m\) 位地址前缀。则不存在函数
\[
\Phi:\operatorname{Im}(\mathcal C_{m,S_0})\to [0,\infty]
\]
使得对所有局部维数存在的点都有
\[
d_{\mu_B}(x)=\Phi\bigl(\mathcal C_{m,S_0}(x,\xi)\bigr).
\]

更强地，对任意证书原子
\[
F(C,\eta)
:=
\{(x,\xi):x\in \Xi_B(C),\ \pi_{S\to S_0}(\xi)=\eta\},
\]
以及任意 \(\alpha\in[0,\infty]\)，都存在
\[
(x,\xi)\in F(C,\eta)
\]
满足
\[
d_{\mu_B}(x)=\alpha.
\]
\end{theorem}
\begin{proof}
强结论正是定理 \ref{thm:conclusion-zg-finite-certificate-fiber-full-spectrum} 的直接重述：任意有限地址前缀与有限 prime shadow 所确定的联合纤维，都完整保留第一坐标上的局部维数谱 \([0,\infty]\)。

若存在所述 \(\Phi\)，则局部维数函数
\[
(x,\xi)\longmapsto d_{\mu_B}(x)
\]
必须在每个证书原子 \(F(C,\eta)\) 上取常值，这与上面的全谱保留矛盾。等价地，这也与推论 \ref{cor:conclusion-zg-finite-factor-no-local-dimension-reader} 相矛盾。证毕。
\end{proof}

\begin{corollary}[局部维数的任何非平凡判别都不能由有限证书实现]\label{cor:conclusion-zg-no-nontrivial-local-dimension-predicate-from-finite-certificate}
沿用定理 \ref{thm:conclusion-zg-finite-prefix-shadow-local-dimension-blindness} 的记号。设
\[
E\subsetneq [0,\infty]
\]
为任意非空真子集。则不存在任何仅由有限地址前缀与有限 prime shadow 所生成的判别器
\[
\mathcal D=\mathcal D(C,\eta)
\]
能够在每个非空证书原子上正确决定命题
\[
d_{\mu_B}(x)\in E.
\]
特别地，下列任一性质都不可能由有限可见证书决定：
\[
d_{\mu_B}(x)=\alpha,
\qquad
d_{\mu_B}(x)<\infty,
\qquad
d_{\mu_B}(x)=\infty,
\qquad
d_{\mu_B}(x)=0.
\]
\end{corollary}
\begin{proof}
反设存在这样的有限证书判别器。取其任一非空证书原子
\[
F(C,\eta).
\]
由定理 \ref{thm:conclusion-zg-finite-prefix-shadow-local-dimension-blindness}，在同一原子内对每个 \(\alpha\in[0,\infty]\) 都存在点 \((x,\xi)\) 使
\[
d_{\mu_B}(x)=\alpha.
\]
由于 \(E\) 是非空真子集，必可在 \([0,\infty]\) 中同时取到
\[
\alpha_{\mathrm{in}}\in E,
\qquad
\alpha_{\mathrm{out}}\notin E.
\]
于是同一证书原子 \(F(C,\eta)\) 内同时存在满足
\[
d_{\mu_B}(x)\in E
\]
与
\[
d_{\mu_B}(x)\notin E
\]
的点。任何仅依赖该有限证书原子的判别器都只能在该原子上取常值，故不可能正确决定上述命题。矛盾。证毕。
\end{proof}

\begin{proposition}[局部维数观测在其定义域上处处不连续]\label{prop:conclusion-zg-local-dimension-nowhere-continuous}
\leanverified{paper\_conclusion\_zg\_local\_dimension\_nowhere\_continuous}
设
\[
D:=\{x\in K_{B,v}: d_{\mu_B}(x)\ \text{存在}\}.
\]
则局部维数函数
\[
d_{\mu_B}:D\to [0,\infty]
\]
在子空间 \(D\) 上 nowhere continuous。
\end{proposition}
\begin{proof}
任取 \(x\in D\) 与其在子空间 \(D\) 中的任意邻域 \(U\)。则存在 \(K_{B,v}\) 中的开集 \(O\) 使
\[
U=D\cap O,
\qquad
x\in O.
\]
由于 \(\Xi_B(C)\) 这类 cylinder 像构成 \(K_{B,v}\) 的拓扑基，可取非空 cylinder \(C\) 使
\[
x\in \Xi_B(C)\subset O.
\]
由定理 \ref{thm:conclusion-zg-full-local-dimension-spectrum-every-cylinder}，在同一 \(\Xi_B(C)\) 内既存在局部维数等于 \(0\) 的点，也存在局部维数等于 \(\infty\) 的点；而这些点都属于 \(D\)。故
\[
U\cap E_0\neq\varnothing,
\qquad
U\cap E_\infty\neq\varnothing.
\]
因此，任意邻域 \(U\) 中都同时出现局部维数取值 \(0\) 与 \(\infty\) 的点，\(d_{\mu_B}\) 不可能在 \(x\) 连续。由于 \(x\) 任意，结论成立。证毕。
\end{proof}
