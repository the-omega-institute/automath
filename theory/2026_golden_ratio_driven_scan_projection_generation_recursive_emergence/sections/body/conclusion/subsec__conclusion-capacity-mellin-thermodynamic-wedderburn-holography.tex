\subsection{连续容量曲线的 Mellin 重构、热力学刚性与 Wedderburn 全息}\label{subsec:conclusion-capacity-mellin-thermodynamic-wedderburn-holography}
在定理 \ref{thm:conclusion-continuous-capacity-integral-node-inversion} 已把整数节点上的容量采样反演为尾计数与直方图、定理 \ref{thm:xi-time-part57b-capacity-moment-exact-mellin-stieltjes} 与定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic} 已把连续容量曲线同全部正阶矩谱做成精确 Mellin--Stieltjes 对偶、而定理 \ref{thm:xi-time-part60aaa-wedderburn-information-completeness} 已把同一数据同 Wedderburn 半单块结构对齐之后，还可把负 Mellin 平均、fold-\(\pi\) 的 Bernoulli 层级、容量等价类的热力学不变量以及块秩全息进一步压缩为下列结论。

\begin{theorem}[容量曲线的 Mellin 完备重构]\label{thm:conclusion-capacity-curve-mellin-complete-reconstruction}
沿用定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic} 的记号。对任意 \(p>0\)，定义负 Mellin 平均
\[
A_{m,p}:=\frac{1}{|X_m|}\sum_{w\in X_m}d_m(w)^{-p}.
\]
则有严格恒等式
\[
A_{m,p}
=
p(p+1)\int_{1}^{\infty}
T^{-p-2}\left(
T-\frac{\mathcal C_m^{\mathrm{cont}}(T)}{|X_m|}
\right)\,\dd T
\]
\[
=
p+1-p(p+1)\int_{1}^{\infty}
T^{-p-2}\frac{\mathcal C_m^{\mathrm{cont}}(T)}{|X_m|}\,\dd T.
\]
因此，单独给定整条连续容量曲线 \(\mathcal C_m^{\mathrm{cont}}(\cdot)\)，便可唯一恢复全部负奇矩
\[
A_{m,1},A_{m,3},A_{m,5},\dots,
\]
并因而唯一恢复 fold-gauge 规范常数列 \((C_m)_{m\ge 1}\) 的整个 Stirling--Bernoulli 渐近谱，以及其极限常数 \(2\pi\)。
\end{theorem}
\leanverified{paper\_conclusion\_capacity\_curve\_mellin\_complete\_reconstruction}
\begin{proof}
对任意整数 \(d\ge 1\) 与任意 \(p>0\)，都有单点恒等式
\[
d^{-p}
=
p(p+1)\int_{1}^{\infty}T^{-p-2}\bigl(T-\min(d,T)\bigr)\,\dd T.
\]
事实上，由于 \(d\ge 1\)，上式右端可化为
\[
p(p+1)\int_d^\infty T^{-p-2}(T-d)\,\dd T
\]
\[
=
p(p+1)\left(
\int_d^\infty T^{-p-1}\,\dd T
-d\int_d^\infty T^{-p-2}\,\dd T
\right)
=
d^{-p}.
\]
将 \(d=d_m(w)\) 代入并对 \(w\in X_m\) 求和，得到
\[
\sum_{w\in X_m}d_m(w)^{-p}
=
p(p+1)\int_1^\infty
T^{-p-2}\sum_{w\in X_m}\bigl(T-\min(d_m(w),T)\bigr)\,\dd T.
\]
而
\[
\sum_{w\in X_m}\bigl(T-\min(d_m(w),T)\bigr)
=
T|X_m|-\mathcal C_m^{\mathrm{cont}}(T),
\]
故两边同除以 \(|X_m|\) 即得第一条公式。

第二条公式只需把积分拆开，并利用
\[
\int_1^\infty T^{-p-1}\,\dd T=\frac{1}{p},
\]
从而
\[
p(p+1)\int_1^\infty T^{-p-2}T\,\dd T=p+1.
\]

于是整条 \(\mathcal C_m^{\mathrm{cont}}(\cdot)\) 便唯一决定每个 \(A_{m,p}\)，特别决定全部负奇矩。
再由定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy} 的系数表达，\(\log C_m\) 的每一阶 Stirling--Bernoulli 模态都由这些负奇矩唯一控制；结合推论 \ref{cor:fold-bin-recover-2pi}，极限常数 \(2\pi\) 亦随之唯一确定。证毕。
\end{proof}

\begin{theorem}[fold-\texorpdfstring{$\pi$}{pi} 的 Euler--Mellin 转译公式]\label{thm:conclusion-foldpi-euler-mellin-translation}
沿用定理 \ref{thm:conclusion-capacity-curve-mellin-complete-reconstruction} 的记号，并沿用推论 \ref{cor:fold-bin-recover-2pi} 的规范常数列 \((C_m)_{m\ge 1}\)。则对任意固定整数 \(R\ge 1\)，当 \(m\to\infty\) 时有
\[
\log C_m-\log(2\pi)
=
\sum_{r=1}^{R}\frac{B_{2r}}{r(2r-1)}\,A_{m,2r-1}
+O\!\bigl(A_{m,2R+1}\bigr).
\]
代入前一定理的 Mellin 重构式，可得
\[
\log C_m-\log(2\pi)
=
2\sum_{r=1}^{R}B_{2r}
\int_{1}^{\infty}T^{-2r-1}
\left(
T-\frac{\mathcal C_m^{\mathrm{cont}}(T)}{|X_m|}
\right)\,\dd T
+O\!\bigl(A_{m,2R+1}\bigr).
\]
换言之，本文中的 \(2\pi\) 可视为阈值容量缺陷
\[
T-\frac{\mathcal C_m^{\mathrm{cont}}(T)}{|X_m|}
\]
所生成的 Bernoulli--Mellin 层级的零阶常数项。
\end{theorem}
\begin{proof}
附录定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy} 的证明已给出：对任意固定 \(R\ge 1\)，当 \(m\to\infty\) 时
\[
\log C_m-\log(2\pi)
=
\sum_{r=1}^{R}
\frac{B_{2r}}{r(2r-1)}
\cdot
\frac{1}{|X_m|}\sum_{w\in X_m}d_m(w)^{-(2r-1)}
+O\!\left(
\frac{1}{|X_m|}
\sum_{w\in X_m}d_m(w)^{-(2R+1)}
\right).
\]
这正是第一条公式。

再在定理 \ref{thm:conclusion-capacity-curve-mellin-complete-reconstruction} 中取 \(p=2r-1\)，便得到
\[
A_{m,2r-1}
=
2r(2r-1)\int_1^\infty T^{-2r-1}
\left(
T-\frac{\mathcal C_m^{\mathrm{cont}}(T)}{|X_m|}
\right)\,\dd T.
\]
将其逐项代回即可得到第二条公式。证毕。
\end{proof}

\begin{theorem}[容量等价类的热力学刚性]\label{thm:conclusion-capacity-equivalence-thermodynamic-rigidity}
设两组正整数值纤维多重度数据 \((X_m,d_m)\) 与 \((\widetilde X_m,\widetilde d_m)\) 的连续容量曲线分别为
\[
\mathcal C_m^{\mathrm{cont}}(T):=\sum_{w\in X_m}\min(d_m(w),T),
\qquad
\widetilde{\mathcal C}_m^{\mathrm{cont}}(T):=\sum_{\widetilde w\in \widetilde X_m}\min(\widetilde d_m(\widetilde w),T).
\]
若对一切 \(m\) 与 \(T\ge 0\) 都有
\[
\mathcal C_m^{\mathrm{cont}}(T)=\widetilde{\mathcal C}_m^{\mathrm{cont}}(T),
\]
则对任意 \(q>0\)，定义
\[
S_q(m):=\sum_{w\in X_m}d_m(w)^q,
\qquad
\widetilde S_q(m):=\sum_{\widetilde w\in \widetilde X_m}\widetilde d_m(\widetilde w)^q,
\]
并在各自底空间上定义
\[
\mu_m(w):=\frac{d_m(w)}{2^m},
\qquad
u_m(w):=\frac{1}{|X_m|},
\]
\[
\widetilde\mu_m(\widetilde w):=\frac{\widetilde d_m(\widetilde w)}{2^m},
\qquad
\widetilde u_m(\widetilde w):=\frac{1}{|\widetilde X_m|},
\]
则有
\[
S_q(m)=\widetilde S_q(m).
\]
因此全部 Rényi 熵
\[
H_q(\mu_m)=H_q(\widetilde\mu_m)
\qquad(q>0)
\]
一致，其中 \(q=1\) 按连续延拓解释为 Shannon 熵。

进一步，对任意 \(\alpha>1\) 记
\[
f_\alpha(t):=t^\alpha-1-\alpha(t-1).
\]
则有
\[
D_{f_\alpha}(\mu_m\Vert u_m)
=
D_{f_\alpha}(\widetilde\mu_m\Vert \widetilde u_m).
\]
特别地，若其中一组属于本文的 bin-fold 类，则另一组与之共享全部 \(q\)-阶热力学极限；因而在该容量等价类内，不会出现额外的 \(q\)-依赖熵率分叉，也不存在第二条独立的伪多重分形分支。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic}，固定 \(m\) 时，连续容量曲线 \(\mathcal C_m^{\mathrm{cont}}\) 已与全部正阶矩谱 \(\{S_q(m)\}_{q>0}\) 完全等价。
因此若
\[
\mathcal C_m^{\mathrm{cont}}=\widetilde{\mathcal C}_m^{\mathrm{cont}},
\]
便立即得到
\[
S_q(m)=\widetilde S_q(m)
\qquad(\forall q>0).
\]

又因为
\[
\sum_{w\in X_m}\mu_m(w)^q
=
2^{-qm}S_q(m),
\qquad
\sum_{\widetilde w\in \widetilde X_m}\widetilde\mu_m(\widetilde w)^q
=
2^{-qm}\widetilde S_q(m),
\]
故全部 Rényi 熵 \(H_q\) 亦完全一致；\(q=1\) 的 Shannon 情形由 \(q\to 1\) 的连续延拓得到。

另一方面，由于所有纤维大小都至少为 \(1\)，有
\[
\mathcal C_m^{\mathrm{cont}}(1)=|X_m|,
\qquad
\widetilde{\mathcal C}_m^{\mathrm{cont}}(1)=|\widetilde X_m|,
\]
故容量等价同时推出
\[
|X_m|=|\widetilde X_m|.
\]
于是
\[
\begin{aligned}
D_{f_\alpha}(\mu_m\Vert u_m)
&=
\sum_{w\in X_m}
u_m(w)\left[
\left(\frac{\mu_m(w)}{u_m(w)}\right)^\alpha
-1-\alpha\left(\frac{\mu_m(w)}{u_m(w)}-1\right)
\right]\\
&=
|X_m|^{\alpha-1}2^{-m\alpha}S_\alpha(m)-1,
\end{aligned}
\]
因为线性项满足
\[
\sum_{w\in X_m}u_m(w)\left(\frac{\mu_m(w)}{u_m(w)}-1\right)
=
\sum_{w\in X_m}\mu_m(w)-\sum_{w\in X_m}u_m(w)=0.
\]
对 \((\widetilde\mu_m,\widetilde u_m)\) 同理可得完全相同的表达式，故
\[
D_{f_\alpha}(\mu_m\Vert u_m)
=
D_{f_\alpha}(\widetilde\mu_m\Vert \widetilde u_m).
\]

最后一句由上述逐阶恒等与正文中关于 bin-fold 容量族决定全部压力谱及其冻结面的结论直接推出。证毕。
\end{proof}

\begin{theorem}[连续容量曲线是 multiplicity 热力学塔的完备不变量]\label{thm:conclusion-continuous-capacity-curve-thermodynamic-completeness}
设两组折叠系统在固定分辨率 \(m\) 上的纤维多重度分别为 \((X_m^{(1)},d_m^{(1)})\) 与 \((X_m^{(2)},d_m^{(2)})\)，其连续容量曲线为
\[
\mathcal C_{m,1}^{\mathrm{cont}}(T)
:=
\sum_{x\in X_m^{(1)}}\min(d_m^{(1)}(x),T),
\qquad
\mathcal C_{m,2}^{\mathrm{cont}}(T)
:=
\sum_{y\in X_m^{(2)}}\min(d_m^{(2)}(y),T).
\]
若对每个 \(m\) 与每个 \(T\ge 0\) 都有
\[
\mathcal C_{m,1}^{\mathrm{cont}}(T)=\mathcal C_{m,2}^{\mathrm{cont}}(T),
\]
则二者在该分辨率上的全部尾计数函数相同：
\[
N_m^{(1)}(t)=N_m^{(2)}(t)
\qquad (\forall t\ge 0),
\]
进而全部正阶矩谱相同：
\[
S_q^{(1)}(m)=S_q^{(2)}(m)
\qquad (\forall q>0),
\]
并且全部正阶 Rényi 熵、最大纤维 \(M_m\)、基态简并数 \(K_m\)、次大纤维 \(M_m^{(2)}\) 亦完全一致。特别地，凡在本文中经由这些尾谱数据所定义的冻结阈值、冻结质量指数与同类热力学量，只要在两侧都有定义，也必完全一致。

换言之，连续容量曲线在 multiplicity 侧已经是热力学完备不变量。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-wedderburn-threshold-holography}，
\[
\mathcal C_m^{\mathrm{cont}}(\cdot)
\Longleftrightarrow
N_m(\cdot)
\Longleftrightarrow
h_m(\cdot)
\]
在固定 \(m\) 下彼此唯一决定。因此若两条连续容量曲线处处相等，则尾计数函数 \(N_m^{(1)}\) 与 \(N_m^{(2)}\) 必逐点相等。

再由定理 \ref{thm:conclusion-capacity-equivalence-thermodynamic-rigidity}，容量曲线一致推出对一切 \(q>0\) 都有
\[
S_q^{(1)}(m)=S_q^{(2)}(m),
\]
并且全部正阶 Rényi 熵一致。

另一方面，\(N_m\) 是整数值右连续阶梯函数，故由其可直接读出
\[
M_m=\sup\{t\ge 0:\ N_m(t)\ge 1\},
\qquad
K_m=N_m(M_m),
\]
以及次大纤维
\[
M_m^{(2)}:=\sup\{t<M_m:\ N_m(t)\ge 1\},
\]
（若不存在则按正文约定取平凡值）。因此这些热力学端点量也完全一致。

最后，正文中关于冻结阈值、冻结质量指数以及同类热力学量的定义都只依赖于 \(\{S_q(m)\}_{q>0}\)、\(M_m\)、\(K_m\)、\(M_m^{(2)}\) 与相应尾谱极限。既然上述数据逐项一致，则这些由它们函子性定义的量也必一致。证毕。
\end{proof}

\begin{theorem}[Wedderburn 阈值全息定理]\label{thm:conclusion-wedderburn-threshold-holography}
\leanverified{paper\_conclusion\_wedderburn\_threshold\_holography}
固定 \(m\)。则下列四类数据彼此唯一决定：
\begin{enumerate}
  \item 连续容量曲线 \(\mathcal C_m^{\mathrm{cont}}(\cdot)\)；
  \item 尾计数函数
  \[
  N_m(T):=\#\{w\in X_m:\ d_m(w)\ge T\};
  \]
  \item 退化谱直方图
  \[
  h_m(d):=\#\{w\in X_m:\ d_m(w)=d\}
  \qquad(d\in\NN);
  \]
  \item 纤维群胚代数
  \[
  A_m:=\CC[\mathcal R_m^{\mathrm{bin}}]
  \]
  的半单代数同构类。
\end{enumerate}
从而，fold-gauge 体积定义的规范常数列 \((C_m)_{m\ge 1}\) 及其极限常数 \(2\pi\)，实际上都是序列 \((A_m)_{m\ge 1}\) 的 Wedderburn 块秩统计不变量。
\end{theorem}
\begin{proof}
\(\textup{(1)}\Leftrightarrow\textup{(2)}\) 由定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic} 中的
\[
\mathcal C_m^{\mathrm{cont}}(T)=\int_0^T N_m(t)\,\dd t
\]
立即得到。

\(\textup{(2)}\Leftrightarrow\textup{(3)}\) 由
\[
h_m(d)=N_m(d)-N_m(d+1)
\qquad(d\ge 1)
\]
给出，反向则由
\[
N_m(T)=\sum_{d\ge \lceil T\rceil}h_m(d)
\]
恢复。

\(\textup{(3)}\Leftrightarrow\textup{(4)}\) 则由命题 \ref{prop:fold-bin-groupoid-algebra-wedderburn} 的 Wedderburn 分解
\[
A_m
\cong
\bigoplus_{w\in X_m}M_{d_m(w)}(\CC)
\cong
\bigoplus_{d\ge 1}M_d(\CC)^{\oplus h_m(d)}
\]
给出。有限维半单复代数的同构类由矩阵块尺寸及其重数唯一决定，故半单代数同构类与直方图互相唯一确定。

最后，由 \(h_m\) 可直接恢复
\[
|G_m^{\mathrm{bin}}|
=
\prod_{d\ge 1}(d!)^{h_m(d)},
\qquad
g_m
=
\frac{1}{2^m}\sum_{d\ge 1}h_m(d)\log(d!).
\]
因此，由 fold-gauge 体积账本定义的规范常数列 \((C_m)\) 及其极限 \(2\pi\) 也都成为同一 Wedderburn 块秩统计的函数。证毕。
\end{proof}

\begin{theorem}[粗粒化下连续容量曲线与全矩谱的全局刚性]\label{thm:conclusion-coarsening-global-rigidity-of-capacity-curve}
\leanverified{paper\_conclusion\_coarsening\_global\_rigidity\_of\_capacity\_curve}
设
\[
f:\Omega\to X,
\qquad
g:X\to Y,
\qquad
h:=g\circ f,
\]
并定义纤维多重度
\[
d_f(x):=\#f^{-1}(x),
\qquad
d_h(y):=\#h^{-1}(y).
\]
记连续容量曲线
\[
\mathcal C_f^{\mathrm{cont}}(T):=\sum_{x\in X}\min\bigl(d_f(x),T\bigr),
\qquad
\mathcal C_h^{\mathrm{cont}}(T):=\sum_{y\in Y}\min\bigl(d_h(y),T\bigr),
\]
以及幂和
\[
S_q(f):=\sum_{x\in X}d_f(x)^q,
\qquad
S_q(h):=\sum_{y\in Y}d_h(y)^q
\qquad (q>0).
\]
再记正纤维支撑
\[
X_+:=\{x\in X:d_f(x)>0\}.
\]
则下列三条件等价：
\begin{enumerate}
  \item \(\mathcal C_h^{\mathrm{cont}}(T)=\mathcal C_f^{\mathrm{cont}}(T)\) 对一切 \(T>0\) 成立；
  \item \(S_q(h)=S_q(f)\) 对一切 \(q>0\) 成立；
  \item \(g|_{X_+}\) 为单射。
\end{enumerate}
换言之，粗粒化一旦真正合并了两个正纤维，便不可能同时在容量曲线与全部碰撞矩两侧隐身。
\end{theorem}
\begin{proof}
\(\textup{(3)}\Rightarrow\textup{(1)},\textup{(2)}\)：若 \(g|_{X_+}\) 单射，则 \(h\) 的非零纤维多重度多重集与 \(f\) 完全一致，只是重新标号而已。故连续容量曲线与全部幂和都逐项相等。

\(\textup{(1)}\Rightarrow\textup{(3)}\)：反设 \(g|_{X_+}\) 非单射，则存在 \(y\in Y\) 及至少两个不同点 \(x_1,x_2\in X_+\) 满足
\[
g(x_1)=g(x_2)=y.
\]
记
\[
a_x:=d_f(x)
\qquad (x\in g^{-1}(y)\cap X_+),
\qquad
A:=\sum_{g(x)=y}a_x=d_h(y).
\]
由于至少有两个 \(a_x>0\)，对任意 \(0<T<A\) 都有严格不等式
\[
\min(A,T)<\sum_{g(x)=y}\min(a_x,T).
\]
对所有 \(y'\neq y\) 则有
\[
\min(d_h(y'),T)\le \sum_{g(x)=y'}\min(d_f(x),T),
\]
并且这里其实取等。故
\[
\mathcal C_h^{\mathrm{cont}}(T)<\mathcal C_f^{\mathrm{cont}}(T)
\qquad (0<T<A),
\]
与 \(\textup{(1)}\) 矛盾。故 \(g|_{X_+}\) 必为单射。

\(\textup{(1)}\Rightarrow\textup{(2)}\)：由定理 \ref{thm:conclusion-continuous-capacity-integral-node-inversion} 的离散反演公式，\(\mathcal C_f^{\mathrm{cont}}\) 与 \(\mathcal C_h^{\mathrm{cont}}\) 在全部整数节点上相等便推出其纤维直方图完全相等，从而对一切 \(q>0\) 都有
\[
S_q(f)=S_q(h).
\]

\(\textup{(2)}\Rightarrow\textup{(1)}\)：定义
\[
n_d(f):=\#\{x\in X:d_f(x)=d\},
\qquad
n_d(h):=\#\{y\in Y:d_h(y)=d\}.
\]
则对足够小的 \(z>0\)，有解析恒等式
\[
\sum_{q\ge 1}S_q(f)z^q
=
\sum_{d\ge 1}n_d(f)\frac{dz}{1-dz},
\qquad
\sum_{q\ge 1}S_q(h)z^q
=
\sum_{d\ge 1}n_d(h)\frac{dz}{1-dz}.
\]
若 \(S_q(f)=S_q(h)\) 对一切正整数 \(q\) 成立，则两边作为解析函数恒等，故极点与留数完全一致，从而
\[
n_d(f)=n_d(h)
\qquad (\forall d\ge 1).
\]
于是
\[
\mathcal C_f^{\mathrm{cont}}(T)=\sum_{d\ge 1}n_d(f)\min(d,T)
=\sum_{d\ge 1}n_d(h)\min(d,T)
=\mathcal C_h^{\mathrm{cont}}(T)
\qquad (\forall T>0).
\]
证毕。
\end{proof}

\begin{corollary}[低多重度扇区的局部不可见性判据]\label{cor:conclusion-low-multiplicity-sector-local-invisibility}
沿用定理 \ref{thm:conclusion-coarsening-global-rigidity-of-capacity-curve} 的记号。若存在 \(U>0\) 使
\[
\mathcal C_h^{\mathrm{cont}}(T)=\mathcal C_f^{\mathrm{cont}}(T)
\qquad (0\le T\le U),
\]
则对每个整数 \(1\le k<U\) 都有
\[
\#\{y\in Y:d_h(y)=k\}
=
\#\{x\in X:d_f(x)=k\}.
\]
因此，若粗粒化在预算区间 \([0,U]\) 内完全不可见，则它不仅不改变该区间内的总可逆容量，而且逐级冻结了全部低多重度直方图。
\end{corollary}
\begin{proof}
对任意正整数 \(k<U\)，取 \(\varepsilon>0\) 足够小，使
\[
k-\varepsilon\ge 0,
\qquad
k+\varepsilon<U.
\]
由于
\[
\mathcal C_f^{\mathrm{cont}}(T)=\mathcal C_h^{\mathrm{cont}}(T)
\qquad (0\le T\le U),
\]
两条曲线在 \(k\) 左右邻域内完全一致。另一方面，对任意有限纤维多重度谱，连续容量曲线在整数点的左右斜率分别为
\[
\bigl(\mathcal C^{\mathrm{cont}}\bigr)'_-(k)=\#\{d\ge k\},
\qquad
\bigl(\mathcal C^{\mathrm{cont}}\bigr)'_+(k)=\#\{d>k\},
\]
故其跳跃恰等于
\[
\bigl(\mathcal C^{\mathrm{cont}}\bigr)'_-(k)-\bigl(\mathcal C^{\mathrm{cont}}\bigr)'_+(k)=\#\{d=k\}.
\]
将此分别应用于 \(f\) 与 \(h\)，便得到
\[
\#\{x\in X:d_f(x)=k\}
=
\#\{y\in Y:d_h(y)=k\}.
\]
证毕。
\end{proof}
