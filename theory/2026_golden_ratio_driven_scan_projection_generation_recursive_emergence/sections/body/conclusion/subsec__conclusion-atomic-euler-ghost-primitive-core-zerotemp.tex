\subsection{单原子 Euler 因子的 ghost--primitive 手术、core 热力学不变性与零温有限部位移}\label{subsec:conclusion-atomic-euler-ghost-primitive-core-zerotemp}
在小节 \ref{subsec:conclusion-realinput40-primitive-surgery-residue-escort-geometry} 已把长度 \(2\) 原子在真实输入 \(40\) 状态核上的 \(\zeta\)/ghost/primitive 剥离与 Abel 位移写成显式公式之后，还可把同一机制提升为一般单原子 Euler 因子的结论层表述，并把附录 \ref{app:real-input-40-zeta-u} 中
\[
\Delta(z,u)=(1-uz^2)\,F(z,u)
\]
的后果进一步压缩为下列四条彼此闭合的命题：primitive 层只留下单点缺陷，ghost 层沿倍长度半群形成几何尾迹，pressure 对被剥离原子完全透明，而 Abel 有限部则保留一条在零温端点仍不消失的精确位移。

\begin{theorem}[一般单原子 Euler 因子的 ghost--primitive 完全手术公式]\label{thm:conclusion-atomic-euler-ghost-primitive-complete-surgery}
\leanverified{paper\_conclusion\_atomic\_euler\_ghost\_primitive\_complete\_surgery}
设某个加权有限状态核满足
\[
\zeta(z,u)=\frac{1}{(1-u^Ez^\ell)^m}\,\zeta_{\mathrm{core}}(z,u),
\qquad
m\in\mathbb Z_{\ge 1},\ \ell\in\mathbb Z_{\ge 1},\ E\in\mathbb Z_{\ge 0},
\]
并定义 primitive 生成函数
\[
P(z,u)=\sum_{n\ge 1}p_n(u)z^n,
\qquad
P_{\mathrm{core}}(z,u)=\sum_{n\ge 1}p_n^{\mathrm{core}}(u)z^n,
\]
以及 periodic/ghost 坐标
\[
\log \zeta(z,u)=\sum_{n\ge 1}\frac{a_n(u)}{n}z^n,
\qquad
\log \zeta_{\mathrm{core}}(z,u)=\sum_{n\ge 1}\frac{a_n^{\mathrm{core}}(u)}{n}z^n.
\]
则有严格恒等式
\[
\boxed{\;
p_n(u)-p_n^{\mathrm{core}}(u)
=
m\,u^E\,\mathbf 1_{\{n=\ell\}}.\;}
\]
\[
\boxed{\;
a_n(u)-a_n^{\mathrm{core}}(u)
=
m\ell\,u^{En/\ell}\,\mathbf 1_{\{\ell\mid n\}}.\;}
\]
换言之，单个 Euler 因子在 primitive 层只改动唯一长度 \(\ell\) 的坐标，而在 ghost 层则沿 \(\ell\)-倍长度子半群留下完全显式的几何尾迹。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-finite-euler-primitive-surgery-additivity}，
\[
P(z,u)=P_{\mathrm{core}}(z,u)+m\,u^Ez^\ell,
\]
逐项比较系数即得 primitive 差分公式。

另一方面，
\[
\log\zeta(z,u)-\log\zeta_{\mathrm{core}}(z,u)
=
-m\log(1-u^Ez^\ell)
=
m\sum_{r\ge 1}\frac{u^{Er}}{r}z^{\ell r}.
\]
再与
\[
\log\zeta(z,u)-\log\zeta_{\mathrm{core}}(z,u)
=
\sum_{n\ge 1}\frac{a_n(u)-a_n^{\mathrm{core}}(u)}{n}z^n
\]
比较系数：当 \(n=\ell r\) 时，
\[
\frac{a_n(u)-a_n^{\mathrm{core}}(u)}{n}
=
\frac{m\,u^{Er}}{r},
\]
从而
\[
a_n(u)-a_n^{\mathrm{core}}(u)=m\ell\,u^{Er}=m\ell\,u^{En/\ell}.
\]
若 \(\ell\nmid n\)，则相应系数为 \(0\)。证毕。
\end{proof}

\begin{corollary}[真实输入 \texorpdfstring{$40$}{40} 状态核的偶长度原子缺陷与 polylog ghost]\label{cor:conclusion-realinput40-even-defect-polylog-ghost}
\leanverified{paper\_conclusion\_realinput40\_even\_defect\_polylog\_ghost}
沿用附录 \ref{app:real-input-40-zeta-u} 的记号。对真实输入 \(40\) 状态核，
\[
\Delta(z,u)=(1-uz^2)\,F(z,u),
\]
因此
\[
\boxed{\;
p_n(u)-p_n^{\mathrm{core}}(u)
=
u\,\mathbf 1_{\{n=2\}},\;}
\]
\[
\boxed{\;
a_n(u)-a_n^{\mathrm{core}}(u)
=
2u^{n/2}\,\mathbf 1_{\{2\mid n\}}.\;}
\]
若定义 primitive 累积函数
\[
\Pi_u(N):=\sum_{n\le N}p_n(u),
\qquad
\Pi_u^{\mathrm{core}}(N):=\sum_{n\le N}p_n^{\mathrm{core}}(u),
\]
则有精确 Heaviside 台阶
\[
\boxed{\;
\Pi_u(N)-\Pi_u^{\mathrm{core}}(N)
=
u\,\mathbf 1_{\{N\ge 2\}}.\;}
\]
进一步，若定义 periodic defect 的 Dirichlet 生成函数
\[
\mathcal D_{\mathrm{gh}}(s;u)
:=
\sum_{n\ge 1}\frac{a_n(u)-a_n^{\mathrm{core}}(u)}{n^s},
\qquad
\Re s>1,
\]
则对 \(|u|<1\) 有
\[
\boxed{\;
\mathcal D_{\mathrm{gh}}(s;u)=2^{1-s}\operatorname{Li}_s(u).\;}
\]
而在 \(u=1\) 时，
\[
\boxed{\;
\mathcal D_{\mathrm{gh}}(s;1)=2^{1-s}\zeta(s)
\qquad (\Re s>1).\;}
\]
因此，对真实输入 \(40\) 状态核而言，全部非核心 primitive 信息都压缩在唯一的长度 \(2\) 原子上，而其在 periodic 层的全部遗迹恰是一条显式 polylog ghost。
\end{corollary}
\begin{proof}
将定理 \ref{thm:conclusion-atomic-euler-ghost-primitive-complete-surgery} 应用于
\[
(m,\ell,E)=(1,2,1)
\]
即可得到前两条公式，这也与命题 \ref{prop:atomic-2-orbit-split-logM} 及注 \ref{rem:real-input-40-single-euler} 完全一致。

对 primitive 差分公式在 \(n\le N\) 上求和，立即得到 Heaviside 台阶。

最后，由偶长度 ghost 差分，
\[
\mathcal D_{\mathrm{gh}}(s;u)
=
\sum_{r\ge 1}\frac{2u^r}{(2r)^s}
=
2^{1-s}\sum_{r\ge 1}\frac{u^r}{r^s}
=
2^{1-s}\operatorname{Li}_s(u)
\qquad (|u|<1).
\]
取 \(u=1\) 即得 \(\mathcal D_{\mathrm{gh}}(s;1)=2^{1-s}\zeta(s)\)。证毕。
\end{proof}

\begin{theorem}[压力的 core 不变性与 Abel 有限部的精确平移]\label{thm:conclusion-realinput40-core-pressure-invariance-abel-shift}
\leanverified{paper\_conclusion\_realinput40\_core\_pressure\_invariance\_abel\_shift}
继续考虑附录 \ref{app:real-input-40-zeta-u} 的真实输入 \(40\) 状态核。记
\[
\zeta_{\mathrm{core}}(z,u):=F(z,u)^{-1},
\qquad
\lambda(u):=\rho(M(u)),
\qquad
\lambda_{\mathrm{core}}(u):=z_{\star,\mathrm{core}}(u)^{-1},
\qquad
z_\star(u):=\lambda(u)^{-1},
\]
其中 \(z_{\star,\mathrm{core}}(u)\) 表示 \(\zeta_{\mathrm{core}}(z,u)\) 的主极点。并令
\[
P(\theta):=\log\lambda(e^\theta),
\qquad
P_{\mathrm{core}}(\theta):=\log\lambda_{\mathrm{core}}(e^\theta).
\]
再定义 full/core 的 Abel 有限部常数
\[
\log\mathfrak M(u)
:=
\sum_{n\ge 1}\left(\frac{p_n(u)}{\lambda(u)^n}-\frac1n\right),
\qquad
\log\mathfrak M_{\mathrm{core}}(u)
:=
\sum_{n\ge 1}\left(\frac{p_n^{\mathrm{core}}(u)}{\lambda_{\mathrm{core}}(u)^n}-\frac1n\right).
\]
则有
\[
\boxed{\;
\lambda(u)=\lambda_{\mathrm{core}}(u),
\qquad
P(\theta)=P_{\mathrm{core}}(\theta).\;}
\]
因而一切仅依赖于 \(P\) 及其导数的热力学量都与 core 完全一致，包括累积量、Legendre 率函数以及由局部累积量决定的 Gaussian/Edgeworth 常数。

另一方面，Abel 有限部发生精确平移：
\[
\boxed{\;
\log\mathfrak M(u)-\log\mathfrak M_{\mathrm{core}}(u)
=
u\,z_\star(u)^2
=
u\,\lambda(u)^{-2}.\;}
\]
特别地，在未扭曲点 \(u=1\) 上，
\[
\boxed{\;
\log\mathfrak M(1)-\log\mathfrak M_{\mathrm{core}}(1)
=
\varphi^{-4}
=
\frac{7-3\sqrt5}{2}.\;}
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-atomic-witt-surgery-pressure-ldp-abel-invariance} 在单原子情形
\[
(m,\ell,E)=(1,2,1)
\]
下立即给出
\[
\lambda(u)=\lambda_{\mathrm{core}}(u),
\qquad
P(\theta)=P_{\mathrm{core}}(\theta),
\qquad
\log\mathfrak M(u)-\log\mathfrak M_{\mathrm{core}}(u)=u\,z_\star(u)^2.
\]
其后关于累积量、Legendre 率函数与 Gaussian/Edgeworth 常数的一致性，只是 \(P\) 及其导数完全一致的直接后果。

最后在 \(u=1\) 处，由附录 \ref{app:real-input-40-zeta-u} 中
\[
z_\star(1)=\varphi^{-2}
\]
立得
\[
\log\mathfrak M(1)-\log\mathfrak M_{\mathrm{core}}(1)=z_\star(1)^2=\varphi^{-4}=\frac{7-3\sqrt5}{2}.
\]
证毕。
\end{proof}

\begin{corollary}[零温端点的有限部位移趋于 \texorpdfstring{$\rho(B_\infty)^{-1}$}{rho(B_infty)^-1}]\label{cor:conclusion-realinput40-zero-temp-finitepart-displacement}
沿用定理 \ref{thm:conclusion-realinput40-core-pressure-invariance-abel-shift} 的记号，令 \(u=e^\theta\)，并定义有限部位移
\[
\Delta_{\mathfrak M}(\theta)
:=
\log\mathfrak M(e^\theta)-\log\mathfrak M_{\mathrm{core}}(e^\theta)
=
e^\theta\lambda(e^\theta)^{-2}
=
e^{\theta-2P(\theta)}.
\]
则当 \(\theta\to+\infty\) 时，有
\[
\boxed{\;
\Delta_{\mathfrak M}(\theta)
=
c^{-2}\left(1-\frac{2d}{c}e^{-\theta/2}+O(e^{-\theta})\right),\;}
\]
并且
\[
\boxed{\;
\lim_{\theta\to+\infty}\Delta_{\mathfrak M}(\theta)
=
c^{-2}
=
\rho(B_\infty)^{-1}.\;}
\]
因此，原子 Euler 因子在 Abel 有限部上留下的净位移并不会在零温端点消失，而是精确趋于四态地基 SFT 的 Perron 根倒数。
\end{corollary}
\begin{proof}
由命题 \ref{prop:real-input-40-pressure-freezing}，
\[
P(\theta)
=
\frac{\theta}{2}+\log c+\frac{d}{c}e^{-\theta/2}+O(e^{-\theta}),
\qquad
\theta\to+\infty.
\]
代入
\(
\Delta_{\mathfrak M}(\theta)=e^{\theta-2P(\theta)}
\)
可得
\[
\theta-2P(\theta)
=
-2\log c-\frac{2d}{c}e^{-\theta/2}+O(e^{-\theta}).
\]
指数化即
\[
\Delta_{\mathfrak M}(\theta)
=
c^{-2}\exp\!\left(-\frac{2d}{c}e^{-\theta/2}+O(e^{-\theta})\right)
=
c^{-2}\left(1-\frac{2d}{c}e^{-\theta/2}+O(e^{-\theta})\right).
\]
另一方面，定理 \ref{thm:xi-time-part73-perron-vs-atomic-sqrtu-separation} 给出
\[
c^2=\rho(B_\infty),
\]
故极限值恰为 \(\rho(B_\infty)^{-1}\)。证毕。
\end{proof}

因此，对单原子 Euler 因子而言，primitive 层的全部 full/core 差异都集中为一个有限长度缺陷，ghost 层则把该缺陷扩展为一条完全可积的倍长度尾迹；与此同时，pressure 与由其导数生成的热力学量对该原子完全透明，而 Abel 有限部却以精确位移的形式保留其痕迹，并在零温端点被四态地基 SFT 的 Perron 尺度所接管。由此，Big-Witt/primitive 剥离并不只是局部代数分解，而是同时闭合于 primitive、ghost、pressure 与 finite-part 四个层面的统一手术机制。
