\begin{theorem}[移位可交换算法的超度量稳定性与 Gödel 实值化的非连续性]\label{thm:conclusion-register-ultrametric-lipschitz-and-godel-no-real-extension}
令
\[
\widehat{\mathcal R}:=\NN^{\NN}
\]
为允许无限支撑的寄存器空间。
对 \(r,s\in\widehat{\mathcal R}\) 定义首个分歧坐标
\[
\kappa(r,s):=\min\{t\ge 1:\ r_t\neq s_t\}\in\NN\cup\{\infty\},
\]
并定义超度量（约定 \(2^{-\infty}=0\)）
\[
d(r,s):=2^{-\kappa(r,s)}.
\]
对任意 \(f(z)=\sum_{j=0}^{d}a_j z^j\in\NN[z]\)，按同一 Toeplitz 卷积规则定义映射
\[
(\Phi_f(r))_t:=\sum_{j=0}^{d}a_j\,r_{t-j},
\qquad r_k:=0\ (k\le 0),
\]
则：
\begin{enumerate}
  \item \textbf{（\(1\)-Lipschitz）}
        对任意 \(r,s\in\widehat{\mathcal R}\)，有
        \[
        d(\Phi_f(r),\Phi_f(s))\le d(r,s).
        \]
  \item \textbf{（严格收缩判别）}
        存在统一常数 \(1/2\) 使对一切 \(r\neq s\) 有
        \[
        d(\Phi_f(r),\Phi_f(s))\le \tfrac12\,d(r,s)
        \]
        当且仅当 \(a_0=0\)。
        在 \(a_0=0\) 的情形，对任意有限支撑 \(r\in\NN^{(\NN)}\subset\widehat{\mathcal R}\) 有 \(\Phi_f^{\,n}(r)\to 0\)（关于 \(d\)）。
  \item \textbf{（Gödel 实值化无连续延拓）}
        令 \(p_1<p_2<\cdots\) 为素数序列，并定义有限支撑上的 Gödel 整数化
        \[
        N(r):=\prod_{t\ge 1}p_t^{r_t}\qquad\bigl(r\in\NN^{(\NN)}\bigr).
        \]
        则不存在连续映射 \(\widetilde N:\widehat{\mathcal R}\to\RR_{>0}\) 满足 \(\widetilde N|_{\NN^{(\NN)}}=N\)。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)} 若 \(\kappa(r,s)=k\)，则 \(r_t=s_t\) 对所有 \(t\le k-1\) 成立。
而 \((\Phi_f(r))_t\) 仅依赖于 \((r_1,\dots,r_t)\)；故对所有 \(t\le k-1\) 有 \((\Phi_f(r))_t=(\Phi_f(s))_t\)，从而 \(\kappa(\Phi_f(r),\Phi_f(s))\ge k\)，即 \(d(\Phi_f(r),\Phi_f(s))\le d(r,s)\)。
\par
\textbf{(2)} 若 \(a_0=0\)，则 \((\Phi_f(r))_t\) 仅依赖 \((r_1,\dots,r_{t-1})\)，因而由上一步可得
\(\kappa(\Phi_f(r),\Phi_f(s))\ge \kappa(r,s)+1\)
对 \(r\neq s\) 成立，即
\(
d(\Phi_f(r),\Phi_f(s))\le \tfrac12 d(r,s)
\)。
反之若 \(a_0>0\)，取 \(r=e_k\)、\(s=0\)，则 \((\Phi_f(r))_k=a_0\neq 0=(\Phi_f(s))_k\)，且对 \(t<k\) 两者同为 \(0\)，故 \(\kappa(\Phi_f(r),\Phi_f(s))=k=\kappa(r,s)\)，严格收缩不成立。
在 \(a_0=0\) 时，若 \(r\) 有限支撑，其非零坐标在每次迭代下整体向右漂移，故任意固定坐标 \(t\) 上的值最终为 \(0\)，即 \(\Phi_f^{\,n}(r)\to 0\) 于超度量 \(d\)。
\par
\textbf{(3)} 令 \(e_n\in\NN^{(\NN)}\) 为第 \(n\) 个基向量。
则对任意固定坐标 \(t\)，当 \(n>t\) 时 \((e_n)_t=0\)，故 \(e_n\to 0\)（关于 \(d\)）。
若存在连续延拓 \(\widetilde N\)，则 \(\widetilde N(e_n)\to \widetilde N(0)\)。
但 \(\widetilde N(e_n)=N(e_n)=p_n\)，而素数无界（例如由 Euclid 定理），故 \(p_n\to\infty\)，矛盾。
证毕。
\end{proof}
