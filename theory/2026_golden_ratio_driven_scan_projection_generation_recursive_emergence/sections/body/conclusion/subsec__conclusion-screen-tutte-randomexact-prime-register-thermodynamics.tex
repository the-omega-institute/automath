\subsection{屏幕审计景观的 Tutte 编码、随机精确率与 prime-register 轴数}\label{subsec:conclusion-screen-tutte-randomexact-prime-register-thermodynamics}
沿用定理 \ref{thm:spg-dyadic-cubical-boundary-injective}、定理 \ref{thm:spg-partial-boundary-screen-kernel-relative-homology}、定理 \ref{thm:spg-partial-boundary-godel-audit-optimal}、定理 \ref{thm:spg-dyadic-boundary-image-ldpc}、定理 \ref{thm:conclusion-screen-discrete-continuous-completion-entropy-identification} 与定理 \ref{thm:conclusion-screen-minimal-completion-family-independent-kernel} 的记号。前两小节已经把部分屏幕的可见补全面数、审计余秩、模 \(p\) 纤维熵指数与收缩拟阵基几何严格对齐。若再把同一对象与 dyadic 边界像的 LDPC 阴影并读，则整张屏幕审计景观还可进一步压缩为如下四条全局后果。

固定分辨率 \(m\) 与维数 \(n\)，记
\[
E_{m,n}:=\overrightarrow{\mathcal F_m^{n-1}},
\qquad
M_{m,n}:=(E_{m,n},r),
\qquad
\rho_{m,n}:=r(E_{m,n})=2^{mn},
\]
并令
\[
N_{m,n}:=|E_{m,n}|=n(2^m+1)\,2^{m(n-1)}.
\]
对任意屏幕 \(S\subseteq E_{m,n}\)，定义
\[
\delta(S):=\rho_{m,n}-r(S),
\qquad
\nu(S):=|S|-r(S),
\]
以及可见补全数
\[
\ell(S):=
\min\Bigl\{
|T|:\ T\subseteq E_{m,n}\setminus S,\ \ker f_{S\cup T}=0
\Bigr\}.
\]
由定理 \ref{thm:conclusion-screen-discrete-continuous-completion-entropy-identification}，对每个 \(S\) 都有
\[
\ell(S)=\delta(S).
\]

\begin{theorem}[边界拟阵的 Tutte 多项式完全编码审计景观]\label{thm:conclusion-screen-audit-landscape-tutte-complete-encoding}
边界拟阵 \(M_{m,n}\) 的 Tutte 多项式满足
\[
T_{M_{m,n}}(x,y)
=
\sum_{S\subseteq E_{m,n}}
(x-1)^{\ell(S)}(y-1)^{\nu(S)}.
\]
等价地，其全屏幕分层生成函数满足
\[
\sum_{S\subseteq E_{m,n}} z^{\ell(S)}w^{\nu(S)}
=
T_{M_{m,n}}(1+z,1+w).
\]
因此，全部屏幕的“可见补全数--冗余度”二维景观被单一二变量多项式完全编码。
\end{theorem}
\begin{proof}
由 Tutte 多项式定义，
\[
T_{M_{m,n}}(x,y)
=
\sum_{S\subseteq E_{m,n}}
(x-1)^{\rho_{m,n}-r(S)}(y-1)^{|S|-r(S)}.
\]
由于
\[
\rho_{m,n}-r(S)=\delta(S)=\ell(S),
\qquad
|S|-r(S)=\nu(S),
\]
便得第一式。把 \(x,y\) 分别替换为 \(1+z,1+w\)，即得第二式。证毕。
\end{proof}

\begin{theorem}[可见补全数、相对顶维同调秩与 prime-register 轴数的同一]\label{thm:conclusion-screen-completion-homology-prime-register-axis-identification}
对任意屏幕 \(S\subseteq E_{m,n}\)，有严格恒等式
\[
\ell(S)
=
\delta(S)
=
\rank_{\ZZ}\ker f_S
=
\rank_{\ZZ}H_n(\mathcal Q_m,A_{\neg S};\ZZ).
\]
进一步，在文稿既定的正负分裂 prime-register 审计协议中，规范审计码所需的辅助素数总数恰为
\[
2\ell(S).
\]
亦即，一单位可见缺口恰对应一条整数审计方向，而在正负指数分裂后恰对应两枚辅助素数。
\end{theorem}
\begin{proof}
第一条等式
\[
\ell(S)=\delta(S)
\]
正是定理 \ref{thm:conclusion-screen-discrete-continuous-completion-entropy-identification} 的内容。

另一方面，由定理 \ref{thm:spg-dyadic-cubical-boundary-injective}，\(\partial_n\) 在顶维链群 \(C_n(\mathcal Q_m;\ZZ)\) 上单射；该链群自由秩为 \(\rho_{m,n}=2^{mn}\)。由秩--零度公式，
\[
\rank_{\ZZ}\ker f_S
=
\rho_{m,n}-\rank_{\ZZ}\operatorname{im}(f_S)
=
\rho_{m,n}-r(S)
=
\delta(S).
\]
再由定理 \ref{thm:spg-partial-boundary-screen-kernel-relative-homology}，
\[
\ker f_S\cong H_n(\mathcal Q_m,A_{\neg S};\ZZ),
\]
故四种量完全一致。

最后，由定理 \ref{thm:spg-partial-boundary-godel-audit-optimal}，若
\[
r:=\rank_{\ZZ}\ker f_S,
\]
则 prime-register 审计码为 \(\ker f_S\) 的每个 \(\ZZ\)-基坐标各配置一对互异素数，分别记录其正、负部分，因此总共恰需 \(2r\) 枚辅助素数。代入
\[
r=\rank_{\ZZ}\ker f_S=\ell(S)
\]
便得结论。证毕。
\end{proof}

\begin{proposition}[Bernoulli 随机屏幕精确率的 Tutte 闭式]\label{prop:conclusion-screen-bernoulli-exactness-tutte-law}
设 \(S_p\subseteq E_{m,n}\) 为参数 \(p\in(0,1)\) 的 Bernoulli 随机屏幕，即每张取向 \((n-1)\)-面独立地以概率 \(p\) 入屏。则
\[
\mathbb P\bigl(\ker f_{S_p}=0\bigr)
=
p^{\rho_{m,n}}(1-p)^{N_{m,n}-\rho_{m,n}}
T_{M_{m,n}}\!\left(1,\frac{1}{1-p}\right).
\]
因此，随机屏幕的精确率完全由同一边界拟阵的单变量 Tutte 截面控制。
\end{proposition}
\begin{proof}
由 \(\partial_n\) 的单射性，\(\ker f_S=0\) 当且仅当 \(f_S\) 满秩，亦即
\[
r(S)=\rho_{m,n}.
\]
于是
\[
\mathbb P(\ker f_{S_p}=0)
=
\sum_{r(S)=\rho_{m,n}} p^{|S|}(1-p)^{N_{m,n}-|S|}.
\]
另一方面，按 Tutte 多项式定义，
\[
T_{M_{m,n}}(1,y)
=
\sum_{r(S)=\rho_{m,n}}(y-1)^{|S|-\rho_{m,n}}.
\]
取 \(y=(1-p)^{-1}\)，则 \(y-1=p/(1-p)\)。故
\[
p^{\rho_{m,n}}(1-p)^{N_{m,n}-\rho_{m,n}}
T_{M_{m,n}}\!\left(1,\frac{1}{1-p}\right)
=
\sum_{r(S)=\rho_{m,n}} p^{|S|}(1-p)^{N_{m,n}-|S|},
\]
即得所述公式。证毕。
\end{proof}

\begin{proposition}[加权最优补全等价于收缩拟阵上的最小权基]\label{prop:conclusion-screen-weighted-completion-contraction-minbasis}
\leanverified{paper\_conclusion\_screen\_weighted\_completion\_contraction\_minbasis}
给定非负代价函数
\[
\kappa:E_{m,n}\setminus S\longrightarrow \RR_{\ge 0}.
\]
则在一切使 \(S\cup T\) 精确的补全面
\[
T\subseteq E_{m,n}\setminus S
\]
中，\(\kappa\)-总代价最小者恰为收缩拟阵 \(M_{m,n}/S\) 的最小权基。特别地，一切最优补全都具有同样的基数 \(\ell(S)\)，并且拟阵贪心算法在 \(M_{m,n}/S\) 上给出严格最优解。
\end{proposition}
\begin{proof}
取 \(I\subseteq S\) 为任一极大独立子集，则
\[
r(I)=r(S),
\qquad
\operatorname{cl}_{M_{m,n}}(I)=\operatorname{cl}_{M_{m,n}}(S).
\]
由定理 \ref{thm:conclusion-screen-minimal-completion-family-independent-kernel} 与定理 \ref{thm:conclusion-screen-exactization-independent-kernel}，最优补全问题可等价写为：在收缩拟阵 \(M_{m,n}/I\) 的全部基中寻找最小 \(\kappa\)-权者。

而 \(S\setminus I\subseteq \operatorname{cl}_{M_{m,n}}(I)\)，故这些元素在 \(M_{m,n}/I\) 中全都成为 loop；删去这些 loop 后恰得到标准收缩 \(M_{m,n}/S\)。因此 \(M_{m,n}/I\) 与 \(M_{m,n}/S\) 具有同一基族，故最优补全正是 \(M_{m,n}/S\) 的最小权基。

所有基的基数都等于收缩拟阵的秩
\[
\rho_{m,n}-r(S)=\ell(S),
\]
故全部最优补全自动具有相同基数。最后，由拟阵的贪心定理，最小权基可由按权递增选取且保持独立的贪心算法得到，因而结论成立。证毕。
\end{proof}

\begin{conjecture}[精确屏幕阈值的码率锁定]\label{conj:conclusion-screen-exact-threshold-code-rate-locking}
对固定 \(n\)，存在一条唯一临界曲线 \(p_c(m,n)\) 使得对任意 \(\varepsilon\in(0,1)\)，当
\[
p\le (1-\varepsilon)p_c(m,n)
\]
时有
\[
\mathbb P\bigl(\ker f_{S_p}=0\bigr)\longrightarrow 0,
\]
而当
\[
p\ge (1+\varepsilon)p_c(m,n)
\]
时有
\[
\mathbb P\bigl(\ker f_{S_p}=0\bigr)\longrightarrow 1.
\]
并且该临界曲线满足
\[
p_c(m,n)\sim \frac1n
\qquad(m\to\infty).
\]
\end{conjecture}

该猜想的结构依据是：一方面，命题 \ref{prop:conclusion-screen-bernoulli-exactness-tutte-law} 已表明随机精确率完全由 \(T_{M_{m,n}}(1,\cdot)\) 控制；另一方面，由定理 \ref{thm:spg-dyadic-boundary-image-ldpc}，
\[
\frac{\rho_{m,n}}{N_{m,n}}
=
\frac{2^m}{n(2^m+1)}
\longrightarrow
\frac1n.
\]
因此若屏幕精确性确有 sharp threshold，则其唯一自然候选应与同一边界阴影的码率极限锁定。

综上，screen exactification 的几种看似分离的量在固定分辨率上进一步塌缩为同一审计热力学：Tutte 多项式统一编码全部屏幕的补全数与冗余度分层；可见缺口的每一个单位严格等价于一条相对顶维同调轴与一对 prime-register 审计素数；Bernoulli 随机精确率退化为单一 Tutte 截面；而加权最优补全则不再是启发式搜索，而是收缩拟阵上的最小权基问题。于是，边界拟阵、相对同调、LDPC 阴影与 prime-register 协议在本文的 screen 侧并非平行接口，而是同一对象的四种热力学坐标。
