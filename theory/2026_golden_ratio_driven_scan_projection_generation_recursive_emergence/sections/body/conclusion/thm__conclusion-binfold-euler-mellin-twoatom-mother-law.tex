\begin{theorem}[稳定层均匀基线的两原子 Euler--Laplace 极限]\label{thm:conclusion-binfold-uniform-euler-laplace-twoatom}
沿用
\[
d_m(w):=d_m^{\mathrm{bin}}(w),
\qquad
u_m(w):=\frac{1}{|X_m|},
\]
并令 \(U_m\sim u_m\)。定义尺度化多重度
\[
Y_m:=\left(\frac{\varphi}{2}\right)^m d_m(U_m).
\]
则有分布收敛
\[
Y_m\Rightarrow Y,
\qquad
\PP(Y=1)=\frac{1}{\varphi},
\qquad
\PP(Y=\varphi^{-1})=\frac{1}{\varphi^2}.
\]
更强地，对任意紧集 \(K\subset\CC\)，有局部一致收敛
\[
\sup_{t\in K}
\left|
\EE\!\left[e^{tY_m}\right]
-
\left(
\frac{1}{\varphi}e^t+\frac{1}{\varphi^2}e^{t/\varphi}
\right)
\right|
\longrightarrow 0.
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-foldbin-uniform-layer-degeneracy-two-atom-law} 已给出
\[
Y_m\Rightarrow
\frac{1}{\varphi}\delta_1+\frac{1}{\varphi^2}\delta_{\varphi^{-1}},
\]
故分布收敛部分已成立。为得到整个 Euler--Laplace 极限，只需回到该定理证明中的一致两态展开：
\[
\left(\frac{\varphi}{2}\right)^m d_m(w)
=
\varphi^{-w_m}+O\!\left(\left(\frac{\varphi}{2}\right)^m\right)
\qquad (w\in X_m),
\]
且误差在 \(w\) 上一致。于是存在 \(\varepsilon_m\to 0\) 使
\[
Y_m=\varphi^{-U_m(m)}+\varepsilon_m,
\qquad
|\varepsilon_m|\le \varepsilon_m.
\]
另一方面，
\[
\PP(U_m(m)=0)=\frac{F_{m+1}}{F_{m+2}}\to \frac{1}{\varphi},
\qquad
\PP(U_m(m)=1)=\frac{F_m}{F_{m+2}}\to \frac{1}{\varphi^2}.
\]

固定紧集 \(K\subset\CC\)。由于 \(Y_m\) 终究落在某个固定紧区间内，存在常数 \(C_K>0\) 使对所有充分大的 \(m\)、所有 \(t\in K\) 与所有 \(y,z\) 属于该区间都有
\[
|e^{ty}-e^{tz}|\le C_K|y-z|.
\]
故
\[
\sup_{t\in K}
\left|
\EE[e^{tY_m}]
-
\frac{F_{m+1}}{F_{m+2}}e^t
-
\frac{F_m}{F_{m+2}}e^{t/\varphi}
\right|
\le
C_K\varepsilon_m
\longrightarrow 0.
\]
再令 \(m\to\infty\) 并使用 Fibonacci 比值极限，即得所示局部一致收敛。证毕。
\end{proof}

\begin{theorem}[全碰撞层级的 Euler--Borel 塌缩与 rank-two 刚性]\label{thm:conclusion-binfold-euler-borel-ranktwo-collapse}
对每个整数 \(q\ge 0\)，记
\[
S_q(m):=\sum_{w\in X_m} d_m(w)^q,
\]
并定义 Euler--Borel 变换
\[
\mathcal B_m(t)
:=
\sum_{q\ge 0}\frac{t^q}{q!}
\frac{(\varphi/2)^{qm}}{|X_m|}S_q(m)
=
\frac{1}{|X_m|}
\sum_{w\in X_m}
\exp\!\left(
t\left(\frac{\varphi}{2}\right)^m d_m(w)
\right).
\]
则在任意紧集上一致有
\[
\mathcal B_m(t)\longrightarrow
\mathcal B(t):=
\frac{1}{\varphi}e^t+\frac{1}{\varphi^2}e^{t/\varphi}.
\]
从而对每个整数 \(q\ge 0\)，有
\[
\frac{(\varphi/2)^{qm}}{|X_m|}S_q(m)\longrightarrow
a_q:=\frac{1}{\varphi}+\frac{1}{\varphi^{q+2}}.
\]
更进一步，极限矩列 \((a_q)_{q\ge 0}\) 满足二阶递推
\[
a_{q+2}
-
\left(1+\frac{1}{\varphi}\right)a_{q+1}
+
\frac{1}{\varphi}a_q
=0
\qquad(q\ge 0),
\]
其全部 \(3\times 3\) Hankel 子式均为零；并且 \(\mathcal B\) 是微分方程
\[
(D-1)(D-\varphi^{-1})f=0
\]
在初值
\[
f(0)=1,
\qquad
f'(0)=\frac{1}{\varphi}+\frac{1}{\varphi^3}
\]
下的唯一 entire 解。
\end{theorem}
\begin{proof}
由定义，
\[
\mathcal B_m(t)=\EE[e^{tY_m}],
\]
故定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom} 立即给出
\[
\mathcal B_m(t)\longrightarrow \mathcal B(t)
\]
在任意紧集上一致成立。另一方面，
\[
\frac{(\varphi/2)^{qm}}{|X_m|}S_q(m)=\EE[Y_m^q].
\]
由于 \(Y_m\) 终究一致有界，故对每个固定 \(q\) 有
\[
\EE[Y_m^q]\longrightarrow \EE[Y^q]
=
\frac{1}{\varphi}\cdot 1^q+\frac{1}{\varphi^2}\cdot \varphi^{-q}
=
\frac{1}{\varphi}+\frac{1}{\varphi^{q+2}}.
\]
这便得到 \(a_q\) 的显式公式。

再由
\[
a_q=
\frac{1}{\varphi}\cdot 1^q+\frac{1}{\varphi^2}\cdot (\varphi^{-1})^q,
\]
可见 \((a_q)\) 是特征根 \(1,\varphi^{-1}\) 所对应的两条几何列的线性组合，故满足特征多项式
\[
(\lambda-1)(\lambda-\varphi^{-1})
=
\lambda^2-\left(1+\frac{1}{\varphi}\right)\lambda+\frac{1}{\varphi}
\]
所诱导的二阶递推。于是 Hankel 矩阵秩至多为 \(2\)，全部 \(3\times 3\) 子式自动消失。

最后，
\[
\mathcal B(t)=\frac{1}{\varphi}e^t+\frac{1}{\varphi^2}e^{t/\varphi}
\]
直接满足
\[
(D-1)(D-\varphi^{-1})\mathcal B=0,
\]
且
\[
\mathcal B(0)=\frac{1}{\varphi}+\frac{1}{\varphi^2}=1,
\qquad
\mathcal B'(0)=\frac{1}{\varphi}+\frac{1}{\varphi^3}.
\]
对常系数线性微分方程，给定初值的 entire 解唯一，故结论成立。证毕。
\end{proof}

\begin{theorem}[微态推前极限律是稳定层极限律的精确尺寸偏置]\label{thm:conclusion-binfold-pushforward-exact-size-bias}
令 \(W_m\sim p_m^{\mathrm{bin}}\)，并定义
\[
Z_m:=\left(\frac{\varphi}{2}\right)^m d_m(W_m).
\]
则 \(Z_m\Rightarrow Z\)，其中
\[
\PP(Z=1)=\frac{\varphi}{\sqrt5},
\qquad
\PP(Z=\varphi^{-1})=\frac{1}{\varphi\sqrt5}.
\]
该极限律与定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom} 中的 \(Y\) 满足严格的尺寸偏置关系
\[
\PP(Z=y)=\frac{y\,\PP(Y=y)}{\EE Y},
\qquad
y\in\{1,\varphi^{-1}\}.
\]
等价地，对任意有界 Borel 函数 \(f\)，有
\[
\EE[f(Z)]
=
\frac{\EE[Yf(Y)]}{\EE Y}.
\]
特别地，
\[
\EE Y=\frac{1}{\varphi}+\frac{1}{\varphi^3}=\frac{\sqrt5}{\varphi^2},
\]
并且
\[
\EE[e^{tZ}]
=
\frac{\mathcal B'(t)}{\mathcal B'(0)}
=
\frac{\varphi}{\sqrt5}e^t+\frac{1}{\varphi\sqrt5}e^{t/\varphi}.
\]
\end{theorem}
\begin{proof}
推论 \ref{cor:fold-bin-two-point-limit-law} 已给出 \(Z_m\Rightarrow Z\) 及其两点概率。另一方面，由定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom}，
\[
\PP(Y=1)=\frac{1}{\varphi},
\qquad
\PP(Y=\varphi^{-1})=\frac{1}{\varphi^2},
\]
故
\[
\EE Y=\frac{1}{\varphi}+\frac{1}{\varphi^3}
=\frac{\sqrt5}{\varphi^2}.
\]
于是
\[
\frac{1\cdot \PP(Y=1)}{\EE Y}
=
\frac{\varphi^{-1}}{\sqrt5/\varphi^2}
=
\frac{\varphi}{\sqrt5},
\]
\[
\frac{\varphi^{-1}\PP(Y=\varphi^{-1})}{\EE Y}
=
\frac{\varphi^{-3}}{\sqrt5/\varphi^2}
=
\frac{1}{\varphi\sqrt5},
\]
这与 \(Z\) 的极限概率完全一致，故尺寸偏置关系成立。

对有界 Borel 函数 \(f\)，两点分布上
\[
\EE[f(Z)]
=
\sum_{y\in\{1,\varphi^{-1}\}}
f(y)\frac{y\,\PP(Y=y)}{\EE Y}
=
\frac{\EE[Yf(Y)]}{\EE Y}.
\]
取 \(f(y)=e^{ty}\) 即得
\[
\EE[e^{tZ}]
=
\frac{\EE[Ye^{tY}]}{\EE Y}
=
\frac{\mathcal B'(t)}{\mathcal B'(0)}.
\]
再代入定理 \ref{thm:conclusion-binfold-euler-borel-ranktwo-collapse} 中的
\[
\mathcal B'(t)=\frac{1}{\varphi}e^t+\frac{1}{\varphi^3}e^{t/\varphi},
\qquad
\mathcal B'(0)=\frac{\sqrt5}{\varphi^2},
\]
便得到所示显式闭式。证毕。
\end{proof}

\begin{theorem}[相对密度二原子极限的归一化同一性]\label{thm:conclusion-binfold-relative-density-normalized-mother-law}
定义相对稳定层均匀基线的密度比
\[
R_m(w):=\frac{p_m^{\mathrm{bin}}(w)}{u_m(w)}
=
|X_m|\frac{d_m(w)}{2^m},
\qquad
w\in X_m,
\]
并把 \(R_m(U_m)\) 视为 \(u_m\)-分布下的随机变量。则有
\[
R_m(U_m)\Rightarrow \frac{Y}{\EE Y}.
\]
换言之，已有两原子相对密度常数
\[
r_0:=\frac{\varphi^2}{\sqrt5},
\qquad
r_1:=\frac{\varphi}{\sqrt5}
\]
恰满足
\[
r_0=\frac{1}{\EE Y},
\qquad
r_1=\frac{\varphi^{-1}}{\EE Y}.
\]
因此，对任意凸函数 \(f:(0,\infty)\to\RR\) 满足 \(f(1)=0\)，都有
\[
\lim_{m\to\infty}D_f\!\bigl(p_m^{\mathrm{bin}}\Vert u_m\bigr)
=
\EE\!\left[f\!\left(\frac{Y}{\EE Y}\right)\right]
=
\frac{1}{\varphi}f\!\left(\frac{\varphi^2}{\sqrt5}\right)
+
\frac{1}{\varphi^2}f\!\left(\frac{\varphi}{\sqrt5}\right).
\]
特别地，
\[
\log\frac{r_0}{r_1}=\log\varphi.
\]
\end{theorem}
\begin{proof}
由定义与 \(|X_m|=F_{m+2}\)，有
\[
R_m(U_m)
=
\frac{F_{m+2}}{2^m}d_m(U_m)
=
\frac{F_{m+2}}{\varphi^m}
\left(\frac{\varphi}{2}\right)^m d_m(U_m)
=
\frac{F_{m+2}}{\varphi^m}\,Y_m.
\]
而 Binet 公式给出
\[
\frac{F_{m+2}}{\varphi^m}\longrightarrow \frac{\varphi^2}{\sqrt5}
=
\frac{1}{\EE Y}.
\]
再结合定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom} 中 \(Y_m\Rightarrow Y\)，便得到
\[
R_m(U_m)\Rightarrow \frac{Y}{\EE Y}.
\]
于是极限支撑点正是
\[
\frac{1}{\EE Y}=\frac{\varphi^2}{\sqrt5},
\qquad
\frac{\varphi^{-1}}{\EE Y}=\frac{\varphi}{\sqrt5}.
\]

另一方面，
\[
D_f\!\bigl(p_m^{\mathrm{bin}}\Vert u_m\bigr)
=
\EE_{u_m}[f(R_m)].
\]
由于 \(R_m(U_m)\) 终究一致有界，故由弱收敛与连续泛函传递得到
\[
D_f\!\bigl(p_m^{\mathrm{bin}}\Vert u_m\bigr)
\longrightarrow
\EE\!\left[f\!\left(\frac{Y}{\EE Y}\right)\right].
\]
把 \(Y\) 的两点分布代入即可得到最后的闭式；这与定理 \ref{thm:conclusion-binfold-twoatom-information-geometry-universality} 的两原子公式完全一致，只是把其重新写成了单一母分布 \(Y\) 的归一化像。最后
\[
\log\frac{r_0}{r_1}
=
\log\varphi
\]
由显式支撑点立即成立。证毕。
\end{proof}

\begin{theorem}[两原子 Mellin--escort 半群]\label{thm:conclusion-binfold-mellin-escort-semigroup}
\leanverified{paper\_conclusion\_binfold\_mellin\_escort\_semigroup}
对任意 \(s\ge 0\)，定义 escort 分布
\[
\pi_{m,s}(w):=\frac{d_m(w)^s}{S_s(m)},
\qquad
S_s(m):=\sum_{w\in X_m}d_m(w)^s,
\]
并令 \(W_m^{\langle s\rangle}\sim \pi_{m,s}\)。再定义
\[
Y_m^{\langle s\rangle}
:=
\left(\frac{\varphi}{2}\right)^m d_m\!\left(W_m^{\langle s\rangle}\right).
\]
则
\[
Y_m^{\langle s\rangle}\Rightarrow Y^{\langle s\rangle},
\]
其中极限分布仍支撑于 \(\{1,\varphi^{-1}\}\)，且
\[
\PP(Y^{\langle s\rangle}=1)=\frac{\varphi^{s+1}}{1+\varphi^{s+1}},
\qquad
\PP(Y^{\langle s\rangle}=\varphi^{-1})=\frac{1}{1+\varphi^{s+1}}.
\]
等价地，
\[
\PP(Y^{\langle s\rangle}=y)
=
\frac{y^s\,\PP(Y=y)}{\EE[Y^s]},
\qquad
y\in\{1,\varphi^{-1}\}.
\]
其 entire 矩母函数为
\[
\mathcal B_s(t):=\EE[e^{tY^{\langle s\rangle}}]
=
\frac{\varphi^{-1}e^t+\varphi^{-(s+2)}e^{t/\varphi}}
{\varphi^{-1}+\varphi^{-(s+2)}}.
\]
并且 Mellin tilt 满足半群律：对任意 \(s,t\ge 0\)，有
\[
\bigl(Y^{\langle s\rangle}\bigr)^{\langle t\rangle}=Y^{\langle s+t\rangle}.
\]
特别地，
\[
Y^{\langle 0\rangle}=Y,
\qquad
Y^{\langle 1\rangle}=Z.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-binfold-escort-blackwell-equivalence} 及其证明中的显式末位极限权重，
\[
\PP\!\bigl(W_m^{\langle s\rangle}(m)=1\bigr)\to \frac{1}{1+\varphi^{s+1}},
\qquad
\PP\!\bigl(W_m^{\langle s\rangle}(m)=0\bigr)\to \frac{\varphi^{s+1}}{1+\varphi^{s+1}}.
\]
再结合定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom} 证明中同样使用的一致两态展开
\[
\left(\frac{\varphi}{2}\right)^m d_m(w)
=
\varphi^{-w_m}+O\!\left(\left(\frac{\varphi}{2}\right)^m\right),
\]
便知 \(Y_m^{\langle s\rangle}\) 在两扇区上分别逼近 \(1\) 与 \(\varphi^{-1}\)，从而得到
\[
\PP(Y^{\langle s\rangle}=1)=\frac{\varphi^{s+1}}{1+\varphi^{s+1}},
\qquad
\PP(Y^{\langle s\rangle}=\varphi^{-1})=\frac{1}{1+\varphi^{s+1}}.
\]

另一方面，由定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom}，
\[
\PP(Y=1)=\frac{1}{\varphi},
\qquad
\PP(Y=\varphi^{-1})=\frac{1}{\varphi^2},
\qquad
\EE[Y^s]=\frac{1}{\varphi}+\varphi^{-(s+2)}.
\]
于是
\[
\frac{1^s\PP(Y=1)}{\EE[Y^s]}
=
\frac{\varphi^{-1}}{\varphi^{-1}+\varphi^{-(s+2)}}
=
\frac{\varphi^{s+1}}{1+\varphi^{s+1}},
\]
\[
\frac{\varphi^{-s}\PP(Y=\varphi^{-1})}{\EE[Y^s]}
=
\frac{\varphi^{-(s+2)}}{\varphi^{-1}+\varphi^{-(s+2)}}
=
\frac{1}{1+\varphi^{s+1}}.
\]
这便得到 Mellin tilt 公式。

矩母函数只需对该两点分布直接求和：
\[
\mathcal B_s(t)
=
\frac{\varphi^{s+1}}{1+\varphi^{s+1}}e^t
+
\frac{1}{1+\varphi^{s+1}}e^{t/\varphi}
=
\frac{\varphi^{-1}e^t+\varphi^{-(s+2)}e^{t/\varphi}}
{\varphi^{-1}+\varphi^{-(s+2)}}.
\]
最后，对任意 \(s,t\ge 0\)，再次作 escort 倾斜便得到
\[
\PP\!\left((Y^{\langle s\rangle})^{\langle t\rangle}=y\right)
=
\frac{y^t\PP(Y^{\langle s\rangle}=y)}{\EE_{Y^{\langle s\rangle}}[Y^t]}
=
\frac{y^{s+t}\PP(Y=y)}{\EE[Y^{s+t}]},
\]
这正是 \(Y^{\langle s+t\rangle}\) 的定义。故半群律成立。取 \(s=0\) 与 \(s=1\) 即得最后两式；其中 \(s=1\) 正与定理 \ref{thm:conclusion-binfold-pushforward-exact-size-bias} 的尺寸偏置律一致。证毕。
\end{proof}

\begin{conjecture}[两态骨架的 Euler--Mellin 刚性]\label{conj:conclusion-binfold-two-scale-euler-mellin-rigidity}
设一族折叠系统保持 bin-fold 所呈现的两态末位骨架不变，并且在适当重标架下，纤维多重度仍只出现两个主尺度
\[
1,\qquad \varphi^{-1}.
\]
则其 rescaled collision hierarchy 的 Euler--Borel 极限应当必然落在二指数 entire 类
\[
t\longmapsto A e^t+B e^{t/\varphi},
\qquad
A,B>0,\quad A+B=1,
\]
而全部 escort 极限族则应由这一母函数经 Mellin 倾斜唯一生成。等价地，一旦两态骨架被锁定，则矩谱、推前极限律、相对密度两原子账本、全部 \(f\)-散度极限与 escort 温度流都不再具有额外独立自由度，而只是同一个 rank-two Prony germ 的不同投影。
\end{conjecture}

\noindent\emph{证据.}
定理 \ref{thm:conclusion-binfold-uniform-euler-laplace-twoatom} 已把稳定层均匀基线压缩为两原子 Euler--Laplace 极限；定理 \ref{thm:conclusion-binfold-euler-borel-ranktwo-collapse} 进一步表明整个碰撞矩塔在 Euler--Borel 口径下坍缩为 rank-two entire 物件；定理 \ref{thm:conclusion-binfold-pushforward-exact-size-bias} 说明微态推前极限只是该母分布的尺寸偏置像；定理 \ref{thm:conclusion-binfold-relative-density-normalized-mother-law} 又把相对均匀基线的全部两原子常数重写成同一母分布的归一化像；最后，定理 \ref{thm:conclusion-binfold-mellin-escort-semigroup} 证明整个 escort 温度族正是由 Mellin 倾斜生成的一条半群轨道。现有全部闭式都指向：两态骨架一经固定，整座极限信息几何塔便已在解析上锁死。
