\begin{theorem}[以 \texorpdfstring{$\log p_t$}{log p_t} 为权的尾部算子常数收敛于 \texorpdfstring{$f(1)$}{f(1)}]\label{thm:conclusion-logp-weighted-tail-operator-norm}
\leanverified{paper\_conclusion\_logp\_weighted\_tail\_operator\_norm}
令 \((p_t)_{t\ge 1}\) 为素数序列，并令
\[
\mathcal P_{\ZZ}:=\ZZ^{(\NN)}
\]
为有限支撑整数寄存器。
定义加权范数
\[
\|x\|_{\log p}:=\sum_{t\ge 1}(\log p_t)\,|x_t|
\qquad(x\in\mathcal P_{\ZZ}),
\]
并令尾部子群
\[
\mathcal P_{\ZZ}^{(\ge T)}:=\bigoplus_{t\ge T}\ZZ\,\mathbf e_t.
\]
对 \(f(z)=\sum_{j=0}^{d}a_j z^j\in\NN[z]\) 及其算子 \(\Phi_f=\Theta(f)\)（定理 \ref{thm:conclusion-shift-commuting-algorithms-polynomial}），定义尾部算子范数
\[
L_T:=\sup_{0\neq x\in\mathcal P_{\ZZ}^{(\ge T)}}\frac{\|\Phi_f(x)\|_{\log p}}{\|x\|_{\log p}}.
\]
则极限存在并满足
\[
\boxed{\ \lim_{T\to\infty}L_T=f(1)=\sum_{j=0}^{d}a_j\ }.
\]
\end{theorem}
\begin{proof}
对任意 \(x\in\mathcal P_{\ZZ}\)，由 \(\Phi_f=\sum_{j=0}^d a_j\,\mathrm{Sh}^j\) 与 \(a_j\ge 0\) 得
\[
\|\Phi_f(x)\|_{\log p}
=\left\|\sum_{j=0}^d a_j\,\mathrm{Sh}^j(x)\right\|_{\log p}
\le \sum_{j=0}^d a_j\,\|\mathrm{Sh}^j(x)\|_{\log p}.
\]
而
\[
\|\mathrm{Sh}^j(x)\|_{\log p}
=\sum_{t\ge 1}(\log p_t)\,|(\mathrm{Sh}^j x)_t|
=\sum_{t\ge 1}(\log p_{t+j})\,|x_t|.
\]
若 \(x\in \mathcal P_{\ZZ}^{(\ge T)}\)，则求和仅在 \(t\ge T\) 上进行，因此
\[
\|\mathrm{Sh}^j(x)\|_{\log p}
\le \left(\sup_{t\ge T}\frac{\log p_{t+j}}{\log p_t}\right)\|x\|_{\log p}.
\]
令
\(
R_{T,j}:=\sup_{t\ge T}\frac{\log p_{t+j}}{\log p_t}
\)。
则
\[
\frac{\|\Phi_f(x)\|_{\log p}}{\|x\|_{\log p}}
\le \sum_{j=0}^d a_j R_{T,j},
\qquad
L_T\le \sum_{j=0}^d a_j R_{T,j}.
\]
由素数定理（例如 \cite{Apostol1976}）可得对固定 \(j\)，当 \(T\to\infty\) 时 \(R_{T,j}\to 1\)，故
\[
\limsup_{T\to\infty}L_T\le \sum_{j=0}^d a_j=f(1).
\]
\par
另一方面，对任意 \(t\ge T\)，取 \(x=\mathbf e_t\)，则
\[
\frac{\|\Phi_f(\mathbf e_t)\|_{\log p}}{\|\mathbf e_t\|_{\log p}}
=\frac{\sum_{j=0}^d a_j\log p_{t+j}}{\log p_t}.
\]
取 \(t\to\infty\) 并再次使用 \(\log p_{t+j}/\log p_t\to 1\) 得
\(
\liminf_{t\to\infty}\frac{\|\Phi_f(\mathbf e_t)\|_{\log p}}{\|\mathbf e_t\|_{\log p}}=f(1)
\)。
由 \(L_T\) 为对 \(t\ge T\) 的上确界，推出
\(
\liminf_{T\to\infty}L_T\ge f(1)
\)。
合并上下极限即得 \(\lim_{T\to\infty}L_T=f(1)\)。证毕。
\end{proof}
