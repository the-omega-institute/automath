\begin{theorem}[完成化符号的刚性与中心阶数的奇偶律]\label{thm:conclusion-toyrh-completion-sign-center-parity}
\leanverified{paper\_conclusion\_toyrh\_completion\_sign\_center\_parity}
在完成化函数满足
\[
\Xi_M(s)=C\,\Xi_M(1-s)
\qquad (C\neq 0)
\]
的设定下，必有
\[
C^2=1.
\]
记
\[
\varepsilon:=C\in\{\pm 1\},
\qquad
\widehat\Xi_M(u):=\Xi_M\!\left(\frac12+u\right).
\]
则
\[
\widehat\Xi_M(-u)=\varepsilon\,\widehat\Xi_M(u).
\]
特别地，若 \(\operatorname{ord}_{s=1/2}\Xi_M\) 表示中心点的 Laurent 阶数（零点取正，极点取负），则
\[
\operatorname{ord}_{1/2}\Xi_M\equiv
\begin{cases}
0 \pmod 2,& \varepsilon=+1,\\
1 \pmod 2,& \varepsilon=-1.
\end{cases}
\]
\end{theorem}
\begin{proof}
对函数方程再施一次 \(s\mapsto 1-s\)，得
\[
\Xi_M(s)=C\,\Xi_M(1-s)=C^2\Xi_M(s).
\]
由于 \(\Xi_M\not\equiv 0\)，故必有 \(C^2=1\)。

将 \(s=\frac12+u\) 代入，得到
\[
\widehat\Xi_M(-u)=\Xi_M\!\left(\frac12-u\right)
=\varepsilon\,\Xi_M\!\left(\frac12+u\right)
=\varepsilon\,\widehat\Xi_M(u).
\]
于是 \(\widehat\Xi_M\) 关于 \(u=0\) 为偶函数或奇函数。其 Laurent 展开因而只含偶次幂或奇次幂，最低非零幂次的奇偶性遂与 \(\varepsilon\) 相同。证毕。
\end{proof}

\begin{corollary}[实系数互反核在临界线上的实/虚二分律]\label{cor:conclusion-toyrh-real-kernel-critical-line-real-imaginary-dichotomy}
\leanverified{paper\_conclusion\_toyrh\_real\_kernel\_critical\_line\_real\_imaginary\_dichotomy}
在定理 \ref{thm:conclusion-toyrh-completion-sign-center-parity} 的条件下，进一步假设 \(M\) 的系数为实数。则对一切 \(t\in \RR\)，有
\[
\overline{\Xi_M\!\left(\frac12+it\right)}
=
\varepsilon\,\Xi_M\!\left(\frac12+it\right).
\]
因此：
\begin{enumerate}
  \item 若 \(\varepsilon=+1\)，则
  \[
  \Xi_M\!\left(\frac12+it\right)\in \RR;
  \]
  \item 若 \(\varepsilon=-1\)，则
  \[
  \Xi_M\!\left(\frac12+it\right)\in i\RR,
  \]
  并且中心点 \(s=\frac12\) 必为奇阶零点或奇阶极点。
\end{enumerate}
\end{corollary}
\begin{proof}
实系数保证
\[
\overline{\Xi_M(\overline s)}=\Xi_M(s).
\]
取 \(s=\frac12+it\)，便有
\[
\overline{\Xi_M\!\left(\frac12+it\right)}
=
\Xi_M\!\left(\frac12-it\right).
\]
再由定理 \ref{thm:conclusion-toyrh-completion-sign-center-parity}，
\[
\Xi_M\!\left(\frac12-it\right)
=
\varepsilon\,\Xi_M\!\left(\frac12+it\right),
\]
于是结论成立。若 \(\varepsilon=+1\)，值等于其共轭，故为实数；若 \(\varepsilon=-1\)，则值等于其共轭的负，故为纯虚。中心阶数奇偶结论仍由定理 \ref{thm:conclusion-toyrh-completion-sign-center-parity} 给出。证毕。
\end{proof}

\begin{theorem}[临界线极点在周期圆上的精确极限测度]\label{thm:conclusion-toyrh-critical-line-atomic-limit-measure}
\leanverified{paper\_conclusion\_toyrh\_critical\_line\_atomic\_limit\_measure}
设临界圆上的达界特征值按不同相位写成
\[
\mu_\beta=\sqrt{\lambda}\,e^{i\theta_\beta},
\qquad
0\le \theta_1<\cdots<\theta_r<2\pi,
\]
并记其重数为 \(m_\beta\)，故
\[
\sum_{\beta=1}^r m_\beta=B.
\]
令
\[
\Delta:=\frac{2\pi}{\log\lambda},
\qquad
\tau_\beta:=\frac{\theta_\beta}{\log\lambda}\in [0,\Delta),
\qquad
\TT_\Delta:=\RR/\Delta\ZZ.
\]
定义周期圆上的重标计数测度
\[
\mu_T^{\mathrm{crit}}
:=
\frac1T
\sum_{\substack{s_0\in\operatorname{Poles}(Z_M)\\ \Re s_0=1/2,\ 0<\Im s_0\le T}}
\operatorname{ord}_{s_0}(Z_M)\,
\delta_{\Im s_0 \bmod \Delta}.
\]
则有弱收敛
\[
\mu_T^{\mathrm{crit}}
\xrightarrow[T\to\infty]{w^\ast}
\frac{\log\lambda}{2\pi}
\sum_{\beta=1}^r m_\beta\,\delta_{\tau_\beta}.
\]
因此，有限维 toy-RH 模型在临界线上的全部相位信息并不形成连续云，而严格压缩为有限原子簇。
\end{theorem}
\begin{proof}
由正文的临界线极点描述，临界线极点正是
\[
s=\frac12+i(\tau_\beta+k\Delta),
\qquad
k\in\ZZ,\ \beta=1,\dots,r,
\]
其中第 \(\beta\) 条竖向等差列按重数 \(m_\beta\) 计。对固定 \(\beta\)，区间 \((0,T]\) 内可见项数满足
\[
\#\{k:\ 0<\tau_\beta+k\Delta\le T\}
=
\frac{T}{\Delta}+O(1).
\]
于是对任意连续测试函数 \(\varphi\in C(\TT_\Delta)\)，有
\[
\int \varphi\,\dd\mu_T^{\mathrm{crit}}
=
\frac1T
\sum_{\beta=1}^{r}m_\beta
\sum_{\substack{k\in\ZZ\\ 0<\tau_\beta+k\Delta\le T}}
\varphi(\tau_\beta)
=
\sum_{\beta=1}^r m_\beta
\Bigl(\frac1\Delta+O(T^{-1})\Bigr)\varphi(\tau_\beta).
\]
令 \(T\to\infty\)，得到
\[
\int \varphi\,\dd\mu_T^{\mathrm{crit}}
\to
\frac1\Delta\sum_{\beta=1}^r m_\beta\varphi(\tau_\beta)
=
\frac{\log\lambda}{2\pi}\sum_{\beta=1}^r m_\beta\varphi(\tau_\beta),
\]
即所述弱收敛。证毕。
\end{proof}

\begin{theorem}[临界线最近邻间隔的有限字母表]\label{thm:conclusion-toyrh-critical-line-nearest-neighbor-finite-alphabet}
\leanverified{paper\_conclusion\_toyrh\_critical\_line\_nearest\_neighbor\_finite\_alphabet}
沿用定理 \ref{thm:conclusion-toyrh-critical-line-atomic-limit-measure} 的记号。对 distinct 相位
\[
0\le \tau_1<\tau_2<\cdots<\tau_r<\Delta
\]
定义
\[
g_j:=\tau_{j+1}-\tau_j\quad(1\le j<r),
\qquad
g_r:=\Delta-\tau_r+\tau_1.
\]
则临界线极点的不同位置按虚部升序排列后，最近邻间隔严格周期重复于有限字母表
\[
\mathcal G:=\{g_1,\dots,g_r\}.
\]
若同时计入同一点的重数，则最近邻间隔字母表变为
\[
\mathcal G\cup\{0\}.
\]
在 unfolded 归一化
\[
u\longmapsto \frac{r}{\Delta}u
\]
下，间隔字母表仍然有限。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-toyrh-critical-line-atomic-limit-measure}，临界线极点的 distinct ordinates 是有限个等差列
\[
\tau_\beta+\Delta\ZZ
\qquad (\beta=1,\dots,r)
\]
的并。将一个长度为 \(\Delta\) 的基本周期内的相位按升序写成
\[
\tau_1<\tau_2<\cdots<\tau_r,
\]
则沿虚轴向上的 distinct 点序列正是
\[
\tau_1,\tau_2,\dots,\tau_r,\tau_1+\Delta,\tau_2+\Delta,\dots
\]
的无限重复。故最近邻间隔依次为
\[
g_1,\dots,g_r,g_1,\dots,g_r,\dots
\]
严格周期重复。

若某一相位 \(\tau_\beta\) 的极点重数 \(m_\beta>1\)，则在 multiset 口径下，同一点会额外贡献零间隔，于是字母表加入 \(0\)。unfolding 只做常数倍缩放，不改变字母表的有限性。证毕。
\end{proof}

\begin{theorem}[临界线二点差分统计的精确 Bragg 梳极限]\label{thm:conclusion-toyrh-critical-line-bragg-comb}
\leanverified{paper\_conclusion\_toyrh\_critical\_line\_bragg\_comb}
沿用定理 \ref{thm:conclusion-toyrh-critical-line-atomic-limit-measure}。定义临界线二点差分测度
\[
\nu_T^{\mathrm{crit}}
:=
\frac1T
\sum_{\substack{s,s'\in\operatorname{Poles}(Z_M)\\
\Re s=\Re s'=1/2\\
0<\Im s,\Im s'\le T}}
\operatorname{ord}_s(Z_M)\operatorname{ord}_{s'}(Z_M)\,
\delta_{\Im s-\Im s'}.
\]
则有 vague 极限
\[
\nu_T^{\mathrm{crit}}
\xrightarrow[T\to\infty]{v}
\frac{\log\lambda}{2\pi}
\sum_{\beta,\gamma=1}^{r}
m_\beta m_\gamma
\sum_{n\in\ZZ}\delta_{\tau_\beta-\tau_\gamma+n\Delta}.
\]
因此，有限维 toy-RH 模型的临界线二点统计纯粹由一条显式 Bragg 梳给出，其纯点部分占据全部极限质量。
\end{theorem}
\begin{proof}
第 \(\beta\) 与第 \(\gamma\) 两条极点列分别为
\[
\tau_\beta+\Delta\ZZ,
\qquad
\tau_\gamma+\Delta\ZZ.
\]
二者的差集恰为
\[
\tau_\beta-\tau_\gamma+\Delta\ZZ.
\]
给定 \(n\in\ZZ\)，满足
\[
(\tau_\beta+k\Delta)-(\tau_\gamma+\ell\Delta)
=
\tau_\beta-\tau_\gamma+n\Delta
\]
等价于 \(k-\ell=n\)。当两点同时落在 \((0,T]\) 内时，此类有序对个数为
\[
\frac{T}{\Delta}+O(1).
\]
再乘上重数 \(m_\beta m_\gamma\)，并对 \(\beta,\gamma\) 求和。于是对任意紧支撑测试函数 \(\varphi\)，有
\[
\int \varphi\,\dd \nu_T^{\mathrm{crit}}
=
\sum_{\beta,\gamma}
m_\beta m_\gamma
\Bigl(\frac1\Delta+O(T^{-1})\Bigr)
\sum_{n\in\ZZ}\varphi(\tau_\beta-\tau_\gamma+n\Delta).
\]
令 \(T\to\infty\) 即得所述极限。证毕。
\end{proof}

\begin{theorem}[属六曲线上的 trigonal 结构由 \texorpdfstring{$w$}{w} 的除子精确触发]\label{thm:conclusion-genus6-trigonal-triggered-by-w-divisor}
设 \(C\) 为非超椭圆属三平面四次，\(\pi:X\to C\) 为相应的二次分歧覆盖。记
\[
Q_0:=[0:1:0],\qquad Q_\infty:=[1:0:0]\in C,
\]
以及分歧点
\[
P_0=(0,1),\qquad P_\infty=[1:-1:0]\in C.
\]
则
\[
\operatorname{div}_C(x)=P_0+2Q_0-P_\infty-2Q_\infty.
\]
若 \(\widetilde P_0,\widetilde P_\infty\) 为 \(P_0,P_\infty\) 的唯一上点，而 \(Q_0^\pm,Q_\infty^\pm\) 为 \(Q_0,Q_\infty\) 的两条非分歧上点，则
\[
\operatorname{div}_X(w)=\widetilde P_0+Q_0^++Q_0^- -\widetilde P_\infty-Q_\infty^+-Q_\infty^-.
\]
特别地，函数 \(w\) 在 \(X\) 上的极点总次数为 \(3\)，故
\[
w:X\longrightarrow \PP^1
\]
是一个三重映射；从而 \(X\) 为 trigonal genus-\(6\) 曲线。
\end{theorem}
\begin{proof}
由正文对 \(x\) 的局部除子计算，\(x\) 在 \(P_0\) 为一次零点，在 \(P_\infty\) 为一次极点，在 \(Q_0\) 为二次零点，在 \(Q_\infty\) 为二次极点，且无其他零极点。因此
\[
\operatorname{div}_C(x)=P_0+2Q_0-P_\infty-2Q_\infty.
\]

又 \(Q(X)=Q(C)(\sqrt{x})\) 且 \(\pi^\ast x=w^2\)。在分歧点上，局部分歧指数为 \(2\)，故
\[
\operatorname{ord}_{\widetilde P_0}(w)=1,
\qquad
\operatorname{ord}_{\widetilde P_\infty}(w)=-1.
\]
在 \(Q_0,Q_\infty\) 上覆盖为 \'{e}tale，而 \(\operatorname{ord}_C(x)=\pm 2\)，故每个上点的 \(w\)-阶数都为 \(\pm 1\)。于是
\[
\operatorname{div}_X(w)=\widetilde P_0+Q_0^++Q_0^- -\widetilde P_\infty-Q_\infty^+-Q_\infty^-.
\]
其极点次数总和为 \(3\)，故 \(w\) 给出次数为 \(3\) 的射影映射。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$s$}{s}-投影在无穷远处的极点分拆与有限分歧度 \(20\)]\label{thm:conclusion-sprojection-infinity-pole-splitting-ramification20}
设 \(X\to \PP^1_s\) 为由 \(s\) 坐标给出的六叶覆盖。则在 \(s=\infty\) 上方恰有四个几何点：
\begin{enumerate}
  \item 一个 pole-type 点 \(Q_{\mathrm{pole}}\)，满足 \(w\sim -s\)，故
  \[
  \operatorname{ord}_{Q_{\mathrm{pole}}}(s)=-1;
  \]
  \item 一个 small-type 点 \(Q_{\mathrm{small}}\)，满足 \(w\sim s^{-1}\)，故
  \[
  \operatorname{ord}_{Q_{\mathrm{small}}}(s)=-1;
  \]
  \item 两个平方根型点 \(R_{\mathrm{re}},R_{\mathrm{im}}\)，分别由 \(w_\pm\) 与 \(w_{\pm i}\) 两对 Puiseux 分支给出，且
  \[
  \operatorname{ord}_{R_{\mathrm{re}}}(s)=\operatorname{ord}_{R_{\mathrm{im}}}(s)=-2.
  \]
\end{enumerate}
因而
\[
(s)_\infty=Q_{\mathrm{pole}}+Q_{\mathrm{small}}+2R_{\mathrm{re}}+2R_{\mathrm{im}}.
\]
特别地：
\begin{enumerate}
  \item \(\deg(s)=6\)；
  \item \(s=\infty\) 上恰有两个简单分歧点，即 \(R_{\mathrm{re}},R_{\mathrm{im}}\)；
  \item 该六重映射的有限部分总分歧度恰为 \(20\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由无穷远局部展开，整数 Laurent 型分支以 \(t=1/s\) 为局部参数，故在 \(Q_{\mathrm{pole}},Q_{\mathrm{small}}\) 处都有
\[
s=t^{-1},
\]
于是极点阶均为 \(1\)。

平方根型分支以 \(u=t^{1/2}=s^{-1/2}\) 为局部参数。成对出现的 \(w_\pm\) 与 \(w_{\pm i}\) 只是同一点在参数 \(u\mapsto -u\) 下的两种 Puiseux 表示，因此各对应一个几何点。对这两个点都有
\[
s=u^{-2},
\]
故极点阶为 \(2\)。由此得到
\[
(s)_\infty=Q_{\mathrm{pole}}+Q_{\mathrm{small}}+2R_{\mathrm{re}}+2R_{\mathrm{im}},
\]
其总次数为 \(6\)，所以 \(\deg(s)=6\)。

Riemann--Hurwitz 对 \(s:X\to \PP^1\) 给出
\[
2g(X)-2=-2\deg(s)+R.
\]
代入 \(g(X)=6\)、\(\deg(s)=6\)，得
\[
10=-12+R,
\qquad
R=22.
\]
无穷远处的两个平方根点各贡献 \(e_P-1=1\)，共贡献 \(2\)。故有限部分总分歧度为
\[
22-2=20.
\]
证毕。
\end{proof}

\begin{theorem}[六叶 \texorpdfstring{$s$}{s}-投影的甲板群平凡性]\label{thm:conclusion-sprojection-deck-group-triviality}
设
\[
K:=\QQ(s),
\qquad
L:=\QQ(X)=K(w),
\]
其中 \(w\) 满足正文中的六次方程。则
\[
\operatorname{Aut}_K(L)=1.
\]
换言之，映射
\[
X\longrightarrow \PP^1_s
\]
没有任何非平凡甲板变换。

因此，正文中的 involution
\[
\iota:(w,s)\mapsto (-w,-s)
\]
并不是 \(s\)-投影的 deck involution；它横向生成的是另一条二次商
\[
X\to C=X/\langle \iota\rangle.
\]
\end{theorem}
\begin{proof}
由正文给出的泛型伽罗瓦群计算，
\[
G:=\operatorname{Gal}(\Delta(w,s)/\QQ(s))\cong S_6.
\]
根域 \(L=K(w)\) 对应于 \(S_6\) 对六个根的自然作用中的一点稳定子
\[
H\cong S_5.
\]
一般地，
\[
\operatorname{Aut}_K(L)\cong N_G(H)/H,
\]
其中 \(N_G(H)\) 为 \(H\) 在 \(G\) 中的正规化子。对对称群中的一点稳定子，有
\[
N_{S_6}(S_5)=S_5.
\]
故
\[
\operatorname{Aut}_K(L)\cong S_5/S_5=1.
\]

若 \(\iota\) 保持 \(s\) 不变，则它将给出一个非平凡 \(K\)-自同构，这与上式矛盾。故 \(\iota\) 不属于 \(s\)-投影的甲板群，而只在横向二次商 \(X\to C\) 中出现。证毕。
\end{proof}

\begin{theorem}[三重覆盖 \texorpdfstring{$C\to \PP^1_x$}{C to P1x} 的十点简单分歧与判别双覆]\label{thm:conclusion-cubic-cover-ten-simple-branchpoints-discriminant-double-cover}
设
\[
\phi:C\longrightarrow \PP^1_x
\]
为正文中的三重覆盖，其判别式为
\[
\operatorname{Disc}_y=x\,P_8(x),
\]
其中
\[
P_8(x)=4x^8+85x^7+416x^6-818x^5+1196x^4-871x^3+284x^2-44x+4.
\]
则：
\begin{enumerate}
  \item \(\phi\) 的分歧值恰为
  \[
  x=0,\qquad P_8(x)=0,\qquad x=\infty;
  \]
  \item 这 \(10\) 个分歧值全是简单分歧值；
  \item 相应的判别双覆
  \[
  H:\ v^2=xP_8(x)\longrightarrow \PP^1_x
  \]
  正好在这 \(10\) 个点处分歧，因此 \(H\) 是 genus-\(4\) 的超椭圆曲线。
\end{enumerate}
\end{theorem}
\begin{proof}
由正文的三重覆盖判别公式，有限分歧值由
\[
xP_8(x)=0
\]
给出，而无穷远处另有一次分歧。多项式 \(xP_8(x)\) 的次数为 \(9\)，而同一正文结论同时断言双覆
\[
v^2=xP_8(x)
\]
是 genus-\(4\) 的超椭圆曲线；这迫使 \(xP_8(x)\) 为 squarefree，因此有限分歧值共 \(9\) 个且互异。加上无穷远处，恰得 \(10\) 个分歧值。

对三重覆盖 \(\phi\)，Riemann--Hurwitz 给出
\[
2g(C)-2=3(-2)+R.
\]
代入 \(g(C)=3\)，得
\[
4=-6+R,
\qquad
R=10.
\]
既然已有 \(10\) 个互异分歧值，而每个分歧值至少贡献 \(1\)，则每个分歧值只能恰好贡献 \(1\)。因此全部分歧都是简单分歧。最后，判别双覆恰以这些点为分歧点集，结论遂成。证毕。
\end{proof}

\begin{theorem}[cyclotomic resultant 根的 unit/\texorpdfstring{$p$}{p}-unit 刚性与偶模平方压缩]\label{thm:conclusion-cyclotomic-resultant-unit-punit-even-square-compression}
设 \(m\ge 4\)，并沿正文 cyclotomic elimination resultant \(R_m(w)\) 的记号。则成立：
\begin{enumerate}
  \item 若
  \[
  m\neq 3p^k
  \]
  对任何素数 \(p\) 与整数 \(k\ge 1\) 都不成立，则 \(R_m(w)\) 为首一整系数多项式且常数项为 \(1\)。因此 \(R_m\) 的每一个根都是代数单位。
  \item 若
  \[
  m=3p^k,
  \]
  则 \(R_m\) 的每一个根都是 \(p\)-unit；亦即在任意 \(\ell\neq p\) 的 \(\ell\)-进赋值下都满足
  \[
  v_\ell(\alpha)=0.
  \]
  \item 若 \(m\) 为偶数，则存在唯一多项式
  \[
  S_m(T)\in \ZZ[T],
  \qquad
  \deg S_m=3\varphi(m),
  \]
  使
  \[
  R_m(w)=S_m(w^2).
  \]
  并且 \(S_m\) 的常数项仍为 \(1\)，其首项系数与 \(R_m\) 相同。故：
  \begin{enumerate}
    \item 当 \(m\neq 3\cdot 2^k\) 时，\(S_m\) 的全部根都是代数单位；
    \item 当 \(m=3\cdot 2^k\) 时，\(S_m\) 的全部根都是 \(2\)-unit。
  \end{enumerate}
  特别地，偶模数下 \(R_m\) 的非零根严格成 \(\pm\) 对出现，而它们的平方落在一个次数 \(3\varphi(m)\) 的 unit/\(2\)-unit 多项式上。
\end{enumerate}
\end{theorem}
\begin{proof}
由正文对 \(R_m\) 的 leading/constant term 计算，
\[
R_m(0)=1,
\qquad
\operatorname{LC}(R_m)=
\begin{cases}
p^{\varepsilon(m)},& m=3p^k,\\
1,& m\neq 3p^k,
\end{cases}
\]
其中 \(\varepsilon(m)=1\)（\(m\) 奇）或 \(2\)（\(m\) 偶）。

若 \(m\neq 3p^k\)，则 \(\operatorname{LC}(R_m)=1\)，故 \(R_m\) 首一且常数项为 \(1\)。于是任意根 \(\alpha\) 是代数整数；其逆元 \(\alpha^{-1}\) 是 reciprocal polynomial
\[
R_m^\vee(T):=T^{\deg R_m}R_m(T^{-1})
\]
的根，而 \(R_m^\vee\) 也首一整系数。故 \(\alpha^{-1}\) 亦为代数整数，遂 \(\alpha\) 为代数单位。

若 \(m=3p^k\)，则正文的 \(p\)-unit 结果给出：任意根 \(\alpha\) 在所有 \(\ell\neq p\) 处都是 \(\ell\)-进整数。另一方面，\(\alpha^{-1}\) 仍是 \(R_m^\vee\) 的根，而 \(R_m^\vee\) 始终首一整系数，因为其首项系数就是常数项 \(R_m(0)=1\)。故 \(\alpha^{-1}\) 在所有素位处都是整数，特别在 \(\ell\neq p\) 处满足
\[
v_\ell(\alpha^{-1})\ge 0.
\]
结合 \(v_\ell(\alpha)\ge 0\)，便得
\[
v_\ell(\alpha)=0
\qquad (\ell\neq p),
\]
故 \(\alpha\) 为 \(p\)-unit。

最后，若 \(m\) 为偶数，则由正文的偶模对称性，
\[
R_m(w)=R_m(-w)\in \ZZ[w^2],
\qquad
\deg_{w^2}R_m=3\varphi(m).
\]
故存在唯一 \(S_m(T)\in \ZZ[T]\) 使
\[
R_m(w)=S_m(w^2),
\]
并且 \(\deg S_m=3\varphi(m)\)。其常数项显然为 \(R_m(0)=1\)，首项系数也与 \(R_m\) 相同。将前两段分别应用于 \(S_m\)，即得 unit/\(2\)-unit 结论。根的 \(\pm\) 成对性亦由 \(R_m(w)=S_m(w^2)\) 立即推出。证毕。
\end{proof}

\noindent
综上，完成化 toy-RH 的中心对称式、属六判别曲线的覆盖几何与 cyclotomic resultant 的整系数消去链在此处形成了一条此前未被单独写出的终端闭环：一方面，\(\Xi_M\) 的临界线几何严格原子化，因而最近邻与二点统计都塌缩为有限字母表与显式 Bragg 梳；另一方面，属六曲线 \(X\) 由 \(w\) 的除子直接触发 trigonal 结构，而 \(s\)-投影却没有任何非平凡甲板变换，并与三重覆盖 \(C\to \PP^1_x\) 的十点简单分歧共同锁定了判别双覆；最后，cyclotomic elimination resultant 的根又统一落在 unit/\(p\)-unit 刚性之中，偶模情形则进一步压缩为平方变量上的整系数 unit 多项式。
