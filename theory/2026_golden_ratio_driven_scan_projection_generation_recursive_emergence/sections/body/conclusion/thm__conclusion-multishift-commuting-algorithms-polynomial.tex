\begin{theorem}[多移位可交换 prime-register 算法的多项式完全刻画]\label{thm:conclusion-multishift-commuting-algorithms-polynomial}
\leanverified{paper\_conclusion\_multishift\_commuting\_algorithms\_polynomial}
令 \(d\ge 2\)，并令
\[
\mathcal P_d:=\NN^{(\NN^d)}=\bigoplus_{\mathbf n\in\NN^d}\NN\,\mathbf e_{\mathbf n}
\]
为以 \(\{\mathbf e_{\mathbf n}\}_{\mathbf n\in\NN^d}\) 为基的自由交换幺半群（有限支撑）。
对每个 \(1\le i\le d\) 定义坐标移位端同态
\[
\mathrm{Sh}_i:\mathcal P_d\to\mathcal P_d,
\qquad
\mathrm{Sh}_i(\mathbf e_{\mathbf n})=\mathbf e_{\mathbf n+\mathbf e_i},
\]
其中 \(\mathbf e_i\in\NN^d\) 为标准基向量。
定义
\[
\mathrm{End}_{\mathrm{Sh}}(\mathcal P_d)
:=\left\{
\Phi:\mathcal P_d\to\mathcal P_d:\ 
\Phi\ \text{为幺半群同态且}\ 
\Phi\circ\mathrm{Sh}_i=\mathrm{Sh}_i\circ\Phi\ \forall i
\right\}.
\]
在 \(\mathrm{End}_{\mathrm{Sh}}(\mathcal P_d)\) 上定义逐点加法 \((\Phi+\Psi)(x):=\Phi(x)+\Psi(x)\) 与乘法 \(\Phi\cdot\Psi:=\Phi\circ\Psi\)，得到交换半环。
则存在自然半环同构
\[
\Theta_d:\NN[z_1,\dots,z_d]\xrightarrow{\ \cong\ }\mathrm{End}_{\mathrm{Sh}}(\mathcal P_d),
\]
其对
\(
f(\mathbf z)=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\mathbf z^{\boldsymbol\alpha}
\)
（有限支撑，\(\mathbf z^{\boldsymbol\alpha}:=\prod_{i=1}^d z_i^{\alpha_i}\)）
的像 \(\Theta_d(f)=:\Phi_f\) 由
\[
\Phi_f(\mathbf e_{\mathbf n})
=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\,\mathbf e_{\mathbf n+\boldsymbol\alpha}
\qquad(\mathbf n\in\NN^d)
\]
唯一确定，并满足 \(\Phi_{f+g}=\Phi_f+\Phi_g\)、\(\Phi_{fg}=\Phi_f\circ\Phi_g\)。
\end{theorem}
\begin{proof}
由于 \(\mathcal P_d\) 为自由交换幺半群，任意 \(\Phi\in\mathrm{End}_{\mathrm{Sh}}(\mathcal P_d)\) 由 \(\Phi(\mathbf e_{\mathbf n})\) 在生成元上的取值唯一决定。
交换关系给出对任意 \(\mathbf n\in\NN^d\) 的递推：
\[
\Phi(\mathbf e_{\mathbf n})
=\Phi(\mathrm{Sh}_1^{n_1}\cdots \mathrm{Sh}_d^{n_d}\mathbf e_{\mathbf 0})
=\mathrm{Sh}_1^{n_1}\cdots \mathrm{Sh}_d^{n_d}\Phi(\mathbf e_{\mathbf 0}),
\]
故 \(\Phi\) 由 \(v:=\Phi(\mathbf e_{\mathbf 0})\) 唯一决定。
写
\(
v=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\mathbf e_{\boldsymbol\alpha}
\)
（有限支撑），则对任意 \(\mathbf n\) 有
\(
\Phi(\mathbf e_{\mathbf n})=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\mathbf e_{\mathbf n+\boldsymbol\alpha}
\)。
令 \(f(\mathbf z):=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\mathbf z^{\boldsymbol\alpha}\in\NN[z_1,\dots,z_d]\)，并以此定义 \(\Phi_f\)。
由构造可见 \(\Phi_f\) 与所有 \(\mathrm{Sh}_i\) 交换，且 \(\Phi\mapsto f\) 与 \(f\mapsto \Phi_f\) 互为逆映射，从而得到集合层的双射。
\par
逐点加法对应系数相加显然。
对复合，直接计算（以 \(\mathbf e_{\mathbf n}\) 为基）：
\[
(\Phi_f\circ\Phi_g)(\mathbf e_{\mathbf n})
=\Phi_f\!\left(\sum_{\boldsymbol\beta}b_{\boldsymbol\beta}\mathbf e_{\mathbf n+\boldsymbol\beta}\right)
=\sum_{\boldsymbol\beta}b_{\boldsymbol\beta}\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\mathbf e_{\mathbf n+\boldsymbol\alpha+\boldsymbol\beta}
=\sum_{\boldsymbol\gamma}\left(\sum_{\boldsymbol\alpha+\boldsymbol\beta=\boldsymbol\gamma}a_{\boldsymbol\alpha}b_{\boldsymbol\beta}\right)\mathbf e_{\mathbf n+\boldsymbol\gamma},
\]
而括号内系数即为 \(fg\) 的系数，故 \(\Phi_f\circ\Phi_g=\Phi_{fg}\)。
因此 \(\Theta_d\) 为半环同构。证毕。
\end{proof}
