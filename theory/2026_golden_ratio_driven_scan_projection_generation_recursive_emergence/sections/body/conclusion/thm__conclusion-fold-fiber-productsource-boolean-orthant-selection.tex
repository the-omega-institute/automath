\begin{theorem}[乘积源在折叠纤维上的精确 hard-core 解离]\label{thm:conclusion-fold-fiber-product-source-hardcore-disintegration}
\leanverified{paper\_conclusion\_fold\_fiber\_product\_source\_hardcore\_disintegration}
固定 \(m\ge 1\)。对每个 \(x\in X_m\)，沿用定理 \ref{thm:pom-fiber-multivariate-holographic-conservation} 的记号
\[
\Theta_x:\Fold_m^{-1}(x)\xrightarrow{\sim}\mathsf{Ind}(\Gamma_x),
\qquad
S(\omega):=\Theta_x(\omega),
\qquad
\mathcal H_x(\boldsymbol z)=\sum_{\omega\in\Fold_m^{-1}(x)}\prod_{i=1}^{m}z_i^{\omega_i}.
\]
对任意参数向量 \(\boldsymbol z=(z_1,\dots,z_m)\in(0,\infty)^m\)，定义乘积源
\[
\mu_{\boldsymbol z}(\omega):=\frac{\prod_{i=1}^{m}z_i^{\omega_i}}{\prod_{i=1}^{m}(1+z_i)}
\qquad (\omega\in\Omega_m).
\]
则其经折叠的推前分布满足
\[
(\Fold_m)_\#\mu_{\boldsymbol z}(x)
=
\frac{\mathcal H_x(\boldsymbol z)}{\prod_{i=1}^{m}(1+z_i)}.
\]
更强地，在条件 \(\Fold_m(\omega)=x\) 下，纤维内独立集随机元 \(S(\omega)\) 的条件分布恰为 \(\Gamma_x\) 上活度场 \(w_{x,v}(\boldsymbol z)\) 的 hard-core Gibbs 分布：
\[
\mu_{\boldsymbol z}\!\bigl(S(\omega)=S\mid \Fold_m(\omega)=x\bigr)
=
\frac{\prod_{v\in S}w_{x,v}(\boldsymbol z)}
{\sum_{T\in \mathsf{Ind}(\Gamma_x)}\prod_{v\in T}w_{x,v}(\boldsymbol z)}.
\]
因此，任意乘积源在折叠之后都严格解离为
\[
\textup{稳定类型分布}
\quad\times\quad
\textup{纤维 hard-core 后验}.
\]
\end{theorem}
\begin{proof}
第一式直接由定义与定理 \ref{thm:pom-fiber-multivariate-holographic-conservation} 给出：
\[
(\Fold_m)_\#\mu_{\boldsymbol z}(x)
=
\sum_{\omega\in\Fold_m^{-1}(x)}\mu_{\boldsymbol z}(\omega)
=
\frac{1}{\prod_{i=1}^{m}(1+z_i)}
\sum_{\omega\in\Fold_m^{-1}(x)}\prod_{i=1}^{m}z_i^{\omega_i}
=
\frac{\mathcal H_x(\boldsymbol z)}{\prod_{i=1}^{m}(1+z_i)}.
\]
第二式则是推论 \ref{cor:pom-fiber-inhom-bernoulli-posterior-hardcore} 在 \(a_i=z_i\) 下的直接改写。其条件权重只依赖于
\[
\prod_{v\in S}w_{x,v}(\boldsymbol z),
\]
而与 \(x\) 固定后吸收进归一化的前因子无关。故所示条件分布成立。证毕。
\end{proof}

\begin{theorem}[纤维多变量生成多项式的系数级布尔立方体分割]\label{thm:conclusion-fold-fiber-hologram-boolean-cube-partition}
\leanverified{paper\_conclusion\_fold\_fiber\_hologram\_boolean\_cube\_partition}
对每个 \(x\in X_m\)，定义其单项式支撑
\[
\mathcal M_x:=
\{\alpha\in\{0,1\}^m:[z^\alpha]\mathcal H_x(\boldsymbol z)\neq 0\}.
\]
则族 \(\{\mathcal M_x\}_{x\in X_m}\) 构成 \(\{0,1\}^m\) 的严格分割：
\begin{enumerate}
  \item \(\bigsqcup_{x\in X_m}\mathcal M_x=\{0,1\}^m\)；
  \item 对任意 \(\alpha\in\{0,1\}^m\)，恰存在唯一 \(x(\alpha)\in X_m\) 使
  \[
  [z^\alpha]\mathcal H_{x(\alpha)}(\boldsymbol z)=1;
  \]
  \item 因而每个 \(\mathcal H_x\) 的系数都属于 \(\{0,1\}\)，并且
  \[
  \mathcal H_x(\boldsymbol z)=\sum_{\alpha\in \mathcal M_x}z^\alpha.
  \]
\end{enumerate}
因此，纤维全息守恒
\[
\sum_{x\in X_m}\mathcal H_x(\boldsymbol z)=\prod_{i=1}^{m}(1+z_i)
\]
不仅是分布级守恒，而且给出了一个系数级的布尔立方体剖分。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-multivariate-holographic-conservation}，
\[
\sum_{x\in X_m}\mathcal H_x(\boldsymbol z)=\prod_{i=1}^{m}(1+z_i)=\sum_{\alpha\in\{0,1\}^m}z^\alpha.
\]
另一方面，
\[
\mathcal H_x(\boldsymbol z)=\sum_{\omega\in\Fold_m^{-1}(x)}z^\omega,
\]
其中不同 \(\omega\in\{0,1\}^m\) 给出不同的布尔单项式 \(z^\omega\)。故每个 \(\mathcal H_x\) 的系数都是非负整数，而且只能取 \(0\) 或 \(1\)。

若存在某个 \(\alpha\in\{0,1\}^m\) 同时落在两个不同支撑 \(\mathcal M_x,\mathcal M_y\) 中，则左端 \(z^\alpha\) 的总系数至少为 \(2\)，与右端系数恒等于 \(1\) 矛盾。反之，右端每个布尔单项式都出现一次，故必恰由某个唯一的 \(\mathcal H_x\) 承载。于是三条结论同时成立。证毕。
\end{proof}

\begin{corollary}[零温外场的正交象限选择律]\label{cor:conclusion-fold-zero-temperature-orthant-selection}
对任意外场 \(\boldsymbol h=(h_1,\dots,h_m)\in\RR^m\)，定义布尔立方体上的线性能量
\[
E_{\boldsymbol h}(\alpha):=\langle \boldsymbol h,\alpha\rangle
\qquad (\alpha\in\{0,1\}^m).
\]
若 \(h_i\neq 0\) 对所有 \(i\) 成立，则 \(E_{\boldsymbol h}\) 在 \(\{0,1\}^m\) 上有唯一最大点
\[
\alpha(\boldsymbol h)_i:=\mathbf 1_{\{h_i>0\}}.
\]
由此存在唯一稳定类型 \(x(\boldsymbol h)\in X_m\) 使
\[
\alpha(\boldsymbol h)\in \mathcal M_{x(\boldsymbol h)}.
\]
并且对任意 \(x\in X_m\)，有极限公式
\[
\lim_{t\to\infty}\frac{1}{t}\log \mathcal H_x(e^{t\boldsymbol h})
=
\max_{\alpha\in \mathcal M_x}\langle \boldsymbol h,\alpha\rangle,
\]
其中 \(e^{t\boldsymbol h}:=(e^{th_1},\dots,e^{th_m})\)。进一步，
\[
\max_{x\in X_m}\lim_{t\to\infty}\frac{1}{t}\log \mathcal H_x(e^{t\boldsymbol h})
=
\sum_{i=1}^{m}\max(h_i,0).
\]
因此，稳定类型的零温竞争并不产生新的热带自由度；它只是把全部外场正交象限重新标号为稳定类型。
\end{corollary}
\begin{proof}
唯一最大点 \(\alpha(\boldsymbol h)\) 是布尔立方体上线性泛函的显然解。由定理 \ref{thm:conclusion-fold-fiber-hologram-boolean-cube-partition}，该点恰落在唯一一个 \(\mathcal M_x\) 中，从而唯一决定 \(x(\boldsymbol h)\)。

对固定 \(x\)，由
\[
\mathcal H_x(e^{t\boldsymbol h})
=
\sum_{\alpha\in \mathcal M_x}e^{t\langle \boldsymbol h,\alpha\rangle}
\]
与有限和的 Laplace 原理，立即得到
\[
\lim_{t\to\infty}\frac{1}{t}\log \mathcal H_x(e^{t\boldsymbol h})
=
\max_{\alpha\in \mathcal M_x}\langle \boldsymbol h,\alpha\rangle.
\]
再对 \(x\in X_m\) 取最大，并用 \(\{\mathcal M_x\}_{x\in X_m}\) 对整个 \(\{0,1\}^m\) 的分割，便得
\[
\max_{x\in X_m}\max_{\alpha\in \mathcal M_x}\langle \boldsymbol h,\alpha\rangle
=
\max_{\alpha\in\{0,1\}^m}\langle \boldsymbol h,\alpha\rangle
=
\sum_{i=1}^{m}\max(h_i,0).
\]
证毕。
\end{proof}

\begin{theorem}[纤维 face-poset 的平方自由整除影子化]\label{thm:conclusion-fold-fiber-faceposet-squarefree-divisibility-shadow}
固定 \(m\ge 1\) 与 \(x\in X_m\)。任选一组两两不同的素数
\[
\{q_v:v\in V(\Gamma_x)\},
\]
并定义平方自由乘法影子
\[
G_x:\mathsf{Ind}(\Gamma_x)\longrightarrow \NN_{\mathrm{sf}},
\qquad
G_x(I):=\prod_{v\in I}q_v,
\]
以及其像
\[
\mathcal G_x:=G_x\bigl(\mathsf{Ind}(\Gamma_x)\bigr)\subset \NN_{\mathrm{sf}}.
\]
则 \(G_x\) 把 \(\mathsf{Ind}(\Gamma_x)\) 嵌入为整除偏序下的严格 \(\gcd\)-半格，并满足
\[
I\subseteq J
\iff
G_x(I)\mid G_x(J),
\]
\[
G_x(I\cap J)=\gcd\bigl(G_x(I),G_x(J)\bigr).
\]
特别地，经由规范双射 \(\Theta_x\)，有严格同构
\[
\Fold_m^{-1}(x)\ \cong\ \mathsf{Ind}(\Gamma_x)\ \cong\ \mathcal G_x.
\]
因此，固定纤维的微态空间、hard-core 独立集复形与平方自由整除半格在当前语义中只是同一对象的三种坐标。
\end{theorem}
\begin{proof}
由于 \(\{q_v\}_{v\in V(\Gamma_x)}\) 两两不同，算术基本定理立即给出
\[
G_x(I)=G_x(J)\iff I=J,
\]
故 \(G_x\) 为单射。
又对任意 \(I,J\in\mathsf{Ind}(\Gamma_x)\)，有
\[
I\subseteq J
\iff
\forall v\in I,\ q_v\mid G_x(J)
\iff
G_x(I)\mid G_x(J),
\]
并且
\[
\gcd\bigl(G_x(I),G_x(J)\bigr)
=
\prod_{v\in I\cap J}q_v
=
G_x(I\cap J).
\]
这就说明 \(\mathcal G_x\) 在整除偏序与 \(\gcd\) 运算下严格复制了 \(\mathsf{Ind}(\Gamma_x)\) 的包含与交结构。
最后，\(\Theta_x:\Fold_m^{-1}(x)\xrightarrow{\sim}\mathsf{Ind}(\Gamma_x)\) 是已有的规范双射，故得到所述三重同构。证毕。
\end{proof}

\begin{theorem}[平方自由算术影子上的 hard-core 面数重建与全息守恒重写]\label{thm:conclusion-fold-fiber-squarefree-shadow-hardcore-holography}
沿用定理 \ref{thm:conclusion-fold-fiber-faceposet-squarefree-divisibility-shadow} 的记号。
对每个 \(r\ge 0\)，定义纤维 face 数
\[
N_x(r):=\#\{I\in\mathsf{Ind}(\Gamma_x):|I|=r\}.
\]
则有精确重建公式
\[
N_x(r)=\#\{n\in\mathcal G_x:\Omega(n)=r\},
\]
以及 hard-core 生成函数的算术化表示
\[
Z_x(t):=\sum_{I\in\mathsf{Ind}(\Gamma_x)}t^{|I|}
=
\sum_{n\in\mathcal G_x}t^{\Omega(n)},
\]
其中 \(\Omega(n)\) 为素因子总数。

进一步，若定义
\[
\operatorname{supp}_x(n):=\{v\in V(\Gamma_x):q_v\mid n\}
\qquad (n\in\mathcal G_x),
\]
则多变量全息多项式可重写为
\[
\mathcal H_x(\boldsymbol z)
=
\Bigl(\prod_{i=1}^{m}z_i^{x_i}\Bigr)
\sum_{n\in\mathcal G_x}\prod_{v\in\operatorname{supp}_x(n)}w_{x,v}(\boldsymbol z),
\]
从而全息守恒恒等式变为
\[
\sum_{x\in X_m}
\Bigl(\prod_{i=1}^{m}z_i^{x_i}\Bigr)
\sum_{n\in\mathcal G_x}\prod_{v\in\operatorname{supp}_x(n)}w_{x,v}(\boldsymbol z)
=
\prod_{i=1}^{m}(1+z_i).
\]
因此，纤维 hard-core 组合学与多变量全息守恒都可在平方自由整除影子上无损重写，而不再需要显式回到候选位点图本身。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fold-fiber-faceposet-squarefree-divisibility-shadow}，每个 \(n\in\mathcal G_x\) 唯一对应某个独立集 \(I\in\mathsf{Ind}(\Gamma_x)\)，并且由于 \(n\) 平方自由，
\[
\Omega(n)=|I|.
\]
故
\[
N_x(r)=\#\{n\in\mathcal G_x:\Omega(n)=r\},
\qquad
Z_x(t)=\sum_{n\in\mathcal G_x}t^{\Omega(n)}.
\]

另一方面，由正文中的多变量全息因式分解，
\[
\mathcal H_x(\boldsymbol z)
=
\Bigl(\prod_{i=1}^{m}z_i^{x_i}\Bigr)
\sum_{I\in\mathsf{Ind}(\Gamma_x)}\prod_{v\in I}w_{x,v}(\boldsymbol z).
\]
把每个独立集 \(I\) 用其平方自由影子 \(n=G_x(I)\) 重命名，并注意到 \(I=\operatorname{supp}_x(n)\)，即得
\[
\mathcal H_x(\boldsymbol z)
=
\Bigl(\prod_{i=1}^{m}z_i^{x_i}\Bigr)
\sum_{n\in\mathcal G_x}\prod_{v\in\operatorname{supp}_x(n)}w_{x,v}(\boldsymbol z).
\]
最后对 \(x\in X_m\) 求和，并使用
\[
\sum_{x\in X_m}\mathcal H_x(\boldsymbol z)=\prod_{i=1}^{m}(1+z_i),
\]
便得到所示算术化全息守恒恒等式。证毕。
\end{proof}
