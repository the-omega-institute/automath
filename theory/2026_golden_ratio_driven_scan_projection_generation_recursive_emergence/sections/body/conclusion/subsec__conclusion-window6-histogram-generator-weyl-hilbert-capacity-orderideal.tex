\begin{theorem}[直方图母函数的五重解码]\label{thm:conclusion-window6-histogram-generator-fivefold-decoding}
定义 window-\(6\) 的纤维直方图母函数
\[
H_6(z):=8z^2+4z^3+9z^4.
\]
则有严格恒等式
\[
H_6(1)=21,
\qquad
H_6'(1)=64,
\qquad
H_6'(1)-H_6(1)=43,
\]
\[
H_6''(1)+H_6'(1)=212,
\qquad
\frac12 H_6''(1)=74.
\]
并且这五个数分别具有如下几何--代数解释：
\[
21=|X_6|=\dim_{\CC}Z(A_6),
\qquad
64=|\Omega_6|,
\]
\[
43=\dim_{\CC}\mathscr R_6,
\qquad
212=\dim_{\CC}A_6,
\qquad
74=\#\operatorname{Ref}\!\bigl(G_6^{\mathrm{bin}}\bigr).
\]
因此，\(H_6\) 在 \(z=1\) 附近的前二阶微分数据已经同时锁定可见稳定层规模、微态总数、隐藏反射包维数、关系群胚代数维数以及隐藏 Weyl 包的反射数。
\end{theorem}
\begin{proof}
由推论 \ref{cor:fold-bin-m6-fiber-hist}，window-\(6\) 的纤维直方图为
\[
2:8,\qquad 3:4,\qquad 4:9.
\]
于是
\[
H_6(1)=8+4+9=21.
\]
再由定理 \ref{thm:conclusion-window6-qmoment-triple-geometry}，
\[
S_q(6)=8\cdot 2^q+4\cdot 3^q+9\cdot 4^q,
\]
故
\[
H_6'(1)=\sum_{w\in X_6}d(w)=S_1(6)=64,
\]
\[
H_6''(1)+H_6'(1)=\sum_{w\in X_6}d(w)^2=S_2(6)=212.
\]
同一条定理还给出
\[
\dim_{\CC}Z(A_6)=|X_6|=21.
\]

另一方面，定理 \ref{thm:conclusion-window6-hidden-a-type-weyl-package} 说明
\[
G_6^{\mathrm{bin}}
\cong
S_2^{\times 8}\times S_3^{\times 4}\times S_4^{\times 9}
\cong
W\!\left(A_1^{\oplus 8}\oplus A_2^{\oplus 4}\oplus A_3^{\oplus 9}\right),
\]
且
\[
\dim_{\CC}\mathscr R_6
=
8\cdot 1+4\cdot 2+9\cdot 3
=43.
\]
又
\[
H_6'(1)-H_6(1)=\sum_{w\in X_6}(d(w)-1)=8+8+27=43,
\]
故得到第三个解释。

最后，\(A_1,A_2,A_3\) 型 Weyl 群的反射数分别为
\[
1,\qquad 3,\qquad 6,
\]
于是
\[
\#\operatorname{Ref}\!\bigl(G_6^{\mathrm{bin}}\bigr)
=
8\cdot 1+4\cdot 3+9\cdot 6
=74.
\]
而
\[
\frac12 H_6''(1)
=
8\binom22+4\binom32+9\binom42
=8+12+54
=74.
\]
证毕。
\end{proof}

\begin{theorem}[不变量次数谱与容量尾函数的同一性]\label{thm:conclusion-window6-degree-spectrum-capacity-tail-identity}

g_r:=

则有严格恒等式

g_r=N_6(r)=\partial_T^+\mathcal C_6^{\mathrm{cont}}(r-1)

特别地，

因此，window-\(6\) 的自由不变量次数谱、容量曲线在前三个整数点的右导数，以及纤维 tail counting 三者完全重合。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-hidden-reflection-invariant-polynomial-ring}，

\CC[\mathscr R_6]^{G_6^{\mathrm{bin}}}

故
\[
g_2=21,\qquad g_3=13,\qquad g_4=9.
\]

另一方面，由纤维直方图 \(2{:}8,3{:}4,4{:}9\) 可得

N_6(2)=8+4+9=21,
\qquad
N_6(3)=4+9=13,

故 \(g_r=N_6(r)\)。

再由定理 \ref{thm:xi-foldbin-continuous-capacity-right-derivative-abelian-center}，

证毕。
\end{proof}

\begin{corollary}[连续容量曲线的 Hilbert 化]\label{cor:conclusion-window6-capacity-hilbertization}
隐藏反射包不变量代数的 Hilbert 级数完全由连续容量曲线在前三个整数点的右导数决定：
\[
\operatorname{Hilb}\!\bigl(\CC[\mathscr R_6]^{G_6^{\mathrm{bin}}};t\bigr)
=
\prod_{k=1}^{3}
\bigl(1-t^{k+1}\bigr)^{-\partial_T^+\mathcal C_6^{\mathrm{cont}}(k)}.
\]
特别地，
\[
\operatorname{Hilb}\!\bigl(\CC[\mathscr R_6]^{G_6^{\mathrm{bin}}};t\bigr)

\frac{1}{(1-t^2)^{21}(1-t^3)^{13}(1-t^4)^9}.

因此，window-\(6\) 的隐藏不变量代数并不需要额外的表示论输入；连续容量曲线的前三处右导数已经给出其完整 Hilbert 级数。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-window6-hidden-reflection-invariant-polynomial-ring}，

\operatorname{Hilb}\!\bigl(\CC[\mathscr R_6]^{G_6^{\mathrm{bin}}};t\bigr)

\frac{1}{(1-t^2)^{g_2}(1-t^3)^{g_3}(1-t^4)^{g_4}}.

再由定理 \ref{thm:conclusion-window6-degree-spectrum-capacity-tail-identity}，
\[
g_{k+1}=\partial_T^+\mathcal C_6^{\mathrm{cont}}(k)
\qquad (k=1,2,3).
\]
代入即得。证毕。
\end{proof}

\begin{theorem}[容量导数对对称包的算术重构]\label{thm:conclusion-window6-capacity-derivatives-reconstruct-weyl-arithmetic}
定义
\[
c_2:=\partial_T^+\mathcal C_6^{\mathrm{cont}}(1),

c_3:=\partial_T^+\mathcal C_6^{\mathrm{cont}}(2),
\qquad
c_4:=\partial_T^+\mathcal C_6^{\mathrm{cont}}(3).
\]
则
\[
(c_2,c_3,c_4)=(21,13,9),
\]
并且
\[
\dim_{\CC}\mathscr R_6=c_2+c_3+c_4,
\]
\[
\#\operatorname{Ref}\!\bigl(G_6^{\mathrm{bin}}\bigr)=c_2+2c_3+3c_4,
\]
\[
|G_6^{\mathrm{bin}}|
=
2^{\,c_2+2c_4}3^{\,c_3}
=
2^{39}3^{13}.
\]
因此，容量曲线的三枚离散右导数已经足以重构 hidden Weyl 包的维数、反射总数以及规范群的精确阶数。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-degree-spectrum-capacity-tail-identity}，
\[
(c_2,c_3,c_4)=(21,13,9).
\]
又由定理 \ref{thm:conclusion-window6-hidden-a-type-weyl-package}，
\[
\dim_{\CC}\mathscr R_6
=
8\cdot 1+4\cdot 2+9\cdot 3
=43
=21+13+9
=c_2+c_3+c_4.
\]

对反射数，
\[
\#\operatorname{Ref}\!\bigl(G_6^{\mathrm{bin}}\bigr)
=
8\cdot 1+4\cdot 3+9\cdot 6
=74
=21+2\cdot 13+3\cdot 9
=c_2+2c_3+3c_4.
\]

最后，由同一定理
\[
G_6^{\mathrm{bin}}
\cong
S_2^{\times 8}\times S_3^{\times 4}\times S_4^{\times 9},
\]
且
\[
|S_2|=2,\qquad
|S_3|=2\cdot 3,\qquad
|S_4|=2^3\cdot 3.
\]
故
\[
|G_6^{\mathrm{bin}}|
=
2^8(2\cdot 3)^4(2^3\cdot 3)^9
=
2^{39}3^{13}.
\]
另一方面，
\[
39=21+18=c_2+2c_4,
\qquad
13=c_3,
\]
从而得到所示容量导数表达式。证毕。
\end{proof}

\begin{corollary}[次数谱的离散微分恢复纤维直方图]\label{cor:conclusion-window6-degree-spectrum-discrete-differentiation-recovers-histogram}

A_6,\qquad

G_6^{\mathrm{bin}}.

\end{corollary}
\begin{proof}

而 \(A_6\) 的 Wedderburn 分解由命题 \ref{prop:fold-bin-groupoid-algebra-wedderburn} 唯一由三档块重数决定，\(\mathscr R_6\) 的 \(A_1^8A_2^4A_3^9\) 直和型由定理 \ref{thm:conclusion-window6-hidden-a-type-weyl-package} 唯一由同一组重数决定，\(G_6^{\mathrm{bin}}\cong S_2^{\times 8}\times S_3^{\times 4}\times S_4^{\times 9}\) 亦然。证毕。

因此，window-\(6\) 的 Boolean hidden-fiber 过滤与 hidden Weyl 包的不变量次数过滤在计数层面严格同构。

A_2=\{(0,0),(0,1)\},

另一方面，定理 \ref{thm:conclusion-window6-degree-spectrum-capacity-tail-identity} 已给出次数 \(2,3,4\) 的基本不变量数恰为 \((21,13,9)\)。故两类过滤严格重合。证毕。

\begin{conjecture}[直方图母函数原则]\label{conj:conclusion-window6-histogram-generator-principle}

\cong
\prod_{d\ge 2}W(A_{d-1})^{\,n_d^{(m)}}

=

\[
\operatorname{Hilb}\!\bigl(\CC[\mathscr R_m]^{G_m};t\bigr)
=
\prod_{r\ge 2}(1-t^r)^{-N_m(r)},
\]
\[
\dim \mathscr R_m=H_m'(1)-H_m(1),
\qquad
\dim A_m^{\mathrm{bin}}=H_m''(1)+H_m'(1),
\]
\[

window-\(6\) 给出该原则的完整审计锚点：母函数

\end{conjecture}

便不再是孤立巧合，而是一条真正的普适统摄律。
