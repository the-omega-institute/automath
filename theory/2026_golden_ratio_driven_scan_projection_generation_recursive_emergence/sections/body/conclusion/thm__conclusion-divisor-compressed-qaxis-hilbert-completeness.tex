\begin{theorem}[divisor-compressed \(q\)-轴压缩算子的 Hilbert 有界性]\label{thm:conclusion-divisor-compressed-qaxis-hilbert-boundedness}
\leanverified{paper\_conclusion\_divisor\_compressed\_qaxis\_hilbert\_boundedness}
定义压缩算子
\[
(\mathcal C u)_q:=\sum_{\ell\ge 1}\frac{u_{\ell q}}{\ell}\qquad (q\ge 1)
\]
以及带权范数
\[
\|v\|_{2,-2}^2:=\sum_{q\ge 1}\frac{|v_q|^2}{q^2}.
\]
则 \(\mathcal C\) 从 \(\ell^2(\NN)\) 有界映入带权空间 \((\ell^2(\NN),\|\cdot\|_{2,-2})\)，并满足
\[
\|\mathcal C u\|_{2,-2}^2
\le
\zeta(2)\sum_{n\ge 1}\sigma_{-2}(n)\,|u_n|^2
\le
\zeta(2)^2\|u\|_{\ell^2}^2.
\]
因此
\[
\|\mathcal C\|_{\ell^2\to \ell^2(q^{-2})}\le \zeta(2).
\]
特别地，对 primitive 核 \(M\) 的非 Perron 归一化谱迹
\[
u_n(M):=\sum_{j=2}^d\lambda_j^n,
\qquad
\mathfrak S_q(M)=(\mathcal C u(M))_q,
\]
有
\[
\sum_{q\ge 1}\frac{|\mathfrak S_q(M)|^2}{q^2}
\le
\zeta(2)^2\,\mathcal E_{\partial}(M).
\]
\end{theorem}
\begin{proof}
对任意 \(q\ge 1\)，由 Cauchy--Schwarz 不等式，
\[
|(\mathcal C u)_q|^2
=
\left|\sum_{\ell\ge 1}\frac{u_{\ell q}}{\ell}\right|^2
\le
\left(\sum_{\ell\ge 1}\frac1{\ell^2}\right)
\left(\sum_{\ell\ge 1}|u_{\ell q}|^2\right)
=
\zeta(2)\sum_{\ell\ge 1}|u_{\ell q}|^2.
\]
两边乘以 \(q^{-2}\) 并对 \(q\) 求和，得到
\[
\|\mathcal C u\|_{2,-2}^2
\le
\zeta(2)\sum_{q,\ell\ge 1}\frac{|u_{\ell q}|^2}{q^2}.
\]
令 \(n=\ell q\)，则
\[
\sum_{q,\ell\ge 1}\frac{|u_{\ell q}|^2}{q^2}
=
\sum_{n\ge 1}|u_n|^2\sum_{q\mid n}\frac1{q^2}
=
\sum_{n\ge 1}\sigma_{-2}(n)\,|u_n|^2.
\]
再用 \(\sigma_{-2}(n)\le \zeta(2)\) 便得
\[
\|\mathcal C u\|_{2,-2}^2
\le
\zeta(2)^2\|u\|_{\ell^2}^2.
\]
故算子范数满足 \(\|\mathcal C\|\le \zeta(2)\)。

对 primitive 核 \(M\)，定理 \ref{thm:conclusion-divisor-compressed-qaxis-ramanujan-diagonalization} 给出
\[
\mathfrak S_q(M)=\sum_{\ell\ge 1}\frac{u_{\ell q}(M)}{\ell}=(\mathcal C u(M))_q.
\]
再由定理 \ref{thm:conclusion-ramanujan-kernel-psd-derivative-parseval-control}
\[
\mathcal E_{\partial}(M)=\sum_{n\ge 1}|u_n(M)|^2
\]
即可得到末式。证毕。
\end{proof}

\begin{proposition}[平方根门槛在 \(q\)-轴上的完全显影]\label{prop:conclusion-divisor-compressed-qaxis-sqrt-threshold-visible}
\leanverified{paper\_conclusion\_divisor\_compressed\_qaxis\_sqrt\_threshold\_visible}
对每个 primitive 核 \(M\) 与每个 \(q\ge 1\)，成立显式上界
\[
|\mathfrak S_q(M)|
\le
-(d-1)\log\!\bigl(1-\Lambda_{\mathrm{rel}}(M)^q\bigr).
\]
若进一步满足 kernel-RH 条件
\[
\Lambda_{\mathrm{rel}}(M)\le \rho(M)^{-1/2},
\]
则
\[
|\mathfrak S_q(M)|
\le
-(d-1)\log\!\bigl(1-\rho(M)^{-q/2}\bigr)
=(d-1)\rho(M)^{-q/2}+O\bigl(\rho(M)^{-q}\bigr).
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-divisor-compressed-qaxis-ramanujan-diagonalization}，
\[
\mathfrak S_q(M)=\sum_{\ell\ge 1}\frac{u_{\ell q}(M)}{\ell}.
\]
而
\[
|u_n(M)|
\le
\sum_{j=2}^d|\lambda_j|^n
\le
(d-1)\Lambda_{\mathrm{rel}}(M)^n.
\]
故
\[
|\mathfrak S_q(M)|
\le
(d-1)\sum_{\ell\ge 1}\frac{\Lambda_{\mathrm{rel}}(M)^{\ell q}}{\ell}
=
-(d-1)\log\!\bigl(1-\Lambda_{\mathrm{rel}}(M)^q\bigr).
\]
若再有 \(\Lambda_{\mathrm{rel}}(M)\le \rho(M)^{-1/2}\)，则直接代入得到
\[
|\mathfrak S_q(M)|
\le
-(d-1)\log\!\bigl(1-\rho(M)^{-q/2}\bigr).
\]
最后用
\[
-\log(1-x)=x+O(x^2)\qquad (x\downarrow 0)
\]
在 \(x=\rho(M)^{-q/2}\) 处的展开，便得到
\[
|\mathfrak S_q(M)|
=(d-1)\rho(M)^{-q/2}+O\bigl(\rho(M)^{-q}\bigr).
\]
证毕。
\end{proof}

\begin{conjecture}[\(q\)-轴与深度包质量的双坐标完备性]\label{conj:conclusion-divisor-qaxis-depth-completeness}
设 \(M,N\) 为两个 primitive 核，并记 \(K_d^{(M)}\)、\(K_d^{(N)}\) 为分别附属于 \(M\)、\(N\) 的 exact-depth Ramanujan 包核。若对一切 \(q,d\ge 1\) 都有
\[
\mathfrak S_q(M)=\mathfrak S_q(N),
\qquad
K_d^{(M)}(0)=K_d^{(N)}(0),
\]
则 \(M\) 与 \(N\) 的非 Perron 谱多重集以及 exact-depth Ramanujan 包分解在自然意义下同构。等价地，cyclotomic 扭点坐标与 exact-depth 包质量共同构成压缩谱对象的完备坐标系。
\end{conjecture}
