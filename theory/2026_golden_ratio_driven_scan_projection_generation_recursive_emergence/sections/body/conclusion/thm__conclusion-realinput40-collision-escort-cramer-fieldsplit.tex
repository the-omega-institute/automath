\begin{theorem}[二碰撞 escort 前沿的单参数代数化]\label{thm:conclusion-realinput40-collision-escort-algebraic-parametrization}
\leanverified{paper\_conclusion\_realinput40\_collision\_escort\_algebraic\_parametrization}
定义
\[
u(\rho):=\frac{\rho(\rho^2-2\rho-1)}{\rho-1},
\qquad
\alpha(\rho):=\frac{(\rho-1)(\rho^2-2\rho-1)}{2\rho^3-5\rho^2+4\rho+1},
\qquad
\rho>1+\sqrt2.
\]
则：
\begin{enumerate}
  \item \(u\mapsto \rho(u)\) 在 \(u>0\) 上与 \(\rho\mapsto u(\rho)\) 互为反函数；
  \item \(\alpha(\theta)=P_2'(\theta)\) 可精确写成
  \[
  \alpha(\theta)=\alpha(\rho(e^\theta));
  \]
  \item \(\alpha(\rho)\) 在 \((1+\sqrt2,\infty)\) 上严格递增，并给出双射
  \[
  (1+\sqrt2,\infty)\xrightarrow{\ \alpha\ }\left(0,\frac12\right).
  \]
\end{enumerate}
因而二碰撞 escort 前沿可由单一代数参数 \(\rho\) 完全参数化。
\end{theorem}
\begin{proof}
三次压力方程
\[
\rho^3-2\rho^2-(u+1)\rho+u=0
\]
对 \(u\) 解出，得到
\[
u(\rho)=\frac{\rho(\rho^2-2\rho-1)}{\rho-1}.
\]
当 \(\rho>1\) 时，\(u(\rho)>0\) 当且仅当 \(\rho^2-2\rho-1>0\)，亦即
\[
\rho>1+\sqrt2.
\]
又
\[
u'(\rho)=\frac{2\rho^3-5\rho^2+4\rho+1}{(\rho-1)^2}.
\]
令 \(\rho=1+\sqrt2+t\)（\(t\ge 0\)），则分子化为
\[
2t^3+(1+6\sqrt2)t^2+(12+2\sqrt2)t+(4+4\sqrt2)>0,
\]
故 \(u(\rho)\) 在 \((1+\sqrt2,\infty)\) 上严格递增。再由
\[
\lim_{\rho\downarrow 1+\sqrt2}u(\rho)=0,
\qquad
\lim_{\rho\to\infty}u(\rho)=+\infty,
\]
可知它将 \((1+\sqrt2,\infty)\) 双射到 \((0,\infty)\)，从而与 \(u\mapsto \rho(u)\) 互为反函数。

对三次方程隐式微分得
\[
(3\rho^2-4\rho-u-1)\rho'(u)+(1-\rho)=0.
\]
因此
\[
P_2'(\theta)
=
u\frac{\rho'(u)}{\rho(u)}
=
\frac{u(\rho-1)}{\rho(3\rho^2-4\rho-u-1)}.
\]
代入 \(u=u(\rho)\) 并化简，即得
\[
\alpha(\rho)=\frac{(\rho-1)(\rho^2-2\rho-1)}{2\rho^3-5\rho^2+4\rho+1}.
\]

再对 \(\rho\) 求导，
\[
\alpha'(\rho)
=
\frac{\rho^4+4\rho^3-10\rho^2+4\rho-3}{(2\rho^3-5\rho^2+4\rho+1)^2}.
\]
令 \(\rho=1+\sqrt2+t\)（\(t\ge 0\)），则分子化为
\[
t^4+(8+4\sqrt2)t^3+(20+24\sqrt2)t^2+(48+24\sqrt2)t+(16+16\sqrt2),
\]
故严格为正，于是 \(\alpha'(\rho)>0\)。最后，
\[
\lim_{\rho\downarrow 1+\sqrt2}\alpha(\rho)=0,
\qquad
\lim_{\rho\to\infty}\alpha(\rho)=\frac12,
\]
从而得到所述双射。证毕。
\end{proof}

\begin{corollary}\label{cor:conclusion-realinput40-collision-density-range-balanced-anchor}
\leanverified{paper\_conclusion\_realinput40\_collision\_density\_range\_balanced\_anchor}
二碰撞输出密度的整个可达区间恰为
\[
0<\alpha<\frac12,
\]
并且平衡点
\[
\alpha_0=P_2'(0)=\frac{3-\sqrt5}{10}
\]
对应于
\[
u=1,
\qquad
\rho=\varphi^2.
\]
\end{corollary}
\begin{proof}
可达区间与双射性由定理 \ref{thm:conclusion-realinput40-collision-escort-algebraic-parametrization} 立即得到。再取 \(u=1\) 时，主稿已给出 \(\rho(1)=\varphi^2\) 与
\[
P_2'(0)=\frac{3-\sqrt5}{10},
\]
故结论成立。证毕。
\end{proof}

\begin{theorem}[中心化 Cram\'er 前沿的显式代数参数化与严格凸性]\label{thm:conclusion-realinput40-centered-cramer-explicit-algebraic-frontier}
\leanverified{paper\_conclusion\_realinput40\_centered\_cramer\_explicit\_algebraic\_frontier}
定义中心化压力
\[
\widetilde P_2(\theta):=P_2(\theta)-P_2(0)=P_2(\theta)-\log \varphi^2,
\]
以及其 Legendre--Fenchel 前沿
\[
J_2(\alpha):=\sup_{\theta\in\mathbb R}\{\alpha\theta-\widetilde P_2(\theta)\},
\qquad
0<\alpha<\frac12.
\]
则 \(J_2\) 由 \(\rho\)-参数显式给出：
\[
J_2(\alpha(\rho))
=
\alpha(\rho)\log\!\frac{\rho(\rho^2-2\rho-1)}{\rho-1}
-\log\rho+\log\varphi^2,
\qquad
\rho>1+\sqrt2.
\]
并且
\[
J_2'(\alpha(\rho))=\log u(\rho),
\qquad
J_2''(\alpha)>0.
\]
故 \(J_2\) 在 \((0,\frac12)\) 上严格凸。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-realinput40-collision-escort-algebraic-parametrization}，\(\theta\mapsto \alpha(\theta)=P_2'(\theta)\) 在 \(\mathbb R\) 上严格递增，故对每个
\[
0<\alpha<\frac12
\]
存在唯一 \(\theta\) 使 \(\alpha=P_2'(\theta)\)。因此 Legendre 极大点唯一，并满足
\[
J_2(\alpha)=\alpha\theta-\widetilde P_2(\theta).
\]
记 \(u=e^\theta\)，再用 \(u=u(\rho)\) 与 \(P_2(\theta)=\log\rho(u)\)，即得
\[
J_2(\alpha(\rho))
=
\alpha(\rho)\log u(\rho)-\log\rho+\log\varphi^2.
\]

由 Legendre 对偶的一般关系，
\[
J_2'(\alpha(\rho))=\theta=\log u(\rho).
\]
再对 \(\rho\) 求导，
\[
\frac{d}{d\rho}\log u(\rho)
=
\frac{2\rho^3-5\rho^2+4\rho+1}{\rho(\rho-1)(\rho^2-2\rho-1)}.
\]
当 \(\rho>1+\sqrt2\) 时，分母严格为正；而分子在 \(\rho=1+\sqrt2+t\)（\(t\ge 0\)）下化为
\[
2t^3+(1+6\sqrt2)t^2+(12+2\sqrt2)t+(4+4\sqrt2)>0.
\]
因此
\[
\frac{d}{d\rho}\log u(\rho)>0.
\]
又由定理 \ref{thm:conclusion-realinput40-collision-escort-algebraic-parametrization} 已知 \(\alpha'(\rho)>0\)，遂得
\[
J_2''(\alpha)
=
\frac{d(\log u)/d\rho}{d\alpha/d\rho}>0.
\]
严格凸性成立。证毕。
\end{proof}

\begin{corollary}\label{cor:conclusion-realinput40-centered-cramer-minimum-at-alpha0}\leanverified{paper\_conclusion\_realinput40\_centered\_cramer\_minimum\_at\_alpha0}
\(J_2\) 的极小值唯一出现在
\[
\alpha=\frac{3-\sqrt5}{10},
\]
且
\[
J_2\!\left(\frac{3-\sqrt5}{10}\right)=0.
\]
\end{corollary}
\begin{proof}
严格凸性由定理 \ref{thm:conclusion-realinput40-centered-cramer-explicit-algebraic-frontier} 已得。取 \(\theta=0\) 时，
\[
\alpha=P_2'(0)=\frac{3-\sqrt5}{10},
\]
且
\[
J_2(\alpha)=\alpha\cdot 0-\widetilde P_2(0)=0.
\]
由严格凸性知这是唯一极小点。证毕。
\end{proof}

\begin{theorem}[解析而超临界的密度带]\label{thm:conclusion-realinput40-analytic-supercritical-density-band}
\leanverified{paper\_conclusion\_realinput40\_analytic\_supercritical\_density\_band}
令
\[
\alpha_R:=P_2'(\theta_R),
\qquad
\alpha_{\mathrm{an}}:=\lim_{\theta\uparrow R_\theta}P_2'(\theta).
\]
则
\[
\frac{3-\sqrt5}{10}<\alpha_R<\alpha_{\mathrm{an}}<\frac12,
\]
并且对每个
\[
\alpha\in(\alpha_R,\alpha_{\mathrm{an}})
\]
都有如下双重性质同时成立：
\begin{enumerate}
  \item 生成该 \(\alpha\) 的唯一倾斜参数 \(\theta(\alpha)\) 仍位于压力函数 \(P_2\) 的主解析支内；
  \item 同时
  \[
  \beta(e^{\theta(\alpha)})>\frac12,
  \]
  即平方根误差律已经严格失效。
\end{enumerate}
因此，二碰撞模型存在一个解析而超临界的非空密度带。
\end{theorem}
\begin{proof}
主稿已证明
\[
\theta_R<R_\theta,
\]
故存在非空开区间
\[
(\theta_R,R_\theta)
\]
满足：其中每个 \(\theta\) 仍处于 \(P_2\) 的主解析支内，且
\[
\beta(e^\theta)>\frac12.
\]
另一方面，由定理 \ref{thm:conclusion-realinput40-collision-escort-algebraic-parametrization}，\(\theta\mapsto P_2'(\theta)\) 严格递增，故它把 \((\theta_R,R_\theta)\) 双射到某个非空开区间
\[
(\alpha_R,\alpha_{\mathrm{an}})\subset\left(0,\frac12\right).
\]
又因 \(\theta_R>0\)，故
\[
\alpha_R>P_2'(0)=\frac{3-\sqrt5}{10}.
\]
而 \(\alpha(\theta)\uparrow \frac12\) 仅在 \(\theta\to+\infty\) 时出现，且 \(R_\theta<\infty\)，故
\[
\alpha_{\mathrm{an}}<\frac12.
\]
于是对每个 \(\alpha\in(\alpha_R,\alpha_{\mathrm{an}})\)，其唯一对应参数 \(\theta(\alpha)\) 落于 \((\theta_R,R_\theta)\)，从而同时满足题述两条性质。证毕。
\end{proof}

\begin{corollary}\label{cor:conclusion-realinput40-analytic-supercritical-phase}
\leanverified{paper\_conclusion\_realinput40\_analytic\_supercritical\_phase}
平方根律的破坏并不以解析性崩塌为代价。相反，首先出现的是一个真正的解析超临界相：
\[
\text{解析性仍在}
\qquad\wedge\qquad
\text{平方根误差已破}.
\]
\end{corollary}
\begin{proof}
这正是定理 \ref{thm:conclusion-realinput40-analytic-supercritical-density-band} 的直接释义。证毕。
\end{proof}

\begin{corollary}[分歧半径证书对首次 RH 失效必然失明]\label{cor:conclusion-realinput40-branch-radius-certificate-blind-to-first-rh-failure}
在二碰撞压力模型中，任何只依赖“是否触及最近代数分歧点”的检测器，都不可能成为核版 RH/Ramanujan 首次失效的 sharp 证书。任何 sharp 证书都必须显式追踪次谱半径与 Perron 根平方根之间的比较，或其等价的次主谱观测量。
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-realinput40-analytic-supercritical-density-band}，存在非空开区间
\[
(\theta_R,R_\theta)
\]
使得对每个 \(\theta\) 落于其中，\(P_2\) 仍保持解析，而平方根误差律已经严格失效。故任何只依赖“是否到达 \(|\theta|=R_\theta\)”这一分歧事件的检测器，在整个该区间上都只能输出“尚未分歧”，从而对首次 RH 失效必然失明。

另一方面，主稿对核版 RH 的 sharp 判据正是次谱比较
\[
\beta(e^\theta)\le \frac12
\]
的成立与否；等价地，它必须显式进入 Perron 主模与次主谱之间的比较，而不能仅靠主解析支的分歧半径。证毕。
\end{proof}

\begin{theorem}[局部二次数域与全局三次分歧的算术分裂]\label{thm:conclusion-realinput40-local-quadratic-global-cubic-branch-splitting}
\leanverified{paper\_conclusion\_realinput40\_local\_quadratic\_global\_cubic\_branch\_splitting}
二碰撞压力函数 \(P_2\) 同时具有两种彼此不可约化的算术层：
\begin{enumerate}
  \item 局部层：其在 \(\theta=0\) 的 Taylor germ 满足
  \[
  P_2^{(k)}(0)\in \mathbb Q(\sqrt5)
  \qquad (k\ge 1);
  \]
  \item 全局层：其代数继续的分歧集合由三次判别式
  \[
  \Delta(u)=4u^3+25u^2+88u+8
  \]
  控制；该三次在 \(\mathbb Q\) 上不可约，且其判别式为
  \[
  \Disc(\Delta)=-5{,}324{,}000,
  \]
  因而其分裂域 \(L_\Delta\) 满足
  \[
  \operatorname{Gal}(L_\Delta/\mathbb Q)\cong S_3.
  \]
\end{enumerate}
因此，二碰撞压力在算术上发生严格分裂：局部累积量闭包属于二次数域，而全局分歧几何已经提升到非阿贝尔三次层。
\end{theorem}
\begin{proof}
附录 D 已证明：输出位势的一参数加权下，
\[
P_2^{(k)}(0)\in\mathbb Q(\sqrt5)
\qquad (k\ge 1),
\]
因此全部局部 cumulant 生成代数都落在 \(\mathbb Q(\sqrt5)\) 中。

另一方面，主稿关于二碰撞压力解析半径的判别式描述说明：全部 \(u\)-平面分歧点恰为
\[
\Delta(u)=4u^3+25u^2+88u+8
\]
的根。对该三次做有理根检验，\(\pm 1,\pm 2,\pm 4,\pm 8,\pm \frac12,\pm \frac14\) 皆非根，故它在 \(\mathbb Q\) 上不可约。再直接计算可得
\[
\Disc(\Delta)=-5{,}324{,}000,
\]
其不是平方数。于是不可约三次的分裂域 Galois 群不可能为 \(A_3\)，只能为
\[
\operatorname{Gal}(L_\Delta/\mathbb Q)\cong S_3.
\]
局部与全局两层遂严格分离。证毕。
\end{proof}

\begin{corollary}\label{cor:conclusion-realinput40-local-cumulant-field-not-global-branch-field}
\leanverified{paper\_conclusion\_realinput40\_local\_cumulant\_field\_not\_global\_branch\_field}
局部热力学统计与全局解析延拓并不共享同一最小数域。更精确地说，\(\mathbb Q(\sqrt5)\) 足以容纳全部局部 cumulant algebra，却不足以容纳控制全局分歧的 branch arithmetic。
\end{corollary}
\begin{proof}
这正是定理 \ref{thm:conclusion-realinput40-local-quadratic-global-cubic-branch-splitting} 的直接结论。证毕。
\end{proof}

\begin{conjecture}[高阶碰撞压力的局部--全局超临界分裂]\label{conj:conclusion-realinput40-highq-localglobal-supercritical-bifurcation}
设 \(\{P_q(\theta)\}_{q\ge 2}\) 为由对称幂 moment-kernel 编译得到的高阶碰撞压力族。则应存在如下普遍二分律：
\begin{enumerate}
  \item 对每个固定 \(q\)，\(\theta=0\) 处的全部局部 cumulant 生成代数落在一个有界次数数域 \(K_q/\mathbb Q\) 内；
  \item 与此同时，\(P_q\) 的全局代数继续由某个判别式多项式 \(\Delta_q(u)\) 控制，其分裂域在一般情形下达到最大允许的非阿贝尔 Galois 类型；
  \item 在 \(q\)-escort 前沿上，存在一个非空开区间，使得 \(P_q\) 仍保持解析而相应 prime-orbit 误差指数已越过其 Ramanujan/RH 型阈值。
\end{enumerate}
二碰撞情形给出的原型可压缩为
\[
\text{局部二次数域}
\Longleftrightarrow
\text{全局 }S_3\text{ 分歧}
\Longleftrightarrow
\text{解析超临界带}.
\]
\end{conjecture}
