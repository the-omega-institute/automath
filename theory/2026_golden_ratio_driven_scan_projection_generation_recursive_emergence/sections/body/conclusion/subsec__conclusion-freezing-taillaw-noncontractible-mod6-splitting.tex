\subsection{冻结相的一核普适性、尾律变换等价与模 \texorpdfstring{$6$}{6} 的度量--拓扑分裂}\label{subsec:conclusion-freezing-taillaw-noncontractible-mod6-splitting}
沿用定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse}、定理 \ref{thm:conclusion-frozen-escort-tv-rigidity}、定理 \ref{thm:conclusion-continuous-capacity-integral-node-inversion}、定理 \ref{thm:xi-time-part57b-capacity-moment-exact-mellin-stieltjes}、定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic}、定理 \ref{thm:conclusion-ae-recovery-threshold-sandwich}、定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 与定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios} 的记号。前文已经分别建立了冻结相 escort 分布对最大纤维层的指数集中、连续容量曲线与正阶矩塔的 Mellin--Stieltjes 对偶，以及非可缩最大纤维的模 \(6\) 相图。把这些链条在结论层闭合后，可进一步得到如下三条统一后果。

\begin{theorem}[冻结相是一核普适相]\label{thm:conclusion-frozen-phase-single-core-universality}
设 \(a,b>a_{\mathrm c}\)，并记
\[
A_m^{\max}:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
U_m^{\max}:=\mathrm{Unif}(A_m^{\max}).
\]
则对任意 \(t\in\{a,b\}\) 都存在概率测度 \(\nu_{m,t}\) 与实数 \(p_{m,t}\in[0,1]\)，使得
\[
\pi_{m,t}=p_{m,t}U_m^{\max}+(1-p_{m,t})\nu_{m,t},
\qquad
1-p_{m,t}\le \exp\!\bigl(-m\Delta(t)+o(m)\bigr).
\]
因此
\[
\bigl|\pi_{m,a}-\pi_{m,b}\bigr|_{\mathrm{TV}}
\le
\exp\!\bigl(-m\Delta(a)+o(m)\bigr)
+
\exp\!\bigl(-m\Delta(b)+o(m)\bigr).
\]
进一步，对每个 \(t>a_{\mathrm c}\) 都有
\[
\lim_{m\to\infty}\frac{1}{m}H_\infty(\pi_{m,t})=g_\ast.
\]
故冻结区在指数精度上只保留两项不变量：共同的微正则核心 \(U_m^{\max}\) 与其简并指数 \(g_\ast\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-frozen-escort-tv-rigidity}，
\[
\bigl|\pi_{m,t}-U_m^{\max}\bigr|_{\mathrm{TV}}
=
1-\pi_{m,t}(A_m^{\max})
\le
\exp\!\bigl(-m\Delta(t)+o(m)\bigr).
\]
若 \(\pi_{m,t}(A_m^{\max})<1\)，把 \(\pi_{m,t}\) 在补集 \(X_m\setminus A_m^{\max}\) 上的条件分布记为 \(\nu_{m,t}\)；若 \(\pi_{m,t}(A_m^{\max})=1\)，则任取 \(\nu_{m,t}\)。令
\[
p_{m,t}:=\pi_{m,t}(A_m^{\max}),
\]
便得分解
\[
\pi_{m,t}=p_{m,t}U_m^{\max}+(1-p_{m,t})\nu_{m,t}.
\]
对 \(t=a,b\) 分别应用三角不等式，
\[
\bigl|\pi_{m,a}-\pi_{m,b}\bigr|_{\mathrm{TV}}
\le
\bigl|\pi_{m,a}-U_m^{\max}\bigr|_{\mathrm{TV}}
+
\bigl|\pi_{m,b}-U_m^{\max}\bigr|_{\mathrm{TV}},
\]
即得全变差估计。

最后，由推论 \ref{cor:conclusion-frozen-escort-all-renyi-collapse} 对 \(s=\infty\) 的情形，
\[
\lim_{m\to\infty}\frac{1}{m}H_\infty(\pi_{m,t})=g_\ast
\qquad (t>a_{\mathrm c}).
\]
故冻结区不同 Rényi 阶的 escort 轨道在极限上共同塌缩到同一微正则核 \(U_m^{\max}\) 及同一简并指数 \(g_\ast\)。证毕。
\end{proof}

\begin{theorem}[尾计数、连续容量与全正矩谱的变换等价]\label{thm:conclusion-tail-capacity-moment-tail-law-equivalence}
\leanverified{paper\_conclusion\_tail\_capacity\_moment\_tail\_law\_equivalence}
固定分辨率 \(m\)。定义纤维尾计数函数
\[
N_m(t):=\bigl|\{x\in X_m:\ d_m(x)\ge t\}\bigr|,
\qquad t\ge 0,
\]
连续容量曲线
\[
C_m^{\mathrm{cont}}(T):=\sum_{x\in X_m}\min(d_m(x),T),
\]
以及正阶矩塔
\[
S_q(m):=\sum_{x\in X_m}d_m(x)^q,
\qquad q>0.
\]
则有严格恒等式
\[
C_m^{\mathrm{cont}}(T)=\int_0^T N_m(t)\,dt,
\qquad
S_q(m)=q\int_0^\infty t^{q-1}N_m(t)\,dt.
\]
因此，对每个固定 \(m\)，三类数据
\[
N_m(\cdot),
\qquad
C_m^{\mathrm{cont}}(\cdot),
\qquad
\{S_q(m)\}_{q>0}
\]
彼此唯一决定。特别地，若最优成功率满足
\[
\mathrm{Succ}_m(B)\ge 1-\varepsilon,
\]
则必要位预算满足
\[
B\ge m-\log_2|X_m|+\log_2(1-\varepsilon).
\]
从而在 \(|X_m|=F_{m+2}\sim \varphi^m/\sqrt5\) 时，
\[
\liminf_{m\to\infty}\frac{B}{m}\ge 1-\log_2\varphi=\log_2\!\frac{2}{\varphi}.
\]
\end{theorem}
\begin{proof}
第一式由定理 \ref{thm:conclusion-continuous-capacity-integral-node-inversion} 中连续容量曲线与尾计数函数的积分关系给出。第二式正是定理 \ref{thm:xi-time-part57b-capacity-moment-exact-mellin-stieltjes} 的 Mellin--Stieltjes 表达。

再由定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic}，对固定 \(m\)，连续容量曲线、纤维直方图与全体正阶矩谱彼此唯一决定。由于 \(X_m\) 有限，\(N_m\) 是有限支撑阶梯函数，而纤维直方图与尾计数函数互相差分可逆，故
\[
N_m(\cdot)
\Longleftrightarrow
C_m^{\mathrm{cont}}(\cdot)
\Longleftrightarrow
\{S_q(m)\}_{q>0}.
\]

最后，若 \(\mathrm{Succ}_m(B)\ge 1-\varepsilon\)，则由推论 \ref{cor:fold-oracle-pigeonhole-bound}，
\[
B\ge m-\log_2|X_m|+\log_2(1-\varepsilon).
\]
将 \(|X_m|=F_{m+2}\) 代入并除以 \(m\)，令 \(m\to\infty\) 即得
\[
\liminf_{m\to\infty}\frac{B}{m}\ge 1-\log_2\varphi.
\]
证毕。
\end{proof}

\begin{theorem}[容量族的完备可恢复性与碰撞层级的全决定]\label{thm:conclusion-capacity-family-complete-recovery-collision-hankel-quantum}
设 \(\mathcal F,\mathcal F'\) 为两套折叠系统。对每个分辨率 \(m\)，分别记其稳定类型空间为
\[
X_m,\qquad X'_m,
\]
纤维多重度函数为
\[
d_m:X_m\to\NN,
\qquad
d'_m:X'_m\to\NN,
\]
以及连续容量曲线为
\[
C^{\mathrm{cont}}_{\mathcal F,m}(T):=\sum_{x\in X_m}\min(d_m(x),T),
\qquad
C^{\mathrm{cont}}_{\mathcal F',m}(T):=\sum_{x'\in X'_m}\min(d'_m(x'),T).
\]
若对一切 \(m\ge 1\) 与 \(T\ge 0\) 都有
\[
C^{\mathrm{cont}}_{\mathcal F,m}(T)=C^{\mathrm{cont}}_{\mathcal F',m}(T),
\]
则下列对象逐一完全一致：
\begin{enumerate}
  \item 每个固定 \(m\) 的纤维多重度多重集
  \[
  \{d_m(x):x\in X_m\}
  \quad\text{与}\quad
  \{d'_m(x'):x'\in X'_m\};
  \]
  \item 每个固定 \(q\ge 2\) 的碰撞矩
  \[
  S_q^{\mathcal F}(m):=\sum_{x\in X_m}d_m(x)^q,
  \qquad
  S_q^{\mathcal F'}(m):=\sum_{x'\in X'_m}d'_m(x')^q;
  \]
  \item 对每个固定 \(q\ge 2\)，序列 \(m\mapsto S_q(m)\) 的最小首一整系数递推、Hankel 秩与最小线性 realization 维数。
\end{enumerate}
特别地，在 \(q=2\) 的情形，由相同的容量族唯一决定的二阶碰撞矩
\[
S_2(m)
\]
同时固定折叠量子信道的最小 Kraus 秩，等价地最小 Stinespring 环境维数。
\end{theorem}
\begin{proof}
对每个固定 \(m\)，定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic} 表明：连续容量曲线
\[
C_m^{\mathrm{cont}}(\cdot)
\]
与纤维直方图、尾计数函数以及全体正阶矩谱
\[
\{S_q(m)\}_{q>0}
\]
彼此唯一决定。因而若
\[
C^{\mathrm{cont}}_{\mathcal F,m}(T)=C^{\mathrm{cont}}_{\mathcal F',m}(T)
\qquad(\forall T\ge 0),
\]
则对应的纤维多重度多重集必然一致，并且对一切 \(q\ge 2\) 都有
\[
S_q^{\mathcal F}(m)=S_q^{\mathcal F'}(m).
\]
这就证明了前两点。

进一步，对固定 \(q\ge 2\)，由于两套系统在每个 \(m\) 上都给出同一个整数序列
\[
m\longmapsto S_q(m),
\]
故其最小首一整系数递推完全相同。再由定理 \ref{thm:hankel-rank-minimal}，该最小递推阶数恰等于 Hankel 秩，也恰等于最小线性 realization 维数；而命题 \ref{prop:pom-hankel-realization-protocol} 给出相应最小 realization 的标准构造。于是第三点随之成立。

最后，对 \(q=2\)，定理 \ref{thm:op-algebra-fold-quantum-channel-minimal-environment-equals-s2} 说明折叠量子信道的最小 Kraus 秩（等价地最小 Stinespring 环境维数）精确等于 \(S_2(m)\)。因此它同样由容量族唯一决定。证毕。
\end{proof}

\begin{theorem}[最大纤维与最大非可缩纤维的模 \texorpdfstring{$6$}{6} 分裂]\label{thm:conclusion-maxfiber-vs-noncontractible-mod6-splitting}
记
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
\widetilde D_m:=\max\{d_m(x): |K_x|\not\simeq *\}.
\]
则对任意 \(x\in X_m\) 都有严格同伦奇偶判别
\[
|K_x|\simeq *
\iff
d_m(x)\ \text{为偶数},
\]
\[
|K_x|\simeq S^{\tau(x)-1}
\iff
d_m(x)\ \text{为奇数}.
\]
并且对充分大的 \(m\) 有
\[
\widetilde D_m=
\begin{cases}
D_m, & m\equiv 0,4\pmod 6,\\[4pt]
D_m^{(2)}, & m\equiv 1,5\pmod 6,\\[4pt]
D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]
等价地，
\[
D_m=\widetilde D_m
\iff
m\equiv 0,4\pmod 6.
\]
因此，度量极大性与拓扑极大性只在两个模 \(6\) chamber 中耦合，而在其余四个 chamber 中发生严格分裂。
\end{theorem}
\begin{proof}
第一组等价正是定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 的内容：纤维多重度的奇偶性与独立集复形 \(K_x\) 的同伦型严格对齐。

另一方面，定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios} 的证明已经调用了定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，从而给出非可缩最大纤维 \(\widetilde D_m\) 在六个剩余类上分别由顶层谱的第一、第二或第三层给出，恰即
\[
\widetilde D_m=
\begin{cases}
D_m, & m\equiv 0,4\pmod 6,\\
D_m^{(2)}, & m\equiv 1,5\pmod 6,\\
D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]
于是
\(
D_m=\widetilde D_m
\)
当且仅当 \(m\equiv 0,4\pmod 6\)。其余四个剩余类中，最大纤维层必须落在可缩偶数层，而非可缩拓扑质量只可能退到第二层或第三层。证毕。
\end{proof}

以上三条结果把热力与拓扑两侧的接口压缩为同一条尾律链：冻结区的 escort 轨道在指数精度上塌缩到单一微正则核心；该核心之外的全部纤维信息又完全等价于尾计数函数、连续容量曲线与正阶矩塔的 Mellin--Stieltjes 三重像；而当把同一尾律投到最大层的独立集复形上时，度量极大层与拓扑极大层只在模 \(6\) 的两个 chamber 中重合，在其余四个 chamber 中发生严格的度量--拓扑分裂。
