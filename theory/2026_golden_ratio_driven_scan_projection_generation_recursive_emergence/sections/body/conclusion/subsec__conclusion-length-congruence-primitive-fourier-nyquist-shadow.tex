\subsection{长度同余类的 primitive Fourier 层析、长度 \texorpdfstring{$2$}{2} 原子注入与 Nyquist 精确采样}\label{subsec:conclusion-length-congruence-primitive-fourier-nyquist-shadow}
沿用附录 \ref{app:real-input-40-length-mertens}、定理 \ref{thm:finite-part-nyquist-parseval-aliasing}、定理 \ref{thm:xi-root-unity-filter-single-class-injection} 与定理 \ref{thm:xi-atomic-witt-energy-residue-l2-law} 的记号。记真实输入 \(40\) 状态核的 primitive 轨道生成函数为
\[
P(z)=\sum_{n\ge 1}p_n z^n,
\qquad
\lambda=\varphi^2,
\]
并记总 Mertens 常数为 \(\mathsf M\)。对整数 \(m\ge 2\) 与剩余类 \(r\in \ZZ/m\ZZ\)，记
\[
\mathsf M_{m,r}^{(\mathrm{len})}
\]
为长度同余类 \(|\tau|\equiv r\pmod m\) 的 Mertens 常数，并定义中心化常数
\[
\widetilde{\mathsf M}_{m,r}
:=
\mathsf M_{m,r}^{(\mathrm{len})}-\frac{\mathsf M}{m}.
\]
对单位根 \(\zeta_m=e^{2\pi i/m}\)，再记
\[
S(\omega;\infty):=\sum_{n\ge 1}\frac{p_n\,\omega^n}{\lambda^n}
\qquad (|\omega|=1,\ \omega\neq 1).
\]
在这一口径下，长度同余类的 Mertens 常数、primitive 扭点值表、长度 \(2\) 原子的 residue 局域性以及 Nyquist 无混叠采样可被压缩为一条共同的 cyclotomic 审计链。

\begin{theorem}[长度同余类 Mertens 常数与 torsion primitive 表的精确 Fourier 层析]\label{thm:conclusion-length-congruence-mertens-torsion-fourier-tomography}
\leanverified{paper\_conclusion\_length\_congruence\_mertens\_torsion\_fourier\_tomography}
对任意 \(m\ge 2\) 与 \(r\in \ZZ/m\ZZ\)，有严格恒等式
\[
\widetilde{\mathsf M}_{m,r}
=
\frac1m\sum_{j=1}^{m-1}\zeta_m^{-jr}P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)
=
\frac1m\sum_{j=1}^{m-1}\zeta_m^{-jr}S(\zeta_m^{\,j};\infty).
\]
反之，对每个 \(1\le j\le m-1\)，成立精确反演
\[
P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)
=
S(\zeta_m^{\,j};\infty)
=
\sum_{r=0}^{m-1}\zeta_m^{jr}\widetilde{\mathsf M}_{m,r}.
\]
并且
\[
\sum_{r=0}^{m-1}\mathsf M_{m,r}^{(\mathrm{len})}=\mathsf M,
\qquad
\sum_{r=0}^{m-1}\widetilde{\mathsf M}_{m,r}=0.
\]
因此，长度同余类 Mertens 常数向量正是非平凡 torsion primitive 值表
\[
\left(P\!\left(\frac{\zeta_m}{\lambda}\right),\dots,P\!\left(\frac{\zeta_m^{m-1}}{\lambda}\right)\right)
\]
的离散 Fourier 变换。
\end{theorem}
\begin{proof}
由定理 \ref{thm:real-input-40-length-mertens-prog}，
\[
\mathsf M_{m,r}^{(\mathrm{len})}
=
\frac{\mathsf M}{m}
+
\frac1m\sum_{j=1}^{m-1}\zeta_m^{-jr}P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right).
\]
减去 \(\mathsf M/m\) 即得第一式。再由命题 \ref{prop:real-input-40-length-twist-converge}，
\[
S(\omega;\infty)=P\!\left(\frac{\omega}{\lambda}\right),
\]
故第二个表达式也成立。

对中心化向量作离散 Fourier 反演，得到
\[
\sum_{r=0}^{m-1}\zeta_m^{jr}\widetilde{\mathsf M}_{m,r}
=
\frac1m\sum_{\ell=1}^{m-1}P\!\left(\frac{\zeta_m^{\,\ell}}{\lambda}\right)
\sum_{r=0}^{m-1}\zeta_m^{(j-\ell)r}
=
P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right),
\]
其中最后一步使用
\[
\sum_{r=0}^{m-1}\zeta_m^{(j-\ell)r}=m\,\delta_{j\ell}.
\]
再对 \(r\) 求和即可得到
\[
\sum_{r=0}^{m-1}\widetilde{\mathsf M}_{m,r}=0
\]
以及
\[
\sum_{r=0}^{m-1}\mathsf M_{m,r}^{(\mathrm{len})}=\mathsf M.
\]
证毕。
\end{proof}

\begin{corollary}[长度同余常数的 Parseval 能量恒等式与严格均匀化判据]\label{cor:conclusion-length-congruence-mertens-parseval-uniformization}
对任意 \(m\ge 2\)，成立 Parseval 恒等式
\[
\sum_{r=0}^{m-1}\bigl|\widetilde{\mathsf M}_{m,r}\bigr|^2
=
\frac1m\sum_{j=1}^{m-1}\left|P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)\right|^2
=
\frac1m\sum_{j=1}^{m-1}\left|S(\zeta_m^{\,j};\infty)\right|^2.
\]
从而以下三命题等价：
\[
\mathsf M_{m,0}^{(\mathrm{len})}
=
\mathsf M_{m,1}^{(\mathrm{len})}
=\cdots=
\mathsf M_{m,m-1}^{(\mathrm{len})},
\]
\[
P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)=0
\qquad (1\le j\le m-1),
\]
\[
S(\zeta_m^{\,j};\infty)=0
\qquad (1\le j\le m-1).
\]
因此，长度同余类的完全均匀化等价于全部非平凡 torsion primitive 通道的严格湮灭。
\end{corollary}
\leanverified{paper\_conclusion\_length\_congruence\_mertens\_parseval\_uniformization}
\begin{proof}
由定理 \ref{thm:conclusion-length-congruence-mertens-torsion-fourier-tomography}，向量
\[
\bigl(\widetilde{\mathsf M}_{m,r}\bigr)_{r\in\ZZ/m\ZZ}
\]
与
\[
\left(P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)\right)_{0\le j\le m-1}
\]
互为离散 Fourier 变换，而 \(j=0\) 项恰为 \(0\)。于是离散 Parseval 恒等式给出
\[
\sum_{r=0}^{m-1}\bigl|\widetilde{\mathsf M}_{m,r}\bigr|^2
=
\frac1m\sum_{j=1}^{m-1}\left|P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)\right|^2.
\]
再由命题 \ref{prop:real-input-40-length-twist-converge} 的恒等式
\[
P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)=S(\zeta_m^{\,j};\infty)
\]
便得到第二个表示。由于平方和为零当且仅当每一项皆为零，所列三命题遂互相等价。证毕。
\end{proof}

\begin{theorem}[长度 \(2\) 原子在 root-of-unity 长度筛中的单类注入律]\label{thm:conclusion-length2-atomic-root-unity-singleclass-injection}
\leanverified{paper\_conclusion\_length2\_atomic\_root\_unity\_singleclass\_injection}
固定模数 \(q\ge 2\)。对每个剩余类 \(a\in \ZZ/q\ZZ\)，定义 residue-filtered primitive 与 periodic 对数生成函数
\[
P_{a,q}(z,u):=\frac1q\sum_{j=0}^{q-1}\omega_q^{-aj}P(\omega_q^j z,u),
\qquad
\omega_q:=e^{2\pi i/q},
\]
\[
A_{a,q}(z,u):=\frac1q\sum_{j=0}^{q-1}\omega_q^{-aj}\log\zeta(\omega_q^j z,u),
\]
并同样记相应的 core 部分为 \(P_{a,q}^{\mathrm{core}}(z,u)\) 与 \(A_{a,q}^{\mathrm{core}}(z,u)\)。则有严格恒等式
\[
P_{a,q}(z,u)
=
P_{a,q}^{\mathrm{core}}(z,u)
+
u z^2\,\mathbf 1_{\{a\equiv 2\;(\mathrm{mod}\,q)\}},
\]
\[
A_{a,q}(z,u)
=
A_{a,q}^{\mathrm{core}}(z,u)
+
\sum_{\substack{k\ge 1\\ 2k\equiv a\;(\mathrm{mod}\,q)}}\frac{u^k z^{2k}}{k}.
\]
特别地，长度 \(2\) 的原子因子在 primitive 层只向唯一同余类 \(a\equiv 2\pmod q\) 注入质量，而在 periodic 层也只沿梳齿约束
\[
2k\equiv a\pmod q
\]
发生局域修正。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:xi-root-unity-filter-single-class-injection} 的内容。证毕。
\end{proof}

\begin{corollary}[长度 \(2\) 原子在 torsion 频域上的能量守幅律]\label{cor:conclusion-length2-atomic-torsion-energy-rigidity}
\leanverified{paper\_conclusion\_length2\_atomic\_torsion\_energy\_rigidity}
沿用定理 \ref{thm:conclusion-length2-atomic-root-unity-singleclass-injection} 的记号。对任意模数 \(q\ge 2\)，都有
\[
\frac1q\sum_{j=0}^{q-1}
\left|p_n(\omega_q^j)-p_n^{\mathrm{core}}(\omega_q^j)\right|^2
=
\mathbf 1_{\{n=2\}},
\]
\[
\frac1q\sum_{j=0}^{q-1}
\left|a_n(\omega_q^j)-a_n^{\mathrm{core}}(\omega_q^j)\right|^2
=
4\,\mathbf 1_{\{2\mid n\}}.
\]
因此，长度 \(2\) 原子在 torsion 频域上表现出双重刚性：primitive 层只占据唯一长度 \(n=2\)，而 periodic 层对一切偶长度均保持固定能量 \(4\)，对奇长度则严格为零。
\end{corollary}
\begin{proof}
这正是定理 \ref{thm:xi-atomic-witt-energy-residue-l2-law} 的平方平均公式。证毕。
\end{proof}

\begin{theorem}[torsion 采样的 Nyquist 精确化判据]\label{thm:conclusion-torsion-sampling-nyquist-noalias-exactness}
\leanverified{paper\_conclusion\_torsion\_sampling\_nyquist\_noalias\_exactness}
记
\[
g(\theta):=-\log F_M(e^{i\theta})
=
\sum_{n\ge 1}\frac{u_n(M)}{n}e^{in\theta},
\qquad \theta\in\RR.
\]
对任意整数 \(q\ge 1\)，取采样点 \(\theta_j:=2\pi j/q\)。若系数支撑集
\[
\operatorname{Supp}(u):=\{n\ge 1:\ u_n(M)\neq 0\}
\]
满足模 \(q\) 的无碰撞条件
\[
n\neq n'\in \operatorname{Supp}(u)
\Longrightarrow
n\not\equiv n' \pmod q,
\]
则离散采样能量与连续 Parseval 质量严格一致：
\[
\frac1q\sum_{j=0}^{q-1}|g(\theta_j)|^2
=
\sum_{n\ge 1}\frac{|u_n(M)|^2}{n^2}.
\]
特别地，若存在 \(N_\ast\) 使
\[
u_n(M)=0\qquad (n>N_\ast),
\]
则对一切 \(q>N_\ast\) 都有上式成立。
\end{theorem}
\begin{proof}
由定理 \ref{thm:finite-part-nyquist-parseval-aliasing}，
\[
\frac1q\sum_{j=0}^{q-1}|g(\theta_j)|^2
=
\sum_{n\ge 1}\frac{|u_n(M)|^2}{n^2}
+
\sum_{\ell\neq 0}\sum_{n\ge 1}\frac{u_n(M)\,\overline{u_{n+\ell q}(M)}}{n(n+\ell q)}.
\]
若 \(\operatorname{Supp}(u)\) 在模 \(q\) 下无碰撞，则对任意 \(\ell\neq 0\) 与任意 \(n\ge 1\)，不可能同时有
\[
u_n(M)\neq 0
\qquad\text{且}\qquad
u_{n+\ell q}(M)\neq 0,
\]
因为这将迫使 \(n\equiv n+\ell q\pmod q\) 并违反无碰撞条件。故全部混叠项严格消失，所示等式成立。

若 \(u_n(M)=0\) 对一切 \(n>N_\ast\) 成立，则集合 \(\{1,\dots,N_\ast\}\) 在任何 \(q>N_\ast\) 下都落入互异的剩余类，因此自动满足无碰撞条件。证毕。
\end{proof}

\begin{theorem}[长度奇偶 Mertens 偏置是 primitive 对数在代数扭点处的 Euler 值]\label{thm:conclusion-length-parity-mertens-algebraic-euler-value}
在 \(m=2\) 时，长度同余类 Mertens 常数满足严格二分
\[
\mathsf M_{2,0}^{(\mathrm{len})}
=
\frac{\mathsf M}{2}
+
\frac12\,P\!\left(-\frac1\lambda\right),
\qquad
\mathsf M_{2,1}^{(\mathrm{len})}
=
\frac{\mathsf M}{2}
-
\frac12\,P\!\left(-\frac1\lambda\right).
\]
因而偏置差恰为
\[
\mathsf M_{2,0}^{(\mathrm{len})}-\mathsf M_{2,1}^{(\mathrm{len})}
=
P\!\left(-\frac1\lambda\right)
=
S(-1;\infty).
\]
另一方面，长度符号 Euler 乘积
\[
L_-(s):=
\prod_{\tau\ \mathrm{primitive}}
\left(1-(-1)^{|\tau|}N(\tau)^{-s}\right)^{-1}
\]
在 \(s=1\) 处满足精确闭式
\[
L_-(1)
=
\zeta_M\!\left(-\frac1\lambda\right)
=
\frac{41}{40}+\frac{11}{24}\sqrt5.
\]
并且
\[
P(z)=\sum_{k\ge 1}\frac{\mu(k)}{k}\log\zeta_M(z^k)
\]
在 \(z=-1/\lambda\) 处给出
\[
P\!\left(-\frac1\lambda\right)
=
\sum_{k\ge 1}\frac{\mu(k)}{k}\log\zeta_M\!\left(\left(-\frac1\lambda\right)^k\right).
\]
因此，长度奇偶 Mertens 偏置是 primitive 对数在代数扭点 \(-\lambda^{-1}=-\varphi^{-2}\) 处的精确 Euler 型取值。
\end{theorem}
\begin{proof}
由定理 \ref{thm:real-input-40-length-mertens-prog} 在 \(m=2\) 处的专门化，
\[
\mathsf M_{2,r}^{(\mathrm{len})}
=
\frac{\mathsf M}{2}
+
\frac12(-1)^rP\!\left(-\frac1\lambda\right),
\]
从而得到前两式。又由命题 \ref{prop:real-input-40-length-twist-converge}，
\[
S(-1;\infty)=P\!\left(-\frac1\lambda\right).
\]
而推论 \ref{cor:real-input-40-length-sign-L1} 给出
\[
L_-(1)=\zeta_M\!\left(-\frac1\lambda\right)=\frac{41}{40}+\frac{11}{24}\sqrt5.
\]
最后，命题 \ref{prop:real-input-40-length-twist-converge} 中的 Möbius--log 反演公式在 \(z=-1/\lambda\) 处直接给出最后一个展开。证毕。
\end{proof}

\begin{conjecture}[全模 length-congruence Mertens 常数的最小 cyclotomic 完备性]\label{conj:conclusion-length-congruence-minimal-cyclotomic-shadow}
记
\[
\mathfrak M_{\mathrm{len}}
:=
\left(\mathsf M_{m,r}^{(\mathrm{len})}\right)_{m\ge 2,\ r\in\ZZ/m\ZZ}.
\]
则应有如下完备性图景：
\begin{enumerate}
\item \(\mathfrak M_{\mathrm{len}}\) 与全部 boundary torsion 值
\[
\left(P\!\left(\frac{\zeta_m^{\,j}}{\lambda}\right)\right)_{m\ge 2,\ 1\le j\le m-1}
\]
完全等价；
\item 任意 proper cofinal 子系统
\[
\mathcal M\subsetneq \ZZ_{\ge 2}
\]
在一般情形下都不足以恢复全部 primitive 长度算术；
\item 因而 \(\mathfrak M_{\mathrm{len}}\) 构成长度方向上的最小 cyclotomic 完备阴影。
\end{enumerate}
\end{conjecture}

\paragraph{统一读法}
定理 \ref{thm:conclusion-length-congruence-mertens-torsion-fourier-tomography} 与推论 \ref{cor:conclusion-length-congruence-mertens-parseval-uniformization} 表明，长度同余类 Mertens 常数并不是对 primitive 长度谱的事后分箱，而是一条严格的 torsion primitive Fourier 层析轴；其完全均匀化恰对应于全部非平凡 torsion 通道的同时湮灭。定理 \ref{thm:conclusion-length2-atomic-root-unity-singleclass-injection} 与推论 \ref{cor:conclusion-length2-atomic-torsion-energy-rigidity} 则说明，即便对最基本的长度 \(2\) 原子因子，residue 过滤与 torsion 频域也呈现出单类注入与严格守幅的局域几何，而非扩散型修正。最后，定理 \ref{thm:conclusion-torsion-sampling-nyquist-noalias-exactness} 把这条 cyclotomic 轴推进到采样论层面：一旦离开同余混叠窗口，torsion 采样便成为严格无损的 Nyquist 证书；而定理 \ref{thm:conclusion-length-parity-mertens-algebraic-euler-value} 又把最基本的奇偶偏置压缩为 primitive 对数在代数扭点 \(-\varphi^{-2}\) 处的 Euler 型取值。由此，长度同余类、root-of-unity 扭点、Möbius--log 抽取与 Nyquist 混叠在同一条 cyclotomic 审计轴上闭合为一个统一对象。
