\subsection{weighted sync-kernel 的自对偶完成化、dyadic Chebyshev--Dwork 塔与 odd-prime unit-root 固定点}\label{subsec:conclusion-weighted-synckernel-selfdual-dyadic-unitroot-adelic-splitting}
沿用加权同步核的记号
\[
B(u)=B_0+uB_1,
\qquad
a_n(u):=\Tr(B(u)^n),
\qquad
\widetilde B(\theta):=e^{-\theta/2}B(e^\theta),
\qquad
\widetilde a_n(\theta):=\Tr(\widetilde B(\theta)^n).
\]
此前结果已经分别给出：完成化矩阵的自对偶相似、Ward 型一阶正交恒等式、\(p\)-幂 Dwork 同余、\(p=2\) 时在不变量坐标 \(t=u+u^{-1}\) 上闭合的 Chebyshev--Dwork 链，以及奇素数 unit-root 切片上的 \(p\)-进极限不变量。把这些接口合并之后，可以把 weighted sync-kernel 的局部算术--解析行为压缩成一条此前尚未在正文结论区单独抽出的链：Archimedean 侧只保留偶喷流，dyadic 侧保留真正的重整化动力学，而 odd-prime 的 unit-root 切片则塌缩为 fixed-point 塔。

\begin{proposition}[完成迹的偶性与自对偶信息压缩]\label{prop:conclusion-weighted-sync-completed-trace-even}
若自对偶置换矩阵 \(\Pi\) 满足
\[
\Pi^{-1}B_0\Pi=B_1,
\qquad
\Pi^{-1}B_1\Pi=B_0,
\]
则对任意整数 \(n\ge 1\)，完成迹
\[
\widetilde a_n(\theta)=\Tr\!\bigl(\widetilde B(\theta)^n\bigr)
\]
满足严格偶性
\[
\widetilde a_n(\theta)=\widetilde a_n(-\theta).
\]
因此对任意 \(r\ge 0\) 都有
\[
\widetilde a_n^{(2r+1)}(0)=0.
\]
\end{proposition}
\leanverified{paper\_conclusion\_weighted\_sync\_completed\_trace\_even}

\begin{proof}
由引理 \ref{lem:completed-similarity}，
\[
\Pi^{-1}\widetilde B(\theta)\Pi=\widetilde B(-\theta).
\]
故 \(\widetilde B(\theta)\) 与 \(\widetilde B(-\theta)\) 相似，从而
\[
\Tr\!\bigl(\widetilde B(\theta)^n\bigr)=\Tr\!\bigl(\widetilde B(-\theta)^n\bigr).
\]
因此 \(\widetilde a_n\) 为偶函数，所有奇阶导数在 \(\theta=0\) 处严格消失。
\end{proof}

\begin{remark}[Ward 恒等式只是第一奇喷流的影子]\label{rem:conclusion-weighted-sync-completed-trace-even-upgrades-ward}
命题 \ref{prop:conclusion-weighted-sync-completed-trace-even} 的 \(r=0\) 情形给出
\(
\widetilde a_n'(0)=0
\)。
结合引理 \ref{lem:sync-kernel-ward-traces} 证明中的显式公式
\(
\widetilde B'(0)=B_{\mathrm{asym}}
\)，
便恢复 Ward 恒等式
\[
\Tr\!\bigl(B(1)^{n-1}B_{\mathrm{asym}}\bigr)=0.
\]
因此 Ward 消失律并非孤立的一阶事实，而是完成迹全体奇喷流同时湮灭的首阶投影。
\end{remark}

\begin{theorem}[Archimedean jet 的二分压缩律]\label{thm:conclusion-weighted-sync-archimedean-jet-binary-compression}
定义
\[
A_n^{(m)}:=\left.\frac{\dd^m}{\dd\theta^m}a_n(e^\theta)\right|_{\theta=0},
\qquad
\widetilde A_n^{(2j)}:=\left.\frac{\dd^{2j}}{\dd\theta^{2j}}\widetilde a_n(\theta)\right|_{\theta=0}.
\]
则对任意 \(m\ge 0\) 成立严格展开
\[
A_n^{(m)}
=
\sum_{j=0}^{\lfloor m/2\rfloor}
\binom{m}{2j}
\left(\frac n2\right)^{m-2j}
\widetilde A_n^{(2j)}.
\]
特别地，整个 Archimedean Taylor 塔完全由偶阶完成迹喷流决定；所有奇阶导数都不是独立数据。
\end{theorem}

\begin{proof}
由定义
\[
a_n(e^\theta)=e^{n\theta/2}\widetilde a_n(\theta).
\]
对右端作 Leibniz 展开，
\[
\frac{\dd^m}{\dd\theta^m}a_n(e^\theta)
=
\sum_{r=0}^{m}\binom{m}{r}
\left(\frac n2\right)^{m-r}
e^{n\theta/2}\widetilde a_n^{(r)}(\theta).
\]
在 \(\theta=0\) 处取值，并用命题 \ref{prop:conclusion-weighted-sync-completed-trace-even} 中奇阶导数严格消失的事实，便只剩偶阶项，从而得到所示公式。
\end{proof}

\begin{theorem}[dyadic Chebyshev--Dwork 塔的有限精度闭包]\label{thm:conclusion-weighted-sync-dyadic-chebyshev-dwork-finite-precision}
定义 Chebyshev 迭代
\[
C_0(t):=t,
\qquad
C_{r+1}(t):=C_r(t)^2-2.
\]
设偶次回文迹的不变量表示为
\[
a_{2^m}(u)=u^{2^{m-1}}A_{2^m}(t),
\qquad
t:=u+u^{-1}.
\]
则对任意整数 \(k\ge 2\) 与 \(2\le \ell\le k\)，在 \(\ZZ[t]\) 中有同余
\[
\boxed{\;
A_{2^k}(t)\equiv A_{2^{\ell-1}}\!\bigl(C_{k-\ell+1}(t)\bigr)\pmod{2^\ell}\; }.
\]
因此，在任意固定精度 \(2^\ell\)（\(\ell\ge 2\)）层上，长度 \(2^k\) 的 dyadic 算术信息都被压缩到单变量迭代轨道
\[
t\mapsto C_1(t)\mapsto C_2(t)\mapsto \cdots \mapsto C_{k-\ell+1}(t)
\]
上；其验证深度相对于长度 \(2^k\) 仅作对数增长。
\leanverified{paper\_conclusion\_weighted\_sync\_dyadic\_chebyshev\_dwork\_finite\_precision}
\end{theorem}

\begin{proof}
由命题 \ref{prop:witt-frobenius-iterated-descent}（取 \(p=2\)），对 \(2\le \ell\le k\) 有
\[
a_{2^k}(u)\equiv a_{2^{\ell-1}}\!\bigl(u^{2^{k-\ell+1}}\bigr)\pmod{2^\ell}.
\]
再由引理 \ref{lem:trace-palindrome-even-invariant}，
\[
a_{2^k}(u)=u^{2^{k-1}}A_{2^k}(t),
\]
且
\[
a_{2^{\ell-1}}\!\bigl(u^{2^{k-\ell+1}}\bigr)
=
\bigl(u^{2^{k-\ell+1}}\bigr)^{2^{\ell-2}}
A_{2^{\ell-1}}\!\bigl(u^{2^{k-\ell+1}}+u^{-2^{k-\ell+1}}\bigr)
=
u^{2^{k-1}}A_{2^{\ell-1}}\!\bigl(u^{2^{k-\ell+1}}+u^{-2^{k-\ell+1}}\bigr).
\]
另一方面，由递推定义可归纳得到
\[
u^{2^r}+u^{-2^r}=C_r(u+u^{-1})=C_r(t)\qquad(r\ge 0).
\]
将其代入上式，并在 Laurent 环中消去公共因子 \(u^{2^{k-1}}\)，即可得到
\[
A_{2^k}(t)\equiv A_{2^{\ell-1}}\!\bigl(C_{k-\ell+1}(t)\bigr)\pmod{2^\ell}.
\]
最后，迭代步数为 \(k-\ell+1=O(k)\)，而长度为 \(2^k\)，故验证深度相对于长度只按 \(\log_2(2^k)=k\) 增长。
\end{proof}

\begin{theorem}[奇素数 unit-root 切片的静态化与反旋零点]\label{thm:conclusion-weighted-sync-oddprime-unitroot-staticization}
设 \(p\) 为奇素数，且 \(u\in\mu_{p-1}\subset\ZZ_p^\times\) 为 Teichm\"uller 单位，因此
\[
u^p=u.
\]
则 \(p\)-幂 ghost 塔 \(\{a_{p^k}(u)\}_{k\ge 0}\) 在 \(\ZZ_p\) 中收敛到极限
\[
A_p(u):=\lim_{k\to\infty}a_{p^k}(u)\in\ZZ_p,
\]
并满足强尾估计
\[
A_p(u)-a_{p^k}(u)\in p^{k+1}\ZZ_p
\qquad(k\ge 0).
\]
特别地，对任意 \(\ell\ge 1\) 与 \(k\ge \ell\)，都有
\[
a_{p^k}(u)\equiv A_p(u)\pmod{p^\ell}.
\]
此外，这些极限自动满足自对偶 fixed-point 关系
\[
A_p(u)=u\,A_p(1/u).
\]
因此在唯一非平凡自反 Teichm\"uller 点 \(u=-1\) 上，
\[
A_p(-1)=0.
\]
换言之，奇素数的 unit-root 切片不是重整化动力学塔，而是有限精度上最终静态化的 fixed-point 塔；其反旋点必然湮灭。
\end{theorem}

\begin{proof}
命题 \ref{prop:witt-unit-root-limit} 已给出：当 \(u^p=u\) 时，
\[
v_p\!\bigl(a_{p^k}(u)-a_{p^{k-1}}(u)\bigr)\ge k
\qquad(k\ge 1),
\]
故 \(\{a_{p^k}(u)\}\) 在 \(\ZZ_p\) 中为 Cauchy 列并收敛到某个极限 \(A_p(u)\)。
并且对任意 \(r>k\)，有
\[
a_{p^r}(u)-a_{p^k}(u)
=
\sum_{m=k+1}^{r}\bigl(a_{p^m}(u)-a_{p^{m-1}}(u)\bigr)\in p^{k+1}\ZZ_p.
\]
令 \(r\to\infty\) 即得
\[
A_p(u)-a_{p^k}(u)\in p^{k+1}\ZZ_p,
\]
从而推出所述模 \(p^\ell\) 稳定性。

另一方面，由命题 \ref{prop:trace-palindrome}，
\[
a_{p^k}(u)=u^{p^k}a_{p^k}(1/u).
\]
在 \(u^p=u\) 的 Teichm\"uller 情形下，\(u^{p^k}=u\)，故
\[
a_{p^k}(u)=u\,a_{p^k}(1/u).
\]
令 \(k\to\infty\) 便得到
\[
A_p(u)=u\,A_p(1/u).
\]
最后取 \(u=-1\) 得
\[
A_p(-1)=-A_p(-1).
\]
由于 \(2\) 在 \(\ZZ_p\) 中可逆（\(p\) 为奇素数），故 \(A_p(-1)=0\)。
\end{proof}

\begin{theorem}[weighted sync-kernel 的 adelic 二相分裂]\label{thm:conclusion-weighted-sync-adelic-two-phase-splitting}
设 \(a_n(u)=\Tr(B(u)^n)\) 为 weighted sync-kernel 的周期迹。
则其算术--解析局部行为发生严格的 adelic 二相分裂：
\begin{enumerate}
  \item \textbf{压缩相：}Archimedean 完成迹 \(\widetilde a_n(\theta)\) 为偶函数，因此整个 Archimedean Taylor 塔由偶阶喷流唯一决定；同时，对每个奇素数 \(p\) 的 unit-root 切片 \(u^p=u\)，\(p\)-幂 ghost 塔 \(\{a_{p^k}(u)\}\) 在有限精度层上最终静态化，并收敛到 fixed-point 数据 \(A_p(u)\)。

  \item \textbf{动力相：}在 \(p=2\) 上，\(2\)-幂 trace 塔不是静止的，而由单变量 Chebyshev 迭代
  \[
  t\longmapsto t^2-2
  \]
  精确支配；模 \(2^\ell\) 的任意有限精度信息都沿定理 \ref{thm:conclusion-weighted-sync-dyadic-chebyshev-dwork-finite-precision} 的 renormalization 链传递。
\end{enumerate}
因此，weighted sync-kernel 在 arithmetic--spectral 层面的全部真正局部重整化动力学都集中在 dyadic 分支；奇素数只保留 fixed-point 型局部残差，而 Archimedean 侧只保留偶喷流几何。
\end{theorem}

\begin{proof}
Archimedean 偶喷流压缩由命题 \ref{prop:conclusion-weighted-sync-completed-trace-even} 与定理 \ref{thm:conclusion-weighted-sync-archimedean-jet-binary-compression} 给出；
dyadic 单变量重整化由定理 \ref{thm:conclusion-weighted-sync-dyadic-chebyshev-dwork-finite-precision} 给出；
odd-prime unit-root 切片的固定点静态化由定理 \ref{thm:conclusion-weighted-sync-oddprime-unitroot-staticization} 给出。
三者合并即得所述分裂。
\end{proof}

\begin{conjecture}[dyadic branch 的全局支配性]\label{conj:conclusion-weighted-sync-dyadic-global-dominance}
对每个素数 \(p\)，记 \(p\)-典型局部因子为
\[
Z_p(T;u):=\exp\!\left(\sum_{k\ge 0}\frac{a_{p^k}(u)}{p^k}T^{p^k}\right),
\]
其 \(\ZZ_p[u]\)-整性由定理 \ref{thm:witt-dwork-zp-integrality} 保证。
设 \(\mathscr W(B)\) 为 weighted sync-kernel 的全局 Big-Witt 类。
则应有如下刚性：

若两个加权同步核在
\begin{enumerate}
  \item 全部奇素数 unit-root 切片上的 fixed-point 数据 \(\{A_p(u)\}_{u\in\mu_{p-1}}\)；
  \item 全部 Archimedean 偶喷流 \(\{\widetilde A_n^{(2j)}\}_{n,j}\)；
\end{enumerate}
都完全一致，而在 dyadic 分支的 Chebyshev--Dwork 塔上不同，则它们的全局 Big-Witt 类仍必不同。

换言之，全局算术--谱分类的全部非平凡动力学自由度应当由 \(p=2\) 单独承担：Archimedean 侧只提供偶喷流几何，odd-prime 侧只提供 fixed-point 数据，而真正携带尺度递归与非平凡重整化动力学的，只有 dyadic 分支。
\end{conjecture}

\begin{remark}[统一读法]\label{rem:conclusion-weighted-synckernel-selfdual-dyadic-unitroot-adelic-splitting}
命题 \ref{prop:conclusion-weighted-sync-completed-trace-even} 与定理 \ref{thm:conclusion-weighted-sync-archimedean-jet-binary-compression} 表明，自对偶完成化把 Archimedean 侧压缩成纯偶喷流几何；
定理 \ref{thm:conclusion-weighted-sync-dyadic-chebyshev-dwork-finite-precision} 表明，\(p=2\) 上的全部有限精度算术通过单变量迭代 \(t\mapsto t^2-2\) 闭合；
定理 \ref{thm:conclusion-weighted-sync-oddprime-unitroot-staticization} 则表明，奇素数的 unit-root 切片最终只留下 fixed-point 极限 \(A_p(u)\) 及其自对偶约束。
因此，这一链条的真正结构结论不是“又一组 Dwork 同余”，而是一个明确的 adelic 角色分工：dyadic 分支承载动力学，其余局部分支只承载压缩后的静态证书。
\end{remark}
