\subsection{primitive/ghost Big-Witt 原子的 Ramanujan 长度层析与谐波射线分离}\label{subsec:conclusion-primitive-ramanujan-divisor-ghost-tomography}
在加权 primitive--periodic Witt 分解
\[
p_n(u)=\frac1n\sum_{d\mid n}\mu(d)\,a_{n/d}(u^d),
\qquad
P(z,u)=\sum_{m\ge 1}\frac{\mu(m)}{m}\log\zeta(z^m,u^m),
\]
有限 atomic surgery 的 primitive 剥离公式
\[
\zeta(z,u)=\prod_{j=1}^{J}(1-u^{E_j}z^{\ell_j})^{-m_j}\,\zeta_{\mathrm{core}}(z,u),
\qquad
P(z,u)=P_{\mathrm{core}}(z,u)+\sum_{j=1}^{J}m_j u^{E_j}z^{\ell_j},
\]
以及真实输入 \(40\) 状态核上的特例
\[
\Delta(z,u)=(1-uz^2)F(z,u),
\qquad
P(z,u)=uz^2+P_{\mathrm{core}}(z,u)
\]
都已建立之后，还可把 Ramanujan 和诱导的长度筛接入同一接口，得到一条同时分离 primitive 几何与 ghost 几何的终端命题链。其核心并不在于再一次重写 Witt 反演，而在于揭示同一 Big-Witt 原子在两层 Ramanujan 层析中的严格异相行为：primitive 层只保留长度整除骨架，ghost 层则把同一原子沿其 harmonic semigroup 展开为几何射线。

对 primitive 生成函数
\[
P(z,u)=\sum_{n\ge 1}p_n(u)z^n
\]
定义 Ramanujan primitive 影
\[
\mathfrak P_q(z,u):=\sum_{n\ge 1}c_q(n)\,p_n(u)z^n
\qquad (q\ge 1),
\]
并定义 divisor-sieve primitive 层
\[
\Pi_r(z,u):=\frac1r\sum_{q\mid r}\mathfrak P_q(z,u)
\qquad (r\ge 1).
\]

\begin{theorem}[primitive 轨道谱的 Ramanujan--divisor 精确层析]\label{thm:conclusion-primitive-ramanujan-divisor-exact-tomography}
\leanverified{paper\_conclusion\_primitive\_ramanujan\_divisor\_exact\_tomography}
对任意 \(r\ge 1\)，成立
\[
\Pi_r(z,u)=\sum_{k\ge 1}p_{rk}(u)z^{rk}.
\]
并且对任意 \(n\ge 1\)，有精确反演
\[
p_n(u)z^n=\sum_{t\ge 1}\mu(t)\,\Pi_{nt}(z,u).
\]
换言之，\(\Pi_r\) 精确切出全部长度被 \(r\) 整除的 primitive 层，而整条 primitive 长度谱可由 \((\Pi_r)_{r\ge 1}\) 唯一恢复。
\end{theorem}
\begin{proof}
由定义，
\[
\Pi_r(z,u)
=
\frac1r\sum_{q\mid r}\sum_{n\ge 1}c_q(n)\,p_n(u)z^n
=
\sum_{n\ge 1}\left(\frac1r\sum_{q\mid r}c_q(n)\right)p_n(u)z^n.
\]
经典 Ramanujan--divisor 恒等式给出
\[
\sum_{q\mid r}c_q(n)=r\,\mathbf 1_{\{r\mid n\}},
\]
从而
\[
\Pi_r(z,u)=\sum_{r\mid n}p_n(u)z^n=\sum_{k\ge 1}p_{rk}(u)z^{rk}.
\]

再看反演公式：
\[
\sum_{t\ge 1}\mu(t)\,\Pi_{nt}(z,u)
=
\sum_{t\ge 1}\mu(t)\sum_{k\ge 1}p_{ntk}(u)z^{ntk}
=
\sum_{m\ge 1}\left(\sum_{t\mid m}\mu(t)\right)p_{nm}(u)z^{nm}.
\]
由于
\[
\sum_{t\mid m}\mu(t)=\mathbf 1_{\{m=1\}},
\]
右边恰化为 \(p_n(u)z^n\)。证毕。
\end{proof}

\begin{theorem}[primitive 轨道谱的 Ramanujan--Dirichlet 阴影]\label{thm:conclusion-primitive-ramanujan-dirichlet-shadow}
\leanverified{paper\_conclusion\_primitive\_ramanujan\_dirichlet\_shadow}
当 \(\Re s>1\) 时，按 \(z\)-系数逐项定义 Dirichlet 影
\[
\mathfrak D_P(s;z,u):=\sum_{q\ge 1}\frac{\mathfrak P_q(z,u)}{q^s}.
\]
则有
\[
\mathfrak D_P(s;z,u)
=
\frac{1}{\zeta(s)}
\sum_{n\ge 1}\sigma_{1-s}(n)\,p_n(u)z^n.
\]
进一步，代入加权 Witt--M\"obius 分解可得
\[
\mathfrak D_P(s;z,u)
=
\frac{1}{\zeta(s)}
\sum_{n\ge 1}
\frac{\sigma_{1-s}(n)}{n}
\left(\sum_{d\mid n}\mu(d)\,a_{n/d}(u^d)\right)z^n.
\]
\end{theorem}
\begin{proof}
当 \(\Re s>1\) 时，每个 \(z^n\) 系数上的 Dirichlet 级数绝对收敛，因此可逐项交换求和次序：
\[
\mathfrak D_P(s;z,u)
=
\sum_{n\ge 1}p_n(u)z^n\sum_{q\ge 1}\frac{c_q(n)}{q^s}.
\]
再用经典 Ramanujan--Dirichlet 恒等式
\[
\sum_{q\ge 1}\frac{c_q(n)}{q^s}
=
\frac{\sigma_{1-s}(n)}{\zeta(s)}
\qquad (\Re s>1),
\]
便得第一式。再把
\[
p_n(u)=\frac1n\sum_{d\mid n}\mu(d)\,a_{n/d}(u^d)
\]
逐项代回，即得到第二式。证毕。
\end{proof}

\begin{theorem}[有限 atomic surgery 的 primitive Ramanujan 可见性律]\label{thm:conclusion-atomic-primitive-ramanujan-visibility-law}
设
\[
\zeta(z,u)=\prod_{j=1}^{J}(1-u^{E_j}z^{\ell_j})^{-m_j}\,\zeta_{\mathrm{core}}(z,u),
\qquad
m_j\in\ZZ_{>0},\ \ell_j\ge 1,\ E_j\ge 0,
\]
并记
\[
P(z,u)=P_{\mathrm{core}}(z,u)+\sum_{j=1}^{J}m_j u^{E_j}z^{\ell_j}.
\]
若 \(\mathfrak P_q^{\mathrm{core}}\)、\(\Pi_r^{\mathrm{core}}\) 表示对 \(P_{\mathrm{core}}\) 同样构造的 Ramanujan 影与 divisor-sieve 层，则对任意 \(q,r\ge 1\) 成立
\[
\mathfrak P_q(z,u)
=
\mathfrak P_q^{\mathrm{core}}(z,u)
+
\sum_{j=1}^{J}m_j c_q(\ell_j)\,u^{E_j}z^{\ell_j},
\]
\[
\Pi_r(z,u)
=
\Pi_r^{\mathrm{core}}(z,u)
+
\sum_{\substack{1\le j\le J\\ r\mid \ell_j}}
m_j u^{E_j}z^{\ell_j}.
\]
因此 primitive Ramanujan tomography 对有限 Big-Witt 原子只保留 divisibility poset 上的长度支撑骨架；它记录哪些 divisor layers 被哪些 primitive 原子直接击中，但不记录这些原子在 ghost 层中的谐波重复。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-finite-euler-primitive-surgery-additivity}，
\[
P(z,u)-P_{\mathrm{core}}(z,u)=\sum_{j=1}^{J}m_j u^{E_j}z^{\ell_j}.
\]
逐项乘以 \(c_q(n)\) 并求和，即得
\[
\mathfrak P_q(z,u)-\mathfrak P_q^{\mathrm{core}}(z,u)
=
\sum_{j=1}^{J}m_j c_q(\ell_j)\,u^{E_j}z^{\ell_j}.
\]
再对全部 \(q\mid r\) 作 divisor average，
\[
\Pi_r(z,u)-\Pi_r^{\mathrm{core}}(z,u)
=
\sum_{j=1}^{J}m_j u^{E_j}z^{\ell_j}
\left(\frac1r\sum_{q\mid r}c_q(\ell_j)\right).
\]
再用
\[
\frac1r\sum_{q\mid r}c_q(\ell_j)=\mathbf 1_{\{r\mid \ell_j\}},
\]
即得第二式。证毕。
\end{proof}

\begin{corollary}[真实输入 \texorpdfstring{$40$}{40} 状态核的 universal \texorpdfstring{$2$}{2}-cycle 原子只在前两级 primitive divisor 层可见]\label{cor:conclusion-realinput40-primitive-ramanujan-first-two-layers}
\leanverified{paper\_conclusion\_realinput40\_primitive\_ramanujan\_first\_two\_layers}
对真实输入 \(40\) 状态核，命题 \ref{prop:atomic-2-orbit-split-logM} 给出
\[
\Delta(z,u)=(1-uz^2)F(z,u),
\qquad
P(z,u)=uz^2+P_{\mathrm{core}}(z,u).
\]
因此对任意 \(q\ge 1\)，
\[
\mathfrak P_q(z,u)
=
\mathfrak P_q^{\mathrm{core}}(z,u)+c_q(2)\,uz^2,
\]
而对任意 \(r\ge 1\)，
\[
\Pi_r(z,u)
=
\Pi_r^{\mathrm{core}}(z,u)+uz^2\,\mathbf 1_{\{r\mid 2\}}.
\]
亦即
\[
\Pi_1(z,u)=\Pi_1^{\mathrm{core}}(z,u)+uz^2,
\qquad
\Pi_2(z,u)=\Pi_2^{\mathrm{core}}(z,u)+uz^2,
\qquad
\Pi_r(z,u)=\Pi_r^{\mathrm{core}}(z,u)\quad (r\ge 3).
\]
因此该长度 \(2\) 原子在 primitive 层只触及 divisor levels \(1\) 与 \(2\)；全部 \(r\ge 3\) 的 primitive multiple-length layers 都已是纯 core 现象。
\end{corollary}
\begin{proof}
在定理 \ref{thm:conclusion-atomic-primitive-ramanujan-visibility-law} 中取
\[
J=1,
\qquad
(m_1,\ell_1,E_1)=(1,2,1),
\]
即可得到全部公式。证毕。
\end{proof}

对 periodic/ghost 坐标
\[
\log\zeta(z,u)=\sum_{n\ge 1}\frac{a_n(u)}{n}z^n
\]
定义 ghost Ramanujan 影
\[
\mathfrak A_q(z,u):=\sum_{n\ge 1}c_q(n)\,a_n(u)z^n
\qquad (q\ge 1),
\]
并定义其 divisor-sieve 层
\[
\Gamma_r(z,u):=\frac1r\sum_{q\mid r}\mathfrak A_q(z,u)
=
\sum_{k\ge 1}a_{rk}(u)z^{rk}.
\]

\begin{theorem}[有限 atomic surgery 的 ghost Ramanujan 射线律]\label{thm:conclusion-atomic-ghost-ramanujan-ray-law}
沿用定理 \ref{thm:conclusion-atomic-primitive-ramanujan-visibility-law} 的设定。记 \(\mathfrak A_q^{\mathrm{core}}\)、\(\Gamma_r^{\mathrm{core}}\) 为 core 部分对应的 ghost Ramanujan 影与 divisor-sieve 层。则每个第 \(j\) 个原子在 ghost 坐标上的贡献满足
\[
a_n^{(j)}(u)=m_j\ell_j\,u^{E_j n/\ell_j}\mathbf 1_{\{\ell_j\mid n\}}.
\]
若对给定 \(r\ge 1\) 记
\[
g_{j,r}:=(r,\ell_j),
\qquad
r_{j,r}:=\frac{r}{g_{j,r}},
\]
则有精确几何射线公式
\[
\Gamma_r(z,u)
=
\Gamma_r^{\mathrm{core}}(z,u)
+
\sum_{j=1}^{J}
m_j\ell_j\,
\frac{u^{E_j r_{j,r}}z^{\ell_j r_{j,r}}}{1-u^{E_j r_{j,r}}z^{\ell_j r_{j,r}}}.
\]
因此，ghost Ramanujan tomography 不再只记录长度骨架，而是把每个 atomic length 沿其 harmonic semigroup 延展为一条显式几何射线。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-atomic-surgery-ghost-abel-shift-general}，每个原子贡献满足
\[
a_n^{(j)}(u)=m_j\ell_j\,u^{E_j n/\ell_j}\mathbf 1_{\{\ell_j\mid n\}}.
\]
于是
\[
\Gamma_r(z,u)-\Gamma_r^{\mathrm{core}}(z,u)
=
\sum_{j=1}^{J}\sum_{k\ge 1}a_{rk}^{(j)}(u)z^{rk}
=
\sum_{j=1}^{J}m_j\ell_j
\sum_{\substack{k\ge 1\\ r\mid \ell_j k}}
u^{E_j k}z^{\ell_j k}.
\]
写 \(\ell_j=g_{j,r}\ell_{j,r}^{\prime}\)、\(r=g_{j,r}r_{j,r}\) 且 \((\ell_{j,r}^{\prime},r_{j,r})=1\)。则
\[
r\mid \ell_j k
\iff
r_{j,r}\mid k.
\]
令 \(k=r_{j,r}m\) 后得到
\[
\sum_{\substack{k\ge 1\\ r\mid \ell_j k}}
u^{E_j k}z^{\ell_j k}
=
\sum_{m\ge 1}
\bigl(u^{E_j r_{j,r}}z^{\ell_j r_{j,r}}\bigr)^m
=
\frac{u^{E_j r_{j,r}}z^{\ell_j r_{j,r}}}{1-u^{E_j r_{j,r}}z^{\ell_j r_{j,r}}}.
\]
对 \(j\) 求和即得所示公式。证毕。
\end{proof}

\begin{corollary}[真实输入 \texorpdfstring{$40$}{40} 状态核的 ghost Ramanujan 奇偶分裂]\label{cor:conclusion-realinput40-ghost-ramanujan-parity-split}
对真实输入 \(40\) 状态核，注 \ref{rem:real-input-40-single-euler} 给出唯一原子因子 \((1-uz^2)^{-1}\)，故对任意 \(r\ge 1\) 有
\[
\Gamma_r(z,u)-\Gamma_r^{\mathrm{core}}(z,u)
=
\begin{cases}
\dfrac{2u^{r/2}z^r}{1-u^{r/2}z^r},& 2\mid r,\\[10pt]
\dfrac{2u^r z^{2r}}{1-u^r z^{2r}},& 2\nmid r.
\end{cases}
\]
因此同一个 universal \(2\)-cycle 原子在 primitive 层只影响前两级 divisor layers，而在 ghost 层却渗入全部 divisor layers；区别仅在于偶层从 \(z^r\) 起步，奇层从 \(z^{2r}\) 起步。
\end{corollary}
\begin{proof}
在定理 \ref{thm:conclusion-atomic-ghost-ramanujan-ray-law} 中取
\[
(m_1,\ell_1,E_1)=(1,2,1).
\]
若 \(2\mid r\)，则 \((r,2)=2\)，从而 \(r_{1,r}=r/2\)，故
\[
\Gamma_r(z,u)-\Gamma_r^{\mathrm{core}}(z,u)
=
2\frac{u^{r/2}z^r}{1-u^{r/2}z^r}.
\]
若 \(2\nmid r\)，则 \((r,2)=1\)，故 \(r_{1,r}=r\)，于是
\[
\Gamma_r(z,u)-\Gamma_r^{\mathrm{core}}(z,u)
=
2\frac{u^r z^{2r}}{1-u^r z^{2r}}.
\]
证毕。
\end{proof}

\begin{conjecture}[有限 Big-Witt 原子的 primitive/ghost 双层 Ramanujan 层析完备性]\label{conj:conclusion-bigwitt-primitive-ghost-ramanujan-tomography}
设
\[
\zeta(z,u)=\prod_{j=1}^{J}(1-u^{E_j}z^{\ell_j})^{-m_j}\,\zeta_{\mathrm{core}}(z,u)
\]
是一族允许有限 atomic surgery 的加权核。则 primitive divisor tower
\[
(\Pi_r(z,u))_{r\ge 1}
\]
应当唯一决定原子多重集 \(\{(\ell_j,E_j,m_j)\}_{j=1}^{J}\) 在 divisibility poset 上的支撑骨架；ghost divisor tower
\[
(\Gamma_r(z,u))_{r\ge 1}
\]
应当唯一决定每个原子所生成的完整 harmonic semigroup；并且二者合起来构成有限 Big-Witt atomic surgery 的最小完备层析坐标。

更具体地，若两组有限 atomic surgeries 具有相同的 primitive Ramanujan divisor tower 与相同的 ghost Ramanujan divisor tower，则其原子多重集应当彼此相同；反之，仅有 primitive 塔会丢失谐波尾迹，仅有 ghost 塔则不直接显化 primitive 有限支撑几何，二者不存在进一步可去的冗余。
\end{conjecture}

以上命题把此前彼此分散的三项事实压缩到同一条闭合链上：primitive Witt 分解给出长度坐标，有限 atomic surgery 说明 primitive 层只出现有限支撑缺陷，而 ghost 剥离公式又把同一缺陷沿倍长度半群展开为显式几何尾迹。由此得到的结构分离是刚性的：Big-Witt 原子在 primitive 几何中表现为有限支撑对象，在 ghost 几何中则表现为无穷谐波对象；Ramanujan divisor tomography 恰把这两类语义放在同一坐标系内并列显影。
