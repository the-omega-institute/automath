\subsection{有限秩阿贝尔系统的 exact-additive 唯一性、derived torsion camouflage 与相位压缩强逆}\label{subsec:conclusion-finite-rank-derived-phase-strongconverse}
沿用小节 \ref{subsec:conclusion-cdim-finite-quotient-k0-precision-rigidity} 与小节 \ref{subsec:conclusion-rational-cdim-solenoid-precision-ledger-rigidity} 的记号。前述结论已经分别给出：圆维在有限生成阿贝尔群上的短正合可加唯一性、\(\mathrm{Ab}_{\mathrm{fr}}\) 上的有理秩延拓、最小外置账本成本等于 kernel 的圆维、以及相位压缩导致的幂律精度下界。把这些对象进一步并置后，还可把 finite-rank abelian 侧的唯一可加标尺、derived \(K_0\) 侧的 torsion camouflage、以及 torus phase-compression 侧的强逆壁垒压缩为同一条 rank-driven 终端链：一切 exact-additive 标尺最终都塌缩到 rational rank；有限缺陷在结构层与 derived Euler 层同时透明；多段外置账本代价沿 kernel-rank 严格望远镜相加；而在几何侧，同一 rank 不变量又被迫重显为不可逾越的精度阈值与无穷近零 annihilator 序列。

\begin{theorem}[有限秩 exact-additive 标尺的唯一延拓与有限缺陷透明性]\label{thm:conclusion-finite-rank-exact-additive-unique-extension}
\leanverified{paper\_conclusion\_finite\_rank\_exact\_additive\_unique\_extension}
设
\[
I:\mathrm{Ab}_{\mathrm{fr}}\to \RR_{\ge 0}
\]
满足：
\begin{enumerate}
  \item 同构不变；
  \item 对任意短正合列
  \[
  0\to A\to B\to C\to 0
  \]
  有
  \[
  I(B)=I(A)+I(C);
  \]
  \item 归一化 \(I(\ZZ)=1\)；
  \item 对一切挠群 \(T\) 有 \(I(T)=0\)。
\end{enumerate}
则对任意有限秩离散阿贝尔群 \(G\)，必有
\[
I(G)=\cdim_{\mathrm{fr}}(G)=\dim_{\QQ}(G\otimes_{\ZZ}\QQ).
\]
特别地，若 \(\phi:G\to H\) 为有限秩阿贝尔群同态，且 \(\ker\phi\) 与 \(\operatorname{coker}\phi\) 都为挠群，则
\[
\cdim_{\mathrm{fr}}(G)=\cdim_{\mathrm{fr}}(H).
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:conclusion-cdimq-unique-exact-extension}，\(\cdim_{\QQ}(G):=\dim_{\QQ}(G\otimes_{\ZZ}\QQ)\) 是 \(\mathrm{Ab}_{\mathrm{fr}}\) 上满足挠群消失、短正合可加与 \(I(\ZZ)=1\) 的唯一归一化标尺。又同一命题与定理 \ref{thm:conclusion-cdim-finite-rank-extension} 给出
\[
\cdim_{\QQ}(G)=\cdim_{\mathrm{fr}}(G).
\]
故 \(I=\cdim_{\mathrm{fr}}\)。

若 \(\phi:G\to H\) 的 kernel 与 cokernel 都为挠群，则短正合列
\[
0\to \ker\phi\to G\to \operatorname{im}\phi\to 0,
\qquad
0\to \operatorname{im}\phi\to H\to \operatorname{coker}\phi\to 0
\]
中两端挠群的圆维皆为 \(0\)。由短正合可加性得
\[
\cdim_{\mathrm{fr}}(G)=\cdim_{\mathrm{fr}}(\operatorname{im}\phi)=\cdim_{\mathrm{fr}}(H).
\]
证毕。
\end{proof}

\begin{theorem}[可逆外置的 kernel 链式加法律]\label{thm:conclusion-finite-rank-kernel-chain-rule}
设
\[
G\xrightarrow{f}H\xrightarrow{g}K
\]
为有限秩离散阿贝尔群同态，并记
\[
\bar g:=g|_{\operatorname{im}f}:\operatorname{im}f\to K.
\]
则存在短正合列
\[
0\to \ker f\to \ker(g\circ f)\xrightarrow{\,f\,}\ker \bar g\to 0,
\]
从而
\[
\cdim_{\mathrm{fr}}(\ker(g\circ f))
=
\cdim_{\mathrm{fr}}(\ker f)+\cdim_{\mathrm{fr}}(\ker \bar g).
\]
\end{theorem}
\begin{proof}
映射
\[
\pi:\ker(g\circ f)\to \ker\bar g,
\qquad
\pi(x):=f(x)
\]
良定义，因为 \(x\in\ker(g\circ f)\) 蕴含 \(g(f(x))=0\) 且 \(f(x)\in\operatorname{im}f\)。其核为
\[
\ker\pi=\{x\in\ker(g\circ f):f(x)=0\}=\ker f.
\]
对任意 \(y\in\ker\bar g\subseteq \operatorname{im}f\)，取 \(x\in G\) 使 \(f(x)=y\)。由 \(g(y)=0\) 得 \(g(f(x))=0\)，故 \(x\in\ker(g\circ f)\) 且 \(\pi(x)=y\)。因此该列短正合。

再由定理 \ref{thm:conclusion-finite-rank-exact-additive-unique-extension} 的短正合可加性，得到
\[
\cdim_{\mathrm{fr}}(\ker(g\circ f))
=
\cdim_{\mathrm{fr}}(\ker f)+\cdim_{\mathrm{fr}}(\ker \bar g).
\]
证毕。
\leanverified{paper\_conclusion\_finite\_rank\_kernel\_chain\_rule}
\end{proof}

\begin{corollary}[多段相位压缩的账本望远镜律]\label{cor:conclusion-finite-rank-ledger-telescoping-law}
\leanverified{paper\_conclusion\_finite\_rank\_ledger\_telescoping\_law}
设
\[
G_0\xrightarrow{f_1}G_1\xrightarrow{f_2}\cdots\xrightarrow{f_m}G_m
\]
为有限秩离散阿贝尔群链，并记整体复合
\[
F:=f_m\cdots f_1:G_0\to G_m.
\]
对每个 \(j\ge 1\)，令
\[
\bar f_j:=f_j|_{\operatorname{im}(f_{j-1}\cdots f_1)}.
\]
则
\[
\cdim_{\mathrm{fr}}(\ker F)
=
\sum_{j=1}^{m}\cdim_{\mathrm{fr}}(\ker \bar f_j).
\]
特别地，按定理 \ref{thm:conclusion-cdim-kernel-register-lower-bound}，整体复合同态的最小可逆外置账本圆维恰等于右端的逐段和。
\end{corollary}
\begin{proof}
对 \(m=1\) 平凡。设结论对 \(m-1\) 段成立。记
\[
F=f_m\circ F_{m-1},
\qquad
F_{m-1}:=f_{m-1}\cdots f_1.
\]
由定理 \ref{thm:conclusion-finite-rank-kernel-chain-rule}，
\[
\cdim_{\mathrm{fr}}(\ker F)
=
\cdim_{\mathrm{fr}}(\ker F_{m-1})
+\cdim_{\mathrm{fr}}\bigl(\ker(f_m|_{\operatorname{im}F_{m-1}})\bigr).
\]
对第一项应用归纳假设，即得
\[
\cdim_{\mathrm{fr}}(\ker F)
=
\sum_{j=1}^{m-1}\cdim_{\mathrm{fr}}(\ker \bar f_j)
+\cdim_{\mathrm{fr}}(\ker \bar f_m).
\]
证毕。
\end{proof}

\begin{theorem}[derived torsion camouflage 与 Euler 圆维不变量]\label{thm:conclusion-derived-torsion-camouflage-euler-cdim}
\leanverified{paper\_conclusion\_derived\_torsion\_camouflage\_euler\_cdim}
设
\[
C^\bullet,D^\bullet\in D^b(\mathrm{Ab}_{\mathrm{fg}})
\]
满足对每个 \(i\) 都有有理同构
\[
H^i(C^\bullet)\otimes_{\ZZ}\QQ
\cong
H^i(D^\bullet)\otimes_{\ZZ}\QQ.
\]
则
\[
\chi_{\cdim}(C^\bullet)=\chi_{\cdim}(D^\bullet),
\qquad
\chi_{\cdim}(E^\bullet):=\sum_i(-1)^i\cdim(H^i(E^\bullet)).
\]
特别地，若 \(T^\bullet\) 的全部上同调群均为挠群，则
\[
\chi_{\cdim}(T^\bullet)=0,
\qquad
\chi_{\cdim}(C^\bullet\oplus T^\bullet)=\chi_{\cdim}(C^\bullet).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-cdim-rank-and-short-exact-additivity}，对每个有限生成阿贝尔群 \(G\) 都有
\[
\cdim(G)=\dim_{\QQ}(G\otimes_{\ZZ}\QQ).
\]
因此 \(\chi_{\cdim}(E^\bullet)\) 只依赖于各阶上同调群的有理秩。若 \(C^\bullet\) 与 \(D^\bullet\) 在每个次数上的有理上同调同构，则这些有理秩逐项相等，于是交错和相等。

若 \(T^\bullet\) 的上同调全为挠群，则各阶圆维皆为 \(0\)，故
\[
\chi_{\cdim}(T^\bullet)=0.
\]
再由直和对上同调与交错和都可加，得到
\[
\chi_{\cdim}(C^\bullet\oplus T^\bullet)=\chi_{\cdim}(C^\bullet).
\]
证毕。
\end{proof}

\begin{corollary}[derived quasi-isogeny 的 Euler 圆维不变性]\label{cor:conclusion-derived-quasi-isogeny-euler-cdim-invariance}
\leanverified{paper\_conclusion\_derived\_quasi\_isogeny\_euler\_cdim\_invariance}
设
\[
u:C^\bullet\to D^\bullet
\]
为 \(D^b(\mathrm{Ab}_{\mathrm{fg}})\) 中的态射。若对每个 \(i\)，诱导映射
\[
H^i(u):H^i(C^\bullet)\to H^i(D^\bullet)
\]
都具有挠 kernel 与挠 cokernel，则
\[
\chi_{\cdim}(C^\bullet)=\chi_{\cdim}(D^\bullet).
\]
\end{corollary}
\begin{proof}
每个 \(H^i(u)\) 在张量 \(\QQ\) 后都成为同构，因此可直接应用定理 \ref{thm:conclusion-derived-torsion-camouflage-euler-cdim}。证毕。
\end{proof}

\begin{theorem}[相位压缩的强逆阈值律]\label{thm:conclusion-phase-compression-strong-converse-threshold}
\leanverified{paper\_conclusion\_phase\_compression\_strong\_converse\_threshold}
设
\[
\varphi:\ZZ^r\to \TT^d
\]
为群同态，并对每个 \(N\ge 1\) 记
\[
B_N:=\{0,1,\dots,N\}^r.
\]
若存在唯一反演协议，能在 \(B_N\) 上从 \(\varphi(x)\) 的 \(\delta_N\)-精度观测中恢复 \(x\)，则必有
\[
\delta_N\lesssim (N+1)^{-r/d}.
\]
等价地，若以
\[
\delta_N=2^{-b_N}
\]
定义精度比特数，则必有
\[
b_N\ge \frac{r}{d}\log_2(N+1)-O(1).
\]
特别地，若存在 \(\varepsilon>0\) 与无穷子列 \(N_\nu\to\infty\) 使
\[
b_{N_\nu}\le \left(\frac{r}{d}-\varepsilon\right)\log_2(N_\nu+1)+O(1),
\]
则该协议不可能在该子列上保持唯一反演。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-cdim-phase-compression-power-law}，对每个 \(N\) 都存在不同的 \(u,v\in B_N\) 使
\[
|\varphi(u)-\varphi(v)|_{\TT^d}\le 2(N+1)^{-r/d}.
\]
若测量分辨率 \(\delta_N\) 大于该碰撞尺度的一个固定倍数，则 \(u\) 与 \(v\) 在观测端不可区分，唯一反演失败。故必要有
\[
\delta_N\lesssim (N+1)^{-r/d}.
\]
取 \(\delta_N=2^{-b_N}\) 便得到
\[
b_N\ge \frac{r}{d}\log_2(N+1)-O(1),
\]
这与推论 \ref{cor:conclusion-cdim-phase-compression-bit-lower-bound} 一致。

若存在 \(\varepsilon>0\) 与无穷子列 \(N_\nu\) 使
\[
b_{N_\nu}\le \left(\frac{r}{d}-\varepsilon\right)\log_2(N_\nu+1)+O(1),
\]
则
\[
\delta_{N_\nu}\ge c(N_\nu+1)^{-r/d+\varepsilon}
\]
对某个常数 \(c>0\) 成立，最终严格大于碰撞尺度 \(2(N_\nu+1)^{-r/d}\)。于是唯一反演在该子列上必然失败。证毕。
\end{proof}

\begin{theorem}[相位压缩诱导的无穷近零 annihilator 序列]\label{thm:conclusion-phase-compression-infinite-approximate-annihilators}
\leanverified{paper\_conclusion\_phase\_compression\_infinite\_approximate\_annihilators}
对任意群同态
\[
\varphi:\ZZ^r\to \TT^d,
\]
存在一列非零向量
\[
k_N\in \ZZ^r\setminus\{0\},
\qquad
\|k_N\|_\infty\le N,
\]
使得对每个 \(N\ge 1\) 都有
\[
|\varphi(k_N)|_{\TT^d}\le 2(N+1)^{-r/d}.
\]
特别地，存在无穷序列 \(k_\nu\neq 0\) 满足
\[
\|k_\nu\|_\infty\to\infty
\]
且
\[
|\varphi(k_\nu)|_{\TT^d}\,\|k_\nu\|_\infty^{\,r/d}\le 2(1+o(1)).
\]
因此，任何 \(d<r\) 的相位压缩都不可避免地携带一族多项式级别趋零的 approximate annihilators。
\end{theorem}
\begin{proof}
前半部分正是定理 \ref{thm:conclusion-cdim-phase-compression-power-law} 的等价表述：对每个 \(N\) 存在不同的 \(u_N,v_N\in B_N\)，令
\[
k_N:=u_N-v_N\neq 0,
\]
便有
\[
\|k_N\|_\infty\le N,
\qquad
|\varphi(k_N)|_{\TT^d}\le 2(N+1)^{-r/d}.
\]

若 \(\ker\varphi\neq 0\)，任选非零 \(k\in\ker\varphi\)，再取 \(k_\nu:=\nu k\)。则
\[
\|k_\nu\|_\infty\to\infty,
\qquad
|\varphi(k_\nu)|_{\TT^d}=0,
\]
结论显然成立。

若 \(\ker\varphi=0\)，则 \(\{k_N\}\) 不可能只落在某个有界集合中，否则会有某个非零 \(k\) 在无穷多个 \(N\) 上重复出现，从而
\[
|\varphi(k)|_{\TT^d}\le 2(N+1)^{-r/d}\to 0,
\]
迫使 \(\varphi(k)=0\)，与 \(\ker\varphi=0\) 矛盾。故可取一子列 \(k_\nu\) 使 \(\|k_\nu\|_\infty\to\infty\)。

对该子列，由 \(\|k_\nu\|_\infty\le N_\nu\) 与前半部分估计，
\[
|\varphi(k_\nu)|_{\TT^d}\,\|k_\nu\|_\infty^{\,r/d}
\le
2\Bigl(\frac{\|k_\nu\|_\infty}{N_\nu+1}\Bigr)^{r/d}
\le 2.
\]
在取趋于无穷的子列并重新编号后，可将右端记为 \(2(1+o(1))\) 级控制。证毕。
\end{proof}

\paragraph{统一读法}
至此，finite-rank abelian 侧与 torus phase-compression 侧可被压缩为同一条 rank-driven 结构链：一切 exact-additive 标尺在 \(\mathrm{Ab}_{\mathrm{fr}}\) 上都唯一塌缩为 \(\cdim_{\mathrm{fr}}=\operatorname{rank}_{\QQ}\)，因此有限缺陷在结构层完全透明；kernel 的圆维又沿复合同态满足严格链式加法律，从而多段外置账本成本形成精确望远镜和。进入 derived 层后，同一 rank 不变量继续支配 Euler 圆维，使一切 torsion camouflage 与 quasi-isogeny 都对 \(\chi_{\cdim}\) 不可见。相对地，在几何侧，当 \(\ZZ^r\) 被压入 \(\TT^d\) 时，这个同一 rank 比值 \(r/d\) 又重新显影为不可跨越的幂律精度壁垒与无穷近零 annihilator 序列；而对有限 \(b\)-位时间窗口，若仍要求统一可逆，则任何低于满秩的可见相位压缩都会把剩余信息指数性地推回离散账本。换言之，代数侧的 torsion 失语、几何侧的强逆精度壁垒与时间窗口上的相位饱和律，并不是三种无关现象，而是同一个 rank 不变量在 exact category、derived category 与 phase-compression geometry 中的三重投影。
