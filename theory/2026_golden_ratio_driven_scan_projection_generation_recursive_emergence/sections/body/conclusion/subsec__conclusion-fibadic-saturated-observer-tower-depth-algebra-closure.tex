\subsection{饱和有限观察者塔、精确深度新空间与有限层观测代数的闭链}\label{subsec:conclusion-fibadic-saturated-observer-tower-depth-algebra-closure}
沿用小节 \ref{subsec:conclusion-fibadic-new-harmonics-conductor-cyclotomic}、
\ref{subsec:conclusion-mixed-hiddenstate-finite-abelian-observability}、
\ref{subsec:conclusion-fibadic-kernel-semisimplicity-hiddenstate-truncation}、
\ref{subsec:conclusion-fibadic-packet-primitive-spectrum-cyclic-threshold}
与
\ref{subsec:conclusion-fibadic-finite-layer-gcd-reduced-dominantdepth-conductor-inversion}
中的记号。此前结论已经分别给出：精确深度的导体饱和 \(F_d\)、混合隐藏态在固定导体下的最小有限受体、精确深度新空间 \(\mathcal N_d\) 与 cyclotomic 包 \(\Pi_d\) 的对应、有限层 gcd-代数的 primitive 幂等分解，以及 ordinary conductor-to-depth 映射的完全反演。把这些链条并置之后，还可把有限阿贝尔观察者、精确深度字符块与有限层联合观测代数进一步压缩为一条闭合结构。

\begin{proposition}[精确深度饱和层的共终端性]\label{prop:conclusion-fibadic-saturated-observer-cofinality}
对任意正整数 \(N\)，记
\[
d_N:=\operatorname{lcm}_{p^e\parallel N} z(p^e),
\]
并定义
\[
G_d:=(\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ),
\qquad
\mathcal H_{S,N}:=(\ZZ/2\ZZ)^\beta\oplus \ZZ/(N\ZZ).
\]
则存在自然满射
\[
\pi_{d_N,N}:G_{d_N}\twoheadrightarrow \mathcal H_{S,N},
\]
其中 \(\pi_{d_N,N}\) 在 \((\ZZ/2\ZZ)^\beta\) 上为恒等映射，在地址部分为模 \(N\) 约化。并且 \(d_N\) 是全部使得同类自然满射
\[
\pi_{e,N}:G_e\twoheadrightarrow \mathcal H_{S,N}
\]
存在的深度 \(e\) 中唯一最小者。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-exact-depth-conductor-saturation}，
\[
F_{d_N}
=
\operatorname{lcm}\{M\ge 1:\ d_M=d_N\}.
\]
由于 \(N\) 本身满足 \(d_N=d_N\)，故必有
\[
N\mid F_{d_N}.
\]
因此模 \(N\) 约化给出自然商映射
\[
\ZZ/(F_{d_N}\ZZ)\twoheadrightarrow \ZZ/(N\ZZ),
\]
再与 \((\ZZ/2\ZZ)^\beta\) 上的恒等映射直和，即得 \(\pi_{d_N,N}\)。

若另有某个 \(e\ge 1\) 使上述同类自然满射
\[
\pi_{e,N}:G_e\twoheadrightarrow \mathcal H_{S,N}
\]
则必有 \(N\mid F_e\)。由定理
\ref{thm:conclusion-fibadic-minimal-resolution-for-prime-register-package}，
\[
N\mid F_e
\quad\Longleftrightarrow\quad
d_N\mid e.
\]
于是 \(e\ge d_N\)。因此 \(d_N\) 不仅可行，而且是唯一最小深度。证毕。
\end{proof}

\begin{theorem}[饱和有限观察者塔的逆极限重建]\label{thm:conclusion-fibadic-saturated-observer-inverse-limit}
\leanverified{paper\_conclusion\_fibadic\_saturated\_observer\_inverse\_limit}
对 \(d\mid e\) 定义过渡映射
\[
\rho_{e,d}:G_e\twoheadrightarrow G_d
\]
为 \((\ZZ/2\ZZ)^\beta\) 上的恒等映射与模 \(F_d\) 约化之直和。则存在紧阿贝尔群的自然同构
\[
\mathcal H_S
\cong
\varprojlim_{d} G_d.
\]
特别地，任意连续有限阿贝尔观测都已经在观察者塔 \(\{G_d\}_{d\ge 1}\) 上被完全穷尽。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-profinite-collapse-to-zhat}，
\[
\widehat X_F\cong \widehat{\ZZ}\cong \varprojlim_{N\ge 1}\ZZ/(N\ZZ).
\]
而由命题 \ref{prop:conclusion-fibadic-saturated-observer-cofinality}，对每个正整数 \(N\) 都有
\[
N\mid F_{d_N},
\]
故 \(\{F_d\}_{d\ge 1}\) 在全部正整数模数中构成一个共尾子系统。于是逆极限在该共尾子系统上不变，从而
\[
\widehat X_F
\cong
\varprojlim_d \ZZ/(F_d\ZZ).
\]
再注意 \((\ZZ/2\ZZ)^\beta\) 在塔中沿恒等映射保持常值，得到
\[
\varprojlim_d G_d
\cong
(\ZZ/2\ZZ)^\beta\times \varprojlim_d \ZZ/(F_d\ZZ)
\cong
(\ZZ/2\ZZ)^\beta\times \widehat X_F
=
\mathcal H_S.
\]
最后，由定理 \ref{thm:conclusion-global-mixed-hidden-state-no-faithful-finite-abelian-observation}，任意连续有限阿贝尔观测都唯一经由某个 \(\mathcal H_{S,N}\) 因子化；再由命题 \ref{prop:conclusion-fibadic-saturated-observer-cofinality}，该 \(\mathcal H_{S,N}\) 又是某个 \(G_{d_N}\) 的商。故全部连续有限阿贝尔观测已被 \(\{G_d\}\) 穷尽。证毕。
\end{proof}

\begin{corollary}[有限阿贝尔观测的最小深度判据]\label{cor:conclusion-fibadic-saturated-observer-minimal-depth}
设 \(G\) 为有限阿贝尔群，\(\phi:\mathcal H_S\to G\) 为连续同态。则存在唯一的最小深度 \(d_{\min}(\phi)\) 使 \(\phi\) 因子化为
\[
\mathcal H_S\xrightarrow{\ \mathrm{id}\times \pi_{F_{d_{\min}(\phi)}}\ } G_{d_{\min}(\phi)}\longrightarrow G.
\]
并且若 \(\phi\) 在导体 \(N\) 上唯一因子化，即
\[
\mathcal H_S\twoheadrightarrow \mathcal H_{S,N}\longrightarrow G,
\]
则
\[
d_{\min}(\phi)=d_N.
\]
特别地，\(d_{\min}(\phi)\) 仅由地址导体 \(N\) 决定，与碰撞 \((\ZZ/2\ZZ)^\beta\) 部分无关。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-global-mixed-hidden-state-no-faithful-finite-abelian-observation}，\(\phi\) 唯一经由某个 \(\mathcal H_{S,N}\) 因子化。再由命题 \ref{prop:conclusion-fibadic-saturated-observer-cofinality}，\(\mathcal H_{S,N}\) 唯一最小地经由 \(G_{d_N}\) 因子化。因此 \(d_{\min}(\phi)=d_N\)，且其唯一性亦由命题 \ref{prop:conclusion-fibadic-saturated-observer-cofinality} 给出。证毕。
\end{proof}

\begin{theorem}[有限层联合观测代数的 cyclotomic 分块]\label{thm:conclusion-fibadic-joint-observer-algebra-cyclotomic-product}
对任意 \(N\ge 1\)，记活跃深度集
\[
\mathcal D_N^{\mathrm{act}}:=\{d\mid N:\ a_d\neq 0\},
\qquad
\mathcal H_N=\bigoplus_{d\in\mathcal D_N^{\mathrm{act}}}\mathcal N_d.
\]
令 \(\mathscr A_N\) 为 \(\End(\mathcal H_N)\) 中由
\[
U|_{\mathcal H_N}
\qquad\text{与}\qquad
\{E_d|_{\mathcal H_N}:\ d\mid N\}
\]
生成的含单位 \(^*\)-代数。则存在自然代数同构
\[
\mathscr A_N
\cong
\prod_{d\in\mathcal D_N^{\mathrm{act}}}\CC[X]/(\Pi_d).
\]
因此：
\begin{enumerate}
  \item \(\mathscr A_N\) 是有限维交换 Artinian 代数，并且 Jacobson 半单；
  \item 对每个 \(d\in\mathcal D_N^{\mathrm{act}}\)，算子 \(U|_{\mathcal N_d}\) 的最小多项式恰为 \(\Pi_d\)；
  \item primitive spectrum 具有自然描述
  \[
  \operatorname{Prim}(\mathscr A_N)
  \cong
  \{(d,\zeta):\ d\in\mathcal D_N^{\mathrm{act}},\ \Pi_d(\zeta)=0\}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-visible-depth-mobius-primitive-semisimplicity}，
\[
E_d=\sum_{e\mid d}P_e,
\qquad
P_d=\sum_{e\mid d}\mu\!\left(\frac{d}{e}\right)E_e.
\]
故 \(\{E_d|_{\mathcal H_N}\}\) 与 \(\{P_d|_{\mathcal H_N}\}_{d\in\mathcal D_N^{\mathrm{act}}}\) 生成同一含单位代数。于是 \(\mathscr A_N\) 也可视为由两两正交的块投影 \(P_d\) 与 \(U|_{\mathcal H_N}\) 生成。

由定理 \ref{thm:conclusion-fibadic-cyclotomic-packet-decomposition}，每个 \(\mathcal N_d\) 都在 \(U\) 下不变，故块限制映射
\[
\Theta_N:\mathscr A_N\longrightarrow \prod_{d\in\mathcal D_N^{\mathrm{act}}}\End(\mathcal N_d),
\qquad
\Theta_N(A):=(A|_{\mathcal N_d})_{d\in\mathcal D_N^{\mathrm{act}}},
\]
是单射。又因为 \(P_d\) 可从整体代数中抽出第 \(d\) 个分块，而每个分块代数均由 \(U|_{\mathcal N_d}\) 生成，故 \(\Theta_N\) 的像恰为
\[
\prod_{d\in\mathcal D_N^{\mathrm{act}}}\CC[U|_{\mathcal N_d}].
\]

现固定 \(d\in\mathcal D_N^{\mathrm{act}}\)。由定理 \ref{thm:conclusion-fibadic-cyclotomic-packet-decomposition}，
\[
\chi_{U|_{\mathcal N_d}}(X)=\Pi_d(X)
=
\prod_{M:\ d_M=d}\Phi_M(X).
\]
各 cyclotomic 因子 \(\Phi_M\) 在 \(\CC\) 上两两既约且无重根，因此 \(\Pi_d\) 为平方自由多项式。另一方面，\(\mathcal N_d\) 由连续角色张成，故 \(U|_{\mathcal N_d}\) 在角色基上可对角化，且其全部特征值正是 \(\Pi_d\) 的根。于是 \(U|_{\mathcal N_d}\) 的最小多项式等于 \(\Pi_d\)，从而
\[
\CC[U|_{\mathcal N_d}]
\cong
\CC[X]/(\Pi_d).
\]
合并各分块即得
\[
\mathscr A_N
\cong
\prod_{d\in\mathcal D_N^{\mathrm{act}}}\CC[X]/(\Pi_d).
\]

由于每个因子 \(\CC[X]/(\Pi_d)\) 都是由平方自由多项式定义的有限维交换代数，故它们 reduced，从而 Jacobson 半单；有限直积仍保此性质，并自动 Artinian。最后，有限维交换半单复代数的 primitive ideals 正是各分块上根评价的核，因此
\[
\operatorname{Prim}(\mathscr A_N)
\cong
\{(d,\zeta):\ d\in\mathcal D_N^{\mathrm{act}},\ \Pi_d(\zeta)=0\}.
\]
证毕。
\end{proof}

\begin{corollary}[有限层谱数据的块函数演算完备性]\label{cor:conclusion-fibadic-joint-observer-algebra-spectral-completeness}
沿用定理 \ref{thm:conclusion-fibadic-joint-observer-algebra-cyclotomic-product} 的记号。对任意 \(A\in\mathscr A_N\)，记其在同构
\[
\mathscr A_N
\cong
\prod_{d\in\mathcal D_N^{\mathrm{act}}}\CC[X]/(\Pi_d)
\]
下的第 \(d\) 个分量为 \(f_{A,d}(X)\in \CC[X]/(\Pi_d)\)。则 \(A\) 的全部谱、迹、行列式与最小多项式都由族
\[
\{f_{A,d}\}_{d\in\mathcal D_N^{\mathrm{act}}}
\]
唯一决定。更精确地，
\[
\operatorname{Spec}(A)
=
\bigcup_{d\in\mathcal D_N^{\mathrm{act}}}
\{\,f_{A,d}(\zeta):\ \Pi_d(\zeta)=0\,\},
\]
\[
\operatorname{Tr}(A)
=
\sum_{d\in\mathcal D_N^{\mathrm{act}}}\ \sum_{\Pi_d(\zeta)=0} f_{A,d}(\zeta),
\qquad
\det(A)
=
\prod_{d\in\mathcal D_N^{\mathrm{act}}}\ \prod_{\Pi_d(\zeta)=0} f_{A,d}(\zeta).
\]
特别地，对
\[
T_\alpha=\sum_{m\ge 1}\alpha(m)E_m
\]
有
\[
f_{T_\alpha,d}(X)=\widehat\alpha(d)
\]
为常值类，故 \(T_\alpha|_{\mathcal H_N}\) 的全部有限层谱数据仅依赖于 Möbius--zeta 变换 \(\widehat\alpha\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-joint-observer-algebra-cyclotomic-product}，\(A\) 在第 \(d\) 个分块上恰由 \(f_{A,d}(U|_{\mathcal N_d})\) 给出。由于 \(U|_{\mathcal N_d}\) 在角色基上可对角化且其特征值恰为 \(\Pi_d\) 的根，故 \(f_{A,d}(U|_{\mathcal N_d})\) 的特征值恰为 \(f_{A,d}(\zeta)\)（\(\Pi_d(\zeta)=0\)）。将各分块并合，即得谱、迹与行列式公式；最小多项式则由各分块最小多项式的最小公倍式给出，故同样由 \(\{f_{A,d}\}\) 唯一决定。

对 \(T_\alpha\)，由定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization}，
\[
T_\alpha|_{\mathcal N_d}=\widehat\alpha(d)\,\mathrm{id}_{\mathcal N_d},
\]
故其第 \(d\) 分量就是常值类 \(f_{T_\alpha,d}(X)=\widehat\alpha(d)\)。于是全部有限层谱数据只依赖于 \(\widehat\alpha\)。证毕。
\end{proof}

\begin{theorem}[精确深度新空间与最小忠实观察者的字符对偶]\label{thm:conclusion-fibadic-newspace-minimal-observer-duality}
对每个活跃深度 \(d\in\{1\}\cup\{n\ge 3\}\)，记
\[
G_d=(\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ),
\qquad
\widehat G_d^{\,\mathrm{addr}}\cong \widehat{\ZZ/(F_d\ZZ)}.
\]
定义其中的精确深度 \(d\) 字符子空间
\[
\mathscr P_d
:=
\operatorname{span}\{\chi\in \widehat{\ZZ/(F_d\ZZ)}:\ d_{\operatorname{cond}(\chi)}=d\}.
\]
则存在自然酉同构
\[
\mathcal N_d\cong \mathscr P_d.
\]
此外，由 \(\mathcal N_d\) 在 \(\mathcal H_d\) 中生成的含单位 \(^*\)-代数恰为整个 \(\mathcal H_d\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-newspace-orthogonal-decomposition}，\(\mathcal N_d\) 正是全部满足 \(\delta(\psi)=d\) 的连续角色所张成的闭子空间。另一方面，由定理 \ref{thm:conclusion-fibadic-exact-depth-linear-thin-algebraic-saturation}，
\[
\bigcap_{\psi:\ \delta(\psi)=d}\ker\psi
=
F_d\widehat{\ZZ},
\]
故每个此类 \(\psi\) 都经由
\[
\widehat X_F\twoheadrightarrow \ZZ/(F_d\ZZ)
\]
因子化，并且其 ordinary conductor \(N\) 满足 \(d_N=d\)。反过来，\(\widehat{\ZZ/(F_d\ZZ)}\) 中每个满足 \(d_{\operatorname{cond}(\chi)}=d\) 的字符拉回后都落在 \(\mathcal N_d\) 中。于是两边在角色基上完全对应，从而得到自然酉同构
\[
\mathcal N_d\cong \mathscr P_d.
\]

第二条正是定理 \ref{thm:conclusion-fibadic-exact-depth-linear-thin-algebraic-saturation} 的代数饱和结论：尽管 \(\mathcal N_d\subsetneq \mathcal H_d\) 在线性上为真子空间，但其所生成的含单位 \(^*\)-代数仍恰为整个 \(\mathcal H_d\)。证毕。
\end{proof}

\begin{corollary}[最小忠实观察者中的 primitive 比例极限]\label{cor:conclusion-fibadic-minimal-observer-primitive-density}
记
\[
a_d:=\dim \mathcal N_d,
\qquad
|G_d|=2^\beta F_d.
\]
则当 \(d\to\infty\) 时，
\[
\frac{a_d}{F_d}\longrightarrow 1,
\qquad
\frac{\dim \mathcal N_d}{|G_d|}
=
\frac{a_d}{2^\beta F_d}
\longrightarrow
2^{-\beta}.
\]
\end{corollary}
\begin{proof}
第一式是推论 \ref{cor:conclusion-fibadic-exact-depth-characters-dominate} 的内容。第二式只需利用
\[
|G_d|=2^\beta F_d
\]
并代入第一式即可。证毕。
\end{proof}

\begin{conjecture}[观察者塔与深度包的 Tannakian 重建]\label{conj:conclusion-fibadic-observer-tower-tannakian-reconstruction}
设两套 Fibonacci 深度过滤系统具有同一观察者塔
\[
\{G_d\}_{d\ge 1}
\]
以及同一深度包多项式族
\[
\{\Pi_d\}_{d\ge 1}.
\]
则它们的有限深度可见范畴应当单位酉等价。更强地，应存在一个保测酉等价，把一套系统中的三元组
\[
\bigl(U,\{P_d\}_{d\ge 1},\{K_d\}_{d\ge 1}\bigr)
\]
送到另一套系统中的对应三元组。
\end{conjecture}

\begin{theorem}[Fibonacci 深度过滤中的饱和观察者闭链]\label{thm:conclusion-fibadic-saturated-observer-closure}
\leanverified{paper\_conclusion\_fibadic\_saturated\_observer\_closure}
在 Fibonacci 深度过滤框架下，精确深度、饱和导体、最小有限观察者、精确深度新空间、cyclotomic 深度包与有限层联合观测代数构成一条闭合链。更精确地，对每个活跃深度 \(d\) 与每个有限层 \(N\)：
\begin{enumerate}
  \item 精确深度 \(d\) 唯一决定饱和导体 \(F_d\)；
  \item 饱和导体 \(F_d\) 唯一决定最小忠实有限观察者
  \[
  G_d=(\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ);
  \]
  \item \(G_d\) 的地址字符中 ordinary conductor 满足 \(d_N=d\) 的部分与 \(\mathcal N_d\) 自然酉等价；
  \item \(U|_{\mathcal N_d}\) 的最小多项式恰为 \(\Pi_d\)；
  \item 由全部 \(\{G_d\}\) 与 \(\{\Pi_d\}\) 已可完全控制有限层联合观测代数 \(\mathscr A_N\) 的半单结构。
\end{enumerate}
因此，导体、观察者、精确深度字符块、cyclotomic 包与有限层代数在同一条 Fibonacci-adic 过滤链上被同时锁定。
\end{theorem}
\begin{proof}
第 \(1\) 条是定理 \ref{thm:conclusion-fibadic-exact-depth-conductor-saturation} 的内容；第 \(2\) 条由定理 \ref{thm:conclusion-mixed-hidden-state-exact-depth-minimal-receiver} 给出；第 \(3\) 条由定理 \ref{thm:conclusion-fibadic-newspace-minimal-observer-duality} 给出；第 \(4\) 条由定理 \ref{thm:conclusion-fibadic-cyclotomic-packet-decomposition} 与定理 \ref{thm:conclusion-fibadic-joint-observer-algebra-cyclotomic-product} 给出；第 \(5\) 条则是定理 \ref{thm:conclusion-fibadic-joint-observer-algebra-cyclotomic-product} 与推论 \ref{cor:conclusion-fibadic-joint-observer-algebra-spectral-completeness} 的合成。证毕。
\end{proof}
