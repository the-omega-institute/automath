\begin{theorem}[有限型 Jensen 缺陷的分布曲率分类]\label{thm:conclusion-jensen-defect-distributional-curvature-classification}
在定理 \ref{thm:conclusion-jensen-defect-radial-spline-rigidity} 的记号下，分布意义中有
\[
\Delta''=\sum_{j=1}^{s}m_j\delta_{x_j},
\qquad
R''=-\sum_{j=1}^{s}m_j\delta_{x_j}
\quad\text{于 }(-\infty,0).
\]
反之，任意有限正整数值原子测度
\[
\nu_0=\sum_{j=1}^{s}m_j\delta_{x_j},
\qquad
x_j<0,\ m_j\in \ZZ_{>0},
\]
都唯一决定一条有限型 Jensen 缺陷及其对偶排斥半径曲线：
\[
\Delta_{\nu_0}(x)=\int_{(-\infty,0)}(x-t)_+\,\dd \nu_0(t),
\qquad
R_{\nu_0}(x)=x-\Delta_{\nu_0}(x).
\]
因此，有限型径向缺陷与正整数原子曲率测度严格等价。
\end{theorem}
\leanverified{paper\_conclusion\_jensen\_defect\_distributional\_curvature\_classification}
\begin{proof}
对单个壳层函数 \((x-x_j)_+\) 有
\[
\frac{\dd}{\dd x}(x-x_j)_+=\mathbf 1_{(x_j,\infty)}(x),
\qquad
\frac{\dd^2}{\dd x^2}(x-x_j)_+=\delta_{x_j}
\]
在分布意义下成立。逐项求和即得
\[
\Delta''=\sum_{j=1}^{s}m_j\delta_{x_j}.
\]
由
\[
R=x-\Delta
\]
立得
\[
R''=-\Delta''.
\]

反向方向中，若给定
\[
\nu_0=\sum_{j=1}^{s}m_j\delta_{x_j},
\]
则定义
\[
\Delta_{\nu_0}(x):=\int_{(-\infty,0)}(x-t)_+\,\dd \nu_0(t)=\sum_{j=1}^{s}m_j(x-x_j)_+.
\]
这正是定理 \ref{thm:conclusion-jensen-defect-radial-spline-rigidity} 的有限型 Jensen 缺陷样条表示。唯一性来自原子测度分解的唯一性。证毕。
\end{proof}

\begin{theorem}[固定总重数与终端 Jensen 质量下的凸序分类]\label{thm:conclusion-jensen-defect-convex-order-classification}
\leanverified{paper\_conclusion\_jensen\_defect\_convex\_order\_classification}
设 \(\Delta_1,\Delta_2\) 为两条有限型 Jensen 缺陷，且具有相同的总重数 \(M\) 与终端 Jensen 质量 \(A\)。令其壳层经验测度分别为
\[
\nu_1=\frac{1}{M}\sum_j m_j^{(1)}\delta_{x_j^{(1)}},
\qquad
\nu_2=\frac{1}{M}\sum_j m_j^{(2)}\delta_{x_j^{(2)}}.
\]
则以下两命题等价：
\begin{enumerate}
  \item 对一切 \(x<0\)，有
  \[
  \Delta_1(x)\le \Delta_2(x);
  \]
  \item 在凸序意义下，
  \[
  \nu_1\preceq_{\mathrm{cx}}\nu_2.
  \]
\end{enumerate}
等价地，在固定 \((M,A)\) 的截面上，径向缺陷曲线的点态序关系完全等价于壳层对数半径分布的凸序关系。
\end{theorem}
\begin{proof}
由定义，
\[
\Delta_i(x)=M\int_{(-\infty,0)}(x-t)_+\,\dd \nu_i(t)
\qquad(i=1,2).
\]
又因两者具有相同的 \(M\) 与 \(A\)，故 \(\nu_1,\nu_2\) 具有相同总质量 \(1\) 与相同均值
\[
\int t\,\dd \nu_i(t)=-\frac{A}{M}.
\]
对等均值概率测度，标准的 stop-loss/凸序判据给出
\[
\nu_1\preceq_{\mathrm{cx}}\nu_2
\iff
\int (x-t)_+\,\dd \nu_1(t)\le \int (x-t)_+\,\dd \nu_2(t)
\quad(\forall x\in \RR).
\]
乘上同一正数 \(M\) 即得所述等价。证毕。
\end{proof}

\begin{theorem}[等壳极值的方差缺口恒等式]\label{thm:conclusion-equal-shell-variance-gap-identity}
在固定 \((M,A)\) 下，设
\[
\bar x:=-\frac{A}{M},
\qquad
\Delta_{\mathrm{eq}}(x):=M(x-\bar x)_+,
\qquad
R_{\mathrm{eq}}(x):=x-\Delta_{\mathrm{eq}}(x).
\]
则有精确积分恒等式
\[
\int_{-\infty}^{0}\bigl(\Delta(x)-\Delta_{\mathrm{eq}}(x)\bigr)\,\dd x
=
\frac{M}{2}\Var_\nu(X),
\]
等价地，
\[
\int_{-\infty}^{0}\bigl(R_{\mathrm{eq}}(x)-R(x)\bigr)\,\dd x
=
\frac{M}{2}\Var_\nu(X),
\qquad
X\sim \nu.
\]
因此，等壳极值不仅是点态极值，其全局面积缺口被壳层对数半径的方差精确度量。
\leanverified{paper\_conclusion\_equal\_shell\_variance\_gap\_identity}
\end{theorem}
\begin{proof}
由 Fubini 公式，对任意 \(t<0\) 有
\[
\int_{-\infty}^{0}(x-t)_+\,\dd x
=
\int_{t}^{0}(x-t)\,\dd x
=
\frac{t^2}{2}.
\]
因此
\[
\int_{-\infty}^{0}\Delta(x)\,\dd x
=
M\int_{(-\infty,0)}\frac{t^2}{2}\,\dd \nu(t)
=
\frac{M}{2}\EE_\nu[X^2].
\]
另一方面，
\[
\int_{-\infty}^{0}\Delta_{\mathrm{eq}}(x)\,\dd x
=
M\int_{-\infty}^{0}(x-\bar x)_+\,\dd x
=
\frac{M}{2}\bar x^2.
\]
相减得
\[
\int_{-\infty}^{0}\bigl(\Delta-\Delta_{\mathrm{eq}}\bigr)\,\dd x
=
\frac{M}{2}\bigl(\EE_\nu[X^2]-\bar x^2\bigr)
=
\frac{M}{2}\Var_\nu(X).
\]
再由 \(R=x-\Delta\) 与 \(R_{\mathrm{eq}}=x-\Delta_{\mathrm{eq}}\) 即得第二式。证毕。
\end{proof}

\begin{theorem}[排斥半径尾部的刚性恢复]\label{thm:conclusion-exclusion-radius-tail-affine-recovery}
\leanverified{paper\_conclusion\_exclusion\_radius\_tail\_affine\_recovery}
设 \(x_s:=\max_j x_j=\log \varrho_s\) 为最外壳层的对数半径。则在尾区间 \(x\in[x_s,0)\) 上有精确仿射公式
\[
\Delta(x)=Mx+A,
\qquad
R(x)=(1-M)x-A.
\]
特别地，尾部斜率与截距唯一恢复总重数和总 Jensen 质量：
\[
M=\Delta_+'(x)=1-R_+'(x)\qquad (x\in(x_s,0)),
\]
\[
A=\Delta(x)-Mx=-R(x)-(M-1)x.
\]
因此，有限型径向缺陷的两个全局量 \((M,A)\) 已完整编码在排斥半径曲线的终端 affine tail 中。
\end{theorem}
\begin{proof}
若 \(x\ge x_s\)，则对所有 \(j\) 都有 \(x\ge x_j\)，故
\[
(x-x_j)_+=x-x_j.
\]
于是
\[
\Delta(x)=\sum_{j=1}^{s}m_j(x-x_j)
=
\Bigl(\sum_{j=1}^{s}m_j\Bigr)x-\sum_{j=1}^{s}m_jx_j
=
Mx+A,
\]
其中
\[
A=-\sum_{j=1}^{s}m_jx_j.
\]
由
\[
R=x-\Delta
\]
立即得到
\[
R(x)=(1-M)x-A.
\]
斜率与截距公式由此直接读出。证毕。
\end{proof}

\begin{theorem}[一次内点等号强制有限 Toeplitz 平坦地平线]\label{thm:conclusion-toeplitz-flat-horizon-from-one-inner-equality}
设 \(\mu\) 为单位圆上的有限正 Borel 测度，总质量为 \(\eta\)。若在某个阶数 \(N\ge 1\) 与某个内点 \(w\in \DD\setminus\{0\}\) 上，定理 \ref{thm:conclusion-toeplitz-buffer-equality-regular-polygon-rigidity} 的 Carath\'eodory 缓冲下界取等，则：
\begin{enumerate}
  \item 对一切 \(1\le j\le N\)，其 Fourier 系数满足
  \[
  c_j:=\int_{\TT}\xi^{-j}\,\dd \mu(\xi)=0;
  \]
  \item 对一切 \(0\le m\le N\)，有
  \[
  T_m(\mu)=\eta I_{m+1};
  \]
  \item 但在下一阶，
  \[
  c_{N+1}\neq 0,
  \qquad
  T_{N+1}(\mu)\neq \eta I_{N+2}.
  \]
\end{enumerate}
因此，一次内点等号自动产生长度恰为 \(N\) 的 finite flatness horizon。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-toeplitz-buffer-equality-regular-polygon-rigidity}，
\[
\mu=\mu_{N,\eta,\phi}:=\frac{\eta}{N+1}\sum_{k=0}^{N}\delta_{\xi_k},
\qquad
\xi_k:=\exp\!\left(i\frac{\phi+2\pi k}{N+1}\right).
\]
于是
\[
c_j=\frac{\eta}{N+1}\sum_{k=0}^{N}\xi_k^{-j}.
\]
该和是 \((N+1)\)-次单位根和，因此
\[
c_j=
\begin{cases}
0,& 1\le j\le N,\\[2pt]
\eta e^{-i\phi},& j=N+1.
\end{cases}
\]
对任意 \(m\le N\)，Toeplitz 块
\[
T_m(\mu)=(c_{j-k})_{0\le j,k\le m}
\]
的非对角元素对应 \(1\le |j-k|\le m\le N\)，故全部为零；而对角元素都等于 \(c_0=\eta\)，从而
\[
T_m(\mu)=\eta I_{m+1}.
\]
当 \(m=N+1\) 时，矩阵中出现差值 \(j-k=\pm(N+1)\)，相应条目由 \(c_{N+1}\) 与 \(c_{-(N+1)}\) 给出且不为零，因此 \(T_{N+1}(\mu)\) 不再是标量单位矩阵。证毕。
\end{proof}

\begin{theorem}[正 \texorpdfstring{$(N+1)$}{(N+1)}-边形极值测度的 Carath\'eodory--Schur 单项式律]\label{thm:conclusion-regular-polygon-caratheodory-schur-monomial-law}
对定理 \ref{thm:conclusion-toeplitz-buffer-equality-regular-polygon-rigidity} 的唯一极值测度
\[
\mu_{N,\eta,\phi}=\frac{\eta}{N+1}\sum_{k=0}^{N}\delta_{\xi_k},
\qquad
\xi_k^{N+1}=e^{i\phi},
\]
其归一化 Carath\'eodory 函数与 Schur 函数满足精确公式
\[
\widetilde C_{N,\phi}(w):=\frac{1}{\eta}C_{\mu_{N,\eta,\phi}}(w)
=
\frac{1+e^{-i\phi}w^{N+1}}{1-e^{-i\phi}w^{N+1}},
\]
\[
S_{N,\phi}(w)=e^{-i\phi}w^{N+1}.
\]
从而对每个固定 \(0<r<1\)，有
\[
\|S_{N,\phi}(r\cdot)\|_{H^\infty(\DD)}=r^{N+1},
\]
并且 Carath\'eodory 缓冲下界的等号点恰为圆周 \(|w|=r\) 上满足
\[
(N+1)\theta\equiv \phi+\pi \pmod{2\pi}
\]
的 \(N+1\) 个相位。
\end{theorem}
\begin{proof}
由上一结论，
\[
c_j=0\quad(1\le j\le N),
\qquad
c_{(N+1)\ell}=\eta e^{-i\ell\phi}\quad(\ell\ge 1).
\]
于是
\[
\widetilde C_{N,\phi}(w)
=
1+2\sum_{j\ge 1}\frac{c_j}{\eta}w^j
=
1+2\sum_{\ell\ge 1}e^{-i\ell\phi}w^{(N+1)\ell}.
\]
这是一条几何级数，故
\[
\widetilde C_{N,\phi}(w)
=
\frac{1+e^{-i\phi}w^{N+1}}{1-e^{-i\phi}w^{N+1}}.
\]
再由 Schur 变换
\[
S=\frac{\widetilde C-1}{\widetilde C+1}
\]
即得
\[
S_{N,\phi}(w)=e^{-i\phi}w^{N+1}.
\]
因此在 \(|z|<1\) 上有
\[
|S_{N,\phi}(rz)|=r^{N+1}|z|^{N+1}\le r^{N+1},
\]
并在 \(|z|=1\) 上取到，故
\[
\|S_{N,\phi}(r\cdot)\|_{H^\infty(\DD)}=r^{N+1}.
\]

最后，
\[
\Re \widetilde C_{N,\phi}(re^{i\theta})
=
\frac{1-r^{2N+2}}{|1-e^{-i\phi}r^{N+1}e^{i(N+1)\theta}|^2}.
\]
分母最大当且仅当
\[
e^{-i\phi}r^{N+1}e^{i(N+1)\theta}=-r^{N+1},
\]
即
\[
(N+1)\theta\equiv \phi+\pi \pmod{2\pi}.
\]
此时取得缓冲下界
\[
\eta\,\frac{1-r^{N+1}}{1+r^{N+1}}.
\]
证毕。
\end{proof}

\begin{theorem}[Toeplitz 谱裕度编译的 sharp 阈值律]\label{thm:conclusion-toeplitz-margin-sharp-threshold-law}
对结论 \ref{thm:conclusion-regular-polygon-caratheodory-schur-monomial-law} 的极值测度，目标收缩率 \(0<q<1\) 在可见半径 \(0<r<1\) 上满足
\[
\|S_{N,\phi}(r\cdot)\|_{H^\infty(\DD)}\le q
\iff
r^{N+1}\le q.
\]
因此极值族中的最小阶数精确为
\[
N_{\min}(r,q)=\left\lceil \frac{\log q}{\log r}\right\rceil-1.
\]
特别地，推论 \ref{cor:conclusion-toeplitz-order-compilation-bound} 的阶数下界在正 \((N+1)\)-边形极值族上是 sharp 的。
\end{theorem}
\leanverified{paper\_conclusion\_toeplitz\_margin\_sharp\_threshold\_law}
\begin{proof}
由结论 \ref{thm:conclusion-regular-polygon-caratheodory-schur-monomial-law}，
\[
\|S_{N,\phi}(r\cdot)\|_{H^\infty(\DD)}=r^{N+1}.
\]
故
\[
\|S_{N,\phi}(r\cdot)\|_{H^\infty(\DD)}\le q
\iff
r^{N+1}\le q.
\]
由于 \(0<r,q<1\)，于是
\[
N_{\min}(r,q)
=
\min\{N\ge 0:\ r^{N+1}\le q\}
=
\left\lceil \frac{\log q}{\log r}\right\rceil-1.
\]
这说明一般阈值下界在该极值族上无法再改进。证毕。
\end{proof}

\begin{theorem}[固定 \texorpdfstring{$(M,A,\eta)$}{(M,A,eta)} 的径向--角向同步极值刚性]\label{thm:conclusion-radial-angular-synchronous-extremal-rigidity}
固定整数 \(M\ge 1\)、总 Jensen 质量 \(A>0\) 与圆周总质量 \(\eta>0\)。考虑如下复合极值问题：
\begin{enumerate}
  \item 在全部总重数为 \(M\)、总 Jensen 质量为 \(A\) 的有限型径向缺陷中，要求排斥半径曲线 \(r_*(\rho)\) 对所有 \(0<\rho<1\) 同时最大；
  \item 在全部总质量为 \(\eta\) 的单位圆有限正测度中，要求 \(N=M-1\) 阶 Carath\'eodory 缓冲下界在某个内点取等。
\end{enumerate}
则满足这两项条件的配置唯一具有如下形式：
\begin{itemize}
  \item 径向部分：全部缺陷零点塌缩到同一壳层
  \[
  \bar\varrho=e^{-A/M};
  \]
  \item 角向部分：测度必为某个旋转后的正 \(M\)-边形均匀离散测度
  \[
  \mu_{M-1,\eta,\phi}.
  \]
\end{itemize}
并且相应 Schur 输出严格为单项式
\[
S(w)=e^{-i\phi}w^{M}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-equal-shell-extremal-principle}，在固定 \((M,A)\) 下，排斥半径曲线的逐点最大值由唯一的等壳构型实现，即
\[
\varrho_1=\cdots=\varrho_s=\bar\varrho=e^{-A/M}.
\]
另一方面，由定理 \ref{thm:conclusion-toeplitz-buffer-equality-regular-polygon-rigidity}，若在阶数 \(N=M-1\) 的缓冲下界在某个内点取等，则测度唯一为
\[
\mu_{M-1,\eta,\phi}.
\]
两种极值分别在径向与角向上各自唯一，因此同时满足两者的复合极值对象也唯一。最后由结论 \ref{thm:conclusion-regular-polygon-caratheodory-schur-monomial-law}，
\[
S(w)=e^{-i\phi}w^{N+1}=e^{-i\phi}w^M.
\]
证毕。
\end{proof}

\begin{theorem}[稀疏零点半径谱的一尺度精确反演]\label{thm:conclusion-jensen-defect-single-scale-sparse-radius-inversion}
设圆盘解析函数 \(F\) 的零点对数半径测度为
\[
\mu_F=\sum_j m_j\delta_{t_j},
\qquad
t_j=\log|a_j|<0.
\]
定义 Jensen 缺陷的对数半径表示 \(J_F(x)\)，以及帽核二阶差分
\[
\Delta_h^2 J_F(x):=\int (h-|x-t|)_+\,\dd\mu_F(t).
\]
若支撑点满足严格分离
\[
|t_i-t_j|>2h
\qquad(i\neq j),
\]
则对每个原子位置 \(t_j\) 都有精确恢复公式
\[
\Delta_h^2 J_F(t_j)=h\,m_j,
\]
而对所有满足
\[
\operatorname{dist}(x,\operatorname{supp}\mu_F)\ge h
\]
的点则有
\[
\Delta_h^2 J_F(x)=0.
\]
因此，在最小间距超过 \(2h\) 的稀疏制度下，一条单尺度帽核扫描即可无混叠地恢复全部重数。
\end{theorem}
\begin{proof}
主稿已给出精确恒等式
\[
\Delta_h^2J_F(x)=\sum_j m_j(h-|x-t_j|)_+.
\]
若 \(x=t_j\)，则该式变为
\[
\Delta_h^2J_F(t_j)=hm_j+\sum_{i\neq j}m_i(h-|t_j-t_i|)_+.
\]
由支撑分离条件 \(|t_i-t_j|>2h>h\)，对所有 \(i\neq j\) 都有
\[
(h-|t_j-t_i|)_+=0,
\]
故仅剩 \(hm_j\)。

若 \(\operatorname{dist}(x,\operatorname{supp}\mu_F)\ge h\)，则对每个 \(j\) 都有 \(|x-t_j|\ge h\)，于是
\[
(h-|x-t_j|)_+=0,
\]
从而 \(\Delta_h^2J_F(x)=0\)。证毕。
\end{proof}

\begin{theorem}[Jensen 缺陷核验的 sharp 边界层障碍]\label{thm:conclusion-jensen-defect-sharp-boundary-layer-obstruction}
仍在结论 \ref{thm:conclusion-jensen-defect-single-scale-sparse-radius-inversion} 的框架下，设观测仅允许全半径加性误差不超过 \(\varepsilon>0\)。则有两条严格互补的事实：
\begin{enumerate}
  \item 若希望用三点触发量
  \[
  \widetilde\Delta_h^2J(x)
  \]
  稳定检测“窗口中心存在单重零点”，则必须满足 sharp 门槛
  \[
  \varepsilon<\frac h4.
  \]
  \item 反过来，对任意 \(\eta\in(0,\varepsilon]\)，都存在一对函数 \(F_0,F_1\)，其中 \(F_0\) 盘内无零点而 \(F_1\) 在深度 \(\eta\) 处具有单重零点，且满足
  \[
  \sup_{0<r<1}|D_{F_1}(r)-D_{F_0}(r)|\le \varepsilon.
  \]
\end{enumerate}
故任何只依赖 Jensen 缺陷全域观测、且误差预算为 \(\varepsilon\) 的确定性核验器，都不可能排除全部满足
\[
0<-\log|a|\le \varepsilon
\]
的边界层内部零点；而局部三点触发若要稳定分辨深度 \(h\)，其噪声预算恰须收缩到 \(h/4\) 以下。该门槛既是充分的，也是尖锐的。
\end{theorem}
\begin{proof}
主稿对三点触发机制已证明：理想单重零点的三点响应等于 \(h\)，而每个观测点的加性误差至多为 \(\varepsilon\) 时，二阶差分的最坏总误差恰为 \(4\varepsilon\)。因此，要在最坏情形下仍保证正响应，必须且只需
\[
h-4\varepsilon>0,
\]
即
\[
\varepsilon<\frac h4.
\]

另一方面，主稿还构造了全域不可判别样本：取
\[
F_0\equiv 1,
\qquad
F_1(w)=1-\frac{w}{a},
\qquad
|a|=e^{-\eta},
\]
则 \(F_0\) 在盘内无零点，而 \(F_1\) 在深度 \(\eta\) 处恰有一个单重零点，并且 Jensen 缺陷的全半径偏差满足
\[
\sup_{0<r<1}|D_{F_1}(r)-D_{F_0}(r)|=\eta\le \varepsilon.
\]
这说明只凭误差预算为 \(\varepsilon\) 的全域观测，不可能排除全部深度不超过 \(\varepsilon\) 的边界层零点。证毕。
\end{proof}
