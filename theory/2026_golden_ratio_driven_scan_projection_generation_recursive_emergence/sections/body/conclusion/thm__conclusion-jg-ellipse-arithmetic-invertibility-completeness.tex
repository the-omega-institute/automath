\begin{theorem}[Joukowsky--Gödel 椭圆的算术可逆性与椭圆化外置的完备性]\label{thm:conclusion-jg-ellipse-arithmetic-invertibility-completeness}
令 \(r:\mathbb P\to\NN\) 为有限支撑素数寄存器，并记
\[
N(r):=\prod_{p\in\mathbb P}p^{r(p)}\in\NN_{\ge 1}.
\]
按推论 \ref{cor:xi-joukowsky-prime-register-ellipse-parameterization} 的记号，令
\[
a(r):=\sqrt{N(r)}+\frac{1}{\sqrt{N(r)}},
\qquad
b(r):=\sqrt{N(r)}-\frac{1}{\sqrt{N(r)}},
\]
并定义对应的轴对齐 Joukowsky--Gödel 椭圆为
\[
\mathcal{E}_{\mathrm{JG}}(r)
:=
\left\{\,a(r)\cos\theta+i\,b(r)\sin\theta:\ \theta\in[0,2\pi)\,\right\}
\subset\CC\cong\RR^2.
\]
则成立：
\begin{enumerate}
  \item 映射
  \[
  \Psi:\RR_{>0}\to\RR_{>0}^2,\qquad
  L\mapsto \bigl(a(L),b(L)\bigr),
  \qquad
  a(L)=\sqrt L+\frac{1}{\sqrt L},\ \ b(L)=\sqrt L-\frac{1}{\sqrt L},
  \]
  为单射，并具有显式逆映射
  \[
  \Psi^{-1}(a,b)=\left(\frac{a+b}{2}\right)^2.
  \]
  \item 映射 \(r\mapsto \mathcal{E}_{\mathrm{JG}}(r)\) 为单射；给定 \(\mathcal{E}_{\mathrm{JG}}(r)\) 可先由半轴对 \((a(r),b(r))\) 恢复 \(N(r)\)，再由算术基本定理唯一恢复 \(r\)。
  \item 沿用定义 \ref{def:conclusion-cmon-externalization} 的 \((\pi,s)\) 与
  定理 \ref{thm:conclusion-cmon-externalization-initial-prime-register} 的结论。
  若 \(S:=\Omega\setminus s(X)\) 可数，则存在哥德尔化外置
  \(r_G:\Omega\to\NN_{\ge 1}\) 使扩展读出 \((\pi,r_G)\) 为单射。
  定义椭圆化记录
  \[
  \ell:=\mathcal{E}_{\mathrm{JG}}\circ r_G:\Omega\to\{\text{Joukowsky--Gödel 椭圆}\},
  \]
  则扩展读出 \((\pi,\ell):\Omega\to X\times\{\text{椭圆}\}\) 亦为单射。
\end{enumerate}
\leanverified{paper\_conclusion\_jg\_ellipse\_arithmetic\_invertibility\_completeness}
\leanverified{paper\_conclusion\_jg\_ellipse\_condition\_number\_integer\_threshold}
\end{theorem}
\begin{proof}
第一条由恒等式 \(a(L)+b(L)=2\sqrt L\) 得 \(\sqrt L=(a+b)/2\)，从而 \(L=((a+b)/2)^2\)，并立即推出单射性。
第二条由第一条恢复 \(N(r)\)，再用算术基本定理得到 \(r\) 的唯一性。
第三条：若 \((\pi,\ell)(\omega)=(\pi,\ell)(\omega')\)，则 \(\pi(\omega)=\pi(\omega')\) 且 \(\mathcal{E}_{\mathrm{JG}}(r_G(\omega))=\mathcal{E}_{\mathrm{JG}}(r_G(\omega'))\)。
由第二条知 \(r_G(\omega)=r_G(\omega')\)，再由 \((\pi,r_G)\) 单射得 \(\omega=\omega'\)。证毕。
\end{proof}

\begin{theorem}[椭圆观测的条件数：整数参数的分辨率阈值]\label{thm:conclusion-jg-ellipse-condition-number-integer-threshold}
设 \(N\in\NN_{\ge 2}\)，令 \(a=a(N)\)、\(b=b(N)\) 如定理 \ref{thm:conclusion-jg-ellipse-arithmetic-invertibility-completeness}。
给定观测 \((\tilde a,\tilde b)\) 满足
\[
|\tilde a-a|\le\varepsilon,
\qquad
|\tilde b-b|\le\varepsilon,
\]
并定义
\[
\tilde t:=\frac{\tilde a+\tilde b}{2},
\qquad
\tilde N:=\tilde t^{2}.
\]
则有误差界
\[
|\tilde N-N|
\le
2\varepsilon\sqrt N+\varepsilon^2.
\]
并且：
\begin{enumerate}
  \item 若 \(2\varepsilon\sqrt N+\varepsilon^2<\tfrac12\)，则将 \(\tilde N\) 取最近整数可唯一确定 \(N\)。
  \item 若
  \[
  \varepsilon\ \ge\ \frac12\,\max\!\bigl\{|a(N+1)-a(N)|,\ |b(N+1)-b(N)|\bigr\},
  \]
  则存在观测 \((\tilde a,\tilde b)\) 同时与 \(N\) 与 \(N+1\) 相容；
  因而在该绝对误差模型下，任何仅依赖 \((\tilde a,\tilde b)\) 的解码不可能在最坏情形中保证唯一恢复。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-jg-ellipse-arithmetic-invertibility-completeness} 的恒等式 \(t:=\sqrt N=(a+b)/2\)，有
\[
|\tilde t-t|
\le
\frac{|\tilde a-a|+|\tilde b-b|}{2}
\le
\varepsilon.
\]
故
\[
|\tilde N-N|
=|\tilde t^{2}-t^{2}|
=|\tilde t-t|\,|\tilde t+t|
\le
\varepsilon(2t+\varepsilon)
=2\varepsilon\sqrt N+\varepsilon^2.
\]
若右端小于 \(1/2\)，则最近整数唯一性成立，得第一条。
第二条中，令
\[
\Delta_N:=\max\!\bigl\{|a(N+1)-a(N)|,\ |b(N+1)-b(N)|\bigr\}.
\]
当 \(\varepsilon\ge \Delta_N/2\) 时，矩形
\(
[a(N)-\varepsilon,a(N)+\varepsilon]\times[b(N)-\varepsilon,b(N)+\varepsilon]
\)
与
\(
[a(N+1)-\varepsilon,a(N+1)+\varepsilon]\times[b(N+1)-\varepsilon,b(N+1)+\varepsilon]
\)
相交，取交集内任一点即为所求观测。
由此 \(N\) 与 \(N+1\) 在该噪声约束下不可区分，结论成立。证毕。
\end{proof}

\begin{corollary}[固定精度噪声下的素数寄存器信息屏障]\label{cor:conclusion-jg-ellipse-fixed-epsilon-information-barrier}
固定任意绝对误差门限 \(\varepsilon_0>0\)。
则存在 \(N_0(\varepsilon_0)\) 使得对所有 \(N\ge N_0(\varepsilon_0)\)，存在观测 \((\tilde a,\tilde b)\) 同时与至少两个不同整数参数相容；
因而在定理 \ref{thm:conclusion-jg-ellipse-arithmetic-invertibility-completeness} 的椭圆化外置中，若椭圆半轴仅能以固定精度访问，则对足够大尺度的外置状态，素数寄存器在椭圆层意义下不可判别。
\leanverified{paper\_conclusion\_jg\_ellipse\_fixed\_epsilon\_information\_barrier}
\end{corollary}
\begin{proof}
注意 \(a(N)\) 与 \(b(N)\) 均为严格递增且
\[
\lim_{N\to\infty}\bigl(a(N+1)-a(N)\bigr)=0,
\qquad
\lim_{N\to\infty}\bigl(b(N+1)-b(N)\bigr)=0.
\]
故存在 \(N_0\) 使得对所有 \(N\ge N_0\) 有
\(
\max\{|a(N+1)-a(N)|,|b(N+1)-b(N)|\}\le 2\varepsilon_0
\)。
应用定理 \ref{thm:conclusion-jg-ellipse-condition-number-integer-threshold} 的第二条即得。证毕。
\end{proof}

\begin{theorem}[素数椭圆族的 \texorpdfstring{$\varepsilon$}{ε}-分离容量：\texorpdfstring{$\sqrt{\text{尺度}}$}{sqrt(scale)} 律]\label{thm:conclusion-jg-ellipse-packing-sqrt-law}
对整数 \(N\ge 2\) 令 \(\mathcal{E}_{\mathrm{JG}}(N)\) 为定理 \ref{thm:conclusion-jg-ellipse-arithmetic-invertibility-completeness} 所定义的 Joukowsky--Gödel 椭圆，并以半轴最大差定义距离
\[
d\bigl(\mathcal{E}_{\mathrm{JG}}(N),\mathcal{E}_{\mathrm{JG}}(M)\bigr)
:=
\max\bigl\{|a(N)-a(M)|,\ |b(N)-b(M)|\bigr\}.
\]
令 \(\mathcal P(M,\varepsilon)\) 表示集合 \(\{\mathcal{E}_{\mathrm{JG}}(N):2\le N\le M\}\) 的最大 \(\varepsilon\)-分离子集基数。
则对任意 \(M\ge 2\) 与 \(\varepsilon>0\) 有界
\[
\left\lfloor \frac{\sqrt M-2}{2\varepsilon}\right\rfloor
\ \le\
\mathcal P(M,\varepsilon)
\ \le\
\left\lfloor \frac{\sqrt M-\sqrt 2}{\varepsilon}\right\rfloor+1.
\]
从而当 \(M\to\infty\) 且 \(\varepsilon\) 固定时，
\[
\mathcal P(M,\varepsilon)\asymp \frac{\sqrt M}{\varepsilon}.
\]
\leanverified{paper\_conclusion\_jg\_ellipse\_packing\_sqrt\_law}
\end{theorem}
\begin{proof}
记 \(t_N:=\sqrt N\)。由 \(a(N)+b(N)=2t_N\) 与不等式
\[
\max\{|x|,|y|\}\ge \frac{|x|+|y|}{2}
\]
得对任意 \(N<M\),
\[
d\bigl(\mathcal{E}_{\mathrm{JG}}(N),\mathcal{E}_{\mathrm{JG}}(M)\bigr)
\ge
\frac{|a(M)-a(N)|+|b(M)-b(N)|}{2}
=
t_M-t_N.
\]
上界：若 \(2\le N_1<\cdots<N_k\le M\) 使 \(\{\mathcal{E}_{\mathrm{JG}}(N_i)\}\) \(\varepsilon\)-分离，则 \(t_{N_{i+1}}-t_{N_i}\ge\varepsilon\)；
累加得 \(\sqrt M-\sqrt2\ge t_{N_k}-t_{N_1}\ge(k-1)\varepsilon\)，故 \(k\le (\sqrt M-\sqrt2)/\varepsilon+1\)。
下界：令 \(N_j:=\left\lceil(2+2j\varepsilon)^2\right\rceil\)。
当 \(N_j\le M\) 时有 \(t_{N_{j+1}}-t_{N_j}\ge 2\varepsilon\)，从而相邻两项距离至少为 \(\varepsilon\)，于是得到所示下界。证毕。
\end{proof}

\begin{theorem}[素数方向的椭圆微分生成元：\texorpdfstring{$\log p$}{log p} 的几何化]\label{thm:conclusion-jg-ellipse-prime-log-generator}

对有限支撑寄存器 \(r:\mathbb P\to\NN\) 定义

u(r):=\frac12\sum_{p\in\mathbb P}r(p)\log p

\qquad
b(r)=2\sinh u(r).

\delta a
=\left(\sum_{p}h(p)\log p\right)\sinh u(r),

\delta b

\]
特别地，若仅将某一素数指数 \(r(p)\) 增加 \(1\)，则在一阶近似下

\end{proof}

\begin{theorem}[Joukowsky 椭圆族上的全纯盲性与径向可识别性]\label{thm:conclusion-joukowsky-holomorphic-blindness-radial-identifiability}

\[

  \[
  R_2(r)=r^2+r^{-2},

  \[
  r=
  \sqrt{\frac{R_2(r)+\sqrt{R_2(r)^2-4}}{2}}.

\end{enumerate}

\begin{proof}

且与 \(r\) 无关。后半即主稿命题中的径向二次矩闭式与反演公式。两者并置，即得“全纯盲性--径向可识别性”的严格二分。证毕。

\begin{theorem}[Joukowsky--Gödel 输运的奇部伽罗瓦不变量]\label{thm:conclusion-jg-transport-oddpart-galois-invariance}
设 \(P(z)\in K[z]\) 可分，\(r\in K^\times\)，并令
\[
Q_r(w):=\operatorname{Res}_z\bigl(P(z),\,rz^2-wz+r^{-1}\bigr).
\]
记 \(L_P\) 与 \(L_Q\) 分别为 \(P\) 与 \(Q_r\) 在 \(K(r)\) 上的分裂域。则存在某个有限根集 \(W\)（可取为 \(Q_r\) 的全体根）使
\[
L_P=L_Q\bigl(\sqrt{w^2-4}:w\in W\bigr),
\qquad
[L_P:L_Q]=2^m
\]
对某个 \(m\ge 0\) 成立。因而 \(L_P/K(r)\) 的任意奇次数子扩张都已包含在 \(L_Q\) 中；等价地，Joukowsky--Gödel 输运不会创造任何新的奇部伽罗瓦信息。
\end{theorem}
\begin{proof}
主稿已证明 \(L_P\) 可由 \(L_Q\) 继续 adjoining 有限多个平方根 \(\sqrt{w^2-4}\) 得到，故扩张 \(L_P/L_Q\) 为一条有限 \(2\)-群塔，特别地
\[
[L_P:L_Q]=2^m
\]
对某个 \(m\ge 0\) 成立。

若 \(E/K(r)\subseteq L_P\) 为奇次数子扩张，则
\[
[E:E\cap L_Q]\mid [L_P:L_Q]=2^m,
\]
而左端同时整除奇数 \([E:K(r)]\)。故
\[
[E:E\cap L_Q]=1,
\]
即 \(E\subseteq L_Q\)。这正是奇部伽罗瓦不变量性。证毕。
\end{proof}

\begin{theorem}[椭圆输运零点云的谐波协议完备性]\label{thm:conclusion-elliptic-transport-zero-cloud-harmonic-protocol-completeness}

\[
f:=F\circ J_r,\qquad
f(e^{i\theta})=\sum_{k\in\ZZ}c_k e^{ik\theta},
\]
并假设
\[
\sum_{k\neq 0}|c_k|\,|k|\log(1+|k|)<\infty.
\]
则存在绝对常数 \(C\) 使
\[
\left|\int_{E_r}F\,\dd\nu_{m,r}-\int_{E_r}F\,\dd\mu_r\right|
\le
\frac{C}{D_m}\sum_{k\neq 0}|c_k|\,|k|\log(1+|k|),
\]
其中 \(D_m=\deg P_m\)。因此，任何由有限个此类谐波统计组成的椭圆协议，对零点云所能看到的全部非 Haar 偏差都统一压缩为 \(O(D_m^{-1})\)；在这一测试类中，椭圆输运是渐近统计完备的。
\end{theorem}
\begin{proof}
主稿在单位圆侧已证明：若
\[
f(e^{i\theta})=\sum_{k\in\ZZ}c_k e^{ik\theta}
\]
满足
\[
\sum_{k\neq 0}|c_k|\,|k|\log(1+|k|)<\infty,
\]
则
\[
\left|\int f\,\dd\mu_m-\int f\,\dd\mu_{\mathrm{Haar}}\right|
\le
\frac{C}{D_m}\sum_{k\neq 0}|c_k|\,|k|\log(1+|k|),
\]
其中 \(\mu_m\) 为单位圆上的零点归一化测度，\(D_m=\deg P_m\)。

又由椭圆输运的推前关系，
\[
\nu_{m,r}=(J_r)_\#\mu_m,
\qquad
\mu_r=(J_r)_\#\mu_{\mathrm{Haar}}.
\]
因此
\[
\int_{E_r}F\,\dd\nu_{m,r}=\int f\,\dd\mu_m,
\qquad
\int_{E_r}F\,\dd\mu_r=\int f\,\dd\mu_{\mathrm{Haar}},
\]
代回即得所示估计。证毕。
\end{proof}

\begin{theorem}[代数喷射场与对数分支坐标的正交性]\label{thm:conclusion-algebraic-jet-field-log-branch-orthogonality}
设 \(\lambda(t)\) 为代数特征方程
\[
\chi(\lambda,t)=0
\]
在代数点 \(t_0\in K\) 附近的一条简单解析分支，且 \(\lambda(t_0)=\lambda_0\in \overline K\)。则对任意 \(m\ge 0\)，皆有
\[
\lambda^{(m)}(t_0)\in K(\lambda_0).
\]
另一方面，若 \(y_\ast\in\overline{\QQ}\setminus\{0,1\}\) 为任一非平凡 Lee--Yang 分歧值，任取对数分支
\[
s_\ast=\log y_\ast,
\]
则 \(s_\ast\) 必为超越数。因而，任意由有限阶代数喷射闭包生成的局部域，都不可能包含任何非平凡 Lee--Yang 分歧值的对数坐标；简言之，
\[
\text{algebraic jet field}\ \perp\ \text{logarithmic branch chart}.
\]
\end{theorem}
\begin{proof}
简单根条件 \(\partial_\lambda\chi(\lambda_0,t_0)\neq 0\) 允许在 \(t_0\) 邻域内应用隐函数定理，得到唯一解析分支 \(\lambda(t)\)。对恒等式 \(\chi(\lambda(t),t)=0\) 逐次求导，可递归地把每一阶导数 \(\lambda^{(m)}(t_0)\) 表达为 \(K(\lambda_0)\) 中元素的有理式，因此
\[
\lambda^{(m)}(t_0)\in K(\lambda_0)
\qquad(m\ge 0).
\]

若 \(e^{s_\ast}=y_\ast\) 且 \(s_\ast\) 代数，则当 \(s_\ast\neq 0\) 时，Lindemann 定理推出 \(e^{s_\ast}\) 超越，这与 \(y_\ast\in\overline{\QQ}\) 矛盾；而 \(s_\ast=0\) 又会导致 \(y_\ast=1\)，同样被排除。故 \(s_\ast\) 必为超越数。

于是前者生成的任何有限代数喷射域都仍是代数扩张，而后者的对数坐标必为超越数，二者不可能相交于非平凡 Lee--Yang 分歧坐标。证毕。
\end{proof}
