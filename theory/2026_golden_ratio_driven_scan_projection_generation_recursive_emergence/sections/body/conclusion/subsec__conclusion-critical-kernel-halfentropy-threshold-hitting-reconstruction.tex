\begin{theorem}[临界伪行列式的唯一极小者为均匀分布]\label{thm:conclusion-critical-kernel-pdet-uniform-minimizer}
\leanverified{paper\_conclusion\_critical\_kernel\_pdet\_uniform\_minimizer}
对任意全支撑分布 \(w\in \Delta(X)\)，临界伪行列式满足
\[
\operatorname{pdet}(L_w)\ge \left(\frac{n}{n-1}\right)^{n-1},
\]
且等号成立当且仅当 \(w\) 为 \(X\) 上的均匀分布。
\end{theorem}
\begin{proof}
由主稿第 \(9.1486\) 节的闭式，
\[
\operatorname{pdet}(L_w)
=
\frac{A(w)^{\,n-2}}{C(w)^{\,n-1}\sqrt{\prod_{x\in X}w(x)}}.
\]
对数均值--算术均值不等式施于 \(\{\sqrt{w(x)}\}_{x\in X}\) 给出
\[
\sqrt{\prod_{x\in X}w(x)}
=
\prod_{x\in X}\sqrt{w(x)}
\le
\left(\frac{A(w)}{n}\right)^n.
\]
因此
\[
\operatorname{pdet}(L_w)
\ge
\frac{n^n}{A(w)^2\,C(w)^{n-1}}
=
\frac{n^n}{(1+C(w))\,C(w)^{n-1}}.
\]
另一方面，Cauchy 不等式给出 \(A(w)^2\le n\)，故
\[
C(w)=A(w)^2-1\le n-1.
\]
函数
\[
f(C):=(1+C)C^{n-1}\qquad (C>0)
\]
严格递增，于是
\[
\operatorname{pdet}(L_w)
\ge
\frac{n^n}{n(n-1)^{n-1}}
=
\left(\frac{n}{n-1}\right)^{n-1}.
\]
两处不等式同时取等当且仅当 \(\sqrt{w(x)}\) 全部相等，即 \(w\) 为均匀分布。证毕。
\end{proof}

\begin{theorem}[张量幂伪行列式的半熵单参数化]\label{thm:conclusion-critical-kernel-tensor-pdet-halfentropy-param}
\leanverified{paper\_conclusion\_critical\_kernel\_tensor\_pdet\_halfentropy\_param}
令 \(\omega_k:=w^{\otimes k}\in \Delta(X^k)\)。则对每个 \(k\ge 1\)，有精确恒等式
\[
\operatorname{pdet}(L_{\omega_k})\,G(\omega_k)^{n^k/2}
=
\frac{A(w)^{\,k(n^k-2)}}{\bigl((1+C(w))^k-1\bigr)^{n^k-1}}.
\]
等价地，记
\[
H_{1/2}(w):=2\log A(w),
\]
则
\[
\operatorname{pdet}(L_{\omega_k})\,G(\omega_k)^{n^k/2}
=
\frac{\exp\!\bigl(\frac{k(n^k-2)}{2}H_{1/2}(w)\bigr)}
{\bigl(e^{kH_{1/2}(w)}-1\bigr)^{n^k-1}}.
\]
因此，去除几何均值因子后，整个张量塔 \(\{\operatorname{pdet}(L_{w^{\otimes k}})\}_{k\ge 1}\) 只由单个半阶 Rényi 熵 \(H_{1/2}(w)\) 决定。
\end{theorem}
\begin{proof}
由张量积结构，
\[
A(\omega_k)
=
\sum_{(x_1,\dots,x_k)\in X^k}\sqrt{w(x_1)\cdots w(x_k)}
=
A(w)^k.
\]
故
\[
C(\omega_k)=A(\omega_k)^2-1=A(w)^{2k}-1=(1+C(w))^k-1.
\]
又
\[
G(\omega_k)=G(w)^k.
\]
将这些代入主稿第 \(9.1486\) 节的闭式
\[
\operatorname{pdet}(L_{\omega_k})
=
\frac{A(\omega_k)^{n^k-2}}
{C(\omega_k)^{n^k-1}\sqrt{\prod_{\xi\in X^k}\omega_k(\xi)}},
\]
并注意
\[
\sqrt{\prod_{\xi\in X^k}\omega_k(\xi)}=G(\omega_k)^{n^k/2},
\]
便得到第一式。第二式只是用 \(H_{1/2}(w)=2\log A(w)\) 改写。证毕。
\end{proof}

\begin{corollary}[单层张量伪行列式即可唯一恢复半熵]\label{cor:conclusion-critical-kernel-single-layer-pdet-recovers-halfentropy}
固定字母表大小 \(n\ge 2\) 与张量层数 \(k\ge 1\)。定义
\[
\Pi_k(w):=\operatorname{pdet}(L_{w^{\otimes k}})\,G(w)^{k n^k/2}.
\]
则映射
\[
w\longmapsto \Pi_k(w)
\]
仅依赖于 \(H_{1/2}(w)\)，并且对 \(H_{1/2}(w)\) 严格单调。特别地，若
\[
\Pi_k(w)=\Pi_k(v)
\]
对某个单独的 \(k\) 成立，则
\[
H_{1/2}(w)=H_{1/2}(v),
\]
从而对所有 \(\ell\ge 1\) 都有
\[
\Pi_\ell(w)=\Pi_\ell(v).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-critical-kernel-tensor-pdet-halfentropy-param}，
\[
\Pi_k(w)=F_{n,k}(h),
\qquad
h:=H_{1/2}(w),
\]
其中
\[
F_{n,k}(h)
:=
\frac{\exp\!\bigl(\frac{k(n^k-2)}2h\bigr)}
{(e^{kh}-1)^{n^k-1}}.
\]
于是
\[
\frac{d}{dh}\log F_{n,k}(h)
=
\frac{k(n^k-2)}2
-k(n^k-1)\frac{e^{kh}}{e^{kh}-1}.
\]
由于
\[
\frac{e^{kh}}{e^{kh}-1}>1,
\]
故
\[
\frac{d}{dh}\log F_{n,k}(h)
<
\frac{k(n^k-2)}2-k(n^k-1)
=
-\frac{k n^k}{2}
<0.
\]
因此 \(F_{n,k}\) 严格递减，单个 \(\Pi_k\) 的值便唯一决定 \(H_{1/2}(w)\)。再由定理 \ref{thm:conclusion-critical-kernel-tensor-pdet-halfentropy-param}，整个张量塔都由该半熵唯一决定。证毕。
\end{proof}

\begin{theorem}[单一非平凡本征对的分布恢复定理]\label{thm:conclusion-critical-kernel-single-eigenpair-recovers-distribution}
设
\[
t_x:=\frac{A(w)}{\sqrt{w(x)}}
\]
两两不同，并记
\[
L_0:=C(w)L_w=\diag(t)-\mathbf 1\mathbf 1^{\mathsf T}.
\]
若 \((\lambda,v)\) 是 \(L_0\) 的任一非平凡本征对，则
\[
\sigma:=\sum_{x\in X}v_x\ne 0,
\qquad
t_x=\lambda+\frac{\sigma}{v_x}\quad(\forall x\in X),
\]
从而
\[
w(x)
=
\frac{\bigl(\lambda+\sigma/v_x\bigr)^{-2}}
{\sum_{u\in X}\bigl(\lambda+\sigma/v_u\bigr)^{-2}}.
\]
因此，在谱简单情形下，任意一个非平凡本征向量连同其本征值已经足以重构整个分布 \(w\)。
\end{theorem}
\begin{proof}
本征方程写成
\[
(\diag(t)-\mathbf 1\mathbf 1^{\mathsf T})v=\lambda v,
\]
即
\[
(t_x-\lambda)v_x=\sum_{u\in X}v_u=\sigma
\qquad(\forall x\in X).
\]
若 \(\sigma=0\)，则
\[
(t_x-\lambda)v_x=0\qquad(\forall x).
\]
由于 \(t_x\) 两两不同，满足 \(t_x=\lambda\) 的指标至多一个，从而 \(v\) 的支撑至多一点；但这又与 \(\sigma=\sum_x v_x=0\) 迫使 \(v=0\)，矛盾。故 \(\sigma\ne 0\)。于是每个 \(v_x\) 都非零，并且
\[
t_x=\lambda+\frac{\sigma}{v_x}.
\]
再由 \(t_x=A(w)/\sqrt{w(x)}\) 得
\[
w(x)=\frac{A(w)^2}{t_x^2}.
\]
对 \(x\) 求和可得
\[
1=A(w)^2\sum_{u\in X}t_u^{-2},
\qquad
A(w)^{-2}=\sum_{u\in X}t_u^{-2}.
\]
代回即得
\[
w(x)
=
\frac{t_x^{-2}}{\sum_{u\in X}t_u^{-2}}
=
\frac{\bigl(\lambda+\sigma/v_x\bigr)^{-2}}
{\sum_{u\in X}\bigl(\lambda+\sigma/v_u\bigr)^{-2}}.
\]
证毕。
\end{proof}

\begin{corollary}[非平凡本征向量的阈值切割全图册]\label{cor:conclusion-critical-kernel-threshold-nodal-atlas}
在定理 \ref{thm:conclusion-critical-kernel-single-eigenpair-recovers-distribution} 的假设下，按
\[
t_{(1)}<t_{(2)}<\cdots<t_{(n)}
\]

\[
\lambda_2,\dots,\lambda_n,
\qquad
\lambda_{k+1}\in (t_{(k)},t_{(k+1)}).
\]
则第 \(k+1\) 个本征向量 \(v^{(k+1)}\) 满足
\[
v^{(k+1)}_{(j)}=\frac{\sigma_k}{t_{(j)}-\lambda_{k+1}},
\]

\begin{enumerate}
  \item 每个非平凡本征向量都恰有两个 nodal domains；
  \item 其正负支撑是按 \(t\)-顺序的一个严格阈值切割；
  \item 由于 \(t_x=A(w)/\sqrt{w(x)}\)，同一切割等价于按 \(w\)-大小的阈值切割；

\end{enumerate}
\end{corollary}
\begin{proof}
主稿第 \(9.1493\) 节给出交错关系
\[
\lambda_{k+1}\in (t_{(k)},t_{(k+1)}),
\]

因此当 \(j\le k\) 时分母为负，当 \(j\ge k+1\) 时分母为正，除整体符号外恰有一次符号翻转。于是 nodal domains 恰有两块，并且正负支撑正是按 \(t\)-顺序的阈值切割。

t_x=\frac{A(w)}{\sqrt{w(x)}},

\end{proof}

\begin{theorem}[分离距离的起点无关线性递推]\label{thm:conclusion-critical-kernel-separation-common-recurrence}
设
\[
s_x(m):=\operatorname{sep}_x(m)\qquad(m\ge 0)
\]
为主稿第 \(9.1506\) 节的精确分离距离。由主稿第 \(9.1509\) 节，存在只依赖于 \((w,\delta)\) 的公共多项式
\[
\Delta_\delta(s)=1+a_1 s+\cdots+a_{n-1}s^{n-1},
\]
使得对每个起点 \(x\) 都有
\[
\sum_{m\ge 0}s_x(m)s^m=\frac{N_x(s)}{\Delta_\delta(s)},
\qquad
\deg N_x\le n-2.
\]
于是所有起点共享同一线性递推
\[
s_x(m+n-1)+a_1 s_x(m+n-2)+\cdots+a_{n-1}s_x(m)=0
\qquad(m\ge 0).
\]
因此，分离距离的时间衰减在起点层面并不产生新的谱结构；起点只改变初值，不改变递推律。
\end{theorem}
\begin{proof}
令
\[
S_x(s):=\sum_{m\ge 0}s_x(m)s^m.
\]
由假设
\[
\Delta_\delta(s)S_x(s)=N_x(s).
\]
由于 \(\deg N_x\le n-2\)，右端在所有 \(s^{m+n-1}\)（\(m\ge 0\)）上的系数恒为零。比较左端 \(s^{m+n-1}\) 的系数，便得到
\[
s_x(m+n-1)+a_1 s_x(m+n-2)+\cdots+a_{n-1}s_x(m)=0.
\]
递推系数完全由公共分母 \(\Delta_\delta\) 决定，与起点 \(x\) 无关。证毕。
\end{proof}

证毕。

\begin{theorem}[固定目标的一致 hitting-spectrum 原理]\label{thm:conclusion-critical-kernel-fixed-target-hitting-spectrum}
固定目标点 \(y\in X\)。设 \(t_u\) 在 \(X\setminus\{y\}\) 上两两不同，并记
\[
D_y(z):=zP_{-y}'(z)+\kappa_\delta P_{-y}(z),
\qquad
P_{-y}(z):=\prod_{u\ne y}(t_u-z).
\]
记其正根为
\[
z_{y,1}<\cdots<z_{y,n-1},
\qquad
\lambda_{y,i}:=\frac{\kappa_\delta}{z_{y,i}}\in(0,1).
\]
则对每个起点 \(x\ne y\)，存在唯一实系数 \(\omega_i^{(x,y)}\) 使
\[
\PP_x(\tau_y=k)
=
\sum_{i=1}^{n-1}\omega_i^{(x,y)}(1-\lambda_{y,i})\lambda_{y,i}^{k-1},
\qquad k\ge 1,
\]
\[
\PP_x(\tau_y>k)
=
\sum_{i=1}^{n-1}\omega_i^{(x,y)}\lambda_{y,i}^{k},
\qquad k\ge 0.
\]
因此，在固定目标 \(y\) 下，全部起点的 hitting laws 共享同一组由目标控制的衰减模态；起点只改变模态权重，不改变模态位置。
\end{theorem}
\begin{proof}
主稿第 \(9.1498\) 节给出对每个 \(x\ne y\),
\[
\EE_x[s^{\tau_y}]
=
\kappa_\delta s\,\frac{t_x-\kappa_\delta}{t_y-\kappa_\delta}
\cdot
\frac{P_{-x,-y}(\kappa_\delta s)}{D_y(\kappa_\delta s)}.
\]
因此不同起点的概率母函数具有同一分母 \(D_y(\kappa_\delta s)\)。由主稿第 \(9.1499\) 节，\(D_y\) 的根全为正且严格交错，从而 \(\lambda_{y,i}\in(0,1)\) 且在当前简单谱假设下全部不同。对上述有理函数作部分分式分解，可写成
\[
\EE_x[s^{\tau_y}]
=
\sum_{i=1}^{n-1}\frac{\omega_i^{(x,y)}(1-\lambda_{y,i})s}{1-\lambda_{y,i}s}.
\]
比较幂级数系数即得两式。证毕。
\end{proof}

\begin{theorem}[单一目标分母的完全逆问题]\label{thm:conclusion-critical-kernel-one-target-denominator-reconstruction}
\leanverified{paper\_conclusion\_critical\_kernel\_one\_target\_denominator\_reconstruction}
在结论 \ref{thm:conclusion-critical-kernel-fixed-target-hitting-spectrum} 的设定下，固定一个目标点 \(y\)。则自然 Laguerre 分母
\[
D_y(z)=zP_{-y}'(z)+\kappa_\delta P_{-y}(z)
\]
单独一项已经足以唯一恢复整个参数族 \(\{t_x\}_{x\in X}\)，从而唯一恢复分布 \(w\)。

更具体地，写
\[
P_{-y}(z)=\sum_{j=0}^{n-1}p_j z^j,
\qquad
D_y(z)=\sum_{j=0}^{n-1}d_j z^j.
\]
则
\[
d_j=(j+\kappa_\delta)p_j\qquad(0\le j\le n-1).
\]
由于
\[
p_{n-1}=(-1)^{n-1},
\]
可由最高次系数直接读出
\[
\kappa_\delta=(-1)^{n-1}d_{n-1}-(n-1),
\]
进而
\[
p_j=\frac{d_j}{j+\kappa_\delta}\qquad(0\le j\le n-1).
\]
于是 \(P_{-y}\) 被完全恢复，其根即为 \(\{t_u:\ u\ne y\}\)。最后由归一化关系
\[
\sum_{u\in X}\frac1{t_u-\kappa_\delta}=1
\]
得到
\[
t_y
=
\kappa_\delta+\left(1-\sum_{u\ne y}\frac1{t_u-\kappa_\delta}\right)^{-1},
\]
从而
\[
w(x)=\frac{t_x^{-2}}{\sum_{v\in X}t_v^{-2}}.
\]
\end{theorem}
\begin{proof}
将
\[
P_{-y}(z)=\sum_{j=0}^{n-1}p_j z^j
\]
代入定义
\[
D_y(z)=zP_{-y}'(z)+\kappa_\delta P_{-y}(z),
\]
得到
\[
D_y(z)=\sum_{j=0}^{n-1}(j+\kappa_\delta)p_j z^j.
\]
这便是
\[
d_j=(j+\kappa_\delta)p_j.
\]

另一方面，
\[
P_{-y}(z)=\prod_{u\ne y}(t_u-z)
\]
的最高次系数为 \(p_{n-1}=(-1)^{n-1}\)。故
\[
d_{n-1}=(n-1+\kappa_\delta)(-1)^{n-1},
\]
从而 \(\kappa_\delta\) 可由 \(d_{n-1}\) 唯一读出。于是每个
\[
p_j=\frac{d_j}{j+\kappa_\delta}
\]
都唯一确定，故 \(P_{-y}\) 唯一恢复，其根给出全部 \(t_u\)（\(u\ne y\)）。

剩余的 \(t_y\) 由标准归一化关系
\[
\sum_{u\in X}\frac1{t_u-\kappa_\delta}=1
\]
唯一求出。最后由
\[
w(x)\propto t_x^{-2}
\]
并结合总质量归一化，即得
\[
w(x)=\frac{t_x^{-2}}{\sum_{v\in X}t_v^{-2}}.
\]
证毕。
\end{proof}

\begin{corollary}[单目标分母决定整个最优核]\label{cor:conclusion-critical-kernel-one-target-denominator-complete-certificate}
在结论 \ref{thm:conclusion-critical-kernel-one-target-denominator-reconstruction} 的条件下，固定任意一个目标 \(y\)。若自然分母 \(D_y\) 已知，则不但 \(w\) 唯一确定，而且整个最优核
\[
K_\delta^\ast
\]
亦唯一确定。因而：
\begin{enumerate}
  \item 所有首次命中时间分布 \(\tau_z\)（任意目标 \(z\)）都可重建；
  \item 所有 resolvent 矩元都可重建；
  \item 所有分离距离曲线 \(\operatorname{sep}_x(m)\) 都可重建。
\end{enumerate}
换言之，一个目标的 Laguerre 分母已经是整个最优核的全局完备证书。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-critical-kernel-one-target-denominator-reconstruction}，\(D_y\) 唯一恢复全部 \(t_x\)，从而唯一恢复分布 \(w\)。而主稿第 \(9.1530\) 节已给出最优核的显式公式：对任意 \(x,z\in X\),
\[
K_\delta^\ast(z\mid x)
=
\left(1-\frac{t_x-\kappa_\delta}{t_x}\right)\mathbf 1_{\{x=z\}}
+
\frac{t_x-\kappa_\delta}{t_x}\cdot \frac{1}{t_z-\kappa_\delta}.
\]
故 \(K_\delta^\ast\) 亦唯一确定。随后，主稿第 \(9.1498\)、\(9.1506\)--\(9.1509\) 与 \(9.1532\) 节中的首次命中、resolvent 与分离距离公式都成为显式可计算量。证毕。
\end{proof}

上述十条结论把第 \(9.1482\)--\(9.1532\) 节原本分散的几条局部闭式第一次压成一条单独可调用的终端链：一方面，临界伪行列式的整个张量塔在去除几何均值因子后只保留一个半阶 Rényi 熵参数，且均匀分布是唯一极小者；另一方面，单一本征对已足以恢复分布并生成完整的阈值切割图册，而单一目标的 Laguerre 分母则进一步把这一恢复原则提升到 hitting law、resolvent 与 separation 全部同时可逆的层面。就本章整体结构而言，这意味着临界最优核并非只能以“存在唯一最优解”的变分语言理解；它还具有一套完全显式的半熵刚性、阈值谱图册与单目标全局证书。
