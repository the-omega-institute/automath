\begin{theorem}[完整 determinant 的唯一最小忠实商]\label{thm:conclusion-artin-determinant-minimal-visible-quotient}
沿用正文第 \(13.1645\) 条的记号。
对群代数系数核
\[
\mathcal K\in \CC[G]^{V\times V}
\]
定义 determinant-活跃通道集
\[
\mathcal A_{\det}(\mathcal K)
:=
\left\{
\rho\in\widehat G:
\det(I-z\mathcal K_\rho)\not\equiv 1
\right\},
\]
并令
\[
N_{\det}(\mathcal K)
:=
\bigcap_{\rho\in\mathcal A_{\det}(\mathcal K)}\ker\rho,
\qquad
Q_{\det}(\mathcal K):=G/N_{\det}(\mathcal K).
\]
当 \(\mathcal A_{\det}(\mathcal K)=\varnothing\) 时约定 \(N_{\det}(\mathcal K)=G\)。
则：
\begin{enumerate}
  \item 若 \(q_{\det}:G\twoheadrightarrow Q_{\det}(\mathcal K)\) 为商映射，则
  \[
  \det(I-z,q_{\det\ast}\mathcal K)=\det(I-z,\mathcal K).
  \]
  \item 对任意有限商 \(\pi:G\twoheadrightarrow H\)，下列两条等价：
  \[
  \det(I-z,\pi_\ast\mathcal K)=\det(I-z,\mathcal K),
  \]
  \[
  \ker\pi\subseteq N_{\det}(\mathcal K).
  \]
  \item 因而 \(Q_{\det}(\mathcal K)\) 是保持完整 Artin determinant 的唯一最小商：任意其他保持完整 determinant 的商都唯一再向下满射到 \(Q_{\det}(\mathcal K)\)。
\end{enumerate}
\end{theorem}
\leanverified{paper\_conclusion\_artin\_determinant\_minimal\_visible\_quotient}

\begin{proof}
由正文第 \(13.1645\) 条，
\[
\det(I-z,\pi_\ast\mathcal K)
=
\prod_{\rho\in \mathrm{Inf}_\pi(\widehat H)}
\det(I-z\mathcal K_\rho)^{d_\rho},
\]
其中 \(d_\rho=\dim\rho\)，而 \(\mathrm{Inf}_\pi(\widehat H)\subseteq\widehat G\) 正是经由 \(\pi\) inflation 上来的通道。

先取 \(\pi=q_{\det}\)。
若 \(\rho\in\mathcal A_{\det}(\mathcal K)\)，则由 \(N_{\det}(\mathcal K)\) 的定义有
\[
N_{\det}(\mathcal K)\subseteq \ker\rho,
\]
故 \(\rho\) 经由 \(Q_{\det}(\mathcal K)\) 因子化。
于是所有 determinant-活跃通道都出现在商后的因子化中。
反之，若某个 \(\rho\notin \mathrm{Inf}_{q_{\det}}(\widehat Q_{\det})\)，则它不属于活跃集，从而
\[
\det(I-z\mathcal K_\rho)\equiv 1.
\]
故商侧因子化与全因子化完全相同，得第 \(1\) 条。

再证第 \(2\) 条。
若 \(\ker\pi\subseteq N_{\det}(\mathcal K)\)，则对每个 \(\rho\in\mathcal A_{\det}(\mathcal K)\) 都有
\[
\ker\pi\subseteq \ker\rho,
\]
故 \(\rho\) 经由 \(H\) 因子化，全部活跃因子均被保留，于是 determinant 不变。

反过来，若
\[
\det(I-z,\pi_\ast\mathcal K)=\det(I-z,\mathcal K),
\]
则上式中被删去的因子乘积必须恒等于 \(1\)。
但每个因子都是常数项为 \(1\) 的多项式，故只有当每个被删去的因子自身恒等于 \(1\) 时才可能发生。
于是任何未 inflation 到 \(H\) 的通道 \(\rho\) 都必须满足
\[
\det(I-z\mathcal K_\rho)\equiv 1,
\]
即一切活跃通道都必须 inflation 到 \(H\)。
这正等价于
\[
\ker\pi
\subseteq
\bigcap_{\rho\in\mathcal A_{\det}(\mathcal K)}\ker\rho
=
N_{\det}(\mathcal K).
\]
第 \(3\) 条遂由商映射的泛性质立得。证毕。
\end{proof}

\begin{theorem}[determinant 与类函数证书的联合最小商]\label{thm:conclusion-artin-determinant-classfunction-joint-minimal-quotient}
设 \(\mathcal E\subset Cl(G)_0\) 为有限零均值类函数族。
把每个 \(\varepsilon\in\mathcal E\) 的不可约展开写为
\[
\varepsilon
=
\sum_{\rho\in\widehat G\setminus\{1\}}a_\rho(\varepsilon)\chi_\rho.
\]
定义类函数活跃通道集
\[
\mathcal A_{\mathrm{cls}}(\mathcal E)
:=
\left\{
\rho\in\widehat G\setminus\{1\}:
\exists\,\varepsilon\in\mathcal E,\ a_\rho(\varepsilon)\neq 0
\right\},
\]
联合活跃通道集
\[
\mathcal A_\ast(\mathcal K,\mathcal E)
:=
\mathcal A_{\det}(\mathcal K)\cup \mathcal A_{\mathrm{cls}}(\mathcal E),
\]
以及
\[
N_\ast(\mathcal K,\mathcal E)
:=
\bigcap_{\rho\in\mathcal A_\ast(\mathcal K,\mathcal E)}\ker\rho,
\qquad
Q_\ast(\mathcal K,\mathcal E):=G/N_\ast(\mathcal K,\mathcal E).
\]
则对任意有限商 \(\pi:G\twoheadrightarrow H\)，下列三条等价：
\begin{enumerate}
  \item \(\pi\) 同时保持完整 determinant 且使每个 \(\varepsilon\in\mathcal E\) 经由 \(H\) 因子化；
  \item \(\pi\) 同时保持完整 determinant 且对每个 \(\varepsilon\in\mathcal E\) 都有
  \[
  |\pi_\ast\varepsilon|_{L^2(H)}=|\varepsilon|_{L^2(G)};
  \]
  \item
  \[
  \ker\pi\subseteq N_\ast(\mathcal K,\mathcal E).
  \]
\end{enumerate}
特别地，\(Q_\ast(\mathcal K,\mathcal E)\) 是同时承载这两类信息的唯一最小商。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-artin-determinant-minimal-visible-quotient}，
保持完整 determinant 当且仅当
\[
\ker\pi\subseteq
\bigcap_{\rho\in\mathcal A_{\det}(\mathcal K)}\ker\rho.
\]

另一方面，正文第 \(13.3656\) 条给出：对任意零均值类函数 \(\varepsilon\)，它经由 \(\pi\) 因子化，等价于
\[
|\pi_\ast\varepsilon|_{L^2(H)}=|\varepsilon|_{L^2(G)},
\]
又等价于其展开中所有出现的不可约通道都 inflation 到 \(H\)，亦即
\[
\ker\pi
\subseteq
\bigcap_{a_\rho(\varepsilon)\neq 0}\ker\rho.
\]
对全部 \(\varepsilon\in\mathcal E\) 同时取交，得到
\[
\ker\pi
\subseteq
\bigcap_{\rho\in\mathcal A_{\mathrm{cls}}(\mathcal E)}\ker\rho.
\]
再与 determinant 部分合并，便得
\[
\ker\pi
\subseteq
\bigcap_{\rho\in\mathcal A_\ast(\mathcal K,\mathcal E)}\ker\rho
=
N_\ast(\mathcal K,\mathcal E).
\]
反向推出同理成立，唯一最小性遂由商的泛性质给出。证毕。
\end{proof}

\begin{proposition}[Artin-visible hull 的矩阵实现]\label{prop:conclusion-artin-visible-hull-matrix-realization}
\leanverified{paper\_conclusion\_artin\_visible\_hull\_matrix\_realization}
在定理 \ref{thm:conclusion-artin-determinant-classfunction-joint-minimal-quotient} 的记号下，令
\[
\Pi_\ast(\mathcal K,\mathcal E)
:=
\bigoplus_{\rho\in\mathcal A_\ast(\mathcal K,\mathcal E)}\rho.
\]
则有自然同构
\[
Q_\ast(\mathcal K,\mathcal E)
\cong
\Pi_\ast(\mathcal K,\mathcal E)(G)
\subseteq
\prod_{\rho\in\mathcal A_\ast(\mathcal K,\mathcal E)}GL(V_\rho).
\]
\end{proposition}

\begin{proof}
直接由表示核的交公式，
\[
\ker \Pi_\ast(\mathcal K,\mathcal E)
=
\bigcap_{\rho\in\mathcal A_\ast(\mathcal K,\mathcal E)}\ker\rho
=
N_\ast(\mathcal K,\mathcal E).
\]
故由第一同构定理，
\[
G/N_\ast(\mathcal K,\mathcal E)\cong \Pi_\ast(\mathcal K,\mathcal E)(G).
\]
证毕。
\end{proof}

\begin{corollary}[阿贝尔化无损的充要条件]\label{cor:conclusion-artin-visible-hull-abelianization-criterion}
设
\[
q_{\mathrm{ab}}:G\twoheadrightarrow G^{\mathrm{ab}}:=G/[G,G].
\]
则下列三条等价：
\begin{enumerate}
  \item \(q_{\mathrm{ab}}\) 同时保持 \(\mathcal K\) 的完整 determinant 与 \(\mathcal E\) 中每个零均值类函数的全部 \(L^2\)-能量；
  \item \(\mathcal A_\ast(\mathcal K,\mathcal E)\) 中每个不可约通道均为一维；
  \item Artin-visible hull \(Q_\ast(\mathcal K,\mathcal E)\) 为阿贝尔群。
\end{enumerate}
\end{corollary}

\begin{proof}
有限群中，恰有一维不可约表示能够经由阿贝尔化商 inflation。
故由定理 \ref{thm:conclusion-artin-determinant-classfunction-joint-minimal-quotient}，
\[
q_{\mathrm{ab}}\ \text{无损}
\iff
\text{全部活跃通道皆可 inflation 到 } G^{\mathrm{ab}}
\iff
\dim\rho=1\quad(\forall \rho\in\mathcal A_\ast).
\]
这证明了 \((1)\Leftrightarrow(2)\)。

若 \((2)\) 成立，则 \(\Pi_\ast(\mathcal K,\mathcal E)\) 是若干一维角色的直和，其像必为阿贝尔群；由命题 \ref{prop:conclusion-artin-visible-hull-matrix-realization} 得 \((2)\Rightarrow(3)\)。

反之，若 \(Q_\ast(\mathcal K,\mathcal E)\) 阿贝尔，则任何经由 \(Q_\ast(\mathcal K,\mathcal E)\) 因子化的不可约表示都必须是一维。由全部活跃通道都经由 \(Q_\ast(\mathcal K,\mathcal E)\) 因子化，得 \((3)\Rightarrow(2)\)。证毕。
\end{proof}

\begin{theorem}[极大通道最优见证的唯一最小承载商]\label{thm:conclusion-artin-extremal-witness-minimal-hull}
固定 \(u\in(0,1]\)，并处于正文第 \(13.3657\) 条的假设下。
令
\[
\psi_{\ast,u}\in Cl(G)_0
\]
为唯一满足
\[
c_A=\langle 1_A,\psi_{\ast,u}\rangle_{Cl(G)}
\qquad
(A\subseteq G\ \text{为共轭类并})
\]
的零均值类函数。
定义
\[
\mathcal A_w(u)
:=
\left\{
\rho\in\widehat G\setminus\{1\}: a_\rho(\psi_{\ast,u})\neq 0
\right\},
\qquad
N_w(u):=\bigcap_{\rho\in\mathcal A_w(u)}\ker\rho,
\]
并记
\[
Q_w(u):=G/N_w(u),
\qquad
q_w:G\twoheadrightarrow Q_w(u).
\]
则：
\begin{enumerate}
  \item \(\psi_{\ast,u}\) 唯一因子化为
  \[
  \psi_{\ast,u}=\bar\psi_{\ast,u}\circ q_w
  \qquad
  (\bar\psi_{\ast,u}\in Cl(Q_w(u))_0);
  \]
  \item 对每个 \(\theta\in\RR\)，正文第 \(13.3657\) 条中的极值类并
  \[
  A_\theta=\bigcup_{\Re(e^{-i\theta}\psi_{\ast,u}(C))>0} C
  \]
  都是某个商层类并 \(\bar A_\theta\subseteq Q_w(u)\) 的拉回：
  \[
  A_\theta=q_w^{-1}(\bar A_\theta);
  \]
  \item 任意使 \(\psi_{\ast,u}\) 因子化的有限商 \(G\twoheadrightarrow H\) 都唯一再向下满射到 \(Q_w(u)\)。
\end{enumerate}
\end{theorem}

\begin{proof}
把正文第 \(13.3656\) 条直接应用于零均值类函数 \(\psi_{\ast,u}\)。
其最小类函数商正是
\[
G/K(\psi_{\ast,u}),
\qquad
K(\psi_{\ast,u})
=
\bigcap_{a_\rho(\psi_{\ast,u})\neq 0}\ker\rho
=
N_w(u).
\]
故第 \(1\) 条与第 \(3\) 条立即成立。

再证第 \(2\) 条。
由第 \(1\) 条，
\[
\psi_{\ast,u}(g)=\bar\psi_{\ast,u}(q_w(g)).
\]
因此
\[
\Re(e^{-i\theta}\psi_{\ast,u}(g))>0
\iff
\Re(e^{-i\theta}\bar\psi_{\ast,u}(q_w(g)))>0.
\]
于是 \(A_\theta\) 恰是
\[
\bar A_\theta:=
\left\{
h\in Q_w(u):
\Re(e^{-i\theta}\bar\psi_{\ast,u}(h))>0
\right\}
\]
的拉回。
由于 \(\bar\psi_{\ast,u}\) 为类函数，\(\bar A_\theta\) 是 \(Q_w(u)\) 中的共轭类并。证毕。
\end{proof}

\begin{theorem}[可见壳的融合律]\label{thm:conclusion-artin-visible-hull-fusion-law}
\leanverified{paper\_conclusion\_artin\_visible\_hull\_fusion\_law}
设两组数据
\[
D_1=(\mathcal K^{(1)},\mathcal E_1),
\qquad
D_2=(\mathcal K^{(2)},\mathcal E_2)
\]
定义在同一有限群 \(G\) 上。
分别记其 Artin-visible hull 为
\[
Q_i:=Q_\ast(D_i)=G/N_i
\qquad (i=1,2).
\]
把联合数据记为
\[
D_1\vee D_2:=
(\mathcal K^{(1)},\mathcal K^{(2)},\mathcal E_1\cup\mathcal E_2).
\]
则有自然同构
\[
Q_\ast(D_1\vee D_2)
\cong
G/(N_1\cap N_2)
\cong
\operatorname{Im}\bigl(G\to Q_1\times Q_2\bigr).
\]
\end{theorem}

\begin{proof}
由定义，联合数据的活跃通道集正是两者活跃通道集的并，因此其核交为
\[
N_\ast(D_1\vee D_2)=N_1\cap N_2.
\]
另一方面，自然乘积映射
\[
G\to Q_1\times Q_2
\]
的核恰为 \(N_1\cap N_2\)。
故由第一同构定理，
\[
G/(N_1\cap N_2)\cong \operatorname{Im}(G\to Q_1\times Q_2).
\]
证毕。
\end{proof}
