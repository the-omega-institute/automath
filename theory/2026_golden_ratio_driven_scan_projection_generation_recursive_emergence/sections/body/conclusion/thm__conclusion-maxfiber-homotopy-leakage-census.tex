\begin{theorem}[最大纤维同伦型的模 \(6\) 相图]\label{thm:conclusion-maxfiber-mod6-homotopy-phase-diagram}
\leanverified{paper\_conclusion\_maxfiber\_mod6\_homotopy\_phase\_diagram}
记
\[
M_m:=\{x\in X_m:d_m(x)=D_m\},
\qquad
D_m:=\max_{x\in X_m}d_m(x).
\]
则对任意 \(x\in M_m\) 有如下严格分裂：
\begin{enumerate}
\normalfont
  \item 若 \(m\) 为奇数，则 \(|K_x|\) 可缩；
  \item 若 \(m\equiv 2\pmod 6\)，则 \(|K_x|\) 可缩；
  \item 若 \(m\equiv 0,4\pmod 6\)，则 \(|K_x|\) 非可缩，且同伦等价于某个球面 \(S^{\tau(x)-1}\)。
\end{enumerate}
\end{theorem}
\begin{proof}
纤维独立集复形的同伦理论已给出
\[
|K_x|\ \text{可缩}
\iff
d_m(x)\ \text{为偶数},
\qquad
|K_x|\simeq S^{\tau(x)-1}
\iff
d_m(x)\ \text{为奇数}.
\]
另一方面，最大纤维闭式满足
\[
D_{2k}=F_{k+2},
\qquad
D_{2k+1}=2F_{k+1}.
\]
故奇数窗上 \(D_{2k+1}\) 恒为偶数，从而全部最大达到者都对应可缩型。

再用 Fibonacci 奇偶周期
\[
F_n\ \text{偶}\iff 3\mid n
\]
可得
\[
D_{2k}=F_{k+2}\ \text{偶}
\iff
3\mid(k+2)
\iff
2k\equiv 2\pmod 6.
\]
于是偶数窗恰在 \(m\equiv 2\pmod 6\) 时可缩，而在 \(m\equiv 0,4\pmod 6\) 时为球面型。证毕。
\end{proof}

\begin{theorem}[最大相位的信息泄漏与同伦复杂度严格脱钩]\label{thm:conclusion-maxfiber-leakage-decouples-from-homotopy}
\leanverified{paper\_conclusion\_maxfiber\_leakage\_decouples\_from\_homotopy}
记
\[
H_\varphi:=(3-\varphi)\log_2\varphi,
\qquad
I_{\mathrm{even}}:=1-H_\varphi,
\qquad
I_{\mathrm{odd}}:=\frac{1-H_\varphi}{2}.
\]
则在稳定阈值区间内同时成立：
\begin{enumerate}
\normalfont
  \item 对所有充分大的偶数窗，最大相位在均匀平均下都满足隐藏位精确无偏，但其相位--隐藏位互信息趋于正的常数 \(I_{\mathrm{even}}\)；
  \item 当 \(m\equiv 2\pmod 6\) 时，同一偶数窗最大相位全体对应可缩型；当 \(m\equiv 0,4\pmod 6\) 时，同一互信息极限 \(I_{\mathrm{even}}\) 仍出现于球面型；
  \item 对所有充分大的奇数窗，最大达到者全体都可缩，但其互信息仍趋于正的常数 \(I_{\mathrm{odd}}\)。
\end{enumerate}
因此，最大相位层上的信息泄漏大小不是同伦非可缩性的函数；同一泄漏常数 \(I_{\mathrm{even}}\) 同时出现在球面型与可缩型偶数窗，而严格正的 \(I_{\mathrm{odd}}\) 甚至完全发生在可缩相。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-maxfiber-hiddenbit-unbiased-marginal-biased-correlation}，对充分大的偶数窗与奇数窗分别有
\[
I(X;B)\longrightarrow 1-H_\varphi=I_{\mathrm{even}},
\qquad
I(X;B)\longrightarrow \frac{1-H_\varphi}{2}=I_{\mathrm{odd}}>0,
\]
并且两支的边缘分布都满足 \(\PP(B=1)=1/2\)。

另一方面，定理 \ref{thm:conclusion-maxfiber-mod6-homotopy-phase-diagram} 已给出最大达到者的同伦型：奇数窗全可缩；偶数窗在 \(m\equiv 2\pmod 6\) 时全可缩，在 \(m\equiv 0,4\pmod 6\) 时全为球面型。将二者并置便得结论。证毕。
\end{proof}

\begin{theorem}[奇数窗最大相位的 \texorpdfstring{$\gcd$}{gcd} 算术层析]\label{thm:conclusion-maxfiber-odd-gcd-determines-leakage-type}
\leanverified{paper\_conclusion\_maxfiber\_odd\_gcd\_determines\_leakage\_type}
设 \(m=2k+1\ge 13\)，取 \(x\in M_m\)，并定义
\[
g(x):=\gcd\bigl(d_{m,0}(x),d_{m,1}(x)\bigr),
\qquad
\ell(x):=1-H_b\!\left(\frac{d_{m,1}(x)}{D_m}\right).
\]
则存在严格二分：
\begin{enumerate}
\normalfont
  \item 若
  \[
  g(x)=F_{k+1},
  \]
  则
  \[
  (d_{m,0}(x),d_{m,1}(x))=(F_{k+1},F_{k+1}),
  \qquad
  p_1(x)=\frac12,
  \qquad
  \ell(x)=0;
  \]
  \item 若
  \[
  g(x)=2,
  \]
  则
  \[
  (d_{m,0}(x),d_{m,1}(x))
  \in
  \{(2F_{k-1},2F_k),(2F_k,2F_{k-1})\},
  \]
  从而
  \[
  \{p_1(x),1-p_1(x)\}
  =
  \left\{
  \frac{F_k}{F_{k+1}},
  \frac{F_{k-1}}{F_{k+1}}
  \right\},
  \qquad
  \ell(x)=1-H_b\!\left(\frac{F_k}{F_{k+1}}\right).
  \]
\end{enumerate}
故在奇数窗上，单个算术不变量 \(g(x)\) 已足以恢复最大相位的全部隐藏位信息型。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-maxfiber-hiddenbit-gcd-phase-classification}，奇数稳定相中的最大达到者恰分成两类：平衡相满足
\[
\gcd(d_{m,0}(x),d_{m,1}(x))=F_{k+1},
\qquad
(d_{m,0}(x),d_{m,1}(x))=(F_{k+1},F_{k+1}),
\]
而偏置相满足
\[
\gcd(d_{m,0}(x),d_{m,1}(x))=2,
\qquad
(d_{m,0}(x),d_{m,1}(x))
\in
\{(2F_{k-1},2F_k),(2F_k,2F_{k-1})\}.
\]
再由推论 \ref{cor:conclusion-maxfiber-gcd-determines-hiddenbit-entropy-class}，前一类满足
\[
p_1(x)=\frac12,
\qquad
H_b(x)=1,
\]
后一类满足
\[
\left\{p_1(x),1-p_1(x)\right\}
=
\left\{
\frac{F_k}{F_{k+1}},
\frac{F_{k-1}}{F_{k+1}}
\right\},
\qquad
H_b(x)=H_b\!\left(\frac{F_k}{F_{k+1}}\right).
\]
把 \(\ell(x)=1-H_b(x)\) 代入即可。证毕。
\end{proof}

\begin{proposition}[奇偶泄漏常数的精确二分之一律来自相位普查]\label{prop:conclusion-maxfiber-half-law-from-phase-census}
定义最大相位上的点态泄漏
\[
\ell_m(x):=1-H_b\!\left(\frac{d_{m,1}(x)}{D_m}\right).
\]
则在稳定阈值区间内有
\[
\frac1{|M_{2k}|}\sum_{x\in M_{2k}}\ell_{2k}(x)
=
1-H_b\!\left(\frac{F_k}{F_{k+2}}\right),
\]
\[
\frac1{|M_{2k+1}|}\sum_{x\in M_{2k+1}}\ell_{2k+1}(x)
=
\frac12\left(
1-H_b\!\left(\frac{F_k}{F_{k+1}}\right)
\right).
\]
特别地，
\[
\lim_{k\to\infty}
\frac{
\frac1{|M_{2k+1}|}\sum_{x\in M_{2k+1}}\ell_{2k+1}(x)
}{
\frac1{|M_{2k}|}\sum_{x\in M_{2k}}\ell_{2k}(x)
}
=\frac12.
\]
这一定比值并非来自局部偏置更弱，而是来自相位普查：偶数窗全部相位都是偏置相，而奇数窗四相中只有两相偏置、两相完全平衡。
\end{proposition}
\leanverified{paper\_conclusion\_maxfiber\_half\_law\_from\_phase\_census}
\begin{proof}
由定理 \ref{thm:conclusion-window6-even-maxfiber-hiddenbit-exact-fairization}，充分大的偶数窗上极大达到者恰有两相，且两相 hidden-bit 分裂互补，所以两相具有同一泄漏值
\[
\ell_{2k}(x)=1-H_b\!\left(\frac{F_k}{F_{k+2}}\right).
\]
于是偶数窗平均泄漏即为该共同值。

对奇数窗，由定理 \ref{thm:conclusion-odd-maxfiber-hiddenbit-tristate-crystal} 与定理 \ref{thm:conclusion-maxfiber-hiddenbit-gcd-phase-classification}，四个最大相位中恰有两相是平衡相、两相是偏置相。由定理 \ref{thm:conclusion-maxfiber-odd-gcd-determines-leakage-type}，平衡相泄漏为 \(0\)，偏置相泄漏共同等于
\[
1-H_b\!\left(\frac{F_k}{F_{k+1}}\right).
\]
因此
\[
\frac1{|M_{2k+1}|}\sum_{x\in M_{2k+1}}\ell_{2k+1}(x)
=
\frac{0+0+\ell_{\mathrm{bias}}+\ell_{\mathrm{bias}}}{4}
=
\frac12\,\ell_{\mathrm{bias}}
=
\frac12\left(
1-H_b\!\left(\frac{F_k}{F_{k+1}}\right)
\right).
\]
最后，由
\[
\frac{F_k}{F_{k+2}}\to \varphi^{-2},
\qquad
\frac{F_k}{F_{k+1}}\to \varphi^{-1},
\]
以及
\[
H_b(\varphi^{-2})=H_b(\varphi^{-1})=H_\varphi,
\]
可得所示极限比值为 \(1/2\)。证毕。
\end{proof}
