\subsection{\texorpdfstring{$q$}{q}-重折叠信道的 Choi 秩、单点塌缩困难性与碰撞核维度刚性}\label{subsec:conclusion-qcollision-choi-rank-sharpp-kernel-dimension-pdet}
本小节在附录关于高阶碰撞矩、折叠信道 Choi 秩与复杂度分界的既有结果之上，给出一组统一的终态命题：其核心是把 \(q\)-重碰撞矩、测量--制备信道的谱阶数量与受限结构下的计算困难性压缩到同一可反演接口。

\begin{theorem}[\texorpdfstring{$q$}{q}-重折叠信道的 Choi 秩等于 \texorpdfstring{$q$}{q}-次碰撞矩]\label{thm:conclusion-qfold-channel-choi-rank-equals-qcollision}
\leanverified{paper\_conclusion\_qfold\_channel\_choi\_rank\_equals\_qcollision}
设 \(\Fold:\Omega\twoheadrightarrow X\) 为有限满射，记纤维多重度
\[
d(x):=\card{\Fold^{-1}(x)}\qquad(x\in X).
\]
令 \(H:=\CC^\Omega\)，并以标准正交基定义纤维投影
\[
\Pi_x:=\sum_{\omega:\ \Fold(\omega)=x}\ket{\omega}\bra{\omega}\in\mathcal B(H).
\]
固定整数 \(q\ge 2\)，定义完全正迹保持映射
\[
\mathcal E_{\Fold}^{(q)}:\mathcal B(H)\to\mathcal B\!\left(H^{\otimes(q-1)}\right),\qquad
\mathcal E_{\Fold}^{(q)}(\rho):=\sum_{x\in X}\Tr(\Pi_x\rho)\,\frac{\Pi_x^{\otimes(q-1)}}{d(x)^{q-1}}.
\]
则其 Choi 矩阵满足
\[
\rank J\!\left(\mathcal E_{\Fold}^{(q)}\right)=\sum_{x\in X}d(x)^q=:S_q(\Fold).
\]
\end{theorem}
\begin{proof}
由 Choi 定义（输入端参考基取 \(\{\ket{\omega}\}_{\omega\in\Omega}\)）与
\(
\mathcal E_{\Fold}^{(q)}(\ket{\omega}\bra{\omega'})
=0
\)
（当 \(\omega\neq\omega'\)）可得
\[
J\!\left(\mathcal E_{\Fold}^{(q)}\right)
=
\sum_{x\in X}\frac{\Pi_x^{\otimes(q-1)}}{d(x)^{q-1}}\otimes \Pi_x^{\mathsf T}.
\]
由于 \(\Pi_x\) 在标准基下为实对角投影，\(\Pi_x^{\mathsf T}=\Pi_x\)；且不同 \(x\) 的支持子空间两两正交，故
\[
\rank J\!\left(\mathcal E_{\Fold}^{(q)}\right)
=\sum_{x\in X}\rank\!\left(\Pi_x^{\otimes(q-1)}\otimes\Pi_x\right)
=\sum_{x\in X}\rank(\Pi_x)^{q-1}\rank(\Pi_x)
=\sum_{x\in X}d(x)^q.
\]
证毕。
\end{proof}

\begin{corollary}[Bayes--InfoNCE 在 \texorpdfstring{$K=2$}{K=2} 时等价于二次碰撞矩与 Choi 秩]\label{cor:conclusion-bayes-infonce2-equals-qcollision2-choi-rank}
设 \(\Fold:\Omega\twoheadrightarrow X\) 为有限满射，\(\Omega\) 上取均匀分布，记 \(N:=|\Omega|\) 与输出分布
\[
w(x):=\PP(\Fold(U)=x)=\frac{d(x)}{N},\qquad U\sim\mathrm{Unif}(\Omega).
\]
则 Bayes--InfoNCE 的最优值在 \(K=2\) 时满足
\[
L_2^\star(\Fold)
=(\log 2)\sum_{x\in X}w(x)^2
=(\log 2)\,\frac{S_2(\Fold)}{N^2}
=(\log 2)\,\frac{\rank J\!\left(\mathcal E_{\Fold}^{(2)}\right)}{N^2}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-infonce-partition-optimal-closed-form}，\(L_2^\star(\Fold)=\EE[\log(1+J)]\)，且
\(
J\mid(\Fold(U)=x)\sim\mathrm{Binomial}(1,w(x))
\)。
故
\(
\EE[\log(1+J)\mid \Fold(U)=x]=w(x)\log 2
\)，对 \(x\) 取期望即得第一式。
其余两式由 \(w(x)=d(x)/N\) 与定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 在 \(q=2\) 时给出的
\(
\rank J(\mathcal E_{\Fold}^{(2)})=S_2(\Fold)=\sum_x d(x)^2
\)
立得。证毕。
\end{proof}

\begin{theorem}[固定阶 \texorpdfstring{$q\ge 2$}{q>=2} 时碰撞矩与对应 Choi 秩计算的 \texorpdfstring{\(\#\mathrm{P}\)}{\#P} 完备性]\label{thm:conclusion-fixed-q-collision-choi-rank-sharpp-complete}
对任意固定整数 \(q\ge 2\)，给定电路 \(C:\{0,1\}^n\to\{0,1\}^n\)，定义
\[
S_q(C):=\sum_{y\in\{0,1\}^n}\card{C^{-1}(y)}^q.
\]
则函数问题 \(C\mapsto S_q(C)\) 为 \(\#\mathrm P\)-完备。
并且令 \(\mathcal E_C^{(q)}\) 为定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 对 \(\Fold=C\) 的对应信道，则
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)=S_q(C),
\]
从而 \(C\mapsto \rank J(\mathcal E_C^{(q)})\) 亦为 \(\#\mathrm P\)-完备。
\end{theorem}
\begin{proof}
第一断言即附录定理 \ref{thm:circuit-s2-sharpP-complete}。
第二断言由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 直接代入 \(d(y)=\card{C^{-1}(y)}\) 得到。
两者合并即得结论。证毕。
\end{proof}

\begin{theorem}[限制到单点塌缩电路后 \texorpdfstring{\(\#\mathrm P\)}{\#P} 完备性保持]\label{thm:conclusion-single-collapse-collision-sharpp-complete}
固定整数 \(q\ge 2\)。
称电路 \(C:\{0,1\}^n\to\{0,1\}^n\) 为单点塌缩型，若存在唯一 \(y_0\) 使 \(\card{C^{-1}(y_0)}\) 可能大于 \(1\)，且对 \(y\neq y_0\) 有 \(\card{C^{-1}(y)}\in\{0,1\}\)。
则在该受限类上计算 \(S_q(C)\) 仍为 \(\#\mathrm P\)-完备。
\end{theorem}
\begin{proof}
\(\#\mathrm P\) 属于性与定理 \ref{thm:conclusion-fixed-q-collision-choi-rank-sharpp-complete} 相同。

\(\#\mathrm P\)-困难性取附录定理 \ref{thm:circuit-sk-fixedk-sharpP-complete} 证明中的同一归约电路 \(C_{\varphi'}\)：它把所有满足赋值塌缩到单点 \(0^{n+1}\)，其余赋值保持恒等输出，因此属于单点塌缩类。
按该证明记 \(N:=2^{n+1}\)、\(s:=2\,\#\mathrm{SAT}(\varphi)\in\{0,2,\dots,N\}\)，并有
\[
S_q(C_{\varphi'})=s^q+(N-s).
\]
在偶数集合上定义 \(g_q(t):=t^q+(N-t)\)。
对任意偶数 \(t\in[0,N-2]\)，
\[
g_q(t+2)-g_q(t)=(t+2)^q-t^q-2\ge 2^q-2>0,
\]
故 \(g_q\) 在 \(\{0,2,\dots,N\}\) 上严格单调。
因此给定 \(S_q(C_{\varphi'})\) 可二分恢复唯一偶数 \(s\)，进而恢复 \(\#\mathrm{SAT}(\varphi)=s/2\)。
故受限类仍为 \(\#\mathrm P\)-困难，结论成立。证毕。
\end{proof}

\begin{theorem}[单点塌缩信道的 \texorpdfstring{$q$}{q}-Choi 秩单参数化]\label{thm:conclusion-single-collapse-choi-rank-one-parameterization}
固定整数 \(q\ge 2\)。
设 \(C:\{0,1\}^N\to\{0,1\}^N\) 为单点塌缩型电路，即存在唯一输出点 \(y_0\) 使其纤维大小可能大于 \(1\)，其余非空纤维大小均属于 \(\{0,1\}\)。
记
\[
t:=\card{C^{-1}(y_0)}.
\]
则 \(q\)-重折叠信道的 Choi 秩满足精确闭式
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)=t^q+(2^N-t).
\]
特别地，在固定 \(N,q\) 下，该秩仅由单参数 \(t\) 决定。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision}，
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)=\sum_{y}\card{C^{-1}(y)}^q.
\]
在单点塌缩型假设下，唯一可能大于 \(1\) 的纤维大小为 \(t\)，而其余非空纤维全为单点，总数即 \(2^N-t\)。
因此
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)=t^q+(2^N-t).
\]
证毕。
\end{proof}

\begin{theorem}[极端几何退化下的 \texorpdfstring{\(\#\mathrm P\)}{\#P}-完备性保持]\label{thm:conclusion-single-collapse-choi-rank-sharpp-persists}
固定整数 \(q\ge 2\)。
即使限制在单点塌缩型电路类上，计算
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)
\]
仍然是 \(\#\mathrm P\)-完备的。
更具体地，在定理 \ref{thm:conclusion-single-collapse-collision-sharpp-complete} 的归约中，存在单点塌缩型电路 \(C_{\varphi'}\) 与偶整数
\[
s:=2\,\#\mathrm{SAT}(\varphi)
\]
使得
\[
\rank J\!\left(\mathcal E_{C_{\varphi'}}^{(q)}\right)=s^q+(2^{n+1}-s),
\]
且该表达式在偶数格点上严格单调，故单个 Choi 秩值已足以唯一恢复 \(\#\mathrm{SAT}(\varphi)\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-single-collapse-collision-sharpp-complete} 的同一归约，单点塌缩型电路 \(C_{\varphi'}\) 满足
\[
S_q(C_{\varphi'})=s^q+(N-s),
\qquad
N:=2^{n+1},
\qquad
s:=2\,\#\mathrm{SAT}(\varphi)\in\{0,2,\dots,N\}.
\]
再由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 与定理 \ref{thm:conclusion-single-collapse-choi-rank-one-parameterization},
\[
\rank J\!\left(\mathcal E_{C_{\varphi'}}^{(q)}\right)=s^q+(N-s).
\]
对偶数格点定义
\[
g_q(t):=t^q+(N-t).
\]
则对任意偶数 \(t\in[0,N-2]\),
\[
g_q(t+2)-g_q(t)=(t+2)^q-t^q-2>0
\qquad(q\ge2),
\]
故 \(g_q\) 在 \(\{0,2,\dots,N\}\) 上严格递增。
于是给定单个 Choi 秩值即可唯一恢复 \(s\)，进而恢复
\[
\#\mathrm{SAT}(\varphi)=\frac{s}{2}.
\]
因此在该受限类上计算 Choi 秩仍为 \(\#\mathrm P\)-完备。证毕。
\end{proof}

\begin{theorem}[parity-poor 与 Choi-rank-rich 的严格正交]\label{thm:conclusion-parity-poor-choi-rank-rich-orthogonal}
记 \(N_m\) 为折叠正常化子，\(\alpha_m\) 为定理 \ref{thm:conclusion-fold-normalizer-abelianization-layered-characters} 中的分层奇偶参数。
则全部 \(\{\pm1\}\)-值特征满足
\[
\Hom(N_m,\{\pm1\})\cong (\ZZ/2\ZZ)^{\alpha_m},
\]
并且该一维角色代数只依赖纤维规模分层
\(
\{N_{m,d}\}_{d\ge 1}
\)。

与此相对，对任意有限满射 \(\Fold:\Omega\twoheadrightarrow X\) 及固定 \(q\ge 2\)，\(q\)-重折叠信道满足
\[
\rank J\!\left(\mathcal E_{\Fold}^{(q)}\right)=\sum_{x\in X}d(x)^q.
\]
当 \(\Fold=C\) 取为布尔电路时，函数
\[
C\longmapsto \rank J\!\left(\mathcal E_C^{(q)}\right)
\]
为 \(\#\mathrm P\)-完备；并且即使限制到单点塌缩子类，仍有
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)=t^q+(2^N-t),
\]
其中 \(t\) 为唯一坏纤维大小，而且该受限问题仍保持 \(\#\mathrm P\)-完备。

故在最强受限情形下仍存在严格分裂：正常化子的一维 parity 审计是贫乏的，因为它只能看见一个有界秩 elementary \(2\)-group；而 \(q\)-Choi 秩方向是富足的，因为它在唯一坏纤维极限下仍保留 \(\#\mathrm P\)-级信息。
\end{theorem}
\begin{proof}
前半部分正是定理 \ref{thm:conclusion-fold-normalizer-characters-completely-stratified} 的内容：全部 \(\{\pm1\}\)-值特征唯一经由
\[
N_m\twoheadrightarrow (\ZZ/2\ZZ)^{\alpha_m}
\]
因子化，且该交换化完全由纤维规模分层 \(\{N_{m,d}\}_{d\ge 1}\) 决定。

后半部分由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 立即给出
\[
\rank J\!\left(\mathcal E_{\Fold}^{(q)}\right)=\sum_{x\in X}d(x)^q.
\]
当 \(\Fold=C\) 为布尔电路时，定理 \ref{thm:conclusion-fixed-q-collision-choi-rank-sharpp-complete} 说明 \(C\mapsto \rank J(\mathcal E_C^{(q)})\) 为 \(\#\mathrm P\)-完备。
若再限制到单点塌缩子类，则定理 \ref{thm:conclusion-single-collapse-choi-rank-one-parameterization} 给出精确闭式
\[
\rank J\!\left(\mathcal E_C^{(q)}\right)=t^q+(2^N-t),
\]
而定理 \ref{thm:conclusion-single-collapse-choi-rank-sharpp-persists} 进一步说明该单参数情形仍为 \(\#\mathrm P\)-完备。

因此，一边是完全塌缩到分层 parity 直方图的有界秩 \(2\)-群证书，另一边是在唯一坏纤维极限下仍保留 \(\#\mathrm P\)-完备信息的单参数 Choi 秩。两者由不同不变量控制，不能互相替代。证毕。
\end{proof}

\begin{conjecture}[奇偶特征贫乏与 Choi 秩富足的正交分离]\label{conj:conclusion-parity-poor-choi-rank-rich-orthogonality}
设一族 fold-compatible 信道同时满足两条性质：
\begin{enumerate}
  \item 其正常化子的全部 \(\{\pm1\}\)-值特征经由某个有界秩 elementary \(2\)-group 因子化；
  \item 其允许单点塌缩型纤维直方图作为受限子类。
\end{enumerate}
则应出现如下普遍的正交分离现象：
\begin{itemize}
  \item 在正常化子的一维对称可见量方向上，系统是 parity-poor 的；
  \item 在 \(q\)-重折叠信道的 Choi 秩方向上，系统仍可携带 \(\#\mathrm P\)-完备的 rank-rich 信息。
\end{itemize}
结论 \ref{thm:conclusion-fold-normalizer-characters-completely-stratified}--\ref{cor:conclusion-fold-normalizer-layered-character-histogram-blindness} 与定理 \ref{thm:conclusion-single-collapse-choi-rank-one-parameterization}--\ref{thm:conclusion-single-collapse-choi-rank-sharpp-persists} 给出了这一原则的首个完整样本：一边是一维 parity 证书完全塌缩到纤维规模分层，另一边则是在唯一坏纤维极限下仍保持 \(\#\mathrm P\)-完备的单参数 Choi 秩。
\end{conjecture}

\begin{theorem}[二输出折叠下 Bayes--InfoNCE\texorpdfstring{$\,_2$}{2} 的显式闭式与指数精度近似屏障]\label{thm:conclusion-bayes-infonce2-sat-threshold-explicit}
给定含 \(n\) 个变量的布尔公式 \(\varphi\)，令
\(
\Omega_\varphi:=\{0,1\}^{n+1}
\)
并定义电路 \(C_\varphi:\Omega_\varphi\to\{0,1\}\) 为
\[
C_\varphi(b,x):=b\wedge \varphi(x),
\qquad
(b,x)\in\{0,1\}\times\{0,1\}^n.
\]
令 \(X_\varphi:=\operatorname{Im}(C_\varphi)\subseteq\{0,1\}\)，并把 \(C_\varphi\) 视为满射 \(C_\varphi:\Omega_\varphi\twoheadrightarrow X_\varphi\)。
记 \(N:=|\Omega_\varphi|=2^{n+1}\) 与 \(s:=\#\mathrm{SAT}(\varphi)\in[0,2^n]\)。
则 Bayes--InfoNCE 的最优值满足显式公式
\[
L_2^\star(C_\varphi)
=(\log 2)\left(\Bigl(\frac{s}{N}\Bigr)^2+\Bigl(1-\frac{s}{N}\Bigr)^2\right).
\]
并且：
\begin{enumerate}
  \item 若存在多项式时间算法输出 \(\widetilde L(\varphi)\) 使
  \[
  \bigl|\widetilde L(\varphi)-L_2^\star(C_\varphi)\bigr|<\frac{\log 2}{2N},
  \]
  则可在多项式时间内判定 \(\mathrm{SAT}\)。
  \item 若存在多项式时间算法输出 \(\widetilde L(\varphi)\) 使
  \[
  \bigl|\widetilde L(\varphi)-L_2^\star(C_\varphi)\bigr|<\frac{\log 2}{4N^2},
  \]
  则可在多项式时间内精确计算 \(\#\mathrm{SAT}(\varphi)\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由构造，输出 \(1\) 的纤维基数为
\(
\card{C_\varphi^{-1}(1)}=\#\{(1,x):\varphi(x)=1\}=s
\)，而输出 \(0\) 的纤维基数为 \(N-s\)。
于是由推论 \ref{cor:conclusion-bayes-infonce2-equals-qcollision2-choi-rank} 得
\[
L_2^\star(C_\varphi)
=(\log 2)\Bigl(\bigl(s/N\bigr)^2+\bigl((N-s)/N\bigr)^2\Bigr),
\]
即所示闭式。

为证明 (1)，注意当 \(s=0\) 时 \(L_2^\star(C_\varphi)=\log 2\)，而当 \(s\ge 1\) 时
\[
L_2^\star(C_\varphi)
\le (\log 2)\left(\Bigl(\frac{1}{N}\Bigr)^2+\Bigl(1-\frac{1}{N}\Bigr)^2\right)
=\log 2-(\log 2)\left(\frac{2}{N}-\frac{2}{N^2}\right).
\]
因此 \(s=0\) 与 \(s\ge 1\) 两情形的函数值间隔至少 \(\frac{\log 2}{N}\)（\(N\ge 2\)），从而误差 \(<\frac{\log 2}{2N}\) 足以区分可满足与不可满足。

对 (2)，令
\[
c_2(s):=\Bigl(\frac{s}{N}\Bigr)^2+\Bigl(1-\frac{s}{N}\Bigr)^2
=1-\frac{2s}{N}+\frac{2s^2}{N^2}.
\]
当 \(s\in[0,N/2]\) 时 \(c_2\) 单调递减，且对任意整数 \(0\le s\le N/2-1\) 有
\[
c_2(s)-c_2(s+1)=\frac{2(N-(2s+1))}{N^2}\ge \frac{2}{N^2}.
\]
因此若 \(|\widetilde c_2-c_2(s)|<\frac{1}{2N^2}\)，则可由单调性与离散间距唯一恢复整数 \(s\)。
而
\(
\widetilde c_2:=\widetilde L(\varphi)/\log 2
\)
满足
\(
|\widetilde c_2-c_2(s)|<\frac{1}{4N^2}
\)
时即强于上述阈值，故可多项式时间恢复 \(s=\#\mathrm{SAT}(\varphi)\)。
证毕。
\end{proof}

\begin{theorem}[纤维置换体积对数的常数绝对误差近似仍具 \texorpdfstring{\(\#\mathrm P\)}{\#P} 困难性]\label{thm:conclusion-fiber-permutation-logvolume-constant-error-sharpp-hard}
对有限电路/折叠 \(C:\Omega\to X\) 定义纤维置换群
\[
G_{\mathrm{fib}}(C):=\prod_{x\in X}\Sym\!\bigl(C^{-1}(x)\bigr),
\qquad
\abs{G_{\mathrm{fib}}(C)}=\prod_{x\in X} d(x)!.
\]
令
\[
V(C):=\log_2\abs{G_{\mathrm{fib}}(C)}.
\]
则对任意常数 \(\varepsilon\in(0,\frac12)\)，在单点塌缩电路类上，给定 \(C\) 输出 \(\widetilde V\) 使
\[
\abs{\widetilde V-V(C)}\le \varepsilon
\]
为 \(\#\mathrm P\)-困难。
\end{theorem}
\begin{proof}
仍取定理 \ref{thm:conclusion-single-collapse-collision-sharpp-complete} 的归约族 \(C_{\varphi'}\)。
设唯一非平凡纤维大小为 \(s=2\,\#\mathrm{SAT}(\varphi)\)，其余非空纤维大小均为 \(1\)。
则
\[
\abs{G_{\mathrm{fib}}(C_{\varphi'})}=s!,
\qquad
V(C_{\varphi'})=\log_2(s!).
\]
对任意偶数 \(t\ge 0\)，
\[
\log_2((t+2)!)-\log_2(t!)=\log_2\!\bigl((t+1)(t+2)\bigr)\ge 1.
\]
故 \(t\mapsto\log_2(t!)\) 在偶数集合 \(\{0,2,\dots,N\}\) 上相邻值间隔至少 \(1\)。
当近似绝对误差 \(<\frac12\) 时，可唯一恢复该偶数 \(s\)，进而恢复 \(\#\mathrm{SAT}(\varphi)=s/2\)。
因此该近似任务为 \(\#\mathrm P\)-困难。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{\(\#\)}{\#}ETH 下通用碰撞核矩阵幂表示的指数维度下界]\label{cor:conclusion-collision-kernel-matrix-power-exp-dimension-under-sharpeth}
假设 \(\#\)ETH（例如不存在 \(2^{o(n)}\) 时间算法计算 \(\#\text{\normalfont\textsc{3SAT}}\)）。
设存在统一编译器 \(\mathcal K\)，把任意 \(n\)-比特电路 \(C\) 映射到
\[
(A_C,u_C,v_C,T(n)),
\]
其中 \(A_C\in\ZZ_{\ge 0}^{d(n)\times d(n)}\)、\(u_C,v_C\in\ZZ^{d(n)}\)、\(T(n)=\poly(n)\)，并满足
\[
S_2(C)=u_C^{\mathsf T}A_C^{\,T(n)}v_C.
\]
则不可能有 \(d(n)=2^{o(n)}\)。
等价地，存在常数 \(c>0\) 与无穷多个 \(n\) 使
\[
d(n)\ge 2^{cn}.
\]
\end{corollary}
\begin{proof}
反设 \(d(n)=2^{o(n)}\)。
对 \(n\)-比特电路有 \(0\le S_2(C)\le 2^{2n}\)，故目标值仅需 \(O(n)\) 比特。
对每个 \(O(n)\)-比特素数 \(p\)，可在模 \(p\) 上以重复平方法计算
\(
u_C^{\mathsf T}A_C^{T(n)}v_C\bmod p
\)，
单模时间为 \(d(n)^{O(1)}\poly(n)=2^{o(n)}\)。
取足够多素数使乘积超过 \(2^{2n}\)，再由中国剩余定理恢复精确 \(S_2(C)\)，总时间仍为 \(2^{o(n)}\)。

由附录定理 \ref{thm:circuit-s2-sharpP-complete} 的归约模板（同其证明中的单点塌缩构造），\(\#\textsc{3SAT}\) 可在线性规模映射下归约到 \(S_2\) 计算，从而得到 \(2^{o(n)}\) 时间算法求解 \(\#\textsc{3SAT}\)，与 \(\#\)ETH 矛盾。
故结论成立。证毕。
\end{proof}

\begin{theorem}[Choi 伪行列式对数的显式公式与常数误差 \texorpdfstring{\(\#\mathrm P\)}{\#P} 困难性]\label{thm:conclusion-qfold-choi-pdet-log-sharpp-hard}
沿定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 的记号，定义
\[
\operatorname{pdet}(J):=\prod_{\lambda_i\neq 0}\lambda_i
\]
为 \(J=J(\mathcal E_{\Fold}^{(q)})\) 的伪行列式。
则成立显式恒等式
\[
-\log_2\operatorname{pdet}\!\left(J(\mathcal E_{\Fold}^{(q)})\right)
=(q-1)\sum_{x\in X} d(x)^q\log_2 d(x),
\]
其中约定 \(0\log_20:=0\)。

并且对任意固定 \(q\ge 2\)，在单点塌缩电路类上，以绝对误差 \(<\frac12\) 近似上述量为 \(\#\mathrm P\)-困难。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 的分块表达
\[
J=\sum_{x\in X}\frac{1}{d(x)^{q-1}}\,
\Pi_x^{\otimes(q-1)}\otimes \Pi_x.
\]
对固定 \(x\)，该块是标量 \(d(x)^{-(q-1)}\) 乘以秩 \(d(x)^q\) 投影，故其非零特征值全为 \(d(x)^{-(q-1)}\)，重数 \(d(x)^q\)。
不同 \(x\) 的块正交直和，因而
\[
\operatorname{pdet}(J)
=\prod_{x\in X}\left(d(x)^{-(q-1)}\right)^{d(x)^q}
=\prod_{x\in X} d(x)^{-(q-1)d(x)^q},
\]
取 \(-\log_2\) 即得显式式。

困难性仍取定理 \ref{thm:conclusion-single-collapse-collision-sharpp-complete} 的单点塌缩族 \(C_{\varphi'}\)，记 \(s=2\,\#\mathrm{SAT}(\varphi)\)。
此时
\[
\Psi_q(C_{\varphi'})
:=-\log_2\operatorname{pdet}\!\left(J(\mathcal E_{C_{\varphi'}}^{(q)})\right)
=(q-1)s^q\log_2 s
\]
（约定 \(s=0\) 时该值为 \(0\)）。

对 \(t\ge 2\) 设 \(F_q(t):=(q-1)t^q\log_2 t\)。
有
\[
F_q'(t)
=(q-1)t^{q-1}\!\left(q\log_2 t+\frac{1}{\ln 2}\right)
>0,
\]
故 \(F_q\) 严格递增；并且当 \(t\ge 2\) 时 \(F_q'(t)>1\)，从而
\[
F_q(t+2)-F_q(t)>2.
\]
又 \(F_q(2)-F_q(0)=(q-1)2^q\ge 4\)。
因此 \(\Psi_q\) 在偶数集合 \(\{0,2,\dots,N\}\) 上相邻值间隔 \(>1\)。
给定绝对误差 \(<\frac12\) 的近似值可唯一恢复偶数 \(s\)，进而恢复 \(\#\mathrm{SAT}(\varphi)=s/2\)。
故该近似问题为 \(\#\mathrm P\)-困难。证毕。
\end{proof}

\begin{theorem}[射影压力算子的整数多项式扇区精确闭合]\label{thm:conclusion-projective-pressure-polynomial-sector-exact-closure}
沿用定义 \ref{def:pom-projective-transfer-operator}、定义 \ref{def:pom-homogeneous-polynomials-simplex}、命题 \ref{prop:pom-projective-operator-invariant-homogeneous} 与定理 \ref{thm:pom-projective-operator-degenerates-to-moment-kernel} 的记号。对每个整数 \(q\ge 2\)，记
\[
\mathcal H_q:=\{p\vert_{\Delta}: p \text{ 为齐次次数 } q \text{ 多项式}\}.
\]
则 \(\mathcal H_q\) 在射影压力算子 \(\mathcal L_q\) 下严格不变，并满足分母完全消去公式
\[
(\mathcal L_q p)(w)=\sum_{b\in A_{\mathrm{out}}} p(B_b w)
\qquad (p\in \mathcal H_q).
\]
更进一步，
\[
\mathcal L_q\vert_{\mathcal H_q}=\widetilde A_q,
\qquad
\rho(\mathcal L_q\vert_{\mathcal H_q})=\rho(\widetilde A_q)=r_q,
\]
其中右端等号按任一单项式基下的矩阵表示理解。
因此，整数矩阶上的全部 projective thermodynamics 已在有限维多项式扇区内精确闭合，不存在任何“整数 \(q\) 可见而对称幂核不可见”的额外无限维谱贡献。
\end{theorem}

\begin{proof}
命题 \ref{prop:pom-projective-operator-invariant-homogeneous} 已证明：对任意 \(p\in \mathcal H_q\)，
\[
(\mathcal L_q p)(w)=\sum_{b\in A_{\mathrm{out}}}p(B_b w),
\]
从而 \(\mathcal H_q\) 严格不变，且所有分母在齐次次数 \(q\) 上完全消去。
再由定理 \ref{thm:pom-projective-operator-degenerates-to-moment-kernel}，
\[
\bigl[\mathcal L_q\vert_{\mathcal H_q}\bigr]_{(w^\nu)}=\widetilde A_q,
\qquad
\rho(\mathcal L_q\vert_{\mathcal H_q})=\rho(\widetilde A_q)=r_q.
\]
故整数矩阶的 projective pressure 数据恰由有限维对称幂 moment-kernel 完整承载。证毕。
\end{proof}

\begin{theorem}[矩阶层级 Hankel 秩的多项式驯化]\label{thm:conclusion-moment-hankel-rank-polynomial-taming}
设 \(K\) 为 \(N\)-状态有限核，\(S_q(m)\) 为其 \(q\)-阶碰撞幂和序列，\(d_q\) 为该序列的最小整系数递推阶，等价地其 Hankel 秩或最小 realization 维数。则对一切整数 \(q\ge 2\) 都有
\[
d_q\le \binom{q+N-1}{N-1}.
\]
特别地，
\[
\limsup_{q\to\infty}\frac{\log d_q}{q}=0.
\]
换言之，尽管朴素 \(q\)-重同步直积态空间呈指数爆炸，真实可见的矩阶复杂度在固定状态数 \(N\) 下只按多项式增长。
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-sympower-hankel-polynomial-upper-bound}，对每个固定 \(q\ge 2\)，存在维数不超过
\[
\binom{q+N-1}{N-1}
\]
的非负矩阵 realization 读出 \(S_q(m)\)。而同一定理已说明，最小整系数递推阶与 Hankel 秩满足同样上界，故
\[
d_q\le \binom{q+N-1}{N-1}.
\]
对固定 \(N\)，右端仅按 \(q^{N-1}\) 多项式增长，因此
\[
\log d_q\le (N-1)\log q+O(1),
\]
从而
\[
\limsup_{q\to\infty}\frac{\log d_q}{q}=0.
\]
证毕。
\end{proof}
