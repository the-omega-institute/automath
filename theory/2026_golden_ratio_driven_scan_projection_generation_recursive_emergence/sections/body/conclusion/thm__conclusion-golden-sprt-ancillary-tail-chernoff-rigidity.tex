\begin{theorem}[Esscher 对称化下的边界唯一破缺与停止壳层分裂]\label{thm:conclusion-golden-sprt-esscher-boundary-symmetry}
\leanverified{paper\_conclusion\_golden\_sprt\_esscher\_boundary\_symmetry}
令
\[
S_T:=\operatorname{sign}(K_{\tau_T})\in\{\pm1\},
\qquad
\tau_T:=\inf\{n\ge 0:\ |K_n|=T\},
\]
并记 \(\mathbb Q\) 为命题 \ref{prop:pom-golden-esscher-symmetrization} 中的公平 Esscher 测度。则对每个 \(T\ge 1\) 都有
\[
\mathbb Q(S_T=+1)=\mathbb Q(S_T=-1)=\frac12,
\qquad
\tau_T\perp S_T
\quad\text{在 }\mathbb Q\text{ 下}.
\]
更强地，对每个可达的 \(n\equiv T\pmod 2\) 与每个 \(s\in\{\pm1\}\)，有精确因子化
\[
\mathbb P_{\theta_\pm}(\tau_T=n,S_T=s)
=
\frac12\,\varphi^{\pm sT/2}
\left(\frac{2}{\varphi^{3/2}}\right)^n
\mathbb Q(\tau_T=n),
\]
等价地，
\[
\frac{d\mathbb P_{\theta_\pm}}{d\mathbb Q}\Big|_{\mathcal F_{\tau_T}}
=
\left(\frac{2}{\varphi^{3/2}}\right)^{\tau_T}\varphi^{\pm K_{\tau_T}/2}.
\]
因此，Esscher 去偏之后，停止实验只剩下终端符号的唯一边界破缺；时间壳层与符号壳层在 \(\mathbb Q\) 下严格分离。
\end{theorem}
\begin{proof}
命题 \ref{prop:pom-golden-esscher-symmetrization} 已给出
\[
\frac{d\mathbb Q}{d\mathbb P_{\theta_+}}\Big|_{\mathcal F_n}
=
\left(\frac{\varphi^{3/2}}{2}\right)^n\varphi^{-K_n/2},
\]
故在停止时刻反演得
\[
\frac{d\mathbb P_{\theta_+}}{d\mathbb Q}\Big|_{\mathcal F_{\tau_T}}
=
\left(\frac{2}{\varphi^{3/2}}\right)^{\tau_T}\varphi^{K_{\tau_T}/2}.
\]
另一方面，由定理 \ref{thm:conclusion-golden-sprt-terminal-sign-minimal-sufficiency} 的似然比公式，
\[
\frac{d\mathbb P_{\theta_-}}{d\mathbb P_{\theta_+}}\Big|_{\mathcal F_{\tau_T}}
=
\varphi^{-K_{\tau_T}}.
\]
与上式相乘得到
\[
\frac{d\mathbb P_{\theta_-}}{d\mathbb Q}\Big|_{\mathcal F_{\tau_T}}
=
\left(\frac{2}{\varphi^{3/2}}\right)^{\tau_T}\varphi^{-K_{\tau_T}/2}.
\]
在事件 \(\{\tau_T=n,S_T=s\}\) 上，\(K_{\tau_T}=sT\) 为常数，故
\[
\mathbb P_{\theta_\pm}(\tau_T=n,S_T=s)
=
\left(\frac{2}{\varphi^{3/2}}\right)^n\varphi^{\pm sT/2}
\mathbb Q(\tau_T=n,S_T=s).
\]

在 \(\mathbb Q\) 下，步进 \(Y_i:=2B_i-1\) 为简单对称随机游走。对任意属于事件 \(\{\tau_T=n,S_T=+1\}\) 的路径
\[
(y_1,\dots,y_n)\in\{\pm1\}^n,
\]
全局取反
\[
(y_1,\dots,y_n)\longmapsto (-y_1,\dots,-y_n)
\]
给出到 \(\{\tau_T=n,S_T=-1\}\) 的双射，并且两条路径在 \(\mathbb Q\) 下具有相同概率 \(2^{-n}\)。因此
\[
\mathbb Q(\tau_T=n,S_T=+1)
=
\mathbb Q(\tau_T=n,S_T=-1)
=
\frac12\,\mathbb Q(\tau_T=n),
\]
从而
\[
\mathbb Q(S_T=+1)=\mathbb Q(S_T=-1)=\frac12,
\qquad
\tau_T\perp S_T.
\]
将此代回上式即得联合因子化。证毕。
\end{proof}

\begin{theorem}[停止实验的一比特充分化与时间 ancillary]\label{thm:conclusion-golden-sprt-onebit-ancillary}
\leanverified{paper\_conclusion\_golden\_sprt\_onebit\_ancillary}
在等先验
\[
\mathbb P(\Theta=\theta_+)=\mathbb P(\Theta=\theta_-)=\frac12
\]
下，定义二元熵函数
\[
h_2(r):=-r\log_2 r-(1-r)\log_2(1-r),
\qquad
\varepsilon_T:=\frac{1}{1+\varphi^T}.
\]
则有：
\begin{enumerate}
  \item \(\tau_T\) 与参数 \(\Theta\) 独立；
  \item \(S_T\) 是停止样本 \(B_1^{\tau_T}\) 关于 \(\Theta\) 的极小充分统计量；
  \item 整个停止样本的参数信息严格塌缩为一比特：
  \[
  \boxed{\;
  I(\Theta;B_1^{\tau_T})
  =
  I(\Theta;S_T)
  =
  1-h_2(\varepsilon_T),
  \qquad
  I(\Theta;\tau_T)=0.
  \;}
  \]
\end{enumerate}
因此，时间坐标在推断论上严格退化为陪集变量，而终端符号携带全部参数信息。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-golden-sprt-esscher-boundary-symmetry}，对任意可达的 \(n\) 有
\[
\mathbb P_{\theta_+}(\tau_T=n)
=
\frac12\Bigl(\varphi^{T/2}+\varphi^{-T/2}\Bigr)
\left(\frac{2}{\varphi^{3/2}}\right)^n
\mathbb Q(\tau_T=n).
\]
把 \(\theta_+\) 换成 \(\theta_-\) 后，右端完全相同，因此
\[
\mathbb P_{\theta_+}(\tau_T=n)=\mathbb P_{\theta_-}(\tau_T=n)
\qquad
(\forall n),
\]
从而在等先验下 \(\tau_T\perp \Theta\)，于是 \(I(\Theta;\tau_T)=0\)。

极小充分性已经由定理 \ref{thm:conclusion-golden-sprt-terminal-sign-minimal-sufficiency} 给出。再由该定理中的似然比公式
\[
\frac{d\mathbb P_{\theta_+}}{d\mathbb P_{\theta_-}}(B_1^{\tau_T})
=
\varphi^{TS_T},
\]
停止后的后验只通过 \(S_T\) 进入。于是
\[
I(\Theta;B_1^{\tau_T})=I(\Theta;S_T).
\]

另一方面，由推论 \ref{cor:pom-sprt-stopping-posterior-constant}，
\[
\mathbb P(\Theta=\theta_+\mid S_T=s)
=
\frac{1}{1+\varphi^{-sT}},
\]
故从 \(\Theta\) 到 \(S_T\) 正好是一条交叉概率为
\[
\varepsilon_T=\frac{1}{1+\varphi^T}
\]
的二元对称信道。在均匀先验下，其互信息闭式为
\[
I(\Theta;S_T)=1-h_2(\varepsilon_T).
\]
综上得到所示等式。证毕。
\end{proof}

\begin{corollary}[固定窗与序贯窗样本复杂度的黄金分离常数]\label{cor:conclusion-golden-sprt-fixed-sequential-sample-separation}
令 \(P_e(n)\) 表示同一二相位模型中长度为 \(n\) 的 Bayes 最优固定窗错误率，并定义
\[
N_{\mathrm{fix}}(\varepsilon):=\min\{n\ge 1:\ P_e(n)\le \varepsilon\}.
\]
再令
\[
T(\varepsilon):=
\left\lceil
\log_\varphi\frac{1-\varepsilon}{\varepsilon}
\right\rceil,
\qquad
N_{\mathrm{seq}}(\varepsilon):=\mathbb E_{\theta_+}[\tau_{T(\varepsilon)}].
\]
则当 \(\varepsilon\downarrow0\) 时，
\[
N_{\mathrm{fix}}(\varepsilon)
\sim
\frac{1}{C_\varphi\log 2}\log\frac1\varepsilon,
\qquad
N_{\mathrm{seq}}(\varepsilon)
\sim
\frac{\varphi^3}{\log\varphi}\log\frac1\varepsilon,
\]
其中
\[
C_\varphi=\frac32\log_2\varphi-1.
\]
从而有严格极限
\[
\boxed{\;
\lim_{\varepsilon\downarrow0}
\frac{N_{\mathrm{fix}}(\varepsilon)}{N_{\mathrm{seq}}(\varepsilon)}
=
\frac{\log\varphi}{\varphi^3 C_\varphi\log 2}
=
3.962207716\ldots
\;}
\]
因此，本模型中最优序贯规则相对于最优固定窗规则的样本节约收敛到一个闭式黄金常数。
\end{corollary}
\begin{proof}
由定理 \ref{thm:pom-bayes-error-explicit-sharp-asymptotic}，
\[
P_e(n)=\varphi^2\sqrt{\frac{2}{\pi n}}\,(2\varphi^{-3/2})^n(1+o(1)),
\]
故
\[
-\log P_e(n)
=
n\,C_\varphi\log 2+o(n).
\]
这立即推出
\[
N_{\mathrm{fix}}(\varepsilon)
\sim
\frac{1}{C_\varphi\log 2}\log\frac1\varepsilon.
\]

另一方面，定理 \ref{thm:conclusion-golden-sprt-risk-time-elimination} 给出精确闭式
\[
\mathbb E_{\theta_+}[\tau_T]
=
\varphi^3(1-2\varepsilon_T)
\log_\varphi\frac{1-\varepsilon_T}{\varepsilon_T}.
\]
取 \(T=T(\varepsilon)\) 时，\(\varepsilon_T=\varepsilon(1+o(1))\)，故
\[
N_{\mathrm{seq}}(\varepsilon)
\sim
\frac{\varphi^3}{\log\varphi}\log\frac1\varepsilon.
\]
两式相除即得结论。证毕。
\end{proof}

\begin{theorem}[停时尾谱指数与固定窗 Chernoff 常数的极限同一]\label{thm:conclusion-golden-sprt-tail-exponent-chernoff-identity}
\leanverified{paper\_conclusion\_golden\_sprt\_tail\_exponent\_chernoff\_identity}
令
\[
\rho_T:=2\varphi^{-3/2}\cos\frac{\pi}{2T},
\qquad
\gamma_T:=-\log_2\rho_T.
\]
则有严格恒等式
\[
\boxed{\;
\gamma_T
=
C_\varphi-\log_2\!\Bigl(\cos\frac{\pi}{2T}\Bigr),
\qquad
C_\varphi=\frac32\log_2\varphi-1.
\;}
\]
从而
\[
\gamma_T\downarrow C_\varphi
\qquad (T\to\infty),
\]
并且存在二阶展开
\[
\boxed{\;
\gamma_T
=
C_\varphi+\frac{\pi^2}{8\log 2}\,T^{-2}+O(T^{-4}).
\;}
\]
等价地，对每个固定 \(T\)，命题 \ref{prop:pom-tau-exponential-tail-analytic-disk} 的双边尾界可改写为
\[
\mathbb P_{\theta_+}(\tau_T>n)=2^{-\gamma_T n+O_T(1)}.
\]
因此，序贯停时的时间尾谱在大阈值极限中与固定窗 Bayes 最优错误指数收敛到同一个黄金常数。
\end{theorem}
\begin{proof}
由命题 \ref{prop:pom-tau-exponential-tail-analytic-disk}，
\[
\rho_T=2\varphi^{-3/2}\cos\frac{\pi}{2T}.
\]
故
\[
\gamma_T
=
-\log_2\!\Bigl(2\varphi^{-3/2}\cos\frac{\pi}{2T}\Bigr)
=
\frac32\log_2\varphi-1-\log_2\!\Bigl(\cos\frac{\pi}{2T}\Bigr)
=
C_\varphi-\log_2\!\Bigl(\cos\frac{\pi}{2T}\Bigr),
\]
其中最后一步用到了推论 \ref{cor:pom-error-exponent-triple-identity} 中 \(C_\varphi\) 的闭式。
由于 \(\cos(\pi/2T)\uparrow 1\)，立即得到 \(\gamma_T\downarrow C_\varphi\)。

再用经典展开
\[
-\log\cos x=\frac{x^2}{2}+O(x^4)
\qquad(x\to0),
\]
取 \(x=\pi/(2T)\) 即得
\[
-\log_2\!\Bigl(\cos\frac{\pi}{2T}\Bigr)
=
\frac{1}{\log 2}\cdot \frac{\pi^2}{8T^2}+O(T^{-4}),
\]
从而得到二阶展开。最后，把命题 \ref{prop:pom-tau-exponential-tail-analytic-disk} 的双边界
\[
C_1(T)\rho_T^n\le \mathbb P_{\theta_+}(\tau_T>n)\le C_2(T)\rho_T^n
\]
取 \(\log_2\) 后重写即可。证毕。
\end{proof}

\begin{conjecture}[辅助时间刚性]\label{conj:conclusion-golden-sprt-ancillary-time-rigidity}
在等先验下，令
\[
Z_n:=\log_\varphi
\frac{\mathbb P(\Theta=\theta_+\mid B_1^n)}
{\mathbb P(\Theta=\theta_-\mid B_1^n)}
\in\mathbb Z
\]
为推论 \ref{cor:pom-phi-logit-lattice-translation} 中的 \(\varphi\)-logit 晶格过程。设 \(\tau\) 是适应于 \((\mathcal F_n)\) 的几乎处处有限停时，并满足：
\begin{enumerate}
  \item \(\tau\) 对参数 \(\Theta\) 为 ancillary；
  \item 存在 \(r\in(0,\frac12)\) 使
  \[
  \mathbb P(\Theta=\theta_+\mid \mathcal F_\tau)\in\{r,1-r\}
  \qquad\text{a.s.};
  \]
  \item 上述二值后验由某个 \(\mathcal F_\tau\)-可测符号统计量 \(S\in\{\pm1\}\) 区分。
\end{enumerate}
则应存在唯一整数 \(T\ge 1\) 使
\[
\tau=\inf\{n\ge 0:\ |Z_n|=T\}
\qquad\text{a.s.}
\]
亦即，在 \(\varphi\)-logit 晶格上，时间 ancillary 与后验二点塌缩若同时成立，则对称 first-exit barrier 应是唯一可能的机制。
\end{conjecture}
