\[
q:=\varphi^{-1}\in(0,1),
\qquad
X:=\sum_{n=2}^{\infty}\varepsilon_n q^n,
\qquad
\mathbb P(\varepsilon_n=1)=\mathbb P(\varepsilon_n=-1)=\frac12,
\]
并记 \(\mu_q^{(2)}\) 为 \(X\) 的分布。再记
\[
C_\varphi(u):=\bigl|\widehat\mu_q^{(2)}(\pi u)\bigr|
\qquad (u\in\mathbb Z).
\]

\begin{theorem}[Pisot--Bernoulli 采样的非 Rajchman 刚性与碰撞膨胀]\label{thm:conclusion-pisot-bernoulli-nonrajchman-collision-inflation}
其 Fourier--Stieltjes 变换满足
\[
\widehat\mu_q^{(2)}(t)=\prod_{n=2}^{\infty}\cos(q^n t),
\qquad
C_\varphi(u)=\bigl|\widehat\mu_q^{(2)}(\pi u)\bigr|.
\]
进一步，沿 Fibonacci 与 Lucas 子列分别存在两个不同的正极限
\[
C_\varphi(F_n)\to c_{\mathrm{Fib}}>0,
\qquad
C_\varphi(L_n)\to c_{\mathrm{Luc}}>0,
\qquad
c_{\mathrm{Fib}}\neq c_{\mathrm{Luc}}.
\]
故该算术采样在 \(\pi\mathbb Z\) 上是非 Rajchman 的；并且
\[
\Sigma_\varphi:=1+2\sum_{u=1}^{\infty}C_\varphi(u)^2=+\infty,
\qquad
F_{m+2}\operatorname{Col}_m\longrightarrow +\infty.
\]
\end{theorem}
\begin{proof}
Fourier 乘积公式与
\[
C_\varphi(u)=\bigl|\widehat\mu_q^{(2)}(\pi u)\bigr|
\]
正是主稿对 Pisot--Bernoulli 卷积的采样表象。另一方面，本小节前文已经证明：沿 Fibonacci 与 Lucas 两条递推子列分别存在正极限
\[
C_\varphi(F_n)\to c_{\mathrm{Fib}}>0,
\qquad
C_\varphi(L_n)\to c_{\mathrm{Luc}}>0,
\]
且二者不等。于是
\[
\widehat\mu_q^{(2)}(\pi u)
\]
不可能沿整数格趋于零，故它在该算术格点上是非 Rajchman 的。

同一结论链又给出
\[
\Sigma_\varphi=1+2\sum_{u\ge1}C_\varphi(u)^2=+\infty,
\]
并据此推出碰撞尺度无界增长
\[
F_{m+2}\operatorname{Col}_m\to+\infty.
\]
证毕。
\end{proof}

\begin{theorem}[有限共振闭包不可能性]\label{thm:conclusion-no-finite-resonance-closure}
对任意 \(U\in\mathbb N\)，定义截断共振闭包常数
\[
\Sigma_\varphi^{(U)}:=1+2\sum_{u=1}^{U}C_\varphi(u)^2.
\]
则对每个固定 \(U\) 都有
\[
\liminf_{m\to\infty}F_{m+2}\operatorname{Col}_m
\ge
\Sigma_\varphi^{(U)}.
\]
由于
\[
\Sigma_\varphi^{(U)}\xrightarrow[U\to\infty]{}+\infty,
\]
故不存在任何有限 \(U\) 使得前 \(U\) 个共振常数就能在渐近上封闭全部碰撞尺度。换言之，碰撞异常的闭包在该模型中本质上是无限尾驱动的。
\end{theorem}
\begin{proof}
主稿先证明：对任意 \(U\)，均有下界
\[
\liminf_{m\to\infty}F_{m+2}\operatorname{Col}_m
\ge
1+2\sum_{u=1}^{U}C_\varphi(u)^2
=
\Sigma_\varphi^{(U)}.
\]
另一方面，由定理 \ref{thm:conclusion-pisot-bernoulli-nonrajchman-collision-inflation} 已知
\[
\sum_{u\ge1}C_\varphi(u)^2=+\infty,
\]
故 \(\Sigma_\varphi^{(U)}\to\infty\)。若某个有限 \(U_0\) 足以闭合全部碰撞尺度，则上式右端应被某个有限常数统一控制，这与 \(\Sigma_\varphi^{(U)}\to\infty\) 矛盾。证毕。
\end{proof}

\begin{theorem}[有限共振骨架的无界赤字律]\label{thm:conclusion-finite-resonance-skeleton-unbounded-deficit}
对任意 \(U\in\mathbb N\)，定义有限共振骨架常数
\[
L_U:=1+2\sum_{u=1}^{U}C_\varphi(u)^2.
\]
则有
\[
L_U\le \liminf_{m\to\infty}F_{m+2}\operatorname{Col}_m
\qquad(\forall U\ge 1),
\]
并且
\[
\sup_{U\ge 1}L_U=+\infty.
\]
更强地，对每个固定 \(U\) 都成立
\[
\liminf_{m\to\infty}\bigl(F_{m+2}\operatorname{Col}_m-L_U\bigr)=+\infty.
\]
因此，任意固定有限数目的共振通道都只能给出一个有界骨架，而真实碰撞尺度在该骨架之上始终保留不可消除的无界赤字。
\end{theorem}
\begin{proof}
前两条只是定理 \ref{thm:conclusion-no-finite-resonance-closure} 的直接改写。

对最后一条，由定理 \ref{thm:conclusion-pisot-bernoulli-nonrajchman-collision-inflation}，
\[
F_{m+2}\operatorname{Col}_m\longrightarrow+\infty.
\]
由于 \(L_U\) 对固定 \(U\) 只是一个常数，故
\[
F_{m+2}\operatorname{Col}_m-L_U\longrightarrow+\infty,
\]
从而其下极限也等于 \(+\infty\)。证毕。
\end{proof}

\begin{corollary}[碰撞尺度不存在单常数型重整化]\label{cor:conclusion-collision-scale-no-single-constant-renormalization}
不存在常数 \(C\in(0,\infty)\) 使
\[
F_{m+2}\operatorname{Col}_m\longrightarrow C.
\]
更强地，
\[
F_{m+2}\operatorname{Col}_m\longrightarrow+\infty.
\]
与此同时，仍有严格的两频共振地板
\[
\liminf_{m\to\infty}
F_{m+2}\!\left(\operatorname{Col}_m-\frac1{F_{m+2}}\right)
\ge
2C_\varphi^2>0.
\]
因此，碰撞统计的极限结构严格分裂为：有限共振频对给出常数级地板，而无限共振闭包则强制出发散级主项。
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-pisot-bernoulli-nonrajchman-collision-inflation} 已给出
\[
F_{m+2}\operatorname{Col}_m\to+\infty,
\]
从而任何有限常数极限都不可能存在。

另一方面，定理 \ref{thm:conclusion-fold-golden-resonance-collision-gap-hard-floor} 已经给出
\[
\liminf_{m\to\infty}
F_{m+2}\!\left(\operatorname{Col}_m-\frac1{F_{m+2}}\right)\ge 2C_\varphi^2.
\]
再结合定理 \ref{thm:conclusion-finite-resonance-skeleton-unbounded-deficit}，便知碰撞尺度不可能被任何单一常数重整化。证毕。
\end{proof}
