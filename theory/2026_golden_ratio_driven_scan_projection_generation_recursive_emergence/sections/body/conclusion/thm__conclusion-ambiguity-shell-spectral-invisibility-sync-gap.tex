\begin{theorem}[瞬态壳的谱隐身定理]\label{thm:conclusion-ambiguity-shell-spectral-invisibility}
\leanverified{paper\_conclusion\_ambiguity\_shell\_spectral\_invisibility}
对 \(m\ge 3\) 的 Zeckendorf 折叠右确定化表示，按“歧义态在前、单点态在后”排序，有
\[
A_m^{\det}=
\begin{pmatrix}
N_m & R_m\\
0 & A_m^{\mathrm{ess}}
\end{pmatrix},
\qquad
N_m^m=0.
\]
并且对一切 \(n\ge 1\)，成立
\[
\Tr\!\bigl((A_m^{\det})^n\bigr)=\Tr\!\bigl((A_m^{\mathrm{ess}})^n\bigr),
\qquad
\det(I-zA_m^{\det})=\det(I-zA_m^{\mathrm{ess}}).
\]
同时 canonical 同步预算满足
\[
D_{\mathrm{sync}}(m)=m-1.
\]
故 ambiguity shell 对周期谱、Fredholm 因子与 \(\zeta\)-行列式完全不可见；它只作为有限同步延迟存活。
\end{theorem}
\begin{proof}
主稿对右确定化矩阵已给出上三角块分解
\[
A_m^{\det}=
\begin{pmatrix}
N_m & R_m\\
0 & A_m^{\mathrm{ess}}
\end{pmatrix},
\qquad
N_m^m=0.
\]
于是对任意 \(n\ge 1\)，仍有
\[
(A_m^{\det})^n=
\begin{pmatrix}
N_m^n & *\\
0 & (A_m^{\mathrm{ess}})^n
\end{pmatrix}.
\]
由于 \(N_m\) 幂零，其全部特征值皆为 \(0\)，故
\[
\Tr(N_m^n)=0
\qquad(n\ge 1),
\]
从而
\[
\Tr\!\bigl((A_m^{\det})^n\bigr)=\Tr\!\bigl((A_m^{\mathrm{ess}})^n\bigr).
\]

同理，
\[
\det(I-zA_m^{\det})
=
\det(I-zN_m)\det(I-zA_m^{\mathrm{ess}}).
\]
而 \(N_m\) 幂零迫使
\[
\det(I-zN_m)=1,
\]
故
\[
\det(I-zA_m^{\det})=\det(I-zA_m^{\mathrm{ess}}).
\]
最后，由主稿的同步延迟闭式，
\[
D_{\mathrm{sync}}(m)=m-1.
\]
于是瞬态壳在周期/Fredholm 通道中完全隐身，却在同步预算中留下精确有限延迟。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\zeta$}{zeta}-完备审计必然不同步完备]\label{cor:conclusion-zeta-complete-audit-not-sync-complete}
\leanverified{paper\_conclusion\_zeta\_complete\_audit\_not\_sync\_complete}
任一仅通过
\[
\det(I-zA),\qquad
\zeta_A(z),\qquad
\{\Tr(A^n)\}_{n\ge 1}
\]
构造的审计器，都不可能恢复 canonical 同步预算 \(D_{\mathrm{sync}}(m)\)。因此，若要求同步可审计，就必须额外记录 ambiguity shell 模块，而不能将其压缩到周期/Fredholm 层。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-ambiguity-shell-spectral-invisibility}，上述 determinant/trace/\(\zeta\) 数据全部只依赖于本质块 \(A_m^{\mathrm{ess}}\)，而对瞬态壳 \(N_m\) 完全失明。另一方面，同步预算
\[
D_{\mathrm{sync}}(m)=m-1
\]
恰由瞬态壳的有限延迟给出。故任何仅通过这些谱型数据构造的审计器都无法恢复同步预算。证毕。
\end{proof}

\begin{theorem}[歧义壳的幂零指数恰等于窗口长度]\label{thm:conclusion-ambiguity-shell-nilpotent-index-equals-window}
\leanverified{paper\_conclusion\_ambiguity\_shell\_nilpotent\_index\_equals\_window}
对每个 \(m\ge 3\)，歧义壳块 \(N_m\) 的幂零指数恰为
\[
\nu(N_m)=m.
\]
亦即，
\[
N_m^m=0,\qquad N_m^{m-1}\neq 0.
\]
\end{theorem}
\begin{proof}
第一式已包含在定理 \ref{thm:conclusion-ambiguity-shell-spectral-invisibility} 的假设中。为证第二式，只需利用同步延迟的尖锐性。主稿已经证明
\[
D_{\mathrm{sync}}(m)=m-1.
\]
因此存在某个输出轨道，其确定化滤波候选集在前 \(m-1\) 步始终非单点，而在第 \(m\) 步首次收缩为单点。等价地，在右确定化图中存在一条长度恰为 \(m-1\) 的有向路径完全停留在歧义态集合中。于是 \(N_m^{m-1}\) 至少有一个非零条目，从而
\[
N_m^{m-1}\neq 0.
\]
结合 \(N_m^m=0\)，得 \(\nu(N_m)=m\)。证毕。
\end{proof}

\begin{corollary}[未同步前缀生成函数的多项式退化]\label{cor:conclusion-ambiguity-shell-unsync-prefix-polynomial}
\leanverified{paper\_conclusion\_ambiguity\_shell\_unsync\_prefix\_polynomial}
对 \(m\ge 3\)，定义未同步前缀计数
\[
N_m^{\mathrm{unsync}}(n),
\qquad
U_m(z):=\sum_{n\ge 0}N_m^{\mathrm{unsync}}(n)z^n.
\]
则 \(U_m(z)\) 不是一般 sofic 情形中的真有理函数，而是一个非零多项式，且其次数恰为
\[
\deg U_m=m-1.
\]
等价地，
\[
N_m^{\mathrm{unsync}}(n)=0\quad (n\ge m),
\qquad
N_m^{\mathrm{unsync}}(m-1)>0.
\]
\end{corollary}
\begin{proof}
由歧义块的矩阵计数表示，
\[
N_m^{\mathrm{unsync}}(n)=e_{V_m}^{T}N_m^n\mathbf 1.
\]
定理 \ref{thm:conclusion-ambiguity-shell-nilpotent-index-equals-window} 给出 \(N_m^m=0\)，故对一切 \(n\ge m\)，
\[
N_m^{\mathrm{unsync}}(n)=0.
\]
因此 \(U_m(z)\) 至多是次数 \(m-1\) 的多项式。

另一方面，由同一定理的 \(N_m^{m-1}\neq 0\) 以及同步延迟 \(D_{\mathrm{sync}}(m)=m-1\) 的尖锐性，存在长度恰为 \(m-1\) 的未同步前缀，故
\[
N_m^{\mathrm{unsync}}(m-1)>0.
\]
于是 \(\deg U_m=m-1\)。证毕。
\end{proof}

\begin{theorem}[主奇点星形簇在 Zeckendorf 点完全坍缩]\label{thm:conclusion-zeckendorf-ambiguity-shell-pole-star-collapse}
对 \(m\ge 3\) 的 Zeckendorf 折叠，未同步前缀生成函数 \(U_m(z)\) 在有限复平面内无极点，因此不存在一般 sofic 歧义壳理论中的主奇点圆、等模极点星形簇，亦不存在由 \(n\bmod p\) 驱动的未同步前缀周期调制。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-ambiguity-shell-unsync-prefix-polynomial}，\(U_m(z)\) 是一条多项式，故在 \(\CC\) 上整且无极点。因而一般 sofic 口径中由主极点圆与等模极点簇驱动的解析机制在此情形下全部消失。

与此同时，主稿的一般 sofic 理论把“边界上存在等模主极点簇”与“系数首项具有 \(n\bmod p\) 周期振幅”解析地联系起来。既然 Zeckendorf 点根本不存在这类极点簇，相应的周期调制亦不可能出现。证毕。
\end{proof}

\begin{theorem}[同步指数、熵缺口与分形余维的同一性]\label{thm:conclusion-ambiguity-shell-sync-exponent-fractal-codimension-identity}
\leanverified{paper\_conclusion\_ambiguity\_shell\_sync\_exponent\_fractal\_codimension\_identity}
在一般 sofic 歧义壳口径下，设 \(Y_m^{\mathrm{amb}}\neq\varnothing\)，并假设歧义邻接矩阵 \(A_m^{\mathrm{amb}}\) 的最大谱半径强连通分量唯一。记
\[
\rho_{\mathrm{amb}}(m):=\rho(A_m^{\mathrm{amb}}),
\qquad
\varepsilon_m:=\log 2-\log \rho_{\mathrm{amb}}(m).
\]
则未同步尾满足尖锐渐近
\[
\nu_m(T_m>n)=c_{n\bmod p}(m)e^{-\varepsilon_m n}(1+o(1)),
\]
其中 \(p\) 为该最大分量的周期；并且
\[
\operatorname{codim}_H(Y_m^{\mathrm{amb}})
=
\frac{\varepsilon_m}{\log 2}.
\]
换言之，
\[
\varepsilon_m
=
\log 2-h_{\mathrm{top}}(Y_m^{\mathrm{amb}})
=
(\log 2)\operatorname{codim}_H(Y_m^{\mathrm{amb}}).
\]
\end{theorem}
\begin{proof}
主稿的 Perron--Frobenius 同步尾定理给出
\[
\nu_m(T_m>n)=c_{n\bmod p}(m)\rho_{\mathrm{amb}}(m)^n(1+o(1)),
\]
将其改写为以 \(2^{-n}\) 为背景尺度的指数率，便得到
\[
\nu_m(T_m>n)=c_{n\bmod p}(m)e^{-\varepsilon_m n}(1+o(1)),
\qquad
\varepsilon_m=\log 2-\log\rho_{\mathrm{amb}}(m).
\]
另一方面，在自然前缀超度量下，歧义壳的 Hausdorff 维数等于
\[
\dim_H(Y_m^{\mathrm{amb}})=\frac{\log \rho_{\mathrm{amb}}(m)}{\log 2},
\]
故
\[
\operatorname{codim}_H(Y_m^{\mathrm{amb}})
=
1-\frac{\log \rho_{\mathrm{amb}}(m)}{\log 2}
=
\frac{\log 2-\log \rho_{\mathrm{amb}}(m)}{\log 2}
=
\frac{\varepsilon_m}{\log 2}.
\]
又 \(h_{\mathrm{top}}(Y_m^{\mathrm{amb}})=\log\rho_{\mathrm{amb}}(m)\)，于是三式合并即得结论。证毕。
\end{proof}

\begin{corollary}[周期调制与主奇点星簇的等价性]\label{cor:conclusion-ambiguity-shell-periodic-modulation-pole-star-equivalence}
在定理 \ref{thm:conclusion-ambiguity-shell-sync-exponent-fractal-codimension-identity} 的假设下，下列三项等价：
\begin{enumerate}
  \item 未同步尾 \(\nu_m(T_m>n)\) 具有非平凡的 \(n\bmod p\) residue-class 调制；
  \item 未同步前缀生成函数
  \[
  U_m(z):=\sum_{n\ge 0}N_m^{\mathrm{unsync}}(n)z^n
  \]
  在圆周 \(|z|=\rho_{\mathrm{amb}}(m)^{-1}\) 上具有 \(p>1\) 个等模主极点；
  \item \(A_m^{\mathrm{amb}}\) 为周期 \(p>1\) 的 imprimitive 矩阵。
\end{enumerate}
\end{corollary}
\begin{proof}
由矩阵表达式
\[
U_m(z)=e_{V_m}^{\top}(I-zA_m^{\mathrm{amb}})^{-1}\mathbf 1
\]
可知其主奇点由 \(A_m^{\mathrm{amb}}\) 的最大模特征值决定。Perron--Frobenius 周期理论表明：当且仅当最大强连通分量的周期为 \(p>1\) 时，收敛圆周上会出现 \(p\) 个等模主极点，并且系数首项出现同一模 \(p\) 的 residue-class 调制。反之，若矩阵 primitive，则主极点唯一，系数首项无非平凡周期振幅。故三者等价。证毕。
\end{proof}

\begin{conjecture}[Zeckendorf 点是歧义壳星形簇相图的幂零边界]\label{conj:conclusion-zeckendorf-nilpotent-boundary-of-pole-star-phase}
考虑保持单点骨架 \(A_m^{\mathrm{ess}}\) 不变、但允许歧义层作局部 sofic 变形的相图。则 Zeckendorf 点应当对应于主奇点星形簇的完全坍缩边界，其特征为：
\begin{enumerate}
  \item 未同步前缀生成函数 \(U_m(z)\) 退化为多项式；
  \item 歧义壳不再携带 Perron 主谱，而只保留一座高度 \(m\) 的纯幂零塔；
  \item 一旦越过该边界进入一般 sofic 区域，\(U_m(z)\) 立刻展开为真有理函数，并同时出现主极点圆、周期振幅与
  \[
  \operatorname{codim}(Y_m^{\mathrm{amb}})=\frac{\varepsilon_m}{\log 2}<1
  \]
  的分形余维机制。
\end{enumerate}
因此，Zeckendorf 点不是“低熵 Perron 尾趋零”的端点，而是“Perron 型歧义相”向“纯幂零延迟相”塌缩时的幂零边界点。
\end{conjecture}

这一边界图景由前述结果共同驱动：歧义壳在 Zeckendorf 点保留了最坏线性同步延迟，却把一般 sofic 歧义层中的主极点星形簇、周期振幅与分形余维机制全部压灭；于是该点的本征特征并非“歧义很小”，而是“歧义完全失去 Perron 主导，只剩零谱幂零塔”。
