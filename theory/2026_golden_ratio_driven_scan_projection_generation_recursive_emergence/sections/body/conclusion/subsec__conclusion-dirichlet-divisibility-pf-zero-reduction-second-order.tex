\subsection{Dirichlet 整除传播、黄金共振偶系数的 \texorpdfstring{$PF_\infty$}{PF∞} 刚性与零点削减的二阶校正}\label{subsec:conclusion-dirichlet-divisibility-pf-zero-reduction-second-order}
沿用命题 \ref{prop:real-input-40-output-dirichlet-nonrh}、定理 \ref{thm:fold-resonance-entire-lp}、定理 \ref{thm:fold-zero-uncertainty}、定理 \ref{thm:fold-zero-density-sparse} 与定理 \ref{thm:xi-fold-zero-geometry-collision-gap-blindness} 的记号。前文已经分别建立 Dirichlet 扭曲坏模数的整除传播、黄金共振整函数 \(\mathcal F_\varphi\) 的 Laguerre--P\'olya 闭包与全实根 Jensen 链、以及谱零块的指数稀疏律与碰撞缺口的黄金常数下界。把这三条结构链并置之后，还可继续抽取出如下若干条终端后果。

\begin{theorem}[Dirichlet 扭曲慢模在整除提升下的精确保形]\label{thm:conclusion-dirichlet-twist-divisibility-lift-monotone}
\leanverified{paper\_conclusion\_dirichlet\_twist\_divisibility\_lift\_monotone}
沿用命题 \ref{prop:real-input-40-output-dirichlet-nonrh} 的记号。对任意整数 \(\ell\ge 1\)、\(m\ge 2\) 与 \(1\le j\le m-1\)，记
\[
\omega_m:=e^{2\pi i/m},
\qquad
\rho_{m,j}:=\rho\!\bigl(M(\omega_m^j)\bigr),
\qquad
\theta_{m,j}:=\frac{\log \rho_{m,j}}{\log\lambda},
\]
\[
\rho_m:=\max_{1\le j\le m-1}\rho_{m,j},
\qquad
\theta_m:=\max_{1\le j\le m-1}\theta_{m,j},
\qquad
\lambda:=\rho(M(1))=\varphi^2.
\]
则有精确恒等式
\[
M(\omega_{\ell m}^{\,\ell j})=M(\omega_m^j),
\qquad
\rho_{\ell m,\ell j}=\rho_{m,j},
\qquad
\theta_{\ell m,\ell j}=\theta_{m,j}.
\]
因而必有
\[
\rho_{\ell m}\ge \rho_m,
\qquad
\theta_{\ell m}\ge \theta_m.
\]
\end{theorem}
\begin{proof}
由单位根恒等式
\[
\omega_{\ell m}^{\,\ell j}
=
e^{2\pi i\ell j/(\ell m)}
=
e^{2\pi ij/m}
=
\omega_m^j
\]
可知，参数 \(\omega_{\ell m}^{\,\ell j}\) 与 \(\omega_m^j\) 对应的是同一个复权值，故
\[
M(\omega_{\ell m}^{\,\ell j})=M(\omega_m^j).
\]
于是谱半径完全相同：
\[
\rho_{\ell m,\ell j}=\rho_{m,j}.
\]
再除以正数 \(\log\lambda\)，即得
\[
\theta_{\ell m,\ell j}=\theta_{m,j}.
\]
最后，由于 \(1\le \ell j\le \ell m-\ell<\ell m\)，\(\ell j\) 确为模 \(\ell m\) 的非平凡频率，故
\[
\rho_{\ell m}
=
\max_{1\le r\le \ell m-1}\rho_{\ell m,r}
\ge
\rho_{\ell m,\ell j}
=
\rho_{m,j}.
\]
对 \(j\) 再取最大即可得到 \(\rho_{\ell m}\ge \rho_m\)；\(\theta_{\ell m}\ge \theta_m\) 则由 \(x\mapsto \log x/\log\lambda\) 的单调性立即推出。证毕。
\end{proof}

\begin{corollary}[仅由整除传播便可推出坏模数集的正下密度]\label{cor:conclusion-dirichlet-bad-moduli-positive-lower-density-by-divisibility}
\leanverified{paper\_conclusion\_dirichlet\_bad\_moduli\_positive\_lower\_density\_by\_divisibility}
定义坏模数集
\[
\mathcal B:=\{m\ge 2:\ \theta_m>\tfrac12\}.
\]
则对任意 \(m_0\in\mathcal B\)，都有
\[
m_0\NN_{\ge 1}\subseteq \mathcal B.
\]
特别地，其下自然密度满足
\[
\underline d(\mathcal B)
:=
\liminf_{X\to\infty}\frac{\#(\mathcal B\cap [1,X])}{X}
\ge
\frac{1}{m_0}
>
0.
\]
\end{corollary}
\begin{proof}
若 \(m_0\in\mathcal B\)，则 \(\theta_{m_0}>\tfrac12\)。由定理 \ref{thm:conclusion-dirichlet-twist-divisibility-lift-monotone}，
对任意 \(\ell\ge 1\) 都有
\[
\theta_{\ell m_0}\ge \theta_{m_0}>\tfrac12,
\]
故 \(\ell m_0\in\mathcal B\)。于是
\[
m_0\NN_{\ge 1}\subseteq \mathcal B.
\]
再对任意 \(X\ge 1\) 计数即得
\[
\#(\mathcal B\cap [1,X])
\ge
\left\lfloor \frac{X}{m_0}\right\rfloor.
\]
两边除以 \(X\) 并取 \(\liminf\)，遂得
\[
\underline d(\mathcal B)\ge \frac{1}{m_0}.
\]
证毕。
\end{proof}

\begin{theorem}[黄金共振整函数的符号修正偶系数构成 \texorpdfstring{$PF_\infty$}{PF∞} 序列]\label{thm:conclusion-golden-resonance-even-coeff-pf-infty}
沿用定理 \ref{thm:fold-resonance-entire-lp} 的记号。将黄金共振整函数写成偶幂级数
\[
\mathcal F_\varphi(z)=\sum_{r\ge 0}a_{2r}z^{2r},
\qquad
a_{2r+1}=0,
\]
并定义符号修正后的偶系数
\[
b_r:=(-1)^r a_{2r}
\qquad (r\ge 0).
\]
则序列 \((b_r)_{r\ge 0}\) 为一个 \(PF_\infty\) 序列。等价地，其普通生成函数
\[
E_\varphi(t):=\sum_{r\ge 0}b_r t^r
\]
的全部零点都落在负实轴上，且均为单根。
\end{theorem}
\begin{proof}
将 \(\mathcal F_\varphi\) 的正零点按升序记为
\[
0<\rho_1<\rho_2<\cdots.
\]
由定理 \ref{thm:fold-resonance-entire-lp} 的显式零点公式，
\[
\{\rho_j:\ j\ge 1\}
=
\Bigl\{\Bigl(k+\tfrac12\Bigr)\varphi^n:\ k\in\ZZ_{\ge 0},\ n\ge 2\Bigr\},
\]
故
\[
\sum_{j\ge 1}\rho_j^{-2}
=
\sum_{n\ge 2}\sum_{k\ge 0}
\frac{1}{(k+\tfrac12)^2\varphi^{2n}}
<
\infty.
\]
因此由 Hadamard 乘积、定理 \ref{thm:fold-resonance-entire-lp} 所给出的阶 \(1\) 增长、以及 \(\mathcal F_\varphi(0)=1\)、\(\mathcal F_\varphi\) 为偶函数，可写成
\[
\mathcal F_\varphi(z)=\prod_{j\ge 1}\left(1-\frac{z^2}{\rho_j^2}\right).
\]
于是
\[
E_\varphi(t)
=
\sum_{r\ge 0}b_r t^r
=
\prod_{j\ge 1}\left(1+\frac{t}{\rho_j^2}\right),
\]
从而 \(E_\varphi\) 的全部零点恰为
\[
t=-\rho_j^2<0
\qquad (j\ge 1),
\]
并且每个零点都是单根。

对每个 \(N\ge 1\)，定义截断多项式
\[
E_{\varphi,N}(t):=\prod_{j=1}^{N}\left(1+\frac{t}{\rho_j^2}\right)
=
\sum_{r=0}^{N}b_r^{(N)}t^r.
\]
由于 \(E_{\varphi,N}\) 的全部零点都位于负实轴上，有限序列版的 Aissen--Schoenberg--Whitney 判别给出：系数列
\[
\bigl(b_0^{(N)},b_1^{(N)},\dots,b_N^{(N)},0,0,\dots\bigr)
\]
为 \(PF_\infty\) 序列。另一方面，对每个固定 \(r\) 都有
\[
b_r^{(N)}\longrightarrow b_r
\qquad (N\to\infty),
\]
因为 \(b_r^{(N)}\) 与 \(b_r\) 分别是 \((\rho_j^{-2})_{1\le j\le N}\) 与 \((\rho_j^{-2})_{j\ge 1}\) 的第 \(r\) 个初等对称函数。
于是任意有限 Toeplitz 子式都是非负数列的极限，因而仍非负。
这就证明了 \((b_r)_{r\ge 0}\) 为 \(PF_\infty\) 序列。证毕。
\end{proof}

\begin{corollary}[黄金共振偶系数满足严格 Tur\'an--P\'olya 链]\label{cor:conclusion-golden-resonance-even-coeff-turan-polya}
沿用定理 \ref{thm:conclusion-golden-resonance-even-coeff-pf-infty} 的记号。则有：
\begin{enumerate}
  \item Toeplitz 矩阵
  \[
  \bigl(b_{i-j}\bigr)_{i,j\ge 0}
  \qquad\bigl(\text{约定 }b_n=0\ \text{当 }n<0\bigr)
  \]
  的全部有限子式都非负；
  \item 严格 Tur\'an 不等式成立：
  \[
  b_r^2>b_{r-1}b_{r+1}
  \qquad (r\ge 1);
  \]
  \item 等价地，\((\log b_r)_{r\ge 0}\) 为严格凹序列。
\end{enumerate}
\end{corollary}
\begin{proof}
第 \(1\) 条正是定理 \ref{thm:conclusion-golden-resonance-even-coeff-pf-infty} 中 \(PF_\infty\) 性质的直接重述。

再定义截断多项式
\[
E_{\varphi,N}(t)
:=
\prod_{j=1}^{N}\left(1+\frac{t}{\rho_j^2}\right)
=
\sum_{s=0}^{N}b_s^{(N)}t^s,
\]
可知，对每个固定 \(r\ge 1\) 与每个 \(N>r\)，Newton 不等式给出
\[
\bigl(b_r^{(N)}\bigr)^2
\ge
\frac{r+1}{r}\frac{N-r+1}{N-r}\,
b_{r-1}^{(N)}b_{r+1}^{(N)}.
\]
令 \(N\to\infty\)，便得到
\[
b_r^2
\ge
\frac{r+1}{r}\,b_{r-1}b_{r+1}
>
b_{r-1}b_{r+1},
\]
即第 \(2\) 条。最后，由 \(E_\varphi(t)=\prod_{j\ge 1}(1+t/\rho_j^2)\) 可见 \(b_r>0\) 对所有 \(r\) 都成立，故
\[
b_r^2>b_{r-1}b_{r+1}
\]
等价于
\[
2\log b_r>\log b_{r-1}+\log b_{r+1},
\]
这正是第 \(3\) 条。证毕。
\end{proof}

\begin{theorem}[零点削减在碰撞与 \texorpdfstring{$\mathrm{KL}$}{KL} 标度上仅为二阶校正]\label{thm:conclusion-zero-reduction-second-order-collision-kl}
记
\[
M:=F_{m+2},
\qquad
p_m(r):=\frac{d_m(r)}{2^m},
\qquad
u(r):=\frac{1}{M}
\qquad
\bigl(r\in \ZZ/(M\ZZ)\bigr),
\]
\[
\mathsf{Col}_m:=\sum_{r}p_m(r)^2,
\qquad
\mathcal Z_m:=\{k\in \ZZ/(M\ZZ):\ \widehat d_m(k)=0\},
\qquad
S_m:=\max_{k\neq 0}|\widehat d_m(k)|.
\]
则始终有
\[
\mathsf{Col}_m-\frac1M
\le
\frac{M-1-|\mathcal Z_m|}{M},
\qquad
D_{\mathrm{KL}}(p_m\|u)\le \log(M-|\mathcal Z_m|),
\]
\[
S_m\le 2^m\sqrt{M-1-|\mathcal Z_m|},
\]
并且同一定理还给出对应的最大纤维超额上界。

若进一步 \(M\) 为偶数，则
\[
0\le \frac{|\mathcal Z_m|}{M}\le \tau(m+2)\,O(\varphi^{-m/2}),
\]
而体相 Fibonacci 共振给出
\[
\liminf_{m\to\infty}M\Bigl(\mathsf{Col}_m-\frac1M\Bigr)\ge 2C_\varphi^2>0.
\]
从而
\[
\frac{|\mathcal Z_m|}{M\left(\mathsf{Col}_m-\frac1M\right)}
=
O\!\bigl(\tau(m+2)\varphi^{-m/2}\bigr)
\longrightarrow 0,
\]
并且
\[
\log M-\log(M-|\mathcal Z_m|)
=
O\!\bigl(\tau(m+2)\varphi^{-m/2}\bigr).
\]
特别地，零点削减对碰撞缺口与 \(\log M\) 级信息上界的改变量都只有指数小量；常数量级的主导项只能由体相共振常数 \(C_\varphi\) 承载。
\end{theorem}
\begin{proof}
前三条上界正是定理 \ref{thm:fold-zero-uncertainty} 的碰撞、KL 与峰值不确定性链；同一定理亦给出相应的最大纤维超额上界。
若 \(M\) 为偶数，则由定理 \ref{thm:fold-zero-density-sparse}，
\[
\frac{|\mathcal Z_m|}{M}\le \tau(m+2)\,O(\varphi^{-m/2}).
\]
另一方面，定理 \ref{thm:xi-fold-zero-geometry-collision-gap-blindness} 给出
\[
M\Bigl(\mathsf{Col}_m-\frac1M\Bigr)\ge 2C_\varphi^2+o(1),
\]
故两式相除可得
\[
\frac{|\mathcal Z_m|}{M\left(\mathsf{Col}_m-\frac1M\right)}
\le
\frac{\tau(m+2)\,O(\varphi^{-m/2})}{2C_\varphi^2+o(1)}
=
O\!\bigl(\tau(m+2)\varphi^{-m/2}\bigr),
\]
并因右端趋于 \(0\) 而得到第一条渐近式。

最后，由
\[
\log(M-|\mathcal Z_m|)
=
\log M+\log\!\Bigl(1-\frac{|\mathcal Z_m|}{M}\Bigr)
\]
以及 \(-\log(1-x)=x+O(x^2)\)（当 \(x\to 0\)）可得
\[
\log M-\log(M-|\mathcal Z_m|)
=
\frac{|\mathcal Z_m|}{M}
+O\!\left(\frac{|\mathcal Z_m|^2}{M^2}\right).
\]
再代入
\[
\frac{|\mathcal Z_m|}{M}\le \tau(m+2)\,O(\varphi^{-m/2}),
\]
即得
\[
\log M-\log(M-|\mathcal Z_m|)
=
O\!\bigl(\tau(m+2)\varphi^{-m/2}\bigr).
\]
证毕。
\end{proof}

\begin{theorem}[偶模数上的平方根律统一失效]\label{thm:conclusion-dirichlet-even-modulus-uniform-parity-obstruction}
\leanverified{paper\_conclusion\_dirichlet\_even\_modulus\_uniform\_parity\_obstruction}
沿用定理 \ref{thm:conclusion-dirichlet-twist-divisibility-lift-monotone} 的记号。若 \(m\ge 2\) 为偶整数，则有统一下界
\[
\theta_m\ge \beta_{\mathrm{par}}
:=
\frac{\log \rho(-1)}{\log\lambda}
>
\frac12,
\qquad
\lambda=\varphi^2,
\]
其中 \(\rho(-1)\) 是三次方程
\[
x^3-2x^2-1=0
\]
的最大实根。特别地，
\[
\theta_m\ge \beta_{\mathrm{par}}\approx 0.82186854047.
\]
换言之，偶模数上的 Dirichlet--Artin 扭曲误差指数全部统一突破平方根门槛。
\end{theorem}
\begin{proof}
若 \(m\) 为偶数，则取 \(j=m/2\)，有
\[
\omega_m^j=e^{2\pi i(m/2)/m}=-1.
\]
由定义
\[
\rho_m=\max_{1\le j\le m-1}\rho_{m,j}\ge \rho_{m,m/2}=\rho(-1),
\]
故
\[
\theta_m=\frac{\log \rho_m}{\log\lambda}\ge \frac{\log \rho(-1)}{\log\lambda}=\beta_{\mathrm{par}}.
\]
另一方面，命题 \ref{prop:real-input-40-output-dirichlet-nonrh} 的核心谱计算已经给出
\[
\rho(-1)>\lambda^{1/2}=\varphi,
\]
故
\[
\beta_{\mathrm{par}}>\frac12.
\]
数值值由该三次方程的最大实根直接计算即得。证毕。
\end{proof}

\begin{corollary}[最小非平凡角色已足以触发平方根律失效]\label{cor:conclusion-dirichlet-minimal-character-trigger}
\leanverified{paper\_conclusion\_dirichlet\_minimal\_character\_trigger}
记
\[
\lambda=\varphi^2.
\]
在最小非平凡单位根扭曲
\[
u=-1
\]
下，相应谱半径 \(\rho(-1)\) 恰为三次方程
\[
x^3-2x^2-1=0
\]
的最大实根，并满足
\[
\rho(-1)>\lambda^{1/2}=\varphi.
\]
故
\[
\frac{\log \rho(-1)}{\log \lambda}>\frac12.
\]
因此，RH 型平方根律的失效并不需要高阶角色或精细扭曲；最基本的奇偶角色已经足以触发相变。
\end{corollary}
\begin{proof}
当模数 \(m=2\) 时，唯一的非平凡 Dirichlet 角色正对应于
\[
\omega_2=-1.
\]
于是定理 \ref{thm:conclusion-dirichlet-even-modulus-uniform-parity-obstruction} 直接给出
\[
\frac{\log \rho(-1)}{\log \lambda}
=
\beta_{\mathrm{par}}
>
\frac12.
\]
而 \(\rho(-1)\) 是三次方程
\[
x^3-2x^2-1=0
\]
的最大实根，已在该定理的证明与命题 \ref{prop:real-input-40-output-dirichlet-nonrh} 中固定。证毕。
\end{proof}

\begin{theorem}[偶模数 residue census 的超平方根坏类强制存在]\label{thm:conclusion-dirichlet-even-modulus-bad-residue-census}
对偶整数 \(m\ge 2\)，定义 primitive residue counts
\[
\Pi_{n,a}^{(m)}
:=
\frac1m\sum_{j=0}^{m-1}\omega_m^{-aj}p_n(\omega_m^j),
\qquad
a\in \ZZ/m\ZZ,
\]
并记
\[
E_{n,a}^{(m)}:=\Pi_{n,a}^{(m)}-\frac{p_n(1)}{m}.
\]
则对每个偶模数 \(m\) 与每个长度 \(n\) 都有精确均方恒等式
\[
\frac1m\sum_{a\in \ZZ/m\ZZ}\bigl|E_{n,a}^{(m)}\bigr|^2
=
\frac1{m^2}\sum_{j=1}^{m-1}\bigl|p_n(\omega_m^j)\bigr|^2
\ge
\frac1{m^2}\,\bigl|p_n(-1)\bigr|^2.
\]
因此，对每个偶模数 \(m\) 与每个 \(n\) 都至少存在一个同余类 \(a_n\in \ZZ/m\ZZ\) 使
\[
\bigl|E_{n,a_n}^{(m)}\bigr|\ge \frac{|p_n(-1)|}{m}.
\]
进一步，若记
\[
\rho_-:=\rho(-1),
\]
则存在常数 \(c_m>0\) 与 \(n_0(m)\) 使得对所有 \(n\ge n_0(m)\) 都有
\[
\max_{a\in \ZZ/m\ZZ}\bigl|E_{n,a}^{(m)}\bigr|
\ge
\frac{c_m}{n}\,\rho_-^n.
\]
故每一个偶模数的 residue census 都必然包含至少一个不可被平方根门槛压低的坏类。
\end{theorem}
\leanverified{paper\_conclusion\_dirichlet\_even\_modulus\_bad\_residue\_census}
\begin{proof}
由离散 Fourier 反演，
\[
E_{n,a}^{(m)}
=
\frac1m\sum_{j=1}^{m-1}\omega_m^{-aj}p_n(\omega_m^j).
\]
对 \(a\) 作均方平均并用角色正交性
\[
\frac1m\sum_{a\in \ZZ/m\ZZ}\omega_m^{a(k-j)}=\delta_{j,k},
\]
即可得到
\[
\frac1m\sum_{a\in \ZZ/m\ZZ}\bigl|E_{n,a}^{(m)}\bigr|^2
=
\frac1{m^2}\sum_{j=1}^{m-1}\bigl|p_n(\omega_m^j)\bigr|^2.
\]
若 \(m\) 为偶数，则 \(j=m/2\) 对应 \(\omega_m^{m/2}=-1\)，故
\[
\frac1m\sum_{a}\bigl|E_{n,a}^{(m)}\bigr|^2
\ge
\frac1{m^2}\,\bigl|p_n(-1)\bigr|^2.
\]
因而必存在某个 \(a_n\) 使
\[
\bigl|E_{n,a_n}^{(m)}\bigr|\ge \frac{|p_n(-1)|}{m}.
\]

下证 \(p_n(-1)\) 的指数下界。设
\[
a_n(u):=\sum_{\lambda\in \operatorname{Spec}(M(u))}\lambda^n
\]
为 primitive Witt 反演中的扭曲周期和，则
\[
p_n(-1)=\frac1n\sum_{d\mid n}\mu(d)\,a_{n/d}\!\bigl((-1)^d\bigr).
\]
由命题 \ref{prop:real-input-40-output-dirichlet-nonrh}，\(\rho_-=\rho(-1)\) 是矩阵 \(M(-1)\) 的简单主谱半径，且其余谱点模长严格小于 \(\rho_-\)。因此存在 \(\vartheta_1<\rho_-\) 与 \(c_->0\) 使
\[
a_n(-1)=c_-\,\rho_-^n+O(\vartheta_1^n).
\]
另一方面，
\[
a_n(1)=\lambda^n+O(\vartheta_2^n)
\]
对某个 \(\vartheta_2<\lambda\) 成立。于是在 Witt 反演中，\(d=1\) 给出主项
\[
\frac{c_-}{n}\rho_-^n,
\]
而 \(d\ge 2\) 的各项至多为
\[
O\!\left(\frac{\max(\rho_-^{n/2},\lambda^{n/2})}{n}\right).
\]
由于
\[
\rho_->\lambda^{1/2},
\]
故所有 \(d\ge 2\) 的项都严格次于 \(\rho_-^n/n\)。因此存在 \(\vartheta<\rho_-\) 使
\[
p_n(-1)=\frac{c_-}{n}\rho_-^n+O\!\left(\frac{\vartheta^n}{n}\right),
\]
从而对充分大 \(n\) 有
\[
|p_n(-1)|\ge c\,\frac{\rho_-^n}{n}
\]
对某个常数 \(c>0\) 成立。把该下界代回
\[
\max_a |E_{n,a}^{(m)}|\ge \frac{|p_n(-1)|}{m}
\]
即可得到最后结论。证毕。
\end{proof}

上述结果把三条此前分离的主线压缩为同一闭合图景：一方面，即使不诉诸更强的奇素数模渐近，仅由一个坏模数及其整除传播，平方根门槛失效便已扩张为具有正下密度的倍数塔；更进一步，偶模数上的 \((-1)\)-角色会在每一层 residue census 中强制产生至少一个超平方根坏类，而该最小导体 \(2\) 的 parity 障碍又被证明对一切有限 cyclotomic 手术完全刚性。另一方面，\(\mathcal F_\varphi\) 的 Laguerre--P\'olya 实零点结构并不止于 Jensen 多项式的全实根性，而会向 Taylor 偶系数层降格为 \(PF_\infty\) 总正性与严格 Tur\'an 链；最后，谱零块虽然能够改进碰撞、KL 与峰值的确定性上界，但其相对量级仅为 \(\varphi^{-m/2+o(m)}\)，不足以解释常数量级的碰撞缺口，而真正的主导项只能来自 Fibonacci 体相共振常数 \(C_\varphi\)。因而 Dirichlet 扭曲障碍、共振系数全正性与零块稀疏校正不再是并列现象，而是同一算术--解析刚性在模数塔、系数塔与频域塔上的三重投影。
