\subsection{window-\texorpdfstring{$6$}{6} 的长根 Bloch 三带、双椭圆曲线与规范体积分层}\label{subsec:conclusion-window6-long-bloch-bielliptic-gauge-stratification}
在 window-\(6\) 的 Coxeter 三项链、qutrit--sign--doublet 张量正规形、boundary 三轴的 \((\ZZ/2\ZZ)^3\) 中心解释以及 \(m=6\) 规范不变量的精确数值锚点都已固定之后，还可把 long bulk 的 Bloch 色散、其所诱导的 genus-\(2\) 代数几何以及规范体积在 boundary/short/long 三层上的严格分解压缩为同一条闭合链。其核心现象是：长根层上的半周期 profile \((4,2,3,4,2,3)\) 在 sign 因子上对角化后恰降为一个由 \(z\mapsto z^3\) 驱动的单一三次色散关系，而这一三次式同时控制三条能带、一个双椭圆 Bloch 曲线与 \(g_6,\kappa_6\) 的几何分层。

\begin{theorem}[window-\texorpdfstring{$6$}{6} 长根层的手征 Bloch 三次式]\label{thm:conclusion-window6-long-bloch-chiral-cubic}
沿用定理 \ref{thm:conclusion-window6-qutrit-sign-doublet-normal-form} 的记号，并令
\[
M:=\operatorname{diag}(4,2,3),\qquad
Q:=
\begin{pmatrix}
0&0&1\\
1&0&0\\
0&1&0
\end{pmatrix},
\qquad
Q^{-1}=Q^2.
\]
在一条长根项链块上定义
\[
\widetilde H_{\ell}(z):=D_{\ell}+zP_6+z^{-1}P_6^{-1},
\qquad z\in\CC^\times.
\]
则存在酉基使
\[
\widetilde H_{\ell}(z)\simeq H_+(z)\oplus H_-(z),
\qquad
H_\pm(z):=M\pm\bigl(zQ+z^{-1}Q^2\bigr).
\]
因而整个 \(12\) 维长根层的 Bloch 族
\[
H_{\ell}(z):=D_6|_{A_{\ell}}+z\,U_{\mathfrak c}|_{A_{\ell}}+z^{-1}U_{\mathfrak c}^{-1}|_{A_{\ell}}
\]
满足
\[
H_{\ell}(z)\simeq \bigl(H_+(z)\oplus H_-(z)\bigr)\otimes I_2.
\]
并且其特征多项式存在精确因式分解
\[
\det(\lambda I_3-H_+(z))
=(\lambda-1)(\lambda-3)(\lambda-5)-\bigl(z^3+z^{-3}\bigr),
\]
\[
\det(\lambda I_3-H_-(z))
=(\lambda-1)(\lambda-3)(\lambda-5)+\bigl(z^3+z^{-3}\bigr).
\]
特别地，当 \(z=e^{i\theta}\) 时，长根层的 Bloch 色散由单一三次方程
\[
(\lambda-1)(\lambda-3)(\lambda-5)=2\cos(3\theta)
\]
控制，而另一手征块仅对应于平移 \(\theta\mapsto \theta+\pi/3\)。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-qutrit-sign-doublet-normal-form}，
\[
A_{\ell}\cong \CC^3\otimes \CC^2_{\mathrm{sgn}}\otimes \CC^2_{\mathrm{orb}},
\]
并且在适当基下
\[
D_6|_{A_{\ell}}=M\otimes I_2\otimes I_2,
\qquad
U_{\mathfrak c}|_{A_{\ell}}=Q\otimes S_2\otimes I_2.
\]
因此
\[
H_{\ell}(z)
=\Bigl((M\otimes I_2)+z(Q\otimes S_2)+z^{-1}(Q^2\otimes S_2)\Bigr)\otimes I_2.
\]
取任一把 \(S_2\) 对角化为 \(\operatorname{diag}(1,-1)\) 的酉矩阵 \(R\)，则 sign 因子分裂为
\[
\widetilde H_{\ell}(z)\simeq H_+(z)\oplus H_-(z),
\qquad
H_{\ell}(z)\simeq \bigl(H_+(z)\oplus H_-(z)\bigr)\otimes I_2.
\]
对 \(H_+(z)\) 直接展开得
\[
\lambda I_3-H_+(z)=
\begin{pmatrix}
\lambda-4 & -z^{-1} & -z\\
-z & \lambda-2 & -z^{-1}\\
-z^{-1} & -z & \lambda-3
\end{pmatrix},
\]
其行列式为
\[
\lambda^3-9\lambda^2+23\lambda-15-(z^3+z^{-3})
=(\lambda-1)(\lambda-3)(\lambda-5)-(z^3+z^{-3}).
\]
又因 \(H_-(z)=H_+(-z)\)，第二个特征多项式随之成立。最后代入 \(z=e^{i\theta}\) 即有
\[
z^3+z^{-3}=2\cos(3\theta),
\]
从而得到所述色散关系。证毕。
\end{proof}

\begin{theorem}[long bulk 的精确态密度公式]\label{thm:conclusion-window6-long-bloch-dos}
设 \(\nu_{\ell}\) 为 long bulk 的归一化态密度测度，则 \(\nu_{\ell}\) 绝对连续，且其密度满足
\[
\frac{d\nu_{\ell}}{d\lambda}
=
\frac{|P'(\lambda)|}{3\pi\sqrt{4-P(\lambda)^2}}\,
\mathbf 1_{\Sigma_{\ell}}(\lambda).
\]
换言之，window-\(6\) 的 long bulk 态密度正是标准 arcsine 测度在三次映射 \(P\) 下的精确推前。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-long-bloch-chiral-cubic}，另一手征块仅是 \(\theta\mapsto \theta+\pi/3\) 的平移，而 orbit doublet 只给出整体双重简并，故归一化态密度可只在单一手征块上计算。
记
\[
t:=P(\lambda)=2\cos(3\theta).
\]
由于 \(3\theta\) 在模 \(2\pi\) 意义下仍均匀分布，\(t\) 的分布即为标准 arcsine 测度
\[
\frac{dt}{\pi\sqrt{4-t^2}},
\qquad t\in[-2,2].
\]
再由推论 \ref{cor:conclusion-window6-long-bloch-threeband-trifolding}，\(\Sigma_{\ell}\) 分裂为三段不交区间，而 \(P\) 在每一段上严格单调。故对任意测试函数 \(f\)，有
\[
\int f\,d\nu_{\ell}
=
\frac13\int_{-2}^{2}
\sum_{P(\lambda)=t}f(\lambda)\,
\frac{dt}{\pi\sqrt{4-t^2}}.
\]
在每一单调支上作变量替换 \(t=P(\lambda)\)，即 \(dt=P'(\lambda)\,d\lambda\)，遂得
\[
\int f\,d\nu_{\ell}
=
\int_{\Sigma_{\ell}}
f(\lambda)\,
\frac{|P'(\lambda)|}{3\pi\sqrt{4-P(\lambda)^2}}\,d\lambda.
\]
因此所述密度公式成立。证毕。
\end{proof}

\begin{theorem}[长根 Bloch 曲线的 genus-\texorpdfstring{$2$}{2} 超椭圆模型与分支判别式]\label{thm:conclusion-window6-long-bloch-hyperelliptic-curve}

\operatorname{Disc}\bigl(P(\lambda)^2-4\bigr)=2^{16}37^2.

\end{theorem}
\begin{proof}

因此 \(\mathcal C_6^{\mathrm{long}}\) 的光滑射影化确为 genus \(2\) 的超椭圆曲线。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} long bulk 的双椭圆性]\label{cor:conclusion-window6-long-bloch-bielliptic-quotients}

\iota_+(x,y)=(-x,y),
\qquad
\iota_-(x,y)=(-x,-y).

相应地，\(\mathcal C_6^{\mathrm{long}}\) 存在两条显式二重椭圆商：

E_+:\quad v^2=u^3-8u^2+16u-4

并且 \(E_-\) 进一步双有理等价于 Weierstrass 模型

\end{corollary}
\begin{proof}

该三次多项式的判别式为 \(592\neq 0\)，故 \(E_+\) 光滑，从而为椭圆曲线。

\[

\qquad

即得

证毕。
\end{proof}

\begin{theorem}[long bulk 的精确 Lyapunov--Fuglede--Kadison 公式]\label{thm:conclusion-window6-long-bloch-lyapunov-fk}
对一手征长根块定义
\[
L(\lambda):=
\frac{1}{2\pi}
\int_0^{2\pi}
\log\Bigl|\det\bigl(\lambda I_3-H_+(e^{i\theta})\bigr)\Bigr|\,d\theta.
\]
则有闭式公式
\[
L(\lambda)=
\begin{cases}
0, & |P(\lambda)|\le 2,\\[2mm]
\displaystyle
\log\frac{|P(\lambda)|+\sqrt{P(\lambda)^2-4}}{2},
& |P(\lambda)|>2.
\end{cases}
\]
特别地，Lyapunov 指数在三条能带上恒为零，在三条带隙内严格为正；同一公式亦给出整个 long bulk 的归一化 Fuglede--Kadison 对数体积。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-long-bloch-chiral-cubic}，
\[
\det\bigl(\lambda I_3-H_+(e^{i\theta})\bigr)=P(\lambda)-2\cos(3\theta).
\]
记 \(a:=P(\lambda)\)，并把 \(3\theta\) 重新记作 \(\vartheta\)。则
\[
L(\lambda)=\frac1{2\pi}\int_0^{2\pi}\log|a-2\cos\vartheta|\,d\vartheta.
\]
令 \(u=e^{i\vartheta}\)，则
\[
a-2\cos\vartheta=a-u-u^{-1}=u^{-1}(u^2-au+1).
\]
因 \(|u|=1\)，前因子 \(u^{-1}\) 对平均对数积分无贡献，故
\[
L(\lambda)=\frac1{2\pi}\int_0^{2\pi}\log|u^2-au+1|\,d\vartheta.
\]
若 \(|a|\le 2\)，则二次方程 \(u^2-au+1=0\) 的两根都落在单位圆上，Jensen 公式给出 \(L(\lambda)=0\)。
若 \(|a|>2\)，则其两根为
\[
u_{\pm}=\frac{a\pm \sqrt{a^2-4}}{2},
\qquad
u_+u_-=1,
\]
其中恰有一根在单位圆外，其模为
\[
\frac{|a|+\sqrt{a^2-4}}{2}.
\]
再次应用 Jensen 公式，即得
\[
L(\lambda)=\log\frac{|a|+\sqrt{a^2-4}}{2}.
\]
而另一手征块只是 \(\vartheta\) 的平移，因此同一闭式亦适用于整个 long bulk。证毕。
\end{proof}

结合定理 \ref{thm:conclusion-window6-minimal-fiber-spin-hypercube} 的 boundary 自旋超立方体与本小节的 long bulk 结果，可以把 window-\(6\) 的可见层理解为一个严格分层的体界系统：边界层由 \((\ZZ/2\ZZ)^3\) 在八个自旋权上的符号作用组织成 \(Q_3\)；体相长根层则由定理 \ref{thm:conclusion-window6-long-bloch-chiral-cubic} 至推论 \ref{cor:conclusion-window6-long-bloch-bielliptic-quotients} 的三次 Bloch 关系组织成三带、双椭圆的 genus-\(2\) 曲线；而定理 \ref{thm:conclusion-window6-gauge-volume-geometric-stratification} 说明 \(g_6\) 与 \(\kappa_6\) 又在 boundary、short、long 三层上精确可加。于是，超立方体、Lie 根系、矩阵代数、信息谱与 Bloch 频带不再只是彼此类比的语言，而成为同一审计对象的五种互补坐标。
