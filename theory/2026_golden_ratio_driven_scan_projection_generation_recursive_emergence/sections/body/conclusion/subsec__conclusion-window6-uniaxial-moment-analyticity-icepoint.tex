\subsection{window-\texorpdfstring{$6$}{6} 的单轴矩塌缩、固定窗解析平凡与 ice point 轴向猜想}\label{subsec:conclusion-window6-uniaxial-moment-analyticity-icepoint}

在 window-\(6\) 的 Fibonacci 单变量台阶、\(B_3\) multiplicity pinning、cyclic 根云的 escort 矩张量以及固定窗三能级热力学都已固定之后，还可进一步把隐藏多重度的全部可见输出压缩为一条单轴律。其核心在于：凡只依赖纤维多重度的根值观测，一阶矩都严格塌缩到 \(e_1\) 轴，二阶矩都在同一 Cartan 标架下对角化且只有常权重才恢复各向同性；与此同时，固定窗隐藏扇区的配分函数只是三项指数和，因而沿矩阶参数 \(q\) 的热力学始终保持解析而无内部临界奇点。由此，有限窗的多重度代数不再产生新的连续方向，而只留下一个被严格钉扎的轴向通道。

\begin{theorem}[任意 multiplicity 权重的一阶矩单轴塌缩]\label{thm:conclusion-window6-arbitrary-multiplicity-first-moment-axis-collapse}
沿用定理 \ref{thm:conclusion-window6-multiplicity-pinning} 的 \(B_3\) 字典
\[
\iota_B:X_6^{\mathrm{cyc}}\xrightarrow{\sim}\Phi(B_3).
\]
对任意函数
\[
f:\{2,3,4\}\longrightarrow \RR
\]
定义根值线性观测
\[
\mathfrak m_f
:=
\sum_{w\in X_6^{\mathrm{cyc}}}
f\!\bigl(d_6^{\mathrm{bin}}(w)\bigr)\,\iota_B(w).
\]
则有精确恒等式
\[
\mathfrak m_f=\bigl(f(4)-f(2)\bigr)e_1.
\]
特别地，
\[
\sum_{w\in X_6^{\mathrm{cyc}}} d_6^{\mathrm{bin}}(w)\,\iota_B(w)=2e_1,
\qquad
\sum_{w\in X_6^{\mathrm{cyc}}} \log d_6^{\mathrm{bin}}(w)\,\iota_B(w)=(\log 2)e_1.
\]
因此，任何只依赖于纤维多重度的根值线性观测量，其 \(e_2,e_3\) 分量都严格湮灭；window-\(6\) 的隐藏扭曲只能产生轴向偏置，而不能生成第二条独立横向漂移。
\end{theorem}
\begin{proof}
由双射 \(\iota_B\)，可将求和改写为
\[
\mathfrak m_f
=
\sum_{\alpha\in\Phi(B_3)} f\!\bigl(m_6(\alpha)\bigr)\,\alpha.
\]
再由推论 \ref{cor:conclusion-window6-signed-weyl-collapse}，
\[
R_2=\{-e_1\}\cup\{\pm e_2\pm e_3\},
\qquad
R_3=\{\pm e_1\pm e_3\},
\]
\[
R_4=\{e_1,\pm e_2,\pm e_3,\pm e_1\pm e_2\},
\]
其中 \(R_j:=\{\alpha\in\Phi(B_3):m_6(\alpha)=j\}\)。
于是
\[
\sum_{\alpha\in R_2}\alpha=-e_1,
\qquad
\sum_{\alpha\in R_3}\alpha=0,
\]
因为四个 \(\pm e_2\pm e_3\) 与四个 \(\pm e_1\pm e_3\) 都成对抵消。另一方面，\(B_3\) 全部根的向量和等于 \(0\)，故
\[
\sum_{\alpha\in R_4}\alpha
=
-\sum_{\alpha\in R_2}\alpha-\sum_{\alpha\in R_3}\alpha
=
e_1.
\]
分层求和便得到
\[
\mathfrak m_f
=
f(2)(-e_1)+f(3)\cdot 0+f(4)e_1
=
\bigl(f(4)-f(2)\bigr)e_1.
\]
取 \(f(t)=t\) 与 \(f(t)=\log t\) 即得两条特例。证毕。
\end{proof}

\begin{theorem}[任意 multiplicity 权重的二阶矩刚性与各向同性障碍]\label{thm:conclusion-window6-arbitrary-multiplicity-second-moment-rigidity}
沿用定理 \ref{thm:conclusion-window6-arbitrary-multiplicity-first-moment-axis-collapse} 的记号。对任意函数
\[
f:\{2,3,4\}\longrightarrow \RR
\]
定义二阶矩张量
\[
M_f
:=
\sum_{w\in X_6^{\mathrm{cyc}}}
f\!\bigl(d_6^{\mathrm{bin}}(w)\bigr)\,
\iota_B(w)\iota_B(w)^{\mathsf T}.
\]
则有完全显式的对角公式
\[
M_f
=
\operatorname{diag}\!\Bigl(
f(2)+4f(3)+5f(4),\ 
4f(2)+6f(4),\ 
4f(2)+4f(3)+2f(4)
\Bigr).
\]
从而 \(M_f\) 是标量矩阵当且仅当
\[
f(2)=f(3)=f(4).
\]
换言之，任一非常数的 multiplicity 重整化都会严格打破 \(B_3\) 的各向同性；它不可能生成各向同性的 \(B_3\)-值 Gaussian free field 协方差。
\leanverified{paper\_conclusion\_window6\_arbitrary\_multiplicity\_second\_moment\_rigidity}
\end{theorem}
\begin{proof}
记
\[
Q_j:=\sum_{\alpha\in R_j}\alpha\alpha^{\mathsf T}
\qquad (j=2,3,4),
\]
其中 \(R_j\) 仍取自推论 \ref{cor:conclusion-window6-signed-weyl-collapse}。直接计算得
\[
Q_2
=
(-e_1)(-e_1)^{\mathsf T}
+\sum_{\pm,\pm}(\pm e_2\pm e_3)(\pm e_2\pm e_3)^{\mathsf T}
=
\operatorname{diag}(1,4,4),
\]
\[
Q_3
=
\sum_{\pm,\pm}(\pm e_1\pm e_3)(\pm e_1\pm e_3)^{\mathsf T}
=
\operatorname{diag}(4,0,4).
\]
另一方面，定理 \ref{thm:conclusion-window6-cyclic-escort-second-moment-tensor} 在 \(q=0\) 处给出
\[
\sum_{\alpha\in\Phi(B_3)}\alpha\alpha^{\mathsf T}=10I_3,
\]
故
\[
Q_4=10I_3-Q_2-Q_3=\operatorname{diag}(5,6,2).
\]
于是
\[
M_f=f(2)Q_2+f(3)Q_3+f(4)Q_4,
\]
代入即可得到所示显式对角式。

若 \(M_f\) 为标量矩阵，则三条对角元必须两两相等。由第一与第三对角元相等得
\[
f(4)=f(2),
\]
再与第一、第二对角元相等联立即得
\[
f(3)=f(2).
\]
反过来，若 \(f(2)=f(3)=f(4)\)，则 \(M_f\) 显然为该常数乘以 \(10I_3\)。故各向同性当且仅当 \(f\) 为常函数。证毕。
\end{proof}

\begin{proposition}[固定 window-\texorpdfstring{$6$}{6} 隐藏扇区的解析平凡性]\label{prop:conclusion-window6-fixed-hidden-sector-analytic-triviality}
定义 escort 型配分函数
\[
Z_6(q)
:=
\sum_{w\in X_6}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q
=
9\cdot 4^q+4\cdot 3^q+8\cdot 2^q
\qquad (q\in\RR),
\]
并记
\[
\mathcal P_6(q):=\log Z_6(q).
\]
则 \(Z_6\) 在复变量 \(q\) 上延拓为整函数，而 \(\mathcal P_6\) 在 \(\RR\) 上实解析且严格凸，并满足
\[
\lim_{q\to+\infty}\mathcal P_6'(q)=\log 4,
\qquad
\lim_{q\to-\infty}\mathcal P_6'(q)=\log 2.
\]
因此，固定 window-\(6\) 的隐藏扇区内部不存在任何临界奇点；其 \(q\to+\infty\) 极限只在九条四重点纤维上冻结，而 \(q\to-\infty\) 极限只在八条二重点纤维上冻结。六顶点连续极限中的真正临界机制，因而不可能来自固定 \(m=6\) 的多重度代数，而只能来自增长尺度上的相关组织。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-window6-fixed-window-freezing-closed}，
\[
Z_6(q)=9\cdot 4^q+4\cdot 3^q+8\cdot 2^q.
\]
把 \(d^q\) 视为 \(e^{q\log d}\)，便知 \(Z_6\) 是复变量 \(q\) 上有限个指数整函数之和，故本身为整函数。

在实轴上，\(\mathcal P_6(q)=\log Z_6(q)\) 显然实解析。再定义 escort 概率
\[
\mu_q(w):=\frac{(d_6^{\mathrm{bin}}(w))^q}{Z_6(q)}.
\]
则直接微分得
\[
\mathcal P_6''(q)=\Var_{\mu_q}\!\bigl(\log d_6^{\mathrm{bin}}(w)\bigr).
\]
由于 \(\log 2,\log 3,\log 4\) 三个取值两两不同，且三层计数 \(8,4,9\) 全都非零，上式对任意实数 \(q\) 都严格为正，故 \(\mathcal P_6\) 严格凸。

最后，当 \(q\to+\infty\) 时，\(9\cdot 4^q\) 支配 \(Z_6(q)\)，故
\[
\mathcal P_6'(q)
=
\frac{9\cdot 4^q\log 4+4\cdot 3^q\log 3+8\cdot 2^q\log 2}
{9\cdot 4^q+4\cdot 3^q+8\cdot 2^q}
\longrightarrow \log 4.
\]
同理，当 \(q\to-\infty\) 时，\(8\cdot 2^q\) 成为主导项，从而
\[
\mathcal P_6'(q)\longrightarrow \log 2.
\]
这正对应于四重点与二重点纤维两端的冻结极限。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的三低阶几何量对全矩谱的刚性决定]\label{thm:conclusion-window6-three-low-order-moments-rigidify-full-spectrum}
\leanverified{paper\_conclusion\_window6\_three\_low\_order\_moments\_rigidify\_full\_spectrum}
记
\[
S_q:=\sum_{w\in X_6}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^q
\qquad (q\ge 0),
\]
并对 \(j\in\{2,3,4\}\) 记
\[
N_j:=\#\{w\in X_6:d_6^{\mathrm{bin}}(w)=j\}.
\]
则仅由 window-\(6\) 的三条低阶数据
\[
d_6^{\mathrm{bin}}(w)\in\{2,3,4\},
\qquad
S_0=21,
\qquad
S_1=64,
\qquad
S_2=212,
\]
即可唯一恢复整个纤维谱，且必有
\[
(N_2,N_3,N_4)=(8,4,9).
\]
从而对一切 \(q\ge 0\),
\[
S_q=8\cdot 2^q+4\cdot 3^q+9\cdot 4^q.
\]
于是 \((S_q)_{q\ge 0}\) 满足三阶常系数递推
\[
S_{q+3}=9S_{q+2}-26S_{q+1}+24S_q
\qquad (q\ge 0).
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:fold-bin-m6-fiber-hist}，window-\(6\) 的全部纤维大小只能取 \(2,3,4\)。又由定理 \ref{thm:xi-time-part9z-window6-moment-involutive-algebra-realization}，
\[
S_0=\#X_6=21,
\qquad
S_1=\sum_{w\in X_6}d_6^{\mathrm{bin}}(w)=64,
\qquad
S_2=\sum_{w\in X_6}\bigl(d_6^{\mathrm{bin}}(w)\bigr)^2=212.
\]
因此 \((N_2,N_3,N_4)\) 满足线性系统
\[
N_2+N_3+N_4=21,
\]
\[
2N_2+3N_3+4N_4=64,
\]
\[
4N_2+9N_3+16N_4=212.
\]
由前两式消去 \(N_2\) 得
\[
N_3+2N_4=22,
\]
而由后二式相减得
\[
3N_3+8N_4=84.
\]
联立即得
\[
N_4=9,\qquad N_3=4,\qquad N_2=8.
\]
故
\[
S_q=\sum_{j=2}^4 N_j j^q=8\cdot 2^q+4\cdot 3^q+9\cdot 4^q.
\]
最后，因为 \(2,3,4\) 都是多项式
\[
r^3-9r^2+26r-24=(r-2)(r-3)(r-4)
\]
的根，故每条指数列 \((j^q)_{q\ge 0}\) 都满足
\[
j^{q+3}-9j^{q+2}+26j^{q+1}-24j^q=0,
\]
其线性组合 \(S_q\) 亦满足同一递推。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} 全矩谱的 Euler 微分闭包]\label{cor:conclusion-window6-euler-egf-ode-closure}
定义 Euler 指数型生成函数
\[
\mathcal E_6(t):=\sum_{q=0}^{\infty}\frac{S_q}{q!}t^q.
\]
则 \(\mathcal E_6\) 是三阶线性常微分方程
\[
f'''-9f''+26f'-24f=0
\]
满足初值
\[
f(0)=21,\qquad f'(0)=64,\qquad f''(0)=212
\]
的唯一 entire 解，并且具有闭式
\[
\mathcal E_6(t)=8e^{2t}+4e^{3t}+9e^{4t}.
\]
特别地，
\[
\mathcal E_6(1)=8e^2+4e^3+9e^4.
\]
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-window6-three-low-order-moments-rigidify-full-spectrum}，
\[
S_{q+3}-9S_{q+2}+26S_{q+1}-24S_q=0.
\]
把该递推逐项乘以 \(t^q/q!\) 并对 \(q\ge 0\) 求和，立得
\[
\mathcal E_6'''(t)-9\mathcal E_6''(t)+26\mathcal E_6'(t)-24\mathcal E_6(t)=0.
\]
而
\[
\mathcal E_6^{(n)}(0)=S_n
\qquad (n=0,1,2),
\]
故初值为
\[
\mathcal E_6(0)=21,\qquad
\mathcal E_6'(0)=64,\qquad
\mathcal E_6''(0)=212.
\]
该微分方程的特征根恰为 \(2,3,4\)，因而其一般解为
\[
f(t)=Ae^{2t}+Be^{3t}+Ce^{4t}.
\]
代入上述三条初值，得到线性系统
\[
A+B+C=21,\qquad 2A+3B+4C=64,\qquad 4A+9B+16C=212,
\]
其唯一解为 \((A,B,C)=(8,4,9)\)。遂得
\[
\mathcal E_6(t)=8e^{2t}+4e^{3t}+9e^{4t}.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的两隙冻结律与 rank-two Bernoulli 地态]\label{thm:conclusion-window6-two-gap-freezing-ranktwo-bernoulli-groundstate}
对 \(w\in X_6\) 定义 Gibbs 族
\[
\nu_t(w):=\frac{e^{t d_6^{\mathrm{bin}}(w)}}{\mathcal E_6(t)},
\qquad
\mathcal E_6(t)=8e^{2t}+4e^{3t}+9e^{4t},
\]
并记
\[
G_j:=\{w\in X_6:d_6^{\mathrm{bin}}(w)=j\}
\qquad (j=2,3,4).
\]
则对一切 \(t\in \RR\)，有精确公式
\[
\nu_t(G_4)=\frac{9}{9+4e^{-t}+8e^{-2t}},
\]
\[
\nu_t(G_3)=\frac{4e^{-t}}{9+4e^{-t}+8e^{-2t}},
\]
\[
\nu_t(G_2)=\frac{8e^{-2t}}{9+4e^{-t}+8e^{-2t}}.
\]
因此当 \(t\to+\infty\) 时，
\[
\nu_t(G_4)=1-\frac49 e^{-t}-\frac{56}{81}e^{-2t}+O(e^{-3t}),
\]
\[
\nu_t(G_3)=\frac49 e^{-t}-\frac{16}{81}e^{-2t}+O(e^{-3t}),
\]
\[
\nu_t(G_2)=\frac89 e^{-2t}+O(e^{-3t}).
\]
换言之，window-\(6\) 的低温选态只有两条真正的激发尺度：一级尺度 \(e^{-t}\) 仅来自三点纤维，二级尺度 \(e^{-2t}\) 才首次看见二点纤维。

更强地，在地态扇区 \(G_4\) 上，每条四点纤维的 hidden 偏移分布都恰为
\[
21B_1+34B_2,
\qquad
B_1,B_2\stackrel{\mathrm{i.i.d.}}{\sim}\mathrm{Bernoulli}\!\left(\frac12\right),
\]
故低温极限不仅选出 \(d=4\) 地态，还把地态内部 rigidify 为一个 rank-two Bernoulli 双寄存器。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-window6-three-low-order-moments-rigidify-full-spectrum}，
\[
\#G_2=8,\qquad \#G_3=4,\qquad \#G_4=9.
\]
因此
\[
\nu_t(G_j)=\frac{\#G_j\,e^{jt}}{8e^{2t}+4e^{3t}+9e^{4t}}
\qquad (j=2,3,4).
\]
把分子分母同除以 \(e^{4t}\)，即得
\[
\nu_t(G_4)=\frac{9}{9+4e^{-t}+8e^{-2t}},
\]
\[
\nu_t(G_3)=\frac{4e^{-t}}{9+4e^{-t}+8e^{-2t}},
\]
\[
\nu_t(G_2)=\frac{8e^{-2t}}{9+4e^{-t}+8e^{-2t}}.
\]
再对 \(e^{-t}\) 在 \(0\) 附近作常规幂级数展开，即得
\[
\nu_t(G_4)=1-\frac49 e^{-t}-\frac{56}{81}e^{-2t}+O(e^{-3t}),
\]
\[
\nu_t(G_3)=\frac49 e^{-t}-\frac{16}{81}e^{-2t}+O(e^{-3t}),
\]
\[
\nu_t(G_2)=\frac89 e^{-2t}+O(e^{-3t}).
\]
这表明 \(d=3\) 扇区构成首个激发层，而 \(d=2\) 扇区只在更深一级的 \(e^{-2t}\) 上出现。

最后，由定理 \ref{thm:xi-time-part9xab-window6-fourpoint-fiber-ranktwo-bernoulli-factorization}，任意四点纤维的 hidden 偏移分布都恰为
\[
21B_1+34B_2,
\qquad
B_1,B_2\stackrel{\mathrm{i.i.d.}}{\sim}\mathrm{Bernoulli}\!\left(\frac12\right).
\]
故低温极限在把质量推向 \(G_4\) 的同时，也把地态内部的随机结构 rigidify 为 rank-two Bernoulli 双寄存器。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} escort--Legendre 前沿的二次代数化]\label{thm:conclusion-window6-escort-legendre-frontier-quadratic-algebraization}
定义
\[
K_6(t):=\log\frac{\mathcal E_6(t)}{21}
=\log\frac{8e^{2t}+4e^{3t}+9e^{4t}}{21},
\]
以及其 Legendre 对偶
\[
I_6(\alpha):=\sup_{t\in\RR}\{\alpha t-K_6(t)\}.
\]
则
\[
\alpha_6(t):=K_6'(t)
=\frac{16e^{2t}+12e^{3t}+36e^{4t}}{8e^{2t}+4e^{3t}+9e^{4t}}
=4-\frac{4e^{-t}(1+4e^{-t})}{9+4e^{-t}+8e^{-2t}}
\]
严格单调递增，并把 \(\RR\) 双射到区间 \((2,4)\)。

对任意 \(\alpha\in(2,4)\)，令 \(t(\alpha)\) 为满足 \(\alpha_6(t)=\alpha\) 的唯一参数，并置
\[
y(\alpha):=e^{-t(\alpha)}.
\]
则 \(y(\alpha)\) 满足二次方程
\[
(8\alpha-16)y^2+(4\alpha-12)y+(9\alpha-36)=0,
\]
因而
\[
y(\alpha)=\frac{3-\alpha+\sqrt{-17\alpha^2+102\alpha-135}}{4(\alpha-2)}.
\]
于是 \(I_6\) 具有显式闭式
\[
I_6(\alpha)
=(4-\alpha)\log y(\alpha)-\log\frac{9+4y(\alpha)+8y(\alpha)^2}{21},
\qquad 2<\alpha<4.
\]
并且端点值满足
\[
I_6(2)=\log\frac{21}{8},
\qquad
I_6\!\left(\frac{64}{21}\right)=0,
\qquad
I_6(4)=\log\frac{21}{9}.
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:conclusion-window6-euler-egf-ode-closure}，
\[
K_6(t)=\log\frac{8e^{2t}+4e^{3t}+9e^{4t}}{21}.
\]
直接求导得到
\[
\alpha_6(t)=K_6'(t)
=\frac{16e^{2t}+12e^{3t}+36e^{4t}}{8e^{2t}+4e^{3t}+9e^{4t}}.
\]
又
\[
K_6''(t)=\Var_{\nu_t}\!\bigl(d_6^{\mathrm{bin}}\bigr)>0,
\]
因为 \(\nu_t\) 在 \(\{2,3,4\}\) 上始终具正权重，故 \(\alpha_6(t)\) 严格递增。由显式公式还可读出
\[
\lim_{t\to-\infty}\alpha_6(t)=2,
\qquad
\lim_{t\to+\infty}\alpha_6(t)=4,
\]
从而其像正是 \((2,4)\)。

令 \(y=e^{-t}>0\)，则
\[
\alpha=\frac{16+12y+36y^2}{8+4y+9y^2},
\]
整理即得
\[
(8\alpha-16)y^2+(4\alpha-12)y+(9\alpha-36)=0.
\]
取其中正根，便得到
\[
y(\alpha)=\frac{3-\alpha+\sqrt{-17\alpha^2+102\alpha-135}}{4(\alpha-2)}.
\]

另一方面，
\[
K_6(t)=-4\log y+\log(9+4y+8y^2)-\log 21,
\]
故在驻点 \(t(\alpha)\) 处，
\[
I_6(\alpha)=\alpha(-\log y)-K_6(t)
=(4-\alpha)\log y-\log\frac{9+4y+8y^2}{21}.
\]
这就得到所示闭式表达。

当 \(\alpha\to 2\) 时，\(y(\alpha)\to+\infty\)，故
\[
I_6(2)=\log\frac{21}{8}.
\]
当 \(\alpha=64/21\) 时，对应 \(t=0\) 与 \(y=1\)，因此
\[
I_6\!\left(\frac{64}{21}\right)=0.
\]
当 \(\alpha\to 4\) 时，\(y(\alpha)\to 0\)，故
\[
I_6(4)=\log\frac{21}{9}.
\]
证毕。
\end{proof}

公开文献中，六顶点高度函数到 Gaussian free field 的严格收敛只在 \(-1\le \Delta\le -1/2\) 区间出现；对于 square ice \((\Delta=1/2)\)，文献所给结论仍主要是去局域化、对数方差与 CLT 型输入，而未出现 ice point 的 GFF 定理，参见 \href{https://arxiv.org/abs/2603.06268}{arXiv:2603.06268}。因此，下述连续极限命题只能保持为猜想。

\begin{conjecture}[ice point 的轴向局部系统猜想]\label{conj:conclusion-window6-icepoint-axial-local-system}
设 square ice 的高度函数在某一临界归一化下收敛到标量 Gaussian free field \(\Psi\)。定义扩展根值映射
\[
\rho_6(w):=
\begin{cases}
\iota_B(w),& w\in X_6^{\mathrm{cyc}},\\
0,& w\in X_6^{\mathrm{bdry}},
\end{cases}
\]

\[
f:\{2,3,4\}\longrightarrow \RR
\]
定义多重度重整化向量观测
\[
\mathcal O_{\delta,f}(z)
:=
f\!\bigl(d_6^{\mathrm{bin}}(\omega_\delta(z))\bigr)\,
\rho_6(\omega_\delta(z)).
\]

\end{conjecture}
