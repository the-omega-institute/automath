\subsection{Bernoulli--\texorpdfstring{$\zeta$}{zeta} 反演、有限矩截断与热力学--算术二层分裂}\label{subsec:conclusion-bernoulli-zeta-moment-truncation-pearson-abel-arithmetic-splitting}
沿用定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy}、定理 \ref{thm:fold-collision-spectrum}、推论 \ref{cor:fold-collision-scale-diverges}、定理 \ref{thm:fold-zero-density-sparse}、定理 \ref{thm:fold-bin-renyi-rate-collapse}、定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic}、命题 \ref{prop:primitive-orbit-surgery} 与定理 \ref{thm:conclusion-realinput40-abel-gap-positive-primitive-mass} 的记号。前文已经分别把折叠标量通道的 Stirling--Bernoulli 层级、连续容量曲线的完备统计性、谱零点的指数稀薄率、以及真实输入 \(40\) 状态核的 primitive/Abel 手术完全显式化。把这些接口并置之后，还可进一步抽取出如下八条后果。其共同结构是：bin-fold 的广延 Rényi 热力学在斜率层面完全塌缩，而 even-\(\zeta\) 算术塔、有限尺度直方图、Pearson 缺陷与原子 Abel 位移则全部滞留在次广延或有限尺度层级，因而形成一条与热力学扁平严格分离的第二结构层。

\begin{theorem}[Bernoulli--\texorpdfstring{$\zeta$}{zeta} 塔的显式三角反演]\label{thm:conclusion-bernoulli-zeta-tower-explicit-triangular-inversion}
记
\[
\alpha_r:=\left(\frac{\varphi}{2}\right)^{2r-1},
\qquad
a_r:=
\frac{B_{2r}}{r(2r-1)}
\bigl(\varphi^{-1}+\varphi^{2r-3}\bigr).
\]
则对每个固定整数 \(r\ge 1\)，都有显式极限反演公式
\[
a_r=
\lim_{m\to\infty}\alpha_r^{-m}
\left(
\log C_m-\log(2\pi)-\sum_{s=1}^{r-1}a_s\alpha_s^m
\right).
\]
从而
\[
B_{2r}
=
\frac{r(2r-1)}{\varphi^{-1}+\varphi^{2r-3}}
\lim_{m\to\infty}
\left(\frac{2}{\varphi}\right)^{(2r-1)m}
\left(
\log C_m-\log(2\pi)-\sum_{s=1}^{r-1}a_s\alpha_s^m
\right),
\]
以及
\[
\zeta(2r)
=
(-1)^{r-1}\frac{(2\pi)^{2r}}{2(2r)!}\,
\frac{r(2r-1)}{\varphi^{-1}+\varphi^{2r-3}}
\lim_{m\to\infty}
\left(\frac{2}{\varphi}\right)^{(2r-1)m}
\left(
\log C_m-\log(2\pi)-\sum_{s=1}^{r-1}a_s\alpha_s^m
\right).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy}，对任意固定 \(R\ge 1\) 都有
\[
\log C_m-\log(2\pi)-\sum_{s=1}^{R-1}a_s\alpha_s^m
=
a_R\alpha_R^m+O(\alpha_{R+1}^m).
\]
又因为
\[
\frac{\alpha_{R+1}}{\alpha_R}
=
\left(\frac{\varphi}{2}\right)^2<1,
\]
两边同乘 \(\alpha_R^{-m}\) 并令 \(m\to\infty\)，即得
\[
a_R=
\lim_{m\to\infty}\alpha_R^{-m}
\left(
\log C_m-\log(2\pi)-\sum_{s=1}^{R-1}a_s\alpha_s^m
\right).
\]
再由
\[
a_r=
\frac{B_{2r}}{r(2r-1)}
\bigl(\varphi^{-1}+\varphi^{2r-3}\bigr),
\]
即可反解出 \(B_{2r}\)。最后代入标准 Bernoulli--\(\zeta\) 恒等式
\[
B_{2r}=(-1)^{r-1}\frac{2(2r)!}{(2\pi)^{2r}}\zeta(2r),
\]
便得 \(\zeta(2r)\) 的显式反演式。证毕。
\end{proof}

\begin{theorem}[容量等价同时刚化广延热力学层与次广延 Bernoulli--\texorpdfstring{$\zeta$}{zeta} 层]\label{thm:conclusion-capacity-equivalence-thermo-arithmetic-rigidity}
\leanverified{paper\_conclusion\_capacity\_equivalence\_thermo\_arithmetic\_rigidity}
设两套 bin-fold 型系统在每个分辨率 \(m\) 上的连续容量曲线满足
\[
\mathcal C_m^{\mathrm{cont}}(T)=\widetilde{\mathcal C}_m^{\mathrm{cont}}(T)
\qquad(\forall\,T\ge 0).
\]
则对每个 \(m\) 与每个 \(q>0\)，都有
\[
S_q(m)=\widetilde S_q(m),
\qquad
H_q(\mu_m)=H_q(\widetilde\mu_m).
\]
并且对应的 fold-gauge 规范常数满足
\[
C_m=\widetilde C_m
\qquad(\forall\,m\ge 1).
\]
进一步地，\(\log C_m\) 渐近展开中的全部系数
\[
a_r,
\qquad
B_{2r},
\qquad
\zeta(2r)
\qquad(r\ge 1)
\]
也逐项一致。换言之，容量等价类不仅在广延 Rényi 热力学上重合，而且会把整条次广延 Bernoulli--\(\zeta\) 算术塔 rigidify 到同一条校正链上。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic}，固定 \(m\) 时连续容量曲线 \(\mathcal C_m^{\mathrm{cont}}\) 与纤维直方图、尾计数函数及全部正阶幂和 \(\{S_q(m)\}_{q>0}\) 彼此唯一决定。故
\[
\mathcal C_m^{\mathrm{cont}}=\widetilde{\mathcal C}_m^{\mathrm{cont}}
\Longrightarrow
S_q(m)=\widetilde S_q(m)
\qquad(\forall q>0).
\]
同一定理还唯一恢复直方图，因此对应的推前分布 \(\mu_m\) 与 \(\widetilde\mu_m\) 仅差重排，故全部 Rényi 熵 \(H_q\) 也逐项一致。

另一方面，由定理 \ref{thm:conclusion-infonce-capacity-wedderburn-unified-closure}，连续容量曲线唯一决定固定分辨率上的 Wedderburn 型；于是两套系统在每个 \(m\) 上都具有相同的半单块直方图。再由定理 \ref{thm:gut-wedderburn-spectrum-determines-bernoulli-zeta-tower}，这强制
\[
C_m=\widetilde C_m
\]
并且全部 Bernoulli 系数 \(B_{2r}\) 与偶值 \(\zeta(2r)\) 逐项一致。最后，\(a_r\) 只是 \(B_{2r}\) 乘上仅依赖 \(r\) 的显式非零因子，故 \(a_r\) 亦一致。证毕。
\end{proof}

\begin{theorem}[有限矩谱的严格决定性与容量曲线的有限截断恢复]\label{thm:conclusion-finite-moment-spectrum-strict-determinacy-truncation}
\leanverified{paper\_conclusion\_finite\_moment\_spectrum\_strict\_determinacy\_truncation}
对固定分辨率 \(m\)，记
\[
h_m(t):=\#\{x\in X_m:\ d_m(x)=t\},
\qquad
M_m:=\max_{x\in X_m}d_m(x),
\]
并对整数 \(n\ge 1\) 记
\[
S_n(m):=\sum_{x\in X_m}d_m(x)^n.
\]
则有限向量
\[
\bigl(h_m(1),\dots,h_m(M_m)\bigr)
\]
被前 \(M_m\) 个整数矩
\[
S_1(m),S_2(m),\dots,S_{M_m}(m)
\]
唯一决定。因而尾计数函数
\[
N_m(T):=\#\{x\in X_m:\ d_m(x)\ge T\},
\]
连续容量曲线
\[
\mathcal C_m^{\mathrm{cont}}(T):=\sum_{x\in X_m}\min\bigl(d_m(x),T\bigr),
\]
以及离散预算容量
\[
C_m(B):=\mathcal C_m^{\mathrm{cont}}(2^B)
\]
都可由这 \(M_m\) 个矩有限恢复。
\end{theorem}
\begin{proof}
对每个 \(n=1,\dots,M_m\)，按纤维大小分层可得
\[
S_n(m)=\sum_{t=1}^{M_m} h_m(t)\,t^n.
\]
这是一组关于未知量 \(\bigl(h_m(1),\dots,h_m(M_m)\bigr)\) 的线性方程，其系数矩阵为
\[
V:=(t^n)_{1\le n,t\le M_m}.
\]
把第 \(t\) 列提出一个 \(t\) 因子，得到
\[
\det V
=
\left(\prod_{t=1}^{M_m}t\right)
\det(t^{n-1})_{1\le n,t\le M_m}
=
\left(\prod_{t=1}^{M_m}t\right)
\prod_{1\le s<t\le M_m}(t-s)\neq 0.
\]
故该线性系统唯一可解，从而向量 \(\bigl(h_m(1),\dots,h_m(M_m)\bigr)\) 被前 \(M_m\) 个整数矩唯一决定。

其后由定义直接有
\[
N_m(T)=\sum_{t\ge \lceil T\rceil}h_m(t),
\qquad
\mathcal C_m^{\mathrm{cont}}(T)=\sum_{t=1}^{M_m}h_m(t)\min(t,T),
\]
并且
\[
C_m(B)=\mathcal C_m^{\mathrm{cont}}(2^B).
\]
故尾计数、连续容量曲线与全部离散预算容量都随之有限恢复。证毕。
\end{proof}

\begin{theorem}[固定尺度矩截断阈值的 sharp 性]\label{thm:conclusion-finite-moment-truncation-sharp-threshold}
\leanverified{paper\_conclusion\_finite\_moment\_truncation\_sharp\_threshold}
沿用定理 \ref{thm:conclusion-finite-moment-spectrum-strict-determinacy-truncation} 的记号。则前 \(M_m\) 个整数矩
\[
S_1(m),\dots,S_{M_m}(m)
\]
唯一恢复纤维直方图 \(\bigl(h_m(1),\dots,h_m(M_m)\bigr)\) 的结论是 sharp 的：存在两组不同的正整数直方图
\[
\mathbf h_m^+\neq \mathbf h_m^-
\]
满足
\[
S_n^+(m)=S_n^-(m)
\qquad(1\le n\le M_m-1),
\]
但
\[
S_{M_m}^+(m)\neq S_{M_m}^-(m).
\]
因而前 \(M_m-1\) 个整数矩在一般情形下严格不足以唯一决定连续容量曲线。
\end{theorem}
\begin{proof}
考虑矩阵
\[
W:=(t^n)_{1\le n\le M_m-1,\ 1\le t\le M_m}.
\]
其秩至多为 \(M_m-1\)，故存在非零向量
\[
v=(v_1,\dots,v_{M_m})\in \ker(W)\cap \ZZ^{M_m}.
\]
取整数
\[
L>\max_{1\le t\le M_m}|v_t|,
\]
并定义两组正整数直方图
\[
\mathbf h_m^{\pm}:=(L\pm v_1,\dots,L\pm v_{M_m}).
\]
则对每个 \(1\le n\le M_m-1\) 都有
\[
\sum_{t=1}^{M_m}(L+v_t)t^n
-
\sum_{t=1}^{M_m}(L-v_t)t^n
=
2\sum_{t=1}^{M_m}v_t t^n
=0,
\]
即
\[
S_n^+(m)=S_n^-(m)
\qquad(1\le n\le M_m-1).
\]

若进一步有
\[
S_{M_m}^+(m)=S_{M_m}^-(m),
\]
则同样得到
\[
\sum_{t=1}^{M_m}v_t t^{M_m}=0.
\]
于是 \(v\) 落入矩阵
\[
W':=(t^n)_{1\le n\le M_m,\ 1\le t\le M_m}
\]
的核。但 \(W'\) 是 Vandermonde 型可逆矩阵，因此 \(\ker(W')=0\)，与 \(v\neq 0\) 矛盾。故第 \(M_m\) 个矩必然分离两者。

最后，由定理 \ref{thm:conclusion-finite-moment-spectrum-strict-determinacy-truncation}，不同直方图诱导不同的连续容量曲线；因此前 \(M_m-1\) 个整数矩不足以统一恢复容量曲线。证毕。
\end{proof}

\begin{corollary}[任何幂和型固定尺度精确审计在最坏情形下恰需 \texorpdfstring{$M_m$}{Mm} 个标量]\label{cor:conclusion-finite-moment-audit-needs-exactly-Mm-scalars}
\leanverified{paper\_conclusion\_finite\_moment\_audit\_needs\_exactly\_mm\_scalars}
设某固定尺度精确审计协议在分辨率 \(m\) 上只读取有限个幂和标量
\[
S_{n_1}(m),\dots,S_{n_r}(m)
\qquad(n_j\ge 1),
\]
并要求对所有支撑于 \(\{1,\dots,M_m\}\) 的纤维直方图都能唯一恢复连续容量曲线。则必有
\[
r\ge M_m.
\]
并且该下界可以达到：取
\[
n_j=j
\qquad(1\le j\le M_m)
\]
时即得到一个最小完备协议。
\end{corollary}
\begin{proof}
可达性由定理 \ref{thm:conclusion-finite-moment-spectrum-strict-determinacy-truncation} 立即给出。

反过来，若 \(r<M_m\)，考虑广义矩阵
\[
U:=(t^{n_j})_{1\le j\le r,\ 1\le t\le M_m}.
\]
则 \(\operatorname{rank}(U)\le r<M_m\)，故存在非零整数向量
\[
v=(v_1,\dots,v_{M_m})\in \ker(U)\cap \ZZ^{M_m}.
\]
与上一定理同样地，取充分大的整数 \(L>\max_t|v_t|\)，并令
\[
\mathbf h_m^{\pm}:=(L\pm v_1,\dots,L\pm v_{M_m}).
\]
则对所有被读取的幂和指数 \(n_j\) 都有
\[
S_{n_j}^+(m)=S_{n_j}^-(m).
\]
但 \(\mathbf h_m^+\neq \mathbf h_m^-\)，故由定理 \ref{thm:conclusion-finite-moment-spectrum-strict-determinacy-truncation} 它们诱导不同的连续容量曲线。这与协议的统一唯一恢复性矛盾。故 \(r\ge M_m\) 是必要的。证毕。
\end{proof}

\begin{theorem}[谱零点稀薄性与 Pearson 非均匀性爆裂的严格分离]\label{thm:conclusion-zero-density-pearson-explosion-strict-separation}
\leanverified{paper\_conclusion\_zero\_density\_pearson\_explosion\_strict\_separation}
记
\[
M:=F_{m+2},
\qquad
p_m(r):=\frac{d_m(r)}{2^m},
\qquad
u_m(r):=\frac{1}{M}
\qquad \bigl(r\in \ZZ/(M\ZZ)\bigr),
\]
以及谱零点集合
\[
\mathcal Z_m:=\{k\in \ZZ/(M\ZZ):\ \widehat d_m(k)=0\}.
\]
则有：
\begin{enumerate}
  \item 若 \(3\mid(m+2)\)（等价地 \(M\) 为偶数），则
  \[
  \frac{|\mathcal Z_m|}{M}
  \le
  \tau(m+2)\,O(\varphi^{-m/2})
  \longrightarrow 0.
  \]
  \item 对全部 \(m\)，Pearson 型非均匀性满足精确恒等式
  \[
  \chi^2(p_m\Vert u_m)
  :=
  M\sum_r\bigl(p_m(r)-u_m(r)\bigr)^2
  =
  M\,\mathsf{Col}_m-1,
  \]
  并且
  \[
  \chi^2(p_m\Vert u_m)\longrightarrow +\infty.
  \]
\end{enumerate}
特别地，沿满足 \(3\mid(m+2)\) 的子列，谱零点的相对密度可以趋于零，而均匀性缺陷却在 \(\chi^2\) 量级上无界放大。
\end{theorem}
\begin{proof}
当 \(3\mid(m+2)\) 时，Fibonacci 数 \(M=F_{m+2}\) 为偶数，故定理 \ref{thm:fold-zero-density-sparse} 直接给出
\[
\frac{|\mathcal Z_m|}{M}
\le
\tau(m+2)\,O(\varphi^{-m/2})
\longrightarrow 0.
\]

另一方面，由定理 \ref{thm:fold-collision-spectrum}，
\[
\sum_r\bigl(p_m(r)-u_m(r)\bigr)^2
=
\mathsf{Col}_m-\frac{1}{M}.
\]
两边同乘 \(M\) 即得
\[
\chi^2(p_m\Vert u_m)
=
M\sum_r\bigl(p_m(r)-u_m(r)\bigr)^2
=
M\,\mathsf{Col}_m-1.
\]
再由推论 \ref{cor:fold-collision-scale-diverges}，
\[
M\,\mathsf{Col}_m\longrightarrow +\infty,
\]
故 \(\chi^2(p_m\Vert u_m)\to+\infty\)。由于该发散沿整个序列成立，因而沿任意满足 \(3\mid(m+2)\) 的子列也同样成立。证毕。
\end{proof}

\begin{proposition}[原子 Witt 因子的 Abel 有限部量子化位移]\label{prop:conclusion-atomic-witt-factor-quantized-abel-shift}
\leanverified{paper\_conclusion\_atomic\_witt\_factor\_quantized\_abel\_shift}
在真实输入 \(40\) 状态核的输出位势加权下，长度 \(2\) 的原子 Euler 因子
\[
(1-uz^2)^{-1}
\]
对应唯一 primitive 坐标
\[
p_n^{\mathrm{cycle}}(u)=u\,\mathbf 1_{\{n=2\}}.
\]
在未扭曲点 \(u=1\) 上，若记
\[
P(z):=P(z,1),\qquad
P_{\mathrm{core}}(z):=P_{\mathrm{core}}(z,1),
\]
\[
\log\mathfrak M:=\log\mathfrak M(1),\qquad
\log\mathfrak M_{\mathrm{core}}:=\log\mathfrak M_{\mathrm{core}}(1),
\qquad
z_\star:=\varphi^{-2},
\]
则有精确分解
\[
P(z)=z^2+P_{\mathrm{core}}(z),
\qquad
\log\mathfrak M=\log\mathfrak M_{\mathrm{core}}+z_\star^2.
\]
特别地，
\[
\Delta\log\mathfrak M:=\log\mathfrak M-\log\mathfrak M_{\mathrm{core}}
=
\varphi^{-4}
=
\frac{7-3\sqrt5}{2}.
\]
换言之，在保持 core primitive 谱不变时，该长度 \(2\) 的原子因子对 Abel 有限部施加一条完全刚性的量子化平移。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-realinput40-atomic-surgery-three-level-invertibility}，
\[
p_n^{\mathrm{cycle}}(u)=u\,\mathbf 1_{\{n=2\}},
\qquad
P(z,u)=uz^2+P_{\mathrm{core}}(z,u).
\]
令 \(u=1\) 即得
\[
P(z)=z^2+P_{\mathrm{core}}(z).
\]
另一方面，定理 \ref{thm:conclusion-realinput40-abel-gap-positive-primitive-mass} 给出对任意 \(u>0\)，
\[
\log\mathfrak M(u)-\log\mathfrak M_{\mathrm{core}}(u)=u\,z_\star(u)^2.
\]
在 \(u=1\) 上有 \(z_\star(1)=\varphi^{-2}\)，故
\[
\Delta\log\mathfrak M
=
z_\star^2
=
\varphi^{-4}
=
\frac{7-3\sqrt5}{2}.
\]
证毕。
\end{proof}

\begin{theorem}[热力学扁平之下的算术刚性]\label{thm:conclusion-thermodynamic-flattening-subextensive-arithmetic-rigidity}
设 \(\mu_m\) 为 bin-fold 的推前测度。则存在两条严格不同的渐近层：
\begin{enumerate}
  \item 对任意 \(q>0\)（其中 \(q=1\) 按 Shannon 熵解释），都有
  \[
  \lim_{m\to\infty}\frac{1}{m}H_q(\mu_m)=\log\varphi.
  \]
  因而广延热力学层完全塌缩到同一条斜率 \(\log\varphi\)。
  \item 与此同时，对任意固定 \(R\ge 1\)，规范重整化序列 \(C_m\) 满足
  \[
  \log C_m
  =
  \log(2\pi)
  +
  \sum_{r=1}^{R}
  \frac{B_{2r}}{r(2r-1)}
  \bigl(\varphi^{-1}+\varphi^{2r-3}\bigr)
  \left(\frac{\varphi}{2}\right)^{(2r-1)m}
  +
  O\!\left(\left(\frac{\varphi}{2}\right)^{(2R+1)m}\right),
  \]
  并且每个偶值 \(\zeta(2r)\) 都可由定理 \ref{thm:conclusion-bernoulli-zeta-tower-explicit-triangular-inversion} 的显式极限公式逐阶恢复。
\end{enumerate}
因此，该折叠系统呈现出一条严格的二层分裂：全部广延 Rényi 热力学只剩单一斜率 \(\log\varphi\)，而完整的 even-\(\zeta\) 算术塔并未消失，而是整体迁移到次广延的规范校正层中。
\end{theorem}
\begin{proof}
第一条正是定理 \ref{thm:fold-bin-renyi-rate-collapse} 的结论，其中 \(q=1\) 按 Shannon 熵的连续延拓解释。

第二条则直接来自定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy} 的完整渐近层级；再结合定理 \ref{thm:conclusion-bernoulli-zeta-tower-explicit-triangular-inversion}，便得到每个偶值 \(\zeta(2r)\) 的逐阶显式恢复。故二层分裂成立。证毕。
\end{proof}

上述八条结果把折叠标量通道、容量层析、碰撞几何与 primitive 手术进一步压缩为同一幅图景：连续容量曲线的等价类不仅 rigidify 全部正阶矩与 Rényi 热力学，而且会通过 Wedderburn 型把整条 Bernoulli--\(\zeta\) 校正塔一并锁死；有限尺度上，纤维直方图并不需要整个实参数矩谱，只需前 \(M_m\) 个整数矩即可唯一恢复，而且这一阈值是 sharp 的，任何只读幂和的最小完备协议在最坏情形下都恰需 \(M_m\) 个标量；频域上，谱零点的相对密度可以趋空，但 Pearson 型非均匀性仍然无界爆裂；而在真实输入 \(40\) 状态核中，被剥离的长度 \(2\) 原子又把这一“次广延记账”具体落实为一条量子化的 Abel 有限部位移。由此，热力学扁平、算术刚性、有限矩 sharp 截断与 primitive 常数修正不再是彼此孤立的现象，而成为同一折叠机制在四个层级上的互补投影。
