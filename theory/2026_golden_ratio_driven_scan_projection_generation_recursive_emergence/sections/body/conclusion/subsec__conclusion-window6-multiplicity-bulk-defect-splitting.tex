\subsection{window-\texorpdfstring{$6$}{6} 的多重度破缺场、长根双块代数与 bulk--defect 谱分解}\label{subsec:conclusion-window6-multiplicity-bulk-defect-splitting}
在定理 \ref{thm:conclusion-window6-multiplicity-pinning} 已把纤维多重度场识别为一个恢复有序旗标的对称性破缺场、定理 \ref{thm:conclusion-window6-coxeter-fold-block-normal-form} 已把 cyclic 扇区写成 short/long 的 Coxeter 块分解、而定理 \ref{thm:conclusion-window6-qutrit-sign-doublet-normal-form} 已把 \(12\) 维长根层压缩为 qutrit--sign--doublet 张量正规形之后，还可进一步把 multiplicity 与 Coxeter 传输的非交换耦合压缩为一条更刚性的 bulk--defect 链。其核心在于：同一个多重度场在长根层上强制两块手征 \(3\times 3\) 全矩阵代数与三次中心关系，而在短根层上只留下一个长度恰为 \(3\) 的幂零 Jordan 缺陷；二者直和后便得到 cyclic 非交换子的半单--幂零二分解。

\begin{theorem}[长根层生成代数的精确双块化]\label{thm:conclusion-window6-long-root-generated-algebra-doubleblock}
\leanverified{paper\_conclusion\_window6\_long\_root\_generated\_algebra\_doubleblock}
沿用定理 \ref{thm:conclusion-window6-qutrit-sign-doublet-normal-form} 的长根层
\[
A_{\ell}\cong \CC^3\otimes \CC^2_{\mathrm{sgn}}\otimes \CC^2_{\mathrm{orb}},
\]
并记
\[
M:=\operatorname{diag}(4,2,3),
\qquad
D_{\ell}:=D_6|_{A_{\ell}}=M\otimes I_2\otimes I_2,
\]
\[
U_{\ell}:=U_{\mathfrak c}|_{A_{\ell}}=Q_3\otimes S_2\otimes I_2,
\qquad
\mathcal B_{\ell}:=\CC\langle D_{\ell},U_{\ell}\rangle.
\]
则有精确代数同构
\[
\mathcal B_{\ell}
\cong
\bigl(M_3(\CC)\oplus M_3(\CC)\bigr)\otimes I_2.
\]
因此，window-\(6\) 的长根层并非一般 \(12\) 维矩阵系统，而是精确分裂为两块手征 \(3\times 3\) 全矩阵代数，再张量一个 orbit doublet 因子。
\end{theorem}
\begin{proof}
取酉矩阵 \(R\in U(2)\) 使
\[
R^{-1}S_2R=\operatorname{diag}(1,-1).
\]
在基变换 \(I_3\otimes R\otimes I_2\) 下，
\[
D_{\ell}\sim (M\oplus M)\otimes I_2,
\qquad
U_{\ell}\sim (Q_3\oplus -Q_3)\otimes I_2.
\]
故只需证明
\[
\CC\langle M,Q_3\rangle=M_3(\CC).
\]
由于 \(M\) 的三个特征值 \(4,2,3\) 两两不同，Lagrange 插值给出三个对角矩阵单位
\[
E_{11},\qquad E_{22},\qquad E_{33}\in \CC[M].
\]
再由 \(Q_3\) 与 \(Q_3^2\) 可得全部矩阵单位
\[
E_{12},E_{23},E_{31},E_{13},E_{21},E_{32},
\]
从而 \(\CC\langle M,Q_3\rangle=M_3(\CC)\)。同理
\[
\CC\langle M,-Q_3\rangle=M_3(\CC).
\]
因此
\[
\mathcal B_{\ell}
=
\bigl(\CC\langle M,Q_3\rangle\oplus \CC\langle M,-Q_3\rangle\bigr)\otimes I_2
\cong
\bigl(M_3(\CC)\oplus M_3(\CC)\bigr)\otimes I_2.
\]
证毕。
\end{proof}

\begin{corollary}[长根层对易子代数的闭式]\label{cor:conclusion-window6-long-root-commutant-closed}
\leanverified{paper\_conclusion\_window6\_long\_root\_commutant\_closed}
在定理 \ref{thm:conclusion-window6-long-root-generated-algebra-doubleblock} 的记号下，
\[
\operatorname{Cent}_{\End(A_{\ell})}(\mathcal B_{\ell})
=
\operatorname{Cent}_{\End(A_{\ell})}(D_{\ell},U_{\ell})
\cong
(\CC\oplus \CC)\otimes M_2(\CC).
\]
并且
\[
U_{\ell}^3=I_3\otimes S_2\otimes I_2,
\qquad
\CC[U_{\ell}^3]\cong \CC\oplus \CC,
\]
从而
\[
\operatorname{Cent}_{\End(A_{\ell})}(D_{\ell},U_{\ell})
\cong
\CC[U_{\ell}^3]\otimes M_2(\CC).
\]
特别地，其复维数等于 \(8\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-window6-long-root-generated-algebra-doubleblock}，
\[
\mathcal B_{\ell}\cong \bigl(M_3(\CC)\oplus M_3(\CC)\bigr)\otimes I_2.
\]
有限维双对易子定理立即给出
\[
\operatorname{Cent}_{\End(A_{\ell})}(\mathcal B_{\ell})
\cong
(\CC\oplus \CC)\otimes M_2(\CC).
\]
另一方面，
\[
U_{\ell}^3
=
(Q_3\otimes S_2\otimes I_2)^3
=
I_3\otimes S_2\otimes I_2.
\]
在对角化 \(S_2\) 后，\(U_{\ell}^3\) 恰对应两条手征块上的 \((+1)\) 与 \((-1)\) 投影，因此
\[
\CC[U_{\ell}^3]\cong \CC\oplus \CC.
\]
合并两式即得结论。证毕。
\end{proof}

\begin{theorem}[三次中心关系]\label{thm:conclusion-window6-cubic-central-relation}
\leanverified{paper\_conclusion\_window6\_cubic\_central\_relation}
对任意
\[
M(a,b,c):=\operatorname{diag}(a,b,c),
\qquad
Q_3=
\begin{pmatrix}
0&0&1\\
1&0&0\\
0&1&0
\end{pmatrix},
\]
恒有精确恒等式
\[
[M(a,b,c),Q_3]^3=(a-b)(a-c)(b-c)\,I_3.
\]
特别地，对
\[
M=\operatorname{diag}(4,2,3)
\]
有
\[
[M,Q_3]^3=-2I_3.
\]
相应地，若在长根层上记
\[
C_{\ell}:=[D_{\ell},U_{\ell}],
\]
则
\[
C_{\ell}^3=-2\,U_{\ell}^3,
\qquad
C_{\ell}^6=4I_{12}.
\]
\end{theorem}
\begin{proof}
直接计算得
\[
[M(a,b,c),Q_3]
=
\begin{pmatrix}
0&0&a-c\\
b-a&0&0\\
0&c-b&0
\end{pmatrix}.
\]
再乘方可得
\[
[M(a,b,c),Q_3]^2
=
\begin{pmatrix}
0&(a-c)(c-b)&0\\
0&0&(b-a)(a-c)\\
(c-b)(b-a)&0&0
\end{pmatrix},
\]
从而
\[
[M(a,b,c),Q_3]^3
=(a-c)(c-b)(b-a)\,I_3
=(a-b)(a-c)(b-c)\,I_3.
\]
代入 \((a,b,c)=(4,2,3)\) 即得
\[
[M,Q_3]^3=-2I_3.
\]

另一方面，由定理 \ref{thm:conclusion-window6-qutrit-sign-doublet-normal-form}，
\[
D_{\ell}=M\otimes I_2\otimes I_2,
\qquad
U_{\ell}=Q_3\otimes S_2\otimes I_2.
\]
故
\[
C_{\ell}=[D_{\ell},U_{\ell}]=[M,Q_3]\otimes S_2\otimes I_2.
\]
于是
\[
C_{\ell}^3
=[M,Q_3]^3\otimes S_2^3\otimes I_2
=-2I_3\otimes S_2\otimes I_2
=-2U_{\ell}^3.
\]
再平方即得
\[
C_{\ell}^6=4I_{12}.
\]
证毕。
\end{proof}

\begin{proposition}[短根层缺陷的三阶幂零性]\label{prop:conclusion-window6-short-root-defect-nilpotent}
\leanverified{paper\_conclusion\_window6\_short\_root\_defect\_nilpotent}
沿用定理 \ref{thm:conclusion-window6-coxeter-fold-block-normal-form} 的短根块顺序
\[
(e_1,\ e_2,\ e_3,\ -e_1,\ -e_2,\ -e_3),
\]
并记
\[
D_{\mathrm{s}}=\operatorname{diag}(4,4,4,2,4,4),
\qquad
N_{\mathrm{s}}:=[D_{\mathrm{s}},P_6].
\]
则
\[
N_{\mathrm{s}}=-2E_{4,3}+2E_{5,4},
\]
因而
\[
N_{\mathrm{s}}^2=-4E_{5,3},
\qquad
N_{\mathrm{s}}^3=0,
\qquad
N_{\mathrm{s}}^2\neq 0.
\]
特别地，
\[
\operatorname{rank}N_{\mathrm{s}}=2,
\qquad
\operatorname{rank}N_{\mathrm{s}}^2=1.
\]
故 short sector 中的非交换性并非谱型传播，而是一个长度恰为 \(3\) 的 Jordan 型局域缺陷；其支撑链正是
\[
e_3\longrightarrow -e_1\longrightarrow -e_2.
\]
\end{proposition}
\begin{proof}
在上述顺序下，\(P_6\) 把基向量沿
\[
e_1\to e_2\to e_3\to -e_1\to -e_2\to -e_3\to e_1
\]
循环推进。由于相邻对角元只在
\[
(4,4,4,\underline{2},4,4)
\]
的第三至第五位置发生跳变，故交换子仅有两项非零，直接计算得
\[
N_{\mathrm{s}}=[D_{\mathrm{s}},P_6]=-2E_{4,3}+2E_{5,4}.
\]
于是
\[
N_{\mathrm{s}}^2=(-2E_{4,3}+2E_{5,4})^2=-4E_{5,3},
\]
再乘一次即为零。秩结论随之立即得到。证毕。
\end{proof}

\begin{theorem}[cyclic 非交换子的最小多项式分解]\label{thm:conclusion-window6-cyclic-commutator-minpoly-splitting}
\leanverified{paper\_conclusion\_window6\_cyclic\_commutator\_minpoly\_splitting}
在 cyclic 扇区
\[
\CC^{X_6^{\mathrm{cyc}}}=\CC^{\mathcal O_{\mathrm{s}}}\oplus A_{\ell}
\]
上记
\[
D_{6,\mathrm{cyc}}:=D_{\mathrm{s}}\oplus D_{\ell},
\qquad
U_{6,\mathrm{cyc}}:=P_6\oplus U_{\ell},
\qquad
K_6:=[D_{6,\mathrm{cyc}},U_{6,\mathrm{cyc}}].
\]
则
\[
\chi_{K_6}(\lambda)=\lambda^6(\lambda^6-4)^2,
\qquad
\mu_{K_6}(\lambda)=\lambda^3(\lambda^6-4).
\]
因此，window-\(6\) 的 cyclic 非交换子具有严格的半单--幂零二分解：一方面存在由方程
\[
\lambda^6=4
\]
控制的长根 bulk；另一方面存在一个三阶幂零的 short-root defect。
\end{theorem}
\begin{proof}
由定义，
\[
K_6
=
[D_{\mathrm{s}},P_6]\oplus [D_{\ell},U_{\ell}]
=
N_{\mathrm{s}}\oplus C_{\ell}.
\]
由命题 \ref{prop:conclusion-window6-short-root-defect-nilpotent}，
\[
\chi_{N_{\mathrm{s}}}(\lambda)=\lambda^6,
\qquad
\mu_{N_{\mathrm{s}}}(\lambda)=\lambda^3.
\]

再由定理 \ref{thm:conclusion-window6-long-root-generated-algebra-doubleblock} 的证明，存在基变换使
\[
C_{\ell}\sim \bigl([M,Q_3]\oplus -[M,Q_3]\bigr)\otimes I_2.
\]
而定理 \ref{thm:conclusion-window6-cubic-central-relation} 给出
\[
[M,Q_3]^3=-2I_3.
\]
对这里的显式矩阵
\[
[M,Q_3]=
\begin{pmatrix}
0&0&1\\
-2&0&0\\
0&1&0
\end{pmatrix},
\]
直接计算可得
\[
\chi_{[M,Q_3]}(\lambda)=\lambda^3+2,
\qquad
\chi_{-[M,Q_3]}(\lambda)=\lambda^3-2.
\]
因此
\[
\chi_{C_{\ell}}(\lambda)
=(\lambda^3+2)^2(\lambda^3-2)^2
=(\lambda^6-4)^2.
\]
又因为
\[
C_{\ell}^3=-2U_{\ell}^3,
\qquad
(U_{\ell}^3)^2=I_{12},
\qquad
U_{\ell}^3\neq \pm I_{12},
\]
故 \(C_{\ell}\) 同时满足 \((C_{\ell}^3-2I)(C_{\ell}^3+2I)=0\)，但任何单一三次因子都不能湮灭 \(C_{\ell}\)。于是
\[
\mu_{C_{\ell}}(\lambda)=\lambda^6-4.
\]

最后，对直和取特征多项式乘积、取最小多项式的最小公倍式，即得
\[
\chi_{K_6}(\lambda)=\chi_{N_{\mathrm{s}}}(\lambda)\chi_{C_{\ell}}(\lambda)
=\lambda^6(\lambda^6-4)^2,
\]
\[
\mu_{K_6}(\lambda)
=
\operatorname{lcm}\bigl(\mu_{N_{\mathrm{s}}}(\lambda),\mu_{C_{\ell}}(\lambda)\bigr)
=
\operatorname{lcm}(\lambda^3,\lambda^6-4)
=
\lambda^3(\lambda^6-4).
\]
证毕。
\end{proof}

\begin{conjecture}[偶数窗的半单 bulk--幂零 defect 普适律]\label{conj:conclusion-window6-evenwindow-bulk-defect-universality}
对任意偶数窗口 \(m\)，设其 cyclic sector 已被某个不可约秩 \(r\) 根系刚性选定，并令
\[
K_m:=[D_m^{\mathrm{cyc}},U_m]
\]
为 multiplicity--Coxeter 对易子。则应存在规范分解
\[
K_m\sim K_m^{\mathrm{ss}}\oplus K_m^{\mathrm{nil}},
\]
满足：
\begin{enumerate}
  \item \(K_m^{\mathrm{ss}}\) 被某个无重根的标量多项式 \(p_m(\lambda)\) 消去；
  \item \(K_m^{\mathrm{nil}}\) 的支撑恰等于由多重度不对称选出的 exceptional short-root defect 集；
  \item \(K_m^{\mathrm{nil}}\) 的幂零指数等于该 defect 链的长度。
\end{enumerate}
对 \(m=6\)，定理 \ref{thm:conclusion-window6-cyclic-commutator-minpoly-splitting} 给出这一原理的首个严格实例：
\[
p_6(\lambda)=\lambda^6-4,
\qquad
(K_6^{\mathrm{nil}})^3=0.
\]
若该猜想成立，则信息层中的非均匀多重度将不再只是组合统计，而将与一条规范的非交换谱分解严格同一：bulk 部分记录可对角化的长根传播壳，defect 部分记录局域 Jordan 缺陷，而 boundary 中心则独立承载 rank-\(3\) 的 sheet parity。
\end{conjecture}

\paragraph{统一读法}
上述链条说明：window-\(6\) 的 fibred multiplicity 并非附着在 \(B_3\) 根多面体上的次要标签，而本身就是一个对称性破缺场。它先在长根层上把十二条根切成三张严格有序的 \((A_1\times A_1)\) 坐标平面，并恢复有序旗标；随后在矩阵层强制
\[
\mathcal B_{\ell}\cong \bigl(M_3(\CC)\oplus M_3(\CC)\bigr)\otimes I_2,
