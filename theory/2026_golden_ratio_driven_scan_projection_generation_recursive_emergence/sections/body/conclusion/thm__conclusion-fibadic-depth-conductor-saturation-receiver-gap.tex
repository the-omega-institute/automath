\begin{theorem}[精确深度的导体饱和]\label{thm:conclusion-fibadic-exact-depth-conductor-saturation}
\leanverified{paper\_conclusion\_fibadic\_exact\_depth\_conductor\_saturation}
对每个 \(d=1\) 或 \(d\ge 3\)，定义
\[
S_d:=\{N\ge 1:\ d_N=d\},
\qquad
M_d:=\operatorname{lcm}\{N:\ N\in S_d\}.
\]
则成立严格恒等式
\[
M_d=F_d.
\]
\end{theorem}
\begin{proof}
先证 \(M_d\mid F_d\)。若 \(N\in S_d\)，则由定理
\ref{thm:conclusion-fibadic-visible-depth-mobius-count} 与定理
\ref{thm:conclusion-fibadic-minimal-resolution-for-prime-register-package}，
\[
d_N=d
\iff
N\mid F_d.
\]
故 \(S_d\) 中每个 \(N\) 都整除 \(F_d\)，从而其最小公倍数 \(M_d\) 也整除 \(F_d\)。

当 \(d=1\) 时，平凡特征的导体为 \(1\)，故 \(1\in S_1\)，于是 \(M_1=1=F_1\)。以下设 \(d\ge 3\)。

由定理 \ref{thm:conclusion-fibadic-visible-depth-mobius-count}，
\[
a_d=\sum_{e\mid d}\mu(d/e)F_e.
\]
对任意真因子 \(e<d\) 都有 \(e\le d/2\)，故
\[
a_d
\ge
F_d-\sum_{\substack{e\mid d\\ e<d}}F_e
\ge
F_d-\sum_{j=1}^{\lfloor d/2\rfloor}F_j
=
F_d-F_{\lfloor d/2\rfloor+2}+1
>0.
\]
于是存在某个连续特征 \(\chi\) 满足 \(\delta(\chi)=d\)。记其普通导体为 \(N_0\)，则
\[
N_0\in S_d.
\]

现取任意素幂 \(p^k\Vert F_d\)。由定理
\ref{thm:conclusion-fibadic-minimal-resolution-for-prime-register-package}，
\[
p^k\mid F_d
\iff
z(p^k)\mid d.
\]
再由 \(d_{N_0}=d\) 与 \(z(p^k)\mid d\)，得到
\[
d_{\operatorname{lcm}(N_0,p^k)}
=
\operatorname{lcm}(d_{N_0},z(p^k))
=
\operatorname{lcm}(d,z(p^k))
=
d.
\]
故 \(\operatorname{lcm}(N_0,p^k)\in S_d\)，从而 \(p^k\mid M_d\)。遍历 \(F_d\) 的全部素幂因子，即得 \(F_d\mid M_d\)。综上，
\[
M_d=F_d.
\]
证毕。
\end{proof}

\begin{theorem}[线性上真新生而代数上全穷尽]\label{thm:conclusion-fibadic-exact-depth-linear-thin-algebraic-saturation}
对每个 \(d\ge 3\)，精确深度新空间 \(\mathcal N_d\) 满足
\[
\mathcal N_d\subsetneq \mathcal H_d,
\]
但由 \(\mathcal N_d\) 生成的含单位 \(^*\)-代数恰为整个有限层代数 \(\mathcal H_d\)。等价地，
\[
\bigcap_{\chi:\ \delta(\chi)=d}\ker\chi
=
F_d\widehat{\ZZ},
\]
并且 exact-depth-\(d\) 角色已在有限商层面饱和整个
\[
\ZZ/(F_d\ZZ).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-newspace-orthogonal-decomposition}，
\[
\mathcal H_d=\bigoplus_{e\mid d}\mathcal N_e.
\]
由于 \(\mathcal N_1\neq 0\)，故当 \(d>1\) 时，
\[
\mathcal N_d\subsetneq \mathcal H_d.
\]

另一方面，由定理 \ref{thm:conclusion-fibadic-exact-depth-conductor-saturation}，存在 \(N\in S_d\) 满足
\[
N=F_d.
\]
于是存在某个连续特征 \(\chi\) 满足
\[
\delta(\chi)=d,
\qquad
\operatorname{cond}(\chi)=F_d.
\]
该特征经由有限商
\[
\pi_d:\widehat X_F\twoheadrightarrow \ZZ/(F_d\ZZ)
\]
因子化，并在此循环群上给出一个生成元角色。故
\[
1,\chi,\chi^2,\dots,\chi^{F_d-1}
\]
恰好张成 \(\ZZ/(F_d\ZZ)\) 的全部角色，也即张成 \(\mathcal H_d\)。由于 \(\chi\in \mathcal N_d\)，由 \(\mathcal N_d\) 生成的含单位 \(^*\)-代数必包含这全部幂次，从而等于 \(\mathcal H_d\)。

再看核。任意满足 \(\delta(\psi)=d\) 的角色都经由 \(\pi_d\) 因子化，故
\[
F_d\widehat{\ZZ}\subseteq \ker\psi.
\]
对全部此类 \(\psi\) 取交，得到
\[
F_d\widehat{\ZZ}
\subseteq
\bigcap_{\psi:\ \delta(\psi)=d}\ker\psi.
\]
而对上面所取的导体恰为 \(F_d\) 的生成元角色 \(\chi\)，有
\[
\ker\chi=F_d\widehat{\ZZ}.
\]
故反向包含也成立，于是
\[
\bigcap_{\psi:\ \delta(\psi)=d}\ker\psi
=
F_d\widehat{\ZZ}.
\]
证毕。
\end{proof}

\begin{theorem}[导体预算的乘法分裂]\label{thm:conclusion-fibadic-conductor-budget-new-old-splitting}
\leanverified{paper\_conclusion\_fibadic\_conductor\_budget\_new\_old\_splitting}
对每个 \(d\ge 3\)，定义
\[
Q_d:=\frac{F_d}{\Pi_d(1)}.
\]
则有精确乘法分裂
\[
Q_d=\prod_{\substack{e\mid d\\ 1<e<d}}\Pi_e(1),
\qquad
F_d=\Pi_d(1)\cdot \prod_{\substack{e\mid d\\ 1<e<d}}\Pi_e(1).
\]
此外，对任意素数 \(p\)，成立
\[
v_p(Q_d)
=
\#\{k\ge 1:\ z(p^k)\mid d,\ z(p^k)<d\}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-depth-packet-primitive-factorization}，
\[
F_d=\prod_{\substack{e\mid d\\ e>1}}\Pi_e(1).
\]
两边除以 \(\Pi_d(1)\) 即得
\[
Q_d=\prod_{\substack{e\mid d\\ 1<e<d}}\Pi_e(1).
\]

再由定理 \ref{thm:conclusion-fibadic-cyclotomic-packet-padic-content}，
\[
v_p(\Pi_e(1))
=
\#\{k\ge 1:\ z(p^k)=e\}.
\]
故
\[
v_p(Q_d)
=
\sum_{\substack{e\mid d\\ 1<e<d}}v_p(\Pi_e(1))
=
\sum_{\substack{e\mid d\\ 1<e<d}}
\#\{k\ge 1:\ z(p^k)=e\}
=
\#\{k\ge 1:\ z(p^k)\mid d,\ z(p^k)<d\}.
\]
证毕。
\end{proof}

\begin{theorem}[混合隐藏态的精确深度最小受体刚性]\label{thm:conclusion-mixed-hidden-state-exact-depth-minimal-receiver}
固定一个可见屏幕 \(S\)，并记
\[
\beta:=\beta_n(\mathcal Q_m,A_{\neg S};\FF_2).
\]
对每个 \(d\ge 3\)，设有限阿贝尔群 \(G\) 承载一条连续观测
\[
\phi:\mathcal H_S\to G,
\qquad
\mathcal H_S=K_S^{\mathrm{col}}\times \widehat X_F,
\]
并满足：
\begin{enumerate}
  \item \(\phi|_{K_S^{\mathrm{col}}}\) 为单射；
  \item \(\phi|_{\widehat X_F}\) 的核恰为 \(F_d\widehat{\ZZ}\)。
\end{enumerate}
则此类观测存在，当且仅当
\[
(\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ)\hookrightarrow G.
\]
特别地，最小可能基数恰为
\[
|G|_{\min}=2^\beta F_d,
\]
且达到该下界时必有唯一同构类型
\[
G\cong (\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-exact-depth-linear-thin-algebraic-saturation}，exact-depth-\(d\) 地址方向上的最小忠实有限商正是
\[
\widehat X_F/F_d\widehat{\ZZ}\cong \ZZ/(F_d\ZZ).
\]
因此，满足第二条的连续观测在地址方向上恰等价于一个经由 \(\ZZ/(F_d\ZZ)\) 的忠实有限读出。

另一方面，由定理 \ref{thm:conclusion-mixed-hidden-state-finite-abelian-classification}，
\[
K_S^{\mathrm{col}}\cong (\ZZ/2\ZZ)^\beta.
\]
于是同时满足两条条件的观测，等价于把有限截断混合态
\[
\mathcal H_{S,F_d}
=
K_S^{\mathrm{col}}\times \ZZ/(F_d\ZZ)
\cong
(\ZZ/2\ZZ)^\beta\times \ZZ/(F_d\ZZ)
\]
单射送入 \(G\)。

因此，存在性当且仅当
\[
(\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ)\hookrightarrow G.
\]
最小基数与极小同构类型则直接由定理
\ref{thm:conclusion-mixed-hidden-state-minimal-finite-receiver}
在 \(N=F_d\) 的特例给出，即
\[
|G|_{\min}=2^\beta F_d,
\qquad
G\cong (\ZZ/2\ZZ)^\beta\oplus \ZZ/(F_d\ZZ)
\]
当且仅当达到最小值。证毕。
\end{proof}

\begin{corollary}[三倍数深度的奇偶分裂门槛]\label{cor:conclusion-fibadic-depth-triple-divisibility-odd-address-threshold}\leanverified{paper\_conclusion\_fibadic\_depth\_triple\_divisibility\_odd\_address\_threshold}
对每个 \(d\ge 3\)，若只要求地址方向 exact-depth-\(d\) 忠实，而不要求碰撞扭量方向单射，则最小纯地址受体的基数恰为 \(F_d\)。并且
\[
F_d\ \text{为奇数}
\iff
3\nmid d.
\]
因此，当且仅当 \(3\nmid d\) 时，存在一个既是最小纯地址受体、又对碰撞扭量完全失明的有限阿贝尔受体；反之若 \(3\mid d\)，则任意最小纯地址受体都不可避免地带有非平凡 \(2\)-主分量。
\end{corollary}
\begin{proof}
若不要求碰撞扭量方向单射，则只需在地址方向上达到核为 \(F_d\widehat{\ZZ}\) 的忠实读出。由定理
\ref{thm:conclusion-fibadic-exact-depth-linear-thin-algebraic-saturation}，最小此类有限商正是
\[
\ZZ/(F_d\ZZ),
\]
故最小纯地址受体基数为 \(F_d\)。

再由 Fibonacci 数列模 \(2\) 的周期律
\[
1,1,0,1,1,0,\dots,
\]
可知
\[
F_d\equiv 0\pmod 2
\iff
3\mid d.
\]
若 \(3\nmid d\)，则 \(F_d\) 为奇数，故最小纯地址受体 \(\ZZ/(F_d\ZZ)\) 也是奇阶有限阿贝尔群。由推论
\ref{cor:conclusion-mixed-hidden-state-odd-order-blindness}，任何奇阶受体都对碰撞扭量完全失明，故其自动满足“完全碰撞盲”。

反之，若 \(3\mid d\)，则 \(F_d\) 为偶数，故任意阶为 \(F_d\) 的有限阿贝尔群都含非平凡 \(2\)-主分量，于是最小纯地址受体不可能避开偶主成分。证毕。
\end{proof}

\begin{conjecture}[新生内容主导]\label{conj:conclusion-fibadic-new-content-dominates-conductor}
当 \(d\to\infty\) 且 \(d\ge 3\) 时，成立
\[
\log Q_d=o(d),
\qquad\text{等价地}\qquad
\Pi_d(1)=F_d^{\,1-o(1)}.
\]
\end{conjecture}

\noindent
这表明，尽管 exact-depth-\(d\) 的最小忠实地址预算由整个 \(F_d\) 控制，但其大部分算术内容应当来自首次在深度 \(d\) 出现的真正新生素幂频率 \(\Pi_d(1)\)，而来自真因子深度的遗产项 \(Q_d\) 只占次指数级修正。若该猜想成立，则前述
\[
\frac{a_d}{F_d}\to 1
\]
的维数主导律将进一步提升为导体主导律：不仅新空间 \(\mathcal N_d\) 在维数上几乎饱和，连其有限受体预算中的绝大部分也由真正的新生内容承担。
