\begin{theorem}[有限窗口上的连通 Lie 因子失效]\label{thm:conclusion-finite-window-connected-lie-factor-collapse}
\leanverified{paper\_conclusion\_finite\_window\_connected\_lie\_factor\_collapse}

\[
\pi_0(K):=K/K^\circ
\]
的像决定。

特别地，若某个紧 Lie 群作用的轨道分解恰实现 window-\(6\) 的 bin-fold 纤维分解
\[
F_6:\Omega_6\to X_6,
\]
则该实现在有限窗口上必经由一个有限群作用；结合定理 \ref{thm:xi-window6-group-quotient-isotropy-strata} 与定理 \ref{thm:xi-time-part67-window6-nonfree-group-quotient-obstruction}，其最小对称对象必携带随点变化的稳定子，因此自然落在 Lie 群胚而非主 Lie 群商。
\end{theorem}
\begin{proof}
作用给出连续同态
\[
\rho:K^\circ\longrightarrow \mathrm{Sym}(X).
\]
右端是有限离散群，故 \(\rho(K^\circ)\) 既连通又离散，只能是平凡子群。于是 \(K^\circ\) 固定 \(X\) 的每一点，全部轨道信息均由 \(K\to \pi_0(K)\) 的离散像承载。

对 window-\(6\) 而言，若某个紧 Lie 群作用的轨道分解与 \(F_6\) 的纤维分解一致，则上一段说明该轨道分解已由其在 \(\mathrm{Sym}(\Omega_6)\) 中的有限像实现。再由定理 \ref{thm:xi-window6-group-quotient-isotropy-strata}，任何这样的有限群实现都强制三层稳定子分层；而定理 \ref{thm:xi-time-part67-window6-nonfree-group-quotient-obstruction} 进一步排除了自由群商与阿贝尔线性陪集两类刚性实现。故 window-\(6\) 的最小对称描述必须允许稳定子随点变化，其自然语言正是 Lie 群胚。证毕。
\end{proof}

\begin{theorem}[有限素数局部化数系的 Pontryagin 对偶刚性分类]\label{thm:conclusion-finite-prime-localization-pontryagin-rigidity}
\leanverified{paper\_conclusion\_finite\_prime\_localization\_pontryagin\_rigidity}
对有限素数集 \(S,T\subset \mathbb P\)，记
\[
G_S:=\ZZ[S^{-1}],
\qquad
\Sigma_S:=\widehat{G_S}.
\]
则下列三条件等价：
\[
\Sigma_S\cong \Sigma_T
\quad\text{（作为紧交换群）},
\]
\[
G_S\cong G_T
\quad\text{（作为离散交换群）},
\]
\[
S=T.
\]
换言之，有限局部化数系在 Pontryagin 对偶层面不存在任何非平凡的 prime-register 伪装。
\end{theorem}
\begin{proof}
Pontryagin 对偶在离散交换群与紧交换群之间给出反变等价，因此
\[
\Sigma_S\cong \Sigma_T
\iff
G_S\cong G_T.
\]
以下只需证明
\[
G_S\cong G_T
\iff
S=T.
\]

若 \(S=T\) 则显然。反过来，设
\[
\Phi:G_S\xrightarrow{\sim}G_T
\]
为群同构。对任意素数 \(p\)，性质
\[
[p]:G_S\to G_S\ \text{为满射}
\]
当且仅当 \(p\in S\)：若 \(p\in S\)，则对任意 \(x\in G_S\) 都有 \(x/p\in G_S\)，故 \([p]\) 满射；若 \(p\notin S\)，则 \(1\in G_S\) 不可能写成 \(py\) 且 \(y\in G_S\)，因为这将迫使 \(y=1/p\notin G_S\)。

而乘法映射的满射性由群同构保持，故
\[
p\in S
\iff
[p]\text{ 在 }G_S\text{ 上满射}
\iff
[p]\text{ 在 }G_T\text{ 上满射}
\iff
p\in T.
\]
于是 \(S=T\)。

最后，由定理 \ref{thm:conclusion-finite-prime-solenoid-terminal-object} 的短正合列
\[
0\to \prod_{p\in S}\ZZ_p\to \Sigma_S\to \TT\to 0,
\]
上述等价亦可理解为：圆因子只给出唯一的一维 Lie 可见层，而真正完成分类的全部离散算术信息都集中在核的 \(p\)-主分量上。证毕。
\end{proof}

:=

\]
则有严格闭式

\]

\kappa_2(Y_n)=2n,\qquad
\kappa_4(Y_n)=-4n,\qquad
\kappa_6(Y_n)=32n,

且所有奇数阶累积量严格为零。

换言之，五分支因子 \(5\) 只在未归一化配分函数中贡献一个加法常数 \(\log 5\)，却不贡献任何高阶体相涨落；branch multiplicity 与 body-address fluctuations 在累积量层严格分裂。

\frac{Z_n(t)}{Z_n(0)}
=
\frac{5(2\cosh t)^{2n}}{5\cdot 4^n}

(\cosh t)^{2n}.
\]
故
\[
\log \EE\!\left[e^{tY_n}\right]=2n\log\cosh t.
\]
右端正是 \(2n\) 个独立 Rademacher 变量之和的累积母函数，因此全部累积量逐阶一致。

把它与

\[
\kappa_2(Y_n)=2n,\qquad
\kappa_4(Y_n)=4!\left(-\frac{n}{6}\right)=-4n,
\qquad
\kappa_6(Y_n)=6!\left(\frac{2n}{45}\right)=32n.
\]
证毕。
\end{proof}

\begin{theorem}[有限素数账本的一圆可见上界与不分裂壁垒]\label{thm:conclusion-finite-prime-one-circle-upper-bound-nonsplit-barrier}

\begin{enumerate}

\end{enumerate}

\begin{enumerate}

  \[

  \]
  给出；
  \item 任意连续环面值观测
  \[
  f:\Sigma_S\to \TT^n

  \item 零维账本核 \(P_S\) 不能被任何额外的连续圆因子替代。
\end{enumerate}

\qquad
P_S=\prod_{p\in S}\ZZ_p,
\]

0\to P_S\to \Sigma_S\to \TT\to 0
\]
严格不分裂。故此类分离模型不存在。更一般的结论只是同一矛盾的重述。证毕。
\end{proof}

\begin{theorem}[单位圆零点约束下的对数振幅全息刚性]\label{thm:conclusion-unit-circle-zero-logamplitude-holographic-rigidity}
\leanverified{paper\_conclusion\_unit\_circle\_zero\_logamplitude\_holographic\_rigidity}
对任意有限支撑素数账本
\[
r\in \NN^{(\mathbb P)},
\qquad
L(r)(z):=\prod_{p\in \mathbb P}(1+z^p)^{r(p)},
\qquad
A_r(\theta):=\log\bigl|L(r)(e^{i\theta})\bigr|,
\]
成立：
\begin{enumerate}
  \item \(L(r)\) 的全部零点都落在单位圆 \(S^1\)，并且都是偶阶单位根；
  \item 对每个素数 \(p\)，有逐素 Fourier 反演公式
  \[
  r(p)=\frac{1}{\pi}\int_{-\pi}^{\pi}A_r(\theta)\cos(p\theta)\,\dd\theta.
  \]
\end{enumerate}
因此映射 \(r\mapsto A_r\) 为单射。更强地，对任意 \(r,s\in \NN^{(\mathbb P)}\)，都有
\[
\|A_r-A_s\|_{L^2[-\pi,\pi]}
\ge
\sqrt{\pi}\,\sup_{p\in\mathbb P}|r(p)-s(p)|.
\]
\end{theorem}
\begin{proof}
对任意素数 \(p\)，
\[
1+z^p=0
\iff
z^{2p}=1,\ z^p=-1,
\]
故每个因子的零点都位于 \(S^1\)，且均为偶阶单位根；乘积 \(L(r)\) 的零点仍满足同一性质。

另一方面，经典 Fourier 展开
\[
\log|1+e^{i\phi}|
=
\sum_{m\ge 1}\frac{(-1)^{m+1}}{m}\cos(m\phi)
\qquad\text{于 }L^2[-\pi,\pi]\text{ 中成立},
\]
故
\[
A_r(\theta)
=
\sum_{q\in\mathbb P}r(q)\sum_{m\ge 1}\frac{(-1)^{m+1}}{m}\cos(mq\theta).
\]
将其与 \(\cos(p\theta)\) 配对，并利用余弦正交性，得到
\[
\frac{1}{\pi}\int_{-\pi}^{\pi}A_r(\theta)\cos(p\theta)\,\dd\theta
=
\sum_{q\in\mathbb P}r(q)\sum_{m\ge 1}\frac{(-1)^{m+1}}{m}\mathbf 1_{\{mq=p\}}.
\]
由于 \(p\) 与 \(q\) 都是素数，只有 \(q=p,m=1\) 这一项存活，故上式恰等于 \(r(p)\)。

令 \(f:=A_r-A_s\)。则对每个素数 \(p\)，有
\[
|r(p)-s(p)|
=
\left|
\frac{1}{\pi}\int_{-\pi}^{\pi}f(\theta)\cos(p\theta)\,\dd\theta
\right|
\le
\frac{1}{\pi}\|f\|_{L^2[-\pi,\pi]}\|\cos(p\theta)\|_{L^2[-\pi,\pi]}.
\]
而
\[
\|\cos(p\theta)\|_{L^2[-\pi,\pi]}=\sqrt{\pi},
\]
故
\[
|r(p)-s(p)|\le \frac{1}{\sqrt{\pi}}\|A_r-A_s\|_{L^2[-\pi,\pi]}.
\]
对全部素数取上确界即得定量下界。证毕。
\end{proof}

\begin{theorem}[Haar 零点普适与固定素数账本的完全失耦]\label{thm:conclusion-haar-zero-universality-fixed-prime-ledger-decoupling}
\leanverified{paper\_conclusion\_haar\_zero\_universality\_fixed\_prime\_ledger\_decoupling}
固定一素数 \(p_0\)，并取两整数 \(a\neq b\)。令 \((q_N)_{N\ge 1}\) 为趋于无穷的素数列，定义
\[
r_N^{(a)}:=a\,e_{p_0}+e_{q_N},
\qquad
r_N^{(b)}:=b\,e_{p_0}+e_{q_N},
\]
其中 \(e_p\) 为素数账本基元。再记 \(\mu_r\) 为 \(L(r)\) 的零点经验测度（按重数归一化到总质量 \(1\)）。则：
\begin{enumerate}
  \item 两列零点经验测度都满足
  \[
  \mu_{r_N^{(a)}}\Rightarrow m_{S^1},
  \qquad
  \mu_{r_N^{(b)}}\Rightarrow m_{S^1},
  \]
  其中 \(m_{S^1}\) 为单位圆 Haar 概率测度；
  \item 但对所有 \(N\)，都有
  \[
  \|A_{r_N^{(a)}}-A_{r_N^{(b)}}\|_{L^2[-\pi,\pi]}
  \ge
  \sqrt{\pi}\,|a-b|.
  \]
\end{enumerate}
故相同的 Haar 极限零点律与固定素数账本的刚性分离可以同时成立：渐近零点几何对固定素数寄存量完全失明，而振幅层却以不衰减的 \(L^2\) 幅度将其保存。
\end{theorem}
\begin{proof}
记
\[
D_N^{(a)}:=\sum_{p\in\mathbb P}p\,r_N^{(a)}(p)=ap_0+q_N,
\qquad
D_N^{(b)}:=bp_0+q_N.
\]
由于 \(q_N\to\infty\)，两列总次数都趋于无穷；同时对每个固定素数 \(p\)，
\[
\sup_N r_N^{(a)}(p)<\infty,
\qquad
\sup_N r_N^{(b)}(p)<\infty.
\]
故可直接调用正文中 bounded-local regime 的 Lee--Yang--Gödel 零点普适律，得到
\[
\mu_{r_N^{(a)}}\Rightarrow m_{S^1},
\qquad
\mu_{r_N^{(b)}}\Rightarrow m_{S^1}.
\]

另一方面，
\[
r_N^{(a)}(p_0)-r_N^{(b)}(p_0)=a-b
\qquad(\forall N),
\]
故由定理 \ref{thm:conclusion-unit-circle-zero-logamplitude-holographic-rigidity} 的定量下界，
\[
\|A_{r_N^{(a)}}-A_{r_N^{(b)}}\|_{L^2[-\pi,\pi]}
\ge
\sqrt{\pi}\,|a-b|
\qquad(\forall N).
\]
证毕。
\end{proof}

\begin{corollary}[仅依赖渐近零点律的相位证书无法恢复固定素数记账]\label{cor:conclusion-asymptotic-zero-law-certificate-cannot-recover-fixed-prime}
\leanverified{paper\_conclusion\_asymptotic\_zero\_law\_certificate\_cannot\_recover\_fixed\_prime}
不存在任何只把对象压缩到其渐近零点经验律
\[
\mu_{r_N}\Rightarrow \nu
\]
的证书范式，能够在 bounded-local 且 \(D_N\to\infty\) 的 Lee--Yang--Gödel 口径下恢复任意固定素数坐标 \(r(p_0)\)。更精确地，对每个固定素数 \(p_0\)，同一个极限相位对象 \(m_{S^1}\) 兼容无穷多个彼此不同的 \(p_0\)-账本值。
\end{corollary}
\begin{proof}
对任意 \(a\neq b\)，定理 \ref{thm:conclusion-haar-zero-universality-fixed-prime-ledger-decoupling} 构造了两列账本 \((r_N^{(a)})\)、\((r_N^{(b)})\)，使得
\[
\mu_{r_N^{(a)}}\Rightarrow m_{S^1},
\qquad
\mu_{r_N^{(b)}}\Rightarrow m_{S^1},
\]
但
\[
r_N^{(a)}(p_0)=a,
\qquad
r_N^{(b)}(p_0)=b
\qquad(\forall N).
\]
故任何仅以极限零点律为输入的读出都不可能区分 \(a\) 与 \(b\)。证毕。
\end{proof}

\begin{theorem}[Haar 级零点普适并不诱导振幅层的普适极限]\label{thm:conclusion-haar-zero-universality-does-not-force-amplitude-universality}
\leanverified{paper\_conclusion\_haar\_zero\_universality\_does\_not\_force\_amplitude\_universality}
在 bounded-local 且 \(D_N\to\infty\) 的 Lee--Yang--Gödel 假设类中，不存在单一函数
\[
A_\infty\in L^2[-\pi,\pi]
\]
使得对每一列账本族 \((r_N)\) 都有
\[
A_{r_N}\longrightarrow A_\infty
\qquad\text{于 }L^2[-\pi,\pi].
\]
换言之，Haar 级的零点普适并不推出振幅层的普适极限；两者之间存在严格断裂。
\end{theorem}
\begin{proof}
反设存在这样的 \(A_\infty\)。取定理 \ref{thm:conclusion-haar-zero-universality-fixed-prime-ledger-decoupling} 中的两列 \((r_N^{(a)})\)、\((r_N^{(b)})\)，则应有
\[
A_{r_N^{(a)}}\to A_\infty,
\qquad
A_{r_N^{(b)}}\to A_\infty
\qquad\text{于 }L^2[-\pi,\pi].
\]
从而
\[
\|A_{r_N^{(a)}}-A_{r_N^{(b)}}\|_{L^2[-\pi,\pi]}\to 0.
\]
但定理 \ref{thm:conclusion-haar-zero-universality-fixed-prime-ledger-decoupling} 已给出统一下界
\[
\|A_{r_N^{(a)}}-A_{r_N^{(b)}}\|_{L^2[-\pi,\pi]}
\ge
\sqrt{\pi}\,|a-b|>0
\qquad(\forall N),
\]
矛盾。证毕。
\end{proof}
