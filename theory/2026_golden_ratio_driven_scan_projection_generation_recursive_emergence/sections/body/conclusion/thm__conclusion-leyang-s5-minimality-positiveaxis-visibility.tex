\begin{theorem}[\texorpdfstring{$S_5$}{S5} 的 Lee--Yang 双通道层析极小完备性]\label{thm:conclusion-leyang-s5-two-channel-minimal-completeness}
\leanverified{paper\_conclusion\_leyang\_s5\_two\_channel\_minimal\_completeness}
设
\[
\mathbf d=
\bigl(
d_{(1^5)},d_{(2\cdot1^3)},d_{(2^2\cdot1)},d_{(3\cdot1^2)},d_{(3\cdot2)},d_{(4\cdot1)},d_{(5)}
\bigr)
\]
为 \(S_5\)-Galois 扩张的 Chebotarev 密度向量。则两条不可约通道 \(\rho_4,\rho_5\) 的极限 Lee--Yang 原子测度
\[
\mu^{(\rho_4)},\qquad \mu^{(\rho_5)}
\]
已足以唯一恢复 \(\mathbf d\)。更强地，这对通道在信息论意义上是极小完备的：删去任一通道，重建即不再唯一。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-rho45-affine-coordinate-system-on-s5-simplex}，六个原子坐标
\[
x_1:=\mu^{(\rho_5)}(\zeta_5),\quad
x_2:=\mu^{(\rho_5)}(i),\quad
x_3:=\mu^{(\rho_5)}(\zeta_6),\quad
x_4:=\mu^{(\rho_5)}(\zeta_3),\quad
x_5:=\mu^{(\rho_5)}(-1),\quad
x_6:=\mu^{(\rho_4)}(-1)
\]
在 \(\mathcal D_{S_5}\) 上构成仿射坐标。其显式反演为
\[
d_{(5)}=5x_1,\qquad
d_{(4\cdot1)}=5x_2,\qquad
d_{(3\cdot2)}=5x_3,
\]
\[
d_{(3\cdot1^2)}=\frac{5x_4-d_{(3\cdot2)}}{2},
\]
以及
\[
S:=d_{(2\cdot1^3)}+d_{(2^2\cdot1)}=\frac52x_5-d_{(4\cdot1)},
\]
\[
T:=d_{(2\cdot1^3)}+2d_{(2^2\cdot1)}=4x_6-d_{(3\cdot2)}-d_{(4\cdot1)}.
\]
故
\[
d_{(2^2\cdot1)}=T-S,\qquad
d_{(2\cdot1^3)}=2S-T,
\]
再由总和为 \(1\) 恢复 \(d_{(1^5)}\)。于是 \((\mu^{(\rho_4)},\mu^{(\rho_5)})\) 唯一决定 \(\mathbf d\)。

若只保留 \(\rho_5\)-通道，则由定理 \ref{thm:conclusion-leyang-rho5-wall-compression-codimension-one-residual}，仅能恢复
\[
S=d_{(2\cdot1^3)}+d_{(2^2\cdot1)},
\]
而不能区分 \((d_{(2\cdot1^3)},d_{(2^2\cdot1)})\) 的拆分。若只保留 \(\rho_4\)-通道，则主文关于显式反演的公式已表明，\((5),(4\cdot1),(3\cdot2),(3\cdot1^2)\) 的质量无法被单独恢复。故任一单通道都不完备，从而 \((\rho_4,\rho_5)\) 为极小完备对。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\rho_5$}{rho5} 单通道层析的亏格恰为一维]\label{cor:conclusion-leyang-rho5-single-channel-codim-one-fiber}
固定 \(\mu^{(\rho_5)}\)。则与之相容的 \(S_5\) 共轭类密度向量全体构成一条仿射线段；其唯一自由参数可取为
\[
t:=d_{(2^2\cdot1)},
\qquad
0\le t\le S,
\]
其中
\[
S=d_{(2\cdot1^3)}+d_{(2^2\cdot1)}
\]
由 \(\mu^{(\rho_5)}\) 唯一确定，而
\[
d_{(2\cdot1^3)}=S-t.
\]
因此，\(\rho_5\)-单通道层析的全部缺失信息恰好集中在
\[
(2\cdot1^3)\quad\text{与}\quad (2^2\cdot1)
\]
这两个循环型之间，且其残余仿射纤维维数恰为 \(1\)。
\leanverified{paper\_conclusion\_leyang\_rho5\_single\_channel\_codim\_one\_fiber}
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-s5-two-channel-minimal-completeness} 的证明与定理 \ref{thm:conclusion-leyang-rho5-wall-compression-codimension-one-residual}，\(\mu^{(\rho_5)}\) 已唯一决定
\[
d_{(5)},\ d_{(4\cdot1)},\ d_{(3\cdot2)},\ d_{(3\cdot1^2)},\ d_{(1^5)},\ S,
\]
而未决部分仅是
\[
(d_{(2\cdot1^3)},d_{(2^2\cdot1)}).
\]
因此相容纤维恰为
\[
\Bigl\{(d_{(2\cdot1^3)},d_{(2^2\cdot1)})=(S-t,t):\ 0\le t\le S\Bigr\},
\]
这是一条仿射线段。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$S_5$}{S5} 极限自由能在正实轴上的奇点排斥原理]\label{thm:conclusion-leyang-rho456-positive-axis-singularity-exclusion}
\leanverified{paper\_conclusion\_leyang\_rho456\_positive\_axis\_singularity\_exclusion}
对三条不可约通道 \(\rho_4,\rho_5,\rho_6\) 的极限自由能 \(F_{\rho_i}(u)\)，正实轴上唯一可能的非解析点是
\[
u=1.
\]
更精确地，
\[
F_{\rho_4},\ F_{\rho_5},\ F_{\rho_6}
\]
在每个区间 \((0,1)\) 与 \((1,\infty)\) 上都为实解析函数；并且若取
\[
D_4=240,\qquad D_5=300,\qquad D_6=180,
\]
则
\[
\exp\!\bigl(D_iF_{\rho_i}(u)\bigr)
\]
在 \(\QQ(u)\) 上都是代数单值函数。因而，凡出现在正实轴上且远离 \(u=1\) 的非解析行为，都不可能来自该 \(S_5\)-cyclotomic Lee--Yang 极限扇区。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-rho456-wall-potential-identities} 及主文给出的 cyclotomic 闭式，\(F_{\rho_4},F_{\rho_5},F_{\rho_6}\) 都写成
\[
a_i\log|u-1|+b_i\log(u+1)+
c_{i,3}\log\Phi_3(u)+
c_{i,4}\log\Phi_4(u)+
c_{i,5}\log\Phi_5(u)+
c_{i,6}\log\Phi_6(u)
\]
的线性组合。这里 \(\Phi_3,\Phi_4,\Phi_5,\Phi_6\) 的全部零点均位于单位圆上，且不落在正实轴；\(u+1\) 的零点则位于 \(-1\)。故正实轴上的唯一可能分歧点只能是 \(u=1\)。

再取
\[
D_4=240,\qquad D_5=300,\qquad D_6=180,
\]
清除各对数系数的分母，则
\[
\exp\!\bigl(D_iF_{\rho_i}(u)\bigr)
\]
成为 \((u-1)\)、\((u+1)\) 与 \(\Phi_3,\Phi_4,\Phi_5,\Phi_6\) 的有理幂乘积，因此在 \(\QQ(u)\) 上为代数单值函数。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\Phi_6$}{Phi6} 分歧精确刻画 \texorpdfstring{$\rho_4$}{rho4} 盲区与 \texorpdfstring{$\{\rho_5,\rho_6\}$}{rho5,rho6} 可见层]\label{cor:conclusion-leyang-rho456-phi6-visibility-separation}
对任意非负组合
\[
F(u)=aF_{\rho_4}(u)+bF_{\rho_5}(u)+cF_{\rho_6}(u),
\qquad a,b,c\ge 0,
\]
其在 primitive sixth roots 处出现分歧，当且仅当
\[
\frac{b}{30}+\frac{c}{36}>0.
\]
等价地，\(\Phi_6\)-stratum 对 \(\rho_4\) 完全不可见，而对 \(\rho_5,\rho_6\) 则严格可见；因此 \(\rho_4\)-sector 与 \(\{\rho_5,\rho_6\}\)-sector 在解析分歧支撑上存在一条精确可分离界面。
\end{corollary}
\begin{proof}
由主文关于三条自由能的 cyclotomic 展开，\(\rho_4\)-通道中不含 \(\log\Phi_6(u)\) 项，而
\[
F_{\rho_5}(u)\ \text{中的}\ \log\Phi_6(u)\ \text{系数为}\ \frac1{30},
\qquad
F_{\rho_6}(u)\ \text{中的}\ \log\Phi_6(u)\ \text{系数为}\ \frac1{36}.
\]
故组合 \(F\) 的 \(\Phi_6\)-系数恰为
\[
\frac{b}{30}+\frac{c}{36}.
\]
在 \(a,b,c\ge 0\) 下，该系数为零当且仅当 \(b=c=0\)。因此 primitive sixth-root 处分歧的出现与否恰好刻画了 \(\rho_4\) 的盲区以及 \(\rho_5,\rho_6\) 的可见层。证毕。
\end{proof}
