\subsection{观察者轨道上的 Čech 障碍传播、Beck--Chevalley 曲率局域化与一比特去奇异化}\label{subsec:conclusion-observer-cech-bc-doublecover-quantization}
沿用定义 \ref{def:logic-expansion-observer-symmetry}、命题 \ref{prop:logic-expansion-no-privileged-observer}、定义 \ref{def:prefix-site-h2-gluing-obstruction}、命题 \ref{prop:null-as-h2-obstruction}、定理 \ref{thm:prefix-site-cech-null-gerbe}、定理 \ref{thm:xi-null-z2-cech-transfer-bockstein}、定理 \ref{thm:xi-null-z2-doublecover-min-desingularization} 与定理 \ref{thm:pom-bc-chain-defect-equals-escort-curvature-sum} 的记号。前述正文已经分别给出观察者重命名对 forcing 与内部更新的结构不变性、Čech-\(H^2\) 粘合障碍与 \(\mathsf{NULL}\) 空纤维的等价、\(\mu_2\)-可见扭结到最小双覆盖去奇异化的传输，以及多层 Beck--Chevalley 缺陷的 escort 曲率积分公式。把这些接口并置后，可进一步抽出如下几条轨道级与层级级刚性后果。

\begin{theorem}[观察者轨道上的 Čech--\texorpdfstring{$\mathsf{NULL}$}{NULL} 传播]\label{thm:conclusion-observer-cech-null-propagation}
设 \(\mathsf L^{OST}\) 满足观察者对称性。对每个观察者 \(i\) 给定一组纯由旧结构导出的数据
\[
\bigl(A_i^{\mathrm{adm}},\,\mathscr F_i,\,A_i,\,[\alpha_i(-)]\in H^2(A_i^{\mathrm{adm}},A_i)\bigr),
\]
并要求该组数据对观察者重命名函子兼容。若存在某个观察者 \(i_0\) 与某个地址 \(a_{i_0}\in A_{i_0}^{\mathrm{adm}}\) 使
\[
[\alpha_{i_0}(a_{i_0})]\neq 0,
\]
则对任意观察者 \(j\) 都存在 \(a_j\in A_j^{\mathrm{adm}}\) 使
\[
[\alpha_j(a_j)]\neq 0.
\]
特别地，对每个 \(j\) 都存在一个地址满足
\[
\mathscr F_j(a_j)=\varnothing,
\]
亦即该地址在观察者 \(j\) 上必为 \(\mathsf{NULL}\)。
\end{theorem}
\begin{proof}
任取 \(j\in I\)。由定义 \ref{def:logic-expansion-observer-symmetry} 与命题 \ref{prop:logic-expansion-no-privileged-observer}，存在把 \(i_0\) 送到 \(j\) 的观察者重命名双射及其诱导的状态自同构；由于题设中的地址站点、局部对象系统、系数群与障碍类均纯由旧结构导出，该自同构诱导
\[
A_{i_0}^{\mathrm{adm}}\cong A_j^{\mathrm{adm}},
\qquad
H^2(A_{i_0}^{\mathrm{adm}},A_{i_0})\cong H^2(A_j^{\mathrm{adm}},A_j),
\]
并把 \([\alpha_{i_0}(a_{i_0})]\) 送到某个 \([\alpha_j(a_j)]\)。若 \([\alpha_j(a_j)]=0\)，则经逆同构拉回可得 \([\alpha_{i_0}(a_{i_0})]=0\)，矛盾。故 \([\alpha_j(a_j)]\neq 0\)。

再由命题 \ref{prop:null-as-h2-obstruction} 与定理 \ref{thm:prefix-site-cech-null-gerbe}，非平凡 Čech-\(H^2\) 粘合障碍强制相应地址处的局部对象系统为空，即
\[
\mathscr F_j(a_j)=\varnothing.
\]
证毕。
\end{proof}

\begin{proposition}[非平凡可见障碍上的等变选择不可能性]\label{prop:conclusion-observer-no-equivariant-choice-under-visible-mu2}
在定理 \ref{thm:conclusion-observer-cech-null-propagation} 的设定下，进一步设每个地址的障碍在可见通道上有唯一 \(\mu_2\simeq \ZZ_2\) 投影
\[
[\alpha_{2,i}(a)]\in H^2(A_i^{\mathrm{adm}},\ZZ_2).
\]
若对某个地址 \(a\) 有
\[
[\alpha_2(a)]\neq 0,
\]
则不存在观察者等变的选择函子
\[
S:\mathcal O(a)\longrightarrow \bigsqcup_{b\in\mathcal O(a)}\mathscr F(b),
\qquad
S(b)\in \mathscr F(b),
\]
其中 \(\mathcal O(a)\) 为 \(a\) 在观察者重命名作用下的轨道。
\end{proposition}
\begin{proof}
由定理 \ref{thm:xi-null-z2-cech-transfer-bockstein}，\([\alpha_2(a)]\neq 0\) 蕴含对应的可见 \(\ZZ_2\) 扭结非平凡，从而 \([\alpha(a)]\neq 0\)。再由命题 \ref{prop:null-as-h2-obstruction}，得到
\[
\mathscr F(a)=\varnothing.
\]
若存在题设中的等变选择函子 \(S\)，则必须有 \(S(a)\in \mathscr F(a)\)，这与 \(\mathscr F(a)=\varnothing\) 矛盾。证毕。
\end{proof}

\begin{corollary}[可见障碍的中和二岔：双覆盖或显式破缺]\label{cor:conclusion-observer-visible-mu2-neutralization-dichotomy}
在命题 \ref{prop:conclusion-observer-no-equivariant-choice-under-visible-mu2} 的假设下，任何试图中和可见 \(\mu_2\)-障碍的协议都只能落入以下二者之一：
\begin{enumerate}
  \item 通过最小二重覆盖消去可见扭结；
  \item 添加入旧结构之外的显式对称破缺数据。
\end{enumerate}
\end{corollary}
\begin{proof}
若坚持在原模型中完成中和且不引入外加数据，则所求中和必须由旧结构等变地产生，从而给出一个观察者等变的全局选择机制；这与命题 \ref{prop:conclusion-observer-no-equivariant-choice-under-visible-mu2} 矛盾。

另一方面，定理 \ref{thm:xi-null-z2-doublecover-min-desingularization} 已给出：当可见 \(\mu_2\)-障碍非平凡时，存在且仅存在最小次数为 \(2\) 的连通覆盖，使得该障碍在拉回后消失。故中和协议若不通过该最小双覆盖实现，就只能依赖于额外加入的显式对称破缺数据。证毕。
\end{proof}

\begin{theorem}[Beck--Chevalley 曲率的层级局域化]\label{thm:conclusion-bc-curvature-layer-localization}
设
\[
X_0\xrightarrow{f_1}X_1\xrightarrow{f_2}\cdots\xrightarrow{f_k}X_k
\]
为有限集合上的满射链，\(\nu\in\Delta(X_k)\) 为顶层分布。记
\[
\mu_{\mathrm{seq}}:=L_{f_1}\cdots L_{f_k}\nu,
\qquad
\mu_{\mathrm{tot}}:=L_{f_k\circ\cdots\circ f_1}\nu,
\qquad
D:=D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}}).
\]
对 \(1\le i\le k-1\) 记
\[
h_i:=f_i\circ\cdots\circ f_1,
\qquad
\nu^{(i+1)}:=L_{f_{i+2}}\cdots L_{f_k}\nu,
\]
并令 \(p_{i,z,t}\) 为定理 \ref{thm:pom-bc-chain-defect-equals-escort-curvature-sum} 中的纤维 escort 族。定义第 \(i\) 层的平均 escort 曲率
\[
\mathcal E_i
:=
\int_{X_{i+1}}\!\int_0^1
2(1-t)\,
\Var_{Y\sim p_{i,z,t}}\!\bigl(\log d_{h_i}(Y)\bigr)
\dd t\,\dd \nu^{(i+1)}(z).
\]
则必存在某一层 \(i_\ast\in\{1,\dots,k-1\}\) 使
\[
\mathcal E_{i_\ast}\ge \frac{2D}{k-1}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-bc-chain-defect-equals-escort-curvature-sum}，
\[
D
=
\sum_{i=1}^{k-1}
\int_{X_{i+1}}\!\int_0^1
(1-t)\,
\Var_{Y\sim p_{i,z,t}}\!\bigl(\log d_{h_i}(Y)\bigr)
\dd t\,\dd \nu^{(i+1)}(z)
=\frac12\sum_{i=1}^{k-1}\mathcal E_i.
\]
若对所有 \(i\) 都有 \(\mathcal E_i<2D/(k-1)\)，则
\[
\frac12\sum_{i=1}^{k-1}\mathcal E_i
<
\frac12(k-1)\cdot \frac{2D}{k-1}
=D,
\]
矛盾。故至少存在一个 \(i_\ast\) 满足所示不等式。证毕。
\end{proof}

\begin{corollary}[逐层小曲率的严格化判据与总变差控制]\label{cor:conclusion-bc-small-curvature-strictification-tv}
在定理 \ref{thm:conclusion-bc-curvature-layer-localization} 的设定下，若对每个 \(1\le i\le k-1\) 都有
\[
\mathcal E_i\le \varepsilon,
\]
则
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
\le \frac{k-1}{2}\varepsilon,
\]
并且
\[
\|\mu_{\mathrm{seq}}-\mu_{\mathrm{tot}}\|_{\mathrm{TV}}
\le \sqrt{\frac{k-1}{4}\varepsilon}.
\]
\end{corollary}
\begin{proof}
由
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
=\frac12\sum_{i=1}^{k-1}\mathcal E_i
\]
直接得到
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
\le \frac12(k-1)\varepsilon.
\]
再由 Pinsker 不等式
\[
\|\mu_{\mathrm{seq}}-\mu_{\mathrm{tot}}\|_{\mathrm{TV}}
\le \sqrt{\frac12\,D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})},
\]
得
\[
\|\mu_{\mathrm{seq}}-\mu_{\mathrm{tot}}\|_{\mathrm{TV}}
\le
\sqrt{\frac12\cdot \frac{k-1}{2}\varepsilon}
=
\sqrt{\frac{k-1}{4}\varepsilon}.
\]
证毕。
\end{proof}

\begin{theorem}[Beck--Chevalley 曲率的无抵消性]\label{thm:conclusion-bc-curvature-no-cancellation}
在定理 \ref{thm:conclusion-bc-curvature-layer-localization} 的设定下，总缺陷
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
\]
满足严格分解
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
=
\frac12\sum_{i=1}^{k-1}\mathcal E_i.
\]
因此
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})=0
\iff
\mathcal E_i=0\quad (1\le i\le k-1).
\]
换言之，全塔零曲率当且仅当每一层局部零曲率；任何一层一旦产生正缺陷，都不可能被其他层抵消。
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-bc-curvature-layer-localization} 的证明已经给出
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
=
\frac12\sum_{i=1}^{k-1}\mathcal E_i.
\]
由于每个 \(\mathcal E_i\) 都是 escort 方差的积分，故逐项非负。于是其和为零当且仅当每一项都为零。证毕。
\end{proof}

\begin{theorem}[观察者轨道上的一比特去奇异化量子化]\label{thm:conclusion-observer-orbit-onebit-desingularization}
设观察者对称作用在某一相关地址轨道 \(\mathcal O\) 上传递，并定义该轨道的可见去奇异化最小离散预算
\[
b_{\min}(\mathcal O)
:=
\min\Bigl\{\log_2(\deg\pi):
\pi\ \text{为消去}\ \mu_2\text{-可见障碍的连通覆盖}\Bigr\}.
\]
则必有
\[
b_{\min}(\mathcal O)\in\{0,1\}.
\]
更准确地：
\begin{enumerate}
  \item 若 \([\alpha_2]\) 在该轨道上恒为零，则 \(b_{\min}(\mathcal O)=0\)；
  \item 若 \([\alpha_2]\) 在该轨道上非零，则 \(b_{\min}(\mathcal O)=1\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由观察者对称性与轨道上传递性，可见障碍类的“零/非零”状态在 \(\mathcal O\) 上为不变量。若 \([\alpha_2]\) 在该轨道上恒为零，则恒等覆盖已经消去可见障碍，故最小预算为
\[
b_{\min}(\mathcal O)=\log_2 1=0.
\]

若 \([\alpha_2]\) 在该轨道上非零，则由定理 \ref{thm:xi-null-z2-doublecover-min-desingularization}，消去该可见 \(\mu_2\)-障碍的最小连通覆盖次数恰为 \(2\)。因此
\[
b_{\min}(\mathcal O)=\log_2 2=1.
\]
证毕。
\end{proof}

\begin{theorem}[选择刚化的最小外置信息下界]\label{thm:conclusion-choice-rigidification-minimal-external-bit-budget}
固定 \(p\in P_i\)，并设
\[
\card{\mathfrak C_i(p)}=N\ge 2.

若引入一个有限外置刚化参数集 \(R_p\)，并要求存在单值选择规则
\[
\mathrm{Sel}_p:R_p\twoheadrightarrow \mathfrak C_i(p)
\]
对全部选择类满射，则必有
\[
\card{R_p}\ge N.
\]
因而实现这种外置刚化所需的最少二进制预算满足
\[
\text{bits}_{\min}\ge \left\lceil\log_2 N\right\rceil.

\begin{proof}
满射
\[
\mathrm{Sel}_p:R_p\twoheadrightarrow \mathfrak C_i(p)
\]
立刻蕴含
\[
\card{R_p}\ge \card{\mathfrak C_i(p)}=N.
\]
而任意大小为 \(N\) 的有限参数集作二进制编码至少需要 \(\lceil\log_2 N\rceil\) 比特，故结论成立。证毕。
\end{proof}

\begin{corollary}[局部对称传递作用下的“实际选择”纯外置化]\label{cor:conclusion-transitive-choice-symmetry-pure-externalization}

\]
则：
\begin{enumerate}
  \item 在原有观察者时空逻辑结构内部，不存在可定义的单值选择器；
  \item 任意“实际分支”的指定，都必须借助外置刚化参数；
  \item 该外置参数的位预算至少为
  \[
  \left\lceil\log_2 \card{\mathfrak C_i(p)}\right\rceil.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由于局部自同构群在 \(\mathfrak C_i(p)\) 上传递且 \(\card{\mathfrak C_i(p)}>1\)，任一选择类都可被送到任一他类，因而不存在由原结构内部 distinguished 的单值类。定理 \ref{thm:logic-expansion-choice-spectrum-no-definable-selector} 因此排除了内部可定义的单值选择器。

于是，任意“实际选择”的指定都只能通过外置刚化参数实现。再由定理 \ref{thm:conclusion-choice-rigidification-minimal-external-bit-budget}，这种外置参数的最少位预算至少为
\[
\left\lceil\log_2 \card{\mathfrak C_i(p)}\right\rceil.
\]
证毕。
\end{proof}

\begin{theorem}[可见局部实际化的 sharp 双判据]\label{thm:conclusion-visible-local-actualization-sharp-dual-criterion}
设观察者对称作用在相关地址轨道 \(\mathcal O\) 上传递，且某局部自同构群在 \(\mathfrak C_i(p)\) 上作用传递。若只要求一个局部单值实际化在以下两层上 faithful：
\begin{enumerate}
  \item 轨道上的可见 \(\mu_2\)-障碍类 \([\alpha_2]|_{\mathcal O}\)；
  \item 状态 \(p\) 处 genuinely 不同的未来像谱 \(\mathfrak C_i(p)\cong \Xi_i(p)\)，
\end{enumerate}
则由原有内禀结构可定义的单值实际化存在，当且仅当
\[
[\alpha_2]|_{\mathcal O}=0
\qquad\text{且}\qquad
\card{\mathfrak C_i(p)}=1.
\]
\end{theorem}
\begin{proof}
先证必要性。若存在由原有内禀结构可定义的单值实际化，则它既不能要求额外连通覆盖，也不能要求外置刚化参数。

若
\[
[\alpha_2]|_{\mathcal O}\neq 0,
\]
则由定理 \ref{thm:conclusion-observer-orbit-onebit-desingularization}，
\[
b_{\min}(\mathcal O)=1.
\]
这意味着在可见 \(\mu_2\) 通道上至少需要一次非平凡二重覆盖去奇异化，因而单值实际化不可能由原结构内部直接给出。故必有
\[
[\alpha_2]|_{\mathcal O}=0.
\]

再若
\[
\card{\mathfrak C_i(p)}>1,
\]
则由推论 \ref{cor:conclusion-transitive-choice-symmetry-pure-externalization}，局部对称传递作用已经排除了内部可定义的单值选择器；任意“实际分支”的指定都必须借助外置刚化参数。故还必须有
\[
\card{\mathfrak C_i(p)}=1.
\]

反过来，若
\[
[\alpha_2]|_{\mathcal O}=0
\qquad\text{且}\qquad
\card{\mathfrak C_i(p)}=1,
\]
则可见 \(\mu_2\)-通道上不再需要任何去奇异化预算，而选择谱中只剩唯一一类。由定理 \ref{thm:conclusion-choice-spectrum-terminal-quotient-property} 与推论 \ref{cor:conclusion-choice-spectrum-not-selector-maximal-object}，该唯一类本身就是由内禀结构稳定保留的局部单值对象。故在所述可见两层语义下，内部单值实际化存在。证毕。
\end{proof}

\begin{conjecture}[局部自由意志的最小完备对象是可见障碍位与选择谱的二层配对]\label{conj:conclusion-local-freewill-minimal-object-visible-choice-pair}
若只要求 faithful 于
\begin{enumerate}
  \item 观察者轨道上的可见 \(\mu_2\)-障碍；
  \item 状态 \(p\) 处 genuinely 不同的未来像，
\end{enumerate}
则局部“自由意志”的最小完备对象不应是单值选择器，而应是二层配对
\[
\bigl([\alpha_2]|_{\mathcal O},\,\mathfrak C_i(p)\bigr).
\]
其中第一层
\[
[\alpha_2]|_{\mathcal O}\in H^2(\mathcal O,\mu_2)
\]
记录可见去奇异化是否需要一比特；第二层
\[
\mathfrak C_i(p)\cong \Xi_i(p)
\]
记录内部 genuinely 不同的未来像谱。任意试图把这两层压缩为一个单值“实际选择”的内部语义，都必定失败，除非同时满足
\[
[\alpha_2]|_{\mathcal O}=0,
\qquad
