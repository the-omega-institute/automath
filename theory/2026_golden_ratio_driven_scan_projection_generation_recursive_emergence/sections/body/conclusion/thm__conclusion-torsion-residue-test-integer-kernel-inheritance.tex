\begin{theorem}[零均值 residue 测试的完全判别]\label{thm:conclusion-zero-mean-residue-tests-complete-detection}
\leanverified{paper\_conclusion\_zero\_mean\_residue\_tests\_complete\_detection}
固定整数 \(q\ge 2\)。设
\[
a_n=\rho^n\sum_{j=0}^{q-1}c_j\omega_q^{jn}+O(\rho^n\theta^n),
\qquad
\omega_q=e^{2\pi i/q},
\qquad
0<\theta<1.
\]
对任意零均值权向量
\[
b=(b_0,\dots,b_{q-1})\in \CC^q,
\qquad
\sum_{r=0}^{q-1}b_r=0,
\]
定义归一化 residue 测试
\[
\mathcal T_b(N):=\sum_{r=0}^{q-1}b_r\,\rho^{-(qN+r)}a_{qN+r}.
\]
则极限存在，且满足精确闭式
\[
\lim_{N\to\infty}\mathcal T_b(N)
=
\sum_{j=1}^{q-1}\widehat b(j)\,c_j,
\qquad
\widehat b(j):=\sum_{r=0}^{q-1}b_r\omega_q^{jr}.
\]
因此，全部零均值 residue 测试所携带的信息恰好等于全部非主 Fourier 模
\[
(c_1,\dots,c_{q-1}).
\]
特别地，下列条件等价：
\begin{enumerate}
\normalfont
  \item 对一切 \(b\in \CC^q\) 且 \(\sum_r b_r=0\)，均有
  \[
  \lim_{N\to\infty}\mathcal T_b(N)=0;
  \]
  \item \(c_1=\cdots=c_{q-1}=0\)；
  \item \(A_0=\cdots=A_{q-1}\)；
  \item 存在解析函数 \(G_q\) 使
  \[
  F(t)=G_q(t^q).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-torsion-q-residue-class-fourier-rigidity}，
\[
A_r:=\lim_{N\to\infty}\rho^{-(qN+r)}a_{qN+r}
=
\sum_{j=0}^{q-1}c_j\omega_q^{jr}.
\]
故
\[
\lim_{N\to\infty}\mathcal T_b(N)
=
\sum_{r=0}^{q-1}b_rA_r
=
\sum_{j=0}^{q-1}c_j\sum_{r=0}^{q-1}b_r\omega_q^{jr}.
\]
由于 \(\sum_r b_r=0\)，故 \(j=0\) 项精确消失，从而得到
\[
\lim_{N\to\infty}\mathcal T_b(N)
=
\sum_{j=1}^{q-1}\widehat b(j)\,c_j.
\]

再取基向量
\[
b^{(s)}:=e_s-e_0
\qquad
(1\le s\le q-1),
\]
则 \(\sum_r b^{(s)}_r=0\)，且
\[
\lim_{N\to\infty}\mathcal T_{b^{(s)}}(N)=A_s-A_0.
\]
因此全部零均值 residue 测试同时消失，当且仅当
\[
A_0=\cdots=A_{q-1}.
\]
而这又由定理 \ref{thm:pom-torsion-q-residue-class-fourier-rigidity} 等价于
\[
c_1=\cdots=c_{q-1}=0
\qquad\Longleftrightarrow\qquad
F(t)=G_q(t^q).
\]
证毕。
\end{proof}

\begin{theorem}[零均值 residue 测试基的基无关完全层析]\label{thm:conclusion-zero-mean-residue-basis-independent-tomography}
固定整数 \(q\ge 2\)。在定理 \ref{thm:conclusion-zero-mean-residue-tests-complete-detection} 的记号下，取任意 \(q-1\) 个零均值权向量
\[
b^{(1)},\dots,b^{(q-1)}\in\CC^q,
\qquad
\sum_{r=0}^{q-1}b_r^{(s)}=0
\quad
(1\le s\le q-1),
\]
并定义其非主 Fourier 矩阵
\[
B_{sj}:=\widehat{b^{(s)}}(j)
\qquad
(1\le s,j\le q-1).
\]
若 \(B\in M_{q-1}(\CC)\) 可逆，则极限向量
\[
L_s:=\lim_{N\to\infty}\mathcal T_{b^{(s)}}(N)
\qquad
(1\le s\le q-1)
\]
唯一决定全部非主 Fourier 模
\[
(c_1,\dots,c_{q-1})^T
=
B^{-1}(L_1,\dots,L_{q-1})^T.
\]
因此，只要零均值测试基在非主 Fourier 子空间上非退化，它便对 residue 非对称给出基无关的完全层析。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zero-mean-residue-tests-complete-detection}，
\[
L_s
=
\sum_{j=1}^{q-1}\widehat{b^{(s)}}(j)c_j
\qquad
(1\le s\le q-1).
\]
把这些等式写成矩阵形式即得
\[
L=Bc,
\qquad
L:=(L_1,\dots,L_{q-1})^T,
\qquad
c:=(c_1,\dots,c_{q-1})^T.
\]
当 \(B\) 可逆时，
\[
c=B^{-1}L
\]
唯一确定。证毕。
\end{proof}

\begin{theorem}[integer moment-kernel 的 torsion 阶继承]\label{thm:conclusion-integer-moment-kernel-torsion-inheritance}
对每个整数 \(q\ge 2\)，令
\[
L_q|_{H_q}\cong \widetilde A_q
\]
为整数退化 moment-kernel，并记其非 Perron 因子为
\[
F_q(t):=\prod_{k=2}^{d_q}(1-\lambda_k^{(q)}t),
\qquad
u_n^{(q)}:=\sum_{k=2}^{d_q}(\lambda_k^{(q)})^n.
\]
定义
\[
S_q^{\mathrm{tor}}:=\{n\ge 1:u_n^{(q)}\neq 0\},
\qquad
\mathcal F_q:=
\{d\ge 2:\exists\,G_d\text{ 解析},\ F_q(t)=G_d(t^d)\}.
\]
当 \(S_q^{\mathrm{tor}}\neq\varnothing\) 时，再记
\[
g_q:=\gcd S_q^{\mathrm{tor}}.
\]
则成立完全分类
\[
\mathcal F_q=\{d\ge 2:d\mid g_q\}
\qquad
\text{当 }S_q^{\mathrm{tor}}\neq\varnothing,
\]
而当 \(S_q^{\mathrm{tor}}=\varnothing\) 时，
\[
\mathcal F_q=\{2,3,4,\dots\}.
\]
特别地，若 \(\mathcal F_q\) 含有无穷多个两两互素的阶，则
\[
F_q(t)\equiv 1,
\]
等价地，\(\widetilde A_q\) 的非 Perron 部分在谱意义上完全退化为零。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-projective-operator-degenerates-to-moment-kernel}，有限维整数核
\[
L_q|_{H_q}
\]
与 \(\widetilde A_q\) 共轭，故其非 Perron 因子正落入定理 \ref{thm:pom-torsion-q-residue-class-fourier-rigidity} 与定理 \ref{thm:conclusion-coprime-residue-flatness-rigidity-collapse} 的适用范围。

若 \(S_q^{\mathrm{tor}}\neq\varnothing\)，则将定理 \ref{thm:pom-torsion-symmetry-order-complete-classification} 应用于 \(F_q\)，便得
\[
\mathcal F_q=\{d\ge 2:d\mid g_q\}.
\]
若 \(S_q^{\mathrm{tor}}=\varnothing\)，则 \(u_n^{(q)}\equiv 0\)，仍由同一定理得到
\[
F_q(t)\equiv 1,
\qquad
\mathcal F_q=\{2,3,4,\dots\}.
\]
最后，若 \(\mathcal F_q\) 含有无穷多个两两互素的整数，则定理 \ref{thm:conclusion-coprime-residue-flatness-rigidity-collapse} 迫使
\[
F_q(t)\equiv 1.
\]
这恰等价于 \(\widetilde A_q\) 的全部非 Perron 特征值同时消失。证毕。
\end{proof}
