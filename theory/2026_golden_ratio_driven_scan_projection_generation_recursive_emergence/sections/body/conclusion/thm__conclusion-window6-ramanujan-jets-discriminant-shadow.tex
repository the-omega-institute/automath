\begin{theorem}[window-\texorpdfstring{$6$}{6} 的一阶 Ramanujan 射流精确恢复 Archimedean 对数尺度]\label{thm:conclusion-window6-output-first-ramanujan-jets-recover-log-scales}
定义一阶 Ramanujan 射流
\[
L_q:=M_q'(0)=\sum_{r\in R_6}c_q(r)\log d(r)
\qquad (q\mid 21).
\]
则有显式闭式
\[
L_3=2\log\frac32,
\qquad
L_7=\log\frac{512}{81},
\qquad
L_{21}=\log\frac{512}{9},
\]
并满足严格线性关系
\[
5L_{21}=9L_3+7L_7.
\]
更重要的是，\(\log 2,\log 3\) 可由两个素导子射流唯一恢复：
\[
\log 2=\frac{2L_3+L_7}{5},
\qquad
\log 3=\frac{9L_3+2L_7}{10}.
\]
\leanverified{paper\_conclusion\_window6\_output\_first\_ramanujan\_jets\_recover\_log\_scales}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-output-ramanujan-layer-charge-closedforms}，三层电荷向量分别为
\[
(-2,2,0),\qquad (-1,-4,5),\qquad (-5,-2,7),
\]
并且 window-\(6\) 的三层多重度恰为 \(2,3,4\)。因此
\[
L_q
=
\Bigl(\sum_{r\in A_2}c_q(r)\Bigr)\log 2
+
\Bigl(\sum_{r\in A_3}c_q(r)\Bigr)\log 3
+
\Bigl(\sum_{r\in A_4}c_q(r)\Bigr)\log 4.
\]
于是
\[
L_3=-2\log 2+2\log 3=2\log\frac32,
\]
\[
L_7=-\log 2-4\log 3+5\log 4
=9\log 2-4\log 3
=\log\frac{512}{81},
\]
\[
L_{21}=-5\log 2-2\log 3+7\log 4
=9\log 2-2\log 3
=\log\frac{512}{9}.
\]
前两式构成线性方程组
\[
L_3=-2\log 2+2\log 3,
\qquad
L_7=9\log 2-4\log 3,
\]
直接求解得
\[
\log 2=\frac{2L_3+L_7}{5},
\qquad
\log 3=\frac{9L_3+2L_7}{10}.
\]
将其代入 \(L_{21}\) 的表达式即得
\[
5L_{21}=9L_3+7L_7.
\]
证毕。
\end{proof}

\qquad

\[
g_6=\frac{273}{640}L_3+\frac{13}{80}L_7.

D_{\mathrm{KL}}(p_6\Vert u_6)

\end{corollary}
\begin{proof}
由
\[
|G_6^{\mathrm{bin}}|=(2!)^8(3!)^4(4!)^9=2^{39}3^{13},

得
\[
g_6=\frac{39}{64}\log 2+\frac{13}{64}\log 3.

再将定理 \ref{thm:conclusion-window6-output-first-ramanujan-jets-recover-log-scales} 中的
\[

\qquad
\log 3=\frac{9L_3+2L_7}{10}

代入，即得
\[
\kappa_6=\frac{23}{32}L_3+\frac{5}{16}L_7,
\qquad
g_6=\frac{273}{640}L_3+\frac{13}{80}L_7.

最后，由前文 window-\(6\) 输出几何的精确关系
\[
D_{\mathrm{KL}}(p_6\Vert u_6)=\kappa_6-\log\frac{64}{21},

代入 \(\kappa_6\) 的射流表达即可。证毕。
\end{proof}

\begin{proposition}[factorial Ramanujan 射流精确恢复全局 gauge 体积]\label{prop:conclusion-window6-output-factorial-jets-recover-gauge-volume}
定义 factorial Ramanujan 射流
\[
\Gamma_q:=\sum_{r\in R_6}c_q(r)\log(d(r)!)
\qquad (q\mid 21).
\]
则
\[
\Gamma_3=2\log 3,
\qquad
\Gamma_7=10\log 2+\log 3=\log(3072),
\qquad
\Gamma_{21}=14\log 2+5\log 3=\log(2^{14}3^5),
\]
并满足
\[
5\Gamma_{21}=9\Gamma_3+7\Gamma_7.
\]
此外，全局 gauge 体积对数由两个素导子射流严格决定：
\[
\log |G_6^{\mathrm{bin}}|
=
\frac{91}{20}\Gamma_3+\frac{39}{10}\Gamma_7.
\]
\end{proposition}
\begin{proof}
由
\[
\log(2!)=\log 2,
\qquad
\log(3!)=\log 2+\log 3,
\qquad
\log(4!)=3\log 2+\log 3,
\]
及定理 \ref{thm:conclusion-window6-output-ramanujan-layer-charge-closedforms} 的三层电荷向量，直接计算得
\[
\Gamma_3=-2\log(2!)+2\log(3!)=2\log 3,
\]
\[
\Gamma_7=-\log(2!)-4\log(3!)+5\log(4!)
=10\log 2+\log 3,
\]
\[
\Gamma_{21}=-5\log(2!)-2\log(3!)+7\log(4!)
=14\log 2+5\log 3.
\]
故
\[
5\Gamma_{21}=9\Gamma_3+7\Gamma_7.
\]
另一方面，
\[
\log |G_6^{\mathrm{bin}}|
=8\log(2!)+4\log(3!)+9\log(4!)
=39\log 2+13\log 3.
\]
由
\[
\Gamma_3=2\log 3,
\qquad
\Gamma_7=10\log 2+\log 3
\]
解出 \(\log 2,\log 3\) 并代回，即得
\[
\log |G_6^{\mathrm{bin}}|
=
\frac{91}{20}\Gamma_3+\frac{39}{10}\Gamma_7.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 判别式次数的 conductor-\texorpdfstring{$21$}{21} 二阶矩恒等式]\label{thm:conclusion-window6-output-discriminant-degree-equals-conductor21-second-moment}
记
\[
\mathfrak D_6:=\deg \Delta_6,
\qquad
A_6:=\dim_{\mathbb F_2}(G_6^{\mathrm{bin}})^{ab},
\qquad
C_6:=\frac14 M_1(2).
\]
则成立精确恒等式
\[
\mathfrak D_6=M_{21}(2)=A_6+C_6=M_1(0)+\frac14 M_1(2).
\]
在 window-\(6\) 上这具体化为
\[
\mathfrak D_6=M_{21}(2)=74=21+53.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-output-ramanujan-layer-charge-closedforms}，在 \(s=2\) 处有
\[
M_{21}(2)=-5\cdot 2^2-2\cdot 3^2+7\cdot 4^2
=-20-18+112
=74.
\]
另一方面，window-\(6\) 的基本审计量满足
\[
A_6=\dim_{\mathbb F_2}(G_6^{\mathrm{bin}})^{ab}=21,
\qquad
\frac14\sum_{r\in R_6}d(r)^2=53,
\qquad
\deg\Delta_6=74.
\]
而
\[
M_1(0)=\sum_{r\in R_6}1=|X_6|=21,
\qquad
M_1(2)=\sum_{r\in R_6}d(r)^2=212.
\]
故
\[
M_1(0)+\frac14 M_1(2)=21+\frac{212}{4}=21+53=74.
\]
于是
\[
\mathfrak D_6=M_{21}(2)=A_6+C_6=M_1(0)+\frac14 M_1(2).
\]
证毕。
\end{proof}

\begin{corollary}[判别式与 anomaly--collision 分裂的素导子层析]\label{cor:conclusion-window6-output-discriminant-anomaly-collision-prime-conductor-splitting}
在 window-\(6\) 上，二阶矩层面有
\[
M_3(2)=10,
\qquad
M_7(2)=40.
\]
从而
\[
\mathfrak D_6=\frac{9M_3(2)+7M_7(2)}{5},
\]
\[
A_6=\mathfrak D_6-\frac14 M_1(2)
=\frac{36M_3(2)+28M_7(2)-5M_1(2)}{20},
\]
\[
C_6=\frac14 M_1(2).
\]
因此，window-\(6\) 的 hidden 判别复杂度并不需要合成导子 \(21\) 作为独立输入；在二阶矩层面，平凡导子与两个素导子 \(3,7\) 已足以把
\[
\deg\Delta_6=A_6+C_6
\]
完全重建。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-window6-output-ramanujan-layer-charge-closedforms} 在 \(s=2\) 处代入，得
\[
M_3(2)=2(3^2-2^2)=10,
\]
\[
M_7(2)=5\cdot 4^2-4\cdot 3^2-2^2=80-36-4=40.
\]
再由定理 \ref{thm:conclusion-window6-output-composite-conductor-collapse}，
\[
\mathfrak D_6=M_{21}(2)=\frac{9M_3(2)+7M_7(2)}{5}.
\]
最后结合定理 \ref{thm:conclusion-window6-output-discriminant-degree-equals-conductor21-second-moment} 的
\[
A_6=\mathfrak D_6-\frac14 M_1(2),
\qquad
C_6=\frac14 M_1(2),
\]
即可得到所示分裂公式。证毕。
\end{proof}
