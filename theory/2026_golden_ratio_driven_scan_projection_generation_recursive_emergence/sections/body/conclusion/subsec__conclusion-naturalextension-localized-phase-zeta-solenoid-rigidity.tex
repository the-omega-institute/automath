\begin{theorem}[精确历史恢复的双完备外置]\label{thm:conclusion-history-recovery-radial-profinite-double-completeness}
设
\[
T:X\twoheadrightarrow X
\]
为可数集上的有限纤维满射，并假设对每个深度 \(N\ge 1\) 都存在一条长度 \(N\) 的前史链，在每一层至少发生一次真分支。若对其自然扩张
\[
X^{\leftarrow}
:=
\{(x_0,x_{-1},x_{-2},\dots):\ T(x_{-n})=x_{-n+1}\}
\]
施行纯相位审计式的精确恢复，即连续辅助读出仅允许取自连续横截寄存器范畴 \(\mathcal R_{\mathrm{cont}}\)，而离散证书仅允许由 prime-slice 账本组成，则有两条不可互换的刚性结论：
\begin{enumerate}
\item 任意有限族连续辅助寄存器
\[
r_1,\dots,r_k\in \mathcal R_{\mathrm{cont}}
\]
都必经由同一条径向轴因子化；更精确地，存在连续嵌入
\[
\Psi:\RR\hookrightarrow \prod_{j=1}^k\operatorname{Im}(r_j)
\]
使
\[
\bigl(r_1,\dots,r_k\bigr)=\Psi\circ \varrho,
\qquad
\varrho(u):=\log|u|.
\]
\item 不存在任何只使用有限个 active prime slices 的精确逆极限恢复方案；等价地，任一对整个 \(X^{\leftarrow}\) 的精确恢复都必须调用无限 prime 支撑的全不连通证书轴。
\end{enumerate}
因此，精确历史恢复中的连通外置部分严格塌缩为唯一的径向寄存器，而其余全部历史歧义都被强制外置到无限 prime 支撑的 profinite 账本中。
\end{theorem}
\begin{proof}
第 \(1\) 条正是定理 \ref{thm:conclusion-continuous-transverse-category-collapse} 的直接应用：在纯相位制度下，任意有限个连续横截寄存器的联合信息量仍不超过一条连续轴，并且该轴规范地由
\[
\varrho(u)=\log|u|
\]
参数化。

对第 \(2\) 条，定理 \ref{thm:conclusion-prime-adjunction-finite-depth-inverse-limit-budget-phase} 已经证明：在“每层至少一处分支”的最坏情形下，长度 \(N\) 前史的精确恢复所需最小有效 prime-slice 数恰为 \(N\)，并且整个逆极限 \(X^{\leftarrow}\) 不可能由有限多个 prime slices 完成精确外置。故任何精确恢复一旦超出有限深度，便必然要求无限 prime 支撑。两条结论合并即得所述双完备外置分解。证毕。
\end{proof}

\begin{corollary}[活动 prime-slice 的线性深度律]\label{cor:conclusion-history-recovery-active-prime-slice-linear-depth}
\leanverified{paper\_conclusion\_history\_recovery\_active\_prime\_slice\_linear\_depth}
在定理 \ref{thm:conclusion-history-recovery-radial-profinite-double-completeness} 的设定下，对每个 \(N\ge 1\) 记
\[
a_N:=\text{任一能够精确恢复长度 }N\text{ 前史的方案所启用的 active prime-slice 数}.
\]
则必有
\[
a_N\ge N,
\]
并且该下界是锐的：存在方案达到
\[
a_N=N.
\]
特别地，只要
\[
a_N=o(N)
\qquad\text{或更弱地}\qquad
\limsup_{N\to\infty}\frac{a_N}{N}<1,
\]
则该方案不可能在充分大深度上保持精确恢复。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-prime-adjunction-finite-depth-inverse-limit-budget-phase}，长度 \(N\) 前史的最小有效 prime-slice 数满足
\[
b(N)=N.
\]
任意精确恢复方案都必须满足 \(a_N\ge b(N)\)，故 \(a_N\ge N\)。同一定理同时给出可达性，于是 \(a_N=N\) 可实现。

若 \(a_N=o(N)\)，则对充分大的 \(N\) 有 \(a_N<N\)，与上式矛盾；若仅有
\[
\limsup_{N\to\infty}\frac{a_N}{N}<1,
\]
则存在 \(\varepsilon>0\) 与无穷多 \(N\) 使
\[
a_N\le (1-\varepsilon)N<N,
\]
同样矛盾。证毕。
\end{proof}

\begin{theorem}[有限素支撑局部化数系的六重刚性]\label{thm:conclusion-finite-localization-sixfold-rigidity}
\leanverified{paper\_conclusion\_finite\_localization\_sixfold\_rigidity}
对有限素数集 \(S,T\subset\mathbb P\)，记
\[
G_S:=\ZZ[S^{-1}],
\qquad
G_T:=\ZZ[T^{-1}],
\qquad
\Sigma_S:=\widehat{G_S},
\qquad
\Sigma_T:=\widehat{G_T}.
\]
再定义局部化自同构熵谱
\[
\mathscr E_S:=\{\,h_{\mathrm{top}}(\alpha_x):\ x\in G_S^\times\,\},
\qquad
\mathscr E_T:=\{\,h_{\mathrm{top}}(\alpha_x):\ x\in G_T^\times\,\}.
\]
则以下六个条件彼此等价：
\begin{enumerate}
\item \(S=T\)；
\item \(G_S\cong G_T\) 作为阿贝尔群；
\item 相位采样函数完全相同：
\[
\Sigma^{\mathrm{ph}}_{G_S}(N)=\Sigma^{\mathrm{ph}}_{G_T}(N)
\qquad (\forall N\ge 1);
\]
\item subgroup \(\zeta\) 完全相同：
\[
\zeta_{G_S}^{\le}(s)=\zeta_{G_T}^{\le}(s);
\]
\item 局部化 solenoid 上全部代数自同构的熵谱完全相同：
\[
\mathscr E_S=\mathscr E_T;
\]
\item Pontryagin 对偶紧群同构：
\[
\Sigma_S\cong \Sigma_T.
\]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-finite-prime-localization-pontryagin-rigidity}，条件 \(1\)、\(2\)、\(6\) 三者彼此等价。又若 \(S=T\)，则 \(G_S\)、\(\Sigma_S\)、\(\Sigma^{\mathrm{ph}}_{G_S}\)、\(\zeta_{G_S}^{\le}\) 与 \(\mathscr E_S\) 显然全部相同，故 \(1\Rightarrow 3,4,5\)。

现证逆向蕴含。若 \(3\) 成立，则由定理 \ref{thm:xi-localized-phase-sampling-closed-form}，
\[
\Sigma^{\mathrm{ph}}_{G_S}(p)=
\begin{cases}
1,& p\in S,\\
p,& p\notin S,
\end{cases}
\qquad
\Sigma^{\mathrm{ph}}_{G_T}(p)=
\begin{cases}
1,& p\in T,\\
p,& p\notin T.
\end{cases}
\]
对每个素数 \(p\) 比较这两个取值，立得
\[
p\in S\iff p\in T.
\]
故 \(S=T\)，即 \(3\Rightarrow 1\)。

若 \(4\) 成立，则由定理 \ref{thm:xi-localized-subgroup-zeta-euler-product}，
\[
\zeta_{G_S}^{\le}(s)=\zeta(s)\prod_{p\in S}(1-p^{-s}),
\qquad
\zeta_{G_T}^{\le}(s)=\zeta(s)\prod_{p\in T}(1-p^{-s}).
\]
消去公共因子 \(\zeta(s)\) 后，Euler 乘积唯一性给出
\[
\prod_{p\in S}(1-p^{-s})=\prod_{p\in T}(1-p^{-s})
\Longrightarrow
S=T.
\]
故 \(4\Rightarrow 1\)。

若 \(5\) 成立，则由推论 \ref{cor:xi-localized-solenoid-unit-entropy-max-scale}，任意
\[
x=\frac{a}{b}\in G_S^\times
\]
都满足
\[
h_{\mathrm{top}}(\alpha_x)=\log\max\{|a|,|b|\}.
\]
由于 \(G_S^\times\) 恰由 \(\pm\) 号与有限个 \(S\)-素数的整数幂生成，\(\max\{|a|,|b|\}\) 取遍且只取遍全部 \(S\)-smooth 正整数。故
\[
\mathscr E_S=\{\log n:\ n\ \text{为 }S\text{-smooth 正整数}\},
\]
同理
\[
\mathscr E_T=\{\log n:\ n\ \text{为 }T\text{-smooth 正整数}\}.
\]
若 \(\mathscr E_S=\mathscr E_T\)，则对任意素数 \(p\)，
\[
\log p\in \mathscr E_S
\iff
p\in S,
\qquad
\log p\in \mathscr E_T
\iff
p\in T.
\]
于是 \(p\in S\iff p\in T\) 对所有 \(p\) 成立，故 \(S=T\)。即 \(5\Rightarrow 1\)。

综上，六个条件彼此等价。证毕。
\end{proof}

\begin{corollary}[dyadic 审计的不完备性与完备增广]\label{cor:conclusion-dyadic-audit-incomplete-zeta-entropy-complete}
对任意有限素数集 \(S\subset\mathbb P\)，记
\[
\Sigma^{(2)}_{G_S}(a):=\Sigma^{\mathrm{ph}}_{G_S}(2^a)
\qquad (a\ge 1).
\]
则有：
\begin{enumerate}
\item 对所有有限 \(S\)，恒有
\[
\cdim_{\mathrm{fr}}(G_S)=1;
\]
\item dyadic 相位采样仅能判定 \(2\) 是否属于 \(S\)：
\[
\Sigma^{(2)}_{G_S}(a)=
\begin{cases}
1,& 2\in S,\\
2^a,& 2\notin S;
\end{cases}
\]
\item 因而任何仅依赖于
\[
\bigl(\cdim_{\mathrm{fr}}(G_S),\Sigma^{(2)}_{G_S}\bigr)
\]
的审计不变量，都不可能在有限素支撑局部化族上成为完备不变量；
\item 但若将其增广为
\[
\bigl(\cdim_{\mathrm{fr}}(G_S),\Sigma^{(2)}_{G_S},\zeta_{G_S}^{\le}\bigr)
\qquad\text{或}\qquad
\bigl(\cdim_{\mathrm{fr}}(G_S),\Sigma^{(2)}_{G_S},\mathscr E_S\bigr),
\]
则得到完备不变量。
\end{enumerate}
\end{corollary}
\begin{proof}
第一条已包含在定理 \ref{thm:xi-localized-phase-sampling-closed-form} 与定理 \ref{thm:conclusion-finite-prime-solenoid-terminal-object} 的结论中：所有有限素支撑局部化数系都只有一个连通圆维。

第二条正是推论 \ref{cor:xi-localized-dyadic-phase-blindness} 的闭式。

若有限素数集 \(S,T\subset\mathbb P\) 满足
\[
2\in S\iff 2\in T
\qquad\text{但}\qquad
S\neq T,
\]
则它们具有相同的 \(\cdim_{\mathrm{fr}}\) 与相同的 dyadic 相位采样列，故第三条成立。

第四条则由定理 \ref{thm:conclusion-finite-localization-sixfold-rigidity} 立即得到：一旦补入 subgroup \(\zeta\) 或完整熵谱 \(\mathscr E_S\)，便可唯一恢复 \(S\)。证毕。
\end{proof}

\begin{proposition}[相位采样 Dirichlet 级数与 subgroup \texorpdfstring{$\zeta$}{zeta} 的 Euler 裂解恒等式]\label{prop:conclusion-phase-sampling-dirichlet-subgroup-zeta-euler-splitting}
对有限素数集 \(S\subset\mathbb P\)，定义
\[
D_S(s):=\sum_{N\ge 1}\frac{\Sigma^{\mathrm{ph}}_{G_S}(N)}{N^s},
\qquad
\Re(s)>2.
\]
则有精确恒等式
\[
D_S(s)
=
\zeta(s-1)\prod_{p\in S}\frac{1-p^{1-s}}{1-p^{-s}}
=
\zeta_{G_S}^{\le}(s-1)\prod_{p\in S}(1-p^{-s})^{-1}.
\]
特别地，
\[
\frac{D_S(s)}{\zeta_{G_S}^{\le}(s-1)}
=
\prod_{p\in S}(1-p^{-s})^{-1},
\]
其右端恰是被 subgroup 统计遗漏的 prime-support Euler 因子。
\end{proposition}
\begin{proof}
由定理 \ref{thm:xi-localized-phase-sampling-closed-form}，
\[
\Sigma^{\mathrm{ph}}_{G_S}(N)
=
\frac{N}{\prod_{p\in S}p^{v_p(N)}}
\]
是完全乘法函数。故对任意素数 \(p\)，其局部 Euler 因子为
\[
\sum_{k\ge 0}\frac{\Sigma^{\mathrm{ph}}_{G_S}(p^k)}{p^{ks}}
=
\begin{cases}
\sum_{k\ge 0}p^{k(1-s)}=(1-p^{1-s})^{-1},& p\notin S,\\[1mm]
\sum_{k\ge 0}p^{-ks}=(1-p^{-s})^{-1},& p\in S.
\end{cases}
\]
因此
\[
D_S(s)
=
\prod_{p\notin S}(1-p^{1-s})^{-1}
\prod_{p\in S}(1-p^{-s})^{-1}
=
\zeta(s-1)\prod_{p\in S}\frac{1-p^{1-s}}{1-p^{-s}}.
\]

另一方面，由定理 \ref{thm:xi-localized-subgroup-zeta-euler-product}，
\[
\zeta_{G_S}^{\le}(s-1)
=
\zeta(s-1)\prod_{p\in S}(1-p^{1-s}).
\]
将两式相除即得
\[
\frac{D_S(s)}{\zeta_{G_S}^{\le}(s-1)}
=
\prod_{p\in S}(1-p^{-s})^{-1}.
\]
再重排即可得到第二个恒等式。证毕。
\end{proof}

\begin{theorem}[有限乘法族的 Artin--Mazur 塔完备性恰等于底数互素]\label{thm:conclusion-localized-solenoid-coprime-artin-mazur-completeness}
设 \(\mathcal B\subset \NN_{\ge 2}\) 为有限集合。对每个有限素数集 \(S\subset \mathbb P\)，记
\[
\mathcal N_{S,\mathcal B}
:=
\{\,\#\operatorname{Fix}([B]^k):\ B\in\mathcal B,\ k\ge 1\,\},
\]
并把相应的 Artin--Mazur 函数组写成
\[
\mathcal Z_{S,\mathcal B}
:=
\{\,\zeta_{S,B}(z):\ B\in\mathcal B\,\}.
\]
则下列条件彼此等价：
\begin{enumerate}
\item 对任意有限素数集 \(S,T\subset \mathbb P\)，若
\[
\#\operatorname{Fix}([B]^k\!:\Sigma_S\to\Sigma_S)
=
\#\operatorname{Fix}([B]^k\!:\Sigma_T\to\Sigma_T)
\qquad (\forall\,B\in\mathcal B,\ \forall\,k\ge 1),
\]
则 \(S=T\)；
\item 对任意有限素数集 \(S,T\subset \mathbb P\)，若
\[
\mathcal Z_{S,\mathcal B}=\mathcal Z_{T,\mathcal B},
\]
则 \(S=T\)；
\item
\[
\gcd(\mathcal B)=1.
\]
\end{enumerate}
特别地，两条互素底数塔
\[
\bigl(\zeta_{S,2},\zeta_{S,3}\bigr)
\]
已经足以唯一恢复任意有限素数支撑 \(S\)。
\end{theorem}
\begin{proof}
先证 \(1\Leftrightarrow 2\)。对每个固定的 \(B\)，推论 \ref{cor:conclusion-localized-dual-B-periodic-count-S-outside} 已将
\[
\zeta_{S,B}(z)
=
\exp\!\left(
\sum_{k\ge 1}\frac{z^k}{k}\,\#\operatorname{Fix}([B]^k)
\right)
\]
写成固定点计数列的指数生成函数，故 \(\zeta_{S,B}\) 与序列
\[
k\longmapsto \#\operatorname{Fix}([B]^k)
\]
彼此唯一决定。于是 \(1\) 与 \(2\) 等价。

再证 \(3\Rightarrow 1\)。设 \(\gcd(\mathcal B)=1\)，并假定 \(S,T\) 在全部 \(B\in\mathcal B\) 与全部 \(k\ge 1\) 上具有相同的固定点计数。任取素数 \(p\)。由 \(\gcd(\mathcal B)=1\)，至少存在某个 \(B\in\mathcal B\) 使 \(p\nmid B\)。于是定理 \ref{thm:conclusion-solenoid-profinite-blindness-fixedpoint-ledger} 第 \(2\) 条表明：单一底数 \(B\) 的整个周期点计数塔已经唯一恢复
\[
S\setminus \operatorname{Supp}(B).
\]
由于 \(p\notin \operatorname{Supp}(B)\)，便得到
\[
p\in S\iff p\in T.
\]
此对每个素数 \(p\) 都成立，故 \(S=T\)。

最后证 \(1\Rightarrow 3\)。若 \(\gcd(\mathcal B)>1\)，取素数 \(p\mid \gcd(\mathcal B)\)。则 \(p\mid B\) 对所有 \(B\in\mathcal B\) 成立，从而 \(p\nmid (B^k-1)\) 对全部 \(k\ge 1\) 成立。由推论 \ref{cor:conclusion-localized-dual-B-periodic-count-S-outside}，
\[
\#\operatorname{Fix}([B]^k)
=(B^k-1)_{S^c}
\]
从不涉及该素数 \(p\)。因此对任意有限 \(S\subset\mathbb P\)，把 \(p\) 加入或移出 \(S\) 都不会改变整个族 \(\mathcal N_{S,\mathcal B}\)，这与 \(1\) 矛盾。故必有 \(\gcd(\mathcal B)=1\)。证毕。
\end{proof}
\leanverified{paper\_conclusion\_localized\_solenoid\_coprime\_artin\_mazur\_completeness}

\begin{corollary}[两条互素 Artin--Mazur 塔与 circle--profinite 结构参数的完全等价]\label{cor:conclusion-localized-solenoid-two-coprime-zeta-structure-equivalence}
对任意有限素数集 \(S,T\subset \mathbb P\)，下列条件彼此等价：

\qquad\text{且}\qquad
\zeta_{S,3}=\zeta_{T,3};

\Sigma_S\cong \Sigma_T;

与

中，profinite 核的 \(p\)-主支撑完全相同。
\end{enumerate}
因此，\((\zeta_{S,2},\zeta_{S,3})\) 不仅是周期不变量，而且在有限局部化族中与结构参数 \(S\) 本身完全等价。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-localized-solenoid-coprime-artin-mazur-completeness}，底数集合 \(\{2,3\}\) 的最大公因子为 \(1\)，故 \(1\Rightarrow 2\)。反向蕴含显然。

\(2\Leftrightarrow 3\) 正是定理 \ref{thm:conclusion-finite-prime-localization-pontryagin-rigidity} 的内容。又由定理 \ref{thm:killo-localization-circle-dimension-prime-ledger-splitting}，

0\longrightarrow \prod_{p\in S}\ZZ_p\longrightarrow \Sigma_S\longrightarrow \TT\longrightarrow 0

中的核之 \(p\)-主因子非平凡当且仅当 \(p\in S\)，因此 \(2\Leftrightarrow 4\)。证毕。
\end{proof}

\begin{theorem}[有限窗口周期审计的无限盲素数集]\label{thm:conclusion-localized-solenoid-finitewindow-periodic-audit-blind-primes}
\leanverified{paper\_conclusion\_localized\_solenoid\_finitewindow\_periodic\_audit\_blind\_primes}
设
\[
\mathcal Q=\{(B_1,k_1),\dots,(B_N,k_N)\}\subset \NN_{\ge 2}\times \NN
\]
为任意有限查询集，并定义
\[
\mathcal P_{\mathcal Q}
:=
\bigcup_{i=1}^{N}\operatorname{Supp}(B_i^{k_i}-1).
\]
则 \(\mathcal P_{\mathcal Q}\) 为有限集合，并且对任意有限素数集 \(S\subset \mathbb P\) 与任意有限集
\[
R\subset \mathbb P\setminus \mathcal P_{\mathcal Q},
\]
都有
\[
\#\operatorname{Fix}([B_i]^{k_i}\!:\Sigma_{S\cup R}\to \Sigma_{S\cup R})
=
\#\operatorname{Fix}([B_i]^{k_i}\!:\Sigma_S\to \Sigma_S)
\qquad (1\le i\le N).
\]
特别地，任何只查询有限多个 \((B,k)\) 的周期点审计都不可能唯一恢复 \(S\)；更强地，它总会留下一个无限盲素数集
\[
\mathbb P\setminus \mathcal P_{\mathcal Q}.
\]
\end{theorem}
\begin{proof}
\(\mathcal P_{\mathcal Q}\) 是有限多个非零整数 \(B_i^{k_i}-1\) 的素因子并，故有限。若 \(p\notin \mathcal P_{\mathcal Q}\)，则

v_p(B_i^{k_i}-1)=0
\qquad (1\le i\le N).
\]
由推论 \ref{cor:conclusion-localized-dual-B-periodic-count-S-outside}，
\[
\#\operatorname{Fix}([B_i]^{k_i})
=
(B_i^{k_i}-1)_{S^c},
\]
而对这些固定的 \(i\)，把任意 \(p\notin \mathcal P_{\mathcal Q}\) 加入或移出 \(S\) 都不会改变右端的 \(p\)-进赋值。对 \(R\) 中各素数逐个施行，即得所示恒等式。

由于素数集合 \(\mathbb P\) 无穷而 \(\mathcal P_{\mathcal Q}\) 有限，补集 \(\mathbb P\setminus \mathcal P_{\mathcal Q}\) 必为无限集。故任何 finite-window 周期审计都存在无限盲素数集。证毕。
\end{proof}

\begin{theorem}[coprime Artin--Mazur 塔完备而 profinite 连续商全盲]\label{thm:conclusion-localized-solenoid-coprime-periodic-vs-profinite-split}
在有限素支撑局部化 solenoid 家族
\[
\{\Sigma_S=\widehat{\ZZ[S^{-1}]}:\ S\subset \mathbb P,\ S\ \text{有限}\}
\]
上，同时存在如下两条严格分裂的可见层：
\begin{enumerate}
\item coprime Artin--Mazur 塔，例如
\[
\bigl(\zeta_{S,2},\zeta_{S,3}\bigr),
\]
已经构成完备不变量；
\item 任意经由 profinite 群的连续观测都完全失明。更精确地，对任意 profinite 群 \(K\) 与任意连续群同态
\[
\Phi:\Sigma_S\to K,
\]
都有 \(\Phi\) 平凡，因此 \(\Sigma_S\) 不存在任何非平凡有限连续商。
\end{enumerate}
因此，在这一家族中，周期点算术与 finite continuous quotient 可见性并非强弱关系，而是信息论意义上的正交分层。
\end{theorem}
\begin{proof}
第 \(1\) 条由推论 \ref{cor:conclusion-localized-solenoid-two-coprime-zeta-structure-equivalence} 立即得到。

第 \(2\) 条正是定理 \ref{thm:conclusion-localized-dual-no-profinite-hom} 的内容。于是，任何只允许通过有限连续商或更一般的 profinite 连续像读取信息的可见理论，在每个 \(\Sigma_S\) 上都只能给出同一个平凡输出；但由第 \(1\) 条，两条互素 Artin--Mazur 塔却已足以唯一恢复 \(S\)。故这两类可见层严格正交。证毕。
\end{proof}

\begin{conjecture}[局部化 solenoid 的最小完备可见包应为 coprime Artin--Mazur 塔]\label{conj:conclusion-localized-solenoid-minimal-complete-visible-package}
在家族
\[
\{\Sigma_S=\widehat{\ZZ[S^{-1}]}:\ S\subset \mathbb P,\ S\ \text{有限}\}
\]
上，设某可见包 \(\mathcal V\) 同时满足：
\begin{enumerate}
\item 对每个 \(S\)，\(\mathcal V\) 只使用由有限个连续自同态 \([B]\) 产生的离线可审计数据；
\item \(\mathcal V\) 对全部有限素数支撑 \(S\) 完备；
\item \(\mathcal V\) 不额外引入外置 prime-ledger。
\end{enumerate}
则 \(\mathcal V\) 的最小完备实现应当等价于一组满足
\[
\gcd(\mathcal B)=1
\]
的乘法塔
\[
\{\zeta_{S,B}:\ B\in \mathcal B\}.
\]
换言之，在这一局部化家族中，最小完备可见包应是 coprime Artin--Mazur tower，而不是任何 finite quotient tower。
\end{conjecture}
