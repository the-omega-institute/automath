\begin{theorem}[碰撞盈余谱的严格去局域化]\label{thm:conclusion-fourier-excess-complete-delocalization}
记
\[
\mathcal E_m(k):=a_m(k)^2=\frac{|\widehat d_m(k)|^2}{2^{2m}}
\qquad
\bigl(k\in \ZZ/(M_m\ZZ)\setminus\{0\}\bigr),
\]
\[
W_m:=\sum_{k\ne 0}\mathcal E_m(k)=M_m\mathsf{Col}_m-1,
\qquad
\Pi_m:=\frac{W_m^2}{\sum_{k\ne 0}\mathcal E_m(k)^2}.
\]
\leanverified{paper\_conclusion\_fourier\_excess\_complete\_delocalization}
则有
\[
W_m\longrightarrow +\infty,
\qquad
\max_{k\ne 0}\frac{\mathcal E_m(k)}{W_m}\longrightarrow 0,
\qquad
\Pi_m\ge W_m\longrightarrow +\infty.
\]
因此，碰撞盈余在 Fourier 空间中不可能由有限个模式、乃至任何保持正比例能量密度的局域模式簇承担；其热力学承载体只能是一团严格扩展的频谱云。
\end{theorem}

\begin{proof}
由定理 \ref{thm:fold-collision-spectrum} 的 Parseval 恒等式与 \(a_m(0)=1\)，有
\[
\mathsf{Col}_m
=
\frac1{M_m}\sum_{k\in \ZZ/(M_m\ZZ)}a_m(k)^2
=
\frac1{M_m}\left(1+\sum_{k\ne 0}\mathcal E_m(k)\right),
\]
故
\[
W_m=M_m\mathsf{Col}_m-1.
\]
而定理 \ref{thm:conclusion-renyi2-effective-support-zero-density-collapse} 已给出
\[
M_m\mathsf{Col}_m\to+\infty,
\]
于是 \(W_m\to+\infty\)。

另一方面，对每个 \(k\) 都有
\[
0\le \mathcal E_m(k)=a_m(k)^2\le 1,
\]
因为 \(a_m(k)=|2^{-m}\widehat d_m(k)|\) 是单位模相位平均的绝对值。于是
\[
\max_{k\ne 0}\frac{\mathcal E_m(k)}{W_m}\le \frac{1}{W_m}\to 0.
\]
再由 \(0\le \mathcal E_m(k)\le 1\) 可知 \(\mathcal E_m(k)^2\le \mathcal E_m(k)\)，故
\[
\sum_{k\ne 0}\mathcal E_m(k)^2\le \sum_{k\ne 0}\mathcal E_m(k)=W_m.
\]
代回参与数定义即得
\[
\Pi_m=\frac{W_m^2}{\sum_{k\ne 0}\mathcal E_m(k)^2}\ge W_m\to+\infty.
\]
证毕。
\end{proof}

\begin{corollary}[次线性谱截断的能量不完备性]\label{cor:conclusion-sublinear-rank-spectral-truncation-incomplete}
若频率子集 \(A_m\subseteq \ZZ/(M_m\ZZ)\setminus\{0\}\) 满足
\[
|A_m|=o(W_m),
\]
则有
\[
\frac{\sum_{k\in A_m}\mathcal E_m(k)}{\sum_{k\ne 0}\mathcal E_m(k)}\longrightarrow 0.
\]
特别地，任意固定个数、任意对数级个数、乃至任意 \(o(W_m)\)-秩的 Fourier 模态截断，都不可能在能量意义下解释总碰撞盈余。
\end{corollary}

\begin{proof}
由 \(0\le \mathcal E_m(k)\le 1\)，有
\[
\sum_{k\in A_m}\mathcal E_m(k)\le |A_m|.
\]
再除以
\[
\sum_{k\ne 0}\mathcal E_m(k)=W_m
\]
即可得到
\[
0\le
\frac{\sum_{k\in A_m}\mathcal E_m(k)}{\sum_{k\ne 0}\mathcal E_m(k)}
\le
\frac{|A_m|}{W_m}\to 0.
\]
证毕。
\end{proof}

\begin{corollary}[Fibonacci 共轭双模的微观可见性与宏观失重]\label{cor:conclusion-fibonacci-twin-resonance-macro-weightlessness}
记
\[
A_m^{\mathrm{Fib}}:=\{F_m,F_{m+1}\}\subseteq \ZZ/(M_m\ZZ).
\]
则有
\[
\mathcal E_m(F_m)+\mathcal E_m(F_{m+1})\longrightarrow 2C_\varphi^2>0,
\]
但同时
\[
\frac{\mathcal E_m(F_m)+\mathcal E_m(F_{m+1})}{W_m}\longrightarrow 0.
\]
因此，Fibonacci 共轭双模在局部上保持常数级可检测共振，却在总碰撞盈余账本中失去任何正比例权重。
\end{corollary}

\begin{proof}
由定理 \ref{thm:fold-critical-resonance-constant}，
\[
a_m(F_m)\to C_\varphi,
\qquad
a_m(F_{m+1})\to C_\varphi,
\]
故
\[
\mathcal E_m(F_m)+\mathcal E_m(F_{m+1})
=
a_m(F_m)^2+a_m(F_{m+1})^2
\longrightarrow 2C_\varphi^2.
\]
另一方面，
\[
|A_m^{\mathrm{Fib}}|=2=o(W_m),
\]
故由推论 \ref{cor:conclusion-sublinear-rank-spectral-truncation-incomplete} 直接得到其相对权重趋于零。证毕。
\end{proof}

\begin{conjecture}[双尺度谱极限中的原子共振与扩散背景]\label{conj:conclusion-two-scale-spectrum-atomic-resonance-diffuse-background}
定义未归一化局部谱测度与归一化背景谱测度
\[
\mu_m:=\sum_{k\ne 0}\mathcal E_m(k)\,\delta_{k/M_m},
\qquad
\nu_m:=\frac1{W_m}\sum_{k\ne 0}\mathcal E_m(k)\,\delta_{k/M_m}.
\]
则存在两列弧邻域
\[
I_m^-\ni \frac{F_m}{M_m}\to \varphi^{-2},
\qquad
I_m^+\ni \frac{F_{m+1}}{M_m}\to \varphi^{-1},
\]
其长度趋于零，并存在圆周上的无原子概率测度 \(\nu_\infty\)，使得
\[
\mu_m(I_m^-)\to C_\varphi^2,
\qquad
\mu_m(I_m^+)\to C_\varphi^2,
\]
且
\[
\left.\nu_m\right|_{(\RR/\ZZ)\setminus (I_m^-\cup I_m^+)}
\Longrightarrow
\nu_\infty.
\]
若该猜想成立，则碰撞谱的极限对象并不是单一最大模或单一原子，而是一对未归一化共振原子嵌入一片经 \(W_m\) 归一化后才显形的扩散背景之中。
\end{conjecture}
