\begin{theorem}[素因子赋值度量的中值群结构]\label{thm:conclusion-valuation-median-group}
记素数集合为 \(\mathbb P\)。
对任意 \(x\in\QQ_{>0}\)，写其素因子赋值分解
\[
x=\prod_{p\in\mathbb P}p^{v_p(x)},\qquad v_p(x)\in\ZZ,
\]
其中仅有限多个 \(v_p(x)\neq 0\)。
定义度量
\[
\delta(x,y):=\sum_{p\in\mathbb P}\bigl|v_p(x)-v_p(y)\bigr|\log p.
\]
则 \((\QQ_{>0},\cdot,\delta)\) 为中值群：对任意三元组 \((x,y,z)\)，令
\[
m(x,y,z):=\prod_{p\in\mathbb P}p^{\operatorname{med}\bigl(v_p(x),v_p(y),v_p(z)\bigr)},
\]
其中 \(\operatorname{med}(a,b,c)\) 为三整数的中值。
则 \(m(x,y,z)\) 为满足
\[
\delta(x,y)=\delta(x,m)+\delta(m,y),\quad
\delta(y,z)=\delta(y,m)+\delta(m,z),\quad
\delta(z,x)=\delta(z,m)+\delta(m,x)
\]
的唯一点，并且中值对群左乘等变：
\[
m(ax,ay,az)=a\,m(x,y,z)\qquad(\forall a,x,y,z\in\QQ_{>0}).
\]
特别地，对 \(\NN_{\ge 1}\subset\QQ_{>0}\)，有
\[
m(1,n,m)=\gcd(n,m).
\]
\leanverified{paper\_conclusion\_valuation\_median\_group}
\leanverified{gcd\_as\_median\_instances}
\leanverified{fib\_prime\_entry\_point}
\end{theorem}

\begin{proof}
令赋值向量
\[
v:\QQ_{>0}\to \bigoplus_{p\in\mathbb P}\ZZ,\qquad v(x):=(v_p(x))_{p}.
\]
则 \(\delta(x,y)\) 为在坐标权 \(w(p)=\log p\) 下的赋权 \(\ell^1\) 距离：
\(
\delta(x,y)=\sum_p w(p)\,|v_p(x)-v_p(y)|
\)。
在单坐标 \((\ZZ,|\cdot|)\) 上，中值 \(t=\operatorname{med}(a,b,c)\) 同时位于三段闭区间 \([a,b],[b,c],[c,a]\) 上，从而对三对点 \((a,b),(b,c),(c,a)\) 同时满足三角等号的分解。
逐坐标取中值并在赋权 \(\ell^1\) 下求和，即得三条分解恒等式；逐坐标唯一性推出全体唯一性。
等变性由 \(v_p(ax)=v_p(a)+v_p(x)\) 与 \(\operatorname{med}\) 的平移不变性逐坐标推出。
最后，对 \(n,m\in\NN_{\ge 1}\) 有 \(v_p(1)=0\)，故
\(
\operatorname{med}(0,v_p(n),v_p(m))=\min\{v_p(n),v_p(m)\}
\)，从而 \(m(1,n,m)=\gcd(n,m)\)。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$(\QQ_{>0},\delta)$}{(Q>0,delta)} 的等距群刻画]\label{thm:conclusion-valuation-isometry-classification}
设 \(F:\QQ_{>0}\to\QQ_{>0}\) 为 \(\delta\)-等距同构（双射且 \(\delta(Fx,Fy)=\delta(x,y)\)）。
则存在唯一的常数 \(c\in\QQ_{>0}\) 与唯一符号函数 \(\varepsilon:\mathbb P\to\{\pm 1\}\) 使对一切 \(x\in\QQ_{>0}\) 有
\[
F(x)=c\cdot \prod_{p\in\mathbb P}p^{\varepsilon(p)v_p(x)}.
\]
因此
\[
\operatorname{Isom}(\QQ_{>0},\delta)\ \cong\ \QQ_{>0}\rtimes \{\pm 1\}^{\mathbb P},
\]
其中 \(\QQ_{>0}\) 以左乘给出平移，\(\{\pm 1\}^{\mathbb P}\) 以逐素数的反射 \(v_p\mapsto \pm v_p\) 给出坐标对称。
\leanverified{paper\_conclusion\_valuation\_isometry\_classification}
\end{theorem}

\begin{proof}
令 \(c:=F(1)\) 并取归一化 \(\widetilde F(x):=c^{-1}F(x)\)，则 \(\widetilde F\) 仍为等距且 \(\widetilde F(1)=1\)。
\par
首先，对任意素数 \(p\)，由唯一分解可得
\[
\delta(1,x)=\log p\quad\Longleftrightarrow\quad x\in\{p,p^{-1}\}.
\]
故 \(\widetilde F\) 在集合 \(\{p,p^{-1}\}\) 上置换，从而存在 \(\varepsilon(p)\in\{\pm 1\}\) 使 \(\widetilde F(p)=p^{\varepsilon(p)}\)。
\par
其次，对任意 \(x\in\QQ_{>0}\) 与素数 \(p\)，同样由唯一分解得到
\[
\delta(x,y)=\log p\quad\Longleftrightarrow\quad y=xp\ \text{或}\ y=x/p.
\]
因此对每个 \(x,p\) 存在符号 \(\sigma_p(x)\in\{\pm 1\}\) 使
\[
\widetilde F(xp)=\widetilde F(x)\,p^{\sigma_p(x)}.
\]
取两素数 \(p\neq q\)，四点 \(x,xp,xq,xpq\) 构成一张交换方块；等距性强制其像仍为交换方块，从而
\[
\widetilde F(x)\,p^{\sigma_p(x)}q^{\sigma_q(xp)}
=\widetilde F(xpq)
=\widetilde F(x)\,q^{\sigma_q(x)}p^{\sigma_p(xq)}.
\]
由素因子表示唯一性推出
\(
\sigma_p(xq)=\sigma_p(x)
\) 与
\(
\sigma_q(xp)=\sigma_q(x)
\)。
由于任意 \(x\) 可由有限次乘除素数从 \(1\) 达到，上式表明对每个 \(p\)，\(\sigma_p(x)\) 在全空间上为常值；并且由
\(
\widetilde F(p)=\widetilde F(1\cdot p)=p^{\sigma_p(1)}
\)
可知该常值即为 \(\varepsilon(p)\)。
\par
最后，对 \(x=\prod_p p^{v_p(x)}\) 反复应用上式（正负幂分别用乘除）得
\[
\widetilde F(x)=\prod_p p^{\varepsilon(p)v_p(x)}.
\]
于是 \(F(x)=c\,\widetilde F(x)\)。唯一性由 \(c=F(1)\) 与 \(\varepsilon(p)\) 在 \(\{p,p^{-1}\}\) 上的决定性给出。证毕。
\end{proof}

\begin{theorem}[\(\NN_{\ge 1}\) 上的绝对刚性与度量可定义算术]\label{thm:conclusion-metric-defines-arithmetic}
\leanverified{paper\_conclusion\_metric\_defines\_arithmetic}
将 \(\delta\) 限制到 \(\NN_{\ge 1}\)。
则：
\begin{enumerate}
  \item \(\operatorname{Isom}(\NN_{\ge 1},\delta)=\{\mathrm{id}\}\)。
  \item 在仅含度量谓词 \(\delta\) 的纯度量结构 \((\NN_{\ge 1},\delta)\) 中，单位元 \(1\)、乘法、\(\gcd\)、\(\mathrm{lcm}\)、整除关系与素数谓词均可由一阶方式定义且由结构唯一确定。
\end{enumerate}
\end{theorem}

\begin{proof}
令性质
\[
\mathbf P(u):\ \forall x\neq y,\ \delta(u,x)\neq \delta(u,y),
\]
即以 \(u\) 为中心的距离函数为单射。
由 \(\delta(1,n)=\sum_p v_p(n)\log p=\log n\)（\(n\in\NN_{\ge 1}\)）可知 \(\mathbf P(1)\) 成立。
若 \(u>1\)，取任一素因子 \(p\mid u\)，则 \(u/p\in\NN_{\ge 1}\) 且
\[
\delta(u,up)=\log p=\delta(u,u/p),
\]
故 \(\mathbf P(u)\) 不成立。
因此 \(1\) 是满足 \(\mathbf P\) 的唯一元素，从而在纯度量语言中可定义且任何等距自同构必固定 \(1\)。
\par
设 \(F\) 为等距自同构，则 \(F(1)=1\)，并且 \(\delta(1,\cdot)\) 在 \(\NN_{\ge 1}\) 上单射，故对任意 \(n\) 有
\[
\delta(1,F(n))=\delta(F(1),F(n))=\delta(1,n)\ \Longrightarrow\ F(n)=n,
\]
从而 \(\operatorname{Isom}(\NN_{\ge 1},\delta)\) 平凡。
\par
记 \(r(n):=\delta(1,n)\)。
由 \(r\) 的单射性，关系 \(r(z)=r(x)+r(y)\) 至多确定一个 \(z\)；而在本结构中 \(r(n)=\log n\)，故该 \(z\) 存在且等于 \(xy\)，从而乘法可由该唯一性刻画定义。
再由恒等式
\[
\delta(n,m)=\log\frac{\mathrm{lcm}(n,m)}{\gcd(n,m)},\qquad
r(n)+r(m)=\log\gcd(n,m)+\log\mathrm{lcm}(n,m),
\]
得到
\[
r(\gcd(n,m))=\frac{r(n)+r(m)-\delta(n,m)}{2},\qquad
r(\mathrm{lcm}(n,m))=\frac{r(n)+r(m)+\delta(n,m)}{2},
\]
并由 \(r\) 单射把 \(\gcd,\mathrm{lcm}\) 定义为满足上述等式的唯一元素。
整除可由 \(\gcd\) 定义为 \(a\mid b\iff \gcd(a,b)=a\)，素数谓词可由“\(>1\) 且正因子仅为 \(1,p\)”在一阶语言中表达，故全部算术对象由 \(\delta\) 内禀确定。证毕。
\end{proof}

\begin{theorem}[赋权自由交换幺半群的度量决定代数]\label{thm:conclusion-weighted-free-cmon-metric-rigidity}
设 \(S\) 为集合，令
\[
M:=\bigoplus_{s\in S}\NN\,e_s
\]
为自由交换幺半群（有限支撑指数向量）。
给定权函数 \(w:S\to(0,\infty)\)，定义赋权 \(\ell^1\) 度量
\[
d_w\!\left(\sum_s a_s e_s,\sum_s b_s e_s\right):=\sum_{s\in S} w(s)\,|a_s-b_s|.
\]
若映射
\[
\Psi:M\to\RR_{\ge 0},\qquad
\Psi\!\left(\sum_s a_s e_s\right):=\sum_s a_s w(s)
\]
为单射，则：
\begin{enumerate}
  \item \(\operatorname{Isom}(M,d_w)=\{\mathrm{id}\}\)；
  \item 幺半群运算可由度量定义：记 \(0\) 为幺元并令 \(r(x):=d_w(0,x)=\Psi(x)\)，则对任意 \(x,y\in M\)，元素
  \[
  x\oplus y:=\text{满足 }r(z)=r(x)+r(y)\text{ 的唯一 }z
  \]
  良定义且恰等于幺半群加法（逐坐标相加）。
\end{enumerate}
\leanverified{paper\_conclusion\_weighted\_free\_cmon\_metric\_rigidity}
\end{theorem}

\begin{proof}
由假设 \(r(x)=d_w(0,x)\) 单射。
与定理 \ref{thm:conclusion-metric-defines-arithmetic} 同型地，\(0\) 是使 \(x\mapsto d_w(u,x)\) 单射的唯一点：若 \(u\neq 0\)，取某坐标 \(s\) 使 \(u_s>0\)，则
\(
d_w(u,u+e_s)=w(s)=d_w(u,u-e_s)
\)
且 \(u-e_s\in M\)，故单射性失败。
因此任何等距自同构固定 \(0\)，再由 \(r\) 单射推出其逐点固定全部元素，得到等距群平凡。
第二条由 \(r\) 单射与幺半群加法满足 \(r(x+y)=r(x)+r(y)\) 直接推出。证毕。
\end{proof}

\begin{theorem}[中值几何的素数支撑极小刚性]\label{thm:conclusion-median-prime-support-minimal-rigidity}
设 \(G=(V,E)\) 为连通中值图，\(F:V\to\NN_{\square}\) 为满足局部素数比值律与方形一致性的素数比值嵌入。若 \(F\) 为单射，则其实际使用的不同素数个数恰等于
\[
|\Theta(G)|=\dim_{\square}(G).
\]
并且在固定基点 \(o\) 与归一化值 \(F(o)\) 后，\(F\) 除素数重标号外唯一。特别地，任何保持三元中值、图距离与全局可寻址性的平方自由素数嵌入，都不能把 \(G\) 压缩到少于 \(|\Theta(G)|\) 条可见素数轴上。
\leanverified{paper\_conclusion\_median\_prime\_support\_minimal\_rigidity}
\end{theorem}
\begin{proof}
定理 \ref{thm:cdim-median-theta-rigidity-prime-ratio} 表明，在固定基点与归一化后，任一满足局部素数比值律与方形一致性的嵌入都唯一来自某个 \(\Theta(G)\hookrightarrow\mathbb P\) 的素数标号；故其全部自由度仅剩素数重标号。另一方面，推论 \ref{cor:cdim-median-min-prime-spectrum-cube-dimension} 已证明：一旦要求可寻址性，所用素数个数至少为 \(|\Theta(G)|\)，且该阈值可达。于是最小素数支撑数恰为 \(|\Theta(G)|\)，并且在该极小口径下除素数重标号外不再有额外自由度。证毕。
\end{proof}

\begin{conjecture}[唯一分解交换幺半群的度量--分解刚性]\label{conj:conclusion-ufm-metric-factor-rigidity}
设 \(A\) 为原子、消去的交换幺半群，且具有唯一分解到不可约元集合 \(\mathcal I(A)\)（按置换意义下唯一）。
给定权函数 \(w:\mathcal I(A)\to(0,\infty)\)，定义赋权度量
\[
d_w(x,y):=\sum_{\pi\in\mathcal I(A)} w(\pi)\,|\nu_\pi(x)-\nu_\pi(y)|,
\]
其中 \(\nu_\pi\) 为分解指数。
若加权和映射 \(x\mapsto d_w(1,x)\) 在 \(A\) 上单射，则 \((A,d_w)\) 的等距自同构群应为平凡，且幺半群乘法应可在纯度量语言中定义。
\end{conjecture}
