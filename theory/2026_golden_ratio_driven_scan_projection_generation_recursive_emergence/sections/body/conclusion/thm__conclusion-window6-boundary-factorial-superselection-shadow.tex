\begin{theorem}[canonical microstate 模上的 boundary-parity 角色塌缩]\label{thm:conclusion-window6-canonical-microstate-boundary-character-collapse}
\leanverified{paper\_conclusion\_window6\_canonical\_microstate\_boundary\_character\_collapse}
记
\[
\Omega_6=\{0,1,\dots,63\},
\qquad
M_{\mathrm{can}}:=\CC^{\Omega_6},
\]
\[
b_1:=\texttt{100001},
\qquad
b_2:=\texttt{101001},
\qquad
b_3:=\texttt{100101}.
\]
由 window-\(6\) 的 boundary 二点纤维显式式
\[
F_{b_1}=\{14,48\},
\qquad
F_{b_2}=\{17,51\},
\qquad
F_{b_3}=\{19,53\}
\]
定义三个对合 \(\tau_j\in P_6\)，其中 \(\tau_j\) 交换 \(F_{b_j}\) 的两点并在其余微态上恒等。令
\[
\widehat P_6=
\{\mathbf 1,\varepsilon_1,\varepsilon_2,\varepsilon_3,
\varepsilon_1\varepsilon_2,\varepsilon_1\varepsilon_3,\varepsilon_2\varepsilon_3,
\delta:=\varepsilon_1\varepsilon_2\varepsilon_3\},
\]
其中
\[
\varepsilon_j(\tau_i)=(-1)^{\delta_{ij}}.
\]
则 \(P_6\) 在 \(M_{\mathrm{can}}\) 上的置换表示满足
\[
M_{\mathrm{can}}
\cong
\mathbf 1^{\oplus 61}\oplus \varepsilon_1\oplus \varepsilon_2\oplus \varepsilon_3.
\]
特别地，抽象上存在的八个角色扇区中，真正出现在 canonical microstate 模上的只有
\[
\mathbf 1,\qquad \varepsilon_1,\qquad \varepsilon_2,\qquad \varepsilon_3,
\]
而
\[
\varepsilon_1\varepsilon_2,\qquad
\varepsilon_1\varepsilon_3,\qquad
\varepsilon_2\varepsilon_3,\qquad
\delta
\]
全部缺席。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-window6-boundary-two-sheet-permutation-decomposition}，window-\(6\) 的 boundary 非平凡部分恰由三条互不相交的二点纤维组成。记 \(\mathbf e_n\) 为 \(M_{\mathrm{can}}\) 的标准基。则
\[
M_{\mathrm{can}}
=
U_0\oplus U_1\oplus U_2\oplus U_3,
\]
其中
\[
U_0:=\Span\{\mathbf e_n:\ n\notin F_{b_1}\cup F_{b_2}\cup F_{b_3}\},
\]
\[
U_1:=\Span\{\mathbf e_{14},\mathbf e_{48}\},
\qquad
U_2:=\Span\{\mathbf e_{17},\mathbf e_{51}\},
\qquad
U_3:=\Span\{\mathbf e_{19},\mathbf e_{53}\}.
\]
由于 \(P_6\) 在其余 \(58\) 个微态上恒等，\(U_0\cong \mathbf 1^{\oplus 58}\)。再对每个 \(j\)，\(U_j\) 上只有 \(\tau_j\) 非平凡，故
\[
U_j\cong \mathbf 1\oplus \varepsilon_j.
\]
于是
\[
M_{\mathrm{can}}
\cong
\mathbf 1^{\oplus 58}\oplus
(\mathbf 1\oplus \varepsilon_1)\oplus
(\mathbf 1\oplus \varepsilon_2)\oplus
(\mathbf 1\oplus \varepsilon_3)
\cong
\mathbf 1^{\oplus 61}\oplus \varepsilon_1\oplus \varepsilon_2\oplus \varepsilon_3.
\]
由此立刻看出其余四个角色的等价类不出现。证毕。
\end{proof}

\begin{proposition}[parity-central 秩谱的离散化与唯一奇部]\label{prop:conclusion-window6-parity-central-rank-spectrum}
\leanverified{paper\_conclusion\_window6\_parity\_central\_rank\_spectrum}
记
\[
e_{\chi}:=\frac18\sum_{g\in P_6}\chi(g)\,g\in \CC[P_6]
\qquad (\chi\in \widehat P_6).
\]
则任意 parity-central 幂等元在 \(M_{\mathrm{can}}\) 上的像维数只能取
\[
0,\ 1,\ 2,\ 3,\ 61,\ 62,\ 63,\ 64.
\]
更精确地，存在唯一的三维 parity-central 非平凡奇部
\[
B^-:=e_{\varepsilon_1}M_{\mathrm{can}}
\oplus e_{\varepsilon_2}M_{\mathrm{can}}
\oplus e_{\varepsilon_3}M_{\mathrm{can}}
=
\Span\{
\mathbf e_{14}-\mathbf e_{48},\
\mathbf e_{17}-\mathbf e_{51},\
\mathbf e_{19}-\mathbf e_{53}
\}.
\]
另一方面，不存在 parity-central 幂等元，其像恰为
\[
B^+:=
\Span\{
\mathbf e_{14}+\mathbf e_{48},\
\mathbf e_{17}+\mathbf e_{51},\
\mathbf e_{19}+\mathbf e_{53}
\}
\]
或整个 boundary 六维平面
\[
B^{\mathrm{tot}}
:=
\Span\{
\mathbf e_{14},\mathbf e_{48},
\mathbf e_{17},\mathbf e_{51},
\mathbf e_{19},\mathbf e_{53}
\}.
\]
\end{proposition}
\begin{proof}
由于 \(P_6\) 为有限交换群，其复群代数分解为
\[
\CC[P_6]\cong \bigoplus_{\chi\in\widehat P_6}\CC e_{\chi},
\]
故任意 parity-central 幂等元都可唯一写成某个子集和
\[
e_{\mathcal S}:=\sum_{\chi\in\mathcal S}e_{\chi},
\qquad
\mathcal S\subseteq \widehat P_6.
\]
由定理 \ref{thm:conclusion-window6-canonical-microstate-boundary-character-collapse}，各角色重数恰为
\[
m(\mathbf 1)=61,
\qquad
m(\varepsilon_1)=m(\varepsilon_2)=m(\varepsilon_3)=1,
\]
其余四个角色重数为零。于是
\[
\operatorname{rank}(e_{\mathcal S}|_{M_{\mathrm{can}}})
\]
只能是 \(\{61,1,1,1\}\) 的某个子集和，即所列八个数值。

再由同一表示分解，
\[
e_{\varepsilon_1}M_{\mathrm{can}}
=
\CC(\mathbf e_{14}-\mathbf e_{48}),
\quad
e_{\varepsilon_2}M_{\mathrm{can}}
=
\CC(\mathbf e_{17}-\mathbf e_{51}),
\quad
e_{\varepsilon_3}M_{\mathrm{can}}
=
\CC(\mathbf e_{19}-\mathbf e_{53}),
\]
故 \(B^-\) 正是全部非平凡出现角色的直和，从而是唯一的三维 parity-central 非平凡奇部。

另一方面，\(B^+\subset e_{\mathbf 1}M_{\mathrm{can}}\)，而
\[
e_{\mathbf 1}M_{\mathrm{can}}\cong \mathbf 1^{\oplus 61}
\]
是整个 trivial isotypic 部。任意 parity-central 幂等元在该 isotypic 部上只能取 \(0\) 或整块 \(e_{\mathbf 1}\) 的投影，不可能在其中再切出一个三维中心子模；故 \(B^+\) 不能是 parity-central 幂等像。由于
\[
B^{\mathrm{tot}}=B^+\oplus B^-,
\qquad
\dim B^{\mathrm{tot}}=6,
\]
而 \(6\) 也不在上述可实现秩谱中，故 \(B^{\mathrm{tot}}\) 同样不可能成为 parity-central 幂等像。证毕。
\end{proof}

\begin{theorem}[factorial superselection trace 的精确闭式与四个暗扇区]\label{thm:conclusion-window6-factorial-superselection-trace-closed-form}
\leanverified{paper\_conclusion\_window6\_factorial\_superselection\_trace\_closed\_form}
对任意 \(t\in \CC\) 定义对角算子
\[
D_t:=\operatorname{diag}\!\left(\frac{t^n}{n!}\right)_{0\le n\le 63},
\qquad
Z_{\chi}(t):=\operatorname{Tr}(D_t e_{\chi}).
\]
则有精确公式
\[
Z_{\varepsilon_1}(t)=\frac12\left(\frac{t^{14}}{14!}+\frac{t^{48}}{48!}\right),
\]
\[
Z_{\varepsilon_2}(t)=\frac12\left(\frac{t^{17}}{17!}+\frac{t^{51}}{51!}\right),
\]
\[
Z_{\varepsilon_3}(t)=\frac12\left(\frac{t^{19}}{19!}+\frac{t^{53}}{53!}\right),
\]
而
\[
Z_{\varepsilon_1\varepsilon_2}(t)
=
Z_{\varepsilon_1\varepsilon_3}(t)
=
Z_{\varepsilon_2\varepsilon_3}(t)
=
Z_{\delta}(t)
=0.
\]
此外
\[
Z_{\mathbf 1}(t)
=
E_{64}(t)-Z_{\varepsilon_1}(t)-Z_{\varepsilon_2}(t)-Z_{\varepsilon_3}(t).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-canonical-microstate-boundary-character-collapse}，只有 \(\mathbf 1,\varepsilon_1,\varepsilon_2,\varepsilon_3\) 四个角色在 \(M_{\mathrm{can}}\) 上出现，因此其余四个角色的 isotypic 部为零，从而 sector trace 恒为零。

对 \(F_{b_1}=\{14,48\}\)，在基
\[
(\mathbf e_{14},\mathbf e_{48})
\]
下，\(\tau_1\) 交换两基向量，而 \(\tau_2,\tau_3\) 恒等。故
\[
e_{\varepsilon_1}\!\restriction_{U_1}
=
\frac12
\begin{pmatrix}
1&-1\\
-1&1
\end{pmatrix},
\qquad
e_{\mathbf 1}\!\restriction_{U_1}
=
\frac12
\begin{pmatrix}
1&1\\
1&1
\end{pmatrix},
\]
且其余六个角色幂等元在 \(U_1\) 上皆为零。于是
\[
\operatorname{Tr}\!\left(
\begin{pmatrix}
\frac{t^{14}}{14!}&0\\
0&\frac{t^{48}}{48!}
\end{pmatrix}
e_{\varepsilon_1}\!\restriction_{U_1}
\right)
=
\frac12\left(\frac{t^{14}}{14!}+\frac{t^{48}}{48!}\right).
\]
对 \(F_{b_2}\) 与 \(F_{b_3}\) 完全同理，得到 \(Z_{\varepsilon_2}(t)\) 与 \(Z_{\varepsilon_3}(t)\) 的公式。

最后，由
\[
\sum_{\chi\in\widehat P_6}e_{\chi}=1
\]
可得
\[
\sum_{\chi\in\widehat P_6}Z_{\chi}(t)=\operatorname{Tr}(D_t)=E_{64}(t),
\]
再利用其余非平凡角色扇区全暗，便得到 \(Z_{\mathbf 1}(t)\) 的表达式。证毕。
\end{proof}

\begin{corollary}[几何余商的谱暗性]\label{cor:conclusion-window6-geometric-quotient-spectral-darkness}
记
\[
P_6^{\mathrm{geo}}=\langle \tau_1\tau_2\tau_3\rangle,
\qquad
Q_6:=P_6/P_6^{\mathrm{geo}}\cong (\ZZ/2\ZZ)^2.
\]
则 \(Q_6\) 的三个非平凡角色在 \(P_6\) 上的充气恰为
\[
\varepsilon_1\varepsilon_2,
\qquad
\varepsilon_1\varepsilon_3,
\qquad
\varepsilon_2\varepsilon_3,
\]
并对一切 \(t\in\CC\) 满足
\[
Z_{\varepsilon_1\varepsilon_2}(t)
=
Z_{\varepsilon_1\varepsilon_3}(t)
=
Z_{\varepsilon_2\varepsilon_3}(t)
=0.
\]
特别地，几何余商中的两比特自由度在 factorial/exponential 微态权下严格谱暗；更强地，对角奇角色
\[
\delta=\varepsilon_1\varepsilon_2\varepsilon_3
\]
同样全暗。
\end{corollary}
\begin{proof}
一个角色能够下降到
\[
Q_6=P_6/P_6^{\mathrm{geo}}
\]
当且仅当它在对角元 \(\tau_1\tau_2\tau_3\) 上取值 \(+1\)。在 \(\widehat P_6\) 中满足该条件的非平凡角色恰是
\[
\varepsilon_1\varepsilon_2,\qquad
\varepsilon_1\varepsilon_3,\qquad
\varepsilon_2\varepsilon_3.
\]
而这些角色的 sector trace 在定理 \ref{thm:conclusion-window6-factorial-superselection-trace-closed-form} 中已被证明恒为零。对 \(\delta\) 同理。证毕。
\end{proof}

\begin{theorem}[Cartan 边界、anti-Cartan 阴影与有理不可见性的严格三分]\label{thm:conclusion-window6-cartan-anticartan-rational-spectral-splitting}
window-\(6\) 的 boundary 结构在三个彼此不可混同的层级上发生严格分裂：
\begin{enumerate}
  \item 在 root--Cartan 完成中，boundary 三词给出零权 Cartan 部分
  \[
  \mathfrak h_{\FF_2}\cong (\ZZ/2\ZZ)^3,
  \]
  并作为 \(T_6[2]\) 的离散锚点承载三条 Cartan 轴；
  \item 在 parity-central 的 canonical 微态谱学中，真正被隔离的并不是 Cartan 边界平面
  \[
  B^+=
  \Span\{
  \mathbf e_{14}+\mathbf e_{48},\
  \mathbf e_{17}+\mathbf e_{51},\
  \mathbf e_{19}+\mathbf e_{53}
  \},
  \]
  而是其反对称阴影
  \[
  B^-=
  \Span\{
  \mathbf e_{14}-\mathbf e_{48},\
  \mathbf e_{17}-\mathbf e_{51},\
  \mathbf e_{19}-\mathbf e_{53}
  \};
  \]
  其中 \(B^-\) 是唯一三维 parity-central 非平凡奇部，而 \(B^+\) 被并入 \(61\) 维 trivial 扇区；
  \item 在连通有理连续证书层，一切正次数有理特征类都对整个 boundary parity 全盲。
\end{enumerate}
因此，boundary parity 在微态层面看见的不是 Cartan 边界本体，而是 Cartan 边界的 anti-Cartan 阴影；而在有理连续包络中，二者又同时消失。
\end{theorem}
\begin{proof}
第一条正是定理 \ref{thm:window6-abelianized-parity-charge-root-cartan-splitting} 的内容：boundary parity 在 Abel 化电荷格中对应零权 Cartan 部分，而非根方向电荷。

第二条由命题 \ref{prop:conclusion-window6-parity-central-rank-spectrum} 与定理 \ref{thm:conclusion-window6-factorial-superselection-trace-closed-form} 直接给出。它们表明：在 canonical 微态模上，parity-central Fourier 分解只实现
\[
\mathbf 1,\varepsilon_1,\varepsilon_2,\varepsilon_3
\]
四个扇区；由此唯一可分离的三维非平凡部分正是 \(B^-\)，而 \(B^+\) 由于处在 \(e_{\mathbf 1}M_{\mathrm{can}}\cong \mathbf 1^{\oplus 61}\) 内，不能被 parity-central 谱学单独切出。

第三条则由定理 \ref{thm:conclusion-window6-boundary-parity-directsummand-rational-blindness} 给出：任何连通连续包络的正次数有理特征类在 \(P_6\) 上都诱导零映射。

三条合并便得到所述严格三分。证毕。
\end{proof}

\begin{conjecture}[anchored two-sheet boundary principle]\label{conj:conclusion-window6-anchored-two-sheet-boundary-principle}
设某一审计分辨率 \(m\) 满足：
\begin{enumerate}
  \item boundary 全部纤维都是 canonical 的二层对，并生成
  \[
  C_{\mathrm{bdry},m}\cong (\ZZ/2\ZZ)^r;
  \]
  \item 可见微态模是这些二层对与其余固定态组成的自然置换模；
  \item 权函数取 factorial/exponential 型
  \[
  a_n(t)=\frac{t^n}{n!}.
  \]
\end{enumerate}
则在 canonical 微态表示中，只应出现 trivial 角色与 \(r\) 条坐标轴角色；任意由至少两条坐标轴相乘得到的高阶 Walsh 角色都应当是谱暗的，亦即其对应的 sector trace 恒等于零。window-\(6\) 的 \(r=3\) 情形给出该原则的首个刚性样本。
\end{conjecture}
