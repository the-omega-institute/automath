\begin{theorem}[可回放 Gödel 外置与信息论最小外置的对数级鸿沟]\label{thm:conclusion-godel-vs-minimal-gap}
固定分辨率 \(m\ge 1\)。令 \(A\sim\mathrm{Unif}(\Omega_m)\)，并记稳定读出 \(X:=\Fold_m(A)\in X_m\)。
定义纤维预算
\[
\kappa_m:=\EE\!\bigl[\log d_m(X)\bigr],
\qquad d_m(x):=\#\Fold_m^{-1}(x).
\]
则：
\begin{enumerate}
  \item 若 \(R\) 为任意记录轴且由 \(A\) 决定（即 \(R=f(A)\)），并且映射 \(A\mapsto(X,R)\) 为单射，则
  \[
  H(R\mid X)\ \ge\ \kappa_m.
  \]
  因而“仅要求信息可逆（存在右逆）”的最小外置长度在均匀基线下至少为 \(\kappa_m\)。
  \item \(\kappa_m\) 具有线性下界
  \[
  \kappa_m\ \ge\ \log\frac{2^m}{|X_m|}
  =m\log 2-\log F_{m+2},
  \]
  从而 \(\kappa_m=\Theta(m)\)。
  \item 对任何固定确定性策略 \(\mathsf S\)，可回放 prime-register 历史 Gödel 值满足
  \[
  \liminf_{m\to\infty}\frac{\EE\!\bigl[\log N(r_{\mathsf S}(\omega))\bigr]}{m\log m}>0,
  \]
  其中 \(\omega\sim\mathrm{Unif}(\Omega_{m+1})\)（见定理 \ref{thm:conclusion-godel-history-expected-liminf}）。因此“要求可回放（轨迹可逆）”与“只要求信息可逆”之间存在不可消除的 \(\Theta(\log m)\) 级开销差。
\end{enumerate}
\end{theorem}
\begin{proof}
对第一条，由 \(R=f(A)\) 得 \(H(A\mid X,R)=0\) 与链式法则
\[
H(A\mid X)=H(R\mid X)+H(A\mid X,R)\le H(R\mid X),
\]
而在均匀基线下 \(A\mid(X=x)\) 在纤维 \(\Fold_m^{-1}(x)\) 上均匀，故
\(
H(A\mid X)=\EE[\log d_m(X)]=\kappa_m
\)。
\par
对第二条，由 \(H(A)=m\log 2\) 且 \(H(A)=H(X)+H(A\mid X)\) 得
\[
\kappa_m=H(A\mid X)=m\log 2-H(X)\ge m\log 2-\log|X_m|.
\]
再用 \(|X_m|=F_{m+2}=\Theta(\varphi^m)\) 得 \(\kappa_m=\Theta(m)\)。
\par
第三条由定理 \ref{thm:conclusion-godel-history-expected-liminf} 直接得到；将其与 \(\kappa_m=\Theta(m)\) 进行归一化比较即可得到 \(\Theta(\log m)\) 分离。证毕。
\end{proof}

\begin{theorem}[CRT-并行 prime-register 的“局部性—轴数”下界]\label{thm:conclusion-crt-parallel-prime-register-locality-axis-lb}
设规模参数 \(B\to\infty\)。
对一组素幂模轴配置 \(B=\prod_p p^{m_p}\) 定义资源计数
\[
S_{\mathrm{mod}}:=\card{\{p:\ m_p>0\}},
\qquad
R_{\mathrm{mod}}:=\sum_p m_p.
\]
在局部 prime-register 约束下，设存在素数上限 \(P\) 使仅允许使用 \(p\le P\) 的素轴，并且每条轴的 \(p\)-进深度满足统一上界 \(m_p\le M\)，其中 \(M\) 不随 \(B\) 增长。
若要求覆盖规模 \(B\)（即 \(\prod_{p\le P}p^{m_p}\ge B\)），则必有
\[
\boxed{\;
S_{\mathrm{mod}}
\ \ge\
\left(1+o(1)\right)\frac{\log B}{M\log\log B}\;}
\qquad(B\to\infty),
\]
其中 \(o(1)\) 仅依赖于素数分布的渐近。
\end{theorem}

\begin{proof}
在约束 \(m_p\le M\) 下有
\[
\log B
\ \le\
\sum_{p\le P}m_p\log p
\ \le\
M\sum_{p\le P}\log p
=M\,\vartheta(P),
\]
其中 \(\vartheta\) 为 Chebyshev 函数。
由素数定理 \(\vartheta(P)\sim P\)，得
\(
P\ge (1+o(1))\frac{1}{M}\log B
\)。
因此
\[
S_{\mathrm{mod}}\ \ge\ \pi(P)
\ \sim\ \frac{P}{\log P}
\ \ge\
\left(1+o(1)\right)\frac{\log B}{M\log\log B},
\]
其中 \(\pi\) 为素数计数函数且 \(\log P\sim\log\log B\)。证毕。
\end{proof}

\begin{theorem}[faithful 平方自由乘法全息的对数乘性胀量]\label{thm:conclusion-squarefree-multiplicative-holography-log-gap}
\leanverified{paper\_conclusion\_squarefree\_multiplicative\_holography\_log\_gap}
固定分辨率 \(m\ge 1\) 与稳定类型 \(x\in X_m\)。
令
\[
s:=|\mathsf{Adm}(x)|,
\]
并对候选位点族任选一组两两不同的素数
\[
\{q_v:v\in \mathsf{Adm}(x)\}.
\]
定义平方自由乘法外置
\[
G_x(I):=\prod_{v\in I}q_v
\qquad
\bigl(I\in\mathsf{Ind}(\Gamma_x)\bigr),
\]
及其最坏码长
\[
L_{\max}(x):=\max_{I\in\mathsf{Ind}(\Gamma_x)}\log_2 G_x(I).
\]
则即使在允许自由选择互异素数、以最小化最坏码长的最有利意义下，仍存在绝对常数 \(c_1,c_2>0\) 使对充分大的 \(s\) 有
\[
L_{\max}(x)\ge c_1\,s\log_2 s.
\]
若进一步 \(\Gamma_x\) 含有长度 \(\Theta(s)\) 的路径分量，特别地若 \(\Gamma_x\simeq P_s\)，则还有
\[
\frac{L_{\max}(x)}{\log_2 d_m(x)}\ge c_2\log s.
\]
因此，平方自由乘法全息虽然 faithful 地恢复了 face-poset 的整除结构，但其最坏码长相对于信息论极限 \(\log_2 d_m(x)\) 必然带有不可消去的 \(\log s\) 乘性损失。
\end{theorem}
\begin{proof}
这正是正文关于平方自由素数外置最坏码长的 sharp 下界在结论层的重写。更具体地，定理 \ref{thm:xi-fold-fiber-squarefree-prime-externalization} 给出
\[
\Fold_m^{-1}(x)\cong \mathsf{Ind}(\Gamma_x)
\]
的 faithful 平方自由乘法外置，而推论 \ref{cor:xi-squarefree-externalization-log-overhead} 已证明：在最优选素意义下，最坏码长满足
\[
L_{\max}(x)\ge c_1\,s\log_2 s.
\]
同一推论还说明：若 \(\Gamma_x\) 含有长度 \(\Theta(s)\) 的路径分量，则
\[
L_{\max}(x)\ge c_2\,\log s\cdot \log_2 d_m(x),
\]
从而得到所示比值下界。证毕。
\end{proof}

\begin{corollary}[path-dominated 纤维族上平方自由乘法全息永不 near-optimal]\label{cor:conclusion-path-dominated-squarefree-holography-never-near-optimal}
设 \((x_\nu)\) 为一列稳定类型，并记
\[
s_\nu:=|\mathsf{Adm}(x_\nu)|\longrightarrow\infty.
\]
若每个 \(\Gamma_{x_\nu}\) 都含有长度 \(\Theta(s_\nu)\) 的路径分量，定义纯信息论最优长度
\[
L_{\mathrm{opt}}(x_\nu):=\left\lceil\log_2 d_m(x_\nu)\right\rceil,
\]
则任意 faithful 的平方自由乘法外置都必须满足
\[
\liminf_{\nu\to\infty}
\frac{L_{\max}(x_\nu)}{L_{\mathrm{opt}}(x_\nu)\log s_\nu}>0.
\]
特别地，
\[
\frac{L_{\max}(x_\nu)}{L_{\mathrm{opt}}(x_\nu)}\longrightarrow\infty.
\]
故在这类纤维上，faithful 乘法算术全息与最优二进制外置之间存在真正的类别分裂：前者恢复了整除/交结构，却永远不能接近无结构编码的极限长度。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-squarefree-multiplicative-holography-log-gap}，
\[
L_{\max}(x_\nu)\ge c\,\log s_\nu\cdot \log_2 d_m(x_\nu)
\]
对某个绝对常数 \(c>0\) 成立。
而
\[
L_{\mathrm{opt}}(x_\nu)=\left\lceil\log_2 d_m(x_\nu)\right\rceil
\sim \log_2 d_m(x_\nu),
\]
故
\[
\frac{L_{\max}(x_\nu)}{L_{\mathrm{opt}}(x_\nu)\log s_\nu}\ge c+o(1),
\]
从而得到所述 \(\liminf\) 下界，并推出比值发散。证毕。
\end{proof}

\begin{conjecture}[faithful 平方自由乘法外置的普适对数乘性刚性]\label{conj:conclusion-squarefree-faithful-multiplicative-holography-universal-log-gap}
设 \((x_\nu)\) 为一列稳定类型，并记
\[
s_\nu:=|\mathsf{Adm}(x_\nu)|\to\infty.
\]
考虑一切满足以下两条的外置族
\[
\mathrm{Code}_\nu:\mathsf{Ind}(\Gamma_{x_\nu})\hookrightarrow \NN_{\mathrm{sf}}:
\]
\begin{enumerate}
  \item \(\mathrm{Code}_\nu\) 为单射；
  \item 对任意 \(I,J\in\mathsf{Ind}(\Gamma_{x_\nu})\)，都有
  \[
  I\subseteq J
  \iff
  \mathrm{Code}_\nu(I)\mid \mathrm{Code}_\nu(J),
  \qquad
  \mathrm{Code}_\nu(I\cap J)=\gcd\bigl(\mathrm{Code}_\nu(I),\mathrm{Code}_\nu(J)\bigr).
  \]
\end{enumerate}
若候选位点图族 \((\Gamma_{x_\nu})\) 具有统一有界度且 path-rich，则应存在普适常数 \(c>0\) 使
\[
\max_{I\in\mathsf{Ind}(\Gamma_{x_\nu})}\log_2\mathrm{Code}_\nu(I)
\ge
c\,s_\nu\log s_\nu
\qquad (\nu\gg 1).
\]
等价地，结论 \ref{thm:conclusion-squarefree-multiplicative-holography-log-gap} 与推论 \ref{cor:conclusion-path-dominated-squarefree-holography-never-near-optimal} 中出现的 \(\log s\) 乘性胀量不应是路径模型的偶然现象，而应是 faithful 乘法算术化本身的普适刚性；路径模型只是这一刚性的极小原型。
\end{conjecture}
