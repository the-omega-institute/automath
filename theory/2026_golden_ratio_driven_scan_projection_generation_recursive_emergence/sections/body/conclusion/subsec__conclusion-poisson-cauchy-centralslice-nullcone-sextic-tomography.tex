\subsection{Poisson--Cauchy 中心剖面、四阶熵矩层析与 quartic 零锥的普适瓶颈}\label{subsec:conclusion-poisson-cauchy-centralslice-nullcone-sextic-tomography}
沿用命题 \ref{prop:cdim-poisson-central-slice-stieltjes-identifiability}、推论 \ref{cor:cdim-poisson-central-slice-kernel}、命题 \ref{prop:cdim-poisson-center-moment-spectrum}、定理 \ref{thm:cdim-poisson-entropy-moment-tomography-up-to-fourth}、推论 \ref{cor:cdim-poisson-bivariate-fdiv-fourth-order-complex-m2}、命题 \ref{prop:cdim-poisson-bivariate-second-order-cancellation-realizable}、定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion} 与定理 \ref{thm:cdim-poisson-second-order-edgeworth-benchmark} 的记号。前述链条已经分别把单点 Poisson 剖面的 Stieltjes 反演、双 KL 通道的四阶熵矩层析、一维位置混合的六阶相对熵展开，以及二维位置--尺度扰动的 quartic 可分辨常数写成显式闭式。把这些接口并置之后，还可进一步抽出如下几条后果：中心单点剖面恰好锁定中心对称化与偶矩喷射；两条 KL 剖面在四阶层面已经给出最小完备的熵矩层析；而二维 quartic 可分辨障碍则收敛到全部平滑 \(f\)-散度共享的一条普适零锥，其上真正的首个非对称指纹被推迟到 sextic 层级。

\begin{theorem}[中心单点剖面的等价关系恰为中心对称化]\label{thm:conclusion-poisson-central-slice-symmetrization-equivalence}
\leanverified{paper\_conclusion\_poisson\_central\_slice\_symmetrization\_equivalence}
设 \(\nu_1,\nu_2\) 为 \(\RR\) 上具有共同重心 \(\bar\gamma\) 的概率测度，并定义中心单点剖面
\[
\Phi_{\nu}(t):=\pi t\,h_t(\bar\gamma)
=
\int_{\RR}\frac{t^2}{(\gamma-\bar\gamma)^2+t^2}\,\dd\nu(\gamma)
\qquad (t>0).
\]
则
\[
\Phi_{\nu_1}\equiv \Phi_{\nu_2}
\qquad\Longleftrightarrow\qquad
\nu_1^{\mathrm{sym}}=\nu_2^{\mathrm{sym}},
\]
其中
\[
\nu^{\mathrm{sym}}:=\frac12(\nu+\check\nu),
\qquad
\check\nu(A):=\nu(2\bar\gamma-A).
\]
因此中心单点剖面所遗失的信息，恰好且仅仅是关于重心反射的反对称部分。
\end{theorem}
\begin{proof}
这正是推论 \ref{cor:cdim-poisson-central-slice-kernel} 的内容。若写
\[
Z:=(\gamma-\bar\gamma)^2,
\qquad
\mu_Z:=\mathrm{Law}(Z),
\]
则命题 \ref{prop:cdim-poisson-central-slice-stieltjes-identifiability} 给出
\[
m_\nu(s):=\int_{[0,\infty)}\frac{1}{z+s}\,\dd\mu_Z(z)
=\frac{\Phi_\nu(\sqrt s)}{s}
\qquad (s>0),
\]
从而 \(\Phi_\nu\) 唯一决定 \(\mu_Z\)。而 \(\mu_Z\) 恰好记录 \(|\gamma-\bar\gamma|\) 的分布，因此唯一决定 \(\nu\) 关于 \(\bar\gamma\) 的中心对称化。证毕。
\end{proof}

\begin{corollary}[中心单点剖面的有限阶喷射严格等价于偶矩锁定]\label{cor:conclusion-poisson-central-slice-even-moment-jet-rigidity}
设 \(\nu_1,\nu_2\) 具有共同重心 \(\bar\gamma\)，并且都具有有限 \(2m\) 阶中心矩。则
\[
\Phi_{\nu_1}(t)-\Phi_{\nu_2}(t)=o(t^{-2m})
\qquad (t\to\infty)
\]
当且仅当
\[
\mu_{2k}(\nu_1)=\mu_{2k}(\nu_2)
\qquad (1\le k\le m).
\]
换言之，中心单点剖面的 \(2m\) 阶无穷远喷射，不多不少，恰好编码前 \(m\) 个偶中心矩。
\end{corollary}
\begin{proof}
由命题 \ref{prop:cdim-poisson-center-moment-spectrum} 的展开式 \eqref{eq:cdim-poisson-center-moment-expansion}，
\[
\Phi_\nu(t)
=
1-\frac{\mu_2}{t^2}+\frac{\mu_4}{t^4}-\cdots+(-1)^m\frac{\mu_{2m}}{t^{2m}}+o(t^{-2m}).
\]
两测度相减得到
\[
\Phi_{\nu_1}(t)-\Phi_{\nu_2}(t)
=
-\frac{\Delta\mu_2}{t^2}
+\frac{\Delta\mu_4}{t^4}
-\cdots
+(-1)^m\frac{\Delta\mu_{2m}}{t^{2m}}
+o(t^{-2m}),
\]
其中 \(\Delta\mu_{2k}:=\mu_{2k}(\nu_1)-\mu_{2k}(\nu_2)\)。于是 \(o(t^{-2m})\) 与全部前导系数 \(\Delta\mu_{2k}\) 同时为零完全等价。证毕。
\end{proof}

\begin{theorem}[两条 KL 剖面在四阶层面给出最小完备的熵矩层析]\label{thm:conclusion-poisson-two-kl-minimal-entropy-moment-tomography}
\leanverified{paper\_conclusion\_poisson\_two\_kl\_minimal\_entropy\_moment\_tomography}
在一维位置混合模型
\[
h_t:=P_t*\nu,
\qquad
g_t(x):=P_t(x-\bar\gamma)
\]
与其二阶匹配基准
\[
g_t^{(2)}(x):=g_t(x)+\sigma^2 r_t(x),
\qquad
r_t(x):=\frac12\,\partial_x^2 g_t(x)
\]
之下，定义
\[
A_4:=\lim_{t\to\infty} t^4 D_{\mathrm{KL}}(h_t\mid g_t),
\]
\[
A_6:=\lim_{t\to\infty} t^6\!\left(D_{\mathrm{KL}}(h_t\mid g_t)-\frac{A_4}{t^4}\right),
\]
\[
B_6^{(2)}:=\lim_{t\to\infty} t^6 D_{\mathrm{KL}}(h_t\mid g_t^{(2)}).
\]
则三元组 \((A_4,A_6,B_6^{(2)})\) 与 \((\sigma^2,\mu_4,\mu_3^2)\) 互相可逆，并满足显式反演公式
\[
\sigma^2=\sqrt{8A_4},
\qquad
\mu_3^2=\frac{32}{3}B_6^{(2)},
\qquad
\mu_4=\frac{\sigma^6+6\mu_3^2-64A_6}{8\sigma^2}.
\]
因此，在四阶熵矩层析口径下，唯一无法由两条 KL 剖面分辨的三阶信息，仅剩偏度的符号。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:cdim-poisson-entropy-moment-tomography-up-to-fourth} 的结论。其核心闭式为
\[
A_4=\frac{\sigma^4}{8},
\qquad
A_6=\frac{\sigma^6-8\sigma^2\mu_4+6\mu_3^2}{64},
\qquad
B_6^{(2)}=\frac{3\mu_3^2}{32}.
\]
先由 \(A_4\) 反演 \(\sigma^2\)，再由 \(B_6^{(2)}\) 反演 \(\mu_3^2\)，最后解 \(A_6\) 的线性方程即可得 \(\mu_4\)。而中心反射
\[
\gamma-\bar\gamma\longmapsto -(\gamma-\bar\gamma)
\]
保持 \(\sigma^2,\mu_4\) 不变，只把 \(\mu_3\) 变号，故四阶层析只能恢复 \(\mu_3^2\) 而无法分辨偏度符号。证毕。
\end{proof}

\begin{theorem}[二维位置--尺度扰动的 quartic 零锥恰为协方差平衡面]\label{thm:conclusion-poisson-bivariate-quartic-null-cone-covariance-balance}
\leanverified{paper\_conclusion\_poisson\_bivariate\_quartic\_null\_cone\_covariance\_balance}
在二维 Poisson--Cauchy 粗粒化
\[
h_t(x):=\EE[P_{t+\Delta}(x-\Gamma)],
\qquad
g_t(x):=P_{t+\bar\delta}(x-\bar\gamma)
\]
中，定义
\[
A:=\Var(\Gamma)-\Var(\Delta),
\qquad
B:=\Cov(\Gamma,\Delta),
\]
以及复二阶矩
\[
m_2:=\EE\!\left[\bigl((\Gamma-\bar\gamma)+\mathrm{i}(\Delta-\bar\delta)\bigr)^2\right].
\]
则下列三条等价：
\[
D_{\mathrm{KL}}(h_t\mid g_t)=o((t+\bar\delta)^{-4}),
\]
\[
A=0\ \text{且}\ B=0,
\]
\[
m_2=0.
\]
并且该 null cone 并非退化对象：存在 \(\Var(\Gamma)>0\) 且 \(\Delta>0\) a.s. 的非平凡分布落在其上。
\end{theorem}
\begin{proof}
由推论 \ref{cor:cdim-poisson-bivariate-fdiv-fourth-order-complex-m2}（取 \(f(u)=u\log u\)）与命题 \ref{prop:cdim-poisson-bivariate-second-order-cancellation-realizable}，
\[
D_{\mathrm{KL}}(h_t\mid g_t)=o((t+\bar\delta)^{-4})
\iff
m_2=0.
\]
另一方面，
\[
m_2
=
\EE\bigl[(\Gamma-\bar\gamma)^2-(\Delta-\bar\delta)^2+2\mathrm{i}(\Gamma-\bar\gamma)(\Delta-\bar\delta)\bigr]
=
A+2\mathrm{i}B,
\]
故 \(m_2=0\iff A=0,\ B=0\)。

至于非退化实现，命题 \ref{prop:cdim-poisson-bivariate-second-order-cancellation-realizable} 已显式构造了等概率三点复向量模型，使
\[
\EE[Z]=0,\qquad \EE[Z^2]=0,\qquad \EE[Z^3]\neq 0,
\]
并在适当平移后满足 \(\Delta>0\) a.s. 且 \(\Var(\Gamma)>0\)。证毕。
\end{proof}

\begin{corollary}[quartic 零锥与散度选择无关]\label{cor:conclusion-poisson-bivariate-quartic-null-cone-fdiv-universal}
设 \(D_f\) 为任意满足 \(f''(1)\neq 0\) 的光滑 Csisz\'ar \(f\)-散度。则在定理 \ref{thm:conclusion-poisson-bivariate-quartic-null-cone-covariance-balance} 的设定下，
\[
(t+\bar\delta)^4 D_f(h_t\mid g_t)\longrightarrow \frac{f''(1)}{8}\,|m_2|^2
=
\frac{f''(1)}{8}\,(A^2+4B^2).
\]
因而对任何这类 \(f\)，都有等价判据
\[
D_f(h_t\mid g_t)=o((t+\bar\delta)^{-4})
\iff
A=0,\ B=0.
\]
故 quartic 级别的可分辨性障碍不是散度选择的产物，而是 Poisson--Cauchy 机制本身的普适零锥。
\end{corollary}
\begin{proof}
推论 \ref{cor:cdim-poisson-bivariate-fdiv-fourth-order-complex-m2} 已给出
\[
\lim_{t\to\infty}(t+\bar\delta)^4 D_f(h_t\mid g_t)=\frac{f''(1)}{8}|m_2|^2.
\]
由于 \(f''(1)\neq 0\)，其主项消失当且仅当 \(m_2=0\)。再由定理 \ref{thm:conclusion-poisson-bivariate-quartic-null-cone-covariance-balance}，这又等价于 \(A=0,\ B=0\)。证毕。
\end{proof}

\begin{theorem}[一维 Poisson--KL 的六阶系数严格为负，且归一化熵缺口从下逼近极限]\label{thm:conclusion-poisson-kl-sixth-coefficient-negative-monotone-approach}
在一维位置混合模型中，定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion} 给出
\[
D_{\mathrm{KL}}(h_t\mid g_t)
=
\frac{\sigma^4}{8t^4}
+
\frac{1}{t^6}
\left(
\frac{\sigma^6}{64}
-\frac{\sigma^2\mu_4}{8}
+\frac{3\mu_3^2}{32}
\right)
+o(t^{-6}).
\]
记
\[
B_6:=\frac{\sigma^6}{64}-\frac{\sigma^2\mu_4}{8}+\frac{3\mu_3^2}{32}.
\]
则恒有
\[
B_6<0.
\]
从而
\[
t^4D_{\mathrm{KL}}(h_t\mid g_t)=\frac{\sigma^4}{8}+\frac{B_6}{t^2}+o(t^{-2}),
\]
故该归一化量在充分大 \(t\) 上严格小于 \(\sigma^4/8\)，并最终严格单调递增地从下方收敛到 \(\sigma^4/8\)。
\end{theorem}
\begin{proof}
由中心矩不等式
\[
\mu_4\ge \sigma^4,
\qquad
\mu_3^2\le \sigma^2\mu_4
\]
可得
\[
B_6
\le
\frac{\sigma^6}{64}
-\frac{\sigma^2\mu_4}{8}
+\frac{3\sigma^2\mu_4}{32}
=
\frac{\sigma^6}{64}-\frac{\sigma^2\mu_4}{32}
\le
\frac{\sigma^6}{64}-\frac{\sigma^6}{32}
=
-\frac{\sigma^6}{64}<0.
\]
将其代回定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion}，便得
\[
t^4D_{\mathrm{KL}}(h_t\mid g_t)=\frac{\sigma^4}{8}+\frac{B_6}{t^2}+o(t^{-2}),
\]
故对充分大 \(t\) 有 \(t^4D_{\mathrm{KL}}(h_t\mid g_t)<\sigma^4/8\)。再逐项求导得
\[
\frac{\dd}{\dd t}\Bigl(t^4D_{\mathrm{KL}}(h_t\mid g_t)\Bigr)
=
-\frac{2B_6}{t^3}+o(t^{-3})>0
\]
在充分大 \(t\) 上成立，因此该归一化量最终严格单调递增并从下收敛到 \(\sigma^4/8\)。证毕。
\end{proof}

\begin{corollary}[反射不对称首次进入熵层析，必在 sextic 而非 quartic]\label{cor:conclusion-poisson-skewness-first-enters-at-sextic}
沿用定理 \ref{thm:conclusion-poisson-two-kl-minimal-entropy-moment-tomography} 的记号，则
\[
A_4=\frac{\sigma^4}{8}
\]
只依赖于方差，与任何反射不对称无关；而二阶匹配通道的六阶系数
\[
B_6^{(2)}=\frac{3\mu_3^2}{32}
\]
恰是第一项反映中心反射不对称的熵指纹。于是：
\begin{enumerate}
  \item quartic 级别的 KL 层析对偏态完全失明；
  \item sextic 级别才首次显影，而且显影方式仅为 \(\mu_3^2\)；
  \item 若 \(\mu_3=0\)，则二阶匹配通道满足
  \[
  D_{\mathrm{KL}}(h_t\mid g_t^{(2)})=O(t^{-8}).
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-poisson-two-kl-minimal-entropy-moment-tomography}，
\[
A_4=\frac{\sigma^4}{8},
\qquad
\mu_3^2=\frac{32}{3}B_6^{(2)}.
\]
故 quartic 主项仅由方差决定，而 sextic 的 matched KL 通道才首次携带偏态信息，并且仅通过 \(\mu_3^2\) 发生依赖。
最后，若 \(\mu_3=0\)，则定理 \ref{thm:cdim-poisson-second-order-edgeworth-benchmark} 直接给出
\[
D_{\mathrm{KL}}(h_t\mid g_t^{(2)})=O(t^{-8}).
\]
证毕。
\end{proof}

\begin{theorem}[Poisson 粗粒化下平滑 \texorpdfstring{$f$}{f}-散度的普适比值定理]\label{thm:conclusion-poisson-fdivergence-universal-ratio}
设 \(h_t,g_t\) 为主稿定义的混合位置--尺度 Poisson 粗粒化密度，并令复二阶中心矩
\[
m_2:=\EE\!\left[\bigl((\Gamma-\bar\gamma)+\mathrm{i}(\Delta-\bar\delta)\bigr)^2\right],
\qquad
s_t:=t+\bar\delta.
\]
对任意在 \(1\) 邻域 \(C^2\) 且满足 \(f(1)=0\) 的凸函数 \(f\)，主稿给出
\[
D_f(h_t\mid g_t)=\frac{f''(1)}{8}\frac{|m_2|^2}{s_t^4}+o(s_t^{-4}).
\]
在非退化情形 \(|m_2|\neq 0\) 下，对任意两类此类函数 \(f,g\) 且 \(g''(1)\neq 0\)，有
\[
\lim_{t\to\infty}\frac{D_f(h_t\mid g_t)}{D_g(h_t\mid g_t)}
=
\frac{f''(1)}{g''(1)}.
\]
因此，在大尺度 Poisson 粗粒化极限中，所有足够光滑的 \(f\)-散度只保留一个与 \(f\) 无关的几何模量 \(|m_2|^2/s_t^4\)；函数 \(f\) 只通过纯局部系数 \(f''(1)\) 进入。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-poisson-bivariate-quartic-null-cone-fdiv-universal}，
\[
D_f(h_t\mid g_t)=\frac{f''(1)}{8}\frac{|m_2|^2}{s_t^4}+o(s_t^{-4}),
\qquad
D_g(h_t\mid g_t)=\frac{g''(1)}{8}\frac{|m_2|^2}{s_t^4}+o(s_t^{-4}).
\]
若 \(|m_2|\neq 0\)，直接相除即得
\[
\frac{D_f(h_t\mid g_t)}{D_g(h_t\mid g_t)}
=
\frac{f''(1)+o(1)}{g''(1)+o(1)}
\longrightarrow
\frac{f''(1)}{g''(1)}.
\]
若 \(|m_2|=0\)，则两者 quartic 主项同时消失，此时命题退化为同一零锥上的空比较；因此非平凡情形下极限比值完全由二阶局部系数决定。证毕。
\end{proof}

\begin{theorem}[复二阶中心矩的全 \texorpdfstring{$f$}{f}-几何完备性]\label{thm:conclusion-poisson-complex-second-moment-complete-for-all-f}
在结论 \ref{thm:conclusion-poisson-fdivergence-universal-ratio} 的同一设定下，固定总质量、重心与平均尺度，即固定
\[
M_0,\ \bar\gamma,\ \bar\delta,
\]
并考虑所有满足主稿正则性假设的粗粒化家族。则对任意两组数据 \(\mu,\mu'\)，下列条件等价：
\begin{enumerate}
  \item 对某个满足 \(f''(1)\neq 0\) 的光滑 \(f\)，有
  \[
  D_f(h_t\mid g_t)-D_f(h_t'\mid g_t')=o(s_t^{-4});
  \]
  \item 对每个此类 \(f\)，都有
  \[
  D_f(h_t\mid g_t)-D_f(h_t'\mid g_t')=o(s_t^{-4});
  \]
  \item 二者满足
  \[
  |m_2(\mu)|=|m_2(\mu')|.
  \]
\end{enumerate}
因此，在固定一阶几何数据之后，\(|m_2|\) 是全部平滑 \(f\)-散度 quartic 渐近的完备统计量。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-poisson-bivariate-quartic-null-cone-fdiv-universal}，对任意满足 \(f''(1)\neq 0\) 的光滑 \(f\)，都有
\[
D_f(h_t\mid g_t)
=
\frac{f''(1)}{8}\frac{|m_2(\mu)|^2}{s_t^4}+o(s_t^{-4}),
\qquad
D_f(h_t'\mid g_t')
=
\frac{f''(1)}{8}\frac{|m_2(\mu')|^2}{s_t^4}+o(s_t^{-4}).
\]
若对某个 \(f\) 有
\[
D_f(h_t\mid g_t)-D_f(h_t'\mid g_t')=o(s_t^{-4}),
\]
则 quartic 主项系数必须相等，即
\[
|m_2(\mu)|^2=|m_2(\mu')|^2,
\]
从而
\[
|m_2(\mu)|=|m_2(\mu')|.
\]
反之，一旦 \(|m_2|\) 相等，则所有此类 \(f\) 的 quartic 主项都同步相等，于是差额统一降为 \(o(s_t^{-4})\)。故三条条件彼此等价。证毕。
\end{proof}

\begin{conjecture}[共动缺陷测度的自然规一化应落在 quartic 零锥上]\label{conj:conclusion-zeta-comoving-defect-natural-normalization-quartic-null-cone}
在 \(\zeta\)-动力学共动缺陷框架中，设有限缺陷块经自然规一化后产生二维随机向量 \((\Gamma,\Delta)\)。则应有块状平衡律
\[
\Var(\Gamma)=\Var(\Delta),
\qquad
\Cov(\Gamma,\Delta)=0,
\]
等价地，
\[
m_2=\EE\!\left[\bigl((\Gamma-\bar\gamma)+\mathrm{i}(\Delta-\bar\delta)\bigr)^2\right]=0.
\]
若该猜想成立，则所有满足 \(f''(1)\neq 0\) 的光滑 \(f\)-散度在 quartic 层的主项都会统一湮灭，粗粒化可分辨度将制度性地从 \(t^{-4}\) 提升到 \(t^{-6}\) 级别；于是首个真正的信息瓶颈不再是协方差几何，而会转移到三阶张量指纹。
\end{conjecture}
