\begin{theorem}[环熵阈值的强制边判据]\label{thm:conclusion-chowliu-forced-edge-threshold}
\leanverified{paper\_conclusion\_chowliu\_forced\_edge\_threshold}
设 \(R\in\RR^{\kappa\times\kappa}\) 为正定相关矩阵，\(T\) 为顶点集 \([,\kappa,]\) 上的生成树，并记
\[
L_T(R):=
-\log\det R+\sum_{(i,j)\in T}\log(1-R_{ij}^2),
\qquad
L_\ast(R):=\min_T L_T(R).
\]
对任意边 \(e=(u,v)\)，若存在不含 \(e\) 的生成树，则定义
\[
\underline I_e(R):=
\inf_{T:\,e\notin T}
\Bigl[-\log\bigl(1-\rho_{uv\mid S_e^T}^2\bigr)\Bigr],
\]
其中 \(S_e^T\) 为树 \(T\) 上连接 \(u,v\) 的唯一路径除去端点后的分离集；若不存在不含 \(e\) 的生成树，则约定 \(\underline I_e(R)=+\infty\)。则有严格蕴含
\[
\underline I_e(R)>L_\ast(R)
\Longrightarrow
e\ \text{属于每一棵最优 Chow--Liu 生成树}.
\]
\end{theorem}
\begin{proof}
若不存在不含 \(e\) 的生成树，则结论显然。现设存在不含 \(e\) 的生成树，并取任意一棵最优树 \(T_\ast\) 使
\[
L_{T_\ast}(R)=L_\ast(R).
\]
若 \(e\notin T_\ast\)，则由结论 \ref{con:xi-det-entropy-tree-basecycle-cmi-certificate}，
\[
-\log\bigl(1-\rho_{uv\mid S_e^{T_\ast}}^2\bigr)\le L_{T_\ast}(R)=L_\ast(R).
\]
而按 \(\underline I_e(R)\) 的定义，
\[
\underline I_e(R)\le -\log\bigl(1-\rho_{uv\mid S_e^{T_\ast}}^2\bigr),
\]
从而得到
\[
\underline I_e(R)\le L_\ast(R),
\]
与假设矛盾。故任何最优树都必须包含 \(e\)。证毕。
\end{proof}

\begin{theorem}[超临界骨架的唯一树判据]\label{thm:conclusion-chowliu-unique-tree-threshold}
\leanverified{paper\_conclusion\_chowliu\_unique\_tree\_threshold}
设 \(T_0\) 为顶点集 \([,\kappa,]\) 上的一棵生成树。若
\[
L_{T_0}(R)<\min_{e\in T_0}\underline I_e(R),
\]
则 \(T_0\) 不仅是 Chow--Liu 最优树，而且是唯一最优树。等价地，\(T_0\) 是唯一使
\[
\sum_{(i,j)\in T}-\log(1-R_{ij}^2)
\]
达到最大值的生成树。
\end{theorem}
\begin{proof}
由于 \(-\log\det R\) 与树 \(T\) 无关，最小化 \(L_T(R)\) 与最大化
\[
\mathcal W_R(T):=\sum_{(i,j)\in T}-\log(1-R_{ij}^2)
\]
完全等价。

现取任意一棵最优树 \(T_\ast\)。由最优性，
\[
L_\ast(R)=L_{T_\ast}(R)\le L_{T_0}(R).
\]
若存在某条 \(e\in T_0\) 不属于 \(T_\ast\)，则由定理
\ref{thm:conclusion-chowliu-forced-edge-threshold} 的逆否命题必有
\[
\underline I_e(R)\le L_\ast(R)\le L_{T_0}(R),
\]
这与假设
\[
L_{T_0}(R)<\min_{e\in T_0}\underline I_e(R)
\]
矛盾。故 \(T_0\subseteq T_\ast\)。两者同为生成树且边数都等于 \(\kappa-1\)，故只能有
\[
T_\ast=T_0.
\]
唯一性得证。证毕。
\end{proof}

\begin{theorem}[等距弱相干相中的链树刚性缺口]\label{thm:conclusion-weak-coherence-chain-tree-gap}
\leanverified{paper\_conclusion\_weak\_coherence\_chain\_tree\_gap}
置顶点为线性顺序 \(\{1,\dots,\kappa\}\)，并设边权只依赖距离：
\[
w_{ij}=W(|i-j|),
\qquad
W(1)>W(2)>\cdots>0.
\]
定义总树权
\[
\mathcal W(T):=\sum_{(i,j)\in T}w_{ij},
\]
以及相邻链树
\[
T_{\mathrm{chain}}:=\{(1,2),(2,3),\dots,(\kappa-1,\kappa)\}.
\]
则对一切 \(\kappa\ge 3\)，有精确组合公式
\[
\mathcal W(T_{\mathrm{chain}})
-\max_{T\neq T_{\mathrm{chain}}}\mathcal W(T)
=
W(1)-W(2).
\]

进一步，在结论 \ref{thm:conclusion-long-range-kernel-quartic-memory-law} 的等距弱相干制度
\[
s_j=s_0+(j-1)\Delta,
\qquad
A(\Delta)=\frac{\Delta}{2\sinh(\Delta/2)},
\qquad
\Delta\to\infty
\]
中，令
\[
a:=A(\Delta)\sim \Delta e^{-\Delta/2},
\qquad
b:=A(2\Delta)\sim 2\Delta e^{-\Delta},
\]
并记
\[
w_1:=-\log(1-a^2),
\qquad
w_2:=-\log(1-b^2).
\]
则
\[
\mathcal W(T_{\mathrm{chain}})
-\max_{T\neq T_{\mathrm{chain}}}\mathcal W(T)
=
w_1-w_2
\sim
\Delta^2e^{-\Delta}.
\]
\end{theorem}
\begin{proof}
任意生成树若不等于 \(T_{\mathrm{chain}}\)，则不可能全部由长度 \(1\) 的边组成；否则它只能是唯一的路径树 \(T_{\mathrm{chain}}\)。因此任意 \(T\neq T_{\mathrm{chain}}\) 至少含有一条长度 \(\ge 2\) 的边，并至多含有 \(\kappa-2\) 条长度 \(1\) 的边。由于 \(W(d)\) 关于 \(d\) 严格递减，所有长度 \(\ge 2\) 的边中最重者恰为长度 \(2\) 的边，权为 \(W(2)\)。故
\[
\mathcal W(T)\le (\kappa-2)W(1)+W(2)
\qquad (T\neq T_{\mathrm{chain}}).
\]
另一方面，
\[
\mathcal W(T_{\mathrm{chain}})=(\kappa-1)W(1).
\]
于是
\[
\mathcal W(T_{\mathrm{chain}})-\mathcal W(T)\ge W(1)-W(2).
\]

反过来，对任意 \(j\in\{1,\dots,\kappa-2\}\)，删去边 \((j+1,j+2)\) 并加入 \((j,j+2)\)，得到一棵生成树 \(T_j\)。它恰含有 \(\kappa-2\) 条长度 \(1\) 的边与一条长度 \(2\) 的边，因此
\[
\mathcal W(T_j)=(\kappa-2)W(1)+W(2),
\]
上界可达，从而缺口恰为 \(W(1)-W(2)\)。

在弱相干显式核上，\(W(1)=w_1\)、\(W(2)=w_2\)。又
\[
w_1=-\log(1-a^2)=a^2+O(a^4)\sim \Delta^2e^{-\Delta},
\]
\[
w_2=-\log(1-b^2)=b^2+O(b^4)\sim 4\Delta^2e^{-2\Delta}.
\]
故
\[
w_1-w_2\sim \Delta^2e^{-\Delta}.
\]
证毕。
\end{proof}

\begin{theorem}[组合冻结先于环熵显影的两尺度分离]\label{thm:conclusion-combinatorial-freezing-precedes-loop-entropy}
在定理 \ref{thm:conclusion-weak-coherence-chain-tree-gap} 的等距弱相干制度下，记链树环熵为
\[
L_{\mathrm{chain}}(R):=\mathcal L_{\mathrm{chain}}(R),
\]
并令
\[
G_{\mathrm{chain}}(\Delta):=
\mathcal W(T_{\mathrm{chain}})
-\max_{T\neq T_{\mathrm{chain}}}\mathcal W(T).
\]
则有
\[
\frac{1}{\kappa}L_{\mathrm{chain}}(R)\sim \Delta^4e^{-2\Delta},
\qquad
G_{\mathrm{chain}}(\Delta)\sim \Delta^2e^{-\Delta},
\]
从而
\[
\frac{L_{\mathrm{chain}}(R)/\kappa}{G_{\mathrm{chain}}(\Delta)}
\sim
\Delta^2e^{-\Delta}\longrightarrow 0
\qquad (\Delta\to\infty).
\]
\end{theorem}
\begin{proof}
由结论 \ref{con:xi-det-entropy-equispaced-loop-entropy-scaling}，
\[
\frac{1}{\kappa}L_{\mathrm{chain}}(R)
=
\frac{1}{\kappa}\mathcal L_{\mathrm{chain}}(R)
\sim \Delta^4e^{-2\Delta}.
\]
而定理 \ref{thm:conclusion-weak-coherence-chain-tree-gap} 已给出
\[
G_{\mathrm{chain}}(\Delta)\sim \Delta^2e^{-\Delta}.
\]
两式相除即可得到
\[
\frac{\Delta^4e^{-2\Delta}}{\Delta^2e^{-\Delta}}
=
\Delta^2e^{-\Delta}\to 0.
\]
证毕。
\end{proof}

\begin{theorem}[基本割容量乘积对回路格协体积的 Hadamard 刚性]\label{thm:conclusion-fundamental-cut-hadamard-rigidity}
\leanverified{paper\_conclusion\_fundamental\_cut\_hadamard\_rigidity}
设 \(G=(V,E)\) 为有限连通正权图，边权记为 \(w_e>0\)，并令
\[
W:=\mathrm{diag}(w_e)_{e\in E}.
\]
取任意生成树 \(T\subseteq E\)。对每条树边 \(t\in T\)，记其基本割为 \(\delta_t\)，容量为
\[
c_t:=\sum_{e\in\delta_t}w_e.
\]
再令 \(\Lambda_{\mathrm{fl}}:=\ker(B)\cap\ZZ^E\) 为相应整数流格，并以定理 \ref{thm:xi-flow-lattice-gram-determinant-tree-weight} 的电阻型二次型记其协体积为
\[
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}}).
\]
则有
\[
\tau_w(G)\le \prod_{t\in T}c_t,
\qquad
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})
\le
\left(\frac{\prod_{t\in T}c_t}{\det W}\right)^{1/2}.
\]

更强地，若 \(C\) 为树 \(T\) 的基本割矩阵，并定义加权割向量
\[
u_t:=(\sqrt{w_e}\,C_{t,e})_{e\in E}\in\RR^E,
\]
则上述两个不等式取等，当且仅当
\[
\langle u_t,u_s\rangle=0
\qquad (t\neq s),
\]
亦即树 \(T\) 的全部基本割在加权边空间 \(\ell^2(E,w)\) 中两两正交。
\end{theorem}
\begin{proof}
由命题 \ref{prop:spg-kirchhoff-fundamental-cut-capacity-hadamard}，
\[
\tau_w(G)\le \prod_{t\in T}c_t.
\]
在该命题的证明中，若记
\[
M:=C\,\mathrm{diag}(w)\,C^\top,
\]
则 \(M_{tt}=c_t\)，故 Hadamard 不等式给出
\[
\det M\le \prod_{t\in T}M_{tt}=\prod_{t\in T}c_t.
\]
而 \(M\) 的第 \(t\) 行恰是向量 \(u_t\)，所以 Hadamard 不等式取等当且仅当这些行向量两两正交，即
\[
\langle u_t,u_s\rangle=0
\qquad (t\neq s).
\]

另一方面，由定理 \ref{thm:xi-flow-lattice-gram-determinant-tree-weight}，
\[
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})
=
\sqrt{\tau_w(G)/\det W}.
\]
将前一不等式代入即得
\[
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})
\le
\left(\frac{\prod_{t\in T}c_t}{\det W}\right)^{1/2}.
\]
等号判据与 \(\tau_w(G)\) 的等号判据完全一致。证毕。
\end{proof}

\begin{theorem}[回路格 \texorpdfstring{$\theta$}{theta} 前因子的树预算下界]\label{thm:conclusion-cycle-lattice-theta-tree-budget-lower-bound}
沿用定理 \ref{thm:conclusion-fundamental-cut-hadamard-rigidity} 的记号，并令
\[
r:=\operatorname{rank}\Lambda_{\mathrm{fl}}.
\]
定义电阻型回路格热核
\[
\Theta_{\Lambda_{\mathrm{fl}},W^{-1}}(t)
:=
\sum_{x\in\Lambda_{\mathrm{fl}}}
\exp\!\bigl(-\pi t\,x^\top W^{-1}x\bigr),
\qquad t>0.
\]
则对任意生成树 \(T\) 都有
\[
\Theta_{\Lambda_{\mathrm{fl}},W^{-1}}(t)
\ge
t^{-r/2}
\left(\frac{\det W}{\prod_{t\in T}c_t}\right)^{1/2}
\bigl(1+o(1)\bigr)
\qquad (t\downarrow 0).
\]
若记
\[
\mathcal C_\ast(G):=\min_T \prod_{t\in T}c_t,
\]
则进一步得到统一下界
\[
\Theta_{\Lambda_{\mathrm{fl}},W^{-1}}(t)
\ge
t^{-r/2}
\left(\frac{\det W}{\mathcal C_\ast(G)}\right)^{1/2}
\bigl(1+o(1)\bigr)
\qquad (t\downarrow 0).
\]
并且该下界在首项常数上取等，当且仅当存在一棵达到 \(\mathcal C_\ast(G)\) 的生成树，其基本割向量在 \(\ell^2(E,w)\) 中两两正交。
\end{theorem}
\begin{proof}
对定理 \ref{thm:pom-cycle-lattice-theta-poisson-duality} 取导纳
\[
c_e:=1/w_e,
\]
则其二次型正是
\[
Q_{W^{-1}}(x)=x^\top W^{-1}x.
\]
令 \(x_0=0\)，由 Poisson 对偶公式，\(t\downarrow 0\) 时有
\[
\Theta_{\Lambda_{\mathrm{fl}},W^{-1}}(t)
=
t^{-r/2}\det_{\Lambda_{\mathrm{fl}}}(Q_{W^{-1}})^{-1/2}\bigl(1+o(1)\bigr).
\]
再由定理 \ref{thm:xi-flow-lattice-gram-determinant-tree-weight}，
\[
\det_{\Lambda_{\mathrm{fl}}}(Q_{W^{-1}})
=
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})^2.
\]
故
\[
\Theta_{\Lambda_{\mathrm{fl}},W^{-1}}(t)
=
t^{-r/2}\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})^{-1}\bigl(1+o(1)\bigr).
\]

由定理 \ref{thm:conclusion-fundamental-cut-hadamard-rigidity}，
\[
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})
\le
\left(\frac{\prod_{t\in T}c_t}{\det W}\right)^{1/2},
\]
取倒数即得
\[
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})^{-1}
\ge
\left(\frac{\det W}{\prod_{t\in T}c_t}\right)^{1/2}.
\]
代回便得第一条下界。对所有生成树取最优即得第二条。

若存在达到 \(\mathcal C_\ast(G)\) 的生成树 \(T_\ast\) 且其基本割向量两两正交，则定理 \ref{thm:conclusion-fundamental-cut-hadamard-rigidity} 的等号成立，因而
\[
\mathrm{covol}_{W^{-1}}(\Lambda_{\mathrm{fl}})
=
\left(\frac{\mathcal C_\ast(G)}{\det W}\right)^{1/2}.
\]
于是统一下界在首项常数上取等。反向同理。证毕。
\end{proof}

\begin{conjecture}[弱相干链相中的同一骨架双最优]\label{conj:conclusion-weak-coherence-chain-double-optimality}
在定理 \ref{thm:conclusion-weak-coherence-chain-tree-gap} 与定理 \ref{thm:conclusion-cycle-lattice-theta-tree-budget-lower-bound} 的等距弱相干制度下，设 \(G_\Delta\) 为由相关矩阵 \(R(\Delta)\) 诱导的完全加权图。则当 \(\Delta\to\infty\) 时，相邻链树 \(T_{\mathrm{chain}}\) 同时满足以下两种渐近最优性：
\[
T_{\mathrm{chain}}
=
\arg\max_T\sum_{e\in T}w_e,
\qquad
T_{\mathrm{chain}}
=
\arg\min_T\prod_{t\in T}c_t(\Delta).
\]
更强地，若对链树的基本割向量作归一化，则存在
\[
\langle u_t,u_s\rangle
=
o\!\bigl(\|u_t\|\,\|u_s\|\bigr)
\qquad
(t\neq s,\ \Delta\to\infty),
\]
从而链树不仅在 Chow--Liu 意义下唯一冻结，而且在回路格 \(\theta\) 前因子意义下也渐近饱和树预算下界。
\end{conjecture}

\noindent
由此，弱相干链相中的树选择问题与回路格热核问题不再是两个彼此平行的附属模块：一方面，条件互信息阈值把某些边直接提升为所有最优树的强制骨架，并把唯一树判据压缩为一条显式不等式；另一方面，等距弱相干显式核又表明，相邻链树的组合冻结以 \(\Delta^2e^{-\Delta}\) 的尺度先行发生，而每点环熵直到 \(\Delta^4e^{-2\Delta}\) 才开始显影，二者之间存在严格的一指数尺度分离。再进一步，基本割容量乘积通过 Hadamard 刚性把同一树骨架投影到回路格协体积与短时 \(\theta\) 前因子上，从而把“加法权最优”的 Chow--Liu 机制与“乘法预算最优”的 cut/cycle-lattice 机制压到同一骨架问题的两种投影之中。
