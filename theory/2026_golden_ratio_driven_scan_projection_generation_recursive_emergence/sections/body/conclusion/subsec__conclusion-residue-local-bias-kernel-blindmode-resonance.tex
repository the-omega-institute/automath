\subsection{余数剖面反演、局部偏置盲模与 Fibonacci 共振的双重锁定}\label{subsec:conclusion-residue-local-bias-kernel-blindmode-resonance}
沿用结论 \ref{con:pom-fold-factor-chain-explicit-kernel}、定理 \ref{thm:xi-fold-bias-tangent-factorization-gram-psd}、定理 \ref{thm:conclusion-fold-golden-resonance-collision-gap-hard-floor} 与定理 \ref{thm:fold-collision-spectrum} 的记号。记
\[
M:=F_{m+2},
\qquad
d_m(r):=\#\{\omega\in\{0,1\}^m:\ r(\omega)=r\},
\qquad
\mu_m(r):=\frac{d_m(r)}{2^m},
\]
并在结论 \ref{con:pom-fold-factor-chain-explicit-kernel} 的同余坐标识别下记
\[
P_m:=\overline P_m^{\square}.
\]
对 \(1\le k\le m\) 记
\[
b_{m,k}(r):=\#\{\omega:\ r(\omega)=r,\ \omega_k=1\},
\qquad
p_k(r):=\frac{b_{m,k}(r)}{d_m(r)}
\quad (d_m(r)>0).
\]
前文已经把单坐标条件偏置的 Fourier 乘子、推前核的显式步长公式以及 bulk Fibonacci 共振的碰撞硬下界完全写成闭式。把这些接口并置之后，还可进一步抽出如下几条彼此扣合的后果：奇窗口中余数多重度剖面已足以重建局部动力学；偶窗口中这一完备性则沿一条严格的一维奇偶角色方向破裂，但这条缺口又只需单个横截标量即可补全；而第一坐标偏置能量又被与全局碰撞缺口相同的两条 Fibonacci 共振频率以常数级强制激活。

\begin{theorem}[奇窗口上的有限坐标约束完全反演]\label{thm:conclusion-residue-local-statistics-complete-reconstruction}
\leanverified{paper\_conclusion\_residue\_local\_statistics\_complete\_reconstruction}
对任意非空 \(I\subseteq\{1,\dots,m\}\) 与 \(v\in\{0,1\}^{I}\)，定义模式计数
\[
A_{m,I}(r;v)
:=
\#\{\omega\in\{0,1\}^m:\ r(\omega)=r,\ \omega_i=v_i\ \forall i\in I\}.
\]
再记
\[
a_{m,I}(r):=A_{m,I}(r;0_I),
\qquad
K_I:=\sum_{S\subseteq I}\delta_{\sum_{i\in S}F_{i+1}}
\quad\text{on }\ZZ/(M\ZZ).
\]
若 \(M=F_{m+2}\) 为奇数，则余数多重度剖面 \(d_m(\cdot)\) 唯一决定一切有限坐标模式计数 \(A_{m,I}(r;v)\)。特别地，\(d_m\) 唯一决定全部单坐标条件概率 \(p_k(r)\)，因而唯一决定推前核 \(P_m\)。
\end{theorem}
\begin{proof}
对任意固定 \(I\)，按 \(I\)-坐标是否取 \(1\) 分层，可得严格卷积恒等式
\[
d_m=K_I*a_{m,I}.
\]
而一般模式只是在余类坐标上作平移：
\[
A_{m,I}(r;v)
=
a_{m,I}\!\left(r-\sum_{i\in I:\ v_i=1}F_{i+1}\right).
\]
因此只需证明卷积算子 \(f\mapsto K_I*f\) 可逆。

对任意角色 \(\chi\in \widehat{\ZZ/(M\ZZ)}\)，有
\[
\widehat K_I(\chi)=\prod_{i\in I}\bigl(1+\chi(F_{i+1})\bigr).
\]
当 \(M\) 为奇数时，每个 \(\chi(F_{i+1})\) 的阶都整除奇数 \(M\)，故不可能等于 \(-1\)。于是
\[
\widehat K_I(\chi)\neq 0
\qquad (\forall \chi).
\]
Fourier 对角化遂给出唯一反演，因此 \(a_{m,I}\) 由 \(d_m\) 唯一决定，进而一切 \(A_{m,I}(r;v)\) 均唯一。

取 \(I=\{k\}\) 即得 \(b_{m,k}(r)=A_{m,\{k\}}(r;1)\)，从而 \(p_k(r)\) 被唯一决定。再由结论 \ref{con:pom-fold-factor-chain-explicit-kernel} 的显式核公式
\[
P_m(r,r+F_{k+1})=\frac{1}{m}\bigl(1-p_k(r)\bigr),
\qquad
P_m(r,r-F_{k+1})=\frac{1}{m}p_k(r),
\]
可知 \(P_m\) 亦由 \(d_m\) 唯一恢复。证毕。
\end{proof}

\begin{theorem}[单坐标约束逆问题的精确障碍秩]\label{thm:conclusion-residue-single-coordinate-obstruction-rank}
\leanverified{paper\_conclusion\_residue\_single\_coordinate\_obstruction\_rank}
对 \(w\in\ZZ/(M\ZZ)\) 定义平移算子
\[
(T_wf)(r):=f(r+w),
\qquad
L_w:=I+T_w
\quad\text{on }\CC^{\ZZ/(M\ZZ)}.
\]
再对 \(1\le k\le m\) 记
\[
g_{m,k}:=\gcd(F_{m+2},F_{k+1})=F_{\gcd(m+2,k+1)}.
\]
则
\[
\dim_{\CC}\ker\bigl(I+T_{F_{k+1}}\bigr)
=
\begin{cases}
0,& \dfrac{F_{m+2}}{g_{m,k}}\ \text{为奇数},\\[1.2ex]
g_{m,k},& \dfrac{F_{m+2}}{g_{m,k}}\ \text{为偶数}.
\end{cases}
\]
特别地，在仿射逆问题
\[
\bigl(I+T_{F_{k+1}}\bigr)b=d_m
\]
可解的前提下，其解空间维数恰等于上式；因而要从 \(d_m\) 中唯一恢复 \(b_{m,k}\)，所需附加独立标量参数的最小数目正是该维数。
\end{theorem}
\begin{proof}
对标准角色
\[
\chi_t(r):=e^{2\pi \mathrm{i}tr/M}
\qquad (t\in\ZZ/(M\ZZ)),
\]
有
\[
T_w\chi_t=\chi_t(w)\chi_t,
\]
故
\[
\ker(L_w)=\operatorname{span}\{\chi_t:\ \chi_t(w)=-1\}.
\]
条件 \(\chi_t(w)=-1\) 等价于
\[
e^{2\pi \mathrm{i}tw/M}=-1
\iff
tw\equiv \frac{M}{2}\pmod M.
\]
设 \(g:=\gcd(M,w)\)。该同余可解当且仅当 \(g\mid M/2\)，亦即 \(M/g\) 为偶数；一旦可解，模 \(M\) 的解恰有 \(g\) 个。于是
\[
\dim_{\CC}\ker(L_w)
=
\begin{cases}
0,& M/g\ \text{为奇数},\\
g,& M/g\ \text{为偶数}.
\end{cases}
\]
代入
\[
M=F_{m+2},
\qquad
w=F_{k+1},
\qquad
\gcd(F_a,F_b)=F_{\gcd(a,b)},
\]
即得所示公式。对仿射方程 \(L_w b=d_m\)，任意两解之差属于 \(\ker(L_w)\)，故可解时其仿射解空间维数正等于核维数；\(b_{m,k}\) 本身已给出一条解，故最小唯一化参数数目也正是此维数。证毕。
\end{proof}

\begin{corollary}[模三相位与局部可逆性的精确二分]\label{cor:conclusion-residue-mod3-local-reversibility-phase-transition}
\leanverified{paper\_conclusion\_residue\_mod3\_local\_reversibility\_phase\_transition}
下列命题彼此等价：
\[
F_{m+2}\ \text{为奇数},
\qquad
m\not\equiv 1\pmod 3,
\qquad
d_m\ \text{唯一决定全部单坐标统计 }b_{m,k},p_k\ \text{与推前核 }P_m.
\]
相反，当
\[
m\equiv 1\pmod 3
\]
时，\(d_m\) 已不足以唯一确定最简单的单坐标统计 \(b_{m,1}\)；于是 \(P_m\) 在该模 \(3\) 相位上必然存在局部不可逆缺陷。
\end{corollary}
\begin{proof}
Fibonacci 奇偶律给出
\[
F_n\ \text{为偶数}\iff 3\mid n.
\]
因此
\[
F_{m+2}\ \text{为奇数}
\iff
m+2\not\equiv 0\pmod 3
\iff
m\not\equiv 1\pmod 3.
\]
若 \(F_{m+2}\) 为奇数，则定理 \ref{thm:conclusion-residue-local-statistics-complete-reconstruction} 立即给出唯一恢复性。

若 \(m\equiv 1\pmod 3\)，则 \(M=F_{m+2}\) 为偶数。对 \(k=1\) 有 \(F_{k+1}=F_2=1\)，并且
\[
g_{m,1}=\gcd(M,1)=1.
\]
由定理 \ref{thm:conclusion-residue-single-coordinate-obstruction-rank}，
\[
\dim_{\CC}\ker(I+T_1)=1.
\]
由于 \(b_{m,1}\) 本身是一条解，方程 \((I+T_1)b=d_m\) 的仿射解空间便非单点，从而 \(b_{m,1}\) 不由 \(d_m\) 唯一确定，\(p_1\) 与 \(P_m\) 亦随之不唯一。证毕。
\end{proof}

\begin{corollary}[偶窗口中的普适一维奇偶盲模]\label{cor:conclusion-residue-even-window-universal-parity-blind-mode}

\ker(I+T_1)=\CC\cdot \chi_{M/2},

\chi_{M/2}(r)=(-1)^r.

因此偶窗口中的最小局部不可逆性并非高维弥散障碍，而是一条严格的一维奇偶角色盲模。
\end{corollary}
\begin{proof}
由前一推论，\(m\equiv 1\pmod 3\) 时 \(M\) 为偶数。对 \(k=1\) 已有

g_{m,1}=1,

\dim_{\CC}\ker(I+T_1)=1.

而

\chi_t(1)=-1
\iff
e^{2\pi \mathrm{i}t/M}=-1
\iff
t\equiv M/2\pmod M.

故唯一的核角色正是
\[
\chi_{M/2}(r)=e^{\pi \mathrm{i}r}=(-1)^r.
\]
证毕。
\end{proof}

\end{proof}

\begin{theorem}[条件漂移的一切奇矩湮灭]\label{thm:conclusion-residue-drift-all-odd-moments-vanish}
对 \(1\le k\le m\) 定义单坐标漂移
\[
\mathrm{drift}_k(r)
:=
P_m(r,r+F_{k+1})-P_m(r,r-F_{k+1})
=
\frac{1}{m}\bigl(1-2p_k(r)\bigr).
\]
再定义补码对合
\[
\tau_m(r)\equiv F_{m+1}-2-r \pmod{F_{m+2}}.
\]
则对任意 \(k\) 与任意奇多项式 \(Q\) 都有
\[
\sum_{r\in\ZZ/(M\ZZ)}\mu_m(r)\,Q(\mathrm{drift}_k(r))=0.
\]
更一般地，对任意满足 \(f\circ\tau_m=f\) 的测试函数 \(f\)，成立
\[
\sum_{r\in\ZZ/(M\ZZ)}\mu_m(r)\,\mathrm{drift}_k(r)\,f(r)=0.
\]
\leanverified{paper\_conclusion\_residue\_drift\_all\_odd\_moments\_vanish}
\end{theorem}
\begin{proof}
对每个 \(\omega\in\{0,1\}^m\)，记其补码为 \(\omega^c:=1-\omega\)。则
\[
r(\omega^c)
\equiv
\sum_{j=1}^{m}(1-\omega_j)F_{j+1}
\equiv
\sum_{j=1}^{m}F_{j+1}-r(\omega)
\pmod M.
\]
而
\[
\sum_{j=1}^{m}F_{j+1}=F_{m+3}-2\equiv F_{m+1}-2\pmod{F_{m+2}},
\]
故
\[
r(\omega^c)=\tau_m(r(\omega)).
\]
补码映射是双射，遂有
\[
\mu_m(\tau_m(r))=\mu_m(r).
\]
同时，\(\omega_k^c=1-\omega_k\)，故
\[
b_{m,k}(\tau_m(r))=d_m(r)-b_{m,k}(r),
\qquad
p_k(\tau_m(r))=1-p_k(r).
\]
于是
\[
\mathrm{drift}_k(\tau_m(r))
=
\frac{1}{m}\bigl(1-2p_k(\tau_m(r))\bigr)
=
-\frac{1}{m}\bigl(1-2p_k(r)\bigr)
=
-\mathrm{drift}_k(r).
\]
因此 \(\mathrm{drift}_k\) 关于 \(\tau_m\) 为奇函数，而 \(\mu_m\) 关于 \(\tau_m\) 不变。对每一对 \(\{r,\tau_m(r)\}\) 配对求和，奇多项式 \(Q(\mathrm{drift}_k(r))\) 或一般的 \(\mathrm{drift}_k(r)f(r)\) 都逐对相消；若 \(r=\tau_m(r)\)，则上式强制 \(\mathrm{drift}_k(r)=0\)。证毕。
\end{proof}

\begin{theorem}[第一坐标偏置能量的谱忠实性]\label{thm:conclusion-first-coordinate-bias-energy-spectral-faithfulness}
\leanverified{paper\_conclusion\_first\_coordinate\_bias\_energy\_spectral\_faithfulness}
定义第一坐标的条件偏置密度
\[
\nu_{m,1}(r):=\mu_m(r)\bigl(2p_1(r)-1\bigr)
=
\frac{2b_{m,1}(r)-d_m(r)}{2^m},
\qquad
E_{m,1}:=\sum_{r\in\ZZ/(M\ZZ)}\nu_{m,1}(r)^2.
\]
则下列命题彼此等价：
\[
E_{m,1}=0,
\qquad
d_m(r)\equiv \frac{2^m}{F_{m+2}},
\qquad
\mathsf{Col}_m=\frac{1}{F_{m+2}}.
\]
因此第一坐标已经是全局均匀性的忠实局部探针；任意非均匀折叠都无法在该偏置能量下隐身。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-fold-bias-tangent-factorization-gram-psd} 在 \(k=1\) 处的 Fourier 乘子公式与 Parseval 恒等式，
\[
E_{m,1}
=
\frac{1}{4M}
\sum_{t\in\ZZ/(M\ZZ)}
\tan^2\!\Bigl(\frac{\pi t}{M}\Bigr)\,
|\widehat\mu_m(t)|^2,
\]
其中在 \(\cos(\pi t/M)=0\) 时按连续延拓解释该项。若 \(t=M/2\) 使分母为零，则
\[
\widehat d_m(t)=\prod_{j=1}^{m}\bigl(1+e^{-2\pi \mathrm{i}tF_{j+1}/M}\bigr)=0
\]
已由 \(j=1\) 的因子 \(1+e^{-\pi \mathrm{i}}\) 强制，因此该奇点对应的 \(|\widehat\mu_m(t)|\) 本身为零。故对一切满足 \(\widehat\mu_m(t)\neq 0\) 的非零频率，权重均严格为正。

于是
\[
E_{m,1}=0
\iff
\widehat\mu_m(t)=0\qquad(\forall t\neq 0).
\]
Fourier 反演表明这等价于 \(\mu_m\) 在 \(\ZZ/(M\ZZ)\) 上均匀，即
\[
d_m(r)\equiv \frac{2^m}{M}.
\]
另一方面，由定理 \ref{thm:fold-collision-spectrum}，
\[
\mathsf{Col}_m
=
\frac{1}{M}\sum_{t\in\ZZ/(M\ZZ)}|\widehat\mu_m(t)|^2,
\]
故仅有零模时恰得
\[
\mathsf{Col}_m=\frac{1}{M}.
\]
反之，只要 \(\mathsf{Col}_m=1/M\)，所有非零 Fourier 模便必须全部消失，从而 \(d_m\) 均匀且 \(E_{m,1}=0\)。证毕。
\end{proof}

\begin{theorem}[第一坐标偏置能量的共振硬下界]\label{thm:conclusion-first-coordinate-bias-energy-resonant-hard-floor}
\leanverified{paper\_conclusion\_first\_coordinate\_bias\_energy\_resonant\_hard\_floor}
记
\[
C_\varphi:=\prod_{n=2}^{\infty}\bigl|\cos(\pi\varphi^{-n})\bigr|>0.
\]
则第一坐标偏置能量满足
\[
\liminf_{m\to\infty}F_{m+2}\,E_{m,1}
\ge
\frac12\tan^2(\pi\varphi^{-2})\,C_\varphi^2
>0.
\]
\end{theorem}
\begin{proof}
沿用上一证明中的谱平均式，并记
\[
R_m(t):=\tan^2\!\Bigl(\frac{\pi t}{F_{m+2}}\Bigr).
\]
则
\[
F_{m+2}E_{m,1}
=
\frac14\sum_{t\in\ZZ/(F_{m+2}\ZZ)}R_m(t)\,|\widehat\mu_m(t)|^2.
\]
保留两项 \(t=F_m\) 与 \(t=F_{m+1}\) 即得
\[
F_{m+2}E_{m,1}
\ge
\frac14\Bigl(
R_m(F_m)\,a_m(F_m)^2
+
R_m(F_{m+1})\,a_m(F_{m+1})^2
\Bigr),
\]
因为 \(|\widehat\mu_m(F_m)|=a_m(F_m)\)、\(|\widehat\mu_m(F_{m+1})|=a_m(F_{m+1})\)。

由定理 \ref{thm:conclusion-fold-golden-resonance-collision-gap-hard-floor} 所调用的 bulk 共振极限，
\[
a_m(F_m)\longrightarrow C_\varphi,
\qquad
a_m(F_{m+1})\longrightarrow C_\varphi.
\]
另一方面，
\[
\frac{F_m}{F_{m+2}}\longrightarrow \varphi^{-2},
\qquad
\frac{F_{m+1}}{F_{m+2}}\longrightarrow \varphi^{-1}=1-\varphi^{-2},
\]
故由 \(\tan^2(\pi(1-x))=\tan^2(\pi x)\) 得
\[
R_m(F_m),\ R_m(F_{m+1})\longrightarrow \tan^2(\pi\varphi^{-2}).
\]
代入并取下极限，即得
\[
\liminf_{m\to\infty}F_{m+2}E_{m,1}
\ge
\frac12\tan^2(\pi\varphi^{-2})\,C_\varphi^2.
\]
证毕。
\end{proof}

\leanverified{paper\_conclusion\_first\_coordinate\_bias\_and\_global\_gap\_common\_resonant\_origin}

\end{proof}

\begin{conjecture}[固定尺度 bin-fold 的 Schur--parity 双层完备审计]\label{conj:conclusion-fixedscale-capacity-parity-two-layer-audit}
对每个固定分辨率 \(m\)，记由余数多重度剖面 \(d_m\) 定义的连续容量曲线为
\[
\mathcal C_m^{\mathrm{cont}}(T):=\sum_{r\in \ZZ/(F_{m+2}\ZZ)}\min\bigl(d_m(r),T\bigr).

则应存在一个有限维奇偶角色空间 \(\mathcal B_m\)，满足：
\begin{enumerate}
  \item \(\mathcal B_m=0\) 当且仅当 \(m\not\equiv 1\pmod 3\)；
  \item 当 \(m\equiv 1\pmod 3\) 时，\(\mathcal B_m\) 由全部局部逆问题的盲模角色生成，且其维数恰等于局部不可逆缺陷的总秩；
  \item 一旦附加一份奇偶补账
  \[
  \beta_m\in \mathcal B_m^\vee,
  \]
  数据对
  \[
  \bigl(\mathcal C_m^{\mathrm{cont}},\beta_m\bigr)
  \]
  就足以唯一恢复全部有限坐标模式计数 \(A_{m,I}(r;v)\) 与完整推前核 \(P_m\)。
\end{enumerate}
换言之，固定尺度 bin-fold 的最小完备审计对象应当分裂为一条由连续容量曲线承载的 Schur--统计层，与一条只在 \(m\equiv 1\pmod 3\) 时出现的 parity--局部补全层。
\end{conjecture}

\paragraph{固定尺度审计读法}
把上面的链条连在一起，固定尺度 bin-fold 的局部审计已经显出一种二层分裂：连续容量曲线与余数多重度剖面承载全部全局可见统计几何，而模 \(3\) 的偶窗口只额外析出一条严格的一维奇偶盲模；这条盲模既不会扩散成高维不可逆噪声，也不会改写全局碰撞缺口的共振来源，而只要求一个横截标量即可补全。若上述猜想成立，则 fixed-resolution 的真正最小完备对象将不再是单独的容量曲线，而是
\[
\bigl(\mathcal C_m^{\mathrm{cont}},\beta_m\bigr)
\]
这一类“Schur 坐标加 parity 补账”的二层审计对。
