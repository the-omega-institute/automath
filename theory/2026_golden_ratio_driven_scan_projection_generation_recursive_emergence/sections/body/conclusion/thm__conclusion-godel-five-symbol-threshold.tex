\begin{theorem}[五元事件字母表的不可降门槛]\label{thm:conclusion-godel-five-symbol-threshold}
设
\[
\Psi_m:\Omega_m\to X_m\times \Sigma^{\le L_m}
\]
为任意确定性可逆提升，满足
\[
\mathrm{pr}_{X_m}\circ\Psi_m=\Fold_m,
\qquad
|\Sigma|=M\ge 2.
\]
则必有
\[
\boxed{\;
L_m\ge \Bigl\lceil \log_M\!\Bigl(\frac{2^m}{F_{m+2}}\Bigr)\Bigr\rceil-1,\qquad
\liminf_{m\to\infty}\frac{L_m}{m}\ge \frac{\log(2/\varphi)}{\log M}.\;}
\]
特别地，若要求
\[
L_m\le \Bigl(\frac{4}{27}+o(1)\Bigr)m
\]
且每步只写入一个事件符号，则必有
\[
\boxed{\;M\ge 5.\;}
\]
\leanverified{godelLift\_alphabet\_threshold\_mge5}
\leanverified{paper\_conclusion\_godel\_five\_symbol\_threshold}
因此，五元事件字母表并非实现层的偶然选择，而是线性密度级可逆重写的一个信息论相变门槛。
\end{theorem}
\begin{proof}
正文关于可逆提升的计数下界已经给出
\[
\frac{2^m}{F_{m+2}}\le |\Sigma^{\le L_m}|<M^{L_m+1}.
\]
于是
\[
L_m+1>\log_M\!\Bigl(\frac{2^m}{F_{m+2}}\Bigr),
\]
从而
\[
L_m\ge \Bigl\lceil \log_M\!\Bigl(\frac{2^m}{F_{m+2}}\Bigr)\Bigr\rceil-1.
\]
再利用
\[
F_{m+2}\sim \frac{\varphi^{m+2}}{\sqrt5},
\]
即可得到
\[
\liminf_{m\to\infty}\frac{L_m}{m}\ge \frac{\log(2/\varphi)}{\log M}.
\]

若进一步要求
\[
L_m\le \Bigl(\frac{4}{27}+o(1)\Bigr)m,
\]
则必须有
\[
\frac{\log(2/\varphi)}{\log M}\le \frac{4}{27}.
\]
而
\[
\frac{\log(2/\varphi)}{\log 4}\approx 0.15288>\frac{4}{27}\approx 0.14815,
\]
故 \(M\le 4\) 不可能；于是必有 \(M\ge 5\)。证毕。
\end{proof}
