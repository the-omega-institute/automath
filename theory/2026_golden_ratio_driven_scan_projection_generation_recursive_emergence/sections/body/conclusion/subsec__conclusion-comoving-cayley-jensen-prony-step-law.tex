\subsection{共动 Cayley--Jensen 缺陷的离线约束与 Stieltjes--Prony 步长二分律}\label{subsec:conclusion-comoving-cayley-jensen-prony-step-law}
本小节把“共动 Jensen 缺陷 \(\Rightarrow\) 离线量上界”与“尺度递归步长 \(\Rightarrow\) Prony 病态性分区”合并为同一组可审计定理。

\begin{theorem}[偏移共动坐标下离线量的根式上界]\label{thm:conclusion-comoving-cayley-jensen-offline-root-bound}
\leanverified{paper\_conclusion\_comoving\_cayley\_jensen\_offline\_root\_bound}
设 Cayley 变换
\[
C:\mathbb H\to\mathbb D,\qquad
C(z)=\frac{1+iz}{1-iz}.
\]
令 \(\rho=\beta+i\gamma\) 为 \(\zeta(s)\) 的非平凡零点，且 \(\beta<\tfrac12\)。
记离线量
\[
\delta:=\frac12-\beta>0.
\]
对任意 \(x_0\in\RR\)，定义
\[
u:=|\gamma-x_0|,\qquad
w_{\rho,x_0}:=C\big((\gamma-x_0)+i\delta\big),\qquad
r_{\rho,x_0}:=|w_{\rho,x_0}|.
\]
设 \(F_{x_0}\) 为单位圆盘内全纯函数，且其零点集合（按重数）包含 \(w_{\rho,x_0}\)。
定义 Jensen 缺陷
\[
D_{x_0}(\varrho)
:=
\frac1{2\pi}\int_0^{2\pi}\log|F_{x_0}(\varrho e^{i\theta})|\,\dd\theta
-\log|F_{x_0}(0)|
\quad(0<\varrho<1).
\]
若某 \(\varrho\in(0,1)\)、\(\varepsilon\ge 0\) 满足
\[
D_{x_0}(\varrho)\le \varepsilon,
\]
并置
\[
r_0:=\varrho e^{-\varepsilon},
\qquad
A(r_0):=\frac{1+r_0^2}{1-r_0^2},
\]
则任意该共动截面上的离线零点都满足
\[
\delta^2-2A(r_0)\delta+(1+u^2)\ge 0.
\]
当
\[
u^2\le A(r_0)^2-1
\]
时，上式等价于根式上界
\[
\delta\le \Delta(u;r_0):=
A(r_0)-\sqrt{A(r_0)^2-(1+u^2)}.
\]
特别地，若 \(u=0\)（即 \(x_0=\gamma\)）则
\[
\delta\le \frac{1-r_0}{1+r_0}.
\]
\end{theorem}
\begin{proof}
由 Jensen 公式，若存在零点 \(a\) 满足 \(|a|<\varrho e^{-\varepsilon}\)，则
\[
D_{x_0}(\varrho)\ge \log\frac{\varrho}{|a|}>\varepsilon,
\]
与假设矛盾。故盘内零点均满足 \(|a|\ge r_0\)，特别地
\[
r_{\rho,x_0}\ge r_0.
\]

对 \(z=u+i\delta\) 直接计算
\[
|C(z)|^2
=
\frac{u^2+(1-\delta)^2}{u^2+(1+\delta)^2}.
\]
代入 \(r_{\rho,x_0}^2\ge r_0^2\) 得
\[
u^2+(1-\delta)^2\ge r_0^2\bigl(u^2+(1+\delta)^2\bigr),
\]
等价化简为
\[
\delta^2-2A(r_0)\delta+(1+u^2)\ge 0.
\]
若 \(u^2\le A(r_0)^2-1\)，判别式非负，得到两支根区间。
对离线量 \(\delta\in(0,\tfrac12)\) 的相关分支取小根，得到
\[
\delta\le A(r_0)-\sqrt{A(r_0)^2-(1+u^2)}.
\]
当 \(u=0\) 时
\[
|C(i\delta)|=\frac{1-\delta}{1+\delta}\ge r_0
\]
直接解得 \(\delta\le(1-r_0)/(1+r_0)\)。证毕。
\end{proof}

\begin{theorem}[中心偏移--半径精度的可行域等价与平方根换算]\label{thm:conclusion-comoving-cayley-jensen-pareto-feasible-region}
沿用定理 \ref{thm:conclusion-comoving-cayley-jensen-offline-root-bound} 记号。
固定目标阈值 \(\delta_0>0\)，并令
\[
A:=\frac{1+r_0^2}{1-r_0^2}.
\]
定义
\[
u_{\max}(\delta_0;r_0):=
\sqrt{\,2A\delta_0-\delta_0^2-1\,},
\]
其中右端取实数当且仅当
\[
2A\delta_0-\delta_0^2-1\ge 0.
\]
则以下两条等价：
\begin{enumerate}
  \item 在给定 \(r_0\) 下，任何满足 \(|\gamma-x_0|\le u_{\max}(\delta_0;r_0)\) 的共动中心均强制 \(\delta\le \delta_0\)；
  \item \((u,r_0)\) 落在二次可行域
  \[
  1+u^2\le 2A\delta_0-\delta_0^2.
  \]
\end{enumerate}
并且当 \(r_0\uparrow 1\) 时，
\[
u_{\max}(\delta_0;r_0)\sim 2\sqrt{\frac{\delta_0}{1-r_0^2}}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-comoving-cayley-jensen-offline-root-bound}，欲保证 \(\delta\le\delta_0\)，
等价于强制
\[
\Delta(u;r_0)\le\delta_0.
\]
即
\[
A-\sqrt{A^2-(1+u^2)}\le \delta_0
\]
等价于
\[
\sqrt{A^2-(1+u^2)}\ge A-\delta_0.
\]
平方并化简得
\[
1+u^2\le 2A\delta_0-\delta_0^2.
\]
反向推导同理，故两条等价。
由 \(u_{\max}\) 的定义立得第一条。

最后，
\[
A=\frac{1+r_0^2}{1-r_0^2}\sim \frac{2}{1-r_0^2}\qquad(r_0\uparrow 1),
\]
代入
\(
u_{\max}^2=2A\delta_0-\delta_0^2-1
\)
即得
\[
u_{\max}(\delta_0;r_0)\sim 2\sqrt{\frac{\delta_0}{1-r_0^2}}.
\]
证毕。
\end{proof}

\begin{theorem}[Stieltjes--Prony 节点分离的尺度步长律]\label{thm:conclusion-stieltjes-prony-step-optimal-law}
\leanverified{paper\_conclusion\_stieltjes\_prony\_step\_optimal\_law}
设
\[
u_n(t;\Delta)=\sum_{j=1}^{\kappa}c_j\,z_j(t;\Delta)^n,\qquad n=0,\dots,2\kappa-1,
\]
\[
z_j(t;\Delta):=\exp\!\bigl(-(t+\alpha_j)\Delta\bigr)e^{-i\gamma_j\Delta},
\]
并假设
\[
0<\underline\alpha\le \alpha_j\le \overline\alpha,\qquad
g_0:=\min_{j\neq \ell}|\gamma_j-\gamma_\ell|>0.
\]
定义节点最小分离度
\[
\operatorname{sep}(t;\Delta):=\min_{j\neq \ell}|z_j(t;\Delta)-z_\ell(t;\Delta)|.
\]
则对
\[
0<\Delta\le \frac{\pi}{g_0}
\]
成立下界
\[
\operatorname{sep}(t;\Delta)\ge
e^{-(t+\overline\alpha)\Delta}\cdot \frac{2}{\pi}\,g_0\,\Delta.
\]
对应的“主尺度目标” \(\Phi_t(\Delta):=e^{-t\Delta}\Delta\) 在 \((0,\infty)\) 上的唯一极大点为
\[
\Delta^\ast(t)=\frac1t.
\]
因此取递归步长 \(\Delta(t)=1/t\) 时，有
\[
\operatorname{sep}\!\left(t;\frac1t\right)
\ge
\frac{2}{\pi e}\,\frac{g_0}{t}\,e^{-\overline\alpha/t}.
\]
\end{theorem}
\begin{proof}
取任意 \(j\neq \ell\)，写
\[
a_j:=e^{-(t+\alpha_j)\Delta},\qquad
a_\ell:=e^{-(t+\alpha_\ell)\Delta}.
\]
则
\[
|z_j-z_\ell|^2
=
a_j^2+a_\ell^2-2a_ja_\ell\cos\big((\gamma_j-\gamma_\ell)\Delta\big)
=
(a_j-a_\ell)^2+4a_ja_\ell\sin^2\!\frac{(\gamma_j-\gamma_\ell)\Delta}{2}.
\]
故
\[
|z_j-z_\ell|
\ge
2\sqrt{a_ja_\ell}\left|\sin\frac{(\gamma_j-\gamma_\ell)\Delta}{2}\right|
\ge
2e^{-(t+\overline\alpha)\Delta}\left|\sin\frac{(\gamma_j-\gamma_\ell)\Delta}{2}\right|.
\]
当 \(0<\Delta\le\pi/g_0\) 时，
\[
\left|\frac{(\gamma_j-\gamma_\ell)\Delta}{2}\right|\le \frac{\pi}{2},
\]
用不等式 \(|\sin x|\ge \frac{2}{\pi}|x|\) 得
\[
|z_j-z_\ell|
\ge
2e^{-(t+\overline\alpha)\Delta}\cdot
\frac{2}{\pi}\cdot\frac{|\gamma_j-\gamma_\ell|\Delta}{2}
\ge
e^{-(t+\overline\alpha)\Delta}\frac{2}{\pi}g_0\Delta.
\]
取最小即得分离下界。

再看 \(\Phi_t(\Delta)=e^{-t\Delta}\Delta\)：
\[
\Phi_t'(\Delta)=e^{-t\Delta}(1-t\Delta),
\]
故唯一极大点为 \(\Delta^\ast=1/t\)。
代入分离下界即得最后一式。证毕。
\end{proof}

\begin{theorem}[尺度递归中的步长--病态性二分律]\label{thm:conclusion-prony-step-conditioning-dichotomy}
沿用定理 \ref{thm:conclusion-stieltjes-prony-step-optimal-law} 设定，令 \(\Delta(t)>0\)。
则：
\begin{enumerate}
  \item 若
  \[
  \liminf_{t\to\infty}t\Delta(t)=+\infty,
  \]
  则存在常数 \(c>0\) 与 \(t_0\) 使
  \[
  \operatorname{sep}(t;\Delta(t))\le e^{-c\,t\Delta(t)}
  \qquad (t\ge t_0),
  \]
  因而任何以 \(\operatorname{sep}^{-(\kappa-1)}\) 为主控量的 Vandermonde/Prony 稳定性界都满足
  \[
  \operatorname{sep}(t;\Delta(t))^{-(\kappa-1)}
  \ge
  \exp\!\bigl(\Omega(t\Delta(t))\bigr).
  \]
  \item 若取
  \[
  \Delta(t)=\frac1t,
  \]
  则
  \[
  \operatorname{sep}\!\left(t;\frac1t\right)\gg t^{-1},
  \]
  从而同类稳定性界可压到多项式量级：
  \[
  \operatorname{sep}\!\left(t;\frac1t\right)^{-(\kappa-1)}
  =O\!\left(t^{\kappa-1}\right).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
先证 (1)。
由定义
\[
|z_j(t;\Delta)-z_\ell(t;\Delta)|
\le
|z_j|+|z_\ell|
\le
2e^{-(t+\underline\alpha)\Delta}.
\]
故
\[
\operatorname{sep}(t;\Delta(t))
\le
2e^{-(t+\underline\alpha)\Delta(t)}
\le
e^{-c\,t\Delta(t)}
\]
（取任意 \(c\in(0,1)\)，当 \(t\) 足够大时成立）。
于是
\[
\operatorname{sep}(t;\Delta(t))^{-(\kappa-1)}
\ge
\exp\!\bigl(c(\kappa-1)t\Delta(t)\bigr),
\]
得到指数病态区。

再证 (2)。
由定理 \ref{thm:conclusion-stieltjes-prony-step-optimal-law}，
\[
\operatorname{sep}\!\left(t;\frac1t\right)
\ge
\frac{2g_0}{\pi e}\,e^{-\overline\alpha/t}\,t^{-1}
\gg t^{-1}.
\]
故
\[
\operatorname{sep}\!\left(t;\frac1t\right)^{-(\kappa-1)}
=
O\!\left(t^{\kappa-1}\right).
\]
证毕。
\end{proof}
