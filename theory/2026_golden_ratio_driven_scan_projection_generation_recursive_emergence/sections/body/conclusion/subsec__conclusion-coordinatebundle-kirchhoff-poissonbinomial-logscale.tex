\subsection{内部坐标束屏幕的 Kirchhoff 因子化、随机延迟审计与对数标度唯一性}\label{subsec:conclusion-coordinatebundle-kirchhoff-poissonbinomial-logscale}

沿用小节 \ref{subsec:conclusion-screen-kirchhoff-gibbs-whitney-rigidity}、
小节 \ref{subsec:conclusion-screen-hypercube-boundary-torsor-shapley} 与
定理 \ref{thm:spg-internal-coordinate-bundle-screen-cost-closed-form}
的记号。固定
\[
J\subseteq [n],
\qquad
J\neq\varnothing,
\qquad
s:=|J|,
\qquad
L_J:=2^{m(n-s)},
\]
并记内部坐标束屏幕为 \(S_J^{\mathrm{int}}\)。对每个
\[
a_{J^c}\in\{0,\dots,2^m-1\}^{J^c},
\]
记对应板层为 \(\Sigma_{a_{J^c}}\)，其外边界面集为
\[
E_{a_{J^c}}:=\partial_{\mathrm{out}}\Sigma_{a_{J^c}}.
\]
前一小节已经表明：内部坐标束屏幕的极小精确化在组合意义上恰等于“每条板层命中一次”的独立选择问题，而一般部分屏幕的极小精确化 Gibbs 系综则由 Kirchhoff 多项式与有效电阻控制。把这两组结构合并之后，还可进一步把内部坐标束屏幕压缩为一条完全可积的层分离链。

\begin{theorem}[内部坐标束屏幕极小精确化的 Kirchhoff 完全因子化]\label{thm:conclusion-coordinate-bundle-kirchhoff-complete-factorization}
记 \(\mathfrak E_J^{\min}\) 为全部按包含极小且使
\[
f_{S_J^{\mathrm{int}}\cup B}
\]
成为单射的外边界补全面集所成之族。定义其加权生成多项式
\[
\mathcal K_J(\mathbf x)
:=
\sum_{B\in \mathfrak E_J^{\min}}
\prod_{e\in B}x_e.
\]
则有严格因子化
\[
\mathcal K_J(\mathbf x)
=
\prod_{a_{J^c}\in\{0,\dots,2^m-1\}^{J^c}}
\left(
\sum_{e\in E_{a_{J^c}}}x_e
\right).
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-coordinate-bundle-minimal-outer-boundary-threshold}，
\[
f_{S_J^{\mathrm{int}}\cup B}\ \text{单射}
\iff
\forall a_{J^c},\quad B\cap E_{a_{J^c}}\neq\varnothing.
\]
再由同一推论的证明，一张外边界面只会命中唯一一条板层。故 \(B\) 按包含极小时，必且只必从每个 \(E_{a_{J^c}}\) 中各选取一张面。于是任一
\[
B\in \mathfrak E_J^{\min}
\]
都与一族独立选择
\[
e_{a_{J^c}}\in E_{a_{J^c}}
\qquad
\bigl(a_{J^c}\in\{0,\dots,2^m-1\}^{J^c}\bigr)
\]
一一对应，而对应单项式恰为
\[
\prod_{a_{J^c}}x_{e_{a_{J^c}}}.
\]
对全部独立选择求和便得到所示乘积公式。证毕。
\end{proof}

\begin{corollary}[极小精确化 Gibbs 系综的板层独立与有效电阻闭式]\label{cor:conclusion-coordinate-bundle-gibbs-resistance-closed-form}
设全部 \(x_e>0\)，并在 \(\mathfrak E_J^{\min}\) 上定义 Gibbs 测度
\[
\mu_{J,\mathbf x}(B)
:=
\frac{1}{\mathcal K_J(\mathbf x)}
\prod_{e\in B}x_e.
\]
则：
\begin{enumerate}
  \item 各板层上的选择彼此独立；
  \item 对任意 \(e\in E_{a_{J^c}}\)，有
  \[
  \mu_{J,\mathbf x}(e\in B)
  =
  \frac{x_e}{\sum_{f\in E_{a_{J^c}}}x_f};
  \]
  \item 压缩图 \(Q_{S_J^{\mathrm{int}}}\) 中，叶顶点 \(a_{J^c}\) 与外部顶点 \(\infty\) 之间的有效电阻满足
  \[
  R^{\mathrm{eff}}_{Q_{S_J^{\mathrm{int}}},\mathbf x}(\infty,a_{J^c})
  =
  \left(
  \sum_{f\in E_{a_{J^c}}}x_f
  \right)^{-1};
  \]
  \item 在无权情形 \(x_e\equiv 1\) 下，对任意 \(e\in E_{a_{J^c}}\) 都有
  \[
  \kappa_{S_J^{\mathrm{int}}}(e)=\frac{1}{|E_{a_{J^c}}|}.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-coordinate-bundle-kirchhoff-complete-factorization}，
\[
\mathcal K_J(\mathbf x)
=
\prod_{a_{J^c}}
\left(
\sum_{f\in E_{a_{J^c}}}x_f
\right).
\]
故 \(\mu_{J,\mathbf x}\) 直接分解为各板层上的乘积测度，每层恰选择一张面，且选择概率按线性权重归一化，于是前两条成立。

另一方面，在内部坐标束情形下，压缩图 \(Q_{S_J^{\mathrm{int}}}\) 是以 \(\infty\) 为中心、各板层为叶子的多重星图；顶点 \(a_{J^c}\) 与 \(\infty\) 之间的并联总导通率恰为
\[
\sum_{f\in E_{a_{J^c}}}x_f.
\]
故这两点之间的有效电阻正是其倒数，从而得到第三条。再由定理 \ref{thm:conclusion-screen-kirchhoff-gibbs-edge-occupation} 的无权情形，
\[
\kappa_{S_J^{\mathrm{int}}}(e)
=
\mu_{J,\mathbf 1}(e\in B)
=
\frac{1}{|E_{a_{J^c}}|},
\]
得到第四条。证毕。
\end{proof}

\begin{theorem}[随机延迟审计的独立 Poisson--binomial 残余律]\label{thm:conclusion-coordinate-bundle-random-audit-poisson-binomial}
设每张外边界面独立地以概率 \(p\in(0,1)\) 被后续审计揭示，并记随机补全面集为 \(B_p\)。定义随机残余歧义秩
\[
\Lambda_{J,p}
:=
\rank_{\ZZ}\ker f_{S_J^{\mathrm{int}}\cup B_p}.
\]
对每条板层记
\[
m_{a_{J^c}}:=|E_{a_{J^c}}|,
\qquad
\theta_{a_{J^c}}:=(1-p)^{m_{a_{J^c}}}.
\]
则存在相互独立的 Bernoulli 变量族
\[
\bigl(\xi_{a_{J^c}}\bigr)_{a_{J^c}\in\{0,\dots,2^m-1\}^{J^c}}
\]
满足
\[
\mathbb P\bigl(\xi_{a_{J^c}}=1\bigr)=\theta_{a_{J^c}},
\qquad
\Lambda_{J,p}=\sum_{a_{J^c}}\xi_{a_{J^c}}.
\]
从而
\[
\mathbb E[\Lambda_{J,p}]
=
\sum_{a_{J^c}}(1-p)^{m_{a_{J^c}}},
\]
\[
\operatorname{Var}(\Lambda_{J,p})
=
\sum_{a_{J^c}}(1-p)^{m_{a_{J^c}}}\Bigl(1-(1-p)^{m_{a_{J^c}}}\Bigr),
\]
并且其 ordinary generating function 为
\[
F_{J,p}(s)
:=
\mathbb E[s^{\Lambda_{J,p}}]
=
\prod_{a_{J^c}}
\Bigl((1-\theta_{a_{J^c}})+s\,\theta_{a_{J^c}}\Bigr).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-coordinate-bundle-outer-boundary-residual-rank}，
\[
\Lambda_{J,p}
=
L_J-|\mathcal T_J(B_p)|,
\]
故 \(\Lambda_{J,p}\) 正是未被 \(B_p\) 命中的板层数。对固定 \(a_{J^c}\)，其未命中事件等价于
\[
B_p\cap E_{a_{J^c}}=\varnothing,
\]
而这要求该层全部 \(m_{a_{J^c}}\) 张外边界面都未被揭示，因此概率恰为
\[
(1-p)^{m_{a_{J^c}}}=\theta_{a_{J^c}}.
\]
由于不同板层对应的外边界面集彼此不交，这些未命中事件相互独立。于是 \(\Lambda_{J,p}\) 正是上述独立 Bernoulli 变量之和，故均值、方差与 ordinary generating function 公式同时成立。证毕。
\end{proof}

\begin{theorem}[对数标度的唯一无交互性]\label{thm:conclusion-coordinate-bundle-logscale-unique-nointeraction}
记 dyadic 网格
\[
\mathcal D_{m,n}:=\{2^{m\ell}:0\le \ell\le n\}.
\]
对任意函数 \(\phi:\mathcal D_{m,n}\to\RR\)，定义布尔格上的集合函数
\[
h_\phi(J):=\phi(\Delta_m(J)),
\qquad
\Delta_m(J)=2^{m(n-|J|)}.
\]
再记其 Möbius 反演为
\[
\widehat h_\phi(T)
:=
\sum_{U\subseteq T}(-1)^{|T|-|U|}h_\phi(U).
\]
则下列命题等价：
\begin{enumerate}
  \item 对所有 \(|T|\ge 2\)，都有 \(\widehat h_\phi(T)=0\)；
  \item 存在常数 \(A,B\in\RR\)，使对所有 \(0\le s\le n\) 都有
  \[
  \phi(2^{m(n-s)})=A+Bs;
  \]
  \item 存在常数 \(A',C\in\RR\)，使
  \[
  \phi(t)=A'+C\log_2 t
  \qquad
  (t\in \mathcal D_{m,n}).
  \]
\end{enumerate}
因此，在内部坐标束屏幕的隐藏体积标度上，能够把全部高阶 Möbius 交互同时消去的标度，除仿射变换外只有对数标度。
\end{theorem}
\begin{proof}
记
\[
u_s:=\phi(2^{m(n-s)}),
\qquad
0\le s\le n.
\]
则对任意 \(J\subseteq [n]\) 都有 \(h_\phi(J)=u_{|J|}\)。于是当 \(|T|=k\) 时，
\[
\widehat h_\phi(T)
=
\sum_{j=0}^{k}(-1)^{k-j}\binom{k}{j}u_j,
\]
这正是序列 \((u_s)\) 的 \(k\) 阶有限差分。故所有二阶及以上 Möbius 原子同时消失，当且仅当
\[
u_{s+2}-2u_{s+1}+u_s=0
\qquad
(0\le s\le n-2),
\]
亦即 \((u_s)\) 为等差数列，这就得到 \((1)\Longleftrightarrow(2)\)。

若 \(u_s=A+Bs\)，则对 \(t=2^{m(n-s)}\) 有
\[
s=n-\frac{1}{m}\log_2 t,
\]
从而
\[
\phi(t)=A+Bn-\frac{B}{m}\log_2 t.
\]
这正是 \((3)\)。反过来，任意 \((3)\) 形式在 \(\mathcal D_{m,n}\) 上代回后都给出关于 \(s\) 的等差数列，故 \((3)\Longleftrightarrow(2)\)。证毕。
\end{proof}

\noindent
因此，内部坐标束屏幕在一般 Kirchhoff 审计几何之内还保留着一条更强的可积子链：极小精确化多项式完全分裂为逐板层的一次因子；相应 Gibbs 系综退化为各层独立的单次抽样，并把有效电阻、边占用与无权关键度全部压成单层并联和的闭式；若外边界只以独立随机延迟方式被揭示，则残余歧义秩不再需要 Tutte/FK 型全局配分函数，而直接成为一条 Poisson--binomial 和；最后，隐藏体积的全部高阶协同只有在对数标度下才会同时消失。换言之，内部坐标束屏幕并不是一般部分屏幕理论中的一个普通特例，而是其中唯一把 Kirchhoff、概率、信息与 Möbius 结构同时压到逐层独立形态的完全可积分支。
