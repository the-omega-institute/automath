\begin{proposition}[容量曲线对截面账本的积分表示]\label{prop:conclusion-section-ledger-capacity-functional}
沿用定理 \ref{thm:conclusion-section-ledger-kl-identity} 的记号。
定义连续容量曲线
\[
C^{\mathrm{cont}}(T):=\sum_{x\in X}\min(d(x),T),
\qquad T\ge 0.
\]
则有严格恒等式
\[
\sum_{x\in X}\log d(x)
\ =\
-\,n\ +\ \int_{1}^{\infty}\frac{C^{\mathrm{cont}}(T)}{T^{2}}\,dT.
\]
从而
\[
\log|\mathcal S(F)|
\ =\
-\,n\ +\ \int_{1}^{\infty}\frac{C^{\mathrm{cont}}(T)}{T^{2}}\,dT.
\]
\end{proposition}

\begin{proof}
对 \(T\ge 0\) 定义尾部计数
\[
N(T):=\#\{x\in X:\ d(x)\ge T\}.
\]
对任意整数 \(d\ge 1\) 有
\[
\log d=\int_{1}^{d}\frac{dT}{T},
\]
于是
\[
\sum_{x\in X}\log d(x)
=\sum_{x\in X}\int_{1}^{d(x)}\frac{dT}{T}
=\int_{1}^{\infty}\frac{N(T)}{T}\,dT,
\]
其中交换求和与积分由于有限和成立。
\par
另一方面，\(C^{\mathrm{cont}}\) 为分段 \(C^1\) 函数，且几乎处处满足
\[
\frac{d}{dT}C^{\mathrm{cont}}(T)=N(T),
\]
因为每一项 \(\min(d(x),T)\) 的导数为 \(\mathbf 1_{T\le d(x)}\)。
因此分部积分得到
\[
\int_{1}^{\infty}\frac{N(T)}{T}\,dT
=\int_{1}^{\infty}\frac{(C^{\mathrm{cont}})'(T)}{T}\,dT
=\Bigl[\frac{C^{\mathrm{cont}}(T)}{T}\Bigr]_{1}^{\infty}+\int_{1}^{\infty}\frac{C^{\mathrm{cont}}(T)}{T^{2}}\,dT.
\]
当 \(T\ge \max_x d(x)\) 时 \(C^{\mathrm{cont}}(T)=\sum_x d(x)=N\) 为常数，故 \(\lim_{T\to\infty}C^{\mathrm{cont}}(T)/T=0\)；
并且 \(C^{\mathrm{cont}}(1)=\sum_x 1=n\)。
代入即得
\(
\sum_x\log d(x)=-n+\int_1^\infty C^{\mathrm{cont}}(T)T^{-2}\,dT
\)。
证毕。
\end{proof}
