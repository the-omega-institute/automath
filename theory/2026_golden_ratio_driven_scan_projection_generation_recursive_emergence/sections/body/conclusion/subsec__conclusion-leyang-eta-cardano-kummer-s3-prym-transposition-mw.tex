\subsection{Lee--Yang 的 \texorpdfstring{$\eta$}{eta} Cardano--Kummer 正规形、显式 \texorpdfstring{$S_3$}{S3} 作用与 Prym 识别}\label{subsec:conclusion-leyang-eta-cardano-kummer-s3-prym-transposition-mw}
沿用子节 \ref{subsec:conclusion-leyang-eta-cubic-discriminant-s3-galois-jacobian-lfactor} 的记号，特别是
\(\mathscr D(\eta)\)、\(E_\Delta\)、\(H_{\mathrm{LY}}\) 与 \(\widetilde H_\eta\) 的定义。

\begin{theorem}[\texorpdfstring{\(\widetilde H_\eta\)}{Htilde eta} 的 Cardano--Kummer 正规形]\label{thm:conclusion-leyang-eta-cardano-kummer-normal-form}
定义
\[
Q_4(\eta):=512\eta^4-2240\eta^3+3465\eta^2-2240\eta+512,
\]
并考虑曲线
\begin{equation}\label{eq:conclusion-leyang-eta-cardano-kummer-curve}
\mathcal C:\quad
\begin{cases}
v^2=\mathscr D(\eta),\\[1mm]
W^3=\eta\bigl(3Q_4(\eta)+32(\eta-1)^2v\bigr).
\end{cases}
\end{equation}
则 \(\mathcal C\) 的射影正规化在 \(K\) 上光滑，且与定理
\ref{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice}
中的 \(S_3\)-闭包曲线 \(\widetilde H_\eta\) 在 \(K\) 上双有理等价。其函数域层塔为
\[
K(\eta)\subset K(\eta,v)\subset K(\eta,v,W),
\]
其中 \(K(\eta,v)\) 是判别式对应的二次子域，而 \(K(\eta,v,W)\) 是
\eqref{eq:conclusion-leyang-eta-cubic-star} 的分裂域。
\end{theorem}
\begin{proof}
从 \eqref{eq:conclusion-leyang-eta-cubic-star} 出发，将三次方程除以首项系数 \(64(\eta-1)^2\) 并写作
\[
y^3+ay^2+by+c=0,\qquad
a=\frac{3(32\eta^2-73\eta+32)}{64(\eta-1)^2}.
\]
作 Tschirnhaus 变换 \(y=x-\frac a3\)，得到压低三次
\[
x^3+px+q=0
\]
且
\[
p=\frac{27\eta(128\eta^2-265\eta+128)}{4096(\eta-1)^4},\qquad
q=-\frac{81\eta\,Q_4(\eta)}{131072(\eta-1)^6}.
\]
Cardano 判别式满足
\[
\Delta_{\mathrm{Car}}
=\left(\frac q2\right)^2+\left(\frac p3\right)^3
=\frac{729\,\eta^2\,\mathscr D(\eta)}{67108864(\eta-1)^8},
\]
故在 \(K(\eta,v)\) 中
\[
\sqrt{\Delta_{\mathrm{Car}}}
=\frac{27\eta\,v}{8192(\eta-1)^4}.
\]
令 \(u^3:=-\frac q2+\sqrt{\Delta_{\mathrm{Car}}}\)，则
\[
u^3
=\frac{27\eta}{2^{18}(\eta-1)^6}
\Bigl(3Q_4(\eta)+32(\eta-1)^2v\Bigr).
\]
再令
\[
W:=\frac{2^6(\eta-1)^2}{3}\,u,
\]
即得
\[
W^3=\eta\bigl(3Q_4(\eta)+32(\eta-1)^2v\bigr).
\]
于是得到模型 \eqref{eq:conclusion-leyang-eta-cardano-kummer-curve}。
因 \(K=\QQ(\sqrt{-3})\) 含 \(\zeta_3\)，三次方程分裂域由
\(u\)（等价于 \(W\)）与判别式平方根 \(v\) 生成，
即 \(K(\eta,v,W)\)。
该分裂域与定理
\ref{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice}
给出的 \(S_3\)-闭包一致，故 \(\mathcal C\) 与 \(\widetilde H_\eta\) 双有理等价。证毕。
\end{proof}

\begin{theorem}[二次共轭范数的立方坍缩恒等式]\label{thm:conclusion-leyang-eta-theta-norm-cubic-collapse}
\leanverified{paper\_conclusion\_leyang\_eta\_theta\_norm\_cubic\_collapse}
定义
\[
B(\eta):=128\eta^2-265\eta+128,\qquad
\Theta:=3Q_4(\eta)+32(\eta-1)^2v\in K(E_\Delta)^\times.
\]
记 \(E_\Delta\) 上双曲 involution 为
\(
\iota:(\eta,v)\mapsto(\eta,-v)
\)。
则恒有
\begin{equation}\label{eq:conclusion-leyang-eta-theta-norm-collapse}
\Theta\cdot\iota(\Theta)=-\eta\,B(\eta)^3.
\end{equation}
等价地，
\[
\operatorname{Norm}_{K(E_\Delta)/K(\eta)}(\Theta)
=-\eta\,B(\eta)^3.
\]
\end{theorem}
\begin{proof}
由定义
\[
\Theta\cdot\iota(\Theta)
=\bigl(3Q_4(\eta)\bigr)^2-\bigl(32(\eta-1)^2v\bigr)^2
=9Q_4(\eta)^2-1024(\eta-1)^4v^2.
\]
代入 \(v^2=\mathscr D(\eta)\) 并在 \(K[\eta]\) 中直接展开化简，得到
\[
9Q_4(\eta)^2-1024(\eta-1)^4\mathscr D(\eta)
=-\eta\,B(\eta)^3,
\]
即 \eqref{eq:conclusion-leyang-eta-theta-norm-collapse}。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{\(\mathcal C\)}{C} 上的显式 \texorpdfstring{\(S_3\)}{S3} 作用]\label{thm:conclusion-leyang-eta-explicit-s3-action-on-cardano-model}
在 \(\mathcal C\) 的函数域上定义
\[
\tau:(\eta,v,W)\longmapsto(\eta,v,\zeta_3W),
\]
\[
\sigma:(\eta,v,W)\longmapsto
\Bigl(\eta,-v,-\frac{\eta B(\eta)}{W}\Bigr).
\]
则 \(\tau\) 为三阶、\(\sigma\) 为二阶，并满足
\[
\sigma\tau\sigma=\tau^{-1}.
\]
因而 \(\langle\tau,\sigma\rangle\cong S_3\)，且
\[
\mathcal C/\langle\tau\rangle\cong E_\Delta,\qquad
\mathcal C/S_3\cong\PP^1_\eta.
\]
\end{theorem}
\begin{proof}
\(\tau\) 保持 \(v^2=\mathscr D(\eta)\)，并且
\((\zeta_3W)^3=W^3\)，故为三阶自同构。

对 \(\sigma\)，第一方程在 \(v\mapsto -v\) 下不变。记
\(
\Theta=3Q_4(\eta)+32(\eta-1)^2v
\)。
由定理 \ref{thm:conclusion-leyang-eta-theta-norm-cubic-collapse}，
\[
\iota(\Theta)=3Q_4(\eta)-32(\eta-1)^2v
=-\frac{\eta B(\eta)^3}{\Theta}.
\]
故
\[
\sigma(W)^3
=\left(-\frac{\eta B(\eta)}{W}\right)^3
=-\frac{\eta^3B(\eta)^3}{W^3}
=-\frac{\eta^2B(\eta)^3}{\Theta}
=\eta\,\iota(\Theta),
\]
正是 \(v\mapsto -v\) 后右端应取之值，故 \(\sigma\) 保持方程。
再由
\[
\sigma^2(W)
=-\frac{\eta B(\eta)}{-\eta B(\eta)/W}
=W
\]
得 \(\sigma^2=\mathrm{id}\)。

最后计算
\[
\sigma\tau\sigma(W)
=\sigma\tau\!\left(-\frac{\eta B(\eta)}{W}\right)
=\sigma\!\left(-\frac{\eta B(\eta)}{\zeta_3W}\right)
=\zeta_3^{-1}W
=\tau^{-1}(W),
\]
并且 \(\sigma\tau\sigma\) 固定 \(\eta,v\)，故关系成立，群同构于 \(S_3\)。
商曲线结论由生成元定义与 Galois 对应直接得到。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{\(\mathcal C\to E_\Delta\)}{C to EDelta} 的全分歧结构与属数]\label{thm:conclusion-leyang-eta-kummer-cover-ramification-genus}
设
\[
\pi:\mathcal C\longrightarrow E_\Delta,\qquad
(\eta,v,W)\longmapsto(\eta,v).
\]
则 \(\pi\) 为循环三次覆盖，且分歧恰发生在四点
\[
R_\pm=(0,\pm48),\qquad O_\pm\in E_\Delta(\overline K)\ \text{（无穷远两点）},
\]
并且四点上皆为全分歧（分歧指数 \(e=3\)）。因此
\[
g(\mathcal C)=5.
\]
\end{theorem}
\begin{proof}
令
\[
f:=\eta\Theta,\qquad
\Theta=3Q_4(\eta)+32(\eta-1)^2v,
\]
则 \(\mathcal C\) 由 Kummer 方程 \(W^3=f\) 给出。
Kummer 判据表明：在 \(P\in E_\Delta\) 处，
\(
\ord_P(f)\not\equiv 0\pmod3
\)
当且仅当 \(\pi\) 在 \(P\) 上分歧，且此时 \(e_P=3\)。

在 \(R_+\) 处，\(\ord_{R_+}(\eta)=1\) 且
\(\Theta(R_+)=3072\neq0\)，故 \(\ord_{R_+}(f)=1\)。
在 \(R_-\) 处，\(\Theta(R_-)=0\)，并由
\eqref{eq:conclusion-leyang-eta-theta-norm-collapse}
与 \(\iota(R_+)=R_-\) 得 \(\ord_{R_-}(\Theta)=1\)，故
\(\ord_{R_-}(f)=2\)。

在无穷远点，取局部参数 \(t=\eta^{-1}\)。
由 \(v^2=\mathscr D(\eta)\) 得两支展开
\[
v=48t^{-2}+O(t^{-1})\quad(\text{于 }O_+),\qquad
v=-48t^{-2}+O(t^{-1})\quad(\text{于 }O_-).
\]
代入 \(\Theta\) 可得
\[
\ord_{O_+}(\Theta)=-4,\qquad
\ord_{O_-}(\Theta)=-3.
\]
又 \(\ord_{O_\pm}(\eta)=-1\)，故
\[
\ord_{O_+}(f)=-5,\qquad
\ord_{O_-}(f)=-4,
\]
均不被 \(3\) 整除，因而 \(O_\pm\) 亦全分歧。

于是四点共贡献
\(
4\times(e-1)=8
\)。
另一方面，由定理
\ref{thm:conclusion-leyang-eta-cardano-kummer-normal-form}
与定理
\ref{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice}
知 \(g(\mathcal C)=g(\widetilde H_\eta)=5\)。
将 \(g(E_\Delta)=1\)、\(\deg\pi=3\) 代入 Hurwitz 公式
\[
2g(\mathcal C)-2=3(2g(E_\Delta)-2)+\sum_{P}(e_P-1)
\]
得到总分歧度必须为 \(8\)，与上述四点贡献完全一致，故无其余分歧点。证毕。
\end{proof}

\begin{theorem}[Prym 识别与极化型]\label{thm:conclusion-leyang-eta-prym-identification-polarization-type}
设 \(\pi:\mathcal C\to E_\Delta\) 如定理
\ref{thm:conclusion-leyang-eta-kummer-cover-ramification-genus}，
并记
\[
\Prym(\pi):=\ker\!\bigl(\operatorname{Nm}_\pi:\Jac(\mathcal C)\to \Jac(E_\Delta)\bigr)^0.
\]
则：
\begin{enumerate}
  \item \(\dim \Prym(\pi)=4\)；
  \item 存在 \(K\)-同源
  \[
  \Prym(\pi)\sim_K \Jac(H_{\mathrm{LY}})^2;
  \]
  \item 由 \(\Jac(\mathcal C)\) 主极化诱导于 \(\Prym(\pi)\) 的极化型为
  \[
  (1,1,3,3).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
第一点由维数公式
\[
\dim \Prym(\pi)=g(\mathcal C)-g(E_\Delta)=5-1=4
\]
立即得到。

对含 \(\zeta_3\) 的底域上的循环三次覆盖，
\(\Jac(\mathcal C)\) 在 \(K\)-同源意义下分解为
\[
\Jac(\mathcal C)\sim_K E_\Delta\times \Prym(\pi).
\]
另一方面，由定理
\ref{thm:conclusion-leyang-eta-cardano-kummer-normal-form}
可将 \(\mathcal C\) 视作 \(\widetilde H_\eta\) 的模型，
并由命题
\ref{prop:conclusion-leyang-eta-s3-jacobian-isogeny-decomposition}
得到
\[
\Jac(\mathcal C)\sim_K \Jac(H_{\mathrm{LY}})^2\times E_\Delta.
\]
比较两式即得
\(
\Prym(\pi)\sim_K\Jac(H_{\mathrm{LY}})^2
\)。

最后，对素数阶 \(3\) 的循环分歧覆盖，
当基曲线为属 \(1\) 且总分歧点数为 \(4\) 时，
Prym 上诱导极化型由经典循环覆盖公式给出为
\((1,1,3,3)\)。证毕。
\end{proof}

\begin{theorem}[transposition 商的显式生成与立方二次消元分解]\label{thm:conclusion-leyang-eta-transposition-quotient-explicit-generator}
在定理 \ref{thm:conclusion-leyang-eta-explicit-s3-action-on-cardano-model}
的 \(\sigma\) 作用下，定义
\[
y=\frac{3\left(W-\dfrac{\eta B(\eta)}{W}\right)-(32\eta^2-73\eta+32)}
{64(\eta-1)^2}\in K(\mathcal C).
\]
则：
\begin{enumerate}
  \item \(y\) 为 \(\sigma\)-不变量，且
  \[
  K(\mathcal C)^{\langle\sigma\rangle}=K(\eta,y);
  \]
  \item 消去 \((v,W)\) 后，\((\eta,y)\) 满足的关系在去除重因子与垂直分量后可写为
  \[
  \eta^2(\eta-1)^2B(\eta)^3\,
  \Phi(y,\eta)=0,
  \]
  其中
  \[
  \Phi(y,\eta)
  =64(\eta-1)^2y^3+3(32\eta^2-73\eta+32)y^2
  +3(16\eta^2-23\eta+16)y+8(\eta-1)^2.
  \]
  从而 \(\mathcal C/\langle\sigma\rangle\) 的规范化由
  \(\Phi(y,\eta)=0\) 给出，并与 \(H_{\mathrm{LY}}\) 同构（故属为 \(2\)）。
\end{enumerate}
\end{theorem}
\begin{proof}
由
\(
\sigma(W)=-\eta B(\eta)/W
\)
可见
\(
W-\eta B(\eta)/W
\)
为 \(\sigma\)-不变，故 \(y\) 不变。
将定义式改写为
\begin{equation}\label{eq:conclusion-leyang-eta-transposition-quadratic-W}
3W^2-\bigl(64(\eta-1)^2y+32\eta^2-73\eta+32\bigr)W-3\eta B(\eta)=0,
\end{equation}
故 \(W\) 在 \(K(\eta,y)\) 上至多二次，因而
\(
[K(\mathcal C):K(\eta,y)]\le 2
\)。

又由定理
\ref{thm:conclusion-leyang-eta-cardano-kummer-normal-form}
有 \([K(\mathcal C):K(\eta)]=6\)，故
\(
[K(\eta,y):K(\eta)]\ge 3
\)。
另一方面，将
\[
v^2-\mathscr D(\eta)=0,\quad
W^3-\eta\bigl(3Q_4(\eta)+32(\eta-1)^2v\bigr)=0,\quad
\eqref{eq:conclusion-leyang-eta-transposition-quadratic-W}
\]
作消元，得到（差一个非零常数）
\[
\operatorname{Res}_{v,W}
=\eta^3(\eta-1)^8B(\eta)^3\Phi(y,\eta)^2.
\]
去重并去除 \(\eta=0,1,\infty\) 及 \(B(\eta)=0\) 的垂直分量后，
得到声明中的消元分解。
因此 \(y\) 满足三次关系 \(\Phi(y,\eta)=0\)，从而
\(
[K(\eta,y):K(\eta)]=3
\)；
进而
\(
[K(\mathcal C):K(\eta,y)]=2
\)。
由于 \(K(\eta,y)\subseteq K(\mathcal C)^{\langle\sigma\rangle}\) 且两者对
\(K(\eta)\) 次数同为 \(3\)，故二者相等。最后，定理
\ref{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice}
已给出任一 transposition 商与 \(H_{\mathrm{LY}}\) 同构，故其属为 \(2\)。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{\(E_\Delta(\QQ)\)}{EDelta(Q)} 的挠子结构与正秩性]\label{prop:conclusion-leyang-eta-edelta-mw-torsion-rank}
在定理
\ref{thm:conclusion-leyang-eta-discriminant-elliptic-j-factorization}
的 \(\QQ\)-Weierstrass 模型
\[
E_\Delta:\quad Y^2=X^3+16437789X+40904021790
\]
上，有：
\begin{enumerate}
  \item 挠子子群
  \[
  E_\Delta(\QQ)_{\mathrm{tors}}\cong \ZZ/2\ZZ,
  \]
  其非平凡挠点可取
  \[
  T=(-2001,0);
  \]
  \item 点
  \[
  P=(303,214272)\in E_\Delta(\QQ)
  \]
  为无限阶，故
  \[
  \operatorname{rank}E_\Delta(\QQ)\ge 1.
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
首先 \(T=(-2001,0)\) 代入方程成立，且 \(Y(T)=0\) 蕴含 \(2T=\mathcal O\)，故存在非平凡 \(2\)-挠点。

由定理
\ref{thm:conclusion-leyang-eta-discriminant-elliptic-j-factorization}
中的判别式分解可知，\(p=101,103\) 为良好约化素数。
直接计数得到
\[
\#E_\Delta(\mathbf F_{101})=106,\qquad
\#E_\Delta(\mathbf F_{103})=116.
\]
良好约化下挠子约化单射，故
\[
\#E_\Delta(\QQ)_{\mathrm{tors}}
\mid \gcd(106,116)=2.
\]
结合已存在的非平凡 \(2\)-挠点，得到
\(
E_\Delta(\QQ)_{\mathrm{tors}}\cong\ZZ/2\ZZ
\)。

再看点 \(P\)。其约化阶满足
\[
\operatorname{ord}_{101}(\overline P)=53,\qquad
\operatorname{ord}_{103}(\overline P)=58.
\]
若 \(P\) 为有理挠点，其阶必须同时整除 \(53\) 与 \(58\)，不可能成立。
故 \(P\) 为无限阶点，从而 \(\operatorname{rank}E_\Delta(\QQ)\ge1\)。证毕。
\end{proof}
