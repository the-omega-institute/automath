\begin{theorem}[arity-charge 热力学的共边界规约刚性]\label{thm:conclusion-realinput40-arity-coboundary-reduction-rigidity}
设 \(M(q)\) 为 essential \(20\) 状态核心上每条边赋权 \(q^{\chi(e)}\) 的加权邻接矩阵。则存在势函数
\[
V:\Sigma_{\mathrm{ess}}\to\{0,1\}
\]
与逐边 \(0/1\) 标记
\[
g(e):=\chi(e)+V(v)-V(u)\in\{0,1\}\qquad (e:u\to v),
\]
使得对任意 \(q>0\)，都有精确相似关系
\[
M(q)=D(q)\,(M_0+qM_1)\,D(q)^{-1},
\qquad
D(q):=\operatorname{diag}(q^{V(u)})_{u\in\Sigma_{\mathrm{ess}}},
\]
其中
\[
(M_b)_{uv}:=\#\{e:u\to v\ \text{且}\ g(e)=b\},
\qquad b\in\{0,1\}.
\]
从而
\[
\det(I-zM(q))=\det(I-z(M_0+qM_1)),
\]
\[
\rho(M(q))=\rho(M_0+qM_1),
\qquad
\operatorname{Tr}(M(q)^n)=\operatorname{Tr}((M_0+qM_1)^n)\quad(\forall n\ge 1).
\]
进一步，若记
\[
c_n(q):=\operatorname{Tr}(M(q)^n),
\]
则 primitive 计数多项式满足加权 Möbius 反演
\[
c_n(q)=\sum_{d\mid n} d\,P_d(q^{n/d}),
\qquad
P_n(q)=\frac{1}{n}\sum_{d\mid n}\mu(d)\,c_{n/d}(q^d).
\]
因此全部 primitive 多项式 \(P_n(q)\)、压力函数
\[
P_\chi(\theta):=\log\rho(M(e^\theta)),
\]
以及其 Legendre 前沿，都只依赖于二值边拆分 \((M_0,M_1)\)，而与所选规约势 \(V\) 无关。
\end{theorem}
\begin{proof}
定理 \ref{thm:real-input-40-arity-charge-coboundary} 已给出共边界正规形
\[
\chi(e)=g(e)-V(v)+V(u),
\qquad
g(e)\in\{0,1\},
\qquad
V\in\{0,1\}.
\]
命题 \ref{prop:real-input-40-arity-charge-similarity-reduction} 进一步证明
\[
M(q)=D(q)\,(M_0+qM_1)\,D(q)^{-1},
\]
并导出行列式、谱半径与迹在该相似变换下的不变性。于是前三组盒装公式全部成立。

再由 Fredholm 行列式的标准 Witt--Möbius 关系，
\[
\log\det(I-zM(q))^{-1}
=
\sum_{n\ge 1}\frac{c_n(q)}{n}z^n
=
\sum_{n\ge 1}\sum_{d\mid n}\frac{d\,P_d(q^{n/d})}{n}z^n,
\]
逐项比较即可得到
\[
c_n(q)=\sum_{d\mid n} d\,P_d(q^{n/d}),
\qquad
P_n(q)=\frac{1}{n}\sum_{d\mid n}\mu(d)\,c_{n/d}(q^d).
\]
由于 \(c_n(q)\) 与 \(\rho(M(q))\) 都已由 \((M_0,M_1)\) 唯一决定，primitive 多项式、压力函数与其 Legendre 对偶也都只依赖于该二值边拆分，而与规约势 \(V\) 的具体选取无关。证毕。
\end{proof}

\begin{corollary}[arity-charge 热力学是纯二值边拆分不变量]\label{cor:conclusion-realinput40-arity-binary-edge-split-invariant}
arity-charge 的热力学并不依赖于“选取哪一个共边界规约势”；它完全压缩为二值边拆分 \((M_0,M_1)\) 的不变量。
\end{corollary}
\begin{proof}
这是定理 \ref{thm:conclusion-realinput40-arity-coboundary-reduction-rigidity} 的直接重述。证毕。
\end{proof}

\begin{theorem}[primitive arity 平均斜率的硬墙完备性]\label{thm:conclusion-realinput40-arity-hardwall-completeness}
\leanverified{paper\_conclusion\_realinput40\_arity\_hardwall\_completeness}
对 essential \(20\) 状态核心上的任意 primitive 周期轨道 \(\gamma\)，其平均 arity-charge 满足严格区间约束
\[
0\le \frac{\chi(\gamma)}{|\gamma|}\le \frac12.
\]
并且两端点同时可达：
\begin{enumerate}
  \item 存在 primitive 轨道使 \(\chi(\gamma)=0\)，因而
  \[
  \inf_\gamma \frac{\chi(\gamma)}{|\gamma|}=0;
  \]
  \item 存在长度 \(2\) 的 primitive 轨道使 \(\chi(\gamma)=1\)，因而
  \[
  \sup_\gamma \frac{\chi(\gamma)}{|\gamma|}=\frac12.
  \]
\end{enumerate}
于是 primitive arity 统计的极端支持斜率恰为闭区间
\[
\left[0,\frac12\right].
\]
特别地，对每个 \(n\ge 1\)，
\[
P_n(q)\in \ZZ_{\ge 0}[q],
\qquad
\deg P_n(q)\le \Bigl\lfloor\frac{n}{2}\Bigr\rfloor.
\]
\end{theorem}
\begin{proof}
下界
\[
\chi(\gamma)\ge 0
\]
来自定理 \ref{thm:real-input-40-arity-charge-coboundary}。另一方面，定理 \ref{thm:real-input-40-arity-charge-density-bound} 已给出
\[
\chi(\gamma)\le \left\lfloor\frac{|\gamma|}{2}\right\rfloor,
\]
并且存在长度 \(2\) 的 primitive 轨道实现 \(\chi(\gamma)=1\)。于是
\[
0\le \frac{\chi(\gamma)}{|\gamma|}\le \frac12
\]
且右端点可达。

左端点可达则由定理 \ref{thm:real-input-40-arity-charge-coboundary} 所诱导的零荷 primitive 轨道存在性给出；等价地，零荷子系统的 \(\zeta\) 非平凡，故存在 primitive \(\gamma\) 使 \(\chi(\gamma)=0\)。因此极端支持斜率正是 \([0,\frac12]\)。

最后，由推论 \ref{cor:real-input-40-arity-charge-no-laurent}，
\[
P_n(q)\in \ZZ_{\ge 0}[q],
\]
而推论 \ref{cor:real-input-40-arity-charge-degree-bound} 给出
\[
\deg P_n(q)\le \left\lfloor\frac{n}{2}\right\rfloor.
\]
证毕。
\end{proof}

\begin{corollary}[primitive arity 大偏差支持的硬墙已在周期层实现]\label{cor:conclusion-realinput40-arity-hardwalls-realized}
arity-charge 大偏差的左右硬墙
\[
0,\qquad \frac12
\]
并不是解析延拓所猜测的边界，而是在 primitive 周期层已经被完全实现的极端斜率。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-realinput40-arity-hardwall-completeness}，两端点都由实际 primitive 轨道达到。证毕。
\end{proof}

\begin{theorem}[有限路径负荷的单单位边界债]\label{thm:conclusion-realinput40-arity-single-unit-boundary-debt}
对 essential \(20\) 状态核心上的任意有限允许路径 \(\pi:u\leadsto v\)，都有
\[
\sum_{e\in\pi}\chi(e)\ge -1.
\]
更精确地，
\[
\chi(\pi)=\sum_{e\in\pi}g(e)+V(u)-V(v),
\qquad
g(e)\in\{0,1\},
\qquad
V\in\{0,1\}.
\]
因此：
\begin{enumerate}
  \item 路径级别的一切负 charge 都是纯边界债；
  \item 闭轨情形中该债自动消失，故 \(\chi(\gamma)\ge 0\)；
  \item 任何试图通过 bulk 内部负贡献制造 Laurent 负幂的机制，在该核上都被严格禁止。
\end{enumerate}
\end{theorem}
\begin{proof}
这正是命题 \ref{prop:real-input-40-arity-charge-prefix-debt-structure} 的路径分解公式。由
\[
\chi(\pi)=\sum_{e\in\pi}g(e)+V(u)-V(v),
\]
且 \(\sum g(e)\ge 0\)、\(V(u)-V(v)\ge -1\)，立得
\[
\chi(\pi)\ge -1.
\]
因此一切负贡献都只能来自端点差 \(V(u)-V(v)\)，即纯边界债。若 \(\pi\) 为闭轨，则 \(u=v\)，边界项自动消失，从而
\[
\chi(\gamma)=\sum_{e\in\gamma}g(e)\ge 0.
\]
于是 bulk 内部不存在真正的负 charge 机制，也就不可能制造 Laurent 负幂。证毕。
\end{proof}

\begin{corollary}[arity-charge 的非负性是深度一的边界债模型]\label{cor:conclusion-realinput40-arity-boundary-debt-model}
从路径语义看，arity-charge 的非负性并非全局巧合，而是一个深度至多一单位的边界债模型：闭轨时债务自动清零，于是只剩纯正的 \(0/1\) 计数。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-realinput40-arity-single-unit-boundary-debt}，所有负荷都只能来自 \(V(u)-V(v)\in\{-1,0,1\}\) 的端点差；闭轨时该项为零。证毕。
\end{proof}

\begin{conjecture}[稳定骨架--瞬态塔--边界债的三层分裂]\label{conj:conclusion-realinput40-k-jordan-boundarydebt-triple-split}
设一类有限同步核同时满足：
\begin{enumerate}
  \item 存在 essential 核 \(A\) 与 full 核 \(M\)，二者共享同一 Bowen--Franks 群；
  \item full 核具有最短全局同步词，其长度恰等于零谱最大 Jordan 深度；
  \item 某个闭轨观测量 \(\chi\) 可规约为
  \[
  \chi(e)=g(e)-V(v)+V(u),
  \qquad
  g(e)\in\{0,1\},
  \qquad
  V\in\{0,1\}.
  \]
\end{enumerate}
则应存在一个普遍三层分裂：
\[
\boxed{\text{稳定 }K\text{-骨架}}
\quad\oplus\quad
\boxed{\text{同步前瞬态 Jordan 塔}}
\quad\oplus\quad
\boxed{\text{深度 }1\text{ 的边界债 charge}}.
\]
其中第一层由 Smith/Bowen--Franks 不变量承载，第二层由同步长度与零谱幂零指数控制，第三层由共边界规约势 \(V\) 的端点跳变承载。真实输入 \(40\) 状态核给出该分裂的首个完整样本，其三组参数分别为
\[
\ZZ^2\oplus(\ZZ/2\ZZ),\qquad (3,5),\qquad 1.
\]
\end{conjecture}
