\begin{theorem}[时间视窗的渐近满相位律]\label{thm:conclusion-time-window-asymptotically-full-phase-law}
\leanverified{paper\_conclusion\_time\_window\_asymptotically\_full\_phase\_law}
设时间视窗群
\[
G_T:=\ZZ^T,
\]
并定义 \(b\)-位窗口对象
\[
W_b(G_T):=(\ZZ/2^b\ZZ)^T,
\qquad
|W_b(G_T)|=2^{bT}.
\]
若存在可逆编码
\[
E_{T,b}:W_b(G_T)\hookrightarrow \mathrm{Phase}_b(k)\times R_{T,b},
\qquad
|\mathrm{Phase}_b(k)|=2^{bk},
\]
则必有
\[
|R_{T,b}|\ge 2^{(T-k)b}.
\]
特别地，若 \(|R_{T,b}|\le T^C\)，则
\[
k\ge T-\frac{C}{b}\log_2 T,
\]
而当 \(b\gtrsim \log T\) 时更强地得到
\[
k\ge T-O(1).
\]
因此，只要要求时间视窗统一可逆，可见相位维数就必须渐近饱和到全部时间自由度；任何低于此的相位压缩都会以指数规模的残差账本回流。
\end{theorem}
\begin{proof}
由编码单射性，
\[
|W_b(G_T)|
\le
|\mathrm{Phase}_b(k)|\,|R_{T,b}|.
\]
代入
\(
|W_b(G_T)|=2^{bT}
\)
与
\(
|\mathrm{Phase}_b(k)|=2^{bk}
\)，得
\[
2^{bT}\le 2^{bk}|R_{T,b}|,
\]
从而
\[
|R_{T,b}|\ge 2^{(T-k)b}.
\]

若 \(|R_{T,b}|\le T^C\)，则取对数得到
\[
(T-k)b\le C\log_2 T,
\]
即
\[
k\ge T-\frac{C}{b}\log_2 T.
\]
当 \(b\gtrsim \log T\) 时，右端第二项为 \(O(1)\)，故
\[
k\ge T-O(1).
\]
这说明低于满秩的连续相位压缩只可能以指数规模的离散残差为代价实现。证毕。
\end{proof}

\begin{corollary}[多项式残差口径下的时间窗口相位饱和]\label{cor:conclusion-time-window-polynomial-ledger-phase-saturation}
在定理 \ref{thm:conclusion-time-window-asymptotically-full-phase-law} 的设定下，若残差账本仅允许多项式增长，即
\[
|R_{T,b}|=T^{O(1)},
\]
则对任何满足 \(b\asymp \log T\) 的精度制度，都有
\[
k=T-O(1).
\]
特别地，若 \(k=o(T)\) 或 \(k\) 固定，则残差账本规模必为
\[
|R_{T,b}|=\exp(\Omega(T\log T))
\]
乃至更强的指数级增长。
\end{corollary}
\begin{proof}
前半部分由定理 \ref{thm:conclusion-time-window-asymptotically-full-phase-law} 直接得出。
若 \(k=o(T)\) 或 \(k\) 固定，则该定理中的下界
\[
|R_{T,b}|\ge 2^{(T-k)b}
\]
在 \(b\asymp \log T\) 下给出
\[
|R_{T,b}|\ge 2^{\Omega(T\log T)}=\exp(\Omega(T\log T)).
\]
证毕。
\end{proof}

\begin{conjecture}[可审计可逆性强迫时间圆维不可压缩]\label{conj:conclusion-time-window-auditably-reversible-circle-dim-rigidity}
设
\[
\phi:\ZZ^r\to \TT^d
\]
为群同态。若存在算法 \(\mathrm{Dec}\)，使得对任意 \(N\) 与任意 \(x\in B_N:=\{0,1,\dots,N\}^r\)，给定 \(\phi(x)\) 的 \(O(\log N)\)-bit 近似后，\(\mathrm{Dec}\) 可在 \(\mathrm{poly}(\log N)\) 时间内恢复 \(x\)，则必有
\[
d\ge r.
\]
将此应用于时间视窗 \(r=T\)，再与定理 \ref{thm:conclusion-time-window-asymptotically-full-phase-law} 联合，可得如下结构读法：在多项式可审计、近似读出且要求统一可逆的制度下，时间自由度不能被压到严格更低的连续圆维上；否则，要么解码复杂度失控，要么残差账本指数爆炸。
\end{conjecture}
