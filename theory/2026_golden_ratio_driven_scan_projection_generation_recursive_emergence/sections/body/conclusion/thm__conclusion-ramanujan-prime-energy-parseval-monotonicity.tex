\[
C_1(M):=\mathcal A_1(M)=-\log F_M(1),\qquad
\Psi_q(M):=\mathfrak S_q(M),\qquad
B_q(M):=\frac{\Psi_q(M)}{q},
\qquad
a_n(M):=\frac{u_n(M)}{n}.
\]

\begin{theorem}[Ramanujan 原子轴与 \(q\)-轴的 \(\zeta\)-剥离因子化]\label{thm:conclusion-ramanujan-atomic-qaxis-zeta-stripping}
\leanverified{paper\_conclusion\_ramanujan\_atomic\_qaxis\_zeta\_stripping}
对任意 \(\Re s>1\)，定义 Dirichlet 级数
\[
\mathfrak C_M(s):=\sum_{m\ge 1}\frac{C_m(M)}{m^s},
\qquad
\mathfrak \Psi_M(s):=\sum_{q\ge 1}\frac{\Psi_q(M)}{q^s}.
\]
则成立严格恒等式
\[
\mathfrak C_M(s)
=
\frac{1}{\zeta(s)}\,\mathfrak \Psi_M(s)
=
\frac{1}{\zeta(s)}
\sum_{n\ge 1}\frac{u_n(M)\sigma_{1-s}(n)}{n}.
\]
\end{theorem}
\begin{proof}
由 \(|u_n(M)|\le (d-1)\Lambda^n\) 与 \(|c_m(n)|\le n\)，当 \(\Re s>1\) 时，
\[
\sum_{m\ge 1}\frac1{|m^s|}
\sum_{n\ge 1}\frac{|u_n(M)|}{n}|c_m(n)|
\le
\zeta(\Re s)\sum_{n\ge 1}|u_n(M)|<\infty,
\]
故双重级数绝对收敛并可交换求和。对 \(m\ge 2\)，主稿给出
\[
C_m(M)=\sum_{n\ge 1}\frac{u_n(M)}{n}\,c_m(n),
\]
而对 \(m=1\) 由 \(c_1(n)\equiv 1\) 与定义 \(C_1(M)=-\log F_M(1)\) 得
\[
C_1(M)=\sum_{n\ge 1}\frac{u_n(M)}{n}\,c_1(n).
\]
于是
\[
\mathfrak C_M(s)
=
\sum_{n\ge 1}\frac{u_n(M)}{n}\sum_{m\ge 1}\frac{c_m(n)}{m^s}.
\]

再由
\[
\eta_q(n):=\sum_{a=1}^q e^{2\pi i an/q}=q\,\mathbf 1_{q\mid n},
\qquad
\eta_q(n)=\sum_{d\mid q}c_d(n),
\]
对 \(q\) 作 Möbius 反演，得到
\[
c_m(n)=\sum_{d\mid (m,n)} d\,\mu(m/d).
\]
故在 \(\Re s>1\) 时，
\[
\sum_{m\ge 1}\frac{c_m(n)}{m^s}
=
\sum_{d\mid n}d\sum_{r\ge 1}\frac{\mu(r)}{(dr)^s}
=
\frac{1}{\zeta(s)}\sum_{d\mid n}d^{1-s}
=
\frac{\sigma_{1-s}(n)}{\zeta(s)}.
\]
代回即得
\[
\mathfrak C_M(s)
=
\frac{1}{\zeta(s)}
\sum_{n\ge 1}\frac{u_n(M)\sigma_{1-s}(n)}{n}.
\]
最后由
\[
\sigma_{1-s}(n)=\sum_{d\mid n}d^{1-s}
\]
按 \(n=d\ell\) 重排，得到
\[
\sum_{n\ge 1}\frac{u_n(M)\sigma_{1-s}(n)}{n}
=
\sum_{d\ge 1}\frac1{d^s}\sum_{\ell\ge 1}\frac{u_{d\ell}(M)}{\ell}
=
\sum_{d\ge 1}\frac{\Psi_d(M)}{d^s}
=
\mathfrak \Psi_M(s).
\]
证毕。
\end{proof}

\begin{corollary}[原子是 Möbius-primitive 坐标，\(q\)-轴是其约数阴影]\label{cor:conclusion-ramanujan-atoms-are-mobius-primitive}
\leanverified{paper\_conclusion\_ramanujan\_atoms\_are\_mobius\_primitive}
对每个 \(q\ge 1\)，有精确反演
\[
\Psi_q(M)=\sum_{d\mid q}C_d(M),
\qquad
C_q(M)=\sum_{d\mid q}\mu(q/d)\,\Psi_d(M).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-ramanujan-atomic-qaxis-zeta-stripping}，
\[
\mathfrak \Psi_M(s)=\zeta(s)\mathfrak C_M(s)
=
\left(\sum_{r\ge 1}\frac1{r^s}\right)
\left(\sum_{d\ge 1}\frac{C_d(M)}{d^s}\right)
=
\sum_{q\ge 1}\frac{1}{q^s}\sum_{d\mid q}C_d(M).
\]
Dirichlet 系数唯一性即给出
\[
\Psi_q(M)=\sum_{d\mid q}C_d(M).
\]
第二式为 Möbius 反演。证毕。
\end{proof}

\begin{theorem}[素数模的非主角色能量等于单位同余类切片势能的中心方差]\label{thm:conclusion-prime-character-energy-centered-variance}
\leanverified{paper\_conclusion\_prime\_character\_energy\_centered\_variance}
设 \(p\ge 3\) 为素数，记
\[
U(p):=(\ZZ/p\ZZ)^\times,
\qquad
\bar V_p(M):=\frac1{p-1}\sum_{r\in U(p)}V_{p,r}(M),
\]
\[
\mathcal E_p(M):=\frac1{p(p-1)}\sum_{\chi\neq \chi_0}|L_M(\chi;p)|^2.
\]
则有严格 Parseval 型恒等式
\[
\sum_{r\in U(p)}|V_{p,r}(M)-\bar V_p(M)|^2
=
\mathcal E_p(M).
\]
\end{theorem}
\begin{proof}
主稿给出素数模角色反演
\[
V_{p,r}(M)=\frac1{p-1}\sum_{\chi}A(\chi)\chi(r),
\]
其中对非平凡角色
\[
A(\chi)=\frac{L_M(\chi;p)}{\tau(\chi)}.
\]
对 \(r\) 取平均，并利用非主角色在 \(U(p)\) 上求和为零，得到
\[
\bar V_p(M)=\frac{A(\chi_0)}{p-1}.
\]
故
\[
V_{p,r}(M)-\bar V_p(M)=\frac1{p-1}\sum_{\chi\neq\chi_0}A(\chi)\chi(r).
\]
再由有限交换群 \(U(p)\) 上角色系的正交性，
\[
\sum_{r\in U(p)}|V_{p,r}(M)-\bar V_p(M)|^2
=
\frac1{p-1}\sum_{\chi\neq\chi_0}|A(\chi)|^2.
\]
而模素数的非平凡本原角色满足 \(|\tau(\chi)|^2=p\)，从而
\[
\frac1{p-1}\sum_{\chi\neq\chi_0}|A(\chi)|^2
=
\frac1{p(p-1)}\sum_{\chi\neq\chi_0}|L_M(\chi;p)|^2
=
\mathcal E_p(M).
\]
证毕。
\end{proof}

\begin{theorem}[素数平坦化的支持指数 gcd 分类]\label{thm:conclusion-prime-flattening-support-gcd-classification}
\leanverified{paper\_conclusion\_prime\_flattening\_support\_gcd\_classification}
写
\[
F_M(t)=1+\sum_{j\ge 1}f_j(M)t^j,
\qquad
g(M):=\gcd\{j\ge 1:f_j(M)\neq 0\},
\]
并约定若 \(F_M\equiv 1\)，则 \(g(M)=0\)。则对任意素数 \(p\)，下列命题等价：
\[
\text{(i) }V_{p,r}(M)\ \text{与}\ r\in U(p)\ \text{无关};
\]
\[
\text{(ii) }\mathcal E_p(M)=0;
\]
\[
\text{(iii) }F_M(t)=G_p(t^p)\ \text{对某个}\ G_p\in\CC[t];
\]
\[
\text{(iv) }p\mid g(M).
\]
特别地，发生素数平坦化的素数集合恰为 \(g(M)\) 的素因子集合；因此除去 \(F_M\equiv1\) 的平凡情形外，该集合必为有限集。
\end{theorem}
\begin{proof}
(i)\(\Leftrightarrow\)(ii) 由定理 \ref{thm:conclusion-prime-character-energy-centered-variance} 立即成立。

(ii)\(\Leftrightarrow\)(iii) 由主稿的 cyclotomic 对称刚性判据：所有非主角色同时消失当且仅当
\[
F_M(t)=G_p(t^p),
\]
等价地，归一化非 Perron 谱多重集具有 \(\omega_p\)-闭合性。

(iii)\(\Leftrightarrow\)(iv) 则是纯代数事实：\(F_M(t)=G_p(t^p)\) 当且仅当 \(F_M\) 的每个非零系数指数均为 \(p\) 的倍数，亦即当且仅当 \(p\mid g(M)\)。证毕。
\end{proof}

\begin{theorem}[大素数扭点能量的二尺度渐近：离散角色表趋于连续 Parseval 质量]\label{thm:conclusion-large-prime-torsion-energy-two-scale-parseval}
定义连续 Parseval 质量
\[
\mathcal P(M):=\sum_{n\ge 1}\frac{|u_n(M)|^2}{n^2}.
\]
则当 \(p\to\infty\) 遍历素数时，有精确二尺度展开
\[
\mathcal E_p(M)
=
\mathcal P(M)-\frac{|C_1(M)|^2}{p-1}+O(\Lambda^p).
\]
若再记
\[
g(\theta):=-\log F_M(e^{i\theta})=\sum_{n\ge 1}\frac{u_n(M)}{n}e^{in\theta},
\]
则
\[
\mathcal P(M)=\frac1{2\pi}\int_0^{2\pi}|g(\theta)|^2\,d\theta,
\]
从而
\[
\mathcal E_p(M)
=
\frac1{2\pi}\int_0^{2\pi}|g(\theta)|^2\,d\theta
-\frac{|C_1(M)|^2}{p-1}
+O(\Lambda^p).
\]
\end{theorem}
\begin{proof}
令
\[
a_n:=\frac{u_n(M)}{n},
\]
则 \(|a_n|\le C\Lambda^n\)。对 \(1\le r\le p-1\) 写
\[
V_{p,r}(M)=\sum_{m\ge 0}a_{r+mp}.
\]
因此
\[
\sum_{r\in U(p)}|V_{p,r}(M)|^2
=
\sum_{r=1}^{p-1}\sum_{m,\ell\ge 0}a_{r+mp}\overline{a_{r+\ell p}}.
\]
其中对角部分给出
\[
\sum_{r=1}^{p-1}\sum_{m\ge 0}|a_{r+mp}|^2
=
\sum_{\substack{n\ge 1\\(n,p)=1}}|a_n|^2
=
\mathcal P(M)+O(\Lambda^{2p}),
\]
因为被剔除的仅是 \(p\) 的倍数项，而
\[
\sum_{m\ge 1}|a_{mp}|^2=O(\Lambda^{2p}).
\]
非对角部分满足
\[
\sum_{r=1}^{p-1}\sum_{m\neq \ell}|a_{r+mp}\overline{a_{r+\ell p}}|
=
O(\Lambda^p),
\]
因为每个乘积至少携带一个额外的 \(p\)-级指数衰减。故
\[
\sum_{r\in U(p)}|V_{p,r}(M)|^2
=
\mathcal P(M)+O(\Lambda^p).
\]

另一方面，由主角色项公式，
\[
(p-1)\bar V_p(M)=C_1(M)-B_p(M),
\]
而
\[
B_p(M)=\sum_{m\ge 1}a_{mp}=O(\Lambda^p).
\]
故
\[
(p-1)|\bar V_p(M)|^2
=
\frac{|C_1(M)-B_p(M)|^2}{p-1}
=
\frac{|C_1(M)|^2}{p-1}+O(\Lambda^p).
\]
结合定理 \ref{thm:conclusion-prime-character-energy-centered-variance}，
\[
\mathcal E_p(M)
=
\sum_{r\in U(p)}|V_{p,r}(M)-\bar V_p(M)|^2
=
\sum_{r\in U(p)}|V_{p,r}(M)|^2-(p-1)|\bar V_p(M)|^2,
\]
遂得
\[
\mathcal E_p(M)=\mathcal P(M)-\frac{|C_1(M)|^2}{p-1}+O(\Lambda^p).
\]
最后一式只是 Fourier 级数 \(g(\theta)=\sum_{n\ge 1}a_ne^{in\theta}\) 的 Parseval 恒等式。证毕。
\end{proof}

\begin{theorem}[主项非零时的大素数扭点能量终将严格单调地自下逼近 Parseval 极限]\label{thm:conclusion-large-prime-torsion-energy-eventual-monotone-from-below}
若 \(C_1(M)\neq 0\) 且 \(F_M\not\equiv 1\)，则存在 \(p_0\) 使得对任意素数 \(p'>p\ge p_0\)，成立
\[
0<\mathcal E_p(M)<\mathcal E_{p'}(M)<\mathcal P(M).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-prime-flattening-support-gcd-classification}，素数平坦化只能出现在有限多个素数上。再由定理 \ref{thm:conclusion-large-prime-torsion-energy-two-scale-parseval}，
\[
\mathcal E_p(M)=\mathcal P(M)-\frac{|C_1(M)|^2}{p-1}+\varepsilon_p,
\qquad
|\varepsilon_p|\le K\Lambda^p.
\]
于是
\[
\mathcal P(M)-\mathcal E_p(M)=\frac{|C_1(M)|^2}{p-1}+O(\Lambda^p)>0
\]
对充分大之 \(p\) 成立，从而
\[
\mathcal E_p(M)<\mathcal P(M).
\]
又对任意素数 \(p'>p\)，有
\[
\mathcal E_{p'}(M)-\mathcal E_p(M)
=
|C_1(M)|^2\left(\frac1{p-1}-\frac1{p'-1}\right)+O(\Lambda^p).
\]
前导项为正且量级至少为 \(c/p^2\)，而误差项指数小；故当 \(p\) 充分大时，上式严格为正。于是
\[
\mathcal E_p(M)<\mathcal E_{p'}(M).
\]
再取 \(p_0\) 足够大，使得同时避开有限个平坦化素数并保证上式严格成立，便得
\[
0<\mathcal E_p(M)<\mathcal E_{p'}(M)<\mathcal P(M)
\]
对所有素数 \(p'>p\ge p_0\) 成立。证毕。
\end{proof}

\begin{proposition}[素数 Ramanujan 原子是“倍数轴—质心轴”差额]\label{prop:conclusion-prime-ramanujan-atom-multiple-axis-barycenter-gap}
对任意素数 \(p\)，记
\[
\bar V_p(M):=\frac1{p-1}\sum_{r\in U(p)}V_{p,r}(M).
\]
则有精确公式
\[
B_p(M)=\frac{C_1(M)+C_p(M)}{p},
\qquad
\bar V_p(M)=\frac1p\left(C_1(M)-\frac{C_p(M)}{p-1}\right).
\]
等价地，
\[
C_p(M)=pB_p(M)-C_1(M)=(p-1)\bigl(C_1(M)-p\bar V_p(M)\bigr).
\]
\end{proposition}
\begin{proof}
由推论 \ref{cor:conclusion-ramanujan-atoms-are-mobius-primitive}，
\[
\Psi_p(M)=C_1(M)+C_p(M).
\]
因此
\[
B_p(M)=\frac{\Psi_p(M)}{p}=\frac{C_1(M)+C_p(M)}{p}.
\]
另一方面，由主角色项公式，
\[
(p-1)\bar V_p(M)=C_1(M)-B_p(M).
\]
代入上式并整理即可得到
\[
\bar V_p(M)=\frac1p\left(C_1(M)-\frac{C_p(M)}{p-1}\right),
\]
以及
\[
C_p(M)=pB_p(M)-C_1(M)=(p-1)\bigl(C_1(M)-p\bar V_p(M)\bigr).
\]
证毕。
\end{proof}

\begin{conjecture}[即使主项消失，非主扭点能量仍应最终单调地自下逼近 Parseval 极限]\label{conj:conclusion-prime-torsion-energy-monotone-critical-mainterm-zero}
设 \(F_M\not\equiv 1\)，并记
\[
\mathcal E_p(M)=\frac1{p(p-1)}\sum_{\chi\neq\chi_0}|L_M(\chi;p)|^2.
\]
则在剔除定理 \ref{thm:conclusion-prime-flattening-support-gcd-classification} 所分类的有限个平坦化素数后，序列 \((\mathcal E_p(M))\) 仍应沿全部充分大素数严格递增，并满足
\[
\mathcal E_p(M)\uparrow \mathcal P(M).
\]
\end{conjecture}
