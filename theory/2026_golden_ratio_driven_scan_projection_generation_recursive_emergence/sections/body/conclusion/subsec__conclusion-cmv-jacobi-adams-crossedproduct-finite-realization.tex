\begin{theorem}[有限窗谱函子在 CMV 证书塔上的有限表示性]\label{thm:conclusion-cmv-finite-window-spectral-functor-finite-representability}

\[
0\le i,j\le N,
\qquad
0\le r\le m
\]

\[
\langle e_i,\widetilde U_L^{\,r}e_j\rangle
=
\langle e_i,U_\infty^{\,r}e_j\rangle.
\]
因此有限窗谱函子
\[
\mathfrak M_{N,m}(U)
:=
\bigl(\langle e_i,U^{r}e_j\rangle\bigr)_{0\le i,j\le N,\ 0\le r\le m}
\]
沿逆系统 \((\widetilde U_L)_L\) 最终常值。等价地，\(F_L\) 在 \(0\) 处的任意有限阶 Taylor 喷流都在某一有限级别后完全冻结。
\end{theorem}
\begin{proof}

\langle e_i,\widetilde U_L^{\,r}e_j\rangle
=
\langle e_i,U_\infty^{\,r}e_j\rangle
\qquad(0\le i,j\le N).
\]
对有限集合 \(r=0,1,\dots,m\) 取最大值
\[
L_\ast(N,m):=\max_{0\le r\le m}L_\ast(N,r),
\]
即可得到所示同时稳定。于是 \(\mathfrak M_{N,m}\) 沿证书塔最终常值。

Carath\'eodory 函数在 \(0\) 处的 Taylor 系数由有限阶谱矩唯一确定，而结论 \ref{con:xi-cmv-finite-window-freeze} 也已经给出了二者的等价稳定性；故最后一条随之成立。证毕。
\end{proof}

\begin{theorem}[Jacobi 阴影的最小半正定修复秩等于禁区根数]\label{thm:conclusion-jacobi-shadow-minimal-psd-repair-rank-equals-forbidden-root-count}
设 \(P_L\) 的全部根均为实数，且
\[
\widetilde P_L(S)=\det(SI-J_L)
\]
为其实对称三对角实现。记
\[
\nu_{\mathrm{forb}}(L)
:=
\#\Bigl(\mathrm{Roots}(P_L)\cap\bigl((2,c_L)\cup(-c_L,-2)\bigr)\Bigr),
\]
并定义 Hermitian 修复秩
\[
\operatorname{rep}(L)
:=
\min\bigl\{\rank K:\ K\succeq0,\ Q_L^{\mathrm{strip}}(J_L)+K\succeq0\bigr\}.
\]
则有精确恒等式
\[
\operatorname{rep}(L)=\nu_{\mathrm{forb}}(L).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part28b-strip-jacobi-quartic-psd}，
\[
\nu_{\mathrm{forb}}(L)
=
\nu_-\!\bigl(Q_L^{\mathrm{strip}}(J_L)\bigr).
\]
记
\[
H:=Q_L^{\mathrm{strip}}(J_L),
\qquad
r:=\nu_-(H).
\]
取 \(H\) 的谱分解
\[
H=U\diag(\lambda_1,\dots,\lambda_n)U^\ast,
\]
其中恰有 \(r\) 个特征值严格为负。不妨将它们排在前面，并令
\[
K:=U\diag(-\lambda_1,\dots,-\lambda_r,0,\dots,0)U^\ast.
\]
则 \(K\succeq 0\)、\(\rank K=r\)，并且 \(H+K\succeq0\)。于是
\[
\operatorname{rep}(L)\le r.
\]

反过来，若某个 \(K\succeq0\) 使 \(H+K\succeq0\) 且 \(\rank K<r\)，记 \(E^-\) 为 \(H\) 的负谱子空间，则 \(\dim E^-=r\)。由于 \(\rank(K|_{E^-})<r\)，存在非零向量 \(x\in E^-\cap\ker K\)。于是
\[
\langle x,(H+K)x\rangle
=
\langle x,Hx\rangle
<
0,
\]
与 \(H+K\succeq0\) 矛盾。故 \(\operatorname{rep}(L)\ge r\)。综上
\[
\operatorname{rep}(L)=r=\nu_{\mathrm{forb}}(L).
\]
证毕。
\end{proof}

\begin{theorem}[端点无原子且支撑有谱隙时 Toeplitz 最小特征值指数塌缩]\label{thm:conclusion-endpoint-gap-forces-exponential-collapse-of-toeplitz-minimum}
设 \(\mu(\{-1\})=0\)，并存在闭集 \(K\subset \TT\setminus\{-1\}\) 使 \(\mu(\TT\setminus K)=0\)。定义
\[
q(K):=\sup_{\xi\in K}\left|\frac{1-\xi}{2}\right|\in(0,1).
\]
则对一切 \(N\ge0\) 都有
\[
\lambda_{\min}(T_N)
\le
\mu(K)\,q(K)^{2N}\,\frac{4^N}{\binom{2N}{N}}.
\]
特别地，
\[
\lambda_{\min}(T_N)
\le
\mu(K)\,\sqrt{\pi N}\,q(K)^{2N}(1+o(1)),
\qquad N\to\infty.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:xi-endpoint-heat-probe-toeplitz-herglotz-identity} 与推论 \ref{cor:xi-endpoint-heat-probe-monotone-limit-gap}，
\[
a_N
=
\int_{\TT}\left|\frac{1-\xi}{2}\right|^{2N}\dd\mu(\xi)
\le
\mu(K)\,q(K)^{2N},
\]
这里使用了 \(\mu(\{-1\})=0\) 与 \(\operatorname{supp}\mu\subseteq K\)。

另一方面，由定理 \ref{thm:xi-endpoint-heat-probe-spectral-margin-lower-bound}，
\[
a_N\ge \lambda_{\min}(T_N)\frac{\binom{2N}{N}}{4^N}.
\]
两式合并即得
\[
\lambda_{\min}(T_N)
\le
\mu(K)\,q(K)^{2N}\,\frac{4^N}{\binom{2N}{N}}.
\]
再用中心二项式渐近
\[
\binom{2N}{N}\sim \frac{4^N}{\sqrt{\pi N}}
\]
得到第二式。证毕。
\end{proof}

\begin{theorem}[有限秩端点探针的完全标量刚性]\label{thm:conclusion-finite-rank-endpoint-probe-complete-scalar-rigidity}
设 \(\mu\) 为 \(\TT\) 上有限正 Borel 测度，并定义 untwisted 端点探针
\[
a_N:=4^{-N}\int_{\TT}|1-\xi|^{2N}\,\dd\mu(\xi),
\qquad
m_{-1}:=\mu(\{-1\}),
\qquad
s_N:=a_N-m_{-1}.
\]
若 \((s_N)_{N\ge 0}\) 的 Hankel 秩有限，记为 \(R\)，则存在唯一的
\[
x_1,\dots,x_R\in(0,1),
\qquad
w_1,\dots,w_R>0
\]
使
\[
s_N=\sum_{j=1}^{R}w_jx_j^N.
\]
并且 \(\mu\) 在 \(\TT\setminus\{-1\}\) 上的支撑必包含于至多 \(R\) 对共轭点
\[
\{e^{i\theta_j},e^{-i\theta_j}\}_{j=1}^{R},
\qquad
x_j=\frac{1-\cos\theta_j}{2}.
\]

若进一步给出全部旋相端点探针族
\[
a_N(\omega):=
4^{-N}\int_{\TT}|1-\omega\xi|^{2N}\,\dd\mu(\xi)
\qquad (N\ge0,\ \omega\in\TT),
\]
则 \(\mu\) 被唯一确定；等价地，相应 Carath\'eodory 函数被唯一确定到一个纯虚常数。
\end{theorem}
\begin{proof}
令
\[
r(\xi):=\frac{|1-\xi|^2}{4}\in[0,1].
\]
则
\[
s_N
=
\int_{\TT\setminus\{-1\}} r(\xi)^N\,\dd\mu(\xi)
=
\int_{[0,1)}x^N\,\dd(r_\#\mu^\circ)(x),
\]
其中 \(\mu^\circ:=\mu|_{\TT\setminus\{-1\}}\)。由正文关于端点探针的 Hausdorff--Hankel 刚性，\((s_N)\) 的 Hankel 秩有限当且仅当推前测度 \(r_\#\mu^\circ\) 是有限原子；于是存在唯一表示
\[
r_\#\mu^\circ=\sum_{j=1}^{R}w_j\delta_{x_j},
\qquad
x_j\in(0,1),\ w_j>0,
\]
从而
\[
s_N=\sum_{j=1}^{R}w_jx_j^N.
\]

对每个 \(x_j\in(0,1)\)，取 \(\theta_j\in(0,\pi)\) 满足
\[
x_j=\frac{1-\cos\theta_j}{2}.
\]
由于 \(r(e^{i\theta})=r(e^{-i\theta})=(1-\cos\theta)/2\)，故 \(\mu^\circ\) 的支撑只能落在这些原子的原像上，即至多 \(R\) 对共轭点
\[
\{e^{i\theta_j},e^{-i\theta_j}\}_{j=1}^{R}.
\]

若进一步知道全部旋相端点探针 \(a_N(\omega)\)，则由定理 \ref{thm:conclusion-single-adams-phase-profile-arithmetic-fourier-microscope} 在 \(d=1\) 的情形，\(\omega\mapsto a_N(\omega)\) 的 Fourier 系数对所有 \(|m|\le N\) 精确恢复 \(\mu\) 的 Fourier 系数 \(c_{-m}\)。令 \(N\) 变化，便得到全部 Fourier 系数 \((c_h)_{h\in\ZZ}\)。由 Herglotz 唯一性，这些 Fourier 系数唯一决定 \(\mu\)；等价地，相应 Carath\'eodory 函数唯一到一个纯虚常数。证毕。
\end{proof}

\begin{corollary}[有限秩端点律诱导全部 Adams 扭曲探针的统一递推闭包]\label{cor:conclusion-finite-rank-endpoint-probe-uniform-adams-recursion-closure}
在定理 \ref{thm:conclusion-finite-rank-endpoint-probe-complete-scalar-rigidity} 的设定下，固定任意整数 \(d\ge1\) 与任意 \(\omega\in\TT\)。定义 Adams 扭曲端点探针
\[
a_N^{(d)}(\omega):=
4^{-N}\int_{\TT}|1-\omega\xi^d|^{2N}\,\dd\mu(\xi),
\qquad
\Sigma_{d,\omega}:=\{\xi\in\TT:\ \omega\xi^d=-1\}.
\]
则序列
\[
a_N^{(d)}(\omega)-\mu(\Sigma_{d,\omega})
\]
是至多 \(2R\) 项指数序列之和；因此其 Hankel 秩至多为 \(2R\)，并满足一个阶数至多 \(2R\) 的常系数线性递推。若相应的 \(2R\) 个数
\[
\frac{|1-\omega e^{\pm i d\theta_j}|^2}{4},
\qquad 1\le j\le R,
\]
彼此互异，则该递推阶数 generically 正好等于 \(2R\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-finite-rank-endpoint-probe-complete-scalar-rigidity}，\(\mu|_{\TT\setminus\{-1\}}\) 的支撑包含于至多 \(2R\) 个点
\[
\xi_1,\dots,\xi_{M_{d,\omega}},
\qquad M_{d,\omega}\le 2R.
\]
将这一有限支撑表示代入 Adams 扭曲端点探针，便得
\[
a_N^{(d)}(\omega)
=
\mu(\Sigma_{d,\omega})
\sum_{\ell=1}^{M_{d,\omega}}
\alpha_\ell
\left(\frac{|1-\omega\xi_\ell^d|^2}{4}\right)^N,
\]
其中 \(\alpha_\ell>0\) 为对应原子质量。故去掉端点常数项后，所得序列是一个至多 \(2R\) 项的指数和。

有限指数和的 Hankel 秩恰等于其不同基数的个数，因而该序列的 Hankel 秩至多为 \(2R\)，并满足阶数至多 \(2R\) 的常系数线性递推。若列出的 \(2R\) 个基数彼此互异，则一般位置下没有代数重合，递推阶数便达到 \(2R\)。证毕。
\end{proof}

\begin{theorem}[互素 Adams 扭曲在有限 cyclotomic 支撑上给出逐点端点层析]\label{thm:conclusion-single-coprime-adams-twist-pointwise-cyclotomic-tomography}
设 \(\mu\) 支撑于 \(M\)-次单位根群
\[
\mu_M:=\{\zeta\in\TT:\ \zeta^M=1\}
\]
上，并取整数 \(d\ge1\) 满足 \(\gcd(d,M)=1\)。则对每个 \(\zeta\in\mu_M\)，都有精确恢复公式
\[
\mu(\{\zeta\})
=
\lim_{N\to\infty}a_N^{(d)}(-\zeta^{-d}).
\]
因此，一个与 \(M\) 互素的 Adams 扭曲阶 \(d\) 连同旋相参数 \(\omega\) 的扫描，就足以逐点恢复全部 cyclotomic 原子质量。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-endpoint-heat-probe-adams-twist-roots}，
\[
a_N^{(d)}(\omega)\downarrow \mu(\Sigma_{d,\omega}),
\qquad
\Sigma_{d,\omega}:=\{\xi\in\TT:\ \omega\xi^d=-1\}.
\]
固定 \(\zeta\in\mu_M\)，取
\[
\omega:=-\zeta^{-d}.
\]
则
\[
\Sigma_{d,-\zeta^{-d}}
=
\{\xi\in\TT:\ \xi^d=\zeta^d\}.
\]
由于 \(\gcd(d,M)=1\)，幂映射
\[
\phi_d:\mu_M\to \mu_M,
\qquad
\xi\mapsto \xi^d
\]
是双射。因此
\[
\Sigma_{d,-\zeta^{-d}}\cap \mu_M=\{\zeta\}.
\]
再结合 \(\operatorname{supp}\mu\subseteq\mu_M\)，便有
\[
\mu(\Sigma_{d,-\zeta^{-d}})=\mu(\{\zeta\}).
\]
对 \(N\to\infty\) 取极限即得结论。证毕。
\end{proof}

\begin{theorem}[单一 Adams 旋相剖面对等差 Fourier 系数的精确显微性]\label{thm:conclusion-single-adams-phase-profile-arithmetic-fourier-microscope}
固定整数 \(d\ge1\) 与 \(N\ge0\)。则函数
\[
\omega\longmapsto a_N^{(d)}(\omega)
\]
是次数不超过 \(N\) 的 Laurent 多项式，并具有显式 Fourier 展开
\[
a_N^{(d)}(\omega)
=
\sum_{m=-N}^{N}
\Bigl(
(-1)^m4^{-N}\binom{2N}{N+m}c_{-dm}
\Bigr)\omega^m.
\]
因此对每个 \(|m|\le N\)，都有精确反演公式
\[
c_{-dm}
=
(-1)^m4^N\binom{2N}{N+m}^{-1}\widehat a_N^{(d)}(m),
\]
其中
\[
\widehat a_N^{(d)}(m)
:=
\frac1{2\pi}\int_0^{2\pi}a_N^{(d)}(e^{i\theta})e^{-im\theta}\,\dd\theta.
\]
故单一扭曲阶 \(d\) 的单一旋相剖面，在深度 \(N\) 上精确恢复了 Fourier 系数子列 \(\{c_{dn}\}\) 的带限窗口。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:xi-adams-binomial-probe-fourier-diagonalization} 的 Fourier 展开与反演公式的直接重述。由于
\[
(-1)^m4^{-N}\binom{2N}{N+m}\neq 0
\qquad(|m|\le N),
\]
每个 Fourier 模态与对应的 \(c_{-dm}\) 一一对应，故旋相剖面对等差指标 \(d\ZZ\) 上的 Fourier 系数给出精确显微读出。证毕。
\end{proof}

\begin{corollary}[有限支撑指数已知时单一有限阶 Adams 剖面即可完全重构测度]\label{cor:conclusion-single-finite-order-adams-profile-reconstructs-finite-cyclotomic-measure}
设 \(\mu\) 支撑于 \(\mu_M\)，并取 \(\gcd(d,M)=1\)。若 \(N\ge M-1\)，则单一函数
\[
\omega\longmapsto a_N^{(d)}(\omega)
\]
已经唯一决定 \(\mu\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-single-adams-phase-profile-arithmetic-fourier-microscope}，知道 \(a_N^{(d)}\) 即等价于知道全部
\[
c_{-dm}
\qquad(-N\le m\le N).
\]
当 \(N\ge M-1\) 时，区间 \([-N,N]\) 模 \(M\) 的像覆盖整个 \(\ZZ/M\ZZ\)；再由 \(\gcd(d,M)=1\)，乘以 \(d\) 在 \(\ZZ/M\ZZ\) 上为置换，故 \(\{c_r:r\in\ZZ/M\ZZ\}\) 全部由 \(a_N^{(d)}\) 唯一恢复。

另一方面，\(\mu\) 支撑于有限循环群 \(\mu_M\)，其原子质量由有限 Fourier 反演唯一决定：
\[
\mu(\{\zeta\})
=
\frac1M\sum_{r=0}^{M-1}c_r\,\zeta^r
\qquad(\zeta\in\mu_M).
\]
故 \(\mu\) 被唯一重构。证毕。
\end{proof}

\begin{theorem}[交叉积规范迹的有限窗有限词长完全有限化]\label{thm:conclusion-crossed-product-canonical-trace-finite-window-finite-word-realization}
固定 \(N,m\ge0\)。令 \(\mathfrak W_{N,m}\) 表示由全部形式词
\[
P_NU^{\varepsilon_1 n_1}P_N\cdots P_NU^{\varepsilon_r n_r}P_N
\]
所张成的复向量空间，其中 \(r\ge1\)、\(\varepsilon_j\in\{\pm1\}\)、\(n_j\ge1\)，且
\[
\sum_{j=1}^r n_j\le m.
\]
若 \(V\) 是定义在 \(\ell^2(\NN_0)\) 上的幺正算子，则记 \(\mathfrak W_{N,m}(V)\) 为把上式中的形式符号 \(U\) 替换为 \(V\) 后得到的算子空间。则存在 \(L_\ast(N,m)\)，使得对每个形式词 \(A\in \mathfrak W_{N,m}\) 与一切 \(L\ge L_\ast(N,m)\) 都有
\[
A(\widetilde U_L)=A(U_\infty).
\]
特别地，一切由有限时间平移、有限窗口投影和有限乘积生成的有限窗有限词长读出，在某一有限级别 CMV 证书上已经被精确实现。结合附录定理 \ref{thm:hsp-period-trace}，这说明交叉积语义中的规范迹有限观测在该口径下不携带任何超出有限证书的数据。
\end{theorem}
\begin{proof}
先看单个压缩因子。若 \(1\le n\le m\)，则由定理 \ref{thm:conclusion-cmv-finite-window-spectral-functor-finite-representability}，存在 \(L_\ast(N,m)\) 使得对一切 \(L\ge L_\ast(N,m)\)，
\[
P_N\widetilde U_L^{\,n}P_N=P_NU_\infty^{\,n}P_N.
\]
又因为 \(\widetilde U_L\) 与 \(U_\infty\) 都是幺正算子，
\[
P_N\widetilde U_L^{-n}P_N
=
\bigl(P_N\widetilde U_L^{\,n}P_N\bigr)^\ast
=
\bigl(P_NU_\infty^{\,n}P_N\bigr)^\ast
=
P_NU_\infty^{-n}P_N.
\]
故所有满足 \(\sum_j n_j\le m\) 的生成词在 \(L\ge L_\ast(N,m)\) 后都逐字一致。

由于这样的生成词只有有限多个，且 \(\mathfrak W_{N,m}\) 是它们张成的有限维空间，线性扩张便得到：对任意形式词 \(A\in\mathfrak W_{N,m}\)，当 \(L\ge L_\ast(N,m)\) 时都有
\[
A(\widetilde U_L)=A(U_\infty).
\]
最后，附录定理 \ref{thm:hsp-period-trace} 已将扫描协议的时间平均值提升为交叉积代数上的规范迹值。因此，在交叉积读出经由有限时间平移、有限窗口投影与有限乘积实现的口径下，上述稳定性说明其有限观测已经在某个有限证书级别上完全实现。证毕。
\end{proof}

\begin{conjecture}[全局提升的唯一剩余障碍应当集中于 Tate--Maslov 指数类]\label{conj:conclusion-tate-maslov-only-remaining-global-obstruction}
设一条协议族同时满足以下三类闭合：
\begin{enumerate}
  \item 所有有限窗 CMV 谱矩最终冻结；
  \item 所有有限级 Jacobi 阴影的条带修复秩最终归零；
  \item 对每个 \(d\ge1\) 与每个 \(\omega\in\TT\)，Adams 扭曲端点剖面已经给出全部 cyclotomic 边界原子。
\end{enumerate}
则任何阻止该协议进一步提升为全局 de Branges / Pontryagin 实现的剩余障碍，都应当集中在 Tate--Maslov 指数类，而不再可能来自有限窗谱矩、条带禁区根计数或 boundary cyclotomic 原子。
\end{conjecture}

\noindent
因此，这里出现的并不是又一条孤立的数值恒等式，而是一条有限实现链：有限 CMV 证书塔先把有限窗非交换谱函子压缩成最终常值对象；Jacobi 阴影再把开临界条带上的离线根数精确改写为最小半正定修复秩；端点热核与 Adams 扭曲则把边界 cyclotomic 原子的逐点层析降为单参数旋相剖面；最后，交叉积规范迹口径下的一切有限窗有限词长读出都已经在某一有限级别证书上完全实现。于是，在有限观测层面，局部谱矩、半正定几何、边界层析与非交换读出不再是四套彼此分离的机制，而被焊接成了一条同一的有限证书链；若其上仍有全局 no-go 剩余，则它只能属于 Tate--Maslov 型的全局指数障碍。
