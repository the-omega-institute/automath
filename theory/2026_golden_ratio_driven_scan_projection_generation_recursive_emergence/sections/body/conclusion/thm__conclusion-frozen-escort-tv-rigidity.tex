\begin{corollary}[冻结相 escort 测度在全 \texorpdfstring{$[1,\infty]$}{[1,infty]} Rényi 轴上的统一塌缩]\label{cor:conclusion-frozen-escort-all-renyi-collapse}
\leanverified{paper\_conclusion\_frozen\_escort\_all\_renyi\_collapse}
沿用定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 与定理 \ref{thm:conclusion-frozen-escort-integer-renyi-collapse} 的记号。若
\[
a>a_{\mathrm c},
\]
则对每个 Rényi 阶 \(s\in[1,\infty]\) 都有
\[
\lim_{m\to\infty}\frac{1}{m}H_s(\pi_{m,a})=g_\ast.
\]
亦即，一旦进入冻结相，escort 分布在全部正则阶上都只保留基态简并指数 \(g_\ast\)，而完全遗忘非基态尾部。
\end{corollary}
\begin{proof}
当 \(s=1\) 时，结论正是定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 的 Shannon 情形；当 \(1<s<\infty\) 时，结论亦由同一定理对固定 Rényi 阶的结论直接给出；当 \(s=\infty\) 时，结论由定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 的 min-entropy 情形给出。故全区间 \(s\in[1,\infty]\) 上统一成立。证毕。
\end{proof}

\begin{theorem}[冻结相到基态均匀分布的精确全变差刚性]\label{thm:conclusion-frozen-escort-tv-rigidity}
沿用定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 的记号，并令
\[
U_m^{\max}:=\mathrm{Unif}(A_m^{\max}),
\qquad
A_m^{\max}:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
K_m:=|A_m^{\max}|.
\]
若 \(a>a_{\mathrm c}\)，则有精确恒等式
\[
\bigl|\pi_{m,a}-U_m^{\max}\bigr|_{\mathrm{TV}}
=
1-\pi_{m,a}(A_m^{\max}).
\]
\leanverified{paper\_conclusion\_frozen\_escort\_tv\_rigidity}
特别地，
\[
\bigl|\pi_{m,a}-U_m^{\max}\bigr|_{\mathrm{TV}}
\le
\exp\!\bigl(-m\Delta(a)+o(m)\bigr),
\qquad
\Delta(a):=a(\alpha_\ast-\alpha_\ast^{(2)})-(\log\varphi-g_\ast)>0.
\]
因此，冻结相不仅是熵率塌缩，而且以指数速度在全变差意义下逼近基态均匀分布。
\end{theorem}
\begin{proof}
记
\[
\varepsilon_m:=1-\pi_{m,a}(A_m^{\max}).
\]
由定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse}，
\[
\varepsilon_m\le \exp\!\bigl(-m\Delta(a)+o(m)\bigr).
\]
另一方面，对每个 \(x\in A_m^{\max}\) 都有 \(d_m(x)=M_m\)，故
\[
\pi_{m,a}(x)=\frac{M_m^a}{S_a(m)}=\frac{1-\varepsilon_m}{K_m},
\qquad
U_m^{\max}(x)=\frac{1}{K_m}.
\]
因此
\[
\sum_{x\in A_m^{\max}}\bigl|\pi_{m,a}(x)-U_m^{\max}(x)\bigr|
=
K_m\left|\frac{1-\varepsilon_m}{K_m}-\frac{1}{K_m}\right|
=
\varepsilon_m.
\]
在补集 \(X_m\setminus A_m^{\max}\) 上，\(U_m^{\max}\equiv 0\)，故
\[
\sum_{x\notin A_m^{\max}}\bigl|\pi_{m,a}(x)-U_m^{\max}(x)\bigr|
=
\sum_{x\notin A_m^{\max}}\pi_{m,a}(x)
=
\varepsilon_m.
\]
于是
\[
\bigl|\pi_{m,a}-U_m^{\max}\bigr|_{\mathrm{TV}}
=
\frac12\left(\varepsilon_m+\varepsilon_m\right)
=
\varepsilon_m
=
1-\pi_{m,a}(A_m^{\max}).
\]
再代入 \(\varepsilon_m\) 的指数界即得结论。证毕。
\end{proof}
