\begin{theorem}[稀疏爆破模板的分母逼近普适律]\label{thm:conclusion-kronecker-sparse-blowup-universal-denominator-law}
设
\[
\alpha_{\mathbf b}:=[0;1,b_1,1,1,b_2,1,1,b_3,\ldots],
\qquad
b_k\to\infty,
\]
其收敛分数记为 \(p_n/q_n\)，并记
\[
\mathsf T_q(\alpha):=W_1\!\left(\frac1q\sum_{t=0}^{q-1}\delta_{\{t\alpha\}},\lambda\right).
\]
再置
\[
\delta_n:=\alpha_{\mathbf b}-\frac{p_n}{q_n}.
\]
则有
\[
q_n^2\left|\delta_n\right|
\longrightarrow
\begin{cases}
\dfrac12,& n\equiv 0,2 \pmod 3,\\[4pt]
0,& n\equiv 1 \pmod 3.
\end{cases}
\]
\end{theorem}
\begin{proof}
记部分商为 \(a_n\)，则
\[
a_{3k+1}=1,\qquad a_{3k+2}=b_{k+1},\qquad a_{3k+3}=1.
\]
再记
\[
r_n:=\frac{q_{n-1}}{q_n},
\qquad
\beta_n:=[a_{n+1};a_{n+2},a_{n+3},\ldots].
\]
由收敛分母递推 \(q_{n+1}=a_{n+1}q_n+q_{n-1}\) 得
\[
r_{n+1}=\frac1{a_{n+1}+r_n},
\qquad
0<r_n<1.
\]
因此
\[
r_{3k+2}=\frac1{b_{k+1}+r_{3k+1}}\to0,
\]
从而
\[
r_{3k+3}=\frac1{1+r_{3k+2}}\to1,
\qquad
r_{3k+4}=\frac1{1+r_{3k+3}}\to\frac12.
\]
重编号即得
\[
r_{3k}\to1,\qquad r_{3k+1}\to\frac12,\qquad r_{3k+2}\to0.
\]

另一方面，
\[
\beta_{3k}=[1;b_{k+1},1,1,b_{k+2},\ldots]\to1,
\]
\[
\beta_{3k+1}=[b_{k+1};1,1,b_{k+2},\ldots]\to\infty,
\]
\[
\beta_{3k+2}=[1;1,b_{k+2},1,1,b_{k+3},\ldots]\to2.
\]
又由 continued fraction 的标准恒等式
\[
\alpha_{\mathbf b}
=
\frac{\beta_n p_n+p_{n-1}}{\beta_n q_n+q_{n-1}}
\]
及
\[
p_{n-1}q_n-p_nq_{n-1}=(-1)^n
\]
可得
\[
\delta_n
=
\frac{(-1)^n}{q_n(\beta_n q_n+q_{n-1})},
\qquad
q_n^2|\delta_n|=\frac1{\beta_n+r_n}.
\]
将上面的三个极限代入即得结论。证毕。
\end{proof}

\begin{theorem}[Euler--Kronecker 六相输运律]\label{thm:conclusion-euler-kronecker-sixphase-transport-law}
\leanverified{paper\_conclusion\_euler\_kronecker\_sixphase\_transport\_law}
取
\[
\alpha=e-2=[0;1,2,1,1,4,1,1,6,\ldots],
\]
并记 \(p_n/q_n\) 为其收敛分数。则沿分母刻度有
\[
q_n\,\mathsf T_{q_n}(\alpha)\longrightarrow
\begin{cases}
\dfrac13,& n\equiv 0,2 \pmod 6,\\[4pt]
\dfrac12,& n\equiv 1,4 \pmod 6,\\[4pt]
\dfrac34,& n\equiv 3,5 \pmod 6.
\end{cases}
\]
\end{theorem}
\begin{proof}
这是定理 \ref{thm:conclusion-kronecker-sparse-blowup-universal-denominator-law} 在 \(b_k=2k\) 时的特例，因此
\[
c_n:=q_n^2\left|\alpha-\frac{p_n}{q_n}\right|
\longrightarrow
\begin{cases}
\dfrac12,& n\equiv 0,2 \pmod 3,\\[4pt]
0,& n\equiv 1 \pmod 3.
\end{cases}
\]
又由
\[
\alpha-\frac{p_n}{q_n}
=
\frac{(-1)^n}{q_n(\beta_n q_n+q_{n-1})}
\]
可知 \(\delta_n\) 与 \((-1)^n\) 同号。正文的分母刻度闭式于是给出：

当 \(n\) 为奇数时（坏侧），
\[
q_n\mathsf T_{q_n}(\alpha)
=
\frac12+\frac{q_n(q_n-1)}{2}\left|\delta_n\right|
=
\frac12+\frac{c_n}{2}+o(1).
\]
当 \(n\) 为偶数时（好侧），
\[
q_n\mathsf T_{q_n}(\alpha)
=
\frac12-\frac{q_n(q_n-1)}{2}\left|\delta_n\right|
+\frac{q_n^2(q_n-1)(2q_n-1)}{6}\left|\delta_n\right|^2
=
\frac12-\frac{c_n}{2}+\frac{c_n^2}{3}+o(1).
\]
分别代入 \(c_n\to 1/2\) 与 \(c_n\to0\)，并按 \(n\bmod 6\) 区分奇偶，即得
\[
\frac12-\frac14+\frac1{12}=\frac13,\qquad
\frac12,\qquad
\frac12+\frac14=\frac34.
\]
证毕。
\end{proof}

\begin{corollary}[Euler 分母刻度的三点输运谱]\label{cor:conclusion-euler-kronecker-threepoint-transport-spectrum}
\leanverified{paper\_conclusion\_euler\_kronecker\_threepoint\_transport\_spectrum}
对 \(\alpha=e-2\)，有
\[
\operatorname{Acc}\bigl\{q_n\mathsf T_{q_n}(\alpha)\bigr\}_{n\ge1}
=
\left\{\frac13,\frac12,\frac34\right\}.
\]
特别地，
\[
\liminf_{n\to\infty}q_n\mathsf T_{q_n}(\alpha)=\frac13,
\qquad
\limsup_{n\to\infty}q_n\mathsf T_{q_n}(\alpha)=\frac34.
\]
并且在 \(n\equiv1,4\pmod6\) 的两条子序列上有精确平坦化
\[
\mathsf T_{q_n}(e-2)=\frac1{2q_n}+o(q_n^{-1}).
\]
\end{corollary}
\begin{proof}
三点极限谱直接由定理 \ref{thm:conclusion-euler-kronecker-sixphase-transport-law} 读出。最后一式对应于 \(q_n\mathsf T_{q_n}(\alpha)\to 1/2\) 的两条剩余类。证毕。
\end{proof}

\begin{theorem}[Euler 分母刻度的 Lipschitz 审计三谱界]\label{thm:conclusion-euler-kronecker-lipschitz-threephase-audit}
对任意 Lipschitz 函数 \(f:[0,1)\to\RR\)，沿 \(\alpha=e-2\) 的收敛分母 \(q_n\) 有
\[
q_n\left|
\frac1{q_n}\sum_{t=0}^{q_n-1}f(\{t\alpha\})
-\int_0^1 f(x)\,dx
\right|
\le
\operatorname{Lip}(f)\,\sigma_n+o(1),
\]
其中
\[
\sigma_n=
\begin{cases}
\dfrac13,& n\equiv 0,2 \pmod 6,\\[4pt]
\dfrac12,& n\equiv 1,4 \pmod 6,\\[4pt]
\dfrac34,& n\equiv 3,5 \pmod 6.
\end{cases}
\]
特别地，
\[
\limsup_{n\to\infty}
q_n\left|
\frac1{q_n}\sum_{t=0}^{q_n-1}f(\{t\alpha\})-\int_0^1 f
\right|
\le
\frac34\,\operatorname{Lip}(f),
\]
而在 \(n\equiv0,2\pmod6\) 的两条子序列上，上述审计预算降至
\[
\frac13\,\operatorname{Lip}(f)+o(1).
\]
\end{theorem}
\begin{proof}
正文的 Lipschitz 读出不等式给出
\[
\left|
\frac1N\sum_{t=0}^{N-1}f(\{t\alpha\})-\int_0^1 f
\right|
\le
\operatorname{Lip}(f)\,\mathsf T_N(\alpha).
\]
取 \(N=q_n\)，再代入定理 \ref{thm:conclusion-euler-kronecker-sixphase-transport-law} 即得结论。证毕。
\end{proof}

\begin{proposition}[局部精确性与全局非自动性的严格分离]\label{prop:conclusion-euler-kronecker-local-exact-global-nonautomatic}
\(\alpha=e-2\) 同时满足以下两条性质：
\begin{enumerate}
\item \(\alpha\) 不属于二次数域，因此不落在经典二次 Ostrowski 系统的有限自动机稳定区；
\item 尽管如此，定理 \ref{thm:conclusion-euler-kronecker-sixphase-transport-law}--\ref{thm:conclusion-euler-kronecker-lipschitz-threephase-audit} 仍在全部收敛分母刻度上给出精确六相输运常数与 Lipschitz 审计预算。
\end{enumerate}
因此，分母刻度上的精确输运闭合严格弱于全局 Ostrowski 自动机闭合。
\end{proposition}
\begin{proof}
由于 \(e\) 超越，\(\alpha=e-2\) 亦非二次无理数，故不属于经典二次 Ostrowski 有限自动机稳定区。另一方面，前述三个结论已经在不诉诸任何全局自动机闭包的前提下，直接从分母刻度闭式导出 \(\alpha=e-2\) 的完整六相常数律与审计谱。故“局部分母精确输运”与“全局自动机闭合”之间存在严格分离。证毕。
\end{proof}

\begin{conjecture}[爆破模板的输运相图刚性]\label{conj:conclusion-kronecker-sparse-blowup-transport-phase-rigidity}
设
\[
\alpha=[0;c_1,\ldots,c_{j-1},b_1,c_{j+1},\ldots,c_r,
c_1,\ldots,c_{j-1},b_2,c_{j+1},\ldots,c_r,\ldots],
\]
其中 \((c_1,\ldots,c_r)\) 固定有界，而 \(b_k\to\infty\)。则分母刻度极限谱
\[
\operatorname{Acc}\bigl\{q_n\mathsf T_{q_n}(\alpha)\bigr\}
\]
仅由有界模板 \((c_1,\ldots,c_r)\) 与爆破位置 \(j\) 决定，而与 \(b_k\) 的具体增长律无关。定理 \ref{thm:conclusion-kronecker-sparse-blowup-universal-denominator-law} 给出模板 \((1,\ast,1)\) 的首个非平凡实例，而 \(\alpha=e-2\) 则提供了该 universality class 的自然超越端点。
\end{conjecture}
