\begin{theorem}[最大可见熵缺口与极大纤维的对数等值]\label{thm:conclusion-maxentropy-gap-equals-log-maxfiber}
沿用 bin-fold 口径，记
\[
\Fold_m^{\mathrm{bin}}:\Omega_m\to X_m,
\qquad
\mathfrak M_m:=\max_{x\in X_m}d_m^{\mathrm{bin}}(x),
\]
并定义
\[
\Delta H_m
:=
\sup_{\mu\in\Delta(\Omega_m)}
\Bigl(H(\mu)-H\bigl((\Fold_m^{\mathrm{bin}})_*\mu\bigr)\Bigr).
\]
则有严格恒等式
\[
\Delta H_m=\log \mathfrak M_m.
\]
\leanverified{paper\_conclusion\_maxentropy\_gap\_equals\_log\_maxfiber}
\end{theorem}
\begin{proof}
记
\[
\pi:=(\Fold_m^{\mathrm{bin}})_*\mu.
\]
把 \(\mu\) 按纤维 disintegrate 为
\[
\mu=\sum_{x\in X_m}\pi(x)\mu_x,
\qquad
\operatorname{supp}(\mu_x)\subseteq \Fold_m^{\mathrm{bin}}{}^{-1}(x).
\]
则 Shannon 链式分解给出
\[
H(\mu)
=
H(\pi)+\sum_{x\in X_m}\pi(x)H(\mu_x).
\]
由于
\[
H(\mu_x)\le \log d_m^{\mathrm{bin}}(x)\le \log \mathfrak M_m,
\]
遂得
\[
H(\mu)-H(\pi)\le \log \mathfrak M_m.
\]
因此
\[
\Delta H_m\le \log \mathfrak M_m.
\]

反过来，取 \(x_\ast\in X_m\) 使
\[
d_m^{\mathrm{bin}}(x_\ast)=\mathfrak M_m,
\]
并令 \(\mu_\ast\) 为纤维 \(\Fold_m^{\mathrm{bin}}{}^{-1}(x_\ast)\) 上的均匀分布。则
\[
(\Fold_m^{\mathrm{bin}})_*\mu_\ast=\delta_{x_\ast},
\qquad
H\bigl((\Fold_m^{\mathrm{bin}})_*\mu_\ast\bigr)=0,
\qquad
H(\mu_\ast)=\log \mathfrak M_m.
\]
故上界取等，即
\[
\Delta H_m=\log \mathfrak M_m.
\]
证毕。
\end{proof}

\begin{corollary}[精确反演阈值的熵化表达]\label{cor:conclusion-full-inversion-threshold-entropy-gap}
\leanverified{paper\_conclusion\_full\_inversion\_threshold\_entropy\_gap}
记
\[
B_m^\star
:=
\min\{B\in\ZZ_{\ge 0}:C_m^{\mathrm{bin}}(B)=2^m\}.
\]
则有
\[
B_m^\star
=
\left\lceil \log_2\mathfrak M_m\right\rceil
=
\left\lceil \frac{\Delta H_m}{\log 2}\right\rceil.
\]
亦即，最坏情形下的精确反演 bit 预算，恰是最大可见熵缺口的二进制化。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-foldbin-faithful-vs-fullinversion-exponential-separation}，
\[
B_m^\star=\left\lceil \log_2\mathfrak M_m\right\rceil.
\]
再由定理 \ref{thm:conclusion-maxentropy-gap-equals-log-maxfiber}，
\[
\Delta H_m=\log \mathfrak M_m.
\]
代入即得
\[
B_m^\star
=
\left\lceil \frac{\Delta H_m}{\log 2}\right\rceil.
\]
证毕。
\end{proof}

\begin{corollary}[精确反演存在严格的谱税下界]\label{cor:conclusion-full-inversion-positive-spectral-tax}
记
\[
N_m:=F_{m+2},
\qquad
\bar d_m:=\frac{2^m}{N_m}.
\]
则对任意 \(\varepsilon\in(0,C_\varphi)\)，存在 \(m_0(\varepsilon)\)，使得对一切 \(m\ge m_0(\varepsilon)\) 都有
\[
B_m^\star
\ge
\left\lceil
m-\log_2F_{m+2}+\log_2(1+C_\varphi-\varepsilon)
\right\rceil.
\]
因而
\[
\liminf_{m\to\infty}
\Bigl(
B_m^\star-m+\log_2F_{m+2}
\Bigr)
\ge
\log_2(1+C_\varphi).
\]
这表明纯鸽巢下界
\[
m-\log_2|X_m|
=
m-\log_2F_{m+2}
\]
对精确反演始终缺少一条严格正的谱税。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-fold-singlepair-visible-obstructions}，对充分大的 \(m\) 有
\[
\|p_m-u_m\|_{\ell^\infty}
\ge
\frac{C_\varphi-\varepsilon}{N_m}.
\]
另一方面，
\[
\|p_m-u_m\|_{\ell^\infty}
=
\frac{\mathfrak M_m-\bar d_m}{2^m},
\qquad
\bar d_m=\frac{2^m}{N_m}.
\]
故
\[
\mathfrak M_m-\bar d_m
\ge
\frac{2^m}{N_m}(C_\varphi-\varepsilon)
=
\bar d_m(C_\varphi-\varepsilon),
\]
即
\[
\mathfrak M_m\ge (1+C_\varphi-\varepsilon)\bar d_m.
\]
取对数并用推论 \ref{cor:conclusion-full-inversion-threshold-entropy-gap}，
\[
B_m^\star
=
\left\lceil \log_2\mathfrak M_m\right\rceil
\ge
\left\lceil
\log_2\bar d_m+\log_2(1+C_\varphi-\varepsilon)
\right\rceil.
\]
再代入
\[
\log_2\bar d_m=m-\log_2F_{m+2}
\]
即得第一式。令 \(m\to\infty\) 再令 \(\varepsilon\downarrow 0\)，得到所示 \(\liminf\) 下界。证毕。
\end{proof}

\begin{conjecture}[Fibonacci 共轭频对对精确反演税的饱和]\label{conj:conclusion-fibonacci-conjugate-pair-saturates-inversion-tax}
记
\[
S_m:=\max_{k\ne 0}\bigl|\widehat d_m(k)\bigr|.
\]
若非平凡最大 Fourier 幅度最终总由共轭 Fibonacci 频对
\[
\pm F_m\pmod{F_{m+2}}
\]
达到，即
\[
S_m=\bigl|\widehat d_m(F_m)\bigr|=\bigl|\widehat d_m(F_{m+1})\bigr|
\qquad
(m\gg 1),
\]
则应有精确渐近
\[
\mathfrak M_m
=
\frac{2^m}{F_{m+2}}\bigl(1+C_\varphi+o(1)\bigr),
\]
从而
\[
B_m^\star
=
m-\log_2F_{m+2}+\log_2(1+C_\varphi)+o(1).
\]
换言之，精确反演的全部额外常数成本应当最终由单个 Fibonacci 共轭频对完全饱和。
\end{conjecture}
