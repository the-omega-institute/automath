\subsection{软核例外谱的字迹分解、二项逆矩阵与尺度分层}\label{subsec:conclusion-softcore-wordtrace-binomial-inverse-scale-separation}
本小节在 \(q\)-体软核矩阵的对称幂框架下，给出 Kronecker 分解、全阶例外幂和的字迹闭式、Fibonacci 乘积律、重量表示矩阵的显式逆与行列式符号律，以及固定 \(m\) 时 \(q\)-方向递推和谱尺度分离估计。
设 \(q\ge 2\)，\(U_q:=\{0,1\}^q\)，并定义
\[
T^{(q)}_{u,v}:=\frac12\Bigl(1+\mathbf 1_{\{u\wedge v=0\}}\Bigr)\qquad (u,v\in U_q).
\]
记
\[
J:=\begin{pmatrix}1&1\\[2pt]1&1\end{pmatrix},
\qquad
K:=\begin{pmatrix}1&1\\[2pt]1&0\end{pmatrix},
\]
以及 \(\operatorname{Sym}^q(\CC^2)\subset(\CC^2)^{\otimes q}\) 为 \(S_q\)-不变对称张量子空间。

\begin{theorem}[Kronecker 与对称幂双分解]\label{thm:conclusion-softcore-kronecker-sympower-decomposition}
\leanverified{paper\_conclusion\_softcore\_kronecker\_sympower\_decomposition}
在上述记号下有严格恒等式
\[
T^{(q)}=\frac12\bigl(J^{\otimes q}+K^{\otimes q}\bigr).
\]
并且其在对称子空间上的限制满足
\[
T^{(q)}\big|_{\operatorname{Sym}^q(\CC^2)}
=\frac12\bigl(\operatorname{Sym}^q(J)+\operatorname{Sym}^q(K)\bigr).
\]
\end{theorem}
\begin{proof}
对任意单坐标 \((a,b)\in\{0,1\}^2\)，\(K_{a,b}=1\) 当且仅当 \((a,b)\neq(1,1)\)。
故
\[
\bigl(K^{\otimes q}\bigr)_{u,v}
=\prod_{t=1}^{q}K_{u_t,v_t}
=\mathbf 1_{\forall t,\,(u_t,v_t)\neq(1,1)}
=\mathbf 1_{\{u\wedge v=0\}}.
\]
又 \(\bigl(J^{\otimes q}\bigr)_{u,v}\equiv 1\)。
代入 \(T^{(q)}_{u,v}\) 的定义即得第一式。

由于 \(J^{\otimes q}\) 与 \(K^{\otimes q}\) 均与张量因子置换作用交换，故保持 \(\operatorname{Sym}^q(\CC^2)\)；
其在该子空间的限制分别为 \(\operatorname{Sym}^q(J)\) 与 \(\operatorname{Sym}^q(K)\)，第二式成立。证毕。
\end{proof}

\begin{theorem}[重量表示矩阵的显式整数逆]\label{thm:conclusion-softcore-weight-matrix-explicit-inverse}
令 \(B^{(q)}\in M_{q+1}(\QQ)\) 为
\(
T^{(q)}|_{\operatorname{Sym}^q(\CC^2)}
\)
在重量基 \((k=0,\dots,q)\) 下的矩阵，即
\[
B^{(q)}_{k,j}
=\frac12\binom{q}{j}+\frac12\binom{q-k}{j},
\qquad 0\le k,j\le q,
\]
并约定 \(\binom{n}{j}=0\)（当 \(j>n\)）。
则 \(B^{(q)}\) 可逆，且其逆矩阵满足
\[
\bigl(B^{(q)}\bigr)^{-1}_{r,j}=
\begin{cases}
-1,&(r,j)=(0,0),\\[3pt]
2,&(r,j)=(0,q),\\[3pt]
0,&r=0,\ 1\le j\le q-1,\\[3pt]
0,&1\le r\le q,\ j<q-r,\\[4pt]
2(-1)^{j-q+r}\binom{r}{q-j},&1\le r\le q,\ j\ge q-r.
\end{cases}
\]
\leanverified{paper\_conclusion\_softcore\_weight\_matrix\_explicit\_inverse}
\end{theorem}
\begin{proof}
设 Pascal 矩阵 \(P\) 与反序矩阵 \(R\) 为
\[
P_{n,j}:=\binom{n}{j},\qquad
(Rv)_k:=v_{q-k},
\]
并记 \(e_q\) 为第 \(q\) 个标准基向量、\(\mathbf 1\) 为全 \(1\) 向量。
对任意列向量 \(f\)，有
\[
(Pf)_n=\sum_{j=0}^{q}\binom{n}{j}f_j,\qquad
(RPf)_k=\sum_{j=0}^{q}\binom{q-k}{j}f_j,
\]
\[
\bigl(\mathbf 1\,e_q^{\mathsf T}Pf\bigr)_k
=\sum_{j=0}^{q}\binom{q}{j}f_j.
\]
故
\[
2B^{(q)}=(R+\mathbf 1 e_q^{\mathsf T})P.
\]

又 \(P^{-1}_{n,j}=(-1)^{n-j}\binom{n}{j}\)（\(n\ge j\)），\(R^{-1}=R\)，且 \(P^{-1}\mathbf 1=e_0\)。
对 \(R+\mathbf 1e_q^{\mathsf T}\) 使用 Sherman--Morrison 公式：
\[
(R+\mathbf 1e_q^{\mathsf T})^{-1}
=R-\frac12\,\mathbf 1e_0^{\mathsf T}.
\]
于是
\[
\bigl(B^{(q)}\bigr)^{-1}
=2P^{-1}\!\left(R-\frac12\,\mathbf 1e_0^{\mathsf T}\right)
=2P^{-1}R-e_0e_0^{\mathsf T}.
\]
逐项展开即得分段条目公式。证毕。
\end{proof}

\begin{theorem}[主谱与余谱的尺度分离]\label{thm:conclusion-softcore-spectral-scale-separation}

\(
T^{(q)}|_{\operatorname{Sym}^q(\CC^2)}

\[
\lambda_0^{(q)}\ge\lambda_1^{(q)}\ge\cdots\ge\lambda_q^{(q)}.

\lambda_0^{(q)}\le 2^{q-1}+\frac12\varphi^q,

\lambda_i^{(q)}\le\frac12\varphi^q\ \ (1\le i\le q),

\[
\lambda_q^{(q)}\ge
\frac12\min \sigma\!\bigl(\operatorname{Sym}^q(K)\bigr).

T^{(q)}\big|_{\operatorname{Sym}^q}

A_q:=\frac12\operatorname{Sym}^q(J),\ 
B_q:=\frac12\operatorname{Sym}^q(K).

\begin{proposition}[割线函数的二尺度自相似递推]\label{prop:conclusion-disjointness-secant-selfsimilar-recurrence}
\leanverified{paper\_conclusion\_disjointness\_secant\_selfsimilar\_recurrence}
对任意 \(q\ge 0\) 与任意 \(\lambda\)（避开极点）成立严格恒等式
\[
\mathcal F_{q+1}(\lambda)
=\frac{\alpha}{\varphi}\,\mathcal F_q\!\left(\frac{\lambda}{\varphi}\right)
\frac{\beta}{\psi}\,\mathcal F_q\!\left(\frac{\lambda}{\psi}\right).
\]
等价地，利用
\(
\frac{\alpha}{\varphi}=\frac12+\frac{3\sqrt5}{10}
\)
与
\(
\frac{\beta}{\psi}=\frac12-\frac{3\sqrt5}{10}
\)，可将递推写为无 \((\alpha,\beta)\) 形式。
\end{proposition}
\begin{proof}
利用节点缩放分裂
\[
\mu_i^{(q+1)}=\varphi\,\mu_i^{(q)}\ (0\le i\le q),
\qquad
\mu_{i+1}^{(q+1)}=\psi\,\mu_i^{(q)}\ (0\le i\le q),
\]
以及权重的 Pascal 分解
\[
w_i^{(q+1)}
=\binom{q+1}{i}\alpha^{q+1-i}\beta^i
=\alpha\,w_i^{(q)}+\beta\,w_{i-1}^{(q)}
\qquad(0\le i\le q+1),

将 \(\mathcal F_{q+1}(\lambda)=\sum_{i=0}^{q+1}w_i^{(q+1)}/(\lambda-d\mu_i^{(q+1)})\) 分拆为两段并作指标平移：

\[
\sum_{i=0}^{q}\frac{\alpha\,w_i^{(q)}}{\lambda-d\varphi\mu_i^{(q)}}
=\frac{\alpha}{\varphi}\sum_{i=0}^{q}\frac{w_i^{(q)}}{\lambda/\varphi-d\mu_i^{(q)}}
=\frac{\alpha}{\varphi}\,\mathcal F_q(\lambda/\varphi).
\]
第二段对应 \(\psi\mu_i^{(q)}\)，得到
\[
\sum_{i=0}^{q}\frac{\beta\,w_i^{(q)}}{\lambda-d\psi\mu_i^{(q)}}
=\frac{\beta}{\psi}\sum_{i=0}^{q}\frac{w_i^{(q)}}{\lambda/\psi-d\mu_i^{(q)}}
=\frac{\beta}{\psi}\,\mathcal F_q(\lambda/\psi).
\]
相加即得所述递推。证毕。
\end{proof}

并分别用 \(P_{q,a,b}(\lambda/\varphi)=G_{q,d}(\lambda/\varphi)(1-b\mathcal F_q(\lambda/\varphi))\)、
\(P_{q,a,b}(\lambda/\psi)=G_{q,d}(\lambda/\psi)(1-b\mathcal F_q(\lambda/\psi))\) 代换，即得所示递推。
初值由 \(q=0\) 时 \(B^{(0)}(a,b)=[a]\) 直接给出。证毕。
\end{proof}

\begin{proposition}[逐对刚性结果式：\texorpdfstring{$\sqrt5$}{sqrt5} 与 \texorpdfstring{$(\alpha,\beta)$}{(alpha,beta)} 的消隐]\label{prop:conclusion-disjointness-pairwise-resultant-rigidity}
\leanverified{paper\_conclusion\_disjointness\_pairwise\_resultant\_rigidity}
设 \(q\ge 2\) 且 \(0\le i\le \lfloor (q-1)/2\rfloor\)。
定义二次因子
\[
Q_{q,i}(\lambda):=\lambda^2-d(-1)^iL_{q-2i}\lambda+d^2(-1)^q,
\]
其根恰为 \(d\mu_i^{(q)}\) 与 \(d\mu_{q-i}^{(q)}\)（命题 \ref{prop:conclusion-lucas-quadratic-factorization} 在 \((S,P)=(1,-1)\) 的特化）。
则有严格闭式
\[
\Res\bigl(P_{q,a,b},Q_{q,i}\bigr)
=(-1)^{\frac{i(i+1)+(q-i)(q-i+1)}{2}}\,
b^{2}\,d^{2q}\,\binom{q}{i}^{2}\,
\left(\prod_{k=1}^{i}F_k\prod_{k=1}^{q-i}F_k\right)^{\!2}.
\]
若 \(q\) 为偶数，则 \(G_{q,d}\) 另含一次因子 \((\lambda-d(-1)^{q/2})\)，并且
\[
P_{q,a,b}\bigl(d(-1)^{q/2}\bigr)
=-(-1)^{\,\frac{q}{2}(q+1)-\binom{\frac{q}{2}+1}{2}}\,
b\,d^{q}\,\binom{q}{q/2}\,
\left(\prod_{k=1}^{q/2}F_k\right)^{\!2}.
\]
\end{proposition}
\begin{proof}
由节点取值公式 \eqref{eq:xi-two-param-disjointness-exceptional-poly-nodes}，
\[
P_{q,a,b}(d\mu_r^{(q)})
=-b\,d^{q}\,w_r\prod_{j\ne r}\bigl(\mu_r^{(q)}-\mu_j^{(q)}\bigr).
\]
对 \(r=i\) 与 \(r=q-i\) 相乘，并用
\(
w_iw_{q-i}=\binom{q}{i}^2(\alpha\beta)^q=\binom{q}{i}^2 5^{-q}
\)；
同时对差积使用 \eqref{eq:xi-two-param-disjointness-mu-vandermonde-fibonacci}，可得
\[
\prod_{j\ne i}\bigl(\mu_i^{(q)}-\mu_j^{(q)}\bigr)\,
\prod_{j\ne q-i}\bigl(\mu_{q-i}^{(q)}-\mu_j^{(q)}\bigr)
=(-1)^{\frac{i(i+1)+(q-i)(q-i+1)}{2}}\,
5^{q}\,
\left(\prod_{k=1}^{i}F_k\prod_{k=1}^{q-i}F_k\right)^{\!2},
\]
其中 \(\varphi\)-幂在共轭配对下抵消。
代回即得结果式闭式，并且 \(5^{q}\) 与 \(5^{-q}\) 精确相消，从而消去 \(\sqrt5\) 与 \((\alpha,\beta)\)。
\par
若 \(q\) 为偶数，取 \(r=q/2\) 则 \(\mu_{q/2}^{(q)}=(\varphi\psi)^{q/2}=(-1)^{q/2}\in\ZZ\)。
对该节点直接应用同一节点取值公式并使用 \(w_{q/2}=\binom{q}{q/2}(\alpha\beta)^{q/2}=\binom{q}{q/2}5^{-q/2}\)，
再与差积的 \((\sqrt5)^{q}\) 因子抵消，即得所示一次节点闭式。证毕。
\end{proof}

\begin{theorem}[停机不可判定性不可能内嵌于单个固定分辨率的 \(q\)-剖面]\label{thm:conclusion-halting-undecidability-externalized-from-fixedm-qprofile}
\leanverified{paper\_conclusion\_halting\_undecidability\_externalized\_from\_fixedm\_qprofile}
在定理 \ref{thm:conclusion-softcore-fixedm-qrecurrence-generic-minimality} 的 generic 假设下，对任意固定 \(m\) 的 \(q\)-向碰撞剖面 \(q\mapsto S_m(q)\) 同时成立：
\begin{enumerate}
  \item 整条序列属于有限阶 \(C\)-finite / rational 类，并已由连续窗口
  \[
  S_m(0),S_m(1),\dots,S_m(2r+3)
  \]
  唯一决定；
  \item 其完整非交换词迹谱
  \[
  \{\tau_1,\dots,\tau_r,\omega_{m,+},\omega_{m,-},\mu_m(\tau_1),\dots,\mu_m(\tau_r)\}
  \]
  可由上述有限窗口完全恢复；
  \item 当 \(q\to\infty\) 时，其高矩阶尾部进入单模主导 regime，并满足
  \[
  \Tr\!\Bigl(\bigl(T^{(q)}|_{\operatorname{Sym}^q(\CC^2)}\bigr)^m\Bigr)
  =
  \bigl(\lambda_0^{(q)}\bigr)^m
  \left(
  1+O\!\left(q\left(\frac{\varphi}{2}\right)^{qm}\right)
  \right).
  \]
\end{enumerate}
因此，定理 \ref{thm:conclusion-temperature-kernel-subexponential-perturbation-rigidity}、推论 \ref{cor:conclusion-temperature-kernel-free-energy-nonapproximable}、命题 \ref{prop:conclusion-reversible-aux-bits-equals-log-budget} 与命题 \ref{prop:conclusion-min-latches-equals-log-states} 中出现的停机型不可判定性，不可能来源于某个单独 fixed-\(m\) 的 \(q\)-向热力学剖面的内部不可压缩性；它只能沿实例族参数 \((M,x)\) 的变化被外置。
\end{theorem}
\begin{proof}
第一条由定理 \ref{thm:conclusion-softcore-fixedm-qseries-rational-ogf} 与定理 \ref{thm:conclusion-softcore-fixedm-qrecurrence-generic-minimality} 直接给出。第二条正是定理 \ref{thm:conclusion-softcore-fixedm-wordtrace-spectrum-window-recovery} 的内容。第三条则由定理 \ref{thm:conclusion-softcore-highq-one-mode-dominance} 给出。

故对每个固定 \(m\)，整条 \(q\)-剖面一方面已由有限窗口完全反演，另一方面在高矩阶上又被压缩到单一主模与指数小的余谱壳中。换言之，它属于一个有限可恢复、尾部单模化的可编译类，而不携带可支持停机不可判定性的内部 \(q\)-轴粗糙度。

与此相反，定理 \ref{thm:conclusion-temperature-kernel-subexponential-perturbation-rigidity} 与推论 \ref{cor:conclusion-temperature-kernel-free-energy-nonapproximable} 已把停机问题编码到实例族 \((M,x)\mapsto F_{M,x}\) 的二值谱缺口；命题 \ref{prop:conclusion-reversible-aux-bits-equals-log-budget} 与命题 \ref{prop:conclusion-min-latches-equals-log-states} 又把同一二值化外置到可逆辅助位指数与最小锁存器数。于是这些不可判定现象只能沿 family parameter 方向出现，而不能由某个单独 fixed-\(m\) 的 \(q\)-profile 内部产生。证毕。
\end{proof}

\end{conjecture}
