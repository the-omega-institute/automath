\subsection{局部化 torsion 轮廓、规范仿射固定点与 \texorpdfstring{$2$}{2}-进 Cantor 闭包}\label{subsec:conclusion-localized-torsion-godel-fixedpoint-cantor-geometry}
本小节把两条已在正文中分别定型的链进一步压缩为统一接口。
其一是局部化数系 \(G_S=\ZZ[S^{-1}]\) 及其 Pontryagin 对偶 \(\widehat{G_S}\) 的倍映射 torsion 轮廓；其二是规范 Gödel 半直积在 \(T_2(E_{\min})\) 上的 \(B\)-进仿射固定点与单向量地址像的自相似几何。
这两条链在结论层的共同特征是：有限核大小函数精确恢复素数账本，而规范 slice 的纯周期固定点在闭包中恰生成一条完全显式的 \(2\)-进 Cantor 子集。

\begin{theorem}[局部化对偶群的 torsion 轮廓完全恢复素数账本]\label{thm:conclusion-localized-dual-torsion-profile-reconstructs-S}
\leanverified{paper\_conclusion\_localized\_dual\_torsion\_profile\_reconstructs\_s}
设 \(S\subset \mathbb P\) 为有限素数集，令
\[
G_S:=\ZZ[S^{-1}],
\qquad
\Sigma_S:=\widehat{G_S}.
\]
对任意整数 \(n\ge 1\)，写
\[
n=n_S\,n_{S^c},
\qquad
n_S:=\prod_{p\in S}p^{v_p(n)},
\qquad
n_{S^c}:=\prod_{p\notin S}p^{v_p(n)}.
\]
记 \([n]:\Sigma_S\to \Sigma_S\) 为乘法端同态。则有
\[
\boxed{\;
[n]\ \text{总是满射},
\qquad
|\ker[n]|=n_{S^c}.\;}
\]
特别地，对任意素数 \(p\) 都有
\[
\boxed{\;
p\in S\iff |\ker[p]|=1,
\qquad
p\notin S\iff |\ker[p]|=p.\;}
\]
因此，素数账本 \(S\) 可以仅由函数
\[
n\longmapsto |\ker[n]|
\]
在 \(\Sigma_S\) 上完全恢复。
\end{theorem}
\begin{proof}
由于 \(G_S\subset \QQ\) 无挠，乘法 \(n:G_S\to G_S\) 总是单射；Pontryagin 对偶的反变等价遂给出 \([n]:\Sigma_S\to \Sigma_S\) 总是满射。

再看核大小。把 \(n\) 分解为
\[
n=n_{S^c}\cdot n_S.
\]
其中 \(n_S\in \ZZ[S^{-1}]^\times\) 是单位，故
\[
nG_S=n_{S^c}G_S.
\]
由定理 \ref{thm:conclusion-localized-groups-aut-and-finite-kernels} 中关于局部化数系商群的核公式，
\[
G_S/nG_S
=
G_S/n_{S^c}G_S
\cong
\ZZ/n_{S^c}\ZZ.
\]
对偶化后
\[
\ker[n]\cong \widehat{G_S/nG_S}
\cong
\widehat{\ZZ/n_{S^c}\ZZ},
\]
故 \(|\ker[n]|=n_{S^c}\)。
取 \(n=p\) 即得关于素数的判据。
最后，由 \(p\in S\iff p_{S^c}=1\) 与 \(p\notin S\iff p_{S^c}=p\)，可从 \(p\mapsto |\ker[p]|\) 唯一恢复 \(S\)。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$[B]$}{[B]} 动力学的周期点计数由 \texorpdfstring{$S$}{S}-外部分严格控制]\label{cor:conclusion-localized-dual-B-periodic-count-S-outside}
沿用定理 \ref{thm:conclusion-localized-dual-torsion-profile-reconstructs-S} 的记号，固定整数 \(B\ge 2\)。
则对每个 \(k\ge 1\)，\([B]\) 在 \(\Sigma_S\) 上的 \(k\)-步固定点个数满足
\[
\boxed{\;
\#\operatorname{Fix}([B]^k)
=
(B^k-1)_{S^c}
=
\prod_{p\notin S}p^{v_p(B^k-1)}.\;}
\]
相应的 Artin--Mazur 型动力学 \(\zeta\)-函数可写为
\[
\boxed{\;
\zeta_{S,B}(z)
:=
\exp\!\left(
\sum_{k\ge 1}\frac{z^k}{k}\,
\prod_{p\notin S}p^{v_p(B^k-1)}
\right).\;}
\]
特别地，当 \(B=2^L\) 且 \(2\in S\) 时，全部固定点计数完全由奇素数对 \(2^{Lk}-1\) 的分解控制。
\end{corollary}
\begin{proof}
由定义，
\[
\operatorname{Fix}([B]^k)=\ker([B]^k-\mathrm{id})=\ker([B^k-1]).
\]
将定理 \ref{thm:conclusion-localized-dual-torsion-profile-reconstructs-S} 应用于 \(n=B^k-1\) 即得
\[
\#\operatorname{Fix}([B]^k)=|(B^k-1)_{S^c}|=(B^k-1)_{S^c}.
\]
再代入 Artin--Mazur \(\zeta\)-函数定义即可。
若 \(B=2^L\) 且 \(2\in S\)，则 \(2^{Lk}-1\) 为奇数，从而 \(2\) 不贡献任何额外因子；全部固定点计数只由 \(S\)-外奇素数的分解决定。证毕。
\end{proof}

\begin{theorem}[局部化对偶到 profinite 群的连续同态全平凡]\label{thm:conclusion-localized-dual-no-profinite-hom}
沿用定理 \ref{thm:conclusion-localized-dual-torsion-profile-reconstructs-S} 的记号。设 \(K\) 为 profinite 群，且
\[
\Phi:\Sigma_S\longrightarrow K
\]
为连续群同态。则 \(\Phi\) 必为平凡同态。特别地，\(\Sigma_S\) 不存在任何非平凡有限连续商。
\end{theorem}
\begin{proof}
记
\[
I:=\Phi(\Sigma_S)\subseteq K.
\]
由于 \(\Sigma_S\) 为交换群，\(I\) 是紧交换群；又因 \(K\) 为 profinite 群，\(I\) 也是 profinite 交换群。把 \(\Phi\) 视为满同态
\[
\Sigma_S\twoheadrightarrow I,
\]
并作 Pontryagin 对偶，得到单射
\[
\widehat I\hookrightarrow G_S.
\]
另一方面，profinite 交换群的 Pontryagin 对偶总是离散挠群，故 \(\widehat I\) 为挠群。而
\[
G_S=\ZZ[S^{-1}]\subset \QQ
\]
是无挠交换群，因此其挠子群只能为零。于是 \(\widehat I=0\)，再由 Pontryagin 对偶的忠实性可知 \(I\) 只能是平凡群。故 \(\Phi\) 为平凡同态。

有限群是 profinite 群的特例，故最后一条随即成立。证毕。
\end{proof}

\begin{corollary}[经有限群压缩的局部化对偶证书必为常值]\label{cor:conclusion-localized-dual-finite-group-certificate-constant}
\leanverified{paper\_conclusion\_localized\_dual\_finite\_group\_certificate\_constant}
沿用定理 \ref{thm:conclusion-localized-dual-no-profinite-hom} 的记号。设 \(H\) 为有限群，\(F\) 为有限集合，并给定连续群同态
\[
\pi:\Sigma_S\longrightarrow H
\]
与任意映射
\[
\beta:H\longrightarrow F.
\]
则复合映射
\[
\beta\circ\pi:\Sigma_S\longrightarrow F
\]
必为常值。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-localized-dual-no-profinite-hom}，\(\pi\) 必为平凡同态，故 \(\pi(\Sigma_S)=\{e_H\}\)。因此
\[
(\beta\circ\pi)(x)=\beta(e_H)
\qquad(\forall\,x\in\Sigma_S),
\]
从而 \(\beta\circ\pi\) 为常值映射。证毕。
\end{proof}

\begin{theorem}[规范 slice 上每一层恰有 \texorpdfstring{$(M+1)^k$}{(M+1)^k} 个仿射固定点]\label{thm:conclusion-canonical-slice-exact-fixedpoint-count}
设
\[
T:=T_2(E_{\min})\cong \ZZ_2^2,
\qquad
B:=2^L>M,
\qquad
v\in T\setminus\{0\},
\]
并采用规范 Gödel 半直积在 \(T\) 上的仿射表示
\[
\Psi(r,k)(x)=B^k x+\operatorname{ev}_B(r),
\qquad
\operatorname{ev}_B(r):=\sum_{t\ge 1}r(p_t)B^{t-1}v,
\]
其中
\[
H_{\mathrm{can}}
:=
\left\{
(r,k)\in \NN^{(\mathbb P)}\times\NN:\ 
\operatorname{supp}(r)\subseteq\{p_1,\dots,p_k\},\
0\le r(p_t)\le M
\right\}.
\]
对固定 \(k\ge 1\)，记
\[
H_{\mathrm{can}}(k):=\{(r,k)\in H_{\mathrm{can}}\}.
\]
则对任意 \((r,k)\in H_{\mathrm{can}}(k)\)，仿射映射 \(\Psi(r,k)\) 在 \(T\) 上有且仅有一个固定点，显式为
\[
\boxed{\;
x_{r,k}
=(1-B^k)^{-1}\operatorname{ev}_B(r)
=\sum_{j\ge 0}B^{jk}\operatorname{ev}_B(r).\;}
\]
若写
\[
a_t:=r(p_t)\in\{0,\dots,M\}
\qquad(1\le t\le k),
\]
则该固定点具有纯周期 \(B\)-进地址展开
\[
\boxed{\;
x_{r,k}
=
\sum_{j\ge 0}\sum_{t=1}^{k}a_t\,B^{jk+t-1}v.\;}
\]
并且对固定 \(k\)，不同的 \(r\) 给出不同固定点，因此
\[
\boxed{\;
\#\{x\in T:\ \exists\,(r,k)\in H_{\mathrm{can}}(k),\ \Psi(r,k)(x)=x\}
=(M+1)^k.\;}
\]
\end{theorem}
\begin{proof}
固定 \((r,k)\in H_{\mathrm{can}}(k)\)。
由于 \(B=2^L\) 为 \(2\)-进非单位，故 \(B^k\in 2\ZZ_2\)；于是
\[
1-B^k\in \ZZ_2^\times
\]
是奇单位。因为 \(T\cong \ZZ_2^2\) 是自由 \(\ZZ_2\)-模，乘以 \(1-B^k\) 在 \(T\) 上可逆，所以方程
\[
x=B^k x+\operatorname{ev}_B(r)
\iff
(1-B^k)x=\operatorname{ev}_B(r)
\]
有唯一解，即
\[
x_{r,k}=(1-B^k)^{-1}\operatorname{ev}_B(r).
\]
再由 \(|B^k|_2<1\)，几何级数在 \(\ZZ_2\) 中收敛，并给出
\[
(1-B^k)^{-1}=\sum_{j\ge 0}B^{jk}.
\]
代入
\[
\operatorname{ev}_B(r)=\sum_{t=1}^{k}a_t B^{t-1}v
\]
便得
\[
x_{r,k}
=
\left(\sum_{j\ge 0}B^{jk}\right)\left(\sum_{t=1}^{k}a_tB^{t-1}v\right)
=
\sum_{j\ge 0}\sum_{t=1}^{k}a_t\,B^{jk+t-1}v,
\]
即纯周期展开式。

最后证明同层互异性。若 \(x_{r,k}=x_{s,k}\)，写
\[
b_t:=s(p_t)\in\{0,\dots,M\}
\qquad(1\le t\le k),
\]
则有两条纯周期 \(B\)-进展开
\[
x_{r,k}
=
\sum_{j\ge 0}\sum_{t=1}^{k}a_t\,B^{jk+t-1}v,
\qquad
x_{s,k}
=
\sum_{j\ge 0}\sum_{t=1}^{k}b_t\,B^{jk+t-1}v.
\]
由于 \(T\cong \ZZ_2^2\) 对 \(\ZZ_2\) 无挠且 \(v\neq 0\)，标量映射
\[
\ZZ_2\longrightarrow T,
\qquad
a\longmapsto av
\]
是单射，故上述两式在 \(T\) 中相等蕴含对应的标量级数在 \(\ZZ_2\) 中相等。
又全部 digits 都落在 \(\{0,\dots,M\}\subset [0,B-1]\) 内，故 \(B\)-进展开唯一性推出 \(a_t=b_t\) 对所有 \(t\) 成立，即 \(r=s\)。
由于每个 \(a_t\) 有 \(M+1\) 种选择，固定层 \(k\) 上的固定点总数恰为 \((M+1)^k\)。证毕。
\end{proof}

\begin{theorem}[规范仿射固定点的闭包是一条 \texorpdfstring{$2$}{2}-进自相似 Cantor 子集]\label{thm:conclusion-canonical-fixedpoints-cantor-closure}
沿用定理 \ref{thm:conclusion-canonical-slice-exact-fixedpoint-count} 的设定，定义规范固定点全集
\[
\operatorname{Fix}_{\mathrm{can}}
:=
\bigcup_{k\ge 1}
\left\{
x_{r,k}:\ (r,k)\in H_{\mathrm{can}}(k)
\right\}\subset T
\]
以及数字自相似集
\[
\mathcal C_{B,M}(v)
:=
\left\{
\sum_{n\ge 0}a_nB^n v:\ a_n\in\{0,\dots,M\}
\right\}\subset T.
\]
则成立：
\begin{enumerate}
  \item 闭包完全一致：
  \[
  \boxed{\;
  \overline{\operatorname{Fix}_{\mathrm{can}}}
  =
  \mathcal C_{B,M}(v).\;}
  \]
  \item 自相似分解：
  \[
  \boxed{\;
  \mathcal C_{B,M}(v)
  =
  \bigsqcup_{a=0}^{M}\bigl(av+B\,\mathcal C_{B,M}(v)\bigr).\;}
  \]
  \item 若在地址线 \(T_v:=\ZZ_2 v\cong \ZZ_2\) 上取标准 \(2\)-进超度量，则其 Hausdorff 维数与上 Minkowski 维数都存在且相等，并满足
  \[
  \boxed{\;
  \dim_{\mathrm H}\mathcal C_{B,M}(v)
  =
  \overline{\dim}_{\mathrm M}\mathcal C_{B,M}(v)
  =
  \frac{\log_2(M+1)}{L}.\;}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
先证闭包等式。
由定理 \ref{thm:conclusion-canonical-slice-exact-fixedpoint-count}，每个 \(x_{r,k}\) 都具有 digits 取自 \(\{0,\dots,M\}\) 的纯周期 \(B\)-进展开，故
\[
\operatorname{Fix}_{\mathrm{can}}\subseteq \mathcal C_{B,M}(v).
\]
反过来，任取
\[
x=\sum_{n\ge 0}a_nB^n v\in \mathcal C_{B,M}(v).
\]
对每个 \(k\ge 1\)，以首 \(k\) 个 digits 周期化，定义
\[
x^{(k)}:=
\sum_{j\ge 0}\sum_{t=0}^{k-1}a_t\,B^{jk+t}v.
\]
则 \(x^{(k)}\in \operatorname{Fix}_{\mathrm{can}}\)，因为它正对应于长度 \(k\) 的 digit block \((a_0,\dots,a_{k-1})\) 的纯周期重复。又 \(x^{(k)}\) 与 \(x\) 的前 \(k\) 个 \(B\)-进 digits 相同，所以
\[
x^{(k)}\longrightarrow x
\qquad
\text{于 }T_v\cong \ZZ_2 \text{ 的 }2\text{-进拓扑中}.
\]
故 \(\mathcal C_{B,M}(v)\subseteq \overline{\operatorname{Fix}_{\mathrm{can}}}\)，从而得到第 \(1\) 条。

第 \(2\) 条可由首位 digit 分解直接得到：任意 \(x\in \mathcal C_{B,M}(v)\) 唯一写成
\[
x=a_0v+B\sum_{n\ge 1}a_nB^{n-1}v,
\qquad
a_0\in\{0,\dots,M\},
\]
而括号中的第二项再次属于 \(\mathcal C_{B,M}(v)\)。
于是
\[
\mathcal C_{B,M}(v)=\bigcup_{a=0}^{M}\bigl(av+B\,\mathcal C_{B,M}(v)\bigr).
\]
由于 \(0\le a\le M<B\)，不同首 digit 对应不同的 \(B T_v\)-陪集，故该并实际上两两不交。

最后证明维数公式。
令 \(D:=\{0,\dots,M\}\)。则
\[
\mathcal C_{B,M}(v)
=
\left\{
\sum_{n\ge 0}d_nB^n v:\ d_n\in D
\right\},
\]
正是结论 \ref{con:xi-time-part62b-rankone-hologram-selfsimilar-measure-dimension} 中单向量全息像集 \(\mathcal X_D\) 的特例。
因此，该结论的第 \(1\) 条给出上面的互不相交自相似分解，而其第 \(3\) 条给出
\[
\dim_{\mathrm H}\mathcal C_{B,M}(v)
=
\dim_{\mathrm M}\mathcal C_{B,M}(v)
=
\frac{\log |D|}{\log B}.
\]
代入 \(|D|=M+1\) 与 \(B=2^L\) 即得
\[
\frac{\log(M+1)}{\log(2^L)}
=
\frac{\log_2(M+1)}{L}.
\]
证毕。
\end{proof}

\begin{theorem}[规范固定点层的最小周期分解与 Möbius 反演]\label{thm:conclusion-canonical-fixedpoint-primitive-mobius}
沿用定理 \ref{thm:conclusion-canonical-slice-exact-fixedpoint-count} 的记号。对每个 \(k\ge 1\) 记
\[
\mathcal F_k:=\{x_{r,k}:\ (r,k)\in H_{\mathrm{can}}(k)\}\subset T,
\]
并定义 primitive 层
\[
\Pi_k:=
\left\{
x\in \mathcal F_k:\ 
x\notin \mathcal F_d\ \text{对一切真因子 }d\mid k
\right\}.
\]
则成立：
\begin{enumerate}
  \item
  \[
  |\mathcal F_k|=(M+1)^k;
  \]
  \item
  \[
  \mathcal F_k=\bigsqcup_{d\mid k}\Pi_d,
  \qquad
  |\mathcal F_k|=\sum_{d\mid k}|\Pi_d|;
  \]
  \item Möbius 反演给出
  \[
  |\Pi_k|=\sum_{d\mid k}\mu(d)(M+1)^{k/d};
  \]
  \item 若 \(M\ge 1\)，则
  \[
  \lim_{k\to\infty}\frac{1}{k}\log |\Pi_k|=\log(M+1).
  \]
\end{enumerate}
\leanverified{paper\_conclusion\_canonical\_fixedpoint\_primitive\_mobius}
\end{theorem}
\begin{proof}
第 \(1\) 条正是定理 \ref{thm:conclusion-canonical-slice-exact-fixedpoint-count} 的计数公式。

对任意 \(x\in \mathcal F_k\)，定理 \ref{thm:conclusion-canonical-slice-exact-fixedpoint-count} 给出一个长度 \(k\) 的纯周期 digit block，而推论 \ref{cor:xi-time-part9g-fixed-point-primitive-word-classification} 表明：该纯周期块存在唯一的最小 primitive 长度 \(d\mid k\)，并且这一最小长度只依赖于点 \(x\) 本身。于是
\[
\mathcal F_k=\bigsqcup_{d\mid k}\Pi_d,
\]
从而得到第 \(2\) 条。

对计数恒等式
\[
(M+1)^k=|\mathcal F_k|=\sum_{d\mid k}|\Pi_d|
\]
应用 Möbius 反演，即得
\[
|\Pi_k|=\sum_{d\mid k}\mu(d)(M+1)^{k/d}.
\]
这证明了第 \(3\) 条。

最后设 \(M\ge 1\)。由上式，
\[
|\Pi_k|
=(M+1)^k+\sum_{\substack{d\mid k\\ d>1}}\mu(d)(M+1)^{k/d}.
\]
对每个真因子 \(d>1\) 都有 \(k/d\le k/2\)，故
\[
|\Pi_k|
=(M+1)^k+O\!\bigl(\tau(k)(M+1)^{k/2}\bigr),
\]
其中 \(\tau(k)\) 为约数函数。于是
\[
|\Pi_k|=(M+1)^k\bigl(1+o(1)\bigr),
\]
进而
\[
\frac{1}{k}\log|\Pi_k|\longrightarrow \log(M+1).
\]
证毕。
\end{proof}

\begin{corollary}[规范固定点目录的 Witt--Euler 乘积与有理 \texorpdfstring{$\zeta$}{zeta} 函数]\label{cor:conclusion-canonical-fixedpoint-witt-euler-rational-zeta}
沿用定理 \ref{thm:conclusion-canonical-fixedpoint-primitive-mobius} 的记号。定义 primitive 周期类数
\[
N_k:=\frac{|\Pi_k|}{k}
\qquad(k\ge 1).
\]
则 \(N_k\in\ZZ_{\ge 0}\)，且对每个 \(n\ge 1\) 都有
\[
(M+1)^n=\sum_{d\mid n}d\,N_d.
\]
因而成立 Witt--Euler 乘积
\[
\prod_{k\ge 1}(1-z^k)^{-N_k}=\frac{1}{1-(M+1)z}.
\]
等价地，规范固定点目录的 Artin--Mazur 型形式 \(\zeta\) 函数
\[
\mathfrak Z_{\mathrm{can}}(z)
:=
\exp\!\left(\sum_{n\ge 1}\frac{(M+1)^n}{n}z^n\right)
\]
恰满足
\[
\mathfrak Z_{\mathrm{can}}(z)=\frac{1}{1-(M+1)z}.
\]
特别地，其极点半径为 \((M+1)^{-1}\)，相应增长熵为 \(\log(M+1)\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-canonical-slice-exact-fixedpoint-count}，长度 \(k\) 的 digit block 与 \(\mathcal F_k\) 中的固定点一一对应。对 primitive block 施加循环群 \(C_k\) 的旋转作用，可得 \(\Pi_k\) 被分解为若干条长度恰为 \(k\) 的循环轨道；这是因为推论 \ref{cor:xi-time-part9g-fixed-point-primitive-word-classification} 已说明，primitive block 的最小周期正是 \(k\)。故 \(k\mid |\Pi_k|\)，从而 \(N_k\in\ZZ_{\ge 0}\)。

对任意 \(n\ge 1\)，长度 \(n\) 的全部块按照最小周期 \(d\mid n\) 分层；每条 primitive 周期类长度为 \(d\) 的轨道恰贡献 \(d\) 个不同的长度 \(n\) 块。故
\[
(M+1)^n=|\mathcal F_n|=\sum_{d\mid n}d\,N_d.
\]
于是
\[
\begin{aligned}
\log\prod_{k\ge 1}(1-z^k)^{-N_k}
&=
\sum_{k\ge 1}N_k\sum_{m\ge 1}\frac{z^{km}}{m}\\
&=
\sum_{n\ge 1}\frac{1}{n}\left(\sum_{d\mid n}d\,N_d\right)z^n\\
&=
\sum_{n\ge 1}\frac{(M+1)^n}{n}z^n\\
&=
-\log(1-(M+1)z).
\end{aligned}
\]
指数化即得所示 Witt--Euler 乘积与有理 \(\zeta\) 函数公式。最后一条由极点位置直接读出；这也与结论 \ref{con:xi-time-part62c-godel-tate-address-fullshift-zeta-entropy} 在 \(N=M+1\) 的全移位情形完全一致。证毕。
\end{proof}
