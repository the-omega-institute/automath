\begin{theorem}[固定径向窗内有限点状缺陷的不可隐身性]\label{thm:conclusion-comoving-defect-fixed-radial-window-nonhiding}
设有限点状缺陷按
\[
B=\prod_{j=1}^{\kappa} b_{a_j},
\qquad
a_j=C(\gamma_j+i\delta_j)\in \DD,
\qquad
\delta_j>0
\]
展开，并令对应共动扫描剖面为
\[
H(x)=\sum_{j=1}^{\kappa}\frac{4\delta_j}{(x-\gamma_j)^2+(1+\delta_j)^2}.
\]
若存在常数
\[
0<\delta_-\le \delta_j\le \delta_+<\infty
\qquad (1\le j\le \kappa),
\]
且 \(\kappa\ge 1\)，则必有统一可见性下界
\[
\|H\|_{L^1(\RR)}
\ge
4\pi\frac{\delta_-}{1+\delta_+}\,\kappa,
\]
以及与 \(\kappa\) 无关的峰值底噪下界
\[
\|H\|_{L^\infty(\RR)}
\ge
2\,\frac{\delta_-^2(1+\delta_-)}{\delta_+(1+\delta_+)^3}.
\]
\end{theorem}
\leanverified{paper\_conclusion\_comoving\_defect\_fixed\_radial\_window\_nonhiding}
\begin{proof}
正文已给出精确公式
\[
\|H\|_{L^1(\RR)}
=
4\pi\sum_{j=1}^{\kappa}\frac{\delta_j}{1+\delta_j},
\]
以及峰值下界
\[
\|H\|_{L^\infty(\RR)}
\ge
2\cdot
\frac{\sum_{j=1}^{\kappa}\delta_j^2(1+\delta_j)^{-3}}
{\sum_{j=1}^{\kappa}\delta_j(1+\delta_j)^{-1}}.
\]
由径向窗约束，
\[
\frac{\delta_j}{1+\delta_j}\ge \frac{\delta_-}{1+\delta_+},
\qquad
\frac{\delta_j^2}{(1+\delta_j)^3}\ge \frac{\delta_-^2}{(1+\delta_+)^3},
\qquad
\frac{\delta_j}{1+\delta_j}\le \frac{\delta_+}{1+\delta_-}.
\]
代回第一式即可得到
\[
\|H\|_{L^1(\RR)}
\ge
4\pi\kappa\frac{\delta_-}{1+\delta_+}.
\]
对第二式，分子至少为
\[
\kappa\,\frac{\delta_-^2}{(1+\delta_+)^3},
\]
分母至多为
\[
\kappa\,\frac{\delta_+}{1+\delta_-},
\]
故
\[
\|H\|_{L^\infty(\RR)}
\ge
2\,
\frac{\delta_-^2(1+\delta_-)}{\delta_+(1+\delta_+)^3}.
\]
证毕。
\end{proof}

\begin{corollary}[零剖面极限与零缺陷最终性的等价]\label{cor:conclusion-comoving-zero-profile-zero-defect-equivalence}
在定理 \ref{thm:conclusion-comoving-defect-fixed-radial-window-nonhiding} 的固定径向窗假设下，对任意一列有限点状缺陷族 \((H_\nu)\)，下列命题等价：
\[
\|H_\nu\|_{L^1(\RR)}\to 0,
\qquad
\|H_\nu\|_{L^\infty(\RR)}\to 0,
\qquad
\kappa_\nu=0\ \text{最终恒成立}.
\]
\end{corollary}
\begin{proof}
若 \(\kappa_\nu=0\) 最终成立，则 \(H_\nu\equiv 0\) 最终成立，前两条显然。

反之，若 \(\kappa_\nu\ge 1\) 沿某子列无穷多次发生，则由定理 \ref{thm:conclusion-comoving-defect-fixed-radial-window-nonhiding}，沿该子列恒有统一正下界
\[
\|H_\nu\|_{L^1(\RR)}\ge 4\pi\frac{\delta_-}{1+\delta_+}>0,
\qquad
\|H_\nu\|_{L^\infty(\RR)}
\ge
2\,\frac{\delta_-^2(1+\delta_-)}{\delta_+(1+\delta_+)^3}>0.
\]
故两种范数都不可能收敛到 \(0\)。证毕。
\end{proof}

\begin{theorem}[缺陷计数与缺陷定位之间的严格复杂度裂缝]\label{thm:conclusion-comoving-defect-count-vs-localization-complexity-gap}
在上述记号下，缺陷总数 \(\kappa\) 已可由单标量统计量给出双侧审计夹逼：
\[
\frac{1+\delta_-}{4\pi\delta_+}\,\|H\|_{L^1(\RR)}
\le
\kappa
\le
\frac{1+\delta_+}{4\pi\delta_-}\,\|H\|_{L^1(\RR)},
\]
并且
\[
\kappa
\ge
\frac{(1+\delta_-)^2}{4\delta_+}\,\|H\|_{L^\infty(\RR)}.
\]
然而，在一般位置与反混叠约束下，完整参数多重集
\[
\{(\gamma_j,\delta_j,c_j)\}_{j=1}^{\kappa}
\]
的精确恢复却恰需且仅需长度为 \(2\kappa\) 的样本前缀
\[
(u_0,u_1,\dots,u_{2\kappa-1}),
\]
而任何长度不超过 \(2\kappa-1\) 的前缀一般都不足以保证唯一恢复。
\end{theorem}
\begin{proof}
前两组不等式即正文对 \(\kappa\) 与 \(\|H\|_{L^1},\|H\|_{L^\infty}\) 给出的双侧审计界。

另一方面，对指数和模型
\[
u_n=\sum_{j=1}^{\kappa}c_jr_j^n
\]
正文已经证明：在一般位置下，长度 \(2\kappa\) 的前缀唯一锁定节点与权重，而任何长度不超过 \(2\kappa-1\) 的前缀都存在不同参数给出同一数据。故缺陷计数属于粗标量可审计层，而缺陷定位是 \(2\kappa\)-样本临界任务。证毕。
\end{proof}

\begin{theorem}[最小识别前缀同时是负方向的完整状态向量]\label{thm:conclusion-comoving-minimal-prefix-complete-negative-state}
设
\[
\mathcal U_{2\kappa}:=(u_0,u_1,\dots,u_{2\kappa-1}),
\qquad
\mathcal H_{\kappa-1}:=(u_{p+q})_{0\le p,q\le \kappa-1},
\qquad
G_{\kappa-1}:=\mathcal H_{\kappa-1}^*\mathcal H_{\kappa-1}.
\]
则在一般位置下，\(\mathcal U_{2\kappa}\) 同时决定下列三类对象：
\begin{enumerate}
  \item 完整缺陷参数多重集 \(\{(\gamma_j,\delta_j,c_j)\}_{j=1}^{\kappa}\)；
  \item Hankel Gram 块 \(G_{\kappa-1}\)；
  \item 对所有充分大的 \(N\)，Toeplitz 证书矩阵 \(T_N\) 的负指数
  \[
  \nu_-(T_N)=\kappa.
  \]
\end{enumerate}
更强地，Toeplitz 的全部负方向在合同意义下由 \(-G_{\kappa-1}\) 完全张成。
\end{theorem}
\begin{proof}
由正文关于 \(2\kappa\)-点锁定的定理，\(\mathcal U_{2\kappa}\) 唯一决定完整参数多重集。

又由于
\[
\mathcal H_{\kappa-1}=(u_{p+q})_{0\le p,q\le \kappa-1}
\]
只用到
\[
u_0,\dots,u_{2\kappa-2},
\]
故它已包含在 \(\mathcal U_{2\kappa}\) 中，从而 \(G_{\kappa-1}=\mathcal H_{\kappa-1}^*\mathcal H_{\kappa-1}\) 也被同一最小前缀唯一决定。

另一方面，由定理 \ref{thm:conclusion-visible-toeplitz-negative-gram-chart}，对充分大的 \(N\) 存在可逆矩阵 \(R_N\) 与正半定块 \(P_N\)，使
\[
R_N^*T_NR_N=
\begin{pmatrix}
P_N&0\\
0&-G_{\kappa-1}
\end{pmatrix}.
\]
因此 Toeplitz 的负方向、负指数以及其合同类型都由同一 Gram 块 \(-G_{\kappa-1}\) 完全决定；特别地，
\[
\nu_-(T_N)=\rank G_{\kappa-1}=\kappa
\]
对充分大的 \(N\) 成立。证毕。
\end{proof}

\begin{proposition}[最小前缀对几何参数与负 Gram 块同时具局部 Lipschitz 稳定性]\label{prop:conclusion-comoving-minimal-prefix-local-lipschitz-stability}
在最小分离
\[
|\gamma_j-\gamma_k|\ge s>0
\]
及紧径向窗
\[
\delta_j\in[\delta_-,\delta_+]\subset(0,\infty)
\]
的假设下，存在常数 \(C_1,C_2>0\)，使对任意两组满足同类约束的数据 \(\Theta,\widetilde\Theta\)，若记
\[
\varepsilon:=\max_{0\le n\le 2\kappa-1}|u_n(\Theta)-u_n(\widetilde\Theta)|,
\]
则有
\[
d_{\mathrm{match}}(\Theta,\widetilde\Theta)\le C_1\varepsilon,
\]
并且其负 Gram 块满足
\[
\|G_{\kappa-1}(\Theta)-G_{\kappa-1}(\widetilde\Theta)\|_{\mathrm{op}}
\le
C_2\varepsilon.
\]
\end{proposition}
\begin{proof}
正文已证明：在节点分离与径向有界的约束下，参数到有限谱指纹的映射在一般位置上局部双 Lipschitz，故第一式成立。

对第二式，记
\[
\mathcal H:=\mathcal H_{\kappa-1}(\Theta),
\qquad
\widetilde{\mathcal H}:=\mathcal H_{\kappa-1}(\widetilde\Theta),
\qquad
E:=\mathcal H-\widetilde{\mathcal H}.
\]
则 \(\mathcal H,\widetilde{\mathcal H}\) 的每个条目差都不超过 \(\varepsilon\)，故
\[
\|E\|_{\mathrm{op}}\le \kappa\varepsilon.
\]
又由于参数类在当前约束下是紧的，存在统一常数 \(M>0\) 使
\[
\|\mathcal H\|_{\mathrm{op}},\ \|\widetilde{\mathcal H}\|_{\mathrm{op}}\le M.
\]
于是
\[
\begin{aligned}
\|\mathcal H^*\mathcal H-\widetilde{\mathcal H}^*\widetilde{\mathcal H}\|_{\mathrm{op}}
&\le
\|\mathcal H^*E\|_{\mathrm{op}}+\|E^*\widetilde{\mathcal H}\|_{\mathrm{op}}\\
&\le
\bigl(\|\mathcal H\|_{\mathrm{op}}+\|\widetilde{\mathcal H}\|_{\mathrm{op}}\bigr)\|E\|_{\mathrm{op}}\\
&\le
2M\kappa\,\varepsilon.
\end{aligned}
\]
取 \(C_2:=2M\kappa\) 即得。证毕。
\end{proof}

\begin{theorem}[六重临界中心附近，统一条件数有界性强制秩一]\label{thm:conclusion-comoving-sextic-center-uniform-conditioning-forces-rankone}
设参数沿六重临界分歧尺度趋向中心：
\[
|u-1|\asymp \left|\alpha-\frac12\right|^{1/6},
\]
并假设各谱节点在 \(u=1\) 处横截裂分。若 \(\kappa\ge 2\)，则任意基于长度 \(2\kappa-1\) 的 Prony/Hankel 局部右逆，其 Lipschitz 常数 \(L(\alpha)\) 必满足
\[
L(\alpha)\ge C_\kappa\left|\alpha-\frac12\right|^{-(\kappa-1)/6}.
\]
因此，若某一审计方案要求在该中心附近保持统一条件数有界，则必有
\[
\kappa\le 1.
\]
\end{theorem}
\begin{proof}
正文关于六重临界分歧向有限节点反演病态传递的普适下界已经给出：在
\[
|u-1|\asymp \left|\alpha-\frac12\right|^{1/6}
\]
且节点横截裂分的前提下，任一相应局部右逆的 Lipschitz 常数都满足
\[
L(\alpha)\ge C_\kappa\left|\alpha-\frac12\right|^{-(\kappa-1)/6}.
\]
若 \(\kappa\ge 2\)，则右端随 \(\alpha\to \tfrac12\) 发散。故若要求在该中心附近保持统一条件数有界，则只能有
\[
\kappa\le 1.
\]
证毕。
\end{proof}

\begin{corollary}[代数可识别性与数值可识别性在临界中心处分裂]\label{cor:conclusion-comoving-algebraic-vs-numerical-identifiability-split}
对任意固定 \(\kappa\ge 2\)，长度 \(2\kappa\) 的最小前缀仍然在逐点意义上唯一决定缺陷参数；然而在六重临界中心附近，不存在一族在参数上统一 Lipschitz 的恢复映射。故在该共动扫描制度中存在严格分离：
\[
\text{代数可识别性}
\not\Rightarrow
\text{数值稳定可识别性}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-comoving-defect-count-vs-localization-complexity-gap}，长度 \(2\kappa\) 的最小前缀在一般位置下对参数恢复是代数上充分且最小的。由定理 \ref{thm:conclusion-comoving-sextic-center-uniform-conditioning-forces-rankone}，若 \(\kappa\ge 2\)，则任意局部恢复映射的 Lipschitz 常数都在临界中心附近发散。两者合并即得结论。证毕。
\end{proof}
