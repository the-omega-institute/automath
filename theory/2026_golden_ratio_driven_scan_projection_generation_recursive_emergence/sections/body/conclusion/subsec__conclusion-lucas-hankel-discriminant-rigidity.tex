\subsection{Lucas 幂 Hankel 主子式的判别式分解与刚性谱律}\label{subsec:conclusion-lucas-hankel-discriminant-rigidity}
本小节给出一组闭式可核验结论：Lucas 幂序列的 Hankel 主子式可写成二项系数、判别式与 Fibonacci 因子的完全乘积；对应对称幂特征多项式判别式与 Hankel 行列式之间存在无剩余恒等式；在归一化后，其离散曲率、Archimedean/\(p\)-adic 刚性与特征 \(p\) 下的 residue-lumping 都可压缩为显式的 Ramanujan 坐标；并在移位、特征 \(p\) 与黄金比例特例下给出刚性分类与 \(5\)-进显式公式。

\begin{definition}[Lucas/Fibonacci 型输入]\label{def:conclusion-lucas-fib-input}
设 \(S,P\in\ZZ\) 且 \(\Delta:=S^2-4P\neq 0\)，并假设 \(P=\pm 1\)。
令 \(\alpha,\beta\) 为
\[
x^2-Sx+P=0
\]
的两根。定义
\[
L_n:=\alpha^n+\beta^n,\qquad
F_n:=\frac{\alpha^n-\beta^n}{\alpha-\beta}\qquad (n\ge 0).
\]
固定 \(q\ge 1\)，置
\[
a_n:=L_n^q,\qquad
H_q:=(a_{i+j})_{0\le i,j\le q}.
\]
\end{definition}

\leanverified{paper\_conclusion\_lucas\_power\_hankel\_closed\_form}
\begin{theorem}[Lucas 幂 Hankel 主子式闭式]\label{thm:conclusion-lucas-power-hankel-closed-form}
在定义 \ref{def:conclusion-lucas-fib-input} 的设定下，有
\[
\det(H_q)
=
\left(\prod_{k=0}^{q}\binom{q}{k}\right)
\Delta^{\frac{q(q+1)}{2}}
\prod_{d=1}^{q}F_d^{\,2(q+1-d)}.
\]
特别地，在黄金比例特例 \((S,P)=(1,-1)\)（\(\Delta=5\)）中，
\[
\det(H_q)
=
\left(\prod_{k=0}^{q}\binom{q}{k}\right)
5^{\frac{q(q+1)}{2}}
\prod_{d=1}^{q}F_d^{\,2(q+1-d)}.
\]
\end{theorem}
\begin{proof}
由二项式展开
\[
a_n=(\alpha^n+\beta^n)^q
=\sum_{k=0}^{q}\binom{q}{k}r_k^n,\qquad
r_k:=\alpha^{q-k}\beta^k.
\]
令 \(c_k:=\binom{q}{k}\)，并定义 Vandermonde 矩阵
\[
V:=(r_k^i)_{0\le i,k\le q}.
\]
则
\[
H_q=V\,\mathrm{diag}(c_0,\dots,c_q)\,V^{\mathsf T},
\]
故
\[
\det(H_q)=\det(V)^2\prod_{k=0}^{q}c_k.
\]

记 \(s:=\beta/\alpha\)，则 \(r_k=\alpha^q s^k\)。从而
\[
\det(V)=\prod_{0\le i<j\le q}(r_j-r_i)
=\alpha^{q\binom{q+1}{2}}
\prod_{0\le i<j\le q}(s^j-s^i).
\]
几何节点的 Vandermonde 乘积恒等式给出
\[
\prod_{0\le i<j\le q}(s^j-s^i)
=
s^{\frac{q(q+1)(q-1)}{6}}
\prod_{d=1}^{q}(s^d-1)^{q+1-d}.
\]
因此
\[
\det(V)^2
=
\alpha^{q^2(q+1)}
s^{\frac{q(q+1)(q-1)}{3}}
\prod_{d=1}^{q}(s^d-1)^{2(q+1-d)}.
\]

又
\[
s^d-1
=\frac{\beta^d-\alpha^d}{\alpha^d}
=-\frac{(\alpha-\beta)F_d}{\alpha^d}.
\]
代入并整理得
\[
\det(V)^2
=
(\alpha\beta)^{\frac{q(q+1)(q-1)}{3}}
(\alpha-\beta)^{q(q+1)}
\prod_{d=1}^{q}F_d^{2(q+1-d)}.
\]
由于 \(P=\alpha\beta=\pm1\)，且
\[
\frac{q(q+1)(q-1)}{3}
\]
恒为偶数，故
\(
(\alpha\beta)^{q(q+1)(q-1)/3}=1
\)。
再用 \((\alpha-\beta)^2=\Delta\)，得
\[
\det(V)^2
=
\Delta^{\frac{q(q+1)}{2}}
\prod_{d=1}^{q}F_d^{2(q+1-d)}.
\]
与 \(\det(H_q)=\det(V)^2\prod_{k=0}^{q}c_k\) 合并即得结论。证毕。
\end{proof}

\begin{theorem}[对称幂特征多项式判别式分解]\label{thm:conclusion-sympower-discriminant-factorization}
在定义 \ref{def:conclusion-lucas-fib-input} 下，定义首一多项式
\[
\Pi_q(t):=\prod_{k=0}^{q}(t-r_k),
\qquad r_k=\alpha^{q-k}\beta^k.
\]
则
\[
\mathrm{Disc}(\Pi_q)
=
\Delta^{\frac{q(q+1)}{2}}
\prod_{d=1}^{q}F_d^{\,2(q+1-d)},
\]
并且
\[
\det(H_q)
=
\left(\prod_{k=0}^{q}\binom{q}{k}\right)\mathrm{Disc}(\Pi_q).
\]
\end{theorem}
\begin{proof}
由判别式定义
\[
\mathrm{Disc}(\Pi_q)
=\prod_{0\le i<j\le q}(r_i-r_j)^2
=\det(V)^2.
\]
再用定理 \ref{thm:conclusion-lucas-power-hankel-closed-form} 中已得的 \(\det(V)^2\) 闭式与
\[
\det(H_q)=\det(V)^2\prod_{k=0}^{q}\binom{q}{k}
\]
即得。证毕。
\end{proof}

\begin{proposition}[Lucas--二次分解与二次域闭包]\label{prop:conclusion-lucas-quadratic-factorization}
沿用定义 \ref{def:conclusion-lucas-fib-input} 的记号，并记
\(
r_k=\alpha^{q-k}\beta^k
\)（\(0\le k\le q\)）与
\(
\Pi_q(t)=\prod_{k=0}^{q}(t-r_k)
\)。
则在 \(\ZZ[t]\) 中有完全显式分解：
\[
\Pi_q(t)
=\bigl(t-P^{q/2}\bigr)^{\mathbf 1_{2\mid q}}
\prod_{k=0}^{\lfloor (q-1)/2\rfloor}
\Bigl(t^2-P^{k}L_{q-2k}\,t+P^{q}\Bigr).
\]
并且每个二次因子的判别式为
\[
\Disc\!\Bigl(t^2-P^{k}L_{q-2k}\,t+P^{q}\Bigr)
=\Delta\,F_{q-2k}^{\,2}.
\]
从而 \(\Pi_q\) 的分裂域包含于 \(\QQ(\sqrt{\Delta})\)，且除退化情形（\(F_{q-2k}=0\) 于所论基域）外，上述二次因子在 \(\QQ\) 上均不可约。
\end{proposition}
\begin{proof}
由定义，
\[
r_{q-k}=\alpha^{k}\beta^{q-k}.
\]
因此对 \(0\le k<q-k\) 有配对恒等式
\[
(t-r_k)(t-r_{q-k})
=t^2-(r_k+r_{q-k})t+r_kr_{q-k}.
\]
又
\[
r_kr_{q-k}=(\alpha\beta)^q=P^{q},
\]
并且
\[
r_k+r_{q-k}
=\alpha^{q-k}\beta^{k}+\alpha^{k}\beta^{q-k}
=(\alpha\beta)^{k}\bigl(\alpha^{q-2k}+\beta^{q-2k}\bigr)
=P^{k}L_{q-2k}.
\]
故得到二次因子 \(t^2-P^{k}L_{q-2k}t+P^{q}\)。当 \(q\) 为偶数且 \(k=q/2\) 时，
\(
r_{q/2}=(\alpha\beta)^{q/2}=P^{q/2}\in\ZZ
\)，对应一次因子 \(t-P^{q/2}\)。
将全部配对因子与可能的一次因子相乘即得分解式。
\par
判别式计算为
\[
\bigl(P^{k}L_{q-2k}\bigr)^2-4P^{q}
=L_{q-2k}^2-4P^{q-2k}
=L_{q-2k}^2-4P^{q-2k},
\]
而恒等式
\(
L_n^2-\Delta F_n^2=4P^{n}
\)
给出
\(
L_n^2-4P^n=\Delta F_n^2
\)。
令 \(n=q-2k\)（与 \(q\) 同奇偶）即得 \(\Delta F_{q-2k}^2\)。证毕。
\end{proof}

\begin{corollary}[尺度参数下的体谱多项式与判别式闭式]\label{cor:conclusion-scaled-spectrum-poly-discriminant}
在命题 \ref{prop:conclusion-lucas-quadratic-factorization} 的设定下，取 \(d\) 为一个标量参数，并定义首一多项式
\[
G_{q,d}(\lambda):=\prod_{k=0}^{q}\bigl(\lambda-d\,r_k\bigr)\in\ZZ[d][\lambda].
\]
则：
\begin{enumerate}
  \item \(G_{q,d}(\lambda)=d^{q+1}\Pi_q(\lambda/d)\)，因此其分裂域包含于 \(\QQ(\sqrt{\Delta})\)；
  \item 其判别式满足
  \[
  \Disc(G_{q,d})
  =d^{q(q+1)}\,\Disc(\Pi_q)
  =d^{q(q+1)}\,\Delta^{\frac{q(q+1)}{2}}
  \prod_{k=1}^{q}F_k^{\,2(q+1-k)}\in\ZZ[d].
  \]
  \item 对任意素数 \(p\)，其 \(p\)-进估值为
  \[
  v_p\!\bigl(\Disc(G_{q,d})\bigr)
  =q(q+1)\,v_p(d)
  +\frac{q(q+1)}{2}\,v_p(\Delta)
  +\sum_{k=1}^{q}2(q+1-k)\,v_p(F_k).
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
第一条由变量替换直接得到。
第二条中，判别式在首一多项式的线性缩放下满足
\(
\Disc(d^{q+1}\Pi_q(\lambda/d))=d^{q(q+1)}\Disc(\Pi_q)
\)，
再代入定理 \ref{thm:conclusion-sympower-discriminant-factorization} 的闭式即得。
第三条为对第二条逐项取 \(v_p\) 的直接推论。证毕。
\end{proof}

\begin{proposition}[移位 Hankel 行列式刚性]\label{prop:conclusion-shifted-hankel-rigidity}
\leanverified{paper\_conclusion\_shifted\_hankel\_rigidity}
对任意 \(r\ge 0\)，定义移位 Hankel 矩阵
\[
H_q^{(r)}:=(a_{i+j+r})_{0\le i,j\le q}.
\]
则
\[
\det(H_q^{(r)})
=
P^{\,r\cdot \frac{q(q+1)}{2}}
\det(H_q).
\]
因此：当 \(P=1\) 时 \(\det(H_q^{(r)})\) 与 \(r\) 无关；当 \(P=-1\) 时绝对值不变，仅符号按
\(
r\cdot q(q+1)/2
\)
的奇偶切换。
\end{proposition}
\begin{proof}
由
\[
a_{n+r}=\sum_{k=0}^{q} c_k r_k^{n+r}
=\sum_{k=0}^{q}(c_kr_k^r)r_k^n,
\]
得
\[
H_q^{(r)}=V\,\mathrm{diag}(c_0r_0^r,\dots,c_qr_q^r)\,V^{\mathsf T},
\]
故
\[
\det(H_q^{(r)})
=\det(V)^2\prod_{k=0}^{q}(c_kr_k^r)
=\det(H_q)\left(\prod_{k=0}^{q}r_k\right)^r.
\]
而
\[
\prod_{k=0}^{q}r_k
=\prod_{k=0}^{q}\alpha^{q-k}\beta^k
=(\alpha\beta)^{\frac{q(q+1)}{2}}
=P^{\frac{q(q+1)}{2}},
\]
代回即得。证毕。
\end{proof}

\begin{theorem}[特征 \texorpdfstring{$p$}{p} 下 Hankel 秩塌缩分类]\label{thm:conclusion-charp-hankel-rank-classification}
设 \(K\) 为域，\(\operatorname{char}(K)=0\) 或 \(\operatorname{char}(K)=p>q\)。
在定义 \ref{def:conclusion-lucas-fib-input} 中记
\[
s:=\beta/\alpha,\qquad m:=\operatorname{ord}(s)\in\NN\cup\{\infty\}.
\]
若 \(m<\infty\)，定义
\[
C_a:=\sum_{\substack{0\le k\le q\\ k\equiv a\ (\mathrm{mod}\ m)}}\binom{q}{k},
\qquad
a=0,\dots,m-1,
\]
并令
\[
M:=\#\{a\in\{0,\dots,m-1\}:C_a\neq 0\}.
\]
则序列 \(a_n=L_n^q\) 的 Hankel 秩满足
\[
\operatorname{rank}_H(a)=
\begin{cases}
q+1,& m=\infty,\\
q+1,& m>q,\\
M,& m\le q.
\end{cases}
\]
且在 \(m\le q\) 情形有合并表示
\[
a_n=\sum_{a=0}^{m-1}C_a(\alpha^q s^a)^n,
\]
其非零项数恰为 \(M\)；因此存在非零 \(M\times M\) Hankel 主子式，而任意 \((M+1)\times(M+1)\) Hankel 主子式恒为零。
\end{theorem}
\begin{proof}
基础表示
\[
a_n=\sum_{k=0}^{q}\binom{q}{k}(\alpha^q s^k)^n
\]
已给出。
当 \(m=\infty\) 或 \(m>q\) 时，\(\{s^k\}_{k=0}^{q}\) 两两不同，且 \(\operatorname{char}(K)=0\) 或 \(p>q\) 保证 \(\binom{q}{k}\neq 0\)。
因此有 \(q+1\) 个不同节点、非零权重的指数和表示，Hankel 秩为 \(q+1\)。

当 \(m\le q\) 时，\(s^{k+m}=s^k\)，按同余类合并得
\[
a_n=\sum_{a=0}^{m-1}C_a(\alpha^q s^a)^n.
\]
删去 \(C_a=0\) 项后得到 \(M\) 个互异节点的指数和表示，故 Hankel 秩为 \(M\)。
同理，\(M\) 阶主块可写为 \(V_M\mathrm{diag}(C_{a_1},\dots,C_{a_M})V_M^{\mathsf T}\)（其中 \(a_j\) 枚举非零类），行列式非零；而高一阶主子式因递推阶不超过 \(M\) 必为零。证毕。
\end{proof}

\begin{proposition}[最小递推与倒易对称]\label{prop:conclusion-lucas-power-minpoly-reciprocity}
在定义 \ref{def:conclusion-lucas-fib-input} 下：
\begin{enumerate}
  \item 序列 \(a_n=L_n^q\) 满足以 \(\Pi_q\) 为湮灭多项式的线性递推
  \[
  a_{n+q+1}
  -e_1 a_{n+q}
  +e_2 a_{n+q-1}
  -\cdots
  +(-1)^{q+1}e_{q+1}a_n=0,
  \]
  其中 \(e_j=e_j(r_0,\dots,r_q)\) 为基本对称多项式。
  若 \(\prod_{d=1}^{q}F_d\neq 0\)（等价于 \(r_0,\dots,r_q\) 两两不同），则该阶数 \(q+1\) 最小。
  \item \(\Pi_q\) 满足严格倒易对称
  \[
  t^{q+1}\Pi_q(P^q/t)
  =
  (-1)^{q+1}P^{\frac{q(q+1)}{2}}\Pi_q(t).
  \]
  等价地，若 \(\rho\) 为 \(\Pi_q\) 的根，则 \(P^q/\rho\) 亦为其根（同重数）。
\end{enumerate}
\end{proposition}
\begin{proof}
第一条由指数和表示直接得到：每个 \(r_k^n\) 被 \(\Pi_q(\mathsf S)\) 湮灭，线性组合同理。
当节点互异时，Hankel 秩为 \(q+1\)，故最小阶即 \(q+1\)。

第二条由
\[
r_k r_{q-k}=P^q
\]
与根集封闭性给出。直接计算
\[
t^{q+1}\Pi_q(P^q/t)
=\prod_{k=0}^{q}(P^q-r_kt)
=(-1)^{q+1}\left(\prod_{k=0}^{q}r_k\right)\Pi_q(t),
\]
再用
\(
\prod_{k=0}^{q}r_k=P^{q(q+1)/2}
\)
即得。证毕。
\end{proof}

\begin{theorem}[黄金比例特例的 \texorpdfstring{$5$}{5}-进赋值闭式]\label{thm:conclusion-golden-lucas-hankel-v5-rigidity}
在黄金比例特例 \((S,P)=(1,-1)\) 下，记
\[
D_q:=\det\!\big(L_{i+j}^{\,q}\big)_{0\le i,j\le q}.
\]
则
\[
v_5(D_q)
=
v_5\!\left(\prod_{k=0}^{q}\binom{q}{k}\right)
+\frac{q(q+1)}{2}
+2\sum_{d=1}^{q}(q+1-d)\,v_5(F_d).
\]
若进一步使用经典恒等式
\[
v_5(F_d)=
\begin{cases}
v_5(d),&5\mid d,\\
0,&5\nmid d,
\end{cases}
\]
则可化为
\[
v_5(D_q)
=
v_5\!\left(\prod_{k=0}^{q}\binom{q}{k}\right)
+\frac{q(q+1)}{2}
+2\sum_{m=1}^{\lfloor q/5\rfloor}(q+1-5m)\,v_5(5m).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-lucas-power-hankel-closed-form} 在 \(\Delta=5\) 下取 \(v_5\) 即得第一式。
第二式仅将加权和按 \(d=5m\) 与 \(5\nmid d\) 分组改写。证毕。
\end{proof}

\begin{corollary}[黄金比例特例 Hankel 主子式的 \texorpdfstring{$5$}{5}-进二次增长]\label{cor:conclusion-golden-lucas-hankel-v5-quadratic-growth}
在定理 \ref{thm:conclusion-golden-lucas-hankel-v5-rigidity} 的记号下，当 \(q\to\infty\) 时有
\[
v_5(D_q)=\frac34\,q^2+O(q\log q).
\]
因此 \(D_q\) 的 \(5\)-进规模满足 \(v_5(D_q)=(\frac34+o(1))q^2\)，并在该特例下诱导 Hankel/Prony 型恢复的超指数级病态性（除非显式引入非阿基米德分解或结构性消去）。
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-golden-lucas-hankel-v5-rigidity}，记
\(
B_q:=\prod_{k=0}^{q}\binom{q}{k}
\)
与 \(M:=\lfloor q/5\rfloor\)，则
\[
v_5(D_q)=v_5(B_q)+\frac{q(q+1)}{2}
+2\sum_{m=1}^{M}(q+1-5m)\,v_5(5m).
\]
先估计 \(v_5(B_q)\)。
由恒等式
\(
B_q=(q!)^{q+1}/\prod_{k=0}^{q}(k!)^2
\)
得
\[
v_5(B_q)=(q+1)v_5(q!)-2\sum_{k=0}^{q}v_5(k!).
\]
对 \(v_5(n!)\) 施用 Legendre 公式 \(v_5(n!)=(n-s_5(n))/4\)（\(s_5(n)\) 为 \(5\)-进数字和），并用 \(\sum_{k=0}^{q}k=q(q+1)/2\) 可消去二次主项，得到
\[
v_5(B_q)=\frac12\sum_{k=0}^{q}s_5(k)-\frac{q+1}{4}\,s_5(q)=O(q\log q).
\]
\par
再对加权和使用 \(v_5(5m)=1+v_5(m)\)：
\[
\sum_{m=1}^{M}(q+1-5m)\,v_5(5m)
=\sum_{m=1}^{M}(q+1-5m)
\;+\;
\sum_{m=1}^{M}(q+1-5m)\,v_5(m).
\]
其中
\(
\sum_{m=1}^{M}(q+1-5m)=M(q+1)-\frac{5}{2}M(M+1)=\frac{q^2}{10}+O(q)
\)。
并且由分解 \(v_5(m)=\sum_{j\ge 1}\ind[5^j\mid m]\) 得到
\[
\sum_{m\le M}v_5(m)=\sum_{j\ge 1}\Bigl\lfloor\frac{M}{5^j}\Bigr\rfloor=\frac{M}{4}+O(\log M),
\qquad
\sum_{m\le M}m\,v_5(m)=\frac{M^2}{8}+O(M\log M).
\]
从而
\[
\sum_{m=1}^{M}(q+1-5m)\,v_5(m)
=(q+1)\sum_{m\le M}v_5(m)-5\sum_{m\le M}m\,v_5(m)
=\frac{q^2}{40}+O(q\log q).
\]
综上
\(
2\sum_{m=1}^{M}(q+1-5m)v_5(5m)=\frac{q^2}{4}+O(q\log q)
\)，
并与 \(\frac{q(q+1)}{2}=\frac12q^2+O(q)\)、\(v_5(B_q)=O(q\log q)\) 合并，即得
\(
v_5(D_q)=\frac34q^2+O(q\log q)
\)。
证毕。
\end{proof}
