\subsection{Lee--Yang 四重覆盖的双观测量读出、最小采样可辨识与扭曲 \texorpdfstring{$\zeta$}{zeta} 统一}\label{subsec:conclusion-leyang-s4-double-observable-chebotarev-sampling-zeta}
沿用定理 \ref{thm:xi-terminal-zm-leyang-elliptic-structure}、结论 \ref{con:xi-terminal-zm-generic-galois-s4} 与结论 \ref{con:xi-terminal-zm-s4-modp-splitting-density} 的记号，本文已得到 Lee--Yang 四次谱覆盖在几何单值群意义下的 \(S_4\) 刚性。
本小节将该刚性进一步压缩为一组可直接读出的有限域观测量，并将其与时间可辨识、圆维下界及扭曲 \(\zeta\) 的局部因子框架并联起来。

\begin{theorem}[纤维根数与判别式符号构成的 \texorpdfstring{$S_4$}{S4} 共轭类完备判别]\label{thm:conclusion-leyang-s4-double-observable-classification}
设 \(\pi_y:E\to\PP^1_y\) 为上述 \(S_4\)-四重覆盖，取奇素数 \(p\) 使覆盖在 \(\FF_p\) 上处于驯良约化，并记
\[
U_p:=\PP^1(\FF_p)\setminus B(\FF_p).
\]
对任意 \(y\in U_p\)，定义
\[
N_p(y):=\card{\{\lambda\in\FF_p:\ \Pi(\lambda,y)=0\}}\in\{0,1,2,4\},
\]
并令
\[
\varepsilon_p(y):=
\left(\frac{\Disc_\lambda(\Pi(\lambda,y))}{p}\right)\in\{\pm1\}.
\]
记 \(\sigma_{p,y}\in S_4\) 为 \(y\) 处几何纤维上的 Frobenius 置换，则二元观测量
\[
\bigl(N_p(y),\varepsilon_p(y)\bigr)
\]
唯一决定 \(\sigma_{p,y}\) 的共轭类，且对应关系为
\begin{equation}\label{eq:conclusion-leyang-s4-double-observable-class-map}
\begin{aligned}
N_p(y)=4
&\Longleftrightarrow
\sigma_{p,y}\sim (1)(1)(1)(1),\\
N_p(y)=2
&\Longleftrightarrow
\sigma_{p,y}\sim (2)(1)(1),\\
N_p(y)=1
&\Longleftrightarrow
\sigma_{p,y}\sim (3)(1),\\
N_p(y)=0,\ \varepsilon_p(y)=+1
&\Longleftrightarrow
\sigma_{p,y}\sim (2)(2),\\
N_p(y)=0,\ \varepsilon_p(y)=-1
&\Longleftrightarrow
\sigma_{p,y}\sim (4).
\end{aligned}
\end{equation}
\end{theorem}
\begin{proof}
对 \(S_4\) 在四点上的自然作用，不动点数分别为：恒等元 \(4\)、换位 \(2\)、三循环 \(1\)、双换位与四循环均为 \(0\)。
故当 \(N_p(y)\in\{4,2,1\}\) 时，共轭类已由不动点数唯一确定，仅 \(N_p(y)=0\) 仍有 \((2)(2)\) 与 \((4)\) 两类歧义。

设 \(\alpha_1,\dots,\alpha_4\) 为 \(\Pi(\lambda,y)\) 在代数闭包中的四根，令
\[
\Delta:=\prod_{i<j}(\alpha_i-\alpha_j),\qquad
\Disc_\lambda(\Pi(\lambda,y))=\Delta^2.
\]
Frobenius 在根上的置换为 \(\sigma_{p,y}\)，故
\[
\mathrm{Frob}_p(\Delta)=\mathrm{sgn}(\sigma_{p,y})\,\Delta.
\]
另一方面，\(\mathrm{Frob}_p\) 在分裂域中与 \(x\mapsto x^p\) 一致，故
\[
\Delta^p=\mathrm{sgn}(\sigma_{p,y})\,\Delta,\qquad
\Delta^{p-1}=\mathrm{sgn}(\sigma_{p,y}).
\]
两边平方后得到
\[
\Disc_\lambda(\Pi(\lambda,y))^{\frac{p-1}{2}}
=\Delta^{p-1}
=\mathrm{sgn}(\sigma_{p,y})\in\{\pm1\},
\]
即 \(\varepsilon_p(y)=\mathrm{sgn}(\sigma_{p,y})\)。
因 \((2)(2)\) 为偶置换而 \((4)\) 为奇置换，\(N_p(y)=0\) 的两类由 \(\varepsilon_p(y)\) 完全分离，从而式 \eqref{eq:conclusion-leyang-s4-double-observable-class-map} 成立。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$S_4$}{S4} 不可约特征的可观测多项式化]\label{thm:conclusion-leyang-s4-character-polynomial-observable}
在定理 \ref{thm:conclusion-leyang-s4-double-observable-classification} 的记号下，置
\[
X_p(y):=N_p(y)-1.
\]
则 \(S_4\) 的全部五个不可约特征均可写成 \((X_p(y),\varepsilon_p(y))\) 的整系数多项式，并可取
\[
\chi_{\mathrm{triv}}=1,\qquad
\chi_{\mathrm{sgn}}=\varepsilon_p(y),\qquad
\chi_{\mathrm{std}}=X_p(y),
\]
\[
\chi_{\mathrm{std}\otimes\mathrm{sgn}}
=\varepsilon_p(y)\,X_p(y),
\qquad
\chi_{2}
=X_p(y)^2-X_p(y)\bigl(1+\varepsilon_p(y)\bigr)-1.
\]
因此，\((N_p(y),\varepsilon_p(y))\) 对 \(S_4\) 共轭类函数环给出完备充分统计：任一不可约局部 Artin 迹均为该二元观测量的代数读出。
\end{theorem}
\begin{proof}
四点置换表示的特征即不动点数 \(N_p(y)\)，故
\[
\chi_{\mathrm{std}}=N_p(y)-1=X_p(y).
\]
符号扭曲表示满足
\[
\chi_{\mathrm{std}\otimes\mathrm{sgn}}
=\chi_{\mathrm{std}}\chi_{\mathrm{sgn}}
=\varepsilon_p(y)\,X_p(y).
\]

对二维不可约表示，定理 \ref{thm:conclusion-leyang-s4-double-observable-classification} 表明共轭类可由五个观测点
\[
(X,\varepsilon)\in
\{(3,+1),(1,-1),(0,+1),(-1,+1),(-1,-1)\}
\]
唯一标识。
将
\[
F(X,\varepsilon):=X^2-X(1+\varepsilon)-1
\]
代入上述五点，可得取值
\[
2,\ 0,\ -1,\ 2,\ 0,
\]
与 \(S_4\) 二维不可约特征表一致，故 \(\chi_2=F(X_p(y),\varepsilon_p(y))\)。
其余结论随之成立。证毕。
\end{proof}

\begin{corollary}[Lee--Yang 四重覆盖的联合 Chebotarev 统计律]\label{cor:conclusion-leyang-s4-joint-chebotarev-double-observable}
设
\[
U_p:=\PP^1(\FF_p)\setminus B(\FF_p),
\]
并令 \(p\) 避开坏约化与分歧碰撞的有限素集。
对每一组可实现观测对 \((n,\varepsilon)\in\{4,2,1,0\}\times\{\pm1\}\)，有
\[
\card{\{y\in U_p:\ N_p(y)=n,\ \varepsilon_p(y)=\varepsilon\}}
=\frac{|C_{n,\varepsilon}|}{|S_4|}\,p+O(\sqrt p),
\]
其中 \(C_{n,\varepsilon}\subset S_4\) 为式 \eqref{eq:conclusion-leyang-s4-double-observable-class-map} 对应的共轭类。
特别地，联合极限分布为
\[
\mathrm{Prob}(N=4,\varepsilon=+1)=\frac1{24},\qquad
\mathrm{Prob}(N=2,\varepsilon=-1)=\frac14,
\]
\[
\mathrm{Prob}(N=1,\varepsilon=+1)=\frac13,\qquad
\mathrm{Prob}(N=0,\varepsilon=+1)=\frac18,\qquad
\mathrm{Prob}(N=0,\varepsilon=-1)=\frac14.
\]
从而在无 \(\FF_p\)-点纤维 \(N=0\) 条件下，
\[
\mathrm{Prob}(\varepsilon=+1\mid N=0)=\frac13,\qquad
\mathrm{Prob}(\varepsilon=-1\mid N=0)=\frac23.
\]
\end{corollary}
\begin{proof}
由结论 \ref{con:xi-terminal-zm-s4-modp-splitting-density}，Frobenius 共轭类在未分歧参数上按 Chebotarev 密度等分布，且误差为 \(O(\sqrt p)\)。
再由定理 \ref{thm:conclusion-leyang-s4-double-observable-classification} 的双观测量--共轭类对应，可将共轭类密度直接推送为 \((N,\varepsilon)\) 的联合密度。
显式概率由 \(S_4\) 五类基数
\[
1,\ 6,\ 8,\ 3,\ 6
\]
归一化得到。
条件概率由 \((2)(2)\) 与 \((4)\) 两类在 \(N=0\) 事件中的相对基数比 \(3:6\) 直接给出。证毕。
\end{proof}

\begin{theorem}[未分歧四次纤维根数的 quartic Poisson 伪装]\label{thm:conclusion-leyang-unramified-fiber-count-quartic-poisson-cloak}
沿用推论 \ref{cor:conclusion-leyang-s4-joint-chebotarev-double-observable} 的记号。
设
\[
U_p^{\mathrm{aff}}
:=
\FF_p\setminus \bigl(B(\overline{\FF}_p)\cap \mathbb A^1_y\bigr)
=
\{y_0\in \FF_p:\ y_0(y_0-1)P_{\mathrm{LY}}(y_0)\neq 0\},
\]
并在 \(U_p^{\mathrm{aff}}\) 上取均匀分布。
对
\[
N_p(y_0):=\card{\{\lambda\in \FF_p:\ \Pi(\lambda,y_0)=0\}}
\]
的概率生成函数有
\[
\frac{1}{|U_p^{\mathrm{aff}}|}
\sum_{y_0\in U_p^{\mathrm{aff}}} z^{N_p(y_0)}
=
\frac38+\frac13 z+\frac14 z^2+\frac1{24}z^4+O_z(p^{-1/2}).
\]
特别地，令 \(z=1+t\)，则
\[
\frac{1}{|U_p^{\mathrm{aff}}|}
\sum_{y_0\in U_p^{\mathrm{aff}}} (1+t)^{N_p(y_0)}
=
1+t+\frac{t^2}{2}+\frac{t^3}{6}+\frac{t^4}{24}+O_t(p^{-1/2}).
\]
从而对任意次数不超过 \(4\) 的多项式 \(Q\in \CC[T]\)，都有
\[
\frac{1}{|U_p^{\mathrm{aff}}|}
\sum_{y_0\in U_p^{\mathrm{aff}}} Q\!\bigl(N_p(y_0)\bigr)
=
\EE_{\mathrm{Pois}(1)}[Q(N)]+O_Q(p^{-1/2}).
\]
\end{theorem}
\begin{proof}
由结论 \ref{con:xi-terminal-zm-s4-modp-fiber-rootcount-fixedpoints}，
\[
\PP(N_p=0)=\frac38+O(p^{-1/2}),\quad
\PP(N_p=1)=\frac13+O(p^{-1/2}),\quad
\PP(N_p=2)=\frac14+O(p^{-1/2}),\quad
\PP(N_p=4)=\frac1{24}+O(p^{-1/2}),
\]
且 \(N_p=3\) 仅在分歧集合的模 \(p\) 化上出现，故其贡献为 \(O(p^{-1})\)，可吸收到 \(O(p^{-1/2})\) 误差中。
于是概率生成函数展开立得
\[
\frac{1}{|U_p^{\mathrm{aff}}|}
\sum_{y_0\in U_p^{\mathrm{aff}}} z^{N_p(y_0)}
=
\frac38+\frac13 z+\frac14 z^2+\frac1{24}z^4+O_z(p^{-1/2}).
\]

把 \(z=1+t\) 代入，直接计算得
\[
\frac38+\frac13(1+t)+\frac14(1+t)^2+\frac1{24}(1+t)^4
=
1+t+\frac{t^2}{2}+\frac{t^3}{6}+\frac{t^4}{24}.
\]
右端恰是 \(e^t\) 在四阶处的截断。
因此前四阶下降阶乘矩
\[
\EE\bigl[(N_p)_k\bigr]
\qquad (1\le k\le 4)
\]
与 \(\mathrm{Pois}(1)\) 一致，误差为 \(O(p^{-1/2})\)。
而 \(\{(T)_k:0\le k\le 4\}\) 张成全部次数不超过 \(4\) 的多项式空间，故任意此类多项式 \(Q\) 的期望亦与 \(\mathrm{Pois}(1)\) 匹配到同一误差阶。证毕。
\end{proof}

\begin{corollary}[五阶下降阶乘是首个可见破缺阶]\label{cor:conclusion-leyang-fifth-falling-factorial-first-visible-break}
\leanverified{paper\_conclusion\_leyang\_fifth\_falling\_factorial\_first\_visible\_break}
在定理 \ref{thm:conclusion-leyang-unramified-fiber-count-quartic-poisson-cloak} 的设定下，记
\[
(N)_k:=N(N-1)\cdots (N-k+1).
\]
则对每个良素 \(p\notin \{2,3,31,37\}\) 都有逐点恒等式
\[
\bigl(N_p(y_0)\bigr)_5=0
\qquad (y_0\in U_p^{\mathrm{aff}}),
\]
从而
\[
\frac{1}{|U_p^{\mathrm{aff}}|}
\sum_{y_0\in U_p^{\mathrm{aff}}}\bigl(N_p(y_0)\bigr)_5=0.
\]
另一方面，
\[
\EE_{\mathrm{Pois}(1)}[(N)_5]=1.
\]
因此，在全部多项式观测中，五阶下降阶乘是区分未分歧四次纤维与 \(\mathrm{Pois}(1)\) 的最小阶：次数不超过 \(4\) 的观测全部被 quartic Poisson 伪装屏蔽，而到五阶才首次出现不可消去的精确破缺。
\end{corollary}
\begin{proof}
由于 \(\Pi(\lambda,y_0)\) 关于 \(\lambda\) 的次数为 \(4\)，对任意 \(y_0\in U_p^{\mathrm{aff}}\) 都有
\[
0\le N_p(y_0)\le 4.
\]
故逐点恒有
\[
\bigl(N_p(y_0)\bigr)_5=0,
\]
平均后仍为 \(0\)。
而 Poisson 分布满足
\[
\EE_{\mathrm{Pois}(1)}[(N)_k]=1
\qquad (k\ge 1),
\]
特别在 \(k=5\) 时得到严格差异。
再结合定理 \ref{thm:conclusion-leyang-unramified-fiber-count-quartic-poisson-cloak} 已知 \(k\le 4\) 的全部多项式观测与 \(\mathrm{Pois}(1)\) 匹配，故五阶确为首个可见破缺阶。证毕。
\end{proof}

\begin{theorem}[Hecke 迹的 KL--信息误差控制界]\label{thm:conclusion-leyang-hecke-trace-kl-bound}
沿用推论 \ref{cor:conclusion-leyang-s4-joint-chebotarev-double-observable} 的记号，并设 \(p\neq 37\) 为良素。
令
\[
U_p^{\mathrm{aff}}:=\FF_p\setminus \bigl(B(\overline{\FF}_p)\cap \mathbb A^1_y\bigr)
\]
为去除有限分歧值后的仿射参数集。
对 \(y\in U_p^{\mathrm{aff}}\) 记
\(
N_p(y):=\card{\{\lambda\in\FF_p:\Pi(\lambda,y)=0\}}\in\{0,1,2,4\}
\)，并令 \(\mu_p\) 为 \(N_p(y)\) 在 \(U_p^{\mathrm{aff}}\) 上的经验分布：
\[
\mu_p(k):=\frac{1}{|U_p^{\mathrm{aff}}|}\card{\{y\in U_p^{\mathrm{aff}}:\ N_p(y)=k\}},
\qquad k\in\{0,1,2,4\}.
\]
令 \(\pi\) 为 \(N\) 的 \(S_4\)-极限分布（由推论 \ref{cor:conclusion-leyang-s4-joint-chebotarev-double-observable} 的联合分布取边缘）：
\[
\pi_0=\frac{3}{8},\qquad
\pi_1=\frac{1}{3},\qquad
\pi_2=\frac{1}{4},\qquad
\pi_4=\frac{1}{24},
\qquad
\EE_\pi[N]=1.
\]
则存在仅依赖于分歧集合 \(B\) 的常数 \(C<\infty\)，使得
\[
\boxed{\;
|a_p|
\le
\frac{3}{\sqrt 2}\,p\,\sqrt{D_{\mathrm{KL}}(\mu_p\Vert \pi)}
\ +\ C\; }.
\]
\end{theorem}

\begin{proof}
由结论 \ref{con:xi-terminal-zm-elliptic-finite-field-point-count}，
\[
\#C_p(\FF_p)=p-a_p,
\qquad
\sum_{y\in\FF_p}N_p(y)=\#C_p(\FF_p),
\]
从而
\[
\sum_{y\in\FF_p}\bigl(N_p(y)-1\bigr)=-a_p.
\]
将和拆分为 \(U_p^{\mathrm{aff}}\) 与其补集，并注意 \(\card{\FF_p\setminus U_p^{\mathrm{aff}}}\le \card{B}\) 且对任意 \(y\) 有 \(|N_p(y)-1|\le 3\)，得到
\[
\left|\EE_{\mu_p}[N]-1\right|
=\left|\frac{1}{|U_p^{\mathrm{aff}}|}\sum_{y\in U_p^{\mathrm{aff}}}(N_p(y)-1)\right|
\le
\frac{|a_p|}{p}+O\!\left(\frac{1}{p}\right),
\]
其中隐含常数仅依赖于 \(\card{B}\)。
\par
另一方面，由 \(N\in\{0,1,2,4\}\) 得对任意两分布 \(\mu,\nu\) 有
\[
\bigl|\EE_\mu[N]-\EE_\nu[N]\bigr|
\le
3\,\mathrm{TV}(\mu,\nu),
\]
其中 \(\mathrm{TV}\) 为全变差距离。
取 \(\nu=\pi\) 并使用 Pinsker 不等式
\(
\mathrm{TV}(\mu_p,\pi)\le \sqrt{\tfrac12 D_{\mathrm{KL}}(\mu_p\Vert\pi)}
\)，得
\[
\left|\EE_{\mu_p}[N]-1\right|
\le
\frac{3}{\sqrt 2}\sqrt{D_{\mathrm{KL}}(\mu_p\Vert\pi)}.
\]
合并两式并乘以 \(p\)，将 \(O(1)\) 项并入常数 \(C\) 即得结论。证毕。
\end{proof}

\begin{corollary}[Chebotarev 统计误差推出 Hasse 级别的迹界]\label{cor:conclusion-leyang-chebotarev-hasse-via-kl}
在定理 \ref{thm:conclusion-leyang-hecke-trace-kl-bound} 的设定下，若存在常数 \(B<\infty\) 使
\[
\max_{k\in\{0,1,2,4\}}|\mu_p(k)-\pi_k|\le \frac{B}{\sqrt p},
\]
则存在只依赖于 \(\pi\) 的常数 \(A(\pi)<\infty\) 与只依赖于分歧集合 \(B\) 的常数 \(C<\infty\)，使得
\[
\boxed{\;
|a_p|
\le
\frac{3}{\sqrt 2}\,\sqrt{A(\pi)}\,B\,\sqrt p
\ +\ C\; }.
\]
特别地，若 \(\mu_p=\pi+O(p^{-1/2})\)，则 \(D_{\mathrm{KL}}(\mu_p\Vert\pi)=O(p^{-1})\)，从而 \(|a_p|=O(\sqrt p)\)。
\end{corollary}

\begin{proof}
在有限字母表上，\(\mu=\pi+\delta\) 且 \(\max_k|\delta_k|\le \epsilon\) 蕴含
\[
D_{\mathrm{KL}}(\mu\Vert\pi)\le A(\pi)\,\epsilon^2
\]
其中可取 \(A(\pi)=\sum_k \pi_k^{-1}\) 的一个常数倍；将 \(\epsilon=B/\sqrt p\) 代入并应用定理 \ref{thm:conclusion-leyang-hecke-trace-kl-bound} 即得所示界。证毕。
\end{proof}

\begin{theorem}[Abel--Jacobi 时间读出的三采样可辨识与圆维下界]\label{thm:conclusion-leyang-time-identifiability-three-sampling}
沿用结论 \ref{con:xi-terminal-zm-leyang-abel-jacobi-ode} 的时间坐标 \(u\) 与 \(\delta=\frac{\dd y}{\dd u}\)。
在归一化 Lee--Yang 口径下，考虑抽象微分代数关系
\[
\delta^4-2(y+1)\delta^2-8y^{3/2}\delta-y(y-1)P(y)\equiv 0,
\qquad
P(y)=y^3+ay^2+by+c.
\]
取样本 \(u_1,u_2,u_3\)，记 \(y_i=y(u_i)\notin\{0,1\}\) 且两两不同，\(\delta_i=\delta(u_i)\)。
则成立：
\begin{enumerate}
  \item 系数 \((a,b,c)\) 由三对观测 \(\{(y_i,\delta_i)\}_{i=1}^3\) 唯一确定；
  \item \((a,b,c)\) 是线性系统
  \[
  a y_i^2+b y_i+c
  =
  \frac{\delta_i^4-2(y_i+1)\delta_i^2-8y_i^{3/2}\delta_i}{y_i(y_i-1)}-y_i^3,
  \qquad i=1,2,3
  \]
  的唯一解；
  \item 若观测含噪 \((\widehat y_i,\widehat\delta_i)\) 满足
  \[
  |\widehat y_i-y_i|\le\varepsilon,\qquad
  |\widehat\delta_i-\delta_i|\le\varepsilon,
  \]
  且存在 \(\eta,\rho>0\) 使
  \[
  \min_{i\ne j}|y_i-y_j|\ge\eta,\qquad
  \min_i |y_i(y_i-1)|\ge\rho,
  \]
  则存在常数 \(C=C(\eta,\rho,\max_i|y_i|,\max_i|\delta_i|)\) 使
  \[
  \max\{|\widehat a-a|,\ |\widehat b-b|,\ |\widehat c-c|\}\le C\,\varepsilon.
  \]
\end{enumerate}
此外，就参数格 \(\ZZ^3\) 而言，\(\cdim(\ZZ^3)=3\) 给出可辨识自由度下界：少于三点的纯时间采样在一般位置下无法唯一恢复 \((a,b,c)\)。
\end{theorem}
\begin{proof}
对每个样本点定义
\[
v_i:=
\frac{\delta_i^4-2(y_i+1)\delta_i^2-8y_i^{3/2}\delta_i}{y_i(y_i-1)}
=P(y_i).
\]
因 \(P\) 首一，
\[
v_i-y_i^3=a y_i^2+b y_i+c,\qquad i=1,2,3,
\]
得到未知向量 \((a,b,c)^\top\) 的 \(3\times 3\) 线性系统，其系数矩阵为
\[
V=
\begin{pmatrix}
y_1^2&y_1&1\\
y_2^2&y_2&1\\
y_3^2&y_3&1
\end{pmatrix},
\qquad
\det V=(y_1-y_2)(y_1-y_3)(y_2-y_3)\neq 0.
\]
故唯一性与显式解立即成立。

在稳定性部分，映射
\[
(y,\delta)\longmapsto
\frac{\delta^4-2(y+1)\delta^2-8y^{3/2}\delta}{y(y-1)}
\]
在分母远离零点、且 \((y,\delta)\) 有界的紧域上为局部 Lipschitz，故
\[
|\widehat v_i-v_i|\le C_0\varepsilon.
\]
线性系统扰动估计给出
\[
\|(\widehat a,\widehat b,\widehat c)-(a,b,c)\|
\le \|V^{-1}\|\,\|(\widehat v_i-v_i)_{i=1}^3\|,
\]
而 \(\|V^{-1}\|\) 在 \(\min_{i\ne j}|y_i-y_j|\ge\eta\) 下有上界，故得到所需 \(C\varepsilon\) 估计。

最后，\((a,b,c)\) 为三自由参数，少于三条独立观测方程一般不足以唯一确定；对参数格 \(\ZZ^3\) 而言，该事实与 \(\cdim(\ZZ^3)=3\) 一致。证毕。
\end{proof}
