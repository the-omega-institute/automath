\begin{theorem}[多维 prime-register 轮廓的随机游走表示与极限律]\label{thm:conclusion-multishift-random-walk-profile}
\leanverified{paper\_conclusion\_multishift\_random\_walk\_profile}
固定整数 \(d\ge 1\)。
取
\(
f(\mathbf z)=\sum_{\boldsymbol\alpha\in F}a_{\boldsymbol\alpha}\mathbf z^{\boldsymbol\alpha}\in\NN[z_1,\dots,z_d]
\)
（有限支撑 \(F\subset\NN^d\)），并令
\[
A:=f(\mathbf 1)=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\ge 2.
\]
令 \(\Phi_f=\Theta_d(f)\) 为定理 \ref{thm:conclusion-multishift-commuting-algorithms-polynomial} 所给算子，并定义迭代轨道
\[
r^{(n)}:=\Phi_f^{\,n}(\mathbf e_{\mathbf 0})
=\sum_{\mathbf k\in\NN^d}c_{n,\mathbf k}\,\mathbf e_{\mathbf k}
\qquad(n\ge 0).
\]
令离散随机向量 \(J\) 取值于 \(F\)，并满足
\[
\PP(J=\boldsymbol\alpha)=\frac{a_{\boldsymbol\alpha}}{A}.
\]
取独立同分布序列 \((J_i)_{i\ge 1}\)，并令 \(S_n:=J_1+\cdots+J_n\)。
则对所有 \(\mathbf k\in\NN^d\) 有系数恒等式
\[
\boxed{\ c_{n,\mathbf k}=A^n\,\PP(S_n=\mathbf k)\ }.
\]
令归一化权 \(\pi_n(\mathbf k):=c_{n,\mathbf k}/A^n\)。
记 \(\boldsymbol\mu:=\EE[J]\in\RR^d\)、\(\Sigma:=\Cov(J)\)。
则成立如下极限律：
\begin{enumerate}
  \item \textbf{大数律.}
        对任意有界连续函数 \(\varphi:\RR^d\to\RR\)，有
        \[
        \sum_{\mathbf k\in\NN^d}\pi_n(\mathbf k)\,\varphi(\mathbf k/n)\ \longrightarrow\ \varphi(\boldsymbol\mu).
        \]
  \item \textbf{中心极限定理.}
        若 \(\Sigma\) 非退化，则对任意有界连续函数 \(\psi:\RR^d\to\RR\)，有
        \[
        \sum_{\mathbf k\in\NN^d}\pi_n(\mathbf k)\,
        \psi\!\left(\frac{\mathbf k-n\boldsymbol\mu}{\sqrt n}\right)
        \ \longrightarrow\
        \int_{\RR^d}\psi(x)\,d\mathcal N(0,\Sigma)(x).
        \]
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(\Phi_f(\mathbf e_{\mathbf n})=\sum_{\boldsymbol\alpha}a_{\boldsymbol\alpha}\mathbf e_{\mathbf n+\boldsymbol\alpha}\) 递推可得生成函数恒等式：
\[
\sum_{\mathbf k}c_{n,\mathbf k}\mathbf z^{\mathbf k}=f(\mathbf z)^n.
\]
另一方面，
\[
\EE[\mathbf z^{S_n}]
=\Bigl(\sum_{\boldsymbol\alpha}\PP(J=\boldsymbol\alpha)\mathbf z^{\boldsymbol\alpha}\Bigr)^n
=\Bigl(\frac{f(\mathbf z)}{A}\Bigr)^n.
\]
比较系数得到 \(c_{n,\mathbf k}=A^n\PP(S_n=\mathbf k)\)。
大数律与中心极限定理分别由独立同分布和的标准结论推出。证毕。
\end{proof}
