\begin{theorem}[Boolean 除数格上的精确 Ramanujan 层析]\label{thm:conclusion-dyadic-boundary-primeshell-boolean-divisor-tomography}
\leanverified{paper\_conclusion\_dyadic\_boundary\_primeshell\_boolean\_divisor\_tomography}
固定 \(m,n\ge 1\)。令 \(\overrightarrow{\mathcal F}^{\,n-1}_m\) 为 \(Q_m\) 的全部取向 \((n-1)\)-面，并固定一组互异素数
\[
\{\mathfrak p_{\widetilde f}\}_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}.
\]
对任意分辨率 \(m\) 的 dyadic polycube \(U\subset I^n\)，定义边界 Gödel 整数
\[
H_m(U):=
\prod_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\mathfrak p_{\widetilde f}^{\varepsilon_{\widetilde f}(U)},
\qquad
\varepsilon_{\widetilde f}(U)\in\{0,1\},
\]
其中 \(\varepsilon_{\widetilde f}(U)\) 表示 \(\widetilde f\) 是否出现在 \(\partial_n[U]\) 的取向支撑中。对任意平方自由导子 \(q\mid \Pi_m:=\prod_{\widetilde f}\mathfrak p_{\widetilde f}\)，记
\[
\mathcal C_q(U):=c_q\!\bigl(H_m(U)\bigr).
\]
则对任意有限子集 \(S\subseteq \overrightarrow{\mathcal F}^{\,n-1}_m\)，若记
\[
q_S:=\prod_{\widetilde f\in S}\mathfrak p_{\widetilde f},
\]
则成立精确恒等式
\[
\mathbf 1_{\{S\subseteq \operatorname{supp}(\partial_n[U])\}}
=
\frac{1}{q_S}\sum_{d\mid q_S}\mathcal C_d(U).
\]
因此，边界取向面支撑的任意有限模式都可由 Ramanujan 影在 Boolean 除数格上的一次望远镜求和精确读出。
\end{theorem}

\begin{proof}
由于 \(H_m(U)\) 为平方自由整数，对任意 \(S\subseteq \overrightarrow{\mathcal F}^{\,n-1}_m\) 都有
\[
q_S\mid H_m(U)
\iff
S\subseteq \operatorname{supp}(\partial_n[U]).
\]
另一方面，Ramanujan 和的经典除数恒等式给出
\[
\sum_{d\mid q_S}c_d\!\bigl(H_m(U)\bigr)
=
q_S\,\mathbf 1_{\{q_S\mid H_m(U)\}}.
\]
将 \(\mathcal C_d(U)=c_d(H_m(U))\) 代入并除以 \(q_S\)，即得所示公式。证毕。
\end{proof}

\begin{corollary}[单面、双面与边界面积的 prime-shell 闭式]\label{cor:conclusion-dyadic-boundary-primeshell-single-double-area-formulas}
\leanverified{paper\_conclusion\_dyadic\_boundary\_primeshell\_single\_double\_area\_formulas}
在上述记号下，对任意互异取向面 \(\widetilde f,\widetilde g\in \overrightarrow{\mathcal F}^{\,n-1}_m\)，有
\[
\varepsilon_{\widetilde f}(U)
=
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U)}{\mathfrak p_{\widetilde f}},
\]
\[
\varepsilon_{\widetilde f}(U)\varepsilon_{\widetilde g}(U)
=
\frac{
1+\mathcal C_{\mathfrak p_{\widetilde f}}(U)
+\mathcal C_{\mathfrak p_{\widetilde g}}(U)
+\mathcal C_{\mathfrak p_{\widetilde f}\mathfrak p_{\widetilde g}}(U)
}{
\mathfrak p_{\widetilde f}\mathfrak p_{\widetilde g}
}.
\]
特别地，
\[
\operatorname{Area}(\partial U)
=
2^{-m(n-1)}
\sum_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U)}{\mathfrak p_{\widetilde f}}.
\]
\end{corollary}

\begin{proof}
分别取定理 \ref{thm:conclusion-dyadic-boundary-primeshell-boolean-divisor-tomography} 中的
\[
S=\{\widetilde f\},
\qquad
S=\{\widetilde f,\widetilde g\},
\]
即可得到前两式。再对全部取向面求和，并注意
\[
\sum_{\widetilde f}\varepsilon_{\widetilde f}(U)=|\partial_n[U]|_{\ell^1},
\]
而单个 dyadic \((n-1)\)-面的面积为 \(2^{-m(n-1)}\)，即得面积公式。证毕。
\end{proof}

\begin{theorem}[合成导子的严格冗余与 prime-shell 的有限 Euler 因子化]\label{thm:conclusion-dyadic-boundary-primeshell-composite-conductors-redundant}
\leanverified{paper\_conclusion\_dyadic\_boundary\_primeshell\_composite\_conductors\_redundant}
对任意平方自由导子 \(q\mid \Pi_m\)，成立
\[
\mathcal C_q(U)=\prod_{\mathfrak p\mid q}\mathcal C_{\mathfrak p}(U).
\]
因此有限 Dirichlet 生成函数
\[
\mathfrak R_{U,m}(s):=\sum_{q\mid \Pi_m}\frac{\mathcal C_q(U)}{q^s}
\]
具有精确 Euler 因子化
\[
\mathfrak R_{U,m}(s)
=
\prod_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\left(
1+\frac{\mathcal C_{\mathfrak p_{\widetilde f}}(U)}{\mathfrak p_{\widetilde f}^{\,s}}
\right).
\]
更具体地，
\[
1+\frac{\mathcal C_{\mathfrak p_{\widetilde f}}(U)}{\mathfrak p_{\widetilde f}^{\,s}}
=
\begin{cases}
1+\dfrac{\mathfrak p_{\widetilde f}-1}{\mathfrak p_{\widetilde f}^{\,s}},
& \varepsilon_{\widetilde f}(U)=1,
\\[8pt]
1-\dfrac{1}{\mathfrak p_{\widetilde f}^{\,s}},
& \varepsilon_{\widetilde f}(U)=0.
\end{cases}
\]
故所有合成导子只携带 prime-shell 的乘法合成，而不再承载额外几何自由度。
\end{theorem}

\begin{proof}
对固定整数 \(n\)，Ramanujan 和关于导子 \(q\) 是乘法性的。由于这里只取 \(q\mid \Pi_m\) 的平方自由导子，故
\[
c_q\!\bigl(H_m(U)\bigr)
=
\prod_{\mathfrak p\mid q}c_{\mathfrak p}\!\bigl(H_m(U)\bigr),
\]
即得第一式。对全部平方自由因子求和并按 Euler 乘积展开，即得生成函数分解。

又对素数 \(\mathfrak p_{\widetilde f}\)，若 \(\varepsilon_{\widetilde f}(U)=1\)，则 \(\mathfrak p_{\widetilde f}\mid H_m(U)\)，从而
\[
\mathcal C_{\mathfrak p_{\widetilde f}}(U)=\mathfrak p_{\widetilde f}-1;
\]
若 \(\varepsilon_{\widetilde f}(U)=0\)，则 \(\mathfrak p_{\widetilde f}\nmid H_m(U)\)，从而
\[
\mathcal C_{\mathfrak p_{\widetilde f}}(U)=-1.
\]
代回即得局部因子闭式。证毕。
\end{proof}

\begin{theorem}[单素数 Ramanujan 壳的全息完备性]\label{thm:conclusion-dyadic-boundary-single-primeshell-complete-holographic-coordinates}
单素数 Ramanujan 影族
\[
\Bigl\{\mathcal C_{\mathfrak p_{\widetilde f}}(U)\Bigr\}_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\]
唯一决定 \(H_m(U)\)、\(\partial_n[U]\) 与 \(U\) 本身。
\end{theorem}

\begin{proof}
由推论 \ref{cor:conclusion-dyadic-boundary-primeshell-single-double-area-formulas}，
\[
\varepsilon_{\widetilde f}(U)
=
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U)}{\mathfrak p_{\widetilde f}}
\]
对每个 \(\widetilde f\) 都唯一确定。故全部单素数影族唯一恢复边界取向支撑向量 \((\varepsilon_{\widetilde f}(U))_{\widetilde f}\)，进而唯一恢复平方自由整数
\[
H_m(U)
=
\prod_{\widetilde f}\mathfrak p_{\widetilde f}^{\varepsilon_{\widetilde f}(U)}.
\]
再由定理 \ref{thm:spg-boundary-godelization-holographic-dictionary}，\(H_m(U)\) 的素因子分解唯一决定 \(\partial_n[U]\) 并唯一恢复 \(U\)。证毕。
\end{proof}

\begin{theorem}[对称差边界的 Ramanujan--Stokes 精确律]\label{thm:conclusion-dyadic-boundary-primeshell-ramanujan-stokes-symmetric-difference}
\leanverified{paper\_conclusion\_dyadic\_boundary\_primeshell\_ramanujan\_stokes\_symmetric\_difference}
对任意两 dyadic polycube \(U,V\subset I^n\)，定义对称差边界 Gödel 码
\[
R_m(U,V)
:=
\frac{H_m(U)H_m(V)}{\gcd(H_m(U),H_m(V))^2},
\]
并记
\[
\mathcal C_q(U,V):=c_q\!\bigl(R_m(U,V)\bigr).
\]
则有
\[
\operatorname{Dist}_m(U,V)
=
\sum_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U,V)}{\mathfrak p_{\widetilde f}},
\]
\[
\operatorname{Area}(\partial U-\partial V)
=
2^{-m(n-1)}
\sum_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U,V)}{\mathfrak p_{\widetilde f}},
\]
以及
\[
\bigl|\operatorname{Vol}(U)-\operatorname{Vol}(V)\bigr|
\le
2^{-m(n-1)}
\sum_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U,V)}{\mathfrak p_{\widetilde f}}.
\]
\end{theorem}

\begin{proof}
平方自由整数 \(R_m(U,V)\) 的素支撑恰为 \(\partial_n[U]\) 与 \(\partial_n[V]\) 的取向面对称差支撑。因此对每个 \(\widetilde f\)，
\[
\mathbf 1_{\{\mathfrak p_{\widetilde f}\mid R_m(U,V)\}}
=
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U,V)}{\mathfrak p_{\widetilde f}}.
\]
对全部 \(\widetilde f\) 求和便得到对称差面数，即
\[
\operatorname{Dist}_m(U,V)
=
\sum_{\widetilde f}
\frac{1+\mathcal C_{\mathfrak p_{\widetilde f}}(U,V)}{\mathfrak p_{\widetilde f}}.
\]
再由定理 \ref{thm:spg-godel-boundary-gcd-lipschitz-stability}，有
\[
\operatorname{Area}(\partial U-\partial V)=2^{-m(n-1)}\operatorname{Dist}_m(U,V),
\]
以及
\[
\bigl|\operatorname{Vol}(U)-\operatorname{Vol}(V)\bigr|
\le
2^{-m(n-1)}\operatorname{Dist}_m(U,V).
\]
代回即得其余两式。证毕。
\end{proof}

\begin{theorem}[单素数 Ramanujan 壳决定全部单项式矩与有限矩盒]\label{thm:conclusion-dyadic-boundary-single-primeshell-determines-moments}
单素数 Ramanujan 影族
\[
\Bigl\{\mathcal C_{\mathfrak p_{\widetilde f}}(U)\Bigr\}_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\]
唯一决定全部单项式矩
\[
\mu_\alpha(U):=\int_U x^\alpha\,dx
\qquad
(\alpha\in \NN^n),
\]
并且唯一决定有限矩盒
\[
\Bigl\{\mu_\alpha(U):\ \alpha\in\{0,\dots,2^m-1\}^n\Bigr\}.
\]
因而它也唯一决定 \(U\) 本身。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-dyadic-boundary-single-primeshell-complete-holographic-coordinates}，单素数 Ramanujan 壳唯一决定 \(H_m(U)\)。而定理 \ref{thm:spg-boundary-godel-moment-readout} 已表明 \(H_m(U)\) 决定全部单项式矩，推论 \ref{cor:spg-boundary-godel-finite-moment-completeness} 表明它尤其决定有限矩盒，并由定理 \ref{thm:conclusion-fixedresolution-boundary-godel-momentbox-equivalence} 在固定分辨率类内唯一恢复 \(U\)。证毕。
\end{proof}

\begin{theorem}[平方自由边界码的 Ramanujan 实现判据与线性时间译码]\label{thm:conclusion-dyadic-boundary-primeshell-realizability-linear-decode}
\leanverified{paper\_conclusion\_dyadic\_boundary\_primeshell\_realizability\_linear\_decode}
设 \(N\mid \Pi_m\) 为任意平方自由整数，并定义
\[
\varepsilon_{\widetilde f}(N):=
\frac{1+c_{\mathfrak p_{\widetilde f}}(N)}{\mathfrak p_{\widetilde f}},
\qquad
b(N):=
\sum_{\widetilde f\in \overrightarrow{\mathcal F}^{\,n-1}_m}
\varepsilon_{\widetilde f}(N)\,\widetilde f.
\]
则下列命题等价：
\begin{enumerate}
  \item 存在分辨率 \(m\) 的 dyadic polycube \(U\subset I^n\) 使 \(H_m(U)=N\)；
  \item \(b(N)\) 满足平方类综合征的局部闭合条件；
  \item \(b(N)\in \operatorname{im}\partial_n\)。
\end{enumerate}
并且在这些条件成立时，\(U\) 唯一存在，且可由 spanning-tree bulk 译码算法在线性时间内恢复。
\end{theorem}

\begin{proof}
由素数 Ramanujan 和在平方自由整数上的显式取值，
\[
\varepsilon_{\widetilde f}(N)=1
\iff
\mathfrak p_{\widetilde f}\mid N.
\]
故 \(b(N)\) 正是由 \(N\) 的 prime-shell 数据恢复出的边界取向支撑链。于是
\[
N=H_m(U)
\iff
\mathbf G_{n-1}^{-1}(N)=\partial_n[U]
\iff
b(N)\in \operatorname{im}\partial_n.
\]
而定理 \ref{thm:conclusion-dyadic-boundary-squareclass-syndrome-unique-fill} 已证明
\[
b(N)\in \operatorname{im}\partial_n
\iff
\delta_{n-1}(N)=1,
\]
并且一旦该条件成立，对应填充唯一，且可由定理 \ref{thm:spg-linear-time-bulk-decode-from-boundary} 的生成树递推在线性时间恢复。证毕。
\end{proof}

\end{conjecture}
