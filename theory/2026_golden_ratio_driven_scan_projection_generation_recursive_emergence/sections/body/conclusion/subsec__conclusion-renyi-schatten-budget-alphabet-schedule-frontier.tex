\subsection{固定阶 Rényi--Schatten 阶跃、线性预算计数壁与最终周期调度锋面}\label{subsec:conclusion-renyi-schatten-budget-alphabet-schedule-frontier}
沿用小节 \ref{subsec:conclusion-renyi-schatten-choi-sat-gap-copy-amplification}、
\ref{subsec:conclusion-fold-ledger-dual-ldp-holography-undecidability-langlands}、
\ref{subsec:conclusion-window6-symmetry-obstruction-linear-externalization}
与 \ref{subsec:conclusion-collision-genfunc-renyi-soft-firstfit} 的记号。
前述链条已经分别给出：固定阶 Rényi--Schatten 幂矩对 SAT 的二值分离、纤维复制放大的 \((t+1)^{\alpha-1}\) 阶 gap、线性预算下的双大偏差容量变分律、可逆侧信息的五符号门槛，以及 soft 序门在最终周期调度上的可判定性锋面猜想。
把这些接口继续压缩后，还可得到如下同一条资源分层链。

\begin{theorem}[固定阶 Rényi--Schatten 缺口的刚性单比特 SAT 序参量]\label{thm:conclusion-renyi-schatten-gap-rigid-onebit-sat-order-parameter}
\leanverified{paper\_conclusion\_renyi\_schatten\_gap\_rigid\_onebit\_sat\_order\_parameter}
固定 \(\alpha>2\)。
对附录推论 \ref{cor:fold-channel-kraus-rank-hardness-template} 与定理 \ref{thm:fold-choi-maxeig-approx-sat-hard} 给出的两纤维 SAT 折叠信道，记
\[
\rho_\varphi:=\frac{1}{N}J(\mathcal E_\varphi),
\qquad
N:=|\Omega_\varphi|=2^{n+1},
\]
并定义
\[
H_\alpha(\rho):=\frac{1}{1-\alpha}\log\Tr(\rho^\alpha),
\qquad
\Delta_\alpha(\varphi):=2\log N-H_\alpha(\rho_\varphi).
\]
则有严格二值化
\[
\varphi\notin \mathrm{SAT}
\iff
\Delta_\alpha(\varphi)=0,
\]
并且
\[
\varphi\in \mathrm{SAT}
\Longrightarrow
\Delta_\alpha(\varphi)\ge \log 2.
\]
因此，若存在多项式时间近似器 \(\widehat\Delta_\alpha(\varphi)\) 满足
\[
\bigl|\widehat\Delta_\alpha(\varphi)-\Delta_\alpha(\varphi)\bigr|<\frac12\log 2
\qquad(\forall \varphi),
\]
则 \(P=NP\)。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-fixed-alpha-renyi-entropy-gap-sat-hard}，对任意固定 \(\alpha>2\) 都有
\[
\varphi\notin \mathrm{SAT}
\Longrightarrow
H_\alpha(\rho_\varphi)=2\log N,
\]
\[
\varphi\in \mathrm{SAT}
\Longrightarrow
H_\alpha(\rho_\varphi)\le 2\log N-\log 2.
\]
把两式改写为 \(\Delta_\alpha(\varphi)=2\log N-H_\alpha(\rho_\varphi)\) 即得
\[
\Delta_\alpha(\varphi)=0
\quad\text{或}\quad
\Delta_\alpha(\varphi)\ge \log 2,
\]
并且两种情形分别等价于 UNSAT 与 SAT。
最后，若加性误差严格小于 \(\frac12\log 2\)，则比较 \(\widehat\Delta_\alpha(\varphi)\) 与 \(\frac12\log 2\) 即可在多项式时间内区分 SAT 与 UNSAT，故 \(P=NP\)。证毕。
\end{proof}

\begin{theorem}[纤维复制把 Rényi--Schatten 缺口放大为任意预设尺度]\label{thm:conclusion-renyi-schatten-gap-copy-amplified-order-parameter}
固定 \(\alpha>2\) 与整数 \(t\ge 1\)。
沿用定理 \ref{thm:conclusion-renyi-schatten-copy-amplification} 的复制放大态
\[
\rho_{\varphi,t}:=\frac{1}{N_t}J(\mathcal E_{\varphi,t}),
\qquad
N_t:=2^n(t+1),
\]
并定义
\[
\Delta_{\alpha,t}(\varphi):=2\log N_t-H_\alpha(\rho_{\varphi,t}).
\]
则成立
\[
\varphi\notin \mathrm{SAT}
\iff
\Delta_{\alpha,t}(\varphi)=0,
\]
而
\[
\varphi\in \mathrm{SAT}
\Longrightarrow
\Delta_{\alpha,t}(\varphi)\ge \log(t+1).
\]
特别地，若 \(t=n^k\)（\(k\) 固定），则
\[
\Delta_{\alpha,t}(\varphi)\ge k\log n+o(1),
\]
而若 \(t=2^n\)，则
\[
\Delta_{\alpha,t}(\varphi)\ge n\log 2+o(1).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-renyi-schatten-copy-amplification}，
\[
\varphi\notin \mathrm{SAT}
\Longrightarrow
\Tr(\rho_{\varphi,t}^\alpha)=N_t^{\,2-2\alpha},
\]
\[
\varphi\in \mathrm{SAT}
\Longrightarrow
\Tr(\rho_{\varphi,t}^\alpha)\ge (t+1)^{\alpha-1}N_t^{\,2-2\alpha}.
\]
代入
\[
H_\alpha(\rho)=\frac{1}{1-\alpha}\log\Tr(\rho^\alpha)
\]
可得
\[
\varphi\notin \mathrm{SAT}
\Longrightarrow
H_\alpha(\rho_{\varphi,t})=2\log N_t,
\]
\[
\varphi\in \mathrm{SAT}
\Longrightarrow
H_\alpha(\rho_{\varphi,t})\le 2\log N_t-\log(t+1).
\]
于是
\[
\Delta_{\alpha,t}(\varphi)=0
\quad\text{或}\quad
\Delta_{\alpha,t}(\varphi)\ge \log(t+1).
\]
最后把 \(t=n^k\) 或 \(t=2^n\) 代入 \(\log(t+1)\) 即得相应增长律。证毕。
\end{proof}

\begin{theorem}[线性预算的双大偏差变分律之上存在谱无关的绝对计数壁]\label{thm:conclusion-linear-budget-absolute-counting-wall}
\leanverified{paper\_conclusion\_linear\_budget\_absolute\_counting\_wall}
对预算 \(B\in\ZZ_{\ge 0}\) 定义最优成功率
\[
\mathrm{Succ}_m(B):=\frac{1}{2^m}\sum_{x\in X_m}\min\!\bigl(d_m(x),2^B\bigr).
\]
令 \(B_m=\lfloor \beta m\rfloor\)、\(a:=\beta\log 2\)。
在定理 \ref{thm:conclusion-oracle-success-double-ldp-variational} 的大偏差假设下，
\[
\lim_{m\to\infty}\frac1m\log \mathrm{Succ}_m(B_m)
=
\max\!\Bigl\{
-\inf_{\alpha\le a}I_w(\alpha),\
\log\varphi+(\beta-1)\log 2-\inf_{\alpha>a}I_U(\alpha)
\Bigr\}.
\]
但无论是否假设该大偏差原理，都存在一个与纤维谱细节无关的绝对计数壁
\[
\beta_{\mathrm{cnt}}:=\log_2\!\Bigl(\frac{2}{\varphi}\Bigr)\approx 0.305758086369\ldots
\]
使得对一切 \(\beta<\beta_{\mathrm{cnt}}\) 都有
\[
\mathrm{Succ}_m(\lfloor \beta m\rfloor)
\le
\exp\!\Bigl(-\bigl((1-\beta)\log 2-\log\varphi\bigr)m+o(m)\Bigr).
\]
特别地，任何线性预算若低于 \(\beta_{\mathrm{cnt}}m\)，则最优恢复成功率必指数衰减到 \(0\)。
\end{theorem}
\begin{proof}
第一式正是定理 \ref{thm:conclusion-oracle-success-double-ldp-variational}。

另一方面，不借助任何额外谱假设，直接由
\[
\min(d_m(x),2^{B_m})\le 2^{B_m}
\]
得
\[
\mathrm{Succ}_m(B_m)
\le
\frac{1}{2^m}\sum_{x\in X_m}2^{B_m}
=
|X_m|\,2^{B_m-m}.
\]
再由 \(|X_m|=F_{m+2}=\exp(m\log\varphi+o(m))\)，得到
\[
\frac1m\log \mathrm{Succ}_m(B_m)
\le
\log\varphi+(\beta-1)\log 2+o(1).
\]
当
\[
\beta<\log_2\!\Bigl(\frac{2}{\varphi}\Bigr)
\]
时，右端严格为负，从而得到所述指数上界。证毕。
\end{proof}

\begin{corollary}[目标线性密度强制最小事件字母表]\label{cor:conclusion-linear-density-forces-minimal-event-alphabet}
设每一步仅写入一个事件符号，字母表大小为 \(M\ge 2\)，并要求存在一族确定性可逆提升，其最坏情形侧信息长度满足
\[
L_m\le (\ell+o(1))m.
\]
则必有
\[
\ell\ge \frac{\log(2/\varphi)}{\log M},
\qquad\text{等价地}\qquad
M\ge \exp\!\left(\frac{\log(2/\varphi)}{\ell}\right).
\]
特别地，当
\[
\ell=\frac{4}{27},
\]
则
\[
M\ge 5.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-side-info-length-lower-bound}，
\[
\liminf_{m\to\infty}\frac{L_m}{m}\ge \frac{\log(2/\varphi)}{\log M}.
\]
把假设 \(L_m\le (\ell+o(1))m\) 代入即得
\[
\ell\ge \frac{\log(2/\varphi)}{\log M},
\]
亦即
\[
M\ge \exp\!\left(\frac{\log(2/\varphi)}{\ell}\right).
\]
当 \(\ell=\frac{4}{27}\) 时，
\[
\exp\!\left(\frac{\log(2/\varphi)}{4/27}\right)\approx 4.460\ldots,
\]
故整数门槛恰为 \(M\ge 5\)。
这与推论 \ref{cor:conclusion-alphabet-threshold-mge5} 及定理 \ref{thm:conclusion-godel-five-symbol-threshold} 一致。证毕。
\end{proof}

\begin{theorem}[第一可行字母表并不消除最大纤维上的指数失败]\label{thm:conclusion-first-feasible-alphabet-still-fails-on-maxfiber}
取临界线性密度
\[
\ell=\frac{4}{27}
\]
与第一可行字母表大小
\[
M=5.
\]
若一族编码--译码方案满足
\[
L_m\le \frac{4}{27}m+O(1),
\]
则对任意最大纤维达到者 \(x_m^\star\in X_m\)（即 \(d_m(x_m^\star)=D_m\)），在条件分布
\[
\omega\sim \mathrm{Unif}\bigl(\Fold_m^{-1}(x_m^\star)\bigr)
\]
下，条件成功率满足
\[
\PP\!\left(
\mathrm{dec}_m(x_m^\star,\mathrm{enc}_m(\omega))=\omega
\ \middle|\ X=x_m^\star
\right)
\le
\exp(-c_5m+C),
\]
其中
\[
c_5:=\frac12\log\varphi-\frac{4}{27}\log 5
\approx 0.002170666243\ldots>0,
\]
而 \(C\) 为与方案有关但与 \(m\) 无关的常数。
因此，“字母表首次可行”与“最坏纤维上可靠”并不是同一门槛：在 \(M=5\) 处，前者已成立，而后者仍以指数速度失败。
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-local-rewrite-maxfiber-strong-converse-finite-alphabet}，对任意有限字母表 \(|\Sigma|=M\ge 2\)，若
\[
L_m\le \frac{4}{27}m+O(1),
\]
则对任意最大纤维达到者 \(x_m^\star\) 都有
\[
\PP\!\left(
\mathrm{dec}_m(x_m^\star,\mathrm{enc}_m(\omega))=\omega
\ \middle|\ X=x_m^\star
\right)
\le
\exp\!\left(-\Bigl(\frac12\log\varphi-\frac{4}{27}\log M\Bigr)m+C\right).
\]
代入 \(M=5\) 即得
\[
\PP(\mathrm{correct}\mid X=x_m^\star)\le \exp(-c_5m+C),
\]
其中
\[
c_5=\frac12\log\varphi-\frac{4}{27}\log 5>0.
\]
另一方面，推论 \ref{cor:conclusion-linear-density-forces-minimal-event-alphabet} 已说明 \(M=5\) 正是 \(\ell=\frac4{27}\) 下的第一可行字母表。
故“可行性门槛”与“最坏纤维上的可靠性门槛”之间确有严格缝隙。证毕。
\end{proof}

\begin{conjecture}[一旦越过计数壁，真正的可判定性锋面转移到最终周期调度]\label{conj:conclusion-budget-alphabet-crossed-schedule-frontier}
在线性预算已越过绝对计数壁、事件字母表也已越过可逆化的最小门槛之后，资源约束本身不再决定 soft 序门分布语义的可判定性。
真正的剩余锋面应当转移到调度 \(u:\NN\to(0,1)\) 的最终周期性。
更精确地，预计此时的可判定/不可判定分裂应与猜想 \ref{conj:conclusion-eventually-periodic-schedule-frontier} 完全一致：
\begin{enumerate}
\normalfont
  \item 若 \(u\) 最终周期，则对任意两有限态骨架 \(\mathcal S_1,\mathcal S_2\)，性质
  \[
  \forall t\ge 0,\ \mu_t^{u,\mathcal S_1}=\mu_t^{u,\mathcal S_2}
  \]
  应为可判定问题；
  \item 若允许一般可计算但非最终周期的 \(u\)，则存在固定有限态骨架对 \((\mathcal S_1,\mathcal S_2)\)，使上述等价性在算术层级意义下达到 \(\Pi^0_1\)-完备。
\end{enumerate}
\end{conjecture}

因此，固定阶 Rényi--Schatten 缺口、纤维复制放大的熵跃迁、线性预算的双大偏差变分律、绝对计数壁、最小事件字母表门槛以及最大纤维强逆界，在结论层并不共用同一条临界曲线。
它们形成了严格分层的三重锋面：信息论计数壁、语法实现门槛与最坏纤维可靠性门槛；
而当这三者都被跨越之后，残余的真正语义锋面才应由最终周期调度给出。
