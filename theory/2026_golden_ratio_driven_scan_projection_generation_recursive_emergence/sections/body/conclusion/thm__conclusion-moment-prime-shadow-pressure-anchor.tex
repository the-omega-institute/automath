\begin{theorem}[几乎所有素数阴影的刚性完备性]\label{thm:conclusion-moment-good-prime-shadow-rigidity}
固定整数 \(q\ge 2\)，令
\[
a_n:=S_q(n+2)\qquad(n\ge 0),
\]
并记其唯一首一最小整系数递推多项式为
\[
P_q(\lambda)\in \ZZ[\lambda],
\qquad
\deg P_q=d_q.
\]
设 \(H_0(q)\) 为定义 \ref{def:pom-hankel-normal-form} 中对应于该最小递推的首个可逆 \(d_q\times d_q\) Hankel 主块。则对每个满足
\[
p\nmid \det H_0(q)
\]
的素数 \(p\)，模 \(p\) 约化序列 \(\bar a_n:=a_n\bmod p\) 在 \(\FF_p\) 上仍具有同阶最小首一递推，且其最小递推多项式恰为
\[
\overline{P_q}(\lambda)=P_q(\lambda)\bmod p.
\]
因而，共有限个良素数上的最小递推阴影族
\[
\{\overline{P_q}\}_{p\nmid \det H_0(q)}
\]
唯一决定 \(P_q\) 本身；进一步，\(P_q\) 连同任意 \(d_q\) 个连续初值唯一决定整个序列 \((S_q(m))_{m\ge 2}\)。
\end{theorem}

\begin{proof}
由命题 \ref{prop:pom-moment-minrecurrence-hankel}，\(d_q\) 同时等于序列 \((a_n)\) 的最小递推阶数、Hankel 秩与最小 realization 维数。再由推论 \ref{cor:xi-hankel-minimal-recurrence-good-prime-preservation}，对每个满足
\[
p\nmid \det H_0(q)
\]
的素数，\(\bar a_n\) 在 \(\FF_p\) 上仍具有同阶最小首一递推，且其最小递推多项式正是 \(\overline{P_q}\)。

若另有首一整系数多项式 \(Q(\lambda)\neq P_q(\lambda)\) 在全部良素数上给出同一阴影，则 \(P_q-Q\) 至少有一个非零整数系数。该系数只被有限多个素数整除，故可取一良素数 \(p\nmid \det H_0(q)\) 避开这些除子，使
\[
\overline{Q}\neq \overline{P_q},
\]
与良素数上的最小性矛盾。故良素数阴影族唯一决定 \(P_q\)。

最后，由最小递推阶为 \(d_q\)，任意 \(d_q\) 个连续初值都唯一生成 \((a_n)\)，等价地唯一生成 \((S_q(m))_{m\ge 2}\)。证毕。
\end{proof}

\begin{corollary}[有限良素数即可作整数级重建]\label{cor:conclusion-moment-finite-good-primes-crt-reconstruction}
\leanverified{paper\_conclusion\_moment\_finite\_good\_primes\_crt\_reconstruction}
在定理 \ref{thm:conclusion-moment-good-prime-shadow-rigidity} 的记号下，若事先给定一个对 \(P_q\) 全部系数与所选 \(d_q\) 个连续初值的统一高度上界 \(B_q\)，并取若干良素数

\begin{proof}
由定理 \ref{thm:conclusion-moment-good-prime-shadow-rigidity}，每个良素数 \(p_i\) 上的阴影都给出真递推多项式 \(P_q\) 与所选初值块的模 \(p_i\) 余数。于是中国剩余定理给出模

\end{proof}

\begin{proposition}[坏素数集即 Hankel 刚性支撑]\label{prop:conclusion-moment-hankel-bad-prime-stiffness-support}
在定理 \ref{thm:conclusion-moment-good-prime-shadow-rigidity} 的设定下，可能导致递推阶下降、最小多项式变形或最小 realization 算术失真的素数集必包含于
\[
\operatorname{Supp}\bigl(\det H_0(q)\bigr).
\]
更精确地，对每个素数 \(p\) 定义
\[
\operatorname{Stiff}_p(q):=v_p\!\bigl(\det H_0(q)\bigr),
\]
并记
\[
H_1(q):=\bigl(a_{i+j+1}\bigr)_{0\le i,j\le d_q-1}
\]
为与 \(H_0(q)\) 相邻的偏移 Hankel 块，则 \(\operatorname{Stiff}_p(q)\) 同时控制：
\begin{enumerate}
  \item 模 \(p\) 阴影是否可能失真；
  \item 最小有理 realization
  \[
  A=H_0(q)^{-1}H_1(q)
  \]
  的 \(p\)-进恢复门槛；
  \item 整数递推与 \(p\)-进最小模型之间的算术刚性。
\end{enumerate}
\end{proposition}

\begin{proof}
由推论 \ref{cor:xi-hankel-minimal-recurrence-good-prime-preservation}，当 \(p\nmid \det H_0(q)\) 时，模 \(p\) 阴影保持最小递推阶与最小首一多项式；故任何模 \(p\) 失真都只能发生在
\[
p\mid \det H_0(q)
\]
时。

另一方面，最小 Hankel realization 的 companion-normal form 由
\[
A=H_0(q)^{-1}H_1(q)
=\frac{\operatorname{adj}(H_0(q))\,H_1(q)}{\det H_0(q)}
\]
给出，因此其全部有理分母都整除 \(\det H_0(q)\)。故 \(p\)-进恢复精度与算术稳定性只可能在
\[
v_p\!\bigl(\det H_0(q)\bigr)>0
\]
时出现损失，而其损失强度恰由该赋值 \(\operatorname{Stiff}_p(q)\) 控制。证毕。
\end{proof}

\begin{theorem}[连续压力的整数阶代数锚点]\label{thm:conclusion-projective-pressure-integer-algebraic-anchor}
对每个整数 \(q\ge 2\)，射影压力算子 \(\mathcal L_q\) 在齐次次数 \(q\) 子空间 \(\mathcal H_q\) 上满足严格代数退化
\[
\mathcal L_q\vert_{\mathcal H_q}=\widetilde A_q,
\qquad
\rho(\mathcal L_q\vert_{\mathcal H_q})=\rho(\widetilde A_q)=r_q.
\]
若连续压力函数按注 \ref{rem:pom-projective-pressure-route} 定义为
\[
\Lambda(t):=\log\lambda(t+1)-\log |A_{\mathrm{out}}|,
\]
其中 \(\lambda(t+1)\) 是 \(\mathcal L_{t+1}\) 的 Perron 主特征值，则在所有整数锚点 \(t=q-1\) 处都有精确恒等式
\[
\Lambda(q-1)=\log r_q-\log |A_{\mathrm{out}}|.
\]
对于 Fold 主核，\(|A_{\mathrm{out}}|=2\)，故
\[
\Lambda(q-1)=\log r_q-\log 2.
\]
\end{theorem}

\begin{proof}
定理 \ref{thm:pom-projective-operator-degenerates-to-moment-kernel} 已证明：对每个整数 \(q\ge 2\)，\(\mathcal L_q\) 在 \(\mathcal H_q\) 上的限制在任一单项式基下恰由 \(\widetilde A_q\) 表示，且
\[
\rho(\mathcal L_q\vert_{\mathcal H_q})=\rho(\widetilde A_q)=r_q.
\]
再由注 \ref{rem:pom-projective-pressure-route} 对连续压力的定义，整数点 \(t=q-1\) 正是该有限维退化的精确插值节点，因此
\[
\Lambda(q-1)=\log\lambda(q)-\log |A_{\mathrm{out}}|
=\log r_q-\log |A_{\mathrm{out}}|.
\]
对 Fold 主核代入 \(|A_{\mathrm{out}}|=2\) 即得最后一式。证毕。
\end{proof}

\begin{theorem}[固定分辨率碰撞矩列的 Stieltjes 正性与完全对数凸性]\label{thm:conclusion-collision-moment-stieltjes-hankel-positivity}
\leanverified{paper\_conclusion\_collision\_moment\_stieltjes\_hankel\_positivity}
固定分辨率 \(m\)。则序列 \(\{S_q(m)\}_{q\ge 0}\) 是一条有限支撑 Stieltjes 矩列。更精确地，对任意 \(R\ge 0\)，Hankel 矩阵
\[
H_R^{(m)}:=\bigl(S_{i+j}(m)\bigr)_{0\le i,j\le R}
\]
正半定，因此全部主子式非负。特别地，对一切 \(q\ge 0\)，有
\[
S_{q+1}(m)^2\le S_q(m)S_{q+2}(m),
\]
并且比值序列
\[
q\longmapsto \frac{S_{q+1}(m)}{S_q(m)}
\]
单调不减。
\end{theorem}

\begin{proof}
定义离散正测度
\[
\nu_m:=\sum_{x\in X_m}\delta_{d_m(x)}
=\sum_{j=1}^{r_m}\kappa_j^{(m)}\delta_{\lambda_j^{(m)}}.
\]
则
\[
S_q(m)=\int t^q\,d\nu_m(t),
\]
故 \(\{S_q(m)\}_{q\ge 0}\) 是一条 Stieltjes 矩列。对任意 \(u=(u_0,\dots,u_R)\in\CC^{R+1}\)，有
\[
u^\ast H_R^{(m)}u
=
\sum_{i,j=0}^{R}\overline{u_i}u_jS_{i+j}(m)
=
\int\left|\sum_{i=0}^{R}u_it^i\right|^2d\nu_m(t)\ge 0.
\]
故 \(H_R^{(m)}\succeq 0\)。

取 \(R=1\) 并把指标整体平移到 \((q,q+1,q+2)\)，得到
\[
\det
\begin{pmatrix}
S_q(m) & S_{q+1}(m)\\
S_{q+1}(m) & S_{q+2}(m)
\end{pmatrix}
\ge 0,
\]
即
\[
S_{q+1}(m)^2\le S_q(m)S_{q+2}(m).
\]
对正数列，这正等价于相邻比值单调不减。证毕。
\end{proof}

\begin{corollary}[碰撞核 Perron 根的完全对数凸性与 Rényi 熵率的离散凹性]\label{cor:conclusion-collision-perron-logconvex-renyi-concavity}
\leanverified{paper\_conclusion\_collision\_perron\_logconvex\_renyi\_concavity}
设 \(r_q:=\rho(A_q)\) 且 \(P_q=\log r_q\)。则对一切整数 \(q\ge 1\)，有
\[
r_{q+1}^2\le r_qr_{q+2},
\qquad
P_{q+1}-P_q\le P_{q+2}-P_{q+1}.
\]
若定义 Rényi 型投影熵率
\[
h_q:=q\log 2-P_q,
\]
则
\[
h_{q+2}-2h_{q+1}+h_q\le 0.
\]
\end{corollary}

\begin{proof}
正文关于 \(q\)-重碰撞核 \(A_q\) 的 realization 已给出
\[
\lim_{m\to\infty}\frac{1}{m}\log S_q(m)=\log r_q=P_q.
\]
对定理 \ref{thm:conclusion-collision-moment-stieltjes-hankel-positivity} 中的不等式
\[
S_{q+1}(m)^2\le S_q(m)S_{q+2}(m)
\]
取对数、除以 \(m\)，再令 \(m\to\infty\)，即得
\[
2\log r_{q+1}\le \log r_q+\log r_{q+2}.
\]
这同时给出 Perron 根的对数凸性与 \(P_q\) 的离散凸性。最后
\[
h_{q+2}-2h_{q+1}+h_q
=
-(P_{q+2}-2P_{q+1}+P_q)\le 0.
\]
证毕。
\end{proof}

\begin{theorem}[单一有限阶饱和强制整个正阶压力射线线性冻结]\label{thm:conclusion-one-order-saturation-freezes-pressure-ray}
对任意整数 \(q\ge 1\)，都有普适上界
\[
P_q\le \log 2+(q-1)\Lambda_\infty.
\]
更强地，若存在某个 \(q_0\ge 2\) 使
\[
P_{q_0}=\log 2+(q_0-1)\Lambda_\infty,
\]
则
\[
I(\Lambda_\infty)=0,
\]
并且对每个 \(q\ge 1\),
\[
P_q=\log 2+(q-1)\Lambda_\infty.
\]
\end{theorem}

\begin{proof}
由
\[
S_q(m)=\sum_{x\in X_m}d_m(x)^q
\le D_m^{q-1}\sum_{x\in X_m}d_m(x)
=2^mD_m^{q-1},
\]
取对数、除以 \(m\)，再令 \(m\to\infty\)，即得
\[
P_q\le \log 2+(q-1)\Lambda_\infty.
\]

若对某个 \(q_0\ge 2\) 上界取等，则
\[
\Lambda(q_0-1)=P_{q_0}-\log 2=(q_0-1)\Lambda_\infty.
\]
另一方面，由大偏差对偶公式
\[
\Lambda(a)=\sup_{0\le y\le \Lambda_\infty}\bigl(ay-I(y)\bigr),
\]
可得
\[
\Lambda(q_0-1)
\le
\sup_{0\le y\le \Lambda_\infty}(q_0-1)y
=(q_0-1)\Lambda_\infty.
\]
要达到等号，极大点只能取在 \(y=\Lambda_\infty\)，并且必须满足
\[
I(\Lambda_\infty)=0.
\]

一旦 \(I(\Lambda_\infty)=0\)，对任意 \(a\ge 0\) 都有
\[
\Lambda(a)\ge a\Lambda_\infty-I(\Lambda_\infty)=a\Lambda_\infty.
\]
而由同样的上界论证，
\[
\Lambda(a)\le a\Lambda_\infty.
\]
故
\[
\Lambda(a)=a\Lambda_\infty\qquad(\forall a\ge 0).
\]
取 \(a=q-1\) 并代回 \(P_q=\log 2+\Lambda(q-1)\)，即得
\[
P_q=\log 2+(q-1)\Lambda_\infty
\qquad(\forall q\ge 1).
\]
证毕。
\end{proof}

\begin{corollary}[单阶谱半径饱和的全阶刚化]\label{cor:conclusion-one-order-spectral-radius-saturation-freezes-all-orders}
\leanverified{paper\_conclusion\_one\_order\_spectral\_radius\_saturation\_freezes\_all\_orders}

\begin{theorem}[正阶压力谱的 sharp 二歧]\label{thm:conclusion-pressure-spectrum-sharp-frozen-unfrozen-dichotomy}
以下两类情形互斥且穷尽：
\begin{enumerate}
  \item \textbf{Frozen phase}: 若
  \[
  I(\Lambda_\infty)=0,
  \]
  则对每个 \(q\ge 1\),
  \[
  P_q=\log 2+(q-1)\Lambda_\infty.
  \]
  \item \textbf{Unfrozen phase}: 若
  \[
  I(\Lambda_\infty)>0,
  \]
  则对每个 \(q\ge 2\)，存在严格正数
  \[
  \delta_q:=\inf_{0\le y\le \Lambda_\infty}\Bigl(I(y)+(q-1)(\Lambda_\infty-y)\Bigr)>0
  \]
  使
  \[
  P_q=\log 2+(q-1)\Lambda_\infty-\delta_q.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
Frozen phase 已由定理 \ref{thm:conclusion-one-order-saturation-freezes-pressure-ray} 得到。现设
\[
I(\Lambda_\infty)>0.
\]
对固定 \(q\ge 2\)，定义
\[
F_q(y):=I(y)+(q-1)(\Lambda_\infty-y)
\qquad (0\le y\le \Lambda_\infty).
\]
由于 \(I\) 为 good rate function，\(F_q\) 在紧区间 \([0,\Lambda_\infty]\) 上达到最小值。并且
\[
F_q(\Lambda_\infty)=I(\Lambda_\infty)>0,
\]
而对任意 \(y<\Lambda_\infty\),
\[
F_q(y)\ge (q-1)(\Lambda_\infty-y)>0.
\]
故
\[
\delta_q:=\min_{0\le y\le \Lambda_\infty}F_q(y)>0.
\]
另一方面，
\[
\Lambda(q-1)
=
\sup_{0\le y\le \Lambda_\infty}\bigl((q-1)y-I(y)\bigr)
=
(q-1)\Lambda_\infty-\delta_q.
\]
代回
\[
P_q=\log 2+\Lambda(q-1)
\]
即得
\[
P_q=\log 2+(q-1)\Lambda_\infty-\delta_q.
\]
两相显然互斥，并由 \(I(\Lambda_\infty)\) 的零/正二值性穷尽。证毕。
\end{proof}

\begin{theorem}[escort 倾斜下的精确 tilted-LDP 与最大纤维吸引子]\label{thm:conclusion-escort-tilted-ldp-maxfiber-attractor}
对任意 \(a>0\)，定义 escort 测度

\pi_m^{\langle a\rangle}(x)
:=
\frac{d_m(x)^a\pi_m(x)}{\sum_{y}d_m(y)^a\pi_m(y)}

\frac{d_m(x)^{a+1}}{S_{a+1}(m)}.

Y_m:=\frac{1}{m}\log d_m(X_m)
\qquad (X_m\sim \pi_m^{\langle a\rangle})
\]
满足 tilted 大偏差原理，其率函数为
\[
I_a(y)=I(y)-ay+\Lambda(a).
\]
特别地，在 frozen phase 中，
\[
I_a(y)=I(y)+a(\Lambda_\infty-y).
\]
因此对任意 \(\varepsilon>0\)，都有

\le
\exp\!\bigl(-(a\varepsilon+o(1))m\bigr).

\end{theorem}

\begin{proof}

\pi_m^{\langle a\rangle}(A)

因此正文关于指数倾斜的大偏差原理直接给出新的率函数
\[
I_a(y)=I(y)-ay+\Lambda(a).
\]

若处于 frozen phase，则由定理 \ref{thm:conclusion-one-order-saturation-freezes-pressure-ray}，

\]
故
\[
I_a(y)=I(y)+a(\Lambda_\infty-y).
\]
当 \(y\le \Lambda_\infty-\varepsilon\) 时，
\[
I_a(y)\ge a\varepsilon.
\]
对闭集 \((-\infty,\Lambda_\infty-\varepsilon]\) 应用大偏差上界，即得
\[
\pi_m^{\langle a\rangle}\!\bigl(Y_m\le \Lambda_\infty-\varepsilon\bigr)
\le
\exp\!\bigl(-(a\varepsilon+o(1))m\bigr).
\]
证毕。
\end{proof}

  \item 无穷 Hankel 矩阵
  \[
  H_\infty^{(m)}:=\bigl(S_{i+j}(m)\bigr)_{i,j\ge 0}
  \]
  的秩恰为
  \[
  \operatorname{rank}H_\infty^{(m)}=r_m;
  \]
  \item 所有 \((r_m+1)\times (r_m+1)\) Hankel 主子式都为零，而某个 \(r_m\times r_m\) 主子式严格正；
  \item 仅凭前 \(2r_m\) 个矩
  \[
  S_0(m),S_1(m),\dots,S_{2r_m-1}(m),
  \]
  即可唯一恢复整个有限多重度谱
  \[
  \{(\lambda_j^{(m)},\kappa_j^{(m)})\}_{j=1}^{r_m}.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
记
\[
V:=(\lambda_j^{(m)\,i})_{\substack{0\le i<\infty\\1\le j\le r_m}},
\qquad
K:=\operatorname{diag}(\kappa_1^{(m)},\dots,\kappa_{r_m}^{(m)}).
\]
则
\[
H_\infty^{(m)}=VKV^\top.
\]
由于节点 \(\lambda_j^{(m)}\) 两两不同且 \(\kappa_j^{(m)}>0\)，前 \(r_m\) 行形成的 Vandermonde 子矩阵满秩，故
\[
\operatorname{rank}H_\infty^{(m)}=r_m.
\]
这立刻推出所有 \((r_m+1)\times (r_m+1)\) Hankel 主子式为零。

另一方面，取前 \(r_m\) 行与前 \(r_m\) 列构成的 Hankel 块，其行列式为
\[
\left(\prod_{j=1}^{r_m}\kappa_j^{(m)}\right)
\left(\prod_{1\le i<j\le r_m}(\lambda_j^{(m)}-\lambda_i^{(m)})^2\right)>0.
\]
故某个 \(r_m\times r_m\) 主子式严格正。

现在令
\[
Q_m(z):=\prod_{j=1}^{r_m}(z-\lambda_j^{(m)})
=
z^{r_m}-c_{r_m-1}z^{r_m-1}-\cdots-c_0.
\]
因为每个 \(\lambda_j^{(m)}\) 都满足 \(Q_m(\lambda_j^{(m)})=0\)，故对所有 \(q\ge 0\),
\[
S_{q+r_m}(m)=c_{r_m-1}S_{q+r_m-1}(m)+\cdots+c_0S_q(m).
\]
因此前 \(2r_m\) 个矩唯一决定该最小递推系数，进而唯一决定 \(Q_m\) 及全部节点 \(\lambda_j^{(m)}\)。

节点恢复后，再由前 \(r_m\) 个矩组成 Vandermonde 线性系统
\[
\sum_{j=1}^{r_m}\kappa_j^{(m)}(\lambda_j^{(m)})^q=S_q(m),
\qquad
0\le q\le r_m-1,
\]
因 Vandermonde 矩阵可逆而唯一解出全部权重 \(\kappa_j^{(m)}\)。证毕。
\end{proof}
