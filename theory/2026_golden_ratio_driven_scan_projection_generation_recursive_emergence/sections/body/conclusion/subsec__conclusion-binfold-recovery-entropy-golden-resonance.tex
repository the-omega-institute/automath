\subsection{二进恢复率、可见熵率与黄金共振的三模刚性}\label{subsec:conclusion-binfold-recovery-entropy-golden-resonance}
沿用小节 \ref{subsec:conclusion-foldbin-faithful-dimension-fullinversion-obstruction}、小节 \ref{subsec:conclusion-binfold-critical-capacity-golden-threshold}、小节 \ref{subsec:conclusion-maxfiber-average-scale-rigidity}、定理 \ref{thm:xi-time-part68-entropy-deficit-triple-identification}、定理 \ref{thm:fold-critical-resonance-constant}、定理 \ref{thm:killo-fold-three-mode-witness-critical-gap}、推论 \ref{cor:fold-critical-resonance-gap} 与推论 \ref{cor:fold-max-fiber-constant-bias} 的记号。前述链条已经分别把完全反演的最小位预算、全阶 Rényi 熵率、碰撞缺口与最大纤维偏置写成黄金闭式。把这些接口并置，可再抽出如下四条彼此咬合的后果。

\begin{theorem}[二进恢复率与可见熵率的精确一比特分解]\label{thm:conclusion-binfold-fullrecovery-visible-entropy-onebit-splitting}
设
\[
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m}
\qquad (w\in X_m),
\]
并记
\[
B_m^{\mathrm{full}}
:=
\min\bigl\{B\in\ZZ_{\ge 0}:\ \mathrm{Succ}_m^{\mathrm{bin}}(B)=1\bigr\}.
\]
其中 \(H_q\) 在 \(q=1\) 时按 Shannon 熵解释。则对任意 \(q>0\) 都有
\[
\lim_{m\to\infty}\frac{B_m^{\mathrm{full}}}{m}
=
\log_2\frac{2}{\varphi},
\qquad
\lim_{m\to\infty}\frac{H_q(\mu_m)}{m\log 2}
=
\log_2\varphi.
\]
从而
\[
\lim_{m\to\infty}
\left(
\frac{B_m^{\mathrm{full}}}{m}
+
\frac{H_q(\mu_m)}{m\log 2}
\right)
=1.
\]
\leanverified{paper\_binfold\_recovery\_m13}
\leanverified{paper\_binfold\_recovery\_m15}
\leanverified{paper\_binfold\_recovery\_m17}
\leanverified{paper\_binfold\_recovery\_extended\_13\_15\_17}
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-foldbin-faithful-vs-fullinversion-exponential-separation} 已给出
\[
B_m^{\mathrm{full}}
=
\bigl(1-\log_2\varphi\bigr)m+O(1)
=
\log_2\frac{2}{\varphi}\,m+O(1),
\]
故
\[
\lim_{m\to\infty}\frac{B_m^{\mathrm{full}}}{m}
=
\log_2\frac{2}{\varphi}.
\]
另一方面，定理 \ref{thm:xi-time-part68-entropy-deficit-triple-identification} 表明对任意 \(q>0\)（其中 \(q=1\) 按 Shannon 熵理解）都有
\[
\frac{1}{m}\bigl(m\log 2-H_q(\mu_m)\bigr)\longrightarrow \log\frac{2}{\varphi}.
\]
移项并除以 \(\log 2\)，即得
\[
\frac{H_q(\mu_m)}{m\log 2}\longrightarrow \log_2\varphi.
\]
最后利用恒等式
\[
\log_2\varphi+\log_2\frac{2}{\varphi}=1
\]
便得到所示严格守恒律。证毕。
\leanverified{paper\_binfold\_recovery\_onebit\_splitting}
\leanverified{paper\_binfold\_recovery\_fibonacci\_scaling}
\end{proof}

\begin{theorem}[黄金共振导致的碰撞缺口硬下界]\label{thm:conclusion-fold-golden-resonance-collision-gap-hard-floor}
记
\[
a_m(k):=\frac{|\widehat d_m(k)|}{2^m},
\]
并令
\[
C_\varphi:=\prod_{n=2}^{\infty}\bigl|\cos(\pi\varphi^{-n})\bigr|>0.
\]
则折叠碰撞概率满足
\[
\liminf_{m\to\infty}F_{m+2}\,\mathsf{Col}_m
\ge
1+2C_\varphi^2.
\]
等价地，
\[
\liminf_{m\to\infty}
F_{m+2}\left(\mathsf{Col}_m-\frac{1}{F_{m+2}}\right)
\ge
2C_\varphi^2.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-collision-spectrum} 的 Parseval 恒等式，
\[
\mathsf{Col}_m=\frac{1}{F_{m+2}}\sum_{k\in \ZZ/(F_{m+2}\ZZ)}a_m(k)^2.
\]
其中
\[
a_m(0)=1.
\]
再由定理 \ref{thm:fold-critical-resonance-constant}，
\[
a_m(F_m)\longrightarrow C_\varphi,
\qquad
a_m(F_{m+1})\longrightarrow C_\varphi.
\]
由于
\[
0<F_m<F_{m+1}<F_{m+2},
\]
故 \(0,F_m,F_{m+1}\) 在 \(\ZZ/(F_{m+2}\ZZ)\) 中彼此不同，于是
\[
F_{m+2}\,\mathsf{Col}_m
=
\sum_k a_m(k)^2
\ge
1+a_m(F_m)^2+a_m(F_{m+1})^2.
\]
令 \(m\to\infty\) 取下极限，即得
\[
\liminf_{m\to\infty}F_{m+2}\,\mathsf{Col}_m
\ge
1+2C_\varphi^2.
\]
再减去均匀基线 \(1/F_{m+2}\) 便得到第二式。证毕。
\end{proof}

\begin{theorem}[极大纤维的黄金共振抬升]\label{thm:conclusion-fold-golden-resonance-maxfiber-lift}
\leanverified{paper\_conclusion\_fold\_golden\_resonance\_maxfiber\_lift}
设
\[
M_m:=\max_{r\in \ZZ/(F_{m+2}\ZZ)} d_m(r),
\qquad
\overline d_m:=\frac{2^m}{F_{m+2}}.
\]
则有
\[
\liminf_{m\to\infty}\frac{F_{m+2}}{2^m}\bigl(M_m-\overline d_m\bigr)\ge C_\varphi.
\]
等价地，
\[
M_m
\ge
\bigl(1+C_\varphi+o(1)\bigr)\frac{2^m}{F_{m+2}}
=
\bigl(\sqrt5(1+C_\varphi)+o(1)\bigr)\left(\frac{2}{\varphi}\right)^m.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-max-fiber-fourier}，
\[
M_m\ge \overline d_m+\frac{S_m}{F_{m+2}},
\qquad
S_m:=\max_{k\ne 0}|\widehat d_m(k)|.
\]
又由定义，
\[
S_m\ge |\widehat d_m(F_m)|=2^m a_m(F_m).
\]
因此
\[
M_m-\overline d_m
\ge
\frac{2^m}{F_{m+2}}\,a_m(F_m).
\]
令 \(m\to\infty\) 取下极限，并应用定理 \ref{thm:fold-critical-resonance-constant} 中
\[
a_m(F_m)\to C_\varphi,
\]
便得到
\[
\liminf_{m\to\infty}\frac{F_{m+2}}{2^m}\bigl(M_m-\overline d_m\bigr)\ge C_\varphi.
\]
再用
\[
\frac{2^m}{F_{m+2}}=\bigl(\sqrt5+o(1)\bigr)\left(\frac{2}{\varphi}\right)^m
\]
改写，即得第二式。证毕。
\end{proof}

\begin{conjecture}[共振三模完备性]\label{conj:conclusion-fold-three-mode-completeness}
在算术 fold 中，除
\[
k=0,\qquad k=F_m,\qquad k=F_{m+1}
\]
之外，其余 Fourier 模态在宏观尺度上全部湮灭，即
\[
\sum_{k\notin\{0,F_m,F_{m+1}\}} a_m(k)^2=o(1).
\]
若该猜想成立，则定理 \ref{thm:conclusion-fold-golden-resonance-collision-gap-hard-floor} 中的下界提升为极限公式
\[
\lim_{m\to\infty}F_{m+2}\,\mathsf{Col}_m
=
1+2C_\varphi^2.
\]
若再与附录中的 Fibonacci 极值模猜想 \ref{conj:fold-fourier-fibonacci-extremizer} 合并，则定理 \ref{thm:conclusion-fold-golden-resonance-maxfiber-lift} 的下界也相应提升为
\[
\lim_{m\to\infty}\frac{F_{m+2}}{2^m}\bigl(M_m-\overline d_m\bigr)
=
C_\varphi.
\]
\end{conjecture}

\noindent\emph{证据.}
定理 \ref{thm:killo-fold-three-mode-witness-critical-gap} 已经表明，碰撞缺口的主阶下界可由三模
\[
\{0,F_m,F_{m+1}\}
\]
单独见证；定理 \ref{thm:xi-time-part63b-three-mode-gap-zero-mode-nonrenormalizable} 又说明，仅靠零模归一化不可能消除这一下界。现有严格正极限也恰只在两条 Fibonacci 临界频率 \(F_m\) 与 \(F_{m+1}\) 上被锁定为 \(C_\varphi\)。因而剩余问题只在于：其余模态是否在平方能量尺度上真的退化为 \(o(1)\)。若答案为是，则碰撞主项便完全由三模闭合；再结合猜想 \ref{conj:fold-fourier-fibonacci-extremizer} 对极值模的定位，峰值偏置也会被同一个常数 \(C_\varphi\) 彻底锁定。

\noindent
因此，bin-fold 的每一步微观二进位在宏观极限上被严格拆成两部分：一部分以
\[
\log_2\varphi
\]
的速率外显为可见熵，另一部分以
\[
\log_2\frac{2}{\varphi}
\]
的速率外显为完全恢复所需的侧信息。而在算术 fold 一侧，黄金共振又把这条隐藏份额重新写回碰撞缺口与极大纤维超均值。于是，“可见统计的熵率”与“恢复所需的外置位率”并非两张彼此无关的账本；它们与共振底噪一道，构成同一条一比特资源分解在信息、Fourier 与峰值三个层面的同步外显。

若再把稳定层均匀基线、微态推前基线、相对均匀密度比与 escort 温度流并置，则这条一比特分解还会进一步坍缩为同一个两原子母分布的 Euler--Mellin 轨道：稳定层均匀律给出母分布本身，微态推前只是其尺寸偏置像，相对密度只是其归一化像，而全部 escort 极限族则只是其 Mellin 倾斜半群。于是 bin-fold 的整座极限信息几何塔，不再是彼此平行的若干闭式，而是同一个 rank-two Prony germ 在四条不同读出通道上的同步投影。
