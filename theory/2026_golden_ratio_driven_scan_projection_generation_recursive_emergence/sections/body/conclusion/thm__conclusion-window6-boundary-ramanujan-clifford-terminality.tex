\begin{theorem}[boundary Clifford 代数的 Ramanujan--Clifford \texorpdfstring{$(4+8)$}{(4+8)} 分裂]\label{thm:conclusion-window6-boundary-ramanujan-clifford-augmentation-splitting}
\leanverified{paper\_conclusion\_window6\_boundary\_ramanujan\_clifford\_augmentation\_splitting}

\[
X_6^{\mathrm{bdry}}
=
\{\texttt{100001},\texttt{100101},\texttt{101001}\},

A_{\partial}:=
\bigoplus_{w\in X_6^{\mathrm{bdry}}}
\End_{\CC}\!\bigl(\CC[(\Fold_6^{\mathrm{bin}})^{-1}(w)]\bigr).

\[
A_{\partial}\cong \bigoplus_{w\in X_6^{\mathrm{bdry}}}M_2(\CC)
\cong
\bigoplus_{w\in X_6^{\mathrm{bdry}}}\mathrm{Cl}_2(\CC).

令

\Sigma(X_1,X_2,X_3)=(X_2,X_3,X_1)

为三块 \(M_2(\CC)\) 的循环置换，并定义

E_{\mathrm{geo}}:=\frac13(1+\Sigma+\Sigma^2),
\qquad
E_{\mathrm{def}}:=\frac13(2-\Sigma-\Sigma^2).

则
\[
E_{\mathrm{geo}}(X_1,X_2,X_3)=(\bar X,\bar X,\bar X),
\qquad

故 \(E_{\mathrm{geo}}\) 是到同步 boundary 子代数

A_{\partial}^{\mathrm{geo}}

的忠实条件期望；同时
\[
E_{\mathrm{def}}(X_1,X_2,X_3)
=(X_1-\bar X,\ X_2-\bar X,\ X_3-\bar X),
\]
其像为
\[
A_{\partial}^{\mathrm{def}}
:=
\{(X_1,X_2,X_3):X_1+X_2+X_3=0\}.
\]
从而
\[
A_{\partial}=A_{\partial}^{\mathrm{geo}}\oplus A_{\partial}^{\mathrm{def}}
\]
作为 \(A_{\partial}^{\mathrm{geo}}\)-双模严格分裂，且
\[
\dim_{\CC}A_{\partial}^{\mathrm{geo}}=4,
\qquad
\dim_{\CC}A_{\partial}^{\mathrm{def}}=8.
\]
等价地，
\[
A_{\partial}\cong M_2(\CC)\otimes \CC[C_3],
\qquad
A_{\partial}^{\mathrm{def}}
\cong
M_2(\CC)\otimes \mathrm{Aug}(\CC[C_3]).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-window6-boundary-two-sheet-permutation-decomposition}，window-\(6\) 的 boundary 部分恰由三条二点纤维组成。故每条 boundary 纤维的端同态代数均同构于 \(M_2(\CC)\)，从而
\[
A_{\partial}\cong \bigoplus_{w\in X_6^{\mathrm{bdry}}}M_2(\CC).
\]
这也可视为三块复 Clifford 二阶块的直和。

算子
\[
E_{\mathrm{geo}}=\frac13(1+\Sigma+\Sigma^2)
\]
是有限群 \(C_3=\langle \Sigma\rangle\) 的平均投影，故其像正是 \(\Sigma\)-不动点子代数，也即
\[
A_{\partial}^{\mathrm{geo}}
=
\{(X,X,X):X\in M_2(\CC)\}.
\]
显式平均公式立刻给出
\[
E_{\mathrm{geo}}(X_1,X_2,X_3)=(\bar X,\bar X,\bar X).
\]
于是
\[
E_{\mathrm{def}}=I-E_{\mathrm{geo}}
\]
满足
\[
E_{\mathrm{def}}(X_1,X_2,X_3)=(X_1-\bar X,\ X_2-\bar X,\ X_3-\bar X),
\]
故其像恰为和为零的子空间 \(A_{\partial}^{\mathrm{def}}\)。由此得到严格直和分裂
\[
A_{\partial}=A_{\partial}^{\mathrm{geo}}\oplus A_{\partial}^{\mathrm{def}}.
\]
维数计算给出
\[
\dim_{\CC}A_{\partial}^{\mathrm{geo}}=4,\qquad
\dim_{\CC}A_{\partial}^{\mathrm{def}}=12-4=8.
\]
最后，三块 \(M_2(\CC)\) 的循环排列与 \(C_3\) 的正则表示一致，故
\[
A_{\partial}\cong M_2(\CC)\otimes \CC[C_3].
\]
而 \(E_{\mathrm{def}}\) 正对应群代数中的增广投影，故
\[
A_{\partial}^{\mathrm{def}}
\cong
M_2(\CC)\otimes \mathrm{Aug}(\CC[C_3]).
\]
证毕。
\end{proof}

\begin{theorem}[boundary parity 八扇区上的 exponential heat-trace 精确等分]\label{thm:conclusion-window6-boundary-eightsector-heattrace-equipartition}
记
\[
Z_6^{\mathrm{bd}}\cong (\ZZ/2\ZZ)^3,
\qquad
\widehat Z_6^{\mathrm{bd}}:=\Hom(Z_6^{\mathrm{bd}},\{\pm 1\}).
\]
设
\[
\CC[G_{\mathrm{bin},6}]
\cong
\bigoplus_{\chi\in\widehat Z_6^{\mathrm{bd}}}
e_{\chi}\,\CC[G_{\mathrm{bin},6}]
\cong
\bigoplus_{\chi\in\widehat Z_6^{\mathrm{bd}}}\CC[G_{\mathrm{int},6}],
\]
其中
\[
e_{\chi}
=
\frac{1}{8}\sum_{g\in Z_6^{\mathrm{bd}}}\chi(g)\,g.
\]
则对任意 \(A\in \CC[G_{\mathrm{int},6}]\) 与任意整函数 \(f\)，若 \(A_{\chi}\) 表示 \(A\) 在第 \(\chi\)-扇区中的拷贝，则
\[
\operatorname{Tr}_{\mathrm{reg}}\!\bigl(f(A_{\chi})\bigr)
\]
与 \(\chi\) 无关。特别地，
\[
\operatorname{Tr}_{\mathrm{reg}}\!\bigl(e^{A_{\chi}}\bigr)
\]
在八个 boundary parity 扇区上完全相同。
\end{theorem}
\begin{proof}
文稿已给出
\[
G_{\mathrm{bin},6}\cong Z_6^{\mathrm{bd}}\times G_{\mathrm{int},6},
\]
从而
\[
\CC[G_{\mathrm{bin},6}]
\cong
\CC[Z_6^{\mathrm{bd}}]\otimes \CC[G_{\mathrm{int},6}].
\]
又由于 \(Z_6^{\mathrm{bd}}\) 为有限交换群，其复群代数分解为
\[
\CC[Z_6^{\mathrm{bd}}]\cong \bigoplus_{\chi\in\widehat Z_6^{\mathrm{bd}}}\CC e_{\chi}.
\]
张量化后得到
\[
\CC[G_{\mathrm{bin},6}]
\cong
\bigoplus_{\chi\in\widehat Z_6^{\mathrm{bd}}}
(\CC e_{\chi}\otimes \CC[G_{\mathrm{int},6}])
\cong
\bigoplus_{\chi\in\widehat Z_6^{\mathrm{bd}}}\CC[G_{\mathrm{int},6}].
\]
若 \(A\in \CC[G_{\mathrm{int},6}]\)，则其在总代数中的提升是 \(1\otimes A\)。在上述直和分解下，\(1\otimes A\) 在每个扇区上的限制都等于同一个算子 \(A\)。整函数演算逐块进行，故 \(f(1\otimes A)\) 在八个扇区上的对角块全部相同，从而正则表示下的块迹完全一致。取 \(f(z)=e^z\) 即得最后一式。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 隐结构的 \texorpdfstring{$(2+3)$}{(2+3)} 严格分裂]\label{thm:conclusion-window6-hidden-structure-two-plus-three-splitting}
window-\(6\) 中存在两类本质不同且不可互相吸收的隐藏自由度：
\begin{enumerate}
  \item 系数活跃部分恰由二维 Boolean 寄存器
  \[
  \{0,1\}^2\xrightarrow{\ \beta\ }\{0,21,34,55\}
  \]
  控制，并完全负责定理 \ref{thm:conclusion-window6-hidden-strip-fourlayer-partition}、推论 \ref{cor:conclusion-window6-dyadic-geometric-series-hidden-strip-telescope} 与定理 \ref{thm:conclusion-window6-e64-finite-borel-partition} 中的四层 hidden-strip 与有限 Borel 剖分；
  \item 系数中性部分恰为 boundary parity 的三维二挠环面
  \[
  Z_6^{\mathrm{bd}}\cong T_6[2]\cong (\ZZ/2\ZZ)^3,
\qquad
  r_{\mathrm{tor}}(Z_6^{\mathrm{bd}})=3,
  \]
  它只负责八重超选分解与定理 \ref{thm:conclusion-window6-boundary-eightsector-heattrace-equipartition} 的 heat-trace 等分，而不改动任何 finite Borel 系数。
\end{enumerate}
更强地，boundary parity 在交换化电荷格中正对应零权 Cartan 部分，而非根方向电荷；因此它既不进入 \(\beta\)-寄存器控制的四层 strip，也不进入 \(E_{64}(t)\) 的系数重组。反之，二维 \(\beta\)-寄存器也不可能忠实承载三维 boundary parity。
\end{theorem}
\begin{proof}
第一部分由定理 \ref{thm:conclusion-window6-hidden-strip-fourlayer-partition}、推论 \ref{cor:conclusion-window6-dyadic-geometric-series-hidden-strip-telescope} 与定理 \ref{thm:conclusion-window6-e64-finite-borel-partition} 直接给出：window-\(6\) 的前 \(64\) 个 dyadic 位置被精确拆成四层 strip，而这些 strip 正是由
\[
\beta(0,0)=0,\qquad
\beta(1,0)=21,\qquad
\beta(0,1)=34,\qquad
\beta(1,1)=55
\]
四个偏移组织出来的。因此有限 Borel 剖分的系数几何完全由二维 Boolean 寄存器控制。

第二部分由定理 \ref{thm:conclusion-window6-boundary-torus-2torsion} 与定理 \ref{thm:conclusion-window6-boundary-parity-minimal-torus-rank} 给出：
\[
Z_6^{\mathrm{bd}}\cong T_6[2]\cong (\ZZ/2\ZZ)^3,
\qquad
r_{\mathrm{tor}}(Z_6^{\mathrm{bd}})=3.
\]
再由定理 \ref{thm:conclusion-window6-boundary-eightsector-heattrace-equipartition}，这一三维二挠只在八个 boundary 角色扇区的 heat-trace 分配上留下作用，而对每个扇区中的 interior 整函数演算系数都保持中性。

此外，定理 \ref{thm:window6-abelianized-parity-charge-root-cartan-splitting} 已将 boundary parity 识别为交换化电荷格中的零权 Cartan 部分，而非根方向电荷；故它不可能回写到由 \(\beta\)-偏移控制的 strip 系数重组。反过来，二维 Boolean 寄存器至多提供两个独立的二值方向，不能忠实承载秩为 \(3\) 的 \((\ZZ/2\ZZ)^3\)。故这两类隐藏自由度严格分裂且不可互相吸收。证毕。
\end{proof}
