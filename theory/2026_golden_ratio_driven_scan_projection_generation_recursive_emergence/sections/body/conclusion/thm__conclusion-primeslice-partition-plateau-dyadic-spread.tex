\begin{theorem}[Smith \texorpdfstring{$p$}{p}-核谱函数的分拆坐标、局部曲率恢复与 plateau 最短前缀]\label{thm:conclusion-smith-pkernel-partition-curvature-plateau}
固定素数 \(p\)。设 \(f\not\equiv 0\) 且
\[
f(k)=\sum_{\ell\ge 1}m_\ell g_\ell(k),
\qquad
g_\ell(k):=\min(k,\ell),
\qquad
m_\ell\in \ZZ_{\ge 0},
\]
且仅有限多个 \(m_\ell\) 非零。定义
\[
L(f):=\sum_{\ell\ge 1}m_\ell,
\qquad
H(f):=\max\{\ell:m_\ell\neq 0\},
\qquad
M(f):=\sum_{\ell\ge 1}\ell\,m_\ell.
\]
则成立：
\begin{enumerate}
\item 前向差分满足
\[
\Delta f(k):=f(k)-f(k-1)=\sum_{\ell\ge k}m_\ell
\qquad(k\ge 1,\ f(0):=0),
\]
因而 \((\Delta f(k))_{k\ge 1}\) 是有限支撑、非负、非增整数序列；
\item 映射
\[
f\longmapsto \lambda(f):=\bigl(1^{m_1}2^{m_2}3^{m_3}\cdots\bigr)
\]
给出 Smith \(p\)-核谱函数与整数分拆之间的自然双射，并满足
\[
|\lambda(f)|=M(f),
\qquad
\ell(\lambda(f))=L(f),
\qquad
\lambda_1(f)=H(f);
\]
\item 局部二阶差分完全恢复原子系数：
\[
m_\ell=2f(\ell)-f(\ell-1)-f(\ell+1)
\qquad(1\le \ell<H(f)),
\]
\[
m_{H(f)}=f(H(f))-f(H(f)-1),
\]
其中约定 \(f(H(f)+1)=f(H(f))\)；
\item 前缀
\[
\bigl(f(1),\dots,f(H(f))\bigr)
\]
已经足以唯一恢复整个谱函数 \(f\)，并且这是统一意义下的最短充分前缀；等价地，
\[
H(f)=\min\{k\ge 0:\ f(k)=f(k+1)\}.
\]
\end{enumerate}
\leanverified{paper\_conclusion\_smith\_p\_kernel\_partition\_curvature\_plateau}
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-smith-p-kernel-spectrum-discrete-concave-characterization} 与推论 \ref{cor:conclusion-smith-p-kernel-spectrum-free-commutative-monoid} 已给出：每个 Smith \(p\)-核谱函数都唯一表示为
\[
f=\sum_{\ell\ge 1}m_\ell g_\ell,
\qquad
g_\ell(k)=\min(k,\ell),
\]
其中 \(m_\ell\in \ZZ_{\ge 0}\) 且仅有限多个非零。由
\[
g_\ell(k)-g_\ell(k-1)=
\begin{cases}
1,& \ell\ge k,\\
0,& \ell<k,
\end{cases}
\]
立得
\[
\Delta f(k)=\sum_{\ell\ge k}m_\ell.
\]
这立即给出 \((\Delta f(k))_{k\ge 1}\) 的有限支撑、非负与非增性，并说明
\[
\lambda(f):=\bigl(1^{m_1}2^{m_2}3^{m_3}\cdots\bigr)
\]
确为一个整数分拆；反向上，任意分拆都唯一决定 \((m_\ell)\)，再由原子分解唯一恢复 \(f\)，从而得到双射。三条标量恒等式
\[
|\lambda(f)|=\sum_{\ell\ge 1}\ell\,m_\ell=M(f),
\qquad
\ell(\lambda(f))=\sum_{\ell\ge 1}m_\ell=L(f),
\qquad
\lambda_1(f)=H(f)
\]
也随之成立。

再由
\[
m_\ell=\Delta f(\ell)-\Delta f(\ell+1)
\]
可得当 \(1\le \ell<H(f)\) 时，
\[
m_\ell
=
\bigl(f(\ell)-f(\ell-1)\bigr)-\bigl(f(\ell+1)-f(\ell)\bigr)
=
2f(\ell)-f(\ell-1)-f(\ell+1).
\]
而对 \(\ell=H(f)\)，由于 \(H(f)\) 之后无更高原子，故 \(\Delta f(H(f)+1)=0\)，从而
\[
m_{H(f)}=\Delta f(H(f))=f(H(f))-f(H(f)-1).
\]
这证明了局部曲率恢复公式。

由上式可知，只要给定前缀 \((f(1),\dots,f(H(f)))\)，便可先恢复 \(m_{H(f)}\)，再自上而下依次恢复全部 \(m_\ell\)，从而恢复整个 \(f\)。另一方面，对任意 \(K\ge 1\)，两个原子函数
\[
g_{K+1},
\qquad
g_{K+2}
\]
在 \(1,\dots,K\) 上完全一致，但它们显然不同，因此不存在只依赖于长度 \(K\) 前缀的统一恢复规则。这证明了最短性。最后，由
\[
\Delta f(k+1)=0
\iff
f(k+1)=f(k)
\]
及
\[
\Delta f(k+1)=\sum_{\ell\ge k+1}m_\ell
\]
可知首次 plateau 恰发生在 \(k=H(f)\)。证毕。
\end{proof}

\begin{theorem}[固定总亏损质量与原子数下的 pointwise 双极值及 plateau 区间]\label{thm:conclusion-smith-pkernel-fixedmass-balance-spike-extremal}
固定正整数 \(E,s\)。考虑全部满足
\[
M(f)=E,
\qquad
L(f)=s
\]
的 Smith \(p\)-核谱函数。把 \(E\) 作 Euclidean 分解
\[
E=sq+r,
\qquad
0\le r<s.
\]
则对每个 \(k\ge 1\)，所有此类谱函数 \(f\) 都满足 pointwise 夹逼
\[
f_{\mathrm{sp}}(k)\le f(k)\le f_{\mathrm{bal}}(k),
\]
其中
\[
f_{\mathrm{sp}}(k):=(s-1)\min(k,1)+\min(k,E-s+1),
\]
\[
f_{\mathrm{bal}}(k):=(s-r)\min(k,q)+r\min(k,q+1).
\]
上下界分别由集中分拆
\[
(E-s+1,1,\dots,1)
\]
与均衡分拆
\[
(\underbrace{q+1,\dots,q+1}_{r},\underbrace{q,\dots,q}_{s-r})
\]
唯一达到。特别地，plateau 检测门槛满足
\[
\left\lceil\frac{E}{s}\right\rceil\le H(f)\le E-s+1,
\]
并且两端分别由 \(f_{\mathrm{bal}}\) 与 \(f_{\mathrm{sp}}\) 取到。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-smith-pkernel-partition-curvature-plateau}，条件
\[
M(f)=E,
\qquad
L(f)=s
\]
等价于把整数 \(E\) 分拆成恰好 \(s\) 个正部件
\[
\lambda_1,\dots,\lambda_s,
\qquad
\sum_{i=1}^{s}\lambda_i=E,
\]
并写成
\[
f(k)=\sum_{i=1}^{s}\min(k,\lambda_i).
\]
对每个固定 \(k\)，函数
\[
\phi_k(x):=\min(k,x)
\qquad(x\in \RR_{\ge 0})
\]
是凹函数。故由 Karamata 不等式，\(\sum_i\phi_k(\lambda_i)\) 在所有具有固定和 \(E\) 与固定部件数 \(s\) 的分拆中，于最均衡分拆处取最大，于最集中的分拆处取最小。最均衡分拆正是
\[
(\underbrace{q+1,\dots,q+1}_{r},\underbrace{q,\dots,q}_{s-r}),
\]
其对应值为
\[
f_{\mathrm{bal}}(k)=(s-r)\min(k,q)+r\min(k,q+1).
\]
最集中分拆则为
\[
(E-s+1,1,\dots,1),
\]
其对应值为
\[
f_{\mathrm{sp}}(k)=(s-1)\min(k,1)+\min(k,E-s+1).
\]
于是 pointwise 夹逼与 sharp 性同时成立。

最后，由定理 \ref{thm:conclusion-smith-pkernel-partition-curvature-plateau}，
\[
H(f)=\max_{1\le i\le s}\lambda_i.
\]
在固定总和与部件数的分拆中，最大部件最小时恰为均衡分拆，其值为 \(\lceil E/s\rceil\)；最大部件最大时恰为集中分拆，其值为 \(E-s+1\)。故得所示 plateau 区间。证毕。
\end{proof}

\begin{theorem}[prime-slice 层号的 \texorpdfstring{$2$}{2}-进赋值识别与基础设施的 spread/inventory 刚性分裂]\label{thm:conclusion-primeslice-v2-layer-spread-inventory}
沿用分层 prime-slice 记号
\[
\mathcal P_j:=\{p_{2^j(2n-1)}:n\ge 1\}
\qquad(j\ge 0).
\]
则：
\begin{enumerate}
\item 对任意素数标签 \(p_t\)，其所属层唯一由
\[
j=v_2(t)
\]
决定；亦即
\[
p_t\in \mathcal P_j
\iff
j=v_2(t).
\]
\item 因而任意有限 prime-label 集合 \(\Lambda\subset \mathbb P\) 都具有规范正交分解
\[
\Lambda=\bigsqcup_{j\ge 0}\bigl(\Lambda\cap \mathcal P_j\bigr),
\]
并且该分块只由素数序号的 \(2\)-进赋值决定。
\item 若深度为 \(k\) 的精确分层外置在每层都取最小字母表
\[
\Lambda_j^{\min}
=
\{p_{2^j},p_{3\cdot 2^j},\dots,p_{(2B_j-1)2^j}\},
\qquad
B_j\le B,
\]
则
\[
\max_{0\le j\le k-1}\max \Lambda_j^{\min}
=
p_{(2B-1)2^{k-1}},
\]
并且
\[
\log p_{(2B-1)2^{k-1}}
=
(k-1)\log 2+\log(2B-1)+\log k+O(1).
\]
另一方面，
\[
\sum_{j=0}^{k-1}\#\Lambda_j^{\min}
=
\sum_{j=0}^{k-1}B_j
\le kB.
\]
因此，在统一分支上界 \(B\) 下，最远 prime-label 的指标尺度呈 dyadic 指数扩张，而 total inventory 只线性增长。
\item 特别地，对固定 \(B\)，任何只含 \(O(k)\) 个连续 prime-index 的窗口，都不可能在所有充分大的 \(k\) 上同时承载深度 \(k\) 的全部最优层字母表。
\end{enumerate}
\end{theorem}
\begin{proof}
任意正整数 \(t\) 都唯一表示为
\[
t=2^j(2n-1),
\qquad
j=v_2(t),
\]
其中 \(2n-1\) 为奇数。由 \(\mathcal P_j\) 的定义，\(p_t\) 恰在且只在这一层出现，从而得到前两条。

第三条的最小字母表显示公式与最大 prime 的 asymptotic 正是定理 \ref{thm:conclusion-layered-primeslice-dyadic-head-law} 的内容；而 inventory 公式只是逐层求和。二者并列即得 spread/inventory 的刚性分裂。

最后，若一个连续 prime-index 窗口只含 \(O(k)\) 个索引，则其可见的最大 prime-index 也仅为线性量级；但第三条表明，深度 \(k\) 的最优层字母表所需的最远 prime-index 已达到
\[
(2B-1)2^{k-1},
\]
其增长为指数级。故当 \(k\) 足够大时，任何线性长度窗口都必遗漏顶层所需标签，从而不可能同时承载全部最优层字母表。证毕。
\end{proof}

\begin{conjecture}[slice-respecting 最优实现数只依赖于原子分拆型]\label{conj:conclusion-primeslice-optimal-count-partition-type}
设
\[
f=\sum_{\ell\ge 1}m_\ell g_\ell
\]
为一个 Smith \(p\)-核谱函数，并在与定理 \ref{thm:conclusion-primeslice-optimal-implementations-falling-factorial} 相同的 slice-respecting 最优实现语义下，记其最优 prime-slice 实现总数为 \(N_{\mathrm{opt}}(f)\)。则 \(N_{\mathrm{opt}}(f)\) 应仅依赖于原子分拆型
\[
\lambda(f)=\bigl(1^{m_1}2^{m_2}3^{m_3}\cdots\bigr),
\]
并可分解为若干仅由 \((m_\ell)_{\ell\ge 1}\) 决定的 falling factorial 因子之乘积。特别地，任何两条具有相同原子分拆型的谱函数，在最优 prime-slice 计数上都应不可区分。
\end{conjecture}
