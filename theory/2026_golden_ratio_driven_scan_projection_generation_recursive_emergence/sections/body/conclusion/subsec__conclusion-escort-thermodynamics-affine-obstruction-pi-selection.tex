\subsection{escort 热力学的二点演算、输出位势的仿射障碍与 \texorpdfstring{$\pi$}{pi} 恢复的临界选择}\label{subsec:conclusion-escort-thermodynamics-affine-obstruction-pi-selection}
沿用定理 \ref{thm:conclusion-fold-onebit-information-geometry-collapse}、定理 \ref{thm:xi-time-part9ob-escort-loginfo-cumulants-negative-skew}、推论 \ref{cor:xi-time-part9o-escort-fiberinfo-constantterm-inversion}、命题 \ref{prop:real-input-40-output-cumulants-closed}、表 \ref{tab:real_input_40_output_potential_cumulants_closed}、推论 \ref{cor:fold-bin-recover-2pi} 与定理 \ref{thm:xi-fold-hidden-logmultiplicity-kappa-gauge-two-scale-explicit} 的记号。前述链条已经分别把 escort 采样下的末位二点塌缩、纤维信息常数项、真实输入 \(40\) 状态核输出位势的局部累积量，以及 fold-gauge 常数列对 \(2\pi\) 的内生恢复固定为显式闭式。把这些接口并置，可进一步抽出如下六条彼此锁定的后果。

\begin{theorem}[escort 热力学的二点泛函演算]\label{thm:conclusion-escort-thermodynamics-two-point-functional-calculus}
设
\[
a:=\log\!\Bigl(\frac{2}{\varphi}\Bigr),
\qquad
W_m^{(q)}\sim \pi_{m,q},
\qquad
Y_m^{(q)}:=ma-\log d_m^{\mathrm{bin}}(W_m^{(q)}).
\]
则对每个固定 \(q\ge 0\)，都有分布收敛
\[
Y_m^{(q)}\Rightarrow Y^{(q)},
\]
其中极限随机变量 \(Y^{(q)}\) 仅取两值
\[
Y^{(q)}\in\{0,\log\varphi\},
\qquad
\mathbb P\!\bigl(Y^{(q)}=\log\varphi\bigr)=\theta(q):=\frac{1}{1+\varphi^{q+1}},
\]
\[
\mathbb P\!\bigl(Y^{(q)}=0\bigr)=1-\theta(q)=\frac{\varphi^{q+1}}{1+\varphi^{q+1}}.
\]
从而对任意有界连续函数 \(F\) 都有
\[
\lim_{m\to\infty}\EE\!\bigl[F(Y_m^{(q)})\bigr]
=
\frac{\varphi^{q+1}}{1+\varphi^{q+1}}\,F(0)
+
\frac{1}{1+\varphi^{q+1}}\,F(\log\varphi).
\]
特别地，其极限累积母函数为
\[
K_q(t)
:=
\lim_{m\to\infty}\log\EE\!\bigl[e^{tY_m^{(q)}}\bigr]
=
\log\!\Bigl(\frac{\varphi^{q+1}+\varphi^{t}}{1+\varphi^{q+1}}\Bigr)
\qquad (t\in\RR).
\]
\leanverified{paper\_conclusion\_escort\_thermodynamics\_two\_point\_functional\_calculus}
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-time-part9ob-escort-loginfo-cumulants-negative-skew} 已经证明
\[
\log d_m^{\mathrm{bin}}(W_m^{(q)})-ma
\Rightarrow
\widetilde Y^{(q)},
\]
其中 \(\widetilde Y^{(q)}\) 仅取 \(0\) 与 \(-\log\varphi\) 两值，且
\[
\mathbb P\!\bigl(\widetilde Y^{(q)}=-\log\varphi\bigr)=\theta(q).
\]
由于
\[
Y_m^{(q)}
=
-\Bigl(\log d_m^{\mathrm{bin}}(W_m^{(q)})-ma\Bigr),
\]
故 \(Y_m^{(q)}\Rightarrow-\widetilde Y^{(q)}\)，即得所示两点极限律。对有界连续函数 \(F\) 的泛函演算由弱收敛立即推出。

最后，仍由定理 \ref{thm:xi-time-part9ob-escort-loginfo-cumulants-negative-skew} 的累积母函数公式，只需取 \(t\mapsto -t\)，便得到
\[
K_q(t)
=
\log\!\bigl(1-\theta(q)+\theta(q)\varphi^t\bigr)
=
\log\!\Bigl(\frac{\varphi^{q+1}+\varphi^t}{1+\varphi^{q+1}}\Bigr).
\]
证毕。
\end{proof}

\begin{corollary}[物理温度 \texorpdfstring{$q=1$}{q=1} 的黄金 Bernoulli 累积律]\label{cor:conclusion-physical-temperature-golden-bernoulli-cumulant-law}
\leanverified{paper\_conclusion\_physical\_temperature\_golden\_bernoulli\_cumulant\_law}
在定理 \ref{thm:conclusion-escort-thermodynamics-two-point-functional-calculus} 中取 \(q=1\)。记
\[
p_*:=\theta(1)=\frac{1}{1+\varphi^2}=\frac{1}{\varphi\sqrt5}
=\frac12-\frac{\sqrt5}{10},
\qquad
L:=\log\varphi.
\]
则
\[
Y^{(1)}=L\,B_*,
\qquad
B_*\sim \mathrm{Bernoulli}(p_*),
\]
并且其前四阶累积量满足
\[
\kappa_1=p_*L,
\qquad
\kappa_2=\frac15L^2,
\qquad
\kappa_3=\frac{\sqrt5}{25}L^3,
\qquad
\kappa_4=-\frac1{25}L^4.
\]
于是两项仿射不变量
\[
R_3:=\frac{\kappa_3^2}{\kappa_2^3}=1,
\qquad
R_4:=\frac{\kappa_4}{\kappa_2^2}=-1
\]
被完全锁定。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-escort-thermodynamics-two-point-functional-calculus}，
\[
Y^{(1)}\in\{0,L\},
\qquad
\mathbb P\!\bigl(Y^{(1)}=L\bigr)=p_*,
\]
故 \(Y^{(1)}=LB_*\)。

Bernoulli 变量的标准累积量公式给出
\[
\kappa_2(B_*)=p_*(1-p_*),
\qquad
\kappa_3(B_*)=p_*(1-p_*)(1-2p_*),
\]
\[
\kappa_4(B_*)=p_*(1-p_*)\bigl(1-6p_*(1-p_*)\bigr).
\]
而
\[
p_*(1-p_*)=\frac15,
\qquad
1-2p_*=\frac{\sqrt5}{5},
\]
故乘上尺度因子 \(L^r\) 后即得四个闭式。最后由定义直接计算，得到
\[
R_3=\frac{\kappa_3^2}{\kappa_2^3}=1,
\qquad
R_4=\frac{\kappa_4}{\kappa_2^2}=-1.
\]
证毕。
\end{proof}

\begin{theorem}[bin-fold 微观热场与 \texorpdfstring{$40$}{40} 状态输出位势之间的仿射累积障碍]\label{thm:conclusion-binfold-realinput40-affine-cumulant-obstruction}
记 \(P(\theta)=\log\lambda(e^\theta)\) 为真实输入 \(40\) 状态核的输出位势，并令 \(X\) 为满足
\[
\kappa_r(X)=P^{(r)}(0)
\qquad (r=1,2,3,4)
\]
的局部统计场。则不存在仿射变换 \(x\mapsto a+bx\)（其中 \(b\neq 0\)），使得推论 \ref{cor:conclusion-physical-temperature-golden-bernoulli-cumulant-law} 中的极限场 \(Y^{(1)}\) 与 \(X\) 在二、三、四阶累积量上同时匹配。等价地，真实输入 \(40\) 状态输出位势并非 bin-fold 物理温度微观热场的仿射热力学影像。
\end{theorem}
\begin{proof}
对任意随机变量 \(Z\)，记
\[
R_3(Z):=\frac{\kappa_3(Z)^2}{\kappa_2(Z)^3},
\qquad
R_4(Z):=\frac{\kappa_4(Z)}{\kappa_2(Z)^2}.
\]
若 \(b\neq 0\)，则高阶累积量在仿射变换下满足
\[
\kappa_r(a+bZ)=b^r\kappa_r(Z)
\qquad (r\ge 2),
\]
故 \(R_3\) 与 \(R_4\) 都是仿射不变量。

由推论 \ref{cor:conclusion-physical-temperature-golden-bernoulli-cumulant-law}，
\[
R_3\!\bigl(Y^{(1)}\bigr)=1,
\qquad
R_4\!\bigl(Y^{(1)}\bigr)=-1.
\]
另一方面，命题 \ref{prop:real-input-40-output-cumulants-closed} 与表 \ref{tab:real_input_40_output_potential_cumulants_closed} 给出
\[
P''(0)\approx 0.0487412456183,
\qquad
P^{(3)}(0)\approx -0.0158881374259,
\qquad
P^{(4)}(0)\approx 0.00568478997567,
\]
从而
\[
R_3(X)=\frac{P^{(3)}(0)^2}{P''(0)^3}\approx 2.1799980451,
\qquad
R_4(X)=\frac{P^{(4)}(0)}{P''(0)^2}\approx 2.3928814165.
\]
二者与 \((1,-1)\) 均不相等，故不存在所述仿射匹配。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\pi$}{pi} 内生恢复的余维一 counterterm 定理]\label{thm:conclusion-pi-endogenous-recovery-codimone-counterterm}
记
\[
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
\eta_*:=\theta(1)=\frac{1}{1+\varphi^2}=\frac{1}{\varphi\sqrt5},
\]
\[
\kappa_m:=\EE_{\mu_m}\!\bigl[\log d_m^{\mathrm{bin}}(W_m)\bigr],
\qquad
W_m\sim \mu_m.
\]
沿用推论 \ref{cor:fold-bin-recover-2pi} 的常数列
\[
C_m
:=
\exp\!\Biggl(
\frac{2^{m+1}}{|X_m|}(g_m-\kappa_m+1)
-\frac{1}{|X_m|}\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)
\Biggr).
\]
现取任意实数 \(\eta\) 与任意序列 \((T_m)_{m\ge 1}\) 满足
\[
T_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\eta\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr),
\]
并定义对应的 counterterm 估计
\[
\mathcal C_m[T]
:=
\exp\!\Biggl(
\frac{2^{m+1}}{|X_m|}(g_m-T_m+1)
-\frac{1}{|X_m|}\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)
\Biggr).
\]
则有
\[
\log \mathcal C_m[T]
=
\log(2\pi)
+
\frac{2\sqrt5}{\varphi^2}\,(\eta-\eta_*)\log\varphi
\Bigl(\frac{2}{\varphi}\Bigr)^m
+O(1).
\]
因此：
\begin{enumerate}
  \item 若 \(\eta>\eta_*\)，则 \(\mathcal C_m[T]\to+\infty\)；
  \item 若 \(\eta<\eta_*\)，则 \(\mathcal C_m[T]\to 0\)；
  \item 只有在 \(\eta=\eta_*\) 这一余维 \(1\) 的超曲面上，才可能出现有限非零极限。
\end{enumerate}
更进一步，若 \(\eta=\eta_*\) 且
\[
\frac{2^{m+1}}{|X_m|}\bigl(T_m-\kappa_m\bigr)\to 0,
\]
则
\[
\mathcal C_m[T]\to 2\pi.
\]
\end{theorem}
\begin{proof}
由定义直接相减可得
\[
\log \mathcal C_m[T]-\log C_m
=
\frac{2^{m+1}}{|X_m|}\,(\kappa_m-T_m).
\]
而定理 \ref{thm:xi-fold-hidden-logmultiplicity-kappa-gauge-two-scale-explicit} 给出
\[
\kappa_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\eta_*\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr),
\]
故
\[
\kappa_m-T_m
=
(\eta-\eta_*)\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr).
\]
再由 Binet 公式，
\[
|X_m|=F_{m+2}
=
\frac{\varphi^{m+2}}{\sqrt5}+O(\varphi^{-m}),
\]
从而
\[
\frac{2^{m+1}}{|X_m|}
=
\frac{2\sqrt5}{\varphi^2}
\Bigl(\frac{2}{\varphi}\Bigr)^m(1+o(1)).
\]
于是
\[
\log \mathcal C_m[T]-\log C_m
=
\frac{2\sqrt5}{\varphi^2}\,(\eta-\eta_*)\log\varphi
\Bigl(\frac{2}{\varphi}\Bigr)^m
+O(1).
\]
最后，由推论 \ref{cor:fold-bin-recover-2pi}，\(\log C_m\to\log(2\pi)\)，故上式立即推出三分结论。

若进一步有
\[
\frac{2^{m+1}}{|X_m|}\bigl(T_m-\kappa_m\bigr)\to 0,
\]
则
\[
\log \mathcal C_m[T]-\log C_m
=
-\frac{2^{m+1}}{|X_m|}\bigl(T_m-\kappa_m\bigr)\to 0.
\]
再结合 \(C_m\to 2\pi\)，便得 \(\mathcal C_m[T]\to 2\pi\)。证毕。
\end{proof}

\begin{corollary}[指数倾斜温度的临界选择：\texorpdfstring{$\beta=\log\varphi$}{beta=log phi} 唯一]\label{cor:conclusion-lastbit-gibbs-critical-temperature-logphi}
\leanverified{paper\_conclusion\_lastbit\_gibbs\_critical\_temperature\_logphi}
对每个固定 \(\beta\in\RR\)，定义末位指数倾斜测度
\[
\nu_{m,\beta}(w):=\frac{1}{Z_{m,\beta}}\,u_m(w)e^{-\beta\ell(w)},
\qquad
u_m(w):=\frac{1}{|X_m|},
\qquad
\ell(w):=w_m\in\{0,1\},
\]
并按末位分块
\[
A_i:=\{w\in X_m:\ \ell(w)=i\}
\qquad (i=0,1),
\]
再记
\[
T_m(\beta):=\EE_{\nu_{m,\beta}}\!\bigl[\log d_m^{\mathrm{bin}}(W)\bigr],
\qquad
W\sim \nu_{m,\beta}.
\]
则有
\[
T_m(\beta)
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\vartheta(\beta)\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr),
\qquad
\vartheta(\beta):=\frac{1}{1+\varphi e^\beta}.
\]
令
\[
C_m^{(\beta)}:=\mathcal C_m[T(\beta)].
\]
则
\begin{enumerate}
  \item \(\beta<\log\varphi\) 时，\(C_m^{(\beta)}\to+\infty\)；
  \item \(\beta>\log\varphi\) 时，\(C_m^{(\beta)}\to0\)；
  \item \(\beta=\log\varphi\) 时，\(C_m^{(\beta)}\to 2\pi\)。
\end{enumerate}
故在全部末位 Gibbs 温度族中，\(\beta=\log\varphi\) 是唯一产生有限非零 \(\pi\)-极限的临界温度。
\end{corollary}
\begin{proof}
由 \(|A_0|=F_{m+1}\)、\(|A_1|=F_m\) 可得
\[
\nu_{m,\beta}(A_1)
=
\frac{F_m e^{-\beta}}{F_{m+1}+F_m e^{-\beta}}
\to
\frac{1}{1+\varphi e^\beta}
=
\vartheta(\beta).
\]
另一方面，由定理 \ref{thm:fold-bin-two-state-asymptotic} 的一致对数展开，
\[
\log d_m^{\mathrm{bin}}(w)
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\ell(w)\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr)
\]
在 \(X_m\) 上一致成立。对 \(\nu_{m,\beta}\) 取期望即得
\[
T_m(\beta)
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\vartheta(\beta)\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr).
\]

又因为
\[
\vartheta(\beta)=\eta_*
\iff
\frac{1}{1+\varphi e^\beta}=\frac{1}{1+\varphi^2}
\iff
e^\beta=\varphi
\iff
\beta=\log\varphi,
\]
故定理 \ref{thm:conclusion-pi-endogenous-recovery-codimone-counterterm} 立即给出前两条结论，并说明只有 \(\beta=\log\varphi\) 处才可能保留有限非零极限。

最后，定理 \ref{thm:conclusion-fold-onebit-information-geometry-collapse} 的第 \(1\) 条已经说明：物理分布 \(\mu_m\) 在指数精度下正由稳定层均匀分布的末位 Gibbs 倾斜实现，而其倾斜参数趋于 \(\log\varphi\)。因此 \(\beta=\log\varphi\) 恰对应于内生恢复序列 \(C_m\to 2\pi\) 的物理固定点，从而
\[
C_m^{(\log\varphi)}\to 2\pi.
\]
证毕。
\end{proof}

\begin{corollary}[escort 温度族的 \texorpdfstring{$\pi$}{pi} 灾变：唯有 \texorpdfstring{$q=1$}{q=1} 存活]\label{cor:conclusion-escort-temperature-pi-catastrophe-q1-only}
令
\[
\kappa_m(q):=\EE_{\pi_{m,q}}\!\bigl[\log d_m^{\mathrm{bin}}(W_m^{(q)})\bigr],
\qquad
W_m^{(q)}\sim \pi_{m,q},
\qquad
C_m^{\langle q\rangle}:=\mathcal C_m[\kappa(q)].
\]
则有
\[
\kappa_m(q)
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\frac{\log\varphi}{1+\varphi^{q+1}}
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr).
\]
于是
\[
\log C_m^{\langle q\rangle}
=
\log(2\pi)
+
\frac{2\sqrt5\log\varphi}{\varphi^2}
\left(
\frac{1}{1+\varphi^{q+1}}-\frac{1}{1+\varphi^2}
\right)
\Bigl(\frac{2}{\varphi}\Bigr)^m
+O(1).
\]
特别地，
\begin{enumerate}
  \item \(q<1\) 时，\(C_m^{\langle q\rangle}\to+\infty\)；
  \item \(q>1\) 时，\(C_m^{\langle q\rangle}\to0\)；
  \item \(q=1\) 时，\(C_m^{\langle q\rangle}\to 2\pi\)。
\end{enumerate}
故在项目内部，\(\pi\) 并不沿 escort 温度 \(q\) 形成一条可变形常数族；它是 \(q=1\) 处的唯一临界固定点。
\end{corollary}
\begin{proof}
推论 \ref{cor:xi-time-part9o-escort-fiberinfo-constantterm-inversion} 已给出
\[
\kappa_m(q)
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)-\eta(q)\log\varphi
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr),
\qquad
\eta(q):=\frac{1}{1+\varphi^{q+1}}.
\]
把 \(\eta(q)\) 代入定理 \ref{thm:conclusion-pi-endogenous-recovery-codimone-counterterm} 便得到第二式与前两条三分结论。

当 \(q=1\) 时，
\[
C_m^{\langle 1\rangle}
=
\mathcal C_m[\kappa(1)]
=
\mathcal C_m[\kappa]
=
C_m,
\]
故由推论 \ref{cor:fold-bin-recover-2pi} 直接得到
\[
C_m^{\langle 1\rangle}\to 2\pi.
\]
证毕。
\end{proof}

\begin{corollary}[物理温度 \texorpdfstring{$q=1$}{q=1} 的唯一三重锁定]\label{cor:conclusion-physical-temperature-q1-unique-triple-lock}
\leanverified{paper\_conclusion\_physical\_temperature\_q1\_unique\_triple\_lock}
记
\[
\eta_*:=\frac{1}{1+\varphi^2}.
\]
则在末位 Gibbs 温度线
\[
\vartheta(\beta):=\frac{1}{1+\varphi e^\beta}
\]
与 escort 温度线
\[
\eta(q):=\frac{1}{1+\varphi^{q+1}}
\]
上，方程
\[
\vartheta(\beta)=\eta_*,
\qquad
\eta(q)=\eta_*
\]
的唯一解分别为
\[
\beta=\log\varphi,
\qquad
q=1.
\]
并且在这一唯一热临界点上，极限两点热场 \(Y^{(1)}\) 的仿射不变量锁定为
\[
R_3\bigl(Y^{(1)}\bigr)=1,
\qquad
R_4\bigl(Y^{(1)}\bigr)=-1,
\]
而真实输入 \(40\) 状态输出位势对应的局部统计场 \(X\) 满足
\[
R_3(X)\approx 2.1799980451,
\qquad
R_4(X)\approx 2.3928814165.
\]
故 \(q=1\) 不只是产生有限非零 \(\pi\)-极限的唯一 escort 温度；它也是唯一同时命中内生 \(\pi\)-恢复临界面与两点热场标度的温度点，而该热场又与真实输入 \(40\) 状态位势严格非仿射同一。
\end{corollary}
\begin{proof}
由推论 \ref{cor:conclusion-lastbit-gibbs-critical-temperature-logphi}，
\[
\vartheta(\beta)=\eta_*
\iff
\beta=\log\varphi,
\]
且只有在该点才有
\(
C_m^{(\beta)}\to 2\pi
\)。
由推论 \ref{cor:conclusion-escort-temperature-pi-catastrophe-q1-only}，
\[
\eta(q)=\eta_*
\iff
q=1,
\]
且只有在该点才有
\(
C_m^{\langle q\rangle}\to 2\pi
\)。

另一方面，推论 \ref{cor:conclusion-physical-temperature-golden-bernoulli-cumulant-law} 已给出
\[
R_3\bigl(Y^{(1)}\bigr)=1,
\qquad
R_4\bigl(Y^{(1)}\bigr)=-1,
\]
而定理 \ref{thm:conclusion-binfold-realinput40-affine-cumulant-obstruction} 给出
\[
R_3(X)\approx 2.1799980451,
\qquad
R_4(X)\approx 2.3928814165,
\]
从而 \(Y^{(1)}\) 与 \(X\) 不可能仿射同一。把这些结论并置即得所述三重锁定。证毕。
\end{proof}

\noindent
因此，escort 热力学在极限上并不保留一条可自由变形的连续微观场谱，而是塌缩为单一的 Bernoulli 两点场；这一两点场的仿射累积不变量又与真实输入 \(40\) 状态输出位势的局部热统计严格不兼容；同时，fold-gauge 的 \(\pi\) 恢复也不是一族可任意变温的常数公式，而是一个余维一的 counterterm 选择问题，其中只有物理温度 \(q=1\)（等价的末位 Gibbs 临界温度 \(\beta=\log\varphi\)）保留有限非零极限。由此，末位二点热场、输出位势的非仿射性与 \(2\pi\) 的内生恢复便被压缩为同一条热力学选择原理。
