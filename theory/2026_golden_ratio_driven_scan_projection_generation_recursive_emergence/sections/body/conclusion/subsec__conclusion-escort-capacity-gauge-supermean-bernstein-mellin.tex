沿用小节 \ref{subsec:conclusion-maxfiber-freezing-capacity-entropy-separation}、小节 \ref{subsec:conclusion-capacity-mellin-thermodynamic-wedderburn-holography}、小节 \ref{subsec:conclusion-maxfiber-average-scale-rigidity} 与小节 \ref{subsec:conclusion-binfold-recovery-entropy-golden-resonance} 的记号。前述链条已经分别把冻结相 escort 分布、连续容量曲线的 Wedderburn 全息、fold-gauge 体积密度以及黄金共振造成的碰撞/峰值偏置压缩为显式闭式。将这些接口并置之后，还可进一步抽取出如下几条派生结论。

\begin{theorem}[escort Rényi 熵的全阶冻结与亚单位阶阈值重整化]\label{thm:conclusion-escort-renyi-all-orders-renormalized-threshold}
沿用定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 的记号。对 \(q\in(0,\infty]\)，定义
\[
a_{\mathrm c}(q):=
\frac{\log\varphi-g_\ast}{\min\{q,1\}\,(\alpha_\ast-\alpha_\ast^{(2)})}.
\]
若
\[
a>a_{\mathrm c}(q),
\]
则 escort 分布
\[
\pi_{m,a}(x):=\frac{d_m(x)^a}{S_a(m)},
\qquad
S_a(m):=\sum_{x\in X_m}d_m(x)^a
\]
的 Rényi 熵率满足
\[
\lim_{m\to\infty}\frac1m H_q(\pi_{m,a})=g_\ast,
\]
其中 \(q=1\) 按 Shannon 熵解释，而
\[
H_\infty(\pi_{m,a})=-\log\max_x\pi_{m,a}(x).
\]
换言之，一旦 \(a\) 超过经 \(q\) 重整化后的冻结阈值，全部 Rényi 阶都塌缩到同一基态简并指数 \(g_\ast\)。
\end{theorem}
\begin{proof}
当 \(q\ge 1\) 时有 \(a_{\mathrm c}(q)=a_{\mathrm c}\)，故结论已由定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 给出；其中 \(q=\infty\) 的情形亦包含在该定理之内。

现设 \(0<q<1\)，并记
\[
\Delta\alpha:=\alpha_\ast-\alpha_\ast^{(2)}>0,
\qquad
A_m^{\max}:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
p_m:=\pi_{m,a}(A_m^{\max}).
\]
由 \(a>a_{\mathrm c}(q)=a_{\mathrm c}/q>a_{\mathrm c}\)，定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 的证明机制给出
\[
p_m\ge 1-\exp(-m\Delta(a)+o(m)),
\]
其中 \(\Delta(a)>0\)。又由于 \(A_m^{\max}\) 上每个点都具有相同权重 \(M_m^a/S_a(m)\)，故
\[
\sum_{x\in A_m^{\max}}\pi_{m,a}(x)^q
=
K_m\left(\frac{p_m}{K_m}\right)^q
=
p_m^q K_m^{1-q}
=
K_m^{1-q}(1+o(1)).
\]

对非基态部分，利用
\[
S_a(m)\ge K_mM_m^a
\]
可得
\[
\max_{x\notin A_m^{\max}}\pi_{m,a}(x)
\le
\frac{(M_m^{(2)})^a}{K_mM_m^a}
=
\exp\!\bigl(-m(a\Delta\alpha+g_\ast)+o(m)\bigr).
\]
于是
\[
\sum_{x\notin A_m^{\max}}\pi_{m,a}(x)^q
\le
|X_m|\,
\exp\!\bigl(-mq(a\Delta\alpha+g_\ast)+o(m)\bigr).
\]
由于 \(|X_m|=\exp(m\log\varphi+o(m))\)，右端指数为
\[
\log\varphi-q(a\Delta\alpha+g_\ast).
\]
而由 \(a>a_{\mathrm c}/q\) 得
\[
qa\Delta\alpha>\log\varphi-g_\ast,
\]
故
\[
\log\varphi-q(a\Delta\alpha+g_\ast)<(1-q)g_\ast.
\]
这表明尾和的指数阶严格低于主项
\[
K_m^{1-q}=\exp((1-q)g_\ast m+o(m)).
\]
因此
\[
\sum_{x\in X_m}\pi_{m,a}(x)^q
=
K_m^{1-q}(1+o(1)),
\]
从而
\[
H_q(\pi_{m,a})
=
\frac{1}{1-q}\log\sum_x \pi_{m,a}(x)^q
=
\log K_m+o(m).
\]
再用 \(\log K_m=g_\ast m+o(m)\)，即得
\[
\lim_{m\to\infty}\frac1m H_q(\pi_{m,a})=g_\ast.
\]
证毕。
\end{proof}

\begin{theorem}[连续容量曲线与纤维群胚半单代数的互逆完备性]\label{thm:conclusion-capacity-groupoid-bidirectional-complete-invariant}
\leanverified{paper\_conclusion\_capacity\_groupoid\_bidirectional\_complete\_invariant}
在 bin-fold 口径下，记
\[
\mathcal C_m^{\mathrm{cont}}(T):=
\sum_{w\in X_m}\min\bigl(d_m^{\mathrm{bin}}(w),T\bigr),
\qquad
h_m(n):=\#\{w\in X_m:\ d_m^{\mathrm{bin}}(w)=n\},
\]
并记
\[
A_m^{\mathrm{bin}}:=\CC[\mathcal R_m^{\mathrm{bin}}].
\]
则 \(\mathcal C_m^{\mathrm{cont}}(\cdot)\) 与 \(A_m^{\mathrm{bin}}\) 的半单复代数同构类互为完备不变量。更具体地，
\[
A_m^{\mathrm{bin}}
\cong
\bigoplus_{n\ge 1} M_n(\CC)^{\oplus h_m(n)},
\]
而直方图 \((h_m(n))_{n\ge 1}\) 可由 \(\mathcal C_m^{\mathrm{cont}}\) 唯一恢复；反之，代数同构类也唯一决定全部 \(h_m(n)\)，从而唯一决定 \(\mathcal C_m^{\mathrm{cont}}\)。
\end{theorem}
\begin{proof}
定理 \ref{thm:conclusion-wedderburn-threshold-holography} 已给出：连续容量曲线、尾计数函数、纤维直方图以及半单代数同构类四类数据彼此唯一决定。故只需把其中两端抽出即可。

更具体地，由定理 \ref{thm:conclusion-continuous-capacity-integral-node-inversion}，
\[
N_m^{\ge}(n)=\mathcal C_m^{\mathrm{cont}}(n)-\mathcal C_m^{\mathrm{cont}}(n-1),
\]
从而
\[
h_m(n)=N_m^{\ge}(n)-N_m^{\ge}(n+1)
\]
唯一恢复。另一方面，由命题 \ref{prop:fold-bin-groupoid-algebra-wedderburn}，
\[
A_m^{\mathrm{bin}}
\cong
\bigoplus_{w\in X_m}M_{d_m^{\mathrm{bin}}(w)}(\CC)
\cong
\bigoplus_{n\ge 1} M_n(\CC)^{\oplus h_m(n)}.
\]
有限维半单复代数的 Artin--Wedderburn 分解在块尺寸与重数上唯一，故代数同构类反向唯一决定 \((h_m(n))\)。最后代回
\[
\mathcal C_m^{\mathrm{cont}}(T)=\sum_{n\ge 1}h_m(n)\min(n,T)
\]
即可恢复容量曲线。证毕。
\end{proof}

\begin{theorem}[fold-gauge 体积的双指数熵律]\label{thm:conclusion-binfold-gauge-volume-double-exponential-law}
\leanverified{paper\_conclusion\_binfold\_gauge\_volume\_double\_exponential\_law}
在 bin-fold 口径下，令
\[
|G_m^{\mathrm{bin}}|:=\prod_{w\in X_m} d_m^{\mathrm{bin}}(w)!,
\qquad
g_m:=2^{-m}\log|G_m^{\mathrm{bin}}|,
\qquad
\kappa_m:=2^{-m}\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w).
\]
则有
\[
g_m=\kappa_m-1+o(1),
\qquad
\kappa_m=m\log\!\Bigl(\frac{2}{\varphi}\Bigr)+O(1),
\]
从而
\[
g_m=m\log\!\Bigl(\frac{2}{\varphi}\Bigr)+O(1).
\]
等价地，
\[
\log |G_m^{\mathrm{bin}}|
=
m\,2^m\log\!\Bigl(\frac{2}{\varphi}\Bigr)+O(2^m),
\]
以及
\[
\log_2 |G_m^{\mathrm{bin}}|
=
m\,2^m\log_2\!\Bigl(\frac{2}{\varphi}\Bigr)+O(2^m).
\]
因此 fold-gauge 的体积主阶服从斜率精确等于 \(\log(2/\varphi)\) 的双指数熵律。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-fold-kappa-gauge-first-order-constants}，
\[
\kappa_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\Bigl(\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr),
\]
以及
\[
g_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-1
-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\Bigl(m\Bigl(\frac{\varphi}{2}\Bigr)^m\Bigr).
\]
故
\[
\kappa_m=m\log\!\Bigl(\frac{2}{\varphi}\Bigr)+O(1),
\qquad
g_m=\kappa_m-1+o(1),
\]
并因此
\[
g_m=m\log\!\Bigl(\frac{2}{\varphi}\Bigr)+O(1).
\]
再乘以 \(2^m\) 便得到
\[
\log |G_m^{\mathrm{bin}}|=2^m g_m
=
m\,2^m\log\!\Bigl(\frac{2}{\varphi}\Bigr)+O(2^m).
\]
除以 \(\log 2\) 即得二进形式。证毕。
\end{proof}

\begin{theorem}[规范体积密度与纤维信息的精确仿射律]\label{thm:conclusion-binfold-kappa-gauge-affine-law}
在 bin-fold 口径下，令
\[
\kappa_m:=2^{-m}\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w),
\qquad
g_m:=2^{-m}\log|G_m^{\mathrm{bin}}|.
\]
则当 \(m\to\infty\) 时成立
\[
\kappa_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right),
\]
以及
\[
g_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-1-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
特别地，
\[
g_m=\kappa_m-1+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
\end{theorem}
\begin{proof}
第一式正是定理 \ref{thm:xi-fold-kappa-gauge-first-order-constants} 在 \(q=1\) 处的专门化，因为
\[
\kappa_m=\kappa_m^{\mathrm{bin}}(1),
\qquad
\kappa_m^{\mathrm{bin}}(q)
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-\frac{\log\varphi}{1+\varphi^{q+1}}
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
代入 \(q=1\) 并利用
\[
1+\varphi^2=\varphi\sqrt5
\]
即得
\[
\kappa_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
第二式由同一结果中关于 \(g_m\) 的闭式直接给出：
\[
g_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-1-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(m\left(\frac{\varphi}{2}\right)^m\right).
\]
两式相减便得到最后的精确仿射差。证毕。
\end{proof}

\begin{corollary}[规范体积与纤维信息的主斜率严格一致]\label{cor:conclusion-binfold-kappa-gauge-common-slope}
\leanverified{paper\_conclusion\_binfold\_kappa\_gauge\_common\_slope}
在定理 \ref{thm:conclusion-binfold-kappa-gauge-affine-law} 的设定下，
\[
\lim_{m\to\infty}\frac{\kappa_m}{m}
=
\lim_{m\to\infty}\frac{g_m}{m}
=
\log\!\Bigl(\frac{2}{\varphi}\Bigr).
\]
因此，规范体积的主斜率与纤维信息的主斜率严格一致，而二者之间仅相差显式常数 \(-1\) 与指数小修正。
\end{corollary}
\begin{proof}
将定理 \ref{thm:conclusion-binfold-kappa-gauge-affine-law} 的两式同时除以 \(m\)，再令 \(m\to\infty\) 即得。证毕。
\end{proof}

\begin{theorem}[对数纤维规模的二原子极限律]\label{thm:conclusion-binfold-logfiber-two-atom-limit}
令
\[
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
L_m(w):=\log d_m^{\mathrm{bin}}(w)-m\log\!\Bigl(\frac{2}{\varphi}\Bigr).
\]
若 \(W_m\sim \mu_m\)，则
\[
L_m(W_m)\Longrightarrow L_\infty,
\]
其中 \(L_\infty\) 为二原子随机变量
\[
\PP(L_\infty=0)=\frac{\varphi}{\sqrt5},
\qquad
\PP(L_\infty=-\log\varphi)=\frac{1}{\varphi\sqrt5}.
\]
其矩母函数与累积母函数分别为
\[
\EE[e^{tL_\infty}]
=
\frac{\varphi}{\sqrt5}+\frac{\varphi^{-t-1}}{\sqrt5},
\qquad
\Lambda_\infty(t)
=
\log\!\left(\frac{\varphi}{\sqrt5}+\frac{\varphi^{-t-1}}{\sqrt5}\right).
\]
特别地，
\[
\lim_{m\to\infty}\Var_{\mu_m}\!\bigl(\log d_m^{\mathrm{bin}}\bigr)
=
\frac{(\log\varphi)^2}{5}.
\]
\end{theorem}
\begin{proof}
由 bin-fold 的两态一致渐近，存在对 \(w\in X_m\) 一致的估计
\[
d_m^{\mathrm{bin}}(w)
=
\varphi^{-w_m}\left(\frac{2}{\varphi}\right)^m+O(1).
\]
由于主项指数增长，取对数后得到
\[
L_m(w)
=
-w_m\log\varphi+O\!\left(\left(\frac{\varphi}{2}\right)^m\right)
\]
一致成立。另一方面，escort 权重在末位两扇区上的极限质量已知为
\[
\mu_m(\{w:\ w_m=0\})\to \frac{\varphi}{\sqrt5},
\qquad
\mu_m(\{w:\ w_m=1\})\to \frac{1}{\varphi\sqrt5}.
\]
故 \(L_m(W_m)\) 在分布上收敛到上述二原子极限。

矩母函数由两点分布直接求得：
\[
\EE[e^{tL_\infty}]
=
\frac{\varphi}{\sqrt5}e^{0}
+\frac{1}{\varphi\sqrt5}e^{-t\log\varphi}
=
\frac{\varphi}{\sqrt5}+\frac{\varphi^{-t-1}}{\sqrt5}.
\]
累积母函数即其对数。

最后，对取值 \(0\) 与 \(-\log\varphi\) 的二点分布，方差等于
\[
\frac{\varphi}{\sqrt5}\cdot \frac{1}{\varphi\sqrt5}\,(\log\varphi)^2
=
\frac{(\log\varphi)^2}{5}.
\]
故方差极限成立。证毕。
\end{proof}

\paragraph{关于共振常数与顶端预算的读法}
定理 \ref{thm:conclusion-fold-golden-resonance-collision-gap-hard-floor} 已给出由 \(C_\varphi\) 驱动的二阶碰撞底噪，而定理 \ref{thm:conclusion-fold-golden-resonance-maxfiber-lift} 也表明该常数能在峰值偏置的粗下界中留下痕迹。然而，结合定理 \ref{thm:conclusion-maxfiber-average-scale-rigidity} 与推论 \ref{cor:xi-time-part70d-maxfiber-excess-exact-main-term} 的精确主项
\[
M_m-\bar d_m
=
\varphi^{-4}\left(\frac{2}{\varphi}\right)^m
+o\!\left(\left(\frac{2}{\varphi}\right)^m\right),
\]
可见 \(C_\varphi\) 在本文现有证明链中更自然地应被理解为碰撞/Fourier 层的二阶刚性常数，而不宜直接提升为 \(\log_2 M_m\) 的独立正加性修正项。就此而言，共振常数对预算的影响是间接的：它首先锁定二阶非均匀性，再经容量亏损、碰撞矩与 Fourier 能量等对象层读取，而不是直接改写顶端精确反演阈值的常数项。

\begin{proposition}[容量亏损的 Bernstein--Mellin 表达]\label{prop:conclusion-capacity-defect-bernstein-mellin}
\leanverified{paper\_conclusion\_capacity\_defect\_bernstein\_mellin}
对任意 \(q>1\)，记
\[
S_q(m):=\sum_{x\in X_m}d_m(x)^q,
\qquad
\mathcal C_m^{\mathrm{cont}}(T):=\sum_{x\in X_m}\min(d_m(x),T),
\qquad
\mathcal D_m(T):=2^m-\mathcal C_m^{\mathrm{cont}}(T).
\]
则有严格恒等式
\[
S_q(m)
=
q(q-1)\int_0^\infty T^{q-2}\,\mathcal D_m(T)\,\dd T.
\]
因此，高阶矩谱可视为容量亏损曲线的 Bernstein--Mellin 变换。
\end{proposition}
\begin{proof}
对任意实数 \(d\ge 0\) 与 \(q>1\)，有标量恒等式
\[
d^q
=
q(q-1)\int_0^\infty T^{q-2}(d-T)_+\,\dd T.
\]
将 \(d=d_m(x)\) 代入并对 \(x\in X_m\) 求和，得到
\[
S_q(m)
=
q(q-1)\int_0^\infty T^{q-2}
\sum_{x\in X_m}(d_m(x)-T)_+\,\dd T.
\]
而
\[
(d_m(x)-T)_+=d_m(x)-\min(d_m(x),T),
\]
故
\[
\sum_{x\in X_m}(d_m(x)-T)_+
=
\sum_{x\in X_m}d_m(x)-\sum_{x\in X_m}\min(d_m(x),T)
=
2^m-\mathcal C_m^{\mathrm{cont}}(T)
=
\mathcal D_m(T).
\]
代回即得结论。证毕。
\end{proof}

\begin{theorem}[冻结相中的预言机肩部刚性]\label{thm:conclusion-frozen-phase-oracle-shoulder-rigidity}
沿用定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse} 的记号。设 \(a>a_{\mathrm c}\)，并记
\[
M_m:=\max_{x\in X_m}d_m(x),
\qquad
K_m:=\#A_m^{\max},
\qquad
A_m^{\max}:=\{x\in X_m:\ d_m(x)=M_m\}.
\]
对任意 \(\beta\ge 0\)，定义预算截断随机变量
\[
L_{m,\beta}:=\frac1m\log \min\bigl(d_m(X_{m,a}),e^{\beta m}\bigr),
\]
其中 \(X_{m,a}\sim \pi_{m,a}\)。再定义最大纤维层的截断容量
\[
\mathcal C_m^{\max}(u):=\sum_{x\in A_m^{\max}}\min\bigl(d_m(x),u\bigr).
\]
则有：
\begin{enumerate}
  \item \(L_{m,\beta}\to \min\{\alpha_\ast,\beta\}\) 依概率收敛；
  \item 若取 \(u_m:=e^{\beta m}\)，则
  \[
  \lim_{m\to\infty}\frac1m\log \mathcal C_m^{\max}(u_m)
  =
  g_\ast+\min\{\alpha_\ast,\beta\}.
  \]
\end{enumerate}
换言之，冻结相并不只给出“质量集中于基态层”，而且在任意指数预算下强制出现一条于 \(\beta=\alpha_\ast\) 折转的刚性线性--饱和肩部。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-frozen-escort-shannon-renyi-collapse}，
\[
\pi_{m,a}(A_m^{\max})
\ge
1-\exp\!\bigl(-m\Delta(a)+o(m)\bigr)
\to 1,
\]
其中
\[
\Delta(a)=a(\alpha_\ast-\alpha_\ast^{(2)})-(\log\varphi-g_\ast)>0.
\]
因此以趋于 \(1\) 的概率有 \(X_{m,a}\in A_m^{\max}\)。在该事件上，
\[
L_{m,\beta}
=
\frac1m\log \min(M_m,e^{\beta m})
=
\min\Bigl\{\frac1m\log M_m,\beta\Bigr\}
\to
\min\{\alpha_\ast,\beta\},
\]
故 \(\textup{(1)}\) 成立。

对 \(\textup{(2)}\)，由 \(d_m(x)=M_m\) 在 \(A_m^{\max}\) 上恒定，直接得到
\[
\mathcal C_m^{\max}(u_m)
=
K_m\min(M_m,e^{\beta m}).
\]
两边取对数并除以 \(m\)，再用
\[
\frac1m\log K_m\to g_\ast,
\qquad
\frac1m\log M_m\to \alpha_\ast,
\]
便得
\[
\frac1m\log \mathcal C_m^{\max}(u_m)
\to
g_\ast+\min\{\alpha_\ast,\beta\}.
\]
证毕。
\end{proof}

而其 Mellin 端则由容量亏损的 Bernstein--Mellin 变换把全部高阶矩谱重新拉回，并在冻结区间上进一步显露出以 \(\alpha_\ast\) 为拐点的刚性预算肩部。至于黄金共振常数 \(C_\varphi\)，现阶段更自然的落点仍是碰撞、Fourier 与容量亏损这些二阶对象，而不是顶端预算常数项本身。故在本文框架中，冻结热力学、半单对象层、fold-gauge 体积、预算肩部几何与 Mellin 端矩谱最终都可通过容量曲线这一硬不变量得到统一组织。
