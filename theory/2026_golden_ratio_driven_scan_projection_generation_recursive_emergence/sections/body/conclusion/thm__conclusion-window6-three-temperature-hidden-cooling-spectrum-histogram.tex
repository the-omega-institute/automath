\begin{theorem}[window-\texorpdfstring{$6$}{6} hidden sector 的三温冷却律与谱全息恢复]\label{thm:conclusion-window6-three-temperature-hidden-cooling-spectral-holography}
\leanverified{paper\_conclusion\_window6\_three\_temperature\_hidden\_cooling\_spectral\_holography}
沿用定理 \ref{thm:conclusion-window6-fiber-gauge-laplacian-spectrum} 与推论 \ref{cor:conclusion-window6-fiber-gauge-laplacian-projectors} 的记号。则对任意 \(t\ge 0\)，有精确热半群分解
\[
e^{-t\Delta_6^{\mathrm{fib}}}
=
P_0+e^{-2t}P_2+e^{-3t}P_3+e^{-4t}P_4.
\]
因此：
\begin{enumerate}
  \item 规范 coarse-graining 极限满足
  \[
  \lim_{t\to\infty}e^{-t\Delta_6^{\mathrm{fib}}}=P_0;
  \]
  \item 全空间上的最优衰减率为
  \[
  \bigl\|e^{-t\Delta_6^{\mathrm{fib}}}-P_0\bigr\|_{2\to 2}=e^{-2t};
  \]
  \item 在 \(M_{(2)}^\perp=M_6^{\mathrm{vis}}\oplus M_{(3)}\oplus M_{(4)}\) 上，衰减率提升为
  \[
  \bigl\|\bigl(e^{-t\Delta_6^{\mathrm{fib}}}-P_0\bigr)\big|_{M_{(2)}^\perp}\bigr\|_{2\to 2}=e^{-3t};
  \]
  \item hidden heat trace、谱 \(\zeta\) 与伪行列式分别为
  \[
  \Theta_6^{\mathrm{hid}}(t)
  :=\Tr\!\bigl(e^{-t\Delta_6^{\mathrm{fib}}}-P_0\bigr)
  =
  8e^{-2t}+8e^{-3t}+27e^{-4t},
  \]
  \[
  \zeta_{\Delta_6^{\mathrm{fib}}}^{\mathrm{hid}}(s)
  :=
  \Tr\!\bigl((\Delta_6^{\mathrm{fib}}|_{M_6^{\mathrm{hid}}})^{-s}\bigr)
  =
  8\cdot 2^{-s}+8\cdot 3^{-s}+27\cdot 4^{-s},
  \]
  \[
  \det\nolimits'\!\bigl(\Delta_6^{\mathrm{fib}}\bigr)=2^8\,3^8\,4^{27}=2^{62}3^8.
  \]
\end{enumerate}
进一步，\(\Delta_6^{\mathrm{fib}}\) 的非零谱测度已经足以恢复全部纤维直方图：
\[
N_{6,2}=\operatorname{mult}(2),
\qquad
N_{6,3}=\frac{\operatorname{mult}(3)}{2},
\qquad
N_{6,4}=\frac{\operatorname{mult}(4)}{3}.
\]
对 window-\(6\) 而言，这给回
\[
2:8,\qquad 3:4,\qquad 4:9.
\]
故纤维直方图与规范 Laplacian 的非零谱在 window-\(6\) 上是严格等价的两种编码。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-fiber-gauge-laplacian-spectrum}，\(\Delta_6^{\mathrm{fib}}\) 的谱投影分解正是
\[
\Delta_6^{\mathrm{fib}}=0\cdot P_0+2P_2+3P_3+4P_4.
\]
对该有限谱作函数演算，立刻得到
\[
e^{-t\Delta_6^{\mathrm{fib}}}
=
e^0P_0+e^{-2t}P_2+e^{-3t}P_3+e^{-4t}P_4.
\]
第 \(\textup{(1)}\) 条随 \(t\to\infty\) 直接成立。

对第 \(\textup{(2)}\) 条，算子
\[
e^{-t\Delta_6^{\mathrm{fib}}}-P_0=e^{-2t}P_2+e^{-3t}P_3+e^{-4t}P_4
\]
在三条正交壳层上的特征值分别为 \(e^{-2t},e^{-3t},e^{-4t}\)，故其算子范数为三者最大值 \(e^{-2t}\)。若限制到 \(M_{(2)}^\perp\)，则最小非零特征值变为 \(3\)，故第 \(\textup{(3)}\) 条成立。

第 \(\textup{(4)}\) 条只是按重数
\[
\dim M_{(2)}=8,\qquad \dim M_{(3)}=8,\qquad \dim M_{(4)}=27
\]
逐项求和或相乘，这与推论 \ref{cor:conclusion-window6-fiber-gauge-laplacian-spectral-functions} 一致。

最后，若把 multiplicity \(d\) 的纤维数记为 \(N_{6,d}\)，则由定理 \ref{thm:conclusion-window6-fiber-gauge-laplacian-spectrum}
\[
\operatorname{mult}(d)=(d-1)N_{6,d}\qquad (d=2,3,4).
\]
故
\[
N_{6,2}=\operatorname{mult}(2),\qquad
N_{6,3}=\frac{\operatorname{mult}(3)}{2},\qquad
N_{6,4}=\frac{\operatorname{mult}(4)}{3}.
\]
代入
\[
\operatorname{mult}(2)=8,\qquad \operatorname{mult}(3)=8,\qquad \operatorname{mult}(4)=27
\]
便恢复 window-\(6\) 的纤维直方图 \(2{:}8,\ 3{:}4,\ 4{:}9\)。证毕。
\end{proof}

\begin{conjecture}[有限 fold-gauge 置换模的谱--直方图全息]\label{conj:conclusion-finite-foldgauge-laplacian-spectrum-histogram-holography}

  \]

  \[
  M_m\cong \bigoplus_{w\in X_m}\bigl(\mathbf 1_w\oplus \mathrm{Std}_w\bigr).

\end{enumerate}

\[
\Delta_m:=\bigoplus_{w\in X_m}\sum_{\tau\in \mathrm{Trans}(F_w)}(I-\tau),

\[
\operatorname{Spec}(\Delta_m)
=
\{0^{(|X_m|)}\}\cup \bigcup_{d\ge 2}\{d^{((d-1)N_{m,d})}\},
\qquad
N_{m,d}:=\#\{w\in X_m:d_m(w)=d\}.
\]
从而
\[
\Theta_m^{\mathrm{hid}}(t)=\sum_{d\ge 2}(d-1)N_{m,d}e^{-dt},
\qquad

\]
并且 \(\Delta_m\) 的非零谱测度与纤维直方图 \(\{N_{m,d}\}_{d\ge2}\) 互为完备不变量。window-\(6\) 正是这一谱--直方图全息原则的第一个完全刚性样本。
\end{conjecture}
