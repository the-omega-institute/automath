\begin{theorem}[window-\texorpdfstring{$6$}{6} 可见 CRT 层的精确 Ramanujan 扇区代数]\label{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer}
\leanverified{paper\_conclusion\_window6\_ramanujan\_sector\_algebra\_on\_crt\_layer}
沿用定理 \ref{thm:conclusion-window6-visible-crt-arithmetic-phase-space} 的识别
\[
X_6\cong R_6:=\ZZ/(21\ZZ),
\]
并对每个 \(\delta\mid 21\) 记
\[
S_\delta:=\{r\in R_6:\ (r,21)=\delta\}.
\]
定义
\[
\mathcal A_{21}:=\operatorname{span}_{\CC}\{1,c_3,c_7,c_{21}\}\subset \CC^{R_6}.
\]
则 \(\mathcal A_{21}\) 恰为 \(R_6\) 上按 exact-gcd strata 常值的函数代数。更具体地，四个扇区示性函数满足
\[
\mathbf 1_{S_{21}}(r)=\frac{1+c_3(r)+c_7(r)+c_{21}(r)}{21},
\]
\[
\mathbf 1_{S_3}(r)=\frac{6+6c_3(r)-c_7(r)-c_{21}(r)}{21},
\]
\[
\mathbf 1_{S_7}(r)=\frac{2-c_3(r)+2c_7(r)-c_{21}(r)}{21},
\]
\[
\mathbf 1_{S_1}(r)=\frac{12-6c_3(r)-2c_7(r)+c_{21}(r)}{21}.
\]
\end{theorem}

\begin{proof}
由于 \(21=3\cdot 7\) 为平方自由数，Ramanujan 和满足
\[
c_3(r)=
\begin{cases}
2,& 3\mid r,\\
-1,& 3\nmid r,
\end{cases}
\qquad
c_7(r)=
\begin{cases}
6,& 7\mid r,\\
-1,& 7\nmid r.
\end{cases}
\]
于是
\[
\mathbf 1_{3\mid r}=\frac{1+c_3(r)}{3},
\qquad
\mathbf 1_{3\nmid r}=\frac{2-c_3(r)}{3},
\]
\[
\mathbf 1_{7\mid r}=\frac{1+c_7(r)}{7},
\qquad
\mathbf 1_{7\nmid r}=\frac{6-c_7(r)}{7}.
\]
再由互素模数下 Ramanujan 和的乘法性，
\[
c_{21}(r)=c_3(r)c_7(r).
\]
因此
\[
\mathbf 1_{S_{21}}=\mathbf 1_{3\mid r}\mathbf 1_{7\mid r},\qquad
\mathbf 1_{S_3}=\mathbf 1_{3\mid r}\mathbf 1_{7\nmid r},
\]
\[
\mathbf 1_{S_7}=\mathbf 1_{3\nmid r}\mathbf 1_{7\mid r},\qquad
\mathbf 1_{S_1}=\mathbf 1_{3\nmid r}\mathbf 1_{7\nmid r},
\]
逐项展开即得所列闭式。四个示性函数显然张成按 exact-gcd strata 常值的函数代数，而上式表明它们全落在 \(\mathcal A_{21}\) 中；反向包含显然，故结论成立。证毕。
\end{proof}

\begin{theorem}[hidden multiplicity 在全部非零可见 conductor strata 上的严格中性律]\label{thm:conclusion-window6-hidden-multiplicity-neutrality-on-nonzero-strata}
\leanverified{paper\_conclusion\_window6\_hidden\_multiplicity\_neutrality\_on\_nonzero\_strata}
将 \(R_6\) 与其标准代表系 \(\{0,1,\dots,20\}\) 识别，并定义
\[
d(r):=\bigl|F_{V_6^{-1}(r)}\bigr|,
\qquad
h(r):=d(r)-1.
\]
则 \(d(r)\) 具有显式分段公式
\[
d(r)=
\begin{cases}
4,& 0\le r\le 8,\\
3,& 9\le r\le 12,\\
2,& 13\le r\le 20,
\end{cases}
\]
并且对每个非零 conductor stratum \(\delta\in\{1,3,7\}\)，都有
\[
\frac1{|S_\delta|}\sum_{r\in S_\delta} d(r)=3,
\qquad
\frac1{|S_\delta|}\sum_{r\in S_\delta} h(r)=2.
\]
另一方面，
\[
d(0)=4,\qquad h(0)=3.
\]
\end{theorem}

\begin{proof}
主稿对 window-\(6\) hidden fiber 的显式审计表明：若 \(V:=V_6(w)\)，则
\[
|F_w|=
\begin{cases}
2,& w_6=1,\\
4,& w_6=0,\ V\le 8,\\
3,& w_6=0,\ V>8,
\end{cases}
\]
且 \(w_6=1\) 必推出 \(V\ge 13\)。于是经由 \(V_6\) 的识别可得
\[
d(r)=4\ (0\le r\le 8),\qquad
d(r)=3\ (9\le r\le 12),\qquad
d(r)=2\ (13\le r\le 20).
\]

记
\[
I_4:=\{0,\dots,8\},\qquad
I_3:=\{9,10,11,12\},\qquad
I_2:=\{13,\dots,20\}.
\]
则直接枚举得到
\[
|S_1\cap I_4|=5,\quad |S_1\cap I_3|=2,\quad |S_1\cap I_2|=5,
\]
\[
|S_3\cap I_4|=|S_3\cap I_3|=|S_3\cap I_2|=2,
\]
\[
|S_7\cap I_4|=1,\quad |S_7\cap I_3|=0,\quad |S_7\cap I_2|=1.
\]
因此
\[
\sum_{r\in S_1}d(r)=5\cdot 4+2\cdot 3+5\cdot 2=36=3|S_1|,
\]
\[
\sum_{r\in S_3}d(r)=2\cdot 4+2\cdot 3+2\cdot 2=18=3|S_3|,
\]
\[
\sum_{r\in S_7}d(r)=1\cdot 4+1\cdot 2=6=3|S_7|.
\]
故全部非零 strata 上的平均 fiber 大小都等于 \(3\)。对 \(h=d-1\) 只需减去各 strata 的基数，便得到平均值统一等于 \(2\)。零层公式显然成立。证毕。
\end{proof}

\begin{theorem}[hidden rank 的 Ramanujan-visible 正交投影闭式]\label{thm:conclusion-window6-hidden-rank-ramanujan-visible-orthogonal-projection}
在计数内积
\[
\langle f,g\rangle:=\sum_{r\in R_6}f(r)\overline{g(r)}
\]
下，记
\[
P_{21}:\CC^{R_6}\to \mathcal A_{21}
\]
为到定理 \ref{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer} 中 Ramanujan 扇区代数的正交投影。则
\[
P_{21}d=3+\mathbf 1_{S_{21}},
\qquad
P_{21}h=2+\mathbf 1_{S_{21}}.
\]
等价地，
\[
P_{21}d
=
3+\frac{1+c_3+c_7+c_{21}}{21},
\]
\[
P_{21}h
=
2+\frac{1+c_3+c_7+c_{21}}{21}.
\]
\leanverified{paper\_conclusion\_window6\_hidden\_rank\_ramanujan\_visible\_orthogonal\_projection}
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer}，\(\mathcal A_{21}\) 恰为按四个 exact-gcd strata 常值的函数代数。故 \(P_{21}\) 正是相对于该分层的条件期望：即在每个 \(S_\delta\) 上取平均值。

由定理 \ref{thm:conclusion-window6-hidden-multiplicity-neutrality-on-nonzero-strata}，函数 \(d\) 在零层 \(S_{21}=\{0\}\) 上取值 \(4\)，而在所有非零 strata \(S_1,S_3,S_7\) 上的平均值都等于 \(3\)。因此
\[
P_{21}d=3+\mathbf 1_{S_{21}}.
\]
同理，\(h=d-1\) 在 \(S_{21}\) 上取值 \(3\)，而在全部非零 strata 上平均值都等于 \(2\)，故
\[
P_{21}h=2+\mathbf 1_{S_{21}}.
\]
再把定理 \ref{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer} 中
\[
\mathbf 1_{S_{21}}=\frac{1+c_3+c_7+c_{21}}{21}
\]
代入，即得后两式。证毕。
\end{proof}

\begin{corollary}[非平凡 conductor 的单位权 Ramanujan 定律]\label{cor:conclusion-window6-nontrivial-conductor-unit-ramanujan-law}
对每个非平凡除数 \(q\in\{3,7,21\}\)，均有
\[
\sum_{r\in R_6} d(r)c_q(r)=\varphi(q),
\qquad
\sum_{r\in R_6} h(r)c_q(r)=\varphi(q).
\]
等价地，若定义 hidden rank 的 squarefree Ramanujan 指纹
\[
\mathfrak R_h(s):=
\sum_{\substack{q\mid 21\\ q>1}}
\frac{1}{\varphi(q)}
\Bigl(\sum_{r\in R_6}h(r)c_q(r)\Bigr)q^{-s},
\]
则有精确闭式
\[
\mathfrak R_h(s)=3^{-s}+7^{-s}+21^{-s}
=(1+3^{-s})(1+7^{-s})-1.
\]
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-window6-hidden-rank-ramanujan-visible-orthogonal-projection}，
\[
h=P_{21}h+\bigl(h-P_{21}h\bigr),
\]
其中后项与 \(\mathcal A_{21}\) 正交。因此对任意 \(q\mid 21\) 都有
\[
\sum_{r\in R_6}h(r)c_q(r)=\sum_{r\in R_6}(P_{21}h)(r)c_q(r).
\]
当 \(q>1\) 时，
\[
\sum_{r\in R_6}c_q(r)=0.
\]
再用
\[
P_{21}h=2+\mathbf 1_{S_{21}},
\]
便得
\[
\sum_{r\in R_6}h(r)c_q(r)
=
2\sum_{r\in R_6}c_q(r)+\sum_{r\in R_6}\mathbf 1_{S_{21}}(r)c_q(r)
=
c_q(0)
=
\varphi(q).
\]
对 \(d\) 同理，因为 \(P_{21}d=3+\mathbf 1_{S_{21}}\)。

最后，把 \(q=3,7,21\) 三项逐一代入定义，即得
\[
\mathfrak R_h(s)=3^{-s}+7^{-s}+21^{-s}.
\]
证毕。
\end{proof}

\begin{proposition}[Ramanujan-blind residual channel 的精确形状]\label{prop:conclusion-window6-ramanujan-blind-residual-channel}
定义残余函数
\[
\varepsilon(r):=h(r)-P_{21}h(r)=d(r)-P_{21}d(r).
\]
则
\[
\varepsilon(0)=0,
\qquad
\varepsilon(r)=1\ \ (1\le r\le 8),
\qquad
\varepsilon(r)=0\ \ (9\le r\le 12),
\qquad
\varepsilon(r)=-1\ \ (13\le r\le 20).
\]
并且
\[
\sum_{r\in R_6}\varepsilon(r)c_q(r)=0
\qquad (q=1,3,7,21),
\]
\[
\|\varepsilon\|_{\ell^2(R_6)}^2=16.
\]
\end{proposition}

\begin{proof}
由定理 \ref{thm:conclusion-window6-hidden-multiplicity-neutrality-on-nonzero-strata}，
\[
h(r)=
\begin{cases}
3,& r=0,\\
3,& 1\le r\le 8,\\
2,& 9\le r\le 12,\\
1,& 13\le r\le 20.
\end{cases}
\]
而由定理 \ref{thm:conclusion-window6-hidden-rank-ramanujan-visible-orthogonal-projection}，
\[
P_{21}h(r)=
\begin{cases}
3,& r=0,\\
2,& r\neq 0.
\end{cases}
\]
两式相减即得 \(\varepsilon\) 的分段公式。

由于 \(\varepsilon=h-P_{21}h\) 正是到 \(\mathcal A_{21}\) 的正交余项，故它与 \(1,c_3,c_7,c_{21}\) 全部正交，即
\[
\sum_{r\in R_6}\varepsilon(r)c_q(r)=0
\qquad (q=1,3,7,21).
\]
最后，
\[
\|\varepsilon\|_{\ell^2(R_6)}^2
=
8\cdot 1^2+4\cdot 0^2+8\cdot (-1)^2
=
16.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 仿射完成缺陷的 conductor-selective Ramanujan 矩]\label{thm:conclusion-window6-affine-defect-ramanujan-moments}
\leanverified{paper\_conclusion\_window6\_affine\_defect\_ramanujan\_moments}
沿用定理 \ref{thm:conclusion-window6-ramanujan-sector-algebra-on-crt-layer} 中的 Ramanujan 和 \(c_q\)。对每个平方自由导子
\[
q\in\{3,7,21\},
\]
定义仿射完成缺陷的 Ramanujan 矩
\[
\mathfrak R_q^{\mathrm{aff}}
:=
\sum_{r\in R_6}\Delta_{\mathrm{aff}}(r)c_q(r),
\]
其中 \(\Delta_{\mathrm{aff}}\) 由命题 \ref{prop:conclusion-window6-affine-completion-defect-law} 给出。则有严格恒等式
\[
\mathfrak R_3^{\mathrm{aff}}=2,
\qquad
\mathfrak R_7^{\mathrm{aff}}=0,
\qquad
\mathfrak R_{21}^{\mathrm{aff}}=-30.
\]
因此，导子 \(7\) 对 window-\(6\) 的仿射完成缺陷完全失明；导子 \(3\) 仅留下最小正信号；而导子 \(21\) 则记录一个强负响应。
\end{theorem}
\begin{proof}
由命题 \ref{prop:conclusion-window6-affine-completion-defect-law}，
\[
\Delta_{\mathrm{aff}}
=
4\,\mathbf 1_{R_{4}^{\mathrm{skew}}}
+\mathbf 1_{R_{3}^{\mathrm{punc}}}.
\]
因此只需在
\[
R_{4}^{\mathrm{skew}}=\{1,2,3,5,6,7\},
\qquad
R_{3}^{\mathrm{punc}}=\{9,10,11,12\}
\]
上计算 Ramanujan 和。

当 \(q=3\) 时，
\[
c_3(r)=
\begin{cases}
2,& 3\mid r,\\
-1,& 3\nmid r,
\end{cases}
\]
故
\[
\sum_{r\in R_{4}^{\mathrm{skew}}}4c_3(r)
=
4(-1-1+2-1+2-1)=0,
\]
\[
\sum_{r\in R_{3}^{\mathrm{punc}}}c_3(r)
=
2-1-1+2=2.
\]
于是
\[
\mathfrak R_3^{\mathrm{aff}}=2.
\]

当 \(q=7\) 时，
\[
c_7(r)=
\begin{cases}
6,& 7\mid r,\\
-1,& 7\nmid r,
\end{cases}
\]
从而
\[
\sum_{r\in R_{4}^{\mathrm{skew}}}4c_7(r)
=
4(-1-1-1-1-1+6)=4,
\]
\[
\sum_{r\in R_{3}^{\mathrm{punc}}}c_7(r)
=
-1-1-1-1=-4.
\]
故
\[
\mathfrak R_7^{\mathrm{aff}}=0.
\]

最后，由 \(21=3\cdot 7\) 的平方自由性，
\[
c_{21}(r)=c_3(r)c_7(r).
\]
因此
\[
\sum_{r\in R_{4}^{\mathrm{skew}}}4c_{21}(r)
=
4(1+1-2+1-2-6)=-28,
\]
\[
\sum_{r\in R_{3}^{\mathrm{punc}}}c_{21}(r)
=
-2+1+1-2=-2.
\]
合并得
\[
\mathfrak R_{21}^{\mathrm{aff}}=-30.
\]
证毕。
\end{proof}

\begin{corollary}[raw multiplicity 与 affine completion 之间的 conductor-selective 位移]\label{cor:conclusion-window6-raw-affine-conductor-shift}
定义
\[
\mathfrak D_q:=\sum_{r\in R_6}d(r)c_q(r),
\qquad
\mathfrak A_q:=\sum_{r\in R_6}a(r)c_q(r),
\]
其中 \(a(r)=d(r)+\Delta_{\mathrm{aff}}(r)\)。则
\[
\mathfrak A_q-\mathfrak D_q=\mathfrak R_q^{\mathrm{aff}}.
\]
更具体地，
\[
\mathfrak D_3=2,\qquad
\mathfrak D_7=6,\qquad
\mathfrak D_{21}=12,
\]
\[
\mathfrak A_3=4,\qquad
\mathfrak A_7=6,\qquad
\mathfrak A_{21}=-18.
\]
于是
\[
\mathfrak A_7=\mathfrak D_7,
\qquad
\mathfrak A_3-\mathfrak D_3=2,
\qquad
\mathfrak A_{21}-\mathfrak D_{21}=-30.
\]
换言之，从 raw multiplicity 过渡到 affine completion 时，导子 \(7\) 完全不变，导子 \(3\) 仅发生最小位移，而导子 \(21\) 则出现大幅翻转。
\end{corollary}
\begin{proof}
由定义
\[
a(r)=d(r)+\Delta_{\mathrm{aff}}(r),
\]
立得
\[
\mathfrak A_q-\mathfrak D_q=\mathfrak R_q^{\mathrm{aff}}.
\]

另一方面，推论 \ref{cor:conclusion-window6-nontrivial-conductor-unit-ramanujan-law} 已证明对每个非平凡导子 \(q\mid 21\) 都有
\[
\mathfrak D_q=\sum_{r\in R_6}d(r)c_q(r)=\varphi(q).
\]
因此
\[
\mathfrak D_3=2,\qquad
\mathfrak D_7=6,\qquad
\mathfrak D_{21}=12.
\]
再与定理 \ref{thm:conclusion-window6-affine-defect-ramanujan-moments} 合并，即得
\[
\mathfrak A_3=2+2=4,
\qquad
\mathfrak A_7=6+0=6,
\qquad
\mathfrak A_{21}=12-30=-18.
\]
其余结论随之成立。证毕。
\end{proof}
