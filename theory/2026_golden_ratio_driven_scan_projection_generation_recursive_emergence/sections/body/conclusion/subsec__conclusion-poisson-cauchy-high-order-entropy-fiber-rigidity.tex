\subsection{Poisson--Cauchy 粗粒化的高阶信息渐近与时间纤维端点几何}\label{subsec:conclusion-poisson-cauchy-high-order-entropy-fiber-rigidity}
本小节在 \S\ref{subsec:conclusion-poisson-cauchy-bivariate-information-fingerprint} 的二阶框架上，
对一维位置混合模型给出三阶总变差常数、KL/R\'enyi 六阶系数、耗散七阶系数以及时间纤维的 Cauchy 化、端点 Poisson 权的 Cayley--Cauchy 精确同构与端点视界的 reciprocal--arcsine 对偶。

\paragraph{基础设定}
设 \(\tilde\nu\) 为 \(\RR\) 上概率测度，定义
\[
\mathsf P_t(x):=\frac1\pi\frac{t}{x^2+t^2},
\]
\[
\tilde h_t(x):=\int_{\RR}\mathsf P_t(x-\gamma)\,\dd\tilde\nu(\gamma),\qquad
\tilde g_t(x):=\mathsf P_t(x-\bar\gamma),\qquad
\bar\gamma:=\int_{\RR}\gamma\,\dd\tilde\nu(\gamma).
\]
记中心矩
\[
\mu_k:=\int_{\RR}(\gamma-\bar\gamma)^k\,\dd\tilde\nu(\gamma)\qquad(k\ge 2),
\]
并定义
\[
\tilde r_t(x):=\frac12\,\partial_x^2\tilde g_t(x).
\]

\begin{theorem}[三阶去方差总变差常数]\label{thm:conclusion-poisson-cauchy-third-order-tv-constant}
\leanverified{paper\_conclusion\_poisson\_cauchy\_third\_order\_tv\_constant}
若 \(\mu_4<\infty\)，定义二阶剥离余项
\[
\Delta_t(x):=\tilde h_t(x)-\tilde g_t(x)-\mu_2\tilde r_t(x),
\]
及三阶轮廓核
\[
\tilde s_t(x):=-\frac16\,\partial_x^3\tilde g_t(x).
\]
则
\[
t^3\|\Delta_t-\mu_3\tilde s_t\|_{L^1(\RR)}\longrightarrow 0,
\]
并且
\[
\lim_{t\to\infty}\frac{t^3}{|\mu_3|}\,\|\Delta_t\|_{L^1(\RR)}=\frac1\pi.
\]
\end{theorem}
\begin{proof}
令 \(u:=x-\bar\gamma\)，并以随机变量 \(Z:=\gamma-\bar\gamma\)（\(\EE Z=0\)）表示混合：
\[
\tilde h_t(x)=\EE\bigl[\mathsf P_t(u-Z)\bigr].
\]
对 \(z\mapsto \mathsf P_t(u-z)\) 在 \(z=0\) 作三阶 Taylor 展开并取期望：
\[
\tilde h_t(u)
=
\mathsf P_t(u)
\frac{\mu_2}{2}\mathsf P_t''(u)
-\frac{\mu_3}{6}\mathsf P_t^{(3)}(u)
+R_t(u),
\]
其中 \(\|R_t\|_\infty=O(t^{-5})\)（由 \(\mu_4<\infty\) 与 \(\sup_u|\mathsf P_t^{(4)}(u)|=O(t^{-5})\)）。
由 \(\tilde g_t(u)=\mathsf P_t(u)\)、\(\tilde r_t=\frac12\mathsf P_t''\)，得
\[
\Delta_t(u)= -\frac{\mu_3}{6}\mathsf P_t^{(3)}(u)+R_t(u)=\mu_3\tilde s_t(u)+R_t(u).
\]
故
\[
t^3\|\Delta_t-\mu_3\tilde s_t\|_{L^1}\le t^3\|R_t\|_{L^1}=o(1).
\]

再用尺度律
\[
\mathsf P_t(x)=t^{-1}\mathsf P_1(x/t)
\]
可得
\[
\|\mathsf P_t^{(3)}\|_{L^1}=t^{-3}\|\mathsf P_1^{(3)}\|_{L^1},
\qquad
\|\tilde s_t\|_{L^1}=\frac16\|\mathsf P_t^{(3)}\|_{L^1}.
\]
直接积分得到
\[
\|\mathsf P_1^{(3)}\|_{L^1}=\frac6\pi,
\]
故
\[
t^3\|\tilde s_t\|_{L^1}=\frac1\pi.
\]
于是
\[
\frac{t^3}{|\mu_3|}\|\Delta_t\|_{L^1}
=
\frac{t^3}{|\mu_3|}\|\mu_3\tilde s_t+o(t^{-3})\|_{L^1}
\longrightarrow
\frac1\pi.
\]
证毕。
\end{proof}

\begin{theorem}[KL 相对熵的六阶精细渐近]\label{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion}\leanverified{paper\_conclusion\_poisson\_cauchy\_kl\_sixth\_order\_expansion}
若 \(\mu_5^\ast:=\int_{\RR}|\gamma-\bar\gamma|^5\,\dd\tilde\nu(\gamma)<\infty\)，则
\[
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)
:=
\int_{\RR}\tilde h_t\log\!\frac{\tilde h_t}{\tilde g_t}\,\dd x
\]
满足
\[
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)
=
\frac{\mu_2^2}{8\,t^4}
+
\frac1{t^6}\left(
\frac{\mu_2^3}{64}
-\frac{\mu_2\mu_4}{8}
+\frac{3\mu_3^2}{32}
\right)
+o(t^{-6}).
\]
\end{theorem}
\begin{proof}
令
\[
\delta_t(x):=\frac{\tilde h_t(x)}{\tilde g_t(x)}-1.
\]
由四阶展开与尺度变换，存在有界轮廓函数
\[
U_2(y)=\frac{3y^2-1}{(1+y^2)^2},\quad
U_3(y)=\frac{4y(y^2-1)}{(1+y^2)^3},\quad
U_4(y)=\frac{5y^4-10y^2+1}{(1+y^2)^4},
\]
使得（\(y=(x-\bar\gamma)/t\)）
\[
\delta_t(x)=\frac{\mu_2}{t^2}U_2(y)+\frac{\mu_3}{t^3}U_3(y)+\frac{\mu_4}{t^4}U_4(y)+O(t^{-5})
\]
一致成立。
又
\[
\int_{\RR}\tilde g_t\,\delta_t\,\dd x=0.
\]
因此
\[
D_{\mathrm{KL}}
=
\int \tilde g_t\,(1+\delta_t)\log(1+\delta_t)\,\dd x
=
\frac12\int\tilde g_t\delta_t^2\,\dd x
-\frac16\int\tilde g_t\delta_t^3\,\dd x
+O(\|\delta_t\|_\infty^4),
\]
而 \(\|\delta_t\|_\infty=O(t^{-2})\)，余项为 \(O(t^{-8})\)。

换元 \(x=\bar\gamma+ty\) 后
\[
\tilde g_t(x)\,\dd x=\frac1\pi\frac{\dd y}{1+y^2}.
\]
利用奇偶性 \(U_2\) 偶、\(U_3\) 奇，\(t^{-5}\) 项消失。
所需积分为
\[
\int_{\RR}\frac{U_2^2}{\pi(1+y^2)}\,\dd y=\frac14,\quad
\int_{\RR}\frac{U_3^2}{\pi(1+y^2)}\,\dd y=\frac{3}{16},
\]
\[
\int_{\RR}\frac{U_2U_4}{\pi(1+y^2)}\,\dd y=-\frac18,\quad
\int_{\RR}\frac{U_2^3}{\pi(1+y^2)}\,\dd y=-\frac{3}{32}.
\]
代回并整理即得公式。证毕。
\end{proof}

\begin{theorem}[KL 相对熵的八阶喷流与奇次项消隐]\label{thm:conclusion-poisson-cauchy-kl-eighth-order-jet}
若中心化绝对八阶矩有限
\[
\mu_8^\ast:=\int_{\RR}|\gamma-\bar\gamma|^8\,\dd\tilde\nu(\gamma)<\infty,
\]
并且高阶 Taylor 余项可交换到积分号内，则当 \(t\to\infty\) 有八阶喷流展开
\[
D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)
=\frac{\mu_2^2}{8\,t^4}+\frac{B_6}{t^6}+\frac{B_8}{t^8}+o(t^{-8}),
\]
其中
\[
B_6=\frac{\mu_2^3}{64}-\frac{\mu_2\mu_4}{8}+\frac{3\mu_3^2}{32},
\]
\[
B_8=\frac1{256}\Bigl(3\mu_2^4-12\mu_2^2\mu_4+9\mu_2\mu_3^2+12\mu_2\mu_6+20\mu_4^2-30\mu_3\mu_5\Bigr).
\]
并且该展开仅含偶次幂：\(t^{-5}\) 与 \(t^{-7}\) 项的系数恒等为零。
\end{theorem}
\begin{proof}[证明要点]
令 \(\delta_t(x)=\tilde h_t(x)/\tilde g_t(x)-1\) 并作尺度置换 \(y=(x-\bar\gamma)/t\)，则
\[
\tilde g_t(x)\,\dd x=\frac{1}{\pi}\frac{\dd y}{1+y^2}.
\]
对 \(z\mapsto (1+(y-z)^2)^{-1}\) 在 \(z=0\) 作六阶 Taylor 展开并取期望，得到
\[
\delta_t(x)=\sum_{k=2}^{6}\frac{\mu_k}{t^k}U_k(y)+O(t^{-7})
\]
（于 \(\tilde g_t\dd x\) 的加权 \(L^p\) 意义下），其中
\[
U_2(y)=\frac{3y^2-1}{(1+y^2)^2},\quad
U_3(y)=\frac{4y(y^2-1)}{(1+y^2)^3},\quad
U_4(y)=\frac{5y^4-10y^2+1}{(1+y^2)^4},
\]
\[
U_5(y)=\frac{2y(y^2-3)(3y^2-1)}{(1+y^2)^5},\quad
U_6(y)=\frac{7y^6-35y^4+21y^2-1}{(1+y^2)^6}.
\]
将
\(
(1+\delta)\log(1+\delta)=\sum_{n\ge 2}\frac{(-1)^n}{n(n-1)}\delta^n
\)
代入
\(
D_{\mathrm{KL}}=\int \tilde g_t(1+\delta_t)\log(1+\delta_t)
\)
并按 \(t^{-k}\) 收集项。
由于基准密度为偶函数、\(U_{2\ell}\) 为偶函数而 \(U_{2\ell+1}\) 为奇函数，所有导致 \(t^{-5}\)、\(t^{-7}\) 的积分项均为整体奇函数的期望，故严格为零。
其余系数可归结为有限个有理积分常数；将这些常数代回即可得到 \(B_6,B_8\) 的闭式表达。证毕。
\end{proof}

\begin{theorem}[四阶可微 \texorpdfstring{$f$}{f}-散度的八阶喷流]\label{thm:conclusion-poisson-cauchy-fdiv-eighth-order-jet}
令 \(f:(0,\infty)\to\RR\) 为凸函数，满足 \(f(1)=0\)，并在 \(1\) 的邻域 \(C^4\) 且 \(f^{(j)}(1)\)（\(j=2,3,4\)）有限。
定义 Csisz\'ar \texorpdfstring{$f$}{f}-散度
\[
D_f(\tilde h_t\mid\tilde g_t):=\int_{\RR}\tilde g_t(x)\,f\!\left(\frac{\tilde h_t(x)}{\tilde g_t(x)}\right)\dd x.
\]
在定理 \ref{thm:conclusion-poisson-cauchy-kl-eighth-order-jet} 的矩条件下，当 \(t\to\infty\) 有
\[
D_f(\tilde h_t\mid\tilde g_t)
=\frac{f''(1)\mu_2^2}{8t^4}+\frac{C_6(f)}{t^6}+\frac{C_8(f)}{t^8}+o(t^{-8}),
\]
其中
\[
C_6(f)=-\frac{f^{(3)}(1)\mu_2^3}{64}-\frac{f''(1)\mu_2 \mu_4}{8}+\frac{3f''(1)\mu_3^2}{32},
\]
\[
C_8(f)=\frac1{512}\Bigl(
f''(1)\bigl(24\mu_2 \mu_6-60\mu_3\mu_5+40\mu_4^2\bigr)
+f^{(3)}(1)\bigl(24\mu_2^2\mu_4-18\mu_2\mu_3^2\bigr)
+f^{(4)}(1)\cdot 3\mu_2^4
\Bigr).
\]
\end{theorem}
\begin{proof}[证明要点]
令 \(\delta_t=\tilde h_t/\tilde g_t-1\) 并对 \(f(1+\delta)\) 在 \(\delta=0\) 处作四阶 Taylor 展开。
由于质量守恒 \(\int \tilde g_t\delta_t=0\)，一次项在积分中消失。
再把 \(\delta_t\) 的六阶喷流代入并按 \(t^{-k}\) 收集至 \(k=8\)，即可把系数归结为同一组有理积分常数；将常数代回得到所示闭式。证毕。
\end{proof}

\begin{theorem}[R\'enyi 散度的六阶一参数闭式]\label{thm:conclusion-poisson-cauchy-renyi-sixth-order-expansion}
在定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion} 的假设下，取 \(\alpha>0,\ \alpha\neq 1\)，定义
\[
D_\alpha(\tilde h_t\mid \tilde g_t)
:=
\frac{1}{\alpha-1}\log\!\int_{\RR}\tilde h_t^\alpha \tilde g_t^{1-\alpha}\,\dd x
=
\frac{1}{\alpha-1}\log\!\int_{\RR}\tilde g_t(1+\delta_t)^\alpha\,\dd x.
\]
则
\[
D_\alpha(\tilde h_t\mid \tilde g_t)
=
\frac{\alpha\mu_2^2}{8\,t^4}
+
\frac1{t^6}\left[
\alpha\left(\frac{3\mu_3^2}{32}-\frac{\mu_2\mu_4}{8}\right)
-\frac{\alpha(\alpha-2)\mu_2^3}{64}
\right]
+o(t^{-6}).
\]
且当 \(\alpha\to 1\) 时该展开连续退化为
定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion}。
\end{theorem}
\begin{proof}
由二项式展开
\[
(1+\delta_t)^\alpha
=
1+\alpha\delta_t+\frac{\alpha(\alpha-1)}{2}\delta_t^2+\frac{\alpha(\alpha-1)(\alpha-2)}{6}\delta_t^3+O(\delta_t^4),
\]
积分后用 \(\int \tilde g_t\delta_t=0\) 得
\[
\int \tilde g_t(1+\delta_t)^\alpha
=
1+\frac{\alpha(\alpha-1)}{2}\int\tilde g_t\delta_t^2
+
\frac{\alpha(\alpha-1)(\alpha-2)}{6}\int\tilde g_t\delta_t^3
+O(t^{-8}).
\]
再用 \(\log(1+u)=u+O(u^2)\)（此处 \(u=O(t^{-4})\)）及前一条证明中的同一积分常数，即得所示系数。
\(\alpha\to1\) 极限由系数显式连续性直接给出。证毕。
\end{proof}

\begin{corollary}[信息耗散泛函的七阶精细渐近]\label{cor:conclusion-poisson-cauchy-i1-seventh-order-expansion}
沿用定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion} 的假设。
若 \(I_1\) 满足耗散恒等式
\[
\frac{\dd}{\dd t}D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)=-I_1(\tilde h_t\mid \tilde g_t),
\]
则
\[
I_1(\tilde h_t\mid \tilde g_t)
=
\frac{\mu_2^2}{2\,t^5}
+
\frac1{t^7}
\left(
\frac{3\mu_2^3}{32}
-\frac{3\mu_2\mu_4}{4}
+\frac{9\mu_3^2}{16}
\right)
+o(t^{-7}).
\]
\end{corollary}
\begin{proof}
将定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion} 的展开逐项求导并取负号：
\[
I_1=-\frac{\dd}{\dd t}D_{\mathrm{KL}}
=4\cdot\frac{\mu_2^2}{8}t^{-5}
+6\left(
\frac{\mu_2^3}{64}
-\frac{\mu_2\mu_4}{8}
+\frac{3\mu_3^2}{32}
\right)t^{-7}
+o(t^{-7}),
\]
化简即得。证毕。
\end{proof}

\begin{corollary}[信息耗散泛函的九阶喷流]\label{cor:conclusion-poisson-cauchy-i1-ninth-order-jet}
在定理 \ref{thm:conclusion-poisson-cauchy-kl-eighth-order-jet} 的假设下，若 \(I_1\) 满足耗散恒等式
\[
\frac{\dd}{\dd t}D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)=-I_1(\tilde h_t\mid \tilde g_t),
\]
则当 \(t\to\infty\) 有
\[
I_1(\tilde h_t\mid \tilde g_t)
=\frac{\mu_2^2}{2t^5}+\frac{6B_6}{t^7}+\frac{8B_8}{t^9}+o(t^{-9}),
\]
其中 \(B_6,B_8\) 与定理 \ref{thm:conclusion-poisson-cauchy-kl-eighth-order-jet} 相同。
\end{corollary}
\begin{proof}
对定理 \ref{thm:conclusion-poisson-cauchy-kl-eighth-order-jet} 的展开逐项求导并取负号即可；余项 \(o(t^{-8})\) 的导数为 \(o(t^{-9})\) 由支配收敛与差分形式的耗散律正则化给出。证毕。
\end{proof}

\begin{proposition}[四阶中心矩的显式解码：由 \texorpdfstring{$L^1$}{L1} 三阶喷流与 KL 六阶喷流确定]\label{prop:conclusion-poisson-cauchy-m4-decode}
\leanverified{paper\_conclusion\_poisson\_cauchy\_m4\_decode}
设 \(\mu_2>0\)、\(\mu_5^\ast<\infty\) 且 \(\mu_4<\infty\)。
令
\[
\Delta_t(x):=\tilde h_t(x)-\tilde g_t(x)-\mu_2\tilde r_t(x)
\]
如定理 \ref{thm:conclusion-poisson-cauchy-third-order-tv-constant} 所定义，并令
\[
A_3:=\lim_{t\to\infty} t^3\|\Delta_t\|_{L^1(\RR)}
\qquad(\text{从而 }A_3=\frac{|\mu_3|}{\pi}).
\]
若再令
\[
B_6:=\lim_{t\to\infty}t^6\!\left(D_{\mathrm{KL}}(\tilde h_t\mid \tilde g_t)-\frac{\mu_2^2}{8t^4}\right)
\qquad(\text{从而 }B_6=\frac{\mu_2^3}{64}-\frac{\mu_2\mu_4}{8}+\frac{3\mu_3^2}{32}),
\]
则第四中心矩 \(\mu_4\) 可由 \((\mu_2,A_3,B_6)\) 显式重构：
\[
\mu_4
=\frac{\mu_2^2}{8}+\frac{3\mu_3^2}{4\mu_2}-\frac{8B_6}{\mu_2}
=\frac{\mu_2^2}{8}+\frac{3\pi^2A_3^2}{4\mu_2}-\frac{8B_6}{\mu_2}.
\]
特别地，若 \(\tilde\nu\) 关于 \(\bar\gamma\) 对称（等价于 \(\mu_3=0\)），则无需三阶 \(\,L^1\) 喷流即可由 KL 六阶喷流反演
\[
\mu_4=\frac{\mu_2^2}{8}-\frac{8B_6}{\mu_2}.
\]
若进一步满足 \(\mu_5=0\) 且定理 \ref{thm:conclusion-poisson-cauchy-kl-eighth-order-jet} 的八阶喷流成立，则由 \(B_8\) 还可显式解码第六中心矩：
\[
\mu_6=\frac{256B_8-3\mu_2^4+12\mu_2^2\mu_4-20\mu_4^2}{12\mu_2}.
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-poisson-cauchy-third-order-tv-constant} 得 \(A_3=|\mu_3|/\pi\)，从而 \(\mu_3^2=\pi^2A_3^2\)。
另一方面，由定理 \ref{thm:conclusion-poisson-cauchy-kl-sixth-order-expansion} 可得 \(B_6\) 的中心矩表达式。
对该代数恒等式解出 \(\mu_4\) 即得结论。证毕。
\end{proof}

\begin{theorem}[时间纤维的 Cauchy 化与 Jensen 缺陷期望表示]\label{thm:conclusion-time-fiber-cauchy-jensen-representation}
定义时间纤维
\[
t(\varrho,\theta)
=
\mathsf C^{-1}(\varrho e^{i\theta})
=
\frac{2\varrho\sin\theta}{1+\varrho^2+2\varrho\cos\theta}
+
i\,\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta},
\]
并令
\[
D_\zeta(\varrho)
=
\frac1{2\pi}\int_0^{2\pi}
\log\Bigl|\Xi\!\bigl(\tfrac12+i\,t(\varrho,\theta)\bigr)\Bigr|\,\dd\theta
-\log|\Xi(-\tfrac12)|.
\]
取半角变量 \(u=\tan(\theta/2)\)，则
\[
\frac{\dd\theta}{2\pi}=\frac{\dd u}{\pi(1+u^2)}.
\]
再令
\[
\lambda:=\frac{1-\varrho}{1+\varrho}\in(0,1),
\]
则
\[
t(\varrho,\theta(u))=x_\lambda(u)+i\,y_\lambda(u),
\]
其中
\[
x_\lambda(u)=\frac{(1-\lambda^2)u}{1+\lambda^2u^2},\qquad
y_\lambda(u)=\frac{\lambda(1+u^2)}{1+\lambda^2u^2}.
\]
故
\[
D_\zeta(\varrho)
=
\int_{\RR}
\log\Bigl|\Xi\!\bigl(\tfrac12+i(x_\lambda(u)+iy_\lambda(u))\bigr)\Bigr|
\frac{\dd u}{\pi(1+u^2)}
-\log|\Xi(-\tfrac12)|.
\]
\end{theorem}
\begin{proof}
由恒等式
\[
\sin\theta=\frac{2u}{1+u^2},\qquad
\cos\theta=\frac{1-u^2}{1+u^2},\qquad
\dd\theta=\frac{2\,\dd u}{1+u^2},
\]
可得测度变换式。
代入 \(t(\varrho,\theta)\) 的分子分母并统一化简：
\[
1+\varrho^2+2\varrho\cos\theta
=
\frac{(1+\varrho)^2+(1-\varrho)^2u^2}{1+u^2}.
\]
于是
\[
\Re t(\varrho,\theta(u))
=
\frac{4\varrho u}{(1+\varrho)^2+(1-\varrho)^2u^2},
\]
\[
\Im t(\varrho,\theta(u))
=
\frac{(1-\varrho^2)(1+u^2)}{(1+\varrho)^2+(1-\varrho)^2u^2}.
\]
再用
\[
\lambda=\frac{1-\varrho}{1+\varrho},\quad
\frac{4\varrho}{(1+\varrho)^2}=1-\lambda^2,\quad
\frac{1-\varrho^2}{(1+\varrho)^2}=\lambda
\]
得到 \(x_\lambda,y_\lambda\) 的显式式子。
将其与 Cauchy 权重变换代回 \(D_\zeta(\varrho)\) 即得结论。证毕。
\end{proof}

\begin{theorem}[端点爆破的普适半圆极限]\label{thm:conclusion-time-fiber-endpoint-universal-semicircle}
沿用定理 \ref{thm:conclusion-time-fiber-cauchy-jensen-representation} 记号，令
\[
\lambda=\frac{1-\varrho}{1+\varrho}\to 0^+,\qquad
u=\frac{v}{\lambda},
\]
并定义端点爆破缩放
\[
T_\lambda(v):=\lambda\,t\!\left(\varrho,\theta\!\left(\frac{v}{\lambda}\right)\right).
\]
则对任意 \(M>0\)，在 \([-M,M]\) 上一致收敛
\[
T_\lambda(v)\longrightarrow T_\ast(v):=\frac{v+i\,v^2}{1+v^2}.
\]
且其像满足
\[
\bigl(\Re T_\ast(v)\bigr)^2+\left(\Im T_\ast(v)-\frac12\right)^2=\frac14.
\]
故极限轨道为圆心 \(i/2\)、半径 \(1/2\) 的半圆。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-time-fiber-cauchy-jensen-representation}，
\[
x_\lambda\!\left(\frac{v}{\lambda}\right)
=
\frac{(1-\lambda^2)(v/\lambda)}{1+v^2}
=
\frac{v}{\lambda(1+v^2)}+O(\lambda),
\]
\[
y_\lambda\!\left(\frac{v}{\lambda}\right)
=
\frac{\lambda(1+v^2/\lambda^2)}{1+v^2}
=
\frac{v^2}{\lambda(1+v^2)}+O(\lambda),
\]
故
\[
T_\lambda(v)
=
\lambda\bigl(x_\lambda(v/\lambda)+i\,y_\lambda(v/\lambda)\bigr)
=
\frac{v+i\,v^2}{1+v^2}+O(\lambda^2),
\]
并在任意紧集上一致成立。
圆方程由代数恒等式
\[
\left(\frac{v}{1+v^2}\right)^2
+
\left(\frac{v^2}{1+v^2}-\frac12\right)^2
=\frac14
\]
直接得到。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$L^2$}{L2} 与 \texorpdfstring{$L^\infty$}{Linf} 的二阶锐衰减常数]\label{thm:conclusion-poisson-cauchy-l2-linf-sharp}
在定理 \ref{thm:conclusion-poisson-cauchy-third-order-tv-constant} 假设下（\(\mu_4<\infty\)），有
\[
\|\tilde h_t-\tilde g_t\|_{L^2(\RR)}
=
\frac{\sqrt3}{4\sqrt\pi}\frac{\mu_2}{t^{5/2}}+o(t^{-5/2}),
\]
\[
\|\tilde h_t-\tilde g_t\|_{L^\infty(\RR)}
=
\frac{1}{\pi}\frac{\mu_2}{t^{3}}+o(t^{-3}).
\]
\end{theorem}
\begin{proof}
由二阶正规形
\[
\tilde h_t-\tilde g_t=\mu_2\tilde r_t+o(t^{-3})
\]
（由 \(\mu_4<\infty\) 的 Taylor 余项控制）可知只需计算 \(\tilde r_t\) 的常数。
写 \(y=(x-\bar\gamma)/t\)，则
\[
\tilde r_t(x)=\frac1\pi\,t^{-3}\frac{3y^2-1}{(1+y^2)^3}.
\]
于是
\[
\|\tilde r_t\|_{L^2}^2
=
\frac{t^{-5}}{\pi^2}\int_{\RR}\frac{(3y^2-1)^2}{(1+y^2)^6}\,\dd y.
\]
直接积分得
\[
\int_{\RR}\frac{(3y^2-1)^2}{(1+y^2)^6}\,\dd y=\frac{3\pi}{16},
\]
故
\[
\|\tilde r_t\|_{L^2}
=
\frac{\sqrt3}{4\sqrt\pi}\,t^{-5/2}.
\]
再看 \(L^\infty\)：
\[
\left|\frac{3y^2-1}{(1+y^2)^3}\right|
\]
在 \(y=0\) 处取最大值 \(1\)，故
\[
\|\tilde r_t\|_{L^\infty}=\frac1\pi t^{-3}.
\]
与主项分解合并即得结论。证毕。
\end{proof}
