\paragraph{纤维开关不变代数的布尔谱、扇区全息多项式与抽象塌缩}
沿用推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric}、定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor}、推论 \ref{cor:pom-toggle-gauge-sector-decomposition-projections}、推论 \ref{cor:pom-fiber-reconstruction-spectrum-complete} 与定理 \ref{thm:pom-fiber-reconstruction-aut-group} 的记号。对固定稳定类型 \(x\)，记
\[
\Gamma_x\cong\bigsqcup_{i=1}^{r(x)}P_{\ell_i(x)},
\qquad
\mathcal F_x\cong \prod_{i=1}^{r(x)}\Gamma_{\ell_i(x)},
\qquad
n_i:=F_{\ell_i(x)+2},
\]
\[
\mathcal H_x:=\ell^2(\mathcal F_x)\cong \bigotimes_{i=1}^{r(x)}\ell^2([n_i]),
\qquad
\mathcal T_x\cong \prod_{i=1}^{r(x)}\mathfrak S_{n_i}.
\]
前述结果已经给出：\(\End(\mathcal H_x)^{\mathcal T_x}\) 是一个 \(2^{r(x)}\)-维交换半单代数，并由各分量上的平凡/标准投影张量生成。由此还可进一步把“完整几何可恢复”与“抽象代数极端遗忘”压缩为同一条链。

\begin{theorem}[开关不变代数的布尔谱与扇区维数多项式]\label{thm:conclusion-fiber-toggle-boolean-spectrum-polynomial}
\leanverified{paper\_conclusion\_fiber\_toggle\_boolean\_spectrum\_polynomial}
对每个 \(1\le i\le r(x)\)，记
\[
P_i^{\mathrm{triv}}:=\frac{1}{n_i}J_i,
\qquad
P_i^{\mathrm{std}}:=I_i-\frac{1}{n_i}J_i.
\]
对任意子集 \(S\subseteq [r(x)]\)，定义
\[
E_S:=
\bigotimes_{i\in S}P_i^{\mathrm{std}}
\otimes
\bigotimes_{i\notin S}P_i^{\mathrm{triv}}.
\]
则 \(\{E_S\}_{S\subseteq[r(x)]}\) 构成 \(\End(\mathcal H_x)^{\mathcal T_x}\) 的一组规范极小幂等元，并满足
\[
E_SE_T=\delta_{S,T}E_S,
\qquad
\sum_{S\subseteq[r(x)]}E_S=I_{\mathcal H_x}.
\]
从而
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \CC^{2^{r(x)}}.
\]

若进一步定义扇区维数生成多项式
\[
\mathscr P_x(t):=\sum_{S\subseteq[r(x)]}\dim(E_S\mathcal H_x)\,t^{|S|},
\]
则有闭式
\[
\mathscr P_x(t)=\prod_{i=1}^{r(x)}\bigl(1+(n_i-1)t\bigr).
\]
特别地，
\[
\dim(E_S\mathcal H_x)=\prod_{i\in S}(n_i-1),
\qquad
\mathscr P_x(1)=|\mathcal F_x|=\prod_{i=1}^{r(x)}n_i.
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:pom-toggle-gauge-invariant-endomorphisms-c2-tensor} 已给出
\[
\End(\mathcal H_x)^{\mathcal T_x}
\cong
\bigotimes_{i=1}^{r(x)}\mathrm{span}\{I_i,J_i\}
\cong
\bigotimes_{i=1}^{r(x)}\CC^2,
\]
而推论 \ref{cor:pom-toggle-gauge-sector-decomposition-projections} 给出正交分解
\[
\mathcal H_x
=
\bigoplus_{S\subseteq[r(x)]}
\Bigl(\bigotimes_{i\in S}\mathrm{Im}(P_i^{\mathrm{std}})\Bigr)
\otimes
\Bigl(\bigotimes_{i\notin S}\mathrm{Im}(P_i^{\mathrm{triv}})\Bigr).
\]
在该分解下，每个 \(\End(\mathcal H_x)^{\mathcal T_x}\)-元在每个扇区上都按标量作用，故 \(E_S\) 正是该交换半单代数的 primitive 幂等元。由扇区正交性立即得到
\[
E_SE_T=\delta_{S,T}E_S,
\qquad
\sum_{S}E_S=I_{\mathcal H_x}.
\]
于是
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \bigoplus_{S\subseteq[r(x)]}\CC E_S\cong \CC^{2^{r(x)}}.
\]

又
\[
\dim \mathrm{Im}(P_i^{\mathrm{triv}})=1,
\qquad
\dim \mathrm{Im}(P_i^{\mathrm{std}})=n_i-1,
\]
故
\[
\dim(E_S\mathcal H_x)=\prod_{i\in S}(n_i-1).
\]
对全部 \(S\subseteq[r(x)]\) 求和即得
\[
\mathscr P_x(t)
=
\sum_{S\subseteq[r(x)]}\Bigl(\prod_{i\in S}(n_i-1)\Bigr)t^{|S|}
=
\prod_{i=1}^{r(x)}\bigl(1+(n_i-1)t\bigr).
\]
最后令 \(t=1\)，便得
\[
\mathscr P_x(1)=\prod_i n_i.
\]
而 \(\prod_i n_i=|\mathcal F_x|\) 来自定理 \ref{thm:pom-fiber-reconstruction-cartesian-product} 的 Cartesian 乘积公式。证毕。
\end{proof}

\begin{theorem}[扇区维数多项式对纤维谱的完全反演]\label{thm:conclusion-fiber-toggle-sector-polynomial-complete-reconstruction}
\leanverified{paper\_conclusion\_fiber\_toggle\_sector\_polynomial\_complete\_reconstruction}
多项式 \(\mathscr P_x(t)\) 唯一决定多重集
\[
\{n_1,\dots,n_{r(x)}\}
=
\{F_{\ell_1(x)+2},\dots,F_{\ell_{r(x)}(x)+2}\},
\]
从而唯一决定长度多重集
\[
\mathsf{Spec}_{\square}(x):=\{\ell_1(x),\dots,\ell_{r(x)}(x)\}.
\]
特别地，若
\[
\mathscr P_x(t)=\mathscr P_y(t),
\]
则
\[
\mathsf{Spec}_{\square}(x)=\mathsf{Spec}_{\square}(y)
\qquad\text{且}\qquad
\mathcal F_x\cong \mathcal F_y.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fiber-toggle-boolean-spectrum-polynomial}，
\[
\mathscr P_x(t)=\prod_{i=1}^{r(x)}\bigl(1+(n_i-1)t\bigr).
\]
因此 \(\mathscr P_x\) 的根恰为
\[
-\frac{1}{n_i-1}
\qquad (1\le i\le r(x)),
\]
按重数计。故 \(\mathscr P_x\) 唯一确定多重集 \(\{n_i-1\}\)，亦即唯一确定 \(\{n_i\}\)。

又 Fibonacci 数列严格递增，故每个
\[
n_i=F_{\ell_i(x)+2}
\]
唯一对应一个 \(\ell_i(x)\)。因此 \(\mathscr P_x\) 唯一决定 \(\mathsf{Spec}_{\square}(x)\)。
最后由推论 \ref{cor:pom-fiber-reconstruction-spectrum-complete}，长度多重集是 \(\mathcal F_x\) 的完备图同构不变量，遂得
\[
\mathscr P_x=\mathscr P_y
\Longrightarrow
\mathcal F_x\cong \mathcal F_y.
\]
证毕。
\end{proof}

\begin{corollary}[扇区维数多项式对几何与对称的全恢复性]\label{cor:conclusion-fiber-toggle-sector-polynomial-recovers-geometry-symmetry}
\leanverified{paper\_conclusion\_fiber\_toggle\_sector\_polynomial\_recovers\_geometry\_symmetry}
一旦 \(\mathscr P_x(t)\) 给定，下列数据都可被唯一恢复：
\[
\mathcal T_x\cong \prod_{i=1}^{r(x)}\mathfrak S_{n_i},
\]
\[
\Aut(\mathcal F_x)\cong (\ZZ/2\ZZ)^{r(x)}\rtimes \prod_{\ell\ge 1}\mathfrak S_{m_\ell(x)},
\qquad
m_\ell(x):=\#\{i:\ell_i(x)=\ell\},
\]
\[
|\mathcal F_x|=\mathscr P_x(1),
\qquad
\mathrm{diam}(\mathcal F_x)=\sum_{i=1}^{r(x)}\ell_i(x),
\qquad
\dim_{\square}(\mathcal F_x)=\sum_{i=1}^{r(x)}\left\lceil\frac{\ell_i(x)}{2}\right\rceil.
\]
因此，\(\mathscr P_x\) 同时编码了纤维几何、有限群对称与立方维结构。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-fiber-toggle-sector-polynomial-complete-reconstruction}，\(\mathscr P_x\) 唯一决定多重集 \(\{n_i\}\) 以及长度谱 \(\mathsf{Spec}_{\square}(x)\)。

于是推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric} 立即给出
\[
\mathcal T_x\cong \prod_i \mathfrak S_{n_i}.
\]
定理 \ref{thm:pom-fiber-reconstruction-aut-group} 给出
\[
\Aut(\mathcal F_x)\cong (\ZZ/2\ZZ)^{r(x)}\rtimes \prod_{\ell\ge 1}\mathfrak S_{m_\ell(x)}.
\]
而定理 \ref{thm:pom-fiber-reconstruction-cartesian-product} 给出
\[
|\mathcal F_x|=\prod_i n_i,
\qquad
\mathrm{diam}(\mathcal F_x)=\sum_i\ell_i(x),
\qquad
\dim_{\square}(\mathcal F_x)=\sum_i\left\lceil\frac{\ell_i(x)}{2}\right\rceil.
\]
结合定理 \ref{thm:conclusion-fiber-toggle-boolean-spectrum-polynomial} 中
\(
\mathscr P_x(1)=\prod_i n_i
\)
即可得到全部结论。证毕。
\end{proof}

\begin{theorem}[开关不变代数的布尔函数表示与抽象塌缩]\label{thm:conclusion-fiber-toggle-boolean-function-algebra-collapse}
\leanverified{paper\_conclusion\_fiber\_toggle\_boolean\_function\_algebra\_collapse}
定义代数同态
\[
\Theta_x:
\CC[z_1,\dots,z_{r(x)}]/(z_1^2-z_1,\dots,z_{r(x)}^2-z_{r(x)})
\longrightarrow
\End(\mathcal H_x)^{\mathcal T_x},
\qquad
z_i\longmapsto P_i^{\mathrm{std}}.
\]
则 \(\Theta_x\) 为代数同构。等价地，
\[
\End(\mathcal H_x)^{\mathcal T_x}
\cong
\mathrm{Fun}(\{0,1\}^{r(x)},\CC).
\]
在该识别下，定理 \ref{thm:conclusion-fiber-toggle-boolean-spectrum-polynomial} 中的幂等元 \(E_S\) 正对应于布尔立方体点
\[
\mathbf 1_{\{\varepsilon_i=1\iff i\in S\}}.
\]

特别地，对任意两条纤维 \(x,y\)，下列命题等价：
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \End(\mathcal H_y)^{\mathcal T_y}
\quad (\text{作为复代数}),
\]
\[
r(x)=r(y).
\]
因此，只要 Cartesian 素因子个数相同，就有
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \CC^{2^{r(x)}}\cong \End(\mathcal H_y)^{\mathcal T_y},
\]
即便
\[
\mathcal F_x\not\cong \mathcal F_y,
\qquad
\mathsf{Spec}_{\square}(x)\neq \mathsf{Spec}_{\square}(y),
\qquad
\mathcal T_x\not\cong \mathcal T_y.
\]
\end{theorem}
\begin{proof}
各投影 \(P_i^{\mathrm{std}}\) 彼此交换，并满足幂等关系
\[
(P_i^{\mathrm{std}})^2=P_i^{\mathrm{std}},
\]
故 \(\Theta_x\) 良定。
又因
\[
P_i^{\mathrm{triv}}=I_i-P_i^{\mathrm{std}},
\]
所以 \(\End(\mathcal H_x)^{\mathcal T_x}\) 由 \(\{P_i^{\mathrm{std}}\}_{i=1}^{r(x)}\) 生成；这说明 \(\Theta_x\) 为满射。

另一方面，由定理 \ref{thm:conclusion-fiber-toggle-boolean-spectrum-polynomial}，
\[
\dim_\CC \End(\mathcal H_x)^{\mathcal T_x}=2^{r(x)}.
\]
而
\[
\dim_\CC \CC[z_1,\dots,z_{r(x)}]/(z_i^2-z_i)=2^{r(x)}.
\]
故 \(\Theta_x\) 亦为单射，从而是同构。有限集 \(\{0,1\}^{r(x)}\) 上的函数代数与上述商代数标准同构，于是得到
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \mathrm{Fun}(\{0,1\}^{r(x)},\CC).
\]
在这一识别下，极小幂等元自然对应于各点的示性函数，故 \(E_S\) 对应于
\(
\mathbf 1_{\{\varepsilon_i=1\iff i\in S\}}
\)。

最后，由
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \CC^{2^{r(x)}},
\qquad
\End(\mathcal H_y)^{\mathcal T_y}\cong \CC^{2^{r(y)}},
\]
而有限维交换半单复代数的同构型仅由其维数决定，遂有
\[
\End(\mathcal H_x)^{\mathcal T_x}\cong \End(\mathcal H_y)^{\mathcal T_y}
\iff
2^{r(x)}=2^{r(y)}
\iff
r(x)=r(y).
\]
若 \(r(x)=r(y)\) 但 \(\mathsf{Spec}_{\square}(x)\neq \mathsf{Spec}_{\square}(y)\)，则由推论 \ref{cor:pom-fiber-reconstruction-spectrum-complete} 得 \(\mathcal F_x\not\cong \mathcal F_y\)；再由推论 \ref{cor:pom-toggle-group-general-fiber-full-symmetric}，一般也有 \(\mathcal T_x\not\cong \mathcal T_y\)。故抽象不变代数确实发生严格的 many-to-one 塌缩。证毕。
\end{proof}

\begin{conjecture}[二重传递局部开关群的普适布尔塌缩]\label{conj:conclusion-fiber-toggle-2transitive-boolean-collapse}
设一类有限纤维系统满足：
\[
\mathcal F_x\cong \prod_{i=1}^{r(x)} G_i,
\qquad
\mathcal H_x\cong \bigotimes_{i=1}^{r(x)}\ell^2(V(G_i)),
\qquad
\mathcal T_x\cong \prod_{i=1}^{r(x)}T_i,
\]
其中每个 \(T_i\) 在 \(V(G_i)\) 上的自然作用是二重传递的。则应有普适同构
\[
\End(\mathcal H_x)^{\mathcal T_x}
\cong
(\CC^2)^{\otimes r(x)}
\cong
\mathrm{Fun}(\{0,1\}^{r(x)},\CC),
\]
从而不变算子代数的抽象同构型只依赖于因子数 \(r(x)\)，而与各因子的大小及更细图结构无关。

当前的 Fibonacci-cube 纤维族给出了该原则的完整样本：此时
\[
G_i=\Gamma_{\ell_i(x)},
\qquad
T_i\cong \mathfrak S_{F_{\ell_i(x)+2}},
\]
而对称群的自然作用正是二重传递的。
\end{conjecture}

\noindent
由此，同一条纤维开关链呈现出一对同时成立的现象。一方面，扇区维数多项式 \(\mathscr P_x\) 足以完整恢复长度谱 \(\mathsf{Spec}_{\square}(x)\)，进而恢复纤维几何、自同构群与局部开关群；另一方面，一旦只取抽象不变算子代数 \(\End(\mathcal H_x)^{\mathcal T_x}\)，其同构型又彻底塌缩为 \(\CC^{2^{r(x)}}\)，只记住 Cartesian 素因子个数。于是，精确谱全息恢复机制与极端布尔化遗忘机制并不是两条冲突叙述，而是同一开关不变接口在“带维数的扇区分解”与“忘掉扇区权重的抽象半单代数”这两个层次上的严格并存。
