\begin{theorem}[Walsh--Stokes 总荷守恒与退化度的精确 Parseval 反演]\label{thm:conclusion-fold-walsh-total-charge-parseval-degeneracy}
\leanverified{paper\_conclusion\_fold\_walsh\_total\_charge\_parseval\_degeneracy}
固定分辨率 \(m\ge 1\)。对每个 \(I\subseteq[m]\) 与 \(x\in X_m\)，记
\[
W_x(I):=\sum_{\omega\in \Fold_m^{-1}(x)}\chi_I(\omega),
\qquad
\chi_I(\omega):=(-1)^{\sum_{i\in I}\omega_i}.
\]
则成立全息总荷守恒律
\[
\sum_{x\in X_m}W_x(I)=
\begin{cases}
2^m,& I=\varnothing,\\[1ex]
0,& I\neq \varnothing,
\end{cases}
\]
并且每条纤维的退化度满足严格谱反演公式
\[
\boxed{\;
|\Fold_m^{-1}(x)|=2^{-m}\sum_{I\subseteq[m]}W_x(I)^2.\;}
\]
从而全体纤维的总 Walsh 能量满足
\[
\sum_{x\in X_m}\sum_{I\subseteq[m]}W_x(I)^2=4^m.
\]
\end{theorem}
\begin{proof}
对给定 \(I\subseteq[m]\)，令 \(\varepsilon^I\in\{\pm 1\}^m\) 为
\[
\varepsilon_i^I=
\begin{cases}
-1,& i\in I,\\
1,& i\notin I.
\end{cases}
\]
由定理 \ref{thm:pom-fiber-multivariate-holographic-conservation} 与定理 \ref{thm:pom-fiber-walsh-signature-rademacher}(1)，
\[
\sum_{x\in X_m}W_x(I)
=\sum_{x\in X_m}\mathcal H_x(\varepsilon^I)
=\prod_{i=1}^{m}(1+\varepsilon_i^I).
\]
若 \(I\neq\varnothing\)，则右端至少含一个零因子；若 \(I=\varnothing\)，则右端等于 \(2^m\)。故总荷守恒成立。

另一方面，由推论 \ref{cor:pom-fiber-walsh-stokes-boundary-tomography-reconstruction}，
\[
\ind_{\Fold_m^{-1}(x)}(\omega)
=2^{-m}\sum_{I\subseteq[m]}W_x(I)\chi_I(\omega)
\qquad (\omega\in\Omega_m).
\]
两边平方后对 \(\omega\in\Omega_m\) 求和，并用 Walsh 正交关系
\[
\sum_{\omega\in\Omega_m}\chi_I(\omega)\chi_J(\omega)=2^m\delta_{IJ},
\]
得到
\[
|\Fold_m^{-1}(x)|
=\sum_{\omega\in\Omega_m}\ind_{\Fold_m^{-1}(x)}(\omega)^2
=2^{-m}\sum_{I\subseteq[m]}W_x(I)^2.
\]
最后对 \(x\in X_m\) 求和，并用 \(\sum_x|\Fold_m^{-1}(x)|=|\Omega_m|=2^m\)，即得
\[
\sum_{x\in X_m}\sum_{I\subseteq[m]}W_x(I)^2
=2^m\sum_{x\in X_m}|\Fold_m^{-1}(x)|
=4^m.
\]
证毕。
\end{proof}

\begin{theorem}[odd-saturating Walsh 模态的 Euler--sphere 二分塌缩]\label{thm:conclusion-fold-odd-saturating-walsh-euler-sphere-collapse}

=-(-1)^{|x\cap I|}\,\widetilde\chi(|K_x|).
\]

\begin{enumerate}
  \item 若 \(|K_x|\) 可缩，则
  \[
  W_x(I)=0
  \qquad (\forall\,I\ \text{为 }x\text{-odd-saturating});
  \]
  \item 若 \(|K_x|\simeq S^{\tau(x)-1}\)，则
  \[
  W_x(I)=-(-1)^{|x\cap I|+\tau(x)-1},
  \qquad
  |W_x(I)|=1.

\end{enumerate}
换言之，全部 odd-saturating Walsh 模态整层塌缩为纤维独立集复形的球面/可缩二分及其奇偶符号。
\end{theorem}
\begin{proof}

\[
W_x(I)=(-1)^{|x\cap I|}
\sum_{S\in\mathsf{Ind}(\Gamma_x)}\prod_{v\in S}\sigma_x(I)_v.
\]
若 \(I\) 为 \(x\)-odd-saturating，则对每个 \(v\in V(\Gamma_x)\) 都有
\[
\sigma_x(I)_v=(-1)^{|I\cap J_x(v)|}=-1.
\]
故
\[
W_x(I)=(-1)^{|x\cap I|}
\sum_{S\in\mathsf{Ind}(\Gamma_x)}(-1)^{|S|}
=(-1)^{|x\cap I|}Z_x(-1).

再由命题 \ref{prop:pom-fiber-witten-index-zx-minus-one}，

\widetilde\chi(|K_x|)=-Z_x(-1),

若 \(|K_x|\simeq S^{\tau(x)-1}\)，则
\[
\widetilde\chi(|K_x|)=(-1)^{\tau(x)-1},
\]
代入即得
\[
W_x(I)=-(-1)^{|x\cap I|+\tau(x)-1},
\]
并且其模长等于 \(1\)。证毕。
\end{proof}

\begin{theorem}[边界数据决定整数射影压力骨架]\label{thm:conclusion-fold-boundary-determines-integer-projective-pressure-skeleton}
设对每个整数 \(q\ge 2\)，极限
\[
r_q:=\lim_{m\to\infty} S_q(m)^{1/m}
\]
存在。则跨分辨率边界谱逆系统
\[
\{\mathfrak B^{(m)}\}_{m\ge 1}
\]
唯一决定整条整数谱骨架 \(\{r_q\}_{q\ge 2}\)。并且对每个整数 \(q\ge 2\)，有
\[
r_q=\rho(\mathcal L_q\vert_{\mathcal H_q})=\rho(\widetilde A_q),
\]
其中 \(\mathcal L_q\) 为射影压力算子，\(\mathcal H_q\) 为齐次次数 \(q\) 多项式扇区，\(\widetilde A_q\) 为对称幂商 moment-kernel。
因此，边界 Walsh--Stokes 数据决定了整数点上的 projective-pressure 谱骨架。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-fold-boundary-spectrum-fiber-histogram-reconstruction} 与定理 \ref{thm:conclusion-fold-collision-resolvent-finite-pole}，\(\mathfrak B^{(m)}\) 唯一决定每个 \(m\) 的全部碰撞矩 \(S_q(m)\)。对固定整数 \(q\ge 2\) 取 \(m\to\infty\)，便唯一决定相应的指数极限 \(r_q\)。

另一方面，由定理 \ref{thm:conclusion-projective-pressure-polynomial-sector-exact-closure}，
\[
\rho(\mathcal L_q\vert_{\mathcal H_q})=\rho(\widetilde A_q)=r_q.
\]
故整数射影压力骨架恰由边界谱逆系统决定。证毕。
\end{proof}
