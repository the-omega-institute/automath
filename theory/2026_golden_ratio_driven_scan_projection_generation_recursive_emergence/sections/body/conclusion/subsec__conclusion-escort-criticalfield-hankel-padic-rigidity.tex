\subsection{escort 全 Rényi 冻结、临界数域解耦与 Hankel 见证的 \texorpdfstring{$p$}{p}-进刚性}\label{subsec:conclusion-escort-criticalfield-hankel-padic-rigidity}

沿用主稿第 \(11.52\)–\(11.68\) 段的记号，并结合定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy}、定理 \ref{thm:conclusion-finite-primitive-surgery-additive-harmonic-diagonalization}、推论 \ref{cor:conclusion-realinput40-atomic-fourier-fingerprint} 与定理 \ref{thm:conclusion-golden-resonance-integer-trace-not-l2} 的接口。前述链条已经分别给出了 bin-fold 体积密度的 Stirling--Bernoulli 展开、escort 纤维信息的显式常数项、真实输入核长度 \(2\) primitive 原子的 residue/Fourier 读出，以及黄金共振整函数的严格 \(\ell^2\) 越界。将这些热力学、算术与谱见证机制并置之后，还可进一步抽出如下几条后继刚性后果。它们共同说明：bin-fold 的 escort 塔在全部固定温度与全部 Rényi 阶上都冻结到同一主斜率；临界 Lee--Yang / RH 尺度之间只能通过有理常数相遇；六倍窗零块的列表审计存在不可压缩的半速率门槛；而 Hankel 见证的坏素与 \(p\)-进刚性则完全由系数型别与谱差型别锁定。

\begin{theorem}[bin-fold escort 塔的全 Rényi 率冻结]\label{thm:conclusion-binfold-escort-all-renyi-rate-freezing}
\leanverified{paper\_conclusion\_binfold\_escort\_all\_renyi\_rate\_freezing}
记
\[
S_a(m):=\sum_{w\in X_m}\bigl(d_m^{\mathrm{bin}}(w)\bigr)^a,
\qquad
\pi_{m,a}^{\mathrm{bin}}(w):=\frac{(d_m^{\mathrm{bin}}(w))^a}{S_a(m)}
\qquad (a\ge 0).
\]
则对任意固定 \(a\ge 0\) 与任意 Rényi 阶 \(q\in(0,\infty]\)，都有
\[
H_q(\pi_{m,a}^{\mathrm{bin}})=m\log\varphi+O_{a,q}(1).
\]
特别地，
\[
\lim_{m\to\infty}\frac1m H_q(\pi_{m,a}^{\mathrm{bin}})=\log\varphi
\qquad (\forall a\ge0,\ \forall q\in(0,\infty]).
\]
因此，bin-fold 的 escort 族在主阶上不存在真正的 \(q\)-多重分形；所有固定温度 \(a\) 与所有 Rényi 阶 \(q\) 的熵率都冻结到同一常数 \(\log\varphi\)。
\end{theorem}
\begin{proof}
当 \(a=0\) 时，\(\pi_{m,0}^{\mathrm{bin}}\) 是 \(X_m\) 上的均匀分布，故对每个 \(q\in(0,\infty]\) 都有
\[
H_q(\pi_{m,0}^{\mathrm{bin}})=\log|X_m|=\log F_{m+2}=m\log\varphi+O(1).
\]
下设 \(a>0\)。

由主稿第 \(11.52\) 段，对任意固定 \(b>0\) 有
\[
H_b(\mu_m)=m\log\varphi+O_b(1),
\qquad
\mu_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m}.
\]
故当 \(b\neq 1\) 时，
\[
\log\sum_{w\in X_m}\mu_m(w)^b=(1-b)m\log\varphi+O_b(1).
\]
又因为
\[
S_b(m)=2^{bm}\sum_{w\in X_m}\mu_m(w)^b,
\]
便得
\[
\log S_b(m)=bm\log2+(1-b)m\log\varphi+O_b(1)
\qquad (b>0).
\]
当 \(b=1\) 时，同一公式也精确成立，因为
\[
S_1(m)=\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)=2^m.
\]

若 \(q\in(0,\infty)\setminus\{1\}\)，则
\[
\sum_w \bigl(\pi_{m,a}^{\mathrm{bin}}(w)\bigr)^q
=
\frac{S_{aq}(m)}{S_a(m)^q},
\]
于是
\[
H_q(\pi_{m,a}^{\mathrm{bin}})
=
\frac{1}{1-q}\bigl(\log S_{aq}(m)-q\log S_a(m)\bigr)
=
m\log\varphi+O_{a,q}(1).
\]

若 \(q=1\)，则
\[
H(\pi_{m,a}^{\mathrm{bin}})
=
\log S_a(m)-a\,\mathbb E_{\pi_{m,a}^{\mathrm{bin}}}\!\bigl[\log d_m^{\mathrm{bin}}(W)\bigr].
\]
而主稿第 \(11.59\) 段已给出
\[
\mathbb E_{\pi_{m,a}^{\mathrm{bin}}}\!\bigl[\log d_m^{\mathrm{bin}}(W)\bigr]
=
m\log(2/\varphi)+O_a(1),
\]
故
\[
H(\pi_{m,a}^{\mathrm{bin}})
=
\bigl(am\log2+(1-a)m\log\varphi\bigr)-am\log(2/\varphi)+O_a(1)
=
m\log\varphi+O_a(1).
\]

若 \(q=\infty\)，则
\[
H_\infty(\pi_{m,a}^{\mathrm{bin}})
=
-\log \max_w \pi_{m,a}^{\mathrm{bin}}(w)
=
\log S_a(m)-a\log M_m,
\]
其中 \(M_m:=\max_w d_m^{\mathrm{bin}}(w)\)。由 bin-fold 两态一致渐近，
\[
\log M_m=m\log(2/\varphi)+O(1),
\]
故
\[
H_\infty(\pi_{m,a}^{\mathrm{bin}})
=
\bigl(am\log2+(1-a)m\log\varphi\bigr)-am\log(2/\varphi)+O_a(1)
=
m\log\varphi+O_a(1).
\]
综上即得结论。证毕。
\end{proof}

\begin{theorem}[escort 族在总变差上塌缩为末位二块 Gibbs 模型]\label{thm:conclusion-binfold-escort-tv-collapse-lastbit-gibbs}
\leanverified{paper\_conclusion\_binfold\_escort\_tv\_collapse\_lastbit\_gibbs}
记
\[
B_{m,0}:=\{w\in X_m:\ w_m=0\},
\qquad
B_{m,1}:=\{w\in X_m:\ w_m=1\}.
\]
定义二块 Gibbs 近似分布
\[
Q_{m,a}(w):=
\frac{1}{F_{m+1}+\varphi^{-a}F_m}
\begin{cases}
1, & w\in B_{m,0},\\[1mm]
\varphi^{-a}, & w\in B_{m,1}.
\end{cases}
\]
则对任意固定 \(a\ge0\)，有一致比率收敛
\[
\sup_{w\in X_m}\left|\frac{\pi_{m,a}^{\mathrm{bin}}(w)}{Q_{m,a}(w)}-1\right|\longrightarrow0.
\]
因而
\[
\bigl|\pi_{m,a}^{\mathrm{bin}}-Q_{m,a}\bigr|_{\mathrm{TV}}\to0,
\qquad
D_{\mathrm{KL}}(\pi_{m,a}^{\mathrm{bin}}\Vert Q_{m,a})\to0.
\]
这表明：在主阶与一阶误差的联合口径下，escort 分布并不保留整个词结构；它在概率论上真正规约为“末位比特 + 条件均匀”的二块 Gibbs 模型。
\end{theorem}
\begin{proof}
当 \(a=0\) 时，\(Q_{m,0}\) 与 \(\pi_{m,0}^{\mathrm{bin}}\) 都是 \(X_m\) 上的均匀分布，结论显然。下设 \(a>0\)。

由主稿第 \(11.58\) 段的统一两态渐近，
\[
\log d_m^{\mathrm{bin}}(w)
=
m\log(2/\varphi)-w_m\log\varphi+O\!\bigl((\varphi/2)^m\bigr)
\qquad (w\in X_m),
\]
且误差对 \(w\) 一致。故对固定 \(a\ge0\)，一致地有
\[
\bigl(d_m^{\mathrm{bin}}(w)\bigr)^a
=
e^{am\log(2/\varphi)}\varphi^{-aw_m}
\Bigl(1+O\!\bigl((\varphi/2)^m\bigr)\Bigr).
\]
于是
\[
S_a(m)
=
e^{am\log(2/\varphi)}
\Bigl(F_{m+1}+\varphi^{-a}F_m\Bigr)
\Bigl(1+O\!\bigl((\varphi/2)^m\bigr)\Bigr),
\]
其中已用到标准 Fibonacci 计数
\[
|B_{m,0}|=F_{m+1},
\qquad
|B_{m,1}|=F_m.
\]
将其代回 \(\pi_{m,a}^{\mathrm{bin}}\) 的定义，便得到
\[
\pi_{m,a}^{\mathrm{bin}}(w)
=
Q_{m,a}(w)\Bigl(1+O\!\bigl((\varphi/2)^m\bigr)\Bigr),
\]
且误差对 \(w\) 一致，从而得到所述一致比率收敛。

总变差收敛随之成立，因为
\[
\bigl|\pi_{m,a}^{\mathrm{bin}}-Q_{m,a}\bigr|_{\mathrm{TV}}
\le
\frac12\sup_w\left|\frac{\pi_{m,a}^{\mathrm{bin}}(w)}{Q_{m,a}(w)}-1\right|.
\]
又由一致比率收敛，
\[
\sup_w\left|\log\frac{\pi_{m,a}^{\mathrm{bin}}(w)}{Q_{m,a}(w)}\right|\to0,
\]
故
\[
D_{\mathrm{KL}}(\pi_{m,a}^{\mathrm{bin}}\Vert Q_{m,a})
=
\sum_w \pi_{m,a}^{\mathrm{bin}}(w)\log\frac{\pi_{m,a}^{\mathrm{bin}}(w)}{Q_{m,a}(w)}
\longrightarrow 0.
\]
证毕。
\end{proof}

\begin{theorem}[Lee--Yang 临界尺度与 RH 临界尺度的有理函数解耦]\label{thm:conclusion-critical-scales-rational-function-decoupling}
\leanverified{paper\_conclusion\_critical\_scales\_rational\_function\_decoupling}
设
\[
u_{LY},\quad b,\quad u_R
\]
如主稿第 \(11.53\) 段所定。对任意有理函数
\[
F,G\in \QQ(T),
\]
若 \(F(u_R)\) 与 \(G(u_{LY})\) 都有定义且
\[
F(u_R)=G(u_{LY}),
\]
则该公共值必属于 \(\QQ\)。同理，若
\[
F(u_R)=G(b),
\]
则该公共值也必属于 \(\QQ\)。

换言之，在已审计的两条临界尺度链之间，任何非平凡的有理函数匹配都被严格禁止；它们只可能通过纯有理常数相遇。
\end{theorem}
\begin{proof}
主稿第 \(11.53\) 段已给出
\[
\QQ(u_R)\cap \QQ(u_{LY})=\QQ,
\qquad
\QQ(b)=\QQ(u_{LY}).
\]
若 \(F(u_R)=G(u_{LY})\)，则左端属于 \(\QQ(u_R)\)，右端属于 \(\QQ(u_{LY})\)，故其公共值属于
\[
\QQ(u_R)\cap \QQ(u_{LY})=\QQ.
\]
这便得到第一结论。

若 \(F(u_R)=G(b)\)，则右端属于 \(\QQ(b)=\QQ(u_{LY})\)，故同理也落在
\[
\QQ(u_R)\cap \QQ(b)=\QQ(u_R)\cap \QQ(u_{LY})=\QQ.
\]
证毕。
\end{proof}

\begin{corollary}[十五次临界合成域上的对数全体超越性]\label{cor:conclusion-critical-degree-fifteen-compositum-log-transcendence}

K:=\QQ(u_R,u_{LY})=\QQ(u_R,b).

[K:\QQ]=15.

则对任意
\[
\alpha\in K^\times\setminus\{1\},

每一个满足
\[
e^L=\alpha
\]
的复数 \(L\) 都是超越数。

因此，主稿第 \(11.54\) 段对若干特殊尺度参数逐个建立的对数超越性，自动扩张为整个十五次临界合成域上的全体对数超越性。

\end{proof}

\begin{theorem}[六倍窗零块的次指数列表审计不可能性]\label{thm:conclusion-sixfold-zero-block-subexponential-list-impossibility}
\leanverified{paper\_conclusion\_sixfold\_zero\_block\_subexponential\_list\_impossibility}
设 \(m+2\equiv 0\ (\mathrm{mod}\ 6)\)，并令
\[
\mathcal E_m: Z_m\to\{0,1\}^{\ell_m}
\]
为任意二进制审计标号。假设每个码字的原像大小至多为 \(L_m\)，即
\[
|\mathcal E_m^{-1}(c)|\le L_m
\qquad
(\forall c\in\{0,1\}^{\ell_m}).
\]
则必有
\[
2^{\ell_m}L_m\ge |Z_m|.
\]
特别地，若
\[
\log L_m=o(m),
\]
则
\[
\liminf_{\substack{m\to\infty\\ m+2\equiv0\ (\mathrm{mod}\ 6)}}\frac{\ell_m}{m}
\ge \frac12\log_2\varphi.
\]

因此，主稿第 \(11.55\) 段中的半速率下界并不只排斥单射编码；它同样排斥一切次指数列表大小的多值审计。
\end{theorem}
\begin{proof}
由 pigeonhole 原理，
\[
|Z_m|
=
\sum_{c\in\{0,1\}^{\ell_m}} |\mathcal E_m^{-1}(c)|
\le 2^{\ell_m}L_m.
\]
这便得到第一式。

另一方面，主稿第 \(11.55\) 段已给出
\[
\log_2|Z_m|\ge \frac{m}{2}\log_2\varphi+O(1)
\qquad
(m+2\equiv0\ (\mathrm{mod}\ 6)).
\]
故
\[
\ell_m
\ge
\log_2|Z_m|-\log_2L_m
\ge
\frac{m}{2}\log_2\varphi+O(1)-\log_2L_m.
\]
若 \(\log L_m=o(m)\)，则两边除以 \(m\) 并取 \(\liminf\) 即得结论。证毕。
\end{proof}

\begin{theorem}[primitive 长度 \texorpdfstring{$2$}{2} 原子的加性 Fourier 秩一性]\label{thm:conclusion-primitive-length-two-additive-fourier-rank-one}
\leanverified{paper\_conclusion\_primitive\_length\_two\_additive\_fourier\_rank\_one}
对任意模数 \(q\ge2\) 与剩余类 \(a\in\ZZ/q\ZZ\)，定义原子剩余轮廓
\[
A_q(a;z,u):=P^{[a,q]}(z,u)-P_{\mathrm{core}}^{[a,q]}(z,u).
\]
则
\[
A_q(a;z,u)=\mathbf 1_{a\equiv 2\ (\mathrm{mod}\ q)}\,u z^2.
\]
其加性离散 Fourier 变换满足
\[
\widehat A_q(j;z,u)
:=
\sum_{a\in\ZZ/q\ZZ}\zeta_q^{aj}A_q(a;z,u)
=
u z^2\,\zeta_q^{2j},
\qquad
j\in\ZZ/q\ZZ.
\]
因此，长度 \(2\) 的 primitive 原子在完整同余分解上不是一个扩散分布，而是一条秩一的加性相位直线；任意一个 Fourier 模式都已足够完全恢复该原子。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:conclusion-finite-primitive-surgery-additive-harmonic-diagonalization} 在唯一原子
\[
(m,\ell,E)=(1,2,1)
\]
情形下的直接特化。第一式成立后，
\[
\widehat A_q(j;z,u)
=
\sum_{a\in\ZZ/q\ZZ}\zeta_q^{aj}\mathbf 1_{a\equiv 2\ (\mathrm{mod}\ q)}\,u z^2
=
u z^2\,\zeta_q^{2j},
\]
即得第二式。证毕。
\end{proof}

\begin{corollary}[primitive 原子的字符刚性]\label{cor:conclusion-primitive-atom-character-rigidity}
沿用定理 \ref{thm:conclusion-primitive-length-two-additive-fourier-rank-one} 的记号。对任意 Dirichlet 角色 \(\chi\pmod q\)（按标准约定扩张到全部剩余类），定义字符投影
\[
A_\chi(z,u):=\sum_{a\in\ZZ/q\ZZ}\chi(a)\,A_q(a;z,u).
\]
则有精确公式
\[
A_\chi(z,u)=\chi(2)\,u z^2.
\]
特别地，若 \(q\) 为奇数，则对所有角色都有 \(\chi(2)\neq0\)，故没有任何角色通道能够湮灭该 primitive 长度 \(2\) 原子；而当 \(q\) 为偶数时，原子的消失与否完全由 \(\chi(2)\) 单独决定。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-primitive-length-two-additive-fourier-rank-one}，
\[
A_q(a;z,u)=\mathbf 1_{a\equiv 2\ (\mathrm{mod}\ q)}\,u z^2,
\]
故
\[
A_\chi(z,u)
=
\sum_{a\in\ZZ/q\ZZ}\chi(a)\mathbf 1_{a\equiv 2\ (\mathrm{mod}\ q)}\,u z^2
=
\chi(2)\,u z^2.
\]
当 \(q\) 为奇数时，\(2\) 在模 \(q\) 下可逆，故任意 Dirichlet 角色在 \(2\) 处不取零，结论随之成立。证毕。
\end{proof}

\begin{theorem}[fold-gauge 常数的首缺陷指数律]\label{thm:conclusion-fold-gauge-first-defect-exponential-law}
设另一列正数 \(\widetilde C_m\) 也 admit 主稿第 \(11.64\)–\(11.66\) 段所述的同型 Stirling--Bernoulli 展开
\[
\log \widetilde C_m-\log(2\pi)
=
\sum_{r\ge1}\widetilde A_r \lambda_r^m,
\qquad
\lambda_r=q^{2r-1},
\qquad
q=\varphi/2.
\]
其中 \(A_r\) 记原 fold-gauge 常数链 \(C_m\) 的对应展开系数：
\[
\log C_m-\log(2\pi)=\sum_{r\ge1}A_r\lambda_r^m.
\]
若 \(r_0\) 是满足
\[
\widetilde A_{r_0}\neq A_{r_0}
\]
的最小指标，则有 sharp 渐近
\[
\log \widetilde C_m-\log C_m
=
(\widetilde A_{r_0}-A_{r_0})\lambda_{r_0}^m
+ O(\lambda_{r_0+1}^m),
\]
从而
\[
\lim_{m\to\infty}\lambda_{r_0}^{-m}
\bigl(\log \widetilde C_m-\log C_m\bigr)
=
\widetilde A_{r_0}-A_{r_0}\neq0.
\]

因此，两条 fold-gauge 常数链之间的首个 jet 缺口不会被更深层指数模态掩盖；它必以唯一的首层尺度 \(\lambda_{r_0}^m\) 显影。
\end{theorem}
\begin{proof}
由主稿第 \(11.64\)–\(11.66\) 段，
\[
\log C_m-\log(2\pi)=\sum_{r\ge1}A_r\lambda_r^m.
\]
两式相减得
\[
\log \widetilde C_m-\log C_m
=
\sum_{r\ge1}(\widetilde A_r-A_r)\lambda_r^m.
\]
若 \(r_0\) 最小且 \(\widetilde A_{r_0}\neq A_{r_0}\)，则
\[
\log \widetilde C_m-\log C_m
=
(\widetilde A_{r_0}-A_{r_0})\lambda_{r_0}^m
+ \sum_{r>r_0}(\widetilde A_r-A_r)\lambda_r^m.
\]
又因为
\[
\frac{\lambda_{r+1}}{\lambda_r}=q^2<1,
\]
故余项为 \(O(\lambda_{r_0+1}^m)\)。两边除以 \(\lambda_{r_0}^m\) 并令 \(m\to\infty\)，即可得到极限公式。证毕。
\end{proof}

\begin{theorem}[Hankel 见证的全局坏素支撑定理]\label{thm:conclusion-hankel-witness-global-bad-prime-support}
在主稿第 \(11.67\) 段的设定下，令
\[
a_n=\sum_{j=1}^r \alpha_j \lambda_j^n,
\qquad
H_r=(a_{i+j})_{0\le i,j<r},
\]
并记
\[
\Delta_{\mathrm{spec}}:=\prod_{1\le i<j\le r}(\lambda_j-\lambda_i)^2.
\]
则对任意非阿基米德赋值 \(v\) 都有
\[
v(\det H_r)=v\!\left(\prod_{j=1}^r\alpha_j\right)+v(\Delta_{\mathrm{spec}}).
\]
因此，\(\det H_r\) 的 rational bad-prime set 必包含于：
\begin{enumerate}
  \item 系数积 \(\prod_j\alpha_j\) 的坏素支撑；
  \item 谱判别式 \(\Delta_{\mathrm{spec}}\) 的坏素支撑；
  \item 底层数域 \(K/\QQ\) 的分歧素数。
\end{enumerate}
换言之，Hankel 见证的全部系统性坏素，不可能凭空生成；它们只能源自系数退化、谱碰撞或域分歧三种机制。
\end{theorem}
\begin{proof}
主稿第 \(11.67\) 段已给出精确分解
\[
\det(H_r)
=
\Bigl(\prod_{j=1}^r\alpha_j\Bigr)
\Bigl(\prod_{1\le i<j\le r}(\lambda_j-\lambda_i)^2\Bigr).
\]
对任意非阿基米德赋值 \(v\) 取赋值，便得
\[
v(\det H_r)
=
\sum_{j=1}^r v(\alpha_j)+2\sum_{1\le i<j\le r}v(\lambda_j-\lambda_i)
=
v\!\left(\prod_{j=1}^r\alpha_j\right)+v(\Delta_{\mathrm{spec}}).
\]
这便得到第一式。

若某个有理素数 \(p\) 不属于上述三类支撑，则对每个 \(v\mid p\) 都有
\[
v\!\left(\prod_j\alpha_j\right)=0,
\qquad
v(\Delta_{\mathrm{spec}})=0,
\]
并且 \(p\) 在底层数域中不分歧。于是 \(v(\det H_r)=0\) 对所有 \(v\mid p\) 成立，故 \(p\) 不会在 Hankel 见证上留下有理坏素痕迹。换言之，任一 rational bad prime 都只能来自三类来源之一。证毕。
\end{proof}

\begin{corollary}[Hankel 见证对 \texorpdfstring{$p$}{p}-adic 单位扰动的刚性]\label{cor:conclusion-hankel-witness-padic-unit-perturbation-rigidity}
沿用定理 \ref{thm:conclusion-hankel-witness-global-bad-prime-support} 的记号。设另一组数据 \((\widetilde\alpha_j,\widetilde\lambda_j)\) 满足：对某个非阿基米德赋值 \(v\)，有
\[
v(\widetilde\alpha_j)=v(\alpha_j)
\qquad (1\le j\le r),
\]
\[
v(\widetilde\lambda_j-\widetilde\lambda_i)=v(\lambda_j-\lambda_i)
\qquad (1\le i<j\le r).
\]
令 \(\widetilde H_r\) 为相应 Hankel 块。则
\[
v(\det \widetilde H_r)=v(\det H_r).
\]
特别地，任何保持“系数 \(v\)-单位型别”与“谱差 \(v\)-单位型别”不变的 \(p\)-adic 扰动，都不可能改变 Hankel 见证的刚性指数。
\end{corollary}
\begin{proof}
对原数据与新数据分别应用定理 \ref{thm:conclusion-hankel-witness-global-bad-prime-support}，得到
\[
v(\det H_r)=\sum_{j=1}^r v(\alpha_j)+2\sum_{1\le i<j\le r}v(\lambda_j-\lambda_i),
\]
\[
v(\det \widetilde H_r)=\sum_{j=1}^r v(\widetilde\alpha_j)+2\sum_{1\le i<j\le r}v(\widetilde\lambda_j-\widetilde\lambda_i).
\]
由假设，两式右端逐项相同，故结论成立。证毕。
\end{proof}
