\subsection{Schur 子群固定向量层析与 primitive 同余的陪集 Fourier 过滤}\label{subsec:conclusion-schur-subgroup-coset-fourier-tomography}
在 Schur 断层成像公式与有限阿贝尔群 primitive 同余分层公式都已建立之后，还可把二者进一步压缩为同一类子群投影机制：在对称群侧，任意子群平均恰由 Specht 模的固定向量维数给出；在有限阿贝尔群侧，对陪集的聚合则严格截去一切不在 annihilator 中的角色通道，并由活动 Fourier 支撑唯一决定最小承载商。

\begin{theorem}[Schur 子群平均的固定向量层析公式]\label{thm:conclusion-schur-subgroup-average-fixed-vector-tomography}
固定整数 \(q\ge 2\)。对任意子群
\[
H\le S_q,
\]
定义其循环型平均
\[
\mathcal A_H(m):=\frac{1}{|H|}\sum_{\sigma\in H}\mathcal C_\sigma(m),
\]
其中 \(\mathcal C_\sigma(m)\) 由定理 \ref{thm:pom-schur-cycle-index-tomography} 给出。再对每个分拆 \(\lambda\vdash q\) 定义
\[
m_H(\lambda)
:=
\dim\bigl(S^\lambda\bigr)^H
=
\frac{1}{|H|}\sum_{\sigma\in H}\chi^\lambda(\sigma),
\]
其中 \(S^\lambda\) 为 Specht 模，\(f^\lambda:=\dim S^\lambda\)。则有严格恒等式
\[
\boxed{\;
\mathcal A_H(m)
=
\sum_{\lambda\vdash q}
\frac{m_H(\lambda)}{f^\lambda}\,T_{q,\lambda}(m).\;}
\]
换言之，任意子群的循环型平均都恰好对应于 Schur 通道上的固定向量投影。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-schur-cycle-index-tomography}，
\[
\frac{T_{q,\lambda}(m)}{f^\lambda}
=
\frac{1}{q!}\sum_{\tau\in S_q}\chi^\lambda(\tau)\,\mathcal C_\tau(m).
\]
由于 \(\tau\mapsto \mathcal C_\tau(m)\) 是 \(S_q\) 上的类函数，而 \(\{\chi^\lambda\}_{\lambda\vdash q}\) 构成类函数空间的正交基，角色正交性给出反演式
\[
\mathcal C_\sigma(m)
=
\sum_{\lambda\vdash q}
\frac{\chi^\lambda(\sigma)}{f^\lambda}\,T_{q,\lambda}(m).
\]
对 \(\sigma\in H\) 取平均，得到
\[
\begin{aligned}
\mathcal A_H(m)
&=
\frac{1}{|H|}\sum_{\sigma\in H}\mathcal C_\sigma(m)\\
&=
\sum_{\lambda\vdash q}
\frac{1}{f^\lambda}
\left(
\frac{1}{|H|}\sum_{\sigma\in H}\chi^\lambda(\sigma)
\right)
T_{q,\lambda}(m).
\end{aligned}
\]
括号中的平均正是平均投影算子
\[
\frac{1}{|H|}\sum_{\sigma\in H}\rho_\lambda(\sigma)
\]
在 \(S^\lambda\) 上的迹，因此等于 \(H\)-不变向量空间 \((S^\lambda)^H\) 的维数，即 \(m_H(\lambda)\)。于是结论成立。证毕。
\end{proof}

\begin{corollary}[偶置换平均只保留平凡通道与交替通道]\label{cor:conclusion-schur-even-odd-average-trivial-alternating}
固定 \(q\ge 2\)，取交替子群 \(A_q\le S_q\)。定义偶/奇循环型平均
\[
\mathcal A_q^{+}(m):=\frac{1}{|A_q|}\sum_{\sigma\in A_q}\mathcal C_\sigma(m),
\qquad
\mathcal A_q^{-}(m):=\frac{1}{|A_q|}\sum_{\sigma\in S_q\setminus A_q}\mathcal C_\sigma(m).
\]
则有完全闭式
\[
\boxed{\;
\mathcal A_q^{+}(m)=T_{q,(q)}(m)+T_{q,(1^q)}(m),\;}
\]
\[
\boxed{\;
\mathcal A_q^{-}(m)=T_{q,(q)}(m)-T_{q,(1^q)}(m).\;}
\]
因而
\[
\boxed{\;
T_{q,(1^q)}(m)=\frac{\mathcal A_q^{+}(m)-\mathcal A_q^{-}(m)}{2},
\qquad
T_{q,(q)}(m)=\frac{\mathcal A_q^{+}(m)+\mathcal A_q^{-}(m)}{2}.\;}
\]
特别地，
\[
\boxed{\;
\bigl|T_{q,(1^q)}(m)\bigr|\le T_{q,(q)}(m).\;}
\]
\end{corollary}
\begin{proof}
由 Frobenius 互反，
\[
m_{A_q}(\lambda)
=
\Bigl\langle \operatorname{Res}^{S_q}_{A_q}S^\lambda,\mathbf 1\Bigr\rangle
=
\Bigl\langle S^\lambda,\operatorname{Ind}_{A_q}^{S_q}\mathbf 1\Bigr\rangle.
\]
而指数为 \(2\) 的子群诱导表示满足
\[
\operatorname{Ind}_{A_q}^{S_q}\mathbf 1
\cong
\mathbf 1\oplus \mathrm{sgn}.
\]
因此
\[
m_{A_q}(\lambda)=
\begin{cases}
1,& \lambda=(q)\ \text{或}\ \lambda=(1^q),\\[2pt]
0,& \text{其余分拆}.
\end{cases}
\]
代入定理 \ref{thm:conclusion-schur-subgroup-average-fixed-vector-tomography} 即得
\[
\mathcal A_q^{+}(m)=T_{q,(q)}(m)+T_{q,(1^q)}(m).
\]

另一方面，由定理 \ref{thm:pom-schur-cycle-index-tomography} 在平凡分拆 \((q)\) 上的特例，
\[
T_{q,(q)}(m)=\frac{1}{q!}\sum_{\sigma\in S_q}\mathcal C_\sigma(m)
=
\frac{\mathcal A_q^{+}(m)+\mathcal A_q^{-}(m)}{2},
\]
因为 \(q!\!=2|A_q|\)。整理即得
\[
\mathcal A_q^{-}(m)=T_{q,(q)}(m)-T_{q,(1^q)}(m)
\]
与后续两式。

最后，由于 \(\mathcal C_\sigma(m)\ge 0\) 对所有 \(\sigma\in S_q\) 都成立，故
\[
\mathcal A_q^{+}(m)\ge 0,
\qquad
\mathcal A_q^{-}(m)\ge 0.
\]
于是
\[
T_{q,(q)}(m)\pm T_{q,(1^q)}(m)\ge 0,
\]
从而
\[
\bigl|T_{q,(1^q)}(m)\bigr|\le T_{q,(q)}(m).
\]
证毕。
\end{proof}

\begin{theorem}[primitive 同余计数的陪集 Fourier 过滤公式]\label{thm:conclusion-primitive-coset-fourier-filter}
\leanverified{paper\_conclusion\_primitive\_coset\_fourier\_filter}
设 \(G\) 为有限阿贝尔群，并固定 \(n\ge 1\) 与 \(u\in(0,1]\)。令 \(H\le G\) 为子群。对陪集 \(cH\in G/H\)，定义聚合 primitive 计数
\[
B_{n,cH}(u):=\sum_{h\in H}B_{n,ch}(u).
\]
再定义
\[
H^\perp:=\operatorname{Ann}(H):=
\{\chi\in\widehat G:\chi|_H\equiv 1\}.
\]
则有精确公式
\[
\boxed{\;
B_{n,cH}(u)
=
\frac{|H|}{|G|}
\sum_{\chi\in H^\perp}\overline{\chi(c)}\,B_n(\chi;u).\;}
\]
换言之，对陪集求和会严格消去全部不属于 \(H^\perp\) 的角色通道。
\end{theorem}
\begin{proof}
由命题 \ref{prop:kernel-chebotarev-fourier}，
\[
B_{n,ch}(u)=
\frac{1}{|G|}\sum_{\chi\in\widehat G}
\overline{\chi(ch)}\,B_n(\chi;u)
=
\frac{1}{|G|}\sum_{\chi\in\widehat G}
\overline{\chi(c)}\,\overline{\chi(h)}\,B_n(\chi;u).
\]
对 \(h\in H\) 求和，得到
\[
\begin{aligned}
B_{n,cH}(u)
&=
\sum_{h\in H}B_{n,ch}(u)\\
&=
\frac{1}{|G|}\sum_{\chi\in\widehat G}
\overline{\chi(c)}\,B_n(\chi;u)
\sum_{h\in H}\overline{\chi(h)}.
\end{aligned}
\]
有限阿贝尔群角色在子群上的正交关系给出
\[
\sum_{h\in H}\overline{\chi(h)}
=
\begin{cases}
|H|,& \chi\in H^\perp,\\[2pt]
0,& \chi\notin H^\perp.
\end{cases}
\]
因此仅剩 \(\chi\in H^\perp\) 的项，遂得所示公式。证毕。
\end{proof}

\begin{theorem}[primitive 同余层析的最小承载商与最大陪集平坦子群]\label{thm:conclusion-primitive-minimal-carrier-quotient}
设 \(G\) 为有限阿贝尔群，并固定 \(n\ge 1\) 与 \(u\in(0,1]\)。定义活动角色支撑
\[
\Sigma_n(u):=\{\chi\in\widehat G:\ B_n(\chi;u)\neq 0\},
\]
以及其公共核
\[
H_n(u):=
\bigcap_{\chi\in\Sigma_n(u)}\ker\chi,
\]
并在 \(\Sigma_n(u)=\varnothing\) 时约定 \(H_n(u)=G\)。则函数
\[
c\longmapsto B_{n,c}(u)
\]
满足：
\begin{enumerate}
  \item 存在唯一函数
  \[
  \widetilde B_{n,u}:G/H_n(u)\to\CC
  \]
  使得
  \[
  \boxed{\;
  B_{n,c}(u)=\widetilde B_{n,u}\bigl(cH_n(u)\bigr)
  \qquad(\forall\,c\in G).\;}
  \]
  \item \(H_n(u)\) 是此性质下的最大子群：若 \(K\le G\) 且
  \[
  B_{n,ck}(u)=B_{n,c}(u)
  \qquad(\forall\,c\in G,\ \forall\,k\in K),
  \]
  则必有
  \[
  \boxed{\;
  K\subseteq H_n(u).\;}
  \]
  \item 因而 \(G/H_n(u)\) 是 primitive 同余层析的唯一最小承载商。等价地，在标准嵌入
  \[
  \widehat{G/H_n(u)}\hookrightarrow \widehat G
  \]
  下，\(\Sigma_n(u)\) 正是诱导函数 \(\widetilde B_{n,u}\) 的 Fourier 支撑。
\end{enumerate}
\end{theorem}
\begin{proof}
先证第 \(1\) 条。若 \(h\in H_n(u)\)，则对每个 \(\chi\in\Sigma_n(u)\) 都有 \(\chi(h)=1\)。由命题 \ref{prop:kernel-chebotarev-fourier}，
\[
B_{n,ch}(u)
=
\frac{1}{|G|}\sum_{\chi\in\Sigma_n(u)}
\overline{\chi(ch)}\,B_n(\chi;u)
=
\frac{1}{|G|}\sum_{\chi\in\Sigma_n(u)}
\overline{\chi(c)}\,\overline{\chi(h)}\,B_n(\chi;u)
=
B_{n,c}(u).
\]
故 \(B_{n,c}(u)\) 在 \(H_n(u)\)-陪集上常值，从而唯一因子化为某个
\[
\widetilde B_{n,u}:G/H_n(u)\to\CC.
\]

再证第 \(2\) 条。若 \(K\le G\) 使 \(B_{n,ck}(u)=B_{n,c}(u)\) 对所有 \(c\in G\)、\(k\in K\) 成立，则对任意 \(k\in K\) 有
\[
0
=
B_{n,ck}(u)-B_{n,c}(u)
=
\frac{1}{|G|}\sum_{\chi\in\Sigma_n(u)}
\overline{\chi(c)}\bigl(\overline{\chi(k)}-1\bigr)B_n(\chi;u).
\]
这是一条关于 \(c\) 的有限 Fourier 展开恒等式。由于角色系构成 \(\CC[G]\) 的正交基，展开唯一性迫使对每个 \(\chi\in\Sigma_n(u)\) 都有
\[
\bigl(\overline{\chi(k)}-1\bigr)B_n(\chi;u)=0.
\]
而 \(B_n(\chi;u)\neq 0\)，故 \(\chi(k)=1\)。于是 \(k\in\ker\chi\) 对所有 \(\chi\in\Sigma_n(u)\) 成立，即
\[
k\in H_n(u).
\]
因为 \(k\in K\) 任意，得到 \(K\subseteq H_n(u)\)。

第 \(3\) 条现在立即成立：任何使 \(c\mapsto B_{n,c}(u)\) 因子化的商 \(G/K\) 都满足 \(K\subseteq H_n(u)\)，故存在唯一满同态
\[
G/K\twoheadrightarrow G/H_n(u).
\]
因此 \(G/H_n(u)\) 是唯一最小承载商。又因为 \(G/H_n(u)\) 的角色恰对应于 \(\widehat G\) 中在 \(H_n(u)\) 上平凡的角色，而 \(\widetilde B_{n,u}\) 的非零 Fourier 系数正是那些满足 \(B_n(\chi;u)\neq 0\) 的角色，于是 \(\Sigma_n(u)\) 正好给出诱导函数的 Fourier 支撑。证毕。
\end{proof}
