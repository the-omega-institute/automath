\subsection{cyclotomic--Ramanujan--Nyquist 主线中的 divisor-compressed 轴、平方根门槛与外围相位层析}\label{subsec:conclusion-cyclotomic-ramanujan-nyquist-divisoraxis}
沿用第 \(13\) 章关于 primitive 扭点平均、Ramanujan 展开与连续 Parseval / 离散混叠恒等式的记号。设
\[
M\in \ZZ_{\ge 0}^{d\times d}\ \text{primitive},\qquad
\rho=\rho(M)>1,
\]
\[
\lambda_j=\mu_j/\rho\quad (j=2,\dots,d),\qquad
u_n:=\sum_{j=2}^d \lambda_j^n,\qquad
F_M(t):=\prod_{j=2}^d(1-\lambda_j t).
\]
对 \(m\ge 2\) 记文稿中的 cyclotomic 原子为 \(C_m(M)\)，并作自然补全
\[
\mathcal A_1(M):=-\log F_M(1),\qquad
\mathcal A_m(M):=C_m(M)\quad (m\ge 2).
\]
再定义 divisor-compressed 的 \(q\)-轴信号
\[
\mathfrak S_q(M):=\sum_{m\mid q}\mathcal A_m(M)\qquad (q\ge 1).
\]
在这一口径下，primitive 扭点原子、Ramanujan 投影、Nyquist 能量与 kernel-RH 的平方根门槛可被进一步压缩为同一条算术轴上的闭式信号理论。

\begin{theorem}[divisor-compressed \(q\)-轴的 Ramanujan 对角化]\label{thm:conclusion-divisor-compressed-qaxis-ramanujan-diagonalization}
对每个 \(q\ge 1\)，有精确恒等式
\[
\mathfrak S_q(M)=\sum_{\ell\ge 1}\frac{u_{\ell q}(M)}{\ell}.
\]
\end{theorem}
\begin{proof}
对 \(m\ge 2\)，文稿给出 Ramanujan 展开
\[
\mathcal A_m(M)=C_m(M)=\sum_{n\ge 1}\frac{u_n(M)}{n}\,c_m(n).
\]
而按定义，
\[
\mathcal A_1(M)=-\log F_M(1)=\sum_{n\ge 1}\frac{u_n(M)}{n},
\]
这正对应于 \(c_1(n)\equiv 1\) 的情形。于是
\[
\mathfrak S_q(M)
=\sum_{m\mid q}\sum_{n\ge 1}\frac{u_n(M)}{n}c_m(n)
=\sum_{n\ge 1}\frac{u_n(M)}{n}\sum_{m\mid q}c_m(n).
\]
再用经典恒等式
\[
\sum_{m\mid q}c_m(n)=q\,\mathbf 1_{q\mid n},
\]
便得到
\[
\mathfrak S_q(M)
=\sum_{q\mid n}\frac{q\,u_n(M)}{n}
=\sum_{\ell\ge 1}\frac{u_{\ell q}(M)}{\ell}.
\]
证毕。
\end{proof}

\begin{theorem}[Ramanujan 核的正半定性与原子线性型的导数 Parseval 控制]\label{thm:conclusion-ramanujan-kernel-psd-derivative-parseval-control}
设 \(\alpha=(\alpha_m)_{m\ge 1}\) 为有限支撑复数列，并定义
\[
L_\alpha(M):=\sum_{m\ge 1}\alpha_m \mathcal A_m(M),
\qquad
R_\alpha(n):=\sum_{m\ge 1}\alpha_m c_m(n).
\]
则有展开
\[
L_\alpha(M)=\sum_{n\ge 1}\frac{u_n(M)}{n}\,R_\alpha(n).
\]
进一步，定义导数谱能量
\[
\mathcal E_{\partial}(M):=\frac{1}{2\pi}\int_0^{2\pi}|g'(\theta)|^2\,d\theta,
\qquad
g(\theta):=-\log F_M(e^{i\theta}).
\]
则
\[
\mathcal E_{\partial}(M)=\sum_{n\ge 1}|u_n(M)|^2,
\]
并且成立最优型 Cauchy--Schwarz 上界
\[
|L_\alpha(M)|^2\le \mathcal E_{\partial}(M)\,\mathcal Q(\alpha),
\]
其中
\[
\mathcal Q(\alpha):=\sum_{n\ge 1}\frac{|R_\alpha(n)|^2}{n^2}
=\sum_{m,n\ge 1}\alpha_m\overline{\alpha_n}\,K_{m,n},
\]
且
\[
K_{m,n}:=\sum_{r\ge 1}\frac{c_m(r)c_n(r)}{r^2}
\]
定义了一个正半定核。
\end{theorem}
\begin{proof}
第一式由定理 \ref{thm:conclusion-divisor-compressed-qaxis-ramanujan-diagonalization} 对有限线性组合逐项展开即得：
\[
L_\alpha(M)
=\sum_{m\ge 1}\alpha_m\sum_{n\ge 1}\frac{u_n(M)}{n}c_m(n)
=\sum_{n\ge 1}\frac{u_n(M)}{n}R_\alpha(n).
\]

又由文稿给出的 Fourier 展开，
\[
g(\theta)=\sum_{n\ge 1}\frac{u_n(M)}{n}e^{in\theta}.
\]
由于 \(\max_{j\ge 2}|\lambda_j|<1\)，可逐项求导得到
\[
g'(\theta)=i\sum_{n\ge 1}u_n(M)e^{in\theta}.
\]
Parseval 恒等式遂给出
\[
\mathcal E_{\partial}(M)
=\frac{1}{2\pi}\int_0^{2\pi}|g'(\theta)|^2\,d\theta
=\sum_{n\ge 1}|u_n(M)|^2.
\]

再把 \(L_\alpha(M)\) 改写为
\[
L_\alpha(M)=\sum_{n\ge 1}u_n(M)\frac{R_\alpha(n)}{n},
\]
直接施以 Cauchy--Schwarz，即得
\[
|L_\alpha(M)|^2
\le
\left(\sum_{n\ge 1}|u_n(M)|^2\right)
\left(\sum_{n\ge 1}\frac{|R_\alpha(n)|^2}{n^2}\right)
=\mathcal E_{\partial}(M)\,\mathcal Q(\alpha).
\]

最后，
\[
\mathcal Q(\alpha)
=\sum_{r\ge 1}\frac{1}{r^2}\left|\sum_{m\ge 1}\alpha_m c_m(r)\right|^2
=\sum_{m,n\ge 1}\alpha_m\overline{\alpha_n}\sum_{r\ge 1}\frac{c_m(r)c_n(r)}{r^2},
\]
故核 \(K_{m,n}\) 的 Gram 型表示成立，从而 \(K\) 正半定。证毕。
\end{proof}

\begin{corollary}[\(q\)-轴累计原子信号的一致 Parseval 上界]\label{cor:conclusion-qaxis-cumulative-atomic-signal-uniform-parseval-bound}
对每个 \(q\ge 1\)，都有
\[
|\mathfrak S_q(M)|^2\le \frac{\pi^2}{6}\,\mathcal E_{\partial}(M),
\]
亦即
\[
\sup_{q\ge 1}|\mathfrak S_q(M)|
\le
\frac{\pi}{\sqrt 6}\,\mathcal E_{\partial}(M)^{1/2}.
\]
\end{corollary}
\begin{proof}
在定理 \ref{thm:conclusion-ramanujan-kernel-psd-derivative-parseval-control} 中取
\[
\alpha_m:=\mathbf 1_{m\mid q}.
\]
则
\[
R_\alpha(n)=\sum_{m\mid q}c_m(n)=q\,\mathbf 1_{q\mid n}.
\]
于是
\[
\mathcal Q(\alpha)
=\sum_{n\ge 1}\frac{q^2\mathbf 1_{q\mid n}}{n^2}
=\sum_{\ell\ge 1}\frac{q^2}{(\ell q)^2}
=\sum_{\ell\ge 1}\frac{1}{\ell^2}
=\frac{\pi^2}{6}.
\]
又 \(L_\alpha(M)=\mathfrak S_q(M)\)，代回定理 \ref{thm:conclusion-ramanujan-kernel-psd-derivative-parseval-control} 即得所示结论。证毕。
\end{proof}

\begin{theorem}[divisor-compressed \(q\)-轴的平方根门槛]\label{thm:conclusion-divisor-compressed-qaxis-squareroot-threshold}
\leanverified{paper\_conclusion\_divisor\_compressed\_qaxis\_squareroot\_threshold}
记
\[
\Lambda_{\mathrm{rel}}(M):=\max_{j\ge 2}|\lambda_j|<1.
\]
则有精确指数公式
\[
\limsup_{q\to\infty}|\mathfrak S_q(M)|^{1/q}
=
\Lambda_{\mathrm{rel}}(M).
\]
因此，kernel-RH 条件
\[
\max_{j\ge 2}|\mu_j|\le \rho^{1/2}
\]
等价于
\[
\limsup_{q\to\infty}|\mathfrak S_q(M)|^{1/q}\le \rho^{-1/2}.
\]
更进一步，在 kernel-RH 成立时，对一切 \(q\ge 1\) 都有显式上界
\[
|\mathfrak S_q(M)|
\le
(d-1)\frac{\rho^{-q/2}}{1-\rho^{-q/2}}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-divisor-compressed-qaxis-ramanujan-diagonalization}，
\[
\mathfrak S_q(M)=u_q(M)+\sum_{\ell\ge 2}\frac{u_{\ell q}(M)}{\ell}.
\]
因此
\[
\left|\sum_{\ell\ge 2}\frac{u_{\ell q}(M)}{\ell}\right|
\le
(d-1)\sum_{\ell\ge 2}\frac{\Lambda_{\mathrm{rel}}(M)^{\ell q}}{\ell}
=O\bigl(\Lambda_{\mathrm{rel}}(M)^{2q}\bigr).
\]
于是 \(\mathfrak S_q(M)\) 与 \(u_q(M)\) 具有相同的主指数。

把所有满足 \(|\lambda_j|=\Lambda_{\mathrm{rel}}(M)\) 的项写成
\[
\lambda_j=\Lambda_{\mathrm{rel}}(M)\xi_j,\qquad |\xi_j|=1.
\]
则存在 \(\Lambda'<\Lambda_{\mathrm{rel}}(M)\) 使
\[
\mathfrak S_q(M)
=
\Lambda_{\mathrm{rel}}(M)^q\sum_{j=1}^{J}\xi_j^q
+O\bigl((\Lambda')^q\bigr).
\]
由定理 \ref{thm:conclusion-peripheral-phase-multiplicity-cesaro-quadratic-law}，归一化序列
\[
\frac{\mathfrak S_q(M)}{\Lambda_{\mathrm{rel}}(M)^q}
\]
的平方模 Ces\`aro 平均收敛到正数
\(
\sum_{\zeta\in S^1}m(\zeta)^2\ge 1
\)；
故该归一化序列沿某无穷子列保持与零有正距离，从而
\[
\limsup_{q\to\infty}|\mathfrak S_q(M)|^{1/q}
\ge \Lambda_{\mathrm{rel}}(M).
\]
反过来，由上式的显式分解立刻有
\[
|\mathfrak S_q(M)|\le C\,\Lambda_{\mathrm{rel}}(M)^q
\]
对某常数 \(C>0\) 成立，因此
\[
\limsup_{q\to\infty}|\mathfrak S_q(M)|^{1/q}
\le \Lambda_{\mathrm{rel}}(M).
\]
两侧合并即得指数公式。

而
\[
\Lambda_{\mathrm{rel}}(M)=\frac{\max_{j\ge 2}|\mu_j|}{\rho},
\]
故
\[
\Lambda_{\mathrm{rel}}(M)\le \rho^{-1/2}
\iff
\max_{j\ge 2}|\mu_j|\le \rho^{1/2},
\]
这就得到 kernel-RH 的等价刻画。

若 kernel-RH 成立，则
\[
|u_{\ell q}(M)|\le (d-1)\rho^{-\ell q/2},
\]
因此
\[
|\mathfrak S_q(M)|
\le
(d-1)\sum_{\ell\ge 1}\frac{\rho^{-\ell q/2}}{\ell}
=(d-1)\bigl(-\log(1-\rho^{-q/2})\bigr).
\]
再用
\[
-\log(1-x)\le \frac{x}{1-x}\qquad (0<x<1)
\]
即可得到
\[
|\mathfrak S_q(M)|
\le
(d-1)\frac{\rho^{-q/2}}{1-\rho^{-q/2}}.
\]
证毕。
\end{proof}

\begin{theorem}[外围相位多重度的 Ces\`aro 二次律]\label{thm:conclusion-peripheral-phase-multiplicity-cesaro-quadratic-law}
设模长达到 \(\Lambda_{\mathrm{rel}}(M)\) 的归一化非 Perron 谱点为
\[
\{\Lambda_{\mathrm{rel}}(M)\xi_1,\dots,\Lambda_{\mathrm{rel}}(M)\xi_J\},
\qquad |\xi_j|=1.
\]
对每个 \(\zeta\in S^1\)，记其多重度
\[
m(\zeta):=\#\{j:\xi_j=\zeta\}.
\]
则成立精确极限
\[
\lim_{N\to\infty}\frac1N\sum_{q=1}^{N}
\left|
\frac{\mathfrak S_q(M)}{\Lambda_{\mathrm{rel}}(M)^q}
\right|^2
=
\sum_{\zeta\in S^1}m(\zeta)^2.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-divisor-compressed-qaxis-squareroot-threshold} 证明中的主项分解，存在 \(0\le \eta<1\) 使
\[
\frac{\mathfrak S_q(M)}{\Lambda_{\mathrm{rel}}(M)^q}
=
\sum_{j=1}^{J}\xi_j^q+O(\eta^q).
\]
因此只需计算
\[
\frac1N\sum_{q=1}^{N}\left|\sum_{j=1}^{J}\xi_j^q\right|^2
=
\frac1N\sum_{q=1}^{N}\sum_{i,j=1}^{J}(\xi_i\overline{\xi_j})^q.
\]
逐项取 Ces\`aro 极限。若 \(\xi_i=\xi_j\)，则相应项恒为 \(1\)；若 \(\xi_i\neq \xi_j\)，则 \(\xi_i\overline{\xi_j}\neq 1\)，而
\[
\frac1N\sum_{q=1}^{N}(\xi_i\overline{\xi_j})^q\longrightarrow 0.
\]
故极限恰为
\[
\#\{(i,j):\xi_i=\xi_j\}
=
\sum_{\zeta\in S^1}m(\zeta)^2.
\]
误差项 \(O(\eta^q)\) 在 Ces\`aro 平均下趋于 \(0\)，故结论成立。证毕。
\end{proof}

\begin{corollary}[有限周期外围相位的精确 Fourier 层析]\label{cor:conclusion-finite-periodic-peripheral-phase-fourier-tomography}
\leanverified{paper\_conclusion\_finite\_periodic\_peripheral\_phase\_fourier\_tomography}
若外围相位全都属于某个有限循环群 \(\mu_L\)，即
\[
\xi_j^L=1\qquad (j=1,\dots,J),
\]
并记
\[
m_r:=\#\{j:\xi_j=\omega_L^r\}\qquad (0\le r\le L-1),
\qquad
\omega_L:=e^{2\pi i/L},
\]
则存在精确的 \(L\)-周期主项
\[
s_q:=\sum_{r=0}^{L-1}m_r\,\omega_L^{rq},
\]
使得
\[
\frac{\mathfrak S_q(M)}{\Lambda_{\mathrm{rel}}(M)^q}=s_q+O(\eta^q)
\]
对某个 \(0\le \eta<1\) 成立。从而对每个 \(0\le r\le L-1\)，有恢复公式
\[
m_r=\frac1L\sum_{q=0}^{L-1}s_q\,\omega_L^{-rq}.
\]
等价地，
\[
m_r
=
\frac1L\sum_{q=0}^{L-1}
\left(
\lim_{k\to\infty}\frac{\mathfrak S_{q+kL}(M)}{\Lambda_{\mathrm{rel}}(M)^{q+kL}}
\right)\omega_L^{-rq}.
\]
\end{corollary}
\begin{proof}
在 \(\xi_j\in \mu_L\) 的假设下，
\[
\sum_{j=1}^{J}\xi_j^q
=
\sum_{r=0}^{L-1}m_r\omega_L^{rq}
=:s_q,
\]
这是一个严格 \(L\)-周期函数。由定理 \ref{thm:conclusion-divisor-compressed-qaxis-squareroot-threshold} 证明中的分解，
\[
\frac{\mathfrak S_q(M)}{\Lambda_{\mathrm{rel}}(M)^q}
=s_q+O(\eta^q).
\]
故对每个固定的同余类 \(q\bmod L\)，令 \(k\to\infty\) 即得
\[
\lim_{k\to\infty}\frac{\mathfrak S_{q+kL}(M)}{\Lambda_{\mathrm{rel}}(M)^{q+kL}}
=s_q.
\]
最后对 \(L\)-周期序列 \((s_q)\) 作离散 Fourier 反演，
\[
m_r=\frac1L\sum_{q=0}^{L-1}s_q\,\omega_L^{-rq},
\]
便得到 multiplicity 向量 \((m_r)\) 的恢复公式。证毕。
\end{proof}

\begin{theorem}[平方根阈值的 \(q\)-轴能量三分律]\label{thm:conclusion-qaxis-energy-trichotomy-sqrt-threshold}
设 \(M\in \mathbf Z_{\ge 0}^{d\times d}\) primitive，\(\rho:=\rho(M)>1\)，并记
\[
\Lambda:=\Lambda_{\mathrm{rel}}(M):=\max_{j\ge 2}|\lambda_j|,
\qquad
E_N(M):=\frac1N\sum_{q=1}^N \rho^{q}\,|\Psi_q(M)|^2.
\]
则存在严格三分：
\[
\Lambda<\rho^{-1/2}\ \Longrightarrow\ E_N(M)\to 0,
\]
\[
\Lambda=\rho^{-1/2}\ \Longrightarrow\ E_N(M)\to \sum_{\zeta\in\mathbf T} m(\zeta)^2,
\]
\[
\Lambda>\rho^{-1/2}\ \Longrightarrow\ \limsup_{N\to\infty}E_N(M)=+\infty,
\]
其中 \(m(\zeta)\) 为外谱壳层
\[
E_\Lambda:=\{\lambda_j:|\lambda_j|=\Lambda\}
\]
在相位 \(\zeta\) 处的重数。
\end{theorem}
\begin{proof}
当 \(\Lambda<\rho^{-1/2}\) 时，主稿关于常数层 \(q\)-轴的外谱展开给出
\[
\Psi_q(M)=\Lambda^q\sum_{k=1}^r \eta_k^q+O(\Lambda_0^q),
\qquad
0\le \Lambda_0<\Lambda,
\]
故存在常数 \(C>0\) 使
\[
|\Psi_q(M)|\le C\Lambda^q.
\]
于是
\[
\rho^q|\Psi_q(M)|^2\le C^2(\rho\Lambda^2)^q,
\]
右端指数可和，故其 Ces\`aro 平均 \(E_N(M)\) 趋于 \(0\)。

当 \(\Lambda=\rho^{-1/2}\) 时，主稿的 \(q\)-轴 Ces\`aro 能量定理直接给出
\[
\frac1N\sum_{q=1}^N\frac{|\Psi_q(M)|^2}{\Lambda^{2q}}
\longrightarrow
\sum_{\zeta\in\mathbf T} m(\zeta)^2.
\]
由于 \(\Lambda^{-2}=\rho\)，这正是
\[
E_N(M)\longrightarrow \sum_{\zeta\in\mathbf T} m(\zeta)^2.
\]

当 \(\Lambda>\rho^{-1/2}\) 时，主稿关于 \(q\)-轴的下界推论断言：存在常数 \(c>0\) 与无穷多 \(q\) 使
\[
|\Psi_q(M)|\ge c\,\Lambda^q.
\]
对这些 \(q\) 取 \(N=q\)，便有
\[
E_q(M)
\ge
\frac1q\,\rho^q|\Psi_q(M)|^2
\ge
\frac{c^2}{q}(\rho\Lambda^2)^q.
\]
因 \(\rho\Lambda^2>1\)，右端趋于 \(+\infty\)，故
\[
\limsup_{N\to\infty}E_N(M)=+\infty.
\]
证毕。
\end{proof}

\begin{corollary}[外谱壳层的单通道层析]\label{cor:conclusion-outer-spectral-shell-single-channel-tomography}
在定理 \ref{thm:conclusion-qaxis-energy-trichotomy-sqrt-threshold} 的设定下，单一标量序列 \((\Psi_q(M))_{q\ge1}\) 已足以层析外谱壳层。具体地，
\[
\Lambda=\limsup_{q\to\infty} |\Psi_q(M)|^{1/q},
\]
并且
\[
\lim_{N\to\infty}\frac1N\sum_{q=1}^N\frac{|\Psi_q(M)|^2}{\Lambda^{2q}}
=
\sum_{\zeta\in\mathbf T}m(\zeta)^2 .
\]
若外谱壳层满足一般位置条件
\[
\eta_i\eta_j^{-1}\notin \mu_\infty \qquad (i\ne j),
\]
则右端退化为壳层总重数 \(r=|E_\Lambda|\)。换言之，在一般位置下，\((\Psi_q(M))\) 单独恢复 \((\Lambda,r)\)。
\end{corollary}
\begin{proof}
第一式即主稿中常数层 \(q\)-轴指数率恢复定理的内容。第二式由定理 \ref{thm:conclusion-qaxis-energy-trichotomy-sqrt-threshold} 在临界归一化下的 Ces\`aro 极限得到。

若外谱壳层满足一般位置条件，则各外围相位簇均为单重簇，故
\[
\sum_{\zeta\in\mathbf T}m(\zeta)^2
=
\sum_{\zeta\in\mathbf T}m(\zeta)
=
r.
\]
于是 \((\Psi_q(M))\) 单独恢复 \(\Lambda\) 与外谱壳层总重数 \(r\)。证毕。
\end{proof}
