\subsection{共动 Jensen 缺陷的熵--能量分解、Poisson--KL 桥接与 RH 证书链}\label{subsec:conclusion-comoving-jensen-entropy-energy-poisson-kl-rh}
沿用
\eqref{eq:xi-comoving-shifted-Fzeta}--\eqref{eq:xi-comoving-shifted-jensen-defect}
中的 \(F_{\zeta,x_0}\)、\(D_{\zeta,x_0}\) 记号，以及定理 \ref{thm:xi-rh-iff-uniform-comoving-defect} 的共动缺陷等价判据。
在涉及 Poisson 粗粒化时，沿用定理 \ref{thm:xi-poisson-cauchy-mixture-kl-t4-minimal-third} 的四阶 KL 标度工具。

\begin{theorem}[Jensen 缺陷的能量--反向相对熵精确分解]\label{thm:conclusion-jensen-defect-energy-reversekl-identity}
设 \(F\) 在单位圆盘 \(\mathbb D\) 内解析且 \(F(0)\neq 0\)。
记归一化圆周测度
\[
m(\dd\theta):=\frac{\dd\theta}{2\pi}.
\]
对 \(0<r<1\) 定义
\[
E_2(r):=\int_0^{2\pi}\bigl|F(re^{\mathrm{i}\theta})\bigr|^2\,m(\dd\theta),
\qquad
p_r(\theta):=\frac{|F(re^{\mathrm{i}\theta})|^2}{E_2(r)}.
\]
则 Jensen 缺陷
\[
D_F(r):=\int_0^{2\pi}\log|F(re^{\mathrm{i}\theta})|\,m(\dd\theta)-\log|F(0)|
\]
满足恒等式
\begin{equation}\label{eq:conclusion-jensen-defect-energy-reversekl}
\boxed{\;
D_F(r)
=\frac12\log\!\Bigl(\frac{E_2(r)}{|F(0)|^2}\Bigr)
-\frac12\,D_{\mathrm{KL}}\!\bigl(m\mid p_r\,m\bigr).\;}
\end{equation}
特别地，
\begin{equation}\label{eq:conclusion-jensen-defect-energy-upperbound}
D_F(r)\le \frac12\log\!\Bigl(\frac{E_2(r)}{|F(0)|^2}\Bigr).
\end{equation}
\end{theorem}
\begin{proof}
由定义
\[
D_{\mathrm{KL}}(m\mid p_r m)=\int_0^{2\pi}-\log p_r(\theta)\,m(\dd\theta).
\]
而
\[
\log p_r(\theta)=\log|F(re^{\mathrm{i}\theta})|^2-\log E_2(r),
\]
故
\[
D_{\mathrm{KL}}(m\mid p_r m)
=-2\int_0^{2\pi}\log|F(re^{\mathrm{i}\theta})|\,m(\dd\theta)+\log E_2(r).
\]
移项得
\[
\int_0^{2\pi}\log|F(re^{\mathrm{i}\theta})|\,m(\dd\theta)
=\frac12\log E_2(r)-\frac12D_{\mathrm{KL}}(m\mid p_r m).
\]
再减去 \(\log|F(0)|\) 即得 \eqref{eq:conclusion-jensen-defect-energy-reversekl}。
由 \(D_{\mathrm{KL}}\ge 0\) 立即得到 \eqref{eq:conclusion-jensen-defect-energy-upperbound}。证毕。
\end{proof}

\begin{theorem}[零点库存的能量--反熵精确恒等式]\label{thm:conclusion-jensen-defect-zero-stock-identity}
在定理 \ref{thm:conclusion-jensen-defect-energy-reversekl-identity} 设定下，记 \(F\) 在 \(\{|z|<r\}\) 内全部零点（按重数）为 \(\{a_k\}_{|a_k|<r}\)。
定义
\[
\mathcal G_F(r):=\log\!\Bigl(\frac{E_2(r)}{|F(0)|^2}\Bigr)-D_{\mathrm{KL}}(m\mid p_r m).
\]
则有
\begin{equation}\label{eq:conclusion-jensen-defect-zero-stock}
\boxed{\;
\mathcal G_F(r)
=2D_F(r)
=2\sum_{|a_k|<r}\log\!\Bigl(\frac{r}{|a_k|}\Bigr).\;}
\end{equation}
因此 \(\mathcal G_F(r)\ge 0\)，且每个盘内零点贡献严格正的可加项。
\end{theorem}
\begin{proof}
由 \eqref{eq:conclusion-jensen-defect-energy-reversekl} 直接得
\[
\mathcal G_F(r)=2D_F(r).
\]
再由 Jensen 公式
\[
D_F(r)=\sum_{|a_k|<r}\log\!\Bigl(\frac{r}{|a_k|}\Bigr)
\]
即得 \eqref{eq:conclusion-jensen-defect-zero-stock}。证毕。
\end{proof}

\begin{corollary}[反熵饱和的零点排斥判据与定量半径]\label{cor:conclusion-jensen-defect-entropy-saturation-exclusion}
沿用定理 \ref{thm:conclusion-jensen-defect-zero-stock-identity} 记号。
则：
\begin{enumerate}
  \item \(F\) 在 \(|z|<r\) 内无零点，当且仅当
  \begin{equation}\label{eq:conclusion-jensen-defect-entropy-saturation}
  \boxed{\;
  D_{\mathrm{KL}}(m\mid p_r m)=\log\!\Bigl(\frac{E_2(r)}{|F(0)|^2}\Bigr).\;}
  \end{equation}
  \item 若某个 \(\varepsilon\ge 0\) 满足
  \begin{equation}\label{eq:conclusion-jensen-defect-gap-leq-2eps}
  \log\!\Bigl(\frac{E_2(r)}{|F(0)|^2}\Bigr)-D_{\mathrm{KL}}(m\mid p_r m)\le 2\varepsilon,
  \end{equation}
  则 \(F\) 在 \(|z|<re^{-\varepsilon}\) 内无零点。
\end{enumerate}
\end{corollary}
\begin{proof}
由 \eqref{eq:conclusion-jensen-defect-zero-stock}，
\(
\mathcal G_F(r)=0
\)
当且仅当
\(
\sum_{|a_k|<r}\log(r/|a_k|)=0
\)，等价于盘内无零点，得 (1)。

由 \eqref{eq:conclusion-jensen-defect-gap-leq-2eps} 与
\eqref{eq:conclusion-jensen-defect-zero-stock} 可得 \(D_F(r)\le \varepsilon\)。
若存在零点 \(a\) 满足 \(|a|<re^{-\varepsilon}\)，则
\[
\log\!\Bigl(\frac{r}{|a|}\Bigr)>\varepsilon
\]
并由 Jensen 和式得 \(D_F(r)\ge \log(r/|a|)>\varepsilon\)，矛盾。证毕。
\end{proof}

\begin{proposition}[Poisson 卷积 Jensen 间隙的纤维反向 KL 表示]\label{prop:conclusion-poisson-jensen-gap-reversekl}
设 \(f:\RR\to(0,\infty)\) 可测，\(t>0\)，并记
\[
P_t(x):=\frac{t}{\pi(x^2+t^2)}.
\]
假设 \((P_t*f)(x)\) 与 \((P_t*\log f)(x)\) 在所论 \(x\) 处有限。
定义概率测度
\[
\nu_{x,t}(\dd y):=P_t(x-y)\,\dd y,
\qquad
\nu_{x,t}^{f}(\dd y):=\frac{f(y)}{(P_t*f)(x)}\,\nu_{x,t}(\dd y).
\]
则有点态恒等式
\begin{equation}\label{eq:conclusion-poisson-jensen-gap-reversekl}
\boxed{\;
\log(P_t*f)(x)-(P_t*\log f)(x)
=D_{\mathrm{KL}}(\nu_{x,t}\mid \nu_{x,t}^{f})\ge 0.\;}
\end{equation}
\end{proposition}
\begin{proof}
由定义
\[
\frac{\dd\nu_{x,t}}{\dd\nu_{x,t}^{f}}(y)=\frac{(P_t*f)(x)}{f(y)}.
\]
故
\[
D_{\mathrm{KL}}(\nu_{x,t}\mid \nu_{x,t}^{f})
=\int \log\!\Bigl(\frac{(P_t*f)(x)}{f(y)}\Bigr)\,\nu_{x,t}(\dd y)
\]
\[
=\log(P_t*f)(x)-\int \log f(y)\,\nu_{x,t}(\dd y)
=\log(P_t*f)(x)-(P_t*\log f)(x).
\]
非负性由相对熵基本性质给出。证毕。
\end{proof}

\begin{proposition}[Cayley 模长的 Poisson 原函数表示]\label{prop:conclusion-cayley-modulus-poisson-primitive}
令 \(u\in\RR\)、\(\delta>0\)，并置 \(z=u+i\delta\in\mathbb H\)。
定义 Cayley 变换
\[
C:\mathbb H\to\mathbb D,\qquad C(z)=\frac{1+iz}{1-iz},
\]
以及 Poisson--Cauchy 核
\[
P_s(u):=\frac{s}{\pi(u^2+s^2)}\qquad(s\in\RR).
\]
则成立精确恒等式
\begin{equation}\label{eq:conclusion-cayley-modulus-poisson-primitive}
\boxed{\;
-\log|C(u+i\delta)|
=\pi\int_{1-\delta}^{1+\delta}P_s(u)\,\dd s
=\frac12\log\frac{u^2+(1+\delta)^2}{u^2+(1-\delta)^2}.\;}
\end{equation}
其中当 \(1-\delta<0\) 时按有向积分理解（因 \(s/(u^2+s^2)\) 在 \(s=0\) 连续）。
\end{proposition}
\begin{proof}
直接计算得到
\[
|C(u+i\delta)|^2=\frac{u^2+(1-\delta)^2}{u^2+(1+\delta)^2},
\]
从而
\[
-\log|C(u+i\delta)|
=\frac12\log\frac{u^2+(1+\delta)^2}{u^2+(1-\delta)^2}.
\]
另一方面，
\[
\frac{\dd}{\dd s}\log(u^2+s^2)=\frac{2s}{u^2+s^2}=2\pi P_s(u),
\]
对 \(s\in[1-\delta,1+\delta]\) 积分即得
\[
\pi\int_{1-\delta}^{1+\delta}P_s(u)\,\dd s
=\frac12\log\frac{u^2+(1+\delta)^2}{u^2+(1-\delta)^2}.
\]
合并即得 \eqref{eq:conclusion-cayley-modulus-poisson-primitive}。证毕。
\end{proof}

\begin{corollary}[有限 Cayley 零点的 Jensen 缺陷尺度切片公式]\label{cor:conclusion-jensen-defect-poisson-slice-finite}
在定理 \ref{thm:conclusion-jensen-defect-zero-stock-identity} 的设定下，进一步假设 \(F\) 在 \(\mathbb D\) 内仅有有限个零点 \(\{a_k\}\)（按重数 \(m_k\)），并且每个零点可写为
\[
a_k=C(u_k+i\delta_k),\qquad u_k\in\RR,\ \delta_k>0.
\]
则
\[
\lim_{r\uparrow 1}D_F(r)
=\sum_k m_k\bigl(-\log|a_k|\bigr)
=\pi\sum_k m_k\int_{1-\delta_k}^{1+\delta_k}P_s(u_k)\,\dd s.
\]
\end{corollary}
\begin{proof}
由 Jensen 公式，
\[
D_F(r)=\sum_{|a_k|<r}m_k\log\frac{r}{|a_k|}.
\]
零点有限时，当 \(r\uparrow 1\) 右端最终包含全部零点，且 \(\log r\to 0\)，故
\[
\lim_{r\uparrow 1}D_F(r)=\sum_k m_k\bigl(-\log|a_k|\bigr).
\]
再由 \eqref{eq:conclusion-cayley-modulus-poisson-primitive} 对每个 \(a_k=C(u_k+i\delta_k)\) 逐项代入即得尺度切片表示。证毕。
\end{proof}

\begin{theorem}[RH 的熵--能量饱和一致判据]\label{thm:conclusion-rh-entropy-energy-saturation-criterion}
对每个 \((x_0,\varrho)\)（\(x_0\in\RR,\ 0<\varrho<1\)）定义
\[
E_{\zeta,x_0}(\varrho):=
\frac{\int_0^{2\pi}|F_{\zeta,x_0}(\varrho e^{\mathrm{i}\theta})|^2\,m(\dd\theta)}
{|F_{\zeta,x_0}(0)|^2},
\]
\[
p_{\zeta,x_0,\varrho}(\theta):=
\frac{|F_{\zeta,x_0}(\varrho e^{\mathrm{i}\theta})|^2}
{\int_0^{2\pi}|F_{\zeta,x_0}(\varrho e^{\mathrm{i}\phi})|^2\,m(\dd\phi)}.
\]
则下述命题等价：
\begin{equation}\label{eq:conclusion-rh-entropy-energy-criterion}
\mathrm{RH}
\Longleftrightarrow
\exists\,\varrho_n\uparrow 1\ \text{s.t.}\ 
\sup_{x_0\in\RR}
\Bigl[
\log E_{\zeta,x_0}(\varrho_n)
-D_{\mathrm{KL}}(m\mid p_{\zeta,x_0,\varrho_n}m)
\Bigr]\xrightarrow[n\to\infty]{}0.
\end{equation}
\end{theorem}
\begin{proof}
把定理 \ref{thm:conclusion-jensen-defect-energy-reversekl-identity} 应用于 \(F_{\zeta,x_0}\)，得到
\[
D_{\zeta,x_0}(\varrho)
=\frac12\log E_{\zeta,x_0}(\varrho)
-\frac12D_{\mathrm{KL}}(m\mid p_{\zeta,x_0,\varrho}m).
\]
因此
\[
\sup_{x_0}D_{\zeta,x_0}(\varrho_n)\le \varepsilon_n
\Longleftrightarrow
\sup_{x_0}\!\Bigl[\log E_{\zeta,x_0}(\varrho_n)-D_{\mathrm{KL}}(m\mid p_{\zeta,x_0,\varrho_n}m)\Bigr]\le 2\varepsilon_n.
\]
再与定理 \ref{thm:xi-rh-iff-uniform-comoving-defect} 的等价判据对照即得 \eqref{eq:conclusion-rh-entropy-energy-criterion}。证毕。
\end{proof}

\begin{corollary}[一致 \(L^2\) 稳定推出 RH]\label{cor:conclusion-rh-from-uniform-l2-stability}
若存在 \(\varrho_n\uparrow 1\) 使
\begin{equation}\label{eq:conclusion-uniform-l2-stability}
\sup_{x_0\in\RR}\log E_{\zeta,x_0}(\varrho_n)\xrightarrow[n\to\infty]{}0,
\end{equation}
则 RH 成立。
\end{corollary}
\begin{proof}
由 \eqref{eq:conclusion-jensen-defect-energy-upperbound} 对 \(F_{\zeta,x_0}\) 的应用，
\[
D_{\zeta,x_0}(\varrho)\le \frac12\log E_{\zeta,x_0}(\varrho).
\]
对 \(x_0\) 取上确界并代入 \eqref{eq:conclusion-uniform-l2-stability}，得
\(
\sup_{x_0}D_{\zeta,x_0}(\varrho_n)\to 0
\)。
由定理 \ref{thm:xi-rh-iff-uniform-comoving-defect} 得 RH。证毕。
\end{proof}

\begin{theorem}[调和发散度与 Rényi/\(\chi^2\) 的精确关系]\label{thm:conclusion-harmonic-dispersion-renyi-chi2}
\leanverified{paper\_conclusion\_harmonic\_dispersion\_renyi\_chi2}
设 \(X\) 为有限集，\(w:X\to(0,1)\) 为处处正的概率分布，\(U\) 为 \(X\) 上均匀分布。
定义
\[
\mathfrak D(w):=|X|^{-2}\sum_{x\in X}\frac{1}{w(x)}.
\]
则：
\begin{enumerate}
  \item （Rényi-2 精确等式）
  \begin{equation}\label{eq:conclusion-harmonic-dispersion-renyi2}
  \boxed{\;D_2(U\|w)=\log \mathfrak D(w).\;}
  \end{equation}
  \item （\([1,2]\) 阶 Rényi 一致上界）对任意 \(\alpha\in[1,2]\)，
  \begin{equation}\label{eq:conclusion-harmonic-dispersion-renyi-alpha}
  \boxed{\;D_\alpha(U\|w)\le \log \mathfrak D(w).\;}
  \end{equation}
  特别地，\(\alpha\downarrow 1\) 时恢复
  \(
  D_{\mathrm{KL}}(U\|w)\le \log\mathfrak D(w)
  \)
  （参见命题 \ref{prop:pom-reverse-kl-bound-by-dispersion}）。
  \item （\(\chi^2\) 精确辨识）
  \begin{equation}\label{eq:conclusion-harmonic-dispersion-chi2}
  \boxed{\;\chi^2(U\|w)=\mathfrak D(w)-1.\;}
  \end{equation}
\end{enumerate}
\end{theorem}
\begin{proof}
由 Rényi-2 定义
\[
D_2(P\|Q)=\log\sum_x\frac{P(x)^2}{Q(x)}.
\]
取 \(P=U,\ Q=w\)，得到
\[
D_2(U\|w)=\log\sum_x\frac{|X|^{-2}}{w(x)}=\log\mathfrak D(w),
\]
即 \eqref{eq:conclusion-harmonic-dispersion-renyi2}。

Rényi 散度随阶数单调不减，故对 \(\alpha\in[1,2]\),
\[
D_\alpha(U\|w)\le D_2(U\|w)=\log\mathfrak D(w),
\]
即 \eqref{eq:conclusion-harmonic-dispersion-renyi-alpha}。

最后由
\[
\chi^2(P\|Q)=\sum_x\frac{P(x)^2}{Q(x)}-1
\]
取 \(P=U,\ Q=w\) 得
\[
\chi^2(U\|w)=\sum_x\frac{|X|^{-2}}{w(x)}-1=\mathfrak D(w)-1.
\]
证毕。
\end{proof}

\begin{conjecture}[共动缺陷与 Poisson--KL 的桥接函子]\label{conj:conclusion-comoving-defect-poisson-kl-bridge}
存在一个由 \((x_0,\varrho)\) 诱导的概率密度函子族
\[
\mathscr H:\ (x_0,\varrho)\mapsto h^{(x_0,\varrho)},
\]
及尺度换元 \(t=t(\varrho)\to\infty\)（\(\varrho\uparrow 1\)），使得对某个绝对常数 \(C>0\) 有一致比较
\begin{equation}\label{eq:conclusion-comoving-defect-poisson-kl-bridge}
\boxed{\;
D_{\zeta,x_0}(\varrho)
\le
C\,D_{\mathrm{KL}}\!\bigl(h_{t}^{(x_0,\varrho)}\mid g_t^{(x_0,\varrho)}\bigr)
+o_{\varrho\uparrow 1}(1),\quad \text{在 }x_0\text{ 上一致},\;}
\end{equation}
其中
\(
h_t^{(x_0,\varrho)}:=P_t*h^{(x_0,\varrho)}
\)，
\(g_t^{(x_0,\varrho)}\) 为同质量与同重心匹配的 Cauchy 基准。
\end{conjecture}

\begin{theorem}[桥接接口与四阶 KL 标度蕴含 RH 及速率]\label{thm:conclusion-bridge-plus-t4-implies-rh-rate}
假设猜想 \ref{conj:conclusion-comoving-defect-poisson-kl-bridge} 成立，并且桥接族满足如下统一矩条件：存在常数 \(V_2,V_6<\infty\)，使对所有参数 \((x_0,\varrho)\) 有
\[
\int y\,h^{(x_0,\varrho)}(y)\,\dd y=0,\qquad
\int y^2 h^{(x_0,\varrho)}(y)\,\dd y\le V_2,\qquad
\int |y|^6 h^{(x_0,\varrho)}(y)\,\dd y\le V_6.
\]
则存在 \(\varrho_n\uparrow 1\) 使
\begin{equation}\label{eq:conclusion-bridge-plus-t4-rate}
\sup_{x_0\in\RR}D_{\zeta,x_0}(\varrho_n)=O\!\bigl(t(\varrho_n)^{-4}\bigr),
\end{equation}
从而 RH 成立。
\end{theorem}
\begin{proof}
在统一矩条件下，对每个固定参数族分量可应用定理
\ref{thm:xi-poisson-cauchy-mixture-kl-t4-minimal-third}，得到一致估计
\[
D_{\mathrm{KL}}\!\bigl(h_t^{(x_0,\varrho)}\mid g_t^{(x_0,\varrho)}\bigr)
=O(t^{-4})
\]
（常数仅依赖 \(V_2,V_6\)）。
代入桥接不等式 \eqref{eq:conclusion-comoving-defect-poisson-kl-bridge} 得
\[
\sup_{x_0}D_{\zeta,x_0}(\varrho)=O(t(\varrho)^{-4})+o(1).
\]
取任意 \(\varrho_n\uparrow 1\) 使 \(t(\varrho_n)\to\infty\)，即得
\eqref{eq:conclusion-bridge-plus-t4-rate}，从而
\(
\sup_{x_0}D_{\zeta,x_0}(\varrho_n)\to 0
\)。
由定理 \ref{thm:xi-rh-iff-uniform-comoving-defect} 推出 RH。证毕。
\end{proof}

\begin{remark}[结构性缺口的精确位置]\label{rem:conclusion-comoving-defect-poisson-kl-gap}
上述链条中，决定性缺口是桥接函子 \(\mathscr H\) 的可构造性与一致误差控制，即把
\(D_{\zeta,x_0}(\varrho)\)
严格转译为可由 Poisson--KL 四阶主项驱动的可审计量。
一旦该接口闭合，RH 判据可由统一速率证书直接触发。
\end{remark}
