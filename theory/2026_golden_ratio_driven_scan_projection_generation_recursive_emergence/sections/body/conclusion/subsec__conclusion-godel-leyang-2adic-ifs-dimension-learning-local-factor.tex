\begin{theorem}[强分离 Gödel--Lee--Yang 仿射 IFS 的同胚编码与精确维数]\label{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension}
设
\[
T:=T_2(E_{\min})\cong \ZZ_2^2,\qquad L\ge 1,\qquad B:=2^L,
\]
并取非空有限集 \(D\subset T\) 满足强分离同余条件
\begin{equation}\label{eq:conclusion-godel-leyang-ssc-modBT}
d\not\equiv d'\pmod{BT}\qquad(d\neq d',\ d,d'\in D).
\end{equation}
定义
\begin{equation}\label{eq:conclusion-godel-leyang-kd-def}
K_D:=
\left\{
\sum_{n\ge 0}B^n d_n:\ d_n\in D
\right\}\subset T,
\end{equation}
地址空间 \(\Sigma_D:=D^{\NN}\)，及编码映射
\begin{equation}\label{eq:conclusion-godel-leyang-pi-def}
\pi:\Sigma_D\to T,\qquad
\pi((d_n)_{n\ge 0})=\sum_{n\ge 0}B^n d_n.
\end{equation}
则有：
\begin{enumerate}
  \item \(K_D\) 是 Hutchinson 方程
  \begin{equation}\label{eq:conclusion-godel-leyang-hutchinson}
  K_D=\bigcup_{d\in D}(d+B K_D)
  \end{equation}
  的唯一非空紧解；
  \item \(\pi:\Sigma_D\to K_D\) 为同胚，故每个 \(x\in K_D\) 具有唯一地址；
  \item \(K_D\) 的 Hausdorff 维数满足
  \begin{equation}\label{eq:conclusion-godel-leyang-dimh-exact}
  \dim_H(K_D)=\frac{\log_2|D|}{L};
  \end{equation}
  \item 若 \(\mu_D\) 为 \(\Sigma_D\) 上等权 Bernoulli 测度经 \(\pi\) 的推前，则对任意 \(x\in K_D\) 与任意 \(n\ge 0\),
  \begin{equation}\label{eq:conclusion-godel-leyang-ball-mass-exact}
  \mu_D\!\bigl(B(x,2^{-nL})\bigr)=|D|^{-n}
  =2^{-nL\dim_H(K_D)}.
  \end{equation}
\end{enumerate}
\leanverified{paper\_conclusion\_godel\_leyang\_ifs\_dimension\_seeds}
\leanverified{paper\_conclusion\_godel\_leyang\_ifs\_dimension\_package}
\end{theorem}
\begin{proof}
在 \(T\) 上取超度量 \(\|x\|_2:=\max(|x_1|_2,|x_2|_2)\)。

\textbf{(1)} 对每个 \(d\in D\)，仿射映射 \(\tau_d(x):=d+Bx\) 的 Lipschitz 常数为 \(|B|_2=2^{-L}<1\)。
故 \(\{\tau_d\}_{d\in D}\) 构成严格压缩 IFS，紧集空间上的 Banach 不动点原理给出唯一非空紧吸引子，且其级数表示正是 \eqref{eq:conclusion-godel-leyang-kd-def}。

\textbf{(2)} 由定义，\(\pi(\Sigma_D)=K_D\)。
设
\[
\sum_{n\ge 0}B^n d_n=\sum_{n\ge 0}B^n d_n'.
\]
模 \(BT\) 约化得 \(d_0\equiv d_0'\pmod{BT}\)，由 \eqref{eq:conclusion-godel-leyang-ssc-modBT} 得 \(d_0=d_0'\)。
两边相减并提取 \(B\) 后，利用乘 \(B\) 在 \(T\) 上的单射性得到
\[
\sum_{n\ge 1}B^{n-1}d_n=\sum_{n\ge 1}B^{n-1}d_n'.
\]
递归即得 \(d_n=d_n'\)（\(\forall n\)），故 \(\pi\) 单射。
又 \(\Sigma_D\) 紧、\(K_D\subset T\) Hausdorff，连续双射 \(\pi\) 必为同胚。

\textbf{(3)} 由 \eqref{eq:conclusion-godel-leyang-hutchinson} 与强分离，长度 \(n\) 的圆柱像对应 \( |D|^n\) 个两两不交的闭球，半径均为 \(2^{-nL}\)。
故尺度 \(2^{-nL}\) 的最小覆盖数满足
\[
N(2^{-nL};K_D)=|D|^n.
\]
于是
\[
\dim_H(K_D)
=\lim_{n\to\infty}
\frac{\log N(2^{-nL};K_D)}{-\log(2^{-nL})}
=\frac{\log_2|D|}{L}.
\]

\textbf{(4)} 等权 Bernoulli 测度使每个长度 \(n\) 圆柱质量为 \(|D|^{-n}\)。
由 (2)--(3)，\(K_D\) 中半径 \(2^{-nL}\) 的球恰对应唯一长度 \(n\) 圆柱像，故
\[
\mu_D(B(x,2^{-nL}))=|D|^{-n}.
\]
再用 \eqref{eq:conclusion-godel-leyang-dimh-exact} 即得 \eqref{eq:conclusion-godel-leyang-ball-mass-exact}。证毕。
\end{proof}

\begin{proposition}[前缀一致与 \(B^n\)-同余的等价刻画]\label{prop:conclusion-godel-leyang-prefix-congruence-equivalence}
在定理 \ref{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension} 的设定下，令
\[
x=\sum_{j\ge 0}B^j d_j,\qquad
y=\sum_{j\ge 0}B^j e_j
\qquad (d_j,e_j\in D)
\]
为 \(K_D\) 中两点的唯一地址展开。
则对任意 \(n\ge 1\)，下列两项等价：
\[
d_0=e_0,\ \dots,\ d_{n-1}=e_{n-1}
\quad\Longleftrightarrow\quad
x\equiv y\pmod{B^nT}.
\]
\leanverified{paper\_conclusion\_godel\_leyang\_prefix\_congruence\_equivalence}
\end{proposition}
\begin{proof}
若前 \(n\) 位一致，则
\[
x-y=\sum_{j\ge n}B^j(d_j-e_j)\in B^nT,
\]
故 \(x\equiv y\pmod{B^nT}\)。

反之，若 \(x\equiv y\pmod{B^nT}\)，则模 \(BT\) 约化得 \(d_0\equiv e_0\pmod{BT}\)。
由强分离条件 \eqref{eq:conclusion-godel-leyang-ssc-modBT}，必有 \(d_0=e_0\)。
两边同减 \(d_0\) 并除以 \(B\)（在 \(T\) 上单射）得到
\(
\sum_{j\ge 0}B^j d_{j+1}\equiv \sum_{j\ge 0}B^j e_{j+1}\pmod{B^{n-1}T}
\)，
归纳即得前 \(n\) 位一致。证毕。
\end{proof}

\begin{proposition}[截断误差最优界与分辨率纤维几何]\label{prop:conclusion-godel-leyang-truncation-fiber}
在定理 \ref{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension} 设定下，设
\[
x=\sum_{n\ge 0}B^n d_n\in K_D,\qquad
x^{(n)}:=\sum_{k=0}^{n-1}B^k d_k.
\]
则：
\begin{enumerate}
  \item 对任意 \(n\ge 1\)，有确定性误差界
  \begin{equation}\label{eq:conclusion-godel-leyang-trunc-error}
  \|x-x^{(n)}\|_2\le 2^{-nL};
  \end{equation}
  \item 令 \(\rho_n:T\to T/B^nT\) 为自然投影，则对任意 \(y\in \rho_n(K_D)\)，其纤维满足
  \begin{equation}\label{eq:conclusion-godel-leyang-fiber-affine-copy}
  (\rho_n|_{K_D})^{-1}(y)=x^{(n)}+B^nK_D
  \end{equation}
  （其中 \(x^{(n)}\) 由 \(y\) 的唯一前缀决定）；
  \item 上述纤维直径满足
  \begin{equation}\label{eq:conclusion-godel-leyang-fiber-diameter}
  \operatorname{diam}\!\bigl((\rho_n|_{K_D})^{-1}(y)\bigr)\le 2^{-nL},
  \end{equation}
  且当 \(|D|\ge 2\) 时等号成立。
\end{enumerate}
\leanverified{paper\_conclusion\_godel\_leyang\_truncation\_fiber}
\end{proposition}
\begin{proof}
\textbf{(1)} 由级数展开，
\[
x-x^{(n)}=\sum_{k=n}^{\infty}B^k d_k
=B^n\sum_{j=0}^{\infty}B^j d_{n+j}.
\]
括号内属于 \(K_D\subset T\)，故其 \(2\)-进范数不超过 \(1\)。
于是
\[
\|x-x^{(n)}\|_2\le |B^n|_2=2^{-nL}.
\]

\textbf{(2)} \(\rho_n(x)\) 仅依赖前 \(n\) 位地址 \((d_0,\dots,d_{n-1})\)。
定理 \ref{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension}(2) 给出该前缀唯一性，故固定 \(y\) 后，前缀唯一确定某个 \(x^{(n)}\)。
尾部 \((d_n,d_{n+1},\dots)\) 可在 \(D^{\NN}\) 任取，正好得到 \(x^{(n)}+B^nK_D\)，即 \eqref{eq:conclusion-godel-leyang-fiber-affine-copy}。

\textbf{(3)} 由 \eqref{eq:conclusion-godel-leyang-fiber-affine-copy}，
\[
\operatorname{diam}(x^{(n)}+B^nK_D)=|B^n|_2\,\operatorname{diam}(K_D)\le 2^{-nL}.
\]
若 \(|D|\ge 2\)，取 \(d\neq d'\in D\) 及任意固定尾部，可构造 \(x_1,x_2\in K_D\) 使 \(x_1-x_2=d-d'\)，由强分离得 \(\|d-d'\|_2=1\)，故 \(\operatorname{diam}(K_D)=1\)，从而 \eqref{eq:conclusion-godel-leyang-fiber-diameter} 取等。证毕。
\end{proof}

\begin{theorem}[信息守恒的有限群判据：Haar 测度当且仅当数字完备]\label{thm:conclusion-godel-leyang-haar-complete-digit}
在定理 \ref{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension} 设定下，记
\[
G:=T/BT,\qquad \bar D:=D\bmod BT\subset G.
\]
则以下三项等价：
\begin{enumerate}
  \item \(\mu_D\) 是 \(T\) 上的 Haar 概率测度（从而 \(K_D=T\)）；
  \item \(|D|=|G|=B^2\)，且 \(\bar D=G\)（即 \(D\) 在模 \(B\) 意义下完备）；
  \item 对任意非平凡角色 \(\chi\in\widehat G\setminus\{1\}\)，数字掩码
  \begin{equation}\label{eq:conclusion-godel-leyang-mask}
  m_D(\chi):=\frac{1}{|D|}\sum_{d\in D}\chi(\bar d)
  \end{equation}
  满足 \(m_D(\chi)=0\)。
\end{enumerate}
\leanverified{paper\_conclusion\_godel\_leyang\_haar\_complete\_digit\_seeds}
\end{theorem}
\begin{proof}
\textbf{(2) \(\Rightarrow\) (1).}
对每个 \(n\ge 1\)，映射
\[
(d_0,\dots,d_{n-1})\longmapsto
\sum_{j=0}^{n-1}B^j d_j \pmod{B^nT}
\]
在 \(D^n\) 与 \(T/B^nT\) 之间为双射（完备数字系的逐级 \(B\)-进展开唯一性）。
等权 Bernoulli 测度在 \(D^n\) 上均匀，故 \(\mu_D\) 在每个有限商 \(T/B^nT\) 的投影均匀。
由逆极限一致性，\(\mu_D\) 即 \(T\) 上 Haar 测度，且支撑为全体 \(T\)。

\textbf{(1) \(\Rightarrow\) (3).}
取 \(\chi\in\widehat G\setminus\{1\}\)，并把 \(\chi\) 视作 \(T\) 上在 \(BT\) 上平凡的连续角色。
由 \(\mu_D\) 的自相似关系
\[
\mu_D=\frac1{|D|}\sum_{d\in D}(\tau_d\circ B)_*\mu_D,\qquad \tau_d(x)=x+d,
\]
得
\[
\int_T\chi\,\dd\mu_D
=\frac1{|D|}\sum_{d\in D}\chi(d)\int_T\chi(Bx)\,\dd\mu_D(x).
\]
因 \(\chi\) 在 \(BT\) 上平凡，\(\chi(Bx)\equiv 1\)，故
\[
\int_T\chi\,\dd\mu_D=m_D(\chi).
\]
而 Haar 测度对非平凡角色积分为 \(0\)，故 \(m_D(\chi)=0\)。

\textbf{(3) \(\Rightarrow\) (2).}
在有限阿贝尔群 \(G\) 上取指示函数 \(f:=\mathbf 1_{\bar D}\)。
其 Fourier 系数
\[
\widehat f(\chi)=\sum_{g\in G}f(g)\chi(g)=\sum_{d\in D}\chi(\bar d)=|D|\,m_D(\chi).
\]
由假设，\(\widehat f(\chi)=0\) 对所有非平凡 \(\chi\) 成立。
Fourier 反演给出
\[
f(g)=\frac1{|G|}\sum_{\chi\in\widehat G}\widehat f(\chi)\chi(g)
=\frac{\widehat f(1)}{|G|}
=\frac{|D|}{|G|},
\]
即 \(f\) 为常函数。
因 \(f\in\{0,1\}\) 且 \(D\neq\varnothing\)，必有 \(f\equiv 1\)，故 \(\bar D=G\)，并且
\[
|D|=|\bar D|=|G|=B^2.
\]
证毕。
\end{proof}

\begin{proposition}[数字不足强制 Haar-零测]\label{prop:conclusion-godel-leyang-haar-null-small-digit}
在定理 \ref{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension} 的设定下，令 \(m_T\) 为紧群 \(T\) 的归一化 Haar 概率测度。
若 \(|D|<B^2\)，则
\[
m_T(K_D)=0.
\]
\end{proposition}
\begin{proof}
由 Hutchinson 方程 \eqref{eq:conclusion-godel-leyang-hutchinson}、Haar 测度的平移不变性与缩放性 \(m_T(BE)=B^{-2}m_T(E)\)，有
\[
m_T(K_D)
=m_T\!\left(\bigcup_{d\in D}(d+BK_D)\right)
\le \sum_{d\in D}m_T(d+BK_D)
=|D|\,B^{-2}m_T(K_D).
\]
当 \(|D|<B^2\) 时，系数 \(|D|B^{-2}<1\)，从而 \(m_T(K_D)=0\)。证毕。
\end{proof}

\begin{proposition}[Haar 随机分辨率相干深度的指数尾律]\label{prop:conclusion-godel-leyang-haar-coherence-depth}
令 \(X,Y\) 为 \(T\) 上独立的 Haar 随机变量。
定义分辨率相干深度
\[
\mathrm{coh}_B(X,Y):=\sup\{n\in\NN:\ X\equiv Y\pmod{B^nT}\}\in\NN\cup\{\infty\}.
\]
则对任意 \(n\ge 0\)，有
\[
\PP\!\bigl(\mathrm{coh}_B(X,Y)\ge n\bigr)=B^{-2n},
\qquad
\PP\!\bigl(\mathrm{coh}_B(X,Y)=n\bigr)=(1-B^{-2})B^{-2n}.
\]
\leanverified{paper\_conclusion\_godel\_leyang\_haar\_coherence\_depth}
\end{proposition}
\begin{proof}
差值 \(Z:=X-Y\) 仍为 Haar 分布，且事件 \(\mathrm{coh}_B(X,Y)\ge n\) 等价于 \(Z\in B^nT\)。
又 \(T/B^nT\) 的基数为 \(B^{2n}\)，故 \(m_T(B^nT)=B^{-2n}\)，从而
\[
\PP(\mathrm{coh}_B(X,Y)\ge n)=m_T(B^nT)=B^{-2n}.
\]
精确分布由尾概率差分得到。证毕。
\end{proof}

\begin{theorem}[Lee--Yang 分辨率塔多项式的单位圆零点与等分布极限]\label{thm:conclusion-godel-leyang-resolution-tower-circle-law}
设 \(B=2^L\ge 2\)，并取整数 \(M\) 满足 \(1\le M<B\)。
定义
\begin{equation}\label{eq:conclusion-godel-leyang-resolution-partition-poly}
Z_n(z):=\prod_{k=0}^{n-1}\left(\sum_{a=0}^{M}z^{aB^k}\right)\in\ZZ[z],\qquad n\ge 1.
\end{equation}
则成立：
\begin{enumerate}
  \item \(Z_n\) 的全部零点均位于单位圆 \(S^1\)；
  \item 记 \(\nu_n\) 为 \(Z_n\) 的零点计数测度（按重数）归一化到总质量 \(1\)，则
  \begin{equation}\label{eq:conclusion-godel-leyang-zeros-weak-limit}
  \nu_n\Longrightarrow m_{S^1}\qquad(n\to\infty),
  \end{equation}
  其中 \(m_{S^1}\) 为单位圆 Haar 概率测度；
  \item 零点集合的聚点集为整个 \(S^1\)。
\end{enumerate}
\end{theorem}
\begin{proof}
令
\[
S_k(z):=\sum_{a=0}^{M}z^{aB^k}.
\]
当 \(z^{B^k}\neq 1\) 时，
\[
S_k(z)=\frac{z^{(M+1)B^k}-1}{z^{B^k}-1}.
\]
故 \(S_k\) 的零点恰为满足 \(z^{(M+1)B^k}=1\) 且 \(z^{B^k}\neq 1\) 的单位根，均在 \(S^1\) 上。
因此 \(Z_n\) 的全部零点属于 \(S^1\)。

记 \(\mathcal R_k\) 为 \(S_k\) 的零点多重集。
其基数为 \(|\mathcal R_k|=MB^k\)，而
\[
\deg Z_n=\sum_{k=0}^{n-1}MB^k=M\frac{B^n-1}{B-1}.
\]
设 \(\nu_k\) 为 \(\mathcal R_k\) 的归一化计数测度，则
\[
\nu_n=\sum_{k=0}^{n-1}w_{n,k}\nu_k,\qquad
w_{n,k}:=\frac{MB^k}{\deg Z_n}.
\]

取 \(m\in\ZZ\setminus\{0\}\)。
利用单位根求和公式，
\[
\widehat{\nu_k}(m)
=\frac{1}{MB^k}
\left(
\sum_{\zeta^{(M+1)B^k}=1}\zeta^m
-\sum_{\xi^{B^k}=1}\xi^m
\right)
\]
\[
=\frac{1}{MB^k}
\left(
(M+1)B^k\,\mathbf 1_{(M+1)B^k\mid m}
-B^k\,\mathbf 1_{B^k\mid m}
\right).
\]
对固定 \(m\neq 0\)，当 \(B^k>|m|\) 时，两指示函数均为 \(0\)，故 \(\widehat{\nu_k}(m)=0\)。
于是
\[
\widehat{\nu_n}(m)
=\sum_{0\le k\le \lfloor \log_B|m|\rfloor}w_{n,k}\widehat{\nu_k}(m)\xrightarrow[n\to\infty]{}0,
\]
因为每个固定 \(k\) 的权重 \(w_{n,k}\to 0\)。
同时 \(\widehat{\nu_n}(0)=1\)。
由三角多项式稠密性与 Fourier 唯一性，\(\nu_n\Rightarrow m_{S^1}\)。

最后，对任意开弧 \(I\subset S^1\)（端点集测度零），有
\[
\nu_n(I)\to m_{S^1}(I)>0,
\]
故 \(I\) 中无穷多 \(n\) 均出现零点，任意开弧皆然，故聚点集为 \(S^1\)。证毕。
\end{proof}

\begin{theorem}[有限样本下数字扭点集的可识别恢复]\label{thm:conclusion-godel-leyang-finite-sample-identifiability}
仍在定理 \ref{thm:conclusion-godel-leyang-2adic-ifs-exact-dimension} 设定下，设 \(X_1,\dots,X_N\) 独立同分布且
\[
X_i\sim \mu_D.
\]
记
\[
\bar X_i:=X_i\bmod BT\in T/BT,\qquad
\bar D:=D\bmod BT.
\]
则：
\begin{enumerate}
  \item \(\bar X_i\) 在 \(\bar D\) 上服从均匀分布；
  \item 若定义经验支持
  \[
  \widehat D_N:=\{\bar X_i:\ 1\le i\le N\}\subset T/BT,
  \]
  则对任意 \(0<\delta<1\)，当
  \begin{equation}\label{eq:conclusion-godel-leyang-coupon-bound}
  N\ge |D|\!\left(\log|D|+\log\frac1\delta\right)
  \end{equation}
  时，
  \[
  \PP\!\left(\widehat D_N=\bar D\right)\ge 1-\delta;
  \]
  \item 若额外固定一个代表截面 \(s:T/BT\to T\) 且 \(D=s(\bar D)\)，则 \(\widehat D_N\) 以同样概率界唯一确定 \(D\)。
\end{enumerate}
\end{theorem}
\begin{proof}
令 \(X=\sum_{n\ge 0}B^nD_n\)（\(D_n\) 为 \(D\)-值等权独立位）。
模 \(BT\) 后仅保留首位：
\[
\bar X=D_0\bmod BT.
\]
由强分离 \eqref{eq:conclusion-godel-leyang-ssc-modBT}，\(D\to\bar D\) 单射，且 \(D_0\) 在 \(D\) 等概率，故 \(\bar X\) 在 \(\bar D\) 等概率。得 (1)。

对任意 \(d\in\bar D\)，其未被 \(N\) 个样本命中的概率为
\[
\PP(d\notin\widehat D_N)=\left(1-\frac1{|D|}\right)^N\le e^{-N/|D|}.
\]
并合界给出
\[
\PP(\widehat D_N\neq \bar D)
\le |D|e^{-N/|D|}.
\]
当 \eqref{eq:conclusion-godel-leyang-coupon-bound} 成立时，右端不超过 \(\delta\)，即得 (2)。

(3) 由固定截面假设 \(D=s(\bar D)\) 立即推出。证毕。
\end{proof}

\begin{proposition}[单位圆谱无限乘积的有理性障碍]\label{prop:conclusion-digital-local-factor-rationality-obstruction}
设 \(\{M_k\}_{k\ge 0}\) 为复有限维矩阵序列，并假设对每个 \(k\) 有
\(
\mathrm{Spec}(M_k)\subseteq\{z\in\CC:\ |z|=1\}
\)。
令
\[
F(t):=\prod_{k\ge 0}\det(I-tM_k)^{-1},
\]
并假设该乘积在 \(t=0\) 的某邻域内收敛为亚纯函数。
若 \(F(t)\) 为有理函数，则除有限多个 \(k\) 外必有 \(M_k=0\)（等价地：仅有限多个因子非平凡）。
特别地，若额外要求 \(\dim(M_k)\ge 1\) 对无穷多个 \(k\) 成立，则 \(F(t)\) 不可能为有理函数。
\leanverified{paper\_conclusion\_digital\_local\_factor\_rationality\_obstruction}
\end{proposition}

\begin{proof}
对每个 \(k\)，因子 \(\det(I-tM_k)^{-1}\) 的极点集合包含 \(\{t=\lambda^{-1}:\ \lambda\in \mathrm{Spec}(M_k)\setminus\{0\}\}\subseteq\{|t|=1\}\)，且极点阶由代数重数给出。
若无穷多个 \(k\) 贡献非零谱，则要么产生无穷多个互异极点，要么在某个极点上产生无界阶数；两者均与有理函数仅有有限多个极点且各极点阶有限矛盾。
故若 \(F\) 有理，则除有限多个 \(k\) 外各因子必须无极点，即 \(\det(I-tM_k)\equiv 1\)，从而 \(M_k=0\)。证毕。
\end{proof}

\begin{corollary}[对局部因子化形式的正则化约束]\label{cor:conclusion-digital-local-factor-regularization-constraint}
\leanverified{paper\_conclusion\_digital\_local\_factor\_regularization\_constraint}
在猜想 \ref{conj:conclusion-godel-leyang-local-langlands-digital-spectrum} 的形式 \eqref{eq:conclusion-godel-leyang-digital-local-factor}--\eqref{eq:conclusion-godel-leyang-local-factor-identity} 下，把变量写作 \(t=2^{-s}\)。
若右侧局部因子 \(L_2(E_{\min},s)\) 作为 \(t\) 的函数为有理函数而同时无穷多个算子 \(\mathbf M_D(2^{Lk})\) 非零，则必须引入与逐因子乘积不同的正则化/重整解释（例如 Fredholm 行列式口径的迹类实现），以避免命题 \ref{prop:conclusion-digital-local-factor-rationality-obstruction} 的有理性障碍。
\end{corollary}

\begin{proposition}[非单位根相位强制连续圆因子]\label{prop:conclusion-noncyclotomic-phase-forces-cdim}
\leanverified{paper\_conclusion\_noncyclotomic\_phase\_forces\_cdim}
设 \(A\) 为离散交换群，\(\widehat A\) 为其 Pontryagin 对偶，并令 \(\cdim^*(A):=\dim_{\mathrm{top}}\bigl((\widehat A)_0\bigr)\)（定义 \ref{def:circle-dimension}）。
若存在 \(a\in A\) 与角色 \(\chi\in\widehat A\) 使 \(\chi(a)\in S^1\) 不是单位根，则 \(\cdim^*(A)\ge 1\)。
\end{proposition}

\begin{proof}
若 \(a\) 为有限阶元，则对任意角色 \(\chi\) 都有 \(\chi(a)\) 为单位根；故假设推出 \(a\) 具有无限阶，从而 \(\langle a\rangle\cong \ZZ\) 嵌入 \(A\)。
对偶上限制映射给出连续满射 \(\widehat A\twoheadrightarrow \widehat{\ZZ}\cong S^1\)。
因此 \((\widehat A)_0\) 的拓扑维数至少为 \(1\)，即 \(\cdim^*(A)\ge 1\)。证毕。
\end{proof}
