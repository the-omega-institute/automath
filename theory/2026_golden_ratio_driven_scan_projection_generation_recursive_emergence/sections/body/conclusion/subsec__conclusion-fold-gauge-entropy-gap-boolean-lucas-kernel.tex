\subsection{规范体积的熵差额与布尔二值核的 Lucas 谱律}\label{subsec:conclusion-fold-gauge-entropy-gap-boolean-lucas-kernel}
本小节把 fold-gauge 体积密度的 Stirling 缺口压缩为输出分布熵的精确差额公式与热力学常数项，并把布尔二值跳跃核的张量谱进一步凝练为 Lucas 幂迹、惯性分解与精确范数闭式。

\begin{proposition}[fold-gauge 体积密度的熵差额精确公式]\label{prop:conclusion-fold-gauge-entropy-defect-exact}
沿用命题 \ref{prop:fold-bin-gauge-volume} 的记号。
记
\[
\pi_m(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
H(\pi_m):=-\sum_{w\in X_m}\pi_m(w)\log \pi_m(w).
\]
则有严格恒等式
\[
\kappa_m=m\log 2-H(\pi_m).
\]
因此结合命题 \ref{prop:fold-bin-gauge-volume} 的夹逼
\(
\kappa_m-1\le g_m\le \kappa_m
\),
得到
\[
m\log 2-H(\pi_m)-1\le g_m\le m\log 2-H(\pi_m),
\]
等价地，
\[
m\log 2-g_m-1\le H(\pi_m)\le m\log 2-g_m.
\]
\end{proposition}
\begin{proof}
由定义
\[
\kappa_m
=
\frac{1}{2^m}\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w)
=
\sum_{w\in X_m}\pi_m(w)\log d_m^{\mathrm{bin}}(w).
\]
再由
\(
d_m^{\mathrm{bin}}(w)=2^m\pi_m(w)
\)
可得
\[
\kappa_m
=
\sum_{w\in X_m}\pi_m(w)\log\bigl(2^m\pi_m(w)\bigr)
=
m\log 2+\sum_{w\in X_m}\pi_m(w)\log \pi_m(w)
=
m\log 2-H(\pi_m).
\]
把该恒等式代回命题 \ref{prop:fold-bin-gauge-volume} 即得其余不等式。证毕。
\leanverified{paper\_fold\_gauge\_entropy\_defect\_witness}
\leanverified{paper\_conclusion\_fold\_gauge\_entropy\_defect\_exact}
\end{proof}

\begin{theorem}[规范体积密度的精确常数项]\label{thm:conclusion-fold-gauge-entropy-density-exact-constant}
沿用命题 \ref{prop:conclusion-fold-gauge-entropy-defect-exact} 的记号，并记
\[
\kappa_m:=\frac{1}{2^m}\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w),
\qquad
g_m:=2^{-m}\log\bigl|\mathcal G_m^{\mathrm{bin}}\bigr|.
\]
则有精确展开
\[
\kappa_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-\frac{\log\varphi}{1+\varphi^2}
+O\!\left(\Bigl(\frac{\varphi}{2}\Bigr)^m\right),
\]
以及
\[
g_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-1
-\frac{\log\varphi}{1+\varphi^2}
+O\!\left(m\Bigl(\frac{\varphi}{2}\Bigr)^m\right).
\]
由于
\[
1+\varphi^2=\varphi\sqrt5,
\]
第二式亦可写为
\[
g_m
=
m\log\!\Bigl(\frac{2}{\varphi}\Bigr)
-1
-\frac{\log\varphi}{\varphi\sqrt5}
+O\!\left(m\Bigl(\frac{\varphi}{2}\Bigr)^m\right).
\]
\end{theorem}
\begin{proof}
第一式与第二式正是推论 \ref{cor:xi-foldbin-shannon-gauge-constant-mirror} 的两条展开。最后一式只需使用恒等式
\[
1+\varphi^2=\varphi\sqrt5
\]
作代数改写。证毕。
\end{proof}
\leanverified{paper\_conclusion\_fold\_gauge\_entropy\_density\_exact\_constant}

\begin{theorem}[布尔二值跳跃核的 Lucas 迹、惯性分解与精确范数]\label{thm:conclusion-boolean-jump-kernel-lucas-inertia-norm}
沿用定理 \ref{thm:xi-boolean-binary-jump-kernels-tensor-spectrum} 的记号。
对 \(q\ge 1\) 记布尔二值跳跃核为
\[
\mathsf C_q(u,v):=\mathbf 1_{\{u\vee v=\mathbf 1\}},
\qquad
\mathsf D_q(u,v):=\mathbf 1_{\{u\wedge v=\mathbf 0\}}
\qquad
(u,v\in\{0,1\}^q),
\]
并令 Lucas 数列写为
\[
L_t:=\varphi^t+(-\varphi^{-1})^t
\qquad (t\ge 0).
\]
则成立：
\begin{enumerate}
  \item 对任意整数 \(t\ge 1\)，
  \[
  \operatorname{tr}(\mathsf C_q^{\,t})
  =
  \operatorname{tr}(\mathsf D_q^{\,t})
  =
  (L_t)^q.
  \]
  \item \(\mathsf C_q\) 与 \(\mathsf D_q\) 的正特征值总重数与负特征值总重数都等于 \(2^{q-1}\)，从而二者签名都为 \(0\)。
  \item 记 \(\|\cdot\|_2\) 为谱范数、\(\|\cdot\|_F\) 为 Frobenius 范数，则
  \[
  \|\mathsf C_q\|_2=\|\mathsf D_q\|_2=\varphi^q,
  \qquad
  \|\mathsf C_q\|_F=\|\mathsf D_q\|_F=3^{q/2}.
  \]
  并且最小奇异值满足
  \[
  \sigma_{\min}(\mathsf C_q)=\sigma_{\min}(\mathsf D_q)=\varphi^{-q},
  \]
  因而 \(2\)-条件数满足
  \[
  \kappa_2(\mathsf C_q)=\kappa_2(\mathsf D_q)=\varphi^{2q}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-boolean-binary-jump-kernels-tensor-spectrum}，
\(\mathsf C_q\) 与 \(\mathsf D_q\) 的特征值多重集都为
\[
\lambda_k^{(q)}=(-1)^{q-k}\varphi^{2k-q}
\qquad (0\le k\le q),
\]
其中重数为 \(\binom qk\)。
因此对任意 \(t\ge 1\)，
\[
\operatorname{tr}(\mathsf C_q^{\,t})
=
\sum_{k=0}^{q}\binom qk\bigl(\lambda_k^{(q)}\bigr)^t
=
\sum_{k=0}^{q}\binom qk
\bigl(\varphi^t\bigr)^k
\bigl((-\varphi^{-1})^t\bigr)^{q-k}
=
\bigl(\varphi^t+(-\varphi^{-1})^t\bigr)^q
=
(L_t)^q.
\]
\(\mathsf D_q\) 的迹公式完全相同，得第 (1) 条。

第 (2) 条中，\(\lambda_k^{(q)}\) 的符号仅由 \(q-k\) 的奇偶决定。
故正特征值总重数为
\[
\sum_{\substack{0\le k\le q\\ k\equiv q\!\!\!\pmod 2}}\binom qk=2^{q-1},
\]
负特征值总重数同理也为 \(2^{q-1}\)。
于是两者签名都为 \(0\)。

对第 (3) 条，由同一定理，\(\mathsf C_q,\mathsf D_q\) 都是实对称矩阵，故奇异值恰为特征值的绝对值。
最大特征值模为 \(\varphi^q\)（取 \(k=q\)），最小特征值模为 \(\varphi^{-q}\)（取 \(k=0\)），于是
\[
\|\mathsf C_q\|_2=\|\mathsf D_q\|_2=\varphi^q,
\qquad
\sigma_{\min}(\mathsf C_q)=\sigma_{\min}(\mathsf D_q)=\varphi^{-q},
\]
从而
\[
\kappa_2(\mathsf C_q)=\kappa_2(\mathsf D_q)=\varphi^{2q}.
\]
再由定理 \ref{thm:xi-boolean-binary-jump-kernels-tensor-spectrum} 的张量分解
\[
\mathsf C_q=A^{\otimes q},
\qquad
\mathsf D_q=B^{\otimes q},
\]
以及 Frobenius 范数在 Kronecker 幂下的乘法性，
\[
\|\mathsf C_q\|_F=\|A\|_F^{\,q},
\qquad
\|\mathsf D_q\|_F=\|B\|_F^{\,q}.
\]
而
\[
\|A\|_F^2=\|B\|_F^2=0^2+1^2+1^2+1^2=3,
\]
故
\[
\|\mathsf C_q\|_F=\|\mathsf D_q\|_F=3^{q/2}.
\]
证毕。
\end{proof}
\leanverified{paper\_conclusion\_boolean\_jump\_kernel\_lucas\_inertia\_norm}

\noindent
由此，fold-gauge 体积密度的主阶斜率被 \(m\log(2/\varphi)\) 锁定，而其常数层缺口精确等于 \(1+\log\varphi/(\varphi\sqrt5)\)；另一方面，布尔二值跳跃核的全部谱侧数据又被 Lucas 数列、符号平衡与黄金条件数完全决定。于是，gauge 侧的体积增长、熵差额与组合谱律便在同一条黄金常数链上完成闭合。
