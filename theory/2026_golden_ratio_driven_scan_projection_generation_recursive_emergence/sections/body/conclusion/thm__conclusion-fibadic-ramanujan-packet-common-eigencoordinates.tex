\begin{theorem}[Ramanujan 包核的 gcd-卷积共同本征性]\label{thm:conclusion-fibadic-ramanujan-packet-common-eigenpacket}
\leanverified{paper\_conclusion\_fibadic\_ramanujan\_packet\_common\_eigenpacket}
沿用定理 \ref{thm:conclusion-fibadic-conditional-expectation-gcd-algebra} 与定理 \ref{thm:conclusion-fibadic-exact-depth-kernel-ramanujan-packet} 的记号。对任意有限支撑算术函数 \(\alpha:\NN\to\CC\)，记
\[
T_\alpha:=\sum_{n\ge 1}\alpha(n)E_n,
\qquad
\widehat\alpha(d):=\sum_{d\mid n}\alpha(n).
\]
则对每个 \(d\ge 1\)，精确深度 Ramanujan 包核 \(K_d\) 满足严格本征关系
\[
T_\alpha K_d=\widehat\alpha(d)\,K_d.
\]
特别地，对单点权 \(\alpha=\mathbf 1_{\{m\}}\) 有
\[
E_mK_d=\mathbf 1_{d\mid m}\,K_d.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-exact-depth-kernel-ramanujan-packet}，\(K_d\) 是所有精确深度为 \(d\) 的特征之和，故
\[
K_d\in \mathcal N_d.
\]
再由定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization}，
\[
T_\alpha|_{\mathcal N_d}=\widehat\alpha(d)\,\mathrm{id}_{\mathcal N_d}.
\]
将该标量作用直接施于 \(K_d\) 即得
\[
T_\alpha K_d=\widehat\alpha(d)\,K_d.
\]
当 \(\alpha=\mathbf 1_{\{m\}}\) 时，
\[
\widehat\alpha(d)=\sum_{d\mid n}\mathbf 1_{\{m\}}(n)=\mathbf 1_{d\mid m},
\]
从而
\[
E_mK_d=\mathbf 1_{d\mid m}\,K_d.
\]
证毕。
\end{proof}

\begin{corollary}[Bernoulli 射流的一阶 Ramanujan 锁定]\label{cor:conclusion-fibadic-bernoulli-jet-kd-zero-lock}
\leanverified{paper\_conclusion\_fibadic\_bernoulli\_jet\_kd\_zero\_lock}
对每个 \(d\ge 1\)，有
\[
\left.\frac{\dd}{\dd u}\log \Pi_d(e^u)\right|_{u=0}=\frac{K_d(0)}{2},
\qquad
2\,\Pi_d'(1)=K_d(0)\,\Pi_d(1).
\]
\end{corollary}
\begin{proof}
由前文关于 Bernoulli--M\"obius 射流的结论，
\[
\left.\frac{\dd}{\dd u}\log \Pi_d(e^u)\right|_{u=0}=\frac{a_d}{2},
\qquad
2\,\Pi_d'(1)=a_d\,\Pi_d(1).
\]
另一方面，定理 \ref{thm:conclusion-fibadic-exact-depth-kernel-ramanujan-packet} 给出
\[
K_d(0)=a_d.
\]
代入即得所述两式。证毕。
\end{proof}
