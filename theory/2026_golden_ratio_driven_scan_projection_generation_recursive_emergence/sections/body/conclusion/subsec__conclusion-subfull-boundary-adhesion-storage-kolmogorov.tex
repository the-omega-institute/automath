\subsection{subfull 维 regime 中的粘连饱和、边界位长指数与 Kolmogorov 下界}\label{subsec:conclusion-subfull-boundary-adhesion-storage-kolmogorov}
沿用小节 \ref{subsec:conclusion-boundary-godel-gluing-defect-kolmogorov-roughness} 的记号。前述结果已经给出精确相图
\[
d_{\partial G}(A)=\max\{d_G(A)-1,\gamma_{\mathrm{adh}}(A)\},
\]
并且在 \(d_G(A)<n\) 时退化为
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A).
\]
在此基础上，还可继续抽出一条更细的终端链：正距并集下粘连缺陷服从极大律，bulk 主导体与 edge 主导体可以由互不接触的分量分别承载，bulk primorial 双对数的二阶喷射对边界类并不完备，而精确边界 Gödel 码的位长增长率与条件 Kolmogorov 下界又都可直接读出同一个粘连指数。

\begin{theorem}[正距并集下的粘连缺陷极大律]\label{thm:conclusion-boundary-godel-separated-union-adhesion-maxlaw}
\leanverified{paper\_conclusion\_boundary\_godel\_separated\_union\_adhesion\_maxlaw}
设 \(A,B\subset I^n\) 为紧集，并满足
\[
\operatorname{dist}(A,B)>0,
\qquad
\max\{d_G(A),d_G(B)\}<n.
\]
则
\[
\gamma_{\mathrm{adh}}(A\cup B)
=
\max\{\gamma_{\mathrm{adh}}(A),\gamma_{\mathrm{adh}}(B)\}.
\]
换言之，在正距并集中，边界粗糙度既不平均化，也不相互抵消，而只由较粗糙分量单独主导。
\end{theorem}
\begin{proof}
由 \(\operatorname{dist}(A,B)>0\)，存在 \(m_0\) 使对一切 \(m\ge m_0\)，dyadic 外近似 \(U_m(A)\) 与 \(U_m(B)\) 互不相交。于是
\[
N_m(A\cup B)=N_m(A)+N_m(B)
\]
对充分大 \(m\) 成立，从而
\[
d_G(A\cup B)=\max\{d_G(A),d_G(B)\}<n.
\]
另一方面，由命题 \ref{prop:spg-boundary-godel-dimension-separated-union-max}，
\[
d_{\partial G}(A\cup B)=\max\{d_{\partial G}(A),d_{\partial G}(B)\}.
\]
再对 \(A\)、\(B\) 与 \(A\cup B\) 分别应用定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram} 在 \(d_G<n\) 下的特例，得
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A),\qquad
d_{\partial G}(B)=\gamma_{\mathrm{adh}}(B),\qquad
d_{\partial G}(A\cup B)=\gamma_{\mathrm{adh}}(A\cup B).
\]
代回即得所示极大律。证毕。
\end{proof}

\begin{theorem}[bulk 主导体与 edge 主导体的正距分离]\label{thm:conclusion-boundary-godel-bulk-edge-dominant-component-separation}
\leanverified{paper\_conclusion\_boundary\_godel\_bulk\_edge\_dominant\_component\_separation}
设 \(A,B\subset I^n\) 为紧集，并满足
\[
\operatorname{dist}(A,B)>0,\qquad d_G(A),d_G(B)<n,
\]
\[
d_G(A)<d_G(B),\qquad
\gamma_{\mathrm{adh}}(A)>\gamma_{\mathrm{adh}}(B).
\]
则并集同时满足
\[
d_G(A\cup B)=d_G(B),
\qquad
d_{\partial G}(A\cup B)=\gamma_{\mathrm{adh}}(A).
\]
因此，同一并集对象的 bulk 增长与边界粗糙度可以分别由两个彼此正距分离的分量独立承载。
\end{theorem}
\begin{proof}
由正距分离下的体积可加性，
\[
N_m(A\cup B)=N_m(A)+N_m(B)
\]
对充分大 \(m\) 成立，因此
\[
d_G(A\cup B)=\max\{d_G(A),d_G(B)\}=d_G(B).
\]
另一方面，由定理 \ref{thm:conclusion-boundary-godel-separated-union-adhesion-maxlaw}，
\[
\gamma_{\mathrm{adh}}(A\cup B)
=
\max\{\gamma_{\mathrm{adh}}(A),\gamma_{\mathrm{adh}}(B)\}
=\gamma_{\mathrm{adh}}(A).
\]
再由定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram} 在 \(d_G(A\cup B)<n\) 下的特例，
\[
d_{\partial G}(A\cup B)=\gamma_{\mathrm{adh}}(A\cup B)=\gamma_{\mathrm{adh}}(A).
\]
证毕。
\end{proof}

\begin{theorem}[bulk primorial 二阶喷射对边界粗糙度的不完备性]\label{thm:conclusion-boundary-godel-bulk-secondorder-jet-incomplete}
设 \(A,B\subset I^n\) 为紧集，并满足
\[
d_G(A)=d_G(B)=d<n.
\]
若存在同一常数 \(M\in(0,\infty)\) 使
\[
N_m(A)=M\,2^{md}+o(2^{md}),
\qquad
N_m(B)=M\,2^{md}+o(2^{md}),
\]
而
\[
\gamma_{\mathrm{adh}}(A)\neq \gamma_{\mathrm{adh}}(B),
\]
则
\[
d_{\partial G}(A)\neq d_{\partial G}(B).
\]
特别地，bulk primorial Gödel 码在双对数一阶斜率与二阶含量项上给出的数据 \((d,M)\) 并不足以决定边界 Gödel 维。
\end{theorem}
\begin{proof}
由定理 \ref{thm:spg-godel-doublelog-second-order-minkowski-content}，
\[
\log\log G_m^\ast(A)
=md\log 2+\log M+\log m+\log(d\log 2)+o(1),
\]
对 \(B\) 亦同。故 \(A\) 与 \(B\) 在 bulk primorial 双对数的一阶斜率与二阶含量项上完全一致。

另一方面，由定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram} 在 \(d_G<n\) 下的特例，
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A),
\qquad
d_{\partial G}(B)=\gamma_{\mathrm{adh}}(B).
\]
若 \(\gamma_{\mathrm{adh}}(A)\neq \gamma_{\mathrm{adh}}(B)\)，则边界 Gödel 维必不相同。证毕。
\end{proof}

\begin{theorem}[高环境维极限中的粘连饱和]\label{thm:conclusion-boundary-godel-highambient-adhesion-saturation}
\leanverified{paper\_conclusion\_boundary\_godel\_highambient\_adhesion\_saturation}
设 \((A_n)_{n\ge 1}\) 是一列紧集，满足
\[
d_G(A_n)<n\qquad(\forall n),
\qquad
\sup_{n} d_G(A_n)<\infty.
\]
则
\[
0\le d_G(A_n)-\gamma_{\mathrm{adh}}(A_n)\le \frac{d_G(A_n)}{n}\xrightarrow[n\to\infty]{}0.
\]
若进一步存在常数 \(c>0\) 使对充分大 \(n\) 都有
\[
d_G(A_n)\ge c,
\]
则
\[
\frac{\gamma_{\mathrm{adh}}(A_n)}{d_G(A_n)}\longrightarrow 1.
\]
因此，在高环境维而 bulk 双对数维保持有界的体制中，粘连缺陷指数渐近饱和 bulk 双对数维。
\end{theorem}
\begin{proof}
由定理 \ref{thm:spg-bulk-boundary-godel-dimension-sandwich}，
\[
0\le d_G(A_n)-d_{\partial G}(A_n)\le \frac{d_G(A_n)}{n}.
\]
而由定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram} 在 \(d_G(A_n)<n\) 下的特例，
\[
d_{\partial G}(A_n)=\gamma_{\mathrm{adh}}(A_n).
\]
合并即得
\[
0\le d_G(A_n)-\gamma_{\mathrm{adh}}(A_n)\le \frac{d_G(A_n)}{n}.
\]
由于 \(\sup_n d_G(A_n)<\infty\)，右端趋于 \(0\)，故差额趋于 \(0\)。若再有 \(d_G(A_n)\ge c>0\)，则两边同除以 \(d_G(A_n)\) 即得比值极限。证毕。
\end{proof}

\begin{theorem}[精确边界 Gödel 码位长的指数等于粘连缺陷指数]\label{thm:conclusion-boundary-godel-bitlength-exponent-equals-adhesion}
\leanverified{paper\_conclusion\_boundary\_godel\_bitlength\_exponent\_equals\_adhesion}
设 \(A\subset I^n\) 为紧集，并满足
\[
d_G(A)<n.
\]
定义精确边界 Gödel 码的位长
\[
L_{\partial,m}(A):=\left\lceil \log_2 H_m(U_m(A))\right\rceil .
\]
则
\[
\limsup_{m\to\infty}\frac{\log L_{\partial,m}(A)}{m\log 2}
=
\gamma_{\mathrm{adh}}(A).
\]
等价地，
\[
L_{\partial,m}(A)=2^{\,\gamma_{\mathrm{adh}}(A)m+o_{\limsup}(m)}.
\]
故粘连缺陷指数可被精确边界整数码的存储位长增长率直接读取。
\end{theorem}
\begin{proof}
由命题 \ref{prop:spg-dyadic-boundary-godel-doublelog-scale}，
\[
\log\log H_m(U_m(A))=\log F(U_m(A))+O(\log m).
\]
故
\[
d_{\partial G}(A)=\limsup_{m\to\infty}\frac{\log\log H_m(U_m(A))}{m\log 2}.
\]
另一方面，由 \(L_{\partial,m}(A)=\log_2 H_m(U_m(A))+O(1)\) 可得
\[
\log L_{\partial,m}(A)=\log\log H_m(U_m(A))+O(1).
\]
两边同除以 \(m\log 2\) 并取上极限，得到
\[
\limsup_{m\to\infty}\frac{\log L_{\partial,m}(A)}{m\log 2}
=
d_{\partial G}(A).
\]
最后再由定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram} 在 \(d_G(A)<n\) 下的特例，
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A),
\]
即得结论。指数形式只是上式的等价改写。证毕。
\end{proof}

\begin{corollary}[正距并集的精确边界存储由最粗糙分量单独决定]\label{cor:conclusion-boundary-godel-separated-union-storage-maxlaw}
\leanverified{paper\_conclusion\_boundary\_godel\_separated\_union\_storage\_maxlaw}
设 \(A,B\subset I^n\) 为紧集，并满足
\[
\operatorname{dist}(A,B)>0,
\qquad
\max\{d_G(A),d_G(B)\}<n.
\]
则
\[
\limsup_{m\to\infty}\frac{\log L_{\partial,m}(A\cup B)}{m\log 2}
=
\max\left\{
\limsup_{m\to\infty}\frac{\log L_{\partial,m}(A)}{m\log 2},
\limsup_{m\to\infty}\frac{\log L_{\partial,m}(B)}{m\log 2}
\right\}
\]
\[
=
\max\{\gamma_{\mathrm{adh}}(A),\gamma_{\mathrm{adh}}(B)\}.
\]
因此，在精确整数化存储层面，正距并集的最坏边界复杂度由最粗糙的单一分量独占。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-boundary-godel-bitlength-exponent-equals-adhesion}，各项位长指数分别等于对应对象的 \(\gamma_{\mathrm{adh}}\)。再由定理 \ref{thm:conclusion-boundary-godel-separated-union-adhesion-maxlaw} 即得结论。证毕。
\end{proof}

\begin{theorem}[Kolmogorov 下界迫使粘连粗糙度下界]\label{thm:conclusion-boundary-godel-kolmogorov-lowerbound-forces-adhesion}
\leanverified{paper\_conclusion\_boundary\_godel\_kolmogorov\_lowerbound\_forces\_adhesion}
设 \(A\subset I^n\) 为紧集，并满足
\[
d_G(A)<n.
\]
若存在 \(\kappa>0\) 与无穷子序列 \(m_j\to\infty\)，使边界 Gödel 整数满足
\[
K\!\left(H_{m_j}(U_{m_j}(A))\,\middle|\,m_j\right)\ge 2^{\kappa m_j},
\]
则必有
\[
\gamma_{\mathrm{adh}}(A)\ge \kappa.
\]
换言之，若边界整数码在条件 Kolmogorov 复杂度上不能被压缩到低于 \(2^{\kappa m}\) 的量级，则粘连缺陷指数至少为 \(\kappa\)。
\end{theorem}
\begin{proof}
对任意正整数 \(N\)，总有
\[
K(N\mid m)\le \log_2 N+O(1).
\]
因此题设下沿子序列 \(m_j\) 必有
\[
\log_2 H_{m_j}(U_{m_j}(A))\ge 2^{\kappa m_j-O(1)}.
\]
再取对数得到
\[
\log\log H_{m_j}(U_{m_j}(A))
\ge \kappa m_j\log 2+O(1).
\]
于是
\[
d_{\partial G}(A)
=
\limsup_{m\to\infty}\frac{\log\log H_m(U_m(A))}{m\log 2}
\ge \kappa.
\]
最后由定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram} 在 \(d_G(A)<n\) 下的特例，
\[
\gamma_{\mathrm{adh}}(A)=d_{\partial G}(A)\ge \kappa.
\]
证毕。
\end{proof}

\noindent
由此，subfull 维 regime 中的边界复杂度可被进一步压缩为一条纯粘连链：正距并集下它服从极大律，并允许 bulk 主导体与 edge 主导体由不同分量分别承载；bulk primorial 双对数的一阶斜率与二阶含量项虽然完全锁定体积增长，却仍不足以决定边界类；在高环境维而 bulk 维保持有界时，粘连指数又渐近饱和 bulk 双对数维。更进一步，\(\gamma_{\mathrm{adh}}(A)\) 既是精确边界 Gödel 码位长的增长指数，也是边界整数码条件 Kolmogorov 不可压缩性的刚性下界。于是，在 \(d_G(A)<n\) 的体制中，boundary complexity 不再由 volumetric 分支主导，而成为一种可由几何极大律、算术位长与算法复杂度三重读取的纯粘连不变量。
