\[
M_m:=F_{m+2},
\qquad
p_m(r):=\frac{d_m(r)}{2^m}
\quad
\left(r\in \mathbb Z/(M_m\mathbb Z)\right),
\qquad
u_m(r):=\frac1{M_m},
\]
\[
a_m(k):=\frac{|\widehat d_m(k)|}{2^m},
\qquad
C_\varphi:=\lim_{m\to\infty}a_m(F_m)=\lim_{m\to\infty}a_m(F_{m+1})>0.
\]

\begin{theorem}[单一共轭 Fibonacci 共振对的四标量强制]\label{thm:conclusion-fold-singlepair-visible-obstructions}
\leanverified{paper\_conclusion\_fold\_singlepair\_visible\_obstructions}
对任意 \(\varepsilon\in(0,C_\varphi)\)，存在 \(m_0(\varepsilon)\)，使得对一切 \(m\ge m_0(\varepsilon)\) 同时成立
\[
\|p_m-u_m\|_{\ell^\infty}
\ge
\frac{C_\varphi-\varepsilon}{M_m},
\]
\[
\|p_m-u_m\|_{\mathrm{TV}}
\ge
\frac{C_\varphi-\varepsilon}{M_m},
\]
\[
\operatorname{Col}_m-\frac1{M_m}
\ge
\frac{2(C_\varphi-\varepsilon)^2}{M_m},
\]
\[
D_{\mathrm{KL}}(p_m\Vert u_m)
\ge
\frac{2(C_\varphi-\varepsilon)^2}{M_m^2}.
\]
因此，可见层在 \(\ell^\infty\)、总变差、碰撞缺口与相对熵四种标量上的首个不可消去量级，已经由共轭 Fibonacci 模对
\[
k\equiv \pm F_m \pmod{F_{m+2}}
\]
单独强制形成。
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-critical-resonance-constant}，
\[
a_m(F_m)\to C_\varphi,
\qquad
a_m(F_{m+1})\to C_\varphi.
\]
故当 \(m\) 充分大时，
\[
a_m(F_m)\ge C_\varphi-\varepsilon,
\qquad
a_m(F_{m+1})\ge C_\varphi-\varepsilon.
\]
再由 Parseval 恒等式
\[
\operatorname{Col}_m-\frac1{M_m}
=
\frac1{M_m}\sum_{k\ne 0}a_m(k)^2,
\]
便得到
\[
\operatorname{Col}_m-\frac1{M_m}
\ge
\frac{a_m(F_m)^2+a_m(F_{m+1})^2}{M_m}
\ge
\frac{2(C_\varphi-\varepsilon)^2}{M_m}.
\]

另一方面，令
\[
D_m:=\max_{r\in \mathbb Z/(M_m\mathbb Z)}d_m(r),
\qquad
\overline d_m:=\frac{2^m}{M_m}.
\]
定理 \ref{thm:conclusion-fold-golden-resonance-maxfiber-lift} 给出
\[
\liminf_{m\to\infty}\frac{M_m}{2^m}(D_m-\overline d_m)\ge C_\varphi.
\]
故对充分大的 \(m\)，有
\[
\frac{D_m-\overline d_m}{2^m}\ge \frac{C_\varphi-\varepsilon}{M_m}.
\]
取 \(r_m^\ast\) 使 \(d_m(r_m^\ast)=D_m\)，则
\[
\|p_m-u_m\|_{\ell^\infty}
\ge
p_m(r_m^\ast)-u_m(r_m^\ast)
=
\frac{D_m-\overline d_m}{2^m}
\ge
\frac{C_\varphi-\varepsilon}{M_m}.
\]
又因为总变差满足
\[
\|p_m-u_m\|_{\mathrm{TV}}
=
\sup_{A\subseteq \mathbb Z/(M_m\mathbb Z)}|p_m(A)-u_m(A)|,
\]
对单点集 \(A=\{r_m^\ast\}\) 取上确界便得
\[
\|p_m-u_m\|_{\mathrm{TV}}
\ge
p_m(r_m^\ast)-u_m(r_m^\ast)
\ge
\frac{C_\varphi-\varepsilon}{M_m}.
\]

最后，由 Pinsker 不等式
\[
D_{\mathrm{KL}}(p_m\Vert u_m)\ge 2\|p_m-u_m\|_{\mathrm{TV}}^2,
\]
遂有
\[
D_{\mathrm{KL}}(p_m\Vert u_m)
\ge
\frac{2(C_\varphi-\varepsilon)^2}{M_m^2}.
\]
证毕。
\end{proof}
