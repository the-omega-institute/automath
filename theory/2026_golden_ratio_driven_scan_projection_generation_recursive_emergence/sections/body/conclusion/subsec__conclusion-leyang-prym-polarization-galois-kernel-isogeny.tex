\subsection{Lee--Yang 子塔中的 Prym 三维体极化、模二伽罗瓦作用与同源核刻画}\label{subsec:conclusion-leyang-prym-polarization-galois-kernel-isogeny}
沿用定理 \ref{thm:xi-resolvent-dihedral-elliptic-birational}、定理 \ref{thm:xi-s3-fiberproduct-branch-parity-xa-xv-ed}、结论 \ref{con:xi-terminal-zm-prym-polarization-type-rigidity-11112} 与结论 \ref{con:xi-terminal-zm-discriminant-ridge-3isogeny-descent} 的记号。
本小节把 \(X_A\)、\(X_V\)、\(E_{\mathrm D}\) 三层对象上的 Prym 极化与有限核结构压缩为可直接调用的结论链。

\begin{theorem}[双覆盖 Prym 三维体的极化型、核与截面维数]\label{thm:conclusion-leyang-prym3-polarization-kernel-h0}
\leanverified{paper\_conclusion\_leyang\_prym3\_polarization\_kernel\_h0}
令
\[
\varphi:Y\to E_{\mathrm D}
\]
为定理 \ref{thm:xi-s3-fiberproduct-branch-parity-xa-xv-ed} 中的二次覆盖 \(q:X_V\to E_{\mathrm D}\)（即 \(Y:=X_V\)）。
定义
\[
B:=\ker\!\bigl(N_{\varphi}: \operatorname{Jac}(Y)\to \operatorname{Jac}(E_{\mathrm D})\bigr)^0,
\]
并记 \(\lambda_B\) 为 \(\operatorname{Jac}(Y)\) 主极化在 \(B\) 上的限制。
则：
\begin{enumerate}
  \item \(\deg(\lambda_B)=4\)，因而 \(\mathrm{type}(\lambda_B)=(1,1,2)\)；
  \item \(\ker(\lambda_B)=\varphi^*\!\bigl(E_{\mathrm D}[2]\bigr)\subset B\)（作为有限平坦群概形）；
  \item 若 \(L\) 为表示 \(\lambda_B\) 的充足线丛，则 \(h^0(B,L)=2\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-s3-fiberproduct-branch-parity-xa-xv-ed}，\(\varphi\) 为分歧二次覆盖且分歧点数为 \(6\)；
又由结论 \ref{con:xi-terminal-zm-s4-galois-closure-genus-tower-holomorphic-differentials} 可取 \(g(Y)=g(X_V)=4\)、\(g(E_{\mathrm D})=1\)，故 \(\dim B=3\)。

对分歧二次覆盖到椭圆基曲线的 Prym 极化，标准结构定理给出
\[
\ker(\lambda_B)=\varphi^*\!\bigl(E_{\mathrm D}[2]\bigr),\qquad
\deg(\lambda_B)=4,\qquad
h^0(B,L)=2.
\]
再由三维阿贝尔簇极化型 \((d_1,d_2,d_3)\) 满足
\[
\deg(\lambda_B)=(d_1d_2d_3)^2,\qquad d_1\mid d_2\mid d_3,
\]
代入 \(\deg(\lambda_B)=4\) 得 \(d_1d_2d_3=2\)，故唯一可能为 \((1,1,2)\)。
证毕。
\end{proof}

\begin{theorem}[模二伽罗瓦表示在 \texorpdfstring{$\ker(\lambda_B)$}{ker(lambdaB)} 上的满像]\label{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel}
设椭圆曲线 \(E_{\mathrm D}\) 取定理 \ref{thm:xi-resolvent-dihedral-elliptic-birational} 的模型
\[
E_{\mathrm D}:\ u^2=f(m),\qquad
f(m):=-4m^3-16m^2+16m+65.
\]
则
\[
\rho_{E_{\mathrm D},2}:G_{\QQ}\longrightarrow
\operatorname{Aut}\!\bigl(E_{\mathrm D}[2]\bigr)\cong \GL_2(\FF_2)
\]
为满射，且其像同构于 \(S_3\)。
结合定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0} 的
\(
\ker(\lambda_B)=\varphi^*(E_{\mathrm D}[2])
\)，
得到 \(G_{\QQ}\) 在 \(\ker(\lambda_B)\) 上的作用像亦为 \(\GL_2(\FF_2)\cong S_3\)。
\end{theorem}
\begin{proof}
椭圆模型 \(u^2=f(m)\) 的非零 \(2\)-挠点与 \(f\) 的三根一一对应，故
\[
\operatorname{Gal}\!\bigl(\QQ(E_{\mathrm D}[2])/\QQ\bigr)
\cong
\operatorname{Gal}\!\bigl(\text{\(f\) 的分裂域}/\QQ\bigr).
\]
直接计算：
\[
f(0)\equiv 2,\quad f(1)\equiv 1,\quad f(2)\equiv 1\pmod 3,
\]
故 \(f\) 在 \(\FF_3\) 上无根，从而在 \(\QQ\) 上不可约。
又
\[
\operatorname{Disc}(f)=2^4\cdot 31^2\cdot 37
\]
不是平方，因此不可约三次多项式 \(f\) 的分裂域伽罗瓦群为 \(S_3\)。
由 \(\GL_2(\FF_2)\cong S_3\)（且阶同为 \(6\)）可得 \(\rho_{E_{\mathrm D},2}\) 满射。
最后利用定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0} 的核识别即得结论。
证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$(B,\lambda_B)$}{(B,lambdaB)} 不存在 \texorpdfstring{$\QQ$}{Q}-定义的 \(2\)-同源主极化化改造]\label{cor:conclusion-leyang-prym3-no-q-principalization-degree2}
在定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0} 与定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel} 的设定下，
\(\ker(\lambda_B)\cong (\ZZ/2\ZZ)^2\) 的三个阶 \(2\) 子群在 \(G_{\QQ}\) 作用下构成单轨道，因而无 \(\QQ\)-定义者。
特别地，不存在定义于 \(\QQ\) 的度 \(2\) 同源
\[
(B,\lambda_B)\dashrightarrow (A,\lambda_A)
\]
使 \(\lambda_A\) 为主极化。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel}，\(G_{\QQ}\) 在
\(\ker(\lambda_B)\setminus\{0\}\) 上的作用等同于 \(S_3\) 在三点上的标准传递作用。
每个阶 \(2\) 子群由其唯一非零元生成，故若某阶 \(2\) 子群在 \(\QQ\) 上定义，则其非零元应为 \(G_{\QQ}\)-不动点，与传递性矛盾。
而 \((1,1,2)\)-极化的度 \(2\) 主极化化需由 \(\ker(\lambda_B)\) 中的 \(\QQ\)-定义极大各向同性子群作商，故此路径在 \(\QQ\) 上不存在。
证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$(B,\lambda_B)$}{(B,lambdaB)} 的对角拼接主极化六维商]\label{thm:conclusion-leyang-prym3-diagonal-gluing-principal-sixfold}
沿用定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0} 的 \(B/\QQ\) 与 \((1,1,2)\)-极化 \(\lambda_B\)，并令
\[
V:=\ker(\lambda_B)\subseteq B[2].
\]
记 \(e_{\lambda_B}\) 为 \(\lambda_B\) 在 \(V\) 上诱导的交换子配对（取值于 \(\mu_2=\{\pm1\}\)），并将其外积到 \(V\times V\) 上得到 \(\lambda_B\boxtimes \lambda_B\) 的配对 \(e_{\lambda_B\boxtimes\lambda_B}\)。
则存在主极化阿贝尔六维簇 \((P_B,\Theta_B)\) 与定义于 \(\QQ\) 的同源
\[
q:\ B\times B\longrightarrow P_B
\]
满足：
\begin{enumerate}
  \item \(\ker(q)=\Delta(V):=\{(v,v):\ v\in V\}\subset V\times V\)；
  \item \(q^{\ast}\Theta_B\sim \lambda_B\boxtimes \lambda_B\)。
\end{enumerate}
并且若 \(\Gamma\subset V\times V\) 为任何定义于 \(\QQ\) 的极大各向同性子群（相对于 \(e_{\lambda_B\boxtimes\lambda_B}\)；等价地 \(|\Gamma|=4\)），则必有 \(\Gamma=\Delta(V)\)。
因此上述 \((P_B,\Theta_B,q)\) 在 \(\QQ\)-同构意义下唯一。
\end{theorem}
\begin{proof}
由 \(\mu_2\) 上的取值，\(e_{\lambda_B}(v,w)^2=1\) 对任意 \(v,w\in V\) 成立，故
\[
e_{\lambda_B\boxtimes\lambda_B}\bigl((v,v),(w,w)\bigr)
=e_{\lambda_B}(v,w)\,e_{\lambda_B}(v,w)
=1.
\]
于是 \(\Delta(V)\) 为 \(V\times V\) 的各向同性子群。
又 \(|V|=4\)，故 \(|V\times V|=16\)，而 \(\Delta(V)\) 的阶为 \(4\)，从而 \(\Delta(V)\) 为极大各向同性。
\par
对任意极化 \(\lambda\) 与其核中的极大各向同性子群 \(K\)，极化下降理论给出：商映射 \(\pi:A\to A/K\) 上存在唯一的主极化 \(\Theta\) 使 \(\pi^{\ast}\Theta\sim \lambda\)。
将该标准构造应用于 \(A=B\times B\)、\(\lambda=\lambda_B\boxtimes\lambda_B\) 与 \(K=\Delta(V)\)，得到存在性。
\par
下面证 \(\QQ\)-定义的极大各向同性子群唯一性。
设 \(\Gamma\subset V\times V\) 为 \(e_{\lambda_B\boxtimes\lambda_B}\)-极大各向同性，且在 \(G_{\QQ}\) 作用下不变。
记 \(\mathrm{pr}_1,\mathrm{pr}_2\) 为两个坐标投影。
由于 \(G_{\QQ}\) 在 \(V\times V\) 上按对角作用，\(\mathrm{pr}_1(\Gamma)\subseteq V\) 与 \(\mathrm{pr}_2(\Gamma)\subseteq V\) 都是 \(G_{\QQ}\)-稳定的 \(\FF_2\)-子空间。
由定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel}，\(G_{\QQ}\) 在 \(V\) 上的作用像为 \(\GL_2(\FF_2)\)，故 \(V\) 的唯一非平凡稳定子空间只能是 \(V\) 自身。
另一方面，若 \(\mathrm{pr}_1(\Gamma)=0\)，则 \(\Gamma\subseteq \{0\}\times V\)，而在 \(\{0\}\times V\) 上配对退化为 \(e_{\lambda_B}\)，其各向同性子空间维数至多为 \(1\)，与 \(\dim_{\FF_2}\Gamma=2\) 矛盾。
故 \(\mathrm{pr}_1(\Gamma)=V\)。
同理 \(\mathrm{pr}_2(\Gamma)=V\)。
由 \(\dim_{\FF_2}\Gamma=\dim_{\FF_2}V=2\)，两投影限制 \(\mathrm{pr}_i|_{\Gamma}:\Gamma\to V\) 皆为同构。
于是存在唯一线性同构 \(f:V\to V\) 使
\[
\Gamma=\{(v,f(v)):\ v\in V\}.
\]
将该表达式代入各向同性条件得 \(f\in\operatorname{Aut}(V)\) 且保持配对 \(e_{\lambda_B}\)（在 \(\mu_2\) 上等价于 \(\mathrm{Sp}(V,e_{\lambda_B})=\GL_2(\FF_2)\)）。
\par
由定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel}，\(G_{\QQ}\) 在 \(V\) 上的作用像为 \(\GL_2(\FF_2)\)，并在 \(V\setminus\{0\}\) 上传递。
又因 \(\Gamma\) 为 \(\QQ\)-定义，故对任意 \(\sigma\in G_{\QQ}\) 与任意 \(v\in V\) 有
\[
(\sigma v,\,\sigma f(v))\in\Gamma
\quad\Longrightarrow\quad
\sigma f(v)=f(\sigma v),
\]
即 \(f\) 与 \(G_{\QQ}\) 的像可交换。
若 \(f\neq 0\)，取 \(v\neq 0\) 并设 \(f(v)=w\neq 0\)。
若 \(w\neq v\)，取 \(g\in \GL_2(\FF_2)\) 固定 \(v\) 而交换 \(V\setminus\{0,v\}\) 的两点，则
\[
f(v)=f(gv)=g f(v)=g w\neq w,
\]
矛盾；故必有 \(f(v)=v\)。
再由传递性，对任意 \(u\neq 0\) 存在 \(h\) 使 \(h(v)=u\)，从而
\(
f(u)=f(hv)=h f(v)=hv=u
\)。
于是 \(f=\mathrm{id}_V\)，从而 \(\Gamma=\Delta(V)\)。
\par
最后，既然唯一的 \(\QQ\)-定义极大各向同性子群为 \(\Delta(V)\)，则由极化下降的唯一性得到 \((P_B,\Theta_B,q)\) 在 \(\QQ\)-同构意义下唯一。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$A_4$}{A4} 商曲线的属 \(2\) 超椭圆模型]\label{thm:conclusion-leyang-xa-genus2-hyperelliptic-model}
\leanverified{paper\_conclusion\_leyang\_xa\_genus2\_hyperelliptic\_model}
设 \(X\to \PP^1_y\) 为 Lee--Yang \(S_4\)-Galois 覆盖，分歧值为
\(
y=0,1,\infty
\)
与三次多项式
\(
g(y)=256y^3+411y^2+165y+32
\)
的三根。
记 \(X_A:=X/A_4\)。
则 \(X_A\to \PP^1_y\) 为二次覆盖，且 \(X_A\) 为属 \(2\) 超椭圆曲线，并可写为
\[
X_A:\ t^2=c\,y(y-1)g(y),\qquad c\in\QQ^\times.
\]
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-s3-fiberproduct-branch-parity-xa-xv-ed} 已给出
\[
X_A:\ t^2=-y(y-1)g(y),
\]
且分歧支撑恰对应六个点
\(
\{0,1,\infty\}\cup\{g(y)=0\}
\)。
故 \(X_A\to\PP^1_y\) 为二次覆盖且总分歧点数为 \(6\)。
由 Riemann--Hurwitz 公式
\[
2g(X_A)-2=2(-2)+6
\]
得 \(g(X_A)=2\)。
将 \(t\) 作有理比例缩放即可把常数写成任意固定 \(c\in\QQ^\times\)。
证毕。
\end{proof}

\begin{proposition}[三次无分歧层诱导的 \texorpdfstring{$\QQ$}{Q}-稳定 \(3\)-挠子群与下降同源]\label{prop:conclusion-leyang-xa-c3-subgroup-descent}

则 \(\operatorname{Jac}(X_A)[3]\) 中存在非平凡 \(G_{\QQ}\)-稳定阶 \(3\) 子群方案
\[
G_3\subset \operatorname{Jac}(X_A)[3],\qquad \#G_3=3,
\]
从而存在定义于 \(\QQ\) 的度 \(3\) 同源
\[
\psi:\ \operatorname{Jac}(X_A)\longrightarrow \operatorname{Jac}(X_A)/G_3.
\]
\end{proposition}
\begin{proof}
定理 \ref{thm:xi-s3-fiberproduct-branch-parity-xa-xv-ed} 给出 \(p\) 为循环 \(C_3\)-étale 覆盖；
结论 \ref{con:xi-terminal-zm-discriminant-ridge-3isogeny-descent} 在同一属 \(2\) 对象上已构造出

\tau\in \operatorname{Jac}(X_A)\!\bigl(\QQ(\sqrt{-3})\bigr)[3]\setminus\{0\},
\]
且 \(G_3:=\langle\tau\rangle\) 在 \(G_{\QQ}\) 作用下稳定。
对阿贝尔簇，任何 \(G_{\QQ}\)-稳定有限子群方案均可下降到基域上的商同源，
故得到所述 \(\QQ\)-定义度 \(3\) 同源 \(\psi\)。
证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\operatorname{Jac}(Y)$}{Jac(Y)} 与 \texorpdfstring{$E_{\mathrm D}\times B$}{ED x B} 的度 \(4\) 同源及图形核]\label{prop:conclusion-leyang-jacy-ed-b-degree4-graph-kernel}
在定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0} 记号下，定义
\[
\alpha:\ E_{\mathrm D}\times B\longrightarrow \operatorname{Jac}(Y),\qquad
(a,b)\longmapsto \varphi^*a+b.
\]
则 \(\alpha\) 为度 \(4\) 同源，且
\[
\ker(\alpha)=\{(x,-\varphi^*x):\ x\in E_{\mathrm D}[2]\}.
\]
特别地，\(\ker(\alpha)\cong E_{\mathrm D}[2]\)（作为 \(G_{\QQ}\)-模）。
\end{proposition}
\begin{proof}
若 \(\alpha(a,b)=0\)，则 \(\varphi^*a=-b\in B\)。
两边施加范数映射并用恒等式 \(N_{\varphi}\circ \varphi^*=[2]\) 得
\[
2a=0,
\]
故 \(a\in E_{\mathrm D}[2]\)。
反之，若 \(x\in E_{\mathrm D}[2]\)，则
\[
N_{\varphi}(\varphi^*x)=2x=0,
\]
故 \(\varphi^*x\in B\)，且 \(\alpha(x,-\varphi^*x)=0\)。
因此核即为所示图形子群。

由 \(\#E_{\mathrm D}[2]=4\) 知 \(\ker(\alpha)\) 有限且阶 \(4\)；
又 \(\dim(E_{\mathrm D}\times B)=1+3=4=\dim\operatorname{Jac}(Y)\)，故 \(\alpha\) 为度 \(4\) 同源。
证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$\operatorname{Jac}(X_V)$}{Jac(XV)} 的 \texorpdfstring{$3^4$}{3^4} 度粘合与 \texorpdfstring{$\operatorname{Jac}(X_A)[3]$}{Jac(XA)[3]} 图形核]\label{prop:conclusion-leyang-jacxv-jacxa-prym-degree81-graph-kernel}
记
\[
P:=\ker\!\bigl(N_p:\operatorname{Jac}(X_V)\to \operatorname{Jac}(X_A)\bigr)^0,
\]
并定义
\[
\beta:\ \operatorname{Jac}(X_A)\times P\longrightarrow \operatorname{Jac}(X_V),\qquad
(a,b)\longmapsto p^*a+b.
\]
则 \(\beta\) 为度 \(3^4=81\) 同源，且
\[
\ker(\beta)=\{(x,-p^*x):\ x\in \operatorname{Jac}(X_A)[3]\}.
\]
\end{proposition}
\begin{proof}
由范数与拉回的基本关系
\[
N_p\circ p^*=[3],
\]
若 \(\beta(a,b)=0\)，则 \(p^*a=-b\in P\)，故
\[
3a=N_p(p^*a)=0,
\]
即 \(a\in \operatorname{Jac}(X_A)[3]\)。
反向同理：若 \(x\in \operatorname{Jac}(X_A)[3]\)，则 \(N_p(p^*x)=3x=0\)，从而 \(p^*x\in P\)，并有 \(\beta(x,-p^*x)=0\)。
故核为所示图形子群。

由定理 \ref{thm:conclusion-leyang-xa-genus2-hyperelliptic-model}，\(g(X_A)=2\)，故
\[
\#\operatorname{Jac}(X_A)[3]=3^{2g(X_A)}=3^4.
\]
因此 \(\deg(\beta)=3^4=81\)。
证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$(B,\lambda_B)$}{(B,lambdaB)} 主极化化的最小定义域与唯一性]\label{thm:conclusion-leyang-prym3-principalization-minimal-field}
沿用定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0} 与定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel} 的记号。设
\[
\mathfrak f(m):=4m^3+16m^2-16m-65,
\]
并令 \(m_1,m_2,m_3\) 为 \(\mathfrak f\) 在代数闭包中的三根，\(K_i:=\QQ(m_i)\)。
则：
\begin{enumerate}
  \item 对每个 \(i\in\{1,2,3\}\)，存在且仅存在（同构意义下）一个主极化阿贝尔三维体 \((A_i,\lambda_i)\) 与度 \(2\) 同源
  \[
  \pi_i:(B,\lambda_B)_{K_i}\to (A_i,\lambda_i),
  \qquad
  \pi_i^*\lambda_i=\lambda_B.
  \]
  \item 若数域 \(F/\QQ\) 上存在
  \[
  \pi:(B,\lambda_B)_F\to (A,\lambda),
  \qquad
  \deg\pi=2,\ \lambda\ \text{主极化},\ \pi^*\lambda=\lambda_B,
  \]
  则必有 \(K_i\subseteq F\) 对某个 \(i\) 成立。特别地，满足该性质的最小扩张次数为 \(3\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-prym3-polarization-kernel-h0}，\(\lambda_B\) 型为 \((1,1,2)\)，故
\[
\ker(\lambda_B)\cong (\ZZ/2\ZZ)^2.
\]
对 \((1,1,2)\)-极化，度 \(2\) 主极化化等价于在 \(\ker(\lambda_B)\) 中选取一个阶 \(2\) 极大各向同性子群；而在 \((\ZZ/2\ZZ)^2\) 中此类子群与非零元一一对应。

再由定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel}，
\[
\ker(\lambda_B)\cong E_{\mathrm D}[2],
\]
且 \(E_{\mathrm D}\) 的非零 \(2\)-挠点恰对应 \(\mathfrak f(m)=0\) 的根。
因此，在域 \(K\) 上存在度 \(2\) 主极化化当且仅当 \(E_{\mathrm D}[2](K)\) 含非零点，当且仅当 \(\mathfrak f\) 在 \(K\) 上有根。

对 \(K_i=\QQ(m_i)\)，根 \(m_i\) 显然在 \(K_i\) 中，故存在相应主极化化。
又由定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel} 的证明，\(\mathfrak f\) 在 \(\QQ\) 上不可约且其分裂域伽罗瓦群为 \(S_3\)，故每个三次数域 \(K_i\) 中恰有一个根；于是 \(E_{\mathrm D}[2](K_i)\setminus\{0\}\) 恰一元素，对应唯一阶 \(2\) 子群，故主极化化唯一。得 (1)。

若 \(F\) 上存在所述 \(\pi\)，则 \(\ker(\pi)\subset\ker(\lambda_B)\) 为 \(F\)-有理阶 \(2\) 子群，故 \(E_{\mathrm D}[2](F)\) 含非零点，等价于 \(\mathfrak f\) 在 \(F\) 上有根，即 \(m_i\in F\) 对某个 \(i\) 成立，故 \(K_i\subseteq F\)。得 (2) 的前半。
又 \(\mathfrak f\) 不可约，故 \([K_i:\QQ]=3\)；结合推论 \ref{cor:conclusion-leyang-prym3-no-q-principalization-degree2} 排除 \(F=\QQ\)，最小扩张次数即为 \(3\)。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$(B,\lambda_B)$}{(B,lambdaB)} 主极化化参数空间的三点有限 \'{e}tale 表示]\label{prop:conclusion-leyang-prym3-principalization-moduli-threepoint-etale}
定义函子 \(\mathscr M_B\)：对任意 \(\QQ\)-代数 \(R\)，\(\mathscr M_B(R)\) 为 \((B,\lambda_B)_R\) 的度 \(2\) 主极化化数据集合（等价地，\(\ker(\lambda_B)_R\) 中阶 \(2\) 极大各向同性子群集合）。
则 \(\mathscr M_B\) 由有限 \'{e}tale \(\QQ\)-概形表示，且有自然同构
\[
\mathscr M_B\cong E_{\mathrm D}[2]\setminus\{0\}
\cong \Spec\!\bigl(\QQ[m]/(\mathfrak f(m))\bigr),
\qquad
\mathfrak f(m)=4m^3+16m^2-16m-65.
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-prym3-principalization-minimal-field} 的证明，主极化化数据与 \(\ker(\lambda_B)\setminus\{0\}\) 一一对应，故
\[
\mathscr M_B\simeq \ker(\lambda_B)\setminus\{0\}.
\]
再由 \(\ker(\lambda_B)\cong E_{\mathrm D}[2]\) 得第一同构。
在模型 \(E_{\mathrm D}:u^2=-\mathfrak f(m)\) 下，非零 \(2\)-挠点满足 \(u=0\) 且 \(\mathfrak f(m)=0\)，故得到第二同构。
此外 \(\Disc(\mathfrak f)=2^4\cdot 31^2\cdot 37\neq 0\)（定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel}），故该三点概形在 \(\QQ\) 上有限 \'{e}tale。证毕。
\end{proof}

\begin{theorem}[两条本征 \texorpdfstring{$S_3$}{S3} 扩张的线性无交与直积伽罗瓦群]\label{thm:conclusion-leyang-fg-splitting-linear-disjoint-s3xs3}
定义
\[
K_f:=\QQ\bigl(\text{\(\mathfrak f(m)=0\) 的分裂域}\bigr),
\qquad
\mathfrak f(m)=4m^3+16m^2-16m-65,
\]
\[
K_g:=\QQ\bigl(\text{\(g(y)=0\) 的分裂域}\bigr),
\qquad
g(y)=256y^3+411y^2+165y+32.
\]
则：
\begin{enumerate}
  \item \(\Gal(K_f/\QQ)\cong S_3\)，且其唯一二次子域为 \(\QQ(\sqrt{37})\)；
  \item \(\Gal(K_g/\QQ)\cong S_3\)，且其唯一二次子域为 \(\QQ(\sqrt{-111})\)；
  \item \(K_f\cap K_g=\QQ\)，并且
  \[
  \Gal(K_fK_g/\QQ)\cong S_3\times S_3.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
对 \(\mathfrak f\)，定理 \ref{thm:conclusion-leyang-ed2-galois-image-on-prym-kernel} 已给出其在 \(\QQ\) 上不可约且分裂域伽罗瓦群为 \(S_3\)，并有
\[
\Disc(\mathfrak f)=2^4\cdot 31^2\cdot 37.
\]
对 \(g\)，可取模 \(5\) 检验：\(g\) 在 \(\FF_5\) 上无根，故在 \(\QQ\) 上不可约；再由三次判别式公式得
\[
\Disc(g)=-3^9\cdot 31^2\cdot 37,
\]
非平方，故 \(\Gal(K_g/\QQ)\cong S_3\)。

对 \(S_3\)-伽罗瓦扩张，唯一二次子域由判别式平方类给出：
\(\mathfrak f\) 的平方自由部分为 \(37\)，\(g\) 的平方自由部分为 \(-3\cdot 37=-111\)。
故二次子域分别为 \(\QQ(\sqrt{37})\)、\(\QQ(\sqrt{-111})\)，且二者不同。

由于 \(K_f,K_g\) 均为 Galois，交域 \(K_f\cap K_g\) 亦为 Galois。
在 \(S_3\)-扩张中，真 Galois 子域仅可能为唯一二次子域；二次子域既不同，故交域只能是 \(\QQ\)。
于是 \(K_f,K_g\) 线性无交，得到
\[
\Gal(K_fK_g/\QQ)\cong \Gal(K_f/\QQ)\times \Gal(K_g/\QQ)\cong S_3\times S_3.
\]
证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\mathfrak f$}{f} 与 \texorpdfstring{$g$}{g} 在素数处分解型的独立性与精确密度]\label{thm:conclusion-leyang-fg-splitting-type-independent-density}
设素数 \(p\nmid 2\cdot 3\cdot 31\cdot 37\)。
以三次分解型记号
\[
(1,1,1),\ (1,2),\ (3)
\]
分别表示完全分裂、一次因子乘不可约二次、不可约三次。
则 \(\mathfrak f\) 与 \(g\) 在 \(p\) 处的分解型独立，联合 Dirichlet 密度为
\[
\begin{array}{c|ccc}
& g:(1,1,1) & g:(1,2) & g:(3)\\ \hline
\mathfrak f:(1,1,1) & \frac1{36} & \frac1{12} & \frac1{18}\\
\mathfrak f:(1,2)   & \frac1{12} & \frac1{4}  & \frac1{6}\\
\mathfrak f:(3)     & \frac1{18} & \frac1{6}  & \frac1{9}
\end{array}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-fg-splitting-linear-disjoint-s3xs3}，
\[
\Gal(K_fK_g/\QQ)\cong S_3\times S_3.
\]
Chebotarev 定理给出：未分歧素数的 Frobenius 共轭类按类大小除以群阶分布。
在 \(S_3\) 中：
恒等类大小 \(1\) 对应 \((1,1,1)\)；
换位类大小 \(3\) 对应 \((1,2)\)；
三循环类大小 \(2\) 对应 \((3)\)。
故单边密度分别为 \(1/6,1/2,1/3\)。
直积群中共轭类大小相乘，故联合密度为上述单边密度乘积，表格即得。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$(B_p,\lambda_{B_p})$}{(Bp,lambdaBp)} 的主极化化出现频率与计数]\label{cor:conclusion-leyang-prym3-finitefield-principalization-frequency}
设 \(p\nmid 2\cdot 3\cdot 31\cdot 37\)，并假定 \((B,\lambda_B)\) 在 \(\FF_p\) 上良好约化为 \((B_p,\lambda_{B_p})\)。
则：
\begin{enumerate}
  \item \((B_p,\lambda_{B_p})\) 存在 \(\FF_p\)-定义的度 \(2\) 主极化化，当且仅当 \(\mathfrak f\) 在 \(\FF_p\) 上有一次因子；其素数密度为
  \[
  \frac16+\frac12=\frac23.
  \]
  \item 主极化化个数等于 \(\mathfrak f\) 在 \(\FF_p\) 上的根数：
  \[
  \#\mathrm{Prin}_2(B_p,\lambda_{B_p})=
  \begin{cases}
  3,&\mathfrak f\ \text{型 }(1,1,1),\\
  1,&\mathfrak f\ \text{型 }(1,2),\\
  0,&\mathfrak f\ \text{型 }(3).
  \end{cases}
  \]
  对应密度分别为 \(1/6,1/2,1/3\)。
  \item 若再要求 \(g\) 在 \(\FF_p\) 上不可约，则“存在主极化化且 \(g\) 不可约”的联合密度为
  \[
  \frac23\cdot\frac13=\frac29.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由命题 \ref{prop:conclusion-leyang-prym3-principalization-moduli-threepoint-etale}，
\(\mathscr M_B\cong \Spec(\QQ[m]/(\mathfrak f(m)))\)。
约化到 \(\FF_p\) 后，\(\FF_p\)-主极化化数据与 \(\bar{\mathfrak f}\) 的 \(\FF_p\)-根一一对应，故存在性与计数结论立得。
密度由定理 \ref{thm:conclusion-leyang-fg-splitting-type-independent-density} 的边际分布与联合分布直接给出；最后一条由表中
\((\mathfrak f:(1,1,1),g:(3))\) 与 \((\mathfrak f:(1,2),g:(3))\) 两格相加即得。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$X_H\to E_{\mathrm D}$}{XH->ED} 的 Prym 椭圆因子与同源辨识]\label{thm:conclusion-leyang-xh-ed-prym-elliptic-isogeny}
令
\[
H=\langle\delta,\tau\rangle\le D_8,\qquad
X_H:=X/H,\qquad
\pi_H:X_H\to E_{\mathrm D}
\]
如结论 \ref{cor:xi-terminal-zm-s4-v4-three-quad-subcovers-branch-parity} 所述。
定义
\[
P_H:=\ker\!\bigl(N_{\pi_H}:\Jac(X_H)\to E_{\mathrm D}\bigr)^0.
\]
则：
\begin{enumerate}
  \item \(P_H\) 为一维阿贝尔簇，故为椭圆曲线；
  \item 在 \(\QQ\) 上有同源
  \[
  P_H\sim_{\QQ}E,
  \]
  其中 \(E\) 为结论 \ref{con:xi-terminal-zm-s4-xh-jacobian-splitting-ed-e} 中出现的椭圆因子；
  \item \(P_H\) 上由 \(\Jac(X_H)\) 限制得到的 Prym 极化为 \((2)\)-型，故极化次数为 \(4\)。
\end{enumerate}
\end{theorem}
\begin{proof}
由结论 \ref{cor:xi-terminal-zm-s4-v4-three-quad-subcovers-branch-parity}，\(g(X_H)=2\) 且 \(g(E_{\mathrm D})=1\)，故
\[
\dim P_H=g(X_H)-g(E_{\mathrm D})=1,
\]
得 (1)。

对二次覆盖 \(\pi_H\) 的范数映射，标准 Prym 构造给出
\[
\Jac(X_H)\sim_{\QQ}E_{\mathrm D}\times P_H.
\]
另一方面，结论 \ref{con:xi-terminal-zm-s4-xh-jacobian-splitting-ed-e} 已给出
\[
\Jac(X_H)\sim_{\QQ}E\times E_{\mathrm D}.
\]
比较同源分解即得 \(P_H\sim_{\QQ}E\)，得 (2)。

最后，\(\pi_H\) 为到椭圆基曲线的分歧二次覆盖（分支次数 \(2\)），其 Prym 极化在一维情形为 \((2)\)-型；对椭圆曲线，\((2)\)-型极化次数为 \(2^2=4\)。得 (3)。证毕。
\end{proof}

\end{remark}
