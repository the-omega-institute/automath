\begin{theorem}[实现压缩与同态线性化的严格分叉]\label{thm:conclusion-implementation-compression-hom-linearization-split}
\leanverified{paper\_conclusion\_implementation\_compression\_hom\_linearization\_split}
在实现层与结构层半圆维口径下，有精确等式
\[
\cdim^{\mathrm{hom}}_{+}(\NN,+)=\frac12,
\qquad
\cdim^{\mathrm{hom}}_{+}(\NN_{\times})=+\infty,
\qquad
\cdim^{\mathrm{impl}}(\NN_{\times})=\frac12.
\]
因此，“可被单寄存器实现”与“可被有限加法账本忠实线性化”是两个彼此独立的性质：前者只测集合论单射，后者测乘法结构能否被压入有限生成的交换可消去加法对象。特别地，不存在任何有限生成交换可消去加法幺半群可作为 \(\NN_{\times}\) 的忠实对数账本。
\end{theorem}
\begin{proof}
前两条等式与最后一条等式正是命题 \ref{prop:killo-implementation-vs-structural-half-circle-dimension} 的内容。另一方面，定理 \ref{thm:killo-prime-freedom-non-finitizability} 已排除
\[
\NN_{\times}\hookrightarrow M
\]
对任意有限生成交换可消去幺半群 \(M\) 的单射幺半群同态，而定理 \ref{thm:killo-godel-compression-not-finite-rank-homologizable} 进一步排除了所有有限生成 \(\ZZ\oplus L\) 型加法账本上的忠实同态化。故实现层单寄存器可编址与结构层有限账本可线性化之间存在严格分叉。证毕。
\end{proof}

\begin{theorem}[有限素局部化的一圆商与全不连通核分解]\label{thm:conclusion-finite-localization-circlequotient-profinite-kernel}
设 \(S\subset \mathbb P\) 为有限素数集，并记
\[
G_S:=\ZZ[S^{-1}],
\qquad
\Sigma_S:=\widehat{G_S},
\qquad
P_S:=\prod_{p\in S}\ZZ_p.
\]
则存在短正合列
\[
0\longrightarrow P_S\longrightarrow \Sigma_S\longrightarrow \TT\longrightarrow 0,
\qquad
\cdim_{\mathrm{fr}}(G_S)=1.
\]
并且对任意两个有限素数集 \(S,T\subset \mathbb P\)，都有
\[
\Sigma_S/P_S\cong \TT \cong \Sigma_T/P_T,
\]
但同时
\[
\Sigma_S\cong \Sigma_T
\iff
S=T.
\]
换言之，所有有限素局部化共享同一个一维连续相位商，而全部算术特异性都被压入全不连通的 \(p\)-进核 \(P_S\) 中。
\leanverified{paper\_conclusion\_finite\_localization\_circlequotient\_profinite\_kernel}
\end{theorem}
\begin{proof}
短正合列与 \(\cdim_{\mathrm{fr}}(G_S)=1\) 由定理 \ref{thm:conclusion-finite-prime-solenoid-terminal-object} 给出；其中特别说明 \(\Sigma_S\) 的任意非平凡连通紧 Lie 商都同构于 \(\TT\)。因此
\[
\Sigma_S/P_S\cong \TT
\]
对每个有限 \(S\) 都成立。另一方面，定理 \ref{thm:conclusion-finite-prime-localization-pontryagin-rigidity} 断言
\[
\Sigma_S\cong \Sigma_T\iff S=T,
\]
故连续圆商并不完备，真正完成分类的全部信息都滞留在全不连通核的 \(p\)-主因子中。证毕。
\end{proof}

\begin{corollary}[局部化数系的圆维坍缩与素支撑完备]\label{cor:conclusion-finite-localization-circle-dimension-collapse-profinite-support-completeness}
\leanverified{paper\_conclusion\_finite\_localization\_circle\_dimension\_collapse\_profinite\_support\_completeness}

在有限秩扩展口径下都满足同一不变量

\[
P_S=\prod_{p\in S}\ZZ_p
\]
却完全恢复素支撑：
\[
P_S\cong P_T
\iff
S=T.
\]
因此，连通圆因子把全部有限素局部化压缩到同一“圆维 \(1\)”类，而 profinite 核则把全部坏素支撑逐点记回。
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-finite-localization-circlequotient-profinite-kernel} 已经同时给出

\cdim_{\mathrm{fr}}(G_S)=1
\]
以及

\Sigma_S\cong \Sigma_T\iff S=T.
\]
而 \(P_S=\prod_{p\in S}\ZZ_p\) 的 \(p\)-主因子非平凡当且仅当 \(p\in S\)，故 \(P_S\) 自身已经唯一决定 \(S\)。证毕。
\end{proof}

\begin{corollary}[有限 prime 支撑、Pontryagin 对偶与圆维失明的严格分离]\label{cor:conclusion-finite-primesupport-pontryagin-rigidity-circle-dimension-blindness}
对任意有限素数集 \(S,T\subset \mathbb P\)，下列条件等价：

G_S\cong G_T,
\qquad
\widehat{G_S}\cong \widehat{G_T},
\qquad
S=T.
\]
特别地，族

\{G_S:S\subset \mathbb P,\ |S|<\infty\}

由两两非同构的有限秩离散交换群组成；然而对每个有限 \(S\)，它们又都满足同一圆维不变量

\cdim_{\mathrm{fr}}(G_S)=1.

因此，圆维只记录唯一连通圆因子的秩，而对有限 prime ledger 的组合型支撑完全失明。
\end{corollary}
\begin{proof}
第一组等价正是定理 \ref{thm:conclusion-finite-prime-localization-pontryagin-rigidity} 的内容：Pontryagin 对偶在离散交换群与紧交换群之间给出反变等价，而有限素支撑 \(S\) 又可由局部化群本身唯一恢复。第二条则由推论 \ref{cor:conclusion-finite-localization-circle-dimension-collapse-profinite-support-completeness} 立即给出：对所有有限 \(S\) 都有
\[
\cdim_{\mathrm{fr}}(G_S)=1,
\]
但其 profinite 核与 Pontryagin 对偶同构类型仍然完全记住 \(S\)。证毕。
\end{proof}

\begin{theorem}[Lee--Yang 分支塔的五片 dyadic 覆盖]\label{thm:conclusion-leyang-five-sheet-dyadic-cover}

\qquad
\deg R(y_n)=5\cdot 4^n.

\mathcal R_\infty:=\varprojlim_n R(y_n)

\bigsqcup_{i=0}^{4}T_2(E_{\min})
\cong
\bigsqcup_{i=0}^{4}\ZZ_2^2.

换言之，Lee--Yang 分支塔的极限对象并不是单一 \(2\)-进地址面，而是其五片不相交拷贝的有限覆盖；连续基底由 \(T_2(E_{\min})\) 承载，而分支多重性则被五值 sheet 变量精确保留。

\begin{proof}
五分片分解与分支度公式是定理 \ref{thm:xi-time-part62d-leyang-branch-inverse-limit-five-torsors} 及其上游的有限层地址分解的直接内容。再结合推论 \ref{cor:killo-2adic-tate-hypercube-limit} 对 \(T_2(E_{\min})\cong\ZZ_2^2\) 的识别，便得到所示逆极限分解。由于过渡映射在每个分片上都由 \([2]\) 诱导，故其坐标形式正是最高位截断。证毕。
\end{proof}

\begin{theorem}[Lee--Yang 全息地址的规范离散--连续分裂]\label{thm:conclusion-leyang-canonical-address-discrete-continuous-splitting}
\leanverified{paper\_conclusion\_leyang\_canonical\_address\_discrete\_continuous\_splitting}

\[
\mathcal R_\infty:=\varprojlim_n R(y_n)
\]
为 Lee--Yang 分支塔的逆极限。对每个有限层 \(n\)，定义规范地址
\[
\operatorname{Addr}_n(y):=(i,\xi_n),
\qquad
y\in P_i+E_{\min}[2^n],
\qquad
\xi_n:=y-P_i\in E_{\min}[2^n].
\]
则 \(\operatorname{Addr}_n\) 与层间截断相容，并在逆极限上诱导规范双射
\[
\operatorname{Addr}_{\mathrm{can}}:\mathcal R_\infty\xrightarrow{\sim}\{0,1,2,3,4\}\times T_2(E_{\min}).
\]
因此，Lee--Yang 的完整全息地址严格分解为一个五值离散 sheet 寄存器与一个连续的 \(2\)-进 Tate 坐标；此处分裂不是集合论偶然，而是由有限层五支平行族与乘二截断相容性共同强迫的规范因子化。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-five-sheet-dyadic-cover}，每一层都有严格分解

R(y_n)=\bigsqcup_{i=0}^{4}\bigl(P_i+E_{\min}[2^n]\bigr),

且五个分支彼此不交。
故对任意 \(y\in R(y_n)\)，存在唯一的 \(i\in\{0,\dots,4\}\) 与唯一的 \(\xi_n\in E_{\min}[2^n]\) 使 \(y=P_i+\xi_n\)，从而 \(\operatorname{Addr}_n\) 良定义且双射。
另一方面，层间映射在每个分片上都由 \([2]\) 诱导，而
\[
\varprojlim_n E_{\min}[2^n]\cong T_2(E_{\min})
\]
由推论 \ref{cor:killo-2adic-tate-hypercube-limit} 给出。
故 \(\operatorname{Addr}_n\) 与截断相容，取逆极限即得所示规范双射。证毕。
\end{proof}

\begin{corollary}[Lee--Yang 规范二进编码的三比特下界]\label{cor:conclusion-leyang-canonical-binary-sheet-lower-bound}
\leanverified{paper\_conclusion\_leyang\_canonical\_binary\_sheet\_lower\_bound}
在定理 \ref{thm:conclusion-leyang-canonical-address-discrete-continuous-splitting} 的规范地址语义下，任意精确且规范的二进制编码器若欲完整承载 \(\mathcal R_\infty\)，则除连续 Tate 坐标外，至少还需一个能够区分五个 sheet 的离散寄存器。
若该寄存器以纯二进制实现，其位数 \(b\) 必满足
\[
2^b\ge 5,
\qquad\text{因而}\qquad
b\ge 3.
\]
特别地，任何只依赖 \(T_2(E_{\min})\) 坐标而忘却 sheet 标签的解码器都是 sheet-oblivious 的；它必将五个几何上不同的分支点压缩为同一个去平移后的 Tate 地址。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-leyang-canonical-address-discrete-continuous-splitting}，完整规范地址正是
\[
(i,x)\in\{0,1,2,3,4\}\times T_2(E_{\min}).
\]
因此若编码器不显式保留 \(i\)，则它至多记住 \(x\)，从而五个不同元素
\[
(0,x),\dots,(4,x)
\]
在解码时不可区分。故任何精确规范编码都必须额外记录五值标签。
纯二进制寄存器容量为 \(2^b\)，于是必有 \(2^b\ge 5\)，即 \(b\ge 3\)。证毕。
\end{proof}

\begin{proposition}[branch-compatible 仿射动力学的五张量 \texorpdfstring{$\zeta$}{zeta} 分解]\label{prop:conclusion-leyang-branch-compatible-zeta-fivefold-factorization}
设 \(F:\mathcal R_\infty\to\mathcal R_\infty\) 为一自映射，满足在定理 \ref{thm:conclusion-leyang-canonical-address-discrete-continuous-splitting} 的规范坐标下
\[
\operatorname{Addr}_{\mathrm{can}}\circ F\circ \operatorname{Addr}_{\mathrm{can}}^{-1}(i,x)
=(i,\psi_i(x)),
\]
其中每个 \(\psi_i\) 都是 \(T_2(E_{\min})\) 上的 \(2^L\)-进仿射映射。
则 Artin--Mazur \(\zeta\)-函数满足
\[
\zeta_F(z)=\prod_{i=0}^{4}\zeta_{\psi_i}(z).
\]
特别地，若 \(\psi_i\equiv \psi\) 与 \(i\) 无关，则
\[
\zeta_F(z)=\zeta_{\psi}(z)^5.
\]
因此，在 branch-compatible 口径下，任何 primitive 轨道都被严格束缚在单一 sheet 内；全局 \(\zeta\)-数据只是五个局部仿射动力学的外乘。
\end{proposition}
\begin{proof}
由假设，每个分片
\[
\mathcal R_\infty^{(i)}:=\{i\}\times T_2(E_{\min})
\]
都在 \(F\) 下不变，且 \(F|_{\mathcal R_\infty^{(i)}}=\psi_i\)。
因此对每个 \(n\ge 1\) 有不交并
\[
\operatorname{Fix}(F^n)
=
\bigsqcup_{i=0}^{4}\operatorname{Fix}(\psi_i^n),
\qquad
\#\operatorname{Fix}(F^n)=\sum_{i=0}^{4}\#\operatorname{Fix}(\psi_i^n).
\]
代入 Artin--Mazur 定义
\[
\zeta_F(z)=\exp\!\left(\sum_{n\ge 1}\frac{\#\operatorname{Fix}(F^n)}{n}z^n\right)
\]
便得
\[
\zeta_F(z)
=
\prod_{i=0}^{4}
\exp\!\left(\sum_{n\ge 1}\frac{\#\operatorname{Fix}(\psi_i^n)}{n}z^n\right)
=
\prod_{i=0}^{4}\zeta_{\psi_i}(z).
\]
若 \(\psi_i\equiv \psi\)，则右端退化为 \(\zeta_\psi(z)^5\)。证毕。
\end{proof}

\begin{theorem}[连续基底的普适性与非完备性]\label{thm:conclusion-continuous-base-universal-incomplete}
在有限素局部化与 Lee--Yang 分支塔两条链上，连续基底都具有普适性，但都不是完备承载体：
\begin{enumerate}
  \item 对有限素局部化，所有 \(\Sigma_S\) 的连续商都统一坍缩为 \(\TT\)，但 \(\TT\) 本身完全遗忘素数支撑；只有附加全不连通核 \(P_S=\prod_{p\in S}\ZZ_p\) 后，算术类型才可恢复。
  \item 对 Lee--Yang 分支塔，所有分支片都共享同一个连续基底 \(T_2(E_{\min})\)，但忽略 sheet 标号后便丢失完整分支对象；只有附加离散标签 \(i\in\{0,\dots,4\}\) 后，逆极限全对象才可恢复。
\end{enumerate}
因此，在本文框架中，连续相位基底从来只能作为统一几何底座出现；一旦要求可恢复的算术记忆或分支多重性，就必须外接一条与之正交的离散账本或有限覆盖。
\end{theorem}
\begin{proof}
第一部分直接由定理 \ref{thm:conclusion-finite-localization-circlequotient-profinite-kernel} 给出：对任意有限 \(S,T\)，均有
\[
\Sigma_S/P_S\cong \TT\cong \Sigma_T/P_T,
\]
但 \(\Sigma_S\cong\Sigma_T\iff S=T\)。第二部分则是定理 \ref{thm:conclusion-leyang-five-sheet-dyadic-cover} 的直接重述：五片分支共享同一 \(T_2(E_{\min})\) 基底，而片标一旦遗忘，完整 branch 对象便不再可恢复。两部分合并即得结论。证毕。
\end{proof}

\begin{theorem}[有限长度可见统计的一维 Archimedean 压缩与本质素数账本的双不可约分裂]\label{thm:conclusion-finite-visible-archimedean-essential-prime-ledger-splitting}
设 \(Z_n(u)\) 为自对偶同步核的 canonical family，并沿定义 \ref{def:conclusion-essential-prime-spectrum} 记绝对不可约代数覆盖
\[
\pi_y:C\to \PP^1_y
\]
在稳定反演语义下的本质素数谱为
\[
S_{\mathrm{ess}}(\pi_y)\subset \mathbb P.
\]
则同一框架中存在如下双不可约分裂：
\begin{enumerate}
  \item 对每个 \(n\ge 1\)，存在唯一多项式 \(Q_n\in\ZZ[T]\) 使
  \[
  Z_n(u)=u^{n/2}Q_n\!\left(u^{1/2}+u^{-1/2}\right).
  \]
  因而任意有限长度的可见统计都严格压缩到单一实坐标
  \[
  T=u^{1/2}+u^{-1/2}.
  \]
  \item 令
  \[
  G_{S_{\mathrm{ess}}}:=\ZZ[S_{\mathrm{ess}}(\pi_y)^{-1}],
  \qquad
  \Sigma_{S_{\mathrm{ess}}}:=\widehat{G_{S_{\mathrm{ess}}}}.
  \]
  若要求对 \(\pi_y\) 的稳定反演在全部好素数方向上只使用有限索引标签、并对坏素数方向保持忠实外置，则最小忠实完成必包含短正合列
  \[
  0\longrightarrow \prod_{p\in S_{\mathrm{ess}}(\pi_y)}\ZZ_p
  \longrightarrow \Sigma_{S_{\mathrm{ess}}}
  \longrightarrow \TT
  \longrightarrow 0.
  \]
\end{enumerate}
特别地，有限长度可见统计的全部 Archimedean 连续可见量只占据一条实坐标，而稳定反演所需的算术记忆则严格滞留在
\[
\prod_{p\in S_{\mathrm{ess}}(\pi_y)}\ZZ_p
\]
这一全不连通账本核中；二者不能互相替代，也不能被同一有限秩连续载体同时吸收。
\end{theorem}
\begin{proof}
第一条正是命题 \ref{prop:conclusion-sync-kernel-finite-length-single-coordinate} 的内容：自对偶同步核的每一个有限长度配分函数都唯一因子化经由
\[
T=u^{1/2}+u^{-1/2}.
\]

对第二条，由定理 \ref{thm:conclusion-essential-prime-axis-minimality}，若 \(p\notin S_{\mathrm{ess}}(\pi_y)\)，则该素数方向上的分支提升可由有限集合 \(A_p\) 稳定分离；而一旦 \(p\in S_{\mathrm{ess}}(\pi_y)\)，任何稳定反演都必须保留无限精度的 \(p\)-进记录轴。由于 \(S_{\mathrm{ess}}(\pi_y)\) 是有限集，定理 \ref{thm:conclusion-finite-prime-solenoid-terminal-object} 遂给出相应最小局部化完成的短正合列
\[
0\longrightarrow \prod_{p\in S_{\mathrm{ess}}(\pi_y)}\ZZ_p
\longrightarrow \Sigma_{S_{\mathrm{ess}}}
\longrightarrow \TT
\longrightarrow 0.
\]
其中连通可见部分严格压缩到唯一圆因子 \(\TT\)，而全部素数账本精确滞留在全不连通核中。

两部分合并即得所述双不可约分裂：Archimedean 可见统计与 non-Archimedean 稳定记忆分别落在不同承载层上。证毕。
\end{proof}

\begin{conjecture}[相位、账本与分支的最小三分裂原则]\label{conj:conclusion-phase-ledger-branch-minimal-trisplitting}
任何与扫描--投影生成器兼容、并同时忠实承载乘法账本与 Lee--Yang 分支数据的实现，都必须在最小层级上分裂为三种彼此不可替代的承载：
\[
\text{连续相位商 } \TT
\quad+\quad
\text{全不连通 prime-ledger 核 } P_S
\quad+\quad
\text{有限 sheet 覆盖 } F.
\]
更精确地，应存在一个最小因子化
\[
\mathcal X\longrightarrow \TT
\]
使其纤维再分解为
\[
P_S\times F,
\qquad
P_S=\prod_{p\in S}\ZZ_p,
\]
其中 \(F\) 为有限离散分支集；对当前 Lee--Yang 链，预期 \(|F|=5\)。删去这三部分中的任意一部分，都将分别破坏相位输运、一致的素支撑恢复或分支全对象恢复。
\end{conjecture}

\begin{proposition}[completion-side 忠实 \texorpdfstring{$\pi$}{pi} 通道的 phase--ledger 双层障碍]\label{prop:conclusion-completion-side-faithful-pi-channel-phase-ledger-double-obstruction}
设某 \(\pi\)-通道在 completion-side 的可见层只经由以下两部分读出：
\begin{enumerate}
  \item 零点经验相位层 \(\mu_r\)；
  \item 一轴 profinite 外置寄存器 \(L\)，且 \(\pcdim(L)=1\)。
\end{enumerate}
则该通道不可能同时满足下列两项：
\begin{enumerate}
  \item 对 bounded-local、\(D_N\to\infty\) 扇区保持 fixed-prime 精确可辨性；
  \item 对某个 \(r\ge 2\) 的 \(p\)-进全秩 hidden sector 保持忠实性。
\end{enumerate}
\end{proposition}
\begin{proof}
第一项已被推论 \ref{cor:conclusion-asymptotic-zero-law-certificate-cannot-recover-fixed-prime} 排除：仅依赖渐近零点经验律的相位层对固定素数账本值严格失明。

第二项则被定理 \ref{thm:conclusion-uniaxial-prime-register-universality-incompatible-high-rank-faithfulness} 排除：任何 \(\pcdim=1\) 的一轴寄存器都不可能忠实承载 rank \(\ge 2\) 的 \(p\)-进全秩扇区。

故一旦 completion-side 的证书层最终塌缩为“单相位层 + 单轴寄存器”，它至多是值层近似算法，而不可能同时保有 fixed-prime exactness 与 high-rank \(p\)-adic fidelity。证毕。
\end{proof}

\begin{conjecture}[faithful \texorpdfstring{$\pi$}{pi}-formula 的 phase--ledger 分裂原理]\label{conj:conclusion-faithful-pi-formula-phase-ledger-splitting}
任一若要在本文语义中忠实承载 \(\pi\)-通道 completion 端的公式族，都应至少分裂为两个正交因子
\[
\mathfrak P_{\mathrm{phase}}
\quad\text{与}\quad
\mathfrak R_{\mathrm{ledger}}.
\]
其中 \(\mathfrak P_{\mathrm{phase}}\) 只负责零点经验律、相位统计及其可能的 Haar 普适极限；而 \(\mathfrak R_{\mathrm{ledger}}\) 负责精确保存逐素账本
\[
r(p)=\frac1\pi\int_{-\pi}^{\pi}A_r(\theta)\cos(p\theta)\,\dd\theta,
\]
并且其最小 \(p\)-进秩恰等于被忠实保留的独立 prime directions 的最大数目。于是：
\begin{enumerate}
  \item rank-one 扇区中，一轴 profinite 寄存器或许足够；
  \item rank \(\ge 2\) 扇区中，一轴普适必然失效；
  \item 任何试图把 \(\mathfrak P_{\mathrm{phase}}\) 与 \(\mathfrak R_{\mathrm{ledger}}\) 再次压缩成单一标量可见量的方案，必将丢失 fixed-prime exactness 或 high-rank \(p\)-adic fidelity。
\end{enumerate}
\end{conjecture}
