\begin{theorem}[boundary 导子 \texorpdfstring{$2$}{2} 中央扇区的表示 \texorpdfstring{$\zeta$}{zeta} 严格平坦]\label{thm:conclusion-window6-boundary-conductor-two-zeta-flatness}
\leanverified{paper\_conclusion\_window6\_boundary\_conductor\_two\_zeta\_flatness}
记
\[
G:=G_6^{\mathrm{bin}},
\qquad
Z_{\partial}:=Z_6^{\mathrm{bd}}\cong (\ZZ/2\ZZ)^3,
\qquad
\widehat Z_{\partial}:=\Hom(Z_{\partial},\{\pm1\}).
\]
对每个 \(\xi\in \widehat Z_{\partial}\)，定义 boundary 中央扇区的细化表示 \(\zeta\)-函数
\[
\zeta^{\mathrm{irr}}_{G,\xi}(s)
:=
\sum_{\substack{\pi\in \widehat G\\ \omega_\pi=\xi}}
(\dim \pi)^{-s},
\]
其中 \(\omega_\pi\) 表示 \(\pi\) 在 \(Z_{\partial}\) 上的中心特征。则对一切 \(\xi\in\widehat Z_{\partial}\) 都有
\[
\zeta^{\mathrm{irr}}_{G,\xi}(s)
=
\frac18\,\zeta^{\mathrm{irr}}_{G}(s)
=
2^5(2+2^{-s})^4(2+2^{-s}+2\cdot 3^{-s})^9.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:derived-window6-representation-zeta-boundary-extension}，restriction 映射
\[
\Hom(G,\{\pm1\})\longrightarrow \Hom(Z_{\partial},\{\pm1\})=\widehat Z_{\partial}
\]
是满射。故对任意 \(\xi,\eta\in \widehat Z_{\partial}\)，可取一维特征 \(\chi\in \Hom(G,\{\pm1\})\) 使
\[
\chi|_{Z_{\partial}}=\eta\xi^{-1}.
\]
于是张量扭曲
\[
\pi\longmapsto \pi\otimes \chi
\]
给出中心特征为 \(\xi\) 与 \(\eta\) 的不可约表示之间的维数保持双射，从而
\[
\zeta^{\mathrm{irr}}_{G,\xi}(s)=\zeta^{\mathrm{irr}}_{G,\eta}(s)
\qquad
(\forall\,\xi,\eta\in\widehat Z_{\partial}).
\]
把全部八个 \(\xi\) 求和即得
\[
\zeta^{\mathrm{irr}}_{G,\xi}(s)=\frac18\,\zeta^{\mathrm{irr}}_{G}(s).
\]
再由定理 \ref{thm:derived-window6-representation-zeta-boundary-extension} 中的全局闭式
\[
\zeta^{\mathrm{irr}}_{G}(s)
=
2^8(2+2^{-s})^4(2+2^{-s}+2\cdot 3^{-s})^9
\]
立得所述公式。证毕。
\end{proof}

\begin{corollary}[boundary 导子 \texorpdfstring{$2$}{2} 扇区的 Plancherel 质量与一维相位数完全等分]\label{cor:conclusion-window6-boundary-conductor-two-plancherel-uniformity}
\leanverified{paper\_conclusion\_window6\_boundary\_conductor\_two\_plancherel\_uniformity}
沿用定理 \ref{thm:conclusion-window6-boundary-conductor-two-zeta-flatness} 的记号。则对每个 \(\xi\in\widehat Z_{\partial}\) 都有
\[
\#\{\chi\in \Hom(G,\{\pm1\}):\ \chi|_{Z_{\partial}}=\xi\}=2^{18},
\]
以及
\[
\sum_{\substack{\pi\in\widehat G\\ \omega_\pi=\xi}}(\dim\pi)^2
=
\frac{|G|}{8}
=
2^{36}3^{13}.
\]
更进一步，八个 boundary 中央扇区的不可约维数多重集彼此完全相同。
\end{corollary}
\begin{proof}
第一式正是定理 \ref{thm:derived-window6-representation-zeta-boundary-extension} 的延拓计数公式。第二式可由定理 \ref{thm:conclusion-window6-boundary-conductor-two-zeta-flatness} 在 \(s=-2\) 处取值：
\[
\sum_{\substack{\pi\in\widehat G\\ \omega_\pi=\xi}}(\dim\pi)^2
=
\zeta^{\mathrm{irr}}_{G,\xi}(-2)
=
2^5(2+2^2)^4(2+2^2+2\cdot 3^2)^9
=
2^5\cdot 6^4\cdot 24^9
=
2^{36}3^{13}.
\]
另一方面，由定理 \ref{thm:derived-window6-gauge-volume-defect-identity}，
\[
|G|=2^{39}3^{13},
\]
故上式亦即 \(|G|/8\)。最后，由定理 \ref{thm:derived-window6-boundary-sector-groupalgebra-isotypy} 与定理 \ref{thm:derived-window6-boundary-sector-irrep-moment-uniformity}，八个扇区在 Wedderburn 分块、不可约维数多重集与 Plancherel 权重分布上完全等型，故维数多重集彼此相同。证毕。
\end{proof}

\begin{proposition}[几何对角商后的四类 boundary 导子 \texorpdfstring{$2$}{2} 扇区仍然精确平坦]\label{prop:conclusion-window6-boundary-conductor-two-geometric-quotient-flatness}
记
\[
Z_{\mathrm{geo}}:=P_{\mathrm{geo}}^6=\langle(1,1,1)\rangle\subset Z_{\partial},
\qquad
Z_{\partial}/Z_{\mathrm{geo}}\cong (\ZZ/2\ZZ)^2.
\]
对任意
\[
\bar\xi\in \widehat{Z_{\partial}/Z_{\mathrm{geo}}},
\]
定义几何等价类细化表示 \(\zeta\)-函数
\[
\zeta^{\mathrm{irr}}_{G,\bar\xi}(s)
:=
\sum_{\substack{\pi\in \widehat G\\ \omega_\pi\mapsto \bar\xi}}
(\dim\pi)^{-s}.
\]
则有精确公式
\[
\zeta^{\mathrm{irr}}_{G,\bar\xi}(s)=\frac14\,\zeta^{\mathrm{irr}}_{G}(s),
\]
并且
\[
\#\{\chi\in \Hom(G,\{\pm1\}):\ \chi|_{Z_{\partial}/Z_{\mathrm{geo}}}=\bar\xi\}=2^{19},
\]
\[
\sum_{\substack{\pi\in\widehat G\\ \omega_\pi\mapsto \bar\xi}}(\dim\pi)^2
=
\frac{|G|}{4}
=
2^{37}3^{13}.
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-window6-boundary-parity-three-layer-blind-filtration}，
\[
Z_{\partial}/Z_{\mathrm{geo}}\cong (\ZZ/2\ZZ)^2,
\]
故每个 \(\bar\xi\in \widehat{Z_{\partial}/Z_{\mathrm{geo}}}\) 恰对应两个 boundary 中央特征 \(\xi\in \widehat Z_{\partial}\)。于是由定理 \ref{thm:conclusion-window6-boundary-conductor-two-zeta-flatness}，
\[
\zeta^{\mathrm{irr}}_{G,\bar\xi}(s)
=
2\cdot \frac18\,\zeta^{\mathrm{irr}}_{G}(s)
=
\frac14\,\zeta^{\mathrm{irr}}_{G}(s).
\]
同理，延拓计数满足
\[
\#\{\chi\in \Hom(G,\{\pm1\}):\ \chi|_{Z_{\partial}/Z_{\mathrm{geo}}}=\bar\xi\}
=
2\cdot 2^{18}
=
2^{19},
\]
Plancherel 质量则由推论 \ref{cor:conclusion-window6-boundary-conductor-two-plancherel-uniformity} 给出
\[
\sum_{\omega_\pi\mapsto \bar\xi}(\dim\pi)^2
=
2\cdot 2^{36}3^{13}
=
2^{37}3^{13}
=
\frac{|G|}{4}.
\]
证毕。
\end{proof}

\begin{theorem}[表示 \texorpdfstring{$\zeta$}{zeta} 在 \texorpdfstring{$+\infty$}{+∞} 的两级喷射精确恢复奇球秩向量]\label{thm:conclusion-window6-boundary-zeta-jets-recover-odd-sphere-ranks}
定义全局归一化与扇区归一化函数
\[
\Psi_6(s):=2^{-21}\zeta^{\mathrm{irr}}_{G}(s),
\qquad
\Psi_{6,\xi}(s):=2^{-18}\zeta^{\mathrm{irr}}_{G,\xi}(s)
\qquad
(\xi\in\widehat Z_{\partial}).
\]
则对每个 \(\xi\in\widehat Z_{\partial}\) 都有严格恒等式
\[
\Psi_{6,\xi}(s)=\Psi_6(s)
=(1+2^{-s-1})^4(1+2^{-s-1}+3^{-s})^9.
\]
从而当 \(s\to+\infty\) 时，
\[
\Psi_6(s)
=
1+13\cdot 2^{-s-1}+9\cdot 3^{-s}
+O(4^{-s}+6^{-s}+9^{-s}).
\]
特别地，
\[
21=\log_2\!\Bigl(\lim_{s\to+\infty}\zeta^{\mathrm{irr}}_{G}(s)\Bigr),
\]
\[
13=\lim_{s\to+\infty}2^{s+1}\bigl(\Psi_6(s)-1\bigr),
\]
\[
9=\lim_{s\to+\infty}3^{s}\bigl(\Psi_6(s)-1-13\cdot 2^{-s-1}\bigr).
\]
另一方面，window-\(6\) 的有理连续包络满足
\[
K_6\simeq_{\QQ}(S^3)^{21}\times (S^5)^{13}\times (S^7)^9.
\]
故奇球秩向量 \((21,13,9)\) 可由离散表示 \(\zeta\)-函数在 \(+\infty\) 的前两级非平凡喷射完全重建。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-boundary-conductor-two-zeta-flatness}，
\[
\zeta^{\mathrm{irr}}_{G,\xi}(s)
=
2^5(2+2^{-s})^4(2+2^{-s}+2\cdot 3^{-s})^9,
\]
而全局闭式为
\[
\zeta^{\mathrm{irr}}_{G}(s)
=
2^8(2+2^{-s})^4(2+2^{-s}+2\cdot 3^{-s})^9.
\]
分别乘以 \(2^{-18}\) 与 \(2^{-21}\) 后，即得
\[
\Psi_{6,\xi}(s)=\Psi_6(s)
=(1+2^{-s-1})^4(1+2^{-s-1}+3^{-s})^9.
\]

令
\[
a:=2^{-s-1},
\qquad
b:=3^{-s},
\]
则 \(a,b\to0\)，并有
\[
(1+a)^4(1+a+b)^9
=
1+(4+9)a+9b+O(a^2+ab+b^2),
\]
即
\[
\Psi_6(s)=1+13\cdot 2^{-s-1}+9\cdot 3^{-s}+O(4^{-s}+6^{-s}+9^{-s}).
\]
这就给出后两条极限公式。

又因 \(s\to+\infty\) 时只有一维表示对 \(\zeta_G^{\mathrm{irr}}(s)\) 留下贡献，而由定理 \ref{thm:derived-window6-representation-zeta-boundary-extension}，
\[
\#\Hom(G,\{\pm1\})=2^{21},
\]
故
\[
\lim_{s\to+\infty}\zeta_G^{\mathrm{irr}}(s)=2^{21},
\]
从而得到第一条极限公式。

最后，由推论 \ref{cor:fold-groupoid-window6-rational-tail-ladder}，
\[
K_6\simeq_{\QQ}(S^3)^{21}\times (S^5)^{13}\times (S^7)^9.
\]
因此，离散表示 \(\zeta\)-函数在 \(+\infty\) 的全局极限与前两级非平凡喷射所恢复的恰是同一奇球秩向量 \((21,13,9)\)。证毕。
\end{proof}

\begin{conjecture}[boundary 导子 \texorpdfstring{$2$}{2} 扇区与高维盲核的终端张量分裂]\label{conj:conclusion-window6-boundary-conductor-two-tensor-splitting}
存在一个以中心特征完备化的表示半环分裂
\[
K_0(\mathrm{Rep}(G_6^{\mathrm{bin}}))^{\wedge}
\cong
\ZZ[\widehat Z_{\partial}]\,\widehat\otimes\,\mathcal K_6,
\]
满足：
\begin{enumerate}
  \item \(\ZZ[\widehat Z_{\partial}]\) 恰承载全部 boundary 导子 \(2\) 相位；
  \item \(\mathcal K_6\) 的表示 \(\zeta\)-函数满足
  \[
  \zeta_{\mathcal K_6}(s)=(1+2^{-s-1})^4(1+2^{-s-1}+3^{-s})^9;
  \]
  \item \(\mathcal K_6\) 在 \(+\infty\) 的喷射只恢复 \((13,9)\)，而全局 rank \(21\) 仅由 conductor-\(2\) 的中央相位表贡献。
\end{enumerate}
若此分裂成立，则 window-\(6\) 的奇球秩向量 \((21,13,9)\) 将被函子化地拆成“boundary 二进中央片 \(21\)”与“高维盲核片 \((13,9)\)”的张量积数据。
\end{conjecture}
