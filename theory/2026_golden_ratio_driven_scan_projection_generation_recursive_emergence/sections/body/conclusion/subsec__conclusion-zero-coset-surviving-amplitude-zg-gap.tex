\begin{theorem}[同一 \texorpdfstring{$2$}{2}-进层中的任意有限刚性余类族之交再次落回 \texorpdfstring{$\gcd$}{gcd} 余类]\label{thm:conclusion-zero-coset-fixed-v2-finite-intersection-gcd}
设 \(M\) 为偶整数。对任意 \(g\mid M/2\)，记
\[
S_g(M):=
\left\{
k\in \ZZ/(M\ZZ):\ k\equiv \frac{M}{2g}\pmod{M/g}
\right\}.
\]
若
\[
g_1,\dots,g_r\mid \frac{M}{2}
\]
满足
\[
v_2(g_1)=v_2(g_2)=\cdots=v_2(g_r),
\]
并记
\[
g:=\gcd(g_1,\dots,g_r),
\]
则有严格恒等式
\[
\boxed{\;
\bigcap_{i=1}^{r}S_{g_i}(M)=S_g(M).\;}
\]
特别地，
\[
\boxed{\;
\left|\bigcap_{i=1}^{r}S_{g_i}(M)\right|=g.\;}
\]
\leanverified{paper\_conclusion\_zero\_coset\_fixed\_v2\_finite\_intersection\_gcd}
\end{theorem}
\begin{proof}
对 \(r=1\) 结论平凡。对 \(r=2\)，这正是定理 \ref{thm:xi-fold-zero-coset-v2-intersection-rigidity} 在
\[
v_2(g_1)=v_2(g_2)
\]
时的同层情形。

现设结论已对 \(r-1\) 个余类成立，并令
\[
d:=\gcd(g_1,\dots,g_{r-1}).
\]
由于 \(g_1,\dots,g_r\) 具有相同的 \(2\)-进赋值，故
\[
v_2(d)=v_2(g_r)=v_2(g_1).
\]
由归纳假设，
\[
\bigcap_{i=1}^{r-1}S_{g_i}(M)=S_d(M).
\]
再对 \(S_d(M)\) 与 \(S_{g_r}(M)\) 应用定理 \ref{thm:xi-fold-zero-coset-v2-intersection-rigidity}，得到
\[
\bigcap_{i=1}^{r}S_{g_i}(M)
=
S_d(M)\cap S_{g_r}(M)
=
S_{\gcd(d,g_r)}(M)
=
S_g(M).
\]
这就证明了第一条。

最后，由定理 \ref{thm:fold-zero-coset-rigidity} 的显式表示，
\[
S_g(M)=
\left\{
\frac{M}{2g}+t\frac{M}{g}\ \ (\bmod M):\ t=0,1,\dots,g-1
\right\},
\]
可见其恰有 \(g\) 个不同代表元。因而
\[
\left|\bigcap_{i=1}^{r}S_{g_i}(M)\right|
=
|S_g(M)|
=
g.
\]
证毕。
\end{proof}

\begin{corollary}[固定 \texorpdfstring{$2$}{2}-进层上的零余类并具有精确容斥公式，且由极大元完全生成]\label{cor:conclusion-zero-coset-fixed-v2-union-inclusion-exclusion-maximal}
\leanverified{paper\_conclusion\_zero\_coset\_fixed\_v2\_union\_inclusion\_exclusion\_maximal}
在定理 \ref{thm:conclusion-zero-coset-fixed-v2-finite-intersection-gcd} 的设定下，取任意有限集合
\[
\mathcal D\subseteq \left\{g:\ g\mid \frac{M}{2}\right\}
\]
并假设 \(\mathcal D\) 中所有元素具有相同的 \(2\)-进赋值。则有精确并集公式
\[
\boxed{\;
\left|\bigcup_{g\in\mathcal D}S_g(M)\right|
=
\sum_{\varnothing\neq J\subseteq \mathcal D}
(-1)^{|J|+1}\,\gcd(J),\;}
\]
其中 \(\gcd(J)\) 表示子族 \(J\) 全部元素的最大公因子。

进一步，若按整除关系记 \(\mathcal D\) 的极大元集合为
\[
\operatorname{Max}(\mathcal D),
\]
则有
\[
\boxed{\;
\bigcup_{g\in\mathcal D}S_g(M)
=
\bigcup_{g\in \operatorname{Max}(\mathcal D)}S_g(M).\;}
\]
\end{corollary}
\begin{proof}
对任意非空子族 \(J\subseteq\mathcal D\)，由定理 \ref{thm:conclusion-zero-coset-fixed-v2-finite-intersection-gcd}，
\[
\bigcap_{g\in J}S_g(M)=S_{\gcd(J)}(M),
\qquad
\left|\bigcap_{g\in J}S_g(M)\right|=\gcd(J).
\]
因此对并集 \(\bigcup_{g\in\mathcal D}S_g(M)\) 直接应用包含--排斥原理，即得所示精确公式。

再证第二条。若 \(g,h\in\mathcal D\) 且 \(g\mid h\)，则因二者具有相同的 \(2\)-进赋值，商 \(h/g\) 必为奇数，故仍位于同一 \(2\)-进层。于是由定理 \ref{thm:conclusion-zero-coset-fixed-v2-finite-intersection-gcd}（对二元子族 \(\{g,h\}\) 应用）可得
\[
S_g(M)\cap S_h(M)=S_{\gcd(g,h)}(M)=S_g(M),
\]
从而
\[
S_g(M)\subseteq S_h(M).
\]
因此，每个非极大元对应的余类都已包含于某个极大元对应的余类中，删去它们不会改变并集。证毕。
\end{proof}

\begin{theorem}[幸存非零频率的平均平方振幅具有严格的共振下界]\label{thm:conclusion-surviving-fourier-energy-lower-bound}
记
\[
M:=F_{m+2},
\qquad
a_m(k):=\frac{|\widehat d_m(k)|}{2^m},
\qquad
\mathsf{Col}_m:=\frac{1}{M}\sum_{k\in \ZZ/(M\ZZ)}a_m(k)^2,
\]
并记谱零点集合
\[
\mathcal Z_m:=\{k\in \ZZ/(M\ZZ):\ a_m(k)=0\}.
\]
当
\[
M-1-|\mathcal Z_m|>0
\]
时，定义幸存非零频率上的平均平方振幅
\[
\mathcal E_m^{\mathrm{surv}}
:=
\frac{1}{M-1-|\mathcal Z_m|}
\sum_{\substack{k\neq 0\\ k\notin \mathcal Z_m}}a_m(k)^2.
\]
则对充分大的 \(m\)（此时该量由定理 \ref{thm:conclusion-foldbin-active-fourier-modes-diverge} 保证良定），有精确恒等式
\[
\mathcal E_m^{\mathrm{surv}}
=
\frac{M\mathsf{Col}_m-1}{M-1-|\mathcal Z_m|},
\]
并因此满足严格下界
\[
\boxed{\;
\mathcal E_m^{\mathrm{surv}}
\ge
\frac{2C_\varphi^2+o(1)}{M-1-|\mathcal Z_m|}.\;}
\]
特别地，
\[
\boxed{\;
\mathcal E_m^{\mathrm{surv}}
\ge
\frac{2C_\varphi^2+o(1)}{M}.\;}
\]
等价地，幸存频率的均方根振幅满足
\[
\boxed{\;
\bigl(\mathcal E_m^{\mathrm{surv}}\bigr)^{1/2}
\ge
\frac{\sqrt{2}\,C_\varphi+o(1)}{\sqrt{M}}.\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-collision-spectrum}，
\[
\mathsf{Col}_m=\frac{1}{M}\sum_{k\in \ZZ/(M\ZZ)}a_m(k)^2.
\]
又由定义，
\[
a_m(0)=1,
\]
且对 \(k\in \mathcal Z_m\) 有 \(a_m(k)=0\)。因此
\[
\sum_{\substack{k\neq 0\\ k\notin \mathcal Z_m}}a_m(k)^2
=
\sum_{k}a_m(k)^2-a_m(0)^2
=
M\mathsf{Col}_m-1.
\]
代回 \(\mathcal E_m^{\mathrm{surv}}\) 的定义便得到
\[
\mathcal E_m^{\mathrm{surv}}
=
\frac{M\mathsf{Col}_m-1}{M-1-|\mathcal Z_m|}.
\]

另一方面，定理 \ref{thm:xi-time-part9pb-rh-window-uniform-gap-coexistence} 给出
\[
M\mathsf{Col}_m-1
\ge
2C_\varphi^2+o(1),
\]
故
\[
\mathcal E_m^{\mathrm{surv}}
\ge
\frac{2C_\varphi^2+o(1)}{M-1-|\mathcal Z_m|}.
\]
再由定理 \ref{thm:fold-zero-density-sparse}，
\[
\frac{|\mathcal Z_m|}{M}
\le
\tau(m+2)\,O(\varphi^{-m/2})
\to 0,
\]
从而分母满足
\[
M-1-|\mathcal Z_m|=M(1+o(1)).
\]
特别地，因 \(M-1-|\mathcal Z_m|\le M\)，便有更粗但更透明的下界
\[
\mathcal E_m^{\mathrm{surv}}
\ge
\frac{2C_\varphi^2+o(1)}{M}.
\]

最后对两边开平方，即得
\[
\bigl(\mathcal E_m^{\mathrm{surv}}\bigr)^{1/2}
\ge
\frac{\sqrt{2}\,C_\varphi+o(1)}{\sqrt{M}}.
\]
证毕。
\end{proof}

\begin{theorem}[有限模态碰撞能量的不完备性]\label{thm:conclusion-finite-modal-collision-energy-incompleteness}
沿用定理 \ref{thm:conclusion-surviving-fourier-energy-lower-bound} 的记号。若一列频率子集
\[
A_m\subseteq \ZZ/(F_{m+2}\ZZ)
\]
满足
\[
|A_m|=o\!\bigl(F_{m+2}\mathsf{Col}_m\bigr),
\]
则有
\[
\frac{1}{F_{m+2}\mathsf{Col}_m}\sum_{k\in A_m}a_m(k)^2\longrightarrow 0.
\]
特别地，由定理 \ref{thm:xi-time-part9pb-rh-window-uniform-gap-coexistence} 的
\[
F_{m+2}\mathsf{Col}_m\to+\infty,
\]
可知任意基数一致有界的频率族都只能携带碰撞能量的渐近零份额。
\end{theorem}
\begin{proof}
由 Parseval 型恒等式
\[
\mathsf{Col}_m=\frac1{F_{m+2}}\sum_{k\in \ZZ/(F_{m+2}\ZZ)}a_m(k)^2,
\]
且对每个 \(k\) 都有
\[
0\le a_m(k)^2\le 1.
\]
因此
\[
0\le
\frac1{F_{m+2}}\sum_{k\in A_m}a_m(k)^2
\le
\frac{|A_m|}{F_{m+2}}.
\]
两边同除以 \(\mathsf{Col}_m\)，得到
\[
0\le
\frac{1}{F_{m+2}\mathsf{Col}_m}\sum_{k\in A_m}a_m(k)^2
\le
\frac{|A_m|}{F_{m+2}\mathsf{Col}_m}.
\]
右端按假设趋于 \(0\)，故结论成立。最后一句只是在
\[
|A_m|=O(1)
\]
时应用同一估计，并结合
\[
F_{m+2}\mathsf{Col}_m\to+\infty
\]
所得。证毕。
\end{proof}

\begin{corollary}[Fibonacci 共轭频对只是共振种子而非热力学承载体]\label{cor:conclusion-fibonacci-resonance-seed-not-thermal-carrier}
记
\[
A_m^{\mathrm{Fib}}:=\{F_m,F_{m+1}\}\subseteq \ZZ/(F_{m+2}\ZZ).
\]
则有
\[
\frac{a_m(F_m)^2+a_m(F_{m+1})^2}{F_{m+2}\mathsf{Col}_m}\longrightarrow 0,
\]
但同时
\[
F_{m+2}\left(\mathsf{Col}_m-\frac1{F_{m+2}}\right)\ge 2C_\varphi^2+o(1),
\qquad
C_\varphi:=\prod_{n\ge 2}\left|\cos(\pi\varphi^{-n})\right|>0.
\]
因此，Fibonacci 共轭频对在动力学上构成不可消除的共振种子，却在热力学意义上永远不是碰撞能量的主承载体。
\end{corollary}
\begin{proof}
由于
\[
|A_m^{\mathrm{Fib}}|=2,
\]
第一式是定理 \ref{thm:conclusion-finite-modal-collision-energy-incompleteness} 的直接特化。
另一方面，定理 \ref{thm:xi-time-part9pb-rh-window-uniform-gap-coexistence} 已给出
\[
a_m(F_m)\to C_\varphi,
\qquad
a_m(F_{m+1})\to C_\varphi,
\]
从而
\[
\mathsf{Col}_m
\ge
\frac{1+a_m(F_m)^2+a_m(F_{m+1})^2}{F_{m+2}}
=
\frac{1+2C_\varphi^2+o(1)}{F_{m+2}}.
\]
移项后即得第二式。于是这两条频率虽然强制出常数级碰撞缺口，却因其能量份额与总碰撞能之比趋于零，不能充当热力学承载体。证毕。
\end{proof}

\begin{theorem}[谱零点计数机制的宏观无力]\label{thm:conclusion-zero-set-count-macro-impotence}
记
\[
M:=F_{m+2},
\qquad
\mathcal Z_m:=\{k\in \ZZ/(M\ZZ):\ a_m(k)=0\},
\qquad
\mathsf{Col}_m:=\sum_{r\in\ZZ/(M\ZZ)}p_m(r)^2.
\]
则在由零点计数单独驱动的普适上界
\[
\mathsf{Col}_m\le 1-\frac{|\mathcal Z_m|}{M}
\]
中，相对于平凡上界 \(\mathsf{Col}_m\le 1\) 的全部改进量恰为
\[
\frac{|\mathcal Z_m|}{M}.
\]
当 \(M\) 为偶数时，这一改进量满足
\[
0\le \frac{|\mathcal Z_m|}{M}\le \tau(m+2)\,O(\varphi^{-m/2}).
\]
然而沿六倍窗同余子列
\[
m+2\equiv 0\pmod 6
\]
又同时有绝对下界
\[
|\mathcal Z_m|\ge F_{(m+2)/2}.
\]
因此谱零点块在绝对计数上可以达到 \(\exp(c m)\) 级别，而其对碰撞上界的相对改进永远至多是 \(\exp(-c m/2+o(m))\) 级别。特别地，任何超过 \(O(\tau(m+2)\varphi^{-m/2})\) 的宏观碰撞缺口都不可能仅由零点个数机制解释，而必须来自幸存非零频率上的振幅重分配。
\end{theorem}
\begin{proof}
上界
\[
\mathsf{Col}_m\le 1-\frac{|\mathcal Z_m|}{M}
\]
正是定理 \ref{thm:fold-zero-density-sparse} 的内容，因此相对于平凡界 \(\mathsf{Col}_m\le 1\) 的改进量恰为 \(|\mathcal Z_m|/M\)。同一定理同时给出
\[
\frac{|\mathcal Z_m|}{M}\le \tau(m+2)\,O(\varphi^{-m/2}),
\]
这就证明了零点计数机制在相对尺度上的指数无力。

另一方面，沿 \(m+2\equiv 0\pmod 6\) 的子列，由定理 \ref{thm:xi-fold-zero-dyadic-tower-disjoint-fibonacci-cosets}，
\[
|\mathcal Z_m|\ge F_{(m+2)/2}.
\]
结合 Fibonacci 渐近 \(F_n=\varphi^{n+o(n)}\)，可知零点块在绝对计数上至少达到
\[
|\mathcal Z_m|\ge \exp\!\left(\frac12(\log\varphi)m+o(m)\right).
\]
于是绝对大块与相对无力严格并存。若某一碰撞缺口在宏观上大于 \(O(\tau(m+2)\varphi^{-m/2})\)，则其来源不能只是“有多少频率被杀死”，而必须来自其余非零频率上的振幅分布本身。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$ZG$}{ZG} 与完整平方自由 Dirichlet 级数之间的首缺口项恰为 \texorpdfstring{$6^{-s}$}{6^-s}]\label{thm:conclusion-zg-vs-squarefree-first-gap-6s}
\leanverified{paper\_conclusion\_zg\_vs\_squarefree\_first\_gap\_6s}
记
\[
D_{\mathrm{ZG}}(s):=\sum_{n\in \mathrm{ZG}}n^{-s},
\qquad
D_{\mathrm{sqf}}(s):=\sum_{\substack{n\ge 1\\ n\ \mathrm{squarefree}}}n^{-s}
\qquad (\Re s>1).
\]
则沿实轴 \(s\to +\infty\) 时有严格渐近
\[
\boxed{\;
D_{\mathrm{sqf}}(s)-D_{\mathrm{ZG}}(s)=6^{-s}+O(15^{-s}).\;}
\]
特别地，
\[
\boxed{\;
\lim_{s\to+\infty}
6^s\bigl(D_{\mathrm{sqf}}(s)-D_{\mathrm{ZG}}(s)\bigr)=1.\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part62dgc-zg-matrix-euler-no-scalar-product}，\(\mathrm{ZG}\) 恰对应于素数指标链上的有限独立集；等价地，\(\mathrm{ZG}\) 正是那些不含相邻素数指标同时出现的平方自由整数。因而其补集
\[
\mathrm{SQF}\setminus \mathrm{ZG}
\]
恰由那些至少含有一对相邻素数指标的平方自由整数构成，其中 \(\mathrm{SQF}\) 表示全部平方自由正整数集合。

最小的被排除元素显然是
\[
2\cdot 3=6.
\]
现说明除 \(6\) 外，所有被排除元素都至少为 \(15\)。若某个平方自由整数不在 \(\mathrm{ZG}\) 中且不同于 \(6\)，则它必含有某一对相邻素数指标。
若该对为 \((3,5)\)，则该整数至少为 \(15\)；若该对为 \((2,3)\) 且还含有其他素因子，则该整数至少为 \(30\)；若该对从 \((5,7)\) 或更后面开始，则该整数至少为 \(35\)。因此，\(6\) 之外的所有被排除元素都满足
\[
n\ge 15.
\]

于是
\[
D_{\mathrm{sqf}}(s)-D_{\mathrm{ZG}}(s)
=
\sum_{n\in \mathrm{SQF}\setminus \mathrm{ZG}}n^{-s}
=
6^{-s}
+
\sum_{\substack{n\in \mathrm{SQF}\setminus \mathrm{ZG}\\ n\ge 15}}n^{-s}.
\]
取任意固定的 \(s_0>1\)。对所有 \(s\ge s_0\)，有
\[
\sum_{\substack{n\in \mathrm{SQF}\setminus \mathrm{ZG}\\ n\ge 15}}n^{-s}
=
15^{-s}
\sum_{\substack{n\in \mathrm{SQF}\setminus \mathrm{ZG}\\ n\ge 15}}
\left(\frac{15}{n}\right)^s
\le
15^{-s}
\sum_{\substack{n\in \mathrm{SQF}\setminus \mathrm{ZG}\\ n\ge 15}}
\left(\frac{15}{n}\right)^{s_0}.
\]
右端括号中的级数绝对收敛，因为
\[
\mathrm{SQF}\setminus \mathrm{ZG}\subseteq \mathrm{SQF},
\]
而
\[
\sum_{\substack{n\ge 1\\ n\ \mathrm{squarefree}}}n^{-s_0}
\]
在 \(s_0>1\) 时收敛。故
\[
\sum_{\substack{n\in \mathrm{SQF}\setminus \mathrm{ZG}\\ n\ge 15}}n^{-s}
=
O(15^{-s}),
\]
从而得到
\[
D_{\mathrm{sqf}}(s)-D_{\mathrm{ZG}}(s)=6^{-s}+O(15^{-s}).
\]
两边同乘 \(6^s\) 并令 \(s\to+\infty\)，便得到极限为 \(1\)。证毕。
\end{proof}

\begin{theorem}[统一有界频窗不可能穷尽碰撞能量]\label{thm:conclusion-bounded-fourier-window-cannot-exhaust-collision-energy}
\leanverified{paper\_conclusion\_bounded\_fourier\_window\_cannot\_exhaust\_collision\_energy}
记
\[
a_m(k):=\frac{|\widehat d_m(k)|}{2^m},
\qquad
F_{m+2}\,\mathsf{Col}_m=\sum_{k\in \ZZ/(F_{m+2}\ZZ)} a_m(k)^2.
\]
设 \(\{A_m\}_{m\ge 1}\) 是任意一族频率集合，满足
\[
\sup_m |A_m|<\infty.
\]
则有
\[
\sum_{k\notin A_m} a_m(k)^2 \longrightarrow +\infty.
\]
换言之，任何统一有界个数的 Fourier 模都不可能在渐近上穷尽折叠碰撞能量。
\end{theorem}
\begin{proof}
由命题 \ref{prop:conclusion-foldbin-remove-main-resonance-peaks-still-diverges}，碰撞能量满足
\[
F_{m+2}\,\mathsf{Col}_m\to +\infty.
\]
另一方面，对任意 \(k\) 都有
\[
0\le a_m(k)\le 1,
\]
因为
\[
|\widehat d_m(k)|\le \sum_r d_m(r)=2^m.
\]
若 \(|A_m|\le C\) 对所有 \(m\) 成立，则
\[
\sum_{k\in A_m} a_m(k)^2\le C.
\]
于是
\[
\sum_{k\notin A_m} a_m(k)^2
=
F_{m+2}\,\mathsf{Col}_m-\sum_{k\in A_m} a_m(k)^2
\ge
F_{m+2}\,\mathsf{Col}_m-C\to +\infty.
\]
证毕。
\end{proof}
