\begin{theorem}[规范固定点闭包与单边满移位的严格共轭]\label{thm:conclusion-canonical-fixedpoint-fullshift-conjugacy}
\leanverified{paper\_conclusion\_canonical\_fixedpoint\_fullshift\_conjugacy}
沿用定理 \ref{thm:conclusion-canonical-fixedpoints-cantor-closure} 的记号，置
\[
D:=\{0,\dots,M\},
\qquad
\iota:D^{\NN}\longrightarrow \mathcal C_{B,M}(v),
\qquad
\iota((a_n)_{n\ge 0}):=\sum_{n\ge 0}a_nB^nv.
\]
则 \(\iota\) 是一个同胚。若 \(\sigma:D^{\NN}\to D^{\NN}\) 为左移位
\[
\sigma(a_0,a_1,a_2,\dots)=(a_1,a_2,\dots),
\]
并对 \(x\in \mathcal C_{B,M}(v)\) 记 \(a_0(x)\in D\) 为其唯一首位 digit，定义
\[
\mathcal T(x):=\frac{x-a_0(x)v}{B},
\]
则有严格共轭关系
\[
\mathcal T\circ \iota=\iota\circ \sigma.
\]
因此，\(\mathcal C_{B,M}(v)\) 不仅是一个 \(2\)-进自相似 Cantor 闭包，而且与 \((M+1)\) 字母单边满移位严格同胚且动力学共轭。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-canonical-fixedpoints-cantor-closure}，
\[
\mathcal C_{B,M}(v)=\bigsqcup_{a=0}^{M}\bigl(av+B\mathcal C_{B,M}(v)\bigr),
\]
且该并两两不交。对每个 \(x\in \mathcal C_{B,M}(v)\) 反复施行首位分解，得到唯一 digit 展开
\[
x=\sum_{n\ge 0}a_nB^nv
\qquad (a_n\in D),
\]
故 \(\iota\) 为双射。

再证连续性。若两列 digits 在前 \(k\) 位相同，则它们之差可写为
\[
B^k\sum_{n\ge 0}c_nB^nv
\qquad (c_n\in\{-M,\dots,M\}),
\]
故在 \(T_v\cong \ZZ_2\) 上的 \(2\)-进距离至多为 \(2^{-Lk}\)。反过来，若两点 \(2\)-进距离小于 \(2^{-Lk}\)，则其差属于 \(B^kT_v\)，由 digits 落在 \(\{0,\dots,M\}\subset [0,B-1]\) 的唯一性，前 \(k\) 位必相同。于是 \(\iota\) 与 \(\iota^{-1}\) 都连续，故 \(\iota\) 为同胚。

最后，对任意 \((a_n)_{n\ge 0}\in D^{\NN}\)，有
\[
\mathcal T(\iota((a_n)_{n\ge 0}))
=
\frac{\sum_{n\ge 0}a_nB^nv-a_0v}{B}
=
\sum_{n\ge 0}a_{n+1}B^nv
=
\iota(\sigma((a_n)_{n\ge 0})).
\]
故共轭关系成立。证毕。
\end{proof}

\begin{corollary}[规范固定点闭包上的 exact expanding endomorphism]\label{cor:conclusion-canonical-fixedpoint-exact-expanding-entropy}\leanverified{paper\_conclusion\_canonical\_fixedpoint\_exact\_expanding\_entropy}
沿用定理 \ref{thm:conclusion-canonical-fixedpoint-fullshift-conjugacy} 的记号。对每个 \(a\in D\) 记
\[
C_a:=av+B\mathcal C_{B,M}(v)\subset \mathcal C_{B,M}(v).
\]
则 \(\mathcal T\) 在每个一级柱面 \(C_a\) 上都是一条 \(2\)-进相似扩张：
\[
d_2(\mathcal T x,\mathcal T y)=2^L\,d_2(x,y)
\qquad (x,y\in C_a).
\]
并且其拓扑熵满足
\[
h_{\mathrm{top}}(\mathcal T)=\log(M+1).
\]
\end{corollary}
\begin{proof}
任取 \(x,y\in C_a\)。由定义可唯一写成
\[
x=av+Bx',
\qquad
y=av+By'
\qquad (x',y'\in \mathcal C_{B,M}(v)).
\]
于是
\[
\mathcal T x=x',
\qquad
\mathcal T y=y'.
\]
在 \(T_v\cong \ZZ_2\) 上，
\[
d_2(x,y)=d_2(Bx',By')=|B|_2\,d_2(x',y')=2^{-L}d_2(x',y'),
\]
故
\[
d_2(\mathcal T x,\mathcal T y)=d_2(x',y')=2^L\,d_2(x,y).
\]
拓扑熵则由定理 \ref{thm:conclusion-canonical-fixedpoint-fullshift-conjugacy} 与 \((M+1)\) 字母单边满移位的标准熵公式直接得到。证毕。
\end{proof}

\begin{theorem}[均匀 Bernoulli 推前的 exact Ahlfors 正则性]\label{thm:conclusion-canonical-fixedpoint-exact-ahlfors-regularity}\leanverified{paper\_conclusion\_canonical\_fixedpoint\_exact\_ahlfors\_regularity}
设 \(\mathbf u_D\) 为 \(D=\{0,\dots,M\}\) 上的均匀分布，并定义推前测度
\[
\mu_{B,M,v}:=\iota_*(\mathbf u_D^{\otimes \NN}).
\]
记
\[
s:=\frac{\log_2(M+1)}{L}.
\]
则对任意 \(x\in \mathcal C_{B,M}(v)\) 与任意整数 \(k\ge 0\)，都有
\[
\mu_{B,M,v}\bigl(B_{2^{-Lk}}(x)\bigr)=(M+1)^{-k}=2^{-Lks}.
\]
从而 \(\mu_{B,M,v}\) 是一条 exact \(s\)-Ahlfors 正则测度，并且
\[
\dim_{\mathrm H}\mu_{B,M,v}
=
\dim_{\mathrm H}\mathcal C_{B,M}(v)
=
\overline{\dim}_{\mathrm M}\mathcal C_{B,M}(v)
=
\frac{\log_2(M+1)}{L}.
\]
\end{theorem}
\begin{proof}
任取 \(x\in \mathcal C_{B,M}(v)\)。由定理 \ref{thm:conclusion-canonical-fixedpoint-fullshift-conjugacy}，\(x\) 唯一对应一串 digits \((a_n(x))_{n\ge 0}\in D^{\NN}\)。在 \(2\)-进超度量下，闭球 \(B_{2^{-Lk}}(x)\) 恰对应前 \(k\) 个 digits 固定的 cylinder：
\[
B_{2^{-Lk}}(x)
=
\left\{
\iota((b_n)_{n\ge 0}):\ b_j=a_j(x)\ \text{for }0\le j\le k-1
\right\}.
\]
该 cylinder 在 \(\mathbf u_D^{\otimes\NN}\) 下的质量显然为 \((M+1)^{-k}\)，故
\[
\mu_{B,M,v}\bigl(B_{2^{-Lk}}(x)\bigr)=(M+1)^{-k}.
\]
再由
\[
(M+1)^{-k}=2^{-Lk\log_2(M+1)/L}=2^{-Lks},
\]
得到 exact \(s\)-Ahlfors 正则性。最后结合定理 \ref{thm:conclusion-canonical-fixedpoints-cantor-closure} 的集合维数闭式，即得测度维数与集合维数的一致性。证毕。
\end{proof}

\begin{theorem}[primitive 周期层的 sharp 误差与 prime-power 闭式]\label{thm:conclusion-canonical-fixedpoint-primitive-sharp-error-primepower}
\leanverified{paper\_conclusion\_canonical\_fixedpoint\_primitive\_sharp\_error\_primepower}
沿用定理 \ref{thm:conclusion-canonical-fixedpoint-primitive-mobius} 与推论 \ref{cor:conclusion-canonical-fixedpoint-witt-euler-rational-zeta} 的记号。则对每个 \(k\ge 1\)，有
\[
0\le (M+1)^k-|\Pi_k|
\le (\tau(k)-1)(M+1)^{k/2},
\]
等价地，
\[
N_k=\frac{(M+1)^k}{k}
+O\!\left(\frac{\tau(k)(M+1)^{k/2}}{k}\right),
\qquad
\frac{|\Pi_k|}{(M+1)^k}\longrightarrow 1.
\]
进一步，对任意素数 \(p\) 与整数 \(\ell\ge 1\)，有精确公式
\[
|\Pi_{p^\ell}|=(M+1)^{p^\ell}-(M+1)^{p^{\ell-1}},
\qquad
N_{p^\ell}=\frac{(M+1)^{p^\ell}-(M+1)^{p^{\ell-1}}}{p^\ell}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-canonical-fixedpoint-primitive-mobius}，
\[
|\Pi_k|=\sum_{d\mid k}\mu(d)(M+1)^{k/d}.
\]
分离 \(d=1\) 项得
\[
|\Pi_k|
=(M+1)^k+\sum_{\substack{d\mid k\\ d>1}}\mu(d)(M+1)^{k/d}.
\]
由于 \(d>1\) 时 \(k/d\le k/2\)，故
\[
\left|
\sum_{\substack{d\mid k\\ d>1}}\mu(d)(M+1)^{k/d}
\right|
\le (\tau(k)-1)(M+1)^{k/2}.
\]
又显然 \(|\Pi_k|\le |\mathcal F_k|=(M+1)^k\)，从而得到第一组 sharp 误差公式。再除以 \(k\) 即得 \(N_k\) 的估计。由于
\[
(\tau(k)-1)(M+1)^{-k/2}\to 0,
\]
故 \(|\Pi_k|/(M+1)^k\to 1\)。

若 \(k=p^\ell\)，则 Möbius 反演和式中只有 \(d=1,p\) 两项非零，遂
\[
|\Pi_{p^\ell}|=(M+1)^{p^\ell}-(M+1)^{p^{\ell-1}}.
\]
再由 \(N_{p^\ell}=|\Pi_{p^\ell}|/p^\ell\) 得第二式。证毕。
\end{proof}

\begin{corollary}[primitive 目录的 Lambert--Witt 生成函数]\label{cor:conclusion-canonical-fixedpoint-lambert-witt-series}
\leanverified{paper\_conclusion\_canonical\_fixedpoint\_lambert\_witt\_series}
在圆盘
\[
|z|<(M+1)^{-1}
\]
内，primitive 周期类数满足 ordinary generating function 恒等式
\[
\sum_{k\ge 1}N_k z^k
=
\sum_{d\ge 1}\frac{\mu(d)}{d}\log\frac{1}{1-(M+1)z^d}.
\]
等价地，成立 Lambert 级数闭式
\[
\sum_{k\ge 1}\frac{kN_k z^k}{1-z^k}
=
\frac{(M+1)z}{1-(M+1)z}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-canonical-fixedpoint-primitive-mobius}，
\[
kN_k=|\Pi_k|=\sum_{d\mid k}\mu(d)(M+1)^{k/d}.
\]
于是
\[
\sum_{k\ge 1}N_k z^k
=
\sum_{k\ge 1}\frac{z^k}{k}\sum_{d\mid k}\mu(d)(M+1)^{k/d}.
\]
写 \(k=dm\) 并交换求和，得
\[
\sum_{k\ge 1}N_k z^k
=
\sum_{d\ge 1}\frac{\mu(d)}{d}\sum_{m\ge 1}\frac{((M+1)z^d)^m}{m}.
\]
当 \(|z|<(M+1)^{-1}\) 时，上式绝对收敛，故
\[
\sum_{k\ge 1}N_k z^k
=
\sum_{d\ge 1}\frac{\mu(d)}{d}\log\frac{1}{1-(M+1)z^d}.
\]

另一方面，由推论 \ref{cor:conclusion-canonical-fixedpoint-witt-euler-rational-zeta}，
\[
(M+1)^n=\sum_{d\mid n}d\,N_d.
\]
故
\[
\sum_{k\ge 1}\frac{kN_k z^k}{1-z^k}
=
\sum_{k\ge 1}\sum_{m\ge 1}kN_k z^{km}
=
\sum_{n\ge 1}\left(\sum_{d\mid n}d\,N_d\right)z^n
=
\sum_{n\ge 1}(M+1)^n z^n
=
\frac{(M+1)z}{1-(M+1)z}.
\]
证毕。
\end{proof}
