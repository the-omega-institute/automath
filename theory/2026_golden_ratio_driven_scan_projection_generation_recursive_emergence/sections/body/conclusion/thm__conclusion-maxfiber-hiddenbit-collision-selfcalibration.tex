\begin{theorem}[碰撞谱对隐藏位泄露常数的自校准]\label{thm:conclusion-maxfiber-hiddenbit-collision-selfcalibration}
\leanverified{paper\_conclusion\_maxfiber\_hiddenbit\_collision\_selfcalibration}
记碰撞 Perron 根序列为 \(\{r_q\}_{q\ge 2}\)，并定义其端点常数
\[
\varphi_{\mathrm{coll}}
:=
\left(\lim_{q\to\infty}r_q^{1/q}\right)^2.
\]
再记
\[
H_\varphi:=H_b(\varphi^{-1})=H_b(\varphi^{-2})=(3-\varphi)\log_2\varphi,
\]
以及
\[
c_{\mathrm{even}}:=1-H_\varphi,
\qquad
c_{\mathrm{odd}}:=\frac{1-H_\varphi}{2}.
\]
则有严格恒等式
\[
\varphi_{\mathrm{coll}}=\varphi,
\]
从而
\[
H_\varphi=(3-\varphi_{\mathrm{coll}})\log_2\varphi_{\mathrm{coll}},
\]
\[
c_{\mathrm{even}}
=
1-(3-\varphi_{\mathrm{coll}})\log_2\varphi_{\mathrm{coll}},
\qquad
c_{\mathrm{odd}}=\frac{c_{\mathrm{even}}}{2}.
\]
因此，最大纤维相位到隐藏位通道的全部已审计极限常数，并不是独立输入，而是碰撞谱端点常数的内生函数。
\end{theorem}

\begin{proof}
由命题 \ref{prop:pom-rq-qinfty-endpoint}，
\[
\lim_{q\to\infty}r_q^{1/q}=\sqrt{\varphi},
\]
故 \(\varphi_{\mathrm{coll}}=\varphi\)。
另一方面，推论 \ref{cor:pom-max-fiber-achievers-hidden-bit-entropy} 与注 \ref{rem:pom-max-fiber-hidden-bit-leakage-constants} 已给出
\[
H_\varphi=(3-\varphi)\log_2\varphi,
\qquad
c_{\mathrm{even}}=1-H_\varphi,
\qquad
c_{\mathrm{odd}}=\frac{1-H_\varphi}{2}.
\]
把 \(\varphi\) 用 \(\varphi_{\mathrm{coll}}\) 替换即可。证毕。
\end{proof}

\begin{theorem}[最大达到者隐藏位谱的三点极限与 Hausdorff 收敛]\label{thm:conclusion-maxfiber-hiddenbit-spectrum-hausdorff-convergence}
定义最大达到者的隐藏位概率谱
\[
\Sigma_m^{\max}:=\left\{p_1(x):x\in M_m\right\},
\qquad
M_m:=\left\{x\in X_m:d_m(x)=D_m\right\}.
\]
则在稳定阈值区间内有精确闭式
\[
\Sigma_{2k}^{\max}
=
\left\{
\frac{F_k}{F_{k+2}},\ \frac{F_{k+1}}{F_{k+2}}
\right\}
\qquad (2k\ge 10),
\]
\[
\Sigma_{2k+1}^{\max}
=
\left\{
\frac12,\ \frac12+\frac{F_{k-2}}{2F_{k+1}},\ \frac12-\frac{F_{k-2}}{2F_{k+1}}
\right\}
\qquad (2k+1\ge 13).
\]
因此两支统一收敛到三点极限集
\[
\Sigma_\infty=\left\{\varphi^{-2},\ \frac12,\ \varphi^{-1}\right\},
\]
并且存在常数 \(C>0\) 使
\[
d_H(\Sigma_m^{\max},\Sigma_\infty)\le C\varphi^{-m}
\qquad (m\to\infty),
\]
其中 \(d_H\) 表示 \([0,1]\) 中有限子集的 Hausdorff 距离。
\end{theorem}

\begin{proof}
偶数支由定理 \ref{thm:conclusion-window6-even-maxfiber-hiddenbit-exact-fairization} 给出
\[
\Sigma_{2k}^{\max}
=
\left\{
\frac{F_k}{F_{k+2}},\ \frac{F_{k+1}}{F_{k+2}}
\right\}.
\]
奇数支由定理 \ref{thm:conclusion-odd-maxfiber-hiddenbit-tristate-crystal} 给出
\[
\Sigma_{2k+1}^{\max}
=
\left\{
\frac12,\ \frac12\pm\frac{F_{k-2}}{2F_{k+1}}
\right\}.
\]
再由 Binet 公式，
\[
\frac{F_k}{F_{k+2}}=\varphi^{-2}+O(\varphi^{-2k}),
\qquad
\frac{F_{k+1}}{F_{k+2}}=\varphi^{-1}+O(\varphi^{-2k}),
\]
\[
\frac{F_{k-2}}{F_{k+1}}=\varphi^{-3}+O(\varphi^{-2k}),
\]
从而
\[
\frac12\pm\frac{F_{k-2}}{2F_{k+1}}
=
\frac12\pm\frac{\varphi^{-3}}2+O(\varphi^{-2k})
\]
因为
\[
\frac12+\frac{\varphi^{-3}}2=\varphi^{-1},
\qquad
\frac12-\frac{\varphi^{-3}}2=\varphi^{-2}.
\]
故奇数支中每个非平衡谱点到 \(\{\varphi^{-2},\varphi^{-1}\}\) 的距离都为 \(O(\varphi^{-2k})\)，而中间点恒等于 \(1/2\)。
两支共同以指数速度收敛到 \(\Sigma_\infty\)，并可统一写为
\(
d_H(\Sigma_m^{\max},\Sigma_\infty)\le C\varphi^{-m}
\)。
证毕。
\end{proof}

\begin{theorem}[最大达到者分裂对的 \texorpdfstring{$\gcd$}{gcd} 三值律构成相位类型的完备算术分类]\label{thm:conclusion-maxfiber-hiddenbit-gcd-phase-classification}
记
\[
\mathcal M_m:=\left\{(d_{m,0}(x),d_{m,1}(x)):x\in M_m\right\}.
\]
则在稳定阈值区间内，映射
\[
(d_{m,0},d_{m,1})\longmapsto \gcd(d_{m,0},d_{m,1})
\]
在最大达到者族上给出完备相位分类：
\begin{itemize}
  \item 若 \(m=2k\ge 10\)，则对一切 \((d_{m,0},d_{m,1})\in\mathcal M_m\) 都有
  \[
  \gcd(d_{m,0},d_{m,1})=1.
  \]
  \item 若 \(m=2k+1\ge 13\)，则恰有两种可能：
  \[
  \gcd(d_{m,0},d_{m,1})=
  \begin{cases}
  F_{k+1},& (d_{m,0},d_{m,1})=(F_{k+1},F_{k+1})\quad(\text{平衡相}),\\[1ex]
  2,& (d_{m,0},d_{m,1})\in\{(2F_{k-1},2F_k),(2F_k,2F_{k-1})\}\quad(\text{偏置相}).
  \end{cases}
  \]
\end{itemize}
因此，\(\gcd(d_{m,0},d_{m,1})\) 不是粗糙的伴随量，而是最大相位类型的完备算术读出。
\end{theorem}

\begin{proof}
这正是推论 \ref{cor:pom-max-fiber-achievers-bsplit-gcd-trichotomy} 的结论。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\gcd$}{gcd} 分类唯一决定隐藏位熵类]\label{cor:conclusion-maxfiber-gcd-determines-hiddenbit-entropy-class}
在结论 \ref{thm:conclusion-maxfiber-hiddenbit-gcd-phase-classification} 的设定下，最大达到者的隐藏位二元熵
\[
H_b(x):=
-p_1(x)\log_2 p_1(x)-(1-p_1(x))\log_2(1-p_1(x))
\]
满足严格三分：
\begin{itemize}
  \item 若
  \[
  \gcd(d_{m,0}(x),d_{m,1}(x))=F_{k+1},
  \]
  则 \(x\) 必为平衡相，因而
  \[
  H_b(x)=1.
  \]
  \item 若
  \[
  \gcd(d_{m,0}(x),d_{m,1}(x))=1
  \qquad (m=2k\ge 10),
  \]
  则
  \[
  H_b(x)
  =
  H_b\!\left(\frac{F_k}{F_{k+2}}\right)
  =
  H_b\!\left(\frac{F_{k+1}}{F_{k+2}}\right).
  \]
  \item 若
  \[
  \gcd(d_{m,0}(x),d_{m,1}(x))=2
  \qquad (m=2k+1\ge 13),
  \]
  则
  \[
  H_b(x)
  =
  H_b\!\left(\frac{F_k}{F_{k+1}}\right)
  =
  H_b\!\left(\frac{F_{k-1}}{F_{k+1}}\right).
  \]
\end{itemize}
特别地，后两类均满足
\[
H_b(x)\longrightarrow H_\varphi.
\]
\end{corollary}

\begin{proof}
若 \(\gcd(d_{m,0}(x),d_{m,1}(x))=F_{k+1}\)，由结论 \ref{thm:conclusion-maxfiber-hiddenbit-gcd-phase-classification} 必属平衡相，故
\[
p_1(x)=\frac12,
\qquad
H_b(x)=1.
\]
若 \(\gcd=1\)，则必属偶数支最大相位，且
\[
p_1(x)\in
\left\{
\frac{F_k}{F_{k+2}},\ \frac{F_{k+1}}{F_{k+2}}
\right\}.
\]
两值互补，故熵相同。
若 \(\gcd=2\)，则必属奇数支偏置相，且
\[
p_1(x)\in
\left\{
\frac{F_k}{F_{k+1}},\ \frac{F_{k-1}}{F_{k+1}}
\right\},
\]
同样互补，故熵相同。
最后由 Binet 公式
\[
\frac{F_k}{F_{k+2}}\to\varphi^{-2},
\qquad
\frac{F_k}{F_{k+1}}\to\varphi^{-1},
\]
并用
\[
H_b(\varphi^{-1})=H_b(\varphi^{-2})=H_\varphi
\]
即可得到极限。证毕。
\end{proof}

\begin{theorem}[最大相位族满足“边缘无偏—相关有偏”的严格二支律]\label{thm:conclusion-maxfiber-hiddenbit-unbiased-marginal-biased-correlation}
令 \(X\sim \mathrm{Unif}(M_m)\) 为最大达到者相位；条件于 \(X=x\) 时，再取
\[
\omega\sim \mathrm{Unif}(\Fold_m^{-1}(x)),
\qquad
B:=b(\omega)\in\{0,1\}.
\]
则在稳定阈值区间内有如下严格公式：
\begin{itemize}
  \item 若 \(m=2k\ge 10\)，则
  \[
  \PP(B=1)=\frac12,
  \qquad
  I(X;B)=1-H_b\!\left(\frac{F_k}{F_{k+2}}\right),
  \]
  从而
  \[
  I(X;B)\longrightarrow 1-H_\varphi=c_{\mathrm{even}}.
  \]
  \item 若 \(m=2k+1\ge 13\)，则
  \[
  \PP(B=1)=\frac12,
  \qquad
  I(X;B)
  =
  1-\frac12\left(1+H_b\!\left(\frac12+\frac{F_{k-2}}{2F_{k+1}}\right)\right),
  \]
  从而
  \[
  I(X;B)\longrightarrow \frac{1-H_\varphi}{2}=c_{\mathrm{odd}}.
  \]
\end{itemize}
故最大相位族在两支上都严格边缘无偏，但相关泄露常数满足精确二倍律
\[
\lim_{k\to\infty}\frac{I_{2k}(X;B)}{I_{2k+1}(X;B)}=2.
\]
\end{theorem}

\begin{proof}
这正是推论 \ref{cor:pom-max-fiber-phase-hidden-bit-mi} 与注 \ref{rem:pom-max-fiber-hidden-bit-leakage-constants} 的合并表述。
偶数支与奇数支都满足 \(\PP(B=1)=1/2\)，故边缘分布严格无偏；
而互信息极限分别趋于
\[
1-H_\varphi,\qquad \frac{1-H_\varphi}{2},
\]
于是二倍律立即成立。证毕。
\end{proof}

\begin{theorem}[奇数稳定相的四相—二相塌缩阈值是可由碰撞谱自校准的黄金阈值]\label{thm:conclusion-maxfiber-hiddenbit-odd-collapse-threshold-collision-selfcalibrated}
在奇数稳定相 \(m=2k+1\ge 13\) 中，文稿给出的有效二相塌缩阈值为
\[
q_\delta=
\left\lceil
\frac{\log(1/\delta)}{\log(1+\varphi^{-3})}
\right\rceil
\qquad (\delta\in(0,1)).
\]
由结论 \ref{thm:conclusion-maxfiber-hiddenbit-collision-selfcalibration}，该阈值可完全重写为碰撞谱形式
\[
q_\delta=
\left\lceil
\frac{\log(1/\delta)}{\log(1+\varphi_{\mathrm{coll}}^{-3})}
\right\rceil.
\]
再由
\[
\varphi^{-3}=\sqrt5-2
\]
可进一步写成
\[
q_\delta=
\left\lceil
\frac{\log(1/\delta)}{\log(\sqrt5-1)}
\right\rceil.
\]
因此，奇数稳定相中“最大达到者族从四相有效塌缩到二相”的阈值并不是额外组合参数，而是碰撞 Perron 塔自校准出来的黄金常数。
\end{theorem}

\begin{proof}
第一式即结论 \ref{con:xi-terminal-odd-fourphase-two-phase-collapse-threshold}。
由结论 \ref{thm:conclusion-maxfiber-hiddenbit-collision-selfcalibration} 已知
\(
\varphi_{\mathrm{coll}}=\varphi
\)，
故第二式只是变量重写。
最后由
\(
\varphi^3=2+\sqrt5
\)
得到
\[
\varphi^{-3}=\sqrt5-2,
\qquad
1+\varphi^{-3}=\sqrt5-1,
\]
从而得到第三式。证毕。
\end{proof}

\begin{conjecture}[最大相位隐藏位统计的全体极限常数均由碰撞 Perron 塔生成]\label{conj:conclusion-maxfiber-hiddenbit-all-constants-from-collision-spectrum}

\begin{enumerate}

  \[
  \bigl(d_{m,0}(x),d_{m,1}(x)\bigr)_{x\in M_m};
  \]
  \item \(\mathcal I_m\) 在相位互补
  \[
  (d_{m,0},d_{m,1})\longleftrightarrow(d_{m,1},d_{m,0})

\end{enumerate}

总可写为 \(\varphi\) 的初等函数，因而由碰撞 Perron 塔 \(\{r_q\}_{q\ge2}\) 唯一决定。

换言之，在最大相位—隐藏位界面上，稳定极限常数不应再需要额外的 hidden-bit 组合输入；它们应全部塌缩到单一自校准参数 \(\varphi_{\mathrm{coll}}\) 所生成的函数域之中。
\end{conjecture}
