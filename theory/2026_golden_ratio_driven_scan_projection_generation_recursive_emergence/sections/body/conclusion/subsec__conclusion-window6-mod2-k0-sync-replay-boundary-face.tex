\subsection{window-\texorpdfstring{$6$}{6} 的模二 \texorpdfstring{$K_0$}{K0} 阴影、fold 同步代价与回放位数分层}\label{subsec:conclusion-window6-mod2-k0-sync-replay-boundary-face}
沿用小节 \ref{subsec:conclusion-foldbin-stable-k0-faithful-boundary-parity-groupoid}、小节 \ref{subsec:conclusion-foldbin-faithful-dimension-fullinversion-obstruction}、小节 \ref{subsec:conclusion-window6-phase-budget-rigidity} 与小节 \ref{subsec:conclusion-window6-spin-minuscule-hypercube-signature} 的记号。此前链条已经分别把 bin-fold 纤维群胚代数的 Wedderburn 分解、有序 \(K_0\) 分类、完美回放阈值、window-\(6\) 的 \(21\)-维 anomaly 电荷格以及 boundary parity 与 \(T_6[2]\) 的识别固定下来。由此还可继续抽出如下后果：anomaly 荷格正是非平凡 Wedderburn 块的模二 \(K_0\) 阴影；window-\(6\) 的 fold 同步最小互信息恰等于输出熵；而完美回放、boundary 超选分辨与 anomaly census 则构成三条严格分离的位数制度。最后，boundary 三扇区在半单代数的 \(K_0\)-正锥中切出一个刚性秩三面，其模二影像恰为 toral two-torsion。

\begin{theorem}[anomaly 荷格的模二 \texorpdfstring{$K_0$}{K0} 实现]\label{thm:conclusion-foldbin-mod2-k0-anomaly-realization}
\leanverified{paper\_conclusion\_foldbin\_mod2\_k0\_anomaly\_realization}
对任意分辨率 \(m\ge 1\)，记
\[
A_m^{\mathrm{nt}}
:=
\bigoplus_{\{w\in X_m:\ d_m^{\mathrm{bin}}(w)\ge 2\}}
M_{d_m^{\mathrm{bin}}(w)}(\CC),
\qquad
G_m^{\mathrm{bin}}
:=
\prod_{w\in X_m} S_{d_m^{\mathrm{bin}}(w)}.
\]
则存在自然同构
\[
K_0(A_m^{\mathrm{nt}})\otimes_{\ZZ}\FF_2
\cong
\bigl(G_m^{\mathrm{bin}}\bigr)^{\mathrm{ab}}.
\]
特别地，在 window-\(6\) 中所有纤维都满足
\[
d_6^{\mathrm{bin}}(w)\in\{2,3,4\},
\]
故
\[
A_6^{\mathrm{nt}}=\CC[\mathcal R_6^{\mathrm{bin}}],
\qquad
K_0\!\bigl(\CC[\mathcal R_6^{\mathrm{bin}}]\bigr)\otimes_{\ZZ}\FF_2
\cong
\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}
\cong
(\ZZ/2\ZZ)^{21}.
\]
因此，window-\(6\) 的 \(21\)-维 anomaly 荷格恰是纤维群胚半单代数之 \(K_0\)-群在模二下的像，而不是外加的 parity 计数。
\end{theorem}
\begin{proof}
由 Wedderburn 分解，
\[
\CC[\mathcal R_m^{\mathrm{bin}}]
\cong
\bigoplus_{w\in X_m}M_{d_m^{\mathrm{bin}}(w)}(\CC),
\]
于是
\[
K_0(A_m^{\mathrm{nt}})
\cong
\bigoplus_{\{w:\ d_m^{\mathrm{bin}}(w)\ge 2\}}
K_0\!\bigl(M_{d_m^{\mathrm{bin}}(w)}(\CC)\bigr)
\cong
\ZZ^{r_m},
\]
其中
\[
r_m:=\#\{w\in X_m:\ d_m^{\mathrm{bin}}(w)\ge 2\}.
\]
另一方面，
\[
G_m^{\mathrm{bin}}
\cong
\prod_{w\in X_m}S_{d_m^{\mathrm{bin}}(w)},
\]
而
\[
S_d^{\mathrm{ab}}
\cong
\begin{cases}
0, & d=1,\\
\ZZ/2\ZZ, & d\ge 2.
\end{cases}
\]
故
\[
\bigl(G_m^{\mathrm{bin}}\bigr)^{\mathrm{ab}}
\cong
(\ZZ/2\ZZ)^{r_m}.
\]
将
\[
K_0(A_m^{\mathrm{nt}})\cong \ZZ^{r_m}
\]
模二约化，便得到
\[
K_0(A_m^{\mathrm{nt}})\otimes_{\ZZ}\FF_2
\cong
\FF_2^{r_m}
\cong
(\ZZ/2\ZZ)^{r_m}
\cong
\bigl(G_m^{\mathrm{bin}}\bigr)^{\mathrm{ab}}.
\]
在 \(m=6\) 时，由推论 \ref{cor:fold-bin-m6-fiber-hist} 的直方图
\[
2:8,\qquad 3:4,\qquad 4:9
\]
知 \(r_6=21\)，而全部纤维都非平凡，故 \(A_6^{\mathrm{nt}}=\CC[\mathcal R_6^{\mathrm{bin}}]\)。代回即可得到 window-\(6\) 的结论。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} anomaly 荷格的根--环面二挠分裂]\label{cor:conclusion-window6-anomaly-root-torus-mod2-splitting}
记
\[
X_6=X_6^{\mathrm{cyc}}\sqcup X_6^{\mathrm{bdry}},
\qquad
|X_6^{\mathrm{cyc}}|=18,
\qquad
X_6^{\mathrm{bdry}}=\{100001,100101,101001\}.
\]
则有规范直和分解
\[
\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}
\cong
\FF_2^{X_6^{\mathrm{cyc}}}\oplus \FF_2^{X_6^{\mathrm{bdry}}}.
\]
进一步，经由 window-\(6\) 的 root--Cartan 字典与 boundary parity 的 toral 识别，有
\[
\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}
\cong
\FF_2^{\Phi(B_3)}\oplus T_6[2]
\cong
\FF_2^{\Phi(C_3)}\oplus T_6[2].
\]
因此，window-\(6\) 的 \(21\)-维 anomaly 荷格并非均匀立方体，而是一个 \(18\)-维根索引扇区与一个 \(3\)-维 toral parity 扇区的严格直和。
\end{corollary}
\begin{proof}
第一条分解只是集合分拆
\[
X_6=X_6^{\mathrm{cyc}}\sqcup X_6^{\mathrm{bdry}}
\]
在定理 \ref{thm:conclusion-foldbin-mod2-k0-anomaly-realization} 给出的坐标模型中的直接翻译。再由定理 \ref{thm:window6-abelianized-parity-charge-root-cartan-splitting}，\(X_6^{\mathrm{cyc}}\) 规范识别为 \(B_3\) 或 \(C_3\) 的 \(18\) 条根方向；由定理 \ref{thm:conclusion-window6-boundary-torus-2torsion}，
\[
\FF_2^{X_6^{\mathrm{bdry}}}
\cong
T_6[2]\cong (\ZZ/2\ZZ)^3.
\]
合并即得所述分裂。证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的精确 fold 同步互信息]\label{thm:conclusion-window6-exact-fold-synchronization-mutual-information}
\leanverified{paper\_conclusion\_window6\_exact\_fold\_synchronization\_mutual\_information}
令
\[
F:=\Fold_6^{\mathrm{bin}}:\Omega_6\to X_6,
\qquad
\Omega_6=\{0,\dots,63\}.
\]
考虑耦合集
\[
\mathcal C_6
:=
\Bigl\{
P_{XY}:\ P_X=P_Y=\mathrm{Unif}(\Omega_6),\ F(X)=F(Y)\ \text{a.s.}
\Bigr\}.
\]
再记
\[
W:=F(X)\sim p_6,
\qquad
p_6(w)=\frac{d_6^{\mathrm{bin}}(w)}{64}.
\]
则
\[
\inf_{P_{XY}\in\mathcal C_6} I(X;Y)=H(W).
\]
亦即，在边缘分布均匀且要求两端 fold 输出严格同步的约束下，最小互信息恰等于 fold 输出本身的熵。
\end{theorem}
\begin{proof}
对任意 \(P_{XY}\in\mathcal C_6\)，由于
\[
W=F(X)=F(Y)\qquad\text{a.s.},
\]
数据处理不等式给出
\[
I(X;Y)\ge I(F(X);F(Y))=I(W;W)=H(W).
\]
故
\[
\inf_{P_{XY}\in\mathcal C_6} I(X;Y)\ge H(W).
\]

反过来，构造达到下界的耦合如下。先取
\[
W\sim p_6.
\]
在条件 \(W=w\) 下，令 \(X\) 与 \(Y\) 在纤维 \(F^{-1}(w)\) 上条件独立且均匀，即
\[
P(X=x\mid W=w)=P(Y=y\mid W=w)=\frac{1}{d_6^{\mathrm{bin}}(w)}\mathbf 1_{\{F(x)=w\}}.
\]
则 \(F(X)=F(Y)=W\) 几乎必然。并且对任意 \(x\in\Omega_6\)，
\[
P(X=x)
=
\frac{p_6(F(x))}{d_6^{\mathrm{bin}}(F(x))}
=
\frac{d_6^{\mathrm{bin}}(F(x))/64}{d_6^{\mathrm{bin}}(F(x))}
=
\frac{1}{64},
\]
故 \(X\) 为均匀分布；\(Y\) 同理。又因 \(X\perp Y\mid W\)，且 \(W\) 是 \(Y\) 的确定函数，故
\[
I(X;Y)=I(X;Y,W)=I(X;W)+I(X;Y\mid W)=I(X;W)=H(W).
\]
于是下界可达。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} 同步代价的熵缺口、KL 缺口与 anomaly 秩公式]\label{cor:conclusion-window6-synchronization-entropy-kl-anomaly-identity}
\leanverified{paper\_conclusion\_window6\_synchronization\_entropy\_kl\_anomaly\_identity}
沿用定理 \ref{thm:conclusion-window6-exact-fold-synchronization-mutual-information} 的记号。定义
\[
\kappa_6:=\frac{1}{64}\sum_{w\in X_6}d_6^{\mathrm{bin}}(w)\log d_6^{\mathrm{bin}}(w),
\qquad
u_6(w):=\frac{1}{21}.
\]
则有精确恒等式
\[
\inf_{P_{XY}\in\mathcal C_6} I(X;Y)
=
\log 64-\kappa_6
=
\log 21-D_{\mathrm{KL}}(p_6\mid u_6).
\]
再由
\[
\dim_{\FF_2}\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}=21,
\]
上式可改写为
\[
\inf_{P_{XY}\in\mathcal C_6} I(X;Y)
=
\log \dim_{\FF_2}\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}
-D_{\mathrm{KL}}(p_6\mid u_6).
\]
因此，window-\(6\) 的最小精确同步代价，恰等于 anomaly 秩的对数减去相对均匀性的 KL 缺口。特别地，利用主稿给出的 \(\kappa_6\approx 1.159067\)，可得
\[
\inf_{P_{XY}\in\mathcal C_6} I(X;Y)\approx 2.999816.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-window6-exact-fold-synchronization-mutual-information}，
\[
\inf_{P_{XY}\in\mathcal C_6} I(X;Y)=H(W)
=
-\sum_{w\in X_6}p_6(w)\log p_6(w).
\]
又
\[
p_6(w)=\frac{d_6^{\mathrm{bin}}(w)}{64},
\]
故
\[
H(W)
=
-\sum_{w\in X_6}\frac{d_6^{\mathrm{bin}}(w)}{64}
\log\frac{d_6^{\mathrm{bin}}(w)}{64}
=
\log 64-\kappa_6.
\]
另一方面，
\[
D_{\mathrm{KL}}(p_6\mid u_6)
=
\sum_{w\in X_6}p_6(w)\log\frac{p_6(w)}{1/21}
=
\kappa_6-\log\frac{64}{21}.
\]
整理即得
\[
\log 64-\kappa_6=\log 21-D_{\mathrm{KL}}(p_6\mid u_6).
\]
再代入
\[
\dim_{\FF_2}\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}=21
\]
便得到最后一式。数值近似由 \(\log 64-\kappa_6\) 直接算出。证毕。
\end{proof}

\begin{theorem}[完美回放阈值与 window-\texorpdfstring{$6$}{6} 的一比特缺口]\label{thm:conclusion-window6-perfect-replay-threshold-onebit-gap}
\leanverified{paper\_conclusion\_window6\_perfect\_replay\_threshold\_onebit\_gap}
对一般折叠通道 \(\Fold_m:\Omega_m\to X_m\)，记
\[
M_m:=\max_{x\in X_m}|\Fold_m^{-1}(x)|.
\]
则完美回放所需的最小侧信息位数满足
\[
B_m^{\mathrm{perf}}=\left\lceil \log_2 M_m\right\rceil.
\]
在 window-\(6\) 的二进制区间折叠中，由
\[
d_6^{\mathrm{bin}}(w)\in\{2,3,4\}
\]
知
\[
M_6=4,
\qquad
B_6^{\mathrm{perf}}=2.
\]
并且当只允许 \(B=1\) 比特时，最优成功率为
\[
\mathrm{Succ}_6(1)=\frac{21}{32}.
\]
故在 window-\(6\) 中，两比特恰足以完美回放，而一比特与完美回放之间仍保留不可消去的缺口。
\end{theorem}
\begin{proof}
一般公式正是定理 \ref{thm:xi-full-microstate-exact-inversion-bitrate-threshold} 的内容。对 window-\(6\)，由推论 \ref{cor:fold-bin-m6-fiber-hist} 的直方图
\[
2:8,\qquad 3:4,\qquad 4:9
\]
知 \(M_6=4\)，故
\[
B_6^{\mathrm{perf}}=\lceil \log_2 4\rceil=2.
\]
而当 \(B=1\) 时，容量闭式给出
\[
C_6(1)=\sum_{w\in X_6}\min(d_6^{\mathrm{bin}}(w),2)=21\cdot 2=42,
\]
从而
\[
\mathrm{Succ}_6(1)=\frac{C_6(1)}{64}=\frac{42}{64}=\frac{21}{32}.
\]
证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} 中回放、boundary 超选与 anomaly census 的三重位数分离]\label{cor:conclusion-window6-replay-boundary-anomaly-bit-separation}
\leanverified{paper\_conclusion\_window6\_replay\_boundary\_anomaly\_bit\_separation}
记
\[
B_6^{\mathrm{replay}}:=B_6^{\mathrm{perf}},
\qquad
B_6^{\mathrm{bdry}}
:=
\left\lceil \log_2 \bigl|(Z_6^{\mathrm{bd}})^\vee\bigr|\right\rceil,
\qquad
B_6^{\mathrm{anom}}
:=
\dim_{\FF_2}\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}.
\]
则
\[
B_6^{\mathrm{replay}}=2,
\qquad
B_6^{\mathrm{bdry}}=3,
\qquad
B_6^{\mathrm{anom}}=21.
\]
从而
\[
B_6^{\mathrm{replay}}
<
B_6^{\mathrm{bdry}}
<
B_6^{\mathrm{anom}},
\qquad
B_6^{\mathrm{bdry}}-B_6^{\mathrm{replay}}=1,
\qquad
B_6^{\mathrm{anom}}-B_6^{\mathrm{replay}}=19.
\]
因此，在 window-\(6\) 中，完美回放、boundary 八重超选分辨与完整 anomaly census 分别属于三种彼此不同的最小位数制度。
\end{corollary}
\begin{proof}
第一式由定理 \ref{thm:conclusion-window6-perfect-replay-threshold-onebit-gap} 给出。又由定理 \ref{thm:conclusion-window6-boundary-torus-2torsion}，
\[
Z_6^{\mathrm{bd}}\cong (\ZZ/2\ZZ)^3,
\]
故其特征群的大小为 \(8\)，从而
\[
B_6^{\mathrm{bdry}}=\lceil \log_2 8\rceil=3.
\]
最后，由定理 \ref{thm:xi-foldbin-parity-charge-split-exact-minrank}，
\[
\bigl(G_6^{\mathrm{bin}}\bigr)^{\mathrm{ab}}\cong (\ZZ/2\ZZ)^{21},
\]
于是
\[
B_6^{\mathrm{anom}}=21.
\]
其余不等式与差值关系直接读出。证毕。
\end{proof}

\begin{proposition}[boundary 扇区在 \texorpdfstring{$K_0$}{K0} 正锥中的秩三原始面]\label{prop:conclusion-window6-boundary-k0-positive-cone-face}
令
\[
A_6:=\CC[\mathcal R_6^{\mathrm{bin}}],
\qquad
A_6^{\mathrm{cyc}}
:=
\bigoplus_{w\in X_6^{\mathrm{cyc}}}M_{d_6^{\mathrm{bin}}(w)}(\CC),
\qquad
A_6^{\mathrm{bd}}
:=
\bigoplus_{w\in X_6^{\mathrm{bdry}}}M_2(\CC).
\]
则
\[
A_6=A_6^{\mathrm{cyc}}\oplus A_6^{\mathrm{bd}},
\qquad
K_0(A_6)^+=K_0(A_6^{\mathrm{cyc}})^+\oplus K_0(A_6^{\mathrm{bd}})^+,
\]
并且
\[
K_0(A_6^{\mathrm{bd}})^+\cong \NN^3,
\qquad
K_0(A_6^{\mathrm{bd}})\otimes_{\ZZ}\FF_2
\cong
T_6[2].
\]
因此，boundary parity 不只是中心子群中的一个附属成分；它在半单群胚代数的 \(K_0\)-正锥中切出一个由 rigid split 唯一指定的秩三原始面，而其模二影像恰与 boundary 三轴的 toral two-torsion 重合。
\end{proposition}
\begin{proof}
由 rigid split，
\[
X_6=X_6^{\mathrm{cyc}}\sqcup X_6^{\mathrm{bdry}},
\qquad
X_6^{\mathrm{bdry}}=\{100001,100101,101001\}.
\]
再由推论 \ref{cor:fold-bin-m6-fiber-hist}，每个 \(w\in X_6^{\mathrm{bdry}}\) 都满足
\[
d_6^{\mathrm{bin}}(w)=2.
\]
故
\[
A_6^{\mathrm{bd}}\cong M_2(\CC)^{\oplus 3},
\]
从而
\[
K_0(A_6^{\mathrm{bd}})\cong \ZZ^3,
\qquad
K_0(A_6^{\mathrm{bd}})^+\cong \NN^3.
\]
由于
\[
A_6=A_6^{\mathrm{cyc}}\oplus A_6^{\mathrm{bd}},
\]
有序 \(K_0\) 亦随之分裂为
\[
K_0(A_6)^+=K_0(A_6^{\mathrm{cyc}})^+\oplus K_0(A_6^{\mathrm{bd}})^+.
\]
因此 \(K_0(A_6^{\mathrm{bd}})^+\) 构成 \(K_0(A_6)^+\) 中一个秩三原始面。

最后，模二约化给出
\[
K_0(A_6^{\mathrm{bd}})\otimes_{\ZZ}\FF_2\cong (\ZZ/2\ZZ)^3.
\]
由定理 \ref{thm:conclusion-window6-boundary-torus-2torsion}，
\[
(\ZZ/2\ZZ)^3\cong T_6[2].
\]
故其模二影像恰为 boundary 三轴的 toral two-torsion。证毕。
\end{proof}

\paragraph{统一读法}
至此，window-\(6\) 的一条对象层链可被压缩为如下四级结构：其非平凡 Wedderburn 块的 \(K_0\)-群在模二下恰好投影为 \(21\)-维 anomaly 荷格；在 root--Cartan 字典下，这个荷格又分裂为一个 \(18\)-维根索引扇区与一个 \(3\)-维 toral parity 扇区；在动力学侧，强制两端 fold 输出同步的最小互信息恰等于输出熵，并可写为 anomaly 秩的对数减去 KL 缺口；而在操作侧，完美回放、boundary 超选分辨与 anomaly census 的最小位数则分别冻结在
\[
2,\qquad 3,\qquad 21.
\]
因此，window-\(6\) 中的非交换对象、模二异常、连续相位与操作回放并不是平行记账，而是同一纤维多重度结构在 \(K\)-理论、信息论与群表示三种坐标中的不同投影。
