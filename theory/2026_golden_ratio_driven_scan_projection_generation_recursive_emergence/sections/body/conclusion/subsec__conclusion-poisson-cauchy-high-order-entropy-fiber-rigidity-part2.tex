\begin{theorem}[端点 Poisson 权的 Cayley--Cauchy 精确尺度同构]\label{thm:conclusion-endpoint-poisson-cayley-cauchy-exact-scale}
\leanverified{paper\_conclusion\_endpoint\_poisson\_cayley\_cauchy\_exact\_scale}
对任意 \(0<r<1\)，令
\[
z_r:=-r\in\mathbb D,
\qquad
\mathcal C(t):=\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\in\TT.
\]
记其 Poisson 核为
\[
P_{z_r}(\theta):=\frac{1-r^2}{|e^{\mathrm{i}\theta}+r|^2}.
\]
若经 Cayley 参数化
\[
e^{\mathrm{i}\theta}=\mathcal C(t)
\qquad\Longleftrightarrow\qquad
\theta=2\arctan t,
\]
则有严格恒等式
\[
P_{z_r}(\theta)\,\frac{\dd\theta}{2\pi}
=
\frac{1}{\pi}\frac{\Gamma_r}{t^2+\Gamma_r^2}\,\dd t,
\qquad
\Gamma_r:=\frac{1+r}{1-r}.
\]
因此，圆盘端点 \(-1\) 处的内层 Poisson 权在 Cayley 反推下不是近似 Cauchy，而是尺度参数 \(\Gamma_r\) 的精确 Cauchy 核。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:xi-endpoint-poisson-cauchy-cayley-pullback} 的内容。其证明只需使用
\[
\cos\theta=\frac{1-t^2}{1+t^2},
\qquad
\dd\theta=\frac{2\,\dd t}{1+t^2},
\]
以及
\[
|e^{\mathrm{i}\theta}+r|^2
=
1+r^2+2r\cos\theta
=
\frac{(1+r)^2+(1-r)^2t^2}{1+t^2},
\]
代回 Poisson 核并整理即可。证毕。
\end{proof}

\begin{corollary}[端点半径、Cauchy 尺度与对偶窗口端点的三重同一]\label{cor:conclusion-endpoint-radius-cauchy-window-triple-identity}
定义
\[
y_-(r):=\frac{1-r}{1+r},
\qquad
y_+(r):=\frac{1+r}{1-r}.
\]
则
\[
y_-(r)=\Gamma_r^{-1},
\qquad
y_+(r)=\Gamma_r,
\qquad
y_-(r)y_+(r)=1,
\]
并且
\[
1-r^2=\frac{4\Gamma_r}{(1+\Gamma_r)^2}.
\]
因此，同一端点参数 \(r\) 同时决定 Poisson 权的 Cauchy 尺度、端点视界变量的支撑窗口端点与对数缺陷常数 \(1-r^2\)。
\end{corollary}
\begin{proof}
前两式由
\(
\Gamma_r=(1+r)/(1-r)
\)
直接整理得到。再由
\[
r=\frac{\Gamma_r-1}{\Gamma_r+1}
\]
代回 \(1-r^2\)，即可得到
\[
1-r^2=\frac{4\Gamma_r}{(1+\Gamma_r)^2}.
\]
证毕。
\end{proof}

\begin{theorem}[端点视界变量的 reciprocal--arcsine 双对偶]\label{thm:conclusion-endpoint-reciprocal-arcsine-duality}
设 \(0<\rho<1\)，并定义
\[
y_-(\rho):=\frac{1-\rho}{1+\rho},
\qquad
y_+(\rho):=\frac{1+\rho}{1-\rho}.
\]
记附录中的端点视界变量为 \(Y_\rho\)，并令
\[
Z_\rho:=Y_\rho^{-1}.
\]
则 \(Y_\rho\) 与 \(Z_\rho\) 的支撑都等于
\[
[y_-(\rho),y_+(\rho)],
\]
并分别具有密度
\[
f_\rho(y)=\frac{1}{\pi y\sqrt{(y-y_-(\rho))(y_+(\rho)-y)}},
\]
\[
g_\rho(z)=\frac{1}{\pi\sqrt{(z-y_-(\rho))(y_+(\rho)-z)}},
\qquad
f_\rho(y)=\frac{g_\rho(y)}{y}.
\]
此外，
\[
\mathbb E[Y_\rho]=1,
\qquad
\mathbb E[\log Y_\rho]=\log(1-\rho^2),
\qquad
\mathbb P(Y_\rho\ge 1)=\frac{\arccos \rho}{\pi}.
\]
因此，\(Z_\rho\) 是一条平移缩放后的 arcsine 分布，而 \(Y_\rho\) 则是其带 \(1/y\) 权的 reciprocal 对偶像。
\end{theorem}
\begin{proof}
这正是附录中端点视界变量定理的内容。其证明基于：先把 \(Z_\rho\) 写成标准 arcsine 分布的仿射像，得到
\[
g_\rho(z)=\frac{1}{\pi\sqrt{(z-y_-(\rho))(y_+(\rho)-z)}}.
\]
再由变量变换 \(Y_\rho=Z_\rho^{-1}\) 及
\[
\left|\frac{\dd}{\dd y}(y^{-1})\right|=\frac1{y^2},
\qquad
y_-(\rho)y_+(\rho)=1,
\]
即可推出
\[
f_\rho(y)=\frac{g_\rho(y)}{y}.
\]
期望、对数矩与尾概率则由上述密度公式直接积分得到。证毕。
\end{proof}

\begin{corollary}[Poisson--Cauchy 与端点视界的同参双几何]\label{cor:conclusion-poisson-cauchy-endpoint-arcsine-common-parameter}
同一个参数 \(r\in(0,1)\) 同时生成两类严格耦合的几何：
\begin{enumerate}
  \item 端点 Poisson 权经 Cayley 反推得到尺度为
  \[
  \Gamma_r=\frac{1+r}{1-r}
  \]
  的 Cauchy 核；
  \item 端点视界变量得到支撑为
  \[
  [\Gamma_r^{-1},\Gamma_r]
  \]
  的 reciprocal--arcsine 对偶分布。
\end{enumerate}
并且这两类对象共享同一对数缺陷常数
\[
\mathbb E[\log Y_r]
=
\log(1-r^2)
=
\log\frac{4\Gamma_r}{(1+\Gamma_r)^2}.
\]
故端点 Poisson 权、实轴 Cauchy 尺度与端点视界的对数缺陷并非三套独立数据，而是同一参数 \(r\) 的三重表象。
\end{corollary}
\begin{proof}
第一条由定理 \ref{thm:conclusion-endpoint-poisson-cayley-cauchy-exact-scale} 给出。第二条与对数矩闭式由定理 \ref{thm:conclusion-endpoint-reciprocal-arcsine-duality} 给出。最后一式再结合推论 \ref{cor:conclusion-endpoint-radius-cauchy-window-triple-identity} 中
\[
1-r^2=\frac{4\Gamma_r}{(1+\Gamma_r)^2}
\]
即可。证毕。
\end{proof}
