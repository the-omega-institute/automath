\subsection{Fibonacci 深度包上的有限层 gcd-代数、dominant-depth 选择与导体反演}\label{subsec:conclusion-fibadic-finite-layer-gcd-reduced-dominantdepth-conductor-inversion}
沿用小节 \ref{subsec:conclusion-fibadic-new-harmonics-conductor-cyclotomic}、
\ref{subsec:conclusion-fibadic-gcd-operator-bernoulli-ramanujan}、
\ref{subsec:conclusion-fibadic-kernel-semisimplicity-hiddenstate-truncation}
与
\ref{subsec:conclusion-fibadic-packet-primitive-spectrum-cyclic-threshold}
中的记号。此前结论已经分别把 Fibonacci 深度分解写成
\[
\mathcal H_N=\bigoplus_{d\mid N}\mathcal N_d,
\qquad
E_N=\sum_{d\mid N}P_d,
\qquad
T_\alpha=\sum_{n\ge 1}\widehat\alpha(n)P_n,
\]
并把有限层连续观测的谱导体固定为
\[
D(f)=d_{N(f)}.
\]
在此基础上，还可把有限层 gcd-算子的全部代数后果进一步压缩为如下几条终端结构：primitive 中心幂等元正是各深度包投影，有限层代数天然 reduced，长时动力学自动选择唯一 dominant depth，而全部 cyclotomic 包本身已经足以反演 ordinary conductor-to-depth 映射。

\begin{theorem}[有限层 gcd-代数的 primitive 中心幂等元分类]\label{thm:conclusion-fibadic-finite-layer-gcd-primitive-central-idempotents}
对任意 \(N\ge 1\)，定义有限层可见代数
\[
\mathcal A_N:=\{\,T_\alpha|_{\mathcal H_N}:\ \alpha\in c_{00}(\NN)\,\}
\subseteq \End(\mathcal H_N).
\]
则：
\begin{enumerate}
\item \(\mathcal A_N\) 是交换半单代数，并具有完全对角分解
\[
T_\alpha|_{\mathcal H_N}
=
\sum_{d\mid N}\widehat\alpha(d)P_d;
\]
\item \(\mathcal A_N\) 的全部幂等元恰为
\[
\sum_{d\in S}P_d,
\qquad
S\subseteq \{d\mid N:\ a_d\neq 0\};
\]
\item 其中 primitive 中心幂等元恰为单项
\[
P_d
\qquad
(d\mid N,\ a_d\neq 0).
\]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-visible-depth-mobius-primitive-semisimplicity} 与定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization}，
\[
T_\alpha=\sum_{n\ge 1}\widehat\alpha(n)P_n,
\]
而限制到 \(\mathcal H_N=\bigoplus_{d\mid N}\mathcal N_d\) 后即得
\[
T_\alpha|_{\mathcal H_N}
=
\sum_{d\mid N}\widehat\alpha(d)P_d.
\]
因此 \(\mathcal A_N\) 在两两正交的投影族 \(\{P_d\}_{d\mid N,\ a_d\neq 0}\) 下完全对角化，故为交换半单代数。

若 \(E\in\mathcal A_N\) 为幂等元，则存在标量 \(c_d\) 使
\[
E=\sum_{d\mid N}c_dP_d.
\]
由 \(E^2=E\) 与 \(P_dP_e=\delta_{d,e}P_d\)，得到
\[
c_d^2=c_d
\qquad
(\forall d\mid N,\ a_d\neq 0),
\]
故每个 \(c_d\in\{0,1\}\)。于是全部幂等元 precisely 为若干 \(P_d\) 的和。

最后，若某幂等元包含至少两个非零分量，则它可分解为两个非零正交幂等元之和，因而不 primitive；反之，单个 \(P_d\) 显然不可再分裂。由于 \(\mathcal A_N\) 交换，所有幂等元自动中心，结论成立。证毕。
\end{proof}

\begin{corollary}[有限层可见条件期望代数的半单坐标化]\label{cor:conclusion-fibadic-finite-layer-conditional-expectation-semisimple-coordinates}
沿用定理 \ref{thm:conclusion-fibadic-finite-layer-gcd-primitive-central-idempotents} 的记号，并记活跃深度集
\[
\mathcal D_N^{\mathrm{act}}:=\{d\mid N:\ a_d\neq 0\}.
\]
则同一有限层代数 \(\mathcal A_N\) 可等价写成
\[
\mathcal A_N
=
\operatorname{span}\{E_m|_{\mathcal H_N}:m\ge 1\}
=
\operatorname{span}\{P_d:d\in \mathcal D_N^{\mathrm{act}}\}
\cong
\CC^{|\mathcal D_N^{\mathrm{act}}|}.
\]
特别地，\(\mathcal A_N\) 是交换半单代数，其 primitive 幂等元恰为
\[
\{P_d:\ d\in \mathcal D_N^{\mathrm{act}}\}.
\]
\end{corollary}
\begin{proof}
对单点指示函数 \(\mathbf 1_{\{m\}}\) 应用定理 \ref{thm:conclusion-fibadic-conditional-expectation-gcd-algebra} 中
\[
T_\alpha:=\sum_{d\ge 1}\alpha(d)E_d
\]
的定义，可得
\[
E_m=T_{\mathbf 1_{\{m\}}}.
\]
故每个 \(E_m|_{\mathcal H_N}\) 都属于 \(\mathcal A_N\)，从而
\[
\operatorname{span}\{E_m|_{\mathcal H_N}:m\ge 1\}\subseteq \mathcal A_N.
\]
反过来，由定理 \ref{thm:conclusion-fibadic-visible-depth-mobius-primitive-semisimplicity}，
\[
P_d=\sum_{e\mid d}\mu\!\left(\frac{d}{e}\right)E_e.
\]
再结合定理 \ref{thm:conclusion-fibadic-finite-layer-gcd-primitive-central-idempotents} 中
\[
T_\alpha|_{\mathcal H_N}
=
\sum_{d\mid N}\widehat\alpha(d)P_d,
\]
可知 \(\mathcal A_N\) 同时由 \(\{E_m|_{\mathcal H_N}\}\) 与 \(\{P_d\}_{d\in\mathcal D_N^{\mathrm{act}}}\) 张成。

由于各 \(P_d\) 两两正交且其和为 \(\mathcal H_N\) 上恒等算子，映射
\[
\mathcal A_N\longrightarrow \CC^{|\mathcal D_N^{\mathrm{act}}|},
\qquad
\sum_{d\in\mathcal D_N^{\mathrm{act}}}c_dP_d\longmapsto (c_d)_{d\in\mathcal D_N^{\mathrm{act}}}
\]
给出代数同构。primitive 幂等元的描述即为定理 \ref{thm:conclusion-fibadic-finite-layer-gcd-primitive-central-idempotents} 的直接重述。证毕。
\end{proof}

\begin{theorem}[单层条件期望的 gcd--rank 指纹刚性]\label{thm:conclusion-fibadic-gcd-rank-fingerprint-rigidity}
\leanverified{paper\_conclusion\_fibadic\_gcd\_rank\_fingerprint\_rigidity}
对每个 \(N\ge 3\)，定义秩指纹
\[
\mathfrak r_N(m):=\operatorname{rank}(E_m|_{\mathcal H_N})
\qquad
(m\ge 1).
\]
则有显式公式
\[
\mathfrak r_N(m)=F_{\gcd(m,N)}.
\]
并且映射
\[
N\longmapsto \mathfrak r_N(\cdot)
\]
在 \(\{3,4,5,\dots\}\) 上是单射。换言之，有限层 \(\mathcal H_N\) 的全部 gcd--rank 指纹唯一决定 \(N\)。
\end{theorem}
\begin{proof}
第一式正是推论 \ref{cor:conclusion-fibadic-single-depth-projection-rank-gcd}。

若 \(\mathfrak r_N=\mathfrak r_{N'}\)，则取 \(m=N\) 得
\[
F_N=\mathfrak r_N(N)=\mathfrak r_{N'}(N)=F_{\gcd(N,N')}.
\]
由于 Fibonacci 数列在 \(n\ge 2\) 上严格递增且 \(N\ge 3\)，故
\[
\gcd(N,N')=N,
\qquad
N\mid N'.
\]
交换 \(N,N'\) 再取 \(m=N'\) 同理得到 \(N'\mid N\)。于是 \(N=N'\)。证毕。
\end{proof}

\begin{theorem}[有限层 gcd-算子的核、像、秩、谱与行列式闭式]\label{thm:conclusion-fibadic-finite-layer-gcd-kernel-image-spectrum-closedform}
\leanverified{paper\_conclusion\_fibadic\_finite\_layer\_gcd\_kernel\_image\_spectrum\_closedform}
对任意 \(N\ge 1\) 与任意有限支撑算术函数 \(\alpha\)，有
\[
\ker(T_\alpha|_{\mathcal H_N})
=
\bigoplus_{\substack{d\mid N\\ \widehat\alpha(d)=0}}\mathcal N_d,
\qquad
\operatorname{im}(T_\alpha|_{\mathcal H_N})
=
\bigoplus_{\substack{d\mid N\\ \widehat\alpha(d)\neq 0}}\mathcal N_d.
\]
从而
\[
\operatorname{rank}(T_\alpha|_{\mathcal H_N})
=
\sum_{\substack{d\mid N\\ \widehat\alpha(d)\neq 0}} a_d,
\qquad
\dim\ker(T_\alpha|_{\mathcal H_N})
=
\sum_{\substack{d\mid N\\ \widehat\alpha(d)= 0}} a_d.
\]
此外，
\[
\operatorname{Spec}(T_\alpha|_{\mathcal H_N})
=
\{\,\widehat\alpha(d):\ d\mid N,\ a_d\neq 0\,\},
\]
其中 \(\widehat\alpha(d)\) 的代数重数恰为 \(a_d\)。特别地，
\[
\rho(T_\alpha|_{\mathcal H_N})
=
\max_{\substack{d\mid N\\ a_d\neq 0}}|\widehat\alpha(d)|,
\]
\[
\operatorname{Tr}(T_\alpha|_{\mathcal H_N})
=
\sum_{d\mid N}a_d\,\widehat\alpha(d),
\qquad
\det(T_\alpha|_{\mathcal H_N})
=
\prod_{d\mid N}\widehat\alpha(d)^{a_d}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization}，
\[
T_\alpha|_{\mathcal N_d}=\widehat\alpha(d)\,\mathrm{id}_{\mathcal N_d}.
\]
故当 \(\widehat\alpha(d)=0\) 时，整个 \(\mathcal N_d\) 落入核；当 \(\widehat\alpha(d)\neq 0\) 时，\(T_\alpha|_{\mathcal N_d}\) 为同构，故 \(\mathcal N_d\subseteq \operatorname{im}(T_\alpha|_{\mathcal H_N})\)。再用
\[
\mathcal H_N=\bigoplus_{d\mid N}\mathcal N_d,
\]
即可得到核与像的直和分解，以及秩和核维数公式。

同一块对角分解也立即给出谱：每个 \(\mathcal N_d\) 贡献唯一特征值 \(\widehat\alpha(d)\)，重数为
\[
\dim \mathcal N_d=a_d.
\]
故谱半径是这些特征值模长的最大值，而迹与行列式分别是各块标量迹与标量行列式之和与乘积。证毕。
\end{proof}

\begin{theorem}[有限层 gcd-算子的可逆性与 Möbius 型逆算子闭式]\label{thm:conclusion-fibadic-finite-layer-gcd-invertibility-mobius-inverse}
\leanverified{paper\_conclusion\_fibadic\_finite\_layer\_gcd\_invertibility\_mobius\_inverse}
对任意 \(N\ge 1\)，限制算子
\[
T_\alpha|_{\mathcal H_N}
\]
可逆，当且仅当
\[
\widehat\alpha(d)\neq 0
\qquad
(\forall d\mid N\ \text{且}\ a_d\neq 0).
\]
在此情形下，
\[
(T_\alpha|_{\mathcal H_N})^{-1}
=
\sum_{\substack{d\mid N\\ a_d\neq 0}}\widehat\alpha(d)^{-1}P_d.
\]
更进一步，若定义有限支撑函数 \(\lambda_N\) 为
\[
\lambda_N(d):=
\begin{cases}
\widehat\alpha(d)^{-1},& d\mid N,\ a_d\neq 0,\\
0,& \text{否则},
\end{cases}
\]
则存在唯一有限支撑算术函数 \(\beta_N\) 使
\[
(T_\alpha|_{\mathcal H_N})^{-1}=T_{\beta_N}|_{\mathcal H_N},
\]
且
\[
\beta_N(n)=\sum_{k\ge 1}\mu(k)\lambda_N(nk)
=
\sum_{\substack{k\ge 1\\ nk\mid N}}\mu(k)\lambda_N(nk).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-finite-layer-gcd-kernel-image-spectrum-closedform}，可逆当且仅当核为零，也就是每个活跃深度块 \(\mathcal N_d\) 上的标量 \(\widehat\alpha(d)\) 都非零。在该条件下，每一块上的逆恰为
\[
\widehat\alpha(d)^{-1}\,\mathrm{id}_{\mathcal N_d},
\]
因此
\[
(T_\alpha|_{\mathcal H_N})^{-1}
=
\sum_{\substack{d\mid N\\ a_d\neq 0}}\widehat\alpha(d)^{-1}P_d.
\]

再由定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization} 中的 Möbius 反演，给定有限支撑标量函数 \(\lambda_N\) 后，存在唯一有限支撑算术函数 \(\beta_N\) 使
\[
T_{\beta_N}=\sum_{n\ge 1}\lambda_N(n)P_n,
\qquad
\beta_N(n)=\sum_{k\ge 1}\mu(k)\lambda_N(nk).
\]
由于 \(\lambda_N\) 仅支撑在 \(\{d:d\mid N\}\) 上，上式可写成有限和
\[
\beta_N(n)=\sum_{\substack{k\ge 1\\ nk\mid N}}\mu(k)\lambda_N(nk).
\]
将 \(\lambda_N(d)=\widehat\alpha(d)^{-1}\) 代入活跃块即可得到所述逆算子。证毕。
\end{proof}

\begin{corollary}[有限层 gcd-代数是 reduced 的，且零因子具有逐层互补支撑]\label{cor:conclusion-fibadic-finite-layer-gcd-reduced-zero-divisor-complement}
\leanverified{paper\_conclusion\_fibadic\_finite\_layer\_gcd\_reduced\_zero\_divisor\_complement}
对每个 \(N\ge 1\)，有限层代数 \(\mathcal A_N\) 没有非零幂零元。更强地，对任意
\[
T_\alpha|_{\mathcal H_N},\ T_\beta|_{\mathcal H_N}\in \mathcal A_N,
\]
有
\[
T_\alpha T_\beta|_{\mathcal H_N}=0
\iff
\widehat\alpha(d)\widehat\beta(d)=0
\quad
(\forall d\mid N\ \text{且}\ a_d\neq 0).
\]
换言之，零因子恰对应于活跃深度谱支撑的逐层互补。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization}，
\[
T_\alpha|_{\mathcal N_d}=\widehat\alpha(d)\,\mathrm{id},
\qquad
T_\beta|_{\mathcal N_d}=\widehat\beta(d)\,\mathrm{id}.
\]
因此
\[
(T_\alpha T_\beta)|_{\mathcal N_d}
=
\widehat\alpha(d)\widehat\beta(d)\,\mathrm{id}_{\mathcal N_d},
\]
故乘积为零当且仅当每个活跃深度块上都有
\[
\widehat\alpha(d)\widehat\beta(d)=0.
\]

若 \(T_\alpha|_{\mathcal H_N}\) 幂零，则存在 \(m\ge 1\) 使
\[
(T_\alpha|_{\mathcal H_N})^m=0.
\]
于是对每个活跃 \(d\mid N\) 都有
\[
\widehat\alpha(d)^m=0,
\]
故 \(\widehat\alpha(d)=0\)。再由定理 \ref{thm:conclusion-fibadic-finite-layer-gcd-kernel-image-spectrum-closedform}，这迫使
\[
T_\alpha|_{\mathcal H_N}=0.
\]
因此 \(\mathcal A_N\) 是 reduced。证毕。
\end{proof}

\begin{theorem}[有限层 gcd-动力学的 dominant-depth 选择]\label{thm:conclusion-fibadic-finite-layer-dominant-depth-selection}
\leanverified{paper\_conclusion\_fibadic\_finite\_layer\_dominant\_depth\_selection}
设 \(N\ge 1\)，并假定存在唯一的活跃除数 \(d_\ast\mid N\) 使
\[
|\widehat\alpha(d_\ast)|
>
\max_{\substack{d\mid N\\ d\neq d_\ast,\ a_d\neq 0}}
|\widehat\alpha(d)|.
\]
则在算子范数下有极限
\[
\widehat\alpha(d_\ast)^{-m}\,T_\alpha^m|_{\mathcal H_N}
\longrightarrow
P_{d_\ast}
\qquad
(m\to\infty).
\]
因此，有限层 gcd-动力学的长时行为会自动选择唯一 dominant depth，并将其余全部精确深度块指数压缩掉。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-gcd-convolution-diagonalization}，
\[
T_\alpha^m|_{\mathcal H_N}
=
\sum_{d\mid N}\widehat\alpha(d)^mP_d.
\]
两边同除以 \(\widehat\alpha(d_\ast)^m\)，得到
\[
\widehat\alpha(d_\ast)^{-m}T_\alpha^m|_{\mathcal H_N}
=
P_{d_\ast}
+
\sum_{\substack{d\mid N\\ d\neq d_\ast}}
\left(\frac{\widehat\alpha(d)}{\widehat\alpha(d_\ast)}\right)^m P_d.
\]
由唯一 dominant-depth 假设，对每个其余活跃块都有
\[
\left|\frac{\widehat\alpha(d)}{\widehat\alpha(d_\ast)}\right|<1.
\]
因此右端余项在算子范数下指数趋零，极限即为 \(P_{d_\ast}\)。证毕。
\end{proof}

\begin{theorem}[有限层连续观测的 \texorpdfstring{$\operatorname{lcm}$}{lcm} 过滤律]\label{thm:conclusion-fibadic-finite-layer-lcm-filtration-law}
\leanverified{paper\_conclusion\_fibadic\_lcm\_filtration\_law\_seeds}
\leanverified{paper\_conclusion\_fibadic\_lcm\_filtration\_law\_package}
设 \(f,g\in L^2(\widehat X_F,m_H)\) 都经有限层因子化，并记其谱导体深度为
\[
D(f),\qquad D(g).
\]
则
\[
D(f+g)\mid \operatorname{lcm}(D(f),D(g)),
\qquad
D(fg)\mid \operatorname{lcm}(D(f),D(g)).
\]
更强地，若
\[
f\in \mathcal H_d,
\qquad
g\in \mathcal H_e,
\]
则
\[
f+g\in \mathcal H_{\operatorname{lcm}(d,e)},
\qquad
fg\in \mathcal H_{\operatorname{lcm}(d,e)}.
\]
\end{theorem}
\begin{proof}
令
\[
\ell:=\operatorname{lcm}(d,e).
\]
由于 \(d\mid \ell\) 与 \(e\mid \ell\)，由定理 \ref{thm:conclusion-fibadic-newspace-orthogonal-decomposition}，
\[
\mathcal H_d,\mathcal H_e\subseteq \mathcal H_\ell.
\]
而 \(\mathcal H_\ell\) 中的函数都经由有限循环商
\[
\pi_\ell:\widehat X_F\twoheadrightarrow \ZZ/(F_\ell\ZZ)
\]
因子化，故 \(\mathcal H_\ell\) 是一个有限维交换函数代数，因而对加法与乘法封闭。于是
\[
f+g,\ fg\in \mathcal H_\ell.
\]

把 \(d=D(f)\)、\(e=D(g)\) 代入，便得
\[
f+g,\ fg\in \mathcal H_{\operatorname{lcm}(D(f),D(g))}.
\]
再由 \(D(\cdot)\) 的定义，它是使函数落入相应有限层的最小深度，故
\[
D(f+g)\mid \operatorname{lcm}(D(f),D(g)),
\qquad
D(fg)\mid \operatorname{lcm}(D(f),D(g)).
\]
证毕。
\end{proof}

\begin{corollary}[cyclotomic 包对 ordinary conductor-to-depth 的完全反演]\label{cor:conclusion-fibadic-cyclotomic-bundles-recover-conductor-depth-map}
若给定全部 cyclotomic 包多项式
\[
\Pi_d(X)
\qquad
(d\ge 1),
\]
则 ordinary conductor-to-depth 映射
\[
N\longmapsto d_N
\]
可被唯一恢复。更精确地，
\[
d_N=d
\iff
\Phi_N(X)\ \text{在}\ \Pi_d(X)\ \text{中出现且仅出现于此处}.
\]
因此，对任意有限层连续观测 \(f\)，其谱导体
\[
D(f)=d_{N(f)}
\]
也由全部 \(\{\Pi_d\}_{d\ge 1}\) 唯一决定。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-depth-packet-triple-completeness}，
\[
\Pi_d(X)=\prod_{N:\ d_N=d}\Phi_N(X).
\]
而不同 \(N\) 的 cyclotomic 多项式 \(\Phi_N(X)\) 在 \(\ZZ[X]\) 中两两不同且不可约，故从全部 \(\Pi_d\) 的因式分解中，可以对每个 \(N\) 唯一读出它恰落在哪一个深度包里。这便唯一恢复了映射
\[
N\mapsto d_N.
\]

再由定理 \ref{thm:conclusion-fibadic-finite-layer-observable-spectral-conductor}，
\[
D(f)=d_{N(f)},
\]
其中
\[
N(f)
=
\operatorname{lcm}\{\operatorname{cond}(\chi):\ \widehat f(\chi)\neq 0\}.
\]
因此一旦 \(N\mapsto d_N\) 已知，任意有限层连续观测的谱导体 \(D(f)\) 亦随之唯一恢复。证毕。
\end{proof}
