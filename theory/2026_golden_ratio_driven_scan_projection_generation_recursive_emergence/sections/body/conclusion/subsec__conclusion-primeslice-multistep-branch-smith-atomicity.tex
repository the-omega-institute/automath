\begin{theorem}[最优分层 prime-slice 外置的实现数具有 falling-factorial 乘积公式]\label{thm:conclusion-primeslice-optimal-implementations-falling-factorial}
\leanverified{paper\_conclusion\_primeslice\_optimal\_implementations\_falling\_factorial}
设
\[
X_0\xrightarrow{\pi_0}X_1\xrightarrow{\pi_1}\cdots\xrightarrow{\pi_{k-1}}X_k
\]
为有限集合上的满射塔。沿用定理 \ref{thm:killo-layered-prime-slices-finite-depth} 的 prime-slice 记号
\[
\mathcal P_j:=\{p_{2^j(2n-1)}:n\ge 1\}
\qquad(0\le j\le k-1),
\]
并对每层 \(j\) 与每个 \(y\in X_{j+1}\) 记
\[
b_j(y):=\#\pi_j^{-1}(y),
\qquad
B_j:=\max_{y\in X_{j+1}}b_j(y).
\]
固定每层的最小标签集
\[
\Lambda_j\subset \mathcal P_j,
\qquad
|\Lambda_j|=B_j.
\]
把“第 \(j\) 层精确标签”定义为满足下述条件的映射
\[
q_j:X_j\to \Lambda_j,
\]
对每个 \(y\in X_{j+1}\)，纤维限制
\[
q_j|_{\pi_j^{-1}(y)}:\pi_j^{-1}(y)\to \Lambda_j
\]
为单射，从而 \((y,q_j(x_j))\) 唯一确定 \(x_j\)。则恰好使用这组最小标签集的全部精确分层编码族 \((q_j)_{j=0}^{k-1}\) 的总数为
\[
N_{\mathrm{opt}}
=
\prod_{j=0}^{k-1}\ \prod_{y\in X_{j+1}} (B_j)_{b_j(y)},
\]
其中
\[
(B)_r:=\prod_{t=0}^{r-1}(B-t)
\]
为 falling factorial。
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-layered-primeslice-local-alphabet-fibermax}，第 \(j\) 层的最小可逆标签字母表大小恰为 \(B_j\)。在此极小字母表 \(\Lambda_j\) 上，一个映射 \(q_j\) 在层 \(j\) 合法，当且仅当它在每条纤维 \(\pi_j^{-1}(y)\) 上都是单射。

现固定 \(j\) 与 \(y\in X_{j+1}\)。由于
\[
\#\pi_j^{-1}(y)=b_j(y),
\qquad
|\Lambda_j|=B_j,
\]
合法限制
\[
q_j|_{\pi_j^{-1}(y)}:\pi_j^{-1}(y)\hookrightarrow \Lambda_j
\]
的个数恰为把一个 \(b_j(y)\)-元集合单射到一个 \(B_j\)-元集合中的方式数，即
\[
(B_j)_{b_j(y)}.
\]
不同纤维之间的标签选择彼此独立，故对固定层 \(j\)，可用精确标签的总数为
\[
\prod_{y\in X_{j+1}}(B_j)_{b_j(y)}.
\]
再由各层 prime-slice \(\mathcal P_j\) 两两不交，层与层之间的选择亦彼此独立，于是全部精确分层编码族的总数为
\[
\prod_{j=0}^{k-1}\ \prod_{y\in X_{j+1}} (B_j)_{b_j(y)}.
\]
证毕。
\end{proof}

\begin{corollary}[最优 prime-slice 实现空间的 Schur--凹性]\label{cor:conclusion-primeslice-optimal-implementation-schur-concavity}
沿用定理 \ref{thm:conclusion-primeslice-optimal-implementations-falling-factorial} 的记号。
对每一层 \(j\)，把纤维尺寸谱记为
\[
\mathbf b^{(j)}=\bigl(b_j(y)\bigr)_{y\in X_{j+1}},
\qquad
\sum_{y\in X_{j+1}}b_j(y)=|X_j|.
\]
则函数
\[
\mathbf b^{(j)}
\longmapsto
\log\!\Bigl(\prod_{y\in X_{j+1}}(B_j)_{b_j(y)}\Bigr)
\]
在固定总和 \(\sum_y b_j(y)\) 下是 Schur--凹的。等价地，层内纤维尺寸越均衡，最优 prime-slice 实现空间越大；层内纤维越尖峰，最优实现空间越小。
\end{corollary}
\begin{proof}
记
\[
\phi_B(r):=\log(B)_r=\sum_{t=0}^{r-1}\log(B-t)
\qquad(0\le r\le B).
\]
则其离散二阶差分为
\[
\phi_B(r+1)-2\phi_B(r)+\phi_B(r-1)
=
\log\frac{B-r}{B-r+1}<0,
\]
故 \(\phi_B\) 在整数区间上严格凹。于是
\[
\sum_{y\in X_{j+1}}\phi_{B_j}(b_j(y))
=
\log\!\Bigl(\prod_{y\in X_{j+1}}(B_j)_{b_j(y)}\Bigr)
\]
在固定总质量下为 Schur--凹函数。证毕。
\end{proof}

\begin{theorem}[自然扩张的 \texorpdfstring{$\ell$}{l}-步前史极小字母表恰等于最大 \texorpdfstring{$\ell$}{l}-步分支数]\label{thm:conclusion-natural-extension-lstep-minimal-alphabet-submultiplicative}
设 \(X\) 为可数集合，\(T:X\twoheadrightarrow X\) 为有限纤维满射。对每个 \(\ell\ge 1\)，定义最大 \(\ell\)-步分支数
\[
D_\ell(T):=\sup_{y\in X}\#(T^{-\ell}(y))\in \NN\cup\{\infty\}.
\]
则成立：
\begin{enumerate}
  \item 若 \(D_\ell(T)<\infty\)，则存在长度-\(\ell\) 的单符号精确编码
  \[
  \mathcal C^{(\ell)}:X^{\leftarrow}\longrightarrow X\times [D_\ell(T)]^{\NN},
  \]
  使得每个符号只记录一个 \(\ell\)-步前史块，而由当前状态与该符号可以唯一恢复整段 \(\ell\)-步前史；
  \item 更一般地，对任意整数 \(M<D_\ell(T)\)，不存在任何以大小 \(M\) 的字母表实现的长度-\(\ell\) 单符号精确编码；特别地，若 \(D_\ell(T)=\infty\)，则不存在任何有限字母表实现；
  \item 序列 \((D_\ell(T))_{\ell\ge 1}\) 满足次乘法
  \[
  D_{\ell+r}(T)\le D_\ell(T)\,D_r(T)
  \qquad(\ell,r\ge 1),
  \]
  因而极限
  \[
  h_{\mathrm{br}}(T):=\lim_{\ell\to\infty}\frac{1}{\ell}\log D_\ell(T)
  =
  \inf_{\ell\ge 1}\frac{1}{\ell}\log D_\ell(T)
  \]
  存在（允许取值 \(+\infty\)）。
\end{enumerate}
\end{theorem}
\begin{proof}
固定 \(\ell\ge 1\)。映射 \(T^\ell:X\twoheadrightarrow X\) 仍为有限纤维满射，且
\[
\sup_{y\in X}\#(T^\ell)^{-1}(y)=D_\ell(T).
\]
若 \(D_\ell(T)<\infty\)，则可对每个 \(y\in X\) 选取单射
\[
\iota^{(\ell)}_y:(T^\ell)^{-1}(y)\hookrightarrow [D_\ell(T)].
\]
定义
\[
\beta^{(\ell)}(x):=\iota^{(\ell)}_{T^\ell(x)}(x),
\]
并对自然扩张点 \((x_0,x_{-1},x_{-2},\dots)\in X^{\leftarrow}\) 置
\[
\mathcal C^{(\ell)}((x_0,x_{-1},x_{-2},\dots))
:=
\bigl(x_0,(\beta^{(\ell)}(x_{-\ell}),\beta^{(\ell)}(x_{-2\ell}),\beta^{(\ell)}(x_{-3\ell}),\dots)\bigr).
\]
由于 \(x_{-n\ell}\in (T^\ell)^{-1}(x_{-(n-1)\ell})\)，已知当前状态 \(x_{-(n-1)\ell}\) 与符号 \(\beta^{(\ell)}(x_{-n\ell})\) 时，可由 \((\iota^{(\ell)}_{x_{-(n-1)\ell}})^{-1}\) 恢复唯一的 \(x_{-n\ell}\)。而整段块
\[
(x_{-(n-1)\ell-1},\dots,x_{-n\ell})
\]
又由末端点 \(x_{-n\ell}\) 通过向前迭代 \(T\) 唯一确定。故该编码确为长度-\(\ell\) 的单符号精确编码。

反之，若某个字母表大小 \(M\) 满足 \(M<D_\ell(T)\)，则存在 \(y\in X\) 使
\[
\#(T^\ell)^{-1}(y)>M.
\]
任何长度-\(\ell\) 单符号编码在该 \((T^\ell)\)-纤维上都必须把不同 \(\ell\)-步前史块送入大小仅为 \(M\) 的字母表，故由抽屉原理必有两个不同前史块得到相同符号，从而不能由 \((y,\text{symbol})\) 唯一恢复。这证明了第 \(1\)、\(2\) 条。

对第 \(3\) 条，固定 \(y\in X\)。有不交并
\[
T^{-(\ell+r)}(y)
=
\bigsqcup_{z\in T^{-r}(y)}T^{-\ell}(z),
\]
故
\[
\#T^{-(\ell+r)}(y)
\le
\Bigl(\sup_{z\in X}\#T^{-\ell}(z)\Bigr)\cdot \#T^{-r}(y)
\le
D_\ell(T)D_r(T).
\]
对 \(y\) 取上确界即得
\[
D_{\ell+r}(T)\le D_\ell(T)D_r(T).
\]
于是 \((\log D_\ell(T))_{\ell\ge 1}\) 为次可加序列；Fekete 引理遂给出所示极限存在。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$p$}{p}-核谱函数的离散凹完全表征]\label{thm:conclusion-smith-p-kernel-spectrum-discrete-concave-characterization}
设 \(f:\ZZ_{\ge 0}\to \ZZ_{\ge 0}\) 满足 \(f(0)=0\)，并对 \(k\ge 1\) 定义
\[
\Delta f(k):=f(k)-f(k-1).
\]
则下列条件等价：
\begin{enumerate}
  \item 存在某个满行秩整数矩阵 \(A\in M_{m\times n}(\ZZ)\) 与某个素数 \(p\)，使
  \[
  f(k)=\log_p\#\ker(A\bmod p^k)-k(n-m)
  \qquad(k\ge 1),
  \]
  且 \(f(0)=0\)；
  \item 序列 \((\Delta f(k))_{k\ge 1}\) 是非负、单调不增且最终为零的整数列；
  \item \(f\) 是单调不减、离散凹且最终常值的整数值函数。
\end{enumerate}
在这些条件成立时，定义
\[
m_\ell:=\Delta f(\ell)-\Delta f(\ell+1)
\qquad(\ell\ge 1).
\]
则 \(m_\ell\in \ZZ_{\ge 0}\)，且仅有限多个 \(m_\ell\) 非零，并有唯一表示
\[
f(k)=\sum_{\ell\ge 1}m_\ell\min(k,\ell)
\qquad(k\ge 0).
\]
等价地，一个显式实现矩阵可取为
\[
A_f=
\operatorname{diag}\bigl(
\underbrace{p,\dots,p}_{m_1},
\underbrace{p^2,\dots,p^2}_{m_2},
\underbrace{p^3,\dots,p^3}_{m_3},
\dots
\bigr),
\]
其中只有有限多个对角块非空；若需置于任意给定的满行秩尺寸口径，只需在此对角块外附加单位对角元与零列，而不改变局部谱函数。
\end{theorem}
\begin{proof}
由定理 \ref{thm:killo-smith-loss-spectrum}，任意满行秩整数矩阵 \(A\) 的 \(p\)-局部核谱函数都可写为
\[
f_p(k)=\sum_{i=1}^{m}\min(k,e_i(p)),
\qquad
\Delta f_p(k)=\#\{i:e_i(p)\ge k\}.
\]
因此 \(\Delta f_p(k)\) 显然非负、单调不增且最终为零，\(f_p\) 亦单调不减、离散凹且最终常值。这证明了 \(1\Rightarrow 2\Rightarrow 3\) 与 \(1\Rightarrow 3\)。

反过来，若 \((\Delta f(k))_{k\ge 1}\) 非负、单调不增且最终为零，则
\[
m_\ell=\Delta f(\ell)-\Delta f(\ell+1)\ge 0,
\]
并且只有有限多个 \(m_\ell\) 非零。对任意 \(k\ge 1\)，望远镜求和给出
\[
\sum_{\ell\ge k}m_\ell=\Delta f(k).
\]
于是
\[
f(k)
=
\sum_{j=1}^{k}\Delta f(j)
=
\sum_{j=1}^{k}\sum_{\ell\ge j}m_\ell
=
\sum_{\ell\ge 1}m_\ell\min(k,\ell).
\]
按重数把每个高度 \(\ell\) 复制 \(m_\ell\) 次，即得一组有限多重集 \(\{e_i\}\)，并令
\[
A_f=\operatorname{diag}(p^{e_1},\dots,p^{e_r}).
\]
再由定理 \ref{thm:killo-smith-loss-spectrum} 便有
\[
\log_p\#\ker(A_f\bmod p^k)=f(k)
\]
（此处 \(n=m=r\)），从而得到 \(2\Rightarrow 1\)。

最后，整数值函数的离散凹性恰等价于前向差分单调不增；单调不减恰等价于前向差分非负；最终常值恰等价于前向差分最终为零。因此 \(2\) 与 \(3\) 等价。唯一性则由
\[
m_\ell=\Delta f(\ell)-\Delta f(\ell+1)
\]
立即得到。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$p$}{p}-核谱函数在直和下形成自由交换幺半群]\label{cor:conclusion-smith-p-kernel-spectrum-free-commutative-monoid}
\leanverified{paper\_conclusion\_smith\_p\_kernel\_spectrum\_free\_commutative\_monoid}
对每个 \(\ell\ge 1\) 定义原子谱函数
\[
g_\ell(k):=\min(k,\ell)
\qquad(k\ge 0).
\]
则所有 \(p\)-核谱函数在逐点加法下构成一个以 \(\{g_\ell\}_{\ell\ge 1}\) 为自由生成系的交换幺半群。更精确地：
\begin{enumerate}
  \item 每个 \(p\)-核谱函数 \(f\) 都唯一分解为
  \[
  f=\sum_{\ell\ge 1}m_\ell g_\ell,
  \qquad
  m_\ell\in \ZZ_{\ge 0},
  \]
  且只有有限多个 \(m_\ell\) 非零；
  \item 对应 Smith 指数多重集里，“高度 \(\ell\)” 的出现次数恰为 \(m_\ell\)；
  \item 下列三个不变量可直接由该原子分解读出：
  \[
  \Delta f(1)=\sum_{\ell\ge 1}m_\ell,
  \qquad
  \min\{k\ge 0:\ f(k)=f(k+1)\}=\max\{\ell:\ m_\ell\neq 0\},
  \]
  \[
  \lim_{k\to\infty}f(k)=\sum_{\ell\ge 1}\ell\,m_\ell.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-smith-p-kernel-spectrum-discrete-concave-characterization} 已给出唯一表示
\[
f(k)=\sum_{\ell\ge 1}m_\ell\min(k,\ell)=\sum_{\ell\ge 1}m_\ell g_\ell(k).
\]
因此 \(\{g_\ell\}_{\ell\ge 1}\) 生成全部 \(p\)-核谱函数，且由系数
\[
m_\ell=\Delta f(\ell)-\Delta f(\ell+1)
\]
的唯一恢复知该生成是自由的。

f_{A\oplus B,p}=f_{A,p}+f_{B,p}.
\]
故上述自由原子分解与矩阵直和完全相容，于是得到交换幺半群结构。

最后三条数值公式都由

\Delta g_\ell(k)=
\begin{cases}
1,& k\le \ell,\\
0,& k>\ell
\end{cases}
\]
直接读出：\(\Delta f(1)\) 统计全部原子数，平台起点是最大原子高度，而平台值是所有原子高度之和。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$p$}{p}-核谱函数与有限支撑正测度锥的严格同构]\label{thm:conclusion-smith-p-kernel-spectrum-measure-cone-isomorphism}

\qquad
m_\ell\in \ZZ_{\ge 0},
\]

f(k)=\int_{\ZZ_{\ge 1}}\min(k,\ell)\,\dd\mu_f(\ell)
=
\sum_{\ell\ge 1}m_\ell\min(k,\ell)
\qquad(k\ge 0),
\]

\[
\Delta f(k)=f(k)-f(k-1)=\mu_f([k,\infty)\cap \ZZ)
\qquad(k\ge 1).

f(k)=\sum_{\ell\ge 1}m_\ell\min(k,\ell),

\[
\mu_f:=\sum_{\ell\ge 1}m_\ell\,\delta_\ell
\]
即得积分表示。

\[
\Delta\min(k,\ell)=
\begin{cases}
1,& \ell\ge k,\\
0,& \ell<k,
\end{cases}
\]
遂有
\[
\Delta f(k)
=
\sum_{\ell\ge 1}m_\ell\,\Delta\min(k,\ell)
=
\sum_{\ell\ge k}m_\ell
=
\mu_f([k,\infty)\cap \ZZ).
\]
唯一性则由
\[
m_\ell=\Delta f(\ell)-\Delta f(\ell+1)
\]
直接恢复。最后，矩阵直和对应局部核谱函数逐点加法，而测度加法显然对应原子重数逐项相加，故得到严格同构。证毕。
\end{proof}

  \mathcal C((x_0,x_{-1},x_{-2},\dots))
  =
  \bigl(x_0,(\beta(x_{-1}),\beta(x_{-2}),\dots)\bigr)\in X\times\NN^\NN
  \]
  给出精确可逆化，并且“一层一坐标”的使用率在最坏情形下已是最小。
\end{enumerate}
\end{theorem}
\begin{proof}
固定 \(k\ge 1\)。题设正是定理 \ref{thm:xi-layered-prime-register-sharp-natural-extension} 的最坏情形假设，因此任意精确分层外置在深度 \(k\) 上都至少需要 \(k\) 个有效 prime slices；而该定理同时给出逐层写入一枚有效 prime 的可达构造。这证明了第 \(1\) 条。

若存在某个有限层外置能够精确编码全部无限前历史，则其限制到任意深度 \(k\) 都给出一个只使用固定有限层数的 \(k\)-步精确外置。取 \(k\) 大于该层数，便与第 \(1\) 条矛盾，故第 \(2\) 条成立。

最后，第 \(3\) 条的充分性正是定理 \ref{thm:killo-natural-extension-branch-register} 的内容；而其最坏情形最优性则由第 \(1\) 条立即给出。证毕。
\end{proof}
