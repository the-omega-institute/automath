\subsection{Zeckendorf--Gödel 自然密度测度的点态局部维数谱与有限证书全谱}\label{subsec:conclusion-zg-density-pointwise-spectrum-finite-certificate}
在前一小节已经获得统一柱渐近、零密度稀疏律与零信息维数之后，结合地址前缀等距性与无穷 prime-register solenoid 的有限可见因子化，可把算术自然密度进一步提升为一套真正的点态几何结论。其核心特征在于：柱质量的非齐次对数修正不仅决定几乎处处零维的典型几何，也同时允许在每个非空局部地址片内实现从 \(0\) 到 \(+\infty\) 的全部局部维数。

\begin{theorem}[算术自然密度测度的点态柱渐近与局部维数公式]\label{thm:conclusion-zg-pointwise-local-dimension-formula}
沿用定理 \ref{thm:xi-time-part60f-zg-cylinder-measure-local-dimension-zero} 的记号，固定
\[
\Xi_B:\Sigma_{\mathrm{ZG}}\to K_{B,v},
\qquad
\mu_B:=(\Xi_B)_\#\nu_{\mathrm{ZG}},
\qquad
B=2^L.
\]
对任意
\[
z=(z_k)_{k\ge 1}\in \Sigma_{\mathrm{ZG}},
\qquad
x:=\Xi_B(z),
\]
都有柱集渐近
\[
\nu_{\mathrm{ZG}}([z_1,\dots,z_n])
=
\frac{e^{-\gamma}}{\delta_{\mathrm{ZG}}\log p_n}
\exp\!\left(-\sum_{k=1}^{n}z_k\log p_k\right)
\bigl(1+o(1)\bigr).
\]
因此 \(\mu_B\) 在 \(x\) 处的上下局部维数满足
\[
\underline d_{\mu_B}(x)
:=
\liminf_{r\downarrow 0}
\frac{\log \mu_B\!\bigl(\mathrm B_{K_{B,v}}(x,r)\bigr)}{\log r}
=
\frac{1}{\log B}
\liminf_{n\to\infty}
\frac{\sum_{k=1}^{n}z_k\log p_k+\log\log p_n}{n},
\]
\[
\overline d_{\mu_B}(x)
:=
\limsup_{r\downarrow 0}
\frac{\log \mu_B\!\bigl(\mathrm B_{K_{B,v}}(x,r)\bigr)}{\log r}
=
\frac{1}{\log B}
\limsup_{n\to\infty}
\frac{\sum_{k=1}^{n}z_k\log p_k+\log\log p_n}{n}.
\]
\end{theorem}
\begin{proof}
第一条正是定理 \ref{thm:conclusion-zg-cylinder-uniform-asymptotic} 对前缀
\(
(z_1,\dots,z_n)
\)
的直接重写。

另一方面，由定理 \ref{thm:xi-time-part9g-holographic-prefix-isometry-on-line} 的柱--球对应，
\[
\Xi_B([z_1,\dots,z_n])
=
\mathrm B_{K_{B,v}}(x,B^{-n}).
\]
由于 \(K_{B,v}\) 继承的是 \(B\)-进超度量，任意充分小的半径 \(r\) 都与某个离散尺度 \(B^{-n}\) 给出同一球，因此局部维数可沿 \(r=B^{-n}\) 计算。
故
\[
\mu_B\!\bigl(\mathrm B_{K_{B,v}}(x,B^{-n})\bigr)
=
\nu_{\mathrm{ZG}}([z_1,\dots,z_n]).
\]
将前述渐近取对数，可得
\[
-\log \mu_B\!\bigl(\mathrm B_{K_{B,v}}(x,B^{-n})\bigr)
=
\sum_{k=1}^{n}z_k\log p_k+\log\log p_n+O(1).
\]
再除以
\[
-\log B^{-n}=n\log B
\]
并分别取 \(\liminf\) 与 \(\limsup\)，便得到两条局部维数公式。证毕。
\end{proof}

\begin{theorem}[每个非空地址 cylinder 内的完整局部维数谱]\leanverified{paper\_conclusion\_zg\_full\_local\_dimension\_spectrum\_every\_cylinder}\label{thm:conclusion-zg-full-local-dimension-spectrum-every-cylinder}
对任意非空 cylinder
\[
C=[a_1,\dots,a_m]\subset \Sigma_{\mathrm{ZG}}
\]
与任意
\[
\alpha\in[0,\infty],
\]
都存在点
\[
x\in \Xi_B(C)\subset K_{B,v}
\]
使得 \(\mu_B\) 在 \(x\) 处的局部维数存在且满足
\[
d_{\mu_B}(x)=\alpha.
\]
因此，\(\mu_B\) 的点态局部维数谱恰为整个区间
\[
[0,\infty].
\]
\end{theorem}
\begin{proof}
有限前缀 \((a_1,\dots,a_m)\) 仅对定理
\ref{thm:conclusion-zg-pointwise-local-dimension-formula} 中的分子贡献 \(O(1)\)，因而不影响极限；我们只需在其后附加一条 admissible 尾部。

\begin{enumerate}
  \item 若 \(\alpha=0\)，则取尾部最终恒零。此时
  \[
  \sum_{k=1}^{n}z_k\log p_k=O(1),
  \]
  由定理 \ref{thm:conclusion-zg-pointwise-local-dimension-formula} 立即得到
  \[
  d_{\mu_B}(x)=0.
  \]

  \item 若 \(\alpha=\infty\)，则在前缀之后取交替尾部
  \[
  z_{m+2j-1}=0,
  \qquad
  z_{m+2j}=1
  \qquad (j\ge 1).
  \]
  于是对每个 \(n\ge 1\)，
  \[
  \sum_{k=1}^{m+2n}z_k\log p_k
  \ge
  \sum_{j=1}^{n}\log p_{m+2j}.
  \]
  由素数定理 \(p_n\sim n\log n\) 可知 \(\log p_{m+2j}\sim \log j\)，故右端为
  \[
  n\log n+O(n).
  \]
  因而
  \[
  \frac{1}{m+2n}\sum_{k=1}^{m+2n}z_k\log p_k\longrightarrow \infty.
  \]
  再由定理 \ref{thm:conclusion-zg-pointwise-local-dimension-formula}，得到
  \[
  d_{\mu_B}(x)=\infty.
  \]

  \item 若 \(0<\alpha<\infty\)，令
  \[
  w_j:=\log p_{m+2j},
  \qquad
  c:=2\alpha\log B.
  \]
  由素数定理，\(w_j\asymp \log j\)，特别地 \(w_j\to\infty\)。取 \(J\) 充分大，使得
  \[
  \frac{c}{w_j}\le \frac12
  \qquad (j\ge J).
  \]
  对 \(j\ge J\)，取独立 Bernoulli 随机变量 \(Y_j\) 满足
  \[
  \mathbb P(Y_j=1)=\frac{c}{w_j},
  \qquad
  \mathbb P(Y_j=0)=1-\frac{c}{w_j},
  \]
  并对 \(j<J\) 置 \(Y_j=0\)。记
  \[
  X_j:=Y_j w_j-c\,\mathbf 1_{\{j\ge J\}}.
  \]
  则 \(\EE[X_j]=0\)，且
  \[
  \operatorname{Var}(X_j)\le c\,w_j\ll \log j.
  \]
  因而
  \[
  \sum_{j=1}^{\infty}\frac{\operatorname{Var}(X_j)}{j^2}<\infty.
  \]
  由 Kolmogorov 强大数定律，
  \[
  \frac{1}{n}\sum_{j=1}^{n}X_j\longrightarrow 0
  \qquad \text{几乎处处}.
  \]
  于是对几乎处处 realization，都有
  \[
  \frac{1}{n}\sum_{j=1}^{n}Y_j w_j
  =
  c\,\frac{n-J+1}{n}+o(1)
  \longrightarrow c.
  \]
  因而可固定其中一条 realization，记为 \(y_j\in\{0,1\}\)。
  现在定义尾部
  \[
  z_{m+2j-1}=0,
  \qquad
  z_{m+2j}=y_j
  \qquad (j\ge 1).
  \]
  该序列显然 admissible。并且
  \[
  \sum_{k=1}^{m+2n}z_k\log p_k
  =
  O(1)+\sum_{j=1}^{n}y_j w_j
  =
  2\alpha n\log B+o(n).
  \]
  因而
  \[
  \frac{1}{m+2n}\sum_{k=1}^{m+2n}z_k\log p_k
  \longrightarrow
  \alpha\log B.
  \]
  对奇数长度 \(m+2n+1\) 亦同样成立，因为新增一位恒为 \(0\)。再由定理 \ref{thm:conclusion-zg-pointwise-local-dimension-formula}，便得到
  \[
  d_{\mu_B}(x)=\alpha.
  \]
\end{enumerate}

三种情形合并即得结论。证毕。
\end{proof}

\begin{corollary}[零维层与无限维层的拓扑共存]\label{cor:conclusion-zg-zero-infinite-local-dimension-topological-coexistence}
定义
\[
E_0:=\{x\in K_{B,v}: d_{\mu_B}(x)=0\},
\qquad
E_\infty:=\{x\in K_{B,v}: \overline d_{\mu_B}(x)=\infty\}.
\]
则 \(E_0\) 在 \(K_{B,v}\) 中稠密，而 \(E_\infty\) 是 \(K_{B,v}\) 中的稠密 \(G_\delta\) 集。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-zg-full-local-dimension-spectrum-every-cylinder} 取 \(\alpha=0\)，可知每个非空 cylinder 中都含有零维点；经 \(\Xi_B\) 推前便得 \(E_0\) 在 \(K_{B,v}\) 中稠密。

对 \(M,R\in\NN\)，定义
\[
\mathcal U_{M,R}
:=
\bigcup_{n\ge M}
\ \bigcup_{\substack{\mathbf z\in\{0,1\}^{n}\ \mathrm{admissible}\\[1mm]
\sum_{k=1}^{n}z_k\log p_k\ge Rn}}
[\mathbf z]
\subset \Sigma_{\mathrm{ZG}}.
\]
它按定义是开集。现证其稠密。任取非空 cylinder
\[
C=[a_1,\dots,a_m].
\]
取整数 \(\ell\) 充分大，使得
\[
\ell\ge \max\{M,2m+2\},
\qquad
\log p_\ell\ge 6R.
\]
在 \(C\) 后附加长度 \(2\ell+1\) 的 admissible 词
\[
0(10)^\ell,
\]
得到更长 cylinder \(C'\subset C\)。若 \(N:=m+2\ell+1\) 为其长度，则 \(N\le 3\ell\)，并且
\[
\sum_{k=1}^{N}z_k\log p_k
\ge
\sum_{j=\lceil \ell/2\rceil}^{\ell}\log p_{m+2j}
\ge
\frac{\ell}{2}\log p_\ell
\ge
3R\ell
\ge
RN.
\]
故 \(C'\subset \mathcal U_{M,R}\)，从而 \(\mathcal U_{M,R}\) 稠密。

于是
\[
\mathcal G_\infty:=\bigcap_{M=1}^{\infty}\bigcap_{R=1}^{\infty}\mathcal U_{M,R}
\]
是 \(\Sigma_{\mathrm{ZG}}\) 中的稠密 \(G_\delta\) 集。对任意 \(z\in \mathcal G_\infty\) 与任意 \(R\ge 1\)，存在任意大的 \(n\) 使
\[
\sum_{k=1}^{n}z_k\log p_k\ge Rn.
\]
由定理 \ref{thm:conclusion-zg-pointwise-local-dimension-formula}，这蕴含
\[
\overline d_{\mu_B}(\Xi_B(z))\ge \frac{R}{\log B}.
\]
令 \(R\to\infty\)，得
\[
\overline d_{\mu_B}(\Xi_B(z))=\infty.
\]
因此
\[
\Xi_B(\mathcal G_\infty)\subseteq E_\infty.
\]
又 \(\Xi_B\) 是从 \(\Sigma_{\mathrm{ZG}}\) 到 \(K_{B,v}\) 的同胚，故 \(\Xi_B(\mathcal G_\infty)\) 仍为 \(K_{B,v}\) 中的稠密 \(G_\delta\) 集，从而 \(E_\infty\) 本身也是稠密 \(G_\delta\) 集。证毕。
\end{proof}

\begin{theorem}[全息推前测度的几乎处处 exact dimension 为零]\label{thm:conclusion-zg-pushforward-exact-dimension-zero}
对 \(\mu_B\)-几乎处处的 \(x\in K_{B,v}\)，局部维数存在且满足
\[
d_{\mu_B}(x)=0.
\]
换言之，\(\mu_B\) 是一个 exact-dimensional 测度，且其 exact dimension 等于 \(0\)。
\end{theorem}
\begin{proof}
在 \((\Sigma_{\mathrm{ZG}},\nu_{\mathrm{ZG}})\) 上记
\[
N_n(z):=\sum_{k=1}^{n}z_k.
\]
由定理 \ref{thm:conclusion-zg-density-measure-sparsity-law}，
\[
\EE_{\nu_{\mathrm{ZG}}}[N_n]
=
\log\log p_n+O(1)
=
O(\log\log n).
\]
取 dyadic 子序列 \(n_j:=2^j\)。对任意 \(\varepsilon>0\)，由 Markov 不等式，
\[
\nu_{\mathrm{ZG}}\!\left(
N_{2^j}>\frac{\varepsilon\,2^j}{\log 2^j}
\right)
\le
\frac{\log 2^j}{\varepsilon\,2^j}\EE[N_{2^j}]
\ll
\frac{j\log j}{2^j}.
\]
右端可求和，故由 Borel--Cantelli 引理，
\[
\frac{N_{2^j}\log 2^j}{2^j}\longrightarrow 0
\qquad
\nu_{\mathrm{ZG}}\text{-a.s.}
\]
若 \(2^j\le n<2^{j+1}\)，则由单调性
\[
\frac{N_n\log n}{n}
\le
2\,
\frac{N_{2^{j+1}}\log 2^{j+1}}{2^{j+1}}.
\]
因此
\[
\frac{N_n\log n}{n}\longrightarrow 0
\qquad
\nu_{\mathrm{ZG}}\text{-a.s.}
\]
再由 \(\log p_n\asymp \log n\)，得到
\[
\sum_{k=1}^{n}z_k\log p_k
\le
N_n\log p_n
=
o(n)
\qquad
\nu_{\mathrm{ZG}}\text{-a.s.}
\]
并且
\[
\log\log p_n=o(n).
\]
代入定理 \ref{thm:conclusion-zg-pointwise-local-dimension-formula}，便得
\[
d_{\mu_B}(\Xi_B(z))=0
\qquad
\nu_{\mathrm{ZG}}\text{-a.s.}
\]
最后利用
\[
\mu_B=(\Xi_B)_\#\nu_{\mathrm{ZG}},
\]
即可把结论推前到 \(\mu_B\) 上。证毕。
\end{proof}

任意非空 cylinder

使得

\begin{theorem}[有限证书对点态局部维数的绝对盲性]\label{thm:conclusion-zg-finite-certificate-local-dimension-blindness}
设 \(S\subset \mathbb P\) 为可数无穷素数集，\(S_0\Subset S\) 为任意有限子集，\(C\subset\Sigma_{\mathrm{ZG}}\) 为任意非空 cylinder，并取任意固定
\[
\eta\in\Sigma_{S_0}.
\]
在乘积空间 \(K_{B,v}\times \Sigma_S\) 上定义有限证书原子
\[
\mathcal F(C,\eta)
:=
\{(\Xi_B(z),\xi):\ z\in C,\ \pi_{S\to S_0}(\xi)=\eta\},
\]
并记
\[
U_C:=\Xi_B(C)\subset K_{B,v},
\]
以及局部维数函数
\[
\mathfrak d(x,\xi):=d_{\mu_B}(x).
\]
则 \(\mathfrak d\) 在 \(\mathcal F(C,\eta)\) 上取满整个区间
\[
[0,\infty].
\]
因此，对有限证书 \(\sigma\)-代数
\[
\mathcal G(C,S_0)
:=
\sigma(U_C\times \Sigma_S)\vee \sigma(\pi_{S\to S_0}),
\]
\(\mathfrak d\) 不可测。换言之，任何仅依赖有限地址前缀与有限 prime shadow 的证书，都不能对点态局部维数作出任何非平凡分类。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zg-finite-certificate-fiber-full-spectrum}，对任意
\[
\alpha\in[0,\infty]
\]
都存在
\[
(x,\xi)\in \mathcal F(C,\eta)
\]
使得
\[
\mathfrak d(x,\xi)=d_{\mu_B}(x)=\alpha.
\]
故 \(\mathfrak d\) 在每个非空有限证书原子 \(\mathcal F(C,\eta)\) 上都取满整个区间 \([0,\infty]\)。

若 \(\mathfrak d\) 对 \(\mathcal G(C,S_0)\) 可测，则它必须在每个 \(\mathcal G(C,S_0)\)-原子上取常值；而 \(\mathcal F(C,\eta)\) 正是这类原子之一，故与前述全谱结论矛盾。证毕。
\end{proof}

\begin{corollary}[有限证书因子不存在局部维数读出器]\label{cor:conclusion-zg-finite-factor-no-local-dimension-reader}
设
\[
\pi:K_{B,v}\times \Sigma_S\to Y
\]
是任意一个仅由有限地址前缀与有限 prime shadow 决定的因子映射。则不存在函数
\[
G:Y\to [0,\infty]
\]
使得对所有 \((x,\xi)\in K_{B,v}\times \Sigma_S\) 都有
\[
d_{\mu_B}(x)=G(\pi(x,\xi)).
\]
特别地，任何有限证书因子都不能把 \(\mu_B\) 的点态局部维数压缩成一个有限可读出的结构标签。
\end{corollary}
\begin{proof}
这类因子的纤维正是有限证书原子的并。若存在 \(G\) 满足
\[
d_{\mu_B}(x)=G(\pi(x,\xi)),
\]
则局部维数函数 \(\mathfrak d(x,\xi):=d_{\mu_B}(x)\) 必在每条因子纤维上取常值，从而对相应的有限证书 \(\sigma\)-代数可测。这与定理 \ref{thm:conclusion-zg-finite-certificate-local-dimension-blindness} 矛盾。证毕。
\end{proof}

\begin{theorem}[每个非空局部地址片内的零维典型性与 residual infinite spikes]\label{thm:conclusion-zg-local-cylinder-zero-typical-residual-infinite}
对任意非空 cylinder
\[
C\subset \Sigma_{\mathrm{ZG}},
\qquad
U_C:=\Xi_B(C)\subset K_{B,v},
\]
都有：
\[
\mu_B\text{-a.e. }x\in U_C
\Longrightarrow
d_{\mu_B}(x)=0,
\]
并且
\[
E_\infty
:=
\{x\in K_{B,v}:\overline d_{\mu_B}(x)=\infty\}
\]
与 \(U_C\) 的交是 \(U_C\) 中的稠密 \(G_\delta\) 集。
因而在每一个局部地址片内，\(\mu_B\)-典型几何完全塌缩为零维，而拓扑上却始终保留一束 residual 的无限上局部维数尖峰。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zg-pushforward-exact-dimension-zero}，对 \(\mu_B\)-几乎处处的 \(x\in K_{B,v}\) 有
\[
d_{\mu_B}(x)=0.
\]
又因 \(\Xi_B\) 是从 \(\Sigma_{\mathrm{ZG}}\) 到 \(K_{B,v}\) 的同胚，任意非空 cylinder 的像 \(U_C\) 都是 \(K_{B,v}\) 中的非空 clopen 集；同时 \(\nu_{\mathrm{ZG}}(C)>0\)，故
\[
\mu_B(U_C)=\nu_{\mathrm{ZG}}(C)>0.
\]
于是上述几乎处处结论限制到 \(U_C\) 上，得到第一条。

另一方面，推论 \ref{cor:conclusion-zg-zero-infinite-local-dimension-topological-coexistence} 已给出：\(E_\infty\) 在 \(K_{B,v}\) 中是稠密 \(G_\delta\) 集。与任意非空开集 \(U_C\) 相交后，\(E_\infty\cap U_C\) 仍是 \(U_C\) 中的稠密 \(G_\delta\) 集。证毕。
\end{proof}

\begin{theorem}[全息推前测度在其支撑上处处非局部 doubling]\label{thm:conclusion-zg-nowhere-locally-doubling}
测度 \(\mu_B\) 在其支撑 \(K_{B,v}\) 的任何非空开集上都不满足局部 doubling 性。更精确地，不存在非空开集
\[
U\subset K_{B,v},
\]
常数 \(C<\infty\) 与半径 \(r_0>0\)，使得对一切 \(x\in U\) 及 \(0<r<r_0\) 都有
\[
\mu_B\bigl(B(x,2r)\bigr)\le C\,\mu_B\bigl(B(x,r)\bigr).
\]
\end{theorem}
\begin{proof}
反设存在这样的 \(U,C,r_0\)。固定 \(x\in U\)，取 \(0<r<r_0\) 并令 \(n\ge 1\) 使
\[
2^{-n}r_0\le r<2^{-n+1}r_0.
\]
迭代 doubling 不等式得到
\[
\mu_B\bigl(B(x,r_0)\bigr)
\le
C^n\,\mu_B\bigl(B(x,2^{-n}r_0)\bigr)
\le
C^n\,\mu_B\bigl(B(x,r)\bigr),
\]
从而
\[
\mu_B\bigl(B(x,r)\bigr)\ge C^{-n}\mu_B\bigl(B(x,r_0)\bigr).
\]
由于 \(n\asymp \log_2(r_0/r)\)，遂有
\[
\overline d_{\mu_B}(x)
:=
\limsup_{r\downarrow 0}
\frac{\log \mu_B(B(x,r))}{\log r}
\le
\log_2 C
\qquad (x\in U).
\]
这表明 \(U\) 中每一点的上局部维数都有限。

但由定理 \ref{thm:conclusion-zg-local-cylinder-zero-typical-residual-infinite}，每个非空开集都包含稠密的 \(E_\infty\)-点，亦即包含满足
\[
\overline d_{\mu_B}(x)=\infty
\]
的点，与上式矛盾。故 \(\mu_B\) nowhere locally doubling。证毕。
\end{proof}
