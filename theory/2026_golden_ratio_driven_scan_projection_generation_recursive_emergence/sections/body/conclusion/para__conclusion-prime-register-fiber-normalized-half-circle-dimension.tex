\paragraph{素数寄存器子纤维的归一化半圆维与伸缩嵌入准则}\label{par:conclusion-prime-register-fiber-normalized-half-circle-dimension}
沿用 \S\ref{subsec:conclusion-prime-register-rigidity-complexity-conjugacy-composition} 的素数寄存器幺半群
\(
\mathcal P=\NN^{(\mathbb P)}
\)
与素数序列 \((p_k)_{k\ge 1}\)。
对任意素数子集 \(S\subseteq\mathbb P\)，记其子纤维
\[
\mathcal P_S:=\NN^{(S)}\subseteq \mathcal P,
\qquad
S_n:=S\cap\{p_1,\dots,p_n\}.
\]
由于 \(S_n\) 有限，有自然同构 \(\mathcal P_{S_n}\cong \NN^{|S_n|}\)。
据此引入归一化半圆维上/下极限：
\[
\overline{\cdim}_+(\mathcal P_S)
:=\limsup_{n\to\infty}\frac{\cdim_+(\mathcal P_{S_n})}{\cdim_+(\NN^n)}\in[0,1],
\qquad
\underline{\cdim}_+(\mathcal P_S)
:=\liminf_{n\to\infty}\frac{\cdim_+(\mathcal P_{S_n})}{\cdim_+(\NN^n)}\in[0,1],
\]
其中 \(\cdim_+\) 为定义 \ref{def:half-circle-dimension} 的半圆维。

\begin{theorem}[归一化半圆维等同于素数索引密度]\label{thm:conclusion-prime-register-fiber-cdim-density}
对任意 \(S\subseteq\mathbb P\) 有恒等式
\[
\overline{\cdim}_+(\mathcal P_S)=\limsup_{n\to\infty}\frac{|S_n|}{n},
\qquad
\underline{\cdim}_+(\mathcal P_S)=\liminf_{n\to\infty}\frac{|S_n|}{n}.
\]
若极限
\(
\rho(S):=\lim_{n\to\infty}|S_n|/n
\)
存在，则
\(
\overline{\cdim}_+(\mathcal P_S)=\underline{\cdim}_+(\mathcal P_S)=\rho(S)
\)。
并且该值在有限对称差变形 \(S\triangle S'\) 下保持不变。
\end{theorem}

\begin{proof}
由 \(\mathcal P_{S_n}\cong \NN^{|S_n|}\) 与例 \ref{ex:circle-dimension-examples} 给出的 \(\cdim_+(\NN^d)=d/2\)，有
\[
\frac{\cdim_+(\mathcal P_{S_n})}{\cdim_+(\NN^n)}
=\frac{\cdim_+(\NN^{|S_n|})}{\cdim_+(\NN^n)}
=\frac{|S_n|/2}{n/2}
=\frac{|S_n|}{n}.
\]
取 \(\limsup\) 与 \(\liminf\) 即得首两式。
若 \(\rho(S)\) 存在，则两者同为 \(\rho(S)\)。
若 \(S\triangle S'\) 有限，则 \(|S_n|-|S'_n|=O(1)\)，故 \(|S_n|/n\) 的极限（或上下极限）不变，从而不变性成立。
证毕。
\end{proof}

\begin{proposition}[任意分数半圆维的可实现性]\label{prop:conclusion-prime-register-fiber-cdim-realize-any-alpha}
对任意 \(\alpha\in[0,1]\)，存在素数子集 \(S\subseteq\mathbb P\) 使
\[
\overline{\cdim}_+(\mathcal P_S)=\underline{\cdim}_+(\mathcal P_S)=\alpha.
\]
\end{proposition}

\begin{proof}
取任意无理数 \(\theta\in(0,1)\) 并令
\[
A:=\{n\in\NN:\ \{n\theta\}<\alpha\},
\]
其中 \(\{\cdot\}\) 为小数部分。
由 Weyl 等分布定理，集合 \(A\) 的自然密度存在且等于 \(\alpha\)，即
\(
\lim_{n\to\infty}|A\cap\{1,\dots,n\}|/n=\alpha
\)。
令
\(
S:=\{p_k:\ k\in A\}
\)，则 \(|S_n|=|A\cap\{1,\dots,n\}|\)。
由定理 \ref{thm:conclusion-prime-register-fiber-cdim-density} 得结论。
证毕。
\end{proof}

\begin{definition}[\texorpdfstring{$K$}{K}-伸缩型生成元单射]\label{def:conclusion-prime-register-K-stretch-embedding}
令 \(S,T\subseteq\mathbb P\)，并写其索引集为
\[
A:=\{k\in\NN:\ p_k\in S\},\qquad B:=\{k\in\NN:\ p_k\in T\}.
\]
称幺半群同态 \(\Phi:\mathcal P_S\to \mathcal P_T\) 为\textbf{\(K\)-伸缩型生成元单射}，若存在 \(k_0\) 使得对所有 \(k\in A\) 且 \(k\ge k_0\)，存在某个 \(\ell\in B\) 满足 \(\ell\le Kk\) 且
\[
\Phi(e_k)=e_\ell,
\]
其中 \(e_k\in\mathcal P\) 表示基向量 \(e_k(p_k)=1\)（其余为 \(0\)）。
并要求上述赋值在充分大处对生成元为单射（不同 \(k\) 的像对应不同 \(\ell\)）。
\end{definition}

\begin{theorem}[\texorpdfstring{$K$}{K}-伸缩嵌入的密度障碍与充分条件]\label{thm:conclusion-prime-register-K-stretch-density-criterion}
令 \(S,T\subseteq\mathbb P\) 且 \(K\ge 1\)。
\begin{enumerate}
  \item 若存在 \(K\)-伸缩型生成元单射 \(\Phi:\mathcal P_S\hookrightarrow \mathcal P_T\)，则必有
  \[
  \overline{\cdim}_+(\mathcal P_S)\le K\cdot \overline{\cdim}_+(\mathcal P_T).
  \]
  \item 若
  \[
  \overline{\cdim}_+(\mathcal P_S)\ <\ K\cdot \underline{\cdim}_+(\mathcal P_T),
  \]
  则存在某个 \(K\)-伸缩型生成元单射 \(\Phi:\mathcal P_S\hookrightarrow \mathcal P_T\)。
\end{enumerate}
\end{theorem}

\begin{proof}
对 (1)，固定充分大的 \(n\)，考虑 \(A\cap[1,n]\) 对应的生成元集合 \(\{e_k:k\in A\cap[1,n]\}\)。
由 \(K\)-伸缩条件，其像均落在 \(\{e_\ell:\ell\in B\cap[1,Kn]\}\) 中；又生成元层单射给出
\[
|A\cap[1,n]|\le |B\cap[1,Kn]|.
\]
两边除以 \(n\) 并取 \(\limsup\) 得
\[
\limsup_{n\to\infty}\frac{|A\cap[1,n]|}{n}
\le
K\cdot \limsup_{m\to\infty}\frac{|B\cap[1,m]|}{m}.
\]
由定理 \ref{thm:conclusion-prime-register-fiber-cdim-density}，左侧为 \(\overline{\cdim}_+(\mathcal P_S)\)，右侧为 \(K\overline{\cdim}_+(\mathcal P_T)\)，得到所示必要条件。

对 (2)，取 \(\varepsilon>0\) 使
\(
\overline{\cdim}_+(\mathcal P_S)+\varepsilon<K(\underline{\cdim}_+(\mathcal P_T)-\varepsilon)
\)。
由上下极限定义，存在 \(N\) 使对所有 \(n\ge N\) 有
\[
|A\cap[1,n]|\le (\overline{\cdim}_+(\mathcal P_S)+\varepsilon)n,
\qquad
|B\cap[1,Kn]|\ge (\underline{\cdim}_+(\mathcal P_T)-\varepsilon)Kn.
\]
从而对所有 \(n\ge N\) 成立 \(|B\cap[1,Kn]|\ge |A\cap[1,n]|\)。
将 \(A\cap[N,\infty)\) 递增枚举为 \(a_1<a_2<\cdots\)。
定义贪心匹配：令 \(f(a_i)\) 为集合 \(B\cap[1,Ka_i]\) 中尚未使用的最小元素。
由于在第 \(i\) 步前使用的元素数不超过 \(|A\cap[1,a_i]|\le i+O(1)\)，且 \(|B\cap[1,Ka_i]|\ge |A\cap[1,a_i]|\)，故该过程不会阻塞，并得到单射 \(f:A\to B\) 满足 \(f(k)\le Kk\) 对所有充分大的 \(k\) 成立。
据此令 \(\Phi(e_k):=e_{f(k)}\)（\(k\in A\) 充分大），并在有限多个小 \(k\) 上任意补齐为生成元层单射，再由自由交换幺半群的泛性质延拓为幺半群同态，即得所需的 \(K\)-伸缩型生成元单射。
证毕。
\end{proof}

\begin{corollary}[伸缩常数的刚性下界]\label{cor:conclusion-prime-register-K-stretch-rigidity-lb}
在定义 \ref{def:conclusion-prime-register-K-stretch-embedding} 的设定下，若存在 \(K\)-伸缩型生成元单射 \(\mathcal P_S\hookrightarrow \mathcal P_T\) 且 \(\overline{\cdim}_+(\mathcal P_T)>0\)，则必有
\[
K\ \ge\ \frac{\overline{\cdim}_+(\mathcal P_S)}{\overline{\cdim}_+(\mathcal P_T)}.
\]
若 \(\overline{\cdim}_+(\mathcal P_T)=0\) 而 \(\overline{\cdim}_+(\mathcal P_S)>0\)，则不存在任何有限 \(K\) 的 \(K\)-伸缩型生成元单射。
\end{corollary}

\begin{proof}
由定理 \ref{thm:conclusion-prime-register-K-stretch-density-criterion}(1) 直接整理得到。
证毕。
\end{proof}

\begin{theorem}[素数完备化轴的不可混叠与同构刚性]\label{thm:conclusion-prime-adic-axis-no-aliasing-rigidity}
令 \(S,T\subseteq\mathbb P\)，并定义
\[
\widehat{\ZZ}_S:=\prod_{p\in S}\ZZ_p,
\qquad
\widehat{\ZZ}_T:=\prod_{p\in T}\ZZ_p
\]
取乘积拓扑。
则：
\begin{enumerate}
  \item 任意连续群同态 \(F:\widehat{\ZZ}_S\to\widehat{\ZZ}_T\) 按素数坐标分离：对每个 \(q\in T\)，复合投影 \(\pi_q\circ F\) 在 \(p\neq q\) 的坐标上恒为零；特别地若 \(q\notin S\)，则 \(\pi_q\circ F\equiv 0\)。
  \item \(\widehat{\ZZ}_S\simeq \widehat{\ZZ}_T\)（拓扑群同构）当且仅当 \(S=T\)。
  \item 若存在连续单射 \(F:\widehat{\ZZ}_S\hookrightarrow \widehat{\ZZ}_T\)，则必有 \(S\subseteq T\)。
\end{enumerate}
\end{theorem}

\begin{proof}
固定 \(q\in T\)。连续性保证 \(\pi_q\circ F:\widehat{\ZZ}_S\to \ZZ_q\) 仅依赖于有限多个坐标，因而因子化为
\(
\prod_{p\in S_0}\ZZ_p\to \ZZ_q
\)
对某个有限 \(S_0\subseteq S\)。
当 \(p\neq q\) 时，\(\ZZ_p\) 为交换 pro-\(p\) 群而 \(\ZZ_q\) 为 pro-\(q\) 群；连续同态的像同时为 pro-\(p\) 与 pro-\(q\) 群，只能为平凡群。
故 \(\pi_q\circ F\) 仅可能通过 \(p=q\) 坐标起作用，且若 \(q\notin S\) 则必为零，得到 (1)。

(2) 若 \(\widehat{\ZZ}_S\simeq\widehat{\ZZ}_T\)，将 (1) 施用于同构及其逆得到 \(S\subseteq T\) 与 \(T\subseteq S\)，故 \(S=T\)；反向显然。

(3) 若 \(F\) 为单射，则对任意 \(q\in S\)，\(q\)-坐标方向上的非平凡元不能全部落入核，故存在某个 \(t\in T\) 使 \(\pi_t\circ F\) 在 \(q\)-坐标上非零。
由 (1) 必有 \(t=q\)，从而 \(q\in T\)。故 \(S\subseteq T\)。
证毕。
\end{proof}

\begin{proposition}[连通相位轴与 profinite 寄存轴的结构正交]\label{prop:conclusion-connected-profinite-orthogonality}
设 \(K\) 为紧连通交换群，\(P\) 为非平凡 profinite 交换群。
则：
\begin{enumerate}
  \item 任意连续群同态 \(K\to P\) 皆为零同态；
  \item 任意连续群同态 \(P\to K\) 的像必为有限子群；
  \item 若 \(P\) 无穷，则不存在连续单射 \(P\hookrightarrow \TT^d\)（任意有限维环面）。
\end{enumerate}
\end{proposition}

\begin{proof}
(1) 连通集在连续映射下的像连通，而 profinite 群全不连通，故其唯一连通子群为平凡群，从而像必平凡。
(2) \(P\) 全不连通，连续像亦全不连通；而 \(\TT^d\) 的任意闭子群同构于 \(\TT^k\times F\)（\(F\) 有限），故其全不连通闭子群只能是有限群。
(3) 若存在连续单射，则其像为 \(\TT^d\) 的无穷全不连通闭子群，与 (2) 矛盾。
证毕。
\end{proof}

\begin{conjecture}[受控伸缩实现类中的归一化半圆维完备性]\label{conj:conclusion-prime-register-normalized-cdim-completeness}
以 \(\{\mathcal P_S\}_{S\subseteq\mathbb P}\) 为对象，取存在某个有限 \(K\) 的 \(K\)-伸缩型生成元单射作为态射，得到范畴 \(\mathsf{Reg}_{\mathrm{temp}}\)。
猜想对任意 \(S,T\subseteq\mathbb P\)，不等式
\[
\overline{\cdim}_+(\mathcal P_S)\le K\cdot \overline{\cdim}_+(\mathcal P_T)
\]
不仅是 \(K\)-伸缩型生成元单射 \(\mathcal P_S\hookrightarrow \mathcal P_T\) 的必要条件，而且在 \(\mathsf{Reg}_{\mathrm{temp}}\) 中也是充分条件（不再需要 \(\underline{\cdim}_+\) 的额外假设）。
\end{conjecture}
