\subsection{Fibonacci 深度投影的 Haar 幂等演算与 bin-fold 规范群的中尺度表示论}\label{subsec:conclusion-fibadic-haar-randomdepth-binfold-conjugacy-schur}
沿用小节 \ref{subsec:conclusion-fibadic-profinite-collapse-character-conductor-rigidity}、小节 \ref{subsec:conclusion-fibadic-gcd-operator-bernoulli-ramanujan}、小节 \ref{subsec:conclusion-foldbin-center-zeta-fibonacci-measure-rigidity}、定理 \ref{thm:conclusion-binfold-kappa-gauge-affine-law} 与定理 \ref{thm:conclusion-foldbin-gauge-zeta-conjugacy-commuting} 的记号。此前结论已经把 Fibonacci-adic 逆极限、精确深度分解
\[
\mathcal H_d=\bigoplus_{e\mid d}\mathcal N_e
\]
及深度包 \(\Pi_d\) 压缩为 gcd-卷积算子代数与 Ramanujan 包核，又把 bin-fold 规范群
\[
G_m\cong \prod_{w\in X_m}\mathfrak S_{d_m^{\mathrm{bin}}(w)}
\]
的中心、表示 \(\zeta\) 函数与体积密度固定为显式公式。继续把这两条链并置，可进一步抽出如下七条后果：一方面，有限深度投影本身可以在 Haar 测度代数中实现为一族卷积幂等元；另一方面，bin-fold 规范群的共轭类复杂度、交换概率与二阶同调则在 \((2\varphi)^{m/2}\)、\(2^m\) 与 \(\varphi^{2m}\) 三个不同尺度上形成分离的表示论层级。

\begin{theorem}[有限深度条件期望的 Haar--卷积实现]\label{thm:conclusion-fibadic-haar-conditional-expectation-convolution}
\leanverified{paper\_conclusion\_fibadic\_haar\_conditional\_expectation\_convolution}
设
\[
\widehat X_F\cong \widehat{\ZZ},
\qquad
\pi_d:\widehat X_F\twoheadrightarrow \ZZ/(F_d\ZZ),
\qquad
H_d:=\ker \pi_d.
\]
记 \(\sigma_d\) 为 \(H_d\) 上的归一化 Haar 概率测度。则对任意
\[
f\in L^2(\widehat X_F,m_H),
\]
有限深度条件期望 \(E_d\) 具有精确卷积表示
\[
E_df=f*\sigma_d.
\]
此外，对任意 \(d,e\ge 1\) 都有
\[
\sigma_d*\sigma_e=\sigma_{\gcd(d,e)}.
\]
\end{theorem}
\begin{proof}
先对任意连续特征 \(\chi\in \widehat X_F^\vee\) 计算：
\[
(\chi*\sigma_d)(x)
=
\chi(x)\int_{H_d}\chi(h)\,\dd \sigma_d(h).
\]
积分项等于 \(1\) 当且仅当 \(\chi\) 在 \(H_d\) 上平凡，否则等于 \(0\)。而 \(\chi\) 在 \(H_d\) 上平凡，等价于 \(\chi\) 经由有限商 \(\pi_d\) 因子化，也即 \(\chi\) 落在可见深度子空间 \(\mathcal H_d\) 的角色基中。故 \(\chi*\sigma_d\) 正是 \(\chi\) 在 \(\mathcal H_d\) 上的正交投影。由于角色构成 \(L^2(\widehat X_F,m_H)\) 的正交基，由线性与闭包即得
\[
E_df=f*\sigma_d.
\]

再对任意角色 \(\chi\) 取 Fourier 变换，得到
\[
\widehat{\sigma_d*\sigma_e}(\chi)
=
\widehat{\sigma_d}(\chi)\widehat{\sigma_e}(\chi)
=
\mathbf 1_{\delta(\chi)\mid d}\,\mathbf 1_{\delta(\chi)\mid e}
=
\mathbf 1_{\delta(\chi)\mid \gcd(d,e)}
=
\widehat{\sigma_{\gcd(d,e)}}(\chi).
\]
Pontryagin 对偶下 Fourier 变换对有界 Borel 测度是忠实的，故
\[
\sigma_d*\sigma_e=\sigma_{\gcd(d,e)}.
\]
证毕。
\end{proof}

\begin{theorem}[精确深度投影的 signed Haar 幂等元与 Ramanujan--shell 公式]\label{thm:conclusion-fibadic-signed-haar-idempotent-ramanujan-shell}
对每个 \(d\ge 1\)，定义 signed measure
\[
\nu_d:=\sum_{e\mid d}\mu\!\left(\frac{d}{e}\right)\sigma_e.
\]
则：
\begin{enumerate}
  \item 精确深度投影 \(P_d\) 具有卷积表示
  \[
  P_df=f*\nu_d;
  \]
  \item \(\{\nu_d\}_{d\ge 1}\) 在测度代数中形成一族两两正交的自伴幂等元：
  \[
  \nu_d*\nu_e=\delta_{d,e}\,\nu_d;
  \]
  \item 其 Fourier 变换满足
  \[
  \widehat{\nu_d}(\chi)=\mathbf 1_{\delta(\chi)=d};
  \]
  \item 相应的精确深度核
  \[
  K_d(x):=\sum_{\delta(\chi)=d}\chi(x)
  \]
  具有显式物理空间表示
  \[
  K_d(x)=\sum_{e\mid d}\mu\!\left(\frac{d}{e}\right)F_e\,\mathbf 1_{H_e}(x).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-fibadic-haar-conditional-expectation-convolution}，
\[
E_d=\cdot*\sigma_d.
\]
另一方面，小节 \ref{subsec:conclusion-fibadic-gcd-operator-bernoulli-ramanujan} 中已证明
\[
E_d=\sum_{e\mid d}P_e.
\]
对整除格施行 Möbius 反演，得到
\[
P_d=\sum_{e\mid d}\mu(d/e)E_e=\cdot *\nu_d,
\]
这就是第一条。

第二条可由算子恒等式直接得到。因为 \(\{P_d\}\) 是两两正交投影，故
\[
P_dP_e=\delta_{d,e}P_d.
\]
将其改写为卷积核语言，即有
\[
(\cdot *\nu_d)*\nu_e=\delta_{d,e}(\cdot *\nu_d),
\]
从而
\[
\nu_d*\nu_e=\delta_{d,e}\nu_d.
\]

第三条则由第一条在角色基上的作用立即得到：
\[
\widehat{\nu_d}(\chi)
=
\sum_{e\mid d}\mu(d/e)\widehat{\sigma_e}(\chi)
=
\sum_{\substack{e\mid d\\ \delta(\chi)\mid e}}\mu(d/e)
=
\mathbf 1_{\delta(\chi)=d}.
\]

最后，对每个 \(e\ge 1\) 定义
\[
D_e(x):=\sum_{\delta(\chi)\mid e}\chi(x).
\]
这些角色恰是经 \(\pi_e\) 从有限循环群 \(\ZZ/(F_e\ZZ)\) 拉回的全部角色。有限 Fourier 正交性给出
\[
D_e(x)=F_e\,\mathbf 1_{H_e}(x).
\]
而
\[
D_e=\sum_{a\mid e}K_a.
\]
再对整除格施行一次 Möbius 反演，即得
\[
K_d(x)=\sum_{e\mid d}\mu(d/e)\,F_e\,\mathbf 1_{H_e}(x).
\]
证毕。
\end{proof}

\begin{theorem}[随机深度平均的完全谱刻画与精确混合判据]\label{thm:conclusion-fibadic-random-depth-average-spectral-criterion}
设 \(\alpha:\NN_{\ge 1}\to[0,1]\) 为有限支撑概率权，满足
\[
\sum_{d\ge 1}\alpha(d)=1.
\]
定义随机深度平均算子
\[
M_\alpha:=\sum_{d\ge 1}\alpha(d)E_d.
\]
则：
\begin{enumerate}
  \item \(M_\alpha\) 是 \(L^2(\widehat X_F,m_H)\) 上的自伴 Markov 收缩；
  \item 在每个精确深度新空间 \(\mathcal N_n\) 上，
  \[
  M_\alpha|_{\mathcal N_n}=\widehat\alpha(n)\,\mathrm{id},
  \qquad
  \widehat\alpha(n):=\sum_{n\mid d}\alpha(d);
  \]
  \item \(M_\alpha^k\) 向常数投影 \(E_1\) 在算子范数下收敛，当且仅当
  \[
  \widehat\alpha(n)<1
  \qquad
  \text{对所有 }n>1\text{ 且 }\mathcal N_n\neq 0;
  \]
  \item 在上述情形下，对一切 \(k\ge 1\) 都有精确范数公式
  \[
  \|M_\alpha^k-E_1\|_{2\to 2}
  =
  \sup_{n>1:\,\mathcal N_n\neq 0}\widehat\alpha(n)^k.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
每个 \(E_d\) 都是条件期望，因此正、保 \(1\)、自伴，且为 \(L^2\)-收缩。有限凸组合 \(M_\alpha\) 仍保持这些性质，故第一条成立。

由
\[
E_d=\sum_{n\mid d}P_n
\]
可得
\[
M_\alpha
=
\sum_d\alpha(d)\sum_{n\mid d}P_n
=
\sum_n\left(\sum_{n\mid d}\alpha(d)\right)P_n
=
\sum_n\widehat\alpha(n)P_n.
\]
于是第二条立即成立。

再由 \(E_1=P_1\) 得
\[
M_\alpha^k-E_1=\sum_{n>1}\widehat\alpha(n)^kP_n.
\]
因为 \(\{P_n\}\) 两两正交，故
\[
\|M_\alpha^k-E_1\|_{2\to 2}
=
\sup_{n>1:\,\mathcal N_n\neq 0}\widehat\alpha(n)^k.
\]
这既给出第四条，也说明当且仅当所有非平凡分块上都满足 \(\widehat\alpha(n)<1\) 时，\(M_\alpha^k\) 才向 \(E_1\) 收敛。证毕。
\end{proof}

\begin{corollary}[均匀除数随机化的精确谱隙公式]\label{cor:conclusion-fibadic-uniform-divisor-randomization-gap}
\leanverified{paper\_conclusion\_fibadic\_uniform\_divisor\_randomization\_gap}
设
\[
M=\prod_{j=1}^r p_j^{a_j}
\]
为正整数，并定义均匀除数平均算子
\[
A_M:=\frac{1}{\tau(M)}\sum_{d\mid M}E_d.
\]
则在每个 \(\mathcal N_n\) 上，
\[
A_M|_{\mathcal N_n}=
\begin{cases}
\dfrac{\tau(M/n)}{\tau(M)}\,\mathrm{id}, & n\mid M,\\[1.2ex]
0, & n\nmid M.
\end{cases}
\]
从而其第二特征值满足精确公式
\[
\rho_2(A_M)=\max_{1\le j\le r}\frac{a_j}{a_j+1},
\]
故谱隙为
\[
1-\rho_2(A_M)=\min_{1\le j\le r}\frac{1}{a_j+1}.
\]
\end{corollary}
\begin{proof}
在定理 \ref{thm:conclusion-fibadic-random-depth-average-spectral-criterion} 中取
\[
\alpha(d)=\frac{\mathbf 1_{d\mid M}}{\tau(M)}.
\]
则
\[
\widehat\alpha(n)
=
\frac{1}{\tau(M)}\#\{d\mid M:\ n\mid d\}.
\]
若 \(n\nmid M\)，该计数为 \(0\)。若 \(n\mid M\)，则计数恰为 \(\tau(M/n)\)，故第一式成立。

若
\[
n=\prod_{j=1}^r p_j^{b_j},
\qquad
0\le b_j\le a_j,
\]
则
\[
\frac{\tau(M/n)}{\tau(M)}
=
\prod_{j=1}^r \frac{a_j-b_j+1}{a_j+1}.
\]
在所有非平凡 \(n>1\) 中，为使此值最大，只需令某一个 \(b_j=1\) 而其余 \(b_i=0\)。于是
\[
\rho_2(A_M)=\max_j\frac{a_j}{a_j+1},
\]
谱隙即为
\[
1-\rho_2(A_M)=\min_j\frac{1}{a_j+1}.
\]
证毕。
\end{proof}

\begin{theorem}[规范群共轭类数的显式主项]\label{thm:conclusion-binfold-gauge-conjugacy-midscale-asymptotic}
设
\[
G_m=\prod_{w\in X_m}\mathfrak S_{d_m^{\mathrm{bin}}(w)},
\qquad
k(G_m):=\#\{\text{\(G_m\) 的共轭类}\}.
\]
记
\[
\Lambda:=\frac{2}{\varphi},
\qquad
C_{\mathrm{conj}}
:=
\frac{\pi}{\sqrt5}\sqrt{\frac23}\left(\varphi+\varphi^{-1/2}\right).
\]
则
\[
\log k(G_m)
=
C_{\mathrm{conj}}(2\varphi)^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr)
\qquad(m\to\infty).
\]
\end{theorem}
\begin{proof}
由 \(k(\mathfrak S_n)=p(n)\) 可得
\[
k(G_m)=\prod_{w\in X_m}p\!\bigl(d_m^{\mathrm{bin}}(w)\bigr),
\qquad
\log k(G_m)=\sum_{w\in X_m}\log p\!\bigl(d_m^{\mathrm{bin}}(w)\bigr).
\]
Hardy--Ramanujan 渐近给出
\[
\log p(n)=\pi\sqrt{\frac{2n}{3}}+O(\log n).
\]
故
\[
\log k(G_m)
=
\pi\sqrt{\frac23}\sum_{w\in X_m}\sqrt{d_m^{\mathrm{bin}}(w)}
+O\!\left(\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)\right).
\]

由两态一致渐近，
\[
d_m^{\mathrm{bin}}(w)=\varphi^{-w_m}\Lambda^m+O(1)
\]
且误差对 \(w\) 一致；并且
\[
\#\{w\in X_m:w_m=0\}=F_{m+1},
\qquad
\#\{w\in X_m:w_m=1\}=F_m.
\]
因此
\[
\sum_{w\in X_m}\sqrt{d_m^{\mathrm{bin}}(w)}
=
F_{m+1}\Lambda^{m/2}
+\varphi^{-1/2}F_m\Lambda^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr).
\]
由 Binet 公式，
\[
F_{m+1}\Lambda^{m/2}\sim \frac{\varphi}{\sqrt5}(2\varphi)^{m/2},
\qquad
\varphi^{-1/2}F_m\Lambda^{m/2}\sim \frac{\varphi^{-1/2}}{\sqrt5}(2\varphi)^{m/2}.
\]
另一方面，
\[
\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)=O(m|X_m|)=O(m\varphi^m)=o\!\bigl((2\varphi)^{m/2}\bigr).
\]
把这些估计代回，即得
\[
\log k(G_m)
=
\frac{\pi}{\sqrt5}\sqrt{\frac23}\left(\varphi+\varphi^{-1/2}\right)(2\varphi)^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr).
\]
证毕。
\end{proof}

\begin{theorem}[规范群交换概率的超指数非阿贝尔律]\label{thm:conclusion-binfold-gauge-commuting-probability-superexponential}
\leanverified{paper\_conclusion\_binfold\_gauge\_commuting\_probability\_superexponential}
设
\[
\mathrm{cp}(G_m):=
\frac{1}{|G_m|^2}\#\{(g,h)\in G_m^2:\ gh=hg\}
\]
为 \(G_m\) 的交换概率，并记
\[
C_{\mathrm{vol}}:=1+\frac{\log\varphi}{\varphi\sqrt5}.
\]
则有精确渐近
\[
-\log \mathrm{cp}(G_m)
=
2^m\left(m\log\frac{2}{\varphi}-C_{\mathrm{vol}}\right)
-C_{\mathrm{conj}}(2\varphi)^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr),
\]
其中 \(C_{\mathrm{conj}}\) 如定理 \ref{thm:conclusion-binfold-gauge-conjugacy-midscale-asymptotic}。
\end{theorem}
\begin{proof}
对任意有限群 \(G\) 都有
\[
\mathrm{cp}(G)=\frac{k(G)}{|G|}.
\]
故
\[
-\log \mathrm{cp}(G_m)=\log |G_m|-\log k(G_m).
\]
其中，定理 \ref{thm:conclusion-binfold-kappa-gauge-affine-law} 给出
\[
\log |G_m|
=
2^m\left(
m\log\frac{2}{\varphi}
-1-\frac{\log\varphi}{\varphi\sqrt5}
\right)
+O(m\varphi^m),
\]
也就是
\[
\log |G_m|
=
2^m\left(m\log\frac{2}{\varphi}-C_{\mathrm{vol}}\right)
+o\!\bigl((2\varphi)^{m/2}\bigr),
\]
因为 \(m\varphi^m=o((2\varphi)^{m/2})\)。再由定理 \ref{thm:conclusion-binfold-gauge-conjugacy-midscale-asymptotic},
\[
\log k(G_m)
=
C_{\mathrm{conj}}(2\varphi)^{m/2}
+o\!\bigl((2\varphi)^{m/2}\bigr).
\]
相减即得结论。证毕。
\end{proof}

\begin{theorem}[规范群 Schur 乘子的二次爆炸律]\label{thm:conclusion-binfold-gauge-schur-multiplier-quadratic-explosion}
\leanverified{paper\_conclusion\_binfold\_gauge\_schur\_multiplier\_quadratic\_explosion}
记
\[
n_m:=|X_m|=F_{m+2}.
\]
则存在 \(m_0\) 使得对一切 \(m\ge m_0\)，都有
\[
H_2(G_m,\ZZ)\cong (\ZZ/2\ZZ)^{\,n_m(n_m+1)/2}.
\]
\end{theorem}
\begin{proof}
由两态一致渐近，
\[
\min_{w\in X_m}d_m^{\mathrm{bin}}(w)\longrightarrow+\infty
\qquad(m\to\infty).
\]
故存在 \(m_0\) 使得对所有 \(m\ge m_0\) 与一切 \(w\in X_m\)，都有
\[
d_m^{\mathrm{bin}}(w)\ge 7.
\]
于是每个对称群因子都满足标准事实
\[
H_2(\mathfrak S_n,\ZZ)\cong \ZZ/2\ZZ,
\qquad
\mathfrak S_n^{\mathrm{ab}}\cong \ZZ/2\ZZ
\qquad(n\ge 7).
\]

对有限直积群，Schur 乘子满足
\[
H_2\!\left(\prod_{i=1}^r G_i,\ZZ\right)
\cong
\bigoplus_{i=1}^r H_2(G_i,\ZZ)
\oplus
\bigoplus_{1\le i<j\le r} G_i^{\mathrm{ab}}\otimes G_j^{\mathrm{ab}}.
\]
将
\[
G_i=\mathfrak S_{d_i},
\qquad
r=n_m
\]
代入，便得
\[
H_2(G_m,\ZZ)
\cong
(\ZZ/2\ZZ)^{n_m}
\oplus
(\ZZ/2\ZZ)^{\binom{n_m}{2}}
\cong
(\ZZ/2\ZZ)^{n_m(n_m+1)/2}.
\]
证毕。
\end{proof}

因此，Fibonacci 深度过滤与 bin-fold 规范群之间还存在一条更深的中尺度接口：在 \(\widehat X_F\) 一侧，有限深度条件期望并不是抽象投影，而可在紧开子群的 Haar 测度代数中实现为卷积幂等元；经 Möbius 反演后，这一演算又进一步细化为精确深度投影的 signed Haar 体系与 Ramanujan shell 公式，并把任意有限支撑深度随机化完全压缩为 \(\alpha\mapsto \widehat\alpha\) 的算术谱判据。与之平行，在 \(G_m\) 一侧，规范群的表示复杂度则在三个不同尺度上分裂：共轭类数增长精确落在 \((2\varphi)^{m/2}\) 的混合指数层，交换概率由 \(2^m\) 体积主项与 \((2\varphi)^{m/2}\) 共轭类修正共同控制，而 Schur 乘子却在 \(|X_m|^2\) 尺度上发生二次爆炸。由此，Fibonacci-adic 深度几何、有限深度随机化与 bin-fold 规范群的中尺度表示论不再是彼此独立的三套对象，而被同一条 gcd--Möbius--Haar 过滤与纤维直方图增长律同时锁定。
