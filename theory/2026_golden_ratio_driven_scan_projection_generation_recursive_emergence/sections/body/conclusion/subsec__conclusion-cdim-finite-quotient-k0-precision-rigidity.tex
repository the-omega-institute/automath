\subsection{圆维的有限商刻画、\texorpdfstring{$K_0$}{K0} 唯一性与相位压缩刚性}\label{subsec:conclusion-cdim-finite-quotient-k0-precision-rigidity}
本小节在定义 \ref{def:circle-dimension} 与定义 \ref{def:half-circle-dimension} 的口径下，组织一条“可测量刻画 \(\to\) 公理唯一性 \(\to\) 有限秩扩张 \(\to\) 账本下界 \(\to\) 去范畴化唯一存活量 \(\to\) 相位压缩复杂度约束”的闭合结论链。

\begin{theorem}[圆维的有限商增长刻画]\label{thm:conclusion-cdim-finite-quotient-growth}
设 \(G\) 为有限生成阿贝尔群，并定义
\[
q_G(n):=|G/nG|\qquad(n\ge 1).
\]
则极限存在且
\[
\lim_{n\to\infty}\frac{\log q_G(n)}{\log n}=\cdim(G).
\]
更精细地，存在有界函数 \(u:\NN\to\NN\) 使
\[
q_G(n)=n^{\cdim(G)}u(n),\qquad \sup_{n\ge 1}u(n)<\infty.
\]
\leanverified{paper_conclusion_cdim_finite_quotient_growth_seeds}
\end{theorem}
\begin{proof}
由有限生成阿贝尔群结构定理，
\[
G\simeq \ZZ^r\oplus T,\qquad r=\cdim(G),\ T\ \text{有限}.
\]
故
\[
G/nG\simeq (\ZZ/n\ZZ)^r\oplus (T/nT),
\]
从而
\[
q_G(n)=n^r|T/nT|.
\]
令 \(u(n):=|T/nT|\)，则 \(1\le u(n)\le |T|\)。
于是
\[
\frac{\log q_G(n)}{\log n}
=r+\frac{\log u(n)}{\log n}\xrightarrow[n\to\infty]{}r=\cdim(G).
\]
证毕。
\end{proof}

\begin{theorem}[短正合可加标尺的圆维唯一性]\label{thm:conclusion-cdim-exact-additive-uniqueness}
设 \(\mathcal A=\mathrm{Ab}_{\mathrm{fg}}\)。
令
\[
I:\mathrm{Ob}(\mathcal A)\to\RR_{\ge 0}
\]
满足：
\begin{enumerate}
  \item 同构不变；
  \item 对任意短正合列
  \(
  0\to A\to B\to C\to 0
  \)
  有
  \(
  I(B)=I(A)+I(C)
  \)；
  \item 归一化 \(I(\ZZ)=1\)。
\end{enumerate}
则对一切 \(G\in\mathcal A\) 都有
\[
I(G)=\cdim(G).
\]
特别地，任意有限群 \(F\) 自动满足 \(I(F)=0\)。
\end{theorem}
\begin{proof}
对任意 \(n\ge 2\)，短正合列
\[
0\to \ZZ\xrightarrow{\times n}\ZZ\to \ZZ/n\ZZ\to 0
\]
给出
\[
I(\ZZ/n\ZZ)=I(\ZZ)-I(\ZZ)=0.
\]
由有限群的循环分解与可加性得 \(I(F)=0\)（\(F\) 有限）。

再写
\[
G\simeq \ZZ^r\oplus T,\qquad T\ \text{有限},
\]
于是
\[
I(G)=I(\ZZ^r)+I(T)=rI(\ZZ)+0=r.
\]
而 \(r=\cdim(G)\)，故 \(I(G)=\cdim(G)\)。
证毕。
\end{proof}

\begin{definition}[有限秩扩张口径]\label{def:conclusion-cdim-finite-rank-extension}
对离散阿贝尔群 \(G\)，若
\[
\operatorname{rank}_{\QQ}(G):=\dim_{\QQ}(G\otimes_{\ZZ}\QQ)<\infty,
\]
则称 \(G\) 为有限秩群。
在本口径下定义
\[
\cdim_{\mathrm{fr}}(G):=\dim\bigl((\widehat G)^0\bigr).
\]
当 \(G\) 有限生成时，\(\cdim_{\mathrm{fr}}(G)=\cdim(G)\)。
\end{definition}

\begin{theorem}[圆维向有限秩无挠群的自然扩张]\label{thm:conclusion-cdim-finite-rank-extension}
设 \(G\) 为有限秩无挠离散阿贝尔群，令 \(r=\operatorname{rank}_{\QQ}(G)\)。
则存在自由子群 \(L\simeq \ZZ^r\) 使 \(G/L\) 为挠群，且在 Pontryagin 对偶下有短正合列
\[
0\longrightarrow \widehat{G/L}\longrightarrow \widehat G\longrightarrow \widehat L\simeq \TT^r\longrightarrow 0,
\]
其中 \(\widehat{G/L}\) 为紧全不连通群。
因而
\[
(\widehat G)^0\simeq \TT^r,\qquad \cdim_{\mathrm{fr}}(G)=r.
\]
特别地，
\[
\cdim_{\mathrm{fr}}(\QQ,+)=1,\qquad
\cdim_{\mathrm{fr}}(K,+)=[K:\QQ]\ \ (\text{任意数域 }K).
\]
\end{theorem}
\begin{proof}
取 \(G\otimes\QQ\) 的一组 \(\QQ\)-基 \(\{g_1\otimes 1,\dots,g_r\otimes 1\}\)，令
\[
L:=\langle g_1,\dots,g_r\rangle\cong \ZZ^r.
\]
任意 \(g\in G\) 在 \(G\otimes\QQ\) 中可写为
\(
g\otimes 1=\sum_i a_i(g_i\otimes 1)
\)
（\(a_i\in\QQ\)），清分母后存在 \(m\ge 1\) 使 \(mg\in L\)，故 \(G/L\) 为挠群。

对短正合列 \(0\to L\to G\to G/L\to 0\) 取 Pontryagin 对偶得
\[
0\to \widehat{G/L}\to \widehat G\to \widehat L\to 0.
\]
挠群的对偶为紧全不连通群，故 \(\widehat{G/L}\) 连通分量平凡。
因此 \((\widehat G)^0\to \widehat L\simeq \TT^r\) 为同构，得 \(\cdim_{\mathrm{fr}}(G)=r\)。

\(\QQ\) 的 \(\QQ\)-秩为 \(1\)，数域 \(K\) 的加法群 \(\QQ\)-秩为 \([K:\QQ]\)，代入即得。
证毕。
\end{proof}

\begin{theorem}[可逆外置的最小圆维账本下界]\label{thm:conclusion-cdim-kernel-register-lower-bound}
设 \(f:G\to H\) 为有限秩阿贝尔群同态，\(r:G\to R\) 为寄存器读出。
若联合映射
\[
(f,r):G\longrightarrow H\oplus R
\]
为单射，则
\[
\cdim_{\mathrm{fr}}(R)\ge \cdim_{\mathrm{fr}}(\ker f).
\]
并且该下界可达：取 \(R=\ker f\) 并令 \(r\) 在 \(\ker f\) 上为恒等嵌入，即可实现等号。
\leanverified{paper_conclusion_cdim_kernel_register_lower_bound}
\end{theorem}
\begin{proof}
\((f,r)\) 单射等价于 \(r|_{\ker f}\) 单射，故有群嵌入
\[
\ker f\hookrightarrow R.
\]
张量 \(\QQ\) 后得
\[
(\ker f)\otimes\QQ\hookrightarrow R\otimes\QQ,
\]
于是 \(\operatorname{rank}_{\QQ}(\ker f)\le \operatorname{rank}_{\QQ}(R)\)。
由定理 \ref{thm:conclusion-cdim-finite-rank-extension}，有限秩口径下
\[
\cdim_{\mathrm{fr}}(X)=\operatorname{rank}_{\QQ}(X),
\]
故结论成立。可达性由构造直接给出。证毕。
\end{proof}

\begin{theorem}[保持直和结构的自然数编码不可能性]\label{thm:conclusion-cdim-no-faithful-additive-natural-code}
记 \(\mathrm{Iso}(\mathrm{Ab}_{\mathrm{fg}})\) 为有限生成阿贝尔群同构类在直和 \(\oplus\) 下的交换幺半群。
则不存在单射交换幺半群同态
\[
\Phi:\bigl(\mathrm{Iso}(\mathrm{Ab}_{\mathrm{fg}}),\oplus\bigr)\longrightarrow (\ZZ_{\ge 0},+).
\]
\end{theorem}
\begin{proof}
反设存在此类 \(\Phi\)。则像
\[
M:=\Phi\bigl(\mathrm{Iso}(\mathrm{Ab}_{\mathrm{fg}})\bigr)\subseteq \ZZ_{\ge 0}
\]
为加法子幺半群。经典事实：\(\ZZ_{\ge 0}\) 的任意加法子幺半群均有限生成。
故 \(M\) 有限生成，进而 \(\mathrm{Iso}(\mathrm{Ab}_{\mathrm{fg}})\) 也有限生成（由 \(\Phi\) 单射）。

设生成元为 \(G_1,\dots,G_m\)，令
\[
P:=\bigcup_{i=1}^{m}\mathrm{Supp}\bigl(\mathrm{Tor}(G_i)\bigr)
\]
为其挠部分阶的素因子并集（有限集合）。
任取素数 \(p\notin P\)。由生成性，\(\ZZ/p\ZZ\) 的同构类应可写为若干 \(G_i\) 的直和。
但该直和的挠部分素因子支撑包含于 \(P\)，不可能等于 \(\{p\}\)，矛盾。
故 \(\Phi\) 不存在。证毕。
\end{proof}

\begin{theorem}[圆维作为 \texorpdfstring{$K_0$}{K0} 的唯一存活可加标尺]\label{thm:conclusion-cdim-k0-unique-survivor}
设 \(\mathrm{Ab}_{\mathrm{fg}}\) 为有限生成阿贝尔群的正合范畴。
则存在自然同构
\[
K_0(\mathrm{Ab}_{\mathrm{fg}})\cong \ZZ,
\qquad
[G]\longmapsto \cdim(G).
\]
等价地，任意短正合可加标尺均唯一经 \(\cdim\) 因子化，挠信息在 \(K_0\) 中严格消失。
\end{theorem}
\begin{proof}
短正合列
\[
0\to \ZZ\xrightarrow{\times n}\ZZ\to \ZZ/n\ZZ\to 0
\]
在 \(K_0\) 中给出
\(
[\ZZ/n\ZZ]=0
\)，故一切有限挠群类为 \(0\)。
对
\(
G\simeq \ZZ^r\oplus T
\)
有
\[
[G]=r[\ZZ]+[T]=r[\ZZ].
\]
因此 \(K_0\) 由 \([\ZZ]\) 生成，并由映射
\[
[G]\mapsto r=\cdim(G)
\]
识别为 \(\ZZ\)。
证毕。
\leanverified{paper\_conclusion\_cdim\_k0\_unique\_survivor}
\end{proof}

\begin{corollary}[圆维在有界导出范畴上的唯一归一化扩张]\label{cor:conclusion-cdim-derived-unique-extension}
对任意 \(C^\bullet\in D^b(\mathrm{Ab}_{\mathrm{fg}})\)，定义
\[
\chi_{\cdim}(C^\bullet):=\sum_i(-1)^i\,\cdim\!\bigl(H^i(C^\bullet)\bigr).
\]
则 \(\chi_{\cdim}\) 对三角正合列可加，且是满足
\(
\chi_{\cdim}(\ZZ[0])=1
\)
的唯一此类扩张。
\end{corollary}
\begin{proof}
由 \(K_0(D^b(\mathrm{Ab}_{\mathrm{fg}}))\cong K_0(\mathrm{Ab}_{\mathrm{fg}})\) 与定理 \ref{thm:conclusion-cdim-k0-unique-survivor}，一切三角可加标尺由 \([\ZZ[0]]\) 唯一决定。
\(\chi_{\cdim}\) 恰对应该生成元取值为 \(1\) 的同态，故唯一。证毕。
\end{proof}

\begin{theorem}[圆维压缩导致的幂律精度账本下界]\label{thm:conclusion-cdim-phase-compression-power-law}
\leanverified{paper\_conclusion\_cdim\_phase\_compression\_power\_law}
令
\[
\TT^d=(\RR/\ZZ)^d,\qquad
|x|_{\TT^d}:=\inf_{m\in\ZZ^d}\|x-m\|_\infty.
\]
设 \(\varphi:\ZZ^r\to \TT^d\) 为群同态，并记
\[
B_N:=\{0,1,\dots,N\}^r\subset\ZZ^r.
\]
则对任意 \(N\ge 1\)，存在不同 \(u,v\in B_N\) 使
\[
|\varphi(u)-\varphi(v)|_{\TT^d}\le 2(N+1)^{-r/d}.
\]
等价地，存在非零 \(k\in\ZZ^r\) 满足 \(\|k\|_\infty\le N\) 且
\[
|\varphi(k)|_{\TT^d}\le 2(N+1)^{-r/d}.
\]
\end{theorem}
\begin{proof}
令
\[
M:=\bigl\lfloor (N+1)^{r/d}\bigr\rfloor.
\]
把 \([0,1)^d\) 分成边长 \(1/M\) 的半开立方体，共 \(M^d\) 个。
考虑点集
\[
\{\varphi(x):x\in B_N\},
\qquad |B_N|=(N+1)^r.
\]
若其中有重复，则结论显然成立（距离为 \(0\)）。
否则它有 \((N+1)^r\) 个不同点。

若 \(M^d<(N+1)^r\)，鸽巢原理给出两个点落在同一立方体，故其 \(\TT^d\)-距离不超过 \(1/M\)。
若 \(M^d=(N+1)^r\)，改取 \(M-1\) 分割（\(M\ge 2\) 时），同理得到距离不超过 \(1/(M-1)\le 2/M\)；当 \(M=1\) 时结论亦平凡成立。
综合得
\[
|\varphi(u)-\varphi(v)|_{\TT^d}\le \frac{2}{M}
\le 2(N+1)^{-r/d}.
\]
置 \(k=u-v\neq 0\) 即得等价式。证毕。
\end{proof}

\begin{corollary}[相位压缩与可逆读出的最小比特代价]\label{cor:conclusion-cdim-phase-compression-bit-lower-bound}
\leanverified{paper\_conclusion\_cdim\_phase\_compression\_bit\_lower\_bound}
在定理 \ref{thm:conclusion-cdim-phase-compression-power-law} 设定下，
若要求在盒 \(B_N\) 上实现唯一反演，则相位测量分辨率 \(\delta_N\) 至少满足
\[
\delta_N\lesssim (N+1)^{-r/d},
\]
等价地所需精度比特数满足
\[
\mathrm{bits}(N)\ \ge\ \frac{r}{d}\log_2(N+1)-O(1).
\]
\end{corollary}
\begin{proof}
若 \(\delta_N\) 大于定理 \ref{thm:conclusion-cdim-phase-compression-power-law} 给出的最坏近碰撞尺度，则至少存在两点在编码端不可区分，唯一反演失败。
取 \(\delta_N=2^{-b_N}\) 即得
\(
b_N\ge (r/d)\log_2(N+1)-O(1)
\)。
证毕。
\end{proof}

\begin{conjecture}[多项式可审计算法下的圆维不可压缩性]\label{conj:conclusion-cdim-polynomial-decodability-rigidity}
设 \(\varphi:\ZZ^r\to\TT^d\) 为群同态。若存在算法 \(\mathsf{Dec}\) 满足：

d\ge r.

该命题刻画“集合论注入可存在”与“结构可审计可逆”之间的复杂度分离：前者不约束圆维，后者强制圆维下界。
\end{conjecture}

\paragraph{素数纤维并行度与圆维内化接口}
为把 residue/逆极限轴中的 prime-register 资源与 \(\cdim\)、\(\cdim^+\) 的圆维语言统一，引入“素数剖面”及其最小连通内化维数。

\begin{definition}[素数剖面与素数圆维]\label{def:conclusion-prime-profile-pcdim}
设 \(K\) 为拓扑有限生成（topologically finitely generated）的交换 profinite 群。对每个素数 \(p\) 定义
\[
d_p(K):=\dim_{\FF_p}(K/pK),
\]
并定义
\[
\operatorname{pcdim}(K):=\sup_{p}d_p(K)\in\NN\cup\{0\}.
\]
量 \(d_p(K)\) 表示 \(p\)-进方向上的并行独立流数，\(\operatorname{pcdim}(K)\) 表示所有素数方向中的最大并行度。
\end{definition}

\begin{theorem}[素数纤维的最小圆维内化原理]\label{thm:conclusion-prime-fiber-min-circle-internalization}
设 \(K\) 为拓扑有限生成的交换 profinite 群，令
\[
r:=\operatorname{pcdim}(K)<\infty.
\]
则存在紧连通交换群 \(\Sigma_K\) 使
\[
0\longrightarrow K\longrightarrow \Sigma_K\longrightarrow \TT^{\,r}\longrightarrow 0
\]
为短正合列。并且该 \(r\) 为最小值：若存在紧连通交换群 \(G\) 与正合列
\[
0\longrightarrow K\longrightarrow G\longrightarrow \TT^{\,m}\longrightarrow 0,
\]
则必有 \(m\ge r\)。
\end{theorem}
\begin{proof}
记 \(T:=\widehat K\)。由 Pontryagin 对偶，\(T\) 为离散挠群，且
\[
\bigl(K/pK\bigr)^\wedge\simeq T[p],\qquad
d_p(K)=\dim_{\FF_p}T[p].
\]
对每个 \(p\)，\(K\) 的 \(p\)-Sylow 分量拓扑有限生成，从而 \(T[p^\infty]\) 可写为有限个 Prüfer 群与有限 \(p\)-群之直和；其独立生成数不超过 \(d_p(K)\le r\)。因此存在嵌入
\[
T[p^\infty]\hookrightarrow \bigl(C_{p^\infty}\bigr)^r.
\]
对全部素数求直和得到
\[
\iota:T\hookrightarrow \bigoplus_{p}\bigl(C_{p^\infty}\bigr)^r
\simeq (\QQ/\ZZ)^r.
\]
令
\[
\pi:\QQ^r\twoheadrightarrow (\QQ/\ZZ)^r,\qquad
A:=\pi^{-1}\!\bigl(\iota(T)\bigr).
\]
则有正合列
\[
0\to \ZZ^r\to A\to T\to 0,
\]
且 \(A\subset \QQ^r\) 无挠。置 \(\Sigma_K:=\widehat A\)，则 \(\Sigma_K\) 紧且连通，并由对偶化得到
\[
0\to K\to \Sigma_K\to \TT^{\,r}\to 0.
\]

下证最小性。设
\[
0\to K\to G\to \TT^{\,m}\to 0
\]
正合，且 \(G\) 紧连通交换。对偶得
\[
0\to \ZZ^m\to B:=\widehat G\to T\to 0,
\]
其中 \(B\) 无挠。由于 \(B/\ZZ^m\) 挠，\(B\) 与 \(\ZZ^m\) 同秩，故可嵌入 \(\QQ^m\)，并有
\[
T\simeq B/\ZZ^m\hookrightarrow \QQ^m/\ZZ^m\simeq (\QQ/\ZZ)^m.
\]
于是
\[
T[p]\hookrightarrow \bigl((\QQ/\ZZ)^m\bigr)[p]\simeq (\ZZ/p\ZZ)^m,
\]
故 \(\dim_{\FF_p}T[p]\le m\)，即 \(d_p(K)\le m\) 对所有 \(p\) 成立，从而
\[
r=\sup_p d_p(K)\le m.
\]
证毕。
\end{proof}

\begin{corollary}[CRT 型 prime-register 的一圆维律]\label{cor:conclusion-crt-prime-register-one-circle}
若 prime-register 由单一 CRT 模数轴（等价地，pro-cyclic 逆极限轴）实现，记其群为 \(K\)。则
\[
d_p(K)\le 1\quad(\forall p),
\qquad
\operatorname{pcdim}(K)\le 1.
\]
当该轴非平凡且含至少一个 \(p\)-进无限分量时，\(\operatorname{pcdim}(K)=1\)。因此 \(K\) 可内化为一维相位扩张
\[
0\longrightarrow K\longrightarrow \Sigma_K\longrightarrow \TT\longrightarrow 0.
\]
\end{corollary}
\begin{proof}
pro-cyclic 交换 profinite 群在每个 \(p\)-分量上至多一生成，故 \(K/pK\) 的 \(\FF_p\)-维数不超过 \(1\)。若存在非平凡 \(p\)-进分量，则对应维数为 \(1\)，得 \(\operatorname{pcdim}(K)=1\)。最后应用定理 \ref{thm:conclusion-prime-fiber-min-circle-internalization} 即得。
\end{proof}

\begin{theorem}[紧李范畴中的无限 prime-register 障碍]\label{thm:conclusion-compact-lie-profinite-kernel-obstruction}
设 \(G\) 为紧李群，且存在正合列
\[
0\longrightarrow K\longrightarrow G\longrightarrow \TT^{\,r}\longrightarrow 0,
\]
其中 \(K\) 为 profinite 群。则 \(K\) 必有限。特别地，无限 profinite prime-register 不能作为紧李群到环面的核被内化。
\end{theorem}
\begin{proof}
\(K\) 为紧李群 \(G\) 的闭子群。由闭子群定理，\(K\) 仍为李群。又 \(K\) 为 profinite，故全不连通；作为李群只能是零维，从而离散。离散且紧的群必有限，结论成立。
\end{proof}

\begin{theorem}[连通环面扩张的 Pontryagin 对偶分类]\label{thm:conclusion-torus-extension-pontryagin-classification}
设 \(K\) 为拓扑有限生成的交换 profinite 群，\(r\ge 0\)。
下列数据按同构自然一一对应：
\begin{enumerate}
  \item 紧连通交换群短正合列
  \[
  0\to K\to G\to \TT^{\,r}\to 0;
  \]
  \item 离散无挠群 \(A\) 与嵌入 \(\ZZ^r\hookrightarrow A\)，满足
  \[
  A/\ZZ^r\simeq K^\wedge.
  \]
\end{enumerate}
对应由 \(G\simeq \widehat A\) 给出。
\end{theorem}
\begin{proof}
Pontryagin 对偶在局部紧交换群范畴上给出反变等价并保持正合。对 \((1)\) 对偶化即得
\[
0\to \ZZ^r\to \widehat G\to K^\wedge\to 0.
\]
因 \(G\) 连通，\(\widehat G\) 无挠，得 \((2)\)。
反向地，给定 \((2)\)，取对偶得到
\[
0\to K\to \widehat A\to \TT^r\to 0,
\]
且 \(\widehat A\) 连通（因 \(A\) 无挠）。故得到 \((1)\)。证毕。
\end{proof}

\begin{corollary}[圆维--素数剖面的资源边界完备性]\label{cor:conclusion-pcdim-resource-completeness}
对拓扑有限生成的交换 profinite 群 \(K\)，
\[
r\ge \sup_p d_p(K)
\]
是存在连通环面扩张
\[
0\to K\to G\to \TT^{\,r}\to 0
\]
的充要资源门槛；其中最小可行值即 \(\operatorname{pcdim}(K)\)。
\end{corollary}
\begin{proof}
充分性由定理 \ref{thm:conclusion-prime-fiber-min-circle-internalization} 的构造给出；必要性由同一定理的最小性给出。
\end{proof}

\begin{proposition}[素数并行的最大值法则与非加性]\label{prop:conclusion-pcdim-max-law}
设 \(K,L\) 为拓扑有限生成的交换 profinite 群。则对每个素数 \(p\),
\[
d_p(K\times L)=d_p(K)+d_p(L),
\]
并且
\[
\operatorname{pcdim}(K\times L)
=
\sup_p\bigl(d_p(K)+d_p(L)\bigr).
\]
特别地，若 \(K,L\) 的素数支撑互不相交（即每个 \(p\) 至少有一个维数为 \(0\)），则
\[
\operatorname{pcdim}(K\times L)=
\max\{\operatorname{pcdim}(K),\operatorname{pcdim}(L)\}.
\]
\leanverified{paper\_conclusion\_pcdim\_max\_law}
\end{proposition}
\begin{proof}
有同构
\[
\frac{K\times L}{p(K\times L)}
\simeq
\frac{K}{pK}\times \frac{L}{pL},
\]
故 \(\FF_p\)-维数可加。对 \(p\) 取上确界得第二式。若支撑互不相交，则每个 \(p\) 处和式退化为两者之一，故上确界即最大值。
\end{proof}

\begin{corollary}[无限素数族的一圆维内化]\label{cor:conclusion-infinite-primes-one-circle}
令 \(S\subseteq \mathcal P_{\mathrm{pr}}\) 为任意素数集合（可无限），并设
\[
K_S:=\prod_{p\in S}\ZZ_p.
\]
则
\[
d_p(K_S)=
\begin{cases}
1,&p\in S,\\
0,&p\notin S,
\end{cases}
\qquad
\operatorname{pcdim}(K_S)=1.
\]
并且存在紧连通交换群 \(\Sigma_{K_S}\) 使
\[
0\to K_S\to \Sigma_{K_S}\to \TT\to 0
\]
正合。若 \(S\) 无限，则 \(\Sigma_{K_S}\) 不能是紧李群。
\end{corollary}
\begin{proof}
前两式由
\(
K_S/pK_S\simeq \FF_p
\)
（\(p\in S\)）或 \(0\)（\(p\notin S\)）直接得到。设
\[
T_S:=K_S^\wedge=\bigoplus_{p\in S}C_{p^\infty}.
\]
利用分解 \(\QQ/\ZZ\simeq\bigoplus_p C_{p^\infty}\)，可得嵌入 \(T_S\hookrightarrow \QQ/\ZZ\)。令
\[
\pi:\QQ\twoheadrightarrow \QQ/\ZZ,\qquad
A_S:=\pi^{-1}(T_S)\subset\QQ.
\]
则 \(0\to \ZZ\to A_S\to T_S\to 0\) 正合，且 \(A_S\) 无挠。对偶化得
\[
0\to K_S\to \widehat{A_S}\to \TT\to 0.
\]
记 \(\Sigma_{K_S}:=\widehat{A_S}\) 即得存在性。若 \(S\) 无限，则 \(K_S\) 无限 profinite；由定理 \ref{thm:conclusion-compact-lie-profinite-kernel-obstruction}，\(\Sigma_{K_S}\) 不可能为紧李群。证毕。
\end{proof}
