\subsection{边界 Gödel 维的粘连缺陷相图与 Kolmogorov 粗糙度下界}\label{subsec:conclusion-boundary-godel-gluing-defect-kolmogorov-roughness}
在体--边界 Gödel 维夹逼、内部粘连恒等式以及边界 Gödel 编码的算法复杂度守恒都已建立之后，还可把边界增长的主导机制进一步压缩为一条精确相图：一部分来自外层厚度 \(2^{-m}N_m\)，另一部分来自内部粘连亏损；与此同时，若体相在固定分辨率下携带接近满量级的算法信息，则其边界面数必被迫进入粗糙区。

\leanverified{paper\_conclusion\_boundary\_godel\_adhesion\_defect\_phase\_diagram}
\begin{theorem}[边界 Gödel 维的粘连缺陷精确相图]\label{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram}
设 \(A\subset I^n\)，并沿用 dyadic 外近似 \(U_m(A)\) 的记号。记
\[
N_m:=N_m(A),
\qquad
E_m:=E\!\bigl(U_m(A)\bigr),
\qquad
F_m:=F\!\bigl(U_m(A)\bigr).
\]
定义粘连缺陷
\[
\Delta_m^{\mathrm{adh}}
:=
nN_m(1-2^{-m})-E_m
\ge 0,
\]
以及其指数
\[
\gamma_{\mathrm{adh}}(A)
:=
\limsup_{m\to\infty}
\frac{\log\!\bigl(\Delta_m^{\mathrm{adh}}+1\bigr)}{m\log 2}.
\]
则边界 Gödel 维满足精确分解公式
\[
d_{\partial G}(A)
=
\max\bigl\{d_G(A)-1,\ \gamma_{\mathrm{adh}}(A)\bigr\}.
\]
特别地，若 \(d_G(A)<n\)，则
\[
d_G(A)-1
<
\frac{n-1}{n}d_G(A)
\le
d_{\partial G}(A),
\]
从而
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A).
\]
换言之，在非满维饱和情形下，边界双对数增长的主导项并非来自外层厚度项 \(2^{-m}N_m\)，而完全由内部粘连亏损决定。
\end{theorem}
\begin{proof}
由命题 \ref{prop:spg-dyadic-polyclube-boundary-vs-adjacency-identity}，
\[
F_m=2nN_m-2E_m.
\]
再由定理 \ref{thm:spg-polyclube-adhesion-sharp-upper-bound}，
\[
E_m\le nN_m(1-2^{-m}),
\]
故 \(\Delta_m^{\mathrm{adh}}\ge 0\)，并且
\[
F_m
=
2nN_m-2E_m
=
2n\,2^{-m}N_m+2\Delta_m^{\mathrm{adh}}.
\]
记
\[
a_m:=2n\,2^{-m}N_m,
\qquad
b_m:=2\Delta_m^{\mathrm{adh}}.
\]
则 \(F_m=a_m+b_m\) 且 \(a_m,b_m\ge 0\)。由
\[
\max\{a_m,b_m\}\le F_m\le 2\max\{a_m,b_m\}
\]
可知
\[
\limsup_{m\to\infty}\frac{\log F_m}{m\log 2}
=
\max\left\{
\limsup_{m\to\infty}\frac{\log a_m}{m\log 2},
\limsup_{m\to\infty}\frac{\log(b_m+1)}{m\log 2}
\right\}.
\]

另一方面，命题 \ref{prop:spg-dyadic-boundary-godel-doublelog-scale} 与定理 \ref{thm:spg-stokes-godel-doublelog-dimension} 分别给出
\[
\log\log H_m\!\bigl(U_m(A)\bigr)=\log F_m+O(\log m),
\]
\[
\log\log G_m(A)=\log N_m(A)+O(\log m),
\qquad
d_G(A)=\limsup_{m\to\infty}\frac{\log N_m}{m\log 2}.
\]
故
\[
d_{\partial G}(A)=\limsup_{m\to\infty}\frac{\log F_m}{m\log 2}.
\]
又因为
\[
\log a_m=\log N_m-m\log 2+O(1),
\]
遂有
\[
\limsup_{m\to\infty}\frac{\log a_m}{m\log 2}=d_G(A)-1.
\]
而按定义，
\[
\limsup_{m\to\infty}\frac{\log(b_m+1)}{m\log 2}
=
\gamma_{\mathrm{adh}}(A).
\]
合并即得
\[
d_{\partial G}(A)=\max\bigl\{d_G(A)-1,\gamma_{\mathrm{adh}}(A)\bigr\}.
\]

最后，由定理 \ref{thm:spg-bulk-boundary-godel-dimension-sandwich}，
\[
\frac{n-1}{n}d_G(A)\le d_{\partial G}(A).
\]
若 \(d_G(A)<n\)，则
\[
d_G(A)-1<\frac{n-1}{n}d_G(A)\le d_{\partial G}(A),
\]
故上式中的极大值不可能由第一项取得，只能由 \(\gamma_{\mathrm{adh}}(A)\) 取得，于是
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A).
\]
证毕。
\end{proof}

\begin{theorem}[Kolmogorov 不可压缩性强迫粗糙边界]\label{thm:conclusion-boundary-kolmogorov-incompressibility-forces-roughness}
\leanverified{paper\_conclusion\_boundary\_kolmogorov\_incompressibility\_forces\_roughness}
对每个分辨率 \(m\)，固定推论 \ref{cor:spg-dyadic-topdim-holography-exactness} 中全部有向 \((n-1)\)-面的一组规范枚举，并以其前
\[
M_{n-1}(m):=n(2^m+1)\,2^{m(n-1)}
\]
个素数构造边界 Gödel 整数 \(H_m(U)\)。则存在常数 \(C_n,c_n>0\) 使得对任意 \(m\)-级 dyadic polycube
\[
U\subset I^n
\]
都成立
\[
K(U\mid m,n)\le C_n\,m\,F(U)+O(1),
\]
其中 \(F(U)\) 为边界有向面数，\(K(\cdot\mid\cdot)\) 为前缀型 Kolmogorov 复杂度。

因此，若某族 \((U_m)\) 满足：存在 \(\kappa>0\) 使对充分大 \(m\) 都有
\[
K(U_m\mid m,n)\ge \kappa\,2^{mn},
\]
则必有
\[
F(U_m)\ge c_n\,\frac{\kappa\,2^{mn}}{m}.
\]
特别地，任何满足
\[
F(U_m)=O\!\bigl(2^{m(n-1)}\bigr)
\]
的家族，都不可能承载 \(2^{mn}\)-量级的算法信息。
\end{theorem}
\begin{proof}
记 \(u_U\in C_n(\mathcal Q_m;\FF_2)\) 为 \(U\) 的 \(0/1\) 顶维占据链。由于 \(U\leftrightarrow u_U\) 是关于参数 \((m,n)\) 的可计算双向识别，故
\[
K(U\mid m,n)=K(u_U\mid m,n)+O(1).
\]
再由定理 \ref{thm:spg-stokes-godel-algorithmic-holographic-completeness}，存在常数 \(C_0\) 使
\[
\bigl|K(u_U\mid m,n)-K(H_m(U)\mid m,n)\bigr|\le C_0.
\]
于是
\[
K(U\mid m,n)\le K(H_m(U)\mid m,n)+O(1).
\]
再记
\[
\ell(N):=\lfloor \log_2 N\rfloor+1
\qquad(N\ge 1),
\]
则
\[
K(H_m(U)\mid m,n)\le \ell(H_m(U))+O(1).
\]
由于 \(H_m(U)\) 是 \(F(U)\) 个互异素数之积，且所用素数都不超过第 \(M_{n-1}(m)\) 个素数，故
\[
\ell(H_m(U))
\le
F(U)\log_2 p_{M_{n-1}(m)}+1.
\]
又由素数增长粗界 \(p_k\ll k\log k\) 与
\[
M_{n-1}(m)=n(2^m+1)2^{m(n-1)}\asymp_n 2^{mn},
\]
可得对充分大 \(m\) 有
\[
\log_2 p_{M_{n-1}(m)}\le C_n\,m.
\]
代回即得
\[
K(U\mid m,n)\le C_n\,m\,F(U)+O(1).
\]

若对某族 \((U_m)\) 有 \(K(U_m\mid m,n)\ge \kappa 2^{mn}\)，则整理上式得到
\[
F(U_m)\ge c_n\,\frac{\kappa\,2^{mn}}{m}
\]
对充分大 \(m\) 成立，其中 \(c_n>0\) 仅依赖于 \(n\)。

最后，若 \(F(U_m)=O(2^{m(n-1)})\)，则上式反推出
\[
K(U_m\mid m,n)=O\!\bigl(m\,2^{m(n-1)}\bigr)=o(2^{mn}),
\]
因而不可能具有 \(2^{mn}\)-量级的算法信息。证毕。
\end{proof}

\begin{theorem}[体相算法饱和强制边界 Gödel 维饱和]\label{thm:conclusion-boundary-kolmogorov-saturation-forces-godel-saturation}
对任意 \(m\)-级 dyadic polycube 序列 \((U_m)\subset I^n\)，定义边界 Gödel 维
\[
d_{\partial G}\bigl((U_m)\bigr)
:=
\limsup_{m\to\infty}
\frac{\log\log H_m(U_m)}{m\log 2}.
\]
若存在 \(\kappa>0\) 使对充分大 \(m\) 都有
\[
K(U_m\mid m,n)\ge \kappa\,2^{mn},
\]
则必有
\[
d_{\partial G}\bigl((U_m)\bigr)=n.
\]
换言之，一旦体相在固定分辨率上达到 \(2^{mn}\)-量级的算法饱和，其边界 Gödel 双对数斜率便被强制拉满到环境维数 \(n\)。
\leanverified{paper\_conclusion\_boundary\_kolmogorov\_saturation\_forces\_godel\_saturation}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-boundary-kolmogorov-incompressibility-forces-roughness}，存在常数 \(c_n>0\) 使
\[
F(U_m)\ge c_n\,\frac{\kappa\,2^{mn}}{m}
\]
对充分大 \(m\) 成立。

另一方面，命题 \ref{prop:spg-dyadic-boundary-godel-doublelog-scale} 给出
\[
\log\log H_m(U_m)=\log F(U_m)+O(\log m).
\]
故
\[
\log\log H_m(U_m)
\ge
mn\log 2-\log m+O(1),
\]
从而
\[
d_{\partial G}\bigl((U_m)\bigr)\ge n.
\]

另一方面，\(m\)-级 dyadic 立方复形的全部有向 \((n-1)\)-面总数至多为
\[
n(2^m+1)2^{m(n-1)}=O(2^{mn}),
\]
故同样由
\[
\log\log H_m(U_m)=\log F(U_m)+O(\log m)
\]
得到
\[
\log\log H_m(U_m)\le mn\log 2+O(\log m),
\]
从而
\[
d_{\partial G}\bigl((U_m)\bigr)\le n.
\]
两式合并即得
\[
d_{\partial G}\bigl((U_m)\bigr)=n.
\]
证毕。
\end{proof}

\begin{corollary}[亚最大边界 Gödel 维排斥算法饱和]\label{cor:conclusion-submaximal-boundary-godel-dimension-excludes-kolmogorov-saturation}
设 \(A\subset I^n\) 的 canonical dyadic 外逼近记为 \(U_m(A)\)，并设
\[
d_{\partial G}(A)=\beta<n.
\]
则对任意 \(\varepsilon>0\)，都有
\[
K(U_m(A)\mid m,n)=O_\varepsilon\!\bigl(m\,2^{(\beta+\varepsilon)m}\bigr).
\]
特别地，
\[
K(U_m(A)\mid m,n)=o(2^{mn}).
\]
\end{corollary}
\begin{proof}
由
\[
d_{\partial G}(A)
=
\limsup_{m\to\infty}
\frac{\log\log H_m(U_m(A))}{m\log 2}
=
\beta,
\]
对任意 \(\varepsilon>0\)，充分大 \(m\) 时有
\[
\log\log H_m(U_m(A))
\le
\left(\beta+\frac{\varepsilon}{2}\right)m\log 2.
\]
再由命题 \ref{prop:spg-dyadic-boundary-godel-doublelog-scale},
\[
\log F(U_m(A))
=
\log\log H_m(U_m(A))+O(\log m)
\le
\left(\beta+\frac{\varepsilon}{2}\right)m\log 2+O(\log m).
\]
对充分大 \(m\)，可把 \(O(\log m)\) 吸收到 \(\frac{\varepsilon}{2}m\log 2\) 中，故
\[
F(U_m(A))\le C_\varepsilon\,2^{(\beta+\varepsilon)m}.
\]
最后由定理 \ref{thm:conclusion-boundary-kolmogorov-incompressibility-forces-roughness},
\[
K(U_m(A)\mid m,n)\le C_n\,m\,F(U_m(A))+O(1)
=
O_\varepsilon\!\bigl(m\,2^{(\beta+\varepsilon)m}\bigr).
\]
由于 \(\beta+\varepsilon<n\) 可任意取，遂得
\[
K(U_m(A)\mid m,n)=o(2^{mn}).
\]
证毕。
\end{proof}

\begin{corollary}[非满维体相中的算法饱和只能由最大粘连缺陷承载]\label{cor:conclusion-boundary-kolmogorov-saturation-implies-maximal-adhesion}
设 \(A\subset I^n\) 为固定集合，\(U_m(A)\) 为其 dyadic 外近似。若
\[
d_G(A)<n
\]
且存在 \(\kappa>0\) 使
\[
K(U_m(A)\mid m,n)\ge \kappa\,2^{mn}
\]
对充分大 \(m\) 成立，则其粘连缺陷指数必达最大值：
\[
\gamma_{\mathrm{adh}}(A)=n.
\]
因此，在非满维体相中，若仍要承载 \(2^{mn}\)-量级的算法信息，则边界增长不可能由外层厚度主导，而只能完全压在内部粘连亏损之上。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-boundary-kolmogorov-saturation-forces-godel-saturation}，
\[
d_{\partial G}(A)=n.
\]
另一方面，由定理 \ref{thm:conclusion-boundary-godel-adhesion-defect-phase-diagram}，在 \(d_G(A)<n\) 时有
\[
d_{\partial G}(A)=\gamma_{\mathrm{adh}}(A).
\]
两式合并即得
\[
\gamma_{\mathrm{adh}}(A)=n.
\]
证毕。
\end{proof}

\begin{conjecture}[边界 Gödel 饱和与体相算法饱和的等价性]\label{conj:conclusion-boundary-godel-kolmogorov-saturation-equivalence}
对固定 \(A\subset I^n\) 及其 dyadic 外近似 \(U_m(A)\)，在
\[
d_G(A)<n
\]
的前提下，下列三条应当等价：
\begin{enumerate}
  \item
        \[
        K(U_m(A)\mid m,n)=\Theta(2^{mn});
        \]
  \item
        \[
        d_{\partial G}(A)=n;
        \]
  \item
        \[
        \gamma_{\mathrm{adh}}(A)=n.
        \]
\end{enumerate}
现有证据已给出
\[
(1)\Longrightarrow(2),
\qquad
(2)\Longleftrightarrow(3)\ \text{在 }d_G(A)<n\text{ 下成立}.
\]
尚待补足的方向只是
\[
d_{\partial G}(A)=n
\Longrightarrow
K(U_m(A)\mid m,n)=\Theta(2^{mn}),
\]
亦即把边界 Gödel 维的满饱和从“可恢复的 combinatorial 负载”提升为“不可压缩的算法负载”。
\end{conjecture}

\noindent
由此，边界双对数增长的主导机制可被精确拆成厚度项与粘连亏损项；在 \(d_G(A)<n\) 的主导区间中，前者严格让位于后者。与此同时，一旦体相本身携带接近满量级的 Kolmogorov 信息，其边界面数便必须至少达到 \(2^{mn}/m\) 量级，从而边界 Gödel 维被强制饱和到 \(n\)；而在非满维体相中，这种饱和又只能完全由粘连亏损承载。于是，低边界区与高复杂度区在 dyadic 体--界全息中不只形成定量互斥，而且在饱和极限下精确汇合为“算法满载 \(\Rightarrow\) 边界 Gödel 满载 \(\Rightarrow\) 粘连缺陷满载”的单向链。
