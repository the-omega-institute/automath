沿用定理 \ref{thm:xi-zg-zeta-factorization}、定理 \ref{thm:xi-zg-abel-residue-log-density}、推论 \ref{cor:xi-zg-absolute-convergence-critical-line} 与定理 \ref{thm:conclusion-zg-density-exact-inhomogeneous-markov} 的记号。对任意 \(\Re(\sigma)>1\)，记
\[
D_{\mathrm{ZG}}(\sigma):=\sum_{n\in \mathrm{ZG}}n^{-\sigma},
\qquad
D_d(\sigma):=\sum_{\substack{n\in \mathrm{ZG}\\ d\mid n}}n^{-\sigma},
\qquad
\mathcal R_q^{\mathrm{ZG}}(\sigma):=\sum_{n\in \mathrm{ZG}}c_q(n)n^{-\sigma}.
\]
再令 \((X_i^{(\sigma)})_{i\ge 1}\) 为相互独立的 Bernoulli 变量，满足
\[
\PP_\sigma(X_i^{(\sigma)}=1)=\frac{1}{p_i^\sigma+1},
\]
并记硬核事件
\[
E_\sigma:=\{X_i^{(\sigma)}X_{i+1}^{(\sigma)}=0\ \forall i\ge 1\}.
\]
若 \(d\) 为平方自由整数，记
\[
S(d):=\{i:\ p_i\mid d\},
\qquad
A_d:=\{X_i^{(\sigma)}=1\ \forall i\in S(d)\}.
\]

\begin{theorem}[强制整除 Dirichlet 级数与硬核柱事件质量的精确对应]\label{thm:conclusion-zg-divisibility-series-hardcore-cylinder-mass}
对任意整数 \(d\ge 1\) 与 \(\Re(\sigma)>1\)，都有
\[
D_d(\sigma)
=
\frac{\zeta(\sigma)}{\zeta(2\sigma)}\PP_\sigma(E_\sigma\cap A_d).
\]
若 \(d\) 含平方因子，或 \(S(d)\) 在 prime-index 链上不是独立集，则两边同为 \(0\)。
\end{theorem}
\begin{proof}
若 \(d\) 含平方因子，则 \(\mathrm{ZG}\) 只含平方自由整数，故 \(D_d(\sigma)=0\)。若 \(S(d)\) 不是独立集，则 \(d\) 含有某一对相邻素数因子，而 \(\mathrm{ZG}\) 的定义排除了这种情形，故左端同样为 \(0\)。另一方面，这两种情形下 \(E_\sigma\cap A_d=\varnothing\)，故右端也为 \(0\)。

下设 \(d\) 平方自由且 \(S(d)\) 独立。记
\[
a_i:=p_i^{-\sigma}
=
\frac{\PP_\sigma(X_i^{(\sigma)}=1)}{\PP_\sigma(X_i^{(\sigma)}=0)}.
\]
对任意有限独立集 \(T\subset \mathbb N\)，都有
\[
\PP_\sigma\bigl(X_i^{(\sigma)}=1\iff i\in T,\ X_i^{(\sigma)}=0\iff i\notin T\bigr)
=
\left(\prod_{i\ge 1}\PP_\sigma(X_i^{(\sigma)}=0)\right)\prod_{i\in T}a_i.
\]
于是
\[
\PP_\sigma(E_\sigma\cap A_d)
=
\left(\prod_{i\ge 1}\PP_\sigma(X_i^{(\sigma)}=0)\right)
\sum_{\substack{T\supseteq S(d)\\ T\text{ 有限独立}}}\prod_{i\in T}a_i.
\]
右侧求和恰与所有满足 \(n\in \mathrm{ZG}\) 且 \(d\mid n\) 的平方自由素因子集一一对应，因此
\[
\sum_{\substack{T\supseteq S(d)\\ T\text{ 有限独立}}}\prod_{i\in T}a_i=D_d(\sigma).
\]
再由
\[
\prod_{i\ge 1}\PP_\sigma(X_i^{(\sigma)}=0)^{-1}
=
\prod_{i\ge 1}(1+p_i^{-\sigma})
=
\frac{\zeta(\sigma)}{\zeta(2\sigma)},
\]
便得到所述恒等式。证毕。
\end{proof}

\begin{theorem}[Ramanujan 阴影与整除柱事件的 Möbius--divisor 对偶]\label{thm:conclusion-zg-ramanujan-shadow-divisor-cylinder-inversion}
对任意整数 \(q\ge 1\)，成立
\[
\mathcal R_q^{\mathrm{ZG}}(\sigma)
=
\sum_{d\mid q}\mu\!\left(\frac{q}{d}\right)d\,D_d(\sigma),
\]
并且对任意 \(d\ge 1\)，有精确逆变换
\[
D_d(\sigma)
=
\frac1d\sum_{q\mid d}\mathcal R_q^{\mathrm{ZG}}(\sigma).
\]
从而
\[
\mathcal R_q^{\mathrm{ZG}}(\sigma)
=
\frac{\zeta(\sigma)}{\zeta(2\sigma)}
\sum_{d\mid q}\mu\!\left(\frac{q}{d}\right)d\,\PP_\sigma(E_\sigma\cap A_d),
\]
并且一切非独立支撑的 \(d\)-项自动消失。
\end{theorem}
\begin{proof}
由经典恒等式
\[
c_q(n)=\sum_{d\mid(q,n)}\mu\!\left(\frac{q}{d}\right)d,
\]
得
\[
\mathcal R_q^{\mathrm{ZG}}(\sigma)
=
\sum_{n\in \mathrm{ZG}}n^{-\sigma}\sum_{d\mid(q,n)}\mu\!\left(\frac{q}{d}\right)d
=
\sum_{d\mid q}\mu\!\left(\frac{q}{d}\right)d
\sum_{\substack{n\in \mathrm{ZG}\\ d\mid n}}n^{-\sigma},
\]
即第一式。

另一方面，经典公式
\[
\sum_{q\mid d}c_q(n)=d\,\mathbf 1_{d\mid n}
\]
给出
\[
\frac1d\sum_{q\mid d}\mathcal R_q^{\mathrm{ZG}}(\sigma)
=
\sum_{n\in \mathrm{ZG}}n^{-\sigma}\frac1d\sum_{q\mid d}c_q(n)
=
\sum_{\substack{n\in \mathrm{ZG}\\ d\mid n}}n^{-\sigma}
=
D_d(\sigma).
\]
最后把定理 \ref{thm:conclusion-zg-divisibility-series-hardcore-cylinder-mass} 代入即可。证毕。
\end{proof}

\begin{theorem}[有限 primorial 截断上的 Ramanujan 阴影完全恢复柱事件]\label{thm:conclusion-zg-finite-primorial-shadow-cylinder-recovery}
设
\[
P_N:=\prod_{i=1}^{N}p_i.
\]
则有限阴影族
\[
\{\mathcal R_q^{\mathrm{ZG}}(\sigma):\ q\mid P_N\}
\]
唯一决定前 \(N\) 个 prime-index 位上的全部 admissible 柱事件概率。更精确地，对任意满足 \(\eta_i\eta_{i+1}=0\) 的模式
\[
\eta=(\eta_1,\dots,\eta_N)\in\{0,1\}^N,
\]
记
\[
S_1(\eta):=\{i\le N:\eta_i=1\},
\qquad
S_0(\eta):=\{i\le N:\eta_i=0\},
\qquad
d_S:=\prod_{i\in S}p_i.
\]
则有
\[
\PP_\sigma\!\left(E_\sigma\cap\{X_1=\eta_1,\dots,X_N=\eta_N\}\right)
=
\sum_{T\subseteq S_0(\eta)}(-1)^{|T|}
\frac{\zeta(2\sigma)}{\zeta(\sigma)}
\frac{1}{d_{S_1(\eta)\cup T}}
\sum_{q\mid d_{S_1(\eta)\cup T}}\mathcal R_q^{\mathrm{ZG}}(\sigma),
\]
其中一切使 \(S_1(\eta)\cup T\) 失去独立性的项按 \(0\) 计。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zg-ramanujan-shadow-divisor-cylinder-inversion}，
\[
\PP_\sigma(E_\sigma\cap A_{d_S})
=
\frac{\zeta(2\sigma)}{\zeta(\sigma)}
\frac{1}{d_S}
\sum_{q\mid d_S}\mathcal R_q^{\mathrm{ZG}}(\sigma)
\]
对每个独立集 \(S\subseteq\{1,\dots,N\}\) 都成立。

现在固定 admissible 模式 \(\eta\)。其对应柱事件可写成
\[
E_\sigma\cap\{X_1=\eta_1,\dots,X_N=\eta_N\}
=
\bigl(E_\sigma\cap A_{d_{S_1(\eta)}}\bigr)\cap\bigcap_{j\in S_0(\eta)}\{X_j=0\}.
\]
对零位约束应用 inclusion--exclusion，
\[
\mathbf 1_{\cap_{j\in S_0(\eta)}\{X_j=0\}}
=
\sum_{T\subseteq S_0(\eta)}(-1)^{|T|}\mathbf 1_{A_{d_T}},
\]
再与 \(\mathbf 1_{E_\sigma\cap A_{d_{S_1(\eta)}}}\) 相乘并取期望，便得到所述公式。证毕。
\end{proof}

\begin{theorem}[prime shadow 与 pair shadow 的精确相关律]\label{thm:conclusion-zg-prime-pair-shadow-correlation-law}
定义条件占据量
\[
\pi_i(\sigma):=\PP_\sigma(X_i^{(\sigma)}=1\mid E_\sigma),
\qquad
\pi_{ij}(\sigma):=\PP_\sigma(X_i^{(\sigma)}=X_j^{(\sigma)}=1\mid E_\sigma).
\]
则对任意 \(i<j\)，有
\[
\frac{\mathcal R_{p_i}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
=
p_i\pi_i(\sigma)-1,
\]
\[
\frac{\mathcal R_{p_ip_j}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
=
1-p_i\pi_i(\sigma)-p_j\pi_j(\sigma)+p_ip_j\pi_{ij}(\sigma),
\]
并且
\[
\frac{\mathcal R_{p_ip_j}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
-
\frac{\mathcal R_{p_i}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
\frac{\mathcal R_{p_j}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
=
p_ip_j\bigl(\pi_{ij}(\sigma)-\pi_i(\sigma)\pi_j(\sigma)\bigr).
\]
特别地，当 \(j=i+1\) 时，
\[
\pi_{i,i+1}(\sigma)=0
\]
且
\[
\frac{\mathcal R_{p_ip_{i+1}}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
-
\frac{\mathcal R_{p_i}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
\frac{\mathcal R_{p_{i+1}}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
=
-p_ip_{i+1}\pi_i(\sigma)\pi_{i+1}(\sigma)<0.
\]
\end{theorem}
\begin{proof}
把定理 \ref{thm:conclusion-zg-divisibility-series-hardcore-cylinder-mass} 应用于 \(d=p_i\) 与 \(d=p_ip_j\)，得到
\[
D_{p_i}(\sigma)=D_{\mathrm{ZG}}(\sigma)\pi_i(\sigma),
\qquad
D_{p_ip_j}(\sigma)=D_{\mathrm{ZG}}(\sigma)\pi_{ij}(\sigma).
\]
另一方面，由定理 \ref{thm:conclusion-zg-ramanujan-shadow-divisor-cylinder-inversion}，
\[
\mathcal R_{p_i}^{\mathrm{ZG}}(\sigma)=-D_{\mathrm{ZG}}(\sigma)+p_iD_{p_i}(\sigma),
\]
\[
\mathcal R_{p_ip_j}^{\mathrm{ZG}}(\sigma)
=
D_{\mathrm{ZG}}(\sigma)-p_iD_{p_i}(\sigma)-p_jD_{p_j}(\sigma)+p_ip_jD_{p_ip_j}(\sigma).
\]
代入即得前两式；再作差便得到相关律。

当 \(j=i+1\) 时，硬核事件 \(E_\sigma\) 强制 \(X_i^{(\sigma)}X_{i+1}^{(\sigma)}=0\)，故 \(\pi_{i,i+1}(\sigma)=0\)。同时，事件“仅第 \(i\) 位取 \(1\)”与“仅第 \(i+1\) 位取 \(1\)”都具有正概率，故 \(\pi_i(\sigma),\pi_{i+1}(\sigma)>0\)。于是最后一式严格为负。证毕。
\end{proof}

\begin{proposition}[固定导子 Ramanujan 阴影的极点阶刚性]\label{prop:conclusion-zg-fixed-conductor-shadow-pole-rigidity}
对每个固定整数 \(q\ge 1\)，\(\mathcal R_q^{\mathrm{ZG}}(\sigma)\) 在 \(\Re(\sigma)>1\) 上绝对收敛，并满足
\[
\bigl|\mathcal R_q^{\mathrm{ZG}}(\sigma)\bigr|
\le
\sum_{d\mid q}d\,D_d(\sigma)
\le
\sigma_1(q)\,D_{\mathrm{ZG}}(\sigma).
\]
因此，当 \(\sigma\downarrow 1\) 时，每个固定导子阴影至多具有与 \(D_{\mathrm{ZG}}(\sigma)\) 同阶的简单极点；任何高阶奇性都被排除。
\end{proposition}
\begin{proof}
由定理 \ref{thm:conclusion-zg-ramanujan-shadow-divisor-cylinder-inversion}，
\[
\bigl|\mathcal R_q^{\mathrm{ZG}}(\sigma)\bigr|
\le
\sum_{d\mid q}d\,D_d(\sigma).
\]
再由定理 \ref{thm:conclusion-zg-divisibility-series-hardcore-cylinder-mass} 的非负性，
\[
0\le D_d(\sigma)\le D_{\mathrm{ZG}}(\sigma),
\]
便得到所示估计。最后，由定理 \ref{thm:xi-zg-abel-residue-log-density}，\(D_{\mathrm{ZG}}(\sigma)\) 在 \(\sigma=1\) 处仅有简单极点，故同阶上界立即推出结论。证毕。
\end{proof}

\begin{theorem}[归一化 Ramanujan 阴影族的最近邻不可乘性]\label{thm:conclusion-zg-shadow-nonmultiplicativity-nearest-neighbor}
不存在一个在平方自由导子上逐素数可乘的归一化阴影族
\[
\mathfrak a_q(\sigma)
\]
满足
\[
\mathfrak a_q(\sigma)=\frac{\mathcal R_q^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
\qquad
\text{对所有平方自由 }q,
\]
并且
\[
\mathfrak a_{q_1q_2}(\sigma)=\mathfrak a_{q_1}(\sigma)\mathfrak a_{q_2}(\sigma)
\qquad
((q_1,q_2)=1).
\]
更具体地，这一乘法性的失败恰由最近邻 conditional covariance
\[
\pi_{ij}(\sigma)-\pi_i(\sigma)\pi_j(\sigma)
\]
控制；对于 \(j=i+1\)，该缺陷严格为负。
\end{theorem}
\begin{proof}
若存在这样的乘法归一化，则对任意 \(i<j\) 都必须有
\[
\frac{\mathcal R_{p_ip_j}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
=
\frac{\mathcal R_{p_i}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}
\frac{\mathcal R_{p_j}^{\mathrm{ZG}}(\sigma)}{D_{\mathrm{ZG}}(\sigma)}.
\]
由定理 \ref{thm:conclusion-zg-prime-pair-shadow-correlation-law}，这等价于
\[
\pi_{ij}(\sigma)=\pi_i(\sigma)\pi_j(\sigma)
\qquad
\forall\, i<j.
\]
取 \(j=i+1\)，左端因硬核约束恒等于 \(0\)，而右端严格为正，矛盾。故所述乘法性不可能成立。最后一句正是同一定理给出的显式相关差。证毕。
\end{proof}

\begin{conjecture}[Ramanujan 阴影对非齐次 Markov 包与局部维数包的完备性]\label{conj:conclusion-zg-ramanujan-shadow-markov-dimension-completeness}

\[
\{\mathcal R_q^{\mathrm{ZG}}(\sigma)\}_{q\ge 1}
\]
应当唯一决定定理 \ref{thm:conclusion-zg-density-exact-inhomogeneous-markov} 所刻画的非齐次 Markov 密度测度；进一步，经由其全部有限柱事件概率，定理 \ref{thm:conclusion-zg-holographic-pushforward-entropy-information-dimension} 及相关局部维数结论中的信息维数与有限证书谱也应当被唯一确定。
\end{conjecture}
\begin{proof}[论据]
定理 \ref{thm:conclusion-zg-finite-primorial-shadow-cylinder-recovery} 已经表明：对每个有限 \(N\)，有限阴影族
\[
\{\mathcal R_q^{\mathrm{ZG}}(\sigma):q\mid P_N\}
\]
唯一决定前 \(N\) 个 prime-index 位上的全部 admissible 柱事件概率。由 Kolmogorov 一致性，这一族有限维分布应当唯一恢复整条非齐次硬核链。另一方面，本小节其余定理已经把该链条组织为精确的非齐次一阶 Markov 结构、块熵增长律与全息推前的零信息维数退化。因而全体 Ramanujan 阴影理应构成这一密度几何包的完备坐标。当前尚缺的是把有限截断的一致恢复与全息局部维数读出拼合成严格的唯一性定理。证毕。
\end{proof}
