\begin{theorem}[Latt\`es 逆像树的挠子主齐次空间结构与仿射伽罗瓦表示]\label{thm:conclusion-leyang-lattes-preimage-torsor-affine-galois}
沿用本小节固定的极小椭圆模型
\[
E_{\min}:\ u^2+u=x^3-x,
\]
并沿用结论 \ref{con:xi-terminal-zm-elliptic-lattes-doubling} 中由倍点映射诱导的四次 Latt\`es 自映射
\[
\Phi(X)=\frac{X^4+2X^2-2X+1}{4X^3-4X+1}\in\QQ(X).
\]
设 \(K\) 为数域，取 \(t\in\PP^1(K)\) 满足 \(t\neq \infty\) 且 \(t\notin \mathrm{PostCrit}(\Phi)\)。
对 \(n\ge 1\) 记
\[
V_n(t):=\Phi^{-n}(t)\subset \PP^1(\overline K),
\qquad
E_{\min}[2^n]\cong (\ZZ/2^n\ZZ)^2.
\]
则：
\begin{enumerate}
  \item 对每个 \(n\ge 1\)，集合 \(V_n(t)\) 上存在一个自然的、随 \(n\) 相容的 \(E_{\min}[2^n]\)-自由且传递作用，使 \(V_n(t)\) 成为 \(E_{\min}[2^n]\) 的主齐次空间。
  \item 逆极限 \(\varprojlim_n V_n(t)\) 为 \(T_2(E_{\min})=\varprojlim_n E_{\min}[2^n]\) 的主齐次空间。
  \item 令 \(G_K:=\Gal(\overline K/K)\)。
  将 \(V_n(t)\) 视为 \(G_K\) 的 \(n\)-层作用集合族，则存在连续仿射型 arboreal 表示
  \[
  \rho_t:\ G_K\longrightarrow \mathrm{Aff}\bigl(T_2(E_{\min})\bigr)\cong T_2(E_{\min})\rtimes \GL_2(\ZZ_2),
  \]
  其线性部分等于 \(2\)-进 Tate 模表示 \(\rho_{E_{\min},2}:G_K\to \GL_2(\ZZ_2)\)，而平移部分由一条连续 \(1\)-上同调类 \(c_t\in Z^1(G_K,T_2(E_{\min}))\) 给出。
\end{enumerate}
\end{theorem}
\begin{proof}[证明要点]
取 \(P\in E_{\min}(\overline K)\) 使 \(x(P)=t\)。
由于 \(t\notin \mathrm{PostCrit}(\Phi)\)，可取 \(P\) 不为 \(2\)-挠点，从而 \(P\neq -P\) 且 \(x^{-1}(t)=\{P,-P\}\)。
对每个 \(n\) 取 \(Q_n\in E_{\min}(\overline K)\) 满足 \([2^n]Q_n=P\)。
定义
\[
\alpha_n:E_{\min}[2^n]\to V_n(t),\qquad T\longmapsto x(Q_n+T).
\]
若 \(X\in V_n(t)\)，取 \(R\in E_{\min}(\overline K)\) 使 \(x(R)=X\)。
由 \(\Phi^n(X)=t\iff x([2^n]R)=x(P)\) 得 \([2^n]R=\pm P\)；
若 \([2^n]R=-P\)，则 \([2^n](-R)=P\) 且 \(x(-R)=x(R)\)，故可改取 \(-R\)。
于是 \([2^n](R-Q_n)=0\)，即 \(R-Q_n\in E_{\min}[2^n]\)，从而 \(\alpha_n\) 满射。
若 \(\alpha_n(T)=\alpha_n(T')\)，则 \(Q_n+T'=\pm(Q_n+T)\)；
对两边施加 \([2^n]\) 得 \(P=\pm P\)，由 \(P\neq -P\) 知只能取正号，故 \(T'=T\)。
因此 \(\alpha_n\) 为双射，并把 \(E_{\min}[2^n]\) 的平移作用转移到 \(V_n(t)\) 上，给出主齐次空间结构。
\par
逆极限与相容性由 \([2]:E_{\min}[2^{n+1}]\to E_{\min}[2^n]\) 与 \(\Phi:V_{n+1}(t)\to V_n(t)\) 在上述双射下的交换性推出。
对 \(g\in G_K\) 定义
\(
c_{t,n}(g):=g(Q_n)-Q_n\in E_{\min}[2^n]
\)，则
\(
c_{t,n}(gh)=c_{t,n}(g)+\rho_{E_{\min},2^n}(g)c_{t,n}(h)
\)。
令 \(n\to\infty\) 得 \(c_t\in Z^1(G_K,T_2(E_{\min}))\)，并得到 \(\rho_t\) 的仿射分解。证毕。
\end{proof}

\begin{theorem}[倍点自然扩张的溶子结构、圆维分解与测度级一致性]\label{thm:conclusion-leyang-doubling-solenoid-extension}
\leanverified{paper\_conclusion\_leyang\_doubling\_solenoid\_extension}
视 \(E_{\min}(\CC)\) 为复环面 \(\CC/\Lambda\)，并考虑倍点映射 \([2]:E_{\min}(\CC)\to E_{\min}(\CC)\)。
定义逆极限紧群
\[
\Sigma_{E_{\min}}:=\varprojlim\Bigl(E_{\min}(\CC)\xleftarrow{[2]}E_{\min}(\CC)\xleftarrow{[2]}\cdots\Bigr)
=\{(P_0,P_1,\dots):[2]P_{n+1}=P_n\}.
\]
则：
\begin{enumerate}
  \item \(\Sigma_{E_{\min}}\) 为紧致阿贝尔群，并存在短正合列
  \[
  0\to T_2(E_{\min})\longrightarrow \Sigma_{E_{\min}}\xrightarrow{\ \pi_0\ }E_{\min}(\CC)\to 0,
  \]
  其中 \(\pi_0(P_0,P_1,\dots)=P_0\)。
  \item 左移同胚
  \(
  \sigma(P_0,P_1,\dots)=(P_1,P_2,\dots)
  \)
  的拓扑熵满足 \(h_{\mathrm{top}}(\sigma)=\log\deg([2])=\log 4\)。
  \item 记 \(\mu_{\Sigma}\) 为 \(\Sigma_{E_{\min}}\) 的 Haar 概率测度，则 \(\mu_{\Sigma}\) 在 \(\sigma\) 下不变。
  并且其在核纤维 \(T_2(E_{\min})=\ker(\pi_0)\) 的各有限商 \(E_{\min}[2^n]\) 上的推前均为均匀分布。
\end{enumerate}
\end{theorem}
\begin{proof}[证明要点]
逆极限在紧群与满射连续同态下保持紧致群结构。
\(\pi_0\) 为满射群同态，核由 \(P_0=O\) 的兼容序列组成，等价于兼容的 \(2^n\)-挠点系，故核同构于 \(T_2(E_{\min})\)，得短正合列。
\par
\(\sigma\) 为群自同胚。
对覆盖映射 \(f\) 的自然扩张，其拓扑熵与 \(f\) 相同，而 \([2]\) 的度为 \(4\)，故 \(h_{\mathrm{top}}(\sigma)=\log 4\)。
\par
紧群上的 Haar 测度在群自同胚下不变。
核纤维上的 Haar 测度在任意满射群同态下推前为 Haar 测度；有限群上的 Haar 测度即均匀分布，从而得到有限层一致性。证毕。
\end{proof}

\begin{proposition}\label{prop:conclusion-leyang-solenoid-stokes-period-module-halflocalized}
\leanverified{paper\_conclusion\_leyang\_solenoid\_stokes\_period\_module\_halflocalized}
在定理 \ref{thm:conclusion-leyang-doubling-solenoid-extension} 的设定下，
取 \(E_{\min}(\CC)\cong \CC/\Lambda\) 上的平移不变全纯 \(1\)-形式 \(\omega\)（例如 N\'eron 微分），并记周期格
\[
\Lambda_\omega:=\left\{\int_{\alpha}\omega:\ \alpha\in H_1(E_{\min}(\CC),\ZZ)\right\}\subset\CC.
\]
令 \(\pi_k:\Sigma_{E_{\min}}\to E_{\min}(\CC)\) 为第 \(k\) 层投影。
定义归一化 Stokes 周期模块
\[
\Pi_{\mathrm{St}}(\Sigma_{E_{\min}};\omega)
:=\bigcup_{k\ge 0}2^{-k}
\left\{\int_{\pi_k\circ\gamma}\omega:\ \gamma:[0,1]\to\Sigma_{E_{\min}},\ \gamma(0)=O,\ \pi_k\circ\gamma\ \text{为回路}\right\}\subset\CC.
\]
则有恒等式
\[
\Pi_{\mathrm{St}}(\Sigma_{E_{\min}};\omega)=\Lambda_\omega[1/2]:=\bigcup_{k\ge 0}2^{-k}\Lambda_\omega.
\]
并且在选定 \(\Lambda_\omega\) 的基后，存在自然同构
\[
\Lambda_\omega[1/2]/\Lambda_\omega\ \cong\ (\QQ_2/\ZZ_2)^2,
\qquad
\widehat{T_2(E_{\min})}\ \cong\ \Lambda_\omega[1/2]/\Lambda_\omega .
\]
\end{proposition}
\begin{proof}
固定 \(k\)。
任取 \(E_{\min}(\CC)\) 中基于 \(O\) 的回路 \(\alpha\)，则 \(\int_\alpha\omega\in\Lambda_\omega\)。
由 \([2]\) 为覆盖映射，\(\alpha\) 可自第 \(k\) 层逐层提升为一族兼容路径，从而存在 \(\gamma\) 使 \(\pi_k\circ\gamma=\alpha\)。
因此 \(2^{-k}\int_{\pi_k\circ\gamma}\omega\in 2^{-k}\Lambda_\omega\)，从而 \(\Lambda_\omega[1/2]\subseteq \Pi_{\mathrm{St}}(\Sigma_{E_{\min}};\omega)\)。
\par
反向包含：任取 \(p\in \Pi_{\mathrm{St}}(\Sigma_{E_{\min}};\omega)\)，则存在 \(k\) 与回路 \(\alpha=\pi_k\circ\gamma\) 使 \(p=2^{-k}\int_{\alpha}\omega\)。
而 \(\int_{\alpha}\omega\in\Lambda_\omega\)，故 \(p\in 2^{-k}\Lambda_\omega\subseteq \Lambda_\omega[1/2]\)，得等式成立。
\par
最后，由 \(\Lambda_\omega\cong \ZZ^2\) 得 \(\Lambda_\omega[1/2]/\Lambda_\omega\cong (\ZZ[1/2]/\ZZ)^2\cong (\QQ_2/\ZZ_2)^2\)。
并且 \(T_2(E_{\min})\cong \ZZ_2^2\)，其 Pontryagin 对偶同构于 \((\QQ_2/\ZZ_2)^2\)，与上述商群对应。证毕。
\end{proof}

\begin{proposition}\label{prop:conclusion-leyang-lattes-hypercube-exact-form-zero-curvature}
\leanverified{paper\_conclusion\_leyang\_lattes\_hypercube\_exact\_form\_zero\_curvature}
在定理 \ref{thm:conclusion-leyang-lattes-preimage-torsor-affine-galois} 的设定下，
固定 \(k\ge 1\) 并取同构 \(E_{\min}[2^k]\cong (\ZZ/2^k\ZZ)^2\)。
在复统一化 \(E_{\min}(\CC)\cong \CC/\Lambda\) 下，选取 \(\Lambda_\omega\) 的一组基 \((\lambda_1,\lambda_2)\)。
将 \((\ZZ/2^k\ZZ)^2\) 按二进展开识别为 \(\{0,1\}^{2k}\)：
对 \(\varepsilon=(\varepsilon_{1,1},\dots,\varepsilon_{1,k},\varepsilon_{2,1},\dots,\varepsilon_{2,k})\) 置
\[
a(\varepsilon):=\sum_{j=1}^k \varepsilon_{1,j}2^{k-j}\in\ZZ/2^k\ZZ,\qquad
b(\varepsilon):=\sum_{j=1}^k \varepsilon_{2,j}2^{k-j}\in\ZZ/2^k\ZZ,
\]
并令其对应点为 \(a(\varepsilon)P_k+b(\varepsilon)Q_k\in E_{\min}[2^k]\)。
定义势函数（取值于 \(\Lambda_\omega[1/2]/\Lambda_\omega\)）
\[
\Phi_k(\varepsilon)
:=\sum_{j=1}^k \varepsilon_{1,j}\frac{\lambda_1}{2^j}+\sum_{j=1}^k \varepsilon_{2,j}\frac{\lambda_2}{2^j}\quad(\mathrm{mod}\ \Lambda_\omega).
\]
则离散 \(1\)-形式 \(d\Phi_k\) 在每个坐标方向上为常值增量：沿翻转 \(\varepsilon_{1,j}\) 的边增量恒为 \(\lambda_1/2^j\)，沿翻转 \(\varepsilon_{2,j}\) 的边增量恒为 \(\lambda_2/2^j\)（均模 \(\Lambda_\omega\)）。
并且对任意二维方形面 \(\square\subset \{0,1\}^{2k}\)，有
\[
\sum_{e\in \partial\square}(d\Phi_k)(e)=0,
\]
即该离散 \(1\)-形式为精确形式且离散曲率为零。
\end{proposition}
\begin{proof}
势函数按坐标分解为各位的线性叠加，因此沿翻转单一坐标位的边增量等于对应项系数，得到常值增量结论。
对任意方形面，其边界由两次可交换的翻转组成；由于 \(\Phi_k\) 为势函数，边界增量望远镜消去，故 \(\partial\square\) 上增量和为零。证毕。
\end{proof}

\begin{corollary}[Latt\`es 预像超立方体上的最优缺陷控制：方形环和支配路径依赖]\label{cor:conclusion-leyang-lattes-hypercube-optimal-defect-control}
沿用定理 \ref{thm:conclusion-leyang-lattes-preimage-torsor-affine-galois} 与命题 \ref{prop:conclusion-leyang-lattes-hypercube-exact-form-zero-curvature} 的识别。

对任意反对称边余链 \(\omega\in C^1(I^{2k};\RR)\) 记其方形缺陷
\[
\varepsilon:=\|d\omega\|_\infty,
\]
其中 \(d\omega\in C^2(I^{2k};\RR)\) 在每个方形面上给出有向边界环和。
则存在势函数 \(f\in C^0(I^{2k};\RR)\) 与残差 \(r\in C^1(I^{2k};\RR)\) 使
\[
\omega=df+r,\qquad dr=d\omega,\qquad \|r\|_\infty\le \frac14\,\varepsilon,
\]
并且常数 \(\frac14\) 为最优。
从而对任意路径 \(\gamma\)（在 \(\Omega_{2k}\) 的边图上）有
\[
\Bigl|\sum_{e\in\gamma}\omega(e)-\bigl(f(\gamma_{\mathrm{end}})-f(\gamma_{\mathrm{start}})\bigr)\Bigr|
\le |\gamma|\,\|r\|_\infty
\le \frac{|\gamma|}{4}\,\varepsilon.
\]
特别地，对任意两点间最短路 \(|\gamma|\le \mathrm{diam}(\Omega_{2k})=2k\)，得到全局路径依赖上界
\[
\sup_{\gamma\ \mathrm{shortest}}
\Bigl|\sum_{e\in\gamma}\omega(e)-\Delta f\Bigr|
\le \frac{k}{2}\,\varepsilon.
\]
\end{corollary}

\begin{proof}
该结论是推论 \ref{cor:spg-hypercube-gradient-consistency-by-square-defect} 在维数 \(2k\) 的直接应用；其将任意边余链分解为势差项与残差项，并给出最优常数 \(\frac14\)。
对路径估计由
\(
\sum_{e\in\gamma}\omega(e)=\Delta f+\sum_{e\in\gamma}r(e)
\)
与三角不等式直接推出。证毕。
\end{proof}

\[
\begin{CD}
E_{\min}[2^{n+1}] @>{\mathcal G_{n+1}}>> H_{n+1}\\

\end{proof}

\begin{theorem}[二进分辨率与 Pontryagin--Fourier 频率截断的严格同一]\label{thm:conclusion-leyang-t2-fourier-truncation-equals-precision}
\leanverified{paper\_conclusion\_leyang\_t2\_fourier\_truncation\_equals\_precision}

f_n:=\EE[f\mid \mathcal F_n].

\[
\boxed{\;
\|f-f_n\|_{L^2(G,\mu)}^2
=\sum_{\substack{\chi\in \widehat G\\ \mathrm{ord}(\chi)>2^n}}|\widehat f(\chi)|^2,\;}
\qquad
\widehat f(\chi):=\int_G f(x)\overline{\chi(x)}\,\dd\mu(x),

\end{theorem}

\end{proof}
