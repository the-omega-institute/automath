\begin{theorem}[算术自然密度测度的精确一阶非齐次 Markov 结构]\label{thm:conclusion-zg-density-exact-inhomogeneous-markov}
记
\[
\Sigma_{\mathrm{ZG}}
:=
\{(z_k)_{k\ge 1}\in\{0,1\}^{\NN}: z_kz_{k+1}=0\ \forall k\},
\]
并令 \(\nu_{\mathrm{ZG}}\) 为定理 \ref{thm:xi-time-part60f-zg-cylinder-measure-local-dimension-zero} 所定义的自然密度诱导概率测度。再记
\[
M_k^{(0)}
:=
\begin{pmatrix}
\dfrac{p_k}{p_k+1} & \dfrac{p_k}{p_k+1}\\[2mm]
0&0
\end{pmatrix},
\qquad
M_k^{(1)}
:=
\begin{pmatrix}
0&0\\[2mm]
\dfrac{1}{p_k+1}&0
\end{pmatrix},
\qquad
M_k:=M_k^{(0)}+M_k^{(1)},
\]
\[
\mathbf e_0:=\begin{pmatrix}1\\0\end{pmatrix},
\qquad
\mathbf e_1:=\begin{pmatrix}0\\1\end{pmatrix},
\qquad
\mathbf 1:=\begin{pmatrix}1\\1\end{pmatrix},
\qquad
\delta_{\mathrm{ZG}}:=\mathrm{dens}(\mathrm{ZG}).
\]
对每个 \(n\ge 0\)，定义尾因子
\[
A_n
:=
\frac{1}{\delta_{\mathrm{ZG}}\zeta(2)}
\lim_{N\to\infty}\mathbf 1^\top M_NM_{N-1}\cdots M_{n+1}\mathbf e_0,
\]
\[
B_n
:=
\frac{1}{\delta_{\mathrm{ZG}}\zeta(2)}
\lim_{N\to\infty}\mathbf 1^\top M_NM_{N-1}\cdots M_{n+1}\mathbf e_1.
\]
则成立：
\begin{enumerate}
  \item 对任意 admissible 前缀
  \[
  \mathbf z=(z_1,\dots,z_n)\in\{0,1\}^n,
  \qquad z_iz_{i+1}=0\ (1\le i<n),
  \]
  其柱集概率满足精确乘积公式
  \[
  \nu_{\mathrm{ZG}}([\mathbf z])
  =
  \left(
  \prod_{k=1}^{n}\frac{p_k^{\,1-z_k}}{p_k+1}
  \right)
  \times
  \begin{cases}
  A_n,& z_n=0,\\
  B_n,& z_n=1;
  \end{cases}
  \]
  \item \(\nu_{\mathrm{ZG}}\) 是一个非齐次一阶 Markov 测度：存在
  \[
  q_n:=\nu_{\mathrm{ZG}}(z_{n+1}=1\mid z_n=0)\in(0,1)
  \]
  使得
  \[
  \nu_{\mathrm{ZG}}(z_{n+1}=1\mid z_1,\dots,z_n)
  =
  \begin{cases}
  0,& z_n=1,\\
  q_n,& z_n=0;
  \end{cases}
  \]
  其中
  \[
  q_n=\frac{B_{n+1}}{p_{n+1}A_{n+1}+B_{n+1}};
  \]
  \item 若记
  \[
  r_n:=\frac{B_n}{A_n},
  \]
  则有精确 Riccati 递推
  \[
  r_n=\frac{p_{n+1}}{p_{n+1}+r_{n+1}},
  \qquad
  q_n=\frac{r_{n+1}}{p_{n+1}+r_{n+1}}.
  \]
\end{enumerate}
\leanverified{paper\_conclusion\_zg\_density\_exact\_inhomogeneous\_markov}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-time-part60f-zg-cylinder-density-matrix-positive}，对任意 admissible 前缀 \(\mathbf z=(z_1,\dots,z_n)\) 有
\[
\nu_{\mathrm{ZG}}([\mathbf z])
=
\frac{1}{\delta_{\mathrm{ZG}}\zeta(2)}
\lim_{N\to\infty}
\mathbf 1^\top M_N\cdots M_{n+1}M_n^{(z_n)}\cdots M_1^{(z_1)}\mathbf e_0.
\]
而直接计算得
\[
M_k^{(0)}\mathbf e_0=\frac{p_k}{p_k+1}\mathbf e_0,\qquad
M_k^{(1)}\mathbf e_0=\frac{1}{p_k+1}\mathbf e_1,
\]
\[
M_k^{(0)}\mathbf e_1=\frac{p_k}{p_k+1}\mathbf e_0,\qquad
M_k^{(1)}\mathbf e_1=0.
\]
因此，对 admissible \(\mathbf z\) 归纳可得
\[
M_n^{(z_n)}\cdots M_1^{(z_1)}\mathbf e_0
=
\left(\prod_{k=1}^{n}\frac{p_k^{\,1-z_k}}{p_k+1}\right)\mathbf e_{z_n}.
\]
代回即得第一条。

对第二条，若 \(z_n=1\)，则由于黄金约束禁止相邻的 \(1\)，必有
\[
\nu_{\mathrm{ZG}}(z_{n+1}=1\mid z_1,\dots,z_n)=0.
\]
若 \(z_n=0\)，则把公共前缀权重约去，得到
\[
\nu_{\mathrm{ZG}}(z_{n+1}=1\mid z_1,\dots,z_n)
=
\frac{\frac{1}{p_{n+1}+1}B_{n+1}}
{\frac{p_{n+1}}{p_{n+1}+1}A_{n+1}+\frac{1}{p_{n+1}+1}B_{n+1}}
=
\frac{B_{n+1}}{p_{n+1}A_{n+1}+B_{n+1}}.
\]
该式仅依赖于末位 \(z_n\)，故 \(\nu_{\mathrm{ZG}}\) 是非齐次一阶 Markov 测度。

最后，由定义
\[
A_n=\frac{1}{\delta_{\mathrm{ZG}}\zeta(2)}
\lim_{N\to\infty}\mathbf 1^\top M_N\cdots M_{n+2}M_{n+1}\mathbf e_0,
\]
\[
B_n=\frac{1}{\delta_{\mathrm{ZG}}\zeta(2)}
\lim_{N\to\infty}\mathbf 1^\top M_N\cdots M_{n+2}M_{n+1}\mathbf e_1,
\]
并代入
\[
M_{n+1}\mathbf e_0
=
\frac{p_{n+1}}{p_{n+1}+1}\mathbf e_0
+\frac{1}{p_{n+1}+1}\mathbf e_1,
\qquad
M_{n+1}\mathbf e_1
=
\frac{p_{n+1}}{p_{n+1}+1}\mathbf e_0,
\]
得
\[
A_n=\frac{p_{n+1}}{p_{n+1}+1}A_{n+1}
+\frac{1}{p_{n+1}+1}B_{n+1},
\qquad
B_n=\frac{p_{n+1}}{p_{n+1}+1}A_{n+1}.
\]
两式相除即得
\[
r_n=\frac{p_{n+1}}{p_{n+1}+r_{n+1}},
\]
再与第二条中的 \(q_n\) 公式合并，便得到
\[
q_n=\frac{r_{n+1}}{p_{n+1}+r_{n+1}}.
\]
证毕。
\end{proof}

\begin{theorem}[任意 admissible 前缀柱集的统一渐近密度公式]\label{thm:conclusion-zg-cylinder-uniform-asymptotic}
对任意 admissible 前缀
\[
\mathbf z=(z_1,\dots,z_n)\in\{0,1\}^n
\]
都有统一渐近
\[
\nu_{\mathrm{ZG}}([\mathbf z])
=
\frac{e^{-\gamma}}{\delta_{\mathrm{ZG}}\log p_n}
\left(
\prod_{k:\,z_k=1}\frac1{p_k}
\right)
\bigl(1+o(1)\bigr),
\qquad n\to\infty,
\]
其中 \(o(1)\) 仅依赖于 \(n\)，而与具体 admissible 词形无关。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zg-density-exact-inhomogeneous-markov} 的第一条，
\[
\nu_{\mathrm{ZG}}([\mathbf z])
=
\left(\prod_{k=1}^{n}\frac{p_k^{\,1-z_k}}{p_k+1}\right)
\times
\begin{cases}
A_n,& z_n=0,\\
B_n,& z_n=1.
\end{cases}
\]
定义
\[
\alpha_n:=\delta_{\mathrm{ZG}}\zeta(2)A_n,
\qquad
\beta_n:=\delta_{\mathrm{ZG}}\zeta(2)B_n.
\]
则 \(\alpha_n,\beta_n\) 正是从第 \(n\) 位之后开始的尾部 admissibility 权重。若当前末位为 \(0\)，则尾部违约只能发生在某个相邻素数对同时被选中，故
\[
1-\alpha_n
\le
\sum_{k>n}\frac{1}{(p_k+1)(p_{k+1}+1)}
=o(1).
\]
若当前末位为 \(1\)，则还必须排除第 \(n+1\) 位取 \(1\) 的事件，因此
\[
1-\beta_n
\le
\frac{1}{p_{n+1}+1}
+
\sum_{k>n+1}\frac{1}{(p_k+1)(p_{k+1}+1)}
=o(1).
\]
于是
\[
A_n=\frac{1+o(1)}{\delta_{\mathrm{ZG}}\zeta(2)},
\qquad
B_n=\frac{1+o(1)}{\delta_{\mathrm{ZG}}\zeta(2)},
\]
并且该 \(o(1)\) 与前缀形状无关。

再把局部因子改写为
\[
\prod_{k=1}^{n}\frac{p_k^{\,1-z_k}}{p_k+1}
=
\left(\prod_{k=1}^{n}\frac{p_k}{p_k+1}\right)
\prod_{k:\,z_k=1}\frac1{p_k}.
\]
由恒等式
\[
\frac{p}{p+1}=\frac{1-p^{-1}}{1-p^{-2}},
\]
得
\[
\prod_{k=1}^{n}\frac{p_k}{p_k+1}
=
\left(\prod_{p\le p_n}\left(1-\frac1p\right)\right)
\left(\prod_{p\le p_n}\left(1-\frac1{p^2}\right)^{-1}\right).
\]
对第一因子应用 Mertens 乘积公式，
\[
\prod_{p\le x}\left(1-\frac1p\right)\sim \frac{e^{-\gamma}}{\log x},
\]
并对第二因子使用
\[
\prod_{p}\left(1-\frac1{p^2}\right)=\frac1{\zeta(2)},
\]
便得到
\[
\prod_{k=1}^{n}\frac{p_k}{p_k+1}
=
\frac{\zeta(2)e^{-\gamma}}{\log p_n}\bigl(1+o(1)\bigr).
\]
与 \(A_n,B_n\) 的渐近合并，即得
\[
\nu_{\mathrm{ZG}}([\mathbf z])
=
\frac{e^{-\gamma}}{\delta_{\mathrm{ZG}}\log p_n}
\left(\prod_{k:\,z_k=1}\frac1{p_k}\right)
\bigl(1+o(1)\bigr).
\]
证毕。
\end{proof}

\begin{theorem}[算术自然密度测度的稀疏律]\label{thm:conclusion-zg-density-measure-sparsity-law}
在概率空间 \((\Sigma_{\mathrm{ZG}},\nu_{\mathrm{ZG}})\) 上定义计数过程
\[
N_n(z):=\sum_{k=1}^{n}z_k.
\]
则有
\[
\EE_{\nu_{\mathrm{ZG}}}[N_n]=\log\log p_n+O(1),
\]
并且
\[
\frac{N_n}{n}\longrightarrow 0
\qquad
\nu_{\mathrm{ZG}}\text{-a.s.}
\]
\end{theorem}
\leanverified{paper\_conclusion\_zg\_density\_measure\_sparsity\_law}
\begin{proof}
记
\[
\pi_n:=\nu_{\mathrm{ZG}}(z_n=1).
\]
由定理 \ref{thm:conclusion-zg-density-exact-inhomogeneous-markov} 的第二条，
\[
\pi_{n+1}=(1-\pi_n)q_n,
\qquad
q_n=\frac{r_{n+1}}{p_{n+1}+r_{n+1}}.
\]
又由第三条，
\[
0<r_n=\frac{p_{n+1}}{p_{n+1}+r_{n+1}}<1,
\]
从而
\[
0<1-r_n=\frac{r_{n+1}}{p_{n+1}+r_{n+1}}\le \frac1{p_{n+1}}.
\]
故
\[
r_n=1+O\!\left(\frac1{p_{n+1}}\right).
\]
将其代回 \(q_n\) 可得
\[
q_n-\frac{1}{p_{n+1}+1}
=
\frac{p_{n+1}(r_{n+1}-1)}
{(p_{n+1}+r_{n+1})(p_{n+1}+1)}
=
O\!\left(\frac1{p_{n+1}^2}\right),
\]
其中使用了 \(r_{n+1}=1+O(p_{n+2}^{-1})\) 与相邻素数的渐近可比性。于是
\[
q_n=\frac{1}{p_{n+1}+1}+O\!\left(\frac1{p_{n+1}^2}\right).
\]
再由 \(0\le \pi_n\le q_{n-1}=O(p_n^{-1})\)，得到
\[
\pi_n
=
(1-\pi_{n-1})q_{n-1}
=
\frac{1}{p_n+1}+O\!\left(\frac1{p_n^2}\right).
\]
于是
\[
\EE_{\nu_{\mathrm{ZG}}}[N_n]
=
\sum_{k=1}^{n}\pi_k
=
\sum_{k=1}^{n}\frac1{p_k}+O(1)
=
\log\log p_n+O(1),
\]
其中最后一步使用素数调和和的经典渐近。

再证几乎处处零密度。对任意 \(\varepsilon>0\)，令 \(n_j:=2^j\)。由 Markov 不等式，
\[
\nu_{\mathrm{ZG}}\!\left(\frac{N_{2^j}}{2^j}>\varepsilon\right)
\le
\frac{\EE[N_{2^j}]}{\varepsilon\,2^j}
\ll
\frac{\log\log p_{2^j}}{2^j}.
\]
由素数定理 \(p_n\sim n\log n\) 知 \(\log\log p_{2^j}=O(j)\)，故
\[
\sum_{j\ge 1}\nu_{\mathrm{ZG}}\!\left(\frac{N_{2^j}}{2^j}>\varepsilon\right)<\infty.
\]
由 Borel--Cantelli 引理，
\[
\frac{N_{2^j}}{2^j}\to 0
\qquad
\nu_{\mathrm{ZG}}\text{-a.s.}
\]
对任意 \(2^j\le n<2^{j+1}\)，由于 \(N_n\le N_{2^{j+1}}\)，有
\[
\frac{N_n}{n}\le \frac{N_{2^{j+1}}}{2^j}
=
2\,\frac{N_{2^{j+1}}}{2^{j+1}},
\]
故全序列同样满足 \(N_n/n\to 0\) 几乎处处。证毕。
\end{proof}

\begin{theorem}[全息推前测度的块熵渐近与零信息维数]\label{thm:conclusion-zg-holographic-pushforward-entropy-information-dimension}
沿用定理 \ref{thm:xi-time-part60f-zg-cylinder-measure-local-dimension-zero} 的记号，固定
\[
\Xi_B:\Sigma_{\mathrm{ZG}}\to K_{B,v},
\qquad
\mu_B:=(\Xi_B)_\#\nu_{\mathrm{ZG}},
\qquad B=2^L.
\]
对每个 \(n\ge 1\)，记
\[
\mathcal P_n:=\{[\mathbf z]: \mathbf z\in\{0,1\}^n,\ \mathbf z\ \text{admissible}\}
\]
为长度 \(n\) 的 cylinder 分割，并定义块熵
\[
H_n(\nu_{\mathrm{ZG}})
:=
-\sum_{C\in\mathcal P_n}\nu_{\mathrm{ZG}}(C)\log \nu_{\mathrm{ZG}}(C).
\]
则有渐近公式
\[
H_n(\nu_{\mathrm{ZG}})=\log p_n+\log\log p_n+O(1).
\]
从而 \(\mu_B\) 的信息维数满足
\[
\dim_I(\mu_B)=0.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-zg-density-exact-inhomogeneous-markov}，\(\nu_{\mathrm{ZG}}\) 为一阶非齐次 Markov 测度。记二元 Shannon 熵函数
\[
h(u):=-u\log u-(1-u)\log(1-u).
\]
再记
\[
\pi_n:=\nu_{\mathrm{ZG}}(z_n=1).
\]
则链式法则给出
\[
H_n(\nu_{\mathrm{ZG}})
=
h(\pi_1)+\sum_{k=1}^{n-1}H(z_{k+1}\mid z_1,\dots,z_k).
\]
而由一阶 Markov 结构，
\[
H(z_{k+1}\mid z_1,\dots,z_k)
=
(1-\pi_k)\,h(q_k),
\]
因为在 \(z_k=1\) 时下一位被强制为 \(0\)。由定理 \ref{thm:conclusion-zg-density-measure-sparsity-law} 的证明，
\[
\pi_k=\frac{1}{p_k+1}+O\!\left(\frac1{p_k^2}\right),
\qquad
q_k=\frac{1}{p_{k+1}+1}+O\!\left(\frac1{p_{k+1}^2}\right).
\]
对小 \(u\)，有
\[
h(u)=u\log\frac1u+u+O(u^2).
\]
故
\[
h(q_k)
=
\frac{\log p_{k+1}}{p_{k+1}}
+\frac{1}{p_{k+1}}
+O\!\left(\frac{\log p_{k+1}}{p_{k+1}^2}\right),
\]
进而
\[
(1-\pi_k)h(q_k)
=
\frac{\log p_{k+1}}{p_{k+1}}
+\frac{1}{p_{k+1}}
+O\!\left(\frac{\log p_{k+1}}{p_{k+1}^2}\right).
\]
对 \(k\) 求和，并使用经典素数和公式
\[
\sum_{p\le x}\frac{\log p}{p}=\log x+O(1),
\qquad
\sum_{p\le x}\frac1p=\log\log x+O(1),
\]
即可得到
\[
H_n(\nu_{\mathrm{ZG}})
=
\log p_n+\log\log p_n+O(1).
\]

再由定理 \ref{thm:xi-time-part9g-holographic-prefix-isometry-on-line} 的柱--球对应，长度 \(n\) 的 cylinder 恰对应于 \(K_{B,v}\) 上半径 \(B^{-n}\) 的地址球，因此 \(\mu_B\) 在自然 \(B\)-进尺度上的分割熵正是 \(H_n(\nu_{\mathrm{ZG}})\)。于是
\[
\dim_I(\mu_B)
=
\lim_{n\to\infty}\frac{H_n(\nu_{\mathrm{ZG}})}{n\log B}.
\]
由素数定理 \(p_n\sim n\log n\)，上式分子为 \(O(\log n)\)，故极限为 \(0\)。证毕。
\end{proof}

\begin{corollary}[算术自然密度测度不可能是任何平稳有限阶 Markov 测度]\label{cor:conclusion-zg-density-not-stationary-finite-order-markov}
\(\nu_{\mathrm{ZG}}\) 不等于 \(\Sigma_{\mathrm{ZG}}\) 上任何平稳有限阶 Markov 测度。等价地，它不能与任何平稳有限阶 Markov 机制在全部 cylinder 概率上完全一致。
\end{corollary}
\leanverified{paper\_conclusion\_zg\_density\_not\_stationary\_finite\_order\_markov}
\begin{proof}
反设 \(\eta=\nu_{\mathrm{ZG}}\) 是某个平稳有限阶 Markov 测度。若其阶数为 \(0\)，则它同时也是平稳 \(1\)-阶 Markov 测度；故不妨设其阶数为某个 \(r\ge 1\)。由定理 \ref{thm:xi-time-part60f-zg-cylinder-measure-local-dimension-zero}，\(\nu_{\mathrm{ZG}}\) 具有全支撑，故
\[
\eta([0^r1])=\nu_{\mathrm{ZG}}([0^r1])>0.
\]
于是存在常数
\[
c:=\eta(z_{r+1}=1\mid z_1=\cdots=z_r=0)>0.
\]
由平稳 \(r\)-阶 Markov 性，对一切 \(n\ge r\) 都有
\[
\eta(z_{n+1}=1\mid z_{n-r+1}=\cdots=z_n=0)=c.
\]
另一方面，定理 \ref{thm:conclusion-zg-density-exact-inhomogeneous-markov} 表明 \(\nu_{\mathrm{ZG}}\) 的相同条件概率只取决于末位，并且恰为
\[
\nu_{\mathrm{ZG}}(z_{n+1}=1\mid z_{n-r+1}=\cdots=z_n=0)=q_n.
\]
但由定理 \ref{thm:conclusion-zg-density-measure-sparsity-law} 的证明，
\[
q_n=\frac{1}{p_{n+1}+1}+O\!\left(\frac1{p_{n+1}^2}\right)\longrightarrow 0.
\]
这与 \(c>0\) 矛盾。故 \(\nu_{\mathrm{ZG}}\) 不能是任何平稳有限阶 Markov 测度。证毕。
\end{proof}

以上结果把 prime 位置权重、Ramanujan 阴影、黄金约束与 \(2^L\)-进全息地址读出收束为同一条精确链：算术上，\(\mathrm{ZG}\) 的强制整除 Dirichlet 级数与 Ramanujan 阴影在 Möbius--divisor 对偶下可逆等价，并在每个有限 primorial 截断上完整恢复非齐次硬核链的全部 admissible 柱事件；局部上，\(\nu_{\mathrm{ZG}}\) 因而由随素数位置缓慢变化的非齐次一阶 Markov 机制精确驱动，且最近邻排斥在 shadow 侧表现为相邻 prime 通道的严格负偏差，从而给出标量 Euler 不可约性的局部来源；全局上，典型路径中的激活位数仅按 \(\log\log p_n\) 增长，频率几乎处处趋于零；几何上，全息推前测度虽然支撑于正 Hausdorff 维的地址集，但其块熵只以 \(\log p_n\) 量级增长，因而信息维数坍缩为零；动力学上，这种绝对时间依赖的条件概率衰减又严格排除了任何平稳有限阶 Markov 实现。于是，\(\nu_{\mathrm{ZG}}\) 既不同于连续势 Gibbs 态，也不同于平稳有限阶 Markov 测度，而是一类其算术阴影、有限柱结构与全息几何都被 prime-index 最近邻约束同时刚化的非齐次算术测度。
