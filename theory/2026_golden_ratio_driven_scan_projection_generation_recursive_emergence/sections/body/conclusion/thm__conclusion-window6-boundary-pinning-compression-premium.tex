\begin{theorem}[canonical boundary 子格的双层钉扎熵与记忆差]\label{thm:conclusion-window6-boundary-pinning-two-scales-memory-gap}
记
\[
G:=G_6^{\mathrm{bin}},
\qquad
Z_{\partial}\subset Z(G)\cong (\ZZ/2\ZZ)^8,
\qquad
A_6\cong M_2(\CC)^{\oplus 8}\oplus M_3(\CC)^{\oplus 4}\oplus M_4(\CC)^{\oplus 9},
\]
并令
\[
I_{\partial}\cong M_2(\CC)^{\oplus 3}
\]
为 canonical boundary 三块理想。则 canonical boundary 结构在抽象群层与抽象半单代数层分别具有两条不可约的离线钉扎尺度：
\[
B_{\mathrm{pin}}^{\mathrm{grp}}
=
\left\lceil \log_2 97155\right\rceil
=
17,
\qquad
B_{\mathrm{pin}}^{\mathrm{alg}}
=
\left\lceil \log_2 \binom{8}{3}\right\rceil
=
6.
\]
并且两层之间存在精确记忆差下界
\[
B_{\mathrm{pin}}^{\mathrm{grp}}-B_{\mathrm{pin}}^{\mathrm{alg}}\ge 11.
\]
更进一步，即使把全部三维中心商
\[
G/U,\qquad \dim_{\FF_2}U=3,
\]
一并列入可观测接口，boundary 的几何来源仍在群层被完全抹平。
\leanverified{paper\_conclusion\_boundary\_pinning\_two\_scales\_memory\_gap}
\leanverified{paper\_conclusion\_prime\_register\_one\_register\_serialization\_gap\_seeds}
\leanverified{paper\_conclusion\_prime\_register\_one\_register\_serialization\_gap\_package}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-boundary-center-seventeen-bit-barrier}，
\[
\bigl|\operatorname{Orb}_{\operatorname{Aut}(G)}(Z_{\partial})\bigr|=97155,
\]
故群层的最小二进制钉扎账本满足
\[
B_{\mathrm{pin}}^{\mathrm{grp}}=\left\lceil \log_2 97155\right\rceil=17.
\]
再由推论 \ref{cor:conclusion-window6-center-quotient-grassmann-collapse}，任意两个三维中心子空间 \(U,V\le Z(G)\) 都满足
\[
G/U\cong G/V\cong (\ZZ/2\ZZ)^5\times S_3^{\,4}\times S_4^{\,9},
\]
故即使把全部三维中心商都列为可观测对象，boundary 的几何来源仍无法在群层恢复。

另一方面，由定理 \ref{thm:conclusion-window6-semisimple-boundary-forgetfulness}，
\[
\Aut_{\CC\text{-alg}}(A_6)
\]
在全部 \(M_2(\CC)^{\oplus 3}\)-型双边理想上作用传递，轨道大小恰为
\[
\binom{8}{3}=56.
\]
故抽象半单代数层的最小二进制钉扎账本满足
\[
B_{\mathrm{pin}}^{\mathrm{alg}}=\left\lceil \log_2 56\right\rceil=6.
\]
相减即得
\[
B_{\mathrm{pin}}^{\mathrm{grp}}-B_{\mathrm{pin}}^{\mathrm{alg}}\ge 17-6=11.
\]
证毕。
\end{proof}

\begin{theorem}[abstract 群层上 canonical boundary 子格的函子性不可恢复]\label{thm:conclusion-window6-boundary-subspace-no-aut-equivariant-projector}
记
\[
G:=G_6^{\mathrm{bin}},
\qquad
Z_{\partial}\subset Z(G)\cong (\ZZ/2\ZZ)^8
\]
为 canonical boundary 三维中心子空间。则不存在任何 \(\Aut(G)\)-等变幂等投影
\[
\Pi:Z(G)\to Z(G)
\]
满足
\[
\operatorname{Im}\Pi=Z_{\partial}.
\]
更强地，任何仅依赖抽象群 \(G\) 而不附加 boundary 几何 pinning 的函子性程序，都不能从
\[
\Gr(3,8;\FF_2)
\]
的
\[
\binom{8}{3}_2=97155
\]
个三维子空间中规范挑出 \(Z_{\partial}\)。
\end{theorem}
\begin{proof}
若存在这样的 \(\Aut(G)\)-等变幂等投影 \(\Pi\)，则对任意 \(\alpha\in\Aut(G)\) 都有
\[
\alpha\circ \Pi=\Pi\circ \alpha.
\]
于是
\[
\alpha(\operatorname{Im}\Pi)=\operatorname{Im}\Pi,
\]
故 \(\operatorname{Im}\Pi\) 必为 \(\Aut(G)\)-不变子群，亦即 \(G\) 的特征子群。将 \(\operatorname{Im}\Pi=Z_{\partial}\) 代入，便与定理 \ref{thm:conclusion-window6-center-characteristic-rigidity} 矛盾；该定理已证明 \(Z_{\partial}\) 不是特征子群。

更强地，由定理 \ref{thm:conclusion-window6-boundary-center-seventeen-bit-barrier}，
\[
\Aut(G)
\]
在
\[
\Gr(3,8;\FF_2)
\]
上对 \(Z_{\partial}\) 的轨道大小恰为
\[
\bigl|\operatorname{Orb}_{\Aut(G)}(Z_{\partial})\bigr|
=
\binom{8}{3}_2
=
97155.
\]
故若无额外几何 pinning，抽象群层面并不存在把 \(Z_{\partial}\) 与其余三维中心子空间区分开的规范选择。证毕。
\leanverified{paper\_conclusion\_window6\_boundary\_subspace\_no\_aut\_equivariant\_projector}
\end{proof}

\begin{theorem}[几何对角坍缩的 \texorpdfstring{\(\mathfrak{so}(10)\)}{so(10)} 最优性与连续包络压缩阶梯]\label{thm:conclusion-window6-boundary-center-compression-ladder}
记 canonical boundary parity 为
\[
P_6\cong (\ZZ/2\ZZ)^3,
\qquad
P_6^{\mathrm{geo}}=\langle(1,1,1)\rangle\cong \ZZ/2\ZZ.
\]
则：
\begin{enumerate}
  \item 对任意紧连通 Lie 群 \(H\) 若
  \[
  \Lie(H)\cong \mathfrak{so}(10),
  \]
  则任意群同态
  \[
  f:P_6\longrightarrow Z(H)[2]
  \]
  都满足
  \[
  \dim_{\FF_2}\ker f\ge 2;
  \]
  因而几何对角坍缩
  \[
  P_6\twoheadrightarrow P_6^{\mathrm{geo}}
  \]
  恰以最小可能代价丢失两位 parity，是 \(\mathfrak{so}(10)\)-型连通包络中的核维数最优压缩。
  \item 对任意紧连通 Lie 群 \(G_{\mathrm{SM}}\) 若
  \[
  \Lie(G_{\mathrm{SM}})\cong \mathfrak u(1)\oplus \mathfrak{su}(2)\oplus \mathfrak{su}(3),
  \]
  则任意
  \[
  f:P_6\longrightarrow Z(G_{\mathrm{SM}})[2]
  \]
  都满足
  \[
  \dim_{\FF_2}\ker f\ge 1.
  \]
\end{enumerate}
因此连续统一包络在中心层面存在严格压缩阶梯
\[
(\ZZ/2\ZZ)^3
\rightsquigarrow
\text{SM-type envelope：至少损失 \(1\) bit}
\rightsquigarrow
\text{\(\mathfrak{so}(10)\)-type envelope：至少损失 \(2\) bits}.
\]
\leanverified{paper\_conclusion\_window6\_compression\_ladder\_spin10}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-boundary-parity-zero-one-three-law}，
\[
P_6^{\mathrm{geo}}=\langle(1,1,1)\rangle,
\qquad
P_6/P_6^{\mathrm{geo}}\cong (\ZZ/2\ZZ)^2,
\]
故几何可实现压缩的像秩为 \(1\)、核秩为 \(2\)。

若 \(H\) 紧连通且 \(\Lie(H)\cong \mathfrak{so}(10)\)，则
\[
H\cong \mathrm{Spin}(10)/\Gamma
\]
其中 \(\Gamma\le Z(\mathrm{Spin}(10))\cong \ZZ/4\ZZ\)。于是
\[
Z(H)[2]
\]
作为 \(\ZZ/4\ZZ\) 的商的 \(2\)-挠部分，\(\FF_2\)-维数至多为 \(1\)。故任意
\[
f:P_6\to Z(H)[2]
\]
的像秩至多 \(1\)，从而
\[
\dim_{\FF_2}\ker f\ge 3-1=2.
\]
因此几何对角坍缩已达到所有 \(\mathfrak{so}(10)\)-型连通包络中的最小核维数。

另一方面，由定理 \ref{thm:xi-window6-sm-envelope-center-2torsion-obstruction}，
\[
\dim_{\FF_2}Z(G_{\mathrm{SM}})[2]\le 2.
\]
故从 \(P_6\cong (\ZZ/2\ZZ)^3\) 到 \(Z(G_{\mathrm{SM}})[2]\) 的任意群同态必满足
\[
\dim_{\FF_2}\ker f\ge 1.
\]
证毕。
\end{proof}

\begin{theorem}[对称 completion 的离线/在线双重溢价分解]\label{thm:conclusion-window6-boundary-offline-online-premium-bifurcation}
window-\(6\) 中，对称保持的 hidden-sector 成本天然分裂为两类不可替代的溢价：
\begin{enumerate}
  \item \textbf{离线钉扎溢价：}
  \[
  B_{\mathrm{pin}}^{\mathrm{grp}}=17,
  \qquad
  B_{\mathrm{pin}}^{\mathrm{alg}}=6;
  \]
  \item \textbf{在线 completion 溢价：}
  相对于裸可逆预算
  \[
  b_{\mathrm{inv}}=2,
  \]
  boundary-parity 等变 completion 至少额外支付
  \[
  +1\ \text{bit},
  \]
  而 full-center 等变 completion 至少额外支付
  \[
  +2\ \text{bits}.
  \]
\end{enumerate}
特别地，不存在 boundary-equivariant 的四值伴随场。
因此，“识别哪一个 hidden sector”与“在已识别 sector 内完成局部解码”属于两种不同资源：前者不能由后者替代，后者也不能回收前者。
\end{theorem}
\begin{proof}
离线部分由定理 \ref{thm:conclusion-window6-boundary-pinning-two-scales-memory-gap} 直接给出。

在线部分由定理 \ref{thm:conclusion-window6-local-index-binary-log-symmetry-surcharge} 给出：
\[
b_{\mathrm{inv}}=2,
\qquad
B_{\partial}\ge 3,
\qquad
B_{\mathrm{center}}\ge 4.
\]
故
\[
B_{\partial}-b_{\mathrm{inv}}\ge 1,
\qquad
B_{\mathrm{center}}-b_{\mathrm{inv}}\ge 2.
\]
又由推论 \ref{cor:xi-time-part9x-no-boundary-equivariant-four-valued-companion-field}，四值标签即 \(B=2\) completion，而它与 boundary 等变性不相容，因此不存在 boundary-equivariant 的四值伴随场。

二者不可互换的原因在于：若 canonical 三维中心子格尚未被钉扎，则“对某个固定 \(Z_{\partial}\) 等变”的约束尚无对象层意义；反过来，即使 \(Z_{\partial}\) 已被离线定位，局部注入性的在线下界
\[
B_{\partial}\ge 3,\qquad B_{\mathrm{center}}\ge 4
\]
仍然独立存在。证毕。
\end{proof}

\begin{corollary}[window-\texorpdfstring{$6$}{6} parity 的锚点性与跨尺度非内禀性]\label{cor:conclusion-window6-boundary-anchor-nonintrinsic-crossscale}
window-\(6\) 的 boundary parity 不是一个可以自动向更高分辨率传播的内禀 \(\ZZ_2\)-场，而是严格锚定在 \(m=6\) 的最小二层缺陷：
\begin{enumerate}
  \item 只有在
  \[
  \Delta_6^{\mathrm{bdry}}=\{0,34\}
  \]
  时，boundary uplift 才由唯一非零增量支配，从而 parity 可无附加选择地恢复；
  \item 一旦进入
  \[
  \Delta_7^{\mathrm{bdry}}=\{0,55,89\},
  \qquad
  \Delta_8^{\mathrm{bdry}}=\{0,89,144\},
  \]
  三层 boundary uplift 已迫使任何二元化引入额外协议门或外加寄存器；
  \item 若还同时要求尾部局域性、无寄存器的内禀 parity 延拓以及 \(\mathfrak{so}(10)\)-型中心保持，则该机制在首个 uplift 步即被严格阻断。
\end{enumerate}
\end{corollary}
\begin{proof}
前两条由定理 \ref{thm:conclusion-window6-choicefree-parity-anchor-isolated} 与定理 \ref{thm:conclusion-window6-no-gateless-crossscale-parity-tower} 直接给出：choice-free parity 只在 \(m=6\) 出现，而 \(m=7,8\) 已失去这种无协议唯一性。

第三条则正是定理 \ref{thm:conclusion-window6-first-uplift-so10-trilemma} 的内容：局域 tail-patch、无寄存器 parity 延拓与 \(\mathfrak{so}(10)\)-型中心保持三者不能同时成立。故该 parity 不能被解释为跨尺度自发传播的内禀 \(\ZZ_2\)-场。证毕。
\end{proof}
