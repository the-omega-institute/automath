\begin{theorem}[有限缺陷剖面的最小 Cauchy 秩]\label{thm:conclusion-finite-defect-profile-minimal-cauchy-rank}
设有限缺陷扫描剖面可写成
\[
H(x)=\sum_{j=1}^{\kappa}\frac{w_j}{(x-\gamma_j)^2+a_j^2},
\qquad
w_j>0,\ a_j>0,
\]
其中 \((\gamma_j,a_j)\) 两两不同。考虑 Cauchy 型正核类
\[
\mathcal C_r:=
\left\{
K(x,y)=\sum_{\ell=1}^{r}
\frac{\omega_\ell}{(x-\alpha_\ell+\mathrm{i}b_\ell)(y-\alpha_\ell-\mathrm{i}b_\ell)}:
\ \omega_\ell>0,\ b_\ell>0
\right\}.
\]
若某个 \(K\in\mathcal C_r\) 满足
\[
K(x,x)=H(x)\qquad(\forall x\in\RR),
\]
则必有 \(r\ge \kappa\)。并且存在 \(K_\star\in\mathcal C_\kappa\) 使
\[
K_\star(x,x)=H(x)\qquad(\forall x\in\RR).
\]
故有限缺陷剖面的内禀最小 Cauchy 秩恰等于 \(\kappa\)。
\end{theorem}
\begin{proof}
存在性由定理 \ref{thm:xi-basepoint-scan-finite-rank-rkhs} 直接给出：取
\[
p_j=\gamma_j-\mathrm{i}a_j,
\qquad
K_\star(x,y)=\sum_{j=1}^{\kappa}\frac{w_j}{(x-p_j)(y-\overline{p_j})},
\]
则 \(K_\star\) 为正定核且满足 \(K_\star(x,x)=H(x)\)。故最小可能秩至多为 \(\kappa\)。

反之，若 \(K\in\mathcal C_r\) 亦满足 \(K(x,x)=H(x)\)，则其对角剖面可写成
\[
H(x)=\sum_{\ell=1}^{r}\frac{\omega_\ell}{(x-\alpha_\ell)^2+b_\ell^2}.
\]
把该表示视为有限原子缺陷测度
\[
\nu_K:=\sum_{\ell=1}^{r}\omega_\ell\,\delta_{(\alpha_\ell,b_\ell-1)},
\]
而原剖面对应
\[
\nu_H:=\sum_{j=1}^{\kappa}w_j\,\delta_{(\gamma_j,a_j-1)}.
\]
由定理 \ref{thm:xi-comoving-horizon-scan-fourier-inversion} 的有限原子可辨识性，边界剖面 \(H\) 唯一决定有限原子数据，故 \(\nu_K=\nu_H\)。特别地，\(\nu_K\) 的原子数不得少于 \(\kappa\)，即 \(r\ge \kappa\)。上下界相遇，故最小秩恰为 \(\kappa\)。证毕。
\end{proof}

\begin{corollary}[有限缺陷剖面的内禀最小核秩]\label{cor:conclusion-finite-defect-profile-intrinsic-minimal-rank}
有限缺陷剖面中的原子数 \(\kappa\) 不是表象依赖的拟合参数，而是由边界剖面 \(H\) 自身强制给出的内禀最小核秩。
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-finite-defect-profile-minimal-cauchy-rank} 已证明：任何精确核化都必须至少使用 \(\kappa\) 个 Cauchy 原子，而 \(\kappa\) 个原子又总是可达。故 \(\kappa\) 即剖面本身的内禀最小核秩。证毕。
\end{proof}

\begin{theorem}[深度层的 Jordan--H\"older 型唯一旗标]\label{thm:conclusion-finite-defect-depth-jordan-holder-flag}
设有限原子缺陷测度为
\[
\nu=\sum_{j=1}^{M}m_j\,\delta_{(\gamma_j,\delta_j)},
\qquad
0<\delta^{(1)}<\delta^{(2)}<\cdots<\delta^{(r)}
\]
为全部不同深度值。定义严格递增部分和
\[
\nu^{(<1)}:=0,
\qquad
\nu^{(<s+1)}:=\sum_{\delta_j\le \delta^{(s)}}m_j\,\delta_{(\gamma_j,\delta_j)}
\qquad(1\le s\le r).
\]
则边界剖面 \(H\) 唯一决定旗标
\[
0=\nu^{(<1)}<\nu^{(<2)}<\cdots<\nu^{(<r+1)}=\nu,
\]
并且每一层满足
\[
\nu^{(<s+1)}-\nu^{(<s)}
=
\sum_{\delta_j=\delta^{(s)}}m_j\,\delta_{(\gamma_j,\delta_j)}.
\]
若记第 \(s\) 步剥离后的残差 Fourier 变换为
\[
\widehat H^{(s)}(\xi)
:=
\widehat H(\xi)-\sum_{t=1}^{s-1}
4\pi\sum_{\delta_j=\delta^{(t)}}
m_j\frac{\delta^{(t)}}{1+\delta^{(t)}}e^{-(1+\delta^{(t)})|\xi|}e^{-\mathrm{i}\gamma_j\xi},
\]
则有精确渐近
\[
e^{(1+\delta^{(s)})|\xi|}\widehat H^{(s)}(\xi)
=
G_s(\xi)+O\!\left(e^{-(\delta^{(s+1)}-\delta^{(s)})|\xi|}\right)
\qquad(|\xi|\to\infty),
\]
其中
\[
G_s(\xi)=
4\pi\sum_{\delta_j=\delta^{(s)}}
m_j\frac{\delta^{(s)}}{1+\delta^{(s)}}e^{-\mathrm{i}\gamma_j\xi},
\]
并约定当 \(s=r\) 时误差项为空。
因此，有限缺陷测度的恢复不是一次性识别，而是按深度递增次序唯一生成的一条分层分解。
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-comoving-horizon-scan-fourier-inversion} 给出了有限缺陷剖面的 Fourier--Laplace 指纹
\[
\widehat H(\xi)=4\pi\sum_{j=1}^{M}m_j\frac{\delta_j}{1+\delta_j}e^{-(1+\delta_j)|\xi|}e^{-\mathrm{i}\gamma_j\xi}.
\]
把最浅深度 \(\delta^{(1)}\) 的指数因子提出，可得
\[
e^{(1+\delta^{(1)})|\xi|}\widehat H(\xi)
=
G_1(\xi)+O\!\left(e^{-(\delta^{(2)}-\delta^{(1)})|\xi|}\right),
\]
故第一层的横向三角多项式 \(G_1\) 被唯一锁定，因而全部满足 \(\delta_j=\delta^{(1)}\) 的 \((\gamma_j,m_j)\) 唯一恢复。随后从 \(\widehat H\) 中剥去这一层并重复同一过程，便得到第 \(s\) 步的残差展开。每一步的唯一恢复性都由同一定理保证，故整条旗标唯一。证毕。
\end{proof}

\begin{corollary}[最近临界层作为第一 graded piece]\label{cor:conclusion-finite-defect-nearest-critical-layer-graded-piece}
对有限缺陷剖面而言，“距离临界线最近的缺陷层”不是附加语义，而是由高频尾部自动生成的第一层 graded piece；其后的全部层仅表现为更快衰减的残差项。
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-finite-defect-depth-jordan-holder-flag} 表明，经过指数归一化 \(e^{(1+\delta^{(1)})|\xi|}\) 后，\(\widehat H\) 的首项极限恰为第一层三角多项式 \(G_1\)，而其余各层全部落入更快衰减的误差项中。故最近临界层自动成为第一 graded piece。证毕。
\end{proof}

\begin{theorem}[非公度双频的全局去缠绕]\label{thm:conclusion-finite-defect-incommensurable-bifrequency-global-dealias}
在定理 \ref{thm:xi-bifrequency-finite-sampling-holographic-rigidity} 的深度互异设定下，取两个非零频率
\[
\xi,\xi'\in\RR\setminus\{0\},
\qquad
\frac{\xi}{\xi'}\notin\QQ.
\]
则无需任何先验横向窗口约束，仅由两组各 \(2\kappa\) 个样本
\[
\{F(\xi,s_0+n\Delta s)\}_{n=0}^{2\kappa-1},
\qquad
\{F(\xi',s_0+n\Delta s)\}_{n=0}^{2\kappa-1}
\]
即可唯一恢复整个有限原子族
\[
\{(\gamma_j,\delta_j,a_j)\}_{j=1}^{\kappa}.
\]
换言之，非公度双频在全局上已自动解除横向别名，任何先验窗口条件都不再必需。
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-bifrequency-finite-sampling-holographic-rigidity} 已经证明：在 \(\delta_j\) 两两不同且 \(\xi/\xi'\notin\QQ\) 时，全部 \(4\kappa\) 个双频有限样本唯一确定整个位相--高度缺陷测度。其证明中对每个原子都得到
\[
c_j(\xi)=A_j e^{-\mathrm{i}\gamma_j\xi},
\qquad
c_j(\xi')=A_j e^{-\mathrm{i}\gamma_j\xi'},
\qquad A_j>0.
\]
若另有 \(\widetilde\gamma_j\) 产生同一对系数，则
\[
e^{-\mathrm{i}(\gamma_j-\widetilde\gamma_j)\xi}=1,
\qquad
e^{-\mathrm{i}(\gamma_j-\widetilde\gamma_j)\xi'}=1.
\]
若 \(\gamma_j\neq \widetilde\gamma_j\)，便推出 \(\xi/\xi'\in\QQ\)，与假设矛盾。故每个 \(\gamma_j\) 全局唯一，继而全部 \(a_j\) 亦由振幅模恢复。证毕。
\end{proof}

\begin{corollary}[非公度双频的全局唯一化]\label{cor:conclusion-finite-defect-incommensurable-bifrequency-global-uniqueness}
非公度双频提供的不是局部去缠绕改良，而是全局唯一化：横向先验窗口在该口径下已不再是恢复所必需的数据。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-finite-defect-incommensurable-bifrequency-global-dealias}，当 \(\xi/\xi'\notin\QQ\) 时，双频有限样本已经全局唯一决定整个原子族。故横向窗口只剩冗余先验，而非恢复必需量。证毕。
\end{proof}

\begin{theorem}[公度双频的不可去别名格]\label{thm:conclusion-finite-defect-commensurable-bifrequency-alias-lattice}
仍在双频有限采样的设定下，若
\[
\frac{\xi}{\xi'}\in\QQ,
\]
则存在正数 \(L>0\) 使得对任意有限原子族
\[
\{(\gamma_j,\delta_j,a_j)\}_{j=1}^{\kappa}
\]
与任意整数向量 \((n_1,\dots,n_\kappa)\in\ZZ^\kappa\)，平移后的原子族
\[
\{(\gamma_j+n_jL,\delta_j,a_j)\}_{j=1}^{\kappa}
\]
产生完全相同的双频样本数据。因而若不附加额外横向先验，公度双频的全局恢复必然失败。
\end{theorem}
\begin{proof}
设
\[
\xi=m\omega,
\qquad
\xi'=n\omega,
\]
其中 \(m,n\in\ZZ\) 互素且 \(\omega\neq 0\)。令
\[
L:=\frac{2\pi}{|\omega|}.
\]
则
\[
e^{-\mathrm{i}L\xi}=e^{-2\pi \mathrm{i}m}=1,
\qquad
e^{-\mathrm{i}L\xi'}=e^{-2\pi \mathrm{i}n}=1.
\]
因此，对任意整数 \(r\) 都有
\[
e^{-\mathrm{i}(\gamma_j+rL)\xi}=e^{-\mathrm{i}\gamma_j\xi},
\qquad
e^{-\mathrm{i}(\gamma_j+rL)\xi'}=e^{-\mathrm{i}\gamma_j\xi'}.
\]
而双频样本只通过这些相位因子与 \((\delta_j,a_j)\) 进入，故对每个原子独立施加 \(\gamma_j\mapsto\gamma_j+n_jL\) 不改变任何样本。于是全局唯一恢复不可能成立。证毕。
\end{proof}

\begin{corollary}[横向窗口阈值的几何必需性]\label{cor:conclusion-finite-defect-commensurable-window-threshold-necessity}
在公度双频口径下，任何横向窗口条件都不是技术附加，而是解除别名格所必需的精确几何约束。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-finite-defect-commensurable-bifrequency-alias-lattice}，若不附加横向先验，则每个原子都带有独立的别名格平移自由度。故要恢复唯一性，必须加入窗口或等价的横向定位约束。证毕。
\end{proof}
