\subsection{Lee--Yang 的 \texorpdfstring{$\eta$}{eta} 消元三次、判别式椭圆脊与 \texorpdfstring{$S_3$}{S3} 闭包分解}\label{subsec:conclusion-leyang-eta-cubic-discriminant-s3-galois-jacobian-lfactor}
沿用定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion}、定理 \ref{thm:xi-leyang-eta-degree3-complete-ramification} 与结论 \ref{con:xi-terminal-zm-leyang-compression-cubic-preimage-discriminant-s3} 的记号。
记
\[
H_{\mathrm{LY}}:\ u^2=-y(y-1)P(y),\qquad
P(y)=256y^3+411y^2+165y+32,
\]
\[
A(y)=128y^3+219y^2+69y+16,\qquad
K=\QQ(\sqrt{-3}),
\]
并固定
\[
\eta=\frac{A(y)-3\sqrt{-3}\,u}{16(2y+1)^3}\in K(H_{\mathrm{LY}})^\times,
\qquad
\eta^\iota=\eta^{-1}.
\]

\begin{theorem}[\texorpdfstring{$\eta$}{eta} 坐标下的三次消元方程与判别式闭式分解]\label{thm:conclusion-leyang-eta-cubic-elimination-discriminant-palindromic}
\leanverified{paper\_conclusion\_leyang\_eta\_cubic\_elimination\_discriminant\_palindromic}
在 \(K(H_{\mathrm{LY}})\) 中，\(y\) 作为 \(K(\eta)\)-代数元满足严格三次方程
\begin{equation}\label{eq:conclusion-leyang-eta-cubic-star}
\begin{aligned}
64(\eta-1)^2y^3
&+3(32\eta^2-73\eta+32)y^2\\
&+3(16\eta^2-23\eta+16)y+8(\eta-1)^2=0.
\end{aligned}
\end{equation}
其关于 \(y\) 的判别式分解为
\begin{equation}\label{eq:conclusion-leyang-eta-cubic-discriminant-dagger}
{\operatorname{Disc}_{y}\!\left(
64(\eta-1)^2y^3
+3(32\eta^2-73\eta+32)y^2
+3(16\eta^2-23\eta+16)y
+8(\eta-1)^2\right)}
=-3^9\eta^2\,\mathscr D(\eta),
\qquad
\mathscr D(\eta):=
2304\eta^4-8896\eta^3+13157\eta^2-8896\eta+2304.
\end{equation}
并且 \(\mathscr D\) 为自互反四次多项式：
\[
\mathscr D(\eta)=\eta^4\mathscr D(\eta^{-1}).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion}，
\[
\eta+\eta^{-1}
=\frac{A(y)}{8(2y+1)^3}.
\]
移项并乘以 \(\eta\) 得
\[
A(y)\eta=8(2y+1)^3(\eta^2+1).
\]
代入
\(
A(y)=128y^3+219y^2+69y+16
\)
与
\(
(2y+1)^3=8y^3+12y^2+6y+1
\)
并整理，得到 \eqref{eq:conclusion-leyang-eta-cubic-star}。

把 \eqref{eq:conclusion-leyang-eta-cubic-star} 视为 \(ay^3+by^2+cy+d\)，即
\[
a=64(\eta-1)^2,\ 
b=3(32\eta^2-73\eta+32),\ 
c=3(16\eta^2-23\eta+16),\ 
d=8(\eta-1)^2,
\]
代入三次判别式公式并化简即得 \eqref{eq:conclusion-leyang-eta-cubic-discriminant-dagger}。
最后，\(\mathscr D\) 的首末项与次末项系数对称，故满足自互反关系。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\eta:H_{\mathrm{LY}}\to\PP^1$}{eta map} 的分歧值完全显式刻画]\label{thm:conclusion-leyang-eta-branch-values-explicit}
将 \(\eta\) 视为 \(K\)-上次数 \(3\) 态射
\[
\eta:H_{\mathrm{LY}}\longrightarrow \PP^1_\eta.
\]
则其分歧值集合恰为
\[
\operatorname{Br}(\eta)=\{0,\infty\}\cup\{\eta\in\overline K:\ \mathscr D(\eta)=0\},
\]
其中 \(\mathscr D\) 如 \eqref{eq:conclusion-leyang-eta-cubic-discriminant-dagger}。
并且在 \(\QQ(\sqrt7)\) 上有分解
\[
\mathscr D(\eta)
=2304\bigl(\eta^2-t_+\eta+1\bigr)\bigl(\eta^2-t_-\eta+1\bigr),
\qquad
t_\pm=\frac{139}{72}\pm\frac{7}{144}\sqrt7.
\]
从而除 \(0,\infty\) 外，四个有限分歧值恰为
\(
\eta^2-t_\pm\eta+1=0
\)
的全部根，并在 \(\eta\mapsto\eta^{-1}\) 下成对互逆。
\end{theorem}
\begin{proof}
映射 \(\eta\) 的几何分歧值对应扩张 \(K(H_{\mathrm{LY}})/K(\eta)\) 的分歧处，
即 \eqref{eq:conclusion-leyang-eta-cubic-star} 特化后出现重根的参数值。
由判别式判据与 \eqref{eq:conclusion-leyang-eta-cubic-discriminant-dagger}，
有限分歧值由 \(\eta=0\) 或 \(\mathscr D(\eta)=0\) 刻画。
又由
\(
\div(\eta)=3(Q_--Q_+)
\)
（定理 \ref{thm:xi-leyang-norm-one-unit-explicit-3torsion}），
\(\eta=0,\infty\) 分别对应全分歧点 \(Q_-,Q_+\)，故 \(0,\infty\in\operatorname{Br}(\eta)\)。
遂得分歧值描述。

将 \(\mathscr D\) 直接在 \(\QQ(\sqrt7)\) 上因式分解即得所示二次分解与 \(t_\pm\)。
每个二次因子常数项为 \(1\)，故根互为倒数。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\eta$}{eta} 分歧四次的二次脊柱椭圆曲线与 \texorpdfstring{$j$}{j} 不变量分解]\label{thm:conclusion-leyang-eta-discriminant-elliptic-j-factorization}
定义曲线
\[
E_\Delta:\ v^2=\mathscr D(\eta)
=2304\eta^4-8896\eta^3+13157\eta^2-8896\eta+2304.
\]
则 \(E_\Delta\) 的射影正规化为属 \(1\) 光滑曲线，并可取 \(\QQ\)-Weierstrass 模型
\[
E_\Delta\cong
\bigl\{Y^2=X^3+16437789X+40904021790\bigr\}.
\]
其 \(j\)-不变量为
\[
j(E_\Delta)
=\frac{225651856947733943}{462633485561856}
=\frac{(608807)^3}{2^{12}3^3 7^6 31^2 37}
=\frac{(419\cdot1453)^3}{2^{12}3^3 7^6 31^2 37}.
\]
对应二元四次不变量可取
\[
(I,J)=(-608807,\,-1514963770),
\]
并满足
\[
4I^3-J^2=-2^{20}3^6 7^6 31^2 37.
\]
\end{theorem}
\begin{proof}
由 \(\gcd(\mathscr D,\mathscr D')=1\) 知四次多项式无重根，故 \(v^2=\mathscr D(\eta)\) 的射影模型光滑且属为 \(1\)。
对四次系数
\[
(a,b,c,d,e)=(2304,-8896,13157,-8896,2304),
\]
按经典公式
\[
I=12ae-3bd+c^2,\qquad
J=72ace+9bcd-27ad^2-27b^2e-2c^3
\]
得
\(
I=-608807,\ J=-1514963770
\)。
其 Jacobian 椭圆曲线可写为
\[
Y^2=X^3-27IX-27J
=X^3+16437789X+40904021790.
\]
再用
\[
j=1728\cdot\frac{4I^3}{4I^3-J^2}
\]
及整数分解即可得到所示 \(j(E_\Delta)\) 与
\(
4I^3-J^2
\)
的素因子结构。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\eta$}{eta} 的 \texorpdfstring{$S_3$}{S3} 伽罗瓦闭包、惯性型与子覆盖格]\label{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice}
令 \(\widetilde H_\eta\) 为仿射系统
\begin{equation}\label{eq:conclusion-leyang-eta-s3-galois-system}
\begin{cases}
64(\eta-1)^2y^3
+3(32\eta^2-73\eta+32)y^2
+3(16\eta^2-23\eta+16)y
+8(\eta-1)^2=0,\\[1mm]
v^2=\mathscr D(\eta),
\end{cases}
\end{equation}
在 \(K\) 上的射影正规化。则：
\begin{enumerate}
  \item \(K(\widetilde H_\eta)/K(\eta)\) 为 \(S_3\)-伽罗瓦扩张，故 \(\widetilde H_\eta\to\PP^1_\eta\) 为正规 \(S_3\)-覆盖；
  \item 惯性型为
  \[
  (3,3,2,2,2,2),
  \]
  即在 \(\eta=0,\infty\) 处惯性阶为 \(3\)，在 \(\mathscr D(\eta)=0\) 的四点处惯性阶为 \(2\)；
  \item 曲线属为
  \[
  g(\widetilde H_\eta)=5;
  \]
  \item 子覆盖格由 \(S_3\) 子群格完全决定：唯一二次中间商为
  \[
  \widetilde H_\eta/A_3\cong E_\Delta,
  \]
  任一阶 \(2\) 子群的商曲线均与 \(H_{\mathrm{LY}}\) 在 \(K\) 上同构。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-leyang-eta-degree3-complete-ramification}，\(\eta:H_{\mathrm{LY}}\to\PP^1\) 次数为 \(3\)，故
\(
[K(H_{\mathrm{LY}}):K(\eta)]=3
\)。
结合定理 \ref{thm:conclusion-leyang-eta-cubic-elimination-discriminant-palindromic}，
\eqref{eq:conclusion-leyang-eta-cubic-star} 给出该三次扩张的判别式
\[
\operatorname{Disc}_{y}=-3^9\eta^2\mathscr D(\eta).
\]
在 \(K=\QQ(\sqrt{-3})\) 中，\(-3^9\) 为平方；\(\eta^2\) 亦为平方。
故加入 \(v\) 满足 \(v^2=\mathscr D(\eta)\) 等价于加入 \(\sqrt{\operatorname{Disc}_{y}}\)（差一个平方因子），
从而得到三次多项式的分裂域；因此伽罗瓦群为 \(S_3\)，且二次子域由 \(A_3\) 不变子给出，即 \(K(E_\Delta)\)。

分歧与惯性阶由判别式零因子与 \(\div(\eta)=3(Q_--Q_+)\) 直接读出：\(\eta=0,\infty\) 对应三循环惯性，\(\mathscr D(\eta)=0\) 对应换位惯性。
对正规 \(S_3\)-覆盖应用 Hurwitz 公式
\[
2g(\widetilde H_\eta)-2
=|S_3|\cdot(-2)
+2\cdot |S_3|\!\left(1-\frac13\right)
+4\cdot |S_3|\!\left(1-\frac12\right)
=8,
\]
故 \(g(\widetilde H_\eta)=5\)。

最后由 Galois 对应：指数 \(2\) 正规子群 \(A_3\) 给出唯一二次商 \(E_\Delta\)；
三个阶 \(2\) 子群互共轭，商曲线互相 \(K\)-同构，且任取其一即回到原三次子域 \(K(H_{\mathrm{LY}})=K(\eta,y)\)。
证毕。
\end{proof}

\begin{proposition}[Jacobian 的 \texorpdfstring{$S_3$}{S3} 同源分裂]\label{prop:conclusion-leyang-eta-s3-jacobian-isogeny-decomposition}
在定理 \ref{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice} 的记号下，存在 \(K\)-同源
\[
\operatorname{Jac}(\widetilde H_\eta)\sim_K \operatorname{Jac}(H_{\mathrm{LY}})^2\times E_\Delta.
\]
\end{proposition}
\begin{proof}
对 \(G=S_3\) 的 Burnside 关系
\[
[G/1]-2[G/C_2]-[G/C_3]+2[G/G]=0
\]
通过 Kani--Rosen 提升为 Jacobian 同源关系：
\[
\operatorname{Jac}(\widetilde H_\eta)\times \operatorname{Jac}(\widetilde H_\eta/G)^2
\sim
\operatorname{Jac}(\widetilde H_\eta/C_2)^2\times \operatorname{Jac}(\widetilde H_\eta/C_3).
\]
由定理 \ref{thm:conclusion-leyang-eta-s3-galois-closure-genus5-lattice}，
\[
\widetilde H_\eta/G\cong \PP^1,\qquad
\widetilde H_\eta/C_2\cong H_{\mathrm{LY}},\qquad
\widetilde H_\eta/C_3\cong E_\Delta.
\]
且 \(\operatorname{Jac}(\PP^1)=0\)，故得结论。证毕。
\end{proof}

\begin{corollary}[良好约化处的局部 \texorpdfstring{$L$}{L} 因子平方型分解]\label{cor:conclusion-leyang-eta-local-lfactor-square-rigidity}
设 \(\mathfrak p\) 为 \(K\) 上使 \(\widetilde H_\eta\)、\(H_{\mathrm{LY}}\)、\(E_\Delta\) 同时良好约化的素理想。
记对应局部 \(L\)-多项式为
\[
P_{\widetilde H_\eta,\mathfrak p}(T),\qquad
P_{H_{\mathrm{LY}},\mathfrak p}(T),\qquad
P_{E_\Delta,\mathfrak p}(T)=1-a_{\mathfrak p}(E_\Delta)T+q_{\mathfrak p}T^2,\qquad
q_{\mathfrak p}:=\#\kappa(\mathfrak p).
\]
则有严格分解
\[
P_{\widetilde H_\eta,\mathfrak p}(T)
=
P_{H_{\mathrm{LY}},\mathfrak p}(T)^2\,
P_{E_\Delta,\mathfrak p}(T).
\]
并且 Frobenius 迹满足
\[
a_{\mathfrak p}(\widetilde H_\eta)
=2\,a_{\mathfrak p}(H_{\mathrm{LY}})+a_{\mathfrak p}(E_\Delta).
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:conclusion-leyang-eta-s3-jacobian-isogeny-decomposition}，
在良好约化处的 \(\ell\)-进 Tate 模上（取半单化）Frobenius 作用按直和分解。
因此 \(H^1_{\acute et}\) 的特征多项式乘法分解即得第一式；
迹加法由一次系数可加性立即得到。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$S_3$}{S3} Artin 迹在互素同余类上的统计盲性]\label{thm:conclusion-leyang-s3-artin-trace-congruence-blindness}
设 \(K/\QQ\) 为伽罗瓦群同构于 \(S_3\) 的有限扩张，并令
\[
\rho_{\mathrm{std}}:S_3\to \GL_2(\CC)
\]
为标准二维表示。
对任意未分歧素数 \(p\)，定义
\[
a_p:=\Tr\bigl(\rho_{\mathrm{std}}(\mathrm{Frob}_p)\bigr)\in\{2,0,-1\}.
\]
令 \(m\ge 1\) 满足 \(\gcd(m,\Disc(K))=1\)，并取任意既约剩余类 \(r\in(\ZZ/m\ZZ)^\times\)。
则在素数集合 \(\{p:\ p\nmid m\Disc(K),\ p\equiv r\ (\mathrm{mod}\ m)\}\) 上，\(a_p\) 的 Dirichlet 密度分布为
\[
\delta(a_p=2\mid p\equiv r)=\frac16,\qquad
\delta(a_p=0\mid p\equiv r)=\frac12,\qquad
\delta(a_p=-1\mid p\equiv r)=\frac13.
\]
因此，任意仅以 \(p\bmod m\) 为输入的分类器 \(g:(\ZZ/m\ZZ)^\times\to\{2,0,-1\}\) 的极限正确率至多为 \(1/2\)，极限错误率至少为 \(1/2\)。
\end{theorem}

\begin{proof}
由 \(\gcd(m,\Disc(K))=1\) 得 \(K\) 与 \(\QQ(\zeta_m)\) 线性无关，复合域的伽罗瓦群同构
\[
\Gal\bigl(K\QQ(\zeta_m)/\QQ\bigr)\cong S_3\times(\ZZ/m\ZZ)^\times.
\]
Chebotarev 密度定理给出 Frobenius 元在该直积群中的等分布。
在条件 \(p\equiv r\ (\mathrm{mod}\ m)\) 下，第二因子固定为 \(r\)，第一因子仍在 \(S_3\) 的共轭类上按类大小比例分布。
而 \(\rho_{\mathrm{std}}\) 的迹在三类上取值分别为：恒等元 \(2\)（类大小 \(1\)）、换位 \(0\)（类大小 \(3\)）、三循环 \(-1\)（类大小 \(2\)）。
故条件密度为 \((1/6,1/2,1/3)\)，且与 \(r\) 无关。

对任意分类器 \(g\)，其在给定 \(r\) 下的极限正确率至多为 \(\max\{1/6,1/2,1/3\}=1/2\)；
对 \(r\) 加权取平均得到总体极限正确率 \(\le 1/2\)，从而极限错误率 \(\ge 1/2\)。证毕。
\end{proof}
