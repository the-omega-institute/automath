\begin{theorem}[tower defect 的精确对偶范数]\label{thm:conclusion-pom-tower-defect-exact-tv-duality}
\leanverified{paper\_conclusion\_pom\_tower\_defect\_exact\_tv\_duality}
设
\[
Y\xrightarrow{g} Z\xrightarrow{h} W
\]
为有限集上的满射。对每个 \(w\in W\)，在纤维 \(h^{-1}(w)\) 上定义两种概率分布
\[
u_w(z):=\frac{1}{d_h(w)},
\qquad
\eta_w(z):=\frac{d_g(z)}{\sum_{z'\in h^{-1}(w)}d_g(z')},
\]
并记
\[
\beta_w:=u_w-\eta_w.
\]
对任意 \(\psi:Y\to\RR\)，若记
\[
\Psi:=(\mathsf A_g\psi)\big|_{h^{-1}(w)},
\]
则有
\[
\mathfrak T_{g,h}(\psi)(w)
=
\sum_{z\in h^{-1}(w)}\beta_w(z)\Psi(z).
\]
更强地，成立精确对偶公式
\[
\sup_{\substack{\Psi:h^{-1}(w)\to\RR\\
\operatorname{osc}_{h^{-1}(w)}(\Psi)\le 1}}
\bigl|\EE_{u_w}[\Psi]-\EE_{\eta_w}[\Psi]\bigr|
=
\operatorname{TV}(u_w,\eta_w).
\]
等价地，
\[
\sup_{\substack{\psi:Y\to\RR\\
\operatorname{osc}_{h^{-1}(w)}(\mathsf A_g\psi)\le 1}}
\bigl|\mathfrak T_{g,h}(\psi)(w)\bigr|
=
\operatorname{TV}(u_w,\eta_w).
\]
特别地，下列命题等价：
\begin{enumerate}
  \item 对一切 \(\psi:Y\to\RR\)，都有 \(\mathfrak T_{g,h}(\psi)(w)=0\)；
  \item \(u_w=\eta_w\)；
  \item 函数 \(d_g\) 在 \(h^{-1}(w)\) 上处处常值。
\end{enumerate}
\end{theorem}
\begin{proof}
由推论 \ref{cor:pom-tower-defect-pinsker-bound}，
\[
\mathfrak T_{g,h}(\psi)(w)=\EE_{u_w}[\Psi]-\EE_{\eta_w}[\Psi]
=
\sum_{z\in h^{-1}(w)}\beta_w(z)\Psi(z).
\]

对偶范数公式只需在有限集 \(h^{-1}(w)\) 上证明。由于
\[
\beta_w(h^{-1}(w))=0,
\]
向 \(\Psi\) 加任意常数不改变配对值，故
\[
\sup_{\operatorname{osc}\le 1}|\langle \beta_w,\Psi\rangle|
=
\sup_{0\le \Psi\le 1}|\langle \beta_w,\Psi\rangle|.
\]
而在有限集上，右端正是总变差的 Hahn 对偶表达：
\[
\sup_{0\le \Psi\le 1}|\langle \beta_w,\Psi\rangle|
=
\sup_{A\subseteq h^{-1}(w)}|\beta_w(A)|
=
\operatorname{TV}(u_w,\eta_w).
\]
再注意任意 \(\Psi:h^{-1}(w)\to\RR\) 都可取
\[
\psi(y):=
\begin{cases}
\Psi(g(y)), & g(y)\in h^{-1}(w),\\
0, & g(y)\notin h^{-1}(w),
\end{cases}
\]
从而在 \(h^{-1}(w)\) 上实现 \((\mathsf A_g\psi)=\Psi\)。于是第二个上确界公式随之成立。

最后，\(\operatorname{TV}(u_w,\eta_w)=0\iff u_w=\eta_w\)。而
\[
u_w(z)=\eta_w(z)
\iff
\frac{1}{d_h(w)}
=
\frac{d_g(z)}{\sum_{z'\in h^{-1}(w)}d_g(z')}
\qquad(\forall z\in h^{-1}(w)),
\]
这恰等价于 \(d_g\) 在 \(h^{-1}(w)\) 上处处常值。证毕。
\end{proof}

\begin{theorem}[局部 Beck--Chevalley 曲率即 size-bias 偏斜的 KL 势]\label{thm:conclusion-pom-local-curvature-is-kl-sizebias}
\leanverified{paper\_conclusion\_pom\_local\_curvature\_is\_kl\_sizebias}
在定理 \ref{thm:conclusion-pom-tower-defect-exact-tv-duality} 的设定下，对每个 \(w\in W\) 定义局部曲率
\[
\mathcal B_{g,h}(w)
:=
\log\!\Bigl(\frac1{d_h(w)}\sum_{z\in h^{-1}(w)}d_g(z)\Bigr)
-\frac1{d_h(w)}\sum_{z\in h^{-1}(w)}\log d_g(z).
\]
则有严格恒等式
\[
\mathcal B_{g,h}(w)=D_{\mathrm{KL}}(u_w\Vert \eta_w).
\]
因此
\[
\operatorname{TV}(u_w,\eta_w)\le \sqrt{\frac12\,\mathcal B_{g,h}(w)},
\]
以及
\[
\sup_{\substack{\psi:Y\to\RR\\
\operatorname{osc}_{h^{-1}(w)}(\mathsf A_g\psi)\le 1}}
\bigl|\mathfrak T_{g,h}(\psi)(w)\bigr|
\le
\sqrt{\frac12\,\mathcal B_{g,h}(w)}.
\]
进一步，对 \(t\in[0,1]\) 定义 escort 权重
\[
p_{w,t}(z):=
\frac{d_g(z)^t}{\sum_{z'\in h^{-1}(w)}d_g(z')^t},
\]
则有能量表示
\[
\mathcal B_{g,h}(w)
=
\int_0^1(1-t)\,
\Var_{Z\sim p_{w,t}}\!\bigl(\log d_g(Z)\bigr)\,\dd t.
\]
\end{theorem}
\begin{proof}
直接计算
\[
\begin{aligned}
D_{\mathrm{KL}}(u_w\Vert \eta_w)
&=
\sum_{z\in h^{-1}(w)}
\frac{1}{d_h(w)}
\log\frac{1/d_h(w)}{d_g(z)/\sum_{z'}d_g(z')}\\
&=
\log\!\Bigl(\frac1{d_h(w)}\sum_{z'\in h^{-1}(w)}d_g(z')\Bigr)
-\frac1{d_h(w)}\sum_{z\in h^{-1}(w)}\log d_g(z),
\end{aligned}
\]
即得 \(\mathcal B_{g,h}(w)=D_{\mathrm{KL}}(u_w\Vert \eta_w)\)。

再由 Pinsker 不等式
\[
\operatorname{TV}(u_w,\eta_w)\le \sqrt{\frac12\,D_{\mathrm{KL}}(u_w\Vert \eta_w)}
\]
与定理 \ref{thm:conclusion-pom-tower-defect-exact-tv-duality} 的精确对偶公式，得到前两条不等式。

最后，把命题 \ref{prop:pom-bc-amgm-defect-escort-variance-energy} 应用于满射对
\(
g:Y\twoheadrightarrow Z
\)
与
\(
h:Z\twoheadrightarrow W
\)
即可得到 escort 方差能量表示。证毕。
\end{proof}

\begin{theorem}[全局 tower defect 半范数的精确闭式与平方根曲率界]\label{thm:conclusion-pom-global-tower-defect-locosc-exact}
\leanverified{paper\_conclusion\_pom\_global\_tower\_defect\_locosc\_exact}
设 \(\nu\in\Delta(W)\)。定义全局 tower defect 半范数
\[
\|\mathcal T_{g,h,\nu}\|_{\mathrm{loc\mbox{-}osc}}
:=
\sup_{\substack{\Psi:Z\to\RR\\
\operatorname{osc}_{h^{-1}(w)}(\Psi)\le 1,\ \forall w\in W}}
\left|
\sum_{w\in W}\nu(w)\bigl(\EE_{u_w}[\Psi]-\EE_{\eta_w}[\Psi]\bigr)
\right|.
\]
则有精确闭式
\[
\|\mathcal T_{g,h,\nu}\|_{\mathrm{loc\mbox{-}osc}}
=
\sum_{w\in W}\nu(w)\,\operatorname{TV}(u_w,\eta_w).
\]
若进一步记
\[
\Omega_{\mathrm{KL}}(g,h;\nu)
:=
D_{\mathrm{KL}}(L_gL_h\nu\Vert L_{h\circ g}\nu),
\]
则
\[
\|\mathcal T_{g,h,\nu}\|_{\mathrm{loc\mbox{-}osc}}
\le
\sqrt{\frac12\,\Omega_{\mathrm{KL}}(g,h;\nu)}.
\]
\end{theorem}
\begin{proof}
对任意 \(\Psi:Z\to\RR\)，由定理 \ref{thm:conclusion-pom-tower-defect-exact-tv-duality} 在每条 \(h\)-纤维上的精确局部估计，
\[
\bigl|\EE_{u_w}[\Psi]-\EE_{\eta_w}[\Psi]\bigr|
\le
\operatorname{TV}(u_w,\eta_w).
\]
故
\[
\left|
\sum_{w\in W}\nu(w)\bigl(\EE_{u_w}[\Psi]-\EE_{\eta_w}[\Psi]\bigr)
\right|
\le
\sum_{w\in W}\nu(w)\operatorname{TV}(u_w,\eta_w),
\]
从而得到上界。

反向取等由纤维分离性给出。对每个 \(w\)，取 \(A_w\subseteq h^{-1}(w)\) 为 \(\beta_w=u_w-\eta_w\) 的 Hahn 正部分支撑，使
\[
\beta_w(A_w)=\operatorname{TV}(u_w,\eta_w).
\]
定义全局函数
\[
\Psi(z):=\mathbf 1_{A_{h(z)}}(z).
\]
则在每条 \(h^{-1}(w)\) 上均有 \(\operatorname{osc}\le 1\)，并且
\[
\EE_{u_w}[\Psi]-\EE_{\eta_w}[\Psi]
=
\beta_w(A_w)
=
\operatorname{TV}(u_w,\eta_w).
\]
于是上界可达，故精确闭式成立。

再由定理 \ref{thm:conclusion-pom-local-curvature-is-kl-sizebias}，
\[
\operatorname{TV}(u_w,\eta_w)\le \sqrt{\frac12\,\mathcal B_{g,h}(w)}.
\]
因此
\[
\|\mathcal T_{g,h,\nu}\|_{\mathrm{loc\mbox{-}osc}}
\le
\sum_{w\in W}\nu(w)\sqrt{\frac12\,\mathcal B_{g,h}(w)}
\le
\sqrt{\frac12\sum_{w\in W}\nu(w)\mathcal B_{g,h}(w)},
\]
其中第二步由 Jensen 不等式给出。最后，由定理 \ref{thm:pom-bc-information-stokes-coboundary}，
\[
\sum_{w\in W}\nu(w)\mathcal B_{g,h}(w)=\Omega_{\mathrm{KL}}(g,h;\nu),
\]
即得结论。证毕。
\end{proof}

\begin{corollary}[Beck--Chevalley 曲率与 tower defect 的同核性]\label{cor:conclusion-pom-curvature-and-tower-defect-same-kernel}
\leanverified{paper\_conclusion\_pom\_curvature\_and\_tower\_defect\_same\_kernel}
设 \(\nu\in\Delta(W)\) 具满支撑。则下列命题等价：
\begin{enumerate}
  \item \(d_g\) 在每条 \(h\)-纤维 \(h^{-1}(w)\) 上处处常值；
  \item \(\Omega_{\mathrm{KL}}(g,h;\nu)=0\)；
  \item 对任意 \(\nu'\in\Delta(W)\)，都有 \(\Omega_{\mathrm{KL}}(g,h;\nu')=0\)；
  \item 对任意 \(\psi:Y\to\RR\) 与任意 \(w\in W\)，都有 \(\mathfrak T_{g,h}(\psi)(w)=0\)；
  \item \(\|\mathcal T_{g,h,\nu}\|_{\mathrm{loc\mbox{-}osc}}=0\)。
\end{enumerate}
\end{corollary}
\begin{proof}
\(\textup{(1)}\Rightarrow\textup{(3)}\) 与 \(\textup{(1)}\Rightarrow\textup{(4)}\) 都是直接的：若 \(d_g\) 在每条 \(h\)-纤维上常值，则 \(u_w=\eta_w\)，故各局部曲率与各局部 tower defect 全部为零。

\(\textup{(3)}\Rightarrow\textup{(2)}\) 显然。

对 \(\textup{(2)}\Rightarrow\textup{(1)}\)，由定理 \ref{thm:pom-bc-information-stokes-coboundary}，
\[
\Omega_{\mathrm{KL}}(g,h;\nu)=\sum_{w\in W}\nu(w)\mathcal B_{g,h}(w).
\]
由于 \(\nu(w)>0\) 且 \(\mathcal B_{g,h}(w)\ge 0\)，故 \(\Omega_{\mathrm{KL}}(g,h;\nu)=0\) 迫使 \(\mathcal B_{g,h}(w)=0\) 对所有 \(w\) 成立。再由定理 \ref{thm:conclusion-pom-local-curvature-is-kl-sizebias}，
\[
\mathcal B_{g,h}(w)=D_{\mathrm{KL}}(u_w\Vert \eta_w),
\]
于是 \(u_w=\eta_w\)，从而 \(d_g\) 在 \(h^{-1}(w)\) 上常值。

\(\textup{(4)}\Rightarrow\textup{(1)}\) 由定理 \ref{thm:conclusion-pom-tower-defect-exact-tv-duality} 立即得出：若一切 \(\psi\) 上 tower defect 全为零，则 \(\operatorname{TV}(u_w,\eta_w)=0\) 对所有 \(w\) 成立。

\(\textup{(5)}\Rightarrow\textup{(1)}\) 则由定理 \ref{thm:conclusion-pom-global-tower-defect-locosc-exact} 的精确闭式
\[
\|\mathcal T_{g,h,\nu}\|_{\mathrm{loc\mbox{-}osc}}
=
\sum_{w\in W}\nu(w)\operatorname{TV}(u_w,\eta_w)
\]
与 \(\nu\) 满支撑性推出。证毕。
\end{proof}

\begin{theorem}[size-bias 单调律]\label{thm:conclusion-pom-size-bias-monotonicity}
\leanverified{paper\_conclusion\_pom\_size\_bias\_monotonicity}
沿用定理 \ref{thm:pom-tower-defect-covariance-law} 的记号，固定 \(w\in W\)。若存在单调函数 \(\Phi\) 使
\[
\Psi(z)=\Phi(d_g(z))
\qquad (z\in h^{-1}(w)),
\]
则有：
\begin{enumerate}
  \item 若 \(\Phi\) 非减，则 \(\mathfrak T_{g,h}(\psi)(w)\le 0\)；
  \item 若 \(\Phi\) 非增，则 \(\mathfrak T_{g,h}(\psi)(w)\ge 0\)。
\end{enumerate}
若 \(d_g\) 在 \(h^{-1}(w)\) 上并非常值，且 \(\Phi\) 在所出现的度值上严格单调，则上述不等号严格。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-tower-defect-covariance-law}，
\[
\mathfrak T_{g,h}(\psi)(w)=-\frac{\Cov(N,\Psi)}{\EE[N]},
\qquad
N:=d_g(Z),
\]
其中 \(Z\) 在 \(h^{-1}(w)\) 上均匀分布。若 \(\Psi=\Phi(N)\)，再取 \(N'\) 为 \(N\) 的独立同分布拷贝，则
\[
2\Cov(N,\Phi(N))
=
\EE\bigl[(N-N')(\Phi(N)-\Phi(N'))\bigr].
\]
当 \(\Phi\) 非减时，被积函数几乎处处非负，故 \(\Cov(N,\Psi)\ge 0\)，于是 \(\mathfrak T_{g,h}(\psi)(w)\le 0\)。当 \(\Phi\) 非增时，同理得到 \(\Cov(N,\Psi)\le 0\)，故 \(\mathfrak T_{g,h}(\psi)(w)\ge 0\)。

若 \(N\) 非常值且 \(\Phi\) 严格单调，则上式被积函数在正概率集合上严格正或严格负，故协方差严格非零，于是相应不等号严格。证毕。
\end{proof}

\begin{theorem}[三阶余圈的符号与曲率包络]\label{thm:conclusion-pom-third-obstruction-curvature-envelope}
设
\[
X\xrightarrow{f}Y\xrightarrow{g}Z\xrightarrow{h}W
\]
为可复合满射三元组。定义
\[
\Theta_{f,g}(z):=(\mathsf A_g\log d_f)(z),
\qquad
\Xi_{f,g,h}(w):=\mathfrak T_{g,h}(\log d_f)(w).
\]
则对每个 \(w\in W\)，有
\[
|\Xi_{f,g,h}(w)|
\le
\operatorname{osc}_{h^{-1}(w)}(\Theta_{f,g})\,
\operatorname{TV}(u_w,\eta_w)
\le
\operatorname{osc}_{h^{-1}(w)}(\Theta_{f,g})\,
\sqrt{\frac12\,\mathcal B_{g,h}(w)}.
\]
若进一步存在单调函数 \(\Phi_w\) 使
\[
\Theta_{f,g}(z)=\Phi_w(d_g(z))
\qquad (z\in h^{-1}(w)),
\]
则：
\begin{enumerate}
  \item 当 \(\Phi_w\) 非减时，\(\Xi_{f,g,h}(w)\le 0\)；
  \item 当 \(\Phi_w\) 非增时，\(\Xi_{f,g,h}(w)\ge 0\)。
\end{enumerate}
若 \(d_g\) 在 \(h^{-1}(w)\) 上并非常值且 \(\Phi_w\) 严格单调，则不等号严格。
\end{theorem}
\begin{proof}
由定义，
\[
\Xi_{f,g,h}(w)=\mathfrak T_{g,h}(\log d_f)(w)=\EE_{u_w}[\Theta_{f,g}]-\EE_{\eta_w}[\Theta_{f,g}].
\]
故由定理 \ref{thm:conclusion-pom-tower-defect-exact-tv-duality}，
\[
|\Xi_{f,g,h}(w)|
\le
\operatorname{osc}_{h^{-1}(w)}(\Theta_{f,g})\,
\operatorname{TV}(u_w,\eta_w).
\]
再由定理 \ref{thm:conclusion-pom-local-curvature-is-kl-sizebias} 的 Pinsker 型界，即得第二个不等式。

若 \(\Theta_{f,g}=\Phi_w(d_g)\) 在 \(h^{-1}(w)\) 上成立，则 \(\Xi_{f,g,h}(w)\) 正是定理 \ref{thm:conclusion-pom-size-bias-monotonicity} 中 \(\Psi=\Theta_{f,g}\) 的专门化，故符号与严格性结论直接跟随。证毕。
\end{proof}

\begin{theorem}[复合均匀提升的双障碍完全分解]\label{thm:conclusion-pom-composite-uniform-lift-dual-obstruction}
对任意有限满射链
\[
X_0\xrightarrow{f_1}X_1\xrightarrow{f_2}\cdots\xrightarrow{f_k}X_k
\]
与任意外层观测通道 \(Y\xrightarrow{g}Z\xrightarrow{h}W\)，复合均匀提升的失真分裂为两类彼此独立的障碍：分布侧由逐层 AM--GM 缺陷 \(\alpha_i\) 控制，观测侧由 tower defect
\[
\mathfrak T_{g,h}(\psi)(w)=-\frac{\Cov(N,\Psi)}{\EE[N]}
\]
控制。更精确地，
\[
D_{\mathrm{KL}}(\mu_{\mathrm{seq}}\|\mu_{\mathrm{tot}})
=
\sum_{i=1}^{k-1}\EE_{\nu^{(i+1)}}[\alpha_i(X_{i+1})],
\]
而任意外层纤维上的观测偏差都恰等于上述协方差项。因而，复合均匀提升的严格函子性并不等价于单一缺陷为零，而等价于这两套障碍同时闭合：分布账本无熵缺陷，且观测账本无协方差扭结。
\end{theorem}
\begin{proof}
链式分布偏差公式正是定理 \ref{thm:pom-bc-chain-defect-equals-escort-curvature-sum}；其零缺陷判据由推论 \ref{cor:pom-bc-zero-defect-rigidity-parenthesization} 给出，说明顺序提升与总提升的差异完全由逐层 \(\alpha_i\) 累加而成。另一方面，观测层的复合失败由定理 \ref{thm:pom-tower-defect-covariance-law} 精确写成 \(-\Cov(N,\Psi)/\EE[N]\)。前者是分布侧的严格化条件，后者是观测侧的严格化条件；二者分别作用于不同接口，彼此不能互相推出。故只有当逐层 \(\alpha_i\) 与外层协方差项同时消失时，复合均匀提升才在分布与观测两侧一起严格闭合。证毕。
\end{proof}

对每个可复合对
\[
Y\xrightarrow{g}Z\xrightarrow{h}W
\]
及每个 \(w\in W\)，考虑局部 tower defect 的 Hahn 优化问题，并记
\[
\mathscr H_w

\Bigl\{
\Psi:h^{-1}(w)\to[0,1]:
\langle u_w-\eta_w,\Psi\rangle=\operatorname{TV}(u_w,\eta_w)
\Bigr\}.
\]

\[
\Theta_{f,g}\big|_{h^{-1}(w)}

在仿射变换后属于 \(\mathscr H_w\)。

\end{conjecture}
