\subsection{屏幕纤维熵、闭包下降链与收缩拟阵补全的同一化}\label{subsec:conclusion-screen-fiber-entropy-closure-contraction-geodesic}
沿用定理 \ref{thm:spg-screen-kernel-connected-components}、推论 \ref{cor:spg-partial-boundary-min-audit-cost-kernel-rank}、定理 \ref{thm:conclusion-screen-visible-external-audit-conservation} 与定理 \ref{thm:conclusion-screen-audit-gap-arithmetic-lipschitz} 的记号。前述链条已经把部分屏幕的核秩、最小审计补全成本、边界行拟阵的余秩以及边界算术超立方体上的 \(1\)-Lipschitz 缺口刚性完全对齐。把这些接口与收缩拟阵的 exactization 几何继续合并之后，还可得到如下更紧的闭环：模 \(p\) 纤维熵指数恰等于审计余秩；单面增广的缺口下降完全由拟阵闭包控制；可见面数、隐藏熵与秩亏之间存在严格守恒律；全部最小补全则由任一独立核的收缩拟阵基族完全参数化。

\begin{theorem}[部分屏幕模 \texorpdfstring{$p$}{p} 纤维熵指数与审计余秩的严格同一]\label{thm:conclusion-screen-modp-fiber-entropy-audit-corank}
固定
\[
M_{m,n}:=\bigl(\overrightarrow{\mathcal F_m^{n-1}},r\bigr),
\qquad
\rho_{m,n}:=2^{mn}.
\]
对任意部分屏幕
\[
S\subseteq \overrightarrow{\mathcal F_m^{n-1}}
\]
与任意素数 \(p\)，记
\[
f_{S,p}:C_n(\mathcal Q_m;\mathbb F_p)\longrightarrow \mathbb F_p^{\,S}
\]
为部分边界观测算子的模 \(p\) 版本。若
\[
y\in \operatorname{im}(f_{S,p}),
\qquad
\mathcal F_{S,p}(y):=f_{S,p}^{-1}(y),
\]
则有精确公式
\[
|\mathcal F_{S,p}(y)|=p^{\,a(S)}.
\]
等价地，
\[
\log_p |\mathcal F_{S,p}(y)|=a(S)=\rho_{m,n}-r(S).
\]
\leanverified{paper\_conclusion\_screen\_modp\_fiber\_entropy\_audit\_corank}
\end{theorem}
\begin{proof}
定理 \ref{thm:spg-screen-kernel-connected-components} 的证明把部分屏幕算子识别为屏幕图 \(\Gamma_S\) 的有向关联矩阵；因此，对任意域 \(\mathbb F_p\)，其秩都等于
\[
|V(\Gamma_S)|-c(\Gamma_S),
\]
从而
\[
\dim_{\mathbb F_p}\ker f_{S,p}=c(\Gamma_S)-1.
\]
于是每个非空纤维 \(\mathcal F_{S,p}(y)\) 都是 \(\ker f_{S,p}\) 的一个仿射平移，故
\[
|\mathcal F_{S,p}(y)|=p^{\,c(\Gamma_S)-1}.
\]
另一方面，由推论 \ref{cor:spg-partial-boundary-min-audit-cost-kernel-rank} 与定理 \ref{thm:conclusion-screen-visible-external-audit-conservation}，
\[
a(S)=c(\Gamma_S)-1=\rho_{m,n}-r(S).
\]
代回即得
\[
|\mathcal F_{S,p}(y)|=p^{\,a(S)},
\qquad
\log_p |\mathcal F_{S,p}(y)|=a(S)=\rho_{m,n}-r(S).
\]
证毕。
\end{proof}

\begin{theorem}[单面增广的闭包判据与严格缺口阶梯]\label{thm:conclusion-screen-single-face-closure-entropy-staircase}
\leanverified{paper\_conclusion\_screen\_single\_face\_closure\_entropy\_staircase}
对任意部分屏幕
\[
S\subseteq \overrightarrow{\mathcal F_m^{n-1}}
\]
与任意未观测面
\[
e\in \overrightarrow{\mathcal F_m^{n-1}}\setminus S,
\]
都有
\[
a(S\cup\{e\})\in \{a(S),a(S)-1\}.
\]
并且以下条件等价：
\[
a(S\cup\{e\})=a(S)-1
\quad\Longleftrightarrow\quad
e\notin \operatorname{cl}_{M_{m,n}}(S).
\]
特别地，只要 \(a(S)>0\)，便存在一条严格下降链
\[
S=S_0\subsetneq S_1\subsetneq \cdots \subsetneq S_{a(S)}
\]
满足
\[
|S_j|=|S|+j,
\qquad
a(S_j)=a(S)-j
\qquad
(0\le j\le a(S)),
\]
且终点 \(S_{a(S)}\) 为精确屏幕。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-screen-visible-external-audit-conservation}，
\[
a(S)=\rho_{m,n}-r(S).
\]
故
\[
a(S\cup\{e\})-a(S)=r(S)-r(S\cup\{e\}).
\]
由于拟阵秩对单元素扩张的增量只能取 \(0\) 或 \(1\)，遂得
\[
a(S\cup\{e\})\in \{a(S),a(S)-1\}.
\]
再由拟阵闭包定义，
\[
r(S\cup\{e\})=r(S)+1
\quad\Longleftrightarrow\quad
e\notin \operatorname{cl}_{M_{m,n}}(S),
\]
故等价性成立。这与定理 \ref{thm:conclusion-screen-rank-gain-closure-diminishing-returns} 的 \(0\text{--}1\) 边际律完全一致。

若 \(a(S)>0\)，则 \(r(S)<\rho_{m,n}\)，故 \(S\) 非张成，从而存在某个
\[
e_0\notin \operatorname{cl}_{M_{m,n}}(S).
\]
于是
\[
a(S\cup\{e_0\})=a(S)-1.
\]
递归重复即可得到所述严格下降链；当 \(j=a(S)\) 时，
\[
a(S_{a(S)})=0,
\]
故终点为精确屏幕。该构造亦可视为定理 \ref{thm:conclusion-screen-deficit-unit-descent-chain} 的闭包判据版本。证毕。
\end{proof}

\begin{theorem}[可见面数、隐藏 \texorpdfstring{$p$}{p} 进熵与秩亏的严格守恒律]\label{thm:conclusion-screen-visible-hidden-padic-entropy-conservation}
定义部分屏幕的冗余度
\[
\nu(S):=|S|-r(S)\ge 0.
\]
则对任意素数 \(p\) 与任意非空纤维 \(\mathcal F_{S,p}(y)\)，都有严格恒等式
\[
|S|+\log_p |\mathcal F_{S,p}(y)|
=
\rho_{m,n}+\nu(S).
\]
因此始终成立下界
\[
|S|+\log_p |\mathcal F_{S,p}(y)|\ge \rho_{m,n},
\]
且取等当且仅当 \(S\) 在边界行拟阵 \(M_{m,n}\) 中独立。若再要求纤维退化为单点，则取等当且仅当 \(S\) 是一组极小精确屏幕。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-screen-modp-fiber-entropy-audit-corank}，
\[
\log_p |\mathcal F_{S,p}(y)|=a(S)=\rho_{m,n}-r(S).
\]
于是
\[
|S|+\log_p |\mathcal F_{S,p}(y)|
=
|S|+\rho_{m,n}-r(S)
=
\rho_{m,n}+\bigl(|S|-r(S)\bigr)
=
\rho_{m,n}+\nu(S).
\]
由拟阵秩公理 \(r(S)\le |S|\)，即得 \(\nu(S)\ge 0\)，故下界成立。

取等当且仅当 \(\nu(S)=0\)，亦即
\[
r(S)=|S|,
\]
这正是 \(S\) 独立的定义。若进一步要求纤维为单点，则
\[
\log_p |\mathcal F_{S,p}(y)|=0
\iff
a(S)=0.
\]
与独立性合并可得
\[
r(S)=|S|=\rho_{m,n},
\]
故 \(S\) 同时独立且张成，即为边界行拟阵的一组基。由定理 \ref{thm:spg-exact-screen-minima-are-matroid-bases} 与定理 \ref{thm:conclusion-screen-zero-slack-matroid-basis-criterion}，这恰是全部极小精确屏幕。证毕。
\end{proof}

\begin{theorem}[最小离散补全面数、连续补账维与隐藏熵指数的三重同一]\label{thm:conclusion-screen-discrete-continuous-completion-entropy-identification}
定义离散补全成本
\[
\kappa_{\mathrm{vis}}(S)
:=
\min\Bigl\{
|T|:\ 
T\subseteq \overrightarrow{\mathcal F_m^{n-1}}\setminus S,\
\ker f_{S\cup T}=0
\Bigr\}.
\]
则对任意部分屏幕 \(S\) 与任意素数 \(p\)，都有
\[
\kappa_{\mathrm{vis}}(S)=a(S)=\log_p |\mathcal F_{S,p}(y)|
\]
对每个 \(y\in \operatorname{im}(f_{S,p})\) 成立。换言之，杀死全部可见歧义所需的最少新增面数，恰等于最小连续审计补账维，也恰等于任一非空模 \(p\) 纤维的对数基数。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-screen-exactization-independent-kernel} 在无权情形 \(c\equiv 1\) 下的公式，
\[
\kappa_{\mathrm{vis}}(S)=\rho_{m,n}-r(S).
\]
再由定理 \ref{thm:conclusion-screen-visible-external-audit-conservation}，
\[
\rho_{m,n}-r(S)=a(S).
\]
最后，定理 \ref{thm:conclusion-screen-modp-fiber-entropy-audit-corank} 已给出
\[
\log_p |\mathcal F_{S,p}(y)|=a(S).
\]
合并三式即得结论。证毕。
\end{proof}

\begin{theorem}[最小补全族的独立核分类与闭包不变性]\label{thm:conclusion-screen-minimal-completion-family-independent-kernel}
给定部分屏幕
\[
S\subseteq \overrightarrow{\mathcal F_m^{n-1}},
\]
取 \(I\subseteq S\) 为任一极大独立子集，于是
\[
|I|=r(S),
\qquad
\operatorname{cl}_{M_{m,n}}(I)=\operatorname{cl}_{M_{m,n}}(S).
\]
定义最小补全族
\[
\mathcal C_{\min}(S)
:=
\Bigl\{
T\subseteq \overrightarrow{\mathcal F_m^{n-1}}\setminus S:\ 
\ker f_{S\cup T}=0,\
|T|=\kappa_{\mathrm{vis}}(S)
\Bigr\}.
\]
则有精确刻画
\[
\mathcal C_{\min}(S)
=
\Bigl\{
B\setminus I:\ 
B\in \mathcal B(M_{m,n}),\
I\subseteq B
\Bigr\}.
\]
特别地，映射
\[
T\longmapsto I\cup T
\]
给出 \(\mathcal C_{\min}(S)\) 与收缩拟阵 \(M_{m,n}/I\) 的基族之间的自然双射。因此 \(\mathcal C_{\min}(S)\) 只依赖于 \(\operatorname{cl}_{M_{m,n}}(S)\)，与 \(I\) 的具体选取无关。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-screen-discrete-continuous-completion-entropy-identification}，
\[
|T|=\kappa_{\mathrm{vis}}(S)=a(S)=\rho_{m,n}-r(S)=\rho_{m,n}-|I|
\]
对任意 \(T\in\mathcal C_{\min}(S)\) 成立。又由
\[
\operatorname{cl}_{M_{m,n}}(I)=\operatorname{cl}_{M_{m,n}}(S),
\]
以及定理 \ref{thm:conclusion-screen-exactization-independent-kernel}，可知 \(S\) 与 \(I\) 具有完全相同的 exactization 可行补全族。因此
\[
\ker f_{S\cup T}=0
\quad\Longleftrightarrow\quad
\ker f_{I\cup T}=0
\quad\Longleftrightarrow\quad
r(I\cup T)=\rho_{m,n}.
\]
而
\[
|I\cup T|=|I|+|T|=\rho_{m,n},
\]
故 \(I\cup T\) 必为 \(M_{m,n}\) 的一组基。于是每个 \(T\in\mathcal C_{\min}(S)\) 都可写成
\[
T=B\setminus I
\]
其中 \(B:=I\cup T\in \mathcal B(M_{m,n})\) 且 \(I\subseteq B\)。

反之，若 \(B\in \mathcal B(M_{m,n})\) 且 \(I\subseteq B\)，令
\[
T:=B\setminus I.
\]
则
\[
|T|=\rho_{m,n}-|I|=\rho_{m,n}-r(S)=\kappa_{\mathrm{vis}}(S).
\]
又因为 \(S\setminus I\subseteq \operatorname{cl}(I)\)，这些元素在收缩拟阵 \(M_{m,n}/I\) 中都是 loop，而 \(B\setminus I\) 是 \(M_{m,n}/I\) 的一组基，故自动有
\[
T=B\setminus I\subseteq \overrightarrow{\mathcal F_m^{n-1}}\setminus S.
\]
并且
\[
S\cup T\supseteq I\cup T=B,
\]
所以 \(S\cup T\) 张成全秩，亦即 \(\ker f_{S\cup T}=0\)。故 \(T\in \mathcal C_{\min}(S)\)。

这就证明了所述刻画。按收缩拟阵的基族定义，
\[
\mathcal B(M_{m,n}/I)
=
\{B\setminus I:\ B\in \mathcal B(M_{m,n}),\ I\subseteq B\},
\]
故 \(\mathcal C_{\min}(S)\) 与 \(\mathcal B(M_{m,n}/I)\) 自然双射。最后，不同极大独立子集 \(I\subseteq S\) 都生成同一 flat \(\operatorname{cl}(S)\)，其收缩拟阵基族因而规范对应，于是 \(\mathcal C_{\min}(S)\) 只依赖于 \(\operatorname{cl}(S)\)。证毕。
\end{proof}

\begin{corollary}[极小精确屏幕的测地交换性]\label{cor:conclusion-screen-minimal-exact-bases-geodesic-exchange}
设
\[
B,B'\in \mathcal B(M_{m,n})
\]
为任意两组极小精确屏幕。则存在一条链
\[
B=B_0,\ B_1,\ \dots,\ B_t=B'
\]
满足每个 \(B_j\) 仍为极小精确屏幕，且
\[
|B_{j+1}\triangle B_j|=2
\qquad
(0\le j<t),
\]
并且可取
\[
t=|B\setminus B'|=\frac12|B\triangle B'|.
\]
换言之，极小精确屏幕在边界算术超立方体中构成一张以单次基交换为边的测地连通图。
\end{corollary}
\begin{proof}
由定理 \ref{thm:spg-exact-screen-minima-are-matroid-bases}，极小精确屏幕恰为边界行拟阵 \(M_{m,n}\) 的全部基。于是结论正是定理 \ref{thm:conclusion-screen-basis-exchange-graph-geodesic-rigidity} 的直接特例：其交换距离公式给出
\[
d_{\Gamma_{\mathrm{ex}}}(B,B')
=
\frac{|B\triangle B'|}{2}
=
|B\setminus B'|,
\]
并实现于一条每步仅作一次换入换出的基交换链。证毕。
\end{proof}

\noindent
因此，在 dyadic 屏幕语义中，部分可见性的四个表面上彼此不同的量事实上已经塌缩为同一不变量：
\[
\rho_{m,n}-r(S)
=
a(S)
=
\kappa_{\mathrm{vis}}(S)
=
\log_p |\mathcal F_{S,p}(y)|.
\]
其中，单面增广的全部边际行为由拟阵闭包完全控制，而所有最小补全则由任一独立核 \(I\subseteq S\) 的收缩拟阵基族精确参数化。于是，屏幕余秩、审计圆维、模 \(p\) 纤维熵指数与离散补全面数不再是四条平行读法，而成为同一收缩拟阵几何在不同语义层中的同一投影；极小精确屏幕之间的测地交换链则给出这一几何在边界算术超立方体中的最短连通骨架。
