\subsection{有限状态 dyadic 通量的极点格、加权探针的几何消元与 primorial 主芽的纯算术恢复}\label{subsec:conclusion-dyadic-flux-polegrid-weighted-probe-primorial-germ}
沿用命题 \ref{prop:spg-regular-language-stokes-dyadic-zeta-rationality}、命题 \ref{prop:spg-weighted-stokes-flux-vanishing-dimension-shift}、定理 \ref{thm:conclusion-serrin-wulff-dyadic-codim-law} 与定理 \ref{thm:conclusion-primorial-bodycode-pole-residue-inversion} 的记号。前述结果已经分别给出：有限状态 dyadic 占据族的 Stokes--dyadic 通量 \(\zeta\) 具有严格分式延拓；primitive 正则家族的主极点与主留数由 Perron 主模控制；按距离消失的加权探针把 Minkowski 余维平移为通量指数；而 primorial 体码又把主极点位置与主留数压缩为双对数一、二阶项。将这四个接口并读后，还可进一步抽出如下统一链条。

\begin{theorem}[有限状态 dyadic 通量 \(\zeta\) 的极点落在有限条垂直算术格上]\label{thm:conclusion-dyadic-flux-finite-state-vertical-pole-grids}
\leanverified{paper\_conclusion\_dyadic\_flux\_finite\_state\_vertical\_pole\_grids}
设 \(L\subset \Sigma^\ast\) 为具有确定型有限状态线性表示的 dyadic polycube 族，满足
\[
N_m(L)=u^{\mathsf T}M^m v,
\qquad
\zeta_L(s)=\sum_{m\ge 0}N_m(L)\,2^{m(s-n)}.
\]
记 \(M\) 的非零特征值集合为 \(\Spec(M)\setminus\{0\}\)。则 \(\zeta_L\) 的极点集合满足
\[
\operatorname{Pole}(\zeta_L)\subseteq
\bigcup_{\lambda\in\Spec(M)\setminus\{0\}}
\left\{
n-\log_2|\lambda|
+\frac{i\bigl(-\arg\lambda+2\pi k\bigr)}{\log 2}
\;:\;k\in\ZZ
\right\}.
\]
特别地，\(\zeta_L\) 的全部极点都落在有限条垂直直线上，且虚部方向的公周期统一为
\[
\frac{2\pi}{\log 2}.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:spg-regular-language-stokes-dyadic-zeta-rationality}，
\[
\zeta_L(s)=u^{\mathsf T}\bigl(I-2^{s-n}M\bigr)^{-1}v.
\]
因此极点只能来自
\[
\det\!\bigl(I-2^{s-n}M\bigr)=0.
\]
这等价于存在某个非零特征值 \(\lambda\in\Spec(M)\setminus\{0\}\) 使
\[
2^{s-n}=\lambda^{-1}.
\]
将 \(\lambda\) 写成
\[
\lambda=|\lambda|e^{i\arg\lambda},
\]
则
\[
(s-n)\log 2
=
\log(\lambda^{-1})
=
-\log|\lambda|-i\arg\lambda+2\pi i k
\qquad(k\in\ZZ).
\]
解出 \(s\) 即得
\[
s
=
n-\log_2|\lambda|
+\frac{i\bigl(-\arg\lambda+2\pi k\bigr)}{\log 2},
\qquad
k\in\ZZ.
\]
故所有极点均落在所示有限并的垂直算术格上，而虚部步长统一为 \(2\pi/\log 2\)。证毕。
\end{proof}

\leanverified{paper\_conclusion\_dyadic\_flux\_primitive\_no\_leading\_logperiodic}
\begin{theorem}[primitive 正则 dyadic 家族的主导极点不携带对数周期拍频]\label{thm:conclusion-dyadic-flux-primitive-no-leading-logperiodic}
设 \(L\subset \Sigma^\ast\) 为 primitive 正则 dyadic polycube 族。记 Perron 根为 \(\rho>1\)，次谱模满足
\[
|\lambda_2|<\rho,
\qquad
d(L):=\log_2\rho,
\qquad
s_\ast(L):=n-d(L).
\]
则成立：
\begin{enumerate}
  \item \(\zeta_L\) 在右端谱线上仅有唯一主导极点 \(s_\ast(L)\)，且该极点为实单极点；
  \item 归一化通量满足指数收敛
  \[
  \left|
  2^{m(n-d(L))}\int_{\partial U_m(L)}\omega_0-\mathcal M_2(L)
  \right|
  \le
  C_0\left(\frac{|\lambda_2|}{\rho}\right)^m
  \qquad(m\ge 0);
  \]
  \item 因而主项中不存在任何对数周期振荡；所有振荡只能出现在指数衰减误差层。
\end{enumerate}
\end{theorem}
\begin{proof}
命题 \ref{prop:spg-regular-language-stokes-dyadic-zeta-rationality} 已给出：\(\zeta_L\) 在
\[
s_\ast(L)=n-\log_2\rho
\]
处具有单极点，并满足
\[
\Res_{s=s_\ast(L)}\zeta_L(s)=\frac{\mathcal M_2(L)}{\log 2},
\]
以及指数收敛估计
\[
\left|
2^{m(n-d(L))}\int_{\partial U_m(L)}\omega_0-\mathcal M_2(L)
\right|
\le
C_0\left(\frac{|\lambda_2|}{\rho}\right)^m.
\]
这给出第二条。

另一方面，由 Perron--Frobenius 定理，primitive 非负矩阵在模 \(\rho\) 上只有唯一特征值 \(\rho\)，并且该特征值为正实单根。由定理 \ref{thm:conclusion-dyadic-flux-finite-state-vertical-pole-grids}，位于最右直线
\[
\Re s=n-\log_2\rho
\]
上的极点只能来自满足 \(|\lambda|=\rho\) 的特征值，因此只能对应 \(\lambda=\rho\)，从而只能是实点 \(s_\ast(L)\)。于是右端谱线上不存在第二个共模极点，更不存在非实共轭极点对。

因此，主导项只可能来自单一实单极点；若出现对数周期拍频，则必须有同一最右实部上的非实极点参与主项，已被排除。故第三条成立。证毕。
\end{proof}

\begin{theorem}[双权探针的几何消元律]\label{thm:conclusion-dyadic-flux-two-weight-geometric-elimination}
\leanverified{paper\_conclusion\_dyadic\_flux\_two\_weight\_geometric\_elimination}
设 \(A\subset I^n\) 为 Minkowski 可测紧集，\(\dim_{\mathrm M}(A)=d\)。取 \(\alpha,\beta\ge 0\)，并令 \(f_\alpha,f_\beta\in C^\infty(I^n)\) 为两组非负权函数，满足在 \(A\) 的某个邻域内
\[
f_\alpha(x)\asymp \dist(x,A)^\alpha,
\qquad
f_\beta(x)\asymp \dist(x,A)^\beta.
\]
记相应加权通量为
\[
\Phi_m(f;A):=\int_{\partial U_m(A)}\omega_f=\int_{U_m(A)}f(x)\,dx.
\]
则有严格比值律
\[
\frac{\Phi_m(f_\beta;A)}{\Phi_m(f_\alpha;A)}
=
\Theta\!\bigl(2^{-m(\beta-\alpha)}\bigr),
\]
并且
\[
\lim_{m\to\infty}
\frac{\log \Phi_m(f_\beta;A)-\log \Phi_m(f_\alpha;A)}{m\log 2}
=-(\beta-\alpha).
\]
因此，两个消失阶之差 \(\beta-\alpha\) 可在完全消去几何未知量 \(d\) 后由纯比值读出。
\end{theorem}
\begin{proof}
由命题 \ref{prop:spg-weighted-stokes-flux-vanishing-dimension-shift}，当 \(f(x)\asymp \dist(x,A)^\gamma\) 时有
\[
\Phi_m(f;A)=\Theta\!\bigl(2^{-m(n-d+\gamma)}\bigr).
\]
分别对 \(\gamma=\alpha\) 与 \(\gamma=\beta\) 应用，得到
\[
\Phi_m(f_\alpha;A)=\Theta\!\bigl(2^{-m(n-d+\alpha)}\bigr),
\qquad
\Phi_m(f_\beta;A)=\Theta\!\bigl(2^{-m(n-d+\beta)}\bigr).
\]
相除即得
\[
\frac{\Phi_m(f_\beta;A)}{\Phi_m(f_\alpha;A)}
=
\Theta\!\bigl(2^{-m(\beta-\alpha)}\bigr).
\]
换言之，
\[
\log \Phi_m(f_\beta;A)-\log \Phi_m(f_\alpha;A)
=
-m(\beta-\alpha)\log 2+O(1),
\]
两边除以 \(m\log 2\) 并令 \(m\to\infty\)，即得所示极限。证毕。
\end{proof}

\begin{corollary}[可实现各向异性 Serrin 边界的加权通量响应落在普适仿射律上]\label{cor:conclusion-serrin-wulff-weighted-flux-affine-law}
\leanverified{paper\_conclusion\_serrin\_wulff\_weighted\_flux\_affine\_law}
设 \(\Omega\subset\RR^n\) 为 \(H\)-Serrin 可实现域，因而
\[
\Omega=x_0+\rho W_H.
\]
令 \(A:=\partial\Omega\)。则：
\begin{enumerate}
  \item \(A\) 满足
  \[
  d_G(A)=d_{\partial G}(A)=n-1;
  \]
  \item 非加权 Stokes--dyadic 通量 \(\zeta_A(s)\) 在 \(s=1\) 处具有单极点；
  \item 对任意 \(\alpha\ge 0\) 及任意满足
  \[
  f(x)\asymp \dist(x,A)^\alpha
  \]
  的非负权函数，都有
  \[
  \Phi_m(f;A)=\Theta\!\bigl(2^{-m(1+\alpha)}\bigr).
  \]
\end{enumerate}
因此，在可实现各向异性 Serrin 边界上，全部局部 vanishing-order probe 的尺度指数都落在同一条普适仿射直线
\[
\sigma(\alpha)=1+\alpha
\]
上，而与各向异性 \(H\) 的细节无关。
\end{corollary}
\begin{proof}
第一、二条正是定理 \ref{thm:conclusion-serrin-wulff-dyadic-codim-law} 的内容。对第三条，将其中得到的
\[
\dim_{\mathrm M}(A)=n-1
\]
代入定理 \ref{thm:conclusion-dyadic-flux-two-weight-geometric-elimination} 所依赖的加权标度律
\[
\Phi_m(f;A)=\Theta\!\bigl(2^{-m(n-d+\alpha)}\bigr)
\]
即可得到
\[
\Phi_m(f;A)=\Theta\!\bigl(2^{-m(1+\alpha)}\bigr).
\]
故通量指数关于 \(\alpha\) 呈斜率为 \(1\) 的仿射律。证毕。
\end{proof}

\begin{theorem}[primitive 正则家族的主导 meromorphic germ 可由 primorial 体码纯算术恢复]\label{thm:conclusion-primorial-bodycode-dominant-germ-arithmetic-recovery}
\leanverified{paper\_conclusion\_primorial\_bodycode\_dominant\_germ\_arithmetic\_recovery}
设 \(L\subset \Sigma^\ast\) 为 primitive 正则 dyadic polycube 族，并记 primorial 体码
\[
G_m^\ast(L):=\prod_{k=1}^{N_m(L)}p_k,
\qquad
d(L):=\log_2\rho,
\qquad
s_\ast(L):=n-d(L).
\]
则主导 meromorphic germ 的位置与留数满足
\[
s_\ast(L)
=
n-\lim_{m\to\infty}\frac{\log\log G_m^\ast(L)}{m\log 2},
\]
以及
\[
\Res_{s=s_\ast(L)}\zeta_L(s)
=
\lim_{m\to\infty}
\frac{
\exp\!\bigl(\log\log G_m^\ast(L)-m(n-s_\ast(L))\log 2\bigr)
}{
m\,(n-s_\ast(L))(\log 2)^2
}.
\]
进一步，
\[
2^{m s_\ast(L)}\int_{\partial U_m(L)}\omega_0
\longrightarrow
(\log 2)\Res_{s=s_\ast(L)}\zeta_L(s)
=
\mathcal M_2(L),
\]
且该收敛具有指数速度。

因此，primitive 正则 dyadic 家族的主导 meromorphic germ，即主极点位置、主留数及归一化主振幅，都可由 primorial 体码的双对数一、二阶项纯算术恢复，而无需显式对角化传递矩阵。
\end{theorem}
\begin{proof}
前两条极限公式正是定理 \ref{thm:conclusion-primorial-bodycode-pole-residue-inversion}。另一方面，由定理 \ref{thm:conclusion-dyadic-flux-primitive-no-leading-logperiodic}，
\[
\left|
2^{m(n-d(L))}\int_{\partial U_m(L)}\omega_0-\mathcal M_2(L)
\right|
\le
C_0\left(\frac{|\lambda_2|}{\rho}\right)^m,
\]
并且
\[
\mathcal M_2(L)=(\log 2)\Res_{s=s_\ast(L)}\zeta_L(s).
\]
再用
\[
n-d(L)=s_\ast(L)
\]
代回，即得所示归一化主振幅公式与指数收敛。证毕。
\end{proof}

\begin{conjecture}[可实现 Wulff 边界的加权通量 Mellin 变换存在平移极点律]\label{conj:conclusion-serrin-wulff-weighted-mellin-pole-shift}
设 \(\Omega=x_0+\rho W_H\) 为 \(H\)-Serrin 可实现域，\(A:=\partial\Omega\)。对每个 \(\alpha\ge 0\)，取
\[
f_\alpha(x)=\dist(x,A)^\alpha \chi(x),
\]
其中 \(\chi\) 为 \(A\) 邻域内的 cutoff，并定义加权 Stokes--dyadic Mellin 生成函数
\[
\zeta_{A,\alpha}(s):=\sum_{m\ge 0}2^{ms}\Phi_m(f_\alpha;A).
\]
则预期：
\begin{enumerate}
  \item \(\zeta_{A,\alpha}(s)\) 在 \(s=1+\alpha\) 处具有单极点；
  \item 其留数满足严格尺度律
  \[
  \Res_{s=1+\alpha}\zeta_{A,\alpha}(s)=C_{H,\alpha}\,\rho^{n-1+\alpha},
  \qquad
  C_{H,\alpha}>0;
  \]
  \item 当 \(\alpha=0\) 时，上式退化为
  \[
  \Res_{s=1}\zeta_A(s)
  =
  \frac{2\rho^{n-1}\mathcal H^{n-1}(\partial W_H)}{\log 2}.
  \]
\end{enumerate}
\end{conjecture}

\noindent
这一组结果表明，有限状态 dyadic 通量的复奇性几何、primitive 主项的无拍频刚性、加权探针对 vanishing order 的纯比值读出、Serrin/Wulff 边界上的普适仿射响应，以及 primorial 体码对主导 meromorphic germ 的纯算术恢复，并不是若干彼此平行的接口。它们实际上共同刻画了同一个机制：有限状态动力学决定复极点晶格，Minkowski 余维与消失阶共同决定实部平移，而 primorial 双对数展开则把主导奇性进一步压缩为一条可审计的整数码序列。
