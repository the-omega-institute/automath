\subsection{window-\texorpdfstring{$6$}{6} 的 ternary scar、pinned \texorpdfstring{$B_3$}{B3} 数据与 \texorpdfstring{$\mathrm{Spin}(7)$}{Spin(7)} 混合骨架}\label{subsec:conclusion-window6-ternary-scar-pinned-b3-spin7-mixed-object}

在定理 \ref{thm:conclusion-window6-gauge-defect-exact-log-gap}、定理 \ref{thm:conclusion-window6-boundary-quotient-cyclic-cardinality}--推论 \ref{cor:conclusion-window6-minimal-faithful-residual-budget-18}、定理 \ref{thm:conclusion-golden-alternating-constants-recover-phi}--推论 \ref{cor:conclusion-golden-fibonacci-two-phase-renormalization}、定理 \ref{thm:conclusion-window6-multiplicity-pinning}、定理 \ref{thm:conclusion-window6-191dim-semisimple-shadow}、定理 \ref{thm:conclusion-window6-minuscule-eightfold-b3-spin7} 以及定理 \ref{thm:conclusion-window6-minimal-fiber-spin-hypercube}--命题 \ref{prop:conclusion-window6-dyadic-mixed-signature-minimality} 已经分别建立之后，window-\(6\) 的可见末端结构还可进一步压缩为一条统一的 mixed Lie--topological 链：其 gauge 缺口并不是弥散的多素数异常，而是一枚唯一的 ternary scar；其 boundary parity 零模剥离之后剩余的异常预算恰与 cyclic 扇区同维；其 multiplicity field 唯一钉扎一份 \(B_3\) 的 pinned root datum，并进一步把极小 Lie 型压缩到 \(\mathrm{Spin}(7)\) 及其八点自旋超立方体；而连续环面部分与 dyadic 地址部分则由最小混合签名 \((3,2)\) 不可再降地耦合在一起。

\begin{theorem}[规范缺口的唯一 ternary scar]\label{thm:conclusion-window6-gauge-gap-unique-ternary-scar}
记
\[
\Delta_6:=\kappa_6-g_6.
\]
则有精确闭式
\[
64\Delta_6=49\log 2-\log 3,
\qquad
e^{64\Delta_6}=\frac{2^{49}}{3}.
\]
因此，window-\(6\) 的规范缺口在算术上完全分裂为一项纯 dyadic 的 \(49\)-比特体量与一枚唯一不可约的 ternary scar \(-\log 3\)。
\leanverified{paper\_conclusion\_window6\_gauge\_gap\_unique\_ternary\_scar}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-gauge-defect-exact-log-gap}，
\[
\Delta_6=\kappa_6-g_6=\frac{49\log 2-\log 3}{64}.
\]
两边乘以 \(64\) 即得第一式。再指数化得到
\[
e^{64\Delta_6}=e^{49\log 2-\log 3}=\frac{2^{49}}{3}.
\]
证毕。
\end{proof}

\begin{corollary}[剥离 boundary parity 之后的残余异常预算]\label{cor:conclusion-window6-boundary-stripped-residual-budget-cyclic-saturation}
记
\[
C_{\partial}:=(\ZZ/2\ZZ)^{X_6^{\mathrm{bdry}}}\subset (G_6^{\mathrm{bin}})^{\mathrm{ab}}.
\]
则
\[
(G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\cong (\ZZ/2\ZZ)^{18},
\qquad
\dim_{\FF_2}\bigl((G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\bigr)=18=|X_6^{\mathrm{cyc}}|.
\]
更强地，任意 \(\FF_2\)-线性读出
\[
\Lambda:(G_6^{\mathrm{bin}})^{\mathrm{ab}}\longrightarrow (\ZZ/2\ZZ)^r
\]
若满足 \(C_{\partial}\subset \ker\Lambda\) 且在商上仍忠实，则必有
\[
r\ge 18,
\]
并且该下界可达。
\end{corollary}
\begin{proof}
定理 \ref{thm:conclusion-window6-boundary-quotient-cyclic-cardinality} 已给出
\[
(G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\cong (\ZZ/2\ZZ)^{18},
\qquad
\dim_{\FF_2}\bigl((G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\bigr)=18=|X_6^{\mathrm{cyc}}|.
\]
若 \(\Lambda\) 在商上仍忠实，则诱导映射
\[
\overline\Lambda:(G_6^{\mathrm{bin}})^{\mathrm{ab}}/C_{\partial}\hookrightarrow (\ZZ/2\ZZ)^r
\]
必须为单射，因此 \(r\ge 18\)。而下界可达性正是推论 \ref{cor:conclusion-window6-minimal-faithful-residual-budget-18} 的内容。证毕。
\end{proof}

\begin{theorem}[黄金二相极限常数的完备反演对]\label{thm:conclusion-window6-golden-two-phase-constants-complete-inversion-pair}
记
\[
c_-:=
\lim_{\substack{n\to\infty\\ \delta_n<0}}
F_n\,\mathsf T_{F_n}(\varphi^{-1}),
\qquad
c_+:=
\lim_{\substack{n\to\infty\\ \delta_n>0}}
F_n\,\mathsf T_{F_n}(\varphi^{-1}).
\]
则
\[
c_-=\frac12+\frac{1}{2\sqrt5},
\qquad
c_+=\frac12-\frac{1}{2\sqrt5}+\frac1{15},
\]
并且
\[
\sqrt5=\frac{1}{c_--c_++\frac1{15}},
\qquad
\varphi=\frac12\left(1+\frac{1}{c_--c_++\frac1{15}}\right).
\]
因此 \((c_-,c_+)\) 构成 \(\varphi\) 的一组显式完备反演对。
\end{theorem}
\begin{proof}
由推论 \ref{cor:conclusion-golden-fibonacci-two-phase-renormalization}，
\[
c_-=\frac12+\frac{1}{2\sqrt5},
\qquad
c_+=\frac12-\frac{1}{2\sqrt5}+\frac1{15}.
\]
故
\[
c_--c_+=\frac1{\sqrt5}-\frac1{15},
\]
从而
\[
\frac1{\sqrt5}=c_--c_++\frac1{15},
\qquad
\sqrt5=\frac{1}{c_--c_++\frac1{15}}.
\]
再代入
\[
\varphi=\frac{1+\sqrt5}{2}
\]
便得到第二式。这与定理 \ref{thm:conclusion-golden-alternating-constants-recover-phi} 完全一致。证毕。
\end{proof}

\begin{corollary}[黄金分支最小重整化对象的二相性]\label{cor:conclusion-window6-golden-two-phase-cocycle-minimality}
\leanverified{paper\_conclusion\_window6\_golden\_two\_phase\_cocycle\_minimality}
沿用定理 \ref{thm:conclusion-window6-golden-two-phase-constants-complete-inversion-pair} 的记号，则
\[
\frac{c_-+c_+}{2}=\frac{8}{15},
\qquad
\frac{c_--c_+}{2}=\frac{1}{2\sqrt5}-\frac1{30}.
\]
特别地，序列
\[
F_n\,\mathsf T_{F_n}(\varphi^{-1})
\]
恰有两条聚点而非一条。因此，任何单标量重整化都不足以闭合该极限行为；其最小闭合对象必为一个奇偶二相 cocycle。
\end{corollary}
\begin{proof}
这正是推论 \ref{cor:conclusion-golden-fibonacci-two-phase-renormalization} 的内容。由于 \(c_-\neq c_+\)，全序列不可能收敛到单一常数；而由定理 \ref{thm:conclusion-window6-golden-two-phase-constants-complete-inversion-pair}，这一二相对已经足以闭式恢复 \(\varphi\)。故二相 cocycle 正是最小闭合对象。证毕。
\end{proof}

\begin{theorem}[multiplicity field 的 pinned \texorpdfstring{$B_3$}{B3} 刚性与 Klein 四稳定子]\label{thm:conclusion-window6-multiplicity-field-pinned-b3-klein-four}
在 \(B_3\) 字典下定义多重度场
\[
m_6:\Phi(B_3)\longrightarrow \{2,3,4\}.
\]
则其长根层精确分解为
\[
L_2=\{\alpha\in\Phi_{\mathrm{long}}(B_3):\alpha_1=0\},
\qquad
L_3=\{\alpha\in\Phi_{\mathrm{long}}(B_3):\alpha_2=0\},
\qquad
L_4=\{\alpha\in\Phi_{\mathrm{long}}(B_3):\alpha_3=0\},
\]
并满足
\[
m_6(\alpha)=2 \iff \alpha=-e_1\ \text{或}\ \alpha\in L_2,
\]
\[
m_6(\alpha)=3 \iff \alpha\in L_3,
\]
\[
m_6(\alpha)=4 \iff \alpha\in L_4\ \text{或}\ \alpha\in \{e_1,\pm e_2,\pm e_3\}.
\]
因此 \(m_6\) 唯一恢复有序旗标
\[
\RR e_1\subset \RR e_1\oplus \RR e_2\subset \RR^3,
\]
并把 signed Weyl 对称严格压缩为
\[
\operatorname{Stab}_{W(B_3)}(m_6)
=
\left\{
\operatorname{diag}(1,\varepsilon_2,\varepsilon_3):\ \varepsilon_2,\varepsilon_3\in\{\pm1\}
\right\}
\cong (\ZZ/2\ZZ)^2.
\]
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:conclusion-window6-multiplicity-pinning} 的重述。该定理表明，三层 \(L_2,L_3,L_4\) 分别切出三张有序坐标超平面，而短根中唯一的二重点 \(-e_1\) 进一步固定第一坐标轴的方向，故 \(m_6\) 不仅恢复抽象根系，而且恢复一份 pinned \(B_3\) 数据。相应稳定子也正由该定理给出。证毕。
\end{proof}

\begin{theorem}[可见 \texorpdfstring{$21$}{21} 维完成只是 \texorpdfstring{$191$}{191} 维 Lie 阴影中的 codimension-\texorpdfstring{$170$}{170} 扇区]\label{thm:conclusion-window6-visible-21dim-codim170-lie-shadow}
\leanverified{paper\_conclusion\_window6\_visible\_21dim\_codim170\_lie\_shadow}
设
\[
\mathcal A_6:=\CC[\mathcal R_6^{\mathrm{bin}}].
\]
则
\[
[\mathcal A_6,\mathcal A_6]
\cong
\mathfrak{sl}_2(\CC)^{\oplus 8}
\oplus
\mathfrak{sl}_3(\CC)^{\oplus 4}
\oplus
\mathfrak{sl}_4(\CC)^{\oplus 9},
\]
从而
\[
\dim_{\CC}[\mathcal A_6,\mathcal A_6]
=
8\cdot 3+4\cdot 8+9\cdot 15
=191.
\]
另一方面，window-\(6\) 的 root--Cartan 可见完成无论取 \(B_3\) 侧还是 \(C_3\) 侧，其复维都等于
\[
\dim_{\CC}\mathfrak{so}_7(\CC)=\dim_{\CC}\mathfrak{sp}_6(\CC)=21.
\]
故可见 Lie 完成只是一个 codimension-\(170\) 的显影扇区：
\[
191-21=170.
\]
\end{theorem}
\begin{proof}
前半部分即定理 \ref{thm:conclusion-window6-191dim-semisimple-shadow}。后半部分则来自定理 \ref{thm:conclusion-window6-abstract-chevalley-completion}，该定理表明 window-\(6\) 的可见 root--Cartan 完成只能是 \(B_3\) 或 \(C_3\) 型，而两者的复维都为 \(21\)。因此可见扇区相对于 \(191\) 维半单 Lie 阴影的余维恰为
\[
191-21=170.
\]
证毕。
\end{proof}

\begin{theorem}[八个最小纤维唯一选出 \texorpdfstring{$B_3$}{B3}，从而唯一选出 \texorpdfstring{$\mathrm{Spin}(7)$}{Spin(7)}]\label{thm:conclusion-window6-eight-minimal-fibers-select-spin7}
记最小纤维层
\[
Y_2:=\{w\in X_6:\ d_6^{\mathrm{bin}}(w)=2\}.
\]
则 \(|Y_2|=8\)。若要求 \(Y_2\) 在同一极小 Lie 完成的某个辅助非零 minuscule 表示中实现为单一 Weyl 权轨道，则 \(C_3\) 侧被唯一排除，只剩
\[
K_{6,\mathrm{sc}}\cong \mathrm{Spin}(7).
\]
并且存在 Weyl-等变识别
\[
Y_2\longleftrightarrow
\left\{
\left(\pm\frac12,\pm\frac12,\pm\frac12\right)
\right\},
\]
即与 \(\mathrm{Spin}(7)\) 自旋表示的八个权完全对应。
\end{theorem}
\begin{proof}
这正是定理 \ref{thm:conclusion-window6-minuscule-eightfold-b3-spin7} 的内容。该定理利用 \(|Y_2|=8\) 与非零 minuscule 轨道基数的比较，唯一排除了 \(C_3\) 并选出 \(B_3\)；其单连通紧实现正是 \(\mathrm{Spin}(7)\)。证毕。
\end{proof}

\begin{theorem}[最小纤维层的自旋超立方体、boundary 二挠作用与极小混合签名]\label{thm:conclusion-window6-spin-hypercube-boundary-torsion-mixed-signature}
在定理 \ref{thm:conclusion-window6-eight-minimal-fibers-select-spin7} 的 \(\mathrm{Spin}(7)\) 完成下，若把两点权仅在一个坐标上变号视为相邻，则 \(Y_2\) 的邻接图同构于三维超立方体
\[
Q_3.
\]
同时，boundary 三个中心 involution 经由
\[
T_6[2]\cong (\ZZ/2\ZZ)^3
\]
在自旋模的八个权空间上按特征标对角作用，因而可同时实现为三组彼此交换的符号矩阵。再令
\[
P_2:=T_2(E_{\min})\cong \ZZ_2^2,
\qquad
\mathcal E_6^{\mathrm{mix}}:=T_6\times P_2,
\]
则其 dyadic 混合签名满足
\[
\sigma_2(\mathcal E_6^{\mathrm{mix}})=(3,2),
\]
且这是保持连续环面部分与 dyadic 地址部分都忠实的最小可能值：若
\[
f:\mathcal E_6^{\mathrm{mix}}\longrightarrow \TT^r\times \ZZ_2^s
\]
在两侧限制都忠实，则必有
\[
r\ge 3,
\qquad
s\ge 2.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-window6-minimal-fiber-spin-hypercube}，\(Y_2\) 已可 Weyl-等变识别为
\[
\left\{
\left(\pm\frac12,\pm\frac12,\pm\frac12\right)
\right\},
\]
而“仅在一个坐标上变号”的邻接恰是三维超立方体 \(Q_3\) 的标准邻接。该定理同时表明，boundary 三个中心 involution 通过
\[
T_6[2]\cong (\ZZ/2\ZZ)^3
\]
在各个权空间上以 \(\{\pm1\}\)-字符对角作用，因此可实现为三组彼此交换的符号矩阵。

另一方面，由命题 \ref{prop:conclusion-window6-dyadic-mixed-signature-minimality}，
\[
\sigma_2(\mathcal E_6^{\mathrm{mix}})=(3,2),
\]
并且对任意保持 \(T_6\) 与 \(P_2\) 两侧都忠实的连续同态
\[
f:\mathcal E_6^{\mathrm{mix}}\to \TT^r\times \ZZ_2^s
\]
都必须满足
\[
r\ge 3,\qquad s\ge 2.
\]
故 \((3,2)\) 正是最小混合签名。证毕。
\end{proof}

\begin{conjecture}[minuscule 边界刚性的三元计数强化]\label{conj:conclusion-window6-minuscule-boundary-rigidity-triple-count}
作为猜想 \ref{conj:conclusion-minuscule-boundary-rigidity} 的强化版本，设某个分辨率窗口 \(m\) 满足刚性分裂
\[
X_m=X_m^{\mathrm{cyc}}\sqcup X_m^{\mathrm{bdry}},
\]
并记
\[
Y_m^{\min}:=\{x\in X_m:\ d_m^{\mathrm{bin}}(x)=d_{\min}\}.
\]
若 \(Y_m^{\min}\) 能在某个极小 Lie 完成的非零 minuscule 表示中实现为单一 Weyl 权轨道，则该极小 Lie 型应当由三元计数
\[
\bigl(|X_m^{\mathrm{cyc}}|,\ |Y_m^{\min}|,\ \dim_{\FF_2}T_m[2]\bigr)
\]
唯一决定，至多差 Langlands 对偶。

对 window-\(6\)，这一三元数据为
\[
\bigl(|X_6^{\mathrm{cyc}}|,\ |Y_2|,\ \dim_{\FF_2}T_6[2]\bigr)=(18,8,3),
\]
并且正是这组数据同时压出 \(m_6\) 的 pinned \(B_3\) 数据、排除 \(C_3\) 并唯一选出 \(\mathrm{Spin}(7)\)。
\end{conjecture}

由此，window-\(6\) 的末端可见对象不再适宜被理解为一个孤立的有限 combinatorial gadget。更合适的压缩是：其规范缺口携带一枚唯一的 ternary scar，其 boundary parity 零模剥离之后留下一个 \(18\) 维 cyclic 残余预算，其 multiplicity field 唯一钉扎一份 \(B_3\) 的 pinned 根数据，其最小纤维层唯一选出 \(\mathrm{Spin}(7)\) 的八点自旋超立方体，而其连续相位与 dyadic 地址账本则被最小混合签名 \((3,2)\) 不可再降地耦合为一个 mixed Lie--topological object。
