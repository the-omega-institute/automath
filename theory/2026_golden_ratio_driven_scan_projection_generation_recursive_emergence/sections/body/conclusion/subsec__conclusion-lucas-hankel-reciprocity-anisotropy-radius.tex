\subsection{Lucas--Hankel 证书的倒易系数刚性、\texorpdfstring{$(\infty,5)$}{(∞,5)}-各向异性与半径二歧}\label{subsec:conclusion-lucas-hankel-reciprocity-anisotropy-radius}
沿用定义 \ref{def:conclusion-lucas-fib-input}、命题 \ref{prop:conclusion-lucas-power-minpoly-reciprocity}、命题 \ref{prop:conclusion-shifted-hankel-rigidity}、定理 \ref{thm:conclusion-lucas-power-hankel-closed-form} 与推论 \ref{cor:conclusion-golden-lucas-hankel-v5-quadratic-growth} 的记号。前述结论已经把 Lucas 幂 Hankel 主子式写成二项系数、判别式与 Fibonacci 因子的完全乘积，并给出最小递推、倒易对称、移位刚性与黄金特例的 \(5\)-进二次增长。把这些接口继续压缩后，还可再抽出如下几条终端后果：\(\Pi_q\) 的系数本身满足精确倒易律；黄金特例中的一次 cyclotomic 因子与移位符号完全由模 \(4\) 剩余类控制；同一 Hankel 证书在 archimedean 与 \(5\)-进两端呈现严格不同的主导标度，并最终把其母函数推入“复零半径 / \(5\)-进整函数”的分裂 regime。

\begin{corollary}[\texorpdfstring{$\Pi_q$}{Pi_q} 的精确系数倒易律]\label{cor:conclusion-lucas-piq-coefficient-reciprocity}
\leanverified{paper\_conclusion\_lucas\_piq\_coefficient\_reciprocity}
将
\[
\Pi_q(t)
=
t^{q+1}-e_1t^q+e_2t^{q-1}-\cdots+(-1)^{q+1}e_{q+1},
\qquad
e_0:=1,
\]
写成基本对称系数形式。则对每个 \(0\le j\le q+1\) 都有
\[
e_{q+1-j}
=
P^{\,\frac{q(q+1-2j)}{2}}\,e_j.
\]
因此，\(\Pi_q\) 的全部系数并非自由参数；后一半系数由前一半系数以显式 \(P\)-幂因子唯一锁定。
\end{corollary}
\begin{proof}
由命题 \ref{prop:conclusion-lucas-power-minpoly-reciprocity}，
\[
t^{q+1}\Pi_q(P^q/t)
=
(-1)^{q+1}P^{\frac{q(q+1)}{2}}\Pi_q(t).
\]
把
\[
\Pi_q(t)=\sum_{j=0}^{q+1}(-1)^je_jt^{q+1-j}
\]
代入，左边化为
\[
\sum_{j=0}^{q+1}(-1)^j e_j P^{q(q+1-j)} t^j,
\]
而右边化为
\[
\sum_{j=0}^{q+1}(-1)^j P^{\frac{q(q+1)}{2}} e_{q+1-j} t^j.
\]
比较 \(t^j\) 系数便得
\[
e_jP^{q(q+1-j)}
=
P^{\frac{q(q+1)}{2}}e_{q+1-j},
\]
整理即得所述公式。证毕。
\end{proof}

\begin{theorem}[黄金比例特例的一次 cyclotomic 闸门]\label{thm:conclusion-golden-lucas-linear-cyclotomic-gate}
\leanverified{paper\_conclusion\_golden\_lucas\_linear\_cyclotomic\_gate}
在黄金比例特例
\[
(S,P)=(1,-1)
\]
下，\(\Pi_q\) 满足：
\begin{enumerate}
  \item 若 \(q\equiv 0\pmod 4\)，则
  \[
  t-1\mid \Pi_q(t);
  \]
  \item 若 \(q\equiv 2\pmod 4\)，则
  \[
  t+1\mid \Pi_q(t);
  \]
  \item 若 \(q\) 为奇数，则
  \[
  \Pi_q(1)\neq 0,\qquad \Pi_q(-1)\neq 0.
  \]
\end{enumerate}
换言之，黄金特例中唯一可能出现的一次 cyclotomic 因子由 \(q\bmod 4\) 完全决定。
\end{theorem}
\begin{proof}
在 \((S,P)=(1,-1)\) 下，
\[
r_k=\alpha^{q-k}\beta^k=(-1)^k\alpha^{q-2k}.
\]
若 \(q\) 为偶数，取 \(k=q/2\)，则
\[
r_{q/2}=(-1)^{q/2}.
\]
因此 \(q\equiv 0\pmod 4\) 时根为 \(1\)，而 \(q\equiv 2\pmod 4\) 时根为 \(-1\)，于是分别得到
\[
t-1\mid \Pi_q(t),
\qquad
t+1\mid \Pi_q(t).
\]

若 \(q\) 为奇数，而 \(\Pi_q(\pm 1)=0\)，则某个 \(r_k=\pm 1\)。取绝对值便有
\[
1=|r_k|=|\alpha|^{q-2k}.
\]
由于 \(|\alpha|=\varphi>1\)，只能有 \(q-2k=0\)，这与 \(q\) 为奇数矛盾。故 \(\pm 1\) 都不是根。证毕。
\end{proof}

\begin{corollary}[黄金比例特例移位 Hankel 主子式的模四符号周期律]\label{cor:conclusion-golden-shifted-hankel-mod4-sign-periodicity}\leanverified{paper\_conclusion\_golden\_shifted\_hankel\_mod4\_sign\_periodicity}
在黄金比例特例下，记
\[
D_q:=\det(H_q),
\qquad
D_q^{(r)}:=\det(H_q^{(r)}).
\]
则
\[
D_q^{(r)}=(-1)^{\,r q(q+1)/2}D_q.
\]
等价地，
\[
D_q^{(r)}=
\begin{cases}
D_q, & q\equiv 0,3\pmod 4,\\[2pt]
(-1)^rD_q, & q\equiv 1,2\pmod 4.
\end{cases}
\]
因此，移位只可能诱发一种严格的模四符号翻转，而不存在更细的 archimedean 失真。
\end{corollary}
\begin{proof}
由命题 \ref{prop:conclusion-shifted-hankel-rigidity}，
\[
\det(H_q^{(r)})
=
P^{\,r q(q+1)/2}\det(H_q).
\]
在 \(P=-1\) 下即得
\[
D_q^{(r)}=(-1)^{\,r q(q+1)/2}D_q.
\]
再注意
\[
\frac{q(q+1)}{2}\equiv
\begin{cases}
0 \pmod 2, & q\equiv 0,3\pmod 4,\\
1 \pmod 2, & q\equiv 1,2\pmod 4,
\end{cases}
\]
便得到分段表示。证毕。
\end{proof}

\begin{theorem}[黄金比例特例 Hankel 主子式的 archimedean 立方增长律]\label{thm:conclusion-golden-lucas-hankel-archimedean-cubic-growth}
\leanverified{paper\_conclusion\_golden\_lucas\_hankel\_archimedean\_cubic\_growth}
在黄金比例特例下，
\[
\log |D_q|
=
\frac{q(q+1)(q+2)}{3}\log\varphi
+ O(q^2).
\]
特别地，
\[
\lim_{q\to\infty}\frac{1}{q^3}\log |D_q|
=
\frac13\log\varphi.
\]
因此，Lucas 幂 Hankel 主子式的 archimedean 体量不是二次阶，而是严格的立方阶指数增长。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-lucas-power-hankel-closed-form}，
\[
D_q
=
5^{\frac{q(q+1)}2}
\left(\prod_{k=0}^q \binom{q}{k}\right)
\prod_{d=1}^q F_d^{\,2(q+1-d)}.
\]
取对数得
\[
\log|D_q|
=
\frac{q(q+1)}2\log 5
+
\sum_{k=0}^q\log\binom{q}{k}
+
2\sum_{d=1}^q(q+1-d)\log F_d.
\]
由 Binet 公式，
\[
\log F_d=d\log\varphi-\frac12\log 5+O(\varphi^{-2d}),
\]
故
\[
2\sum_{d=1}^q(q+1-d)\log F_d
=
2\log\varphi\sum_{d=1}^q(q+1-d)d
-\log 5\sum_{d=1}^q(q+1-d)
+O(q).
\]
其中
\[
\sum_{d=1}^q(q+1-d)d=\frac{q(q+1)(q+2)}6,
\qquad
\sum_{d=1}^q(q+1-d)=\frac{q(q+1)}2.
\]
因此 \(\log 5\) 的二次主项完全抵消，得到
\[
\log|D_q|
=
\frac{q(q+1)(q+2)}{3}\log\varphi
+
\sum_{k=0}^q\log\binom{q}{k}
+O(q).
\]
最后，
\[
0\le \sum_{k=0}^q\log\binom{q}{k}\le q(q+1)\log 2=O(q^2),
\]
故结论成立。证毕。
\end{proof}

\begin{theorem}[黄金比例特例的 \texorpdfstring{$(\infty,5)$}{(∞,5)}-各向异性律]\label{thm:conclusion-golden-lucas-hankel-inf5-anisotropy}
\leanverified{paper\_conclusion\_golden\_lucas\_hankel\_inf5\_anisotropy}
在黄金比例特例下，
\[
\frac{1}{q^3}\log |D_q|\to \frac13\log\varphi,
\qquad
\frac{1}{q^2}v_5(D_q)\to \frac34.
\]
从而
\[
\frac{v_5(D_q)}{\log|D_q|}\longrightarrow 0.
\]
因此，同一整数序列 \(D_q\) 在 archimedean 与 \(5\)-进两端呈现严格不同的主导尺度：前者为立方阶，后者仅为二次阶。
\end{theorem}
\begin{proof}
第一式由定理 \ref{thm:conclusion-golden-lucas-hankel-archimedean-cubic-growth} 给出；第二式由推论 \ref{cor:conclusion-golden-lucas-hankel-v5-quadratic-growth} 给出。于是
\[
v_5(D_q)=\frac34\,q^2+O(q\log q),
\qquad
\log|D_q|=\frac13(\log\varphi)q^3+O(q^2).
\]
两式相除即得
\[
\frac{v_5(D_q)}{\log|D_q|}=O(q^{-1})\to 0.
\]
证毕。
\end{proof}

\begin{corollary}[黄金特例 shifted Hankel 主子式的 \texorpdfstring{$5$}{5}-进刚度完全不变]\label{cor:conclusion-golden-shifted-hankel-v5-invariance}
\leanverified{paper\_conclusion\_golden\_shifted\_hankel\_v5\_invariance}
在黄金比例特例下，对任意移位 \(r\ge 0\) 都有
\[
v_5\!\bigl(\det(H_q^{(r)})\bigr)=v_5(D_q).
\]
特别地，所有 shifted Hankel 证书在 \(5\)-进方向上携带完全相同的刚度指数。
\end{corollary}
\begin{proof}
由推论 \ref{cor:conclusion-golden-shifted-hankel-mod4-sign-periodicity}，
\[
\det(H_q^{(r)})=\pm D_q.
\]
而 \(5\)-进赋值对符号不敏感，故
\[
v_5\!\bigl(\det(H_q^{(r)})\bigr)=v_5(D_q).
\]
证毕。
\end{proof}

\begin{theorem}[Hankel 行列式母函数的复/\texorpdfstring{$5$}{5}-进半径二歧]\label{thm:conclusion-golden-hankel-ogf-radius-dichotomy}
\leanverified{paper\_conclusion\_golden\_hankel\_ogf\_radius\_dichotomy}
约定
\[
D_0:=1,
\]
并定义幂级数
\[
\mathcal D(T):=\sum_{q\ge 0}D_qT^q.
\]
则：
\begin{enumerate}
  \item 作为复幂级数，\(\mathcal D(T)\) 的收敛半径为
  \[
  R_\infty=0;
  \]
  \item 作为 \(\QQ_5\)-幂级数，\(\mathcal D(T)\) 的收敛半径为
  \[
  R_5=+\infty.
  \]
\end{enumerate}
换言之，同一整数序列 \(D_q\) 在复域上生成零半径级数，却在 \(5\)-进域上生成整函数。
\end{theorem}
\begin{proof}
复收敛半径由 Cauchy--Hadamard 公式给出
\[
R_\infty^{-1}
=
\limsup_{q\to\infty}|D_q|^{1/q}.
\]
由定理 \ref{thm:conclusion-golden-lucas-hankel-archimedean-cubic-growth}，
\[
\frac1q\log |D_q|
=
\frac13(\log\varphi)q^2+O(q)\to +\infty,
\]
故 \(|D_q|^{1/q}\to +\infty\)，从而 \(R_\infty=0\)。

另一方面，
\[
R_5^{-1}
=
\limsup_{q\to\infty}|D_q|_5^{1/q}
=
\limsup_{q\to\infty}5^{-v_5(D_q)/q}.
\]
由推论 \ref{cor:conclusion-golden-lucas-hankel-v5-quadratic-growth}，
\[
\frac{v_5(D_q)}{q}\to+\infty,
\]
故
\[
|D_q|_5^{1/q}=5^{-v_5(D_q)/q}\to 0.
\]
于是 \(R_5^{-1}=0\)，即 \(R_5=+\infty\)。证毕。
\end{proof}

\noindent
因此，Lucas 幂 Hankel 证书在结论层又进一步裂解为三条相互咬合而不再等价的刚性：其一，\(\Pi_q\) 的根集倒易对称会在系数层直接压出精确倒易律，并在黄金特例中把全部一次 cyclotomic 因子压缩到 \(q\bmod 4\) 的单一闸门；其二，shift 作用在 archimedean 侧至多制造模四符号翻转，而在 \(5\)-进侧则完全看不见，从而暴露出一种严格的 \((\infty,5)\)-各向异性；其三，同一 Hankel 行列式序列在复域上的母函数立刻坍缩为零半径，而在 \(5\)-进域上却延拓为整函数。于是，Lucas–Hankel 证书的刚性不再只是某个单一判别式恒等式，而是一组同时作用在系数、相位、shift 与解析半径层面的多端口约束。
