\subsection{单导子 torsion 积的 cyclotomic--Euler 编译与 Schur--容量塔的单坐标完备性}\label{subsec:conclusion-ramanujan-torsion-schur-capacity-singleconductor}
沿用定理 \ref{thm:conclusion-schur-master-kernel-cauchy-externalization} 与命题 \ref{prop:pom-schur-trace-jacobi-trudi-euler-primitive-fiber} 的记号，固定分辨率 \(m\)，并定义平凡 Schur 通道生成函数
\[
H_m(z):=\sum_{q\ge 0}T_{q,(q)}(m)z^q
=
\prod_{x\in X_m}(1-d_m(x)z)^{-1}
=
\prod_{n\ge 1}(1-z^n)^{-b_n(m)}.
\]
再记纤维直方图
\[
h_m(s):=\#\{x\in X_m:\ d_m(x)=s\},
\qquad
Q_m(z):=\prod_{x\in X_m}(1-d_m(x)z)=\prod_{s\ge 1}(1-sz)^{h_m(s)}.
\]
对任意整数 \(r\ge 2\)，令
\[
U(r):=(\ZZ/r\ZZ)^\times,
\qquad
\omega_r:=e^{2\pi i/r},
\]
并定义单导子 primitive torsion 积
\[
\Theta_{m,r}(z):=\prod_{a\in U(r)}H_m(\omega_r^a z).
\]
这一对象同时把平凡通道的 rational 积、primitive 坐标的 Euler 积，以及碰撞矩的 Ramanujan 变换压入同一条单变量解析链。

\begin{theorem}[单导子 primitive torsion 积的 cyclotomic--Euler 双因子化]\label{thm:conclusion-single-conductor-torsion-cyclotomic-euler-factorization}
\leanverified{paper\_conclusion\_single\_conductor\_torsion\_cyclotomic\_euler\_factorization}
对每个整数 \(r\ge 2\)，记 \(\Phi_r\) 为第 \(r\) 个 cyclotomic 多项式，\(\phi\) 为 Euler totient。则有严格恒等式
\[
\Theta_{m,r}(z)
=
\prod_{x\in X_m}\Phi_r\!\bigl(d_m(x)z\bigr)^{-1}
=
\prod_{n\ge 1}
\Phi_{\,r/(r,n)}(z^n)^{
-\,b_n(m)\,\phi(r)/\phi(r/(r,n))
}.
\]
\end{theorem}

\begin{proof}
由
\[
H_m(z)=\prod_{x\in X_m}(1-d_m(x)z)^{-1}
\]
得到
\[
\Theta_{m,r}(z)
=
\prod_{x\in X_m}\prod_{a\in U(r)}(1-\omega_r^a d_m(x)z)^{-1}.
\]
而经典恒等式
\[
\prod_{a\in U(r)}(1-\omega_r^a t)=\Phi_r(t)
\]
立即给出第一重因子化。

另一方面，由 Euler 分解
\[
H_m(z)=\prod_{n\ge 1}(1-z^n)^{-b_n(m)}
\]
可得
\[
\Theta_{m,r}(z)
=
\prod_{n\ge 1}\prod_{a\in U(r)}(1-\omega_r^{an}z^n)^{-b_n(m)}.
\]
令 \(d:=(r,n)\)。则 \(\omega_r^n\) 的阶为 \(r/d\)，且当 \(a\) 遍历 \(U(r)\) 时，\(\omega_r^{an}\) 恰遍历全部 primitive \((r/d)\)-th roots，每个出现
\[
\frac{\phi(r)}{\phi(r/d)}
\]
次。因此
\[
\prod_{a\in U(r)}(1-\omega_r^{an}z^n)
=
\Phi_{r/d}(z^n)^{\phi(r)/\phi(r/d)}.
\]
代回并将 \(r/d\) 写作 \(r/(r,n)\)，便得第二重因子化。证毕。
\end{proof}

\begin{theorem}[单导子极点除子对纤维直方图的刚性恢复]\label{thm:conclusion-single-conductor-torsion-pole-divisor-recovers-histogram}
\leanverified{paper\_conclusion\_single\_conductor\_torsion\_pole\_divisor\_recovers\_histogram}
对每个固定 \(r\ge 2\)，函数 \(\Theta_{m,r}(z)\) 的极点除子满足
\[
\operatorname{div}_{\infty}\!\bigl(\Theta_{m,r}\bigr)
=
\sum_{s\ge 1}h_m(s)\sum_{a\in U(r)}\Bigl[\frac{\omega_r^{-a}}{s}\Bigr].
\]
特别地，\(\Theta_{m,r}\) 的极点半径集合恰为
\[
\left\{\frac{1}{s}: h_m(s)\neq 0\right\},
\]
并且在每个半径 \(1/s\) 上，全部 primitive 角点都具有相同极点重数 \(h_m(s)\)。因此任意单个非平凡导子 \(r\ge 2\) 的 torsion 积已经唯一决定完整纤维直方图 \(\{h_m(s)\}_{s\ge 1}\)。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-single-conductor-torsion-cyclotomic-euler-factorization}，
\[
\Theta_{m,r}(z)=\prod_{s\ge 1}\Phi_r(sz)^{-h_m(s)}.
\]
而 \(\Phi_r(sz)\) 的零点恰为
\[
z=\frac{\omega_r^{-a}}{s}
\qquad
(a\in U(r)),
\]
并且这些零点都是单的。故 \(\Theta_{m,r}\) 在上述点恰有重数 \(h_m(s)\) 的极点，从而得到所示极点除子。

由于不同的 \(s\) 给出不同的半径 \(1/s\)，极点半径集合便精确恢复了所有出现的纤维值；再由各半径上的极点重数即可恢复 \(h_m(s)\)。证毕。
\end{proof}

\begin{corollary}[导体 \texorpdfstring{$2$}{2} 的极小完备性与 window-\texorpdfstring{$6$}{6} 的单导子全恢复]\label{cor:conclusion-single-conductor-r2-minimal-completeness-window6}
当 \(r=2\) 时，
\[
U(2)=\{1\},
\qquad
\Phi_2(t)=1+t,
\]
从而
\[
\Theta_{m,2}(z)=H_m(-z)=Q_m(-z)^{-1}.
\]
因此导体 \(2\) 已经是最小的非平凡完备 torsion 坐标。

在 \(m=6\) 时，由推论 \ref{cor:xi-time-part63c-window6-schur-packet-locked-by-charpoly}，
\[
Q_6(z)=(1-2z)^8(1-3z)^4(1-4z)^9,
\]
故
\[
\Theta_{6,2}(z)=(1+2z)^{-8}(1+3z)^{-4}(1+4z)^{-9}.
\]
更一般地，对任意 \(r\ge 2\) 都有
\[
\Theta_{6,r}(z)=\Phi_r(2z)^{-8}\Phi_r(3z)^{-4}\Phi_r(4z)^{-9}.
\]
于是，window-\(6\) 的全部 Schur 断层、全部碰撞矩、全部 primitive 纤维坐标与全部连续容量数据，都可由任意单个非平凡导子的 torsion 积恢复；尤其，导体 \(2\) 已经足够。
\end{corollary}

\begin{proof}
第一式由 \(U(2)=\{1\}\) 与 \(\Phi_2(t)=1+t\) 直接得出。对 \(m=6\) 的显式式，只需将推论 \ref{cor:xi-time-part63c-window6-schur-packet-locked-by-charpoly} 中的
\[
Q_6(z)=(1-2z)^8(1-3z)^4(1-4z)^9
\]
代入 \(\Theta_{m,2}(z)=Q_m(-z)^{-1}\) 即得。最后一式则由定理 \ref{thm:conclusion-single-conductor-torsion-cyclotomic-euler-factorization} 的第一重因子化与 window-\(6\) 的纤维直方图直接给出。证毕。
\end{proof}

\begin{theorem}[单导子 torsion 积的对数导数精确生成 Ramanujan 变换后的碰撞矩]\label{thm:conclusion-single-conductor-logderivative-ramanujan-moments}
对每个 \(r\ge 2\)，定义
\[
\mathcal G_{m,r}(z):=
z\frac{\Theta_{m,r}'(z)}{\Theta_{m,r}(z)},
\qquad
c_r(q):=\sum_{a\in U(r)}\omega_r^{aq}.
\]
则有双重显式公式
\[
\mathcal G_{m,r}(z)=\sum_{q\ge 1}c_r(q)\,S_q(m)\,z^q,
\]
\[
\mathcal G_{m,r}(z)
=
-\sum_{s\ge 1}h_m(s)\,s z\,\frac{\Phi_r'(sz)}{\Phi_r(sz)}.
\]
若再约定
\[
\Theta_{m,1}(z):=H_m(z),
\qquad
\mathcal G_{m,1}(z):=z\frac{H_m'(z)}{H_m(z)},
\qquad
c_1(q):=1,
\]
则对任意整数 \(r\ge 1\) 还成立 divisor-sieve 恒等式
\[
\frac1r\sum_{d\mid r}\mathcal G_{m,d}(z)
=
\sum_{\ell\ge 1}S_{r\ell}(m)\,z^{r\ell}.
\]
\end{theorem}

\begin{proof}
由
\[
\log H_m(z)=\sum_{x\in X_m}\sum_{q\ge 1}\frac{(d_m(x)z)^q}{q}
=
\sum_{q\ge 1}\frac{S_q(m)}{q}z^q
\]
可得
\[
\log \Theta_{m,r}(z)
=
\sum_{a\in U(r)}\log H_m(\omega_r^a z)
=
\sum_{q\ge 1}\frac{S_q(m)}{q}c_r(q)z^q.
\]
对 \(z\) 施加算子 \(z\frac{\dd}{\dd z}\) 便得到第一式。

另一方面，由定理 \ref{thm:conclusion-single-conductor-torsion-cyclotomic-euler-factorization}，
\[
\log\Theta_{m,r}(z)
=
-\sum_{s\ge 1}h_m(s)\log\Phi_r(sz),
\]
再对 \(z\) 施加 \(z\frac{\dd}{\dd z}\)，即得第二式。

最后利用经典 Ramanujan 恒等式
\[
\sum_{d\mid r}c_d(q)=r\,\mathbf 1_{\{r\mid q\}},
\]
把第一式对 \(d\mid r\) 求和，便有
\[
\frac1r\sum_{d\mid r}\mathcal G_{m,d}(z)
=
\sum_{q\ge 1}\mathbf 1_{\{r\mid q\}}S_q(m)z^q
=
\sum_{\ell\ge 1}S_{r\ell}(m)z^{r\ell}.
\]
证毕。
\end{proof}

\begin{theorem}[单导子 torsion 积对 Schur--容量塔的完全充分性]\label{thm:conclusion-single-conductor-torsion-complete-sufficient-statistic}
对任意固定 \(r\ge 2\)，单个解析对象 \(\Theta_{m,r}\) 唯一决定下列全部数据：
\begin{enumerate}
  \item 完整纤维直方图 \(\{h_m(s)\}_{s\ge 1}\)；
  \item 特征多项式与平凡通道生成函数
  \[
  Q_m(z)=\prod_{s\ge 1}(1-sz)^{h_m(s)},
  \qquad
  H_m(z)=Q_m(z)^{-1};
  \]
  \item primitive 纤维坐标 \(\{b_n(m)\}_{n\ge 1}\)，由 Euler 分解
  \[
  H_m(z)=\prod_{n\ge 1}(1-z^n)^{-b_n(m)};
  \]
  \item 全部碰撞矩 \(\{S_q(m)\}_{q\ge 1}\)；
  \item 全部 Schur 通道迹 \(\{T_{q,\lambda}(m)\}_{q\ge 1,\ \lambda\vdash q}\)；
  \item 连续容量曲线
  \[
  \mathcal C_m^{\mathrm{cont}}(T)=\sum_{s\ge 1}h_m(s)\min(s,T).
  \]
\end{enumerate}
因此，任意单个非平凡导子的 torsion 积已经是整个 Schur--容量塔的完全充分统计量。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-single-conductor-torsion-pole-divisor-recovers-histogram}，\(\Theta_{m,r}\) 唯一恢复纤维直方图 \(\{h_m(s)\}\)。于是
\[
Q_m(z)=\prod_{s\ge 1}(1-sz)^{h_m(s)}
\]
与
\[
H_m(z)=Q_m(z)^{-1}
\]
被唯一决定，得到第 \(1\) 与第 \(2\) 条。

第 \(3\) 条随即由命题 \ref{prop:pom-schur-trace-jacobi-trudi-euler-primitive-fiber} 的 Euler 分解给出。第 \(4\) 条可由
\[
H_m(z)=\prod_{x\in X_m}(1-d_m(x)z)^{-1}
\]
直接恢复，亦可由定理 \ref{thm:conclusion-single-conductor-logderivative-ramanujan-moments} 通过 Ramanujan 变换恢复。

对第 \(5\) 条，由定理 \ref{thm:conclusion-schur-single-row-complete-recovery}，单行通道
\[
T_{q,(q)}(m)=[z^q]H_m(z)
\]
已经足以恢复全部 Schur 通道迹。第 \(6\) 条则由定理 \ref{thm:xi-capacity-tail-histogram-moment-complete-statistic} 直接得到；事实上，一旦 \(\{h_m(s)\}\) 已知，容量曲线的显式公式便立刻成立。证毕。
\end{proof}

\begin{theorem}[缩放 torsion 势只依赖于归一化纤维经验谱]\label{thm:conclusion-single-conductor-scaled-torsion-potential-empirical-measure}
记
\[
M_m:=\max_{x\in X_m}d_m(x),
\qquad
\nu_m:=\frac1{|X_m|}\sum_{x\in X_m}\delta_{d_m(x)/M_m}.
\]
对固定 \(r\ge 2\) 与 \(|\zeta|<1\)，定义缩放 torsion 势
\[
\Psi_{m,r}(\zeta):=\frac1{|X_m|}\log\Theta_{m,r}\!\left(\frac{\zeta}{M_m}\right),
\]
其中对数取单位圆盘上的主支。则有精确积分表示
\[
\Psi_{m,r}(\zeta)
=
-\int_{[0,1]}\log \Phi_r(\zeta t)\,\dd\nu_m(t).
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-single-conductor-torsion-cyclotomic-euler-factorization} 的第一重因子化，
\[
\log\Theta_{m,r}\!\left(\frac{\zeta}{M_m}\right)
=
-\sum_{x\in X_m}\log\Phi_r\!\left(\zeta\frac{d_m(x)}{M_m}\right).
\]
当 \(|\zeta|<1\) 且 \(0\le d_m(x)/M_m\le 1\) 时，\(\Phi_r(\zeta d_m(x)/M_m)\neq 0\)，故单位圆盘上的主支对数良定义。将上式两边除以 \(|X_m|\)，便得到
\[
\Psi_{m,r}(\zeta)
=
-\frac1{|X_m|}\sum_{x\in X_m}\log\Phi_r\!\left(\zeta\frac{d_m(x)}{M_m}\right)
=
-\int_{[0,1]}\log \Phi_r(\zeta t)\,\dd\nu_m(t).
\]
证毕。
\end{proof}

\begin{conjecture}[单导子 torsion 势的热力学充分性]\label{conj:conclusion-single-conductor-torsion-thermodynamic-sufficiency}
设归一化纤维经验谱满足
\[
\nu_m\Rightarrow \nu
\qquad
\text{于 }[0,1]\text{ 上弱收敛},
\]
并固定任意 \(r\ge 2\)。则由定理 \ref{thm:conclusion-single-conductor-scaled-torsion-potential-empirical-measure} 所定义的极限势
\[
\Psi_r(\zeta):=
-\int_{[0,1]}\log \Phi_r(\zeta t)\,\dd\nu(t)
\qquad
(|\zeta|<1)
\]
应当已经唯一决定相应的极限容量曲线与极限 Schur 可见性相图。换言之，有限窗层面的“单导子完全充分性”应当在缩放极限下继续保持。
\end{conjecture}

\noindent
由此，Ramanujan 理论在这里不再只是为碰撞矩提供一组变换系数，而是把平凡通道、primitive 纤维坐标、Schur 断层、碰撞矩与容量几何压缩进同一个固定导子的 torsion 函数。与定理 \ref{thm:conclusion-schur-single-row-complete-recovery} 相比，上述链条更进一步：不再需要整条单行 Schur 通道，只需任意一个非平凡 conductor，便已足以无损恢复整个 Schur--容量塔。
