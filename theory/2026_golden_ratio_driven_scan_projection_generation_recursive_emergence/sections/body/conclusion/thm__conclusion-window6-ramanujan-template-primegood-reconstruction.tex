\begin{theorem}[window-\(6\) hidden template 的 prime-to-\(39270\) 单导子 Ramanujan 断层公式]\label{thm:conclusion-window6-hidden-template-primegood-ramanujan-faultlaw}
\leanverified{paper\_conclusion\_window6\_hidden\_template\_primegood\_ramanujan\_faultlaw}
记
\[
m_*:=\operatorname{lcm}(21,34,55)=39270.
\]
对每个 \(w\in X_6\)，把 centered hidden template 写成
\[
F_w=V_6(w)+D_w,
\qquad
d(w):=|D_w|\in\{2,3,4\},
\]
并定义局部归一化 Ramanujan 矩
\[
\mathcal R_w(q):=\frac1{d(w)}\sum_{\delta\in D_w}c_q(\delta)
\qquad (q\ge 1).
\]
若
\[
(q,m_*)=1,
\]
则有精确恒等式
\[
\mathcal R_w(q)
=
\frac{\varphi(q)+(d(w)-1)\mu(q)}{d(w)}
=
\mu(q)+\frac{\varphi(q)-\mu(q)}{d(w)}.
\]
\end{theorem}
\begin{proof}
由 window-\(6\) hidden fiber 的 affine Boolean-square 标准形，
\[
D_w=
\begin{cases}
\{0,34\},& d(w)=2,\\
\{0,21,34\},& d(w)=3,\\
\{0,21,34,55\},& d(w)=4.
\end{cases}
\]
若
\[
(q,m_*)=1,
\]
则 \(21,34,55\) 在 \(\ZZ/q\ZZ\) 中都为单位。于是对任意
\[
\delta\in\{21,34,55\},
\]
乘法
\[
a\longmapsto a\delta
\]
在 \((\ZZ/q\ZZ)^\times\) 上给出置换，故
\[
c_q(\delta)=c_q(1)=\mu(q).
\]
另一方面，
\[
c_q(0)=\sum_{\substack{a\bmod q\\(a,q)=1}}1=\varphi(q).
\]
因此 \(D_w\) 中唯一的零偏移贡献 \(\varphi(q)\)，其余 \(d(w)-1\) 个非零偏移都贡献 \(\mu(q)\)。于是
\[
d(w)\mathcal R_w(q)=\varphi(q)+(d(w)-1)\mu(q),
\]
化简即得结论。证毕。
\end{proof}

\begin{corollary}[单一良导子足以重建局部 hidden template]\label{cor:conclusion-window6-single-good-conductor-reconstructs-hidden-template}
\leanverified{paper\_conclusion\_window6\_single\_good\_conductor\_reconstructs\_hidden\_template}
若 \(q\ge 2\) 且
\[
(q,m_*)=1,
\]
则
\[
d(w)=\frac{\varphi(q)-\mu(q)}{\mathcal R_w(q)-\mu(q)}.
\]
特别地，若 \(p\nmid m_*\) 为素数，则
\[
\mathcal R_w(p)=\frac{p-d(w)}{d(w)},
\qquad
d(w)=\frac{p}{\mathcal R_w(p)+1}.
\]
因此，一旦 visible 基点 \(V_6(w)\) 已知，任取一个与 \(39270\) 互素的单导子 \(q\)，便可由 \(\mathcal R_w(q)\) 唯一恢复整个局部 hidden fiber
\[
F_w=V_6(w)+D_w.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-window6-hidden-template-primegood-ramanujan-faultlaw}，
\[
\mathcal R_w(q)-\mu(q)=\frac{\varphi(q)-\mu(q)}{d(w)},
\]
直接移项即得
\[
d(w)=\frac{\varphi(q)-\mu(q)}{\mathcal R_w(q)-\mu(q)}.
\]
当 \(q=p\) 为素数且 \(p\nmid m_*\) 时，
\[
\varphi(p)=p-1,\qquad \mu(p)=-1,
\]
故
\[
\mathcal R_w(p)=\frac{p-d(w)}{d(w)},
\qquad
d(w)=\frac{p}{\mathcal R_w(p)+1}.
\]
又 window-\(6\) 中 \(D_w\) 仅由
\[
d(w)\in\{2,3,4\}
\]
唯一决定，故 \(d(w)\) 一经恢复，\(D_w\) 与 \(F_w=V_6(w)+D_w\) 也随之唯一确定。证毕。
\end{proof}
