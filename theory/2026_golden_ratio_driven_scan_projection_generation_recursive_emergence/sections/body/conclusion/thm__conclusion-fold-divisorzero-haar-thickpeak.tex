\[
M:=F_{m+2},
\qquad
Z_m:=\left\{k\in \mathbb Z/(M\mathbb Z):\widehat d_m(k)=0\right\},
\qquad
\overline d_m:=\frac{2^m}{F_{m+2}},
\qquad
D_m:=\max_{x\in X_m}d_m(x).

对每个满足 \(g\mid M/2\) 的正整数 \(g\)，记

S_g(M):=
\left\{
\frac{M}{2g}+t\frac{M}{g}\pmod M:\ 0\le t\le g-1

\]

\begin{theorem}[谱零点的约数格--对合见证双重刻画]\label{thm:conclusion-fold-zero-divisor-lattice-involution-witness}
若 \(M=F_{m+2}\) 为偶数，则谱零点集合满足精确分解
\[
Z_m=
\bigcup_{\substack{1\le j\le m\\ g_j\mid M/2}}S_{g_j}(M),
\qquad
g_j:=\gcd(F_{j+1},M)=F_{\gcd(j+1,m+2)}.
\]
因此 \(Z_m\) 完全由整数 \(m+2\) 的约数格决定，而与其余微观数据无关。更强地，对任意 \(k\in Z_m\)，存在某个 \(j\in\{1,\dots,m\}\) 使得对合
\[
\iota_j:\Omega_m\to\Omega_m,
\qquad
\iota_j(\omega)=\omega\oplus e_j
\]
无固定点，且逐点满足
\[
e^{-2\pi i kN(\omega)/M}+e^{-2\pi i kN(\iota_j(\omega))/M}=0.
\]
故每个谱零点都拥有一个坐标翻转型的微观配对湮灭见证。
\end{theorem}
\begin{proof}
主稿给出零点判别
\[
\widehat d_m(k)=0
\iff
\exists\, j\in\{1,\dots,m\}\ \text{使}\ e^{-2\pi i kF_{j+1}/M}=-1,
\]
等价地，
\[
2kF_{j+1}\equiv M\pmod{2M}.
\]
对固定 \(w:=F_{j+1}\)，记 \(g:=\gcd(w,M)\)。主稿关于该同余方程的解集分类表明：若 \(g\mid M/2\)，则全部解恰为 \(S_g(M)\)；若 \(g\nmid M/2\)，则无解。于是
\[
Z_m=
\bigcup_{\substack{1\le j\le m\\ g_j\mid M/2}}S_{g_j}(M).
\]
再由 Fibonacci 最大公因子恒等式
\[
\gcd(F_a,F_b)=F_{\gcd(a,b)}
\]
得
\[
g_j=\gcd(F_{j+1},F_{m+2})=F_{\gcd(j+1,m+2)},
\]
从而 \(Z_m\) 只依赖于 \(m+2\) 的约数格。

另一方面，只要
\[
e^{-2\pi i kF_{j+1}/M}=-1,
\]
主稿已构造无固定点对合 \(\iota_j(\omega)=\omega\oplus e_j\)，并证明逐点相消恒等式
\[
e^{-2\pi i kN(\omega)/M}+e^{-2\pi i kN(\iota_j(\omega))/M}=0.
\]
对 \(\Omega_m\) 求和即得 \(\widehat d_m(k)=0\)。证毕。
\end{proof}

\begin{theorem}[模 \texorpdfstring{$2$}{2} 商上的精确 Haar 化]\label{thm:conclusion-fold-mod2-exact-haarization}
\leanverified{paper\_conclusion\_fold\_mod2\_exact\_haarization}
若
\[
m+2\equiv 0\pmod 3,
\]
则 \(M=F_{m+2}\) 为偶数，且
\[
k=\frac{M}{2}\in Z_m.
\]
因此折叠值模 \(M\) 的奇偶商分布精确均匀：对任意函数
\[
f:\mathbb Z/2\mathbb Z\to\mathbb R,
\]
都有
\[
2^{-m}\sum_{\omega\in\Omega_m}f\!\bigl(\pi_m(\omega)\bmod 2\bigr)
=
\frac{f(0)+f(1)}{2},
\]
其中
\[
\pi_m(\omega):=N(\omega)\bmod F_{m+2}.
\]
换言之，在模 \(2\) 商上，fold 推前测度恰为 Haar 测度。
\end{theorem}
\begin{proof}
当 \(m+2\equiv0\pmod3\) 时，主稿已证明 \(F_{m+2}\) 为偶数，且频率 \(k=M/2\) 处有严格零点
\[
\widehat d_m(M/2)=0.
\]
但该角色正是奇偶角色
\[
\chi_{M/2}(r)=(-1)^r.
\]
若记
\[
E_m:=\#\{\omega\in\Omega_m:\pi_m(\omega)\equiv0\pmod2\},
\qquad
O_m:=\#\{\omega\in\Omega_m:\pi_m(\omega)\equiv1\pmod2\},
\]
则
\[
E_m-O_m=\widehat d_m(M/2)=0,
\qquad
E_m+O_m=2^m.
\]
故
\[
E_m=O_m=2^{m-1}.
\]
将任意 \(f\) 写成
\[
f(0)\mathbf 1_{\mathrm{even}}+f(1)\mathbf 1_{\mathrm{odd}},
\]
便得到所示平均恒等式。证毕。
\end{proof}

\begin{theorem}[稀疏零集--厚峰偏置二分]\label{thm:conclusion-sparse-zero-thick-peak-dichotomy}
尽管谱零点由约数格强制产生，但其密度在 \(m\to\infty\) 时仍指数稀疏：
\[
\frac{|Z_m|}{F_{m+2}}
\le
\tau(m+2)\frac{F_{\lfloor (m+2)/2\rfloor}}{F_{m+2}}
=
\tau(m+2)O(\varphi^{-m/2}).
\]
与此同时，最大纤维偏置却保持常数量级的厚度：
\[
\liminf_{m\to\infty}\frac{D_m}{\overline d_m}\ge 1+C_\varphi\approx 1.2244178274047294.
\]
因此，fold 非均匀性的主导机制不能归因于“零点很多”；它必须由一个指数稀薄但振幅厚实的共振骨架承载。
\end{theorem}
\begin{proof}
零点密度上界正是主稿关于 \(Z_m\) 的指数稀疏定理：
\[
|Z_m|
\le
\tau(m+2)F_{\lfloor (m+2)/2\rfloor}.
\]
两边除以 \(F_{m+2}\) 即得第一式。另一方面，主稿关于最大纤维的常数级 \(L^\infty\) 偏置下界给出
\[
\liminf_{m\to\infty}\frac{D_m}{\overline d_m}\ge 1+C_\varphi>1.
\]
二者合并说明：精确零点的总量在比例上趋于零，而主峰相对均值的偏移却并不消失。故 fold 非均匀性呈现“稀疏零集--厚峰偏置”二分结构。证毕。
\end{proof}
