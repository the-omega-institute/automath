\begin{theorem}[秩缺口与 characteristic-\texorpdfstring{$2$}{2} 终端记忆深度的独立性]\label{thm:conclusion-resonance-window-gap-vs-mod2-terminal-memory-independent}
在共振窗 \(q=9,\dots,17\) 中，设
\[
\Delta(q):=\bigl(2\lfloor q/2\rfloor+1\bigr)-d_q
\]
为秩缺口，并记
\[
\operatorname{Per}_2^{\max}(q)
\]
为定理 \ref{thm:conclusion-resonance-window-dyadic-chambers} 给出的模 \(2\) 最终周期上界。则不存在函数
\[
F:\ZZ_{\ge 0}\to \ZZ_{\ge 1}
\]
使得
\[
\operatorname{Per}_2^{\max}(q)=F(\Delta(q))
\qquad (q=9,\dots,17).
\]
换言之，\(\Delta(q)\) 与 characteristic-\(2\) 的终端记忆深度是相互独立的两条终端轴。
\leanverified{resonance\_window\_gap\_not\_terminal\_type\_function}
\leanverified{resonance\_window\_delta2\_witnesses}
\leanverified{resonance\_window\_five\_terminal\_types\_distinct}
\leanverified{paper\_resonance\_window\_terminal\_extended}
\end{theorem}

\begin{proof}
表 \ref{tab:fold_collision_resonance_gap_delta_q_9_17} 给出
\[
\Delta(9)=\Delta(10)=\Delta(11)=\Delta(13)=\Delta(14)=2,
\qquad
\Delta(12)=0,
\]
\[
\Delta(15)=\Delta(16)=\Delta(17)=4.
\]
另一方面，定理 \ref{thm:conclusion-resonance-window-dyadic-chambers} 给出
\[
\operatorname{Per}_2^{\max}(q)\mid 4 \quad (q=9,10),
\]
\[
\operatorname{Per}_2^{\max}(q)\mid 8 \quad (q=11,\dots,16),
\]
\[
\operatorname{Per}_2^{\max}(q)\mid 16 \quad (q=17).
\]
因此同一缺口层 \(\Delta=2\) 内同时出现周期级别 \(4\) 与 \(8\)，而同一缺口层 \(\Delta=4\) 内同时出现 \(8\) 与 \(16\)。故 \(\operatorname{Per}_2^{\max}(q)\) 不可能仅由 \(\Delta(q)\) 决定。证毕。
\end{proof}

\begin{corollary}[共振窗至少具有五种有限特征终端型]\label{cor:conclusion-resonance-window-five-terminal-types}
若以有序对
\[
\bigl(\Delta(q),\operatorname{Per}_2^{\max}(q)\bigr)
\]
作为共振窗 \(q=9,\dots,17\) 的有限特征终端型，则至少出现下列五种互异类型：
\[
(2,4),\qquad
(2,8),\qquad
(0,8),\qquad
(4,8),\qquad
(4,16).
\]
更具体地，
\[
q=9,10 \leadsto (2,4),\qquad
q=11,13,14 \leadsto (2,8),\qquad
q=12 \leadsto (0,8),
\]
\[
q=15,16 \leadsto (4,8),\qquad
q=17 \leadsto (4,16).
\]
\leanverified{resonance\_window\_five\_types\_pairwise\_distinct}
\end{corollary}

\begin{proof}
这只是定理 \ref{thm:conclusion-resonance-window-gap-vs-mod2-terminal-memory-independent} 证明中两张已审计表格的逐项合并。五种不同的有序对均在共振窗内实际出现，故至少有五种互异的有限特征终端型。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$q=13,15$}{q=13,15} 的 \texorpdfstring{$3$}{3}-进三相分裂与任意 \texorpdfstring{$3^\ell$}{3^l} 周期壳]\label{thm:conclusion-q13-q15-z3-threephase-periodic-shells}
\leanverified{paper\_conclusion\_q13\_q15\_z3\_threephase\_periodic\_shells}
在定理 \ref{thm:pom-moment-resonance} 的递推生效区间内，对 \(q\in\{13,15\}\) 与任意 \(\ell\ge 1\)，序列
\[
S_q(m)\pmod{3^\ell}
\]
在去掉至多 \(2\ell\) 项前导之后进入纯周期壳，其周期满足
\[
\Pi_{q,\ell}\mid 2\cdot 3^{\,\ell+\lceil\log_3(5\ell)\rceil-1}.
\]
更精确地，存在唯一 Hensel 分解
\[
P_q(\lambda)=U_{0,q}(\lambda)\,U_{+,q}(\lambda)\,U_{-,q}(\lambda)
\qquad\text{于 }\ZZ_3[\lambda],
\]
满足
\[
U_{0,q}\equiv \lambda^2,\qquad
U_{+,q}\equiv (\lambda-1)^4,\qquad
U_{-,q}\equiv (\lambda+1)^5\pmod 3.
\]
相应地，\(3\)-进解模分裂为
\[
\mathscr M_q^{(3)}
=
\{f\in \ZZ_3^{\NN}:\ P_q(E)f=0\}
=
\mathscr M_{q,0}^{(3)}\oplus
\mathscr M_{q,+}^{(3)}\oplus
\mathscr M_{q,-}^{(3)}.
\]
其中幂零部分在模 \(3^\ell\) 下至多 \(2\ell\) 步后消失，而剩余 semisimple 局部相位仅来自 \(+1\) 与 \(-1\) 两个扇区；不存在新的扩域相位通道。
\end{theorem}
\begin{proof}
由注 \ref{rem:pom-charpoly-mod3-selection} 与表 \ref{tab:fold_collision_charpoly_mod3_selection_q13_15}，
\[
P_q(\lambda)\equiv \lambda^2(\lambda-1)^4(\lambda+1)^5\pmod 3
\qquad (q\in\{13,15\}).
\]
三因子在 \(\FF_3[\lambda]\) 中两两互素，故由 Hensel 引理可唯一提升为
\[
P_q=U_{0,q}U_{+,q}U_{-,q}
\qquad\text{于 }\ZZ_3[\lambda],
\]
并保持上述模 \(3\) 同余。由中国剩余定理得到解模的直和分裂
\[
\mathscr M_q^{(3)}
=
\mathscr M_{q,0}^{(3)}\oplus
\mathscr M_{q,+}^{(3)}\oplus
\mathscr M_{q,-}^{(3)}.
\]

在 \(\mathscr M_{q,0}^{(3)}\) 上，写
\[
U_{0,q}(E)=E^2+3R_0(E).
\]
于是
\[
E^2\mathscr M_{q,0}^{(3)}\subset 3\,\mathscr M_{q,0}^{(3)},
\qquad
E^{2\ell}\mathscr M_{q,0}^{(3)}\subset 3^\ell\,\mathscr M_{q,0}^{(3)}.
\]
故幂零分量在模 \(3^\ell\) 下至多 \(2\ell\) 步后消失。

在 \(\mathscr M_{q,+}^{(3)}\) 上令
\[
D_+:=E-1.
\]
由
\[
U_{+,q}(E)=D_+^4+3R_+(E)
\]
得
\[
D_+^{4\ell}\mathscr M_{q,+}^{(3)}\subset 3^\ell\,\mathscr M_{q,+}^{(3)}.
\]
若记
\[
s_+:=\ell+\lceil\log_3(4\ell)\rceil-1,
\]
则对 \(1\le t<4\ell\) 有
\[
v_3\!\binom{3^{s_+}}{t}=s_+-v_3(t)\ge \ell.
\]
故在模 \(3^\ell\) 下，
\[
E^{3^{s_+}}=(1+D_+)^{3^{s_+}}\equiv 1
\qquad\text{于 }\mathscr M_{q,+}^{(3)}.
\]

在 \(\mathscr M_{q,-}^{(3)}\) 上令
\[
D_-:=E+1.
\]
由
\[
U_{-,q}(E)=D_-^5+3R_-(E)
\]
得
\[
D_-^{5\ell}\mathscr M_{q,-}^{(3)}\subset 3^\ell\,\mathscr M_{q,-}^{(3)}.
\]
若记
\[
s_-:=\ell+\lceil\log_3(5\ell)\rceil-1,
\]
则同样有
\[
v_3\!\binom{3^{s_-}}{t}=s_--v_3(t)\ge \ell
\qquad (1\le t<5\ell).
\]
于是
\[
E^{3^{s_-}}=(-1+D_-)^{3^{s_-}}\equiv -1
\qquad\text{于 }\mathscr M_{q,-}^{(3)}.
\]
故 \(\mathscr M_{q,-}^{(3)}\) 的周期整除 \(2\cdot 3^{s_-}\)，而 \(\mathscr M_{q,+}^{(3)}\) 的周期整除 \(3^{s_+}\)。又因 \(s_-\ge s_+\)，合并三部分即得
\[
\Pi_{q,\ell}\mid 2\cdot 3^{s_-}
=
2\cdot 3^{\,\ell+\lceil\log_3(5\ell)\rceil-1}.
\]
最后，由于 semisimple 局部因子只来自 \((\lambda-1)\) 与 \((\lambda+1)\)，故 \(3\)-进侧不存在任何新的扩域相位通道。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$q=13,15$}{q=13,15} 的混合 \texorpdfstring{$(2^k3^\ell)$}{(2^k3^l)} 周期壳]\label{thm:conclusion-q13-q15-mixed-2k3l-periodic-shells}

S_q(m)\pmod{2^k3^\ell}

在递推生效区间内最终周期化。其前导长度可取

L_{q;k,\ell}\le \max(3k,2\ell),

\Theta_{q;k,\ell}\mid
\operatorname{lcm}\!\Bigl(

2\cdot 3^{\,\ell+\lceil\log_3(5\ell)\rceil-1}

\]
特别地，当 \(k=\ell=n\) 时，
\[
\Theta_{q;n,n}\mid
\operatorname{lcm}\!\Bigl(
2^{\,n+\lceil\log_2(8n)\rceil-1},
\;
2\cdot 3^{\,n+\lceil\log_3(5n)\rceil-1}
\Bigr).

\end{theorem}
\leanverified{paper\_conclusion\_q13\_q15\_mixed\_2k3l\_periodic\_shells}
\begin{proof}

\[
2^{\,k+\lceil\log_2(8k)\rceil-1}.
\]
另一方面，定理 \ref{thm:conclusion-q13-q15-z3-threephase-periodic-shells} 给出模 \(3^\ell\) 的前导长度至多 \(2\ell\)，而周期整除
\[
2\cdot 3^{\,\ell+\lceil\log_3(5\ell)\rceil-1}.

\[
L_{q;k,\ell}\le \max(3k,2\ell)

并施用中国剩余分解

\begin{theorem}[\texorpdfstring{$q=13,15$}{q=13,15} 的局部相位排斥与 mixed-residue 尾复杂度的幂零厚化性]\label{thm:conclusion-q13-q15-local-phase-exclusion-6adic}
\leanverified{paper\_conclusion\_q13\_q15\_local\_phase\_exclusion\_6adic}
对 \(q\in\{13,15\}\) 与任意 \(n\ge 1\)，模 \(6^n\) 的最终尾动力学不需要任何新的局部 semisimple 相位通道：其 \(2\)-进侧仅保留 \((-1)\)-扇区上的 unipotent 厚化，其 \(3\)-进侧仅保留 \(\pm1\) 两个扇区上的厚化，而全部复杂性都来自这些既有扇区的幂零加厚，而非新的不可约相位因子或新的扩域模态。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-resonance-window-z2-hensel-tail-periodicity}，对 \(q=13,15\) 的 dyadic 尾动力学，唯一的局部单位因子来自
\[
V_q(\lambda)\equiv (\lambda+1)^8\pmod 2,
\]
故 \(2\)-进侧不存在第二个单位扇区，也不存在任何更高次数的局部不可约因子。另一方面，定理 \ref{thm:conclusion-q13-q15-z3-threephase-periodic-shells} 说明 \(3\)-进侧的全部局部相位只来自
\[
\lambda-1,
\qquad
\lambda+1,
\]
而幂零因子 \(\lambda^2\) 在有限步后消失。于是，对任意 \(n\ge 1\)，模 \(6^n\) 的 mixed-residue 尾复杂度只能由这两条局部链的 unipotent 厚化与 CRT 拼接产生，而不需要任何新的局部 semisimple 通道。证毕。
\end{proof}
