\subsection{最大非可缩纤维的模 \texorpdfstring{$6$}{6} 密度与六倍窗谱零块的黄金耦合}\label{subsec:conclusion-max-noncontractible-fiber-zero-block-golden-coupling}
沿用定义 \ref{def:pom-max-noncontractible-fiber}、定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}、定理 \ref{thm:pom-max-noncontractible-fiber-exponent-mod6-ratio} 与推论 \ref{cor:fold-zero-half-index-multiple6} 的记号。
在该框架下，纤维同伦判别所给出的模 \(6\) 顶层相图还可继续压缩为一组全局后果：最大纤维何时仍保持非可缩拓扑可见性，不仅具有精确的自然密度，而且其相对代价常数、指数尺度守恒以及与六倍窗谱零块的乘法耦合都可以完全显式化。

\begin{theorem}[最大纤维同伦可见性的自然密度]\label{thm:conclusion-maxfiber-homotopy-visibility-density}
\leanverified{paper\_conclusion\_maxfiber\_homotopy\_visibility\_density}
记
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
\widetilde D_m:=\max\{d_m(x):x\in X_m,\ |K_x|\not\simeq *\}.
\]
定义
\[
\mathcal A:=\{m\ge 17:\ \widetilde D_m=D_m\}.
\]
则有精确描述
\[
\boxed{\;
\mathcal A
=
\{m\ge 17:\ m\equiv 0,4\pmod 6\}.\;}
\]
特别地，\(\mathcal A\) 的自然密度存在且满足
\[
\boxed{\;
\lim_{N\to\infty}\frac1N\,\#\bigl(\mathcal A\cap [1,N]\bigr)=\frac13.\;}
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，
\[
\widetilde D_m=
\begin{cases}
D_m,& m\equiv 0,4\pmod 6,\\[2pt]
D_m^{(2)},& m\equiv 1,5\pmod 6,\\[2pt]
D_m^{(3)},& m\equiv 2,3\pmod 6.
\end{cases}
\qquad (m\ge 17).
\]
故
\[
\mathcal A=\{m\ge 17:\ m\equiv 0,4\pmod 6\}.
\]
模 \(6\) 的两个剩余类在自然数中的密度为 \(2/6\)，而去掉有限多个初始项不改变自然密度，遂得
\[
\lim_{N\to\infty}\frac1N\,\#\bigl(\mathcal A\cap [1,N]\bigr)=\frac{2}{6}=\frac13.
\]
证毕。
\end{proof}

\begin{theorem}[极大纤维同伦型的全同模 \texorpdfstring{$6$}{6} 相律]\label{thm:conclusion-maxfiber-uniphase-mod6-law}
对每个 \(m\ge 17\)，下列三条互相等价：
\begin{enumerate}
  \item 存在 \(x\in X_m\) 使 \(d_m(x)=D_m\) 且 \(|K_x|\not\simeq *\)；
  \item 对一切满足 \(d_m(x)=D_m\) 的 \(x\in X_m\)，都有 \(|K_x|\not\simeq *\)；
  \item \(m\equiv 0,4\pmod 6\)。
\end{enumerate}
等价地，
\[
\boxed{\;
\forall x\in X_m,\ d_m(x)=D_m
\quad\Longrightarrow\quad
|K_x|\not\simeq *
\iff
m\equiv 0,4\pmod 6.\;}
\]
因此，在每个固定分辨率层上，全部极大纤维的同伦型不存在混相：它们要么同时可缩，要么同时非可缩。
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence},
\[
|K_x|\not\simeq *
\iff
d_m(x)\ \text{为奇数}.
\]
因此，对任意满足 \(d_m(x)=D_m\) 的极大纤维，其同伦型仅由同一个整数 \(D_m\) 的奇偶决定；特别地，一旦存在一个非可缩极大纤维，则全部极大纤维都非可缩，反之亦然。于是第 \(1\) 条与第 \(2\) 条等价。

另一方面，第 \(1\) 条等价于
\[
\widetilde D_m=D_m,
\]
而定理 \ref{thm:conclusion-maxfiber-homotopy-visibility-density} 已证明这恰在
\[
m\equiv 0,4\pmod 6
\]
时成立。故三条互相等价，所示全同相律亦随之成立。证毕。
\end{proof}

\begin{theorem}[非可缩最大纤维的模 \texorpdfstring{$6$}{6} 极限比例与黄金常数]\label{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios}
沿用定理 \ref{thm:conclusion-maxfiber-homotopy-visibility-density} 的记号。则沿模 \(6\) 的四类剩余类有精确极限
\[
\boxed{\;
\lim_{\substack{m\to\infty\\ m\equiv 0,4\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}=1,\;}
\]
\[
\boxed{\;
\lim_{\substack{m\to\infty\\ m\equiv 1,5\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=
1-\frac{1}{2\varphi^{5}},\;}
\]
\[
\boxed{\;
\lim_{\substack{m\to\infty\\ m\equiv 2\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=
1-\varphi^{-5},\;}
\]
\[
\boxed{\;
\lim_{\substack{m\to\infty\\ m\equiv 3\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=
1-\frac12\bigl(\varphi^{-5}+\varphi^{-9}\bigr).\;}
\]
等价地，相对损失
\[
1-\frac{\widetilde D_m}{D_m}
\]
在四类扇区上分别趋于
\[
0,\qquad
\frac{1}{2\varphi^5},\qquad
\varphi^{-5},\qquad
\frac12\bigl(\varphi^{-5}+\varphi^{-9}\bigr).
\]
\end{theorem}
\begin{proof}
这四个极限正是定理 \ref{thm:pom-max-noncontractible-fiber-exponent-mod6-ratio} 的四条模 \(6\) 比值公式。
对每一类剩余类从 \(1\) 中减去该极限，即得相应的相对损失常数。证毕。
\end{proof}

\begin{corollary}[非可缩最大纤维的指数率守恒]\label{cor:conclusion-max-noncontractible-fiber-exponential-rate}
沿用上述记号，则有
\[
\boxed{\;
\widetilde D_m=\Theta(D_m).\;}
\]
特别地，
\[
\boxed{\;
\lim_{m\to\infty}\frac1m\log \widetilde D_m
=
\frac12\log\varphi.\;}
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-max-noncontractible-fiber-mod6-golden-ratios}，沿每个模 \(6\) 剩余类，
\(
\widetilde D_m/D_m
\)
都收敛到严格正的常数。
因此存在常数 \(c,C>0\) 与 \(m_0\)，使 \(m\ge m_0\) 时
\[
cD_m\le \widetilde D_m\le CD_m,
\]
故 \(\widetilde D_m=\Theta(D_m)\)。
再由定理 \ref{thm:pom-max-fiber} 的闭式，
\[
\lim_{m\to\infty}\frac1m\log D_m=\frac12\log\varphi,
\]
结合上式即得
\[
\lim_{m\to\infty}\frac1m\log \widetilde D_m=\frac12\log\varphi.
\]
证毕。
\end{proof}

\begin{theorem}[六倍窗中的谱--拓扑黄金乘积下界]\label{thm:conclusion-window6-spectral-topological-golden-product}
记
\[
\mathcal Z_m:=\{k\in \ZZ/(F_{m+2}\ZZ):\ \widehat d_m(k)=0\}
\]
为谱零点集合。若
\[
m+2\equiv 0\pmod 6,
\]
则有乘积下界
\[
\boxed{\;
\widetilde D_m\,|\mathcal Z_m|
\ge
F_{(m+2)/2}\,F_{(m+4)/2}.\;}
\]
进一步，
\[
\boxed{\;
\liminf_{\substack{m\to\infty\\ m+2\equiv 0\ (\mathrm{mod}\ 6)}}
\frac{\widetilde D_m\,|\mathcal Z_m|}{F_{m+2}}
\ge
\frac{\varphi}{\sqrt5}.\;}
\]
\end{theorem}
\begin{proof}
设
\[
d:=\frac{m+2}{2}.
\]
由推论 \ref{cor:fold-zero-half-index-multiple6}，
\[
|\mathcal Z_m|\ge F_d=F_{(m+2)/2}.
\]
同一条件下 \(m\equiv 4\pmod 6\)，故定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase} 给出
\[
\widetilde D_m=D_m.
\]
又由定理 \ref{thm:pom-max-fiber}，若 \(m=2k\) 则
\[
D_m=F_{k+2}.
\]
而 \(d=(m+2)/2=k+1\)，从而
\[
\widetilde D_m=D_m=F_{k+2}=F_{d+1}=F_{(m+4)/2}.
\]
于是
\[
\widetilde D_m\,|\mathcal Z_m|
\ge
F_dF_{d+1}
=
F_{(m+2)/2}F_{(m+4)/2},
\]
得到第一条。

再除以
\[
F_{m+2}=F_{2d}.
\]
由恒等式 \(F_{2d}=F_dL_d\)（其中 \(L_d\) 为 Lucas 数），可得
\[
\frac{\widetilde D_m\,|\mathcal Z_m|}{F_{m+2}}
\ge
\frac{F_dF_{d+1}}{F_{2d}}
=
\frac{F_{d+1}}{L_d}.
\]
由 Binet 公式，
\[
F_{d+1}\sim \frac{\varphi^{d+1}}{\sqrt5},
\qquad
L_d\sim \varphi^d,
\]
故
\[
\frac{F_{d+1}}{L_d}\longrightarrow \frac{\varphi}{\sqrt5}.
\]
因此
\[
\liminf_{\substack{m\to\infty\\ m+2\equiv 0\ (\mathrm{mod}\ 6)}}
\frac{\widetilde D_m\,|\mathcal Z_m|}{F_{m+2}}
\ge
\frac{\varphi}{\sqrt5}.
\]
证毕。
\end{proof}
