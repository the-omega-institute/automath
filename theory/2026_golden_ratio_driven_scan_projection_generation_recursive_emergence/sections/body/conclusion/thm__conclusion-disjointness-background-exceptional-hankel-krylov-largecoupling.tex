\paragraph{背景节点的整块保留、Hankel 笔刻画与大耦合双相分裂}
沿用本小节关于两参数不交核的记号。特别地，
\[
B^{(q)}(a,b)=dD+bJ,\qquad d:=a-b,
\]
并记
\[
\mu_i^{(q)}:=\varphi^{\,q-i}\psi^i\qquad(0\le i\le q).
\]
又由定理 \ref{thm:conclusion-disjointness-krylov-hankel-symmetric-subspace}，对称扇区
\[
\mathcal H_{\mathrm{sym}}
:=
\{f:\mathcal P([q])\to \RR:\ f(U)=f(V)\ \text{whenever }|U|=|V|\}
\]
满足 \(\dim \mathcal H_{\mathrm{sym}}=q+1\)，且
\[
u_r:=D^r\mathbf 1\qquad(0\le r\le q)
\]
构成其一组基。

\begin{theorem}[全特征多项式的背景--例外分裂]\label{thm:conclusion-disjointness-full-characteristic-polynomial-splitting}
\leanverified{paper\_conclusion\_disjointness\_full\_characteristic\_polynomial\_splitting}
记
\[
\chi_q(\lambda):=\det\!\bigl(\lambda I-B^{(q)}(a,b)\bigr).
\]
则有精确分解
\[
\chi_q(\lambda)
=
\prod_{i=0}^{q}\bigl(\lambda-d\mu_i^{(q)}\bigr)^{\binom qi-1}\,
P_{q,a,b}(\lambda).
\]
特别地，除对称扇区 \(\mathcal H_{\mathrm{sym}}\) 贡献的 \(q+1\) 个例外根之外，其余全部谱严格保留为背景节点
\[
d\mu_0^{(q)},\dots,d\mu_q^{(q)}
\]
并按多重度 \(\binom qi-1\) 出现。
\end{theorem}
\begin{proof}
由于 \(D=K^{\otimes q}\) 与 \(J=\mathbf 1\mathbf 1^{\mathsf T}\) 都与 \(S_q\) 的置换作用交换，\(\mathcal H_{\mathrm{sym}}\) 及其正交补 \(\mathcal H_{\mathrm{sym}}^\perp\) 都在 \(D\) 与 \(J\) 下不变。若 \(x\in \mathcal H_{\mathrm{sym}}^\perp\)，则
\[
Jx=\mathbf 1(\mathbf 1^{\mathsf T}x)=0,
\]
因为 \(\mathbf 1\in \mathcal H_{\mathrm{sym}}\)。故
\[
B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}^\perp}
=
d\,D\big|_{\mathcal H_{\mathrm{sym}}^\perp}.
\]

另一方面，\(D=K^{\otimes q}\) 的全谱为 \(\{\mu_i^{(q)}\}_{i=0}^{q}\)，其代数重数分别为 \(\binom qi\)。在对称扇区上，\(D\big|_{\mathcal H_{\mathrm{sym}}}\cong \mathrm{Sym}^q(K)\)，故每个 \(\mu_i^{(q)}\) 在 \(\mathcal H_{\mathrm{sym}}\) 中恰出现一次。因此在 \(\mathcal H_{\mathrm{sym}}^\perp\) 上，\(d\mu_i^{(q)}\) 的重数恰为 \(\binom qi-1\)。

再由世俗方程的定义，\(B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}}\) 的特征多项式正是 \(P_{q,a,b}\)。将两部分相乘即得结论。证毕。
\end{proof}

\begin{corollary}[背景节点与例外节点的严格分离]\label{cor:conclusion-disjointness-background-exceptional-coprime-separation}
\leanverified{paper\_conclusion\_disjointness\_background\_exceptional\_coprime\_separation}
若 \(bd\neq 0\)，则
\[
\gcd\!\bigl(P_{q,a,b},G_{q,d}\bigr)=1,
\qquad
G_{q,d}(\lambda):=\prod_{i=0}^{q}\bigl(\lambda-d\mu_i^{(q)}\bigr).
\]
因此每个背景节点 \(d\mu_i^{(q)}\) 在 \(\chi_q\) 中的精确重数恰为
\[
\binom qi-1,
\]
不会再被例外因子额外碰撞。
\end{corollary}
\begin{proof}
命题 \ref{prop:conclusion-disjointness-pairwise-resultant-rigidity} 已对每一对共轭背景节点对应的二次因子
\[
Q_{q,i}(\lambda)=\lambda^2-d(-1)^iL_{q-2i}\lambda+d^2(-1)^q
\]
给出显式非零结果式
\[
\Res\bigl(P_{q,a,b},Q_{q,i}\bigr)\neq 0
\qquad (bd\neq 0).
\]
若 \(q\) 为偶数，同一命题还给出中心一次因子 \(\lambda-d(-1)^{q/2}\) 处的显式非零取值。故 \(P_{q,a,b}\) 与 \(G_{q,d}\) 没有公共根，从而互素。

再由定理 \ref{thm:conclusion-disjointness-full-characteristic-polynomial-splitting}，\(\chi_q\) 中背景节点的全部重数正由 \(\prod_i(\lambda-d\mu_i^{(q)})^{\binom qi-1}\) 给出，故其精确重数为 \(\binom qi-1\)。证毕。
\end{proof}

\begin{theorem}[例外迹矩的阶 \texorpdfstring{$q+1$}{q+1} 递推刚性]\label{thm:conclusion-disjointness-exceptional-trace-moment-recurrence}
定义
\[
\Theta_q(m):=
\Tr\!\bigl((B^{(q)}(a,b))^m\bigr)
-d^m\sum_{i=0}^{q}\Bigl(\binom qi-1\Bigr)\bigl(\mu_i^{(q)}\bigr)^m
\qquad (m\ge 0).
\]
则
\[
\Theta_q(m)=\Tr\!\Bigl(\bigl(B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}}\bigr)^m\Bigr),
\]
并且序列 \(m\mapsto \Theta_q(m)\) 满足由 \(P_{q,a,b}\) 给出的阶 \((q+1)\) 常系数线性递推。
\leanverified{paper\_conclusion\_disjointness\_exceptional\_trace\_moment\_recurrence}
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-disjointness-full-characteristic-polynomial-splitting}，
\[
\Tr\!\bigl((B^{(q)}(a,b))^m\bigr)
=
d^m\sum_{i=0}^{q}\Bigl(\binom qi-1\Bigr)\bigl(\mu_i^{(q)}\bigr)^m
\,
\Tr\!\Bigl(\bigl(B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}}\bigr)^m\Bigr),
\]
故第一式成立。

再由 Cayley--Hamilton 定理，
\[
P_{q,a,b}\!\Bigl(B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}}\Bigr)=0.
\]
将其右乘 \(\bigl(B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}}\bigr)^m\) 并取迹，即得 \(\Theta_q(m)\) 满足以 \(P_{q,a,b}\) 为递推多项式的线性关系。由于 \(\deg P_{q,a,b}=q+1\)，递推阶数为 \(q+1\)。证毕。
\end{proof}

\begin{corollary}[例外谱的 \texorpdfstring{$(q+1)$}{q+1} 矩完全恢复]\label{cor:conclusion-disjointness-exceptional-spectrum-reconstruction-from-qplus1-traces}
\leanverified{paper\_conclusion\_disjointness\_exceptional\_spectrum\_reconstruction\_from\_qplus1\_traces}
仅给定
\[
\Tr\!\bigl((B^{(q)}(a,b))^m\bigr)\qquad (1\le m\le q+1),
\]
便可唯一恢复整个例外特征因子 \(P_{q,a,b}\)，无需求出任何 \(2^q\times 2^q\) 实现矩阵的全部特征值。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-disjointness-exceptional-trace-moment-recurrence}，对 \(1\le m\le q+1\) 可显式减去背景项
\[
d^m\sum_{i=0}^{q}\Bigl(\binom qi-1\Bigr)\bigl(\mu_i^{(q)}\bigr)^m
\]
而得到例外 power sums
\[
\Theta_q(m)=\sum_{j=0}^{q}\Lambda_j^m,
\]
其中 \(\Lambda_0,\dots,\Lambda_q\) 为 \(P_{q,a,b}\) 的根（按重数计）。Newton 恒等式将前 \(q+1\) 个 power sums 与次数 \(q+1\) 首一多项式的全部 elementary symmetric coefficients 一一对应，故 \(P_{q,a,b}\) 被唯一恢复。证毕。
\end{proof}

\begin{theorem}[例外特征因子的 Hankel--Gram 广义特征笔表示]\label{thm:conclusion-disjointness-exceptional-hankel-gram-pencil}
定义 Hankel 矩阵与边界向量
\[
H_q:=\bigl(F_{r+s+3}^{\,q}\bigr)_{0\le r,s\le q},
\qquad
H_q^{+}:=\bigl(F_{r+s+4}^{\,q}\bigr)_{0\le r,s\le q},
\qquad
h_q:=\bigl(F_{r+3}^{\,q}\bigr)_{0\le r\le q}.
\]
则成立精确公式
\[
P_{q,a,b}(\lambda)
=
\frac{\det\!\bigl(\lambda H_q-dH_q^{+}-b\,h_qh_q^{\mathsf T}\bigr)}{\det H_q}.
\]
因此，例外谱恰是广义特征笔
\[
\bigl(dH_q^{+}+b\,h_qh_q^{\mathsf T},\,H_q\bigr)
\]
的全部广义特征值。
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-disjointness-krylov-hankel-symmetric-subspace}，\(\{u_0,\dots,u_q\}\) 是 \(\mathcal H_{\mathrm{sym}}\) 的一组基，且其 Gram 矩阵为
\[
\bigl(\langle u_r,u_s\rangle\bigr)_{r,s}=H_q.
\]
又
\[
\langle u_r,Du_s\rangle=\langle u_r,u_{s+1}\rangle=F_{r+s+4}^{\,q},
\]
故 \(D\big|_{\mathcal H_{\mathrm{sym}}}\) 在该非正交基下的笔矩阵为 \(H_q^{+}\)。

另一方面，
\[
Ju_s=(\mathbf 1^{\mathsf T}u_s)\mathbf 1=F_{s+3}^{\,q}u_0,
\]
从而
\[
\langle u_r,Ju_s\rangle
=
F_{r+3}^{\,q}F_{s+3}^{\,q},
\]
即 \(J\big|_{\mathcal H_{\mathrm{sym}}}\) 在该基下对应矩阵为 \(h_qh_q^{\mathsf T}\)。

因此 \(B^{(q)}(a,b)\big|_{\mathcal H_{\mathrm{sym}}}\) 的广义特征方程为
\[
\det\!\bigl(\lambda H_q-dH_q^{+}-b\,h_qh_q^{\mathsf T}\bigr)=0.
\]
再除以 \(\det H_q\) 并利用 \(\det H_q\neq 0\)（定理 \ref{thm:conclusion-disjointness-krylov-hankel-symmetric-subspace}），便得到所示特征多项式表达。证毕。
\end{proof}
\leanverified{paper\_conclusion\_disjointness\_exceptional\_hankel\_gram\_pencil}

\begin{theorem}[割线函数的 Hankel--resolvent 标量化]\label{thm:conclusion-disjointness-secant-hankel-resolvent-scalarization}
当
\[
\lambda\notin\{d\mu_i^{(q)}:0\le i\le q\}
\]
时，成立精确恒等式
\[
\mathcal F_q(\lambda)=h_q^{\mathsf T}(\lambda H_q-dH_q^{+})^{-1}h_q,
\]
并且
\[
P_{q,a,b}(\lambda)
=
G_{q,d}(\lambda)\Bigl(1-b\,h_q^{\mathsf T}(\lambda H_q-dH_q^{+})^{-1}h_q\Bigr).
\]
因此 \(\lambda\) 为例外根当且仅当
\[
1=b\,h_q^{\mathsf T}(\lambda H_q-dH_q^{+})^{-1}h_q.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:conclusion-disjointness-exceptional-hankel-gram-pencil}，
\[
P_{q,a,b}(\lambda)
=
\frac{\det\!\bigl(\lambda H_q-dH_q^{+}-b\,h_qh_q^{\mathsf T}\bigr)}{\det H_q}.
\]
对矩阵行列式引理
\[
\det(M-buu^{\mathsf T})
=
\det(M)\bigl(1-b\,u^{\mathsf T}M^{-1}u\bigr)
\]
取
\[
M=\lambda H_q-dH_q^{+},
\qquad
u=h_q,
\]
得到
\[
P_{q,a,b}(\lambda)
=
\frac{\det(\lambda H_q-dH_q^{+})}{\det H_q}
\Bigl(1-b\,h_q^{\mathsf T}(\lambda H_q-dH_q^{+})^{-1}h_q\Bigr).
\]
当 \(b=0\) 时，左端退化为 \(P_{q,a,0}(\lambda)=G_{q,d}(\lambda)\)，故
\[
\det(\lambda H_q-dH_q^{+})=\det(H_q)\,G_{q,d}(\lambda).
\]
代回并与定义
\[
P_{q,a,b}(\lambda)=G_{q,d}(\lambda)\bigl(1-b\mathcal F_q(\lambda)\bigr)
\]
比较，即得
\[
\mathcal F_q(\lambda)=h_q^{\mathsf T}(\lambda H_q-dH_q^{+})^{-1}h_q.
\]
其余结论立即推出。证毕。
\end{proof}

\begin{theorem}[软核动力学的双向时间对称 Krylov 压缩]\label{thm:conclusion-softcore-bi-infinite-symmetric-krylov-compression}
记
\[
T^{(q)}:=\frac12(D+J).
\]
则对任意整数 \(m\ge 1\)，都存在实矩阵
\[
C_m^{+}(q),\ \widehat C_m^{-}(q)\in M_{q+1}(\RR)
\]
使得
\[
\bigl(T^{(q)}\bigr)^m
=
2^{-m}D^m+U_q\,C_m^{+}(q)\,U_q^{\mathsf T},
\]
\[
\bigl(T^{(q)}\bigr)^{-m}
=
2^{m}D^{-m}+U_q\,\widehat C_m^{-}(q)\,U_q^{\mathsf T}.
\]
特别地，
\[
\rank\!\Bigl(\bigl(T^{(q)}\bigr)^{\pm m}-2^{\mp m}D^{\pm m}\Bigr)\le q+1,
\]
且双向时间上的全部例外动力学都严格局限于同一 \((q+1)\)-维对称 Krylov 扇区 \(\mathcal H_{\mathrm{sym}}\) 内。
\end{theorem}
\begin{proof}
先看正时间。由于
\[
D+J=B^{(q)}(2,1),
\]
定理 \ref{thm:conclusion-disjointness-fixedq-symmetric-krylov-sector} 在 \((a,b)=(2,1)\) 时给出
\[
(D+J)^m=D^m+U_q\,\widetilde C_m(q)\,U_q^{\mathsf T}
\]
对某个 \(\widetilde C_m(q)\in M_{q+1}(\RR)\) 成立。两边除以 \(2^m\) 即得第一式。

再看负时间。由定理 \ref{thm:conclusion-softcore-inverse-power-entrywise} 的证明，
\[
(D+J)^{-1}=D^{-1}-\frac12\,e_{\varnothing}e_{\varnothing}^{\mathsf T},
\]
故 \((D+J)^{-m}\) 可按 \(D^{-1}\) 与 \(e_{\varnothing}e_{\varnothing}^{\mathsf T}\) 的字展开。任一含秩一因子的项都收缩为若干向量 \(D^{-r}\mathbf 1\) 的左右张量积。由于 \(\mathcal H_{\mathrm{sym}}\) 在 \(D^{-1}\) 下不变且 \(\mathbf 1\in \mathcal H_{\mathrm{sym}}\)，所有 \(D^{-r}\mathbf 1\) 都属于 \(\mathcal H_{\mathrm{sym}}\)；再由定理 \ref{thm:conclusion-disjointness-krylov-hankel-symmetric-subspace}，它们全部落在 \(\operatorname{span}\{u_0,\dots,u_q\}\) 中。故存在 \(\widehat C_m(q)\in M_{q+1}(\RR)\) 使
\[
(D+J)^{-m}=D^{-m}+U_q\,\widehat C_m(q)\,U_q^{\mathsf T}.
\]
再乘以 \(2^m\) 即得第二式。

秩界由两式右端的秩一览无遗；而 \(\operatorname{im}(U_q)=\mathcal H_{\mathrm{sym}}\) 已见于定理 \ref{thm:conclusion-disjointness-krylov-hankel-symmetric-subspace}，故全部修正项都局限在该扇区内。证毕。
\end{proof}

\begin{corollary}[对称扇区之外的双向完全不可见性]\label{cor:conclusion-softcore-bi-infinite-exact-invisibility-off-symmetric-sector}
\leanverified{paper\_conclusion\_softcore\_bi\_infinite\_exact\_invisibility\_off\_symmetric\_sector}
对任意 \(x\in \mathcal H_{\mathrm{sym}}^\perp\) 与任意整数 \(m\ge 1\)，有
\[
\bigl(T^{(q)}\bigr)^m x=2^{-m}D^m x,
\qquad
\bigl(T^{(q)}\bigr)^{-m}x=2^{m}D^{-m}x.
\]
等价地，对任意 \(y\in \RR^{2^q}\) 都有
\[
\langle y,\bigl(T^{(q)}\bigr)^m x\rangle
=
2^{-m}\langle y,D^m x\rangle,
\qquad
\langle y,\bigl(T^{(q)}\bigr)^{-m}x\rangle
=
2^{m}\langle y,D^{-m}x\rangle.
\]
因此，双向时间轴上的全部例外修正，对 \(\mathcal H_{\mathrm{sym}}^\perp\) 都是精确不可见的。
\end{corollary}
\begin{proof}
由定理 \ref{thm:conclusion-softcore-bi-infinite-symmetric-krylov-compression}，
\[
\bigl(T^{(q)}\bigr)^{\pm m}-2^{\mp m}D^{\pm m}
=
U_q C U_q^{\mathsf T}
\]
对某个矩阵 \(C\) 成立，而 \(\operatorname{im}(U_q)=\mathcal H_{\mathrm{sym}}\)。若 \(x\in \mathcal H_{\mathrm{sym}}^\perp\)，则 \(U_q^{\mathsf T}x=0\)，故修正项作用于 \(x\) 必为零。内积形式只是同一等式左右再与任意 \(y\) 配对。证毕。
\end{proof}

\begin{theorem}[秩一例外谱的大耦合二相分裂]\label{thm:conclusion-rankone-large-coupling-two-phase-separation}
沿用命题 \ref{prop:conclusion-rankone-cauchy-orthogonality} 的记号。令
\[
W:=\sum_{i=0}^{q}w_i,
\qquad
N(x):=\sum_{i=0}^{q}w_i\!\!\prod_{\substack{0\le j\le q\\ j\ne i}}\!\!(x-t_j),
\]
并设 \(t_0<\cdots<t_q\)。则 \(N\) 具有 \(q\) 个实根
\[
\zeta_1<\cdots<\zeta_q,
\qquad
\zeta_j\in(t_{j-1},t_j),
\]
而 \(A(b)=D_0+bgg^{\mathsf T}\) 的特征值 \(\lambda_0(b)<\cdots<\lambda_q(b)\) 满足
\[
\lambda_q(b)=bW+\frac{\sum_{i=0}^{q}w_it_i}{W}+O(b^{-1}),
\]
\[
\lambda_j(b)=\zeta_{j+1}+O(b^{-1})
\qquad (0\le j\le q-1).
\]
因此当 \(b\to+\infty\) 时，秩一例外谱发生严格二相分裂：一条宏观分支线性逃逸，其余 \(q\) 条微观分支冻结到一组 interlacing 内部平衡点。
\end{theorem}
\begin{proof}
由
\[
\Phi(x):=\sum_{i=0}^{q}\frac{w_i}{x-t_i}
=\frac{N(x)}{G(x)}
\]
可知 \(\Phi\) 在每个区间 \((t_{j-1},t_j)\) 上连续且严格递减，并从 \(+\infty\) 变化到 \(-\infty\)。因此方程 \(\Phi(x)=0\) 在每个此类区间内恰有一个实根，这些根正是 \(N\) 的 \(q\) 个实根 \(\zeta_1,\dots,\zeta_q\)。

另一方面，主稿的大耦合谱分裂定理 \ref{thm:xi-rankone-secular-small-large-coupling} 已给出：当 \(b\to+\infty\) 时，存在唯一一条根分支趋于 \(+\infty\)，并满足
\[
\lambda_q(b)=bW+\frac{\sum_i w_i t_i}{W}+O(b^{-1}),
\]
其余 \(q\) 条根分支分别收敛到 \(N\) 的 \(q\) 个实根。上述 interlacing 结构已唯一标定了收敛目标，故
\[
\lambda_j(b)=\zeta_{j+1}+O(b^{-1})
\qquad (0\le j\le q-1).
\]
证毕。
\end{proof}

\begin{theorem}[大耦合判别式与微观谱速度的幂律]\label{thm:conclusion-rankone-large-coupling-discriminant-speed-powerlaw}
沿用定理 \ref{thm:conclusion-rankone-large-coupling-two-phase-separation} 的记号，则有
\[
\mathrm{Disc}(P_b)=b^{2q}W^{2}\mathrm{Disc}(N)+O(b^{2q-1}),
\]
并且
\[
\prod_{j=0}^{q}\lambda_j'(b)
=
\frac{\bigl(\prod_{i=0}^{q}w_i\bigr)\mathrm{Disc}(G)}{\mathrm{Disc}(P_b)}
=
\frac{\bigl(\prod_i w_i\bigr)\mathrm{Disc}(G)}
{b^{2q}W^{2}\mathrm{Disc}(N)}
\bigl(1+O(b^{-1})\bigr).
\]
进一步，由 \(\lambda_q'(b)\to W\) 可得
\[
\prod_{j=0}^{q-1}\lambda_j'(b)
=
\frac{\bigl(\prod_i w_i\bigr)\mathrm{Disc}(G)}
{b^{2q}W^{3}\mathrm{Disc}(N)}
\bigl(1+O(b^{-1})\bigr).
\]
因此，大耦合极限下的 \(q\) 条微观分支并非仅仅趋于冻结；其总谱速度满足严格的 \(b^{-2q}\) 衰减律。
\end{theorem}
\begin{proof}
设 \(\beta(b):=\lambda_q(b)\)。由定理 \ref{thm:conclusion-rankone-large-coupling-two-phase-separation}，
\[
\beta(b)=bW+O(1),
\qquad
\lambda_j(b)=\zeta_{j+1}+O(b^{-1})
\quad (0\le j\le q-1).
\]
于是
\[
\mathrm{Disc}(P_b)
=
\prod_{0\le i<j\le q}\bigl(\lambda_j(b)-\lambda_i(b)\bigr)^2
\]
中，涉及 \(\beta(b)\) 的 \(q\) 个差因子各自贡献
\[
\bigl(\beta(b)-\zeta_j\bigr)^2=(bW)^2\bigl(1+O(b^{-1})\bigr).
\]
因此这些因子的总乘积为
\[
b^{2q}W^{2q}\bigl(1+O(b^{-1})\bigr).
\]
另一方面，\(N(x)=W\prod_{j=1}^{q}(x-\zeta_j)\)，故
\[
\prod_{1\le i<j\le q}(\zeta_j-\zeta_i)^2
=
\frac{\mathrm{Disc}(N)}{W^{2q-2}}.
\]
两式相乘即得
\[
\mathrm{Disc}(P_b)=b^{2q}W^{2}\mathrm{Disc}(N)+O(b^{2q-1}).
\]

再由推论 \ref{cor:conclusion-rankone-eigenvalue-speed-discriminant-identity}，
\[
\prod_{j=0}^{q}\lambda_j'(b)
=
\frac{\bigl(\prod_i w_i\bigr)\mathrm{Disc}(G)}{\mathrm{Disc}(P_b)},
\]
代入刚得到的判别式展开即可得第二式。

最后，由
\[
\lambda_q(b)=bW+\frac{\sum_i w_i t_i}{W}+O(b^{-1})
\]
直接求导，得 \(\lambda_q'(b)=W+O(b^{-2})\)，特别地 \(\lambda_q'(b)\to W\)。故将第二式除以 \(\lambda_q'(b)\) 即得最后一式。证毕。
\end{proof}

以上结果表明，两参数不交核的谱结构可被严格拆分为三层互补对象：一层是保留全部二项背景节点的高重数块；一层是由 \(q+1\) 维对称 Krylov 扇区完全承载的例外谱；最后一层则是在一般秩一耦合下于大耦合极限显现出的宏观逃逸分支与微观冻结簇。于是，例外因子的恢复、Hankel 笔表示、双向时间修正与判别式幂律不再是彼此孤立的现象，而构成同一背景--例外分裂机制在代数、解析与动力学三个方向上的互补投影。
