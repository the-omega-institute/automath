\begin{theorem}[bin-fold 渐近信息几何的两原子普适性]\label{thm:conclusion-binfold-twoatom-information-geometry-universality}
记
\[
p_m^{\mathrm{bin}}(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
u_m(w):=\frac{1}{|X_m|},
\qquad
R_m(w):=\frac{p_m^{\mathrm{bin}}(w)}{u_m(w)}.
\]
再记
\[
r_0:=\frac{\varphi^2}{\sqrt5},
\qquad
r_1:=\frac{\varphi}{\sqrt5}.
\]
则对任意凸函数 \(f:(0,\infty)\to\RR\) 满足 \(f(1)=0\)，其相对均匀基线的 \(f\)-散度
\[
D_f\!\bigl(p_m^{\mathrm{bin}}\Vert u_m\bigr)
:=
\sum_{w\in X_m}u_m(w)\,f(R_m(w))
\]
满足极限
\[
\lim_{m\to\infty}D_f\!\bigl(p_m^{\mathrm{bin}}\Vert u_m\bigr)
=
\frac1\varphi f(r_0)+\frac1{\varphi^2}f(r_1).
\]
因此，bin-fold 相对于稳定层均匀基线的全部渐近 \(f\)-散度，都因子化为同一两原子概率律
\[
\left\{
r_0\ \text{with weight }\varphi^{-1},
\qquad
r_1\ \text{with weight }\varphi^{-2}
\right\}.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-foldbin-likelihood-ratio-two-atom-transfer}，
\[
R_m(W_m)\Rightarrow R_\infty,
\qquad
R_\infty=
\begin{cases}
r_0,& \text{以概率 }\varphi^{-1},\\
r_1,& \text{以概率 }\varphi^{-2},
\end{cases}
\]
其中 \(W_m\sim u_m\)。而
\[
D_f\!\bigl(p_m^{\mathrm{bin}}\Vert u_m\bigr)=\EE[f(R_m(W_m))].
\]
对连续凸函数 \(f\) 直接应用该定理中的泛函传递原则，即得
\[
\EE[f(R_m(W_m))]
\longrightarrow
\frac1\varphi f(r_0)+\frac1{\varphi^2}f(r_1).
\]
证毕。
\end{proof}

\begin{theorem}[两枚极限标量完全重构 bin-fold 的渐近温度学]\label{thm:conclusion-binfold-two-scalar-complete-reconstruction}
固定 \(\alpha>1\) 与 \(q\ge 0\)。定义两枚极限标量
\[
A_\alpha:=\lim_{m\to\infty}D_{f_\alpha}\!\bigl(\mu_m\Vert u_m\bigr),
\qquad
K_q:=\lim_{m\to\infty}D_{\mathrm{KL}}\!\bigl(\pi_{m,q}\Vert \pi_{m,0}\bigr),
\]
其中
\[
f_\alpha(t)=t^\alpha-1-\alpha(t-1).
\]
则二元组 \((A_\alpha,K_q)\) 唯一决定下列全部数据：
\[
\varphi,
\qquad
q,
\qquad
\theta(q):=\frac{1}{1+\varphi^{q+1}},
\]
以及尺度化残差
\[
Z_m^{(q)}:=\left(\frac{\varphi}{2}\right)^m d_m^{\mathrm{bin}}(W_m^{(q)})
\]
的极限分布
\[
\PP\!\left(Z^{(q)}=\varphi^{-1}\right)=\theta(q),
\qquad
\PP\!\left(Z^{(q)}=1\right)=1-\theta(q).
\]
换言之，bin-fold 的渐近 escort 温度学在信息论意义上只需两枚实标量即可完全解码：一枚恢复黄金参数，另一枚恢复温度参数。
\end{theorem}

\begin{proof}
由推论 \ref{cor:conclusion-binfold-single-scalar-recovers-golden-parameter}，固定 \(\alpha>1\) 时，
\[
A_\alpha
=
5^{-\alpha/2}\bigl(\varphi^{2\alpha-1}+\varphi^{\alpha-2}\bigr)-1
\]
作为 \(\varphi\) 的函数严格单调，故 \(A_\alpha\) 唯一恢复 \(\varphi\)。

再记
\[
\theta_0:=\theta(0)=\frac{1}{1+\varphi}.
\]
由命题 \ref{prop:fold-bin-kl-escort-limit-constant}，
\[
K_q
=
\theta(q)\log\frac{\theta(q)}{\theta_0}
+\bigl(1-\theta(q)\bigr)\log\frac{1-\theta(q)}{1-\theta_0}.
\]
对固定的 \(\theta_0\in(0,1/2)\)，定义
\[
F_{\theta_0}(\theta)
:=
\theta\log\frac{\theta}{\theta_0}
+(1-\theta)\log\frac{1-\theta}{1-\theta_0}.
\]
则
\[
F_{\theta_0}'(\theta)
=
\log\frac{\theta(1-\theta_0)}{\theta_0(1-\theta)}.
\]
当 \(\theta\in(0,\theta_0)\) 时有 \(F'_{\theta_0}(\theta)<0\)，故 \(F_{\theta_0}\) 在该区间上严格递减。另一方面，
\[
\theta(q)=\frac{1}{1+\varphi^{q+1}}
\]
随 \(q\) 严格递减，且其值域正落在 \((0,\theta_0]\) 内。因此
\[
q\longmapsto K_q=F_{\theta_0}(\theta(q))
\]
严格递增，故 \(K_q\) 在已知 \(\varphi\) 后唯一恢复 \(q\)，并随之唯一恢复 \(\theta(q)\)。

最后，由主稿中对固定 \(q\) 的尺度化残差两点极限律，
\[
Z_m^{(q)}\Rightarrow Z^{(q)},
\qquad
Z^{(q)}=
\begin{cases}
\varphi^{-1},& \text{以概率 }\theta(q),\\
1,& \text{以概率 }1-\theta(q).
\end{cases}
\]
故极限分布也由 \((A_\alpha,K_q)\) 唯一决定。证毕。
\end{proof}

\[
\lim_{m\to\infty}D_{\mathrm{KL}}\!\bigl(\pi_{m,q}\Vert \pi_{m,p}\bigr)

D_{\mathrm{KL}}\!\bigl(\Bern(\theta(q))\Vert \Bern(\theta(p))\bigr),

\bigl(1-\theta(q)\bigr)F(1)+\theta(q)F(\varphi^{-1}).

\begin{corollary}[Rényi 层级的指数盲性]\label{cor:conclusion-binfold-renyi-exponential-blindness}

\frac1m H_q(\mu_m)\to \log\varphi,
\]

\end{corollary}

\frac1m H_q(\mu_m)\to \log\varphi.
\]

\end{proof}

\begin{theorem}[严格失饱和区中的两原子持久性]\label{thm:conclusion-binfold-twoatom-persistence-in-strict-nonsaturation-region}

  \item 然而两原子极限律、上尾单跳塌缩与 Rényi 速率塌缩对全部充分大的 \(m\) 持续成立。

因此，bin-fold 的两原子信息几何并非由“最大纤维恰好达到组合上界”驱动；相反，它在最终严格失饱和区中仍保持稳定。

\end{proof}
