\subsection{黄金传输误差与幂散度二原子识别的桥接}\label{subsec:conclusion-golden-transport-powerdivergence-bridge}
在定理 \ref{thm:conclusion-golden-fibonacci-audit-two-periodic-rigidity} 已将黄金分支 Fibonacci 分母上的 Wasserstein 审计振荡压缩为显式的上、下极限与中心值之后，还可进一步把这一几何误差链与 bin-fold 相对均匀基线的二原子 \(f\)-散度极限统一到同一条可辨识叙事中。其核心特征是：一方面，transport audit 在坏侧精确等于星差证书的一半、在好侧则继续分辨箱内偏移，从而与通用 \((\log N)/N\) 型星差预算产生不可消除的对数级冗余分离；另一方面，幂散度极限谱由同一黄金参数 \(\varphi\) 完全控制，且在 \(\chi^2\)-型截面上退化为线性反演公式。

\begin{theorem}[黄金分支上 transport audit 相对星差预算的对数级不可逆冗余]\label{thm:conclusion-golden-transport-star-log-overhead}
沿用附录 \ref{app:kronecker-w1-transport} 与附录 \ref{app:kronecker-discrepancy} 的记号，记
\[
D_n^\ast:=D_{F_n}^\ast\!\bigl(P_{F_n}(\varphi^{-1})\bigr),
\qquad
\mathsf T_n:=\mathsf T_{F_n}(\varphi^{-1}),
\]
并定义显式星差预算
\[
U_n:=\frac{3+\left\lceil \dfrac{\log(\sqrt5\,F_n)}{\log\varphi}\right\rceil}{F_n}.
\]
则有：
\begin{enumerate}
  \item 当 \(n\) 为偶数时，
  \[
  \mathsf T_n=\frac12\,D_n^\ast
  \]
  精确成立，并且
  \[
  \boxed{\;
  \lim_{\substack{n\to\infty\\ n\ \mathrm{even}}}F_nD_n^\ast
  =
  1+\frac{1}{\sqrt5}.\;}
  \]
  \item 当 \(n\) 为奇数时，
  \[
  D_n^\ast=\frac1{F_n}
  \]
  精确成立，并且
  \[
  \boxed{\;
  \lim_{\substack{n\to\infty\\ n\ \mathrm{odd}}}F_n\mathsf T_n
  =
  \frac12-\frac{1}{2\sqrt5}+\frac1{15}.\;}
  \]
  \item 通用星差预算相对真实 transport audit 的冗余满足
  \[
  \boxed{\;
  \frac{U_n}{\mathsf T_n}\xrightarrow[n\to\infty]{}+\infty,
  \qquad
  \frac{U_n}{\mathsf T_n}=\Theta(\log F_n).\;}
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-golden-w1-star-certificate-parity-splitting}，当 \(n\) 为偶数时有精确恒等式
\[
\mathsf T_n=\frac12\,D_n^\ast.
\]
再由推论 \ref{cor:kronecker-w1-fibonacci-limits}，
\[
F_n\mathsf T_n\longrightarrow \frac12+\frac{1}{2\sqrt5}
\qquad
\bigl(n\to\infty,\ n\ \mathrm{even}\bigr),
\]
故
\[
F_nD_n^\ast=2F_n\mathsf T_n\longrightarrow 1+\frac{1}{\sqrt5}.
\]
这证明第一条。

当 \(n\) 为奇数时，由定理 \ref{thm:kronecker-w1-denominator-closed-form} 的好侧公式可知
\[
D_n^\ast=\frac1{F_n},
\]
而推论 \ref{cor:kronecker-w1-fibonacci-limits} 给出
\[
F_n\mathsf T_n\longrightarrow \frac12-\frac{1}{2\sqrt5}+\frac1{15}.
\]
于是第二条成立。

对第三条，由附录 \ref{app:kronecker-discrepancy} 中的显式证书，
\[
U_n=\frac{3+\left\lceil \dfrac{\log(\sqrt5\,F_n)}{\log\varphi}\right\rceil}{F_n}
\asymp \frac{\log F_n}{F_n}.
\]
另一方面，定理 \ref{thm:conclusion-golden-fibonacci-audit-two-periodic-rigidity} 给出
\[
0<\liminf_{n\to\infty}F_n\mathsf T_n
\le
\limsup_{n\to\infty}F_n\mathsf T_n
<\infty,
\]
故 \(\mathsf T_n\asymp F_n^{-1}\)。因此
\[
\frac{U_n}{\mathsf T_n}
=
\frac{3+\left\lceil \dfrac{\log(\sqrt5\,F_n)}{\log\varphi}\right\rceil}{F_n\mathsf T_n}
\asymp
\log F_n,
\]
从而既有 \(\Theta(\log F_n)\) 级增长，也必然趋于 \(+\infty\)。证毕。
\end{proof}

\begin{theorem}[单个 \texorpdfstring{$\chi^2$}{chi2}-型极限常数对黄金参数的线性恢复]\label{thm:conclusion-chi2-linear-recovers-phi}
\leanverified{paper\_conclusion\_chi2\_linear\_recovers\_phi}
对 \(\alpha>1\) 定义
\[
f_\alpha(t):=t^\alpha-1-\alpha(t-1),
\qquad
D_\infty^{f_\alpha}:=\lim_{m\to\infty}D_{f_\alpha}(\mu_m\Vert u_m).
\]
则在 \(\alpha=2\) 处，
\[
f_2(t)=(t-1)^2,
\]
对应 Pearson \(\chi^2\)-型极限常数，并且有精确线性公式
\[
\boxed{\;
D_\infty^{f_2}
=
\frac{2\varphi-3}{5},
\qquad
\varphi=\frac{5D_\infty^{f_2}+3}{2}.\;}
\]
特别地，单个 \(\chi^2\)-型极限标量已足以线性恢复黄金参数 \(\varphi\)。
\end{theorem}
\begin{proof}
由命题 \ref{prop:fold-bin-phi-identifiable-power-divergence}，
\[
D_\infty^{f_\alpha}
=
\frac{1}{5^{\alpha/2}}\bigl(\varphi^{2\alpha-1}+\varphi^{\alpha-2}\bigr)-1.
\]
取 \(\alpha=2\) 得
\[
D_\infty^{f_2}
=
\frac{\varphi^3+1}{5}-1.
\]
又因 \(\varphi^2=\varphi+1\)，故
\[
\varphi^3=\varphi(\varphi+1)=2\varphi+1.
\]
代回即得
\[
D_\infty^{f_2}
=
\frac{2\varphi-3}{5}.
\]
移项便得到
\[
\varphi=\frac{5D_\infty^{f_2}+3}{2}.
\]
证毕。
\end{proof}

\begin{proposition}[极限 power-divergence 族的严格对数凸性与主导密度比选取]\label{prop:conclusion-power-divergence-logconvex-dominant-ratio}
\leanverified{paper\_conclusion\_power\_divergence\_logconvex\_dominant\_ratio}
沿用定理 \ref{thm:conclusion-chi2-linear-recovers-phi} 的记号，定义
\[
E(\alpha):=D_\infty^{f_\alpha}+1
=
\frac{1}{5^{\alpha/2}}\bigl(\varphi^{2\alpha-1}+\varphi^{\alpha-2}\bigr)
=
\varphi^{-1}\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr)^\alpha
+
\varphi^{-2}\Bigl(\frac{\varphi}{\sqrt5}\Bigr)^\alpha.
\]
则：
\begin{enumerate}
  \item \(E(\alpha)\) 在 \((1,\infty)\) 上严格对数凸；
  \item 极限比率与指数率满足
  \[
  \boxed{\;
  \lim_{\alpha\to\infty}\frac{E(\alpha+1)}{E(\alpha)}
  =
  \frac{\varphi^2}{\sqrt5},
  \qquad
  \lim_{\alpha\to\infty}\frac1\alpha\log E(\alpha)
  =
  \log\!\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr).\;}
  \]
\end{enumerate}
换言之，power-divergence 族的高阶端点精确选出二原子极限中的较大密度比 \(\varphi^2/\sqrt5\)。
\end{proposition}
\begin{proof}
写成
\[
E(\alpha)=Ae^{c\alpha}+Be^{d\alpha},
\qquad
A:=\varphi^{-1},\quad B:=\varphi^{-2},
\]
\[
c:=\log\!\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr),
\qquad
d:=\log\!\Bigl(\frac{\varphi}{\sqrt5}\Bigr).
\]
由于 \(A,B>0\) 且 \(c>d\)，有
\[
(\log E)''(\alpha)
=
\frac{AB(c-d)^2e^{(c+d)\alpha}}{(Ae^{c\alpha}+Be^{d\alpha})^2}>0,
\]
故 \(E\) 严格对数凸。

又因为 \(c>d\)，首项 \(Ae^{c\alpha}\) 在 \(\alpha\to\infty\) 时严格主导 \(Be^{d\alpha}\)，故
\[
\frac{E(\alpha+1)}{E(\alpha)}
=
\frac{Ae^{c(\alpha+1)}+Be^{d(\alpha+1)}}{Ae^{c\alpha}+Be^{d\alpha}}
\longrightarrow e^c
=
\frac{\varphi^2}{\sqrt5},
\]
并且
\[
\frac1\alpha\log E(\alpha)\longrightarrow c
=
\log\!\Bigl(\frac{\varphi^2}{\sqrt5}\Bigr).
\]
最后，由定理 \ref{thm:conclusion-foldbin-likelihood-ratio-two-atom-transfer}，\(\varphi/\sqrt5\) 与 \(\varphi^2/\sqrt5\) 正是 bin-fold 相对均匀基线的两点极限密度比；故上述高阶极限恰选出其中的较大者。证毕。
\end{proof}
