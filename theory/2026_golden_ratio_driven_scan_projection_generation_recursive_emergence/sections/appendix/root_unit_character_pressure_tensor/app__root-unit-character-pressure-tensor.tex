\section{根单位点完成化轨道、压力代数谱曲线与伸缩链的角色--精度张量化}\label{app:root-unit-character-pressure-tensor}

本附录在既有“折叠投影/多重度谱---稳定类型”数据结构之上，针对统一图景给出三条结构性结论：根单位点完成化轨道在 Chebyshev--Adams 运算下的角色对角化、规范差压力函数的四次代数谱曲线、以及伸缩链在角色谱通道与精度势账本通道上的正交张量化闭环。所用折叠直方图与最大纤维谱均来自现有可审计表格数据。

\subsection{根单位点完成化轨道模与角色谱分解}\label{subsec:root-unit-completion-orbit-character}

取单位圆扭曲 \(u=e^{\mathrm{i}\theta}\)、半角变量 \(r=e^{\mathrm{i}\theta/2}\)、完成化坐标
\(s=r+r^{-1}=2\cos(\theta/2)\in[-2,2]\)，并令
\begin{equation}\label{eq:chebyshev-adams-Cm}
C_m(s):=r^m+r^{-m}=2\cos(m\theta/2)\in\ZZ[s],
\end{equation}
即 Chebyshev--Adams 运算（与 \S\ref{subsec:unit-disk-witt-chebyshev} 及推论 \ref{cor:primitive-completion-hatp} 一致）。

对任意整数 \(q\ge 3\)，定义有限\textbf{根单位点完成化轨道}
\begin{equation}\label{eq:root-unit-orbit-Sq}
\mathcal{S}_q:=\Bigl\{\, s_a:=2\cos\!\Bigl(\frac{\pi a}{q}\Bigr) : a=1,2,\dots,q-1 \Bigr\}\subset[-2,2].
\end{equation}
则 \(\mathcal{S}_q\) 在所有 \(s\mapsto C_m(s)\) 下封闭，并满足置换律
\begin{equation}\label{eq:root-unit-Cm-permutation}
C_m(s_a)=2\cos\!\Bigl(\frac{\pi m a}{q}\Bigr)=s_{\langle ma\rangle},
\end{equation}
其中 \(\langle ma\rangle\) 表示在同值识别 \(a\sim 2q-a\)（余弦偶性）下的唯一代表元。

令 \(\CC^{\mathcal{S}_q}\) 为在 \(\mathcal{S}_q\) 上的函数空间，并定义有限维表示
\begin{equation}\label{eq:root-unit-Rm}
(\mathsf{R}_m F)(s):=F(C_m(s)),\qquad F\in\CC^{\mathcal{S}_q}.
\end{equation}
则 \(\{\mathsf{R}_m\}_{m\ge 1}\) 为由乘法半群作用诱导的有限维表示，其不可约分解由模 \(2q\) 的偶 Dirichlet 角色（满足 \(\chi(-1)=1\)）对角化。

\leanverified{paper\_root\_unit\_character\_diagonalization}
\begin{theorem}[根单位点角色对角化]\label{thm:root-unit-character-diagonalization}
对任意 \(\sigma>0\)，定义“完成化--Witt 伸缩”在 \(\CC^{\mathcal{S}_q}\) 上的正则化算子
\begin{equation}\label{eq:root-unit-W-sigma}
(\mathsf{W}_{\sigma,q}F)(s):=\sum_{m\ge 1}\frac{1}{m^{1+\sigma}}\,F(C_m(s)).
\end{equation}
则对每个模 \(2q\) 的偶 Dirichlet 角色 \(\chi\) 存在非零向量 \(v_\chi\in\CC^{\mathcal{S}_q}\) 使得
\[
\mathsf{W}_{\sigma,q}v_\chi=L(1+\sigma,\chi)\,v_\chi,
\]
其中 \(L(1+\sigma,\chi)=\sum_{m\ge 1}\chi(m)m^{-(1+\sigma)}\) 为 Dirichlet \(L\)-级数。并且 \(\CC^{\mathcal{S}_q}\) 分解为这些偶角色本征子空间的直和（按 \(a\sim -a\) 识别后的等变子空间）。
\end{theorem}

该结论在体系内部给出一条离散正交通道的系统生成律：当把连续相位压缩到根单位点集合时，伸缩链不再以端点三态退化出现（附录 \ref{app:unit-circle-phase-gate} 的 \(\mathcal{S}_\star=\{0,2,-2\}\)），而自动提升为完整的角色谱，随 \(q\) 增大出现可控的有限维正交通道。

\subsection{规范差压力函数的代数谱曲线}\label{subsec:gauge-pressure-algebraic-curve}

文稿将规范差逐位失配的指数矩母函数写成有限维加权邻接矩阵的谱半径（命题 \ref{prop:fold-gauge-anomaly-pressure}）：存在矩阵 \(A_\theta\) 使压力函数满足
\[
P_G(\theta)=\log\rho(A_\theta),
\]
其中 \(A_\theta\) 在文中给出显式形式。

\begin{theorem}[压力谱曲线的四次代数方程]\label{thm:gauge-pressure-quartic}
\leanverified{paper\_gauge\_pressure\_quartic}
设 \(x=e^\theta>0\)，令 \(y=\rho(A_\theta)\)。则 \(y\) 为以下四次方程的最大正实根：
\begin{equation}\label{eq:gauge-pressure-quartic}
8y^4-4y^3-(4x+2)y^2+(2x-x^3)y+x=0.
\end{equation}
从而 \(P_G(\theta)\) 为代数函数分支的对数；其全局正则性与凸性可在该代数曲线层面以判别式与分支点结构进行判定。
\end{theorem}

\begin{corollary}[倾斜平均失配率的闭式表达]\label{cor:gauge-pressure-tilt-derivative}
\leanverified{paper\_gauge\_pressure\_tilt\_derivative}
令 \(F(y,x)=8y^4-4y^3-(4x+2)y^2+(2x-x^3)y+x\)。则
\begin{equation}\label{eq:gauge-pressure-P-prime}
P_G'(\theta)=-\frac{x\,\partial_x F(y,x)}{y\,\partial_y F(y,x)}
=\frac{x\bigl(4y^2-(2-3x^2)y-1\bigr)}{y\bigl(32y^3-12y^2-(8x+4)y+(2x-x^3)\bigr)}.
\end{equation}
该式将“规范差的倾斜平均”完全还原为 \((x,y)\) 的代数函数，从而把规范差的统计学输出（均值、方差、偏差律）统一纳入可审计的代数谱曲线机制之中。
\end{corollary}

\subsubsection{压力谱四次曲线的射影模型、预解式与 \texorpdfstring{$S_4$}{S4} 闭包}\label{subsubsec:gauge-pressure-s4-geometry}

为与命题 \ref{prop:fold-gauge-anomaly-pressure} 的矩阵表示保持一致，令
\[
u=e^\theta,\qquad \mu(u):=2\rho(A_\theta).
\]
则 \(\mu\) 满足四次方程
\begin{equation}\label{eq:gauge-pressure-mu-quartic}
F(\mu,u):=\mu^4-\mu^3-(2u+1)\mu^2+(2u-u^3)\mu+2u=0.
\end{equation}
该式与式 \eqref{eq:gauge-pressure-quartic} 的关系为 \(\mu=2y,\ u=x\)。

\leanverified{paper\_gauge\_pressure\_plane\_quartic\_genus3}
\begin{theorem}[压力谱四次曲线的光滑平面四次模型]\label{thm:gauge-pressure-plane-quartic-genus3}
记 \(\overline{C}_{\mathsf G}\subset\PP^2_{[\mu:u:z]}\) 为 \eqref{eq:gauge-pressure-mu-quartic} 的总次数齐次化射影闭包。则 \(\overline{C}_{\mathsf G}\) 为光滑平面四次曲线，因而
\[
g(\overline{C}_{\mathsf G})=\frac{(4-1)(4-2)}{2}=3.
\]
\end{theorem}
\begin{proof}
设仿射方程 \(f(\mu,u)=F(\mu,u)\)。有限点奇性满足
\(
f=\partial_\mu f=\partial_u f=0
\)。
对该理想作 Gröbner 消元可得单位理想，故仿射片无奇点。

在无穷远直线 \(z=0\) 上，齐次方程化为
\(
\mu(\mu^3-u^3)=0
\)，交点为 \([0:1:0]\) 与 \([\zeta:1:0]\)（\(\zeta^3=1\)）。
由齐次多项式
\[
F_h=\mu^{4}-\mu^{3}z-2\mu^{2}uz-\mu^{2}z^{2}-\mu u^{3}+2\mu uz^{2}+2uz^{3}
\]
的梯度计算得：在 \([0:1:0]\) 处 \(\partial_\mu F_h=-1\neq 0\)；在 \([\zeta:1:0]\) 处 \(\partial_u F_h=-3\zeta\neq 0\)。
故无穷远点也皆光滑。于是 \(\overline{C}_{\mathsf G}\) 光滑。
平面光滑四次曲线属由标准公式给出为 \(3\)。证毕。
\end{proof}

\leanverified{paper\_gauge\_pressure\_generic\_galois\_s4}
\begin{theorem}[函数域 \texorpdfstring{$\QQ(u)$}{Q(u)} 上的泛型伽罗瓦群]\label{thm:gauge-pressure-generic-galois-s4}
四次多项式 \(F(\mu,u)\in \QQ(u)[\mu]\) 可分且不可约，并满足
\[
\mathrm{Gal}\bigl(F(\mu,u)/\QQ(u)\bigr)\cong S_4.
\]
\end{theorem}
\begin{proof}
取特化 \(u=2\) 得
\[
F(\mu,2)=\mu^4-\mu^3-5\mu^2-4\mu+4.
\]
其三次预解式为
\[
R_2(z)=z^3+5z^2-12z-100.
\]
有理根判别给出 \(R_2\) 在 \(\QQ\) 上不可约；并且
\(
\mathrm{Disc}\,F(\mu,2)=-2^4\cdot 6343
\)
非平方。
按四次多项式经典判别，特化伽罗瓦群为 \(S_4\)。
由伽罗瓦群特化原理，泛型伽罗瓦群含有同构于 \(S_4\) 的子群，故只能为 \(S_4\)。
可分性由 \(\mathrm{Disc}_\mu F\not\equiv 0\) 立得。证毕。
\end{proof}

\leanverified{paper\_gauge\_pressure\_resolvent\_disc\_identity}
\begin{proposition}[三次预解式与判别式恒等]\label{prop:gauge-pressure-resolvent-disc-identity}
设 \(F(\mu,u)\) 如 \eqref{eq:gauge-pressure-mu-quartic}，其标准三次预解式可写为
\[
R(z,u):=z^3+(2u+1)z^2+(u^3-10u)z-\bigl(u^6-4u^4+20u^2+10u\bigr).
\]
则有恒等式
\begin{equation}\label{eq:gauge-pressure-disc-factorization-p10}
\mathrm{Disc}_{z}R(z,u)=\mathrm{Disc}_{\mu}F(\mu,u)=-u(u-1)P_{10}(u),
\end{equation}
其中
\[
\begin{aligned}
P_{10}(u)=\;&27u^{10}+27u^9-153u^8-163u^7+793u^6+1061u^5\\
&-832u^4-816u^3+1320u^2-440u+40.
\end{aligned}
\]
\end{proposition}
\begin{proof}
对一般四次多项式 \(x^4+ax^3+bx^2+cx+d\) 取标准预解式
\[
z^3-bz^2+(ac-4d)z+(4bd-a^2d-c^2).
\]
代入
\(
(a,b,c,d)=(-1,\,-(2u+1),\,2u-u^3,\,2u)
\)
即得所述 \(R(z,u)\)。
再直接计算两者判别式并化简，可得
\(
\mathrm{Disc}_{z}R=\mathrm{Disc}_{\mu}F
\)；
最后在 \(\ZZ[u]\) 中因式分解即得式 \eqref{eq:gauge-pressure-disc-factorization-p10}。证毕。
\end{proof}

\leanverified{paper\_gauge\_pressure\_resolvent\_cover\_genus4}
\begin{theorem}[预解式三次覆叠的规范化属]\label{thm:gauge-pressure-resolvent-cover-genus4}
令 \(C_{\mathrm{res}}\) 为射影曲线 \(R(z,u)=0\) 的规范化，并以 \(u\) 记到 \(\PP^1_u\) 的坐标映射。则：
\begin{enumerate}
  \item \(u:C_{\mathrm{res}}\to \PP^1_u\) 为次数 \(3\) 的覆叠；
  \item 分支集合为
  \[
  B_{\mathrm{res}}=\{u=0,\ u=1\}\cup\{P_{10}(u)=0\},
  \]
  共 \(12\) 个有限分支值，且均为单分歧；
  \item \(g(C_{\mathrm{res}})=4\)。
\end{enumerate}
\end{theorem}
\begin{proof}
\(R\) 关于 \(z\) 为首一三次式，故映射次数为 \(3\)。
分支当且仅当 \(\mathrm{Disc}_z R=0\)，由命题 \ref{prop:gauge-pressure-resolvent-disc-identity} 即得分支集合。
又 \(-u(u-1)P_{10}(u)\) 为平方自由，故每个分支值对应一次分歧，分歧总量为 \(12\)。
Riemann--Hurwitz 公式给出
\[
2g(C_{\mathrm{res}})-2=3(-2)+12=6,
\]
故 \(g(C_{\mathrm{res}})=4\)。证毕。
\end{proof}

\leanverified{paper\_gauge\_pressure\_sign\_cover\_genus5}
\begin{theorem}[判别式二次子覆叠与唯一二次子扩张]\label{thm:gauge-pressure-sign-cover-genus5}
设
\[
C_{\mathrm{sgn}}:\quad w^2=\mathrm{Disc}_{\mu}F(\mu,u)=-u(u-1)P_{10}(u)
\]
为其光滑射影模型。则：
\begin{enumerate}
  \item \(C_{\mathrm{sgn}}\) 为超椭圆曲线，且 \(g(C_{\mathrm{sgn}})=5\)；
  \item 在定理 \ref{thm:gauge-pressure-generic-galois-s4} 的 \(S_4\) 情形下，\(\QQ(u,w)\) 给出分裂域中的唯一二次子扩张（对应符号特征 \(\mathrm{sgn}:S_4\to\{\pm1\}\)）。
\end{enumerate}
\end{theorem}
\begin{proof}
右端多项式次数为 \(12\) 且平方自由，故 \(C_{\mathrm{sgn}}\to\PP^1_u\) 为在 \(12\) 点分支的二次覆盖，从而
\(
g=(12-2)/2=5
\)。
对不可约四次，\(\sqrt{\mathrm{Disc}}\) 生成的二次扩张对应商 \(S_4/A_4\cong \ZZ/2\ZZ\)。
当伽罗瓦群为 \(S_4\) 时，指数 \(2\) 正规子群唯一，故二次子扩张唯一。证毕。
\end{proof}

\leanverified{paper\_gauge\_pressure\_s4\_closure\_genus49}
\begin{theorem}[\texorpdfstring{$S_4$}{S4} 伽罗瓦闭包的属与中间商塔]\label{thm:gauge-pressure-s4-closure-genus49}
设 \(X_{\mathsf G}\to\PP^1_u\) 为四次覆盖 \(F(\mu,u)=0\) 的 \(S_4\)-伽罗瓦闭包。则：
\begin{enumerate}

  \item \(g(X_{\mathsf G})=49\)；

  X_{\mathsf G}/A_4\simeq C_{\mathrm{sgn}},\qquad

\end{theorem}

\subsection{四次压力谱覆盖与七次 arity-charge 谱接口的线性互不相交}\label{subsec:root-unit-s4xs7-linearly-disjoint}

本节把 \ref{subsec:gauge-pressure-algebraic-curve} 的四次压力谱曲线与真实输入核的 arity-charge 七次谱接口统一置于同一基域上，从而得到可审计的直积伽罗瓦闭包与联合 Frobenius 循环型的乘积律。

\begin{theorem}[四次压力谱与七次谱接口的线性互不相交]\label{thm:root-unit-s4xs7-linearly-disjoint}
\leanverified{paper\_root\_unit\_s4xs7\_linearly\_disjoint}
令 \(K:=\QQ(u,q)\)。
设 \(L_4/K\) 为四次多项式 \(F(\mu,u)\in\QQ(u)[\mu]\)（见式 \eqref{eq:gauge-pressure-mu-quartic}）在 \(K\) 上的分裂域；
设 \(L_7/K\) 为七次多项式 \(P_7(\Lambda,q)\in\QQ(q)[\Lambda]\)（见定理 \ref{thm:real-input-40-arity-charge-det-closed}）在 \(K\) 上的分裂域。
则
\[
L_4\cap L_7=K,
\qquad
\Gal(L_4L_7/K)\cong S_4\times S_7.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:gauge-pressure-generic-galois-s4} 可知 \(\Gal(L_4/K)\cong S_4\)，且 \(L_4\) 的唯一二次子扩张由判别式生成（定理 \ref{thm:gauge-pressure-sign-cover-genus5}）：
\[
K\subset K\bigl(\sqrt{\Disc_{\mu}F(\mu,u)}\bigr)\subset L_4,
\qquad
\Disc_{\mu}F(\mu,u)=-u(u-1)P_{10}(u)
\]
（命题 \ref{prop:gauge-pressure-resolvent-disc-identity}）。
另一方面，由定理 \ref{thm:real-input-40-p7-generic-galois-s7} 得 \(\Gal(L_7/K)\cong S_7\)，且同样存在唯一二次子扩张
\[
K\subset K\bigl(\sqrt{\Disc_{\Lambda}P_7(\Lambda,q)}\bigr)\subset L_7,
\qquad
\Disc_{\Lambda}\bigl(P_7(\Lambda,q)\bigr)=-4q\,D(q)
\]
（定理 \ref{thm:real-input-40-p7-discriminant-branch-s17}）。
\par
设 \(F:=L_4\cap L_7\)。
若 \(F\neq K\)，则 \(F/K\) 为非平凡伽罗瓦扩张，并诱导出 \(S_4\) 与 \(S_7\) 的公共非平凡商群。
由于 \(S_7\) 的唯一非平凡真正规商为 \(S_7/A_7\cong \ZZ/2\ZZ\)，故 \(F/K\) 必为二次扩张。
因此必有二次子域相等
\[
K\bigl(\sqrt{-u(u-1)P_{10}(u)}\bigr)=K\bigl(\sqrt{-4q\,D(q)}\bigr),
\]
等价于存在 \(g(u,q)\in K^\times\) 使
\(
-u(u-1)P_{10}(u)=g(u,q)^2\cdot(-4q\,D(q))
\)。
对 \(q\) 作特化 \(q=2\)，由定理 \ref{thm:real-input-40-p7-generic-galois-s7} 的特化审计（例如其给出的 \(\Disc(P_7(\Lambda,2))\) 含奇指数素因子）可知 \(-4\cdot 2\cdot D(2)\in\QQ^\times\) 为非平方常数。
于是上式在 \(\QQ(u)\) 中推出 \(-u(u-1)P_{10}(u)\) 为常数乘以平方。
但 \(-u(u-1)P_{10}(u)\) 在 \(\QQ[u]\) 中平方自由（命题 \ref{prop:gauge-pressure-resolvent-disc-identity}），不可能为常数乘以平方，矛盾。
故 \(F=K\)。
\par
线性互不相交推出复合分裂域的伽罗瓦群为直积 \(S_4\times S_7\)（例如由伽罗瓦理论的标准结论：两正则伽罗瓦扩张在基域上交为基域当且仅当伽罗瓦群为直积）。证毕。
\end{proof}

\leanverified{paper\_root\_unit\_s4xs7\_chebotarev\_product}
\begin{corollary}[联合 Frobenius 循环型的乘积律]\label{cor:root-unit-s4xs7-chebotarev-product}
在定理 \ref{thm:root-unit-s4xs7-linearly-disjoint} 的设定下，取有理特化 \((u,q)=(u_0,q_0)\in\QQ^2\) 落在 Hilbert 不可约性所排除的薄集之外，使得两条覆盖的特化伽罗瓦群仍为 \(S_4\) 与 \(S_7\)，且相应分裂域在 \(\QQ\) 上线性互不相交。
记复合分裂域的伽罗瓦群为 \(G\simeq S_4\times S_7\)。
对所有避开判别式的素数 \(p\)，令 \(\mathrm{Frob}_p\) 在 \(G\) 中的共轭类记为 \((C_4(p),C_7(p))\)。
则对任意共轭类 \(C_4\subset S_4\)、\(C_7\subset S_7\)，满足自然密度恒等式
\[
\delta\bigl(\{p:\ C_4(p)=C_4,\ C_7(p)=C_7\}\bigr)
=\frac{|C_4|}{|S_4|}\cdot\frac{|C_7|}{|S_7|}
=\frac{|C_4|}{24}\cdot\frac{|C_7|}{5040}.
\]
\end{corollary}

\begin{proof}
由特化假设，复合分裂域的伽罗瓦群为直积 \(S_4\times S_7\)。
对该直积伽罗瓦扩张应用 Chebotarev 密度定理，并利用直积群中共轭类计数的分解性，即得所示密度为类比例乘积。证毕。
\end{proof}

\subsection{多模数角格的贴边极点漂移比：稠密性、经验分布与 Hermite 饱和猜想}\label{subsec:root-unit-angle-lattice-drift-ratio}

在 \S\ref{app:real-input-40-arity-3d} 的局部二次几何中，near-\(1\) 主极点的缺口与有限部漂移由同一组二阶不变量控制：Fisher 度量 \(g=\Sigma\succ 0\) 与有限部 Hessian \(H\)。
沿纯虚扭曲切片，空间--时间换算律（命题 \ref{prop:real-input-40-space-time-conversion-law}）把二者耦合为方向 Rayleigh 商 \(\mathcal{W}(v)=(v^{\mathsf T}Hv)/(v^{\mathsf T}\Sigma v)\)（定义 \ref{def:real-input-40-directional-coupling}）。
本节说明：当扭曲角被组织成多模数角格时，可实现的归一化漂移比在广义特征值带内呈现可控稠密性，并在体积比尺度下满足一条显式的经验分布律。

\begin{definition}[多模数角格与预算]\label{def:root-unit-angle-lattice}
固定维数 \(d\ge 2\) 与模数组 \(m=(m_1,\dots,m_d)\in(\ZZ_{\ge 2})^d\)，记预算
\[
B:=\prod_{k=1}^d m_k.
\]
定义未包裹角格
\[
L(m):=\frac{2\pi}{m_1}\ZZ\times\cdots\times \frac{2\pi}{m_d}\ZZ\subset\RR^d,
\qquad
\det L(m)=\frac{(2\pi)^d}{B},
\]
并令 \(\Lambda(m)\subset[-\pi,\pi]^d\) 为逐坐标取主值代表元后的包裹角格点集（除去零点）。
\end{definition}

\begin{definition}[归一化漂移比]\label{def:root-unit-normalized-drift-ratio}
沿用命题 \ref{prop:real-input-40-space-time-conversion-law} 的记号，令 \(\lambda=\rho(M_0)\)。
对小角 \(\vartheta\in\RR^d\) 定义
\[
s(\vartheta):=\frac{P(i\vartheta)}{\log\lambda},
\qquad
\Delta_{\mathfrak{M}}(\vartheta):=\log\mathfrak{M}(0)-\Re\log\mathfrak{M}(i\vartheta),
\]
并定义\textbf{归一化漂移比}
\[
R(\vartheta):=\frac{\Delta_{\mathfrak{M}}(\vartheta)}{\log\lambda\cdot(1-\Re s(\vartheta))}
\]
（当 \(\vartheta\) 足够小时分母为正）。
由命题 \ref{prop:real-input-40-space-time-conversion-law}，当 \(\vartheta\to 0\) 且 \(v(\vartheta):=\vartheta/\|\vartheta\|\) 时有
\[
R(\vartheta)=\mathcal{W}(v(\vartheta))+O(\|\vartheta\|^2),
\]
\[
\mathcal{W}(v):=\frac{v^{\mathsf T}Hv}{v^{\mathsf T}\Sigma v}\qquad(v\in\mathbb S^{d-1}).
\]
\end{definition}

\begin{theorem}[漂移比的连续谱带与可实现稠密性]\label{thm:root-unit-drift-ratio-dense-band}
\leanverified{paper\_root\_unit\_drift\_ratio\_dense\_band}
设 \(d\ge 2\)，并取 \(\Sigma\succ 0\) 与对称矩阵 \(H\)。
令
\[
\mu_{\min}:=\min_{\|v\|=1}\mathcal{W}(v),\quad
\mu_{\max}:=\max_{\|v\|=1}\mathcal{W}(v).
\]
则对任意 \(\mu\in[\mu_{\min},\mu_{\max}]\) 与任意 \(\eta>0\)，存在 \(B_0(\mu,\eta)\) 使得：若预算 \(B\ge B_0(\mu,\eta)\)，则必存在某个 \(\vartheta\in \Lambda(m)\setminus\{0\}\) 满足
\[
|R(\vartheta)-\mu|<\eta,
\qquad
1-\Re s(\vartheta)\ll B^{-2/d}.
\]
从而当 \(B\to\infty\) 时，可实现的漂移比集合 \(\{R(\vartheta):\vartheta\in\Lambda(m)\setminus\{0\}\}\) 在闭区间 \([\mu_{\min},\mu_{\max}]\) 内稠密。
\end{theorem}

\begin{proof}
函数 \(\mathcal{W}\) 在紧致球面 \(\mathbb S^{d-1}\) 上连续，故取到极值 \(\mu_{\min},\mu_{\max}\)；并且对任意 \(\mu\in[\mu_{\min},\mu_{\max}]\)，由介值性可取 \(u\in\mathbb S^{d-1}\) 使 \(\mathcal{W}(u)=\mu\)。
\par
取以 \(u\) 为轴的对称双锥凸体
\[
K_{L,\varepsilon}:=\{\,t u+w:\ |t|\le L,\ w\perp u,\ \|w\|\le \varepsilon |t|\,\},
\]
其体积满足 \(\mathrm{vol}(K_{L,\varepsilon})\asymp \varepsilon^{d-1}L^d\)。
取 \(L\) 使 \(\mathrm{vol}(K_{L,\varepsilon})>2^d\det L(m)\)，由 Minkowski 凸体定理可得
\(
x\in L(m)\setminus\{0\}\cap K_{L,\varepsilon}
\)。
当 \(B\) 足够大时可令 \(L<\pi\)，从而该 \(x\) 的逐坐标包裹不改变其值，故 \(x\in\Lambda(m)\setminus\{0\}\)。
\par
由 \(x\in K_{L,\varepsilon}\) 可知其方向 \(v(x)=x/\|x\|\) 与 \(u\) 的夹角由 \(\varepsilon\) 控制。
由 \(\mathcal{W}\) 的一致连续性，可取 \(\varepsilon\) 足够小使得 \(\|v(x)-u\|\) 足够小时有 \(|\mathcal{W}(v(x))-\mu|<\eta/2\)。
\par
另一方面，由定义 \ref{def:root-unit-normalized-drift-ratio} 中的二次耦合式，\(R(x)=\mathcal{W}(v(x))+O(\|x\|^2)\)。
而 Minkowski 构造给出 \(\|x\|\asymp (\det L(m))^{1/d}=\Theta(B^{-1/d})\)，故当 \(B\) 足够大时 \(|R(x)-\mathcal{W}(v(x))|<\eta/2\)。
合并即得 \(|R(x)-\mu|<\eta\)。
最后，由命题 \ref{prop:real-input-40-space-time-conversion-law} 的缺口二次近似 \(1-\Re s(x)\asymp x^{\mathsf T}\Sigma x/\log\lambda\) 与 \(\|x\|=O(B^{-1/d})\) 得 \(1-\Re s(x)\ll B^{-2/d}\)。
证毕。
\end{proof}

\begin{theorem}[漂移比的经验分布律]\label{thm:root-unit-drift-ratio-empirical-law}
\leanverified{paper\_root\_unit\_drift\_ratio\_empirical\_law}
在定义 \ref{def:root-unit-angle-lattice}--\ref{def:root-unit-normalized-drift-ratio} 的设定下，取 \(\varepsilon>0\) 并定义截断集合
\[
\mathcal P_{\varepsilon}(m):=\{\vartheta\in\Lambda(m)\setminus\{0\}:\ 1-\Re s(\vartheta)\le \varepsilon\}.
\]
在小角有效域内，上述条件与椭球约束
\(
\vartheta^{\mathsf T}\Sigma\vartheta\le 2\log\lambda\cdot \varepsilon
\)
主导等价。
并且在极限 \(\varepsilon\downarrow 0\)、\(B\to\infty\) 且 \(B\varepsilon^{d/2}\to\infty\) 的稳定体积比 regime 内，定义球面测度为 \(d\sigma\)，并令
\[
w_{\Sigma}(u):=(u^{\mathsf T}\Sigma u)^{-d/2}\quad(u\in\mathbb S^{d-1}),
\qquad
d\nu_{\Sigma}(u):=\frac{w_{\Sigma}(u)}{\int_{\mathbb S^{d-1}} w_{\Sigma}\,d\sigma}\,d\sigma(u).
\]
则对任意有界连续函数 \(\Phi:\RR\to\RR\)，有经验分布收敛
\[
\frac{1}{\#\mathcal P_{\varepsilon}(m)}
\sum_{\vartheta\in\mathcal P_{\varepsilon}(m)} \Phi\bigl(R(\vartheta)\bigr)
\;=\;
\int_{\mathbb S^{d-1}} \Phi\bigl(\mathcal{W}(u)\bigr)\,d\nu_{\Sigma}(u)\ +\ o(1),
\]
其中 \(\mathcal{W}(u)=(u^{\mathsf T}Hu)/(u^{\mathsf T}\Sigma u)\)。
\end{theorem}

\begin{proof}
令 \(R:=\sqrt{2\log\lambda\cdot\varepsilon}\)，并记椭球
\(
E_R:=\{x\in\RR^d:\ x^{\mathsf T}\Sigma x\le R^2\}
\)。
当 \(\varepsilon\) 足够小时，命题 \ref{prop:real-input-40-space-time-conversion-law} 的二次近似给出
\(
\mathcal P_{\varepsilon}(m)=\Lambda(m)\cap E_R+o(\#\mathcal P_{\varepsilon}(m))
\)（以计数意义理解）。
对任意球面可测集 \(A\subset\mathbb S^{d-1}\)，考虑锥域 \(C(A):=\{ru:\ r\ge 0,\ u\in A\}\)。
按几何数论的 Lipschitz 原理对凸体 \(E_R\cap C(A)\) 的格点计数作体积比估计，可得在 \(B\varepsilon^{d/2}\to\infty\) 下
\[
\#\bigl(\Lambda(m)\cap E_R\cap C(A)\bigr)
=\frac{1}{\det L(m)}\,\mathrm{vol}(E_R\cap C(A))+o(B\varepsilon^{d/2}).
\]
又对固定方向 \(u\)，椭球半径上界为 \(r_{\max}(u)=R/\sqrt{u^{\mathsf T}\Sigma u}\)，从而
\[
\mathrm{vol}(E_R\cap C(A))
=\int_A\left(\int_0^{r_{\max}(u)} r^{d-1}\,dr\right)d\sigma(u)
=\frac{R^d}{d}\int_A (u^{\mathsf T}\Sigma u)^{-d/2}\,d\sigma(u).
\]
与 \(A=\mathbb S^{d-1}\) 的总计数相除，即得方向经验测度收敛到 \(\nu_{\Sigma}\)。
\par
最后，由定义 \ref{def:root-unit-normalized-drift-ratio} 中 \(R(\vartheta)=\mathcal{W}(v(\vartheta))+O(\|\vartheta\|^2)\) 及 \(\|\vartheta\|^2=O(\varepsilon)\)，可得在 \(\mathcal P_{\varepsilon}(m)\) 上一致估计 \(R(\vartheta)=\mathcal{W}(v(\vartheta))+O(\varepsilon)\)。
令 \(\varepsilon\downarrow 0\)，由 \(\Phi\) 的一致连续性与方向经验测度收敛，结合连续映射定理得到所述经验分布极限。证毕。
\end{proof}

\begin{conjecture}[prime-register 角格的 Hermite 饱和与漂移比极值采样]\label{conj:root-unit-prime-register-hermite-saturation}
在小角 regime 内，记
\[
q(m):=\min_{\vartheta\in\Lambda(m)\setminus\{0\}}\ \vartheta^{\mathsf T}\Sigma\vartheta.
\]
在固定共体积类 \(\det L(m)=(2\pi)^d/B\) 上，对所有格形状的最优上确界由 Hermite 常数 \(\gamma_d\) 给出
\[
\sup_{\Lambda:\ \det\Lambda=(2\pi)^d/B}\ \min_{0\ne x\in\Lambda} x^{\mathsf T}\Sigma x
=\gamma_d(\det\Sigma)^{1/d}\Bigl(\frac{(2\pi)^d}{B}\Bigr)^{2/d}.
\]
设 \(\Lambda(m)\) 由 prime-register 机制生成（例如由互异素数与可审计的外置协议给出一族模数组 \(m^{(n)}\)，其预算 \(B_n=\prod_k m^{(n)}_k\to\infty\)）。
则存在无穷子列使得
\[
\limsup_{n\to\infty}\ \frac{B_n^{2/d}}{(2\pi)^2(\det\Sigma)^{1/d}}\ q\bigl(m^{(n)}\bigr)=\gamma_d.
\]
并且若 \(\vartheta_n\) 取到 \(q(m^{(n)})\) 的最短向量，则其归一化方向 \(u_n=\vartheta_n/\|\vartheta_n\|\) 在球面上按定理 \ref{thm:root-unit-drift-ratio-empirical-law} 的权测度 \(\nu_{\Sigma}\) 等分布，从而 \(R(\vartheta_n)\) 在 \([\mu_{\min},\mu_{\max}]\) 上对广义特征方向给出无偏采样。
\end{conjecture}

\subsection{伸缩链的双残差正交分解：角色谱通道与精度势账本通道的张量化闭环}\label{subsec:dilation-character-precision-tensor}

文稿在 Dirichlet 变量上引入 Witt--dilation（伸缩）算子（\S\ref{app:real-input-40-dirichlet}）
\[
(Wf)(s):=\sum_{m\ge 1}\frac{1}{m}\,f(ms),
\]
并给出严格恒等式 \(\log\mathcal{Z}(s)=(W\mathcal{P})(s)\) 及 Möbius 反演
\(\mathcal{P}(s)=\sum_{m\ge 1}\frac{\mu(m)}{m}\,\log\mathcal{Z}(ms)\)。

将“尺度链”解释为深度线上的平移卷积，并与 \ref{subsec:root-unit-completion-orbit-character} 的角色表示合并，可得到无需额外结构的张量化闭环。

\leanverified{paper\_character\_precision\_tensor}
\begin{theorem}[角色--精度的正交张量化]\label{thm:character-precision-tensor}
取深度坐标 \(y\in\RR\) 的平移表示 \(T_{\log m}g(y)=g(y+\log m)\)，并取根单位点轨道模上的角色表示 \(\mathsf{R}_m\)（定理 \ref{thm:root-unit-character-diagonalization}）。对 \(\sigma>0\) 定义联合伸缩算子
\[
\mathcal{K}_{\sigma,q}:=\sum_{m\ge 1}\frac{1}{m^{1+\sigma}}\,T_{\log m}\otimes \mathsf{R}_m
\quad\text{作用于}\quad L^2(\RR,\dd y)\otimes \CC^{\mathcal{S}_q}.
\]
则在定理 \ref{thm:root-unit-character-diagonalization} 的角色直和分解下，\(\mathcal{K}_{\sigma,q}\) 分解为标量卷积算子族的直和：
\[
\mathcal{K}_{\sigma,q}\ \simeq\ \bigoplus_{\chi(-1)=1}\ \Bigl(\sum_{m\ge 1}\frac{\chi(m)}{m^{1+\sigma}}\,T_{\log m}\Bigr),
\]
并且对每个偶角色分量，其“零频”本征值为 \(L(1+\sigma,\chi)\)。

因此，伸缩链的残差空间在体系内部具有可检验的正交二分：
\begin{enumerate}
  \item \textbf{角色谱通道}（有限维）：由 \(q\) 控制，产生 \(L(1+\sigma,\chi)\) 的离散谱；
  \item \textbf{精度势账本通道}（无限维）：深度平移的连续通道，承载压缩后的精度预算。
\end{enumerate}
该正交分解与文稿的“整数层审计先行、谱层指标后置”的分层原则一致：整数层（如 Frobenius 缺陷与 Adams 结构）锁定可实现性与一致性，谱层的伸缩残差再通过角色谱通道与精度势账本通道分离进入可核对的通道化分析。
\end{theorem}
