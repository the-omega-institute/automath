\section{比值结式、Chebyshev 压缩与判别式平方类}\label{app:ratio-resultant-discriminant-rigidity}

本附录固定一类在低信息损耗下连接“比值结式/互反压缩”与原多项式判别式二次域、分裂域结构的代数机制。正文中出现的相位共振结果式与压缩判别式计算可视为该机制在特定谱多项式上的实现。

\subsection{设定与比值结式}
设 \(K\) 为特征不等于 \(2\) 的域，取首一可分多项式 \(f(x)\in K[x]\) 满足 \(\deg f=n\ge 2\) 且 \(f(0)\neq 0\)。
记 \(L/K\) 为 \(f\) 的分裂域，并在 \(\bar K\) 中记根为 \(\alpha_1,\dots,\alpha_n\)。

\begin{definition}[比值结式与归一化互反多项式]\label{def:ratio-resultant}
定义关于伸缩参数 \(u\) 的比值结式
\[
\mathcal R_f(u):=\Res_x\bigl(f(x),f(ux)\bigr)\in K[u],
\]
以及去除平凡因子后的归一化多项式
\[
\Phi_f(u):=(-1)^n\frac{\mathcal R_f(u)}{f(0)(u-1)^n}\in K[u].
\]
\end{definition}

\begin{theorem}[比值结式的二次判别式刚性]\label{thm:ratio-resultant-disc-rigidity}
\leanverified{paper\_ratio\_resultant\_disc\_rigidity}
在平方类群 \(K^{\times}/(K^{\times})^{2}\) 中成立
\[
\Disc_u(\Phi_f)\ \equiv\ \Disc_x(f).
\]
等价地，
\[
K\!\left(\sqrt{\Disc_u(\Phi_f)}\right)=K\!\left(\sqrt{\Disc_x(f)}\right).
\]
\end{theorem}
\begin{proof}
在分裂域 \(L\) 中写
\[
f(x)=\prod_{i=1}^n(x-\alpha_i),
\qquad
f(ux)=u^n\prod_{j=1}^n\left(x-\frac{\alpha_j}{u}\right).
\]
由结式的根表达式得
\[
\mathcal R_f(u)=\prod_{i=1}^n\prod_{j=1}^n (u\alpha_i-\alpha_j).
\]
其中 \(i=j\) 的因子给出
\[
\prod_{i=1}^n(u\alpha_i-\alpha_i)=(u-1)^n\prod_{i=1}^n\alpha_i=(u-1)^n f(0),
\]
从而 \(\Phi_f(u)\) 的根集（按集合计）恰为所有比值 \(\alpha_j/\alpha_i\)（\(i\neq j\)）。
令
\[
\Omega:=\{(i,j):1\le i,j\le n,\ i\neq j\}.
\]
对 \(\sigma\in S_n\) 定义其在 \(\Omega\) 上的诱导置换
\(
\tau(\sigma):(i,j)\mapsto(\sigma(i),\sigma(j))
\)。
将 \(\Phi_f\) 的根按 \(\Omega\) 重标号时，\(\sqrt{\Disc_u(\Phi_f)}\)（Vandermonde 型乘积）按 \(\operatorname{sgn}(\tau(\sigma))\) 变号；而
\(\sqrt{\Disc_x(f)}=\prod_{i<j}(\alpha_i-\alpha_j)\)
按 \(\operatorname{sgn}(\sigma)\) 变号。
因此只需证明同态 \(\sigma\mapsto \operatorname{sgn}(\tau(\sigma))\) 与 \(\operatorname{sgn}(\sigma)\) 在 \(S_n\) 上一致。

取换位 \((12)\in S_n\)。
其在 \(\Omega\) 上的作用分解为互不相交的换位：
\((1,2)\leftrightarrow(2,1)\) 一对；
对每个 \(k\ge 3\)，有 \((1,k)\leftrightarrow(2,k)\) 一对；
以及 \((k,1)\leftrightarrow(k,2)\) 一对；
其余元素不动。
于是 \(\tau((12))\) 含 \(1+(n-2)+(n-2)=2n-3\) 个换位，故为奇置换，
\[
 \operatorname{sgn}(\tau((12)))=-1=\operatorname{sgn}((12)).
\]
由于 \(S_n\to\{\pm1\}\) 的非平凡同态唯一为符号表示，得到 \(\operatorname{sgn}\circ\tau=\operatorname{sgn}\)。
两判别式诱导同一二次特征，等价于其平方类相同。
\end{proof}

\begin{theorem}[Chebyshev 压缩下的判别式奇偶律]\label{thm:ratio-resultant-chebyshev-parity}
令 \(m:=n(n-1)\)，则 \(\deg\Phi_f=m\)，且 \(\Phi_f\) 自互反：
\[
u^{m}\Phi_f(1/u)=\Phi_f(u).
\]
因此存在唯一 \(\Psi_f(s)\in K[s]\)（\(\deg \Psi_f=m/2\)）使得
\[
\Phi_f(u)=u^{m/2}\Psi_f\!\bigl(u+u^{-1}\bigr).
\]
若 \(\Psi_f\) 可分（等价于 \(\Disc_s(\Psi_f)\neq 0\)），则在 \(K^{\times}/(K^{\times})^{2}\) 中有
\[
\Disc_s(\Psi_f)\ \equiv\ \Disc_x(f)^{\,n-2}.
\]
特别地，当 \(n\) 为偶数时 \(\Disc_s(\Psi_f)\) 在 \(K\) 中为平方；当 \(n\) 为奇数时
\[
K\!\left(\sqrt{\Disc_s(\Psi_f)}\right)=K\!\left(\sqrt{\Disc_x(f)}\right).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:ratio-resultant-disc-rigidity} 的根描述，\(\Phi_f\) 的根集为 \(\{\alpha_j/\alpha_i:i\neq j\}\)，并对取逆封闭。
因此 \(\Phi_f\) 为自互反多项式，且 \(\deg\Phi_f=\card(\Omega)=n(n-1)=m\)。
对任意自互反多项式 \(P(u)\in K[u]\)（\(\deg P\) 为偶数，且常数项非零），存在唯一 \(Q(s)\in K[s]\) 使 \(P(u)=u^{\deg P/2}Q(u+u^{-1})\)；将此应用于 \(P=\Phi_f\) 即得 \(\Psi_f\) 的存在唯一性。

令
\[
s_{ij}:=\frac{\alpha_i}{\alpha_j}+\frac{\alpha_j}{\alpha_i}\qquad (1\le i<j\le n).
\]
由压缩关系可知 \(\Psi_f(s)\) 的根（按集合计）恰为所有 \(s_{ij}\)，并由对称性得 \(\Psi_f\in K[s]\)。

记 \(\binom{[n]}{2}\) 为 \(2\)-子集集合。
置换 \(\sigma\in S_n\) 在其上诱导置换 \(\rho(\sigma):\{i,j\}\mapsto\{\sigma(i),\sigma(j)\}\)。
将 \(\Psi_f\) 的根按 \(2\)-子集重标号时，\(\sqrt{\Disc_s(\Psi_f)}\) 按 \(\operatorname{sgn}(\rho(\sigma))\) 变号；
因此 \(\Disc_s(\Psi_f)\) 所对应的二次特征为 \(\operatorname{sgn}\circ\rho\)。

计算换位 \((12)\) 在 \(2\)-子集上的符号：
\(\{1,2\}\) 不动；
对每个 \(k\ge 3\)，\(\{1,k\}\leftrightarrow\{2,k\}\) 构成一对换位，共 \(n-2\) 对；
其余不动。
故
\[
 \operatorname{sgn}(\rho((12)))=(-1)^{n-2}=\operatorname{sgn}((12))^{\,n-2}.
\]
同理由于 \(S_n\to\{\pm1\}\) 的同态由换位取值决定，得到
\(
 \operatorname{sgn}\circ\rho=\operatorname{sgn}^{\,n-2}
\)。
而 \(\Disc_x(f)\) 对应 \(\operatorname{sgn}\)，从而 \(\Disc_x(f)^{\,n-2}\) 对应 \(\operatorname{sgn}^{\,n-2}\)。
两者诱导同一二次扩张，等价于平方类相同。
\end{proof}

\subsection{比值域与分裂域}
记
\[
M:=K\bigl(\alpha_i/\alpha_j:\ i\neq j\bigr)\subseteq L.
\]

\begin{proposition}[比值域与分裂域的次数整除律及其对称性判别]\label{prop:ratio-field-splitting}
在上述设定下：
\begin{enumerate}
  \item 存在等式 \(L=M(\alpha_1)\)，并且 \([L:M]\mid n\)。
  \item 设 \(G:=\Gal(L/K)\) 作用于比值集合 \(\{\alpha_i/\alpha_j:i\neq j\}\) 的自然同态为
  \[
  \pi:G\to \Sym\bigl(\{\alpha_i/\alpha_j:i\neq j\}\bigr),
  \]
  并记 \(H:=\ker\pi\)。则 \(H\) 为循环群，且其阶 \(\card(H)=d\) 满足 \(d\mid n\)。
  \item \(H\neq 1\) 当且仅当存在单位根 \(\zeta\neq 1\) 使 \(f(\zeta x)=f(x)\)。
  在首一且 \(f(0)\neq 0\) 的情形，这进一步等价于存在 \(d>1\) 与 \(g(x)\in K[x]\) 使
  \(
  f(x)=g(x^d)
  \)；
  并且此时 \([L:M]=d\)，扩张 \(L/M\) 为循环扩张。
\end{enumerate}
\end{proposition}
\begin{proof}
\textbf{(1)} 由于 \(\alpha_i=(\alpha_i/\alpha_1)\alpha_1\) 且 \(\alpha_i/\alpha_1\in M\)，故 \(L\subseteq M(\alpha_1)\)，反向包含显然，得 \(L=M(\alpha_1)\)。
又由 \(f(0)=\prod_{i=1}^n\alpha_i\) 有
\[
\alpha_1^n=\frac{\prod_{i=1}^n\alpha_i}{\prod_{i=2}^n(\alpha_i/\alpha_1)}
=\frac{f(0)}{\prod_{i=2}^n(\alpha_i/\alpha_1)}\in M,
\]
从而 \(\alpha_1\) 在 \(M\) 上的次数整除 \(n\)，即 \([L:M]\mid n\)。

\textbf{(2)} 若 \(\sigma\in H\)，则对一切 \(i\) 有
\(
\sigma(\alpha_i/\alpha_1)=\alpha_i/\alpha_1
\)，从而存在 \(c_{\sigma}:=\sigma(\alpha_1)/\alpha_1\in L^{\times}\) 使
\(
\sigma(\alpha_i)=c_{\sigma}\alpha_i
\) 对所有 \(i\) 成立。
由于 \(\sigma\) 置换根集，集合 \(\{c_{\sigma}\alpha_i\}_{i=1}^n\) 与 \(\{\alpha_i\}_{i=1}^n\) 相同，比较乘积得 \(c_{\sigma}^n=1\)。
并且 \(c_{\sigma}=\sigma(\alpha_i)/\alpha_i\) 为两个根之比，故对任意 \(\tau\in H\) 有 \(\tau(c_{\sigma})=c_{\sigma}\)。
于是映射
\[
H\to \mu_n(L),\qquad \sigma\mapsto c_{\sigma}
\]
为群同态且单射，故 \(H\) 同构于循环群 \(\mu_n(L)\) 的某个子群，从而 \(H\) 循环，且 \(\card(H)=d\mid n\)。

\textbf{(3)} 若 \(H\neq 1\)，取非平凡 \(\sigma\in H\)，令 \(\zeta:=c_{\sigma}\neq 1\)。
则 \(\sigma(\alpha_i)=\zeta\alpha_i\) 对所有 \(i\) 成立，故根集对乘以 \(\zeta\) 封闭，从而 \(f(\zeta x)\) 与 \(f(x)\) 在 \(\bar K\) 中具有同一根集；二者又同为首一同次数多项式，故 \(f(\zeta x)=f(x)\)。
反之若存在单位根 \(\zeta\neq 1\) 使 \(f(\zeta x)=f(x)\)，则根集在乘以 \(\zeta\) 下不变，给出 \(L\) 的一个 \(K\)-自同构将每个根送至 \(\zeta\) 倍的某个根；该自同构固定全部比值 \(\alpha_i/\alpha_j\)，故 \(H\neq 1\)。

在首一且 \(f(0)\neq 0\) 时，将 \(f(x)=\sum_{k=0}^n a_k x^k\) 代入 \(f(\zeta x)=f(x)\) 得 \(a_k(\zeta^k-1)=0\)。
令 \(d:=\ord(\zeta)>1\)，则 \(a_k\neq 0\Rightarrow d\mid k\)，从而 \(f(x)\in K[x^d]\)，即存在 \(g\in K[x]\) 使 \(f(x)=g(x^d)\)。
此时 \(\card(H)=d\)，并由 (1) 得 \([L:M]=d\)，而 \(L/M\) 由 \(H\) 生成，故为循环扩张。
\end{proof}

\begin{corollary}[全对称群下的比值完备性]\label{cor:ratio-field-full-symmetric}
设 \(n\ge 4\)，并假设 \(\Gal(f/K)\cong S_n\)。
则命题 \ref{prop:ratio-field-splitting} 中的核 \(H\) 必为平凡群，从而
\[
M=L.
\]
因此 \(\Phi_f(u)\) 的分裂域与 \(f\) 的分裂域一致。
此外，若 \(\Phi_f\) 可分且比值 \(\alpha_i/\alpha_j\)（\(i\neq j\)）两两不相等，则 \(\Phi_f\) 在 \(K\) 上不可约且 \(\deg\Phi_f=n(n-1)\)。
\end{corollary}
\begin{proof}
由命题 \ref{prop:ratio-field-splitting}，\(H\) 为循环正规子群。
对 \(n\ge 4\)，\(S_n\) 不含非平凡循环正规子群，因此 \(H=1\) 且 \(M=L\)。

在“比值互异”假设下，\(\Phi_f\) 的根可与 \(\Omega\) 双射对应。
由于 \(S_n\) 在 \(\Omega\) 上的作用是传递的，任一根的 \(G\)-轨道大小为 \(\card(\Omega)=n(n-1)\)；
而 \(H=1\) 保证该作用忠实，从而该根在 \(K\) 上的最小多项式次数为 \(n(n-1)\)，即 \(\Phi_f\) 不可约。
\end{proof}
