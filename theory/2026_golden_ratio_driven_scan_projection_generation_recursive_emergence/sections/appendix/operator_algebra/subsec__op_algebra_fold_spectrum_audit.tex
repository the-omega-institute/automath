\begin{corollary}[奇异值谱—退化直方图的完备互译]\label{cor:fold-conditional-expectation-spectrum-histogram}

\end{corollary}

\par
\textbf{(2)} 由命题 \ref{prop:fold-fiber-multiplicity-trace-moments}，
\(\Tr(\mathcal{F}_m^{t})=\sum_x d_m(x)^{t}\) 给出 \(\mathcal{F}_m\) 的幂和。
对一个 \(r\times r\) 矩阵（此处 \(r=|X_m|\)），幂和 \(\Tr(\mathcal{F}_m^{t})\)（\(t=1,\dots,r\)）由 Newton 恒等式唯一确定特征多项式的系数，
从而确定其本征值多重集，即 \(\{d_m(x)\}\)。证毕。
\leanverified{paper\_fold\_conditional\_expectation\_spectrum\_histogram}
\end{proof}

\begin{proposition}[反射审计的谱交错界：稳定层证书对微观谱的两侧界]\label{prop:fold-audit-spectral-interlacing}
固定分辨率 \(m\)，令 \(N:=|\Omega_m|=2^m\)、\(r:=|X_m|\)。
在 Hilbert 空间 \(\ell^2(\Omega_m)\)（计数内积）上，令 \(U_m:\ell^2(X_m)\to \ell^2(\Omega_m)\) 为等距嵌入
\[
(U_m g)(a):=\frac{g(\Fold_m(a))}{\sqrt{d_m(\Fold_m(a))}},\qquad a\in\Omega_m,
\]
并令 \(\Pi_m:=U_mU_m^\ast\) 为到其像空间的正交投影（等价地，\(\Pi_m=\iota_m\circ E_m\) 在 \(\ell^2\) 口径下的实现）。
对任意自伴算子 \(T\)（\(\ell^2(\Omega_m)\) 上）定义稳定层压缩与反射后可见算子
\[
T_m^{\mathrm{stab}}:=U_m^\ast T U_m\quad(\ell^2(X_m)\ \text{上}),
\qquad
T_m^{\mathrm{aud}}:=\Pi_m T \Pi_m\quad(\ell^2(\Omega_m)\ \text{上}).
\]
记 \(T\) 的本征值按降序为 \(\lambda_1(T)\ge\cdots\ge\lambda_N(T)\)，
记 \(T_m^{\mathrm{stab}}\)（等价于 \(T_m^{\mathrm{aud}}\) 在 \(\mathrm{Im}(\Pi_m)\) 上的限制）的本征值按降序为
\(\lambda_1(T_m^{\mathrm{stab}})\ge\cdots\ge\lambda_r(T_m^{\mathrm{stab}})\)。
则对一切 \(k=1,\dots,r\) 成立压缩交错界
\[
\boxed{\;
\lambda_k(T)\ \ge\ \lambda_k(T_m^{\mathrm{stab}})\ \ge\ \lambda_{k+N-r}(T).
\;}
\]
从而 \(\|T_m^{\mathrm{stab}}\|_{op}\le \|T\|_{op}\)。
因此，任何在稳定层 \(\ell^2(X_m)\) 上形成的谱半径/谱隙证书自动推出对微观算子 \(T\) 的两侧谱界证书。
\end{proposition}
\begin{proof}
\(U_m\) 为等距嵌入，故 \(T_m^{\mathrm{stab}}\) 是 \(T\) 在子空间 \(\mathrm{Im}(U_m)=\mathrm{Im}(\Pi_m)\) 上的正交压缩。
压缩交错界是 Hermitian 算子在正交压缩下的标准 Cauchy 交错定理（等价地，由 min--max 原理直接推出）。
算子范数不等式由交错界对最大/最小本征值的夹逼立即得到。证毕。
\end{proof}

\subsubsection{Jones 基本构造的闭式分解与纤维矩阵单位：退化直方图的代数完备证书}\label{subsubsec:op-algebra-fold-jones-basic-construction}
本节前文把折叠条件期望 \(E_m\) 的奇异值谱与退化直方图互译为等价审计接口。
在有限维交换代数口径下，进一步可将“纤维直方图 \(\{d_m(x)\}\)”提升为一个完全显式的 \emph{Jones 基本构造}分解：
基本构造代数分裂为各纤维上的全矩阵代数直和，并且存在一组由点投影与 Jones 投影生成的规范矩阵单位。

\begin{theorem}[折叠包含的 Jones 基本构造闭式：直接和矩阵代数分解与深度二完备性]\label{thm:fold-jones-basic-construction-directsum}
\leanverified{paper\_op\_algebra\_fold\_jones\_basic\_construction\_directsum}
固定分辨率 \(m\)。
令
\(
A_m:=\CC^{\Omega_m}
\)
为微态集合 \(\Omega_m\) 上的复值函数代数（等价地：\(\ell^2(\Omega_m)\) 上的对角算子代数），
令
\(
B_m:=\CC^{X_m}
\)
为稳定层 \(X_m\) 上的函数代数，并令 \(\iota_m:B_m\hookrightarrow A_m\) 为拉回嵌入 \((\iota_m f)(\omega)=f(\Fold_m(\omega))\)。
取纤维平均条件期望 \(E_m:A_m\to B_m\)（命题 \ref{prop:fold-conditional-expectation}），并令
\[
e_m:=\iota_m\circ E_m:\ \ell^2(\Omega_m)\to \ell^2(\Omega_m)
\]
为对应的（\(\ell^2\) 口径下）Jones 投影，即对每条纤维做算术平均并回填到该纤维上的正交投影。
则基本构造代数满足自然同构
\[
\boxed{\;
\langle A_m,e_m\rangle
\ \cong\
\bigoplus_{x\in X_m} M_{d_m(x)}(\CC)\; }.
\]
因此该包含 \(B_m\subset A_m\) 的全部有限维标准不变量在本层级上由退化直方图
\(
\{d:\#\{x\in X_m:d_m(x)=d\}\}
\)
完全决定；并且其主图为以 \(x\in X_m\) 为中心、度数 \(d_m(x)\) 的星形图之不交并，从而在 Jones 塔意义下为深度二对象（基本构造在一步后闭合）。
\end{theorem}

\begin{proof}[证明要点]
在标准基 \(\{\delta_\omega\}_{\omega\in\Omega_m}\) 下，\(\ell^2(\Omega_m)\) 分解为纤维直和
\[
\ell^2(\Omega_m)=\bigoplus_{x\in X_m}\ell^2\!\bigl(\Fold_m^{-1}(x)\bigr).
\]
代数 \(A_m\) 为全体对角算子，故保持该分解且在每个纤维子空间上包含所有坐标点投影；
而 \(e_m\) 在每个纤维上是秩 \(1\) 的平均投影，亦保持该直和分解。
因此 \(\langle A_m,e_m\rangle\) 在上述分解下为块对角代数，其每一块由“纤维上的点投影 + 纤维平均投影”生成。
由命题 \ref{prop:fold-jones-matrix-units}（见下）给出的规范矩阵单位可知：每条纤维上的生成代数恰为 \(M_{d_m(x)}(\CC)\)。
对 \(x\) 取直和即得所示同构。
主图/深度二陈述在该情形下等价于“块分解为星形不交并且一步闭合”，由直和结构直接读出。证毕。
\end{proof}

\begin{proposition}[纤维内矩阵元的规范生成：由点投影与 \texorpdfstring{$e_m$}{e_m} 生成全矩阵]\label{prop:fold-jones-matrix-units}
\leanverified{paper\_fold\_jones\_matrix\_units}
沿用定理 \ref{thm:fold-jones-basic-construction-directsum} 的设定。
对任意 \(x\in X_m\)，记纤维点投影族为
\(
\{P_\omega\}_{\omega\in \Fold_m^{-1}(x)}\subset A_m
\)
（在 \(\ell^2(\Omega_m)\) 上即为 \(\delta_\omega\) 的坐标投影）。
对 \(\omega,\omega'\in \Fold_m^{-1}(x)\) 定义
\[
E_{\omega,\omega'}:=d_m(x)\,P_\omega\,e_m\,P_{\omega'}\ \in\ \langle A_m,e_m\rangle.
\]
则在每条纤维上成立矩阵元关系
\[
\boxed{\;
E_{\omega,\omega'}E_{\eta,\eta'}=\delta_{\omega',\eta}\,E_{\omega,\eta'},
\qquad
E_{\omega,\omega'}^{\ast}=E_{\omega',\omega}\; }.
\]
从而 \(\{E_{\omega,\omega'}\}_{\omega,\omega'\in\Fold_m^{-1}(x)}\) 构成 \(M_{d_m(x)}(\CC)\) 的一组规范矩阵单位，并给出一份完全可审计的基；
特别地，“纤维内的全部可逆微观重标号”在代数层面被实现为
\(
\prod_{x\in X_m}S_{d_m(x)}
\)
对该矩阵单位基的置换作用。
\end{proposition}

\begin{proof}
注意 \(e_m\) 在纤维 \(\Fold_m^{-1}(x)\) 上为平均投影：
\(
e_m\,\delta_{\omega'}=d_m(x)^{-1}\sum_{\eta\in\Fold_m^{-1}(x)}\delta_\eta
\)。
因此
\(
P_\omega e_m P_{\omega'}\delta_{\omega'}
=d_m(x)^{-1}\delta_\omega
\)，
从而 \(E_{\omega,\omega'}\delta_{\omega'}=\delta_\omega\)，且 \(E_{\omega,\omega'}\) 在该纤维外为零。
于是直接计算得到矩阵单位乘法关系与伴随关系；这与标准矩阵单位 \(e_{\omega,\omega'}\) 在基 \(\{\delta_\omega\}\) 下的作用完全一致。证毕。
\end{proof}

\begin{proposition}[容量曲线的折点与纤维谱的层数]\label{prop:op-algebra-capacity-kinks-equal-spectrum-levels}
\leanverified{paper\_op\_algebra\_capacity\_kinks\_equal\_spectrum\_levels}
对固定分辨率 \(m\)，令纤维谱为多重集合 \(\{d_m(x)\}_{x\in X_m}\)。
定义连续容量曲线
\[
\mathcal C_m^{\mathrm{cont}}(T):=\sum_{x\in X_m}\min\bigl(d_m(x),T\bigr),\qquad T\ge 0,
\]
并定义尾计数函数
\[
N_m(T):=\#\{x\in X_m:\ d_m(x)\ge T\},\qquad T\ge 0.
\]
则 \(\mathcal C_m^{\mathrm{cont}}\) 为分段线性凹函数，且几乎处处满足
\[
\frac{\dd}{\dd T}\mathcal C_m^{\mathrm{cont}}(T)=N_m(T).
\]
其不可导点集合恰为纤维谱的不同取值集合 \(\{d_m(x):x\in X_m\}\)。
特别地，\(\mathcal C_m^{\mathrm{cont}}\) 的折点数（不计 \(T=0\)）等于纤维谱不同取值的个数。
\end{proposition}
\begin{proof}
对任意整数 \(d\ge 0\)，函数 \(T\mapsto \min(d,T)\) 在 \(T=d\) 处出现斜率跳变，其余处线性；其导数几乎处处等于 \(\mathbf 1_{\{T\le d\}}\)。
对 \(x\in X_m\) 求和即得 \((\mathcal C_m^{\mathrm{cont}})'(T)=\sum_x \mathbf 1_{\{T\le d_m(x)\}}=N_m(T)\) 于几乎处处成立。
因此 \(\mathcal C_m^{\mathrm{cont}}\) 的斜率跳变点仅可能位于某个 \(d_m(x)\) 处；反之若某值 \(t\) 在谱中出现，则对应项在 \(T=t\) 处产生非零斜率跳变，故 \(t\) 必为折点。证毕。
\end{proof}

\begin{theorem}[限秩投影的变分极值与连续容量曲线]\label{thm:op-algebra-oracle-projection-variational-capacity}
\leanverified{paper\_op\_algebra\_oracle\_projection\_variational\_capacity}
沿用定理 \ref{thm:fold-jones-basic-construction-directsum} 的分解
\(
\langle A_m,e_m\rangle\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC)
\)。
对整数 \(L\ge 0\)，令 \(\mathcal P_m(L)\) 为所有满足逐块秩约束的投影集合：
\[
\mathcal P_m(L):=
\Bigl\{
p\in \langle A_m,e_m\rangle:\ p^2=p=p^\ast,\ 
\mathrm{rank}(p|_{M_{d_m(x)}(\CC)})\le L,\ \forall x\in X_m
\Bigr\}.
\]
则有严格变分恒等式
\[
\max_{p\in\mathcal P_m(L)}\ \sum_{x\in X_m}\mathrm{rank}\bigl(p|_{M_{d_m(x)}(\CC)}\bigr)
=\sum_{x\in X_m}\min\bigl(d_m(x),L\bigr)
=\mathcal C_m^{\mathrm{cont}}(L).
\]
并且对每个 \(x\) 的块内极大化子在酉共轭意义下唯一：其块分量可取为秩 \(\min(d_m(x),L)\) 的标准对角投影。
\end{theorem}
\begin{proof}
在直和分解下，任一投影 \(p\) 等价于一族块投影 \(p_x\in M_{d_m(x)}(\CC)\)。
约束 \(\mathrm{rank}(p_x)\le L\) 逐块独立，而目标函数 \(\sum_x \mathrm{rank}(p_x)\) 亦为逐块可加。
因此全局最大化分解为对每个 \(x\) 分别最大化 \(\mathrm{rank}(p_x)\)。
在 \(M_d(\CC)\) 中，投影的秩可取 \(0,1,\dots,d\) 的整数，故在约束 \(\le L\) 下最大秩为 \(\min(d,L)\)；
对所有块求和即得
\(
\max \sum_x \mathrm{rank}(p_x)=\sum_x \min(d_m(x),L)=\mathcal C_m^{\mathrm{cont}}(L)
\)。
块内唯一性由“同秩投影在 \(M_d(\CC)\) 中必酉共轭”等价分类给出。证毕。
\end{proof}

\begin{corollary}[容量曲线的离散可逆性：由整数点取值恢复纤维谱多重集合]\label{cor:op-algebra-capacity-curve-complete-invariant}
\leanverified{paper\_op\_algebra\_capacity\_curve\_complete\_invariant}
令 \(\mathcal C_m^{\mathrm{cont}}(T)=\sum_{x\in X_m}\min(d_m(x),T)\)。
则仅由整数点序列 \(\{\mathcal C_m^{\mathrm{cont}}(t)\}_{t\in\NN}\) 可唯一恢复纤维谱的多重集合：
\begin{enumerate}
  \item 对 \(t\ge 1\)，有尾计数恢复公式
  \[
  N_m(t)=\mathcal C_m^{\mathrm{cont}}(t)-\mathcal C_m^{\mathrm{cont}}(t-1).
  \]
  \item 对 \(t\ge 1\)，出现次数满足
  \[
  \#\{x\in X_m:\ d_m(x)=t\}=N_m(t)-N_m(t+1).
  \]
\end{enumerate}
因此 \(\mathcal C_m^{\mathrm{cont}}\) 与 \(\{d_m(x)\}_{x\in X_m}\) 互相唯一确定；
结合定理 \ref{thm:fold-jones-basic-construction-directsum}，\(\mathcal C_m^{\mathrm{cont}}\) 在固定 \(m\) 上等价决定 Jones 基本构造代数 \(\langle A_m,e_m\rangle\) 的 \(^\ast\)-同构类型。
\end{corollary}
\begin{proof}
对单项 \(c_d(t):=\min(d,t)\)（\(t\in\NN\)），其差分满足
\(
c_d(t)-c_d(t-1)=\mathbf 1_{\{d\ge t\}}
\)。
对 \(x\) 求和得
\(
\mathcal C_m^{\mathrm{cont}}(t)-\mathcal C_m^{\mathrm{cont}}(t-1)=\sum_x \mathbf 1_{\{d_m(x)\ge t\}}=N_m(t)
\)，得到第一式。
第二式由
\(
\#\{d=t\}=\#\{d\ge t\}-\#\{d\ge t+1\}
\)
立得。
最后，“有限维 \(C^\ast\)-代数的 \(^\ast\)-同构分类由矩阵块大小多重集合决定”，与定理 \ref{thm:fold-jones-basic-construction-directsum} 合并即得结论。证毕。
\end{proof}

\begin{theorem}[规范群体积的离散 Stieltjes--Dirichlet 表示]\label{thm:op-algebra-fold-gauge-log-volume-stieltjes}
\leanverified{paper\_op\_algebra\_fold\_gauge\_log\_volume\_stieltjes}
沿用命题 \ref{prop:op-algebra-fold-visible-permutation-lifts} 与推论 \ref{cor:op-algebra-capacity-curve-complete-invariant} 的记号，记
\[
G_m:=\{\tau\in\mathrm{Sym}(\Omega_m):\ \Fold_m\circ\tau=\Fold_m\},
\qquad
N_m(k):=\#\{x\in X_m:\ d_m(x)\ge k\}\quad(k\ge 1).
\]
则有精确恒等式
\[
\log|G_m|
=
\sum_{k\ge 1}N_m(k)\log k
=
\sum_{k\ge 1}\bigl(\mathcal C_m^{\mathrm{cont}}(k)-\mathcal C_m^{\mathrm{cont}}(k-1)\bigr)\log k.
\]
若再定义
\[
\zeta_m^{\mathrm{tail}}(s):=\sum_{k\ge 1}N_m(k)\,k^{-s},
\]
则该和实际上有限，并满足
\[
-\frac{\dd}{\dd s}\zeta_m^{\mathrm{tail}}(s)\Big|_{s=0}=\log|G_m|.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:op-algebra-fold-visible-permutation-lifts}，
\[
|G_m|=\prod_{x\in X_m}d_m(x)!.
\]
取对数并展开阶乘，得到
\[
\log|G_m|
=
\sum_{x\in X_m}\log(d_m(x)!)
=
\sum_{x\in X_m}\sum_{k=1}^{d_m(x)}\log k.
\]
交换有限求和次序即得
\[
\log|G_m|
=
\sum_{k\ge 1}\#\{x\in X_m:\ d_m(x)\ge k\}\log k
=
\sum_{k\ge 1}N_m(k)\log k.
\]
再由推论 \ref{cor:op-algebra-capacity-curve-complete-invariant}，
\[
N_m(k)=\mathcal C_m^{\mathrm{cont}}(k)-\mathcal C_m^{\mathrm{cont}}(k-1),
\]
代回即得第二个表示式。

最后，由于 \(N_m(k)=0\) 对一切 \(k>\max_{x\in X_m}d_m(x)\) 成立，\(\zeta_m^{\mathrm{tail}}(s)\) 实为有限 Dirichlet 多项式，可逐项求导：
\[
-\frac{\dd}{\dd s}\zeta_m^{\mathrm{tail}}(s)\Big|_{s=0}
=
\sum_{k\ge 1}N_m(k)\log k
=
\log|G_m|.
\]
证毕。
\end{proof}
