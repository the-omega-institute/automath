\begin{theorem}[Jones 交换子的纤维秩二谱刚性]\label{thm:op-algebra-fold-jones-commutator-rank2-spectrum}
沿用命题 \ref{prop:op-algebra-fold-index-weighted-commutator-energy} 的记号。
对任意 \(f\in\mathcal A_m\)，记
\[
C_f:=[M_f,e_m].
\]
对每个 \(x\in X_m\)，记
\[
F_x:=\Fold_m^{-1}(x),
\qquad
d_x:=|F_x|,
\]
并定义纤维均值与纤维方差
\[
\mu_x(f):=\frac1{d_x}\sum_{a\in F_x}f(a),
\qquad
\sigma_x(f)^2:=\frac1{d_x}\sum_{a\in F_x}|f(a)-\mu_x(f)|^2.
\]
则成立：
\begin{enumerate}
  \item 在纤维正交分解
  \[
  L^2(\mathcal A_m,\tau_m)\cong \bigoplus_{x\in X_m}\ell^2(F_x)
  \]
  下，\(C_f\) 按纤维分块：
  \[
  C_f=\bigoplus_{x\in X_m} C_{f,x},
  \qquad
  C_{f,x}:=C_f|_{\ell^2(F_x)},
  \]
  且每个 \(C_{f,x}\) 的秩至多为 \(2\)。
  \item 若 \(\sigma_x(f)=0\)，则 \(C_{f,x}=0\)；若 \(\sigma_x(f)>0\)，则 \(C_{f,x}\) 恰有两个非零奇异值，且二者相等，均为
  \[
  \sigma_x(f).
  \]
  \item 若 \(\|\cdot\|_{\mathrm{HS},\Omega_m}\) 表示 \(\ell^2(\Omega_m)\) 上未经 \(2^{-m}\) 归一化的 Hilbert--Schmidt 范数，即
  \[
  \|T\|_{\mathrm{HS},\Omega_m}^2:=\Tr(T^\ast T),
  \]
  则全局范数满足精确公式
  \[
  \|C_f\|_{\mathrm{op}}=\max_{x\in X_m}\sigma_x(f),
  \qquad
  \|C_f\|_{\mathrm{HS},\Omega_m}^{2}=2\sum_{x\in X_m}\sigma_x(f)^2.
  \]
  \item 进一步，索引加权交换子
  \[
  \widetilde C_f:=\mathrm{Ind}_{\mathrm W}(E_m)^{1/2}C_f
  \]
  满足
  \[
  \|\widetilde C_f\|_{\mathrm{HS},\Omega_m}^{2}
  =
  2\sum_{x\in X_m}d_x\,\sigma_x(f)^2
  =
  2\sum_{x\in X_m}\sum_{a\in F_x}|f(a)-\mu_x(f)|^2.
  \]
  特别地，命题 \ref{prop:op-algebra-fold-index-weighted-commutator-energy} 的归一化恒等式正是上式除以 \(2^m\) 后的等价写法。
\end{enumerate}
\end{theorem}
\begin{proof}
由 \(M_f\) 与 \(e_m\) 都保持纤维分解
\(
L^2(\mathcal A_m,\tau_m)\cong \bigoplus_{x\in X_m}\ell^2(F_x)
\)，
交换子 \(C_f=[M_f,e_m]\) 亦必按纤维分块。
固定 \(x\in X_m\)，在 \(\ell^2(F_x)\cong \CC^{d_x}\) 上记
\[
u_x:=d_x^{-1/2}\mathbf 1_x,
\qquad
e_x:=u_xu_x^\ast,
\qquad
D_x:=\mathrm{diag}\bigl(f(a)\bigr)_{a\in F_x}.
\]
则 \(C_{f,x}=D_xe_x-e_xD_x\)。
再定义
\[
b_x:=(I-e_x)D_xu_x,
\qquad
c_x:=(I-e_x)D_x^\ast u_x,
\]
于是 \(b_x,c_x\in u_x^\perp\)，并且
\[
\|b_x\|^2
=
\|c_x\|^2
=
\frac1{d_x}\sum_{a\in F_x}|f(a)-\mu_x(f)|^2
=
\sigma_x(f)^2.
\]
对任意 \(w\in u_x^\perp\)，有
\[
C_{f,x}u_x=b_x,
\qquad
C_{f,x}w=-u_x\,\langle c_x,w\rangle,
\]
故相对于分解 \(\CC u_x\oplus u_x^\perp\)，\(C_{f,x}\) 的矩阵为
\[
C_{f,x}=
\begin{pmatrix}
0 & -c_x^\ast\\
b_x & 0
\end{pmatrix}.
\]
因此 \(\mathrm{rank}(C_{f,x})\le 2\)。
若 \(\sigma_x(f)=0\)，则 \(b_x=c_x=0\)，从而 \(C_{f,x}=0\)。

若 \(\sigma_x(f)>0\)，则
\[
C_{f,x}^\ast C_{f,x}
=
\begin{pmatrix}
\|b_x\|^2 & 0\\
0 & c_xc_x^\ast
\end{pmatrix}.
\]
其中 \(c_xc_x^\ast\) 是 \(u_x^\perp\) 上的秩一正算子，其唯一非零特征值为 \(\|c_x\|^2=\sigma_x(f)^2\)。
故 \(C_{f,x}\) 恰有两个非零奇异值，且均等于 \(\sigma_x(f)\)。

全局算子范数由块对角分解直接给出：
\[
\|C_f\|_{\mathrm{op}}=\max_x \|C_{f,x}\|_{\mathrm{op}}=\max_x \sigma_x(f).
\]
同理，Hilbert--Schmidt 范数按块平方可加，故
\[
\|C_f\|_{\mathrm{HS},\Omega_m}^2
=
\sum_{x\in X_m}\|C_{f,x}\|_{\mathrm{HS},\Omega_m}^2
=
\sum_{x\in X_m}2\,\sigma_x(f)^2.
\]

最后，由定理 \ref{thm:op-algebra-fold-watatani-index-equals-multiplicity-field}，
\(\mathrm{Ind}_{\mathrm W}(E_m)\) 在 \(x\)-纤维块上恰为标量 \(d_x\)。
因此 \(\widetilde C_f\) 在该块上的两个非零奇异值均为 \(\sqrt{d_x}\,\sigma_x(f)\)，从而
\[
\|\widetilde C_f\|_{\mathrm{HS},\Omega_m}^2
=
\sum_{x\in X_m}2d_x\,\sigma_x(f)^2.
\]
再由
\[
d_x\,\sigma_x(f)^2
=
\sum_{a\in F_x}|f(a)-\mu_x(f)|^2
\]
即可得到第二个等式。
又因 \(\partial_m(f)=\mathrm{Ind}_{\mathrm W}(E_m)^{1/2}[e_m,M_f]=-\widetilde C_f\)，且命题 \ref{prop:op-algebra-fold-index-weighted-commutator-energy} 使用的 Hilbert--Schmidt 范数满足
\[
\|T\|_{\mathsf{HS}}^2=2^{-m}\|T\|_{\mathrm{HS},\Omega_m}^2,
\]
故该命题正是本结论最后一式除以 \(2^m\) 后的归一化表述。证毕。
\end{proof}
