\begin{theorem}[折叠兼容置换群的正合列、非规范裂开与花环分解]\label{thm:op-algebra-fold-fiber-normalizer-wreath}
\leanverified{paper\_op\_algebra\_fold\_fiber\_normalizer\_wreath}
沿用 \S\ref{subsec:op-algebra-fold-gauge-commutant} 的记号。
定义折叠兼容置换群（纤维规范化子）
\[
N_m:=\left\{\tau\in\mathrm{Sym}(\Omega_m):\ \exists\, f\in\mathrm{Sym}(X_m)\ \ \text{s.t.}\ \ \Fold_m\circ\tau=f\circ\Fold_m\right\}.
\]
并定义保持纤维谱的可见置换群
\[
\mathrm{Sym}_{d_m}(X_m):=\{f\in\mathrm{Sym}(X_m):\ d_m\circ f=d_m\}.
\]
令 \(\pi:N_m\to \mathrm{Sym}(X_m)\) 为诱导映射 \(\pi(\tau):=f\)（其中 \(f\) 满足 \(\Fold_m\circ\tau=f\circ\Fold_m\)）。
则：
\begin{enumerate}
  \item \(\pi\) 良定义且像为 \(\mathrm{Sym}_{d_m}(X_m)\)，并且存在正合列
        \[
        1\longrightarrow G_m\longrightarrow N_m\xrightarrow{\ \pi\ }\mathrm{Sym}_{d_m}(X_m)\longrightarrow 1,
        \qquad
        G_m=\Aut(\Fold_m).
        \]
  \item 该正合列总是非规范地裂开：存在群同态截面 \(s:\mathrm{Sym}_{d_m}(X_m)\to N_m\) 满足 \(\pi\circ s=\mathrm{id}\)。
        因而
        \[
        N_m\ \cong\ G_m\rtimes \mathrm{Sym}_{d_m}(X_m).
        \]
  \item 记分层
        \(
        X_m^{(d)}:=\{x\in X_m:\ d_m(x)=d\}
        \)
        与
        \(N_{m,d}:=|X_m^{(d)}|\)。
        则有同构
        \[
        \mathrm{Sym}_{d_m}(X_m)\ \cong\ \prod_{d\ge 1}\mathfrak S_{N_{m,d}},
        \qquad
        G_m\ \cong\ \prod_{d\ge 1}(\mathfrak S_d)^{N_{m,d}},
        \]
        并因此得到（非规范的）花环分解
        \[
        N_m\ \cong\ \prod_{d\ge 1}\bigl(\mathfrak S_d\wr \mathfrak S_{N_{m,d}}\bigr).
        \]
        特别地，\(N_m\) 的群同构类仅依赖于多重集合 \(\{d_m(x):x\in X_m\}\)（纤维谱）。
\end{enumerate}
\end{theorem}
\begin{proof}
对 \(\tau\in N_m\)，若 \(\Fold_m\circ\tau=f\circ\Fold_m=g\circ\Fold_m\)，则由 \(\Fold_m\) 满射立得 \(f=g\)，故 \(\pi\) 良定义。
若 \(\Fold_m\circ\tau=f\circ\Fold_m\)，则 \(\tau\) 给出双射 \(\Fold_m^{-1}(x)\to\Fold_m^{-1}(f(x))\)，从而 \(d_m(f(x))=d_m(x)\)；故 \(\mathrm{Im}(\pi)\subseteq \mathrm{Sym}_{d_m}(X_m)\)。
反之，若 \(f\in \mathrm{Sym}_{d_m}(X_m)\)，则由命题 \ref{prop:op-algebra-fold-visible-permutation-lifts} 存在其 Fold-lift \(\tau\)，从而 \(f\in\mathrm{Im}(\pi)\)。
并且 \(\ker(\pi)\) 恰为满足 \(\Fold_m\circ\tau=\Fold_m\) 的置换，即 \(G_m\)。
于是得到正合列。
\par
裂开：对每个 \(x\in X_m\) 任取一组纤维标号同构
\(
\ell_x:[d_m(x)]\xrightarrow{\sim}\Fold_m^{-1}(x)
\)。
对 \(f\in\mathrm{Sym}_{d_m}(X_m)\) 定义 \(\tau_f\in \mathrm{Sym}(\Omega_m)\) 为逐纤维的“纯搬运提升”
\[
\tau_f\bigl(\ell_x(i)\bigr):=\ell_{f(x)}(i)\qquad(1\le i\le d_m(x)).
\]
则 \(\Fold_m\circ\tau_f=f\circ\Fold_m\)，故 \(\tau_f\in N_m\) 且 \(\pi(\tau_f)=f\)。
并且 \(f\mapsto \tau_f\) 为群同态（由逐块拼接直接核对），从而给出截面 \(s=f\mapsto \tau_f\)，得到半直积同构。
\par
最后，\(f\in \mathrm{Sym}_{d_m}(X_m)\) 必保持分层集合 \(X_m^{(d)}\)，故
\(
\mathrm{Sym}_{d_m}(X_m)\cong\prod_{d\ge 1}\mathfrak S(X_m^{(d)})\cong\prod_{d\ge 1}\mathfrak S_{N_{m,d}}
\)。
而 \(G_m\cong\prod_{x\in X_m}\mathfrak S_{d_m(x)}\cong\prod_{d\ge 1}(\mathfrak S_d)^{N_{m,d}}\) 由纤维分解 \(\Omega_m=\bigsqcup_x\Fold_m^{-1}(x)\) 直接给出（亦见 \S\ref{subsec:op-algebra-fold-gauge-commutant}）。
半直积结构在每个 \(d\) 上由 \(\mathfrak S_{N_{m,d}}\) 置换其 \(N_{m,d}\) 个 \(\mathfrak S_d\)-分量给出，故得到 \(\prod_d(\mathfrak S_d\wr \mathfrak S_{N_{m,d}})\)。
证毕。
\end{proof}

\begin{corollary}[纤维谱直方图决定 \(G_m\)、\(N_m\) 的群阶]\label{cor:op-algebra-fold-group-orders}
\leanverified{paper\_op\_algebra\_fold\_group\_orders}
沿定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 的记号，并对每个 \(d\ge 1\) 记
\(
N_{m,d}:=\card{X_m^{(d)}}=\#\{x\in X_m:\ d_m(x)=d\}
\)。
则成立闭式
\[
\boxed{\;
|G_m|=\prod_{x\in X_m} d_m(x)!\ =\ \prod_{d\ge 1}(d!)^{N_{m,d}}\;}
\]
以及
\[
\boxed{\;
|N_m|
\ =\ 
|G_m|\cdot\bigl|\mathrm{Sym}_{d_m}(X_m)\bigr|
\ =\ 
\prod_{d\ge 1}(d!)^{N_{m,d}}\cdot \prod_{d\ge 1}(N_{m,d})!\; }.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 的分解
\(
G_m\cong\prod_{x\in X_m}\mathfrak S_{d_m(x)}
\)
与
\(
\mathrm{Sym}_{d_m}(X_m)\cong \prod_{d\ge 1}\mathfrak S_{N_{m,d}}
\)，
直接取群阶即得两式。
亦可由花环分解 \(N_m\cong\prod_{d\ge 1}(\mathfrak S_d\wr \mathfrak S_{N_{m,d}})\) 与
\(|\mathfrak S_d\wr \mathfrak S_n|=(d!)^{n}\cdot n!\) 推出。证毕。
\end{proof}

\begin{proposition}[折叠兼容提升的全局符号分解]\label{prop:op-algebra-fold-lift-global-sign-factorization}
\leanverified{paper\_op\_algebra\_fold\_lift\_global\_sign\_factorization}
沿定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 的记号。
令 \(X_m^{\mathrm{odd}}:=\{x\in X_m:\ d_m(x)\equiv 1\ (\mathrm{mod}\ 2)\}\) 为奇纤维层。
对任意 \(\tau\in N_m\) 记 \(f:=\pi(\tau)\in \mathrm{Sym}_{d_m}(X_m)\)。
任取一组纤维标号同构 \(\ell_x:[d_m(x)]\xrightarrow{\sim}\Fold_m^{-1}(x)\)，并定义
\[
p_x(\tau):=\ell_{f(x)}^{-1}\circ \tau|_{\Fold_m^{-1}(x)}\circ \ell_x\ \in\ \mathfrak S_{d_m(x)}.
\]
则量
\[
\eta(\tau):=\prod_{x\in X_m}\operatorname{sgn}\bigl(p_x(\tau)\bigr)\in\{\pm1\}
\]
与标号选择无关，并且成立严格恒等式
\[
\boxed{\;
\operatorname{sgn}(\tau)
\ =\
\operatorname{sgn}\bigl(f|_{X_m^{\mathrm{odd}}}\bigr)\cdot \eta(\tau).\;}
\]
\end{proposition}
\begin{proof}
先证 \(\eta(\tau)\) 的标号不变性。
若将 \(\ell_x\) 改为 \(\ell_x\circ u_x\)（其中 \(u_x\in\mathfrak S_{d_m(x)}\)），则
\(
p_x(\tau)\mapsto u_{f(x)}^{-1}p_x(\tau)u_x
\)，从而
\(
\operatorname{sgn}(p_x(\tau))\mapsto \operatorname{sgn}(u_{f(x)})\operatorname{sgn}(p_x(\tau))\operatorname{sgn}(u_x)
\)。
对 \(x\) 取全乘积并用 \(f\) 为双射，得到额外因子为
\(
\prod_x \operatorname{sgn}(u_{f(x)})\cdot \prod_x \operatorname{sgn}(u_x)=(\prod_x \operatorname{sgn}(u_x))^2=1
\)，故 \(\eta(\tau)\) 与标号无关。
\par
再证符号分解。
在给定标号下，把 \(\Omega_m\) 的线性序取为“按纤维拼接”：先固定 \(X_m\) 的一个线性序，再在每个纤维内按 \(\ell_x\) 的顺序排列。
令 \(\tau_0:=s(f)\) 为定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 中由同一标号诱导的纯搬运截面元，则 \(\pi(\tau_0)=f\) 且对每个 \(x\) 有 \(p_x(\tau_0)=\mathrm{id}\)。
因此 \(\tau=g\circ\tau_0\) 对某个唯一 \(g\in G_m\) 成立，并有
\[
\operatorname{sgn}(\tau)=\operatorname{sgn}(g)\operatorname{sgn}(\tau_0).
\]
由 \(g\) 的逐纤维分解，\(\operatorname{sgn}(g)=\prod_x \operatorname{sgn}(p_x(\tau))=\eta(\tau)\)。
于是只需计算 \(\operatorname{sgn}(\tau_0)\)。
\par
\(\tau_0\) 仅搬运纤维块：交换两段长度分别为 \(d_m(x)\)、\(d_m(y)\) 的块产生 \(d_m(x)d_m(y)\) 次相邻换位，
其奇偶性等于 \((d_m(x)\bmod 2)(d_m(y)\bmod 2)\)。
故所有符号贡献仅来自奇块之间的逆序对，等价于 \(f\) 在奇纤维层 \(X_m^{\mathrm{odd}}\) 上的置换符号：
\(
\operatorname{sgn}(\tau_0)=\operatorname{sgn}(f|_{X_m^{\mathrm{odd}}})
\)。
合并即得所示恒等式。证毕。
\end{proof}

\begin{theorem}[花环因子的交换化与 \texorpdfstring{$\{\pm1\}$}{+-1}-角色的完备分类]\label{thm:op-algebra-fold-wreath-abelianization-characters}
\leanverified{paper\_op\_algebra\_fold\_wreath\_abelianization\_characters}
沿定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 的记号，并令
\(
N_m^{\mathrm{ab}}:=N_m/[N_m,N_m]
\)。
则 \(N_m^{\mathrm{ab}}\) 为有限 \(2\)-群，且由花环分解诱导出同构
\[
N_m^{\mathrm{ab}}
\ \cong\
\prod_{d\ge 1}\bigl(\mathfrak S_d\wr \mathfrak S_{N_{m,d}}\bigr)^{\mathrm{ab}}.
\]
对每个 \(d\ge 1\) 与 \(n\ge 0\) 记 \(W_{d,n}:=\mathfrak S_d\wr \mathfrak S_n\)，则其交换化满足
\begin{equation}\label{eq:op-algebra-wreath-abelianization-cases}
W_{d,n}^{\mathrm{ab}}\cong
\begin{cases}
\{e\}, & n=0,\\
\{e\}, & d=1,\ n=1,\\
\ZZ_2, & d=1,\ n\ge 2,\\
\ZZ_2, & d\ge 2,\ n=1,\\
(\ZZ_2)^2, & d\ge 2,\ n\ge 2.
\end{cases}
\end{equation}
等价地，\(\mathrm{Hom}(N_m,\{\pm1\})\) 是一个有限的 \(\ZZ_2\)-向量空间，并由下述两类显式角色（满足相应存在性条件者）给出一组基：
\begin{enumerate}
  \item \textbf{可见层符号角色：}
  对每个满足 \(N_{m,d}\ge 2\) 的 \(d\ge 1\)，定义
  \[
  \chi^{\mathrm{vis}}_{d}(\tau):=\operatorname{sgn}\bigl(\pi(\tau)|_{X_m^{(d)}}\bigr),
  \qquad \tau\in N_m.
  \]
  \item \textbf{纤维内符号角色：}
  对每个满足 \(d\ge 2\) 且 \(N_{m,d}\ge 1\) 的 \(d\)，定义
  \[
  \chi^{\mathrm{fib}}_{d}(\tau):=\prod_{x\in X_m^{(d)}}\operatorname{sgn}\bigl(p_x(\tau)\bigr),
  \qquad \tau\in N_m,
  \]
  其中 \(p_x(\tau)\) 如命题 \ref{prop:op-algebra-fold-lift-global-sign-factorization} 定义。
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 的花环分解，\(N_m\) 为有限直积，故其交换化为对应交换化的直积。
因此只需计算 \(W_{d,n}^{\mathrm{ab}}\)。
\par
\textbf{当 \(d=1\) 时}，\(W_{1,n}\cong \mathfrak S_n\)，且 \(\mathfrak S_n^{\mathrm{ab}}\cong\{e\}\)（\(n\le 1\)）与 \(\mathfrak S_n^{\mathrm{ab}}\cong \ZZ_2\)（\(n\ge 2\)）由符号同态给出，得到 \eqref{eq:op-algebra-wreath-abelianization-cases} 的前三行。
\par
\textbf{当 \(d\ge 2\) 时}，写 \(W_{d,n}=B\rtimes \mathfrak S_n\)，其中 \(B:=(\mathfrak S_d)^n\)。
由于 \(\mathfrak S_d^{\mathrm{ab}}\cong \ZZ_2\)（由 \(\operatorname{sgn}\) 给出），故 \(B^{\mathrm{ab}}\cong (\ZZ_2)^n\)。
而 \(\mathfrak S_n\) 通过置换 \(n\) 个坐标共轭作用于 \(B\)，于是 \(B^{\mathrm{ab}}\) 的 coinvariants
\(
(B^{\mathrm{ab}})_{\mathfrak S_n}
\)
同构于 \(\ZZ_2\)（由“坐标和”生成）。
再由半直积交换化的一般结构
\(
(B\rtimes \mathfrak S_n)^{\mathrm{ab}}\cong (B^{\mathrm{ab}})_{\mathfrak S_n}\oplus \mathfrak S_n^{\mathrm{ab}}
\)，
并注意 \(\mathfrak S_n^{\mathrm{ab}}\cong\{e\}\) 当 \(n\le 1\)、\(\mathfrak S_n^{\mathrm{ab}}\cong\ZZ_2\) 当 \(n\ge 2\)，即得 \(d\ge 2\) 时的两行，从而得到 \eqref{eq:op-algebra-wreath-abelianization-cases}。
\par
最后，对角色的完备性与独立性：
由 \(N_m\cong\prod_{d\ge 1}(\mathfrak S_d\wr \mathfrak S_{N_{m,d}})\)，有
\[
\mathrm{Hom}(N_m,\{\pm1\})
\ \cong\
\prod_{d\ge 1}\mathrm{Hom}\!\bigl(W_{d,N_{m,d}},\{\pm1\}\bigr)
\ \cong\
\prod_{d\ge 1}\mathrm{Hom}\!\bigl(W_{d,N_{m,d}}^{\mathrm{ab}},\ZZ_2\bigr).
\]
当 \(N_{m,d}\ge 2\) 时，\(\chi_d^{\mathrm{vis}}\) 正是投影到 \(\mathfrak S_{N_{m,d}}\) 的符号同态；
当 \(d\ge 2\) 时，\(\chi_d^{\mathrm{fib}}\) 正是投影到基群 \((\mathfrak S_d)^{N_{m,d}}\) 后取坐标符号并在 coinvariants 上取生成元。
由 \eqref{eq:op-algebra-wreath-abelianization-cases}，每个 \(d\) 上的角色群为 \(\{e\}\)、\(\ZZ_2\) 或 \((\ZZ_2)^2\)，且在 \((\ZZ_2)^2\) 情形下 \(\chi_d^{\mathrm{vis}}\)、\(\chi_d^{\mathrm{fib}}\) 分别对应两条独立坐标。
因此由所有非平凡 \(\chi_d^{\mathrm{vis}}\)、\(\chi_d^{\mathrm{fib}}\) 组成的集合在 \(\mathrm{Hom}(N_m,\{\pm1\})\) 中给出一组显式基，完备性与独立性同时成立。证毕。
\end{proof}

\begin{corollary}[全局符号在角色基下的唯一分解]\label{cor:op-algebra-fold-global-sign-unique-decomposition}
沿命题 \ref{prop:op-algebra-fold-lift-global-sign-factorization} 与定理 \ref{thm:op-algebra-fold-wreath-abelianization-characters} 的记号。
则对任意 \(\tau\in N_m\) 有
\[
\operatorname{sgn}(\tau)
=
\Bigl(\prod_{\substack{d\ge 1\\ d\ \mathrm{odd}}}\chi^{\mathrm{vis}}_{d}(\tau)\Bigr)\cdot
\Bigl(\prod_{d\ge 2}\chi^{\mathrm{fib}}_{d}(\tau)\Bigr),
\]
并且该表达在定理 \ref{thm:op-algebra-fold-wreath-abelianization-characters} 给出的 \(\{\pm1\}\)-角色基下是唯一的。
\end{corollary}
\begin{proof}
由命题 \ref{prop:op-algebra-fold-lift-global-sign-factorization}，
\(
\operatorname{sgn}(\tau)=\operatorname{sgn}(f|_{X_m^{\mathrm{odd}}})\cdot \eta(\tau)
\)。
又因 \(f\in\mathrm{Sym}_{d_m}(X_m)\) 保持各层 \(X_m^{(d)}\) 并在层间无混合，故
\(
\operatorname{sgn}(f|_{X_m^{\mathrm{odd}}})=\prod_{d\ \mathrm{odd}}\operatorname{sgn}(f|_{X_m^{(d)}})
=\prod_{d\ \mathrm{odd}}\chi_d^{\mathrm{vis}}(\tau)
\)。
同时由 \(\eta(\tau)=\prod_{x\in X_m}\operatorname{sgn}(p_x(\tau))\) 与 \(p_x(\tau)\in \mathfrak S_{d_m(x)}\)，得
\(
\eta(\tau)=\prod_{d\ge 2}\prod_{x\in X_m^{(d)}}\operatorname{sgn}(p_x(\tau))
=\prod_{d\ge 2}\chi_d^{\mathrm{fib}}(\tau)
\)。
合并即得所示分解。
唯一性由定理 \ref{thm:op-algebra-fold-wreath-abelianization-characters}：\(\mathrm{Hom}(N_m,\{\pm1\})\) 作为 \(\ZZ_2\)-向量空间由所列角色给出一组基，故任意角色（尤其是全局符号）在该基下的 \(\{0,1\}\) 指数展开唯一。证毕。
\end{proof}

\begin{corollary}[交错实现与偶/奇提升的等分计数]\label{cor:op-algebra-fold-alternating-lift-counting}
沿定理 \ref{thm:op-algebra-fold-fiber-normalizer-wreath} 的记号。
\begin{enumerate}
  \item 若存在 \(x\in X_m\) 使 \(d_m(x)\ge 2\)，则 \(\operatorname{sgn}(G_m)=\{\pm1\}\)；
        并且对任意 \(f\in \mathrm{Sym}_{d_m}(X_m)\)，其全部 lifts 集合
        \(
        \{\tau\in N_m:\ \pi(\tau)=f\}
        \)
        在全局符号上严格等分：
        \[
        \#\{\tau:\ \pi(\tau)=f,\ \operatorname{sgn}(\tau)=+1\}
        =
        \#\{\tau:\ \pi(\tau)=f,\ \operatorname{sgn}(\tau)=-1\}
        =\frac{|G_m|}{2},
        \qquad
        |G_m|=\prod_{x\in X_m}d_m(x)!.
        \]
  \item 若对所有 \(x\in X_m\) 均有 \(d_m(x)=1\)（即 \(\Fold_m\) 为双射），则 \(G_m=\{e\}\) 且对每个 \(f\in\mathrm{Sym}(X_m)\) lift 唯一；
        该 lift 即为 \(f\) 本身在 \(\Omega_m=X_m\) 上的作用，并满足 \(\operatorname{sgn}(\tau)=\operatorname{sgn}(f)\)。
\end{enumerate}
\leanverified{paper\_op\_algebra\_fold\_alternating\_lift\_counting}
\end{corollary}
\begin{proof}
(1) 若某条纤维满足 \(d_m(x)\ge 2\)，则对应分量 \(\mathfrak S_{d_m(x)}\le G_m\) 含换位，故 \(\operatorname{sgn}(G_m)=\{\pm1\}\)。
对固定 \(f\in \mathrm{Sym}_{d_m}(X_m)\)，由命题 \ref{prop:op-algebra-fold-visible-permutation-lifts}，其 lifts 构成 \(G_m\) 的一个左陪集 \(G_m\tau_0\)。
取 \(g\in G_m\) 使 \(\operatorname{sgn}(g)=-1\)，则左乘 \(g\) 给出陪集上的无不动点对合
\(
\tau\mapsto g\tau
\)，并且 \(\operatorname{sgn}(g\tau)=-\operatorname{sgn}(\tau)\)。
因此偶、奇 lifts 数目相等，且总数为 \(|G_m|\)，得到等分计数。
\par
(2) 若所有纤维为单点，则 \(\Fold_m\) 为双射并可识别 \(\Omega_m=X_m\)，此时 \(G_m\) 平凡，且 \(\Fold_m\circ\tau=f\circ\Fold_m\) 强制 \(\tau=f\)，从而结论成立。证毕。
\end{proof}
