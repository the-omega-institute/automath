\begin{theorem}[Choi 态的 Petz--/sandwiched Rényi 互信息闭式]\label{thm:op-algebra-fold-quantum-channel-choi-renyi-mutual-information}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_choi\_renyi\_mutual\_information}
令 \(n:=|\Omega_m|=2^m\)，并记 \(\rho_{RB}:=J(\mathcal{E}_m)/n\)。
对任意 \(\alpha>0\)、\(\alpha\neq 1\)，定义
\[
I_\alpha^{\mathrm P}(R\!:\!B)_\rho
:=
D_\alpha\!\bigl(\rho_{RB}\,\|\,\rho_R\otimes\rho_B\bigr),
\qquad
\widetilde I_\alpha(R\!:\!B)_\rho
:=
\widetilde D_\alpha\!\bigl(\rho_{RB}\,\|\,\rho_R\otimes\rho_B\bigr),
\]
其中 \(D_\alpha\) 与 \(\widetilde D_\alpha\) 分别为 Petz Rényi 相对熵与 sandwiched Rényi 相对熵。
则二者完全一致，并满足闭式
\[
I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_\alpha(R\!:\!B)_\rho
=
\frac{1}{\alpha-1}\log\!\Bigl(
n^{\alpha-2}\sum_{x\in X_m} d_m(x)^{2-\alpha}
\Bigr).
\]
特别地，
\[
I_2^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_2(R\!:\!B)_\rho
=
\log|X_m|,
\]
\[
\lim_{\alpha\to\infty}I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\lim_{\alpha\to\infty}\widetilde I_\alpha(R\!:\!B)_\rho
=
\log\frac{n}{d_m^{\min}},
\qquad
d_m^{\min}:=\min_{x\in X_m}d_m(x),
\]
并且
\[
\lim_{\alpha\to1}I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\lim_{\alpha\to1}\widetilde I_\alpha(R\!:\!B)_\rho
=
H(\pi_m).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-quantum-channel-choi-spectrum-mi}，\(\rho_R=\rho_B=I/n\)，故
\[
\rho_R\otimes\rho_B=\frac{I}{n^2}.
\]
因此 \(\rho_{RB}\) 与 \(\rho_R\otimes\rho_B\) 对易，Petz 与 sandwiched Rényi 相对熵在此即为同一表达式：
\[
I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_\alpha(R\!:\!B)_\rho
=
\frac{1}{\alpha-1}
\log\Tr\!\Bigl[
\rho_{RB}^{\alpha}\bigl(\rho_R\otimes\rho_B\bigr)^{1-\alpha}
\Bigr].
\]
又
\[
\bigl(\rho_R\otimes\rho_B\bigr)^{1-\alpha}
=
\Bigl(\frac{I}{n^2}\Bigr)^{1-\alpha}
=
n^{2(\alpha-1)}I,
\]
从而
\[
I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_\alpha(R\!:\!B)_\rho
=
\frac{1}{\alpha-1}
\log\!\Bigl(n^{2(\alpha-1)}\Tr(\rho_{RB}^\alpha)\Bigr).
\]
再由推论 \ref{cor:op-algebra-fold-quantum-channel-choi-renyi-moments}，
\[
\Tr(\rho_{RB}^\alpha)=n^{-\alpha}\sum_{x\in X_m} d_m(x)^{2-\alpha},
\]
代入即得所示闭式。

当 \(\alpha=2\) 时，括号内恰为
\(
\sum_{x\in X_m}d_m(x)^0=|X_m|
\)，故得到
\[
I_2^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_2(R\!:\!B)_\rho
=
\log|X_m|.
\]
对 \(\alpha\to\infty\) 的端点，只需把闭式改写为
\[
I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_\alpha(R\!:\!B)_\rho
=
\frac{\alpha-2}{\alpha-1}\log\frac{n}{d_m^{\min}}
+
\frac{1}{\alpha-1}
\log\!\Bigl(
\sum_{x\in X_m}\Bigl(\frac{d_m(x)}{d_m^{\min}}\Bigr)^{2-\alpha}
\Bigr).
\]
括号内和式始终夹在 \(1\) 与 \(|X_m|\) 之间，故第二项趋于 \(0\)，而第一项趋于 \(\log(n/d_m^{\min})\)。
于是得到共同的极限
\[
\lim_{\alpha\to\infty}I_\alpha^{\mathrm P}(R\!:\!B)_\rho
=
\lim_{\alpha\to\infty}\widetilde I_\alpha(R\!:\!B)_\rho
=
\log\frac{n}{d_m^{\min}}.
\]
最后，\(\alpha\to1\) 时两者都连续延拓到 Umegaki 相对熵，而定理 \ref{thm:op-algebra-fold-quantum-channel-choi-spectrum-mi} 已给出
\[
I(R\!:\!B)_\rho=H(\pi_m).
\]
证毕。
\end{proof}

\begin{corollary}[二阶 Choi Rényi 互信息恢复经典容量]\label{cor:op-algebra-fold-quantum-channel-i2-equals-classical-capacity}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_i2\_equals\_classical\_capacity}
沿用定理 \ref{thm:op-algebra-fold-quantum-channel-choi-renyi-mutual-information} 的记号，则
\[
I_2^{\mathrm P}(R\!:\!B)_\rho
=
\widetilde I_2(R\!:\!B)_\rho
=
C(\mathcal{E}_m)
=
\log|X_m|.
\]
\end{corollary}
\begin{proof}

\(
I_2^{\mathrm P}(R\!:\!B)_\rho

\)；
再由定理 \ref{thm:op-algebra-fold-quantum-channel-capacity}，

\)。
合并即得。证毕。
\end{proof}

\begin{theorem}[全部输出 Rényi 熵范围的完全闭式]\label{thm:op-algebra-fold-quantum-channel-output-renyi-entropy-range}
令 \(n:=|\Omega_m|=2^m\)，并记
\[
d_m^{\min}:=\min_{x\in X_m}d_m(x).
\]
对任意输入态 \(\sigma\in\mathcal{B}(\mathcal{H}_m)\)（\(\sigma\ge 0\)、\(\Tr\sigma=1\)），令
\[
p_x:=\Tr(\Pi_x\sigma),
\qquad
\sum_{x\in X_m}p_x=1.
\]
则
\[
\mathcal{E}_m(\sigma)=\sum_{x\in X_m}\frac{p_x}{d_m(x)}\,\Pi_x,
\]
并且对每个 \(x\in X_m\)，输出态在块 \(\Pi_x\mathcal{H}_m\) 上的唯一非零特征值为 \(p_x/d_m(x)\)，其重数为 \(d_m(x)\)。
从而对任意 \(\alpha>0\)、\(\alpha\neq 1\)，输出 Rényi 熵满足闭式
\[
H_\alpha\bigl(\mathcal{E}_m(\sigma)\bigr)
=
\frac{1}{1-\alpha}\log\!\Bigl(
\sum_{x\in X_m} p_x^\alpha d_m(x)^{1-\alpha}
\Bigr).
\]
此外有精确极值公式
\[
\max_{\sigma}H_\alpha\bigl(\mathcal{E}_m(\sigma)\bigr)=\log n,
\qquad
\min_{\sigma}H_\alpha\bigl(\mathcal{E}_m(\sigma)\bigr)=\log d_m^{\min}.
\]
其中最大值由 \(\sigma=I/n\) 达到，而最小值由任一支撑于最小纤维 \(\Pi_{x_{\min}}\mathcal{H}_m\)（其中 \(d_m(x_{\min})=d_m^{\min}\)）内的纯态达到。
\end{theorem}
\begin{proof}
由定义 \eqref{eq:op-algebra-fold-quantum-channel-def}，
\[
\mathcal{E}_m(\sigma)=\sum_{x\in X_m}\frac{\Tr(\Pi_x\sigma)}{d_m(x)}\,\Pi_x
=
\sum_{x\in X_m}\frac{p_x}{d_m(x)}\,\Pi_x.
\]
由于 \(\{\Pi_x\}_{x\in X_m}\) 两两正交，故输出态按纤维块正交分解；每个块 \(\Pi_x\mathcal{H}_m\) 上的特征值恒为 \(p_x/d_m(x)\)，其重数为 \(d_m(x)\)。
因此
\[
\Tr\bigl(\mathcal{E}_m(\sigma)^\alpha\bigr)
=
\sum_{x\in X_m}d_m(x)\Bigl(\frac{p_x}{d_m(x)}\Bigr)^\alpha
=
\sum_{x\in X_m}p_x^\alpha d_m(x)^{1-\alpha},
\]
从而得到 Rényi 熵闭式。

最大值方面，\(\mathcal{E}_m(\sigma)\) 是 \(n\)-维态，故总有
\(
H_\alpha(\mathcal{E}_m(\sigma))\le \log n
\)。
而当 \(\sigma=I/n\) 时，
\[
\mathcal{E}_m(I/n)
=
\sum_{x\in X_m}\frac{\Tr(\Pi_x)/n}{d_m(x)}\,\Pi_x
=
\frac{1}{n}\sum_{x\in X_m}\Pi_x
=
\frac{I}{n},
\]
故达到
\(
H_\alpha(\mathcal{E}_m(I/n))=\log n
\)。

最小值方面，先取任意 \(x_{\min}\in X_m\) 满足 \(d_m(x_{\min})=d_m^{\min}\)，并令 \(\sigma=\ket{\psi}\!\bra{\psi}\) 为任意支撑于 \(\Pi_{x_{\min}}\mathcal{H}_m\) 内的纯态。
则 \(p_{x_{\min}}=1\)、其余 \(p_x=0\)，于是
\[
\mathcal{E}_m(\sigma)=\frac{1}{d_m^{\min}}\Pi_{x_{\min}},
\]
故
\(
H_\alpha(\mathcal{E}_m(\sigma))=\log d_m^{\min}
\)。
反向地，对任意输入态与任意 \(\alpha>1\)，由 \(\sum_x p_x^\alpha\le 1\) 以及 \(d_m(x)^{1-\alpha}\le (d_m^{\min})^{1-\alpha}\) 得
\[
\sum_{x\in X_m}p_x^\alpha d_m(x)^{1-\alpha}\le (d_m^{\min})^{1-\alpha},
\]
故
\[
H_\alpha\bigl(\mathcal{E}_m(\sigma)\bigr)\ge \log d_m^{\min}.
\]
而当 \(0<\alpha<1\) 时，\(\sum_x p_x^\alpha\ge 1\) 且 \(d_m(x)^{1-\alpha}\ge (d_m^{\min})^{1-\alpha}\)，于是
\[
\sum_{x\in X_m}p_x^\alpha d_m(x)^{1-\alpha}\ge (d_m^{\min})^{1-\alpha},
\]
同样推出
\[
H_\alpha\bigl(\mathcal{E}_m(\sigma)\bigr)\ge \log d_m^{\min}.
\]
故最小值恰为 \(\log d_m^{\min}\)。证毕。
\end{proof}

\begin{theorem}[乘法域、固定点代数与最大可恢复代数完全一致]\label{thm:op-algebra-fold-quantum-channel-md-fix-recoverable}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_md\_fix\_recoverable}
定义
\[
\operatorname{MD}(\mathcal{E}_m)
:=
\Bigl\{
a\in\mathcal{B}(\mathcal{H}_m):
\mathcal{E}_m(a^\ast a)=\mathcal{E}_m(a)^\ast\mathcal{E}_m(a),\quad
\mathcal{E}_m(aa^\ast)=\mathcal{E}_m(a)\mathcal{E}_m(a)^\ast
\Bigr\},
\]
\[
\operatorname{Fix}(\mathcal{E}_m)
:=
\bigl\{
a\in\mathcal{B}(\mathcal{H}_m):\ \mathcal{E}_m(a)=a
\bigr\}.
\]
则有精确恒等式
\[
\operatorname{MD}(\mathcal{E}_m)
=
\operatorname{Fix}(\mathcal{E}_m)
=
\mathcal{Z}_m.
\]
特别地，命题 \ref{prop:op-algebra-fold-quantum-channel-max-recoverable-algebra} 中的最大可恢复代数与固定点代数、乘法域三者完全一致。
\end{theorem}
\begin{proof}
先证 \(\operatorname{Fix}(\mathcal{E}_m)=\mathcal{Z}_m\)。
由 \eqref{eq:op-algebra-fold-quantum-channel-def}，对任意 \(a\in\mathcal{B}(\mathcal{H}_m)\) 都有 \(\mathcal{E}_m(a)\in\mathcal{Z}_m\)。
另一方面，若 \(z=\sum_{x\in X_m}\alpha_x\Pi_x\in\mathcal{Z}_m\)，则
\[
\mathcal{E}_m(z)
=
\sum_{y\in X_m}\frac{\Tr(\Pi_y z)}{d_m(y)}\,\Pi_y
=
\sum_{y\in X_m}\alpha_y\Pi_y
=
z.
\]
故 \(\mathcal{Z}_m\subseteq \operatorname{Fix}(\mathcal{E}_m)\)，而固定点显然必须落在像空间 \(\mathcal{Z}_m\) 中，故
\(
\operatorname{Fix}(\mathcal{E}_m)=\mathcal{Z}_m
\)。

再证 \(\operatorname{MD}(\mathcal{E}_m)=\mathcal{Z}_m\)。
首先，\(\mathcal{E}_m\) 对 \(\mathcal{Z}_m\) 是双模映射：若
\(
z_1=\sum_x\alpha_x\Pi_x
\)、
\(
z_2=\sum_x\beta_x\Pi_x
\)
属于 \(\mathcal{Z}_m\)，则对任意 \(a\) 有
\[
\mathcal{E}_m(z_1 a z_2)
=
\sum_{x\in X_m}\frac{\Tr(\Pi_x z_1 a z_2)}{d_m(x)}\,\Pi_x
=
\sum_{x\in X_m}\alpha_x\beta_x\frac{\Tr(\Pi_x a)}{d_m(x)}\,\Pi_x
=
z_1\mathcal{E}_m(a)z_2.
\]
又因为 \(\mathcal{E}_m\) 迹保持，若 \(\mathcal{E}_m(t^\ast t)=0\)，则
\[
0=\Tr\bigl(\mathcal{E}_m(t^\ast t)\bigr)=\Tr(t^\ast t),
\]
从而 \(t=0\)。
因此 \(\mathcal{E}_m\) 是忠实的。

现取 \(a\in\operatorname{MD}(\mathcal{E}_m)\)，并令 \(z:=\mathcal{E}_m(a)\in\mathcal{Z}_m\)。
由双模性与乘法域定义，
\[
\mathcal{E}_m\bigl((a-z)^\ast(a-z)\bigr)
=
\mathcal{E}_m(a^\ast a)-\mathcal{E}_m(a)^\ast z-z^\ast \mathcal{E}_m(a)+z^\ast z
=
\mathcal{E}_m(a)^\ast\mathcal{E}_m(a)-z^\ast z-z^\ast z+z^\ast z
=
0.
\]
忠实性推出 \(a=z\in\mathcal{Z}_m\)。
故 \(\operatorname{MD}(\mathcal{E}_m)\subseteq \mathcal{Z}_m\)。
反向包含则直接来自 \(\mathcal{E}_m|_{\mathcal{Z}_m}=\mathrm{id}_{\mathcal{Z}_m}\)：对任意 \(z\in\mathcal{Z}_m\)，
\[
\mathcal{E}_m(z^\ast z)=z^\ast z=\mathcal{E}_m(z)^\ast\mathcal{E}_m(z),
\qquad
\mathcal{E}_m(zz^\ast)=zz^\ast=\mathcal{E}_m(z)\mathcal{E}_m(z)^\ast,
\]
故 \(z\in\operatorname{MD}(\mathcal{E}_m)\)。
于是 \(\operatorname{MD}(\mathcal{E}_m)=\mathcal{Z}_m\)。

最后，命题 \ref{prop:op-algebra-fold-quantum-channel-max-recoverable-algebra} 已给出最大可恢复代数恰为 \(\mathcal{Z}_m\)，从而三者完全一致。证毕。
\end{proof}
