\subsection{折叠量子信道与碰撞矩：最小环境维数、Choi 谱与容量}\label{subsec:op-algebra-fold-quantum-channel}
本段在附录 \ref{subsec:op-algebra-fold-audit-spectrum}--\ref{subsec:op-algebra-fold-watatani-index} 的“折叠条件期望/基本构造”口径之外，引入一个与之自然一致的量子信息版本：以 \(\Omega_m\) 的标准基为计算基，把“按折叠纤维做条件化并在纤维内去相干”提升为作用在 \(\mathcal{B}(\mathcal{H}_m)\) 上的完全正映射。
该量子信道的最小 Kraus 秩（等价于最小 Stinespring 环境维数）精确等于二阶碰撞矩 \(S_2(m)\)，其 Choi 谱与互信息可由纤维多重度场 \(\{d_m(x)\}\) 完全显式给出，并与正文关于“隐藏比特预算/可见比特”指标严格对齐。

\paragraph{量子化的折叠条件化算子。}
固定 \(m\ge 1\)。
令 \(\Omega_m=\{0,1\}^m\)，\(X_m\) 为稳定类型集，\(\Fold_m:\Omega_m\to X_m\) 为折叠映射，并记纤维多重度
\[
d_m(x):=\bigl|\Fold_m^{-1}(x)\bigr|,\qquad x\in X_m,
\]
以及二阶碰撞矩（定义 \ref{def:pom-s2}）
\[
S_2(m):=\sum_{x\in X_m} d_m(x)^2.
\]
令 \(\mathcal{H}_m:=\CC^{\Omega_m}\) 并取正交基 \(\{\lvert\omega\rangle\}_{\omega\in\Omega_m}\)。
对每个 \(x\in X_m\) 定义纤维投影
\[
\Pi_x:=\sum_{\omega:\ \Fold_m(\omega)=x}\lvert\omega\rangle\langle\omega\rvert\ \in\ \mathcal{B}(\mathcal{H}_m).
\]

\begin{proposition}[\texorpdfstring{$q$}{q}-重碰撞投影的秩刻画]\label{prop:op-algebra-fold-q-collision-projector-rank}
\leanverified{paper\_op\_algebra\_fold\_q\_collision\_projector\_rank}
对任意整数 \(q\ge 2\)，定义 \(q\)-碰撞矩
\[
S_q(m):=\sum_{x\in X_m} d_m(x)^q,
\]
以及 \(q\)-重碰撞投影算子
\[
\Pi_m^{(q)}:=\sum_{x\in X_m}\Pi_x^{\otimes q}\ \in\ \mathcal{B}(\mathcal{H}_m^{\otimes q}).
\]
则 \(\Pi_m^{(q)}\) 为正交投影，且其秩满足
\[
\rank\bigl(\Pi_m^{(q)}\bigr)=S_q(m).
\]
\end{proposition}
\begin{proof}
由于 \(\{\Pi_x\}_{x\in X_m}\) 两两正交且 \(\sum_x\Pi_x=I_{\mathcal{H}_m}\)，故 \(\{\Pi_x^{\otimes q}\}_{x\in X_m}\) 两两正交且均为投影，从而其和 \(\Pi_m^{(q)}\) 仍为投影。
又 \(\rank(\Pi_x)=d_m(x)\)，故
\[
\rank(\Pi_m^{(q)})
=\sum_{x\in X_m}\rank(\Pi_x^{\otimes q})
=\sum_{x\in X_m}\bigl(\rank(\Pi_x)\bigr)^q
=\sum_{x\in X_m} d_m(x)^q
=S_q(m).
\]
证毕。
\end{proof}

定义完全正且迹保持的线性映射
\begin{equation}\label{eq:op-algebra-fold-quantum-channel-def}
\mathcal{E}_m:\mathcal{B}(\mathcal{H}_m)\to\mathcal{B}(\mathcal{H}_m),\qquad
\mathcal{E}_m(\rho):=\sum_{x\in X_m}\frac{\Tr(\Pi_x\rho)}{d_m(x)}\,\Pi_x.
\end{equation}
该映射是“先以 \(\{\Pi_x\}\) 测量稳定类型，再在对应纤维子空间内制备最大混合态”的测量--制备信道，因此为幺正保持（unital）且纠缠破坏（entanglement-breaking）。

\begin{proposition}[投影在粗粒化下的 \(0/1\) 刚性]\label{prop:op-algebra-fold-quantum-channel-projection-01-rigidity}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_projection\_01\_rigidity}
设 \(\mathcal{E}_m\) 如 \eqref{eq:op-algebra-fold-quantum-channel-def}。
对任意正交投影 \(P\in\mathcal{B}(\mathcal{H}_m)\)（即 \(P=P^\ast=P^2\)），下列条件等价：
\begin{enumerate}
  \item \(\mathcal{E}_m(P)\) 仍为正交投影；
  \item 存在集合 \(S\subseteq X_m\) 使得
  \[
  P=\sum_{x\in S}\Pi_x;
  \]
  \item \(P\in\mathcal{Z}_m:=\Span\{\Pi_x:\ x\in X_m\}\cong \CC^{X_m}\)。
\end{enumerate}
\end{proposition}
\begin{proof}
写
\(
\alpha_x:=\Tr(\Pi_xP)/d_m(x)
\)。
则
\[
\mathcal{E}_m(P)=\sum_{x\in X_m}\alpha_x\,\Pi_x,
\qquad
0\le \alpha_x\le 1
\]
（因为 \(0\le \Pi_xP\Pi_x\le \Pi_x\)）。
若 \(\mathcal{E}_m(P)\) 为投影，则对每个 \(x\) 有 \(\alpha_x^2=\alpha_x\)，从而 \(\alpha_x\in\{0,1\}\)。
因此 \(\Tr(\Pi_xP)\in\{0,d_m(x)\}\)。

当 \(\Tr(\Pi_xP)=d_m(x)\) 时，\(\Pi_x-\Pi_xP\Pi_x\ge 0\) 且迹为零，故 \(\Pi_xP\Pi_x=\Pi_x\)，从而 \(P\Pi_x=\Pi_x\)；
当 \(\Tr(\Pi_xP)=0\) 时，\(\Pi_xP\Pi_x\ge 0\) 且迹为零，故 \(\Pi_xP\Pi_x=0\)，从而 \(P\Pi_x=0\)。
于是
\[
P=\sum_{x\in X_m}P\Pi_x=\sum_{\alpha_x=1}\Pi_x,
\]
得到 (2)，并立刻推出 (3)。
反向蕴含：若 \(P=\sum_{x\in S}\Pi_x\)，则 \(\mathcal{E}_m(P)=P\) 为投影，故 (1) 成立。证毕。
\end{proof}

\begin{theorem}[最小 Stinespring 环境维数的碰撞矩刻画]\label{thm:op-algebra-fold-quantum-channel-minimal-environment-equals-s2}
设 \(\mathcal{E}_m\) 如 \eqref{eq:op-algebra-fold-quantum-channel-def}。
则 \(\mathcal{E}_m\) 的最小 Kraus 秩（等价于最小 Choi 秩）满足
\[
\rank J(\mathcal{E}_m)=\min\{\text{Kraus 个数}\}=S_2(m).
\]
因此 \(\mathcal{E}_m\) 的最小 Stinespring 环境 Hilbert 空间维数亦恰为 \(S_2(m)\)。
并且存在达到该最小值的 Kraus 表示，可取
\[
K^{(x)}_{\omega,\omega'}:=\frac{\lvert\omega\rangle\langle\omega'\rvert}{\sqrt{d_m(x)}},
\qquad
\omega,\omega'\in\Fold_m^{-1}(x),\ x\in X_m,
\]
其总数为 \(\sum_x d_m(x)^2=S_2(m)\)。
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_minimal\_environment\_equals\_s2}
\end{theorem}
\begin{proof}
先验证所给 Kraus 族实现 \(\mathcal{E}_m\)：
对固定 \(x\) 与 \(\omega,\omega'\in\Fold_m^{-1}(x)\)，有
\[
K^{(x)}_{\omega,\omega'}\,\rho\,\bigl(K^{(x)}_{\omega,\omega'}\bigr)^\ast
=\frac{1}{d_m(x)}\ket{\omega}\!\bra{\omega}\cdot \bra{\omega'}\rho\ket{\omega'}.
\]
对 \(\omega'\) 求和得到 \(\frac{1}{d_m(x)}\Tr(\Pi_x\rho)\ket{\omega}\!\bra{\omega}\)，再对 \(\omega\) 求和得到 \(\frac{\Tr(\Pi_x\rho)}{d_m(x)}\Pi_x\)，最后对 \(x\) 求和即为 \(\mathcal{E}_m(\rho)\)。
且
\[
\sum_{x\in X_m}\ \sum_{\omega,\omega'\in\Fold_m^{-1}(x)}
\bigl(K^{(x)}_{\omega,\omega'}\bigr)^\ast K^{(x)}_{\omega,\omega'}
=\sum_{x\in X_m}\sum_{\omega'\in\Fold_m^{-1}(x)}\ket{\omega'}\!\bra{\omega'}
=\sum_{x\in X_m}\Pi_x=I,
\]
从而迹保持成立。

为证最小性，计算 Choi 矩阵 \(J(\mathcal{E}_m)=\sum_{i,j}\ket{i}\!\bra{j}\otimes \mathcal{E}_m(\ket{i}\!\bra{j})\)（以 \(\Omega_m\) 的标准基为参考基）。
若 \(i\neq j\) 则 \(\Tr(\Pi_x\ket{i}\!\bra{j})=0\)，故 \(\mathcal{E}_m(\ket{i}\!\bra{j})=0\)；
若 \(i=j\) 且 \(\Fold_m(i)=x\)，则 \(\Tr(\Pi_x\ket{i}\!\bra{i})=1\) 且
\(
\mathcal{E}_m(\ket{i}\!\bra{i})=\Pi_x/d_m(x)
\)。
于是
\[
J(\mathcal{E}_m)
=\sum_{x\in X_m}\ \sum_{i\in\Fold_m^{-1}(x)} \ket{i}\!\bra{i}\otimes \frac{\Pi_x}{d_m(x)}.
\]
该矩阵在参考系统上按 \(i\) 分块正交直和；对固定 \(x\) 与 \(i\in\Fold_m^{-1}(x)\)，张量因子 \(\Pi_x/d_m(x)\) 的秩为 \(d_m(x)\)。
因此
\[
\rank J(\mathcal{E}_m)
=\sum_{x\in X_m} \bigl|\Fold_m^{-1}(x)\bigr|\cdot d_m(x)
=\sum_{x\in X_m} d_m(x)^2
=S_2(m).
\]
而完全正映射的最小 Kraus 个数等于 Choi 秩，故最小 Stinespring 环境维数亦为 \(S_2(m)\)。
上一步 Kraus 族的个数恰为 \(S_2(m)\)，从而达到下界。证毕。
\end{proof}

\begin{corollary}[球面--信道对偶：最小环境维数等于球面配分函数]\label{cor:op-algebra-fold-quantum-channel-minenv-equals-sphere-partition-function}
设 \(\mathcal{E}_m\) 如 \eqref{eq:op-algebra-fold-quantum-channel-def}。
则其最小 Stinespring 环境维数满足

其中 \(Z_m\) 为定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 所确定的二维定向 TQFT。
\end{corollary}
\begin{proof}

而由推论 \ref{cor:conclusion-tqft-sphere-partition-function-s2}，\(Z_m(S^2)=S_2(m)\)。

\end{proof}

\begin{proposition}[折叠信道的显式混合幺正实现]\label{prop:op-algebra-fold-quantum-channel-mixed-unitary}
设 \(\mathcal{E}_m\) 如 \eqref{eq:op-algebra-fold-quantum-channel-def}。
则 \(\mathcal{E}_m\) 为混合幺正信道：存在有限幺正族 \(\{U_g\}_{g\in\mathcal{G}_m}\subset U(\mathcal{H}_m)\) 使
\[
\mathcal{E}_m(\rho)=\frac{1}{|\mathcal{G}_m|}\sum_{g\in\mathcal{G}_m}U_g\,\rho\,U_g^\ast.
\]
\end{proposition}
\begin{proof}
对每个 \(x\in X_m\) 取任意双射 \(\tau_x:F_x\to\ZZ/d_m(x)\ZZ\)，并以 \(\ket{u}_x\) 表示 \(\mathcal{H}_x\) 上对应的正交基（即 \(\ket{u}_x=\ket{\tau_x^{-1}(u)}\)）。
在 \(\mathcal{H}_x\) 上定义 Heisenberg--Weyl 幺正算子
\[
Z_x\ket{u}_x:=e^{2\pi\mathrm{i}u/d_m(x)}\ket{u}_x,
\qquad
X_x\ket{u}_x:=\ket{u+1}_x,
\qquad
W_{x,a,b}:=Z_x^{a}X_x^{b},
\]
其中 \(a,b\in\ZZ/d_m(x)\ZZ\)。
令
\(
\mathcal{G}_m:=\prod_{x\in X_m}(\ZZ/d_m(x)\ZZ)^2
\)
并定义块对角幺正
\[
U_{(a_x,b_x)_{x\in X_m}}:=\bigoplus_{x\in X_m}W_{x,a_x,b_x}\ \in U(\mathcal{H}_m).
\]
则对任意 \(\rho\in\mathcal{B}(\mathcal{H}_m)\)，共轭平均在每个块 \(\mathcal{B}(\mathcal{H}_x)\) 上实现完全去偏极化：
\[
\frac{1}{d_m(x)^2}\sum_{a,b}W_{x,a,b}\,\rho_{xx}\,W_{x,a,b}^\ast=\frac{\Tr(\rho_{xx})}{d_m(x)}I_{\mathcal{H}_x},
\]
而对 \(x\neq y\) 的交叉块，独立平均使其消失。于是整体平均给出
\(
\rho\mapsto \bigoplus_x \frac{\Tr(\Pi_x\rho)}{d_m(x)}\Pi_x
\)，即 \(\mathcal{E}_m\)。证毕。
\end{proof}

\begin{proposition}[块对角 Haar--twirling 表示]\label{prop:op-algebra-fold-quantum-channel-haar-twirl}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_haar\_twirl}
令 \(\mathcal{H}_m=\bigoplus_{x\in X_m}\mathcal{H}_x\) 为纤维直和分解，其中 \(\mathcal{H}_x:=\Pi_x\mathcal{H}_m\) 且 \(\dim\mathcal{H}_x=d_m(x)\)。
令
\[
\mathcal{U}_m:=\prod_{x\in X_m}U(\mathcal{H}_x)\ \cong\ \prod_{x\in X_m}U(d_m(x))
\]
按块对角方式嵌入 \(U(\mathcal{H}_m)\)，并记其 Haar 概率测度为 \(\dd U\)。
则对任意 \(\rho\in\mathcal{B}(\mathcal{H}_m)\) 有严格恒等式
\[
\mathcal{E}_m(\rho)=\int_{\mathcal{U}_m}U\rho U^\ast\,\dd U.
\]
\end{proposition}
\begin{proof}
将 \(\rho\) 分解为块矩阵 \(\rho=\sum_{x,y}\Pi_x\rho\Pi_y\)。
对 \(x\neq y\) 的交叉块 \(\Pi_x\rho\Pi_y:\mathcal{H}_y\to\mathcal{H}_x\)，积分在左右分别对 \(U(\mathcal{H}_x)\)、\(U(\mathcal{H}_y)\) 作独立共轭平均；由 Haar 不变性，唯一不变的线性映射为零映射，故该部分积分为零。
\par
对对角块 \(x=y\)，由 \(U(d)\) 的标准 twirling 公式，
\[
\int_{U(\mathcal{H}_x)}U_x(\Pi_x\rho\Pi_x)U_x^\ast\,\dd U_x
=\frac{\Tr(\Pi_x\rho)}{d_m(x)}\,\Pi_x.
\]
对 \(x\) 求和即得所示恒等式。证毕。
\end{proof}

\begin{theorem}[Choi 矩阵的纤维块分解与谱公式]\label{thm:op-algebra-fold-quantum-channel-choi-spectrum-mi}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_choi\_spectrum\_mi}
令 \(n:=|\Omega_m|=2^m\)，并记 \(\rho_{RB}:=J(\mathcal{E}_m)/n\) 为 \(\mathcal{E}_m\) 的归一化 Choi 态。
则 Choi 矩阵满足严格分解
\[
J(\mathcal{E}_m)=\sum_{x\in X_m}\frac{1}{d_m(x)}\,\Pi_x\otimes \Pi_x,
\qquad
\rho_{RB}=\frac{1}{n}\sum_{x\in X_m}\frac{1}{d_m(x)}\,\Pi_x\otimes \Pi_x.
\]
再令 \(\pi_m\) 为 \(\Omega_m\) 上均匀分布经 \(\Fold_m\) 推送到 \(X_m\) 的分布，即
\[
\pi_m(x)=\frac{d_m(x)}{2^m}.
\]
则：
\begin{enumerate}
  \item \(\rho_{RB}\) 为可分态，因而 \(\mathcal{E}_m\) 为纠缠破坏通道；并且其边缘态满足
  \[
  \rho_R=\rho_B=\frac{I}{n};
  \]
  \item \(J(\mathcal{E}_m)\) 的全部非零特征值为
  \[
  \frac{1}{d_m(x)}\qquad (x\in X_m),
  \]
  且每个特征值的重数恰为 \(d_m(x)^2\)；等价地，\(\rho_{RB}\) 的全部非零特征值为
  \[
  \lambda_x=\frac{1}{n\,d_m(x)}\qquad (x\in X_m),
  \]
  且每个 \(\lambda_x\) 的重数恰为 \(d_m(x)^2\)；
  \item Choi 互信息满足精确恒等式
  \[
  I(R\!:\!B)_{\rho}=H(\pi_m),
  \]
  其中 \(H(\cdot)\) 为 Shannon 熵（以自然对数计）。等价地，
  \[
  I(R\!:\!B)_{\rho}=m\log 2-\sum_{x\in X_m}\frac{d_m(x)}{2^m}\log d_m(x).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-quantum-channel-minimal-environment-equals-s2} 的 Choi 计算，
\[
J(\mathcal{E}_m)
=\sum_{x\in X_m}\ \sum_{i\in\Fold_m^{-1}(x)} \ket{i}\!\bra{i}\otimes \frac{\Pi_x}{d_m(x)}
=\sum_{x\in X_m}\frac{1}{d_m(x)}\,\Pi_x\otimes \Pi_x,
\]
从而得到 \(\rho_{RB}=J(\mathcal{E}_m)/n\) 的分解式。该态显然是若干乘积态 \(\ket{i}\!\bra{i}\otimes(\Pi_x/d_m(x))\) 的凸组合，故可分，纠缠破坏结论随之成立。

由
\(
J(\mathcal{E}_m)=\sum_x (1/d_m(x))\,\Pi_x\otimes\Pi_x
\)
与 \(\Pi_x\otimes\Pi_x\) 的正交分解可知：
对每个 \(x\)，\(\Pi_x\otimes\Pi_x\) 为秩 \(d_m(x)^2\) 的正交投影，且该谱块的唯一非零特征值为 \(1/d_m(x)\)，重数为 \(d_m(x)^2\)。
归一化后即得 \(\rho_{RB}\) 的特征值 \(\lambda_x=1/(n d_m(x))\) 与重数。

边缘态计算为
\[
\rho_R=\Tr_B\rho_{RB}=\frac{1}{n}\sum_{i\in\Omega_m}\ket{i}\!\bra{i}=\frac{I}{n},
\qquad
\rho_B=\Tr_R\rho_{RB}=\frac{1}{n}\sum_{x\in X_m}\Pi_x=\frac{I}{n},
\]
故 \(S(\rho_R)=S(\rho_B)=\log n=m\log 2\)。
另一方面，由谱分解得
\[
S(\rho_{RB})
=-\sum_{x\in X_m} d_m(x)^2\cdot \frac{1}{n d_m(x)}\log\!\Bigl(\frac{1}{n d_m(x)}\Bigr)
=\sum_{x\in X_m}\frac{d_m(x)}{n}\log\bigl(n d_m(x)\bigr).
\]
代入互信息定义 \(I(R\!:\!B)=S(\rho_R)+S(\rho_B)-S(\rho_{RB})\) 得
\[
I(R\!:\!B)_{\rho}
=\log n-\sum_{x\in X_m}\pi_m(x)\log d_m(x).
\]
而
\[
H(\pi_m)=-\sum_x \pi_m(x)\log \pi_m(x)
=\log n-\sum_x \pi_m(x)\log d_m(x),
\]
故 \(I(R\!:\!B)_{\rho}=H(\pi_m)\)。证毕。
\end{proof}

\begin{corollary}[Rényi--谱矩与纤维统计量]\label{cor:op-algebra-fold-quantum-channel-choi-renyi-moments}
令 \(n:=|\Omega_m|=2^m\)，并记 \(\rho_{RB}:=J(\mathcal{E}_m)/n\)。
对任意实数 \(\alpha>0\)（\(\alpha\neq 1\)）有
\[
\Tr(\rho_{RB}^\alpha)=n^{-\alpha}\sum_{x\in X_m} d_m(x)^{2-\alpha},
\qquad
H_\alpha(\rho_{RB})=\frac{1}{1-\alpha}\log\!\Bigl(n^{-\alpha}\sum_{x\in X_m} d_m(x)^{2-\alpha}\Bigr),
\]
其中 \(H_\alpha\) 为 Rényi 熵。
特别地，令
\[
d_m^{\min}:=\min_{x\in X_m}d_m(x),
\qquad
D_m:=\max_{x\in X_m}d_m(x),
\]
则
\[
\|\rho_{RB}\|_\infty=\frac{1}{n\,d_m^{\min}},
\qquad
\lambda_{\min}^{+}(\rho_{RB})=\frac{1}{n\,D_m},
\qquad
\Tr(\rho_{RB}^2)=\frac{|X_m|}{n^2}.
\]
此外 von Neumann 熵满足闭式
\[
S(\rho_{RB})=\log n+\sum_{x\in X_m}\frac{d_m(x)}{n}\log d_m(x).
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-quantum-channel-choi-spectrum-mi}，\(\rho_{RB}\) 的非零特征值为 \(\lambda_x=1/(n d_m(x))\)，重数为 \(d_m(x)^2\)。
因此
\[
\Tr(\rho_{RB}^\alpha)=\sum_{x\in X_m} d_m(x)^2\Bigl(\frac{1}{n d_m(x)}\Bigr)^\alpha
=n^{-\alpha}\sum_{x\in X_m} d_m(x)^{2-\alpha},
\]
从而 Rényi 熵公式成立。
\(\|\rho_{RB}\|_\infty\) 与 \(\lambda_{\min}^+(\rho_{RB})\) 分别由最大/最小非零特征值给出，得到与 \(d_m^{\min}\)、\(D_m\) 的对应关系。
当 \(\alpha=2\) 时得到
\(
\Tr(\rho_{RB}^2)=n^{-2}\sum_x d_m(x)^{0}=|X_m|/n^2
\)。
von Neumann 熵由谱和给出：
\[
S(\rho_{RB})
=-\sum_{x\in X_m}d_m(x)^2\cdot \frac{1}{n d_m(x)}\log\!\Bigl(\frac{1}{n d_m(x)}\Bigr)
=\log n+\sum_{x\in X_m}\frac{d_m(x)}{n}\log d_m(x).
\]
证毕。
\end{proof}

\begin{theorem}[容量的极值：经典容量为 \texorpdfstring{$\log|X_m|$}{log|Xm|}，量子容量为零]\label{thm:op-algebra-fold-quantum-channel-capacity}
设 \(\mathcal{E}_m\) 如 \eqref{eq:op-algebra-fold-quantum-channel-def}。
则：
\begin{enumerate}
  \item \(\mathcal{E}_m\) 的量子容量 \(Q(\mathcal{E}_m)=0\)，且私密容量亦为 \(0\)；
  \item \(\mathcal{E}_m\) 的经典容量满足
  \[
  C(\mathcal{E}_m)=\log |X_m|=\log F_{m+2}.
  \]
\end{enumerate}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_capacity}
\end{theorem}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-quantum-channel-choi-spectrum-mi}，\(\mathcal{E}_m\) 为纠缠破坏通道，故其量子容量与私密容量均为零。

经典容量方面：令 \(\mathcal{M}(\rho):=\sum_{x\in X_m}\Tr(\Pi_x\rho)\ket{x}\!\bra{x}\) 为以 \(\{\Pi_x\}\) 为 POVM 的量子--经典测量通道，令 \(\mathcal{P}(\ket{x}\!\bra{x}):=\Pi_x/d_m(x)\) 为对应制备通道，则 \(\mathcal{E}_m=\mathcal{P}\circ\mathcal{M}\)。
因此任意经 \(\mathcal{E}_m\) 传输的信息必经由字母表大小为 \(|X_m|\) 的经典寄存器 \(x\)，从而 \(C(\mathcal{E}_m)\le \log|X_m|\)。
另一方面，对每个 \(x\in X_m\) 取任意纯态 \(\rho_x\) 支撑于 \(\Pi_x\mathcal{H}_m\)，则 \(\Tr(\Pi_{x'}\rho_x)=\delta_{x,x'}\)，故 \(\mathcal{M}(\rho_x)=\ket{x}\!\bra{x}\) 可被零误差区分，于是 \(C(\mathcal{E}_m)\ge \log|X_m|\)。
两侧合并得等式。证毕。
\end{proof}

\begin{proposition}[OAQEC 视角下的最大可恢复代数]\label{prop:op-algebra-fold-quantum-channel-max-recoverable-algebra}
把 \(\mathcal{E}_m\) 视作噪声通道。
在算子代数量子纠错（operator-algebra quantum error correction, OAQEC）意义下，其可恢复的最大 \(C^\ast\)-子代数为
\[
\mathcal{Z}_m:=\Span\{\Pi_x:\ x\in X_m\}\ \cong\ \CC^{X_m},
\]
且任意可恢复 \(\ast\)-子代数都包含于 \(\mathcal{Z}_m\)。
此外 \(\mathcal{Z}_m\) 点态固定：对一切 \(a\in\mathcal{Z}_m\) 有 \(\mathcal{E}_m(a)=a\)，故恢复可取恒等映射。
\end{proposition}
\begin{proof}
由定理 \ref{thm:op-algebra-fold-quantum-channel-minimal-environment-equals-s2} 的 Kraus 族 \(\{K^{(x)}_{\omega,\omega'}\}\)（其中 \(\omega,\omega'\in\Fold_m^{-1}(x)\)），可直接计算
\[
\bigl(K^{(x)}_{\omega,\omega'}\bigr)^\ast K^{(x)}_{\eta,\eta'}
=\frac{1}{d_m(x)}\,\delta_{\omega,\eta}\,\ket{\omega'}\!\bra{\eta'}.
\]
因此 \(\Span\{K_i^\ast K_j\}\) 线性张成
\(
\bigoplus_{x\in X_m}\mathcal{B}(\Pi_x\mathcal{H}_m)
\)，其交换子代数恰为每个块上的标量算子，即 \(\mathcal{Z}_m\)。
OAQEC 的 Knill--Laflamme 型判据表明：若 \(\mathcal{A}\subset\mathcal{B}(\mathcal{H}_m)\) 可恢复，则必须满足 \(K_i^\ast K_j\in \mathcal{A}'\)（对一切 \(i,j\)），从而 \(\mathcal{A}\subseteq \bigl(\Span\{K_i^\ast K_j\}\bigr)'=\mathcal{Z}_m\)（参见 \cite{DauphinaisKribsVasmer2024StabilizerFormalismOAQEC}）。
反向方向：任取 \(a=\sum_x \alpha_x\Pi_x\in\mathcal{Z}_m\)，代入 \eqref{eq:op-algebra-fold-quantum-channel-def} 得 \(\mathcal{E}_m(a)=a\)，故 \(\mathcal{Z}_m\) 可恢复且恢复可取恒等映射。证毕。
\end{proof}

\begin{theorem}[纤维群胚的弱 Hopf 代数化与量子操作实现]\label{thm:op-algebra-fold-quantum-channel-weak-hopf-groupoid}
\leanverified{paper\_op\_algebra\_fold\_quantum\_channel\_weak\_hopf\_groupoid}
定义折叠纤维群胚（等价关系群胚）
\[
\mathcal{G}_m:=\Omega_m\times_{X_m}\Omega_m
=\bigl\{(\omega,\omega')\in\Omega_m^2:\ \Fold_m(\omega)=\Fold_m(\omega')\bigr\}.
\]
令 \(\CC[\mathcal{G}_m]\) 为其复群胚代数，以基元 \(e_{\omega,\omega'}\)（\((\omega,\omega')\in\mathcal{G}_m\)）张成，并以
\[
e_{\omega,\omega'}e_{\eta,\eta'}=\delta_{\omega',\eta}\,e_{\omega,\eta'},
\qquad
e_{\omega,\omega'}^\ast=e_{\omega',\omega}
\]
定义乘法与 \(\ast\)。
则：
\begin{enumerate}
  \item \(\CC[\mathcal{G}_m]\) 自然成为有限维弱 Hopf \(\ast\)-代数，其余结构映射可取为
  \[
  \Delta(e_{\omega,\omega'})=\sum_{u:\ \Fold_m(u)=\Fold_m(\omega)} e_{\omega,u}\otimes e_{u,\omega'},
  \qquad
  \varepsilon(e_{\omega,\omega'})=\delta_{\omega,\omega'}.
  \]
  \item \(\CC[\mathcal{G}_m]\) 分解为纤维块的满矩阵代数直和
  \[
  \CC[\mathcal{G}_m]\ \cong\ \bigoplus_{x\in X_m} M_{d_m(x)}(\CC),
  \]
  因而 \(\dim_{\CC}\CC[\mathcal{G}_m]=\sum_x d_m(x)^2=S_2(m)\)。
  \item 令 \(\Phi:\CC[\mathcal{G}_m]\to \End(\mathcal{B}(\mathcal{H}_m))\) 为由基元诱导的完全正表示
  \[
  \Phi(e_{\omega,\omega'})(\rho):=\ket{\omega}\!\bra{\omega'}\,\rho\,\ket{\omega'}\!\bra{\omega},
  \]
  并令
  \[
  h_m:=\sum_{x\in X_m}\ \frac{1}{d_m(x)}\sum_{\omega,\omega'\in\Fold_m^{-1}(x)} e_{\omega,\omega'}\ \in\ \CC[\mathcal{G}_m].
  \]
  则 \(\mathcal{E}_m=\Phi(h_m)\)。
  因此 \(\mathcal{E}_m\) 可视为该弱 Hopf 对称在局域算符上的“量子操作型作用”，在可计算意义下给出非可逆对称以完全正映射作用于算符的显式模型（比较 \cite{OkadaTachikawa2024NonInvertibleQuantumOperations}）。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)}--\textbf{(2)} 群胚代数的弱 Hopf 结构为标准构造；并且在每条纤维 \(\Fold_m^{-1}(x)\) 上，\(\mathcal{G}_m\) 的箭头集合给出“配对群胚”，其代数同构于 \(M_{d_m(x)}(\CC)\)，不同 \(x\) 间无可合成态，故为直和分解。
该分解也可由附录 \ref{subsubsec:op-algebra-fold-jones-basic-construction} 的矩阵单位 \(\{E_{\omega,\omega'}\}\) 直接给出：映射 \(e_{\omega,\omega'}\mapsto E_{\omega,\omega'}\) 在每条纤维上实现 \(M_{d_m(x)}(\CC)\) 的同构。

\textbf{(3)} 对任意 \(\rho\) 有
\[
\Phi(h_m)(\rho)
=\sum_{x\in X_m}\frac{1}{d_m(x)}\sum_{\omega,\omega'\in\Fold_m^{-1}(x)}
\ket{\omega}\!\bra{\omega'}\,\rho\,\ket{\omega'}\!\bra{\omega}.
\]
对固定 \(x\) 先对 \(\omega'\) 求和得到 \(\Tr(\Pi_x\rho)\) 的系数，再对 \(\omega\) 求和得到 \(\Tr(\Pi_x\rho)\Pi_x\)，从而
\(
\Phi(h_m)(\rho)=\sum_x \frac{\Tr(\Pi_x\rho)}{d_m(x)}\Pi_x=\mathcal{E}_m(\rho)
\)。
证毕。
\end{proof}

\begin{theorem}[\(\mathbb Z_2^2\)-分级弱 Hopf 量子群胚的封闭性]\label{thm:op-algebra-fold-weak-hopf-z2x2-sector-closure}
沿用定理 \ref{thm:op-algebra-fold-quantum-channel-weak-hopf-groupoid} 的弱 Hopf \(\ast\)-代数 \(\CC[\mathcal{G}_m]\)。
对任意 \((\alpha,\beta)\in\{\pm1\}^2\)，令
\[
H_m^{\alpha,\beta}:=\bigoplus_{\chi(x)=(\alpha,\beta)}\CC[\mathcal{G}_m]_x
\]
为由取向指纹 \(\chi\)（命题 \ref{prop:fold-groupoid-z2x2-sector-by-orientation}）选取的 \(\mathbb Z_2^2\)-扇区理想，其中
\[
\CC[\mathcal{G}_m]_x:=\Span\{e_{\omega,\omega'}:\ \Fold_m(\omega)=\Fold_m(\omega')=x\}\ \cong\ M_{d_m(x)}(\CC)
\]
为第 \(x\) 个矩阵块。
则：
\begin{enumerate}
  \item \(H_m^{\alpha,\beta}\) 在乘法、\(\ast\)、余乘法 \(\Delta\)、对极 \(S\) 与余单位 \(\varepsilon\) 下封闭，并以自身幺元
  \[
  1_{\alpha,\beta}:=\sum_{\chi(x)=(\alpha,\beta)}\ \sum_{\omega\in\Fold_m^{-1}(x)}e_{\omega,\omega}
  \]
  成为弱 Hopf \(\ast\)-代数；
  \item 存在弱 Hopf \(\ast\)-代数的直和分解
  \[
  \CC[\mathcal{G}_m]\cong\bigoplus_{(\alpha,\beta)\in\{\pm1\}^2}H_m^{\alpha,\beta};
  \]
  \item 有限维表示范畴按中心幂等元分裂：
  \[
  \mathrm{Rep}(\CC[\mathcal{G}_m])\ \simeq\ \prod_{(\alpha,\beta)\in\{\pm1\}^2}\mathrm{Rep}(H_m^{\alpha,\beta}).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
在基元 \(e_{\omega,\omega'}\) 上，余乘法满足
\[
\Delta(e_{\omega,\omega'})=\sum_{u:\ \Fold_m(u)=\Fold_m(\omega)} e_{\omega,u}\otimes e_{u,\omega'}.
\]
因此若 \(\Fold_m(\omega)=\Fold_m(\omega')=x\)，则和式中每一项仍落在同一 \(x\)-块的张量积中，从而 \(\Delta\) 逐块封闭。
\(\ast\) 由 \(e_{\omega,\omega'}^\ast=e_{\omega',\omega}\) 给出，亦保持 \(\Fold_m(\omega)=\Fold_m(\omega')\) 的标签不变。
弱 Hopf 结构的对极在配对群胚情形可取为
\(
S(e_{\omega,\omega'})=e_{\omega',\omega}
\)，因而同样逐块封闭；余单位 \(\varepsilon(e_{\omega,\omega'})=\delta_{\omega,\omega'}\) 显然逐块封闭。
\par
由块直和分解 \(\CC[\mathcal{G}_m]\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC)\)，按 \(\chi(x)\) 收集矩阵块得到两侧理想直和分解；
对固定 \((\alpha,\beta)\) 的直和在乘法下封闭，并具有幺元 \(1_{\alpha,\beta}\)（各块单位之和）。
由此 \(H_m^{\alpha,\beta}\) 继承弱 Hopf \(\ast\)-代数结构，得到 (1) 与 (2)。
\par
(3) 为有限维半单代数的中心幂等元分解的标准结论：\(\CC[\mathcal{G}_m]\) 的中心幂等元把任意表示分解为扇区直积，从而得到所示范畴等价。证毕。
\end{proof}
