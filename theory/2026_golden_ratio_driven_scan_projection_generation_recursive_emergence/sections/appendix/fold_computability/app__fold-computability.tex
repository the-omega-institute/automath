\section{折叠算子 \texorpdfstring{$\Fold_m$}{Fold\_m} 的可计算性与基本性质}
\label{app:fold-computability}

Zeckendorf 规范形可由贪心算法计算：对给定 $N\in\NN$，反复减去不超过当前余量的最大 Fibonacci 权重，并跳过相邻权重以强制无相邻 $1$。算法终止且输出唯一规范形 $Z(N)=(c_1,c_2,\dots)$（\cite{Zeckendorf1972}）。

对任意 $\omega=\omega_1\cdots\omega_m\in\Omega_m$，按定义 \ref{def:fold-word} 先计算 Fibonacci 加权值
\[
N(\omega)=\sum_{k=1}^m \omega_k F_{k+1},
\]
再取 $Z(N(\omega))$ 的前 $m$ 位得到 $\Fold_m(\omega)=\pi_m(Z(N(\omega)))\in X_m$。上述过程由有限次整数运算与 Zeckendorf 贪心规范化组成，因而给出对 $\Fold_m$ 的显式算法。

命题 \ref{prop:fold-basic} 的三个性质可直接由 Zeckendorf 唯一性推出：
\begin{itemize}
  \item \textbf{规范性：} $\pi_m(Z(\cdot))$ 输出的前 $m$ 位满足无相邻 $1$，故落在 $X_m$；
  \item \textbf{幂等性：}若 $\omega\in X_m$，则其加权值 $N(\omega)$ 已是合法 Zeckendorf 表示，唯一性迫使 $Z(N(\omega))$ 的前 $m$ 位等于 $\omega$；
  \item \textbf{满射性：}对任意 $x\in X_m$，取 $\omega=x$ 即有 $\Fold_m(\omega)=x$。
\end{itemize}

\begin{lemma}[Zeckendorf 范围上的双射]\label{lem:zeckendorf-range-bijection}
对任意 $m\ge 1$，若 $0\le N\le F_{m+2}-1$，则 $Z(N)$ 的非零位索引不超过 $m$，从而
\[
\pi_m(Z(N))=Z(N)\ \text{（作为长度 }m\text{ 的补零截断）}.
\]
因此映射 $N\mapsto \pi_m(Z(N))$ 在 $\{0,1,\dots,F_{m+2}-1\}$ 上给出到 $X_m$ 的双射，其逆映射为 $w\mapsto V_m(w)$。
\leanverified{paper\_zeckendorf\_range\_bijection}
\end{lemma}

\textbf{证明：}
若 $Z(N)$ 在第 $m+1$ 位有 $1$，则 $N\ge F_{m+2}$（因为该位权重即为 $F_{m+2}$ 且其余项非负），与 $N\le F_{m+2}-1$ 矛盾。故 $c_{m+1}=c_{m+2}=\cdots=0$，从而截断不丢信息并给出双射结论。
