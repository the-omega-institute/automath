\paragraph{以端点 \texorpdfstring{$w=-1$}{w=-1} 为基点的 Busemann 坐标与 horocycle 分层}\label{par:app-horocycle-busemann}
在 \eqref{eq:app-cayley-disk} 的 Cayley 紧化下，端点 \(w=-1\) 是 \(t\to\pm\infty\) 的共同像。
围绕该端点，可以把“离临界线偏移”严格表达为一个以 \(-1\) 为基点的双曲几何坐标，并由此得到与 Jensen 缺陷分辨率门槛对齐的 horocycle 分层。

\begin{proposition}[端点 Busemann/Poisson 坐标（指数化 Busemann 函数）]\label{prop:app-busemann-poisson-minus1}
\leanverified{paper\_app\_busemann\_poisson\_minus1}
对 \(w\in\mathbb{D}\) 定义
\[
B^{-1}(w):=\Im\bigl(\mathcal{C}^{-1}(w)\bigr).
\]
则 \(B^{-1}(w)\) 具有完全显式的闭式
\begin{equation}\label{eq:app-busemann-poisson-minus1-closed}
\boxed{\;
B^{-1}(w)
=\frac{1-|w|^{2}}{|1+w|^{2}}
\; }.
\end{equation}
并且 \(b_{-1}(w):=\log B^{-1}(w)\)（差一个常数）可取作圆盘模型中以边界点 \(-1\) 为基点的 Busemann 函数；\(B^{-1}\) 因而是其指数化坐标（亦即相应的 Poisson 核）。
\end{proposition}

\begin{proof}
令 \(t=\mathcal{C}^{-1}(w)=x+\mathrm{i}y\in\mathbb{H}\)。
由 \eqref{eq:app-cayley-disk} 得
\(
1+w=1+\frac{1+\mathrm{i}t}{1-\mathrm{i}t}=\frac{2}{1-\mathrm{i}t}
\)，从而
\[
|1+w|^{2}=\frac{4}{|1-\mathrm{i}t|^{2}}
=\frac{4}{x^{2}+(1+y)^{2}}.
\]
另一方面，
\(
|w|^{2}=\frac{|1+\mathrm{i}t|^{2}}{|1-\mathrm{i}t|^{2}}
=\frac{x^{2}+(1-y)^{2}}{x^{2}+(1+y)^{2}}
\)，故
\[
1-|w|^{2}
=\frac{(1+y)^{2}-(1-y)^{2}}{x^{2}+(1+y)^{2}}
=\frac{4y}{x^{2}+(1+y)^{2}}.
\]
将两式相除即得
\(
\frac{1-|w|^{2}}{|1+w|^{2}}=y=\Im(t)=B^{-1}(w)
\)，从而 \eqref{eq:app-busemann-poisson-minus1-closed} 成立。
关于 Busemann 函数的陈述是标准双曲几何事实：圆盘模型中以 \(\xi\in\partial\mathbb{D}\) 为基点的 Busemann 函数可取为 \(\log\frac{1-|w|^{2}}{|w-\xi|^{2}}\)（差一个常数），故在 \(\xi=-1\) 时得到 \(b_{-1}(w)=\log B^{-1}(w)\)。证毕。
\end{proof}

\begin{theorem}[horocycle 分层：\texorpdfstring{$B^{-1}(w)=\delta$}{Bminus1=delta} 的水平集是内切圆]\label{thm:app-horocycle-foliation-minus1}
\leanverified{paper\_app\_horocycle\_foliation\_minus1}
对任意 \(\delta>0\) 定义
\[
\mathcal{H}_{\delta}:=\Bigl\{w\in\mathbb{D}:\ B^{-1}(w)=\delta\Bigr\}
=\Bigl\{w\in\mathbb{D}:\ \frac{1-|w|^{2}}{|1+w|^{2}}=\delta\Bigr\}.
\]
则 \(\mathcal{H}_{\delta}\) 是一个与单位圆 \(\partial\mathbb{D}\) 在点 \(-1\) 内切的欧氏圆（双曲几何中的 horocycle），并具有显式圆方程
\begin{equation}\label{eq:app-horocycle-circle-euclid}
\boxed{\;
\mathcal{H}_{\delta}
=\left\{w\in\mathbb{D}:\ \left|w+\frac{\delta}{1+\delta}\right|=\frac{1}{1+\delta}\right\}.
\;}
\end{equation}
在上半平面坐标中，\(\mathcal{C}^{-1}(\mathcal{H}_{\delta})=\{t\in\mathbb{H}:\Im(t)=\delta\}\) 为水平直线；因此 \(\delta\) 可理解为以端点 \(w=-1\) 为基点的 Busemann 坐标层。
\end{theorem}

\begin{proof}
由命题 \ref{prop:app-busemann-poisson-minus1}，\(\mathcal{C}^{-1}(\mathcal{H}_{\delta})\) 等于 \(\Im(t)=\delta\) 的水平直线，其在上半平面模型中即为以 \(\infty\) 为基点的 horocycle。
将其经 \(\mathcal{C}\) 推回圆盘即得到以 \(-1\) 为基点的 horocycle。
\par
为得到显式欧氏圆方程，设 \(w=x+\mathrm{i}y\)。
由 \(\frac{1-|w|^{2}}{|1+w|^{2}}=\delta\) 得
\[
1-(x^{2}+y^{2})=\delta\bigl((1+x)^{2}+y^{2}\bigr).
\]
整理并配方得到
\[
\left(x+\frac{\delta}{1+\delta}\right)^{2}+y^{2}=\frac{1}{(1+\delta)^{2}},
\]
即 \eqref{eq:app-horocycle-circle-euclid}。
该圆心为 \(-\frac{\delta}{1+\delta}\in(-1,0)\)，半径为 \(\frac{1}{1+\delta}\)，其最左端点为 \(-1\)，故与 \(\partial\mathbb{D}\) 在 \(-1\) 内切。证毕。
\end{proof}

\begin{corollary}[离临界线偏移的 horocycle 几何实现与二阶抑制]\label{cor:app-offcritical-horocycle-busemann}
\leanverified{paper\_app\_offcritical\_horocycle\_busemann}
在命题 \ref{prop:app-offcritical-radius-compression} 的设定下，若 \(\rho=\beta+\mathrm{i}\gamma\) 为离临界线零点（\(\beta<\tfrac12\)），令 \(\delta=\tfrac12-\beta>0\) 且 \(w_{\rho}=\mathcal{C}(\gamma+\mathrm{i}\delta)\)。
则
\[
\boxed{\;\delta=B^{-1}(w_{\rho})=\frac{1-|w_{\rho}|^{2}}{|1+w_{\rho}|^{2}}\;},
\qquad\text{从而}\qquad
w_{\rho}\in\mathcal{H}_{\delta}.
\]
并且有完全显式恒等式
\[
\boxed{\;
|1+w_{\rho}|^{2}=\frac{4}{\gamma^{2}+(1+\delta)^{2}},
\qquad
1-|w_{\rho}|^{2}=\frac{4\delta}{\gamma^{2}+(1+\delta)^{2}}
=\delta\,|1+w_{\rho}|^{2}\; }.
\]
因此在端点邻域 \(|1+w|\ll 1\) 的自然尺度下，“离临界线偏移 \(\delta\)”以\emph{端点 chordal 距离的平方}进入：径向缺口 \(1-|w|^{2}\) 被严格写成 \(\delta|1+w|^{2}\)，与注 \ref{rem:app-offcritical-second-order-radial-suppression} 的“二阶径向通道”表述一致。
\end{corollary}

\begin{proof}
由命题 \ref{prop:app-offcritical-radius-compression}，\(t_{\rho}=\gamma+\mathrm{i}\delta\) 且 \(w_{\rho}=\mathcal{C}(t_{\rho})\)。
故 \(\delta=\Im(t_{\rho})=\Im(\mathcal{C}^{-1}(w_{\rho}))=B^{-1}(w_{\rho})\)，并由定理 \ref{thm:app-horocycle-foliation-minus1} 得 \(w_{\rho}\in\mathcal{H}_{\delta}\)。
另一方面，由 \(1+w_{\rho}=2/(1-\mathrm{i}t_{\rho})\) 得
\[
|1+w_{\rho}|^{2}=\frac{4}{|1-\mathrm{i}t_{\rho}|^{2}}
=\frac{4}{\gamma^{2}+(1+\delta)^{2}}.
\]
再由 \eqref{eq:app-offcritical-radius-closed} 得
\(
1-|w_{\rho}|^{2}=\frac{4\delta}{\gamma^{2}+(1+\delta)^{2}}
\)，合并即得 \(1-|w_{\rho}|^{2}=\delta|1+w_{\rho}|^{2}\)。证毕。
\end{proof}

\begin{theorem}[共动视界局部化：移位 Cayley 的深度公式]\label{thm:app-comoving-horizon-localization}
\leanverified{paper\_app\_comoving\_horizon\_localization}
对任意 \(x_{0}\in\RR\) 定义移位 Cayley 紧化
\[
\mathcal{C}_{x_{0}}(t):=\mathcal{C}(t-x_{0}).
\]
在命题 \ref{prop:app-offcritical-radius-compression} 的设定下，若 \(\rho=\beta+\mathrm{i}\gamma\) 为离临界线零点（\(\beta<\tfrac12\)），令 \(\delta=\tfrac12-\beta>0\)，并定义
\[
w_{\rho,x_{0}}:=\mathcal{C}_{x_{0}}(\gamma+\mathrm{i}\delta)=\mathcal{C}\bigl((\gamma-x_{0})+\mathrm{i}\delta\bigr)\in\mathbb{D},
\qquad
h_{x_{0}}(\rho):=1-|w_{\rho,x_{0}}|^{2}.
\]
则有闭式
\[
\boxed{\;
B^{-1}(w_{\rho,x_{0}})=\delta,\qquad
|1+w_{\rho,x_{0}}|^{2}=\frac{4}{(\gamma-x_{0})^{2}+(1+\delta)^{2}},\qquad
h_{x_{0}}(\rho)=\frac{4\delta}{(\gamma-x_{0})^{2}+(1+\delta)^{2}}
=\delta\,|1+w_{\rho,x_{0}}|^{2}\; }.
\]
特别地，取 \(x_{0}=\gamma\) 得
\[
w_{\rho,\gamma}=\mathcal{C}(\mathrm{i}\delta)=\frac{1-\delta}{1+\delta}\in(0,1),
\qquad
h_{\gamma}(\rho)=\frac{4\delta}{(1+\delta)^{2}}\sim 4\delta\qquad(\delta\downarrow 0),
\]
从而在共动参数 \(x_{0}=\gamma\) 的局部化坐标中，离临界线偏移不再随高度 \(|\gamma|\) 产生二阶压缩。
\end{theorem}

\begin{proof}
由定义 \(w_{\rho,x_{0}}=\mathcal{C}(t)\) 其中 \(t=(\gamma-x_{0})+\mathrm{i}\delta\)，故
\(
B^{-1}(w_{\rho,x_{0}})=\Im(\mathcal{C}^{-1}(w_{\rho,x_{0}}))=\Im(t)=\delta
\)。
其余恒等式与推论 \ref{cor:app-offcritical-horocycle-busemann} 的证明完全同型：由 \(1+w_{\rho,x_{0}}=2/(1-\mathrm{i}t)\) 得
\[
|1+w_{\rho,x_{0}}|^{2}=\frac{4}{|1-\mathrm{i}t|^{2}}=\frac{4}{(\gamma-x_{0})^{2}+(1+\delta)^{2}},
\]
并由 \(|\mathcal{C}(t)|^{2}=\frac{|1+\mathrm{i}t|^{2}}{|1-\mathrm{i}t|^{2}}=\frac{(\gamma-x_{0})^{2}+(1-\delta)^{2}}{(\gamma-x_{0})^{2}+(1+\delta)^{2}}\) 得
\(
h_{x_{0}}(\rho)=1-|w_{\rho,x_{0}}|^{2}=\frac{4\delta}{(\gamma-x_{0})^{2}+(1+\delta)^{2}}
\)，并立即推出 \(h_{x_{0}}(\rho)=\delta\,|1+w_{\rho,x_{0}}|^{2}\)。
取 \(x_{0}=\gamma\) 时 \(t=\mathrm{i}\delta\)，从而 \(w_{\rho,\gamma}=\mathcal{C}(\mathrm{i}\delta)=\frac{1-\delta}{1+\delta}\) 并代入闭式得 \(h_{\gamma}(\rho)=\frac{4\delta}{(1+\delta)^{2}}\)。证毕。
\end{proof}

\begin{corollary}[端点分辨率下界（horocycle 表述）]\label{cor:app-horocycle-visible-resolution}
\leanverified{paper\_app\_horocycle\_visible\_resolution}
设存在离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\)（\(\beta<\tfrac12\)），并令 \(w_{\rho}\) 如推论 \ref{cor:app-offcritical-horocycle-busemann}。
若希望在半径审计点 \(\varrho\in(0,1)\) 处强制出现由该零点贡献导致的正缺陷（即保证 \(\varrho>|w_{\rho}|\)，从而 \(D_{\zeta}(\varrho)\ge \log(\varrho/|w_{\rho}|)>0\)，推论 \ref{cor:app-jensen-visible-threshold-offcritical}），则必须满足边界层分辨率门槛
\[
1-\varrho<1-|w_{\rho}|
\le 1-|w_{\rho}|^{2}
=\delta\,|1+w_{\rho}|^{2}
=\frac{4\delta}{\gamma^{2}+(1+\delta)^{2}}.
\]
特别地，在 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，该必要条件化为
\[
1-\varrho\ \lesssim\ \frac{\delta}{\gamma^{2}},
\]
即高高度区间中“排除离线零点”的半径审计必须进入 \(O(\delta/\gamma^{2})\) 的端点边界层分辨率。
\end{corollary}

\begin{proof}
若 \(\varrho\le |w_{\rho}|\)，则圆盘 \(|w|<\varrho\) 不包含零点 \(w_{\rho}\)，Jensen 求和式 \eqref{eq:app-jensen-sum} 在 \(\varrho\) 处不含该零点项，故不能强制得到由该零点导致的正缺陷下界。
因此必须有 \(\varrho>|w_{\rho}|\)，等价于 \(1-\varrho<1-|w_{\rho}|\)。
而 \(1-|w_{\rho}|\le 1-|w_{\rho}|^{2}\) 对所有 \(|w_{\rho}|<1\) 成立；最后两步由推论 \ref{cor:app-offcritical-horocycle-busemann} 给出的恒等式得到。
高高度渐近由推论 \ref{cor:app-offcritical-radius-high-height} 或直接对 \(\frac{4\delta}{\gamma^{2}+(1+\delta)^{2}}\) 取 \(|\gamma|\to\infty\) 得到。证毕。
\end{proof}
