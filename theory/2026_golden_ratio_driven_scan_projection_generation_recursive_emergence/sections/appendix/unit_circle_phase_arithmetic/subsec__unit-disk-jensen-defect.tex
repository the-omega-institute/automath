\subsection{Jensen 缺陷：把盘内零点计数压缩为一族有限圆周平均}\label{subsec:unit-disk-jensen-defect}
定理 \ref{thm:app-rh-iff-disk-zero-free} 把 RH 转写为“开圆盘内无零点”。
这一条件可进一步完全有限化：零点信息被 Jensen 公式压缩为一族半径 \(\varrho\in(0,1)\) 的\textbf{有限圆周积分缺陷}。

\begin{definition}[Jensen 缺陷]\label{def:app-jensen-defect}
令 \(F_{\zeta}\) 如 \eqref{eq:app-Fzeta-disk}。
对任意 \(0<\varrho<1\) 定义 Jensen 缺陷
\begin{equation}\label{eq:app-jensen-defect}
\boxed{\;
D_{\zeta}(\varrho)
:=\frac{1}{2\pi}\int_{0}^{2\pi}\log\bigl|F_{\zeta}(\varrho e^{\mathrm{i}\theta})\bigr|\,\dd\theta
-\log|F_{\zeta}(0)|\; }.
\end{equation}
\end{definition}

\begin{proposition}[Jensen 公式与非负缺陷]\label{prop:app-jensen-formula-defect}
设 \(0<\varrho<1\)。
令 \(\{a_k\}\) 为 \(F_{\zeta}\) 在 \(|w|<\varrho\) 内的零点（按重数计）。
则 Jensen 公式给出
\begin{equation}\label{eq:app-jensen-sum}
\boxed{\;
D_{\zeta}(\varrho)
=\sum_{|a_k|<\varrho}\log\!\Bigl(\frac{\varrho}{|a_k|}\Bigr)\ \ge\ 0\; }.
\end{equation}
从而 \(D_{\zeta}(\varrho)=0\) 当且仅当 \(F_{\zeta}\) 在 \(|w|<\varrho\) 内无零点。
\leanverified{paper\_app\_jensen\_formula\_defect}
\end{proposition}
\begin{proof}
由于 \(F_{\zeta}\) 在 \(|w|<1\) 内解析且 \(F_{\zeta}(0)=\Xi_{\zeta}(-\tfrac12)\neq 0\)，可对任意 \(\varrho<1\) 在圆盘 \(|w|<\varrho\) 上应用 Jensen 公式。
结论 \eqref{eq:app-jensen-sum} 以及非负性是标准推论（参见 \cite{Duren1970TheoryHp}）。证毕。
\end{proof}

\begin{proposition}[幂替换的缺陷共变性：幂半群仅重参数化半径]\label{prop:app-jensen-defect-power-covariance}
令 \(F\) 在单位圆盘 \(\mathbb{D}\) 内解析且 \(F(0)\neq 0\)，并对 \(0<\varrho<1\) 定义 Jensen 缺陷
\[
D_F(\varrho):=\frac{1}{2\pi}\int_{0}^{2\pi}\log\bigl|F(\varrho e^{\mathrm{i}\theta})\bigr|\,\dd\theta-\log|F(0)|.
\]
对整数 \(m\ge 1\) 定义幂拉回 \(F^{(m)}(w):=F(w^m)\)。
则对任意 \(0<\varrho<1\) 有严格恒等式
\begin{equation}\label{eq:app-jensen-defect-power-covariance}
\boxed{\ D_{F^{(m)}}(\varrho)=D_F(\varrho^m)\ }.
\end{equation}
\leanverified{paper\_app\_jensen\_defect\_power\_covariance}
特别地，将 \(F\) 取为本附录的圆盘化完成化 \(F_{\zeta}\)（定义 \ref{def:app-jensen-defect}）时，上式表明：幂半群 \(w\mapsto w^m\) 在 Jensen 缺陷层面不引入新的连续自由度，而仅对半径变量施加确定性重参数化 \(\varrho\mapsto \varrho^m\)。
\end{proposition}
\begin{proof}
由定义，
\[
D_{F^{(m)}}(\varrho)
=\frac{1}{2\pi}\int_{0}^{2\pi}\log\bigl|F\!\bigl((\varrho e^{\mathrm{i}\theta})^m\bigr)\bigr|\,\dd\theta-\log|F(0)|
=\frac{1}{2\pi}\int_{0}^{2\pi}\log\bigl|F(\varrho^m e^{\mathrm{i}m\theta})\bigr|\,\dd\theta-\log|F(0)|.
\]
令 \(\phi:=m\theta\)，则 \(\dd\phi=m\,\dd\theta\)，从而
\[
D_{F^{(m)}}(\varrho)
=\frac{1}{2\pi m}\int_{0}^{2\pi m}\log\bigl|F(\varrho^m e^{\mathrm{i}\phi})\bigr|\,\dd\phi-\log|F(0)|.
\]
注意被积函数关于 \(\phi\) 具有 \(2\pi\)-周期性，故
\(
\int_{0}^{2\pi m}\log|F(\varrho^m e^{\mathrm{i}\phi})|\,\dd\phi
=m\int_{0}^{2\pi}\log|F(\varrho^m e^{\mathrm{i}\phi})|\,\dd\phi
\)。
代回即得 \(D_{F^{(m)}}(\varrho)=D_F(\varrho^m)\)，证毕。
\end{proof}

\begin{proposition}[单点盘内零点诱导的 Jensen 缺陷刚性下界]\label{prop:app-jensen-single-zero-lower-bound}
\leanverified{paper\_app\_jensen\_single\_zero\_lower\_bound}
若 \(F_{\zeta}\) 在开单位圆盘内存在零点 \(a\) 满足 \(|a|<1\)，则对任意 \(\varrho\in(|a|,1)\) 有
\begin{equation}\label{eq:app-jensen-single-zero-lower-bound}
\boxed{\;
D_{\zeta}(\varrho)\ge \log\!\Bigl(\frac{\varrho}{|a|}\Bigr),
\qquad
\liminf_{\varrho\uparrow 1}D_{\zeta}(\varrho)\ge \log\!\Bigl(\frac{1}{|a|}\Bigr)>0\; }.
\end{equation}
特别地，任何盘内零点都会在边界层尺度上留下不可消去的正缺陷下界。
\end{proposition}
\begin{proof}
取任意 \(\varrho\in(|a|,1)\)。
由 Jensen 求和式 \eqref{eq:app-jensen-sum}，对 \(|a|<\varrho\) 的任一零点项均有贡献
\(
\log\bigl(\varrho/|a|\bigr)
\)，从而
\(
D_{\zeta}(\varrho)=\sum_{|a_k|<\varrho}\log\bigl(\varrho/|a_k|\bigr)\ge \log\bigl(\varrho/|a|\bigr)
\)。
令 \(\varrho\uparrow 1\) 得到 \(\liminf\) 下界。证毕。
\end{proof}

\begin{corollary}[离临界线零点的圆盘可见阈值]\label{cor:app-jensen-visible-threshold-offcritical}
\leanverified{paper\_app\_jensen\_visible\_threshold\_offcritical}
若存在 \(\Xi_{\zeta}\) 的零点 \(\rho=\beta+\mathrm{i}\gamma\) 满足 \(\beta<\tfrac12\)，令 \(\delta=\tfrac12-\beta>0\) 并置
\(
w_{\rho}:=\mathcal{C}(\gamma+\mathrm{i}\delta)\in\mathbb{D}
\)（命题 \ref{prop:app-offcritical-radius-compression}）。
则对任意 \(\varrho\in(|w_{\rho}|,1)\) 有
\[
\boxed{\;
D_{\zeta}(\varrho)\ge \log\!\Bigl(\frac{\varrho}{|w_{\rho}|}\Bigr),
\qquad
|w_{\rho}|^{2}=\frac{\gamma^{2}+(\delta-1)^{2}}{\gamma^{2}+(\delta+1)^{2}}\; }.
\]
因此，给定半径分辨率 \(1-\varrho\)，只有满足 \(1-|w_{\rho}|=O(1-\varrho)\) 的离线零点才会对 \(D_{\zeta}(\varrho)\) 产生可分辨的正贡献；
并且在 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，由推论 \ref{cor:app-offcritical-radius-high-height} 得该条件等价于
\[
\delta=O\bigl((1-\varrho)\gamma^{2}\bigr).
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:app-offcritical-radius-compression}，\(F_{\zeta}\) 在盘内零点 \(w_{\rho}\) 处消失。
对任意 \(\varrho\in(|w_{\rho}|,1)\)，将命题 \ref{prop:app-jensen-single-zero-lower-bound} 应用于 \(a=w_{\rho}\) 即得所示下界。
最后一条等价式由推论 \ref{cor:app-offcritical-radius-high-height} 的渐近 \(1-|w_{\rho}|\sim 2\delta/\gamma^{2}\) 直接推出。证毕。
\end{proof}

\begin{corollary}[有限采样盲区：未越过阈值半径则离线零点不可判别]\label{cor:app-jensen-finite-sampling-blindspot}
\leanverified{paper\_app\_jensen\_finite\_sampling\_blindspot}
取任意有限半径采样集合 \(\{\varrho_j\}_{j=1}^{M}\subset(0,1)\) 并令 \(\varrho_{\max}:=\max_j\varrho_j\)。
则 Jensen 缺陷值族 \(\{D_{\zeta}(\varrho_j)\}_{j=1}^{M}\) 仅能反映圆盘内零点中满足 \(|a|<\varrho_{\max}\) 的那一部分；任何满足 \(|a|>\varrho_{\max}\) 的盘内零点在该有限采样上\emph{严格不可见}（其对 Jensen 求和式 \eqref{eq:app-jensen-sum} 的贡献在每个 \(\varrho_j\) 处皆为零）。
\par
特别地，若存在离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\)（\(\beta<\tfrac12\)）对应圆盘像 \(w_{\rho}=\mathcal{C}(\gamma+\mathrm{i}\delta)\in\mathbb{D}\)，且 \(\varrho_{\max}\le |w_{\rho}|\)，则仅凭有限采样 \(\{D_{\zeta}(\varrho_j)\}\) 无法给出排除该零点的正缺陷证书；要使该零点在采样上可被强制显现，其必要条件为
\[
\boxed{\ \varrho_{\max}>|w_{\rho}|\ }\qquad\text{等价地}\qquad
1-\varrho_{\max}<1-|w_{\rho}|
\le 1-|w_{\rho}|^{2}
=\frac{4\delta}{\gamma^{2}+(1+\delta)^{2}}.
\]
当 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，这一必要条件化为端点边界层尺度律 \(1-\varrho_{\max}\lesssim \delta/\gamma^{2}\)。
\end{corollary}

\begin{proof}
由 Jensen 求和式 \eqref{eq:app-jensen-sum}，对每个 \(\varrho\in(0,1)\)，缺陷 \(D_{\zeta}(\varrho)\) 只对满足 \(|a|<\varrho\) 的零点求和；故当 \(|a|>\varrho_{\max}\) 时，该零点对所有 \(\varrho_j\le\varrho_{\max}\) 的采样值均不贡献任何项。
因此有限采样只能“看到”半径 \(\varrho_{\max}\) 内的零点集合。
\par
对离线零点 \(\rho\)，若 \(\varrho_{\max}\le |w_{\rho}|\)，则对所有采样半径皆有 \(\varrho_j\le |w_{\rho}|\)，从而不满足推论 \ref{cor:app-jensen-visible-threshold-offcritical} 的可见阈值条件 \(\varrho>|w_{\rho}|\)，故无法由该采样族强制得到由该零点导致的正缺陷下界。
而若希望在采样上“触发”该零点的缺陷贡献，必要条件即为 \(\varrho_{\max}>|w_{\rho}|\)。
最后的几何化不等式由 \(|w_{\rho}|<1\Rightarrow 1-|w_{\rho}|\le 1-|w_{\rho}|^{2}\) 与推论 \ref{cor:app-offcritical-horocycle-busemann} 的闭式恒等式给出；高高度渐近同样由该推论（或推论 \ref{cor:app-offcritical-radius-high-height}）得到。证毕。
\end{proof}

\begin{corollary}[RH 的完全有限化判据：缺陷恒零]\label{cor:app-rh-iff-jensen-defect-zero}
\leanverified{paper\_app\_rh\_iff\_jensen\_defect\_zero}
RH 等价于
\[
\boxed{\ \forall\,\varrho\in(0,1),\quad D_{\zeta}(\varrho)=0\ }.
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:app-jensen-formula-defect}，对每个固定 \(\varrho\)，有 \(D_{\zeta}(\varrho)=0\iff F_{\zeta}\) 在 \(|w|<\varrho\) 内无零点。
令 \(\varrho\uparrow 1\) 即得 “对一切 \(\varrho\in(0,1)\) 缺陷恒零” 等价于 “\(F_{\zeta}\) 在 \(|w|<1\) 内无零点”，再与定理 \ref{thm:app-rh-iff-disk-zero-free} 合并即得。证毕。
\end{proof}

\begin{theorem}[趋近边界的缺陷趋零判据]\label{thm:app-rh-iff-jensen-defect-vanishing-sequence}
\leanverified{paper\_app\_rh\_iff\_jensen\_defect\_vanishing\_sequence}
下列命题等价：
\begin{enumerate}
  \item[(i)] RH 成立；
  \item[(ii)] 存在一列 \(\varrho_n\uparrow 1\) 使 \(D_{\zeta}(\varrho_n)\to 0\)。
\end{enumerate}
\end{theorem}
\begin{proof}
(i)\(\Rightarrow\)(ii) 由推论 \ref{cor:app-rh-iff-jensen-defect-zero} 立得：若 RH 成立，则 \(D_{\zeta}(\varrho)\equiv 0\)，从而任取 \(\varrho_n\uparrow 1\) 即有 \(D_{\zeta}(\varrho_n)\to 0\)。
\par
(ii)\(\Rightarrow\)(i) 反设 RH 不成立。
由定理 \ref{thm:app-rh-iff-disk-zero-free}，存在盘内零点 \(a\in\mathbb{D}\) 使 \(F_{\zeta}(a)=0\)。
则由命题 \ref{prop:app-jensen-single-zero-lower-bound} 得
\(
\liminf_{\varrho\uparrow 1}D_{\zeta}(\varrho)\ge \log(1/|a|)>0
\)，与 \(D_{\zeta}(\varrho_n)\to 0\) 矛盾。
故盘内零点不存在，进而由定理 \ref{thm:app-rh-iff-disk-zero-free} 得 RH 成立。证毕。
\end{proof}

\begin{corollary}[缺陷上界序列给出的零点排除证书]\label{cor:app-jensen-defect-upperbound-certificate}
\leanverified{paper\_app\_jensen\_defect\_upperbound\_certificate}
若存在 \(\varrho_n\uparrow 1\) 与 \(\varepsilon_n\downarrow 0\) 使 \(D_{\zeta}(\varrho_n)\le \varepsilon_n\)，则 \(F_{\zeta}\) 在 \(\mathbb{D}\) 内无零点，从而 RH 成立。
更定量地，对任意盘内零点 \(a\in\mathbb{D}\)，当 \(n\) 充分大使 \(|a|<\varrho_n\) 时必有
\[
|a|\ge \varrho_n e^{-\varepsilon_n}.
\]
\end{corollary}
\begin{proof}
固定任意盘内零点 \(a\in\mathbb{D}\)。
当 \(n\) 充分大使 \(|a|<\varrho_n\) 时，由 \eqref{eq:app-jensen-single-zero-lower-bound} 得
\(
\log(\varrho_n/|a|)\le D_{\zeta}(\varrho_n)\le \varepsilon_n
\)，从而 \(|a|\ge \varrho_n e^{-\varepsilon_n}\)。
令 \(n\to\infty\) 得 \(|a|\ge 1\)，与 \(a\in\mathbb{D}\) 矛盾；故盘内零点不存在。
由定理 \ref{thm:app-rh-iff-disk-zero-free} 或推论 \ref{cor:app-rh-iff-jensen-defect-zero} 即得 RH 成立。证毕。
\end{proof}

\subsubsection{导数律、可数化与端点局部化}\label{subsubsec:app-jensen-defect-new-cluster}

\paragraph{记号与接口（最小复述）}
为便于直接引用，本小节仅列出与后续结论相关的最小接口：
\begin{enumerate}
  \item \textbf{面积相位紧化与相位盒算术：}相位盒 \(\mathbb{U}:=(-\tfrac12,\tfrac12)\) 与紧化门 \(\upsilon(x)=\frac{1}{\pi}\arctan(x)\) 见 \eqref{eq:unit-circle-phase-upsilon}--\eqref{eq:unit-circle-phase-inverse}；
  沿 \(\upsilon\) 的结构搬运在 \(\mathbb{U}\) 上定义与 \((\RR,+,\cdot)\) 严格同构的运算 \((\boxplus,\boxtimes)\)（定义 \ref{def:unit-circle-phase-transport-arith}）。
  \item \textbf{圆盘化完成化：}Cayley 变换 \(\mathcal{C}:\mathbb{H}\to\mathbb{D}\) 及其逆映射见 \eqref{eq:app-cayley-disk}；
  圆盘化完成化函数 \(F_{\zeta}(w)=\Xi_{\zeta}(s(w))\) 由 \eqref{eq:app-s-of-w}--\eqref{eq:app-Fzeta-disk} 定义。
  \item \textbf{Jensen 缺陷：}对 \(\varrho\in(0,1)\)，缺陷 \(D_{\zeta}(\varrho)\) 由定义 \ref{def:app-jensen-defect} 给出，并满足 Jensen 求和式 \eqref{eq:app-jensen-sum}。
\end{enumerate}

\begin{proposition}[对数半径导数律：缺陷的导数即盘内零点计数]\label{prop:app-jensen-defect-log-derivative}
令
\[
N_{F_{\zeta}}(\varrho)
:=\#\bigl\{a:\ F_{\zeta}(a)=0,\ |a|<\varrho\bigr\}
\]
为 \(F_{\zeta}\) 在 \(|w|<\varrho\) 内的零点计数（按重数计）。
则对几乎处处的 \(\varrho\in(0,1)\) 有
\[
\frac{\dd}{\dd(\log\varrho)}D_{\zeta}(\varrho)=N_{F_{\zeta}}(\varrho),
\qquad
D_{\zeta}'(\varrho)=\frac{1}{\varrho}\,N_{F_{\zeta}}(\varrho).
\]
并且 \(D_{\zeta}\) 在 \((0,1)\) 上连续、非减，且若对某 \(\varrho_0\in(0,1)\) 有 \(D_{\zeta}(\varrho_0)=0\)，则对所有 \(0<\varrho\le \varrho_0\) 有 \(D_{\zeta}(\varrho)=0\)。
\leanverified{paper\_app\_jensen\_defect\_log\_derivative}
\end{proposition}
\begin{proof}
由 \eqref{eq:app-jensen-sum}，对每个固定 \(\varrho\in(0,1)\) 盘内零点数有限，故求和为有限和。
当 \(\varrho\) 不取任何 \(|a|\)（其中 \(a\) 遍历 \(F_{\zeta}\) 的盘内零点）时，集合 \(\{a:\ |a|<\varrho\}\) 在某个邻域内恒定，从而
\(
D_{\zeta}(\varrho)=\sum_{|a|<\varrho}\log(\varrho/|a|)
\)
在该邻域内对 \(\log\varrho\) 求导得到 \(\frac{\dd}{\dd(\log\varrho)}D_{\zeta}(\varrho)=N_{F_{\zeta}}(\varrho)\)。
当 \(\varrho\) 穿过某个 \(|a|\) 时，额外项 \(\log(\varrho/|a|)\) 在 \(\varrho\downarrow |a|\) 下趋于 \(0\)，从而 \(D_{\zeta}\) 连续。
非负性与单调性由 \eqref{eq:app-jensen-sum} 立得。
最后若 \(D_{\zeta}(\varrho_0)=0\)，则对 \(\varrho\le \varrho_0\) 有 \(0\le D_{\zeta}(\varrho)\le D_{\zeta}(\varrho_0)=0\)。证毕。
\end{proof}

\begin{corollary}[幂拉回的零点计数尺度律（径向测度公式）]\label{cor:app-jensen-zero-count-power-law}
沿用命题 \ref{prop:app-jensen-defect-power-covariance} 的记号。
令 \(N_F(\varrho)\) 表示 \(F\) 在 \(|w|<\varrho\) 内的零点计数（按重数计），并令 \(N_{F^{(m)}}\) 为 \(F^{(m)}\) 的相应计数函数。
则对任意整数 \(m\ge 1\) 与任意 \(0<\varrho<1\) 有
\begin{equation}\label{eq:app-zero-count-power-law}
\boxed{\ N_{F^{(m)}}(\varrho)=m\,N_F(\varrho^m)\ }.
\end{equation}
等价地，若以零点模长计数测度
\[
\mu_F:=\sum_{a:\,F(a)=0}\mathrm{mult}(a)\,\delta_{|a|}
\]
表示盘内零点的径向分布，则
\[
\mu_{F^{(m)}}=m\,(\varrho\mapsto \varrho^{1/m})_{\ast}\mu_F.
\]
\end{corollary}
\begin{proof}
由假设 \(F(0)\neq 0\)，\(F\) 在 \(w=0\) 处无零点。
若 \(a\neq 0\) 为 \(F\) 的零点且重数为 \(r\)，则 \(F^{(m)}(w)=F(w^m)\) 在方程 \(w^m=a\) 的每个根处同样有重数 \(r\) 的零点。
这些根共有 \(m\) 个，且模长均为 \(|a|^{1/m}\)。
因此，对给定 \(0<\varrho<1\)，盘内条件 \(|w|<\varrho\) 对应于 \(|a|<\varrho^m\)，并且每个满足 \(|a|<\varrho^m\) 的零点贡献 \(m\) 个盘内零点（按重数计），从而得到 \eqref{eq:app-zero-count-power-law}。
测度表述由同一“模长取 \(m\) 次根且权重乘 \(m\)”的对应关系直接推出。证毕。
\end{proof}
\leanverified{paper\_app\_jensen\_zero\_count\_power\_law}

\begin{corollary}[折角结构：盘内零点在缺陷上生成可定位跳变原子]\label{cor:app-jensen-defect-kink-structure}
\leanverified{paper\_app\_jensen\_defect\_kink\_structure}
令 \(r\in(0,1)\) 并记
\[
m(r):=\sum_{a:\ F_{\zeta}(a)=0,\ |a|=r}\mathrm{mult}(a)\in\ZZ_{\ge 0}
\]
为模长恰等于 \(r\) 的盘内零点的总重数（有限和）。
则 \(D_{\zeta}(\varrho)\) 作为 \(\log\varrho\) 的函数在 \(\varrho=r\) 处存在左、右导数，并满足斜率跳变律
\[
\frac{\dd}{\dd(\log\varrho)}D_{\zeta}(\varrho)\Big|_{\varrho=r+0}
-\frac{\dd}{\dd(\log\varrho)}D_{\zeta}(\varrho)\Big|_{\varrho=r-0}
=m(r).
\]
等价地，零点计数函数在 \(\varrho=r\) 处发生跳变
\[
N_{F_{\zeta}}(r+0)-N_{F_{\zeta}}(r-0)=m(r).
\]
因此 \(D_{\zeta}\) 可视为“零点半径分布”的累计函数：每一组模长为 \(r\) 的盘内零点在 \(\varrho=r\) 处强制生成一个不可消去的折角（kink）。
\end{corollary}
\begin{proof}
由 Jensen 求和式 \eqref{eq:app-jensen-sum}，对任意固定 \(r\in(0,1)\)，在 \(r\) 的某个去心邻域内除去有限多个阈值半径 \(|a|\) 外，
零点集合 \(\{a:\ |a|<\varrho\}\) 恒定，从而
\(
\frac{\dd}{\dd(\log\varrho)}D_{\zeta}(\varrho)=N_{F_{\zeta}}(\varrho)
\)
（命题 \ref{prop:app-jensen-defect-log-derivative}）。
当 \(\varrho\) 从 \(r-0\) 增至 \(r+0\) 时，恰有模长为 \(r\) 的零点项从“未计入”变为“计入”，并各贡献一个单位斜率，因此斜率跳变等于其总重数 \(m(r)\)。
证毕。
\end{proof}

\begin{theorem}[任意趋近列的可数化判据]\label{thm:app-jensen-countable-criterion}
\leanverified{paper\_app\_jensen\_countable\_criterion}
设 \((\varrho_n)_{n\ge 1}\subset(0,1)\) 单调递增且 \(\varrho_n\uparrow 1\)。
则下列命题等价：
\begin{enumerate}
  \item[(a)] RH 成立；
  \item[(b)] 对所有 \(n\ge 1\)，有 \(D_{\zeta}(\varrho_n)=0\)。
\end{enumerate}
\end{theorem}
\begin{proof}
(a)\(\Rightarrow\)(b) 由推论 \ref{cor:app-rh-iff-jensen-defect-zero} 立得。
\par
(b)\(\Rightarrow\)(a) 取任意 \(\varrho\in(0,1)\)。
由 \(\varrho_n\uparrow 1\) 可取 \(n\) 使 \(\varrho\le \varrho_n\)。
由命题 \ref{prop:app-jensen-defect-log-derivative} 的单调性，\(0\le D_{\zeta}(\varrho)\le D_{\zeta}(\varrho_n)=0\)，故 \(D_{\zeta}(\varrho)=0\)。
再由推论 \ref{cor:app-rh-iff-jensen-defect-zero} 得 RH。证毕。
\end{proof}

\begin{theorem}[离临界线信息的折角定位与视界深度]\label{thm:app-jensen-offcritical-kink-localization}
\leanverified{paper\_app\_jensen\_offcritical\_kink\_localization}
假设存在 \(\Xi_{\zeta}\) 的零点 \(\rho=\beta+\mathrm{i}\gamma\) 满足 \(0<\beta<\tfrac12\)，并记 \(\delta:=\tfrac12-\beta>0\)，
以及 \(w_{\rho}:=\mathcal{C}(\gamma+\mathrm{i}\delta)\in\mathbb{D}\)（命题 \ref{prop:app-offcritical-radius-compression}）。
令 \(r_{\rho}:=|w_{\rho}|\) 且 \(m_{\rho}:=\mathrm{mult}(w_{\rho})\) 为对应圆盘零点的重数。
则：
\begin{enumerate}
  \item \(F_{\zeta}(w_{\rho})=0\)，从而 \(r_{\rho}\) 是 \(D_{\zeta}\) 的折角阈值半径；
  \item 存在 \(\varepsilon_0>0\) 使得对所有 \(0<\varepsilon<\varepsilon_0\) 有
  \[
  N_{F_{\zeta}}(r_{\rho}+\varepsilon)=N_{F_{\zeta}}(r_{\rho}-\varepsilon)+m_{\rho},
  \]
  等价地，\(D_{\zeta}(\varrho)\) 作为 \(\log\varrho\) 的函数在 \(\varrho=r_{\rho}\) 处出现折角，其斜率跳变为 \(m_{\rho}\)（推论 \ref{cor:app-jensen-defect-kink-structure}）。
  \item 定义视界深度（近边界缺口）
  \[
  h(\rho):=1-|w_{\rho}|^{2},
  \]
  则由 \eqref{eq:app-offcritical-radius-closed} 有完全闭式
  \[
  \boxed{\ h(\rho)=\frac{4\delta}{\gamma^{2}+(\delta+1)^{2}}\ }.
  \]
  特别地，当 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，\(h(\rho)\sim 4\delta/\gamma^{2}\)，从而离临界线信息在圆盘接口中被压入边界层。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)} 由命题 \ref{prop:app-offcritical-radius-compression}。
\par
\textbf{(2)} 由 \(\varrho<1\) 时盘内零点有限，可取 \(\varepsilon_0>0\) 使得在环域
\(
r_{\rho}-\varepsilon_0<|w|<r_{\rho}+\varepsilon_0
\)
内不存在除模长 \(r_{\rho}\) 之外的其它零点模长。
于是当 \(\varrho\) 穿过 \(r_{\rho}\) 时，零点计数函数的跳变恰为该模长层的总重数 \(m_{\rho}\)，从而得到所示等式；折角结论由推论 \ref{cor:app-jensen-defect-kink-structure} 等价推出。
\par
\textbf{(3)} 由 \eqref{eq:app-offcritical-radius-closed} 直接化简得
\(
1-|w_{\rho}|^{2}=\frac{4\delta}{\gamma^{2}+(\delta+1)^{2}}
\)。
当 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时分母主项为 \(\gamma^{2}\)，得渐近 \(h(\rho)\sim 4\delta/\gamma^{2}\)。
证毕。
\end{proof}

\begin{corollary}[采样半径截断下的不可检出边界层]\label{cor:app-jensen-kink-undetectable-threshold}
给定有限采样半径集合 \(\{\varrho_1,\dots,\varrho_M\}\subset(0,1)\)，并记
\(
\varrho_{\max}:=\max_{1\le j\le M}\varrho_j
\)
与 \(\eta:=1-\varrho_{\max}>0\)。
若存在离临界线零点 \(\rho\) 使得 \(1-|w_{\rho}|<\eta\)（例如 \(h(\rho)\ll \eta\)），则其折角阈值半径满足 \(r_{\rho}=|w_{\rho}|>\varrho_j\) 对所有 \(j\)，从而仅凭该有限网格上的缺陷采样 \(\{D_{\zeta}(\varrho_j)\}_{j=1}^{M}\) 无法排除该折角原子。
\end{corollary}
\begin{proof}
由 \(r_{\rho}>\varrho_j\)（对所有 \(j\)）可知零点 \(w_{\rho}\) 从未进入任一采样圆盘 \(|w|<\varrho_j\)，因此 Jensen 求和式 \eqref{eq:app-jensen-sum} 在这些采样点上不含该零点项；折角位置不可由该网格定位，故无法据此排除。证毕。
\end{proof}

\begin{proposition}[由缺陷导出的零点排斥子盘]\label{prop:app-jensen-zero-repulsion-subdisk}
对任意 \(0<\varrho<1\) 定义
\[
r_{\ast}(\varrho):=\varrho\,\exp\!\bigl(-D_{\zeta}(\varrho)\bigr)\in(0,\varrho].
\]
则 \(F_{\zeta}\) 在 \(|w|<r_{\ast}(\varrho)\) 内无零点。
特别地，若 \(D_{\zeta}(\varrho)\le \varepsilon\)，则排斥半径满足 \(r_{\ast}(\varrho)\ge \varrho e^{-\varepsilon}\)。
\end{proposition}
\begin{proof}
反设存在零点 \(a\) 满足 \(|a|<r_{\ast}(\varrho)\le \varrho\)。
则
\[
\log\!\Bigl(\frac{\varrho}{|a|}\Bigr)\ >\ \log\!\Bigl(\frac{\varrho}{r_{\ast}(\varrho)}\Bigr)\ =\ D_{\zeta}(\varrho).
\]
但 \eqref{eq:app-jensen-sum} 表明 \(D_{\zeta}(\varrho)\) 是若干非负项之和，且其中包含项 \(\log(\varrho/|a|)\)，矛盾。证毕。
\end{proof}

\begin{corollary}[排斥半径趋近 \(1\) 的充分条件]\label{cor:app-jensen-repulsion-radius-to-one}
若存在序列 \(\varrho_n\uparrow 1\) 使得 \(r_{\ast}(\varrho_n)\uparrow 1\)，则 \(F_{\zeta}\) 在 \(|w|<1\) 内无零点，从而 RH 成立。
\end{corollary}
\begin{proof}
若存在盘内零点 \(a\) 满足 \(|a|<1\)，则可取 \(n\) 使 \(|a|<r_{\ast}(\varrho_n)\)，这与命题 \ref{prop:app-jensen-zero-repulsion-subdisk} 矛盾。
故 \(F_{\zeta}\) 在 \(\mathbb{D}\) 内无零点；由定理 \ref{thm:app-rh-iff-disk-zero-free} 得 RH。证毕。
\end{proof}

\begin{proposition}[\texorpdfstring{$|w|=\varrho$}{|w|=varrho} 的 Cayley 逆像为一条显式圆]\label{prop:app-cayley-preimage-modulus-circle}
设 \(0<\varrho<1\)。
令 \(t=x+\mathrm{i}y\in\mathbb{H}\) 且 \(w=\mathcal{C}(t)\)。
则 \(|w|=\varrho\) 等价于
\[
x^2+\Bigl(y-\frac{1+\varrho^2}{1-\varrho^2}\Bigr)^2=\Bigl(\frac{2\varrho}{1-\varrho^2}\Bigr)^2.
\]
该圆与虚轴交于两点
\[
y_{\pm}(\varrho)=\frac{1\pm\varrho}{1\mp\varrho}.
\]
并且当 \(\varrho\uparrow 1\) 时，\(y_{-}(\varrho)\downarrow 0\)、\(y_{+}(\varrho)\uparrow\infty\)。
\end{proposition}
\begin{proof}
由 \eqref{eq:app-cayley-disk} 及 \(|\mathcal{C}(t)|^2=\frac{|1+\mathrm{i}t|^2}{|1-\mathrm{i}t|^2}\)，对 \(t=x+\mathrm{i}y\) 计算
\[
|1+\mathrm{i}t|^2=x^2+(1-y)^2,
\qquad
|1-\mathrm{i}t|^2=x^2+(1+y)^2.
\]
令其比值等于 \(\varrho^2\) 并配方即得圆方程。
取 \(x=0\) 得交点公式；极限由显式表达直接推出。证毕。
\end{proof}

\begin{proposition}[缺陷的上半平面圆平均表示与显式参数化]\label{prop:app-jensen-defect-upperhalf-circle-average}
设 \(0<\varrho<1\)。
记
\[
t(\varrho,\theta):=\mathcal{C}^{-1}(\varrho e^{\mathrm{i}\theta})\in\mathbb{H}.
\]
则
\[
D_{\zeta}(\varrho)
=\frac{1}{2\pi}\int_{0}^{2\pi}\log\Bigl|\Xi_{\zeta}\!\Bigl(\tfrac12+\mathrm{i}\,t(\varrho,\theta)\Bigr)\Bigr|\,\dd\theta
-\log\bigl|\Xi_{\zeta}(-\tfrac12)\bigr|.
\]
并且 \(t(\varrho,\theta)\) 具有显式分解
\begin{equation}\label{eq:app-t-rho-theta-explicit}
\boxed{\;
t(\varrho,\theta)
=\frac{2\varrho\sin\theta}{1+\varrho^2+2\varrho\cos\theta}
\;+\;
\mathrm{i}\,\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta}\; }.
\end{equation}
\end{proposition}
\begin{proof}
由 \eqref{eq:app-jensen-defect}、\eqref{eq:app-Fzeta-disk} 与 \eqref{eq:app-s-of-w}，对 \(w=\varrho e^{\mathrm{i}\theta}\) 有
\(
\log|F_{\zeta}(w)|=\log|\Xi_{\zeta}(\tfrac12+\mathrm{i}\mathcal{C}^{-1}(w))|
\)，且 \(F_{\zeta}(0)=\Xi_{\zeta}(-\tfrac12)\)，故得到第一式。
\par
显式分解由 \eqref{eq:app-cayley-disk} 的 \(t=-\mathrm{i}\frac{w-1}{w+1}\) 在 \(w=\varrho e^{\mathrm{i}\theta}\) 上作代数化简得到。证毕。
\end{proof}

\begin{corollary}[端点局部化：\texorpdfstring{$\varrho\uparrow 1$}{varrho->1} 的高虚部贡献集中于 \texorpdfstring{$\theta\approx\pi$}{theta~pi}]\label{cor:app-jensen-endpoint-localization}
由 \eqref{eq:app-t-rho-theta-explicit} 得
\[
\Im\,t(\varrho,\theta)=\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta},
\]
其在 \(\theta=\pi\) 处取最大值
\[
\Im\,t(\varrho,\pi)=\frac{1+\varrho}{1-\varrho}=y_{+}(\varrho)
\]
（命题 \ref{prop:app-cayley-preimage-modulus-circle}）。
因此当 \(\varrho\uparrow 1\) 时，圆周平均的高虚部采样被迫集中于端点 \(w=\varrho e^{\mathrm{i}\pi}\approx-1\) 的窄角邻域；该端点邻域成为控制 \(D_{\zeta}(\varrho)\) 的主要刚性来源。
\end{corollary}
\begin{proof}
由 \eqref{eq:app-t-rho-theta-explicit} 直接计算。
当 \(\theta=\pi\) 时分母为 \((1-\varrho)^2\)，得最大值并随 \(\varrho\uparrow 1\) 发散。证毕。
\end{proof}

\begin{remark}[端点分解的优先奇通道原则（与 Chebyshev--Adams 桥对齐）]\label{rem:app-jensen-odd-channel-priority}
推论 \ref{cor:app-jensen-endpoint-localization} 表明：\(\varrho\uparrow 1\) 的主要困难局部化于端点 \(w\approx-1\)。
在完成化坐标 \(S=\mathfrak{r}+\mathfrak{r}^{-1}\) 下，幂半群 \(w\mapsto w^m\) 对应 \(S\mapsto C_m(S)\)（\S\ref{subsec:unit-disk-witt-chebyshev}）。
由端点导数律（命题 \ref{prop:chebyshev-endpoint-derivative-splitting}）可知：奇 \(m\) 通道在 \(S=0\) 附近保留线性主尺度，而偶 \(m\) 通道的一阶喷射退化，贡献仅进入高阶残差。
因此任何针对端点邻域的多尺度剥离在结构上必然以奇通道为主项、偶通道为高阶修正。
\end{remark}

\paragraph{研究目标（与 RH 的严格契约对齐）}\label{par:app-jensen-research-target-chain}
综合以上接口与结论，单位圆面积相位算术--圆盘化--Jensen 缺陷路线可被组织为如下闭合契约链：
\begin{enumerate}
  \item \textbf{可数化：}任选 \(\varrho_n\uparrow 1\)，由定理 \ref{thm:app-jensen-countable-criterion} 把 RH 等价压缩为可数族约束 \(D_{\zeta}(\varrho_n)=0\)。
  \item \textbf{几何显式化：}对每个 \(\varrho\)，由命题 \ref{prop:app-jensen-defect-upperhalf-circle-average} 将 \(D_{\zeta}(\varrho)\) 解释为 \(\Xi_{\zeta}\) 在上半平面一条显式圆周上的对数模长平均；该圆的几何由命题 \ref{prop:app-cayley-preimage-modulus-circle} 给出。
  \item \textbf{端点局部化与奇偶通道：}由推论 \ref{cor:app-jensen-endpoint-localization}，当 \(\varrho\uparrow 1\) 时主要困难局部化于端点 \(w\approx-1\) 的窄角邻域；端点喷射的奇偶分裂（命题 \ref{prop:chebyshev-endpoint-derivative-splitting}）为端点多尺度剥离提供结构分解（注 \ref{rem:app-jensen-odd-channel-priority}）。
  \item \textbf{证书化充分条件：}若能在某一具体 \(\varrho_n\uparrow 1\) 上构造可审计上界链，使零点排斥半径 \(r_{\ast}(\varrho_n)=\varrho_n e^{-D_{\zeta}(\varrho_n)}\uparrow 1\)，则由推论 \ref{cor:app-jensen-repulsion-radius-to-one} 推出盘内无零点，从而得到 RH。
\end{enumerate}

\paragraph{视界/毛发同构语义的纯数学闭合链}\label{par:app-jensen-horizon-hair-closed-chain}
在“圆盘化完成化 \(F_{\zeta}\)—Jensen 缺陷 \(D_{\zeta}\)—端点紧化 \(w\to-1\)”的闭合接口内，上述结果可进一步组织为一条以\emph{可数审计、折角定位与离散证书}为核心的四段链：
\begin{enumerate}
  \item \textbf{可数审计：}RH 等价于对任意趋近列 \(\varrho_n\uparrow 1\) 的可数约束 \(D_{\zeta}(\varrho_n)=0\)（定理 \ref{thm:app-jensen-countable-criterion}）。
  \item \textbf{折角定位：}盘内零点在缺陷函数上生成可定位折角原子（推论 \ref{cor:app-jensen-defect-kink-structure}）；
  若存在离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\)（\(\beta<\tfrac12\)），则其圆盘像 \(w_{\rho}\) 给出一条近边界折角阈值半径 \(r_{\rho}=|w_{\rho}|\)，并以视界深度
  \(h(\rho)=1-|w_{\rho}|^{2}=\frac{4\delta}{\gamma^{2}+(\delta+1)^{2}}\)（定理 \ref{thm:app-jensen-offcritical-kink-localization}）
  定量刻画其在边界层中的压缩尺度；缺陷上界进一步给出离线偏移的显式上界（定理 \ref{thm:app-jensen-defect-controls-offcritical-shift}）。
  \item \textbf{离散证书（实现层）：}当对象被提升为有限样本可复核的 realization/transfer 表达时，Hankel-正规形提供最小实现维数与分母/赋值的离散证书（定义 \ref{def:pom-hankel-normal-form}、注 \ref{rem:pom-hankel-denominator-stiffness} 及注 \ref{rem:pom-hankel-stiffness-p5}）；
  该证书在 \(p\)-进侧表现为刚性指数门槛，从而把“可逆性所需信息”外置为可审计的不变量。
  \item \textbf{对数深度同余链：}在二幂塔上，Dwork 型同余在不变量坐标 \(t=u+u^{-1}\) 下闭合为 Chebyshev--Dwork 同余链，并给出对数深度的逐级可复现审计（定理 \ref{thm:chebyshev-dwork-congruence-chain} 与推论 \ref{cor:chebyshev-dwork-matrixfree}）。
\end{enumerate}
