\subsection{相位盒上的算术：结构搬运（提升—运算—读出）}\label{subsec:unit-circle-phase-arithmetic}
记相位盒
\[
\mathbb{U}:=\Bigl(-\tfrac12,\tfrac12\Bigr).
\]
把 \(\upsilon:\RR\to\mathbb{U}\) 视作提升门，把 \(\upsilon^{-1}(u)=\tan(\pi u)\) 视作读出门，则可按结构搬运在 \(\mathbb{U}\) 上定义与实数算术严格同构的运算。

\begin{definition}[相位盒算术（结构搬运）]\label{def:unit-circle-phase-transport-arith}
对任意 \(u,v\in\mathbb{U}\)，定义
\[
u\boxplus v
:=\upsilon\!\bigl(\upsilon^{-1}(u)+\upsilon^{-1}(v)\bigr),
\qquad
u\boxtimes v
:=\upsilon\!\bigl(\upsilon^{-1}(u)\cdot\upsilon^{-1}(v)\bigr).
\]
等价地，
\[
u\boxplus v=\frac{1}{\pi}\arctan\!\Bigl(\tan(\pi u)+\tan(\pi v)\Bigr),
\qquad
u\boxtimes v=\frac{1}{\pi}\arctan\!\Bigl(\tan(\pi u)\tan(\pi v)\Bigr).
\]
并记零元与幺元
\[
0_{\mathbb{U}}:=\upsilon(0)=0,
\qquad
1_{\mathbb{U}}:=\upsilon(1)=\frac{1}{\pi}\arctan(1)=\frac14.
\]
\end{definition}

\begin{theorem}[与实数域严格同构]\label{thm:unit-circle-phase-field-iso}
映射
\(
\upsilon:(\RR,+,\cdot)\to(\mathbb{U},\boxplus,\boxtimes)
\)
为域同构；亦即对任意 \(x,y\in\RR\) 有
\[
\upsilon(x+y)=\upsilon(x)\boxplus\upsilon(y),
\qquad
\upsilon(xy)=\upsilon(x)\boxtimes\upsilon(y).
\]
特别地，\((\mathbb{U},\boxplus,\boxtimes)\) 满足交换律、结合律、分配律与可逆性（除 \(0_{\mathbb{U}}\) 外）。
\end{theorem}
\begin{proof}
由定义 \ref{def:unit-circle-phase-transport-arith}，\(\boxplus,\boxtimes\) 即为沿双射 \(\upsilon\) 的结构搬运；同构恒等式由代回 \(\upsilon^{-1}\circ \upsilon=\mathrm{id}_\RR\) 直接得到。其余域结构性质由 \((\RR,+,\cdot)\) 经同构传递。证毕。
\end{proof}

\begin{remark}[有界表示与边界承载发散]\label{rem:unit-circle-phase-bounded}
相位表示始终落在有界盒 \(\mathbb{U}\) 内；“数值发散”被等价搬运为相位趋近边界：
\[
x\to\pm\infty\qquad\Longleftrightarrow\qquad \upsilon(x)\to\pm\tfrac12
\qquad\Longleftrightarrow\qquad \vartheta(x)\to\pm\pi.
\]
因此该坐标门可作为“溢出规避”的无参表示层：主尺度不再以数值大小呈现，而以“边界余量”呈现。
\end{remark}

\begin{remark}[与第 \ref{sec:emergent-arithmetic} 节的运算定义性同构同型]\label{rem:unit-circle-phase-transport-analogy}
第 \ref{sec:emergent-arithmetic} 节把稳定地址上的算术固定为“逐位叠加后经 \(\Fold_\infty\) 归一化”的对象层实现；对加法，这与
\[
\mathrm{Val}^{-1}\!\bigl(\mathrm{Val}(c)+\mathrm{Val}(d)\bigr)
\]
等价（见定义 \ref{def:stable-add} 与定理 \ref{thm:stable-add-value-consistency}），而乘法仍由
\[
\mathrm{Val}^{-1}\!\bigl(\mathrm{Val}(c)\mathrm{Val}(d)\bigr)
\]
给出（见定义 \ref{def:stable-mul}）。
此处仅将“值映射”替换为 \(\upsilon^{-1}=\tan(\pi\cdot)\) 并将值域替换为 \(\RR\)，从而得到一个连续版本的同型结构搬运。
\end{remark}
