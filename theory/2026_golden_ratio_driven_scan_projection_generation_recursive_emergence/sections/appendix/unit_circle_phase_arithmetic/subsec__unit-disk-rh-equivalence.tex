\subsection{圆盘完成化与 RH 的圆盘零点禁入等价：把 \texorpdfstring{$t\to\infty$}{t->infty} 压到边界点 \texorpdfstring{$w=-1$}{w=-1}}\label{subsec:unit-disk-rh-equivalence}
本小节给出一个与正文 unitary slice 口径互补的\textbf{无参圆盘化重述}：把完成化 \(\Xi_{\zeta}\)（其零点与 \(\zeta\) 的非平凡零点一致）通过 Cayley 变换搬运为单位圆盘内的解析函数，从而把 RH 等价改写为“盘内零点禁入”与“边界有限圆周平均缺陷为零”两类判据。
与之对照，正文 \S\ref{subsec:xi-from-self-dual} 通过 Dirichlet 截面 \(u_L(s)=L^{2s-1}\) 将临界线识别为参数侧的单位圆切片（注 \ref{rem:xi-critical-line-unit-circle}）；而此处进一步把临界线参数 \(t\in\RR\) 自身紧化为单位圆上一条单弧，使 \(\pm\infty\) 合并为同一边界点 \(w=-1\)。

\paragraph{自对偶一变量 \texorpdfstring{$\Xi$}{Xi} 的非周期相位化：临界线的“单点无穷”重写}
令 \(\Xi_{\Delta,L}(s)=\Delta(L^{-s},L^{2s-1})\) 如 \S\ref{subsec:xi-from-self-dual}。
在完成化条件下，临界线实值性
\(
\Xi_{\Delta,L}(\tfrac12+\mathrm{i}t)\in\RR
\)
由命题 \ref{prop:xi-delta-L-critical-line-real} 给出。
据此可引入一个与 \eqref{eq:dirichlet-cayley-phase-axis} 一致的\textbf{非周期临界线参数化}：用 Cayley 相位轴把 \(t\in\RR\) 紧化为 \(\TT\setminus\{-1\}\) 上的单位圆参数，并把 \(t\to\pm\infty\) 合并为端点 \(-1\)。

\begin{definition}[非周期临界线相位化]\label{def:xi-nonperiodic-phase}
定义
\begin{equation}\label{eq:xi-nonperiodic-phase}
\boxed{\;
\widetilde{\Xi}_{\Delta,L}(u_{\mathrm C})
:=\Xi_{\Delta,L}\!\Bigl(\frac12+\mathrm{i}\,t(u_{\mathrm C})\Bigr),
\qquad
t(u_{\mathrm C})=-\mathrm{i}\,\frac{u_{\mathrm C}-1}{u_{\mathrm C}+1}
\qquad
(u_{\mathrm C}\in\TT\setminus\{-1\})\; }.
\end{equation}
\end{definition}

\begin{proposition}[单位圆对称性：函数方程在 \texorpdfstring{$\TT$}{T} 上的重写]\label{prop:xi-nonperiodic-phase-symmetry}
\leanverified{paper\_xi\_nonperiodic\_phase\_symmetry}
在命题 \ref{prop:xi-delta-L-critical-line-real} 的条件下，\(\widetilde{\Xi}_{\Delta,L}\) 是定义在 \(\TT\setminus\{-1\}\) 上的实值函数。
并且满足纯单位圆对称性
\begin{equation}\label{eq:xi-unit-circle-inversion-symmetry}
\boxed{\;
\widetilde{\Xi}_{\Delta,L}(u_{\mathrm C})
=\widetilde{\Xi}_{\Delta,L}(1/u_{\mathrm C})\; }.
\end{equation}
其中端点极限 \(u_{\mathrm C}\to-1\) 对应 \(t(u_{\mathrm C})\to\pm\infty\)，从而临界线的无穷高度被严格重写为单位圆上的单点边界层。
\end{proposition}
\begin{proof}
当 \(u_{\mathrm C}\in\TT\setminus\{-1\}\) 时，\(t(u_{\mathrm C})\in\RR\)，故由命题 \ref{prop:xi-delta-L-critical-line-real} 得 \(\widetilde{\Xi}_{\Delta,L}(u_{\mathrm C})\in\RR\)。
又由函数方程 \(\Xi_{\Delta,L}(s)=\Xi_{\Delta,L}(1-s)\)（命题 \ref{prop:xi-delta-L-functional-equation}）在临界线上的等价式
\(
\Xi_{\Delta,L}(\tfrac12+\mathrm{i}t)=\Xi_{\Delta,L}(\tfrac12-\mathrm{i}t)
\)
可知 \(\Xi_{\Delta,L}\) 在临界线上为偶函数。
而 Cayley 反变换满足 \(t(1/u_{\mathrm C})=-t(u_{\mathrm C})\)（例如取 \(u_{\mathrm C}=e^{\mathrm{i}\theta}\) 时 \(t(u_{\mathrm C})=\tan(\theta/2)\)），
于是
\[
\widetilde{\Xi}_{\Delta,L}(1/u_{\mathrm C})
=\Xi_{\Delta,L}\!\Bigl(\tfrac12+\mathrm{i}\,t(1/u_{\mathrm C})\Bigr)
=\Xi_{\Delta,L}\!\Bigl(\tfrac12-\mathrm{i}\,t(u_{\mathrm C})\Bigr)
=\widetilde{\Xi}_{\Delta,L}(u_{\mathrm C}),
\]
得到 \eqref{eq:xi-unit-circle-inversion-symmetry}。
端点对应关系由 \(\mathcal{C}(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\to-1\ (t\to\pm\infty)\)（命题 \ref{prop:app-cayley-upperhalf-disk}）立即得到。证毕。
\end{proof}

\paragraph{完成化对象}
记
\begin{equation}\label{eq:app-xi-classical}
\Xi_{\zeta}(s):=\frac12\,s(s-1)\,\pi^{-s/2}\Gamma\!\left(\frac{s}{2}\right)\zeta(s),
\end{equation}
则 \(\Xi_{\zeta}\) 为整函数且满足严格函数方程 \(\Xi_{\zeta}(s)=\Xi_{\zeta}(1-s)\)，其零点集恰为黎曼 \(\zeta\) 的非平凡零点（含重数）（参见 \cite{Apostol1976}）。

\paragraph{上半平面 \texorpdfstring{$\leftrightarrow$}{<->} 圆盘：无参 Cayley 紧化}
令上半平面
\[
\mathbb{H}:=\{t\in\CC:\Im(t)>0\},
\qquad
\mathbb{D}:=\{w\in\CC:|w|<1\}.
\]
定义 Möbius 变换（与 \eqref{eq:unit-circle-cayley-embed} 在实轴上一致）
\begin{equation}\label{eq:app-cayley-disk}
\boxed{\;
w=\mathcal{C}(t):=\frac{1+\mathrm{i}t}{1-\mathrm{i}t},
\qquad
t=\mathcal{C}^{-1}(w):=-\mathrm{i}\,\frac{w-1}{w+1}\; }.
\end{equation}

\begin{proposition}[Cayley 紧化的圆盘几何]\label{prop:app-cayley-upperhalf-disk}
\leanverified{paper\_app\_cayley\_upperhalf\_disk}
\(\mathcal{C}\) 给出双全纯同构 \(\mathcal{C}:\mathbb{H}\to\mathbb{D}\)，其逆映射为 \eqref{eq:app-cayley-disk} 的 \(\mathcal{C}^{-1}\)。
并且当 \(t\in\RR\) 时有 \(|\mathcal{C}(t)|=1\)，当 \(t\to\pm\infty\) 时有 \(\mathcal{C}(t)\to-1\)。
\end{proposition}
\begin{proof}
直接计算
\[
|\mathcal{C}(t)|^2
=\frac{|1+\mathrm{i}t|^2}{|1-\mathrm{i}t|^2}
=\frac{(1+\mathrm{i}t)(1-\mathrm{i}\overline t)}{(1-\mathrm{i}t)(1+\mathrm{i}\overline t)}
=\frac{1+|t|^2-2\Im(t)}{1+|t|^2+2\Im(t)}.
\]
故 \(\Im(t)>0\Rightarrow |\mathcal{C}(t)|<1\)，\(\Im(t)=0\Rightarrow |\mathcal{C}(t)|=1\)。
逆映射由代数求解得到，并可验证其把 \(|w|<1\) 映到 \(\Im(t)>0\)。
最后，当 \(t\in\RR\) 时 \(|\mathcal{C}(t)|=1\)，而 \(t\to\pm\infty\) 时 \(\mathcal{C}(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\to-1\)。证毕。
\end{proof}

\paragraph{圆盘化完成化函数}
令
\begin{equation}\label{eq:app-s-of-w}
\boxed{\;
s(w):=\frac12+\mathrm{i}\,\mathcal{C}^{-1}(w)
=\frac12+\frac{w-1}{w+1}\; },
\qquad (|w|<1),
\end{equation}
并定义圆盘内解析函数
\begin{equation}\label{eq:app-Fzeta-disk}
\boxed{\;
F_{\zeta}(w):=\Xi_{\zeta}\!\bigl(s(w)\bigr)\qquad(|w|<1)\; }.
\end{equation}
由于 \(\Xi_{\zeta}\) 整且 \(s(w)\) 在 \(|w|<1\) 内全纯（唯一极点 \(w=-1\) 位于边界），故 \(F_{\zeta}\) 在开单位圆盘内解析。
当 \(w=e^{\mathrm{i}\theta}\in\TT\setminus\{-1\}\) 时由 \eqref{eq:app-cayley-disk} 得 \(\mathcal{C}^{-1}(e^{\mathrm{i}\theta})=\tan(\theta/2)\in\RR\)，从而
\begin{equation}\label{eq:app-Fzeta-boundary}
F_{\zeta}(e^{\mathrm{i}\theta})=\Xi_{\zeta}\!\left(\tfrac12+\mathrm{i}\tan(\theta/2)\right),
\end{equation}
即把“临界线上取值”改写为“单位圆边界的相位参数化”。

\begin{theorem}[RH 的圆盘等价式：盘内零点禁入]\label{thm:app-rh-iff-disk-zero-free}
\leanverified{paper\_app\_rh\_iff\_disk\_zero\_free}
RH 等价于
\[
\boxed{\ \forall\,w\in\mathbb{D},\quad F_{\zeta}(w)\neq 0\ }.
\]
\end{theorem}
\begin{proof}
(\(\Rightarrow\)) 若 RH 不成立，则存在 \(\Xi_{\zeta}\) 的零点 \(\rho=\beta+\mathrm{i}\gamma\) 满足 \(\beta\neq \tfrac12\) 且 \(0<\beta<1\)。
由对称性 \(\Xi_{\zeta}(s)=\Xi_{\zeta}(1-s)\)，可不失一般性取 \(\beta<\tfrac12\)。
令
\[
t_{\rho}:=\gamma-\mathrm{i}\left(\beta-\tfrac12\right),
\qquad
\Im(t_{\rho})=\tfrac12-\beta>0,
\]
则 \(t_{\rho}\in\mathbb{H}\) 且 \( \tfrac12+\mathrm{i}t_{\rho}=\rho\)。
由命题 \ref{prop:app-cayley-upperhalf-disk}，取 \(w_{\rho}:=\mathcal{C}(t_{\rho})\in\mathbb{D}\) 并有 \(\mathcal{C}^{-1}(w_{\rho})=t_{\rho}\)。
因此
\[
F_{\zeta}(w_{\rho})
=\Xi_{\zeta}\!\left(\tfrac12+\mathrm{i}\,\mathcal{C}^{-1}(w_{\rho})\right)
=\Xi_{\zeta}\!\left(\tfrac12+\mathrm{i}t_{\rho}\right)
=\Xi_{\zeta}(\rho)
=0,
\]
从而盘内出现零点。

(\(\Leftarrow\)) 反之，若存在 \(w_0\in\mathbb{D}\) 使 \(F_{\zeta}(w_0)=0\)，令 \(t_0:=\mathcal{C}^{-1}(w_0)\in\mathbb{H}\) 并置 \(s_0:=\tfrac12+\mathrm{i}t_0\)。
由于 \(\Im(t_0)>0\)，有 \(\Re(s_0)=\tfrac12-\Im(t_0)<\tfrac12\)，且
\(
\Xi_{\zeta}(s_0)=F_{\zeta}(w_0)=0
\)。
由函数方程与共轭对称性立刻推出存在偏离临界线的零点，故 RH 不成立。证毕。
\end{proof}

\paragraph{谱截断逼近与 Carath\'eodory 闭包}
小节 \ref{subsubsec:app-horizon-weyl-carath-toeplitz} 把 RH 等价转写为视界函数 \(\mathscr{C}_{\zeta}\) 的 Carath\'eodory 正性与 Toeplitz--PSD 层级。
因此，若能构造一列有限维 Carath\'eodory 证书（对应有限正原子谱测度），并在圆盘内以紧一致方式逼近 \(\mathscr{C}_{\zeta}\)，则可由闭包性把正性传递到极限。

\begin{conjecture}[谱截断 Carath\'eodory 逼近的紧一致收敛]\label{conj:app-horizon-truncated-carath-convergence}
存在一列有限正原子测度 \(\{\mu_{\zeta,N}\}_{N\ge 1}\) 于 \(\TT\)，满足自对偶对称
\[
\mu_{\zeta,N}\circ(\xi\mapsto \xi^{-1})^{-1}=\mu_{\zeta,N},
\]
并令
\begin{equation}\label{eq:app-horizon-carath-trunc}
\mathscr{C}_{\zeta,N}(w):=\int_{\TT}\frac{\xi+w}{\xi-w}\,\dd\mu_{\zeta,N}(\xi),
\qquad w\in\mathbb{D}.
\end{equation}
则 \(\mathscr{C}_{\zeta,N}\to \mathscr{C}_{\zeta}\) 在 \(\mathbb{D}\) 的每个紧子集上一致收敛。
在该收敛成立时，\(\mathscr{C}_{\zeta}\) 作为 Carath\'eodory 类函数的局部一致极限仍满足 \(\Re \mathscr{C}_{\zeta}\ge 0\)，从而由推论 \ref{cor:app-horizon-carath-rh} 得 RH。
\end{conjecture}

\begin{remark}\label{rem:app-horizon-trunc-kernel-bound}
对任意 \(0<r<1\) 与 \(|w|\le r\)，核 \(\frac{\xi+w}{\xi-w}\) 在 \(\xi\in\TT\) 上一致有界，并满足
\[
\sup_{\xi\in\TT}\left|\frac{\xi+w}{\xi-w}\right|\le \frac{1+r}{1-r}.
\]
因此，一旦截断测度在总变差意义下收敛，核积分便在 \(|w|\le r\) 上自动获得一致收敛。
若需要将截断谱测度归一化为概率测度，可对 \(\mu_{\zeta,N}\) 作尺度归一化并同步缩放 \(\mathscr{C}_{\zeta,N}\) 的常项与总体幅度，以保持与 \(\mathscr{C}_{\zeta}\) 的归一化约定一致。
\end{remark}

\paragraph{离临界线偏移在圆盘半径上的精确定量压缩}
在定理 \ref{thm:app-rh-iff-disk-zero-free} 的证明中，离临界线零点事件被搬运为圆盘内零点事件。
下述命题给出该搬运在圆盘半径上的完全闭式压缩律：离线偏移 \(\tfrac12-\beta\) 在圆盘坐标中以 \(\Theta(\gamma^{-2})\) 的尺度被压到边界层。

\begin{proposition}[离临界线偏移的圆盘半径闭式压缩]\label{prop:app-offcritical-radius-compression}
\leanverified{paper\_app\_offcritical\_radius\_compression}
设 \(\Xi_{\zeta}\) 存在零点 \(\rho=\beta+\mathrm{i}\gamma\) 满足 \(\beta\neq\tfrac12\) 且 \(0<\beta<1\)。
由对称性 \(\Xi_{\zeta}(s)=\Xi_{\zeta}(1-s)\)，不失一般性取 \(\beta<\tfrac12\)，并令
\[
\delta:=\tfrac12-\beta>0,
\qquad
t_{\rho}:=\gamma+\mathrm{i}\delta\in\mathbb{H},
\qquad
w_{\rho}:=\mathcal{C}(t_{\rho})\in\mathbb{D}.
\]
则 \(F_{\zeta}(w_{\rho})=0\)。
并且 \(w_{\rho}\) 的圆盘半径满足完全闭式
\begin{equation}\label{eq:app-offcritical-radius-closed}
\boxed{\;
|w_{\rho}|^{2}
=\frac{\gamma^{2}+(\delta-1)^{2}}{\gamma^{2}+(\delta+1)^{2}},
\qquad
1-|w_{\rho}|^{2}
=\frac{4\delta}{\gamma^{2}+(\delta+1)^{2}}\; }.
\end{equation}
\end{proposition}
\begin{proof}
由定理 \ref{thm:app-rh-iff-disk-zero-free} 的构造，\(F_{\zeta}(w_{\rho})=\Xi_{\zeta}(\rho)=0\)。
模长闭式由命题 \ref{prop:app-cayley-upperhalf-disk} 的恒等式
\(
|\mathcal{C}(t)|^{2}=\frac{1+|t|^{2}-2\Im(t)}{1+|t|^{2}+2\Im(t)}
\)
取 \(t=t_{\rho}=\gamma+\mathrm{i}\delta\) 并直接化简得到。
\end{proof}

\begin{corollary}[高高度压缩尺度]\label{cor:app-offcritical-radius-high-height}
\leanverified{paper\_app\_offcritical\_radius\_high\_height}
当 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，
\[
1-|w_{\rho}|^{2}\sim \frac{4\delta}{\gamma^{2}},
\qquad
1-|w_{\rho}|\sim \frac{2\delta}{\gamma^{2}}.
\]
因此，在固定离临界线偏移 \(\delta\) 下，圆盘坐标对偏移的可分辨尺度为 \(\Theta(\gamma^{-2})\)：高度越大，径向压缩越强。
\end{corollary}
\begin{proof}
由 \eqref{eq:app-offcritical-radius-closed} 得
\(
1-|w_{\rho}|^{2}=\frac{4\delta}{\gamma^{2}+(\delta+1)^{2}}
\sim \frac{4\delta}{\gamma^{2}}
\)。
又由恒等式
\(
1-|w_{\rho}|=\frac{1-|w_{\rho}|^{2}}{1+|w_{\rho}|}
\)
并注意 \(|w_{\rho}|\to 1\)，即得第二式。证毕。
\end{proof}

\begin{corollary}[面积相位端点深度下的“二阶径向通道”]\label{cor:app-offcritical-radius-second-order-radial-channel}
\leanverified{paper\_app\_offcritical\_radius\_second\_order\_radial\_channel}
沿面积相位坐标 \(\upsilon(x)=\pi^{-1}\arctan(x)\)（式 \eqref{eq:unit-circle-phase-upsilon}），对高度 \(\gamma\in\RR\) 定义端点深度
\[
\delta_{\gamma}:=\frac12-\frac{1}{\pi}\arctan|\gamma|\in\Bigl(0,\frac12\Bigr),
\qquad\text{从而}\qquad
|\gamma|=\cot(\pi\delta_{\gamma}),\quad \delta_{\gamma}\sim\frac{1}{\pi|\gamma|}\ (|\gamma|\to\infty).
\]
设 \(\rho=\beta+\mathrm{i}\gamma\) 为 \(\Xi_{\zeta}\) 的零点且 \(0<\beta<\frac12\)，并令 \(w_{\rho}\in\mathbb{D}\) 如命题 \ref{prop:app-offcritical-radius-compression}。
则当 \(\delta_{\gamma}\downarrow 0\)（等价于 \(|\gamma|\to\infty\)）且 \(\beta\) 有界时，
\[
\boxed{\;
1-|w_{\rho}|^{2}
=
4\pi^{2}\Bigl(\tfrac12-\beta\Bigr)\,\delta_{\gamma}^{2}\bigl(1+o(1)\bigr),
\qquad
1-|w_{\rho}|
=
2\pi^{2}\Bigl(\tfrac12-\beta\Bigr)\,\delta_{\gamma}^{2}\bigl(1+o(1)\bigr)\; }.
\]
\end{corollary}
\begin{proof}
由推论 \ref{cor:app-offcritical-radius-high-height} 得
\(
1-|w_{\rho}|^{2}=\frac{4(\tfrac12-\beta)}{\gamma^{2}}(1+o(1))
\)
且 \(1-|w_{\rho}|=\frac{2(\tfrac12-\beta)}{\gamma^{2}}(1+o(1))\)。
另一方面，由 \(\delta_{\gamma}=\frac12-\pi^{-1}\arctan|\gamma|\) 得 \(|\gamma|=\cot(\pi\delta_{\gamma})\sim(\pi\delta_{\gamma})^{-1}\)，从而 \(\gamma^{-2}=\pi^{2}\delta_{\gamma}^{2}(1+o(1))\)。
代回即得结论。证毕。
\end{proof}

\begin{remark}[“偏离临界线”的二阶径向抑制]\label{rem:app-offcritical-second-order-radial-suppression}
推论 \ref{cor:app-offcritical-radius-second-order-radial-channel} 给出一个严格的结构事实：在“单位圆面积相位”紧化下，高度端点深度 \(\delta_{\gamma}\) 是一阶可见尺度（\(|\gamma|\sim(\pi\delta_{\gamma})^{-1}\)），而离临界线偏移 \(\tfrac12-\beta\) 并不以一阶相位扰动出现，而是被压缩到端点深度的\emph{二阶径向项} \(\delta_{\gamma}^{2}\) 上。
因此，在仅允许相位切片（unitary slice）而拒绝径向自由度的闭包内，\(\beta-\tfrac12\) 的信息被结构性地推入不可见通道；该现象将在 \S\ref{subsec:xi-pw-type-safety-null} 的空值（\(\mathsf{NULL}\)）语义中体现为“不可寻址”的硬拒绝。
\end{remark}

\begin{remark}[精度势表述：高度的指数压缩因子]\label{rem:app-offcritical-radius-precision-potential}
由定义 \ref{def:unit-circle-phase-precision-potential}，精度势满足
\(
e^{-\mathcal{P}(\gamma)}=\bigl(\pi(1+\gamma^{2})\bigr)^{-1}
\)。
因此 \eqref{eq:app-offcritical-radius-closed} 可改写为
\begin{equation}\label{eq:app-offcritical-radius-via-precision-potential}
\boxed{\;
1-|w_{\rho}|^{2}
=4\pi\,\delta\,e^{-\mathcal{P}(\gamma)}\cdot\frac{1+\gamma^{2}}{\gamma^{2}+(\delta+1)^{2}}\; }.
\end{equation}
特别地，当 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，
\(
1-|w_{\rho}|^{2}\sim 4\pi\,\delta\,e^{-\mathcal{P}(\gamma)}
\)。
该等价式将“离临界线偏移的圆盘可见性压缩”精确归因于精度势的指数因子 \(e^{-\mathcal{P}(\gamma)}\)。
\end{remark}

\begin{remark}[端点层的普适常数：\texorpdfstring{$T\log T$}{T log T} 密度在相位紧化下的对数弱发散]\label{rem:app-rvm-endpoint-constant}
令 \(N(T)\) 为 \(\zeta\) 非平凡零点在高度窗 \(0<\Im(\rho)\le T\) 内的计数函数。
黎曼--冯\,曼戈尔特公式给出主项
\[
N(T)=\frac{T}{2\pi}\log\!\Bigl(\frac{T}{2\pi}\Bigr)-\frac{T}{2\pi}+O(\log T)\qquad(T\to\infty)
\]
（参见 \cite{Apostol1976}）。
记其主项为
\(
M(T):=\frac{T}{2\pi}\log(\frac{T}{2\pi})-\frac{T}{2\pi}
\)，并用无参紧化 \(u=\upsilon(T)=\frac{1}{\pi}\arctan(T)\) 定义端点压缩主项
\(
M_{\Phi}(u):=M(\tan(\pi u))
\)。
则当 \(\delta:=\tfrac12-u\downarrow 0\) 时有极限
\begin{equation}\label{eq:app-rvm-endpoint-constant}
\boxed{\;
\lim_{\delta\downarrow 0}\;
\frac{\delta^{2}}{\log(1/\delta)}\,
\frac{\dd}{\dd u}M_{\Phi}\!\left(\tfrac12-\delta\right)
\;=\;\frac{1}{2\pi^{2}}\; }.
\end{equation}
该常数表明：在相位紧化坐标中，“无穷高度”的 \(T\log T\) 增长被重写为端点层 \(\delta\downarrow 0\) 的对数弱发散，并且其主尺度被剥离为一个与次阶振荡无关的普适常数。
几何上，\(\delta\downarrow 0\) 等价于 \(\vartheta(T)\to\pi\)，亦即边界点 \(w=e^{\mathrm{i}\vartheta(T)}\to-1\)（命题 \ref{prop:app-cayley-upperhalf-disk}）。
\end{remark}
\begin{proof}
由 \(M'(T)=\frac{1}{2\pi}\log(\frac{T}{2\pi})\) 与 \(\frac{\dd}{\dd u}\tan(\pi u)=\pi(1+\tan^2(\pi u))\)，得
\[
\frac{\dd}{\dd u}M_{\Phi}(u)
=\frac{1}{2\pi}\log\!\Bigl(\frac{\tan(\pi u)}{2\pi}\Bigr)\cdot \pi\bigl(1+\tan^2(\pi u)\bigr).
\]
取 \(u=\tfrac12-\delta\) 并用渐近 \(\tan(\pi(\tfrac12-\delta))=\cot(\pi\delta)\sim \frac{1}{\pi\delta}\)，则
\(
1+\tan^2(\pi u)\sim \frac{1}{\pi^2\delta^2}
\)
且
\(
\log(\tan(\pi u))=\log(1/\delta)+O(1)
\)。
代回即得
\(
\frac{\dd}{\dd u}M_{\Phi}(\tfrac12-\delta)
=\frac{1}{2\pi^{2}\delta^{2}}\log(1/\delta)\,(1+o(1))
\)，从而 \eqref{eq:app-rvm-endpoint-constant} 成立。证毕。
\end{proof}

\begin{proposition}[端点紧化下的平均可分辨尺度：相邻零点的 dyadic 位深度]\label{prop:app-adjacent-zero-dyadic-resolution}
令 \(\rho_n=\beta_n+\mathrm{i}\gamma_n\) 为 \(\zeta\) 的非平凡零点（按重数计），并按虚部递增排序 \(\gamma_1\le \gamma_2\le\cdots\)。
取端点相位紧化坐标
\(
u_n:=\upsilon(\gamma_n)=\pi^{-1}\arctan(\gamma_n)\in(-\tfrac12,\tfrac12)
\)
（式 \eqref{eq:unit-circle-phase-derivative}）。
则在 \(\gamma_n\sim T\to\infty\) 的平均尺度上有
\begin{equation}\label{eq:app-adjacent-zero-u-gap-scale}
\boxed{\;
|u_{n+1}-u_n|\ \asymp\ \frac{1}{T^{2}\log T}\; }.
\end{equation}
因此，若以 dyadic 盒宽度 \(2^{-b}\) 作为端点分辨率预算（亦即以 \(b\) 位精度给出 \(u\) 的 dyadic 括号），则区分相邻零点所需的最小位深度满足
\begin{equation}\label{eq:app-adjacent-zero-dyadic-depth}
\boxed{\;
b_{n}:=\Bigl\lceil-\log_{2}|u_{n+1}-u_n|\Bigr\rceil
\;=\;2\log_{2}T+\log_{2}\log T+O(1)\qquad(T\to\infty)\; }.
\end{equation}
\end{proposition}

\begin{proof}
由黎曼--冯\,曼戈尔特主项（备注 \ref{rem:app-rvm-endpoint-constant}），零点密度在高度 \(T\) 的尺度为
\(
N(T+H)-N(T)\asymp \frac{H}{2\pi}\log T
\)
（对 \(1\ll H\ll T\) 的局部窗），故“平均间距尺度”满足
\begin{equation}\label{eq:app-adjacent-zero-mean-gap-scale}
\gamma_{n+1}-\gamma_n\ \asymp\ \frac{2\pi}{\log T}
\qquad(\gamma_n\sim T\to\infty).
\end{equation}
另一方面，\(\upsilon'(t)=\bigl(\pi(1+t^2)\bigr)^{-1}\)（式 \eqref{eq:unit-circle-phase-derivative}），故由中值定理存在 \(\xi_n\in[\gamma_n,\gamma_{n+1}]\) 使
\[
u_{n+1}-u_n=\upsilon'(\xi_n)\,(\gamma_{n+1}-\gamma_n)
=\frac{\gamma_{n+1}-\gamma_n}{\pi(1+\xi_n^{2})}.
\]
当 \(\gamma_n\sim T\) 且 \(\gamma_{n+1}-\gamma_n\) 处于上述平均尺度时，亦有 \(\xi_n\sim T\)，从而 \((1+\xi_n^2)^{-1}\asymp T^{-2}\)。
与 \eqref{eq:app-adjacent-zero-mean-gap-scale} 合并即得 \eqref{eq:app-adjacent-zero-u-gap-scale}。
最后，对 \eqref{eq:app-adjacent-zero-u-gap-scale} 取 \(-\log_2\) 并用 \(-\log_2(T^{-2}(\log T)^{-1})=2\log_2 T+\log_2\log T+O(1)\) 即得 \eqref{eq:app-adjacent-zero-dyadic-depth}。证毕。
\end{proof}
