\subsubsection{视界 Weyl--Carath\'eodory 正性与 Toeplitz 层级：对数导数的 PSD 可审计判别}\label{subsubsec:app-horizon-weyl-carath-toeplitz}
本小节在附录 \ref{subsec:unit-disk-rh-equivalence} 的圆盘化完成化接口之上，给出一个与“盘内零点禁入”等价、但更贴近证书语义的闭合判别：把 RH 转写为视界对数导数的 Carath\'eodory 正性（正实部），并进一步等价为一个 Toeplitz 半正定（PSD）层级。
该层级属于截断矩问题的标准形式，因此每一级都可被视作可审计的半正定可行性约束。
此外，在 \(|w|<1/3\) 的子圆盘内，可借函数方程把该视界函数转写为绝对收敛的欧拉积/Dirichlet 级数，从而提供一个“远离端点”的可审计补丁：算术项完全进入一致绝对收敛域，困难被严格隔离到端点邻域的解析延拓与正性保持问题。
\par
为避免与 Cayley 变换 \(\mathcal{C}\)（式 \eqref{eq:app-cayley-disk}）混淆，本文用花体字母 \(\mathscr{C}\) 记 Carath\'eodory 类函数。

\paragraph{视界一变量与 Weyl（对数导数）函数}
沿用式 \eqref{eq:app-xi-classical} 的完成化整函数 \(\Xi_{\zeta}\)，定义
\[
E_{\zeta}(z):=\Xi_{\zeta}\!\left(\tfrac12+\mathrm{i}z\right),\qquad z\in\CC .
\]
则 \(E_{\zeta}\) 为实整函数（\(E_{\zeta}(\overline z)=\overline{E_{\zeta}(z)}\)），且 RH 等价于 \(E_{\zeta}\) 的零点全在实轴：若 \(\rho=\beta+\mathrm{i}\gamma\) 为 \(\Xi_{\zeta}\) 的零点，则令
\[
z:=\gamma+\mathrm{i}\Bigl(\tfrac12-\beta\Bigr)
\]
有 \(E_{\zeta}(z)=0\)，且 \(\beta<\tfrac12 \iff \Im z>0\)。

定义视界 Weyl（对数导数）函数
\[
m_{\zeta}(z):=-\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}\qquad(z\in\mathbb{H}).
\]

\begin{theorem}[视界 Weyl 函数的 Herglotz 等价]\label{thm:app-horizon-weyl-herglotz}
RH 成立当且仅当 \(m_{\zeta}\) 在上半平面 \(\mathbb{H}\) 上解析且满足
\[
\Im m_{\zeta}(z)\ge 0,\qquad z\in\mathbb{H},
\]
即 \(m_{\zeta}\) 为 Herglotz（Nevanlinna）函数。
\end{theorem}

\begin{proof}
(\(\Rightarrow\)) 若 RH 成立，则 \(E_{\zeta}\) 的全部零点为实数（按重数计为 \(\{\gamma_n\}\subset\RR\)）。
因此 \(E_{\zeta}'/E_{\zeta}\) 在 \(\mathbb{H}\) 上解析，并具有部分分式展开
\[
\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}=\sum_n\frac{1}{z-\gamma_n}+c,
\qquad c\in\RR,
\]
其中级数按标准收敛意义理解。
对任意 \(\gamma_n\in\RR\) 与 \(\Im z>0\)，有 \(\Im\frac{1}{z-\gamma_n}<0\)，从而
\(\Im m_{\zeta}(z)=-\Im\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}\ge 0\)。
\par
(\(\Leftarrow\)) 若 \(m_{\zeta}\) 为 Herglotz，则 \(m_{\zeta}\) 在 \(\mathbb{H}\) 上解析。
而 \(m_{\zeta}\) 的极点恰为 \(E_{\zeta}\) 在 \(\mathbb{H}\) 内的零点；故 \(E_{\zeta}\) 在 \(\mathbb{H}\) 内无零点。
由 \(E_{\zeta}\) 的实整性，任一非实零点必成共轭对，必有一根落在 \(\mathbb{H}\)，与“\(\mathbb{H}\) 内无零点”矛盾。
故 \(E_{\zeta}\) 零点全在 \(\RR\)，等价于 RH。证毕。
\end{proof}
\leanverified{paper\_app\_horizon\_weyl\_herglotz}

\paragraph{圆盘视界端点与 Carath\'eodory 正性}
沿用 Cayley 变换 \(\mathcal{C}:\mathbb{H}\to\mathbb{D}\) 及其逆 \(\mathcal{C}^{-1}\)（式 \eqref{eq:app-cayley-disk}），并沿用圆盘化截面 \(s(w)=\tfrac12+\mathrm{i}\,\mathcal{C}^{-1}(w)\)（式 \eqref{eq:app-s-of-w}）。
定义视界 Carath\'eodory（对数导数）函数
\begin{equation}\label{eq:app-horizon-carath-def}
\boxed{\;
\mathscr{C}_{\zeta}(w):=-\frac{\Xi_{\zeta}'(s(w))}{\Xi_{\zeta}(s(w))}\qquad(w\in\mathbb{D}).\;}
\end{equation}
注意到 \(E_{\zeta}'(z)=\mathrm{i}\,\Xi_{\zeta}'(\tfrac12+\mathrm{i}z)\)，从而对 \(w=\mathcal{C}(z)\)（\(z\in\mathbb{H}\)）有严格恒等式
\begin{equation}\label{eq:app-horizon-m-c-relation}
\boxed{\;
\mathscr{C}_{\zeta}(w)=-\mathrm{i}\,m_{\zeta}\!\bigl(\mathcal{C}^{-1}(w)\bigr),\qquad
m_{\zeta}(z)=\mathrm{i}\,\mathscr{C}_{\zeta}\!\bigl(\mathcal{C}(z)\bigr).\;}
\end{equation}

\begin{corollary}[RH 的圆盘正性形式]\label{cor:app-horizon-carath-rh}
\leanverified{paper\_app\_horizon\_carath\_rh}
RH 成立当且仅当 \(\mathscr{C}_{\zeta}\) 在 \(\mathbb{D}\) 上解析且满足
\[
\Re \mathscr{C}_{\zeta}(w)\ge 0,\qquad w\in\mathbb{D},
\]
即 \(\mathscr{C}_{\zeta}\) 属于 Carath\'eodory 类（正实部函数）。
\end{corollary}

\begin{proof}
由 \eqref{eq:app-horizon-m-c-relation}，对 \(z\in\mathbb{H}\) 有
\(\Re \mathscr{C}_{\zeta}(\mathcal{C}(z))=\Im m_{\zeta}(z)\)。
因此 \(\Im m_{\zeta}\ge 0\) 与 \(\Re\mathscr{C}_{\zeta}\ge 0\) 等价；
再与定理 \ref{thm:app-horizon-weyl-herglotz} 合并即得结论。
解析性条件亦一致：若 \(\Xi_{\zeta}\) 在 \(\mathbb{H}\) 内有零点，则 \(m_{\zeta}\) 在该点有极点；等价地，\(\mathscr{C}_{\zeta}\) 在其圆盘像处出现极点，从而不可能属于 Carath\'eodory 类。证毕。
\end{proof}

\paragraph{Toeplitz 半正定层级：Carath\'eodory \texorpdfstring{$\Leftrightarrow$}{<->} PSD 矩序列}
设 \(f\) 为 \(\mathbb{D}\) 上解析函数并满足 \(\Re f(w)\ge 0\)。
Carath\'eodory/Herglotz 表示（例如 \cite{Duren1970TheoryHp,Garnett2007BoundedAnalyticFunctions,Simon2005OPUC1}）给出：存在有限正 Borel 测度 \(\mu\) 于单位圆 \(\TT\)（或等价地，于 \([0,2\pi)\)）与实常数 \(\beta\) 使
\begin{equation}\label{eq:app-horizon-herglotz-disk}
f(w)=\mathrm{i}\beta+\int_{\TT}\frac{\xi+w}{\xi-w}\,\dd\mu(\xi).
\end{equation}
当 \(f(\overline w)=\overline{f(w)}\) 时，其 Maclaurin 系数满足共轭对称性；并可取归一化幂级数形式
\begin{equation}\label{eq:app-horizon-carath-series}
f(w)=c_0+2\sum_{n\ge 1}c_n w^n,\qquad c_{-n}:=\overline{c_n},
\end{equation}
其中 \(c_n\) 为 \(\mu\) 的 Fourier 矩：\(c_n=\int_{\TT}\xi^{-n}\dd\mu(\xi)\)。

\paragraph{RH 假设下的视界谱测度显式化：零点的 Cayley 像原子化}
推论 \ref{cor:app-horizon-carath-rh} 表明：RH 成立时 \(\mathscr{C}_{\zeta}\) 属于 Carath\'eodory 类，因此其表示 \eqref{eq:app-horizon-herglotz-disk} 中的谱测度 \(\mu\)（以及矩 \(\{c_n\}\)）在原则上由 \(\mathscr{C}_{\zeta}\) 唯一决定。
下述定理给出该谱测度在 RH 前提下的一个闭式原子化正规形：它把零点集合直接转写为单位圆上的纯原子谱对象，并同时给出矩序列的绝对收敛求和表示。

\begin{theorem}[RH 假设下的视界谱测度原子化公式]\label{thm:app-horizon-spectral-measure-atomic}
\leanverified{paper\_app\_horizon\_spectral\_measure\_atomic}
假设 RH 成立。
令 \(E_{\zeta}(z)=\Xi_{\zeta}(\tfrac12+\mathrm{i}z)\)，并令 \(\Gamma_+\subset(0,\infty)\) 为 \(E_{\zeta}\) 的正零点多重集（按重数计）。
对每个 \(\gamma\in\Gamma_+\) 记零点重数为 \(\mathrm{mult}(\gamma)\)，并定义
\[
\xi_{\gamma}:=\mathcal{C}(\gamma)\in\TT,
\qquad
\xi_{-\gamma}=\mathcal{C}(-\gamma)=\xi_{\gamma}^{-1}.
\]
则 \(\mathscr{C}_{\zeta}\) 的 Herglotz--Carath\'eodory 表示 \eqref{eq:app-horizon-herglotz-disk} 中可取 \(\beta=0\)，并且可取一纯原子测度 \(\mu_{\zeta}\) 使
\begin{equation}\label{eq:app-horizon-mu-atomic}
\boxed{\;
\mu_{\zeta}
=
\sum_{\gamma\in\Gamma_+}\frac{\mathrm{mult}(\gamma)}{1+\gamma^2}\Bigl(\delta_{\xi_{\gamma}}+\delta_{\xi_{\gamma}^{-1}}\Bigr).
\;}
\end{equation}
在归一化幂级数 \eqref{eq:app-horizon-carath-series} 下，其 Fourier 矩（视界多极矩）满足绝对收敛求和公式：对任意 \(n\ge 0\)，
\begin{equation}\label{eq:app-horizon-moment-sum}
\boxed{\;
c_n=\int_{\TT}\xi^{-n}\,\dd\mu_{\zeta}(\xi)
=
\sum_{\gamma\in\Gamma_+}\frac{\mathrm{mult}(\gamma)}{1+\gamma^2}\bigl(\xi_{\gamma}^{-n}+\xi_{\gamma}^{n}\bigr)
=
2\sum_{\gamma\in\Gamma_+}\frac{\mathrm{mult}(\gamma)}{1+\gamma^2}\cos\!\bigl(n\theta_{\gamma}\bigr),
\qquad \xi_{\gamma}=e^{\mathrm{i}\theta_{\gamma}}.
\;}
\end{equation}
\end{theorem}

\begin{proof}
由 \(\Xi_{\zeta}(s)=\Xi_{\zeta}(1-s)\) 得 \(E_{\zeta}(z)=E_{\zeta}(-z)\)，故 \(E_{\zeta}\) 为偶函数，从而 \((E_{\zeta}'/E_{\zeta})(z)\) 为奇函数。
在 RH 成立时 \(E_{\zeta}\) 的全部零点皆为实数且关于原点成对出现；因此（按重数计）可取对称求和使
\begin{equation}\label{eq:app-horizon-Elog-odd-pf}
\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}
=
\sum_{\gamma\in\Gamma_+}\mathrm{mult}(\gamma)\!\left(\frac{1}{z-\gamma}+\frac{1}{z+\gamma}\right),
\qquad z\in\mathbb{H},
\end{equation}
其中级数按标准部分分式意义理解（与定理 \ref{thm:app-horizon-weyl-herglotz} 的展开一致，并由奇性固定常数项为零）。
从而
\[
m_{\zeta}(z):=-\frac{E_{\zeta}'(z)}{E_{\zeta}(z)}
=
\sum_{\gamma\in\Gamma_+}\mathrm{mult}(\gamma)\!\left(\frac{1}{\gamma-z}+\frac{1}{-\gamma-z}\right).
\]

令 \(t\in\RR\)、\(z\in\mathbb{H}\)，并置 \(\xi=\mathcal{C}(t)\)、\(w=\mathcal{C}(z)\)。
由 \eqref{eq:app-cayley-disk} 的代数化简得到核恒等式
\begin{equation}\label{eq:app-horizon-cayley-kernel-identity}
\frac{\xi+w}{\xi-w}
=-\mathrm{i}\,\frac{1+t^{2}}{t-z}+\mathrm{i}t,
\end{equation}
等价地，
\begin{equation}\label{eq:app-horizon-cayley-kernel-solve}
\frac{1}{t-z}
=\frac{\mathrm{i}}{1+t^{2}}\frac{\xi+w}{\xi-w}+\frac{t}{1+t^{2}}.
\end{equation}
将 \eqref{eq:app-horizon-cayley-kernel-solve} 代入 \(t=\gamma\) 与 \(t=-\gamma\) 并相加，常数项 \(\pm\gamma/(1+\gamma^2)\) 抵消，得到
\[
\frac{1}{\gamma-z}+\frac{1}{-\gamma-z}
=\frac{\mathrm{i}}{1+\gamma^{2}}
\left(
\frac{\xi_{\gamma}+w}{\xi_{\gamma}-w}
+
\frac{\xi_{\gamma}^{-1}+w}{\xi_{\gamma}^{-1}-w}
\right).
\]
代回 \(m_{\zeta}\) 的展开并用关系式 \eqref{eq:app-horizon-m-c-relation}（\(\mathscr{C}_{\zeta}(w)=-\mathrm{i}\,m_{\zeta}(\mathcal{C}^{-1}(w))\)），即可得到
\[
\mathscr{C}_{\zeta}(w)
=
\sum_{\gamma\in\Gamma_+}\frac{\mathrm{mult}(\gamma)}{1+\gamma^2}
\left(
\frac{\xi_{\gamma}+w}{\xi_{\gamma}-w}
+
\frac{\xi_{\gamma}^{-1}+w}{\xi_{\gamma}^{-1}-w}
\right),
\qquad w\in\mathbb{D}.
\]
与一般表示 \eqref{eq:app-horizon-herglotz-disk} 对照，识别出 \(\beta=0\) 且谱测度 \(\mu_{\zeta}\) 可取为 \eqref{eq:app-horizon-mu-atomic}。
矩公式 \eqref{eq:app-horizon-moment-sum} 由 \(c_n=\int_{\TT}\xi^{-n}\dd\mu_{\zeta}(\xi)\) 与 \(|\xi_{\gamma}|=1\) 直接推出；绝对收敛性来自 \(\sum_{\gamma\in\Gamma_+}\mathrm{mult}(\gamma)/(1+\gamma^2)<\infty\)。
证毕。
\end{proof}

\begin{corollary}[Ramanujan 自对偶对称的测度级实现]\label{cor:app-horizon-ramanujan-selfdual-measure}
\leanverified{paper\_app\_horizon\_ramanujan\_selfdual\_measure}
在定理 \ref{thm:app-horizon-spectral-measure-atomic} 的表示下，\(\mu_{\zeta}\) 满足严格自对偶对称性
\[
\mu_{\zeta}\circ(\xi\mapsto \xi^{-1})^{-1}=\mu_{\zeta},
\]
即谱点以 \((\xi,\xi^{-1})\) 成对出现且权重相等；因此 \(\{c_n\}_{n\ge 0}\) 为实序列且满足 \(c_{-n}=c_n\)。
\end{corollary}

\paragraph{共动平移族的谱测度（正文层析定理的证明）}\label{par:app-horizon-comoving-spectral-measure}
正文小节 \ref{subsubsec:xi-horizon-spectral-measure-tomography} 引入共动平移族
\(
\mathscr{C}_{\zeta,x_0}(w):=-\mathrm{i}\,m_{\zeta}\!\bigl(x_0+\mathcal{C}^{-1}(w)\bigr)
\)
以实现对视界矩序列的层析扫描。
这里给出其谱测度的闭式原子化与“加权显式公式”接口的技术性推导。

\begin{proof}[定理 \ref{thm:xi-horizon-comoving-spectral-measure} 的证明]
令 \(x_0\in\RR\) 固定，并定义上半平面中的平移族
\(
m_{\zeta,x_0}(z):=m_{\zeta}(z+x_0)
\)。
由定理 \ref{thm:app-horizon-weyl-herglotz}，在 RH 前提下 \(m_{\zeta}\) 为 Herglotz 函数（\(\Im m_{\zeta}\ge 0\) 于 \(\mathbb{H}\)），而平移 \(z\mapsto z+x_0\) 为 \(\mathbb{H}\) 的共形自同构，故 \(m_{\zeta,x_0}\) 仍为 Herglotz。
将其经 Cayley 紧化推回圆盘并乘以 \(-\mathrm{i}\) 得
\[
\mathscr{C}_{\zeta,x_0}(w):=-\mathrm{i}\,m_{\zeta}\!\bigl(x_0+\mathcal{C}^{-1}(w)\bigr),
\qquad w\in\mathbb{D},
\]
从而 \(\Re \mathscr{C}_{\zeta,x_0}(w)\ge 0\) 于 \(\mathbb{D}\)，即 \(\mathscr{C}_{\zeta,x_0}\) 为 Carath\'eodory 函数。
\par
在 RH 成立时，\(E_{\zeta}\) 的全部零点为实数且关于原点成对出现。
沿用 \eqref{eq:app-horizon-Elog-odd-pf}（按重数计）的部分分式展开并将 \(z\) 替换为 \(z+x_0\)，得对 \(z\in\mathbb{H}\)
\[
m_{\zeta,x_0}(z)
=
\sum_{\gamma\in\Gamma_+}\mathrm{mult}(\gamma)\!\left(\frac{1}{\gamma-(z+x_0)}+\frac{1}{-\gamma-(z+x_0)}\right).
\]
将两项分别写成 Cauchy 核 \(\frac{1}{t-z}\) 的形式：令
\(
t_{\pm}:=\pm\gamma-x_0\in\RR
\)，则上式为
\[
m_{\zeta,x_0}(z)
=
\sum_{\gamma\in\Gamma_+}\mathrm{mult}(\gamma)\!\left(\frac{1}{t_+-z}+\frac{1}{t_--z}\right).
\]
对每个实参数 \(t\in\RR\)，令 \(\xi:=\mathcal{C}(t)\in\TT\) 且 \(w:=\mathcal{C}(z)\in\mathbb{D}\)。
由核恒等式 \eqref{eq:app-horizon-cayley-kernel-solve}，
\[
\frac{1}{t-z}
=\frac{\mathrm{i}}{1+t^{2}}\frac{\xi+w}{\xi-w}+\frac{t}{1+t^{2}}.
\]
将该恒等式逐项代入并乘以 \(-\mathrm{i}\)，得到
\[
\mathscr{C}_{\zeta,x_0}(w)
=\mathrm{i}\beta(x_0)
+\sum_{\gamma\in\Gamma_+}\mathrm{mult}(\gamma)
\left(
\frac{1}{1+t_+^2}\frac{\xi_{t_+}+w}{\xi_{t_+}-w}
\;+\;
\frac{1}{1+t_-^2}\frac{\xi_{t_-}+w}{\xi_{t_-}-w}
\right).
\]
其中 \(\xi_{t_\pm}:=\mathcal{C}(t_\pm)\in\TT\)，而 \(\beta(x_0)\in\RR\) 为 Carath\'eodory/Herglotz 表示中的实常数项（可吸收全部 \(t/(1+t^2)\) 型常数贡献）；其具体值不影响矩与 PSD 证书。
将求和指标从 \(\Gamma_+\) 改写为全零点多重集 \(\Gamma:=\Gamma_+\cup(-\Gamma_+)\subset\RR\)（按重数计），并记
\(
\xi_{\gamma,x_0}:=\mathcal{C}(\gamma-x_0)
\)，即可识别出 Carath\'eodory 表示的纯原子测度
\[
\mu_{\zeta,x_0}
=
\sum_{\gamma\in\Gamma}\frac{\mathrm{mult}(\gamma)}{1+(\gamma-x_0)^2}\,\delta_{\xi_{\gamma,x_0}},
\]
从而得到正文式 \eqref{eq:xi-horizon-mu-comoving} 与矩恒等式 \eqref{eq:xi-horizon-moment-comoving}（由 \(c_n(x_0)=\int_{\TT}\xi^{-n}\dd\mu_{\zeta,x_0}(\xi)\) 直接推出）。
\par
最后证明 \eqref{eq:xi-horizon-c0-im-m}。
由 \(\mathcal{C}^{-1}(0)=\mathrm{i}\)（式 \eqref{eq:app-cayley-disk}），有
\(
\mathscr{C}_{\zeta,x_0}(0)=-\mathrm{i}\,m_{\zeta}(x_0+\mathrm{i})
\)。
另一方面，对任意 Carath\'eodory 表示 \(f(w)=\mathrm{i}\beta+\int_{\TT}\frac{\xi+w}{\xi-w}\dd\mu(\xi)\) 有
\(
\Re f(0)=\mu(\TT)
\)。
取 \(f=\mathscr{C}_{\zeta,x_0}\) 与 \(\mu=\mu_{\zeta,x_0}\) 即得
\[
c_0(x_0)=\mu_{\zeta,x_0}(\TT)=\Re \mathscr{C}_{\zeta,x_0}(0)=\Im m_{\zeta}(x_0+\mathrm{i}).
\]
证毕。
\end{proof}

\begin{remark}[\texorpdfstring{$|w|<1/3$}{|w|<1/3} 的共动欧拉积补丁与相位因子]\label{rem:app-horizon-euler-patch-comoving}
沿用 \eqref{eq:app-horizon-fe-rewrite} 与 \eqref{eq:app-horizon-sigma-def}。
注意
\(
1-(s(w)+\mathrm{i}x_0)=\sigma(w)-\mathrm{i}x_0
\)，
从而
\[
\mathscr{C}_{\zeta,x_0}(w)
=-\frac{\Xi_{\zeta}'(s(w)+\mathrm{i}x_0)}{\Xi_{\zeta}(s(w)+\mathrm{i}x_0)}
=\frac{\Xi_{\zeta}'(\sigma(w)-\mathrm{i}x_0)}{\Xi_{\zeta}(\sigma(w)-\mathrm{i}x_0)}.
\]
当 \(|w|<1/3\) 时仍有 \(\Re\sigma(w)>1\)（定理 \ref{thm:app-horizon-euler-patch}），故在该子圆盘内可得到与 \eqref{eq:app-horizon-euler-patch} 同型的一致绝对收敛展开，只需将 \(s\) 替换为 \(\sigma(w)-\mathrm{i}x_0\)：
\[
\mathscr{C}_{\zeta,x_0}(w)
=\frac{1}{\sigma(w)-\mathrm{i}x_0}+\frac{1}{\sigma(w)-1-\mathrm{i}x_0}
-\frac12\log\pi+\frac12\,\psi\!\left(\frac{\sigma(w)-\mathrm{i}x_0}{2}\right)
-\sum_{n\ge2}\frac{\Lambda(n)}{n^{\sigma(w)-\mathrm{i}x_0}}.
\]
其中 Dirichlet 项等于 \(-\sum_{n\ge2}\Lambda(n)\,n^{\mathrm{i}x_0}n^{-\sigma(w)}\)，即相位因子 \(n^{\mathrm{i}x_0}\) 被结构性引入，而绝对收敛性与逐项求导/系数提取的合法性不受 \(x_0\) 影响（因为 \(\Re(\sigma(w)-\mathrm{i}x_0)=\Re\sigma(w)\)）。
这给出正文推论 \ref{cor:xi-horizon-weighted-explicit-formula} 所述的“加权显式公式族”的可审计算术接口。
\end{remark}

\begin{theorem}[端点 \texorpdfstring{$\xi=-1$}{xi=-1} 角窗质量的一阶渐近]\label{thm:app-horizon-endpoint-window-mass}
\leanverified{paper\_app\_horizon\_endpoint\_window\_mass}
假设 RH 成立并令 \(\mu_{\zeta}\) 为定理 \ref{thm:app-horizon-spectral-measure-atomic} 给出的视界谱测度。
对 \(0<\varepsilon<\pi\) 定义端点角窗
\[
I_{\varepsilon}:=\Bigl\{\xi=e^{\mathrm{i}\theta}\in\TT:\ |\theta-\pi|<\varepsilon\Bigr\},
\qquad \theta\in[0,2\pi).
\]
则当 \(\varepsilon\downarrow 0\) 时有
\[
\mu_{\zeta}(I_{\varepsilon})
=
\frac{\varepsilon}{2\pi}\left(\log\frac{1}{\pi\varepsilon}+1\right)
+o(\varepsilon).
\]
特别地，\(\mu_{\zeta}(I_{\varepsilon})\sim \frac{\varepsilon}{2\pi}\log\frac{1}{\varepsilon}\)。
\end{theorem}

\begin{proof}
由定理 \ref{thm:app-horizon-spectral-measure-atomic} 的原子化公式 \eqref{eq:app-horizon-mu-atomic}，有
\[
\mu_{\zeta}(I_{\varepsilon})
=
2\sum_{\substack{\gamma\in\Gamma_+\\ \xi_{\gamma}\in I_{\varepsilon}}}
\frac{\mathrm{mult}(\gamma)}{1+\gamma^2}.
\]
对 \(\gamma>0\) 记 \(\xi_{\gamma}=e^{\mathrm{i}\theta_{\gamma}}\)（\(\theta_{\gamma}\in(0,\pi)\)）。
由 \(\theta_{\gamma}=2\arctan(\gamma)\)（由 \eqref{eq:app-cayley-disk} 取 \(w=e^{\mathrm{i}\theta}\) 化简可得）可知
\(\xi_{\gamma}\in I_{\varepsilon}\iff \theta_{\gamma}>\pi-\varepsilon\iff \gamma\ge \cot(\varepsilon/2)\)。
令 \(T_{\varepsilon}:=\cot(\varepsilon/2)\to\infty\)（\(\varepsilon\downarrow 0\)），并令 \(N(T)\) 为正零点计数函数（按重数计），则
\[
\mu_{\zeta}(I_{\varepsilon})
=2\int_{T_{\varepsilon}}^{\infty}\frac{1}{1+t^2}\,\dd N(t).
\]
由黎曼--冯\,曼戈尔特公式主项（备注 \ref{rem:app-rvm-endpoint-constant}），有
\(
N(t)=M(t)+O(\log t)
\)
且 \(M'(t)=\frac{1}{2\pi}\log(\frac{t}{2\pi})\)。
因此对误差项用分部积分并利用 \((1+t^2)^{-1}\) 与其导数的衰减，得到
\[
\mu_{\zeta}(I_{\varepsilon})
=2\int_{T_{\varepsilon}}^{\infty}\frac{1}{1+t^2}M'(t)\,\dd t
+O\!\left(\frac{\log T_{\varepsilon}}{T_{\varepsilon}^2}\right).
\]
对主项积分，利用 \(t\to\infty\) 下 \( (1+t^2)^{-1}=t^{-2}+O(t^{-4}) \) 得
\[
\int_{T_{\varepsilon}}^{\infty}\frac{1}{1+t^2}M'(t)\,\dd t
=\frac{1}{2\pi}\int_{T_{\varepsilon}}^{\infty}\frac{\log(\frac{t}{2\pi})}{t^2}\,\dd t
+O\!\left(\frac{\log T_{\varepsilon}}{T_{\varepsilon}^3}\right)
=\frac{1}{2\pi}\frac{\log(\frac{T_{\varepsilon}}{2\pi})+1}{T_{\varepsilon}}
+O\!\left(\frac{\log T_{\varepsilon}}{T_{\varepsilon}^3}\right).
\]
从而
\[
\mu_{\zeta}(I_{\varepsilon})
=\frac{1}{\pi}\frac{\log(\frac{T_{\varepsilon}}{2\pi})+1}{T_{\varepsilon}}
+O\!\left(\frac{\log T_{\varepsilon}}{T_{\varepsilon}^2}\right).
\]
最后用 \(T_{\varepsilon}^{-1}=\tan(\varepsilon/2)=\varepsilon/2+O(\varepsilon^{3})\) 与
\(\log(\tfrac{T_{\varepsilon}}{2\pi})=\log\!\bigl(\tfrac{1}{2\pi\tan(\varepsilon/2)}\bigr)=\log(\tfrac{1}{\pi\varepsilon})+O(\varepsilon^{2})\)，即可得到所示渐近式；误差项为 \(o(\varepsilon)\)。证毕。
\end{proof}

\begin{remark}[与精度势权重的对齐]\label{rem:app-horizon-endpoint-window-mass-precision-potential}
由备注 \ref{rem:app-offcritical-radius-precision-potential}，有
\(
(1+\gamma^2)^{-1}=\pi\,e^{-\mathcal{P}(\gamma)}
\)。
因此端点角窗质量的首阶尺度 \(\varepsilon\log(1/\varepsilon)\) 可视为：在相位紧化下由 \(T\log T\) 密度诱导的对数弱发散，经精度势权重的二次红移抑制后的可见端点层质量。
\end{remark}

由此得到 Toeplitz 半正定约束：对任意 \(N\ge 0\) 定义 Toeplitz 矩阵
\[
\mathbf{T}_N:=(c_{j-k})_{0\le j,k\le N}.
\]
则 \(\mathbf{T}_N\succeq 0\)，因为 \(\mathbf{T}_N\) 是 \(L^2(\mu)\) 中向量 \(\{1,\xi,\dots,\xi^N\}\) 的 Gram 矩阵。
反过来，\(\{\mathbf{T}_N\succeq 0\}_{N\ge 0}\) 恰为截断矩问题的全体主子式正性条件，保证存在正测度 \(\mu\) 使 \eqref{eq:app-horizon-herglotz-disk} 成立，进而 \(f\) 为 Carath\'eodory。

\begin{theorem}[Toeplitz--PSD 层级与 RH 的等价性]\label{thm:app-horizon-toeplitz-lmi}
令 \(\mathscr{C}_{\zeta}\) 如 \eqref{eq:app-horizon-carath-def} 定义，并将其在 \(w=0\) 处的幂级数写作
\[
\mathscr{C}_{\zeta}(w)=c_0+2\sum_{n\ge 1}c_n w^n,\qquad c_{-n}:=\overline{c_n}.
\]
则下列陈述等价：\vspace{-2pt}
\begin{enumerate}
  \item RH 成立；
  \item \(\mathscr{C}_{\zeta}\) 属于 Carath\'eodory 类（推论 \ref{cor:app-horizon-carath-rh}）；
  \item 对所有 \(N\ge 0\)，Toeplitz 主子矩阵
  \[
  \boxed{\;\mathbf{T}_N=(c_{j-k})_{0\le j,k\le N}\succeq 0\;}
  \]
  成立。
\end{enumerate}
换言之，RH 等价于一族可逐级验证、且每一级均为半正定可行性问题的 PSD 层级。
\end{theorem}

\begin{proof}
\((1)\Leftrightarrow(2)\) 由推论 \ref{cor:app-horizon-carath-rh}。
\((2)\Rightarrow(3)\) 由 \eqref{eq:app-horizon-herglotz-disk} 的 Fourier 矩展开与 Gram 矩阵解释立得。
\((3)\Rightarrow(2)\) 属于 Carath\'eodory--Fej\'er/截断矩问题的标准完备性：若所有 Toeplitz 主子矩阵半正定，则存在正测度 \(\mu\) 使得矩序列 \(\{c_n\}\) 实现为 \(\mu\) 的 Fourier 矩，从而由 \eqref{eq:app-horizon-herglotz-disk} 得到 \(\Re f\ge 0\)（参见 \cite{Duren1970TheoryHp,Simon2005OPUC1}）。证毕。
\end{proof}
\leanverified{paper\_app\_horizon\_toeplitz\_lmi}

\begin{proposition}[Toeplitz--PSD 的 \texorpdfstring{$\ZZ/2$}{Z/2} 扇区分解（反转对合下的偶/奇块）]\label{prop:app-horizon-toeplitz-z2-splitting}
\leanverified{paper\_app\_horizon\_toeplitz\_z2\_splitting}
在 \eqref{eq:app-horizon-carath-series} 的共轭对称口径下，进一步假设系数满足实结构
\[
c_n\in\RR\qquad(n\in\ZZ),
\]
从而 \(c_{-n}=c_n\) 且每个 \(\mathbf{T}_N\) 为实对称 Toeplitz 矩阵。
令 \(\mathbf{J}_N\in\RR^{(N+1)\times(N+1)}\) 为反转矩阵（\( \mathbf{J}_N e_k=e_{N-k} \)），并令
\[
E_N^{\pm}:=\ker(\mathbf{J}_N\mp I).
\]
则 \( \RR^{N+1}=E_N^{+}\oplus E_N^{-} \) 为正交直和，并且 \(\mathbf{T}_N\) 与 \(\mathbf{J}_N\) 对易，从而 \(E_N^{\pm}\) 均为 \(\mathbf{T}_N\) 的不变子空间。
取一组适配分解 \(E_N^{+}\oplus E_N^{-}\) 的正交基，则存在正交矩阵 \(Q_N\) 使
\[
Q_N^{\mathsf T}\mathbf{T}_N Q_N
=
\begin{pmatrix}
\mathbf{T}_N^{+} & 0\\
0 & \mathbf{T}_N^{-}
\end{pmatrix},
\]
其中 \(\mathbf{T}_N^{\pm}\) 为 \(\mathbf{T}_N\) 在 \(E_N^{\pm}\) 上的限制矩阵（在不同正交基下相互正交共轭）。
因此
\[
\mathbf{T}_N\succeq 0
\quad\Longleftrightarrow\quad
\mathbf{T}_N^{+}\succeq 0\ \text{且}\ \mathbf{T}_N^{-}\succeq 0.
\]
\end{proposition}

\begin{proof}
对任意 \(0\le j,k\le N\)，由 \(\mathbf{T}_N=(c_{j-k})\) 与 \(\mathbf{J}_N e_k=e_{N-k}\) 得
\[
(\mathbf{J}_N\mathbf{T}_N)_{jk}=(\mathbf{T}_N)_{N-j,k}=c_{(N-j)-k},
\qquad
(\mathbf{T}_N\mathbf{J}_N)_{jk}=(\mathbf{T}_N)_{j,N-k}=c_{j-(N-k)}.
\]
注意到 \(j-(N-k)=-( (N-j)-k)\)，并由 \(c_{-n}=c_n\) 得 \((\mathbf{J}_N\mathbf{T}_N)_{jk}=(\mathbf{T}_N\mathbf{J}_N)_{jk}\)，故 \(\mathbf{J}_N\mathbf{T}_N=\mathbf{T}_N\mathbf{J}_N\)。
由于 \(\mathbf{J}_N\) 为对称正交对合（\(\mathbf{J}_N^{\mathsf T}=\mathbf{J}_N\)、\(\mathbf{J}_N^2=I\)），其 \(\pm 1\) 特征子空间 \(E_N^{\pm}\) 构成 \(\RR^{N+1}\) 的正交直和；又因对易性，\(\mathbf{T}_N\) 保持 \(E_N^{\pm}\) 不变，故取 \(Q_N\) 的列向量为 \(E_N^{+}\) 与 \(E_N^{-}\) 的一组适配正交基即可得到块对角形式。
\par
最后设 \(x=x_{+}+x_{-}\) 为相应分解（\(x_{\pm}\in E_N^{\pm}\)）。由 \(\mathbf{T}_N\) 的对称性与对易性有
\[
x_{+}^{\mathsf T}\mathbf{T}_N x_{-}
=
x_{+}^{\mathsf T}\mathbf{J}_N\mathbf{T}_N\mathbf{J}_N x_{-}
=
x_{+}^{\mathsf T}\mathbf{T}_N(-x_{-})
=
-\,x_{+}^{\mathsf T}\mathbf{T}_N x_{-},
\]
从而交叉项为零并得到二次型分裂
\(
x^{\mathsf T}\mathbf{T}_N x=x_{+}^{\mathsf T}\mathbf{T}_N x_{+}+x_{-}^{\mathsf T}\mathbf{T}_N x_{-}
\)。
于是 \(\mathbf{T}_N\succeq 0\) 当且仅当其在 \(E_N^{+}\) 与 \(E_N^{-}\) 上的限制二次型分别半正定；等价地，\(\mathbf{T}_N^{\pm}\succeq 0\)。证毕。
\end{proof}

\begin{corollary}[Toeplitz--PSD 层级的偶/奇证书分裂与失败扇区定位]\label{cor:app-horizon-toeplitz-z2-cert-split}
\leanverified{paper\_app\_horizon\_toeplitz\_z2\_cert\_split}
在命题 \ref{prop:app-horizon-toeplitz-z2-splitting} 的实结构假设下，Toeplitz--PSD 层级
\(\{\mathbf{T}_N\succeq 0\}_{N\ge 0}\)
可分裂为两条相互独立的 PSD 子层级：
\[
\forall N\ge 0,\qquad
\mathbf{T}_N\succeq 0
\quad\Longleftrightarrow\quad
\mathbf{T}_N^{+}\succeq 0\ \text{且}\ \mathbf{T}_N^{-}\succeq 0.
\]
因此，若某一级 \(N\) 上出现 PSD 失败，则存在最小 \(N_\star\) 与扇区符号 \(\epsilon\in\{+,-\}\)，使得 \(\mathbf{T}_{N_\star}^{\epsilon}\) 出现负特征值；该 \((N_\star,\epsilon)\) 给出 Toeplitz 证书在有限级别上的可审计失败定位。
\end{corollary}

\paragraph{系数的局部封装：仅用 \texorpdfstring{$\Xi_{\zeta}$}{Xi_zeta} 在 \texorpdfstring{$s=-\tfrac12$}{s=-1/2} 的导数}
由于 \(s(0)=-\tfrac12\) 且 \(\Xi_{\zeta}(-\tfrac12)\neq 0\)，可在该点定义对数导数的泰勒系数。
为书写简明，令
\[
\ell_k:=\left.\frac{\dd^k}{\dd s^k}\log\Xi_{\zeta}(s)\right|_{s=-1/2}\in\RR
\qquad(k\ge 1).
\]
由 \(\mathscr{C}_{\zeta}(w)=-\bigl(\log\Xi_{\zeta}\bigr)'\!\bigl(s(w)\bigr)\) 与恒等式
\begin{equation}\label{eq:app-horizon-s-of-w-shift}
s(w)=-\tfrac12+\frac{2w}{1+w}
\end{equation}
可得到 \(\mathscr{C}_{\zeta}\) 的系数完全由有限阶 \(\ell_k\) 显式给出。

\begin{proposition}[前几阶系数的显式表达]\label{prop:app-horizon-coeff-local}
在 \(\mathscr{C}_{\zeta}(w)=c_0+2\sum_{n\ge 1}c_n w^n\) 的归一化下，有
\begin{equation}\label{eq:app-horizon-c0c3}
\begin{aligned}
c_0&=-\ell_1,\\
c_1&=-\ell_2,\\
c_2&=\ell_2-\ell_3,\\
c_3&=-\ell_2+2\ell_3-\tfrac23\ell_4.
\end{aligned}
\end{equation}
并且一般地，\(c_n\) 是 \((\ell_2,\dots,\ell_{n+1})\) 的显式有理线性组合（由 \(\frac{2w}{1+w}=\sum_{m\ge 1}2(-1)^{m-1}w^m\) 代入得到）。
\leanverified{paper\_app\_horizon\_coeff\_local}
\end{proposition}

\begin{proof}
由 \eqref{eq:app-horizon-s-of-w-shift}，令 \(u(w):=s(w)+\tfrac12=\frac{2w}{1+w}=2w-2w^2+2w^3+O(w^4)\)。
在 \(s=-\tfrac12\) 处对 \( (\log\Xi_{\zeta})'(s) \) 作泰勒展开：
\[
\bigl(\log\Xi_{\zeta}\bigr)'\!\left(-\tfrac12+u\right)
=\ell_1+\ell_2 u+\tfrac12\ell_3 u^2+\tfrac16\ell_4 u^3+O(u^4).
\]
代入 \(u=u(w)\) 并取负号得到 \(\mathscr{C}_{\zeta}(w)\) 的 \(w\)-展开；
对照 \(\mathscr{C}_{\zeta}(w)=c_0+2c_1w+2c_2w^2+2c_3w^3+O(w^4)\) 即得 \eqref{eq:app-horizon-c0c3}。
一般形式同理由复合幂级数唯一性推出。证毕。
\end{proof}

\begin{remark}[低阶 PSD 约束的显式必要条件]\label{rem:app-horizon-low-level-psd}
由定理 \ref{thm:app-horizon-toeplitz-lmi}，RH \(\Rightarrow\) \(\mathbf{T}_N\succeq 0\)（对一切 \(N\)）。
其前两级分别给出
\[
\mathbf{T}_0=(c_0)\succeq 0,
\qquad
\mathbf{T}_1=\begin{pmatrix}c_0&c_1\\c_1&c_0\end{pmatrix}\succeq 0,
\]
即 \(c_0\ge 0\) 与 \(c_0^2-c_1^2\ge 0\)。
将 \eqref{eq:app-horizon-c0c3} 代入可得以 \(\ell_1,\ell_2\) 表示的显式不等式；更高阶约束同理可由有限阶 \(\ell_k\) 逐级生成。
这些约束本身并不足以推出 RH，但它们提供了一个完全内部的、可逐级复核的必要条件族。
\end{remark}

\paragraph{\texorpdfstring{$|w|<1/3$}{|w|<1/3} 欧拉积可审计补丁：把算术项隔离到绝对收敛域}
记
\begin{equation}\label{eq:app-horizon-sigma-def}
\boxed{\;\sigma(w):=1-s(w)=\frac{3-w}{2(1+w)}\; }.
\end{equation}
由函数方程 \(\Xi_{\zeta}(s)=\Xi_{\zeta}(1-s)\) 得对数导数反对称性
\(
\frac{\Xi_{\zeta}'(s)}{\Xi_{\zeta}(s)}=-\frac{\Xi_{\zeta}'(1-s)}{\Xi_{\zeta}(1-s)}
\)，
从而
\begin{equation}\label{eq:app-horizon-fe-rewrite}
\boxed{\;
\mathscr{C}_{\zeta}(w)
=-\frac{\Xi_{\zeta}'(s(w))}{\Xi_{\zeta}(s(w))}
=\frac{\Xi_{\zeta}'(\sigma(w))}{\Xi_{\zeta}(\sigma(w))}\; }.
\end{equation}

\begin{theorem}[\texorpdfstring{$|w|<1/3$}{|w|<1/3} 的欧拉积补丁（绝对收敛素数展开）]\label{thm:app-horizon-euler-patch}
\leanverified{paper\_app\_horizon\_euler\_patch}
若 \(|w|<\frac13\)，则 \(\Re\sigma(w)>1\)。
因此在该子圆盘内，\(\mathscr{C}_{\zeta}\) 具有一致绝对收敛表示
\begin{equation}\label{eq:app-horizon-euler-patch}
\boxed{\;
\mathscr{C}_{\zeta}(w)=\frac{1}{\sigma(w)}+\frac{1}{\sigma(w)-1}
-\frac12\log\pi+\frac12\,\psi\!\left(\frac{\sigma(w)}{2}\right)
-\sum_{n\ge2}\frac{\Lambda(n)}{n^{\sigma(w)}},\qquad \sigma(w)=1-s(w),\;}
\end{equation}
其中 \(\psi=\Gamma'/\Gamma\) 为 digamma 函数，\(\Lambda\) 为 von Mangoldt 函数；并且级数在任意闭子圆盘 \(|w|\le r<1/3\) 上一致绝对收敛。
\end{theorem}

\begin{proof}
由 \eqref{eq:app-horizon-sigma-def}，\(\sigma\) 是具有实系数的 Möbius 变换，故把圆 \(|w|=r\) 映为一条关于实轴对称的圆；其 \(\Re\)-最小值在该像圆的最左端取得，而最左端必位于实轴上。
于是对固定 \(r\in(0,1)\) 有
\[
\min_{|w|=r}\Re\sigma(w)=\min\{\sigma(r),\sigma(-r)\}=\sigma(r)=\frac{3-r}{2(1+r)}.
\]
当 \(r<1/3\) 时右端严格大于 \(1\)，从而 \(|w|<1/3\Rightarrow \Re\sigma(w)>1\)。
\par
由 \eqref{eq:app-horizon-fe-rewrite}，\(\mathscr{C}_{\zeta}(w)=\frac{\Xi_{\zeta}'(\sigma(w))}{\Xi_{\zeta}(\sigma(w))}\)。
将 \(\Xi_{\zeta}(s)=\frac12\,s(s-1)\pi^{-s/2}\Gamma(\tfrac{s}{2})\zeta(s)\) 取对数导数，得到
\[
\frac{\Xi_{\zeta}'(s)}{\Xi_{\zeta}(s)}
=\frac{1}{s}+\frac{1}{s-1}-\frac12\log\pi+\frac12\,\psi\!\left(\frac{s}{2}\right)+\frac{\zeta'}{\zeta}(s).
\]
当 \(\Re s>1\) 时有标准恒等式
\(
\frac{\zeta'}{\zeta}(s)=-\sum_{n\ge 2}\frac{\Lambda(n)}{n^s}
\)，
且右端绝对收敛。
将 \(s=\sigma(w)\) 代入并合并即得 \eqref{eq:app-horizon-euler-patch}。
一致绝对收敛由 \(|w|\le r<1/3\Rightarrow \inf\Re\sigma(w)>1\) 与 Dirichlet 级数在半平面 \(\Re s\ge 1+\varepsilon\) 的一致绝对收敛性得到。证毕。
\end{proof}

\begin{corollary}[系数的素数矩表示（可审计接口）]\label{cor:app-horizon-prime-moment-interface}
\leanverified{paper\_app\_horizon\_prime\_moment\_interface}
在 \(|w|<1/3\) 内，\(\mathscr{C}_{\zeta}\) 的 Taylor 系数 \(c_n\) 可通过对 \eqref{eq:app-horizon-euler-patch} 在 \(w=0\) 处逐项求导得到。
其中算术部分的贡献完全由收敛和式
\[
\sum_{n\ge2}\frac{\Lambda(n)(\log n)^k}{n^{3/2}}
\qquad(k\ge 0)
\]
及有限个有理组合构成（源于 \(\sigma(0)=3/2\) 与 \(\sigma(w)=\frac{3-w}{2(1+w)}\) 的链式展开）。
因此，每一有限级 Toeplitz 主子矩阵 \(\mathbf{T}_N\) 的构造可在绝对收敛域内逐项审计；困难被结构性地隔离到“将局部补丁的正性跨越到全圆盘并保持 \(\Re\mathscr{C}_{\zeta}\ge 0\)”这一端点延拓问题。
\end{corollary}

\begin{remark}[视界多极矩与反射正性（与“毛发”语义的对齐）]\label{rem:app-horizon-multipole-positivity-hair}
Herglotz 表示 \eqref{eq:app-horizon-herglotz-disk} 中的测度 \(\mu\) 可被视作“视界谱测度”：
其 Fourier 矩 \(\{c_n\}\) 给出一组视界多极矩（亦可理解为“可见毛发系数”），而 Toeplitz PSD 条件 \(\mathbf{T}_N\succeq 0\) 要求该矩序列必须来自某个正测度（即来自真正的 Hilbert 空间谱对象）。
若存在盘内零点（等价于 RH 失败），则 \(\mathscr{C}_{\zeta}\) 在盘内出现极点并失去 Carath\'eodory 正性；在矩层级上，这必然迫使某一级 \(\mathbf{T}_N\) 出现负特征值，从而把“非单元性/反射正性破缺”压缩为一个有限维可审计失败见证。
\end{remark}
\paragraph{盘内极点的系数印迹与 Toeplitz--PSD 的检出门槛}
本段把“盘内极点”对 \(\{c_n\}\) 的主阶贡献写成闭式，并据此给出一个与端点压缩相容的、可审计的 PSD 检出门槛：该门槛把“极端共振”（端点薄层中的离切片信息）转写为“需要多高阶矩/多极矩才能稳定触发有限维否证器”的尺度律。

\begin{proposition}[盘内极点的系数印迹（主部与 Taylor 主项）]\label{prop:app-horizon-pole-imprint}
\leanverified{paper\_app\_horizon\_pole\_imprint}
设 \(\rho=\beta+\mathrm{i}\gamma\) 为 \(\Xi_{\zeta}\) 的简单零点且 \(\beta<\tfrac12\)。
令 \(\delta:=\tfrac12-\beta>0\)、\(z_\rho:=\gamma+\mathrm{i}\delta\in\mathbb{H}\)，并令 \(w_\rho:=\mathcal{C}(z_\rho)\in\mathbb{D}\) 为其圆盘像（Cayley 变换 \(\mathcal{C}\) 见式 \eqref{eq:app-cayley-disk}）。
则 \(\mathscr{C}_{\zeta}\) 在 \(w=w_\rho\) 处具有简单极点，并可写为
\begin{equation}\label{eq:app-horizon-pole-principal-part}
\boxed{\;
\mathscr{C}_{\zeta}(w)
=-\frac{(1+w_\rho)^2}{2}\cdot\frac{1}{w-w_\rho}+H_\rho(w),\;}
\end{equation}
其中 \(H_\rho\) 在 \(\mathbb{D}\) 内解析。
将 \(H_\rho\) 在 \(w=0\) 处按同一归一化幂级数展开
\[
H_\rho(w)=h_0+2\sum_{n\ge 1}h_n w^n,
\]
并沿用 \(\mathscr{C}_{\zeta}(w)=c_0+2\sum_{n\ge 1}c_n w^n\)（式 \eqref{eq:app-horizon-carath-series}），则系数满足
\[
c_0=\frac{(1+w_\rho)^2}{2w_\rho}+h_0,
\qquad
c_n=\frac{(1+w_\rho)^2}{4w_\rho^{n+1}}+h_n\quad(n\ge 1).
\]
\end{proposition}

\begin{proof}
由定义 \eqref{eq:app-horizon-carath-def}，\(\mathscr{C}_{\zeta}(w)=-\bigl(\log\Xi_{\zeta}\bigr)'\!\bigl(s(w)\bigr)\)。
若 \(\rho\) 为 \(\Xi_{\zeta}\) 的简单零点，则 \(\bigl(\log\Xi_{\zeta}\bigr)'(s)\) 在 \(s=\rho\) 处具有留数 \(1\) 的简单极点，从而 \(\mathscr{C}_{\zeta}\) 在 \(w=w_\rho\) 处具有留数 \(-1\) 的简单极点。
用链式法则在 \(w_\rho\) 处线性化 \(s(w)=s(w_\rho)+s'(w_\rho)(w-w_\rho)+\cdots\)，得到主部
\[
\mathscr{C}_{\zeta}(w)
=-\frac{1}{s'(w_\rho)}\cdot\frac{1}{w-w_\rho}+O(1).
\]
而由 \eqref{eq:app-horizon-s-of-w-shift} 有 \(s'(w)=\frac{2}{(1+w)^2}\)，故 \(-1/s'(w_\rho)=-(1+w_\rho)^2/2\)，从而得到 \eqref{eq:app-horizon-pole-principal-part}。
\par
将 \eqref{eq:app-horizon-pole-principal-part} 的极点项在 \(w=0\) 处作几何级数展开：
\[
-\frac{(1+w_\rho)^2}{2}\cdot\frac{1}{w-w_\rho}
=\frac{(1+w_\rho)^2}{2}\sum_{n\ge 0}\frac{w^n}{w_\rho^{n+1}},
\]
再与 \(H_\rho(w)=h_0+2\sum_{n\ge 1}h_n w^n\) 合并，并对照 \(\mathscr{C}_{\zeta}(w)=c_0+2\sum_{n\ge 1}c_n w^n\) 的系数即可得到所示结论。证毕。
\end{proof}

\begin{theorem}[共振品质因数与 Toeplitz--PSD 的检出门槛]\label{thm:app-horizon-toeplitz-detection-threshold}
\leanverified{paper\_app\_horizon\_toeplitz\_detection\_threshold}
在命题 \ref{prop:app-horizon-pole-imprint} 的设定下，定义共振品质因数
\begin{equation}\label{eq:app-horizon-Qrho-def}
\boxed{\;
Q_\rho:=\frac{1}{1-|w_\rho|^2}
=\frac{\gamma^2+(1+\delta)^2}{4\delta}\;}
\end{equation}
（后一等式见推论 \ref{cor:app-offcritical-horocycle-busemann}）。
假设余项系数满足一份可审计上界 \(|h_n|\le H\)（对所有 \(n\ge 0\) 成立）。
\par
若希望仅通过有限级 Toeplitz--PSD 核验来\emph{排除}该盘内极点（等价地：要求 \(\mathbf{T}_N\succeq 0\) 在某个足够大的截断阶 \(N\) 上仍成立，从而拒绝“有限维失败见证”），则该核验阶数必须满足下界
\begin{equation}\label{eq:app-horizon-toeplitz-detect-threshold}
\boxed{\;
N\ \ge\ C^{-1}\,Q_\rho\,\log\!\bigl(1+\delta Q_\rho\bigr)\;},
\end{equation}
其中常数 \(C>0\) 仅依赖于 \(c_0\) 与 \(H\)（不依赖于 \(\gamma,\delta\)）。
特别地，当 \(|\gamma|\to\infty\) 且 \(\delta\) 有界时，\(\delta Q_\rho\asymp \gamma^2\)，故该下界等价于
\[
N\ \gtrsim\ \frac{\gamma^2}{\delta}\,\log(1+\gamma^2).
\]
\par
反向表述：若对某个 \(N\) 已核验 \(\mathbf{T}_0,\dots,\mathbf{T}_N\succeq 0\)，则任何可能存在的离临界线零点 \(\rho=\beta+\mathrm{i}\gamma\) 必满足
\[
\delta\ \le\ C\,\frac{\gamma^2\,\log(1+\gamma^2)}{N},
\]
从而在高高度区间 \(|\gamma|\gg 1\) 上，固定阶的 Toeplitz--PSD 核验只能排除“偏移不进入端点薄层”的离线零点。
\end{theorem}

\begin{proof}
由定理 \ref{thm:app-horizon-toeplitz-lmi}，\(\mathbf{T}_N\succeq 0\) 蕴含对任意 \(1\le n\le N\) 其 \(\{0,n\}\) 主子矩阵
\(
\begin{psmallmatrix}
c_0 & c_n\\
c_{-n} & c_0
\end{psmallmatrix}
\)
半正定。
由于 \(c_{-n}=\overline{c_n}\)，该条件等价于
\begin{equation}\label{eq:app-horizon-2x2-necessary}
|c_n|\le c_0,\qquad 1\le n\le N.
\end{equation}
另一方面，由命题 \ref{prop:app-horizon-pole-imprint} 与 \(|h_n|\le H\) 得到对任意 \(n\ge 1\) 的下界
\begin{equation}\label{eq:app-horizon-cn-lowerbound}
|c_n|
\ge
\frac{|1+w_\rho|^2}{4\,|w_\rho|^{n+1}}-H.
\end{equation}
因此若 \(\mathbf{T}_N\succeq 0\) 成立，则对 \(n=N\) 合并 \eqref{eq:app-horizon-2x2-necessary}--\eqref{eq:app-horizon-cn-lowerbound} 得
\[
\frac{|1+w_\rho|^2}{4\,|w_\rho|^{N+1}}\le c_0+H.
\]
由推论 \ref{cor:app-offcritical-horocycle-busemann} 的恒等式
\(
|1+w_\rho|^2=(1-|w_\rho|^2)/\delta
\)
与定义 \(Q_\rho=(1-|w_\rho|^2)^{-1}\) 得 \(|1+w_\rho|^2=(\delta Q_\rho)^{-1}\)，从而
\begin{equation}\label{eq:app-horizon-wrho-growth-bound}
|w_\rho|^{-(N+1)}\le 4(c_0+H)\,\delta Q_\rho.
\end{equation}
令 \(r:=|w_\rho|^2=1-1/Q_\rho\in(0,1)\)。
对 \(r\in(0,1)\) 有标准不等式
\(
1-r\le -\log r \le \frac{1-r}{r}
\)，
于是当 \(Q_\rho>1\) 时
\begin{equation}\label{eq:app-horizon-logw-sandwich}
\frac{1}{2Q_\rho}\ \le\ -\log|w_\rho|\ \le\ \frac{1}{2(Q_\rho-1)}.
\end{equation}
将 \eqref{eq:app-horizon-wrho-growth-bound} 取对数并用 \eqref{eq:app-horizon-logw-sandwich} 的右端估计（使左侧增长最慢，从而得到必要的“检出门槛”下界），得到存在常数 \(C>0\) 仅依赖于 \(c_0,H\) 使
\[
N+1
\ge
C^{-1}\,Q_\rho\,\log\!\bigl(1+\delta Q_\rho\bigr),
\]
即得到 \eqref{eq:app-horizon-toeplitz-detect-threshold}（把 \(\log(4(c_0+H)\delta Q_\rho)\) 吸收进 \(\log(1+\delta Q_\rho)\) 与常数即可）。
\par
最后由 \eqref{eq:app-horizon-Qrho-def} 的闭式与 \(\delta Q_\rho\asymp \gamma^2\)（\(|\gamma|\to\infty\)、\(\delta\) 有界）得到所示渐近形式；反向表述由 \(Q_\rho\asymp \gamma^2/\delta\) 代回 \eqref{eq:app-horizon-toeplitz-detect-threshold} 并整理即得。证毕。
\end{proof}

\begin{remark}[失败见证的端点聚焦（角分辨率口径）]\label{rem:app-horizon-toeplitz-witness-endpoint-focus}
Toeplitz--PSD 的有限级否证器等价于在 \(\deg\le N\) 的三角多项式/OPUC 截断内找到一条使二次型为负的方向（推论 \ref{cor:app-horizon-toeplitz-z2-cert-split} 给出其偶/奇扇区定位）。
在命题 \ref{prop:app-horizon-pole-imprint} 的几何语义中，盘内极点对应于端点 \(w=-1\) 邻域的薄层压缩：当 \(|\gamma|\) 大时有 \(\arg w_\rho\approx\pi\) 且 \(\delta Q_\rho=|1+w_\rho|^{-2}\asymp \gamma^2\)（推论 \ref{cor:app-offcritical-horocycle-busemann}）。
因此，若否证器在阶 \(N\) 上首次出现，其“角分辨率”自然受 \(N^{-1}\) 量级约束，并倾向于在 \(\theta\approx\pi\) 的端点角窗内聚焦；这与定理 \ref{thm:app-horizon-endpoint-window-mass} 的端点窗质量渐近以及端点紧化的可见化机制相一致。
\end{remark}
