\begin{proposition}[端点角窗律：高虚部阈值的窗口测度]\label{prop:app-jensen-endpoint-angle-window}
\leanverified{paper\_app\_jensen\_endpoint\_angle\_window}
沿用命题 \ref{prop:app-jensen-defect-upperhalf-circle-average} 的记号，令
\[
y(\varrho,\theta):=\Im\,t(\varrho,\theta)=\frac{1-\varrho^2}{1+\varrho^2+2\varrho\cos\theta},
\qquad 0<\varrho<1,\ \theta\in[0,2\pi).
\]
对任意阈值 \(Y>0\) 定义角窗
\[
A_Y(\varrho):=\{\theta\in[0,2\pi): y(\varrho,\theta)\ge Y\}.
\]
并记
\(
y_{-}(\varrho)=\frac{1-\varrho}{1+\varrho}
\)
与
\(
y_{+}(\varrho)=\frac{1+\varrho}{1-\varrho}
\)
（命题 \ref{prop:app-cayley-preimage-modulus-circle}）。
则成立分段测度公式：
\begin{enumerate}
  \item 若 \(0<Y\le y_{-}(\varrho)\)，则 \(A_Y(\varrho)=[0,2\pi)\)，从而 \(\mathrm{meas}\,A_Y(\varrho)=2\pi\)；
  \item 若 \(y_{-}(\varrho)<Y<y_{+}(\varrho)\)，则 \(A_Y(\varrho)\) 为以 \(\pi\) 为中心的对称区间，且
  \[
  \boxed{\;
  \mathrm{meas}\,A_Y(\varrho)=2\Bigl(\pi-\arccos\bigl(c_Y(\varrho)\bigr)\Bigr),\qquad
  c_Y(\varrho):=\frac{(1-\varrho^2)-Y(1+\varrho^2)}{2\varrho Y}\; }.
  \]
  \item 若 \(Y\ge y_{+}(\varrho)\)，则 \(A_Y(\varrho)=\varnothing\)，从而 \(\mathrm{meas}\,A_Y(\varrho)=0\)。
\end{enumerate}
此外，当 \(\varrho=1-\eta\) 且 \(\eta\downarrow 0\)（等价于 \(\varrho\uparrow 1\)）并且 \(Y>0\) 固定时，有边界层渐近
\[
\boxed{\;
\mathrm{meas}\,A_Y(\varrho)
=2\sqrt{\frac{2\eta}{Y}}\ \bigl(1+o(1)\bigr)\qquad(\eta\downarrow 0)\; }.
\]
\end{proposition}

\begin{proof}
由 \(y(\varrho,\theta)\ge Y\) 等价于
\[
1-\varrho^2\ \ge\ Y(1+\varrho^2+2\varrho\cos\theta)
\quad\Longleftrightarrow\quad
\cos\theta\ \le\ c_Y(\varrho).
\]
由 \(y(\varrho,0)=y_{-}(\varrho)\)、\(y(\varrho,\pi)=y_{+}(\varrho)\)（命题 \ref{prop:app-cayley-preimage-modulus-circle}）可知：
当 \(Y\le y_{-}(\varrho)\) 时上式对所有 \(\theta\) 成立；当 \(Y\ge y_{+}(\varrho)\) 时无解；而在中间情形 \(y_{-}(\varrho)<Y<y_{+}(\varrho)\) 时 \(c_Y(\varrho)\in(-1,1)\)，解集为 \(\theta\) 以 \(\pi\) 为中心的对称区间。
由于集合 \(\{\theta\in[0,2\pi):\cos\theta\ge c\}\) 的测度为 \(2\arccos(c)\)，取补集得到
\(\mathrm{meas}\,A_Y(\varrho)=2\pi-2\arccos(c_Y(\varrho))\)，即所示闭式。
\par
最后令 \(\varrho=1-\eta\) 且 \(\eta\downarrow 0\)。对固定 \(Y>0\) 有
\(
c_Y(\varrho)=-1+\eta/Y+O(\eta^2)
\)。
并用端点展开 \(\arccos(-1+\epsilon)=\pi-\sqrt{2\epsilon}+o(\sqrt{\epsilon})\)（\(\epsilon\downarrow 0\)），得到
\[
\mathrm{meas}\,A_Y(\varrho)
=2\bigl(\pi-\arccos(c_Y(\varrho))\bigr)
=2\sqrt{\frac{2\eta}{Y}}\,(1+o(1)).
\]
证毕。
\end{proof}

\begin{theorem}[端点角窗稀有指数与 min-entropy 端点常数的闭式匹配]\label{thm:app-endpoint-window-hinfty-match}
令 \(h_\infty=\log(2/\sqrt{\varphi})\) 为折叠输出测度的 min-entropy 熵率端点常数（推论 \ref{cor:pom-min-entropy-rate}）。
取常数 \(c>0\)，并对 \(m\to\infty\) 定义
\[
Y_m:=\Bigl(\frac{4}{\varphi^{2}}\Bigr)^{m},
\qquad
\varrho_m:=1-\eta_m,\qquad \eta_m:=c\,\varphi^{-m}.
\]
则当 \(m\to\infty\) 时有指数匹配
\[
\mathrm{meas}\,A_{Y_m}(\varrho_m)
=2\sqrt{2c}\,e^{-m h_\infty}\,\bigl(1+o(1)\bigr).
\]
等价地，若 \(\Theta\sim\mathrm{Unif}[0,2\pi)\)，则稀有事件概率满足
\[
\mathbb{P}\!\left(\Im\,t(\varrho_m,\Theta)\ge Y_m\right)
=\frac{1}{2\pi}\,\mathrm{meas}\,A_{Y_m}(\varrho_m)
=\frac{\sqrt{2c}}{\pi}\,e^{-m h_\infty}\,\bigl(1+o(1)\bigr).
\]
\leanverified{paper\_app\_endpoint\_window\_hinfty\_match}
\end{theorem}

\begin{proof}
由定义 \(y_\pm(\varrho)=\frac{1\pm\varrho}{1\mp\varrho}\) 与 \(\eta_m=c\varphi^{-m}\) 得
\[
y_-(\varrho_m)\asymp \eta_m,\qquad
y_+(\varrho_m)\asymp \eta_m^{-1}\asymp \varphi^{m}.
\]
另一方面 \(Y_m=(4/\varphi^2)^m\)，且
\(
Y_m/y_+(\varrho_m)\asymp (4/\varphi^3)^m\to 0
\)，故当 \(m\) 充分大时必有 \(y_-(\varrho_m)<Y_m<y_+(\varrho_m)\)，可使用命题 \ref{prop:app-jensen-endpoint-angle-window} 的中间情形闭式：
\[
\mathrm{meas}\,A_{Y_m}(\varrho_m)
=2\Bigl(\pi-\arccos\bigl(c_{Y_m}(\varrho_m)\bigr)\Bigr),
\qquad
c_Y(\varrho):=\frac{(1-\varrho^2)-Y(1+\varrho^2)}{2\varrho Y}.
\]
令 \(\varrho_m=1-\eta_m\)。直接展开得
\[
c_{Y_m}(\varrho_m)
=-1+\frac{\eta_m}{Y_m}+O(\eta_m^2),
\]
且由于
\(
\eta_m^2/(\eta_m/Y_m)=\eta_m Y_m=c(4/\varphi^3)^m\to 0
\)，故亦可写为
\(
c_{Y_m}(\varrho_m)=-1+\frac{\eta_m}{Y_m}\,(1+o(1))
\)。
再由端点展开 \(\arccos(-1+\epsilon)=\pi-\sqrt{2\epsilon}+o(\sqrt\epsilon)\)（\(\epsilon\downarrow 0\)），得到
\[
\mathrm{meas}\,A_{Y_m}(\varrho_m)
=2\sqrt{\frac{2\eta_m}{Y_m}}\,(1+o(1)).
\]
代入 \(\eta_m=c\varphi^{-m}\)、\(Y_m=(4/\varphi^2)^m\) 得
\[
\mathrm{meas}\,A_{Y_m}(\varrho_m)
=2\sqrt{2c}\Bigl(\frac{\sqrt{\varphi}}{2}\Bigr)^m(1+o(1))
=2\sqrt{2c}\,e^{-m\log(2/\sqrt{\varphi})}(1+o(1)).
\]
最后一步用 \(h_\infty=\log(2/\sqrt{\varphi})\)（推论 \ref{cor:pom-min-entropy-rate}）即得结论。
\end{proof}

\begin{remark}[window-\(6\) 锚点的端点角窗读出]\label{rem:app-endpoint-window-window6}
取 \(m=6\) 时，上述定理给出端点角窗在阈值 \(Y_6=(4/\varphi^2)^6\)、半径 \(\varrho_6=1-c\varphi^{-6}\) 下的稀有测度指数率
\[
\mathrm{meas}\,A_{Y_6}(\varrho_6)
=2\sqrt{2c}\,e^{-6h_\infty}\,\bigl(1+o(1)\bigr),
\qquad
\mathbb{P}\!\left(\Im\,t(\varrho_6,\Theta)\ge Y_6\right)
=\frac{\sqrt{2c}}{\pi}\,e^{-6h_\infty}\,\bigl(1+o(1)\bigr).
\]
因此在 window-\(6\) 的最小通道锚点处，min-entropy 端点常数 \(h_\infty\) 可被直接转译为端点视界角窗稀有事件的指数衰减率。
\end{remark}

\begin{theorem}[缺陷上界推出离线偏移的显式上界]\label{thm:app-jensen-defect-controls-offcritical-shift}
\leanverified{paper\_app\_jensen\_defect\_controls\_offcritical\_shift}
固定 \(0<\varrho<1\)。
若存在 \(\varepsilon\ge 0\) 使 \(D_{\zeta}(\varrho)\le \varepsilon\)，令
\[
r_0:=\varrho e^{-\varepsilon}\in(0,1).
\]
则对任意满足 \(\beta<\tfrac12\) 的零点 \(\rho=\beta+\mathrm{i}\gamma\)（若存在），其离线偏移 \(\delta:=\tfrac12-\beta>0\) 与圆盘半径 \(r=|w_{\rho}|\)（命题 \ref{prop:app-offcritical-radius-compression}）满足
\[
r=|w_{\rho}|\ \ge\ r_0.
\]
并且在满足
\[
1-\frac{(1-r_0^2)^2}{(1+r_0^2)^2}\,(\gamma^2+1)\ \ge\ 0
\]
（从而下述根式为实数）的条件下，有显式上界
\[
\boxed{\;
\delta\ \le\ \Delta(\gamma;r_0),\qquad
\Delta(\gamma;r)
:=\frac{1+r^{2}}{1-r^{2}}
\left(1-\sqrt{1-\frac{(1-r^{2})^{2}}{(1+r^{2})^{2}}(\gamma^{2}+1)}\right)\; }.
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:app-jensen-visible-threshold-offcritical}，当 \(|w_{\rho}|<\varrho\) 时有
\(
D_{\zeta}(\varrho)\ge \log(\varrho/|w_{\rho}|)
\)；
与 \(D_{\zeta}(\varrho)\le \varepsilon\) 合并得到 \(|w_{\rho}|\ge \varrho e^{-\varepsilon}=r_0\)。
当 \(|w_{\rho}|\ge \varrho\) 时上述下界不适用，但此时 \(|w_{\rho}|\ge \varrho\ge r_0\) 仍成立，故总有 \(r\ge r_0\)。
\par
另一方面，命题 \ref{prop:app-offcritical-radius-compression} 给出闭式关系
\[
r^{2}=\frac{\gamma^{2}+(\delta-1)^{2}}{\gamma^{2}+(\delta+1)^{2}}.
\]
对固定 \(\gamma\)，上述右端关于 \(\delta\) 在 \([0,\sqrt{\gamma^{2}+1}]\) 上严格递减，故 \(r\ge r_0\) 导出 \(\delta\le \delta_0\)，其中 \(\delta_0\) 为方程在 \(r=r_0\) 时的（较小）实根。
将该方程对 \(\delta\) 显式求解（取较小根）即得到所示 \(\Delta(\gamma;r_0)\)。证毕。
\end{proof}

\begin{corollary}[边界层展开：\texorpdfstring{$r_0\uparrow 1$}{r0->1} 时的主阶上界]\label{cor:app-jensen-offcritical-shift-boundary}
在定理 \ref{thm:app-jensen-defect-controls-offcritical-shift} 的条件下，若进一步满足 \(r_0\uparrow 1\) 且 \(\gamma\) 固定，则有渐近展开
\[
\Delta(\gamma;r_0)
=\frac{\gamma^2+1}{2}\cdot\frac{1-r_0^2}{1+r_0^2}
+O\!\left((\gamma^2+1)^2(1-r_0^2)^2\right)
\qquad(r_0\uparrow 1).
\]
\end{corollary}

\begin{proof}
在 \(r_0\uparrow 1\) 时，
\(
q:=\frac{(1-r_0^2)^2}{(1+r_0^2)^2}(\gamma^2+1)\to 0
\)。
将
\(
\sqrt{1-q}=1-\frac{q}{2}+O(q^2)
\)
代入 \(\Delta(\gamma;r_0)\) 的闭式，并整理即可得到所示展开。证毕。
\end{proof}
