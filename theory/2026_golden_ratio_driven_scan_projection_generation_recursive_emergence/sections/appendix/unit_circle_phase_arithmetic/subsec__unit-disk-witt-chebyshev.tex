\subsection{与 Witt--Chebyshev--Adams 桥包的对齐：幂半群 \texorpdfstring{$w\mapsto w^m$}{w->w(m)} 与对数主尺度}\label{subsec:unit-disk-witt-chebyshev}
本文在 unitary slice 上已反复使用“倍角/幂映射 \(\leftrightarrow\) Chebyshev--Adams”桥（见 \S\ref{subsec:xi-from-self-dual} 与注 \ref{rem:xi-joukowski-toy-rh-alignment} 的 \(y\mapsto y+y^{-1}\) 压缩）。
在本附录的圆盘参数 \(w\) 下，该桥可以以完全无参形式写出。

\paragraph{边界倍角与 Chebyshev 多项式}
当 \(w=e^{\mathrm{i}\theta}\in\TT\) 时，令
\[
\mathfrak{r}:=e^{\mathrm{i}\theta/2},\qquad S:=\mathfrak{r}+\mathfrak{r}^{-1}=2\cos(\theta/2)\in[-2,2].
\]
则对任意整数 \(m\ge 1\)，有
\[
w^m=e^{\mathrm{i}m\theta},\qquad
\mathfrak{r}^m+\mathfrak{r}^{-m}=2\cos(m\theta/2)=:C_m(S),
\]
其中 \(C_m\in\ZZ[S]\) 为 Chebyshev--Adams 多项式表示（参见 \cite{MasonHandscomb2003Chebyshev}）。
因此幂半群 \(w\mapsto w^m\) 在 \(S\)-坐标中恰对应 \(S\mapsto C_m(S)\) 的代数化倍角运算。

\paragraph{Chebyshev--Frobenius 差额与 \texorpdfstring{$p$}{p}-导数}
令 \(p\) 为素数。由 \(C_p(S)=\mathfrak{r}^p+\mathfrak{r}^{-p}\) 与二项式系数的 \(p\)-整除性，在 \(\ZZ[S]\) 中可定义差额多项式
\begin{equation}\label{eq:chebyshev-frobenius-delta-poly}
\boxed{\;\Delta_p(S):=\frac{C_p(S)-S^p}{p}\in\ZZ[S]\;}
\end{equation}
并据此把代入算子 \(f(S)\mapsto f(C_p(S))\) 解释为一条 Frobenius 提升。

\begin{definition}[\texorpdfstring{$\delta_p$}{delta_p}-结构：Chebyshev 作为 Frobenius-lift]\label{def:chebyshev-frobenius-pderivation}
对任意 \(f\in\ZZ[S]\)，定义
\begin{equation}\label{eq:chebyshev-frobenius-pderivation}
\boxed{\;
\delta_p(f):=\frac{f(C_p(S))-\bigl(f(S)\bigr)^p}{p}\in\QQ[S]\; }.
\end{equation}
\end{definition}

\begin{proposition}[\texorpdfstring{$\delta_p$}{delta_p}-导数的整系数闭包：\texorpdfstring{$\ZZ[S]$}{Z[S]} 上的可审计 Frobenius-lift]\label{prop:chebyshev-frobenius-pderivation-integrality}
在定义 \ref{def:chebyshev-frobenius-pderivation} 的设定下，有：
\begin{enumerate}
  \item \(\Delta_p(S)\in\ZZ[S]\)（等价于 \(\delta_p(S)=\Delta_p(S)\)）；
  \item 对任意 \(f\in\ZZ[S]\)，有 \(\delta_p(f)\in\ZZ[S]\)，并且满足 Frobenius-lift 恒等式
  \[
  \boxed{\ f(C_p(S))=\bigl(f(S)\bigr)^p+p\,\delta_p(f)\ }.
  \]
\end{enumerate}
特别地，\(\bmod p\) 层面 \(f(C_p(S))\equiv f(S)^p\) 把完成化 Adams 作用与纯 Frobenius 严格对齐（并与推论 \ref{cor:completed-dwork-frobenius-tower} 的 \(\widehat a_{p^k}\) 塌缩链相容）。
\leanverified{paper\_chebyshev\_frobenius\_pderivation\_integrality}
\end{proposition}
\begin{proof}
由 \((\mathfrak{r}+\mathfrak{r}^{-1})^p\equiv \mathfrak{r}^p+\mathfrak{r}^{-p}\ (\mathrm{mod}\ p)\)（因 \(\binom{p}{k}\equiv 0\ (\mathrm{mod}\ p)\) 对 \(1\le k\le p-1\) 成立），得
\(
C_p(S)\equiv S^p\ (\mathrm{mod}\ p)
\) 于 \(\ZZ[S]\)，故 \(\Delta_p(S)\in\ZZ[S]\)。
对一般 \(f\in\ZZ[S]\)，由同余的代入不变性得
\(
f(C_p(S))\equiv f(S^p)\ (\mathrm{mod}\ p)
\)；
另一方面在 \(\ZZ/p\ZZ\) 上 Frobenius 满足 \(f(S)^p\equiv f(S^p)\ (\mathrm{mod}\ p)\)。
两式合并即得 \(f(C_p(S))-\bigl(f(S)\bigr)^p\in p\,\ZZ[S]\)，从而 \(\delta_p(f)\in\ZZ[S]\) 且 Frobenius-lift 恒等式成立。证毕。
\end{proof}

\subsubsection{相位坐标与完成化迹坐标的全有理交换图}\label{subsubsec:phase-completion-rational-exchange}
本段给出一个对 unitary-slice 接口尤为关键的“全有理”事实：若以 Cayley 紧化坐标 \(x\in\overline{\RR}\) 参数化半角变量 \(\mathfrak{r}\)，则完成化迹坐标 \(S=\mathfrak{r}+\mathfrak{r}^{-1}\) 与 Chebyshev--Adams 动力学 \(S\mapsto C_d(S)\) 都可被实现为纯有理/整系数运算。

\paragraph{完成化迹坐标的 Cayley 有理参数化}
取 \(\mathfrak{r}\in\TT\) 并用 Cayley 形式 \(\mathfrak{r}=\iota(x)=\frac{1+\mathrm{i}x}{1-\mathrm{i}x}\)（式 \eqref{eq:unit-circle-cayley-embed}）参数化。
则完成化迹坐标在 \(x\)-坐标下具有全有理闭式
\begin{equation}\label{eq:phase-completion-trace-rational}
\boxed{\;
S(x):=\iota(x)+\iota(x)^{-1}
=\frac{2(1-x^2)}{1+x^2}\in[-2,2]\; }.
\end{equation}
因此，任何仅以 \(S\in[-2,2]\) 为连续参数的 unitary-slice 审计接口，都可等价改写为以 \(x\in\overline{\RR}\) 为连续参数的接口，且改写不引入根式或三角函数。

\paragraph{Chebyshev--Adams 的有理提升}
幂半群在半角变量上的作用为 \(\mathfrak{r}\mapsto \mathfrak{r}^{d}\)（\(d\ge 1\)），其在迹坐标上压缩为 \(S\mapsto C_d(S)\)，其中 \(C_d\in\ZZ[S]\) 且满足
\[
C_d(S)=\mathfrak{r}^{d}+\mathfrak{r}^{-d}\qquad\text{当}\qquad S=\mathfrak{r}+\mathfrak{r}^{-1}.
\]
更强地，在 Cayley 坐标 \(x\) 上存在有理函数 \(\tau_d\in\QQ(x)\) 使得 \(\mathfrak{r}^d=\iota(\tau_d(x))\)，从而交换图严格成立：
\[
\begin{CD}
x @>\iota>> \mathfrak{r} @>>> S=\mathfrak{r}+\mathfrak{r}^{-1}\\
@V\tau_d VV @VV(\cdot)^d V @VV C_d V\\
\tau_d(x) @>\iota>> \mathfrak{r}^d @>>> C_d(S)
\end{CD}
\]
其中 \(C_d(S(x))=S(\tau_d(x))\)。

\begin{proposition}[Chebyshev--Adams 作用的全有理实现]\label{prop:chebyshev-adams-fully-rational}
\leanverified{paper\_chebyshev\_adams\_fully\_rational}
对任意整数 \(d\ge 1\)，存在 \(A_d,B_d\in\ZZ[x]\) 使
\begin{equation}\label{eq:adams-Ab-polynomials}
\boxed{\;
(1+\mathrm{i}x)^d=A_d(x)+\mathrm{i}B_d(x),\qquad
(1-\mathrm{i}x)^d=A_d(x)-\mathrm{i}B_d(x)\; }.
\end{equation}
并令
\begin{equation}\label{eq:adams-tau-d}
\boxed{\;\tau_d(x):=\frac{B_d(x)}{A_d(x)}\in\QQ(x)\; }.
\end{equation}
则有
\begin{equation}\label{eq:adams-cayley-lift}
\boxed{\;
\iota(x)^d=\iota\!\bigl(\tau_d(x)\bigr)\; }.
\end{equation}
从而在迹坐标上满足严格一致性
\begin{equation}\label{eq:adams-trace-commute}
\boxed{\;
C_d\!\bigl(S(x)\bigr)=S\!\bigl(\tau_d(x)\bigr)
=\frac{2\bigl(1-\tau_d(x)^2\bigr)}{1+\tau_d(x)^2}\; }.
\end{equation}
此外，\((A_d,B_d)\) 可由纯乘加递推在 \(\ZZ[x]\) 中计算：
\[
A_0=1,\ B_0=0,\qquad
A_{d+1}=A_d-xB_d,\quad
B_{d+1}=B_d+xA_d.
\]
因此任何依赖 \(S\mapsto C_d(S)\) 的 unitary-slice 审计或证书构造，都可在不调用三角函数与代数根式的前提下，以整系数多项式的乘加运算实现。
\end{proposition}
\begin{proof}
由二项式展开，\((1+\mathrm{i}x)^d\) 的实部与虚部分别为整系数多项式，故存在 \(A_d,B_d\in\ZZ[x]\) 使 \eqref{eq:adams-Ab-polynomials} 成立。
由 \(\iota(x)=\frac{1+\mathrm{i}x}{1-\mathrm{i}x}\) 得
\[
\iota(x)^d=\frac{(1+\mathrm{i}x)^d}{(1-\mathrm{i}x)^d}
=\frac{A_d(x)+\mathrm{i}B_d(x)}{A_d(x)-\mathrm{i}B_d(x)}
=\frac{1+\mathrm{i}\tau_d(x)}{1-\mathrm{i}\tau_d(x)}
=\iota\!\bigl(\tau_d(x)\bigr),
\]
得到 \eqref{eq:adams-cayley-lift}。
再将 \(S(t)=\iota(t)+\iota(t)^{-1}=\frac{2(1-t^2)}{1+t^2}\) 代入 \(\tau_d(x)\) 得 \eqref{eq:adams-trace-commute}。
递推式由
\(
(1+\mathrm{i}x)^{d+1}=(1+\mathrm{i}x)(A_d+\mathrm{i}B_d)
\)
展开并对实/虚部分别比较得到。证毕。
\end{proof}

\begin{corollary}[Chebyshev--Adams 的有理半群表示：\texorpdfstring{$\tau_{mn}=\tau_m\circ\tau_n$}{tau_mn=tau_m∘tau_n}]\label{cor:chebyshev-adams-rational-semigroup}
\leanverified{paper\_chebyshev\_adams\_rational\_semigroup}
对任意整数 \(m,n\ge 1\)，有作为有理函数的恒等式
\[
\tau_{mn}=\tau_m\circ\tau_n,
\qquad
\tau_1=\mathrm{id}.
\]
因此 \(d\mapsto\tau_d\) 给出从乘法半群 \((\NN_{\ge 1},\cdot)\) 到有理自映射半群 \((\QQ(x),\circ)\) 的同态。
特别地，一切依赖 \(\mathfrak r\mapsto \mathfrak r^d\) 或 \(S\mapsto C_d(S)\) 的 unitary-slice 审计与证书构造，均可被编译为 \(\tau_d\) 的复合与整系数递推运算，从而在实现层彻底剥离三角函数与代数根式。
\end{corollary}
\begin{proof}
由 \eqref{eq:adams-cayley-lift}，
\[
\iota\!\bigl(\tau_m(\tau_n(x))\bigr)
=\bigl(\iota(\tau_n(x))\bigr)^m
=\iota(x)^{mn}.
\]
另一方面仍由 \eqref{eq:adams-cayley-lift}，\(\iota(\tau_{mn}(x))=\iota(x)^{mn}\)。
由于 \(\iota\) 在 \(\overline{\RR}\) 上为单射（M\"obius 同胚），故得到 \(\tau_{mn}(x)=\tau_m(\tau_n(x))\) 作为 \(\QQ(x)\) 中的恒等式。证毕。
\end{proof}

\begin{theorem}[unitary-slice 投影门的素数正规形]\label{thm:unitary-slice-tau-prime-normal-form}
\leanverified{paper\_unitary\_slice\_tau\_prime\_normal\_form}
沿用命题 \ref{prop:chebyshev-adams-fully-rational} 与推论 \ref{cor:chebyshev-adams-rational-semigroup} 的记号。
映射
\[
\Theta:(\NN_{\ge 1},\cdot)\longrightarrow (\{\tau_d\}_{d\ge 1},\circ),
\qquad d\longmapsto \tau_d
\]
为单射半群同态，因此 \(\{\tau_d\}\) 在复合下与 \((\NN_{\ge 1},\cdot)\) 同构。
从而：
\begin{enumerate}
  \item \(\tau_d\) 在该半群中可分解当且仅当 \(d\) 为合数；
  \item 每个 \(\tau_d\) 具有唯一的素数正规形：若 \(d=\prod_{j=1}^{k}p_j\) 为按重数计的素因子分解，则
  \[
  \tau_d=\tau_{p_1}\circ\cdots\circ \tau_{p_k},
  \]
  且由于 \(\tau_{mn}=\tau_m\circ\tau_n=\tau_n\circ\tau_m\)，该表示在重排意义下唯一。
\end{enumerate}
\end{theorem}
\begin{proof}
同态性由 \(\tau_{mn}=\tau_m\circ\tau_n\) 直接给出。
为证单射，若 \(\tau_d=\tau_{d'}\) 作为有理自映射恒等，则对所有定义域内的 \(x\) 有
\[
\iota(x)^d=\iota(\tau_d(x))=\iota(\tau_{d'}(x))=\iota(x)^{d'}.
\]
因此 \(\iota(x)^{d-d'}=1\) 在 \(\iota(\overline{\RR})\subset\TT\) 的无限子集上成立。
由于单位圆上满足 \(z^{d-d'}=1\) 的点集有限，必有 \(d-d'=0\)，即 \(d=d'\)。
故 \(\Theta\) 为单射半群同态，并给出同构。
分解性与素数正规形随之归结为整数的合数性与唯一素因子分解（并用交换性消去排列歧义）。证毕。
\end{proof}

\paragraph{半角提升作为最小不可见纤维：从 \(w\) 到 \(\mathfrak{r}\) 的分支信息}
映射
\[
q:\TT\to\TT,\qquad q(\mathfrak{r})=\mathfrak{r}^{2}=w
\]
为二重覆盖，其核为 \(\{\pm 1\}\cong \ZZ/2\ZZ\)。
因此在 \(w\)-层（亦即正文的 unitary twist 变量 \(u\) 层）不可见的最小离散自由度，正是该覆盖的 deck 自同构（半角分支）\(\mathfrak{r}\mapsto-\mathfrak{r}\)。
在完成化坐标 \(S=\mathfrak{r}+\mathfrak{r}^{-1}\) 中，该分支体现为
\(
S\mapsto -S
\)，从而在 \(S\) 的奇偶喷射分裂中被显式激活。

\begin{proposition}[端点 \texorpdfstring{$w=-1$}{w=-1} 的半角残差与 \texorpdfstring{$\ZZ/4$}{Z/4} 型取值]\label{prop:half-angle-z4-residue}
在端点 \(w=-1\)（等价于 \(\theta=\pi\)）处，完成化坐标满足 \(S=0\)。
并且对任意 \(m\ge 1\)，Chebyshev--Adams 多项式在端点的取值为
\begin{equation}\label{eq:chebyshev-endpoint-values}
\boxed{\;
C_m(0)=\mathfrak{r}^{m}+\mathfrak{r}^{-m}=2\cos(m\pi/2)\in\{0,\pm 2\}\; }.
\end{equation}
特别地，\(C_m(0)\) 仅由 \(m\bmod 4\) 决定，从而端点处的幂半群作用携带一个 \(\ZZ/4\) 型离散残差。
\leanverified{paper\_half\_angle\_z4\_residue}
\end{proposition}
\begin{proof}
端点 \(w=-1=e^{i\pi}\) 的半角满足 \(\mathfrak{r}^{2}=-1\)，故 \(\mathfrak{r}=\pm i=e^{\pm i\pi/2}\)，于是 \(S=\mathfrak{r}+\mathfrak{r}^{-1}=0\)。
又由于 \(C_m(S)=\mathfrak{r}^m+\mathfrak{r}^{-m}=2\cos(m\theta/2)\)，在 \(\theta=\pi\) 处得
\(
C_m(0)=2\cos(m\pi/2)
\)，其取值仅依赖 \(m\bmod 4\)。证毕。
\end{proof}

\begin{proposition}[端点正交分解：\texorpdfstring{$\ZZ/4$}{Z/4} 离散载荷与 \texorpdfstring{$\RR_{+}$}{R+} 精度载荷]\label{prop:endpoint-orthogonal-decomposition-z4-Rplus}
\leanverified{paper\_endpoint\_orthogonal\_decomposition\_z4\_Rplus}
在端点 \(w=-1\)（完成化坐标 \(S=0\)）的邻域内，伸缩半群 \(m\in\NN_{\ge 1}\) 的作用可分解为两条互不耦合的载荷：
\begin{enumerate}
  \item \textbf{离散端点载荷：}端点态 \(C_m(0)\in\{0,\pm 2\}\) 仅由 \(m\bmod 4\) 决定（命题 \ref{prop:half-angle-z4-residue}）。
  \item \textbf{连续精度载荷：}取 Cayley 相位轴 \(u_{\mathrm C}(t)=\frac{1+\mathrm{i}t}{1-\mathrm{i}t}\) 并令半平面线性化坐标 \(\mathfrak{w}(u)=\frac{1+u}{1-u}\)。
  则在 \(t\neq 0\) 时有 \(\mathfrak{w}(u_{\mathrm C}(t))=\mathrm{i}/t\)，并且在伸缩 \(t\mapsto mt\) 下满足
  \[
  \mathfrak{w}\!\bigl(u_{\mathrm C}(mt)\bigr)=\frac{1}{m}\,\mathfrak{w}\!\bigl(u_{\mathrm C}(t)\bigr),
  \qquad
  y\!\bigl(u_{\mathrm C}(mt)\bigr)=y\!\bigl(u_{\mathrm C}(t)\bigr)+\log m,
  \]
  其中 \(y(u):=-\log|\mathfrak{w}(u)|\) 为端点精度势 \eqref{eq:endpoint-busemann-potential}。
\end{enumerate}
因此，端点处的非紧性沿精度势 \(y\) 轴累积为 \(\log m\) 的平移账本，而端点的“半角分支记忆”由有限离散残差 \(m\bmod 4\) 完全承载。
\end{proposition}
\begin{proof}
第一条即命题 \ref{prop:half-angle-z4-residue}。
第二条由 \eqref{eq:scale-linearization-halfplane-coordinate}--\eqref{eq:scale-linearization-w-vs-x} 得出：当 \(u=u_{\mathrm C}(t)\) 时 \(x(u)=t\) 且 \(\mathfrak{w}(u)=\mathrm{i}/x(u)=\mathrm{i}/t\)。
伸缩 \(t\mapsto mt\) 在 \(u\)-变量上对应于 \(u\mapsto\psi_m(u)\)，且由命题 \ref{prop:scale-linearization-identity-multipliers} 有 \(\mathfrak{w}(\psi_m(u))=\mathfrak{w}(u)/m\)，从而 \(\mathfrak{w}(u_{\mathrm C}(mt))=\mathfrak{w}(u_{\mathrm C}(t))/m\)。
最后由定义 \(y=-\log|\mathfrak{w}|\)（式 \eqref{eq:endpoint-busemann-potential}）即得加法律，亦可直接视为 \eqref{eq:endpoint-busemann-addition-law} 在 unitary Cayley 轴上的限制。证毕。
\end{proof}

\begin{proposition}[端点导数律：奇伸缩线性放大，偶伸缩二阶退化]\label{prop:chebyshev-endpoint-derivative-splitting}
\leanverified{paper\_chebyshev\_endpoint\_derivative\_splitting}
令 \(m\ge 1\) 且 \(C_m(S)=2T_m(S/2)\)（\(T_m\) 为第一类 Chebyshev 多项式）。
则在 \(S=0\) 处有
\begin{equation}\label{eq:chebyshev-endpoint-derivative}
\boxed{\;
C_m'(0)=m\sin(m\pi/2)\; }.
\end{equation}
从而：
\begin{enumerate}
  \item 若 \(m\) 为偶数，则 \(C_m'(0)=0\)，且 \(C_m\) 为偶多项式，端点喷射至少二阶起跳；
  \item 若 \(m\) 为奇数，则 \(C_m(0)=0\) 且 \(|C_m'(0)|=m\)，端点喷射为线性主导。
\end{enumerate}
\end{proposition}
\begin{proof}
由恒等式 \(C_m(S)=2T_m(S/2)\) 与导数公式 \(T_m'(x)=mU_{m-1}(x)\)（\(U_{m-1}\) 为第二类 Chebyshev 多项式），得
\[
C_m'(0)=2\cdot T_m'(0)\cdot \frac12=T_m'(0)=mU_{m-1}(0).
\]
又 \(U_{m-1}(\cos\phi)=\frac{\sin(m\phi)}{\sin\phi}\)，取 \(\cos\phi=0\Rightarrow \phi=\pi/2\) 得 \(U_{m-1}(0)=\sin(m\pi/2)\)，从而得到 \eqref{eq:chebyshev-endpoint-derivative}。
奇偶分裂由 \(\sin(m\pi/2)\in\{0,\pm 1\}\) 立得；并且由 \(C_m(-S)=(-1)^m C_m(S)\)（对应半角分支 \(\mathfrak{r}\mapsto-\mathfrak{r}\)）可知偶 \(m\) 时 \(C_m\) 为偶多项式、奇 \(m\) 时为奇多项式。证毕。
\end{proof}

\begin{corollary}[端点喷射的两类正则化：奇通道控制一阶，偶通道进入高阶]\label{cor:chebyshev-endpoint-jet-regularization}
\leanverified{paper\_chebyshev\_endpoint\_jet\_regularization}
设 \(g\) 在包含区间 \([-2,2]\) 的某个开邻域上解析。
则当 \(S\to 0\) 时有展开
\[
g(C_m(S))=
\begin{cases}
g(C_m(0)) + O(S^2), & m\ \text{为偶数},\\[4pt]
g(0) + g'(0)\,C_m'(0)\,S+O(S^2), & m\ \text{为奇数}.
\end{cases}
\]
因此在端点邻域，奇 \(m\) 通道保留线性主尺度，而偶 \(m\) 通道不产生一阶项，只能改变高阶喷射与有限部常数。
\end{corollary}
\begin{proof}
偶 \(m\) 时 \(C_m'(0)=0\) 且 \(C_m(0)\in\{\pm 2\}\)，故 \(C_m(S)=C_m(0)+O(S^2)\)，在 \(C_m(0)\) 处对 \(g\) 作 Taylor 展开得到 \(g(C_m(S))=g(C_m(0))+O(S^2)\)。
奇 \(m\) 时 \(C_m(0)=0\) 且 \(C_m'(0)\neq 0\)，故 \(C_m(S)=C_m'(0)S+O(S^3)\)，在 \(0\) 处对 \(g\) 作 Taylor 展开即得第二行。证毕。
\end{proof}

\paragraph{primitive 奇偶扇区的端点消失：主奇性由偶长度独占}
令 \(p_n(u)\in\ZZ[u]\) 为 Witt/Möbius 反演得到的 primitive 多项式（例如 \S\ref{subsec:operator-zeta-interface} 的 \(\mathrm{Mob}\) 输出；并见附录 \ref{app:kernel-completion} 及其后的 primitive 接口）。
在大量核族中，primitive 系列满足有限长度镜像对称 \(p_n(u)=u^n p_n(1/u)\)（见推论 \ref{cor:primitive-finite-sym}），并对奇长度出现更强的整除律：

\begin{corollary}[奇长度 primitive 的端点消失]\label{cor:primitive-odd-vanish-u-minus1}
若 \(n\ge 3\) 为奇数，则 \(u(u+1)\mid p_n(u)\)（推论 \ref{cor:primitive-odd-div}）。
特别地 \(p_n(-1)=0\)。
因此把 primitive 生成函数 \(P(z,u)=\sum_{n\ge 1}p_n(u)z^n\) 分解为
\[
P_{\mathrm{odd}}(z,u):=\sum_{n\ \mathrm{odd}}p_n(u)z^n,\qquad
P_{\mathrm{even}}(z,u):=\sum_{n\ \mathrm{even}}p_n(u)z^n,
\]
则除去有限个低长项（至多 \(n=1\)）后，\(P_{\mathrm{odd}}(z,u)\) 在 \(u=-1\) 处至少一阶消失；
从而端点 \(u=-1\) 的任何不可去主奇性只能由偶长度扇区 \(P_{\mathrm{even}}(z,u)\) 控制。
\end{corollary}
\leanverified{paper\_primitive\_odd\_vanish\_u\_minus1}

\begin{proposition}[完成化子环与“唯一角度”的代数刻画]\label{prop:completion-subring-unique-angle}
\leanverified{paper\_completion\_subring\_unique\_angle}
令 \(\mathfrak{r}\) 为半角变量并记 \(S=\mathfrak{r}+\mathfrak{r}^{-1}\)。
则反演作用 \(\mathfrak{r}\mapsto \mathfrak{r}^{-1}\) 的不变量子环满足
\[
\boxed{\;\ZZ[\mathfrak{r},\mathfrak{r}^{-1}]^{\mathfrak{r}\mapsto \mathfrak{r}^{-1}}
=\ZZ[\mathfrak{r}+\mathfrak{r}^{-1}]
=\ZZ[S]\; }.
\]
因此，“从半角层下降到唯一角度”在代数上等价于取反演不变子环；任何超出 \(\ZZ[S]\) 的自由度都等价于半角分支（以及端点喷射）所携带的不可见信息分量。
\end{proposition}
\begin{proof}
对不变量 \(f(\mathfrak{r})=f(\mathfrak{r}^{-1})\)，将 Laurent 系数按 \(k\) 与 \(-k\) 配对，得
\(
f(\mathfrak{r})=\sum_{k\ge 0}a_k(\mathfrak{r}^k+\mathfrak{r}^{-k})
\)。
而 \(\mathfrak{r}^k+\mathfrak{r}^{-k}=C_k(\mathfrak{r}+\mathfrak{r}^{-1})=C_k(S)\in\ZZ[S]\)，故不变量子环包含于 \(\ZZ[S]\)；反向包含由定义直接得到。证毕。
\end{proof}

\begin{proposition}[端点分解范式：Witt 伸缩链的奇/偶通道与 primitive 奇/偶扇区]\label{prop:endpoint-decomposition-paradigm}
\leanverified{paper\_endpoint\_decomposition\_paradigm}
在完成化坐标 \((w,\mathfrak{r},S)\) 下（\(w=\mathfrak{r}^2,\ S=\mathfrak{r}+\mathfrak{r}^{-1}\)），幂半群 \(w\mapsto w^m\) 对应的完成化动作为 \(S\mapsto C_m(S)\)。
于是端点 \(w=-1\)（\(S=0\)）处，“无穷伸缩链”的局部结构满足：
\begin{enumerate}
  \item \textbf{（伸缩通道分裂）}端点喷射按 \(m\) 的奇偶严格分裂：奇 \(m\) 通道控制线性主尺度，而偶 \(m\) 通道只进入高阶喷射与有限部（命题 \ref{prop:chebyshev-endpoint-derivative-splitting} 与推论 \ref{cor:chebyshev-endpoint-jet-regularization}）。
  \item \textbf{（primitive 扇区分裂）}在 primitive 多项式序列中，奇长度 \(n\ge 3\) 的扇区在 \(u=-1\) 处严格消失（推论 \ref{cor:primitive-odd-vanish-u-minus1}），因此端点主奇性完全由偶长度扇区控制。
  \item \textbf{（完成化子环与不可见分量）}所有可见代数结构严格下降到反演不变子环 \(\ZZ[S]\)（命题 \ref{prop:completion-subring-unique-angle}）；被投影压扁的信息以半角分支（\(\ZZ/2\)）与端点离散残差（\(\ZZ/4\)，命题 \ref{prop:half-angle-z4-residue}）以及高阶喷射精度的形式出现。
\end{enumerate}
\end{proposition}

\paragraph{圆盘内的 Witt 幂和算子与对数主尺度}
对任意在 \(\mathbb{D}\) 内解析的函数 \(f\)，定义
\begin{equation}\label{eq:app-witt-pow-operator}
\boxed{\;
(W_{\mathrm{pow}}f)(w):=\sum_{m\ge 1}\frac{1}{m}f(w^m),\qquad |w|<1\; }.
\end{equation}
由于 \(|w|<1\Rightarrow |w^m|\to 0\)，上式在 \(|w|<1\) 内绝对收敛并定义一个解析函数。
作为最小测试例，取 \(f(w)=w\) 得
\[
(W_{\mathrm{pow}}f)(w)=\sum_{m\ge 1}\frac{w^m}{m}=-\log(1-w),
\]
从而“幂链叠加”的主奇性以边界对数尺度出现。
该对数主尺度是结构性的：在保持 \(W_{\mathrm{pow}}\) 的 \(1/m\) 加权与幂替换接口不变的前提下，其边界对数行为只能在正则化/重整化口径中被剥离为可审计的符号项（例如 Dirichlet 正则化下由 \(\zeta(1+\sigma)\) 的极点控制的发散系数；参见命题 \ref{prop:endpoint-dirichlet-symbol-weights}）。
该现象与本文 Witt/Möbius 反演桥包（命题 \ref{prop:zetaK-mobius-primitive}）在结构上同型：在形式层面，\(\log Z\) 的展开天然产生 \(\sum_{m\ge 1}\frac{1}{m}(\cdot)\circ[m]\) 的幂链算子；而在圆盘解析口径下，该幂链算子具有直接的收敛域与边界奇性。
