\paragraph{圆维语义下的素数账本与结构不可有限化}
以下结果在定义 \ref{def:circle-dimension}--\ref{def:half-circle-dimension} 的语义框架下推进，并严格区分实现层编码与结构层同态资源。

\begin{theorem}[短正合列的圆维可加性与秩识别]\label{thm:killo-cdim-short-exact-additivity}
设
\[
0\longrightarrow A\longrightarrow B\longrightarrow C\longrightarrow 0
\]
为有限生成离散交换群短正合列，则
\[
\cdim(B)=\cdim(A)+\cdim(C).
\]
并且对任意有限生成离散交换群 \(G\) 有等式
\[
\cdim(G)=\operatorname{rk}_{\ZZ}(G)=\dim_{\QQ}(G\otimes_{\ZZ}\QQ).
\]
\leanverified{paper\_killo\_cdim\_short\_exact\_additivity}
\end{theorem}
\begin{proof}
有限生成交换群结构定理给出
\[
G\simeq \ZZ^r\oplus F,\qquad F\ \text{有限}.
\]
Pontryagin 对偶满足
\[
\widehat G\simeq \widehat{\ZZ^r}\times\widehat F\simeq \TT^r\times \widehat F.
\]
其中 \(\widehat F\) 为有限紧群，故
\[
(\widehat G)^0\simeq \TT^r,\qquad \cdim(G)=\dim((\widehat G)^0)=r=\operatorname{rk}_{\ZZ}(G).
\]
又有限生成交换群在短正合列中秩可加：
\[
\operatorname{rk}_{\ZZ}(B)=\operatorname{rk}_{\ZZ}(A)+\operatorname{rk}_{\ZZ}(C).
\]
代入 \(\cdim=\operatorname{rk}_{\ZZ}\) 即得可加性，且同时给出与 \(\dim_{\QQ}(G\otimes\QQ)\) 的一致识别。证毕。
\end{proof}

\begin{proposition}[可消去幺半群的单射在 Grothendieck 完成上保持单射]\label{prop:killo-grothendieck-completion-preserves-injection}
设 \(M,N\) 为交换可消去幺半群，并满足定义 \ref{def:half-circle-dimension} 的嵌入/完备性口径。若 \(f:M\to N\) 为单射幺半群同态，则诱导群同态
\[
G(f):G(M)\longrightarrow G(N)
\]
亦为单射。
\leanverified{paper\_killo\_grothendieck\_completion\_preserves\_injection}
\end{proposition}
\begin{proof}
设 \(a,b\in M\) 且 \(G(f)(a-b)=0\)。在 \(G(N)\) 中等价于 \(f(a)-f(b)=0\)，即存在 \(c\in N\) 使
\[
f(a)+c=f(b)+c.
\]
由 \(N\) 的可消去性得 \(f(a)=f(b)\)，再由 \(f\) 单射得 \(a=b\)，故 \(a-b=0\)。从而 \(G(f)\) 单射。证毕。
\end{proof}

\begin{theorem}[素数自由度导致结构半圆维不可有限化]\label{thm:killo-prime-freedom-non-finitizability}
\leanverified{paper\_killo\_prime\_freedom\_non\_finitizability}
记
\[
\NN_{\times}:=(\NN_{\ge 1},\times).
\]
则：
\begin{enumerate}
  \item 其 Grothendieck 群完成同构于可数秩自由交换群
  \[
  G(\NN_{\times})\cong \bigoplus_{p\in\mathbb P}\ZZ;
  \]
  \item 对任意有限生成交换可消去幺半群 \(M\)（满足定义 \ref{def:half-circle-dimension} 的口径），不存在幺半群单射
  \[
  \NN_{\times}\hookrightarrow M.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
唯一分解给出同态
\[
v:\NN_{\times}\to \bigoplus_{p\in\mathbb P}\NN,\qquad
n\mapsto (v_p(n))_{p\in\mathbb P},
\]
且仅有限多个分量非零。群完成允许指数取负，得到
\[
G(\NN_{\times})\cong \bigoplus_{p\in\mathbb P}\ZZ.
\]
得第 1 条。

若存在单射 \(f:\NN_{\times}\to M\)，则由命题 \ref{prop:killo-grothendieck-completion-preserves-injection}，\(G(f)\) 单射：
\[
G(\NN_{\times})\hookrightarrow G(M).
\]
但 \(M\) 有限生成 \(\Rightarrow G(M)\) 有限生成交换群，故 \(\operatorname{rk}_{\ZZ}(G(M))<\infty\)；这不可能容纳可数秩自由群 \(\bigoplus_{p}\ZZ\) 的单射像，矛盾。证毕。
\end{proof}

\begin{corollary}[乘法到有限寄存器加法的同态线性化不可能性]\label{cor:killo-no-finite-additive-register-linearization}
\leanverified{paper\_killo\_no\_finite\_additive\_register\_linearization}
对任意有限 \(k\in\NN\)，不存在幺半群单射同态
\[
\NN_{\times}\hookrightarrow (\NN^k,+).
\]
等价地，若坚持“乘法 \(\mapsto\) 寄存器加法”的严格同态语义，则所需单侧寄存器最小基数为可数无穷；并且
\[
n\longmapsto (v_p(n))_{p\in\mathbb P}
\]
给出可数寄存器实现。
\end{corollary}
\begin{proof}
\((\NN^k,+)\) 为有限生成交换可消去幺半群，套用定理 \ref{thm:killo-prime-freedom-non-finitizability} 第 2 条即得不存在单射同态。另一方面，估值向量映射显然为同态单射，故可数无穷寄存器足够。证毕。
\end{proof}

\begin{theorem}[Gödel 式实现层压缩不可由有限秩账本同态化]\label{thm:killo-godel-compression-not-finite-rank-homologizable}
\leanverified{paper\_killo\_godel\_compression\_not\_finite\_rank\_homologizable}
令 \(e:\NN_{\times}\to\ZZ\) 为任意单射。称一对 \((L,s)\) 为 \(e\) 的同态化账本，若 \(L\) 为交换群且
\[
\Phi:\NN_{\times}\to \ZZ\oplus L,\qquad \Phi(n):=(e(n),s(n))
\]
满足
\[
\Phi(ab)=\Phi(a)+\Phi(b)\quad(\forall a,b\in\NN_{\times}),
\]
并且 \(\Phi\) 为单射。则当 \(L\) 有限生成时，此类同态化账本不存在。
\end{theorem}
\begin{proof}
若存在，则 \(\Phi\) 给出
\[
\NN_{\times}\hookrightarrow (\ZZ\oplus L,+)
\]
的单射幺半群同态。\((\ZZ\oplus L,+)\) 为有限生成交换可消去幺半群，故与定理 \ref{thm:killo-prime-freedom-non-finitizability} 第 2 条矛盾。证毕。
\end{proof}

\begin{proposition}[实现层与结构层半圆维的严格分离]\label{prop:killo-implementation-vs-structural-half-circle-dimension}
\leanverified{paper\_killo\_implementation\_vs\_structural\_half\_circle\_dimension}
定义两类不可混同的量：
\begin{enumerate}
  \item 实现层半圆维
  \[
  \cdim^{\mathrm{impl}}(X):=\inf\left\{\frac{m}{2}: \exists\,\text{单射 }X\hookrightarrow \NN^m\right\};
  \]
  \item 结构层同态半圆维
  \[
  \cdim^{\mathrm{hom}}_{+}(M):=\inf\left\{\frac{k}{2}: \exists\,\text{幺半群单射同态 }M\hookrightarrow (\NN^k,+)\right\}.
  \]
\end{enumerate}
则
\[
\cdim^{\mathrm{hom}}_{+}(\NN,+)=\frac12,\qquad
\cdim^{\mathrm{hom}}_{+}(\NN_{\times})=+\infty,
\]
而同时
\[
\cdim^{\mathrm{impl}}(\NN_{\times})=\frac12.
\]
\end{proposition}
\begin{proof}
\((\NN,+)\) 到 \((\NN,+)\) 的恒等映射即单射同态，故 \(\cdim^{\mathrm{hom}}_{+}(\NN,+)\le \frac12\)；按定义显然不可能小于 \(\frac12\)，故取等。

由推论 \ref{cor:killo-no-finite-additive-register-linearization}，\(\NN_{\times}\) 对任意有限 \(k\) 无法单射同态到 \((\NN^k,+)\)，故 \(\cdim^{\mathrm{hom}}_{+}(\NN_{\times})=+\infty\)。

\(\NN_{\times}\) 为可数无限集，存在集合单射 \(\NN_{\times}\hookrightarrow \NN\)，故 \(\cdim^{\mathrm{impl}}(\NN_{\times})\le \frac12\)；同样由定义下界得等号。证毕。
\end{proof}

\begin{theorem}[局部化数系的圆维—素数账本二分解与可恢复性]\label{thm:killo-localization-circle-dimension-prime-ledger-splitting}
\leanverified{paper\_killo\_localization\_circle\_dimension\_prime\_ledger\_splitting}
设 \(S\subset\mathbb P\) 为有限素数集，并记
\[
G_S:=\ZZ[S^{-1}]
=\left\{\frac{a}{\prod_{p\in S}p^{k_p}}:a\in\ZZ,\ k_p\in\NN\right\}
\]
为其加法群。则存在短正合列
\[
0\longrightarrow \ZZ\longrightarrow G_S\longrightarrow \bigoplus_{p\in S}C_{p^\infty}\longrightarrow 0,
\]
其中 \(C_{p^\infty}\) 为 Prüfer \(p\)-群。Pontryagin 对偶给出紧群短正合列
\[
0\longrightarrow \prod_{p\in S}\ZZ_p\longrightarrow \widehat{G_S}\longrightarrow \TT\longrightarrow 0.
\]
并且：
\begin{enumerate}
  \item 在有限秩扩展口径（定义 \ref{def:conclusion-cdim-finite-rank-extension}）下，
  \[
  \cdim_{\mathrm{fr}}(G_S)=1;
  \]
  \item 核 \(\prod_{p\in S}\ZZ_p\) 为全不连通素数账本因子，且 \(S\) 可由其完全恢复：
  \[
  p\in S\iff \text{核的 \(p\)-主因子非平凡}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
商群 \(G_S/\ZZ\) 的元素皆为有限阶，且其 \(p\)-primary 分量对每个 \(p\in S\) 同构于 \(C_{p^\infty}\)，故得第一条短正合列。对偶化并用
\[
\widehat{C_{p^\infty}}\simeq \ZZ_p,\qquad \widehat{\ZZ}\simeq \TT
\]
得第二条短正合列。

由定理 \ref{thm:conclusion-cdim-finite-rank-extension}，\(\cdim_{\mathrm{fr}}(G)=\operatorname{rank}_{\QQ}(G)\)。而 \(G_S\otimes_{\ZZ}\QQ\simeq \QQ\)（秩 \(1\)），故 \(\cdim_{\mathrm{fr}}(G_S)=1\)。

最后，核分解为 \(\prod_{p\in S}\ZZ_p\)，其 \(p\)-主因子非平凡当且仅当 \(p\in S\)。故 \(S\) 可由核唯一恢复。证毕。
\end{proof}

\begin{corollary}[固定素数基数的局部化族在圆维口径下不可分类]\label{cor:killo-localization-fixed-cardinality-nonclassification}
对任意固定整数 \(r\ge 1\)，定义
\[
\mathscr G_r:=\{\,G_S:\ S\subset\mathbb P,\ |S|=r\,\}.
\]
则：
\begin{enumerate}
  \item \(\mathscr G_r\) 中存在无穷多个两两不同构的群；
  \item 对每个 \(G_S\in \mathscr G_r\)，都有
  \[
  \cdim_{\mathrm{fr}}(G_S)=1,
  \qquad
  (\widehat{G_S})^0\simeq \TT.
  \]
\end{enumerate}
因此，任何仅依赖 \(\cdim_{\mathrm{fr}}\) 或仅依赖连通圆因子 \(\TT\) 的分类不变量，在族 \(\mathscr G_r\) 上都必然是无限不忠实的。
\leanverified{paper\_killo\_localization\_fixed\_cardinality\_nonclassification}
\end{corollary}
\begin{proof}
由定理 \ref{thm:killo-localization-circle-dimension-prime-ledger-splitting}，对每个 \(S\) 都有 \(\cdim_{\mathrm{fr}}(G_S)=1\)，并存在短正合列
\[
0\longrightarrow \prod_{p\in S}\ZZ_p\longrightarrow \widehat{G_S}\longrightarrow \TT\longrightarrow 0.
\]
由于 \(\prod_{p\in S}\ZZ_p\) 全不连通，交
\(
(\widehat{G_S})^0\cap \prod_{p\in S}\ZZ_p
\)
作为全不连通群中的连通子群只能是平凡子群。另一方面，\((\widehat{G_S})^0\) 在商映射下的像是 \(\TT\) 的连通闭子群，且包含单位元，故必为整个 \(\TT\)。因此该商映射在 \((\widehat{G_S})^0\) 上限制为同构，于是
\(
(\widehat{G_S})^0\simeq \TT
\)。

另一方面，满足 \(|S|=r\) 的有限素数集有无穷多个。若 \(S\neq S'\)，则由定理 \ref{thm:killo-localization-circle-dimension-prime-ledger-splitting} 的可恢复性，
\[
\prod_{p\in S}\ZZ_p\not\cong \prod_{p\in S'}\ZZ_p,
\]
从而 \(G_S\not\cong G_{S'}\)。故 \(\mathscr G_r\) 中存在无穷多个两两不同构对象，而它们共享相同的有限秩圆维与相同的连通圆因子。证毕。
\end{proof}

上述链条给出一条严格分界：实现层可将可数自由度压缩至单寄存器，而结构层同态保持要求会在乘法对象上强制出现不可有限化的素数账本核；在局部化数系中，该核与相位圆维可在对偶层面二分并可完全反演。

\begin{theorem}[偶子群自由作用与无分歧 \(A_4\)-提升]\label{thm:killo-s4-even-subgroup-free-action-unramified-a4-lift}

  \item \(\mathcal H\to \PP^1_u\) 为分歧于上述 \(12\) 点的二重覆盖，且

\[
\mathcal C_6=\mathcal X/V_4\to \mathcal H=\mathcal X/A_4
\]
为循环三重覆盖。惯性与 \(A_4\) 交平凡同样给出无分歧；故
\[
g(\mathcal C_6)=1+3(g(\mathcal H)-1)=1+3\cdot 4=13.

证毕。
\end{proof}

在定理 \ref{thm:killo-s4-even-subgroup-free-action-unramified-a4-lift} 与定理 \ref{thm:killo-s4-genus49-jacobian-computable-isogeny} 的记号下：

  \QQ(\zeta_3)\subseteq \operatorname{End}^0\!\bigl(\operatorname{Prym}(\mathcal C_6/\mathcal H)\bigr).
  \]
  \item \(P=\operatorname{Prym}(Y/Z)\) 的诱导极化型为
  \[
  (1,1,1,1,2,2,2,2,2).
  \]
\end{enumerate}
\end{proposition}
\begin{proof}
第 1 条：\(\mathcal C_6\to\mathcal H\) 为无分歧循环三重覆盖，且 \(g(\mathcal H)=5\)。循环素数度无分歧覆盖的 Prym 极化型为
\[
(1^{g(\mathcal H)-1},3^{g(\mathcal H)-1})=(1^4,3^4).
\]
又甲板变换群 \(C_3\) 在 Prym 上作用，故有 \(\QQ(\zeta_3)\)-乘法嵌入；其包含关系亦与定理 \ref{thm:fold-gauge-anomaly-a2-eisenstein-multiplication} 一致。

第 2 条：\(\pi:Y\to Z\) 为二重覆盖，\(g(Y)=13\)、\(g(Z)=4\)。由 Hurwitz 公式
\[
g(Y)=2g(Z)-1+r
\]
得 \(r=6\)，即分歧点数 \(2r=12\)。二重分歧覆盖的 Prym 极化型为
\[
(1^{g(Z)},2^{r-1})=(1^4,2^5).
\]
证毕。
\end{proof}

\begin{conjecture}[大维 Prym 因子的最小额外自同态]\label{conj:killo-minimal-extra-endomorphism-large-prym}
在上述 Hurwitz 数据点上，预期：
\begin{enumerate}
  \item \(P=\operatorname{Prym}(Y/Z)\) 在 \(\overline{\QQ}\) 上绝对单，且
  \[
  \operatorname{End}_{\overline{\QQ}}(P)=\ZZ;
  \]
  \item
  \[
  \operatorname{End}^0_{\overline{\QQ}}\!\bigl(\operatorname{Prym}(\mathcal C_6/\mathcal H)\bigr)=\QQ(\zeta_3).
  \]
\end{enumerate}
该预期成立时，定理 \ref{thm:killo-s4-genus49-jacobian-computable-isogeny} 在简单因子层面达到最简，且与结论 \ref{conj:fold-gauge-anomaly-isotypic-absolute-simplicity} 的极小内自同态口径兼容。
\end{conjecture}
