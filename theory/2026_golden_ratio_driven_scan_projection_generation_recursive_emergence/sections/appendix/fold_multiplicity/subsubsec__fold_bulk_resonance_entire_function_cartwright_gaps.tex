\paragraph{虚轴精确渐近、Cartwright 指示函数与零点间隙的尺度律}
本段在定理 \ref{thm:fold-resonance-entire-lp} 的整函数闭包基础上，给出沿虚轴的精确增长次主项，
并据此确定指示函数与 Cartwright 类结构；进一步给出零点计数的线性律与最小零点间隙的 \(R^{-1}\) 标度。

\begin{theorem}[虚轴精确渐近与指数型的确定]\label{thm:fold-resonance-imag-asymptotic-type-pi}
令 \(\mathcal{F}_\varphi\) 如定理 \ref{thm:fold-resonance-entire-lp}。当 \(|y|\to\infty\) 时，有渐近式
\begin{equation}\label{eq:fold-resonance-imag-asymptotic}
\log\bigl|\mathcal{F}_\varphi(\mathrm{i}y)\bigr|
=\pi|y|-\frac{\log 2}{\log\varphi}\,\log|y|+O(1).
\end{equation}
特别地，\(\mathcal{F}_\varphi\) 的指数型恰为 \(\pi\)。
\end{theorem}

\begin{proof}
由 \(\cos(\mathrm{i}t)=\cosh t\) 与 \(\mathcal{F}_\varphi\) 的定义，对 \(y>0\) 有
\[
\log \mathcal{F}_\varphi(\mathrm{i}y)=\sum_{n=2}^{\infty}\log\cosh(\pi y\varphi^{-n}),
\]
其中每一项为正实数的对数。取整数 \(N=N(y)\) 使
\[
y\varphi^{-N}\in[1,\varphi),
\]
于是 \(N=\log_\varphi y+O(1)\)。将级数分解为
\[
S_1:=\sum_{n=2}^{N}\log\cosh(\pi y\varphi^{-n}),\qquad
S_2:=\sum_{n>N}\log\cosh(\pi y\varphi^{-n}).
\]

当 \(2\le n\le N\) 时，\(\pi y\varphi^{-n}\ge \pi y\varphi^{-N}\ge \pi\)，并可用一致估计
\[
\log\cosh t=t-\log 2+O(e^{-2t})\qquad (t\ge \pi).
\]
从而
\[
S_1
=\pi y\sum_{n=2}^{N}\varphi^{-n}-(N-1)\log 2
+O\!\Bigl(\sum_{n=2}^{N}e^{-2\pi y\varphi^{-n}}\Bigr).
\]
注意到 \(\sum_{n=2}^{N}\varphi^{-n}=1-\varphi^{-(N-1)}\)，并且指数尾和一致有界，故
\[
S_1=\pi y-\pi y\varphi^{-(N-1)}-(N-1)\log 2+O(1).
\]
由 \(y\varphi^{-N}\in[1,\varphi)\) 可得 \(y\varphi^{-(N-1)}\in[\varphi,\varphi^2)\)，从而 \(\pi y\varphi^{-(N-1)}=O(1)\)。

对 \(S_2\)，当 \(n>N\) 时有 \(y\varphi^{-n}\le y\varphi^{-(N+1)}<1\)。利用 \(\log\cosh t=O(t^2)\)（\(t\to 0\)），得
\[
S_2=O\!\Bigl(\sum_{n>N} y^2\varphi^{-2n}\Bigr)
=O\!\bigl(y^2\varphi^{-2N}\bigr)
=O(1),
\]
其中最后一步使用 \(y\varphi^{-N}=O(1)\)。

综上，
\[
\log \mathcal{F}_\varphi(\mathrm{i}y)
=\pi y-(N-1)\log 2+O(1)
=\pi y-\frac{\log 2}{\log\varphi}\,\log y+O(1).
\]
由于 \(\mathcal{F}_\varphi\) 为偶函数，得到 \eqref{eq:fold-resonance-imag-asymptotic} 对 \(|y|\to\infty\) 成立。

另一方面，定理 \ref{thm:fold-resonance-entire-lp} 给出全平面上界
\(\log|\mathcal{F}_\varphi(z)|\le \pi|\Im z|\)，故指数型不超过 \(\pi\)；
而 \eqref{eq:fold-resonance-imag-asymptotic} 表明沿虚轴的增长率达到 \(\pi\)，因此指数型恰为 \(\pi\)。证毕。
\end{proof}

\begin{theorem}[Cartwright 类与指示函数]\label{thm:fold-resonance-cartwright-indicator}
令指示函数
\[
h(\theta):=\limsup_{r\to\infty}\frac{1}{r}\log\bigl|\mathcal{F}_\varphi(re^{\mathrm{i}\theta})\bigr|.
\]
则对任意 \(\theta\in\RR\) 成立
\[
h(\theta)=\pi|\sin\theta|.
\]
从而 \(\mathcal{F}_\varphi\) 属于 Cartwright 类，且其指示图为虚轴方向的线段 \([-\mathrm{i}\pi,\mathrm{i}\pi]\)。
\end{theorem}

\begin{proof}
由定理 \ref{thm:fold-resonance-entire-lp} 的全平面估计，
\[
\log\bigl|\mathcal{F}_\varphi(re^{\mathrm{i}\theta})\bigr|\le \pi r|\sin\theta|,
\]
故 \(h(\theta)\le \pi|\sin\theta|\)。

为得下界，先给出与实部无关的粗下界：对任意 \(x\in\RR\) 与 \(|y|\ge 1\)，由恒等式
\[
|\cos(x+\mathrm{i}y)|^2=\cos^2x\,\cosh^2y+\sin^2x\,\sinh^2y\ge \sinh^2|y|
\]
并用 \(\sinh|y|\ge \frac{1-e^{-2}}{2}e^{|y|}\)（\(|y|\ge 1\)）得
\[
|\cos(x+\mathrm{i}y)|\ge c\,e^{|y|},\qquad c:=\frac{1-e^{-2}}{2}.
\]

固定 \(\theta\) 且 \(\sin\theta\neq 0\)。对给定 \(r\) 取 \(N=N(r,\theta)\) 使
\[
\pi r|\sin\theta|\,\varphi^{-N}\ge 1>\pi r|\sin\theta|\,\varphi^{-(N+1)}.
\]
对 \(2\le n\le N\) 令 \(w_n:=\pi r e^{\mathrm{i}\theta}\varphi^{-n}\)，则 \(|\Im w_n|\ge 1\)，于是
\[
\log\bigl|\mathcal{F}_\varphi(re^{\mathrm{i}\theta})\bigr|
=\sum_{n=2}^{\infty}\log|\cos w_n|
\ge \sum_{n=2}^{N}\log|\cos w_n|
\ge (N-1)\log c+\sum_{n=2}^{N}|\Im w_n|.
\]
而
\[
\sum_{n=2}^{N}|\Im w_n|
=\pi r|\sin\theta|\sum_{n=2}^{N}\varphi^{-n}
=\pi r|\sin\theta|\bigl(1-\varphi^{-(N-1)}\bigr)
=\pi r|\sin\theta|+O(1).
\]
同时 \(N=\log_\varphi r+O(1)\)，故 \((N-1)\log c=O(\log r)=o(r)\)。从而
\[
\liminf_{r\to\infty}\frac{1}{r}\log\bigl|\mathcal{F}_\varphi(re^{\mathrm{i}\theta})\bigr|
\ge \pi|\sin\theta|.
\]
与上界合并得 \(h(\theta)=\pi|\sin\theta|\)。

当 \(\sin\theta=0\) 时，\(z=re^{\mathrm{i}\theta}\in\RR\) 且每个因子满足 \(|\cos(\pi z\varphi^{-n})|\le 1\)，故
\(\log|\mathcal{F}_\varphi(z)|\le 0\)，从而 \(h(\theta)=0=\pi|\sin\theta|\)。

最后，由 \(|\mathcal{F}_\varphi(x)|\le 1\)（\(x\in\RR\)）可知 \(\log^+|\mathcal{F}_\varphi(x)|\equiv 0\)，从而
\(\int_{\RR}\frac{\log^+|\mathcal{F}_\varphi(x)|}{1+x^2}\,\dd x=0\)；
结合定理 \ref{thm:fold-resonance-imag-asymptotic-type-pi} 给出的指数型为 \(\pi\)，即得 \(\mathcal{F}_\varphi\) 属 Cartwright 类。证毕。
\end{proof}

\begin{corollary}[零点计数的线性律与显式余项阶]\label{cor:fold-resonance-zero-counting-linear}
记
\[
N_+(R):=\#\{x\in(0,R]:\ \mathcal{F}_\varphi(x)=0\},
\qquad
N(R):=\#\{x\in[-R,R]:\ \mathcal{F}_\varphi(x)=0\}.
\]
则当 \(R\to\infty\) 时，
\[
N_+(R)=R+O(\log R),
\qquad
N(R)=2R+O(\log R).
\]
\end{corollary}
\leanverified{paper\_fold\_resonance\_zero\_counting\_linear}

\begin{proof}
由定理 \ref{thm:fold-resonance-entire-lp}，正零点为 \(\{(k+\tfrac12)\varphi^n:\ k\in\ZZ_{\ge 0},\ n\ge 2\}\)。
固定 \(n\ge 2\)，该层在 \((0,R]\) 内的零点数为
\[
\#\Bigl\{k\ge 0:\ (k+\tfrac12)\varphi^n\le R\Bigr\}
=\left\lfloor \frac{R}{\varphi^n}+\frac12\right\rfloor
=\frac{R}{\varphi^n}+O(1).
\]
仅当 \(\varphi^n\le 2R\) 时该项可能非零，故求和至 \(n\le \lfloor\log_\varphi(2R)\rfloor\) 即止，得到
\[
N_+(R)
=\sum_{n=2}^{\lfloor\log_\varphi(2R)\rfloor}\Bigl(\frac{R}{\varphi^n}+O(1)\Bigr)
=R\sum_{n=2}^{\lfloor\log_\varphi(2R)\rfloor}\varphi^{-n}+O(\log R)
=R+O(\log R),
\]
其中使用 \(\sum_{n=2}^{\infty}\varphi^{-n}=1\)。由零点集关于原点对称，得
\(
N(R)=2N_+(R)+O(1)=2R+O(\log R)
\)。
证毕。
\end{proof}

\begin{theorem}[Lucas 近共振与零点最小间隙上界]\label{thm:fold-resonance-lucas-near-resonance-gap}
令 Lucas 数列 \((L_n)_{n\ge 0}\) 由 \(L_0=2\)、\(L_1=1\)、\(L_{n+2}=L_{n+1}+L_n\) 定义。
对每个整数 \(m\ge 2\) 且 \(3\nmid m\)，定义两点
\[
x_m:=\frac12\,\varphi^{2m+2},
\qquad
y_m:=\frac{L_{2m}}{2}\,\varphi^2.
\]
则 \(x_m,y_m\) 均为 \(\mathcal{F}_\varphi\) 的零点，并满足精确间隙恒等式
\[
\abs{x_m-y_m}
=\frac12\,\varphi^{2-2m}
=\frac{\varphi^4}{4}\cdot\frac{1}{x_m}.
\]
从而零点集并非一致分离。若记
\[
\delta(R):=\min\{\abs{u-v}:\ u\ne v,\ \mathcal{F}_\varphi(u)=\mathcal{F}_\varphi(v)=0,\ \abs{u},\abs{v}\le R\},
\]
则对无穷多个 \(R\)（取 \(R=x_m\)）有
\[
\delta(R)\le \frac{\varphi^4}{4R}.
\]
\leanverified{paper\_fold\_resonance\_lucas\_near\_resonance\_gap}
\end{theorem}

\begin{proof}
由 Lucas 闭式 \(L_n=\varphi^n+\psi^n\)（\(\psi:=-\varphi^{-1}\)）可知 \(L_n\) 为偶数当且仅当 \(n\equiv 0\pmod 3\)；
由 \(3\nmid m\) 得 \(2m\not\equiv 0\pmod 3\)，故 \(L_{2m}\) 为奇数。
于是
\[
y_m=\Bigl(\frac{L_{2m}-1}{2}+\frac12\Bigr)\varphi^2
\]
具有零点参数形态 \((k+\tfrac12)\varphi^n\)（取 \(n=2\)），而 \(x_m=(0+\tfrac12)\varphi^{2m+2}\) 亦为零点。

又因 \(2m\) 为偶数，有 \(\psi^{2m}=\varphi^{-2m}\)，从而
\[
\varphi^{2m}-L_{2m}
=-\psi^{2m}
=-\varphi^{-2m}.
\]
于是
\[
x_m-y_m
=\frac12\bigl(\varphi^{2m+2}-L_{2m}\varphi^2\bigr)
=\frac{\varphi^2}{2}\bigl(\varphi^{2m}-L_{2m}\bigr)
=-\frac12\,\varphi^{2-2m},
\]
取绝对值得第一式。由 \(x_m=\frac12\varphi^{2m+2}\) 立得
\(\frac12\varphi^{2-2m}=\frac{\varphi^4}{4}\cdot \frac{1}{x_m}\)。
最后取 \(R=x_m\) 即得 \(\delta(R)\le \abs{x_m-y_m}=\varphi^4/(4R)\)。证毕。
\end{proof}

\begin{proposition}[零点最小间隙的全局下界]\label{prop:fold-resonance-min-gap-global-lower}
对任意两不同零点 \(u\ne v\) 且 \(\abs{u}\le \abs{v}\)，恒有
\[
\abs{u-v}\ \ge\ \frac{\varphi^4}{8\abs{v}}.
\]
因此对一切 \(R\ge 1\)，
\[
\delta(R)\ \ge\ \frac{\varphi^4}{8R}.
\]
结合定理 \ref{thm:fold-resonance-lucas-near-resonance-gap} 的上界，可得最小零点间隙满足 \(\delta(R)\asymp R^{-1}\)。
\end{proposition}

\begin{proof}
由零点结构，可写
\[
u=\frac{K}{2}\varphi^{m},\qquad v=\frac{L}{2}\varphi^{n},
\]
其中 \(m,n\ge 2\)，且 \(K,L\) 为奇整数。设 \(m\ge n\)（否则交换），记 \(p:=m-n\ge 0\)，则
\[
u-v=\frac{\varphi^{n}}{2}\,(K\varphi^{p}-L).
\]
令 \(\alpha:=K\varphi^{p}-L\in\ZZ[\varphi]\)。由 \(u\ne v\) 得 \(\alpha\ne 0\)。
取共轭嵌入 \(\varphi\mapsto\psi:=-\varphi^{-1}\)，则 \(\alpha' = K\psi^{p}-L\)。
代数整数范数 \(N(\alpha):=\alpha\alpha'\in\ZZ\setminus\{0\}\)，故
\[
\abs{\alpha}\ge \frac{1}{\abs{\alpha'}}.
\]
又因 \(\abs{\psi}=\varphi^{-1}\) 且 \(\abs{\psi}^{p}\le 1\)，有
\[
\abs{\alpha'}=\abs{K\psi^{p}-L}\le \abs{L}+\abs{K}\abs{\psi}^{p}\le \abs{L}+\abs{K}.
\]
由 \(\abs{v}=\frac{\abs{L}}{2}\varphi^{n}\) 得 \(\abs{L}=2\abs{v}\varphi^{-n}\)。
又由 \(\abs{u}\le \abs{v}\) 与 \(\abs{u}=\frac{\abs{K}}{2}\varphi^{n+p}\ge \frac{\abs{K}}{2}\varphi^n\) 得 \(\abs{K}\le 2\abs{v}\varphi^{-n}\)。
于是
\[
\abs{\alpha'}\le 4\abs{v}\varphi^{-n},
\qquad
\abs{\alpha}\ge \frac{\varphi^{n}}{4\abs{v}}.
\]
从而
\[
\abs{u-v}
=\frac{\varphi^{n}}{2}\abs{\alpha}
\ge \frac{\varphi^{n}}{2}\cdot\frac{\varphi^{n}}{4\abs{v}}
=\frac{\varphi^{2n}}{8\abs{v}}
\ge \frac{\varphi^4}{8\abs{v}},
\]
其中最后一步用到 \(n\ge 2\)。取 \(\abs{v}\le R\) 即得 \(\delta(R)\ge \varphi^4/(8R)\)。证毕。
\end{proof}
