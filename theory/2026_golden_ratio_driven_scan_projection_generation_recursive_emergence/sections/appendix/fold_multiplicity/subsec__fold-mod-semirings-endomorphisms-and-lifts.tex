\subsection{模半环的端同态、乘法奇异性与纤维 lift 群}\label{subsec:fold-mod-semirings-endomorphisms-and-lifts}
折叠同余半环的模表示（定理 \ref{thm:fold-congruence-mod-semirings}）不仅给出中间商格的约数分类（命题 \ref{prop:fold-intermediate-semirings-divisor-lattice}），亦强制了可见算术层上的对称、投影与奇异结构。
本小节将其端同态、乘法退化与 lift 对称作完全闭式刻画。

\begin{theorem}[折叠同余半环的幺自同构刚性]\label{thm:fold-mod-semirings-unital-automorphism-rigidity}
设 \(m\ge 1\)，令 \(n:=F_{m+2}\)。通过定理 \ref{thm:fold-congruence-mod-semirings} 的半环同构
\[
(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m)\cong (\ZZ/(n\ZZ),+,\times)
\]
识别 \(\Omega_m/\!\equiv_m\) 为幺交换半环。
则其全部幺半环自同构为恒等映射：
\[
\Aut_{\mathrm{unital\; semiring}}(\Omega_m/\!\equiv_m)=\{\mathrm{id}\}.
\]
\end{theorem}
\begin{proof}
在同构下只需证明：任意幺半环自同构 \(\varphi:\ZZ/(n\ZZ)\to \ZZ/(n\ZZ)\) 必为恒等。
由于 \(\varphi\) 保持幺元，\(\varphi(\overline 1)=\overline 1\)。而对任意 \(\overline k\in\ZZ/(n\ZZ)\)，有 \(\overline k=k\cdot \overline 1\)，从而
\[
\varphi(\overline k)=\varphi(k\cdot \overline 1)=k\cdot \varphi(\overline 1)=\overline k.
\]
故 \(\varphi=\mathrm{id}\)。证毕。
\end{proof}

\begin{theorem}[折叠同余半环的端同态与幂等元分类]\label{thm:fold-mod-semirings-endomorphisms-idempotents}
设 \(m\ge 1\)，令 \(n:=F_{m+2}\)，并仍以定理 \ref{thm:fold-congruence-mod-semirings} 识别 \(\Omega_m/\!\equiv_m\cong \ZZ/(n\ZZ)\)。
则半环端同态单子 \(\End_{\mathrm{semiring}}(\Omega_m/\!\equiv_m)\) 与 \(\ZZ/(n\ZZ)\) 的幂等元集合自然双射：
\[
\End_{\mathrm{semiring}}(\Omega_m/\!\equiv_m)\ \cong\ \{e\in \ZZ/(n\ZZ):\ e^2=e\}.
\]
在该对应下，幂等元 \(e\) 对应的端同态为乘法投影
\[
\varphi_e(\overline x):=e\,\overline x.
\]
\end{theorem}
\begin{proof}
在 \(\ZZ/(n\ZZ)\) 上证明即可。设 \(\varphi\) 为半环端同态，记 \(e:=\varphi(\overline 1)\)。
由乘法保持性，
\[
e^2=\varphi(\overline 1)\varphi(\overline 1)=\varphi(\overline 1\cdot \overline 1)=\varphi(\overline 1)=e,
\]
故 \(e\) 幂等。又任取 \(\overline x\in\ZZ/(n\ZZ)\)，有 \(\overline x=x\cdot \overline 1\)，于是
\[
\varphi(\overline x)=\varphi(x\cdot \overline 1)=x\cdot \varphi(\overline 1)=x e=e\,\overline x.
\]
从而 \(\varphi=\varphi_e\)。反之，若 \(e^2=e\)，则映射 \(\varphi_e(\overline x)=e\,\overline x\) 显然保持加法与乘法（乘法保持性仅用到 \(e^2=e\)），因而为半环端同态。证毕。
\end{proof}

\begin{corollary}[CRT 幂等元的子集参数化与布尔骨架]\label{cor:fold-mod-semirings-idempotents-crt-boolean}
\leanverified{paper\_fold\_mod\_semirings\_idempotents\_crt\_boolean}
设
\[
n=\prod_{i=1}^r p_i^{\alpha_i}
\]
为 \(n\ge 1\) 的互异素数分解。对任意子集 \(S\subseteq \{1,\dots,r\}\)，存在唯一幂等元 \(e_S\in \ZZ/(n\ZZ)\) 满足
\[
e_S\equiv 1\pmod{p_i^{\alpha_i}}\quad(i\in S),
\qquad
e_S\equiv 0\pmod{p_i^{\alpha_i}}\quad(i\notin S).
\]
并且
\[
\operatorname{Idem}(\ZZ/(n\ZZ))
=
\{e_S:\ S\subseteq \{1,\dots,r\}\},
\qquad
\#\operatorname{Idem}(\ZZ/(n\ZZ))=2^r=2^{\omega(n)}.
\]
在中国剩余分解下，幂等元集合与子集格 \(\mathcal P(\{1,\dots,r\})\) 同构；其交、并与补分别由
\[
e_S e_T=e_{S\cap T},
\qquad
e_S+e_T-e_Se_T=e_{S\cup T},
\qquad
1-e_S=e_{S^c}
\]
给出。
\end{corollary}
\begin{proof}
由中国剩余定理
\[
\ZZ/(n\ZZ)\cong \prod_{i=1}^r \ZZ/(p_i^{\alpha_i}\ZZ).
\]
在每个局部因子 \(\ZZ/(p_i^{\alpha_i}\ZZ)\) 中，幂等方程 \(u^2=u\) 只有 \(u=0,1\) 两个解。
故任一幂等元都唯一对应于一个 \(0/1\) 坐标向量，亦即唯一对应于一个子集 \(S\subseteq \{1,\dots,r\}\)，其中第 \(i\) 个坐标取 \(1\) 当且仅当 \(i\in S\)。
计数公式遂即成立。

在直积描述下，乘法、\(e+f-ef\) 与 \(1-e\) 分别按坐标实现逻辑与、逻辑或与取补，因此诱导出与子集格完全一致的布尔结构。证毕。
\end{proof}

\begin{corollary}[CRT 原始正交幂等框架与规范分量分解]\label{cor:fold-mod-semirings-crt-orthogonal-frame}
\leanverified{paper\_fold\_mod\_semirings\_crt\_orthogonal\_frame}
设 \(m\ge 1\)，令
\[
n:=F_{m+2}=\prod_{i=1}^r p_i^{\alpha_i},
\qquad
R_m:=(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m),
\]
并记 \(\pi_m:R_m\to \ZZ/(n\ZZ)\) 为定理 \ref{thm:fold-congruence-mod-semirings} 给出的半环同构。
对每个 \(i\)，置
\[
n_i:=\frac{n}{p_i^{\alpha_i}},
\]
并取整数 \(s_i\) 满足
\[
s_i n_i\equiv 1\pmod{p_i^{\alpha_i}}.
\]
令
\[
\varepsilon_i:=\overline{s_i n_i}\in \ZZ/(n\ZZ),
\qquad
e_i:=\pi_m^{-1}(\varepsilon_i)\in R_m.
\]
则 \(e_1,\dots,e_r\) 构成 \(R_m\) 的一组原始中心正交幂等框架：
\[
e_i\boxtimes_m e_i=e_i,
\qquad
e_i\boxtimes_m e_j=0\quad(i\neq j),
\qquad
e_1\boxplus_m\cdots\boxplus_m e_r=1.
\]
并且任意 \(x\in R_m\) 都有唯一分量分解
\[
x=(e_1\boxtimes_m x)\boxplus_m\cdots\boxplus_m(e_r\boxtimes_m x),
\]
其中 \(\pi_m(e_i\boxtimes_m x)\) 恰为 \(\pi_m(x)\) 在第 \(i\) 个 CRT 因子 \(\ZZ/(p_i^{\alpha_i}\ZZ)\) 上的投影再嵌入。
\end{corollary}
\begin{proof}
由中国剩余定理，
\[
\ZZ/(n\ZZ)\cong \prod_{i=1}^r \ZZ/(p_i^{\alpha_i}\ZZ).
\]
在该分解下，\(\varepsilon_i\) 的第 \(i\) 个分量为 \(1\)，其余分量全为 \(0\)；因此每个 \(\varepsilon_i\) 都是原始中心幂等元，且彼此正交并满足
\[
\varepsilon_1+\cdots+\varepsilon_r=\overline 1.
\]
通过半环同构 \(\pi_m\) 拉回，得到 \(e_i\) 的幂等、正交与求和关系。

对任意 \(\overline a\in \ZZ/(n\ZZ)\)，有
\[
\overline a=(\varepsilon_1\overline a)+\cdots+(\varepsilon_r\overline a),
\]
且第 \(i\) 项只保留第 \(i\) 个 CRT 分量。故对任意 \(x\in R_m\) 有
\[
x=(e_1\boxtimes_m x)\boxplus_m\cdots\boxplus_m(e_r\boxtimes_m x).
\]
若另有 \(x=y_1\boxplus_m\cdots\boxplus_m y_r\) 且每个 \(y_i=e_i\boxtimes_m y_i\)，则对任意 \(j\) 左乘 \(e_j\) 并利用正交关系可得
\[
e_j\boxtimes_m x=y_j,
\]
从而分解唯一。证毕。
\end{proof}

除幂等投影外，可见乘法层的核像大小、幂零阴影与湮灭子格也都由 \(F_{m+2}\) 的素因子分解逐项决定。

\begin{theorem}[乘法核像的 \texorpdfstring{$\gcd$}{gcd} 互反公式]\label{thm:fold-mod-semirings-kernel-image-gcd}
\leanverified{paper\_fold\_mod\_semirings\_kernel\_image\_gcd}
设
\[
n_m:=F_{m+2},
\qquad
\mathfrak R_m:=(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m),
\]
并记 \(\pi_m:\mathfrak R_m\to \ZZ/(n_m\ZZ)\) 为定理 \ref{thm:fold-congruence-mod-semirings} 的半环同构。
对任意 \(x\in \mathfrak R_m\)，取整数 \(a_x\) 满足 \(\pi_m(x)=\overline{a_x}\)，并定义乘法端同态
\[
\mu_x:\mathfrak R_m\to \mathfrak R_m,
\qquad
y\longmapsto x\boxtimes_m y.
\]
则有精确公式
\[
\#\ker(\mu_x)=\gcd(a_x,n_m),
\qquad
\#\operatorname{im}(\mu_x)=\frac{n_m}{\gcd(a_x,n_m)},
\]
从而
\[
\#\ker(\mu_x)\cdot \#\operatorname{im}(\mu_x)=n_m.
\]
特别地，
\[
x\in \mathfrak R_m^\times
\iff
\gcd(a_x,n_m)=1,
\]
并且对 \(x\neq 0\)，有
\[
x\ \text{为零因子}
\iff
1<\gcd(a_x,n_m)<n_m.
\]
\end{theorem}
\begin{proof}
由于 \(\gcd(a_x,n_m)\) 对 \(a_x\) 的选取无关，只需在 \(\ZZ/(n_m\ZZ)\) 上证明。
记
\[
g:=\gcd(a_x,n_m),
\qquad
a_x=g a',
\qquad
n_m=g n',
\qquad
\gcd(a',n')=1.
\]
方程
\[
a_x y\equiv 0\pmod{n_m}
\]
等价于
\[
g a' y\equiv 0\pmod{g n'}
\iff
a'y\equiv 0\pmod{n'}.
\]
由 \(\gcd(a',n')=1\) 得 \(y\equiv 0\pmod{n'}\)。
模 \(n_m\) 下满足此条件的剩余类恰有 \(g\) 个，故 \(\#\ker(\mu_x)=g\)。

另一方面，\(\mu_x\) 是有限加法群 \((\ZZ/(n_m\ZZ),+)\) 的自同态，因而由核像公式得到
\[
\#\operatorname{im}(\mu_x)=\frac{n_m}{g}.
\]
单位判据与零因子判据随即成立。证毕。
\end{proof}

\begin{theorem}[平方自由分岔与幂零层的精确计数]\label{thm:fold-mod-semirings-squarefree-nilpotent-branch}
设
\[
n_m=F_{m+2}=\prod_{i=1}^{r_m} p_i^{\alpha_i},
\qquad
\mathfrak R_m=(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m).
\]
若 \(\pi_m(x)=\overline{a_x}\)，则
\[
\operatorname{Nil}(\mathfrak R_m)
=
\{x\in \mathfrak R_m:\ \operatorname{rad}(n_m)\mid a_x\},
\qquad
\#\operatorname{Nil}(\mathfrak R_m)=\frac{n_m}{\operatorname{rad}(n_m)}.
\]
因此以下命题等价：
\[
\mathfrak R_m\ \text{无非零幂零元},
\qquad
\mathfrak R_m\ \text{为约化环},
\qquad
F_{m+2}\ \text{平方自由}.
\]
\end{theorem}
\begin{proof}
仍在 \(\ZZ/(n_m\ZZ)\) 上证明即可。
由中国剩余定理，
\[
\ZZ/(n_m\ZZ)\cong \prod_{i=1}^{r_m}\ZZ/(p_i^{\alpha_i}\ZZ).
\]
在每个局部因子 \(\ZZ/(p_i^{\alpha_i}\ZZ)\) 中，一个元素为幂零元当且仅当它被 \(p_i\) 整除。
因此直积中的元素幂零，当且仅当其各坐标都落在各自极大理想中；这等价于其对应整数剩余类同时被所有 \(p_i\) 整除，也即 \(\operatorname{rad}(n_m)\mid a_x\)。

模 \(n_m\) 下满足该整除条件的剩余类恰有 \(n_m/\operatorname{rad}(n_m)\) 个，从而得到计数公式。
最后，存在非零幂零元当且仅当某个 \(\alpha_i\ge 2\)，这恰等价于 \(n_m\) 非平方自由。证毕。
\end{proof}

\begin{theorem}[湮灭子理想的 valuation 格]\label{thm:fold-mod-semirings-annihilator-valuation-lattice}
设
\[
n_m=F_{m+2}=\prod_{i=1}^{r_m} p_i^{\alpha_i},
\qquad
\mathfrak R_m=(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m).
\]
对 \(x\in \mathfrak R_m\)，若 \(\pi_m(x)=\overline{a_x}\)，则定义截断估值向量
\[
\nu_i(x):=\min\{v_{p_i}(a_x),\alpha_i\},
\qquad
1\le i\le r_m,
\]
其中约定 \(v_{p_i}(0)=\infty\)。
再定义湮灭子理想
\[
\operatorname{Ann}(x):=\{y\in \mathfrak R_m:\ x\boxtimes_m y=0\}.
\]
则在 \(\mathfrak R_m\cong \ZZ/(n_m\ZZ)\) 的识别下，有
\[
\operatorname{Ann}(x)
=
\left(\prod_{i=1}^{r_m} p_i^{\alpha_i-\nu_i(x)}\right)\Big/\!(n_m).
\]
若记
\[
\mathcal A_m:=\{\operatorname{Ann}(x):\ x\in \mathfrak R_m\},
\]
并按包含关系赋序，则映射
\[
\operatorname{Ann}(x)\longmapsto (\nu_1(x),\dots,\nu_{r_m}(x))
\]
给出规范格同构
\[
\mathcal A_m\cong \prod_{i=1}^{r_m}[0,\alpha_i].
\]
特别地，
\[
\#\mathcal A_m=\prod_{i=1}^{r_m}(\alpha_i+1),
\]
并且 \(\mathcal A_m\) 为布尔代数当且仅当 \(n_m\) 平方自由。
\end{theorem}
\begin{proof}
在 \(\ZZ/(n_m\ZZ)\) 中，\(\operatorname{Ann}(x)\) 就是乘法映射 \(\mu_x\) 的核。
若记
\[
g_x:=\gcd(a_x,n_m)=\prod_{i=1}^{r_m} p_i^{\nu_i(x)},
\]
则由定理 \ref{thm:fold-mod-semirings-kernel-image-gcd} 的证明，
\[
\operatorname{Ann}(x)
=
\left(\frac{n_m}{g_x}\right)\Big/\!(n_m)
=
\left(\prod_{i=1}^{r_m} p_i^{\alpha_i-\nu_i(x)}\right)\Big/\!(n_m).
\]

反过来，任意向量 \((\nu_1,\dots,\nu_{r_m})\in \prod_i [0,\alpha_i]\) 都可由
\[
a=\prod_{i=1}^{r_m} p_i^{\nu_i}
\]
实现，故所有湮灭子理想恰由这些截断估值向量参数化。
对两个元素 \(x,y\)，有
\[
\operatorname{Ann}(x)\subseteq \operatorname{Ann}(y)
\iff
\alpha_i-\nu_i(x)\ge \alpha_i-\nu_i(y)\ \text{对一切 }i
\iff
\nu_i(x)\le \nu_i(y)\ \text{对一切 }i.
\]
故 \(\mathcal A_m\) 与链乘积 \(\prod_i [0,\alpha_i]\) 作为有序格同构。
基数公式随即成立；而链乘积成为布尔代数当且仅当每条链都只有两个点，也即每个 \(\alpha_i=1\)，等价于 \(n_m\) 平方自由。证毕。
\end{proof}

\begin{theorem}[纤维保持可观测置换的 lift 群与 wreath 分解]\label{thm:fold-lift-wreath-decomposition}
\leanverified{paper\_fold\_lift\_wreath\_decomposition}
设 \(p:\Omega\to X\) 为有限集合之间的满射。
对 \(d\ge 1\) 记
\(
X(d):=\{x\in X:\ |p^{-1}(x)|=d\}
\)。
定义 lift 群
\[
\mathcal L(p)
:=\left\{\tau\in\mathrm{Sym}(\Omega):\ \exists\,f\in\mathrm{Sym}(X)\ \text{s.t.}\ p\circ\tau=f\circ p\right\}.
\]
则在为每个 \(x\in X(d)\) 选取任意双射 \(p^{-1}(x)\cong\{1,\dots,d\}\) 之后，存在（非规范的）群同构
\[
\mathcal L(p)\ \cong\ \prod_{d\ge 1}\Bigl(\mathfrak S_d\wr \mathfrak S_{\abs{X(d)}}\Bigr),
\]
其中 wreath 乘积 \(\mathfrak S_d\wr \mathfrak S_n:=\mathfrak S_d^{\,n}\rtimes \mathfrak S_n\) 的半直积由 \(\mathfrak S_n\) 对 \(n\) 个 \(\mathfrak S_d\) 因子的置换作用给出。
\end{theorem}
\begin{proof}
若 \(\tau\in\mathcal L(p)\) 覆盖 \(f\in\mathrm{Sym}(X)\)，则对任意 \(x\in X\) 有
\(
\tau(p^{-1}(x))=p^{-1}(f(x))
\)，
从而 \(|p^{-1}(x)|=|p^{-1}(f(x))|\)，即 \(f\) 必逐层保持 \(X(d)\) 并诱导一个 \(\mathfrak S_{\abs{X(d)}}\) 的置换。
在固定的纤维识别下，\(\tau\) 在每个纤维 \(p^{-1}(x)\) 上给出一个 \(\mathfrak S_d\) 元素；当 \(f\) 置换 \(X(d)\) 时，这些 \(\mathfrak S_d\) 因子相应被重排，恰形成半直积 \(\mathfrak S_d^{\,\abs{X(d)}}\rtimes \mathfrak S_{\abs{X(d)}}\)。
对各个 \(d\) 的分层彼此独立，得到乘积分解。证毕。
\end{proof}

则以下两类现象同步出现：
\begin{enumerate}
  \item \(N_m\) 平方自由，因而 \(\mathfrak R_m\) 为约化环，且其湮灭子格退化为由 CRT 幂等元控制的布尔骨架；
  \item 一切乘法型异常与外置信息审计都可压缩到 CRT 扇区层面，而无需额外的幂零隐层。
\end{enumerate}
相反，若 \(N_m\) 含平方因子，则 \(\mathfrak R_m\) 必出现非零幂零元，且乘法型隐层所需的最小额外深度由
\[
\max_{1\le i\le s} e_i
\]
控制。
window-\(6\) 的 \(N_6=21\) 提供了这一分岔在平方自由端的基准例。
\end{conjecture}
