\subsection{模半环端同态、幺自同构刚性与纤维 lift 群}\label{subsec:fold-mod-semirings-endomorphisms-and-lifts}
折叠同余半环的模表示（定理 \ref{thm:fold-congruence-mod-semirings}）不仅给出中间商格的约数分类（命题 \ref{prop:fold-intermediate-semirings-divisor-lattice}），亦强制了可见算术层上的对称与投影结构。
本小节将其端同态与 lift 对称作完全闭式刻画。

\begin{theorem}[折叠同余半环的幺自同构刚性]\label{thm:fold-mod-semirings-unital-automorphism-rigidity}
设 \(m\ge 1\)，令 \(n:=F_{m+2}\)。通过定理 \ref{thm:fold-congruence-mod-semirings} 的半环同构
\[
(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m)\cong (\ZZ/(n\ZZ),+,\times)
\]
识别 \(\Omega_m/\!\equiv_m\) 为幺交换半环。
则其全部幺半环自同构为恒等映射：
\[
\Aut_{\mathrm{unital\; semiring}}(\Omega_m/\!\equiv_m)=\{\mathrm{id}\}.
\]
\end{theorem}
\begin{proof}
在同构下只需证明：任意幺半环自同构 \(\varphi:\ZZ/(n\ZZ)\to \ZZ/(n\ZZ)\) 必为恒等。
由于 \(\varphi\) 保持幺元，\(\varphi(\overline 1)=\overline 1\)。而对任意 \(\overline k\in\ZZ/(n\ZZ)\)，有 \(\overline k=k\cdot \overline 1\)，从而
\[
\varphi(\overline k)=\varphi(k\cdot \overline 1)=k\cdot \varphi(\overline 1)=\overline k.
\]
故 \(\varphi=\mathrm{id}\)。证毕。
\end{proof}

\begin{theorem}[折叠同余半环的端同态与幂等元分类]\label{thm:fold-mod-semirings-endomorphisms-idempotents}
设 \(m\ge 1\)，令 \(n:=F_{m+2}\)，并仍以定理 \ref{thm:fold-congruence-mod-semirings} 识别 \(\Omega_m/\!\equiv_m\cong \ZZ/(n\ZZ)\)。
则半环端同态单子 \(\End_{\mathrm{semiring}}(\Omega_m/\!\equiv_m)\) 与 \(\ZZ/(n\ZZ)\) 的幂等元集合自然双射：
\[
\End_{\mathrm{semiring}}(\Omega_m/\!\equiv_m)\ \cong\ \{e\in \ZZ/(n\ZZ):\ e^2=e\}.
\]
在该对应下，幂等元 \(e\) 对应的端同态为乘法投影
\[
\varphi_e(\overline x):=e\,\overline x.
\]
\end{theorem}
\begin{proof}
在 \(\ZZ/(n\ZZ)\) 上证明即可。设 \(\varphi\) 为半环端同态，记 \(e:=\varphi(\overline 1)\)。
由乘法保持性，
\[
e^2=\varphi(\overline 1)\varphi(\overline 1)=\varphi(\overline 1\cdot \overline 1)=\varphi(\overline 1)=e,
\]
故 \(e\) 幂等。又任取 \(\overline x\in\ZZ/(n\ZZ)\)，有 \(\overline x=x\cdot \overline 1\)，于是
\[
\varphi(\overline x)=\varphi(x\cdot \overline 1)=x\cdot \varphi(\overline 1)=x e=e\,\overline x.
\]
从而 \(\varphi=\varphi_e\)。反之，若 \(e^2=e\)，则映射 \(\varphi_e(\overline x)=e\,\overline x\) 显然保持加法与乘法（乘法保持性仅用到 \(e^2=e\)），因而为半环端同态。证毕。
\end{proof}

\begin{corollary}[幂等投影的计数与 CRT 布尔格]\label{cor:fold-mod-semirings-idempotents-crt-boolean}
设 \(n\ge 1\)，记 \(\omega(n)\) 为 \(n\) 的不同素因子个数。
则 \(\ZZ/(n\ZZ)\) 的幂等元个数满足
\[
\#\{e\in\ZZ/(n\ZZ):e^2=e\}=2^{\omega(n)}.
\]
并且幂等元集合在乘法下构成布尔代数，其结构同构于对互异素幂分量的子集选择。
\end{corollary}
\begin{proof}
设 \(n=\prod_{i=1}^r p_i^{\alpha_i}\) 为互异素数分解（\(r=\omega(n)\)）。
由中国剩余定理
\[
\ZZ/(n\ZZ)\ \cong\ \prod_{i=1}^r \ZZ/(p_i^{\alpha_i}\ZZ).
\]
而在每个局部环 \(\ZZ/(p^\alpha\ZZ)\) 中幂等元只能取 \(0,1\)。因此幂等元在直积分量上逐坐标独立选择 \(0/1\)，计数为 \(2^r\)。
在该同构下，乘法为逐坐标乘法，互补由 \(e\mapsto 1-e\) 给出，从而幂等元构成布尔代数并与子集选择同构。证毕。
\end{proof}

\begin{corollary}[CRT 原始正交幂等框架与规范分量分解]\label{cor:fold-mod-semirings-crt-orthogonal-frame}
设 \(m\ge 1\)，令
\[
n:=F_{m+2}=\prod_{i=1}^r p_i^{\alpha_i},
\qquad
R_m:=(\Omega_m/\!\equiv_m,\boxplus_m,\boxtimes_m),
\]
并记 \(\pi_m:R_m\to \ZZ/(n\ZZ)\) 为定理 \ref{thm:fold-congruence-mod-semirings} 给出的半环同构。
对每个 \(i\)，置
\[
n_i:=\frac{n}{p_i^{\alpha_i}},
\]
并取整数 \(s_i\) 满足
\[
s_i n_i\equiv 1\pmod{p_i^{\alpha_i}}.
\]
令
\[
\varepsilon_i:=\overline{s_i n_i}\in \ZZ/(n\ZZ),
\qquad
e_i:=\pi_m^{-1}(\varepsilon_i)\in R_m.
\]
则 \(e_1,\dots,e_r\) 构成 \(R_m\) 的一组原始中心正交幂等框架：
\[
e_i\boxtimes_m e_i=e_i,
\qquad
e_i\boxtimes_m e_j=0\quad(i\neq j),
\qquad
e_1\boxplus_m\cdots\boxplus_m e_r=1.
\]
并且任意 \(x\in R_m\) 都有唯一分量分解
\[
x=(e_1\boxtimes_m x)\boxplus_m\cdots\boxplus_m(e_r\boxtimes_m x),
\]
其中 \(\pi_m(e_i\boxtimes_m x)\) 恰为 \(\pi_m(x)\) 在第 \(i\) 个 CRT 因子 \(\ZZ/(p_i^{\alpha_i}\ZZ)\) 上的投影再嵌入。
\end{corollary}
\begin{proof}
由中国剩余定理，
\[
\ZZ/(n\ZZ)\cong \prod_{i=1}^r \ZZ/(p_i^{\alpha_i}\ZZ).
\]
在该分解下，\(\varepsilon_i\) 的第 \(i\) 个分量为 \(1\)，其余分量全为 \(0\)；因此每个 \(\varepsilon_i\) 都是原始中心幂等元，且彼此正交并满足
\[
\varepsilon_1+\cdots+\varepsilon_r=\overline 1.
\]
通过半环同构 \(\pi_m\) 拉回，得到 \(e_i\) 的幂等、正交与求和关系。

对任意 \(\overline a\in \ZZ/(n\ZZ)\)，有
\[
\overline a=(\varepsilon_1\overline a)+\cdots+(\varepsilon_r\overline a),
\]
且第 \(i\) 项只保留第 \(i\) 个 CRT 分量。故对任意 \(x\in R_m\) 有
\[
x=(e_1\boxtimes_m x)\boxplus_m\cdots\boxplus_m(e_r\boxtimes_m x).
\]
若另有 \(x=y_1\boxplus_m\cdots\boxplus_m y_r\) 且每个 \(y_i=e_i\boxtimes_m y_i\)，则对任意 \(j\) 左乘 \(e_j\) 并利用正交关系可得
\[
e_j\boxtimes_m x=y_j,
\]
从而分解唯一。证毕。
\end{proof}

\begin{example}[window-\texorpdfstring{\(6\)}{6} 的四投影算术根源]\label{ex:fold-mod-semirings-m6-idempotents}
在 \(m=6\) 时 \(n=F_{8}=21=3\cdot 7\)，故 \(\omega(n)=2\)。
由推论 \ref{cor:fold-mod-semirings-idempotents-crt-boolean} 得幂等元恰有 \(4\) 个；除 \(\overline 0,\overline 1\) 外，两枚原始正交幂等元可显式取为
\[
e_{(3)}=\overline 7,\qquad e_{(7)}=\overline{15}\qquad(\text{于 }\ZZ/(21\ZZ)),
\]
并满足 \(e_{(3)}+e_{(7)}=\overline 1\)、\(e_{(3)}e_{(7)}=\overline 0\)。
从而
\[
\ZZ/(21\ZZ)\ \cong\ \ZZ/(3\ZZ)\times \ZZ/(7\ZZ),
\]
且与定理 \ref{thm:fold-mod-semirings-endomorphisms-idempotents} 对应的半环端同态给出一个完全显式的四扇区投影格。
\end{example}

\begin{corollary}[稳定类型上的幺半环对称不可见性]\label{cor:fold-visible-unital-semirings-no-nontrivial-symmetry}
通过 \(V_m:X_m\to\{0,1,\dots,F_{m+2}-1\}\)（引理 \ref{lem:zeckendorf-range-bijection}）将稳定类型集 \(X_m\) 识别为 \(\ZZ/(F_{m+2}\ZZ)\) 的代表系。
若双射 \(f:X_m\to X_m\) 在该识别下诱导 \(\ZZ/(F_{m+2}\ZZ)\) 的幺半环自同构，则 \(f=\mathrm{id}_{X_m}\)。
\end{corollary}
\begin{proof}
由定理 \ref{thm:fold-mod-semirings-unital-automorphism-rigidity}，幺半环自同构只能为恒等，因而 \(f\) 必为恒等。证毕。
\end{proof}

\begin{theorem}[纤维保持可观测置换的 lift 群与 wreath 分解]\label{thm:fold-lift-wreath-decomposition}
设 \(p:\Omega\to X\) 为有限集合之间的满射。
对 \(d\ge 1\) 记
\(
X(d):=\{x\in X:\ |p^{-1}(x)|=d\}
\)。
定义 lift 群
\[
\mathcal L(p)
:=\left\{\tau\in\mathrm{Sym}(\Omega):\ \exists\,f\in\mathrm{Sym}(X)\ \text{s.t.}\ p\circ\tau=f\circ p\right\}.
\]
则在为每个 \(x\in X(d)\) 选取任意双射 \(p^{-1}(x)\cong\{1,\dots,d\}\) 之后，存在（非规范的）群同构
\[
\mathcal L(p)\ \cong\ \prod_{d\ge 1}\Bigl(\mathfrak S_d\wr \mathfrak S_{\abs{X(d)}}\Bigr),
\]
其中 wreath 乘积 \(\mathfrak S_d\wr \mathfrak S_n:=\mathfrak S_d^{\,n}\rtimes \mathfrak S_n\) 的半直积由 \(\mathfrak S_n\) 对 \(n\) 个 \(\mathfrak S_d\) 因子的置换作用给出。
\end{theorem}
\begin{proof}
若 \(\tau\in\mathcal L(p)\) 覆盖 \(f\in\mathrm{Sym}(X)\)，则对任意 \(x\in X\) 有
\(
\tau(p^{-1}(x))=p^{-1}(f(x))
\)，
从而 \(|p^{-1}(x)|=|p^{-1}(f(x))|\)，即 \(f\) 必逐层保持 \(X(d)\) 并诱导一个 \(\mathfrak S_{\abs{X(d)}}\) 的置换。
在固定的纤维识别下，\(\tau\) 在每个纤维 \(p^{-1}(x)\) 上给出一个 \(\mathfrak S_d\) 元素；当 \(f\) 置换 \(X(d)\) 时，这些 \(\mathfrak S_d\) 因子相应被重排，恰形成半直积 \(\mathfrak S_d^{\,\abs{X(d)}}\rtimes \mathfrak S_{\abs{X(d)}}\)。
对各个 \(d\) 的分层彼此独立，得到乘积分解。证毕。
\end{proof}

\begin{remark}[window-\texorpdfstring{\(6\)}{6} 的 bin-fold lift 标准型]\label{rem:foldbin6-lift-wreath-normal-form}
将定理 \ref{thm:fold-lift-wreath-decomposition} 应用于二进制区间折叠 \(\Fold^{\mathrm{bin}}_6:\Omega_6\to X_6\)，并用推论 \ref{cor:fold-bin-m6-fiber-hist} 的直方图 \(2:8,\,3:4,\,4:9\)，可得
\[
\mathcal L(\Fold^{\mathrm{bin}}_6)\ \cong\ (\mathfrak S_2\wr \mathfrak S_{8})\times(\mathfrak S_3\wr \mathfrak S_{4})\times(\mathfrak S_4\wr \mathfrak S_{9}).
\]
该分解仅依赖纤维分层数据，因而可由有限枚举逐项复核。
\end{remark}

\begin{conjecture}[素因子—扇区对应原则]\label{conj:fold-prime-sector-omega-bound}
令 \(m\ge 1\) 且 \(n:=F_{m+2}\)。
设在稳定类型集 \(X_m\) 上给定一个可审计的扇区分解，其生成的子集族在并、交与补下闭合，因而形成一个有限布尔代数 \(\mathcal B_m\subseteq \mathcal P(X_m)\)。
则存在一个满的布尔代数态射
\[
\Pi_m:\mathcal B_m\longtwoheadrightarrow \{e\in\ZZ/(n\ZZ):\ e^2=e\},
\]
其中右端以推论 \ref{cor:fold-mod-semirings-idempotents-crt-boolean} 的幂等投影布尔格结构为运算。
因此任何此类可审计扇区分解的最大独立 \(\ZZ_2\) 秩受 \(\omega(n)\) 控制，并在满射存在时达到 \(\omega(F_{m+2})\) 的上界。
\end{conjecture}
