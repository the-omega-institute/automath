\paragraph{Lee--Yang 分歧域与十次分歧域的联合算术独立性}
以下结论在分歧域层面给出可直接用于 Chebotarev 统计的独立性框架，并与文中既有 \(S_{10}\) 与 \(S_3\) 证书链保持符号一致。

\begin{theorem}[两类 Lee--Yang 分歧域的算术独立性与直积伽罗瓦群]\label{thm:killo-leyang-two-branch-fields-product-galois}
\leanverified{paper\_killo\_leyang\_two\_branch\_fields\_product\_galois}
设 \(P_{10}(u)\in\ZZ[u]\) 为谱四次分歧十次多项式，分裂域记为 \(L_{10}\)；并设
\[
K_{10}:=\QQ\!\left(\sqrt{66269989}\right),\qquad 66269989=1931\cdot 34319.
\]
再设
\[
g(y):=256y^3+411y^2+165y+32\in\ZZ[y]
\]
的分裂域为 \(L_{\mathrm{LY}}\)，并记
\[
K_{\mathrm{LY}}:=\QQ(\sqrt{-111}).
\]
若
\[
\mathrm{Gal}(L_{10}/\QQ)\cong S_{10},\qquad
\mathrm{Gal}(L_{\mathrm{LY}}/\QQ)\cong S_{3},
\]
且 \(K_{10}\) 与 \(K_{\mathrm{LY}}\) 分别为两分裂域的唯一非平凡二次正规子域，并满足
\[
\mathrm{Ram}(K_{10}/\QQ)=\{1931,34319\},
\qquad
\mathrm{Ram}(L_{\mathrm{LY}}/\QQ)\subseteq\{3,37\},
\]
则
\[
L_{10}\cap L_{\mathrm{LY}}=\QQ,
\qquad
\mathrm{Gal}(L_{10}L_{\mathrm{LY}}/\QQ)\cong S_{10}\times S_{3}.
\]
\end{theorem}
\begin{proof}
两域均为 \(\QQ\) 上伽罗瓦扩张，故交域
\(
E:=L_{10}\cap L_{\mathrm{LY}}
\)
亦为 \(\QQ\) 上伽罗瓦扩张。由
\(
\mathrm{Gal}(L_{10}/\QQ)\cong S_{10}
\)
（\(n\ge 5\) 时正规子群仅有 \(\{1\},A_n,S_n\)）知
\(
E\in\{\QQ,K_{10}\}
\)；
由
\(
\mathrm{Gal}(L_{\mathrm{LY}}/\QQ)\cong S_3
\)
知
\(
E\in\{\QQ,K_{\mathrm{LY}}\}
\)。
若 \(E\neq \QQ\)，则由上式必有 \(E=K_{10}\subseteq L_{\mathrm{LY}}\)。
但 \(K_{10}\) 在 \(1931,34319\) 处分歧，而 \(L_{\mathrm{LY}}\) 仅在 \(\{3,37\}\) 内可能分歧，矛盾。故 \(E=\QQ\)。
于是两域线性互素，且二者伽罗瓦，标准限制映射给出
\[
\mathrm{Gal}(L_{10}L_{\mathrm{LY}}/\QQ)
\xrightarrow{\ \sim\ }
\mathrm{Gal}(L_{10}/\QQ)\times \mathrm{Gal}(L_{\mathrm{LY}}/\QQ)
\cong S_{10}\times S_3.
\]
证毕。
\end{proof}

\begin{theorem}[联合 Chebotarev 乘积律与双指纹密度闭式]\label{thm:killo-leyang-joint-chebotarev-product-law}
设 \(C\subset S_{10}\)、\(D\subset S_3\) 为共轭类。对除去有限分歧素数后的素数 \(p\)，记
\[
\sigma_{10}(p):=\mathrm{Frob}_p|_{L_{10}},\qquad
\sigma_{3}(p):=\mathrm{Frob}_p|_{L_{\mathrm{LY}}}.
\]
则集合
\[
\{\,p:\ \sigma_{10}(p)\in C,\ \sigma_3(p)\in D\,\}
\]
具有自然密度
\[
\delta(C,D)=\frac{|C|}{10!}\cdot \frac{|D|}{6}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:killo-leyang-two-branch-fields-product-galois}，
\(
\mathrm{Gal}(L_{10}L_{\mathrm{LY}}/\QQ)\cong S_{10}\times S_3
\)。
Chebotarev 密度定理给出 Frobenius 共轭类按群中共轭类比例分布；在直积群中，共轭类即为直积共轭类 \(C\times D\)，其比例为
\(
\frac{|C|}{|S_{10}|}\frac{|D|}{|S_3|}
\)。
证毕。
\end{proof}

\begin{corollary}[联合事件的显式密度]\label{cor:killo-leyang-joint-density-explicit}
定义事件
\[
A:=\{\text{\(P_{10}\bmod p\) 含一次因子}\},
\]
\[
B_{\mathrm{split}}:=\{\text{\(g\bmod p\) 全分裂}\},
\qquad
B_{\mathrm{root}}:=\{\text{\(g\bmod p\) 至少有一根}\}.
\]
则有
\[
\delta(A\cap B_{\mathrm{split}})
=\frac{28319}{44800}\cdot\frac{1}{6}
=\frac{28319}{268800},
\]
\[
\delta(A\cap B_{\mathrm{root}})
=\frac{28319}{44800}\cdot\frac{2}{3}
=\frac{28319}{67200}.
\]
\end{corollary}
\begin{proof}
由推论 \ref{cor:fold-gauge-anomaly-P10-modp-splitting-densities}，
\(
\delta(A)=28319/44800
\)。
而在 \(S_3\) 中，全分裂对应恒等元类，故
\(
\delta(B_{\mathrm{split}})=1/6
\)；
“至少一根”对应 \((1)(1)(1)\) 与 \((1)(2)\) 两类，故
\(
\delta(B_{\mathrm{root}})=1/6+1/2=2/3
\)。
由定理 \ref{thm:killo-leyang-joint-chebotarev-product-law} 的乘积律即得结论。
\end{proof}

\begin{theorem}[判别式二次特征的解耦与条件密度不变性]\label{thm:killo-leyang-quadratic-character-decoupling}
对不分歧素数 \(p\)，定义
\[
\chi_{10}(p):=\mathrm{sgn}\!\bigl(\sigma_{10}(p)\bigr)\in\{\pm1\},
\qquad
\chi_{\mathrm{LY}}(p):=\mathrm{sgn}\!\bigl(\sigma_{3}(p)\bigr)\in\{\pm1\}.
\]
则：
\begin{enumerate}
  \item 对 \(p\nmid 66269989\)，有
  \[
  \chi_{10}(p)=\left(\frac{66269989}{p}\right);
  \]
  \item 对 \(p\nmid 111\)，有
  \[
  \chi_{\mathrm{LY}}(p)=\left(\frac{-111}{p}\right);
  \]
  \item \(\bigl(\chi_{10}(p),\chi_{\mathrm{LY}}(p)\bigr)\) 在 \(\{\pm1\}^2\) 上等分布，每一对符号密度均为 \(1/4\)；
  \item 在任一条件 \(\chi_{10}(p)=\pm1\) 下，\(g\bmod p\) 的三种分解型密度保持不变：
  \[
  \delta\!\left((1)(2)\,\middle|\,\chi_{10}(p)=\pm1\right)=\frac12,\quad
  \delta\!\left((1)(1)(1)\,\middle|\,\chi_{10}(p)=\pm1\right)=\frac16,\quad
  \delta\!\left((3)\,\middle|\,\chi_{10}(p)=\pm1\right)=\frac13.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
第 1、2 条是“符号特征 \(\leftrightarrow\) 唯一二次正规子域”的 Artin 特征对应：分别对应 \(K_{10}\) 与 \(K_{\mathrm{LY}}\)。

第 3 条由定理 \ref{thm:killo-leyang-two-branch-fields-product-galois} 的直积结构推出。等价地，合成双二次域
\[
K_{10}K_{\mathrm{LY}}/\QQ
\]
的伽罗瓦群同构于 \(C_2\times C_2\)，Chebotarev 给出四类等密度 \(1/4\)。

第 4 条由定理 \ref{thm:killo-leyang-joint-chebotarev-product-law} 直接得到：\(\chi_{10}\) 仅依赖 \(S_{10}\) 分量，而 \(g\bmod p\) 分解型仅依赖 \(S_3\) 分量，二者独立；故对 \(\chi_{10}(p)=\pm1\) 的条件化不改变 \(S_3\) 共轭类分布。将 \(S_3\) 的三类大小 \(3,1,2\) 代入即得 \(1/2,1/6,1/3\)。证毕。
\end{proof}

\begin{theorem}[并入 Latt\`es 临界六次后的三重直积与三域同验密度]\label{thm:killo-leyang-lattes-triple-product-galois}
令
\[
C(X):=X^{6}-8X^{5}+24X^{4}-36X^{3}+34X^{2}-20X+4\in\ZZ[X],
\]
其分裂域记为 \(L_4\)。并假设
\[
\mathrm{Disc}(C)=-2^{19}\cdot 37^{3},\qquad
\mathrm{Gal}(L_4/\QQ)\cong S_4\times C_2.
\]
则 \(L_{10},L_{\mathrm{LY}},L_4\) 线性互素，且
\[
\mathrm{Gal}(L_{10}L_{\mathrm{LY}}L_4/\QQ)
\cong
S_{10}\times S_3\times (S_4\times C_2).
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:killo-leyang-two-branch-fields-product-galois}，
\(
L_{10}\cap L_{\mathrm{LY}}=\QQ
\)。

再证 \(L_{\mathrm{LY}}\cap L_4=\QQ\)。交域是 \(L_{\mathrm{LY}}\) 的伽罗瓦子扩张，故只能为 \(\QQ\) 或 \(K_{\mathrm{LY}}=\QQ(\sqrt{-111})\)。
若为后者，则其在素数 \(3\) 处分歧；而由 \(\mathrm{Disc}(C)\) 可知 \(L_4/\QQ\) 仅在 \(\{2,37\}\) 处分歧，矛盾。故交域平凡。

设 \(M:=L_{\mathrm{LY}}L_4\)。则 \(M/\QQ\) 仅在 \(\{2,3,37\}\) 处分歧。若 \(L_{10}\cap M\neq \QQ\)，作为 \(L_{10}\) 的伽罗瓦子扩张，必含其唯一非平凡二次正规子域 \(K_{10}\)；但 \(K_{10}\) 在 \(1931,34319\) 处分歧，不可能嵌入 \(M\)，矛盾。
故 \(L_{10}\cap M=\QQ\)。

综上，三域两两在所需层面交域平凡，故复合域伽罗瓦群为所示三重直积。证毕。
\end{proof}

\begin{corollary}[三域同验的全分裂密度与混合事件密度]\label{cor:killo-leyang-lattes-triple-density}
对除去有限分歧素数后的素数 \(p\)，若同时要求
\begin{enumerate}
  \item \(P_{10}\bmod p\) 全分裂，
  \item \(g\bmod p\) 全分裂，
  \item \(C\bmod p\) 全分裂，
\end{enumerate}
则对应素数集合自然密度为
\[
\frac{1}{10!}\cdot\frac{1}{6}\cdot\frac{1}{48}.
\]
并且事件
\[
A\cap B_{\mathrm{split}}\cap\{\text{\(C\bmod p\) 全分裂}\}
\]
的密度为
\[
\frac{28319}{44800}\cdot\frac{1}{6}\cdot\frac{1}{48}
=\frac{28319}{12902400}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:killo-leyang-lattes-triple-product-galois} 与 Chebotarev 乘积律，
三组 Frobenius 条件的密度等于三边际密度之积。全分裂对应各群的恒等元共轭类，故第一式成立；第二式再代入推论 \ref{cor:killo-leyang-joint-density-explicit} 的边际因子即可。证毕。
\end{proof}

\begin{theorem}[三域 Frobenius 类函数统计的完全张量分解]\label{thm:killo-leyang-lattes-triple-class-function-factorization}
沿用定理 \ref{thm:killo-leyang-lattes-triple-product-galois} 的记号。对任意类函数
\[
f:S_{10}\to\CC,
\qquad
g:S_3\to\CC,
\qquad
h:S_4\times C_2\to\CC,
\]
定义群平均
\[
\langle f\rangle_{S_{10}}:=\frac{1}{10!}\sum_{\tau\in S_{10}}f(\tau),
\qquad
\langle g\rangle_{S_3}:=\frac{1}{6}\sum_{\upsilon\in S_3}g(\upsilon),
\]
\[
\langle h\rangle_{S_4\times C_2}:=
\frac{1}{48}\sum_{\xi\in S_4\times C_2}h(\xi).
\]
再记
\[
\pi^\ast_{\mathrm{tr}}(x):=
\#\left\{p\le x:\ p\ \text{在}\ L_{10}L_{\mathrm{LY}}L_4/\QQ\ \text{中不分歧}\right\}.
\]
对每个此类素数 \(p\)，记
\[
\sigma_{10}(p):=\mathrm{Frob}_p|_{L_{10}},
\qquad
\sigma_{3}(p):=\mathrm{Frob}_p|_{L_{\mathrm{LY}}},
\qquad
\sigma_{4}(p):=\mathrm{Frob}_p|_{L_4}.
\]
则有极限恒等式
\[
\boxed{\;
\lim_{x\to\infty}
\frac{1}{\pi^\ast_{\mathrm{tr}}(x)}
\sum_{\substack{p\le x\\ p\ \text{在}\ L_{10}L_{\mathrm{LY}}L_4/\QQ\ \text{中不分歧}}}
f\!\bigl(\sigma_{10}(p)\bigr)\,
g\!\bigl(\sigma_3(p)\bigr)\,
h\!\bigl(\sigma_4(p)\bigr)
=
\langle f\rangle_{S_{10}}\,
\langle g\rangle_{S_3}\,
\langle h\rangle_{S_4\times C_2}.\;}
\]
特别地，若三者中至少一个是相应群上的非平凡不可约特征标，则上述平均值为 \(0\)；仅当三者全为平凡表示时，该平均值才等于 \(1\)。
\end{theorem}
\begin{proof}
把三者分别展开为各自群上共轭类指示函数的线性组合。于是被求平均的乘积可写成
\[
\sum_{C,D,E} a_C b_D c_E\,
\mathbf 1_{\{\sigma_{10}(p)\in C,\ \sigma_3(p)\in D,\ \sigma_4(p)\in E\}},
\]
其中 \(C\subset S_{10}\)、\(D\subset S_3\)、\(E\subset S_4\times C_2\) 遍历共轭类。
由定理 \ref{thm:killo-leyang-lattes-triple-product-galois}，三域合成扩张的伽罗瓦群为
\[
S_{10}\times S_3\times (S_4\times C_2).
\]
Chebotarev 密度定理在直积群上给出
\[
\delta(C,D,E)=\frac{|C|}{10!}\cdot \frac{|D|}{6}\cdot \frac{|E|}{48}.
\]
故对极限求和后得到
\[
\sum_{C,D,E} a_C b_D c_E\frac{|C|}{10!}\frac{|D|}{6}\frac{|E|}{48}
=
\left(\sum_C a_C\frac{|C|}{10!}\right)
\left(\sum_D b_D\frac{|D|}{6}\right)
\left(\sum_E c_E\frac{|E|}{48}\right),
\]
这正是所需张量分解。

最后，对不可约特征标使用有限群平均恒等式
\[
\frac{1}{|G|}\sum_{g\in G}\chi(g)=0
\qquad(\chi\neq \mathbf 1),
\]
即可得到正交结论。证毕。
\end{proof}

\begin{proposition}[分歧素数集的硬分离与双分歧不可兼得]\label{prop:killo-leyang-ramification-hard-separation}
令 \(\mathfrak R_{10}\) 与 \(\mathfrak R_{\mathrm{LY}}\) 分别为 \(L_{10}/\QQ\)、\(L_{\mathrm{LY}}/\QQ\) 的分歧素数集。则
\[
\mathfrak R_{10}\cap \mathfrak R_{\mathrm{LY}}\subseteq \{3\}.
\]
\end{proposition}
\begin{proof}
由结论 \ref{con:xi-terminal-zm-leyang-cubic-arithmetic}，\(\mathfrak R_{\mathrm{LY}}\subseteq\{3,37\}\)。
另一方面，命题 \ref{prop:fold-gauge-anomaly-P10-galois-discriminant} 给出
\[
\mathrm{Disc}(P_{10})
=2^{38}\cdot 3^{24}\cdot 5^{2}\cdot 1931\cdot 9013^{2}\cdot 34319^{3},
\]
故 \(37\nmid \mathrm{Disc}(P_{10})\)。分裂域分歧素数只能落在多项式判别式素因子之内，因而 \(37\notin\mathfrak R_{10}\)。
于是
\(
\mathfrak R_{10}\cap \mathfrak R_{\mathrm{LY}}
\subseteq
\{3\}
\)。
证毕。
\end{proof}
