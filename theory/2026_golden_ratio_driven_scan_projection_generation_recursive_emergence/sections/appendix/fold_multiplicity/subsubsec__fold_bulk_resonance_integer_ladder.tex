\paragraph{解析闭包与实零结构}
\begin{theorem}[共振梯的整函数闭包、函数方程与实零点结构]\label{thm:fold-resonance-entire-lp}
定义整函数候选
\[
\mathcal{F}_\varphi(z):=\prod_{n=2}^{\infty}\cos(\pi z\,\varphi^{-n}),\qquad z\in\CC.
\]
则成立：
\begin{enumerate}
  \item \(\mathcal{F}_\varphi\) 为整函数，且为指数型（阶为 \(1\)），并满足
  \[
  |\mathcal{F}_\varphi(z)|\le \exp\!\bigl(\pi|\Im z|\bigr)\qquad(\forall z\in\CC).
  \]
  \item \(\mathcal{F}_\varphi\) 满足严格自相似函数方程
  \[
  \mathcal{F}_\varphi(z)=\cos\!\Bigl(\frac{\pi z}{\varphi^2}\Bigr)\,\mathcal{F}_\varphi\!\Bigl(\frac{z}{\varphi}\Bigr).
  \]
  \item \(\mathcal{F}_\varphi\) 的零点全部为实且均为单零，零点集为
  \[
  \mathcal{Z}(\mathcal{F}_\varphi)=\Bigl\{(k+\tfrac12)\varphi^n:\ k\in\ZZ,\ n\ge 2\Bigr\}.
  \]
  \item \(\mathcal{F}_\varphi\) 属于 Laguerre--P\'olya 类。
\end{enumerate}
并且对任意 \(u\in\RR\)，有
\[
C_\varphi(u)=|\mathcal{F}_\varphi(u)|.
\]
\end{theorem}

\begin{proof}
记 \(w_n(z):=\pi z\varphi^{-n}\)（\(n\ge 2\)）。对任意紧集 \(K_R:=\{z:|z|\le R\}\)，有
\(
|w_n(z)|\le \pi R\varphi^{-n}
\)。
利用估计
\[
|\cos w-1|\le \frac{|w|^2}{2}e^{|\Im w|}\qquad(\forall w\in\CC),
\]
可得
\[
\sum_{n\ge 2}\sup_{z\in K_R}|\cos w_n(z)-1|
\le
\frac{\pi^2R^2}{2}e^{\pi R}\sum_{n\ge 2}\varphi^{-2n}<\infty.
\]
故无穷乘积在紧集上一致收敛并定义整函数。

再由 \(|\cos(x+iy)|\le e^{|y|}\) 得
\[
|\mathcal{F}_\varphi(z)|
\le
\prod_{n\ge 2}\exp\!\bigl(\pi|\Im z|\,\varphi^{-n}\bigr)
=
\exp\!\Bigl(\pi|\Im z|\sum_{n\ge 2}\varphi^{-n}\Bigr)
=\exp\!\bigl(\pi|\Im z|\bigr),
\]
其中使用 \(\sum_{n\ge 2}\varphi^{-n}=1\)。从而得到指数型上界。

函数方程由首因子分离得到：
\[
\mathcal{F}_\varphi(z)
=
\cos\!\Bigl(\frac{\pi z}{\varphi^2}\Bigr)\prod_{n\ge 3}\cos(\pi z\varphi^{-n})
=
\cos\!\Bigl(\frac{\pi z}{\varphi^2}\Bigr)\,\mathcal{F}_\varphi\!\Bigl(\frac{z}{\varphi}\Bigr).
\]

每个因子 \(\cos(\pi z\varphi^{-n})\) 的零点为 \(\{(k+\tfrac12)\varphi^n:k\in\ZZ\}\)。
若 \(n\neq m\) 时两层零点重合，则有
\[
\varphi^{n-m}=\frac{\ell+\tfrac12}{k+\tfrac12}\in\QQ,
\]
与 \(\varphi\) 的无理性矛盾，故不同层零点不相交；在任一零点处仅一因子消失，故皆为单零。

令有限部分积
\[
\mathcal{F}_{\varphi,N}(z):=\prod_{n=2}^{N}\cos(\pi z\varphi^{-n}).
\]
每个 \(\mathcal{F}_{\varphi,N}\) 为仅有实零点的实整函数，故属于 Laguerre--P\'olya 类。由前述紧集一致收敛，
\(\mathcal{F}_{\varphi,N}\to\mathcal{F}_\varphi\)；
Laguerre--P\'olya 类对局部一致极限封闭，故 \(\mathcal{F}_\varphi\in\mathrm{LP}\)。

最后，对 \(u\in\RR\) 取绝对值即得
\[
|\mathcal{F}_\varphi(u)|
=
\prod_{n=2}^{\infty}|\cos(\pi u\varphi^{-n})|
=
C_\varphi(u).
\]
证毕。
\end{proof}

\begin{proposition}[共振梯常数的截断误差上界]\label{prop:fold-resonance-truncation-error}
\leanverified{paper\_fold\_resonance\_truncation\_error}
对 \(u\in\RR\) 与 \(N\ge 2\)，记
\[
C_{\varphi,N}(u):=\prod_{n=2}^{N}\Bigl|\cos\bigl(\pi u\varphi^{-n}\bigr)\Bigr|.
\]
若满足
\[
\pi|u|\varphi^{-(N+1)}\le \frac12,
\]
则有显式双边界
\[
C_{\varphi,N}(u)\,
\exp\!\Bigl(
-\frac{\pi^2u^2\,\varphi^{-2(N+1)}}{1-\varphi^{-2}}
\Bigr)
\le
C_\varphi(u)
\le
C_{\varphi,N}(u).
\]
其中右侧不等式对所有 \(N\ge 2\) 恒成立。
\end{proposition}

\begin{proof}
由 \(|\cos x|\le 1\)（\(x\in\RR\)）直接得到上界
\(
C_\varphi(u)\le C_{\varphi,N}(u)
\)。

下界方面，令 \(x_n:=\pi|u|\varphi^{-n}\)。
当 \(n\ge N+1\) 时有 \(x_n\le 1/2\)。
在区间 \([0,1/2]\) 上记
\[
h(x):=x^2+\log(\cos x),
\]
则 \(h(0)=0\) 且
\[
h'(x)=2x-\tan x.
\]
又 \(g(x):=2x-\tan x\) 满足 \(g(0)=0\)、\(g'(x)=2-\sec^2x>0\)（\(0\le x\le 1/2<\pi/4\)），故 \(h'(x)\ge 0\)。
于是 \(h(x)\ge 0\)，等价于
\[
-\log(\cos x)\le x^2\qquad(0\le x\le 1/2).
\]
因此
\[
-\sum_{n>N}\log\Bigl|\cos\bigl(\pi u\varphi^{-n}\bigr)\Bigr|
\le
\sum_{n>N}x_n^2
=
\pi^2u^2\sum_{n>N}\varphi^{-2n}
=
\frac{\pi^2u^2\,\varphi^{-2(N+1)}}{1-\varphi^{-2}}.
\]
指数化并与前 \(N\) 项分离即得
\[
C_\varphi(u)
=
C_{\varphi,N}(u)\prod_{n>N}\Bigl|\cos\bigl(\pi u\varphi^{-n}\bigr)\Bigr|
\ge
C_{\varphi,N}(u)\exp\!\Bigl(
-\frac{\pi^2u^2\,\varphi^{-2(N+1)}}{1-\varphi^{-2}}
\Bigr).
\]
证毕。
\end{proof}

\begin{theorem}[bulk 共振梯的全整数推广]\label{thm:fold-critical-resonance-constant-integer-ladder}
\leanverified{paper\_fold\_critical\_resonance\_constant\_integer\_ladder}
对每个固定整数 \(u\in\ZZ\)，乘积 \(C_\varphi(u)\) 收敛且严格为正。
并且若对每个 \(m\ge 1\) 定义同余代表
\[
k_m(u)\equiv uF_m\pmod{F_{m+2}},
\qquad 0\le k_m(u)<F_{m+2},
\]
则存在极限
\[
\lim_{m\to\infty}a_m\!\bigl(k_m(u)\bigr)=C_\varphi(u).
\]
同样地，亦有
\[
\lim_{m\to\infty}a_m\!\bigl(uF_{m+1}\bigr)=C_\varphi(u).
\]
当 \(u=1\) 时，上述结论退化为定理 \ref{thm:fold-critical-resonance-constant}（即 \(C_\varphi(1)=C_\varphi\)）。
\end{theorem}

\begin{proof}
先证乘积收敛与正性。
对任意固定 \(u\)，当 \(x\) 足够小且 \(|x|\le 1/4\) 时存在常数 \(C>0\) 使
\(
|\log|\cos(\pi u x)||\le C u^2 x^2
\)。
取 \(x=\varphi^{-n}\) 得
\(
\sum_{n\ge 2}u^2\varphi^{-2n}<\infty
\)，
故对数级数绝对收敛，从而乘积收敛。
由于每个因子均为正且收敛到 \(1\)，故极限严格为正。
\par
再证幅度极限。
由定义 \ref{def:fold-normalized-amplitude}，
\[
a_m\!\bigl(k_m(u)\bigr)
=\prod_{j=1}^{m}\Bigl|\cos\Bigl(\pi k_m(u)\,F_{j+1}/F_{m+2}\Bigr)\Bigr|.
\]
以 \(k_m(u)\equiv uF_m\ (\mathrm{mod}\ F_{m+2})\) 代入，并用引理 \ref{lem:fold-resonance-docagne} 的同余压缩
\(
F_mF_{j+1}\equiv (-1)^{j+1}F_{m+1-j}\ (\mathrm{mod}\ F_{m+2})
\)
以及余弦偶性，得到精确表示
\[
a_m\!\bigl(k_m(u)\bigr)
=\prod_{j=1}^{m}\Bigl|\cos\Bigl(\pi u\,\frac{F_{m+1-j}}{F_{m+2}}\Bigr)\Bigr|.
\]
固定 \(j\) 令 \(m\to\infty\)，由 Binet 渐近 \(F_n=\varphi^n/\sqrt 5+O(\varphi^{-n})\) 得
\(
F_{m+1-j}/F_{m+2}\to \varphi^{-(j+1)}
\)。
并且由前述二次控制可对对数级数使用支配收敛，故乘积极限存在且等于
\(
\prod_{j=1}^{\infty}|\cos(\pi u\,\varphi^{-(j+1)})|
=\prod_{n=2}^{\infty}|\cos(\pi u\,\varphi^{-n})|
=C_\varphi(u)
\)。
从而第一个极限成立。
\par
最后，因 \(F_{m+1}\equiv -F_m\pmod{F_{m+2}}\) 与余弦偶性，知
\(
a_m(uF_{m+1})=a_m(-uF_m)=a_m(uF_m)=a_m(k_m(u))
\)，
故第二个极限同样成立。证毕。
\end{proof}

\paragraph{Pisot--Bernoulli 卷积表象：共振常数的测度化}

令 \(q=\varphi^{-1}\in(0,1)\)。
定义随机级数
\[
X=\sum_{n=2}^{\infty}\varepsilon_n q^n,
\qquad
\mathbb{P}(\varepsilon_n=1)=\mathbb{P}(\varepsilon_n=-1)=\tfrac12,
\]
并令 \(\mu_q^{(2)}\) 为 \(X\) 的分布（一个起始指标为 \(n=2\) 的 Pisot 参数 Bernoulli 卷积移位版本）。
则其 Fourier--Stieltjes 变换满足乘积公式
\[
\widehat{\mu_q^{(2)}}(t)
=\int_{\RR}e^{\mathrm{i}tx}\,d\mu_q^{(2)}(x)
=\prod_{n=2}^{\infty}\cos(q^n t).
\]
从而对所有整数 \(u\in\ZZ\)，有恒等式
\[
C_\varphi(u)=\Bigl|\widehat{\mu_q^{(2)}}(\pi u)\Bigr|.

并且 \S\ref{subsubsec:fold-critical-resonance} 的 edge 极限常数
\(
C_\varphi^{\mathrm{edge}}(k)=\prod_{n=1}^{\infty}|\cos(\pi k\varphi^{-n})|

亦可视为同一 Bernoulli 卷积在算术格点上的 Fourier 采样（仅起始指标不同）。
\end{theorem}

\begin{proof}
由独立性与 \(\mathbb{E}(e^{\mathrm{i}t\varepsilon_n q^n})=\cos(tq^n)\)，有
\(
\widehat{\mu_q^{(2)}}(t)=\prod_{n=2}^{\infty}\cos(q^n t)
\)。
取 \(t=\pi u\) 并取绝对值，即得
\(
|\widehat{\mu_q^{(2)}}(\pi u)|=\prod_{n=2}^{\infty}|\cos(\pi u\varphi^{-n})|=C_\varphi(u)

最后一段仅为指标平移的直接比较。证毕。
\end{proof}

\paragraph{Fibonacci 型递推方向的极限分解：两个可审计的正极限常数}
\begin{theorem}[Fibonacci 型整数递推方向的共振梯极限分解]\label{thm:fold-resonance-ladder-fibonacci-directional-limit}
令 \(\varphi=(1+\sqrt5)/2\)，并令 \(C_\varphi(u)\) 由定义 \ref{def:fold-critical-resonance-ladder} 给出。
取任意整数初值对 \((u_0,u_1)\in\ZZ^2\setminus\{(0,0)\}\)，并由 Fibonacci 递推
\(
u_{n+2}=u_{n+1}+u_n
\)
生成整数序列 \((u_n)_{n\ge 0}\)。
记
\[
\psi:=\frac{1-\sqrt5}{2}=-\varphi^{-1},\qquad
\alpha:=\frac{u_1-u_0\psi}{\sqrt5},\qquad
\beta:=\frac{u_0\varphi-u_1}{\sqrt5}.
\]
则极限
\(
\lim_{n\to\infty} C_\varphi(u_n)
\)
存在且严格为正，并满足显式乘积公式
\[
\lim_{n\to\infty}C_\varphi(u_n)
=\Bigl|\cos(\pi\alpha)\Bigr|\cdot
\prod_{k=1}^{\infty}\Bigl|\cos\bigl(\pi\alpha\varphi^{-k}\bigr)\Bigr|\cdot
\Bigl|\cos\bigl(\pi\beta\varphi^{-k}\bigr)\Bigr|.
\]
\end{theorem}

\begin{proof}
由 Binet 表示，存在实数 \(\alpha,\beta\) 使得对所有 \(n\ge 0\),
\[
u_n=\alpha\varphi^n+\beta\psi^n,
\qquad
\psi=\frac{1-\sqrt5}{2}.
\]
将 \(n=0,1\) 代入并解得系数正是题设所定义的 \(\alpha,\beta\)。
此外 \(u_0=\alpha+\beta\in\ZZ\)。
\par
对每个 \(n\) 写
\[
C_\varphi(u_n)=\prod_{m=2}^{\infty}\bigl|\cos(\pi u_n\varphi^{-m})\bigr|
=\Bigl(\prod_{m=2}^{n-1}\bigl|\cos(\pi u_n\varphi^{-m})\bigr|\Bigr)\cdot
\Bigl(\prod_{m=n}^{\infty}\bigl|\cos(\pi u_n\varphi^{-m})\bigr|\Bigr).
\]
先处理尾部乘积。对每个固定 \(k\ge 0\)，令 \(m=n+k\)，则
\[
u_n\varphi^{-(n+k)}
=\alpha\varphi^{-k}+\beta\psi^n\varphi^{-(n+k)}
\xrightarrow[n\to\infty]{}\alpha\varphi^{-k},
\]
从而逐项收敛到 \(|\cos(\pi\alpha\varphi^{-k})|\)。
由于当 \(k\) 足够大时 \(|\alpha\varphi^{-k}|\) 足够小，并存在常数 \(C>0\) 使得
\(
|\log|\cos(\pi x)||\le Cx^2
\)
对所有充分小的 \(|x|\) 成立，
而 \(\sum_{k\ge 0}\varphi^{-2k}<\infty\)，故可对对数级数使用支配收敛，得到
\[
\prod_{m=n}^{\infty}\bigl|\cos(\pi u_n\varphi^{-m})\bigr|
\xrightarrow[n\to\infty]{}
\prod_{k=0}^{\infty}\bigl|\cos(\pi\alpha\varphi^{-k})\bigr|
=\bigl|\cos(\pi\alpha)\bigr|\prod_{k=1}^{\infty}\bigl|\cos(\pi\alpha\varphi^{-k})\bigr|.
\]
\par
再处理首段乘积。对 \(2\le m\le n-1\) 令 \(k:=n-m\)（于是 \(1\le k\le n-2\)）。
由于 \(u_k\in\ZZ\) 且 \(|\cos(\pi(x+u_k))|=|\cos(\pi x)|\)，有
\[
\bigl|\cos(\pi u_n\varphi^{-m})\bigr|
=\bigl|\cos\bigl(\pi(u_n\varphi^{-(n-k)}-u_k)\bigr)\bigr|.
\]
利用 Binet 表示，
\[
u_n\varphi^{-(n-k)}-u_k
=\beta\bigl(\psi^n\varphi^{-(n-k)}-\psi^k\bigr)
\xrightarrow[n\to\infty]{}-\beta\psi^k.
\]
又因 \(\psi=-\varphi^{-1}\) 且 \(\cos\) 为偶函数，得到逐项极限
\(
\bigl|\cos\bigl(\pi(u_n\varphi^{-(n-k)}-u_k)\bigr)\bigr|
\to
\bigl|\cos(\pi\beta\varphi^{-k})\bigr|
\)。
同样地，对 \(k\) 足够大时 \(|\beta\varphi^{-k}|\) 足够小并可对对数级数支配收敛，
从而
\[
\prod_{m=2}^{n-1}\bigl|\cos(\pi u_n\varphi^{-m})\bigr|
\xrightarrow[n\to\infty]{}
\prod_{k=1}^{\infty}\bigl|\cos(\pi\beta\varphi^{-k})\bigr|.
\]
合并两段极限即得结论。
严格正性由 \(\alpha,\beta\) 与 \(\varphi^{-k}\) 的无理性确保各因子均非零，
并由前述对数级数绝对可和性推出乘积不退化为零。证毕。
\end{proof}

\begin{corollary}[Fibonacci 与 Lucas 子列的极限常数]\label{cor:fold-resonance-ladder-fib-luc-limits}
令 \((F_n)_{n\ge 0}\) 为 Fibonacci 数列（\(F_0=0,F_1=1\)），\((L_n)_{n\ge 0}\) 为 Lucas 数列（\(L_0=2,L_1=1\)），均满足递推 \(x_{n+2}=x_{n+1}+x_n\)。
则存在严格正极限
\[
\lim_{n\to\infty}C_\varphi(F_n)
=\Bigl|\cos\!\Bigl(\frac{\pi}{\sqrt5}\Bigr)\Bigr|
\cdot \prod_{k=1}^{\infty}\Bigl|\cos\!\Bigl(\frac{\pi}{\sqrt5\,\varphi^{k}}\Bigr)\Bigr|^{2},
\]
\[
\lim_{n\to\infty}C_\varphi(L_n)

\begin{corollary}[整数方向的非衰减与无单一极限]\label{cor:fold-resonance-ladder-no-decay-no-limit}

\paragraph{全共振闭包常数的发散与碰撞尺度的无界性}
\begin{theorem}[全共振闭包常数 \texorpdfstring{$\Sigma_\varphi$}{Sigma_phi} 的发散]\label{thm:fold-sigma-phi-diverges}
\leanverified{paper\_fold\_sigma\_phi\_diverges}
形式定义
\[
\Sigma_\varphi:=1+2\sum_{u=1}^{\infty}C_\varphi(u)^2
\]
在黄金比例情形下必然发散；更具体地，存在常数 \(c_*>0\) 与 \(N_0\) 使得对所有 \(N\ge N_0\),
\[
\sum_{u=1}^{N}C_\varphi(u)^2\ \ge\ c_*^2\,\card{\{n\ge 1:\ F_n\le N\}}\ \gtrsim\ (\log N).
\]
从而 \(\Sigma_\varphi=+\infty\)。
\end{theorem}

\begin{proof}
由推论 \ref{cor:fold-resonance-ladder-fib-luc-limits}，存在 \(n_0\) 使得对所有 \(n\ge n_0\) 都有 \(C_\varphi(F_n)\ge c_*>0\)。
于是对任意足够大的 \(N\),
\[
\sum_{u=1}^{N}C_\varphi(u)^2
\ge
\sum_{\substack{n\ge n_0\\ F_n\le N}}C_\varphi(F_n)^2
\ge
c_*^2\,\card{\{n\ge n_0:\ F_n\le N\}}.
\]
又由 \(F_n\asymp \varphi^n\) 可知 \(\card{\{n:\ F_n\le N\}}\asymp \log_\varphi N\)，从而给出至少对数级的下界并推出级数发散。证毕。
\end{proof}

\begin{corollary}[碰撞尺度的无界增长]\label{cor:fold-collision-scale-diverges}
碰撞尺度满足
\[
F_{m+2}\,\mathsf{Col}_m\ \xrightarrow[m\to\infty]{}\ +\infty.
\]
\end{corollary}

\begin{proof}
由推论 \ref{cor:fold-collision-gap-resonance-ladder-U}，对任意 \(U\in\NN\) 有

\end{proof}

\begin{corollary}[算术格点 Fourier 采样的非平方可和性]\label{cor:fold-bernoulli-fourier-sampling-not-l2}
令 \(\mu_q^{(2)}\) 如定理 \ref{thm:fold-pisot-bernoulli-convolution-representation}，其中 \(q=\varphi^{-1}\)。

\end{corollary}

\begin{proof}

若 \(\nu\) 具有 \(L^2(\mathbb{T}_2)\) 密度，则由 Parseval 恒等式其 Fourier 系数必为 \(\ell^2\)，与前段矛盾。证毕。
\end{proof}
