\begin{proposition}[二阶角色的反射湮灭与模二完全平衡]\label{prop:fold-second-character-reflection-annihilation}
记
\[
N:=F_{m+2},\qquad
G:=\ZZ/(N\ZZ),
\]
并沿用定理 \ref{thm:fold-congruence-universal-property} 的参数化 \(w:G\to X_m\)。
设
\[
\tau_m(r)\equiv (F_{m+1}-2)-r\pmod N.
\]
由命题 \ref{prop:fold-complement-duality}，有
\[
d_m(w(r))=d_m(w(\tau_m(r)))\qquad(\forall r\in G).
\]
若 \(N\) 为偶数（等价于 \(m\equiv 1\pmod 3\)），则对任意整数 \(q\ge 1\) 恒有
\[
\sum_{r=0}^{N-1}(-1)^r\,d_m(w(r))^q=0.
\]
等价地，函数
\[
r\longmapsto d_m(w(r))^q
\]
在循环群 \(G\) 的 Fourier 系数于频率 \(N/2\) 处严格消失：
\[
\widehat{d_m^{(q)}}(N/2)=0,\qquad
d_m^{(q)}(r):=d_m(w(r))^q.
\]
进一步，偶剩余类与奇剩余类在权重 \(d_m(w(r))^q\) 下总质量完全相等：
\[
\sum_{\substack{0\le r\le N-1\\ r\equiv 0\ (\mathrm{mod}\ 2)}} d_m(w(r))^q
=
\sum_{\substack{0\le r\le N-1\\ r\equiv 1\ (\mathrm{mod}\ 2)}} d_m(w(r))^q.
\]
\end{proposition}
\begin{proof}
记 \(a:=F_{m+1}-2\)，并令
\[
T_q:=\sum_{r=0}^{N-1}(-1)^r\,d_m(w(r))^q.
\]
由 \(\tau_m(r)\equiv a-r\) 的换元（在 \(G\) 上为置换）与命题 \ref{prop:fold-complement-duality} 的多重度守恒，得
\[
\begin{aligned}
T_q
&=\sum_{r=0}^{N-1}(-1)^{a-r}\,d_m(w(a-r))^q\\
&=(-1)^a\sum_{r=0}^{N-1}(-1)^r\,d_m(w(r))^q
=(-1)^a T_q.
\end{aligned}
\]
当 \(N\) 为偶数时，\(F_{m+2}\) 为偶数，按 Fibonacci 奇偶周期 \(F_n\equiv 0\ (\mathrm{mod}\ 2)\iff 3\mid n\)，等价于 \(m\equiv 1\ (\mathrm{mod}\ 3)\)，从而 \(F_{m+1}\) 为奇数，故 \(a=F_{m+1}-2\) 为奇数。
于是 \((-1)^a=-1\)，得到 \(T_q=-T_q\)，即 \(T_q=0\)。

而
\[
\widehat{d_m^{(q)}}(N/2)
=
\sum_{r=0}^{N-1}d_m^{(q)}(r)\exp\!\left(-2\pi i\frac{(N/2)r}{N}\right)
=
\sum_{r=0}^{N-1}(-1)^r d_m(w(r))^q
=T_q=0.
\]
最后一式与偶奇类权重和相等互为等价改写。证毕。
\end{proof}
