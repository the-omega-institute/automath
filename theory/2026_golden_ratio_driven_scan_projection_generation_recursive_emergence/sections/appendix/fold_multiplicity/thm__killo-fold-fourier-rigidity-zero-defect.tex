\begin{theorem}[折叠多重度谱的中心反射对称与 Fourier 相位刚性]\label{thm:killo-fold-fourier-reflection-phase-rigidity}
令
\[
M:=F_{m+2},
\qquad
D_m(x):=\sum_{r\in \ZZ/(M\ZZ)}d_m(r)\,x^r\in \CC[x]/(x^M-1),
\]
并记
\[
S_m:=\sum_{j=1}^{m}F_{j+1}=F_{m+3}-2\equiv F_{m+1}-2\pmod M.
\]
则有严格反射对称
\[
d_m(r)=d_m(S_m-r)
\qquad
\bigl(r\in \ZZ/(M\ZZ)\bigr).
\]
从而离散 Fourier 系数满足相位刚性
\[
\widehat d_m(k)=e^{-2\pi i kS_m/M}\,\overline{\widehat d_m(k)}
\qquad
\bigl(k\in \ZZ/(M\ZZ)\bigr),
\]
等价地，
\[
e^{\pi i kS_m/M}\widehat d_m(k)\in \RR.
\]
更精确地，
\[
\widehat d_m(k)
=
e^{-\pi i kS_m/M}\,
2^m\prod_{j=1}^{m}\cos\!\left(\frac{\pi kF_{j+1}}{M}\right).
\]
\leanverified{paper\_killo\_fold\_fourier\_reflection\_phase\_rigidity}
\end{theorem}
\begin{proof}
由命题 \ref{prop:fold-multiplicity-group-algebra} 的群环恒等式，
\[
D_m(x^{-1})
=
\prod_{j=1}^{m}\bigl(1+x^{-F_{j+1}}\bigr)
=
x^{-S_m}\prod_{j=1}^{m}\bigl(1+x^{F_{j+1}}\bigr)
\equiv
x^{-S_m}D_m(x)
\pmod{x^M-1}.
\]
两边同乘 \(x^{S_m}\) 得
\[
x^{S_m}D_m(x^{-1})\equiv D_m(x)\pmod{x^M-1}.
\]
在基 \(1,x,\dots,x^{M-1}\) 下比较系数，即得
\(
d_m(r)=d_m(S_m-r)
\)
对一切 \(r\in \ZZ/(M\ZZ)\) 成立。

再由 Fourier 变换定义，
\[
\widehat d_m(k)
=
\sum_{r}d_m(r)e^{-2\pi i kr/M}
=
\sum_{r}d_m(S_m-r)e^{-2\pi i kr/M}.
\]
将变量替换为 \(r\mapsto S_m-r\) 后得到
\[
\widehat d_m(k)
=
e^{-2\pi i kS_m/M}\sum_{r}d_m(r)e^{2\pi i kr/M}
=
e^{-2\pi i kS_m/M}\,\overline{\widehat d_m(k)}.
\]
因此
\[
\overline{e^{\pi i kS_m/M}\widehat d_m(k)}
=
e^{-\pi i kS_m/M}\,\overline{\widehat d_m(k)}
=
e^{\pi i kS_m/M}\widehat d_m(k),
\]
故 \(e^{\pi i kS_m/M}\widehat d_m(k)\in\RR\)。

最后，由定理 \ref{thm:fold-spectrum-factorization}
\[
\widehat d_m(k)=\prod_{j=1}^{m}\left(1+e^{-2\pi i kF_{j+1}/M}\right),
\]
并对每个因子应用恒等式
\(
1+e^{-i\theta}=e^{-i\theta/2}\,2\cos(\theta/2)
\)，
便得
\[
\widehat d_m(k)
=
e^{-\pi i k\sum_{j=1}^{m}F_{j+1}/M}\,
2^m\prod_{j=1}^{m}\cos\!\left(\frac{\pi kF_{j+1}}{M}\right)
=
e^{-\pi i kS_m/M}\,
2^m\prod_{j=1}^{m}\cos\!\left(\frac{\pi kF_{j+1}}{M}\right).
\]
证毕。
\end{proof}

\begin{theorem}[谱零点集合的同余与仿射陪集刻画]\label{thm:killo-fold-zero-spectrum-affine-congruence}
\leanverified{paper\_killo\_fold\_zero\_spectrum\_affine\_congruence}
令
\[
M:=F_{m+2},
\qquad
\mathcal Z_m:=\{\,k\in \ZZ/(M\ZZ):\ \widehat d_m(k)=0\,\}.
\]
对每个 \(1\le j\le m\)，定义
\[
\mathcal Z_{m,j}
:=
\left\{
k\in \ZZ/(M\ZZ):
e^{-2\pi i kF_{j+1}/M}=-1
\right\}.
\]
则
\[
\mathcal Z_m=\bigcup_{j=1}^{m}\mathcal Z_{m,j}.
\]
若 \(M\) 为偶数，则
\[
\mathcal Z_{m,j}
=
\left\{
k\in \ZZ/(M\ZZ):
kF_{j+1}\equiv \frac{M}{2}\pmod M
\right\}.
\]
进一步，记
\[
g_{m,j}:=\gcd(F_{j+1},M)=F_{\gcd(j+1,m+2)}.
\]
则成立：
\begin{enumerate}
\normalfont
  \item \(\mathcal Z_{m,j}\neq\varnothing\) 当且仅当 \(g_{m,j}\mid M/2\)；
  \item 一旦 \(\mathcal Z_{m,j}\neq\varnothing\)，任取其一解 \(k_{m,j}\)，都有
  \[
  \mathcal Z_{m,j}
  =
  k_{m,j}+\frac{M}{g_{m,j}}\ZZ/M\ZZ,
  \qquad
  |\mathcal Z_{m,j}|=g_{m,j};
  \]
  \item 若 \(3\nmid (m+2)\)（等价于 \(M\) 为奇数），则
  \[
  \mathcal Z_m=\varnothing.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-spectrum-factorization}，
\[
\widehat d_m(k)=\prod_{j=1}^{m}\left(1+e^{-2\pi i kF_{j+1}/M}\right),
\]
故 \(\widehat d_m(k)=0\) 当且仅当存在某个 \(j\) 使
\(
e^{-2\pi i kF_{j+1}/M}=-1
\)。
于是
\(
\mathcal Z_m=\bigcup_{j=1}^{m}\mathcal Z_{m,j}
\)。

若 \(M\) 为偶数，则
\[
e^{-2\pi i kF_{j+1}/M}=-1
\iff
\frac{kF_{j+1}}{M}\equiv \frac12\pmod 1
\iff
kF_{j+1}\equiv \frac{M}{2}\pmod M.
\]
因此 \(\mathcal Z_{m,j}\) 正是该线性同余的解集。
标准同余理论表明：该方程可解当且仅当
\(
\gcd(F_{j+1},M)\mid M/2
\)，
即 \(g_{m,j}\mid M/2\)；
若可解，则全体解为任一特解 \(k_{m,j}\) 加上齐次解子群
\(
(M/g_{m,j})\ZZ/M\ZZ
\)，
从而
\(
|\mathcal Z_{m,j}|=g_{m,j}
\)。
而
\[
g_{m,j}
=
\gcd(F_{j+1},F_{m+2})
=
F_{\gcd(j+1,m+2)}
\]
由 Fibonacci 最大公因子恒等式给出。

最后，若 \(M\) 为奇数，则 \(-1\) 的阶为 \(2\)，不可能属于 \(M\)-次单位根群；
因而对任意 \(j\) 都不可能有
\(
e^{-2\pi i kF_{j+1}/M}=-1
\)。
于是全部 \(\mathcal Z_{m,j}\) 皆空，从而 \(\mathcal Z_m=\varnothing\)。
证毕。
\end{proof}

\begin{corollary}[多重度卷积算子的可逆--缺陷二分]\label{cor:killo-fold-multiplicity-convolution-invertibility-defect}
\leanverified{paper\_killo\_fold\_multiplicity\_convolution\_invertibility\_defect}
令 \(G:=\ZZ/(F_{m+2}\ZZ)\)，并定义卷积算子
\[
T_m:\CC[G]\to \CC[G],
\qquad
(T_mf)(r):=\sum_{s\in G}d_m(r-s)f(s).
\]
则 Fourier 基把 \(T_m\) 对角化，并且
\[
\ker(T_m)
=
\bigoplus_{k\in \mathcal Z_m}\CC\cdot e_k,
\qquad
\dim_{\CC}\ker(T_m)=|\mathcal Z_m|.
\]
因此出现严格二分：
\begin{enumerate}
\normalfont
  \item 若 \(3\nmid(m+2)\)，则 \(T_m\) 可逆；
  \item 若 \(3\mid(m+2)\)，则
  \[
  0\le
  \frac{\dim_{\CC}\ker(T_m)}{F_{m+2}}
  =
  \frac{|\mathcal Z_m|}{F_{m+2}}
  \le
  \tau(m+2)\,\frac{F_{\lfloor (m+2)/2\rfloor}}{F_{m+2}}
  =
  \tau(m+2)\,O(\varphi^{-m/2}).
  \]
\end{enumerate}
换言之，多重度卷积算子在奇模长上严格可逆；在偶模长上，其不可逆方向至多占频谱空间的指数稀薄部分。
\end{corollary}
\begin{proof}
对角色基函数 \(e_k(r):=e^{2\pi i kr/F_{m+2}}\)，有限 Abel 群上的离散 Fourier 变换给出
\[
T_me_k=\widehat d_m(k)\,e_k.
\]
因此
\[
\ker(T_m)=\bigoplus_{k:\,\widehat d_m(k)=0}\CC\cdot e_k
=
\bigoplus_{k\in \mathcal Z_m}\CC\cdot e_k,
\]
并得到 \(\dim_{\CC}\ker(T_m)=|\mathcal Z_m|\)。
若 \(3\nmid(m+2)\)，则由定理 \ref{thm:killo-fold-zero-spectrum-affine-congruence} 有 \(\mathcal Z_m=\varnothing\)，故 \(T_m\) 可逆。
若 \(3\mid(m+2)\)，则由定理 \ref{thm:fold-zero-density-sparse} 得所示上界。
证毕。
\end{proof}
