\begin{theorem}[稳定类型分布的 Rényi--2 常数级均匀性缺口]\label{thm:killo-fold-renyi2-uniformity-gap}
\leanverified{paper\_killo\_fold\_renyi2\_uniformity\_gap}
沿用附录 \ref{app:fold-multiplicity} 的记号。令
\[
M_m:=F_{m+2},
\qquad
p_m(r):=\frac{d_m(r)}{2^m}\quad \bigl(r\in \ZZ/(M_m\ZZ)\bigr),
\]
并令 \(u_m\) 为 \(\ZZ/(M_m\ZZ)\) 上的均匀分布。则 Rényi--2 散度满足精确恒等式
\[
D_2(p_m\Vert u_m)=\log\!\bigl(M_m\,\mathsf{Col}_m\bigr).
\]
从而
\[
\liminf_{m\to\infty}D_2(p_m\Vert u_m)\ge \log(1+2C_\varphi^2)>0.
\]
特别地，由
\(
C_\varphi\approx 0.2244178274047294
\)
可得
\[
\log(1+2C_\varphi^2)\approx 0.09597061849768761\ \text{nats}
\approx 0.13845633537766053\ \text{bits}.
\]
\end{theorem}
\begin{proof}
由 Rényi--2 散度定义，
\[
D_2(p_m\Vert u_m)
=
\log\sum_{r}\frac{p_m(r)^2}{u_m(r)}.
\]
由于 \(u_m(r)\equiv 1/M_m\)，故
\[
D_2(p_m\Vert u_m)
=
\log\!\left(M_m\sum_r p_m(r)^2\right)
=
\log\!\bigl(M_m\,\mathsf{Col}_m\bigr).
\]
再由推论 \ref{cor:fold-critical-resonance-gap}，
\[
M_m\,\mathsf{Col}_m
=
F_{m+2}\,\mathsf{Col}_m
\ge
1+2C_\varphi^2+o(1).
\]
对数函数单调，遂得
\[
\liminf_{m\to\infty}D_2(p_m\Vert u_m)\ge \log(1+2C_\varphi^2)>0.
\]
最后把
\(
1+2C_\varphi^2\approx 1.1007267225141177
\)
代入即得数值近似。证毕。
\end{proof}

\begin{theorem}[常数级碰撞缺口的三模见证与双共振承载]\label{thm:killo-fold-three-mode-witness-critical-gap}
沿用定义 \ref{def:fold-normalized-amplitude} 的记号，定义
\[
W_m:=
\frac{1}{F_{m+2}}
\Bigl(1+a_m(F_m)^2+a_m(F_{m+1})^2\Bigr).
\]
则对一切 \(m\ge 1\) 都有
\[
\mathsf{Col}_m\ge W_m.
\]
并且
\[
F_{m+2}\Bigl(W_m-\frac{1}{F_{m+2}}\Bigr)
=
a_m(F_m)^2+a_m(F_{m+1})^2
\longrightarrow 2C_\varphi^2.
\]
此外，两条非平凡见证频率的归一化极限满足
\[
\frac{F_m}{F_{m+2}}\to \varphi^{-2},
\qquad
\frac{F_{m+1}}{F_{m+2}}\to \varphi^{-1}.
\]
因此，常数级碰撞缺口的主阶部分完全由零模态与两条 Fibonacci 共振模态
\[
\{0,F_m,F_{m+1}\}
\]
见证，而其中真正承载非平凡主项的仅是 \(\varphi^{-2}\) 与 \(\varphi^{-1}\) 两条 bulk 共振频率。
\end{theorem}
\begin{proof}
由附录 \ref{app:fold-multiplicity} 中碰撞概率的谱恒等式，
\[
\mathsf{Col}_m
=
\frac{1}{F_{m+2}}\sum_{k\in \ZZ/(F_{m+2}\ZZ)}a_m(k)^2.
\]
保留三项 \(k=0,F_m,F_{m+1}\) 即得
\[
\mathsf{Col}_m
\ge
\frac{1}{F_{m+2}}
\Bigl(a_m(0)^2+a_m(F_m)^2+a_m(F_{m+1})^2\Bigr)
=
W_m,
\]
其中 \(a_m(0)=1\)。

再由定理 \ref{thm:fold-critical-resonance-constant}，
\[
a_m(F_m)\to C_\varphi,
\qquad
a_m(F_{m+1})\to C_\varphi.
\]
故
\[
F_{m+2}\Bigl(W_m-\frac{1}{F_{m+2}}\Bigr)
=
a_m(F_m)^2+a_m(F_{m+1})^2
\longrightarrow 2C_\varphi^2.
\]
最后，由 Fibonacci 比值极限
\(
F_{m+1}/F_m\to \varphi
\)
可得
\[
\frac{F_m}{F_{m+2}}\to \varphi^{-2},
\qquad
\frac{F_{m+1}}{F_{m+2}}\to \varphi^{-1}.
\]
证毕。
\end{proof}
