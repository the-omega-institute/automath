\begin{proposition}[末位约束下的截断右逆唯一性]\label{prop:fold-bin-minsector-lastbit1-section-unique}
\leanverified{paper\_fold\_bin\_minsector\_lastbit1\_section\_unique}
对任意 \(m\ge 2\)，令 \(X_m=\{w\in\{0,1\}^m:\ w\ \text{不含子词 }11\}\)，并令截断算子
\[
\tau_m:X_m\to X_{m-2},
\qquad
\tau_m(w_1\cdots w_m):=w_1\cdots w_{m-2}.
\]
设 \(\iota_{m-2}:X_{m-2}\hookrightarrow X_m\) 为单射且满足 \(\tau_m\circ \iota_{m-2}=\mathrm{id}_{X_{m-2}}\)。
若进一步要求
\[
\mathrm{Im}(\iota_{m-2})\subseteq \{w\in X_m:\ w_m=1\},
\]
则 \(\iota_{m-2}\) 被强制唯一，并且对所有 \(u\in X_{m-2}\) 有显式公式
\[
\iota_{m-2}(u)=u01.
\]
\leanverified{paper\_fold\_bin\_minsector\_lastbit1\_section\_unique}
\end{proposition}
\begin{proof}
取任意 \(u\in X_{m-2}\)，并令 \(w=\iota_{m-2}(u)\in X_m\)。
由 \(\tau_m(w)=u\) 知 \(w\) 的前 \(m-2\) 位固定为 \(u\)，从而 \(w=uab\)（其中 \(a,b\in\{0,1\}\)）。
由假设 \(w_m=b=1\)。
又因 \(w\in X_m\) 禁止相邻 \(11\)，故 \(a=w_{m-1}=0\)。
因此 \(w=u01\)，并且该扩展不依赖于额外选择，故 \(\iota_{m-2}\) 唯一且必为 \(u\mapsto u01\)。
证毕。
\end{proof}

\begin{remark}
命题 \ref{prop:fold-bin-minsector-lastbit1-section-unique} 给出的唯一候选与猜想 \ref{conj:fold-bin-min-sector-regular-subcover} 中的自然提升 \(\iota_{m-2}(v)=(v,0,1)\) 一致；因此在一旦把最小退化扇区的子覆盖像限制在末位为 \(1\) 的层时，提升映射不再具有自由度。
\end{remark}

\begin{corollary}[稳定层坐标代数的 Fibonacci 分裂与 \(K_0\) 坐标正合列]\label{cor:fold-stable-type-coordinate-algebra-fibonacci-splitting}
\leanverified{paper\_fold\_stable\_type\_coordinate\_algebra\_fibonacci\_splitting}
沿用命题 \ref{prop:fold-bin-minsector-lastbit1-section-unique} 的记号。
令坐标代数
\[
\mathcal{Z}_m:=\Span\{\Pi_w:\ w\in X_m\}\ \cong\ \CC^{X_m},
\]
其中 \(\Pi_w\) 为纤维投影（见 \eqref{eq:op-algebra-fold-quantum-channel-def}）。
则存在自然的 \(\ast\)-同构
\[
\boxed{\;\mathcal{Z}_m\ \cong\ \mathcal{Z}_{m-1}\ \oplus\ \mathcal{Z}_{m-2}\;}
\]
其中第一项对应末位为 \(0\) 的稳定类型嵌入 \(X_{m-1}\ni v\mapsto v0\in X_m\)，第二项对应末位为 \(1\) 的稳定类型通过唯一右逆 \(X_{m-2}\ni u\mapsto u01\in X_m\) 的嵌入。
并且在 \(K\)-理论层面给出自然正合列
\[
0\longrightarrow K_0(\mathcal{Z}_{m-2})\xrightarrow{\ \iota_\ast\ }K_0(\mathcal{Z}_m)\xrightarrow{\ \pi_\ast\ }K_0(\mathcal{Z}_{m-1})\longrightarrow 0,
\]
其中 \(\pi_\ast\) 为“丢弃末位为 \(1\) 的坐标”之投影同态。
\end{corollary}
\begin{proof}
由禁词 \(11\) 与末位分拆：
若 \(w\in X_m\) 满足 \(w_m=0\)，则 \(w=v0\) 且 \(v\in X_{m-1}\) 唯一；
若 \(w_m=1\)，则必有 \(w_{m-1}=0\)，从而 \(w=u01\) 且 \(u\in X_{m-2}\) 唯一，并且该右逆由命题 \ref{prop:fold-bin-minsector-lastbit1-section-unique} 强制为 \(u\mapsto u01\)。
因此
\[
X_m=(X_{m-1}\cdot 0)\ \sqcup\ (X_{m-2}\cdot 01)
\]
为不交并，遂 \(\CC^{X_m}\cong \CC^{X_{m-1}}\oplus \CC^{X_{m-2}}\)，并在基元 \(\Pi_w\) 下给出所述 \(\ast\)-同构。
由于 \(\mathcal{Z}_m\) 为有限维交换 \(C^\ast\)-代数，\(K_0(\mathcal{Z}_m)\cong \ZZ^{X_m}\)，坐标分裂诱导所示正合列，其中 \(\pi_\ast\) 为到末位 \(0\) 坐标的投影，而 \(\iota_\ast\) 识别末位 \(1\) 坐标为子群。证毕。
\end{proof}
