\begin{theorem}[分层素数片上的有限深度投影可逆化外置]\label{thm:killo-layered-prime-slices-finite-depth}
设 \((X_j)_{j=0}^{k}\) 为可数集合列，且对每个 \(j=0,\dots,k-1\) 给定有限纤维满射
\[
\pi_j:X_j\twoheadrightarrow X_{j+1},
\qquad
\forall y\in X_{j+1},\ \#\pi_j^{-1}(y)<\infty.
\]
固定素数枚举 \((p_n)_{n\ge 1}\)。定义分层素数片
\[
\mathcal P_j:=\{p_{2^j(2n-1)}:n\ge 1\},\qquad j\ge 0.
\]
则 \(\mathcal P=\bigsqcup_{j\ge 0}\mathcal P_j\) 为素数集合的不交分解。并且存在映射
\[
\Phi:X_0\longrightarrow X_k\times \ZZ_{>0},\qquad
\Phi(x_0)=(x_k,G(x_0)),
\]
满足：
\begin{enumerate}
\item \(\Phi\) 为单射；
\item \(G(x_0)\) 在唯一分解下具有严格分层分离
\[
G(x_0)=\prod_{j=0}^{k-1}q_j(x_0),\qquad q_j(x_0)\in \mathcal P_j,
\]
且由 \((x_k,G(x_0))\) 可逐层确定性重构唯一 \(x_0\)。
\end{enumerate}
\leanverified{paper\_killo\_layered\_prime\_slices\_finite\_depth}
\end{theorem}
\begin{proof}
任意整数 \(m\ge 1\) 唯一表示为 \(m=2^j(2n-1)\)，故指标层由 \(v_2(m)=j\) 给出不交覆盖；经枚举 \(m\mapsto p_m\) 即得 \(\mathcal P=\bigsqcup_{j\ge 0}\mathcal P_j\)。

固定 \(j\in\{0,\dots,k-1\}\)。因 \(X_{j+1}\times\NN\) 与 \(\mathcal P_j\) 均可数，取单射
\[
\alpha_j:X_{j+1}\times\NN\hookrightarrow \mathcal P_j.
\]
对每个 \(y\in X_{j+1}\)，取纤维枚举单射
\[
\iota_{j,y}:\pi_j^{-1}(y)\hookrightarrow\NN.
\]
对 \(x_j\in X_j\) 记 \(y_{j+1}:=\pi_j(x_j)\)，并定义
\[
q_j(x_j):=\alpha_j\!\bigl(y_{j+1},\iota_{j,y_{j+1}}(x_j)\bigr)\in\mathcal P_j.
\]
对 \(x_0\in X_0\) 递推 \(x_{j+1}:=\pi_j(x_j)\)，设
\[
G(x_0):=\prod_{j=0}^{k-1}q_j(x_j),\qquad
\Phi(x_0):=(x_k,G(x_0)).
\]

给定 \((x_k,N)\) 属于 \(\Phi\) 的像。由唯一分解，按分层片提取
\[
N=\prod_{j\ge 0}N_j,\qquad
N_j:=\prod_{p\in\mathcal P_j}p^{v_p(N)}.
\]
在像内有 \(N_j=1\)（\(j\ge k\)），且 \(N_j=q_j\in\mathcal P_j\)（\(0\le j\le k-1\)）。于是可逆向递推：已知 \(x_{j+1}\) 与 \(q_j\)，由 \(\alpha_j\) 的像上反演得到唯一索引，再由 \(\iota_{j,x_{j+1}}\) 的像上反演得到唯一 \(x_j\in\pi_j^{-1}(x_{j+1})\)。逐层下推得到唯一 \(x_0\)。故 \(\Phi\) 为单射，且像上逆映射显式可构造。
\end{proof}

\begin{corollary}[分层外置的最坏情形必要性与最小实现]\label{cor:killo-layered-necessity-minimality}
沿用定理 \ref{thm:killo-layered-prime-slices-finite-depth}，并假设对每层 \(j=0,\dots,k-1\)，存在 \(y_j\in X_{j+1}\) 使 \(\#\pi_j^{-1}(y_j)\ge 2\)。设
\[
\Psi(x_0)=(x_k,H(x_0)),\qquad
H(x_0)=\prod_{j=0}^{k-1}h_j(x_0),
\]
其中每个 \(h_j(x_0)\) 的全部素因子均落在预先固定片 \(\mathcal P_j\)。若 \(\Psi\) 为单射，则对每一层 \(j\) 至少存在一个微态 \(x_0^{(j)}\)（可取在同一宏态纤维内）使 \(h_j(x_0^{(j)})\neq 1\)。因此在最坏情形分析中，系统必须为全部 \(k\) 个分层素数片预留有效通道。

定理 \ref{thm:killo-layered-prime-slices-finite-depth} 的构造在最坏情形下每层恰使用一个素数因子，达到该必要下界。
\end{corollary}
\begin{proof}
固定一层 \(j\)。由 \(\#\pi_j^{-1}(y_j)\ge 2\)，取不同点 \(u_j\neq v_j\in\pi_j^{-1}(y_j)\)。固定其余层的选择并沿后续映射延拓到同一宏态，可构造两个微态 \(x_0',x_0''\) 使它们仅在第 \(j\) 层分支选择不同而宏态相同。若对所有此类微态有 \(h_j\equiv 1\)，则分层分离导致 \(\Psi(x_0')=\Psi(x_0'')\)，与单射性矛盾。故存在 \(x_0^{(j)}\) 使 \(h_j(x_0^{(j)})\neq 1\)。

对每个 \(j\) 重复即得结论。定理 \ref{thm:killo-layered-prime-slices-finite-depth} 中每层直接写入一枚 \(q_j\in\mathcal P_j\)，故在最坏情形下达到每层一枚的最小实现。
\end{proof}

\begin{theorem}[有限纤维自映射的自然扩张与分支指标寄存器实现]\label{thm:killo-natural-extension-branch-register}
设 \(X\) 为可数集合，\(T:X\twoheadrightarrow X\) 为有限纤维满射。对每个 \(y\in X\) 固定单射
\[
\iota_y:T^{-1}(y)\hookrightarrow\NN,\qquad
\beta(x):=\iota_{T(x)}(x).
\]
定义自然扩张
\[
X^{\leftarrow}:=\{(x_0,x_{-1},x_{-2},\dots):T(x_{-n})=x_{-n+1}\},
\]
及其左移同构 \(\sigma\)。令编码映射
\[
\mathcal C:X^{\leftarrow}\to X\times\NN^\NN,\qquad
\mathcal C((x_0,x_{-1},x_{-2},\dots))=(x_0,(\beta(x_{-1}),\beta(x_{-2}),\dots)).
\]
记 \(\widehat X:=\mathcal C(X^{\leftarrow})\subseteq X\times\NN^\NN\)，并定义
\[
\widehat T:=\mathcal C\circ \sigma\circ \mathcal C^{-1}:\widehat X\to\widehat X.
\]
则 \(\widehat T\) 为双射；在 \(\widehat X\) 上其作用可写成“右移插入”
\[
\widehat T(x,(a_1,a_2,\dots))=(T(x),(\beta(x),a_1,a_2,\dots)).
\]
故 \((\widehat X,\widehat T)\) 与 \((X^{\leftarrow},\sigma)\) 共轭。
\end{theorem}
\begin{proof}
\(\mathcal C\) 的单射性来自递推反演：给定 \((x_0,(a_1,a_2,\dots))\in\widehat X\)，由其属于像空间可知存在且唯一前史满足
\[
x_{-1}=\iota_{x_0}^{-1}(a_1),\quad
x_{-2}=\iota_{x_{-1}}^{-1}(a_2),\ \dots
\]
从而可重构唯一元素于 \(X^{\leftarrow}\)。满射性对 \(\widehat X\) 为定义所致。

对任意 \(\mathbf x=(x_0,x_{-1},\dots)\in X^{\leftarrow}\)，
\[
(\mathcal C\circ \sigma)(\mathbf x)
=(T(x_0),(\beta(x_0),\beta(x_{-1}),\beta(x_{-2}),\dots)),
\]
而这正对应右移插入形式。故 \(\widehat T=\mathcal C\circ\sigma\circ\mathcal C^{-1}\) 为双射并与自然扩张共轭。
\end{proof}

\begin{theorem}[整数线性投影的信息亏损谱与 Smith 不变量可逆读出]\label{thm:killo-smith-loss-spectrum}
\leanverified{paper\_killo\_smith\_loss\_spectrum}
设 \(A\in M_{m\times n}(\ZZ)\) 且 \(\operatorname{rank}_{\QQ}(A)=m\)。其 Smith 不变量记为
\[
d_1\mid d_2\mid\cdots\mid d_m.
\]
对素数 \(p\) 与 \(k\ge 1\)，定义
\[
K_p(k):=\#\ker(A\bmod p^k),\qquad
f_p(k):=\log_p K_p(k)-k(n-m).
\]
若记 \(e_i(p):=v_p(d_i)\)，则：
\begin{enumerate}
\item
\[
f_p(k)=\sum_{i=1}^m\min(k,e_i(p));
\]
\item 由 \(f_p\) 可逆恢复多重集 \(\{e_i(p)\}_{i=1}^m\)，且
\[
\Delta_p(k):=f_p(k)-f_p(k-1)=\#\{i:e_i(p)\ge k\},\qquad k\ge 1.
\]
\end{enumerate}
因此 \(\{K_p(k)\}_{p,k}\)（等价地 \(\{f_p(k)\}_{p,k}\)）可逆确定全部 Smith 不变量。
\end{theorem}
\begin{proof}
由 Smith 形式，存在 \(U\in GL_m(\ZZ),V\in GL_n(\ZZ)\) 使
\[
UAV=\operatorname{diag}(d_1,\dots,d_m,0,\dots,0).
\]
模 \(p^k\) 后 \(U,V\) 仍可逆，故核大小不变。对 \(i\le m\)，同余
\[
d_i y_i\equiv 0\pmod{p^k}
\]
的解数为 \(p^{\min(k,e_i(p))}\)；其余 \(n-m\) 个变量自由，各有 \(p^k\) 种。故
\[
K_p(k)=p^{k(n-m)+\sum_{i=1}^m\min(k,e_i(p))},
\]
即得第一式。

再用恒等式
\[
\min(k,e)-\min(k-1,e)=\mathbf 1_{\{e\ge k\}},
\]
得到
\[
\Delta_p(k)=\sum_{i=1}^m\mathbf 1_{\{e_i(p)\ge k\}}=\#\{i:e_i(p)\ge k\}.
\]
这正是分拆与共轭分拆的标准反演，故 \(\{e_i(p)\}\) 可逆恢复。
\end{proof}

\begin{corollary}[任意模数下的信息亏损闭式与素数可加分解]\label{cor:killo-kernel-size-general-modulus}
\leanverified{paper\_killo\_kernel\_size\_general\_modulus}
在定理 \ref{thm:killo-smith-loss-spectrum} 的设定下，对任意 \(b\ge 1\)，诱导映射
\[
A_b:(\ZZ/b\ZZ)^n\to(\ZZ/b\ZZ)^m
\]
满足
\[
\#\ker(A_b)=b^{\,n-m}\prod_{i=1}^m\gcd(d_i,b).
\]
若 \(b=\prod_p p^{k_p}\)，则
\[
\log \#\ker(A_b)
=\sum_{p}\Bigl(k_p(n-m)+\sum_{i=1}^m\min(k_p,e_i(p))\Bigr)\log p.
\]
因此信息亏损的对数量在素数方向严格可加分解。
\end{corollary}
\begin{proof}
Smith 形下核方程分解为 \(d_i y_i\equiv 0\pmod b\) 与自由变量部分。映射 \(y\mapsto d_i y\) 在 \(\ZZ/b\ZZ\) 上的核大小为 \(\gcd(d_i,b)\)，自由变量贡献 \(b^{n-m}\)，得第一式。

再由
\[
\gcd(d_i,b)=\prod_p p^{\min(v_p(d_i),k_p)}
=\prod_p p^{\min(e_i(p),k_p)},
\]
取对数并交换求和即得第二式。
\end{proof}

\begin{theorem}[整数轴对齐椭圆与对角算法的素数寄存器等价]\label{thm:killo-ellipse-diagonal-prime-register-equivalence}
\leanverified{paper\_killo\_ellipse\_diagonal\_prime\_register\_equivalence}
令
\[
\mathsf E_{\NN}:=\left\{
E_{a,b}: \frac{x^2}{a^2}+\frac{y^2}{b^2}=1,\ a,b\in\NN_{>0}
\right\},
\]
并定义运算
\[
E_{a,b}\odot E_{c,d}:=E_{ac,bd}.
\]
则 \((\mathsf E_{\NN},\odot)\) 为交换幺半群，且存在幺半群同构
\[
(\NN^{(\mathcal P)}\times \NN^{(\mathcal P)},+)
\cong
(\mathsf E_{\NN},\odot),\qquad
(r_1,r_2)\mapsto E_{N(r_1),N(r_2)},
\]
其中
\[
N(r):=\prod_{p\in\mathcal P}p^{r(p)},\qquad
\NN^{(\mathcal P)}:=\{r:\mathcal P\to\NN\text{ 有限支撑}\}.
\]
因此在该可计算子类中，椭圆参数与对角算法 \(\operatorname{diag}(a,b)\) 的素数指数寄存器完全等价，且算法复合对应 \(\odot\)。
\end{theorem}
\begin{proof}
由 \(\operatorname{diag}(a,b)(S^1)=E_{a,b}\) 与
\[
\operatorname{diag}(a,b)\circ\operatorname{diag}(c,d)=\operatorname{diag}(ac,bd),
\]
可得
\[
(\mathsf E_{\NN},\odot)\cong(\NN_{>0}\times\NN_{>0},\times).
\]
唯一分解给出
\[
(\NN_{>0},\times)\cong(\NN^{(\mathcal P)},+),\qquad r\mapsto N(r).
\]
取直积即得所述同构。
\end{proof}

\paragraph{Lee--Yang 扭点地址化与 Gödel--Zeckendorf 外置链条}
在上述分层寄存器与椭圆化框架基础上，可进一步得到 \(2\)-进扭点地址化、半直积仿射表示与 Zeckendorf 像集解析之间的一致闭环。

\begin{theorem}[Lee--Yang 椭圆曲线的 \(2^n\)-扭点超立方体 Cayley 实现]\label{thm:killo-leyang-2power-torsion-cayley-hypercube}
设 Lee--Yang 频谱曲线的极小 Weierstrass 模型为
\[
E_{\min}:\quad u^2+u=x^3-x,
\]
其同构类对应 Cremona 记号 \(37a1\)。对任意 \(n\ge 1\)，记 \(E_{\min}[2^n]\) 为 \(2^n\)-扭点群，\(\Omega_{2n}:=\{0,1\}^{2n}\)，并记 \(Q_{2n}\) 为 \(\Omega_{2n}\) 上的 Hamming 立方图（相邻当且仅当 Hamming 距离为 \(1\)）。任选同构
\[
E_{\min}[2^n]\cong(\ZZ/2^n\ZZ)^2
\]
的一组 \(\ZZ/2^n\ZZ\)-基 \((P_n,Q_n)\)，定义
\[
\iota_n(\mathbf b,\mathbf c)
:=\Bigl(\sum_{j=1}^{n} b_j2^{j-1}\Bigr)P_n
 +\Bigl(\sum_{j=1}^{n} c_j2^{j-1}\Bigr)Q_n,
\]
其中 \(\mathbf b=(b_1,\dots,b_n),\ \mathbf c=(c_1,\dots,c_n)\in\{0,1\}^n\)。则 \(\iota_n:\Omega_{2n}\to E_{\min}[2^n]\) 为双射。并且若定义对称生成集
\[
S_n:=\{\pm 2^{j-1}P_n,\ \pm 2^{j-1}Q_n:\ 1\le j\le n\}\subset E_{\min}[2^n],
\]
则有图同构
\[
Q_{2n}\cong \mathrm{Cay}(E_{\min}[2^n],S_n).
\]
\end{theorem}
\begin{proof}
由 \(E_{\min}[2^n]\cong(\ZZ/2^n\ZZ)^2\)，在固定基 \((P_n,Q_n)\) 后，每个扭点唯一表示为 \(aP_n+bQ_n\)（\(a,b\in\{0,\dots,2^n-1\}\)）。映射 \(\iota_n\) 正是将 \(\Omega_{2n}\) 的两段长度 \(n\) 的二进制向量分别解释为 \(a,b\) 的二进展开，故为双射。

若 \(\omega,\omega'\in\Omega_{2n}\) 仅在一位翻转，则对应的 \((a,b)\) 仅在某一坐标改变 \(\pm 2^{j-1}\)，故
\[
\iota_n(\omega')-\iota_n(\omega)\in S_n.
\]
反之，若两点差在 \(S_n\) 内，则仅有一位二进制位发生翻转，对应 \(Q_{2n}\) 的一条边。故邻接关系一一对应，图同构成立。
\end{proof}

\begin{corollary}[分辨率塔与 \(2\)-进 Tate 模块的超立方体逆极限同构]\label{cor:killo-2adic-tate-hypercube-limit}
在定理 \ref{thm:killo-leyang-2power-torsion-cayley-hypercube} 的设定下，取相容基 \((P_n,Q_n)_{n\ge 1}\) 使得
\[
P_n=2P_{n+1},\qquad Q_n=2Q_{n+1}.
\]
则乘二映射
\[
[2]:E_{\min}[2^{n+1}]\longrightarrow E_{\min}[2^n]
\]
在坐标表示下等同于
\[
(a\bmod 2^{n+1},\,b\bmod 2^{n+1})\longmapsto (a\bmod 2^n,\,b\bmod 2^n),
\]
而在 \(\Omega_{2(n+1)}\to\Omega_{2n}\) 上对应于丢弃两段二进制展开最高位的截断算子。因而
\[
\varprojlim_n E_{\min}[2^n]\cong T_2(E_{\min})\cong\ZZ_2^2,
\]
且在上述截断兼容下，图塔 \((Q_{2n})_{n\ge 1}\) 的逆极限同构于 \(T_2(E_{\min})\) 的 \(2\)-进地址空间。
\end{corollary}
\begin{proof}
相容基给出
\[
E_{\min}[2^{n+1}]\cong(\ZZ/2^{n+1}\ZZ)^2,\qquad
E_{\min}[2^n]\cong(\ZZ/2^n\ZZ)^2
\]
的兼容识别，并使 \([2]\) 将 \(aP_{n+1}+bQ_{n+1}\) 送至 \(aP_n+bQ_n\)。由于 \(P_n,Q_n\) 的阶为 \(2^n\)，系数自然按 \(\bmod 2^n\) 约化。二进截断与该约化一致，故逆极限即 \(\ZZ_2^2\)，并与 Tate 模块定义一致。
\end{proof}

\begin{theorem}[Gödel 半直积在 \(T_2(E_{\min})\) 上的 \(2^L\)-进仿射表示与规范忠实性的最小阈值]\label{thm:killo-godel-semidir-affine-2adic}
设 \(\mathcal E\) 为有限原子事件集，并固定注入编码
\[
\mathrm{code}:\mathcal E\hookrightarrow\NN,\qquad
M:=\max_{e\in\mathcal E}\mathrm{code}(e),\qquad B:=2^L\ge 2.
\]
记 \((p_t)_{t\ge 1}\) 为素数序列，\(\mathcal R:=\NN^{(\mathbb P)}\) 为素数指标上的有限支撑向量空间，\(\Sigma\) 为素数坐标移位算子，\(H:=\mathcal R\rtimes_{\Sigma}\NN\) 为对应半直积。令
\[
T:=T_2(E_{\min})\cong\ZZ_2^2,
\]
并任选非零向量 \(v\in T\)。定义
\[
\mathrm{ev}_B:\mathcal R\to T,\qquad
\mathrm{ev}_B(r):=\sum_{t\ge 1} r(p_t)\,B^{t-1}v,
\]
（因 \(r\) 有限支撑，该和有限），以及
\[
\Psi:H\to\mathrm{Aff}(T),\qquad
\Psi(r,k)(x):=B^k x+\mathrm{ev}_B(r).
\]
则 \(\Psi\) 为单子同态。进一步，若取规范编码子集
\[
H_{\mathrm{can}}
:=\{(r,k)\in H:\ \mathrm{supp}(r)\subseteq\{p_1,\dots,p_k\},\ 0\le r(p_t)\le M\},
\]
则
\[
\Psi|_{H_{\mathrm{can}}}\ \text{为忠实表示}
\quad\Longleftrightarrow\quad
B>M.
\]
\end{theorem}
\begin{proof}
由移位定义可得
\[
\mathrm{ev}_B(\Sigma^k s)
=\sum_{t\ge 1}(\Sigma^k s)(p_t)\,B^{t-1}v
=\sum_{u\ge 1}s(p_u)\,B^{u+k-1}v
=B^k\mathrm{ev}_B(s).
\]
故对任意 \((r,k),(s,\ell)\in H\),
\[
\Psi(r,k)\!\circ\!\Psi(s,\ell)(x)
=B^{k+\ell}x+B^k\mathrm{ev}_B(s)+\mathrm{ev}_B(r)
=B^{k+\ell}x+\mathrm{ev}_B(r+\Sigma^k s),
\]
与半直积乘法
\[
(r,k)\cdot(s,\ell)=(r+\Sigma^k s,k+\ell)
\]
一致，故 \(\Psi\) 为同态。

先证充分性。设 \(B>M\)，并取 \((r,k),(s,\ell)\in H_{\mathrm{can}}\) 使
\(
\Psi(r,k)=\Psi(s,\ell)
\)。
比较线性部分得 \(B^k=B^\ell\)，故 \(k=\ell\)。再比较平移部分得 \(\mathrm{ev}_B(r)=\mathrm{ev}_B(s)\)。由 \(H_{\mathrm{can}}\) 的支撑约束，二者仅涉及 \(p_1,\dots,p_k\) 的系数；由 \(0\le r(p_t),s(p_t)\le M<B\) 以及 \(B\)-进展开唯一性，得到 \(r(p_t)=s(p_t)\)（\(1\le t\le k\)），故 \(r=s\)。因此 \(\Psi|_{H_{\mathrm{can}}}\) 忠实。

再证必要性。若 \(B\le M\)，取 \(k=2\)，并定义 \(r,s\in\mathcal R\) 为
\[
r(p_1)=B,\qquad r(p_2)=0,
\]
\[
s(p_1)=0,\qquad s(p_2)=1,
\]
其余素数坐标全为 \(0\)。则 \((r,2),(s,2)\in H_{\mathrm{can}}\)，因为二者支撑都包含于 \(\{p_1,p_2\}\)，且各坐标都落在区间 \([0,M]\) 中。另一方面，
\[
\mathrm{ev}_B(r)=Bv=\mathrm{ev}_B(s),
\]
故
\[
\Psi(r,2)(x)=B^2x+Bv=\Psi(s,2)(x)
\qquad (x\in T).
\]
但 \((r,2)\neq (s,2)\)，从而 \(\Psi|_{H_{\mathrm{can}}}\) 不忠实。
综上，规范忠实性当且仅当 \(B>M\)。
\end{proof}

\begin{corollary}[无限事件流的 \(2^L\)-进全息点与地址读取解释]\label{cor:killo-infinite-stream-2adic-holographic-point}
沿用定理 \ref{thm:killo-godel-semidir-affine-2adic} 的设定。对任意无限事件流 \(\mathbf e=(e_1,e_2,\dots)\in\mathcal E^{\NN}\)，定义
\[
X(\mathbf e):=\sum_{t\ge 1}\mathrm{code}(e_t)\,B^{t-1}v\in T.
\]
则该级数在 \(2\)-进拓扑下收敛；其前缀和
\[
X_n(\mathbf e):=\sum_{t=1}^{n}\mathrm{code}(e_t)\,B^{t-1}v
\]
满足
\[
X_n(\mathbf e)\equiv X(\mathbf e)\pmod{B^nT}.
\]
此外，当 \(B>M\) 时，映射 \(\mathbf e\mapsto X(\mathbf e)\) 为单射。因而在该模型中，时间推进可解释为同一 \(2^L\)-进地址点的分辨率提升过程。
\end{corollary}
\begin{proof}
因 \(B=2^L\) 满足 \(|B|_2<1\)，且 \(\mathrm{code}(e_t)\) 有界，故 \(B^{t-1}v\to 0\) 且级数按 \(2\)-进范数绝对收敛。尾和
\[
\sum_{t>n}\mathrm{code}(e_t)\,B^{t-1}v\in B^nT
\]
立即给出同余关系。若 \(X(\mathbf e)=X(\mathbf e')\)，则
\[
\sum_{t\ge 1}\bigl(\mathrm{code}(e_t)-\mathrm{code}(e_t')\bigr)B^{t-1}v=0.
\]
由各位系数落在 \([-(B-1),B-1]\) 且 \(B\)-进展开唯一，得
\(\mathrm{code}(e_t)=\mathrm{code}(e_t')\) 对一切 \(t\) 成立。编码注入性推出 \(\mathbf e=\mathbf e'\)。
\end{proof}

\begin{theorem}[Zeckendorf--Gödel 像集 \(ZG\) 的自然密度与素数转移矩阵闭式]\label{thm:killo-zg-natural-density-transfer-matrix}
定义
\[
ZG:=\left\{
\prod_{k\ge 1}p_k^{z_k}\;:\;
z_k\in\{0,1\},\ z_kz_{k+1}=0,\ z_k=0\text{ 对充分大 }k
\right\},
\]
即平方自由且不含相邻素因子的整数集。则其自然密度
\[
\delta(ZG):=\lim_{N\to\infty}\frac{1}{N}\#\{n\le N:\ n\in ZG\}
\]
存在，并满足
\[
\delta(ZG)
=\frac{1}{\zeta(2)}
\cdot
\lim_{n\to\infty}
\mathbf 1^\top
\Bigl(\prod_{i=1}^{n}
\begin{pmatrix}
\dfrac{p_i}{p_i+1} & \dfrac{p_i}{p_i+1}\\[4pt]
\dfrac{1}{p_i+1} & 0
\end{pmatrix}
\Bigr)
\begin{pmatrix}1\\0\end{pmatrix},
\qquad
\mathbf 1=\begin{pmatrix}1\\1\end{pmatrix}.
\]
且该极限严格为正，因此 \(\delta(ZG)\in(0,1)\)。
\end{theorem}
\begin{proof}
对每个素数 \(p\)，事件 \(p^2\nmid n\) 的密度为 \(1-1/p^2\)，并在不同素数间可乘。条件化于平方自由后，
\[
\PP(p\mid n\mid p^2\nmid n)
=\frac{(1/p-1/p^2)}{1-1/p^2}
=\frac{1}{p+1}.
\]
故在平方自由模型下，指标变量 \(\mathbf 1_{\{p_i\mid n\}}\) 独立，参数为 \(1/(p_i+1)\)。而 \(ZG\) 的约束即链上禁相邻 \(1\)。

对前缀 \(\{p_1,\dots,p_n\}\)，令 \(a_i\) 为第 \(i\) 位取 \(0\) 的合法概率，\(b_i\) 为第 \(i\) 位取 \(1\) 的合法概率，则
\[
a_i=(a_{i-1}+b_{i-1})\frac{p_i}{p_i+1},\qquad
b_i=a_{i-1}\frac{1}{p_i+1},\qquad
(a_0,b_0)=(1,0),
\]
即所示矩阵递推。再乘回平方自由因子 \(\prod_i(1-1/p_i^2)=1/\zeta(2)\) 即得截断密度表达。由 \(\sum_i p_i^{-2}<\infty\) 与上述递推的 Cauchy 收敛性，极限存在。每一阶段概率均正且尾部校正因子正，故极限严格正。
\end{proof}

\begin{theorem}[\(ZG\) 的 Dirichlet 生成函数矩阵 Euler 积表达]\label{thm:killo-zg-dirichlet-matrix-euler}
\leanverified{paper\_killo\_zg\_dirichlet\_matrix\_euler}
对 \(\Re(s)>1\)，定义
\[
D_{ZG}(s):=\sum_{n\in ZG}\frac{1}{n^s}.
\]
则 \(D_{ZG}(s)\) 绝对收敛，并可表示为
\[
D_{ZG}(s)
=\lim_{n\to\infty}
\mathbf 1^\top
\Bigl(\prod_{i=1}^{n}
\begin{pmatrix}
1 & 1\\[2pt]
p_i^{-s} & 0
\end{pmatrix}
\Bigr)
\begin{pmatrix}1\\0\end{pmatrix}.
\]
上式即 \(ZG\) 的矩阵 Euler 积表示。
\end{theorem}
\begin{proof}
\(ZG\) 的元素与素数指标链上的独立集 \(S\subset\NN\)（禁止相邻）对应，权重为 \(\prod_{i\in S}p_i^{-s}\)，故
\[
D_{ZG}(s)
=\sum_{\substack{S\subset\NN\\S\ \text{无相邻}}}\ \prod_{i\in S}p_i^{-s}.
\]
对前 \(n\) 个素数，设 \(A_n\) 为合法加权和，\(B_n\) 为“且选取第 \(n\) 位”的合法加权和，则
\[
A_n=A_{n-1}+B_{n-1},\qquad
B_n=p_n^{-s}A_{n-1},\qquad
(A_0,B_0)=(1,0).
\]
故 \((A_n,B_n)^\top\) 由所示矩阵迭代给出，且
\[
A_n+B_n
=\mathbf 1^\top
\Bigl(\prod_{i=1}^{n}
\begin{pmatrix}
1 & 1\\
p_i^{-s} & 0
\end{pmatrix}
\Bigr)
\begin{pmatrix}1\\0\end{pmatrix}.
\]
当 \(\sigma:=\Re(s)>1\) 时，\(\sum_i p_i^{-\sigma}<\infty\)，故独立集权重和绝对收敛，\(\lim_{n\to\infty}(A_n+B_n)\) 存在并等于 \(D_{ZG}(s)\)。
\end{proof}

\begin{proposition}[Lee--Yang 分支集的超立方体平行族分解与二进地址几何]\label{prop:killo-leyang-branchset-hypercube-families}
设 \(y_n\) 为 Lee--Yang 塔的第 \(n\) 层对象。其分支集满足
\[
R(y_n)=\bigcup_{i=1}^{5}\Pi_{i,n},
\qquad
\Pi_{i,n}:=[2^n]^{-1}(P_i),
\]
并且
\[
\deg R(y_n)=5\cdot 4^n.
\]
结合定理 \ref{thm:killo-leyang-2power-torsion-cayley-hypercube}，每个 \(\Pi_{i,n}\) 都是 \(E_{\min}[2^n]\) 的主齐次空间，因而（在选取基点后）可识别为一个 \(2n\)-超立方体顶点集。于是 \(R(y_n)\) 在 \(E_{\min}\) 内分解为五个互不相交的超立方体平行族。再由推论 \ref{cor:killo-2adic-tate-hypercube-limit} 与倍点塔的相容性，乘二映射 \([2]\) 将第 \(n+1\) 层平行族按去最高位截断投影到第 \(n\) 层。
\end{proposition}
\begin{proof}
分支集分解与次数公式由 Lee--Yang 塔的既有分歧计算给出；并且 \(\Pi_{i,n}=[2^n]^{-1}(P_i)\) 对每个 \(i\) 都是 \(E_{\min}[2^n]\) 的 torsor。
任取 \(P_{i,n}\in\Pi_{i,n}\)，则 \(\Pi_{i,n}=P_{i,n}+E_{\min}[2^n]\)，且平移
\[
E_{\min}[2^n]\to \Pi_{i,n},\qquad Q\mapsto P_{i,n}+Q
\]
为双射并将 Cayley 图结构从 \(E_{\min}[2^n]\) 传递到 \(\Pi_{i,n}\)。
由定理 \ref{thm:killo-leyang-2power-torsion-cayley-hypercube}，\(E_{\min}[2^n]\) 具有 \(Q_{2n}\) 结构，故每个 \(\Pi_{i,n}\)（在选定基点 \(P_{i,n}\) 后）可识别为 \(2n\)-超立方体顶点集。
推论 \ref{cor:killo-2adic-tate-hypercube-limit} 已说明 \([2]\) 在扭点坐标上等同模约化，即二进最高位截断；在选取相容基点族 \(([2]P_{i,n+1}=P_{i,n})\) 后，该结论逐层传递到 \(\Pi_{i,n}\) 上，从而得到所述投影关系。
\end{proof}
