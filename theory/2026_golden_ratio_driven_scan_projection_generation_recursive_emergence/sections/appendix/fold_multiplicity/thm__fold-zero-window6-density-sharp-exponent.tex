\begin{theorem}[六倍窗子序列上零点集合密度的精确指数级]\label{thm:fold-zero-window6-density-sharp-exponent}
沿用推论 \ref{cor:fold-zero-half-index-multiple6} 与定理 \ref{thm:fold-zero-density-sparse} 的记号。若写
\[
m=6n-2,
\qquad n\to\infty,
\]
则有
\[
\boxed{\;
\liminf_{n\to\infty}
\varphi^{m/2}\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\ge
\varphi^{-1}.\;}
\]
并且
\[
\boxed{\;
\lim_{n\to\infty}
\frac{1}{m}\log\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
=
-\frac{1}{2}\log\varphi.\;}
\]
换言之，在六倍窗子子序列上，谱零点的相对密度恰以 \(\varphi^{-m/2}\) 的指数尺度衰减。
\leanverified{paper\_fold\_zero\_window6\_density\_sharp\_exponent}
\end{theorem}
\begin{proof}
当 \(m=6n-2\) 时，\(m+2=6n\) 被 \(6\) 整除。由推论 \ref{cor:fold-zero-half-index-multiple6}，
\[
\card{\mathcal{Z}_m}\ge F_{(m+2)/2}=F_{3n}.
\]
因此
\[
\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\ge
\frac{F_{(m+2)/2}}{F_{m+2}}
=
\frac{F_{3n}}{F_{6n}}.
\]
由 Binet 公式 \(F_t=(\varphi^t-\psi^t)/\sqrt5\)（其中 \(\psi=-\varphi^{-1}\)），可得
\[
\frac{F_{3n}}{F_{6n}}
\sim
\varphi^{-3n}.
\]
又因为 \(m/2=3n-1\)，于是
\[
\varphi^{m/2}\frac{F_{3n}}{F_{6n}}
=
\varphi^{3n-1}\frac{F_{3n}}{F_{6n}}
\longrightarrow
\varphi^{-1}.
\]
故
\[
\liminf_{n\to\infty}
\varphi^{m/2}\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\ge
\varphi^{-1}.
\]

另一方面，由定理 \ref{thm:fold-zero-density-sparse}，
\[
\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\le
\tau(m+2)\,\frac{F_{\lfloor (m+2)/2\rfloor}}{F_{m+2}}.
\]
在当前子序列上 \(\lfloor (m+2)/2\rfloor=3n\)，故
\[
\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\le
\tau(6n)\,\frac{F_{3n}}{F_{6n}}.
\]
由于 \(\tau(N)=e^{o(N)}\)，有
\[
\frac{1}{m}\log\tau(6n)\to 0.
\]
再结合
\[
\frac{1}{m}\log\frac{F_{3n}}{F_{6n}}
\longrightarrow
-\frac{1}{2}\log\varphi,
\]
得到
\[
\limsup_{n\to\infty}
\frac{1}{m}\log\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\le
-\frac{1}{2}\log\varphi.
\]
而下界由
\(
\card{\mathcal{Z}_m}\ge F_{3n}
\)
立刻给出：
\[
\liminf_{n\to\infty}
\frac{1}{m}\log\frac{\card{\mathcal{Z}_m}}{F_{m+2}}
\ge
\lim_{n\to\infty}
\frac{1}{m}\log\frac{F_{3n}}{F_{6n}}
=
-\frac{1}{2}\log\varphi.
\]
综上极限存在且等于 \(-\tfrac12\log\varphi\)。证毕。
\end{proof}
