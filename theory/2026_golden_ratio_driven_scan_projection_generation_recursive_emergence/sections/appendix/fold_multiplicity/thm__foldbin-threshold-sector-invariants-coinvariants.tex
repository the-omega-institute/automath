\paragraph{bin-fold 退化谱的阈值几何与规范代数后果}
以下结果把命题 \ref{prop:fold-bin-fiber-tail}、推论 \ref{cor:fold-bin-saturation}、推论 \ref{cor:fold-bin-m6-fiber-hist}、命题 \ref{prop:fold-bin-gauge-decomposition}、命题 \ref{prop:fold-bin-groupoid-algebra-wedderburn} 与推论 \ref{cor:fold-bin-gauge-m6} 的对象层信息推进为若干直接的结构定理。

\begin{theorem}[bin-fold 组合上界饱和扇区的精确阈值刻画]\label{thm:foldbin-saturation-sector-threshold}
设 \(m\ge 2\)，取 \(K=K(m)\) 如定义 \ref{def:K-of-m}，并记尾长 \(L:=K-m\ge 1\)。
再令
\[
S_{\max}(m):=\max\Bigl\{\sum_{k=m+1}^{K} c_kF_{k+1}:\ (c_{m+1},\dots,c_K)\in\{0,1\}^{L},\ c_kc_{k+1}=0\Bigr\},
\]
以及
\[
d^{\mathrm{bin}}_{\max}(m):=\max_{w\in X_m} d_m^{\mathrm{bin}}(w)=d_m^{\mathrm{bin}}(0^m).
\]
则达到组合上界 \(F_{L+2}\) 的扇区恰为
\begin{equation}\label{eq:foldbin-saturation-sector-threshold}
\mathcal S_{m,F_{L+2}}
=
\Bigl\{
w\in X_m:\ w_m=0,\ \ V_m(w)\le 2^m-1-S_{\max}(m)
\Bigr\}.
\end{equation}
特别地，
\begin{equation}\label{eq:foldbin-saturation-sector-nonempty}
d^{\mathrm{bin}}_{\max}(m)=F_{L+2}
\iff
\mathcal S_{m,F_{L+2}}\neq\varnothing
\iff
2^m-1\ge S_{\max}(m).
\end{equation}
在这些等价条件成立时，最大纤维扇区满足
\[
\mathcal S_{m,d^{\mathrm{bin}}_{\max}(m)}=\mathcal S_{m,F_{L+2}}.
\]
\end{theorem}
\leanverified{paper\_foldbin\_saturation\_sector\_threshold}
\begin{proof}
由命题 \ref{prop:fold-bin-fiber-tail}，对任意 \(w\in X_m\)，纤维 \((\Fold_m^{\mathrm{bin}})^{-1}(w)\) 与满足
\[
c_kc_{k+1}=0,\qquad
w_mc_{m+1}=0,\qquad
V_m(w)+\sum_{k=m+1}^{K}c_kF_{k+1}\le 2^m-1
\]
的尾部位串一一对应。

先设 \(w_m=0\) 且 \(V_m(w)\le 2^m-1-S_{\max}(m)\)。
则对任意满足相邻约束的尾部位串，都有
\[
V_m(w)+\sum_{k=m+1}^{K}c_kF_{k+1}
\le
V_m(w)+S_{\max}(m)
\le
2^m-1,
\]
故阈值约束自动冗余，而边界约束亦消失。
于是所有长度 \(L\) 上不含相邻 \(1\) 的二进制串都可实现，从而
\[
d_m^{\mathrm{bin}}(w)=F_{L+2},
\]
其中 \(F_{L+2}\) 是长度 \(L\) 的无相邻 \(1\) 串的标准 Fibonacci 计数。
这证明了 \eqref{eq:foldbin-saturation-sector-threshold} 的包含“\(\supseteq\)”。

反之，若 \(w_m=1\)，则边界约束强迫 \(c_{m+1}=0\)。
由于 \(L\ge 1\)，尾部 \((1,0,\dots,0)\) 满足相邻约束，但被边界约束排除，因此
\[
d_m^{\mathrm{bin}}(w)<F_{L+2}.
\]
若 \(w_m=0\) 但 \(V_m(w)>2^m-1-S_{\max}(m)\)，取实现 \(S_{\max}(m)\) 的某条尾部位串 \(c^{\max}\)。
该位串满足相邻约束，却使
\[
V_m(w)+\sum_{k=m+1}^{K}c_k^{\max}F_{k+1}
>
2^m-1,
\]
故阈值约束至少排除了这一合法尾部，因此同样有
\[
d_m^{\mathrm{bin}}(w)<F_{L+2}.
\]
于是 \eqref{eq:foldbin-saturation-sector-threshold} 得证。

再由推论 \ref{cor:fold-bin-saturation}，\(d^{\mathrm{bin}}_{\max}(m)\) 由 \(0^m\) 取得，且 \(d^{\mathrm{bin}}_{\max}(m)\le F_{L+2}\)。
因此 \(d^{\mathrm{bin}}_{\max}(m)=F_{L+2}\) 当且仅当某个 \(w\in X_m\) 取得该上界，亦即当且仅当 \(\mathcal S_{m,F_{L+2}}\neq\varnothing\)。
由 \eqref{eq:foldbin-saturation-sector-threshold}，这又等价于 \(0^m\in\mathcal S_{m,F_{L+2}}\)，亦即 \(2^m-1\ge S_{\max}(m)\)。
最后，在饱和情形下最大值就是 \(F_{L+2}\)，故 \(\mathcal S_{m,d^{\mathrm{bin}}_{\max}(m)}=\mathcal S_{m,F_{L+2}}\)。证毕。
\end{proof}

\begin{theorem}[bin-fold 规范群不变量环的显式多项式表示]\label{thm:foldbin-gauge-invariant-ring}
记
\[
G_m^{\mathrm{bin}}:=\mathcal G_m^{\mathrm{bin}}
=\mathrm{Aut}\!\bigl(\Fold_m^{\mathrm{bin}}\bigr),
\qquad
F_w:=(\Fold_m^{\mathrm{bin}})^{-1}(w),
\qquad
d_w:=|F_w|.
\]
并令
\[
R_m:=\CC[x_n\mid 0\le n<2^m]
\]
为微态坐标多项式环，其中 \(G_m^{\mathrm{bin}}\) 通过纤维内坐标置换作用于 \(R_m\)。
则存在自然的分次同构
\[
R_m^{G_m^{\mathrm{bin}}}
\cong
\bigotimes_{w\in X_m}\CC[e_{w,1},e_{w,2},\dots,e_{w,d_w}],
\]
其中 \(e_{w,j}\) 是纤维 \(F_w\) 上 \(d_w\) 个变量的第 \(j\) 个初等对称多项式。
因此 \(R_m^{G_m^{\mathrm{bin}}}\) 是多项式环，且 Hilbert 级数为
\begin{equation}\label{eq:foldbin-gauge-invariant-ring-hilbert}
\mathrm{Hilb}\bigl(R_m^{G_m^{\mathrm{bin}}},t\bigr)
=
\prod_{w\in X_m}\ \prod_{j=1}^{d_w}\frac{1}{1-t^j}.
\end{equation}
特别地，在 \(m=6\) 时，由
\[
G_6^{\mathrm{bin}}\cong S_2^{\,8}\times S_3^{\,4}\times S_4^{\,9}
\]
与直方图 \(2:8,\ 3:4,\ 4:9\)，其齐次生成元的次数多重集满足：一次 \(21\) 个、二次 \(21\) 个、三次 \(13\) 个、四次 \(9\) 个；并且
\[
\mathrm{Hilb}\bigl(R_6^{G_6^{\mathrm{bin}}},t\bigr)
=
\frac{1}{(1-t)^{21}(1-t^2)^{21}(1-t^3)^{13}(1-t^4)^9}.
\]
\end{theorem}
\leanverified{paper\_foldbin\_gauge\_invariant\_ring}
\begin{proof}
对每个 \(w\in X_m\) 固定纤维 \(F_w\) 的一个排序 \(F_w=\{n_{w,1},\dots,n_{w,d_w}\}\)，并记 \(x_{w,i}:=x_{n_{w,i}}\)。
由于不同纤维不交，故
\[
R_m\cong \bigotimes_{w\in X_m}\CC[x_{w,1},\dots,x_{w,d_w}].
\]
由命题 \ref{prop:fold-bin-gauge-decomposition}，
\[
G_m^{\mathrm{bin}}\cong \prod_{w\in X_m}S_{d_w},
\]
且第 \(w\) 个对称群只作用于第 \(w\) 个张量因子。
因此
\[
R_m^{G_m^{\mathrm{bin}}}
\cong
\bigotimes_{w\in X_m}\CC[x_{w,1},\dots,x_{w,d_w}]^{S_{d_w}}.
\]
由对称多项式基本定理，
\[
\CC[x_{w,1},\dots,x_{w,d_w}]^{S_{d_w}}
\cong
\CC[e_{w,1},\dots,e_{w,d_w}],
\qquad \deg e_{w,j}=j.
\]
将这些局部结果张量化，便得到所述多项式结构与 Hilbert 级数公式 \eqref{eq:foldbin-gauge-invariant-ring-hilbert}。

当 \(m=6\) 时，推论 \ref{cor:fold-bin-m6-fiber-hist} 与推论 \ref{cor:fold-bin-gauge-m6} 给出 \(2\)-纤维 \(8\) 条、\(3\)-纤维 \(4\) 条、\(4\)-纤维 \(9\) 条。
于是次数 \(j\) 的生成元个数等于满足 \(d_w\ge j\) 的纤维数，分别为
\[
21,\qquad 21,\qquad 13,\qquad 9,
\]
从而得到所示显式分级剖面与 Hilbert 级数。证毕。
\end{proof}

\begin{theorem}[微态置换表示的纤维规范分裂与交换子代数维数]\label{thm:foldbin-permutation-representation-commutant}
沿用定理 \ref{thm:foldbin-gauge-invariant-ring} 的记号，并设
\[
V_m:=\CC^{\{0,1,\dots,2^m-1\}}
\cong
\bigoplus_{w\in X_m}\CC^{F_w}
\]
为 \(G_m^{\mathrm{bin}}\) 在微态集合上的置换表示。
对每个 \(w\in X_m\)，令 \(\mathbf 1\) 表示全群 \(G_m^{\mathrm{bin}}\) 的平凡表示，并令 \(\mathrm{Std}_w\) 表示沿投影 \(G_m^{\mathrm{bin}}\to S_{d_w}\) 拉回的标准表示；若 \(d_w=1\)，约定 \(\mathrm{Std}_w=0\)。
则作为 \(G_m^{\mathrm{bin}}\)-表示，有规范直和分解
\[
V_m\cong \mathbf 1^{\oplus |X_m|}\ \oplus\ \bigoplus_{w\in X_m}\mathrm{Std}_w.
\]
特别地：
\begin{enumerate}
  \item 不变量子空间维数满足
  \[
  \dim V_m^{G_m^{\mathrm{bin}}}=|X_m|;
  \]
  \item 非平凡标准部分维数满足
  \[
  \dim\Bigl(\bigoplus_{w\in X_m}\mathrm{Std}_w\Bigr)=2^m-|X_m|;
  \]
  \item 若记
  \[
  N_m:=\#\{w\in X_m:\ d_w\ge 2\},
  \]
  则交换子代数满足
  \[
  \End_{G_m^{\mathrm{bin}}}(V_m)
  \cong
  M_{|X_m|}(\CC)\ \oplus\ \bigoplus_{\substack{w\in X_m\\ d_w\ge 2}}\CC,
  \]
  从而
  \begin{equation}\label{eq:foldbin-commutant-dimension}
  \dim \End_{G_m^{\mathrm{bin}}}(V_m)=|X_m|^2+N_m.
  \end{equation}
\end{enumerate}
在审计偶数窗口 \(m\in\{6,8,10,12\}\) 上，由定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 的 \(d_{\min}(m)\ge 2\) 可知 \(N_m=|X_m|\)，故
\[
\dim \End_{G_m^{\mathrm{bin}}}(V_m)=|X_m|(|X_m|+1).
\]
特别地，当 \(m=6\) 时 \(|X_6|=21\)，因此
\[
\dim \End_{G_6^{\mathrm{bin}}}(V_6)=21\cdot 22=462.
\]
\end{theorem}
\begin{proof}
对每个 \(w\in X_m\)，纤维置换表示 \(\CC^{F_w}\) 作为第 \(w\) 个因子 \(S_{d_w}\) 的表示，按对称群的基本分解可写为
\[
\CC^{F_w}\cong \mathbf 1\oplus \mathrm{Std}_w.
\]
将这些分解对全部纤维求直和，即得所述整体分裂。

第一条随即成立，因为每条纤维上的常数函数都给出一个全群不变量，且这些不变量彼此线性独立。
第二条由
\[
\sum_{w\in X_m}(d_w-1)=\sum_{w\in X_m}d_w-|X_m|=2^m-|X_m|
\]
立即得到。

对第三条，注意 \(\mathbf 1^{\oplus |X_m|}\) 是平凡表示的 \(|X_m|\)-重同构和，因此其交换子代数为 \(M_{|X_m|}(\CC)\)。
另一方面，各 \(\mathrm{Std}_w\) 两两非同构：因为 \(\mathrm{Std}_w\) 的核包含除第 \(w\) 个因子外的全部对称群因子，而对不同 \(w\) 得到的核不同。
故由 Schur 引理，
\[
\End_{G_m^{\mathrm{bin}}}\!\Bigl(\bigoplus_{w\in X_m}\mathrm{Std}_w\Bigr)
\cong
\bigoplus_{\substack{w\in X_m\\ d_w\ge 2}}\CC,
\]
且平凡部分与标准部分之间不存在非零 \(G_m^{\mathrm{bin}}\)-同态。
于是得到所述交换子代数分解与维数公式 \eqref{eq:foldbin-commutant-dimension}。

在审计偶数窗口上，定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 给出所有纤维均满足 \(d_w\ge 2\)，故 \(N_m=|X_m|\)。
取 \(m=6\) 且 \(|X_6|=F_8=21\)，便得到 \(21^2+21=462\)。证毕。
\end{proof}

\begin{theorem}[bin-fold 规范群的 coinvariant 代数与显式 Hilbert 系列]\label{thm:foldbin-coinvariant-gorenstein}
\leanverified{paper\_foldbin\_coinvariant\_gorenstein}
沿用定理 \ref{thm:foldbin-gauge-invariant-ring} 的记号。
记不变量理想
\[
\bigl(R_m^{G_m^{\mathrm{bin}}}\bigr)_+
:=
\langle \text{所有正次数 }G_m^{\mathrm{bin}}\text{-不变量}\rangle
\subset R_m,
\]
并定义 coinvariant 代数
\[
\mathrm{Coinv}_m
:=
R_m\big/\langle (R_m^{G_m^{\mathrm{bin}}})_+\rangle.
\]
则存在分次代数同构
\[
\mathrm{Coinv}_m
\cong
\bigotimes_{w\in X_m}\mathrm{Coinv}(S_{d_w}),
\]
其中 \(\mathrm{Coinv}(S_d)\) 表示对称群 \(S_d\) 的经典 coinvariant 代数。
因此：
\begin{enumerate}
  \item \(\mathrm{Coinv}_m\) 是 Artinian Gorenstein 代数；
  \item 其 Hilbert--Poincar\'e 系列为
  \begin{equation}\label{eq:foldbin-coinvariant-hilbert}
  \mathrm{Hilb}(\mathrm{Coinv}_m,t)
  =
  \prod_{w\in X_m}\ \prod_{j=1}^{d_w}(1+t+\cdots+t^{j-1});
  \end{equation}
  \item 其总维数为
  \begin{equation}\label{eq:foldbin-coinvariant-dimension}
  \dim_{\CC}\mathrm{Coinv}_m
  =
  \prod_{w\in X_m}d_w!
  =
  |G_m^{\mathrm{bin}}|;
  \end{equation}
  \item 其 socle 次数为
  \begin{equation}\label{eq:foldbin-coinvariant-top-degree}
  \deg_{\mathrm{top}}(\mathrm{Coinv}_m)
  =
  \sum_{w\in X_m}\binom{d_w}{2}.
  \end{equation}
\end{enumerate}
特别地，在 \(m=6\) 时，
\[
\mathrm{Hilb}(\mathrm{Coinv}_6,t)
=(1+t)^{21}(1+t+t^2)^{13}(1+t+t^2+t^3)^9,
\]
\[
\dim_{\CC}\mathrm{Coinv}_6
=(2!)^8(3!)^4(4!)^9
=|G_6^{\mathrm{bin}}|,
\]
并且
\[
\deg_{\mathrm{top}}(\mathrm{Coinv}_6)=8\cdot 1+4\cdot 3+9\cdot 6=74.
\]
\end{theorem}
\begin{proof}
沿定理 \ref{thm:foldbin-gauge-invariant-ring} 的记号写
\[
R_m\cong \bigotimes_{w\in X_m}R_w,
\qquad
R_w:=\CC[x_{w,1},\dots,x_{w,d_w}],
\]
且
\[
R_m^{G_m^{\mathrm{bin}}}
\cong
\bigotimes_{w\in X_m}R_w^{S_{d_w}}.
\]
由作用的纤维独立性，不变量理想正是各局部正次数不变量理想的张量和，于是
\[
\mathrm{Coinv}_m
\cong
\bigotimes_{w\in X_m}
\frac{R_w}{\langle (R_w^{S_{d_w}})_+\rangle}
\cong
\bigotimes_{w\in X_m}\mathrm{Coinv}(S_{d_w}).
\]

对任意 \(d\ge 1\)，经典对称群 coinvariant 代数满足
\[
\mathrm{Coinv}(S_d)
\cong
\CC[x_1,\dots,x_d]/(e_1,\dots,e_d),
\]
因此是由次数 \(1,\dots,d\) 的完全交生成元给出的分次 Artinian Gorenstein 代数。
其 Hilbert 系列、总维数与顶次数分别为
\[
\prod_{j=1}^{d}(1+t+\cdots+t^{j-1}),
\qquad
d!,
\qquad
\sum_{j=1}^{d}(j-1)=\binom{d}{2}.
\]
将这些局部公式在 \(w\in X_m\) 上取张量积，便得到 \eqref{eq:foldbin-coinvariant-hilbert}、\eqref{eq:foldbin-coinvariant-dimension} 与 \eqref{eq:foldbin-coinvariant-top-degree}。
而
\[
|G_m^{\mathrm{bin}}|
=
\prod_{w\in X_m}|S_{d_w}|
=
\prod_{w\in X_m}d_w!
\]
由命题 \ref{prop:fold-bin-gauge-decomposition} 立即成立。

当 \(m=6\) 时，推论 \ref{cor:fold-bin-m6-fiber-hist} 给出直方图 \(2:8,\ 3:4,\ 4:9\)。
于是
\[
\prod_{w\in X_6}\prod_{j=1}^{d_w}(1+t+\cdots+t^{j-1})
=(1+t)^{8+4+9}(1+t+t^2)^{4+9}(1+t+t^2+t^3)^9,
\]
即得到 \((1+t)^{21}(1+t+t^2)^{13}(1+t+t^2+t^3)^9\)。
总维数与顶次数则分别由同一直方图直接汇总为
\[
(2!)^8(3!)^4(4!)^9,
\qquad
8\cdot \binom22+4\cdot \binom32+9\cdot \binom42=74.
\]
证毕。
\end{proof}
