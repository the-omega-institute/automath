\begin{theorem}[结论 9（2026-04-14 12:43:32 +08）：六个 Perron 谱半径的强乘法独立性]\label{thm:killo-fold-resonance-perron-independence-q12-17}
对 \(q\in\{12,13,14,15,16,17\}\) 记 \(r_q\) 为 \(P_q\) 的 Perron 根。
若整数向量 \((n_{12},\dots,n_{17})\in\ZZ^6\) 满足
\[
\prod_{q=12}^{17} r_q^{\,n_q}\in\QQ^\times,
\]
则必有
\[
n_{12}=n_{13}=n_{14}=n_{15}=n_{16}=n_{17}=0.
\]
换言之，\(\{r_{12},r_{13},r_{14},r_{15},r_{16},r_{17}\}\) 在 \(\QQ^\times\) 上强乘法独立。
\leanverified{paper\_killo\_fold\_resonance\_perron\_independence\_q12\_17}
\end{theorem}

\begin{proof}
设存在非零整数向量 \((n_q)\) 使
\[
\prod_{q=12}^{17} r_q^{\,n_q}=c\in\QQ^\times.
\]
取其中某个 \(q_0\) 使 \(n_{q_0}\neq 0\)。

由定理 \ref{thm:killo-fold-resonance-product-galois-q12-17}，
存在自同构
\[
\sigma\in\Gal(L_{12}\cdots L_{17}/\QQ)
\]
在除 \(L_{q_0}\) 之外的五个因子上恒等，而在 \(L_{q_0}\) 上非平凡。
由于 \(r_{q_0}\) 的代数共轭正是 \(P_{q_0}\) 的全部根，\(\sigma(r_{q_0})\) 是 \(P_{q_0}\) 的某个与 \(r_{q_0}\) 不同的根。

再由表 \ref{tab:fold_collision_kernel_rh_scan_q2_17}，对全部
\(
q\in\{12,\dots,17\}
\)
都严格成立
\[
\Lambda_q<r_q,
\]
其中 \(\Lambda_q\) 是非 Perron 最大模根。
故任一非平凡共轭根 \(\alpha\neq r_{q_0}\) 都满足
\(
|\alpha|\le \Lambda_{q_0}<r_{q_0}
\)。
特别地，
\[
|\sigma(r_{q_0})|<r_{q_0}.
\]

对原关系施加 \(\sigma\) 得
\[
\sigma(r_{q_0})^{\,n_{q_0}}\prod_{q\neq q_0}r_q^{\,n_q}=c.
\]
与原式相除，得到
\[
\left(\frac{\sigma(r_{q_0})}{r_{q_0}}\right)^{n_{q_0}}=1.
\]
取绝对值即得
\[
\left|\frac{\sigma(r_{q_0})}{r_{q_0}}\right|^{|n_{q_0}|}=1,
\]
但左端严格小于 \(1\)，矛盾。
故不存在非零指数关系，结论成立。
证毕。
\end{proof}

\begin{corollary}[六个 bulk 指数的 \texorpdfstring{$\QQ$}{Q}-线性独立性]\label{cor:killo-fold-resonance-log-linear-independence-q12-17}
在定理 \ref{thm:killo-fold-resonance-perron-independence-q12-17} 的设定下，
\[
\log r_{12},\ \log r_{13},\ \log r_{14},\ \log r_{15},\ \log r_{16},\ \log r_{17}
\]
在 \(\QQ\) 上线性无关。
\end{corollary}

\begin{proof}
若存在有理系数关系
\(
\sum_{q=12}^{17}a_q\log r_q=0
\)，
清分母后得整数关系
\(
\prod_{q=12}^{17}r_q^{\,n_q}=1\in\QQ^\times
\)，
与定理 \ref{thm:killo-fold-resonance-perron-independence-q12-17} 矛盾。
证毕。
\end{proof}
