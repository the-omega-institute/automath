\begin{corollary}[补位对合强制的 Fourier 相位锁定]\label{cor:fold-fourier-phase-locking-by-complement}
\leanverified{paper\_fold\_fourier\_phase\_locking\_by\_complement}
沿用定理 \ref{thm:fold-spectrum-factorization} 的记号，并令
\[
S_m:=\sum_{j=1}^m F_{j+1}.
\]
则对任意 \(k\in\ZZ/(F_{m+2}\ZZ)\)，都有严格相位关系
\[
\widehat d_m(k)=e^{-2\pi i kS_m/F_{m+2}}\ \overline{\widehat d_m(k)}.
\]
等价地，
\[
e^{\pi i kS_m/F_{m+2}}\widehat d_m(k)\in\RR.
\]
\end{corollary}
\begin{proof}
由 \(r(\omega)\equiv N(\omega)\pmod{F_{m+2}}\) 与 \(\widehat d_m\) 的定义，
\[
\widehat d_m(k)=\sum_{r} d_m(r)e^{-2\pi i kr/F_{m+2}}
=\sum_{\omega\in\Omega_m}e^{-2\pi i kN(\omega)/F_{m+2}}.
\]
再由命题 \ref{prop:fold-complement-duality} 中 \(N(C(\omega))=S_m-N(\omega)\)，
\[
\widehat d_m(k)
=\sum_{\omega}e^{-2\pi i k(S_m-N(\omega))/F_{m+2}}
=e^{-2\pi i kS_m/F_{m+2}}\sum_{\omega}e^{2\pi i kN(\omega)/F_{m+2}}
=e^{-2\pi i kS_m/F_{m+2}}\ \overline{\widehat d_m(k)}.
\]
证毕。
\end{proof}

\begin{proposition}[补对偶的固定点与退化直方图的偶性约束]\label{prop:fold-complement-duality-fixedpoints-parity}
\leanverified{paper\_fold\_complement\_duality\_fixedpoints\_parity}
令 \(\jmath_m:X_m\to X_m\) 为命题 \ref{prop:fold-complement-duality} 给出的稳定类型对合，并记
\(
\mathrm{Fix}(\jmath_m):=\{x\in X_m:\ \jmath_m(x)=x\}
\)。
则：
\begin{enumerate}
\normalfont
  \item 固定点集合满足
  \[
  \mathrm{Fix}(\jmath_m)=\{\,w(r):\ 2r\equiv F_{m+1}-2\pmod{F_{m+2}}\,\}.
  \]
  从而当 \(F_{m+2}\) 为奇数时 \(\card{\mathrm{Fix}(\jmath_m)}=1\)；当 \(F_{m+2}\) 为偶数时 \(\mathrm{Fix}(\jmath_m)=\varnothing\)。
  \item 令纤维退化直方图
  \[
  H_m(d):=\#\{x\in X_m:\ d_m(x)=d\}\qquad(d\in\NN).
  \]
  则对任意 \(d\) 有同余
  \[
  H_m(d)\equiv \#\{x\in \mathrm{Fix}(\jmath_m):\ d_m(x)=d\}\pmod 2.
  \]
  特别地：若 \(F_{m+2}\) 为偶数，则对所有 \(d\) 都有 \(H_m(d)\) 为偶数；若 \(F_{m+2}\) 为奇数，则至多一个 \(d\) 使 \(H_m(d)\) 为奇数。
\end{enumerate}
\end{proposition}
\begin{proof}
(1) 由 \(\jmath_m(w(r))=w(\tau_m(r))\) 与 \(w(\cdot)\) 的单射性，固定点条件等价于 \(\tau_m(r)\equiv r\pmod{F_{m+2}}\)，即 \(2r\equiv F_{m+1}-2\pmod{F_{m+2}}\)。
当 \(F_{m+2}\) 奇时，\(2\) 在模 \(F_{m+2}\) 下可逆，解唯一。
当 \(F_{m+2}\) 偶时（等价于 \(m+2\equiv 0\pmod 3\)），\(F_{m+1}\) 为奇数，从而 \(F_{m+1}-2\) 为奇数；因此线性同余无解。
\par
(2) 将 \(X_m\) 按 \(\jmath_m\) 的轨道分解为若干二点轨道与固定点轨道。
由命题 \ref{prop:fold-complement-duality} 的多重度守恒 \(d_m(x)=d_m(\jmath_m(x))\)，每条二点轨道对 \(H_m(d)\) 的贡献为 \(0\) 或 \(2\)，在模 \(2\) 意义下消失；故 \(H_m(d)\) 的奇偶性完全由固定点部分贡献，得到所示同余与推论。证毕。
\end{proof}

\begin{theorem}[自补稳定类型的周期 \texorpdfstring{$3$}{3} 前缀刻画]\label{thm:fold-complement-fixed-word-period3}
\leanverified{paper\_fold\_complement\_fixed\_word\_period3}
令 \(s^{(m)}\in X_m\) 为无限周期词 \((001)^\infty\) 的长度 \(m\) 前缀：
\[
s^{(m)}:=\pi_m\bigl((001)^\infty\bigr).
\]
则
\[
\mathrm{Fix}(\jmath_m)=
\begin{cases}
\{s^{(m)}\}, & m\not\equiv 1\pmod 3,\\[2pt]
\varnothing, & m\equiv 1\pmod 3.
\end{cases}
\]
因此在存在自补稳定类型的所有分辨率上，其词法形状被强制为周期 \(3\) 的前缀（密度 \(1/3\)）。
\end{theorem}
\begin{proof}
先给出一个求和恒等式：对任意 \(n\ge 1\)，令
\(
S_n:=\sum_{j=1}^{n}F_{3j+1}
\)，
则由递推 \(F_{k+3}=2F_{k+1}+F_k\) 归纳得
\[
2S_n=F_{3n+3}-2.
\]
\par
令 \(m\ge 1\) 且写成 \(m=3n\)、\(3n+1\) 或 \(3n+2\)。
当 \(m=3n\) 或 \(m=3n+2\) 时，\(s^{(m)}\) 的 \(1\) 恰出现在位置 \(k=3,6,\dots,3n\)，从而
\[
V_m(s^{(m)})=\sum_{j=1}^{n}F_{3j+1}=S_n.
\]
若 \(m=3n+2\)，则 \(2V_m(s^{(m)})=F_{m+1}-2\)；
若 \(m=3n\)，则 \(2V_m(s^{(m)})=F_{m+3}-2\equiv F_{m+1}-2\pmod{F_{m+2}}\)。
两种情形均推出 \(s^{(m)}\in \mathrm{Fix}(\jmath_m)\)（命题 \ref{prop:fold-complement-duality-fixedpoints-parity}(1)）。
又当 \(m\not\equiv 1\pmod 3\) 时 \(F_{m+2}\) 为奇数，固定点唯一，故 \(\mathrm{Fix}(\jmath_m)=\{s^{(m)}\}\)。
\par
当 \(m=3n+1\) 时，\(F_{m+2}=F_{3n+3}\) 为偶数且 \(F_{m+1}=F_{3n+2}\) 为奇数，从而 \(F_{m+1}-2\) 为奇数，故同余 \(2r\equiv F_{m+1}-2\pmod{F_{m+2}}\) 无解，固定点为空。证毕。
\end{proof}
