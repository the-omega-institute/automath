\begin{theorem}[参数碰撞核 \(A_4(t)\) 的谱自对偶恒等式]\label{thm:killo-fold-collision-spectral-self-duality}
设碰撞核 \(A_4(t)\) 为五阶参数伴随矩阵族，其特征多项式为
\[
\chi(t,\lambda):=\det(\lambda I-A_4(t))
=\lambda^5-2\lambda^4-t\lambda^3-2\lambda+2.
\]
则成立恒等式
\[
\boxed{\;
\chi(t,\lambda)+\chi(t,-\lambda)=4(1-\lambda^4),
\;}
\]
其右端与参数 \(t\) 无关。
\end{theorem}

\begin{proof}
代入 \(-\lambda\)：
\begin{align*}
\chi(t,-\lambda)
&=(-\lambda)^5-2(-\lambda)^4-t(-\lambda)^3-2(-\lambda)+2\\
&=-\lambda^5-2\lambda^4+t\lambda^3+2\lambda+2.
\end{align*}
与 \(\chi(t,\lambda)=\lambda^5-2\lambda^4-t\lambda^3-2\lambda+2\) 相加，
\(\lambda^5\) 项与 \(t\lambda^3\) 项及 \(\lambda\) 项逐对抵消，余
\[
-4\lambda^4+4=4(1-\lambda^4).\qedhere
\]
\end{proof}

\begin{corollary}[度量-张量分离]\label{cor:killo-fold-collision-metric-tensor-separation}
\leanverified{paper\_killo\_fold\_collision\_metric\_tensor\_separation}
恒等式 \(\chi(t,\lambda)+\chi(t,-\lambda)=4(1-\lambda^4)\)
将特征多项式的系数分为两类：
\begin{enumerate}
\item 偶数幂系数（\(\lambda^4,\lambda^2,\lambda^0\)）为 \((-2,0,2)\)，
      仅依赖碰撞核的二阶结构，与参数 \(t\) 无关；
\item 奇数幂系数（\(\lambda^5,\lambda^3,\lambda^1\)）为 \((1,-t,-2)\)，
      携带参数 \(t\) 的全部信息。
\end{enumerate}
在 Amari 信息几何框架中，偶数幂对应 Fisher 度量 \(g_{ij}\)（二阶结构），
奇数幂对应 Amari-Chentsov 三阶张量 \(C_{ijk}\)。
因此谱自对偶恒等式是如下原理的特征多项式实现：
Fisher 度量在三阶统计量变化下不变。
\end{corollary}

\begin{corollary}[自对偶零点的 Fibonacci 约束]\label{cor:killo-fold-collision-self-dual-zeros}
令 \(\lambda_0\) 为 \(\chi(t,\lambda_0)=0\) 的根。
则 \(\chi(t,-\lambda_0)=-4(1-\lambda_0^4)\)。
特别地，\(\lambda_0\) 与 \(-\lambda_0\) 同时为 \(\chi(t,\cdot)\) 的根
当且仅当 \(\lambda_0^4=1\)，
即 \(\lambda_0\in\{+1,-1,+i,-i\}\)。
对实伴随矩阵，此约束排除虚根，
仅允许 \(\lambda_0=\pm 1\) 上的对称谱。
\end{corollary}
