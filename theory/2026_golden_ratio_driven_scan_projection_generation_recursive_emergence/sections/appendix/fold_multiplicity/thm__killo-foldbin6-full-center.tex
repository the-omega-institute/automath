\begin{theorem}[window-\(6\) bin-fold 规范群中心的完整刻画]\label{thm:killo-foldbin6-full-center}
\leanverified{paper\_killo\_foldbin6\_full\_center}
在 \(m=6\) 的二进制区间折叠口径下，
\[
\mathcal{G}^{\mathrm{bin}}_6\cong S_2^{8}\times S_3^{4}\times S_4^{9},
\qquad
Z(\mathcal{G}^{\mathrm{bin}}_6)\cong (\ZZ_2)^8.
\]
并且由推论 \ref{cor:fold-bin-boundary-center-z2} 给出的 boundary sheet-flip 子群
\[
(\ZZ_2)^{X^{\mathrm{bdry}}_6}\hookrightarrow Z(\mathcal{G}^{\mathrm{bin}}_6)
\]
是一个三维坐标子格，从而
\[
Z(\mathcal{G}^{\mathrm{bin}}_6)\big/(\ZZ_2)^{X^{\mathrm{bdry}}_6}\cong (\ZZ_2)^5.
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:fold-bin-gauge-m6}，
\[
\mathcal{G}^{\mathrm{bin}}_6\cong S_2^{8}\times S_3^{4}\times S_4^{9}.
\]
直积群的中心等于各因子中心之直积，而
\[
Z(S_2)=S_2\cong \ZZ_2,
\qquad
Z(S_n)=1\quad(n\ge 3),
\]
故
\[
Z(\mathcal{G}^{\mathrm{bin}}_6)\cong Z(S_2)^8\times Z(S_3)^4\times Z(S_4)^9\cong (\ZZ_2)^8.
\]
再由推论 \ref{cor:fold-bin-boundary-center-z2}，boundary sheet-flip 已给出规范嵌入
\[
(\ZZ_2)^{X^{\mathrm{bdry}}_6}\cong (\ZZ_2)^3\hookrightarrow Z(\mathcal{G}^{\mathrm{bin}}_6).
\]
因此商群仍为初等 \(2\)-群，且维数为 \(8-3=5\)，即
\[
Z(\mathcal{G}^{\mathrm{bin}}_6)\big/(\ZZ_2)^{X^{\mathrm{bdry}}_6}\cong (\ZZ_2)^5.
\]
\end{proof}

\begin{corollary}[window-\(6\) 中心的 boundary--interior 规范直和]\label{cor:killo-foldbin6-center-boundary-interior}
\leanverified{paper\_killo\_foldbin6\_center\_boundary\_interior}
记
\[
X^{(2)}_6:=\{w\in X_6:\ d^{\mathrm{bin}}_6(w)=2\}.
\]
则由表 \ref{tab:fold_bin_gauge_m6_fibers} 有
\[
X^{(2)}_6=
\{\texttt{000001},\ \texttt{000101},\ \texttt{001001},\ \texttt{010001},\ \texttt{010101},\ \texttt{100001},\ \texttt{100101},\ \texttt{101001}\}.
\]
令
\[
X^{(2),\mathrm{int}}_6:=X^{(2)}_6\setminus X^{\mathrm{bdry}}_6
=
\{\texttt{000001},\ \texttt{000101},\ \texttt{001001},\ \texttt{010001},\ \texttt{010101}\}.
\]
则存在规范直和分解
\[
Z(\mathcal{G}^{\mathrm{bin}}_6)
\cong
(\ZZ_2)^{X^{\mathrm{bdry}}_6}\oplus (\ZZ_2)^{X^{(2),\mathrm{int}}_6}
\cong (\ZZ_2)^3\oplus(\ZZ_2)^5.
\]
\end{corollary}
\begin{proof}
表 \ref{tab:fold_bin_gauge_m6_fibers} 显示恰有上述八条纤维满足 \(d^{\mathrm{bin}}_6(w)=2\)。
由定理 \ref{thm:killo-foldbin6-full-center}，中心全部来自这八个 \(S_2\) 因子的非平凡元。
其中三条 boundary 纤维构成 \(X^{\mathrm{bdry}}_6\)，其对应中心子群正是推论 \ref{cor:fold-bin-boundary-center-z2} 的 \((\ZZ_2)^{X^{\mathrm{bdry}}_6}\)。
其余五条纤维对应的中心 involution 分属不交的直积因子，因而彼此可交换且线性独立，给出补空间 \((\ZZ_2)^{X^{(2),\mathrm{int}}_6}\)。
故得到所示规范直和分解。
\end{proof}
