\begin{corollary}[跨模型恒等式：最小退化扇区补集规模与 Fold 极大纤维的双尺度锁定]\label{cor:fold-bin-minsector-complement-maxfiber-audited}
\leanverified{paper\_fold\_bin\_minsector\_complement\_maxfiber\_audited}
在审计窗口 \(m\in\{6,8,10,12\}\) 上，令 \(\mathcal{S}_{m,d}\subseteq X_m\) 为 bin-fold 的退化谱扇区与最小退化度 \(d_{\min}(m)\) 如定义 \ref{def:fold-bin-degeneracy-sectors}。
记 Fold 在层级 \(n\) 的最大纤维基数为
\(
D_n:=\max_{x\in X_n} d_n(x)
\)
（见定理 \ref{thm:pom-max-fiber}）。
则成立精确恒等式
\[
\bigl|X_m\setminus \mathcal{S}_{m,d_{\min}(m)}\bigr| = D_{2m-2}.
\]
\end{corollary}
\begin{proof}
由定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 在上述 \(m\) 上有 \(\bigl|\mathcal{S}_{m,d_{\min}(m)}\bigr|=|X_{m-2}|\)。
又稳定类型计数律给出 \(|X_n|=F_{n+2}\)，从而
\[
\bigl|X_m\setminus \mathcal{S}_{m,d_{\min}(m)}\bigr|
=|X_m|-|X_{m-2}|
=F_{m+2}-F_m
=F_{m+1}.
\]
另一方面，\(2m-2=2(m-1)\) 为偶数；由定理 \ref{thm:pom-max-fiber} 得
\(
D_{2m-2}=F_{(m-1)+2}=F_{m+1}
\)。
两式相等，证毕。
\end{proof}

\begin{corollary}[最小退化 Fibonacci 律的不可无穷外推]\label{cor:fold-bin-min-degeneracy-fib-nonpersistent}
\leanverified{paper\_fold\_bin\_min\_degeneracy\_fib\_nonpersistent}
沿用定义 \ref{def:fold-bin-degeneracy-sectors} 的记号。
若对无穷多个充分大的偶数 \(m\) 同时成立
\[
d_{\min}(m)=F_{m/2},
\]
则与总预像计数恒等式 \(|\Omega_m|=2^m\) 矛盾。
因此，定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 的已审计等式 \(d_{\min}(m)=F_{m/2}\) 不可能在偶数窗上无条件延拓为“对所有 \(m\ge 6\) 恒成立”的规律。
\end{corollary}
\begin{proof}
由定义，bin-fold 的纤维大小满足
\[
2^m
=|\Omega_m|
=\sum_{w\in X_m}\bigl|(\Fold_m^{\mathrm{bin}})^{-1}(w)\bigr|
\ge |X_m|\,d_{\min}(m).
\]
又稳定类型计数律给出 \(|X_m|=F_{m+2}\)，故若 \(d_{\min}(m)=F_{m/2}\) 则
\[
2^m\ \ge\ F_{m+2}\,F_{m/2}.
\]
由 Fibonacci 数的指数渐近 \(F_n\asymp \varphi^n\) 得 \(F_{m+2}F_{m/2}\asymp \varphi^{3m/2}\)。
注意 \(\varphi^3=2+\sqrt5>4\)，从而 \(\varphi^{3/2}>2\)，故存在 \(m_0\) 使对所有偶数 \(m\ge m_0\) 有 \(F_{m+2}F_{m/2}>2^m\)，与上式矛盾。证毕。
\end{proof}

\begin{corollary}[最小退化 Fibonacci 律的显式截止阈值]\label{cor:fold-bin-min-degeneracy-fib-explicit-cutoff}
\leanverified{paper\_fold\_bin\_min\_degeneracy\_fib\_explicit\_cutoff}
沿用推论 \ref{cor:fold-bin-min-degeneracy-fib-nonpersistent} 的记号。
对任意偶数 \(m\ge 24\)，等式
\[
d_{\min}(m)=F_{m/2}
\]
不可能成立。
\end{corollary}
\begin{proof}
若 \(d_{\min}(m)=F_{m/2}\)，则由推论 \ref{cor:fold-bin-min-degeneracy-fib-nonpersistent} 的推导必有
\begin{equation}\label{eq:fold-bin-min-degeneracy-product-ineq}
2^m\ \ge\ F_{m+2}\,F_{m/2}.
\end{equation}
取 \(m=24\)，直接计算得到
\[
F_{26}F_{12}=121393\cdot 144=17480592\ >\ 2^{24}=16777216,
\]
从而 \eqref{eq:fold-bin-min-degeneracy-product-ineq} 在 \(m=24\) 已矛盾。
\par
下面说明对所有更大的偶数亦不可能。
令
\[
R(m):=\frac{F_{m+2}F_{m/2}}{2^m}\qquad(m\ \text{为偶数}).
\]
则
\[
\frac{R(m+2)}{R(m)}
=\frac{F_{m+4}}{F_{m+2}}\cdot \frac{F_{m/2+1}}{F_{m/2}}\cdot \frac{1}{4}.
\]
利用 \(\varphi=[1;1,1,1,\dots]\) 的连分数收敛子性质：对任意 \(n\ge 1\) 有
\[
\left|\frac{F_{n+1}}{F_n}-\varphi\right|<\frac{1}{F_n^2},
\]
从而当 \(n\ge 12\) 时
\[
\frac{F_{n+1}}{F_n}>\varphi-\frac{1}{F_{12}^2},
\qquad
\frac{F_{n+2}}{F_n}=1+\frac{F_{n+1}}{F_n}>\varphi^2-\frac{1}{F_{12}^2}.
\]
当 \(m\ge 24\) 时同时有 \(m/2\ge 12\) 与 \(m+2\ge 26\ge 12\)，故
\[
\frac{R(m+2)}{R(m)}
>\frac{\bigl(\varphi^2-\frac{1}{F_{12}^2}\bigr)\bigl(\varphi-\frac{1}{F_{12}^2}\bigr)}{4}.
\]
注意 \(F_{12}=144\)，令 \(\varepsilon:=1/F_{12}^2=1/20736\)。
则
\[
\bigl(\varphi^2-\varepsilon\bigr)\bigl(\varphi-\varepsilon\bigr)
=\varphi^3-\varepsilon(\varphi^2+\varphi)+\varepsilon^2
>4,
\]
因为 \(\varphi^3=2+\sqrt5\) 且 \(\varepsilon(\varphi^2+\varphi)<5/20736\)。
因此 \(\frac{R(m+2)}{R(m)}>1\)，即 \(R(m)\) 在所有偶数 \(m\ge 24\) 上严格递增。
既然 \(R(24)>1\)，便对所有偶数 \(m\ge 24\) 都有 \(R(m)>1\)，亦即 \(F_{m+2}F_{m/2}>2^m\)，与 \eqref{eq:fold-bin-min-degeneracy-product-ineq} 矛盾。
证毕。
\end{proof}

\begin{corollary}[子覆盖同构外推下的最小扇区密度极限]\label{cor:fold-bin-minsector-density-phi-minus2}
\leanverified{paper\_fold\_bin\_minsector\_density\_phi\_minus2}
沿用定义 \ref{def:fold-bin-degeneracy-sectors} 的记号。
若“子覆盖同构”恒等式
\(
\bigl|\mathcal{S}_{m,d_{\min}(m)}\bigr|=|X_{m-2}|
\)
对一列趋于无穷的偶数 \(m\) 成立，则存在极限
\[
\lim_{\substack{m\to\infty\\ m\ \mathrm{even}}}
\frac{\bigl|\mathcal{S}_{m,d_{\min}(m)}\bigr|}{|X_m|}
\;=\;
\varphi^{-2}.
\]
\end{corollary}
\begin{proof}
在上述 \(m\) 上，
\(
\frac{|\mathcal{S}_{m,d_{\min}(m)}|}{|X_m|}=\frac{|X_{m-2}|}{|X_m|}=\frac{F_m}{F_{m+2}}
\)。
由 \(F_{m+2}\sim \varphi^2 F_m\) 得极限为 \(\varphi^{-2}\)。证毕。
\end{proof}

\begin{corollary}[退化谱多值性强制非自由作用：自由群作用与线性商障碍]\label{cor:foldbin6-degeneracy-multivalued-nonfree}
在 \(m=6\) 上，bin-fold 纤维退化度直方图为 \(2:8,3:4,4:9\)（注 \ref{rem:fold-bin-gauge-volume-m6}），因而纤维大小在同一窗口内取多值。
因此等价关系
\(
\omega\sim\eta\iff \Fold^{\mathrm{bin}}_6(\omega)=\Fold^{\mathrm{bin}}_6(\eta)
\)
不可能由任何有限群在 \(\Omega_6\) 上的自由作用之轨道划分实现；亦不可能来自任何阿贝尔群的线性陪集划分（推论 \ref{cor:terminal-foldbin6-no-linear-kernel}）。
从而任何将该多对一压缩解释为“对称/规范冗余”的方案必然要求变稳定子（非自由作用）或更一般的群胚结构。
\end{corollary}
\begin{proof}
若有限群 \(H\) 在 \(\Omega_6\) 上自由作用，则任一轨道大小恒等于 \(|H|\)，从而所有轨道同势；与退化度多值性矛盾，故不可能。
对线性陪集划分，纤维亦同势；其在 window-\(6\) 上的不可实现性由推论 \ref{cor:terminal-foldbin6-no-linear-kernel} 给出。证毕。
\end{proof}
