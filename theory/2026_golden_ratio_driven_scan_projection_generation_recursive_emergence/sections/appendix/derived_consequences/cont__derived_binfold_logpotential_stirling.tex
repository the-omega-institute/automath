\subsection{bin-fold 对数势与规范体积的 Stirling 常数}

在 bin-fold 退化谱的双折点极限、fold-gauge 常数列 \(C_m\) 的 \(2\pi\) 回复以及规范体积密度的二阶 Stirling 展开已经建立之后，还可把均匀纤维平均的对数势与规范体积差额继续压缩为完全显式的常数律。

\begin{theorem}[bin-fold 平均纤维对数势的精确常数项]\label{thm:derived-binfold-average-log-potential-constant}
\leanverified{paper\_derived\_binfold\_average\_log\_potential\_constant}
记
\[
L_m:=\frac{1}{|X_m|}\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w).
\]
则当 \(m\to\infty\) 时，
\[
L_m
=
m\log\!\left(\frac{2}{\varphi}\right)
-\varphi^{-2}\log\varphi
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
特别地，
\[
\lim_{m\to\infty}\left(
L_m-m\log\!\left(\frac{2}{\varphi}\right)
\right)
=
-\varphi^{-2}\log\varphi.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-bin-two-state-asymptotic}，对一切 \(w\in X_m\) 都有一致展开
\[
d_m^{\mathrm{bin}}(w)
=
\varphi^{-w_m}\left(\frac{2}{\varphi}\right)^m
\left(1+O\!\left(\left(\frac{\varphi}{2}\right)^m\right)\right).
\]
取对数得
\[
\log d_m^{\mathrm{bin}}(w)
=
m\log\!\left(\frac{2}{\varphi}\right)-w_m\log\varphi
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right),
\]
误差在 \(w\) 上一致。对 \(w\in X_m\) 求平均，并用
\[
\card\{w\in X_m:w_m=0\}=F_{m+1},
\qquad
\card\{w\in X_m:w_m=1\}=F_m,
\qquad
|X_m|=F_{m+2},
\]
得到
\[
L_m
=
m\log\!\left(\frac{2}{\varphi}\right)
-\frac{F_m}{F_{m+2}}\log\varphi
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
再由 Binet 公式，
\[
\frac{F_m}{F_{m+2}}=\varphi^{-2}+O(\varphi^{-m}),
\]
而 \(O(\varphi^{-m})\) 可并入 \(O((\varphi/2)^m)\)，遂得所示展开。证毕。
\end{proof}

\begin{theorem}[bin-fold 规范体积差额的第二渐近式]\label{thm:derived-binfold-gauge-volume-second-asymptotic}
沿用定理 \ref{thm:fold-bin-gauge-volume-stirling-second-order} 与推论 \ref{cor:fold-bin-recover-2pi} 的记号，记
\[
g_m:=2^{-m}\log|\mathcal G_m^{\mathrm{bin}}|,
\qquad
\kappa_m:=2^{-m}\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w),
\]
\[
C_m:=
\exp\!\left(
\frac{2^{m+1}}{|X_m|}(g_m-\kappa_m+1)
-\frac{1}{|X_m|}\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)
\right).
\]
则有
\[
\frac{2^{m+1}}{|X_m|}(g_m-\kappa_m+1)
=
m\log\!\left(\frac{2}{\varphi}\right)
+\log(2\pi)
-\varphi^{-2}\log\varphi
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
等价地，
\[
g_m
=
\kappa_m-1
+\frac{|X_m|}{2^{m+1}}
\left(
m\log\!\left(\frac{2}{\varphi}\right)
+\log(2\pi)
-\varphi^{-2}\log\varphi
\right)
+o\!\left(\frac{|X_m|}{2^m}\right).
\]
再进一步，
\[
\log|\mathcal G_m^{\mathrm{bin}}|
=
\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w)
-2^m
+\frac12\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)
+\frac{|X_m|}{2}\log(2\pi)
+o(|X_m|).
\]
\end{theorem}
\begin{proof}
由 \(C_m\) 的定义，
\[
\log C_m
=
\frac{2^{m+1}}{|X_m|}(g_m-\kappa_m+1)
-\frac{1}{|X_m|}\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w).
\]
推论 \ref{cor:fold-bin-recover-2pi} 给出
\[
\log C_m=\log(2\pi)+O\!\left(\left(\frac{\varphi}{2}\right)^m\right),
\]
而定理 \ref{thm:derived-binfold-average-log-potential-constant} 给出
\[
\frac{1}{|X_m|}\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)
=
m\log\!\left(\frac{2}{\varphi}\right)
-\varphi^{-2}\log\varphi
+O\!\left(\left(\frac{\varphi}{2}\right)^m\right).
\]
两式相加即得第一条展开。

第二条只需把第一条乘上 \(|X_m|/2^{m+1}\)。由于
\[
\frac{|X_m|}{2^m}
=
\Theta\!\left(\left(\frac{\varphi}{2}\right)^m\right),
\]
其误差项满足
\[
\frac{|X_m|}{2^{m+1}}\,O\!\left(\left(\frac{\varphi}{2}\right)^m\right)
=
O\!\left(\left(\frac{\varphi}{2}\right)^{2m}\right)
=
o\!\left(\frac{|X_m|}{2^m}\right).
\]

最后，把第二条乘以 \(2^m\)，并用
\[
2^m\kappa_m=\sum_{w\in X_m}d_m^{\mathrm{bin}}(w)\log d_m^{\mathrm{bin}}(w),
\qquad
|X_m|\,L_m=\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w),
\]
再结合定理 \ref{thm:derived-binfold-average-log-potential-constant} 的
\[
\frac{|X_m|}{2}
\left(
m\log\!\left(\frac{2}{\varphi}\right)-\varphi^{-2}\log\varphi
\right)
=
\frac12\sum_{w\in X_m}\log d_m^{\mathrm{bin}}(w)+o(|X_m|),
\]
即可得到第三条。证毕。
\end{proof}
