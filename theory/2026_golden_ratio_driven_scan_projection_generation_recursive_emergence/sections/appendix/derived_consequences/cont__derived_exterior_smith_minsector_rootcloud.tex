\subsection{外幂 Smith 几何、最小退化扇区与 window-\texorpdfstring{$6$}{6} 根云}

定理 \ref{thm:xi-exterior-power-smith-schubert}、推论 \ref{cor:xi-exterior-power-poincare-duality} 与命题 \ref{prop:xi-exterior-power-gaussian-binom-hilbert} 已把外幂余核 \(M_{q,k}\) 的 \(\pi_\varphi\)-指数谱参数化为 Schubert 矩形中的分拆面积，并给出镜像互补与 Hilbert 级数闭式。另一方面，命题 \ref{prop:fold-bin-chi2-col}、定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 与推论 \ref{cor:fold-bin-minsector-complement-maxfiber-audited} 已把 bin-fold 的 \(\chi^2\) 偏离、最小退化扇区及其补集规模压缩为可审计的有限尺度数据；表 \ref{tab:fold6_b3c3_root_dictionary_b3}、表 \ref{tab:fold6_b3c3_root_dictionary_c3} 与定理 \ref{thm:xi-window6-c3-quadratic-energy-equipartition} 则给出了 window-\(6\) 循环扇区的 \(B_3/C_3\) 根字典与 \(C_3\) 二次能量等分。将这些接口进一步收束，可得到下列派生结构。

\begin{theorem}[\texorpdfstring{$\pi_\varphi$}{pi}-进核增长、Hilbert 层函数与镜像互补]\label{thm:derived-exterior-power-kernel-growth-hilbert-mirror}
沿用定理 \ref{thm:xi-exterior-power-smith-schubert} 与命题 \ref{prop:xi-exterior-power-gaussian-binom-hilbert} 的记号。令
\[
\mathcal O=\ZZ[\varphi],
\qquad
\pi_\varphi=\varphi+2,
\qquad
M_{q,k}:=\operatorname{coker}_{\mathcal O}(\Lambda^k C_q),
\qquad
N:=\binom{q+1}{k},
\]
并置
\[
a:=q+1-k,
\qquad
e(\lambda):=\frac{k(k-1)}{2}+|\lambda|
\qquad
(\lambda\subseteq a^k).
\]
对每个整数 \(r\ge 1\)，定义模 \((\pi_\varphi^r)\) 的核增长函数
\[
\Sigma_{q,k}(r):=
\log_5 \#
\ker\!\left(
(\Lambda^k C_q)_r:(\mathcal O/(\pi_\varphi^r))^N\to (\mathcal O/(\pi_\varphi^r))^N
\right),
\]
其中 \((\Lambda^k C_q)_r\) 表示 \(\Lambda^k C_q\) 的模 \((\pi_\varphi^r)\) 约化。再记
\[
H_{q,k}(t)=\sum_{s\ge 0} h_s t^s
=
\frac{N-E_{q,k}(t)}{1-t},
\qquad
E_{q,k}(t)=\sum_{\lambda\subseteq a^k} t^{e(\lambda)}.
\]
则有：
\begin{enumerate}
  \item 核增长满足显式公式
  \[
  \Sigma_{q,k}(r)=\sum_{\lambda\subseteq a^k}\min\bigl(e(\lambda),r\bigr).
  \]
  \item Hilbert 层函数是核增长的一阶差分：
  \[
  h_s=\Sigma_{q,k}(s+1)-\Sigma_{q,k}(s)
  \qquad (0\le s\le kq-1).
  \]
  \item 层函数满足镜像互补律
  \[
  h_s+h_{kq-1-s}=N
  \qquad (0\le s\le kq-1).
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-exterior-power-smith-schubert}，
\[
M_{q,k}\cong \bigoplus_{\lambda\subseteq a^k}\mathcal O/(\pi_\varphi^{e(\lambda)}).
\]
这等价于存在幺模矩阵把 \(\Lambda^k C_q\) 化为对角矩阵
\[
\operatorname{diag}\bigl(\pi_\varphi^{e(\lambda)}\bigr)_{\lambda\subseteq a^k}.
\]
模 \((\pi_\varphi^r)\) 约化后，每个直和因子都化为
\[
\mathcal O/(\pi_\varphi^r)\xrightarrow{\ \cdot \pi_\varphi^{e(\lambda)}\ }\mathcal O/(\pi_\varphi^r).
\]
对单个指数 \(e\)，该乘法的核恰为
\[
(\pi_\varphi^{r-\min(e,r)})/(\pi_\varphi^r),
\]
其基数为 \(5^{\min(e,r)}\)，因为 \(\#(\mathcal O/\pi_\varphi^m)=5^m\)。乘法性立即给出
\[
\#\ker((\Lambda^k C_q)_r)
=
5^{\sum_{\lambda}\min(e(\lambda),r)},
\]
对 \(\log_5\) 取对数便得第一式。

再由命题 \ref{prop:xi-exterior-power-gaussian-binom-hilbert} 的直和分解解释，
\[
h_s=\sum_{\lambda\subseteq a^k}\mathbf 1_{\{e(\lambda)\ge s+1\}},
\]
因为单个循环因子 \(\mathcal O/(\pi_\varphi^e)\) 在第 \(s\) 层
\(
\pi_\varphi^s/\pi_\varphi^{s+1}
\)
贡献 \(1\) 当且仅当 \(e\ge s+1\)。另一方面，
\[
\Sigma_{q,k}(s+1)-\Sigma_{q,k}(s)
=
\sum_{\lambda}
\bigl(\min(e(\lambda),s+1)-\min(e(\lambda),s)\bigr),
\]
而括号中的差恰等于 \(\mathbf 1_{\{e(\lambda)\ge s+1\}}\)。故得第二式。

最后，由推论 \ref{cor:xi-exterior-power-poincare-duality}，
\[
e(\lambda^\vee)=kq-e(\lambda).
\]
于是
\[
h_{kq-1-s}
=
\#\{\lambda:\ e(\lambda)\ge kq-s\}
=
\#\{\lambda:\ e(\lambda^\vee)\ge kq-s\}
=
\#\{\lambda:\ e(\lambda)\le s\}.
\]
该集合与 \(\{e(\lambda)\ge s+1\}\) 互补，总数为 \(N\)，故
\[
h_s+h_{kq-1-s}=N.
\]
证毕。
\end{proof}

\begin{theorem}[Schubert--Smith 指数谱的自对偶矩定律]\label{thm:derived-schubert-smith-selfdual-moment-law}
\leanverified{paper\_derived\_schubert\_smith\_selfdual\_moment\_law}
沿用定理 \ref{thm:derived-exterior-power-kernel-growth-hilbert-mirror} 的记号，令
\[
\mathcal E_{q,k}:=\{e(\lambda):\ \lambda\subseteq a^k\}
\]
按重数计为大小 \(N=\binom{q+1}{k}\) 的多重集。则：
\begin{enumerate}
  \item 指数生成函数满足自倒数函数方程
  \[
  E_{q,k}(t)=t^{kq}E_{q,k}(t^{-1}).
  \]
  \item 一阶矩满足
  \[
  \frac1N\sum_{\lambda\subseteq a^k}e(\lambda)=\frac{kq}{2}.
  \]
  \item 全部奇数中心矩消失：
  \[
  \sum_{\lambda\subseteq a^k}\left(e(\lambda)-\frac{kq}{2}\right)^{2r+1}=0
  \qquad (r\ge 0).
  \]
  \item 二阶中心矩具有闭式
  \[
  \frac1N\sum_{\lambda\subseteq a^k}\left(e(\lambda)-\frac{kq}{2}\right)^2
  =
  \frac{k(q+2)(q+1-k)}{12}.
  \]
\end{enumerate}
\end{theorem}
\begin{proof}
由推论 \ref{cor:xi-exterior-power-poincare-duality}，
\[
e(\lambda^\vee)=kq-e(\lambda).
\]
因此
\[
E_{q,k}(t)
=
\sum_{\lambda\subseteq a^k}t^{e(\lambda)}
=
\sum_{\lambda\subseteq a^k}t^{kq-e(\lambda^\vee)}
=
t^{kq}\sum_{\lambda\subseteq a^k}t^{-e(\lambda)}
=
t^{kq}E_{q,k}(t^{-1}),
\]
得到第一式。

第二式正是命题 \ref{prop:xi-exterior-power-det-coker-size} 中
\[
\sum_{i=1}^{N}e_i
=
\binom{q+1}{k}\frac{kq}{2}
\]
的平均值写法。

第三式由中心对称性直接推出。因为多重集 \(\mathcal E_{q,k}\) 在变换
\[
e\longmapsto kq-e
\]
下不变，故关于中心 \(kq/2\) 的任意奇数幂项都按互补对偶逐项相消。

对第四式，由定理 \ref{thm:xi-exterior-power-smith-schubert}，\(\mathcal E_{q,k}\) 正是全部 \(k\)-子集和多重集。令随机变量 \(X\) 为从 \(\{0,1,\dots,q\}\) 中均匀抽取一个 \(k\)-子集 \(S\) 后的子集和
\[
X:=\sum_{i\in S} i.
\]
则 \(\mathcal E_{q,k}\) 的二阶中心矩恰为 \(\operatorname{Var}(X)\)。

记总体大小 \(N_0:=q+1\)，并设
\[
I_i:=\mathbf 1_{\{i\in S\}}
\qquad (0\le i\le q).
\]
则
\[
\mathbb E[I_i]=\frac{k}{N_0},
\qquad
\mathbb E[I_iI_j]=\frac{k(k-1)}{N_0(N_0-1)}
\qquad (i\ne j),
\]
从而
\[
\operatorname{Var}(I_i)=\frac{k(N_0-k)}{N_0^2},
\qquad
\operatorname{Cov}(I_i,I_j)=-\frac{k(N_0-k)}{N_0^2(N_0-1)}.
\]
又有
\[
\sum_{i=0}^{q} i=\frac{q(q+1)}{2},
\qquad
\sum_{i=0}^{q} i^2=\frac{q(q+1)(2q+1)}{6}.
\]
于是
\[
\mathbb E[X]=\sum_{i=0}^{q}i\,\mathbb E[I_i]
=
\frac{k}{q+1}\cdot \frac{q(q+1)}{2}
=
\frac{kq}{2},
\]
与第二式一致。再由
\[
\operatorname{Var}(X)=\sum_i i^2\operatorname{Var}(I_i)+2\sum_{i<j}ij\,\operatorname{Cov}(I_i,I_j)
\]
及
\[
2\sum_{i<j}ij
=
\left(\sum_i i\right)^2-\sum_i i^2,
\]
代入整理即可得到
\[
\operatorname{Var}(X)=\frac{k(q+2)(q+1-k)}{12}.
\]
这正是所求二阶中心矩。证毕。
\end{proof}

\begin{theorem}[最小退化扇区给出的 \texorpdfstring{$\chi^2$}{chi2} 两块下界]\label{thm:derived-binfold-minsector-chi2-two-block-bound}
\leanverified{paper\_derived\_binfold\_minsector\_chi2\_two\_block\_bound}
沿用定义 \ref{def:fold-bin-degeneracy-sectors}、定理 \ref{thm:fold-bin-min-degeneracy-fib-audited}、推论 \ref{cor:fold-bin-minsector-complement-maxfiber-audited} 与命题 \ref{prop:fold-bin-chi2-col} 的记号。对审计偶数窗
\[
m\in\{6,8,10,12\},
\]
记
\[
\mathcal S_{m,*}:=\mathcal S_{m,d_{\min}(m)},
\qquad
p_m^{\mathrm{bin}}(w):=\frac{d_m^{\mathrm{bin}}(w)}{2^m},
\qquad
u_m(w):=\frac{1}{|X_m|}.
\]
则最小退化扇区与其补集给出两块精确下界
\[
\chi^2\!\bigl(p_m^{\mathrm{bin}}\|u_m\bigr)
\ge
\frac{\Delta_m^2}{u_m(\mathcal S_{m,*})}
+
\frac{\Delta_m^2}{u_m(X_m\setminus \mathcal S_{m,*})},
\]
其中
\[
\Delta_m:=p_m^{\mathrm{bin}}(\mathcal S_{m,*})-u_m(\mathcal S_{m,*}).
\]
进一步代入
\[
d_{\min}(m)=F_{m/2},
\qquad
|\mathcal S_{m,*}|=F_m,
\qquad
|X_m\setminus \mathcal S_{m,*}|=F_{m+1},
\qquad
|X_m|=F_{m+2},
\]
得到显式形式
\[
\chi^2\!\bigl(p_m^{\mathrm{bin}}\|u_m\bigr)
\ge
\frac{F_m}{F_{m+1}}
\left(
\frac{F_{m+2}F_{m/2}}{2^m}-1
\right)^2.
\]
\end{theorem}
\begin{proof}
对任意概率分布 \(p,u\) 与任意子集 \(A\)，有
\[
\sum_{x\in A}\frac{(p(x)-u(x))^2}{u(x)}
\ge
\frac{(p(A)-u(A))^2}{u(A)},
\]
这是 Cauchy--Schwarz 不等式的直接推论。将其分别应用于
\[
A=\mathcal S_{m,*},
\qquad
A^c=X_m\setminus \mathcal S_{m,*},
\]
得到
\[
\chi^2\!\bigl(p_m^{\mathrm{bin}}\|u_m\bigr)
\ge
\frac{\bigl(p_m^{\mathrm{bin}}(\mathcal S_{m,*})-u_m(\mathcal S_{m,*})\bigr)^2}{u_m(\mathcal S_{m,*})}
+
\frac{\bigl(p_m^{\mathrm{bin}}(X_m\setminus \mathcal S_{m,*})-u_m(X_m\setminus \mathcal S_{m,*})\bigr)^2}{u_m(X_m\setminus \mathcal S_{m,*})}.
\]
由于两块概率和都等于 \(1\)，第二个分子中的差恰为 \(-\Delta_m\)，于是得到第一条两块下界。

再由定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 与推论 \ref{cor:fold-bin-minsector-complement-maxfiber-audited}，
\[
u_m(\mathcal S_{m,*})=\frac{F_m}{F_{m+2}},
\qquad
u_m(X_m\setminus \mathcal S_{m,*})=\frac{F_{m+1}}{F_{m+2}}.
\]
另一方面，\(\mathcal S_{m,*}\) 上共有 \(F_m\) 条纤维，每条纤维大小都等于 \(F_{m/2}\)，故
\[
p_m^{\mathrm{bin}}(\mathcal S_{m,*})
=
\frac{F_mF_{m/2}}{2^m}.
\]
因此
\[
\Delta_m
=
\frac{F_mF_{m/2}}{2^m}-\frac{F_m}{F_{m+2}}
=
\frac{F_m}{F_{m+2}}
\left(
\frac{F_{m+2}F_{m/2}}{2^m}-1
\right).
\]
将这些量代入两块下界，得到
\[
\chi^2\!\bigl(p_m^{\mathrm{bin}}\|u_m\bigr)
\ge
\Delta_m^2
\left(
\frac{F_{m+2}}{F_m}+\frac{F_{m+2}}{F_{m+1}}
\right).
\]
利用 Fibonacci 递推 \(F_{m+2}=F_{m+1}+F_m\)，化简为
\[
\frac{F_{m+2}}{F_m}+\frac{F_{m+2}}{F_{m+1}}
=
\frac{F_{m+2}^2}{F_mF_{m+1}}.
\]
再代入 \(\Delta_m\) 的表达式，即得
\[
\chi^2\!\bigl(p_m^{\mathrm{bin}}\|u_m\bigr)
\ge
\frac{F_m}{F_{m+1}}
\left(
\frac{F_{m+2}F_{m/2}}{2^m}-1
\right)^2.
\]
证毕。
\end{proof}

\begin{theorem}[window-\texorpdfstring{$6$}{6} 的 \texorpdfstring{$B_3/C_3$}{B3/C3} 根云是严格二阶各向同性设计]\label{thm:derived-window6-b3c3-rootcloud-isotropic-design}
沿用定理 \ref{thm:xi-window6-cyclic-kernel-bc3-root-datum} 的记号，令
\[
\iota_B:X_6^{\mathrm{cyc}}\to \mathbb R^3
\]
为表 \ref{tab:fold6_b3c3_root_dictionary_b3} 给出的 \(B_3\) 根字典，
\[
\iota_C:X_6^{\mathrm{cyc}}\to \mathbb R^3
\]
为表 \ref{tab:fold6_b3c3_root_dictionary_c3} 给出的 \(C_3\) 根字典。则有
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_B(w)=0,
\qquad
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_B(w)\iota_B(w)^{\mathsf T}=10I_3,
\]
以及
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_C(w)=0,
\qquad
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_C(w)\iota_C(w)^{\mathsf T}=16I_3.
\]
等价地，对任意 \(u\in\mathbb R^3\)，
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\langle u,\iota_B(w)\rangle^2=10\,|u|^2,
\qquad
\sum_{w\in X_6^{\mathrm{cyc}}}\langle u,\iota_C(w)\rangle^2=16\,|u|^2.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:xi-window6-cyclic-kernel-bc3-root-datum}，在 \(B_3\) 规范下，
\[
\iota_B(X_6^{\mathrm{cyc}})
=
\{\pm e_1,\pm e_2,\pm e_3\}
\sqcup
\{\pm e_i\pm e_j:\ 1\le i<j\le 3\}.
\]
该集合关于原点中心对称，因此
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_B(w)=0.
\]
其二阶矩分为两部分。短根部分给出
\[
\sum_{\alpha\in\{\pm e_1,\pm e_2,\pm e_3\}}\alpha\alpha^{\mathsf T}=2I_3.
\]
对每一对 \(1\le i<j\le 3\)，四个长根 \(\pm e_i\pm e_j\) 的外积和为
\[
\sum_{\varepsilon,\eta\in\{\pm1\}}
(\varepsilon e_i+\eta e_j)(\varepsilon e_i+\eta e_j)^{\mathsf T}
=
4\bigl(e_ie_i^{\mathsf T}+e_je_j^{\mathsf T}\bigr),
\]
因为所有交叉项都按符号对称相消。对三对 \((i,j)\) 求和，每个坐标方向恰出现两次，故长根总贡献为
\[
8I_3.
\]
两部分相加即得
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_B(w)\iota_B(w)^{\mathsf T}=10I_3.
\]

在 \(C_3\) 规范下，由定理 \ref{thm:xi-window6-cyclic-kernel-bc3-root-datum}，非轴向十二个向量保持不变，而六个轴向向量从 \(\pm e_i\) 变为 \(\pm 2e_i\)。因此总和仍因中心对称而为零。又由定理 \ref{thm:xi-window6-c3-quadratic-energy-equipartition}，
\[
\sum_{w\in X_6^{\mathrm{cyc}}}\iota_C(w)\iota_C(w)^{\mathsf T}=16I_3.
\]
最后，对任意 \(u\in\mathbb R^3\)，利用
\[
\langle u,v\rangle^2=u^{\mathsf T}(vv^{\mathsf T})u
\]
即可将矩阵恒等式改写为二次型恒等式。证毕。
\end{proof}

\begin{remark}[根云与二次统计的旋转不变性]\label{rem:derived-window6-rootcloud-quadratic-isotropy}
定理 \ref{thm:derived-window6-b3c3-rootcloud-isotropic-design} 表明，window-\(6\) 的 cyclic 扇区不仅可由 \(B_3/C_3\) 根字典确定，而且其根云在二阶矩层面已经严格实现了各向同性框架常数。
\end{remark}

\begin{corollary}[带多重度根云的 signed-Weyl 压缩与定向轴刚化]\label{cor:derived-window6-multiplicity-rootcloud-signed-weyl-collapse}
沿用推论 \ref{cor:conclusion-window6-signed-weyl-collapse} 的记号，记
\[
R_j:=\{\alpha\in \Phi(B_3):\ m_6(\alpha)=j\}
\qquad (j=2,3,4).
\]
则任一保持三层 \(R_2,R_3,R_4\) 逐层不变的 signed Weyl 变换 \(g\) 都满足
\[
g\in
K_4:=
\left\{
\operatorname{diag}(1,\varepsilon_2,\varepsilon_3):\
\varepsilon_2,\varepsilon_3\in\{\pm 1\}
\right\}
\cong (\ZZ/2\ZZ)^2.
\]
特别地，带多重度标记的根云把 \(W(B_3)\) 的 signed hyperoctahedral 对称严格压缩到 Klein 四群，并唯一刚化出有向轴
\[
\RR e_1.
\]
\end{corollary}
\begin{proof}
由推论 \ref{cor:conclusion-window6-signed-weyl-collapse}，
\[
R_2=\{-e_1\}\cup\{\pm e_2\pm e_3\},
\qquad
R_3=\{\pm e_1\pm e_3\},
\qquad
R_4=\{e_1,\pm e_2,\pm e_3,\pm e_1\pm e_2\},
\]
并且
\[
\Aut_{\pm}\!\bigl(\Phi(B_3),m_6\bigr)
=
\left\{
\operatorname{diag}(1,\varepsilon_2,\varepsilon_3):\ \varepsilon_2,\varepsilon_3\in\{\pm1\}
\right\}.
\]
任何逐层保持 \(R_2,R_3,R_4\) 的 signed Weyl 变换恰是 \(\Aut_{\pm}(\Phi(B_3),m_6)\) 的元素，因此只能保留第二、第三坐标的独立变号，而第一坐标轴的方向被唯一固定。证毕。
\end{proof}
