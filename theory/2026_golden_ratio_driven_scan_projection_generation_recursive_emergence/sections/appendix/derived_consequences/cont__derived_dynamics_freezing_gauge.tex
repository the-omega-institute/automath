\subsection{Lee--Yang 地址塔的动力学 \texorpdfstring{$\zeta$}{zeta} 与算子代数刚性}

在定理 \ref{thm:derived-leyang-profinite-rigidity} 已将 Lee--Yang 分支塔识别为五份四符号满移位之后，还可进一步把周期点计数、Artin--Mazur \(\zeta\) 函数以及相应的 Bowen--Franks/Cuntz--Krieger 不变量写成完全闭式。

\begin{theorem}[Lee--Yang 地址塔的 Artin--Mazur \texorpdfstring{$\zeta$}{zeta} 函数与原始轨道计数闭式]\label{thm:derived-leyang-artin-mazur-zeta}
\leanverified{paper\_derived\_leyang\_artin\_mazur\_zeta}
记
\[
\mathfrak X_{\mathrm{LY}}:=\varprojlim_n \mathcal R_n,
\]
并令 \(\sigma\) 为由层间截断诱导的地址移位。则对每个整数 \(n\ge 1\) 都有
\[
\#\operatorname{Fix}(\sigma^n)=5\cdot 4^n.
\]
因而其 Artin--Mazur 动力学 \(\zeta\)-函数满足
\[
\zeta_{\sigma}(z)
:=
\exp\!\Bigl(\sum_{n\ge 1}\frac{\#\operatorname{Fix}(\sigma^n)}{n}z^n\Bigr)
=(1-4z)^{-5}.
\]
若再记长度恰为 \(n\) 的原始闭轨道数为 \(\operatorname{Prim}_n(\sigma)\)，则
\[
\operatorname{Prim}_n(\sigma)
=
\frac{5}{n}\sum_{d\mid n}\mu(d)\,4^{n/d}.
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:derived-leyang-profinite-rigidity}，
\[
\mathfrak X_{\mathrm{LY}}
\cong
\bigsqcup_{i=0}^{4}\mathcal A^{\NN},
\qquad
\mathcal A:=\{00,01,10,11\},
\qquad
|\mathcal A|=4,
\]
并且 \(\sigma\) 在每个分支上都对应于满四元单侧移位。于是 \(\sigma^n\)-不动点恰为长度 \(n\) 的周期字在该分支上的无限重复，因此每个分支恰有 \(4^n\) 个不动点，总数为 \(5\cdot 4^n\)。

代入 Artin--Mazur \(\zeta\) 函数定义，
\[
\zeta_{\sigma}(z)
=
\exp\!\Bigl(5\sum_{n\ge 1}\frac{(4z)^n}{n}\Bigr)
=
\exp\!\bigl(-5\log(1-4z)\bigr)
=(1-4z)^{-5}.
\]

最后，对任意动力系统都成立
\[
\#\operatorname{Fix}(\sigma^n)
=
\sum_{d\mid n} d\,\operatorname{Prim}_d(\sigma).
\]
对该除数和作 Möbius 反演即得
\[
\operatorname{Prim}_n(\sigma)
=
\frac{1}{n}\sum_{d\mid n}\mu(d)\,\#\operatorname{Fix}(\sigma^{n/d})
=
\frac{5}{n}\sum_{d\mid n}\mu(d)\,4^{n/d}.
\]
证毕。
\end{proof}

\begin{corollary}[Lee--Yang 地址系统的 Bowen--Franks 群与 Cuntz--Krieger 代数]\label{cor:derived-leyang-bf-ck-rigidity}
\leanverified{paper\_derived\_leyang\_bf\_ck\_rigidity}
在定理 \ref{thm:derived-leyang-artin-mazur-zeta} 的设定下，Lee--Yang 地址系统可由块对角邻接矩阵
\[
A_{\mathrm{LY}}:=\operatorname{diag}(J_4,J_4,J_4,J_4,J_4)
\]
表示，其中 \(J_4\) 为 \(4\times 4\) 全 \(1\) 矩阵。于是
\[
\mathrm{BF}(A_{\mathrm{LY}})
=
\operatorname{coker}(I-A_{\mathrm{LY}})
\cong
(\ZZ/3\ZZ)^5,
\qquad
\ker(I-A_{\mathrm{LY}})=0.
\]
并且其 Cuntz--Krieger 代数满足
\[
\mathcal O_{A_{\mathrm{LY}}}\cong \mathcal O_4^{\oplus 5}.
\]
\end{corollary}

\begin{proof}
由定理 \ref{thm:derived-leyang-profinite-rigidity} 与定理 \ref{thm:derived-leyang-artin-mazur-zeta}，每个分支都是满四元移位，故单分支邻接矩阵为 \(J_4\)，整体邻接矩阵即为五块 \(J_4\) 的直和。

对单块而言，
\[
I_4-J_4=
\begin{pmatrix}
0&-1&-1&-1\\
-1&0&-1&-1\\
-1&-1&0&-1\\
-1&-1&-1&0
\end{pmatrix}
\]
经整数初等行列变换化为 Smith 标准形
\[
\operatorname{SNF}(I_4-J_4)=\operatorname{diag}(1,1,1,3).
\]
因此
\[
\operatorname{coker}(I_4-J_4)\cong \ZZ/3\ZZ,
\qquad
\ker(I_4-J_4)=0.
\]
块对角直和将 kernel 与 cokernel 逐块直和，于是
\[
\mathrm{BF}(A_{\mathrm{LY}})
\cong
\bigl(\ZZ/3\ZZ\bigr)^5,
\qquad
\ker(I-A_{\mathrm{LY}})=0.
\]

最后，由块对角矩阵对应图的不交并，Cuntz--Krieger 代数分解为对应分支代数的直和；而单个满四元移位的 Cuntz--Krieger 代数即 \(\mathcal O_4\)。故
\[
\mathcal O_{A_{\mathrm{LY}}}\cong \mathcal O_4^{\oplus 5}.
\]
证毕。
\end{proof}

\subsection{冻结相 escort 分布的微正则化与全阶 Rényi 塌缩}

固定冻结相中的最大纤维集合不仅承载质量集中，而且在指数精度上已经足以强制 escort 分布退化为该集合上的微正则分布。由此还可得到一条随 Rényi 阶变化的阈值曲面。

\begin{theorem}[冻结相 escort 分布的指数微正则化]\label{thm:derived-fixed-freezing-microcanonical-tv}
\leanverified{paper\_derived\_fixed\_freezing\_microcanonical\_tv}
沿用结论 \ref{con:xi-terminal-fixed-freezing-escort-oracle-sharp-threshold} 与推论 \ref{cor:xi-terminal-fixed-order-escort-maxfiber-concentration} 的记号。对整数 \(a\ge 1\) 且 \(a>a_{\mathrm c}\) 定义
\[
A_m^{\max}:=\{x\in X_m:\ d_m(x)=M_m\},
\qquad
K_m:=|A_m^{\max}|,
\qquad
\pi_{m,a}(x):=\frac{d_m(x)^a}{S_a(m)}.
\]
令 \(u_m\) 为 \(A_m^{\max}\) 上的均匀分布（视为 \(X_m\) 上的概率测度）。则存在
\[
\varepsilon_m:=1-\pi_{m,a}(A_m^{\max})
\le
\exp\!\bigl(-m\Delta(a)+o(m)\bigr)
\]
使得
\[
\|\pi_{m,a}-u_m\|_{\mathrm{TV}}=\varepsilon_m.
\]
特别地，
\[
\|\pi_{m,a}-u_m\|_{\mathrm{TV}}
\le
\exp\!\bigl(-m\Delta(a)+o(m)\bigr).
\]
从而对任意有界可观测量 \(f:X_m\to\RR\) 都有
\[
\bigl|\EE_{\pi_{m,a}}f-\EE_{u_m}f\bigr|
\le
2\|f\|_{\infty}\,
\exp\!\bigl(-m\Delta(a)+o(m)\bigr).
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:xi-terminal-fixed-order-escort-maxfiber-concentration}，
\[
\varepsilon_m
:=
1-\pi_{m,a}(A_m^{\max})
\le
\exp\!\bigl(-m\Delta(a)+o(m)\bigr).
\]
而对任意 \(x\in A_m^{\max}\) 都有 \(d_m(x)=M_m\)，故 \(\pi_{m,a}\) 在 \(A_m^{\max}\) 上严格均匀：
\[
\pi_{m,a}(x)=\frac{M_m^a}{S_a(m)}=\frac{1-\varepsilon_m}{K_m},
\qquad
u_m(x)=\frac{1}{K_m}
\qquad
(x\in A_m^{\max}).
\]
因此
\[
\begin{aligned}
\|\pi_{m,a}-u_m\|_{\mathrm{TV}}
&=
\frac12\sum_{x\in A_m^{\max}}
\left|
\frac{1-\varepsilon_m}{K_m}-\frac{1}{K_m}
\right|
\;+\;
\frac12\sum_{x\notin A_m^{\max}}\pi_{m,a}(x) \\
&=
\frac12\sum_{x\in A_m^{\max}}\frac{\varepsilon_m}{K_m}
\;+\;
\frac12\sum_{x\notin A_m^{\max}}\pi_{m,a}(x)
=
\varepsilon_m.
\end{aligned}
\]
上界随即成立。最后一式由总变差对有界函数积分差的标准控制不等式给出。
\end{proof}

\begin{theorem}[冻结相之上的全阶 Rényi 熵塌缩面]\label{thm:derived-fixed-freezing-renyi-surface}
\leanverified{paper\_derived\_fixed\_freezing\_renyi\_surface}
仍在定理 \ref{thm:derived-fixed-freezing-microcanonical-tv} 的设定下，对 \(r\in(0,\infty)\setminus\{1\}\) 定义
\[
H_r(\pi):=\frac{1}{1-r}\log\sum_x \pi(x)^r,
\]
对 \(r=1\) 取 Shannon 熵 \(H_1(\pi):=-\sum_x\pi(x)\log\pi(x)\)，对 \(r=\infty\) 取
\(
H_{\infty}(\pi):=-\log\max_x\pi(x)
\)。
再定义阈值
\[
a_r:=
\begin{cases}
a_{\mathrm c}/r, & 0<r<1,\\[1mm]
a_{\mathrm c}, & 1\le r\le \infty.
\end{cases}
\]
则对任意 \(r\in(0,\infty]\) 与任意整数 \(a\ge 1\) 且 \(a>a_r\)，极限
\[
h_r(a):=\lim_{m\to\infty}\frac{1}{m}H_r(\pi_{m,a})
\]
存在，并满足
\[
h_r(a)=g_{\ast}.
\]
亦即，一旦 \(a\) 超过相应阈值，escort 分布的全部宏观 Rényi 熵率均塌缩到同一个最大层简并指数 \(g_{\ast}\)。
\end{theorem}

\begin{proof}
记
\[
\varepsilon_m:=1-\pi_{m,a}(A_m^{\max}).
\]
由定理 \ref{thm:derived-fixed-freezing-microcanonical-tv}，
\[
\varepsilon_m\le \exp\!\bigl(-m\Delta(a)+o(m)\bigr),
\qquad
\frac1m\log K_m\to g_{\ast},
\qquad
\frac1m\log|X_m|\to \log\varphi.
\]

先设 \(1<r<\infty\)。因为 \(\pi_{m,a}\) 在 \(A_m^{\max}\) 上均匀，
\[
\sum_x \pi_{m,a}(x)^r
\ge
K_m\Bigl(\frac{1-\varepsilon_m}{K_m}\Bigr)^r
=(1-\varepsilon_m)^r K_m^{1-r},
\]
从而
\[
H_r(\pi_{m,a})
\le
\log K_m-\frac{r}{r-1}\log(1-\varepsilon_m).
\]
另一方面 \(H_r\ge H_{\infty}\)，且
\[
H_{\infty}(\pi_{m,a})
=
-\log\frac{1-\varepsilon_m}{K_m}
=
\log K_m-\log(1-\varepsilon_m).
\]
由于 \(\varepsilon_m\to 0\) 指数成立，遂有
\[
\frac1m H_r(\pi_{m,a})\to g_{\ast}.
\]

对 \(r=\infty\)，上式已经给出
\[
\frac1m H_{\infty}(\pi_{m,a})\to g_{\ast}.
\]

对 \(r=1\)，令 \(\widetilde\pi_{m,a}\) 为 \(\pi_{m,a}\) 在 \(X_m\setminus A_m^{\max}\) 上的条件分布，则熵分解给出
\[
H_1(\pi_{m,a})
=
h(\varepsilon_m)+(1-\varepsilon_m)\log K_m+\varepsilon_m H_1(\widetilde\pi_{m,a}),
\]
其中 \(h(\cdot)\) 为二元熵函数。又
\[
H_1(\widetilde\pi_{m,a})\le \log|X_m|.
\]
由于 \(\varepsilon_m\) 指数趋零，故
\[
\frac1m H_1(\pi_{m,a})
=
\frac1m\log K_m+o(1)\to g_{\ast}.
\]

最后设 \(0<r<1\)。由 concavity 与总质量分解，
\[
\sum_x \pi_{m,a}(x)^r
\le
K_m^{1-r}(1-\varepsilon_m)^r
+
|X_m|^{1-r}\varepsilon_m^r,
\]
同时仍有下界
\[
\sum_x \pi_{m,a}(x)^r
\ge
K_m^{1-r}(1-\varepsilon_m)^r.
\]
故
\[
H_r(\pi_{m,a})
=
\log K_m
+
\frac{r}{1-r}\log(1-\varepsilon_m)
+
\frac{1}{1-r}\log(1+\Xi_m),
\]
其中
\[
\Xi_m:=
K_m^{r-1}|X_m|^{1-r}
\Bigl(\frac{\varepsilon_m}{1-\varepsilon_m}\Bigr)^r.
\]
取 \(\frac1m\log\) 并使用上述三个指数，得到
\[
\frac1m\log \Xi_m
\le
(1-r)(\log\varphi-g_{\ast})-r\Delta(a)+o(1).
\]
而
\[
\Delta(a)=a(\alpha_{\ast}-\alpha_{\ast}^{(2)})-(\log\varphi-g_{\ast}),
\]
故若 \(a>a_{\mathrm c}/r\)，则
\[
r\Delta(a)
=
ra(\alpha_{\ast}-\alpha_{\ast}^{(2)})-r(\log\varphi-g_{\ast})
>
(1-r)(\log\varphi-g_{\ast}),
\]
从而 \(\Xi_m\to 0\) 指数成立。于是
\[
\frac1m H_r(\pi_{m,a})\to g_{\ast}.
\]
综上，对全部 \(r\in(0,\infty]\) 都有 \(h_r(a)=g_{\ast}\)。
\end{proof}

\subsection{审计偶窗上的 gauge 群交换化刚性}

除 \(m=6\) 的显式直方图外，现有审计链实际上已经足以给出整个偶窗族的交换化刚性。

\begin{theorem}[已审计偶窗上的 bin-fold gauge 群交换化刚性]\label{thm:derived-foldbin-gauge-abelianization-even-windows}
\leanverified{paper\_derived\_foldbin\_gauge\_abelianization\_even\_windows}
在二进制区间折叠口径下，记
\[
\Gamma_m:=\mathcal G_m^{\mathrm{bin}}
=
\operatorname{Aut}\!\bigl(\mathrm{Fold}^{\mathrm{bin}}_m\bigr).
\]
则对审计偶数窗口
\[
m\in\{6,8,10,12\},
\]
其交换化满足
\[
\Gamma_m^{\mathrm{ab}}
\cong
(\ZZ/2\ZZ)^{F_{m+2}}.
\]
特别地，任何取值于二元立方体的忠实异常签名
\[
\Gamma_m^{\mathrm{ab}}\hookrightarrow (\ZZ/2\ZZ)^B
\]
都必满足
\[
B\ge F_{m+2}.
\]
\end{theorem}

\begin{proof}
由命题 \ref{prop:fold-bin-gauge-decomposition}，
\[
\Gamma_m\cong \prod_{w\in X_m} S_{d_m^{\mathrm{bin}}(w)}.
\]
而定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 在审计偶窗上给出
\[
d_{\min}(m)=F_{m/2}\ge 2.
\]
因此对每个 \(w\in X_m\) 都有 \(d_m^{\mathrm{bin}}(w)\ge 2\)，从而
\[
S_{d_m^{\mathrm{bin}}(w)}^{\mathrm{ab}}\cong \ZZ/2\ZZ.
\]
于是
\[
\Gamma_m^{\mathrm{ab}}
\cong
\bigoplus_{w\in X_m}\bigl(S_{d_m^{\mathrm{bin}}(w)}^{\mathrm{ab}}\bigr)
\cong
\bigoplus_{w\in X_m}\ZZ/2\ZZ
\cong
(\ZZ/2\ZZ)^{|X_m|}.
\]
最后由稳定类型计数律 \(|X_m|=F_{m+2}\) 得
\[
\Gamma_m^{\mathrm{ab}}\cong (\ZZ/2\ZZ)^{F_{m+2}}.
\]

若存在忠实异常签名 \(\Gamma_m^{\mathrm{ab}}\hookrightarrow (\ZZ/2\ZZ)^B\)，则它是 \(\FF_2\)-向量空间的单射同态；比较维数即得 \(B\ge F_{m+2}\)。
\end{proof}
