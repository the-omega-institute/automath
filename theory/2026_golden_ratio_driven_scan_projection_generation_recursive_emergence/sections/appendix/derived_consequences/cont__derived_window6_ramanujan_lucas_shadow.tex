\subsection{六倍窗半阶见证块的 Ramanujan 阴影与 Lucas 半因子恢复}

附录 \ref{app:fold-multiplicity} 已给出零点余类的刚性表示
\(
S_g(M)
\)
以及六倍窗上的半阶大陪集零点块
\(
S_{F_d}(F_{2d})\subseteq \mathcal Z_m
\)；
正文又已证明：六倍窗零点相对密度恰处在 \(\varphi^{-m/2}\) 尺度，而缩放碰撞量
\(
F_{m+2}\operatorname{Col}_m
\)
仍趋于 \(+\infty\)，并且其边界对象由整条共振梯 \(u\mapsto C_\varphi(u)\) 驱动。把这些接口与 Ramanujan 坐标重新并置后，还可得到下列几条更细的派生后果。

\begin{theorem}[单陪集 Ramanujan 阴影的精确闭式]\label{thm:derived-single-coset-ramanujan-shadow-closed-form}
\leanverified{paper\_derived\_single\_coset\_ramanujan\_shadow\_closed\_form}
设 \(M\) 为偶数，\(g\mid M/2\)，并记
\[
S_g(M):=
\left\{
k\in \ZZ/(M\ZZ):\
k\equiv \frac{M}{2g}\pmod{M/g}
\right\}.
\]
对每个 \(q\mid M\)，定义其 Ramanujan 阴影
\[
\Sigma_{g,M}(q):=\sum_{k\in S_g(M)}c_q(k),
\qquad
c_q(n):=\sum_{(a,q)=1}e^{2\pi i an/q}.
\]
则有精确公式
\[
\boxed{\;
\Sigma_{g,M}(q)
=
g\,c_q\!\left(\frac{M}{2g}\right)\mathbf 1_{\{q\mid M/g\}}.\;}
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:fold-zero-coset-rigidity} 的显式表示，
\[
S_g(M)=
\left\{
\frac{M}{2g}+t\frac{M}{g}:\ 0\le t\le g-1
\right\}\subset \ZZ/(M\ZZ).
\]
于是
\[
\Sigma_{g,M}(q)
=
\sum_{(a,q)=1}
\sum_{t=0}^{g-1}
e^{2\pi i a(M/(2g)+tM/g)/q}.
\]
交换求和次序得
\[
\Sigma_{g,M}(q)
=
\sum_{(a,q)=1}
e^{2\pi i aM/(2gq)}
\sum_{t=0}^{g-1}e^{2\pi i atM/(gq)}.
\]
内层是长度 \(g\) 的等比和，其公比为
\(
\exp(2\pi i aM/(gq))
\)。
由于 \(q\mid M\)，其 \(g\)-次幂恒为 \(1\)。
若 \(q\nmid M/g\)，则该公比不等于 \(1\)，故内层和为 \(0\)；
若 \(q\mid M/g\)，则公比恰为 \(1\)，从而内层和等于 \(g\)。
因此
\[
\Sigma_{g,M}(q)
=
g\,\mathbf 1_{\{q\mid M/g\}}
\sum_{(a,q)=1}e^{2\pi i aM/(2gq)}
=
g\,c_q\!\left(\frac{M}{2g}\right)\mathbf 1_{\{q\mid M/g\}}.
\]
证毕。
\end{proof}

\begin{theorem}[六倍窗半阶见证块的 Lucas 半因子 Ramanujan 因子化]\label{thm:derived-window6-halfblock-lucas-ramanujan-factorization}
设
\[
m+2=2d\equiv 0\pmod 6,
\qquad
M:=F_{m+2}=F_{2d}=F_dL_d,
\]
并记六倍窗半阶见证块
\[
W_d:=S_{F_d}(F_{2d})\subseteq \mathcal Z_m.
\]
对每个正整数 \(q\)，定义
\[
\Sigma_d(q):=\sum_{k\in W_d}c_q(k).
\]
则对每个 \(q\mid F_{2d}\) 都有
\[
\boxed{\;
\Sigma_d(q)
=
F_d\,c_q\!\left(\frac{L_d}{2}\right)\mathbf 1_{\{q\mid L_d\}},\;}
\]
因而其归一化阴影满足
\[
\boxed{\;
\frac{\Sigma_d(q)}{F_{2d}}
=
\frac{c_q(L_d/2)}{L_d}\,\mathbf 1_{\{q\mid L_d\}}.\;}
\]
\end{theorem}

\begin{proof}
由推论 \ref{cor:fold-zero-half-index-multiple6}，
\[
W_d=S_{F_d}(F_{2d})\subseteq \mathcal Z_m,
\qquad
|W_d|=F_d.
\]
将定理 \ref{thm:derived-single-coset-ramanujan-shadow-closed-form}
应用于 \(g=F_d\)、\(M=F_{2d}\)，得到
\[
\Sigma_d(q)
=
F_d\,
c_q\!\left(\frac{F_{2d}}{2F_d}\right)
\mathbf 1_{\{q\mid F_{2d}/F_d\}}.
\]
再用恒等式 \(F_{2d}=F_dL_d\)，并注意 \(d\equiv 0\pmod 3\) 时 Lucas 数 \(L_d\) 为偶数（见推论 \ref{cor:fold-zero-half-index-multiple6} 的证明），即可改写为
\[
\frac{F_{2d}}{2F_d}=\frac{L_d}{2},
\qquad
\frac{F_{2d}}{F_d}=L_d,
\]
从而得到所示两式。证毕。
\end{proof}
\leanverified{paper\_derived\_window6\_halfblock\_lucas\_ramanujan\_factorization}

\begin{corollary}[半阶见证块完整恢复 \texorpdfstring{$L_d/2$}{Ld/2} 的除数格]\label{cor:derived-window6-halfblock-recovers-lucas-divisor-lattice}
在定理 \ref{thm:derived-window6-halfblock-lucas-ramanujan-factorization} 的设定下，对任意 \(r\mid L_d\) 都有
\[
\boxed{\;
\frac{1}{rF_d}\sum_{q\mid r}\Sigma_d(q)
=
\mathbf 1_{\{r\mid L_d/2\}}.\;}
\]
特别地，\(W_d\) 的 Ramanujan 阴影唯一决定 \(L_d/2\) 的全部除数格；其除数 Dirichlet 级数可写为
\[
\boxed{\;
\sum_{r\mid L_d/2}r^{-s}
=
\frac{1}{F_d}\sum_{r\mid L_d}\frac{1}{r^{s+1}}
\sum_{q\mid r}\Sigma_d(q)
\qquad (\Re s>0).\;}
\]
\end{corollary}

\begin{proof}
由定理 \ref{thm:derived-window6-halfblock-lucas-ramanujan-factorization}，
\[
\sum_{q\mid r}\Sigma_d(q)
=
F_d\sum_{q\mid r}c_q(L_d/2).
\]
再用 Ramanujan 和的经典除数恒等式
\[
\sum_{q\mid r}c_q(n)=r\,\mathbf 1_{\{r\mid n\}},
\]
取 \(n=L_d/2\) 即得第一式。
将其乘以 \(r^{-s}\) 并对 \(r\mid L_d\) 求和，即得到第二式。证毕。
\end{proof}
\leanverified{paper\_derived\_window6\_halfblock\_recovers\_lucas\_divisor\_lattice}

\begin{theorem}[固定导子对半阶见证块的指数级失明]\label{thm:derived-window6-halfblock-fixed-conductor-blindness}
固定有限导子集 \(Q\subset \NN\)。沿六倍窗子列 \(m+2=2d\equiv 0\pmod 6\)，有
\[
\boxed{\;
\max_{q\in Q}
\left|
\frac{\Sigma_d(q)}{F_{2d}}
\right|
\le
\frac{\max_{q\in Q}\varphi(q)}{L_d}
=
O_Q(\varphi^{-m/2}).\;}
\]
\end{theorem}

\begin{proof}
对任意正整数 \(q\)，由标准界 \(|c_q(n)|\le \varphi(q)\) 与 \(|W_d|=F_d\) 得
\[
\left|
\frac{\Sigma_d(q)}{F_{2d}}
\right|
\le
\frac{\varphi(q)}{L_d},
\]
其中用到了 \(F_{2d}=F_dL_d\)。
对 \(q\in Q\) 取最大值得到
\[
\max_{q\in Q}
\left|
\frac{\Sigma_d(q)}{F_{2d}}
\right|
\le
\frac{\max_{q\in Q}\varphi(q)}{L_d}.
\]
最后由 Binet 公式
\[
L_d=\varphi^d+\psi^d
\qquad
(\psi=-\varphi^{-1})
\]
可得
\[
\frac1{L_d}
=
\varphi^{-d}+O(\varphi^{-3d})
=
O(\varphi^{-m/2}),
\]
而 \(d=(m+2)/2\)，故结论成立。证毕。
\end{proof}
\leanverified{paper\_derived\_window6\_halfblock\_fixed\_conductor\_blindness}

\begin{theorem}[有限固定导子 Ramanujan 观测不可能承载 bulk 碰撞尺度]\label{thm:derived-window6-fixed-conductor-ramanujan-nogo-bulk-collision}
固定有限集合 \(Q\subset \NN\)，并沿六倍窗子列 \(m+2=2d\equiv 0\pmod 6\) 定义
\[
z_{m,q}:=\frac{\Sigma_d(q)}{F_{m+2}}
\qquad (q\in Q).
\]
若
\[
P\in \CC[(z_q)_{q\in Q}]
\]
是不依赖于 \(m\) 且常数项为 \(0\) 的多项式，则有
\[
\boxed{\;
P\bigl((z_{m,q})_{q\in Q}\bigr)=O_{P,Q}(\varphi^{-m/2}).\;}
\]
另一方面，
\[
F_{m+2}\operatorname{Col}_m\longrightarrow +\infty.
\]
因此，不存在任何有限固定导子 Ramanujan 见证块观测的多项式组合，能够恢复 bulk 碰撞尺度 \(F_{m+2}\operatorname{Col}_m\)。
\end{theorem}

\begin{proof}
由定理 \ref{thm:derived-window6-halfblock-fixed-conductor-blindness}，对每个 \(q\in Q\) 都有
\[
z_{m,q}=O_Q(\varphi^{-m/2}).
\]
由于 \(P\) 的常数项为 \(0\)，其每一单项式都至少含有一个变量 \(z_q\)，故当 \(m\to\infty\) 时
\[
P\bigl((z_{m,q})_{q\in Q}\bigr)=O_{P,Q}(\varphi^{-m/2}).
\]
另一方面，由定理 \ref{thm:conclusion-pisot-bernoulli-nonrajchman-collision-inflation}，
\[
F_{m+2}\operatorname{Col}_m\to +\infty.
\]
若后者可由某个此类多项式 \(P((z_{m,q})_{q\in Q})\) 渐近恢复，则左端趋于 \(+\infty\)，而右端趋于 \(0\)，矛盾。证毕。
\end{proof}
\leanverified{paper\_derived\_window6\_fixed\_conductor\_ramanujan\_nogo\_bulk\_collision}

\begin{remark}[从零点计数盲性到固定导子 Ramanujan 盲性]\label{rem:derived-window6-ramanujan-shadow-strengthens-zero-count-blindness}
推论 \ref{cor:conclusion-fold-zero-count-cannot-control-gap} 已排除了“零点计数本身”对缩放碰撞缺口的渐近控制。
定理 \ref{thm:derived-window6-fixed-conductor-ramanujan-nogo-bulk-collision} 进一步表明：即使把六倍窗的 canonical 半阶见证块投到任意有限个固定导子的 Ramanujan 坐标上，所得到的全部读数仍然只有 \(\varphi^{-m/2}\) 量级，因而同样不可能承载真实的 bulk 碰撞尺度。
换言之，零点几何在固定导子 Ramanujan 层上的全部可见部分，依然只构成振幅主项之外的半尺度余项。
\end{remark}

\begin{conjecture}[六倍窗碰撞尺度的振幅主项与 Ramanujan 零几何余项分裂]\label{conj:derived-window6-collision-amplitude-ramanujan-remainder-splitting}
沿六倍窗子列 \(m+2=2d\equiv 0\pmod 6\)，应存在分解
\[
\operatorname{Col}_m
=
\mathcal A_m^{\mathrm{amp}}
+\mathcal R_m^{\mathrm{Ram}},
\]
满足：
\begin{enumerate}
\normalfont
  \item \(\mathcal A_m^{\mathrm{amp}}\) 仅由振幅律
  \[
  \frac{|\widehat d_m(k)|^2}{2^{2m}}
  =
  \prod_{j=1}^{m}\cos^2\!\left(\pi kF_{j+1}/F_{m+2}\right)
  \]
  所决定，并承载全部发散主项
  \[
  F_{m+2}\mathcal A_m^{\mathrm{amp}}\to +\infty;
  \]
  \item \(\mathcal R_m^{\mathrm{Ram}}\) 仅由零点余类并
  \[
  \mathcal Z_m=\bigcup_{g}S_g(F_{m+2})
  \]
  的 Ramanujan 阴影决定，并满足尖锐余项界
  \[
  \mathcal R_m^{\mathrm{Ram}}=O(L_d^{-1})=O(\varphi^{-m/2});
  \]
  \item 六倍窗的半阶见证块
  \[
  W_d=S_{F_d}(F_{2d})
  \]
  已经达到该余项的正确指数尺度，即
  \[
  \mathcal R_m^{\mathrm{Ram}}\not=o(L_d^{-1}).
  \]
\end{enumerate}
若该猜想成立，则定理 \ref{thm:conclusion-zero-reduction-second-order-collision-kl} 关于“零点削减仅为二阶校正”的现象，将可被提升为一条严格的主项--余项分裂；而定理 \ref{thm:conclusion-pisot-boundary-shadow-not-l2} 所揭示的非 \(\ell^2\) 共振梯边界，也正是该主项无法压缩回有限固定导子 Ramanujan 读出的解析原因。
\end{conjecture}
