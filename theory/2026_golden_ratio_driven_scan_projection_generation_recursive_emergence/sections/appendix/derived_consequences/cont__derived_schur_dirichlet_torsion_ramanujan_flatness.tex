\subsection{Schur primitive torsion 表的 Ramanujan 迹、Dirichlet 能量与 cyclotomic 刚性}

正文在 \S\ref{subsubsec:pom-schur-dirichlet-torsion-factorization} 已经建立了三件彼此相邻但尚未在同一处封口的事实：其一，Schur 通道的角色扭点泛函 \(\mathcal L_{q,\lambda}(\chi)\) 与 primitive torsion 表在 Dirichlet 角色变换下互为精确反演；其二，在次数界 \(\deg F_{q,\lambda}\le \varphi(q)\) 下，有限 torsion 表已足以恢复归一化谱包络与常数异常；其三，通道 prime-shadow 的平方根门槛由这些有限 torsion 数据唯一确定。把这三组接口并读后，可得到如下 Ramanujan 迹、能量方差与 torsion-flat 刚性的闭链。

\begin{theorem}[主角色的 Ramanujan 迹与跨通道 Parseval--Gram 恒等式]\label{thm:derived-schur-dirichlet-ramanujan-parseval}
\leanverified{paper\_derived\_schur\_dirichlet\_ramanujan\_parseval}
固定整数 \(q\ge 2\)，并沿用定义 \ref{def:pom-schur-dirichlet-torsion-functional} 与定理 \ref{thm:pom-schur-dirichlet-factorization} 的记号。对每个 \(\lambda\vdash q\) 定义
\[
f_{q,\lambda}(a):=-\log F_{q,\lambda}(\omega_q^{\,a})
\qquad (a\in U(q)),
\]
并记主角色为 \(\chi_0\)。则对任意两条 Schur 通道 \(\lambda,\mu\vdash q\) 都有：
\begin{enumerate}
\normalfont
  \item 主角色值给出真正的 Ramanujan 迹
  \[
  \mathcal L_{q,\lambda}(\chi_0)
  =
  \sum_{n\ge 1}\frac{u_{n,\lambda}}{n}\,c_q(n),
  \qquad
  c_q(n):=\sum_{a\in U(q)}\omega_q^{\,an};
  \]
  \item 全角色满足跨通道 Parseval--Gram 恒等式
  \[
  \sum_{\chi\in\widehat{U(q)}}\mathcal L_{q,\lambda}(\chi)\,
  \overline{\mathcal L_{q,\mu}(\chi)}
  =
  \varphi(q)\sum_{a\in U(q)}f_{q,\lambda}(a)\,\overline{f_{q,\mu}(a)}.
  \]
\end{enumerate}
特别地，
\[
\sum_{\chi\in\widehat{U(q)}}|\mathcal L_{q,\lambda}(\chi)|^2
=
\varphi(q)\sum_{a\in U(q)}|f_{q,\lambda}(a)|^2.
\]
因此，Schur 通道的 primitive torsion 表、Dirichlet 角色能量与 Ramanujan 主迹不再是平行数据，而是同一有限 Fourier 几何的三张坐标图。
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-schur-dirichlet-factorization}，
\[
\mathcal L_{q,\lambda}(\chi)
=
\sum_{n\ge 1}\frac{u_{n,\lambda}}{n}\,G(\chi;n),
\qquad
G(\chi;n):=\sum_{a\in U(q)}\chi(a)\omega_q^{\,an}.
\]
取 \(\chi=\chi_0\) 时，
\[
G(\chi_0;n)=\sum_{a\in U(q)}\omega_q^{\,an}=c_q(n),
\]
从而得到第一式。

另一方面，由定义
\[
\mathcal L_{q,\lambda}(\chi)=\sum_{a\in U(q)}\chi(a)f_{q,\lambda}(a),
\qquad
\mathcal L_{q,\mu}(\chi)=\sum_{b\in U(q)}\chi(b)f_{q,\mu}(b).
\]
于是
\[
\sum_{\chi}\mathcal L_{q,\lambda}(\chi)\overline{\mathcal L_{q,\mu}(\chi)}
=
\sum_{a,b}f_{q,\lambda}(a)\overline{f_{q,\mu}(b)}
\sum_{\chi}\chi(a)\overline{\chi(b)}.
\]
由有限阿贝尔群 \(U(q)\) 上的角色正交性，
\[
\sum_{\chi\in\widehat{U(q)}}\chi(a)\overline{\chi(b)}
=
\varphi(q)\,\mathbf 1_{\{a=b\}},
\]
故得到 Parseval--Gram 恒等式。令 \(\mu=\lambda\) 即得平方范数版本。证毕。
\end{proof}

\begin{corollary}[非主角色能量恰等于 primitive torsion 表的中心化方差]\label{cor:derived-schur-dirichlet-nonprincipal-energy-variance}
\leanverified{paper\_derived\_schur\_dirichlet\_nonprincipal\_energy\_variance}
在定理 \ref{thm:derived-schur-dirichlet-ramanujan-parseval} 的设定下，定义中心化 torsion 振幅
\[
f_{q,\lambda}^{\circ}(a)
:=
f_{q,\lambda}(a)-\frac{1}{\varphi(q)}\mathcal L_{q,\lambda}(\chi_0).
\]
则有精确恒等式
\[
\sum_{a\in U(q)}|f_{q,\lambda}^{\circ}(a)|^2
=
\frac{1}{\varphi(q)}
\sum_{\chi\neq\chi_0}|\mathcal L_{q,\lambda}(\chi)|^2.
\]
并且对任意 \(a\in U(q)\) 都有
\[
\left|f_{q,\lambda}(a)-\frac{1}{\varphi(q)}\mathcal L_{q,\lambda}(\chi_0)\right|
\le
\frac{1}{\sqrt{\varphi(q)}}
\left(\sum_{\chi\neq\chi_0}|\mathcal L_{q,\lambda}(\chi)|^2\right)^{1/2}.
\]
因此，全部非主 Dirichlet 通道的总能量，正好等于 primitive torsion 表相对于其 Ramanujan 主均值的中心化方差。
\end{corollary}

\begin{proof}
由定理 \ref{thm:derived-schur-dirichlet-ramanujan-parseval}，
\[
\sum_{\chi}|\mathcal L_{q,\lambda}(\chi)|^2
=
\varphi(q)\sum_{a}|f_{q,\lambda}(a)|^2.
\]
把 \(f_{q,\lambda}=f_{q,\lambda}^{\circ}+\varphi(q)^{-1}\mathcal L_{q,\lambda}(\chi_0)\) 代入右端，并注意
\[
\sum_{a\in U(q)}f_{q,\lambda}^{\circ}(a)=0,
\]
便得
\[
\varphi(q)\sum_a |f_{q,\lambda}(a)|^2
=
|\mathcal L_{q,\lambda}(\chi_0)|^2
+
\varphi(q)\sum_a|f_{q,\lambda}^{\circ}(a)|^2.
\]
与左端按主/非主角色拆分比较，即得第一式。

再由定理 \ref{thm:pom-schur-dirichlet-factorization} 的反演公式，
\[
f_{q,\lambda}^{\circ}(a)
=
\frac{1}{\varphi(q)}\sum_{\chi\neq\chi_0}\overline{\chi}(a)\,\mathcal L_{q,\lambda}(\chi).
\]
对右端施 Cauchy--Schwarz，并用 \(|\chi(a)|=1\) 与
\(
\#(\widehat{U(q)}\setminus\{\chi_0\})=\varphi(q)-1<\varphi(q)
\)
即可得到上界。证毕。
\end{proof}

\begin{theorem}[torsion-flat 通道的 cyclotomic pencil 刚性]\label{thm:derived-schur-dirichlet-torsion-flat-cyclotomic-pencil}
\leanverified{paper\_derived\_schur\_dirichlet\_torsion\_flat\_cyclotomic\_pencil}
在定理 \ref{thm:pom-schur-dirichlet-torsion-triple-interface} 的次数界
\[
\deg F_{q,\lambda}\le \varphi(q)
\]
下，以下三条对任意固定通道 \(\lambda\vdash q\) 等价：
\begin{enumerate}
\normalfont
  \item 所有非主角色扭点泛函都消失：
  \[
  \mathcal L_{q,\lambda}(\chi)=0
  \qquad (\forall\,\chi\neq\chi_0);
  \]
  \item primitive torsion 表在 \(U(q)\) 上常值：
  \[
  F_{q,\lambda}(\omega_q^{\,a})\ \text{与}\ a\in U(q)\ \text{无关};
  \]
  \item 存在唯一常数 \(C_{q,\lambda}\in\CC^\times\) 使
  \[
  F_{q,\lambda}(t)
  =
  C_{q,\lambda}
  +
  \bigl(1-C_{q,\lambda}\bigr)\Phi_q^{\mathrm{cyc}}(t),
  \]
  其中 \(\Phi_q^{\mathrm{cyc}}\) 是第 \(q\) 个 cyclotomic 多项式。
\end{enumerate}
因此，一旦 Schur 通道在全部非主 Dirichlet 方向上消失，它就不再只是“在 primitive roots 上偶然平坦”的多项式，而被强迫坍缩为唯一的 cyclotomic pencil。
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-schur-dirichlet-factorization} 的反演公式，
\[
f_{q,\lambda}(a)
=
\frac{1}{\varphi(q)}
\sum_{\chi\in\widehat{U(q)}}\overline{\chi}(a)\,\mathcal L_{q,\lambda}(\chi).
\]
因此，所有非主角色系数同时为零，当且仅当 \(f_{q,\lambda}(a)\) 在 \(U(q)\) 上为常数；又因为指数函数在此使用正文 remark \ref{rem:pom-schur-dirichlet-log-branch-normalization} 所固定的规范对数分支，故这又等价于 \(F_{q,\lambda}(\omega_q^{\,a})\) 在 \(U(q)\) 上为常数。这证明了 (1)\(\Leftrightarrow\)(2)。

现设 (2) 成立，并记该常值为 \(C_{q,\lambda}\)。则多项式
\[
F_{q,\lambda}(t)-C_{q,\lambda}
\]
在全部 primitive \(q\)-次单位根上为零。由于 \(\Phi_q^{\mathrm{cyc}}(t)\) 正是这些点的最小多项式，且
\[
\deg(F_{q,\lambda}-C_{q,\lambda})\le \deg F_{q,\lambda}\le \varphi(q)=\deg \Phi_q^{\mathrm{cyc}},
\]
故必有
\[
F_{q,\lambda}(t)-C_{q,\lambda}=\alpha\,\Phi_q^{\mathrm{cyc}}(t)
\]
其中 \(\alpha\) 为常数。再由 \(F_{q,\lambda}(0)=1\) 与 \(\Phi_q^{\mathrm{cyc}}(0)=1\)，得到
\[
1-C_{q,\lambda}=\alpha,
\]
从而得到 (3)。

(3)\(\Rightarrow\)(2) 则是直接的，因为对任意 \(a\in U(q)\) 都有 \(\Phi_q^{\mathrm{cyc}}(\omega_q^{\,a})=0\)，故
\[
F_{q,\lambda}(\omega_q^{\,a})=C_{q,\lambda}.
\]
证毕。
\end{proof}

\begin{corollary}[prime-power 导子记住常数异常，非 prime-power 导子则完全失忆]\label{cor:derived-schur-dirichlet-primepower-memory-split}
\leanverified{paper\_derived\_schur\_dirichlet\_primepower\_memory\_split}
在定理 \ref{thm:derived-schur-dirichlet-torsion-flat-cyclotomic-pencil} 的等价条件下，常数 \(C_{q,\lambda}\) 与主角色值、常数异常之间满足
\[
\mathcal L_{q,\lambda}(\chi_0)
=
-\varphi(q)\log C_{q,\lambda},
\]
以及
\[
\mathcal M_{q,\lambda}^{-1}
=
F_{q,\lambda}(1)
=
C_{q,\lambda}+\bigl(1-C_{q,\lambda}\bigr)\Phi_q^{\mathrm{cyc}}(1).
\]
于是发生严格的 conductor 分岔：
\begin{enumerate}
\normalfont
  \item 若 \(q=p^r\) 为素数幂，则
  \[
  \mathcal M_{q,\lambda}^{-1}
  =
  p-(p-1)C_{q,\lambda};
  \]
  \item 若 \(q\) 含至少两个不同素因子，则
  \[
  \Phi_q^{\mathrm{cyc}}(1)=1,
  \qquad
  \mathcal M_{q,\lambda}=1.
  \]
\end{enumerate}
因此，在 torsion-flat 族上，prime-power 导子仍能记住常数异常；而一旦导子含至少两个不同素因子，常数异常层便完全失忆。
\end{corollary}

\begin{proof}
在 torsion-flat 条件下，由上一条定理可知
\[
F_{q,\lambda}(\omega_q^{\,a})=C_{q,\lambda}
\qquad (a\in U(q)).
\]
因此
\[
\mathcal L_{q,\lambda}(\chi_0)
=
-\sum_{a\in U(q)}\log F_{q,\lambda}(\omega_q^{\,a})
=
-\varphi(q)\log C_{q,\lambda}.
\]
又由定理 \ref{thm:pom-schur-dirichlet-torsion-triple-interface}，
\(
\mathcal M_{q,\lambda}=F_{q,\lambda}(1)^{-1}
\)，而上一条定理中的 pencil 表达式在 \(t=1\) 处给出
\[
F_{q,\lambda}(1)
=
C_{q,\lambda}+\bigl(1-C_{q,\lambda}\bigr)\Phi_q^{\mathrm{cyc}}(1).
\]
最后利用标准 cyclotomic 评价
\[
\Phi_q^{\mathrm{cyc}}(1)=
\begin{cases}
p,& q=p^r,\\
1,& q\ \text{含至少两个不同素因子},
\end{cases}
\]
即可得到两种情形。证毕。
\end{proof}

\begin{proposition}[primitive torsion 表的均匀平坦度由非主能量精确控制]\label{prop:derived-schur-dirichlet-uniform-flatness-bound}
\leanverified{paper\_derived\_schur\_dirichlet\_uniform\_flatness\_bound}
对任意 \(a,b\in U(q)\) 与任意通道 \(\lambda\vdash q\)，都有
\[
\bigl|f_{q,\lambda}(a)-f_{q,\lambda}(b)\bigr|
\le
\frac{2}{\sqrt{\varphi(q)}}
\left(\sum_{\chi\neq\chi_0}|\mathcal L_{q,\lambda}(\chi)|^2\right)^{1/2}.
\]
因此，若某列通道 \((\lambda_j)\) 满足
\[
\sum_{\chi\neq\chi_0}|\mathcal L_{q,\lambda_j}(\chi)|^2\longrightarrow 0,
\]
则它们的 primitive torsion 表 \(f_{q,\lambda_j}\) 在 \(U(q)\) 上一致趋于常数。
\end{proposition}

\begin{proof}
由推论 \ref{cor:derived-schur-dirichlet-nonprincipal-energy-variance}，
\[
\left|f_{q,\lambda}(a)-\frac{1}{\varphi(q)}\mathcal L_{q,\lambda}(\chi_0)\right|
\le
\frac{1}{\sqrt{\varphi(q)}}
\left(\sum_{\chi\neq\chi_0}|\mathcal L_{q,\lambda}(\chi)|^2\right)^{1/2}
\]
对每个 \(a\in U(q)\) 都成立。分别对 \(a\) 与 \(b\) 应用该估计，再用三角不等式，即得两点差上界。第二条随即推出。证毕。
\end{proof}

\begin{conjecture}[越过平方根门槛必然伴随 torsion-energy gap]\label{conj:derived-schur-dirichlet-overcritical-energy-gap}
固定 \(q\ge 2\)，并处在定理 \ref{thm:pom-schur-chebotarev-prime-deviation} 与定理 \ref{thm:pom-schur-near-rh-linear-inequality-complete} 的非退化制度下。对每个非平凡通道 \(\lambda\neq (q)\) 定义其 local near-RH 缺口
\[
\Delta_{q,\lambda}:=2\log\rho(B_{q,\lambda})-\log\rho_q.
\]
则应当存在函数
\[
\varepsilon_q:(0,\infty)\to(0,\infty)
\]
使得只要
\[
\Delta_{q,\lambda}\ge \delta>0,
\]
便必有
\[
\sum_{\chi\neq\chi_0}|\mathcal L_{q,\lambda}(\chi)|^2\ge \varepsilon_q(\delta).
\]
换言之，任何 genuinely overcritical 的 Schur prime-shadow 都不可能在 primitive torsion 表上趋于平坦；越过平方根门槛必须携带一个定量不可消去的非主 torsion 能量。
\end{conjecture}

\begin{remark}[统一链]\label{rem:derived-schur-dirichlet-torsion-unified-chain}
定理 \ref{thm:derived-schur-dirichlet-ramanujan-parseval}、推论 \ref{cor:derived-schur-dirichlet-nonprincipal-energy-variance}、定理 \ref{thm:derived-schur-dirichlet-torsion-flat-cyclotomic-pencil} 与推论 \ref{cor:derived-schur-dirichlet-primepower-memory-split} 共同表明：对固定 \(q\) 而言，
\[
\text{Ramanujan 主迹}
\Longrightarrow
\text{Dirichlet 能量分解}
\Longrightarrow
\text{primitive torsion 平坦度}
\Longrightarrow
\text{常数异常与 prime-shadow 门槛分岔}
\]
并不是一串松散类比，而是同一份 primitive torsion 数据在主/非主角色分解下的严格闭链。
\end{remark}
