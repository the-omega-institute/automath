\subsection{顶层双谱同坡、精确反演阈值与 Watatani 碰撞压力}

正文已经分别给出 Fold 顶端纤维谱的 Fibonacci 闭式、严格基态间隙下的通用冻结判据、全态精确反演阈值的一般公式，以及二阶碰撞矩的 Watatani 指标表示。
把这些接口在 regular Fold 上并读后，可进一步抽出一条更细的后果链：顶层双谱在指数尺度上同坡，从而严格单隙冻结机制在此失去适用性；而与之正交地，全态精确反演的 dyadic 阈值仍由最大纤维闭式精确控制，并把 Rényi--\(2\) 碰撞压力落实为一个 Watatani 指数可观测量的增长指数。

\begin{theorem}[顶层双谱的同坡性]\label{thm:derived-fold-top-two-same-slope}
\leanverified{paper\_derived\_fold\_top\_two\_same\_slope}
记
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
D_m^{(2)}:=\max\{d_m(x):d_m(x)<D_m\}.
\]
则最大纤维与次大纤维具有相同的指数斜率：
\[
\lim_{m\to\infty}\frac1m\log D_m
=
\lim_{m\to\infty}\frac1m\log D_m^{(2)}
=
\frac12\log\varphi.
\]
更精细地，
\[
\lim_{k\to\infty}\frac{D_{2k}^{(2)}}{D_{2k}}
=
1-\frac{1}{9+4\sqrt5}
=
1-\varphi^{-6},
\]
\[
\lim_{k\to\infty}\frac{D_{2k+1}^{(2)}}{D_{2k+1}}
=
1-\frac{1}{11+5\sqrt5}
=
1-\frac12\varphi^{-5}.
\]
因此，Fold 顶层纤维谱在最大层与次大层之间并不存在指数隔离，而只发生乘性常数级分裂。
\end{theorem}

\begin{proof}
由定理 \ref{thm:pom-max-fiber}，
\[
D_{2k}=F_{k+2},
\qquad
D_{2k+1}=2F_{k+1},
\]
而由定理 \ref{thm:pom-second-max-fiber-closed-form} 与推论
\ref{cor:pom-second-max-gap-constant}，
\[
\frac{D_{2k}^{(2)}}{D_{2k}}\to 1-\varphi^{-6},
\qquad
\frac{D_{2k+1}^{(2)}}{D_{2k+1}}\to 1-\frac12\varphi^{-5}.
\]
两极限常数都严格位于 \((0,1)\) 内，因此
\[
\frac1m\log D_m^{(2)}
=
\frac1m\log D_m+\frac1m\log\!\left(\frac{D_m^{(2)}}{D_m}\right),
\]
其中第二项趋于 \(0\)。
再由 Binet 公式或定理 \ref{thm:pom-max-fiber} 的 Fibonacci 闭式，得到
\[
\lim_{m\to\infty}\frac1m\log D_m=\frac12\log\varphi.
\]
故同样的极限也成立于 \(D_m^{(2)}\)。证毕。
\end{proof}

\begin{corollary}[strict one-gap freezing 机制在 Fold 上不适用]\label{cor:derived-fold-not-in-onegap-freezing-regime}
\leanverified{paper\_derived\_fold\_not\_in\_onegap\_freezing\_regime}
附录定理 \ref{thm:fold-pressure-freezing-threshold} 所要求的严格基态间隙假设
\[
\alpha_\ast^{(2)}<\alpha_\ast
\]
在 regular Fold 上失效。确切地说，
\[
\alpha_\ast^{(2)}=\alpha_\ast=\frac12\log\varphi.
\]
因此，任何在 Fold 上出现的有限阶仿射压力段，都不能由该 strict one-gap 冻结机制推出。
\end{corollary}

\begin{proof}
定理 \ref{thm:fold-pressure-freezing-threshold} 的前提正是
\(
\alpha_\ast^{(2)}<\alpha_\ast
\)。
而定理 \ref{thm:derived-fold-top-two-same-slope} 已给出
\(
\alpha_\ast^{(2)}=\alpha_\ast=\frac12\log\varphi
\)。
故该定理在 Fold 上不适用，所述结论随即成立。证毕。
\end{proof}

\begin{theorem}[全态精确反演的 dyadic 阈值闭式]\label{thm:derived-fold-exact-inversion-dyadic-threshold}
\leanverified{paper\_derived\_fold\_exact\_inversion\_dyadic\_threshold}
令
\[
\mathcal C_m(B):=\sum_{x\in X_m}\min\bigl(d_m(x),2^B\bigr),
\qquad
b_{\mathrm{inv}}(m):=\min\{B\in\ZZ_{\ge 0}:\ \mathcal C_m(B)=2^m\}.
\]
则对一切 \(m\ge 2\) 都有
\[
b_{\mathrm{inv}}(m)=\Bigl\lceil\log_2 D_m\Bigr\rceil.
\]
特别地，
\[
b_{\mathrm{inv}}(2k)=\Bigl\lceil \log_2 F_{k+2}\Bigr\rceil,
\qquad
b_{\mathrm{inv}}(2k+1)=\Bigl\lceil 1+\log_2 F_{k+1}\Bigr\rceil.
\]
因此，regular Fold 在每个分辨率上的最小精确反演预算可由顶层纤维闭式逐点完全读出。
\end{theorem}

\begin{proof}
由定理 \ref{thm:xi-exact-inversion-threshold-peak-index}，
\[
b_{\mathrm{inv}}(m)=\Bigl\lceil \log_2 \max_{x\in X_m}d_m(x)\Bigr\rceil
=
\Bigl\lceil \log_2 D_m\Bigr\rceil.
\]
再将定理 \ref{thm:pom-max-fiber} 的闭式
\(
D_{2k}=F_{k+2}
\)
与
\(
D_{2k+1}=2F_{k+1}
\)
代入即可。证毕。
\end{proof}

\begin{corollary}[精确反演率的奇偶展开与 sharp 位率]\label{cor:derived-fold-exact-inversion-parity-rate}
定义临界位率
\[
\beta_c:=\frac12\log_2\varphi.
\]
则有奇偶展开
\[
b_{\mathrm{inv}}(2k)
=
\Bigl\lceil
k\log_2\varphi
+2\log_2\varphi
-\frac12\log_2 5
+o(1)
\Bigr\rceil,
\]
\[
b_{\mathrm{inv}}(2k+1)
=
\Bigl\lceil
k\log_2\varphi
+1+\log_2\varphi
-\frac12\log_2 5
+o(1)
\Bigr\rceil.
\]
从而
\[
\lim_{m\to\infty}\frac{b_{\mathrm{inv}}(m)}{m}=\beta_c.
\]
并且对任意 \(\varepsilon>0\) 都有：
\[
B_m\le (\beta_c-\varepsilon)m\ \text{eventually}
\quad\Longrightarrow\quad
\mathcal C_m(B_m)<2^m\ \text{eventually},
\]
\[
B_m\ge (\beta_c+\varepsilon)m\ \text{eventually}
\quad\Longrightarrow\quad
\mathcal C_m(B_m)=2^m\ \text{eventually}.
\]
因此 \(\beta_c\) 是 regular Fold 全态精确反演的 sharp dyadic 位率。
\end{corollary}

\begin{proof}
由定理 \ref{thm:derived-fold-exact-inversion-dyadic-threshold} 及 Binet 公式
\[
F_n=\frac{\varphi^n}{\sqrt5}(1+o(1)),
\]
直接得到两条奇偶展开。
于是
\[
b_{\mathrm{inv}}(m)=\beta_c\,m+O(1),
\]
故极限位率为 \(\beta_c\)。

若 \(B_m\le(\beta_c-\varepsilon)m\) eventually，则由
\(
b_{\mathrm{inv}}(m)=\beta_c m+O(1)
\)
知对充分大 \(m\) 有 \(B_m<b_{\mathrm{inv}}(m)\)，故
\(
\mathcal C_m(B_m)<2^m
\)。
反向情形同理：若 \(B_m\ge(\beta_c+\varepsilon)m\) eventually，则对充分大 \(m\) 有 \(B_m\ge b_{\mathrm{inv}}(m)\)，于是
\(
\mathcal C_m(B_m)=2^m
\)。
证毕。
\end{proof}

\begin{theorem}[Watatani 指数与二阶碰撞压力的同一性]\label{thm:derived-fold-watatani-collision-pressure-identity}
\leanverified{paper\_derived\_fold\_watatani\_collision\_pressure\_identity}
记
\[
\mathcal E_{m,1}:\mathcal A_{m,1}\to Z(\mathcal A_{m,1})
\]
为附录中的中心值条件期望，并令
\[
I_m:=\mathrm{Ind}_{\mathrm W}(\mathcal E_{m,1}).
\]
则有精确恒等式
\[
\tau_{m,1}(I_m)=\frac{S_2(m)}{2^m}.
\]
进一步，令 \(r_2=\rho(\widetilde A_2)\) 为二阶 moment-kernel 的 Perron 根，\(h_2=\log(4/r_2)\) 为 Rényi--\(2\) 投影熵率。
在实参数压力曲线的整数锚点口径下，
\[
\tau(2)=\log r_2=\log \rho(\mathcal L_2|_{\mathcal H_2}).
\]
于是
\[
\lim_{m\to\infty}\frac1m\log \tau_{m,1}(I_m)
=
\log r_2-\log 2
=
\tau(2)-\log 2
=
\log 2-h_2.
\]
因此，Rényi--\(2\) 碰撞压力在 Fold 上恰可由一个 Watatani 指数可观测量的归一化增长指数读取。
\end{theorem}

\begin{proof}
第一式正是定理 \ref{thm:op-algebra-fold-watatani-index-equals-multiplicity-field}
中的闭式：
\[
\tau_{m,1}\!\left(\mathrm{Ind}_{\mathrm W}(\mathcal E_{m,1})\right)
=
\frac{1}{2^m}\sum_{x\in X_m}d_m(x)^2
=
\frac{S_2(m)}{2^m}.
\]
另一方面，推论 \ref{cor:pom-s2-asymptotic} 给出
\[
S_2(m)=\Theta(r_2^m),
\qquad
h_2=\log\!\left(\frac{4}{r_2}\right).
\]
故
\[
\lim_{m\to\infty}\frac1m\log \tau_{m,1}(I_m)
=
\lim_{m\to\infty}\left(\frac1m\log S_2(m)-\log 2\right)
=
\log r_2-\log 2
=
\log 2-h_2.
\]
再由定理 \ref{thm:pom-real-q-pressure-analytic-interpolation}，
\[
\tau(2)=\log r_2=\log \rho(\mathcal L_2|_{\mathcal H_2}),
\]
即可把同一增长指数改写到实参数射影压力算子口径。证毕。
\end{proof}

\begin{conjecture}[regular Fold 的无有限温冻结]\label{conj:derived-fold-no-finite-temperature-freezing}
在定理 \ref{thm:pom-real-q-pressure-analytic-interpolation} 的实参数制度下，regular Fold 的压力曲线 \(\tau(q)\) 在 \((1,\infty)\) 上满足：
\begin{enumerate}
\normalfont
  \item \(\tau\) 实解析且严格凸；
  \item 对每个有限 \(q>1\)，都有
  \[
  \tau'(q)>\frac12\log\varphi;
  \]
  \item 且
  \[
  \lim_{q\to\infty}\tau'(q)=\frac12\log\varphi.
  \]
\end{enumerate}
等价地，regular Fold 仅在 zero-temperature 极限接触顶层斜率，而不存在任何有限温度上的冻结射线。
\end{conjecture}

\begin{remark}[同坡边界上的压力几何]\label{rem:derived-fold-sameslope-pressure-geometry}
定理 \ref{thm:derived-fold-top-two-same-slope} 与推论
\ref{cor:derived-fold-not-in-onegap-freezing-regime}
说明：regular Fold 正好位于 strict one-gap 冻结机制的边界。
若再结合定理 \ref{thm:pom-real-q-pressure-analytic-interpolation} 的实参数解析延拓、整数锚点 \(\tau(q)=\log r_q\) 的凸性，以及定理 \ref{thm:derived-fold-watatani-collision-pressure-identity} 中 \(q=2\) 的 Watatani 读出，则 regular Fold 的热力学图像更接近于“严格凸直到无穷远才贴住零温斜率”，而不是任何有限整阶或有限实阶上的线性冻结。
\end{remark}
