\subsection{对数密度、偶值 \texorpdfstring{$\zeta$}{zeta} 抽取、解析缓冲层与素数扭曲乘积}

在 Lee--Yang 分支塔的有限层精确等分、\(\mathrm{ZG}\) 的 prime-hard-core 归一化、fold-gauge 常数的 Stirling--Bernoulli 层级以及碰撞势的三次压力方程之外，尚可继续抽取若干彼此独立而可直接调用的后果。其中，解析半径与 RH 临界的相对位置给出一条严格的内部缓冲层，而最小素数模扭曲则收束为一个正有限的归一化 Euler 乘积。

\begin{theorem}[\texorpdfstring{\(\mathrm{ZG}\)}{ZG} 在平方自由层中的条件对数密度等于硬核常数]\label{thm:derived-zg-squarefree-conditional-logdensity-hardcore}
\leanverified{paper\_derived\_zg\_squarefree\_conditional\_logdensity\_hardcore}
沿用推论 \ref{cor:xi-zg-hardcore-constant-residue} 与推论 \ref{cor:xi-prime-hardcore-deltaHC-matches-C_HC} 的记号，记
\[
\mathfrak C_{\mathrm{HC}}
:=
\delta_{\mathrm{HC}}(1)
=
\lim_{N\to\infty}\frac{P_N(1)}{\prod_{k=1}^{N}\left(1+\frac1{p_k}\right)}
\in(0,1).
\]
再定义调和计数函数
\[
A_{\mathrm{ZG}}(x):=
\sum_{\substack{n\le x\\ n\in \mathrm{ZG}}}\frac1n,
\qquad
A_{\mathrm{sf}}(x):=
\sum_{\substack{n\le x\\ \mu^2(n)=1}}\frac1n.
\]
则当 \(x\to\infty\) 时有
\[
A_{\mathrm{ZG}}(x)\sim \frac{\mathfrak C_{\mathrm{HC}}}{\zeta(2)}\log x,
\qquad
A_{\mathrm{sf}}(x)\sim \frac{1}{\zeta(2)}\log x,
\]
因而
\[
\frac{A_{\mathrm{ZG}}(x)}{A_{\mathrm{sf}}(x)}\longrightarrow \mathfrak C_{\mathrm{HC}}.
\]
换言之，\(\mathrm{ZG}\) 在自然数中的对数密度等于 \(\mathfrak C_{\mathrm{HC}}/\zeta(2)\)，而在平方自由整数中的条件对数密度恰等于 prime-hard-core 常数 \(\mathfrak C_{\mathrm{HC}}\)。
\end{theorem}
\begin{proof}
由推论 \ref{cor:xi-zg-hardcore-constant-residue} 与推论 \ref{cor:xi-prime-hardcore-deltaHC-matches-C_HC}，
\[
\delta_{\mathrm{ZG}}=\frac{\mathfrak C_{\mathrm{HC}}}{\zeta(2)}.
\]
再由推论 \ref{cor:xi-zg-reciprocal-harmonic-asymptotic}，
\[
A_{\mathrm{ZG}}(x)\sim \delta_{\mathrm{ZG}}\log x
=\frac{\mathfrak C_{\mathrm{HC}}}{\zeta(2)}\log x.
\]

另一方面，
\[
\sum_{n\ge 1}\mu^2(n)n^{-s}=\frac{\zeta(s)}{\zeta(2s)}
\qquad(\Re(s)>1).
\]
其在 \(s=1\) 处具有留数 \(1/\zeta(2)\)。对这一非负系数 Dirichlet 级数应用与定理 \ref{thm:xi-zg-abel-residue-log-density} 同型的 Abel--Tauberian 论证，得到
\[
A_{\mathrm{sf}}(x)\sim \frac{1}{\zeta(2)}\log x.
\]
两式相除即得
\[
\frac{A_{\mathrm{ZG}}(x)}{A_{\mathrm{sf}}(x)}\to \mathfrak C_{\mathrm{HC}}.
\]
证毕。
\end{proof}

\begin{theorem}[fold-gauge 常数的有限阶移位消元与偶值 \texorpdfstring{$\zeta$}{zeta} 抽取]\label{thm:derived-fold-gauge-shift-annihilator-even-zeta}
沿用定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy} 的记号，令
\[
f_m:=\log C_m-\log(2\pi),
\qquad
q_r:=\left(\frac{\varphi}{2}\right)^{2r-1},
\qquad
a_r:=\frac{B_{2r}}{r(2r-1)}\bigl(\varphi^{-1}+\varphi^{2r-3}\bigr).
\]
在序列空间上记移位算子 \(E\) 为
\[
(Ef)_m:=f_{m+1}.
\]
对每个 \(r\ge 1\) 约定
\[
\mathcal P_{r-1}(E):=\prod_{j=1}^{r-1}(E-q_j),
\]
其中 \(r=1\) 时取空积 \(\mathcal P_0(E)=1\)。则极限
\[
\mathcal L_r
:=
\lim_{m\to\infty}q_r^{-m}\bigl(\mathcal P_{r-1}(E)f\bigr)_m
\]
存在，并满足显式恒等式
\[
\mathcal L_r
=
a_r\prod_{j=1}^{r-1}(q_r-q_j).
\]
从而对每个 \(r\ge 1\) 都有闭式恢复公式
\[
\zeta(2r)
=
(-1)^{r-1}\frac{(2\pi)^{2r}}{2(2r)!}\,
\frac{r(2r-1)}{\varphi^{-1}+\varphi^{2r-3}}\,
\frac{\mathcal L_r}{\prod_{j=1}^{r-1}(q_r-q_j)}.
\]
因此，单一序列 \((\log C_m)_{m\ge 1}\) 在有限阶移位消元之后即可逐模态抽取每个偶值 \(\zeta(2r)\)，而无需递归剥离全部低阶尾项。
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-bin-gauge-constant-stirling-bernoulli-hierarchy}，对每个固定 \(r\ge 1\) 有
\[
f_m=\sum_{s=1}^{r}a_s q_s^m+O(q_{r+1}^m).
\]
对任意固定 \(s\ge 1\)，记几何模态序列 \(g^{(s)}=(g_m^{(s)})_{m\ge 1}\) 为
\[
g_m^{(s)}:=q_s^m.
\]
则直接计算得
\[
\bigl((E-q_j)g^{(s)}\bigr)_m
=q_s^{m+1}-q_j q_s^m
=(q_s-q_j)q_s^m.
\]
因此
\[
\bigl(\mathcal P_{r-1}(E)g^{(s)}\bigr)_m
=
\left(\prod_{j=1}^{r-1}(q_s-q_j)\right)q_s^m.
\]
于是
\[
\bigl(\mathcal P_{r-1}(E)f\bigr)_m
=
\sum_{s=1}^{r}a_s\left(\prod_{j=1}^{r-1}(q_s-q_j)\right)q_s^m
+O(q_{r+1}^m).
\]
当 \(s<r\) 时，上式中的乘积在 \(j=s\) 处出现因子 \(q_s-q_s=0\)，故前 \(r-1\) 个模态全部被精确消去，从而
\[
\bigl(\mathcal P_{r-1}(E)f\bigr)_m
=
a_r\left(\prod_{j=1}^{r-1}(q_r-q_j)\right)q_r^m
+O(q_{r+1}^m).
\]
由于 \(0<q_{r+1}<q_r<1\)，两边同除 \(q_r^m\) 并令 \(m\to\infty\)，即得
\[
\mathcal L_r
=
a_r\prod_{j=1}^{r-1}(q_r-q_j).
\]
最后代入
\[
a_r=\frac{B_{2r}}{r(2r-1)}\bigl(\varphi^{-1}+\varphi^{2r-3}\bigr),
\qquad
B_{2r}=(-1)^{r-1}\frac{2(2r)!}{(2\pi)^{2r}}\zeta(2r),
\]
整理即得 \(\zeta(2r)\) 的显示公式。证毕。
\end{proof}

\begin{theorem}[碰撞势累积量的二次数域刚性与非 holonomic 障碍]\label{thm:derived-collision-cumulants-quadraticfield-nonholonomic}
沿用命题 \ref{prop:real-input-40-collision-pressure} 的记号，定义
\[
P_2(\theta):=\log \rho(e^\theta),
\qquad
\kappa_n:=P_2^{(n)}(0)\qquad(n\ge 1).
\]
则成立：
\begin{enumerate}
  \item 对每个 \(n\ge 1\)，都有
  \[
  \kappa_n\in\QQ(\sqrt5).
  \]
  \item 解析函数芽 \(P_2(\theta)-P_2(0)\) 在 \(\theta=0\) 处不是 D-有限函数。
  \item 因而序列 \((\kappa_n)_{n\ge 1}\) 不是 \(P\)-递归序列。
\end{enumerate}
\end{theorem}
\begin{proof}
记
\[
K:=\QQ(\sqrt5),
\qquad
F(\rho,u):=\rho^3-2\rho^2-(u+1)\rho+u.
\]
由命题 \ref{prop:real-input-40-collision-pressure}，
\[
F(\varphi^2,1)=0.
\]
并且
\[
\partial_\rho F(\varphi^2,1)
=3\varphi^4-4\varphi^2-2
=5\varphi\neq 0.
\]
故形式隐函数定理在域 \(K\) 上适用：存在唯一形式幂级数
\[
\widehat\rho(u)\in K[[u-1]]
\]
使
\[
F(\widehat\rho(u),u)=0,
\qquad
\widehat\rho(1)=\varphi^2.
\]
由于 \(e^\theta-1\in\QQ[[\theta]]\)，合成得到
\[
\widehat\rho(e^\theta)\in K[[\theta]].
\]
再注意 \(\widehat\rho(e^\theta)/\varphi^2\) 的常数项等于 \(1\)，故其形式对数良定义，并满足
\[
P_2(\theta)-P_2(0)
=
\log\!\left(\frac{\widehat\rho(e^\theta)}{\varphi^2}\right)\in K[[\theta]].
\]
于是全部 Taylor 系数 \(\kappa_n\)（\(n\ge 1\)）都落在 \(K=\QQ(\sqrt5)\) 中，这证明了第一条。

第二条由推论 \ref{cor:real-input-40-collision-branch-radius} 直接给出：\(P_2(\theta)\) 的全部代数分歧点恰为
\[
\theta=\log u_j+2\pi i k,
\qquad
k\in\ZZ,
\]
其中 \(u_j\) 遍历三次判别式方程
\[
4u^3+25u^2+88u+8=0
\]
的三个根。故 \(P_2(\theta)-P_2(0)\) 在 \(\CC\) 中具有一条 \(2\pi i\)-平移晶格型的无限奇点集。若它是 D-有限函数芽，则必满足某个具有多项式系数的线性微分方程；从而其有限奇点只能落在该微分方程首项系数的零点集内，因而至多有限个。这与上述无限奇点集矛盾，故它不是 D-有限函数。

最后，\((\kappa_n)_{n\ge 1}\) 的指数型生成函数正是
\[
\sum_{n\ge 1}\kappa_n\frac{\theta^n}{n!}
=
P_2(\theta)-P_2(0).
\]
而 \(P\)-递归序列与 D-有限指数型生成函数等价，因此第二条立即推出第三条。证毕。
\end{proof}

\begin{theorem}[碰撞势压力函数的奇异环：分支多项式显式化与解析延拓半径]\label{thm:derived-collision-pressure-singular-ring}
\leanverified{paper\_derived\_collision\_pressure\_singular\_ring}
沿用命题 \ref{prop:real-input-40-collision-pressure} 的记号，记
\[
P_2(\theta):=\log \rho(e^\theta),
\qquad
f(\rho,u):=\rho^3-2\rho^2-(u+1)\rho+u.
\]
则：
\begin{enumerate}
  \item \(\rho(u)\) 的有限分支点恰由判别式方程
  \[
  \Delta(u):=\operatorname{Disc}_\rho f(\rho,u)=4u^3+25u^2+88u+8=0
  \]
  给出；
  \item 若记 \(\Sigma\) 为 \(P_2(\theta)\) 在 \(\theta\)-平面中的分歧点全体，则
  \[
  \Sigma=
  \{\log u_j+2\pi i k:\ \Delta(u_j)=0,\ k\in\ZZ\};
  \]
  \item 设 \(\Delta(u)=0\) 的三根为
  \[
  u_0\approx -0.09334762302241694,
  \qquad
  u_\pm\approx -3.078326188488792 \pm 3.456761347287067\,i.
  \]
  取主值对数后，\(\theta=0\) 处的 Taylor 收敛半径恰为
  \[
  R_{\mathrm{coll}}
  =
  \min_{\sigma\in\Sigma}|\sigma|
  =
  |\log u_+|
  \approx 2.76230289346087.
  \]
\end{enumerate}
因此，圆盘
\[
|\theta|<R_{\mathrm{coll}}
\]
正是基点 \(\theta=0\) 的最大 Taylor 解析域；一旦穿越边界 \(|\theta|=R_{\mathrm{coll}}\)，便必然遭遇由代数分支引起的解析障碍。换言之，\(P_2\) 的最近奇性在 \(\theta\)-平面上形成一条由 \(\Sigma\) 生成的“奇异环”。
\end{theorem}
\begin{proof}
由代数函数的一般消元原理，有限分支点恰由
\[
f(\rho,u)=0,
\qquad
\partial_\rho f(\rho,u)=0
\]
联立消元得到。对这里的三次多项式直接计算判别式，便得
\[
\Delta(u)=4u^3+25u^2+88u+8.
\]
故第一条成立。

第二条正是把附录推论 \ref{cor:real-input-40-collision-branch-radius} 对 \(u\)-平面分支点的描述，经变量代换 \(u=e^\theta\) 拉回到 \(\theta\)-平面：每个 \(u_j\) 的对数分支在 \(\theta\)-平面中形成整条 \(2\pi i\ZZ\)-平移轨道，因此
\[
\Sigma=
\{\log u_j+2\pi i k:\ \Delta(u_j)=0,\ k\in\ZZ\}.
\]

最后，对 \(\Delta(u)\) 求根即可得到上述三个数值近似。由推论 \ref{cor:real-input-40-collision-branch-radius}，\(\theta=0\) 处的 Taylor 收敛半径等于最近分歧点到原点的距离；而最近者正是 \(\log u_+\) 及其共轭，故
\[
R_{\mathrm{coll}}=|\log u_+|\approx 2.76230289346087.
\]
于是 \(|\theta|<R_{\mathrm{coll}}\) 为最大 Taylor 解析盘，穿越其边界必碰到分支障碍。证毕。
\end{proof}

\begin{corollary}[碰撞势的解析半径、Cauchy 阶乘界与根单位采样阈值]\label{cor:derived-collision-cumulants-radius-cauchy-rootunity}
沿用定理 \ref{thm:derived-collision-cumulants-quadraticfield-nonholonomic} 的记号，并令
\[
\Delta(u):=4u^3+25u^2+88u+8.
\]
记其三个根为
\[
u_0\approx -0.09334762302241694,
\qquad
u_\pm\approx -3.078326188488792 \pm 3.456761347287067\,i.
\]
取主值对数
\[
\theta_{\mathrm{coll}}:=\log u_+
\approx 1.532286026474262 + 2.298350888853574\,i,
\]
并记
\[
R_{\mathrm{coll}}:=|\theta_{\mathrm{coll}}|
\approx 2.76230289346087.
\]
则：
\begin{enumerate}
  \item \(P_2(\theta)\) 在 \(\theta=0\) 处的 Taylor 收敛半径恰为
  \[
  \operatorname{rad}_0 P_2=R_{\mathrm{coll}}.
  \]
  \item 对任意 \(0<R<R_{\mathrm{coll}}\)，记
  \[
  M_R:=\max_{|\theta|=R}|P_2(\theta)|.
  \]
  则对每个 \(n\ge 1\) 都有 Cauchy 型界
  \[
  |\kappa_n|\le n!\,M_R\,R^{-n}.
  \]
  \item 对根单位采样点
  \[
  \theta_p:=\frac{2\pi i}{p}\qquad(p\ge 2),
  \]
  \(\theta=0\) 处的 Taylor 级数在 \(\theta_p\) 收敛当且仅当
  \[
  \frac{2\pi}{p}<R_{\mathrm{coll}}.
  \]
  因而满足该判据的最小整数模数为
  \[
  p_{\min}^{\mathrm{coll}}=3.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
由推论 \ref{cor:real-input-40-collision-branch-radius}，\(P_2(\theta)\) 的分歧点全体为
\[
\theta=\log u_j+2\pi i k,
\qquad
k\in\ZZ,
\]
其中 \(u_j\) 遍历 \(\Delta(u)=0\) 的三个根。对三根直接求值可得上式中的数值近似，而最近的分歧点正是主值对数 \(\theta_{\mathrm{coll}}=\log u_+\) 及其共轭，故
\[
\operatorname{rad}_0 P_2=|\theta_{\mathrm{coll}}|=R_{\mathrm{coll}}.
\]

对任意 \(0<R<R_{\mathrm{coll}}\)，函数 \(P_2\) 在闭圆周 \(|\theta|=R\) 上全纯，故由 Cauchy 导数估计直接得到
\[
|\kappa_n|=|P_2^{(n)}(0)|\le n!\,M_R\,R^{-n}.
\]

最后，\(|\theta_p|=2\pi/p\)。因此 Taylor 级数在 \(\theta_p\) 收敛当且仅当 \(|\theta_p|<R_{\mathrm{coll}}\)，亦即
\[
\frac{2\pi}{p}<R_{\mathrm{coll}}.
\]
由于
\[
\frac{2\pi}{R_{\mathrm{coll}}}\approx 2.2743,
\]
满足该不等式的最小整数恰为 \(p=3\)。证毕。
\end{proof}

R_\theta:=R_{\mathrm{coll}},
\qquad
\theta_R:=\log u_R,

则

0<\theta_R<R_\theta,

\delta_{\mathrm{an}}>0,

特别地，实区间

(\theta_R,R_\theta)\neq \varnothing

非空，并且对任意 \(\theta\in(\theta_R,R_\theta)\)，函数 \(P_2(\theta)\) 仍解析，而 kernel RH/Ramanujan 条件已经失效。因而 RH 型阈值的破裂发生在解析域内部，而非由最近代数分歧点触发。

R_\theta=R_{\mathrm{coll}}\approx 2.76230289346087.

\theta_R=\log u_R\approx 1.27888,

\textsf{RH}(u)\iff 0<u\le u_R.

\end{proof}

\begin{theorem}[素数模最小扭曲的归一化乘积极限与尾误差]\label{thm:derived-collision-prime-twist-normalized-product}
沿用推论 \ref{cor:real-input-40-collision-twist-fourth} 与定理 \ref{thm:xi-time-part9p-odd-prime-shadow-product} 的记号，记
\[
\lambda:=\rho(1)=\varphi^2,
\qquad
\mathcal P(X):=\prod_{\substack{3\le p\le X\\ p\ \mathrm{prime}}}\frac{\rho(2\pi/p)}{\lambda}.
\]
则极限
\[
\mathcal P_\infty
:=
\lim_{X\to\infty}\mathcal P(X)
=
\prod_{\substack{p\ge 3\\ p\ \mathrm{prime}}}\frac{\rho(2\pi/p)}{\lambda}
\]
存在，并满足
\[
0<\mathcal P_\infty<1.
\]
而且尾积具有定量控制
\[
0\le
\log\frac{\mathcal P(X)}{\mathcal P_\infty}
=
O\!\left(\sum_{p>X}\frac1{p^2}\right)
=
O\!\left(\frac1{X\log X}\right).
\]
若再把模 \(2\) 的奇偶扭曲一并纳入，则完整归一化乘积
\[
\widetilde{\mathcal P}_\infty
:=
\frac{\rho(-1)}{\lambda}\,\mathcal P_\infty
\]
同样满足
\[
0<\widetilde{\mathcal P}_\infty<1.
\]
\end{theorem}
\leanverified{paper\_derived\_collision\_prime\_twist\_normalized\_product}
\begin{proof}
由推论 \ref{cor:real-input-40-collision-twist-fourth}，对奇素数 \(p\) 的最小非平凡角 \(t_p=2\pi/p\) 有
\[
\log\frac{\rho(t_p)}{\lambda}
=
-\frac{6\sqrt5-5}{250}\left(\frac{2\pi}{p}\right)^2
+\frac{7+24\sqrt5}{75000}\left(\frac{2\pi}{p}\right)^4
+O\!\left(p^{-6}\right).
\]
因此存在常数 \(C>0\) 使对充分大奇素数 \(p\) 成立
\[
\left|\log\frac{\rho(2\pi/p)}{\lambda}\right|\le \frac{C}{p^2}.
\]
由于
\[
\sum_{p\ \mathrm{prime}}\frac1{p^2}<\infty,
\]
可知级数
\[
\sum_{\substack{p\ge 3\\ p\ \mathrm{prime}}}\left|\log\frac{\rho(2\pi/p)}{\lambda}\right|
\]
绝对收敛。再由定理 \ref{thm:xi-time-part9p-odd-prime-shadow-product}，每个因子都严格落在 \((0,1)\) 内，故奇素数归一化乘积极限 \(\mathcal P_\infty\) 存在且满足 \(0<\mathcal P_\infty<1\)。

对任意 \(X\ge 3\)，由单调收敛与对数可加性，
\[
\log\frac{\mathcal P(X)}{\mathcal P_\infty}
=
-\sum_{\substack{p>X\\ p\ \mathrm{prime}}}
\log\frac{\rho(2\pi/p)}{\lambda}
=
O\!\left(\sum_{p>X}\frac1{p^2}\right).
\]
而由素数定理的分部求和形式，
\[
\sum_{\substack{p>X\\ p\ \mathrm{prime}}}\frac1{p^2}
\ll
\int_X^\infty \frac{\dd t}{t^2\log t}
\ll
\frac1{X\log X},
\]
遂得所示尾误差。

最后，由注 \ref{rem:real-input-40-collision-parity}，
\[
0<\frac{\rho(-1)}{\lambda}<1.
\]
把该模 \(2\) 因子乘到 \(\mathcal P_\infty\) 上，即得
\[
0<\widetilde{\mathcal P}_\infty<1.
\]
证毕。
\end{proof}
