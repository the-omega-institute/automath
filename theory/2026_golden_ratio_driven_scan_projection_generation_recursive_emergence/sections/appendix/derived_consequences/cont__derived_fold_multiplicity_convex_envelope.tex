\subsection{折叠多重度剖面的凸序极值与容量包络}

定理 \ref{thm:fold-congruence-mod-semirings}、定理 \ref{thm:fold-multiplicity-energy} 与定理 \ref{thm:fold-collision-spectrum} 已把折叠多重度剖面 \(d_m\) 识别为定义在有限循环群 \(\ZZ/(F_{m+2}\ZZ)\) 上的整数值类函数，并将碰撞缺口严格等价为非平凡 Fourier 模能量。结论 \ref{con:xi-time-part9l-fold-fourier-energy-arithmetic-lower-bound} 已给出平方势的整除障碍，而定理 \ref{thm:xi-dyadic-oracle-capacity-mellin-recovery} 则把连续容量曲线与实阶矩谱联结为 Mellin--Stieltjes 对偶。把这些接口统一到固定总质量的离散凸序问题上，可进一步得到如下精确极值原理。

\begin{theorem}[折叠多重度剖面的精确凸序极值原理]\label{thm:derived-fold-multiplicity-convex-order-extremal}
\leanverified{paper\_derived\_fold\_multiplicity\_convex\_order\_extremal}
固定 \(m\ge 1\)，并记
\[
F:=F_{m+2},
\qquad
M:=2^m,
\qquad
M=aF+\rho,
\qquad
0\le \rho<F.
\]
对任意函数 \(\Phi:\RR_{\ge 0}\to \RR\)，把折叠多重度剖面 \(d_m\) 视为定义在 \(\ZZ/(F\ZZ)\) 上的整数值函数。则：
\begin{enumerate}
  \item 若 \(\Phi\) 为凸函数，则
  \[
  \sum_{u\in \ZZ/(F\ZZ)}\Phi\bigl(d_m(u)\bigr)
  \ge
  (F-\rho)\Phi(a)+\rho\,\Phi(a+1).
  \]
  \item 若 \(\Phi\) 为凹函数，则
  \[
  \sum_{u\in \ZZ/(F\ZZ)}\Phi\bigl(d_m(u)\bigr)
  \le
  (F-\rho)\Phi(a)+\rho\,\Phi(a+1).
  \]
  \item 若 \(\Phi\) 严格凸或严格凹，则等号成立当且仅当 \(d_m\) 为平衡剖面，即
  \[
  d_m(u)\in\{a,a+1\}\qquad (\forall u\in \ZZ/(F\ZZ)),
  \]
  且恰有 \(\rho\) 个余类取值 \(a+1\)，其余 \(F-\rho\) 个余类取值 \(a\)。
\end{enumerate}
\end{theorem}
\begin{proof}
记
\[
v_u:=d_m(u)\in \ZZ_{\ge 0},
\qquad
\sum_{u\in \ZZ/(F\ZZ)}v_u=M.
\]
若存在一对余类 \(u_1,u_2\) 使
\[
v_{u_1}\ge v_{u_2}+2,
\]
则作平衡化操作
\[
(v_{u_1},v_{u_2})\longmapsto (v_{u_1}-1,v_{u_2}+1).
\]
设 \(\Delta_\Phi(n):=\Phi(n+1)-\Phi(n)\)（\(n\in\ZZ_{\ge 0}\)）。当 \(\Phi\) 凸时，\(\Delta_\Phi(n)\) 单调不减，故
\[
\Phi(v_{u_1})-\Phi(v_{u_1}-1)
=
\Delta_\Phi(v_{u_1}-1)
\ge
\Delta_\Phi(v_{u_2})
=
\Phi(v_{u_2}+1)-\Phi(v_{u_2}),
\]
亦即
\[
\Phi(v_{u_1})+\Phi(v_{u_2})
\ge
\Phi(v_{u_1}-1)+\Phi(v_{u_2}+1).
\]
若 \(\Phi\) 严格凸，则只要 \(v_{u_1}\ge v_{u_2}+2\)，上述不等式即为严格不等式。对凹函数情形，不等号全部反向。

因此，在保持总和 \(M\) 不变的前提下，反复施行该平衡化操作，会使
\[
\sum_u \Phi(v_u)
\]
在凸情形单调下降、在凹情形单调上升；而该过程在有限步后终止于所有坐标彼此相差至多 \(1\) 的剖面。由于
\[
M=aF+\rho,
\qquad
0\le \rho<F,
\]
终止态只能是 \(\rho\) 个坐标取 \(a+1\)、其余 \(F-\rho\) 个坐标取 \(a\)。这便给出极值值
\[
(F-\rho)\Phi(a)+\rho\,\Phi(a+1).
\]
若 \(\Phi\) 严格凸或严格凹，则每一步只要尚未达到平衡剖面就会严格改进，故等号条件亦随之确定。证毕。
\end{proof}

\leanverified{paper\_derived\_fold\_multiplicity\_fourier\_collision\_floor}
\begin{corollary}[非平凡 Fourier 能量与碰撞概率的精确整数障碍]\label{cor:derived-fold-multiplicity-fourier-collision-floor}
沿用定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 的记号。则有
\[
\sum_{k\ne 0}\bigl|\widehat d_m(k)\bigr|^2
\ge
\rho(F-\rho),
\]
以及
\[
\mathsf{Col}_m
\ge
\frac{(F-\rho)a^2+\rho(a+1)^2}{M^2}
=
\frac1F+\frac{\rho(F-\rho)}{F\,M^2}.
\]
并且等号成立当且仅当 \(d_m\) 为定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 的平衡剖面。
\end{corollary}
\begin{proof}
在定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 中取 \(\Phi(t)=t^2\)，得到
\[
\sum_{u}d_m(u)^2
\ge
(F-\rho)a^2+\rho(a+1)^2.
\]
又由定理 \ref{thm:fold-multiplicity-energy}，
\[
\sum_{k\ne 0}\bigl|\widehat d_m(k)\bigr|^2
=
F\sum_u d_m(u)^2-M^2.
\]
代入上式并化简即得
\[
\sum_{k\ne 0}\bigl|\widehat d_m(k)\bigr|^2
\ge
F\bigl((F-\rho)a^2+\rho(a+1)^2\bigr)-M^2
=
\rho(F-\rho).
\]
再由定理 \ref{thm:fold-collision-spectrum}，
\[
\mathsf{Col}_m
=
\frac1F+\frac{1}{M^2F}\sum_{k\ne 0}\bigl|\widehat d_m(k)\bigr|^2,
\]
便得到碰撞概率的下界。由于 \(t^2\) 严格凸，等号条件由定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 直接给出。这正是结论 \ref{con:xi-time-part9l-fold-fourier-energy-arithmetic-lower-bound} 的凸序极值化表述。证毕。
\end{proof}

\begin{theorem}[连续容量曲线与实阶幂和的精确包络]\label{thm:derived-fold-multiplicity-capacity-renyi-envelope}
\leanverified{paper\_derived\_fold\_multiplicity\_capacity\_renyi\_envelope}
沿用定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 的记号。定义连续容量曲线
\[
\mathcal C_m^{\mathrm{cont}}(T):=\sum_{u\in \ZZ/(F\ZZ)}\min\bigl(d_m(u),T\bigr)
\qquad (T\ge 0),
\]
以及实阶幂和
\[
S_q(m):=\sum_{u\in \ZZ/(F\ZZ)} d_m(u)^q
\qquad (q>0).
\]
则有：
\begin{enumerate}
  \item 对任意 \(T\ge 0\)，连续容量满足精确上界
  \[
  \mathcal C_m^{\mathrm{cont}}(T)
  \le
  (F-\rho)\min(a,T)+\rho\min(a+1,T).
  \]
  等价地，
  \[
  \mathcal C_m^{\mathrm{cont}}(T)\le
  \begin{cases}
  FT, & 0\le T\le a,\\[4pt]
  (F-\rho)a+\rho T, & a\le T\le a+1,\\[4pt]
  M, & T\ge a+1.
  \end{cases}
  \]
  平衡剖面对一切 \(T\ge 0\) 都达到等号。

  \item 对 \(q>1\)，有
  \[
  S_q(m)\ge (F-\rho)a^q+\rho(a+1)^q.
  \]
  对 \(0<q<1\)，有
  \[
  S_q(m)\le (F-\rho)a^q+\rho(a+1)^q.
  \]
  当 \(q=1\) 时，
  \[
  S_1(m)=M
  \]
  为恒等式。对于 \(q\ne 1\)，等号成立当且仅当 \(d_m\) 为平衡剖面。
\end{enumerate}
\end{theorem}
\begin{proof}
对任意固定 \(T\ge 0\)，函数
\[
\Phi_T(t):=\min(t,T)
\]
在 \(\RR_{\ge 0}\) 上凹。由定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 的凹函数情形，
\[
\mathcal C_m^{\mathrm{cont}}(T)
=
\sum_u \Phi_T(d_m(u))
\le
(F-\rho)\Phi_T(a)+\rho\,\Phi_T(a+1),
\]
即得第一条。分段写法只是把右端在三个区间上显式化。平衡剖面代入后显然处处达到等号。

对第二条，当 \(q>1\) 时，\(t\mapsto t^q\) 在 \(\RR_{\ge 0}\) 上严格凸；当 \(0<q<1\) 时，它在 \(\RR_{\ge 0}\) 上严格凹。于是再由定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal} 便得相应不等式与等号条件。\(q=1\) 时 \(S_1(m)=\sum_u d_m(u)=M\) 为总质量恒等式。证毕。
\end{proof}

\begin{theorem}[零纤维约束下的精确包络]\label{thm:derived-fold-multiplicity-zerofiber-envelope}
\leanverified{paper\_derived\_fold\_multiplicity\_zerofiber\_envelope}
固定整数 \(z\) 并假设折叠多重度剖面满足
\[
\#\{u\in \ZZ/(F\ZZ):\ d_m(u)=0\}\ge z,
\qquad
0\le z<F.
\]
记
\[
n:=F-z,
\qquad
M=a_z n+\rho_z,
\qquad
0\le \rho_z<n.
\]
则有：
\begin{enumerate}
  \item 平方和、Fourier 能量与碰撞概率满足
  \[
  \sum_{u}d_m(u)^2
  \ge
  (n-\rho_z)a_z^2+\rho_z(a_z+1)^2,
  \]
  \[
  \sum_{k\ne 0}\bigl|\widehat d_m(k)\bigr|^2
  \ge
  F\Bigl((n-\rho_z)a_z^2+\rho_z(a_z+1)^2\Bigr)-M^2,
  \]
  \[
  \mathsf{Col}_m
  \ge
  \frac{(n-\rho_z)a_z^2+\rho_z(a_z+1)^2}{M^2}.
  \]

  \item 连续容量曲线满足
  \[
  \mathcal C_m^{\mathrm{cont}}(T)
  \le
  (n-\rho_z)\min(a_z,T)+\rho_z\min(a_z+1,T)
  \qquad (T\ge 0).
  \]

  \item 最大纤维满足
  \[
  \max_{u} d_m(u)
  \ge
  \left\lceil \frac{M}{F-z}\right\rceil
  =
  a_z+\mathbf 1_{\{\rho_z>0\}}.
  \]
\end{enumerate}
上述各界都由下述剖面同时达到：恰有 \(z\) 个零纤维，其余 \(n\) 个纤维形成平衡剖面，即在非零支撑上仅取 \(a_z\) 与 \(a_z+1\) 两值，且恰有 \(\rho_z\) 个取 \(a_z+1\)。
\end{theorem}
\begin{proof}
由零纤维约束，可把 \(d_m\) 的全部取值重排为一个长度 \(n\) 的非负整数列
\[
b_1,\dots,b_n
\]
与 \(z\) 个附加的零，其中
\[
\sum_{j=1}^{n} b_j=M.
\]
对该长度 \(n\) 的列应用定理 \ref{thm:derived-fold-multiplicity-convex-order-extremal}，即可得到
\[
\sum_{j=1}^{n} b_j^2
\ge
(n-\rho_z)a_z^2+\rho_z(a_z+1)^2,
\]
以及
\[
\sum_{j=1}^{n}\min(b_j,T)
\le
(n-\rho_z)\min(a_z,T)+\rho_z\min(a_z+1,T).
\]
把 \(z\) 个零项补回便得第一式与容量上界。

Fourier 能量下界再由定理 \ref{thm:fold-multiplicity-energy}
\[
\sum_{k\ne 0}\bigl|\widehat d_m(k)\bigr|^2
=
F\sum_u d_m(u)^2-M^2
\]
直接推出；碰撞概率下界则由
\[
\mathsf{Col}_m=\frac{1}{M^2}\sum_u d_m(u)^2
\]
得到。

最后，若全部非零纤维都严格小于 \(\lceil M/n\rceil\)，则其总和严格小于 \(M\)，矛盾，故
\[
\max_{u}d_m(u)\ge \left\lceil \frac{M}{n}\right\rceil.
\]
而构造“\(z\) 个零纤维 \(+\) 其余 \(n\) 个纤维平衡分配”的剖面即可同时达到上述各界。证毕。
\end{proof}

\begin{corollary}[Fibonacci 尺度零纤维约束的专门化]\label{cor:derived-fold-multiplicity-fibonacci-zerofiber-specialization}
在定理 \ref{thm:derived-fold-multiplicity-zerofiber-envelope} 的设定下，若额外知道
\[
\#\{u\in \ZZ/(F_{m+2}\ZZ):\ d_m(u)=0\}\ge F_{(m+2)/2},
\]
则记
\[
n:=F_{m+2}-F_{(m+2)/2},
\qquad
2^m=a_\ast n+\rho_\ast,
\qquad
0\le \rho_\ast<n,
\]
便有
\[
\mathsf{Col}_m
\ge
\frac{(n-\rho_\ast)a_\ast^2+\rho_\ast(a_\ast+1)^2}{2^{2m}},
\]
\[
\mathcal C_m^{\mathrm{cont}}(T)
\le
(n-\rho_\ast)\min(a_\ast,T)+\rho_\ast\min(a_\ast+1,T)
\qquad (T\ge 0),
\]
\[
\max_u d_m(u)\ge \left\lceil\frac{2^m}{F_{m+2}-F_{(m+2)/2}}\right\rceil.
\]
\end{corollary}
\begin{proof}
这只是定理 \ref{thm:derived-fold-multiplicity-zerofiber-envelope} 在
\[
z=F_{(m+2)/2}
\]
处的直接专门化。证毕。
\leanverified{paper\_derived\_fold\_multiplicity\_fibonacci\_zerofiber\_specialization}
\end{proof}

\begin{remark}[与六倍窗半阶零块的接口]\label{rem:derived-fold-multiplicity-zerofiber-vs-spectral-zero}
推论 \ref{cor:fold-zero-half-index-multiple6} 已在频域上给出六倍窗的半阶零块
\[
|\mathcal Z_m|\ge F_{(m+2)/2},
\qquad
S_{F_{(m+2)/2}}(F_{m+2})\subseteq \mathcal Z_m.
\]
因此，Fibonacci 尺度 \(F_{(m+2)/2}\) 已在 Fourier 侧被严格锁定。推论 \ref{cor:derived-fold-multiplicity-fibonacci-zerofiber-specialization} 说明：一旦余类侧获得同尺度的零纤维约束，上述碰撞、容量与峰值界便会立即收束为精确包络。
\end{remark}
