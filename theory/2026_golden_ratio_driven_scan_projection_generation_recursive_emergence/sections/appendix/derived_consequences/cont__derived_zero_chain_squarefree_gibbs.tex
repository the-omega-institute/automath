\paragraph{谱零点的半维钉扎与平方自由 Gibbs 的乘积化}
六倍窗上的谱零点下上界与链上内算子的平方自由算术，分别把零点计数与固定点格统计压缩到约数结构与有限 Euler 因子之中。把这两条压缩机制再推进一步，即得下列两个直接后果。

\begin{theorem}[六倍窗谱零点集合的乘法型半维指数钉扎]\label{thm:derived-window6-zero-set-multiplicative-half-exponent}
\leanverified{paper\_derived\_window6\_zero\_set\_multiplicative\_half\_exponent}
沿用推论 \ref{cor:fold-zero-half-index-multiple6} 与定理 \ref{thm:fold-zero-density-sparse} 的记号，记
\[
\mathcal Z_m:=\{k\in \ZZ/(F_{m+2}\ZZ):\ \widehat d_m(k)=0\}.
\]
若
\[
m\equiv 4\pmod 6,
\]
则有双边估计
\[
F_{(m+2)/2}\le |\mathcal Z_m|\le \tau(m+2)\,F_{(m+2)/2},
\]
其中 \(\tau(\cdot)\) 为约数函数。因而
\[
|\mathcal Z_m|=F_{(m+2)/2}\exp(o(m)),
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 4\ (6)}}\frac1m\log|\mathcal Z_m|
=
\frac12\log\varphi.
\]
\end{theorem}
\begin{proof}
条件 \(m\equiv 4\pmod 6\) 等价于 \(m+2\equiv 0\pmod 6\)。令
\[
d:=\frac{m+2}{2}.
\]
由推论 \ref{cor:fold-zero-half-index-multiple6}，
\[
|\mathcal Z_m|\ge F_d.
\]
另一方面，定理 \ref{thm:fold-zero-density-sparse} 给出
\[
|\mathcal Z_m|
\le
\tau(m+2)\,F_{\lfloor (m+2)/2\rfloor}
=
\tau(m+2)\,F_d.
\]
这就得到所示双边估计。

再由经典估计 \(\tau(n)=e^{o(n)}\)，可将上式改写为
\[
|\mathcal Z_m|=F_d\,e^{o(m)}=F_{(m+2)/2}\exp(o(m)).
\]
最后，Binet 公式给出
\[
\log F_d=d\log\varphi+o(d)=\frac{m+2}{2}\log\varphi+o(m),
\]
从而
\[
\frac1m\log|\mathcal Z_m|
=
\frac1m\log F_{(m+2)/2}+o(1)
\longrightarrow
\frac12\log\varphi.
\]
证毕。
\end{proof}

\begin{theorem}[链上规范投影的平方自由 Gibbs 系综的 Bernoulli 因子化]\label{thm:derived-chain-squarefree-gibbs-bernoulli-factorization}
\leanverified{paper\_derived\_chain\_squarefree\_gibbs\_bernoulli\_factorization}
沿用定理 \ref{thm:killo-chain-interior-godel-gcd-lcm} 的记号。对任意 \(s>0\)，定义 Gibbs 权
\[
Z_n(s):=\sum_{J\in \mathrm{Int}(C_n)}\kappa(J)^{-s},
\qquad
\mathbb P_s(I):=\frac{\kappa(I)^{-s}}{Z_n(s)}.
\]
再对 \(1\le i\le n-1\) 定义坐标随机变量
\[
X_i(I):=\mathbf 1_{\{i\in \mathrm{Fix}(I)\}}.
\]
则 \(X_1,\dots,X_{n-1}\) 相互独立，且
\[
\mathbb P_s(X_i=1)=\frac{1}{1+q_i^s},
\qquad
\mathbb P_s(X_i=0)=\frac{q_i^s}{1+q_i^s}.
\]
并且配分函数与低阶统计量都具有闭式
\[
Z_n(s)=q_0^{-s}\prod_{i=1}^{n-1}(1+q_i^{-s}),
\]
\[
\mathbb E_s\bigl(|\mathrm{Fix}(I)|-1\bigr)
=
\sum_{i=1}^{n-1}\frac{1}{1+q_i^s},
\]
\[
\mathrm{Var}_s\bigl(|\mathrm{Fix}(I)|-1\bigr)
=
\sum_{i=1}^{n-1}\frac{q_i^s}{(1+q_i^s)^2}.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:killo-chain-interior-godel-gcd-lcm}，每个 \(I\in \mathrm{Int}(C_n)\) 唯一对应于某个子集
\[
S\subseteq \{1,\dots,n-1\}
\]
使得
\[
\mathrm{Fix}(I)=\{0\}\cup S,
\qquad
\kappa(I)=q_0\prod_{i\in S}q_i.
\]
因此
\[
Z_n(s)
=
\sum_{S\subseteq \{1,\dots,n-1\}}
\left(q_0\prod_{i\in S}q_i\right)^{-s}
=
q_0^{-s}\prod_{i=1}^{n-1}(1+q_i^{-s}).
\]

若仍以 \(S\) 表示 \(I\) 的自由固定点集合，则
\[
\mathbb P_s(S)
=
\frac{q_0^{-s}\prod_{i\in S}q_i^{-s}}{q_0^{-s}\prod_{i=1}^{n-1}(1+q_i^{-s})}
=
\prod_{i\in S}\frac{q_i^{-s}}{1+q_i^{-s}}
\prod_{i\notin S}\frac{1}{1+q_i^{-s}}.
\]
这正是独立 Bernoulli 变量的乘积分布，其中
\[
\frac{q_i^{-s}}{1+q_i^{-s}}=\frac{1}{1+q_i^s},
\qquad
\frac{1}{1+q_i^{-s}}=\frac{q_i^s}{1+q_i^s}.
\]
故 \(X_1,\dots,X_{n-1}\) 相互独立，且边际分布如上。

最后，由
\[
|\mathrm{Fix}(I)|-1=\sum_{i=1}^{n-1}X_i
\]
以及独立 Bernoulli 求和公式，立即得到所示期望与方差闭式。证毕。
\end{proof}

\begin{theorem}[ZG 硬核 Dirichlet 级数的一阶 Tauberian 完备与二阶 Euler 不可约]\label{thm:derived-zg-tauberian-complete-euler-irreducible}
记
\[
\mathcal D_{\mathrm{ZG}}(\sigma):=\sum_{n\in \mathrm{ZG}}n^{-\sigma}
\qquad (\sigma>1).
\]
则存在严格正函数 \(H_{\mathrm{ZG}}(\sigma)\) 使
\[
\mathcal D_{\mathrm{ZG}}(\sigma)
=
\frac{\zeta(\sigma)}{\zeta(2\sigma)}\,H_{\mathrm{ZG}}(\sigma),
\qquad
H_{\mathrm{ZG}}(1+)=\zeta(2)\,\delta_{\mathrm{ZG}},
\]
其中 \(\delta_{\mathrm{ZG}}\) 为 \(\mathrm{ZG}\) 的自然密度。

另一方面，不存在任何标量 Euler 分解
\[
\mathcal D_{\mathrm{ZG}}(s)=\prod_{i\ge 1}F_i(p_i^{-s}),
\qquad
F_i(T)\in 1+T\CC[[T]].
\]
更强地，这种不可约性在二次层已经出现：若强行匹配常数项与单素数项，则每个局部因子都被迫等于
\[
F_i(T)=1+T,
\]
从而相邻素数积 \(p_ip_{i+1}\) 的系数必为 \(1\)；但在 \(\mathrm{ZG}\) 中该系数严格为 \(0\)。
\end{theorem}
\begin{proof}
前半部分正是定理 \ref{thm:xi-zg-zeta-factorization} 与推论 \ref{cor:xi-zg-hardcore-constant-residue} 的合并表述。

现证 Euler 不可约性。反设存在
\[
\mathcal D_{\mathrm{ZG}}(s)=\prod_{i\ge 1}F_i(p_i^{-s}),
\qquad
F_i(T)=1+\sum_{m\ge 1}c_{i,m}T^m.
\]
由于 \(\mathrm{ZG}\) 只含平方自由整数，所有 \(p_i^m\ (m\ge 2)\) 的系数都必须为 \(0\)。故每个 \(F_i\) 只能线性化为
\[
F_i(T)=1+c_{i,1}T.
\]
再由单素数 \(p_i\in \mathrm{ZG}\) 且其系数等于 \(1\)，得 \(c_{i,1}=1\)，所以
\[
F_i(T)=1+T
\qquad (\forall i).
\]
但此时形式乘积中相邻素数积 \(p_ip_{i+1}\) 的系数等于 \(1\)，而 \(\mathrm{ZG}\) 的相邻禁约束恰给出
\[
p_ip_{i+1}\notin \mathrm{ZG},
\]
故该系数应为 \(0\)。矛盾。因此不存在标量 Euler 分解。证毕。
\end{proof}

\begin{theorem}[素数硬核 Gibbs 场的渐近 Poisson 平衡]\label{thm:derived-zg-prime-hardcore-poisson-balance}
固定 \(t>0\)。在 \(\{1,\dots,N\}\) 的无相邻子集上按权重
\[
\propto t^{|S|}\prod_{k\in S}p_k^{-1}
\]
取样，记选中指标数为 \(K_N\)。则存在只依赖于 \(t\) 的常数 \(\mu(t),\nu(t)\) 使
\[
\EE[K_N]=t\log\log p_N+\mu(t)+o(1),
\]
\[
\Var(K_N)=t\log\log p_N+\nu(t)+o(1),
\]
并且
\[
\frac{\Var(K_N)}{\EE[K_N]}\longrightarrow 1,
\qquad
\frac{K_N-\EE[K_N]}{\sqrt{\Var(K_N)}}\Longrightarrow \mathcal N(0,1).
\]
因此，尽管 \(\mathrm{ZG}\) 在 Euler 层保留不可压缩的最近邻耦合，prime-hard-core Gibbs 场在第一、第二累积量尺度上仍恢复到 Poisson 平衡。
\leanverified{paper\_derived\_zg\_prime\_hardcore\_poisson\_balance}
\end{theorem}
\begin{proof}
定理 \ref{thm:xi-prime-hardcore-gibbs-clt} 已经给出
\[
\EE[K_N]=t\log\log p_N+\mu(t)+o(1),
\qquad
\Var(K_N)=t\log\log p_N+\nu(t)+o(1),
\]
以及中心极限定律。于是
\[
\frac{\Var(K_N)}{\EE[K_N]}
=
\frac{t\log\log p_N+\nu(t)+o(1)}{t\log\log p_N+\mu(t)+o(1)}
\longrightarrow 1.
\]
这正是 Poisson 平衡在前两阶累积量上的表述，而正态极限则由同一定理直接给出。证毕。
\end{proof}
