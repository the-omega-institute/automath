\subsection{纤维独立集复形的 Witten 指数与非可缩相位}

定理 \ref{thm:pom-fiber-independence-complex-classification} 与定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 已把纤维独立集复形的同伦型压缩为“偶重可缩、奇重球面”的严格二分；命题 \ref{prop:pom-fiber-witten-index-zx-minus-one} 与推论 \ref{cor:pom-fiber-independence-complex-homology} 则进一步把交错计数、约化 Euler 特征与球面维数联结起来。另一方面，定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 已在审计偶数窗上锁定最小退化扇区的 Fibonacci 规模，而定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase} 与定理 \ref{thm:pom-max-noncontractible-fiber-exponent-mod6-ratio} 已给出最大非可缩纤维的模 \(6\) 相图与常数层缺陷。把这些接口并置后，可得到下列派生后果。

\begin{theorem}[纤维独立集复形的交错 Witten 指数由纤维奇偶完全决定]\label{thm:derived-fiber-indcomplex-alternating-witten-parity}
\leanverified{paper\_derived\_fiber\_indcomplex\_alternating\_witten\_parity}
设 \(m\ge 1\)，\(x\in X_m\) 为稳定类型，并沿用定理 \ref{thm:pom-fiber-parity-homotopy-equivalence} 的记号
\[
\Gamma_x\cong \bigsqcup_{i=1}^{r}P_{\ell_i},
\qquad
K_x:=\Ind^{\Delta}(\Gamma_x),
\qquad
\tau(x):=\sum_{i=1}^{r}\left\lfloor\frac{\ell_i+1}{3}\right\rfloor.
\]
定义交错计数
\[
W_x:=\sum_{S\in \mathsf{Ind}(\Gamma_x)}(-1)^{|S|}.
\]
则有严格闭式
\[
W_x=
\begin{cases}
0, & d_m(x)\ \text{为偶数},\\[4pt]
(-1)^{\tau(x)}, & d_m(x)\ \text{为奇数}.
\end{cases}
\]
等价地，
\[
|W_x|=1\iff d_m(x)\ \text{为奇数},
\qquad
|W_x|=0\iff d_m(x)\ \text{为偶数}.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:pom-fiber-witten-index-zx-minus-one}，
\[
W_x=-\widetilde\chi(|K_x|),
\]
其中 \(\widetilde\chi\) 为约化 Euler 特征。若 \(d_m(x)\) 为偶数，则由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，\(|K_x|\) 可缩，故
\[
\widetilde\chi(|K_x|)=0,
\qquad
W_x=0.
\]
若 \(d_m(x)\) 为奇数，则同一定理给出
\[
|K_x|\simeq S^{\tau(x)-1}.
\]
再由推论 \ref{cor:pom-fiber-independence-complex-homology}，
\[
\widetilde\chi(|K_x|)=(-1)^{\tau(x)-1},
\]
于是
\[
W_x=-(-1)^{\tau(x)-1}=(-1)^{\tau(x)}.
\]
证毕。
\end{proof}

\begin{corollary}[纤维独立集多项式在 \texorpdfstring{$t=-1$}{t=-1} 的三值压缩]\label{cor:derived-fiber-independence-polynomial-minus-one-trichotomy}
\leanverified{paper\_derived\_fiber\_independence\_polynomial\_minus\_one\_trichotomy}
在定理 \ref{thm:derived-fiber-indcomplex-alternating-witten-parity} 的记号下，定义纤维独立集多项式
\[
I_x(t):=\sum_{S\in \mathsf{Ind}(\Gamma_x)} t^{|S|}.
\]
则有严格闭式
\[
I_x(-1)=
\begin{cases}
0, & d_m(x)\ \text{为偶数},\\[4pt]
(-1)^{\tau(x)}, & d_m(x)\ \text{为奇数}.
\end{cases}
\]
特别地，
\[
I_x(-1)\in\{0,\pm 1\},
\]
并且该值完全由 \(d_m(x)\) 的奇偶与 \(\tau(x)\) 的奇偶决定。因而在 \(t=-1\) 的 Lee--Yang 读数上，纤维尺度被完全压缩，只剩下可缩性与球面维数奇偶这两层拓扑信息。
\end{corollary}
\begin{proof}
由定义
\[
I_x(-1)=\sum_{S\in \mathsf{Ind}(\Gamma_x)}(-1)^{|S|}=W_x,
\]
故结论即由定理 \ref{thm:derived-fiber-indcomplex-alternating-witten-parity} 直接给出。证毕。
\end{proof}

\begin{theorem}[审计偶数窗口最小退化扇区的拓扑相位]\label{thm:derived-audited-even-minsector-topological-phase}
\leanverified{paper\_derived\_audited\_even\_minsector\_topological\_phase}
沿用定义 \ref{def:fold-bin-degeneracy-sectors} 与定理 \ref{thm:fold-bin-min-degeneracy-fib-audited} 的记号。对审计偶数窗
\[
m\in\{6,8,10,12\},
\]
记最小退化扇区
\[
\mathcal S_{m,\min}:=\mathcal S_{m,d_{\min}(m)}.
\]
则
\[
d_{\min}(m)=F_{m/2},
\qquad
|\mathcal S_{m,\min}|=F_m=|X_{m-2}|,
\]
并且对任意 \(x\in \mathcal S_{m,\min}\) 都有
\[
|K_x|\simeq
\begin{cases}
\ast, & m\equiv 0\pmod 6,\\[4pt]
S^0, & m\equiv 2,4\pmod 6.
\end{cases}
\]
更具体地，
\[
m=6,12:\ \text{全部最小退化纤维可缩},
\qquad
m=8,10:\ \text{全部最小退化纤维同伦于 }S^0.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:fold-bin-min-degeneracy-fib-audited}，
\[
d_{\min}(6)=2,\qquad d_{\min}(8)=3,\qquad d_{\min}(10)=5,\qquad d_{\min}(12)=8.
\]
再由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，纤维独立集复形是否可缩完全由纤维基数的奇偶决定：偶重时可缩，奇重时同伦于球面。

因此当 \(m=6,12\) 时，\(d_{\min}(m)\) 为偶数，故任意
\(
x\in \mathcal S_{m,\min}
\)
都满足
\[
|K_x|\simeq \ast.
\]
当 \(m=8,10\) 时，\(d_{\min}(m)\) 为奇数，故
\[
|K_x|\simeq S^{\tau(x)-1}.
\]
此时还需确定球面维数。若 \(d_m(x)=3\)，则由
\[
d_m(x)=\prod_i F_{\ell_i+2}
\]
并注意 \(3\) 在正 Fibonacci 值中仅出现为 \(F_4\)，故唯一可能是
\[
3=F_4,
\]
故 \(\Gamma_x\cong P_2\)，从而
\[
\tau(x)=\left\lfloor\frac{2+1}{3}\right\rfloor=1.
\]
若 \(d_m(x)=5\)，同理由于 \(5\) 在正 Fibonacci 值中仅出现为 \(F_5\)，唯一可能是
\[
5=F_5,
\]
故 \(\Gamma_x\cong P_3\)，从而
\[
\tau(x)=\left\lfloor\frac{3+1}{3}\right\rfloor=1.
\]
因此在 \(m=8,10\) 两种情形中均有
\[
|K_x|\simeq S^{\tau(x)-1}=S^0.
\]
证毕。
\end{proof}

\begin{theorem}[最大非可缩纤维的模 \texorpdfstring{$6$}{6} 缺陷常数]\label{thm:derived-max-noncontractible-fiber-defect-constants}
\leanverified{paper\_derived\_max\_noncontractible\_fiber\_defect\_constants}
沿用定义 \ref{def:pom-top-fiber-spectrum}、定义 \ref{def:pom-max-noncontractible-fiber} 与定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase} 的记号。记
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
\widetilde D_m:=\max\{d_m(x):x\in X_m,\ |K_x|\not\simeq \ast\}.
\]
则对所有 \(m\ge 17\)，\(\widetilde D_m/D_m\) 具有如下精确相图：
\[
\frac{\widetilde D_m}{D_m}=
\begin{cases}
1, & m\equiv 0,4\pmod 6,\\[8pt]
1-\dfrac{F_{k-4}}{2F_{k+1}}, & m=2k+1\equiv 1,5\pmod 6,\\[12pt]
1-\dfrac{F_{k-3}}{F_{k+2}}, & m=2k\equiv 2\pmod 6,\\[12pt]
1-\dfrac{F_{k-4}+F_{k-8}}{2F_{k+1}}, & m=2k+1\equiv 3\pmod 6.
\end{cases}
\]
从而沿各模 \(6\) 相位有极限
\[
\lim_{\substack{m\to\infty\\ m\equiv 0,4\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1,
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 1,5\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1-\frac{1}{2\varphi^{5}}
\approx 0.9549150281252629,
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 2\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1-\frac{1}{\varphi^{5}}
\approx 0.9098300562505258,
\]
\[
\lim_{\substack{m\to\infty\\ m\equiv 3\!\!\!\!\pmod 6}}
\frac{\widetilde D_m}{D_m}
=1-\frac{1}{2\varphi^{5}}-\frac{1}{2\varphi^{9}}
\approx 0.9483372193770504.
\]
\end{theorem}
\begin{proof}
由定理 \ref{thm:pom-max-noncontractible-fiber-mod6-phase}，
\[
\widetilde D_m=
\begin{cases}
D_m, & m\equiv 0,4\pmod 6,\\[2pt]
D_m^{(2)}, & m\equiv 1,5\pmod 6,\\[2pt]
D_m^{(3)}, & m\equiv 2,3\pmod 6.
\end{cases}
\]
再由定理 \ref{thm:pom-max-fiber}、定理 \ref{thm:pom-second-max-fiber-closed-form}、定理 \ref{thm:pom-third-max-fiber-even-closed-form} 与注 \ref{rem:pom-third-max-fiber-odd-closed-form}，
\[
D_{2k}=F_{k+2},\qquad D_{2k+1}=2F_{k+1},
\]
\[
D_{2k+1}^{(2)}=2F_{k+1}-F_{k-4},
\qquad
D_{2k}^{(3)}=F_{k+2}-F_{k-3},
\]
\[
D_{2k+1}^{(3)}=2F_{k+1}-F_{k-4}-F_{k-8}.
\]
将这些闭式逐项除以 \(D_m\) 即得有限尺度相图。

最后，由 Binet 渐近 \(F_n\sim \varphi^n/\sqrt5\) 可知
\[
\frac{F_{k-4}}{F_{k+1}}\to \varphi^{-5},
\qquad
\frac{F_{k-3}}{F_{k+2}}\to \varphi^{-5},
\qquad
\frac{F_{k-8}}{F_{k+1}}\to \varphi^{-9},
\]
代回即得三个非平凡极限常数。证毕。
\end{proof}

\begin{corollary}[最大非可缩纤维在指数尺度上无额外罚项]\label{cor:derived-max-noncontractible-fiber-no-exponential-penalty}
沿用定理 \ref{thm:derived-max-noncontractible-fiber-defect-constants} 的记号。则
\[
\lim_{m\to\infty}\frac1m\log \widetilde D_m
=
\lim_{m\to\infty}\frac1m\log D_m
=
\frac12\log\varphi.
\]
换言之，最大非可缩纤维与最大纤维在指数尺度上完全同阶；它们之间的全部差异都被压缩为定理 \ref{thm:derived-max-noncontractible-fiber-defect-constants} 给出的模 \(6\) 常数层相位，而不存在额外的指数级罚项。
\leanverified{paper\_derived\_max\_noncontractible\_fiber\_no\_exponential\_penalty}
\end{corollary}
\begin{proof}
由定理 \ref{thm:derived-max-noncontractible-fiber-defect-constants}，比值
\[
\frac{\widetilde D_m}{D_m}
\]
沿每个模 \(6\) 剩余类都收敛到严格正的有限常数。于是
\[
\frac1m\log \widetilde D_m
=
\frac1m\log D_m+\frac1m\log\!\left(\frac{\widetilde D_m}{D_m}\right),
\]
其中第二项趋于 \(0\)。故
\[
\lim_{m\to\infty}\frac1m\log \widetilde D_m
=
\lim_{m\to\infty}\frac1m\log D_m.
\]
再由定理 \ref{thm:pom-max-fiber} 的闭式
\[
D_{2k}=F_{k+2},
\qquad
D_{2k+1}=2F_{k+1},
\]
结合 Fibonacci 渐近 \(F_n\sim \varphi^n/\sqrt5\)，得到
\[
\lim_{m\to\infty}\frac1m\log D_m=\frac12\log\varphi.
\]
证毕。
\end{proof}

\begin{corollary}[指数最优球面纤维的 Lee--Yang 符号压缩]\label{cor:derived-optimal-spherical-fiber-sign-compression}
\leanverified{paper\_derived\_optimal\_spherical\_fiber\_sign\_compression}
对所有 \(m\ge 17\)，存在稳定类型 \(x_m\in X_m\) 使
\[
d_m(x_m)=\widetilde D_m,
\qquad
|K_{x_m}|\simeq S^{\tau(x_m)-1},
\qquad
I_{x_m}(-1)=(-1)^{\tau(x_m)}.
\]
并且
\[
\lim_{m\to\infty}\frac1m\log d_m(x_m)=\frac12\log\varphi.
\]
因此，指数尺度上最优的非可缩纤维并不是可缩例外，而是一族球面纤维；然而它们在 \(t=-1\) 的读数始终被压缩为纯符号。
\end{corollary}
\begin{proof}
由 \(\widetilde D_m\) 的定义，可取 \(x_m\in X_m\) 使
\[
d_m(x_m)=\widetilde D_m,
\qquad
|K_{x_m}|\not\simeq \ast.
\]
再由定理 \ref{thm:pom-fiber-parity-homotopy-equivalence}，
\[
|K_{x_m}|\simeq S^{\tau(x_m)-1}.
\]
推论 \ref{cor:derived-fiber-independence-polynomial-minus-one-trichotomy} 遂给出
\[
I_{x_m}(-1)=(-1)^{\tau(x_m)}.
\]
最后，由推论 \ref{cor:derived-max-noncontractible-fiber-no-exponential-penalty}，
\[
\lim_{m\to\infty}\frac1m\log d_m(x_m)
=
\lim_{m\to\infty}\frac1m\log \widetilde D_m
=
\frac12\log\varphi.
\]
证毕。
\end{proof}
