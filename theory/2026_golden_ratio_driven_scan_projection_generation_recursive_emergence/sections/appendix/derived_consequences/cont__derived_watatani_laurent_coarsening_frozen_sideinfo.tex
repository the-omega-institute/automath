\subsection{Watatani Laurent 矩塔、粗化单调性与冻结侧信息阈值}

正文与附录已经分别建立了三条彼此相连但尚未在同一处封口的接口：其一，折叠条件期望的 Watatani 指标元把正整数矩识别为碰撞矩；其二，二维定向 TQFT 的 genus 配分函数把负整数矩识别为同一中心元的 Laurent 迹；其三，固定分辨率的 Hankel--Prony 有限决定性把有限矩列恢复为完整纤维直方图。把这些接口与粗化、\(q\)-重测量--制备信道以及冻结 escort 输入下的逐纤维可逆化预算并读后，可进一步压缩出如下派生后果。

\begin{theorem}[粗化分割的 genus 相变单调律]\label{thm:derived-coarsening-genus-sign-law}
\leanverified{paper\_derived\_coarsening\_genus\_sign\_law}
固定分辨率 \(m\)，设
\[
(=)\ \subseteq\ \approx_1\ \subseteq\ \approx_2\ \subseteq\ (\sim)
\]
为定义在 \(\Omega_m\) 上的两条等价关系，并记对应等价类族为
\[
\Omega_m/\!\approx_i=\mathcal C_i.
\]
对每个 \(g\ge 0\) 定义 genus-\(g\) 分区和
\[
Z_{\approx_i}(\Sigma_g):=\sum_{C\in \mathcal C_i}|C|^{2-2g}.
\]
若 \(\approx_2\) 真粗于 \(\approx_1\)，则有严格符号分裂
\[
Z_{\approx_2}(\Sigma_0)>Z_{\approx_1}(\Sigma_0),
\]
\[
Z_{\approx_2}(\Sigma_1)<Z_{\approx_1}(\Sigma_1),
\]
\[
Z_{\approx_2}(\Sigma_g)<Z_{\approx_1}(\Sigma_g)
\qquad(g\ge 2).
\]
因此，类的合并对球面层给出严格增益，而对环面及更高 genus 层一律给出严格损耗；\(g=1\) 是唯一的符号翻转层。
\end{theorem}

\begin{proof}
任一从 \(\approx_1\) 到 \(\approx_2\) 的粗化都可分解为有限步“把两类 \(A,B\) 合并为 \(A\sqcup B\)”的操作，故只需处理单步合并。记
\[
a:=|A|,
\qquad
b:=|B|,
\qquad
\alpha_g:=2-2g.
\]
则分区和的改变量恰为
\[
\Delta_g:=(a+b)^{\alpha_g}-a^{\alpha_g}-b^{\alpha_g}.
\]
当 \(g=0\) 时，\(\alpha_0=2\)，故
\[
\Delta_0=(a+b)^2-a^2-b^2=2ab>0.
\]
当 \(g=1\) 时，\(\alpha_1=0\)，故
\[
\Delta_1=1-1-1=-1<0.
\]
当 \(g\ge 2\) 时，\(\alpha_g<0\)，函数 \(x\mapsto x^{\alpha_g}\) 在 \((0,\infty)\) 上严格递减，因此
\[
(a+b)^{\alpha_g}<a^{\alpha_g},
\qquad
(a+b)^{\alpha_g}<b^{\alpha_g},
\]
从而
\[
\Delta_g<0.
\]
把有限步粗化的改变量累加即可得到全部结论。证毕。
\end{proof}

\begin{theorem}[同一 Watatani 谱元同时生成 \(q\)-重环境塔与 genus Laurent 塔]\label{thm:derived-watatani-positive-laurent-towers}
固定分辨率 \(m\)，记折叠条件期望的中心 Watatani 指标元为
\[
\mathrm{Ind}_{\mathrm W}(E_m)=\sum_{x\in X_m}d_m(x)\,\widetilde P_x.
\]
则同一中心正元同时承载两座资源塔：
\begin{enumerate}
\normalfont
  \item 对每个整数 \(q\ge 2\)，\(q\)-重折叠测量--制备信道满足
  \[
  \rank J\!\left(\mathcal E_{\Fold_m}^{(q)}\right)
  =
  2^m\,\tau_m\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^{q-1}\right);
  \]
  \item 对每个 genus \(g\ge 0\)，二维定向 TQFT 配分函数满足
  \[
  Z_m(\Sigma_g)
  =
  2^m\,\tau_{m,1}\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^{1-2g}\right).
  \]
\end{enumerate}
因而正整数矩给出量子实现中的最小环境维数塔，而负整数矩给出同一谱元的 genus-Laurent 配分塔；二者不是独立对象族，而是同一中心谱的正/负矩切片。
\end{theorem}

\begin{proof}
由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision}，
\[
\rank J\!\left(\mathcal E_{\Fold_m}^{(q)}\right)=S_q(m).
\]
再由推论 \ref{cor:op-algebra-fold-watatani-index-moments}，
\[
\tau_m\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^{q-1}\right)=\frac{S_q(m)}{2^m}.
\]
两式合并即得第一条。

第二条正是命题 \ref{prop:conclusion-watatani-handle-identity-trace-moment} 的迹矩公式。证毕。
\end{proof}

\begin{theorem}[有限矩列对算子--拓扑--容量三联体的完备性]\label{thm:derived-finite-moments-complete-operator-topology-capacity}
固定分辨率 \(m\)。记纤维多重度的不同取值为
\[
1\le D_1<\cdots<D_r,
\qquad
n_i:=\#\{x\in X_m:\ d_m(x)=D_i\}.
\]
则有限矩列
\[
S_0(m),S_1(m),\dots,S_{2r-1}(m)
\]
唯一决定直方图
\[
\{(D_i,n_i)\}_{i=1}^r.
\]
从而它同时唯一决定下列全部对象：
\begin{enumerate}
\normalfont
  \item 同一中心元的全部整数 Watatani 矩
  \[
  \tau_m\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^s\right)
  \qquad (s\in \ZZ);
  \]
  \item 全部 \(q\)-重折叠信道的 Choi 秩
  \[
  \rank J\!\left(\mathcal E_{\Fold_m}^{(q)}\right)
  \qquad (q\ge 2);
  \]
  \item 全部二维 genus 配分函数
  \[
  Z_m(\Sigma_g)
  \qquad (g\ge 0);
  \]
  \item 全部预言机容量曲线与成功率曲线
  \[
  \mathcal C_m(B),
  \qquad
  \mathrm{Succ}_m(B)
  \qquad (B\ge 0).
  \]
\end{enumerate}
因此，在固定分辨率上，长度 \(2r\) 的有限矩列已经构成项目中“算子值资源、二维拓扑配分、可逆化侧信息预算”的完备有限证书。
\end{theorem}

\begin{proof}
由推论 \ref{cor:conclusion-fixedresolution-first-2sm-sufficiency}，首 \(2r\) 阶矩唯一决定全部谱点与重数；也即唯一决定直方图 \(\{(D_i,n_i)\}_{i=1}^r\)。

一旦该直方图已知，则对任意整数 \(s\in\ZZ\) 都有
\[
\tau_m\!\left(\mathrm{Ind}_{\mathrm W}(E_m)^s\right)
=
\frac1{2^m}\sum_{i=1}^{r}n_iD_i^{s+1},
\]
因而全部整数 Watatani 矩被唯一决定。又对每个 \(q\ge 2\)，
\[
\rank J\!\left(\mathcal E_{\Fold_m}^{(q)}\right)
=
\sum_{i=1}^{r}n_iD_i^q
\]
由定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 或定理 \ref{thm:derived-watatani-positive-laurent-towers} 唯一决定。对 genus 配分函数同理，
\[
Z_m(\Sigma_g)=\sum_{i=1}^{r}n_iD_i^{2-2g}
\]
由定理 \ref{thm:conclusion-fold-symtft-partition-function-collision-moments} 唯一决定。最后，对每个 \(B\ge 0\)，
\[
\mathcal C_m(B)=\sum_{i=1}^{r}n_i\min(D_i,2^B),
\qquad
\mathrm{Succ}_m(B)=\frac{\mathcal C_m(B)}{2^m},
\]
由定理 \ref{thm:fold-oracle-capacity-closed-form} 唯一决定。证毕。
\end{proof}

\begin{theorem}[冻结 escort 输入下的侧信息线性阈值]\label{thm:derived-frozen-escort-sideinfo-linear-threshold}
在定理 \ref{thm:fold-pressure-freezing-threshold} 与定理 \ref{thm:fold-escort-groundstate-concentration} 的假设下，取整数 \(a>a_{\mathrm c}\)，并记
\[
\pi_{m,a}(x):=\frac{d_m(x)^a}{S_a(m)},
\qquad
\mu_{m,a}(\omega):=\frac{\pi_{m,a}(\Fold_m(\omega))}{d_m(\Fold_m(\omega))}.
\]
设 \(\widetilde{\mathrm{Succ}}_{m,a}(B)\) 为在输入分布 \(\mu_{m,a}\) 下、允许 \(B\) 比特侧信息且要求解码永不越纤维时的最优成功率。则有统一上界
\[
\widetilde{\mathrm{Succ}}_{m,a}(B)
\le
\min\!\left(1,\frac{2^B}{M_m}\right)
+
\exp\!\bigl(-m\Delta(a)+o(m)\bigr),
\]
其中
\[
M_m:=\max_{x\in X_m}d_m(x),
\qquad
\Delta(a):=a(\alpha_\ast-\alpha_\ast^{(2)})-(\log\varphi-g_\ast)>0.
\]
特别地，若存在 \(\varepsilon>0\) 使
\[
B_m\le \left(\frac{\alpha_\ast}{\log 2}-\varepsilon\right)m
\qquad\text{eventually},
\]
则
\[
\widetilde{\mathrm{Succ}}_{m,a}(B_m)
\le
\exp\!\bigl(-\varepsilon\log 2\cdot m+o(m)\bigr)
+
\exp\!\bigl(-m\Delta(a)+o(m)\bigr),
\]
从而 \(\widetilde{\mathrm{Succ}}_{m,a}(B_m)\) 在指数尺度上趋于 \(0\)。
换言之，一旦冻结相输入已把质量压到最大纤维层，而位率仍严格低于
\[
R_{\mathrm{fr}}:=\frac{\alpha_\ast}{\log 2},
\]
则任何纤维内可逆化机制都不可能维持非退化成功率。
\end{theorem}

\begin{proof}
固定一个满足“解码永不越纤维”的 \(B\)-比特机制。对每个纤维 \(F_x:=\Fold_m^{-1}(x)\)，该机制至多能在 \(F_x\) 中可逆恢复
\[
\min(d_m(x),2^B)
\]
个微态。因此，在先按 \(\pi_{m,a}\) 抽样 \(x\)、再在 \(F_x\) 中均匀抽样的输入制度下，其条件成功率至多为
\[
\min\!\left(1,\frac{2^B}{d_m(x)}\right).
\]
对 \(x\) 取 \(\pi_{m,a}\)-期望，得到
\[
\widetilde{\mathrm{Succ}}_{m,a}(B)
\le
\sum_{x\in X_m}\pi_{m,a}(x)\min\!\left(1,\frac{2^B}{d_m(x)}\right).
\]
把求和拆到最大纤维集合
\[
A_m^{\max}:=\{x\in X_m:\ d_m(x)=M_m\}
\]
及其补集。对 \(A_m^{\max}\) 上每一点，积分核恒等于 \(\min(1,2^B/M_m)\)；对补集，积分核至多为 \(1\)。故
\[
\widetilde{\mathrm{Succ}}_{m,a}(B)
\le
\pi_{m,a}(A_m^{\max})\min\!\left(1,\frac{2^B}{M_m}\right)
+\bigl(1-\pi_{m,a}(A_m^{\max})\bigr)
\]
\[
\le
\min\!\left(1,\frac{2^B}{M_m}\right)
+\bigl(1-\pi_{m,a}(A_m^{\max})\bigr).
\]
再由定理 \ref{thm:fold-escort-groundstate-concentration}，
\[
1-\pi_{m,a}(A_m^{\max})
\le
\exp\!\bigl(-m\Delta(a)+o(m)\bigr),
\]
即得第一条上界。

若进一步
\[
B_m\le \left(\frac{\alpha_\ast}{\log 2}-\varepsilon\right)m
\qquad\text{eventually},
\]
则由 \(M_m=\exp(\alpha_\ast m+o(m))\) 可得
\[
\frac{2^{B_m}}{M_m}
\le
\exp\!\bigl(-\varepsilon\log 2\cdot m+o(m)\bigr).
\]
将其代回前述上界即得第二条。证毕。
\end{proof}

\begin{corollary}[真粗化下的 \(q\)-重环境资源严格膨胀]\label{cor:derived-coarsening-strictly-increases-qfold-environment-rank}
沿用定理 \ref{thm:derived-coarsening-genus-sign-law} 的记号。对每个等价关系 \(\approx\) 与每个 \(q>1\)，定义相应的 \(q\)-重测量--制备信道
\[
\mathcal E_{\approx}^{(q)}(\rho)
:=
\sum_{C\in \Omega_m/\!\approx}
\Tr(\Pi_C\rho)\,
\frac{\Pi_C^{\otimes(q-1)}}{|C|^{q-1}},
\qquad
\Pi_C:=\sum_{\omega\in C}\ket{\omega}\bra{\omega}.
\]
若 \(\approx_2\) 真粗于 \(\approx_1\)，则
\[
\rank J\!\left(\mathcal E_{\approx_2}^{(q)}\right)
>
\rank J\!\left(\mathcal E_{\approx_1}^{(q)}\right).
\]
因此，在不允许跨类泄漏的制度下，任何真实 coarse-graining 都会严格增加 \(q\)-重折叠信道的最小 Kraus 数，也即严格增加最小 Stinespring 环境维数。
\end{corollary}

\begin{proof}
与定理 \ref{thm:conclusion-qfold-channel-choi-rank-equals-qcollision} 完全同型的计算给出
\[
\rank J\!\left(\mathcal E_{\approx}^{(q)}\right)=\sum_{C\in \Omega_m/\!\approx}|C|^q.
\]
若 \(\approx_2\) 由 \(\approx_1\) 经过一步合并 \(A,B\mapsto A\sqcup B\) 获得，则秩差恰为
\[
(a+b)^q-a^q-b^q,
\qquad
a:=|A|,
\quad
b:=|B|.
\]
由于 \(q>1\) 时函数 \(x\mapsto x^q\) 在 \((0,\infty)\) 上严格凸，故
\[
(a+b)^q>a^q+b^q.
\]
对有限步粗化逐步迭代即得一般情形。证毕。
\end{proof}
