\begin{definition}[去 prime-powers 的 primitive（Perron 剥离）]\label{def:primitive-sharp}
沿用命题 \ref{prop:finite-rh-sqrt-resonance-general} 的记号，定义
\[
p_n^\sharp
:=\frac{1}{n}\sum_{d\mid n}\mu(d)\,\bigl(a_{n/d}-\lambda^{n/d}\bigr),
\qquad
\mathsf{E}_n^\sharp
:=\frac{n}{\lambda^{n/2}}\,p_n^\sharp.
\]
它等价于在 ghost 坐标中把 Perron 模态 \(\lambda^k\) 从 \((a_k)\) 中逐项剥离，再施加 Witt/Möbius 反演；因此 \(p_n^\sharp\) 是一个\textbf{纯非 Perron 谱驱动}的 primitive 观测量。
\end{definition}

\begin{proposition}[去 Perron 后的纯相位极限]\label{prop:primitive-sharp-phase-limit}
\leanverified{paper\_primitive\_sharp\_phase\_limit}
在命题 \ref{prop:finite-rh-sqrt-resonance-general} 的假设下，若临界模谱 \(\mathcal{C}\) 在 \(|\mu|=\sqrt{\lambda}\) 上半单，则当 \(n\to\infty\) 有
\[
p_n^\sharp
=\frac{\lambda^{n/2}}{n}\sum_{\mu\in\mathcal{C}}\omega_\mu^{\,n}
+O\!\left(\frac{\Lambda^n}{n}\right),
\qquad
\mathsf{E}_n^\sharp
=\sum_{\mu\in\mathcal{C}}\omega_\mu^{\,n}
+O\!\left(\Bigl(\frac{\Lambda}{\sqrt{\lambda}}\Bigr)^n\right),
\]
其中 \(\omega_\mu=\mu/\sqrt{\lambda}\) 且 \(\Lambda=\max\{R,\lambda^{1/3}\}\) 与命题 \ref{prop:finite-rh-sqrt-resonance-general} 相同。
\end{proposition}
\begin{proof}
由命题 \ref{prop:finite-rh-sqrt-resonance-general} 证明中的谱分解，
对任意 \(k\ge 1\) 有
\(
a_k-\lambda^k=\lambda^{k/2}\sum_{\mu\in\mathcal{C}}\omega_\mu^{\,k}+O(R^k)
\)。
代入定义得
\[
np_n^\sharp=\sum_{d\mid n}\mu(d)\,\bigl(a_{n/d}-\lambda^{n/d}\bigr)
=(a_n-\lambda^n)+\sum_{\substack{d\mid n\\ d\ge 2}}\mu(d)\,\bigl(a_{n/d}-\lambda^{n/d}\bigr).
\]
第一项等于 \(\lambda^{n/2}\sum_{\mu\in\mathcal{C}}\omega_\mu^{\,n}+O(R^n)\)。
对 \(d\ge 2\)，令 \(k=n/d\le n/2\)，则
\(
|a_k-\lambda^k|
\ll \lambda^{k/2}+R^k
\ll \lambda^{n/4}+R^{n/2}
\ll \Lambda^n
\)
（其中 \(\Lambda=\max\{R,\lambda^{1/3}\}\) 且 \(\Lambda<\sqrt{\lambda}\)）。
于是
\(
np_n^\sharp
=\lambda^{n/2}\sum_{\mu\in\mathcal{C}}\omega_\mu^{\,n}+O(\Lambda^n)
\)，
两边除以 \(n\) 并再归一化即得结论。证毕。
\end{proof}

\begin{remark}[多频准随机极限：\texorpdfstring{$\mathsf{E}_n^\sharp$}{E_n^sharp} 的极限分布]\label{rem:primitive-sharp-limit-law}
在命题 \ref{prop:primitive-sharp-phase-limit} 的假设下，设临界模谱（按代数重数计）为
\[
\mu_j=\sqrt{\lambda}\,e^{i\alpha_j}\qquad(j=1,\dots,J),
\]
并假设 \(\alpha_1,\dots,\alpha_J\) 在 \(2\pi\) 上有理无关（等价地 \(\{\alpha_j/2\pi\}\) 在 \(\QQ\) 上线性无关）。
令
\(
S_n:=\sum_{j=1}^{J}e^{in\alpha_j}
\)。
则 \((n\alpha_1,\dots,n\alpha_J)\bmod 2\pi\) 在 \(\mathbb{T}^J\) 上等分布（Kronecker/Weyl），从而对任意连续函数 \(f:\CC\to\CC\) 有 Ces\`aro 极限
\[
\lim_{N\to\infty}\frac{1}{N}\sum_{n=1}^{N} f(S_n)
=\int_{\mathbb{T}^J} f\!\left(\sum_{j=1}^{J}e^{i\theta_j}\right)\,d\theta,
\]
其中 \(d\theta\) 为 \(\mathbb{T}^J\) 的 Haar 概率测度。
由于命题 \ref{prop:primitive-sharp-phase-limit} 给出 \(\mathsf{E}_n^\sharp=S_n+o(1)\)，故 \(\mathsf{E}_n^\sharp\) 具有同一极限分布：它是 Haar 测度在映射 \((\theta_1,\dots,\theta_J)\mapsto\sum_{j=1}^{J}e^{i\theta_j}\) 下的推前测度。
\end{remark}

\begin{remark}[偶长层的“超强化”主次项]\label{rem:finite-rh-even-strengthen}
在偶长层，$\lambda^{n/2}$ 项由 $d=1$ 与 $d=2$ 精确抵消。进一步分情形可得更强的主次项：
\[
p_n=\frac{\lambda^n}{n}-\frac{\lambda^{n/3}}{n}+O\!\left(\frac{\lambda^{n/4}}{n}\right),
\qquad n\ \text{偶且}\ 3\mid n,
\]
\[
p_n=\frac{\lambda^n}{n}+(-1)^{n/2+1}\frac{\lambda^{n/4}}{n}
O\!\left(\frac{\lambda^{n/5}}{n}\right),
\qquad n\ \text{偶且}\ 3\nmid n.
\]
其中第一式的 $-\lambda^{n/3}$ 来自 $d=3$ 项（$\mu(3)=-1$），第二式的 $\lambda^{n/4}$ 来自 $d=2$ 项中的次级振荡
$(-\varphi)^{n/2}=(-1)^{n/2}\lambda^{n/4}$；而 $\mu(4)=0$ 使 $d=4$ 不贡献主次级项。
\end{remark}

\begin{remark}[次级项由“谱信息 $\times$ 因子结构”共同锁定]\label{rem:finite-rh-factor-controls-secondary}
在本核上，唯一满足 $|\mu|=\sqrt{\lambda}$ 的非平凡特征值为 $-\sqrt{\lambda}$，因此 primitive 侧的“平方根级误差”与“偶长层相消”都由临界模谱相位与 Möbius 反演的 $d=1,2$ 两项共同锁定（命题 \ref{prop:finite-rh-sqrt-resonance-general}，以及其特例推论 \ref{cor:finite-rh-explicit-formula}）。
一旦该层级被抵消，新的主次级项不再只由谱决定，还会显式依赖 $n$ 的因子结构：对 $n$ 的任意奇素因子 $p\mid n$，Möbius 反演中的 $d=p$ 项会贡献一个量级为 $\lambda^{n/p}/n$ 的修正（其符号由 $\mu(p)=-1$ 决定）。
\par
更紧凑地，定义
\[
p_{\min}(n):=\min\{p:\ p\ \text{为奇素数且}\ p\mid n\},
\qquad(\text{若无奇素因子则令 }p_{\min}(n)=+\infty).
\]
则对偶数 $n$，在 $d=1,2$ 的 $\lambda^{n/2}$ 级精确相消之后，剩余次级项的指数尺度由
\[
\max\Bigl(\frac14,\ \frac{1}{p_{\min}(n)}\Bigr)
\]
决定：当 $3\mid n$ 时由 $d=3$ 项给出 $\lambda^{n/3}/n$，当 $3\nmid n$ 时由 $d=2$ 项中的次级振荡给出 $\lambda^{n/4}/n$，而更大奇素因子只会产生更低阶的 $\lambda^{n/p}$ 修正。
\end{remark}

\begin{corollary}[RH-型误差的锐性（长度与 $x$ 口径）]\label{cor:finite-rh-prime-orbit-osc}
令
\[
\Pi(N):=\sum_{n\le N}p_n,\qquad x:=\lambda^N.
\]
由推论 \ref{cor:finite-rh-explicit-formula} 得
\[
\Pi(N)=\frac{\lambda}{\lambda-1}\cdot\frac{\lambda^N}{N}+O\!\left(\frac{\lambda^{N/2}}{N}\right),
\]
且由于 $(-\varphi)$ 的实际存在，上式中的平方根量级不可改进到 $\lambda^{(1/2-\varepsilon)N}$。等价地
\[
\pi(x)=\frac{\lambda\log\lambda}{\lambda-1}\cdot\frac{x}{\log x}
\;+\Omega_{\pm}\!\left(\frac{x^{1/2}}{\log x}\right),
\]
并且振荡频率由 $\Im(s)=\frac{(2k+1)\pi}{\log\lambda}$ 给出（周期 $2\pi/\log\lambda$）。
\end{corollary}

\begin{proof}
由上文的闭式
\(
a_n=\varphi^{2n}+(-\varphi)^n+2+3(-1)^n+\varphi^{-2n}+\varphi^{-n}
\)
可得
\(
a_n=\lambda^n+(-1)^n\lambda^{n/2}+O(1)
\)。

由 Möbius 反演 $p_n=\frac1n\sum_{d\mid n}\mu(d)a_{n/d}$：
若 $n$ 为奇数，则所有 $d\ge 2$ 满足 $d\ge 3$，故 $a_{n/d}=O(\lambda^{n/3})$。于是
\[
np_n=a_n+\sum_{\substack{d\mid n\\ d\ge 3}}\mu(d)a_{n/d}
=\lambda^n-\lambda^{n/2}+O(\lambda^{n/3}),
\]
得到第一式。

若 $n$ 为偶数，则写出 $d=1,2$ 两项：
\[
np_n=a_n-\;a_{n/2}+\sum_{\substack{d\mid n\\ d\ge 3}}\mu(d)a_{n/d}.
\]
此时 $a_n=\lambda^n+\lambda^{n/2}+O(1)$（因为 $(-1)^n=1$），而 $a_{n/2}=\lambda^{n/2}+O(\lambda^{n/4})$，故
\[
a_n-a_{n/2}=\lambda^n+O(\lambda^{n/4}),
\]
并且余项由 $d\ge 3$ 给出 $O(\lambda^{n/3})$，从而得到第二式。这里 $\lambda^{n/2}$ 的抵消正是 $a_n$ 的平方根项（$d=1$）与 $a_{n/2}$ 的主项（$d=2$）之间的精确抵消。
\end{proof}
