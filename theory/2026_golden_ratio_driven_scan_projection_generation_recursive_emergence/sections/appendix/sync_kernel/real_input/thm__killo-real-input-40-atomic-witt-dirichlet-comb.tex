\begin{theorem}[原子 Witt 因子在周期层给出精确的偶数长度 Dirichlet 梳齿]\label{thm:killo-real-input-40-atomic-witt-dirichlet-comb}
\leanverified{paper\_killo\_real\_input\_40\_atomic\_witt\_dirichlet\_comb}
沿用本小节的记号，写
\[
\zeta_{\mathrm{atom}}(z,u):=(1-uz^2)^{-1},
\qquad
\log \zeta_{\mathrm{atom}}(z,u)=\sum_{n\ge 1}\frac{a_n^{\mathrm{atom}}(u)}{n}z^n,
\]
并记其 primitive 生成函数为
\[
P_{\mathrm{atom}}(z,u)=\sum_{n\ge 1}p_n^{\mathrm{atom}}(u)z^n.
\]
则原子因子对 ghost 系数与 primitive 系数的贡献完全闭式：
\[
\boxed{
a_n^{\mathrm{atom}}(u)=
\begin{cases}
0,& n\ \text{为奇数},\\[2mm]
2u^{n/2},& n\ \text{为偶数},
\end{cases}
\qquad
p_n^{\mathrm{atom}}(u)=
\begin{cases}
u,& n=2,\\[2mm]
0,& n\neq 2.
\end{cases}}
\]
特别地，对任意整数 \(m\ge 2\) 与 \(\omega_m:=e^{2\pi i/m}\)，原子 ghost 系数在单位根扭曲平均下满足精确梳齿律
\[
\boxed{
\frac1m\sum_{j=0}^{m-1}a_n^{\mathrm{atom}}(\omega_m^j)=
\begin{cases}
2,& 2m\mid n,\\[2mm]
0,& 2m\nmid n.
\end{cases}}
\]
而 primitive 层上，原子部分仅在 \(n=2\) 处留下单一脉冲，除此之外完全消失。
\end{theorem}
\begin{proof}
由
\[
\zeta_{\mathrm{atom}}(z,u)=(1-uz^2)^{-1}
\]
得
\[
\log \zeta_{\mathrm{atom}}(z,u)
=
-\log(1-uz^2)
=
\sum_{k\ge 1}\frac{u^k z^{2k}}{k}
=
\sum_{k\ge 1}\frac{2u^k}{2k}z^{2k}.
\]
因此立得
\[
a_{2k}^{\mathrm{atom}}(u)=2u^k,
\qquad
a_{2k+1}^{\mathrm{atom}}(u)=0.
\]
另一方面，由命题 \ref{prop:atomic-2-orbit-split-logM} 的原子剥离，
\[
P(z,u)=uz^2+P_{\mathrm{core}}(z,u),
\]
故原子 primitive 生成函数正是
\[
P_{\mathrm{atom}}(z,u)=uz^2.
\]
于是
\[
p_2^{\mathrm{atom}}(u)=u,
\qquad
p_n^{\mathrm{atom}}(u)=0\quad(n\neq 2).
\]

再看单位根扭曲平均。若 \(n\) 为奇数，则 \(a_n^{\mathrm{atom}}(\omega_m^j)=0\) 对所有 \(j\) 成立。若 \(n=2\ell\) 为偶数，则
\[
\frac1m\sum_{j=0}^{m-1}a_n^{\mathrm{atom}}(\omega_m^j)
=
\frac{2}{m}\sum_{j=0}^{m-1}\omega_m^{j\ell}.
\]
单位根求和在且仅在 \(m\mid \ell\) 时等于 \(m\)，否则为 \(0\)。这恰等价于 \(2m\mid n\)。故得到所示 Dirichlet 梳齿公式。证毕。
\end{proof}
