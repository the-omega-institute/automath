\subsubsection{Abel--Mertens 接口：primitive 谱、主极点与有限部（通用闭式）}\label{app:abel-mertens-universal}
令 $A$ 为有限有向图（或 SFT）的邻接矩阵，并记
\[
a_n:=\Tr(A^n),\qquad
\zeta_A(z):=\exp\!\left(\sum_{n\ge 1}\frac{a_n}{n}z^n\right)=\frac{1}{\det(I-zA)}.
\]
令 $p_n$ 为长度为 $n$ 的 primitive 闭轨道数（按轨道计），并记 primitive 生成函数
\[
P(z):=\sum_{n\ge 1}p_n z^n.
\]

\begin{theorem}[primitive 对数接口与 Abel 有限部常数]\label{thm:abel-mertens-universal}
\leanverified{paper\_abel\_mertens\_universal}
有恒等式
\[
P(z)=\sum_{k\ge 1}\frac{\mu(k)}{k}\log\zeta_A(z^k)
\qquad(|z|\ \text{充分小}).
\]
进一步假设 $\zeta_A(z)$ 在 $z_\star$ 处有唯一主极点且为简单极点，并记
\[
\lambda:=\rho(A)=\frac{1}{z_\star},
\qquad
C:=\lim_{z\to z_\star}(1-\lambda z)\zeta_A(z).
\]
则存在 Abel/Hadamard 有限部常数 $\log\mathfrak{M}$ 使得
\[
P(z)=-\log(1-\lambda z)+\log\mathfrak{M}+o(1)\qquad(z\to z_\star),
\]
并且常数项满足闭式分解
\[
\log\mathfrak{M}
=\log C+\sum_{k\ge 2}\frac{\mu(k)}{k}\log\zeta_A(z_\star^{\,k}).
\]
若写 $\Delta_A(z):=\det(I-zA)$，则当 $z_\star$ 为 $\Delta_A$ 的简单零点时有
\[
C=\frac{1}{-z_\star\,\Delta_A'(z_\star)}=-\frac{\lambda}{\Delta_A'(z_\star)}.
\]
\end{theorem}

\begin{proof}
由 $\zeta_A(z)=\prod_{\tau\ \mathrm{primitive}}(1-z^{|\tau|})^{-1}$（闭轨道欧拉乘积）取对数得
\[
\log\zeta_A(z)=\sum_{\tau\ \mathrm{primitive}}\sum_{m\ge 1}\frac{z^{m|\tau|}}{m}
=\sum_{m\ge 1}\frac{P(z^m)}{m}.
\]
对上式作 Möbius 反演即得
\(
P(z)=\sum_{k\ge 1}\frac{\mu(k)}{k}\log\zeta_A(z^k)
\)。

在 $z\to z_\star$ 时，$\zeta_A(z)=\frac{C}{1-\lambda z}+O(1)$，故
\[
\log\zeta_A(z)=-\log(1-\lambda z)+\log C+o(1).
\]
将该展开代入 primitive 对数接口，并将 $k=1$ 项单独提出：当 $k\ge 2$ 时 $z^k\to z_\star^k$ 且 $|z_\star^k|<|z_\star|$，
因此 $\log\zeta_A(z^k)$ 在 $z=z_\star$ 处解析并收敛到 $\log\zeta_A(z_\star^k)$。于是得到
\[
P(z)=-\log(1-\lambda z)+\log C+\sum_{k\ge 2}\frac{\mu(k)}{k}\log\zeta_A(z_\star^{\,k})+o(1),
\]
从而有限部常数为所示的闭式分解。
最后，当 $\Delta_A(z)=\det(I-zA)$ 在 $z_\star$ 为简单零点时，
\(
\zeta_A(z)=1/\Delta_A(z)
\)
线性化即可得
\(
C=\lim_{z\to z_\star}(1-\lambda z)/\Delta_A(z)=1/(-z_\star\Delta_A'(z_\star))
\)。
\end{proof}

\begin{corollary}[Mertens 常数与 Abel 有限部常数]\label{cor:abel-mertens-constant-universal}
\leanverified{paper\_abel\_mertens\_constant\_universal}
令
\[
\mathcal{M}(N):=\sum_{n\le N}\frac{p_n}{\lambda^n}.
\]
则存在常数 $\mathsf{M}$ 使得
\[
\mathcal{M}(N)=\log N+\mathsf{M}+o(1),
\qquad
\boxed{\ \mathsf{M}=\gamma+\log\mathfrak{M}\ }.
\]
\end{corollary}

\begin{proof}
由主极点谱隙（次大特征值模 $<\lambda$）可得 $p_n=\lambda^n/n+O(\lambda_2^n/n)$，从而
\(
b_n:=p_n/\lambda^n-1/n
\)
绝对可和。于是
\[
\sum_{n\le N}\frac{p_n}{\lambda^n}
=\sum_{n\le N}\frac{1}{n}+\sum_{n\le N}b_n
=\log N+\gamma+\sum_{n\ge 1}b_n+o(1).
\]
另一方面，
\[
P(z)+\log(1-\lambda z)=\sum_{n\ge 1}\left(\frac{p_n}{\lambda^n}-\frac{1}{n}\right)(\lambda z)^n,
\]
故在 $z\to z_\star$（即 $\lambda z\to 1$）时，其 Abel 有限部常数正是 $\sum_{n\ge 1}b_n=\log\mathfrak{M}$。
\end{proof}

\paragraph{可计算配方（只用 $\zeta_A$ 与主留数）}

\begin{proposition}[Abel 有限部常数的加法分解：输入骨架 $+$ 垂直同步器]\label{prop:real-input-40-logM-split}

\[
\Delta_M(z):=\det(I-zM)=(1-z)^2(1+z)\,\Delta_B(z)\,\Delta_C(z),
\]
其中
\[
\Delta_B(z):=\det(I-zB)=(1+z)^2(1-3z+z^2),
\qquad
\Delta_C(z):=\det(I-zC)=1+z-z^2.
\]
记
\(
\zeta_M(z)=\Delta_M(z)^{-1}
\)
与主极点
\(
z_\star=\lambda^{-1}=\varphi^{-2}
\)
（由 $\Delta_B(z_\star)=0$ 给出，且 $z_\star\in(0,1)$）。
定义“输入骨架”与“垂直同步器”两部分
\[
\zeta_{\mathrm{in}}(z):=\Delta_B(z)^{-1},
\qquad
\zeta_{\mathrm{vert}}(z):=\bigl((1-z)^2(1+z)\Delta_C(z)\bigr)^{-1}.
\]
则 $\zeta_M=\zeta_{\mathrm{in}}\cdot \zeta_{\mathrm{vert}}$，并且 Abel 有限部常数满足严格恒等式
\[
\boxed{\ \log\mathfrak{M}_M=\log\mathfrak{M}_{\mathrm{in}}+P_{\mathrm{vert}}(z_\star)\ },
\]
 \leanverified{paper\_real\_input\_40\_logM\_split}
 \leanverified{paper\_real\_input\_40\_logm\_split}
其中 $\log\mathfrak{M}_M$ 与 $\log\mathfrak{M}_{\mathrm{in}}$ 分别由定理 \ref{thm:abel-mertens-universal} 应用于 $\zeta_M$ 与 $\zeta_{\mathrm{in}}$ 得到，而
\[
P_{\mathrm{vert}}(z):=\sum_{m\ge 1}\frac{\mu(m)}{m}\log \zeta_{\mathrm{vert}}(z^m)

在 $|z|<1$ 内解析，故 $P_{\mathrm{vert}}(z_\star)$ 为良定义的正实数对数。
\end{proposition}
\begin{proof}
由 $\zeta_M=\zeta_{\mathrm{in}}\zeta_{\mathrm{vert}}$，Möbius--$\log$ 算子 $P(z)=\sum_{m\ge 1}\mu(m)\log\zeta(z^m)/m$ 对乘积是线性的：

P_M=P_{\mathrm{in}}+P_{\mathrm{vert}}
\)。

\(
C_M=C_{\mathrm{in}}\cdot \zeta_{\mathrm{vert}}(z_\star)
\)。

\(
\log\mathfrak{M}=\log C+\sum_{m\ge 2}\mu(m)\log\zeta(z_\star^m)/m
\)，
并把 $m=1$ 项 $\log\zeta_{\mathrm{vert}}(z_\star)$ 与尾和合并，即得所示恒等式。证毕。
\end{proof}

\begin{corollary}[垂直同步器的“单一剩余常数”半闭式]\label{cor:real-input-40-vert-single-series}
\leanverified{paper\_real\_input\_40\_vert\_single\_series}
沿用命题 \ref{prop:real-input-40-logM-split} 的记号，并令
\[
S_F(z):=\sum_{m\ge 1}\frac{\mu(m)}{m}\log\bigl(1+z^m-z^{2m}\bigr)\qquad(|z|<1).
\]
则在 \(|z|<1\) 内有严格恒等式
\[
\boxed{\;
P_{\mathrm{vert}}(z)=z+z^2-S_F(z)\; }.
\]

P_{\mathrm{vert}}(z_\star)
\)
的非代数内容完全压缩为单一 Möbius--$\log$ 级数 \(S_F(z_\star)\)。
\end{corollary}
\begin{proof}
由 \(\zeta_{\mathrm{vert}}(z)=((1-z)^2(1+z)\Delta_C(z))^{-1}\) 与 \(\Delta_C(z)=1+z-z^2\)，有
\[
\log \zeta_{\mathrm{vert}}(z)
=-2\log(1-z)-\log(1+z)-\log(1+z-z^2).

将其代入
\(
P_{\mathrm{vert}}(z)=\sum_{m\ge 1}\mu(m)\log\zeta_{\mathrm{vert}}(z^m)/m

\[
P_{\mathrm{vert}}(z)

=z+z^2-S_F(z).
\]
证毕。
\end{proof}

\begin{theorem}[Fibonacci 张量闭包类：系统地产生核版 RH-sharp 核]\label{thm:fib-tensor-rhsharp-closure}
\leanverified{paper\_fib\_tensor\_rhsharp\_closure}
令
\[
F:=\begin{pmatrix}1&1\\[2pt]1&0\end{pmatrix},\qquad
B:=F\otimes F,\qquad
C:=-F.
\]
对任意 $n\ge 1$ 与非负整数 $a,b$，定义块对角矩阵
\[
M_{n;a,b}:=B^{\otimes n}\ \oplus\ C^{\otimes n}\ \oplus\ I_a\ \oplus\ (-I)_b.
\]
记
\(
\lambda_n:=\rho(M_{n;a,b})
\)。
则：
\begin{enumerate}
  \item \textbf{主尺度：}$\lambda_n=\rho(B^{\otimes n})=\rho(B)^n=\varphi^{2n}$。
  \item \textbf{核版 RH/GRH 的平方根界：}对任意非主特征值 $\mu\in\sigma(M_{n;a,b})$ 有
  \[
  |\mu|\le \sqrt{\lambda_n}=\varphi^{n},
  \]
  且该界在 $C^{\otimes n}$ 的主特征值 $\mu=(-\varphi)^n$ 上达到等号（因此为 RH-sharp）。
  \item \textbf{迹与 primitive 的闭式：}令 Lucas 数 $L_k:=\Tr(F^k)$。则对所有 $k\ge 1$，
  \[
  \Tr(M_{n;a,b}^k)
  =
  \Tr\!\bigl((B^{\otimes n})^k\bigr)+\Tr\!\bigl((C^{\otimes n})^k\bigr)+a+b(-1)^k
  =
  (L_k^2)^{n}+(-1)^{kn}L_k^{n}+a+b(-1)^k.
  \]
  相应 primitive 轨道数满足
  \[
  p_k(M_{n;a,b})=\frac{1}{k}\sum_{d\mid k}\mu(d)\,\Tr(M_{n;a,b}^{\,k/d}),
  \]
  从而得到一族完全显式、可审计的 Lucas--Möbius 闭式。
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(1)} 块对角的谱为并集，且 Kronecker 张量满足 $\rho(A\otimes B)=\rho(A)\rho(B)$，从而 $\rho(B^{\otimes n})=\rho(B)^n$。由 $\rho(F)=\varphi$ 得 $\rho(B)=\rho(F\otimes F)=\varphi^2$。

\textbf{(2)} 由 $\sigma(F)=\{\varphi,-\varphi^{-1}\}$ 得
\(
\sigma(B)=\sigma(F\otimes F)=\{\varphi^2,-1,-1,\varphi^{-2}\}
\)，
因此 $B^{\otimes n}$ 的特征值模长落在 $\{\varphi^{2n},1,\varphi^{-2n}\}$，均不超过 $\varphi^{n}=\sqrt{\varphi^{2n}}$。
又 $C=-F$ 的特征值为 $-\varphi$ 与 $\varphi^{-1}$，故 $C^{\otimes n}$ 的谱半径为 $\varphi^{n}$，并由 $(-\varphi)^n$ 达界。平凡块 $I_a,(-I)_b$ 的特征值模长均为 $1\le \varphi^{n}$。合并即得平方根界与 sharp 性。

\textbf{(3)} 利用 $\Tr((A\otimes B)^k)=\Tr(A^k)\Tr(B^k)$ 得 $\Tr((B^{\otimes n})^k)=\Tr(B^k)^n$ 与 $\Tr((C^{\otimes n})^k)=\Tr(C^k)^n$。
又 $\Tr(B^k)=\Tr((F\otimes F)^k)=\Tr(F^k)^2=L_k^2$，并且 $\Tr(C^k)=\Tr((-F)^k)=(-1)^k L_k$。加上平凡块的迹即可。
最后 primitive 闭式由 Möbius 反演直接给出。证毕。
\end{proof}
