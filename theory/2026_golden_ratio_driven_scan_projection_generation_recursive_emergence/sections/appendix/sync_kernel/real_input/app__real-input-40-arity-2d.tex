\subsubsection{真实输入下的二维 arity--负携带素数谱}\label{app:real-input-40-arity-2d}
\paragraph{定义（二维素数谱）}
定义
\[
\chi:=e-\mathbf{1}_{\{d=2\}}\in\{-1,0,1\},
\qquad
\nu:=\mathbf{1}_{\{t\in Q_-\}}\in\{0,1\},
\]
并对 primitive 周期轨道 $\gamma$ 令
\[
\chi(\gamma):=\sum_{k=1}^{|\gamma|}\chi_k,\qquad
N_-(\gamma):=\sum_{k=1}^{|\gamma|}\nu_k.
\]
二维 arity 生成多项式定义为
\[
P_n(q,r):=\sum_{\substack{\gamma\ \mathrm{primitive}\|\gamma|=n}} q^{\chi(\gamma)}r^{N_-(\gamma)}.
\]
对每条边赋权 $q^\chi r^\nu$ 得到加权矩阵 $M(q,r)$，记
\[
a_n(q,r):=\operatorname{tr}(M(q,r)^n),\qquad
\zeta(z;q,r)=\exp\!\left(\sum_{n\ge 1}\frac{a_n(q,r)}{n}z^n\right)
=\frac{1}{\det(I-zM(q,r))},
\]
则 Möbius 反演给出
\[
P_n(q,r)=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(q^d,r^d).
\]
当 $r=1$ 时退化到一维 $P_n(q)$。本例中前 $8$ 项为
\[
\begin{aligned}
P_1(q,r)&=1,\\
P_2(q,r)&=2qr+4q+r^{2},\\
P_3(q,r)&=4q,\\
P_4(q,r)&=q^{2}r^{2}+3q^{2}+2qr^{3}+4q+r^{2},\\
P_5(q,r)&=12q^{2}+5qr+4q+r^{2},\\
P_6(q,r)&=2q^{3}r^{3}+8q^{3}+4q^{2}r^{4}+18q^{2}+2qr^{5}+2qr^{3}+6qr^{2}+6qr+4q+r^{4},\\
P_7(q,r)&=36q^{3}+10q^{2}r^{2}+12q^{2}r+28q^{2}+7qr^{3}+12qr^{2}+6qr+4q+r^{4},\\
P_8(q,r)&=3q^{4}r^{4}+18q^{4}+8q^{3}r^{5}+84q^{3}+5q^{2}r^{6}+4q^{2}r^{4}+12q^{2}r^{3}\\
&\quad+30q^{2}r^{2}+36q^{2}r+34q^{2}+2qr^{7}+4qr^{5}+6qr^{4}+6qr^{3}+12qr^{2}+6qr+4q+r^{6}.
\end{aligned}
\]
\begin{corollary}[短素数轨道刚性 I：长度 \(3\) 的 primitive 完全同型]\label{cor:real-input-40-arity-2d-rigidity-n3}
\leanverified{paper\_real\_input\_40\_arity\_2d\_rigidity\_n3}
由 \(P_3(q,r)=4q\) 可知：长度 \(3\) 的 primitive 轨道恰有 \(4\) 条，并且全部满足
\[
\chi(\gamma)=1,\qquad N_-(\gamma)=0.
\]
\end{corollary}

\begin{corollary}[短素数轨道刚性 II：长度 \(2\) 的唯一“高负携带但零电荷”类型]\label{cor:real-input-40-arity-2d-rigidity-n2}
\leanverified{paper\_real\_input\_40\_arity\_2d\_rigidity\_n2}
由 \(P_2(q,r)=2qr+4q+r^{2}\) 可知：长度 \(2\) 的 primitive 轨道总数为 \(7\)，并且存在且仅存在 \(1\) 条轨道满足
\[
\chi(\gamma)=0,\qquad N_-(\gamma)=2.
\]
此外，其余轨道按 \((\chi,N_-)\) 的分布为：\(2\) 条满足 \((1,1)\)，\(4\) 条满足 \((1,0)\)。
\end{corollary}

\begin{corollary}[arity-charge 非负性与二维素数谱的 \texorpdfstring{$q$}{q}-非负幂]\label{cor:real-input-40-arity-2d-nonnegative}
\leanverified{paper\_real\_input\_40\_arity\_2d\_nonnegative}
在真实输入 40 状态核的 essential $20$ 状态核心上，对任意周期轨道（尤其 primitive）\(\gamma\)，都有
\[
\chi(\gamma)\ge 0.
\]
因此对一切 \(n\ge 1\)，二维 primitive 加权谱多项式满足
\[
P_n(q,r)\in\ZZ_{\ge 0}[q,r],
\]
从而不会出现 \(q^{-1},q^{-2}\) 等 Laurent 负幂。
\end{corollary}

\begin{proof}
由附录 \ref{app:real-input-40-arity-charge} 的定理 \ref{thm:real-input-40-arity-charge-coboundary}，存在 \(g(e)\in\{0,1\}\) 使得对任意周期轨道
\(\chi(\gamma)=\sum_{e\in\gamma}g(e)\ge 0\)。
二维权重只是在此基础上再乘以 \(r^{N_-(\gamma)}\)（非负指数），故 \(P_n(q,r)\) 必为 \(q,r\) 的非负幂多项式。证毕。
\end{proof}

\begin{remark}[到 \(n\le 8\) 的一致性核对（可审计）]\label{rem:real-input-40-arity-2d-nonneg-evidence}
在上述显式输出的 \(n\le 8\) 的所有 \(P_n(q,r)\) 中，\(q\) 的指数仅出现非负幂（未出现 \(q^{-1},q^{-2}\) 等项）。
该事实与推论 \ref{cor:real-input-40-arity-2d-nonnegative} 一致，可作为对生成/枚举实现的一致性核对；
其逐项可审计输出见脚本 \texttt{scripts/exp\_sync\_kernel\_real\_input\_40\_arity\_2d.py}（\texttt{P\_n\_2d\_terms}）。
\end{remark}
\paragraph{定义（二维 Dirichlet 分层与扭曲常数）}
对模 $(m_1,m_2)$ 的二维 Dirichlet 分层，记
\[
p_{n,(r_1,r_2)}^{(m_1,m_2)}
=\#\{\gamma:\ |\gamma|=n,\ \chi(\gamma)\equiv r_1\ (\mathrm{mod}\ m_1),\ N_-(\gamma)\equiv r_2\ (\mathrm{mod}\ m_2)\},
\]
则离散傅里叶反演给出
\[
p_{n,(r_1,r_2)}^{(m_1,m_2)}
=\frac{1}{m_1m_2}\sum_{j_1=0}^{m_1-1}\sum_{j_2=0}^{m_2-1}
\omega_{m_1}^{-j_1r_1}\omega_{m_2}^{-j_2r_2}\,P_n(\omega_{m_1}^{j_1},\omega_{m_2}^{j_2}).
\]
对非平凡扭曲 $(q,r)\neq(1,1)$，定义收敛常数
\[
C(q,r):=\sum_{n\ge 1}\frac{P_n(q,r)}{\lambda^n}.
\]
\paragraph{定理（扭曲谱半径下降与指数等分布）}
记
\[
\rho_{m_1,m_2}:=\max_{(j_1,j_2)\neq(0,0)}\rho\!\left(M(\omega_{m_1}^{j_1},\omega_{m_2}^{j_2})\right),
\qquad \omega_m=e^{2\pi i/m},
\]
则对任意同余类 $(r_1,r_2)$ 有
\[
p_{n,(r_1,r_2)}^{(m_1,m_2)}=\frac{p_n}{m_1m_2}+O\!\left(\frac{\rho_{m_1,m_2}^n}{n}\right).
\]

\paragraph{推论（二维 Dirichlet--Mertens 常数矩阵）}
定义
\[
C_{r_1,r_2}^{(m_1,m_2)}:=\frac{\mathsf{M}_{\varphi^2}}{m_1m_2}
+\frac{1}{m_1m_2}\sum_{(j_1,j_2)\neq(0,0)}\omega_{m_1}^{-j_1r_1}\omega_{m_2}^{-j_2r_2}
C(\omega_{m_1}^{j_1},\omega_{m_2}^{j_2}),
\]
则
\[
\sum_{n\le N}\frac{p_{n,(r_1,r_2)}^{(m_1,m_2)}}{\lambda^n}
=\frac{1}{m_1m_2}\log N+C_{r_1,r_2}^{(m_1,m_2)}+o(1).
\]

在 $(m_1,m_2)=(2,2)$ 下，二维 Mertens 和满足
\[
\sum_{n\le N}\frac{p_{n,(r_1,r_2)}^{(2,2)}}{\lambda^n}
=\frac{1}{4}\log N+C_{r_1,r_2}+o(1),
\]
其中
\[

\paragraph{算法协议（可审计计算）}
给定 $(m_1,m_2)$ 与截断阶 $N$，可按以下步骤复现上述常数矩阵：
\begin{itemize}
  \item 构造真实输入核状态集 $\Sigma=Q\times\{0,1\}^2$ 与确定性转移；
  \item 对给定 $(q,r)$ 构造加权矩阵 $M(q,r)$（边权为 $q^\chi r^\nu$），并计算 $a_n(q,r)=\mathrm{tr}(M(q,r)^n)$；
  \item 用 Möbius 反演从 $\{a_n(q,r)\}$ 提取 $P_n(q,r)$ 与扭曲常数 $C(q,r)$；
  \item 通过二维离散傅里叶反演得到 $C_{r_1,r_2}^{(m_1,m_2)}$，并计算最坏扭曲谱半径 $\rho_{m_1,m_2}$；
  \item 输出 $P_n(q,r)$、$C(q,r)$、$C_{r_1,r_2}^{(m_1,m_2)}$ 与 $\rho_{m_1,m_2}$。
\end{itemize}
数值复现见脚本 \texttt{scripts/exp\_sync\_kernel\_real\_input\_40\_arity\_2d.py}。
