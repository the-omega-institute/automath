\begin{proposition}[真实输入 40 状态核的非零谱与 primitive 轨道闭式]\label{prop:real-input-40-nonzero-spectrum-trace-primitive}
设 \(M\) 为附录 \ref{app:real-input-40-kernel} 的真实输入 \(40\) 状态核邻接矩阵，并沿用命题 \ref{prop:real-input-40-essential-reduction} 的 essential 约化与命题 \ref{prop:real-input-40-A0-tensor-hidden} 的三因子分解。
则 \(M\) 的全部非零特征值（按代数重数计）为
\[
\lambda\in\Bigl\{1\ \text{(重数 2)},\ -1\ \text{(重数 3)},\ \varphi^2,\ \varphi^{-2},\ -\varphi,\ \varphi^{-1}\Bigr\},
\]
其余特征值均为 \(0\)。
因此对任意 \(n\ge 1\) 有周期迹闭式
\[
\Tr(M^{n})
=\varphi^{2n}+\varphi^{-2n}+(-\varphi)^{n}+\varphi^{-n}+2+3(-1)^{n}.
\]
\end{proposition}

\begin{proof}
由命题 \ref{prop:real-input-40-essential-reduction}，\(\det(I-zM)=\det(I-zM_{\mathrm{ess}})\) 且 \(\Tr(M^n)=\Tr(M_{\mathrm{ess}}^n)\)；
又由命题 \ref{prop:real-input-40-A0-tensor-hidden} 有
\[
\det(I-zM)
=(1-z)^2(1+z)\,\det\!\bigl(I-z(A_0\otimes A_0)\bigr)\,\det(I-z(-A_0)).
\]
其中
\(
\det(I-z(A_0\otimes A_0))=(1+z)^2(1-3z+z^2)
\)
且
\(
\det(I-z(-A_0))=\det(I+zA_0)=1+z-z^2
\)。
故
\[
\det(I-zM)=(1-z)^2(1+z)^3(1-3z+z^2)(1+z-z^2).
\]
由 \(\det(I-zM)=\prod_i(1-z\lambda_i)\) 可知非零特征值为上述各因子零点的倒数，并按幂次给出代数重数：
\((1-z)^2\) 对应 \(1\)（重数 \(2\)），\((1+z)^3\) 对应 \(-1\)（重数 \(3\)），
\((1-3z+z^2)\) 的两根为 \(\varphi^{-2},\varphi^{2}\) 的倒数，故对应特征值为 \(\varphi^{2},\varphi^{-2}\)，
\((1+z-z^2)=0\) 等价于 \(z^2-z-1=0\) 给出根 \(z=\varphi,-\varphi^{-1}\)，故对应特征值为 \(\varphi^{-1},-\varphi\)。
其余特征值为 \(0\)。
最后迹等于全部特征值的 \(n\) 次幂之和，得到所示闭式。证毕。
\end{proof}

\begin{corollary}[primitive 轨道计数的 Möbius 反演与平方根误差]\label{cor:real-input-40-primitive-orbits-mobius-sqrt}
\leanverified{paper\_real\_input\_40\_primitive\_orbits\_mobius\_sqrt}
设 \(p_n\) 为 \(M\) 的 primitive 轨道计数（定义 \ref{def:primitive-sharp}）。
则对每个 \(n\ge 1\) 有显式公式
\[
p_n=\frac{1}{n}\sum_{d\mid n}\mu(d)\,\Tr\!\bigl(M^{n/d}\bigr),
\]
并且存在渐近
\[
p_n=\frac{\varphi^{2n}}{n}+O\!\Bigl(\frac{\varphi^{n}}{n}\Bigr)\qquad(n\to\infty),
\]
其中误差项的指数底由次大模特征值 \(|-\varphi|=\varphi\) 决定。
\end{corollary}

\begin{proof}
对有限有向图的动力学 \(\zeta\) 恒等式给出
\(
\log \zeta_M(z)=\sum_{n\ge 1}\frac{\Tr(M^n)}{n}z^n=\sum_{n\ge 1}\frac{p_n}{n}\sum_{k\ge 1}z^{nk}
\)，比较系数即得 Witt/Möbius 反演公式。
将命题 \ref{prop:real-input-40-nonzero-spectrum-trace-primitive} 的迹闭式代入，并按特征值模长排序：主项来自 \(\varphi^{2n}\)，其次为模长 \(\varphi\) 的项 \((-\varphi)^n\)，其余项均为 \(O(1)\) 或指数底更小；再除以 \(n\) 得所示渐近。证毕。
\end{proof}

\begin{corollary}[primitive 轨道的奇偶两相与平方根主振荡]\label{cor:real-input-40-primitive-orbits-parity-cancellation}
\leanverified{paper\_real\_input\_40\_primitive\_orbits\_parity\_cancellation}
沿用推论 \ref{cor:real-input-40-primitive-orbits-mobius-sqrt} 的记号。
则存在常数 \(C<\infty\) 使对所有 \(n\ge 1\) 有
\[
p_n=\frac{\varphi^{2n}}{n}-\mathbf 1_{\{n\ \mathrm{odd}\}}\frac{\varphi^{n}}{n}
O\!\Bigl(\frac{\varphi^{2n/3}}{n}\Bigr).
\]
特别地，模长为 \(\varphi\) 的“平方根级”谱项在偶数长度上被 Möbius 反演的 \(d=2\) 因子完全消去，而在奇数长度上保留为主振荡项。
\end{corollary}
\begin{proof}
由 Möbius 反演公式
\(
p_n=\frac1n\sum_{d\mid n}\mu(d)\Tr(M^{n/d})
\)
与命题 \ref{prop:real-input-40-nonzero-spectrum-trace-primitive} 的迹闭式，
将
\[
\Tr(M^m)=\varphi^{2m}+(-\varphi)^m+R_m,
\]
其中 \(R_m:=\varphi^{-2m}+\varphi^{-m}+2+3(-1)^m\)，满足 \(|R_m|\le C_0\)（与 \(m\) 无关）。
\par
对 \(n\) 为奇数时，求和中无 \(d=2\) 项，故
\[
p_n=\frac1n\Bigl(\varphi^{2n}-\varphi^n+\sum_{\substack{d\mid n\\ d\ge 3}}\mu(d)\varphi^{2n/d}\Bigr)
\;+\;
O\!\left(\frac{1}{n}\sum_{d\mid n}1\right).
\]
注意当 \(d\ge 3\) 时 \(\varphi^{2n/d}\le \varphi^{2n/3}\)，且因子数 \(d(n)=n^{o(1)}\)，从而
\(
\sum_{d\mid n,\ d\ge 3}\varphi^{2n/d}
=O(\varphi^{2n/3}n^{o(1)})
\)。
常数项同样被 \(n^{o(1)}\) 控制并吸收进 \(O(\varphi^{2n/3})\)。
得到奇数情形的展开。
\par
对 \(n\) 为偶数时，求和中出现 \(d=2\) 项且 \(\mu(2)=-1\)。
因此
\[
p_n=\frac1n\Bigl(\Tr(M^n)-\Tr(M^{n/2})+\sum_{\substack{d\mid n\\ d\ge 3}}\mu(d)\Tr(M^{n/d})\Bigr).
\]
由迹闭式，\(\Tr(M^n)\) 的 \(\varphi^n\) 项为 \((-\varphi)^n=+\varphi^n\)，而 \(\Tr(M^{n/2})\) 的 Perron 主项为 \(\varphi^{2\cdot(n/2)}=\varphi^n\)，二者精确抵消。
其余 \(d\ge 3\) 的 Perron 贡献至多 \(\varphi^{2n/3}\)，其余谱项的模长 \(\le\varphi\) 在 \(n/d\le n/3\) 的尺度上更小，故同样吸收到 \(O(\varphi^{2n/3})\)。
综上得到所示统一渐近。证毕。
\end{proof}

\begin{theorem}[负平方根模态的二因子消去律]\label{thm:primitive-odd-even-sqrt-cancellation}
设 \(M\in\ZZ_{\ge 0}^{N\times N}\) 为有限非负整数矩阵，谱半径 \(\rho=\rho(M)>1\)。
假设存在特征值
\[
\lambda_\star=-\sqrt{\rho}\in\mathrm{Spec}(M),
\]
并且 \(\rho\) 与 \(\lambda_\star\) 均为单根；
此外，对所有其他特征值 \(\lambda\neq \rho,\lambda_\star\) 有 \(|\lambda|<\sqrt{\rho}\)。
记 \(a_n:=\Tr(M^n)\)，并按定义 \ref{def:primitive-sharp} 令
\[
p_n:=\frac1n\sum_{d\mid n}\mu(d)\,a_{n/d}.
\]
则存在 \(\theta<\sqrt{\rho}\) 使得
\[
a_n=\rho^{n}+(-1)^n\rho^{n/2}+O(\theta^{n}),
\]
并且
\[
p_n=\frac{\rho^{n}}{n}-\mathbf 1_{\{n\ \mathrm{odd}\}}\frac{\rho^{n/2}}{n}
O\!\Bigl(\frac{\rho^{n/3}}{n}\Bigr).
\]
\end{theorem}
\begin{proof}
由 \(\rho,\lambda_\star\) 为单根并且其余谱严格落在 \(|\lambda|<\sqrt{\rho}\) 内，可取 \(\theta\in(0,\sqrt{\rho})\) 使
\(
a_n=\rho^n+\lambda_\star^n+O(\theta^n)
=\rho^n+(-1)^n\rho^{n/2}+O(\theta^n)
\)。
\par
将该展开代入 Möbius 反演
\(
p_n=\frac1n\sum_{d\mid n}\mu(d)a_{n/d}
\)
并分离 \(d=1\)、（当 \(2\mid n\) 时的）\(d=2\) 与 \(d\ge 3\) 三部分。
平方根级项 \((-1)^{n/d}\rho^{(n/d)/2}\) 仅在 \(d=1\) 与 \(d=2\) 两项上可能达到 \(\rho^{n/2}\) 的尺度；
当 \(n\) 为偶数时，\(d=1\) 给出 \(+\rho^{n/2}\)，而 \(d=2\) 的 Perron 主项给出 \(-\rho^{n/2}\)，二者抵消；
当 \(n\) 为奇数时，不存在 \(d=2\) 项且 \((-1)^n=-1\)，因此保留 \(-\rho^{n/2}\)。
\par
对 \(d\ge 3\) 的 Perron 项有 \(\rho^{n/d}\le \rho^{n/3}\)，其余谱项贡献为 \(O(\theta^{n/d})\) 且同样 \(\le O(\rho^{n/3})\)。
除以 \(n\) 并吸收因子数 \(d(n)=n^{o(1)}\) 即得所示误差界。证毕。
\end{proof}
