\subsubsection{二值跳跃的谱相似降维、饱和闭轨刚性与 \texorpdfstring{$\chi$}{chi}-扩张覆盖的本原性}\label{app:real-input-40-arity-charge-structural-rigidity}
本小节在附录 \ref{app:real-input-40-arity-charge} 已建立的
\(\chi=e-\mathbf{1}_{\{d=2\}}\)、
\(\chi\) 的 \(0/1\) 共边界正规形（定理 \ref{thm:real-input-40-arity-charge-coboundary}）
以及密度上界证书（定理 \ref{thm:real-input-40-arity-charge-density-bound}）之上，
抽取若干可直接引用的结构性推论：加权谱与 \(\zeta\) 的精确相似降维、饱和闭轨的紧边刚性判别、以及 \(\chi\)-扩张在所有模层上的强连通性。

\begin{proposition}[加权邻接矩阵的精确相似降维]\label{prop:real-input-40-arity-charge-similarity-reduction}
\leanverified{paper\_real\_input\_40\_arity\_charge\_similarity\_reduction}
沿用附录 \ref{app:real-input-40-arity-charge} 的记号。
在 essential $20$ 状态核心上，对任意 \(q>0\) 定义 \(\chi\)-赋权邻接矩阵
\[
M(q)_{uv}:=\sum_{e:u\to v} q^{\chi(e)}\qquad(u,v\in\Sigma_{\mathrm{ess}}).
\]
令 \(V:\Sigma_{\mathrm{ess}}\to\{0,1\}\)、\(g:E_{\mathrm{ess}}\to\{0,1\}\) 为定理 \ref{thm:real-input-40-arity-charge-coboundary} 的共边界正规形数据，
并定义两张 \(0/1\) 拆分矩阵
\[
(M_b)_{uv}:=\#\{e:u\to v\ \text{且}\ g(e)=b\},
\qquad b\in\{0,1\}.
\]
则对角矩阵 \(D(q):=\mathrm{diag}(q^{V(u)})_{u\in\Sigma_{\mathrm{ess}}}\) 满足
\[
\boxed{\;M(q)=D(q)\,\bigl(M_0+qM_1\bigr)\,D(q)^{-1}.\;}
\]
因而对任意 \(z\in\CC\) 与任意 \(n\ge 1\) 有
\[
\det(I-zM(q))=\det(I-z(M_0+qM_1)),
\qquad
\rho(M(q))=\rho(M_0+qM_1),
\qquad
\Tr(M(q)^n)=\Tr((M_0+qM_1)^n).
\]
\end{proposition}
\begin{proof}
由定理 \ref{thm:real-input-40-arity-charge-coboundary}，对每条边 \(e:u\to v\) 有
\(
\chi(e)=g(e)-V(v)+V(u)
\)。
故 \(q^{\chi(e)}=q^{V(u)}q^{-V(v)}q^{g(e)}\)。
对固定 \((u,v)\) 求和得到
\[
M(q)_{uv}
=q^{V(u)}q^{-V(v)}\sum_{e:u\to v}q^{g(e)}
=q^{V(u)}q^{-V(v)}(M_0+qM_1)_{uv},
\]
即所示相似关系。
其余结论由相似不变量与迹的共轭不变性直接推出。证毕。
\end{proof}

\begin{proposition}[密度上界的等号刚性与紧边子图]\label{prop:real-input-40-arity-charge-density-tight-rigidity}
在定理 \ref{thm:real-input-40-arity-charge-density-bound} 的设定中，
令 \(w(e):=1-2g(e)\in\{\pm1\}\)，并取其势函数证书 \(h:\Sigma_{\mathrm{ess}}\to\{-2,-1,0\}\) 满足
\[
w(e)+h(u)-h(v)\ge 0\qquad(e:u\to v).
\]
称边 \(e:u\to v\) 为 \emph{紧边}，若满足
\[
w(e)+h(u)-h(v)=0.
\]
则对任意闭路 \(\gamma\)（尤其 primitive）有
\[
\sum_{e\in\gamma} g(e)\le \frac{|\gamma|}{2},
\]
且等号成立当且仅当 \(\gamma\) 的每条边均为紧边。
在等号情形沿 \(\gamma\) 有步进律
\[
h(v)-h(u)=w(e)=1-2g(e),
\]
从而 \(g(e)=0\) 当且仅当 \(h\) 沿边上升 \(+1\)，而 \(g(e)=1\) 当且仅当 \(h\) 沿边下降 \(-1\)。
因此任何饱和闭路必为偶长度，并且其 \(h\)-轨迹在三层 \(\{-2,-1,0\}\) 上作平衡循环。
\end{proposition}
\begin{proof}
将 \(w(e)+h(u)-h(v)\ge 0\) 沿闭路 \(\gamma\) 求和，势函数项望远镜相消，得到
\[
0\le \sum_{e\in\gamma}w(e)=\sum_{e\in\gamma}(1-2g(e))=|\gamma|-2\sum_{e\in\gamma}g(e),
\]
即 \(\sum g(e)\le |\gamma|/2\)。
若取等，则一族非负项之和为零，故每项必为零，即闭路每边均为紧边；反向显然。
紧边条件等价于 \(h(v)-h(u)=w(e)=1-2g(e)\)，从而得到上升/下降判别与偶长度结论。证毕。
\end{proof}

\begin{proposition}[路径级欠账深度的精确判别]\label{prop:real-input-40-arity-charge-prefix-debt-structure}
对 essential core 上任意有限允许路径 \(\pi:u\leadsto v\)，成立精确恒等式
\[
\boxed{\;\chi(\pi)=\sum_{e\in\pi}g(e)+V(u)-V(v).\;}
\]
特别地：
\[
\chi(\pi)=-1
\quad\Longleftrightarrow\quad
\Bigl(\sum_{e\in\pi}g(e)=0\ \text{且}\ V(u)=0,\ V(v)=1\Bigr).
\]
\leanverified{paper\_real\_input\_40\_arity\_charge\_prefix\_debt\_structure}
\end{proposition}
\begin{proof}
由定理 \ref{thm:real-input-40-arity-charge-coboundary} 的共边界分解 \(\chi(e)=g(e)-V(v)+V(u)\)，沿路径求和即得第一式。
由于 \(g(e)\ge 0\) 且 \(V\in\{0,1\}\)，有 \(\chi(\pi)\ge -1\)，并且 \(\chi(\pi)=-1\) 当且仅当 \(\sum g(e)=0\) 且 \(V(u)-V(v)=-1\)，即 \(V(u)=0,V(v)=1\)。证毕。
\end{proof}

\begin{theorem}[\texorpdfstring{$\chi$}{chi}-扩张覆盖的本原性：整数层与模层均强连通]\label{thm:real-input-40-arity-charge-cover-primitive}
\leanverified{paper\_real\_input\_40\_arity\_charge\_cover\_primitive}
以 \(\chi\) 定义整数覆盖图 \(G_\chi\)：顶点为 \(\Sigma_{\mathrm{ess}}\times\ZZ\)，并对每条边 \(e:u\to v\) 引入边
\[
(u,n)\longrightarrow (v,n+\chi(e)).
\]
则 \(G_\chi\) 强连通。
并且对任意 \(m\ge 2\)，模覆盖 \(G_{\chi\bmod m}\)（顶点 \(\Sigma_{\mathrm{ess}}\times\ZZ/m\ZZ\)）亦强连通。
\end{theorem}
\begin{proof}
基图 \(G_{\mathrm{ess}}\) 强连通（命题 \ref{prop:real-input-40-essential-20}）。
覆盖图的强连通性等价于闭路权和集合生成的子群为全群：即
\(
\langle \chi(\gamma):\ \gamma\ \text{为闭路}\rangle=\ZZ
\)
（模 \(m\) 情形为 \(\ZZ/m\ZZ\)）。
由定理 \ref{thm:real-input-40-arity-charge-density-bound} 的“长度 \(2\) primitive 且 \(\chi(\gamma)=1\)”可达性，可知闭路权和的 \(\gcd\) 为 \(1\)，从而整数层与任意模层均不发生由周期性强迫的分裂。
故 \(G_\chi\) 与 \(G_{\chi\bmod m}\) 强连通。证毕。
\end{proof}
