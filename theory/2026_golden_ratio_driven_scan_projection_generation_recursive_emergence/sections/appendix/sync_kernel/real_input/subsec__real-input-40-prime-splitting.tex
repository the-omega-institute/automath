\subsubsection{真实输入核的 primitive 轨道谱精确拆分：\texorpdfstring{$M=(F\otimes F)\ \boxplus\ (-F)$}{M=(F⊗F) boxplus (-F)}}\label{subsec:real-input-40-prime-splitting}
命题 \ref{prop:real-input-40-fib-tensor} 已在 \(\zeta\)-层面给出因子分解
\(
\det(I-zM)=(1-z)^2(1+z)\det(I-z(F\otimes F))\det(I-z(-F))
\)。
本小节把该分解扩展到 primitive 轨道谱层：给出 \(p_n(M)\) 用 \(p_n(F\otimes F)\) 与 \(p_n(F)\) 的\textbf{精确差分公式}，并据此得到 Abel--Mertens 常数的一个快速收敛表达式。

\paragraph{记号}
令
\[
F:=\begin{pmatrix}1&1\\[2pt]1&0\end{pmatrix},\qquad
B:=F\otimes F,
\qquad
\lambda:=\rho(M)=\rho(B)=\varphi^2,
\qquad
z_\star:=\lambda^{-1}=\varphi^{-2}.
\]
对任意整数邻接矩阵 \(T\)，记 \(p_n(T)\) 为长度为 \(n\) 的 primitive 轨道数（按轨道计，满足 \(a_n(T)=\Tr(T^n)=\sum_{d\mid n} d\,p_d(T)\)）。
并记
\[
\pi_n:=p_n(F),\qquad r_n:=p_n(B)=p_n(F\otimes F).
\]

\begin{theorem}[Artin 型素谱拆分：\texorpdfstring{$p_n(M)$}{p_n(M)} 的精确差分]\label{thm:real-input-40-prime-artin-splitting}
对所有 \(n\ge 3\)，有精确恒等式
\[
\boxed{\;
p_n(M)=r_n+(-1)^n\,\pi_n+\mathbf{1}_{\{n\equiv 2\!\!\pmod 4\}}\ \pi_{n/2}\;}
\]
其中 \(\mathbf{1}_{\{n\equiv 2\!\!\pmod 4\}}\) 为指示函数。
等价地，对 \(n\ge 3\)：
\begin{itemize}
  \item 若 \(n\) 为奇数，则 \(p_n(M)=r_n-\pi_n\)；
  \item 若 \(4\mid n\)，则 \(p_n(M)=r_n+\pi_n\)；
  \item 若 \(n\equiv 2\pmod 4\)，则 \(p_n(M)=r_n+\pi_n+\pi_{n/2}\)。
\end{itemize}
此外低阶项满足
\[
p_1(M)=r_1,
\qquad
p_2(M)=r_2+\pi_2+\pi_1+1.
\]
\leanverified{paper\_real\_input\_40\_prime\_artin\_splitting}
\end{theorem}
\begin{proof}
由命题 \ref{prop:real-input-40-fib-tensor} 与迹的可加性（对块对角和按谱分解求和），可得对任意 \(n\ge 1\)
\[
a_n(M):=\Tr(M^n)
=\Tr(B^n)+\Tr((-F)^n)+2\cdot 1^n+(-1)^n.
\]
记 Lucas 数 \(L_n:=\Tr(F^n)\)（见 \ref{app:abel-mertens-universal} 小节的定义与式 \((\ref{app:abel-mertens-universal})\) 附近），则
\[
\Tr(B^n)=\Tr((F\otimes F)^n)=\Tr(F^n)^2=L_n^2,
\qquad
\Tr((-F)^n)=(-1)^n L_n.
\]
从而得到闭式
\[
a_n(M)=L_n^2+(-1)^n(L_n+1)+2.
\]
另一方面 \(a_n(B)=L_n^2\)，故定义“垂直修正迹”
\[
b_n:=a_n(M)-a_n(B)=2+(-1)^n+(-1)^nL_n.
\]
由 Möbius 反演（式 \(\,p_n=\frac1n\sum_{d\mid n}\mu(d)a_{n/d}\,\)，见 \ref{prop:real-input-40-essential-20} 小节末尾），有
\[
p_n(M)-r_n=\frac{1}{n}\sum_{d\mid n}\mu(d)\,b_{n/d}.
\]
把 \(b_m=2+(-1)^m+(-1)^mL_m\) 分三段处理：
\begin{itemize}
  \item 常数项 \(2\) 的 Möbius 变换只在 \(n=1\) 留下贡献；
  \item 纯符号项 \((-1)^m\) 的 Möbius 变换只在 \(n=1,2\) 留下贡献；
  \item 对 \(n\ge 3\)，只剩下
  \[
  \frac{1}{n}\sum_{d\mid n}\mu(d)(-1)^{n/d}L_{n/d}.
  \]
\end{itemize}
于是对 \(n\ge 3\)，只需证明
\[
\frac{1}{n}\sum_{d\mid n}\mu(d)(-1)^{n/d}L_{n/d}
=(-1)^n\pi_n+\mathbf{1}_{\{n\equiv 2\!\!\pmod 4\}}\pi_{n/2}.
\]

\textbf{(i) \(n\) 为奇数：}此时对所有 \(d\mid n\) 有 \(n/d\) 为奇数，故 \((-1)^{n/d}=-1\) 恒定，得到
\(
\frac{1}{n}\sum_{d\mid n}\mu(d)(-1)^{n/d}L_{n/d}=-\pi_n
\)，即结论。

\textbf{(ii) \(n=2m\) 为偶数：}由于 \(\mu(d)=0\) 当 \(4\mid d\)，在求和中只有 \(d\) 为奇数或 \(d=2d'\)（\(d'\) 奇数）两类项贡献。于是
\[
\frac{1}{2m}\sum_{d\mid 2m}\mu(d)(-1)^{2m/d}L_{2m/d}
=\frac{1}{2m}\sum_{\substack{d\mid m\\ d\ \mathrm{odd}}}\mu(d)\,L_{2m/d}
\ -\ \frac{1}{2m}\sum_{\substack{d\mid m\\ d\ \mathrm{odd}}}\mu(d)\,(-1)^{m/d}L_{m/d}.
\]
若 \(m\) 为偶数（即 \(4\mid n\)），则 \(m/d\) 仍为偶数，故 \((-1)^{m/d}=+1\)，上式恰等于
\(
\frac{1}{2m}\sum_{d\mid 2m}\mu(d)L_{2m/d}=\pi_{2m}
\)
（同样因为 \(d\) 含 \(4\) 的项被 \(\mu\) 消去）。若 \(m\) 为奇数（即 \(n\equiv 2\pmod 4\)），则 \(m/d\) 为奇数，故 \((-1)^{m/d}=-1\)，上式变为
\[
\frac{1}{2m}\sum_{\substack{d\mid m\\ d\ \mathrm{odd}}}\mu(d)\,L_{2m/d}
\ +\ \frac{1}{2m}\sum_{\substack{d\mid m\\ d\ \mathrm{odd}}}\mu(d)\,L_{m/d}
=\pi_{2m}+\pi_m.
\]
合并两种情形，即得对偶数 \(n\ge 4\) 的结论。最后把 \(n=1,2\) 的低阶修正单独列出即可。证毕。
\end{proof}

\begin{corollary}[空间/时间/边界三分解（逐项可核验）]\label{cor:real-input-40-space-twist-boundary-split}
\leanverified{paper\_real\_input\_40\_space\_twist\_boundary\_split}
沿用记号 \(F=\bigl(\begin{smallmatrix}1&1\\[2pt]1&0\end{smallmatrix}\bigr)\)、\(B=F\otimes F\)，并记 Lucas 数 \(L_n=\Tr(F^n)\)。
定义“边界短周期”与“时间扭曲块”的 primitive 系数为
\[
b_n:=\mathbf{1}_{\{n=1\}}+\mathbf{1}_{\{n=2\}},
\qquad
\tau_n:=p_n(-F)=\frac{1}{n}\sum_{d\mid n}\mu(d)\,(-1)^{n/d}L_{n/d}.
\]
则对所有 \(n\ge 1\) 都有严格恒等式
\[
\boxed{\; p_n(M)=p_n(B)+\tau_n+b_n\; }.
\]
并且 \(b_n=0\) 对一切 \(n\ge 3\)，因此在 \(n\ge 3\) 上
\(
p_n(M)-p_n(B)=p_n(-F)
\)
恰由扭曲块给出（见表 \ref{tab:real_input_40_primitive_artin_split} 的前 \(10\) 项）。
\end{corollary}

\begin{corollary}[素数长度的乘积律（可审计的加乘统一）]\label{cor:real-input-40-prime-length-product}
\leanverified{paper\_real\_input\_40\_prime\_length\_product}
设 \(p\) 为奇素数，记 Lucas 数 \(L_p=\Tr(F^p)\) 与 \(\pi_p=p_p(F)\)。则真实输入核在素数长度上满足
\[
\boxed{\;
p_p(M)=\frac{L_p(L_p-1)}{p}=L_p\cdot \pi_p\; }.
\]
\end{corollary}
\begin{proof}
对奇素数 \(p\)，由 \(a_p(M)=L_p^2- (L_p+1)+2=L_p^2-L_p+1\) 与 \(a_1(M)=1\) 得
\[
p_p(M)=\frac{1}{p}\bigl(a_p(M)-a_1(M)\bigr)=\frac{L_p^2-L_p}{p}=\frac{L_p(L_p-1)}{p}.
\]
另一方面 \(\pi_p=\frac{1}{p}(L_p-L_1)=\frac{L_p-1}{p}\)，故 \(p_p(M)=L_p\pi_p\)。证毕。
\end{proof}

\begin{remark}[素数长度的局部一维化]\label{rem:real-input-40-prime-local-1d}
推论 \ref{cor:real-input-40-prime-length-product} 表明：在奇素数长度上，真实输入核的 primitive 轨道数
\(
p_p(M)
\)
不携带额外的“高维自由度”，而是由单一 Lucas 迹参数 \(L_p=\Tr(F^p)\)（等价地：由 \(\pi_p=(L_p-1)/p\)）完全决定。
因此，除 Perron 主增长率所锁定的全局指数尺度之外，任何更细的复杂性只能经由合数长度的 Möbius 干涉与幂通道叠加进入（与定理 \ref{thm:real-input-40-prime-artin-splitting} 的模 \(4\) 选择律相容）。
\end{remark}

\begin{corollary}[primitive 生成函数的闭式恒等式]\label{cor:real-input-40-prime-genfun-identity}
令
\[
P_M(z):=\sum_{n\ge 1}p_n(M)z^n,\qquad
P_B(z):=\sum_{n\ge 1}r_n z^n,\qquad
P_F(z):=\sum_{n\ge 1}\pi_n z^n.
\]
则有恒等式
\[
\boxed{\;
P_M(z)-P_B(z)= z+z^2+P_F(-z)+\frac{P_F(z^2)-P_F(-z^2)}{2}\;}
\qquad(|z|\ \text{充分小}).
\]
\leanverified{paper\_real\_input\_40\_prime\_genfun\_identity}
\end{corollary}
\begin{proof}
比较系数即可。由定理 \ref{thm:real-input-40-prime-artin-splitting}，对 \(n\ge 3\) 有
\(
[z^n](P_M-P_B)=(-1)^n\pi_n+\mathbf{1}_{\{n\equiv 2\pmod 4\}}\pi_{n/2}
\)，而
\(
[z^n]P_F(-z)=(-1)^n\pi_n
\)。
另一方面，
\(
\frac{P_F(z^2)-P_F(-z^2)}{2}
\)
在 \(z^{2m}\) 处的系数为 \(\frac{1-(-1)^m}{2}\pi_m\)，仅当 \(m\) 为奇数（即 \(n\equiv 2\pmod 4\)）时等于 \(\pi_{n/2}\)，否则为 \(0\)。
最后 \(z+z^2\) 恰吸收 \(n=1,2\) 的低阶修正。证毕。
\end{proof}

\begin{corollary}[Abel--Mertens 常数的快速收敛表达（差分/三点值公式）]\label{cor:real-input-40-logM-fast-prime}
\leanverified{paper\_real\_input\_40\_logm\_fast\_prime}
沿用定理 \ref{thm:abel-mertens-universal} 的记号，\(\log\mathfrak{M}_M\) 与 \(\log\mathfrak{M}_{\mathrm{in}}\) 分别对应 \(M\) 与输入骨架 \(B=F\otimes F\)。
则差分满足绝对收敛级数
\[
\boxed{\;
\log\mathfrak{M}_M-\log\mathfrak{M}_{\mathrm{in}}
=\sum_{n\ge 1}\frac{p_n(M)-r_n}{\lambda^n}\;}
\]
并且由推论 \ref{cor:real-input-40-prime-genfun-identity} 得到三点值公式
\[
\boxed{\;
\log\mathfrak{M}_M-\log\mathfrak{M}_{\mathrm{in}}
=\frac{1}{\lambda}+\frac{1}{\lambda^2}
P_F\!\left(-\frac{1}{\lambda}\right)
\frac{
P_F\!\left(\frac{1}{\lambda^2}\right)-P_F\!\left(-\frac{1}{\lambda^2}\right)
}{2}\; }.
\]
\end{corollary}
\begin{proof}
由推论 \ref{cor:abel-mertens-constant-universal} 的证明（式 \(\log\mathfrak{M}=\sum_{n\ge 1}(p_n/\lambda^n-1/n)\)）可得
\[
\log\mathfrak{M}_M-\log\mathfrak{M}_{\mathrm{in}}
=\sum_{n\ge 1}\left(\frac{p_n(M)}{\lambda^n}-\frac{1}{n}\right)
-\sum_{n\ge 1}\left(\frac{r_n}{\lambda^n}-\frac{1}{n}\right)
=\sum_{n\ge 1}\frac{p_n(M)-r_n}{\lambda^n},
\]
其绝对收敛由 \(p_n(M)-r_n=O(\varphi^n/n)\) 与 \(\lambda=\varphi^2\) 给出。
将推论 \ref{cor:real-input-40-prime-genfun-identity} 的右端在 \(z=z_\star=\lambda^{-1}\) 处代入，即得三点值公式。证毕。
\end{proof}

\begin{remark}[端到端可审计算法：仅用 Lucas 迹样本]\label{rem:real-input-40-lucas-pipeline}
定理 \ref{thm:real-input-40-prime-artin-splitting} 与推论 \ref{cor:real-input-40-logM-fast-prime} 可组合为一条“最小数据依赖”的端到端编译管线：输入仅为 Lucas 迹样本 \(L_n=\Tr(F^n)\)（由递推 \(L_1=1,L_2=3,L_n=L_{n-1}+L_{n-2}\) 生成），即可在不显式枚举 primitive 轨道的前提下重建真实核 primitive 谱与 Abel 常数差。
\begin{enumerate}
  \item \textbf{(ghost\(\to\)Witt)} 用 Möbius/Witt 反演计算
  \[
  \pi_n=p_n(F)=\frac1n\sum_{d\mid n}\mu(d)\,L_{n/d},
  \qquad
  r_n=p_n(F\otimes F)=\frac1n\sum_{d\mid n}\mu(d)\,L_{n/d}^2.
  \]
  \item \textbf{(mod 4 选择律)} 用定理 \ref{thm:real-input-40-prime-artin-splitting} 计算 \(p_n(M)\)。
  \item \textbf{(三点取值)} 用推论 \ref{cor:real-input-40-logM-fast-prime} 计算
  \(\log\mathfrak{M}_M-\log\mathfrak{M}_{\mathrm{in}}\)。
\end{enumerate}
\par
\textbf{误差预算（显式尾项界）} 三点值公式只涉及 \(P_F(z)=\sum_{n\ge 1}\pi_n z^n\) 在
\(z\in\{-\lambda^{-1},\pm\lambda^{-2}\}\) 的取值。由于对任意 \(n\ge 1\) 有
\(
n\pi_n\le a_n(F)=L_n\le \varphi^n+1
\),
故 \(|z|<\varphi^{-1}\) 时成立可审计上界
\[
\sum_{n>N}\pi_n|z|^n
\le \sum_{n>N}\frac{(\varphi|z|)^n}{n}+\sum_{n>N}\frac{|z|^n}{n}
\le \frac{(\varphi|z|)^{N+1}}{(N+1)(1-\varphi|z|)}+\frac{|z|^{N+1}}{(N+1)(1-|z|)}.
\]
据此定义
\[
\mathrm{Tail}(N;|z|):=\frac{(\varphi|z|)^{N+1}}{(N+1)(1-\varphi|z|)}+\frac{|z|^{N+1}}{(N+1)(1-|z|)}.
\]
则把 \(P_F\) 在上述三点都截断到 \(N\) 项时，总误差满足
\[
\bigl|\mathrm{Err}_N\bigr|
\le \mathrm{Tail}(N;\lambda^{-1})+\mathrm{Tail}(N;\lambda^{-2}).
\]
由 \(\varphi\lambda^{-1}=\varphi^{-1}\)、\(\varphi\lambda^{-2}=\varphi^{-3}\) 得到
\(
N(\varepsilon)=O(\log(1/\varepsilon))
\)
的最小样本长度结论（因此只需 \(L_1,\dots,L_{N(\varepsilon)}\)）。
\par
\textbf{计算复杂度（筛实现）} 对给定 \(N\)，Möbius 反演可用 \(\mu(1),\dots,\mu(N)\) 的线性筛与除数卷积实现（逐 \(d\) 累加到倍数），从而生成 \(\{\pi_n\}_{n\le N}\)、\(\{r_n\}_{n\le N}\) 的时间复杂度为 \(O(N\log N)\)，空间为 \(O(N)\)；随后模 \(4\) 选择律与三点取值为线性时间。
\par
\textbf{可复现实现：}脚本 \texttt{scripts/exp\_real\_input\_40\_lucas\_pipeline.py} 实现上述管线，并同时审计：
(i) 模 \(4\) 选择律与 Möbius(trace) 计算的一致性；
(ii) 素数长度乘积律 \(p_p(M)=L_p\pi_p\)（奇素数 \(p\)）；
(iii) 三点值公式与级数 \(\sum_{n\ge 1}(p_n(M)-r_n)\lambda^{-n}\) 在尾项界内的一致性。
\end{remark}
