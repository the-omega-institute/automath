\begin{proposition}[\texorpdfstring{$p^k$}{p^k}-稀疏化：模 \texorpdfstring{$p^k$}{p^k} 下只保留 \texorpdfstring{$u^p$}{u^p}-子环]\label{prop:witt-pk-sparsification}
令 \(p\) 为素数、\(k\ge 1\)。将
\[
a_{p^k}(u)=\sum_{j\ge 0}a_{p^k,j}\,u^j\in\ZZ[u]
\]
按 \(u\) 的幂展开。由推论 \ref{cor:witt-dwork-congruence} 的同余
\(
a_{p^k}(u)\equiv a_{p^{k-1}}(u^p)\ (\mathrm{mod}\ p^k)
\)
逐系数比较得到：
\begin{enumerate}
  \item 若 \(p\nmid j\)，则 \(a_{p^k,j}\equiv 0\ (\mathrm{mod}\ p^k)\)；
  \item 若 \(j=p\ell\)，则
  \[
  a_{p^k,p\ell}\equiv a_{p^{k-1},\ell}\pmod{p^k}.
  \]
\end{enumerate}
等价地，在 \(\ZZ[u]/(p^k)\) 中有
\[
a_{p^k}(u)\equiv \sum_{\ell\ge 0}a_{p^{k-1},\ell}\,u^{p\ell}\in \bigl(\ZZ/p^k\ZZ\bigr)[u^p].
\]
\leanverified{paper\_witt\_pk\_sparsification\_seeds}
\end{proposition}
\begin{proof}
由推论 \ref{cor:witt-dwork-congruence} 知
\(
a_{p^k}(u)-a_{p^{k-1}}(u^p)\in p^k\ZZ[u]
\)。
又
\(
a_{p^{k-1}}(u^p)=\sum_{\ell\ge 0}a_{p^{k-1},\ell}\,u^{p\ell}
\)
只含 \(u^{p\ell}\) 项。逐幂次比较即可得到两条系数同余。证毕。
\end{proof}

\begin{proposition}[迭代下降：模 \texorpdfstring{$p^\ell$}{p^ell} 的 Frobenius 推前只依赖低层]\label{prop:witt-frobenius-iterated-descent}
令 \(p\) 为素数，\(k\ge 1\)，并取任意 \(1\le \ell\le k\)。
并约定 \(a_{p^0}(u):=a_1(u)\)。
在推论 \ref{cor:witt-dwork-congruence} 的 Dwork 型同余
\[
a_{p^j}(u)\equiv a_{p^{j-1}}(u^p)\pmod{p^j}\qquad(j\ge 1)
\]
下，在 \(\ZZ[u]\) 中有同余
\begin{equation}\label{eq:witt-frobenius-iterated-descent}
\boxed{\ a_{p^k}(u)\equiv a_{p^{\ell-1}}\!\bigl(u^{p^{k-\ell+1}}\bigr)\pmod{p^\ell}\ }.
\end{equation}
因此在商环 \((\ZZ/p^\ell\ZZ)[u]\) 中，\(a_{p^k}(u)\) 仅依赖于变量 \(u^{p^{k-\ell+1}}\)，即
\[
a_{p^k}(u)\bmod p^\ell\ \in\ (\ZZ/p^\ell\ZZ)\bigl[u^{p^{k-\ell+1}}\bigr].
\]
\end{proposition}
\begin{proof}
当 \(j\ge \ell\) 时，由 \(p^j\ZZ[u]\subset p^\ell\ZZ[u]\) 可将 Dwork 同余降模为
\(
a_{p^j}(u)\equiv a_{p^{j-1}}(u^p)\pmod{p^\ell}
\)。
从 \(j=k\) 起向下迭代至 \(j=\ell\) 得
\[
a_{p^k}(u)\equiv a_{p^\ell}\!\bigl(u^{p^{k-\ell}}\bigr)\pmod{p^\ell}.
\]
再对层级 \(j=\ell\) 应用同余（模 \(p^\ell\)）得到
\[
a_{p^\ell}\!\bigl(u^{p^{k-\ell}}\bigr)\equiv a_{p^{\ell-1}}\!\bigl(u^{p^{k-\ell+1}}\bigr)\pmod{p^\ell}.
\]
合并即得 \eqref{eq:witt-frobenius-iterated-descent}。
最后的子环包含性由右端只含 \(u^{p^{k-\ell+1}}\) 的幂次立得。证毕。
\end{proof}
\leanverified{paper\_witt\_frobenius\_iterated\_descent}

\begin{lemma}[偶次回文迹的 \texorpdfstring{$t=u+u^{-1}$}{t=u+u^{-1}} 不变量分解]\label{lem:trace-palindrome-even-invariant}
设 \(n\ge 2\) 为偶数，且 \(a_n(u)\in\ZZ[u]\) 满足回文对称
\(
a_n(u)=u^n a_n(1/u)
\)
（见命题 \ref{prop:trace-palindrome}）。
则存在唯一多项式 \(A_n(t)\in\ZZ[t]\) 使得在 Laurent 环 \(\ZZ[u,u^{-1}]\) 中成立
\[
\boxed{\ a_n(u)=u^{n/2}\,A_n\!\left(u+u^{-1}\right)\ }.
\]
\end{lemma}
\begin{proof}
令 \(t:=u+u^{-1}\)。
回文对称等价于 \(u^{-n/2}a_n(u)\) 在 \(u\mapsto 1/u\) 下不变，因此
\(
u^{-n/2}a_n(u)\in \ZZ[u,u^{-1}]^{\,u\mapsto 1/u}=\ZZ[u+u^{-1}]=\ZZ[t]
\)。
于是存在 \(A_n(t)\in\ZZ[t]\) 使 \(u^{-n/2}a_n(u)=A_n(t)\)，即得所示分解。
唯一性来自嵌入 \(\ZZ[t]\hookrightarrow \ZZ[u,u^{-1}]\) 的单射性（把 \(t\) 视为 \(u\) 的代数不变量）。证毕。
\end{proof}
\leanverified{paper\_trace\_palindrome\_even\_invariant}

\begin{theorem}[Chebyshev--Dwork 同余链：\texorpdfstring{$p=2$}{p=2} 的 Frobenius 在不变量坐标上闭合]\label{thm:chebyshev-dwork-congruence-chain}
令 \(a_n(u)=\Tr(B(u)^n)\in\ZZ[u]\) 为周期迹多项式，并令 \(t:=u+u^{-1}\)。
对每个 \(k\ge 1\)，令 \(A_{2^k}(t)\in\ZZ[t]\) 为引理 \ref{lem:trace-palindrome-even-invariant} 给出的不变量表示：
\[
a_{2^k}(u)=u^{2^{k-1}}A_{2^k}(t).
\]
则对所有 \(k\ge 2\)，由 Dwork 型同余 \(a_{2^k}(u)\equiv a_{2^{k-1}}(u^2)\ (\mathrm{mod}\ 2^k)\) 推出在 \(\ZZ[t]\) 内闭合的同余递推：
\[
\boxed{\ A_{2^k}(t)\equiv A_{2^{k-1}}(t^2-2)\pmod{2^k}\qquad\text{于 }\ZZ[t].\ }
\]
\end{theorem}
\begin{proof}
由推论 \ref{cor:witt-dwork-congruence}（取 \(p=2\)）有
\(
a_{2^k}(u)\equiv a_{2^{k-1}}(u^2)\ (\mathrm{mod}\ 2^k)
\) 于 \(\ZZ[u]\)。
对偶次 \(n=2^k\) 与 \(n=2^{k-1}\) 应用引理 \ref{lem:trace-palindrome-even-invariant}，写成
\[
a_{2^k}(u)=u^{2^{k-1}}A_{2^k}(t),\qquad
a_{2^{k-1}}(u^2)=(u^2)^{2^{k-2}}A_{2^{k-1}}(u^2+u^{-2}).
\]
注意恒等式
\(
u^2+u^{-2}=(u+u^{-1})^2-2=t^2-2
\)，故
\(
a_{2^{k-1}}(u^2)=u^{2^{k-1}}A_{2^{k-1}}(t^2-2)
\)。
将其代入 Dwork 同余并在 \(\ZZ[u,u^{-1}]\) 中消去共同因子 \(u^{2^{k-1}}\)，得到
\(
A_{2^k}(t)\equiv A_{2^{k-1}}(t^2-2)\ (\mathrm{mod}\ 2^k)
\) 于 \(\ZZ[t]\)。
证毕。
\end{proof}
\leanverified{paper\_chebyshev\_dwork\_congruence\_chain}

\begin{proposition}[\texorpdfstring{$\theta$}{theta}-导数的 \texorpdfstring{$p$}{p}-进滤波：逐阶可审计缩放律]\label{prop:witt-theta-derivative-filter}
\leanverified{paper\_witt\_theta\_derivative\_filter}
令 \(p\) 为素数、\(k\ge 1\)。设 \(a_n(u)\in\ZZ[u]\) 为周期迹多项式（主文与附录统一记号），并令 \(u=e^\theta\)。
定义在 \(\theta=0\) 处的导数矩（等价于“幂次矩”）
$$
A_{n}^{(r)}:=\left.\frac{\mathrm{d}^r}{\mathrm{d}\theta^r}\,a_n(e^\theta)\right|_{\theta=0}\in\ZZ,\qquad r\ge 0.
$$
则 Dwork 型同余 \(a_{p^k}(u)\equiv a_{p^{k-1}}(u^p)\ (\mathrm{mod}\ p^k)\) 蕴含逐阶缩放同余
$$
\boxed{\ A_{p^k}^{(r)}\equiv p^r\,A_{p^{k-1}}^{(r)}\pmod{p^k}\ }\qquad(r\ge 0).
$$
特别地，当 \(0\le r<k\) 时得到 \(p\)-进估值下界
$$
\boxed{\ p^{\,k-r}\ \big|\ A_{p^k}^{(r)}\ }.
$$
\end{proposition}
\begin{proof}
记 Euler 导数算子 \(D:=u\,\frac{\mathrm{d}}{\mathrm{d}u}\)，则
\(
\frac{\mathrm{d}}{\mathrm{d}\theta}f(e^\theta)=(Df)(u)\big|_{u=e^\theta}
\)
并由归纳得
\(
A_n^{(r)}=(D^r a_n)(1)\in\ZZ
\)（因为 \(D^r a_n\in\ZZ[u]\)）。
\par
对 Dwork 同余在 \(\ZZ[u]\) 中施加 \(D^r\) 仍保持模 \(p^k\)（因为 \(D\) 为 \(\ZZ\)-线性微分算子）。
另一方面，对任意 \(f\) 有恒等式
$$
D\bigl(f(u^p)\bigr)=p\,(Df)(u^p),
$$
从而 \(D^r\bigl(f(u^p)\bigr)=p^r\,(D^r f)(u^p)\)。
将其应用于 \(f=a_{p^{k-1}}\) 并在 \(u=1\) 处取值（此时 \(u^p=1\)），得到
$$
(D^r a_{p^k})(1)\equiv p^r (D^r a_{p^{k-1}})(1)\pmod{p^k},
$$
即所示的 \(A^{(r)}\) 同余。
当 \(r<k\) 时，右端含因子 \(p^r\)，故左端被 \(p^{k-r}\) 整除。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$u^p=1$}{u^p=1} 的塔式锁定：扭曲迹被推回无扭曲层]\label{cor:witt-root-of-unity-tower-lock}
令 \(p\) 为素数、\(k\ge 1\)，并令 \(R\) 为交换环。若 \(u\in R\) 满足 \(u^p=1\)，则
\[
a_{p^k}(u)\equiv a_{p^{k-1}}(1)\pmod{p^k}\qquad\text{于}\ R.
\]
更一般地，对任意 \(1\le \ell\le k\)，有
\[
\boxed{\ a_{p^k}(u)\equiv a_{p^{k-\ell}}(1)\pmod{p^\ell}\ }\qquad\text{于}\ R.
\]
\leanverified{paper\_witt\_root\_of\_unity\_tower\_lock}
\end{corollary}
\begin{proof}
由推论 \ref{cor:witt-dwork-congruence}，
\(
a_{p^k}(u)\equiv a_{p^{k-1}}(u^p)\ (\mathrm{mod}\ p^k)
\)。
在 \(u^p=1\) 下得 \(a_{p^{k-1}}(u^p)=a_{p^{k-1}}(1)\)，从而得到第一式。
对第二式，只需将同余在指数链上迭代 \(\ell\) 次：由于 \(u^{p^j}=1\)（\(j\ge 1\)），每一步都把参数 \(u^{p^j}\) 退化为 \(1\)，并且模数每次下降一阶，最终得到模 \(p^\ell\) 的锁定同余。证毕。
\end{proof}

\begin{theorem}[根单位点塔式锁定的同极限收敛]\label{thm:witt-root-of-unity-common-padic-limit}
\leanverified{paper\_witt\_root\_of\_unity\_common\_padic\_limit}
令 \(p\) 为素数，并假设对每个满足 \(u^p=1\) 的 unitary-slice 网格点，所有 \(a_{p^k}(u)\) 都落在 \(\ZZ\) 中。
则序列
\[
\bigl(a_{p^k}(1)\bigr)_{k\ge 0}
\]
在 \(\ZZ_p\) 中收敛于某个极限 \(L_p\in\ZZ_p\)。
进一步，对任意满足 \(u^p=1\) 的此类网格点，序列
\[
\bigl(a_{p^k}(u)\bigr)_{k\ge 0}
\]
亦在 \(\ZZ_p\) 中收敛到同一极限 \(L_p\)，并且对每个 \(k\ge 1\) 成立一致同余
\[
a_{p^k}(u)\equiv L_p\pmod{p^{k-1}}
\qquad\text{于 }\ZZ_p.
\]
\end{theorem}
\begin{proof}
对推论 \ref{cor:witt-dwork-congruence} 取 \(u=1\) 得
\[
a_{p^k}(1)\equiv a_{p^{k-1}}(1)\pmod{p^k}\qquad(k\ge 1).
\]
因此对任意 \(r>s\) 有
\[
a_{p^r}(1)-a_{p^s}(1)
=\sum_{j=s+1}^{r}\bigl(a_{p^j}(1)-a_{p^{j-1}}(1)\bigr)\in p^{s+1}\ZZ_p,
\]
故 \(\bigl(a_{p^k}(1)\bigr)_{k\ge 0}\) 为 \(\ZZ_p\) 中的 Cauchy 列，从而收敛到某个 \(L_p\in\ZZ_p\)。

现固定满足 \(u^p=1\) 的网格点 \(u\)。
由推论 \ref{cor:witt-root-of-unity-tower-lock}，
\[
a_{p^k}(u)-a_{p^{k-1}}(1)\in p^k\ZZ_p.
\]
另一方面，由上式的 Cauchy 性立刻得到
\[
a_{p^{k-1}}(1)-L_p\in p^{k-1}\ZZ_p.
\]
于是
\[
a_{p^k}(u)-L_p
=\bigl(a_{p^k}(u)-a_{p^{k-1}}(1)\bigr)
+\bigl(a_{p^{k-1}}(1)-L_p\bigr)
\in p^k\ZZ_p+p^{k-1}\ZZ_p
=p^{k-1}\ZZ_p.
\]
这既给出所述一致同余，也表明 \(\bigl(a_{p^k}(u)\bigr)_{k\ge 0}\) 收敛到同一极限 \(L_p\)。
证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$\theta$}{theta}-导数的稳定首位残差：\texorpdfstring{$\alpha_{p,r}\in\mathbb{F}_p$}{alpha in F_p}]\label{cor:witt-theta-stable-residue}
\leanverified{paper\_witt\_theta\_stable\_residue}
固定素数 \(p\) 与阶 \(r\ge 1\)。当 \(k>r\) 时，由命题 \ref{prop:witt-theta-derivative-filter} 有 \(p^{k-r}\mid A_{p^k}^{(r)}\)，从而可定义
\[
\alpha_{p,r}:=\left[p^{-(k-r)}A_{p^k}^{(r)}\right]\bmod p\ \in \mathbb{Z}/p\mathbb{Z}\cong\mathbb{F}_p.
\]
则 \(\alpha_{p,r}\) 与 \(k\) 的选择无关（只要 \(k>r\)）。
\end{corollary}
\begin{proof}
由命题 \ref{prop:witt-theta-derivative-filter} 的缩放同余，
\(
A_{p^k}^{(r)}=p^rA_{p^{k-1}}^{(r)}+p^k t
\)
对某个 \(t\in\ZZ\) 成立。两边除以 \(p^{k-r}\) 得
\[
p^{-(k-r)}A_{p^k}^{(r)}=p^{-(k-1-r)}A_{p^{k-1}}^{(r)}+p^{r}t.
\]
对模 \(p\) 取剩余，由 \(r\ge 1\) 知最后一项 \(p^r t\equiv 0\pmod p\)，故
\(
p^{-(k-r)}A_{p^k}^{(r)}\equiv p^{-(k-1-r)}A_{p^{k-1}}^{(r)}\pmod p
\)。
迭代即得 \(\alpha_{p,r}\) 在 \(k>r\) 区间内稳定。证毕。
\end{proof}

\begin{corollary}[\texorpdfstring{$p$}{p}-进平坦球：\texorpdfstring{$u=e^\theta$}{u=exp(theta)} 在 \texorpdfstring{$u=1$}{u=1} 附近的 \texorpdfstring{$p^k$}{p^k}-不敏感性]\label{cor:witt-padic-flatness-ball}
令 \(p\) 为素数、\(k\ge 1\)。取 \(\theta\in p\ZZ_p\)；当 \(p=2\) 时进一步假设 \(\theta\in 4\ZZ_2\)（等价于 \(v_2(\theta)\ge 2\)，以确保 \(e^\theta\) 收敛），并令 \(u=e^\theta\)。
则在 \(\ZZ_p\) 中有
\[
\boxed{\ a_{p^k}(e^\theta)\equiv a_{p^k}(1)\pmod{p^k}\ }.
\]
\leanverified{paper\_witt\_padic\_flatness\_ball}
\end{corollary}
\begin{proof}
当 \(k=1\) 时，由 \(v_p(\theta)\ge 1\) 知对任意 \(r\ge 1\) 有
\(
v_p(\theta^r/r!)=r\,v_p(\theta)-v_p(r!)\ge r-v_p(r!)\ge 1
\)，
故 Taylor 展开中的每个 \(r\ge 1\) 项都落在 \(p\ZZ_p\) 中，从而
\(
a_p(e^\theta)\equiv a_p(1)\pmod p
\)。
下设 \(k\ge 2\)。

在 \(\theta=0\) 处作 Taylor 展开：
\[
a_{p^k}(e^\theta)=\sum_{r\ge 0}\frac{A_{p^k}^{(r)}}{r!}\,\theta^r.
\]
常数项为 \(a_{p^k}(1)=A_{p^k}^{(0)}\)。对非零阶项分两类估值：
\begin{itemize}
  \item \textbf{\(r=1\)：}由命题 \ref{prop:witt-theta-derivative-filter}（取 \(r=1\)）得 \(p^{k-1}\mid A_{p^k}^{(1)}\)，又 \(v_p(\theta)\ge 1\)，故
  \(v_p(A_{p^k}^{(1)}\theta)\ge (k-1)+1=k\)。
  \item \textbf{\(r\ge 2\)：}由同一缩放同余在 \(k=2\) 处知
  \(A_{p^2}^{(r)}\equiv p^rA_{p}^{(r)}\equiv 0\pmod{p^2}\)，从而 \(p^2\mid A_{p^2}^{(r)}\)。
  再对 \(k\ge 3\) 归纳：若 \(p^{k-1}\mid A_{p^{k-1}}^{(r)}\)，则
  \(A_{p^k}^{(r)}\equiv p^rA_{p^{k-1}}^{(r)}\equiv 0\pmod{p^k}\)，故 \(p^k\mid A_{p^k}^{(r)}\)。
  于是对所有 \(k\ge 2\) 与 \(r\ge 2\) 都有 \(A_{p^k}^{(r)}\in p^k\ZZ\)。又因 \(v_p(\theta)\ge 1\) 蕴含 \(v_p(\theta^r/r!)\ge 0\)，故每项 \(\frac{A_{p^k}^{(r)}}{r!}\theta^r\) 的 \(p\)-进估值均不小于 \(k\)。
\end{itemize}
综上，所有非零阶项都被 \(p^k\) 整除，因此 \(a_{p^k}(e^\theta)\equiv a_{p^k}(1)\pmod{p^k}\)。证毕。
\end{proof}

\begin{remark}[模 \texorpdfstring{$p^k$}{p^k} 的计算加速律]\label{rem:witt-root-of-unity-acceleration}
在推论 \ref{cor:witt-root-of-unity-tower-lock} 的条件 \(u^p=1\) 下，模 \(p^k\) 的值
\(
a_{p^k}(u)\bmod p^k
\)
被完全锁定为
\(
a_{p^{k-1}}(1)\bmod p^k
\)：
因此对“非平凡角色扭曲”在 \(p\)-幂长度层的 \(\bmod p^k\) 扫描，不必显式计算高次矩阵幂
\(
B(u)^{p^k}
\)；
只需在无扭曲点 \(u=1\) 计算较短长度的迹 \(a_{p^{k-1}}(1)\)，并用塔式锁定把该值提升回 \(p^k\) 层即可。
\end{remark}

\begin{definition}[Frobenius 缺陷多项式]\label{def:witt-frobenius-defect}
对素数 \(p\) 与 \(k\ge 1\)，定义
\[
\Delta_{p,k}(u):=a_{p^k}(u)-a_{p^{k-1}}(u^p)\in\ZZ[u].
\]
\end{definition}

\leanverified{paper\_witt\_frobenius\_defect\_divisibility}
\begin{proposition}[Frobenius 缺陷的整除深度：\texorpdfstring{$\Delta_{p,k}(u)\in p^k\ZZ[u]$}{Delta in p^k Z[u]}]\label{prop:witt-frobenius-defect-divisibility}
对任意素数 \(p\) 与 \(k\ge 1\)，有
\[
\boxed{\ \Delta_{p,k}(u)\in p^k\,\ZZ[u]\ },
\]
等价地，\(\Delta_{p,k}(u)\) 的全部系数都被 \(p^k\) 整除。
\end{proposition}
\begin{proof}
这正是推论 \ref{cor:witt-dwork-congruence} 的重述：同余
\(
a_{p^k}(u)\equiv a_{p^{k-1}}(u^p)\ (\mathrm{mod}\ p^k)
\)
按定义等价于 \(\Delta_{p,k}(u)\in p^k\ZZ[u]\)。证毕。
\end{proof}

\begin{remark}[整数层审计与谱层指标的分层]\label{rem:witt-frobenius-defect-audit}
在任何一次可复现计算中，可先对一组 \((p,k)\) 检查 \(\Delta_{p,k}(u)\) 在 \(\bmod p^k\) 下的消失（或其系数整除深度），以确认“primitive--periodic 的 Big-Witt 桥包”在实现层没有破坏。
在该整数层审计通过之后，再讨论单位圆扭曲谱半径（例如 \(\rho/\lambda\)）的偏移，才具有可解释的结构意义；否则应优先定位为实现/数据管线层的失败。
\end{remark}
