\begin{proposition}[表示级自对偶：一维角色扭曲的函数方程]\label{prop:kernel-self-dual-character}
\leanverified{paper\_kernel\_self\_dual\_character}
沿用 \S\ref{app:kernel-self-dual} 的记号。设 \(G\) 为有限阿贝尔群，\(\chi:G\to\CC^\times\) 为一维角色，且假设群标签只依赖输出位 \(e\in\{0,1\}\)：对 \(e=0\) 取标签为单位元、对 \(e=1\) 取某固定元 \(g_1\in G\)。
则相应的角色扭曲矩阵可写成
\[
B_\chi(u)=B_0+u\,\chi(g_1)\,B_1.
\]
在自对偶对称 \(\Pi^{-1}B_0\Pi=B_1,\ \Pi^{-1}B_1\Pi=B_0\) 下，有相似恒等式
\[
\boxed{\;
\Pi^{-1}B_\chi(u)\Pi
=u\,\chi(g_1)\,B_{\chi^{-1}}(1/u)\; }.
\]
因此行列式与 \(\zeta\)-函数满足表示级函数方程
\[
\boxed{\;
\Delta_\chi(z,u):=\det(I-zB_\chi(u))
=\Delta_{\chi^{-1}}\!\bigl(u\,\chi(g_1)\,z,\ 1/u\bigr)\; },
\]
\[
\boxed{\;
\zeta_\chi(z,u):=\Delta_\chi(z,u)^{-1}
=\zeta_{\chi^{-1}}\!\bigl(u\,\chi(g_1)\,z,\ 1/u\bigr)\; }.
\]
\end{proposition}
\begin{proof}
由 \(\Pi^{-1}B_0\Pi=B_1,\ \Pi^{-1}B_1\Pi=B_0\) 得
\[
\Pi^{-1}B_\chi(u)\Pi
=B_1+u\,\chi(g_1)\,B_0
=u\,\chi(g_1)\Bigl(B_0+\frac{1}{u}\chi(g_1)^{-1}B_1\Bigr)
=u\,\chi(g_1)\,B_{\chi^{-1}}(1/u).
\]
对两边取 \(\det(I-z\,\cdot)\) 并用 \(\det(I-z\alpha A)=\det(I-(\alpha z)A)\) 即得所示行列式函数方程；\(\zeta\) 版本由取倒数得到。证毕。
\end{proof}

\begin{theorem}[Schur 分量的函数方程：状态级自对偶的 \texorpdfstring{$q$}{q}-重张量提升]\label{thm:kernel-self-dual-character-schur}
\leanverified{paper\_kernel\_self\_dual\_character\_schur}
沿用命题 \ref{prop:kernel-self-dual-character} 的记号，并令 \(V\) 为 \(B_\chi(u)\) 的作用空间（有限维复向量空间）。
对整数 \(q\ge 1\) 考虑张量幂 \(B_\chi(u)^{\otimes q}\) 在 \(V^{\otimes q}\) 上的作用。
令 \(S_q\) 以置换张量因子方式作用于 \(V^{\otimes q}\)，并取其 Schur--Weyl 等变分解
\[
V^{\otimes q}\ \cong\ \bigoplus_{\lambda\vdash q}\mathbb S_\lambda(V)\otimes S^\lambda,
\]
其中 \(\mathbb S_\lambda\) 为 Schur 函子、\(S^\lambda\) 为 Specht 模。
记 \(B_{\chi,\lambda}(u)\) 为 \(B_\chi(u)^{\otimes q}\) 在 \(\mathbb S_\lambda(V)\) 上诱导的 Schur 分量算子。
则对每个 \(\lambda\vdash q\) 存在可逆算子 \(\Pi_\lambda:\mathbb S_\lambda(V)\to\mathbb S_\lambda(V)\) 使得
\[
\boxed{\;
\Pi_\lambda^{-1}B_{\chi,\lambda}(u)\Pi_\lambda
=\bigl(u\,\chi(g_1)\bigr)^{q}\,B_{\chi^{-1},\lambda}(1/u)\; }.
\]
因此对每个 \(\lambda\vdash q\) 均有分量级函数方程
\[
\boxed{\;
\det\!\bigl(I-zB_{\chi,\lambda}(u)\bigr)
=
\det\!\bigl(I-\bigl(u\,\chi(g_1)\bigr)^{q}z\,B_{\chi^{-1},\lambda}(1/u)\bigr)\; }.
\]
\end{theorem}
\begin{proof}
由命题 \ref{prop:kernel-self-dual-character} 的相似恒等式
\(
\Pi^{-1}B_\chi(u)\Pi=u\,\chi(g_1)\,B_{\chi^{-1}}(1/u)
\)，取 \(q\) 次张量幂得
\[
(\Pi^{\otimes q})^{-1}B_\chi(u)^{\otimes q}\Pi^{\otimes q}
=\bigl(u\,\chi(g_1)\bigr)^q\,B_{\chi^{-1}}(1/u)^{\otimes q}.
\]
另一方面，\(\Pi^{\otimes q}\) 逐因子同构地作用于 \(V^{\otimes q}\)，故与 \(S_q\) 的置换作用对易，从而保持 Schur--Weyl 分解的每个等型分量。
将上述相似等式限制到 \(\mathbb S_\lambda(V)\otimes S^\lambda\) 上，并在 Specht 因子上取恒等，即得到 \(\mathbb S_\lambda(V)\) 上的诱导相似算子 \(\Pi_\lambda\) 与第一式。
对两边取 \(\det(I-z\,\cdot)\) 并吸收标量因子到 \(z\) 中，即得分量级函数方程。
\end{proof}

\begin{remark}[表示级函数方程的二面体对称：\texorpdfstring{$(\chi,u,z)\mapsto(\chi^{-1},1/u,u\chi(g_1)z)$}{(chi,u,z)->(chi^{-1},1/u,u chi(g1) z)} 的闭合]\label{rem:kernel-self-dual-character-dihedral}

\begin{proposition}[角色相位的 cyclotomic 消去：\texorpdfstring{$\mu_r$}{mu_r}-不变量压缩为 \texorpdfstring{$u^r$}{u^r}]\label{prop:character-phase-cyclotomic-elimination}
\leanverified{paper\_character\_phase\_cyclotomic\_elimination}
沿用命题 \ref{prop:kernel-self-dual-character} 的设定，并设 \(g_1\) 在 \(G\) 中的阶为 \(r\)。
记
\(
\mu_r:=\{\omega\in\CC:\omega^r=1\}
\)。
定义 cyclotomic 扭曲积
\[
\mathcal{D}_r(z,u):=\prod_{\omega\in\mu_r}\det\!\bigl(I-z(B_0+u\omega B_1)\bigr).
\]
则对任意 \(\omega_0\in\mu_r\) 有 \(\mathcal{D}_r(z,u\omega_0)=\mathcal{D}_r(z,u)\)。
因此存在唯一多项式 \(F_r(z,U)\in\ZZ[z,U]\) 使
\[
\boxed{\ \mathcal{D}_r(z,u)=F_r(z,u^r)\ }.
\]
\end{proposition}
\begin{proof}
对任意 \(\omega_0\in\mu_r\)，由于集合 \(\mu_r\) 在乘法下封闭，
\[
\mathcal{D}_r(z,u\omega_0)
=\prod_{\omega\in\mu_r}\det(I-z(B_0+u\omega_0\omega B_1))
=\prod_{\omega'\in\mu_r}\det(I-z(B_0+u\omega' B_1))
=\mathcal{D}_r(z,u).
\]
并且 \(\mathcal{D}_r(z,u)\) 的系数是 cyclotomic 整数且在 \(\omega\mapsto\omega^k\)（\(\gcd(k,r)=1\)）的共轭作用下不变，
故其系数落在 \(\ZZ\)，从而 \(\mathcal{D}_r\in\ZZ[z,u]\)。
写 \(\mathcal{D}_r(z,u)=\sum_{k\ge 0}d_k(z)u^k\)（\(d_k(z)\in\ZZ[z]\)），由不变性得
\(d_k(z)\omega_0^k=d_k(z)\) 对所有 \(\omega_0\in\mu_r\) 成立，
从而 \(d_k(z)=0\) 除非 \(r\mid k\)。
因此 \(\mathcal{D}_r(z,u)\) 只能含 \(u^{r\ell}\) 项；记 \(U=u^r\) 得所示表示，唯一性由系数比较得到。证毕。
\end{proof}

\begin{remark}[相位消元的 \(r\)-阶压缩]\label{rem:character-phase-cyclotomic-elimination-compress}

\end{remark}

\begin{proposition}[周期迹多项式的回文对称]\label{prop:trace-palindrome}
\leanverified{paper\_trace\_palindrome}
记
\[
a_n(u):=\Tr(B(u)^n)=\sum_{k=0}^{n}a_{n,k}u^k.
\]
则对所有 $n\ge 1$ 有
\[
a_n(u)=u^n a_n(1/u)
\quad\Longleftrightarrow\quad
a_{n,k}=a_{n,n-k}.
\]
\end{proposition}

\begin{proof}
由自对偶 $\Pi^{-1}B(u)\Pi=uB(1/u)$ 与迹不变性，
\[
a_n(u)=\Tr(B(u)^n)=\Tr\!\bigl((uB(1/u))^n\bigr)=u^n\Tr(B(1/u)^n)=u^n a_n(1/u),
\]
比较系数即得结论。证毕。
\end{proof}

\begin{conclusion}[Teichm\"uller 切片上的驻波分解：回文谱只含余弦模态]\label{con:sync-kernel-weighted-teichmueller-standing-wave}
沿用命题 \ref{prop:trace-palindrome} 的记号，并取单位圆扭曲 \(u=e^{\mathrm{i}\theta}\)。
定义归一化离散傅里叶振幅
\[
A_n(\theta):=e^{-\mathrm{i}n\theta/2}\,a_n(e^{\mathrm{i}\theta})
=\sum_{k=0}^{n}a_{n,k}\,e^{\mathrm{i}(k-n/2)\theta}.
\]
则 \(A_n(\theta)\in\RR\) 且 \(A_n(\theta)=A_n(-\theta)\)。
更精确地，\(A_n\) 具有纯余弦展开
\[
\boxed{\ 
A_n(\theta)=
\begin{cases}
a_{n,n/2}+2\sum_{k=0}^{n/2-1} a_{n,k}\cos\!\bigl((k-n/2)\theta\bigr), & 2\mid n,\\[4pt]
2\sum_{k=0}^{(n-1)/2} a_{n,k}\cos\!\bigl((k-n/2)\theta\bigr), & 2\nmid n.
\end{cases}
\ }
\]
因此 \(a_n(e^{\mathrm{i}\theta})=e^{\mathrm{i}n\theta/2}A_n(\theta)\) 为“行波相位”乘以实偶振幅，离散谱在严格意义上呈驻波型而无正弦分量。
\end{conclusion}
\begin{proof}[证明要点]
由 \(a_{n,k}=a_{n,n-k}\)，把和式按 \(k\leftrightarrow n-k\) 配对：
\[
a_{n,k}e^{\mathrm{i}(k-n/2)\theta}+a_{n,n-k}e^{-\mathrm{i}(k-n/2)\theta}
=2a_{n,k}\cos\!\bigl((k-n/2)\theta\bigr).
\]
若 \(n\) 为偶数，尚有中心项 \(k=n/2\) 给出常数项 \(a_{n,n/2}\)。
由此得到所示余弦展开与实偶性。
\end{proof}

\begin{conclusion}[Archimedean--Teichm\"uller 二面体约束：完成化落在 \texorpdfstring{$\mathbb{Z}[s]$}{Z[s]} 并满足奇偶表示]\label{con:sync-kernel-weighted-archimedean-teichmueller-dihedral}
仍沿用命题 \ref{prop:trace-palindrome} 的记号，令 \(u=r^2\)，并记
\[
s:=r+r^{-1},\qquad \widehat a_n(s):=r^{-n}a_n(r^2)\in\mathbb{Z}[r,r^{-1}].
\]
则 \(\widehat a_n(s)\) 实际属于不变量子环 \(\mathbb{Z}[s]\)，并满足二面体奇偶律
\[
\boxed{\ \widehat a_n(-s)=(-1)^n\,\widehat a_n(s)\ }.
\]
在 Archimedean 单位圆切片 \(r=e^{\mathrm{i}\theta/2}\) 上，\(s=2\cos(\theta/2)\in[-2,2]\) 使得
\[
a_n(e^{\mathrm{i}\theta})=e^{\mathrm{i}n\theta/2}\,\widehat a_n(2\cos(\theta/2)),
\]
从而驻波振幅严格由实变量 \(s\) 控制。
该 “\(r\mapsto r^{-1}\)” 反射与 “\(r\mapsto -r\)” 半转共同生成的二面体约束，与正文的自对偶函数方程（命题 \ref{prop:kernel-self-dual-character} 与注 \ref{rem:kernel-self-dual-character-dihedral}）在结构上同源。
\end{conclusion}
\begin{proof}[证明要点]
由命题 \ref{prop:trace-palindrome}，\(a_n(r^2)=r^{2n}a_n(r^{-2})\)，故
\(
r^{-n}a_n(r^2)=r^{n}a_n(r^{-2})
\)，
等价于 \(r^{-n}a_n(r^2)\) 在 \(r\mapsto r^{-1}\) 下不变。
由不变量子环恒等式 \(\mathbb{Z}[r,r^{-1}]^{\,r\mapsto r^{-1}}=\mathbb{Z}[r+r^{-1}]=\mathbb{Z}[s]\) 得 \(\widehat a_n(s)\in\mathbb{Z}[s]\)。
又因 \(\widehat a_n(-r)=(-r)^{-n}a_n((-r)^2)=(-1)^n\widehat a_n(r)\)，而 \(-r+(-r)^{-1}=-s\)，故得到 \(\widehat a_n(-s)=(-1)^n\widehat a_n(s)\)。
\end{proof}

\begin{corollary}[奇阶消失律]\label{cor:trace-odd-vanish}
\leanverified{paper\_trace\_odd\_vanish}
若 $n$ 为奇数，则
\[
a_n(-1)=0,
\qquad
(u+1)\mid a_n(u)\ \ \text{于}\ \mathbb{Z}[u].
\]
\end{corollary}
\begin{proof}
由命题 \ref{prop:trace-palindrome}，$a_n(u)=u^n a_n(1/u)$。取 $u=-1$ 得 $a_n(-1)=(-1)^n a_n(-1)$，奇数 $n$ 迫使 $a_n(-1)=0$，从而 $(u+1)\mid a_n(u)$。证毕。
\end{proof}

\begin{corollary}[周期层的中心矩恒等式：奇阶严格为零]\label{cor:trace-central-moments-zero}
\leanverified{paper\_trace\_central\_moments\_zero}
沿用命题 \ref{prop:trace-palindrome} 的记号。对任意整数 \(m\ge 0\)，都有中心矩恒等式
$$
\boxed{\ \sum_{k=0}^{n}\left(k-\frac{n}{2}\right)^{2m+1}a_{n,k}=0\ }\qquad(\text{于 }\QQ).
$$
特别地，取 \(m=0\) 得
$$
\sum_{k=0}^{n}k\,a_{n,k}=\frac{n}{2}\sum_{k=0}^{n}a_{n,k}=\frac{n}{2}\,a_n(1).
$$
若进一步把系数正规化为 \(\pi^{\mathrm{per}}_{n,k}:=a_{n,k}/a_n(1)\)（当 \(a_n(1)\neq 0\)），则 \(\{\pi^{\mathrm{per}}_{n,k}\}_k\) 关于 \(n/2\) 精确对称，所有奇阶中心矩严格为零。
\end{corollary}
\begin{proof}
由命题 \ref{prop:trace-palindrome} 有 \(a_{n,k}=a_{n,n-k}\)。
对任意函数 \(f\) 满足 \(f(x)=-f(-x)\)（奇函数），把求和按 \(k\leftrightarrow n-k\) 配对得
$$
\sum_{k=0}^{n} f\!\left(k-\frac{n}{2}\right)a_{n,k}
=\frac12\sum_{k=0}^{n}\Bigl(
f\!\left(k-\frac{n}{2}\right)a_{n,k}
f\!\left((n-k)-\frac{n}{2}\right)a_{n,n-k}
\Bigr)=0,
$$
因为 \((n-k)-\frac{n}{2}=-(k-\frac{n}{2})\) 且 \(a_{n,n-k}=a_{n,k}\)。
取 \(f(x)=x^{2m+1}\) 即得所示恒等式；\(m=0\) 的线性式为其特例。证毕。
\end{proof}

\begin{corollary}[回文迹在单位圆上的完成化实值性：纯余弦模]\label{cor:trace-palindrome-completed-real}
\leanverified{paper\_trace\_palindrome\_completed\_real}
在命题 \ref{prop:trace-palindrome} 的设定下，取 \(u=e^{i\theta}\) 并定义完成化迹振幅
\[
\mathcal{A}_n(\theta):=e^{-in\theta/2}a_n(e^{i\theta})
=\sum_{k=0}^{n}a_{n,k}e^{i(k-n/2)\theta}.
\]
则 \(\mathcal{A}_n(\theta)\in\RR\) 且 \(\mathcal{A}_n(\theta)=\mathcal{A}_n(-\theta)\)。
等价地，\(\mathcal{A}_n\) 的 Fourier 展开中正弦分量严格为零，仅含余弦模态。
若进一步设 \(a_n(1)\neq 0\) 并令 \(\pi^{\mathrm{per}}_{n,k}:=a_{n,k}/a_n(1)\)，则其特征函数
\[
\Phi_n(\theta):=\sum_{k=0}^{n}\pi^{\mathrm{per}}_{n,k}e^{i(k-n/2)\theta}
\]
亦为实偶函数。
\end{corollary}
\begin{proof}
由回文关系 \(a_{n,k}=a_{n,n-k}\)，对和式按 \(k\leftrightarrow n-k\) 配对：
\[
a_{n,k}e^{i(k-n/2)\theta}+a_{n,n-k}e^{i((n-k)-n/2)\theta}
=2a_{n,k}\cos\!\bigl((k-n/2)\theta\bigr),
\]
故 \(\mathcal{A}_n\) 为实值且为偶函数。
归一化后的 \(\Phi_n\) 仅是对系数作比例缩放，结论不变。证毕。
\end{proof}

\begin{remark}[频域禁律：正弦分量的结构性消失]\label{rem:trace-palindrome-sine-forbidden}
在幺正切片上，回文完成化将迹振幅强制为实偶函数，从而正弦分量严格为零；
等价地，任何携带净流的行进相位在该层级被结构性剔除。
该频域表述与推论 \ref{cor:trace-central-moments-zero} 的奇阶中心矩消失完全一致。
\end{remark}

\begin{corollary}[$\Delta(z,u)$ 的系数回文结构]\label{cor:delta-coeff-palindrome}
\leanverified{paper\_delta\_coeff\_palindrome}
写
\[
\Delta(z,u)=\det(I-zB(u))=\sum_{j=0}^{10}c_j(u)\,z^j.
\]
则对每个 $j$ 都有
\[
c_j(u)=u^j c_j(1/u),
\]
即 $c_j(u)$ 为次数 $j$ 的回文多项式（完成化后等价为 $u^{j/2}$ 乘以偶函数）。
\end{corollary}

\begin{remark}[不变量表示：系数仅依赖 $u+u^{-1}$]
取 $u=e^\theta$，则 $u^{-j/2}c_j(u)$ 为 $\theta$ 的偶函数，因而可写成 $u+u^{-1}$ 的多项式；等价地，完成化后 $\widehat{\Delta}(w,u)$ 的系数落在不变量子环 $\QQ[u+u^{-1}]$ 中。
\end{remark}

\begin{corollary}[完成化系数的“临界线实值性”]\label{cor:completion-real-line}
\leanverified{paper\_completion\_real\_line}
取 $u=e^{it}$（$|u|=1$），并定义
\[
d_j(u):=u^{-j/2}c_j(u).
\]
则
\[
\overline{d_j(u)}=d_j(u),
\]
即完成化后的系数在单位圆上为实数。
\end{corollary}

\begin{proposition}[回文化系数导致的代数分歧锁定：分支障碍由一元判别式完全控制]\label{prop:delta-branchpoints-discriminant-lock}
\leanverified{paper\_delta\_branchpoints\_discriminant\_lock}
沿用命题 \ref{prop:sync-kernel-weighted-hat-delta} 的完成化表示 \(\Delta(z,u)=\widehat\Delta(w,s)\)，其中 \(w=z\sqrt{u}\)、\(s=\sqrt{u}+1/\sqrt{u}\)。
定义关于 \(w\) 的结式
\[
\mathscr{B}(s):=\operatorname{Res}_w\bigl(\widehat\Delta(w,s),\partial_w\widehat\Delta(w,s)\bigr)\in\ZZ[s].
\]
若 \(w_\star(s)\) 为方程 \(\widehat\Delta(w,s)=0\) 的某一解析分支，且在一点 \(s_0\) 处对应单根，则 \(w_\star\) 在 \(\CC\setminus \mathscr{B}^{-1}(0)\) 上可解析延拓，且任意解析延拓障碍只能发生在 \(\mathscr{B}(s)=0\) 上。
限制到单位圆切片 \(u=e^{i\theta}\)（\(s=2\cos(\theta/2)\)）时，谱分支的带边/拐点候选角集合因此被代数集合
\[
\{\,\theta\in\RR:\ \mathscr{B}(2\cos(\theta/2))=0\,\}
\]
所锁定，并在 \(\theta\mapsto-\theta\) 下闭合。
\end{proposition}
\begin{proof}
隐函数定理失效的充要条件为同时满足 \(\widehat\Delta(w,s)=0\) 与 \(\partial_w\widehat\Delta(w,s)=0\)，
等价于 \(\widehat\Delta(\cdot,s)\) 在 \(w\)-方向具有重根；这当且仅当两者的结式 \(\mathscr{B}(s)\) 为零。
因此在 \(\mathscr{B}(s)\neq 0\) 的开集上解析分支可唯一延拓，而分歧只能出现在 \(\mathscr{B}(s)=0\) 上。
单位圆切片的角集合由 \(s(\theta)=2\cos(\theta/2)\) 的实值与偶性给出。证毕。
\end{proof}

\begin{remark}[带边候选角的一元可审计化]\label{rem:discriminant-univariate-audit}
命题 \ref{prop:delta-branchpoints-discriminant-lock} 将谱分支的解析分歧位置压缩为一元整系数多项式 \(\mathscr{B}\) 的零点集合；
因此在 unitary slice 上的带边/分岐候选角可由代数消元直接审计。
\end{remark}
