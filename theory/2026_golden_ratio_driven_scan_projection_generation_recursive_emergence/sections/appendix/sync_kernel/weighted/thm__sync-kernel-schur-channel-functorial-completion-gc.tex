\begin{theorem}[Schur 通道的严格自对偶函子性]\label{thm:sync-kernel-schur-channel-strict-self-dual-functoriality}
沿用 \S\ref{app:kernel-self-dual} 的记号，设
\[
B(u)=B_0+uB_1,\qquad \Pi^{-1}B(u)\Pi=uB(1/u).
\]
固定 \(q\ge 1\)，令
\[
A_q(u):=B(u)^{\otimes q}\in \End(V^{\otimes q}),
\]
并按 Schur--Weyl 分解写成
\[
V^{\otimes q}\cong \bigoplus_{\lambda\vdash q}\mathcal U_{q,\lambda}\otimes \mathcal S_\lambda,
\qquad
A_q(u)\cong \bigoplus_{\lambda\vdash q} B_{q,\lambda}(u)\otimes \Id_{\mathcal S_\lambda},
\]
其中 \(\mathcal S_\lambda\) 为 Specht 模，\(B_{q,\lambda}(u)\in \End(\mathcal U_{q,\lambda})\) 为对应 Schur 通道算子。
则对每个 \(\lambda\vdash q\)，存在可逆变换
\[
\Pi_{q,\lambda}\in \GL(\mathcal U_{q,\lambda})
\]
（可取为 \(\Pi^{\otimes q}\) 在 \(\lambda\)-等型分量上的限制）使
\[
\Pi_{q,\lambda}^{-1}B_{q,\lambda}(u)\Pi_{q,\lambda}=u^q\,B_{q,\lambda}(1/u).
\]
从而其通道行列式
\[
\Delta_{q,\lambda}(z,u):=\det(I-zB_{q,\lambda}(u))
\]
满足二维自对偶函数方程
\[
\Delta_{q,\lambda}(z,u)=\Delta_{q,\lambda}(u^q z,1/u).
\]
\end{theorem}
\begin{proof}
由状态级相似关系
\(
\Pi^{-1}B(u)\Pi=uB(1/u)
\)
取 \(q\) 次张量幂得
\[
(\Pi^{\otimes q})^{-1}A_q(u)\Pi^{\otimes q}
=
\bigl(uB(1/u)\bigr)^{\otimes q}
=
u^q A_q(1/u).
\]
又 \(\Pi^{\otimes q}\) 与 \(S_q\) 的自然置换作用对易，故保持每个 \(\lambda\)-等型分量并诱导可逆映射 \(\Pi_{q,\lambda}\)。
把上式限制到 \(\mathcal U_{q,\lambda}\otimes\mathcal S_\lambda\) 并在 Specht 因子上取恒等，即得
\[
\Pi_{q,\lambda}^{-1}B_{q,\lambda}(u)\Pi_{q,\lambda}=u^q B_{q,\lambda}(1/u).
\]
对两侧取 \(\det(I-z\,\cdot)\) 即得
\[
\Delta_{q,\lambda}(z,u)=\Delta_{q,\lambda}(u^q z,1/u).
\]
证毕。
\end{proof}

\begin{corollary}[Schur 全通道完成化与单位切片代数判别]\label{cor:sync-kernel-schur-channel-completion-unit-slice}
在定理 \ref{thm:sync-kernel-schur-channel-strict-self-dual-functoriality} 的设定下，取
\[
u=r^2,\qquad w:=zr^q,\qquad S:=r+r^{-1}.
\]
则对任意 \(\lambda\vdash q\) 存在唯一二元多项式
\[
\widehat\Delta_{q,\lambda}(w,S)\in\ZZ[w,S]
\]
使
\[
\Delta_{q,\lambda}(z,u)=\widehat\Delta_{q,\lambda}\!\left(z\,u^{q/2},\,u^{1/2}+u^{-1/2}\right).
\]
进一步，对任意 \(L>0\) 定义单位切片多项式
\[
P_{L,\lambda}(S):=\widehat\Delta_{q,\lambda}(L^{-1/2},S).
\]
在单位切片 \(u=e^{i\theta}\) 上取径向规范
\[
z=L^{-1/2}e^{-iq\theta/2},
\]
则有等价关系
\[
\Delta_{q,\lambda}(z,e^{i\theta})=0
\quad\Longleftrightarrow\quad
P_{L,\lambda}\!\bigl(2\cos(\theta/2)\bigr)=0.
\]
因而“单位切片零点是否全部来自 \(|S|\le 2\)”可转化为 \(P_{L,\lambda}\) 在区间 \([-2,2]\) 的实根定位问题，并可由 Sturm 链在有限步内判定。
\end{corollary}
\begin{proof}
由
\[
\Delta_{q,\lambda}(z,u)=\Delta_{q,\lambda}(u^q z,1/u)
\]
令 \(u=r^2\)、\(w=zr^q\)，得
\[
\Delta_{q,\lambda}(w/r^q,r^2)
\]
在对合 \(r\mapsto r^{-1}\) 下不变。
由不变量环恒等式
\[
\ZZ[r,r^{-1}]^{\,r\mapsto r^{-1}}=\ZZ[r+r^{-1}]=\ZZ[S],
\]
可写为唯一的 \(\widehat\Delta_{q,\lambda}(w,S)\in\ZZ[w,S]\)。

当 \(u=e^{i\theta}\) 时取 \(r=e^{i\theta/2}\)，并代入规范 \(z=L^{-1/2}e^{-iq\theta/2}\)，则
\[
w=zr^q=L^{-1/2},\qquad S=r+r^{-1}=2\cos(\theta/2).
\]
故
\[
\Delta_{q,\lambda}(z,e^{i\theta})
=
\widehat\Delta_{q,\lambda}\!\bigl(L^{-1/2},2\cos(\theta/2)\bigr)
=
P_{L,\lambda}(2\cos(\theta/2)),
\]
从而得到所示等价。
最后，区间实根计数/定位可由 Sturm 链完成，故给出有限步代数判别。
证毕。
\end{proof}

\begin{theorem}[Schur 通道压力函数的 Gallavotti--Cohen 对称律]\label{thm:sync-kernel-schur-channel-pressure-gc}
\leanverified{paper\_sync\_kernel\_schur\_channel\_pressure\_gc}
在定理 \ref{thm:sync-kernel-schur-channel-strict-self-dual-functoriality} 的设定下，固定 \(\lambda\vdash q\)。
若 \(u>0\) 时 \(B_{q,\lambda}(u)\) 为 primitive 非负矩阵，记其 Perron 根为
\[
\rho_{q,\lambda}(u),\qquad
P_{q,\lambda}(\theta):=\log\rho_{q,\lambda}(e^\theta).
\]
则
\[
P_{q,\lambda}(\theta)=q\theta+P_{q,\lambda}(-\theta).
\]
令
\[
I_{q,\lambda}(\alpha):=\sup_{\theta\in\RR}\{\alpha\theta-P_{q,\lambda}(\theta)\}
\]
为 Legendre--Fenchel 变换，则
\[
I_{q,\lambda}(\alpha)=I_{q,\lambda}(q-\alpha).
\]
等价地，定义中心化率函数
\[
J_{q,\lambda}(s):=I_{q,\lambda}\!\left(\frac q2+s\right),
\]
则 \(J_{q,\lambda}(s)=J_{q,\lambda}(-s)\)。
若再定义
\[
\widetilde I_{q,\lambda}(s):=J_{q,\lambda}(s)-\frac q2\,s,
\]
则满足线性反对称形式
\[
\widetilde I_{q,\lambda}(s)-\widetilde I_{q,\lambda}(-s)=-q\,s.
\]
\end{theorem}
\begin{proof}
由相似关系
\[
\Pi_{q,\lambda}^{-1}B_{q,\lambda}(u)\Pi_{q,\lambda}=u^q B_{q,\lambda}(1/u)
\]
可知两侧谱相同，故
\[
\rho_{q,\lambda}(u)=u^q\rho_{q,\lambda}(1/u).
\]
令 \(u=e^\theta\) 取对数即得
\[
P_{q,\lambda}(\theta)=q\theta+P_{q,\lambda}(-\theta).
\]

再证率函数对称。由定义
\[
I_{q,\lambda}(q-\alpha)
=
\sup_{\theta}\{(q-\alpha)\theta-P_{q,\lambda}(\theta)\}.
\]
作代换 \(t:=-\theta\)，并用 \(P_{q,\lambda}(-t)=P_{q,\lambda}(t)-qt\)，得
\[
I_{q,\lambda}(q-\alpha)
=
\sup_t\{\alpha t-P_{q,\lambda}(t)\}
=
I_{q,\lambda}(\alpha).
\]
于是 \(J_{q,\lambda}(s)=J_{q,\lambda}(-s)\)。
最后
\[
\widetilde I_{q,\lambda}(s)-\widetilde I_{q,\lambda}(-s)
=
\bigl(J_{q,\lambda}(s)-J_{q,\lambda}(-s)\bigr)-q s
=-q s.
\]
证毕。
\end{proof}

\begin{conjecture}[Schur 单位切片多项式的实根交错稳定塔]\label{conj:sync-kernel-schur-unit-slice-interlacing-tower}
沿用推论 \ref{cor:sync-kernel-schur-channel-completion-unit-slice} 的记号。
对每个 \(q\) 与 \(\lambda\vdash q\)，考虑单位切片多项式 \(P_{L,\lambda}(S)\)。
猜想对任意 \(L>0\) 成立：
\begin{enumerate}
  \item \(P_{L,\lambda}\) 的全部零点均为单实根，且落在开区间 \((-2,2)\)；
  \item 若 \(\lambda\prec\mu\)（Young 格上加一格的分支关系），则 \(P_{L,\lambda}\) 与 \(P_{L,\mu}\) 的零点严格交错。
\end{enumerate}
若该猜想成立，则 Schur 全通道单位切片谱约束可组织为 Gelfand--Tsetlin 型实根稳定塔，并给出通道化 toy-GRH 的统一表示论结构解释。
\end{conjecture}
