\begin{corollary}[primitive 多项式的完成化：\texorpdfstring{$\widehat p_n(s)\in\ZZ[s]$}{hat p in Z[s]} 与 Chebyshev--Adams 运算]\label{cor:primitive-completion-hatp}
取
\[
u=r^2,\qquad s:=r+r^{-1},
\]
并对 \(n\ge 1\) 定义 primitive 多项式的完成化
\[
\widehat p_n(s):=r^{-n}p_n(r^2).
\]
则对所有 \(n\ge 1\) 有 \(\widehat p_n(s)\in\ZZ[s]\)。并且：
\begin{enumerate}
  \item 若 \(n\) 为偶数，则 \(\widehat p_n(s)\) 为 \(s\) 的偶多项式；
  \item 若 \(n\) 为奇数，则 \(\widehat p_n(s)\) 为 \(s\) 的奇多项式。
\end{enumerate}
进一步，对任意整数 \(d\ge 1\) 定义
\[
C_d(s):=r^d+r^{-d}\in\ZZ[s],
\qquad
C_d(s)=2T_d(s/2)
\]
（\(T_d\) 为第一类 Chebyshev 多项式）。则完成化坐标下的幂替换满足严格的“倍角律”：
\[
r^{-dn}\,p_n(r^{2d})=\widehat p_n\!\bigl(C_d(s)\bigr),
\]
即 Witt/Möbius 反演中的 \(u\mapsto u^d\) 在完成化后等价于 \(s\mapsto C_d(s)\) 的 Chebyshev--Adams 运算。
\leanverified{paper\_primitive\_completion\_hatp}
\end{corollary}

\begin{corollary}[奇长度的严格奇偶平衡：\texorpdfstring{$p_n(-1)=0$}{p_n(-1)=0}]\label{cor:primitive-odd-parity-balance}
若 \(n\ge 3\) 为奇数，则由推论 \ref{cor:primitive-odd-div} 的整除律 \(u(u+1)\mid p_n(u)\) 得
\[
\boxed{\ p_n(-1)=\sum_{k\ge 0}(-1)^k p_{n,k}=0\ }.
\]
因此奇长度 primitive 轨道在“能量奇偶”两侧的计数严格相等（以符号加权意义表述）。
\leanverified{paper\_primitive\_odd\_parity\_balance}
\end{corollary}
\begin{proof}
由系数镜像对称 \(p_{n,k}=p_{n,n-k}\) 可得（见上文）
\(
p_n(u)=u^n p_n(1/u)
\)。
令 \(u=r^2\) 得
\(
p_n(r^2)=r^{2n}p_n(r^{-2})
\)，
从而 \(r^{-n}p_n(r^2)\) 在 \(r\mapsto r^{-1}\) 下不变。由不变量子环恒等式
\(
\ZZ[r,r^{-1}]^{\,r\mapsto r^{-1}}=\ZZ[r+r^{-1}]=\ZZ[s]
\)
即得 \(\widehat p_n(s)\in\ZZ[s]\)。
\par
奇偶性来自指数奇偶：\(p_n(r^2)\) 只含 \(r^{2k}\)，乘上 \(r^{-n}\) 后幂指数与 \(n\) 同奇偶；而 \(\ZZ[s]\) 中 \(s\mapsto -s\) 的奇偶分解与 \(r\mapsto -r\) 的幂奇偶一致，故得到 \(\widehat p_n\) 的奇偶性结论。
\par
最后令 \(s=r+r^{-1}\)，则
\(
C_d(s)=r^d+r^{-d}
\)
满足递推 \(C_{d+1}=sC_d-C_{d-1}\)，因而 \(C_d\in\ZZ[s]\)，并与 Chebyshev 多项式关系 \(C_d(s)=2T_d(s/2)\) 等价。代入 \(u=r^2\) 及 \(u^d=r^{2d}\) 得
\[
r^{-dn}p_n(r^{2d})=\widehat p_n(r^d+r^{-d})=\widehat p_n(C_d(s)),
\]
即所述倍角律。证毕。
\end{proof}

\begin{corollary}[完成化的 Chebyshev--Witt 变换：\texorpdfstring{$\widehat a\leftrightarrow \widehat p$}{hat a <-> hat p}]\label{cor:chebyshev-witt-completed}
\leanverified{paper\_chebyshev\_witt\_completed}
在同一完成化坐标 \(u=r^2,s=r+r^{-1}\) 下，定义 completed ghost（迹）多项式
\[
\widehat a_n(s):=r^{-n}a_n(r^2)=r^{-n}\Tr(B(r^2)^n)\in\ZZ[s].
\]
则对任意 \(n\ge 1\) 有严格恒等式
\[
\boxed{\ 
\widehat p_n(s)=\frac1n\sum_{d\mid n}\mu(d)\,\widehat a_{n/d}\!\bigl(C_d(s)\bigr),
\qquad
\widehat a_n(s)=\sum_{d\mid n}d\,\widehat p_d\!\bigl(C_{n/d}(s)\bigr)
\ }.
\]
\end{corollary}
\begin{proof}
对 \(u=r^2\) 的 Witt/Möbius 反演
\(
p_n(u)=\frac1n\sum_{d\mid n}\mu(d)a_{n/d}(u^d)
\)
两侧同乘 \(r^{-n}\)，并用完成化倍角律
\(
r^{-dn}f(r^{2d})=\widehat f(C_d(s))
\)
（对 \(f=a_{n/d}\) 与 \(f=p_d\) 分别应用）即可得到第一式；第二式同理由
\(
a_n(u)=\sum_{d\mid n} d\,p_d(u^{n/d})
\)
完成化得到。
\end{proof}

\begin{theorem}[完成化 primitive 的素幂差商解码恒等式]\label{thm:completed-primitive-prime-power-difference-quotient}
沿用推论 \ref{cor:chebyshev-witt-completed} 的完成化记号。
对任意素数 \(p\) 与整数 \(k\ge 1\)，有严格恒等式
\begin{equation}\label{eq:completed-primitive-prime-power-diffquot}
\boxed{\;
\widehat p_{p^k}(s)
=\frac{\widehat a_{p^k}(s)-\widehat a_{p^{k-1}}\!\bigl(C_p(s)\bigr)}{p^k}
\ \in\ \ZZ[s].\;}
\end{equation}
\leanverified{paper\_completed\_primitive\_prime\_power\_difference\_quotient}
\end{theorem}
\begin{proof}
将推论 \ref{cor:chebyshev-witt-completed} 的 Möbius 反演式
\(
\widehat p_n(s)=\frac1n\sum_{d\mid n}\mu(d)\,\widehat a_{n/d}(C_d(s))
\)
应用于 \(n=p^k\)。
由于 \(\mu(d)=0\) 除非 \(d\in\{1,p\}\)，且 \(\mu(1)=1\)、\(\mu(p)=-1\)，得到
\[
\widehat p_{p^k}(s)
=\frac{1}{p^k}\Bigl(\widehat a_{p^k}(s)-\widehat a_{p^{k-1}}\!\bigl(C_p(s)\bigr)\Bigr),
\]
即 \eqref{eq:completed-primitive-prime-power-diffquot}。
整系数性由左端 \(\widehat p_{p^k}\in\ZZ[s]\)（推论 \ref{cor:primitive-completion-hatp} 与推论 \ref{cor:chebyshev-witt-completed}）直接给出。证毕。
\end{proof}

\begin{corollary}[素数长度同余：\texorpdfstring{$\widehat a_p(s)\equiv C_p(s)\ (\mathrm{mod}\ p)$}{hat a_p congruent Cp mod p}]\label{cor:completed-prime-congruence}
\leanverified{paper\_completed\_prime\_congruence}
沿用推论 \ref{cor:chebyshev-witt-completed} 的完成化记号。若 \(p\) 为素数，则有整系数同余
\[
\boxed{\ \widehat a_p(s)\equiv C_p(s)\pmod p\ }.
\]
\end{corollary}
\begin{proof}
对素数 \(p\)，由 \(\widehat p_p(s)=\frac1p\bigl(\widehat a_p(s)-\widehat a_1(C_p(s))\bigr)\)（推论 \ref{cor:chebyshev-witt-completed} 的第一式，因子仅为 \(d=1,p\)）可知
\[
\widehat a_p(s)-\widehat a_1(C_p(s))\in p\,\ZZ[s].
\]

\(
\widehat a_1(s)=r^{-1}(1+r^2)=r+r^{-1}=s
\)，从而 \(\widehat a_1(C_p(s))=C_p(s)\)。
代回即得同余。证毕。
\end{proof}

\begin{lemma}[平方代换的判别式裂解：内部判别式以平方进入]\label{lem:disc-square-substitution-split}
\leanverified{paper\_disc\_square\_substitution\_split}
设 \(f(t)=a\prod_{j=1}^{d}(t-\alpha_j)\in\CC[t]\) 为次数 \(d\ge 1\) 的多项式，且 \(c:=f(0)\neq 0\)。
定义 \(g(s):=f(s^2)\in\CC[s]\)（次数 \(2d\) 的偶多项式）。
则判别式满足严格恒等式
\[
\boxed{\ \mathrm{Disc}_s(g)=(-1)^d\,4^d\,a\,c\cdot \mathrm{Disc}_t(f)^2\ }.
\]
\end{lemma}
\begin{proof}
写 \(f(t)=a\prod_{j=1}^{d}(t-\alpha_j)\)，则
\(
g(s)=a\prod_{j=1}^{d}(s^2-\alpha_j)
=a\prod_{j=1}^{d}(s-\sqrt{\alpha_j})(s+\sqrt{\alpha_j})
\)。
对每个 \(j\) 的同根对有差 \(2\sqrt{\alpha_j}\)，贡献
\(\prod_{j}(2\sqrt{\alpha_j})^2=4^d\prod_j \alpha_j\)；
而对不同 \(i\neq j\)，四对差的乘积满足
\[
\prod_{\varepsilon,\varepsilon'\in\{\pm 1\}}\bigl(\varepsilon\sqrt{\alpha_i}-\varepsilon'\sqrt{\alpha_j}\bigr)^2
=\bigl(\alpha_i-\alpha_j\bigr)^4,
\]
故跨对贡献为 \(\prod_{i<j}(\alpha_i-\alpha_j)^4=\mathrm{Disc}_t(f)^2/a^{4d-4}\)。
由判别式的根表示（\(g\) 为次数 \(2d\) 的多项式）得
\[
\mathrm{Disc}_s(g)=a^{4d-2}\cdot 4^d\prod_{j=1}^{d}\alpha_j\cdot\prod_{i<j}(\alpha_i-\alpha_j)^4.
\]
代入上一式，并用
\(
c=f(0)=a(-1)^d\prod_{j=1}^{d}\alpha_j
\)
消去 \(\prod_j\alpha_j\)，即得所述恒等式。证毕。
\end{proof}

\subsubsection{Dirichlet 完成化截面的单例外对与 Salem 多项式}\label{app:sync-kernel-weighted-dirichlet-xi-single-exception}
沿用式 \eqref{eq:sync-kernel-weighted-delta-explicit} 的 \(\Delta(z,u)=\det(I-zB(u))\) 以及命题 \ref{prop:kernel-xi-functional-equation} 的完成化截面
\[
\Xi(s):=\Delta(3^{-s},3^{2s-1}),
\qquad
\Xi(s)=\Xi(1-s).
\]
记
\[
r:=3^{\,s-\frac12}\qquad(\text{故 }u=3^{2s-1}=r^2,\ z=3^{-s}=3^{-1/2}r^{-1}).
\]

\begin{theorem}[完成化零点的单例外对结构]\label{thm:sync-kernel-weighted-xi-single-exception-pair}
在上述记号下，存在多项式 \(N(r)\in\ZZ[\sqrt3][r]\) 使得
\begin{equation}\label{eq:sync-kernel-weighted-xi-Nr-quotient}
\boxed{\;
\Xi(s)=\frac{N(r)}{27r^3},
\qquad
N(r)=\sqrt3\,r^6-2r^5-3\sqrt3\,r^4-8r^3-3\sqrt3\,r^2-2r+\sqrt3\; }.
\end{equation}
并且 \(N\) 为互逆多项式：\(r^6N(1/r)=N(r)\)。
设 \(\sigma:\sqrt3\mapsto-\sqrt3\) 为 \(\QQ(\sqrt3)\) 的非平凡自同构，则 \(N\) 的零点满足如下分层：
\begin{enumerate}
  \item 存在唯一实数 \(\alpha>1\) 使得 \(\{\alpha,\alpha^{-1}\}\) 为 \(N\) 的全部非幺模实零点；
  \item 除 \(\alpha,\alpha^{-1}\) 外，\(N\) 的其余四个零点皆满足 \(|r|=1\)，并成两对共轭。
\end{enumerate}
等价地，\(\Xi(s)=0\) 在临界线 \(\Re(s)=\tfrac12\) 之外仅出现一对例外零点（互为 \(s\mapsto 1-s\) 的对称像）。
\leanverified{paper\_sync\_kernel\_weighted\_xi\_single\_exception\_pair}
\end{theorem}

\begin{proof}
将 \(u=r^2,\ z=3^{-1/2}r^{-1}\) 代入 \(\Delta(z,u)\) 的显式表达式 \eqref{eq:sync-kernel-weighted-delta-explicit} 并清分母，即得 \eqref{eq:sync-kernel-weighted-xi-Nr-quotient}；互逆性由直接代数化简验证。
\par
为刻画零点模长结构，令 \(t:=r+r^{-1}\)。
由互逆性可将 \(N\) 写成
\begin{equation}\label{eq:sync-kernel-weighted-Nr-pt}
\boxed{\;
N(r)=r^3P(t),
\qquad
P(t)=\sqrt3\,t^3-2t^2-6\sqrt3\,t-4\in\RR[t]\; }.
\end{equation}
计算得
\[
P(-2)=4\sqrt3-12<0,\qquad
P(-1)=5\sqrt3-6>0,\qquad
P(0)=-4<0,
\]
\[
P(3)=9\sqrt3-22<0,\qquad
P(4)=40\sqrt3-36>0,
\]
从而 \(P\) 存在三实根 \(t_1<t_2<t_3\) 且满足
\[
t_1\in(-2,-1),\qquad t_2\in(-1,0),\qquad t_3\in(3,4)\subset(2,\infty).
\]
对任一根 \(t=t_j\)，方程 \(t=r+r^{-1}\) 等价于二次方程 \(r^2-tr+1=0\)。
当 \(|t|<2\) 时判别式 \(t^2-4<0\)，两根为共轭且满足 \(|r|=1\)；
当 \(t>2\) 时判别式 \(t^2-4>0\)，两根为互逆正实根 \(\alpha,\alpha^{-1}\)（其中 \(\alpha=\frac{t+\sqrt{t^2-4}}{2}>1\)）。
由 \(t_1,t_2\in(-2,2)\) 与 \(t_3>2\) 即得所述分层。证毕。
\end{proof}

\begin{corollary}[三二次分解与临界/例外分层]\label{cor:sync-kernel-weighted-xi-three-quadratics}
\leanverified{paper\_sync\_kernel\_weighted\_xi\_three\_quadratics}
设 \(t_1<t_2<t_3\) 为 \(P(t)=0\) 的三实根，则在 \(\RR[r]\) 上成立分解
\[
N(r)=\sqrt3\prod_{j=1}^{3}\bigl(r^2-t_jr+1\bigr).
\]
其中 \(r^2-t_3r+1\) 的两根为 \(\alpha,\alpha^{-1}\)（\(\alpha>1\)），而 \(r^2-t_1r+1\) 与 \(r^2-t_2r+1\) 的根皆在单位圆上。
\end{corollary}

\begin{proof}
由 \eqref{eq:sync-kernel-weighted-Nr-pt} 有 \(N(r)=r^3P(r+r^{-1})\)。
将 \(P\) 分解为 \(\sqrt3\prod_{j=1}^{3}(t-t_j)\) 并使用恒等式
\(
r+r^{-1}-t_j=(r^2-t_jr+1)/r
\)
即可得到所示分解；根的模长归类继承自定理 \ref{thm:sync-kernel-weighted-xi-single-exception-pair} 的证明。证毕。
\end{proof}

\begin{theorem}[例外零点的显式坐标与偏离指数]\label{thm:sync-kernel-weighted-xi-exception-coordinate}
\leanverified{paper\_sync\_kernel\_weighted\_xi\_exception\_coordinate}
沿用推论 \ref{cor:sync-kernel-weighted-xi-three-quadratics} 的记号，令
\[
\alpha=\frac{t_3+\sqrt{t_3^2-4}}{2}>1,
\qquad
\beta:=\frac{\log\alpha}{\log 3}.
\]
则 \(\Xi(s)=0\) 的全部零点分为两类：
\begin{enumerate}
  \item 例外实零点恰为 \(s_\pm=\frac12\pm\beta\)；
  \item 其余四个零点全落在临界线 \(\Re(s)=\tfrac12\)，并可写为
  \[
  s=\frac12\pm i\frac{\theta_1}{\log 3},
  \qquad
  s=\frac12\pm i\frac{\theta_2}{\log 3},
  \qquad
  \theta_j:=\arccos(t_j/2)\in(0,\pi)\ (j=1,2).
  \]
\end{enumerate}
数值上，
\[
t_1\approx-1.6423800228,\quad t_2\approx-0.4350479177,\quad t_3\approx3.2321284788,
\]
\[
\alpha\approx 2.8855773846,\qquad \beta\approx 0.9646032765.
\]
\end{theorem}

\begin{proof}
由定义 \(r=3^{s-1/2}\) 与 \eqref{eq:sync-kernel-weighted-xi-Nr-quotient}，\(\Xi(s)=0\iff N(r)=0\)。
当 \(r=\alpha,\alpha^{-1}\) 时 \(\log r\in\RR\)，故 \(s=\frac12+\log r/\log 3\) 为实数并给出 \(s_\pm\)。
当 \(|r|=1\) 时写 \(r=e^{i\theta}\)，则 \(s=\frac12+i\theta/\log 3\) 落在临界线；
由 \(r+r^{-1}=2\cos\theta=t_j\) 得 \(\theta_j=\arccos(t_j/2)\)。
数值近似由三次方程 \(P(t)=0\) 的实根及二次方程 \(r^2-t_3r+1=0\) 的解析表达式给出。证毕。
\end{proof}

\begin{proposition}[范数降阶与整系数 Salem 型六次]\label{prop:sync-kernel-weighted-xi-norm-salem}
\leanverified{paper\_sync\_kernel\_weighted\_xi\_norm\_salem}
设 \(\sigma:\sqrt3\mapsto-\sqrt3\) 为 \(\QQ(\sqrt3)\) 的非平凡自同构，并令
\[
F(r):=-\,\mathrm{Nm}_{\QQ(\sqrt3)/\QQ}\!\bigl(N\bigr)(r):=-\,N(r)\,\sigma(N)(r)\in\ZZ[r].
\]
则有显式恒等式
\[
F(r)=3r^{12}-22r^{10}-23r^{8}-12r^{6}-23r^{4}-22r^{2}+3.
\]
置 \(y=r^2\)，得到整系数六次多项式
\[
f(y):=3y^{6}-22y^{5}-23y^{4}-12y^{3}-23y^{2}-22y+3\in\ZZ[y].
\]
则 \(y_0:=\alpha^2\) 为 \(f\) 的唯一实根（\(>1\)），并且 \(f\) 另有唯一实根 \(y_0^{-1}\in(0,1)\)，其余四根均满足 \(|y|=1\)。
并且 \(f\) 在 \(\QQ\) 上不可约（例如其模 \(7\) 约化不可约）。
\end{proposition}

\begin{proof}
范数闭式由直接展开 \(N(r)\sigma(N)(r)\) 并取负号得到；得到 \(f(y)\) 仅使用代换 \(y=r^2\)。
由定理 \ref{thm:sync-kernel-weighted-xi-single-exception-pair}，\(N(r)\) 的零点模长结构为 \(\{\alpha,\alpha^{-1}\}\cup\{|r|=1\text{ 的四点}\}\)；
共轭嵌入 \(\sigma\) 不改变 \(|r|\)，故 \(F(r)\) 的零点集合为两者并集，从而仍仅含一对正实互逆根 \(\alpha,\alpha^{-1}\) 而其余均在单位圆上。
代换 \(y=r^2\) 得到 \(f\) 的根模长结构为 \(\{\alpha^2,\alpha^{-2}\}\cup\{|y|=1\text{ 的四点}\}\)，从而实根性陈述成立。
不可约性可由有限域不可约性判别：将 \(f\) 模 \(7\) 约化得到 \(\FF_7[y]\) 上不可约六次多项式，故 \(f\) 在 \(\QQ[y]\) 上不可约（Gauss 引理）。证毕。
\end{proof}

\begin{theorem}[例外偏离指数的 Mahler 测度表述]\label{thm:sync-kernel-weighted-xi-beta-mahler}
令 \(\tilde f(y):=f(y)/3\) 为首一化多项式（从而 \(\tilde f\) 的首项与常数项皆为 \(1\)），并令 \(M(\tilde f)\) 为 Mahler 测度。
则
\[
M(\tilde f)=\alpha^2,
\qquad
\log M(\tilde f)=2\log\alpha=2\beta\log 3,
\qquad
\beta=\frac{1}{2\log 3}\,\log M(\tilde f).
\]
\end{theorem}

\begin{proof}
按定义 \(M(\tilde f)=\prod_i\max\{1,|y_i|\}\)，其中 \(\{y_i\}\) 为 \(\tilde f\) 的根（按重数计）。
由命题 \ref{prop:sync-kernel-weighted-xi-norm-salem}，除 \(y_0=\alpha^2>1\) 外其余根均满足 \(|y_i|\le 1\)；并且仅有一根为 \(y_0^{-1}<1\) 而其余四根在单位圆上。
因此 \(M(\tilde f)=\alpha^2\)，取对数并用 \(\beta=\log\alpha/\log 3\) 即得。证毕。
\end{proof}
