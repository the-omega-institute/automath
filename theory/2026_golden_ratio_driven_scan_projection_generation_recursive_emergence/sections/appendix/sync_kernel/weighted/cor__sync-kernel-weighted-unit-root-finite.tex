\begin{corollary}[单位根扭曲的有限共轭性：把“最坏模态”降为有限代数比较]\label{cor:sync-kernel-weighted-unit-root-finite}
\leanverified{paper\_sync\_kernel\_weighted\_unit\_root\_finite}
令 \(\omega_m:=e^{2\pi i/m}\) 且取 \(u=\omega_m^j\)（\(1\le j\le m-1\)）。
在完成化坐标 \(u=r^2,\ s=r+r^{-1}\) 下，可取 \(r=\omega_{2m}^{\,j}\)，从而
\[
s=2\cos(\pi j/m)\in K_{2m}^+:=\QQ(\omega_{2m}+\omega_{2m}^{-1}).
\]
由命题 \ref{prop:sync-kernel-weighted-hat-delta}，存在 \(\widehat\Delta(w,s)\in\ZZ[w,s]\) 使得
\(
\Delta(z,u)=\widehat\Delta(zr,s)
\)；
因此对固定 \(m\)，当 \(j\) 变化时所有单位根扭曲切片 \(\Delta(z,\omega_m^j)\) 的系数都落在同一个实代数数域 \(K_{2m}^+\) 中，并在其 Galois 作用下封闭。
特别地，“最坏扭曲模态”（例如按最小极点模长或按谱半径比较）可被表述为 \(K_{2m}^+\) 的有限个共轭嵌入下的\textbf{有限比较问题}；其需要比较的共轭数目至多为
\[
[K_{2m}^+:\QQ]=\frac{\phi(2m)}{2},
\]
其中 \(\phi\) 为欧拉函数。当 \(m\) 为奇数时 \(\phi(2m)=\phi(m)\)，于是该上界退化为 \(\phi(m)/2\)。
\end{corollary}

\begin{remark}[可计算桥包：推论 \ref{cor:sync-kernel-weighted-unit-root-finite} 的“有限证书”流程]\label{rem:sync-kernel-weighted-unit-root-finite-bridge}
对固定模数 \(m\)，要给出“最坏单位根扭曲”的严格证书，可按如下有限流程执行：
\begin{enumerate}
  \item \textbf{（完成化参数化）} 列出 \(s_j:=2\cos(\pi j/m)\)（\(1\le j\le m-1\)），并按 \(K_{2m}^+\) 的 Galois 作用（等价于 \(\omega_{2m}\mapsto \omega_{2m}^a,\ a\in(\ZZ/2m\ZZ)^\times\)）把这些 \(s_j\) 分成有限多个共轭轨道。
  \item \textbf{（代数求根）} 对每个轨道代表 \(s\)，在数域 \(K_{2m}^+\) 上求解 \(\widehat\Delta(w,s)=0\) 的最小模根 \(w_\star(s)\)（或等价的 \(\Lambda(s)=1/w_\star(s)\)），并由 \(z_\star=w_\star/r\) 得到单位根切片的最小极点模长与相应谱半径。
  \item \textbf{（有限比较）} 在全部轨道代表上做有限次比较即可得到“最坏模态”，并可将比较过程（所用的数域、最小多项式与根的精度/尾界）作为可审计证书对象输出。
\end{enumerate}
该流程把原本的“扫遍全部 \(j\)”压缩为“按共轭轨道做有限比较”，并且天然适配符号/代数证书（与数值谱排序相比更稳健）。
\end{remark}

\begin{corollary}[有限长度的精确镜像对称]\label{cor:primitive-finite-sym}

\subsubsection{完成化行列式与严格偶性}\label{app:kernel-completion}
令 $u=e^\theta$，定义完成化行列式
\[
\widetilde{\Delta}(z,\theta):=\Delta(e^{-\theta/2}z,e^\theta),
\qquad
\widetilde{\zeta}(z,\theta):=\widetilde{\Delta}(z,\theta)^{-1}.
\]
由 $\Delta(z,u)=\Delta(uz,1/u)$ 得
\[
\widetilde{\Delta}(z,-\theta)=\widetilde{\Delta}(z,\theta),
\qquad
\widetilde{\zeta}(z,-\theta)=\widetilde{\zeta}(z,\theta),
\]
即完成化后在参数 $\theta$ 上呈严格偶性。

\subsubsection{自对偶 \texorpdfstring{$\Delta(z,u)=\Delta(uz,1/u)$}{Delta(z,u)=Delta(uz,1/u)} 的 s->1-s 函数方程}\label{app:kernel-functional-equation}
本小节把二变量自对偶对称压缩为 Dirichlet 变量下的一条单变量函数方程，从而与 $\zeta(s)$ 叙事中的“完成化/函数方程/临界线”三件套对齐。

\begin{proposition}[完成化行列式的函数方程]\label{prop:kernel-xi-functional-equation}
\leanverified{paper\_kernel\_xi\_functional\_equation}
设同步核满足自对偶恒等式 $\Delta(z,u)=\Delta(uz,1/u)$（见 \S\ref{app:kernel-self-dual}），并取
\[
\lambda:=\rho(B(1))>1.
\]
定义沿曲线
\[
z(s):=\lambda^{-s},\qquad u(s):=\lambda^{2s-1}
\]
的单变量函数
\[
\Xi(s):=\Delta(z(s),u(s))=\Delta(\lambda^{-s},\lambda^{2s-1}).
\]
则有严格函数方程
\[
\boxed{\ \Xi(s)=\Xi(1-s)\ }.
\]
等价地，定义 $\zeta$ 的同一截面
\[
\zeta_{\mathrm{sd}}(s):=\zeta(z(s),u(s))=\Delta(\lambda^{-s},\lambda^{2s-1})^{-1},
\]
则 $\zeta_{\mathrm{sd}}(s)=\zeta_{\mathrm{sd}}(1-s)$。
\end{proposition}
\begin{proof}
由自对偶恒等式，
\[
\Xi(s)=\Delta(\lambda^{-s},\lambda^{2s-1})
=\Delta(\lambda^{2s-1}\lambda^{-s},\lambda^{1-2s})
=\Delta(\lambda^{s-1},\lambda^{1-2s}).
\]
而对 $1-s$ 有
\(
z(1-s)=\lambda^{-(1-s)}=\lambda^{s-1}
\)
且
\(
u(1-s)=\lambda^{2(1-s)-1}=\lambda^{1-2s}
\)，
故 $\Xi(s)=\Delta(z(1-s),u(1-s))=\Xi(1-s)$。证毕。
\end{proof}

\begin{remark}[临界线与对称固定点]\label{rem:kernel-critical-line-u1}
由定义 $u(s)=\lambda^{2s-1}$ 可见：直线 $\Re(s)=\tfrac12$ 满足 $|u(s)|=1$，并且 $s=\tfrac12$ 为对称固定点（对应 $u=1$）。
因此在该完成化截面上，“临界线”自然等价为“自对偶对称面/单位圆权重”。
\end{remark}

\begin{remark}[Dirichlet 压缩下的平衡原子常数化]\label{rem:dirichlet-balanced-atom-constant}
沿用命题 \ref{prop:kernel-xi-functional-equation} 的记号。对任意整数 \(a,b\)，沿 Dirichlet 曲线有
\[
u(s)^a z(s)^b=\lambda^{-a}\lambda^{(2a-b)s}.
\]
因此当 \(b=2a\) 时 \(u^a z^{2a}=\lambda^{-a}\) 与 \(s\) 无关，任何因子 \((1-u^a z^{2a})\) 在 \(\Xi(s)\) 中仅贡献常数 \((1-\lambda^{-a})\)，不影响零点/极点位置。
特别地，真实输入 40 状态核输出位势中的原子因子 \((1-uz^2)\)（命题 \ref{prop:atomic-2-orbit-split-logM}）在该截面上化为常数 \((1-\lambda^{-1})\)，其作用只体现为 Abel 有限部常数的平移。
\end{remark}

\begin{definition}[核 Hardy 函数：临界线实值化]\label{def:kernel-hardy-function}
沿用命题 \ref{prop:kernel-xi-functional-equation} 的记号。定义

Z(t):=\widetilde{\Delta}\!\bigl(\lambda^{-1/2},\,2it\log\lambda\bigr),
\qquad t\in\RR,
$$
并称其为同步核完成化行列式的 Hardy 型实值化。
\end{definition}

\begin{proposition}[Hardy 实值性、偶性与临界线截面恒等式]\label{prop:kernel-hardy-real-even}
\leanverified{paper\_kernel\_hardy\_real\_even}
对一切实数 \(t\in\RR\)，有
$$
\boxed{\ Z(t)\in\RR,\qquad Z(t)=Z(-t)\ }.
$$
并且若令 \(s=\tfrac12+it\)，则临界线上的截面恒等式成立：
$$
\boxed{\ Z(t)=\Xi\!\left(\tfrac12+it\right)\ }.
$$
\end{proposition}
\begin{proof}
取 \(\theta:=2it\log\lambda\) 与 \(u=e^\theta=\lambda^{2it}\)，则 \(|u|=1\)。
由完成化定义（\S\ref{app:kernel-completion}）
\[
\widetilde{\Delta}(z,\theta)=\Delta(e^{-\theta/2}z,e^\theta)
\]
可得
\[
Z(t)=\widetilde{\Delta}(\lambda^{-1/2},\theta)
=\Delta\!\bigl(e^{-\theta/2}\lambda^{-1/2},\,u\bigr)
=\Delta\!\bigl(\lambda^{-1/2-it},\,\lambda^{2it}\bigr)
=\Delta\!\bigl(\lambda^{-s},\,\lambda^{2s-1}\bigr)
=\Xi(s),
\]
从而得到 \(Z(t)=\Xi(\tfrac12+it)\)。
\par
另一方面，由 \(\Delta(z,u)=\sum_{j=0}^{10}c_j(u)z^j\) 与完成化定义立得
\[
\widetilde{\Delta}(z,\theta)=\sum_{j=0}^{10}c_j(u)\,e^{-j\theta/2}z^j
=\sum_{j=0}^{10}\bigl(u^{-j/2}c_j(u)\bigr)\,z^j
=\sum_{j=0}^{10}d_j(u)\,z^j,
\]
其中 \(d_j(u)=u^{-j/2}c_j(u)\)。
由推论 \ref{cor:completion-real-line}，当 \(|u|=1\) 时 \(d_j(u)\in\RR\)。
因此 \(\widetilde{\Delta}(z,\theta)\) 作为关于 \(z\) 的多项式在 \(z\in\RR\) 上取实值，特别地 \(Z(t)=\widetilde{\Delta}(\lambda^{-1/2},\theta)\in\RR\)。
\par
偶性由完成化偶性 \(\widetilde{\Delta}(z,-\theta)=\widetilde{\Delta}(z,\theta)\)（\S\ref{app:kernel-completion}）直接推出：把 \(\theta=2it\log\lambda\) 代入得 \(Z(-t)=Z(t)\)。证毕。
\end{proof}

\begin{remark}[朝 RH 的可执行核心：谱边界与代数消元]\label{rem:rh-core-elimination}

\begin{proposition}[Legendre 二次互逆律：二次系数的严格互逆]\label{prop:einstein-legendre}
设 \(P(\theta)=\log\lambda(e^\theta)\) 与 \(I(\alpha)\) 由本小节的 Legendre 关系连接，并记 \(\alpha_\star=P'(0)=\tfrac12\)。
则二次展开系数满足严格恒等式
\[
\boxed{\ I''(\alpha_\star)=\frac{1}{P''(0)}\ }.
\]
等价地，以 \(\delta=\alpha-\tfrac12\) 表示中心化，则
\[
I\!\left(\tfrac12+\delta\right)=\frac{\delta^2}{2P''(0)}+O(\delta^3),
\]
从而本文同步核的显式系数给出 \( \frac{1}{2P''(0)}=\frac{51}{11}\)。
\end{proposition}
\begin{proof}
由 \(I(\alpha(\theta))=\theta\alpha(\theta)-\bigl(P(\theta)-P(0)\bigr)\) 与 \(\alpha(\theta)=P'(\theta)\) 可得（在 \(\theta=0\) 的邻域）
\[
\frac{\mathrm{d}I}{\mathrm{d}\alpha}(\alpha(\theta))=\theta,
\qquad
\frac{\mathrm{d}^2 I}{\mathrm{d}\alpha^2}(\alpha(\theta))=\frac{\mathrm{d}\theta}{\mathrm{d}\alpha}=\frac{1}{\alpha'(\theta)}=\frac{1}{P''(\theta)}.
\]
令 \(\theta\to 0\) 即得 \(I''(\alpha_\star)=1/P''(0)\)。证毕。
\end{proof}
由 $p_{n,k}\approx \frac{1}{n}\exp(n(\log 3-I(k/n)))$ 得到局部二次近似：若
\(
k=\frac{n}{2}+x\sqrt{n}
\)
（$x$ 有界），则
\[
p_{n,k}\approx \frac{3^n}{n}\exp\!\left(-\frac{x^2}{2P''(0)}\right)
=\frac{3^n}{n}\exp\!\left(-\frac{51}{11}x^2\right),
\]
即呈现方差尺度 $nP''(0)=\frac{11}{102}n$ 的高斯型指数近似。

\begin{remark}[率函数的代数证书：\texorpdfstring{$(\alpha,I)$}{(alpha,I)} 曲线可消元]\label{rem:rate-algebraic-elimination}
本附录中 \(\lambda(u)\) 由六次多项式方程 \(F(\lambda,u)=0\) 隐式定义，并且
\[
\alpha(u)=P'(\theta)=\frac{u\lambda_u(u)}{\lambda(u)}

\[
F(\lambda,u)=0,\qquad
G(\alpha,\lambda,u):=\alpha\,\lambda\,F_\lambda(\lambda,u)+uF_u(\lambda,u)=0

\begin{remark}[消元曲线的对偶对称、端点奇异簇与 \texorpdfstring{$\alpha=\tfrac12$}{alpha=1/2} 聚合指纹]\label{rem:rate-algebraic-elimination-structure}
在本同步核上，resultant 的原始输出带有一个幂因子 \(u^{v_u}\)（来自 \(u\to 0\) 端的退化分支）。
脚本 \texttt{scripts/exp\_sync\_kernel\_rate\_curve\_elimination.py} 将其除去并得到最小 \(u\)-次数的规范化消元多项式 \(R(\alpha,u)\in\ZZ[\alpha,u]\)。
该 \(R\) 除了给出 Perron 率曲线 \(u\mapsto \alpha(u)\) 的代数闭包外，还强制携带三类结构性指纹：
\begin{itemize}
  \item \textbf{（对偶自对称）} \((u,\alpha)\mapsto(1/u,1-\alpha)\) 在代数曲线层面闭合；
  \item \textbf{（端点与侧翼退化簇）} 在 \(u\to 0\) 与 \(u\to\infty\) 端，既有区间端点 \(\alpha\to0\) 与 \(\alpha\to1\)（Perron 分支，见推论 \ref{cor:alpha-endpoint-series}），也强制携带来自 \(\lambda=0\) 退化谱分支的侧翼四重簇：\((u,\alpha)\to(0,\tfrac14)\) 与 \((u,\alpha)\to(\infty,\tfrac34)\)；
  \item \textbf{（中心聚合）} 在 \(\alpha=\tfrac12\) 截面上出现 \((u-1)^6\) 的六重聚合，其解析意义是：在 \(u=1\) 处全部 \(6\) 条非零谱分支同时满足 \(\alpha=1/2\)（见定理 \ref{thm:sync-spectral-tangent-lock}）。
\end{itemize}
\par
\noindent\textbf{侧翼端点簇的 Puiseux 机制：指数不够，必须给出 jet 与统一化变量。}
令 \(\alpha=\tfrac14+\delta\)。由结构证书中
\(
r_0(\alpha)=\alpha(\alpha-1)(4\alpha-1)^4
\)、
\(
r_1(\alpha)=(4\alpha-1)^2(104\alpha^4-172\alpha^3+50\alpha^2-12\alpha+3)
\)，且括号多项式在 \(\alpha=\tfrac14\) 处取值 \(\tfrac{27}{32}\neq0\)，可知侧翼簇满足 Puiseux 指数
\[
\boxed{\ \alpha-\tfrac14=\Theta(u^{1/2})\quad\bigl((u,\alpha)\to(0,\tfrac14)\bigr)\ }.
\]
但\emph{仅有指数并不足以稳定求根/延拓}：在本核上，侧翼端点的最低阶 jet 发生了完全平方退化，从而精确前导常数与正确统一化变量都被强制锁定。
\par
\noindent\textbf{（端点 jet 完全平方，给出精确常数 \(\boldsymbol{K_{\mathrm{end}}}\)）}
把 \(R(\alpha,u)=\sum_{j=0}^{22}r_j(\alpha)u^j\) 视为 \(u\) 的多项式，并在 \((u,\delta)\to(0,0)\) 展开。
由 \(r_0,r_1\) 的显式形状再加上系数 \(r_2(\tfrac14)=-\tfrac{243}{256}\)（同一脚本符号消元后可直接提取并做有理数审计），得到
\[
R\!\left(\tfrac14+\delta,u\right)
=-48\,\delta^4+\frac{27}{2}\,\delta^2u-\frac{243}{256}\,u^2
+O(\delta^5)+O(u\delta^3)+O(u^2\delta)+O(u^3).
\]
其中二次型部分可因式分解为
\[
-48\,\delta^4+\frac{27}{2}\,\delta^2u-\frac{243}{256}\,u^2
=-\frac{3}{256}\,\bigl(64\delta^2-9u\bigr)^2.
\]
因此侧翼端点簇上所有局部分支都满足
\[
\boxed{\ \delta^2=\frac{9}{64}u+O(u^{5/4})\ },
\qquad
\boxed{\ \alpha(u)=\frac14\pm K_{\mathrm{end}}\sqrt{u}+O(u^{3/4}),\quad K_{\mathrm{end}}:=\frac{3}{8}\ }.
\]
这条“端点常数”是无需额外计算开销且可直接复算的强审计项：若未来重新生成 \(R\) 或更换参数族，端点处 \(r_0,r_1\) 的形状先锁定指数 \(1/2\)，再由 \(r_2(\tfrac14)\) 的有理值检验 jet 是否仍为完全平方，从而直接否决大量潜在的消元/实现错误。
\par
\noindent\textbf{（四分支与统一化变量必须取 \(u^{1/4}\)）}
由于最低阶项为完全平方，侧翼端点的\emph{正确统一化变量}不是 \(t=\sqrt{u}\) 而是
\[
\boxed{\ u=s^4,\qquad s=u^{1/4}\ }.
\]
在该变量下，每一张局部分支 \(\alpha(s)\) 都是解析的；例如（取一条实分支）有
\[
\boxed{\ \alpha(s)=\frac14+\frac38 s^2+\frac{3}{16}s^3-\frac54 s^4+\frac{115}{64}s^5+O(s^6)\ }.
\]
于是局部上形成四页覆盖：对同一 \(u>0\)，四个取值 \(s\in\{\pm u^{1/4},\ \pm i\,u^{1/4}\}\) 给出两条实分支与一对共轭复分支。
上述分支特性对数值求根算法的稳定性具有关键影响：在 \((\alpha,u)\) 平面直接应用 Newton 迭代时，端点簇附近的奇异性易导致迭代轨迹偏离实分支并进入复分支区域；而引入变量替换 \(s=u^{1/4}\) 并结合 continuation/homotopy 策略，则可实现四张分支的稳定分离，从而将端点从几何分歧点描述统一为解析展开口径。
\par
\noindent\textbf{（相对条件数恒等律）}
尽管 \(\alpha'(u)\sim \pm \tfrac{K_{\mathrm{end}}}{2}\,u^{-1/2}\) 在 \(u\to0\) 发散，但相对条件数
\[
\kappa_\alpha(u):=\frac{u\,|\alpha'(u)|}{|\alpha-\tfrac14|}
\]
沿任意侧翼端点分支都有
\[
\boxed{\ \kappa_\alpha(u)\to\frac12\quad\bigl((u,\alpha)\to(0,\tfrac14)\bigr)\ }.
\]
该极限说明端点的不稳定性来自“用 \(u\) 作为绝对步长参数”的尺度选择，而不是来自相对误差意义下的不可控放大。
\par
\noindent\textbf{（对偶端点 \(u\to\infty\) 的同系数强审计）}
由对偶自对称 \((u,\alpha)\mapsto(1/u,1-\alpha)\) 可把上述一切端点结论无损搬运到 \((u,\alpha)\to(\infty,\tfrac34)\)：
令 \(v=1/u\) 与 \(\alpha'=1-\alpha\)，则 \((\alpha',v)\to(\tfrac14,0)\) 给出
\[
\boxed{\ \alpha(u)=\frac34-\frac38\,u^{-1/2}\ \mp\ \frac{3}{16}\,u^{-3/4}\ +\ \frac54\,u^{-1}+O(u^{-5/4})\ }.
\]
因此在 \(u\to0\) 端 \(\alpha\) 必须以系数 \(K_{\mathrm{end}}=\tfrac38\) 以 \(\sqrt{u}\) 速度离开 \(\tfrac14\)，而在 \(u\to\infty\) 端必须以\emph{同一系数} \(K_{\mathrm{end}}\) 以 \(u^{-1/2}\) 速度离开 \(\tfrac34\)；这两端同时构成对 \(R\) 的“无需额外计算开销且可直接复算、极强约束”审计对照。
\par
\noindent\textbf{中心聚合诱导的 \emph{order-6 singular inversion}。}
由 \(R(\tfrac12,u)=-(u-1)^6Q(u)/64\) 且 \(Q(1)\neq0\) 可知 \(u=1\) 在 \(\alpha=\tfrac12\) 截面上为恰 \(6\) 重根，从而
\[
\partial_u^k R\!\left(\tfrac12,u\right)\Big|_{u=1}=0,\qquad k=0,1,\dots,5.
\]

\[
|u-1|\asymp|\alpha-\tfrac12|^{1/6},
\]
因此反解的条件数按 \(|u-1|^{-5}\) 级爆炸。

\begin{proposition}[\texorpdfstring{$v_u(R_{\mathrm{raw}})=7$}{vu=7} 的刚性分解：\texorpdfstring{$7=3+4$}{7=3+4}]\label{prop:rate-resultant-uadic-split-7}
\leanverified{paper\_rate\_resultant\_uadic\_split\_7}
记 \(R_{\mathrm{raw}}(\alpha,u)=\mathrm{Res}_\lambda(F(\lambda,u),G(\alpha,\lambda,u))\) 为原始 resultant（未除去 \(u\)-幂因子），并令 \(v_u\) 为其 \(u\)-adic valuation（表 \ref{tab:sync_kernel_rate_curve_resultant_degree} 给出 \(v_u(R_{\mathrm{raw}})=7\)）。
则该“\(7\)”在 \(u\to 0\) 端可由两类独立退化源作出刚性分解
\[
\boxed{\ 7=3\ (\lambda\text{-退化包/激活端})\ +\ 4\ (\alpha\text{-端点簇})\ }.
\]
\begin{proof}
因为 \(F(\lambda,u)\) 关于 \(\lambda\) 首一，resultant 满足
\(
R_{\mathrm{raw}}(\alpha,u)=\prod_{F(\lambda_i,u)=0}G(\alpha,\lambda_i,u)
\)
（按重数计）。
当 \(u=0\) 时 \(F(\lambda,0)=\lambda^5(\lambda-1)\)，故根包分解为一条 \(\lambda=1\) 的解析支与一个五重零根包。
\par
\noindent 对该零根包作 Newton 多边形分析可得：当 \(u\to0\) 时它刚性分裂为四条 Puiseux 支
\(
\lambda(u)=\zeta\,u^{1/4}+O(u^{1/2})
\)
（\(\zeta^4=1\)）与一条解析小根支
\(
\lambda(u)=-u+O(u^2)
\)。
把这些近似代入
\(
G(\alpha,\lambda,u)=\alpha\lambda F_\lambda(\lambda,u)+uF_u(\lambda,u)
\)
并取首阶平衡：
\begin{itemize}
  \item 在四条 Puiseux 支上，
  \(
  G(\alpha,\lambda(u),u)=(1-4\alpha)\,u\lambda(u)+O(u^{3/2})
  \)，
  因而每条支贡献一个显式 \(u\) 因子，合计给出 \(u^4\)，并同时给出系数因子 \((1-4\alpha)^4=(4\alpha-1)^4\)（端点四重簇）；
  \item 在解析小根支上，
  \(
  G(\alpha,\lambda(u),u)=(1-\alpha)u^2+O(u^3)
  \)，
  而四条 Puiseux 支中的 \(\lambda(u)\sim u^{1/4}\) 还额外贡献总次数 \(u^{4\cdot 1/4}=u\)；二者合并为 \(u^3\)（激活端的刚性退化量）；
  \item 在 \(\lambda\approx 1\) 的分支上有 \(G(\alpha,1,0)=\alpha\)。
\end{itemize}
因此 \(R_{\mathrm{raw}}(\alpha,u)\) 的首项为（差一个整体符号）
\(
u^7\,\alpha(\alpha-1)(4\alpha-1)^4
\)，
从而规范化时除去的 \(u^7\) 并非数值冗余，而是上述两类退化源的刚性叠加；任何修改若改变了 \(v_u\) 的值，必定意味着“激活端”或“端点簇”结构被破坏。证毕。
\end{proof}
\end{proposition}

\subsubsection{能量分层 primitive 谱的局部极限定理（prime-local CLT）}\label{app:prime-local-clt}
本节把上一小节的 LDP（率函数 \(I\)）提升为长度 \(n\) 的能量分层 primitive 计数 \(p_{n,k}\) 的\emph{局部极限定理}，并显式给出 \(n^{-3/2}\) 的前因子（其中 \(n^{-1}\) 来自 primitive 的 Möbius/Witt 反演，额外的 \(n^{-1/2}\) 来自高斯局部化）。

\begin{theorem}[prime-local CLT：能量层的高斯局部极限与显式前因子]\label{thm:prime-local-clt}
令 \(p_{n,k}=[u^k]p_n(u)\) 表示长度 \(n\) 的 primitive 轨道中能量为 \(k\) 的计数（本附录记号），并记
\[
\sigma^2:=P''(0)=\frac{11}{102}.
\]
则对任意固定 \(X>0\)，当 \(n\to\infty\) 且
\[
k=\frac{n}{2}+x\sqrt{n},\qquad |x|\le X,
\]
有一致渐近
\[
\boxed{\;
p_{n,k}
=\frac{1}{n}\cdot \frac{3^n}{\sqrt{2\pi\sigma^2\,n}}\,
\exp\!\left(-\frac{x^2}{2\sigma^2}\right)\cdot \bigl(1+o(1)\bigr)\; }.
\]
等价地，
\[
p_{n,k}\sim \frac{3^n}{\sqrt{2\pi\sigma^2}}\;n^{-3/2}\,
\exp\!\left(-\frac{(k-\frac{n}{2})^2}{2\sigma^2 n}\right),
\qquad n\to\infty,
\]
且上述 \(\sim\) 在 \(|x|\le X\) 上一致成立。
\end{theorem}

\begin{remark}[证明路线]\label{rem:prime-local-clt-proof-idea}
要点是把“周期迹层”的谱扰动局部 CLT 与 primitive 的 Möbius 反演解耦：
\begin{itemize}
  \item 对周期迹 \(a_n(u)=\mathrm{Tr}(B(u)^n)\)，由 \(\lambda(e^\theta)\) 的实解析性与单位圆上的谱隙，可对 \(u\) 的复参数做 Fourier 反演/鞍点法，得到 \(a_{n,k}=[u^k]a_n(u)\) 的局部高斯极限及 \(n^{-1/2}\) 前因子；
  \item 对 primitive 多项式 \(p_n(u)=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(u^d)\)，主项来自 \(d=1\)；而 \(d\ge 2\) 的项在局部尺度上指数小（由 \(\rho(u)<\lambda(u)\) 与 \(n/d\) 缩短），故不影响主阶形状，只留下额外的 primitive 因子 \(1/n\)。
\end{itemize}
因此局部 CLT 的“形状常数”由同一压力 Hessian \(\sigma^2=P''(0)\) 决定，前因子则严格分解为 primitive 的 \(1/n\) 与局部高斯的 \(n^{-1/2}\)。
\end{remark}

\subsubsection{加权 Mertens 常数与 Abel 有限部}\label{app:mertens-weighted}
令 $\lambda(u)$ 为 Perron 根，$\rho(u)$ 为其余特征值最大模，则
\[
a_n(u)=\mathrm{Tr}(B(u)^n)=\lambda(u)^n+O(\rho(u)^n),
\]
由 Witt 反演得
\[
p_n(u)=\frac{\lambda(u)^n}{n}+O\!\left(\frac{\rho(u)^n}{n}\right).
\]
因此
\[
\Pi_\theta(N):=\sum_{n\le N}p_n(\theta)\sim \frac{\lambda(\theta)}{\lambda(\theta)-1}\cdot\frac{\lambda(\theta)^N}{N}.
\]
记
\[
\mathcal{M}_\theta(N):=\sum_{n\le N}\frac{p_n(u)}{\lambda(u)^n},
\]
则
\[
\mathcal{M}_\theta(N)=\log N+M(\theta)+o(1),
\qquad
M(\theta)=\gamma+\sum_{n\ge 1}\left(\frac{p_n(u)}{\lambda(u)^n}-\frac{1}{n}\right),
\]

\begin{proposition}[Mertens 常数的偶性]\label{prop:mertens-even}
\leanverified{paper\_mertens\_even}
由 $p_n(u)=u^n p_n(1/u)$ 与 $\lambda(u)=u\lambda(1/u)$ 得
\[
\frac{p_n(u)}{\lambda(u)^n}=\frac{p_n(1/u)}{\lambda(1/u)^n},
\]
从而
\[
M(\theta)=M(-\theta),\qquad
\log\mathfrak{M}(\theta)=\log\mathfrak{M}(-\theta).
\]
\end{proposition}
其中 $\gamma$ 为 Euler 常数。设
\[
P_\theta(r):=\sum_{n\ge 1}\frac{p_n(u)}{\lambda(u)^n}r^n,\quad 0<r<1,
\]
则 Abel 有限部给出
\[
\log\mathfrak{M}(\theta):=\mathrm{fp}_{r=1}\bigl(P_\theta(r)+\log(1-r)\bigr)
=\sum_{n\ge 1}\left(\frac{p_n(u)}{\lambda(u)^n}-\frac{1}{n}\right),
\]
从而
\[
M(\theta)=\gamma+\log\mathfrak{M}(\theta).
\]
在 $\theta=0$ 处有
\[
M(0)\approx 0.2738887860704445,\qquad
\log\mathfrak{M}(0)\approx -0.3033268788310883.
\]
