\begin{corollary}[鲁棒归一化斯托克斯恒等式与缺陷预算]\label{cor:app-robust-normalized-stokes-with-defect-budget}
在定理 \ref{thm:app-normalized-integration-l1-defect-inverse-tower} 的设定下，设 \((m-1)\)-形式序列 \(\alpha=(\alpha_n)_{n\ge 0}\) 的边界缺陷满足
\[
\sum_{n\ge 0}\frac{1}{D_{n+1}}\int_{\partial M_{n+1}}\bigl|\alpha_{n+1}-\pi_n^\ast\alpha_n\bigr|<\infty,
\]
并设 \(m\)-形式序列 \(\beta=(\beta_n)_{n\ge 0}\) 的层间缺陷满足
\[
\sum_{n\ge 0}\frac{1}{D_{n+1}}\int_{M_{n+1}}\bigl|\beta_{n+1}-\pi_n^\ast\beta_n\bigr|<\infty.
\]
进一步，设存在 \(\varepsilon_n\in\Omega^m(M_n)\) 使得在分布意义下
\[
\beta_n=d\alpha_n+\varepsilon_n\qquad(n\ge 0),
\]
并且归一化误差极限
\[
E_\infty:=\lim_{n\to\infty}\frac{1}{D_n}\int_{M_n}\varepsilon_n
\]
存在（例如当 \(\sum_{n\ge 0}\frac{1}{D_{n+1}}\int_{M_{n+1}}|\varepsilon_{n+1}-\pi_n^\ast\varepsilon_n|<\infty\) 时由定理 \ref{thm:app-normalized-integration-l1-defect-inverse-tower} 自动成立）。
则归一化极限泛函满足鲁棒恒等式
\[
I_\infty(\beta)-B_\infty(\alpha)=E_\infty,
\]
并且具有确定性误差界
\[
\bigl|I_\infty(\beta)-B_\infty(\alpha)\bigr|
\le
\limsup_{n\to\infty}\frac{1}{D_n}\int_{M_n}|\varepsilon_n|.
\]
特别地，若 \(\frac{1}{D_n}\int_{M_n}|\varepsilon_n|\to 0\)，则 \(I_\infty(\beta)=B_\infty(\alpha)\)。
\end{corollary}
\begin{proof}
对每个 \(n\) 的带角流形 \(M_n\) 应用逐层广义 Stokes 得
\[
\frac{1}{D_n}\int_{M_n}\beta_n-\frac{1}{D_n}\int_{\partial M_n}\alpha_n
=
\frac{1}{D_n}\int_{M_n}(d\alpha_n+\varepsilon_n)-\frac{1}{D_n}\int_{\partial M_n}\alpha_n
=
\frac{1}{D_n}\int_{M_n}\varepsilon_n.
\]
令 \(n\to\infty\)，左端分别收敛到 \(I_\infty(\beta)\) 与 \(B_\infty(\alpha)\)，右端按假设收敛到 \(E_\infty\)，得到恒等式。
误差界由三角不等式与上极限立得；最后一段为其直接推论。证毕。
\end{proof}
