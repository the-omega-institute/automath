\begin{theorem}[覆叠逆塔的 Gödel 边界成本守恒与归一化 Stokes 一致性]\label{thm:app-covering-tower-godel-boundary-normalized-stokes}
在定理 \ref{thm:app-normalized-stokes-finite-cover-inverse-tower} 与定理 \ref{thm:app-solenoid-fundamental-current-and-stokes} 的设定下，
令
\[
\pi_{n,0}:=\pi_0\circ\pi_1\circ\cdots\circ\pi_{n-1}:M_n\to M_0,
\qquad
D_n:=\prod_{j=0}^{n-1}d_j,
\qquad
D_0:=1.
\]
设 \(\partial M_0\) 的连通分支为 \(C_1,\dots,C_r\)，并取两两不同的素数 \(q_1,\dots,q_r\)。
对每个 \(n\ge 0\) 与每个边界连通分支 \(C'\subset \partial M_n\)，若 \(\pi_{n,0}(C')\subset C_i\)，则记其标签为 \(q_i\)，并定义 Gödel 边界整数
\[
H_n
:=
\prod_{C'\subset\partial M_n}
q_{\mathrm{lab}(C')}^{\,\deg(\pi_{n,0}|_{C'})}
\in \NN.
\]
则成立：
\begin{enumerate}
  \item \textbf{严格幂律.}
        记 \(H_0=\prod_{i=1}^{r}q_i\)。
        则对所有 \(n\ge 0\)，
        \[
        H_n=H_0^{D_n}.
        \]
        因而归一化 Gödel 成本
        \[
        \frac{1}{D_n}\log H_n=\log H_0
        \]
        与层数无关。
  \item \textbf{边界成本的 Stokes 实现.}
        存在 \(M_0\) 上支撑于边界 collar 邻域的光滑 \((m-1)\)-形式 \(\alpha\)，使得
        \[
        \int_{\partial M_0}\alpha=\log H_0,
        \qquad
        \frac{1}{D_n}\int_{M_n}\dd\bigl(\pi_{n,0}^{\ast}\alpha\bigr)=\log H_0
        \qquad(\forall\, n\ge 0).
        \]
  \item \textbf{与逆极限 Stokes 泛函兼容.}
        令 \(S=\varprojlim(M_n,\pi_n)\)，并以圆柱类记
        \[
        \widetilde{\alpha}:=[\alpha]\in \Omega^{m-1}_{\mathrm{cyl}}(S).
        \]
        则有
        \[
        \mathcal I_S(\dd\widetilde{\alpha})
        =
        \mathcal I_{\partial S}(\widetilde{\alpha}|_{\partial S})
        =
        \log H_0.
        \]
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(1)}
固定 \(i\in\{1,\dots,r\}\)。
总边界覆叠 \(\pi_{n,0}:\partial M_n\to \partial M_0\) 在 \(C_i\) 上的全逆像由若干连通分支 \(C'\) 组成，且各限制映射
\(
\pi_{n,0}|_{C'}:C'\to C_i
\)
均为有限覆盖。
这些覆盖次数之和等于总覆盖次数 \(D_n\)。
因此，素数 \(q_i\) 在 \(H_n\) 中的总指数恰为 \(D_n\)，从而
\[
H_n=\prod_{i=1}^{r}q_i^{D_n}
=\left(\prod_{i=1}^{r}q_i\right)^{D_n}
=H_0^{D_n}.
\]
对数化并除以 \(D_n\) 即得归一化成本恒等式。

\textbf{(2)}
对每个 \(i\)，取边界分支 \(C_i\) 的一个 collar 邻域 \(U_i\simeq C_i\times[0,\varepsilon)\)，并令
\(
\mathrm{pr}_i:U_i\to C_i
\)
为投影。
取支撑于 \(U_i\) 的 cutoff 函数 \(\chi_i\)，使其在 \(C_i\times\{0\}\) 的邻域内恒等于 \(1\)。
记 \(\mathrm{vol}_{C_i}\) 为 \(C_i\) 的诱导体积形式，并令
\[
a_i:=\frac{\log q_i}{\int_{C_i}\mathrm{vol}_{C_i}}.
\]
定义
\[
\alpha:=\sum_{i=1}^{r}\chi_i\,a_i\,\mathrm{pr}_i^{\ast}(\mathrm{vol}_{C_i}).
\]
则 \(\alpha\) 为支撑于边界 collar 的光滑 \((m-1)\)-形式，且
\[
\int_{\partial M_0}\alpha
=
\sum_{i=1}^{r}a_i\int_{C_i}\mathrm{vol}_{C_i}
=
\sum_{i=1}^{r}\log q_i
=
\log H_0.
\]

对任意 \(n\ge 0\)，覆盖换元公式给出
\[
\int_{\partial M_n}\pi_{n,0}^{\ast}\alpha
=
D_n\int_{\partial M_0}\alpha
=
D_n\log H_0.
\]
再由经典广义 Stokes 定理，
\[
\int_{M_n}\dd\bigl(\pi_{n,0}^{\ast}\alpha\bigr)
=
\int_{\partial M_n}\pi_{n,0}^{\ast}\alpha
=
D_n\log H_0.
\]
两边除以 \(D_n\) 即得第二条。

\textbf{(3)}
由定理 \ref{thm:app-solenoid-fundamental-current-and-stokes}，圆柱类 \(\widetilde{\alpha}=[\alpha]\) 满足
\[
\mathcal I_S(\dd\widetilde{\alpha})
=
\mathcal I_{\partial S}(\widetilde{\alpha}|_{\partial S})
=
\frac{1}{D_n}\int_{\partial M_n}\pi_{n,0}^{\ast}\alpha.
\]
将第 (2) 条中的恒等式代入，即得
\[
\mathcal I_S(\dd\widetilde{\alpha})
=
\mathcal I_{\partial S}(\widetilde{\alpha}|_{\partial S})
=
\log H_0.
\]
证毕。
\end{proof}
