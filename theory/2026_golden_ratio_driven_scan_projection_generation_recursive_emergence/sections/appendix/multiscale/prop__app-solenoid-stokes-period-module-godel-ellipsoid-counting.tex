\begin{proposition}[solenoid 周期格的 Gödel 椭球计数律]\label{prop:app-solenoid-stokes-period-module-godel-ellipsoid-counting}
\leanverified{paper\_prop\_app\_solenoid\_stokes\_period\_module\_godel\_ellipsoid\_counting}
\leanverified{paper\_app\_solenoid\_stokes\_period\_module\_godel\_ellipsoid\_counting}
取整数矩阵 \(M\in M_r(\ZZ)\) 满足 \(|\det M|>1\)，并考虑覆盖端同态
\[
f_M:\TT^r\to\TT^r,\qquad [x]\mapsto[Mx].
\]
令常值逆系统的逆极限为
\[
\Sigma_M:=\varprojlim(\TT^r\xleftarrow{f_M}\TT^r\xleftarrow{f_M}\cdots).
\]
由定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module}（取 \(B_n\equiv M\)）可知其 Stokes 周期模块为
\[
\Pi_{\mathrm{St}}(\Sigma_M)=\bigcup_{k\ge 0}(M^{\mathsf T})^{-k}\ZZ^r\subset\QQ^r.
\]
记
\(
\Pi_k:=(M^{\mathsf T})^{-k}\ZZ^r
\)，则 \(\Pi_k\) 为 \(\RR^r\) 中满秩格且
\(
\operatorname{covol}(\Pi_k)=|\det M|^{-k}
\)。
\par
取互异素数 \(p_1,\dots,p_r\) 与指数向量 \(\mathbf r=(r_1,\dots,r_r)\in\NN^r\)，并定义
\[
N(\mathbf r):=\prod_{i=1}^{r}p_i^{r_i},
\qquad
E_r(\mathbf r;R):=\left\{x\in\RR^r:\ \sum_{i=1}^{r}\frac{x_i^2}{p_i^{r_i}}\le R^2\right\}.
\]
则对每个固定 \(k\ge 0\)，存在常数 \(C_{M,k,\mathbf r}>0\) 使得当 \(R\to\infty\) 时
\[
\boxed{\;
\card{\Pi_k\cap E_r(\mathbf r;R)}
=\operatorname{Vol}(B_r)\sqrt{N(\mathbf r)}\,R^r\cdot|\det M|^{k}
+O_{M,k,\mathbf r}(R^{r-1})\;}
\]
其中 \(\operatorname{Vol}(B_r)\) 为 \(\RR^r\) 中单位球体积。
从而
\[
\log\card{\Pi_k\cap E_r(\mathbf r;R)}
=r\log R+\tfrac12\log N(\mathbf r)+k\log|\det M|+\log\operatorname{Vol}(B_r)+o(1).
\]
\end{proposition}

\begin{proof}
由 \(\Pi_k=(M^{\mathsf T})^{-k}\ZZ^r\) 得
\(
\operatorname{covol}(\Pi_k)=|\det(M^{\mathsf T})^{-k}|=|\det M|^{-k}
\)。
对满秩欧氏格与光滑凸体族应用体积近似（Lipschitz 原理）可得
\[
\card{\Pi_k\cap E_r(\mathbf r;R)}
=\frac{\operatorname{Vol}(E_r(\mathbf r;R))}{\operatorname{covol}(\Pi_k)}+O_{M,k,\mathbf r}(R^{r-1}).
\]
另一方面，轴对齐椭球体积满足
\(
\operatorname{Vol}(E_r(\mathbf r;R))=\operatorname{Vol}(B_r)\sqrt{N(\mathbf r)}\,R^r
\)。
代回即得主式；对主式取对数并吸收相对误差得到对数展开。证毕。
\end{proof}
