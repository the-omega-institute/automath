% Normalized Stokes--L2 scaling on N-adic solenoids.

\begin{theorem}[\texorpdfstring{$N$}{N}-adic solenoid 上归一化 Stokes--\(L^2\) 能量的闭式与维数临界]\label{thm:app-solenoid-normalized-stokes-l2-scaling-critical-degree}
取 \(r\ge 1\)、\(N\ge 2\)，在平坦度量下考虑 \(r\)-维环面 \(\TT^r\) 及其覆盖自映射
\[
f:\TT^r\to\TT^r,\qquad f(x)=Nx.
\]
令 \(S_f:=\varprojlim(\TT^r\xleftarrow{f}\TT^r\xleftarrow{f}\cdots)\)，则 \(S_f\) 同构于 \(r\)-维 \(N\)-solenoid \(\Sigma_{N,r}\)（见定理 \ref{thm:cdim-localized-division-prime-fiber-no-growth} 的记号）。
\par
对任意光滑 \(q\)-形式 \(\beta\in\Omega^q(\TT^r)\)（\(0\le q\le r\)），定义归一化动力学 Stokes--\(L^2\) 能量
\[
\mathcal N_n(\beta)
:=
\frac{1}{N^{rn}}\int_{\TT^r}\bigl|f^{n\ast}\beta\bigr|^{2}\,d\mathrm{vol}.
\]
则对所有 \(n\ge 0\) 成立精确闭式
\[
\boxed{\;
\mathcal N_n(\beta)
=
N^{2n(q-r)}\int_{\TT^r}|\beta|^{2}\,d\mathrm{vol}.\;}
\]
从而出现严格的“阶数--维数临界”：
\(\mathcal N_n(\beta)\) 在 \(q=r\) 时恒等不变，在 \(q<r\) 时以比率 \(N^{2(q-r)}<1\) 指数衰减。
\leanverified{paper\_app\_solenoid\_normalized\_stokes\_l2\_scaling\_critical\_degree}
\end{theorem}

\begin{proof}
在平坦坐标下有 \(f^\ast(dx_i)=N\,dx_i\)，故对任意 \(q\)-形式 \(\beta\) 有点态伸缩
\[
|f^{n\ast}\beta(x)|^{2}
=N^{2nq}\,|\beta(f^{n}x)|^{2}.
\]
因此
\[
\int_{\TT^r}|f^{n\ast}\beta(x)|^{2}\,dx
=N^{2nq}\int_{\TT^r}|\beta(f^{n}x)|^{2}\,dx.
\]
令 \(y=f^{n}x\) 并注意 \(dy=N^{rn}dx\)，得到
\[
\int_{\TT^r}|\beta(f^{n}x)|^{2}\,dx
=N^{-rn}\int_{\TT^r}|\beta(y)|^{2}\,dy.
\]
合并并再乘以归一化因子 \(N^{-rn}\) 即得结论。证毕。
\end{proof}
