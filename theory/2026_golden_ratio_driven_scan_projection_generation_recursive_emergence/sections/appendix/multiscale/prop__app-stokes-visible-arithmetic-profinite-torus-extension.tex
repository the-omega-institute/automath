\begin{corollary}\label{cor:app-stokes-period-module-covector-normalization}
在定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 的记号下，
令
\[
D_n:=A_n^{\mathsf T}\in \GL_r(\ZZ)\qquad(n\ge 0),
\]
并记离散群
\[
A:=\bigcup_{n\ge 0}D_n^{-1}\ZZ^r\ \subset\ \QQ^r.
\]
则
\[
\Pi_{\mathrm{St}}(\mathcal S)=A.
\]
此外，条件 \(D_n^{-1}D_{n+1}\in M_r(\ZZ)\) 对所有 \(n\) 成立。
\end{corollary}
\begin{proof}
由定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 的定义，
\(
\Pi_{\mathrm{St}}(\mathcal S)=\bigcup_{n\ge 0}(A_n^{\mathsf T})^{-1}\ZZ^r
\)；
代入 \(D_n:=A_n^{\mathsf T}\) 即得第一式。
又 \(A_{n+1}=B_nA_n\) 给出 \(D_n^{-1}D_{n+1}=(A_n^{\mathsf T})^{-1}(A_{n+1}^{\mathsf T})=B_n^{\mathsf T}\in M_r(\ZZ)\)。证毕。
\end{proof}

\begin{proposition}\label{prop:app-stokes-period-module-path-integral-reconstruction}
在定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 的设定下，
记 \(\pi_n:\mathcal S\to \TT^r\) 为第 \(n\) 层投影，并取 \(\TT^r\) 上标准不变 \(1\)-形式基 \(d\boldsymbol\theta=(d\theta_1,\dots,d\theta_r)\)。
令 \(\mathcal I_{\mathrm{St}}(\mathcal S)\) 为所有连续曲线 \(\gamma:[0,1]\to\mathcal S\) 满足 \(\gamma(0)=0\) 且存在某个 \(n\) 使 \(\pi_n\circ\gamma\) 为 \(\TT^r\) 中基于 \(0\) 的回路之集合。
定义由路径积分生成的周期模块
\[
\Pi_{\mathrm{St}}^{\mathrm{int}}(\mathcal S)
:=\bigcup_{n\ge 0}(A_n^{\mathsf T})^{-1}
\left\{\int_{\pi_n\circ\gamma}d\boldsymbol\theta:\ \gamma\in \mathcal I_{\mathrm{St}}(\mathcal S)\right\}
\subset\QQ^r.
\]
则
\[
\Pi_{\mathrm{St}}^{\mathrm{int}}(\mathcal S)=\Pi_{\mathrm{St}}(\mathcal S).
\]
\end{proposition}
\begin{proof}
固定 \(n\)。
任取 \(\TT^r\) 中基于 \(0\) 的回路 \(\alpha\)，则
\(
\int_\alpha d\boldsymbol\theta\in\ZZ^r
\)。
由于各层过渡映射为覆盖映射，\(\alpha\) 可逐层提升为与逆系统相容的路径族，从而存在 \(\gamma\in\mathcal I_{\mathrm{St}}(\mathcal S)\) 使 \(\pi_n\circ\gamma=\alpha\)。
因此
\(
\{\int_{\pi_n\circ\gamma}d\boldsymbol\theta:\gamma\in\mathcal I_{\mathrm{St}}(\mathcal S)\}=\ZZ^r
\)，从而
\[
\Pi_{\mathrm{St}}^{\mathrm{int}}(\mathcal S)=\bigcup_{n\ge 0}(A_n^{\mathsf T})^{-1}\ZZ^r=\Pi_{\mathrm{St}}(\mathcal S),
\]
其中最后一步为定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 的定义。证毕。
\end{proof}

\begin{proposition}\label{prop:app-profinite-torus-extension-stokes-module-classification}
设 \(E\) 为紧连通交换群，并存在短正合列
\[
0\longrightarrow P\longrightarrow E\overset{\pi}{\longrightarrow}\TT^r\longrightarrow 0,
\]
其中 \(P\) 为 profinite 紧群。
令 \(A:=\widehat E=\mathrm{Hom}_{\mathrm{cont}}(E,\TT)\) 为其 Pontryagin 对偶离散群。
则 \(A\) 为秩 \(r\) 的无挠阿贝尔群且含自然嵌入 \(\ZZ^r\hookrightarrow A\)。
将 \(A\) 视为 \(\QQ^r\) 的满秩子群时，置
\[
\Pi_{\mathrm{St}}(E):=A\subset\QQ^r.
\]
则成立：
\begin{enumerate}
  \item \(\ZZ^r\subseteq \Pi_{\mathrm{St}}(E)\subseteq \QQ^r\)，并且 \(\Pi_{\mathrm{St}}(E)\) 的抽象同构类型仅由 \(E\) 决定。
  \item \(E\) 在紧群意义下分裂（即 \(E\cong \TT^r\times P\)）当且仅当 \(\Pi_{\mathrm{St}}(E)=\ZZ^r\)。
  \item 若 \(E_1,E_2\) 为两组此类扩张，则
  \[
  E_1\cong E_2
  \quad\Longleftrightarrow\quad
  \Pi_{\mathrm{St}}(E_1)\cong \Pi_{\mathrm{St}}(E_2)
  \]
  （右侧为交换群同构）。
\end{enumerate}
\end{proposition}
\begin{proof}
对偶化短正合列得到
\[
0\longrightarrow \widehat{\TT^r}\longrightarrow \widehat E\longrightarrow \widehat P\longrightarrow 0.
\]
其中 \(\widehat{\TT^r}\cong \ZZ^r\)，而 \(\widehat P\) 为纯挠群。
因此 \(A=\widehat E\) 为秩 \(r\) 的无挠群且含 \(\ZZ^r\)，得 (1) 的第一句；其余为 Pontryagin 对偶的反变等价性给出。
\par
对 (2)，\(E\cong \TT^r\times P\) 当且仅当对偶短正合列分裂，即 \(A\cong \ZZ^r\oplus \widehat P\)。
但 \(A\) 无挠，故只能发生在 \(\widehat P=0\) 且 \(A=\ZZ^r\) 的情形；反之若 \(A=\ZZ^r\)，则 \(E\cong \widehat{\ZZ^r}\cong \TT^r\)，并与 \(P=0\) 的分裂一致。
\par
(3) 由 Pontryagin 对偶为范畴等价立得：\(E_1\cong E_2\iff \widehat{E_1}\cong \widehat{E_2}\iff \Pi_{\mathrm{St}}(E_1)\cong \Pi_{\mathrm{St}}(E_2)\)。证毕。
\end{proof}

\begin{proposition}\label{prop:app-toral-solenoid-cech-h1-direct-limit}
取整数矩阵 \(M\in M_r(\ZZ)\) 满足 \(|\det M|>1\)，令 \(f_M:\TT^r\to\TT^r\) 为覆盖端同态 \([x]\mapsto [Mx]\)，并定义逆极限
\[
\Sigma_M:=\varprojlim\bigl(\TT^r\xleftarrow{f_M}\TT^r\xleftarrow{f_M}\cdots\bigr).
\]
则 Čech 上同调满足自然同构
\[
\check H^1(\Sigma_M;\ZZ)\ \cong\ \varinjlim\bigl(\ZZ^r\xrightarrow{M^{\mathsf T}}\ZZ^r\xrightarrow{M^{\mathsf T}}\cdots\bigr)
\ \cong\ \bigcup_{k\ge 0}(M^{\mathsf T})^{-k}\ZZ^r\ \subset\ \QQ^r.
\]
并且在定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 的意义下，
\[
\Pi_{\mathrm{St}}(\Sigma_M)=\bigcup_{k\ge 0}(M^{\mathsf T})^{-k}\ZZ^r.
\]
\end{proposition}
\begin{proof}
由于 \(\TT^r\) 为有限 CW 复形且 \(f_M\) 为覆盖映射，逆极限空间的 Čech 上同调与底空间上同调在拉回 \(f_M^\ast\) 下的直极限一致：
\[
\check H^1(\Sigma_M;\ZZ)\cong \varinjlim\bigl(H^1(\TT^r;\ZZ)\xrightarrow{f_M^\ast}H^1(\TT^r;\ZZ)\to\cdots\bigr).
\]
标准同构 \(H^1(\TT^r;\ZZ)\cong \ZZ^r\) 下，\(f_M^\ast\) 恰为 \(M^{\mathsf T}\) 作用，得第一式。
将直极限嵌入 \(\QQ^r\) 的并集表示为：把第 \(k\) 层的 \(v\in\ZZ^r\) 送入 \((M^{\mathsf T})^{-k}v\)；良定义性与满射性由直极限等价关系直接验证，得到第二式。
最后一式为定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 在常值逆系统 \(B_n\equiv M\) 下的特例。证毕。
\end{proof}
