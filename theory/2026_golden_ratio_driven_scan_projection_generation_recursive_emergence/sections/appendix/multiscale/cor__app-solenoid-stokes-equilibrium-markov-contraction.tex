% Markov contraction induced by equidegree correspondences on solenoids.

\begin{corollary}[等度对应子诱导的 Stokes 平衡测度与 Markov 收缩性]\label{cor:app-solenoid-stokes-equilibrium-markov-contraction}
\leanverified{paper\_app\_solenoid\_stokes\_equilibrium\_markov\_contraction}
令 \(S\) 为一紧连通 solenoid 型紧交换群（例如定理 \ref{thm:app-high-rank-solenoid-pontryagin-stokes-period-module} 的 \(\mathcal S\)，或定理 \ref{thm:cdim-localized-division-prime-fiber-no-growth} 的 \(\Sigma_{N,r}\)），并记 \(\mu\) 为其 Haar 概率测度。
设 \(C=(Z,s,t)\) 为 \(S\) 上的有限对应子，其中 \(Z\) 为紧交换群且
\[
S\xleftarrow{s}Z\xrightarrow{t}S
\]
为连续满射群同态，并满足等度条件 \(|\ker s|=|\ker t|<\infty\)。
对 \(f\in L^2(\mu)\) 定义 Markov 传递算子
\[
(P_C f)(y):=\frac{1}{|\ker s|}\sum_{z\in t^{-1}(y)} f\bigl(s(z)\bigr),
\]
其中求和按有限纤维的计数测度理解。
则：
\begin{enumerate}
  \item \(\mu\) 为 \(P_C\) 的平衡测度：\(\int_S P_C f\,d\mu=\int_S f\,d\mu\) 对所有 \(f\in L^1(\mu)\) 成立；
  \item \(P_C\) 在 \(L^2(\mu)\) 上为收缩：\(\|P_C f\|_{L^2(\mu)}\le \|f\|_{L^2(\mu)}\)。
\end{enumerate}
\end{corollary}

\begin{proof}
由于 \(s,t\) 为满射群同态且核有限，Haar 测度在推前下保持：\(s_\ast \mu_Z=\mu\)、\(t_\ast \mu_Z=\mu\)，其中 \(\mu_Z\) 为 \(Z\) 的 Haar 概率测度。
并且对任意可积函数 \(g\in L^1(\mu_Z)\) 有纤维平均恒等式
\[
\int_S \left(\frac{1}{|\ker t|}\sum_{z\in t^{-1}(y)} g(z)\right)\,d\mu(y)
=
\int_Z g(z)\,d\mu_Z(z).
\]
取 \(g(z):=f(s(z))\) 并用 \(|\ker s|=|\ker t|\) 得
\[
\int_S P_C f\,d\mu
=
\frac{|\ker t|}{|\ker s|}\int_Z f(s(z))\,d\mu_Z(z)
=
\int_S f\,d\mu,
\]
从而得到平衡性。
\par
对每个 \(y\) 的有限平均由 Jensen 不等式给出点态估计
\[
|P_C f(y)|^2\le P_C(|f|^2)(y).
\]
两边积分并用平衡性即得
\[
\|P_C f\|_{L^2(\mu)}^{2}
=\int_S |P_C f|^2\,d\mu
\le \int_S P_C(|f|^2)\,d\mu
=\int_S |f|^2\,d\mu
=\|f\|_{L^2(\mu)}^{2}.
\]
证毕。
\end{proof}
