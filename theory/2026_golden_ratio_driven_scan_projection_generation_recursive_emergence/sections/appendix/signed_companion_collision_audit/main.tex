\section{带符号伴随代理与碰撞矩证书}\label{app:signed-companion-collision-audit}

本附录隔离从碰撞矩递推重构的带符号伴随矩阵背后的线性代数约定。其目的有二：第一，记录由带符号 Frobenius 约定所强制的行列式与外平方系数；第二，将这些形式代理计算与更强的断言（即同一矩阵为非负碰撞核或已流等价于此类核）分离。

\subsection{带符号伴随约定}

设 \(R\) 为交换环，\(d\geq 2\)，令
\[
    c=(c_1,\ldots,c_d)\in R^d.
\]
采用带符号递推伴随
\[
A(c)=
\begin{pmatrix}
0&1&0&\cdots&0\\
0&0&1&\cdots&0\\
\vdots&\vdots&\vdots&\ddots&\vdots\\
0&0&0&\cdots&1\\
-c_d&-c_{d-1}&-c_{d-2}&\cdots&-c_1
\end{pmatrix}.
\]
等价地，经翻转置换共轭后，此即首行为 \((-c_1,\ldots,-c_d)\)、次对角线元素为 \(1\) 的 Frobenius 伴随矩阵。此约定已在 Omega Lean 文件
\texttt{\detokenize{lean4/Omega/Folding/CollisionKernel.lean}} 中形式化。

\begin{lemma}[带符号伴随系数]\label{lem:signed-companion-coefficients}
\leanverified{paper\_signed\_companion\_coefficients}
\(A(c)\) 的特征多项式为
\[
    \chi_{A(c)}(x)=x^d+c_1x^{d-1}+c_2x^{d-2}+\cdots+c_d.
\]
从而
\[
    \tr A(c)=-c_1,\qquad e_2(A(c))=c_2,
\]
其中 \(e_2\) 表示特征值的第二初等对称多项式，等价于 \(\Lambda^2 A(c)\) 在分裂域上的迹。
\end{lemma}

\begin{proof}
所示矩阵为首一多项式 \(x^d+c_1x^{d-1}+\cdots+c_d\) 的通常 Frobenius 伴随矩阵（相差翻转置换共轭）。因此其特征多项式即为所示多项式。与
\[
    \chi_A(x)=x^d-e_1(A)x^{d-1}+e_2(A)x^{d-2}-\cdots
\]
比较系数即得 \(e_1(A(c))=\tr A(c)=-c_1\) 及 \(e_2(A(c))=c_2\)。
\end{proof}

\begin{theorem}[带符号伴随行列式]\label{thm:signed-companion-det}
对带符号伴随 \(A(c)\)，
\[
    \det(I-A(c))=1+\sum_{i=1}^d c_i.
\]
相比之下，普通递推伴随 \(C(c)\) 的特征多项式为 \(x^d-c_1x^{d-1}-\cdots-c_d\)，满足
\[
    \det(I-C(c))=1-\sum_{i=1}^d c_i.
\]
因此两种行列式约定服从万有转换规则
\[
    \det(I-C(c))+\det(I-A(c))=2.
\]
\leanverified{signedCompanionDet}
\leanverified{paper\_signed\_companion\_det}
\end{theorem}

\begin{proof}
带符号恒等式由引理 \ref{lem:signed-companion-coefficients} 在 \(x=1\) 处赋值特征多项式得到：
\[
    \det(I-A(c))=\chi_{A(c)}(1)=1+\sum_{i=1}^d c_i.
\]
普通递推伴随以 \(-c_i\) 替换系数，给出 \(1-\sum_i c_i\)。两式相加即得转换规则。
\end{proof}

\begin{corollary}[\(q=6,7\) 带符号代理行列式]\label{cor:signed-companion-q6-q7}
对审计的递推向量
\[
\begin{aligned}
c_6&=(2,17,28,88,-26,4,-4),\\
c_7&=(2,26,74,311,-34,84,-42),
\end{aligned}
\]
带符号伴随末行为
\[
\begin{aligned}
r_6&=(4,-4,26,-88,-28,-17,-2),\\
r_7&=(42,-84,34,-311,-74,-26,-2).
\end{aligned}
\]
其带符号行列式代理值为
\[
    \det(I-A(c_6))=110,\qquad \det(I-A(c_7))=422.
\]
从而带符号行列式超出偶 Fibonacci 基线的余量为
\[
    110-F_{10}=55,\qquad 422-F_{12}=278.
\]
\leanverified{signedCompanionDet6,signedCompanionDet7}
\end{corollary}

\begin{proof}
末行为 \((-c_d,\ldots,-c_1)\)。应用定理 \ref{thm:signed-companion-det} 得
\[
1+\sum_i c_{6,i}=1+2+17+28+88-26+4-4=110,
\]
以及
\[
1+\sum_i c_{7,i}=1+2+26+74+311-34+84-42=422.
\]
由 \(F_{10}=55\)、\(F_{12}=144\)，两个余量分别为 \(55\) 与 \(278\)。
\end{proof}

\begin{remark}[代理边界]\label{rem:signed-companion-proxy-boundary}
推论 \ref{cor:signed-companion-q6-q7} 是关于从矩递推重构的带符号伴随矩阵的行列式陈述，不应被解读为 \(A(c_6)\) 与 \(A(c_7)\) 已被认证为非负碰撞核。将带符号代理提升为 Bowen--Franks/SFT 不变量需要额外的非负表示或流等价证书。
\end{remark}

\begin{theorem}[有界带符号代理 Lucas 证书]\label{thm:signed-companion-lucas-certificate}
\leanverified{paper\_signed\_companion\_lucas\_certificate}
在审计递推范围 \(2\leq q\leq 23\) 内，等式
\[
    e_2(A_q)=L_q
\]
恰在
\[
    q=3,4,5
\]
处成立。在同一范围内，三重巧合
\[
    e_2(A_q)=L_q=\epsilon_q
\]
恰在 \(q=5\) 处成立。特别地，\(\{6,\ldots,23\}\) 中无 \(q\) 满足 \(e_2(A_q)=L_q\) 或 \(\epsilon_q=L_q\)。
\end{theorem}

\begin{proof}
恒等式 \(e_2(A_q)=c_{q,2}\) 与 \(\epsilon_q=\abs{1+\sum_i c_{q,i}}-F_{2q-2}\) 将声明归结为审计整数递推表。直接枚举给出表 \ref{tab:signed-companion-collision-audit-q2-23}。外部 Lucas 证书含 \(22\) 行（\(q=2,\ldots,23\)），其真行恰为 \(q=3,4,5\)。将余量列与 \(L_q\) 比较后仅余 \(q=5\)。计算由
\[
    \texttt{\detokenize{scripts/equational_theory/signed_companion_collision_audit.py}}
\]
及其 JSON 证书
\[
    \texttt{\detokenize{scripts/equational_theory/exterior_lucas_certificate_q2_23.json}}
\]
验证。
\end{proof}

\begin{corollary}[\(q=5\) 巧合的分解]\label{cor:signed-companion-q5-coincidence-split}
\leanverified{paper\_signed\_companion\_q5\_coincidence\_split}
\(q=5\) 恒等式
\[
    L_5=e_2(A_5)=\epsilon_5=11
\]
是两个独立系数方程的同时满足：
\[
    c_{5,2}=L_5,
    \qquad
    1+\sum_i c_{5,i}-F_8=L_5.
\]
第一个方程仅在审计走廊 \(q=3,4,5\) 内持续，第二个方程在 \(q=5\) 之后无审计解。
\end{corollary}

\begin{proof}
对 \(c_5=(2,11,8,20,-10)\)，引理 \ref{lem:signed-companion-coefficients} 给出 \(e_2(A_5)=c_{5,2}=11=L_5\)。定理 \ref{thm:signed-companion-det} 给出
\[
    1+\sum_i c_{5,i}=1+2+11+8+20-10=32,
\]
又 \(F_8=21\)，故 \(\epsilon_5=32-21=11=L_5\)。其余断言恰为定理 \ref{thm:signed-companion-lucas-certificate} 的有界证书。
\end{proof}

\subsection{融合环影子与障碍}

\begin{corollary}[融合相干性未被带符号代理认证]\label{cor:signed-companion-fusion-obstruction}
形式 Omega 数据支持 Fibonacci 融合环影子：
\[
    M^2=M+I,
\]
其自然中心化子为 \(\NN[M]\)，路径独立集模型与 Fibonacci 融合路径基底双射。但这些数据不认证带符号碰撞矩代理的结合子、\(F\)-符号、辫子或五角相干性。更进一步，任何将 \(q=5\) 等式 \(e_2(A_5)=L_5\) 解释为持续 Fibonacci-anyon 相干律的迹保持机制，在 \(q=6\) 处已被障碍：
\[
    e_2(A_6)=17\neq 18=L_6.
\]
\leanverified{goldenMeanAdjacencyNat_sq}
\leanverified{paper_pom_centralizer_semiring_fibonacci_fusion}
\leanverified{paper_pom_fiber_fibonacci_anyon_fusion_path_bijection}
\end{corollary}

\begin{proof}
融合环与路径基底陈述为声明中引用的已形式化半环层影子。幺范畴或辫子范畴化需要额外数据：结合子、\(F\)-符号、辫子映射以及五角或六角相干性，这些均不属于上述带符号伴随证书。若 \(q=5\) 外部 Lucas 等式由一个随 \(q\) 持续的迹保持律解释，则它将强制 \(e_2(A_q)=L_q\) 在 \(q=5\) 之后继续成立。审计系数 \(c_{6,2}=17\) 而 \(L_6=18\)，在下一个值处即给出矛盾。
\end{proof}

\subsection{非负提升与流等价证书}

带符号伴随代理在推论 \ref{cor:signed-companion-q6-q7} 中记录了 \(q=6,7\) 的行列式与外平方数据，但带符号矩阵 \(A(c_q)\) 含有负元素，因而不直接定义有限型子移位。下述命题构造显式非负不可约矩阵，保持审计的行列式、外平方与 Bowen--Franks 数据。

\begin{proposition}[非负 \(q=6,7\) 提升]\label{prop:nonneg-lift-q6-q7}
\leanverified{paper\_nonneg\_lift\_q6\_q7}
存在非负整数矩阵
\[
B_6=
\begin{pmatrix}
1&0&4&5\\
2&8&0&1\\
1&4&0&2\\
2&1&6&4
\end{pmatrix},
\qquad
B_7=
\begin{pmatrix}
9&6&3&0&2\\
4&1&1&1&2\\
2&1&1&4&3\\
1&1&1&4&0\\
4&4&0&1&1
\end{pmatrix},
\]
满足
\[
    \det(I-B_6)=110,\quad e_2(B_6)=17,\quad \mathrm{BF}(B_6)\cong \ZZ/110\ZZ,
\]
\[
    \det(I-B_7)=422,\quad e_2(B_7)=26,\quad \mathrm{BF}(B_7)\cong \ZZ/422\ZZ.
\]
\(B_6\) 与 \(B_7\) 均为强连通非置换矩阵且具有对角环，因此定义不可约非置换有限型子移位。由 Franks/Bowen--Franks 分类，\(B_6\) 与 \(B_7\) 分别属于不变量为 \((\ZZ/110\ZZ,+)\) 与 \((\ZZ/422\ZZ,+)\) 的正行列式流等价类。
\end{proposition}

\begin{proof}
行列式与外平方系数由 Newton 恒等式 \(e_2(M)=(\tr M)^2-\tr(M^2))/2\) 及 Bareiss 消去法直接计算。对 \(I-B_q\) 的所有 \(k\times k\) 子式（\(1\leq k<n\)）取最大公因数，均为 \(1\)。由行列式因子定理，Smith 标准形为
\[
    \mathrm{SNF}(I-B_6)=\mathrm{diag}(1,1,1,110),\qquad
    \mathrm{SNF}(I-B_7)=\mathrm{diag}(1,1,1,1,422),
\]
从而 Bowen--Franks 群为所述阶的循环群。强连通性由有向支撑图直接检验。每一矩阵均含正对角元素，故周期为 \(1\)；行和分别为 \((10,11,7,13)\) 与 \((20,9,11,7,10)\)，非置换矩阵行模式。验证脚本见
\[
    \texttt{\detokenize{scripts/equational_theory/nonnegative_lift_flow_certificate.py}}.
\]
\end{proof}

\begin{remark}[提升边界]\label{rem:nonneg-lift-scope}
命题 \ref{prop:nonneg-lift-q6-q7} 是审计行列式/外平方/Bowen--Franks 数据的非负提升，不声称原带符号伴随已是正则非负碰撞核，亦不保持迹或完整递推特征多项式。
\end{remark}
