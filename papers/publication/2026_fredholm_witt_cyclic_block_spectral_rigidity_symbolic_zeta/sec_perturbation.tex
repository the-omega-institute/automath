\section{Perturbative Applications}
\label{sec:perturbation}

\subsection{Finite-part constants and determinant-rigidity}

\begin{definition}[Reduced determinant]\label{def:reduced-determinant}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of matrices near
  \(\theta_0\), let \(\lambda(\theta)=\rho(A_\theta)\) be the simple
  Perron eigenvalue, and set
  \[
    B_\theta:=I-\frac{A_\theta}{\lambda(\theta)}.
  \]
  Write the non-Perron Jordan spectrum of \(B_\theta\) as
  \(\{0\}\cup\{\nu_j(\theta):j=2,\dots,d\}\). The reduced determinant is
  \[
    \det\nolimits'(B_\theta)
    :=\prod_{j=2}^{d}\nu_j(\theta)
    =\prod_{\mu\in\spec(A_\theta)\setminus\{\lambda(\theta)\}}
      \left(1-\frac{\mu}{\lambda(\theta)}\right).
  \]
  and \(C(A_\theta)=\det\nolimits'(B_\theta)^{-1}\).
\end{definition}

\begin{theorem}[Reduced-determinant gradient formula]
  \label{thm:finite-part-reduced-determinant-group-inverse-gradient}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Choose left and
  right Perron vectors \(\ell_\theta,r_\theta\) normalised by
  \(\ell_\theta^{\mathsf T}r_\theta=1\), let
  \[
    P_\theta:=r_\theta\ell_\theta^{\mathsf T},\quad
    R_\theta:=B_\theta^{\#},
  \]
  where \(B_\theta=I-A_\theta/\lambda(\theta)\) and \(B_\theta^{\#}\) is
  the group inverse.
  Then
  \[
    \partial_\theta\log C(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right),
  \]
  with
  \[
    \partial_\theta\lambda(\theta)
    =\ell_\theta^{\mathsf T}(\partial_\theta A_\theta)r_\theta.
  \]
\end{theorem}

\begin{proof}
  Since \(\lambda(\theta)\) is a simple eigenvalue
  \cite[Chapter~II, \S1]{Kato1995Perturbation}, \(B_\theta\) has a simple
  zero eigenvalue and all other eigenvalues are non-zero; hence the group
  inverse \(R_\theta\) exists; see \cite[Chapter~4]{CampbellMeyer1979}.
  Because the spectrum depends analytically,
  \(P_\theta\) and \(R_\theta=B_\theta^{\#}\) vary analytically on
  \(\theta\).
  On \((I-P_\theta)\CC^d\), \(\det\nolimits'(B_\theta)\) is the
  ordinary determinant, so by Jacobi's formula
  \[
    \partial_\theta\log\det\nolimits'(B_\theta)
    =\Tr(R_\theta\,\partial_\theta B_\theta).
  \]
  Since \(C(A_\theta)=\det\nolimits'(B_\theta)^{-1}\), this gives
  \[
    \partial_\theta\log C(A_\theta)
    =-\partial_\theta\log\det\nolimits'(B_\theta)
    =-\Tr(R_\theta\,\partial_\theta B_\theta),
  \]
  while
  \[
    \partial_\theta B_\theta
    =-\frac{\partial_\theta A_\theta}{\lambda(\theta)}
      +\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
       A_\theta.
  \]
  Substituting yields the stated formula.
\end{proof}

\begin{theorem}[Holomorphic variation of the Abel finite part]
  \label{thm:logM-holomorphic-variation}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Then
  \(\theta\mapsto\log\MM(A_\theta)\) is holomorphic near \(\theta_0\)
  and
  \[
    \frac{\dd}{\dd\theta}\log\MM(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right)
    +\sum_{k\ge 2}\frac{\mu(k)}{k}
      \frac{\dd}{\dd\theta}
      \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  The correction series converges absolutely and locally uniformly near
  \(\theta_0\).
\end{theorem}

\begin{proof}
  Apply \Cref{thm:finite-part-primitive-closed-form} to each \(A_\theta\).
  For \(k\ge 2\), set
  \[
    \Phi_k(\theta):=\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  Choose \(\epsilon>0\) so that
  \(|\lambda(\theta)|\ge\lambda_{\min}>1\) for
  \(|\theta-\theta_0|\le\epsilon\). Since \(\|A_\theta\|\) is bounded on
  this compact set, Jacobi gives
  \[
    \left|\frac{\mu(k)}{k}\partial_\theta\Phi_k(\theta)\right|
    \le
    \frac{C_1}{k\,\lambda_{\min}^{k}}
    +\frac{C_2}{\lambda_{\min}^{k-1}},
  \]
  and therefore
  \[
    \sum_{k\ge2}\frac{\mu(k)}{k}\partial_\theta\Phi_k
  \]
  converges absolutely and locally uniformly, so differentiation is
  termwise justified. The reduced term is given by
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient}.
  For \(\Phi_k\), Jacobi's formula gives
  \[
    \frac{\dd}{\dd\theta}\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}
    =\Tr\!\left((I-\lambda(\theta)^{-k}A_\theta)^{-1}
      \left(\frac{\partial_\theta A_\theta}{\lambda(\theta)^k}
      -\frac{k\,\partial_\theta\lambda(\theta)}{\lambda(\theta)^{k+1}}
      A_\theta\right)\right),
  \]
  a finite-dimensional Jacobi computation with no matrix-log branch.
\end{proof}

\begin{corollary}[Finite parts depend only on the determinant]
  \label{cor:finite-part-zeta-rigidity}
  Let \(A\) and \(B\) be primitive matrices. If
  \[
    \det(I-zA)=\det(I-zB),
  \]
  equivalently \(\zeta_A=\zeta_B\), then \(A\) and \(B\) have the same
  Perron root and satisfy
  \[
    C(A)=C(B),
    \qquad
    \MM(A)=\MM(B).
  \]
  More generally, let \(\theta\mapsto A_\theta\) and
  \(\theta\mapsto B_\theta\) be holomorphic families of primitive
  matrices near \(\theta_0\). If
  \[
    \det(I-zA_\theta)=\det(I-zB_\theta)
  \]
  in a neighbourhood of \((\theta_0,0)\), then
  \[
    C(A_\theta)=C(B_\theta),
    \qquad
    \MM(A_\theta)=\MM(B_\theta)
  \]
  near \(\theta_0\).
\end{corollary}

\begin{proof}
  By \Cref{thm:fredholm-spectral-rigidity}, determinant equality implies
  equality of the non-zero spectra with algebraic multiplicity. In
  particular the Perron roots agree; write \(\lambda\) for this common
  Perron root. The reduced determinants
  \[
    \prod_{\mu\in\spec(A)\setminus\{\lambda\}}
      \left(1-\frac{\mu}{\lambda}\right)
    \quad\text{and}\quad
    \prod_{\nu\in\spec(B)\setminus\{\lambda\}}
      \left(1-\frac{\nu}{\lambda}\right)
  \]
  coincide, so \(C(A)=C(B)\) by \eqref{eq:reduced-det-constant}. The
  formula in \Cref{thm:finite-part-primitive-closed-form} then gives
  \(\MM(A)=\MM(B)\). The family statement follows pointwise in
  \(\theta\), and hence the resulting holomorphic functions agree near
  \(\theta_0\).
\end{proof}

\subsection{Twisted determinants and spectral-gap limit laws}

Let \((X_A^+,\sigma)\) be the mixing subshift determined by a primitive
matrix \(A\), let \(\mu\) be its Parry measure, and let
\(f:X_A^+\to\RR\) be locally constant. For \(\theta\in\CC\) near \(0\),
define the twisted transfer operator
\[
  (\mathcal L_\theta g)(x)
  :=\sum_{\sigma y=x} e^{\theta f(y)}g(y).
\]

\begin{proposition}[Uniform zero-free disc for twisted determinants]
  \label{prop:twisted-zero-free}
  Assume \(f\) is not a coboundary modulo constants. After recoding if
  needed to a one-step model, let \(M_t\) be the weighted matrices and
  \(\lambda=\rho(B)\) the Perron root of the underlying primitive matrix
  \(B\). For every compact set \(K\Subset\RR\setminus\{0\}\) there
  exists \(\eta_K>0\) such that
  \[
    \det(I-zM_t)\neq 0
    \qquad (t\in K,\ \abs{z}\le (1+\eta_K)\lambda^{-1}).
  \]
  Equivalently, the twisted zeta-functions
  \[
    \zeta_{B,t}(z):=\det(I-zM_t)^{-1}
  \]
  are holomorphic on a disc of radius strictly larger than
  \(\lambda^{-1}\), uniformly for \(t\in K\). Moreover there exist
  constants \(C_K>0\) and \(\vartheta_K\in(0,1)\) such that
  \[
    \sup_{t\in K}\abs{\Tr(M_t^n)}
    \le C_K(\vartheta_K\lambda)^n,
    \qquad n\ge 1.
  \]
  In the one-step coding, these traces are the weighted periodic-orbit
  sums, so periodic data exhibit uniform exponential cancellation away
  from \(t=0\).
\end{proposition}

\begin{proof}
  Recoding if needed, \(f\) becomes a one-step function \(\varphi\) on a
  primitive matrix \(B\), and
  \[
    M_t(b',b):=B_{b',b}e^{it\varphi(b)}.
  \]
  Set \(\rho(t)=\rho(M_t)\). Since \(|M_t|=B\), Perron--Frobenius
  comparison gives \(\rho(t)\le\lambda\). If \(\rho(t)=\lambda\) for
  some \(t\neq0\), a Wielandt-type equality case gives a unimodular
  diagonal \(D_t\) and phase \(c_t\) with
  \(M_t=e^{itc_t}D_t^{-1}BD_t\), hence \(it\varphi\) is a coboundary and
  \(f\) is a coboundary modulo constants by Liv\v{s}ic. Since this is
  excluded, for every compact \(K\Subset\RR\setminus\{0\}\),
  \[
    \rho_K:=\max_{t\in K}\rho(t)<\lambda.
  \]
  Choose \(\eta_K>0\) so that \((1+\eta_K)\rho_K<\lambda\). If
  \(\mu\in\spec(M_t)\) and \(\abs{z}\le (1+\eta_K)\lambda^{-1}\), then
  \[
    \abs{z\mu}\le \frac{(1+\eta_K)\rho_K}{\lambda}<1,
  \]
  so \(1-z\mu\neq0\). Hence \(\det(I-zM_t)\neq0\) on the stated region.

  If \(\mu_1(t),\dots,\mu_d(t)\) are the eigenvalues of \(M_t\), then
  \[
    \abs{\Tr(M_t^n)}
    =\left|\sum_{j=1}^d \mu_j(t)^n\right|
    \le d\,\rho_K^n
    =d\left(\frac{\rho_K}{\lambda}\right)^n\lambda^n.
  \]
  Thus the trace bound holds with \(C_K=d\) and
  \(\vartheta_K=\rho_K/\lambda\). In the one-step model,
  \(\Tr(M_t^n)\) is the usual weighted sum over periodic points of
  period \(n\).
\end{proof}

\begin{proposition}[Spectral-gap central limit theorem]
  \label{thm:clt-spectral}
  Assume \(f\) is not a coboundary modulo constants. Write
  \(\bar f:=f-\int f\,\dd\mu\) for the centred observable. Then there
  exists \(\sigma_f^2>0\) such that
  \[
    \frac{\sum_{k=0}^{n-1}\bar f\circ\sigma^k}{\sqrt n}
    \Longrightarrow \mathcal N(0,\sigma_f^2).
  \]
  Moreover, for every compact set \(K\Subset\RR\setminus\{0\}\) there
  exist constants \(C_K>0\) and \(\vartheta_K\in(0,1)\) such that
  \[
    \sup_{t\in K}
    \left|\int e^{it\sum_{k=0}^{n-1}\bar f\circ\sigma^k}\,\dd\mu\right|
    \le C_K\vartheta_K^n,\qquad n\ge1.
  \]
\end{proposition}

\begin{proof}
  Recoding if needed, \(f\) becomes a one-step function \(\varphi\) on a
  primitive matrix \(B\). For \(t\in\RR\), define weighted matrices
  \[
    M_t(b',b):=B_{b',b}e^{it\varphi(b)}.
  \]
  Let \(\lambda=\rho(B)\) and \(\ell,r\) be Perron vectors of \(B\) with
  \(\ell^\top r=1\). Then
  \[
    \phi_n(t):=
    \int e^{it\sum_{k=0}^{n-1}\bar f\circ\sigma^k}\,\dd\mu
    =\frac{\ell^\top M_t^n r}{\lambda^n}.
  \]

  Set \(\rho(t)=\rho(M_t)\). Since \(|M_t|=B\), Perron--Frobenius
  comparison gives \(\rho(t)\le\lambda\). If \(\rho(t)=\lambda\) for
  some \(t\neq0\), a Wielandt-type equality case gives a unimodular
  diagonal \(D_t\) and phase \(c_t\) with
  \(M_t=e^{itc_t}D_t^{-1}BD_t\), hence \(it\varphi\) is a coboundary and
  \(f\) is a coboundary modulo constants by Liv\v{s}ic. Since this is
  excluded, for every compact \(K\Subset\RR\setminus\{0\}\),
  \[
    \rho_K:=\max_{t\in K}\rho(t)<\lambda.
  \]
  Let \(M_t=P_tJ_tP_t^{-1}\) be a Jordan decomposition. In finite
  dimension there exist \(m\) and \(C_K<\infty\) such that
  \[
    \|M_t^n\|\le C_K n^{m-1}\rho_K^n\le C_K'(\vartheta_K\lambda)^n
  \]
  for some \(\vartheta_K\in(\rho_K/\lambda,1)\). Hence for \(t\in K\),
  \[
    |\phi_n(t)|\le \frac{\|\ell\|\|r\|}{\lambda^n}\|M_t^n\|
    \le C_K'\vartheta_K^n.
  \]

  For the CLT, analytic perturbation of \(M_t\) around \(t=0\) gives a
  simple dominant eigenvalue \(\lambda(t)\) and a spectral projector
  \(P_t\):
  \[
    \phi_n(t)=\lambda(t)^n\ell^\top P_t r + O(\vartheta^n),\qquad |t|<\delta,
  \]
  with \(\lambda(0)=\lambda,\lambda'(0)=0\), and
  \[
    \lambda''(0)=\sigma_f^2\lambda,\qquad \sigma_f^2>0
  \]
  whenever \(f\) is not a coboundary modulo constants
  \cite{Ruelle1969,KellerLiverani1999,HennionHerve2001,Livsic1971}.
  Taking \(t=u/\sqrt n\) and using the above bound away from \(t=0\),
  we obtain
  \[
    \phi_n(u/\sqrt n)\to e^{-\sigma_f^2u^2/2}.
  \]
  L\'evy continuity yields the normal limit law.
\end{proof}

\begin{remark}
  The arguments in
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient,thm:logM-holomorphic-variation,cor:finite-part-zeta-rigidity,prop:twisted-zero-free,thm:clt-spectral}
  rely on three shared ingredients: (1) a simple dominant eigenvalue
  \(\lambda(\theta)\) of the transfer operator, (2) the reduced
  resolvent (or group inverse) on the spectral complement of
  \(\lambda(\theta)\), and (3) analytic dependence of the spectral data
  on the parameter \(\theta\). The operator-theoretic viewpoint of
  \Cref{sec:fredholm} isolates exactly this common core. The point is
  not only that one recovers standard perturbative statements, but that
  determinant data already control the finite-part functionals and the
  zero-free frequency gap away from \(t=0\), so the gradient formulas,
  periodic-orbit cancellation, and the CLT become manifestations of one
  and the same spectral package.
\end{remark}
