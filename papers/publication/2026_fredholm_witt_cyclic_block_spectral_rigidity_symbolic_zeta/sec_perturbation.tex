\section{Applications and Examples}
\label{sec:perturbation}

This section illustrates how the determinant viewpoint from
\Cref{sec:fredholm} organizes standard perturbative and spectral-gap
consequences in symbolic dynamics. All results here are classical or
follow from standard methods; none represent new structural
contributions. Statements are given with brief proof sketches; full
proofs are deferred to \Cref{sec:appendix-proofs}.

\medskip

\noindent\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
\textbf{Standing assumptions for this section.}
Throughout \Cref{sec:perturbation}, the following hypotheses are in
force unless stated otherwise:
\begin{enumerate}[label=\textup{(SA\arabic*)},leftmargin=*,nosep]
  \item\label{SA:primitive}
    \(\theta\mapsto A_\theta\) is a holomorphic family of primitive
    \(d\times d\) matrices near \(\theta_0\in\CC\).
  \item\label{SA:simple-dominant}
    \(A_{\theta_0}\) has a simple dominant eigenvalue
    \(\lambda_0=\rho(A_{\theta_0})>0\) with
    \(\abs{\mu}<\lambda_0\) for every other eigenvalue \(\mu\).
  \item\label{SA:spectral-gap}
    There exist \(q\in(0,1)\) and a neighbourhood \(U\ni\theta_0\) such
    that the spectral-gap ratio satisfies
    \(\sup_{\theta\in U}
    \rho(A_\theta|_{(I-P_\theta)\CC^d})/\abs{\lambda(\theta)}\le q<1\),
    where \(P_\theta\) is the rank-one Riesz projector and
    \(\lambda(\theta)\) the analytic eigenvalue branch from
    \Cref{def:spectral-gap-hypothesis}.
  \item\label{SA:compact-freq}
    In the twisted-determinant and off-origin decay statements,
    frequency sets \(K\) are assumed to be compact subsets of
    \(\RR\setminus\Lambda_f\), where \(\Lambda_f\) is the arithmetic
    lattice obstruction (\Cref{def:arithmetic-lattice}).
  \item\label{SA:non-coboundary}
    When the central limit theorem
    (\Cref{thm:clt-spectral}) or positivity of the asymptotic variance
    is claimed, the observable \(f\) is assumed to be not a coboundary
    modulo constants.
\end{enumerate}}}

\subsection{Analytic eigenvalue branches and algebraic finite-part functionals}

\begin{definition}[Analytic spectral-gap hypothesis]
  \label{def:spectral-gap-hypothesis}
  Let \(\theta\mapsto A(\theta)\) be a holomorphic family of
  \(d\times d\) complex matrices near \(\theta_0\in\CC\). We say that
  \(A(\theta)\) satisfies the \emph{analytic spectral-gap hypothesis at
  \(\theta_0\)} if \(A(\theta_0)\) has a simple eigenvalue
  \(\lambda_0\ne 0\) such that
  \[
    \abs{\mu}<\abs{\lambda_0}
    \quad\text{for every }
    \mu\in\spec\!\bigl(A(\theta_0)\bigr)\setminus\{\lambda_0\}.
  \]
  Under this condition, after shrinking to a simply connected
  neighbourhood \(U\ni\theta_0\), the standard Riesz projection
  \[
    P(\theta)
    :=\frac{1}{2\pi i}
      \int_{\abs{z-\lambda_0}=\delta}
      (zI-A(\theta))^{-1}\,\dd z
  \]
  (for \(\delta>0\) chosen so that no other eigenvalue of
  \(A(\theta_0)\) lies in \(\abs{z-\lambda_0}\le\delta\)) defines a
  holomorphic rank-one projector. The corresponding eigenvalue branch
  \[
    \lambda(\theta):=\Tr(A(\theta)P(\theta))
  \]
  is the unique holomorphic function on \(U\) with \(\lambda(\theta_0)
  =\lambda_0\) and \(A(\theta)P(\theta)=\lambda(\theta)P(\theta)\).
  There exist \(q\in(0,1)\) and a neighbourhood \(U\) such that
  \[
    \sup_{\theta\in U}
    \frac{\rho\!\left(A(\theta)\bigm|_{(I-P(\theta))\CC^d}\right)}
         {\abs{\lambda(\theta)}}
    \le q < 1.
  \]
  When \(A(\theta)\) is a primitive matrix with Perron root \(\lambda_0
  =\rho(A(\theta_0))>0\), this is exactly the Perron--Frobenius
  spectral gap, and \(\lambda(\theta)\) agrees with \(\rho(A(\theta))\)
  on the real slice where \(A(\theta)\) remains primitive.
\end{definition}

\begin{definition}[Algebraic finite-part functionals]
  \label{def:algebraic-finite-part}
  Let \(A\) be a \(d\times d\) matrix satisfying the spectral-gap
  hypothesis with distinguished simple eigenvalue \(\lambda\ne 0\).
  Set \(B:=I-A/\lambda\). The \emph{reduced determinant} is
  \[
    C(A,\lambda)
    :=\det\nolimits'(B)^{-1}
    :=\prod_{\mu\in\spec(A)\setminus\{\lambda\}}
      \!\!\left(1-\frac{\mu}{\lambda}\right)^{-1},
  \]
  where the product is taken with algebraic multiplicity. The
  \emph{Abel finite-part functional} is
  \[
    F(A,\lambda)
    :=\log C(A,\lambda)
      +\sum_{k\ge 2}\frac{\mu(k)}{k}
        \log\det\!\bigl(I-\lambda^{-k}A\bigr)^{-1},
  \]
  where the logarithm is the principal branch.
\end{definition}

\subsection{Finite-part variation and truncation}

The following propositions record the group-inverse gradient formula for
the reduced determinant, normal convergence of the secondary Fredholm
factors, the holomorphic variation of the Abel finite part, and a
rapidly convergent truncation scheme. Assumptions
\ref{SA:primitive}--\ref{SA:spectral-gap} are in force. Proofs are given
in \Cref{sec:appendix-proofs}.

\begin{proposition}[Reduced-determinant gradient formula]
  \label{thm:finite-part-reduced-determinant-group-inverse-gradient}
  Under \Cref{def:spectral-gap-hypothesis}, with left and right
  eigenvectors \(\ell_\theta,r_\theta\) normalized by
  \(\ell_\theta^{\mathsf T}r_\theta=1\) and
  \(P_\theta=r_\theta\ell_\theta^{\mathsf T}\), and with group inverse
  \(R_\theta:=(I-A_\theta/\lambda(\theta))^{\#}\),
  \[
    \partial_\theta\log C(A_\theta,\lambda(\theta))
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right).
  \]
\end{proposition}

\begin{proposition}[Normal convergence of secondary Fredholm factors]
  \label{prop:secondary-fredholm-normal-convergence}
  Under \Cref{def:spectral-gap-hypothesis}, define holomorphic branches
  \(\Phi_k(\theta):=\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}\)
  for \(k\ge 2\). There exist \(C_U>0\) and \(q_U\in(0,1)\) such that
  \[
    \sup_{\theta\in U}\abs{\Phi_k(\theta)}
    +\sup_{\theta\in U}\abs{\partial_\theta\Phi_k(\theta)}
    \le C_U k q_U^k
    \qquad (k\ge 2).
  \]
  Consequently \(\sum_{k\ge 2}\mu(k)\Phi_k(\theta)/k\) converges
  normally and may be differentiated termwise.
\end{proposition}

\begin{proposition}[Holomorphic variation of the Abel finite part]
  \label{thm:logM-holomorphic-variation}
  The function
  \(\theta\mapsto F(A_\theta,\lambda(\theta))\) is holomorphic near
  \(\theta_0\) and
  \[
    \frac{\dd}{\dd\theta}F(A_\theta,\lambda(\theta))
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right)
    +\sum_{k\ge 2}\frac{\mu(k)}{k}
      \partial_\theta\Phi_k(\theta).
  \]
\end{proposition}

\begin{proposition}[Determinant truncation for the Abel finite part]
  \label{prop:logM-truncation}
  For
  \(
    F_N(A_\theta,\lambda(\theta))
    :=\log C(A_\theta,\lambda(\theta))
      +\sum_{2\le k\le N}\mu(k)\Phi_k(\theta)/k
  \),
  there exist \(C_U>0\) and \(q_U\in(0,1)\) such that
  \[
    \sup_{\theta\in U}
    \abs{F(A_\theta,\lambda(\theta))-F_N(A_\theta,\lambda(\theta))}
    \le C_U q_U^{N+1}.
  \]
  The same estimate holds after one \(\theta\)-derivative.
\end{proposition}

\subsection{Twisted determinants, periodic moments, and cumulants}

Let \((X_A^+,\sigma_A)\) be the mixing subshift determined by a primitive
matrix \(A\), let \(\mu_A\) be its Parry measure, and let
\(f:X_A^+\to\RR\) be locally constant. After passing to a suitable
block presentation, there is a primitive adjacency matrix \(B\) and an
edge function \(g\) such that \(S_nf(x)=\sum_{k=0}^{n-1}g(x_k,x_{k+1})\).
For \(t\in\CC\), define the weighted matrix
\(M_t(i,j):=B_{ij}e^{tg(i,j)}\).

\begin{proposition}[Twisted determinants encode periodic-orbit moments]
  \label{prop:twisted-determinant-orbit-moments}
  Assumptions \ref{SA:primitive}--\ref{SA:simple-dominant} apply to
  \(M_0=B\). For every \(n\ge 1\),
  \(\Tr(M_t^n)=\sum_{x\in\Fix(\sigma_A^n)}e^{tS_nf(x)}\). The twisted
  determinant determines every periodic-orbit moment sum via
  differentiation at \(t=0\).
\end{proposition}

\begin{proposition}[Twisted determinant rigidity for locally constant observables]
  \label{thm:twisted-determinant-rigidity}
  Assumptions \ref{SA:primitive}--\ref{SA:simple-dominant} apply.
  If two locally constant observables \(f,g\) on mixing SFTs satisfy
  \(\det(I-zM_t^{(f)})=\det(I-zM_t^{(g)})\) near \((0,0)\), then
  all periodic-orbit moment sums agree and the dominant eigenvalue
  branches coincide near \(t=0\).
\end{proposition}

\begin{corollary}[Cumulant rigidity]
  \label{cor:twisted-determinant-cumulants}
  Under the hypotheses of \Cref{thm:twisted-determinant-rigidity}, the
  centered Birkhoff sums for \(f\) and \(g\) have the same asymptotic
  cumulants of every order.
\end{corollary}

\subsection{Spectral-radius gap and limit laws}

Write \(\bar f:=f-\int f\,\dd\mu_A\) and
\(\widetilde M_u(i,j):=B_{ij}e^{iu\bar g(i,j)}\) for \(u\in\RR\).

\begin{definition}[Arithmetic lattice obstruction]
  \label{def:arithmetic-lattice}
  The set
  \(\Lambda_f:=\{u\in\RR:\rho(\widetilde M_u)=\lambda\}\)
  consists of the frequencies at which the spectral radius of the
  twisted matrix family equals the Perron root. By Parry's cocycle
  criterion \cite[Theorem~1]{Parry1992Cocycle}, \(u\in\Lambda_f\) if
  and only if \(e^{iu\bar g}\) is a multiplicative coboundary.
\end{definition}

\begin{proposition}[Off-origin decay from a spectral-radius gap]
  \label{prop:off-origin-decay}
  Assumptions \ref{SA:primitive}--\ref{SA:compact-freq} apply.
  Let \(K\Subset\RR\setminus\Lambda_f\) be compact. Then
  \(\kappa_K:=\sup_{u\in K}\rho(\widetilde M_u)/\lambda<1\), and for
  every \(\vartheta_K\in((1+\kappa_K)/2,1)\) there exists \(C_K>0\)
  such that
  \[
    \sup_{u\in K}\abs{\phi_n(u)}\le C_K\vartheta_K^n
    \qquad (n\ge 1),
  \]
  where \(\phi_n(u):=\lambda^{-n}\ell_0^{\mathsf T}\widetilde M_u^n r_0\)
  is the characteristic function of the centered Birkhoff sum.
\end{proposition}

\begin{proposition}[Spectral-gap central limit theorem]
  \label{thm:clt-spectral}
  Assumptions \ref{SA:primitive}--\ref{SA:non-coboundary} apply.
  There exists \(\sigma_f^2>0\) such that
  \[
    \frac{\sum_{k=0}^{n-1}\bar f\circ\sigma_A^k}{\sqrt n}
    \Longrightarrow \mathcal N(0,\sigma_f^2).
  \]
\end{proposition}

\subsection{Explicit computational examples}

\begin{remark}[Computational example: the golden-mean family]
  \label{rem:example-golden-mean}
  Let
  \[
    A_s=
    \begin{pmatrix}
      1+s & 1 \\
      1   & 0
    \end{pmatrix},
    \qquad |s|\ll 1.
  \]
  For real \(s\) near \(0\), this is a primitive matrix. The analytic
  simple eigenvalue branch from \Cref{def:spectral-gap-hypothesis} is
  \[
    \lambda(s)=\frac{1+s+\sqrt{(1+s)^2+4}}{2},
  \]
  and the second eigenvalue is \(\mu_2(s)=-\lambda(s)^{-1}\). Hence,
  by \Cref{def:algebraic-finite-part},
  \[
    C(A_s,\lambda(s))
    =\left(1+\lambda(s)^{-2}\right)^{-1},
    \qquad
    \partial_s\log C(A_s,\lambda(s))
    =\frac{2\lambda'(s)}{\lambda(s)(\lambda(s)^2+1)}.
  \]
  At \(s=0\), with \(\varphi=(1+\sqrt5)/2\),
  \[
    C(A_0,\varphi)=\frac{\varphi}{\sqrt5},
    \qquad
    \left.\partial_s\log C(A_s,\lambda(s))\right|_{s=0}
    =\frac{2}{\sqrt5(\varphi^2+1)}
    =0.2472135\ldots.
  \]
  The Abel finite part is
  \[
    \log \MM(A_0)
    =F(A_0,\varphi)
    =\log\frac{\varphi}{\sqrt5}
      +\sum_{k\ge 2}\frac{\mu(k)}{k}
        \log\Bigl((1-\varphi^{1-k})(1+\varphi^{-(k+1)})\Bigr)^{-1},
  \]
  and the determinant truncation from \Cref{prop:logM-truncation} is
  numerically rapid: truncation at \(k=14\) already gives
  \[
    \log\MM(A_0)=-0.82923\ldots.
  \]
  This example validates the derivative formula and the exponential
  truncation scheme in a nontrivial holomorphic family.
\end{remark}

\begin{remark}[Computational example: the full two-shift]
  \label{rem:example-full-two-shift}
  Let
  \[
    A=
    \begin{pmatrix}
      1 & 1 \\
      1 & 1
    \end{pmatrix}
  \]
  and let \(f(x)=\mathbf 1_{\{x_0=1\}}\). Then
  \[
    M_t=
    \begin{pmatrix}
      1 & e^t \\
      1 & e^t
    \end{pmatrix},
    \qquad
    \det(I-zM_t)=1-(1+e^t)z.
  \]
  The analytic leading eigenvalue branch is \(\lambda(t)=1+e^t\),
  and \(C(M_t,\lambda(t))=1\). After centering by \(1/2\), the
  dominant eigenvalue becomes
  \(\lambda_c(t)=e^{-t/2}(1+e^t)=2\cosh(t/2)\). Hence
  \(\sigma_f^2=1/4\), and on every compact set
  \(K\Subset\RR\setminus 2\pi\ZZ\),
  \[
    \sup_{u\in K}|\phi_n(u)|
    \le
    \left(\sup_{u\in K}\abs{\cos(u/2)}\right)^n.
  \]
  This example is arithmetic rather than aperiodic, so the lattice
  obstruction is visible at \(u\in 2\pi\ZZ\). Away from that lattice,
  the determinant package recovers the variance and the exponential
  decay exactly.
\end{remark}
