\section{Perturbative Applications}
\label{sec:perturbation}

\subsection{Finite-part constants and Abel variation}

\begin{definition}[Reduced determinant]\label{def:reduced-determinant}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of matrices near
  \(\theta_0\), let \(\lambda(\theta)=\rho(A_\theta)\) be the simple
  Perron eigenvalue, and set
  \[
    B_\theta:=I-\frac{A_\theta}{\lambda(\theta)}.
  \]
  Write the non-Perron Jordan spectrum of \(B_\theta\) as
  \(\{0\}\cup\{\nu_j(\theta):j=2,\dots,d\}\). The reduced determinant is
  \[
    \det\nolimits'(B_\theta)
    :=\prod_{\nu\in\spec(B_\theta)\setminus\{0\}}\nu
  \]
  with algebraic multiplicity, or equivalently
  \[
    \det\nolimits'(B_\theta)
    =\prod_{\mu\in\spec(A_\theta)\setminus\{\lambda(\theta)\}}
      \left(1-\frac{\mu}{\lambda(\theta)}\right).
  \]
  By \eqref{eq:reduced-det-constant},
  \[
    C(A_\theta)^{-1}
    =\prod_{\mu\in\spec(A_\theta)\setminus\{\lambda(\theta)\}}
      \left(1-\frac{\mu}{\lambda(\theta)}\right)
    =\det\nolimits'(B_\theta),
  \]
  and hence \(C(A_\theta)=\det\nolimits'(B_\theta)^{-1}\).
\end{definition}

\begin{proposition}[Reduced-determinant gradient formula]
  \label{thm:finite-part-reduced-determinant-group-inverse-gradient}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Choose left and
  right Perron vectors \(\ell_\theta,r_\theta\) normalised by
  \(\ell_\theta^{\mathsf T}r_\theta=1\), let
  \[
    P_\theta:=r_\theta\ell_\theta^{\mathsf T},\quad
    R_\theta:=B_\theta^{\#},
  \]
  where \(B_\theta=I-A_\theta/\lambda(\theta)\) and \(B_\theta^{\#}\) is
  the group inverse.
  Then
  \[
    \partial_\theta\log C(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right),
  \]
  with
  \[
    \partial_\theta\lambda(\theta)
    =\ell_\theta^{\mathsf T}(\partial_\theta A_\theta)r_\theta.
  \]
\end{proposition}

\begin{proof}
  Since \(\lambda(\theta)\) is a simple eigenvalue
  \cite[Chapter~II, \S1]{Kato1995Perturbation}, \(B_\theta\) has a simple
  zero eigenvalue and all other eigenvalues are non-zero; hence the group
  inverse \(R_\theta\) exists; see \cite[Chapter~4]{CampbellMeyer1979}.
  Because the spectrum depends analytically,
  \(P_\theta\) and \(R_\theta=B_\theta^{\#}\) vary analytically on
  \(\theta\).
  On \((I-P_\theta)\CC^d\), the restriction of \(B_\theta\) is
  invertible and its ordinary determinant is \(\det\nolimits'(B_\theta)\).
  Jacobi's formula on this complementary spectral subspace gives
  \[
    \partial_\theta\log\det\nolimits'(B_\theta)
    =\Tr(R_\theta\,\partial_\theta B_\theta).
  \]
  By \Cref{def:reduced-determinant},
  \[
    \partial_\theta\log C(A_\theta)
    =-\partial_\theta\log\det\nolimits'(B_\theta)
    =-\Tr(R_\theta\,\partial_\theta B_\theta),
  \]
  while
  \[
    \partial_\theta B_\theta
    =-\frac{\partial_\theta A_\theta}{\lambda(\theta)}
      +\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
       A_\theta.
  \]
  Substituting yields the stated formula.
\end{proof}

\begin{proposition}[Normal convergence of the secondary Fredholm factors]
  \label{prop:secondary-fredholm-normal-convergence}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Choose left and
  right Perron vectors \(\ell_\theta,r_\theta\) normalized by
  \(\ell_\theta^{\mathsf T}r_\theta=1\), put
  \(P_\theta:=r_\theta\ell_\theta^{\mathsf T}\), and for \(k\ge 2\)
  define
  \[
    G_k(\theta):=I-\lambda(\theta)^{-k}A_\theta,
    \qquad
    \Phi_k(\theta):=\log\det G_k(\theta)^{-1}.
  \]
  Then, after shrinking to a neighbourhood \(U\) of \(\theta_0\), there
  exist constants \(C_U>0\) and \(q_U\in(0,1)\) such that
  \[
    \sup_{\theta\in U}
    \rho\!\left(
      \lambda(\theta)^{-1}A_\theta\bigm|_{(I-P_\theta)\CC^d}\right)
    \le q_U
  \]
  and
  \[
    \sup_{\theta\in U}\abs{\Phi_k(\theta)}
    +\sup_{\theta\in U}\abs{\partial_\theta\Phi_k(\theta)}
    \le C_U k q_U^k
    \qquad (k\ge 2).
  \]
  Consequently
  \[
    \sum_{k\ge 2}\frac{\mu(k)}{k}\Phi_k(\theta)
  \]
  converges normally on \(U\), and termwise differentiation is
  justified there.
\end{proposition}

\begin{proof}
  Choose a closed disc \(U\) around \(\theta_0\) so small that
  \(\lambda(\theta)\) stays simple on \(U\). Set
  \[
    \lambda_\ast:=\inf_{\theta\in U}\abs{\lambda(\theta)}>1,
    \qquad
    M_0:=\sup_{\theta\in U}\norm{A_\theta}<\infty,
  \]
  and
  \[
    M_1:=\sup_{\theta\in U}
    \bigl(\norm{\partial_\theta A_\theta}
      +\abs{\partial_\theta\lambda(\theta)}\bigr)<\infty.
  \]
  Because the Perron projector \(P_\theta\) depends holomorphically on
  \(\theta\), the complementary family
  \[
    \theta\longmapsto
    \lambda(\theta)^{-1}A_\theta\bigm|_{(I-P_\theta)\CC^d}
  \]
  is continuous. After shrinking \(U\) again if necessary, simplicity
  of the Perron root gives
  \[
    q_0:=
    \sup_{\theta\in U}
    \rho\!\left(
      \lambda(\theta)^{-1}A_\theta\bigm|_{(I-P_\theta)\CC^d}\right)<1.
  \]
  Fix \(q_U\in(\max\{\lambda_\ast^{-1},q_0\},1)\).

  For each \(k\ge 2\), the operator \(G_k(\theta)\) acts by the scalar
  \(1-\lambda(\theta)^{1-k}\) on the Perron line and by
  \(I-\lambda(\theta)^{-k}A_\theta\) on \((I-P_\theta)\CC^d\). In
  particular every \(G_k(\theta)\) is invertible on \(U\). Since
  \(G_k(\theta)\to I\) uniformly on \(U\) as \(k\to\infty\), there
  exists \(K\ge 2\) such that
  \[
    \sup_{\theta\in U}\norm{I-G_k(\theta)}\le \frac12
    \qquad (k\ge K),
  \]
  and hence
  \[
    \sup_{\theta\in U}\norm{G_k(\theta)^{-1}}\le 2
    \qquad (k\ge K).
  \]
  For the finitely many values \(2\le k<K\), continuity on the compact
  set \(U\) yields
  \[
    C_U^{(0)}:=\max_{2\le k<K}\sup_{\theta\in U}\norm{G_k(\theta)^{-1}}
    <\infty.
  \]
  Therefore
  \[
    \sup_{k\ge 2,\ \theta\in U}\norm{G_k(\theta)^{-1}}
    \le C_U^{(1)}
  \]
  for some constant \(C_U^{(1)}>0\).

  For \(k\ge K\), the absolutely convergent determinant series gives
  \[
    \Phi_k(\theta)
    =\sum_{m\ge 1}
      \frac{\Tr\!\bigl((\lambda(\theta)^{-k}A_\theta)^m\bigr)}{m},
  \]
  so
  \[
    \abs{\Phi_k(\theta)}
    \le d\sum_{m\ge 1}\frac{(M_0\lambda_\ast^{-k})^m}{m}
    \le C_U^{(2)}\lambda_\ast^{-k}.
  \]
  Enlarging the constant to absorb the finitely many indices
  \(2\le k<K\), we obtain
  \[
    \sup_{\theta\in U}\abs{\Phi_k(\theta)}
    \le C_U^{(3)}q_U^k
    \qquad (k\ge 2).
  \]

  Jacobi's formula yields
  \[
    \partial_\theta\Phi_k(\theta)
    =-\Tr\!\bigl(G_k(\theta)^{-1}\partial_\theta G_k(\theta)\bigr)
    =\Tr\!\left(
      G_k(\theta)^{-1}\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)^k}
        -\frac{k\,\partial_\theta\lambda(\theta)}{\lambda(\theta)^{k+1}}
         A_\theta\right)\right).
  \]
  Hence
  \[
    \sup_{\theta\in U}\norm{\partial_\theta G_k(\theta)}
    \le M_1\lambda_\ast^{-k}
      +kM_0M_1\lambda_\ast^{-k-1}
    \le C_U^{(4)}k\lambda_\ast^{-k},
  \]
  and the uniform inverse bound gives
  \[
    \sup_{\theta\in U}\abs{\partial_\theta\Phi_k(\theta)}
    \le C_U^{(5)}k\lambda_\ast^{-k}
    \le C_U^{(5)}kq_U^k.
  \]
  Combining the two estimates and using \(|\mu(k)|\le 1\) proves the
  normal convergence of
  \(\sum_{k\ge 2}\mu(k)\Phi_k(\theta)/k\) and of its derivative series.
\end{proof}

\begin{proposition}[Holomorphic variation of the Abel finite part]
  \label{thm:logM-holomorphic-variation}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Let
  \(R_\theta=(I-A_\theta/\lambda(\theta))^{\#}\). Then
  \(\theta\mapsto\log\MM(A_\theta)\) is holomorphic near \(\theta_0\)
  and
  \[
    \frac{\dd}{\dd\theta}\log\MM(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right)
    +\sum_{k\ge 2}\frac{\mu(k)}{k}
      \frac{\dd}{\dd\theta}
      \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
\end{proposition}

\begin{proof}
  For each \(\theta\) near \(\theta_0\),
  \Cref{thm:finite-part-primitive-closed-form} gives
  \[
    \log\MM(A_\theta)
    =\log C(A_\theta)
      +\sum_{k\ge 2}\frac{\mu(k)}{k}
        \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  By \Cref{prop:secondary-fredholm-normal-convergence}, the correction
  series converges normally and may be differentiated termwise. The
  derivative of \(\log C(A_\theta)\) is given by
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient},
  which proves the stated formula.
\end{proof}

\begin{proposition}[Determinant truncation for the Abel finite part]
  \label{prop:logM-truncation}
  Under the hypotheses of \Cref{thm:logM-holomorphic-variation}, there
  exist a neighbourhood \(U\) of \(\theta_0\), a constant \(C_U>0\), and
  a number \(q_U\in(0,1)\) such that for
  \[
    \log\MM_N(A_\theta)
    :=\log C(A_\theta)
      +\sum_{2\le k\le N}\frac{\mu(k)}{k}
       \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1},
  \]
  one has
  \[
    \sup_{\theta\in U}
    \abs{\log\MM(A_\theta)-\log\MM_N(A_\theta)}
    \le C_U q_U^{N+1}.
  \]
  The same estimate holds after one \(\theta\)-derivative.
\end{proposition}

\begin{proof}
  Let \(U\), \(C_0\), and \(q_U\) be given by
  \Cref{prop:secondary-fredholm-normal-convergence}, and write
  \[
    \Phi_k(\theta):=\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  Then
  \[
    \sum_{k>N}\left|\frac{\mu(k)}{k}\Phi_k(\theta)\right|
    \le C_0\sum_{k>N}q_U^k,
  \]
  and the same estimate holds with \(\Phi_k\) replaced by
  \(\partial_\theta\Phi_k\). Summing the geometric tail gives the
  result.
\end{proof}

\subsection{Twisted determinants, periodic moments, and cumulants}

Let \((X_A^+,\sigma_A)\) be the mixing subshift determined by a primitive
matrix \(A\), let \(\mu_A\) be its Parry measure, and let
\(f:X_A^+\to\RR\) be locally constant. If \(f\) depends on the first
\(m\) coordinates, pass to the \((m-1)\)-block presentation. Then the
recoded system is again a mixing finite-state Markov shift with
primitive adjacency matrix \(B\), and there is a real-valued edge
function \(g\) on the allowed transitions of \(B\) such that
\[
  S_nf(x)=\sum_{k=0}^{n-1}g(x_k,x_{k+1})
\]
on the recoded shift. For \(t\in\CC\), define the weighted matrix
\[
  M_t(i,j):=B_{ij}e^{t g(i,j)}.
\]

\begin{proposition}[Twisted determinants encode periodic-orbit moments]
  \label{prop:twisted-determinant-orbit-moments}
  For every \(n\ge 1\),
  \[
    \Tr(M_t^n)=\sum_{x\in\Fix(\sigma_A^n)}e^{tS_nf(x)}.
  \]
  Hence, for \((z,t)\) near \((0,0)\),
  \[
    \det(I-zM_t)^{-1}
    =\exp\!\left(
      \sum_{n\ge 1}\frac{z^n}{n}
      \sum_{x\in\Fix(\sigma_A^n)}e^{tS_nf(x)}\right)
    =\prod_{\mathcal O}
      \bigl(1-e^{tS_{\mathcal O}f}z^{|\mathcal O|}\bigr)^{-1},
  \]
  where \(\mathcal O\) runs over primitive periodic orbits of the
  recoded shift. For every integer \(r\ge 0\),
  \[
    \left.\partial_t^r\Tr(M_t^n)\right|_{t=0}
    =\sum_{x\in\Fix(\sigma_A^n)}(S_nf(x))^r,
  \]
  and consequently
  \[
    \left.\partial_t^r
      \log\det(I-zM_t)^{-1}\right|_{t=0}
    =\sum_{n\ge 1}\frac{z^n}{n}
      \sum_{x\in\Fix(\sigma_A^n)}(S_nf(x))^r.
  \]
  In particular, the twisted determinant determines every periodic-orbit
  moment sum.
\end{proposition}

\begin{proof}
  For each \(n\ge 1\),
  \[
    \Tr(M_t^n)=\sum_{x\in\Fix(\sigma_A^n)}e^{tS_nf(x)}.
  \]
  Applying \Cref{thm:fredholm-witt-bridge} to the finite matrix \(M_t\)
  gives the first identity. The primitive-orbit product is the
  corresponding Euler-product expansion by M\"obius inversion, exactly as
  in \Cref{prop:prelim-trace-primitive}. Differentiating the logarithm
  with respect to \(t\) at \(t=0\) yields the moment identities.
\end{proof}

\begin{proposition}[Twisted determinant rigidity for locally constant observables]
  \label{thm:twisted-determinant-rigidity}
  Let \(f\) and \(g\) be locally constant observables on mixing
  subshifts of finite type \((X_A^+,\sigma_A)\) and \((X_B^+,\sigma_B)\),
  and let \(M_t^{(f)}\) and \(M_t^{(g)}\) be one-step weighted matrix
  models as above. Assume that
  \[
    \det(I-zM_t^{(f)})=\det(I-zM_t^{(g)})
  \]
  for \((z,t)\) in a neighbourhood of \((0,0)\). Then:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item For every \(n\ge 1\) and every \(r\ge 0\),
      \[
        \sum_{x\in\Fix(\sigma_A^n)}(S_nf(x))^r
        =
        \sum_{y\in\Fix(\sigma_B^n)}(S_ng(y))^r.
      \]
    \item The dominant eigenvalue germs \(\lambda_f(t)\) and
      \(\lambda_g(t)\) agree near \(t=0\).
  \end{enumerate}
\end{proposition}

\begin{proof}
  Equality of determinants near \((0,0)\) implies equality of the
  reciprocal determinants and therefore, by
  \Cref{prop:twisted-determinant-orbit-moments}, equality of the trace
  functions
  \[
    \Tr\!\bigl((M_t^{(f)})^n\bigr)
    =\Tr\!\bigl((M_t^{(g)})^n\bigr)
    \qquad (n\ge 1)
  \]
  for \(t\) near \(0\). Differentiating \(r\) times at \(t=0\) proves
  \textup{(i)}.

  For each fixed \(t\) near \(0\), \Cref{thm:fredholm-spectral-rigidity}
  applied to \(M_t^{(f)}\) and \(M_t^{(g)}\) shows that their non-zero
  spectra agree with multiplicity. In particular the simple Perron
  eigenvalue branch agrees, proving \textup{(ii)}.
\end{proof}

\begin{corollary}[Cumulant rigidity]
  \label{cor:twisted-determinant-cumulants}
  Under the hypotheses of \Cref{thm:twisted-determinant-rigidity}, the
  centered Birkhoff sums for \(f\) and \(g\) have the same asymptotic
  cumulants of every order; in particular, their asymptotic variances
  coincide.
\end{corollary}

\begin{proof}
  By \Cref{thm:twisted-determinant-rigidity}\textup{(ii)}, the dominant
  eigenvalue germs agree near \(t=0\). Since the centered cumulant
  generating function is
  \[
    t\longmapsto
    \log\frac{\lambda_f(t)}{\lambda_f(0)}
    -t\,\frac{\lambda_f'(0)}{\lambda_f(0)},
  \]
  the asymptotic cumulants are the same for \(f\) and \(g\).
\end{proof}

\subsection{Spectral-radius gap and limit laws}

Write
\[
  \bar f:=f-\int f\,\dd\mu_A,
  \qquad
  \bar g(i,j):=g(i,j)-\int f\,\dd\mu_A,
  \qquad
  \widetilde M_u(i,j):=B_{ij}e^{iu\bar g(i,j)}
  \qquad (u\in\RR).
\]
Choose positive left and right Perron vectors \(\ell_0,r_0\) for \(B\),
normalized by
\[
  \ell_0^{\mathsf T}B=\lambda\ell_0^{\mathsf T},
  \qquad
  Br_0=\lambda r_0,
  \qquad
  \ell_0^{\mathsf T}r_0=1,
\]
where \(\lambda=\rho(B)\). Then, in the block presentation,
\[
  \phi_n(u):=
  \int e^{iu\sum_{k=0}^{n-1}\bar f\circ\sigma_A^k}\,\dd\mu_A
  =\lambda^{-n}\ell_0^{\mathsf T}\widetilde M_u^n r_0.
\]

\begin{definition}[Arithmetic obstruction set]
  \label{def:arithmetic-lattice}
  The arithmetic obstruction set of \(f\) is
  \[
    \Lambda_f:=
    \Bigl\{u\in\RR:\ \exists\,\omega\in\mathbb S^1,
      \eta:\mathrm{state}(B)\to\mathbb S^1\text{ such that }
      e^{iu\bar g(i,j)}=\omega\,\eta(i)\eta(j)^{-1}
      \text{ for every allowed edge }(i,j)\Bigr\}.
  \]
  By the finite-state cocycle criterion of Parry
  \cite[Theorem~1]{Parry1992Cocycle}, this is exactly the set of
  frequencies \(u\) for which \(\rho(\widetilde M_u)=\lambda\). We call
  \(f\) \emph{aperiodic}, or \emph{non-lattice}, when
  \(\Lambda_f=\{0\}\).
\end{definition}

\begin{proposition}[Off-origin decay from a spectral-radius gap]
  \label{prop:off-origin-decay}
  Assume \(\bar f\) is not a coboundary modulo constants. Let
  \(K\Subset\RR\setminus\Lambda_f\), and choose \(\eta_K\in(0,1)\) such
  that
  \[
    \sup_{u\in K}\frac{\rho(\widetilde M_u)}{\lambda}<\eta_K<1.
  \]
  Then there exists \(C_K>0\) such that
  \[
    \sup_{u\in K}\abs{\phi_n(u)}
    \le C_K\eta_K^n
    \qquad (n\ge 1).
  \]
  In particular, if \(f\) is aperiodic then the same estimate holds on
  every compact set \(K\Subset\RR\setminus\{0\}\).
\end{proposition}

\begin{proof}
  Put \(r_K:=\eta_K\lambda\). By the choice of \(\eta_K\), one has
  \[
    \sup_{u\in K}\rho(\widetilde M_u)<r_K<\lambda.
  \]
  Hence the compact set \(K\times\{\abs{z}=r_K\}\) avoids the spectrum
  of \(zI-\widetilde M_u\), and therefore
  \[
    C_K^{(0)}:=
    \sup_{u\in K,\ \abs{z}=r_K}\|(zI-\widetilde M_u)^{-1}\|<\infty.
  \]
  By the holomorphic functional calculus,
  \[
    \widetilde M_u^n
    =\frac{1}{2\pi i}\int_{\abs{z}=r_K}
      z^n(zI-\widetilde M_u)^{-1}\,\dd z,
  \]
  so
  \[
    \sup_{u\in K}\|\widetilde M_u^n\|
    \le 2\pi C_K^{(0)}r_K^n.
  \]
  Multiplying by \(\lambda^{-n}\|\ell_0\|\,\|r_0\|\) gives
  \[
    \sup_{u\in K}\abs{\phi_n(u)}
    \le
    2\pi\|\ell_0\|\,\|r_0\|\,C_K^{(0)}
    \left(\frac{r_K}{\lambda}\right)^n,
  \]
  which is the stated estimate.
\end{proof}

\begin{corollary}[Spectral-gap central limit theorem]
  \label{thm:clt-spectral}
  Assume \(f\) is not a coboundary modulo constants. Then there exists
  \(\sigma_f^2>0\) such that
  \[
    \frac{\sum_{k=0}^{n-1}\bar f\circ\sigma_A^k}{\sqrt n}
    \Longrightarrow \mathcal N(0,\sigma_f^2).
  \]
\end{corollary}

\begin{proof}
  The map \(u\mapsto \widetilde M_u\) is entire. Since \(B=\widetilde
  M_0\) is primitive, analytic perturbation theory gives
  \(\delta>0\), an analytic simple eigenvalue \(u\mapsto\lambda_c(u)\),
  an analytic rank-one projector \(u\mapsto P_u\), and constants
  \(C>0\), \(\vartheta\in(0,1)\) such that
  \[
    \widetilde M_u^n=\lambda_c(u)^nP_u+N_u^n,
    \qquad
    \|N_u^n\|\le C(\vartheta\lambda)^n
  \]
  for \(|u|<\delta\) and \(n\ge 1\); see
  \cite[Chapter~II, \S2]{Kato1995Perturbation}. Because \(\bar f\) is
  centered,
  \[
    \lambda_c'(0)=0.
  \]
  The standard finite-state spectral method gives
  \[
    \log\frac{\lambda_c(u)}{\lambda}
    =-\frac{\sigma_f^2}{2}u^2+O(u^3)
    \qquad (u\to 0),
  \]
  where
  \[
    \sigma_f^2
    =-\left.\frac{\dd^2}{\dd u^2}\log\frac{\lambda_c(u)}{\lambda}
      \right|_{u=0}>0
  \]
  whenever \(f\) is not a coboundary modulo constants; see
  \cite[Chapter~5]{Ruelle1978Thermodynamic},
  \cite[Chapter~3]{Baladi2000PositiveTransfer},
  and \cite[Chapter~IV]{HennionHerve2001}. Because
  \(P_0=r_0\ell_0^{\mathsf T}\), one has
  \[
    \ell_0^{\mathsf T}P_{u_0/\sqrt n}r_0=1+O(n^{-1/2}).
  \]
  Hence, for each fixed \(u_0\in\RR\),
  \[
    \phi_n(u_0/\sqrt n)
    =\left(\frac{\lambda_c(u_0/\sqrt n)}{\lambda}\right)^n
      \ell_0^{\mathsf T}P_{u_0/\sqrt n}r_0
      +O(\vartheta^n)
    \longrightarrow e^{-\sigma_f^2u_0^2/2}.
  \]
  L\'evy's continuity theorem yields the central limit theorem.
\end{proof}

\begin{remark}
  The determinant package clarifies why
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient,prop:secondary-fredholm-normal-convergence,thm:logM-holomorphic-variation,prop:logM-truncation,thm:twisted-determinant-rigidity,cor:twisted-determinant-cumulants,prop:off-origin-decay,thm:clt-spectral}
  belong together. In each case one starts from a simple dominant pole
  of a Fredholm determinant, isolates the reduced resolvent on the
  complementary spectral subspace, and then reads a perturbative
  quantity from that reduced determinant or from the motion of the pole.
  The difference between finite-part constants, orbit moments, and limit
  laws is therefore a difference of observable, not of mechanism.
\end{remark}

\subsection{Explicit computational examples}

\begin{remark}[Computational example: the golden-mean family]
  \label{rem:example-golden-mean}
  Let
  \[
    A_s=
    \begin{pmatrix}
      1+s & 1 \\
      1   & 0
    \end{pmatrix},
    \qquad |s|\ll 1.
  \]
  Its Perron root is
  \[
    \lambda(s)=\frac{1+s+\sqrt{(1+s)^2+4}}{2},
  \]
  and the second eigenvalue is \(\mu_2(s)=-\lambda(s)^{-1}\). Hence
  \[
    C(A_s)
    =\left(1+\lambda(s)^{-2}\right)^{-1},
    \qquad
    \partial_s\log C(A_s)
    =\frac{2\lambda'(s)}{\lambda(s)(\lambda(s)^2+1)}.
  \]
  At \(s=0\), with \(\varphi=(1+\sqrt5)/2\),
  \[
    C(A_0)=\frac{\varphi}{\sqrt5},
    \qquad
    \left.\partial_s\log C(A_s)\right|_{s=0}
    =\frac{2}{\sqrt5(\varphi^2+1)}
    =0.2472135\ldots.
  \]
  The Abel finite part is
  \[
    \log \MM(A_0)
    =\log\frac{\varphi}{\sqrt5}
      +\sum_{k\ge 2}\frac{\mu(k)}{k}
        \log\Bigl((1-\varphi^{1-k})(1+\varphi^{-(k+1)})\Bigr)^{-1},
  \]
  and the determinant truncation from \Cref{prop:logM-truncation} is
  numerically rapid: truncation at \(k=14\) already gives
  \[
    \log\MM(A_0)=-0.82923\ldots.
  \]
  This example validates the derivative formula and the exponential
  truncation scheme in a nontrivial holomorphic family, but it does not
  test the twisted non-lattice estimates because no observable is
  involved.
\end{remark}

\begin{remark}[Computational example: the full two-shift]
  \label{rem:example-full-two-shift}
  Let
  \[
    A=
    \begin{pmatrix}
      1 & 1 \\
      1 & 1
    \end{pmatrix}
  \]
  and let \(f(x)=\mathbf 1_{\{x_0=1\}}\). Then
  \[
    M_t=
    \begin{pmatrix}
      1 & e^t \\
      1 & e^t
    \end{pmatrix},
    \qquad
    \det(I-zM_t)=1-(1+e^t)z.
  \]
  Thus \(\lambda(t)=1+e^t\) and \(C(M_t)=1\). After centering by
  \(1/2\), the dominant eigenvalue becomes
  \[
    \lambda_c(t)=e^{-t/2}(1+e^t)=2\cosh(t/2).
  \]
  Therefore, for real \(u\),
  \[
    \int e^{iu\sum_{k=0}^{n-1}(f-\frac12)\circ\sigma_A^k}\,\dd\mu_A
    =\left(\frac{\lambda_c(iu)}{2}\right)^n
    =\cos(u/2)^n.
  \]
  Hence
  \[
    \sigma_f^2=\frac14,
  \]
  and on every compact set \(K\Subset\RR\setminus 2\pi\ZZ\),
  \[
    \sup_{u\in K}
    \left|\int e^{iu\sum_{k=0}^{n-1}(f-\frac12)\circ\sigma_A^k}\,\dd\mu_A\right|
    \le
    \left(\sup_{u\in K}\abs{\cos(u/2)}\right)^n.
  \]
  This example is arithmetic rather than aperiodic, so the lattice
  obstruction is visible at \(u\in 2\pi\ZZ\). Away from that lattice,
  the determinant package recovers the variance and the exponential
  decay exactly, not merely asymptotically. It therefore shows both what
  the off-origin estimate captures and where it stops: the decay is
  uniform only after excluding the lattice obstruction.
\end{remark}

\subsection{Comparison with classical finite-dimensional perturbation theory}

Classical finite-dimensional perturbation theory starts from a fixed
family of matrices \(A_\theta\) and studies the motion of its isolated
eigenvalues, eigenvectors, and reduced resolvent. That machinery is
already enough to prove
\Cref{thm:finite-part-reduced-determinant-group-inverse-gradient,thm:clt-spectral}
for a \emph{single} symbolic model. What it does not single out is the
determinant as a common, model-independent object.

The determinant packaging adds exactly three features needed here.
First, \Cref{prop:twisted-determinant-orbit-moments} is period-graded:
one determinant organizes all periodic lengths simultaneously via the
Euler product and M\"obius inversion. Second,
\Cref{thm:twisted-determinant-rigidity,cor:twisted-determinant-cumulants}
compares different shifts or different observables without putting them
on a common state space; the invariant is the determinant germ, not the
ambient matrix size. Third,
\Cref{prop:secondary-fredholm-normal-convergence,prop:logM-truncation}
split the Abel finite part into a reduced determinant plus a rapidly
convergent hierarchy of secondary factors. That is the only comparison
with ordinary finite-dimensional perturbation theory needed here.
