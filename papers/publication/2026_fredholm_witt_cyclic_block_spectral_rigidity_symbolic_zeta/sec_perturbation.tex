\section{Perturbative Applications}
\label{sec:perturbation}

\subsection{Finite-part constants and determinant-rigidity}

\begin{definition}[Reduced determinant]\label{def:reduced-determinant}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of matrices near
  \(\theta_0\), let \(\lambda(\theta)=\rho(A_\theta)\) be the simple
  Perron eigenvalue, and set
  \[
    B_\theta:=I-\frac{A_\theta}{\lambda(\theta)}.
  \]
  Write the non-Perron Jordan spectrum of \(B_\theta\) as
  \(\{0\}\cup\{\nu_j(\theta):j=2,\dots,d\}\). The reduced determinant is
  \[
    \det\nolimits'(B_\theta)
    :=\prod_{j=2}^{d}\nu_j(\theta)
    =\prod_{\mu\in\spec(A_\theta)\setminus\{\lambda(\theta)\}}
      \left(1-\frac{\mu}{\lambda(\theta)}\right).
  \]
  and \(C(A_\theta)=\det\nolimits'(B_\theta)^{-1}\).
\end{definition}

\begin{theorem}[Reduced-determinant gradient formula]
  \label{thm:finite-part-reduced-determinant-group-inverse-gradient}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Choose left and
  right Perron vectors \(\ell_\theta,r_\theta\) normalised by
  \(\ell_\theta^{\mathsf T}r_\theta=1\), let
  \[
    P_\theta:=r_\theta\ell_\theta^{\mathsf T},\quad
    R_\theta:=B_\theta^{\#},
  \]
  where \(B_\theta=I-A_\theta/\lambda(\theta)\) and \(B_\theta^{\#}\) is
  the group inverse.
  Then
  \[
    \partial_\theta\log C(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right),
  \]
  with
  \[
    \partial_\theta\lambda(\theta)
    =\ell_\theta^{\mathsf T}(\partial_\theta A_\theta)r_\theta.
  \]
\end{theorem}

\begin{proof}
  Since \(\lambda(\theta)\) is a simple eigenvalue
  \cite[Chapter~II, \S1]{Kato1995Perturbation}, \(B_\theta\) has a simple
  zero eigenvalue and all other eigenvalues are non-zero; hence the group
  inverse \(R_\theta\) exists; see \cite[Chapter~4]{CampbellMeyer1979}.
  Because the spectrum depends analytically,
  \(P_\theta\) and \(R_\theta=B_\theta^{\#}\) vary analytically on
  \(\theta\).
  On \((I-P_\theta)\CC^d\), \(\det\nolimits'(B_\theta)\) is the
  ordinary determinant, so by Jacobi's formula
  \[
    \partial_\theta\log\det\nolimits'(B_\theta)
    =\Tr(R_\theta\,\partial_\theta B_\theta).
  \]
  Since \(C(A_\theta)=\det\nolimits'(B_\theta)^{-1}\), this gives
  \[
    \partial_\theta\log C(A_\theta)
    =-\partial_\theta\log\det\nolimits'(B_\theta)
    =-\Tr(R_\theta\,\partial_\theta B_\theta),
  \]
  while
  \[
    \partial_\theta B_\theta
    =-\frac{\partial_\theta A_\theta}{\lambda(\theta)}
      +\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
       A_\theta.
  \]
  Substituting yields the stated formula.
\end{proof}

\begin{theorem}[Holomorphic variation of the Abel finite part]
  \label{thm:logM-holomorphic-variation}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Then
  \(\theta\mapsto\log\MM(A_\theta)\) is holomorphic near \(\theta_0\)
  and
  \[
    \frac{\dd}{\dd\theta}\log\MM(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right)
    +\sum_{k\ge 2}\frac{\mu(k)}{k}
      \frac{\dd}{\dd\theta}
      \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  The correction series converges absolutely and locally uniformly near
  \(\theta_0\).
\end{theorem}

\begin{proof}
  Apply \Cref{thm:finite-part-primitive-closed-form} to each \(A_\theta\).
  For \(k\ge 2\), set
  \[
    \Phi_k(\theta):=\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  Choose \(\epsilon>0\) so that
  \(|\lambda(\theta)|\ge\lambda_{\min}>1\) for
  \(|\theta-\theta_0|\le\epsilon\). Since \(\|A_\theta\|\) is bounded on
  this compact set, Jacobi gives
  \[
    \left|\frac{\mu(k)}{k}\partial_\theta\Phi_k(\theta)\right|
    \le
    \frac{C_1}{k\,\lambda_{\min}^{k}}
    +\frac{C_2}{\lambda_{\min}^{k-1}},
  \]
  and therefore
  \[
    \sum_{k\ge2}\frac{\mu(k)}{k}\partial_\theta\Phi_k
  \]
  converges absolutely and locally uniformly, so differentiation is
  termwise justified. The reduced term is given by
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient}.
  For \(\Phi_k\), Jacobi's formula gives
  \[
    \frac{\dd}{\dd\theta}\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}
    =\Tr\!\left((I-\lambda(\theta)^{-k}A_\theta)^{-1}
      \left(\frac{\partial_\theta A_\theta}{\lambda(\theta)^k}
      -\frac{k\,\partial_\theta\lambda(\theta)}{\lambda(\theta)^{k+1}}
      A_\theta\right)\right),
  \]
  a finite-dimensional Jacobi computation with no matrix-log branch.
\end{proof}

\begin{proposition}[Determinant truncation for the Abel finite part]
  \label{prop:logM-truncation}
  Under the hypotheses of \Cref{thm:logM-holomorphic-variation}, there
  exist a neighbourhood \(U\) of \(\theta_0\), a constant \(C_U>0\), and
  a number \(q_U\in(0,1)\) such that for
  \[
    \log\MM_N(A_\theta)
    :=\log C(A_\theta)
      +\sum_{2\le k\le N}\frac{\mu(k)}{k}
       \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1},
  \]
  one has
  \[
    \sup_{\theta\in U}
    \abs{\log\MM(A_\theta)-\log\MM_N(A_\theta)}
    \le C_U q_U^{N+1}.
  \]
  The same estimate holds after one \(\theta\)-derivative.
\end{proposition}

\begin{proof}
  In the proof of \Cref{thm:logM-holomorphic-variation} we obtained
  bounds
  \[
    |\Phi_k(\theta)|+|\partial_\theta\Phi_k(\theta)|
    \le C_0 q_U^k
  \]
  on a sufficiently small neighbourhood \(U\) of \(\theta_0\). Hence
  \[
    \sum_{k>N}\left|\frac{\mu(k)}{k}\Phi_k(\theta)\right|
    \le C_0\sum_{k>N}q_U^k,
  \]
  and the same estimate holds with \(\Phi_k\) replaced by
  \(\partial_\theta\Phi_k\). Summing the geometric tail gives the
  result.
\end{proof}

\subsection{Twisted determinants and orbit-moment rigidity}

Let \((X_A^+,\sigma_A)\) be the mixing subshift determined by a primitive
matrix \(A\), let \(\mu_A\) be its Parry measure, and let
\(f:X_A^+\to\RR\) be locally constant. After recoding if needed, we may
assume that \(f\) is one-step, so there is a primitive matrix \(B\) and
a function \(\varphi\) on the state set of \(B\) such that
\[
  S_nf(x)=\sum_{k=0}^{n-1}\varphi(x_k)
\]
on the recoded shift. For \(t\in\CC\), define the weighted matrix
\[
  M_t(b',b):=B_{b',b}e^{t\varphi(b)}.
\]

\begin{proposition}[Twisted determinants encode orbit moments]
  \label{prop:twisted-determinant-orbit-moments}
  For \((z,t)\) near \((0,0)\),
  \[
    \det(I-zM_t)^{-1}
    =\exp\!\left(
      \sum_{n\ge 1}\frac{z^n}{n}
      \sum_{x\in\Fix(\sigma_A^n)}e^{tS_nf(x)}\right)
    =\prod_{\mathcal O}
      \bigl(1-e^{tS_{\mathcal O}f}z^{|\mathcal O|}\bigr)^{-1},
  \]
  where \(\mathcal O\) runs over primitive periodic orbits of the
  recoded shift. Consequently, for every integer \(r\ge 0\),
  \[
    \left.\partial_t^r
      \log\det(I-zM_t)^{-1}\right|_{t=0}
    =\sum_{n\ge 1}\frac{z^n}{n}
      \sum_{x\in\Fix(\sigma_A^n)}(S_nf(x))^r.
  \]
\end{proposition}

\begin{proof}
  For each \(n\ge 1\),
  \[
    \Tr(M_t^n)=\sum_{x\in\Fix(\sigma_A^n)}e^{tS_nf(x)}.
  \]
  Applying \Cref{thm:fredholm-witt-bridge} to the finite matrix \(M_t\)
  gives the first identity. The primitive-orbit product is the
  corresponding Euler-product expansion by M\"obius inversion, exactly as
  in \Cref{prop:prelim-trace-primitive}. Differentiating the logarithm
  with respect to \(t\) at \(t=0\) yields the moment identities.
\end{proof}

\begin{theorem}[Twisted determinant rigidity for locally constant observables]
  \label{thm:twisted-determinant-rigidity}
  Let \(f\) and \(g\) be locally constant observables on mixing
  subshifts of finite type \((X_A^+,\sigma_A)\) and \((X_B^+,\sigma_B)\),
  and let \(M_t^{(f)}\) and \(M_t^{(g)}\) be one-step weighted matrix
  models as above. Assume that
  \[
    \det(I-zM_t^{(f)})=\det(I-zM_t^{(g)})
  \]
  for \((z,t)\) in a neighbourhood of \((0,0)\). Then:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item For every \(n\ge 1\) and every \(r\ge 0\),
      \[
        \sum_{x\in\Fix(\sigma_A^n)}(S_nf(x))^r
        =
        \sum_{y\in\Fix(\sigma_B^n)}(S_ng(y))^r.
      \]
    \item The dominant eigenvalue germs \(\lambda_f(t)\) and
      \(\lambda_g(t)\) agree near \(t=0\).
    \item The centered Birkhoff sums for \(f\) and \(g\) have the same
      asymptotic cumulants of every order; in particular their
      asymptotic variances coincide.
  \end{enumerate}
\end{theorem}

\begin{proof}
  Equality of determinants near \((0,0)\) implies equality of the
  reciprocal determinants and therefore, by
  \Cref{prop:twisted-determinant-orbit-moments}, equality of the trace
  functions
  \[
    \Tr\!\bigl((M_t^{(f)})^n\bigr)
    =\Tr\!\bigl((M_t^{(g)})^n\bigr)
    \qquad (n\ge 1)
  \]
  for \(t\) near \(0\). Differentiating \(r\) times at \(t=0\) proves
  \textup{(i)}.

  For each fixed \(t\) near \(0\), \Cref{thm:fredholm-spectral-rigidity}
  applied to \(M_t^{(f)}\) and \(M_t^{(g)}\) shows that their non-zero
  spectra agree with multiplicity. In particular the simple Perron
  eigenvalue branch agrees, proving \textup{(ii)}. Since the centered
  cumulant generating function is
  \[
    t\longmapsto
    \log\frac{\lambda_f(t)}{\lambda_f(0)}
    -t\,\frac{\lambda_f'(0)}{\lambda_f(0)},
  \]
  the equality \(\lambda_f(t)=\lambda_g(t)\) implies \textup{(iii)}.
\end{proof}

\subsection{Spectral-gap limit laws}

\begin{proposition}[Spectral-gap central limit theorem]
  \label{thm:clt-spectral}
  Assume \(f\) is not a coboundary modulo constants. Write
  \(\bar f:=f-\int f\,\dd\mu_A\) for the centred observable. Then there
  exists \(\sigma_f^2>0\) such that
  \[
    \frac{\sum_{k=0}^{n-1}\bar f\circ\sigma_A^k}{\sqrt n}
    \Longrightarrow \mathcal N(0,\sigma_f^2).
  \]
  Moreover, for every compact set \(K\Subset\RR\setminus\{0\}\) there
  exist constants \(C_K>0\) and \(\vartheta_K\in(0,1)\) such that
  \[
    \sup_{t\in K}
    \left|\int e^{it\sum_{k=0}^{n-1}\bar f\circ\sigma_A^k}\,\dd\mu_A\right|
    \le C_K\vartheta_K^n,\qquad n\ge1.
  \]
\end{proposition}

\begin{proof}
  With the one-step recoding above, define
  \[
    \phi_n(t):=
    \int e^{it\sum_{k=0}^{n-1}\bar f\circ\sigma_A^k}\,\dd\mu_A.
  \]
  If \(\widetilde M_t\) denotes the centered weighted matrix, then
  \[
    \phi_n(t)=\frac{\ell^\top \widetilde M_t^n r}{\lambda^n},
  \]
  where \(\lambda=\rho(B)\) and \(\ell,r\) are Perron vectors of \(B\)
  with \(\ell^\top r=1\).

  For real \(t\), Perron--Frobenius comparison gives
  \(\rho(\widetilde M_t)\le\lambda\). If \(\rho(\widetilde M_t)=\lambda\)
  for some \(t\neq0\), then a Wielandt-type equality case yields a
  unimodular diagonal \(D_t\) and phase \(c_t\) with
  \[
    \widetilde M_t=e^{itc_t}D_t^{-1}BD_t,
  \]
  hence \(it\varphi\) is a coboundary and \(f\) is a coboundary modulo
  constants by Liv\v{s}ic, contrary to the hypothesis. Therefore for
  every compact \(K\Subset\RR\setminus\{0\}\),
  \[
    \rho_K:=\max_{t\in K}\rho(\widetilde M_t)<\lambda.
  \]
  In finite dimension this spectral gap implies
  \[
    \|\widetilde M_t^n\|\le C_K\vartheta_K^n\lambda^n
  \]
  uniformly on \(K\) for some \(\vartheta_K\in(0,1)\), proving the
  exponential decay of \(\phi_n(t)\) away from \(0\).

  Near \(t=0\), analytic perturbation gives a simple dominant eigenvalue
  \(\lambda_c(t)\) of \(\widetilde M_t\) and a spectral projector
  \(P_t\) such that
  \[
    \phi_n(t)=\lambda_c(t)^n\ell^\top P_t r + O(\vartheta^n),
    \qquad |t|<\delta,
  \]
  with \(\lambda_c(0)=\lambda\), \(\lambda_c'(0)=0\), and
  \[
    \lambda_c''(0)=\sigma_f^2\lambda,\qquad \sigma_f^2>0
  \]
  whenever \(f\) is not a coboundary modulo constants
  \cite{Ruelle1978Thermodynamic,Baladi2000PositiveTransfer,Kato1995Perturbation}.
  Taking \(t=u/\sqrt n\) and using the exponential bound away from
  \(0\), we obtain
  \[
    \phi_n(u/\sqrt n)\to e^{-\sigma_f^2u^2/2}.
  \]
  L\'evy's continuity theorem yields the normal limit law.
\end{proof}

\begin{remark}
  The determinant package clarifies why
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient,thm:logM-holomorphic-variation,prop:logM-truncation,thm:twisted-determinant-rigidity,thm:clt-spectral}
  belong together. In each case one starts from a simple dominant pole
  of a Fredholm determinant, isolates the reduced resolvent on the
  complementary spectral subspace, and then reads a perturbative
  quantity from that reduced determinant or from the motion of the pole.
  The difference between finite-part constants, orbit moments, and limit
  laws is therefore a difference of observable, not of mechanism.
\end{remark}

\subsection{Explicit computational examples}

\begin{remark}[Computational example: the golden-mean family]
  \label{rem:example-golden-mean}
  Let
  \[
    A_s=
    \begin{pmatrix}
      1+s & 1 \\
      1   & 0
    \end{pmatrix},
    \qquad |s|\ll 1.
  \]
  Its Perron root is
  \[
    \lambda(s)=\frac{1+s+\sqrt{(1+s)^2+4}}{2},
  \]
  and the second eigenvalue is \(\mu_2(s)=-\lambda(s)^{-1}\). Hence
  \[
    C(A_s)
    =\left(1+\lambda(s)^{-2}\right)^{-1},
    \qquad
    \partial_s\log C(A_s)
    =\frac{2\lambda'(s)}{\lambda(s)(\lambda(s)^2+1)}.
  \]
  At \(s=0\), with \(\varphi=(1+\sqrt5)/2\),
  \[
    C(A_0)=\frac{\varphi}{\sqrt5},
    \qquad
    \left.\partial_s\log C(A_s)\right|_{s=0}
    =\frac{2}{\sqrt5(\varphi^2+1)}
    =0.2472135\ldots.
  \]
  The Abel finite part is
  \[
    \log \MM(A_0)
    =\log\frac{\varphi}{\sqrt5}
      +\sum_{k\ge 2}\frac{\mu(k)}{k}
        \log\Bigl((1-\varphi^{1-k})(1+\varphi^{-(k+1)})\Bigr)^{-1},
  \]
  and the determinant truncation from \Cref{prop:logM-truncation} is
  numerically rapid: truncation at \(k=14\) already gives
  \[
    \log\MM(A_0)=-0.82923\ldots.
  \]
\end{remark}

\begin{remark}[Computational example: the full two-shift]
  \label{rem:example-full-two-shift}
  Let
  \[
    A=
    \begin{pmatrix}
      1 & 1 \\
      1 & 1
    \end{pmatrix}
  \]
  and let \(f(x)=\mathbf 1_{\{x_0=1\}}\). Then
  \[
    M_t=
    \begin{pmatrix}
      1 & e^t \\
      1 & e^t
    \end{pmatrix},
    \qquad
    \det(I-zM_t)=1-(1+e^t)z.
  \]
  Thus \(\lambda(t)=1+e^t\) and \(C(M_t)=1\). After centering by
  \(1/2\), the dominant eigenvalue becomes
  \[
    \lambda_c(t)=e^{-t/2}(1+e^t)=2\cosh(t/2).
  \]
  Therefore, for real \(u\),
  \[
    \int e^{iu\sum_{k=0}^{n-1}(f-\frac12)\circ\sigma_A^k}\,\dd\mu_A
    =\left(\frac{\lambda_c(iu)}{2}\right)^n
    =\cos(u/2)^n.
  \]
  Hence
  \[
    \sigma_f^2=\frac14,
  \]
  and on every compact set \(K\Subset\RR\setminus\{0\}\),
  \[
    \sup_{u\in K}
    \left|\int e^{iu\sum_{k=0}^{n-1}(f-\frac12)\circ\sigma_A^k}\,\dd\mu_A\right|
    \le
    \left(\sup_{u\in K}\abs{\cos(u/2)}\right)^n.
  \]
  In this model the determinant package recovers the variance and the
  off-origin decay exactly, not merely asymptotically.
\end{remark}

\subsection{Comparison with classical finite-dimensional perturbation theory}

Classical finite-dimensional perturbation theory starts from a fixed
family of matrices \(A_\theta\) and studies the motion of its isolated
eigenvalues, eigenvectors, and reduced resolvent. That machinery is
already enough to prove
\Cref{thm:finite-part-reduced-determinant-group-inverse-gradient,thm:clt-spectral}
for a \emph{single} symbolic model. What it does not single out is the
determinant as a common, model-independent object.

The Fredholm packaging adds three pieces of structure that are not
visible from Jacobi's formula for one matrix family alone. First,
\Cref{prop:twisted-determinant-orbit-moments} is period-graded: one and
the same determinant organizes all periodic lengths simultaneously via
the Euler product and M\"obius inversion. Second,
\Cref{thm:twisted-determinant-rigidity} compares different shifts or
different observables without putting them on a common state space; the
invariant is the determinant germ, not the ambient matrix size. Third,
\Cref{thm:logM-holomorphic-variation,prop:logM-truncation} split the
Abel finite part into a reduced determinant plus a convergent hierarchy
of secondary Fredholm factors. This gives a canonical approximation
scheme for \(\log\MM(A_\theta)\) that ordinary finite-dimensional
perturbation theory does not isolate on its own.

In short, finite-dimensional perturbation theory explains how a chosen
matrix family moves. The determinant-theoretic formulation also records
what survives when the ambient presentation changes, which orbit-length
data are encoded in the motion, and how to pass from local spectral data
to globally organized symbolic invariants.
