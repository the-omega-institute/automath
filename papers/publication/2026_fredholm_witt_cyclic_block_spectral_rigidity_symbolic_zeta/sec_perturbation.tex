\section{Perturbative Applications}
\label{sec:perturbation}

\subsection{Holomorphic variation of finite-part constants}

\begin{theorem}[Reduced-determinant gradient formula]
  \label{thm:finite-part-reduced-determinant-group-inverse-gradient}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Choose left and
  right Perron vectors \(\ell_\theta,r_\theta\) normalised by
  \(\ell_\theta^{\mathsf T}r_\theta=1\), let
  \(P_\theta:=r_\theta\ell_\theta^{\mathsf T}\), and set
  \[
    B_\theta:=I-\frac{A_\theta}{\lambda(\theta)},
    \qquad
    R_\theta:=B_\theta^{\#},
  \]
  where \(B_\theta^{\#}\) is the group inverse
  \cite{CampbellMeyer1979}, characterised by
  \[
    B_\theta R_\theta=R_\theta B_\theta=I-P_\theta,
    \qquad
    R_\theta P_\theta=P_\theta R_\theta=0.
  \]
  Then
  \[
    \partial_\theta\log C(A_\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right),
  \]
  with
  \[
    \partial_\theta\lambda(\theta)
    =\ell_\theta^{\mathsf T}(\partial_\theta A_\theta)r_\theta.
  \]
\end{theorem}

\begin{proof}
  Since the Perron eigenvalue is simple
  \cite[Chapter~II, \S1]{Kato1995Perturbation}, \(B_\theta\) has a
  simple zero eigenvalue and all other eigenvalues are non-zero; hence
  the group inverse \(R_\theta\) exists. Because \(\lambda(\theta)\) is
  a simple eigenvalue varying analytically, the spectral projection
  \(P_\theta\) and hence the group inverse
  \(R_\theta=B_\theta^{\#}\) depend analytically on \(\theta\) by
  standard resolvent calculus
  \cite[Chapter~II, \S1.4]{Kato1995Perturbation}. On the invariant complement
  \((I-P_\theta)\CC^d\), the reduced determinant is the ordinary
  determinant of \(B_\theta\), so the usual logarithmic derivative
  gives
  \[
    \partial_\theta\log\det\nolimits'(B_\theta)
    =\Tr(R_\theta\,\partial_\theta B_\theta).
  \]
  Since \(C(A_\theta)=\det{}'(B_\theta)^{-1}\) by
  \eqref{eq:reduced-det-constant}, we have
  \(\partial_\theta\log C=-\partial_\theta\log\det\nolimits'(B_\theta)
  =-\Tr(R_\theta\,\partial_\theta B_\theta)\). Now
  \[
    \partial_\theta B_\theta
    =-\frac{\partial_\theta A_\theta}{\lambda(\theta)}
      +\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
       A_\theta,
  \]
  so the two minus signs cancel to give the stated formula. The
  derivative of the Perron root is the
  standard perturbation identity
  \[
    \partial_\theta\lambda(\theta)
    =\ell_\theta^{\mathsf T}(\partial_\theta A_\theta)r_\theta.
  \]
\end{proof}

\begin{theorem}[Holomorphic variation of the Abel finite part]
  \label{thm:logM-holomorphic-variation}
  Let \(\theta\mapsto A_\theta\) be a holomorphic family of primitive
  matrices near \(\theta_0\), and assume that
  \(\lambda(\theta):=\rho(A_\theta)\) stays simple. Then
  \(\theta\mapsto\log\MM(A_\theta)\) is holomorphic near \(\theta_0\)
  and
  \[
    \frac{\dd}{\dd\theta}\log\MM(A_\theta)
    =\frac{\dd}{\dd\theta}\log C(A_\theta)
     +\sum_{k\ge 2}\frac{\mu(k)}{k}
      \frac{\dd}{\dd\theta}
      \log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}.
  \]
  Moreover, for every fixed \(z\) with \(|z|<\lambda(\theta)^{-1}\),
  \[
    \partial_\theta\log\det(I-zA_\theta)^{-1}
    =\Tr\!\bigl((I-zA_\theta)^{-1}z\,\partial_\theta A_\theta\bigr).
  \]
\end{theorem}

\begin{proof}
  Apply \Cref{thm:finite-part-primitive-closed-form} to each
  \(A_\theta\). For \(k\ge 2\) and \(\theta\) in a neighbourhood of
  \(\theta_0\), set
  \(\lambda_{\min}:=\min_{|\theta-\theta_0|\le\epsilon}
  \lambda(\theta)>1\); then
  \(|\lambda(\theta)^{-k}|\le\lambda_{\min}^{-2}\) for all \(k\ge 2\),
  so the series of holomorphic functions
  \(\sum_{k\ge 2}(\mu(k)/k)\log\det(I-\lambda(\theta)^{-k}A_\theta)^{-1}\)
  converges normally in \(\theta\) and may be differentiated term by
  term. The derivative of \(\log C(A_\theta)\) is given by
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient}.

  For the resolvent identity, write
  \[
    \log\det(I-zA_\theta)^{-1}=-\Tr\log(I-zA_\theta)
  \]
  and differentiate inside the trace.
\end{proof}

\subsection{Spectral-gap limit laws}

Let \((X_A^+,\sigma)\) be the mixing subshift determined by a primitive
matrix \(A\), let \(\mu\) be its Parry measure, and let
\(f:X_A^+\to\RR\) be a locally constant observable. For \(\theta\in\CC\)
near \(0\), define the twisted transfer operator
\[
  (\mathcal L_\theta g)(x)
  :=\sum_{\sigma y=x} e^{\theta f(y)}g(y).
\]

\begin{proposition}[Spectral-gap central limit theorem]
  \label{thm:clt-spectral}
  The following is a standard consequence of the Nagaev--Guivarc'h
  method; we include it to illustrate the common spectral-gap mechanism
  underlying the results of this section. Assume that \(f\) is not a
  coboundary modulo constants. Write \(\bar f:=f-\int f\,\dd\mu\) for
  the centred observable. Then there exists \(\sigma_f^2>0\) such that
  \[
    \frac{\sum_{k=0}^{n-1}\bar f\circ\sigma^k}{\sqrt n}
    \Longrightarrow \mathcal N(0,\sigma_f^2).
  \]
  Moreover, for every compact set \(K\Subset\RR\setminus\{0\}\) there
  exist constants \(C_K>0\) and \(\vartheta_K\in(0,1)\) such that
  \[
    \sup_{t\in K}
    \abs{\int e^{it\sum_{k=0}^{n-1}\bar f\circ\sigma^k}\,\dd\mu}
    \le C_K\vartheta_K^n
  \]
  for all \(n\ge 1\).
\end{proposition}

\begin{proof}
  Because \(f\) is locally constant, one may work on a
  finite-dimensional space of cylinder functions of sufficiently large
  range on which the operators \(\mathcal L_\theta\) act analytically.
  Standard perturbation theory
  \cite[Chapter~VII, \S4]{Kato1995Perturbation}
  therefore yields a simple dominant
  eigenvalue \(\lambda(\theta)\) and a spectral gap uniform for
  \(\theta\) near \(0\); see
  \cite[Chapters~1 and~3]{Baladi2000PositiveTransfer} and
  \cite[Chapter~2]{ParryPollicott1990Zeta}. Since \(\bar f\) is
  centred, \(\lambda'(0)=0\) and the cumulant generating function
  \(\log\lambda(\theta)\) satisfies
  \[
    \frac{\dd^2}{\dd\theta^2}\bigg|_{\theta=0}\log\lambda(\theta)
    =\frac{\lambda''(0)}{\lambda(0)}=:\sigma_f^2.
  \]
  Here \(\lambda''(0)/\lambda(0)\) equals the asymptotic variance
  \(\lim_{n\to\infty}n^{-1}\int(\sum_{k=0}^{n-1}\bar f\circ\sigma^k)^2
  \,\dd\mu\), as follows from expanding the exponential moment
  \(\int e^{\theta S_n f}\,\dd\mu\) to second order in \(\theta\).
  Since \(f\) is not a coboundary modulo constants,
  \(\sigma_f^2>0\). The Nagaev--Guivarc'h method
  \cite{Nagaev1957,GuivarchHardy1988} then gives the
  central limit theorem, while the uniform spectral gap for
  \(\theta=it\) with \(t\in K\) gives the stated exponential bound for
  the characteristic functions.
\end{proof}

\begin{remark}
  The arguments in
  \Cref{thm:finite-part-reduced-determinant-group-inverse-gradient,thm:logM-holomorphic-variation,thm:clt-spectral}
  rely on three shared ingredients: (1) a simple dominant eigenvalue
  \(\lambda(\theta)\) of the transfer operator, (2) the reduced
  resolvent (or group inverse) on the spectral complement of
  \(\lambda(\theta)\), and (3) analytic dependence of the spectral
  data on the parameter \(\theta\). The operator-theoretic viewpoint
  of \Cref{sec:fredholm} isolates exactly this common core, and the
  gradient formula and the CLT are two manifestations of the same
  spectral-gap mechanism applied to different functionals.
\end{remark}
