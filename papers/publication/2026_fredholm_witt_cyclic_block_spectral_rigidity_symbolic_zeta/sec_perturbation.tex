\section{A classical finite-part illustration}
\label{sec:perturbation}

This appendix keeps a single finite-dimensional illustration of the
determinant viewpoint from \Cref{sec:fredholm}. The formulas recorded
here are classical consequences of analytic perturbation theory and
Fredholm determinant identities; compare
\cite{Ruelle1978Thermodynamic,Baladi2000PositiveTransfer,PollicottVytnova2016,Jezequel2019Parameter}.
They are not used in the proof of the operator-theoretic results and are
included only to show, in one concrete symbolic setting, how the
determinant package reorganizes standard finite-part data.

\medskip

\noindent\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
\textbf{Standing assumptions for this section.}
Throughout \Cref{sec:perturbation}, the following hypotheses are in
force unless stated otherwise:
\begin{enumerate}[label=\textup{(SA\arabic*)},leftmargin=*,nosep]
  \item\label{SA:primitive}
    \(\theta\mapsto A_\theta\) is a holomorphic family of primitive
    \(d\times d\) matrices near \(\theta_0\in\CC\).
  \item\label{SA:simple-dominant}
    \(A_{\theta_0}\) has a simple dominant eigenvalue
    \(\lambda_0=\rho(A_{\theta_0})>0\) with
    \(\abs{\mu}<\lambda_0\) for every other eigenvalue \(\mu\).
  \item\label{SA:spectral-gap}
    After shrinking to a simply connected neighbourhood \(U\ni\theta_0\),
    the Riesz projector \(P_\theta\) and analytic eigenvalue branch
    \(\lambda(\theta)\) from \Cref{def:spectral-gap-hypothesis} are
    defined on \(U\), and there exists \(q\in(0,1)\) such that
    \[
      \rho\!\bigl(A_\theta\bigm|_{(I-P_\theta)\CC^d}\bigr)
      \le q\,\abs{\lambda(\theta)}
      \qquad (\theta\in U).
    \]
  \item\label{SA:expansive}
    After shrinking \(U\) further if necessary,
    \[
      \Lambda_U:=\inf_{\theta\in U}\abs{\lambda(\theta)}>1.
    \]
\end{enumerate}}}

\subsection{Analytic eigenvalue branches and algebraic finite-part functionals}

\begin{definition}[Analytic spectral-gap hypothesis]
  \label{def:spectral-gap-hypothesis}
  Let \(\theta\mapsto A(\theta)\) be a holomorphic family of
  \(d\times d\) complex matrices near \(\theta_0\in\CC\). We say that
  \(A(\theta)\) satisfies the \emph{analytic spectral-gap hypothesis at
  \(\theta_0\)} if \(A(\theta_0)\) has a simple eigenvalue
  \(\lambda_0\ne 0\) such that
  \[
    \abs{\mu}<\abs{\lambda_0}
    \quad\text{for every }
    \mu\in\spec\!\bigl(A(\theta_0)\bigr)\setminus\{\lambda_0\}.
  \]
  Under this condition, after shrinking to a simply connected
  neighbourhood \(U\ni\theta_0\), the standard Riesz projection
  \[
    P(\theta)
    :=\frac{1}{2\pi i}
      \int_{\abs{z-\lambda_0}=\delta}
      (zI-A(\theta))^{-1}\,\dd z
  \]
  (for \(\delta>0\) chosen so that no other eigenvalue of
  \(A(\theta_0)\) lies in \(\abs{z-\lambda_0}\le\delta\)) defines a
  holomorphic rank-one projector. The corresponding eigenvalue branch
  \[
    \lambda(\theta):=\Tr(A(\theta)P(\theta))
  \]
  is the unique holomorphic function on \(U\) with \(\lambda(\theta_0)
  =\lambda_0\) and \(A(\theta)P(\theta)=\lambda(\theta)P(\theta)\).
  When \(A(\theta)\) is a primitive matrix with Perron root \(\lambda_0
  =\rho(A(\theta_0))>0\), this is exactly the Perron--Frobenius
  spectral gap, and \(\lambda(\theta)\) agrees with \(\rho(A(\theta))\)
  on the real slice where \(A(\theta)\) remains primitive.
\end{definition}

\begin{definition}[Algebraic finite-part functionals]
  \label{def:algebraic-finite-part}
  Let \(A\) be a \(d\times d\) matrix whose leading eigenvalue
  \(\lambda=\rho(A)>1\) has one-dimensional eigenspace and satisfies
  \[
    \abs{\mu}<\lambda
    \quad\text{for every }
    \mu\in\spec(A)\setminus\{\lambda\}.
  \]
  Set \(B:=I-A/\lambda\). The
  \emph{reduced determinant} is
  \[
    C(A,\lambda)
    :=\det\nolimits'(B)^{-1}
    :=\prod_{\mu\in\spec(A)\setminus\{\lambda\}}
      \!\!\left(1-\frac{\mu}{\lambda}\right)^{-1},
  \]
  where the product is taken with algebraic multiplicity. For each
  \(k\ge 2\), the \emph{secondary Fredholm logarithm} is
  \[
    \Phi_k(A,\lambda)
    :=\sum_{m\ge 1}\frac{\Tr(A^m)}{m\,\lambda^{km}}.
  \]
  Because \(\rho(\lambda^{-k}A)=\abs{\lambda}^{1-k}<1\), the series for
  \(\Phi_k(A,\lambda)\) converges absolutely and equals the holomorphic
  logarithm determined by that series:
  \[
    \log\det\!\bigl(I-\lambda^{-k}A\bigr)^{-1}.
  \]
  The \emph{Abel finite-part functional} is
  \[
    F(A,\lambda)
    :=\log C(A,\lambda)
      +\sum_{k\ge 2}\frac{\mu(k)}{k}\,\Phi_k(A,\lambda).
  \]
\end{definition}

\begin{remark}[Why the expansive hypothesis is needed]
  \label{rem:expansive-hypothesis-needed}
  A bare spectral gap at an isolated eigenvalue does not guarantee that
  the matrices \(I-\lambda^{-k}A\) are invertible for \(k\ge 2\). The
  stronger requirement that \(\lambda=\rho(A)\) be a one-dimensional
  leading eigenvalue with \(\lambda>1\) rules out resonances
  \(\mu=\lambda^k\) with \(k\ge 2\) and makes the secondary logarithms
  in \Cref{def:algebraic-finite-part} unambiguous.
\end{remark}

\subsection{Finite-part variation and truncation}

The next proposition is the only result retained in this section. It
collects the standard holomorphic-variation and truncation facts needed
for the example below. Assumptions \ref{SA:primitive}--\ref{SA:expansive}
are in force, and the proof is deferred to
\Cref{sec:appendix-proofs}.

\begin{proposition}[Holomorphic variation and truncation of the Abel finite part]
  \label{prop:finite-part-illustration}
  Define
  \[
    R_\theta:=\bigl(I-A_\theta/\lambda(\theta)\bigr)^{\#},
    \qquad
    \Phi_k(\theta):=\Phi_k(A_\theta,\lambda(\theta))
    \qquad (k\ge 2),
  \]
  and
  \[
    F(\theta):=F(A_\theta,\lambda(\theta))
    =\log C(A_\theta,\lambda(\theta))
      +\sum_{k\ge 2}\frac{\mu(k)}{k}\,\Phi_k(\theta).
  \]
  Then \(R_\theta\) exists for every \(\theta\in U\), the correction
  series for \(F(\theta)\) converges normally on \(U\), and \(F\) is
  holomorphic there. Moreover,
  \[
    \frac{\dd}{\dd\theta}F(\theta)
    =\Tr\!\left(
      R_\theta\left(
        \frac{\partial_\theta A_\theta}{\lambda(\theta)}
        -\frac{\partial_\theta\lambda(\theta)}{\lambda(\theta)^2}
         A_\theta\right)\right)
    +\sum_{k\ge 2}\frac{\mu(k)}{k}\,\partial_\theta\Phi_k(\theta).
  \]
  If
  \[
    F_N(\theta)
    :=\log C(A_\theta,\lambda(\theta))
      +\sum_{2\le k\le N}\frac{\mu(k)}{k}\,\Phi_k(\theta),
  \]
  then there exist \(C_U>0\) and \(\eta_U\in(0,1)\) such that
  \[
    \sup_{\theta\in U}\abs{F(\theta)-F_N(\theta)}
    \le C_U\eta_U^{N+1}
    \qquad (N\ge 2),
  \]
  and the same estimate holds after one \(\theta\)-derivative.
\end{proposition}

\subsection{Explicit computational example}

\begin{remark}[Computational example: the golden-mean family]
  \label{rem:example-golden-mean}
  Let
  \[
    A_s=
    \begin{pmatrix}
      1+s & 1 \\
      1   & 0
    \end{pmatrix},
    \qquad |s|\ll 1.
  \]
  For real \(s\) near \(0\), this is a primitive matrix. The analytic
  simple eigenvalue branch from \Cref{def:spectral-gap-hypothesis} is
  \[
    \lambda(s)=\frac{1+s+\sqrt{(1+s)^2+4}}{2},
  \]
  and the second eigenvalue is \(\mu_2(s)=-\lambda(s)^{-1}\). Hence,
  by \Cref{def:algebraic-finite-part},
  \[
    C(A_s,\lambda(s))
    =\left(1+\lambda(s)^{-2}\right)^{-1},
    \qquad
    \partial_s\log C(A_s,\lambda(s))
    =\frac{2\lambda'(s)}{\lambda(s)(\lambda(s)^2+1)}.
  \]
  At \(s=0\), with \(\varphi=(1+\sqrt5)/2\),
  \[
    C(A_0,\varphi)=\frac{\varphi}{\sqrt5},
    \qquad
    \left.\partial_s\log C(A_s,\lambda(s))\right|_{s=0}
    =\frac{2}{\sqrt5(\varphi^2+1)}
    =0.2472135\ldots.
  \]
  The finite-part functional is
  \[
    F(A_0,\varphi)
    =\log\frac{\varphi}{\sqrt5}
      +\sum_{k\ge 2}\frac{\mu(k)}{k}\,
        \Phi_k(A_0,\varphi),
  \]
  and the truncation from \Cref{prop:finite-part-illustration} is
  numerically rapid: keeping the terms \(2\le k\le 14\) already gives
  \[
    F(A_0,\varphi)=-0.82923\ldots.
  \]
  This provides a concrete determinant-level check of the finite-part
  formulas in a nontrivial holomorphic family.
\end{remark}
