\section{Determinants and Euler products}\label{sec:preliminaries}

We begin by recording the two determinant identities that will be used
throughout the paper. The first is the finite-dimensional formula behind
\eqref{eq:intro-finite-zeta}.

Let \(A\) be a complex \(d\times d\) matrix and set
\[
  a_n(A):=\Tr(A^n),
  \qquad
  p_n(A):=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(A)
  \qquad (n\ge 1).
\]

\begin{proposition}[Finite matrices and Euler products]
  \label{prop:prelim-trace-primitive}
  For every complex \(d\times d\) matrix \(A\),
  \begin{equation}\label{eq:zeta-finite-matrix}
    \det(I-zA)^{-1}
    =\exp\!\left(\sum_{n\ge 1}\frac{\Tr(A^n)}{n}z^n\right)
    =\prod_{n\ge 1}(1-z^n)^{-p_n(A)}
  \end{equation}
  as an identity of holomorphic germs at \(z=0\), equivalently as an
  identity of formal power series. If \(A\) is a primitive
  \(\{0,1\}\)-matrix, then \(p_n(A)\) is the number of primitive orbits
  of length \(n\) for the associated subshift
  \cite{BowenLanford1970Zeta,LindMarcus1995}.
\end{proposition}

\begin{proof}
  The determinant formula is classical and follows from the Newton
  identities. M\"obius inversion gives
  \(
    a_n(A)=\sum_{d\mid n}d\,p_d(A)
  \),
  and hence
  \[
    \sum_{n\ge 1}\frac{a_n(A)}{n}z^n
    =-\sum_{m\ge 1}p_m(A)\log(1-z^m).
  \]
  Exponentiating yields \eqref{eq:zeta-finite-matrix}. Analytically, the
  argument is valid for \(\abs{z}<\min(1,\|A\|^{-1})\), and formally no
  convergence issue arises. The interpretation of \(p_n(A)\) in terms of
  primitive periodic orbits is standard.
\end{proof}

The trace-class analogue is equally standard.

\begin{proposition}[Fredholm determinants and Euler products]
  \label{thm:fredholm-witt-bridge}
  Let \(H\) be a complex Hilbert space and let \(T\in \sone(H)\). Write
  \[
    Z_T(z):=\det(I-zT)^{-1},
    \qquad
    a_n(T):=\Tr(T^n),
  \]
  and
  \[
    p_n(T):=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(T)
    \qquad (n\ge 1).
  \]
  Then, for
  \(\abs{z}<\min(1,\|T\|^{-1})\),
  \[
    Z_T(z)
    =\exp\!\left(\sum_{n\ge 1}\frac{a_n(T)}{n}z^n\right)
    =\prod_{n\ge 1}(1-z^n)^{-p_n(T)},
  \]
  where
  \(
    (1-z^m)^{-p_m(T)}
    :=\exp\!\bigl(-p_m(T)\log(1-z^m)\bigr)
  \)
  and \(\log(1-z^m)\) is the principal branch on \(\abs{z}<1\). In
  particular, the identity holds as an equality of holomorphic germs at
  \(z=0\).
\end{proposition}

\begin{proof}
  The first identity,
  \[
    Z_T(z)
    =\exp\!\left(\sum_{n\ge 1}\frac{\Tr(T^n)}{n}z^n\right),
  \]
  is Grothendieck's trace formula; see \cite{Grothendieck1956} or
  \cite[Chapter~3]{Simon2005TraceIdeals}. Since
  \(a_n(T)=\sum_{d\mid n}d\,p_d(T)\), we find
  \begin{align*}
    \sum_{n\ge 1}\frac{a_n(T)}{n}z^n
      &=\sum_{n\ge 1}\sum_{d\mid n}\frac{d\,p_d(T)}{n}z^n \\
      &=\sum_{m\ge 1}\sum_{r\ge 1}\frac{p_m(T)}{r}z^{mr}
       =-\sum_{m\ge 1}p_m(T)\log(1-z^m).
  \end{align*}
  The logarithm is single-valued for \(\abs{z}<1\), while the trace
  series converges for \(\abs{z}<\|T\|^{-1}\). Exponentiating gives the
  stated product formula.
\end{proof}

\begin{remark}[Euler coordinates]
  \label{rem:witt-audit-constraints}
  In contrast with the finite-dimensional symbolic setting, the
  coefficients \(p_n(T)\) need not be integers and need not be
  non-negative. Accordingly, the product formula is to be understood as
  an identity of holomorphic germs, interpreted through the chosen branch
  of the logarithm.
\end{remark}
