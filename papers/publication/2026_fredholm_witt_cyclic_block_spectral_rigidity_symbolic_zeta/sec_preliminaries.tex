\section{Finite matrices and Fredholm determinants}
\label{sec:preliminaries}

We begin with the two standard determinant identities that underlie the
paper. The first is the familiar finite-dimensional formula from
symbolic dynamics.

Let \(A\) be a complex \(d\times d\) matrix and set
\[
  a_n(A):=\Tr(A^n),
  \qquad
  p_n(A):=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(A)
  \qquad (n\ge 1).
\]

\begin{proposition}[Finite matrices and Euler products]
  \label{prop:prelim-trace-primitive}
  For every complex \(d\times d\) matrix \(A\),
  \begin{equation}\label{eq:zeta-finite-matrix}
    \det(I-zA)^{-1}
    =\exp\!\left(\sum_{n\ge 1}\frac{\Tr(A^n)}{n}z^n\right)
    =\prod_{n\ge 1}(1-z^n)^{-p_n(A)}
  \end{equation}
  as an identity of holomorphic germs at \(z=0\), equivalently as an
  identity of formal power series. If \(A\) is a primitive
  \(\{0,1\}\)-matrix, then \(p_n(A)\) is the number of primitive orbits
  of length \(n\) for the associated subshift
  \cite{BowenLanford1970Zeta,LindMarcus1995}.
\end{proposition}

\begin{proof}
  The determinant identity is classical and follows from the Newton
  identities. By M\"obius inversion,
  \(
    a_n(A)=\sum_{d\mid n}d\,p_d(A)
  \),
  and hence
  \[
    \sum_{n\ge 1}\frac{a_n(A)}{n}z^n
    =-\sum_{m\ge 1}p_m(A)\log(1-z^m).
  \]
  Exponentiating gives \eqref{eq:zeta-finite-matrix}; this is valid
  either formally or, analytically, for
  \(\abs{z}<\min(1,\|A\|^{-1})\).
  The primitive-orbit interpretation is standard.
\end{proof}

The second identity is the trace-class analogue.

\begin{proposition}[Fredholm determinants and Euler products]
  \label{thm:fredholm-witt-bridge}
  Let \(H\) be a complex Hilbert space and let \(T\in \sone(H)\). Write
  \[
    Z_T(z):=\det(I-zT)^{-1},
    \qquad
    a_n(T):=\Tr(T^n),
  \]
  and
  \[
    p_n(T):=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(T)
    \qquad (n\ge 1).
  \]
  Then, for
  \(\abs{z}<\min(1,\|T\|^{-1})\),
  \[
    Z_T(z)
    =\exp\!\left(\sum_{n\ge 1}\frac{a_n(T)}{n}z^n\right)
    =\prod_{n\ge 1}(1-z^n)^{-p_n(T)},
  \]
  where
  \(
    (1-z^m)^{-p_m(T)}
    :=\exp\!\bigl(-p_m(T)\log(1-z^m)\bigr)
  \)
  and \(\log(1-z^m)\) is the principal branch on \(\abs{z}<1\). In
  particular, the identity holds as an equality of holomorphic germs at
  \(z=0\).
\end{proposition}

\begin{proof}
  The Fredholm determinant identity
  \[
    Z_T(z)
    =\exp\!\left(\sum_{n\ge 1}\frac{\Tr(T^n)}{n}z^n\right)
  \]
  is classical; see \cite{Grothendieck1956} and
  \cite[Chapter~3]{Simon2005TraceIdeals}. By definition of \(p_n(T)\),
  M\"obius inversion gives
  \[
    a_n(T)=\sum_{d\mid n}d\,p_d(T).
  \]
  Therefore
  \begin{align*}
    \sum_{n\ge 1}\frac{a_n(T)}{n}z^n
      &=\sum_{n\ge 1}\sum_{d\mid n}\frac{d\,p_d(T)}{n}z^n \\
      &=\sum_{m\ge 1}\sum_{r\ge 1}\frac{p_m(T)}{r}z^{mr}
       =-\sum_{m\ge 1}p_m(T)\log(1-z^m).
  \end{align*}
  The principal branch is single-valued for \(\abs{z}<1\), while the
  trace series converges for \(\abs{z}<\|T\|^{-1}\). Exponentiating
  yields the stated formula.
\end{proof}

\begin{remark}[Euler coordinates]
  \label{rem:witt-audit-constraints}
  In contrast with the finite-dimensional symbolic setting, the numbers
  \(p_n(T)\) need not be integral and need not be non-negative. Thus
  the product formula above is only an identity of germs, interpreted
  through the chosen branch of the logarithm.
\end{remark}
