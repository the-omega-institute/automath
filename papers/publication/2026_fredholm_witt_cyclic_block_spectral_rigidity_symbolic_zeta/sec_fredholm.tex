\section{Cyclic-block realizations in trace class}\label{sec:fredholm}

We begin with the normal trace-class case. For \(T\in \sone(H)\), we set
\[
  N(T):=\sum_{\lambda\in\spec(T)\setminus\{0\}}m_a(\lambda)
  \in \NN\cup\{\infty\},
\]
where \(m_a(\lambda)\) denotes the algebraic multiplicity of the
eigenvalue \(\lambda\). If \(L\) is compact and normal, then
\[
  N(L)
  =\operatorname{rank}\mathbf 1_{(0,\infty)}(|L|)
  =\dim\overline{\operatorname{ran}(L)}.
\]
Thus \(N(L)\) is the rank of the support projection rather than the
dimension of the possibly non-closed range itself.

For \(n\ge 1\), let \(\Pi_n\) be the cyclic permutation matrix on
\(\CC^n\), and for \(\alpha\in\CC\) define
\[
  C(n,\alpha):=\alpha\Pi_n.
\]
Since the eigenvalues of \(\Pi_n\) are the \(n\)th roots of unity, one
has
\begin{equation}\label{eq:cycle-block-det}
  \det(I-zC(n,\alpha))=1-\alpha^n z^n.
\end{equation}
The next theorem shows that these cyclic blocks are exactly the normal
building blocks compatible with a prescribed Euler product.

\begin{theorem}[Cyclic-block realization theorem]
  \label{thm:cyclic-fredholm-realisation}
  Let
  \(
    \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J}
  \)
  be a countable family with
  \[
    \begin{aligned}
      n_j,m_j&\in\NN,\\
      \beta_j&\in\CC\setminus\{0\},
    \end{aligned}
  \]
  and assume that
  \[
    \sum_{j\in J}m_j n_j\abs{\beta_j}^{1/n_j}<\infty.
  \]
  Choose \(\alpha_j\in\CC\) such that \(\alpha_j^{n_j}=\beta_j\), and
  set
  \[
    H_{\mathcal D}
    :=\bigoplus_{j\in J}\bigl(\CC^{n_j}\otimes\CC^{m_j}\bigr),
    \qquad
    L_{\mathcal D}
    :=\bigoplus_{j\in J}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
  \]
  Then \(L_{\mathcal D}\in \sone(H_{\mathcal D})\), and
  \[
    Z_{L_{\mathcal D}}(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  as an identity of holomorphic germs at \(z=0\). In particular, if
  \[
    r_{\mathcal D}:=\sup_{j\in J}\abs{\beta_j}^{1/n_j},
  \]
  then \(r_{\mathcal D}<\infty\) and the identity holds for
  \(\abs{z}<r_{\mathcal D}^{-1}\).

  Moreover, if \(L\in \sone(H)\) is normal, then the following are
  equivalent:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item
      \[
        Z_L(z)
        =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
      \]
      in a neighborhood of \(z=0\).
    \item There exist a Hilbert space \(H_0\) and a unitary operator
      \(U:H\to H_0\oplus H_{\mathcal D}\) such that
      \[
        ULU^{-1}=0_{H_0}\oplus L_{\mathcal D}.
      \]
  \end{enumerate}
  In particular, every normal realization satisfies
  \[
    N(L)=\sum_{j\in J}m_jn_j,
    \qquad
    \|L\|_1=\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
\end{theorem}

\begin{proof}
  Each block \(C(n_j,\alpha_j)\otimes I_{m_j}\) has singular value
  \(\abs{\alpha_j}\) with multiplicity \(m_jn_j\). Therefore
  \[
    \|L_{\mathcal D}\|_1
    =\sum_{j\in J}m_jn_j\abs{\alpha_j}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}
    <\infty,
  \]
  so \(L_{\mathcal D}\) is trace class. Since \(m_jn_j\ge 1\) for every
  \(j\), the same summability implies
  \(
    r_{\mathcal D}<\infty
  \).

  For each \(j\), \eqref{eq:cycle-block-det} gives
  \[
    \det\!\bigl(I-z(C(n_j,\alpha_j)\otimes I_{m_j})\bigr)
    =\det(I-zC(n_j,\alpha_j))^{m_j}
    =(1-\beta_j z^{n_j})^{m_j}.
  \]
  Let
  \[
    L_{\mathcal D}^{(N)}
    :=\bigoplus_{j\le N}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
  \]
  Then
  \[
    \|L_{\mathcal D}-L_{\mathcal D}^{(N)}\|_1
    =\sum_{j>N}m_jn_j\abs{\beta_j}^{1/n_j}\to 0.
  \]
  By the trace-norm continuity of the Fredholm determinant
  \cite[Theorem~3.4]{Simon2005TraceIdeals},
  \[
    \det(I-zL_{\mathcal D})
    =\lim_{N\to\infty}\det(I-zL_{\mathcal D}^{(N)})
  \]
  locally uniformly in \(z\). Since
  \[
    Z_{L_{\mathcal D}^{(N)}}(z)
    =\prod_{j\le N}(1-\beta_j z^{n_j})^{-m_j},
  \]
  passage to the limit yields
  \[
    Z_{L_{\mathcal D}}(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  as a germ at \(0\). The operator norm satisfies
  \[
    \|L_{\mathcal D}\|
    =\sup_{j\in J}\abs{\alpha_j}
    =\sup_{j\in J}\abs{\beta_j}^{1/n_j}
    =r_{\mathcal D},
  \]
  so the identity is valid in particular for \(\abs{z}<r_{\mathcal D}^{-1}\).

  The implication \textup{(ii)}\(\Rightarrow\)\textup{(i)} is immediate.
  Suppose now that \textup{(i)} holds. The first part of the theorem
  furnishes a normal operator \(L_{\mathcal D}\) with the same Fredholm
  zeta germ, and hence
  \[
    \det(I-zL)=\det(I-zL_{\mathcal D})
  \]
  as entire functions. By \cite[Theorem~3.7]{Simon2005TraceIdeals}, the
  zeros of \(\det(I-zT)\) are precisely the reciprocals of the non-zero
  eigenvalues of \(T\), counted with algebraic multiplicity. Therefore
  \(L\) and \(L_{\mathcal D}\) have the same non-zero spectrum with
  multiplicity.

  The operator \(L_{\mathcal D}\) is normal, and its non-zero
  eigenvalues are the numbers
  \[
    \alpha_j\omega,
    \qquad
    \omega^{n_j}=1,
  \]
  each occurring with multiplicity \(m_j\). Since \(L\) is compact and
  normal, the spectral theorem \cite{Conway1990Course} gives an
  orthogonal decomposition
  \[
    H=\ker L\oplus\bigoplus_{\lambda\in\spec(L)\setminus\{0\}}E_\lambda,
    \qquad
    L|_{E_\lambda}=\lambda I.
  \]
  For each non-zero eigenvalue \(\lambda\), let
  \[
    J_\lambda
    :=\bigl\{j\in J:\lambda=\alpha_j\omega
      \text{ for some }\omega^{n_j}=1\bigr\}.
  \]
  Since the multiplicity of \(\lambda\) is finite,
  \[
    \dim E_\lambda=\sum_{j\in J_\lambda}m_j.
  \]
  Choose an orthogonal decomposition
  \[
    E_\lambda=\bigoplus_{j\in J_\lambda}E_{\lambda,j},
    \qquad
    \dim E_{\lambda,j}=m_j.
  \]
  For each \(j\), set
  \[
    H_j:=\bigoplus_{\omega^{n_j}=1}E_{\alpha_j\omega,j}.
  \]
  Then \(H_j\) is reducing for \(L\), the subspaces \(H_j\) are pairwise
  orthogonal, and \(L|_{H_j}\) is unitarily equivalent to
  \[
    \bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}.
  \]
  The discrete Fourier transform on \(\CC^{n_j}\) diagonalizes
  \(\Pi_{n_j}\), so
  \[
    \bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}
    \cong C(n_j,\alpha_j)\otimes I_{m_j}.
  \]
  Hence \(L\) is unitarily equivalent to
  \(
    0_{\ker L}\oplus L_{\mathcal D}
  \),
  which proves \textup{(ii)}. The formulas for \(N(L)\) and \(\|L\|_1\)
  are now immediate from the first part of the proof.
\end{proof}

We shall refer to any normal operator satisfying
\Cref{thm:cyclic-fredholm-realisation}\textup{(ii)} as a
\emph{block-cyclic normal realization} of \(\mathcal D\).

\begin{remark}[Scope of the realization theorem]
  \label{rem:normal-scope-trace}
  \Cref{thm:cyclic-fredholm-realisation} is an exact existence and
  uniqueness statement only in the normal trace-class category. The
  later rigidity and Jordan-classification theorems do not supply a
  separate realization criterion for arbitrary nonnormal trace-class
  operators. Rather, the summability hypothesis is used there only to
  build the canonical normal model \(L_{\mathcal D}\), after which the
  determinant forces the same non-zero spectral data for every
  trace-class realization with the same germ.
\end{remark}

\begin{remark}[Choice of roots]
  \label{rem:choice-of-roots}
  Replacing \(\alpha_j\) by another \(n_j\)th root of \(\beta_j\) merely
  rotates the same regular \(n_j\)-gon of eigenvalues. Hence the block
  \(C(n_j,\alpha_j)\otimes I_{m_j}\) is determined only up to unitary
  equivalence, which is all that is needed in
  \Cref{thm:cyclic-fredholm-realisation}.
\end{remark}

\section{Determinant rigidity in trace class}
\label{sec:rigidity}

We now discard the normality assumption and keep only the Fredholm
determinant. The next results do not enlarge the realization theorem
from \Cref{thm:cyclic-fredholm-realisation}; rather, once an admissible
block-cyclic model exists, Simon's description of the zeros of
\(\det(I-zT)\) forces the non-zero spectral data of every trace-class
realization.

\begin{proposition}[Rigidity of the non-zero spectrum]
  \label{prop:fredholm-factorisation-rigidity}
  \begin{enumerate}[label=\textup{(\alph*)}]
    \item Let
      \[
        \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J},
        \qquad
        \mathcal D'=\{(n'_\ell,\beta'_\ell,m'_\ell)\}_{\ell\in L},
      \]
      and assume that both data sets satisfy the summability condition
      of \Cref{thm:cyclic-fredholm-realisation}. If the corresponding
      Euler products agree as holomorphic germs at \(z=0\), then the
      block-cyclic models attached to \(\mathcal D\) and
      \(\mathcal D'\) have the same non-zero eigenvalue multiset,
      counted with algebraic multiplicity.
      In particular,
      \[
        \sum_{j\in J}m_jn_j=\sum_{\ell\in L}m'_\ell n'_\ell,
      \]
      and the corresponding trace-norm sums
      \[
        \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j},
        \qquad
        \sum_{\ell\in L}m'_\ell n'_\ell\abs{\beta'_\ell}^{1/n'_\ell},
      \]
      are equal.
    \item Let \(T_1,T_2\in \sone(H)\). If
      \(
        Z_{T_1}(z)=Z_{T_2}(z)
      \)
      in a neighborhood of \(z=0\), then \(T_1\) and \(T_2\) have the
      same non-zero spectrum with algebraic multiplicity. Consequently,
      \[
        \Tr(T_1^n)=\Tr(T_2^n),
        \qquad
        p_n(T_1)=p_n(T_2)
        \qquad (n\ge 1).
      \]
  \end{enumerate}
\end{proposition}

\begin{proof}
  For \textup{(a)}, \Cref{thm:cyclic-fredholm-realisation} produces
  normal models \(L_{\mathcal D}\) and \(L_{\mathcal D'}\) with the
  prescribed Euler products as Fredholm zeta germs. Taking reciprocals,
  the entire functions
  \[
    z\mapsto \det(I-zL_{\mathcal D}),
    \qquad
    z\mapsto \det(I-zL_{\mathcal D'})
  \]
  agree near \(0\), hence everywhere. Simon's theorem
  \cite[Theorem~3.7]{Simon2005TraceIdeals} identifies their zeros with
  the reciprocals of the non-zero eigenvalues, counted with algebraic
  multiplicity. This gives the equality of eigenvalue multisets and,
  after summing moduli and multiplicities, the two displayed identities.

  For \textup{(b)}, the same argument applied directly to \(T_1\) and
  \(T_2\) yields equality of the non-zero spectra with multiplicity.
  Comparing the logarithmic expansions of \(Z_{T_1}\) and \(Z_{T_2}\)
  then shows that the trace sequences agree, and hence so do the
  M\"obius-inverted coordinates.
\end{proof}

\begin{corollary}[Determinant invariants in the normal class]
  \label{cor:determinant-intrinsic-normal-data}
  Let \(f\) be a holomorphic germ at \(0\) admitting an Euler
  factorization
  \[
    f(z)=\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  satisfying the summability condition of
  \Cref{thm:cyclic-fredholm-realisation}. If \(L\in\sone(H)\) is normal
  and
  \[
    Z_L(z)=f(z)
  \]
  near \(z=0\), then
  \[
    N(L)=\sum_{j\in J}m_jn_j,
    \qquad
    \|L\|_1=\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  For compact normal \(L\), this is equivalent to
  \[
    \dim\overline{\operatorname{ran}(L)}=\sum_{j\in J}m_jn_j.
  \]
  The right-hand sides depend only on the Fredholm zeta germ \(f\), not
  on the chosen Euler factorization.
\end{corollary}

\begin{proof}
  Apply \Cref{thm:cyclic-fredholm-realisation} to one admissible Euler
  factorization of \(f\), and then use
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(a)}.
\end{proof}

Without the normality assumption one still obtains the following
spectral rigidity statement.

\begin{proposition}[Spectral rigidity for arbitrary realizations]
  \label{thm:general-support-optimality}
  Let \(T\in \sone(H)\) satisfy
  \[
    Z_T(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  near \(z=0\), where the Euler data satisfy the summability condition
  of \Cref{thm:cyclic-fredholm-realisation}. Then the non-zero spectrum
  of \(T\), counted with algebraic multiplicity, is
  \[
    \biguplus_{j\in J}\bigl\{\alpha_j\omega:\omega^{n_j}=1\bigr\}^{m_j},
    \qquad
    \alpha_j^{n_j}=\beta_j.
  \]
  In particular,
  \[
    N(T)=\sum_{j\in J}m_jn_j.
  \]
\end{proposition}

\begin{proof}
  Let \(L_{\mathcal D}\) be the block-cyclic model associated with the
  given Euler data. Then
  \(Z_T(z)=Z_{L_{\mathcal D}}(z)\) near \(0\), so
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(b)} shows that
  \(T\) and \(L_{\mathcal D}\) have the same non-zero eigenvalues with
  algebraic multiplicity. The description of the spectrum of
  \(L_{\mathcal D}\) is immediate from the definition of the cyclic
  blocks.
\end{proof}

\begin{theorem}[Trace-norm minimality]
  \label{thm:trace-norm-minimality}
  Let \(T\in \sone(H)\) satisfy
  \[
    Z_T(z)=\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  near \(z=0\), where the Euler data obey the summability condition of
  \Cref{thm:cyclic-fredholm-realisation}. Then
  \[
    \|T\|_1\ge \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  Equality holds if and only if \(T\) is normal. Equivalently, equality
  holds if and only if there exist a Hilbert space \(H_0\) and a unitary
  operator \(U:H\to H_0\oplus H_{\mathcal D}\) such that
  \[
    UTU^{-1}=0_{H_0}\oplus L_{\mathcal D},
  \]
  where \(L_{\mathcal D}\) is the block-cyclic model of
  \Cref{thm:cyclic-fredholm-realisation}.
\end{theorem}

\begin{proof}
  Let \(L_{\mathcal D}\) be the block-cyclic model associated with the
  given Euler data. By
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(b)},
  \(T\) and \(L_{\mathcal D}\) have the same non-zero eigenvalues with
  algebraic multiplicity. Hence
  \[
    \sum_k\abs{\lambda_k(T)}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}
    =\|L_{\mathcal D}\|_1.
  \]
  Simon's eigenvalue-singular-value inequality
  \cite[Theorem~1.15]{Simon2005TraceIdeals} yields
  \[
    \sum_k\abs{\lambda_k(T)}\le \|T\|_1,
  \]
  and therefore
  \[
    \|T\|_1
    \ge \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  Since \(L_{\mathcal D}\) is normal, equality is attained by
  \(L_{\mathcal D}\), and hence also by every enlargement
  \(0_{H_0}\oplus L_{\mathcal D}\).

  Assume now that equality holds. Choose an orthonormal basis in which
  the compact operator \(T\) is upper triangular, with diagonal entries
  \(\lambda_k(T)\) repeated according to algebraic multiplicity; see
  \cite{Ringrose1971}. Let
  \(D\) be the diagonal unitary with diagonal entries
  \[
    d_k=
    \begin{cases}
      \overline{\lambda_k(T)}/\abs{\lambda_k(T)},
        & \lambda_k(T)\ne 0,\\
      1, & \lambda_k(T)=0.
    \end{cases}
  \]
  Then \(DT\) is upper triangular and
  \[
    \Tr(DT)=\sum_k\abs{\lambda_k(T)}=\|T\|_1.
  \]
  Write the polar decomposition \(T=V|T|\). Since \(\|DV\|\le 1\),
  \[
    0
    =\Tr|T|-\Re\Tr(DT)
    =\Tr\!\bigl(|T|^{1/2}(I-\Re(DV))|T|^{1/2}\bigr).
  \]
  The operator inside the trace is positive, hence vanishes. Thus
  \(\Re(DV)=I\) on \(\overline{\operatorname{ran}(|T|)}\). For
  \(h\in\overline{\operatorname{ran}(|T|)}\),
  \[
    \|h-DVh\|^2
    =\|h\|^2+\|DVh\|^2-2\Re\langle DVh,h\rangle
    \le 2\|h\|^2-2\|h\|^2=0,
  \]
  so \(DVh=h\). Therefore
  \[
    DT=DV|T|=|T|
  \]
  is positive. A positive upper-triangular operator is diagonal, hence
  \(DT\), and therefore \(T\), is diagonal in this basis. In
  particular \(T\) is normal. The converse characterization now follows
  from \Cref{thm:cyclic-fredholm-realisation}.
\end{proof}

\begin{theorem}[Jordan structure away from \(0\)]
  \label{thm:nonnormal-jordan-classification}
  Let
  \(
    \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J}
  \)
  satisfy the summability condition of
  \Cref{thm:cyclic-fredholm-realisation}. Let \(L_{\mathcal D}\) be the
  associated block-cyclic model and write
  \[
    \Lambda_{\mathcal D}:=\spec(L_{\mathcal D})\setminus\{0\}.
  \]
  For each \(\lambda\in \Lambda_{\mathcal D}\), let
  \(a_{\mathcal D}(\lambda)\) be the algebraic multiplicity of
  \(\lambda\) in \(L_{\mathcal D}\). For each \(\lambda\in
  \Lambda_{\mathcal D}\), let
  \[
    \pi_\lambda=(r_{\lambda,1},\dots,r_{\lambda,\ell_\lambda})
  \]
  be a finite tuple of positive integers. Then the following are
  equivalent:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item There exists \(T\in \sone(H)\) such that
      \[
        Z_T(z)=\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
      \]
      near \(z=0\), and for every \(\lambda\in \Lambda_{\mathcal D}\)
      the restriction of \(T\) to the generalized eigenspace at
      \(\lambda\) has Jordan block sizes exactly
      \(\pi_\lambda\).
    \item For every \(\lambda\in \Lambda_{\mathcal D}\),
      \[
        a_{\mathcal D}(\lambda)
        =\sum_{q=1}^{\ell_\lambda}r_{\lambda,q}.
      \]
  \end{enumerate}
  If these conditions hold, then
  \[
    \inf\Bigl\{\|T\|_1:T\text{ satisfies \textup{(i)}}\Bigr\}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  This infimum is attained if and only if
  \(r_{\lambda,q}=1\) for every \(\lambda\in \Lambda_{\mathcal D}\) and
  every \(1\le q\le \ell_\lambda\).
\end{theorem}

\begin{proof}
  Assume \textup{(i)}. By
  \Cref{thm:general-support-optimality}, the non-zero spectrum of \(T\),
  counted with algebraic multiplicity, agrees with that of
  \(L_{\mathcal D}\). Hence the algebraic multiplicity of each
  \(\lambda\in \Lambda_{\mathcal D}\) is
  \(a_{\mathcal D}(\lambda)\). On the finite-dimensional generalized
  eigenspace at \(\lambda\), the sum of the Jordan block sizes is
  exactly the algebraic multiplicity. Therefore \textup{(ii)} follows.

  Assume now that \textup{(ii)} holds. Let \(N_r\) denote the nilpotent
  Jordan block of size \(r\), with ones on the superdiagonal and zeros
  elsewhere. Since
  \[
    \sum_{\lambda\in \Lambda_{\mathcal D}}
      a_{\mathcal D}(\lambda)\abs{\lambda}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}
    <\infty,
  \]
  the set
  \[
    \mathcal B
    :=\{(\lambda,q):\lambda\in\Lambda_{\mathcal D},
      \ 1\le q\le \ell_\lambda\}
  \]
  is countable. Fix \(\varepsilon>0\), and choose numbers
  \(\delta_{\lambda,q}>0\) such that
  \[
    \sum_{(\lambda,q)\in\mathcal B}\delta_{\lambda,q}<\varepsilon.
  \]
  For each \((\lambda,q)\in\mathcal B\), continuity of the trace norm on
  the finite-dimensional matrix algebra
  \(M_{r_{\lambda,q}}(\CC)\) yields a number \(t_{\lambda,q}\ge 0\) such
  that
  \[
    t_{\lambda,q}>0 \text{ if } r_{\lambda,q}\ge 2,
    \qquad
    \left\|\lambda I_{r_{\lambda,q}}
      +t_{\lambda,q}N_{r_{\lambda,q}}\right\|_1
    \le r_{\lambda,q}\abs{\lambda}+\delta_{\lambda,q}.
  \]
  Set
  \[
    B_{\lambda,q}
    :=\lambda I_{r_{\lambda,q}}+t_{\lambda,q}N_{r_{\lambda,q}}.
  \]
  If \(r_{\lambda,q}\ge 2\), then \(B_{\lambda,q}\) is similar to the
  Jordan block \(J_{r_{\lambda,q}}(\lambda)=\lambda I_{r_{\lambda,q}}
  +N_{r_{\lambda,q}}\), because
  \[
    D_t^{-1}J_r(\lambda)D_t=\lambda I_r+tN_r,
    \qquad
    D_t:=\operatorname{diag}(1,t,\dots,t^{r-1}),
  \]
  for every \(t>0\). If \(r_{\lambda,q}=1\), then \(B_{\lambda,q}\) is
  the \(1\times 1\) Jordan block \((\lambda)\).

  Define
  \[
    H_\varepsilon
    :=\bigoplus_{(\lambda,q)\in\mathcal B}\CC^{r_{\lambda,q}},
    \qquad
    T_\varepsilon
    :=\bigoplus_{(\lambda,q)\in\mathcal B}B_{\lambda,q}.
  \]
  The estimate above and \textup{(ii)} give
  \begin{align*}
    \|T_\varepsilon\|_1
    &\le \sum_{(\lambda,q)\in\mathcal B}
      \left(r_{\lambda,q}\abs{\lambda}+\delta_{\lambda,q}\right) \\
    &= \sum_{\lambda\in \Lambda_{\mathcal D}}
      a_{\mathcal D}(\lambda)\abs{\lambda}
      +\sum_{(\lambda,q)\in\mathcal B}\delta_{\lambda,q} \\
    &< \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}+\varepsilon.
  \end{align*}
  Hence \(T_\varepsilon\in \sone(H_\varepsilon)\). By construction, for
  each \(\lambda\in \Lambda_{\mathcal D}\) the generalized eigenspace of
  \(T_\varepsilon\) at \(\lambda\) is the direct sum of the blocks
  \(B_{\lambda,q}\), so the Jordan block sizes are exactly
  \(\pi_\lambda\).

  Let \(\mathcal B_N\subset \mathcal B\) be an increasing exhaustion by
  finite sets, and set
  \[
    T_\varepsilon^{(N)}
    :=\bigoplus_{(\lambda,q)\in\mathcal B_N}B_{\lambda,q}.
  \]
  Then
  \(
    \|T_\varepsilon-T_\varepsilon^{(N)}\|_1\to 0
  \).
  Since each \(B_{\lambda,q}\) is upper triangular with diagonal entry
  \(\lambda\), one has
  \[
    \det(I-zB_{\lambda,q})=(1-\lambda z)^{r_{\lambda,q}}.
  \]
  Therefore
  \[
    \det(I-zT_\varepsilon^{(N)})
    =\prod_{(\lambda,q)\in\mathcal B_N}(1-\lambda z)^{r_{\lambda,q}}.
  \]
  By trace-norm continuity of the Fredholm determinant
  \cite[Theorem~3.4]{Simon2005TraceIdeals},
  \[
    \det(I-zT_\varepsilon)
    =\prod_{\lambda\in\Lambda_{\mathcal D}}
      (1-\lambda z)^{a_{\mathcal D}(\lambda)}
  \]
  as a holomorphic germ at \(z=0\). The same product is
  \(\det(I-zL_{\mathcal D})\), because \(L_{\mathcal D}\) has the
  non-zero eigenvalues \(\lambda\in \Lambda_{\mathcal D}\) with
  multiplicities \(a_{\mathcal D}(\lambda)\). Since
  \(Z_{L_{\mathcal D}}(z)=\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}\) by
  \Cref{thm:cyclic-fredholm-realisation},
  \[
    Z_{T_\varepsilon}(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  near \(z=0\). This proves \textup{(i)}.

  The construction above gives, for every \(\varepsilon>0\), an operator
  \(T_\varepsilon\) satisfying \textup{(i)} and
  \[
    \|T_\varepsilon\|_1
    <\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}+\varepsilon.
  \]
  On the other hand, every operator satisfying \textup{(i)} obeys the
  lower bound from \Cref{thm:trace-norm-minimality}. Consequently,
  \[
    \inf\Bigl\{\|T\|_1:T\text{ satisfies \textup{(i)}}\Bigr\}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]

  If \(r_{\lambda,q}=1\) for every \(\lambda\) and \(q\), then
  \(L_{\mathcal D}\) itself satisfies \textup{(i)} and attains the
  infimum by \Cref{thm:cyclic-fredholm-realisation}. Conversely, assume
  that the infimum is attained by some operator \(T\) satisfying
  \textup{(i)}. Then \(\|T\|_1\) equals the lower bound from
  \Cref{thm:trace-norm-minimality}, so \(T\) is normal. For each
  \(\lambda\in \Lambda_{\mathcal D}\), the restriction of \(T\) to the
  generalized eigenspace at \(\lambda\) is a finite-dimensional normal
  operator with the single eigenvalue \(\lambda\), hence equals
  \(\lambda I\). Therefore every Jordan block has size \(1\).
\end{proof}

\begin{remark}[Freedom at the origin]
  \label{rem:zero-spectrum-freedom}
  If \(Q\in \sone(K)\) is quasinilpotent, then \(\det(I-zQ)\) has no
  zeros and equals \(1\) at \(z=0\); hence \(\det(I-zQ)\equiv 1\). Thus
  \(T\oplus Q\) has the same Fredholm zeta function as \(T\).
  \Cref{thm:nonnormal-jordan-classification} therefore identifies the
  complete determinant-constrained Jordan geometry away from \(0\),
  while the quasinilpotent sector remains completely free.
\end{remark}

\section{Regularized determinants in Schatten classes}
\label{sec:schatten}

Fix an integer \(p\ge 2\). For a compact operator \(T\), let
\(s_k(T)\) denote its singular values. The Schatten class
\(\mathcal S_p(H)\) consists of those compact operators for which
\[
  \|T\|_p^p:=\sum_k s_k(T)^p<\infty.
\]
For \(T\in \mathcal S_p(H)\), set
\[
  Z_{p,T}(z):=\det\nolimits_p(I-zT)^{-1}.
\]
By the regularized determinant theory
\cite[Chapter~9]{Simon2005TraceIdeals}, \(Z_{p,T}\) is entire, its
reciprocal vanishes precisely at the reciprocals of the non-zero
eigenvalues of \(T\), counted with algebraic multiplicity, and
\begin{equation}\label{eq:regularized-log-series}
  Z_{p,T}(z)
  =\exp\!\left(\sum_{q\ge p}\frac{\Tr(T^q)}{q}z^q\right)
\end{equation}
for \(\abs{z}<\|T\|^{-1}\).

For \(n\ge 1\) and \(\beta\in \CC\setminus\{0\}\), define
\begin{equation}\label{eq:regularized-euler-factor}
  \mathcal E_{p,n,\beta}(z)
  :=(1-\beta z^n)^{-1}
    \exp\!\left(
      -\sum_{\ell=1}^{\lfloor (p-1)/n\rfloor}
        \frac{\beta^\ell z^{n\ell}}{\ell}\right).
\end{equation}

\begin{theorem}[Regularized realization theorem]
  \label{thm:schatten-regularized}
  Let \(\mathcal D\) be countable Euler data,
  \[
    \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J},
  \]
  with
  \[
    \begin{aligned}
      n_j,m_j&\in\NN,\\
      \beta_j&\in\CC\setminus\{0\},
    \end{aligned}
  \]
  and assume that
  \[
    \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}<\infty.
  \]
  Choose \(\alpha_j\in\CC\) such that \(\alpha_j^{n_j}=\beta_j\), and
  set
  \[
    H_{\mathcal D}
    :=\bigoplus_{j\in J}\bigl(\CC^{n_j}\otimes\CC^{m_j}\bigr),
    \qquad
    L_{\mathcal D}
    :=\bigoplus_{j\in J}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
  \]
  Then \(L_{\mathcal D}\in \mathcal S_p(H_{\mathcal D})\), and
  \[
    Z_{p,L_{\mathcal D}}(z)
    =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  locally uniformly on \(\CC\).

  Moreover, if \(L\in \mathcal S_p(H)\) is normal, then the following
  are equivalent:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item
      \[
        Z_{p,L}(z)
        =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
      \]
      in a neighborhood of \(z=0\).
    \item There exist a Hilbert space \(H_0\) and a unitary operator
      \(U:H\to H_0\oplus H_{\mathcal D}\) such that
      \[
        ULU^{-1}=0_{H_0}\oplus L_{\mathcal D}.
      \]
  \end{enumerate}
  In particular, every normal realization satisfies
  \[
    \|L\|_p^p=\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]

  If \(T\in \mathcal S_p(H)\) satisfies
  \[
    Z_{p,T}(z)
    =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  near \(z=0\), then
  \[
    \|T\|_p^p\ge \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
  Equality holds if and only if \(T\) is normal. Equivalently,
  equality holds if and only if there exist a Hilbert space \(H_0\) and
  a unitary operator \(U:H\to H_0\oplus H_{\mathcal D}\) such that
  \[
    UTU^{-1}=0_{H_0}\oplus L_{\mathcal D}.
  \]
\end{theorem}

\begin{proof}
  Each block \(C(n_j,\alpha_j)\otimes I_{m_j}\) has singular value
  \(\abs{\alpha_j}\) with multiplicity \(m_jn_j\). Therefore
  \[
    \|L_{\mathcal D}\|_p^p
    =\sum_{j\in J}m_jn_j\abs{\alpha_j}^p
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}
    <\infty,
  \]
  so \(L_{\mathcal D}\in \mathcal S_p(H_{\mathcal D})\). Since
  \(m_jn_j\ge 1\) for every \(j\), the same summability implies
  \(
    \sup_j\abs{\alpha_j}<\infty
  \).

  For each \(j\), set
  \(
    B_j:=C(n_j,\alpha_j)\otimes I_{m_j}
  \).
  If \(q\ge 1\), then
  \[
    \Tr(B_j^q)
    =
    \begin{cases}
      m_jn_j\,\beta_j^{q/n_j}, & n_j\mid q,\\
      0, & n_j\nmid q,
    \end{cases}
  \]
  because the eigenvalues of \(C(n_j,\alpha_j)\) are
  \(\alpha_j\omega\), \(\omega^{n_j}=1\). Hence, for
  \(\abs{z}<\abs{\alpha_j}^{-1}\),
  \begin{align*}
    \log Z_{p,B_j}(z)
    &=\sum_{q\ge p}\frac{\Tr(B_j^q)}{q}z^q \\
    &=m_j\sum_{\ell\ge \lceil p/n_j\rceil}
      \frac{\beta_j^\ell z^{n_j\ell}}{\ell}
     =\log\!\bigl(\mathcal E_{p,n_j,\beta_j}(z)^{m_j}\bigr),
  \end{align*}
  where \eqref{eq:regularized-euler-factor} is used in the last step.
  Therefore
  \[
    Z_{p,B_j}(z)=\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  as entire functions.

  Let
  \[
    L_{\mathcal D}^{(N)}
    :=\bigoplus_{j\le N}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
  \]
  Then
  \[
    \|L_{\mathcal D}-L_{\mathcal D}^{(N)}\|_p^p
    =\sum_{j>N}m_jn_j\abs{\beta_j}^{p/n_j}\to 0.
  \]
  By the continuity of the regularized determinant on \(\mathcal S_p\)
  \cite[Chapter~9]{Simon2005TraceIdeals},
  \[
    Z_{p,L_{\mathcal D}^{(N)}}\longrightarrow Z_{p,L_{\mathcal D}}
  \]
  locally uniformly on \(\CC\). Since
  \[
    Z_{p,L_{\mathcal D}^{(N)}}(z)
    =\prod_{j\le N}\mathcal E_{p,n_j,\beta_j}(z)^{m_j},
  \]
  passage to the limit yields
  \[
    Z_{p,L_{\mathcal D}}(z)
    =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  locally uniformly on \(\CC\).

  The implication \textup{(ii)}\(\Rightarrow\)\textup{(i)} is immediate.
  Suppose now that \textup{(i)} holds. The first part of the theorem
  provides an operator \(L_{\mathcal D}\in\mathcal S_p\) with the same
  regularized determinant germ, so
  \[
    \det\nolimits_p(I-zL)=\det\nolimits_p(I-zL_{\mathcal D})
  \]
  as entire functions. By the regularized determinant theory cited
  above, the zeros of \(\det_p(I-zT)\) are precisely the reciprocals of
  the non-zero eigenvalues of \(T\), counted with algebraic
  multiplicity. Therefore \(L\) and \(L_{\mathcal D}\) have the same
  non-zero spectrum with multiplicity.

  Since \(L\) is compact and normal, the spectral theorem gives an
  orthogonal decomposition
  \[
    H=\ker L\oplus\bigoplus_{\lambda\in\spec(L)\setminus\{0\}}E_\lambda,
    \qquad
    L|_{E_\lambda}=\lambda I.
  \]
  For each non-zero eigenvalue \(\lambda\), let
  \[
    J_\lambda
    :=\bigl\{j\in J:\lambda=\alpha_j\omega
      \text{ for some }\omega^{n_j}=1\bigr\}.
  \]
  Since the multiplicity of \(\lambda\) is finite,
  \[
    \dim E_\lambda=\sum_{j\in J_\lambda}m_j.
  \]
  Choose an orthogonal decomposition
  \[
    E_\lambda=\bigoplus_{j\in J_\lambda}E_{\lambda,j},
    \qquad
    \dim E_{\lambda,j}=m_j.
  \]
  For each \(j\), set
  \[
    H_j:=\bigoplus_{\omega^{n_j}=1}E_{\alpha_j\omega,j}.
  \]
  Then \(H_j\) is reducing for \(L\), the subspaces \(H_j\) are
  pairwise orthogonal, and \(L|_{H_j}\) is unitarily equivalent to
  \[
    \bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}
    \cong C(n_j,\alpha_j)\otimes I_{m_j}.
  \]
  Consequently \(L\) is unitarily equivalent to
  \(0_{\ker L}\oplus L_{\mathcal D}\), which proves \textup{(ii)}. The
  formula for \(\|L\|_p^p\) is immediate from the first part of the
  proof.

  Let \(T\in \mathcal S_p(H)\) satisfy
  \[
    Z_{p,T}(z)
    =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  near \(z=0\). Then \(T\) and \(L_{\mathcal D}\) have the same
  non-zero eigenvalues with algebraic multiplicity, so
  \[
    \sum_k\abs{\lambda_k(T)}^p
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]

  Choose an orthonormal basis in which the compact operator \(T\) is
  upper triangular, with diagonal entries \(\lambda_k(T)\) repeated
  according to algebraic multiplicity \cite{Ringrose1971}. Let \(D\) be
  the diagonal operator with diagonal entries \(\lambda_k(T)\). Then
  \[
    \|D\|_p^p=\sum_k\abs{\lambda_k(T)}^p.
  \]
  By \Cref{thm:schatten-quantitative-stability},
  \[
    \|T\|_p^p
    \ge \|D\|_p^p+\|T-D\|_p^p
    \ge \|D\|_p^p,
  \]
  and therefore
  \[
    \|T\|_p^p
    \ge \sum_k\abs{\lambda_k(T)}^p
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
  If equality holds, then \(\|T-D\|_p=0\), so \(T=D\) is normal. The
  converse characterization follows from the equivalence
  \textup{(i)}\(\Leftrightarrow\)\textup{(ii)} already proved.
\end{proof}

\begin{remark}[Scope of the regularized theorem]
  \label{rem:normal-scope-schatten}
  The summability condition in \Cref{thm:schatten-regularized}
  characterizes normal realizations of the regularized Euler product.
  As in trace class, the later results for arbitrary operators in
  \(\mathcal S_p\) are determinant-constrained classification and
  stability statements relative to the canonical normal model
  \(L_{\mathcal D}\), not new existence theorems beyond the normal
  category.
\end{remark}

The equality case above admits a quantitative strengthening that
controls the full Schatten distance to the normal orbit.

\begin{theorem}[Quantitative Schatten stability]
  \label{thm:schatten-quantitative-stability}
  Fix an integer \(p\ge 2\), let \(T\in \mathcal S_p(H)\), and choose
  an orthonormal basis \((e_k)_{k\ge 1}\) in which \(T\) is upper
  triangular, with diagonal entries \(\lambda_k(T)\) repeated according
  to algebraic multiplicity. Let \(D\) be the diagonal operator
  determined by
  \[
    De_k=\lambda_k(T)e_k
    \qquad (k\ge 1).
  \]
  Then \(D\in \mathcal S_p(H)\), \(T-D\in \mathcal S_p(H)\), and
  \[
    \|T-D\|_p^p
    \le \|T\|_p^p-\|D\|_p^p
    =\|T\|_p^p-\sum_k\abs{\lambda_k(T)}^p.
  \]
  If \(p=2\), then
  \[
    \|T-D\|_2^2
    =\|T\|_2^2-\|D\|_2^2
    =\|T\|_2^2-\sum_k\abs{\lambda_k(T)}^2.
  \]
\end{theorem}

\begin{proof}
  Simon's eigenvalue-singular-value inequality
  \cite[Theorem~1.15]{Simon2005TraceIdeals} gives
  \[
    \sum_k\abs{\lambda_k(T)}^p\le \|T\|_p^p,
  \]
  so \(D\in \mathcal S_p(H)\). Set
  \[
    K:=T-D\in \mathcal S_p(H).
  \]
  The operator \(K\) is strictly upper triangular in the chosen basis.

  For \(n\ge 1\), let \(P_n\) be the orthogonal projection onto
  \(\CC e_n\), and let \(Q_n\) be the orthogonal projection onto
  \(\overline{\operatorname{span}}\{e_k:k>n\}\). Define
  \[
    R_n:=P_nKQ_n.
  \]
  Since \(K\) is strictly upper triangular,
  \[
    K=\sum_{n\ge 1}R_n
  \]
  in the strong operator topology. Moreover,
  \[
    R_nR_m^*=0
    \qquad (n\ne m),
  \]
  because the ranges of the operators \(R_n\) are pairwise orthogonal.
  Consequently,
  \[
    KK^*=\sum_{n\ge 1}R_nR_n^*
  \]
  with orthogonal positive summands, and therefore
  \[
    (KK^*)^{p/2}=\sum_{n\ge 1}(R_nR_n^*)^{p/2}.
  \]
  Taking traces yields
  \begin{equation}\label{eq:schatten-row-sum}
    \|K\|_p^p
    =\sum_{n\ge 1}\|R_n\|_p^p.
  \end{equation}

  For \(n\ge 1\), let \(W_n\) be the diagonal unitary determined by
  \[
    W_ne_n=-e_n,
    \qquad
    W_ne_m=e_m
    \qquad (m\ne n).
  \]
  Set
  \[
    T_0:=T,
    \qquad
    T_n:=D+Q_nKQ_n
    \qquad (n\ge 1).
  \]
  Since \(K\) is strictly upper triangular,
  \[
    T_{n-1}=D+R_n+Q_nKQ_n.
  \]
  The unitary \(W_n\) fixes \(D\) and \(Q_nKQ_n\), while
  \(W_nR_nW_n=-R_n\). Hence
  \[
    T_n=\frac{T_{n-1}+W_nT_{n-1}W_n}{2},
    \qquad
    R_n=\frac{T_{n-1}-W_nT_{n-1}W_n}{2}.
  \]
  The Clarkson--McCarthy inequality for Schatten classes
  \cite{BallCarlenLieb1994}, applied to
  \(A=T_{n-1}\) and \(B=W_nT_{n-1}W_n\), therefore gives
  \[
    \|T_n\|_p^p+\|R_n\|_p^p\le \|T_{n-1}\|_p^p
    \qquad (n\ge 1).
  \]
  Summation over \(1\le n\le N\) yields
  \begin{equation}\label{eq:schatten-stability-telescope}
    \sum_{n=1}^N\|R_n\|_p^p
    \le \|T\|_p^p-\|T_N\|_p^p.
  \end{equation}

  Since \(K\in \mathcal S_p(H)\),
  \[
    T_N-D=Q_NKQ_N\longrightarrow 0
  \]
  in \(\mathcal S_p(H)\). Thus \(T_N\to D\) in \(\mathcal S_p(H)\), and
  passing to the limit in \eqref{eq:schatten-stability-telescope},
  followed by \eqref{eq:schatten-row-sum}, gives
  \[
    \|T-D\|_p^p
    =\sum_{n\ge 1}\|R_n\|_p^p
    \le \|T\|_p^p-\|D\|_p^p.
  \]

  If \(p=2\), then the diagonal operator \(D\) and the strictly upper
  triangular operator \(K\) are orthogonal in the Hilbert--Schmidt inner
  product. Therefore
  \[
    \|T\|_2^2=\|D\|_2^2+\|K\|_2^2,
  \]
  which is the stated identity.
\end{proof}

\begin{corollary}[Coercive rigidity in a determinant class]
  \label{cor:schatten-coercive-rigidity}
  Fix an integer \(p\ge 2\) and Euler data
  \(
    \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J}
  \)
  satisfying
  \[
    \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}<\infty.
  \]
  For a Hilbert space \(H\), define
  \[
    \mathfrak N_{\mathcal D}(H)
    :=\left\{
    \begin{aligned}
      &N\in \mathcal S_p(H): N\text{ is normal and } \\
      &Z_{p,N}(z)
      =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
      \text{ near }z=0
    \end{aligned}
    \right\}.
  \]
  If \(T\in \mathcal S_p(H)\) satisfies
  \[
    Z_{p,T}(z)
    =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  near \(z=0\), then \(\mathfrak N_{\mathcal D}(H)\neq\varnothing\) and
  \[
    \inf_{N\in \mathfrak N_{\mathcal D}(H)}\|T-N\|_p^p
    \le \|T\|_p^p-\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
  Consequently, if \((T_r)_{r\ge 1}\subset \mathcal S_p(H)\) all satisfy
  the same determinant identity and
  \[
    \|T_r\|_p^p
    \longrightarrow \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j},
  \]
  then there exist \(N_r\in \mathfrak N_{\mathcal D}(H)\) such that
  \[
    \|T_r-N_r\|_p\to 0.
  \]
\end{corollary}

\begin{proof}
  Let \(L_{\mathcal D}\) be the block-cyclic model from
  \Cref{thm:schatten-regularized}. Choose an orthonormal basis in which
  \(T\) is upper triangular, with diagonal entries \(\lambda_k(T)\)
  repeated according to algebraic multiplicity, and let \(D\) be the
  diagonal operator with those entries. By
  \Cref{thm:schatten-regularized}, \(T\) and \(L_{\mathcal D}\) have the
  same non-zero spectrum with multiplicity. Since \(D\) is normal and
  has the same non-zero spectral data, the normal-classification
  argument in the proof of \Cref{thm:schatten-regularized} shows that
  \(D\) is unitarily equivalent to \(0\oplus L_{\mathcal D}\). Hence
  \(D\in \mathfrak N_{\mathcal D}(H)\), and
  \[
    \|D\|_p^p=\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
  The estimate from \Cref{thm:schatten-quantitative-stability} now gives
  \[
    \inf_{N\in \mathfrak N_{\mathcal D}(H)}\|T-N\|_p^p
    \le \|T-D\|_p^p
    \le \|T\|_p^p-\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
  Applying this estimate to each \(T_r\) gives the final assertion.
\end{proof}

\begin{corollary}[Jensen formula for the spectral \(p\)-mass]
  \label{cor:schatten-mass}
  Let \(p\ge 2\) and \(T\in \mathcal S_p(H)\). Then
  \[
    \sum_{\lambda\in\spec(T)\setminus\{0\}}m_a(\lambda)\abs{\lambda}^p
    =p^2\int_0^\infty \frac{1}{r^{p+1}}
      \left(
        \frac{1}{2\pi}\int_0^{2\pi}
          \log\abs{\det\nolimits_p(I-re^{it}T)}\,\dd t
      \right)\dd r.
  \]
  In the setting of \Cref{thm:schatten-regularized},
  \[
    \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}
    =p^2\int_0^\infty \frac{1}{r^{p+1}}
      \left(
        \frac{1}{2\pi}\int_0^{2\pi}
          \log\abs{\det\nolimits_p(I-re^{it}L_{\mathcal D})}\,\dd t
      \right)\dd r.
  \]
\end{corollary}

\begin{proof}
  Set \(D_{p,T}(z):=\det_p(I-zT)\). Since \(D_{p,T}(0)=1\), Jensen's
  formula yields
  \[
    \frac{1}{2\pi}\int_0^{2\pi}\log\abs{D_{p,T}(re^{it})}\,\dd t
    =\sum_{\lambda\in\spec(T)\setminus\{0\}}
      m_a(\lambda)\log^+(r\abs{\lambda}),
  \]
  where \(\log^+x:=\max\{\log x,0\}\). The summands are non-negative, so
  Tonelli's theorem gives
  \begin{align*}
    &p^2\int_0^\infty \frac{1}{r^{p+1}}
      \left(
        \frac{1}{2\pi}\int_0^{2\pi}
          \log\abs{D_{p,T}(re^{it})}\,\dd t
      \right)\dd r \\
    &\qquad
      =\sum_{\lambda\in\spec(T)\setminus\{0\}}m_a(\lambda)
        p^2\int_0^\infty r^{-p-1}\log^+(r\abs{\lambda})\,\dd r.
  \end{align*}
  For \(a>0\),
  \[
    p^2\int_0^\infty r^{-p-1}\log^+(ra)\,\dd r
    =p^2a^p\int_1^\infty u^{-p-1}\log u\,\dd u
    =a^p.
  \]
  This proves the first identity.

  Apply the first identity to \(T=L_{\mathcal D}\) and use
  \Cref{thm:schatten-regularized}, which gives
  \[
    \sum_{\lambda\in\spec(L_{\mathcal D})\setminus\{0\}}
      m_a(\lambda)\abs{\lambda}^p
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
\end{proof}

\begin{theorem}[Jordan structure for regularized determinants]
  \label{thm:schatten-jordan}
  Fix an integer \(p\ge 2\), and let
  \(
    \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J}
  \)
  satisfy
  \[
    \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}<\infty.
  \]
  Let \(L_{\mathcal D}\) be the associated block-cyclic model, set
  \[
    \Lambda_{\mathcal D}:=\spec(L_{\mathcal D})\setminus\{0\},
  \]
  and let \(a_{\mathcal D}(\lambda)\) denote the algebraic multiplicity
  of \(\lambda\in\Lambda_{\mathcal D}\). For each
  \(\lambda\in\Lambda_{\mathcal D}\), let
  \[
    \pi_\lambda=(r_{\lambda,1},\dots,r_{\lambda,\ell_\lambda})
  \]
  be a finite tuple of positive integers. Then the following are
  equivalent:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item There exists \(T\in \mathcal S_p(H)\) such that
      \[
        Z_{p,T}(z)
        =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
      \]
      near \(z=0\), and for every \(\lambda\in\Lambda_{\mathcal D}\)
      the restriction of \(T\) to the generalized eigenspace at
      \(\lambda\) has Jordan block sizes exactly \(\pi_\lambda\).
    \item For every \(\lambda\in\Lambda_{\mathcal D}\),
      \[
        a_{\mathcal D}(\lambda)
        =\sum_{q=1}^{\ell_\lambda}r_{\lambda,q}.
      \]
  \end{enumerate}
  If these conditions hold, then
  \[
    \inf\Bigl\{\|T\|_p^p:T\text{ satisfies \textup{(i)}}\Bigr\}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]
  This infimum is attained if and only if
  \(r_{\lambda,q}=1\) for every \(\lambda\in\Lambda_{\mathcal D}\) and
  every \(1\le q\le \ell_\lambda\).
\end{theorem}

\begin{proof}
  Assume \textup{(i)}. By \Cref{thm:schatten-regularized}, the
  non-zero spectrum of \(T\), counted with algebraic multiplicity,
  agrees with that of \(L_{\mathcal D}\). Hence the algebraic
  multiplicity of each \(\lambda\in\Lambda_{\mathcal D}\) is
  \(a_{\mathcal D}(\lambda)\). On the finite-dimensional generalized
  eigenspace at \(\lambda\), the sum of the Jordan block sizes is
  exactly the algebraic multiplicity. Therefore \textup{(ii)} holds.

  Assume now that \textup{(ii)} holds. Let \(N_r\) denote the nilpotent
  Jordan block of size \(r\), with ones on the superdiagonal and zeros
  elsewhere. Since
  \[
    \sum_{\lambda\in\Lambda_{\mathcal D}}
      a_{\mathcal D}(\lambda)\abs{\lambda}^p
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}
    <\infty,
  \]
  the set
  \[
    \mathcal B
    :=\{(\lambda,q):\lambda\in\Lambda_{\mathcal D},
      \ 1\le q\le \ell_\lambda\}
  \]
  is countable. Fix \(\varepsilon>0\), and choose numbers
  \(\delta_{\lambda,q}>0\) such that
  \[
    \sum_{(\lambda,q)\in\mathcal B}\delta_{\lambda,q}<\varepsilon.
  \]
  For each \((\lambda,q)\in\mathcal B\), continuity of the Schatten
  \(p\)-norm on \(M_{r_{\lambda,q}}(\CC)\) yields a number
  \(t_{\lambda,q}\ge 0\) such that \(t_{\lambda,q}>0\) whenever
  \(r_{\lambda,q}\ge 2\), and
  \[
    \begin{aligned}
      \left\|\lambda I_{r_{\lambda,q}}
        +t_{\lambda,q}N_{r_{\lambda,q}}\right\|_p^p
      &\le r_{\lambda,q}\abs{\lambda}^p \\
      &\quad +\delta_{\lambda,q}.
    \end{aligned}
  \]
  Set
  \[
    B_{\lambda,q}
    :=\lambda I_{r_{\lambda,q}}+t_{\lambda,q}N_{r_{\lambda,q}}.
  \]
  If \(r_{\lambda,q}\ge 2\), then \(B_{\lambda,q}\) is similar to the
  Jordan block \(J_{r_{\lambda,q}}(\lambda)=\lambda I_{r_{\lambda,q}}
  +N_{r_{\lambda,q}}\), because
  \[
    D_t^{-1}J_r(\lambda)D_t=\lambda I_r+tN_r,
    \qquad
    D_t:=\operatorname{diag}(1,t,\dots,t^{r-1}),
  \]
  for every \(t>0\). If \(r_{\lambda,q}=1\), then \(B_{\lambda,q}\) is
  the \(1\times 1\) Jordan block \((\lambda)\).

  Define
  \[
    H_\varepsilon
    :=\bigoplus_{(\lambda,q)\in\mathcal B}\CC^{r_{\lambda,q}},
    \qquad
    T_\varepsilon
    :=\bigoplus_{(\lambda,q)\in\mathcal B}B_{\lambda,q}.
  \]
  The estimate above and \textup{(ii)} give
  \begin{align*}
    \|T_\varepsilon\|_p^p
    &\le \sum_{(\lambda,q)\in\mathcal B}
      \left(r_{\lambda,q}\abs{\lambda}^p+\delta_{\lambda,q}\right) \\
    &= \sum_{\lambda\in \Lambda_{\mathcal D}}
      a_{\mathcal D}(\lambda)\abs{\lambda}^p
      +\sum_{(\lambda,q)\in\mathcal B}\delta_{\lambda,q} \\
    &< \sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}+\varepsilon.
  \end{align*}
  Hence \(T_\varepsilon\in \mathcal S_p(H_\varepsilon)\). By
  construction, for each \(\lambda\in \Lambda_{\mathcal D}\) the
  generalized eigenspace of \(T_\varepsilon\) at \(\lambda\) is the
  direct sum of the blocks \(B_{\lambda,q}\), so the Jordan block sizes
  are exactly \(\pi_\lambda\).

  For every \(m\ge p\),
  \[
    \Tr(B_{\lambda,q}^m)=r_{\lambda,q}\lambda^m,
  \]
  because \(B_{\lambda,q}\) is upper triangular with constant diagonal
  \(\lambda\). Since
  \[
    \sup_{\lambda\in\Lambda_{\mathcal D}}\abs{\lambda}<\infty
    \qquad\text{and}\qquad
    \sum_{\lambda\in\Lambda_{\mathcal D}}
      a_{\mathcal D}(\lambda)\abs{\lambda}^p<\infty,
  \]
  the series
  \(
    \sum_{\lambda\in\Lambda_{\mathcal D}}a_{\mathcal D}(\lambda)\lambda^m
  \)
  converges absolutely for every \(m\ge p\). Therefore
  \[
    \Tr(T_\varepsilon^m)
    =\sum_{(\lambda,q)\in\mathcal B}r_{\lambda,q}\lambda^m
    =\sum_{\lambda\in\Lambda_{\mathcal D}}
      a_{\mathcal D}(\lambda)\lambda^m
    =\Tr(L_{\mathcal D}^m)
    \qquad (m\ge p).
  \]
  The local expansion \eqref{eq:regularized-log-series} now gives
  \[
    Z_{p,T_\varepsilon}(z)
    =Z_{p,L_{\mathcal D}}(z)
  \]
  near \(z=0\). The product formula from
  \Cref{thm:schatten-regularized} therefore yields
  \[
    Z_{p,T_\varepsilon}(z)
    =\prod_{j\in J}\mathcal E_{p,n_j,\beta_j}(z)^{m_j}
  \]
  near \(z=0\). This proves \textup{(i)}.

  The construction above gives, for every \(\varepsilon>0\), an operator
  \(T_\varepsilon\) satisfying \textup{(i)} and
  \[
    \|T_\varepsilon\|_p^p
    <\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}+\varepsilon.
  \]
  On the other hand, every operator satisfying \textup{(i)} obeys the
  lower bound from \Cref{thm:schatten-regularized}. Consequently,
  \[
    \inf\Bigl\{\|T\|_p^p:T\text{ satisfies \textup{(i)}}\Bigr\}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{p/n_j}.
  \]

  If \(r_{\lambda,q}=1\) for every \(\lambda\) and \(q\), then
  \(L_{\mathcal D}\) itself satisfies \textup{(i)} and attains the
  infimum by \Cref{thm:schatten-regularized}. Conversely, assume that
  the infimum is attained by some operator \(T\) satisfying
  \textup{(i)}. Then \(\|T\|_p^p\) equals the lower bound from
  \Cref{thm:schatten-regularized}, so \(T\) is normal. For each
  \(\lambda\in \Lambda_{\mathcal D}\), the restriction of \(T\) to the
  generalized eigenspace at \(\lambda\) is a finite-dimensional normal
  operator with the single eigenvalue \(\lambda\), hence equals
  \(\lambda I\). Therefore every Jordan block has size \(1\).
\end{proof}
