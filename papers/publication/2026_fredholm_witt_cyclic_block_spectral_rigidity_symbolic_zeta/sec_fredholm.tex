\section{Cyclic-block normal realizations}\label{sec:fredholm}

For a trace-class operator \(T\) we write
\[
  N(T):=\sum_{\lambda\in\spec(T)\setminus\{0\}}m_a(\lambda)
  \in \NN\cup\{\infty\},
\]
where \(m_a(\lambda)\) denotes the algebraic multiplicity of the
eigenvalue \(\lambda\). If \(L\) is compact and normal, then
\[
  N(L)
  =\operatorname{rank}\mathbf 1_{(0,\infty)}(|L|)
  =\dim\overline{\operatorname{ran}(L)}.
\]
Thus \(N(L)\) is the rank of the support projection rather than the
dimension of the possibly non-closed range itself.

For \(n\ge 1\), let \(\Pi_n\) be the cyclic permutation matrix on
\(\CC^n\), and for \(\alpha\in\CC\) define
\[
  C(n,\alpha):=\alpha\Pi_n.
\]
Since the eigenvalues of \(\Pi_n\) are the \(n\)th roots of unity, one
has
\begin{equation}\label{eq:cycle-block-det}
  \det(I-zC(n,\alpha))=1-\alpha^n z^n.
\end{equation}
The next theorem shows that these cyclic blocks are exactly the normal
trace-class building blocks compatible with a prescribed Euler product.

\begin{theorem}[Cyclic-block characterization]
  \label{thm:cyclic-fredholm-realisation}
  Let
  \(
    \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J}
  \)
  be a countable family with
  \[
    n_j,m_j\in\NN,
    \qquad
    \beta_j\in\CC\setminus\{0\},
  \]
  and assume that
  \[
    \sum_{j\in J}m_j n_j\abs{\beta_j}^{1/n_j}<\infty.
  \]
  Choose \(\alpha_j\in\CC\) such that \(\alpha_j^{n_j}=\beta_j\), and
  set
  \[
    H_{\mathcal D}
    :=\bigoplus_{j\in J}\bigl(\CC^{n_j}\otimes\CC^{m_j}\bigr),
    \qquad
    L_{\mathcal D}
    :=\bigoplus_{j\in J}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
  \]
  Then \(L_{\mathcal D}\in \sone(H_{\mathcal D})\), and
  \[
    Z_{L_{\mathcal D}}(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  as an identity of holomorphic germs at \(z=0\). In particular, if
  \[
    r_{\mathcal D}:=\sup_{j\in J}\abs{\beta_j}^{1/n_j},
  \]
  then \(r_{\mathcal D}<\infty\) and the identity holds for
  \(\abs{z}<r_{\mathcal D}^{-1}\).

  Moreover, if \(L\in \sone(H)\) is normal, then the following are
  equivalent:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item
      \[
        Z_L(z)
        =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
      \]
      in a neighborhood of \(z=0\).
    \item There exist a Hilbert space \(H_0\) and a unitary operator
      \(U:H\to H_0\oplus H_{\mathcal D}\) such that
      \[
        ULU^{-1}=0_{H_0}\oplus L_{\mathcal D}.
      \]
  \end{enumerate}
  In particular, every normal realization satisfies
  \[
    N(L)=\sum_{j\in J}m_jn_j,
    \qquad
    \|L\|_1=\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
\end{theorem}

\begin{proof}
  Each block \(C(n_j,\alpha_j)\otimes I_{m_j}\) has singular value
  \(\abs{\alpha_j}\) with multiplicity \(m_jn_j\). Therefore
  \[
    \|L_{\mathcal D}\|_1
    =\sum_{j\in J}m_jn_j\abs{\alpha_j}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}
    <\infty,
  \]
  so \(L_{\mathcal D}\) is trace class. Since \(m_jn_j\ge 1\) for every
  \(j\), the same summability implies
  \(
    r_{\mathcal D}<\infty
  \).

  For each \(j\), \eqref{eq:cycle-block-det} gives
  \[
    \det\!\bigl(I-z(C(n_j,\alpha_j)\otimes I_{m_j})\bigr)
    =\det(I-zC(n_j,\alpha_j))^{m_j}
    =(1-\beta_j z^{n_j})^{m_j}.
  \]
  Let
  \[
    L_{\mathcal D}^{(N)}
    :=\bigoplus_{j\le N}\bigl(C(n_j,\alpha_j)\otimes I_{m_j}\bigr).
  \]
  Then
  \[
    \|L_{\mathcal D}-L_{\mathcal D}^{(N)}\|_1
    =\sum_{j>N}m_jn_j\abs{\beta_j}^{1/n_j}\to 0.
  \]
  By the trace-norm continuity of the Fredholm determinant
  \cite[Theorem~3.4]{Simon2005TraceIdeals},
  \[
    \det(I-zL_{\mathcal D})
    =\lim_{N\to\infty}\det(I-zL_{\mathcal D}^{(N)})
  \]
  locally uniformly in \(z\). Since
  \[
    Z_{L_{\mathcal D}^{(N)}}(z)
    =\prod_{j\le N}(1-\beta_j z^{n_j})^{-m_j},
  \]
  passage to the limit yields
  \[
    Z_{L_{\mathcal D}}(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  as a germ at \(0\). The operator norm satisfies
  \[
    \|L_{\mathcal D}\|
    =\sup_{j\in J}\abs{\alpha_j}
    =\sup_{j\in J}\abs{\beta_j}^{1/n_j}
    =r_{\mathcal D},
  \]
  so the identity is valid in particular for \(\abs{z}<r_{\mathcal D}^{-1}\).

  The implication \textup{(ii)}\(\Rightarrow\)\textup{(i)} is immediate.
  Conversely, assume \textup{(i)}. The first part of the theorem
  furnishes a normal operator \(L_{\mathcal D}\) with the same Fredholm
  zeta germ, and hence
  \[
    \det(I-zL)=\det(I-zL_{\mathcal D})
  \]
  as entire functions. By \cite[Theorem~3.7]{Simon2005TraceIdeals}, the
  zeros of \(\det(I-zT)\) are precisely the reciprocals of the non-zero
  eigenvalues of \(T\), counted with algebraic multiplicity. Therefore
  \(L\) and \(L_{\mathcal D}\) have the same non-zero spectrum with
  multiplicity.

  The operator \(L_{\mathcal D}\) is normal, and its non-zero
  eigenvalues are the numbers
  \[
    \alpha_j\omega,
    \qquad
    \omega^{n_j}=1,
  \]
  each occurring with multiplicity \(m_j\). Since \(L\) is compact and
  normal, the spectral theorem \cite{Conway1990Course} gives an
  orthogonal decomposition
  \[
    H=\ker L\oplus\bigoplus_{\lambda\in\spec(L)\setminus\{0\}}E_\lambda,
    \qquad
    L|_{E_\lambda}=\lambda I.
  \]
  For each non-zero eigenvalue \(\lambda\), let
  \[
    J_\lambda
    :=\bigl\{j\in J:\lambda=\alpha_j\omega
      \text{ for some }\omega^{n_j}=1\bigr\}.
  \]
  Since the multiplicity of \(\lambda\) is finite,
  \[
    \dim E_\lambda=\sum_{j\in J_\lambda}m_j.
  \]
  Choose an orthogonal decomposition
  \[
    E_\lambda=\bigoplus_{j\in J_\lambda}E_{\lambda,j},
    \qquad
    \dim E_{\lambda,j}=m_j.
  \]
  For each \(j\), set
  \[
    H_j:=\bigoplus_{\omega^{n_j}=1}E_{\alpha_j\omega,j}.
  \]
  Then \(H_j\) is reducing for \(L\), the subspaces \(H_j\) are pairwise
  orthogonal, and \(L|_{H_j}\) is unitarily equivalent to
  \[
    \bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}.
  \]
  The discrete Fourier transform on \(\CC^{n_j}\) diagonalizes
  \(\Pi_{n_j}\), so
  \[
    \bigoplus_{\omega^{n_j}=1}\alpha_j\omega\,I_{m_j}
    \cong C(n_j,\alpha_j)\otimes I_{m_j}.
  \]
  Hence \(L\) is unitarily equivalent to
  \(
    0_{\ker L}\oplus L_{\mathcal D}
  \),
  which proves \textup{(ii)}. The formulas for \(N(L)\) and \(\|L\|_1\)
  are now immediate from the first part of the proof.
\end{proof}

\begin{definition}[Block-cyclic normal realization]
  \label{def:block-cyclic-normal-realisation}
  A normal operator \(L\in \sone(H)\) satisfying
  \Cref{thm:cyclic-fredholm-realisation}\textup{(ii)} is called a
  \emph{block-cyclic normal realization} of the Euler data
  \(\mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J}\).
\end{definition}

\begin{corollary}[Prescribed spectral count and trace norm]
  \label{thm:block-cyclic-rigidity}
  Every block-cyclic normal realization \(L\) of \(\mathcal D\) satisfies
  \[
    N(L)=\sum_{j\in J}m_jn_j,
    \qquad
    \|L\|_1=\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  If \(L\) is compact and normal, this is equivalent to
  \[
    \dim\overline{\operatorname{ran}(L)}=\sum_{j\in J}m_jn_j.
  \]
  Consequently, the summability condition in
  \Cref{thm:cyclic-fredholm-realisation} is necessary and sufficient for
  normal trace-class realizability, and a single factor
  \((1-\beta z^n)^{-m}\) forces exactly \(mn\) non-zero eigenvalues.
\end{corollary}

\begin{proof}
  By definition, \(L\) is unitarily equivalent to
  \(0_{H_0}\oplus L_{\mathcal D}\) for some Hilbert space \(H_0\). Each
  block contributes \(m_jn_j\) non-zero eigenvalues and trace norm
  \(m_jn_j\abs{\beta_j}^{1/n_j}\), so the stated formulas follow. The
  compact-normal interpretation of \(N(L)\) was noted at the beginning
  of the section.
\end{proof}

\begin{remark}[Choice of roots]
  \label{rem:choice-of-roots}
  Replacing \(\alpha_j\) by another \(n_j\)th root of \(\beta_j\) merely
  rotates the same regular \(n_j\)-gon of eigenvalues. Hence the block
  \(C(n_j,\alpha_j)\otimes I_{m_j}\) is determined only up to unitary
  equivalence, which is all that is needed in
  \Cref{thm:cyclic-fredholm-realisation}.
\end{remark}

\section{Spectral rigidity and trace-norm minimality}
\label{sec:rigidity}

We now turn from the normal classification to consequences that depend
only on the Fredholm determinant. Once a single admissible block-cyclic
model is available, Simon's description of the zeros of
\(\det(I-zT)\) forces the non-zero spectrum of every trace-class
realization.

\begin{proposition}[Fredholm factorization rigidity]
  \label{prop:fredholm-factorisation-rigidity}
  \begin{enumerate}[label=\textup{(\alph*)}]
    \item Let
      \[
        \mathcal D=\{(n_j,\beta_j,m_j)\}_{j\in J},
        \qquad
        \mathcal D'=\{(n'_\ell,\beta'_\ell,m'_\ell)\}_{\ell\in L},
      \]
      and assume that both data sets satisfy the summability condition
      of \Cref{thm:cyclic-fredholm-realisation}. If the corresponding
      Euler products agree as holomorphic germs at \(z=0\), then the
      block-cyclic models attached to \(\mathcal D\) and
      \(\mathcal D'\) have the same non-zero eigenvalue multiset,
      counted with algebraic multiplicity.
      In particular,
      \[
        \sum_{j\in J}m_jn_j=\sum_{\ell\in L}m'_\ell n'_\ell,
      \]
      and the corresponding trace-norm sums
      \[
        \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j},
        \qquad
        \sum_{\ell\in L}m'_\ell n'_\ell\abs{\beta'_\ell}^{1/n'_\ell},
      \]
      are equal.
    \item Let \(T_1,T_2\in \sone(H)\). If
      \(
        Z_{T_1}(z)=Z_{T_2}(z)
      \)
      in a neighborhood of \(z=0\), then \(T_1\) and \(T_2\) have the
      same non-zero spectrum with algebraic multiplicity. Consequently,
      \[
        \Tr(T_1^n)=\Tr(T_2^n),
        \qquad
        p_n(T_1)=p_n(T_2)
        \qquad (n\ge 1).
      \]
  \end{enumerate}
\end{proposition}

\begin{proof}
  For \textup{(a)}, \Cref{thm:cyclic-fredholm-realisation} produces
  normal models \(L_{\mathcal D}\) and \(L_{\mathcal D'}\) with the
  prescribed Euler products as Fredholm zeta germs. Taking reciprocals,
  the entire functions
  \[
    z\mapsto \det(I-zL_{\mathcal D}),
    \qquad
    z\mapsto \det(I-zL_{\mathcal D'})
  \]
  agree near \(0\), hence everywhere. Simon's theorem
  \cite[Theorem~3.7]{Simon2005TraceIdeals} identifies their zeros with
  the reciprocals of the non-zero eigenvalues, counted with algebraic
  multiplicity. This gives the equality of eigenvalue multisets and,
  after summing moduli and multiplicities, the two displayed identities.

  For \textup{(b)}, the same argument applied directly to \(T_1\) and
  \(T_2\) yields equality of the non-zero spectra with multiplicity.
  Comparing the logarithmic expansions of \(Z_{T_1}\) and \(Z_{T_2}\)
  then shows that the trace sequences agree, and hence so do the
  M\"obius-inverted coordinates.
\end{proof}

\begin{corollary}[Determinant-intrinsic spectral count and trace norm in the normal class]
  \label{cor:determinant-intrinsic-normal-data}
  Let \(f\) be a holomorphic germ at \(0\) admitting an Euler
  factorization
  \[
    f(z)=\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  satisfying the summability condition of
  \Cref{thm:cyclic-fredholm-realisation}. If \(L\in\sone(H)\) is normal
  and
  \[
    Z_L(z)=f(z)
  \]
  near \(z=0\), then
  \[
    N(L)=\sum_{j\in J}m_jn_j,
    \qquad
    \|L\|_1=\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  For compact normal \(L\), this is equivalent to
  \[
    \dim\overline{\operatorname{ran}(L)}=\sum_{j\in J}m_jn_j.
  \]
  The right-hand sides depend only on the Fredholm zeta germ \(f\), not
  on the chosen Euler factorization.
\end{corollary}

\begin{proof}
  Apply \Cref{thm:block-cyclic-rigidity} to one admissible Euler
  factorization of \(f\), and then use
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(a)}.
\end{proof}

The next statement records the corresponding consequence without any
normality assumption.

\begin{proposition}[Determinant-imposed eigenvalue multiset for trace-class realizations]
  \label{thm:general-support-optimality}
  Let \(T\in \sone(H)\) satisfy
  \[
    Z_T(z)
    =\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  near \(z=0\), where the Euler data satisfy the summability condition
  of \Cref{thm:cyclic-fredholm-realisation}. Then the non-zero spectrum
  of \(T\), counted with algebraic multiplicity, is
  \[
    \biguplus_{j\in J}\bigl\{\alpha_j\omega:\omega^{n_j}=1\bigr\}^{m_j},
    \qquad
    \alpha_j^{n_j}=\beta_j.
  \]
  In particular,
  \[
    N(T)=\sum_{j\in J}m_jn_j.
  \]
\end{proposition}

\begin{proof}
  Let \(L_{\mathcal D}\) be the block-cyclic model associated with the
  given Euler data. Then
  \(Z_T(z)=Z_{L_{\mathcal D}}(z)\) near \(0\), so
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(b)} shows that
  \(T\) and \(L_{\mathcal D}\) have the same non-zero eigenvalues with
  algebraic multiplicity. The description of the spectrum of
  \(L_{\mathcal D}\) is immediate from the definition of the cyclic
  blocks.
\end{proof}

\begin{theorem}[Trace-norm minimality]
  \label{thm:trace-norm-minimality}
  Let \(T\in \sone(H)\) satisfy
  \[
    Z_T(z)=\prod_{j\in J}(1-\beta_j z^{n_j})^{-m_j}
  \]
  near \(z=0\), where the Euler data obey the summability condition of
  \Cref{thm:cyclic-fredholm-realisation}. Then
  \[
    \|T\|_1\ge \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  Equality holds if and only if \(T\) is normal. Equivalently, equality
  holds if and only if there exist a Hilbert space \(H_0\) and a unitary
  operator \(U:H\to H_0\oplus H_{\mathcal D}\) such that
  \[
    UTU^{-1}=0_{H_0}\oplus L_{\mathcal D},
  \]
  where \(L_{\mathcal D}\) is the block-cyclic model of
  \Cref{thm:cyclic-fredholm-realisation}.
\end{theorem}

\begin{proof}
  Let \(L_{\mathcal D}\) be the block-cyclic model associated with the
  given Euler data. By
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(b)},
  \(T\) and \(L_{\mathcal D}\) have the same non-zero eigenvalues with
  algebraic multiplicity. Hence
  \[
    \sum_k\abs{\lambda_k(T)}
    =\sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}
    =\|L_{\mathcal D}\|_1.
  \]
  Simon's eigenvalue-singular-value inequality
  \cite[Theorem~1.15]{Simon2005TraceIdeals} yields
  \[
    \sum_k\abs{\lambda_k(T)}\le \|T\|_1,
  \]
  and therefore
  \[
    \|T\|_1
    \ge \sum_{j\in J}m_jn_j\abs{\beta_j}^{1/n_j}.
  \]
  Since \(L_{\mathcal D}\) is normal, equality is attained by
  \(L_{\mathcal D}\), and hence also by every enlargement
  \(0_{H_0}\oplus L_{\mathcal D}\).

  Assume now that equality holds. Choose an orthonormal basis in which
  the compact operator \(T\) is upper triangular, with diagonal entries
  \(\lambda_k(T)\) repeated according to algebraic multiplicity. Let
  \(D\) be the diagonal unitary with diagonal entries
  \[
    d_k=
    \begin{cases}
      \overline{\lambda_k(T)}/\abs{\lambda_k(T)},
        & \lambda_k(T)\ne 0,\\
      1, & \lambda_k(T)=0.
    \end{cases}
  \]
  Then \(DT\) is upper triangular and
  \[
    \Tr(DT)=\sum_k\abs{\lambda_k(T)}=\|T\|_1.
  \]
  Write the polar decomposition \(T=V|T|\). Since \(\|DV\|\le 1\),
  \[
    0
    =\Tr|T|-\Re\Tr(DT)
    =\Tr\!\bigl(|T|^{1/2}(I-\Re(DV))|T|^{1/2}\bigr).
  \]
  The operator inside the trace is positive, hence vanishes. Thus
  \(\Re(DV)=I\) on \(\overline{\operatorname{ran}(|T|)}\). For
  \(h\in\overline{\operatorname{ran}(|T|)}\),
  \[
    \|h-DVh\|^2
    =\|h\|^2+\|DVh\|^2-2\Re\langle DVh,h\rangle
    \le 2\|h\|^2-2\|h\|^2=0,
  \]
  so \(DVh=h\). Therefore
  \[
    DT=DV|T|=|T|
  \]
  is positive. A positive upper-triangular operator is diagonal, hence
  \(DT\), and therefore \(T\), is diagonal in this basis. In
  particular \(T\) is normal. The converse characterization now follows
  from \Cref{thm:cyclic-fredholm-realisation}.
\end{proof}

\begin{corollary}[Symbolic-dynamical trace-norm rigidity]
  \label{cor:symbolic-trace-norm}
  Let \(A\) be a complex \(d\times d\) matrix and let \(T\in \sone(H)\)
  satisfy
  \[
    Z_T(z)=\det(I-zA)^{-1}
  \]
  near \(z=0\). Denote the non-zero eigenvalues of \(A\) by
  \(\{\lambda_i\}_{i=1}^M\), listed with algebraic multiplicity. Then
  \[
    \|T\|_1\ge \sum_{i=1}^M\abs{\lambda_i},
  \]
  and equality holds if and only if there exist a Hilbert space \(H_0\)
  and a unitary operator
  \[
    U:H\to H_0\oplus \CC^M
  \]
  such that
  \[
    UTU^{-1}=0_{H_0}\oplus\bigoplus_{i=1}^M C(1,\lambda_i).
  \]
  In particular, this applies to the zeta functions of primitive
  subshifts of finite type via
  \Cref{prop:prelim-trace-primitive}.
\end{corollary}

\begin{proof}
  Apply \Cref{thm:trace-norm-minimality} to the Euler data
  \(
    \mathcal D=\{(1,\lambda_i,1)\}_{i=1}^M
  \).
\end{proof}

\begin{corollary}[Finite Euler products and finite-rank normal models]
  \label{cor:finite-rank-realisation}
  If \(J\) is finite in \Cref{thm:cyclic-fredholm-realisation}, then
  \(L_{\mathcal D}\) is finite rank. In particular, every finite Euler
  product
  \[
    \prod_{j=1}^N(1-\beta_j z^{n_j})^{-m_j}
  \]
  admits a finite-rank normal Fredholm realization whose non-zero rank
  is exactly \(\sum_{j=1}^N m_jn_j\).
\end{corollary}

\begin{corollary}[Realizability versus rigidity]
  \label{cor:realisability-vs-rigidity}
  Let \(A\) be a finite matrix and let \(T\in \sone(H)\) satisfy
  \[
    \det(I-zT)=\det(I-zA).
  \]
  Then \(T\) and \(A\) have the same non-zero eigenvalues with algebraic
  multiplicity. Thus enlarging the ambient operator class does not
  create additional spectral freedom once the determinant is fixed.
\end{corollary}

\begin{proof}
  Equality of determinants implies \(Z_T(z)=Z_A(z)\) near \(0\), so the
  conclusion follows from
  \Cref{prop:fredholm-factorisation-rigidity}\textup{(b)}.
\end{proof}
