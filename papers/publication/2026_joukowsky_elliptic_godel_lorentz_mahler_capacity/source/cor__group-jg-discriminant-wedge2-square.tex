\begin{corollary}[Wedge-square factorization of the discriminant]\label{cor:group-jg-discriminant-wedge2-square}
In the setting of Theorem \ref{thm:group-jg-discriminant-factorization}, assume further that \(P\) is monic and separable. Let \(z_1,\dots,z_n\) be the roots of \(P\), and define the second exterior-power root-product polynomial
\[
P^{\wedge 2}(u):=\prod_{1\le i<j\le n}\bigl(u-z_i z_j\bigr).
\]
Then the coefficients of \(P^{\wedge 2}(u)\) lie in \(K\), and the discriminant of the transport polynomial \(Q_r\) satisfies the exact identity
\[
\boxed{\;
\operatorname{Disc}_{w}(Q_r)
=r^{n(n-1)}\,\operatorname{Disc}_{z}(P)\,\Bigl(P^{\wedge 2}(r^{-2})\Bigr)^2.
\;}
\]
\leanverified{r\_sq\_mul\_sub\_one}
\leanverified{sq\_sub\_symm}
\leanverified{pow\_mul\_pow\_two}
\leanverified{sq\_mul\_r\_sq}
\leanverified{sq\_eq\_factor}
\leanverified{sq\_eq\_factor\_symm}
\leanverified{paper\_group\_jg\_discriminant\_wedge2\_square}
\end{corollary}

\begin{proof}
By Theorem \ref{thm:group-jg-discriminant-factorization} (here \(a_n=1\)),
\[
\operatorname{Disc}_{w}(Q_r)
=\operatorname{Disc}_{z}(P)\,
r^{-n(n-1)}\prod_{1\le i<j\le n}(r^2 z_i z_j-1)^2.
\]
Moreover,
\[
r^2 z_i z_j-1=r^2(z_i z_j-r^{-2}),
\]
and therefore
\[
\prod_{i<j}(r^2 z_i z_j-1)^2
=r^{2n(n-1)}\prod_{i<j}(z_i z_j-r^{-2})^2
=r^{2n(n-1)}\bigl(P^{\wedge 2}(r^{-2})\bigr)^2,
\]
where in the last step we use the fact that squaring removes the sign discrepancy,
\(
\prod_{i<j}(z_i z_j-r^{-2})^2=\prod_{i<j}(r^{-2}-z_i z_j)^2
\)。
Substituting this back gives the stated identity. On the other hand, the coefficients of \(P^{\wedge 2}\) are symmetric polynomials in \(\{z_i\}\), hence belong to \(K\). This completes the proof.
\end{proof}
