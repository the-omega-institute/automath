\begin{corollary}[Holographic identity under the Lee--Yang unit-circle hypothesis]\label{cor:group-jg-leyang-holographic-identity}
In the setting of Theorem \ref{thm:group-jg-pullback-factorization}, assume further that \(K\subseteq\CC\) and that all zeros of \(P\) satisfy
\[
|z_j|=1\qquad(1\le j\le n),
\]
and let \(r\in\RR_{>0}\). Write \(J_r(z)=r z+r^{-1}z^{-1}\). Then for every \(z\in\CC^\times\) one has the exact identity
\[
\boxed{\;
Q_r\!\bigl(J_r(z)\bigr)=r^{n}\,P(z)\,P\!\bigl(r^{-2}z^{-1}\bigr).
\;}
\]

\leanverified{J\_r}
\leanverified{P}
\leanverified{Q\_r\_eval\_at\_J}
\leanverified{J\_r\_eq}
\leanverified{P\_at\_reciprocal}
\leanverified{r\_sq\_P\_P\_reciprocal}
\leanverified{Q\_r\_eval\_at\_J\_eq}
\leanverified{leyang\_holographic\_n2}
\end{corollary}

\begin{proof}
By Theorem \ref{thm:group-jg-pullback-factorization},
\[
r^n z^n\,Q_r\!\bigl(J_r(z)\bigr)=P(z)\,P^{\vee}(r^2 z).
\]
By the definition of the reciprocal polynomial \(P^{\vee}(u)=u^n P(1/u)\), substituting \(u=r^2 z\) yields
\[
P^{\vee}(r^2 z)=r^{2n}z^n P(r^{-2}z^{-1}).
\]
Substituting this into the previous identity and canceling the common factor \(z^n\) gives the claim.
\end{proof}

\begin{corollary}[Double-resultant elimination and concentric double spectra]\label{cor:group-jg-leyang-double-resultant}
Retain the hypotheses of Corollary \ref{cor:group-jg-leyang-holographic-identity}, and define
\[
S_r(z):=\operatorname{Res}_w\!\bigl(Q_r(w),\,r z^2-w z+r^{-1}\bigr)\in K(r)[z].
\]
Then in \(K(r)[z]\) one has the exact identity
\[
\boxed{\;
S_r(z)=(-1)^n r^{n}\,P(z)\,\Bigl[z^{n}P\!\bigl(r^{-2}z^{-1}\bigr)\Bigr].
\;}
\]
Consequently, over \(\CC\) its zero set is precisely
\[
\mathrm{Zero}(S_r)
=\{z_j\}_{j=1}^n\ \cup\ \{r^{-2}z_j^{-1}\}_{j=1}^n,
\]
where the two clusters lie respectively on the circles \(|z|=1\) and \(|z|=r^{-2}\).
\end{corollary}

\begin{proof}
Since \(r z^2-w z+r^{-1}\) is linear in \(w\), the linear specialization formula for the resultant gives
\[
\operatorname{Res}_w\!\bigl(Q_r(w),\,r z^2-w z+r^{-1}\bigr)
=(-z)^n\,Q_r\!\left(\frac{r z^2+r^{-1}}{z}\right)
=(-z)^n Q_r\!\bigl(J_r(z)\bigr).
\]
Substituting Corollary \ref{cor:group-jg-leyang-holographic-identity} therefore yields
\[
S_r(z)=(-z)^n r^{n}P(z)P(r^{-2}z^{-1})
=(-1)^n r^{n}P(z)\,\bigl[z^nP(r^{-2}z^{-1})\bigr],
\]
which is exactly the displayed identity. The location of the zeros follows from the two factors.
\end{proof}

\begin{corollary}[Invertibility and holographic reconstruction from a single elliptic transport scale]\label{cor:group-jg-leyang-single-scale-reconstruction}
In the setting of Corollary \ref{cor:group-jg-leyang-double-resultant}, assume further that \(r>1\) and that \(P\) is separable and monic. Then the map
\[
\mathcal{T}_r:\ P\ \longmapsto\ Q_r
\]
is injective, and there is a reconstruction procedure uniquely determined by \((r,Q_r)\): compute \(S_r(z)\) from \(Q_r\), then \(P\) is the unique monic factor of \(S_r\) of degree \(n\) whose zeros all lie on the unit circle \(|z|=1\).
\end{corollary}

\begin{proof}
By Corollary \ref{cor:group-jg-leyang-double-resultant},
\[
S_r(z)=(-1)^n r^n\,P(z)\cdot \bigl[z^nP(r^{-2}z^{-1})\bigr].
\]
When \(r>1\), all zeros of the second factor lie on the circle \(|z|=r^{-2}<1\), whereas all zeros of \(P\) lie on \(|z|=1\). Hence the two zero clusters are disjoint.
Therefore, in the factorization of \(S_r\) over \(\CC[z]\), there exists exactly one degree-\(n\) factor whose zeros all lie on the unit circle; taking its monic normalization recovers \(P\).
If \(\mathcal{T}_r(P_1)=\mathcal{T}_r(P_2)\), then the corresponding polynomials \(S_r\) coincide, hence the unit-circle degree-\(n\) factor described above is the same for both. Thus \(P_1=P_2\).
\end{proof}
