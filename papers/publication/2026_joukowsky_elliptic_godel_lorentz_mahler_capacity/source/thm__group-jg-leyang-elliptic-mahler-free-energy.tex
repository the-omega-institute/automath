\begin{theorem}[Two-scale logarithmic mean identity on the Joukowsky--Gödel elliptic boundary]\label{thm:group-jg-elliptic-two-scale-stokes-mean}
In the notation of Theorem \ref{thm:group-jg-pullback-factorization}, let \(K=\CC\) and \(r>1\). Let \(P(z)\in\CC[z]\) have degree \(n\ge 1\) and satisfy \(P(0)\neq 0\). Set
\[
J_r(e^{i\theta})=r e^{i\theta}+r^{-1}e^{-i\theta},
\qquad
E_r:=J_r(\{e^{i\theta}:\theta\in[0,2\pi)\}),
\]
and let \(\mu_r\) be the pushforward of Haar measure \(\frac{\dd\theta}{2\pi}\) under \(J_r\). Define the circular logarithmic mean functional of radius \(\rho>0\) by
\[
\mathcal M_\rho(P):=\frac1{2\pi}\int_{0}^{2\pi}\log|P(\rho e^{i\theta})|\,\dd\theta.
\]
Then one has the exact identity
\[
\boxed{\;
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)
=
\mathcal M_{1}(P)+\mathcal M_{r^{-2}}(P)+n\log r.
\;}
\]
\end{theorem}
\begin{proof}
By Theorem \ref{thm:group-jg-pullback-factorization}, for \(|z|=1\) one has
\[
r^n z^n Q_r\!\left(r z+r^{-1}z^{-1}\right)=P(z)\,P^{\vee}(r^2 z).
\]
Set \(z=e^{i\theta}\) and take logarithmic absolute values. Since \(\log|z^n|=0\), this gives
\[
\log\bigl|Q_r(J_r(e^{i\theta}))\bigr|
=
\log|P(e^{i\theta})|+\log|P^\vee(r^2 e^{i\theta})|-n\log r.
\]
Average over \(\theta\in[0,2\pi)\), and identify the left-hand side with \(\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)\) by definition of \(\mu_r\). We obtain
\[
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)
=\mathcal M_1(P)+\frac1{2\pi}\int_{0}^{2\pi}\log|P^\vee(r^2 e^{i\theta})|\,\dd\theta-n\log r.
\]
On the other hand, from \(P^\vee(u)=u^nP(1/u)\) we have
\[
P^\vee(r^2 e^{i\theta})=r^{2n}e^{in\theta}P(r^{-2}e^{-i\theta}),
\qquad
\bigl|P^\vee(r^2 e^{i\theta})\bigr|=r^{2n}\bigl|P(r^{-2}e^{-i\theta})\bigr|.
\]
Hence
\[
\frac1{2\pi}\int_{0}^{2\pi}\log|P^\vee(r^2 e^{i\theta})|\,\dd\theta
=2n\log r+\mathcal M_{r^{-2}}(P),
\]
and substituting this into the previous identity gives the claim.
\end{proof}

\begin{theorem}[Closed form for the bivariate exponential moment generating function of a Joukowsky ellipse]\label{thm:group-jg-joukowsky-bivariate-mgf-bessel}
In the notation of Theorem \ref{thm:group-jg-elliptic-two-scale-stokes-mean}, let \(r>1\), and write
\[
J_r(e^{i\theta})=r e^{i\theta}+r^{-1}e^{-i\theta},
\qquad
E_r:=J_r(\TT),
\qquad
\mu_r:=J_{r\ast}\!\left(\frac{\dd\theta}{2\pi}\right).
\]
For \(u,v\in\CC\), define the bivariate exponential moment generating function
\[
M_r(u,v):=\int_{E_r}\exp\bigl(u w+v\overline w\bigr)\,\dd\mu_r(w).
\]
Then one has the closed formula
\[
\boxed{\;
M_r(u,v)=I_0\!\Bigl(2\sqrt{\,u^2+v^2+(r^2+r^{-2})uv\,}\Bigr),
\;}
\]
where \(I_0\) is the modified Bessel function of the first kind of order \(0\). In particular:
\begin{enumerate}
  \item setting \(v=0\) gives \(M_r(u,0)=I_0(2u)\), independent of \(r\);
  \item the dependence on \(r\) enters only through the scalar \(t=r^2+r^{-2}\), hence all mixed moments
  \(\int_{E_r} w^a\overline w^{\,b}\,\dd\mu_r(w)\)
  are polynomials in \(t\), with coefficients depending only on \((a,b)\).
\end{enumerate}
\end{theorem}
\begin{proof}
Let \(z=e^{i\theta}\). Then
\[
w=r z+r^{-1}z^{-1},\qquad \overline w=r z^{-1}+r^{-1}z,
\]
so that
\[
u w+v\overline w
=(ur+vr^{-1})z+(ur^{-1}+vr)z^{-1}.
\]
Write \(A:=ur+vr^{-1}\) and \(B:=ur^{-1}+vr\). Then
\[
M_r(u,v)=\frac1{2\pi}\int_0^{2\pi}\exp\!\bigl(Ae^{i\theta}+Be^{-i\theta}\bigr)\,\dd\theta.
\]
A standard identity yields
\[
\frac1{2\pi}\int_0^{2\pi}\exp\!\bigl(Ae^{i\theta}+Be^{-i\theta}\bigr)\,\dd\theta
=I_0\!\bigl(2\sqrt{AB}\bigr),
\]
while
\[
AB=(ur+vr^{-1})(ur^{-1}+vr)=u^2+v^2+(r^2+r^{-2})uv,
\]
and substitution gives the main formula.
When \(v=0\), one has \(AB=u^2\), hence \(M_r(u,0)=I_0(2u)\).
Finally, the Taylor coefficients of \(M_r\) in \((u,v)\) are exactly the mixed moments, and the main formula depends on \(r\) only through \(t=r^2+r^{-2}\). Therefore every mixed moment is a polynomial in \(t\).
\end{proof}

\begin{corollary}[Multiplicative decomposition of logarithmic means on the elliptic boundary]\label{cor:group-jg-elliptic-two-scale-mahler-multiplicative}
Under the hypotheses of Theorem \ref{thm:group-jg-elliptic-two-scale-stokes-mean},
\[
\exp\!\left(\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)\right)
=
r^n\,\exp(\mathcal M_1(P))\,\exp(\mathcal M_{r^{-2}}(P)).
\]
\end{corollary}
\begin{proof}
Exponentiating both sides of Theorem \ref{thm:group-jg-elliptic-two-scale-stokes-mean} gives the result.
\end{proof}

\begin{theorem}[Mahler--free-energy identity for Lee--Yang elliptic transport]\label{thm:group-jg-leyang-mahler-free-energy}
In the notation of Theorem \ref{thm:group-jg-pullback-factorization}, let \(K=\CC\) and \(r>1\). Let \(P(z)\in\CC[z]\) have degree \(n\ge 1\), and assume the Lee--Yang condition that all zeros satisfy \(|z_j|=1\). Set
\[
J_r(e^{i\theta})=r e^{i\theta}+r^{-1}e^{-i\theta},
\qquad
E_r:=J_r(\{e^{i\theta}:\theta\in[0,2\pi)\}),
\]
and let \(\mu_r\) be the pushforward of Haar measure \(\frac{\dd\theta}{2\pi}\) under \(J_r\). Then one has the exact identity
\[
\boxed{\;
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)=\log|a_n a_0|+n\log r\; }.
\]
In particular, if \(P\) is monic, then
\[
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)=n\log r.
\]
\end{theorem}

\begin{proof}
By Theorem \ref{thm:group-jg-pullback-factorization}, for \(|z|=1\) one has
\[
r^n z^n Q_r\!\left(r z+r^{-1}z^{-1}\right)=P(z)\,P^{\vee}(r^2 z).
\]
Taking logarithmic absolute values and integrating over \(|z|=1\) against \(\frac{\dd\theta}{2\pi}\), while observing that \(\log|z|=0\), gives
\[
\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w)
=\int_{0}^{2\pi}\log|P(e^{i\theta})|\,\frac{\dd\theta}{2\pi}
\ +\ \int_{0}^{2\pi}\log|P^{\vee}(r^2 e^{i\theta})|\,\frac{\dd\theta}{2\pi}
\ -\ n\log r.
\]
Since all zeros of \(P\) lie on the unit circle, Jensen's formula yields
\[
\int_{0}^{2\pi}\log|P(e^{i\theta})|\,\frac{\dd\theta}{2\pi}=\log|a_n|.
\]
On the other hand, the leading coefficient of \(P^\vee(u)=u^n P(1/u)\) is \(a_0\), and the zeros of \(P^\vee(r^2 z)\) are \((r^2 z_j)^{-1}\), all of modulus \(r^{-2}<1\).
Applying Jensen's formula once more, we find
\[
\int_{0}^{2\pi}\log|P^{\vee}(r^2 e^{i\theta})|\,\frac{\dd\theta}{2\pi}
=\log|a_0 r^{2n}|=\log|a_0|+2n\log r.
\]
Substituting this into the previous identity gives the stated formula.
\end{proof}

\begin{corollary}[Linear response of the elliptic transport free energy]\label{cor:group-jg-leyang-free-energy-linear-flux}
Under the hypotheses of Theorem \ref{thm:group-jg-leyang-mahler-free-energy}, define
\[
\mathcal F(r):=\int_{E_r}\log|Q_r(w)|\,\dd\mu_r(w).
\]
Then \(\mathcal F(r)=\log|a_n a_0|+n\log r\), and hence
\[
\frac{\dd \mathcal F}{\dd(\log r)}\equiv n.
\]
\end{corollary}

\begin{proof}
The explicit expression follows directly from Theorem \ref{thm:group-jg-leyang-mahler-free-energy}; differentiating with respect to \(\log r\) gives the conclusion.
\end{proof}

\begin{theorem}[Doubling dynamics and binary symbolic coding on a Lee--Yang ellipse]\label{thm:group-jg-leyang-ellipse-doubling-shift}
Let \(r>1\), and let \(E_r\) and its parametrization \(w(\theta)=r e^{i\theta}+r^{-1}e^{-i\theta}\) be as above. Define
\[
T_r:E_r\to E_r,\qquad T_r(w(\theta)):=w(2\theta)\qquad(\theta\in\RR/2\pi\ZZ).
\]
Then:
\begin{enumerate}
  \item \(T_r\) is topologically conjugate to the doubling map on the circle \(D:\theta\mapsto 2\theta\ (\mathrm{mod}\ 2\pi)\), hence \(h_{\mathrm{top}}(T_r)=\log 2\);
  \item \(\mu_r\) is an invariant probability measure for \(T_r\);
  \item writing \(h(\theta)=w(\theta)\), the Lyapunov exponent with respect to \(\mu_r=h_\ast(\frac{\dd\theta}{2\pi})\) satisfies
  \[
  \chi(T_r,\mu_r):=\int_{E_r}\log|DT_r|\,\dd\mu_r=\log 2.
  \]
  \item after removing the null set corresponding to dyadic angles, there is a measurable isomorphism identifying \(D\) with the left shift on binary sequences; therefore \(T_r\) on \((E_r,\mu_r)\) is measure-theoretically isomorphic to the left shift on \(\{0,1\}^{\NN}\).
\end{enumerate}
\end{theorem}

\begin{proof}
Since \(\theta\mapsto w(\theta)\) is a continuous bijection from \(\RR/2\pi\ZZ\) onto the compact set \(E_r\), it is a homeomorphism \(h:\TT\to E_r\).
By definition, \(T_r\circ h=h\circ D\), so the topological conjugacy is immediate. In particular, \(h_{\mathrm{top}}(T_r)=h_{\mathrm{top}}(D)=\log 2\).
\par
\(\mu_r=h_\ast(\frac{\dd\theta}{2\pi})\) is \(T_r\)-invariant because
\(
D_\ast(\frac{\dd\theta}{2\pi})=\frac{\dd\theta}{2\pi}
\)
on the circle.
\par
In the \(\theta\)-coordinate, the chain rule gives
\[
DT_r\bigl(h(\theta)\bigr)=\frac{h'(2\theta)\cdot 2}{h'(\theta)}.
\]
Therefore
\[
\int_{E_r}\log|DT_r|\,\dd\mu_r
=\int_{0}^{2\pi}\left(\log 2+\log|h'(2\theta)|-\log|h'(\theta)|\right)\frac{\dd\theta}{2\pi}
=\log 2,
\]
where the last step uses the invariance of Haar measure under \(\theta\mapsto 2\theta\).
The final assertion is the classical symbolic coding of the doubling map by binary left shift, after removing the dyadic set on which binary expansion is not unique.
\end{proof}
