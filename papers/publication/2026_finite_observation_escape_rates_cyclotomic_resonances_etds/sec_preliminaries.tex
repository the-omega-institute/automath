\section{Preliminaries}\label{sec:preliminaries}

Throughout the paper, \((X,\mathcal{B},\mu,T)\) denotes a
probability-preserving system.  A finite observation is specified by a
finite alphabet \(\Sigma\) and a measurable partition
\[
  \mathcal W=\{W_i:i\in\Sigma\}
\]
of \(X\).  We write
\(\Sigma^{<\infty}:=\bigcup_{m\ge 0}\Sigma^m\).  The binary case is the
special case \(\Sigma=\{0,1\}\), \(W_1=W\), and
\(W_0=X\setminus W\).

\subsection{Finite readouts, words, and cylinders}

For \(t\in\NNzero\) define the observed symbol \(s_t(x)\in\Sigma\) by
\[
  T^t x\in W_{s_t(x)} .
\]
For $m\ge 1$ and $t\in\NNzero$ set
\[
  \omega_m(x;t):=s_t(x)s_{t+1}(x)\cdots s_{t+m-1}(x)\in\Sigma^m.
\]
If $a=a_0\cdots a_{m-1}\in\Sigma^m$, we write $\abs{a}=m$ and define the
cylinder event
\[
  C_a:=\{x\in X:\omega_m(x;0)=a\}
  =\bigcap_{j=0}^{m-1}T^{-j}W_{a_j}.
\]
More generally,
\[
  C_{a,t}:=\{x\in X:\omega_m(x;t)=a\}=T^{-t}C_a.
\]
The finite cylinder partition at resolution $m$ is
\[
  \mathcal{C}_m:=\{C_a:a\in\Sigma^m\},
\]
and the associated finite $\sigma$-algebra is
\[
  \mathcal{F}_m:=\sigma(C_a:a\in\Sigma^m).
\]
We also write
\[
  \mathcal{F}_\infty:=\sigma\!\Bigl(\bigcup_{m\ge 1}\mathcal{F}_m\Bigr).
\]

\begin{proposition}[Symbolic factor]\label{prop:symbolic-factor}
Define
\[
  \omega_\infty(x):=(s_t(x))_{t\ge 0}\in\Sigma^{\NNzero}.
\]
Let $\sigma:\Sigma^{\NNzero}\to\Sigma^{\NNzero}$ be the left shift,
$(\sigma\xi)_t=\xi_{t+1}$. Then:
\begin{enumerate}[label=(\alph*)]
  \item $\omega_\infty$ is measurable and satisfies
  $\omega_\infty\circ T=\sigma\circ\omega_\infty$;
  \item if $[a]:=\{\xi\in\Sigma^{\NNzero}:\xi_0\cdots\xi_{m-1}=a\}$,
  then $C_a=\omega_\infty^{-1}([a])$;
  \item if $\mathcal{A}_m:=\sigma([a]:a\in\Sigma^m)$, then
  $\mathcal{F}_m=\omega_\infty^{-1}(\mathcal{A}_m)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Part (a) is immediate from the measurability of the partition
\(\mathcal W\) and of \(T\).
Part (b) is the definition of $C_a$ written in symbolic form.
Part (c) follows from (b) because $\mathcal{A}_m$ is generated by length-$m$
symbolic cylinders.
\end{proof}

\subsection{Prefix geometry}

For $\xi,\eta\in\Sigma^{\NNzero}$ define the common-prefix depth
\[
  \ell(\xi,\eta):=\sup\{k\ge 0:\xi_0\cdots\xi_{k-1}=\eta_0\cdots\eta_{k-1}\}
  \in\NNzero\cup\{\infty\},
\]
with the convention $2^{-\infty}=0$. The prefix ultrametric is
\[
  d_{\mathrm{pre}}(\xi,\eta):=2^{-\ell(\xi,\eta)}.
\]
For $x,y\in X$ we define the pullback pseudo-ultrametric
\[
  d_X(x,y):=d_{\mathrm{pre}}(\omega_\infty(x),\omega_\infty(y)).
\]

\begin{lemma}[Prefix balls are cylinders]\label{lem:prefix-ball-cylinder}
Let $a\in\Sigma^m$ and $\xi\in[a]$. Then
\[
  [a]=\overline{B}_{d_{\mathrm{pre}}}(\xi,2^{-m}).
\]
In particular, the topology induced by $d_{\mathrm{pre}}$ is the usual product
topology on $\Sigma^{\NNzero}$.
\end{lemma}

\begin{proof}
If $\eta\in[a]$, then $\ell(\xi,\eta)\ge m$, hence
$d_{\mathrm{pre}}(\xi,\eta)\le 2^{-m}$.
Conversely, if $d_{\mathrm{pre}}(\xi,\eta)\le 2^{-m}$, then
$\ell(\xi,\eta)\ge m$, so $\eta$ has the same first $m$ symbols as $\xi$, and
therefore belongs to $[a]$.
\end{proof}

\subsection{Conditional probabilities at finite resolution}

We use the depth-zero convention
\[
  \mathcal{F}_0:=\{\emptyset,X\}.
\]
If $\mathsf{P}\in\mathcal{F}_\infty$, then we set
\[
  p_0:=\mu(\mathsf{P}\mid\mathcal{F}_0)=\mu(\mathsf{P})
\]
as a constant random variable, and for each $m\ge 1$ we write
\[
  p_m:=\mu(\mathsf{P}\mid\mathcal{F}_m).
\]
The process $(p_m)_{m\ge 0}$ is the finite-resolution conditional-probability
martingale associated with $\mathsf{P}$. It will govern both the optimal
classification rule in Section~\ref{sec:clarity} and the boundary-decay
identities proved there.

\subsection{Additional hypothesis}

When we want the symbolic factor to recover all measurable information, we
assume that the observation partition \(\mathcal W\) is generating, so
\[
  \mathcal{F}_\infty=\mathcal{B}
  \qquad\text{modulo \(\mu\)-null sets.}
\]
This hypothesis is used only for full-information remarks.  The structural
results in the main open-system theorems do not require it.
