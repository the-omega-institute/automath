The escape-rate and amplitude formulas from Section~\ref{sec:open-system}
already contain the main dynamical information.  The present subsection
records supplementary consequences of the same killed-operator
formalism.  The finite-state result is an unconditional resolvent
reformulation.  The H\"older block-recoding statement is conditional on
the explicit Banach-space power bound, represented adjoint, spectral
identification, and residue non-degeneracy hypotheses stated below.

\subsection{Supplementary resolvent reformulations}

\begin{theorem}[Error resolvent and cyclotomic resonance lift]
\label{thm:error-resolvent-cyclotomic}
Under the hypotheses of
Theorem~\ref{thm:first-entry-escape-rate}, set
\[
  \widehat{\mathcal S}:=\mathcal S\times \ZZ/q\ZZ,
\]
and define the lifted killed matrix \(\widehat B_H\) on
\(\widehat{\mathcal S}\) by
\[
  \widehat B_H\bigl((i,r),(j,s)\bigr)
  := B_H(i,j)\mathbf{1}_{\{s=r+1\bmod q\}}.
\]
Let \(\widehat u,\widehat b\in\RR^{\widehat{\mathcal S}}\) be given by
\[
  \widehat u(i,r):=u_i\mathbf{1}_{\{r=0\}},
  \qquad
  \widehat b(i,r):=b_r(i).
\]
Then for every \(m\geq 1\),
\[
  \varepsilon_m(\mathsf{P}_R;\nu_\varphi)
  = \widehat u^{\mathsf T}\widehat B_H^{m-1}\widehat b.
\]
Consequently the error generating function
\[
  \mathcal E_{\mathsf{P}_R}(z)
  := \sum_{m\geq 1}
      \varepsilon_m(\mathsf{P}_R;\nu_\varphi)z^m
\]
satisfies
\[
  \mathcal E_{\mathsf{P}_R}(z)
  = z\,\widehat u^{\mathsf T}
      (I-z\widehat B_H)^{-1}\widehat b
  \qquad (|z|<\rho_H^{-1}),
\]
and therefore extends to a rational function of \(z\).

If \(S_q\) denotes the cyclic permutation matrix on \(\ZZ/q\ZZ\), then
\[
  \widehat B_H=S_q\otimes B_H.
\]
Hence
\[
  \det(I-z\widehat B_H)=\det(I-z^q B_H^q),
\]
and
\[
  \spec(\widehat B_H)
  = \bigcup_{\omega^q=1}\omega\,\spec(B_H).
\]
In particular, every pole of \(\mathcal E_{\mathsf{P}_R}\) belongs to
\[
  \set{(\omega\lambda)^{-1}:
       \omega^q=1,\ \lambda\in\spec(B_H)\setminus\{0\}},
\]
and the positive real pole of smallest modulus is
\[
  z_H=\rho_H^{-1}
      =e^{\gamma_H(\varphi)}
      =e^{P_Y(\varphi)-P_{Y_H}(\varphi)}.
\]
Thus finite-observation error detects the killed open-system spectrum
through a cyclotomic residue lift, while the escape rate is recovered as
the dominant real pole.
\leanverified{paper_scan_projection_address_error_resolvent_cyclotomic}
\end{theorem}

\begin{proof}
For \(n\geq 0\), a direct induction on \(n\) gives
\[
  \widehat B_H^n\bigl((i,r),(j,s)\bigr)
  = B_H^n(i,j)\mathbf{1}_{\{s=r+n\bmod q\}}.
\]
Therefore
\begin{align*}
  \widehat u^{\mathsf T}\widehat B_H^{m-1}\widehat b
  &= \sum_{i,j\in\mathcal S}
      u_i B_H^{m-1}(i,j)b_{m-1\bmod q}(j) \\
  &= u^{\mathsf T}B_H^{m-1}b_{m-1\bmod q},
\end{align*}
which is exactly
\(\varepsilon_m(\mathsf{P}_R;\nu_\varphi)\) by
Theorem~\ref{thm:first-entry-escape-rate}.

For \(|z|<\rho(\widehat B_H)^{-1}\), the geometric series converges
absolutely and yields
\[
  \mathcal E_{\mathsf{P}_R}(z)
  = \sum_{m\geq 1}
      \widehat u^{\mathsf T}\widehat B_H^{m-1}\widehat b\,z^m
  = z\,\widehat u^{\mathsf T}
      \sum_{n\geq 0}(z\widehat B_H)^n\widehat b
  = z\,\widehat u^{\mathsf T}
      (I-z\widehat B_H)^{-1}\widehat b.
\]
This is rational in \(z\).

If \(S_q\) is the cyclic permutation matrix on \(\ZZ/q\ZZ\), then the
definition of \(\widehat B_H\) gives
\[
  \widehat B_H=S_q\otimes B_H.
\]
Since the eigenvalues of \(S_q\) are exactly the \(q\)-th roots of
unity, the standard spectrum formula for Kronecker products implies
\[
  \spec(\widehat B_H)
  = \set{\omega\lambda:
         \omega^q=1,\ \lambda\in\spec(B_H)}.
\]
Multiplying over the spectrum yields
\[
  \det(I-z\widehat B_H)
  = \prod_{\lambda\in\spec(B_H)}
      \prod_{\omega^q=1}(1-z\omega\lambda)
  = \prod_{\lambda\in\spec(B_H)}(1-z^q\lambda^q)
  = \det(I-z^q B_H^q).
\]
Hence the poles of the scalar resolvent
\(\mathcal E_{\mathsf{P}_R}\) are contained in the reciprocal lifted
spectrum.

It remains to identify the dominant real pole.  Let \(v,\ell\geq 0\) be
Perron right and left eigenvectors of \(B_H\) for \(\rho_H\), normalized
by \(\ell^{\mathsf T}v=1\).  Since the transient block \(Q\) is
nilpotent, both \(v\) and \(\ell\) are strictly positive on
\(\mathcal S_\infty\).  Set
\[
  \widehat v:=\mathbf{1}_q\otimes v,
  \qquad
  \widehat\ell:=\mathbf{1}_q\otimes \ell,
\]
where \(\mathbf{1}_q\) is the all-ones vector on \(\ZZ/q\ZZ\).  Then
\(\widehat B_H\widehat v=\rho_H\widehat v\) and
\(\widehat\ell^{\mathsf T}\widehat B_H=\rho_H\widehat\ell^{\mathsf T}\).
Moreover,
\[
  \widehat u^{\mathsf T}\widehat v=u^{\mathsf T}v>0,
  \qquad
  \widehat\ell^{\mathsf T}\widehat b
  = \sum_{a\in\ZZ/q\ZZ}\ell^{\mathsf T}b_a>0,
\]
because \(u_i=\pi_i>0\) on \(\mathcal S_\infty\) and each \(b_a(i)\geq
\theta>0\).  Since \(1\) is a simple eigenvalue of \(S_q\) and
\(\rho_H\) is the simple Perron eigenvalue of \(B_H\), the eigenvalue
\(\rho_H\) of \(\widehat B_H=S_q\otimes B_H\) is also simple.  Thus the
scalar resolvent coefficient does not annihilate the corresponding
eigenprojection, so \(z=\rho_H^{-1}\) is a genuine simple pole of
\(\mathcal E_{\mathsf{P}_R}\).  The pressure identity from
Theorem~\ref{thm:first-entry-escape-rate} gives
\[
  \rho_H=e^{P_{Y_H}(\varphi)-P_Y(\varphi)}
  =e^{-\gamma_H(\varphi)},
\]
and therefore the displayed formula for \(z_H\).
\end{proof}

\begin{theorem}[Conditional H\"older Gibbs cyclotomic lift after block recoding]
\label{thm:holder-open-resonance}
Let \(Y\) be a topologically mixing one-sided subshift of finite type,
let \(\varphi\) be a H\"older potential normalized so that
\(\mathcal L_\varphi\mathbf 1=\mathbf 1\), and let \(H\subset Y\) be a
finite union of cylinders.  Fix \(q\geq 2\) and a non-empty proper subset
\(R\subsetneq \ZZ/q\ZZ\).  Choose \(\ell\geq 1\) so that \(H\) is a union
of \(\ell\)-cylinders, and let \((Y^{(\ell)},\sigma^{(\ell)})\) be the
\(\ell\)-block presentation of \(Y\).  Write \(H^{(\ell)}\) for the
corresponding one-step hole, \(\psi:=\varphi^{(\ell)}\) for the lifted
normalized H\"older potential, and
\[
  \mathcal H_\alpha^-:=\mathcal H_\alpha((Y^{(\ell)})^-),
  \qquad
  \mathcal H_\alpha^+:=\mathcal H_\alpha(Y^{(\ell)})
\]
for the past and future H\"older spaces with their usual
\(\|\cdot\|_\alpha=\|\cdot\|_\infty+[\cdot]_\alpha\) norms.

Assume the following explicit H\"older open-system hypotheses.
\begin{enumerate}[label=(H\arabic*),leftmargin=*]
  \item The natural extension equilibrium state
    \(\widehat\nu_\psi\) has a H\"older one-step backward
    specification
    \[
      G(a\mid \xi^-)
      :=\widehat\nu_\psi(y_1=a\mid y_{(-\infty,0]}=\xi^-),
    \]
    where \(G(a\mid\cdot)\in\mathcal H_\alpha^-\) for every block symbol
    \(a\), \(G(a\mid\xi^-)=0\) unless the transition from the terminal
    symbol of \(\xi^-\) to \(a\) is admissible, and
    \(\sum_aG(a\mid\xi^-)=1\).
  \item If \(\mathcal S\) is the set of safe block symbols and
    \[
      h^-(\xi^-):=\sum_{a\notin\mathcal S}G(a\mid\xi^-),
      \qquad
      (\mathcal K_{\psi,H^{(\ell)}}f)(\xi^-)
      :=\sum_{a\in\mathcal S}G(a\mid\xi^-)f(\xi^-a),
    \]
    then \(h^-\in\mathcal H_\alpha^-\),
    \(\mathcal K_{\psi,H^{(\ell)}}:\mathcal H_\alpha^-\to
    \mathcal H_\alpha^-\) is bounded, and the killed backward operator
    has a H\"older power bound: there are constants
    \(C_\ast<\infty\) and \(0<\sigma_\ast<1\) such that
    \[
      \|\mathcal K_{\psi,H^{(\ell)}}^n f\|_\alpha
      \leq C_\ast\sigma_\ast^n\|f\|_\alpha
      \qquad(n\geq0,\ f\in\mathcal H_\alpha^-).
    \]
  \item The forward open Ruelle operator
    \[
      \mathcal L_{\psi,H^{(\ell)}}f
      :=\mathcal L_\psi(\mathbf 1_{\mathcal S}f)
    \]
    is bounded and quasi-compact on \(\mathcal H_\alpha^+\).  Its leading
    open resonance is
    \(\lambda_H=e^{-\gamma_H(\varphi)}\).  This resonance is simple and
    isolated, its spectral projector is positive, and for some
    \(\eta>0\) both the essential spectral radius and the rest of the
    spectrum have modulus at most
    \(e^{-(\gamma_H(\varphi)+\eta)}\).
  \item With the natural-extension pairing
    \[
      \langle f^-,f^+\rangle
      :=\int f^- f^+\,\dd\widehat\nu_\psi,
    \]
    the operators satisfy the represented adjoint relation
    \[
      \langle\mathcal K_{\psi,H^{(\ell)}}f^-,f^+\rangle
      =
      \langle f^-,\mathcal L_{\psi,H^{(\ell)}}f^+\rangle
      \qquad
      (f^-\in\mathcal H_\alpha^-,\ f^+\in\mathcal H_\alpha^+),
    \]
    and this represented adjoint identifies the finite-dimensional Riesz
    spectral subspaces for all isolated non-zero spectral values outside
    the common essential spectral radius.
  \item For the first-hit functions
    \[
      p_r^-:=
      \sum_{\substack{k\geq1\\ r+k\in R\ {\rm mod}\ q}}
        \mathcal K_{\psi,H^{(\ell)}}^{\,k-1}h^-,
      \qquad
      g_r^-:=\min\{p_r^-,1-p_r^-\},
    \]
    the residue ambiguity is uniformly non-degenerate: there exists
    \(\theta>0\) such that
    \[
      \theta\leq g_r^-(\xi^-)\leq\frac12
      \qquad
      (\xi^-\in (Y^{(\ell)})^-,\ r\in\ZZ/q\ZZ).
    \]
\end{enumerate}

For the first-entry residue event \(\mathsf P_R^{(\ell)}\) in the block
system, let
\(\varepsilon_n^{(\ell)}(\mathsf P_R^{(\ell)};\nu_{\varphi^{(\ell)}})\)
denote the depth-\(n\) Bayes error computed from block prefixes.  Then
the following hold.
\begin{enumerate}[label=(\roman*)]
  \item For every \(m\geq \ell\),
    \[
      \varepsilon_m(\mathsf P_R;\nu_\varphi)
      =
      \varepsilon_{m-\ell+1}^{(\ell)}
        (\mathsf P_R^{(\ell)};\nu_{\varphi^{(\ell)}}).
    \]
  \item The series defining \(p_r^-\) converges absolutely in
    \(\mathcal H_\alpha^-\), and
    \[
      \|p_r^-\|_\alpha
      \leq \frac{C_\ast}{1-\sigma_\ast}\|h^-\|_\alpha,
      \qquad
      \|g_r^-\|_\alpha
      \leq 1+\frac{C_\ast}{1-\sigma_\ast}\|h^-\|_\alpha .
    \]
    With
    \[
      \Lambda^-(f)
      := \int f\,\dd\nu_{\varphi^{(\ell)}}^{-},
    \]
    one has
    \[
      \varepsilon_n^{(\ell)}
      (\mathsf P_R^{(\ell)};\nu_{\varphi^{(\ell)}})
      =
      \Lambda^-\bigl(
        \mathcal K_{\psi,H^{(\ell)}}^{\,n-1}
        g_{n-1\bmod q}^-
      \bigr)
      \qquad (n\geq 1).
    \]
  \item Let
    \[
      (\widehat{\mathcal K}F)_r
      := \mathcal K_{\psi,H^{(\ell)}}(F_{r+1}),
      \qquad
      F=(F_r)_{r\in\ZZ/q\ZZ},\ r\in\ZZ/q\ZZ,
    \]
    on the \(q\)-fold product Banach space, with indices read modulo
    \(q\).  Then the block-error generating function
    \[
      \mathcal E_{\mathsf P_R}^{(\ell)}(z)
      := \sum_{n\geq 1}
         \varepsilon_n^{(\ell)}
           (\mathsf P_R^{(\ell)};\nu_{\varphi^{(\ell)}})z^n
    \]
    satisfies
    \[
      \mathcal E_{\mathsf P_R}^{(\ell)}(z)
      = z\,\widehat\Lambda^-
          (I-z\widehat{\mathcal K})^{-1}\widehat g^-,
    \]
    where \(\widehat g^-:=(g_0^-,\ldots,g_{q-1}^-)\) and
    \(\widehat\Lambda^-(F):=\Lambda^-(F_0)\).  Consequently every pole
    coming from an isolated spectral value of \(\widehat{\mathcal K}\)
    belongs to the reciprocal \(q\)-th cyclotomic lift of
    \(\spec(\mathcal K_{\psi,H^{(\ell)}})\).
  \item The isolated spectrum of
    \(\mathcal K_{\psi,H^{(\ell)}}\) agrees, under the
    represented adjoint hypothesis (H4), with the isolated non-zero
    spectrum of the usual forward open Ruelle operator
    \(\mathcal L_{\varphi,H}\).  Hence every isolated pole of the
    original generating function
    \[
      \mathcal E_{\mathsf P_R}(z)
      := \sum_{m\geq 1}
         \varepsilon_m(\mathsf P_R;\nu_\varphi)z^m
    \]
    belongs to the reciprocal \(q\)-th cyclotomic lift of the open
    Ruelle spectrum.
  \item Under the spectral gap in (H3), there exist positive constants
    \(c_r\) such that
    \[
      \varepsilon_m(\mathsf P_R;\nu_\varphi)
      = c_{m-\ell\bmod q}\,e^{-m\gamma_H(\varphi)}
        + O\bigl(e^{-m(\gamma_H(\varphi)+\eta)}\bigr)
      \qquad (m\to\infty).
    \]
\end{enumerate}
\leanverified{paper_scan_projection_address_holder_open_resonance}
\end{theorem}

\begin{proof}
Part~(i) is immediate from the standard \(\ell\)-block conjugacy:
reading an \(m\)-prefix in the original coding is equivalent to reading
an \((m-\ell+1)\)-prefix in the block system, and the first-entry times
into \(H\) and \(H^{(\ell)}\) agree.

For part~(ii), the \(k\)-th summand
\(\mathcal K_{\psi,H^{(\ell)}}^{\,k-1}h^-\) is the conditional
probability, given the present past history, of making \(k-1\) safe
block transitions and then entering the hole at the next transition.
Indeed, expanding \(\mathcal K_{\psi,H^{(\ell)}}^{\,k-1}h^-\) gives the
sum over all admissible safe words \(a_1,\ldots,a_{k-1}\) of
\[
  G(a_1\mid\xi^-)
  G(a_2\mid\xi^-a_1)\cdots
  G(a_{k-1}\mid\xi^-a_1\cdots a_{k-2})
  h^-(\xi^-a_1\cdots a_{k-1}),
\]
which is exactly the \(g\)-kernel probability of that first-hit
cylinder.  These first-hit cylinders are disjoint as \(k\) varies, and
their union is the event whose first entry has the specified residue.

Hypothesis (H2) gives the summable H\"older-norm estimate
\[
  \sum_{\substack{k\geq1\\ r+k\in R\ {\rm mod}\ q}}
  \|\mathcal K_{\psi,H^{(\ell)}}^{\,k-1}h^-\|_\alpha
  \leq
  \sum_{k\geq1}C_\ast\sigma_\ast^{k-1}\|h^-\|_\alpha
  =
  \frac{C_\ast}{1-\sigma_\ast}\|h^-\|_\alpha .
\]
Thus the first-hit series for \(p_r^-\) converges absolutely in
\(\mathcal H_\alpha^-\), with the displayed bound.  The map
\(t\mapsto\min\{t,1-t\}\) is \(1\)-Lipschitz on \([0,1]\), so
\(g_r^-=\min\{p_r^-,1-p_r^-\}\) is H\"older and satisfies the stated
norm estimate.

It remains to identify the Bayes error.  A block prefix that has already
met \(H^{(\ell)}\) decides the first-entry residue event and contributes
zero.  On a safe block prefix of length \(n\), the conditional
probability of \(\mathsf P_R^{(\ell)}\) depends on the prefix only
through the terminal past history and is
\(p_{n-1\bmod q}^-\).  Hence the conditional Bayes loss on that prefix
is \(g_{n-1\bmod q}^-\).  Integrating over safe prefixes is precisely
iteration of the killed backward kernel, and therefore
\[
  \varepsilon_n^{(\ell)}
  (\mathsf P_R^{(\ell)};\nu_\psi)
  =
  \Lambda^-\bigl(
    \mathcal K_{\psi,H^{(\ell)}}^{\,n-1}g_{n-1\bmod q}^-
  \bigr).
\]

For part~(iii), define \(\widehat{\mathcal K}\) on the \(q\)-fold
product exactly as in the finite-state argument.  Then
\((\widehat{\mathcal K}^n\widehat g^-)_0
=\mathcal K_{\psi,H^{(\ell)}}^{\,n}g_{n\bmod q}^-\), so summing the
geometric series gives the stated resolvent formula.  Since
\(\widehat{\mathcal K}=S_q^{-1}\otimes
\mathcal K_{\psi,H^{(\ell)}}\), its spectrum is the \(q\)-th cyclotomic
lift of \(\spec(\mathcal K_{\psi,H^{(\ell)}})\).  Thus any isolated
spectral value of \(\widehat{\mathcal K}\) that appears as a pole of the
scalar resolvent coefficient has the stated reciprocal form.

For part~(iv), first verify the adjoint relation on cylinder functions.
Disintegrating \(\widehat\nu_\psi\) over past histories and using the
one-step kernel \(G(a\mid\xi^-)\) gives
\[
  \langle\mathcal K_{\psi,H^{(\ell)}}f^-,f^+\rangle
  =
  \langle f^-,\mathcal L_{\psi,H^{(\ell)}}f^+\rangle .
\]
Cylinder functions identify the natural-extension pairing, and (H4)
assumes this identity as a continuous represented adjoint relation on
\(\mathcal H_\alpha^-\times\mathcal H_\alpha^+\).  The second clause of
(H4) is the explicit Banach-space spectral
identification needed here: the represented adjoint pairs the
finite-dimensional Riesz ranges for isolated non-zero spectral values,
so the isolated spectra of the past and future open operators agree
there.  Block recoding changes only finitely many initial coefficients
of the generating function and does not change the open pressure or the
isolated resonance set, so part~(i) transfers the pole statement back to
the original generating function.

For part~(v), hypotheses (H3) and (H4) transfer the leading forward
Riesz projector and spectral gap to the backward operator.  Thus
\[
  \mathcal K_{\psi,H^{(\ell)}}^n
  = \lambda_H^n\Pi_H^- + (R_H^-)^n,
  \qquad
  \lambda_H=e^{-\gamma_H(\varphi)},
\]
with \(\|(R_H^-)^n\|=O(e^{-n(\gamma_H(\varphi)+\eta)})\).  Apply this
decomposition to the representation in part~(ii).  The uniform residue
ambiguity in (H5)
gives \(g_r^-\geq\theta\) on the safe-history space, and positivity of
the leading projector in (H3), transferred to the past operator by (H4),
implies that
\[
  d_r:=\Lambda^-(\Pi_H^- g_r^-)
\]
is strictly positive for every residue class \(r\).  Therefore
\[
  \varepsilon_n^{(\ell)}
  (\mathsf P_R^{(\ell)};\nu_\psi)
  = d_{n-1\bmod q}\,e^{-n\gamma_H(\varphi)}
    + O\bigl(e^{-n(\gamma_H(\varphi)+\eta)}\bigr)
\]
for suitable positive constants \(d_r\).  Finally, use part~(i) with
\(n=m-\ell+1\) and absorb the bounded shift \(\ell-1\) into the
constants.
\end{proof}

\subsection{An explicit example: the doubling map and Fibonacci escape}

\begin{proposition}[Doubling map and the Fibonacci escape law]
\label{prop:doubling-fibonacci-escape}
Let \(T(x)=2x\bmod 1\) on \([0,1)\), equipped with Lebesgue measure
\(\Leb\), and let \(W=[1/2,1)\).  The binary itinerary of \(x\) gives
the full one-sided two-shift with the Bernoulli \((1/2,1/2)\) measure.
Let
\[
  H=[3/4,1),
\]
so that \(H\) corresponds to the cylinder \([11]\).  Define
\[
  \tau_H(x):=\inf\set{n\geq 0:T^n x\in H}
\]
and
\[
  \mathsf{P}_{\mathrm{even}}
  := \set{x:\tau_H(x)<\infty,\ \tau_H(x)\equiv 0 \pmod 2}.
\]
If \((F_n)_{n\geq 0}\) denotes the Fibonacci sequence
\(F_0=0\), \(F_1=1\), then for every \(m\geq 1\),
\[
  \varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)
  = \frac{F_{m+3}}{5\,2^m}.
\]
Moreover,
\[
  \Leb(\tau_H\geq m)=\frac{F_{m+3}}{2^{m+1}},
  \qquad
  \varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)
  = \frac25\,\Leb(\tau_H\geq m),
\]
and therefore
\[
  \lim_{m\to\infty}
  -\frac1m\log\varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)
  = \log\frac{2}{\varphi}
  = \log 2-\log\varphi.
\]
Equivalently, the survivor entropy of the golden-mean subshift is
recovered from finite-observation error by
\[
  h_{\mathrm{top}}(Y_H)
  = \log\varphi
  = \log 2
    + \lim_{m\to\infty}\frac1m
      \log \varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb).
\]
\end{proposition}

\begin{proof}
In binary coding, \(\tau_H\) is the first index at which the block
\(11\) appears.  A length-\(m\) prefix contributes to
\(\varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)\) if and only if it
contains no occurrence of \(11\) at positions \(0,\ldots,m-2\).  Let
\begin{align*}
  G_m^0
  &:= \set{a\in\{0,1\}^m:
           a \text{ avoids }11,\ a_{m-1}=0}, \\
  G_m^1
  &:= \set{a\in\{0,1\}^m:
           a \text{ avoids }11,\ a_{m-1}=1}.
\end{align*}
The usual golden-mean recursion gives
\[
  \card{G_m^0}=F_{m+1},
  \qquad
  \card{G_m^1}=F_m.
\]
Every length-\(m\) cylinder has Lebesgue measure \(2^{-m}\).

Let
\begin{align*}
  a
  &:= \Leb(\mathsf{P}_{\mathrm{even}}
           \mid \text{safe prefix of even length ending in }0), \\
  b
  &:= \Leb(\mathsf{P}_{\mathrm{even}}
           \mid \text{safe prefix of even length ending in }1), \\
  c
  &:= \Leb(\mathsf{P}_{\mathrm{even}}
           \mid \text{safe prefix of odd length ending in }0), \\
  d
  &:= \Leb(\mathsf{P}_{\mathrm{even}}
           \mid \text{safe prefix of odd length ending in }1).
\end{align*}
Since the next binary digit is \(0\) or \(1\) with equal probability,
\[
  a=\frac12 c+\frac12 d,
  \qquad
  b=\frac12 c,
\]
\[
  c=\frac12 a+\frac12 b,
  \qquad
  d=\frac12+\frac12 a.
\]
Solving gives
\[
  a=\frac35,\qquad b=\frac15,\qquad c=\frac25,\qquad d=\frac45.
\]
Hence the Bayes error on a safe prefix depends only on its terminal
symbol:
\[
  \beta(a)=\frac25
  \quad\text{on }G_m^0,
  \qquad
  \beta(a)=\frac15
  \quad\text{on }G_m^1.
\]
Therefore
\[
  \varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)
  = 2^{-m}\left(
      \frac25\card{G_m^0}
      + \frac15\card{G_m^1}
    \right)
  = 2^{-m}\left(
      \frac25F_{m+1}
      + \frac15F_m
    \right).
\]
Using \(F_{m+3}=2F_{m+1}+F_m\), we obtain
\[
  \varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)
  = \frac{F_{m+3}}{5\,2^m}.
\]

The condition \(\tau_H\geq m\) is equivalent to saying that the first
\(m+1\) binary digits avoid \(11\), so
\[
  \Leb(\tau_H\geq m)=\frac{F_{m+3}}{2^{m+1}}.
\]
This proves the exact factor \(\varepsilon_m=\frac25\Leb(\tau_H\geq m)\).
Finally, \(F_{m+3}\sim \varphi^{m+3}/\sqrt5\), so
\[
  \lim_{m\to\infty}
  -\frac1m\log\varepsilon_m(\mathsf{P}_{\mathrm{even}};\Leb)
  = \log\frac{2}{\varphi}.
\]
The survivor subshift is the golden-mean shift, whose topological
entropy is \(\log\varphi\), and the last identity follows.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:main-resonance}]
The exact Markov resolvent statement is
Theorem~\ref{thm:error-resolvent-cyclotomic}.  The conditional H\"older
Gibbs extension under the explicit hypotheses (H1)--(H5), including the
summable H\"older power bound, represented past--future spectral
identification, and spectral gap, is
Theorem~\ref{thm:holder-open-resonance}.
\end{proof}
