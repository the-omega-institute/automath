\section{Information states and forcing}\label{sec:information-states}

We now fix the lower semantic layer. Information states are nonempty teams of valuations equipped with a refinement relation, and forcing is simply pointwise truth holding uniformly across the state.

\subsection{Lower-layer syntax}

We record the syntax explicitly because the nondefinability argument in \Cref{thm:pointwise-irreducibility} is carried out in a constant-free reduct of the lower language.

\begin{definition}[Pointwise language $\mathcal L_0$ and forcing language $\mathcal L_1$]\label{def:lower-layer-syntax}
Fix a many-sorted signature
\[
\Sigma=(S,F,R).
\]
For a context $\Gamma=(x_1:s_1,\dots,x_n:s_n)$, the \emph{$\Gamma$-terms} of the lower language are generated by
\[
t ::= x_i \mid f(t_1,\dots,t_m),
\]
where $x_i$ has sort $s_i$ and $f$ ranges over the function symbols of $\Sigma$. The \emph{$\Gamma$-formulas} are generated by
\[
\varphi ::= (t=t') \mid Q(t_1,\dots,t_m) \mid \neg\varphi \mid (\varphi\land\psi) \mid \exists y:s\,\varphi,
\]
where $Q$ ranges over the relation symbols of $\Sigma$, and the usual derived connectives and universal quantifiers are defined as usual.

The language $\mathcal L_0$ is this ordinary many-sorted first-order language with its Tarskian semantics $M,\rho\models\varphi$. The language $\mathcal L_1$ has the same grammar; only the semantic level changes, namely from pointwise evaluation to forcing at information states.
\end{definition}

Thus $\mathcal L_0$ and $\mathcal L_1$ differ only semantically: formulas of $\mathcal L_0$ are evaluated at single valuations, whereas formulas of $\mathcal L_1$ must hold uniformly across all valuations in a state.

\subsection{States, refinement, and forcing}

Fix a many-sorted first-order signature
\[
\Sigma=(S,F,R).
\]
For a model $M$, write $D_s^M$ for the interpretation of a sort $s\in S$. A \emph{context} is a finite typed list
\[
\Gamma=(x_1:s_1,\dots,x_n:s_n),
\]
and its realization set in $M$ is
\[
\Val_M(\Gamma):=\{\rho\mid \rho(x_i)\in D_{s_i}^M\text{ for }1\le i\le n\}.
\]
If $\Gamma\preceq\Delta$, every $\sigma\in\Val_M(\Delta)$ restricts to an element $\sigma|_{\Gamma}\in\Val_M(\Gamma)$.

\begin{definition}[Information state]\label{def:information-state}
An \emph{information state} in $M$ is a pair
\[
p=(\Gamma_p,R_p)
\]
such that $\Gamma_p$ is a context and $\varnothing\neq R_p\subseteq \Val_M(\Gamma_p)$. The set of information states in $M$ is denoted by $\St(M)$.
\end{definition}

\begin{definition}[Refinement]\label{def:refinement}
For states $p=(\Gamma_p,R_p)$ and $q=(\Gamma_q,R_q)$ in the same model, write $q\le p$ if
\begin{enumerate}[label=(\roman*),nosep]
\item $\Gamma_p\preceq\Gamma_q$,
\item every $\sigma\in R_q$ satisfies $\sigma|_{\Gamma_p}\in R_p$.
\end{enumerate}
\end{definition}

Refinement therefore combines two monotonicities: the context may grow, while the set of admissible realizations may shrink.

\begin{definition}[State forcing]\label{def:state-forcing}
Let $p=(\Gamma_p,R_p)\in\St(M)$ and let $\varphi$ be a formula well formed in context $\Gamma_p$. We define
\[
M,p\Vdash\varphi
\quad:\Longleftrightarrow\quad
\forall \rho\in R_p\;(M,\rho\models\varphi).
\]
\end{definition}

\begin{definition}[Singleton embedding]\label{def:singleton-embedding}
Let $\Gamma$ be a context in $M$ and let $\rho\in \Val_M(\Gamma)$. The \emph{singleton embedding} of $\rho$ is the state
\[
i_M(\rho):=(\Gamma,\{\rho\})\in\St(M).
\]
\end{definition}

\begin{proposition}[Flat team-style reading of lower-layer forcing]\label{prop:flat-team-reading}
Let $p=(\Gamma_p,R_p)$ be an information state. If one reads $R_p$ as a nonempty team of $\Gamma_p$-valuations, then for every lower-layer formula $\varphi$ well formed in context $\Gamma_p$,
\[
M,p\Vdash\varphi
\quad\Longleftrightarrow\quad
\forall \rho\in R_p\;(M,\rho\models\varphi).
\]
\leanverified{paper_information_state_flat_team_reading_seeds}
\end{proposition}

\begin{proof}
This is just \Cref{def:state-forcing}.
\end{proof}

\begin{definition}[State theory]\label{def:state-theory}
For $p\in\St(M)$, define
\[
\Th_M(p):=\{\varphi\mid M,p\Vdash\varphi\}.
\]
\end{definition}

\begin{definition}[Decidedness]\label{def:decidedness}
A formula $\varphi$ is \emph{decided} at $p$, written $M,p\Downarrow\varphi$, if
\[
M,p\Vdash\varphi
\qquad\text{or}\qquad
M,p\Vdash\neg\varphi.
\]
\end{definition}

\begin{proposition}[Persistence of forcing]\label{prop:forcing-persistence}
If $q\le p$ and $M,p\Vdash\varphi$, then $M,q\Vdash\varphi^\uparrow$, where $\varphi^\uparrow$ is the same formula viewed in the larger context $\Gamma_q$.
\leanverified{paper_information_state_forcing_persistence_publication}
\end{proposition}

\begin{proof}
Let $\sigma\in R_q$. By \Cref{def:refinement}, its restriction $\sigma|_{\Gamma_p}$ lies in $R_p$. Since $M,p\Vdash\varphi$, we have $M,\sigma|_{\Gamma_p}\models\varphi$. The formula $\varphi$ mentions only variables from $\Gamma_p$, so the same valuation, viewed in the larger context $\Gamma_q$, satisfies $\varphi^\uparrow$. Hence $M,q\Vdash\varphi^\uparrow$.
\end{proof}

\begin{corollary}[Monotone growth of state theories]\label{cor:state-theory-monotonicity}
If $q\le p$, then $\Th_M(p)\subseteq \Th_M(q)$.
\leanverified{paper_information_state_state_theory_monotonicity_focused_seeds}
\end{corollary}

\begin{proof}
Immediate from \Cref{prop:forcing-persistence}.
\end{proof}

\begin{proposition}[Singleton conservativity]\label{prop:singleton-conservativity}
Let $\rho\in\Val_M(\Gamma)$ and let $i_M(\rho)$ be the state of \Cref{def:singleton-embedding}. Then for every $\Gamma$-formula $\varphi$,
\[
M,i_M(\rho)\Vdash\varphi
\quad\Longleftrightarrow\quad
M,\rho\models\varphi.
\]
\leanverified{paper_information_state_singleton_conservativity_seeds}
\end{proposition}

\begin{proof}
Since $R_{i_M(\rho)}=\{\rho\}$, the forcing clause reduces at once to ordinary satisfaction at $\rho$.
\end{proof}

\subsection{Lifted consequence}

\begin{definition}[State semantic consequence]\label{def:state-semantic-consequence}
For formulas $\Phi\cup\{\varphi\}$, define
\[
\Phi\models_{\mathrm{S}}\varphi
\]
to mean that for every model $M$ and every state $p\in\St(M)$,
\[
M,p\Vdash\Phi
\quad\Longrightarrow\quad
M,p\Vdash\varphi.
\]
\end{definition}

\begin{proposition}[Lifted soundness of pointwise reasoning]\label{prop:lifted-pointwise-soundness}
Let $\vdash_*$ be any derivability relation on formulas that is sound for ordinary many-sorted satisfaction. Then
\[
\Gamma\vdash_*\varphi
\quad\Longrightarrow\quad
\Gamma\models_{\mathrm{S}}\varphi.
\]
\leanverified{paper_information_state_lifted_pointwise_soundness_focused_seeds}
\end{proposition}

\begin{proof}
Assume $\Gamma\vdash_*\varphi$, and let $M$ and $p=(\Gamma_p,R_p)$ satisfy $M,p\Vdash\Gamma$. Then every $\rho\in R_p$ satisfies each formula in $\Gamma$ pointwise. Since $\vdash_*$ is sound for ordinary satisfaction, every such $\rho$ also satisfies $\varphi$. Hence $M,p\Vdash\varphi$, and therefore $\Gamma\models_{\mathrm{S}}\varphi$.
\end{proof}

\Cref{prop:lifted-pointwise-soundness} is the basic link between ordinary satisfaction and state forcing: any derivation that is sound pointwise remains sound when formulas are tested uniformly over a state.

\begin{remark}[Where the new semantics begins]
\Cref{prop:flat-team-reading} shows that forcing is flat on the lower language. The genuinely new behaviour begins only after the local-object predicates are added, since gluing failure and its visible/blind decomposition do not belong to the lower layer itself.
\end{remark}
