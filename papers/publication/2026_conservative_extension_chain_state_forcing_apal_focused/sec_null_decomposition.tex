\section{Local Objects and Gluing Failure}\label{sec:null-decomposition}

We now pass from admissible references to local objects. An admitted reference need not determine a global object, but it may still admit local witnesses and compatible local families. The distinction between these possibilities is the source of the gluing phenomena studied in this section.

\subsection{Refinement axioms}\label{sec:refinement-axioms}

The local-object layer interacts with state refinement through the following standing axioms. Whenever $q\le p$ we fix the variance convention
\[
p\longrightarrow q,
\]
meaning that every object defined at the coarser state $p$ pulls back functorially to $q$. All structural maps below are natural in chains of refinements. The axioms are used explicitly in \Cref{prop:typed-readout-persistence,thm:null-classification-under-refinement} and implicitly wherever refinement stability of visible classes is invoked.

\begin{definition}[Refinement axioms for the local-object layer]\label{def:refinement-axioms}
Let $q\le p$ be a refinement of information states. The local-object layer satisfies:
\begin{enumerate}[label=(RA\arabic*),nosep]
\item\label{ax:ref-restriction} \emph{Reference pullback:} there is a map
\[
\rho^{s}_{p\to q}:\mathcal{R}_s(p)\longrightarrow \mathcal{R}_s(q)
\]
compatible with admissibility, so that if $M,p\Vdash\Adm_s(r)$ then $M,q\Vdash\Adm_s(r)$ and $[r]_q=\rho^{s}_{p\to q}([r]_p)$.
\item\label{ax:cover-restriction} \emph{Cover pullback:} there is a functor
\[
\Phi^{s}_{p\to q}:\mathcal{B}_{p,s}\longrightarrow\mathcal{B}_{q,s}
\]
preserving identities, refinements, and covering families. Concretely, a cover $\{a_i\to a\}$ at $p$ pulls back to a cover $\{\Phi^{s}_{p\to q}(a_i)\to \Phi^{s}_{p\to q}(a)\}$ at $q$.
\item\label{ax:section-restriction} \emph{Section transport:} for every $a\in\mathcal{R}_s(p)$ with image $\rho^{s}_{p\to q}(a)$, there is a natural transformation
\[
T^{s}_{p\to q}:F_{p,s}|_{\mathcal{C}_{p,s}/a}\Longrightarrow F_{q,s}|_{\mathcal{C}_{q,s}/\rho^{s}_{p\to q}(a)}
\]
compatible with the restriction maps of the two slice sites. Consequently, a compatible local family at $p$ pulls back to a compatible local family at $q$.
\item\label{ax:vis-stability} \emph{Visible class transport:} there is a well-defined induced map
\[
\Vis_{p\to q}:\Vis_{p,s}(r)\longrightarrow \Vis_{q,s}(r)
\]
defined by pulling a representative covering family along \ref{ax:cover-restriction} and transporting its sections via \ref{ax:section-restriction}. Whenever $\Vis_{p,s}(r)$ is a singleton, $\Vis_{p\to q}$ sends that unique class to the unique class at $q$.
\end{enumerate}
\end{definition}

We assume \ref{ax:ref-restriction}--\ref{ax:vis-stability} throughout the remainder of this section.

\subsection{References and local objects}

Fix a family of referenceable sorts
\[
S_{\mathrm{ref}}\subseteq S.
\]
For each $s\in S_{\mathrm{ref}}$, the language contains a reference sort $\mathsf{Ref}_s$ and an admissibility predicate $\Adm_s(\cdot)$.

\begin{definition}[Reference classes]\label{def:reference-classes}
Let $p$ be a state and let $r,r'$ be reference terms of sort $\mathsf{Ref}_s$. Define
\[
r\equiv_p r'
\quad:\Longleftrightarrow\quad
M,p\Vdash (r=r').
\]
The set of admitted reference classes at $p$ is
\[
\mathcal{R}_s(p):=\{[r]_p\mid M,p\Vdash \Adm_s(r)\}.
\]
\end{definition}

\begin{definition}[Local reference frame]\label{def:local-reference-frame}
For a state $p$ and a referenceable sort $s$, a \emph{local reference frame} is a triple
\[
\mathcal{B}_{p,s}=\bigl(\mathcal{R}_s(p),\sqsubseteq_{p,s},\Cov_{p,s}\bigr),
\]
where $\sqsubseteq_{p,s}$ is a preorder on $\mathcal{R}_s(p)$ and $\Cov_{p,s}$ assigns to each $a\in\mathcal{R}_s(p)$ a collection of covering families
\[
\{a_i\to a\}_{i\in I},
\qquad
a_i\sqsubseteq_{p,s} a.
\]
These covering families are required to satisfy:
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{identity:} $\{a\to a\}\in\Cov_{p,s}(a)$ for every $a$;
\item \emph{pullback stability:} if $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and $b\sqsubseteq_{p,s} a$, then there exists a covering family
\[
\{b_{ij}\to b\}_{i\in I,\;j\in J_i}\in\Cov_{p,s}(b)
\]
such that $b_{ij}\sqsubseteq_{p,s} a_i$ for all $i,j$;
\item \emph{transitivity:} if $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and, for each $i\in I$,
\[
\{b_{ij}\to a_i\}_{j\in J_i}\in\Cov_{p,s}(a_i),
\]
then
\[
\{b_{ij}\to a\}_{i\in I,\;j\in J_i}\in\Cov_{p,s}(a).
\]
\end{enumerate}
\end{definition}

\begin{proposition}[The cover relation defines a Grothendieck pretopology]\label{prop:local-reference-pretopology}
Let $\mathcal{B}_{p,s}$ be the preorder category whose objects are the admitted reference classes in $\mathcal{R}_s(p)$ and whose morphisms are the refinement arrows
\[
b\to a
\quad:\Longleftrightarrow\quad
b\sqsubseteq_{p,s} a.
\]
Then the covering families $\Cov_{p,s}$ form a Grothendieck pretopology on $\mathcal{B}_{p,s}$.
\end{proposition}

\begin{proof}
See Appendix~\ref{app:deferred-proofs}.
\end{proof}

Accordingly, we use the same symbol $\mathcal{B}_{p,s}$ for the resulting preorder category in the presheaf and sheaf constructions below.

\begin{definition}[Local object functor]\label{def:local-object-functor}
A \emph{local object functor} on $(p,s)$ is a contravariant functor
\[
F_{p,s}:\mathcal{B}_{p,s}^{op}\to\mathsf{Set}.
\]
If $b\sqsubseteq_{p,s} a$, the induced restriction map is denoted by
\[
\res_b^a:F_{p,s}(a)\to F_{p,s}(b).
\]
\end{definition}

\begin{definition}[Three existence levels]\label{def:existence-levels}
Let $r:\mathsf{Ref}_s$ and assume $M,p\Vdash \Adm_s(r)$. Write $a=[r]_p$.
\begin{enumerate}[label=(\roman*),nosep]
\item $M,p\Vdash \LocSec_s(r)$ if there exists a cover $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ such that $F_{p,s}(a_i)\neq\varnothing$ for every $i\in I$.
\item $M,p\Vdash \CompSec_s(r)$ if there exist a cover $\{a_i\to a\}_{i\in I}$ and sections $\sigma_i\in F_{p,s}(a_i)$ such that for every common refinement $c\sqsubseteq_{p,s} a_i,a_j$,
\[
\res_c^{a_i}(\sigma_i)=\res_c^{a_j}(\sigma_j).
\]
\item $M,p\Vdash \Sec_s(r)$ if $F_{p,s}(a)\neq\varnothing$.
\end{enumerate}
\end{definition}

\begin{proposition}[Global sections yield compatible local sections]\label{prop:global-implies-compatible-local}
If $M,p\Vdash \Sec_s(r)$, then
\[
M,p\Vdash \CompSec_s(r)
\qquad\text{and}\qquad
M,p\Vdash \LocSec_s(r).
\]
\end{proposition}

\begin{proof}
See Appendix~\ref{app:deferred-proofs}.
\end{proof}

\begin{example}[Toy behaviors of $\LocSec$, $\CompSec$, and $\Sec$]\label{ex:toy-loc-comp-sec}
Let $p$ be a state with one referenceable sort $s$ and a local reference frame generated by a root $a$, two refinements $u,v\sqsubseteq_{p,s} a$ that cover $a$, and their common refinement $w\sqsubseteq_{p,s} u,v$. Define three references $r_{\mathrm{loc}}, r_{\mathrm{cmp}}, r_{\mathrm{glue}}$ that share this frame but differ in their presheaves:
\begin{itemize}[nosep]
\item $F_{p,s}^{\mathrm{loc}}(u)=F_{p,s}^{\mathrm{loc}}(v)=\varnothing$, so $\LocSec_s(r_{\mathrm{loc}})$ already fails;
\item Let
\[
F_{p,s}^{\mathrm{cmp}}(u)=\{\sigma_u\},\qquad
F_{p,s}^{\mathrm{cmp}}(v)=\{\sigma_v\},\qquad
F_{p,s}^{\mathrm{cmp}}(w)=\{0,1\},
\]
with $\res_w^u(\sigma_u)=0$ and $\res_w^v(\sigma_v)=1$.
Then local witnesses exist but are incompatible on $w$, so $\LocSec_s(r_{\mathrm{cmp}})$ holds and $\CompSec_s(r_{\mathrm{cmp}})$ fails;
\item $F_{p,s}^{\mathrm{glue}}(u)=F_{p,s}^{\mathrm{glue}}(v)=F_{p,s}^{\mathrm{glue}}(w)=\{*\}$ while $F_{p,s}^{\mathrm{glue}}(a)=\varnothing$.
Then both $\LocSec_s(r_{\mathrm{glue}})$ and $\CompSec_s(r_{\mathrm{glue}})$ hold, but $\Sec_s(r_{\mathrm{glue}})$ fails.
\end{itemize}
This toy picture isolates the three semantic levels before any gerbe machinery is invoked: absence may arise from missing local sections, incompatible local sections, or genuine gluing failure.
\end{example}

\subsection{Sheafification and gluing failure}

By \Cref{prop:local-reference-pretopology}, for fixed $(p,s)$ the local reference frame of \Cref{def:local-reference-frame} determines a site
\[
\mathcal{C}_{p,s}:=(\mathcal{B}_{p,s},\Cov_{p,s}),
\]
and the local object functor of \Cref{def:local-object-functor} determines a presheaf
\[
F_{p,s}\in\mathbf{PSh}(\mathcal{B}_{p,s}).
\]
Write
\[
a_{p,s}:\mathbf{PSh}(\mathcal{B}_{p,s})\to\mathbf{Sh}(\mathcal{C}_{p,s})
\]
for the associated sheafification functor.

\begin{proposition}[Sheafification characterizes compatible local sectionability]\label{thm:sheafification-characterization}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and write $a=[r]_p$. Then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
\bigl(a_{p,s}F_{p,s}\bigr)(a)\neq\varnothing.
\]
\end{proposition}

\begin{proof}
See Appendix~\ref{app:deferred-proofs}.
\end{proof}

\begin{remark}\label{rem:sheafification-standard}
\Cref{thm:sheafification-characterization} is a standard fact of sheaf theory, included here to fix notation.
\end{remark}

\begin{corollary}[Sheaf condition at a reference]\label{cor:sheaf-local-to-global}
If $F_{p,s}$ is already a sheaf at $a=[r]_p$, then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r).
\]
\end{corollary}

\begin{proof}
If $F_{p,s}$ is already a sheaf at $a$, then the unit $F_{p,s}\to a_{p,s}F_{p,s}$ is an isomorphism at $a$. The claim therefore follows from \Cref{thm:sheafification-characterization}.
\end{proof}

\begin{corollary}[Sheafification removes gluing failure]\label{cor:sheafification-removes-glue}
Let $\Sec_s^{\sharp}(r)$ denote global sectionability computed with the sheafified local object functor $a_{p,s}F_{p,s}$. Then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s^{\sharp}(r).
\]
In particular, no admitted reference remains in gluing failure after sheafification.
\end{corollary}

\begin{proof}
This is \Cref{thm:sheafification-characterization} rewritten for the sheaf $a_{p,s}F_{p,s}$. Once the presheaf has been replaced by a sheaf, compatible local existence and global existence coincide.
\end{proof}

\subsection{Realization prestacks and connected components}

The presheaf $F_{p,s}$ records only the isomorphism classes of local realizations. To pass from compatible local families to gerbes and cohomology, one works with a chosen groupoid-valued lift whose connected components recover $F_{p,s}$ after sheafification. For separated $F_{p,s}|_{\mathcal C_a}$, the split construction below provides such a lift. For a general presheaf, however, the existence of a banded prestack lift is not automatic. The substantive issue for the later semantics is therefore twofold: first, to specify a lift when one is available, and second, to require that its stackification still reflects the intended gluing semantics at the terminal fibre.

\begin{definition}[Realization prestack and stackification]\label{def:realization-prestack}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, and write
\[
a=[r]_p,
\qquad
\mathcal C_a:=\mathcal C_{p,s}/a.
\]
A \emph{realization prestack} for $r$ at $p$ is a category fibred in groupoids
\[
\mathfrak P_r\to \mathcal C_a
\]
such that:
\begin{enumerate}[label=(\roman*),nosep]
\item the presheaf of isomorphism classes
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r):U\longmapsto \Ob(\mathfrak P_r(U))/\!\cong
\]
is identified with the restriction of the local object functor,
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r)\cong F_{p,s}|_{\mathcal C_a};
\]
\item for every object $U$ of $\mathcal C_a$ and every $x\in \mathfrak P_r(U)$, the automorphism sheaf
\[
\underline{\mathrm{Aut}}(x)
\]
on $\mathcal C_a/U$ is functorially identified with the restriction of a fixed abelian sheaf $\Band$ on $\mathcal C_a$.
\end{enumerate}
Write
\[
\mathfrak L_r:=a(\mathfrak P_r)
\]
for the stackification of $\mathfrak P_r$ in groupoids \cite[Tag 02ZP]{StacksProject}.
\end{definition}

\begin{proposition}[Canonical split realization prestack]\label{prop:split-realization-prestack}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write
\[
a=[r]_p,
\qquad
\mathcal C_a:=\mathcal C_{p,s}/a,
\]
and let $\Band$ be any abelian sheaf on $\mathcal C_a$. Assume that the restriction of the local object functor $F_{p,s}|_{\mathcal C_a}$ is a \emph{separated} presheaf, that is, for every cover $\{U_i\to U\}$ in $\mathcal C_a$ and every pair $\sigma,\tau\in F_{p,s}(U)$,
\[
\res_{U_i}^{U}(\sigma)=\res_{U_i}^{U}(\tau)\text{ for all }i
\quad\Longrightarrow\quad
\sigma=\tau.
\]
Then there exists a realization prestack
\[
\mathfrak P_r^{\mathrm{split}}\to \mathcal C_a
\]
with band $\Band$ such that
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r^{\mathrm{split}})
\cong
F_{p,s}|_{\mathcal C_a}.
\]
Explicitly, for each object $U$ of $\mathcal C_a$, the fibre
\[
\mathfrak P_r^{\mathrm{split}}(U)
\]
has object set $F_{p,s}(U)$ and morphism sets
\[
\mathrm{Hom}_{\mathfrak P_r^{\mathrm{split}}(U)}(\sigma,\tau)
:=
\begin{cases}
\Band(U), & \sigma=\tau,\\
\varnothing, & \sigma\neq\tau,
\end{cases}
\]
with composition given by the group law on $\Band(U)$.
\end{proposition}

\begin{proof}
For a morphism $f:V\to U$ in $\mathcal C_a$, define pullback on objects by
\[
f^*(\sigma):=\res_V^U(\sigma)\in F_{p,s}(V)
\]
and on automorphisms by the sheaf restriction
\[
f^*:\Band(U)\to \Band(V).
\]
This gives a pseudofunctor $\mathcal C_a^{op}\to \mathbf{Grpd}$ and hence a category fibred in groupoids over $\mathcal C_a$.

We must verify the prestack condition: for any two objects $\sigma,\tau\in \mathfrak P_r^{\mathrm{split}}(U)$, the presheaf
\[
\mathrm{Isom}(\sigma,\tau):V\longmapsto \mathrm{Hom}_{\mathfrak P_r^{\mathrm{split}}(V)}(\res_V^U(\sigma),\res_V^U(\tau))
\]
on $\mathcal C_a/U$ must be a sheaf. If $\sigma=\tau$, then $\mathrm{Isom}(\sigma,\sigma)\cong \Band|_U$, which is a sheaf. If $\sigma\neq\tau$, then $\mathrm{Isom}(\sigma,\tau)(V)$ is $\Band(V)$ when $\res_V^U(\sigma)=\res_V^U(\tau)$ and is $\varnothing$ otherwise. Let $\{V_i\to V\}$ be a cover and suppose that $\mathrm{Isom}(\sigma,\tau)$ has matching local sections, i.e.\ $\res_{V_i}^U(\sigma)=\res_{V_i}^U(\tau)$ for every $i$. The separatedness hypothesis on $F_{p,s}|_{\mathcal C_a}$ then forces $\res_V^U(\sigma)=\res_V^U(\tau)$, so $\mathrm{Isom}(\sigma,\tau)(V)=\Band(V)$, and the matching family glues because $\Band$ is a sheaf. Conversely, if $\mathrm{Isom}(\sigma,\tau)(V)=\varnothing$, there are no local sections to glue. In both cases, the sheaf condition is satisfied. Hence $\mathfrak P_r^{\mathrm{split}}$ is a prestack.

By construction, two objects of $\mathfrak P_r^{\mathrm{split}}(U)$ are isomorphic if and only if they are equal as elements of $F_{p,s}(U)$. Therefore
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r^{\mathrm{split}})(U)\cong F_{p,s}(U)
\]
for every $U$, giving condition \textup{(i)} of \Cref{def:realization-prestack}. Condition \textup{(ii)} holds because the automorphism sheaf of every object $\sigma$ is exactly $\Band|_U$.
\end{proof}

\begin{remark}[Why separatedness is needed]\label{rem:separatedness-needed}
The separatedness hypothesis cannot be dropped. If $F_{p,s}|_{\mathcal C_a}$ is not separated, two distinct global sections $\sigma\neq\tau$ may become equal on a cover, so $\mathrm{Isom}(\sigma,\tau)$ would have local sections without a global one, violating the sheaf condition. For a non-separated $F_{p,s}|_{\mathcal C_a}$, one should either pass to the separated presheaf $F_{p,s}^+$ first or replace the split construction by a weaker fibred category and reprove the stackification statements under that weaker setup.
\end{remark}

\begin{proposition}[Components of the stackification]\label{prop:stackification-components}
In the setting of \Cref{def:realization-prestack}, there is a canonical isomorphism of sheaves on $\mathcal C_a$:
\[
\pi_0(\mathfrak L_r)\cong a_{p,s}F_{p,s}|_{\mathcal C_a}.
\]
\end{proposition}

\begin{proof}
See Appendix~\ref{app:deferred-proofs}.
\end{proof}

\begin{definition}[Global conservativity at the terminal fibre]\label{def:global-conservativity}
In the setting of \Cref{def:realization-prestack}, let
\[
\iota_r:\mathfrak P_r\to \mathfrak L_r:=a(\mathfrak P_r)
\]
be the stackification morphism. We say that $\mathfrak P_r$ is \emph{globally conservative at $a$} if the induced functor on the terminal fibre
\[
(\iota_r)_a:(\mathfrak P_r)_a\to (\mathfrak L_r)_a
\]
is essentially surjective.
\end{definition}

\begin{proposition}[No new global objects at $a$]\label{prop:no-new-global-objects}
Assume global conservativity at $a$ in the sense of \Cref{def:global-conservativity}. Then
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\mathfrak L_r(a)\neq\varnothing.
\]
\end{proposition}

\begin{proof}
See Appendix~\ref{app:deferred-proofs}.
\end{proof}

\begin{remark}[What remains conditional]\label{rem:gluing-sensitive-lifts}
\Cref{prop:split-realization-prestack} provides a split lift for every separated presheaf $F_{p,s}|_{\mathcal C_a}$ and fixed band. The later obstruction theory needs more: a lift whose stackification is globally conservative at $a$ and whose component gerbes admit the cohomological description used below. Accordingly, the branch obstruction classes of \Cref{sec:gerbe-obstruction} are attached to such a chosen lift and its band.
\end{remark}

\subsection{Assumptions on the chosen-lift formalism}\label{subsec:scope-chosen-lift}

For later use we collect the verifications that enter the branch-obstruction analysis.

Given an admitted reference $r:\mathsf{Ref}_s$, the chosen-lift formalism applies once the following steps have been carried out.
\begin{enumerate}[label=(\arabic*),nosep]
\item (\emph{Separated presheaf}) Check that $F_{p,s}|_{\mathcal C_a}$ is separated on the slice site of a reference class $a=[r]_p$ so that the split prestack of \Cref{prop:split-realization-prestack} exists.
\item (\emph{Gluing-sensitive lift}) Choose a realization prestack $\mathfrak P_r$ whose stackification $\mathfrak L_r$ is globally conservative at $a$ (\Cref{def:global-conservativity}).
\item (\emph{Finite presentation}) Produce a finite cover $\mathcal U$ with finite nerve $N(\mathcal U)$ and trivialize the band on $\mathcal U$ so that \Cref{def:finite-nerve-presentation} applies.
\item (\emph{Branch hypotheses}) When branch constancy or visibility invariance is needed, verify the bijectivity condition of \Cref{def:branch-constancy} on the chosen cover.
\end{enumerate}
Section~\ref{sec:homological-visibility} turns such a presentation into concrete homological data.

The torus-type construction proved later in \Cref{ex:end-to-end} supplies an explicit instance of this scheme. There the local object functor is separated because every proper intersection carries a singleton fibre; the chosen gluing-sensitive prestack is globally conservative at the terminal fibre because both terminal fibres are empty; the band is constant on a finite good cover, so a finite nerve presentation exists; and the unique visible branch restricts to the unique connected component on every simplex, so branch constancy holds.

A similar pattern appears for finitely generated Kripke frames with Beth semantics \cite{Beth1956,Kripke1965}. If $r$ records admissible partial sections of a predicate, then the associated local object functor is a presheaf on the Alexandrov site of the frame and is separated by monotonicity. Promoting local sections to slice groupoids with identity automorphisms gives a realization prestack that is globally conservative at the limit node, finite antichain covers yield finite nerves, and the finiteness of the maximal nodes forces bijectivity on connected components. Thus the same hypotheses arise in a standard semantic example unrelated to contextuality.

\subsection{Visible value classes and typed readouts}

\begin{definition}[Visible value classes]\label{def:visible-value-classes}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and write $a=[r]_p$. A \emph{visible presentation} of a value at $r$ is a compatible local family
\[
\bigl(\{a_i\to a\}_{i\in I},(\sigma_i)_{i\in I}\bigr)
\]
as in \Cref{def:existence-levels}(ii). Two visible presentations are called \emph{value-equivalent} if after passage to a common refinement of their covers, the restricted local sections agree pointwise. A \emph{visible value class} is an equivalence class of visible presentations. The set of visible value classes of $r$ at $p$ is denoted by $\Vis_{p,s}(r)$.
\end{definition}

\begin{proposition}[Visible value classes as global connected components]\label{thm:visible-value-components}
Assume that $r$ admits a realization prestack in the sense of \Cref{def:realization-prestack}. Then
\[
\Vis_{p,s}(r)\cong \bigl(a_{p,s}F_{p,s}\bigr)(a)\cong \pi_0(\mathfrak L_r)(a).
\]
\end{proposition}

\begin{proof}
See Appendix~\ref{app:deferred-proofs}.
\end{proof}

\begin{definition}[Typed readout]\label{def:typed-readout}
For a reference $r:\mathsf{Ref}_s$ at a state $p$, define the typed readout
\[
\Read_{p,s}(r)
\]
by the following cases:
\[
\Read_{p,s}(r)=
\begin{cases}
\text{undefined}, & M,p\nVdash \Adm_s(r),\\[0.5ex]
\Null, & M,p\Vdash \Adm_s(r)\ \text{and}\ \Vis_{p,s}(r)=\varnothing,\\[0.5ex]
v, & \Vis_{p,s}(r)=\{v\},\\[0.5ex]
\text{undefined}, & |\Vis_{p,s}(r)|>1.
\end{cases}
\]
\end{definition}

\begin{proposition}[Address before value]\label{prop:address-before-value}
If
\[
\Read_{p,s}(r)=v
\qquad\text{with}\qquad
v\neq \Null,
\]
then
\[
M,p\Vdash \Adm_s(r)
\qquad\text{and}\qquad
M,p\Vdash \CompSec_s(r).
\]
\end{proposition}

\begin{proof}
By \Cref{def:typed-readout}, a non-null readout can occur only in the singleton case $\Vis_{p,s}(r)=\{v\}$. By \Cref{def:visible-value-classes}, every element of $\Vis_{p,s}(r)$ is represented by a compatible local family, so $M,p\Vdash \CompSec_s(r)$. Admissibility is built into the definition of $\Vis_{p,s}(r)$, since visible value classes are defined only for admitted references.
\end{proof}

\begin{corollary}[Non-null readout criterion]\label{cor:nonnull-readout-criterion}
If
\[
M,p\Vdash \Adm_s(r)
\qquad\text{and}\qquad
\Vis_{p,s}(r)=\{v\},
\]
then
\[
\Read_{p,s}(r)=v.
\]
\end{corollary}

\begin{proof}
Immediate from \Cref{def:typed-readout}.
\end{proof}

\begin{proposition}[Refinement stability of defined readouts]\label{prop:typed-readout-persistence}
Let $q\le p$, and suppose
\[
\Read_{p,s}(r)=v
\qquad\text{with}\qquad
v\neq \Null.
\]
Then
\[
\Read_{q,s}(r)=v.
\]
\end{proposition}

\begin{proof}
Since $v\neq\Null$, the readout at $p$ is represented by a unique visible value class. By the refinement axioms \ref{ax:section-restriction} and \ref{ax:vis-stability} (\Cref{def:refinement-axioms}), compatible local families pull back functorially from $p$ to $q$ and the induced map $\Vis_{p\to q}$ sends a singleton of visible classes to a singleton. Hence the same class $v$ remains the unique visible class at $q$, so \Cref{def:typed-readout} gives $\Read_{q,s}(r)=v$.
\end{proof}

\subsection{Structural absence}

\begin{definition}[Undefined, total, and null]\label{def:undef-total-null}
For $r:\mathsf{Ref}_s$, define
\[
\Undef_s(r):=\neg\Adm_s(r),\qquad
\Tot_s(r):=\Adm_s(r)\land \Sec_s(r),\qquad
\Null_s(r):=\Adm_s(r)\land \neg\Sec_s(r).
\]
\end{definition}

\begin{definition}[The three null modes]\label{def:null-failure-modes}
For $r:\mathsf{Ref}_s$, define
\[
\Null^{\mathrm{loc}}_s(r):=\Adm_s(r)\land \neg\LocSec_s(r),
\]
\[
\Null^{\mathrm{cmp}}_s(r):=\Adm_s(r)\land \LocSec_s(r)\land \neg\CompSec_s(r),
\]
\[
\Null^{\mathrm{glue}}_s(r):=\Adm_s(r)\land \CompSec_s(r)\land \neg\Sec_s(r).
\]
\end{definition}

\begin{proposition}[The three modes lie inside null]\label{prop:null-modes-imply-null}
Each of $\Null^{\mathrm{loc}}_s(r)$, $\Null^{\mathrm{cmp}}_s(r)$, and $\Null^{\mathrm{glue}}_s(r)$ implies $\Null_s(r)$.
\end{proposition}

\begin{proof}
The gluing case is immediate from \Cref{def:null-failure-modes}. In the local and compatibility cases, a global section would imply compatible local sectionability and local sectionability by \Cref{prop:global-implies-compatible-local}, contradicting the defining negations.
\end{proof}

\begin{proposition}[Mutual exclusivity of the null modes]\label{prop:null-modes-exclusive}
No two distinct null modes can hold simultaneously for the same reference at the same state.
\end{proposition}

\begin{proof}
The mode $\Null^{\mathrm{loc}}_s(r)$ contains $\neg\LocSec_s(r)$, while the other two contain $\LocSec_s(r)$. The mode $\Null^{\mathrm{cmp}}_s(r)$ contains $\neg\CompSec_s(r)$, while $\Null^{\mathrm{glue}}_s(r)$ contains $\CompSec_s(r)$.
\end{proof}

\begin{definition}[Decomposition decidability]\label{def:decomposition-decidability}
We say that $p$ has \emph{decomposition decidability} for $r:\mathsf{Ref}_s$ if both $\LocSec_s(r)$ and $\CompSec_s(r)$ are decided at $p$.
\end{definition}

\begin{proposition}[Null trichotomy]\label{prop:null-trichotomy}
If $p$ has decomposition decidability for $r:\mathsf{Ref}_s$, then
\[
M,p\Vdash \Null_s(r)
\]
if and only if exactly one of
\[
M,p\Vdash \Null^{\mathrm{loc}}_s(r),\qquad
M,p\Vdash \Null^{\mathrm{cmp}}_s(r),\qquad
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\]
holds.
\end{proposition}

\begin{proof}
Assume $M,p\Vdash \Null_s(r)$. Then $r$ is admitted but not globally sectionable. Since $\LocSec_s(r)$ and $\CompSec_s(r)$ are decided at $p$, there are three cases: either local sectionability already fails, local sectionability holds but compatibility fails, or compatibility holds while global sectionability fails. These are exactly the three null modes. The converse follows from \Cref{prop:null-modes-imply-null}, and uniqueness is \Cref{prop:null-modes-exclusive}.
\end{proof}

\begin{proposition}[Classification after refinement]\label{thm:null-classification-under-refinement}
Let $q\le p$ and suppose
\[
M,p\Vdash \Null_s(r).
\]
Assume, in addition, that
\[
M,q\Vdash \Null_s(r).
\]
If $q$ has decomposition decidability for $r$, then exactly one null mode holds at $q$.
\end{proposition}

\begin{proof}
The hypothesis already gives $M,q\Vdash \Null_s(r)$. Together with decomposition decidability at $q$, \Cref{prop:null-trichotomy} applies directly at the refined state $q$.
\end{proof}

\begin{remark}[Nullity need not persist under refinement]\label{rem:null-nonpersistence}
The additional assumption $M,q\Vdash \Null_s(r)$ is essential. The refinement axioms transport admissibility and compatible local families from $p$ to $q$, but they do not rule out the appearance of new global sections at the refined state. Thus one cannot infer $M,q\Vdash \neg\Sec_s(r)$, and hence $M,q\Vdash \Null_s(r)$, from $M,p\Vdash \Null_s(r)$ alone.
\end{remark}

\begin{proposition}[A finite model for the three absence modes]\label{prop:finite-three-mode-model}
There exist a finite state $p$ and admitted references
\[
r_{\mathrm{loc}},\qquad r_{\mathrm{cmp}},\qquad r_{\mathrm{glue}}
\]
of one referenceable sort such that
\[
M,p\Vdash \Null^{\mathrm{loc}}_s(r_{\mathrm{loc}}),\qquad
M,p\Vdash \Null^{\mathrm{cmp}}_s(r_{\mathrm{cmp}}),\qquad
M,p\Vdash \Null^{\mathrm{glue}}_s(r_{\mathrm{glue}}).
\]
\end{proposition}

\begin{proof}
Take a single state $p$ and one referenceable sort $s$. Let the local reference frame be the disjoint union of three rooted components.

For $r_{\mathrm{loc}}$, use one object $a_{\mathrm{loc}}$ with only the trivial cover and define
\[
F(a_{\mathrm{loc}})=\varnothing.
\]
Then local sectionability already fails, so $M,p\Vdash \Null^{\mathrm{loc}}_s(r_{\mathrm{loc}})$.

For $r_{\mathrm{cmp}}$, use objects
\[
c_{\mathrm{cmp}}\sqsubseteq u_{\mathrm{cmp}}\sqsubseteq a_{\mathrm{cmp}},
\qquad
c_{\mathrm{cmp}}\sqsubseteq v_{\mathrm{cmp}}\sqsubseteq a_{\mathrm{cmp}},
\]
with nontrivial cover $\{u_{\mathrm{cmp}}\to a_{\mathrm{cmp}},v_{\mathrm{cmp}}\to a_{\mathrm{cmp}}\}$. Define
\[
F(u_{\mathrm{cmp}})=\{\sigma_u\},\quad
F(v_{\mathrm{cmp}})=\{\sigma_v\},\quad
F(c_{\mathrm{cmp}})=\{0,1\},\quad
F(a_{\mathrm{cmp}})=\varnothing,
\]
and restrictions
\[
\res_{c_{\mathrm{cmp}}}^{u_{\mathrm{cmp}}}(\sigma_u)=0,
\qquad
\res_{c_{\mathrm{cmp}}}^{v_{\mathrm{cmp}}}(\sigma_v)=1.
\]
Local sections exist on the chosen cover, but the only available family is incompatible on the overlap. Hence $M,p\Vdash \Null^{\mathrm{cmp}}_s(r_{\mathrm{cmp}})$.

For $r_{\mathrm{glue}}$, use objects
\[
c_{\mathrm{glue}}\sqsubseteq u_{\mathrm{glue}}\sqsubseteq a_{\mathrm{glue}},
\qquad
c_{\mathrm{glue}}\sqsubseteq v_{\mathrm{glue}}\sqsubseteq a_{\mathrm{glue}},
\]
with cover $\{u_{\mathrm{glue}}\to a_{\mathrm{glue}},v_{\mathrm{glue}}\to a_{\mathrm{glue}}\}$. Define
\[
F(u_{\mathrm{glue}})=F(v_{\mathrm{glue}})=F(c_{\mathrm{glue}})=\{*\},
\qquad
F(a_{\mathrm{glue}})=\varnothing,
\]
and let every restriction map be the identity on $\{*\}$. Then the chosen cover supports a compatible family, but there is no global section at $a_{\mathrm{glue}}$. Thus $M,p\Vdash \Null^{\mathrm{glue}}_s(r_{\mathrm{glue}})$.

Finally, stipulate
\[
M,p\Vdash \Adm_s(r_{\mathrm{loc}})\land \Adm_s(r_{\mathrm{cmp}})\land \Adm_s(r_{\mathrm{glue}})
\]
and identify the three reference classes with $a_{\mathrm{loc}},a_{\mathrm{cmp}},a_{\mathrm{glue}}$ respectively.
\end{proof}

\subsection{Forcing necessity of the enrichment}

The three null modes show that information-state forcing distinguishes situations invisible to pointwise semantics. The following result makes this precise: the gluing predicates introduced at the local-object layer cannot be reduced to the lower layers.

We first isolate the three language layers used in the separation argument and reformulate the local-object predicates in a one-variable, constant-free manner.

\begin{definition}[Lower and enriched one-variable languages]\label{def:lower-enriched-languages}
Fix a referenceable sort $s\in S_{\mathrm{ref}}$.
\begin{enumerate}[label=(\roman*),nosep]
\item $\mathcal L_0$ is the lower many-sorted first-order language of \Cref{def:lower-layer-syntax}, interpreted pointwise by the usual Tarskian relation $M,\rho\models\varphi$.
\item $\mathcal L_1$ has exactly the same formulas as $\mathcal L_0$ and is interpreted at information states by the forcing relation of \Cref{def:state-forcing}.
\item $\mathcal L_2$ extends $\mathcal L_1$ by unary predicates
\[
\Adm_s(x),\qquad \LocSec_s(x),\qquad \CompSec_s(x),\qquad \Sec_s(x)
\]
for variables $x$ of sort $s$.
\end{enumerate}

A \emph{parameter-indexed local-object expansion} of a lower structure $M$ at the sort $s$ consists of:
\begin{enumerate}[label=(\alph*),nosep]
\item a subset $\Adm_s^M\subseteq D_s^M$ of admitted parameters;
\item for each $u\in \Adm_s^M$, a local reference frame with distinguished root $a_u$ and a presheaf
\[
F_u:\mathcal C_u^{op}\to\mathsf{Set}
\]
on the corresponding slice site $\mathcal C_u$.
\end{enumerate}
For each $u\in D_s^M$, let
\[
p_u:=(x:s,\{\rho_u\}),
\qquad
\rho_u(x)=u,
\]
be the singleton information state determined by $u$. The new atomic predicates are evaluated by
\[
M,p_u\Vdash \Adm_s(x)
\quad\Longleftrightarrow\quad
u\in \Adm_s^M,
\]
and, when $u\in \Adm_s^M$, the predicates $\LocSec_s(x)$, $\CompSec_s(x)$, and $\Sec_s(x)$ are interpreted exactly as in \Cref{def:existence-levels}, with the distinguished root $a_u$ playing the role of the reference class $[r]_p$. For $u\notin\Adm_s^M$, these three section predicates are declared false.
\end{definition}

\begin{definition}[Information-state definability]\label{def:information-state-definability}
Let $\mathcal{K}$ be a class of $\mathcal{L}_2$-expansions constructed as above, and let $P(x)$ be one of the $\mathcal{L}_2$-predicates in the variable $x$ of sort $s$. We say that $P$ is \emph{$\mathcal{L}_1$-definable over $\mathcal{K}$} if there exists an $\mathcal{L}_1$-formula $\theta(x)$ such that for every $M\in\mathcal{K}$ and every singleton parameter-state $p_u$,
\[
M,p_u\Vdash \theta(x)
\quad\Longleftrightarrow\quad
M,p_u\Vdash P(x).
\]
\end{definition}

\begin{theorem}[Pointwise irreducibility of gluing predicates]\label{thm:pointwise-irreducibility}
Let $\Sigma_{\mathrm{eq}}$ be the constant-free signature with one referenceable sort $s$ and equality only. Then there exist a $\Sigma_{\mathrm{eq}}$-structure $M$, singleton states
\[
p_a,\qquad p_b,\qquad p_c,\qquad p_d,
\]
and a parameter-indexed local-object expansion of $M$ such that:
\begin{enumerate}[label=(\roman*),nosep]
\item the parameters $a,b\in D_s^M$ lie in the same $\mathrm{Aut}(M)$-orbit, and so do $c,d\in D_s^M$;
\item
\[
M,p_a\Vdash \CompSec_s(x)\land \Sec_s(x),
\qquad
M,p_b\Vdash \CompSec_s(x)\land \neg\Sec_s(x);
\]
\item
\[
M,p_c\Vdash \CompSec_s(x),
\qquad
M,p_d\Vdash \LocSec_s(x)\land \neg\CompSec_s(x).
\]
\end{enumerate}
Consequently, none of the predicates $\CompSec_s(x)$, $\Sec_s(x)$, or
\[
\Null^{\mathrm{glue}}_s(x):=\Adm_s(x)\land \CompSec_s(x)\land \neg\Sec_s(x)
\]
is $\mathcal L_1$-definable over the class of $\mathcal L_2$-expansions in the sense of \Cref{def:information-state-definability}.
\end{theorem}

\begin{proof}
Let $M$ be the pure equality structure on the four-element carrier
\[
D_s^M=\{a,b,c,d\}.
\]
Because $\Sigma_{\mathrm{eq}}$ has equality only, every permutation of $D_s^M$ is an automorphism of $M$. In particular, the transpositions $(a\,b)$ and $(c\,d)$ show that $a,b$ lie in one automorphism orbit and $c,d$ lie in another.

For each $u\in\{a,b,c,d\}$, define a local reference frame with distinguished root $a_u$ and three further objects
\[
w_u\sqsubseteq u_u\sqsubseteq a_u,
\qquad
w_u\sqsubseteq v_u\sqsubseteq a_u,
\]
whose only nontrivial covering family is
\[
\{u_u\to a_u,\ v_u\to a_u\}.
\]
All other covers are trivial identity covers. Declare every parameter admitted, so
\[
\Adm_s^M=\{a,b,c,d\}.
\]

Now define the presheaves $F_u$.

For $u=a$ and $u=c$, let
\[
F_u(a_u)=F_u(u_u)=F_u(v_u)=F_u(w_u)=\{*\},
\]
and let every restriction map be the identity on $\{*\}$. Then there is a compatible local family on the displayed cover, and there is also a global section at the root. Hence
\[
M,p_a\Vdash \CompSec_s(x)\land \Sec_s(x),
\qquad
M,p_c\Vdash \CompSec_s(x)\land \Sec_s(x).
\]

For $u=b$, let
\[
F_b(u_b)=F_b(v_b)=F_b(w_b)=\{*\},
\qquad
F_b(a_b)=\varnothing,
\]
again with identity restrictions on the singleton fibres. Then $b$ has a compatible local family on the cover $\{u_b\to a_b,v_b\to a_b\}$ but no global section at $a_b$, so
\[
M,p_b\Vdash \CompSec_s(x)\land \neg\Sec_s(x).
\]
Since $b$ is admitted, this also gives
\[
M,p_b\Vdash \Null^{\mathrm{glue}}_s(x).
\]

For $u=d$, let
\[
F_d(u_d)=\{\sigma_u\},\qquad
F_d(v_d)=\{\sigma_v\},\qquad
F_d(w_d)=\{0,1\},\qquad
F_d(a_d)=\varnothing,
\]
with restriction maps determined by
\[
\res_{w_d}^{u_d}(\sigma_u)=0,
\qquad
\res_{w_d}^{v_d}(\sigma_v)=1.
\]
Then local sections exist on the cover of $a_d$, but the only available local family is incompatible on the overlap $w_d$. Therefore
\[
M,p_d\Vdash \LocSec_s(x)\land \neg\CompSec_s(x).
\]

We now pass to the lower language. Let $\theta(x)$ be any formula of $\mathcal L_1$. Because $p_a,p_b,p_c,p_d$ are singleton states, \Cref{prop:singleton-conservativity} gives
\[
M,p_u\Vdash\theta(x)
\quad\Longleftrightarrow\quad
M,\rho_u\models\theta(x)
\qquad
(u\in\{a,b,c,d\}).
\]
By invariance of first-order satisfaction under automorphisms \cite[Theorem~1.3.9]{HodgesShorter1997},
\[
M,\rho_a\models\theta(x)
\Longleftrightarrow
M,\rho_b\models\theta(x),
\qquad
M,\rho_c\models\theta(x)
\Longleftrightarrow
M,\rho_d\models\theta(x).
\]
Hence no lower-language formula separates $a$ from $b$, and no lower-language formula separates $c$ from $d$.

If some $\mathcal L_1$-formula defined $\Sec_s(x)$ throughout all $\mathcal L_2$-expansions, it would separate $a$ from $b$, contradicting the previous paragraph. The same contradiction with $c$ and $d$ rules out definability of $\CompSec_s(x)$. Finally, $\Null^{\mathrm{glue}}_s(x)$ holds at $p_b$ but fails at $p_a$, so it too cannot be defined by any $\mathcal L_1$-formula.
\end{proof}
