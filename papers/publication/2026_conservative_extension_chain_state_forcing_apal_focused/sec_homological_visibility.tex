\section{Visible quotients and the pure-\texorpdfstring{$\operatorname{Ext}$}{Ext} residue}\label{sec:homological-visibility}

We now impose finite-presentation hypotheses and isolate the part of a branch obstruction detected by $H_2$. For a fixed finite nerve presentation this leads to a quotient of the coefficient group obtained from the homological evaluation map and recovered through the class-admissible character channels. The homological algebra is standard. The point is to read the universal coefficient sequence in the semantic terms supplied by the chosen lift. By construction the quotient is presentation-relative: it depends on the chosen branch, finite nerve presentation, band trivialization, and lift. The purpose of this section is to isolate what remains invariant once those data have been fixed.

\subsection{Finite nerve presentations}\label{subsec:finite-nerve-criteria}

\begin{lemma}[Finite nerve criterion]\label{lem:finite-nerve-criterion}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and let $\mathfrak G_v$ be a component gerbe banded by a finite abelian group $A$. Assume there is a finite cover
\[
\mathcal U=\{a_i\to a\}
\]
of $a=[r]_p$ such that:
\begin{enumerate}[label=(\roman*),nosep]
\item every finite nonempty intersection of the $a_i$ belongs to $\mathcal C_{p,s}/a$ and the restriction of $F_{p,s}$ to that intersection is separated;
\item the restrictions of $\Band$ to each intersection admit compatible trivializations (possibly after shrinking the cover inside the same finite poset);
\item $H^1(U_i,\Band)=H^1(U_{ij},\Band)=0$ for all single and double intersections in the cover.
\end{enumerate}
Then the \v{C}ech nerve $N(\mathcal U)$ is finite and \Cref{thm:cech-bridge-compatible-realizations} applies, yielding a finite nerve presentation for the branch obstruction class $\omega_v$.
\end{lemma}

\begin{proof}
Because the cover is finite and the intersection poset closes under finite intersections, $N(\mathcal U)$ has finitely many simplices. Hypothesis~(ii) gives a constant coefficient description of the band on every simplex of the nerve, so the obstruction cocycle takes values in the fixed finite group $A$. The $H^1$-vanishing in (iii) is precisely the hypothesis needed in \Cref{thm:cech-bridge-compatible-realizations} to construct $\omega_v$ from a matching family of local objects. Finally, separatedness on intersections (i) ensures uniqueness of the pulled-back sections, so the class is represented on the finite simplicial set $N(\mathcal U)$.
\end{proof}

\begin{definition}[Finite nerve presentation of the obstruction class]\label{def:finite-nerve-presentation}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write $a=[r]_p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item the hypotheses of \Cref{def:cech-gluing-obstruction} hold;
\item a visible branch $v\in \Vis_{p,s}(r)$ has been chosen;
\item the associated branch gerbe $\mathfrak G_v$ is banded by a finite abelian group $A$.
\end{enumerate}
A \emph{finite nerve presentation} of the branch obstruction class consists of:
\begin{enumerate}[label=(\roman*),nosep]
\item a finite covering family
\[
\mathcal U=\{a_i\to a\}_{i\in I},
\]
whose \v{C}ech nerve $N(\mathcal U)$ is finite;
  \item a trivialization of the band on $\mathcal U$, meaning compatible identifications of the restrictions of the band on all nonempty overlaps with the constant coefficient group $A$, so that the band may be viewed as a constant coefficient finite abelian group $A$ on $N(\mathcal U)$;
\item the induced obstruction class
\[
\omega\in H^2(N(\mathcal U),A),
\]
obtained by identifying \v{C}ech cochains on $\mathcal U$ with simplicial cochains on the finite nerve and writing
\[
\omega:=\omega_v
\]
for the chosen visible branch.
\end{enumerate}
For the finite covers considered below, this is the standard identification of \v{C}ech cochains on $\mathcal U$ with simplicial cochains on the nerve $N(\mathcal U)$ \cite[Secs.~4G, 4K]{Hatcher2002}.
Write
\[
\eta_A:H^2(N(\mathcal U),A)\longrightarrow \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)
\]
for the map in the universal coefficient exact sequence
\[
0\to \operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)\to H^2(N(\mathcal U),A)\xrightarrow{\eta_A}\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\to 0
\]
\cite[Sec.~3.1]{Hatcher2002}, and define the \emph{homological evaluation map}
\[
\mathrm{ev}_{\omega}:=\eta_A(\omega).
\]
\end{definition}

\begin{definition}[Presentation-relative visible quotient]\label{def:intrinsic-visible-quotient}
In the setting of \Cref{def:finite-nerve-presentation}, define the \emph{hidden subgroup}
\[
K_{\omega}:=\mathrm{Im}(\mathrm{ev}_{\omega})
\]
and the associated \emph{presentation-relative visible quotient}
\[
A_{\mathrm{vis}}^{\omega}:=\operatorname{coker}(\mathrm{ev}_{\omega})=A/K_{\omega}.
\]
\end{definition}

\begin{definition}[Class-admissible characters]\label{def:class-admissible-character}
In the setting of \Cref{def:finite-nerve-presentation}, let
\[
X_{\omega}:=\{\chi\in\widehat A\mid \chi_*(\omega)=0\},
\qquad
\widehat A:=\mathrm{Hom}(A,\mathbb T).
\]
Elements of $X_{\omega}$ are called \emph{class-admissible} for $\omega$.
\end{definition}

The next proposition is a direct reformulation of the universal coefficient sequence in the language needed later.

\begin{proposition}[Character-theoretic detection of the presentation-relative visible quotient]\label{thm:intrinsic-visible-quotient}
In the setting of \Cref{def:class-admissible-character},
\[
X_{\omega}
=
\{\chi\in\widehat A\mid \chi\circ \mathrm{ev}_{\omega}=0\}
=
\mathrm{Ann}(K_{\omega}).
\]
Consequently,
\[
K_{\omega}
=
\bigcap_{\chi\in X_{\omega}}\ker\chi,
\qquad
A_{\mathrm{vis}}^{\omega}\cong A\Big/\!\bigcap_{\chi\in X_{\omega}}\ker\chi.
\]
\end{proposition}

\begin{proof}
For every character $\chi:A\to\mathbb T$, naturality of the universal coefficient sequence gives a commutative square
\[
\begin{CD}
H^2(N(\mathcal U),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A) \\
@V\chi_*VV @VV(\chi\circ -)V \\
H^2(N(\mathcal U),\mathbb T) @>\eta_{\mathbb T}>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),\mathbb T).
\end{CD}
\]
Since $\mathbb T$ is divisible and hence injective in $\mathbf{Ab}$, one has
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),\mathbb T)=0
\]
\cite[Sec.~2.3]{Weibel1994}. The universal coefficient sequence with coefficients in $\mathbb T$ therefore has vanishing left term, so $\eta_{\mathbb T}$ is an isomorphism. Therefore
\[
\chi_*(\omega)=0
\Longleftrightarrow
\eta_{\mathbb T}(\chi_*(\omega))=0
\Longleftrightarrow
\chi\circ \eta_A(\omega)=0
\Longleftrightarrow
\chi\circ \mathrm{ev}_{\omega}=0.
\]
This proves the first displayed identity, and the second is simply the definition of the annihilator of a subgroup of $A$.

Let $H:=K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})\le A$. By the first part of the proof, $X_{\omega}=\mathrm{Ann}(H)$. Finite abelian duality implies
\[
\bigcap_{\chi\in \mathrm{Ann}(H)}\ker\chi=H
\]
\cite[Chaps.~5, 12]{DummitFoote2004}. Hence
\[
K_{\omega}
=
\bigcap_{\chi\in X_{\omega}}\ker\chi.
\]
The final isomorphism follows immediately:
\[
A_{\mathrm{vis}}^{\omega}
=
A/K_{\omega}
\cong
A\Big/\!\bigcap_{\chi\in X_{\omega}}\ker\chi.
\qedhere
\]
\end{proof}

\begin{corollary}[Character detection of the presentation-relative visible quotient]\label{cor:intrinsic-character-detection}
The quotient map
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}^{\omega}
\]
induces a natural isomorphism
\[
\widehat{A_{\mathrm{vis}}^{\omega}}\cong X_{\omega}.
\]
For elements $x,y\in A$, one has
\[
\pi_{\omega}(x)=\pi_{\omega}(y)
\quad\Longleftrightarrow\quad
\chi(x)=\chi(y)\ \text{for every class-admissible }\chi.
\]
\end{corollary}

\begin{proof}
By \Cref{thm:intrinsic-visible-quotient},
\[
X_{\omega}=\mathrm{Ann}(K_{\omega})=K_{\omega}^{\perp}.
\]
Characters of $A_{\mathrm{vis}}^{\omega}=A/K_{\omega}$ are exactly the characters of $A$ that kill $K_{\omega}$, so precomposition with $\pi_{\omega}$ identifies $\widehat{A_{\mathrm{vis}}^{\omega}}$ with $X_{\omega}$. The separation statement follows from finite abelian duality applied to the finite group $A_{\mathrm{vis}}^{\omega}$ \cite[Chap.~5]{DummitFoote2004}.
\end{proof}

\begin{proposition}[Initiality among pure-extension quotients]\label{thm:intrinsic-pure-ext-initiality}
Let
\[
q:A\twoheadrightarrow B
\]
be a surjective homomorphism onto a finite abelian group $B$. In the setting of \Cref{def:finite-nerve-presentation}, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item the pushed-forward class $q_*(\omega)\in H^2(N(\mathcal U),B)$ lies in the subgroup
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),B);
\]
\item $\eta_B(q_*(\omega))=0$;
\item $q\circ \mathrm{ev}_{\omega}=0$;
\item $\mathrm{Im}(\mathrm{ev}_{\omega})\subseteq \ker q$;
\item $q$ factors uniquely through
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}^{\omega}.
\]
\end{enumerate}
Hence $A_{\mathrm{vis}}^{\omega}$ is the initial coefficient quotient through which the $H_2$-visible part of the obstruction disappears, leaving only a pure $\operatorname{Ext}$-type remnant.
\end{proposition}

\begin{proof}
The equivalence of \textup{(i)} and \textup{(ii)} is exactly the universal coefficient sequence for $B$ \cite[Sec.~3.1]{Hatcher2002}. Naturality gives
\[
\eta_B(q_*(\omega))=q\circ \eta_A(\omega)=q\circ \mathrm{ev}_{\omega},
\]
so \textup{(ii)} and \textup{(iii)} are equivalent. Conditions \textup{(iii)} and \textup{(iv)} are equivalent by definition of the image of a homomorphism. By \Cref{def:intrinsic-visible-quotient}, $\mathrm{Im}(\mathrm{ev}_{\omega})=K_{\omega}$, so \textup{(iv)} is equivalent to
\[
K_{\omega}\subseteq \ker q.
\]
By the universal property of the quotient by $K_{\omega}$, this is equivalent to unique factorization of $q$ through $\pi_{\omega}$, proving \textup{(v)}.
\end{proof}

\begin{corollary}[Class-trivializing initiality when $H_1$ vanishes]\label{cor:h1-vanishing-class-trivializing}
Assume
\[
H_1(N(\mathcal U),\mathbb Z)=0.
\]
Then, for every surjective homomorphism $q:A\twoheadrightarrow B$, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $q_*(\omega)=0$ in $H^2(N(\mathcal U),B)$;
\item $q$ factors uniquely through
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}^{\omega}.
\]
\end{enumerate}
Hence, when $H_1(N(\mathcal U),\mathbb Z)=0$, the presentation-relative visible quotient is initial among coefficient quotients that annihilate the obstruction class itself.
\end{corollary}

\begin{proof}
If $H_1(N(\mathcal U),\mathbb Z)=0$, then
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),B)=0
\]
for every abelian group $B$. Thus the pure-extension condition in \Cref{thm:intrinsic-pure-ext-initiality}(i) is equivalent to $q_*(\omega)=0$, and the conclusion follows.
\end{proof}

\begin{remark}[Well-definedness within a chosen presentation]\label{rem:no-canonical-splitting}
For a fixed branch obstruction class $\omega$ on a fixed finite nerve $N(\mathcal U)$, both $A_{\mathrm{vis}}^{\omega}$ and $K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})$ are determined by $\omega$ and the nerve. The universal coefficient sequence need not split canonically, so the theory is formulated in terms of these intrinsic presentation-relative objects rather than a noncanonical decomposition of $\omega$.
\end{remark}

\begin{proposition}[Character-blind pure-extension obstructions]\label{thm:character-blind-obstructions}
In the setting of \Cref{def:finite-nerve-presentation}, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $K_{\omega}=0$;
\item $A_{\mathrm{vis}}^{\omega}=A$;
\item $\mathrm{ev}_{\omega}=0$;
\item $\omega$ lies in the subgroup
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)\subseteq H^2(N(\mathcal U),A).
\]
\end{enumerate}
If these conditions hold and $\omega\neq 0$, then the chosen branch gerbe $\mathfrak G_v$ is non-neutral, while every character of $A$ is class-admissible for that branch. If, in addition, $|\Vis_{p,s}(r)|=1$ and the hypotheses of \Cref{thm:gerbe-null-semantics} hold for the chosen realization prestack (in particular global conservativity at $a$), then
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]
\end{proposition}

\begin{proof}
By \Cref{def:intrinsic-visible-quotient},
\[
K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega}),
\]
so \textup{(i)} and \textup{(iii)} are equivalent. Since $A_{\mathrm{vis}}^{\omega}=A/K_{\omega}$, conditions \textup{(i)} and \textup{(ii)} are equivalent. By the universal coefficient sequence, \textup{(iii)} holds exactly when $\omega\in\ker(\eta_A)$, and this kernel is the stated $\operatorname{Ext}$-subgroup. Hence all four conditions are equivalent.

If they hold and $\omega\neq 0$, then \Cref{thm:cech-bridge-compatible-realizations}(iv) shows that the branch gerbe $\mathfrak G_v$ is non-neutral.
On the other hand, $\mathrm{ev}_{\omega}=0$ implies
\[
\chi\circ \mathrm{ev}_{\omega}=0
\]
for every $\chi\in\widehat A$, so \Cref{thm:intrinsic-visible-quotient} shows that every character is class-admissible.
Under the additional hypotheses, the unique visible branch is $v$, and the non-neutrality just proved allows us to apply \Cref{thm:gerbe-null-semantics}. Hence $M,p\Vdash \CompSec_s(r)$ and $M,p\Vdash \neg\Sec_s(r)$, so $M,p\Vdash \Null^{\mathrm{glue}}_s(r)$ by definition.
\end{proof}

\begin{corollary}[Automatic blindness when $H_2$ vanishes]\label{cor:h2-vanishing-blindness}
Assume
\[
H_2(N(\mathcal U),\mathbb Z)=0.
\]
Then, for every obstruction class $\omega\in H^2(N(\mathcal U),A)$,
\[
\mathrm{ev}_{\omega}=0,
\qquad
K_{\omega}=0,
\qquad
A_{\mathrm{vis}}^{\omega}=A.
\]
Every nonzero branch obstruction class on such a nerve is therefore character-blind.
\end{corollary}

\begin{proof}
If $H_2(N(\mathcal U),\mathbb Z)=0$, then
\[
\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)=0,
\]
so $\mathrm{ev}_{\omega}=0$ for every $\omega$. Apply \Cref{thm:character-blind-obstructions}.
\end{proof}

\begin{corollary}[No character-blind obstructions when $H_1$ is free]\label{cor:h1-free-no-blindness}
Assume that $H_1(N(\mathcal U),\mathbb Z)$ is free abelian. Then
\[
K_{\omega}=0
\quad\Longleftrightarrow\quad
\omega=0.
\]
Equivalently, every nonzero branch obstruction class has a nontrivial $H_2$-visible part. In particular, if $|\Vis_{p,s}(r)|=1$, every nontrivial gluing obstruction on such a nerve produces a proper presentation-relative visible quotient.
\end{corollary}

\begin{proof}
If $H_1(N(\mathcal U),\mathbb Z)$ is free abelian, then
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)=0.
\]
Hence the universal coefficient map $\eta_A$ is injective. By \Cref{thm:character-blind-obstructions},
\[
K_{\omega}=0
\quad\Longleftrightarrow\quad
\mathrm{ev}_{\omega}=0
\quad\Longleftrightarrow\quad
\eta_A(\omega)=0.
\]
Injectivity of $\eta_A$ therefore gives $K_{\omega}=0$ if and only if $\omega=0$. If $\omega\neq 0$, then $K_{\omega}\neq 0$, so $A_{\mathrm{vis}}^{\omega}=A/K_{\omega}$ is a proper quotient of the finite group $A$.
\end{proof}

\subsection{Mayer--Vietoris generation of the visible part}\label{subsec:mv-visible}

\begin{theorem}[Mayer--Vietoris generation of the hidden subgroup]\label{thm:mv-hidden-subgroup}
In the setting of \Cref{def:finite-nerve-presentation}, write
\[
N:=N(\mathcal U).
\]
Let
\[
N_1,N_2\subseteq N
\]
be subcomplexes such that
\[
N=N_1\cup N_2.
\]
For the inclusion maps
\[
j_i:N_i\hookrightarrow N
\qquad
(i=1,2),
\]
define the restricted obstruction classes
\[
\omega_i:=j_i^*(\omega)\in H^2(N_i,A)
\qquad
(i=1,2),
\]
and their hidden subgroups
\[
K_{\omega_i}:=\operatorname{Im}(\eta_A(\omega_i))
\qquad
(i=1,2).
\]
If the map
\[
(j_{1*},j_{2*}):H_2(N_1,\mathbb Z)\oplus H_2(N_2,\mathbb Z)\longrightarrow H_2(N,\mathbb Z)
\]
is surjective, then
\[
K_{\omega}=K_{\omega_1}+K_{\omega_2}.
\]
Consequently,
\[
A_{\mathrm{vis}}^{\omega}\cong A/(K_{\omega_1}+K_{\omega_2}).
\]
In particular, the same conclusion holds whenever
\[
H_1(N_1\cap N_2,\mathbb Z)=0.
\]
\end{theorem}

\begin{proof}
For each $i\in\{1,2\}$, naturality of the universal coefficient map gives a commutative square
\[
\begin{CD}
H^2(N,A) @>\eta_A>> \mathrm{Hom}(H_2(N,\mathbb Z),A) \\
@Vj_i^*VV @VV(-\circ j_{i*})V \\
H^2(N_i,A) @>\eta_A>> \mathrm{Hom}(H_2(N_i,\mathbb Z),A).
\end{CD}
\]
Hence
\[
\mathrm{ev}_{\omega_i}=\eta_A(\omega_i)=\eta_A(j_i^*\omega)=\mathrm{ev}_{\omega}\circ j_{i*}
\qquad
(i=1,2).
\]
It follows immediately that
\[
K_{\omega_i}\subseteq K_{\omega}
\qquad
(i=1,2),
\]
and therefore
\[
K_{\omega_1}+K_{\omega_2}\subseteq K_{\omega}.
\]

Assume now that $(j_{1*},j_{2*})$ is surjective, and let
\[
a\in K_{\omega}.
\]
Choose
\[
z\in H_2(N,\mathbb Z)
\]
with
\[
a=\mathrm{ev}_{\omega}(z).
\]
By surjectivity, there exist
\[
z_1\in H_2(N_1,\mathbb Z),
\qquad
z_2\in H_2(N_2,\mathbb Z)
\]
such that
\[
z=j_{1*}(z_1)+j_{2*}(z_2).
\]
Therefore
\[
\begin{aligned}
a
&=\mathrm{ev}_{\omega}\bigl(j_{1*}(z_1)+j_{2*}(z_2)\bigr) \\
&=\mathrm{ev}_{\omega}\bigl(j_{1*}(z_1)\bigr)+\mathrm{ev}_{\omega}\bigl(j_{2*}(z_2)\bigr) \\
&=\mathrm{ev}_{\omega_1}(z_1)+\mathrm{ev}_{\omega_2}(z_2),
\end{aligned}
\]
so
\[
a\in K_{\omega_1}+K_{\omega_2}.
\]
Thus
\[
K_{\omega}\subseteq K_{\omega_1}+K_{\omega_2},
\]
and equality follows. The quotient formula is then the definition of $A_{\mathrm{vis}}^{\omega}$.

If
\[
H_1(N_1\cap N_2,\mathbb Z)=0,
\]
the Mayer--Vietoris exact sequence for the decomposition $N=N_1\cup N_2$ gives
\[
H_2(N_1,\mathbb Z)\oplus H_2(N_2,\mathbb Z)\longrightarrow H_2(N,\mathbb Z)\longrightarrow H_1(N_1\cap N_2,\mathbb Z),
\]
so the first map is surjective; see \cite[Sec.~2.2]{Hatcher2002}. The preceding argument therefore applies.
\end{proof}

\begin{corollary}[Piecewise $H_2$-trivial filtrations force pure-$\operatorname{Ext}$ residue]\label{cor:piecewise-h2-trivial-residue}
In the setting of \Cref{def:finite-nerve-presentation}, write
\[
N:=N(\mathcal U).
\]
Assume that there exist subcomplexes
\[
\varnothing=N_0\subseteq N_1\subseteq \cdots \subseteq N_m=N
\]
and
\[
L_t\subseteq N
\qquad
(1\le t\le m)
\]
such that
\[
N_t=N_{t-1}\cup L_t,
\qquad
H_1(N_{t-1}\cap L_t,\mathbb Z)=0,
\qquad
H_2(L_t,\mathbb Z)=0
\]
for every $t$. Then
\[
K_{\omega}=0,
\qquad
A_{\mathrm{vis}}^{\omega}=A.
\]
If
\[
\omega\neq 0,
\]
the obstruction is therefore completely character-blind. If, in addition,
\[
H_1(N,\mathbb Z)
\]
is free abelian, then
\[
\omega=0.
\]
\end{corollary}

\begin{proof}
For each $t$, let
\[
\omega^{(t)}\in H^2(N_t,A)
\]
denote the restriction of $\omega$ to $N_t$. The proof is by induction on $t$ that
\[
K_{\omega^{(t)}}=0.
\]

For $t=1$, one has
\[
N_1=L_1
\]
because $N_0=\varnothing$. Since
\[
H_2(L_1,\mathbb Z)=0,
\]
the target of
\[
\mathrm{ev}_{\omega^{(1)}}:H_2(N_1,\mathbb Z)\to A
\]
has zero source, so
\[
K_{\omega^{(1)}}=0.
\]

Assume
\[
K_{\omega^{(t-1)}}=0
\]
for some $t\ge 2$. Apply \Cref{thm:mv-hidden-subgroup} to the decomposition
\[
N_t=N_{t-1}\cup L_t.
\]
The hypothesis
\[
H_1(N_{t-1}\cap L_t,\mathbb Z)=0
\]
ensures the required surjectivity on $H_2$. Since
\[
H_2(L_t,\mathbb Z)=0,
\]
the restricted hidden subgroup on $L_t$ is zero. Therefore
\[
K_{\omega^{(t)}}=K_{\omega^{(t-1)}}+K_{\omega|_{L_t}}=0+0=0.
\]
This completes the induction, and the case $t=m$ gives
\[
K_{\omega}=0.
\]
Hence
\[
A_{\mathrm{vis}}^{\omega}=A/K_{\omega}=A.
\]

If
\[
\omega\neq 0,
\]
\Cref{thm:character-blind-obstructions} shows that the obstruction is completely character-blind. If, in addition,
\[
H_1(N,\mathbb Z)
\]
is free abelian, \Cref{cor:h1-free-no-blindness} yields
\[
\omega=0.
\qedhere
\]
\end{proof}

\begin{remark}[Overlap creation of $H_2$-visible obstruction]\label{rem:overlap-created-visibility}
\Cref{thm:mv-hidden-subgroup} isolates the mechanism by which a visible quotient can be created globally. If both pieces are $H_2$-trivial and the overlap has vanishing $H_1$, then no $H_2$-visible obstruction survives the gluing. The $S^2$-type example of \Cref{ex:s2-visible-quotient} lies exactly outside this regime: the standard decomposition by two contractible pieces has an $S^1$-type overlap, and the visible class is created on that overlap rather than inherited from either hemisphere.
\end{remark}

\subsection{Change of presentation}\label{sec:visible-quotient-invariance}

The quotient $A_{\mathrm{vis}}^{\omega}$ is attached to a choice of realization prestack and a finite nerve presentation. The next propositions record the changes of presentation that arise most naturally in practice.

\begin{proposition}[Invariance under change of band trivialization]\label{prop:visible-quotient-trivialization-invariance}
In the setting of \Cref{def:finite-nerve-presentation}, keep the chosen cover $\mathcal U$ and the chosen branch gerbe fixed, but replace the chosen trivialization of the band by another one. Let
\[
\alpha:A\xrightarrow{\sim}A
\]
be the resulting automorphism of the coefficient group, and let
\[
\omega':=\alpha_*(\omega)\in H^2(N(\mathcal U),A)
\]
be the same obstruction class written in the new trivialization. Then
\[
\mathrm{ev}_{\omega'}=\alpha\circ \mathrm{ev}_{\omega},
\qquad
K_{\omega'}=\alpha(K_{\omega}),
\]
and $\alpha$ induces a canonical isomorphism
\[
A_{\mathrm{vis}}^{\omega}\xrightarrow{\sim}A_{\mathrm{vis}}^{\omega'}.
\]
\end{proposition}

\begin{proof}
Changing the trivialization applies the coefficient automorphism $\alpha$ to every simplicial cochain, so $\omega'=\alpha_*(\omega)$ by construction. Naturality of the universal coefficient sequence gives
\[
\eta_A(\omega')=\eta_A(\alpha_*(\omega))=\alpha\circ \eta_A(\omega),
\]
hence
\[
\mathrm{ev}_{\omega'}=\alpha\circ \mathrm{ev}_{\omega}.
\]
Taking images yields $K_{\omega'}=\alpha(K_{\omega})$. Since $\alpha$ is an automorphism of $A$ carrying $K_{\omega}$ onto $K_{\omega'}$, it descends to the claimed quotient isomorphism.
\end{proof}

\begin{proposition}[Invariance under equivalence of chosen lifts]\label{prop:visible-quotient-lift-invariance}
In the setting of \Cref{def:finite-nerve-presentation}, let $\mathfrak P_r$ and $\mathfrak P'_r$ be two realization prestacks over $\mathcal C_a$, banded respectively by finite abelian groups $A$ and $A'$, and let
\[
\Phi:\mathfrak P_r\to \mathfrak P'_r
\]
be an equivalence of categories fibred in groupoids over $\mathcal C_a$ whose induced map on automorphism sheaves is an isomorphism
\[
\alpha:A\xrightarrow{\sim}A'.
\]
Let $v$ and $v'=\Phi(v)$ be the corresponding visible branches, and let $\omega$ and $\omega'$ be the branch obstruction classes on the same cover $\mathcal U$. Then
\[
\omega'=\alpha_*(\omega)\in H^2(N(\mathcal U),A'),
\]
and $\alpha$ induces a canonical isomorphism
\[
A_{\mathrm{vis}}^{\omega}\xrightarrow{\sim}A_{\mathrm{vis}}^{\omega'}.
\]
\end{proposition}

\begin{proof}
The equivalence $\Phi$ transports local objects and transition isomorphisms. Specifically, if $x_i\in\mathfrak G_v(U_i)$ are the local objects and $\phi_{ij}$ are the transition isomorphisms producing the cocycle $g=(g_{ijk})$, then $\Phi(x_i)\in\mathfrak G'_{v'}(U_i)$ and $\Phi(\phi_{ij})$ are valid transition isomorphisms for $\mathfrak G'_{v'}$. On automorphism sections, $\Phi$ acts through the band isomorphism $\alpha$, so the transported cocycle is $\alpha(g)=(\alpha(g_{ijk}))$. Hence
\[
\omega'=\alpha_*(\omega).
\]
By naturality of the universal coefficient sequence,
\[
\mathrm{ev}_{\omega'}=\alpha\circ \mathrm{ev}_{\omega},
\]
so $\alpha(K_{\omega})=K_{\omega'}$. Therefore $\alpha$ descends to the claimed quotient isomorphism.
\end{proof}

\begin{proposition}[Invariance under common refinement with $H_2$-surjective transition maps]\label{prop:visible-quotient-refinement-invariance}
In the setting of \Cref{def:finite-nerve-presentation}, let $\mathcal U$ and $\mathcal V$ be two finite covers of $a$ admitting finite \v{C}ech nerves.

If $\mathcal V$ refines $\mathcal U$ via a refinement map $\tau:\mathcal V\to\mathcal U$, and if $\omega_{\mathcal U}\in H^2(N(\mathcal U),A)$ and $\omega_{\mathcal V}\in H^2(N(\mathcal V),A)$ are the corresponding obstruction classes, then
\[
\tau^*(\omega_{\mathcal U})=\omega_{\mathcal V}
\]
and
\[
\mathrm{ev}_{\omega_{\mathcal V}}=\mathrm{ev}_{\omega_{\mathcal U}}\circ \tau_*,
\]
where $\tau_*:H_2(N(\mathcal V),\mathbb Z)\to H_2(N(\mathcal U),\mathbb Z)$ is the map induced on homology. Consequently,
\[
\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal V}})\subseteq \mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}}),
\]
with equality whenever $\tau_*$ is surjective. In that surjective case,
\[
A_{\mathrm{vis}}^{\omega_{\mathcal U}}\xrightarrow{\sim}A_{\mathrm{vis}}^{\omega_{\mathcal V}}.
\]

More generally, if $\mathcal U$ and $\mathcal V$ admit a common refinement $\mathcal W$ such that both induced maps
\[
H_2(N(\mathcal W),\mathbb Z)\to H_2(N(\mathcal U),\mathbb Z),
\qquad
H_2(N(\mathcal W),\mathbb Z)\to H_2(N(\mathcal V),\mathbb Z)
\]
are surjective, then the two presentation-relative visible quotients are canonically isomorphic after comparison through $\mathcal W$.
\end{proposition}

\begin{proof}
The identity $\tau^*(\omega_{\mathcal U})=\omega_{\mathcal V}$ is \Cref{thm:cech-bridge-compatible-realizations}(iii), which shows that the class is stable in the filtered \v{C}ech system. Naturality of the universal coefficient sequence gives the commutative diagram
\[
\begin{CD}
H^2(N(\mathcal U),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A) \\
@V\tau^*VV @VV(-\circ\tau_*)V \\
H^2(N(\mathcal V),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal V),\mathbb Z),A),
\end{CD}
\]
from which
\[
\mathrm{ev}_{\omega_{\mathcal V}}=\mathrm{ev}_{\omega_{\mathcal U}}\circ\tau_*
\]
follows. The inclusion of images is immediate. If $\tau_*$ is surjective, then
\[
\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal V}})
=
\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}}\circ\tau_*)
=
\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}}),
\]
so $K_{\omega_{\mathcal V}}=K_{\omega_{\mathcal U}}$ and the corresponding quotient groups are canonically equal.

For the common-refinement statement, apply the single-refinement argument to both maps $\mathcal W\to\mathcal U$ and $\mathcal W\to\mathcal V$ and compare the two quotients through the common subgroup $K_{\omega_{\mathcal W}}$.
\end{proof}

\begin{definition}[Presentation groupoid]\label{def:presentation-groupoid}
Let $\mathsf{Pres}_r$ be the groupoid whose objects are triples
\[
(\mathfrak P_r,\mathcal U,\tau)
\]
consisting of:
\begin{enumerate}[label=(\roman*),nosep]
\item a chosen realization prestack for $r$;
\item a finite cover with \v{C}ech nerve $N(\mathcal U)$ satisfying \Cref{lem:finite-nerve-criterion};
\item a choice of band trivialization $\tau$ on that cover.
\end{enumerate}
Morphisms are generated by the following operations:
\begin{enumerate}[label=(\alph*),nosep]
\item equivalences of realization prestacks over $\mathcal C_a$ whose induced band isomorphisms respect the trivializations (\Cref{prop:visible-quotient-lift-invariance});
\item changes of band trivialization (\Cref{prop:visible-quotient-trivialization-invariance});
\item refinements $\mathcal V\to \mathcal U$ with surjective maps on $H_2$ (\Cref{prop:visible-quotient-refinement-invariance}).
\end{enumerate}
\end{definition}

\begin{proposition}[Functorial presentation-relative visible quotient]\label{prop:presentation-groupoid}
The assignment
\[
(\mathfrak P_r,\mathcal U,\tau)\longmapsto A_{\mathrm{vis}}^{\omega}
\]
extends to a functor $F:\mathsf{Pres}_r\to \mathbf{FinAb}$ whose values are well defined up to nonunique isomorphism.
\end{proposition}

\begin{proof}
Each generator in \Cref{def:presentation-groupoid} induces an isomorphism of presentation-relative visible quotients by \Cref{prop:visible-quotient-trivialization-invariance,prop:visible-quotient-lift-invariance,prop:visible-quotient-refinement-invariance}. Composition of generators matches composition of the induced isomorphisms, so $F$ is well defined on morphisms. Since the morphisms form a groupoid, all identifications are up to isomorphism, establishing the claim.
\end{proof}

\begin{remark}[On ``invariance'']\label{rem:invariance-scope}
\Cref{prop:visible-quotient-trivialization-invariance}, \Cref{prop:visible-quotient-lift-invariance}, \Cref{prop:visible-quotient-refinement-invariance}, and \Cref{prop:presentation-groupoid} show that the quotient is well defined within the presentation groupoid generated by equivalences of lifts, re-trivialization of the band, and $H_2$-surjective refinement. Nonequivalent realization prestacks may still lead to different quotients.
\end{remark}

The next examples are intended only to calibrate the formalism. The first two are stated purely at the level of finite nerve presentations. The torus-type construction is included only as an explicit witness for the standing assumptions of \S\ref{subsec:scope-chosen-lift}. None of these examples is presented as a canonical construction extracted from the lower semantics. Here a \emph{finite good cover} means a finite cover whose nonempty finite intersections are contractible; then the nerve has the homotopy type of the underlying space, so the displayed homology groups may be computed either on the space or on the finite nerve \cite[Secs.~4G, 4K]{Hatcher2002}.

\begin{example}[A presentation-relative visible quotient on an $S^2$-type nerve]\label{ex:s2-visible-quotient}
Let $\mathcal U$ be a finite good cover of $S^2$ with nerve $N(\mathcal U)$, and let
\[
A=\mathbb Z/4\mathbb Z.
\]
Since
\[
H_1(S^2,\mathbb Z)=0,
\qquad
H_2(S^2,\mathbb Z)\cong \mathbb Z
\]
\cite[Secs.~2.1, 2.2]{Hatcher2002}, the universal coefficient map is an isomorphism
\[
H^2(N(\mathcal U),A)\cong \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\cong A.
\]
Let $\omega$ be the class corresponding to the homomorphism sending a fundamental class to $2\in\mathbb Z/4\mathbb Z$. Then
\[
K_{\omega}=\langle 2\rangle=\{0,2\},
\qquad
A_{\mathrm{vis}}^{\omega}\cong (\mathbb Z/4\mathbb Z)/\langle 2\rangle\cong \mathbb Z/2\mathbb Z.
\]
Thus the obstruction is nontrivial, but only its parity quotient is visible through the character channels. Because $\mathcal U$ is a finite good cover, the corresponding \v{C}ech and derived classes agree in degree $2$. If, in addition, $\omega$ comes from a chosen realization prestack satisfying the hypotheses of \Cref{thm:gerbe-null-semantics} and is the unique visible branch obstruction of the reference under consideration, then \Cref{thm:gerbe-null-semantics} yields a genuine gluing-level absence with a nontrivial presentation-relative visible quotient.
\end{example}

\begin{example}[A character-blind obstruction on an $\mathbb{R}P^2$-type nerve]\label{ex:rp2-character-blind}
Let $\mathcal U$ be a finite good cover of $\mathbb{R}P^2$ with nerve a finite triangulation of $\mathbb{R}P^2$, and take
\[
A=\mathbb Z/2\mathbb Z.
\]
Then
\[
H_2(\mathbb{R}P^2,\mathbb Z)=0,
\qquad
H_1(\mathbb{R}P^2,\mathbb Z)\cong \mathbb Z/2\mathbb Z
\]
\cite[Sec.~2.2]{Hatcher2002}. The universal coefficient sequence therefore yields
\[
H^2(\mathbb{R}P^2,\mathbb Z/2\mathbb Z)
\cong
\operatorname{Ext}^1(\mathbb Z/2\mathbb Z,\mathbb Z/2\mathbb Z)
\cong
\mathbb Z/2\mathbb Z
\]
\cite[Sec.~3.1]{Hatcher2002}. Let $\omega$ be the nonzero class. Then
\[
\mathrm{ev}_{\omega}=0,
\qquad
K_{\omega}=0,
\qquad
A_{\mathrm{vis}}^{\omega}=A,
\]
by \Cref{thm:character-blind-obstructions}. Thus no class-level quotient collapse occurs, although the obstruction class is nonzero. This branch is therefore completely character-blind. Because $\mathcal U$ is a finite good cover, the corresponding \v{C}ech and derived classes agree in degree $2$. If, in addition, it is the unique visible branch of the admitted reference and the chosen realization prestack satisfies the hypotheses of \Cref{thm:gerbe-null-semantics}, then \Cref{thm:gerbe-null-semantics} yields a genuine gluing-level absence that is invisible to every character channel.
\end{example}

\begin{proposition}[A torus-type consistency example]\label{ex:end-to-end}
There exist a state $p$, an admitted reference $r:\mathsf{Ref}_s$, a separated local object functor $F_{p,s}$ on a finite slice site $\mathcal C_a$, and a chosen realization prestack $\mathfrak P_r$ with band $A=\mathbb Z/4\mathbb Z$ such that:
\begin{enumerate}[label=(\roman*),nosep]
\item the standing assumptions collected in \S\ref{subsec:scope-chosen-lift} hold;
\item $r$ is compatibly locally sectionable but not globally sectionable;
\item the unique visible branch $v$ has branch obstruction class $\omega_v$ satisfying
\[
\mathrm{ev}_{\omega_v}(1)=2
\]
under an identification $H_2(N(\mathcal U),\mathbb Z)\cong \mathbb Z$;
\item the associated presentation-relative visible quotient satisfies
\[
A_{\mathrm{vis}}^{\omega_v}\cong \mathbb Z/2\mathbb Z.
\]
\end{enumerate}
\end{proposition}

\begin{proof}
We construct one such witness in five steps.

\emph{Step~1.}
Let $p$ be a state with one referenceable sort $s$ and a single admitted reference $r$ with reference class $a=[r]_p$. Choose a finite good cover
\[
\mathcal U=\{u_0,u_1,u_2,u_3\}
\]
of the torus $T^2$, and let $\mathcal C_a$ be the preorder site whose objects are the nonempty finite intersections of the $u_i$ together with the root object $a$, ordered by reverse inclusion. Because the cover is good, nonempty finite intersections are contractible, so the identity, pullback, and transitivity axioms for the resulting cover families follow directly from the usual intersection calculus \cite[Chap.~IV]{Giraud1971}. Thus \Cref{def:local-reference-frame} holds. The nerve $N(\mathcal U)$ has the homotopy type of $T^2$, hence
\[
H_1(N(\mathcal U),\mathbb Z)\cong\mathbb Z^2,
\qquad
H_2(N(\mathcal U),\mathbb Z)\cong\mathbb Z.
\]
The same acyclic-cover hypothesis identifies the cover-relative \v{C}ech and derived cohomology groups in degree $2$.

\emph{Step~2.}
Define the local object functor by
\[
F(U)=\{*\}\ \text{for every proper intersection }U\subsetneq a,
\qquad
F(a)=\varnothing,
\]
with all restriction maps the unique maps between singletons. Because every fibre is either empty or a singleton, the presheaf is separated. Every cover of $a$ by members of $\mathcal U$ therefore carries exactly one matching family, so $r$ is compatibly locally sectionable, while $F(a)=\varnothing$ shows that $r$ has no global section. Since the matching family is unique, $\Vis_{p,s}(r)$ consists of a single class $v$. Thus $M,p\Vdash \CompSec_s(r)\land \neg\Sec_s(r)$, so the failure at $a$ is already of gluing type. This establishes \textup{(ii)} modulo the chosen-lift data.

\emph{Step~3.}
Let $A=\mathbb Z/4\mathbb Z$, and choose a class
\[
\omega\in H^2(N(\mathcal U),A)
\]
that sends a fundamental class of $T^2$ to $2\in A$. Let $\mathfrak G_{\omega}$ be the standard $A$-banded gerbe on $\mathcal C_a$ classified by $\omega$ \cite[Chap.~IV]{Giraud1971}. Because every nonempty finite intersection of the good cover is contractible, the $H^1$-vanishing hypotheses of \Cref{thm:cech-bridge-compatible-realizations} are satisfied, so $\mathfrak G_{\omega}$ is a banded gerbe whose \v{C}ech cocycle represents $\omega$. For every proper intersection $U\subsetneq a$, fix one object
\[
x_U\in \mathfrak G_{\omega}(U).
\]
Let $\mathfrak P_r$ be the full subprestack of $\mathfrak G_{\omega}$ generated by the $x_U$ and their restrictions. Over each proper intersection $U$, the fibre of $\mathfrak P_r$ has the single object $x_U$, so every isomorphism presheaf in $\mathfrak P_r$ is either empty or the constant sheaf $A$ on $\mathcal C_a/U$. Hence $\mathfrak P_r$ is a prestack, and
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r)(U)\cong F(U)
\]
for every object $U$. Stackification replaces the prestack automorphism presheaves by the sheaf $A$, hence the stackification of $\mathfrak P_r$ is exactly $\mathfrak G_{\omega}$ \cite[Tag~02ZP]{StacksProject}. This provides the chosen lift, while the finite good cover together with the constant band provides the required finite presentation.

\emph{Step~4.}
Since $\omega\neq 0$, \Cref{thm:cech-bridge-compatible-realizations}(iv) shows that $\mathfrak G_{\omega}$ is non-neutral and therefore
\[
\mathfrak G_{\omega}(a)=\varnothing.
\]
Because $(\mathfrak P_r)_a$ and $(\mathfrak G_{\omega})_a$ are both empty, the comparison functor $(\mathfrak P_r)_a\to (\mathfrak G_{\omega})_a$ is vacuously essentially surjective. Thus $\mathfrak P_r$ is globally conservative at $a$. The unique visible branch $v$ therefore determines a single component gerbe $\mathfrak G_v=\mathfrak G_{\omega}$ with obstruction class $\omega_v=\omega$. Moreover, every fibre of the gerbe over a proper intersection is connected, so $\pi_0(\mathfrak L_r)(U)$ is a singleton for every proper intersection $U$, and the restriction maps
\[
\pi_0(\mathfrak L_r)(a)=\{v\}\longrightarrow \pi_0(\mathfrak L_r)(U)
\]
are bijections. Hence branch constancy holds.

\emph{Step~5.}
Since $H_1(N(\mathcal U),\mathbb Z)\cong \mathbb Z^2$ is free, $\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)=0$ and the universal coefficient map
\[
\eta_A:H^2(N(\mathcal U),A)\xrightarrow{\;\sim\;} \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)
\]
is an isomorphism. The chosen class $\omega_v=\omega$ therefore yields the homological evaluation map
\[
\mathrm{ev}_{\omega_v}:H_2(N(\mathcal U),\mathbb Z)\cong\mathbb Z\longrightarrow \mathbb Z/4\mathbb Z,
\qquad
\mathrm{ev}_{\omega_v}(1)=2,
\]
which proves \textup{(iii)}. Consequently
\[
K_{\omega_v}=\mathrm{Im}(\mathrm{ev}_{\omega_v})=\{0,2\},
\qquad
A_{\mathrm{vis}}^{\omega_v}\cong (\mathbb Z/4\mathbb Z)/\langle 2\rangle\cong\mathbb Z/2\mathbb Z,
\]
proving \textup{(iv)}. The class-admissible characters are exactly those characters that annihilate $2$, that is, the characters factoring through the parity quotient. This verifies the standing assumptions from \S\ref{subsec:scope-chosen-lift} and computes the associated presentation-relative visible quotient explicitly. The example serves only to show that the standing hypotheses can coexist in one finite model; it is not intended as a canonical construction.
\end{proof}

\begin{remark}[Why no GHZ gerbe computation is claimed here]\label{rem:ghz-benchmark}
The GHZ model enters here only through its support presheaf, as in \cite{AbramskyBrandenburger2011}. The standard GHZ data do not by themselves determine a canonical banded gerbe, a fixed coefficient band, or a canonical \v{C}ech $H^2$-class in the present framework.
\end{remark}


\subsection{Comparison with sheaf-theoretic contextuality}

The case $|\Vis_{p,s}(r)|=1$ connects the present framework with the sheaf-theoretic analysis of contextuality \cite{AbramskyBrandenburger2011,AbramskyMansfield2012,AbramskyEtAl2015}. The comparison has two layers: clauses \textup{(i)}--\textup{(iii)} are purely presheaf-theoretic, whereas clauses \textup{(iv)}--\textup{(v)} depend on a chosen globally conservative realization prestack and on a finite nerve presentation of its branch obstruction.

\begin{proposition}[Unique-branch comparison with the no-global-section picture]\label{thm:unique-branch-contextuality-comparison}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, write $a=[r]_p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item the slice site $\mathcal C_a$ is the measurement-cover site of a finite Abramsky--Brandenburger scenario, and the local object functor $F_{p,s}|_{\mathcal C_a}$ is identified with the support presheaf $S_e$ of an empirical model $e$ \cite{AbramskyBrandenburger2011}; moreover $S_e$ is separated;
\item $r$ is equipped with a chosen realization prestack that is globally conservative at $a$ in the sense of \Cref{def:global-conservativity};
\item $|\Vis_{p,s}(r)|=1$, with unique visible branch $v$;
\item the branch obstruction class
\[
\omega:=\omega_v
\]
attached to that chosen realization prestack admits a finite nerve presentation with band $A$.
\end{enumerate}
Then:
\begin{enumerate}[label=(\roman*),nosep]
\item
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a compatible family};
\]
\item
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a global section};
\]
\item consequently,
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a compatible family but no global section};
\]
\item
\[
A_{\mathrm{vis}}^{\omega}
=
\operatorname{coker}(\mathrm{ev}_{\omega}),
\]
and this quotient is exactly the presentation-relative quotient of $A$ detected by the class-admissible characters;
\item if $\omega\neq 0$, then
\[
A_{\mathrm{vis}}^{\omega}=A
\quad\Longleftrightarrow\quad
K_{\omega}=0
\quad\Longleftrightarrow\quad
\omega\in \operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A).
\]
Hence the no-global-section obstruction can remain nontrivial while being invisible to every admissible character channel precisely when the surviving class is pure $\operatorname{Ext}$-type.
\end{enumerate}
\end{proposition}

\begin{proof}
The first three clauses are presheaf-level and do not use the chosen realization prestack. Under the identification $F_{p,s}|_{\mathcal C_a}\cong S_e$, a compatible family in the support presheaf is exactly a compatible local family in the sense of \Cref{def:existence-levels}(ii), and a global section of $S_e$ is exactly an element of $F_{p,s}(a)$. This proves \textup{(i)} and \textup{(ii)}.

Statement \textup{(iii)} is then just the definition of $\Null^{\mathrm{glue}}_s(r)$. By \Cref{def:intrinsic-visible-quotient}, the presentation-relative visible quotient is $\operatorname{coker}(\mathrm{ev}_{\omega})$, and \Cref{cor:intrinsic-character-detection} identifies that quotient with the one detected by the class-admissible characters. Finally, \textup{(v)} is exactly \Cref{thm:character-blind-obstructions} rewritten for the unique branch $\omega$.
\end{proof}

\subsection{Minimal neutralizing quotients and quotient completeness}

The free-$H_1$ regime removes the pure-$\operatorname{Ext}$ ambiguity from the universal coefficient sequence. For a visible branch
\[
v\in \Vis_{p,s}(r)
\]
with finite nerve presentation
\[
(\mathcal U_v,A_v,\omega_v)
\]
and a surjective homomorphism
\[
q:A_v\twoheadrightarrow B,
\]
write
\[
q_*\mathfrak G_v
\]
for a $B$-banded gerbe on $\mathcal C_{p,s}/a$ whose finite-presentation obstruction class is $q_*(\omega_v)$. Existence and uniqueness up to equivalence are standard consequences of band change and the classification of banded gerbes \cite[Chap.~IV]{Giraud1971}, \cite[Tag 06NY]{StacksProject}.

\begin{theorem}[Minimal neutralizing quotients and quotient completeness in the free-$H_1$ regime]\label{thm:free-h1-quotient-completeness}
\textbf{[New-1 | 2026-04-17]}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, write $a=[r]_p$, and assume the hypotheses of \Cref{def:cech-gluing-obstruction} together with global conservativity at $a$. Suppose that every visible branch
\[
v\in \Vis_{p,s}(r)
\]
admits a finite nerve presentation
\[
(\mathcal U_v,A_v,\omega_v)
\]
such that
\[
H_1(N(\mathcal U_v),\mathbb Z)
\]
is free abelian. Then:
\begin{enumerate}[label=(\roman*),nosep]
\item for every visible branch $v$ and every surjective homomorphism
\[
q:A_v\twoheadrightarrow B,
\]
the gerbe $q_*\mathfrak G_v$ is neutral if and only if $q$ factors uniquely through
\[
\pi_{\omega_v}:A_v\twoheadrightarrow A_{\mathrm{vis}}^{\omega_v}.
\]
Consequently,
\[
A_{\mathrm{vis}}^{\omega_v}
\]
is the initial coefficient quotient that neutralizes the branch gerbe $\mathfrak G_v$;
\item for every visible branch $v$,
\[
\mathfrak G_v\ \text{is non-neutral}
\quad\Longleftrightarrow\quad
A_{\mathrm{vis}}^{\omega_v}\neq A_v;
\]
\item
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
\Vis_{p,s}(r)\neq\varnothing
\ \text{and}\
A_{\mathrm{vis}}^{\omega_v}\neq A_v
\qquad
(\forall v\in \Vis_{p,s}(r)).
\]
If branch constancy over $a$ also holds, then
\[
q_r:\mathfrak L_r\to \pi_0(\mathfrak L_r)_{\mathrm{disc}}
\ \text{admits a branch selector}
\quad\Longleftrightarrow\quad
A_{\mathrm{vis}}^{\omega_v}=A_v
\qquad
(\forall v\in \Vis_{p,s}(r)),
\]
and, under the standing global-conservativity hypothesis, these conditions are equivalent to surjectivity of
\[
u_a:F_{p,s}(a)\to \bigl(a_{p,s}F_{p,s}\bigr)(a);
\]
\item under the additional hypotheses of \Cref{thm:unique-branch-contextuality-comparison}, if
\[
|\Vis_{p,s}(r)|=1,
\]
then
\[
S_e\ \text{has a compatible family but no global section}
\quad\Longleftrightarrow\quad
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
A_{\mathrm{vis}}^{\omega}\neq A.
\]
Thus, in the unique-branch support-presheaf case with free first homology, the presentation-relative visible quotient is complete for gluing-nullity.
\end{enumerate}
\end{theorem}

\begin{proof}
Fix a visible branch
\[
v\in \Vis_{p,s}(r)
\]
and a surjective homomorphism
\[
q:A_v\twoheadrightarrow B.
\]
By construction, the band-changed gerbe $q_*\mathfrak G_v$ has finite-presentation obstruction class
\[
q_*(\omega_v)\in H^2(N(\mathcal U_v),B).
\]
Therefore \Cref{thm:cech-bridge-compatible-realizations}(iv) gives
\[
q_*\mathfrak G_v\ \text{neutral}
\quad\Longleftrightarrow\quad
q_*(\omega_v)=0.
\]
Since
\[
H_1(N(\mathcal U_v),\mathbb Z)
\]
is free abelian, one has
\[
\operatorname{Ext}^1(H_1(N(\mathcal U_v),\mathbb Z),B)=0.
\]
Hence condition \textup{(i)} of \Cref{thm:intrinsic-pure-ext-initiality} reduces to vanishing of $q_*(\omega_v)$, and that proposition yields
\[
q_*(\omega_v)=0
\quad\Longleftrightarrow\quad
q\ \text{factors uniquely through}\ \pi_{\omega_v}.
\]
This proves \textup{(i)}.

For \textup{(ii)}, apply \textup{(i)} to the identity map
\[
\mathrm{id}_{A_v}:A_v\twoheadrightarrow A_v.
\]
The factorization condition becomes
\[
\mathrm{id}_{A_v}=\theta\circ \pi_{\omega_v}
\]
for some homomorphism
\[
\theta:A_{\mathrm{vis}}^{\omega_v}\to A_v.
\]
Such a factorization exists if and only if $\pi_{\omega_v}$ is an isomorphism, equivalently
\[
A_{\mathrm{vis}}^{\omega_v}=A_v.
\]
Thus $\mathfrak G_v$ is neutral if and only if $A_{\mathrm{vis}}^{\omega_v}=A_v$, proving the displayed equivalence.

Statement \textup{(iii)} follows from \Cref{thm:gerbe-null-semantics}, which identifies gluing-nullity with nonemptiness of $\Vis_{p,s}(r)$ together with non-neutrality of every component gerbe. The characterization by visible quotients is therefore exactly \textup{(ii)} applied branchwise. If branch constancy also holds, \Cref{thm:branch-selector-neutrality} yields
\[
q_r\ \text{admits a branch selector}
\quad\Longleftrightarrow\quad
\mathfrak G_v\ \text{is neutral for every visible branch }v.
\]
Combining this with \textup{(ii)} gives the branch-selector criterion in terms of visible quotients, and the same theorem identifies it with surjectivity of the terminal sheafification unit $u_a$ under the standing global-conservativity hypothesis.

For \textup{(iv)}, \Cref{thm:unique-branch-contextuality-comparison} identifies the support-presheaf condition
\[
S_e\ \text{has a compatible family but no global section}
\]
with the forcing condition
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]
When $|\Vis_{p,s}(r)|=1$, statement \textup{(iii)} reduces to
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
A_{\mathrm{vis}}^{\omega}\neq A.
\]
This proves the final equivalence.
\end{proof}

\subsection{Dimensional constraints on gluing-nullity}

The branch obstruction package is therefore sensitive not only to the coefficient band but also to the homological dimension of the chosen overlap geometry. The next theorem isolates the resulting rigidity phenomenon for chosen finite presentations.

\begin{theorem}[Conditional two-dimensionality of gluing-null branch obstructions]\label{thm:two-dimensionality-gluing-null}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, write $a=[r]_p$, and assume the hypotheses of \Cref{def:cech-gluing-obstruction} together with global conservativity at $a$. Suppose that every visible branch
\[
v\in \Vis_{p,s}(r)
\]
admits a finite nerve presentation
\[
(\mathcal U_v,A_v,\omega_v)
\]
in the sense of \Cref{def:finite-nerve-presentation}. If
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r),
\]
then
\[
\omega_v\neq 0\in H^2(N(\mathcal U_v),A_v)
\qquad
(\forall v\in \Vis_{p,s}(r)).
\]
In particular,
\[
H^2(N(\mathcal U_v),A_v)\neq 0
\qquad
(\forall v\in \Vis_{p,s}(r)).
\]

If, in addition,
\[
H_1(N(\mathcal U_v),\mathbb Z)\ \text{is free abelian}
\qquad
(\forall v\in \Vis_{p,s}(r)),
\]
then
\[
K_{\omega_v}\neq 0,
\qquad
A_{\mathrm{vis}}^{\omega_v}\neq A_v,
\qquad
H_2(N(\mathcal U_v),\mathbb Z)\neq 0
\qquad
(\forall v\in \Vis_{p,s}(r)).
\]

If
\[
H^2(N(\mathcal U_v),A_v)=0
\qquad
(\forall v\in \Vis_{p,s}(r)),
\]
then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r),
\]
and, if branch constancy over $a$ also holds, then the component projection
\[
q_r:\mathfrak L_r\to \pi_0(\mathfrak L_r)_{\mathrm{disc}}
\]
admits a branch selector, and the terminal sheafification unit
\[
u_a:F_{p,s}(a)\to \bigl(a_{p,s}F_{p,s}\bigr)(a)
\]
is surjective.

Under the additional hypotheses of \Cref{thm:unique-branch-contextuality-comparison}, if
\[
|\Vis_{p,s}(r)|=1
\]
and the finite nerve presentation of the unique branch satisfies
\[
H_1(N(\mathcal U),\mathbb Z)\ \text{free abelian},
\qquad
H_2(N(\mathcal U),\mathbb Z)=0,
\]
then the support presheaf $S_e$ cannot have a compatible family without a global section. In particular, the hypotheses of \Cref{thm:unique-branch-contextuality-comparison} are incompatible with any chosen finite nerve presentation having the homotopy type of a graph.
\end{theorem}

\begin{proof}
Assume
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]
By \Cref{thm:gerbe-null-semantics},
\[
\Vis_{p,s}(r)\neq\varnothing
\]
and every component gerbe $\mathfrak G_v$ is non-neutral. Fix
\[
v\in \Vis_{p,s}(r).
\]
The finite nerve presentation of $v$ identifies the branch obstruction class with
\[
\omega_v\in H^2(N(\mathcal U_v),A_v).
\]
If $\omega_v=0$, then \Cref{thm:cech-bridge-compatible-realizations}(iv) gives neutrality of $\mathfrak G_v$, contradicting the previous paragraph. Therefore
\[
\omega_v\neq 0.
\]
Since $v$ was arbitrary, the first assertion follows.

Assume now that $H_1(N(\mathcal U_v),\mathbb Z)$ is free abelian for every visible branch $v$. By \Cref{cor:h1-free-no-blindness},
\[
\omega_v\neq 0
\quad\Longrightarrow\quad
K_{\omega_v}\neq 0.
\]
Hence
\[
A_{\mathrm{vis}}^{\omega_v}=A_v/K_{\omega_v}\neq A_v.
\]
Because
\[
K_{\omega_v}=\mathrm{Im}(\mathrm{ev}_{\omega_v}),
\]
the inequality $K_{\omega_v}\neq 0$ implies that
\[
\mathrm{ev}_{\omega_v}:H_2(N(\mathcal U_v),\mathbb Z)\to A_v
\]
is nonzero, which is possible only when
\[
H_2(N(\mathcal U_v),\mathbb Z)\neq 0.
\]

Assume next that
\[
H^2(N(\mathcal U_v),A_v)=0
\qquad
(\forall v\in \Vis_{p,s}(r)).
\]
Then $\omega_v=0$ for every visible branch $v$. By \Cref{thm:cech-bridge-compatible-realizations}(iv), each component gerbe $\mathfrak G_v$ is neutral. If
\[
M,p\Vdash \CompSec_s(r),
\]
then \Cref{thm:gerbe-null-semantics} gives
\[
\Vis_{p,s}(r)\neq\varnothing.
\]
Choosing any visible branch $v$, the already established neutrality of $\mathfrak G_v$ and the same theorem yield
\[
M,p\Vdash \Sec_s(r).
\]
The converse implication
\[
M,p\Vdash \Sec_s(r)\Longrightarrow M,p\Vdash \CompSec_s(r)
\]
is \Cref{prop:global-implies-compatible-local}. This proves
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r).
\]
If branch constancy over $a$ also holds, then every component gerbe is neutral and \Cref{thm:branch-selector-neutrality} yields a branch selector for $q_r$ together with surjectivity of $u_a$.

For the final assertion, assume the hypotheses of \Cref{thm:unique-branch-contextuality-comparison}, together with
\[
|\Vis_{p,s}(r)|=1,
\qquad
H_1(N(\mathcal U),\mathbb Z)\ \text{free abelian},
\qquad
H_2(N(\mathcal U),\mathbb Z)=0.
\]
The universal coefficient sequence from \Cref{def:finite-nerve-presentation} gives
\[
H^2(N(\mathcal U),A)=0,
\]
because both
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)
\qquad\text{and}\qquad
\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)
\]
vanish. The already proved vanishing-$H^2$ clause therefore shows that
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r).
\]
By \Cref{thm:unique-branch-contextuality-comparison}, these two forcing conditions are respectively equivalent to the existence of a compatible family and of a global section for the support presheaf $S_e$. Hence $S_e$ cannot have a compatible family without a global section.

If the chosen finite nerve presentation has the homotopy type of a graph, then
\[
H_2(N(\mathcal U),\mathbb Z)=0
\]
and
\[
H_1(N(\mathcal U),\mathbb Z)
\]
is free abelian \cite[Secs.~2.1, 2.2]{Hatcher2002}. The preceding paragraph applies directly.
\end{proof}

\begin{remark}[Car\`u-type blind cases as pure $\operatorname{Ext}$-residuals]\label{rem:caru-ext-residual}
Under the hypotheses of \Cref{thm:unique-branch-contextuality-comparison}, part~(v) identifies the cohomological incompleteness observed by Car\`u \cite{Caru2017} with the case in which the surviving obstruction class lies in the $\operatorname{Ext}$-summand of the universal coefficient sequence.
\end{remark}

\begin{corollary}[Piecewise $H_2$-trivial overlap filtrations exclude gluing-nullity]\label{cor:piecewise-h2-trivial-no-glue}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, write
\[
a=[r]_p,
\]
and assume the hypotheses of \Cref{def:cech-gluing-obstruction} together with global conservativity at $a$. Suppose that every visible branch
\[
v\in \Vis_{p,s}(r)
\]
admits a finite nerve presentation
\[
(\mathcal U_v,A_v,\omega_v)
\]
for which the finite nerve
\[
N(\mathcal U_v)
\]
has a filtration
\[
\varnothing=N_{v,0}\subseteq N_{v,1}\subseteq \cdots \subseteq N_{v,m_v}=N(\mathcal U_v)
\]
and subcomplexes
\[
L_{v,t}\subseteq N(\mathcal U_v)
\qquad
(1\le t\le m_v)
\]
satisfying
\[
N_{v,t}=N_{v,t-1}\cup L_{v,t},
\qquad
H_1(N_{v,t-1}\cap L_{v,t},\mathbb Z)=0,
\qquad
H_2(L_{v,t},\mathbb Z)=0
\]
for every visible branch $v$ and every $t$. Assume moreover that
\[
H_1(N(\mathcal U_v),\mathbb Z)
\]
is free abelian for every visible branch $v$. Then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r).
\]
If branch constancy over $a$ also holds, then the component projection
\[
q_r:\mathfrak L_r\to \pi_0(\mathfrak L_r)_{\mathrm{disc}}
\]
admits a branch selector, and the terminal sheafification unit
\[
u_a:F_{p,s}(a)\to \bigl(a_{p,s}F_{p,s}\bigr)(a)
\]
is surjective.
\end{corollary}

\begin{proof}
Fix a visible branch
\[
v\in \Vis_{p,s}(r).
\]
By \Cref{cor:piecewise-h2-trivial-residue}, the filtration hypotheses force
\[
K_{\omega_v}=0.
\]
Since
\[
H_1(N(\mathcal U_v),\mathbb Z)
\]
is free abelian, the same corollary gives
\[
\omega_v=0.
\]
Because $v$ was arbitrary,
\[
\omega_v=0
\qquad
(\forall v\in \Vis_{p,s}(r)).
\]
Therefore every component gerbe is neutral by \Cref{thm:cech-bridge-compatible-realizations}(iv). The vanishing-$H^2$ clause of \Cref{thm:two-dimensionality-gluing-null} now applies verbatim and yields
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r).
\]
If branch constancy also holds, the same theorem gives a branch selector for $q_r$ and surjectivity of $u_a$.
\end{proof}
