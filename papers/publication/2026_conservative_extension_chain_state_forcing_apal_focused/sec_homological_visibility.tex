\subsection{Homological Evaluation, Visible Quotients, and Pure-Ext Blindness}\label{sec:homological-visibility}

To isolate the part of a branch obstruction that belongs to the cohomology class itself, we make the homological evaluation map primary. The visible quotient will be defined from this map and only afterwards recovered through class-admissible character channels.

Here ``visible quotient'' is defined relative to a chosen branch obstruction class $\omega$ once a branch gerbe and a finite nerve presentation have been fixed. It does not depend on the underlying set-valued local object functor before such a realization-theoretic choice has been made. We prove in \Cref{prop:visible-quotient-lift-invariance,prop:visible-quotient-refinement-invariance} below that this quotient is invariant under equivalence of chosen lifts and common refinement of covers.

\begin{definition}[Finite nerve presentation of the obstruction class]\label{def:finite-nerve-presentation}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write $a=[r]_p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item the hypotheses of \Cref{def:cech-gluing-obstruction} hold;
\item a visible branch $v\in \Vis_{p,s}(r)$ has been chosen;
\item the associated branch gerbe $\mathfrak G_v$ is banded by a finite abelian group $A$.
\end{enumerate}
A \emph{finite nerve presentation} of the branch obstruction class consists of:
\begin{enumerate}[label=(\roman*),nosep]
\item a finite covering family
\[
\mathcal U=\{a_i\to a\}_{i\in I},
\]
whose \v{C}ech nerve $N(\mathcal U)$ is finite;
\item a trivialization of the band on $\mathcal U$, so that the band may be viewed as a constant coefficient finite abelian group $A$ on $N(\mathcal U)$;
\item the induced obstruction class
\[
\omega\in H^2(N(\mathcal U),A),
\]
obtained by identifying \v{C}ech cochains on $\mathcal U$ with simplicial cochains on the finite nerve and writing
\[
\omega:=\omega_v
\]
for the chosen visible branch.
\end{enumerate}
For the finite covers considered below, this is the standard identification of \v{C}ech cochains on $\mathcal U$ with simplicial cochains on the nerve $N(\mathcal U)$.
Write
\[
\eta_A:H^2(N(\mathcal U),A)\longrightarrow \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)
\]
for the map in the universal coefficient exact sequence
\[
0\to \operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)\to H^2(N(\mathcal U),A)\xrightarrow{\eta_A}\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\to 0
\]
\cite[Sec.~3.1]{Hatcher2002}, and define the \emph{homological evaluation map}
\[
\mathrm{ev}_{\omega}:=\eta_A(\omega).
\]
\end{definition}

\begin{definition}[Visible quotient]\label{def:intrinsic-visible-quotient}
In the setting of \Cref{def:finite-nerve-presentation}, define the \emph{hidden subgroup}
\[
K_{\omega}:=\mathrm{Im}(\mathrm{ev}_{\omega})
\]
and the associated \emph{visible quotient}
\[
A_{\mathrm{vis}}^{\omega}:=\operatorname{coker}(\mathrm{ev}_{\omega})=A/K_{\omega}.
\]
\end{definition}

\begin{definition}[Class-admissible characters]\label{def:class-admissible-character}
In the setting of \Cref{def:finite-nerve-presentation}, let
\[
X_{\omega}:=\{\chi\in\widehat A\mid \chi_*(\omega)=0\},
\qquad
\widehat A:=\mathrm{Hom}(A,\mathbb T).
\]
Elements of $X_{\omega}$ are called \emph{class-admissible} for $\omega$.
\end{definition}

\begin{proposition}[Character-theoretic detection of the visible quotient]\label{thm:intrinsic-visible-quotient}
In the setting of \Cref{def:class-admissible-character},
\[
X_{\omega}
=
\{\chi\in\widehat A\mid \chi\circ \mathrm{ev}_{\omega}=0\}
=
\mathrm{Ann}(K_{\omega}).
\]
Consequently,
\[
K_{\omega}
=
\bigcap_{\chi\in X_{\omega}}\ker\chi,
\qquad
A_{\mathrm{vis}}^{\omega}\cong A\Big/\!\bigcap_{\chi\in X_{\omega}}\ker\chi.
\]
\end{proposition}

\begin{proof}
For every character $\chi:A\to\mathbb T$, naturality of the universal coefficient sequence gives a commutative square
\[
\begin{CD}
H^2(N(\mathcal U),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A) \\
@V\chi_*VV @VV(\chi\circ -)V \\
H^2(N(\mathcal U),\mathbb T) @>\eta_{\mathbb T}>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),\mathbb T).
\end{CD}
\]
Since $\mathbb T$ is divisible and hence injective in $\mathbf{Ab}$, one has
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),\mathbb T)=0
\]
\cite[Sec.~2.3]{Weibel1994}. The universal coefficient sequence with coefficients in $\mathbb T$ therefore has vanishing left term, so $\eta_{\mathbb T}$ is an isomorphism. Therefore
\[
\chi_*(\omega)=0
\Longleftrightarrow
\eta_{\mathbb T}(\chi_*(\omega))=0
\Longleftrightarrow
\chi\circ \eta_A(\omega)=0
\Longleftrightarrow
\chi\circ \mathrm{ev}_{\omega}=0.
\]
This proves the first displayed identity, and the second is simply the definition of the annihilator of a subgroup of $A$.

Let $H:=K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})\le A$. By the first part of the proof, $X_{\omega}=\mathrm{Ann}(H)$. Finite abelian duality implies
\[
\bigcap_{\chi\in \mathrm{Ann}(H)}\ker\chi=H
\]
\cite[Chaps.~5, 12]{DummitFoote2004}. Hence
\[
K_{\omega}
=
\bigcap_{\chi\in X_{\omega}}\ker\chi.
\]
The final isomorphism follows immediately:
\[
A_{\mathrm{vis}}^{\omega}
=
A/K_{\omega}
\cong
A\Big/\!\bigcap_{\chi\in X_{\omega}}\ker\chi.
\qedhere
\]
\end{proof}

\begin{corollary}[Character detection of the visible quotient]\label{cor:intrinsic-character-detection}
The quotient map
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}^{\omega}
\]
induces a natural isomorphism
\[
\widehat{A_{\mathrm{vis}}^{\omega}}\cong X_{\omega}.
\]
For elements $x,y\in A$, one has
\[
\pi_{\omega}(x)=\pi_{\omega}(y)
\quad\Longleftrightarrow\quad
\chi(x)=\chi(y)\ \text{for every class-admissible }\chi.
\]
\end{corollary}

\begin{proof}
By \Cref{thm:intrinsic-visible-quotient},
\[
X_{\omega}=\mathrm{Ann}(K_{\omega})=K_{\omega}^{\perp}.
\]
Characters of $A_{\mathrm{vis}}^{\omega}=A/K_{\omega}$ are exactly the characters of $A$ that kill $K_{\omega}$, so precomposition with $\pi_{\omega}$ identifies $\widehat{A_{\mathrm{vis}}^{\omega}}$ with $X_{\omega}$. The separation statement follows from finite abelian duality applied to the finite group $A_{\mathrm{vis}}^{\omega}$ \cite[Chap.~5]{DummitFoote2004}.
\end{proof}

\begin{proposition}[Initiality among pure-extension quotients]\label{thm:intrinsic-pure-ext-initiality}
Let
\[
q:A\twoheadrightarrow B
\]
be a surjective homomorphism onto a finite abelian group $B$. In the setting of \Cref{def:finite-nerve-presentation}, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item the pushed-forward class $q_*(\omega)\in H^2(N(\mathcal U),B)$ lies in the subgroup
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),B);
\]
\item $\eta_B(q_*(\omega))=0$;
\item $q\circ \mathrm{ev}_{\omega}=0$;
\item $\mathrm{Im}(\mathrm{ev}_{\omega})\subseteq \ker q$;
\item $q$ factors uniquely through
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}^{\omega}.
\]
\end{enumerate}
Hence $A_{\mathrm{vis}}^{\omega}$ is the initial coefficient quotient through which the $H_2$-visible part of the obstruction disappears, leaving only a pure $\operatorname{Ext}$-type remnant.
\end{proposition}

\begin{proof}
The equivalence of \textup{(i)} and \textup{(ii)} is exactly the universal coefficient sequence for $B$ \cite[Sec.~3.1]{Hatcher2002}. Naturality gives
\[
\eta_B(q_*(\omega))=q\circ \eta_A(\omega)=q\circ \mathrm{ev}_{\omega},
\]
so \textup{(ii)} and \textup{(iii)} are equivalent. Conditions \textup{(iii)} and \textup{(iv)} are equivalent by definition of the image of a homomorphism. By \Cref{def:intrinsic-visible-quotient}, $\mathrm{Im}(\mathrm{ev}_{\omega})=K_{\omega}$, so \textup{(iv)} is equivalent to
\[
K_{\omega}\subseteq \ker q.
\]
By the universal property of the quotient by $K_{\omega}$, this is equivalent to unique factorization of $q$ through $\pi_{\omega}$, proving \textup{(v)}.
\end{proof}

\begin{corollary}[Class-trivializing initiality when $H_1$ vanishes]\label{cor:h1-vanishing-class-trivializing}
Assume
\[
H_1(N(\mathcal U),\mathbb Z)=0.
\]
Then, for every surjective homomorphism $q:A\twoheadrightarrow B$, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $q_*(\omega)=0$ in $H^2(N(\mathcal U),B)$;
\item $q$ factors uniquely through
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}^{\omega}.
\]
\end{enumerate}
Hence, when $H_1(N(\mathcal U),\mathbb Z)=0$, the visible quotient is initial among coefficient quotients that annihilate the obstruction class itself.
\end{corollary}

\begin{proof}
If $H_1(N(\mathcal U),\mathbb Z)=0$, then
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),B)=0
\]
for every abelian group $B$. Thus the pure-extension condition in \Cref{thm:intrinsic-pure-ext-initiality}(i) is equivalent to $q_*(\omega)=0$, and the conclusion follows.
\end{proof}

\begin{remark}[Well-definedness within a chosen presentation]\label{rem:no-canonical-splitting}
Given a fixed branch obstruction class $\omega$ on a fixed finite nerve $N(\mathcal U)$, the quotient $A_{\mathrm{vis}}^{\omega}$ and the subgroup $K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})$ are determined by $\omega$ and the nerve alone. By contrast, the universal coefficient sequence need not split canonically. Thus the theory isolates a well-defined visible quotient and a well-defined blind subgroup for each chosen presentation, without requiring any noncanonical decomposition of $\omega$ into visible and invisible summands. The invariance of these objects under change of presentation is addressed in \Cref{prop:visible-quotient-lift-invariance,prop:visible-quotient-refinement-invariance}.
\end{remark}

\begin{theorem}[Character-blind pure-extension obstructions]\label{thm:character-blind-obstructions}
In the setting of \Cref{def:finite-nerve-presentation}, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $K_{\omega}=0$;
\item $A_{\mathrm{vis}}^{\omega}=A$;
\item $\mathrm{ev}_{\omega}=0$;
\item $\omega$ lies in the subgroup
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)\subseteq H^2(N(\mathcal U),A).
\]
\end{enumerate}
If these conditions hold and $\omega\neq 0$, then the chosen branch gerbe $\mathfrak G_v$ is non-neutral, while every character of $A$ is class-admissible for that branch. In particular, if $|\Vis_{p,s}(r)|=1$, then
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]
\end{theorem}

\begin{proof}
By \Cref{def:intrinsic-visible-quotient},
\[
K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega}),
\]
so \textup{(i)} and \textup{(iii)} are equivalent. Since $A_{\mathrm{vis}}^{\omega}=A/K_{\omega}$, conditions \textup{(i)} and \textup{(ii)} are equivalent. By the universal coefficient sequence, \textup{(iii)} holds exactly when $\omega\in\ker(\eta_A)$, and this kernel is the stated $\operatorname{Ext}$-subgroup. Hence all four conditions are equivalent.

If they hold and $\omega\neq 0$, then \Cref{thm:cech-bridge-compatible-realizations}(iv) shows that the branch gerbe $\mathfrak G_v$ is non-neutral.
On the other hand, $\mathrm{ev}_{\omega}=0$ implies
\[
\chi\circ \mathrm{ev}_{\omega}=0
\]
for every $\chi\in\widehat A$, so \Cref{thm:intrinsic-visible-quotient} shows that every character is class-admissible.
\end{proof}

\begin{corollary}[Automatic blindness when $H_2$ vanishes]\label{cor:h2-vanishing-blindness}
Assume
\[
H_2(N(\mathcal U),\mathbb Z)=0.
\]
Then, for every obstruction class $\omega\in H^2(N(\mathcal U),A)$,
\[
\mathrm{ev}_{\omega}=0,
\qquad
K_{\omega}=0,
\qquad
A_{\mathrm{vis}}^{\omega}=A.
\]
Every nonzero branch obstruction class on such a nerve is therefore character-blind.
\end{corollary}

\begin{proof}
If $H_2(N(\mathcal U),\mathbb Z)=0$, then
\[
\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)=0,
\]
so $\mathrm{ev}_{\omega}=0$ for every $\omega$. Apply \Cref{thm:character-blind-obstructions}.
\end{proof}

\begin{corollary}[No character-blind obstructions when $H_1$ is free]\label{cor:h1-free-no-blindness}
Assume that $H_1(N(\mathcal U),\mathbb Z)$ is free abelian. Then
\[
K_{\omega}=0
\quad\Longleftrightarrow\quad
\omega=0.
\]
Equivalently, every nonzero branch obstruction class has a nontrivial $H_2$-visible part. In particular, if $|\Vis_{p,s}(r)|=1$, every nontrivial gluing obstruction on such a nerve produces a proper visible quotient.
\end{corollary}

\begin{proof}
If $H_1(N(\mathcal U),\mathbb Z)$ is free abelian, then
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)=0.
\]
Hence the universal coefficient map $\eta_A$ is injective. By \Cref{thm:character-blind-obstructions},
\[
K_{\omega}=0
\quad\Longleftrightarrow\quad
\mathrm{ev}_{\omega}=0
\quad\Longleftrightarrow\quad
\eta_A(\omega)=0.
\]
Injectivity of $\eta_A$ therefore gives $K_{\omega}=0$ if and only if $\omega=0$. If $\omega\neq 0$, then $K_{\omega}\neq 0$, so $A_{\mathrm{vis}}^{\omega}=A/K_{\omega}$ is a proper quotient of the finite group $A$.
\end{proof}

\subsection{Invariance of the visible quotient under change of presentation}\label{sec:visible-quotient-invariance}

The visible quotient $A_{\mathrm{vis}}^{\omega}$ is defined from a chosen realization prestack and a chosen finite nerve presentation. The following two propositions show that the quotient is invariant under the natural notions of equivalence for these choices.

\begin{proposition}[Invariance under equivalence of chosen lifts]\label{prop:visible-quotient-lift-invariance}
In the setting of \Cref{def:finite-nerve-presentation}, let $\mathfrak P_r$ and $\mathfrak P'_r$ be two realization prestacks over $\mathcal C_a$ with the same band $A$, and let
\[
\Phi:\mathfrak P_r\to \mathfrak P'_r
\]
be an equivalence of categories fibred in groupoids over $\mathcal C_a$ that is compatible with the band identification (i.e.\ the induced map on automorphism sheaves is the identity on $A$). Let $v$ and $v'=\Phi(v)$ be the corresponding visible branches, and let $\omega$ and $\omega'$ be the branch obstruction classes on the same cover $\mathcal U$. Then $\omega=\omega'$ in $H^2(N(\mathcal U),A)$, and in particular
\[
A_{\mathrm{vis}}^{\omega}=A_{\mathrm{vis}}^{\omega'}.
\]
\end{proposition}

\begin{proof}
The equivalence $\Phi$ transports local objects and transition isomorphisms. Specifically, if $x_i\in\mathfrak G_v(U_i)$ are the local objects and $\phi_{ij}$ are the transition isomorphisms producing the cocycle $g=(g_{ijk})$, then $\Phi(x_i)\in\mathfrak G'_{v'}(U_i)$ and $\Phi(\phi_{ij})$ are valid transition isomorphisms for $\mathfrak G'_{v'}$. Since $\Phi$ acts as the identity on automorphism sections, the resulting cocycle is the same:
\[
\Phi(\phi_{ik}^{-1}\circ\phi_{jk}\circ\phi_{ij})
=
\phi_{ik}^{-1}\circ\phi_{jk}\circ\phi_{ij}
=g_{ijk}.
\]
Hence $\omega'=[g]=\omega$, and the visible quotient is unchanged.
\end{proof}

\begin{proposition}[Invariance under common refinement of covers]\label{prop:visible-quotient-refinement-invariance}
In the setting of \Cref{def:finite-nerve-presentation}, let $\mathcal U$ and $\mathcal V$ be two finite covers of $a$ admitting finite \v{C}ech nerves, with $\mathcal V$ a refinement of $\mathcal U$ via a refinement map $\tau:\mathcal V\to\mathcal U$. Let $\omega_{\mathcal U}\in H^2(N(\mathcal U),A)$ and $\omega_{\mathcal V}\in H^2(N(\mathcal V),A)$ be the branch obstruction classes on the two covers. Then
\[
\tau^*(\omega_{\mathcal U})=\omega_{\mathcal V}
\]
and the induced map $\tau^*:\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\to\mathrm{Hom}(H_2(N(\mathcal V),\mathbb Z),A)$ satisfies
\[
\mathrm{ev}_{\omega_{\mathcal V}}=\mathrm{ev}_{\omega_{\mathcal U}}\circ \tau_*,
\]
where $\tau_*:H_2(N(\mathcal V),\mathbb Z)\to H_2(N(\mathcal U),\mathbb Z)$ is the map induced on homology. In particular,
\[
\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}})\subseteq \mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal V}})
\quad\text{whenever}\quad
\tau_*\text{ is surjective},
\]
and $\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}})= \mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal V}})$ whenever $\tau_*$ is surjective. In the filtered \v{C}ech system, the visible quotient stabilizes:
\[
A_{\mathrm{vis}}^{\omega_{\mathcal U}}\cong A_{\mathrm{vis}}^{\omega_{\mathcal V}}
\]
for all sufficiently fine covers.
\end{proposition}

\begin{proof}
The identity $\tau^*(\omega_{\mathcal U})=\omega_{\mathcal V}$ is \Cref{thm:cech-bridge-compatible-realizations}(iii), which shows that the class is stable in the filtered \v{C}ech system. Naturality of the universal coefficient sequence gives the commutative diagram
\[
\begin{CD}
H^2(N(\mathcal U),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A) \\
@V\tau^*VV @VV(-\circ\tau_*)V \\
H^2(N(\mathcal V),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal V),\mathbb Z),A),
\end{CD}
\]
from which $\mathrm{ev}_{\omega_{\mathcal V}}=\mathrm{ev}_{\omega_{\mathcal U}}\circ\tau_*$ follows. If $\tau_*$ is surjective, then $\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal V}})=\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}}\circ\tau_*)=\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}})$, so
\[
K_{\omega_{\mathcal V}}=K_{\omega_{\mathcal U}}
\quad\text{and}\quad
A_{\mathrm{vis}}^{\omega_{\mathcal V}}=A/K_{\omega_{\mathcal V}}=A/K_{\omega_{\mathcal U}}=A_{\mathrm{vis}}^{\omega_{\mathcal U}}.
\]
In the filtered system, since $A$ is finite and the images $\mathrm{Im}(\mathrm{ev}_{\omega_{\mathcal U}})\le A$ are nested subgroups of a finite group, the system stabilizes after finitely many refinements.
\end{proof}

\begin{remark}[Scope of ``invariance'']\label{rem:invariance-scope}
\Cref{prop:visible-quotient-lift-invariance} and \Cref{prop:visible-quotient-refinement-invariance} together show that the visible quotient depends only on the equivalence class of the chosen realization prestack and the stable \v{C}ech class, not on auxiliary choices of cover or of representative lift within an equivalence class. However, the visible quotient does depend on the choice of realization prestack itself (including the band). We do not claim that an arbitrary set-valued presheaf $F_{p,s}$ determines a canonical visible quotient without such a choice.
\end{remark}

The following examples are stated at the level of finite nerve presentations of chosen branch gerbes. In particular, the cocycles come from gerbe data on the nerve rather than from arbitrarily declared comparison elements.

\begin{example}[A visible quotient on an $S^2$-type nerve]\label{ex:s2-visible-quotient}
Let $\mathcal U$ be a finite good cover of $S^2$ with nerve $N(\mathcal U)$, and let
\[
A=\mathbb Z/4\mathbb Z.
\]
Since
\[
H_1(S^2,\mathbb Z)=0,
\qquad
H_2(S^2,\mathbb Z)\cong \mathbb Z
\]
\cite[Secs.~2.1, 2.2]{Hatcher2002}, the universal coefficient map is an isomorphism
\[
H^2(N(\mathcal U),A)\cong \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\cong A.
\]
Let $\omega$ be the class corresponding to the homomorphism sending a fundamental class to $2\in\mathbb Z/4\mathbb Z$. Then
\[
K_{\omega}=\langle 2\rangle=\{0,2\},
\qquad
A_{\mathrm{vis}}^{\omega}\cong (\mathbb Z/4\mathbb Z)/\langle 2\rangle\cong \mathbb Z/2\mathbb Z.
\]
Thus the obstruction is nontrivial, but only its parity quotient is visible through the character channels. If $\omega$ is the unique visible branch obstruction of the reference under consideration, then \Cref{thm:gerbe-null-semantics} yields a genuine gluing-level absence with a nontrivial visible quotient.
\end{example}

\begin{example}[A character-blind obstruction on an $\mathbb{R}P^2$-type nerve]\label{ex:rp2-character-blind}
Let $\mathcal U$ be a finite cover whose Cech nerve is a finite triangulation of $\mathbb{R}P^2$, and take
\[
A=\mathbb Z/2\mathbb Z.
\]
Then
\[
H_2(\mathbb{R}P^2,\mathbb Z)=0,
\qquad
H_1(\mathbb{R}P^2,\mathbb Z)\cong \mathbb Z/2\mathbb Z
\]
\cite[Sec.~2.2]{Hatcher2002}. The universal coefficient sequence therefore yields
\[
H^2(\mathbb{R}P^2,\mathbb Z/2\mathbb Z)
\cong
\operatorname{Ext}^1(\mathbb Z/2\mathbb Z,\mathbb Z/2\mathbb Z)
\cong
\mathbb Z/2\mathbb Z
\]
\cite[Sec.~3.1]{Hatcher2002}. Let $\omega$ be the nonzero class. Then
\[
\mathrm{ev}_{\omega}=0,
\qquad
K_{\omega}=0,
\qquad
A_{\mathrm{vis}}^{\omega}=A,
\]
by \Cref{thm:character-blind-obstructions}. Thus no class-level quotient collapse occurs at all, yet the obstruction class is nonzero. Hence this branch is completely character-blind. If it is the unique visible branch of the admitted reference under consideration, then \Cref{thm:gerbe-null-semantics} yields a genuine gluing-level absence that is invisible to every character channel.
\end{example}

\begin{example}[End-to-end: from reference to visible quotient on a torus-type nerve]\label{ex:end-to-end}
We construct a complete instance of the pipeline, starting from a reference and a local object functor, through a chosen realization prestack, to the visible quotient.

\emph{Step~1: Reference and local object functor.}
Let $p$ be a state with one referenceable sort $s$ and a single admitted reference $r$ with reference class $a=[r]_p$. We build a local reference frame whose \v{C}ech nerve has the topology of a torus. Use a standard finite good cover
\[
\mathcal U=\{u_0,u_1,u_2,u_3\}
\]
of the torus $T^2$, with pairwise overlaps $u_{ij}$, triple overlaps $u_{ijk}$, and quadruple overlap $u_{0123}$. Define a local reference frame at $a$ with one sub-reference for each open set in the cover, ordered by inclusion, so that the resulting site is the \v{C}ech site of $\mathcal U$. The nerve $N(\mathcal U)$ is a finite simplicial complex with
\[
H_1(N(\mathcal U),\mathbb Z)\cong\mathbb Z^2,\qquad H_2(N(\mathcal U),\mathbb Z)\cong\mathbb Z.
\]
Define the local object functor by setting $F(u)=\{*_u\}$ for every object $u\neq a$ in the \v{C}ech site and $F(a)=\varnothing$, with all restriction maps the unique maps between singletons. Then the family $(*_{u_0},*_{u_1},*_{u_2},*_{u_3})$ is compatible (all overlaps have unique sections that agree), so $r$ is compatibly locally sectionable, but $F(a)=\varnothing$ means there is no global section:
\[
M,p\Vdash\Null^{\mathrm{glue}}_s(r),\qquad |\Vis_{p,s}(r)|=1.
\]

\emph{Step~2: Chosen realization prestack.}
The local object functor $F|_{\mathcal C_a}$ is separated (all fibers have at most one element). Take the band $A=\mathbb Z/4\mathbb Z$. Since $F|_{\mathcal C_a}$ is separated, \Cref{prop:split-realization-prestack} provides a split realization prestack $\mathfrak P_r^{\mathrm{split}}$. Each fiber is a groupoid with one object and automorphism group $\mathbb Z/4\mathbb Z$. The split prestack is neutral (its cocycle is identically zero). To construct a nontrivial gerbe, define a realization prestack $\mathfrak P_r$ with the same objects but with a twisted descent datum: on the double overlaps, choose transition isomorphisms
\[
\phi_{ij}\in\Band(u_{ij})=\mathbb Z/4\mathbb Z
\]
such that the resulting \v{C}ech $2$-cocycle $g=(g_{ijk})$ defined by
\[
g_{ijk}=\phi_{ik}^{-1}-\phi_{jk}+\phi_{ij}\in\mathbb Z/4\mathbb Z
\]
represents the class corresponding to $2\in\mathbb Z/4\mathbb Z$ under the universal coefficient isomorphism (see Step~3). Explicitly, one may choose $\phi_{01}=2$, $\phi_{02}=\phi_{12}=\phi_{03}=\phi_{13}=\phi_{23}=0$, and verify that the resulting cocycle is cohomologous to a representative of the class $2\in H^2(T^2,\mathbb Z/4\mathbb Z)$. The stackification $\mathfrak L_r$ has one visible branch $v$, and the component gerbe $\mathfrak G_v$ has obstruction class
\[
\omega:=\omega_v\in H^2(N(\mathcal U),\mathbb Z/4\mathbb Z)
\]
corresponding to $2$ under $\eta_A$.

\emph{Step~3: Homological data.}
Since $H_1(T^2,\mathbb Z)\cong\mathbb Z^2$ is free, we have
\[
\operatorname{Ext}^1(\mathbb Z^2,\mathbb Z/4\mathbb Z)=0,
\]
so the universal coefficient map is an isomorphism:
\[
\eta_A:H^2(T^2,\mathbb Z/4\mathbb Z)\xrightarrow{\;\sim\;}\mathrm{Hom}(H_2(T^2,\mathbb Z),\mathbb Z/4\mathbb Z)\cong\mathbb Z/4\mathbb Z.
\]
The class $\omega$ corresponds to the homomorphism sending the fundamental class $[T^2]$ to $2\in\mathbb Z/4\mathbb Z$.

\emph{Step~4: Visible quotient.}
The evaluation map is
\[
\mathrm{ev}_{\omega}:H_2(T^2,\mathbb Z)\cong\mathbb Z\to\mathbb Z/4\mathbb Z,\qquad [T^2]\mapsto 2.
\]
Therefore
\[
K_{\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})=\langle 2\rangle=\{0,2\}\le\mathbb Z/4\mathbb Z,
\]
and
\[
A_{\mathrm{vis}}^{\omega}=(\mathbb Z/4\mathbb Z)/\langle 2\rangle\cong\mathbb Z/2\mathbb Z.
\]
The class-admissible characters are those $\chi:\mathbb Z/4\mathbb Z\to\mathbb T$ with $\chi(2)=1$, which is the unique order-$2$ character. In summary: the admitted reference $r$ is in gluing failure, the branch gerbe is non-neutral with obstruction class $\omega=2$, the visible quotient $A_{\mathrm{vis}}^{\omega}\cong\mathbb Z/2\mathbb Z$ captures the parity of the obstruction, and the order-$4$ information is invisible to the character channels. This example validates the full pipeline from local object functor through realization prestack to visible quotient.
\end{example}


\subsection{Conditional connection to sheaf-theoretic contextuality}

The unique-branch case $|\Vis_{p,s}(r)|=1$ deserves separate attention because it connects the present theory to the sheaf-theoretic analysis of contextuality \cite{AbramskyBrandenburger2011,AbramskyMansfield2012,AbramskyEtAl2015}. The comparison is \emph{conditional}: it assumes that a compatible realization prestack has already been chosen. We do not construct such a prestack from an arbitrary empirical model or support presheaf; doing so would require additional structure (a groupoid enrichment of the measurement scenario) that is not part of the standard Abramsky--Brandenburger formalism. The theorem below should therefore be read as a translation result: \emph{if} the extra structure is available, \emph{then} the formalism developed here recovers the no-global-section picture and refines the blind residue.

\begin{theorem}[Conditional unique-branch comparison with the no-global-section picture]\label{thm:unique-branch-contextuality-comparison}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, write $a=[r]_p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item the slice site $\mathcal C_a$ is the measurement-cover site of a finite Abramsky--Brandenburger scenario, and the local object functor $F_{p,s}|_{\mathcal C_a}$ is identified with the support presheaf $S_e$ of an empirical model $e$ \cite{AbramskyBrandenburger2011}; moreover $S_e$ is separated;
\item $r$ is equipped with a chosen realization prestack that is globally conservative at $a$ in the sense of \Cref{def:global-conservativity};
\item $|\Vis_{p,s}(r)|=1$, with unique visible branch $v$;
\item the branch obstruction class
\[
\omega:=\omega_v
\]
attached to that chosen realization prestack admits a finite nerve presentation with band $A$.
\end{enumerate}
Then:
\begin{enumerate}[label=(\roman*),nosep]
\item
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a compatible family};
\]
\item
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a global section};
\]
\item consequently,
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a compatible family but no global section};
\]
\item
\[
A_{\mathrm{vis}}^{\omega}
=
\operatorname{coker}(\mathrm{ev}_{\omega}),
\]
and this quotient is exactly the quotient of $A$ detected by the class-admissible characters;
\item if $\omega\neq 0$, then
\[
A_{\mathrm{vis}}^{\omega}=A
\quad\Longleftrightarrow\quad
K_{\omega}=0
\quad\Longleftrightarrow\quad
\omega\in \operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A).
\]
Hence the no-global-section obstruction can remain nontrivial while being invisible to every admissible character channel precisely when the surviving class is pure $\operatorname{Ext}$-type.
\end{enumerate}
\end{theorem}

\begin{proof}
Under the identification $F_{p,s}|_{\mathcal C_a}\cong S_e$, a compatible family in the support presheaf is exactly a compatible local family in the sense of \Cref{def:existence-levels}(ii), and a global section of $S_e$ is exactly an element of $F_{p,s}(a)$. This proves \textup{(i)} and \textup{(ii)}.

Statement \textup{(iii)} is then just the definition of $\Null^{\mathrm{glue}}_s(r)$. By \Cref{def:intrinsic-visible-quotient}, the visible quotient is $\operatorname{coker}(\mathrm{ev}_{\omega})$, and \Cref{cor:intrinsic-character-detection} identifies that quotient with the one detected by the class-admissible characters. Finally, \textup{(v)} is exactly \Cref{thm:character-blind-obstructions} rewritten for the unique branch $\omega$.
\end{proof}

\begin{remark}[Car\`u-type blind cases as pure $\operatorname{Ext}$-residuals]\label{rem:caru-ext-residual}
Under the conditional hypotheses of \Cref{thm:unique-branch-contextuality-comparison}, part~(v) gives a characterization of the failure of cohomological completeness observed by Car\`u \cite{Caru2017}: in the unique-branch case, when a compatible realization prestack exists, the blind residue is exactly the case in which the surviving obstruction class lies entirely in the $\operatorname{Ext}$-summand of the universal coefficient sequence. This does not improve on the Abramsky--Brandenburger obstruction theory itself, but provides a conditional semantic reinterpretation of the cohomological incompleteness phenomenon.
\end{remark}
