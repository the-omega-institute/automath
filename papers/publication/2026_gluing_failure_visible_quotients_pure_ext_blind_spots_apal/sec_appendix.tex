\section{Auxiliary proofs}\label{app:deferred-proofs}

\begin{proof}[Proof of Proposition~\ref{prop:conservative-composition}]
Compose the language embeddings and forgetful functors:
\[
e_{\ell,n}:=e_{m,n}\circ e_{\ell,m},
\qquad
U_{n,\ell}:=U_{m,\ell}\circ U_{n,m}.
\]
For every $M\in\Mod_n$, define
\[
\pi^M_{n,\ell}:=
\pi^{\,U_{n,m}(M)}_{m,\ell}\circ \pi^M_{n,m}.
\]
If $\varphi\in\mathrm{Form}_\ell$ and $p\in\St_n(M)$, then
\[
M,p\Vdash_n\varphi
\Longleftrightarrow
U_{n,m}(M),\pi^M_{n,m}(p)\Vdash_m\varphi
\Longleftrightarrow
U_{n,\ell}(M),\pi^M_{n,\ell}(p)\Vdash_\ell\varphi,
\]
by two applications of \Cref{def:conservative-extension}. This is exactly the required conservativity condition.
\end{proof}

\begin{proof}[Proof of Proposition~\ref{prop:local-reference-pretopology}]
The identity axiom in \Cref{def:local-reference-frame} gives the covering family $\{a\to a\}$ for every object $a$. If $\{a_i\to a\}_{i\in I}$ is a cover and $b\to a$ is any morphism, the pullback-stability axiom provides a cover
\[
\{b_{ij}\to b\}_{i\in I,\;j\in J_i}
\]
such that each arrow factors through $a_i\to a$. Finally, if $\{a_i\to a\}_{i\in I}$ covers $a$ and each $\{b_{ij}\to a_i\}_{j\in J_i}$ covers $a_i$, the transitivity axiom gives the composite cover
\[
\{b_{ij}\to a\}_{i\in I,\;j\in J_i}.
\]
These are exactly the identity, pullback, and transitivity axioms for a Grothendieck pretopology on a preorder category \cite[Chap.~III, \S2]{MacLaneMoerdijk1994}, \cite[Part~B2.1]{Johnstone2002}.
\end{proof}

\begin{proof}[Proof of Proposition~\ref{thm:sheafification-characterization}]
By the standard site-theoretic description of sheafification, a section of
$a_{p,s}F_{p,s}$ at $a$ is represented by a matching family over some
covering sieve of $a$, equivalently over some covering family of $a$
\cite[Chap.~III, \S2]{MacLaneMoerdijk1994}, \cite[Part~B2.2]{Johnstone2002}.
In the present setting, such a matching family is exactly a compatible local
family as in \Cref{def:existence-levels}(ii). Hence the sheafified fiber is
nonempty exactly when $r$ is compatibly locally sectionable at $p$.
\end{proof}

\begin{proof}[Proof of Proposition~\ref{prop:global-implies-compatible-local}]
Let $a=[r]_p$ and choose $\sigma\in F_{p,s}(a)$. By \Cref{def:local-reference-frame}, the trivial cover $\{a\to a\}$ is available. The singleton family $(\sigma)$ is therefore compatible, so both compatible local sectionability and local sectionability follow immediately.
\end{proof}

\begin{proof}[Proof of Proposition~\ref{thm:visible-value-components}]
By the standard description of sheafification on a site, a section of $a_{p,s}F_{p,s}$ at $a$ is represented by a matching family over a cover of $a$, modulo passage to common refinements. This is exactly the equivalence relation imposed in \Cref{def:visible-value-classes}. Hence
\[
\Vis_{p,s}(r)\cong \bigl(a_{p,s}F_{p,s}\bigr)(a).
\]
The second identification is \Cref{prop:stackification-components} evaluated at $a$.
\end{proof}
