\section{Conditional Component Gerbes and Cover-Relative Gluing Obstructions}\label{sec:gerbe-obstruction}

The stackification $\mathfrak L_r$ may have several global connected-component sections over $a$. Cohomological obstruction theory therefore belongs, in general, not to a single gerbe attached to $r$, but to the component gerbes indexed by the visible value classes of $r$.

Throughout this section we fix a conditional datum $\mathbf D=(r,p,\mathfrak P_r,\Band,v,\mathcal U,A,\omega)$ in the sense of \Cref{def:conditional-datum}; in particular, $\mathfrak P_r$ is gluing-sensitive at $a$, $v$ is a visible value class, and $\mathcal U$ is a gerbe-adapted cover equipped with a finite abelian trivialization of the band.

\begin{definition}[Component gerbes and cover-relative branch obstruction classes]\label{def:compatible-realization-stack}\label{def:cech-gluing-obstruction}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write
\[
a=[r]_p,
\]
and assume that $r$ admits a gluing-sensitive realization prestack $\mathfrak P_r$ (in the sense of \Cref{def:gluing-sensitive}) with stackification $\mathfrak L_r$ and band $\Band$ as in \Cref{def:realization-prestack}. For a visible value class
\[
v\in \Vis_{p,s}(r)\cong \pi_0(\mathfrak L_r)(a),
\]
define the \emph{component substack}
\[
\mathfrak G_v(U):=\{x\in \mathfrak L_r(U)\mid [x]=v|_U\in \pi_0(\mathfrak L_r)(U)\}
\]
for each object $U$ of $\mathcal C_{p,s}/a$. When \Cref{thm:component-gerbe-decomposition} applies, $\mathfrak G_v$ is called the \emph{component gerbe} of the visible branch $v$.

A cover
\[
\mathcal U=\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)
\]
is called \emph{gerbe-adapted for $v$} if its \v{C}ech nerve exists in $\mathcal C_{p,s}/a$ and
\begin{enumerate}[label=(\roman*),nosep]
\item for every $i$, $H^1(\mathcal C_a/a_i,\Band)=0$;
\item for every $i,j$, $H^1(\mathcal C_a/(a_i\wedge a_j),\Band)=0$.
\end{enumerate}
Whenever such a cover $\mathcal U$ has been chosen, \Cref{thm:cech-bridge-compatible-realizations} yields a class
\[
[\omega_v(\mathcal U)]\in\Cech^2(\mathcal U,\Band),
\]
called the \emph{cover-relative branch obstruction class} of $v$ on $\mathcal U$.
\end{definition}

\begin{remark}[Why the obstruction is defined cover-by-cover]\label{rem:cover-relative-obstruction}
No general directed or cofinal family of gerbe-adapted covers is assumed here. The paper works instead with a chosen adapted cover for each branch, and uses refinement only in classes where the induced maps on cohomology or homology are explicitly controlled. If one is given additional site hypotheses producing a directed cofinal family and a \v{C}ech-to-derived comparison isomorphism, then the classes $[\omega_v(\mathcal U)]$ may be assembled into an intrinsic class; that extra hypothesis is not built into the present definitions.
\end{remark}

\begin{remark}[\v{C}ech refinement maps]\label{rem:cech-refinement-maps}
For a refinement $\mathcal V=\{b_j\to a\}_{j\in J}$ of $\mathcal U=\{a_i\to a\}_{i\in I}$ via an index map $\lambda:J\to I$ with $b_j\sqsubseteq_{p,s}a_{\lambda(j)}$, define
\[
(r_{\mathcal V\mathcal U}h)_{jk}:=h_{\lambda(j)\lambda(k)}\big|_{b_j\wedge b_k}
\qquad
\text{for }h=(h_{ij})\in\check C^1(\mathcal U,\Band),
\]
\[
(r_{\mathcal V\mathcal U}g)_{jkl}:=g_{\lambda(j)\lambda(k)\lambda(l)}\big|_{b_j\wedge b_k\wedge b_l}
\qquad
\text{for }g=(g_{ijk})\in\check C^2(\mathcal U,\Band).
\]
These maps commute with coboundaries and satisfy
\[
r_{\mathcal W\mathcal V}\circ r_{\mathcal V\mathcal U}=r_{\mathcal W\mathcal U}
\]
for every further refinement $\mathcal W$ of $\mathcal V$, so they descend to cohomology.
\end{remark}

\begin{theorem}[Component gerbes of visible branches]\label{thm:component-gerbe-decomposition}
Assume $\mathfrak P_r$ is a gluing-sensitive realization prestack at $a$ in the sense of \Cref{def:gluing-sensitive}. Then for every
\[
v\in \Vis_{p,s}(r)\cong \pi_0(\mathfrak L_r)(a),
\]
the full substack $\mathfrak G_v$ is a banded $\Band$-gerbe over $\mathcal C_{p,s}/a$.
\leanverified{paper_gluing_failure_component_gerbe_decomposition}
\end{theorem}

\begin{proof}
Since $\mathfrak L_r$ is a stack in groupoids and $\mathfrak G_v$ is a full substack, $\mathfrak G_v$ is again a stack in groupoids. Local nonemptiness follows from \Cref{lem:component-local-nonempty}: for every object $U$ of $\mathcal C_{p,s}/a$, there exists a cover $\{U_i\to U\}$ and objects $x_i\in\mathfrak L_r(U_i)$ with $[x_i]=v|_{U_i}$, so $(\mathfrak G_v)(U_i)\neq\varnothing$ for all $i$.

Now let $x,y\in \mathfrak G_v(U)$. Then $[x]=[y]=v|_U\in \pi_0(\mathfrak L_r)(U)$. By the definition of the sheaf of connected components, there exists a cover $\{U_i\to U\}$ such that
\[
x|_{U_i}\cong y|_{U_i}
\]
for every $i$. Thus any two objects of $\mathfrak G_v(U)$ are locally isomorphic, so $\mathfrak G_v$ is a gerbe \cite[Tag 06NY]{StacksProject}. The banding is inherited from $\mathfrak L_r$: since $\mathfrak G_v$ is a full substack, the automorphism sheaf of any object $x\in \mathfrak G_v(U)$ coincides with its automorphism sheaf in $\mathfrak L_r(U)$, hence is identified with $\Band|_U$.
\end{proof}

\begin{definition}[Branch constancy]\label{def:branch-constancy}
In the setting of \Cref{def:compatible-realization-stack}, we say that $\pi_0(\mathfrak L_r)$ is \emph{branch-constant over $a$} if for every object $U$ of $\mathcal C_{p,s}/a$, the restriction map
\[
\rho_U:\pi_0(\mathfrak L_r)(a)\longrightarrow \pi_0(\mathfrak L_r)(U)
\]
is bijective.
\end{definition}

\begin{corollary}[Global branch decomposition under branch constancy]\label{cor:branch-decomposition}
Assume branch constancy over $a$ in the sense of \Cref{def:branch-constancy}. Then the canonical functor
\[
\coprod_{v\in \Vis_{p,s}(r)} \mathfrak G_v\longrightarrow \mathfrak L_r
\]
is an equivalence of stacks in groupoids.
\end{corollary}

\begin{proof}
Let $U\in \mathcal C_{p,s}/a$ and $x\in \mathfrak L_r(U)$. Put $\xi:=[x]\in \pi_0(\mathfrak L_r)(U)$. By surjectivity of $\rho_U$, there exists $v\in \pi_0(\mathfrak L_r)(a)$ such that $\rho_U(v)=\xi$. Hence $x\in \mathfrak G_v(U)$, so the displayed functor is essentially surjective on every fibre.

For full faithfulness, let $x\in \mathfrak G_v(U)$ and $y\in \mathfrak G_w(U)$, and suppose there is a morphism $f:x\to y$ in $\mathfrak L_r(U)$. Any morphism in a groupoid is an isomorphism, hence $[x]=[y]$ in $\pi_0(\mathfrak L_r)(U)$. Therefore $\rho_U(v)=\rho_U(w)$. By injectivity of $\rho_U$, one gets $v=w$. Thus there are no morphisms in $\mathfrak L_r(U)$ between objects of distinct summands, and within each summand the morphism sets coincide because $\mathfrak G_v$ is a full substack.
\end{proof}

\begin{proposition}[Constructing the \v{C}ech obstruction of a branch gerbe]\label{thm:cech-bridge-compatible-realizations}
Assume $\mathfrak P_r$ is a gluing-sensitive realization prestack at $a$, and let $\mathfrak G_v$ be a banded component gerbe with band $\Band$ in the sense of \Cref{def:compatible-realization-stack}. Let
\[
\mathcal U=\{a_i\to a\}_{i\in I}\in \Cov_{p,s}(a)
\]
be a gerbe-adapted cover for $v$ in the sense of \Cref{def:cech-gluing-obstruction}, and write
\[
U_{i_0\cdots i_n}
\]
for the corresponding simplices. Assume
\[
H^1(\mathcal C_a/U_i,\\Band)=0
\qquad\text{and}\qquad
H^1(\mathcal C_a/U_{ij},\\Band)=0
\]
for all $i,j\in I$. Choose objects
\[
x_i\in \mathfrak G_v(U_i).
\]
Then:
\begin{enumerate}[label=(\roman*),nosep]
\item for every pair $(i,j)$ there exists an isomorphism
\[
\phi_{ij}:x_i|_{U_{ij}}\xrightarrow{\sim} x_j|_{U_{ij}};
\]
\item on each triple overlap $U_{ijk}$, the element
\[
g_{ijk}:=\phi_{ik}^{-1}|_{U_{ijk}}\circ \phi_{jk}|_{U_{ijk}}\circ \phi_{ij}|_{U_{ijk}}
\in \Band(U_{ijk})
\]
defines a Cech $2$-cocycle $g=(g_{ijk})\in \check Z^2(\mathcal U,\Band)$;
\item for a fixed cover $\mathcal U$, the class $[g]\in \Cech^2(\mathcal U,\Band)$ depends only on the branch gerbe $\mathfrak G_v$ and on $\mathcal U$; we denote it by $[\omega_v(\mathcal U)]$. Moreover, if $\mathcal V$ refines $\mathcal U$, then the class obtained on $\mathcal V$ from the pulled-back local data is exactly $r_{\mathcal V\mathcal U}([\omega_v(\mathcal U)])$;
\item the class $[g]$ vanishes if and only if the branch gerbe $\mathfrak G_v$ is neutral over $a$.
\end{enumerate}
\end{proposition}

\begin{proof}
Because $\mathfrak G_v$ is a banded gerbe, the sheaf
\[
\mathrm{Isom}(x_i|_{U_{ij}},x_j|_{U_{ij}})
\]
is a $\Band|_{U_{ij}}$-torsor for every pair $(i,j)$. By the standard classification of torsors by $H^1$ \cite[Chap.~III]{MacLaneMoerdijk1994}, \cite[Sec.~11]{StacksProject}, the vanishing
\[
H^1(\mathcal C_a/U_{ij},\\Band)=0
\]
forces this torsor to be trivial. Hence global sections exist, which gives the isomorphisms $\phi_{ij}$ and proves \textup{(i)}.

For \textup{(ii)}, the composite
\[
\phi_{ik}^{-1}|_{U_{ijk}}\circ \phi_{jk}|_{U_{ijk}}\circ \phi_{ij}|_{U_{ijk}}
\]
is an automorphism of $x_i|_{U_{ijk}}$, hence a section of $\Band(U_{ijk})$. The usual associativity calculation on quadruple overlaps shows that $(g_{ijk})$ satisfies the Cech cocycle identity \cite[Chap.~IV]{Giraud1971}, \cite[Sec.~11]{StacksProject}.

For \textup{(iii)}, first change the $\phi_{ij}$ while keeping the objects $x_i$ fixed. Any other choice has the form
\[
\phi'_{ij}=\phi_{ij}\circ h_{ij}^{-1}
\]
for a $1$-cochain $h=(h_{ij})\in \check C^1(\mathcal U,\Band)$, and the corresponding cocycle is
\[
g'=(\delta h)\, g.
\]
Thus the Cech class is unchanged.

Next change the local objects. If $x'_i\in \mathfrak G_v(U_i)$ is another choice, then
\[
\mathrm{Isom}(x_i,x'_i)
\]
is a $\Band|_{U_i}$-torsor. Since $H^1(\mathcal C_a/U_i,\\Band)=0$, this torsor is trivial, so we may choose an isomorphism
\[
\psi_i:x_i\xrightarrow{\sim} x'_i.
\]
Transporting the $\phi_{ij}$ across the $\psi_i$ shows that the cocycle attached to $(x'_i)$ differs from $g$ by a coboundary.

Now let $\mathcal V=\{b_j\to a\}_{j\in J}$ be a refinement of $\mathcal U$ via $\lambda:J\to I$. Pull back the local data by setting
\[
y_j:=x_{\lambda(j)}|_{b_j},
\qquad
\theta_{jk}:=\phi_{\lambda(j)\lambda(k)}|_{b_j\wedge b_k}.
\]
The cocycle attached to $(y_j,\theta_{jk})$ is
\[
(r_{\mathcal V\mathcal U}g)_{jkl}
=
g_{\lambda(j)\lambda(k)\lambda(l)}\big|_{b_j\wedge b_k\wedge b_l},
\]
exactly as in \Cref{rem:cech-refinement-maps}. The identities
\[
r_{\mathcal W\mathcal V}\circ r_{\mathcal V\mathcal U}=r_{\mathcal W\mathcal U},
\qquad
r_{\mathcal V\mathcal U}(\delta h)=\delta(r_{\mathcal V\mathcal U}h)
\]
show that the induced cohomology classes are functorial under further refinement and remain independent of the auxiliary choices.

If the gerbe is neutral, choose a global object $x\in \mathfrak G_v(a)$. Then each
\[
\mathrm{Isom}(x|_{U_i},x_i)
\]
is a $\Band|_{U_i}$-torsor, hence trivial by $H^1(\mathcal C_a/U_i,\\Band)=0$. Choose isomorphisms
\[
\psi_i:x|_{U_i}\xrightarrow{\sim} x_i.
\]
With
\[
\phi_{ij}:=\psi_j|_{U_{ij}}\circ \psi_i^{-1}|_{U_{ij}},
\]
the resulting cocycle is identically zero. Thus neutrality implies $[g]=0$.

Conversely, if $[g]=0$, choose a $1$-cochain $h=(h_{ij})$ with
\[
g=\delta h.
\]
Replacing $\phi_{ij}$ by
\[
\phi'_{ij}:=\phi_{ij}\circ h_{ij}^{-1},
\]
we obtain isomorphisms satisfying the exact cocycle condition on triple overlaps. Since $\mathfrak G_v$ is a stack, the descent datum $(x_i,\phi'_{ij})$ is effective and glues to a global object over $a$. Hence the gerbe is neutral.
\end{proof}

\begin{theorem}[Branched gerbe semantics of gluing-level absence]\label{thm:gerbe-null-semantics}
Assume that $\mathfrak P_r$ is a gluing-sensitive realization prestack at $a$ in the sense of \Cref{def:gluing-sensitive}. Then, for the admitted reference $r:\mathsf{Ref}_s$ under consideration,
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
\Vis_{p,s}(r)\neq\varnothing.
\]
Moreover,
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\exists v\in \Vis_{p,s}(r)\ \text{such that}\ \mathfrak G_v\ \text{is neutral}.
\]
Consequently,
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
\Vis_{p,s}(r)\neq\varnothing
\ \text{and}\
\forall v\in \Vis_{p,s}(r),\ \mathfrak G_v\ \text{is non-neutral}.
\]
If, in addition, for each visible branch $v$ one chooses a gerbe-adapted cover $\mathcal U_v$ in the sense of \Cref{def:cech-gluing-obstruction}, then the last condition is equivalent to
\[
\Vis_{p,s}(r)\neq\varnothing
\ \text{and}\
[\,\omega_v(\mathcal U_v)\,]\neq 0\ \text{for every}\ v\in \Vis_{p,s}(r).
\]
In particular, the single obstruction class picture is recovered precisely when $|\Vis_{p,s}(r)|=1$.
\leanverified{paper_gluing_failure_gerbe_null_semantics}
\end{theorem}

\begin{proof}
The first equivalence is the combination of \Cref{thm:sheafification-characterization,thm:visible-value-components}. Indeed, compatible local sectionability is equivalent to the nonemptiness of $(a_{p,s}F_{p,s})(a)$, and this set is exactly $\Vis_{p,s}(r)$.

For the second equivalence, \Cref{prop:no-new-global-objects} gives
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\mathfrak L_r(a)\neq\varnothing.
\]
If $x\in \mathfrak L_r(a)$, then $x$ determines a visible value class
\[
v=[x]\in \pi_0(\mathfrak L_r)(a)\cong \Vis_{p,s}(r),
\]
and $x\in \mathfrak G_v(a)$, so $\mathfrak G_v$ is neutral. Conversely, if $\mathfrak G_v$ is neutral for some visible branch $v$, then any global object of $\mathfrak G_v(a)$ lies in $\mathfrak L_r(a)$, so $\mathfrak L_r(a)\neq\varnothing$.

Since $r$ is admitted, the definition
\[
\Null^{\mathrm{glue}}_s(r):=\Adm_s(r)\land \CompSec_s(r)\land \neg\Sec_s(r)
\]
now yields the third equivalence immediately.

For the final claim, \Cref{thm:cech-bridge-compatible-realizations}(iv) identifies the vanishing of each cover-relative class $[\omega_v(\mathcal U_v)]$ with neutrality of the corresponding component gerbe. If $|\Vis_{p,s}(r)|=1$, there is therefore only one chosen branch obstruction class to test.
\end{proof}
