\section{Local Objects, Visible Branches, and Gluing Failure}\label{sec:null-decomposition}

We next distinguish admissible reference from objecthood. An admitted reference need not yet determine a global object, but it may still support local witnesses and compatible local families. This local-to-global gap is the source of the structural phenomena studied in the present section.

\begin{definition}[Refinement stability]\label{def:refinement-stability}
We say that the local object system is \emph{refinement-stable} if, for every refinement $q\le p$:
\begin{enumerate}[label=(\roman*),nosep]
\item admitted references, covers, and compatible local families at $q$ restrict functorially to those at $p$;
\item a visible value class already uniquely determined at $p$ is not split by further refinement, i.e.\ if $|\Vis_{p,s}(r)|=1$ then $|\Vis_{q,s}(r)|=1$ and the unique class at $q$ restricts to the unique class at $p$.
\end{enumerate}
\end{definition}

Throughout the section we assume that the local object system is refinement-stable in the sense of \Cref{def:refinement-stability}.

\subsection{References and local objects}

Fix a family of referenceable sorts
\[
S_{\mathrm{ref}}\subseteq S.
\]
For each $s\in S_{\mathrm{ref}}$, the language contains a reference sort $\mathsf{Ref}_s$ and an admissibility predicate $\Adm_s(\cdot)$.

\begin{definition}[Reference classes]\label{def:reference-classes}
Let $p$ be a state and let $r,r'$ be reference terms of sort $\mathsf{Ref}_s$. Define
\[
r\equiv_p r'
\quad:\Longleftrightarrow\quad
M,p\Vdash (r=r').
\]
The set of admitted reference classes at $p$ is
\[
\mathcal{R}_s(p):=\{[r]_p\mid M,p\Vdash \Adm_s(r)\}.
\]
\end{definition}

\begin{definition}[Local reference frame and site]\label{def:local-reference-frame}
For a state $p$ and a referenceable sort $s$, a \emph{local reference frame} is a triple
\[
\mathcal{B}_{p,s}=\bigl(\mathcal{R}_s(p),\sqsubseteq_{p,s},\Cov_{p,s}\bigr),
\]
where:
\begin{enumerate}[label=(\roman*),nosep]
\item $\bigl(\mathcal{R}_s(p),\sqsubseteq_{p,s}\bigr)$ is a \emph{meet-semilattice}, viewed as a thin category: objects are admitted reference classes, and there is a unique morphism $b\to a$ whenever $b\sqsubseteq_{p,s} a$. In particular, finite meets $a_i\wedge a_j$ exist for every pair $a_i,a_j$.
\item $\Cov_{p,s}$ is a \emph{Grothendieck pretopology} on the category $\mathcal{R}_s(p)$: it assigns to each $a\in\mathcal{R}_s(p)$ a collection $\Cov_{p,s}(a)$ of \emph{covering families} $\{a_i\to a\}_{i\in I}$ (where each $a_i\sqsubseteq_{p,s} a$) satisfying:
  \begin{enumerate}[label=(\alph*),nosep]
  \item \emph{Identity:} $\{\mathrm{id}_a:a\to a\}\in\Cov_{p,s}(a)$ for every $a$.
  \item \emph{Stability:} if $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and $b\sqsubseteq_{p,s} a$, then $\{a_i\wedge b\to b\}_{i\in I}\in\Cov_{p,s}(b)$.
  \item \emph{Transitivity:} if $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and for each $i$, $\{b_{ij}\to a_i\}_{j\in J_i}\in\Cov_{p,s}(a_i)$, then $\{b_{ij}\to a\}_{i,j}\in\Cov_{p,s}(a)$.
  \end{enumerate}
\end{enumerate}
The resulting site is denoted $\mathcal{C}_{p,s}:=\bigl(\mathcal{R}_s(p),\Cov_{p,s}\bigr)$. The existence of finite meets allows one to define Cech nerves, slice sites $\mathcal{C}_{p,s}/a$, and sheafification over $\mathcal{C}_{p,s}$ by standard means \cite[Chap.~III]{MacLaneMoerdijk1994}.
\end{definition}

\begin{definition}[Local object functor]\label{def:local-object-functor}
A \emph{local object functor} on $(p,s)$ is a contravariant functor
\[
F_{p,s}:\mathcal{C}_{p,s}^{op}\to\mathsf{Set}.
\]
Since $\mathcal{C}_{p,s}$ is the meet-semilattice with the pretopology of \Cref{def:local-reference-frame}, functoriality amounts to: for each $b\sqsubseteq_{p,s} a$, there is a restriction map
\[
\res_b^a:F_{p,s}(a)\to F_{p,s}(b),
\]
and $\res_a^a=\mathrm{id}$ and $\res_c^b\circ\res_b^a=\res_c^a$ for $c\sqsubseteq_{p,s}b\sqsubseteq_{p,s}a$.
\end{definition}

\begin{definition}[Three existence levels]\label{def:existence-levels}
Let $r:\mathsf{Ref}_s$ and assume $M,p\Vdash \Adm_s(r)$. Write $a=[r]_p$.
\begin{enumerate}[label=(\roman*),nosep]
\item $M,p\Vdash \LocSec_s(r)$ if there exists a cover $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ such that $F_{p,s}(a_i)\neq\varnothing$ for every $i\in I$.
\item $M,p\Vdash \CompSec_s(r)$ if there exist a cover $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and sections $\sigma_i\in F_{p,s}(a_i)$ such that for every $i,j\in I$,
\[
\res_{a_i\wedge a_j}^{a_i}(\sigma_i)=\res_{a_i\wedge a_j}^{a_j}(\sigma_j),
\]
where $a_i\wedge a_j$ is the meet in the semilattice $\mathcal{R}_s(p)$. This is the standard matching-family condition: sections agree on pairwise meets, which are the canonical overlaps supplied by the semilattice structure.
\item $M,p\Vdash \Sec_s(r)$ if $F_{p,s}(a)\neq\varnothing$.
\end{enumerate}
\end{definition}

\begin{proposition}[Global sections yield compatible local sections]\label{prop:global-implies-compatible-local}
If $M,p\Vdash \Sec_s(r)$, then
\[
M,p\Vdash \CompSec_s(r)
\qquad\text{and}\qquad
M,p\Vdash \LocSec_s(r).
\]
\end{proposition}

\begin{proof}
Let $a=[r]_p$ and choose $\sigma\in F_{p,s}(a)$. The identity cover $\{\mathrm{id}_a:a\to a\}$ is available by \Cref{def:local-reference-frame}(ii)(a). The singleton family $(\sigma)$ satisfies the compatibility condition vacuously (the only meet is $a\wedge a=a$, and $\res_a^a(\sigma)=\sigma$). Hence both compatible local sectionability and local sectionability follow.
\end{proof}

\subsection{Sheafification and gluing failure}

For fixed $(p,s)$, the site $\mathcal{C}_{p,s}$ has been specified in \Cref{def:local-reference-frame}, and the local object functor of \Cref{def:local-object-functor} determines a presheaf
\[
F_{p,s}\in\mathbf{PSh}(\mathcal{C}_{p,s}).
\]
Write
\[
a_{p,s}:\mathbf{PSh}(\mathcal{C}_{p,s})\to\mathbf{Sh}(\mathcal{C}_{p,s})
\]
for the associated sheafification functor, which exists because $\mathcal{C}_{p,s}$ carries a Grothendieck pretopology \cite[Chap.~III, \S5]{MacLaneMoerdijk1994}.

\begin{proposition}[Sheafification characterizes compatible local sectionability]\label{thm:sheafification-characterization}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and write $a=[r]_p$. Then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
\bigl(a_{p,s}F_{p,s}\bigr)(a)\neq\varnothing.
\]
\end{proposition}

\begin{proof}
By the standard site-theoretic description of the associated sheaf on a pretopology site, a section of $a_{p,s}F_{p,s}$ at $a$ is an equivalence class of matching families over covering sieves of $a$ \cite[Chap.~III, \S2]{MacLaneMoerdijk1994}, \cite[Part~B2.2]{Johnstone2002}. A matching family over a cover $\{a_i\to a\}_{i\in I}$ consists of sections $\sigma_i\in F_{p,s}(a_i)$ such that $\res_{a_i\wedge a_j}^{a_i}(\sigma_i)=\res_{a_i\wedge a_j}^{a_j}(\sigma_j)$ for all $i,j$. Because $\mathcal{C}_{p,s}$ is a meet-semilattice with the pretopology of \Cref{def:local-reference-frame}, these pairwise meet restrictions are the canonical overlaps, and the matching condition is exactly the compatibility condition of \Cref{def:existence-levels}(ii). Hence the sheafified fiber $(a_{p,s}F_{p,s})(a)$ is nonempty if and only if some covering family supports a compatible local family, i.e., if and only if $M,p\Vdash \CompSec_s(r)$.
\end{proof}

\begin{corollary}[Sheaf condition at a reference]\label{cor:sheaf-local-to-global}
If $F_{p,s}$ is already a sheaf at $a=[r]_p$, then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s(r).
\]
\end{corollary}

\begin{proof}
If $F_{p,s}$ is already a sheaf at $a$, then the unit $F_{p,s}\to a_{p,s}F_{p,s}$ is an isomorphism at $a$. The claim therefore follows from \Cref{thm:sheafification-characterization}.
\end{proof}

\begin{corollary}[Sheafification removes gluing failure]\label{cor:sheafification-removes-glue}
Let $\Sec_s^{\sharp}(r)$ denote global sectionability computed with the sheafified local object functor $a_{p,s}F_{p,s}$. Then
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
M,p\Vdash \Sec_s^{\sharp}(r).
\]
In particular, no admitted reference remains in gluing failure after sheafification.
\end{corollary}

\begin{proof}
This is \Cref{thm:sheafification-characterization} rewritten for the sheaf $a_{p,s}F_{p,s}$. Once the presheaf has been replaced by a sheaf, compatible local existence and global existence coincide.
\end{proof}

\subsection{Realization prestacks and connected components}

The presheaf $F_{p,s}$ records only the isomorphism classes of local realizations. To pass from compatible local families to gerbes and cohomology, one needs a groupoid-valued lift whose connected components recover $F_{p,s}$ after sheafification. The first point is purely categorical: once an abelian band has been fixed on the slice site, such lifts always exist. The substantive issue is therefore not bare existence, but whether one has chosen a lift whose stackification still reflects the intended gluing semantics at the terminal fibre.

\begin{definition}[Realization prestack and stackification]\label{def:realization-prestack}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, and write
\[
a=[r]_p,
\qquad
\mathcal C_a:=\mathcal C_{p,s}/a.
\]
A \emph{realization prestack} for $r$ at $p$ is a category fibred in groupoids
\[
\mathfrak P_r\to \mathcal C_a
\]
such that:
\begin{enumerate}[label=(\roman*),nosep]
\item the presheaf of isomorphism classes
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r):U\longmapsto \Ob(\mathfrak P_r(U))/\!\cong
\]
is identified with the restriction of the local object functor,
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r)\cong F_{p,s}|_{\mathcal C_a};
\]
\item for every object $U$ of $\mathcal C_a$ and every $x\in \mathfrak P_r(U)$, the automorphism sheaf
\[
\underline{\mathrm{Aut}}(x)
\]
on $\mathcal C_a/U$ is functorially identified with the restriction of a fixed abelian sheaf $\Band$ on $\mathcal C_a$.
\end{enumerate}
Write
\[
\mathfrak L_r:=a(\mathfrak P_r)
\]
for the stackification of $\mathfrak P_r$ in groupoids \cite[Tag 02ZP]{StacksProject}.
\end{definition}

\begin{proposition}[Canonical split realization prestack]\label{prop:split-realization-prestack}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write
\[
a=[r]_p,
\qquad
\mathcal C_a:=\mathcal C_{p,s}/a,
\]
and let $\Band$ be any abelian sheaf on $\mathcal C_a$. Then there exists a realization prestack
\[
\mathfrak P_r^{\mathrm{split}}\to \mathcal C_a
\]
with band $\Band$ such that
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r^{\mathrm{split}})
\cong
F_{p,s}|_{\mathcal C_a}.
\]
Explicitly, for each object $U$ of $\mathcal C_a$, the fibre
\[
\mathfrak P_r^{\mathrm{split}}(U)
\]
has object set $F_{p,s}(U)$ and morphism sets
\[
\mathrm{Hom}_{\mathfrak P_r^{\mathrm{split}}(U)}(\sigma,\tau)
:=
\begin{cases}
\Band(U), & \sigma=\tau,\\
\varnothing, & \sigma\neq\tau,
\end{cases}
\]
with composition given by the group law on $\Band(U)$.
\end{proposition}

\begin{proof}
For a morphism $f:V\to U$ in $\mathcal C_a$, define pullback on objects by
\[
f^*(\sigma):=\res_V^U(\sigma)\in F_{p,s}(V)
\]
and on automorphisms by the sheaf restriction
\[
f^*:\Band(U)\to \Band(V).
\]
This gives a pseudofunctor $\mathcal C_a^{op}\to \mathbf{Grpd}$ and hence a category fibred in groupoids over $\mathcal C_a$. For any two objects $\sigma,\tau\in \mathfrak P_r^{\mathrm{split}}(U)$, the isomorphism presheaf on $\mathcal C_a/U$ is either $\Band|_U$ or the empty presheaf. Since $\Band$ is a sheaf and the empty presheaf is also a sheaf, $\mathfrak P_r^{\mathrm{split}}$ is a prestack.

By construction, two objects of $\mathfrak P_r^{\mathrm{split}}(U)$ are isomorphic if and only if they are equal as elements of $F_{p,s}(U)$. Therefore
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r^{\mathrm{split}})(U)\cong F_{p,s}(U)
\]
for every $U$, giving condition \textup{(i)} of \Cref{def:realization-prestack}. Condition \textup{(ii)} holds because the automorphism sheaf of every object $\sigma$ is exactly $\Band|_U$.
\end{proof}

\begin{proposition}[Components of the stackification]\label{prop:stackification-components}
In the setting of \Cref{def:realization-prestack}, there is a canonical isomorphism of sheaves on $\mathcal C_a$:
\[
\pi_0(\mathfrak L_r)\cong a_{p,s}F_{p,s}|_{\mathcal C_a}.
\]
\end{proposition}

\begin{proof}
By stackification, the morphism presheaves of $\mathfrak P_r$ are sheafified and every object of $\mathfrak L_r$ is locally in the essential image of $\mathfrak P_r\to \mathfrak L_r$ \cite[Tag 02ZP]{StacksProject}. Passing to isomorphism classes therefore shows that $\pi_0(\mathfrak L_r)$ is the sheafification of $\pi_0^{\mathrm{pre}}(\mathfrak P_r)$. Since stacks in setoids are precisely sheaves of sets \cite[Tag 042Y]{StacksProject}, we obtain
\[
\pi_0(\mathfrak L_r)\cong a\bigl(\pi_0^{\mathrm{pre}}(\mathfrak P_r)\bigr).
\]
Using \Cref{def:realization-prestack}(i), this becomes
\[
\pi_0(\mathfrak L_r)\cong a_{p,s}F_{p,s}|_{\mathcal C_a}.
\qedhere
\]
\end{proof}

\begin{definition}[Global conservativity at the terminal fibre]\label{def:global-conservativity}
In the setting of \Cref{def:realization-prestack}, let
\[
\iota_r:\mathfrak P_r\to \mathfrak L_r:=a(\mathfrak P_r)
\]
be the stackification morphism. We say that $\mathfrak P_r$ is \emph{globally conservative at $a$} if the induced functor on the terminal fibre
\[
(\iota_r)_a:(\mathfrak P_r)_a\to (\mathfrak L_r)_a
\]
is essentially surjective.
\end{definition}

\begin{proposition}[No new global objects at $a$]\label{prop:no-new-global-objects}
Assume global conservativity at $a$ in the sense of \Cref{def:global-conservativity}. Then
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\mathfrak L_r(a)\neq\varnothing.
\]
\end{proposition}

\begin{proof}
By \Cref{def:existence-levels} and \Cref{def:realization-prestack}(i),
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
F_{p,s}(a)\neq\varnothing
\quad\Longleftrightarrow\quad
\pi_0^{\mathrm{pre}}(\mathfrak P_r)(a)\neq\varnothing
\quad\Longleftrightarrow\quad
(\mathfrak P_r)_a\neq\varnothing.
\]
If $(\mathfrak P_r)_a\neq\varnothing$, choose $y\in (\mathfrak P_r)_a$. Then $\iota_r(y)\in \mathfrak L_r(a)$, so $\mathfrak L_r(a)\neq\varnothing$. Conversely, if $x\in \mathfrak L_r(a)$, essential surjectivity of $(\iota_r)_a$ yields $y\in (\mathfrak P_r)_a$ with $\iota_r(y)\cong x$, so $(\mathfrak P_r)_a\neq\varnothing$.
\end{proof}

\begin{definition}[Gluing-sensitive realization prestack]\label{def:gluing-sensitive}
In the setting of \Cref{def:realization-prestack}, the realization prestack $\mathfrak P_r\to\mathcal C_a$ is called \emph{gluing-sensitive at $a$} if it satisfies the following conditions.
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{Connected-component agreement.} The canonical map
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r)\cong F_{p,s}|_{\mathcal C_a}
\]
of \Cref{def:realization-prestack}(i) induces, after passing to the associated sheaf, an isomorphism
\[
\pi_0(\mathfrak L_r)\cong a_{p,s}F_{p,s}|_{\mathcal C_a}.
\]
This is automatic from \Cref{prop:stackification-components} once the conditions of \Cref{def:realization-prestack} hold.
\item \emph{Terminal-fiber essential surjectivity.} The stackification morphism $(\iota_r)_a:(\mathfrak P_r)_a\to(\mathfrak L_r)_a$ is essentially surjective (i.e., $\mathfrak P_r$ is globally conservative at $a$ in the sense of \Cref{def:global-conservativity}).
\end{enumerate}
Conditions (i) and (ii) ensure that the gerbe semantics of gluing failure (\Cref{thm:gerbe-null-semantics}) is valid. The canonical split prestack of \Cref{prop:split-realization-prestack} is \emph{not} gluing-sensitive in general: see \Cref{rem:split-not-conservative} below.
\end{definition}

\begin{lemma}[Component substacks are locally nonempty]\label{lem:component-local-nonempty}
Let $\mathfrak P_r$ be gluing-sensitive at $a$. For every visible value class $v\in\pi_0(\mathfrak L_r)(a)$ and every object $U$ of $\mathcal C_a$, there exists a cover $\{U_i\to U\}$ such that $(\mathfrak G_v)(U_i)\neq\varnothing$ for all $i$.
\end{lemma}

\begin{proof}
Since $v|_U\in\pi_0(\mathfrak L_r)(U)$ is a section of the sheaf of connected components, the definition of $\pi_0(\mathfrak L_r)$ yields a cover $\{U_i\to U\}$ and objects $x_i\in\mathfrak L_r(U_i)$ with $[x_i]=v|_{U_i}$. Each $x_i$ belongs to the fibre of $\mathfrak G_v$ over $U_i$, so $(\mathfrak G_v)(U_i)\neq\varnothing$.
\end{proof}

\begin{remark}[The split prestack is not globally conservative in general]\label{rem:split-not-conservative}
Consider the local reference frame with $u,v\to a$ and $c=u\wedge v$, and define
\[
F(u)=F(v)=F(c)=\{*\},\qquad F(a)=\varnothing.
\]
The canonical split prestack satisfies $(\mathfrak P_r^{\mathrm{split}})_a=\varnothing$ because $F(a)=\varnothing$. However, the stackification creates a global object at $a$: the compatible family $(*,*)$ over the cover $\{u,v\}$ glues after stackification, so $(\mathfrak L_r)_a\neq\varnothing$. The terminal-fiber map $(\mathfrak P_r^{\mathrm{split}})_a\to(\mathfrak L_r)_a$ is therefore empty-to-nonempty, hence not essentially surjective. This example shows that the split prestack is not gluing-sensitive, and explains why the branch-gerbe theory requires a substantive hypothesis rather than merely appealing to the canonical split lift.
\end{remark}

\begin{remark}[What remains conditional]\label{rem:gluing-sensitive-lifts}
\Cref{prop:split-realization-prestack} shows that admitting some banded realization prestack is not an additional existence burden: once a band has been fixed, a canonical split lift is always available. What remains conditional in the later branch-obstruction theory is the existence of a gluing-sensitive lift in the sense of \Cref{def:gluing-sensitive}. No general existence theorem for such lifts is proved in this paper, and the results of \Cref{sec:gerbe-obstruction,sec:homological-visibility} should therefore be read as a conditional framework that starts only after a gluing-sensitive lift at $a$ has been chosen.
\end{remark}

\subsection{Visible value classes and typed readouts}

\begin{definition}[Visible value classes]\label{def:visible-value-classes}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and write $a=[r]_p$. A \emph{visible presentation} of a value at $r$ is a compatible local family
\[
\bigl(\{a_i\to a\}_{i\in I},(\sigma_i)_{i\in I}\bigr)
\]
as in \Cref{def:existence-levels}(ii), i.e., sections $\sigma_i\in F_{p,s}(a_i)$ satisfying $\res_{a_i\wedge a_j}^{a_i}(\sigma_i)=\res_{a_i\wedge a_j}^{a_j}(\sigma_j)$ for all $i,j$. Two visible presentations $(\{a_i\},(\sigma_i))$ and $(\{b_j\},(\tau_j))$ are called \emph{value-equivalent} if there exists a common refinement $\{c_k\to a\}_{k\in K}$ of both covers (which exists by the transitivity and stability axioms of $\Cov_{p,s}$) such that the pulled-back sections agree: $\res_{c_k}^{a_i}(\sigma_i)=\res_{c_k}^{b_j}(\tau_j)$ whenever $c_k\sqsubseteq_{p,s} a_i$ and $c_k\sqsubseteq_{p,s} b_j$. A \emph{visible value class} is an equivalence class of visible presentations under this relation. The set of visible value classes of $r$ at $p$ is denoted by $\Vis_{p,s}(r)$.
\end{definition}

\begin{proposition}[Visible value classes as global connected-component sections]\label{thm:visible-value-components}
Assume that $r$ admits a realization prestack in the sense of \Cref{def:realization-prestack}. Then
\[
\Vis_{p,s}(r)\cong \bigl(a_{p,s}F_{p,s}\bigr)(a)\cong \pi_0(\mathfrak L_r)(a).
\]
\end{proposition}

\begin{proof}
On the pretopology site $\mathcal{C}_{p,s}$, a section of the associated sheaf $a_{p,s}F_{p,s}$ at $a$ is an equivalence class of matching families over covers of $a$, where two matching families over covers $\{a_i\}$ and $\{b_j\}$ are equivalent if they agree on a common refinement (which exists by the transitivity and stability axioms of $\Cov_{p,s}$) \cite[Chap.~III, \S2]{MacLaneMoerdijk1994}. This is exactly the equivalence relation on visible presentations in \Cref{def:visible-value-classes}. Hence
\[
\Vis_{p,s}(r)\cong \bigl(a_{p,s}F_{p,s}\bigr)(a).
\]
The second identification is \Cref{prop:stackification-components} evaluated at $a$.
\end{proof}

\begin{definition}[Typed readout]\label{def:typed-readout}
For a reference $r:\mathsf{Ref}_s$ at a state $p$, define the typed readout
\[
\Read_{p,s}(r)
\]
by the following cases:
\[
\Read_{p,s}(r)=
\begin{cases}
\text{undefined}, & M,p\nVdash \Adm_s(r),\\[0.5ex]
\Null, & M,p\Vdash \Adm_s(r)\ \text{and}\ \Vis_{p,s}(r)=\varnothing,\\[0.5ex]
v, & \Vis_{p,s}(r)=\{v\},\\[0.5ex]
\text{undefined}, & |\Vis_{p,s}(r)|>1.
\end{cases}
\]
\end{definition}

\begin{proposition}[Address before value]\label{prop:address-before-value}
If
\[
\Read_{p,s}(r)=v
\qquad\text{with}\qquad
v\neq \Null,
\]
then
\[
M,p\Vdash \Adm_s(r)
\qquad\text{and}\qquad
M,p\Vdash \CompSec_s(r).
\]
\end{proposition}

\begin{proof}
By \Cref{def:typed-readout}, a non-null readout can occur only in the singleton case $\Vis_{p,s}(r)=\{v\}$. By \Cref{def:visible-value-classes}, every element of $\Vis_{p,s}(r)$ is represented by a compatible local family, so $M,p\Vdash \CompSec_s(r)$. Admissibility is built into the definition of $\Vis_{p,s}(r)$, since visible value classes are defined only for admitted references.
\end{proof}

\begin{corollary}[Non-null readout criterion]\label{cor:nonnull-readout-criterion}
If
\[
M,p\Vdash \Adm_s(r)
\qquad\text{and}\qquad
\Vis_{p,s}(r)=\{v\},
\]
then
\[
\Read_{p,s}(r)=v.
\]
\end{corollary}

\begin{proof}
Immediate from \Cref{def:typed-readout}.
\end{proof}

\begin{proposition}[Refinement stability of defined readouts]\label{prop:typed-readout-persistence}
Let $q\le p$, and suppose
\[
\Read_{p,s}(r)=v
\qquad\text{with}\qquad
v\neq \Null.
\]
Then
\[
\Read_{q,s}(r)=v.
\]
\end{proposition}

\begin{proof}
Since $v\neq\Null$, the readout at $p$ is represented by a unique visible value class. By refinement stability (\Cref{def:refinement-stability}), compatible local families restrict functorially from $q$ to $p$, and a uniquely determined visible class at $p$ cannot split under further refinement. Hence the same class $v$ remains the unique visible class at $q$, so \Cref{def:typed-readout} gives $\Read_{q,s}(r)=v$.
\end{proof}

\subsection{Structural absence}

\begin{definition}[Undefined, total, and null]\label{def:undef-total-null}
For $r:\mathsf{Ref}_s$, define
\[
\Undef_s(r):=\neg\Adm_s(r),\qquad
\Tot_s(r):=\Adm_s(r)\land \Sec_s(r),\qquad
\Null_s(r):=\Adm_s(r)\land \neg\Sec_s(r).
\]
\end{definition}

\begin{definition}[The three null modes]\label{def:null-failure-modes}
For $r:\mathsf{Ref}_s$, define
\[
\Null^{\mathrm{loc}}_s(r):=\Adm_s(r)\land \neg\LocSec_s(r),
\]
\[
\Null^{\mathrm{cmp}}_s(r):=\Adm_s(r)\land \LocSec_s(r)\land \neg\CompSec_s(r),
\]
\[
\Null^{\mathrm{glue}}_s(r):=\Adm_s(r)\land \CompSec_s(r)\land \neg\Sec_s(r).
\]
\end{definition}

\begin{proposition}[The three modes lie inside null]\label{prop:null-modes-imply-null}
Each of $\Null^{\mathrm{loc}}_s(r)$, $\Null^{\mathrm{cmp}}_s(r)$, and $\Null^{\mathrm{glue}}_s(r)$ implies $\Null_s(r)$.
\end{proposition}

\begin{proof}
The gluing case is immediate from \Cref{def:null-failure-modes}. In the local and compatibility cases, a global section would imply compatible local sectionability and local sectionability by \Cref{prop:global-implies-compatible-local}, contradicting the defining negations.
\end{proof}

\begin{proposition}[Mutual exclusivity of the null modes]\label{prop:null-modes-exclusive}
No two distinct null modes can hold simultaneously for the same reference at the same state.
\end{proposition}

\begin{proof}
The mode $\Null^{\mathrm{loc}}_s(r)$ contains $\neg\LocSec_s(r)$, while the other two contain $\LocSec_s(r)$. The mode $\Null^{\mathrm{cmp}}_s(r)$ contains $\neg\CompSec_s(r)$, while $\Null^{\mathrm{glue}}_s(r)$ contains $\CompSec_s(r)$.
\end{proof}

\begin{definition}[Decomposition decidability]\label{def:decomposition-decidability}
We say that $p$ has \emph{decomposition decidability} for $r:\mathsf{Ref}_s$ if both $\LocSec_s(r)$ and $\CompSec_s(r)$ are decided at $p$.
\end{definition}

\begin{proposition}[Null trichotomy]\label{prop:null-trichotomy}
If $p$ has decomposition decidability for $r:\mathsf{Ref}_s$, then
\[
M,p\Vdash \Null_s(r)
\]
if and only if exactly one of
\[
M,p\Vdash \Null^{\mathrm{loc}}_s(r),\qquad
M,p\Vdash \Null^{\mathrm{cmp}}_s(r),\qquad
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\]
holds.
\end{proposition}

\begin{proof}
Assume $M,p\Vdash \Null_s(r)$. Then $r$ is admitted but not globally sectionable. Since $\LocSec_s(r)$ and $\CompSec_s(r)$ are decided at $p$, there are three cases: either local sectionability already fails, local sectionability holds but compatibility fails, or compatibility holds while global sectionability fails. These are exactly the three null modes. The converse follows from \Cref{prop:null-modes-imply-null}, and uniqueness is \Cref{prop:null-modes-exclusive}.
\end{proof}

\begin{proposition}[Classification after refinement]\label{thm:null-classification-under-refinement}
Let $q\le p$ and suppose
\[
M,p\Vdash \Null_s(r).
\]
If $q$ has decomposition decidability for $r$, then exactly one null mode holds at $q$.
\end{proposition}

\begin{proof}
By persistence of forcing, $M,q\Vdash \Null_s(r)$. The hypothesis gives decomposition decidability at $q$, so \Cref{prop:null-trichotomy} applies at the refined state $q$.
\end{proof}

\begin{proposition}[A finite model for the three absence modes]\label{prop:finite-three-mode-model}
There exist a finite state $p$ and admitted references
\[
r_{\mathrm{loc}},\qquad r_{\mathrm{cmp}},\qquad r_{\mathrm{glue}}
\]
of one referenceable sort such that
\[
M,p\Vdash \Null^{\mathrm{loc}}_s(r_{\mathrm{loc}}),\qquad
M,p\Vdash \Null^{\mathrm{cmp}}_s(r_{\mathrm{cmp}}),\qquad
M,p\Vdash \Null^{\mathrm{glue}}_s(r_{\mathrm{glue}}).
\]
\end{proposition}

\begin{proof}
Take a single state $p$ and one referenceable sort $s$. Let the local reference frame consist of three rooted components together with a new bottom element $0$ that lies below every object. Meets inside a component are the usual ones, and the meet of objects belonging to different components is declared to be $0$. Put $F(0)=\{*\}$ and let every restriction map to $0$ send its source element to $*$. Covers are generated by the identity covers and the component-specific covers described below; stability at $0$ holds because every pullback of a cover along $0\to a$ is the trivial cover of $0$.

For $r_{\mathrm{loc}}$, use one object $a_{\mathrm{loc}}$ with only the trivial cover and define
\[
F(a_{\mathrm{loc}})=\varnothing.
\]
Then local sectionability already fails, so $M,p\Vdash \Null^{\mathrm{loc}}_s(r_{\mathrm{loc}})$.

For $r_{\mathrm{cmp}}$, use objects $u_{\mathrm{cmp}},v_{\mathrm{cmp}}\sqsubseteq_{p,s} a_{\mathrm{cmp}}$ with $c_{\mathrm{cmp}}:=u_{\mathrm{cmp}}\wedge v_{\mathrm{cmp}}$ (the meet in the semilattice), and let $\{u_{\mathrm{cmp}}\to a_{\mathrm{cmp}},v_{\mathrm{cmp}}\to a_{\mathrm{cmp}}\}\in\Cov_{p,s}(a_{\mathrm{cmp}})$. Define
\[
F(u_{\mathrm{cmp}})=\{\sigma_u\},\quad
F(v_{\mathrm{cmp}})=\{\sigma_v\},\quad
F(c_{\mathrm{cmp}})=\{0,1\},\quad
F(a_{\mathrm{cmp}})=\varnothing,
\]
with restrictions $\res_{c_{\mathrm{cmp}}}^{u_{\mathrm{cmp}}}(\sigma_u)=0$ and $\res_{c_{\mathrm{cmp}}}^{v_{\mathrm{cmp}}}(\sigma_v)=1$. Local sections exist on the chosen cover, but the compatibility check at the meet $c_{\mathrm{cmp}}=u_{\mathrm{cmp}}\wedge v_{\mathrm{cmp}}$ fails: $0\neq 1$. Hence $M,p\Vdash \Null^{\mathrm{cmp}}_s(r_{\mathrm{cmp}})$.

For $r_{\mathrm{glue}}$, use objects $u_{\mathrm{glue}},v_{\mathrm{glue}}\sqsubseteq_{p,s} a_{\mathrm{glue}}$ with $c_{\mathrm{glue}}:=u_{\mathrm{glue}}\wedge v_{\mathrm{glue}}$, and let $\{u_{\mathrm{glue}}\to a_{\mathrm{glue}},v_{\mathrm{glue}}\to a_{\mathrm{glue}}\}\in\Cov_{p,s}(a_{\mathrm{glue}})$. Define
\[
F(u_{\mathrm{glue}})=F(v_{\mathrm{glue}})=F(c_{\mathrm{glue}})=\{*\},
\qquad
F(a_{\mathrm{glue}})=\varnothing,
\]
and let every restriction map be the identity on $\{*\}$. The compatibility check at the meet $c_{\mathrm{glue}}$ succeeds: $*=*$. Hence the cover supports a compatible family, but there is no global section at $a_{\mathrm{glue}}$. Thus $M,p\Vdash \Null^{\mathrm{glue}}_s(r_{\mathrm{glue}})$.

Finally, stipulate
\[
M,p\Vdash \Adm_s(r_{\mathrm{loc}})\land \Adm_s(r_{\mathrm{cmp}})\land \Adm_s(r_{\mathrm{glue}})
\]
and identify the three reference classes with $a_{\mathrm{loc}},a_{\mathrm{cmp}},a_{\mathrm{glue}}$ respectively. Because $0$ is the meet of cross-component objects and every restriction to $0$ is constant, the stability and matching conditions referenced above remain valid, and the constructed presheaves are defined on an honest meet-semilattice site.
\end{proof}

\subsection{Forcing necessity of the enrichment}

The three null modes show that information-state forcing distinguishes situations invisible to pointwise semantics. The following lemma isolates the automorphism invariance that drives the separation argument.

\begin{lemma}[State-level automorphism invariance]\label{lem:state-automorphism-invariance}
Let $M$ be a model, $p=(\Gamma_p,R_p)$ an information state, and $\alpha$ an automorphism of the $\Sigma_1$-reduct of $M$ that preserves $R_p$ setwise (i.e., $\alpha\circ\rho\in R_p$ for every $\rho\in R_p$). Then for every lower-layer formula $\psi(x)$ and every reference $r$,
\[
M,p\Vdash\psi(r)
\quad\Longleftrightarrow\quad
M,p\Vdash\psi(\alpha(r)).
\]
\end{lemma}

\begin{proof}
By \Cref{def:state-forcing},
\[
M,p\Vdash\psi(r)
\;\Longleftrightarrow\;
\forall\rho\in R_p,\;M,\rho\models\psi(r).
\]
Since $\alpha$ preserves $R_p$ setwise and is an automorphism of the $\Sigma_1$-reduct, first-order truth invariance under automorphisms gives
\[
M,\rho\models\psi(r)
\;\Longleftrightarrow\;
M,\alpha\circ\rho\models\psi(\alpha(r))
\]
for each $\rho\in R_p$. As $\rho\mapsto\alpha\circ\rho$ is a bijection on $R_p$,
\[
\forall\rho\in R_p,\;M,\rho\models\psi(r)
\;\Longleftrightarrow\;
\forall\rho\in R_p,\;M,\rho\models\psi(\alpha(r)).
\qedhere
\]
\end{proof}

The following result makes the irreducibility precise: the gluing predicates introduced at the local-object layer cannot be reduced to the lower layers.

\begin{theorem}[Pointwise irreducibility of gluing predicates]\label{thm:pointwise-irreducibility}
There exist a model $M$, a state $p=(\Gamma_p,R_p)$, and admitted references
\[
r_1,\quad r_2,\quad r_3,\quad r_4:\mathsf{Ref}_s
\]
of the same sort such that for every $\rho\in R_p$ and every lower-layer $\mathrm{Form}_1$-formula $\varphi(x)$ with one free variable $x$ of sort $\mathsf{Ref}_s$,
\[
M,\rho\models\varphi(r_1)
\;\Longleftrightarrow\;
M,\rho\models\varphi(r_2)
\;\Longleftrightarrow\;
M,\rho\models\varphi(r_3)
\;\Longleftrightarrow\;
M,\rho\models\varphi(r_4),
\]
and moreover:
\begin{enumerate}[label=(\roman*),nosep]
\item $M,p\Vdash \CompSec_s(r_1)\land \Sec_s(r_1)$;
\item $M,p\Vdash \CompSec_s(r_2)\land \neg\Sec_s(r_2)$, i.e., $M,p\Vdash\Null^{\mathrm{glue}}_s(r_2)$;
\item $M,p\Vdash \LocSec_s(r_3)\land \neg\CompSec_s(r_3)$;
\item $M,p\Vdash \neg\LocSec_s(r_4)$.
\end{enumerate}
The four references are lower-layer indistinguishable and realize the four possible positions in the implication chain
\[
\Sec_s(r)\Rightarrow \CompSec_s(r)\Rightarrow \LocSec_s(r).
\]
In particular, the indistinguishable pairs $(r_1,r_4)$, $(r_1,r_3)$, and $(r_1,r_2)$ witness that $\LocSec_s$, $\CompSec_s$, and $\Sec_s$ are not definable in the information-state forcing language without the local-object enrichment; the pair $(r_1,r_2)$ also witnesses that $\Null^{\mathrm{glue}}_s$ is not lower-definable.
\end{theorem}

\begin{proof}
Take the context
\[
\Gamma_p=
\bigl(
y_1:\mathsf{Ref}_s,\,
y_2:\mathsf{Ref}_s,\,
y_3:\mathsf{Ref}_s,\,
y_4:\mathsf{Ref}_s
\bigr),
\]
set $r_k:=y_k$ for $k=1,2,3,4$, and let $p=(\Gamma_p,R_p)$ with $R_p=\{\rho_0\}$, where
\[
\rho_0(y_1)=0,\quad
\rho_0(y_2)=1,\quad
\rho_0(y_3)=2,\quad
\rho_0(y_4)=3.
\]

\medskip
\noindent\textit{Lower-layer indistinguishability.}
Let $P\subseteq D_{\mathsf{Ref}_s}^M$ be a set containing the interpretations of all $\Sigma_1$-constants of sort $\mathsf{Ref}_s$ together with the images of every $\Sigma_1$-function symbol whose codomain is $\mathsf{Ref}_s$ when the arguments range over the interpretations of $\Sigma_1$-terms that do not mention $y_1,\dots,y_4$. Enlarge the reference domain to
\[
D_{\mathsf{Ref}_s}^M=P\sqcup\{0,1,2,3\},
\]
where the four highlighted elements are disjoint from $P$. Interpret every $\Sigma_1$-relation symbol whose type involves $\mathsf{Ref}_s$ so that it ignores $\{0,1,2,3\}$ (for example by interpreting each such relation as empty on tuples containing any of these elements), and interpret every $\Sigma_1$-function symbol with a $\mathsf{Ref}_s$-output so that whenever one of the inputs belongs to $\{0,1,2,3\}$, the value lies in $P$ and depends only on the inputs from $P$. In particular, every named parameter in $\Sigma_1$ is interpreted inside $P$ and therefore fixed pointwise by any permutation of $\{0,1,2,3\}$.

With this setup, the subgroup $\langle (01)(23),(02)(13)\rangle\cong(\mathbb Z/2\mathbb Z)^2$ acting on $\{0,1,2,3\}$ and fixing $P$ pointwise extends to automorphisms of the $\Sigma_1$-reduct of $M$. This group acts transitively on $\{0,1,2,3\}$, so by invariance of first-order truth under automorphisms, every lower-layer unary formula $\varphi(x)$ takes the same value at $0,1,2,3$, establishing the lower-layer indistinguishability displayed above (since $R_p=\{\rho_0\}$). Named parameters, constants, and function values remain fixed because they lie in $P$.

\medskip
\noindent\textit{Local reference frames.}
Give $r_1,r_2,r_3,r_4$ pairwise disjoint rooted components in the local reference frame and adjoin a single bottom element $0$ that lies below every object; the meet of any two objects from distinct components is $0$. Write $a_k=[r_k]_p$. For each component we let $F(0)=\{*\}$ and every restriction map to $0$ send its source element to $*$. The pretopology is generated by identity covers, the component-specific covers described below, and the trivial cover of $0$, so the stability axiom holds automatically across components.

\smallskip
\noindent\textit{Data for $r_1$ (global section exists).}
Let $a_1$ carry only the trivial cover and define $F(a_1)=\{*\}$. Then $*$ is a global section, giving \textup{(i)}.

\smallskip
\noindent\textit{Data for $r_2$ (compatible local section, no global section).}
In the semilattice component for $a_2$, set
\[
c_2\sqsubseteq u_2\sqsubseteq a_2,
\qquad
c_2\sqsubseteq v_2\sqsubseteq a_2,
\qquad
c_2=u_2\wedge v_2.
\]
Let $\{u_2\to a_2,\,v_2\to a_2\}\in\Cov_{p,s}(a_2)$, and define
\[
F(u_2)=F(v_2)=F(c_2)=\{*\},\qquad F(a_2)=\varnothing,
\]
with all restriction maps being the identity on $\{*\}$. The unique family $(\sigma_{u_2}=*,\sigma_{v_2}=*)$ is compatible on the meet $c_2=u_2\wedge v_2$. Hence $M,p\Vdash\CompSec_s(r_2)$. But $F(a_2)=\varnothing$, so $M,p\nVdash\Sec_s(r_2)$, giving \textup{(ii)}.

\smallskip
\noindent\textit{Data for $r_3$ (local sections exist, incompatible).}
In the component for $a_3$, set
\[
c_3\sqsubseteq u_3\sqsubseteq a_3,
\qquad
c_3\sqsubseteq v_3\sqsubseteq a_3,
\qquad
c_3=u_3\wedge v_3.
\]
Let $\{u_3\to a_3,\,v_3\to a_3\}\in\Cov_{p,s}(a_3)$, and define
\[
F(u_3)=\{\sigma_u\},\quad
F(v_3)=\{\sigma_v\},\quad
F(c_3)=\{0,1\},\quad
F(a_3)=\varnothing,
\]
with restrictions $\res_{c_3}^{u_3}(\sigma_u)=0$ and $\res_{c_3}^{v_3}(\sigma_v)=1$. Local sections exist, but the compatibility condition at the meet $c_3=u_3\wedge v_3$ fails: $\res_{c_3}^{u_3}(\sigma_u)=0\neq 1=\res_{c_3}^{v_3}(\sigma_v)$. Hence $M,p\Vdash\LocSec_s(r_3)\land\neg\CompSec_s(r_3)$, giving \textup{(iii)}.

\smallskip
\noindent\textit{Data for $r_4$ (no local section exists).}
Let $a_4$ carry only the trivial cover and define
\[
F(a_4)=\varnothing.
\]
Then the only available cover is $\{\mathrm{id}_{a_4}:a_4\to a_4\}$, and its local section set is empty. Hence $M,p\nVdash \LocSec_s(r_4)$, establishing \textup{(iv)}. This fourth reference is included both to realize the remaining position in the implication chain and to ensure that the automorphism group acts transitively on all four values.

\smallskip
\noindent\textit{Separation.}
By lower-layer indistinguishability, every $\mathrm{Form}_1$-formula assigns the same truth value to $r_1,r_2,r_3,r_4$ at $p$. The pair $(r_1,r_4)$ witnesses that $\LocSec_s$ is not lower-definable: $M,p\Vdash\LocSec_s(r_1)$ but $M,p\nVdash\LocSec_s(r_4)$. The pair $(r_1,r_3)$ witnesses that $\CompSec_s$ is not lower-definable: $M,p\Vdash\CompSec_s(r_1)$ but $M,p\nVdash\CompSec_s(r_3)$. The pair $(r_1,r_2)$ witnesses that $\Sec_s$ is not lower-definable: both satisfy $\CompSec_s$, yet $M,p\Vdash\Sec_s(r_1)$ and $M,p\nVdash\Sec_s(r_2)$. The same pair also witnesses that $\Null^{\mathrm{glue}}_s$ is not lower-definable: $M,p\Vdash\Null^{\mathrm{glue}}_s(r_2)$ while $M,p\nVdash\Null^{\mathrm{glue}}_s(r_1)$. Hence none of these predicates is expressible by any formula of the lower information-state forcing language.
Since the meet of any two objects from distinct components is $0$ and all restrictions to $0$ are constant, the resulting local reference frame is a finite meet-semilattice satisfying the stability axiom of \Cref{def:local-reference-frame}.
\end{proof}

\subsection{Conditional data for the gerbe and homological layers}

Sections \ref{sec:gerbe-obstruction} and \ref{sec:homological-visibility} work relative to explicit data, so we package the standing hypotheses once and for all.

\begin{definition}[Conditional datum]\label{def:conditional-datum}
Let $p$ be an information state, let $r:\mathsf{Ref}_s$ be admitted at $p$, and write $a=[r]_p$. A \emph{conditional datum} for $(p,r)$ is a tuple
\[
\mathbf D=(r,p,\mathfrak P_r,\Band,v,\mathcal U,A,\omega)
\]
consisting of
\begin{enumerate}[label=(\roman*),nosep]
\item a gluing-sensitive realization prestack $\mathfrak P_r\to\mathcal C_{p,s}/a$ with band $\Band$ in the sense of \Cref{def:gluing-sensitive};
\item a visible value class $v\in \Vis_{p,s}(r)\cong \pi_0(\mathfrak L_r)(a)$;
\item a gerbe-adapted cover $\mathcal U=\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ for $v$;
\item a finite abelian trivialization of the band on $\mathcal U$, i.e. an identification $\Band|_{\mathcal U}\cong \underline{A}$ with $A$ a finite abelian group;
\item the induced cover-relative obstruction class $\omega=[\omega_v(\mathcal U)]\in H^2(N(\mathcal U),A)$.
\end{enumerate}
\end{definition}

\begin{remark}[Scope of the conditional results]\label{rem:conditional-scope}
Throughout Sections \ref{sec:gerbe-obstruction} and \ref{sec:homological-visibility} we fix a conditional datum $\mathbf D$ whenever the statements mention the gerbe or homological layers. Every theorem in those sections should therefore be read as: given data $\mathbf D$ as in \Cref{def:conditional-datum}, the displayed conclusions follow. When such data are unavailable, the statements apply only after supplying an external choice of $\mathbf D$.
\end{remark}
