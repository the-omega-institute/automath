\section{Local Objects, Visible Branches, and Gluing Failure}\label{sec:null-decomposition}

We next distinguish admissible reference from objecthood. An admitted reference need not yet determine a global object, but it may still support local witnesses and compatible local families. This local-to-global gap is the source of the structural phenomena studied in the present section.

The refinement hypothesis used below is part of the local-object layer, not a
metatheoretic convention; it is isolated in
\Cref{def:refinement-stable-local-object-layer}.

\subsection{References and local objects}

Fix a family of referenceable sorts
\[
S_{\mathrm{ref}}\subseteq S.
\]
For each $s\in S_{\mathrm{ref}}$, the language contains a reference sort $\mathsf{Ref}_s$ and an admissibility predicate $\Adm_s(\cdot)$.

\begin{definition}[Reference classes]\label{def:reference-classes}
Let $p$ be a state and let $r,r'$ be reference terms of sort $\mathsf{Ref}_s$. Define
\[
r\equiv_p r'
\quad:\Longleftrightarrow\quad
M,p\Vdash (r=r').
\]
The set of admitted reference classes at $p$ is
\[
\mathcal{R}_s(p):=\{[r]_p\mid M,p\Vdash \Adm_s(r)\}.
\]
\end{definition}

\begin{definition}[Local reference frame]\label{def:local-reference-frame}
For a state $p$ and a referenceable sort $s$, a \emph{local reference frame} is a triple
\[
\mathcal{B}_{p,s}=\bigl(\mathcal{R}_s(p),\sqsubseteq_{p,s},\Cov_{p,s}\bigr),
\]
where $\sqsubseteq_{p,s}$ is a preorder on $\mathcal{R}_s(p)$ and $\Cov_{p,s}$ assigns to each $a\in\mathcal{R}_s(p)$ a collection of nonempty covering families
\[
\{a_i\to a\}_{i\in I},
\qquad
a_i\sqsubseteq_{p,s} a.
\]
These covering families are required to satisfy:
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{identity:} $\{a\to a\}\in\Cov_{p,s}(a)$ for every $a$;
\item \emph{pullback stability:} if $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and $b\sqsubseteq_{p,s} a$, then there exists a covering family
\[
\{b_{ij}\to b\}_{i\in I,\;j\in J_i}\in\Cov_{p,s}(b)
\]
such that $b_{ij}\sqsubseteq_{p,s} a_i$ for all $i,j$;
\item \emph{transitivity:} if $\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)$ and, for each $i\in I$,
\[
\{b_{ij}\to a_i\}_{j\in J_i}\in\Cov_{p,s}(a_i),
\]
then
\[
\{b_{ij}\to a\}_{i\in I,\;j\in J_i}\in\Cov_{p,s}(a).
\]
\end{enumerate}
\end{definition}

\begin{proposition}[The cover relation defines a Grothendieck pretopology]\label{prop:local-reference-pretopology}
Let $\mathcal{B}_{p,s}$ be the preorder category whose objects are the admitted reference classes in $\mathcal{R}_s(p)$ and whose morphisms are the refinement arrows
\[
b\to a
\quad:\Longleftrightarrow\quad
b\sqsubseteq_{p,s} a.
\]
Then the covering families $\Cov_{p,s}$ form a Grothendieck pretopology on $\mathcal{B}_{p,s}$, in the nonempty-cover convention fixed in \Cref{def:local-reference-frame}.
\end{proposition}

\begin{proof}
The identity axiom in \Cref{def:local-reference-frame} gives the covering family $\{a\to a\}$ for every object $a$. If $\{a_i\to a\}_{i\in I}$ is a cover and $b\to a$ is any morphism, the pullback-stability axiom provides a cover
\[
\{b_{ij}\to b\}_{i\in I,\;j\in J_i}
\]
such that each arrow factors through $a_i\to a$. Finally, if $\{a_i\to a\}_{i\in I}$ covers $a$ and each $\{b_{ij}\to a_i\}_{j\in J_i}$ covers $a_i$, the transitivity axiom gives the composite cover
\[
\{b_{ij}\to a\}_{i\in I,\;j\in J_i}.
\]
These are exactly the identity, pullback, and transitivity axioms for a Grothendieck pretopology on a preorder category \cite[Chap.~III, \S2]{MacLaneMoerdijk1994}, \cite[Part~B2.1]{Johnstone2002}. The nonempty-cover convention is harmless for the semantic applications, where covers represent available local presentations; it also avoids treating the empty family as a vacuous local witness.
\end{proof}

Accordingly, we use the same symbol $\mathcal{B}_{p,s}$ for the resulting preorder category in the presheaf and sheaf constructions below.

\begin{definition}[Local object functor]\label{def:local-object-functor}
For a state $p$ and a referenceable sort $s$, a \emph{local object functor} is
a contravariant set-valued functor
\[
F_{p,s}:\mathcal B_{p,s}^{\mathrm{op}}\longrightarrow \mathbf{Set}.
\]
Thus each admitted reference class $a\in\mathcal R_s(p)$ has a set
$F_{p,s}(a)$ of local objects over $a$, and every refinement arrow
$b\to a$, equivalently $b\sqsubseteq_{p,s}a$, has a restriction map
\[
\res_b^a:F_{p,s}(a)\longrightarrow F_{p,s}(b).
\]
The restriction maps satisfy
\[
\res_a^a=\mathrm{id}_{F_{p,s}(a)}
\]
and, whenever $c\sqsubseteq_{p,s} b\sqsubseteq_{p,s}a$,
\[
\res_c^a=\res_c^b\circ \res_b^a.
\]
When the state and sort are fixed we write $F$ for $F_{p,s}$. The notation
$F_{p,s}|_{\mathcal C_a}$ denotes the restriction of this presheaf to the slice
site
\[
\mathcal C_a:=\mathcal C_{p,s}/a.
\]
\end{definition}

\begin{definition}[Refinement-stable local-object layer]\label{def:refinement-stable-local-object-layer}
Let $q\le p$. The local-object data for a referenceable sort $s$ are
\emph{refinement-stable along $q\le p$} if the following data and properties
are specified.
\begin{enumerate}[label=(\roman*),nosep]
\item There is a functor
\[
\rho_{q,p}^s:\mathcal B_{q,s}\longrightarrow \mathcal B_{p,s}
\]
which sends the refined reference class of every admitted reference to its
coarser class:
\[
\rho_{q,p}^s([r]_q)=[r]_p.
\]
It preserves refinement arrows and sends covering families in
$\Cov_{q,s}$ to covering families in $\Cov_{p,s}$.
\item There is a natural descent transformation
\[
\delta_{q,p}:F_{q,s}\Longrightarrow F_{p,s}\circ \rho_{q,p}^s
\]
of presheaves on $\mathcal B_{q,s}$. Thus, whenever
$c\sqsubseteq_{q,s} b$,
\[
\delta_c\bigl(\res_c^b(\sigma)\bigr)
=
\res_{\rho_{q,p}^s(c)}^{\rho_{q,p}^s(b)}\bigl(\delta_b(\sigma)\bigr)
\qquad(\sigma\in F_{q,s}(b)).
\]
Moreover, this descent preserves compatible families: the image of any
compatible local family over a cover in $\Cov_{q,s}$ is a compatible local
family over the descended cover in $\Cov_{p,s}$.
\item The induced descent map on visible value classes has no singleton
splitting: if $r$ is admitted at $p$ and
\[
\Vis_{p,s}(r)=\{v\},
\]
then the descent map
\[
\delta_{q,p}^{\Vis}:\Vis_{q,s}(r)\longrightarrow \Vis_{p,s}(r)
\]
constructed in \Cref{lem:refinement-descent-visible-classes} is a bijection.
In that case we denote the unique refined class over $v$ again by $v$.
\end{enumerate}
Whenever a result in this section compares states $q\le p$, the local-object
layer is assumed to be refinement-stable along that refinement.
\end{definition}

\begin{definition}[Three existence levels]\label{def:existence-levels}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and put $a=[r]_p$. In the
preorder-pretopology of \Cref{prop:local-reference-pretopology}, with no
pairwise-pullback hypothesis imposed, the three sectionability predicates are
read as follows.
\begin{enumerate}[label=(\roman*),nosep]
\item $M,p\Vdash \LocSec_s(r)$ means that some cover
$\{a_i\to a\}_{i\in I}$ in $\Cov_{p,s}(a)$ has
$F_{p,s}(a_i)\neq\varnothing$ for all indices $i$.
\item $M,p\Vdash \CompSec_s(r)$ means that one can choose such a cover together
with $\sigma_i\in F_{p,s}(a_i)$ so that, whenever
$c\sqsubseteq_{p,s}a_i$ and $c\sqsubseteq_{p,s}a_j$,
\[
\res_c^{a_i}(\sigma_i)=\res_c^{a_j}(\sigma_j).
\]
\item $M,p\Vdash \Sec_s(r)$ means $F_{p,s}(a)$ is nonempty.
\end{enumerate}
\end{definition}

\begin{remark}
The sibling paper states the same three-level interface using pairwise fibre
products on the reference site \cite[Definition~4.4]{HomologicalVisibilityGluingObstructions2026}.
The increment here is that compatibility is formulated for arbitrary preorder
sites by testing all common refinements; \Cref{lem:common-refinement-matching-family}
identifies this with the matching-family condition.
\end{remark}

\begin{lemma}[Common refinements are matching families]\label{lem:common-refinement-matching-family}
Let
\[
\mathcal U=\{a_i\to a\}_{i\in I}\in\Cov_{p,s}(a)
\]
and let $S_{\mathcal U}$ be the covering sieve generated by $\mathcal U$. For
sections $\sigma_i\in F_{p,s}(a_i)$, the following are equivalent.
\begin{enumerate}[label=(\roman*),nosep]
\item For every common refinement $c\sqsubseteq_{p,s}a_i,a_j$,
\[
\res_c^{a_i}(\sigma_i)=\res_c^{a_j}(\sigma_j).
\]
\item The family $(\sigma_i)_{i\in I}$ is a matching family for the cover
$\mathcal U$ in the pretopology of \Cref{prop:local-reference-pretopology}.
\item There is a unique matching family
\[
(\tau_b)_{(b\to a)\in S_{\mathcal U}}
\]
on the covering sieve $S_{\mathcal U}$ such that
$\tau_{a_i}=\sigma_i$ for every $i$.
\end{enumerate}
\end{lemma}

\begin{proof}
In a preorder site, an object $b\to a$ belongs to the sieve generated by
$\mathcal U$ exactly when $b\sqsubseteq_{p,s}a_i$ for at least one $i$. Assume
\textup{(i)}. For such a $b$, choose an index $i$ with
$b\sqsubseteq_{p,s}a_i$ and set
\[
\tau_b:=\res_b^{a_i}(\sigma_i).
\]
This is independent of the chosen $i$: if $b\sqsubseteq a_i,a_j$, then $b$ is a
common refinement of $a_i$ and $a_j$, so \textup{(i)} gives the same restricted
section. If $c\sqsubseteq b$ and both objects lie in the sieve, then functoriality
of restriction gives
\[
\res_c^b(\tau_b)=\tau_c.
\]
Thus $(\tau_b)$ is a matching family on the sieve, proving \textup{(iii)} and
hence \textup{(ii)}.

Conversely, any matching family on $S_{\mathcal U}$ restricts to sections
$\sigma_i$ on the generators $a_i$. If $c\sqsubseteq a_i,a_j$, then $c$ lies in
$S_{\mathcal U}$ and the matching condition gives
\[
\res_c^{a_i}(\sigma_i)=\tau_c=\res_c^{a_j}(\sigma_j),
\]
which is \textup{(i)}. The same calculation also proves uniqueness of the
sieve-level family, since every $\tau_b$ must equal
$\res_b^{a_i}(\sigma_i)$ after choosing any generator through which $b$ factors.
\end{proof}

\begin{proposition}[Global sections yield compatible local sections]\label{prop:global-implies-compatible-local}
In the common-refinement setting of \Cref{def:existence-levels}, if
$M,p\Vdash \Sec_s(r)$, then
\[
M,p\Vdash \CompSec_s(r)
\qquad\text{and}\qquad
M,p\Vdash \LocSec_s(r).
\]
\leanverified{paper_conservative_extension_global_implies_compatible_local_seeds}
\end{proposition}

\begin{remark}
The fibre-product version appears as \cite[Proposition~4.5]{HomologicalVisibilityGluingObstructions2026}.
The local increment is the same implication for preorder sites whose overlap
condition is expressed by common refinements.
\end{remark}

\begin{proof}
Let $a=[r]_p$ and take $\sigma\in F_{p,s}(a)$. The identity cover of $a$ is
available by \Cref{def:local-reference-frame}. Its one chosen section restricts
to itself along every common refinement, so it witnesses both predicates.
\end{proof}

\subsection{Sheafification and gluing failure}

By \Cref{prop:local-reference-pretopology}, for fixed $(p,s)$ the local reference frame of \Cref{def:local-reference-frame} determines a site
\[
\mathcal{C}_{p,s}:=(\mathcal{B}_{p,s},\Cov_{p,s}),
\]
and the local object functor of \Cref{def:local-object-functor} determines a presheaf
\[
F_{p,s}\in\mathbf{PSh}(\mathcal{B}_{p,s}).
\]
Write
\[
a_{p,s}:\mathbf{PSh}(\mathcal{B}_{p,s})\to\mathbf{Sh}(\mathcal{C}_{p,s})
\]
for the associated sheafification functor.

\begin{theorem}[Common-refinement sheafification criterion]\label{thm:sheafification-characterization}
For the common-refinement preorder site of this paper, an admitted reference
$r:\mathsf{Ref}_s$ at $p$ is compatibly locally sectionable exactly when the
associated sheaf $a_{p,s}F_{p,s}$ has a section over $a=[r]_p$. This version does
not assume chosen pairwise fibre products or meets.
\end{theorem}

\begin{remark}
The sibling result is the fibre-product formulation
\cite[Theorem~4.6]{HomologicalVisibilityGluingObstructions2026}. The concrete
increment here is that the equivalence is proved without choosing pairwise
meets, using \Cref{lem:common-refinement-matching-family} instead.
\end{remark}

\begin{proof}
By the standard site-theoretic construction, an element over $a$ in the
associated sheaf is represented by a matching family on a covering sieve, or
equivalently on a cover,
\cite[Chap.~III, \S2]{MacLaneMoerdijk1994},
\cite[Part~B2.2]{Johnstone2002}. For our preorder pretopology,
\Cref{lem:common-refinement-matching-family} translates those representatives
precisely into the families counted by \Cref{def:existence-levels}(ii). Thus
nonemptiness of the sheafified value at $a$ is equivalent to compatible local
sectionability.
\end{proof}

\begin{remark}[Role of the sheafification step]\label{rem:sheafification-bookkeeping}
\Cref{thm:sheafification-characterization} is a bookkeeping statement about
which common-refinement compatible data survive sheafification. It does not
construct a realization prestack, a gerbe, or an obstruction class. Those require
the separate realization input introduced below.
\end{remark}

\begin{corollary}[Sheaf condition at a common-refinement reference]\label{cor:sheaf-local-to-global}
In the common-refinement setting above, if $F_{p,s}$ is already a sheaf at
$a=[r]_p$, then compatible local sectionability and global sectionability agree
for $r$ at $p$.
\end{corollary}

\begin{remark}
The corresponding fibre-product corollary is
\cite[Corollary~4.7]{HomologicalVisibilityGluingObstructions2026}. Here the
same sheaf condition is applied after replacing overlap meets by the
common-refinement criterion of \Cref{def:existence-levels}.
\end{remark}

\begin{proof}
At $a$, the unit into the associated sheaf is an isomorphism under the stated
hypothesis. Combine this with \Cref{thm:sheafification-characterization}.
\end{proof}

\begin{corollary}[Common-refinement sheafification removes gluing failure]\label{cor:sheafification-removes-glue}
Let $\Sec_s^{\sharp}(r)$ be global sectionability after replacing the
common-refinement local-object functor by its associated sheaf. Then
$\CompSec_s(r)$ and $\Sec_s^{\sharp}(r)$ have the same forcing truth at $p$, so
the sheafified layer has no admitted references of the gluing-failure kind.
\end{corollary}

\begin{remark}
The fibre-product-site statement is
\cite[Corollary~4.8]{HomologicalVisibilityGluingObstructions2026}. The
present version records the same conclusion for the wider preorder-site
compatibility notion used in this paper.
\end{remark}

\begin{proof}
For the associated sheaf, global sectionability at $a$ is exactly nonemptiness
of $(a_{p,s}F_{p,s})(a)$. Apply
\Cref{thm:sheafification-characterization}.
\end{proof}

sheafification results above determine visible set-valued classes, but they do
not by themselves produce groupoids of realizations, off-diagonal isomorphism
sheaves, component gerbes, or obstruction classes.

\subsection{Visible value classes and typed readouts}

\begin{definition}[Visible value classes]\label{def:visible-value-classes}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, and put $a=[r]_p$. A visible
presentation is a cover of $a$ together with local sections that satisfy the
common-refinement compatibility clause of \Cref{def:existence-levels}(ii). The
relation of presenting the same value is generated by pulling two such
presentations to a shared refinement cover and comparing the pulled-back
sections there. The quotient set is denoted $\Vis_{p,s}(r)$.
\end{definition}

\begin{remark}
The sibling definition is the fibre-product version
\cite[Definition~4.40]{HomologicalVisibilityGluingObstructions2026}. The local
increment is that value-equivalence is taken over the common-refinement
presentation category attached to the preorder site.
\end{remark}

\begin{lemma}[Refinement descent for visible classes]\label{lem:refinement-descent-visible-classes}
Assume the local-object layer is refinement-stable along $q\le p$ in the sense
of \Cref{def:refinement-stable-local-object-layer}. Then every visible
presentation of an admitted reference $r$ at $q$ descends functorially to a
visible presentation of $r$ at $p$. Passing to value-equivalence classes gives
a well-defined map
\[
\delta_{q,p}^{\Vis}:\Vis_{q,s}(r)\longrightarrow \Vis_{p,s}(r).
\]
If $\Vis_{p,s}(r)=\{v\}$, then this map is a bijection and the unique refined
visible class over $v$ is denoted again by $v$.
\end{lemma}

\begin{proof}
Let the refined visible presentation be
\[
\bigl(\{b_i\to b\}_{i\in I},(\sigma_i)_{i\in I}\bigr),
\qquad b=[r]_q.
\]
By \Cref{def:refinement-stable-local-object-layer}(i), applying
$\rho_{q,p}^s$ gives a cover
\[
\{\rho_{q,p}^s(b_i)\to \rho_{q,p}^s(b)\}_{i\in I}
=
\{\rho_{q,p}^s(b_i)\to [r]_p\}_{i\in I}
\]
in $\Cov_{p,s}([r]_p)$. Define descended local sections by
\[
\bar\sigma_i:=\delta_{b_i}(\sigma_i)\in F_{p,s}(\rho_{q,p}^s(b_i)).
\]
The compatibility-preservation clause in
\Cref{def:refinement-stable-local-object-layer}(ii) says exactly that this
descended family is compatible over the descended cover. Hence it is a visible
presentation at $p$.

If two refined presentations become equal after passage to a common refinement,
then applying the functor $\rho_{q,p}^s$ and the natural maps $\delta_{q,p}$
preserves that equality. Hence equivalent refined presentations have equivalent
descents, so the displayed map on visible classes is well-defined. The final
singleton assertion is exactly the no-splitting clause in
\Cref{def:refinement-stable-local-object-layer}(iii).
\end{proof}

\begin{theorem}[Visible value classes as global connected components]\label{thm:visible-value-components}
\leanverified{paper_conservative_extension_visible_value_components_focused}
Assume that the stack-banded realization prestack input of
\Cref{def:realization-prestack} has been fixed for $r$. For the
common-refinement visible classes above, there are canonical identifications
\[
\Vis_{p,s}(r)\cong \bigl(a_{p,s}F_{p,s}\bigr)(a)\cong \pi_0(\mathfrak L_r)(a).
\]
\end{theorem}

\begin{remark}
The sibling component statement is
\cite[Proposition~4.41]{HomologicalVisibilityGluingObstructions2026}. The
increment here is twofold: the first identification uses the common-refinement
quotient of \Cref{def:visible-value-classes}, and the second is asserted only
after the stack-banded realization input required later in this paper has been
chosen.
\end{remark}

\begin{proof}
The sheafification part follows by combining the construction of the associated
sheaf with \Cref{lem:common-refinement-matching-family}: representatives over
$a$ are precisely the common-refinement compatible presentations, modulo the
same refinement relation used to form $\Vis_{p,s}(r)$. Hence
\[
\Vis_{p,s}(r)\cong \bigl(a_{p,s}F_{p,s}\bigr)(a).
\]
The second identification is \Cref{prop:stackification-components} evaluated at $a$.
\end{proof}

\begin{remark}[Role of the component identification]\label{rem:component-identification-bookkeeping}
The first isomorphism in \Cref{thm:visible-value-components} is again a sheafification statement about the set-valued functor $F_{p,s}$. The second uses the chosen stack-banded realization prestack through \Cref{prop:stackification-components}. Thus the theorem identifies visible classes with components only after a realization prestack has already been supplied or constructed by the separated split-lift criterion. It is not an existence theorem for such a prestack. The stack-level banding used later for component gerbes is the separate datum in \Cref{def:realization-prestack}, item~\textup{(ii)}, or the descended datum of \Cref{prop:stackification-band-descends}; it is not supplied by the component identification alone.
\end{remark}

\begin{definition}[Typed readout]\label{def:typed-readout}
Let $r:\mathsf{Ref}_s$ be a reference term at a state $p$. The typed readout
\[
\Read_{p,s}(r)
\]
takes values in the disjoint union of the visible value classes and a formal
symbol $\Null$. It is defined as follows:
\[
\Read_{p,s}(r)=
\begin{cases}
v, & M,p\Vdash \Adm_s(r)\ \text{and}\ \Vis_{p,s}(r)=\{v\},\\
\Null, & \text{otherwise.}
\end{cases}
\]
Thus a non-null readout is returned only when the reference is admitted and the
common-refinement visible value quotient has a unique class.
\end{definition}

\begin{proposition}[Address before value]\label{prop:address-before-value}
If
\[
\Read_{p,s}(r)=v
\qquad\text{and}\qquad
v\neq\Null,
\]
then $r$ is admitted at $p$, the address
\[
a=[r]_p
\]
is defined in $\mathcal R_s(p)$, and
\[
\Vis_{p,s}(r)=\{v\}.
\]
In particular,
\[
M,p\Vdash \CompSec_s(r).
\]
\leanverified{paper_information_state_address_before_value_seeds}
\end{proposition}

\begin{proof}
The first three assertions are immediate from \Cref{def:typed-readout}: the
only clause that can return a value different from the formal symbol $\Null$ is
the clause in which $M,p\Vdash\Adm_s(r)$ and $\Vis_{p,s}(r)$ is the singleton
$\{v\}$. Since $\Vis_{p,s}(r)$ is nonempty, there is a compatible local
presentation of $r$ by \Cref{def:visible-value-classes}; equivalently
$M,p\Vdash\CompSec_s(r)$ by \Cref{def:existence-levels}.
\end{proof}

\begin{corollary}[Non-null readout criterion]\label{cor:nonnull-readout-criterion}
\[
\Read_{p,s}(r)\neq\Null
\quad\Longleftrightarrow\quad
M,p\Vdash \Adm_s(r)\ \text{and}\ |\Vis_{p,s}(r)|=1.
\]
\leanverified{paper_information_state_nonnull_readout_criterion_seeds}
\end{corollary}

\begin{proof}
If the readout is non-null, \Cref{prop:address-before-value} gives admission
and shows that $\Vis_{p,s}(r)$ is a singleton. Conversely, if $r$ is admitted
and $\Vis_{p,s}(r)=\{v\}$, the first clause of \Cref{def:typed-readout} gives
$\Read_{p,s}(r)=v\neq\Null$.
\end{proof}

\begin{proposition}[Refinement stability of defined readouts]\label{prop:typed-readout-persistence}
Let $q\le p$, and suppose
\[
\Read_{p,s}(r)=v
\qquad\text{with}\qquad
v\neq \Null.
\]
Then
\[
\Read_{q,s}(r)=v.
\]
\leanverified{paper_information_state_typed_readout_persistence_seeds}
\end{proposition}

\begin{proof}
Since $v\neq\Null$, the readout at $p$ is represented by a unique visible value
class:
\[
\Vis_{p,s}(r)=\{v\}.
\]
Also $M,p\Vdash\Adm_s(r)$ by \Cref{prop:address-before-value}, and admission
persists along $q\le p$, so the refined visible set $\Vis_{q,s}(r)$ is defined.
By \Cref{lem:refinement-descent-visible-classes}, the refinement-stable
local-object layer gives a descent map
\[
\delta_{q,p}^{\Vis}:\Vis_{q,s}(r)\to\Vis_{p,s}(r)
\]
which is bijective over this singleton. Hence $\Vis_{q,s}(r)=\{v\}$ under the
canonical identification of \Cref{def:refinement-stable-local-object-layer}, and
\Cref{def:typed-readout} gives $\Read_{q,s}(r)=v$.
\end{proof}

\subsection{Structural absence}

\begin{definition}[Undefined, total, and null]\label{def:undef-total-null}
For a reference term $r:\mathsf{Ref}_s$ at a state $p$, define the three
coarse status predicates by the following abbreviations:
\[
\Undef_s(r):=\neg\Adm_s(r),
\qquad
\Tot_s(r):=\Adm_s(r)\land \Sec_s(r),
\qquad
\Null_s(r):=\Adm_s(r)\land \neg\Sec_s(r).
\]
Thus undefined means that no admitted address is present at $p$, total means
that the admitted address already has a global local object, and null means
that the reference is admitted but has no global section.
\end{definition}

\begin{definition}[The three null modes]\label{def:null-failure-modes}
For an admitted-reference term $r:\mathsf{Ref}_s$ at $p$, the null predicate is
decomposed into three mutually exclusive modes:
\[
\Null^{\mathrm{loc}}_s(r)
:=
\Adm_s(r)\land \neg\LocSec_s(r),
\]
\[
\Null^{\mathrm{cmp}}_s(r)
:=
\Adm_s(r)\land \LocSec_s(r)\land \neg\CompSec_s(r),
\]
and
\[
\Null^{\mathrm{glue}}_s(r)
:=
\Adm_s(r)\land \CompSec_s(r)\land \neg\Sec_s(r).
\]
The first mode says that no local cover carries witnesses, the second says that
local witnesses exist but cannot be made compatible, and the third says that
compatible local witnesses exist but do not glue to a global section.
\end{definition}

\begin{proposition}[The three modes lie inside null]\label{prop:null-modes-imply-null}
For every reference term $r:\mathsf{Ref}_s$ at every state $p$,
\[
M,p\Vdash \Null^{\mathrm{loc}}_s(r)
\quad\text{or}\quad
M,p\Vdash \Null^{\mathrm{cmp}}_s(r)
\quad\text{or}\quad
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\]
implies
\[
M,p\Vdash \Null_s(r).
\]
\end{proposition}

\begin{proof}
In the local mode, $r$ is admitted and $\LocSec_s(r)$ fails. If
$M,p\Vdash\Sec_s(r)$, then
\Cref{prop:global-implies-compatible-local} would give
$M,p\Vdash\LocSec_s(r)$, a contradiction. Hence $\Sec_s(r)$ fails and
\Cref{def:undef-total-null} gives $\Null_s(r)$.

In the compatibility mode, $r$ is admitted, $\LocSec_s(r)$ holds, and
$\CompSec_s(r)$ fails. If a global section existed, then
\Cref{prop:global-implies-compatible-local} would give
$M,p\Vdash\CompSec_s(r)$, again a contradiction. Thus $\Sec_s(r)$ fails and
$\Null_s(r)$ holds.

In the gluing mode, admission and $\neg\Sec_s(r)$ are part of the definition,
so $\Null_s(r)$ follows directly from \Cref{def:undef-total-null}.
\end{proof}

\begin{proposition}[Mutual exclusivity of the null modes]\label{prop:null-modes-exclusive}
No two distinct null modes can hold simultaneously for the same reference at the same state.
\end{proposition}

\begin{proof}
The mode $\Null^{\mathrm{loc}}_s(r)$ contains $\neg\LocSec_s(r)$, while the other two contain $\LocSec_s(r)$. The mode $\Null^{\mathrm{cmp}}_s(r)$ contains $\neg\CompSec_s(r)$, while $\Null^{\mathrm{glue}}_s(r)$ contains $\CompSec_s(r)$.
\end{proof}

\begin{definition}[Decomposition decidability]\label{def:decomposition-decidability}
A state $p$ has \emph{decomposition decidability} for a reference term
$r:\mathsf{Ref}_s$ if the two local-object predicates
\[
\LocSec_s(r)
\qquad\text{and}\qquad
\CompSec_s(r)
\]
are decided at $p$: for each
$P\in\{\LocSec_s(r),\CompSec_s(r)\}$, either $M,p\Vdash P$ or
$M,p\Vdash\neg P$. No realization-prestack or gerbe data are part of this
decidability hypothesis.
\end{definition}

\begin{proposition}[Null trichotomy]\label{prop:null-trichotomy}
See \cite[Proposition~4.15]{HomologicalVisibilityGluingObstructions2026}.
\end{proposition}

\begin{theorem}[Classification after refinement]\label{thm:null-classification-under-refinement}
Let $q\le p$ and suppose
\[
M,p\Vdash \Null_s(r).
\]
Assume the local-object layer is refinement-stable along $q\le p$ in the sense
of \Cref{def:refinement-stable-local-object-layer}. If $q$ has decomposition
decidability for $r$, then the refined state $q$ realizes a unique one of the
three null modes.
\end{theorem}

\begin{remark}
The sibling classification is \cite[Theorem~4.16]{HomologicalVisibilityGluingObstructions2026}.
The concrete increment here is the explicit refinement-stability interface
needed before transporting the local-object predicates from $p$ to $q$.
\end{remark}

\begin{proof}
The refinement-stability interface of
\Cref{def:refinement-stable-local-object-layer} supplies the refined
local-object frame, covers, and section predicates for the same reference at
$q$. By persistence of forcing, $M,q\Vdash \Null_s(r)$. The hypothesis gives
decomposition decidability at $q$, so \Cref{prop:null-trichotomy} applies at the
refined state $q$.
\end{proof}

\subsection{Forcing necessity of the enrichment}

The three null modes show that information-state forcing distinguishes situations invisible to pointwise semantics. The following result makes this precise: the gluing predicates introduced at the local-object layer cannot be reduced to the lower layers. The restriction on formulas is part of the claim: formulas are not allowed to use the local-object or absence predicates themselves, nor to name the distinguished reference constants except through the displayed free variable.

\begin{theorem}[Pointwise irreducibility of gluing predicates]\label{thm:pointwise-irreducibility}
\leanverified{paper_conservative_extension_pointwise_irreducibility_focused}
There exist a model $M$, a state $p=(\Gamma_p,R_p)$, and admitted references
\[
r_1,r_2,r_3,r_4:\mathsf{Ref}_s
\]
of the same sort such that:
\begin{enumerate}[label=(\roman*),nosep]
\item for every $\rho\in R_p$ and every $\mathrm{Form}_1$-formula $\varphi(x)$ with one free variable $x$, containing none of the local-object or absence predicates
\[
\Adm_s,\quad \LocSec_s,\quad \CompSec_s,\quad \Sec_s,\quad
\Null_s,\quad \Null^{\mathrm{loc}}_s,\quad
\Null^{\mathrm{cmp}}_s,\quad \Null^{\mathrm{glue}}_s,
\]
and containing no occurrence of the distinguished constants $r_1,r_2,r_3,r_4$ other than the substituted variable,
\[
M,\rho\models\varphi(r_1)
\quad\Longleftrightarrow\quad
M,\rho\models\varphi(r_2);
\]
\item under the same hypotheses on $\rho$ and $\varphi(x)$,
\[
M,\rho\models\varphi(r_3)
\quad\Longleftrightarrow\quad
M,\rho\models\varphi(r_4);
\]
\item $M,p\Vdash \CompSec_s(r_1)\land \Sec_s(r_1)$;
\item $M,p\Vdash \CompSec_s(r_2)\land \neg\Sec_s(r_2)$;
\item $M,p\Vdash \CompSec_s(r_3)$;
\item $M,p\Vdash \LocSec_s(r_4)\land \neg\CompSec_s(r_4)$.
\end{enumerate}
In particular, no formula of the information-state forcing language that is invariant under $\mathrm{Form}_1$-automorphisms, contains none of the local-object or absence predicates displayed above, and does not name the distinguished reference constants except through its free variable can define the predicates
\[
\CompSec_s,
\qquad
\Sec_s,
\qquad
\Null^{\mathrm{glue}}_s.
\]
\end{theorem}

\begin{proof}
Take a single state $p$ and one referenceable sort $s$. Introduce four reference terms
\[
r_1,r_2,r_3,r_4:\mathsf{Ref}_s
\]
with distinct reference classes
\[
[r_1]_p,\ [r_2]_p,\ [r_3]_p,\ [r_4]_p.
\]
Choose the $\mathrm{Form}_1$-reduct of $M$ so that there are automorphisms
\[
\alpha,\beta:M|_{\mathrm{Form}_1}\xrightarrow{\sim} M|_{\mathrm{Form}_1}
\]
with
\[
\alpha(r_1^M)=r_2^M,
\qquad
\alpha(r_2^M)=r_1^M,
\]
\[
\beta(r_3^M)=r_4^M,
\qquad
\beta(r_4^M)=r_3^M,
\]
and each of $\alpha,\beta$ preserves every realization $\rho\in R_p$. By invariance of first-order truth under automorphisms, every $\mathrm{Form}_1$-formula $\varphi(x)$ satisfying the restriction in~\textup{(i)} therefore obeys
\[
M,\rho\models\varphi(r_1)
\quad\Longleftrightarrow\quad
M,\rho\models\varphi(r_2)
\]
for every $\rho\in R_p$, which proves \textup{(i)}. The same argument with $\beta$ proves \textup{(ii)}.

Write $a_k=[r_k]_p$ for $1\le k\le 4$, and let the local reference frame be the disjoint union of four rooted components, one above each $a_k$. In each component choose a cover
\[
\{u_k\to a_k,v_k\to a_k\}
\]
with common refinement $c_k$. Define a single local object functor $F_{p,s}$ on the disjoint union by specifying it componentwise.

On the $a_1$-component, set
\[
F_{p,s}(u_1)=F_{p,s}(v_1)=F_{p,s}(c_1)=F_{p,s}(a_1)=\{*\},
\]
with all restrictions equal to the identity on $\{*\}$. The unique local family $(*,*)$ is compatible, and $*\in F_{p,s}(a_1)$ is a global section. Hence \textup{(iii)}.

On the $a_2$-component, set
\[
F_{p,s}(u_2)=F_{p,s}(v_2)=F_{p,s}(c_2)=\{*\},
\qquad
F_{p,s}(a_2)=\varnothing,
\]
with all restrictions to $c_2$ equal to the identity on $\{*\}$. The local family $(*,*)$ is compatible on the overlap, but $F_{p,s}(a_2)=\varnothing$. Hence \textup{(iv)}.

On the $a_3$-component, use the same definition as on the $a_1$-component. Then $r_3$ is compatibly locally sectionable, so \textup{(v)} holds.

On the $a_4$-component, set
\[
F_{p,s}(u_4)=\{\sigma_u\},\quad
F_{p,s}(v_4)=\{\sigma_v\},\quad
F_{p,s}(c_4)=\{0,1\},\quad
F_{p,s}(a_4)=\varnothing,
\]
with restriction maps
\[
\res_{c_4}^{u_4}(\sigma_u)=0,
\qquad
\res_{c_4}^{v_4}(\sigma_v)=1.
\]
Then local sections exist on the chosen cover, but the only local family is incompatible on the overlap. Hence \textup{(vi)}.

Finally, stipulate
\[
M,p\Vdash \Adm_s(r_1)\land \Adm_s(r_2)\land \Adm_s(r_3)\land \Adm_s(r_4)
\]
and identify the four reference classes with $a_1,a_2,a_3,a_4$ respectively.

Now let $\theta(x)$ be any lower-language information-state forcing formula satisfying the restrictions in the final sentence of the theorem. Because $\mathbb{L}_1$-forcing is flat over the realizations in $R_p$, the pointwise equivalences in \textup{(i)} and \textup{(ii)} imply
\[
M,p\Vdash \theta(r_1)\leftrightarrow\theta(r_2)
\qquad\text{and}\qquad
M,p\Vdash \theta(r_3)\leftrightarrow\theta(r_4).
\]
But \textup{(iii)} and \textup{(iv)} separate $\Sec_s$ and $\Null^{\mathrm{glue}}_s$, while \textup{(v)} and \textup{(vi)} separate $\CompSec_s$. Therefore no such $\theta$ can define any of the three predicates.
\end{proof}
