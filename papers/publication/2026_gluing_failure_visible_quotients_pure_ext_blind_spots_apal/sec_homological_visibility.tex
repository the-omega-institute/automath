\section{Homological Evaluation, Visible Quotients, and Pure-Ext Blindness}\label{sec:homological-visibility}

This section studies a branch obstruction after a concrete finite presentation has been chosen. To isolate the part of the obstruction that is detected at the level of homology, we make the homological evaluation map primary. The visible quotient is defined from this map and only afterwards recovered through class-admissible character channels. Independence of band trivialization is proved; refinement invariance is asserted only under explicit surjectivity hypotheses on the induced maps in $H_2$ (\Cref{prop:visible-quotient-invariance}).

\begin{definition}[Finite nerve presentation of the obstruction class]\label{def:finite-nerve-presentation}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write $a=[r]_p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item $r$ admits a gluing-sensitive realization prestack, so each visible branch $v\in \Vis_{p,s}(r)$ has a component gerbe $\mathfrak G_v$ in the sense of \Cref{def:compatible-realization-stack};
\item a visible branch $v\in \Vis_{p,s}(r)$ and a gerbe-adapted cover $\mathcal U=\{a_i\to a\}_{i\in I}$ for $v$ have been chosen;
\item the band of $\mathfrak G_v$ is trivialized on $\mathcal U$ by a finite abelian group $A$.
\end{enumerate}
These choices, together with the induced class $\omega=[\omega_v(\mathcal U)]$, constitute the conditional datum $\mathbf D=(r,p,\mathfrak P_r,\Band,v,\mathcal U,A,\omega)$ from \Cref{def:conditional-datum}.
A \emph{finite nerve presentation} of the branch obstruction class consists of:
\begin{enumerate}[label=(\roman*),nosep]
\item the chosen finite covering family
\[
\mathcal U=\{a_i\to a\}_{i\in I},
\]
whose \v{C}ech nerve $N(\mathcal U)$ is finite;
\item the chosen trivialization of the band on $\mathcal U$, so that the band may be viewed as a constant coefficient finite abelian group $A$ on $N(\mathcal U)$;
\item the induced obstruction class
\[
\omega\in H^2(N(\mathcal U),A),
\]
obtained by identifying \v{C}ech cochains on $\mathcal U$ with simplicial cochains on the finite nerve and writing
\[
\omega:=[\omega_v(\mathcal U)]
\]
for the chosen visible branch and chosen cover.
\end{enumerate}
Write
\[
\eta_A:H^2(N(\mathcal U),A)\longrightarrow \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)
\]
for the natural map in the universal coefficient exact sequence
\[
0\to \operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)\to H^2(N(\mathcal U),A)\xrightarrow{\eta_A}\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\to 0
\]
\cite[Sec.~3.1]{Hatcher2002}. Define the \emph{homological evaluation map} to be the concrete homomorphism
\[
\mathrm{ev}_{\omega}:H_2(N(\mathcal U),\mathbb Z)\longrightarrow A,\qquad
h\longmapsto \eta_A(\omega)(h).
\]
Thus $\mathrm{ev}_{\omega}=\eta_A(\omega)\in\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)$.
\end{definition}

\begin{definition}[Visible quotient]\label{def:intrinsic-visible-quotient}
In the setting of \Cref{def:finite-nerve-presentation}, define the \emph{hidden subgroup}
\[
K_{\mathcal U,\omega}:=\mathrm{ev}_{\omega}\bigl(H_2(N(\mathcal U),\mathbb Z)\bigr)=\mathrm{Im}(\mathrm{ev}_{\omega})\leq A
\]
and the associated \emph{visible quotient}
\[
A_{\mathrm{vis}}(\mathcal U,A,\omega):=\operatorname{coker}(\mathrm{ev}_{\omega})=A/K_{\mathcal U,\omega}.
\]
As stated, $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ depends on the choice of cover $\mathcal U$ and band trivialization. \Cref{prop:visible-quotient-invariance} below shows that $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ is independent of the chosen band trivialization. Cover refinement satisfies $K_{\mathcal V,\omega'}\subseteq K_{\mathcal U,\omega}$, and full cover-independence holds when the refinement map on $H_2$ is surjective.
\end{definition}

\begin{lemma}[Finite abelian duality]\label{lem:finite-abelian-duality}
Let $A$ be a finite abelian group and $H\leq A$ a subgroup. Then
\[
H=\bigcap_{\chi\in H^{\perp}}\ker\chi,
\]
where $H^{\perp}:=\{\chi\in\widehat{A}\mid \chi(h)=1\ \forall h\in H\}$ and $\widehat{A}:=\mathrm{Hom}(A,\mathbb T)$.
\end{lemma}

\begin{proof}
This is standard Pontryagin duality for finite abelian groups \cite[Chaps.~5,~12]{DummitFoote2004}. Each $\chi\in H^{\perp}$ factors through $A/H$, so $H\subseteq\bigcap_{\chi\in H^{\perp}}\ker\chi$. For the reverse inclusion, if $a\notin H$, then the quotient $A/H$ is a nontrivial finite abelian group containing the nonzero element $\bar a$, and finite abelian duality for $A/H$ supplies a character $\bar\chi:A/H\to\mathbb T$ with $\bar\chi(\bar a)\neq 1$. The composition $\chi:=\bar\chi\circ\pi_H:A\to\mathbb T$ lies in $H^{\perp}$ but satisfies $\chi(a)=\bar\chi(\bar a)\neq 1$, so $a\notin\bigcap_{\chi\in H^{\perp}}\ker\chi$.
\end{proof}

\begin{proposition}[Invariance of the visible quotient]\label{prop:visible-quotient-invariance}
In the setting of \Cref{def:finite-nerve-presentation}:
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{Change of band trivialization.} If $u:A\xrightarrow{\sim}A'$ is an isomorphism of band groups and $\omega':=u_*(\omega)\in H^2(N(\mathcal U),A')$ is the pushed-forward class, then $\mathrm{ev}_{\omega'}=u\circ\mathrm{ev}_{\omega}$, $K_{\mathcal U,\omega'}=u(K_{\mathcal U,\omega})$, and there is a canonical isomorphism $A'/K_{\mathcal U,\omega'}\cong A/K_{\mathcal U,\omega}$. In particular, $A_{\mathrm{vis}}(\mathcal U,A',\omega')$ is canonically isomorphic to $A_{\mathrm{vis}}(\mathcal U,A,\omega)$.

\item \emph{Cover refinement.} If $\mathcal V=\{b_j\to a\}_{j\in J}$ is a refinement of $\mathcal U$ via an index map $\lambda:J\to I$ and $\omega':=r_{\mathcal V\mathcal U}(\omega)\in H^2(N(\mathcal V),A)$ is the pulled-back class, then
\[
\mathrm{ev}_{\omega'}=\mathrm{ev}_{\omega}\circ f_*,
\]
where $f_*:H_2(N(\mathcal V),\mathbb Z)\to H_2(N(\mathcal U),\mathbb Z)$ is induced by the refinement map $f:N(\mathcal V)\to N(\mathcal U)$. Consequently, $K_{\mathcal V,\omega'}\subseteq K_{\mathcal U,\omega}$ and there is a surjection $A_{\mathrm{vis}}(\mathcal U,A,\omega)\twoheadrightarrow A_{\mathrm{vis}}(\mathcal V,A,\omega')$.

If moreover $f_*$ is surjective on $H_2$ (for example, when the refinement induces a simplicial homotopy equivalence, as in barycentric subdivision), then $K_{\mathcal V,\omega'}=K_{\mathcal U,\omega}$ and $A_{\mathrm{vis}}(\mathcal V,A,\omega')\cong A_{\mathrm{vis}}(\mathcal U,A,\omega)$.
\end{enumerate}
\end{proposition}

\begin{proof}
For \textup{(i)}, naturality of the universal coefficient sequence under the coefficient map $u:A\to A'$ gives a commutative diagram
\[
\begin{CD}
H^2(N(\mathcal U),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A) \\
@Vu_*VV @VV(u\circ-)V \\
H^2(N(\mathcal U),A') @>\eta_{A'}>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A').
\end{CD}
\]
Hence $\eta_{A'}(u_*\omega)=u\circ\eta_A(\omega)$, i.e., $\mathrm{ev}_{\omega'}=u\circ\mathrm{ev}_{\omega}$. Since $u$ is an isomorphism, $K_{\mathcal U,\omega'}=u(K_{\mathcal U,\omega})$ and the isomorphism $u:A\to A'$ induces $A/K_{\mathcal U,\omega}\cong A'/K_{\mathcal U,\omega'}$.

For \textup{(ii)}, the refinement map $f:N(\mathcal V)\to N(\mathcal U)$ (sending simplex $(b_{j_0},\ldots,b_{j_n})$ to $(a_{\lambda(j_0)},\ldots,a_{\lambda(j_n)})$) induces $f^*:H^2(N(\mathcal U),A)\to H^2(N(\mathcal V),A)$ in cohomology and $f_*:H_2(N(\mathcal V),\mathbb Z)\to H_2(N(\mathcal U),\mathbb Z)$ in homology. Naturality of $\eta_A$ with respect to simplicial maps gives
\[
\eta_A(f^*\omega)=\eta_A(\omega)\circ f_*,
\]
i.e.,
\[
\mathrm{ev}_{\omega'}(h)=\mathrm{ev}_{\omega}(f_*(h))
\qquad
\text{for }h\in H_2(N(\mathcal V),\mathbb Z).
\]
Therefore
\[
K_{\mathcal V,\omega'}
=
\mathrm{ev}_{\omega'}(H_2(N(\mathcal V),\mathbb Z))
=
\mathrm{ev}_{\omega}\bigl(f_*(H_2(N(\mathcal V),\mathbb Z))\bigr)
\subseteq
K_{\mathcal U,\omega}.
\]
This gives the surjection $A/K_{\mathcal U,\omega}\twoheadrightarrow A/K_{\mathcal V,\omega'}$. If $f_*$ is surjective on $H_2$, then
\[
f_*\bigl(H_2(N(\mathcal V),\mathbb Z)\bigr)=H_2(N(\mathcal U),\mathbb Z),
\]
so $K_{\mathcal V,\omega'}=K_{\mathcal U,\omega}$.
\end{proof}

\begin{corollary}[Subdivision-type refinements preserve the visible quotient]\label{cor:subdivision-visible-quotient}
In the setting of \Cref{prop:visible-quotient-invariance}, if the refinement map $f:N(\mathcal V)\to N(\mathcal U)$ is a simplicial homotopy equivalence, then
\[
A_{\mathrm{vis}}(\mathcal V,A,\omega')\cong A_{\mathrm{vis}}(\mathcal U,A,\omega).
\]
In particular, barycentric subdivision and iterated subdivision refinements do not change the visible quotient.
\end{corollary}

\begin{proof}
A simplicial homotopy equivalence induces an isomorphism on all homology groups, hence in particular a surjection on $H_2$. Apply \Cref{prop:visible-quotient-invariance}(ii).
\end{proof}

\begin{remark}[What may happen when $H_2$-surjectivity fails]\label{rem:cover-dependence-visible-quotient}
Without surjectivity of the refinement map on $H_2$, one still has only the canonical quotient map
\[
A_{\mathrm{vis}}(\mathcal U,A,\omega)\twoheadrightarrow A_{\mathrm{vis}}(\mathcal V,A,\omega').
\]
Thus refinement can enlarge the visible quotient by forgetting some $H_2$-visible classes. For that reason, presentation-independence is asserted in this paper only for refinement classes whose effect on $H_2$ is explicitly controlled, such as subdivision-type refinements.
\end{remark}

\begin{definition}[Class-admissible characters]\label{def:class-admissible-character}
In the setting of \Cref{def:finite-nerve-presentation}, let
\[
X_{\omega}:=\{\chi\in\widehat A\mid \chi_*(\omega)=0\},
\qquad
\widehat A:=\mathrm{Hom}(A,\mathbb T).
\]
Elements of $X_{\omega}$ are called \emph{class-admissible} for $\omega$.
\end{definition}

\begin{proposition}[Character-theoretic detection of the visible quotient]\label{thm:intrinsic-visible-quotient}
In the setting of \Cref{def:class-admissible-character},
\[
X_{\omega}
=
\{\chi\in\widehat A\mid \chi\circ \mathrm{ev}_{\omega}=0\}
=
\mathrm{Ann}(K_{\mathcal U,\omega}).
\]
Consequently,
\[
K_{\mathcal U,\omega}
=
\bigcap_{\chi\in X_{\omega}}\ker\chi,
\qquad
A_{\mathrm{vis}}(\mathcal U,A,\omega)\cong A\Big/\!\bigcap_{\chi\in X_{\omega}}\ker\chi.
\]
\end{proposition}

\begin{proof}
For every character $\chi:A\to\mathbb T$, naturality of the universal coefficient sequence gives a commutative square
\[
\begin{CD}
H^2(N(\mathcal U),A) @>\eta_A>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A) \\
@V\chi_*VV @VV(\chi\circ -)V \\
H^2(N(\mathcal U),\mathbb T) @>\eta_{\mathbb T}>> \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),\mathbb T).
\end{CD}
\]
Since $\mathbb T$ is divisible and hence injective in $\mathbf{Ab}$, one has
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),\mathbb T)=0
\]
\cite[Sec.~2.3]{Weibel1994}. Thus $\eta_{\mathbb T}$ is an isomorphism by the universal coefficient exact sequence stated in \Cref{def:finite-nerve-presentation}. Therefore
\[
\chi_*(\omega)=0
\Longleftrightarrow
\eta_{\mathbb T}(\chi_*(\omega))=0
\Longleftrightarrow
\chi\circ \eta_A(\omega)=0
\Longleftrightarrow
\chi\circ \mathrm{ev}_{\omega}=0.
\]
This proves the first displayed identity, and the second is simply the definition of the annihilator of a subgroup of $A$.

Let $H:=K_{\mathcal U,\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})\le A$. By the first part of the proof, $X_{\omega}=\mathrm{Ann}(H)=H^{\perp}$. \Cref{lem:finite-abelian-duality} gives
\[
\bigcap_{\chi\in H^{\perp}}\ker\chi=H.
\]
Hence
\[
K_{\mathcal U,\omega}
=
\bigcap_{\chi\in X_{\omega}}\ker\chi.
\]
The final isomorphism follows immediately:
\[
=
\frac{A}{K_{\mathcal U,\omega}}
\cong
A\Big/\!\bigcap_{\chi\in X_{\omega}}\ker\chi.
\qedhere
\]
\end{proof}

\begin{corollary}[Character detection of the visible quotient]\label{cor:intrinsic-character-detection}
The quotient map
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}(\mathcal U,A,\omega)
\]
induces a natural isomorphism
\[
\widehat{A_{\mathrm{vis}}(\mathcal U,A,\omega)}\cong X_{\omega}.
\]
For elements $x,y\in A$, one has
\[
\pi_{\omega}(x)=\pi_{\omega}(y)
\quad\Longleftrightarrow\quad
\chi(x)=\chi(y)\ \text{for every class-admissible }\chi.
\]
\end{corollary}

\begin{proof}
By \Cref{thm:intrinsic-visible-quotient},
\[
X_{\omega}=\mathrm{Ann}(K_{\mathcal U,\omega})=K_{\mathcal U,\omega}^{\perp}.
\]
Characters of $A_{\mathrm{vis}}(\mathcal U,A,\omega)=A/K_{\mathcal U,\omega}$ are exactly the characters of $A$ that kill $K_{\mathcal U,\omega}$, so precomposition with $\pi_{\omega}$ identifies $\widehat{A_{\mathrm{vis}}(\mathcal U,A,\omega)}$ with $X_{\omega}$. The separation statement follows from finite abelian duality applied to the finite group $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ \cite[Chap.~5]{DummitFoote2004}.
\end{proof}

\begin{proposition}[Initiality among pure-extension quotients]\label{thm:intrinsic-pure-ext-initiality}
Let
\[
q:A\twoheadrightarrow B
\]
be a surjective homomorphism onto a finite abelian group $B$. In the setting of \Cref{def:finite-nerve-presentation}, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item the pushed-forward class $q_*(\omega)\in H^2(N(\mathcal U),B)$ lies in the subgroup
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),B);
\]
\item $\eta_B(q_*(\omega))=0$;
\item $q\circ \mathrm{ev}_{\omega}=0$;
\item $\mathrm{Im}(\mathrm{ev}_{\omega})\subseteq \ker q$;
\item $q$ factors uniquely through
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}(\mathcal U,A,\omega).
\]
\end{enumerate}
Hence $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ is the initial coefficient quotient through which the $H_2$-visible part of the obstruction disappears, leaving only the $\ker(\eta_A)$-part of the obstruction.
\end{proposition}

\begin{proof}
The equivalence of \textup{(i)} and \textup{(ii)} is exactly the universal coefficient sequence for $B$ \cite[Sec.~3.1]{Hatcher2002}. Naturality gives
\[
\eta_B(q_*(\omega))=q\circ \eta_A(\omega)=q\circ \mathrm{ev}_{\omega},
\]
so \textup{(ii)} and \textup{(iii)} are equivalent. Conditions \textup{(iii)} and \textup{(iv)} are equivalent by definition of the image of a homomorphism. By \Cref{def:intrinsic-visible-quotient}, $\mathrm{Im}(\mathrm{ev}_{\omega})=K_{\mathcal U,\omega}$, so \textup{(iv)} is equivalent to
\[
K_{\mathcal U,\omega}\subseteq \ker q.
\]
By the universal property of the quotient by $K_{\mathcal U,\omega}$, this is equivalent to unique factorization of $q$ through $\pi_{\omega}$, proving \textup{(v)}.
\end{proof}

\begin{corollary}[Class-trivializing initiality when $H_1$ vanishes]\label{cor:h1-vanishing-class-trivializing}
Assume
\[
H_1(N(\mathcal U),\mathbb Z)=0.
\]
Then, for every surjective homomorphism $q:A\twoheadrightarrow B$, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $q_*(\omega)=0$ in $H^2(N(\mathcal U),B)$;
\item $q$ factors uniquely through
\[
\pi_{\omega}:A\twoheadrightarrow A_{\mathrm{vis}}(\mathcal U,A,\omega).
\]
\end{enumerate}
Hence, when $H_1(N(\mathcal U),\mathbb Z)=0$, the visible quotient $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ is initial among coefficient quotients that annihilate the obstruction class itself.
\end{corollary}

\begin{proof}
If $H_1(N(\mathcal U),\mathbb Z)=0$, then
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),B)=0
\]
for every abelian group $B$. Thus the pure-extension condition in \Cref{thm:intrinsic-pure-ext-initiality}(i) is equivalent to $q_*(\omega)=0$, and the conclusion follows.
\end{proof}

\begin{remark}[Visible quotient without a canonical splitting]\label{rem:no-canonical-splitting}
The quotient $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ and the subgroup $K_{\mathcal U,\omega}=\mathrm{Im}(\mathrm{ev}_{\omega})$ are defined without choosing a splitting of the universal coefficient sequence. Since that sequence need not split canonically, the visible quotient and hidden subgroup are available in settings where a decomposition of $\omega$ into visible and invisible summands would require a noncanonical choice.
\end{remark}

\begin{proposition}[Character-blind pure-extension obstructions]\label{thm:character-blind-obstructions}
In the setting of \Cref{def:finite-nerve-presentation}, the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $K_{\mathcal U,\omega}=0$;
\item $A_{\mathrm{vis}}(\mathcal U,A,\omega)=A$;
\item $\mathrm{ev}_{\omega}=0$;
\item $\omega$ lies in the subgroup
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)\subseteq H^2(N(\mathcal U),A).
\]
\end{enumerate}
If these conditions hold and $\omega\neq 0$, then the chosen branch gerbe $\mathfrak G_v$ is non-neutral, while every character of $A$ is class-admissible for that branch. In particular, if $|\Vis_{p,s}(r)|=1$, then
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]
\end{proposition}

\begin{proof}
By \Cref{def:intrinsic-visible-quotient},
\[
K_{\mathcal U,\omega}=\mathrm{Im}(\mathrm{ev}_{\omega}),
\]
so \textup{(i)} and \textup{(iii)} are equivalent. Since $A_{\mathrm{vis}}(\mathcal U,A,\omega)=A/K_{\mathcal U,\omega}$, conditions \textup{(i)} and \textup{(ii)} are equivalent. By the universal coefficient sequence, \textup{(iii)} holds exactly when $\omega\in\ker(\eta_A)$, and this kernel is the stated $\operatorname{Ext}$-subgroup. Hence all four conditions are equivalent.

If they hold and $\omega\neq 0$, then \Cref{thm:cech-bridge-compatible-realizations}(iv) shows that the branch gerbe $\mathfrak G_v$ is non-neutral.
On the other hand, $\mathrm{ev}_{\omega}=0$ implies
\[
\chi\circ \mathrm{ev}_{\omega}=0
\]
for every $\chi\in\widehat A$, so \Cref{thm:intrinsic-visible-quotient} shows that every character is class-admissible.
Since a finite nerve presentation chooses a visible branch, $\Vis_{p,s}(r)\neq\varnothing$. If moreover $|\Vis_{p,s}(r)|=1$, then \Cref{thm:sheafification-characterization,thm:visible-value-components} gives $M,p\Vdash \CompSec_s(r)$. Let $v$ be the unique visible branch. We already know that $\mathfrak G_v$ is non-neutral. If $M,p\Vdash \Sec_s(r)$, then by \Cref{def:existence-levels} one has $F_{p,s}(a)\neq\varnothing$, so \Cref{def:realization-prestack}(i) yields an object of $(\mathfrak P_r)_a$, hence also an object of $\mathfrak L_r(a)$. Because $v$ is the only visible branch, every object of $\mathfrak L_r(a)$ belongs to $\mathfrak G_v(a)$, contradicting the non-neutrality of $\mathfrak G_v$. Therefore $M,p\nVdash \Sec_s(r)$, and since $r$ is admitted by hypothesis, we conclude
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]
\end{proof}

\begin{corollary}[Automatic blindness when $H_2$ vanishes]\label{cor:h2-vanishing-blindness}
Assume
\[
H_2(N(\mathcal U),\mathbb Z)=0.
\]
Then, for every obstruction class $\omega\in H^2(N(\mathcal U),A)$,
\[
\mathrm{ev}_{\omega}=0,
\qquad
K_{\mathcal U,\omega}=0,
\qquad
A_{\mathrm{vis}}(\mathcal U,A,\omega)=A.
\]
Every nonzero branch obstruction class on such a nerve is therefore character-blind.
\end{corollary}

\begin{proof}
If $H_2(N(\mathcal U),\mathbb Z)=0$, then
\[
\mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)=0,
\]
so $\mathrm{ev}_{\omega}=0$ for every $\omega$. Apply \Cref{thm:character-blind-obstructions}.
\end{proof}

\begin{corollary}[No character-blind obstructions when $H_1$ is free]\label{cor:h1-free-no-blindness}
Assume that $H_1(N(\mathcal U),\mathbb Z)$ is free abelian. Then
\[
K_{\mathcal U,\omega}=0
\quad\Longleftrightarrow\quad
\omega=0.
\]
Equivalently, every nonzero branch obstruction class has a nontrivial $H_2$-visible part. In particular, if $|\Vis_{p,s}(r)|=1$, every nontrivial gluing obstruction on such a nerve produces a proper visible quotient $A_{\mathrm{vis}}(\mathcal U,A,\omega)\subsetneq A$.
\end{corollary}

\begin{proof}
If $H_1(N(\mathcal U),\mathbb Z)$ is free abelian, then
\[
\operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A)=0.
\]
Hence the universal coefficient map $\eta_A$ is injective. By \Cref{thm:character-blind-obstructions},
\[
K_{\mathcal U,\omega}=0
\quad\Longleftrightarrow\quad
\mathrm{ev}_{\omega}=0
\quad\Longleftrightarrow\quad
\eta_A(\omega)=0.
\]
Injectivity of $\eta_A$ therefore gives $K_{\mathcal U,\omega}=0$ if and only if $\omega=0$. If $\omega\neq 0$, then $K_{\mathcal U,\omega}\neq 0$, so $A_{\mathrm{vis}}(\mathcal U,A,\omega)=A/K_{\mathcal U,\omega}$ is a proper quotient of the finite group $A$.
\end{proof}

\begin{example}[A visible quotient on an $S^2$-type nerve]\label{ex:s2-visible-quotient}
Let $\mathcal U$ be a finite good cover of $S^2$ with nerve $N(\mathcal U)$, and let
\[
A=\mathbb Z/4\mathbb Z.
\]
Since
\[
H_1(S^2,\mathbb Z)=0,
\qquad
H_2(S^2,\mathbb Z)\cong \mathbb Z
\]
\cite[Secs.~2.1, 2.2]{Hatcher2002}, the universal coefficient map is an isomorphism
\[
H^2(N(\mathcal U),A)\cong \mathrm{Hom}(H_2(N(\mathcal U),\mathbb Z),A)\cong A.
\]
Let $\omega$ be the class corresponding to the homomorphism sending a fundamental class to $2\in\mathbb Z/4\mathbb Z$. Then
\[
K_{\mathcal U,\omega}=\langle 2\rangle=\{0,2\},
\qquad
A_{\mathrm{vis}}(\mathcal U,A,\omega)\cong (\mathbb Z/4\mathbb Z)/\langle 2\rangle\cong \mathbb Z/2\mathbb Z.
\]
Thus the obstruction is nontrivial, but only its parity quotient survives in $A_{\mathrm{vis}}(\mathcal U,A,\omega)$. For a finite good cover of $S^2$, the nerve theorem identifies $N(\mathcal U)$ with a finite simplicial model of $S^2$, so the chosen cover itself computes the relevant simplicial and \v{C}ech cohomology. If $\omega=[\omega_v(\mathcal U)]$ is the unique visible branch obstruction of the reference under consideration and the chosen realization prestack is gluing-sensitive at $a$, then \Cref{thm:gerbe-null-semantics} yields a genuine gluing-level absence with a nontrivial visible quotient.
\end{example}

\begin{example}[A character-blind obstruction on an $\mathbb{R}P^2$-type nerve]\label{ex:rp2-character-blind}
Let $\mathcal U$ be a finite cover whose Cech nerve is a finite triangulation of $\mathbb{R}P^2$, and take
\[
A=\mathbb Z/2\mathbb Z.
\]
Then
\[
H_2(\mathbb{R}P^2,\mathbb Z)=0,
\qquad
H_1(\mathbb{R}P^2,\mathbb Z)\cong \mathbb Z/2\mathbb Z
\]
\cite[Sec.~2.2]{Hatcher2002}. The universal coefficient sequence therefore yields
\[
H^2(\mathbb{R}P^2,\mathbb Z/2\mathbb Z)
\cong
\operatorname{Ext}^1(\mathbb Z/2\mathbb Z,\mathbb Z/2\mathbb Z)
\cong
\mathbb Z/2\mathbb Z
\]
\cite[Sec.~3.1]{Hatcher2002}. Let $\omega$ be the nonzero class. Then
\[
\mathrm{ev}_{\omega}=0,
\qquad
K_{\mathcal U,\omega}=0,
\qquad
A_{\mathrm{vis}}(\mathcal U,A,\omega)=A,
\]
by \Cref{thm:character-blind-obstructions}. Thus no class-level quotient collapse occurs at all, yet the obstruction class is nonzero. Hence this branch is completely character-blind. For a finite triangulation of $\mathbb{R}P^2$, the nerve is already a finite simplicial complex, so the chosen cover itself computes the relevant simplicial and \v{C}ech cohomology with constant coefficients. If it is the unique visible branch of the admitted reference under consideration and the chosen realization prestack is gluing-sensitive at $a$, then \Cref{thm:gerbe-null-semantics} yields a genuine gluing-level absence that is invisible to every character channel.
\end{example}

\begin{example}[End-to-end: from local reference frame to visible quotient]\label{ex:end-to-end}
We give a single concrete model satisfying the full chain of hypotheses from the local reference frame through the visible quotient.

\smallskip
\noindent\textit{State and local reference frame.}
Fix one information state $p$, one referenceable sort $s$, and one admitted reference $r$ with
\[
M,p\Vdash \Adm_s(r),
\qquad
a=[r]_p.
\]
Let $\mathcal R_s(p)$ be the finite meet-semilattice obtained from the face poset of the boundary of a tetrahedron, together with a top element $a$ and an initial element $0$. Thus the chosen cover of $a$ is
\[
\mathcal U=\{u_0,u_1,u_2,u_3\to a\},
\]
the pairwise meets are $c_{ij}:=u_i\wedge u_j$ for $i<j$, the triple meets are $t_{ijk}:=u_i\wedge u_j\wedge u_k$ for $i<j<k$, and the fourfold meet is
\[
u_0\wedge u_1\wedge u_2\wedge u_3=0.
\]
Let $\Cov_{p,s}$ be the Grothendieck pretopology generated by this cover and the identity covers. The abstract nerve obtained by retaining exactly the non-initial finite meets of $\mathcal U$ is
\[
N(\mathcal U)\cong \partial\Delta^3\simeq S^2.
\]
For every proper object $U<a$, the slice site $\mathcal C_a/U$ inherits only the identity cover (the restriction of $\mathcal U$ along $U\to a$ collapses to $\{\mathrm{id}_U\}$ because $U$ meets at most two of the $u_i$ and the remaining meets are the bottom element). Consequently every Čech $1$-cochain on $\mathcal C_a/U$ is already a coboundary, and
\[
H^1(\mathcal C_a/U,\underline A)=0
\]
for every finite constant abelian sheaf $\underline A$ on such a slice.

\smallskip
\noindent\textit{Local object functor and visible branch.}
Define
\[
F(U)=\{*\}\quad\text{for every }U<a,
\qquad
F(a)=\varnothing,
\]
with all restriction maps between proper objects equal to the unique map on $\{*\}$. On the cover $\mathcal U$, the family $(*)_{i=0}^3$ is compatible on every pairwise meet $c_{ij}$, so
\[
M,p\Vdash \CompSec_s(r).
\]
Because $F(a)=\varnothing$, one has
\[
M,p\nVdash \Sec_s(r).
\]
The compatible family on $\mathcal U$ is unique up to common refinement, so
\[
\Vis_{p,s}(r)=\{v\}
\]
has a single visible branch.

\smallskip
\noindent\textit{Twisted realization prestack.}
Let $A=\mathbb Z/4\mathbb Z$, viewed as a constant band on $\mathcal C_a=\mathcal C_{p,s}/a$. Choose the normalized \v{C}ech $2$-cocycle
\[
g\in \check Z^2(\mathcal U,\underline A)
\]
that sends each positively oriented $2$-simplex of $N(\mathcal U)=\partial\Delta^3$ to $2\in A$ and the opposite orientation to $-2$. Because $\partial\Delta^3$ has no nondegenerate $3$-simplices, every normalized $2$-cochain is automatically a $2$-cocycle; pairing $g$ with the fundamental class of $S^2$ gives $2\in A$, so
\[
[g]\neq 0\in H^2(N(\mathcal U),A)\cong A.
\]

Let $\mathfrak P_r$ be the standard prestack attached to this cocycle \cite[Chap.~IV]{Giraud1971}: for every proper object $U<a$, the fibre $\mathfrak P_r(U)$ is the connected groupoid with one object $x_U$ and automorphism group $A$; restriction functors are the evident identities on objects and on the coefficient group, and the coherence on triple overlaps is twisted by $g$. At the terminal object set
\[
\mathfrak P_r(a)=\varnothing.
\]
Then
\[
\pi_0^{\mathrm{pre}}(\mathfrak P_r)(U)\cong F(U)
\]
for every $U\in\mathcal C_a$, and the automorphism sheaf of each local object is $\underline A$. Hence $\mathfrak P_r$ is a realization prestack with band $\underline A$ in the sense of \Cref{def:realization-prestack}.

Let $\mathfrak L_r:=a(\mathfrak P_r)$ be its stackification. By the standard \v{C}ech $2$-cocycle construction, $\mathfrak L_r$ is the $\underline A$-banded gerbe classified by $[g]$ \cite[Chap.~IV]{Giraud1971}. We claim that $\mathfrak L_r(a)=\varnothing$. Indeed, if $x\in\mathfrak L_r(a)$ existed, then for each $i$ one could choose an isomorphism
\[
\psi_i:x|_{u_i}\xrightarrow{\sim} x_{u_i}.
\]
On each overlap $u_i\wedge u_j$, the transition
\[
h_{ij}:=\psi_j|_{u_i\wedge u_j}\circ \psi_i^{-1}|_{u_i\wedge u_j}\in A
\]
would define a $1$-cochain with $\delta h=g$. This would force $[g]=0$, contrary to the computation above. Hence $\mathfrak L_r(a)=\varnothing$; equivalently, the unique component gerbe $\mathfrak G_v=\mathfrak L_r$ is non-neutral. Because also $\mathfrak P_r(a)=\varnothing$, the terminal-fibre map
\[
(\mathfrak P_r)_a\longrightarrow (\mathfrak L_r)_a
\]
is empty-to-empty and therefore essentially surjective. Local nonemptiness of $\mathfrak G_v$ is immediate from the objects $x_{u_i}$ on the chosen cover. Thus $\mathfrak P_r$ is gluing-sensitive at $a$, and \Cref{thm:gerbe-null-semantics} yields
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r).
\]

\smallskip
\noindent\textit{Visible quotient.}
Since $N(\mathcal U)\simeq S^2$, one has
\[
H_1(N(\mathcal U),\mathbb Z)=0,
\qquad
H_2(N(\mathcal U),\mathbb Z)\cong\mathbb Z.
\]
The universal coefficient sequence therefore identifies
\[
H^2(N(\mathcal U),A)\cong \operatorname{Hom}(H_2(N(\mathcal U),\mathbb Z),A).
\]
Under this identification, the cover-relative branch obstruction class $\omega_v(\mathcal U)=[g]$ corresponds to the homomorphism sending a generator of $H_2$ to $2\in A$. Hence
\[
\mathrm{ev}_{\omega_v(\mathcal U)}(n)=2n,
\qquad
K_{\mathcal U,\omega_v(\mathcal U)}=\{0,2\}=\langle 2\rangle,
\qquad
A_{\mathrm{vis}}(\mathcal U,A,\omega_v(\mathcal U))\cong (\mathbb Z/4\mathbb Z)/\langle 2\rangle\cong \mathbb Z/2\mathbb Z.
\]
The admissible character channels are exactly the characters annihilating $\langle 2\rangle$, in agreement with \Cref{thm:intrinsic-visible-quotient}. This finite example therefore verifies, in a single site, the full passage from a compatible local family with no global object to a non-neutral component gerbe and a proper visible quotient.
\end{example}


\subsection{Connection to sheaf-theoretic contextuality}

The unique-branch case $|\Vis_{p,s}(r)|=1$ deserves separate attention because it connects the present theory to the sheaf-theoretic analysis of contextuality. The comparison has two layers. The first is purely presheaf-theoretic and uses only the support presheaf of an Abramsky--Brandenburger measurement scenario. The second adds genuinely new data: a chosen branch, a gluing-sensitive realization prestack, and a finite obstruction presentation.

\begin{center}
\small
\begin{tabular}{p{0.22\textwidth}p{0.21\textwidth}p{0.25\textwidth}p{0.11\textwidth}}
\textbf{Present paper} & \textbf{Abramsky--Brandenburger analogue} & \textbf{Extra data required here} & \textbf{Status} \\
\hline
support presheaf $S_e$ & support presheaf $S_e$ & none & unconditional \\
visible branch $v$ & compatible family class & a chosen branch in the sheafified support data & conditional \\
component gerbe $\mathfrak G_v$ & no direct analogue & a gluing-sensitive realization prestack and band $\Band$ & conditional \\
visible quotient $A_{\mathrm{vis}}(\mathcal U,A,\omega)$ & no direct analogue & a finite abelian band trivialization and finite nerve presentation & conditional \\
\end{tabular}
\end{center}

\begin{proposition}[Unique-branch comparison with the no-global-section picture: support-presheaf part]\label{thm:unique-branch-contextuality-comparison}
Let $r:\mathsf{Ref}_s$ be admitted at $p$, write $a=[r]_p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item the slice site $\mathcal C_a$ is the measurement-cover site of a finite Abramsky--Brandenburger scenario, and the local object functor $F_{p,s}|_{\mathcal C_a}$ is identified with the support presheaf $S_e$ of an empirical model $e$ \cite{AbramskyBrandenburger2011};
\item $|\Vis_{p,s}(r)|=1$, with unique visible branch $v$.
\end{enumerate}
No realization-prestack hypothesis is used in the following equivalences:
\begin{enumerate}[label=(\roman*),nosep]
\item
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a compatible family};
\]
\item
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a global section};
\]
\item consequently,
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
S_e\ \text{has a compatible family but no global section}.
\]
\end{enumerate}
\end{proposition}

\begin{proof}
Under the identification $F_{p,s}|_{\mathcal C_a}\cong S_e$, a compatible family in $S_e$ is exactly a compatible local family as in \Cref{def:existence-levels}(ii), and a global section of $S_e$ is exactly an element of $F_{p,s}(a)$. Statements \textup{(i)} and \textup{(ii)} are therefore just the definitions of $\CompSec_s(r)$ and $\Sec_s(r)$ specialized to the support presheaf. Statement \textup{(iii)} is then the definition of $\Null^{\mathrm{glue}}_s(r)$.
\end{proof}

\begin{remark}[The discrete prestack is not gluing-sensitive in general]\label{rem:discrete-support-not-gluing-sensitive}
Let $\widetilde S_e$ denote the sheafification of the support presheaf on $\mathcal C_a$, and let $\mathfrak P_r^{\mathrm{disc}}$ be the discrete prestack on $S_e$. If
\[
\widetilde S_e(a)\setminus S_e(a)\neq\varnothing,
\]
then the terminal fibre of $\mathfrak P_r^{\mathrm{disc}}$ is the discrete groupoid on $S_e(a)$, while the terminal fibre of its stackification is the discrete groupoid on $\widetilde S_e(a)$. Every element of $\widetilde S_e(a)\setminus S_e(a)$ therefore defines a global object of the stackification that is not in the essential image of $(\mathfrak P_r^{\mathrm{disc}})_a$. Hence the discrete prestack fails terminal-fibre essential surjectivity whenever sheafification creates a new global section. This is the same no-global-section phenomenon already visible in \Cref{rem:split-not-conservative}.
\end{remark}

\begin{theorem}[Unique-branch comparison: banded-gerbe part]\label{thm:unique-branch-banded-gerbe}
In the setting of \Cref{thm:unique-branch-contextuality-comparison}, assume additionally:
\begin{enumerate}[label=(\roman*),nosep]
\item $r$ admits a gluing-sensitive realization prestack $\mathfrak P_r$ (in the sense of \Cref{def:gluing-sensitive}) whose associated band $\Band$ has a finite abelian trivialization $\Band\cong\underline{A}$ on a gerbe-adapted covering family of $a$;
\item on a chosen gerbe-adapted cover $\mathcal U$ for the unique visible branch $v$, the cover-relative obstruction class
\[
\omega:=[\omega_v(\mathcal U)]
\]
admits a finite nerve presentation with band $A$ as in \Cref{def:finite-nerve-presentation}.
\end{enumerate}
Then:
\begin{enumerate}[label=(\roman*),nosep]
\item
\[
A_{\mathrm{vis}}(\mathcal U,A,\omega)=A/K_{\mathcal U,\omega}=\operatorname{coker}(\mathrm{ev}_{\omega}),
\]
and this quotient is exactly the quotient of $A$ detected by the class-admissible characters (\Cref{thm:intrinsic-visible-quotient});
\item if $\omega\neq 0$, then
\[
A_{\mathrm{vis}}(\mathcal U,A,\omega)=A
\quad\Longleftrightarrow\quad
K_{\mathcal U,\omega}=0
\quad\Longleftrightarrow\quad
\omega\in \operatorname{Ext}^1(H_1(N(\mathcal U),\mathbb Z),A).
\]
Hence the no-global-section obstruction can remain nontrivial while being invisible to every admissible character channel precisely when the surviving class is pure $\operatorname{Ext}$-type.
\end{enumerate}
\end{theorem}

\begin{proof}
By \Cref{def:intrinsic-visible-quotient}, the visible quotient is $\operatorname{coker}(\mathrm{ev}_{\omega})=A/K_{\mathcal U,\omega}$. \Cref{cor:intrinsic-character-detection} identifies this with the quotient detected by class-admissible characters. Statement \textup{(ii)} is exactly \Cref{thm:character-blind-obstructions} applied to the unique visible branch.
\end{proof}

\begin{remark}[Car\`u-type blind cases as pure $\operatorname{Ext}$-residuals]\label{rem:caru-ext-residual}
\Cref{thm:unique-branch-banded-gerbe}(ii) turns the comparison with \cite{Caru2017} into a theorem: the failure of cohomological completeness in the unique-branch case is not an external phenomenon but exactly the case in which the surviving obstruction class lies entirely in the $\operatorname{Ext}$-summand of the universal coefficient sequence. The separation between the support-presheaf part (\Cref{thm:unique-branch-contextuality-comparison}) and the banded-gerbe part (\Cref{thm:unique-branch-banded-gerbe}) makes explicit which conclusions require only the support-presheaf identification and which additionally require a banded lift with finite abelian obstruction.
\end{remark}
