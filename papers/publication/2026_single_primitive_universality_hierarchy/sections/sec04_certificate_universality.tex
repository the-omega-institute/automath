\section{Finite certificates}\label{sec:L2}

A protocol supplies canonical forms.  At a finite resolution one can
also record invariants of the normalization fibers.  For the
Zeckendorf protocol these are the fibers of
\(\Fold_m\colon\Omega_m\to X_m\).

\subsection{Collision moments}
\label{ssec:moments}

For \(x\in X_m\), write
\[
  d_m(x)=|\Fold_m^{-1}(x)|.
\]
Since every legal word folds to itself, \(\Fold_m\) is onto and
\(d_m(x)\ge1\).

\begin{definition}[Collision moment]\label{def:collision-moment}
For \(q\ge0\) and \(m\ge1\), define
\begin{equation}\label{eq:Sq}
  S_q(m)=\sum_{x\in X_m}d_m(x)^q .
\end{equation}
Thus \(S_q(m)\) is the number of \(q\)-tuples
\((\omega^{(1)},\ldots,\omega^{(q)})\in\Omega_m^q\) whose entries
have the same fold image, with the convention
\(S_0(m)=|X_m|\).
\end{definition}

In particular \(S_0(m)=F_{m+2}\) and \(S_1(m)=2^m\).

\subsection{Hankel positivity}
\label{ssec:hankel}

\begin{proposition}[Hankel positivity and log-convexity]
\label{thm:hankel-psd}
For fixed \(m\ge1\), the Hankel matrix
\[
  H_r^{(m)}=\bigl[S_{i+j}(m)\bigr]_{0\le i,j\le r}
\]
is positive semidefinite for every \(r\ge0\).  Consequently:
\begin{enumerate}
\item all principal minors of \(H_r^{(m)}\) are nonnegative;
\item \(S_q(m)^2\le S_{q-1}(m)S_{q+1}(m)\) for \(q\ge1\);
\item the ratios \(S_{q+1}(m)/S_q(m)\) are nondecreasing in \(q\).
\end{enumerate}
\end{proposition}

\begin{proof}
Let
\[
  \nu_m=\sum_{x\in X_m}\delta_{d_m(x)} .
\]
Then \(S_q(m)=\int t^q\,d\nu_m(t)\).  For
\(c_0,\ldots,c_r\in\RR\),
\[
  \sum_{i,j=0}^r c_i c_j S_{i+j}(m)
  =\sum_{x\in X_m}
    \Bigl(\sum_{i=0}^r c_i d_m(x)^i\Bigr)^2\ge0.
\]
This is positive semidefiniteness, the finite case of the Hamburger
moment criterion \cite{Hamburger1920}.  The \(2\times2\) principal
minors give log-convexity, and division by
\(S_q(m)S_{q-1}(m)>0\) gives the monotonicity of the consecutive
ratios.
\end{proof}

\subsection{Finite spectral reconstruction}
\label{ssec:prony}

\begin{proposition}[Prony--Hankel reconstruction]
\label{thm:prony}
Let \(a_1<\cdots<a_{r_m}\) be the distinct values among the fiber
sizes \(d_m(x)\), and put
\[
  n_i=|\{x\in X_m:d_m(x)=a_i\}|.
\]
Let \(N_m=|X_m|\).  The fiber-size spectrum is determined uniquely
by the finite list
\[
  S_0(m),\ldots,S_{2N_m-1}(m).
\]
More sharply, once \(r_m\) is known, it is determined by
\[
  S_0(m),\ldots,S_{2r_m-1}(m).
\]
More precisely:
\begin{enumerate}
\item \(r_m\) is the Hankel rank of the infinite matrix
  \([S_{i+j}(m)]_{i,j\ge0}\), and the preceding finite list with
  \(N_m=|X_m|\) determines this rank;
\item the moment sequence satisfies a unique recurrence of minimum
  order \(r_m\);
\item the recurrence polynomial has roots \(a_1,\ldots,a_{r_m}\),
  and the multiplicities \(n_i\) are then recovered from a
  Vandermonde system.
\end{enumerate}
\end{proposition}

\begin{proof}
The moments have the representation
\[
  S_q(m)=\sum_{i=1}^{r_m} n_i a_i^q .
\]
The functions \(q\mapsto a_i^q\) are linearly independent, since
the \(a_i\) are distinct.  Moreover, the finite Hankel block
\([S_{i+j}(m)]_{0\le i,j<r_m}\) is positive definite: for a
nonzero polynomial \(Q\) of degree \(<r_m\),
\(\sum_i n_i Q(a_i)^2>0\).  Hence the infinite Hankel matrix has
rank \(r_m\).  Since the multiplicities \(n_i\) are positive
integers, \(r_m\le N_m\).  The leading Hankel block of size
\((s+1)\times(s+1)\) is positive definite for \(s<r_m\) and
singular for \(s\ge r_m\).  Therefore the finite moments through
\(S_{2N_m-1}(m)\) determine \(r_m\): it is the least \(s<N_m\) for
which
\(\det[S_{i+j}(m)]_{0\le i,j\le s}=0\), or \(N_m\) if no such
\(s\) occurs.
Let
\[
  P(z)=\prod_{i=1}^{r_m}(z-a_i)
      =z^{r_m}+p_{r_m-1}z^{r_m-1}+\cdots+p_0 .
\]
Since \(P(a_i)=0\), the moments satisfy
\[
  S_{q+r_m}(m)+p_{r_m-1}S_{q+r_m-1}(m)
  +\cdots+p_0S_q(m)=0\qquad(q\ge0).
\]
Minimality follows from the rank statement.  Conversely, the
invertible \(r_m\times r_m\) Hankel block determines the
coefficients \(p_0,\ldots,p_{r_m-1}\) from the square Hankel system
using \(S_0(m),\ldots,S_{2r_m-1}(m)\); its characteristic
polynomial recovers the support \(\{a_i\}\).  Finally,
\[
  S_q(m)=\sum_i n_i a_i^q,\qquad 0\le q<r_m,
\]
is a nonsingular Vandermonde system for the \(n_i\).  This is the
finite Prony reconstruction procedure \cite{BatenkovYomdin2013}.
\end{proof}

Thus the segment through \(S_{2N_m-1}(m)\) certifies the fiber-size
spectrum at resolution \(m\).  Once the Hankel rank has been
identified, the shorter segment through \(S_{2r_m-1}(m)\) suffices.

\subsection{Finite-state compression}
\label{ssec:kernel}

\begin{proposition}[Symmetric-tensor transfer compression]
\label{thm:star-kernel}
Let \(\{B_\alpha\}_{\alpha\in\mathcal A}\subset
M_k(\RR_{\ge0})\) be a finite family and, for \(q\ge1\), set
\[
  A^{(q)}=\sum_{\alpha\in\mathcal A}B_\alpha^{\otimes(q-1)}
\]
on \(V^{\otimes(q-1)}\), where \(V=\RR^k\).  Then:
\begin{enumerate}
\item \(A^{(q)}\) commutes with the action of
  \(\mathfrak S_{q-1}\) by permutation of tensor factors;
\item \(\rho(A^{(q)})=\rho(A^{(q)}|_{\Sym^{q-1}(V)})\);
\item \(\dim\Sym^{q-1}(V)=\binom{k+q-2}{q-1}\), which is
  polynomial in \(q\) for fixed \(k\).
\end{enumerate}
Consequently, if a moment sequence has the transfer form
\[
  S_q(m)=\langle u_q,(A^{(q)})^m v_q\rangle
\]
with \(u_q,v_q\in\Sym^{q-1}(V)\), then it can be computed on the
symmetric subspace.
\end{proposition}

\begin{proof}
For every permutation operator \(P_\pi\) and every matrix \(B\),
\[
  P_\pi B^{\otimes(q-1)}=B^{\otimes(q-1)}P_\pi ;
\]
after summing over \(\alpha\), this proves the first assertion.
Since \(A^{(q)}\) is nonnegative, Perron--Frobenius theory gives a
nonnegative eigenvector \(v\) for its spectral radius
\cite{BermanPlemmons1994}.  The vector
\(\sum_{\pi\in\mathfrak S_{q-1}}P_\pi v\) is nonzero, symmetric,
and still an eigenvector with eigenvalue \(\rho(A^{(q)})\), proving
the second assertion.  The dimension formula is the usual count of
monomials of degree \(q-1\) in \(k\) variables.
\end{proof}

\begin{definition}[Effective finite resolution data]
\label{def:effective-resolution-data}
Finite resolution data for a protocol are called \emph{effective} if
they consist of finite raw-code sets \(C_m\), finite normal-form sets
\(N_m\), and computable surjections
\[
  \mathrm{NF}_m\colon C_m\longrightarrow N_m\qquad(m\ge1),
\]
together with uniform algorithms which, on input \(m\), output
finite lists without repetition containing exactly the elements of
\(C_m\) and \(N_m\).  In particular the cardinalities
\(|C_m|\) and \(|N_m|\) are computable from the resolution data.
All sums, fibers, searches, and moment segments at resolution \(m\)
below are interpreted as bounded computations over these finite
lists.  Thus a bound ``from the finite resolution data'' means a
computable integer obtained uniformly from \(m\), the two lists, and
their cardinalities.
\end{definition}

\begin{definition}[Certificate universality]
\label{def:cert-univ}
A protocol-universal system equipped with effective finite resolution
data is \emph{certificate-universal in the finite-fiber sense} if,
for
\[
  d_m(n)=|\mathrm{NF}_m^{-1}(n)|,\qquad n\in N_m,
\]
the normalization fibers carry the computable moment sequences
\[
  S_q^{(m)}=\sum_{n\in N_m}d_m(n)^q
\]
such that:
\begin{enumerate}
\item for each \(m\), the Hankel matrices
  \([S_{i+j}^{(m)}]_{i,j}\) are positive semidefinite;
\item for each \(m\), a finite initial segment of the moments,
  with a computable bound from the finite resolution data,
  determines the full fiber-multiplicity spectrum.
\end{enumerate}
It is \emph{finite-state certificate-universal} if, in addition,
there is a polynomial \(P\) such that, for each \(q\ge1\), the
sequence in the resolution parameter has a transfer-matrix
representation
\[
  S_q^{(m)}=u_q^{\mathsf T}A_q^m v_q\qquad(m\ge1),
\]
with \(\dim A_q\le P(q)\).  This is the only consequence of
finite-state structure used below: for each fixed \(q\), the
\(m\)-dependence is represented by one fixed finite matrix.
\end{definition}

\begin{proposition}[Finite-fiber certificate universality]
\label{prop:zeckendorf-cert-univ}
Let \(A\) be a nonempty set and let \(g\colon A\to A\) be an
endomorphism of infinite order.  The Zeckendorf protocol of
\Cref{thm:zeckendorf-protocol-univ}, together with the finite fold
maps \(\Fold_m\colon\Omega_m\to X_m\), the standard effective finite
resolution lists, and collision moments
\[
  S_q(m)=\sum_{x\in X_m}|\Fold_m^{-1}(x)|^q,
\]
is certificate-universal in the finite-fiber sense of
\Cref{def:cert-univ}.  More precisely, the Hankel positivity clause
is \Cref{thm:hankel-psd}, and the finite spectral reconstruction
clause is \Cref{thm:prony}, with the computable a priori bound
\(2|X_m|-1=2F_{m+2}-1\) on the required final moment index.
\end{proposition}

\begin{proof}
The underlying address system is protocol-universal for
\(\{+,\times\}\) by \Cref{thm:zeckendorf-protocol-univ}.  It remains
to verify the two finite-fiber certificate clauses.
For each \(m\), the lexicographic list of \(\Omega_m=\{0,1\}^m\) is
computable, and filtering it by the local condition
\(c_kc_{k+1}=0\) gives a computable no-repetition list of \(X_m\).
Thus \(|\Omega_m|=2^m\) and \(|X_m|=F_{m+2}\) are computable from
the finite resolution data.  The map \(\Fold_m\) is a computable
surjection by \Cref{prop:fold-confluence}.  Hence the fiber sizes
\(d_m(x)=|\Fold_m^{-1}(x)|\) are finite and define the moments in
\Cref{def:collision-moment}.  \Cref{thm:hankel-psd} proves that the
Hankel matrices \([S_{i+j}(m)]_{i,j}\) are positive semidefinite for
each fixed resolution.  \Cref{thm:prony} proves that the finite list
\[
  S_0(m),\ldots,S_{2|X_m|-1}(m)
\]
determines the full fiber-size spectrum at that resolution, and
\(|X_m|=F_{m+2}\) by \Cref{prop:fold-confluence}.  These are exactly
the two finite-fiber clauses in \Cref{def:cert-univ}.
\end{proof}

The finite-state strengthening is the additional transfer
representation specified in \Cref{thm:star-kernel}.  Thus the
finite-fiber certificate is tied to the finite normalizer, while the
finite-state condition is a separate computational assertion and is
not a consequence of Hankel positivity or finite Prony
reconstruction.
