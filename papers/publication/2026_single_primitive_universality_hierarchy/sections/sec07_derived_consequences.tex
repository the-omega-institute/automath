\section{Consequences of the finite fiber measure}
\label{sec:consequences}

This appendix records several standard consequences of the finite
measure
\[
  \nu_m=\sum_{x\in X_m}\delta_{d_m(x)} .
\]
They are not used in the proofs of the separations.

\subsection{The collision law}

\begin{proposition}[Finite mixture decomposition]
\label{thm:mixture}
Fix \(m\ge1\) and \(q\ge1\).  Let
\[
  R_m^{(q)}=\{(\omega^{(1)},\ldots,\omega^{(q)})\in\Omega_m^q:
  \Fold_m(\omega^{(1)})=\cdots=\Fold_m(\omega^{(q)})\},
\]
and let \(U_{m,q}\) be the uniform probability measure on
\(R_m^{(q)}\).  For \(x\in X_m\), put
\[
  F_x=\Fold_m^{-1}(x),\qquad \nu_x=\mathrm{Unif}(F_x).
\]
Then
\begin{equation}\label{eq:mixture}
  U_{m,q}
  =\sum_{x\in X_m}\frac{d_m(x)^q}{S_q(m)}\,\nu_x^{\otimes q}.
\end{equation}
Equivalently, if \(X\) denotes the common fold image under
\(U_{m,q}\), then
\[
  \Pr[X=x]=\frac{d_m(x)^q}{S_q(m)},
\]
and, conditional on \(X=x\), the coordinates
\(\omega^{(1)},\ldots,\omega^{(q)}\) are independent and uniform on
\(F_x\).
\end{proposition}

\begin{proof}
The decomposition
\[
  R_m^{(q)}=\bigsqcup_{x\in X_m}F_x^q
\]
is disjoint, and \(|F_x^q|=d_m(x)^q\).  The right hand side of
\eqref{eq:mixture} assigns to each point of \(F_x^q\) the mass
\[
  \frac{d_m(x)^q}{S_q(m)}\,d_m(x)^{-q}=\frac1{S_q(m)}.
\]
Since \(|R_m^{(q)}|=S_q(m)\), this is the uniform law.
\end{proof}

The measures \(\nu_x^{\otimes q}\) have disjoint supports.  Hence
the convex hull
\[
  \mathcal C_{m,q}
  =\operatorname{conv}\{\nu_x^{\otimes q}:x\in X_m\}
\]
is a finite simplex whose extreme points are these product laws.

\subsection{Jacobi realization}

\begin{proposition}[Jacobi realization of collision moments]
\label{thm:jacobi}
Fix \(m\ge1\).  Write
\[
  \nu_m=\sum_{x\in X_m}\delta_{d_m(x)}
       =\sum_{i=1}^{r_m}n_i\,\delta_{a_i},
  \qquad a_1<\cdots<a_{r_m},
\]
where \(a_i\) are the distinct fiber sizes and \(n_i\) their
multiplicities.  Then \(S_q(m)=\int t^q\,d\nu_m(t)\) for all
\(q\ge0\).  There is a unique real Jacobi matrix
\[
  J_m\in M_{r_m}(\RR)
\]
with positive subdiagonal entries
\(\beta_1,\ldots,\beta_{r_m-1}>0\), and a cyclic vector
\(\xi_m\in\RR^{r_m}\), such that
\begin{equation}\label{eq:jacobi-moments}
  S_q(m)=\langle \xi_m,J_m^q\xi_m\rangle\qquad(q\ge0).
\end{equation}
\end{proposition}

\begin{proof}
Take \(H_m=L^2(\nu_m)\), let \(J_m f(t)=t f(t)\), and let
\(\xi_m\equiv1\).  Then
\[
  \langle \xi_m,J_m^q\xi_m\rangle
  =\int t^q\,d\nu_m(t)=S_q(m).
\]
The measure \(\nu_m\) has \(r_m\) atoms, so \(\dim H_m=r_m\).  The
constant function is cyclic because polynomials separate the finite
support.  The cyclic spectral theorem gives uniqueness up to unitary
equivalence, and Gram--Schmidt applied to the monomial basis gives
the Jacobi form.
\end{proof}

\begin{corollary}[Stieltjes continued fraction]
\label{cor:stieltjes}
The Cauchy transform
\[
  G_m(z)=\langle \xi_m,(zI-J_m)^{-1}\xi_m\rangle
\]
has the finite Stieltjes continued fraction
\[
  G_m(z)
  =
  \frac{\|\xi_m\|^2}
  {z-\alpha_1
    - \dfrac{\beta_1^2}
      {z-\alpha_2
        - \dfrac{\beta_2^2}
          {\ddots
            - \dfrac{\beta_{r_m-1}^2}
              {z-\alpha_{r_m}}}}}.
\]
The Jacobi parameters \((\alpha_j,\beta_j)_j\) are uniquely
determined by \(\nu_m\), and hence by the Hankel certificate
\cite{Stieltjes1894}.
\end{corollary}

\begin{proof}
Write the multiplication operator from \Cref{thm:jacobi} in the
orthonormal polynomial basis.  Successive Schur complements evaluate
the resolvent of the resulting finite tridiagonal Jacobi matrix and
give the displayed continued fraction.  Uniqueness is the usual
uniqueness of the Jacobi parameters for the finite positive measure
\(\nu_m\).
\end{proof}

\subsection{Escort concentration}

\begin{proposition}[Escort large-deviation estimate]
\label{thm:escort-ldp}
Fix \(m\ge1\), and define
\[
  \pi_{m,q}(x)=\frac{d_m(x)^q}{S_q(m)},\qquad x\in X_m,\ q>0.
\]
Let
\[
  D_m=\max_x d_m(x),\qquad
  A_m^{\max}=\{x:d_m(x)=D_m\},\qquad
  N_m=|A_m^{\max}|.
\]
If \(A_m^{\max}\ne X_m\), set
\[
  D_m^{(2)}=\max_{x\notin A_m^{\max}}d_m(x),\qquad
  \Delta_m=\log D_m-\log D_m^{(2)};
\]
if \(A_m^{\max}=X_m\), set \(\Delta_m=+\infty\).  Then, in the
finite-alphabet Laplace-principle sense \cite{DemboZeitouni2010},
\[
  -\frac1q\log\pi_{m,q}(x)
  \longrightarrow I_m(x)=\log D_m-\log d_m(x)
  \qquad(q\to\infty).
\]
Moreover,
\begin{equation}\label{eq:escort-tail}
  1-\pi_{m,q}(A_m^{\max})
  \le
  \frac{|X_m|-N_m}{N_m}e^{-q\Delta_m},
\end{equation}
with the right hand side interpreted as \(0\) when
\(A_m^{\max}=X_m\).  If \(A_m^{\max}\ne X_m\), then any
\[
  q>\Delta_m^{-1}
  \log\!\left(\frac{2(|X_m|-N_m)}{N_m}\right)
\]
forces \(\pi_{m,q}(A_m^{\max})>1/2\) whenever the logarithm is
positive; if it is nonpositive, the same conclusion holds for every
\(q>0\).
\end{proposition}

\begin{proof}
If \(A_m^{\max}=X_m\), all assertions are immediate.  Otherwise,
\[
  S_q(m)=D_m^q\left(N_m+
    \sum_{x\notin A_m^{\max}}\left(\frac{d_m(x)}{D_m}\right)^q
  \right).
\]
It follows that
\[
  \pi_{m,q}(x)=
  \frac{(d_m(x)/D_m)^q}
  {N_m+\sum_{y\notin A_m^{\max}}(d_m(y)/D_m)^q},
\]
and the rate limit follows because the denominator tends to
\(N_m\).  For \(x\notin A_m^{\max}\),
\(d_m(x)/D_m\le e^{-\Delta_m}\), which gives
\eqref{eq:escort-tail}; the last assertion is the stated
specialization of that bound.
\end{proof}

\subsection{A cross-resolution kernel}

\begin{proposition}[Cross-resolution positive-definite kernel]
\label{thm:cross-kernel}
Let \(\mu_{1/2}\) be Bernoulli-\(\tfrac12\) product measure on
\(\Omega_\infty=\{0,1\}^{\NN}\), and let
\(\mathrm{pref}_m(\omega)\) be the length-\(m\) prefix.  For
\(q\ge0\), define
\[
  \Phi_{m,q}(\omega)
  =d_m\!\bigl(\Fold_m(\mathrm{pref}_m(\omega))\bigr)^{q/2}
\]
and
\begin{equation}\label{eq:cross-kernel}
  K_q(m,n)
  =\int_{\Omega_\infty}
    \Phi_{m,q}(\omega)\Phi_{n,q}(\omega)\,d\mu_{1/2}(\omega).
\end{equation}
Then \(K_q\) is a positive-definite kernel on \(\NN\) in the
Moore--Aronszajn sense \cite{Aronszajn1950}.  On the diagonal,
\begin{equation}\label{eq:kernel-diagonal}
  K_q(m,m)=2^{-m}S_{q+1}(m).
\end{equation}
\end{proposition}

\begin{proof}
For \(m_1,\ldots,m_r\in\NN\) and \(c_1,\ldots,c_r\in\RR\),
\[
  \sum_{i,j}c_i c_jK_q(m_i,m_j)
  =\int_{\Omega_\infty}
    \left(\sum_i c_i\Phi_{m_i,q}(\omega)\right)^2
    d\mu_{1/2}(\omega)\ge0.
\]
This proves positive definiteness.  For the diagonal,
\[
  2^mK_q(m,m)
  =\sum_{\eta\in\Omega_m}
    d_m(\Fold_m(\eta))^q
  =\sum_{x\in X_m}d_m(x)^{q+1}
  =S_{q+1}(m).
\]
\end{proof}

\subsection{The collision zeta function}

\begin{proposition}[Rationality of the collision zeta function]
\label{thm:zeta}
For fixed \(m\), define
\[
  \zeta_m(z)=
  \exp\left(\sum_{q\ge1}\frac{S_q(m)}q z^q\right).
\]
Then
\begin{equation}\label{eq:zeta-rational}
  \zeta_m(z)
  =\prod_{x\in X_m}(1-d_m(x)z)^{-1}
  =\prod_{j=1}^{r_m}(1-a_jz)^{-n_j},
\end{equation}
with radius of convergence \(1/D_m\).  Moreover,
\begin{equation}\label{eq:zeta-primitive}
  \zeta_m(z)=\prod_{n\ge1}(1-z^n)^{-p_n^{(m)}},
  \qquad
  p_n^{(m)}=\sum_{x\in X_m}M_n(d_m(x)),
\end{equation}
where
\[
  M_n(a)=\frac1n\sum_{d\mid n}\mu(d)a^{n/d}
\]
is the necklace polynomial.  In particular \(p_n^{(m)}\in\NN\).
\end{proposition}

\begin{proof}
Since \(S_q(m)=\sum_x d_m(x)^q\),
\[
  \log\zeta_m(z)
  =\sum_{q\ge1}\sum_x\frac{(d_m(x)z)^q}{q}
  =-\sum_x\log(1-d_m(x)z),
\]
which proves \eqref{eq:zeta-rational}.  The primitive expansion is
the standard Witt--M\"obius inversion \cite{Moree2005}:
\[
  S_n(m)=\sum_{d\mid n}d\,p_d^{(m)},\qquad
  p_n^{(m)}=\frac1n\sum_{d\mid n}\mu(d)S_{n/d}(m).
\]
Substituting \(S_{n/d}(m)=\sum_x d_m(x)^{n/d}\) gives
\eqref{eq:zeta-primitive}.  The integrality of \(M_n(a)\) for
\(a\in\NN\) is classical.
\end{proof}

\subsection{A finite Gibbs principle}

\begin{proposition}[Gibbs variational principle on fibers]
\label{thm:gibbs}
Fix \(m\ge1\) and \(x\in X_m\), and let \(F_x=\Fold_m^{-1}(x)\).
Let \(R_x\colon F_x\to\RR\) be any statistic.  Define
\[
  Z_x(\theta)=\sum_{\omega\in F_x}e^{\theta R_x(\omega)},\qquad
  \psi_x(\theta)=\log Z_x(\theta).
\]
Then the finite Gibbs variational principle gives
\cite{Ellis1985}
\begin{equation}\label{eq:gibbs-vp}
  \psi_x(\theta)=
  \sup_{\pi\in\Delta(F_x)}
  \{\theta\,\mathbb E_\pi[R_x]+H(\pi)\},
\end{equation}
and the unique maximizer is
\[
  \pi_\theta(\omega)=\frac{e^{\theta R_x(\omega)}}{Z_x(\theta)}.
\]
If \(F_x\simeq F_x^{(1)}\times F_x^{(2)}\) and
\[
  R_x(\omega_1,\omega_2)
  =R_x^{(1)}(\omega_1)+R_x^{(2)}(\omega_2),
\]
then \(\psi_x\) decomposes additively.
\end{proposition}

\begin{proof}
For any \(\pi\in\Delta(F_x)\),
\[
  D_{\mathrm{KL}}(\pi\|\pi_\theta)
  =-H(\pi)-\theta\,\mathbb E_\pi[R_x]+\psi_x(\theta)\ge0,
\]
with equality if and only if \(\pi=\pi_\theta\).  This proves
\eqref{eq:gibbs-vp}.  The product assertion follows from
\[
  Z_x(\theta)=Z_x^{(1)}(\theta)Z_x^{(2)}(\theta).
\]
\end{proof}

\subsection{The natural numbers object}

\begin{proposition}[Fold numerals as a natural numbers object]
\label{prop:nno}
Define
\[
  0_X=\Val^{-1}(0)\in X_{\infty}^{\mathrm{fin}},\qquad
  \mathsf S_X(c)=\Val^{-1}(\Val(c)+1).
\]
Then \((X_{\infty}^{\mathrm{fin}},0_X,\mathsf S_X)\) is
isomorphic in \(\mathbf{Set}\) to
\((\NN,0,\mathsf S)\), hence is a natural numbers object
\cite{Lawvere1964}.  Under this identification, the numeral
\[
  c_n=\Val^{-1}(n)
\]
acts on every endomorphism \(f\colon A\to A\) by \(n\)-fold
iteration \(f\mapsto f^{\circ n}\).  The level-zero readout
identifies addition with composition of iterates; the level-one
readout identifies multiplication with composition of the operators
\(f\mapsto f^{\circ n}\).
\end{proposition}

\begin{proof}
The valuation is a bijection by \Cref{prop:fold-confluence}, and it
carries \(0_X\) to \(0\) and \(\mathsf S_X\) to the usual successor
on \(\NN\).  This proves the natural-numbers-object statement in
\(\mathbf{Set}\).  The iteration identities are exactly
\Cref{thm:two-level} transported through this isomorphism.
\end{proof}
