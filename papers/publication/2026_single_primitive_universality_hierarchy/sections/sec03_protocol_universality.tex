\section{Protocol Universality: Scan--Projection Arithmetic
  Genesis}\label{sec:L1}
% ══════════════════════════════════════════════════════════

We now describe a deeper form of universality in which arithmetic
\emph{operations themselves} are generated by a single protocol,
rather than being external primitives applied to a unified
expression library.  Full proofs appear
in~\cite{OmegaSPG2026}; we state the results in the form needed
for the hierarchy comparison.

% ── 3.1  The scan--projection generator ───────────────────
\subsection{The scan--projection generator}\label{ssec:spg}

\begin{definition}[Monogenic action and binary readout]
\label{def:spg}
Let $(X, \mathcal{B}, \mu)$ be a probability space and
$T \colon X \to X$ a measure-preserving transformation generated
by a single element $g$ of a (semi)group~$\Gamma = \langle g
\rangle$, so that $T^{t}(x) = g^{t} \cdot x$ for $t \in \NN$.
Fix a measurable \emph{window} $W \subseteq X$.  The
\emph{binary readout} is
\[
  s_{t}(x) \;=\; \one_{W}(T^{t} x) \;\in\; \{0,1\}.
\]
The \emph{microstate word} of resolution~$m$ starting at time~$t$
is
\[
  \omega_{m}(x;\,t)
    \;=\; s_{t}(x)\, s_{t+1}(x) \cdots s_{t+m-1}(x)
    \;\in\; \Omega_{m} := \{0,1\}^{m}.
\]
\end{definition}

The critical philosophical point: \emph{time is not imported as an
external parameter}; it is the iteration count of the single
generator~$g$.  The only inputs are $(X,\mu,T,W)$.

% ── 3.2  Fold normalization ──────────────────────────────
\subsection{Fibonacci--Zeckendorf fold normalization}
\label{ssec:fold}

\begin{definition}[Fibonacci-weighted value and fold map]
\label{def:fold}
For $\omega = \omega_{1}\cdots\omega_{m} \in \Omega_{m}$, define
the \emph{Fibonacci-weighted value}
\[
  N(\omega) \;:=\; \sum_{k=1}^{m} \omega_{k}\, F_{k+1}
    \;\in\; \NN,
\]
where $F_{k}$ is the $k$-th Fibonacci number.
Let $Z(n) = (c_{1}, c_{2}, \ldots)$ denote the
\emph{Zeckendorf representation} of~$n$~\cite{Zeckendorf1972},
the unique representation as a sum of non-consecutive Fibonacci
numbers.  Efficient algorithms for arithmetic in Zeckendorf
numeration are developed in~\cite{AhlbachEtAl2013}; the
automaton-theoretic framework for Ostrowski numeration (of
which Zeckendorf is the golden-ratio case), including
addition via finite automata, is established
in~\cite{HieronymiTerry2018}.  The \emph{fold map} at resolution~$m$ is
\[
  \Fold_{m} \colon \Omega_{m} \to X_{m},
  \qquad
  \Fold_{m}(\omega) \;:=\; \pi_{m}\!\bigl(Z(N(\omega))\bigr),
\]
where $\pi_{m}$ denotes reduction modulo~$F_{m+2}$
followed by Zeckendorf encoding, so that
$\Val(\Fold_{m}(\omega)) \equiv N(\omega) \pmod{F_{m+2}}$.
Here $X_{m} \subseteq \Omega_{m}$ is the set of Zeckendorf-legal
words (no two consecutive~$1$s), and the fold map realizes the
ring isomorphism $X_{m} \cong \ZZ/F_{m+2}\ZZ$.
\end{definition}

The fold map is computed by four local value-preserving rewrite
rules on digit strings:
\begin{alignat}{2}
  &\text{(R1)}& \quad a_{k} \ge 1,\; a_{k+1} \ge 1
    &\;\longmapsto\; \mathbf{a} - e_{k} - e_{k+1} + e_{k+2},
    \label{eq:R1}\\
  &\text{(R2)}& \quad a_{1} \ge 2
    &\;\longmapsto\; \mathbf{a} - 2e_{1} + e_{2},
    \label{eq:R2}\\
  &\text{(R3)}& \quad a_{2} \ge 2
    &\;\longmapsto\; \mathbf{a} - 2e_{2} + e_{1} + e_{3},
    \label{eq:R3}\\
  &\text{(R4)}& \quad a_{k} \ge 2\; (k \ge 3)
    &\;\longmapsto\; \mathbf{a} - 2e_{k} + e_{k-2} + e_{k+1}.
    \label{eq:R4}
\end{alignat}

\begin{proposition}[Termination, confluence, uniqueness
  {\cite[Prop.~6.3]{OmegaSPG2026}}]
\label{prop:fold-confluence}
The rewrite system \eqref{eq:R1}--\eqref{eq:R4} is strongly
terminating and locally confluent.  By Newman's lemma each
$\mathbf{a} \in \NN^{m}$ has a unique normal form
$\mathrm{NF}(\mathbf{a}) \in X_{m}$.
\begin{enumerate}[label=(\roman*)]
\item At finite resolution,
  $\Val(\mathrm{NF}(\mathbf{a})) \equiv \Val(\mathbf{a})
    \pmod{F_{m+2}}$,
  reflecting the ring isomorphism $X_{m} \cong \ZZ/F_{m+2}\ZZ$.
\item In the inverse-limit completion
  $X_{\infty}^{\mathrm{fin}}$, the value is preserved exactly:
  $\Val(\mathrm{NF}_{\infty}(\mathbf{a})) = \Val(\mathbf{a})$
  for any eventually-zero $\mathbf{a} \in \NN^{(\NN)}$.
\end{enumerate}
The distinction between (i) and~(ii) reflects the fact that
rule~(R1) may emit carries into position $m+2$, which are
retained in $X_{\infty}^{\mathrm{fin}}$ but projected out modulo
$F_{m+2}$ at finite resolution.
\end{proposition}

The spaces $(X_{m})_{m \ge 1}$ form an inverse system under
truncation.  We distinguish two infinite-level completions:

\begin{definition}[Finite and profinite completions]
\label{def:completions}
The \emph{eventually-zero} legal words are
\begin{equation}\label{eq:X-fin}
  X_{\infty}^{\mathrm{fin}}
    \;:=\; \bigl\{(c_{1},c_{2},\ldots) \in \{0,1\}^{\NN}
      : c_{k}\, c_{k+1} = 0\;\forall k,\;
      c_{k} = 0 \text{ for all } k \gg 0\bigr\}.
\end{equation}
This is a countable set in bijection with~$\NN$ via the
Fibonacci-weighted valuation
$\Val(c) := \sum_{k \ge 1} c_{k}\, F_{k+1}$
(a finite sum for eventually-zero words).

The \emph{profinite completion} is the full inverse limit
\begin{equation}\label{eq:X-infty}
  \widehat{X}
    \;:=\; \varprojlim_{m} X_{m}
    \;\cong\; \bigl\{(c_{1},c_{2},\ldots) \in \{0,1\}^{\NN}
      : c_{k}\, c_{k+1} = 0 \;\forall k\bigr\},
\end{equation}
which is uncountable and compact (a Cantor-type set),
analogous to the $p$-adic integers~$\ZZ_{p}$.
\end{definition}

\begin{remark}\label{rem:finite-vs-profinite}
The profinite completion $\widehat{X}$ carries a natural
compact-group or compact-monoid structure, but $\Val$ does
not extend to a bijection $\widehat{X} \to \NN$ since
$\widehat{X}$ is uncountable.  All arithmetic constructions
below use~$X_{\infty}^{\mathrm{fin}}$, not~$\widehat{X}$.
\end{remark}

% ── 3.3  Emergent arithmetic ─────────────────────────────
\subsection{Arithmetic as fold readout}\label{ssec:arith}

\begin{definition}[Stable addition and multiplication]
\label{def:stable-arith}
For $c, d \in X_{\infty}^{\mathrm{fin}}$, define:
\begin{align}
  c \oplus d &\;:=\; \Fold_{\infty}(c + d),
    \label{eq:stable-add}\\
  c \otimes d &\;:=\; \Val^{-1}\!\bigl(\Val(c) \cdot \Val(d)\bigr),
    \label{eq:stable-mul}
\end{align}
where $c + d$ is coordinate-wise addition
(again eventually zero) and
$\Val \colon X_{\infty}^{\mathrm{fin}} \to \NN$ is the
Fibonacci-weighted valuation.
\end{definition}

\begin{theorem}[Semiring structure
  {\cite[Thms.~8.5, 8.9]{OmegaSPG2026}}]
\label{thm:semiring}
$(X_{\infty}^{\mathrm{fin}}, \oplus, \otimes)$ is a commutative
semiring with unit $0_{X} = (0,0,\ldots)$, and the valuation map
\[
  \Val \colon
    (X_{\infty}^{\mathrm{fin}},\, \oplus,\, \otimes)
      \;\xrightarrow{\;\sim\;}\;
    (\NN,\, +,\, \times)
\]
is a semiring isomorphism.  At finite resolution, the induced
structure is $X_{m} \cong \ZZ / F_{m+2}\ZZ$ as a ring.
\end{theorem}

The decisive point: neither $\oplus$ nor $\otimes$ is a primitive
operation.  Both are \emph{derived} from coordinate stacking
followed by fold normalization.

% ── 3.4  Single primitive via two-level readout ──────────
\subsection{Composition as the sole primitive}\label{ssec:sole}

\begin{theorem}[Two-level readout of composition
  {\cite[Thm.~8.11]{OmegaSPG2026}}]
\label{thm:two-level}
Let $A$ be a non-empty set and $g \colon A \to A$ an endomorphism
of infinite order (i.e., $g^{m} \neq g^{n}$ for $m \neq n$).
Define:
\begin{align}
  \text{Level 0:} &\quad S_{0} := \langle g \rangle
    = \{g^{\circ n} : n \in \NN\} \subseteq \End(A),
    \label{eq:level0}\\
  \text{Level 1:} &\quad S_{1} := \{I_{n} : n \in \NN\}
    \subseteq \End(\End(A)),
    \quad I_{n}(f) := f^{\circ n}.
    \label{eq:level1}
\end{align}
With embeddings $\iota_{0}(n) := g^{\circ n}$ and
$\iota_{1}(n) := I_{n}$:
\begin{align}
  \iota_{0}(m + n) &= \iota_{0}(m) \circ \iota_{0}(n)
    &&\text{(addition from composition)},
    \label{eq:add-from-comp}\\
  \iota_{1}(m \cdot n) &= \iota_{1}(m) \circ \iota_{1}(n)
    &&\text{(multiplication from composition)}.
    \label{eq:mul-from-comp}
\end{align}
\end{theorem}

\begin{corollary}[Minimal primitive set
  {\cite[Cor.~8.12]{OmegaSPG2026}}]
\label{cor:minimal}
Arithmetic at the stable-address level can be generated by the
successor map~$\mathsf{S}$ and the fold operator alone.
Addition is a level-0 composition readout; multiplication
is its level-1 stacking.
(This is the Zeckendorf-numeral instantiation of the classical
Church-numeral iteration scheme~\cite{Church1936};
the contribution here is not the iteration identities themselves
but their embedding in the fold-normalization framework with
decidable canonical forms.)
\end{corollary}

% ── 3.5  Formal definition of protocol universality ──────
\subsection{Formal definition}

\begin{definition}[Protocol universality]\label{def:prot-univ}
A \emph{single-primitive protocol} for a class of
operations~$\mathcal{O}$ over a structured domain~$(D, \mathcal{O})$
consists of:
\begin{enumerate}[label=(\alph*)]
\item a single generating operation $\sigma \colon D \to D$
  (or $\sigma \colon D^{k} \to D$),
\item a computable normalization map
  $\mathrm{NF} \colon D^{*} \to D$ with unique normal forms, and
\item a finite family of \emph{readout functors}
  $\{\Phi_{\ell}\}_{\ell=0}^{L}$ such that each
  $\mathcal{O}_{i} \in \mathcal{O}$ is realized as
  $\Phi_{\ell}$-image of~$\sigma$-composition for some level~$\ell$.
\end{enumerate}
We say the protocol is \emph{protocol-universal for~$\mathcal{O}$}
if every operation in~$\mathcal{O}$ is so realized, and
the word problem---whether two $\sigma$-compositions yield the
same normal form---is decidable.
\end{definition}

The scan--projection generator with fold normalization and the
two-level readout $(\Phi_{0}, \Phi_{1})$ of
\Cref{thm:two-level} constitutes a protocol-universal system for
$\mathcal{O} = \{+, \times\}$ over~$(\NN, +, \times)$.
