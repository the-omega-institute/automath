\section{Protocol universality}\label{sec:L1}

Expression universality has no built-in normal form.  The next
setting has a single primitive accompanied by a computable
normalization.  The construction is deliberately
elementary: a monogenic dynamics produces binary words, and the
fold map replaces each word by its Zeckendorf address.

\subsection{Symbolic readout}

\begin{definition}[Monogenic action and binary readout]
\label{def:spg}
Let \((X,\mathcal B,\mu)\) be a probability space and let
\(T\colon X\to X\) be a measure-preserving transformation generated
by a single element \(g\) of a semigroup \(\Gamma=\langle g\rangle\),
so that \(T^t(x)=g^t\cdot x\) for \(t\in\NN\).  For a measurable
window \(W\subseteq X\), put
\[
  s_t(x)=\one_W(T^t x)\in\{0,1\}.
\]
The microstate word of length \(m\) starting at time \(t\) is
\[
  \omega_m(x;t)=s_t(x)s_{t+1}(x)\cdots s_{t+m-1}(x)
  \in\Omega_m:=\{0,1\}^m .
\]
\end{definition}

This is the usual passage from a dynamical system and a partition
element to symbolic itineraries \cite{MorseHedlund1940}.  The only
time-evolution datum is the generator \(g\).

\subsection{The Fibonacci fold}
\label{ssec:fold}

Let \(F_1=F_2=1\) and \(F_{n+2}=F_{n+1}+F_n\).  For \(m\ge1\), let
\[
  X_m=\{c_1\cdots c_m\in\{0,1\}^m:
        c_kc_{k+1}=0 \text{ for } 1\le k<m\}
\]
be the set of legal Fibonacci words, and define
\[
  \Val_m(c_1\cdots c_m)=\sum_{k=1}^m c_kF_{k+1}.
\]

\begin{definition}[Fold map]\label{def:fold}
For \(\omega=\omega_1\cdots\omega_m\in\Omega_m\), set
\[
  N(\omega)=\sum_{k=1}^m \omega_kF_{k+1}.
\]
The fold map at resolution \(m\) is the map
\(\Fold_m\colon\Omega_m\to X_m\) which sends \(\omega\) to the
unique legal word \(c\in X_m\) satisfying
\begin{equation}\label{eq:fold-residue}
  \Val_m(c)\equiv N(\omega)\pmod {F_{m+2}}.
\end{equation}
\end{definition}

\begin{proposition}[Finite fold normal forms]
\label{prop:fold-confluence}
For every \(m\ge1\), the map
\[
  \Val_m\colon X_m\longrightarrow \{0,1,\ldots,F_{m+2}-1\}
\]
is a bijection.  Hence \(\Fold_m\) is well defined and computable.
Moreover, if
\[
  X_{\infty}^{\mathrm{fin}}
  =\{(c_1,c_2,\ldots)\in\{0,1\}^{\NN}:
    c_kc_{k+1}=0 \text{ for all }k,\ c_k=0\text{ for }k\gg0\},
\]
then
\[
  \Val(c)=\sum_{k\ge1}c_kF_{k+1}
\]
is a bijection \(X_{\infty}^{\mathrm{fin}}\to\NN\).
\end{proposition}

\begin{proof}
Zeckendorf's theorem gives a unique expression of every
\(n\ge0\) as a sum of nonconsecutive Fibonacci numbers
\(F_{k+1}\) \cite{Zeckendorf1972}.  If \(0\le n<F_{m+2}\), no
summand \(F_{m+2}\) or larger can occur, so the digits occupy only
the positions \(1,\ldots,m\).  This gives a legal word in \(X_m\).
Conversely, if \(M_m\) denotes the maximum of \(\Val_m\) on \(X_m\),
then
\[
  M_m=\max(M_{m-1},F_{m+1}+M_{m-2}),\qquad M_0=0,\quad M_1=1,
\]
and induction gives \(M_m=F_{m+2}-1\).  Thus every legal word has a
value in the displayed range, and uniqueness in Zeckendorf's
theorem proves the finite bijection.  The infinite statement is the
same theorem without the upper bound; eventual vanishing makes the
sum finite.  Computability follows from the greedy Zeckendorf
algorithm, or from standard finite-automaton methods for
Zeckendorf/Ostrowski arithmetic
\cite{AhlbachEtAl2013,HieronymiTerry2018}.
\end{proof}

The finite legal-word spaces form an inverse system under
coordinate truncation.  We distinguish its countable
eventually-zero part, where the arithmetic is carried out, from its
compact completion.

\begin{definition}[Finite and compact completions]
\label{def:completions}
The eventually-zero space \(X_{\infty}^{\mathrm{fin}}\) is the
countable set in \Cref{prop:fold-confluence}, identified with
\(\NN\) by \(\Val\).  The compact inverse limit is
\begin{equation}\label{eq:X-infty}
  \widehat X
  =\varprojlim_m X_m
  \cong
  \{(c_1,c_2,\ldots)\in\{0,1\}^{\NN}:
    c_kc_{k+1}=0\text{ for all }k\}.
\end{equation}
It is a compact zero-dimensional space, analogous to a profinite
completion \cite{RibesZalesskii2010}.  The arithmetic below,
however, is carried out on \(X_{\infty}^{\mathrm{fin}}\).
\end{definition}

\begin{remark}\label{rem:finite-vs-profinite}
The valuation is not a bijection \(\widehat X\to\NN\), since
\(\widehat X\) is uncountable.  Stable arithmetic is therefore an
operation on eventually-zero addresses, not on the whole compact
inverse limit.
\end{remark}

\subsection{Arithmetic by readout}
\label{ssec:arith}

\begin{definition}[Stable addition and multiplication]
\label{def:stable-arith}
For \(c,d\in X_{\infty}^{\mathrm{fin}}\), define
\begin{align}
  c\oplus d
    &=\Val^{-1}\bigl(\Val(c)+\Val(d)\bigr),\label{eq:stable-add}\\
  c\otimes d
    &=\Val^{-1}\bigl(\Val(c)\Val(d)\bigr).
      \label{eq:stable-mul}
\end{align}
Equivalently, \(c\oplus d\) is obtained by adding the two digit
sequences coordinatewise and applying Zeckendorf normalization.
\end{definition}

\begin{proposition}[Transported semiring structure]
\label{thm:semiring}
\((X_{\infty}^{\mathrm{fin}},\oplus,\otimes)\) is a commutative
semiring with additive identity \(0_X=(0,0,\ldots)\) and
multiplicative identity \(\Val^{-1}(1)\).  The valuation map
\[
  \Val\colon
  (X_{\infty}^{\mathrm{fin}},\oplus,\otimes)
  \xrightarrow{\ \sim\ }(\NN,+,\times)
\]
is a semiring isomorphism.  At finite resolution, transporting the
ring operations on \(\ZZ/F_{m+2}\ZZ\) through the bijection
\(\Val_m\colon X_m\to\ZZ/F_{m+2}\ZZ\) gives a ring structure on
\(X_m\) isomorphic to \(\ZZ/F_{m+2}\ZZ\).
\end{proposition}

\begin{proof}
The operations \(\oplus\) and \(\otimes\) are the pullbacks of
\(+\) and \(\times\) along the bijection \(\Val\).  The semiring
axioms and the asserted isomorphism follow at once.  The finite
statement is the same transport of structure through \(\Val_m\),
with values taken modulo \(F_{m+2}\).
\end{proof}

Thus arithmetic is not an additional primitive operation on words.
It is read from valuation, residue reduction, and normalization.

\subsection{Composition as the primitive}
\label{ssec:sole}

\begin{proposition}[Two-level readout of composition]
\label{thm:two-level}
Let \(A\) be a nonempty set and let \(g\colon A\to A\) be an
endomorphism of infinite order.  Define
\begin{align}
  S_0&=\langle g\rangle
      =\{g^{\circ n}:n\in\NN\}\subseteq\End(A),\label{eq:level0}\\
  S_1&=\{I_n:n\in\NN\}\subseteq\End(\End(A)),
      \qquad I_n(f)=f^{\circ n}.\label{eq:level1}
\end{align}
With \(\iota_0(n)=g^{\circ n}\) and \(\iota_1(n)=I_n\), the maps
\(\iota_0\) and \(\iota_1\) are injective and
\begin{align}
  \iota_0(m+n)&=\iota_0(m)\circ\iota_0(n),
    \label{eq:add-from-comp}\\
  \iota_1(mn)&=\iota_1(m)\circ\iota_1(n).
    \label{eq:mul-from-comp}
\end{align}
\end{proposition}

\begin{proof}
The first identity is
\(g^{\circ m}\circ g^{\circ n}=g^{\circ(m+n)}\); infinite order
also gives injectivity of \(\iota_0\).  The same hypothesis gives
injectivity of \(\iota_1\), since \(I_m(g)=g^{\circ m}\) and
\(I_n(g)=g^{\circ n}\).  For the second identity,
\[
  (I_m\circ I_n)(f)=I_m(f^{\circ n})
    =(f^{\circ n})^{\circ m}=f^{\circ mn}=I_{mn}(f).
\]
\end{proof}

\begin{corollary}[Zeckendorf readout of Church iteration]
\label{cor:minimal}
Under the identification \(X_{\infty}^{\mathrm{fin}}\cong\NN\),
the address \(c_n=\Val^{-1}(n)\) represents the \(n\)-fold iterate.
Addition of addresses is the level-zero composition law, and
multiplication is the level-one composition law.  This is the usual
Church-numeral iteration identity \cite{Church1936}, expressed in
Zeckendorf normal form.
\end{corollary}

\begin{proof}
Apply \Cref{thm:two-level} after replacing \(n\) by its address
\(c_n\).  The semiring identification in \Cref{thm:semiring}
transports the addition and multiplication laws to stable
addresses.
\end{proof}

\begin{theorem}[Level-zero obstruction for multiplicative readout]
\label{thm:level-zero-obstruction}
Let \(M\) be a monoid and let \(g\in M\) have infinite order.  Put
\[
  C=\{g^n:n\in\NN\}\subset M .
\]
There is no injective map
\(\eta\colon\NN_{\ge1}\to C\) satisfying
\[
  \eta(ab)=\eta(a)\eta(b)
  \qquad(a,b\ge1).
\]
Consequently multiplication on the positive integers cannot be
represented by the ambient monoid product on a single free
monogenic orbit, even after arbitrary recoding of the numerals into
that orbit.  In the composition model of
\Cref{thm:two-level}, multiplicative readout therefore requires a
change of level from the orbit \(\{g^{\circ n}\}\) to the power
operators \(f\mapsto f^{\circ n}\), under this level-zero
monogenic-orbit restriction.
\end{theorem}

\begin{proof}
Since \(g\) has infinite order, each element of \(C\) has a unique
form \(g^e\) with \(e\in\NN\).  Write
\[
  \eta(n)=g^{e(n)} .
\]
The displayed multiplicativity condition is equivalent to
\[
  e(ab)=e(a)+e(b)\qquad(a,b\ge1).
\]
Taking \(a=b=1\) gives \(e(1)=0\).  If \(p\) is a prime and
\(e(p)=0\), then \(\eta(p)=\eta(1)\), contradicting injectivity.
Thus injectivity would force \(e(p)>0\) for every prime \(p\).

Choose two distinct primes \(p\) and \(r\), and set
\[
  A=e(p)>0,\qquad B=e(r)>0 .
\]
Then
\[
  e(p^B)=B\,e(p)=AB=A\,e(r)=e(r^A).
\]
Hence \(\eta(p^B)=\eta(r^A)\), while unique factorization gives
\(p^B\ne r^A\).  This contradicts injectivity and proves the
claim.
\end{proof}

\begin{definition}[Protocol universality]\label{def:prot-univ}
A \emph{single-primitive protocol} for a class of operations
\(\mathcal O\) consists of
\begin{enumerate}
\item a syntax generated by one operation \(\sigma\);
\item a computable normalization map \(\mathrm{NF}\) with unique
  normal forms, in the standard sense of canonical rewriting
  \cite{BaaderNipkow1998};
\item a finite family of readout maps
  \(\{\Phi_\ell\}_{\ell=0}^L\) through which the operations in
  \(\mathcal O\) are realized from \(\sigma\)-compositions.
\end{enumerate}
It is \emph{protocol-universal for \(\mathcal O\)} if all
operations in \(\mathcal O\) are realized in this way and equality
of normal forms is decidable.
\end{definition}

The Zeckendorf address system with composition readouts is
protocol-universal for \(\mathcal O=\{+,\times\}\) on \(\NN\).
The normal forms are computable by \Cref{prop:fold-confluence}, and
the readouts are those of \Cref{thm:two-level,cor:minimal}.
