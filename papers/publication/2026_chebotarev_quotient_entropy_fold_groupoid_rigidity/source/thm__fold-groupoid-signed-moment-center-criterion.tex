\begin{proposition}\label{prop:fold-chi-layered-fiber-spectrum-certified-by-signed-powersums}
Fix \(m\), and write \(N_m:=|X_m|\). For \(s,t\in\{0,1\}\) and \(q=0,1,\dots,N_m-1\), define
\[
M_{s,t}(q)
:=
2^m\,\tau_{m,1}\!\left(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\right)
=
\sum_{x\in X_m}\chi_1(x)^s\chi_2(x)^t\,d_m(x)^{q+1},
\]
where \(\chi(x)=(\chi_1(x),\chi_2(x))\in\{\pm1\}^2\). Then, for every \((\alpha,\beta)\in\{\pm1\}^2\), the polynomial
\[
Q_m^{\alpha,\beta}(z):=
\prod_{\substack{x\in X_m\\ \chi(x)=(\alpha,\beta)}}(1-d_m(x)z)\in\ZZ[z]
\]
is determined by the finite family
\[
\{M_{s,t}(q):(s,t)\in\{0,1\}^2,\ 0\le q\le N_m-1\}.
\]
More precisely, the coefficients of \(Q_m^{\alpha,\beta}\) are recovered constructively from these moments by the Newton identities.
\end{proposition}

\begin{proof}
Fix \((\alpha,\beta)\in\{\pm1\}^2\). For each \(k=1,\dots,N_m\), Fourier inversion on \(\{\pm1\}^2\) gives
\[
p_k^{\alpha,\beta}
:=
\sum_{\chi(x)=(\alpha,\beta)}d_m(x)^k
=
\frac14\sum_{s,t\in\{0,1\}}\alpha^s\beta^t\,M_{s,t}(k-1).
\]
Now enlarge the multiset \(\{d_m(x):\chi(x)=(\alpha,\beta)\}\) by adjoining zeros until it has cardinality \(N_m\). The power sums remain \((p_k^{\alpha,\beta})_{1\le k\le N_m}\). Newton's identities therefore recover the monic polynomial whose roots are precisely these numbers from the first \(N_m\) power sums. Deleting the zero roots leaves exactly \(Q_m^{\alpha,\beta}(z)\).
\end{proof}

\begin{proposition}\label{prop:fold-center-resolution-criterion-indw-eps}
Write \(Z_m:=Z(A_{m,1})\), and set
\[
Z_m^{(3)}:=\CC\!\left[\mathrm{IndW}(E_{m,1}),\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}}\right]\subseteq Z_m.
\]
Then:
\begin{enumerate}
\normalfont
  \item
  \[
  Z_m^{(3)}
  =
  \bigoplus_{(\alpha,\beta)\in\{\pm1\}^2}\CC\!\left[\mathrm{IndW}(E_{m,1})\right]\,e_{\alpha,\beta}.
  \]
  \item If
  \[
  V_{\alpha,\beta}:=\{d_m(x):\chi(x)=(\alpha,\beta)\}\subset\NN,
  \]
  then
  \[
  \dim_{\CC}Z_m^{(3)}=\sum_{(\alpha,\beta)\in\{\pm1\}^2}|V_{\alpha,\beta}|.
  \]
  \item Consequently, \(Z_m^{(3)}=Z_m\) if and only if, for every \((\alpha,\beta)\), the map
  \[
  x\longmapsto d_m(x)
  \qquad\text{on}\qquad
  X_m^{\alpha,\beta}:=\{x\in X_m:\chi(x)=(\alpha,\beta)\}
  \]
  is injective.
\end{enumerate}
\end{proposition}

\begin{proof}
Under the Wedderburn identification, \(Z_m\cong\CC^{X_m}\) is spanned by the minimal central projections \(\widetilde P_x\), and
\[
\mathrm{IndW}(E_{m,1})=\sum_x d_m(x)\,\widetilde P_x,
\qquad
\varepsilon_{\mathrm{scan}}=\sum_x \chi_1(x)\,\widetilde P_x,
\qquad
\varepsilon_{\mathrm{rev}}=\sum_x \chi_2(x)\,\widetilde P_x.
\]
It follows that, on the block indexed by \(x\), an element of \(Z_m^{(3)}\) depends only on the pair \((\chi(x),d_m(x))\); inside a fixed sector it is therefore a polynomial function of \(d_m(x)\). Multiplication by \(e_{\alpha,\beta}\) isolates the sector \((\alpha,\beta)\), which gives (1).

For (2), fix \((\alpha,\beta)\). On the finite set \(X_m^{\alpha,\beta}\), the image of the polynomial algebra in the variable \(d_m\) has dimension equal to the number of distinct values taken by \(d_m\): if these values are \(v_1,\dots,v_r\), Lagrange interpolation gives polynomials that are \(1\) at one \(v_i\) and \(0\) at the others. Summing over the four sectors yields the claimed formula for \(\dim_{\CC}Z_m^{(3)}\).

Finally,
\[
\dim_{\CC}Z_m=\sum_{(\alpha,\beta)\in\{\pm1\}^2}\abs{X_m^{\alpha,\beta}}.
\]
Hence \(Z_m^{(3)}=Z_m\) if and only if \(\dim_{\CC}Z_m^{(3)}=\dim_{\CC}Z_m\). By (2), this is equivalent to
\[
|V_{\alpha,\beta}|=\abs{X_m^{\alpha,\beta}}
\qquad\text{for all }(\alpha,\beta)\in\{\pm1\}^2,
\]
which is exactly the injectivity of \(x\mapsto d_m(x)\) on each sector \(X_m^{\alpha,\beta}\).
\end{proof}
