\begin{proposition}\label{prop:fold-chi-layered-fiber-spectrum-certified-by-signed-powersums}
Fix \(m\), and write \(N_m:=|X_m|\). For \(s,t\in\{0,1\}\) and \(q=0,1,\dots,N_m-1\), define
\[
M_{s,t}(q)
:=
2^m\,\tau_{m,1}\!\left(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\right)
=
\sum_{x\in X_m}\chi_1(x)^s\chi_2(x)^t\,d_m(x)^{q+1},
\]
where \(\chi(x)=(\chi_1(x),\chi_2(x))\in\{\pm1\}^2\). Then, for each \((\alpha,\beta)\in\{\pm1\}^2\), the polynomial
\[
Q_m^{\alpha,\beta}(z):=
\prod_{\substack{x\in X_m\\ \chi(x)=(\alpha,\beta)}}(1-d_m(x)z)\in\ZZ[z]
\]
is uniquely determined by the finite family
\[
\{M_{s,t}(q):(s,t)\in\{0,1\}^2,\ 0\le q\le N_m-1\}.
\]
The determination is constructive, via the Newton identities.
\end{proposition}

\begin{proof}
For each \((\alpha,\beta)\in\{\pm1\}^2\) and each \(k=1,\dots,N_m\), character orthogonality on \(\{\pm1\}^2\) gives
\[
p_k^{\alpha,\beta}
:=
\sum_{\chi(x)=(\alpha,\beta)}d_m(x)^k
=
\frac14\sum_{s,t\in\{0,1\}}\alpha^s\beta^t\,M_{s,t}(k-1).
\]
If one pads the multiset \(\{d_m(x):\chi(x)=(\alpha,\beta)\}\) with zeros up to length \(N_m\), its power sums are still exactly \((p_k^{\alpha,\beta})_{1\le k\le N_m}\). The Newton identities recover the coefficients of the monic polynomial with these roots from the first \(N_m\) power sums. Since the zero roots contribute no nontrivial linear factor, this determines \(Q_m^{\alpha,\beta}(z)\).
\end{proof}

\begin{theorem}\label{thm:fold-window6-chi-layered-spectrum-reconstruction-by-12-moments}
Take \(m=6\). By Assumption \ref{cor:fold-bin-m6-fiber-hist},
\[
\begin{aligned}
d_6(x)&\in\{2,3,4\},\\
\#\{x\in X_6:d_6(x)=2\}&=8,\\
\#\{x\in X_6:d_6(x)=3\}&=4,\\
\#\{x\in X_6:d_6(x)=4\}&=9.
\end{aligned}
\]
and \(|X_6|=21\).
Write
\[
\chi(x)=(\chi_1(x),\chi_2(x))\in\{\pm1\}^2.
\]
For \(s,t\in\{0,1\}\) and \(q\ge 0\), define
\[
M_{s,t}(q)
:=
2^6\,\tau_{6,1}\!\left(
\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,
\mathrm{IndW}(E_{6,1})^{\,q}
\right)
=
\sum_{x\in X_6}\chi_1(x)^s\chi_2(x)^t\,d_6(x)^{q+1}.
\]
For each \((\alpha,\beta)\in\{\pm1\}^2\) and \(j\in\{2,3,4\}\), let
\[
n_j^{\alpha,\beta}
:=
\#\{x\in X_6:\chi(x)=(\alpha,\beta),\ d_6(x)=j\}.
\]
Then the twelve numbers
\[
\{M_{s,t}(q):(s,t)\in\{0,1\}^2,\ q=0,1,2\}
\]
determine all twelve integers \(n_j^{\alpha,\beta}\). More precisely, if
\[
p_k^{\alpha,\beta}
:=
\sum_{\chi(x)=(\alpha,\beta)}d_6(x)^k
=
\frac14\sum_{s,t\in\{0,1\}}\alpha^s\beta^t\,M_{s,t}(k-1),
\qquad k=1,2,3,
\]
then
\[
n_2^{\alpha,\beta}
=
3p_1^{\alpha,\beta}-\frac74p_2^{\alpha,\beta}+\frac14p_3^{\alpha,\beta},
\]
\[
n_3^{\alpha,\beta}
=
-\frac83p_1^{\alpha,\beta}+2p_2^{\alpha,\beta}-\frac13p_3^{\alpha,\beta},
\]
\[
n_4^{\alpha,\beta}
=
\frac34p_1^{\alpha,\beta}-\frac58p_2^{\alpha,\beta}+\frac18p_3^{\alpha,\beta}.
\]
Consequently,
\[
Q_6^{\alpha,\beta}(z)
:=
\prod_{\chi(x)=(\alpha,\beta)}(1-d_6(x)z)
=
(1-2z)^{n_2^{\alpha,\beta}}(1-3z)^{n_3^{\alpha,\beta}}(1-4z)^{n_4^{\alpha,\beta}}.
\]
\end{theorem}

\begin{proof}
By Theorem \ref{thm:fold-groupoid-z2x2-joint-spectral-measure}, the quantities \(M_{s,t}(q)\) are the signed power sums of the \(\chi\)-layered fiber spectrum. Fourier-Hadamard inversion gives
\[
p_k^{\alpha,\beta}
=
\frac14\sum_{s,t\in\{0,1\}}\alpha^s\beta^t\,M_{s,t}(k-1),
\qquad k=1,2,3.
\]

Since the only fiber sizes are \(2,3,4\), the multiplicities \(n_2^{\alpha,\beta}\), \(n_3^{\alpha,\beta}\), and \(n_4^{\alpha,\beta}\) satisfy
\[
p_1^{\alpha,\beta}=2n_2^{\alpha,\beta}+3n_3^{\alpha,\beta}+4n_4^{\alpha,\beta},
\]
\[
p_2^{\alpha,\beta}=4n_2^{\alpha,\beta}+9n_3^{\alpha,\beta}+16n_4^{\alpha,\beta},
\]
\[
p_3^{\alpha,\beta}=8n_2^{\alpha,\beta}+27n_3^{\alpha,\beta}+64n_4^{\alpha,\beta}.
\]
The coefficient matrix
\[
\begin{pmatrix}
2&3&4\\
4&9&16\\
8&27&64
\end{pmatrix}
\]
has determinant \(48\neq 0\), so the system is invertible, and direct inversion gives the displayed formulas.
\end{proof}

\begin{proposition}\label{prop:fold-center-resolution-criterion-indw-eps}
Write \(Z_m:=Z(A_{m,1})\), and define
\[
Z_m^{(3)}:=\CC\!\left[\mathrm{IndW}(E_{m,1}),\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}}\right]\subseteq Z_m.
\]
Then:
\begin{enumerate}
\normalfont
  \item
  \[
  Z_m^{(3)}
  =
  \bigoplus_{(\alpha,\beta)\in\{\pm1\}^2}\CC\!\left[\mathrm{IndW}(E_{m,1})\right]\,e_{\alpha,\beta}.
  \]
  \item If
  \[
  V_{\alpha,\beta}:=\{d_m(x):\chi(x)=(\alpha,\beta)\}\subset\NN,
  \]
  then
  \[
  \dim_{\CC}Z_m^{(3)}=\sum_{(\alpha,\beta)\in\{\pm1\}^2}|V_{\alpha,\beta}|.
  \]
  \item Consequently, \(Z_m^{(3)}=Z_m\) if and only if, for every \((\alpha,\beta)\), the map
  \[
  x\longmapsto d_m(x)
  \qquad\text{on}\qquad
  X_m^{\alpha,\beta}:=\{x\in X_m:\chi(x)=(\alpha,\beta)\}
  \]
  is injective.
\end{enumerate}
\end{proposition}

\begin{proof}
Under the Wedderburn identification, \(Z_m\cong\CC^{X_m}\) is spanned by the minimal central projections \(\widetilde P_x\), and
\[
\mathrm{IndW}(E_{m,1})=\sum_x d_m(x)\,\widetilde P_x,
\qquad
\varepsilon_{\mathrm{scan}}=\sum_x \chi_1(x)\,\widetilde P_x,
\qquad
\varepsilon_{\mathrm{rev}}=\sum_x \chi_2(x)\,\widetilde P_x.
\]
Thus every element of \(Z_m^{(3)}\) takes on the block \(x\) a value determined only by \(\chi(x)\) and \(d_m(x)\), and within a fixed sector it is a polynomial function of \(d_m\). Multiplying by \(e_{\alpha,\beta}\) restricts to the sector \((\alpha,\beta)\), proving (1).

On a finite set, the image of the polynomial algebra in one variable has dimension equal to the number of distinct values taken by that variable. This gives (2). Since \(\dim_{\CC}Z_m=|X_m|\), statement (3) follows immediately.
\end{proof}

\begin{theorem}\label{thm:fold-window6-center-three-observables-dimension-defect}
Take \(m=6\), and write
\[
A_{6,1}:=\CC[\mathcal R_6],
\qquad
Z_6:=Z(A_{6,1}),
\qquad
Z_6^{(3)}:=\CC\!\left[\mathrm{IndW}(E_{6,1}),\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}}\right]\subseteq Z_6.
\]
Then
\[
\dim_{\CC}Z_6^{(3)}\le 12.
\]
Since \(|X_6|=21\), one therefore has
\[
\dim_{\CC}Z_6=21,
\qquad
\dim_{\CC}(Z_6/Z_6^{(3)})\ge 9.
\]
Equivalently, the three central observables \(\mathrm{IndW}(E_{6,1})\), \(\varepsilon_{\mathrm{scan}}\), and \(\varepsilon_{\mathrm{rev}}\) never suffice to distinguish all \(21\) stable types in window \(6\).

Moreover,
\[
\dim_{\CC}Z_6^{(3)}=12
\iff
V_{\alpha,\beta}=\{2,3,4\}
\quad\text{for every }(\alpha,\beta)\in\{\pm1\}^2.
\]
\end{theorem}

\begin{proof}
By Assumption \ref{cor:fold-bin-m6-fiber-hist}, the possible fiber multiplicities in window \(6\) are exactly \(2,3,4\). Hence
\[
V_{\alpha,\beta}\subseteq\{2,3,4\},
\qquad
|V_{\alpha,\beta}|\le 3
\]
for every sector \((\alpha,\beta)\). Proposition \ref{prop:fold-center-resolution-criterion-indw-eps} therefore gives
\[
\dim_{\CC}Z_6^{(3)}
=
\sum_{(\alpha,\beta)\in\{\pm1\}^2}|V_{\alpha,\beta}|
\le 12.
\]

On the other hand, the Wedderburn decomposition
\[
A_{6,1}\cong \bigoplus_{x\in X_6}M_{d_6(x)}(\CC)
\]
shows that \(A_{6,1}\) has \(|X_6|=21\) simple blocks, and therefore \(\dim_{\CC}Z_6=21\). The lower bound on \(\dim_{\CC}(Z_6/Z_6^{(3)})\) follows at once.

Finally, equality \(\dim_{\CC}Z_6^{(3)}=12\) holds exactly when \(|V_{\alpha,\beta}|=3\) in every sector, which is equivalent to the condition \(V_{\alpha,\beta}=\{2,3,4\}\) for all \((\alpha,\beta)\).
\end{proof}
