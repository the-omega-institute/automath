\begin{theorem}\label{thm:fold-groupoid-z2x2-central-idempotents}
Let \(A_{m,1}=\CC[\mathcal{R}_m]\) be the fold-groupoid algebra from Definition \ref{def:fold-equivalence-groupoid-algebra}. By Proposition \ref{prop:fold-groupoid-wedderburn},
\[
A_{m,1}\cong \bigoplus_{x\in X_m}M_{d_m(x)}(\CC)
\]
after choosing a Wedderburn identification.
For each \(x\in X_m\), let \(\widetilde P_x\in Z(A_{m,1})\) be the minimal central idempotent corresponding to the simple block \(M_{d_m(x)}(\CC)\), so that
\(
\widetilde P_x A_{m,1}\cong M_{d_m(x)}(\CC)
\).
Let
\(
\chi(x)=(\operatorname{sgn}(c_x),\operatorname{sgn}(\iota_x))\in\{\pm1\}^2
\)
be the scan-reversal fingerprint of Proposition \ref{prop:fold-groupoid-z2x2-sector-by-orientation}.

Define central self-adjoint unitaries
\[
\varepsilon_{\mathrm{scan}}:=\sum_{x\in X_m}\operatorname{sgn}(c_x)\,\widetilde P_x,
\qquad
\varepsilon_{\mathrm{rev}}:=\sum_{x\in X_m}\operatorname{sgn}(\iota_x)\,\widetilde P_x.
\]
Then \(\varepsilon_{\mathrm{scan}}^2=\varepsilon_{\mathrm{rev}}^2=1\) and \(\varepsilon_{\mathrm{scan}}\varepsilon_{\mathrm{rev}}=\varepsilon_{\mathrm{rev}}\varepsilon_{\mathrm{scan}}\).
For any \((\alpha,\beta)\in\{\pm1\}^2\), define
\[
e_{\alpha,\beta}
:=
\frac14\bigl(1+\alpha\,\varepsilon_{\mathrm{scan}}\bigr)\bigl(1+\beta\,\varepsilon_{\mathrm{rev}}\bigr)
\ \in\ Z(A_{m,1}).
\]
Then \(\{e_{\alpha,\beta}\}_{(\alpha,\beta)\in\{\pm1\}^2}\) is a complete family of pairwise orthogonal central idempotents summing to \(1\), and the associated sector ideals satisfy
\[
e_{\alpha,\beta}
=
\sum_{\chi(x)=(\alpha,\beta)}\widetilde P_x,
\]
\[
e_{\alpha,\beta}A_{m,1}
\ \cong\
\bigoplus_{\chi(x)=(\alpha,\beta)}M_{d_m(x)}(\CC).
\]
Equivalently,
\[
\widetilde P_x\le e_{\alpha,\beta}
\quad\Longleftrightarrow\quad
\chi(x)=(\alpha,\beta).
\]
Thus the \(\mathbb Z_2^2\)-sector decomposition is realized internally by the joint spectral decomposition of \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\).
Moreover, the observable subalgebra generated by these two involutions,
\[
\mathcal A_\chi:=C^\ast(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}}),
\]
is the commutative semisimple \(C^\ast\)-algebra
\[
\mathcal A_\chi=\bigoplus_{X_m^{\alpha,\beta}\neq\varnothing}\CC\,e_{\alpha,\beta}.
\]
In particular, \(\mathcal A_\chi\) is a quotient of \(\CC[\mathbb Z_2^2]\); if every sector is nonempty, then \(\mathcal A_\chi\cong \CC[\mathbb Z_2^2]\cong \CC^4\).
\end{theorem}

\begin{proof}
By the Wedderburn decomposition, the center \(Z(A_{m,1})\) is linearly spanned by \(\{\widetilde P_x\}\), and
\[
\widetilde P_x\widetilde P_y=\delta_{xy}\widetilde P_x,
\qquad
\sum_x\widetilde P_x=1.
\]
Hence \(\varepsilon_{\mathrm{scan}}\) and \(\varepsilon_{\mathrm{rev}}\) are central, and on the block \(M_{d_m(x)}(\CC)\) they act by the scalars \(\operatorname{sgn}(c_x)\) and \(\operatorname{sgn}(\iota_x)\), respectively. In particular, they are commuting self-adjoint involutions.

Since they commute, the standard projection formula shows that \(e_{\alpha,\beta}\) is a central idempotent, that distinct \(e_{\alpha,\beta}\) are orthogonal, and that \(\sum_{\alpha,\beta}e_{\alpha,\beta}=1\). On the \(x\)-block one has
\[
e_{\alpha,\beta}\Big|_x
=
\frac14\bigl(1+\alpha\,\operatorname{sgn}(c_x)\bigr)\bigl(1+\beta\,\operatorname{sgn}(\iota_x)\bigr),
\]
and this scalar is \(1\) exactly when \(\chi(x)=(\alpha,\beta)\), and \(0\) otherwise.
Hence
\[
e_{\alpha,\beta}=\sum_{\chi(x)=(\alpha,\beta)}\widetilde P_x,
\]
so \(e_{\alpha,\beta}\) acts as the identity on the blocks with fingerprint \((\alpha,\beta)\) and vanishes on all others. Hence
\[
e_{\alpha,\beta}A_{m,1}
\cong
\bigoplus_{\chi(x)=(\alpha,\beta)}M_{d_m(x)}(\CC).
\]

Finally, \(\mathcal A_\chi\) is generated by two commuting involutions, hence is a commutative semisimple finite-dimensional \(C^\ast\)-algebra. The preceding block computation shows that its minimal central idempotents are precisely the nonzero \(e_{\alpha,\beta}\), and therefore
\[
\mathcal A_\chi=\bigoplus_{X_m^{\alpha,\beta}\neq\varnothing}\CC\,e_{\alpha,\beta}.
\]
Thus \(\mathcal A_\chi\) is a quotient of \(\CC[\mathbb Z_2^2]\), and it is exactly \(\CC^4\) when all four sectors occur.
\end{proof}

\begin{theorem}\label{thm:fold-groupoid-z2x2-joint-spectral-measure}
Under a fixed Wedderburn identification
\[
A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC),
\]
equip \(A_{m,1}\) with the normalized trace
\[
\tau_{m,1}(A):=2^{-m}\sum_{x\in X_m}\Tr(A_x),
\]
where \(A_x\in M_{d_m(x)}(\CC)\) is the \(x\)-block component.
Then, for every minimal central projection \(\widetilde P_x\),
\[
\tau_{m,1}(\widetilde P_x)=\frac{d_m(x)}{2^m}=: \pi_m(x),
\]
so \(\pi_m\) is the pushforward of the uniform measure on \(\Omega_m\) under \(\Fold_m\).

Let \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\) be as in Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}. Then for every function \(f:\{\pm1\}^2\to\CC\),
\[
\tau_{m,1}\!\bigl(f(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\bigr)
=\sum_{x\in X_m}\pi_m(x)\,f\!\bigl(\chi(x)\bigr).
\]
Hence, with respect to \(\tau_{m,1}\), the joint spectral measure of \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\) is \(\chi_\ast\pi_m\).

Furthermore, let
\[
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m} d_m(x)^2\,\widetilde P_x\ \in\ Z(A_{m,1})
\]
be the Watatani index element of the block-depolarizing conditional expectation \(E_{m,1}\) from Proposition \ref{prop:fold-groupoid-block-expectation}.
Then for every integer \(q\ge 0\) and every \(s,t\in\{0,1\}\),
\[
\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\frac{1}{2^m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^{s}\operatorname{sgn}(\iota_x)^{t}\,d_m(x)^{2q+1}.
\]
\end{theorem}

\begin{proof}
Since \(\sum_{x\in X_m}d_m(x)=|\Omega_m|=2^m\), the functional \(\tau_{m,1}\) is a normalized trace on \(A_{m,1}\). Because \(\varepsilon_{\mathrm{scan}}\) and \(\varepsilon_{\mathrm{rev}}\) act by the scalars \(\chi(x)\) on block \(x\), we may write
\[
f(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})
=\sum_{x\in X_m} f(\chi(x))\,\widetilde P_x.
\]
Taking traces and using
\[
\tau_{m,1}(\widetilde P_x)=2^{-m}\Tr(I_{d_m(x)})=\frac{d_m(x)}{2^m}
\]
gives the first identity.

For the second formula, note that \(\mathrm{IndW}(E_{m,1})\) acts on the \(x\)-block by the scalar \(d_m(x)^2\). Therefore
\[
\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}
=\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{2q}\,\widetilde P_x.
\]
Applying the normalized trace once more yields
\[
2^{-m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{2q}\,\Tr(I_{d_m(x)})
=\frac{1}{2^m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{2q+1}.
\]
\end{proof}

\begin{proposition}\label{prop:fold-groupoid-maxblock-chi-homogeneity}
Let \(\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=D_m\}\), and define the maximal-block ideal
\[
I_m:=\bigoplus_{x\in\mathcal{M}_m}M_{D_m}(\CC)\ \subset\ A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC).
\]
Let \(\chi_m\in\{\pm1\}^2\) be the constant fingerprint on maximal fibers supplied by Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}, and let \(e_{\chi_m}:=e_{\alpha,\beta}\), where \(\chi_m=(\alpha,\beta)\), be the corresponding central idempotent from Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}. Then
\[
I_m\subseteq e_{\chi_m}A_{m,1}.
\]
In particular, the Wedderburn decomposition of the quotient algebra \(A_{m,1}/e_{\chi_m}A_{m,1}\) contains no matrix block of dimension \(D_m\).
\end{proposition}
\begin{proof}
By Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}, every \(x\in\mathcal M_m\) satisfies \(\chi(x)=\chi_m\).
By Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}, the central projection \(e_{\chi_m}\) acts by \(1\) on block \(x\) if and only if \(\chi(x)=\chi_m\), and by \(0\) otherwise.
Hence \(\widetilde P_x\le e_{\chi_m}\) for all \(x\in \mathcal M_m\), so
\[
M_{D_m}(\CC)\cong \widetilde P_xA_{m,1}\subseteq e_{\chi_m}A_{m,1}.
\]
Taking the direct sum over all \(x\in\mathcal M_m\) gives \(I_m\subseteq e_{\chi_m}A_{m,1}\). Hence every block of size \(D_m\) is contained in the ideal being quotiented out, so no \(D_m\)-dimensional matrix block survives in the Wedderburn decomposition of \(A_{m,1}/e_{\chi_m}A_{m,1}\).
\end{proof}

\begin{theorem}\label{thm:fold-max-data-processing-saturation-chi}
Let \(E_m\) be the conditional expectation induced by the fold map \(\Fold_m:\Omega_m\to X_m\), that is, uniform averaging along fibers. For a probability distribution \(\rho\) on \(\Omega_m\), let \(\rho^\flat\) be the associated fiberwise-uniform distribution from Proposition \ref{prop:fold-condexp-index-maxfiber}. Define
\[
\Delta_m:=\sup_{\rho}D(\rho\Vert\rho^\flat),
\]
where the supremum runs over all probability distributions on \(\Omega_m\).
Then the following assertions hold.
\begin{enumerate}
\normalfont
  \item \(\Delta_m=\log D_m\) by Proposition \ref{prop:fold-condexp-index-maxfiber}.
  \item If \(\rho\) attains the supremum, then the support of \((\Fold_m)_\ast\rho\) is contained in \(\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=D_m\}\); equivalently,
  \[
  (\Fold_m)_\ast\rho(x)>0\ \Longrightarrow\ d_m(x)=D_m.
  \]
  \item Consequently, for any maximizing distribution \(\rho\), every macroscopic support point \(x^\star\in \mathrm{supp}((\Fold_m)_\ast\rho)\) satisfies
  \[
  \chi(x^\star)=\chi_m,
  \]
  where \(\chi_m\) is the constant maximal-fiber fingerprint from Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}. In particular, when \(m\) is odd and lies in the stable regime, one has \(\operatorname{sgn}(\iota_{x^\star})=+1\).
\end{enumerate}
\end{theorem}
\begin{proof}
The first statement is exactly Proposition \ref{prop:fold-condexp-index-maxfiber}.

For the second statement, let \(\rho_X:=(\Fold_m)_\ast\rho\), and for each \(x\in X_m\) let \(u_x\) denote the uniform distribution on \(\Fold_m^{-1}(x)\). The chain rule for \(\rho^\flat\) gives
\[
D(\rho\Vert\rho^\flat)
=\sum_{x\in X_m}\rho_X(x)\,D\!\bigl(\rho(\cdot\mid x)\,\Vert\,u_x\bigr).
\]
For each \(x\), the conditional divergence is bounded by
\[
D\!\bigl(\rho(\cdot\mid x)\Vert u_x\bigr)\le \log d_m(x)\le \log D_m.
\]
If \(\rho\) attains the value \(\Delta_m=\log D_m\), then this weighted average of quantities bounded above by \(\log D_m\) must itself be equal to \(\log D_m\). Therefore every \(x\) with \(\rho_X(x)>0\) satisfies \(\log d_m(x)=\log D_m\), so \(\rho_X\) is supported on \(\mathcal M_m\).

The third statement follows immediately from the second and Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}: any \(x^\star\in\mathrm{supp}(\rho_X)\subseteq\mathcal M_m\) satisfies \(\chi(x^\star)=\chi_m\).
The assertion about the second component in the odd stable regime is exactly the final clause of Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}.
\end{proof}

\begin{corollary}\label{cor:fold-indw-high-q-extract-maxfiber-and-chi}
Fix a resolution \(m\), and let
\[
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m}d_m(x)^2\,\widetilde P_x\ \in\ Z(A_{m,1})
\]
be as in Theorem \ref{thm:fold-groupoid-z2x2-joint-spectral-measure}.
Write
\[
\begin{aligned}
D_m&:=\max_{x\in X_m}d_m(x),\\
\mathcal M_m&:=\{x\in X_m:\ d_m(x)=D_m\},\\
\chi_m&=(\alpha_m,\beta_m)\in\{\pm1\}^2.
\end{aligned}
\]
Here \(\chi_m\) is furnished by Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}, so that \(\chi(x)=\chi_m\) for every \(x\in\mathcal M_m\).
Then the following assertions hold.
\begin{enumerate}
\normalfont
  \item The maximal fiber size is recovered by the limit
  \[
  D_m
  =
  \lim_{q\to\infty}\Bigl(2^m\,\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)\Bigr)^{1/(2q+1)}.
  \]
  \item The fingerprint \(\chi_m\) is recovered from signed moment ratios: for every \(s,t\in\{0,1\}\),
  \[
  \lim_{q\to\infty}
  \frac{\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)}
  {\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)}
  =
  \alpha_m^{\,s}\beta_m^{\,t}.
  \]
  \item For any \((\alpha,\beta)\in\{\pm1\}^2\), define the sectorwise maximal fiber size by
  \[
  D_m^{\alpha,\beta}
  :=
  \max\{d_m(x):\chi(x)=(\alpha,\beta)\}.
  \]
  If the sector is empty, we set \(D_m^{\alpha,\beta}:=0\).
  Then
  \[
  D_m^{\alpha,\beta}
  =
  \lim_{q\to\infty}\Bigl(2^m\,\tau_{m,1}\!\bigl(e_{\alpha,\beta}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)\Bigr)^{1/(2q+1)}.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
Since \(\widetilde P_x\widetilde P_y=\delta_{xy}\widetilde P_x\) and \(\sum_x\widetilde P_x=1\),
\[
\mathrm{IndW}(E_{m,1})^{\,q}=\sum_{x\in X_m} d_m(x)^{2q}\,\widetilde P_x,
\]
and therefore
\[
2^m\,\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m}d_m(x)^{2q+1}.
\]
For a finite set of positive numbers, the limit of the \((2q+1)\)-st root of the power sum is the maximum, so
\[
\lim_{q\to\infty}\Bigl(\sum_x d_m(x)^{2q+1}\Bigr)^{1/(2q+1)}=D_m.
\]
This proves (1).

For the second statement, Theorem \ref{thm:fold-groupoid-z2x2-joint-spectral-measure} gives
\[
2^m\,\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{2q+1}.
\]
After dividing numerator and denominator by \(D_m^{2q+1}\), all terms with \(d_m(x)<D_m\) vanish as \(q\to\infty\), and only the contribution from \(\mathcal M_m\) remains.
More explicitly, the denominator tends to \(|\mathcal M_m|\), while the numerator tends \(\alpha_m^s\beta_m^t|\mathcal M_m|\) because \(\chi\) is constant on \(\mathcal M_m\) with value \(\chi_m=(\alpha_m,\beta_m)\). Hence the limit is \(\alpha_m^s\beta_m^t\).

For the third statement, the block-projection property from Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents} gives
\[
e_{\alpha,\beta}\widetilde P_x=\mathbf{1}_{\{\chi(x)=(\alpha,\beta)\}}\,\widetilde P_x.
\]
Hence
\[
2^m\,\tau_{m,1}\!\bigl(e_{\alpha,\beta}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{\chi(x)=(\alpha,\beta)} d_m(x)^{2q+1},
\]
and letting \(q\to\infty\) gives \(D_m^{\alpha,\beta}\), with value \(0\) when the sector is empty.
\end{proof}

\begin{theorem}\label{thm:fold-window6-chi-layered-spectrum-reconstruction-by-12-moments}
Take \(m=6\). With the notation of Assumption \ref{cor:fold-bin-m6-fiber-hist},
\[
\begin{aligned}
d_6(x)&\in\{2,3,4\},\\
\#\{x\in X_6:\ d_6(x)=2\}&=8,\\
\#\{x\in X_6:\ d_6(x)=3\}&=4,\\
\#\{x\in X_6:\ d_6(x)=4\}&=9.
\end{aligned}

These data imply \(|X_6|=21\).
Write
\[
\chi(x)=(\chi_1(x),\chi_2(x))\in\{\pm1\}^2,

and for \(s,t\in\{0,1\}\) and \(q\ge 0\), define
\[
M_{s,t}(q)
:=
2^6\,\tau_{6,1}\!\left(
\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,
\mathrm{IndW}(E_{6,1})^{\,q}
\right)
=
\sum_{x\in X_6}\chi_1(x)^s\chi_2(x)^t\,d_6(x)^{2q+1}.
\]
For each \((\alpha,\beta)\in\{\pm1\}^2\) and each \(j\in\{2,3,4\}\), let
\[
n_j^{\alpha,\beta}
:=
\#\{x\in X_6:\ \chi(x)=(\alpha,\beta),\ d_6(x)=j\}.

Then the twelve scalars
\[
\{M_{s,t}(q):\ (s,t)\in\{0,1\}^2,\ q=0,1,2\}
\]
uniquely recover all twelve integers \(n_j^{\alpha,\beta}\).
More explicitly, if one defines
\[
p_{2r+1}^{\alpha,\beta}
:=
\sum_{\chi(x)=(\alpha,\beta)} d_6(x)^{2r+1}

then the explicit inversion formulas are

n_2^{\alpha,\beta}

\[
n_3^{\alpha,\beta}
=
-\frac{64}{105}p_1^{\alpha,\beta}+\frac{4}{21}p_3^{\alpha,\beta}-\frac{1}{105}p_5^{\alpha,\beta},

\[
n_4^{\alpha,\beta}
=
\frac{3}{28}p_1^{\alpha,\beta}-\frac{13}{336}p_3^{\alpha,\beta}+\frac{1}{336}p_5^{\alpha,\beta}.

In particular, for admissible moment data the right-hand sides are automatically integers.
Consequently, the fiber-spectrum polynomial of each sector is explicitly

Q_6^{\alpha,\beta}(z)
:=
\prod_{\chi(x)=(\alpha,\beta)}(1-d_6(x)z)
=
(1-2z)^{n_2^{\alpha,\beta}}(1-3z)^{n_3^{\alpha,\beta}}(1-4z)^{n_4^{\alpha,\beta}}.

\begin{proof}
The proof is a direct computation: one first isolates the sectorwise odd moments by Fourier-Hadamard inversion, and then inverts the resulting \(3\times 3\) system coming from the possible fiber sizes \(2\), \(3\), and \(4\). We record the details in Appendix \ref{app:window6}.

\begin{theorem}\label{thm:fold-window6-center-three-observables-dimension-defect}
Take \(m=6\), and write
\[
A_{6,1}:=\CC[\mathcal R_6],
\qquad
Z_6:=Z(A_{6,1}),
\]
\[
Z_6^{(3)}
:=
\CC\!\left[
\mathrm{IndW}(E_{6,1}),
\varepsilon_{\mathrm{scan}},
\varepsilon_{\mathrm{rev}}
\right]
\subseteq Z_6.
\]
Then one necessarily has the strict upper bound
\[
\dim_{\CC}Z_6^{(3)}\le 12.
\]
On the other hand, since \(|X_6|=21\),
\[
\dim_{\CC}Z_6=21,
\qquad
\dim_{\CC}(Z_6/Z_6^{(3)})\ge 9.
\]
In particular, the three central observables \(\mathrm{IndW}(E_{6,1})\), \(\varepsilon_{\mathrm{scan}}\), and \(\varepsilon_{\mathrm{rev}}\) do not separate all \(21\) minimal central projections in window \(6\); equivalently, the undetected central quotient has dimension at least \(9\).

Moreover, the following are equivalent:
\[
\dim_{\CC}Z_6^{(3)}=12
\iff
V_{\alpha,\beta}=\{2,3,4\}
\quad\text{for every}\quad
(\alpha,\beta)\in\{\pm1\}^2.
\]
\end{theorem}
\begin{proof}
By Assumption \ref{cor:fold-bin-m6-fiber-hist}, the fiber multiplicities in window \(6\) take only the values \(2,3,4\). Hence for every \((\alpha,\beta)\in\{\pm1\}^2\),
\[
V_{\alpha,\beta}\subseteq \{2,3,4\},
\qquad
|V_{\alpha,\beta}|\le 3.
\]
Proposition \ref{prop:fold-center-resolution-criterion-indw-eps} therefore gives
\[
\dim_{\CC}Z_6^{(3)}
=
\sum_{(\alpha,\beta)\in\{\pm1\}^2}|V_{\alpha,\beta}|
\le
4\cdot 3=12.
\]

On the other hand, the Wedderburn decomposition
\[
A_{6,1}\cong \bigoplus_{x\in X_6}M_{d_6(x)}(\CC)
\]
shows that \(A_{6,1}\) has exactly \(|X_6|=21\) simple blocks, so
\[
\dim_{\CC}Z_6=21.
\]
Consequently,
\[
\dim_{\CC}(Z_6/Z_6^{(3)})
=
\dim_{\CC}Z_6-\dim_{\CC}Z_6^{(3)}
\ge
21-12=9.
\]

Finally, \(\dim_{\CC}Z_6^{(3)}=12\) holds if and only if every sector attains the upper bound \(|V_{\alpha,\beta}|=3\). Since each \(V_{\alpha,\beta}\) is already contained in \(\{2,3,4\}\), this is equivalent to requiring
\[
V_{\alpha,\beta}=\{2,3,4\}
\qquad
(\alpha,\beta\in\{\pm1\}^2).
\]
\end{proof}
