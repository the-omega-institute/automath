\mathcal A_\chi=\bigoplus_{X_m^{\alpha,\beta}\neq\varnothing}\CC\,e_{\alpha,\beta}.

\begin{theorem}[The joint spectral measure of \texorpdfstring{$\chi$}{chi} coincides with the pushforward distribution: trace formulas for signed collision moments]\label{thm:fold-groupoid-z2x2-joint-spectral-measure}
Equip \(A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC)\) with the normalized trace
\[
\tau_{m,1}(A):=2^{-m}\sum_{x\in X_m}\Tr(A_x),
\]
where \(A_x\in M_{d_m(x)}(\CC)\) is the \(x\)-block component.
Then for every minimal central projection \(\widetilde P_x\),
\[
\tau_{m,1}(\widetilde P_x)=\frac{d_m(x)}{2^m}=: \pi_m(x),
\]
so \(\pi_m\) is the pushforward of the uniform microstate baseline under the fold map \(\Fold_m\).

Let \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\) be as in Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}. Then for every function \(f:\{\pm1\}^2\to\CC\),
\[
\tau_{m,1}\!\bigl(f(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\bigr)
=\sum_{x\in X_m}\pi_m(x)\,f\!\bigl(\chi(x)\bigr).
\]
Hence, with respect to \(\tau_{m,1}\), the joint spectral measure of \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\) is exactly the pushforward measure \(\chi_\ast\pi_m\).

Furthermore, let
\[
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m} d_m(x)\,\widetilde P_x\ \in\ Z(A_{m,1})
\]
be the Watatani index element of the block-depolarizing conditional expectation \(E_{m,1}\).
Then for every integer \(q\ge 0\) and every \(s,t\in\{0,1\}\),
\[
\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\frac{1}{2^m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^{s}\operatorname{sgn}(\iota_x)^{t}\,d_m(x)^{q+1}.
\]
\end{theorem}

\begin{proof}
Because \(\sum_{x\in X_m}d_m(x)=|\Omega_m|=2^m\), the functional \(\tau_{m,1}\) is indeed a normalized trace on \(A_{m,1}\).
\par
Because \(\varepsilon_{\mathrm{scan}}\) and \(\varepsilon_{\mathrm{rev}}\) act by the scalars \(\chi(x)\) on the block \(x\), we may write
\[
f(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})
=\sum_{x\in X_m} f(\chi(x))\,\widetilde P_x.
\]
Taking traces and using
\[
\tau_{m,1}(\widetilde P_x)=2^{-m}\Tr(I_{d_m(x)})=\frac{d_m(x)}{2^m}
\]
gives the first identity.

For the second formula, note that \(\mathrm{IndW}(E_{m,1})\) acts on the block \(x\) by the scalar \(d_m(x)\). Therefore
\[
\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}
=\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^q\,\widetilde P_x.
\]
Taking the normalized trace once more yields
\[
2^{-m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^q\,\Tr(I_{d_m(x)})
=\frac{1}{2^m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{q+1}.
\]
\end{proof}

\begin{proposition}[Maximal-block \texorpdfstring{$\chi$}{chi}-homogeneity: the largest matrix blocks lie in a single sector]\label{prop:fold-groupoid-maxblock-chi-homogeneity}
Let \(\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=D_m\}\), and define the maximal-block ideal
\[
I_m:=\bigoplus_{x\in\mathcal{M}_m}M_{D_m}(\CC)\ \subset\ A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC).
\]
Let \(\chi_m\in\{\pm1\}^2\) be the constant fingerprint on maximal fibers supplied by Theorem \ref{thm:pom-maxfiber-orientation-charge-periodic}, and let \(e_{\chi_m}:=e_{\alpha,\beta}\), where \(\chi_m=(\alpha,\beta)\), be the corresponding central idempotent from Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}. Then
\[
\boxed{\;I_m\subseteq e_{\chi_m}A_{m,1}.\;}
\]
In particular, the Wedderburn decomposition of the quotient algebra \(A_{m,1}/e_{\chi_m}A_{m,1}\) contains no matrix block of dimension \(D_m\).
\end{proposition}
\begin{proof}
By Theorem \ref{thm:pom-maxfiber-orientation-charge-periodic}, every \(x\in\mathcal M_m\) satisfies \(\chi(x)=\chi_m\).
By Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}, the central projection \(e_{\chi_m}\) acts by \(1\) on block \(x\) if and only if \(\chi(x)=\chi_m\), and by \(0\) otherwise.
Hence \(\widetilde P_x\le e_{\chi_m}\) for all \(x\in \mathcal M_m\), so
\[
M_{D_m}(\CC)\cong \widetilde P_xA_{m,1}\subseteq e_{\chi_m}A_{m,1}.
\]
Taking the direct sum over all \(x\in\mathcal M_m\) gives \(I_m\subseteq e_{\chi_m}A_{m,1}\).
The final assertion is just the Wedderburn-theoretic restatement of this inclusion.
\end{proof}

\begin{theorem}[Support rigidity at maximal entropy loss under coarse-graining and the \texorpdfstring{$\chi$}{chi} anomaly invariant]\label{thm:fold-max-data-processing-saturation-chi}
Let \(E_m\) be the conditional expectation/coarse-graining channel induced by the fold map \(\Fold_m:\Omega_m\to X_m\), that is, uniform averaging along fibers. For a probability distribution \(\rho\) on \(\Omega_m\), write \(\rho^\flat:=\rho\circ E_m\). Define the maximal relative-entropy loss
\[
\Delta_m:=\sup_{\rho}D(\rho\Vert\rho^\flat),
\]
where \(\rho\) ranges over probability distributions on \(\Omega_m\) (equivalently, over states on the commutative algebra \(A_m=\CC^{\Omega_m}\)).
Then:
\begin{enumerate}
\normalfont
  \item \(\Delta_m=\log D_m\) by Proposition \ref{prop:fold-condexp-index-maxfiber}.
  \item If \(\rho\) attains the supremum, then the support of \((\Fold_m)_\ast\rho\) is contained in \(\mathcal{M}_m:=\{x\in X_m:\ d_m(x)=D_m\}\); equivalently,
  \[
  (\Fold_m)_\ast\rho(x)>0\ \Longrightarrow\ d_m(x)=D_m.
  \]
  \item Consequently, for any maximizing distribution \(\rho\), every macroscopic support point \(x^\star\in \mathrm{supp}((\Fold_m)_\ast\rho)\) satisfies
  \[
  \chi(x^\star)=\chi_m,
  \]
  where \(\chi_m\) is the constant maximal-fiber fingerprint from Theorem \ref{thm:pom-maxfiber-orientation-charge-periodic}. In particular, when \(m\) is odd and lies in the stable regime, one has \(\operatorname{sgn}(\iota_{x^\star})=+1\).
\end{enumerate}
\end{theorem}
\begin{proof}
The first statement is exactly Proposition \ref{prop:fold-condexp-index-maxfiber}.

For the second statement, let \(\rho_X:=(\Fold_m)_\ast\rho\), and for each \(x\in X_m\) let \(u_x\) denote the uniform distribution on \(\Fold_m^{-1}(x)\). The chain rule for \(\rho^\flat\) gives
\[
D(\rho\Vert\rho^\flat)
=\sum_{x\in X_m}\rho_X(x)\,D\!\bigl(\rho(\cdot\mid x)\,\Vert\,u_x\bigr).
\]
For each \(x\), the conditional divergence is bounded by
\[
D\!\bigl(\rho(\cdot\mid x)\Vert u_x\bigr)\le \log d_m(x)\le \log D_m.
\]
If \(\rho\) attains the value \(\Delta_m=\log D_m\), then this weighted average of numbers bounded above by \(\log D_m\) must itself equal \(\log D_m\). Therefore every \(x\) with \(\rho_X(x)>0\) satisfies \(\log d_m(x)=\log D_m\), so \(\rho_X\) is supported on \(\mathcal M_m\).

The third statement now follows immediately from the second and from Theorem \ref{thm:pom-maxfiber-orientation-charge-periodic}: any \(x^\star\in\mathrm{supp}(\rho_X)\subseteq\mathcal M_m\) satisfies \(\chi(x^\star)=\chi_m\).
The assertion about the second component in the odd stable regime is exactly the final clause of that theorem.
\end{proof}

\begin{corollary}[High-\texorpdfstring{$q\to\infty$}{q->infty} limits of the Watatani index moments: recovery of the maximal fiber size and the fingerprint \texorpdfstring{$\chi_m$}{chi\_m}]\label{cor:fold-indw-high-q-extract-maxfiber-and-chi}
Fix a resolution \(m\), and let
\[
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m}d_m(x)\,\widetilde P_x\ \in\ Z(A_{m,1})
\]
be as in Theorem \ref{thm:fold-groupoid-z2x2-joint-spectral-measure}.
Write
\[
D_m:=\max_{x\in X_m}d_m(x),\qquad
\mathcal M_m:=\{x\in X_m:\ d_m(x)=D_m\},
\qquad
\chi_m=(\alpha_m,\beta_m)\in\{\pm1\}^2,
\]
where \(\chi_m\) is furnished by Theorem \ref{thm:pom-maxfiber-orientation-charge-periodic}; by Proposition \ref{prop:fold-groupoid-maxblock-chi-homogeneity}, one has \(\chi(x)=\chi_m\) for every \(x\in\mathcal M_m\).
Then:
\begin{enumerate}
\normalfont
  \item The maximal fiber size is recovered by an \(\ell^\infty\)-type limit:
  \[
  D_m
  =
  \lim_{q\to\infty}\Bigl(2^m\,\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)\Bigr)^{1/(q+1)}.
  \]
  \item The fingerprint \(\chi_m\) is recovered by signed moment ratios: for every \(s,t\in\{0,1\}\),
  \[
  \lim_{q\to\infty}
  \frac{\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)}
  {\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)}
  =
  \alpha_m^{\,s}\beta_m^{\,t}.
  \]
  \item For any \((\alpha,\beta)\in\{\pm1\}^2\), define the sectorwise maximal fiber size by
  \[
  D_m^{\alpha,\beta}:=\max\{d_m(x):\chi(x)=(\alpha,\beta)\}\qquad(\text{with the convention }0\text{ for the empty set}).
  \]
  Then
  \[
  D_m^{\alpha,\beta}
  =
  \lim_{q\to\infty}\Bigl(2^m\,\tau_{m,1}\!\bigl(e_{\alpha,\beta}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)\Bigr)^{1/(q+1)}.
  \]
\end{enumerate}
\end{corollary}
\begin{proof}
Since \(\widetilde P_x\widetilde P_y=\delta_{xy}\widetilde P_x\) and \(\sum_x\widetilde P_x=1\),
\[
\mathrm{IndW}(E_{m,1})^{\,q}=\sum_{x\in X_m} d_m(x)^{q}\,\widetilde P_x,
\]
and therefore
\[
2^m\,\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m}d_m(x)^{q+1}.
\]
For a finite set of positive numbers, the limit of the \((q+1)\)-st root of the power sum is the maximum, so
\[
\lim_{q\to\infty}\Bigl(\sum_x d_m(x)^{q+1}\Bigr)^{1/(q+1)}=D_m.
\]
This proves the first statement.

For the second statement, Theorem \ref{thm:fold-groupoid-z2x2-joint-spectral-measure} gives
\[
2^m\,\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{q+1}.
\]
After dividing numerator and denominator by \(D_m^{q+1}\), all terms with \(d_m(x)<D_m\) vanish as \(q\to\infty\), and only the contribution from \(\mathcal M_m\) remains.
Since \(\chi\) is constant on \(\mathcal M_m\) with value \(\chi_m=(\alpha_m,\beta_m)\), the limit equals \(\alpha_m^s\beta_m^t\).

For the third statement, the block-projection property from Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents} gives
\[
e_{\alpha,\beta}\widetilde P_x=\ind[\chi(x)=(\alpha,\beta)]\,\widetilde P_x.
\]
Hence
\[
2^m\,\tau_{m,1}\!\bigl(e_{\alpha,\beta}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{\chi(x)=(\alpha,\beta)} d_m(x)^{q+1},
\]
and taking the \(\ell^\infty\)-limit as \(q\to\infty\) yields \(D_m^{\alpha,\beta}\).
\end{proof}

\begin{proposition}[Two-bit signed power sums uniquely certify the \texorpdfstring{$\chi$}{chi}-layered fiber spectrum]\label{prop:fold-chi-layered-fiber-spectrum-certified-by-signed-powersums}
Fix \(m\), and write \(N_m:=|X_m|\).
For \(s,t\in\{0,1\}\) and \(q=0,1,\dots,N_m-1\), define the two-bit signed power sums
\[
M_{s,t}(q)
:=
2^m\,\tau_{m,1}\!\left(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\right)
=
\sum_{x\in X_m}\chi_1(x)^s\chi_2(x)^t\,d_m(x)^{q+1},
\]
where \(\chi(x)=(\chi_1(x),\chi_2(x))\in\{\pm1\}^2\).
Then for each \((\alpha,\beta)\in\{\pm1\}^2\), the polynomial
\[
Q_m^{\alpha,\beta}(z):=
\prod_{\substack{x\in X_m\\ \chi(x)=(\alpha,\beta)}}\bigl(1-d_m(x)\,z\bigr)\ \in\ \ZZ[z]
\]
is uniquely determined by the finite data set
\(
\{M_{s,t}(q):\ (s,t)\in\{0,1\}^2,\ 0\le q\le N_m-1\}
\),
and this determination is constructive via the Newton identities.
\end{proposition}
\begin{proof}
For each \((\alpha,\beta)\in\{\pm1\}^2\) and each \(k=1,2,\dots,N_m\), the character orthogonality of \(\{\pm1\}^2\) (equivalently, Fourier-Hadamard inversion) gives the sectorwise power sums
\[
p_k^{\alpha,\beta}
:=
\sum_{\chi(x)=(\alpha,\beta)} d_m(x)^k
=
\frac14\sum_{s,t\in\{0,1\}}\alpha^s\beta^t\,M_{s,t}(k-1).
\]
If one pads the multiset \(\{d_m(x):\chi(x)=(\alpha,\beta)\}\) with zeros up to length \(N_m\), its power sums remain exactly \((p_k^{\alpha,\beta})_{1\le k\le N_m}\).
The Newton identities determine uniquely the coefficients of the monic polynomial
\[
\prod_{j=1}^{N_m}(1-\lambda_j z)
\]
from the first \(N_m\) power sums, and conversely.
Since zero roots contribute no nontrivial linear factor, the polynomial \(Q_m^{\alpha,\beta}(z)\) is uniquely determined by the stated data.
\end{proof}

\qquad k=1,2,3.
\]

When \(m=6\), Corollary \ref{cor:fold-bin-m6-fiber-hist} shows that the only possible fiber sizes are \(2,3,4\). Hence for a fixed sector \((\alpha,\beta)\), the counts \(n_2^{\alpha,\beta}\), \(n_3^{\alpha,\beta}\), and \(n_4^{\alpha,\beta}\) satisfy the linear system
\[
p_1^{\alpha,\beta}=2n_2^{\alpha,\beta}+3n_3^{\alpha,\beta}+4n_4^{\alpha,\beta},
\]
\[
p_2^{\alpha,\beta}=4n_2^{\alpha,\beta}+9n_3^{\alpha,\beta}+16n_4^{\alpha,\beta},
\]
\[
p_3^{\alpha,\beta}=8n_2^{\alpha,\beta}+27n_3^{\alpha,\beta}+64n_4^{\alpha,\beta}.
\]
Its coefficient matrix is
\[
\begin{pmatrix}
2&3&4\\
4&9&16\\
8&27&64
\end{pmatrix},
\]
whose determinant is \(48\neq 0\). Therefore the system is invertible, and direct inversion yields the three displayed formulas.

Finally, because \(d_6(x)\) takes only the values \(2,3,4\), the linear factors of \(Q_6^{\alpha,\beta}(z)\) can only be \(1-2z\), \(1-3z\), and \(1-4z\), with exponents exactly \(n_2^{\alpha,\beta}\), \(n_3^{\alpha,\beta}\), and \(n_4^{\alpha,\beta}\), respectively.
\end{proof}

\begin{proposition}[Central resolution criterion: the dimension and surjectivity test for the center generated by \texorpdfstring{$(\mathrm{IndW},\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})$}{(IndW,eps\_scan,eps\_rev)}]\label{prop:fold-center-resolution-criterion-indw-eps}
Write \(Z_m:=Z(A_{m,1})\), and define the central subalgebra
\[
Z_m^{(3)}:=\CC\!\left[\mathrm{IndW}(E_{m,1}),\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}}\right]\ \subseteq\ Z_m.
\]
Then:
\begin{enumerate}
\normalfont
  \item \(Z_m^{(3)}\) decomposes sectorwise as
  \[
  Z_m^{(3)}
  =
  \bigoplus_{(\alpha,\beta)\in\{\pm1\}^2}\CC\!\left[\mathrm{IndW}(E_{m,1})\right]\,e_{\alpha,\beta}.
  \]
  \item If
  \[
  V_{\alpha,\beta}:=\{d_m(x):\ \chi(x)=(\alpha,\beta)\}\subset\NN
  \]
  denotes the set of distinct fiber multiplicities appearing in the sector \((\alpha,\beta)\), then
  \[
  \dim_{\CC}Z_m^{(3)}=\sum_{(\alpha,\beta)\in\{\pm1\}^2}|V_{\alpha,\beta}|.
  \]
  \item Consequently,
  \[
  Z_m^{(3)}=Z_m
  \]
  if and only if, for every \((\alpha,\beta)\), the map
  \[
  x\longmapsto d_m(x)
  \quad\text{on}\quad
  X_m^{\alpha,\beta}:=\{x\in X_m:\ \chi(x)=(\alpha,\beta)\}
  \]
  is injective.
\end{enumerate}
\end{proposition}
\begin{proof}
Under the Wedderburn identification \(A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC)\), the center \(Z_m\cong\CC^{X_m}\) is spanned by \(\{\widetilde P_x\}_{x\in X_m}\), and
\[
\mathrm{IndW}(E_{m,1})=\sum_x d_m(x)\,\widetilde P_x,\qquad
\varepsilon_{\mathrm{scan}}=\sum_x \chi_1(x)\,\widetilde P_x,\qquad
\varepsilon_{\mathrm{rev}}=\sum_x \chi_2(x)\,\widetilde P_x.
\]
Therefore every element of \(Z_m^{(3)}\) takes on block \(x\) a value depending only on \(\chi(x)\) and \(d_m(x)\), and within each fixed sector it is a polynomial function of \(d_m\).
Multiplication by \(e_{\alpha,\beta}\) simply restricts such a function to the sector \(\chi=(\alpha,\beta)\), proving the first statement.

On a finite set, the image of the polynomial algebra \(\CC[d]\) has dimension equal to the number of distinct values taken by \(d\) on that set. This gives the dimension formula in (2).
Finally, since \(\dim_{\CC}Z_m=|X_m|\), the criterion in (3) follows immediately from (2): one has \(Z_m^{(3)}=Z_m\) exactly when there are no collisions of \(d_m\) inside any fixed sector.
\end{proof}

\begin{theorem}[For window \texorpdfstring{$6$}{6}, the three central observables have dimension at most \(12\) and leave an irreducible resolution defect]\label{thm:fold-window6-center-three-observables-dimension-defect}
Take \(m=6\), and write
\[
A_{6,1}:=\CC[\mathcal R_6],
\qquad
Z_6:=Z(A_{6,1}),
\]
\[
Z_6^{(3)}
:=
\CC\!\left[
\mathrm{IndW}(E_{6,1}),
\varepsilon_{\mathrm{scan}},
\varepsilon_{\mathrm{rev}}
\right]
\subseteq Z_6.
\]
Then one necessarily has the strict upper bound
\[
\dim_{\CC}Z_6^{(3)}\le 12.
\]
On the other hand, since \(|X_6|=21\),
\[
\dim_{\CC}Z_6=21,
\qquad
\dim_{\CC}(Z_6/Z_6^{(3)})\ge 9.
\]
Equivalently, the three central observables \(\mathrm{IndW}(E_{6,1})\), \(\varepsilon_{\mathrm{scan}}\), and \(\varepsilon_{\mathrm{rev}}\) can never completely distinguish all \(21\) stable types in window \(6\); at least nine central degrees of freedom remain invisible to this triple.

Moreover, the following are equivalent:
\[
\dim_{\CC}Z_6^{(3)}=12
\iff
V_{\alpha,\beta}=\{2,3,4\}
\quad\text{for every}\quad
(\alpha,\beta)\in\{\pm1\}^2.
\]
\end{theorem}
\begin{proof}
By Corollary \ref{cor:fold-bin-m6-fiber-hist}, the fiber multiplicities in window \(6\) take only the values \(2,3,4\). Hence for every \((\alpha,\beta)\in\{\pm1\}^2\),
\[
V_{\alpha,\beta}\subseteq \{2,3,4\},
\qquad
|V_{\alpha,\beta}|\le 3.
\]
Proposition \ref{prop:fold-center-resolution-criterion-indw-eps} therefore gives
\[
\dim_{\CC}Z_6^{(3)}
=
\sum_{(\alpha,\beta)\in\{\pm1\}^2}|V_{\alpha,\beta}|
\le
4\cdot 3=12.
\]

On the other hand, the Wedderburn decomposition
\[
A_{6,1}\cong \bigoplus_{x\in X_6}M_{d_6(x)}(\CC)
\]
shows that \(A_{6,1}\) has exactly \(|X_6|=21\) simple blocks, so
\[
\dim_{\CC}Z_6=21.
\]
Consequently,
\[
\dim_{\CC}(Z_6/Z_6^{(3)})
=
\dim_{\CC}Z_6-\dim_{\CC}Z_6^{(3)}
\ge
21-12=9.
\]

Finally, \(\dim_{\CC}Z_6^{(3)}=12\) holds if and only if every sector attains the upper bound \(|V_{\alpha,\beta}|=3\). Since each \(V_{\alpha,\beta}\) is already contained in \(\{2,3,4\}\), this is equivalent to requiring
\[
V_{\alpha,\beta}=\{2,3,4\}
\qquad
(\alpha,\beta\in\{\pm1\}^2).
\]
\end{proof}

\begin{conjecture}[Eventual simple spectrum of fiber multiplicities inside each sector]\label{conj:fold-sectorwise-eventual-simple-spectrum}
There exists \(m_0\) such that for every \(m\ge m_0\) and every \((\alpha,\beta)\in\{\pm1\}^2\), the map

In other words, the center eventually reaches its full resolution limit under the observable system generated by the one-variable spectral field \(\mathrm{IndW}\) together with the two-bit fingerprint \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\).
\end{conjecture}
