\begin{theorem}\label{thm:fold-groupoid-z2x2-central-idempotents}
Let \(\mathcal{R}_m=\Omega_m\times_{X_m}\Omega_m\) be the fold equivalence-relation groupoid of Definition \ref{def:fold-equivalence-groupoid-algebra}, and write
\[
A_{m,1}:=\CC[\mathcal{R}_m]\cong \bigoplus_{x\in X_m}M_{d_m(x)}(\CC)
\]
as in Proposition \ref{prop:fold-groupoid-wedderburn}. For each \(x\in X_m\), let \(\widetilde P_x\in Z(A_{m,1})\) be the minimal central idempotent corresponding to the block \(M_{d_m(x)}(\CC)\), and let
\[
\chi(x)=\bigl(\operatorname{sgn}(c_x),\operatorname{sgn}(\iota_x)\bigr)\in\{\pm1\}^2
\]
be the scan-reversal fingerprint of Proposition \ref{prop:fold-groupoid-z2x2-sector-by-orientation}.

Define central self-adjoint unitaries
\[
\varepsilon_{\mathrm{scan}}:=\sum_{x\in X_m}\operatorname{sgn}(c_x)\,\widetilde P_x,
\qquad
\varepsilon_{\mathrm{rev}}:=\sum_{x\in X_m}\operatorname{sgn}(\iota_x)\,\widetilde P_x.
\]
Then \(\varepsilon_{\mathrm{scan}}^2=\varepsilon_{\mathrm{rev}}^2=1\), and the two involutions commute. For \((\alpha,\beta)\in\{\pm1\}^2\), set
\[
e_{\alpha,\beta}
:=
\frac14\bigl(1+\alpha\,\varepsilon_{\mathrm{scan}}\bigr)\bigl(1+\beta\,\varepsilon_{\mathrm{rev}}\bigr)
\in Z(A_{m,1}).
\]
Then \(\{e_{\alpha,\beta}\}_{(\alpha,\beta)\in\{\pm1\}^2}\) is a complete family of pairwise orthogonal central idempotents with sum \(1\), and
\[
e_{\alpha,\beta}A_{m,1}
\cong
\bigoplus_{\chi(x)=(\alpha,\beta)}M_{d_m(x)}(\CC).
\]
In particular,
\[
\mathcal A_\chi:=C^\ast(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})
=
\bigoplus_{X_m^{\alpha,\beta}\neq\varnothing}\CC\,e_{\alpha,\beta}.
\]
Thus \(\mathcal A_\chi\) is canonically a quotient of \(\CC[\ZZ_2\times\ZZ_2]\), and \(\mathcal A_\chi\cong \CC^4\) when all four sectors are nonempty.
\end{theorem}

\begin{proof}
Under the Wedderburn decomposition, the center \(Z(A_{m,1})\) is spanned by \(\{\widetilde P_x\}_{x\in X_m}\), with
\[
\widetilde P_x\widetilde P_y=\delta_{xy}\widetilde P_x,
\qquad
\sum_{x\in X_m}\widetilde P_x=1.
\]
It follows immediately that \(\varepsilon_{\mathrm{scan}}\) and \(\varepsilon_{\mathrm{rev}}\) are commuting central involutions. On the block indexed by \(x\), they act by the scalars \(\operatorname{sgn}(c_x)\) and \(\operatorname{sgn}(\iota_x)\), respectively.

Since the two involutions commute, one checks directly that the elements \(e_{\alpha,\beta}\) are central idempotents, pairwise orthogonal, and sum to \(1\). On the \(x\)-block one has
\[
e_{\alpha,\beta}\big|_x
=
\frac14\bigl(1+\alpha\,\operatorname{sgn}(c_x)\bigr)\bigl(1+\beta\,\operatorname{sgn}(\iota_x)\bigr),
\]
and this scalar is \(1\) precisely when \(\chi(x)=(\alpha,\beta)\), and \(0\) otherwise. Hence \(e_{\alpha,\beta}\) acts as the identity exactly on the blocks in the sector \((\alpha,\beta)\), which proves the decomposition of \(e_{\alpha,\beta}A_{m,1}\).

The description of \(\mathcal A_\chi\) follows at once: it is a commutative finite-dimensional \(C^\ast\)-algebra generated by two commuting involutions, so its minimal central idempotents are precisely the nonzero \(e_{\alpha,\beta}\). Equivalently, \(\mathcal A_\chi\) is the image of the universal commutative \(C^\ast\)-algebra generated by two commuting order-two unitaries, namely \(\CC[\ZZ_2\times\ZZ_2]\), under the map sending the canonical generators to \(\varepsilon_{\mathrm{scan}}\) and \(\varepsilon_{\mathrm{rev}}\).
\end{proof}

\begin{remark}
The projection formula is elementary. What is specific to the present setting is that the relevant involutions are canonically attached to the fold-groupoid model, so the sector decomposition is recovered internally from the algebra.
\end{remark}

\begin{theorem}\label{thm:fold-groupoid-z2x2-joint-spectral-measure}
Equip \(A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC)\) with the normalized trace
\[
\tau_{m,1}(A):=2^{-m}\sum_{x\in X_m}\Tr(A_x),
\]
where \(A_x\in M_{d_m(x)}(\CC)\) denotes the \(x\)-block.
Then, for every minimal central projection \(\widetilde P_x\),
\[
\tau_{m,1}(\widetilde P_x)=\frac{d_m(x)}{2^m}=: \pi_m(x),
\]
so \(\pi_m\) is the pushforward of the uniform measure on \(\Omega_m\) under \(\Fold_m\).

For every function \(f:\{\pm1\}^2\to\CC\),
\[
\tau_{m,1}\!\bigl(f(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\bigr)
=
\sum_{x\in X_m}\pi_m(x)\,f\!\bigl(\chi(x)\bigr).
\]
Thus the joint spectral measure of \((\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})\) with respect to \(\tau_{m,1}\) is exactly \(\chi_\ast\pi_m\).

Moreover, if
\[
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m}d_m(x)\,\widetilde P_x\in Z(A_{m,1})
\]
is the Watatani index element of the block-depolarizing conditional expectation \(E_{m,1}\), then for every \(q\ge 0\) and every \(s,t\in\{0,1\}\),
\[
\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\frac{1}{2^m}\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{q+1}.
\]
\end{theorem}

\begin{proof}
Since \(\sum_{x\in X_m}d_m(x)=|\Omega_m|=2^m\), the functional \(\tau_{m,1}\) is a normalized trace. Moreover,
\[
\tau_{m,1}(\widetilde P_x)=2^{-m}\Tr(I_{d_m(x)})=\frac{d_m(x)}{2^m},
\]
which is exactly the pushforward mass of \(x\) under the uniform measure on \(\Omega_m\).

Because \(\varepsilon_{\mathrm{scan}}\) and \(\varepsilon_{\mathrm{rev}}\) act on the \(x\)-block by the scalars \(\chi(x)\), one has
\[
f(\varepsilon_{\mathrm{scan}},\varepsilon_{\mathrm{rev}})
=
\sum_{x\in X_m}f(\chi(x))\,\widetilde P_x.
\]
Taking traces gives the asserted formula for the joint spectral measure.

For the moment identity, note that \(\mathrm{IndW}(E_{m,1})\) acts on the \(x\)-block by the scalar \(d_m(x)\). Hence
\[
\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}
=
\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^q\,\widetilde P_x,
\]
and tracing once more yields the formula.
\end{proof}

\begin{proposition}\label{prop:fold-groupoid-maxblock-chi-homogeneity}
Let \(\mathcal{M}_m:=\{x\in X_m:d_m(x)=D_m\}\), and define the maximal-block ideal
\[
I_m:=\bigoplus_{x\in\mathcal{M}_m}M_{D_m}(\CC)\subset A_{m,1}\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC).
\]
Let \(\chi_m\in\{\pm1\}^2\) be the constant maximal-fiber fingerprint supplied by Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}, and let \(e_{\chi_m}\) denote the corresponding central idempotent from Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents}. Then
\[
I_m\subseteq e_{\chi_m}A_{m,1}.
\]
In particular, the quotient algebra \(A_{m,1}/e_{\chi_m}A_{m,1}\) has no simple block of dimension \(D_m\).
\end{proposition}

\begin{proof}
By Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}, every \(x\in\mathcal M_m\) satisfies \(\chi(x)=\chi_m\). Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents} therefore implies that \(e_{\chi_m}\) acts as the identity on each maximal block and vanishes on the remaining sectors. Thus \(\widetilde P_x\le e_{\chi_m}\) for all \(x\in\mathcal M_m\), and
\[
M_{D_m}(\CC)\cong \widetilde P_xA_{m,1}\subseteq e_{\chi_m}A_{m,1}.
\]
Taking the direct sum over \(x\in\mathcal M_m\) yields the claim.
\end{proof}

\begin{theorem}\label{thm:fold-max-data-processing-saturation-chi}
Let \(E_m\) be the conditional expectation induced by the fold map \(\Fold_m:\Omega_m\to X_m\), that is, uniform averaging on the fibers. For a probability distribution \(\rho\) on \(\Omega_m\), write \(\rho^\flat:=\rho\circ E_m\), and define
\[
\Delta_m:=\sup_{\rho}D(\rho\Vert \rho^\flat),
\]
where \(\rho\) ranges over all probability distributions on \(\Omega_m\). Then:
\begin{enumerate}
\normalfont
  \item \(\Delta_m=\log D_m\).
  \item If \(\rho\) attains the supremum, then \((\Fold_m)_\ast\rho\) is supported on
  \[
  \mathcal{M}_m:=\{x\in X_m:d_m(x)=D_m\}.
  \]
  \item Consequently, every \(x^\star\in \mathrm{supp}((\Fold_m)_\ast\rho)\) satisfies
  \[
  \chi(x^\star)=\chi_m,
  \]
  where \(\chi_m\) is the maximal-fiber fingerprint from Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}. In particular, in the odd stable regime one has \(\operatorname{sgn}(\iota_{x^\star})=+1\).
\end{enumerate}
\end{theorem}

\begin{proof}
The identity \(\Delta_m=\log D_m\) is exactly Proposition \ref{prop:fold-condexp-index-maxfiber}.

For the support statement, let \(\rho_X:=(\Fold_m)_\ast\rho\), and for each \(x\in X_m\) let \(u_x\) be the uniform measure on the fiber \(\Fold_m^{-1}(x)\). The relative-entropy chain rule gives
\[
D(\rho\Vert\rho^\flat)
=
\sum_{x\in X_m}\rho_X(x)\,D\!\bigl(\rho(\cdot\mid x)\,\Vert\,u_x\bigr).
\]
Each fiber contribution satisfies
\[
D\!\bigl(\rho(\cdot\mid x)\,\Vert\,u_x\bigr)\le \log d_m(x)\le \log D_m.
\]
If \(\rho\) realizes the supremum \(\Delta_m=\log D_m\), then a convex combination of numbers bounded above by \(\log D_m\) attains that bound. Hence every \(x\) with \(\rho_X(x)>0\) must satisfy \(d_m(x)=D_m\), which proves (2).

The final statement follows immediately from (2) and Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}, since that assumption makes \(\chi\) constant on \(\mathcal M_m\).
\end{proof}

\begin{corollary}\label{cor:fold-indw-high-q-extract-maxfiber-and-chi}
Fix \(m\), and let
\[
\mathrm{IndW}(E_{m,1})=\sum_{x\in X_m}d_m(x)\,\widetilde P_x\in Z(A_{m,1})
\]
be as above. Write
\[
D_m:=\max_{x\in X_m}d_m(x),
\qquad
\mathcal M_m:=\{x\in X_m:d_m(x)=D_m\},
\qquad
\chi_m:=(\alpha_m,\beta_m)\in\{\pm1\}^2,
\]
where \(\chi_m\) is furnished by Assumption \ref{thm:pom-maxfiber-orientation-charge-periodic}. Then:
\begin{enumerate}
\normalfont
  \item
  \[
  D_m
  =
  \lim_{q\to\infty}\Bigl(2^m\,\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)\Bigr)^{1/(q+1)}.
  \]
  \item For every \(s,t\in\{0,1\}\),
  \[
  \lim_{q\to\infty}
  \frac{\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)}
  {\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)}
  =
  \alpha_m^{\,s}\beta_m^{\,t}.
  \]
  \item For any \((\alpha,\beta)\in\{\pm1\}^2\), if
  \[
  D_m^{\alpha,\beta}:=
  \max\{d_m(x):\chi(x)=(\alpha,\beta)\},
  \]
  with the convention \(D_m^{\alpha,\beta}=0\) when the sector is empty, then
  \[
  D_m^{\alpha,\beta}
  =
  \lim_{q\to\infty}\Bigl(2^m\,\tau_{m,1}\!\bigl(e_{\alpha,\beta}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)\Bigr)^{1/(q+1)}.
  \]
\end{enumerate}
\end{corollary}

\begin{proof}
Since the \(\widetilde P_x\) are pairwise orthogonal and sum to \(1\),
\[
\mathrm{IndW}(E_{m,1})^{\,q}=\sum_{x\in X_m}d_m(x)^q\,\widetilde P_x,
\]
and therefore
\[
2^m\,\tau_{m,1}\!\bigl(\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m}d_m(x)^{q+1}.
\]
The \((q+1)\)-st root of this power sum tends to the largest fiber size \(D_m\), which proves (1).

For (2), Theorem \ref{thm:fold-groupoid-z2x2-joint-spectral-measure} gives
\[
2^m\,\tau_{m,1}\!\bigl(\varepsilon_{\mathrm{scan}}^{\,s}\varepsilon_{\mathrm{rev}}^{\,t}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{x\in X_m}\operatorname{sgn}(c_x)^s\operatorname{sgn}(\iota_x)^t\,d_m(x)^{q+1}.
\]
After dividing numerator and denominator by \(D_m^{q+1}\), all terms with \(d_m(x)<D_m\) vanish as \(q\to\infty\), while the contribution from \(\mathcal M_m\) survives. Since \(\chi\) is constant on \(\mathcal M_m\) with value \(\chi_m=(\alpha_m,\beta_m)\), the ratio tends to \(\alpha_m^s\beta_m^t\).

For (3), Theorem \ref{thm:fold-groupoid-z2x2-central-idempotents} gives
\[
e_{\alpha,\beta}\widetilde P_x=
\begin{cases}
\widetilde P_x,& \chi(x)=(\alpha,\beta),\\
0,& \chi(x)\neq(\alpha,\beta).
\end{cases}
\]
Hence
\[
2^m\,\tau_{m,1}\!\bigl(e_{\alpha,\beta}\,\mathrm{IndW}(E_{m,1})^{\,q}\bigr)
=
\sum_{\chi(x)=(\alpha,\beta)}d_m(x)^{q+1},
\]
and the same \(\ell^\infty\)-limit argument yields \(D_m^{\alpha,\beta}\).
\end{proof}
