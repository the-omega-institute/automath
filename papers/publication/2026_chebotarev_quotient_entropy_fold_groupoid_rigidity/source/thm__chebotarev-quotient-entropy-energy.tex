\begin{theorem}[Relative-entropy chain decomposition for Chebotarev layers under a group quotient]\label{thm:kernel-chebotarev-quotient-relative-entropy-chain}
Let \(\pi:G_2\twoheadrightarrow G_1\) be a surjective homomorphism of finite abelian groups, and write \(K:=\ker\pi\).
Fix a temperature parameter \(u\in(0,1]\) and a length \(n\ge 1\), and assume that the total primitive count
\[
B_n(\mathbf{1};u):=
\sum_{c_2\in G_2}B^{(2)}_{n,c_2}(u)
=
\sum_{c_1\in G_1}B^{(1)}_{n,c_1}(u)
\]
is strictly positive, where the coarse-layer counts are obtained from the fine layer by pushforward:
\[
B^{(1)}_{n,c_1}(u):=
\sum_{c_2\in\pi^{-1}(c_1)}B^{(2)}_{n,c_2}(u)
\qquad(c_1\in G_1).
\]
Define probability measures
\[
\mu_n^{(i)}(c):=\frac{B^{(i)}_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(i=1,2),
\]
and write \(\mathrm{unif}_{G_i}\) for the uniform measure on \(G_i\).
For each \(c_1\in G_1\) with \(\mu_n^{(1)}(c_1)>0\), define the conditional fiber distribution
\[
\mu_{n,\mathrm{fib}}^{(c_1)}(c_2):=
\frac{\mu_n^{(2)}(c_2)}{\mu_n^{(1)}(c_1)}
\qquad(c_2\in\pi^{-1}(c_1)),
\]
and let \(\mathrm{unif}_{\pi^{-1}(c_1)}\) denote the uniform distribution on that fiber.
Then one has the exact relative-entropy chain identity
\[
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr).
\]
By convention, the contribution of a fiber with \(\mu_n^{(1)}(c_1)=0\) is \(0\).
\end{theorem}

\begin{proof}
By construction, \(\mu_n^{(1)}=\pi_\ast\mu_n^{(2)}\).
Since \(\pi\) is a surjective homomorphism of finite groups, every fiber \(\pi^{-1}(c_1)\) has cardinality \(|K|\). Hence for every \(c_2\in G_2\),
\[
\mathrm{unif}_{G_2}(c_2)
=
\frac{1}{|G_2|}
=
\frac{1}{|G_1|\,|K|}
=
\mathrm{unif}_{G_1}(\pi(c_2))\cdot \mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2).
\]
Whenever \(\mu_n^{(1)}(\pi(c_2))>0\), we also have the factorization
\[
\mu_n^{(2)}(c_2)
=
\mu_n^{(1)}(\pi(c_2))\cdot
\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2).
\]
Therefore,
\[
\log\frac{\mu_n^{(2)}(c_2)}{\mathrm{unif}_{G_2}(c_2)}
=
\log\frac{\mu_n^{(1)}(\pi(c_2))}{\mathrm{unif}_{G_1}(\pi(c_2))}
+
\log\frac{\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2)}{\mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2)}.
\]
Multiplying by \(\mu_n^{(2)}(c_2)\), summing over \(c_2\in G_2\), and regrouping by fibers yields
\[
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr).
\]
This is the desired identity.
\end{proof}

\begin{theorem}[Orthogonal decomposition of visible and hidden character energy]\label{thm:kernel-chebotarev-visible-hidden-character-energy}
In the setting of Theorem \ref{thm:kernel-chebotarev-quotient-relative-entropy-chain}, define the relative densities with respect to the uniform measures by
\[
r_n^{(i)}(c):=|G_i|\,\mu_n^{(i)}(c)
=
\frac{|G_i|\,B^{(i)}_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(i=1,2).
\]
Then one has the exact orthogonal decomposition
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
+
\frac{1}{|G_2|}\sum_{c_1\in G_1}\sum_{c_2\in\pi^{-1}(c_1)}
\bigl(r_n^{(2)}(c_2)-r_n^{(1)}(c_1)\bigr)^2.
\]
Furthermore, let
\[
K^\perp:=\{\chi\in\widehat G_2:\ \chi|_K\equiv 1\}\cong \widehat G_1,
\]
and define the twisted primitive spectrum on the fine layer by
\[
B_n^{(2)}(\chi;u):=\sum_{c_2\in G_2}\chi(c_2)\,B^{(2)}_{n,c_2}(u)
\qquad(\chi\in\widehat G_2).
\]
Then the decomposition admits the Fourier/Plancherel form
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2,
\]
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in K^\perp\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
Consequently, the quotient map removes exactly the hidden character energy carried by characters that are nontrivial on the kernel:
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
\;-\;
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
\;=\;
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus K^\perp}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
\end{theorem}

\begin{proof}
For \(c_1\in G_1\), the pushforward relation gives
\[
r_n^{(1)}(c_1)
=
\frac{1}{|K|}\sum_{c_2\in\pi^{-1}(c_1)}r_n^{(2)}(c_2).
\]
Thus \(r_n^{(1)}\circ\pi\) is the conditional expectation of \(r_n^{(2)}\) onto the closed subspace of functions on \(G_2\) that are constant on fibers of \(\pi\). Equivalently, it is the orthogonal projection of \(r_n^{(2)}\) in \(L^2(\mathrm{unif}_{G_2})\) onto the fiber-constant subspace. Pythagoras' identity in Hilbert space therefore yields
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}\circ\pi-1\|_{L^2(\mathrm{unif}_{G_2})}^2
+
\|r_n^{(2)}-r_n^{(1)}\circ\pi\|_{L^2(\mathrm{unif}_{G_2})}^2.
\]
Because every fiber has cardinality \(|K|\), averaging over \(G_2\) and then over the fibers of \(\pi\) gives
\[
\|r_n^{(1)}\circ\pi-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2.
\]
Expanding the second term fiber by fiber gives the displayed orthogonal decomposition.

For the Fourier formulas, apply Parseval's identity to the character basis of \(G_2\). The Fourier coefficients of \(r_n^{(2)}\) are
\[
\widehat r_n^{(2)}(\chi)
:=
\frac{1}{|G_2|}\sum_{c_2\in G_2}r_n^{(2)}(c_2)\,\overline{\chi(c_2)}
=
\frac{B_n^{(2)}(\overline\chi;u)}{B_n(\mathbf{1};u)}.
\]
Since \(\chi\mapsto \overline\chi\) is a permutation of \(\widehat G_2\), we obtain
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}\abs{\widehat r_n^{(2)}(\chi)}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
Likewise, the characters of \(G_1\) identify under pullback with \(K^\perp\subseteq \widehat G_2\), so
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in K^\perp\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
Subtracting the two identities yields the explicit hidden-energy formula.
\end{proof}

\begin{remark}
The chain rule and Parseval identities used here are classical. The content of Theorems \ref{thm:kernel-chebotarev-quotient-relative-entropy-chain} and \ref{thm:kernel-chebotarev-visible-hidden-character-energy} is that, for primitive Chebotarev layers, the coarse counts are the exact pushforward of the fine counts, so the discarded term is identified canonically with the character energy outside \(K^\perp\).
\end{remark}
