\begin{theorem}[Entropy decomposition under a quotient]\label{thm:kernel-chebotarev-quotient-relative-entropy-chain}
Let \(\pi:G_2\twoheadrightarrow G_1\) be a surjective homomorphism of finite abelian groups, and write \(K:=\ker\pi\).
Fix \(u\in(0,1]\) and \(n\ge 1\), and assume that the total primitive count
\[
B_n(\mathbf{1};u):=
\sum_{c_2\in G_2}B^{(2)}_{n,c_2}(u)
=
\sum_{c_1\in G_1}B^{(1)}_{n,c_1}(u)
\]
is strictly positive, where the coarse-layer counts are obtained from the fine layer by pushforward:
\[
B^{(1)}_{n,c_1}(u):=
\sum_{c_2\in\pi^{-1}(c_1)}B^{(2)}_{n,c_2}(u)
\qquad(c_1\in G_1).
\]
Set
\[
\mu_n^{(i)}(c):=\frac{B^{(i)}_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(i=1,2),
\]
and write \(\mathrm{unif}_{G_i}\) for the uniform measure on \(G_i\).
For each \(c_1\in G_1\) with \(\mu_n^{(1)}(c_1)>0\), define the conditional fiber distribution
\[
\mu_{n,\mathrm{fib}}^{(c_1)}(c_2):=
\frac{\mu_n^{(2)}(c_2)}{\mu_n^{(1)}(c_1)}
\qquad(c_2\in\pi^{-1}(c_1)),
\]
and let \(\mathrm{unif}_{\pi^{-1}(c_1)}\) denote the uniform distribution on that fiber.
Then
\[
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr).
\]
Fibers with \(\mu_n^{(1)}(c_1)=0\) are understood to contribute \(0\).
\end{theorem}

\begin{proof}
By construction \(\mu_n^{(1)}=\pi_\ast\mu_n^{(2)}\). Since \(\pi\) is surjective, every fiber \(\pi^{-1}(c_1)\) has cardinality \(|K|\), and therefore
\[
\mathrm{unif}_{G_2}(c_2)
=
\frac{1}{|G_2|}
=
\frac{1}{|G_1|\,|K|}
=
\mathrm{unif}_{G_1}(\pi(c_2))\cdot \mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2).
\]
If \(\mu_n^{(1)}(c_1)=0\), then \(\mu_n^{(2)}(c_2)=0\) for every \(c_2\in\pi^{-1}(c_1)\), so there is nothing to prove on that fiber. On the remaining fibers,
\[
\mu_n^{(2)}(c_2)
=
\mu_n^{(1)}(\pi(c_2))\cdot
\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2).
\]
It follows that
\[
\log\frac{\mu_n^{(2)}(c_2)}{\mathrm{unif}_{G_2}(c_2)}
=
\log\frac{\mu_n^{(1)}(\pi(c_2))}{\mathrm{unif}_{G_1}(\pi(c_2))}
+
\log\frac{\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2)}{\mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2)}.
\]
Multiplying by \(\mu_n^{(2)}(c_2)\), summing over \(c_2\in G_2\), and regrouping the terms by fibers gives
\[
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr).
\]
\end{proof}

\begin{theorem}[Visible and hidden Fourier energy]\label{thm:kernel-chebotarev-visible-hidden-character-energy}
In the setting of Theorem \ref{thm:kernel-chebotarev-quotient-relative-entropy-chain}, define the relative densities by
\[
r_n^{(i)}(c):=|G_i|\,\mu_n^{(i)}(c)
=
\frac{|G_i|\,B^{(i)}_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(i=1,2).
\]
Then
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
+
\frac{1}{|G_2|}\sum_{c_1\in G_1}\sum_{c_2\in\pi^{-1}(c_1)}
\bigl(r_n^{(2)}(c_2)-r_n^{(1)}(c_1)\bigr)^2.
\]
Furthermore, let
\[
K^\perp:=\{\chi\in\widehat G_2:\ \chi|_K\equiv 1\}\cong \widehat G_1,
\]
and define the twisted primitive spectrum on the fine layer by
\[
B_n^{(2)}(\chi;u):=\sum_{c_2\in G_2}\chi(c_2)\,B^{(2)}_{n,c_2}(u)
\qquad(\chi\in\widehat G_2).
\]
The same decomposition may be written in Fourier form as
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2,
\]
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in K^\perp\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
Consequently, the contribution lost under passage to the quotient is
\[
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus K^\perp}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
\end{theorem}

\begin{proof}
For \(c_1\in G_1\), the pushforward relation gives
\[
r_n^{(1)}(c_1)
=
\frac{1}{|K|}\sum_{c_2\in\pi^{-1}(c_1)}r_n^{(2)}(c_2).
\]
Let
\[
\mathcal H_\pi:=\{f:G_2\to\CC:\ f \text{ is constant on each fiber of }\pi\}.
\]
The pullback \(r_n^{(1)}\circ\pi\) belongs to \(\mathcal H_\pi\), and the averaging formula shows that it is the conditional expectation of \(r_n^{(2)}\) onto \(\mathcal H_\pi\). Since conditional expectation onto a finite partition is the orthogonal projection in \(L^2(\mathrm{unif}_{G_2})\), Pythagoras' theorem gives
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}\circ\pi-1\|_{L^2(\mathrm{unif}_{G_2})}^2
+
\|r_n^{(2)}-r_n^{(1)}\circ\pi\|_{L^2(\mathrm{unif}_{G_2})}^2.
\]
For any function \(h:G_1\to\CC\),
\[
\|h\circ\pi\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\frac{1}{|G_2|}\sum_{c_1\in G_1}\sum_{c_2\in\pi^{-1}(c_1)}|h(c_1)|^2
=
\frac{1}{|G_1|}\sum_{c_1\in G_1}|h(c_1)|^2
=
\|h\|_{L^2(\mathrm{unif}_{G_1})}^2.
\]
Applying this with \(h=r_n^{(1)}-1\) shows that
\[
\|r_n^{(1)}\circ\pi-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2.
\]
Expanding the second term fiberwise yields the first identity.

For the Fourier expressions, apply Parseval's identity to the character basis of \(G_2\). The Fourier coefficients of \(r_n^{(2)}\) are
\[
\widehat r_n^{(2)}(\chi)
:=
\frac{1}{|G_2|}\sum_{c_2\in G_2}r_n^{(2)}(c_2)\,\overline{\chi(c_2)}
=
\frac{B_n^{(2)}(\overline\chi;u)}{B_n(\mathbf{1};u)}.
\]
Since the counts \(B^{(2)}_{n,c_2}(u)\) are real, one has
\[
B_n^{(2)}(\overline\chi;u)=\overline{B_n^{(2)}(\chi;u)},
\]
and hence \(\chi\) and \(\overline\chi\) contribute the same squared modulus. Thus
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}\abs{\widehat r_n^{(2)}(\chi)}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
If \(\psi\in\widehat G_1\) and \(\chi:=\psi\circ\pi\in K^\perp\), then
\[
\widehat r_n^{(1)}(\psi)
=
\frac{1}{|G_1|}\sum_{c_1\in G_1}r_n^{(1)}(c_1)\,\overline{\psi(c_1)}
=
\frac{1}{|G_2|}\sum_{c_2\in G_2}r_n^{(2)}(c_2)\,\overline{\chi(c_2)}
=
\widehat r_n^{(2)}(\chi).
\]
Thus the characters of \(G_1\) identify via pullback with \(K^\perp\subseteq\widehat G_2\), and Parseval gives
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\mathbf{1};u)^2}
\sum_{\chi\in K^\perp\setminus\{\mathbf{1}\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
Subtracting the two identities yields the final formula.
\end{proof}
