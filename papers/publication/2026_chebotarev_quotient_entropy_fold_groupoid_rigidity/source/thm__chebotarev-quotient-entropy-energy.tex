\begin{proposition}\label{thm:kernel-chebotarev-quotient-relative-entropy-chain}
Let \(\pi:G_2\twoheadrightarrow G_1\) be a surjective homomorphism of finite abelian groups, and write \(K:=\ker \pi\). Fix \(u\in(0,1]\) and \(n\ge 1\), and assume that
\[
B_n(\ind;u):=
\sum_{c_2\in G_2}B^{(2)}_{n,c_2}(u)
=
\sum_{c_1\in G_1}B^{(1)}_{n,c_1}(u)
>0,
\]
where the coarse counts are obtained by pushforward,
\[
B^{(1)}_{n,c_1}(u):=
\sum_{c_2\in\pi^{-1}(c_1)}B^{(2)}_{n,c_2}(u)
\qquad(c_1\in G_1).
\]
Define probability measures
\[
\mu_n^{(i)}(c):=\frac{B^{(i)}_{n,c}(u)}{B_n(\ind;u)}
\qquad(i=1,2),
\]
and let \(\mathrm{unif}_{G_i}\) denote the uniform law on \(G_i\). For \(c_1\in G_1\) with \(\mu_n^{(1)}(c_1)>0\), write
\[
\mu_{n,\mathrm{fib}}^{(c_1)}(c_2):=
\frac{\mu_n^{(2)}(c_2)}{\mu_n^{(1)}(c_1)}
\qquad(c_2\in\pi^{-1}(c_1))
\]
for the conditional law on the fiber \(\pi^{-1}(c_1)\). Then
\[
D\!\bigl(\mu_n^{(2)}\,\|\,\mathrm{unif}_{G_2}\bigr)
=
D\!\bigl(\mu_n^{(1)}\,\|\,\mathrm{unif}_{G_1}\bigr)
+
\sum_{c_1\in G_1}\mu_n^{(1)}(c_1)\,
D\!\bigl(\mu_{n,\mathrm{fib}}^{(c_1)}\,\|\,\mathrm{unif}_{\pi^{-1}(c_1)}\bigr),
\]
with the usual convention that fibers carrying no mass contribute \(0\).
\end{proposition}

\begin{proof}
This is simply the relative-entropy chain rule for the partition into fibers of \(\pi\), specialized to the uniform reference measures on \(G_2\), \(G_1\), and the fibers of \(\pi\). Since \(\pi\) is surjective, every fiber has cardinality \(|K|\), hence
\[
\mathrm{unif}_{G_2}(c_2)
=
\frac{1}{|G_2|}
=
\frac{1}{|G_1|\,|K|}
=
\mathrm{unif}_{G_1}(\pi(c_2))\,\mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2).
\]
Moreover \(\mu_n^{(1)}=\pi_\ast\mu_n^{(2)}\). If \(\mu_n^{(1)}(c_1)=0\), then \(\mu_n^{(2)}\) vanishes on the entire fiber above \(c_1\), so that fiber contributes nothing. On the other fibers one has
\[
\mu_n^{(2)}(c_2)
=
\mu_n^{(1)}(\pi(c_2))\,\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2).
\]
Consequently
\[
\log \frac{\mu_n^{(2)}(c_2)}{\mathrm{unif}_{G_2}(c_2)}
=
\log \frac{\mu_n^{(1)}(\pi(c_2))}{\mathrm{unif}_{G_1}(\pi(c_2))}
+
\log \frac{\mu_{n,\mathrm{fib}}^{(\pi(c_2))}(c_2)}{\mathrm{unif}_{\pi^{-1}(\pi(c_2))}(c_2)}.
\]
Multiplying by \(\mu_n^{(2)}(c_2)\), summing over \(c_2\in G_2\), and regrouping by fibers gives the claim.
\end{proof}

The same splitting appears at the quadratic level, and there it admits a particularly transparent Fourier interpretation.

\begin{proposition}\label{thm:kernel-chebotarev-visible-hidden-character-energy}
In the setting of Proposition \ref{thm:kernel-chebotarev-quotient-relative-entropy-chain}, define the relative densities
\[
r_n^{(i)}(c):=|G_i|\,\mu_n^{(i)}(c)
=
\frac{|G_i|\,B^{(i)}_{n,c}(u)}{B_n(\ind;u)}
\qquad(i=1,2).
\]
Then
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
+
\frac{1}{|G_2|}\sum_{c_1\in G_1}\sum_{c_2\in\pi^{-1}(c_1)}
\bigl(r_n^{(2)}(c_2)-r_n^{(1)}(c_1)\bigr)^2.
\]
Equivalently, \(\chi^2(\mu_n^{(2)}\Vert\mathrm{unif}_{G_2})\) splits exactly into the quotient-visible contribution and the fiberwise remainder.

Now let
\[
K^\perp:=\{\chi\in\widehat G_2:\chi|_K\equiv 1\}\cong\widehat G_1,
\]
and define
\[
B_n^{(2)}(\chi;u):=\sum_{c_2\in G_2}\chi(c_2)\,B^{(2)}_{n,c_2}(u)
\qquad(\chi\in\widehat G_2).
\]
Then
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\ind\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2,
\]
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in K^\perp\setminus\{\ind\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
In particular, the contribution lost under passage to the quotient is
\[
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in\widehat G_2\setminus K^\perp}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
\end{proposition}

\begin{proof}
For \(c_1\in G_1\) the pushforward relation may be written as
\[
r_n^{(1)}(c_1)=\frac{1}{|K|}\sum_{c_2\in\pi^{-1}(c_1)}r_n^{(2)}(c_2).
\]
Thus \(r_n^{(1)}\circ \pi\) is the conditional expectation of \(r_n^{(2)}\) onto the subspace
\[
\mathcal H_\pi:=\{f:G_2\to\CC:\ f \text{ is constant on every fiber of }\pi\}.
\]
Since conditional expectation onto a finite partition is the orthogonal projection in \(L^2(\mathrm{unif}_{G_2})\), Pythagoras' theorem gives
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\|r_n^{(1)}\circ\pi-1\|_{L^2(\mathrm{unif}_{G_2})}^2
+
\|r_n^{(2)}-r_n^{(1)}\circ\pi\|_{L^2(\mathrm{unif}_{G_2})}^2.
\]
For any \(h:G_1\to\CC\),
\[
\|h\circ\pi\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\frac{1}{|G_2|}\sum_{c_1\in G_1}\sum_{c_2\in\pi^{-1}(c_1)}|h(c_1)|^2
=
\frac{1}{|G_1|}\sum_{c_1\in G_1}|h(c_1)|^2
=
\|h\|_{L^2(\mathrm{unif}_{G_1})}^2.
\]
Applying this with \(h=r_n^{(1)}-1\) yields the first identity.

For the Fourier formulas we apply Parseval's identity on \(G_2\). The Fourier coefficients of \(r_n^{(2)}\) are
\[
\widehat r_n^{(2)}(\chi)
:=
\frac{1}{|G_2|}\sum_{c_2\in G_2}r_n^{(2)}(c_2)\,\overline{\chi(c_2)}
=
\frac{B_n^{(2)}(\overline\chi;u)}{B_n(\ind;u)}.
\]
Since the counts \(B^{(2)}_{n,c_2}(u)\) are real,
\[
B_n^{(2)}(\overline\chi;u)=\overline{B_n^{(2)}(\chi;u)},
\]
so the squared modulus is unaffected by conjugation. Hence
\[
\|r_n^{(2)}-1\|_{L^2(\mathrm{unif}_{G_2})}^2
=
\sum_{\chi\in\widehat G_2\setminus\{\ind\}}\abs{\widehat r_n^{(2)}(\chi)}^2
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in\widehat G_2\setminus\{\ind\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]

If \(\psi\in\widehat G_1\) and \(\chi:=\psi\circ\pi\in K^\perp\), then
\[
\widehat r_n^{(1)}(\psi)
=
\frac{1}{|G_1|}\sum_{c_1\in G_1}r_n^{(1)}(c_1)\,\overline{\psi(c_1)}
=
\frac{1}{|G_2|}\sum_{c_2\in G_2}r_n^{(2)}(c_2)\,\overline{\chi(c_2)}
=
\widehat r_n^{(2)}(\chi).
\]
Thus the characters of \(G_1\) identify with \(K^\perp\subseteq\widehat G_2\), and another application of Parseval gives
\[
\|r_n^{(1)}-1\|_{L^2(\mathrm{unif}_{G_1})}^2
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in K^\perp\setminus\{\ind\}}
\bigl|B_n^{(2)}(\chi;u)\bigr|^2.
\]
Subtracting the two identities gives the final formula.
\end{proof}
