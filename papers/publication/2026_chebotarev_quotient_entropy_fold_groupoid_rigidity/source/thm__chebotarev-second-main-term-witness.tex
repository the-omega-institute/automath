\begin{theorem}[Second term from a dominant channel]\label{thm:kernel-chebotarev-second-main-term-witness}
Fix a finite group \(G\) and a temperature parameter \(u\in(0,1]\) in the setting of Assumption \ref{thm:kernel-chebotarev-exp}.
Let \(\lambda_1(u)=\spr(M_{K,\ind}(u))\) be the Perron root in the trivial channel, and for each irreducible representation \(\rho\neq \ind\) write \(M_{K,\rho}(u)\) for the corresponding twisted channel matrix.
Assume that \(\eta(u)>0\) and that there exists a nontrivial channel \(\rho_\star\neq \ind\) and a complex number \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)\);
  \item every nontrivial spectral value \(\nu\) with \(\abs{\nu}=\eta(u)\) belongs to \(\{\mu_\star,\overline{\mu_\star}\}\);
  \item each value in \(\{\mu_\star,\overline{\mu_\star}\}\) that occurs in the nontrivial spectrum has algebraic multiplicity \(1\); and
  \item
  \[
  \Gamma(u):=\max\bigl\{\abs{\nu}:\ \nu\in\mathrm{spec}(M_{K,\rho}(u)),\ \rho\neq \ind,\ \nu\notin\{\mu_\star,\overline{\mu_\star}\}\bigr\}
  \]
  satisfies \(\Gamma(u)<\eta(u)\).
\end{enumerate}
Then for every conjugacy-invariant subset \(A\subseteq G\), with
\[
B_{n,A}(u):=\sum_{c\in A}B_{n,c}(u),
\]
there exists a complex coefficient \(c_A\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma(u)^n}{n}\right).
\]
If, in addition, \(c_A\neq 0\), then there exists a constant \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|\ \ge\ c(A)\,\eta(u)^n
\]
for infinitely many integers \(n\).
\end{theorem}

\begin{proof}
For each conjugacy-invariant subset \(A\subseteq G\), Assumption \ref{thm:kernel-chebotarev-exp} gives linear functionals \(L_{A,\rho}\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\frac{1}{n}\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr).
\]
For each nontrivial channel \(\rho\), let \(\Sigma_\rho\) denote the spectrum of \(M_{K,\rho}(u)\). Since the dominant spectral values are simple and separated from the remaining nontrivial spectrum by \(\Gamma(u)<\eta(u)\), the finite-dimensional spectral decomposition gives
\[
L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
\sum_{\nu\in \Sigma_\rho\cap\{\mu_\star,\overline{\mu_\star}\}}
a_{A,\rho,\nu}\,\nu^n
+O\!\bigl(\Gamma(u)^n\bigr),
\]
where \(a_{A,\rho,\nu}:=L_{A,\rho}(P_{\rho,\nu})\) and \(P_{\rho,\nu}\) is the rank-one spectral projector of \(M_{K,\rho}(u)\) associated with \(\nu\). The error term is uniform in \(\rho\) because \(\widehat G\) is finite. Summing over the finitely many nontrivial channels yields
\[
\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
a_A\,\mu_\star^n+b_A\,\overline{\mu_\star}^{\,n}
+O\!\bigl(\Gamma(u)^n\bigr)
\]
for suitable coefficients \(a_A,b_A\in\CC\).

Suppose first that \(\mu_\star\notin\RR\). Since \(B_{n,A}(u)\in\RR\) for every \(n\), comparison with the complex conjugate of the preceding asymptotic formula gives
\[
(a_A-\overline{b_A})\,\mu_\star^n+(b_A-\overline{a_A})\,\overline{\mu_\star}^{\,n}
=
O\!\bigl(\Gamma(u)^n\bigr).
\]
After dividing by \(\overline{\mu_\star}^{\,n}\), we obtain
\[
(a_A-\overline{b_A})\left(\frac{\mu_\star}{\overline{\mu_\star}}\right)^n
+
(b_A-\overline{a_A})
=
O\!\left(\left(\frac{\Gamma(u)}{\eta(u)}\right)^n\right).
\]
If \(a_A\neq \overline{b_A}\), then the left-hand side is either periodic and nonzero infinitely often, or dense in a nondegenerate circle, according as \(\arg(\mu_\star)/\pi\) is rational or irrational. In either case it cannot converge to \(0\), because \(\Gamma(u)<\eta(u)\). Hence \(b_A=\overline{a_A}\). The dominant contribution may therefore be written as
\[
a_A\,\mu_\star^n+\overline{a_A\,\mu_\star^n}
=
\Re\!\bigl(c_A\mu_\star^n\bigr),
\qquad c_A:=2a_A.
\]
If \(\mu_\star\in\RR\), then the dominant term is a real scalar multiple of \(\mu_\star^n\), and we choose \(c_A\in\RR\) to be that scalar. Substituting this into the subset-count expansion gives
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma(u)^n}{n}\right).
\]

Assume now that \(c_A\neq 0\). Write \(\mu_\star=\eta(u)e^{i\theta}\) and \(c_A=|c_A|e^{i\phi}\). Then
\[
\Re(c_A\mu_\star^n)=|c_A|\,\eta(u)^n\cos(\phi+n\theta).
\]
If \(\theta/\pi\in\QQ\), then the sequence \(\cos(\phi+n\theta)\) is periodic. Because \(c_A\neq 0\) by hypothesis, this periodic sequence is not identically zero, so one of its nonzero values recurs infinitely often. If \(\theta/\pi\notin\QQ\), then the set \(\{e^{in\theta}:n\ge 1\}\) is dense in \(\TT\), so there are infinitely many \(n\) for which \(|\cos(\phi+n\theta)|\ge 1/2\). In both cases there exists \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|
\ge c(A)\,\eta(u)^n
\]
for infinitely many \(n\).
\end{proof}
