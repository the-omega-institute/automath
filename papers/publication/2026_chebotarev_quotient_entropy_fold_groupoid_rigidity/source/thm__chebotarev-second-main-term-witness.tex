\begin{theorem}[Second main term and witness sets from a dominant channel]\label{thm:kernel-chebotarev-second-main-term-witness}
Fix a finite group \(G\) and a temperature parameter \(u\in(0,1]\) in the setting of Assumption \ref{thm:kernel-chebotarev-exp}.
Let \(\lambda_1(u)=\spr(M_{K,\ind}(u))\) be the Perron root in the trivial channel, and for each irreducible representation \(\rho\neq \ind\) write \(M_{K,\rho}(u)\) for the corresponding twisted channel matrix.
Assume that there exists a nontrivial channel \(\rho_\star\neq \ind\) and a complex number \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)\);
  \item the nontrivial spectral values of modulus \(\eta(u)\) consist exactly of the pair \(\{\mu_\star,\overline{\mu_\star}\}\), counted with algebraic multiplicity, where the two values may coincide if \(\mu_\star\in\RR\);
  \item each dominant spectral value in \(\{\mu_\star,\overline{\mu_\star}\}\) is simple; and
  \item
  \[
  \Gamma(u):=\max\bigl\{\abs{\nu}:\ \nu\in\mathrm{spec}(M_{K,\rho}(u)),\ \rho\neq \ind,\ \nu\notin\{\mu_\star,\overline{\mu_\star}\}\bigr\}
  \]
  satisfies \(\Gamma(u)<\eta(u)\).
\end{enumerate}
Then for every conjugacy-invariant subset \(A\subseteq G\), with
\[
B_{n,A}(u):=\sum_{c\in A}B_{n,c}(u),
\]
there exists a complex coefficient \(c_A\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma(u)^n}{n}\right).
\]
If, in addition, \(c_A\neq 0\), then there exists a constant \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|\ \ge\ c(A)\,\eta(u)^n
\]
for infinitely many integers \(n\).
\end{theorem}

\begin{proof}
For each conjugacy-invariant subset \(A\subseteq G\), Assumption \ref{thm:kernel-chebotarev-exp} provides linear functionals \(L_{A,\rho}\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\frac{1}{n}\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr).
\]
For each nontrivial channel \(\rho\), let \(\Sigma_\rho\) denote the spectrum of \(M_{K,\rho}(u)\). Because the dominant spectral values are simple and isolated from the rest of the nontrivial spectrum by the strict inequality \(\Gamma(u)<\eta(u)\), the finite-dimensional spectral decomposition gives
\[
L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
\sum_{\nu\in \Sigma_\rho\cap\{\mu_\star,\overline{\mu_\star}\}}
a_{A,\rho,\nu}\,\nu^n
+O\!\bigl(\Gamma(u)^n\bigr),
\]
where \(a_{A,\rho,\nu}:=L_{A,\rho}(P_{\rho,\nu})\) and \(P_{\rho,\nu}\) is the rank-one spectral projector of \(M_{K,\rho}(u)\) associated with \(\nu\). Summing over the finitely many nontrivial channels yields
\[
\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
a_A\,\mu_\star^n+b_A\,\overline{\mu_\star}^{\,n}
+O\!\bigl(\Gamma(u)^n\bigr)
\]
for suitable coefficients \(a_A,b_A\in\CC\).

If \(\mu_\star\notin\RR\), then the left-hand side is real for every \(n\), so necessarily \(b_A=\overline{a_A}\). Hence the dominant contribution may be written as
\[
a_A\,\mu_\star^n+\overline{a_A\,\mu_\star^n}
=
\Re\!\bigl(c_A\mu_\star^n\bigr),
\qquad c_A:=2a_A.
\]
If \(\mu_\star\in\RR\), then the dominant part is a real scalar multiple of \(\mu_\star^n\); in that case we choose \(c_A\in\RR\) to be that scalar. Substituting this into the subset-count expansion proves
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma(u)^n}{n}\right).
\]

It remains to prove the witness lower bound when \(c_A\neq 0\). Write \(\mu_\star=\eta(u)e^{i\theta}\) and \(c_A=|c_A|e^{i\phi}\). Then
\[
\Re(c_A\mu_\star^n)=|c_A|\,\eta(u)^n\cos(\phi+n\theta).
\]
If \(\theta/\pi\in\QQ\), then the sequence \(\cos(\phi+n\theta)\) is periodic. Because \(c_A\neq 0\) by hypothesis, this periodic sequence is not identically zero, so one of its nonzero values recurs infinitely often. If \(\theta/\pi\notin\QQ\), then the set \(\{e^{in\theta}:n\ge 1\}\) is dense in \(\TT\), so there are infinitely many \(n\) for which \(|\cos(\phi+n\theta)|\ge 1/2\). In both cases there exists \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|
\ge c(A)\,\eta(u)^n
\]
for infinitely many \(n\), as claimed.
\end{proof}
