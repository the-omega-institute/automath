\begin{theorem}\label{thm:kernel-chebotarev-second-main-term-witness}
Let \(G\) be a finite group and let \(u\in(0,1]\) satisfy Assumption \ref{thm:kernel-chebotarev-exp}. Write \(\lambda_1(u)=\spr(M_{K,\ind}(u))\) for the Perron root in the trivial channel, and let \(M_{K,\rho}(u)\) denote the twisted channel matrix attached to a nontrivial irreducible representation \(\rho\neq\ind\). Assume that \(\eta(u)>0\), and that there exist a nontrivial channel \(\rho_\star\neq\ind\) and a spectral value \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)\);
  \item every nontrivial spectral value \(\nu\) with \(\abs{\nu}=\eta(u)\) belongs to \(\{\mu_\star,\overline{\mu_\star}\}\);
  \item whenever one of the values in \(\{\mu_\star,\overline{\mu_\star}\}\) occurs in a nontrivial channel, it occurs there with algebraic multiplicity \(1\); and
  \item
  \[
  \Gamma(u):=\max\Bigl(\bigl\{\abs{\nu}:\ \nu\in\mathrm{spec}(M_{K,\rho}(u)),\ \rho\neq\ind,\ \nu\notin\{\mu_\star,\overline{\mu_\star}\}\bigr\}\cup\{0\}\Bigr)
  \]
  satisfies \(\Gamma(u)<\eta(u)\).
\end{enumerate}
Then for every conjugacy-invariant subset \(A\subseteq G\), with
\[
B_{n,A}(u):=\sum_{c\in A}B_{n,c}(u),
\]
there exists a complex constant \(c_A\) such that, for every fixed \(\Gamma_+\) with
\[
\Gamma(u)<\Gamma_+<\eta(u),
\]
one has
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma_+^n}{n}\right).
\]
If \(c_A\neq 0\), then there exists a constant \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|\ge c(A)\,\eta(u)^n
\]
for infinitely many integers \(n\).
\end{theorem}

\begin{proof}
Fix a conjugacy-invariant subset \(A\subseteq G\). By Assumption \ref{thm:kernel-chebotarev-exp}, there are linear functionals \(L_{A,\rho}\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\frac{1}{n}\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr).
\]
Fix \(\Gamma_+\) with \(\Gamma(u)<\Gamma_+<\eta(u)\). For each nontrivial channel \(\rho\), the only eigenvalues of modulus \(\eta(u)\) are among \(\{\mu_\star,\overline{\mu_\star}\}\), and they are simple whenever they occur. It follows from Jordan normal form that there are spectral projectors \(P_{\rho,\nu}\) for \(\nu\in\mathrm{spec}(M_{K,\rho}(u))\cap\{\mu_\star,\overline{\mu_\star}\}\) and matrices \(R_{\rho,n}\) such that
\[
M_{K,\rho}(u)^n
=
\sum_{\nu\in\mathrm{spec}(M_{K,\rho}(u))\cap\{\mu_\star,\overline{\mu_\star}\}}
\nu^nP_{\rho,\nu}
+
R_{\rho,n},
\qquad
\|R_{\rho,n}\|=O\!\bigl(\Gamma_+^n\bigr).
\]
Indeed, every Jordan block contributing to \(R_{\rho,n}\) has eigenvalue modulus at most \(\Gamma(u)\), and its \(n\)-th power is therefore \(O(\Gamma_+^n)\) for every fixed \(\Gamma_+>\Gamma(u)\); see \cite[Chapter 6]{HornJohnson2013}.

Applying \(L_{A,\rho}\) gives
\[
L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
\sum_{\nu\in\mathrm{spec}(M_{K,\rho}(u))\cap\{\mu_\star,\overline{\mu_\star}\}}
a_{A,\rho,\nu}\,\nu^n
+
O\!\bigl(\Gamma_+^n\bigr),
\]
where \(a_{A,\rho,\nu}:=L_{A,\rho}(P_{\rho,\nu})\). Summing over the finitely many nontrivial channels yields constants \(a_A,b_A\in\CC\), with \(b_A=0\) if \(\mu_\star\in\RR\), such that
\[
\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
a_A\,\mu_\star^n+b_A\,\overline{\mu_\star}^{\,n}
+
O\!\bigl(\Gamma_+^n\bigr).
\]

Assume first that \(\mu_\star\notin\RR\). Since \(B_{n,A}(u)\) is real for every \(n\), comparison with the complex conjugate asymptotic shows that
\[
(a_A-\overline{b_A})\,\mu_\star^n
+
(b_A-\overline{a_A})\,\overline{\mu_\star}^{\,n}
=
O\!\bigl(\Gamma_+^n\bigr).
\]
After division by \(\overline{\mu_\star}^{\,n}\), this becomes
\[
(a_A-\overline{b_A})\left(\frac{\mu_\star}{\overline{\mu_\star}}\right)^n
+
(b_A-\overline{a_A})
=
O\!\left(\left(\frac{\Gamma_+}{\eta(u)}\right)^n\right).
\]
If \(a_A\neq\overline{b_A}\), then the left-hand side is a nonconstant affine function of the unimodular sequence \((\mu_\star/\overline{\mu_\star})^n\), and such a sequence cannot converge to \(0\): it is periodic if \(\arg(\mu_\star)/\pi\in\QQ\), and dense in \(\TT\) otherwise. Hence \(b_A=\overline{a_A}\), and the dominant contribution may be written as
\[
a_A\,\mu_\star^n+\overline{a_A\,\mu_\star^n}
=
\Re\!\bigl(c_A\mu_\star^n\bigr),
\qquad
c_A:=2a_A.
\]

If \(\mu_\star\in\RR\), then the dominant term has the form
\[
a_A\,\mu_\star^n+O\!\bigl(\Gamma_+^n\bigr).
\]
Since the left-hand side is real-valued for all \(n\), taking imaginary parts gives
\[
\Im(a_A)\,\mu_\star^n=O\!\bigl(\Gamma_+^n\bigr).
\]
Because \(|\mu_\star|=\eta(u)>\Gamma_+\), division by \(\mu_\star^n\) and passage to the limit show that \(\Im(a_A)=0\). Thus \(a_A\in\RR\), and we set \(c_A:=a_A\). Substituting into the expansion for \(B_{n,A}(u)\) gives
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma_+^n}{n}\right).
\]

Assume now that \(c_A\neq 0\). Writing \(\mu_\star=\eta(u)e^{i\theta}\) and \(c_A=|c_A|e^{i\phi}\), we obtain
\[
\Re(c_A\mu_\star^n)=|c_A|\,\eta(u)^n\cos(\phi+n\theta).
\]
If \(\theta/\pi\in\QQ\), then \(\cos(\phi+n\theta)\) is periodic and not identically zero; if \(\theta/\pi\notin\QQ\), then the sequence \((e^{in\theta})_{n\ge 1}\) is dense in \(\TT\), so \(|\cos(\phi+n\theta)|\ge 1/2\) for infinitely many \(n\). In both cases there is a constant \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|\ge c(A)\,\eta(u)^n
\]
for infinitely many \(n\).
\end{proof}
