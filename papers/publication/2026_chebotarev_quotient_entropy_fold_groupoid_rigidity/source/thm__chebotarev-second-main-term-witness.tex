\begin{theorem}\label{thm:kernel-chebotarev-second-main-term-witness}
Let \(G\) be a finite group and let \(u\in(0,1]\) be as in Assumption \ref{thm:kernel-chebotarev-exp}. Write \(\lambda_1(u)=\spr(M_{K,\ind}(u))\) for the Perron root in the trivial channel, and for each irreducible representation \(\rho\neq \ind\) let \(M_{K,\rho}(u)\) denote the corresponding twisted channel matrix. Assume that \(\eta(u)>0\) and that there exist a nontrivial channel \(\rho_\star\neq \ind\) and a complex number \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)\);
  \item every nontrivial spectral value \(\nu\) with \(\abs{\nu}=\eta(u)\) belongs to \(\{\mu_\star,\overline{\mu_\star}\}\);
  \item whenever one of the values in \(\{\mu_\star,\overline{\mu_\star}\}\) occurs in the spectrum of a nontrivial channel matrix \(M_{K,\rho}(u)\), it occurs there with algebraic multiplicity \(1\); and
  \item
  \[
  \Gamma(u):=\max\Bigl(\bigl\{\abs{\nu}:\ \nu\in\mathrm{spec}(M_{K,\rho}(u)),\ \rho\neq \ind,\ \nu\notin\{\mu_\star,\overline{\mu_\star}\}\bigr\}\cup\{0\}\Bigr)
  \]
  satisfies \(\Gamma(u)<\eta(u)\).
\end{enumerate}
Then for every conjugacy-invariant subset \(A\subseteq G\), with
\[
B_{n,A}(u):=\sum_{c\in A}B_{n,c}(u),
\]
there exists a complex coefficient \(c_A\) such that, for every fixed \(\Gamma_+\) satisfying
\[
\Gamma(u)<\Gamma_+<\eta(u),
\]
one has
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma_+^n}{n}\right).
\]
If, in addition, \(c_A\neq 0\), then there exists a constant \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|\ \ge\ c(A)\,\eta(u)^n
\]
for infinitely many integers \(n\).
\end{theorem}

\begin{proof}
Fix a conjugacy-invariant subset \(A\subseteq G\). By Assumption \ref{thm:kernel-chebotarev-exp}, there exist linear functionals \(L_{A,\rho}\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\frac{1}{n}\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr).
\]
Fix \(\Gamma_+\) with \(\Gamma(u)<\Gamma_+<\eta(u)\). For each nontrivial channel \(\rho\), let \(\Sigma_\rho\) denote the spectrum of \(M_{K,\rho}(u)\). Since the only nontrivial spectral values of modulus \(\eta(u)\) are \(\mu_\star\) and \(\overline{\mu_\star}\), and each such value is simple in every channel where it occurs, Jordan normal form yields spectral projectors \(P_{\rho,\nu}\) for \(\nu\in\Sigma_\rho\cap\{\mu_\star,\overline{\mu_\star}\}\) and remainder matrices \(R_{\rho,n}\) such that
\[
M_{K,\rho}(u)^n
=
\sum_{\nu\in \Sigma_\rho\cap\{\mu_\star,\overline{\mu_\star}\}}
\nu^nP_{\rho,\nu}
+R_{\rho,n},
\qquad
\|R_{\rho,n}\|=O\!\bigl(\Gamma_+^n\bigr).
\]
see, for example, \cite[Chapter 6]{HornJohnson2013}. Applying \(L_{A,\rho}\) gives
\[
L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
\sum_{\nu\in \Sigma_\rho\cap\{\mu_\star,\overline{\mu_\star}\}}
a_{A,\rho,\nu}\,\nu^n
+O\!\bigl(\Gamma_+^n\bigr),
\]
where \(a_{A,\rho,\nu}:=L_{A,\rho}(P_{\rho,\nu})\). Since \(\widehat G\) is finite, the implied constant may be chosen uniformly in \(\rho\). Summing over the nontrivial channels therefore yields coefficients \(a_A,b_A\in\CC\), with \(b_A=0\) when \(\mu_\star\in\RR\), such that
\[
\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
a_A\,\mu_\star^n+b_A\,\overline{\mu_\star}^{\,n}
+O\!\bigl(\Gamma_+^n\bigr).
\]

Assume first that \(\mu_\star\notin\RR\). Since \(B_{n,A}(u)\in\RR\) for every \(n\), comparison with the complex conjugate of the preceding asymptotic formula gives
\[
(a_A-\overline{b_A})\,\mu_\star^n+(b_A-\overline{a_A})\,\overline{\mu_\star}^{\,n}
=
O\!\bigl(\Gamma_+^n\bigr).
\]
After dividing by \(\overline{\mu_\star}^{\,n}\), we obtain
\[
(a_A-\overline{b_A})\left(\frac{\mu_\star}{\overline{\mu_\star}}\right)^n
+
(b_A-\overline{a_A})
=
O\!\left(\left(\frac{\Gamma_+}{\eta(u)}\right)^n\right).
\]
If \(a_A\neq \overline{b_A}\), then the left-hand side is either periodic and nonzero infinitely often or dense in a nondegenerate circle, according as \(\arg(\mu_\star)/\pi\) is rational or irrational. In either case it cannot converge to \(0\), because \(\Gamma_+<\eta(u)\) and hence the right-hand side tends to \(0\). Thus \(b_A=\overline{a_A}\), and the dominant contribution may be written as
\[
a_A\,\mu_\star^n+\overline{a_A\,\mu_\star^n}
=
\Re\!\bigl(c_A\mu_\star^n\bigr),
\qquad c_A:=2a_A.
\]
If \(\mu_\star\in\RR\), then the dominant contribution has the form
\[
a_A\,\mu_\star^n+O\!\bigl(\Gamma_+^n\bigr).
\]
Since the left-hand side is real-valued for every \(n\), taking imaginary parts gives
\[
\Im(a_A)\,\mu_\star^n=O\!\bigl(\Gamma_+^n\bigr).
\]
Because \(|\mu_\star|=\eta(u)>\Gamma_+\), division by \(\mu_\star^n\) and passage to the limit force \(\Im(a_A)=0\). Hence \(a_A\in\RR\), and we set \(c_A:=a_A\). Substituting into the subset-count expansion gives
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma_+^n}{n}\right).
\]

Assume now that \(c_A\neq 0\). Write \(\mu_\star=\eta(u)e^{i\theta}\) and \(c_A=|c_A|e^{i\phi}\). Then
\[
\Re(c_A\mu_\star^n)=|c_A|\,\eta(u)^n\cos(\phi+n\theta).
\]
If \(\theta/\pi\in\QQ\), then the sequence \(\cos(\phi+n\theta)\) is periodic. Because \(c_A\neq 0\) by hypothesis, this periodic sequence is not identically zero, so one of its nonzero values recurs infinitely often. If \(\theta/\pi\notin\QQ\), then the set \(\{e^{in\theta}:n\ge 1\}\) is dense in \(\TT\), so there are infinitely many \(n\) for which \(|\cos(\phi+n\theta)|\ge 1/2\). In both cases there exists \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|
\ge c(A)\,\eta(u)^n
\]
for infinitely many \(n\).
\end{proof}
