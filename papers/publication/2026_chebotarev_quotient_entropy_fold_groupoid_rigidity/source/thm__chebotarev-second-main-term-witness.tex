\begin{theorem}\label{thm:kernel-chebotarev-second-main-term-witness}
Fix a finite group \(G\) and \(u\in(0,1]\) in the setting of Assumption \ref{thm:kernel-chebotarev-exp}. Let \(\lambda_1(u)=\spr(M_{K,\ind}(u))\) be the Perron root in the trivial channel, and write \(M_{K,\rho}(u)\) for the twisted matrix attached to a nontrivial irreducible representation \(\rho\neq \ind\). Assume that there exist a nontrivial channel \(\rho_\star\neq \ind\) and a complex number \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)>0\);
  \item among all nontrivial channels, the spectral values of modulus \(\eta(u)\) are exactly \(\{\mu_\star,\overline{\mu_\star}\}\), counted with algebraic multiplicity, where the two values may coincide if \(\mu_\star\in\RR\);
  \item each dominant spectral value in \(\{\mu_\star,\overline{\mu_\star}\}\) is simple;
  \item
  \[
  \Gamma(u):=\max\bigl\{\abs{\nu}:\nu\in\mathrm{spec}(M_{K,\rho}(u)),\ \rho\neq \ind,\ \nu\notin\{\mu_\star,\overline{\mu_\star}\}\bigr\}
  \]
  satisfies \(\Gamma(u)<\eta(u)\).
\end{enumerate}
Then, for every conjugacy-invariant subset \(A\subseteq G\),
\[
B_{n,A}(u):=\sum_{c\in A}B_{n,c}(u)
\]
admits an expansion
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re\!\bigl(c_A\,\mu_\star^{\,n}\bigr)\frac{1}{n}
\;+\;
O\!\left(\frac{\Gamma(u)^n}{n}\right)
\]
for some coefficient \(c_A\in\CC\). If \(c_A\neq 0\), then there is a constant \(c(A)>0\) such that
\[
\bigl|\Re(c_A\mu_\star^{\,n})\bigr|\ge c(A)\,\eta(u)^n
\]
for infinitely many integers \(n\).
\end{theorem}

\begin{proof}
By Assumption \ref{thm:kernel-chebotarev-exp}, for each conjugacy-invariant subset \(A\subseteq G\) there are linear functionals \(L_{A,\rho}\) such that
\[
B_{n,A}(u)
=
\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}
\;+\;
\frac{1}{n}\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr).
\]
Fix \(\rho\neq \ind\), and write \(T_\rho:=M_{K,\rho}(u)\). The Jordan decomposition of \(T_\rho\), together with the spectral hypotheses, gives
\[
T_\rho^n
=
\sum_{\omega\in\mathrm{spec}(T_\rho)\cap\{\mu_\star,\overline{\mu_\star}\}}\omega^n P_{\rho,\omega}
+R_{\rho,n},
\]
where \(P_{\rho,\omega}\) is the spectral projector onto the one-dimensional eigenspace for \(\omega\), and \(\|R_{\rho,n}\|=O(\Gamma(u)^n)\) for any fixed matrix norm. Since there are only finitely many nontrivial channels, summing over \(\rho\neq \ind\) yields coefficients \(a_{A,\omega}\in\CC\) for which
\[
S_n:=\sum_{\rho\in\widehat G\setminus\{\ind\}}L_{A,\rho}\!\bigl(M_{K,\rho}(u)^n\bigr)
=
\sum_{\omega\in\{\mu_\star,\overline{\mu_\star}\}}a_{A,\omega}\,\omega^n
+O(\Gamma(u)^n).
\]
Because \(B_{n,A}(u)\) and \(\mathrm{unif}_G(A)\lambda_1(u)^n/n\) are real, each \(S_n\) is real.

If \(\mu_\star\notin\RR\), then conjugating the previous expansion and subtracting gives
\[
\bigl(a_{A,\mu_\star}-\overline{a_{A,\overline{\mu_\star}}}\bigr)\mu_\star^n
+
\bigl(a_{A,\overline{\mu_\star}}-\overline{a_{A,\mu_\star}}\bigr)\overline{\mu_\star}^{\,n}
=
O(\Gamma(u)^n).
\]
After division by \(\overline{\mu_\star}^{\,n}\), this becomes
\[
\bigl(a_{A,\mu_\star}-\overline{a_{A,\overline{\mu_\star}}}\bigr)
\left(\frac{\mu_\star}{\overline{\mu_\star}}\right)^n
+
\bigl(a_{A,\overline{\mu_\star}}-\overline{a_{A,\mu_\star}}\bigr)
=
O\!\left(\left(\frac{\Gamma(u)}{\eta(u)}\right)^n\right).
\]
Since \(\Gamma(u)<\eta(u)\), the right-hand side tends to \(0\). The sequence \((\mu_\star/\overline{\mu_\star})^n\) is either periodic or dense in a subgroup of \(\TT\); in either case the only way a sequence of the form \(c_1z_n+c_2\) can converge to \(0\) is that \(c_1=c_2=0\). Therefore
\[
a_{A,\overline{\mu_\star}}=\overline{a_{A,\mu_\star}}.
\]
Setting \(c_A:=2a_{A,\mu_\star}\) gives
\[
S_n
=
\Re\!\bigl(c_A\mu_\star^n\bigr)+O(\Gamma(u)^n).
\]
If \(\mu_\star\in\RR\), then the dominant set consists only of \(\mu_\star\), and the reality of \(S_n\) shows that the corresponding coefficient is real. The same formula therefore holds with \(c_A:=a_{A,\mu_\star}\in\RR\). Substituting into the expression for \(B_{n,A}(u)\) proves the asserted expansion.

Finally, write \(c_A=\abs{c_A}e^{i\phi}\) and \(\mu_\star=\eta(u)e^{i\theta}\). Then
\[
\Re(c_A\mu_\star^n)=\abs{c_A}\,\eta(u)^n\cos(\phi+n\theta).
\]
If \(\theta/\pi\in\QQ\), the sequence \((\cos(\phi+n\theta))_{n\ge 1}\) is periodic, hence attains a nonzero value infinitely often because \(c_A\neq 0\). If \(\theta/\pi\notin\QQ\), the sequence \((\phi+n\theta)\bmod 2\pi\) is dense in \(\TT\), so there are infinitely many \(n\) with \(\abs{\cos(\phi+n\theta)}\ge 1/2\). In either case there exists \(c(A)>0\) such that
\[
\abs{\Re(c_A\mu_\star^n)}\ge c(A)\,\eta(u)^n
\]
for infinitely many \(n\), as claimed.
\end{proof}
