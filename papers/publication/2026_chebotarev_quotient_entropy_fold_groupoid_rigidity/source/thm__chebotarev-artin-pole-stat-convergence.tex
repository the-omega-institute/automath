\begin{theorem}[Artin 因子化下第一非平凡奇点的精确阶数]\label{thm:kernel-artin-dominant-channel-pole-order}
在命题 \ref{prop:kernel-artin-factorization} 的记号下，固定有限群 \(G\)、温度参数 \(u\in(0,1]\)，并设存在某个非平凡不可约表示 \(\rho_\star\neq \mathbf{1}\) 与特征值 \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) 满足
\begin{enumerate}
  \item \(|\mu_\star|=\Lambda(u):=\max_{\rho\neq \mathbf{1}}\rho(M_{K,\rho}(u))\)；
  \item \(\mu_\star\) 在通道 \(\rho_\star\) 中的代数重数等于 \(1\)；
  \item 对所有 \(\rho\neq \rho_\star\) 以及 \(\rho_\star\) 内除 \(\mu_\star\) 外的其余谱点 \(\nu\)，均有 \(|\nu|<\Lambda(u)\)。
\end{enumerate}
则
\[
\boxed{\;
\frac{\zeta_{K,G}(z,u)}{L_{K,\mathbf{1}}(z,u)}
\text{ 在 }z=\mu_\star^{-1}\text{ 处具有精确极点阶 }\dim\rho_\star\; }.
\]
进一步，若 \(A\subseteq G\) 为共轭不变集合，\(U=\pi^{-1}(A)\) 为相应 cylinder 事件，并且 \(1_A\) 在 \(\rho_\star\)-通道上的 Fourier 分量非零，则存在常数 \(c_A\neq 0\) 与 \(\eta>0\) 使
\[
B_{n,U}(u)
=
\mu_{\mathrm{Haar}}(U)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re(c_A\mu_\star^n)\frac{1}{n}
\;+\;
O\!\left(\frac{(\Lambda(u)-\eta)^n}{n}\right),
\]
并且存在常数 \(c(A)>0\) 与无穷多个 \(n\) 使
\[
\left|B_{n,U}(u)-\mu_{\mathrm{Haar}}(U)\frac{\lambda_1(u)^n}{n}\right|
\ge
c(A)\frac{\Lambda(u)^n}{n}.
\]
\end{theorem}

\begin{proof}
由命题 \ref{prop:kernel-artin-factorization}，
\[
\frac{\zeta_{K,G}(z,u)}{L_{K,\mathbf{1}}(z,u)}
=
\prod_{\rho\in\widehat G\setminus\{\mathbf{1}\}}L_{K,\rho}(z,u)^{\dim\rho}.
\]
对 \(\rho=\rho_\star\)，由于 \(\mu_\star\) 是代数重数 \(1\) 的特征值，故
\[
\det(I-zM_{K,\rho_\star}(u))
\]
在 \(z=\mu_\star^{-1}\) 处有一个简单零点，从而 \(L_{K,\rho_\star}(z,u)\) 在该点有简单极点；再取 \(\dim\rho_\star\) 次幂，即得到阶数恰为 \(\dim\rho_\star\) 的极点。
另一方面，对所有其余非平凡通道 \(\rho\neq \rho_\star\) 以及 \(\rho_\star\) 内除 \(\mu_\star\) 外的其余谱点，由假设均有模长严格小于 \(\Lambda(u)=|\mu_\star|\)，故相应因子在 \(z=\mu_\star^{-1}\) 处解析且非零，不会改变极点阶数。于是得到所示精确阶数。
\par
后半部分正是定理 \ref{thm:kernel-chebotarev-second-main-term-witness} 的适用情形：该定理给出了 dominant nontrivial channel 的二主项展开与见证集合振荡。这里只是把其结论与 Artin 因子化的极点阶数粘合到同一条解析链上。证毕。
\end{proof}

\begin{theorem}[GRH 型谱隙的统计后果：TV/\texorpdfstring{$\chi^2$}{chi2}/KL 的指数收敛与 witness 反证]\label{thm:kernel-chebotarev-statistical-convergence}
设 \(G\) 为有限阿贝尔群，固定 \(u\in(0,1]\)，并对充分大的 \(n\) 定义 primitive Chebotarev 标签分布
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(c\in G),
\]
对任意子集 \(A\subseteq G\) 仍写
\[
\mu_{n,u}(A):=\sum_{c\in A}\mu_{n,u}(c),
\qquad
\mathrm{unif}_G(A):=\frac{|A|}{|G|},
\]
记 \(G\) 上的均匀分布为 \(\mathrm{unif}_G\)，并令
\[
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}<1,
\qquad
\Lambda(u):=\max\{|\lambda_2(u)|,\eta(u)\}.
\]
则有统一的统计收敛界
\[
\boxed{\;
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\tau(u)^n\bigr)\; },
\]
\[
\boxed{\;
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr)\; },
\]
\[
\boxed{\;
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr)\; }.
\]
若进一步满足定义 \ref{def:kernel-rh} 的平方根型谱隙条件
\[
\textsf{GRH}_{K,G}(u):\qquad \eta(u)\le \lambda_1(u)^{1/2},
\]
则
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr),
\]
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr),
\qquad
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr).
\]
反之，若定理 \ref{thm:kernel-artin-dominant-channel-pole-order} 的 dominant channel 假设成立，且 \(\Lambda(u)>\lambda_1(u)^{1/2}\)，则存在共轭不变集合 \(A\subseteq G\) 与常数 \(c(A)>0\)，使得沿无穷多个 \(n\) 有
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c(A)\,\frac{\Lambda(u)^n}{\lambda_1(u)^n}.
\]
特别地，此时不可能成立
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr).
\]
\end{theorem}

\begin{proof}
由定理 \ref{thm:kernel-chebotarev-exp}，对每个 \(c\in G\) 有一致渐近式
\[
B_{n,c}(u)
=
\frac{1}{|G|}\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right),
\qquad
B_n(\mathbf{1};u)
=
\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right).
\]
两式相除即得
\[
\mu_{n,u}(c)=\frac{1}{|G|}+O\!\bigl(\tau(u)^n\bigr)
\]
在 \(c\in G\) 上一致成立。
于是
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
\frac12\sum_{c\in G}\left|\mu_{n,u}(c)-\frac{1}{|G|}\right|
=
O\!\bigl(\tau(u)^n\bigr),
\]
得到 total variation 估计。
\par
再令相对密度
\[
r_n(c):=|G|\mu_{n,u}(c)=1+O\!\bigl(\tau(u)^n\bigr)
\]
一致成立。则
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
对于 KL 散度，因 \(r_n(c)\to 1\) 一致成立，故对充分大的 \(n\) 有 \(|r_n(c)-1|\le 1/2\)。对 \(|x|\le 1/2\) 使用标准估计
\[
(1+x)\log(1+x)\le 2x^2,
\]
于是
\[
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}r_n(c)\log r_n(c)
\le
\frac{2}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
这就得到三重统计收敛律。
\par
若进一步满足 \(\textsf{GRH}_{K,G}(u)\)，则 \(\tau(u)\le \lambda_1(u)^{-1/2}\)，把它代入上式即得平方根级上界。
\par
反之，若定理 \ref{thm:kernel-artin-dominant-channel-pole-order} 的假设成立，则该定理给出某个共轭不变集合 \(A\subseteq G\) 使得沿无穷多个 \(n\)
\[
\left|\sum_{c\in A}B_{n,c}(u)-\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}\right|
\ge
c(A)\frac{\Lambda(u)^n}{n}.
\]
又由于
\[
B_n(\mathbf{1};u)=\frac{\lambda_1(u)^n}{n}\bigl(1+o(1)\bigr),
\]
两边同除以 \(B_n(\mathbf{1};u)\) 得
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c'(A)\,\frac{\Lambda(u)^n}{\lambda_1(u)^n}
\]
沿无穷多个 \(n\) 成立，其中 \(c'(A)>0\)。由于 total variation 支配任意事件概率差值，而当 \(\Lambda(u)>\lambda_1(u)^{1/2}\) 时
\[
\left(\frac{\Lambda(u)}{\lambda_1(u)}\right)^n
\gg
\lambda_1(u)^{-n/2},
\]
故不可能再有
\(
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O(\lambda_1(u)^{-n/2})
\)。
证毕。
\end{proof}
