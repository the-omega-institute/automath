\begin{theorem}[Exact order of the first nontrivial singularity under Artin factorization]\label{thm:kernel-artin-dominant-channel-pole-order}
In the notation of Proposition \ref{prop:kernel-artin-factorization}, fix a finite group \(G\), a temperature parameter \(u\in(0,1]\), and assume that there exists a nontrivial irreducible representation \(\rho_\star\neq \mathbf{1}\) with an eigenvalue \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\Lambda(u):=\max_{\rho\neq \mathbf{1}}\rho(M_{K,\rho}(u))\);
  \item \(\mu_\star\) has algebraic multiplicity \(1\) in the channel \(\rho_\star\);
  \item for every \(\rho\neq \rho_\star\), and for every other spectral value \(\nu\) of \(M_{K,\rho_\star}(u)\) with \(\nu\neq \mu_\star\), one has \(\abs{\nu}<\Lambda(u)\).
\end{enumerate}
Then
\[
\boxed{\;
\frac{\zeta_{K,G}(z,u)}{L_{K,\mathbf{1}}(z,u)}
\text{ has a pole of exact order }\dim\rho_\star\text{ at }z=\mu_\star^{-1}\; }.
\]
Moreover, if \(A\subseteq G\) is a conjugacy-invariant subset, \(U=\pi^{-1}(A)\) is the corresponding cylinder event, and the Fourier component of \(1_A\) in the \(\rho_\star\)-channel is nonzero, then there exist constants \(c_A\neq 0\) and \(\eta>0\) such that
\[
B_{n,U}(u)
=
\mu_{\mathrm{Haar}}(U)\frac{\lambda_1(u)^n}{n}
\;+\;
\Re(c_A\mu_\star^n)\frac{1}{n}
\;+\;
O\!\left(\frac{(\Lambda(u)-\eta)^n}{n}\right),
\]
and there exists a constant \(c(A)>0\) such that for infinitely many \(n\),
\[
\left|B_{n,U}(u)-\mu_{\mathrm{Haar}}(U)\frac{\lambda_1(u)^n}{n}\right|
\ge
c(A)\frac{\Lambda(u)^n}{n}.
\]
\end{theorem}

\begin{proof}
By Proposition \ref{prop:kernel-artin-factorization},
\[
\frac{\zeta_{K,G}(z,u)}{L_{K,\mathbf{1}}(z,u)}
=
\prod_{\rho\in\widehat G\setminus\{\mathbf{1}\}}L_{K,\rho}(z,u)^{\dim\rho}.
\]
For \(\rho=\rho_\star\), the assumption that \(\mu_\star\) has algebraic multiplicity \(1\) implies that
\[
\det(I-zM_{K,\rho_\star}(u))
\]
has a simple zero at \(z=\mu_\star^{-1}\). Consequently \(L_{K,\rho_\star}(z,u)\) has a simple pole there, and taking the \(\dim\rho_\star\)-th power yields a pole of exact order \(\dim\rho_\star\).
For every other nontrivial channel \(\rho\neq \rho_\star\), and for every spectral value \(\nu\neq \mu_\star\) within the channel \(\rho_\star\), the hypothesis gives \(\abs{\nu}<\Lambda(u)=\abs{\mu_\star}\). Hence the corresponding factors are analytic and nonzero at \(z=\mu_\star^{-1}\), so they do not alter the pole order. This proves the first assertion.

The second assertion is precisely the situation covered by Theorem \ref{thm:kernel-chebotarev-second-main-term-witness}. That theorem supplies the second main term and the witness-set oscillation for a dominant nontrivial channel, and the Artin factorization above identifies the same channel analytically through the exact pole order.
\end{proof}

\begin{theorem}[Statistical consequences of a GRH-type spectral gap: exponential convergence in TV/\texorpdfstring{$\chi^2$}{chi2}/KL and witness obstructions]\label{thm:kernel-chebotarev-statistical-convergence}
Let \(G\) be a finite abelian group, fix \(u\in(0,1]\), and for all sufficiently large \(n\) define the primitive Chebotarev label distribution
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\mathbf{1};u)}
\qquad(c\in G).
\]
For any subset \(A\subseteq G\), write
\[
\mu_{n,u}(A):=\sum_{c\in A}\mu_{n,u}(c),
\qquad
\mathrm{unif}_G(A):=\frac{\abs{A}}{\abs{G}},
\]
and let \(\mathrm{unif}_G\) denote the uniform measure on \(G\). Define
\[
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}<1,
\qquad
\Lambda(u):=\max\{\abs{\lambda_2(u)},\eta(u)\}.
\]
Then one has the uniform statistical convergence bounds
\[
\boxed{\;
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\tau(u)^n\bigr)\; },
\]
\[
\boxed{\;
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr)\; },
\]
\[
\boxed{\;
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr)\; }.
\]
If, in addition, the square-root spectral-gap condition of Definition \ref{def:kernel-rh} holds,
\[
\textsf{GRH}_{K,G}(u):\qquad \eta(u)\le \lambda_1(u)^{1/2},
\]
then
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr),
\]
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr),
\qquad
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr).
\]
Conversely, if the dominant-channel hypotheses of Theorem \ref{thm:kernel-artin-dominant-channel-pole-order} hold and \(\Lambda(u)>\lambda_1(u)^{1/2}\), then there exist a conjugacy-invariant subset \(A\subseteq G\) and a constant \(c(A)>0\) such that for infinitely many \(n\),
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c(A)\,\frac{\Lambda(u)^n}{\lambda_1(u)^n}.
\]
In particular, one cannot have
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr)
\]
in that regime.
\end{theorem}

\begin{proof}
The spectral expansion of Theorem \ref{thm:kernel-chebotarev-exp} gives, uniformly in \(c\in G\),
\[
B_{n,c}(u)
=
\frac{1}{|G|}\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right),
\qquad
B_n(\mathbf{1};u)
=
\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right).
\]
Taking the ratio of these two asymptotics yields
\[
\mu_{n,u}(c)=\frac{1}{|G|}+O\!\bigl(\tau(u)^n\bigr)
\]
uniformly for \(c\in G\).
Therefore,
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
\frac12\sum_{c\in G}\left|\mu_{n,u}(c)-\frac{1}{|G|}\right|
=
O\!\bigl(\tau(u)^n\bigr),
\]
which proves the total-variation estimate.

Now define the relative density
\[
r_n(c):=|G|\mu_{n,u}(c)=1+O\!\bigl(\tau(u)^n\bigr),
\]
again uniformly in \(c\).
Then
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
For the Kullback-Leibler divergence, uniform convergence \(r_n(c)\to 1\) implies that \(|r_n(c)-1|\le 1/2\) for all sufficiently large \(n\). Using the elementary estimate
\[
(1+x)\log(1+x)\le 2x^2
\qquad(\abs{x}\le 1/2),
\]
we obtain
\[
D_{\mathrm{KL}}\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}r_n(c)\log r_n(c)
\le
\frac{2}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
This proves the triple exponential convergence law.

If \(\textsf{GRH}_{K,G}(u)\) holds, then \(\tau(u)\le \lambda_1(u)^{-1/2}\), and substitution into the previous bounds gives the square-root decay estimates.

For the converse statement, Theorem \ref{thm:kernel-artin-dominant-channel-pole-order} provides a conjugacy-invariant subset \(A\subseteq G\) such that, for infinitely many \(n\),
\[
\left|\sum_{c\in A}B_{n,c}(u)-\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}\right|
\ge
c(A)\frac{\Lambda(u)^n}{n}.
\]
Since
\[
B_n(\mathbf{1};u)=\frac{\lambda_1(u)^n}{n}\bigl(1+o(1)\bigr),
\]
dividing both sides by \(B_n(\mathbf{1};u)\) yields
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c'(A)\,\frac{\Lambda(u)^n}{\lambda_1(u)^n}
\]
along infinitely many \(n\), for some \(c'(A)>0\).
Because total variation dominates the discrepancy of any event, and because
\[
\left(\frac{\Lambda(u)}{\lambda_1(u)}\right)^n
\gg
\lambda_1(u)^{-n/2}
\qquad\text{when }\Lambda(u)>\lambda_1(u)^{1/2},
\]
the square-root bound in total variation cannot hold in this regime.
\end{proof}
