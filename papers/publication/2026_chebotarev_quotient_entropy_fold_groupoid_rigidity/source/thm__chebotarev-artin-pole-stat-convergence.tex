\begin{theorem}[Exact order of the first nontrivial singularity under Artin factorization]\label{thm:kernel-artin-dominant-channel-pole-order}
In the notation of Assumption \ref{prop:kernel-artin-factorization}, fix a finite group \(G\), a temperature parameter \(u\in(0,1]\), and assume that there exists a nontrivial irreducible representation \(\rho_\star\neq \ind\) with an eigenvalue \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)\);
  \item \(\mu_\star\) has algebraic multiplicity \(1\) in the channel \(\rho_\star\); and
  \item every other nontrivial spectral value \(\nu\neq \mu_\star\) satisfies \(\abs{\nu}<\eta(u)\).
\end{enumerate}
Then
\[
\boxed{\;
\frac{\zeta_{K,G}(z,u)}{L_{K,\ind}(z,u)}
\text{ has a pole of exact order }\dim\rho_\star\text{ at }z=\mu_\star^{-1}\; }.
\]
\end{theorem}

\begin{proof}
By Assumption \ref{prop:kernel-artin-factorization},
\[
\frac{\zeta_{K,G}(z,u)}{L_{K,\ind}(z,u)}
=
\prod_{\rho\in\widehat G\setminus\{\ind\}}L_{K,\rho}(z,u)^{\dim\rho}.
\]
For \(\rho=\rho_\star\), the assumption that \(\mu_\star\) has algebraic multiplicity \(1\) implies that
\[
\det(I-zM_{K,\rho_\star}(u))
\]
has a simple zero at \(z=\mu_\star^{-1}\). Consequently \(L_{K,\rho_\star}(z,u)\) has a simple pole there, and taking the \(\dim\rho_\star\)-th power yields a pole of exact order \(\dim\rho_\star\). For every other nontrivial channel \(\rho\neq \rho_\star\), and for every other spectral value \(\nu\neq \mu_\star\) inside the channel \(\rho_\star\), the hypothesis gives \(\abs{\nu}<\eta(u)=\abs{\mu_\star}\). Hence the corresponding factors are analytic and nonzero at \(z=\mu_\star^{-1}\), so they do not alter the pole order.
\end{proof}

\begin{theorem}[Statistical consequences of the spectral gap]\label{thm:kernel-chebotarev-statistical-convergence}
Let \(G\) be a finite abelian group, fix \(u\in(0,1]\), and for all sufficiently large \(n\) define the primitive Chebotarev label distribution
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\ind;u)}
\qquad(c\in G).
\]
For any subset \(A\subseteq G\), write
\[
\mu_{n,u}(A):=\sum_{c\in A}\mu_{n,u}(c),
\qquad
\mathrm{unif}_G(A):=\frac{\abs{A}}{\abs{G}},
\]
and let \(\mathrm{unif}_G\) denote the uniform measure on \(G\). Define
\[
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}<1,
\qquad
\Lambda(u):=\max\{\abs{\lambda_2(u)},\eta(u)\}.
\]
Then one has the uniform statistical convergence bounds
\[
\boxed{\;
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\tau(u)^n\bigr)\; },
\]
\[
\boxed{\;
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr)\; },
\]
\[
\boxed{\;
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr)\; }.
\]
If, in addition, the square-root spectral-gap condition of Definition \ref{def:kernel-rh} holds, then
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr),
\]
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr),
\qquad
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr).
\]
Conversely, suppose there exists a subset \(A\subseteq G\) for which the hypotheses of Theorem \ref{thm:kernel-chebotarev-second-main-term-witness} hold with \(c_A\neq 0\). If \(\eta(u)>\lambda_1(u)^{1/2}\), then there exists a constant \(c(A)>0\) such that for infinitely many \(n\),
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c(A)\,\frac{\eta(u)^n}{\lambda_1(u)^n}.
\]
In particular, one cannot have
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr)
\]
in that regime.
\end{theorem}

\begin{proof}
Assumption \ref{thm:kernel-chebotarev-exp} gives, uniformly in \(c\in G\),
\[
B_{n,c}(u)
=
\frac{1}{|G|}\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right),
\qquad
B_n(\ind;u)
=
\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right).
\]
Equivalently,
\[
B_{n,c}(u)
=
\frac{\lambda_1(u)^n}{n}\left(\frac{1}{|G|}+O\!\bigl(\tau(u)^n\bigr)\right),
\qquad
B_n(\ind;u)
=
\frac{\lambda_1(u)^n}{n}\left(1+O\!\bigl(\tau(u)^n\bigr)\right).
\]
For all sufficiently large \(n\), the second factor is positive and bounded away from \(0\), so its reciprocal is \(1+O(\tau(u)^n)\). Hence
\[
\mu_{n,u}(c)=\frac{1}{|G|}+O\!\bigl(\tau(u)^n\bigr)
\]
uniformly for \(c\in G\). Therefore,
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
\frac12\sum_{c\in G}\left|\mu_{n,u}(c)-\frac{1}{|G|}\right|
=
O\!\bigl(\tau(u)^n\bigr),
\]
which proves the total-variation estimate.

Now define the relative density
\[
r_n(c):=|G|\mu_{n,u}(c)=1+O\!\bigl(\tau(u)^n\bigr),
\]
again uniformly in \(c\). Then
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
For the relative entropy, the inequality \(\log t\le t-1\) for \(t>0\) yields
\[
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}r_n(c)\log r_n(c)
\le
\frac{1}{|G|}\sum_{c\in G}r_n(c)\bigl(r_n(c)-1\bigr).
\]
Because \(\frac{1}{|G|}\sum_{c\in G}r_n(c)=1\), the right-hand side equals
\[
\frac{1}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
This proves the triple exponential convergence law.

If \(\textsf{GRH}_{K,G}(u)\) holds, then \(\tau(u)\le \lambda_1(u)^{-1/2}\), and substitution into the previous bounds gives the square-root decay estimates.

For the converse statement, fix a subset \(A\subseteq G\) as in the hypothesis. Theorem \ref{thm:kernel-chebotarev-second-main-term-witness} gives
\[
\left|B_{n,A}(u)-\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}\right|
\ge
c(A)\frac{\eta(u)^n}{n}
\]
for infinitely many \(n\), after possibly decreasing \(c(A)\). Since
\[
B_n(\ind;u)=\frac{\lambda_1(u)^n}{n}\bigl(1+o(1)\bigr),
\]
dividing both sides by \(B_n(\ind;u)\) yields
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c'(A)\,\frac{\eta(u)^n}{\lambda_1(u)^n}
\]
along infinitely many \(n\), for some \(c'(A)>0\). Because total variation dominates the discrepancy of any event, and because
\[
\left(\frac{\eta(u)}{\lambda_1(u)}\right)^n
\gg
\lambda_1(u)^{-n/2}
\qquad\text{when }\eta(u)>\lambda_1(u)^{1/2},
\]
the square-root bound in total variation cannot hold in this regime.
\end{proof}
