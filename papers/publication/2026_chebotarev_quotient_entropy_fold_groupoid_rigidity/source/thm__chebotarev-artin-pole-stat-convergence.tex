\begin{theorem}\label{thm:kernel-artin-dominant-channel-pole-order}
In the notation of Assumption \ref{prop:kernel-artin-factorization}, fix a finite group \(G\) and \(u\in(0,1]\). Assume that there is a nontrivial irreducible representation \(\rho_\star\neq \ind\) with an eigenvalue \(\mu_\star\in\mathrm{spec}(M_{K,\rho_\star}(u))\) such that
\begin{enumerate}
  \item \(\abs{\mu_\star}=\eta(u)>0\);
  \item \(\mu_\star\) has algebraic multiplicity \(1\) in the channel \(\rho_\star\);
  \item for every nontrivial channel \(\rho\neq \rho_\star\), one has \(\mu_\star\notin\mathrm{spec}(M_{K,\rho}(u))\), and every nontrivial spectral value \(\nu\neq \mu_\star\) satisfies \(\abs{\nu}<\eta(u)\).
\end{enumerate}
Then
\[
\frac{\zeta_{K,G}(z,u)}{L_{K,\ind}(z,u)}
\]
has a pole of exact order \(\dim\rho_\star\) at \(z=\mu_\star^{-1}\).
\end{theorem}

\begin{proof}
Assumption \ref{prop:kernel-artin-factorization} gives
\[
\frac{\zeta_{K,G}(z,u)}{L_{K,\ind}(z,u)}
=
\prod_{\rho\in\widehat G\setminus\{\ind\}}L_{K,\rho}(z,u)^{\dim\rho}.
\]
For \(\rho=\rho_\star\), the eigenvalue \(\mu_\star\) is simple, so \(\det(I-zM_{K,\rho_\star}(u))\) has a simple zero at \(z=\mu_\star^{-1}\). Thus \(L_{K,\rho_\star}(z,u)\) has a simple pole there, and its \(\dim\rho_\star\)-th power has a pole of exact order \(\dim\rho_\star\).

For every other nontrivial channel \(\rho\neq \rho_\star\), the hypothesis implies that \(\mu_\star\notin\mathrm{spec}(M_{K,\rho}(u))\), hence \(L_{K,\rho}(z,u)\) is analytic and nonvanishing at \(z=\mu_\star^{-1}\). The same is true for the factors corresponding to the remaining eigenvalues in the channel \(\rho_\star\). These factors therefore do not alter the pole order.
\end{proof}

\begin{theorem}\label{thm:kernel-chebotarev-statistical-convergence}
Let \(G\) be a finite abelian group, fix \(u\in(0,1]\), and for all sufficiently large \(n\) define
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\ind;u)}
\qquad(c\in G).
\]
For a subset \(A\subseteq G\), write
\[
\mu_{n,u}(A):=\sum_{c\in A}\mu_{n,u}(c),
\qquad
\mathrm{unif}_G(A):=\frac{\abs{A}}{\abs{G}},
\]
and let \(\mathrm{unif}_G\) denote the uniform measure on \(G\). Set
\[
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}<1,
\qquad
\Lambda(u):=\max\{\abs{\lambda_2(u)},\eta(u)\}.
\]
Then
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\tau(u)^n\bigr),
\]
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr),
\]
\[
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
If, in addition, the square-root spectral-gap condition of Definition \ref{def:kernel-rh} holds, then
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
O\!\bigl(\lambda_1(u)^{-n/2}\bigr),
\]
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr),
\qquad
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
O\!\bigl(\lambda_1(u)^{-n}\bigr).
\]
Conversely, suppose that there is a subset \(A\subseteq G\) for which the hypotheses of Theorem \ref{thm:kernel-chebotarev-second-main-term-witness} hold with \(c_A\neq 0\). If \(\eta(u)>\lambda_1(u)^{1/2}\), then there exists \(c(A)>0\) such that, for infinitely many \(n\),
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
c(A)\,\frac{\eta(u)^n}{\lambda_1(u)^n}.
\]
In particular,
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
\neq
O\!\bigl(\lambda_1(u)^{-n/2}\bigr)
\]
in that regime.
\end{theorem}

\begin{proof}
Assumption \ref{thm:kernel-chebotarev-exp} gives, uniformly in \(c\in G\),
\[
B_{n,c}(u)
=
\frac{1}{|G|}\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right),
\qquad
B_n(\ind;u)
=
\frac{\lambda_1(u)^n}{n}
+O\!\left(\frac{\Lambda(u)^n}{n}\right).
\]
Equivalently,
\[
B_n(\ind;u)=\frac{\lambda_1(u)^n}{n}\bigl(1+O(\tau(u)^n)\bigr),
\]
so
\[
\frac{1}{B_n(\ind;u)}
=
\frac{n}{\lambda_1(u)^n}\bigl(1+O(\tau(u)^n)\bigr)
\]
for all sufficiently large \(n\). Multiplying by the asymptotic formula for \(B_{n,c}(u)\) yields
\[
\mu_{n,u}(c)=\frac{1}{|G|}+O\!\bigl(\tau(u)^n\bigr)
\]
uniformly in \(c\in G\). Summing over \(c\) gives
\[
\|\mu_{n,u}-\mathrm{unif}_G\|_{\mathrm{TV}}
=
\frac12\sum_{c\in G}\left|\mu_{n,u}(c)-\frac{1}{|G|}\right|
=
O\!\bigl(\tau(u)^n\bigr).
\]

Now set
\[
r_n(c):=|G|\mu_{n,u}(c)=1+O\!\bigl(\tau(u)^n\bigr)
\]
uniformly in \(c\). Then
\[
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
=
\frac{1}{|G|}\sum_{c\in G}(r_n(c)-1)^2
=
O\!\bigl(\tau(u)^{2n}\bigr).
\]
For the relative entropy we use the standard bound
\[
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)
\le
\log\!\Bigl(1+\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)\Bigr)
\le
\chi^2\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr),
\]
and therefore
\[
D\!\bigl(\mu_{n,u}\,\|\,\mathrm{unif}_G\bigr)=O\!\bigl(\tau(u)^{2n}\bigr).
\]

If \(\textsf{GRH}_{K,G}(u)\) holds, then \(\tau(u)\le \lambda_1(u)^{-1/2}\), and substituting this into the previous bounds gives the square-root estimates.

For the converse statement, fix \(A\subseteq G\) as in the hypothesis. Since \(G\) is abelian, every subset is conjugacy-invariant, so Theorem \ref{thm:kernel-chebotarev-second-main-term-witness} yields
\[
\left|B_{n,A}(u)-\mathrm{unif}_G(A)\frac{\lambda_1(u)^n}{n}\right|
\ge
c(A)\frac{\eta(u)^n}{n}
\]
for infinitely many \(n\), after possibly decreasing \(c(A)\). Because
\[
B_n(\ind;u)=\frac{\lambda_1(u)^n}{n}\bigl(1+O(\tau(u)^n)\bigr),
\]
there exists \(n_0\) such that
\[
\frac{1}{B_n(\ind;u)}\ge \frac{n}{2\lambda_1(u)^n}
\qquad(n\ge n_0).
\]
Restricting to the infinite subsequence furnished above and to \(n\ge n_0\), division by \(B_n(\ind;u)\) gives
\[
\bigl|\mu_{n,u}(A)-\mathrm{unif}_G(A)\bigr|
\ge
\frac{c(A)}{2}\,\frac{\eta(u)^n}{\lambda_1(u)^n}
\]
for infinitely many \(n\). Since total variation dominates the discrepancy on any fixed subset, and
\[
\left(\frac{\eta(u)}{\lambda_1(u)}\right)^n
\gg
\lambda_1(u)^{-n/2}
\qquad\text{when }\eta(u)>\lambda_1(u)^{1/2},
\]
the square-root bound in total variation cannot hold.
\end{proof}
