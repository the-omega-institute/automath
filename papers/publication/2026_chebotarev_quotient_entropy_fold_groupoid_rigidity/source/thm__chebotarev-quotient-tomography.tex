\begin{theorem}\label{thm:kernel-chebotarev-quotient-lattice-tomography}
Let \(G\) be a finite abelian group, fix \(u\in(0,1]\), and let
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\ind;u)}
\qquad(c\in G)
\]
be the primitive Chebotarev label distribution for all sufficiently large \(n\), so that \(B_n(\ind;u)>0\). For a subgroup \(H\le G\), write
\[
\pi_H:G\twoheadrightarrow G/H,
\qquad
\mu_{n,u}^H:=(\pi_H)_\ast\mu_{n,u},
\]
and define the quotient-visible energy
\[
\mathcal E_{n,u}(H)
:=
\chi^2\!\bigl(\mu_{n,u}^H\,\|\,\mathrm{unif}_{G/H}\bigr).
\]
Define also the exact-kernel spectral energy
\[
\mathcal W_{n,u}(H)
:=
\frac{1}{B_n(\ind;u)^2}
\sum_{\substack{\chi\in\widehat G\setminus\{\ind\}\\ \ker\chi=H}}
\bigl|B_n(\chi;u)\bigr|^2.
\]
Then
\[
\mathcal E_{n,u}(H)=\sum_{J\supseteq H}\mathcal W_{n,u}(J).
\]
Equivalently, the quotient-visible \(\chi^2\)-profile is the cumulative exact-kernel spectral mass on the subgroup lattice.

Let \(\mu_{\mathrm{sub}}\) denote the Möbius function of the subgroup lattice of \(G\) \cite{Rota1964}. Then
\[
\mathcal W_{n,u}(H)
=
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal E_{n,u}(J).
\]
In particular, the full family \(\{\mathcal E_{n,u}(H)\}_{H\le G}\) determines exactly the nontrivial Fourier energy carried by characters with kernel \(H\).
\end{theorem}

\begin{proof}
Fix \(H\le G\). Applying Theorem \ref{thm:kernel-chebotarev-visible-hidden-character-energy} to the quotient map \(\pi_H:G\twoheadrightarrow G/H\) gives
\[
\mathcal E_{n,u}(H)
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in H^\perp\setminus\{\ind\}}
\bigl|B_n(\chi;u)\bigr|^2,
\]
where
\[
H^\perp=\{\chi\in\widehat G:\chi|_H\equiv 1\}.
\]
For a character \(\chi\in\widehat G\), the condition \(\chi|_H\equiv 1\) is equivalent to \(H\subseteq\ker\chi\). Therefore
\[
\sum_{\chi\in H^\perp\setminus\{\ind\}}
\bigl|B_n(\chi;u)\bigr|^2
=
\sum_{J\supseteq H}
\sum_{\substack{\chi\in\widehat G\setminus\{\ind\}\\ \ker\chi=J}}
\bigl|B_n(\chi;u)\bigr|^2,
\]
which is exactly the first displayed formula.

Now the subgroup lattice of \(G\) is finite, hence locally finite, so Möbius inversion applies. Since
\[
\mathcal E_{n,u}(H)=\sum_{J\supseteq H}\mathcal W_{n,u}(J),
\]
the inversion formula on the poset \((\{H\le G\},\subseteq)\) yields
\[
\mathcal W_{n,u}(H)
=
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal E_{n,u}(J).
\]
This proves the claim.
\end{proof}

\begin{theorem}\label{thm:kernel-chebotarev-entropy-tomography}
Let \(G\), \(u\), and \(\mu_{n,u}\) be as in Theorem \ref{thm:kernel-chebotarev-quotient-lattice-tomography}, and assume moreover that Assumption \ref{thm:kernel-chebotarev-exp} holds. For \(H\le G\), define the quotient relative entropy
\[
\mathcal D_{n,u}(H)
:=
D\!\bigl(\mu_{n,u}^H\,\|\,\mathrm{unif}_{G/H}\bigr).
\]
Then, uniformly in \(H\le G\),
\[
\mathcal D_{n,u}(H)
=
\frac12\,\mathcal E_{n,u}(H)
+
O\!\bigl(\tau(u)^{3n}\bigr),
\qquad
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}<1.
\]
Consequently,
\[
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal D_{n,u}(J)
=
\frac12\,\mathcal W_{n,u}(H)
+
O\!\bigl(\tau(u)^{3n}\bigr)
\]
uniformly in \(H\le G\).
Thus the family of quotient relative entropies recovers the exact-kernel spectral energy to second order.
\end{theorem}

\begin{proof}
Fix \(H\le G\), and write \(Q_H:=G/H\). For \(\bar c\in Q_H\), set
\[
B^{H}_{n,\bar c}(u):=\sum_{c\in \pi_H^{-1}(\bar c)}B_{n,c}(u).
\]
Since each fiber of \(\pi_H\) has cardinality \(|H|\), Assumption \ref{thm:kernel-chebotarev-exp} implies
\[
B^{H}_{n,\bar c}(u)
=
\frac{1}{|Q_H|}\frac{\lambda_1(u)^n}{n}
+
O\!\left(\frac{\Lambda(u)^n}{n}\right)
\]
uniformly in \(\bar c\in Q_H\), while
\[
B_n(\ind;u)
=
\frac{\lambda_1(u)^n}{n}
+
O\!\left(\frac{\Lambda(u)^n}{n}\right).
\]
Hence
\[
\mu_{n,u}^H(\bar c)
=
\frac{1}{|Q_H|}
+
O\!\bigl(\tau(u)^n\bigr)
\]
uniformly in both \(H\) and \(\bar c\), because \(G\) has only finitely many subgroups. Define the relative density
\[
r_{n,u}^H(\bar c):=|Q_H|\,\mu_{n,u}^H(\bar c)
=
1+\delta_{n,u}^H(\bar c),
\qquad
\delta_{n,u}^H(\bar c)=O\!\bigl(\tau(u)^n\bigr).
\]
Since \(\mu_{n,u}^H\) is a probability measure,
\[
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\delta_{n,u}^H(\bar c)=0.
\]

For all sufficiently large \(n\), uniformly in \(H\), one has \(\|\delta_{n,u}^H\|_\infty\le 1/2\). On \([-1/2,1/2]\), the Taylor expansion of \((1+t)\log(1+t)\) gives
\[
(1+t)\log(1+t)=t+\frac{t^2}{2}+O(t^3).
\]
Applying this with \(t=\delta_{n,u}^H(\bar c)\), averaging over \(Q_H\), and using the vanishing of the linear term yields
\[
\begin{aligned}
\mathcal D_{n,u}(H)
&=
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}
\bigl(1+\delta_{n,u}^H(\bar c)\bigr)
\log\!\bigl(1+\delta_{n,u}^H(\bar c)\bigr)\\
&=
\frac{1}{2|Q_H|}\sum_{\bar c\in Q_H}\delta_{n,u}^H(\bar c)^2
+
O\!\left(\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\abs{\delta_{n,u}^H(\bar c)}^3\right).
\end{aligned}
\]
Because \(|Q_H|\le |G|\) and \(\|\delta_{n,u}^H\|_\infty=O(\tau(u)^n)\) uniformly in \(H\),
\[
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\abs{\delta_{n,u}^H(\bar c)}^3
=
O\!\bigl(\tau(u)^{3n}\bigr).
\]
The quadratic term is exactly \(\frac12\mathcal E_{n,u}(H)\), because
\[
\mathcal E_{n,u}(H)
=
\chi^2\!\bigl(\mu_{n,u}^H\,\|\,\mathrm{unif}_{Q_H}\bigr)
=
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\delta_{n,u}^H(\bar c)^2.
\]
This proves the first asserted asymptotic.

Finally, Theorem \ref{thm:kernel-chebotarev-quotient-lattice-tomography} gives
\[
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal E_{n,u}(J)
=
\mathcal W_{n,u}(H).
\]
Applying the first part to each \(J\supseteq H\) and using the finiteness of the subgroup lattice shows that the Möbius transform of the error term is still \(O(\tau(u)^{3n})\), uniformly in \(H\). This yields the second formula.
\end{proof}
