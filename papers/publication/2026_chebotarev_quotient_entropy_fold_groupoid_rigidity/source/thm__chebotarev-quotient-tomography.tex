\begin{theorem}\label{thm:kernel-chebotarev-quotient-lattice-tomography}
Let \(G\) be a finite abelian group, fix \(u\in(0,1]\), and let
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\ind;u)}
\qquad(c\in G)
\]
be the primitive Chebotarev label distribution for all sufficiently large \(n\), so that \(B_n(\ind;u)>0\). For a subgroup \(H\le G\), write
\[
\pi_H:G\twoheadrightarrow G/H,
\qquad
\mu_{n,u}^H:=(\pi_H)_\ast\mu_{n,u},
\]
and define the quotient-visible energy
\[
\mathcal E_{n,u}(H)
:=
\chi^2\!\bigl(\mu_{n,u}^H\,\|\,\mathrm{unif}_{G/H}\bigr).
\]
Define also the exact-kernel spectral energy
\[
\mathcal W_{n,u}(H)
:=
\frac{1}{B_n(\ind;u)^2}
\sum_{\substack{\chi\in\widehat G\setminus\{\ind\}\\ \ker\chi=H}}
\bigl|B_n(\chi;u)\bigr|^2.
\]
Then
\[
\mathcal E_{n,u}(H)=\sum_{J\supseteq H}\mathcal W_{n,u}(J).
\]
In other words, the quotient-visible \(\chi^2\)-profile is the cumulative exact-kernel spectral mass on the subgroup lattice.

If \(\mu_{\mathrm{sub}}\) denotes the Möbius function of the subgroup lattice of \(G\) \cite{Rota1964}, then
\[
\mathcal W_{n,u}(H)
=
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal E_{n,u}(J).
\]
In particular, the full family \(\{\mathcal E_{n,u}(H)\}_{H\le G}\) determines the nontrivial Fourier energy carried by characters with kernel \(H\).
\end{theorem}

\begin{proof}
Fix \(H\le G\). Proposition \ref{thm:kernel-chebotarev-visible-hidden-character-energy}, applied to the quotient map \(\pi_H:G\twoheadrightarrow G/H\), gives
\[
\mathcal E_{n,u}(H)
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in H^\perp\setminus\{\ind\}}
\bigl|B_n(\chi;u)\bigr|^2,
\qquad
H^\perp:=\{\chi\in\widehat G:\chi|_H\equiv 1\}.
\]
For a character \(\chi\), the condition \(\chi|_H\equiv 1\) is equivalent to \(H\subseteq\ker\chi\). Hence
\[
\sum_{\chi\in H^\perp\setminus\{\ind\}}
\bigl|B_n(\chi;u)\bigr|^2
=
\sum_{J\supseteq H}
\sum_{\substack{\chi\in\widehat G\setminus\{\ind\}\\ \ker\chi=J}}
\bigl|B_n(\chi;u)\bigr|^2,
\]
which is exactly the first formula.

Since the subgroup lattice of \(G\) is finite, Möbius inversion applies immediately to the relation
\[
\mathcal E_{n,u}(H)=\sum_{J\supseteq H}\mathcal W_{n,u}(J).
\]
This yields
\[
\mathcal W_{n,u}(H)
=
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal E_{n,u}(J),
\]
as claimed.
\end{proof}

\begin{theorem}\label{thm:kernel-chebotarev-entropy-tomography}
Let \(G\), \(u\), and \(\mu_{n,u}\) be as in Theorem \ref{thm:kernel-chebotarev-quotient-lattice-tomography}, and assume in addition that Assumption \ref{thm:kernel-chebotarev-exp} holds. For \(H\le G\), define
\[
\mathcal D_{n,u}(H)
:=
D\!\bigl(\mu_{n,u}^H\,\|\,\mathrm{unif}_{G/H}\bigr).
\]
Then, uniformly in \(H\le G\),
\[
\mathcal D_{n,u}(H)
=
\frac12\,\mathcal E_{n,u}(H)
+
O\!\bigl(\tau(u)^{3n}\bigr),
\qquad
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}<1.
\]
Consequently,
\[
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal D_{n,u}(J)
=
\frac12\,\mathcal W_{n,u}(H)
+
O\!\bigl(\tau(u)^{3n}\bigr)
\]
uniformly in \(H\). Thus the family of quotient relative entropies recovers the exact-kernel spectral energy to second order.
\end{theorem}

\begin{proof}
Fix \(H\le G\), and set \(Q_H:=G/H\). For \(\bar c\in Q_H\), let
\[
B^{H}_{n,\bar c}(u):=\sum_{c\in \pi_H^{-1}(\bar c)}B_{n,c}(u).
\]
Because every fiber of \(\pi_H\) has cardinality \(|H|\), Assumption \ref{thm:kernel-chebotarev-exp} gives
\[
B^{H}_{n,\bar c}(u)
=
\frac{1}{|Q_H|}\frac{\lambda_1(u)^n}{n}
+
O\!\left(\frac{\Lambda(u)^n}{n}\right)
\]
uniformly in \(\bar c\in Q_H\), and at the same time
\[
B_n(\ind;u)
=
\frac{\lambda_1(u)^n}{n}
+
O\!\left(\frac{\Lambda(u)^n}{n}\right).
\]
Since \(G\) has only finitely many subgroups, it follows that
\[
\mu_{n,u}^H(\bar c)=\frac{1}{|Q_H|}+O\!\bigl(\tau(u)^n\bigr)
\]
uniformly in both \(H\) and \(\bar c\). Writing
\[
r_{n,u}^H(\bar c):=|Q_H|\,\mu_{n,u}^H(\bar c)=1+\delta_{n,u}^H(\bar c),
\]
we therefore have \(\delta_{n,u}^H(\bar c)=O(\tau(u)^n)\) uniformly in \(H\) and \(\bar c\), and
\[
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\delta_{n,u}^H(\bar c)=0
\]
because \(\mu_{n,u}^H\) is a probability measure.

For \(n\) sufficiently large one has \(\|\delta_{n,u}^H\|_\infty\le 1/2\), uniformly in \(H\). On \([-1/2,1/2]\) the Taylor expansion
\[
(1+t)\log(1+t)=t+\frac{t^2}{2}+O(t^3)
\]
holds with a uniform implied constant. Applying this pointwise with \(t=\delta_{n,u}^H(\bar c)\), averaging over \(Q_H\), and using the vanishing of the linear term gives
\[
\begin{aligned}
\mathcal D_{n,u}(H)
&=
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}
\bigl(1+\delta_{n,u}^H(\bar c)\bigr)
\log\!\bigl(1+\delta_{n,u}^H(\bar c)\bigr)\\
&=
\frac{1}{2|Q_H|}\sum_{\bar c\in Q_H}\delta_{n,u}^H(\bar c)^2
+
O\!\left(\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\abs{\delta_{n,u}^H(\bar c)}^3\right).
\end{aligned}
\]
Since \(|Q_H|\le |G|\) and \(\|\delta_{n,u}^H\|_\infty=O(\tau(u)^n)\) uniformly,
\[
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\abs{\delta_{n,u}^H(\bar c)}^3
=
O\!\bigl(\tau(u)^{3n}\bigr).
\]
The quadratic term is exactly \(\frac12\mathcal E_{n,u}(H)\), because
\[
\mathcal E_{n,u}(H)
=
\chi^2\!\bigl(\mu_{n,u}^H\,\|\,\mathrm{unif}_{Q_H}\bigr)
=
\frac{1}{|Q_H|}\sum_{\bar c\in Q_H}\delta_{n,u}^H(\bar c)^2.
\]
This proves the first assertion.

Applying the first assertion to every \(J\supseteq H\) and then using Theorem \ref{thm:kernel-chebotarev-quotient-lattice-tomography}, we obtain
\[
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal D_{n,u}(J)
=
\frac12
\sum_{J\supseteq H}\mu_{\mathrm{sub}}(H,J)\,\mathcal E_{n,u}(J)
+
O\!\bigl(\tau(u)^{3n}\bigr).
\]
The Möbius transform of the main term is \(\mathcal W_{n,u}(H)\), and the error remains \(O(\tau(u)^{3n})\) because the subgroup lattice is finite. This gives the second formula.
\end{proof}
