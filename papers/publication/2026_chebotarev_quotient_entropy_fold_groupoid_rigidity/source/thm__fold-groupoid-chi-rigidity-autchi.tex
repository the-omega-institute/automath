\begin{proposition}\label{thm:fold-groupoid-chi-rigidity}
With the notation of Proposition \ref{prop:fold-groupoid-z2x2-sector-by-orientation}, fix a Wedderburn decomposition
\[
\CC[\mathcal R_m]\cong\bigoplus_{x\in X_m}M_{d_m(x)}(\CC).
\]
For each \(x\in X_m\), let \(P(c_x),P(\iota_x)\in M_{d_m(x)}(\CC)\) be the permutation matrices of the scan element and the reversal involution on the \(x\)-block. Assume that \(\Phi\in\Aut^\ast(\CC[\mathcal R_m])\) is compatible with these operators in the sense that there exist a permutation \(\sigma:X_m\to X_m\) and unitaries
\[
U_x\in M_{d_m(\sigma(x))}(\CC)
\]
such that
\[
\Phi\!\bigl(M_{d_m(x)}(\CC)\bigr)=M_{d_m(\sigma(x))}(\CC),
\]
and
\[
\Phi\!\bigl(P(c_x)\bigr)=U_xP(c_{\sigma(x)})U_x^{-1},
\qquad
\Phi\!\bigl(P(\iota_x)\bigr)=U_xP(\iota_{\sigma(x)})U_x^{-1}
\]
for every \(x\in X_m\). Then
\[
\chi(\sigma(x))=\chi(x)
\qquad(x\in X_m),
\]
where
\[
\chi(x)=\bigl(\operatorname{sgn}(c_x),\operatorname{sgn}(\iota_x)\bigr)\in\{\pm1\}^2.
\]
\end{proposition}

\begin{proof}
For any permutation \(\pi\), one has
\[
\det P(\pi)=\operatorname{sgn}(\pi)\in\{\pm1\}.
\]
By Proposition \ref{thm:fold-groupoid-aut-semidirect-pi0}, once the \(x\)-block is identified with the \(\sigma(x)\)-block, the restriction of \(\Phi\) is implemented by unitary conjugation. Determinants are therefore preserved on each matrix block. Applying this to the compatibility relation for \(c_x\), we obtain
\[
\det P(c_x)
=
\det\!\bigl(\Phi(P(c_x))\bigr)
=
\det\!\bigl(U_xP(c_{\sigma(x)})U_x^{-1}\bigr)
=
\det P(c_{\sigma(x)}),
\]
hence \(\operatorname{sgn}(c_x)=\operatorname{sgn}(c_{\sigma(x)})\). The same argument applied to \(P(\iota_x)\) gives
\[
\operatorname{sgn}(\iota_x)=\operatorname{sgn}(\iota_{\sigma(x)}).
\]
Together these identities say exactly that \(\chi(\sigma(x))=\chi(x)\).
\end{proof}

\begin{proposition}\label{prop:fold-groupoid-aut-chi-semidirect-classification}
For \((\alpha,\beta)\in\{\pm1\}^2\), let
\[
\CC[\mathcal{R}_m]^{\alpha,\beta}
\]
be the sector ideal introduced in Proposition \ref{prop:fold-groupoid-z2x2-sector-by-orientation}. Define
\[
\Aut^\ast_{\chi}(\CC[\mathcal R_m])
:=
\left\{\Phi\in\Aut^\ast(\CC[\mathcal R_m]):
\Phi\bigl(\CC[\mathcal R_m]^{\alpha,\beta}\bigr)=\CC[\mathcal R_m]^{\alpha,\beta}\ \forall(\alpha,\beta)\right\},
\]
and let
\[
\Pi_{\chi}
:=
\left\{\sigma\in\mathrm{Sym}(X_m):\ d_m(\sigma(x))=d_m(x),\ \chi(\sigma(x))=\chi(x)\ \forall x\in X_m\right\}.
\]
Then there is a natural isomorphism
\[
\Aut^\ast_{\chi}(\CC[\mathcal R_m])
\cong
\Bigl(\prod_{x\in X_m}\mathrm{PU}\bigl(d_m(x)\bigr)\Bigr)\rtimes \Pi_{\chi}.
\]
\end{proposition}

\begin{proof}
Proposition \ref{thm:fold-groupoid-aut-semidirect-pi0} identifies \(\Aut^\ast(\CC[\mathcal R_m])\) with the semidirect product of blockwise projective-unitary automorphisms and permutations of blocks of equal size. If \(\Phi\in\Aut^\ast_{\chi}(\CC[\mathcal R_m])\), then the induced block permutation must preserve the size \(d_m(x)\) and also preserve each sector ideal. Since the sector ideals are precisely the sums of blocks with fixed fingerprint, the induced permutation lies in \(\Pi_\chi\).

Conversely, any \(\sigma\in\Pi_\chi\), together with arbitrary blockwise inner automorphisms, preserves every sector ideal and therefore defines an element of \(\Aut^\ast_{\chi}(\CC[\mathcal R_m])\). This is exactly the asserted semidirect-product description.
\end{proof}
