\begin{definition}[Dominant exact-kernel family]\label{def:kernel-chebotarev-dominant-exact-kernel-family}
Let \(G\) be a finite abelian group, let \(u\in(0,1]\), and assume Assumption \ref{thm:kernel-chebotarev-exp}. We say that the nontrivial twisted spectrum is dominated by the exact kernel \(H_\star\le G\) if there exist
\[
0\le \Gamma_\star(u)<\eta_\star(u)<\lambda_1(u),
\]
a nonempty set \(S_\star(u)\subseteq \widehat G\setminus\{\ind\}\), and nonzero complex amplitudes \(a_\chi(u)\) and dominant spectral values \(\omega_\chi(u)\) for \(\chi\in S_\star(u)\) such that
\[
\ker\chi=H_\star
\qquad(\chi\in S_\star(u)),
\]
\[
\abs{\omega_\chi(u)}=\eta_\star(u)
\qquad(\chi\in S_\star(u)),
\]
and
\[
B_n(\chi;u)
=
\frac{a_\chi(u)\omega_\chi(u)^n}{n}
+
O\!\left(\frac{\Gamma_\star(u)^n}{n}\right)
\qquad(\chi\in S_\star(u)),
\]
\[
B_n(\psi;u)
=
O\!\left(\frac{\Gamma_\star(u)^n}{n}\right)
\qquad(\psi\in \widehat G\setminus(S_\star(u)\cup\{\ind\})).
\]
\end{definition}

\begin{theorem}\label{thm:kernel-chebotarev-dominant-kernel-detection}
Assume the hypotheses of Definition \ref{def:kernel-chebotarev-dominant-exact-kernel-family}. Let
\[
\mu_{n,u}(c):=\frac{B_{n,c}(u)}{B_n(\ind;u)}
\qquad(c\in G)
\]
for all sufficiently large \(n\), so that \(B_n(\ind;u)>0\), and define
\[
\mathcal D_{n,u}(H)
:=
D\!\bigl((\pi_H)_\ast\mu_{n,u}\,\|\,\mathrm{unif}_{G/H}\bigr)
\qquad(H\le G).
\]
Set
\[
C_\star(u):=\sum_{\chi\in S_\star(u)}\abs{a_\chi(u)}^2>0,
\]
\[
\rho_\star(u)
:=
\max\!\left\{
\frac{\eta_\star(u)\Gamma_\star(u)}{\lambda_1(u)^2},
\left(\frac{\Gamma_\star(u)}{\lambda_1(u)}\right)^2,
\tau(u)^3
\right\},
\qquad
\tau(u):=\frac{\Lambda(u)}{\lambda_1(u)}.
\]
Then \(0\le \rho_\star(u)<1\), and for every subgroup \(H\le G\),
\[
\mathcal D_{n,u}(H)
=
\frac{C_\star(u)}{2}
\left(\frac{\eta_\star(u)}{\lambda_1(u)}\right)^{2n}
+
O\!\bigl(\rho_\star(u)^n\bigr)
\qquad\text{if }H\subseteq H_\star,
\]
whereas
\[
\mathcal D_{n,u}(H)=O\!\bigl(\rho_\star(u)^n\bigr)
\qquad\text{if }H\not\subseteq H_\star.
\]
Thus the quotients on which the leading entropy survives are exactly those with \(H\subseteq H_\star\), or equivalently those admitting the natural quotient map \(G/H\twoheadrightarrow G/H_\star\). In this sense \(G/H_\star\) is the coarsest quotient carrying the dominant second-order entropy.

Moreover,
\[
\sum_{J\supseteq H_\star}\mu_{\mathrm{sub}}(H_\star,J)\,\mathcal D_{n,u}(J)
=
\frac{C_\star(u)}{2}
\left(\frac{\eta_\star(u)}{\lambda_1(u)}\right)^{2n}
+
O\!\bigl(\rho_\star(u)^n\bigr).
\]
\end{theorem}

\begin{proof}
The inequalities \(\Gamma_\star(u)<\eta_\star(u)<\lambda_1(u)\) and \(\tau(u)<1\) show immediately that \(0\le \rho_\star(u)<1\).

For \(H\le G\), Theorem \ref{thm:kernel-chebotarev-quotient-lattice-tomography} gives
\[
\mathcal E_{n,u}(H)
:=
\chi^2\!\bigl((\pi_H)_\ast\mu_{n,u}\,\|\,\mathrm{unif}_{G/H}\bigr)
=
\frac{1}{B_n(\ind;u)^2}
\sum_{\chi\in H^\perp\setminus\{\ind\}}
\abs{B_n(\chi;u)}^2.
\]
On the other hand, Assumption \ref{thm:kernel-chebotarev-exp} implies
\[
B_n(\ind;u)
=
\frac{\lambda_1(u)^n}{n}\left(1+O\!\bigl(\tau(u)^n\bigr)\right),
\qquad
\frac{1}{B_n(\ind;u)^2}
=
\frac{n^2}{\lambda_1(u)^{2n}}\left(1+O\!\bigl(\tau(u)^n\bigr)\right).
\]

Fix \(\chi\in S_\star(u)\). By definition,
\[
\abs{B_n(\chi;u)}^2
=
\frac{\abs{a_\chi(u)}^2\eta_\star(u)^{2n}}{n^2}
+
O\!\left(\frac{\eta_\star(u)^n\Gamma_\star(u)^n}{n^2}\right)
+
O\!\left(\frac{\Gamma_\star(u)^{2n}}{n^2}\right).
\]
Summing over \(\chi\in S_\star(u)\) yields
\[
\sum_{\chi\in S_\star(u)}\abs{B_n(\chi;u)}^2
=
\frac{C_\star(u)\eta_\star(u)^{2n}}{n^2}
+
O\!\left(\frac{\eta_\star(u)^n\Gamma_\star(u)^n}{n^2}\right)
+
O\!\left(\frac{\Gamma_\star(u)^{2n}}{n^2}\right).
\]
For every \(\psi\notin S_\star(u)\cup\{\ind\}\), the defining hypothesis gives
\[
\abs{B_n(\psi;u)}^2
=
O\!\left(\frac{\Gamma_\star(u)^{2n}}{n^2}\right).
\]

Suppose first that \(H\subseteq H_\star\). Since every \(\chi\in S_\star(u)\) has kernel \(H_\star\), these characters all lie in \(H^\perp\). Therefore
\[
\sum_{\chi\in H^\perp\setminus\{\ind\}}\abs{B_n(\chi;u)}^2
=
\frac{C_\star(u)\eta_\star(u)^{2n}}{n^2}
+
O\!\left(\frac{\eta_\star(u)^n\Gamma_\star(u)^n}{n^2}\right)
+
O\!\left(\frac{\Gamma_\star(u)^{2n}}{n^2}\right).
\]
Multiplying by \(1/B_n(\ind;u)^2\) gives
\[
\mathcal E_{n,u}(H)
=
C_\star(u)\left(\frac{\eta_\star(u)}{\lambda_1(u)}\right)^{2n}
+
O\!\left(\left(\frac{\eta_\star(u)\Gamma_\star(u)}{\lambda_1(u)^2}\right)^n\right)
+
O\!\left(\left(\frac{\Gamma_\star(u)}{\lambda_1(u)}\right)^{2n}\right)
+
O\!\bigl(\tau(u)^{3n}\bigr).
\]
Here we used the elementary inequality
\[
\tau(u)^n\left(\frac{\eta_\star(u)}{\lambda_1(u)}\right)^{2n}\le \tau(u)^{3n},
\]
which holds because \(\eta_\star(u)\le \eta(u)\le \Lambda(u)\).

Now suppose that \(H\not\subseteq H_\star\). Then no character in \(S_\star(u)\) belongs to \(H^\perp\), since \(H\subseteq\ker\chi\) would force \(H\subseteq H_\star\). Hence only the error terms remain and
\[
\mathcal E_{n,u}(H)
=
O\!\left(\left(\frac{\Gamma_\star(u)}{\lambda_1(u)}\right)^{2n}\right).
\]

Theorem \ref{thm:kernel-chebotarev-entropy-tomography} now yields
\[
\mathcal D_{n,u}(H)=\frac12\,\mathcal E_{n,u}(H)+O\!\bigl(\tau(u)^{3n}\bigr)
\]
uniformly in \(H\). Combining this with the two estimates for \(\mathcal E_{n,u}(H)\) gives the claimed asymptotics for \(\mathcal D_{n,u}(H)\).

Finally, Theorem \ref{thm:kernel-chebotarev-entropy-tomography} also gives
\[
\sum_{J\supseteq H_\star}\mu_{\mathrm{sub}}(H_\star,J)\,\mathcal D_{n,u}(J)
=
\frac12\,\mathcal W_{n,u}(H_\star)+O\!\bigl(\tau(u)^{3n}\bigr).
\]
By Theorem \ref{thm:kernel-chebotarev-quotient-lattice-tomography}, \(\mathcal W_{n,u}(H_\star)\) is precisely the exact-kernel spectral mass at \(H_\star\), and the preceding computation shows that only the characters in \(S_\star(u)\) contribute at dominant size. Hence
\[
\mathcal W_{n,u}(H_\star)
=
C_\star(u)\left(\frac{\eta_\star(u)}{\lambda_1(u)}\right)^{2n}
+
O\!\bigl(\rho_\star(u)^n\bigr),
\]
which is the asserted formula.
\end{proof}
