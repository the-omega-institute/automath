\section{Galois groups in the arithmetic window}
\label{sec:chebotarev}

By \Cref{thm:all-q-transfer}, the dominant root of every collision recurrence is
the same partition constant \(\lambda_q\) that governs Sanna's Fibonacci
partition power sums.  The arithmetic window \(q=9,\dots,17\) therefore studies
the arithmetic of these identified constants rather than fold-specific spectral
artifacts.  In this range the minimal recurrence polynomials already display a
rigid arithmetic profile: integral Hankel relations are generated by the
minimal polynomial, irreducibility and full symmetric Galois groups admit
explicit mod-\(p\) certificates, and the splitting fields for
\(q=12,13,14,15\) are linearly disjoint.  The exact recurrences used to define
the polynomials \(P_q\), together with the provenance of the certificate
tables and the audited secondary spectral data, are recorded separately in
Appendix~\ref{app:certification}. 

\subsection{Integral Hankel relations}

\begin{definition}[Resonance polynomials and null modes]
\label{def:resonance-polynomials}
For \(q\ge 2\), let \(P_q(\lambda)\in\ZZ[\lambda]\) denote the monic minimal
polynomial of the companion matrix of the linear recurrence satisfied by
\((S_q(m))_{m\ge m_0(q)}\), where \(m_0(q)\) is as in
Table~\ref{tab:appendix-recurrences-q9-17}.  Set
\(d_q:=\deg P_q\).  For \(q=9,\dots,17\), set
\[
  n_q:=2\left\lfloor\frac{q}{2}\right\rfloor+1,
  \qquad
  \kappa_q:=n_q-d_q.
\]
Define the square Hankel block
\[
  H_q:=\bigl[S_q(i+j+2)\bigr]_{0\le i,j\le n_q-1}
\]
and its integral null module
\[
  \mathrm{NullModes}(q):=\ker_{\ZZ}(H_q^{\mathsf T})\subset \ZZ^{n_q}.
\]
Finally, let
\[
  \mathcal M(P_q):=
  \bigl\{\operatorname{coeff}_{<n_q}(Q(\lambda)P_q(\lambda)):
  Q\in\ZZ[\lambda],\ \deg Q<\kappa_q\bigr\}\subset\ZZ^{n_q},
\]
where \(\operatorname{coeff}_{<n_q}\) truncates a polynomial to its first
\(n_q\) coefficients.
\end{definition}

\begin{theorem}[Hankel principalization]
\label{thm:principalization}
For every \(q\in\{9,\dots,17\}\),
\[
  \mathrm{NullModes}(q)=\mathcal M(P_q).
\]
In other words, every integral elimination relation visible in the chosen
Hankel window is a short multiple of the minimal recurrence polynomial.
\end{theorem}

\begin{proof}
Since \(P_q(E)S_q(\cdot+2)=0\), every short multiple \(Q(\lambda)P_q(\lambda)\)
gives an annihilating relation on the full sequence and hence a vector in
\(\mathrm{NullModes}(q)\).  Thus \(\mathcal M(P_q)\subseteq \mathrm{NullModes}(q)\).

By minimality of the recurrence, sufficiently large square Hankel blocks of the
sequence have rank \(d_q\); the chosen range is one in which this rank is
already attained.
Hence
\[
  \rk H_q=d_q,
  \qquad
  \rk_{\ZZ}\mathrm{NullModes}(q)=n_q-d_q=\kappa_q.
\]
On the other hand, \(\mathcal M(P_q)\) is a free \(\ZZ\)-module of rank
\(\kappa_q\), generated by the truncated coefficient vectors of
\(P_q,\lambda P_q,\dots,\lambda^{\kappa_q-1}P_q\).  Since
\(\mathcal M(P_q)\subseteq \mathrm{NullModes}(q)\) and both are free
\(\ZZ\)-modules of rank \(\kappa_q\), the inclusion has finite index.  To
verify that this index equals~\(1\), we compute the Smith normal form of the
\(\kappa_q\times\kappa_q\) matrix expressing the generators
\(P_q,\lambda P_q,\dots,\lambda^{\kappa_q-1}P_q\) in a \(\ZZ\)-basis of
\(\mathrm{NullModes}(q)\); for each \(q=9,\dots,17\), all invariant factors
equal~\(1\) (equivalently, the change-of-basis matrix has determinant
\(\pm 1\)).  This computational verification (recorded in
Appendix~\ref{app:certification}) confirms
\(\mathcal M(P_q)=\mathrm{NullModes}(q)\).
\end{proof}

\begin{corollary}[No spurious eliminations]
\label{cor:no-spurious-elimination}
Let \(q\in\{9,\dots,17\}\), and let \(B(\lambda)\in\ZZ[\lambda]\) satisfy
\(\deg B<n_q\).  If its coefficient vector belongs to \(\mathrm{NullModes}(q)\),
then there exists a unique polynomial \(Q(\lambda)\in\ZZ[\lambda]\) with
\(\deg Q<\kappa_q\) such that
\[
  B(\lambda)=Q(\lambda)P_q(\lambda).
\]
In particular, every elimination relation detected in the chosen Hankel block
extends automatically to the full recurrence.
\end{corollary}

\begin{proof}
By \Cref{thm:principalization}, the coefficient vector of \(B\) is the
truncation of \(QP_q\) for some \(Q\) of degree \(<\kappa_q\).  Since
\(\deg(QP_q)<n_q\), truncation does nothing, and uniqueness follows from the
fact that \(P_q\) is monic.
\end{proof}

\begin{corollary}[GCD recovery]
\label{cor:gcd-recovers-minpoly}
Let \(q\in\{9,\dots,17\}\), and choose any \(\ZZ\)-basis
\(\alpha^{(1)},\dots,\alpha^{(\kappa_q)}\) of \(\mathrm{NullModes}(q)\).  Write
\[
  A^{(j)}(\lambda):=\sum_{k=0}^{n_q-1}\alpha^{(j)}_k\lambda^k.
\]
Then
\[
  \gcd\bigl(A^{(1)}(\lambda),\dots,A^{(\kappa_q)}(\lambda)\bigr)=\pm P_q(\lambda).
\]
Thus the minimal recurrence polynomial is uniquely recoverable from any integral
basis of finite Hankel certificates.
\end{corollary}

\begin{proof}
By \Cref{thm:principalization}, each \(A^{(j)}\) is of the form
\(Q^{(j)}P_q\) with \(\deg Q^{(j)}<\kappa_q\), so \(P_q\) is a common divisor.
Conversely, the module \(\mathcal M(P_q)\) contains \(P_q\) itself, because one
may take \(Q\equiv 1\).  Since the \(\alpha^{(j)}\) form a basis of
\(\mathcal M(P_q)\), there are integers \(c_j\) such that
\[
  P_q=\sum_{j=1}^{\kappa_q} c_j A^{(j)}
  =\left(\sum_{j=1}^{\kappa_q} c_jQ^{(j)}\right)P_q.
\]
As \(\ZZ[\lambda]\) is an integral domain, the polynomials \(Q^{(j)}\) generate
the unit ideal.  Hence their greatest common divisor is \(\pm1\), and the gcd of
the \(A^{(j)}\) is exactly \(\pm P_q\).
\end{proof}

\begin{proposition}[Irreducibility in the resonance window]
\label{prop:irreducibility-window}
For \(q=9,\dots,17\), the minimal recurrence polynomial \(P_q\) is irreducible
over \(\QQ\).  Its degree sequence is
\[
  (d_9,d_{10},d_{11},d_{12},d_{13},d_{14},d_{15},d_{16},d_{17})
  =(7,9,9,13,11,13,11,13,13).
\]
\end{proposition}

\begin{proof}
The degree data are the orders of the exact recurrences listed in
Table~\ref{tab:appendix-recurrences-q9-17}.  By
\Cref{tab:galois-certificates-q9-17}, for each \(q\) there exists a prime
\(p_{\mathrm{irr}}\) such that \(P_q\bmod p_{\mathrm{irr}}\) is irreducible in
\(\FF_{p_{\mathrm{irr}}}[\lambda]\).  Since \(P_q\) is monic with integer
coefficients, Gauss's lemma implies that \(P_q\) is irreducible over \(\QQ\).
\end{proof}

\begin{lemma}[A transitive group with an \texorpdfstring{$(n-1)$}{(n-1)}-cycle is primitive]
\label{lem:nminus1-cycle-primitive}
Let \(G\le S_n\) be transitive with \(n\ge 3\).  If \(G\) contains an
\((n-1)\)-cycle, then \(G\) is primitive.
\end{lemma}

\begin{proof}
Let \(\tau\in G\) be an \((n-1)\)-cycle, and let \(\alpha\) be its unique fixed
point.  Consider a block \(B\subseteq\{1,\dots,n\}\) for the action of \(G\).

If \(\alpha\in B\), then \(B\setminus\{\alpha\}\) is \(\tau\)-invariant inside the
single \(\tau\)-orbit \(\{1,\dots,n\}\setminus\{\alpha\}\), so either
\(B=\{\alpha\}\) or \(B=\{1,\dots,n\}\).

If \(\alpha\notin B\), choose \(\beta\in B\) and \(g\in G\) with \(g(\beta)=\alpha\),
using transitivity.  Then \(g(B)\) is a block containing \(\alpha\), hence
trivial or all by the previous paragraph.  Therefore \(B\) itself is trivial or
all.
\end{proof}

\begin{theorem}[Full symmetric Galois groups in the resonance window]
\label{thm:galois-sd-window}
For every \(q=9,\dots,17\),
\[
  \Gal(P_q/\QQ)\cong S_{d_q}.
\]
\end{theorem}

\begin{proof}
Fix \(q\in\{9,\dots,17\}\), and let \(G_q:=\Gal(P_q/\QQ)\le S_{d_q}\) in its
natural action on the roots of \(P_q\).

By \Cref{prop:irreducibility-window}, the polynomial \(P_q\) is irreducible, so
\(G_q\) is transitive.  The factorization type \((d_q-1,1)\) from
\Cref{tab:galois-certificates-q9-17} exhibits an element of \(G_q\) that is an
\((d_q-1)\)-cycle, so \Cref{lem:nminus1-cycle-primitive} shows that \(G_q\) is
primitive.

The same table also provides an element of prime cycle length
\[
  \ell_q\in\{3,5,5,7,7,7,7,7,7\},
\]
and in each case \(\ell_q\le d_q-3\).  Jordan's theorem therefore implies that
\(A_{d_q}\subseteq G_q\); see \cite[Thm.~3.3A]{DixonMortimer1996}.

Finally, an \((d_q-1)\)-cycle is odd because \(d_q-1\) is even for all
\(d_q\in\{7,9,11,13\}\).  Hence \(G_q\) is not contained in \(A_{d_q}\).  Since
\(A_{d_q}\subseteq G_q\le S_{d_q}\), we conclude that \(G_q=S_{d_q}\).
\end{proof}

\begin{corollary}[Chebotarev densities]
\label{cor:individual-chebotarev}
For each \(q=9,\dots,17\), the set of unramified primes \(p\) for which
\(P_q\bmod p\) is irreducible in \(\FF_p[\lambda]\) has natural density
\(1/d_q\).
\end{corollary}

\begin{proof}
By \Cref{thm:galois-sd-window} and the Chebotarev density theorem
\cite{Neukirch1999}, Frobenius classes are equidistributed in \(S_{d_q}\).
Irreducibility modulo \(p\) is equivalent to Frobenius being a
\(d_q\)-cycle, and the proportion of \(d_q\)-cycles in \(S_{d_q}\) is
\((d_q-1)!/d_q!=1/d_q\).
\end{proof}

\begin{corollary}[Rational R\'enyi denominators are large]
\label{cor:renyi-rational-denominator}
For \(q\in\{9,\dots,17\}\), set
\[
  D_q:=\frac{q\log 2-\log \lambda_q}{(q-1)\log\varphi}.
\]
If \(D_q=a/b\in\QQ\) in lowest terms, then
\[
  b\ge \left\lceil\frac{d_q}{2}\right\rceil.
\]
In particular, the range \(q=9,\dots,17\) yields the lower bounds
\[
  b\ge 4,5,5,7,6,7,6,7,7
  \qquad (q=9,\dots,17).
\]
\end{corollary}

\begin{proof}
From the definition of \(D_q\),
\[
  \lambda_q = 2^q\varphi^{-(q-1)a/b}\in \QQ(\varphi^{1/b}).
\]
Since \(\QQ(\varphi)=\QQ(\sqrt5)\) is quadratic, the extension
\(\QQ(\varphi^{1/b})/\QQ\) has degree at most \(2b\).  By
\Cref{prop:irreducibility-window}, the algebraic degree of \(\lambda_q\) is
\(d_q\).
Therefore \(d_q\le 2b\), which is the claimed lower bound.
\end{proof}

\begin{theorem}[Linear disjointness for \texorpdfstring{$q=12,13,14,15$}{q=12,13,14,15}]
\label{thm:linear-disjointness}
Let \(L_q\) be the splitting field of \(P_q\) for
\(q\in\{12,13,14,15\}\).  Then the fields \(L_{12},L_{13},L_{14},L_{15}\) are
linearly disjoint over \(\QQ\), and
\[
  \Gal(L_{12}L_{13}L_{14}L_{15}/\QQ)
  \cong
  S_{13}\times S_{11}\times S_{13}\times S_{11}.
\]
\end{theorem}

\begin{proof}
By \Cref{thm:galois-sd-window}, each \(L_q/\QQ\) has full symmetric Galois
group, so its only nontrivial proper normal subgroup is \(A_{d_q}\).  The
corresponding quadratic subfield is the discriminant field
\(\QQ(\sqrt{\Disc(P_q)})\).

By \Cref{tab:disc-squareclass-q12-15}, the four discriminant square classes are
linearly independent in \(\QQ^\times/(\QQ^\times)^2\).  Hence none of the four
quadratic discriminant fields lies in the compositum of the other three.

Fix \(q\in\{12,13,14,15\}\), and let \(M_q\) be the compositum of the three
remaining splitting fields.  Because both \(L_q/\QQ\) and \(M_q/\QQ\) are
Galois, the intersection \(L_q\cap M_q\) is a normal subextension of
\(L_q/\QQ\).  Therefore \(L_q\cap M_q\) is either \(\QQ\), the discriminant
quadratic field of \(L_q\), or all of \(L_q\).  The last two possibilities are
excluded by squareclass independence.  Thus \(L_q\cap M_q=\QQ\) for each \(q\),
which is exactly linear disjointness.

The direct-product formula for the Galois group follows from the compositum law
for finite Galois extensions with trivial intersections.
\end{proof}

\begin{corollary}[Product Chebotarev densities]
\label{cor:chebotarev-product}
Let \(C_q\subseteq S_{d_q}\) be a conjugacy class for each
\(q\in\{12,13,14,15\}\).  The set of unramified primes \(p\) for which the
Frobenius classes of \(P_{12},P_{13},P_{14},P_{15}\) lie respectively in
\(C_{12},C_{13},C_{14},C_{15}\) has natural density
\[
  \prod_{q\in\{12,13,14,15\}}\frac{\card{C_q}}{\card{S_{d_q}}}.
\]
In particular, the density of primes for which all four polynomials remain
irreducible is
\[
  \frac{1}{13}\cdot\frac{1}{11}\cdot\frac{1}{13}\cdot\frac{1}{11}
  =\frac{1}{20449}.
\]
\end{corollary}

\begin{proof}
Apply the Chebotarev density theorem \cite{Neukirch1999} to the compositum
field of \Cref{thm:linear-disjointness}.  The direct-product Galois group
forces exact independence of Frobenius classes across the four factors.
\end{proof}

\begin{remark}[Additivity of the logarithmic density]
\label{rem:log-density-additivity}
In logarithmic form, \Cref{cor:chebotarev-product} gives
\[
  -\log \delta(C_{12},C_{13},C_{14},C_{15})
  =
  \sum_{q\in\{12,13,14,15\}}
  \Bigl(-\log \frac{\card{C_q}}{\card{S_{d_q}}}\Bigr),
\]
where \(\delta(C_{12},C_{13},C_{14},C_{15})\) denotes the density of primes
whose Frobenius classes lie in the respective conjugacy classes
\(C_q\subseteq S_{d_q}\).  For \(d_q\)-cycles this yields
\(-\log \delta = \log 20449\).
\end{remark}
