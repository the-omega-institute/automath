\section{Weighted Integrals for the Eighth-Order Term}\label{app:entropy-integrals}

We record the weighted identities used in the proof of
Theorem~\ref{thm:poisson-kl-eighth}.

\begin{lemma}\label{lem:entropy-integrals}
Let $u_2,\dots,u_6$ be the mode functions defined by
\eqref{eq:cayley-first-modes} and \eqref{eq:cayley-next-modes}.  Then
\[
\int_{\RR}\frac{u_2(y)^2}{\pi(1+y^2)}\dd y=\frac14,
\qquad
\int_{\RR}\frac{u_3(y)^2}{\pi(1+y^2)}\dd y=\frac{3}{16},
\]
\[
\int_{\RR}\frac{u_2(y)u_4(y)}{\pi(1+y^2)}\dd y=-\frac18,
\qquad
\int_{\RR}\frac{u_2(y)^3}{\pi(1+y^2)}\dd y=-\frac{3}{32},
\]
\[
\int_{\RR}\frac{u_2(y)u_6(y)}{\pi(1+y^2)}\dd y=\frac{3}{64},
\qquad
\int_{\RR}\frac{u_3(y)u_5(y)}{\pi(1+y^2)}\dd y=-\frac{15}{128},
\]
\[
\int_{\RR}\frac{u_4(y)^2}{\pi(1+y^2)}\dd y=\frac{5}{32},
\qquad
\int_{\RR}\frac{u_2(y)^2u_4(y)}{\pi(1+y^2)}\dd y=\frac{3}{32},
\]
\[
\int_{\RR}\frac{u_2(y)u_3(y)^2}{\pi(1+y^2)}\dd y=-\frac{9}{128},
\qquad
\int_{\RR}\frac{u_2(y)^4}{\pi(1+y^2)}\dd y=\frac{9}{64}.
\]
\end{lemma}

\begin{proof}
Set $y=\tan\theta$ with $\theta\in(-\pi/2,\pi/2)$, so that
$\dd y/(1+y^2)=\dd\theta$, and define
\[
c_k(\theta):=\cos(2k\theta),
\qquad
s_k(\theta):=\sin(2k\theta).
\]
By Theorem~\ref{thm:cayley-chebyshev-mode},
\[
u_2(\tan\theta)=-\frac12(c_1+c_2),
\qquad
u_3(\tan\theta)=-\frac14(s_1+2s_2+s_3),
\]
\[
u_4(\tan\theta)=\frac18(c_1+3c_2+3c_3+c_4),
\qquad
u_5(\tan\theta)=\frac1{16}(s_1+4s_2+6s_3+4s_4+s_5),
\]
\[
u_6(\tan\theta)=-\frac1{32}(c_1+5c_2+10c_3+10c_4+5c_5+c_6).
\]
On $(-\pi/2,\pi/2)$, one has
\[
\int_{-\pi/2}^{\pi/2} c_k(\theta)^2\,\dd\theta
=\int_{-\pi/2}^{\pi/2} s_k(\theta)^2\,\dd\theta
=\frac{\pi}{2}\qquad(k\ge1),
\]
and mixed products with different parity or different frequency integrate to
zero.

Each displayed identity follows by expanding the corresponding trigonometric
polynomial and keeping only zero-frequency terms.  For clarity:
\[
\int_{\RR}\frac{u_2(y)^2}{\pi(1+y^2)}\dd y
=\frac14\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)^2\,\dd\theta
=\frac14,
\]
\[
\int_{\RR}\frac{u_3(y)^2}{\pi(1+y^2)}\dd y
=\frac1{16}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(s_1+2s_2+s_3)^2\,\dd\theta
=\frac{3}{16},
\]
\[
\int_{\RR}\frac{u_2(y)u_4(y)}{\pi(1+y^2)}\dd y
=-\frac1{16}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)(c_1+3c_2+3c_3+c_4)\,\dd\theta
=-\frac18,
\]
\[
\int_{\RR}\frac{u_2(y)^3}{\pi(1+y^2)}\dd y
=-\frac1{8}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)^3\,\dd\theta
=-\frac{3}{32},
\]
\[
\int_{\RR}\frac{u_2(y)u_6(y)}{\pi(1+y^2)}\dd y
=\frac1{64}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)(c_1+5c_2+10c_3+10c_4+5c_5+c_6)\,\dd\theta
=\frac{3}{64},
\]
\[
\int_{\RR}\frac{u_3(y)u_5(y)}{\pi(1+y^2)}\dd y
=-\frac1{64}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(s_1+2s_2+s_3)(s_1+4s_2+6s_3+4s_4+s_5)\,\dd\theta
=-\frac{15}{128},
\]
\[
\int_{\RR}\frac{u_4(y)^2}{\pi(1+y^2)}\dd y
=\frac1{64}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+3c_2+3c_3+c_4)^2\,\dd\theta
=\frac{5}{32},
\]
\[
\int_{\RR}\frac{u_2(y)^2u_4(y)}{\pi(1+y^2)}\dd y
=\frac1{32}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)^2(c_1+3c_2+3c_3+c_4)\,\dd\theta
=\frac{3}{32},
\]
\[
\int_{\RR}\frac{u_2(y)u_3(y)^2}{\pi(1+y^2)}\dd y
=-\frac1{32}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)(s_1+2s_2+s_3)^2\,\dd\theta
=-\frac{9}{128},
\]
\[
\int_{\RR}\frac{u_2(y)^4}{\pi(1+y^2)}\dd y
=\frac1{16}\cdot\frac1\pi\int_{-\pi/2}^{\pi/2}(c_1+c_2)^4\,\dd\theta
=\frac{9}{64}.
\]
\end{proof}

\section{Step-by-Step Coefficient Assembly for Theorem~\ref{thm:poisson-kl-eighth}}%
\label{app:coefficient-assembly}

We provide the full bookkeeping requested by the main proof.  The goal
of this appendix is auditability: each surviving monomial in
$\Phi(\delta_t)$ through order $t^{-8}$ is listed explicitly, matched
with the corresponding integral from
Appendix~\ref{app:entropy-integrals}, and simplified coefficient by
coefficient.

\subsection*{Step 1.\ Expansion of $\Phi(\delta_t)$}

With $\delta_t=t^{-2}a_2+t^{-3}a_3+\cdots+t^{-6}a_6+O(t^{-7})$
where $a_k:=\mu_ku_k$ and
$\Phi(z)=(1+z)\log(1+z)-z=\tfrac12z^2-\tfrac16z^3+\tfrac1{12}z^4+O(z^5)$,
the contributions through order $t^{-8}$ are:
\begin{align*}
\tfrac12\delta_t^2 &= \tfrac{1}{2t^4}a_2^2
+\tfrac{1}{t^5}a_2a_3
+\tfrac{1}{t^6}\bigl(a_2a_4+\tfrac12 a_3^2\bigr)
+\tfrac{1}{t^7}(a_2a_5+a_3a_4) \\
&\quad +\tfrac{1}{t^8}\bigl(a_2a_6+a_3a_5+\tfrac12 a_4^2\bigr)
+O(t^{-9}), \\[4pt]
-\tfrac16\delta_t^3 &= -\tfrac{1}{6t^6}a_2^3
-\tfrac{1}{2t^7}a_2^2a_3
-\tfrac{1}{t^8}\bigl(\tfrac12 a_2^2a_4+\tfrac12 a_2a_3^2\bigr)
+O(t^{-9}), \\[4pt]
\tfrac1{12}\delta_t^4 &= \tfrac{1}{12t^8}a_2^4+O(t^{-9}).
\end{align*}

\subsection*{Step 2.\ Odd-order cancellation}

The terms at orders $t^{-5}$ and $t^{-7}$ integrate to zero by
Lemma~\ref{lem:mode-parity-vanish}: the total mode-degree sum is
odd ($2+3=5$, and $2+5=7$, $3+4=7$, $2+2+3=7$), so the weighted
integrals vanish.

\begin{center}
\small
\begin{tabular}{p{0.13\textwidth}p{0.30\textwidth}p{0.13\textwidth}p{0.25\textwidth}}
\textbf{Order} & \textbf{Source term} & \textbf{Mode sum} & \textbf{Reason} \\
$t^{-5}$ & $a_2a_3$ from $\tfrac12\delta_t^2$ & $2+3=5$ & Lemma~\ref{lem:mode-parity-vanish} \\
$t^{-7}$ & $a_2a_5+a_3a_4$ from $\tfrac12\delta_t^2$ & $2+5=7$, $3+4=7$ & Lemma~\ref{lem:mode-parity-vanish} \\
$t^{-7}$ & $-\tfrac12 a_2^2a_3$ from $-\tfrac16\delta_t^3$ & $2+2+3=7$ & Lemma~\ref{lem:mode-parity-vanish}
\end{tabular}
\end{center}

\subsection*{Step 3.\ Assembly of $C_4$}

The only $t^{-4}$ term is $\tfrac12a_2^2/t^4$.  Hence
\[
C_4=\frac{\sigma^4}{2}\int_{\RR}\frac{u_2^2}{\pi(1+y^2)}\dd y
=\frac{\sigma^4}{2}\cdot\frac14=\frac{\sigma^4}{8}.
\]

\subsection*{Step 4.\ Assembly of $C_6$}

The $t^{-6}$ contributions after integration are:
\[
\sigma^2\mu_4\int\frac{u_2u_4}{\pi(1+y^2)}\dd y
+\frac{\mu_3^2}{2}\int\frac{u_3^2}{\pi(1+y^2)}\dd y
-\frac{\sigma^6}{6}\int\frac{u_2^3}{\pi(1+y^2)}\dd y.
\]
Equivalently, the surviving monomials are recorded in the following
ledger.
\begin{center}
\small
\begin{tabular}{p{0.20\textwidth}p{0.22\textwidth}p{0.14\textwidth}p{0.20\textwidth}}
\textbf{Source in $\Phi(\delta_t)$} & \textbf{Monomial} & \textbf{Integral} & \textbf{Contribution} \\
$\tfrac12\delta_t^2$ & $a_2a_4$ & $\int u_2u_4\,\dd\omega=-\tfrac18$ & $-\dfrac{\sigma^2\mu_4}{8}$ \\
$\tfrac12\delta_t^2$ & $\tfrac12 a_3^2$ & $\int u_3^2\,\dd\omega=\tfrac{3}{16}$ & $\dfrac{3\mu_3^2}{32}$ \\
$-\tfrac16\delta_t^3$ & $-\tfrac16 a_2^3$ & $\int u_2^3\,\dd\omega=-\tfrac{3}{32}$ & $\dfrac{\sigma^6}{64}$
\end{tabular}
\end{center}
Substituting the integral values from Lemma~\ref{lem:entropy-integrals}:
\[
=\sigma^2\mu_4\Bigl(-\frac18\Bigr)
+\frac{\mu_3^2}{2}\cdot\frac{3}{16}
-\frac{\sigma^6}{6}\Bigl(-\frac{3}{32}\Bigr)
=-\frac{\sigma^2\mu_4}{8}+\frac{3\mu_3^2}{32}+\frac{\sigma^6}{64}.
\]
Combining over a common denominator of 64:
\[
C_6=\frac{-8\sigma^2\mu_4+6\mu_3^2+\sigma^6}{64}
=\frac{\sigma^6+6\mu_3^2-8\sigma^2\mu_4}{64}.
\]

\subsection*{Step 5.\ Assembly of $C_8$}

The $t^{-8}$ contributions split into two groups.

\medskip\noindent
\emph{Group A} (from $\tfrac12\delta_t^2$):
\[
\sigma^2\mu_6\int\frac{u_2u_6}{\pi(1+y^2)}\dd y
+\mu_3\mu_5\int\frac{u_3u_5}{\pi(1+y^2)}\dd y
+\frac{\mu_4^2}{2}\int\frac{u_4^2}{\pi(1+y^2)}\dd y
\]
\[
=\sigma^2\mu_6\cdot\frac{3}{64}
+\mu_3\mu_5\Bigl(-\frac{15}{128}\Bigr)
+\frac{\mu_4^2}{2}\cdot\frac{5}{32}
=\frac{12\sigma^2\mu_6-30\mu_3\mu_5+20\mu_4^2}{256}.
\]

\medskip\noindent
\emph{Group B} (from $-\tfrac16\delta_t^3$ and $\tfrac1{12}\delta_t^4$):
\[
-\frac{\sigma^4\mu_4}{2}\int\frac{u_2^2u_4}{\pi(1+y^2)}\dd y
-\frac{\sigma^2\mu_3^2}{2}\int\frac{u_2u_3^2}{\pi(1+y^2)}\dd y
+\frac{\sigma^8}{12}\int\frac{u_2^4}{\pi(1+y^2)}\dd y
\]
\[
=-\frac{\sigma^4\mu_4}{2}\cdot\frac{3}{32}
-\frac{\sigma^2\mu_3^2}{2}\Bigl(-\frac{9}{128}\Bigr)
+\frac{\sigma^8}{12}\cdot\frac{9}{64}
=\frac{-12\sigma^4\mu_4+9\sigma^2\mu_3^2+3\sigma^8}{256}.
\]
The full coefficient ledger is therefore:
\begin{center}
\small
\begin{tabular}{p{0.20\textwidth}p{0.22\textwidth}p{0.14\textwidth}p{0.20\textwidth}}
\textbf{Source in $\Phi(\delta_t)$} & \textbf{Monomial} & \textbf{Integral} & \textbf{Contribution} \\
$\tfrac12\delta_t^2$ & $a_2a_6$ & $\int u_2u_6\,\dd\omega=\tfrac{3}{64}$ & $\dfrac{12\sigma^2\mu_6}{256}$ \\
$\tfrac12\delta_t^2$ & $a_3a_5$ & $\int u_3u_5\,\dd\omega=-\tfrac{15}{128}$ & $-\dfrac{30\mu_3\mu_5}{256}$ \\
$\tfrac12\delta_t^2$ & $\tfrac12 a_4^2$ & $\int u_4^2\,\dd\omega=\tfrac{5}{32}$ & $\dfrac{20\mu_4^2}{256}$ \\
$-\tfrac16\delta_t^3$ & $-\tfrac12 a_2^2a_4$ & $\int u_2^2u_4\,\dd\omega=\tfrac{3}{32}$ & $-\dfrac{12\sigma^4\mu_4}{256}$ \\
$-\tfrac16\delta_t^3$ & $-\tfrac12 a_2a_3^2$ & $\int u_2u_3^2\,\dd\omega=-\tfrac{9}{128}$ & $\dfrac{9\sigma^2\mu_3^2}{256}$ \\
$\tfrac1{12}\delta_t^4$ & $\tfrac1{12}a_2^4$ & $\int u_2^4\,\dd\omega=\tfrac{9}{64}$ & $\dfrac{3\sigma^8}{256}$
\end{tabular}
\end{center}
Thus
\[
C_8=\frac{3\sigma^8-12\sigma^4\mu_4+9\sigma^2\mu_3^2}{256}
+\frac{12\sigma^2\mu_6-30\mu_3\mu_5+20\mu_4^2}{256}.
\]

\subsection*{Step 6.\ Conversion to defect variables}

With the normalized variable $X=\Delta/\sigma$, set
$\beta_k:=\EE[X^k]$, $Q_2:=X^2-1-\beta_3X$,
$\kappa:=\|Q_2\|^2=\beta_4-1-\beta_3^2$.
Then:
\[
\frac{64}{\sigma^6}C_6
=1+6\beta_3^2-8\beta_4
=1+6\beta_3^2-8(1+\beta_3^2+\kappa)
=-7-2\beta_3^2-8\kappa.
\]
This gives the defect
$\Delta_6=\sigma^6(\beta_3^2/32+\kappa/8)$.

For $C_8$, writing $\beta_k=\mu_k/\sigma^k$ and using the identities
$\beta_5=\beta_3(\beta_3^2+2+2\kappa)+L$ and
$\beta_6=1+3\beta_3^2+3\kappa+\beta_3^4+3\beta_3^2\kappa+3\beta_3L+M$
(where $L:=\EE[XQ_2^2]$, $M:=\EE[Q_2^3]$), a direct substitution
converts $256C_8/\sigma^8$ into the expression in
Theorem~\ref{thm:poisson-defect-ladder}.
