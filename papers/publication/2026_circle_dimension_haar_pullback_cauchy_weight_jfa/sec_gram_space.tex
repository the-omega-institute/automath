\section{The secant family, the RKHS, and centered reconstruction}%
\label{sec:gram-space}

The secant calculus from Section~\ref{sec:entropy-asymptotics} also has
a natural Hilbert-space interpretation.  For the kernel under
consideration, the Gram orbit, the centered Poisson channels, and the
entropy coefficients are all encoded by the same explicit secant
family.  This permits a concrete description of the reproducing-kernel
Hilbert space generated by~$K$.

\subsection{The Gram kernel and the Hilbert completion}

Let
\[
\omega(\dd y):=\frac{\dd y}{\pi(1+y^2)}.
\]
For $\varepsilon\in\RR$ define the secant family
\begin{equation}\label{eq:mode-secant-family}
\Psi_\varepsilon(y):=\frac{R_\varepsilon(y)-1}{\varepsilon}
=\frac{2y-\varepsilon}{1+(y-\varepsilon)^2},
\qquad y\in\RR,
\end{equation}
where the formula remains valid at $\varepsilon=0$ and gives
\[
\Psi_0(y)=u_1(y)=\frac{2y}{1+y^2}.
\]
Since
\[
R_\varepsilon(y)\,\omega(\dd y)=P_1(y-\varepsilon)\,\dd y,
\]
one has
\[
\int_{\RR}R_\varepsilon(y)\,\omega(\dd y)=1
\qquad(\varepsilon\in\RR),
\]
and therefore $\Psi_\varepsilon\in L^2_0(\omega)$ for every
$\varepsilon\in\RR$.

\begin{theorem}%
\label{thm:mode-gram-kernel}
For every $\varepsilon,\eta\in\RR$,
\begin{equation}\label{eq:mode-gram-kernel}
\langle \Psi_\varepsilon,\Psi_\eta\rangle_{L^2(\omega)}
=\frac{2}{4+(\varepsilon-\eta)^2}.
\end{equation}
Equivalently,
\begin{equation}\label{eq:mode-gram-RR}
\int_{\RR}R_\varepsilon(y)R_\eta(y)\,\omega(\dd y)
=\frac{\varepsilon^2+\eta^2+4}{(\varepsilon-\eta)^2+4}.
\end{equation}
\end{theorem}

\begin{proof}
For $\varepsilon\neq\eta$ define
\[
I(\varepsilon,\eta)
:=\frac{1}{\pi}\int_{\RR}
\frac{1+y^2}{(1+(y-\varepsilon)^2)(1+(y-\eta)^2)}\,\dd y.
\]
The integrand is $O(|z|^{-2})$ on the upper half-plane and has simple
poles at $z=\varepsilon+\ii$ and $z=\eta+\ii$.  A direct residue
computation gives
\[
\operatorname{Res}_{z=\varepsilon+\ii}
=\frac{\varepsilon(\varepsilon+2\ii)}
{2\ii(\varepsilon-\eta)(\varepsilon-\eta+2\ii)},
\]
\[
\operatorname{Res}_{z=\eta+\ii}
=\frac{\eta(\eta+2\ii)}
{2\ii(\eta-\varepsilon)(\eta-\varepsilon+2\ii)}.
\]
Hence, by the residue theorem,
\[
I(\varepsilon,\eta)
=2\ii\Bigl(
\operatorname{Res}_{z=\varepsilon+\ii}
+\operatorname{Res}_{z=\eta+\ii}
\Bigr)
=\frac{\varepsilon^2+\eta^2+4}{(\varepsilon-\eta)^2+4}.
\]
This proves \eqref{eq:mode-gram-RR} for $\varepsilon\neq\eta$, and the
diagonal case follows by continuity.

Since
\[
\int_{\RR}R_\varepsilon(y)\,\omega(\dd y)=1,
\]
one obtains
\[
\int_{\RR}(R_\varepsilon(y)-1)(R_\eta(y)-1)\,\omega(\dd y)
=\frac{2\varepsilon\eta}{4+(\varepsilon-\eta)^2}.
\]
For $\varepsilon\eta\neq 0$, division by $\varepsilon\eta$ yields
\eqref{eq:mode-gram-kernel}; the remaining cases follow by continuity in
$(\varepsilon,\eta)$.
\end{proof}

\begin{corollary}%
\label{cor:mode-quadratic-integrals}
For $|\varepsilon|$ small one has the $L^2(\omega)$ expansion
\[
\Psi_\varepsilon
=\sum_{n=1}^{\infty}u_n\,\varepsilon^{n-1}.
\]
Consequently, for every $p,q\ge 1$,
\[
\langle u_{2p},u_{2q}\rangle_{L^2(\omega)}
=(-1)^{p+q}2^{-(2p+2q-1)}
\binom{2p+2q-2}{2p-1},
\]
for every $p,q\ge 0$,
\[
\langle u_{2p+1},u_{2q+1}\rangle_{L^2(\omega)}
=(-1)^{p+q}2^{-(2p+2q+1)}
\binom{2p+2q}{2p},
\]
and
\[
\langle u_{2p},u_{2q+1}\rangle_{L^2(\omega)}=0.
\]
\end{corollary}

\begin{proof}
By Proposition~\ref{prop:cayley-mode-remainder},
\[
\Psi_\varepsilon
=\sum_{n=1}^{N}u_n\,\varepsilon^{n-1}
+\varepsilon^N v_N(\cdot,\varepsilon)
\]
with $v_N$ bounded uniformly on~$\RR^2$.  Since $\omega$ is a probability
measure, this implies convergence in $L^2(\omega)$ for $|\varepsilon|$
sufficiently small.  Hence
\[
\frac{2}{4+(\varepsilon-\eta)^2}
=\sum_{m,n\ge 1}
\langle u_m,u_n\rangle_{L^2(\omega)}\,
\varepsilon^{m-1}\eta^{n-1}.
\]
On the other hand,
\[
\frac{2}{4+(\varepsilon-\eta)^2}
=\frac12\sum_{\ell=0}^{\infty}
(-1)^\ell 4^{-\ell}(\varepsilon-\eta)^{2\ell}.
\]
Comparing coefficients gives the stated formulas.  Mixed parity terms
vanish because the kernel only contains even total degree.
\end{proof}

\begin{theorem}%
\label{thm:mode-space-rkhs}
Let
\[
L^2_0(\omega):=
\left\{
f\in L^2(\omega):\ \int_{\RR}f(y)\,\omega(\dd y)=0
\right\},
\]
and let $\mathcal H_K$ be the reproducing-kernel Hilbert space generated
by the kernel
\[
K(a,b):=\frac{2}{4+(a-b)^2}.
\]
Set
\[
\mathcal S_0:=\operatorname{span}\{\Psi_\varepsilon:\varepsilon\in\RR\},
\qquad
\mathcal S:=\overline{\mathcal S_0}^{\,L^2(\omega)}
\subset L^2_0(\omega).
\]
Then one has
\[
\mathcal S=L^2_0(\omega).
\]
Equivalently, the algebraic span
$\mathcal S_0$ is dense in~$L^2_0(\omega)$.
Moreover, the linear assignment
\[
\mathcal U_0:\mathcal S_0\longrightarrow \mathcal H_K,
\qquad
\mathcal U_0(\Psi_\varepsilon):=K(\cdot,\varepsilon)
\qquad(\varepsilon\in\RR)
\]
is an isometry and extends uniquely to a unitary operator
\[
\mathcal U:L^2_0(\omega)\longrightarrow \mathcal H_K.
\]
\end{theorem}

\begin{proof}
We already noted that $\Psi_\varepsilon\in L^2_0(\omega)$, so
$\mathcal S_0\subset L^2_0(\omega)$.  Let
$f\in L^2_0(\omega)$ be orthogonal to~$\mathcal S_0$.  Then for every
$\varepsilon\in\RR$,
\[
0=\langle f,\Psi_\varepsilon\rangle_{L^2(\omega)}.
\]
If $\varepsilon\neq 0$, multiplying by~$\varepsilon$ gives
\[
0=\int_{\RR}f(y)\bigl(R_\varepsilon(y)-1\bigr)\,\omega(\dd y)
=\int_{\RR}f(y)R_\varepsilon(y)\,\omega(\dd y),
\]
because $\int f\,\dd\omega=0$.  Since
$R_\varepsilon(y)\,\omega(\dd y)=P_1(y-\varepsilon)\,\dd y$, this means
\[
0=\int_{\RR}f(y)P_1(\varepsilon-y)\,\dd y
\qquad(\varepsilon\in\RR).
\]
The integral above is absolutely convergent, because
\[
\int_{\RR}|f(y)|P_1(\varepsilon-y)\dd y
=\int_{\RR}|f(y)|R_\varepsilon(y)\,\omega(\dd y)
\le \|f\|_{L^2(\omega)}\|R_\varepsilon\|_{L^2(\omega)},
\]
and by \eqref{eq:mode-gram-RR},
\[
\|R_\varepsilon\|_{L^2(\omega)}^2
=\int_{\RR}R_\varepsilon(y)^2\,\omega(\dd y)
=1+\frac{\varepsilon^2}{2}.
\]
Thus the Poisson regularization $T_1f=P_1*f$ is a well-defined function
and vanishes identically on~$\RR$.

Define the finite signed Borel measure
\[
\mu_f(E):=\int_E f(y)\,\omega(\dd y),
\qquad E\subset\RR\ \text{Borel}.
\]
Since $\omega$ is a probability measure and $f\in L^2(\omega)$,
\[
|\mu_f|(\RR)\le \|f\|_{L^1(\omega)}\le \|f\|_{L^2(\omega)}.
\]
The preceding identity is therefore
\[
(P_1*\mu_f)(\varepsilon)=0
\qquad(\varepsilon\in\RR).
\]
Taking Fourier transforms in the finite-measure sense yields
\[
e^{-|\xi|}\widehat{\mu_f}(\xi)
=\widehat{P_1}(\xi)\widehat{\mu_f}(\xi)
=\widehat{P_1*\mu_f}(\xi)
=0
\qquad(\xi\in\RR).
\]
Because $\widehat{\mu_f}$ is a bounded continuous function on~$\RR$, it
follows that $\widehat{\mu_f}\equiv 0$.  Uniqueness of the Fourier
transform for finite signed measures \cite[Chapter~II]{Lukacs1970}
therefore gives $\mu_f=0$.  Since $\omega$ is absolutely continuous with
strictly positive density, $f=0$ $\omega$-almost everywhere.  Thus
$\mathcal S_0^\perp=\{0\}$ in $L^2_0(\omega)$, so
\[
\overline{\mathcal S_0}^{\,L^2(\omega)}=L^2_0(\omega).
\]
By definition this closure is~$\mathcal S$, hence
$\mathcal S=L^2_0(\omega)$.

Theorem~\ref{thm:mode-gram-kernel} shows that
\[
\langle \Psi_\varepsilon,\Psi_\eta\rangle_{L^2(\omega)}
=K(\varepsilon,\eta)
=\langle K(\cdot,\varepsilon),K(\cdot,\eta)\rangle_{\mathcal H_K},
\]
so $\mathcal U_0$ is an isometry on~$\mathcal S_0$.  Since the
kernel sections span a dense subspace of~$\mathcal H_K$, the isometry
extends uniquely to a surjective isometry
$\mathcal U:L^2_0(\omega)\to\mathcal H_K$, that is, to a unitary
operator.
\end{proof}

\begin{remark}
For a translation-invariant positive-definite kernel, the existence of
the associated reproducing-kernel Hilbert space is classical
\cite{Aronszajn1950}.  What is special here is that the abstract space
may be identified explicitly with the concrete weighted space
$L^2_0(\omega)$ generated by the secant orbit.
\end{remark}

\subsection{Higher secant kernels and coefficient representation}

For the kernel under consideration, the multilinear integrals appearing
in the entropy expansion from Section~\ref{sec:entropy-asymptotics} can
be organized as derivatives of explicit multilinear kernels built from
the same secant family.

\begin{theorem}[Higher secant kernels and coefficient representation]
\label{thm:higher-secant-kernels}
For each integer $m\ge 2$ define
\[
\mathcal J_m(\varepsilon_1,\dots,\varepsilon_m)
:=
\int_{\RR}\prod_{j=1}^m R_{\varepsilon_j}(y)\,\omega(\dd y),
\]
\[
\Gamma_m(\varepsilon_1,\dots,\varepsilon_m)
:=
\int_{\RR}\prod_{j=1}^m \Psi_{\varepsilon_j}(y)\,\omega(\dd y).
\]
Then the following hold.
\begin{enumerate}[label=(\roman*)]
\item If $\varepsilon_1,\dots,\varepsilon_m$ are pairwise distinct, then
      \begin{equation}\label{eq:higher-secant-residue}
      \mathcal J_m(\varepsilon_1,\dots,\varepsilon_m)
      =
      \sum_{j=1}^m
      \frac{\varepsilon_j(\varepsilon_j+2\ii)}
      {\displaystyle\prod_{\ell\ne j}
      (\varepsilon_j-\varepsilon_\ell)
      (\varepsilon_j-\varepsilon_\ell+2\ii)}.
      \end{equation}
\item The function $\Gamma_m$ is continuous on $\RR^m$, is
      real-analytic in a neighborhood of $(0,\dots,0)$, and for
      nonzero pairwise distinct parameters satisfies
      \begin{equation}\label{eq:higher-secant-inclusion-exclusion}
      \Gamma_m(\varepsilon_1,\dots,\varepsilon_m)
      =
      \frac{1}{\varepsilon_1\cdots\varepsilon_m}
      \sum_{S\subseteq\{1,\dots,m\}}(-1)^{m-|S|}
      \mathcal J_{|S|}\bigl((\varepsilon_j)_{j\in S}\bigr),
      \end{equation}
      with the conventions $\mathcal J_0:=1$ and
      $\mathcal J_1(\varepsilon):=1$.  For $m=2$ one has
      \[
      \Gamma_2(\varepsilon_1,\varepsilon_2)=K(\varepsilon_1,\varepsilon_2).
      \]
\item Let $N\ge 2$, and assume that $\nu$ has finite centered
      $(2N+1)$st absolute moment.  If
      \[
      D_{\mathrm{KL}}(h_t\|g_t)
      =\sum_{r=2}^{N}c_{2r}\,t^{-2r}+o(t^{-2N})
      \qquad(t\to\infty)
      \]
      is the expansion from
      Theorem~\ref{thm:poisson-kl-even-orders}, then
      \begin{equation}\label{eq:higher-secant-coefficients}
      c_{2N}
      =
      \sum_{k=2}^{N}\frac{(-1)^k}{k(k-1)}
      \sum_{\substack{n_1,\dots,n_k\ge 2\\n_1+\cdots+n_k=2N}}
      \mu_{n_1}\cdots\mu_{n_k}\,
      \frac{
      \partial_1^{n_1-1}\cdots \partial_k^{n_k-1}
      \Gamma_k(0,\dots,0)}
      {(n_1-1)!\cdots(n_k-1)!}.
      \end{equation}
      Consequently, within this model every coefficient in the Poisson
      entropy expansion is recovered as a Taylor coefficient of the
      higher secant kernels $\Gamma_k$.
\end{enumerate}
\end{theorem}

\begin{proof}
Fix $m\ge 2$.  For pairwise distinct
$\varepsilon_1,\dots,\varepsilon_m\in\RR$ set
\[
F_m(z)
:=
\frac{1+z^2}{\prod_{j=1}^m(1+(z-\varepsilon_j)^2)}.
\]
Since $F_m(z)=O(|z|^{2-2m})$ in the upper half-plane, the residue
theorem applies on semicircles of radius tending to infinity.  The poles
in the upper half-plane are the simple poles
$z=\varepsilon_j+\ii$ $(1\le j\le m)$.  For each such $j$,
\[
\operatorname{Res}_{z=\varepsilon_j+\ii}F_m
=
\frac{1+(\varepsilon_j+\ii)^2}
{2\ii\displaystyle\prod_{\ell\ne j}
\bigl(1+(\varepsilon_j+\ii-\varepsilon_\ell)^2\bigr)}.
\]
Because
\[
1+(\varepsilon_j+\ii)^2=\varepsilon_j(\varepsilon_j+2\ii)
\]
and
\[
1+(\varepsilon_j+\ii-\varepsilon_\ell)^2
=(\varepsilon_j-\varepsilon_\ell)
(\varepsilon_j-\varepsilon_\ell+2\ii),
\]
the residue identity becomes
\[
\operatorname{Res}_{z=\varepsilon_j+\ii}F_m
=
\frac{\varepsilon_j(\varepsilon_j+2\ii)}
{2\ii\displaystyle\prod_{\ell\ne j}
(\varepsilon_j-\varepsilon_\ell)
(\varepsilon_j-\varepsilon_\ell+2\ii)}.
\]
Hence
\[
\mathcal J_m(\varepsilon_1,\dots,\varepsilon_m)
=\frac{1}{\pi}\int_{\RR}F_m(y)\,\dd y
=2\ii\sum_{j=1}^m \operatorname{Res}_{z=\varepsilon_j+\ii}F_m,
\]
which is exactly \eqref{eq:higher-secant-residue}.  This proves~(i).

For~(ii), continuity of $\Gamma_m$ on $\RR^m$ follows from dominated
convergence on compact parameter sets.  Indeed, if
$|\varepsilon_j|\le M$ for every~$j$, then
\[
\Psi_{\varepsilon_j}(y)
=\frac{2(y-\varepsilon_j)+\varepsilon_j}{1+(y-\varepsilon_j)^2},
\]
so
\[
\abs{\Psi_{\varepsilon_j}(y)}
\le \frac{2\abs{y-\varepsilon_j}}{1+(y-\varepsilon_j)^2}
+\frac{|\varepsilon_j|}{1+(y-\varepsilon_j)^2}
\le 1+M
\qquad(y\in\RR).
\]
Since $\omega$ is a probability measure,
$\prod_{j=1}^m\Psi_{\varepsilon_j}\in L^1(\omega)$ uniformly on such
compact sets.

If every $\varepsilon_j$ is nonzero, then
\[
\Psi_{\varepsilon_j}=\frac{R_{\varepsilon_j}-1}{\varepsilon_j}.
\]
Multiplying these identities and integrating against $\omega$ gives
\eqref{eq:higher-secant-inclusion-exclusion} by expanding the product
\[
\prod_{j=1}^m(R_{\varepsilon_j}-1)
\]
and using the conventions $\mathcal J_0:=1$ and
$\mathcal J_1(\varepsilon):=1$.

To prove the analyticity statement near the origin, fix $0<\rho<1$.
Proposition~\ref{prop:cayley-mode-remainder} yields, for every integer
$N\ge 1$,
\[
\Psi_\varepsilon(y)
=\sum_{n=1}^{N}u_n(y)\varepsilon^{n-1}
+\varepsilon^N v_N(y,\varepsilon),
\qquad |\varepsilon|\le \rho,
\]
where $v_N$ is bounded on $\RR\times[-\rho,\rho]$.  Since
$|u_n(y)|\le 1$ for every $n\ge 1$ and $y\in\RR$, multiplying the
preceding expansions for $\Psi_{\varepsilon_1},\dots,\Psi_{\varepsilon_m}$
shows that
\[
\Gamma_m(\varepsilon_1,\dots,\varepsilon_m)
=
\sum_{n_1,\dots,n_m=1}^{N}
I_{n_1,\dots,n_m}\,
\varepsilon_1^{n_1-1}\cdots\varepsilon_m^{n_m-1}
+\sum_{j=1}^m\varepsilon_j^N G_{j,N}(\varepsilon_1,\dots,\varepsilon_m),
\]
where each $G_{j,N}$ is bounded on $[-\rho,\rho]^m$ and
\[
I_{n_1,\dots,n_m}
:=
\int_{\RR}
u_{n_1}(y)\cdots u_{n_m}(y)\,\omega(\dd y).
\]
This is exactly the Taylor expansion of $\Gamma_m$ at the origin, so
$\Gamma_m$ is real-analytic there.  The identity $\Gamma_2=K$ is
Theorem~\ref{thm:mode-gram-kernel}.

For~(iii), the Taylor expansion from~(ii) gives
\[
I_{n_1,\dots,n_k}
=
\frac{
\partial_1^{n_1-1}\cdots\partial_k^{n_k-1}\Gamma_k(0,\dots,0)}
{(n_1-1)!\cdots(n_k-1)!}
\qquad(n_1,\dots,n_k\ge 1).
\]
The proof of Theorem~\ref{thm:poisson-kl-even-orders} yields
\[
c_{2N}
=
\sum_{k=2}^{N}\frac{(-1)^k}{k(k-1)}
\sum_{\substack{n_1,\dots,n_k\ge 2\\n_1+\cdots+n_k=2N}}
\mu_{n_1}\cdots\mu_{n_k}\,I_{n_1,\dots,n_k}.
\]
Substituting the derivative formula for $I_{n_1,\dots,n_k}$ proves
\eqref{eq:higher-secant-coefficients}.
\end{proof}

\subsection{Centered reconstruction from Poisson data}

We now turn from the Hilbert-space structure to the information carried
by the Poisson smoothing itself.  Evaluating the smoothed profile and
its derivative at the mean produces two scalar channels from which the
centered law may be recovered.

\begin{theorem}%
\label{thm:poisson-central-two-channel}
Let $\nu$ be a probability measure on~$\RR$ with finite mean~$\bar\gamma$.
Let $\nu_c$ denote the law of the centered variable
$\Delta:=\gamma-\bar\gamma$.  Define
\[
h_t(x):=\int_{\RR}P_t(x-\gamma)\dd\nu(\gamma),
\]
the central observation functions
\[
A(t):=\pi\,h_t(\bar\gamma),
\qquad
B(t):=\pi\,\partial_x h_t(x)\big|_{x=\bar\gamma}
\qquad(t>0).
\]
\[
H(t):=\int_t^\infty B(s)\dd s
\qquad(t>0),
\]
and let
\[
\phi_{\nu_c}(r):=\int_{\RR}e^{ir\Delta}\dd\nu(\gamma),
\qquad r\in\RR,
\]
be the centered characteristic function.  Then for every $t>0$ one has
\begin{equation}\label{eq:poisson-laplace-reconstruction}
A(t)+\ii H(t)
=\int_0^\infty e^{-tr}\phi_{\nu_c}(r)\dd r.
\end{equation}
Consequently, the pair $(A,B)$ determines the centered law~$\nu_c$
uniquely.  Equivalently, the triple $(\bar\gamma,A,B)$ determines the
original measure~$\nu$ uniquely.
\end{theorem}

\begin{proof}
By the elementary Laplace identities
\[
\int_0^\infty e^{-tr}\cos(r\Delta)\dd r=\frac{t}{t^2+\Delta^2},
\qquad
\int_0^\infty e^{-tr}\sin(r\Delta)\dd r=\frac{\Delta}{t^2+\Delta^2},
\]
one obtains
\[
A(t)
=\int_{\RR}\frac{t}{t^2+\Delta^2}\dd\nu(\gamma)
=\int_0^\infty e^{-tr}\Re\phi_{\nu_c}(r)\dd r.
\]
Differentiation under the integral is justified by dominated convergence,
since for each fixed $t>0$,
\[
\left|\frac{2t(x-\gamma)}{((x-\gamma)^2+t^2)^2}\right|
\le \frac{1}{2t}
\qquad(x,\gamma\in\RR),
\]
and
\[
\partial_x h_t(x)\big|_{x=\bar\gamma}
=\frac{2}{\pi}\int_{\RR}\frac{t\,\Delta}{(\Delta^2+t^2)^2}\dd\nu(\gamma),
\]
in particular
\[
B(t)=2t\int_{\RR}\frac{\Delta}{(\Delta^2+t^2)^2}\dd\nu(\gamma).
\]
Since
\[
\int_t^\infty \frac{2s|\Delta|}{(\Delta^2+s^2)^2}\dd s
=\frac{|\Delta|}{\Delta^2+t^2}
\le \frac{1}{2t},
\]
Fubini's theorem applies and gives
\[
H(t)
=\int_{\RR}\frac{\Delta}{t^2+\Delta^2}\dd\nu(\gamma)
=\int_0^\infty e^{-tr}\Im\phi_{\nu_c}(r)\dd r.
\]
Combining the two formulas yields
\eqref{eq:poisson-laplace-reconstruction}.  By uniqueness of the Laplace
transform on bounded continuous functions \cite{Widder1941}, applied to
the bounded continuous functions $\Re\phi_{\nu_c}$ and $\Im\phi_{\nu_c}$,
$(A,H)$, and therefore $(A,B)$, determine $\Re\phi_{\nu_c}(r)$ and
$\Im\phi_{\nu_c}(r)$ for every $r\ge 0$.  Hence $\phi_{\nu_c}(r)$ is known
on $[0,\infty)$.  For $r>0$ set
\[
\phi_{\nu_c}(-r):=\overline{\phi_{\nu_c}(r)}.
\]
This is the standard characteristic-function symmetry, so the entire
characteristic function is determined.  Levy's inversion theorem for
characteristic functions then recovers the centered law~$\nu_c$
\cite{Lukacs1970,Feller1971}, and adding the mean recovers~$\nu$.
\end{proof}

\begin{corollary}%
\label{cor:poisson-symmetrization}
Under the hypotheses of Theorem~\ref{thm:poisson-central-two-channel}, the
function $A$ determines the symmetrization of the centered law~$\nu_c$.
\end{corollary}

\begin{proof}
The real part of $\phi_{\nu_c}$ is exactly the characteristic function of
the centered symmetrization
$\frac12(\nu_c+\check\nu_c)$, where $\check\nu_c(E):=\nu_c(-E)$.  Since
$A$ is the Laplace transform of $\Re\phi_{\nu_c}$, the claim follows from
Laplace inversion.
\end{proof}

\begin{corollary}%
\label{cor:poisson-symmetric-case}
Under the hypotheses of Theorem~\ref{thm:poisson-central-two-channel}, the
centered law~$\nu_c$ is symmetric if and only if $B\equiv 0$.  In that
case $A$ alone determines~$\nu_c$, and the pair $(\bar\gamma,A)$
determines~$\nu$.
\end{corollary}

\begin{proof}
The centered law is symmetric if and only if
$\Im\phi_{\nu_c}\equiv 0$.  By
\eqref{eq:poisson-laplace-reconstruction}, this is equivalent to
$H\equiv 0$, hence to $B\equiv 0$.
\end{proof}

\begin{theorem}%
\label{thm:poisson-channel-rkhs}
Let $\nu$ be a probability measure on~$\RR$ with finite mean~$\bar\gamma$.
For $t>0$ define the centered fluctuation profile
\[
\widetilde\delta_t(y):=
\frac{h_t(\bar\gamma+ty)}{g_t(\bar\gamma+ty)}-1,
\qquad y\in\RR.
\]
Then $\widetilde\delta_t\in L^2_0(\omega)$ and
\begin{equation}\label{eq:delta-t-secant}
\widetilde\delta_t(y)
=\frac{1}{t}\int_{\RR}\Delta\,\Psi_{\Delta/t}(y)\,\dd\nu(\gamma).
\end{equation}
Consequently,
\begin{equation}\label{eq:delta-t-rkhs}
(\mathcal U\widetilde\delta_t)(a)
=\frac{1}{t}\int_{\RR}\Delta\,K(a,\Delta/t)\,\dd\nu(\gamma),
\qquad a\in\RR.
\end{equation}
Moreover,
\begin{equation}\label{eq:delta-t-origin}
\widetilde\delta_t(0)=tA(t)-1,
\end{equation}
and
\begin{equation}\label{eq:delta-t-rkhs-origin}
(\mathcal U\widetilde\delta_t)(0)=2t\,H(2t)
=2t\int_{2t}^{\infty}B(s)\,\dd s.
\end{equation}
\end{theorem}

\begin{proof}
By definition,
\[
\widetilde\delta_t(y)
=\int_{\RR}\bigl(R_{\Delta/t}(y)-1\bigr)\,\dd\nu(\gamma)
=\frac{1}{t}\int_{\RR}\Delta\,\Psi_{\Delta/t}(y)\,\dd\nu(\gamma),
\]
which is \eqref{eq:delta-t-secant}.  By Theorem~\ref{thm:mode-gram-kernel},
\[
\|\Psi_\varepsilon\|_{L^2(\omega)}^2
=\frac{2}{4+(\varepsilon-\varepsilon)^2}
=\frac12.
\]
Hence
\[
\left\|\widetilde\delta_t\right\|_{L^2(\omega)}
\le\frac{1}{t}\int_{\RR}|\Delta|\,\left\|\Psi_{\Delta/t}\right\|_{L^2(\omega)}\,\dd\nu(\gamma)
=\frac{1}{\sqrt2\,t}\int_{\RR}|\Delta|\,\dd\nu(\gamma)
<\infty,
\]
so \eqref{eq:delta-t-secant} is a Bochner integral in $L^2(\omega)$.
Since each $\Psi_{\Delta/t}$ belongs to the closed subspace
$L^2_0(\omega)$, the same identity shows directly that
$\widetilde\delta_t\in L^2_0(\omega)$.  Applying the unitary map
$\mathcal U$ from Theorem~\ref{thm:mode-space-rkhs} to
\eqref{eq:delta-t-secant} yields \eqref{eq:delta-t-rkhs}.

At $y=0$ one has
\[
g_t(\bar\gamma)=\frac{1}{\pi t},
\]
so
\[
\widetilde\delta_t(0)
=\frac{h_t(\bar\gamma)}{g_t(\bar\gamma)}-1
=tA(t)-1,
\]
which proves \eqref{eq:delta-t-origin}.  Finally, evaluating
\eqref{eq:delta-t-rkhs} at $a=0$ and using
$K(0,b)=2/(4+b^2)$ gives
\[
(\mathcal U\widetilde\delta_t)(0)
=\frac{1}{t}\int_{\RR}\Delta\frac{2}{4+(\Delta/t)^2}\,\dd\nu(\gamma)
=2t\int_{\RR}\frac{\Delta}{4t^2+\Delta^2}\,\dd\nu(\gamma).
\]
The definition of $H$ in
Theorem~\ref{thm:poisson-central-two-channel} gives
\[
H(2t)=\int_{\RR}\frac{\Delta}{4t^2+\Delta^2}\,\dd\nu(\gamma),
\]
and \eqref{eq:delta-t-rkhs-origin} follows.
\end{proof}

\begin{corollary}[Completely monotone quadratic secant energy]
\label{cor:quadratic-secant-energy}
Let $\nu$ be a probability measure on~$\RR$ with finite mean~$\bar\gamma$,
and let $\nu_c$ be the centered law of $\Delta=\gamma-\bar\gamma$.
In \eqref{eq:quadratic-secant-energy-kernel}, the notation
$\Delta':=\gamma'-\bar\gamma$ is used for the second integration
variable.
Define
\[
\mathcal E_\nu(t)
:=
\|\mathcal U\widetilde\delta_t\|_{\mathcal H_K}^2
=
\|\widetilde\delta_t\|_{L^2(\omega)}^2
\qquad(t>0).
\]
Then
\begin{equation}\label{eq:quadratic-secant-energy-kernel}
\mathcal E_\nu(t)
=
\frac{1}{t^2}
\iint_{\RR^2}\Delta\Delta'\,
K\!\left(\frac{\Delta}{t},\frac{\Delta'}{t}\right)
\dd\nu(\gamma)\dd\nu(\gamma'),
\end{equation}
\begin{equation}\label{eq:quadratic-secant-energy-laplace}
\mathcal E_\nu(t)
=
\frac{1}{t}\int_0^\infty e^{-2tr}
\abs{\phi_{\nu_c}'(r)}^2\dd r.
\end{equation}
Consequently,
\begin{equation}\label{eq:quadratic-secant-energy-complete-monotone}
(-1)^m\frac{\dd^m}{\dd t^m}\bigl(t\mathcal E_\nu(t)\bigr)
=
\int_0^\infty (2r)^m e^{-2tr}
\abs{\phi_{\nu_c}'(r)}^2\dd r
\ge 0
\qquad(m\ge 0,\ t>0),
\end{equation}
so $t\mapsto t\mathcal E_\nu(t)$ is completely monotone on~$(0,\infty)$
in the sense of Bernstein--Widder \cite{Widder1941}.
\end{corollary}

\begin{proof}
The Bochner representation \eqref{eq:delta-t-rkhs} gives
\[
\mathcal U\widetilde\delta_t
=
\frac{1}{t}\int_{\RR}\Delta\,
K\!\left(\cdot,\frac{\Delta}{t}\right)\dd\nu(\gamma)
\]
in $\mathcal H_K$.  Taking the squared norm and using the reproducing
identity
\[
\left\langle
K\!\left(\cdot,\frac{\Delta}{t}\right),
K\!\left(\cdot,\frac{\Delta'}{t}\right)
\right\rangle_{\mathcal H_K}
=
K\!\left(\frac{\Delta}{t},\frac{\Delta'}{t}\right)
\]
yields \eqref{eq:quadratic-secant-energy-kernel}.

For the Fourier representation, Proposition~\ref{thm:gram-space-poisson-image}
gives
\[
\widehat{K(\cdot,a)}(\xi)=\pi e^{-2|\xi|}e^{-\ii a\xi}.
\]
Hence
\[
\widehat{\mathcal U\widetilde\delta_t}(\xi)
=
\frac{\pi e^{-2|\xi|}}{t}
\int_{\RR}\Delta\,e^{-\ii \xi\Delta/t}\dd\nu(\gamma).
\]
Since
\[
\phi_{\nu_c}'(r)
=\int_{\RR}\ii\Delta e^{\ii r\Delta}\dd\nu(\gamma),
\]
the last integral equals $-\ii\,\phi_{\nu_c}'(-\xi/t)$.  Therefore
Proposition~\ref{thm:gram-space-poisson-image} implies
\begin{align*}
\mathcal E_\nu(t)
&=
\frac{1}{2\pi^2}\int_{\RR}e^{2|\xi|}
\abs{\widehat{\mathcal U\widetilde\delta_t}(\xi)}^2\dd\xi \\
&=
\frac{1}{2t^2}\int_{\RR}e^{-2|\xi|}
\abs{\phi_{\nu_c}'(-\xi/t)}^2\dd\xi.
\end{align*}
The identity
\[
\phi_{\nu_c}'(-r)=-\overline{\phi_{\nu_c}'(r)}
\qquad(r\in\RR)
\]
shows that
$\abs{\phi_{\nu_c}'(-r)}=\abs{\phi_{\nu_c}'(r)}$.  After the change of
variables $\xi=tr$, one obtains
\eqref{eq:quadratic-secant-energy-laplace}.

Since $\nu$ has finite first absolute moment,
$\phi_{\nu_c}'$ is bounded on~$\RR$.  The integrand in
\eqref{eq:quadratic-secant-energy-laplace} is therefore dominated by
$\|\phi_{\nu_c}'\|_{L^\infty}^2 e^{-2tr}$ on every half-line
$[t_0,\infty)$ with $t_0>0$.  Differentiation under the integral sign
	then gives \eqref{eq:quadratic-secant-energy-complete-monotone}.
\end{proof}

\subsection{The quotient flow as a non-autonomous Doob transform}

The Poisson comparison quotient
\[
u_t(x):=\frac{h_t(x)}{g_t(x)}
\qquad(t>0,\ x\in\RR)
\]
admits a closed evolution equation with respect to the Cauchy generator.
This identifies the entropy decay as a nonlocal Bregman dissipation and
connects the quadratic secant energy with the $\chi^2$-channel of the
Poisson smoothing.

\begin{theorem}[Non-autonomous Doob transform and $\Phi$-entropy dissipation]\label{thm:poisson-doob-phi-entropy}
Let $\nu$ be a probability measure on~$\RR$ with mean~$\bar\gamma$ and
finite centered second moment.  For $t>0$ define
\[
h_t:=P_t*\nu,
\qquad
g_t(x):=P_t(x-\bar\gamma),
\qquad
u_t(x):=\frac{h_t(x)}{g_t(x)}.
\]
Let
\[
(Lf)(x):=\frac{1}{\pi}\operatorname{p.v.}\int_{\RR}
\frac{f(y)-f(x)}{(x-y)^2}\dd y
\]
be the one-dimensional Cauchy generator, and set
\begin{equation}\label{eq:doob-generator-definition}
(\mathcal L_t f)(x)
:=\frac{1}{g_t(x)}\bigl(L(g_tf)(x)-f(x)Lg_t(x)\bigr).
\end{equation}
Then the following assertions hold.
\begin{enumerate}[label=(\roman*)]
\item
      The function $u_t$ is bounded and belongs to $C^\infty(\RR)$ for
      every $t>0$.  Moreover,
      \begin{equation}\label{eq:quotient-flow-equation}
      \partial_tu_t=\mathcal L_tu_t
      \qquad\text{on }\RR\times(0,\infty),
      \end{equation}
      and \eqref{eq:doob-generator-definition} may be written explicitly
      as
      \begin{equation}\label{eq:doob-generator-kernel}
      (\mathcal L_tf)(x)
      =\frac{1}{\pi g_t(x)}\operatorname{p.v.}\int_{\RR}
      \frac{g_t(y)\bigl(f(y)-f(x)\bigr)}{(x-y)^2}\dd y.
      \end{equation}
\item
      For every $f,\psi\in C_c^\infty(\RR)$ one has the weighted Green
      identity
      \begin{equation}\label{eq:doob-green-identity}
      \begin{split}
      \int_{\RR}f(x)(\mathcal L_t\psi)(x)\,g_t(x)^2\dd x
      &=
      -\frac{1}{2\pi}
      \iint_{\RR^2}
      \frac{g_t(x)g_t(y)}{(x-y)^2}
      \\
      &\qquad\qquad\times
      \bigl(f(x)-f(y)\bigr)\bigl(\psi(x)-\psi(y)\bigr)\dd x\dd y.
      \end{split}
      \end{equation}
\item
      Let $\Phi:[0,\infty)\to\RR$ be convex, of class $C^1$, with
      locally Lipschitz derivative and $\Phi(1)=0$.  Define
      \[
      \mathcal H_\Phi(t):=\int_{\RR}g_t(x)\Phi(u_t(x))\dd x.
      \]
      Then $\mathcal H_\Phi(t)<\infty$ for every $t>0$,
      $t\mapsto\mathcal H_\Phi(t)$ is absolutely continuous on compact
      subintervals of $(0,\infty)$, and
      \begin{equation}\label{eq:phi-entropy-dissipation}
      -\frac{\dd}{\dd t}\mathcal H_\Phi(t)
      =
      \frac{1}{2\pi}\iint_{\RR^2}
      \frac{
      g_t(x)D_\Phi(u_t(x),u_t(y))
      +g_t(y)D_\Phi(u_t(y),u_t(x))}
      {(x-y)^2}\dd x\dd y,
      \end{equation}
      where
      \[
      D_\Phi(a,b):=\Phi(a)-\Phi(b)-\Phi'(b)(a-b)
      \qquad(a,b\ge 0).
      \]
      In particular, $t\mapsto\mathcal H_\Phi(t)$ is nonincreasing.
\item
      For $\Phi(s)=s\log s$ one has
      \[
      \mathcal H_\Phi(t)=D_{\mathrm{KL}}(h_t\|g_t),
      \]
      and
      \begin{equation}\label{eq:relative-entropy-integrated-dissipation}
      D_{\mathrm{KL}}(h_t\|g_t)
      =
      \int_t^\infty \mathcal I_\nu(s)\dd s,
      \end{equation}
      where
      \begin{equation}\label{eq:relative-entropy-dissipation-rate}
      \mathcal I_\nu(s)
      :=
      \frac{1}{\pi}\iint_{\RR^2}
      \frac{g_s(x)\Lambda(u_s(x),u_s(y))}{(x-y)^2}\dd x\dd y,
      \qquad
      \Lambda(a,b):=a\log\frac{a}{b}-a+b.
      \end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}
For $t>0$ the functions $h_t$ and $g_t$ are positive and belong to
$C^\infty(\RR)$, because they are obtained by convolving the smooth
Poisson kernel with a probability measure.  Writing
$x=\bar\gamma+ty$ and $\varepsilon=\Delta/t$, one has
\[
u_t(x)=\int_{\RR}R_{\Delta/t}(y)\dd\nu(\gamma).
\]
The elementary estimate
\[
1+y^2=1+\bigl((y-\varepsilon)+\varepsilon\bigr)^2
\le 2\bigl(1+(y-\varepsilon)^2\bigr)+2\varepsilon^2
\]
gives
\[
0<R_\varepsilon(y)\le 2+2\varepsilon^2
\qquad(y,\varepsilon\in\RR),
\]
and therefore
\[
0<u_t(x)\le 2+\frac{2\mu_2}{t^2},
\qquad
\mu_2:=\int_{\RR}(\gamma-\bar\gamma)^2\dd\nu(\gamma).
\]
Likewise,
\[
1+(y-\varepsilon)^2
\le 2\bigl(1+y^2\bigr)+2\varepsilon^2
\le 2(1+\varepsilon^2)(1+y^2),
\]
so
\[
R_\varepsilon(y)\ge \frac{1}{2(1+\varepsilon^2)}
\qquad(y,\varepsilon\in\RR).
\]
Hence
\[
u_t(x)\ge \frac12\int_{\RR}\frac{1}{1+(\Delta/t)^2}\dd\nu(\gamma)
\ge \frac{1}{2(1+\mu_2/t^2)},
\]
where the last inequality follows from Jensen's inequality applied to the
convex function $s\mapsto(1+s)^{-1}$ on $[0,\infty)$.
Thus $u_t$ is bounded for every fixed $t>0$.

By Proposition~\ref{thm:poisson-semigroup-law},
$\partial_t h_t=-|D|h_t$ and $\partial_t g_t=-|D|g_t$.
The singular-integral representation of the Cauchy generator
\cite[Definition~2.2 and Proposition~2.4]{Kwasnicki2017} gives
\[
\partial_t h_t=Lh_t,
\qquad
\partial_t g_t=Lg_t,
\]
with $L$ as above.  Since $h_t=u_tg_t$ and $g_t(x)>0$,
\[
\partial_tu_t(x)
=\frac{Lh_t(x)-u_t(x)Lg_t(x)}{g_t(x)}.
\]
Substituting $h_t=u_tg_t$ and combining the two principal-value integrals
yields
\[
\partial_tu_t(x)
=\frac{1}{\pi g_t(x)}\operatorname{p.v.}\int_{\RR}
\frac{g_t(y)\bigl(u_t(y)-u_t(x)\bigr)}{(x-y)^2}\dd y.
\]
This is \eqref{eq:quotient-flow-equation} and proves
\eqref{eq:doob-generator-kernel}; part~(i) follows.

For part~(ii), let $f,\psi\in C_c^\infty(\RR)$.  By
\eqref{eq:doob-generator-kernel},
\[
\int_{\RR}f(x)(\mathcal L_t\psi)(x)\,g_t(x)^2\dd x
=
\frac{1}{\pi}\iint_{\RR^2}
\frac{f(x)g_t(x)g_t(y)\bigl(\psi(y)-\psi(x)\bigr)}
{(x-y)^2}\dd x\dd y.
\]
Interchanging $x$ and $y$ in the same integral and averaging the two
representations gives \eqref{eq:doob-green-identity}.

Fix a compact interval $[a,b]\subset(0,\infty)$.  The preceding bound for
$u_t$ implies that the range of $u_t$ on $\RR\times[a,b]$ is contained in
a compact subinterval of $(0,\infty)$.  Since $\Phi'$ is locally
Lipschitz, there is a constant $C_{\Phi,a,b}$ such that
\[
0\le D_\Phi(\alpha,\beta)\le C_{\Phi,a,b}|\alpha-\beta|^2
\]
for all $\alpha,\beta$ in that range.  Because $u_t$ is smooth and
bounded, this gives
\[
D_\Phi(u_t(x),u_t(y))
\le C_{\Phi,a,b}|u_t(x)-u_t(y)|^2
\le C'_{\Phi,a,b}\min\{|x-y|^2,1\}.
\]
Consequently,
\[
\frac{g_t(x)D_\Phi(u_t(x),u_t(y))}{(x-y)^2}
\]
is dominated by an $L^1(\RR^2)$-function, uniformly for $t\in[a,b]$,
because $g_t\in L^1(\RR)$ and
$\int_{\RR}\min\{|x-y|^{-2},1\}\dd y<\infty$.  It follows that
$\mathcal H_\Phi(t)$ is finite and differentiable on~$[a,b]$, with
\[
\frac{\dd}{\dd t}\mathcal H_\Phi(t)
=
\int_{\RR}\Phi'(u_t)\,\partial_th_t\dd x
+
\int_{\RR}\bigl(\Phi(u_t)-u_t\Phi'(u_t)\bigr)\partial_tg_t\dd x.
\]
Substituting $\partial_th_t=L(u_tg_t)$ and $\partial_tg_t=Lg_t$, and then
applying the bilinear identity associated with the principal-value
formula for~$L$,
\[
\int_{\RR}\varphi\,Lf\dd x
=
-\frac{1}{2\pi}\iint_{\RR^2}
\frac{\bigl(f(x)-f(y)\bigr)\bigl(\varphi(x)-\varphi(y)\bigr)}
{(x-y)^2}\dd x\dd y,
\]
gives
\[
\frac{\dd}{\dd t}\mathcal H_\Phi(t)
=
-\frac{1}{2\pi}\iint_{\RR^2}
\frac{E_t(x,y)}{(x-y)^2}\dd x\dd y,
\]
where
\begin{align*}
E_t(x,y)
&:=
\bigl(u_t(x)g_t(x)-u_t(y)g_t(y)\bigr)
\bigl(\Phi'(u_t(x))-\Phi'(u_t(y))\bigr)
\\
&\quad
+\bigl(g_t(x)-g_t(y)\bigr)
\Bigl(
\Phi(u_t(x))-u_t(x)\Phi'(u_t(x))
-\Phi(u_t(y))+u_t(y)\Phi'(u_t(y))
\Bigr).
\end{align*}
Set
\[
a:=u_t(x),
\qquad
b:=u_t(y),
\qquad
g_x:=g_t(x),
\qquad
g_y:=g_t(y).
\]
A direct rearrangement yields
\[
E_t(x,y)
=g_x\bigl(\Phi(a)-\Phi(b)-\Phi'(b)(a-b)\bigr)
+g_y\bigl(\Phi(b)-\Phi(a)-\Phi'(a)(b-a)\bigr),
\]
that is,
\[
E_t(x,y)=g_t(x)D_\Phi(u_t(x),u_t(y))
+g_t(y)D_\Phi(u_t(y),u_t(x)).
\]
Substituting this identity proves
\eqref{eq:phi-entropy-dissipation}.  Convexity of~$\Phi$ gives
$D_\Phi\ge 0$, so $\mathcal H_\Phi$ is nonincreasing.  Since the choice
of $[a,b]$ was arbitrary, part~(iii) follows.

For part~(iv), if $\Phi(s)=s\log s$, then
\[
\mathcal H_\Phi(t)
=\int_{\RR}g_t(x)u_t(x)\log u_t(x)\dd x
=D_{\mathrm{KL}}(h_t\|g_t).
\]
The corresponding Bregman divergence is exactly
$D_\Phi(a,b)=\Lambda(a,b)$.  Hence
\eqref{eq:relative-entropy-dissipation-rate} is the specialization of
\eqref{eq:phi-entropy-dissipation}.  Proposition~\ref{thm:poisson-second-order}
gives $D_{\mathrm{KL}}(h_t\|g_t)=O(t^{-4})$ as $t\to\infty$, so
$D_{\mathrm{KL}}(h_t\|g_t)\to 0$.  Integrating the derivative identity from
$t$ to $\infty$ proves
\eqref{eq:relative-entropy-integrated-dissipation}.
\end{proof}

\begin{corollary}[Quadratic secant energy as a $\chi^2$-channel]\label{cor:quadratic-energy-chi-square}
Let $\nu$ be a probability measure on~$\RR$ with finite centered second
moment.  Then, for every $t>0$,
\begin{equation}\label{eq:quadratic-energy-chi-square}
\mathcal E_\nu(t)
=
\chi^2(h_t\|g_t)
:=
\int_{\RR}\frac{(h_t(x)-g_t(x))^2}{g_t(x)}\dd x
=
\frac{1}{t}\int_0^\infty e^{-2tr}\abs{\phi_{\nu_c}'(r)}^2\dd r.
\end{equation}
Moreover,
\begin{equation}\label{eq:quadratic-energy-dissipation}
-\frac{\dd}{\dd t}\mathcal E_\nu(t)
=
\frac{1}{\pi}\iint_{\RR^2}
\frac{g_t(x)\bigl(u_t(x)-u_t(y)\bigr)^2}{(x-y)^2}\dd x\dd y.
\end{equation}
\end{corollary}

\begin{proof}
By definition,
\[
\widetilde\delta_t(y)=u_t(\bar\gamma+ty)-1.
\]
Since
\[
g_t(\bar\gamma+ty)\,t\,\dd y
=\frac{1}{\pi(1+y^2)}\dd y
=\omega(\dd y),
\]
one obtains
\[
\mathcal E_\nu(t)
=\|\widetilde\delta_t\|_{L^2(\omega)}^2
=\int_{\RR}g_t(x)\bigl(u_t(x)-1\bigr)^2\dd x
=\int_{\RR}\frac{(h_t(x)-g_t(x))^2}{g_t(x)}\dd x.
\]
The Laplace representation is exactly
\eqref{eq:quadratic-secant-energy-laplace}.  Applying
Theorem~\ref{thm:poisson-doob-phi-entropy} with
$\Phi(s)=(s-1)^2$ gives
\[
D_\Phi(a,b)=(a-b)^2,
\]
hence \eqref{eq:quadratic-energy-dissipation}.
\end{proof}

\begin{remark}
The previous theorem recovers the centered Cauchy transform on the
imaginary axis:
\[
G_{\nu_c}(\ii t)
=\int_{\RR}\frac{1}{\ii t-\Delta}\dd\nu(\gamma)
=-H(t)-\ii A(t)
\qquad(t>0).
\]
Thus the Laplace-transform formulation repackages the same
Cauchy-transform identity in a way that isolates the loss of the
translation parameter.  In particular, the proof of
Theorem~\ref{thm:poisson-central-two-channel} is most naturally
interpreted as a Laplace-uniqueness plus
characteristic-function-inversion argument rather than as a Stieltjes
inversion argument.
\end{remark}
