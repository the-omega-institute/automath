\subsection{Pullback of Haar measure and rigidity}

The Jacobian of the Cayley parametrization identifies normalized arc
length on the circle with the Cauchy weight on the line.  Since this is
the only role played by the circle model in the sequel, we record both
the resulting change-of-variables formula and the associated rigidity
statement.

\subsubsection{The pullback measure}

Set
\begin{equation}\label{eq:cauchy-measure}
\dd\mu(x):=\frac{1}{\pi(1+x^{2})}\dd x.
\end{equation}

\begin{proposition}\label{prop:phase-derivative}
The phase coordinate \eqref{eq:phase-coord} satisfies
\begin{equation}\label{eq:phase-deriv}
\upsilon'(x)=\frac{1}{\pi(1+x^{2})},
\qquad
\vartheta'(x)=\frac{2}{1+x^{2}}.
\end{equation}
Consequently, $\upsilon:(\RR,\mu)\to(\UU,\dd u)$ is measure preserving.
In particular, for every non-negative Borel function $f$ on~$\UU$,
\begin{equation}\label{eq:weighted-integral}
\int_{\RR}f(\upsilon(x))\dd\mu(x)=\int_{\UU}f(u)\dd u.
\end{equation}
\end{proposition}

\begin{proof}
Differentiate \eqref{eq:phase-coord}.  The change-of-variables formula
then gives \eqref{eq:weighted-integral}.
\end{proof}

\subsubsection{Rigidity of the Cayley chart}

\begin{theorem}%
\label{thm:haar-pullback-uniqueness}
The Cayley transform satisfies
\begin{equation}\label{eq:haar-pullback}
\iota^{*}m_{\TT}=\mu.
\end{equation}
More precisely, let
$\varphi:\overline\RR\to\TT$ be an orientation-preserving homeomorphism
such that $\varphi|_{\RR}$ is of class $C^{1}$,
$\varphi(0)=1$, and $\varphi(\infty)=-1$.  Then the following are
equivalent:
\begin{enumerate}[label=(\roman*)]
\item $\varphi^{*}m_{\TT}=\mu$ on~$\RR$;
\item if $\varphi(x)=e^{\ii\alpha(x)}$ with
      $\alpha:\RR\to(-\pi,\pi)$ increasing, then
      $\alpha'(x)=2/(1+x^{2})$ for every $x\in\RR$;
\item $\varphi=\iota$.
\end{enumerate}
\end{theorem}

\begin{proof}
For $\iota$ itself,
\[
\iota^{*}m_{\TT}
=\frac{1}{2\pi}\vartheta'(x)\dd x
=\frac{1}{\pi(1+x^{2})}\dd x
=\dd\mu(x),
\]
so \eqref{eq:haar-pullback} holds.

Now let $\varphi$ be as in the statement.  Because $\varphi$ is
orientation preserving and maps $\RR$ onto $\TT\setminus\{-1\}$, there
is a unique increasing $C^{1}$-bijection
$\alpha:\RR\to(-\pi,\pi)$ such that $\varphi(x)=e^{\ii\alpha(x)}$ and
$\alpha(0)=0$.  Pulling back arc length gives
\[
\varphi^{*}m_{\TT}=\frac{1}{2\pi}\alpha'(x)\dd x.
\]
Hence (i) is equivalent to
\[
\alpha'(x)=\frac{2}{1+x^{2}},
\]
which is exactly (ii).  Integrating and using $\alpha(0)=0$ yields
\[
\alpha(x)=2\arctan x.
\]
Thus $\varphi(x)=e^{\ii\alpha(x)}=\iota(x)$, so (ii) implies~(iii).  The
implication (iii)$\Rightarrow$(i) has already been proved.
\end{proof}

\begin{remark}\label{rem:orientation-reversal}
If the orientation assumption is removed, the same argument shows that
$\varphi^{*}m_{\TT}=\mu$ implies either $\varphi=\iota$ or
$\varphi=\overline{\iota}$.  Thus the chart is unique up to complex
conjugation.
\end{remark}
