\section{Lattice Sampling for the Secant Kernel}%
\label{app:strip-lattice}

This appendix specializes the standard principal shift-invariant-space
framework \cite{deBoorDeVoreRon1994,RonShen1995,AldroubiGrochenig2001}
to the secant Gram kernel $K(a,b)=2/(4+(a-b)^2)$.
Unlike Appendix~\ref{app:gram-space}, it is not used in the proofs of
the main entropy theorems; we retain it only as a companion closed-form
specialization for the same kernel.
The novelty is limited to three explicit formulas: the closed-form
lattice symbol, the cardinal kernel's Fourier description, and the
exact interpolation norm.  All structural claims (Riesz-basis property,
cardinal reconstruction, invertibility) are direct consequences of the
standard theory once the symbol is computed.

\paragraph{Complex inner product convention.}
We work with the complexification of $\mathcal H_K$, inner product
linear in the first variable, with norm
$\|f\|_{\mathcal H_K}^2 := \frac{1}{2\pi^2}\int_\RR e^{2|\xi|}|\widehat f(\xi)|^2\dd\xi$.

\subsection{Lattice symbol, Riesz basis, and cardinal reconstruction}

\begin{proposition}[Lattice sampling]%
\label{thm:poisson-lattice-sampling}
Define
\[
\mathcal H_K(\ZZ)
:=\overline{\operatorname{span}}\{K(\cdot,n):n\in\ZZ\}\subset\mathcal H_K.
\]
Set
\begin{equation}\label{eq:lattice-sampling-constants}
A_{\ZZ}:=\frac{\pi}{\sinh(2\pi)},\qquad B_{\ZZ}:=\pi\coth(2\pi).
\end{equation}
Then $\{K(\cdot,n)\}_{n\in\ZZ}$ is a Riesz basis for $\mathcal H_K(\ZZ)$
with bounds $A_{\ZZ}\le\sigma\le B_{\ZZ}$, where the lattice symbol is
\begin{equation}\label{eq:lattice-symbol}
\sigma(\theta)
=\sum_{n\in\ZZ}K(0,n)e^{-\ii n\theta}
=\frac{\pi\cosh(2(\pi-|\theta|))}{\sinh(2\pi)},
\qquad\theta\in[-\pi,\pi].
\end{equation}
For every $\alpha\in\ell^2(\ZZ)$ there is a unique interpolant
$F_\alpha\in\mathcal H_K(\ZZ)$ with $F_\alpha(n)=\alpha_n$, and
\begin{equation}\label{eq:cardinal-norm-formula}
\|F_\alpha\|_{\mathcal H_K}^2
=\frac{1}{2\pi}\int_{-\pi}^{\pi}
\frac{|\widehat\alpha(\theta)|^2}{\sigma(\theta)}\dd\theta.
\end{equation}
\end{proposition}

\begin{proof}
Set $\widehat c(\theta):=\sum_n c_n e^{-\ii n\theta}$ and
$F_c:=\sum_n c_n K(\cdot,n)$.  By Theorem~\ref{thm:mode-gram-kernel} and
Parseval,
\[
\|F_c\|_{\mathcal H_K}^2
=\frac{1}{2\pi}\int_{-\pi}^\pi\sigma(\theta)|\widehat c(\theta)|^2\dd\theta.
\]
Poisson summation with $\widehat k(\xi)=\pi e^{-2|\xi|}$ gives
\eqref{eq:lattice-symbol}.  The bounds $A_\ZZ\le\sigma\le B_\ZZ$ are
immediate, establishing the Riesz inequalities.  The synthesis map
extends to an isomorphism $\ell^2(\ZZ)\to\mathcal H_K(\ZZ)$ by density.
The Gram operator $G=(K(m,n))_{m,n}$ is bounded and invertible with
symbol $\sigma$; the unique interpolant $F_\alpha=\mathcal C_K(G^{-1}\alpha)$
satisfies \eqref{eq:cardinal-norm-formula}.
\end{proof}

\begin{remark}
The symbol \eqref{eq:lattice-symbol} and the norm identity
\eqref{eq:cardinal-norm-formula} are the explicit new content of this
appendix relative to the general theory in
\cite{deBoorDeVoreRon1994,RonShen1995,AldroubiGrochenig2001}.
Analogous closed-form formulas are known for Gaussian generators
\cite{RomeroUlanovskiiZlotnikov2024,BaranovBelovGrochenig2022};
for the Cauchy/Lorentz integer-shift family, Riesz and interpolation
statements of this type appear in
\cite{KiselevMininNovikov2016,KiselevMininNovikov2019}.
\end{remark}
