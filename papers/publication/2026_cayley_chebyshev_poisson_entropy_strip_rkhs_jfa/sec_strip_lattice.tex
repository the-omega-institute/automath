\section{The Strip RKHS and Lattice Consequences}%
\label{sec:strip-lattice}

\paragraph{Complex Hilbert space convention.}
From this point onward, we work with the complexification of the real
Hilbert space from Section~\ref{sec:gram-space}, still denoted
$\mathcal H_K$.  Its inner product is linear in the first variable and
conjugate-linear in the second:
\[
\langle f,g\rangle_{\mathcal H_K}
:=\frac{1}{2\pi^2}\int_{\RR}e^{2|\xi|}\widehat f(\xi)\,
\overline{\widehat g(\xi)}\,\dd\xi.
\]
All reproducing kernels satisfy the Hermitian symmetry
$K(z,w)=\overline{K(w,z)}$, and evaluation representers depend on
$\overline w$, not on~$w$, in the second slot.

\paragraph{Role of the lattice results.}
This section develops the Hilbert and lattice side of the same secant mode
family used in the entropy expansion.  Proposition~\ref{prop:gram-space-poisson-image}
identifies $\mathcal H_K$ with the Poisson image space
$\sqrt{\pi}\,e^{-|D|}L^2(\RR)$ and provides the strip model, while
Proposition~\ref{prop:poisson-hardy-splitting} separates positive and
negative Fourier supports into two Hardy-type summands.  For the lattice
part we use the standard principal shift-invariant-space mechanism on the
closed integer-orbit spaces $\mathcal H_K(\ZZ)$ and $\mathcal S_{\ZZ}$.
The Poisson-specific inputs are the closed form of the lattice symbol
\eqref{eq:lattice-symbol}, the cardinal kernel formula
\eqref{eq:cardinal-kernel-fourier}, and the exact norm identity
\eqref{eq:cardinal-norm-formula}.  The final theorem then connects this
lattice chain back to the Poisson observation profiles from
Theorem~\ref{thm:poisson-channel-rkhs}: integer samples recover exactly the
lattice projection of the observation channel, and recover the full channel
precisely in the lattice-subspace case.

\subsection{A Poisson image realization of the mode space}

\begin{proposition}[The mode Gram space as a Poisson image space]%
\label{prop:gram-space-poisson-image}
Let
\[
K(a,b):=\frac{2}{4+(a-b)^2}=\pi\,P_2(a-b),
\qquad a,b\in\RR,
\]
and let $\mathcal H_K$ be the reproducing-kernel Hilbert space generated
by~$K$.  Define
\[
\mathcal X
:=
\left\{
f\in L^2(\RR):
\int_{\RR}e^{2|\xi|}\abs{\widehat f(\xi)}^2\dd\xi<\infty
\right\},
\]
with norm
\[
\|f\|_{\mathcal X}^2
:=\frac{1}{2\pi^2}\int_{\RR}e^{2|\xi|}\abs{\widehat f(\xi)}^2\dd\xi.
\]
Then $\mathcal H_K=\mathcal X$ isometrically.  Equivalently,
\[
\mathcal H_K=\sqrt{\pi}\,T_1(L^2(\RR)),
\qquad
\|\sqrt{\pi}\,T_1 g\|_{\mathcal H_K}=\|g\|_{L^2(\RR)}
\quad(g\in L^2(\RR)).
\]
Every $f\in\mathcal H_K$ admits a unique holomorphic extension to the
strip
\[
\mathbb S:=\{z\in\CC:\ |\Im z|<1\},
\]
given by
\[
f(z)=\frac{1}{2\pi}\int_{\RR}\widehat f(\xi)e^{\ii z\xi}\dd\xi,
\qquad z\in\mathbb S,
\]
and satisfying the sharp pointwise estimate
\[
|f(x+\ii y)|\le \frac{1}{\sqrt{2(1-y^2)}}\,\|f\|_{\mathcal H_K}
\qquad(x\in\RR,\ |y|<1).
\]
\end{proposition}

\begin{proof}
The Fourier transform of the translation kernel $k(x):=2/(4+x^2)$ is
\[
\widehat k(\xi)=\pi e^{-2|\xi|}.
\]
Hence
\[
\widehat{K(\cdot,a)}(\xi)=\pi e^{-2|\xi|}e^{-\ii a\xi}.
\]
For $f\in\mathcal X$ one therefore has
\begin{align*}
\langle f,K(\cdot,a)\rangle_{\mathcal X}
&=
\frac{1}{2\pi^2}\int_{\RR}
e^{2|\xi|}\widehat f(\xi)\,
\overline{\widehat{K(\cdot,a)}(\xi)}\dd\xi \\
&=
\frac{1}{2\pi}\int_{\RR}\widehat f(\xi)e^{\ii a\xi}\dd\xi
=f(a)
\end{align*}
by Fourier inversion.  Thus $\mathcal X$ is a reproducing-kernel Hilbert
space with kernel~$K$, and the Moore--Aronszajn uniqueness theorem yields
$\mathcal X=\mathcal H_K$ \cite{Aronszajn1950}.

Now let $g\in L^2(\RR)$ and set $f:=\sqrt{\pi}\,T_1 g$.  Since
\[
\widehat f(\xi)=\sqrt{\pi}\,e^{-|\xi|}\widehat g(\xi),
\]
Plancherel gives
\[
\|f\|_{\mathcal X}^2
=\frac{1}{2\pi^2}\int_{\RR}e^{2|\xi|}\pi e^{-2|\xi|}
\abs{\widehat g(\xi)}^2\dd\xi
=\frac{1}{2\pi}\int_{\RR}\abs{\widehat g(\xi)}^2\dd\xi
=\|g\|_{L^2}^2.
\]
Conversely, if $f\in\mathcal X$, define
\[
\widehat g(\xi):=\pi^{-1/2}e^{|\xi|}\widehat f(\xi).
\]
Then $g\in L^2(\RR)$ and
$\widehat f=\sqrt{\pi}\,e^{-|\xi|}\widehat g$, so
$f=\sqrt{\pi}\,T_1 g$.  This proves the image-space description.

If $z=x+\ii y$ with $|y|<1$, Cauchy--Schwarz gives
\begin{align*}
|f(z)|
&\le \frac{1}{2\pi}
\left(\int_{\RR}e^{2|\xi|}\abs{\widehat f(\xi)}^2\dd\xi\right)^{1/2}
\left(\int_{\RR}e^{-2|\xi|}e^{-2y\xi}\dd\xi\right)^{1/2} \\
&=
\frac{1}{\sqrt{2(1-y^2)}}\,\|f\|_{\mathcal X},
\end{align*}
because
\[
\int_{\RR}e^{-2|\xi|}e^{-2y\xi}\dd\xi=\frac{1}{1-y^2}.
\]
The constant is sharp: at each $z=x+\ii y$ equality is attained by the
normalized reproducing kernel whose Fourier transform is proportional to
$e^{-2|\xi|}e^{-\ii\overline z\xi}$.
Hence the integral defining $f(z)$ converges absolutely and locally
uniformly on $\mathbb S$, so the extension is holomorphic there.  The same
estimate shows uniqueness by continuity from the real axis.
\end{proof}

\begin{remark}\label{rem:poisson-image-framework}
The identity $K(a,b) = \pi P_2(a-b)$ means that the Gram kernel is,
up to the factor~$\pi$, the Poisson kernel at time~$2$.  This explains
why the RKHS is a Poisson image space: the kernel itself arises from the
semigroup.  The holomorphic extension and pointwise estimate above are
the standard $H^2$ Poisson-boundary statements transported to a
translation-invariant strip model; see
\cite{Duren1970,Rudin1987}.
\end{remark}

\subsection{Hardy splitting, lattice sampling, and cardinal reconstruction}

\begin{proposition}[Hardy splitting of the strip model]%
\label{prop:poisson-hardy-splitting}
Let
\[
\mathcal H_K^+
:=
\left\{
f\in\mathcal H_K:\ \widehat f(\xi)=0 \text{ for a.e. }\xi<0
\right\},
\]
\[
\mathcal H_K^-
:=
\left\{
f\in\mathcal H_K:\ \widehat f(\xi)=0 \text{ for a.e. }\xi>0
\right\}.
\]
Then
\[
\mathcal H_K=\mathcal H_K^+\oplus \mathcal H_K^-
\]
orthogonally.  Moreover, each $f_\pm\in\mathcal H_K^\pm$ extends
holomorphically to $\mathbb S$, and $\mathcal H_K^\pm$ is a
reproducing-kernel Hilbert space on~$\mathbb S$ with kernel
\begin{equation}\label{eq:hardy-splitting-kernels}
K_+(z,w)=\frac{1}{2(2-\ii(z-\overline w))},
\qquad
K_-(z,w)=\frac{1}{2(2+\ii(z-\overline w))}.
\end{equation}
For $a,b\in\RR$ one has
\begin{equation}\label{eq:hardy-splitting-real-kernel}
K(a,b)=K_+(a,b)+K_-(a,b).
\end{equation}
\end{proposition}

\begin{proof}
The norm formula from Proposition~\ref{prop:gram-space-poisson-image}
shows that $\mathcal H_K^+$ and $\mathcal H_K^-$ are closed subspaces of
$\mathcal H_K$, and they are orthogonal because their Fourier supports are
disjoint.  The decomposition
\[
\widehat f
=\widehat f\,\mathbf 1_{[0,\infty)}
+\widehat f\,\mathbf 1_{(-\infty,0]}
\]
therefore yields the asserted orthogonal sum.

For $w\in\mathbb S$, define $K_\pm(\cdot,w)$ by
\[
\widehat{K_+(\cdot,w)}(\xi)
:=
\pi e^{-2\xi}e^{-\ii\overline w\,\xi}\mathbf 1_{[0,\infty)}(\xi),
\]
\[
\widehat{K_-(\cdot,w)}(\xi)
:=
\pi e^{2\xi}e^{-\ii\overline w\,\xi}\mathbf 1_{(-\infty,0]}(\xi).
\]
Since $|\Im w|<1$, these functions belong to $\mathcal H_K^\pm$.  If
$f_+\in\mathcal H_K^+$, then
\begin{align*}
\langle f_+,K_+(\cdot,w)\rangle_{\mathcal H_K}
&=
\frac{1}{2\pi^2}\int_0^\infty
e^{2\xi}\widehat f_+(\xi)\,
\overline{\pi e^{-2\xi}e^{-\ii\overline w\,\xi}}\dd\xi \\
&=
\frac{1}{2\pi}\int_0^\infty \widehat f_+(\xi)e^{\ii w\xi}\dd\xi
=f_+(w).
\end{align*}
The same argument gives the reproducing identity for $\mathcal H_K^-$.  By
Fourier inversion,
\[
K_+(z,w)
=\frac{1}{2\pi}\int_0^\infty
\pi e^{-2\xi}e^{\ii(z-\overline w)\xi}\dd\xi
=\frac{1}{2(2-\ii(z-\overline w))},
\]
\[
K_-(z,w)
=\frac{1}{2\pi}\int_{-\infty}^0
\pi e^{2\xi}e^{\ii(z-\overline w)\xi}\dd\xi
=\frac{1}{2(2+\ii(z-\overline w))},
\]
which proves \eqref{eq:hardy-splitting-kernels}.  Adding the two kernels
on the real axis yields \eqref{eq:hardy-splitting-real-kernel}.
\end{proof}

We next specialize the standard fiberization description of a principal
shift-invariant space and its orthogonal projection to the kernel
$K(a,b)=2/(4+(a-b)^2)$; see
\cite{deBoorDeVoreRon1994,RonShen1995,AldroubiGrochenig2001}.  In the
present model both the periodized weight and the projector can be
written in closed form.

\begin{theorem}\label{thm:uniform-lattice-projection}
Let $h>0$ and set
\[
I_h:=\Bigl[-\frac{\pi}{h},\frac{\pi}{h}\Bigr],
\qquad
\Sigma_h(\xi):=\sum_{m\in\ZZ}e^{-2\abs{\xi+2\pi m/h}}
\qquad(\xi\in I_h).
\]
Then
\begin{equation}\label{eq:lattice-periodized-weight}
\Sigma_h(\xi)
=\frac{\cosh(2(\pi/h-\abs{\xi}))}{\sinh(2\pi/h)}
\qquad(\xi\in I_h).
\end{equation}
Define
\[
\mathcal H_K(h\ZZ)
:=
\overline{\operatorname{span}}\{K(\cdot,hn):n\in\ZZ\}
\subset\mathcal H_K
\]
and
\[
\mathcal N_h
:=
\{f\in\mathcal H_K:\ f(hn)=0\ \text{for every }n\in\ZZ\}.
\]
For $f\in\mathcal H_K$ write
\[
q_f(\xi):=e^{2\abs{\xi}}\widehat f(\xi).
\]
Then the following hold.
\begin{enumerate}[label=(\roman*)]
\item
      One has $f\in\mathcal H_K(h\ZZ)$ if and only if $q_f$ is
      $2\pi/h$-periodic and
      \[
      \int_{I_h}\Sigma_h(\xi)\abs{q_f(\xi)}^2\dd\xi<\infty.
      \]
      In that case,
      \begin{equation}\label{eq:lattice-subspace-norm}
      \|f\|_{\mathcal H_K}^2
      =\frac{1}{2\pi^2}\int_{I_h}\Sigma_h(\xi)\abs{q_f(\xi)}^2\dd\xi.
      \end{equation}
\item
      The orthogonal complement of~$\mathcal H_K(h\ZZ)$ is exactly
      $\mathcal N_h$, and
      \begin{equation}\label{eq:lattice-alias-cancellation}
      f\in\mathcal N_h
      \iff
      \sum_{m\in\ZZ}\widehat f\!\left(\xi+\frac{2\pi m}{h}\right)=0
      \quad\text{for a.e. }\xi\in I_h.
      \end{equation}
      Consequently,
      \[
      \mathcal H_K=\mathcal H_K(h\ZZ)\oplus \mathcal N_h
      \]
      orthogonally.
\item
      Let $\Pi_h:\mathcal H_K\to\mathcal H_K(h\ZZ)$ denote the orthogonal
      projection.  Then, for almost every $\xi\in I_h$, the series
      \[
      \sum_{m\in\ZZ}\widehat f\!\left(\xi+\frac{2\pi m}{h}\right)
      \]
      converges absolutely and
      \begin{equation}\label{eq:lattice-alias-projection}
      q_{\Pi_hf}(\xi)
      =\frac{\displaystyle\sum_{m\in\ZZ}
      \widehat f\!\left(\xi+\frac{2\pi m}{h}\right)}
      {\Sigma_h(\xi)},
      \end{equation}
      with $q_{\Pi_hf}$ extended $2\pi/h$-periodically to~$\RR$.
      Equivalently,
      \[
      \widehat{\Pi_hf}(\xi)=e^{-2\abs{\xi}}q_{\Pi_hf}(\xi)
      \qquad\text{for a.e. }\xi\in\RR.
      \]
      Moreover,
      \begin{equation}\label{eq:lattice-projection-norm}
      \|\Pi_hf\|_{\mathcal H_K}^2
      =\frac{1}{2\pi^2}\int_{I_h}
      \frac{\left|\sum_{m\in\ZZ}
      \widehat f\!\left(\xi+\frac{2\pi m}{h}\right)\right|^2}
      {\Sigma_h(\xi)}\,\dd\xi,
      \end{equation}
      and $\Pi_hf$ is the unique element of~$\mathcal H_K(h\ZZ)$
      satisfying
      \[
      (\Pi_hf)(hn)=f(hn)
      \qquad(n\in\ZZ).
      \]
\end{enumerate}
\end{theorem}

\begin{proof}
Let
\[
a:=\frac{2\pi}{h}.
\]
For $\xi\in I_h$ one has $\abs{\xi}\le a/2$, so
\begin{align*}
\Sigma_h(\xi)
&=e^{-2\abs{\xi}}
+\sum_{m=1}^{\infty}e^{-2(\abs{\xi}+ma)}
+\sum_{m=1}^{\infty}e^{-2(ma-\abs{\xi})} \\
&=\frac{e^{-2\abs{\xi}}+e^{-2a}e^{2\abs{\xi}}}{1-e^{-2a}}
=\frac{\cosh(a-2\abs{\xi})}{\sinh a},
\end{align*}
which is \eqref{eq:lattice-periodized-weight}.

If
\[
F=\sum_{n=-N}^{N}c_nK(\cdot,hn),
\]
then
\[
\widehat F(\xi)=\pi e^{-2\abs{\xi}}\sum_{n=-N}^{N}c_ne^{-\ii hn\xi},
\]
hence
\[
q_F(\xi)=\pi\sum_{n=-N}^{N}c_ne^{-\ii hn\xi},
\]
which is $a$-periodic.  Using the norm formula from
Proposition~\ref{prop:gram-space-poisson-image} and partitioning~$\RR$
into the translates $I_h+ma$ gives
\[
\|F\|_{\mathcal H_K}^2
=\frac{1}{2\pi^2}\int_{I_h}
\sum_{m\in\ZZ}e^{-2\abs{\xi+ma}}\abs{q_F(\xi)}^2\dd\xi
=\frac{1}{2\pi^2}\int_{I_h}\Sigma_h(\xi)\abs{q_F(\xi)}^2\dd\xi.
\]
Passing to the closure proves one direction of~(i).

Conversely, let $q$ be $a$-periodic with
\[
\int_{I_h}\Sigma_h(\xi)\abs{q(\xi)}^2\dd\xi<\infty.
\]
Since $\Sigma_h$ is continuous and strictly positive on the compact
interval~$I_h$, the spaces
$L^2(I_h,\Sigma_h(\xi)\dd\xi)$ and $L^2(I_h,\dd\xi)$ coincide with
equivalent norms.  The trigonometric polynomials
\[
\sum_{n=-N}^{N}d_ne^{-\ii hn\xi}
\]
are therefore dense in $L^2(I_h,\Sigma_h(\xi)\dd\xi)$.  Choose such a
sequence $p_N$ with $p_N\to \pi^{-1}q$ in that weighted norm, and define
$F_N\in\mathcal H_K(h\ZZ)$ by
\[
F_N:=\sum_{n=-N}^{N}d_nK(\cdot,hn).
\]
Then $q_{F_N}=\pi p_N$, so the preceding norm identity gives
\[
\|F_N-F_M\|_{\mathcal H_K}^2
=\frac{1}{2\pi^2}\int_{I_h}\Sigma_h(\xi)
\abs{q_{F_N}(\xi)-q_{F_M}(\xi)}^2\dd\xi.
\]
Hence $(F_N)$ is Cauchy in~$\mathcal H_K$ and converges to some
$F\in\mathcal H_K(h\ZZ)$ with $q_F=q$.  This proves~(i).

For every $f\in\mathcal H_K$ and almost every $\xi\in I_h$,
Cauchy--Schwarz gives
\[
\sum_{m\in\ZZ}\abs{\widehat f(\xi+ma)}
\le
\Sigma_h(\xi)^{1/2}
\left(
\sum_{m\in\ZZ}e^{2\abs{\xi+ma}}
\abs{\widehat f(\xi+ma)}^2
\right)^{1/2},
\]
so the alias sum is absolutely convergent for almost every~$\xi$.

If $g\in\mathcal H_K(h\ZZ)$, then by~(i) its profile $q_g$ is
$a$-periodic.  Using the Fourier norm formula again,
\begin{align*}
\langle f,g\rangle_{\mathcal H_K}
&=
\frac{1}{2\pi^2}\int_{I_h}
\sum_{m\in\ZZ}e^{-2\abs{\xi+ma}}q_f(\xi+ma)\,
\overline{q_g(\xi)}\dd\xi \\
&=
\frac{1}{2\pi^2}\int_{I_h}
\left(
\sum_{m\in\ZZ}\widehat f(\xi+ma)
\right)\overline{q_g(\xi)}\dd\xi.
\end{align*}
Since the periodic profiles $q_g$ form a dense subspace of
$L^2(I_h,\Sigma_h(\xi)\dd\xi)$ and $\Sigma_h$ is bounded above and below
on~$I_h$, the same set is dense in $L^2(I_h,\dd\xi)$.  Therefore
$f\perp\mathcal H_K(h\ZZ)$ if and only if the alias sum in
\eqref{eq:lattice-alias-cancellation} vanishes almost everywhere on~$I_h$.

By the reproducing property,
\[
f(hn)=\langle f,K(\cdot,hn)\rangle_{\mathcal H_K}
\qquad(n\in\ZZ),
\]
and the closed linear span of the kernel sections $K(\cdot,hn)$ is
exactly $\mathcal H_K(h\ZZ)$.  Consequently,
\[
f\in\mathcal N_h
\iff
f\perp\mathcal H_K(h\ZZ).
\]
This proves~(ii).

Let $g\in\mathcal H_K(h\ZZ)$ and write $q_g$ for its periodic profile.
Then
\[
\|f-g\|_{\mathcal H_K}^2
=\frac{1}{2\pi^2}\int_{I_h}
\sum_{m\in\ZZ}e^{-2\abs{\xi+ma}}
\abs{q_f(\xi+ma)-q_g(\xi)}^2\dd\xi.
\]
For each $\xi\in I_h$, the integrand is minimized when $q_g(\xi)$ is the
weighted average of the fiber values $q_f(\xi+ma)$ with weights
$e^{-2\abs{\xi+ma}}$.  Since
\[
e^{-2\abs{\xi+ma}}q_f(\xi+ma)=\widehat f(\xi+ma),
\]
the minimizing profile is exactly the one in
\eqref{eq:lattice-alias-projection}.  This proves the projection formula.
Substituting the minimizing value into the norm identity gives
\eqref{eq:lattice-projection-norm}.  Finally,
\[
f-\Pi_hf\in\mathcal N_h
\]
by~(ii), so $(\Pi_hf)(hn)=f(hn)$ for every $n\in\ZZ$.  If
$G\in\mathcal H_K(h\ZZ)$ satisfies $G(hn)=f(hn)$ for all $n$, then
$G-\Pi_hf\in\mathcal H_K(h\ZZ)\cap\mathcal N_h=\{0\}$, hence
$G=\Pi_hf$.  This proves~(iii).
\end{proof}

\paragraph{Positioning against prior shift-invariant and Cauchy sampling theory.}
The Riesz bounds and lattice reconstruction below are instances of the
standard shift-invariant-space machinery for principal generators, as in the
classical structure theory of finitely generated shift-invariant spaces
and stable translates
\cite{deBoorDeVoreRon1994,RonShen1995,Bownik2000,AldroubiGrochenig2001,GrochenigRomeroStockler2020}.
They also fit the general RKHS sampling and Galerkin reconstruction
frameworks of
\cite{NashedWalter1991,NashedSun2010,ChengJiangSun2016}, and more
broadly the cardinal-interpolation and radial-basis-function literature
\cite{Buhmann2003}.
In particular, the symbol criterion in
Proposition~\ref{prop:poisson-lattice-sampling} is the usual Fourier-symbol
approach.  The calculation specific to the present Poisson strip is the
explicit symbol computation
Proposition~\ref{prop:poisson-lattice-symbol}; the later cardinal norm and
Fourier formula are the corresponding closed-form consequences of that
symbol.

For the Cauchy/Lorentz generator, integer-shift explicit nod-function and
Riesz-constant computations appear in the existing Cauchy--type literature,
including integer shifts and nod-function constructions
\cite{KiselevMininNovikov2016,KiselevMininNovikov2019}.
For the present paper, the additional content beyond the abstract
shift-invariant framework is the Poisson-strip realization of the secant
mode space together with the closed-form lattice symbol
\eqref{eq:lattice-symbol}, and the exact cardinal Fourier formula and
norm formula
\eqref{eq:cardinal-kernel-fourier},\eqref{eq:cardinal-norm-formula}.

\begin{proposition}[Explicit Poisson-strip lattice symbol]%
\label{prop:poisson-lattice-symbol}
Let
\[
K(a,b):=\frac{2}{4+(a-b)^2},
\qquad
\sigma(\theta):=\sum_{n\in\ZZ}K(n,0)e^{-\ii n\theta}
=\sum_{n\in\ZZ}\frac{2}{4+n^2}e^{-\ii n\theta}.
\]
Then $\sigma$ is continuous and $2\pi$-periodic, and for
$\theta\in[-\pi,\pi]$ one has
\begin{equation}\label{eq:lattice-symbol}
\sigma(\theta)
=\frac{\pi\cosh(2(\pi-|\theta|))}{\sinh(2\pi)}.
\end{equation}
Consequently the sharp lattice-symbol bounds are
\[
A_{\ZZ}:=\frac{\pi}{\sinh(2\pi)}
\le
\sigma(\theta)
\le
\pi\coth(2\pi)=:B_{\ZZ}
\qquad(\theta\in[-\pi,\pi]).
\]
Moreover, for every finitely supported scalar sequence
$c=(c_n)_{n\in\ZZ}$, with
\[
\widetilde c(\theta):=\sum_{n\in\ZZ}c_ne^{-\ii n\theta},
\]
one has the exact Gram identity
\begin{equation}\label{eq:lattice-symbol-form}
\sum_{m,n\in\ZZ}c_m\overline{c_n}\,K(m,n)
=\frac{1}{2\pi}\int_{-\pi}^{\pi}
\sigma(\theta)|\widetilde c(\theta)|^2\dd\theta .
\end{equation}
\end{proposition}

\begin{proof}
Set $k(x):=K(x,0)=2/(4+x^2)$.  With the Fourier convention used in this
paper,
\[
\widehat k(\xi)=\pi e^{-2|\xi|}.
\]
The defining series for $\sigma$ is absolutely and uniformly convergent,
so $\sigma$ is continuous and $2\pi$-periodic.
The Poisson summation formula \cite[Chapter~III]{Stein1971} gives
\[
\sigma(\theta)=\sum_{m\in\ZZ}\widehat k(\theta+2\pi m)
=\pi\sum_{m\in\ZZ}e^{-2|\theta+2\pi m|}.
\]
For $0\le\theta\le\pi$ this geometric series equals
\[
\pi\left(e^{-2\theta}
+\sum_{j=1}^{\infty}e^{-2(\theta+2\pi j)}
+\sum_{j=1}^{\infty}e^{-2(2\pi j-\theta)}\right)
=\pi\,\frac{\cosh(2(\pi-\theta))}{\sinh(2\pi)}.
\]
The expression is even in~$\theta$, so \eqref{eq:lattice-symbol} follows
on $[-\pi,\pi]$, and periodicity gives the full symbol.
The minimum and maximum of the displayed formula occur respectively at
$|\theta|=\pi$ and $\theta=0$, which gives the sharp constants
$A_{\ZZ}$ and $B_{\ZZ}$.

Finally, since $K(m,n)=k(m-n)$, Parseval's identity for the discrete
Fourier transform gives
\[
\sum_{m,n\in\ZZ}c_m\overline{c_n}\,K(m,n)
=\frac{1}{2\pi}\int_{-\pi}^{\pi}
\left(\sum_{\ell\in\ZZ}k(\ell)e^{-\ii\ell\theta}\right)
|\widetilde c(\theta)|^2\dd\theta,
\]
which is exactly \eqref{eq:lattice-symbol-form}.
\end{proof}

\begin{proposition}[Standard symbol sampling for the Poisson lattice]%
\label{prop:poisson-lattice-sampling}
Define the closed shift-generated Hilbert subspaces
\[
\mathcal H_K(\ZZ)
:=
\overline{\operatorname{span}}\{K(\cdot,n):n\in\ZZ\}
\subset \mathcal H_K,
\]
\[
\mathcal S_{\ZZ}
:=
\overline{\operatorname{span}}\{\Psi_n:n\in\ZZ\}
\subset L^2_0(\omega).
\]
Thus $\mathcal H_K(\ZZ)$ and $\mathcal S_{\ZZ}$ are the Hilbert closures
of finite linear combinations of the corresponding kernel sections and
secants.
Set
\begin{equation}\label{eq:lattice-sampling-constants}
A_{\ZZ}:=\frac{\pi}{\sinh(2\pi)},
\qquad
B_{\ZZ}:=\pi\coth(2\pi).
\end{equation}
Applying the standard positive-symbol criterion for a principal
shift-invariant system to the explicit Poisson-strip symbol
of Proposition~\ref{prop:poisson-lattice-symbol} gives the following
Riesz and sampling conclusions: the families $\{K(\cdot,n)\}_{n\in\ZZ}$ and
$\{\Psi_n\}_{n\in\ZZ}$ are Riesz bases for
$\mathcal H_K(\ZZ)$ and $\mathcal S_{\ZZ}$, respectively.  More
precisely, for every
finitely supported scalar sequence $(c_n)_{n\in\ZZ}$ one has
\begin{equation}\label{eq:lattice-riesz-kernel}
A_{\ZZ}\sum_{n\in\ZZ}|c_n|^2
\le
\left\|
\sum_{n\in\ZZ}c_nK(\cdot,n)
\right\|_{\mathcal H_K}^2
\le
B_{\ZZ}\sum_{n\in\ZZ}|c_n|^2,
\end{equation}
\begin{equation}\label{eq:lattice-riesz-secant}
A_{\ZZ}\sum_{n\in\ZZ}|c_n|^2
\le
\left\|
\sum_{n\in\ZZ}c_n\Psi_n
\right\|_{L^2(\omega)}^2
\le
B_{\ZZ}\sum_{n\in\ZZ}|c_n|^2.
\end{equation}
Consequently, the synthesis maps
\[
\mathcal C_K(c):=\sum_{n\in\ZZ}c_nK(\cdot,n),
\qquad
\mathcal C_S(c):=\sum_{n\in\ZZ}c_n\Psi_n,
\qquad c\in c_{00}(\ZZ),
\]
extend uniquely to topological isomorphisms
\[
\mathcal C_K:\ell^2(\ZZ)\xrightarrow{\sim}\mathcal H_K(\ZZ),
\qquad
\mathcal C_S:\ell^2(\ZZ)\xrightarrow{\sim}\mathcal S_{\ZZ}.
\]
If $F\in\mathcal H_K(\ZZ)$, then
\begin{equation}\label{eq:lattice-sampling-HK}
A_{\ZZ}\|F\|_{\mathcal H_K}^2
\le
\sum_{n\in\ZZ}|F(n)|^2
\le
B_{\ZZ}\|F\|_{\mathcal H_K}^2.
\end{equation}
Consequently, the restriction map
\[
R_{\ZZ}:\mathcal H_K(\ZZ)\to \ell^2(\ZZ),
\qquad
R_{\ZZ}F:=(F(n))_{n\in\ZZ},
\]
is an isomorphism of Hilbert spaces.  Equivalently, every
$\alpha\in\ell^2(\ZZ)$ admits a unique interpolant
$F_\alpha\in\mathcal H_K(\ZZ)$ satisfying
\[
F_\alpha(n)=\alpha_n
\qquad(n\in\ZZ).
\]
\end{proposition}

\begin{proof}
We spell out the specialization of the classical Fourier-symbol criterion
\cite{deBoorDeVoreRon1994,RonShen1995,AldroubiGrochenig2001} because the
explicit constants come from Proposition~\ref{prop:poisson-lattice-symbol}.
For finitely supported $c=(c_n)_{n\in\ZZ}$ set
\[
F_c:=\sum_{n\in\ZZ}c_nK(\cdot,n),
\qquad
\widetilde c(\theta):=\sum_{n\in\ZZ}c_ne^{-\ii n\theta},
\qquad
\theta\in[-\pi,\pi].
\]
By Theorem~\ref{thm:mode-gram-kernel},
\[
\|F_c\|_{\mathcal H_K}^2
=\sum_{m,n\in\ZZ}c_m\overline{c_n}\,K(m,n)
=\sum_{m,n\in\ZZ}c_m\overline{c_n}\,\frac{2}{4+(m-n)^2}.
\]
Thus \eqref{eq:lattice-symbol-form} from
Proposition~\ref{prop:poisson-lattice-symbol} gives
\[
\|F_c\|_{\mathcal H_K}^2
=\frac{1}{2\pi}\int_{-\pi}^{\pi}\sigma(\theta)
|\widetilde c(\theta)|^2\dd\theta.
\]
Combining the sharp symbol bounds from
Proposition~\ref{prop:poisson-lattice-symbol} with Parseval for
$\widetilde c$ proves \eqref{eq:lattice-riesz-kernel}.  Since
$\mathcal U\Psi_n=K(\cdot,n)$ by
Theorem~\ref{thm:mode-space-rkhs}, \eqref{eq:lattice-riesz-secant}
follows immediately.
The two inequalities show that $\mathcal C_K$ and $\mathcal C_S$
are bounded above and below on the dense subspace $c_{00}(\ZZ)$ of
$\ell^2(\ZZ)$.  Hence they extend uniquely to bounded topological
isomorphisms from $\ell^2(\ZZ)$ onto their closed ranges.  By definition
those closed ranges are precisely $\mathcal H_K(\ZZ)$ and
$\mathcal S_{\ZZ}$, so the stated Riesz-basis assertions follow.

Now let $G$ be the Toeplitz Gram operator on $\ell^2(\ZZ)$ associated
with the matrix $(K(m,n))_{m,n\in\ZZ}$.  For finitely supported $c$ one
has $R_{\ZZ}F_c=Gc$, so
\[
\sum_{n\in\ZZ}|F_c(n)|^2
=\|Gc\|_{\ell^2}^2
=\frac{1}{2\pi}\int_{-\pi}^{\pi}\sigma(\theta)^2
|\widetilde c(\theta)|^2\dd\theta.
\]
Since $A_{\ZZ}\le \sigma\le B_{\ZZ}$, comparison with
\eqref{eq:lattice-symbol-form} yields
\[
A_{\ZZ}\|F_c\|_{\mathcal H_K}^2
\le
\sum_{n\in\ZZ}|F_c(n)|^2
\le
B_{\ZZ}\|F_c\|_{\mathcal H_K}^2.
\]
Passing to limits in $\mathcal H_K(\ZZ)$ proves
\eqref{eq:lattice-sampling-HK}.

The symbol bounds imply that $G$ extends to a boundedly invertible
operator on $\ell^2(\ZZ)$.  Thus for every $\alpha\in\ell^2(\ZZ)$ there is
a unique $c\in\ell^2(\ZZ)$ with $Gc=\alpha$.  By
\eqref{eq:lattice-riesz-kernel}, the series $\sum_n c_nK(\cdot,n)$
converges in $\mathcal H_K(\ZZ)$ to some $F_\alpha$, and then
$R_{\ZZ}F_\alpha=\alpha$.  Uniqueness follows from the lower bound in
\eqref{eq:lattice-sampling-HK}.
\end{proof}

\begin{corollary}[Explicit lattice cardinal kernel and norm]%
\label{cor:poisson-cardinal-reconstruction}
Under the hypotheses of Proposition~\ref{prop:poisson-lattice-sampling}, let
\[
L:=R_{\ZZ}^{-1}\delta_0\in\mathcal H_K(\ZZ).
\]
Here $\mathcal H_K(\ZZ)$ is the closed span of the kernel sections
$\{K(\cdot,n):n\in\ZZ\}$.
Then $L(n)=\delta_{0n}$ for every $n\in\ZZ$.  For every
$\alpha=(\alpha_n)_{n\in\ZZ}\in\ell^2(\ZZ)$, the series
\[
F_\alpha(z):=\sum_{n\in\ZZ}\alpha_nL(z-n)
\]
converges in $\mathcal H_K(\ZZ)$ and uniformly on compact subsets of the
strip $\mathbb S$, and it is the unique function in $\mathcal H_K(\ZZ)$
satisfying
\[
F_\alpha(n)=\alpha_n
\qquad(n\in\ZZ).
\]
If
\[
\widetilde\alpha(\theta):=\sum_{n\in\ZZ}\alpha_ne^{-\ii n\theta},
\qquad \theta\in[-\pi,\pi],
\]
then
\begin{equation}\label{eq:cardinal-norm-formula}
\|F_\alpha\|_{\mathcal H_K}^2
=\frac{1}{2\pi}\int_{-\pi}^{\pi}
\frac{|\widetilde\alpha(\theta)|^2}{\sigma(\theta)}\dd\theta,
\end{equation}
where $\sigma$ is the lattice symbol from \eqref{eq:lattice-symbol}.  If
$d\in\ell^2(\ZZ)$ is the unique sequence satisfying
\[
\widetilde d(\theta)=\sigma(\theta)^{-1}
\qquad\text{for a.e. }\theta\in[-\pi,\pi],
\]
then
\[
L(z)=\sum_{n\in\ZZ}d_nK(z,n)
\]
in $\mathcal H_K$, and for almost every $\theta\in[-\pi,\pi]$ and every
$m\in\ZZ$ one has
\begin{equation}\label{eq:cardinal-kernel-fourier}
\widehat L(\theta+2\pi m)
=\frac{\pi e^{-2|\theta+2\pi m|}}{\sigma(\theta)}.
\end{equation}
\end{corollary}

\begin{proof}
For $n\in\ZZ$ define the translation operator
$\tau_nF(z):=F(z-n)$.  Since
\[
K(z-n,m)=K(z,m+n),
\]
integer translations preserve $\mathcal H_K(\ZZ)$.  By definition,
$R_{\ZZ}L=\delta_0$.  If $S_n$ denotes the shift operator on
$\ell^2(\ZZ)$ given by $(S_na)_m:=a_{m-n}$, then
$R_{\ZZ}\tau_n=S_nR_{\ZZ}$, so for every $n\in\ZZ$
one has
\[
R_{\ZZ}(\tau_nL)=S_n\delta_0=\delta_n.
\]
Hence $\tau_nL$ is the unique interpolant of $\delta_n$, and therefore
$L(n)=\delta_{0n}$.

If $\alpha$ has finite support, then
\[
R_{\ZZ}\left(\sum_{n\in\ZZ}\alpha_nL(\cdot-n)\right)
=\sum_{n\in\ZZ}\alpha_n\delta_n
=\alpha.
\]
By the uniqueness part of
Proposition~\ref{prop:poisson-lattice-sampling}, the corresponding interpolant
is exactly
\[
F_\alpha=\sum_{n\in\ZZ}\alpha_nL(\cdot-n).
\]
For general $\alpha\in\ell^2(\ZZ)$, let $\alpha^{(N)}$ be the truncation
to $\{-N,\dots,N\}$.  Then $\alpha^{(N)}\to\alpha$ in $\ell^2(\ZZ)$, and
the continuity of $R_{\ZZ}^{-1}$ implies
\[
F_{\alpha^{(N)}}=R_{\ZZ}^{-1}\alpha^{(N)}
\longrightarrow
R_{\ZZ}^{-1}\alpha=:F_\alpha
\]
in $\mathcal H_K(\ZZ)$.  Since each $F_{\alpha^{(N)}}$ is a finite
cardinal sum, the series for $F_\alpha$ converges in $\mathcal H_K(\ZZ)$.
Proposition~\ref{prop:gram-space-poisson-image} gives the uniform estimate
\[
|F(z)|\le \frac{1}{\sqrt{2(1-r^2)}}\,\|F\|_{\mathcal H_K}
\qquad(|\Im z|\le r<1),
\]
so $\mathcal H_K$-norm convergence implies uniform convergence on every
compact subset of~$\mathbb S$.  The interpolation identity for $F_\alpha$
follows from the continuity of $R_{\ZZ}$.

To prove \eqref{eq:cardinal-norm-formula}, first suppose that $\alpha$
has finite support and write
\[
F_\alpha=\sum_{n\in\ZZ}c_nK(\cdot,n).
\]
Then $Gc=\alpha$, where $G$ is the Toeplitz Gram operator from the proof
of Proposition~\ref{prop:poisson-lattice-sampling}.  In the discrete Fourier
representation this becomes
\[
\widetilde\alpha(\theta)=\sigma(\theta)\widetilde c(\theta)
\qquad\text{for a.e. }\theta\in[-\pi,\pi].
\]
Using \eqref{eq:lattice-symbol-form}, we obtain
\[
\|F_\alpha\|_{\mathcal H_K}^2
=\frac{1}{2\pi}\int_{-\pi}^{\pi}
\sigma(\theta)|\widetilde c(\theta)|^2\dd\theta
=\frac{1}{2\pi}\int_{-\pi}^{\pi}
\frac{|\widetilde\alpha(\theta)|^2}{\sigma(\theta)}\dd\theta.
\]
Since $1/\sigma\in L^\infty([-\pi,\pi])$, the right-hand side is
continuous in $\alpha\in\ell^2(\ZZ)$, so the formula extends to all
$\alpha$ by density.

Now let $d:=G^{-1}\delta_0$.  Then $R_{\ZZ}^{-1}\delta_0=L$ gives
\[
L=\sum_{n\in\ZZ}d_nK(\cdot,n)
\]
in $\mathcal H_K$, and the discrete Fourier transform diagonalizes $G$,
so
\[
\widetilde d(\theta)=\sigma(\theta)^{-1}
\qquad\text{for a.e. }\theta\in[-\pi,\pi].
\]
Taking Fourier transforms on~$\RR$ and using
$\widehat{K(\cdot,n)}(\xi)=\pi e^{-2|\xi|}e^{-\ii n\xi}$, we find
\[
\widehat L(\xi)
=\pi e^{-2|\xi|}\sum_{n\in\ZZ}d_ne^{-\ii n\xi}.
\]
Writing $\xi=\theta+2\pi m$ with $\theta\in[-\pi,\pi]$ and $m\in\ZZ$,
the periodicity of the discrete Fourier series yields
\[
\sum_{n\in\ZZ}d_ne^{-\ii n(\theta+2\pi m)}
=\widetilde d(\theta)
=\sigma(\theta)^{-1},
\]
which proves \eqref{eq:cardinal-kernel-fourier}.
\end{proof}

\begin{theorem}[Integer-orbit sampling of Poisson observation channels]%
\label{thm:poisson-observation-lattice-sampling}
Let $\nu$ be a probability measure on~$\RR$ with finite mean
$\bar\gamma$, let $t>0$, and let $\widetilde\delta_t$ be the centred
fluctuation profile from Theorem~\ref{thm:poisson-channel-rkhs}.  Put
\[
F_t:=\mathcal U\widetilde\delta_t\in\mathcal H_K,
\]
and let $\Pi_{\ZZ}$ denote the orthogonal projection of $\mathcal H_K$ onto
$\mathcal H_K(\ZZ)$.  Define the integer-orbit observation samples
\[
\alpha_n(t):=F_t(n),\qquad n\in\ZZ .
\]
Then
\begin{equation}\label{eq:poisson-observation-integer-samples}
\alpha_n(t)
=\langle \widetilde\delta_t,\Psi_n\rangle_{L^2(\omega)}
=\frac1t\int_{\RR}
(\gamma-\bar\gamma)\,
K\!\left(n,\frac{\gamma-\bar\gamma}{t}\right)\dd\nu(\gamma),
\qquad n\in\ZZ .
\end{equation}
Moreover $\alpha(t):=(\alpha_n(t))_{n\in\ZZ}\in\ell^2(\ZZ)$ and
\begin{equation}\label{eq:poisson-observation-lattice-stability}
A_{\ZZ}\|\Pi_{\ZZ}F_t\|_{\mathcal H_K}^2
\le
\sum_{n\in\ZZ}|\alpha_n(t)|^2
\le
B_{\ZZ}\|\Pi_{\ZZ}F_t\|_{\mathcal H_K}^2
\le
B_{\ZZ}\|\widetilde\delta_t\|_{L^2(\omega)}^2.
\end{equation}
The cardinal reconstruction with these observable samples is exactly the
lattice projection:
\begin{equation}\label{eq:poisson-observation-cardinal-projection}
\Pi_{\ZZ}F_t(z)
=\sum_{n\in\ZZ}\alpha_n(t)L(z-n)
\qquad(z\in\mathbb S),
\end{equation}
with convergence in $\mathcal H_K(\ZZ)$ and uniformly on compact subsets
of~$\mathbb S$, and
\begin{equation}\label{eq:poisson-observation-cardinal-norm}
\|\Pi_{\ZZ}F_t\|_{\mathcal H_K}^2
=\frac1{2\pi}\int_{-\pi}^{\pi}
\frac{|\widetilde\alpha_t(\theta)|^2}{\sigma(\theta)}\dd\theta,
\qquad
\widetilde\alpha_t(\theta):=
\sum_{n\in\ZZ}\alpha_n(t)e^{-\ii n\theta}.
\end{equation}
Finally,
\[
F_t-\Pi_{\ZZ}F_t\in\mathcal H_K(\ZZ)^\perp
\qquad\text{and}\qquad
(F_t-\Pi_{\ZZ}F_t)(n)=0\quad(n\in\ZZ).
\]
Thus the integer orbit recovers the full Poisson observation channel if
and only if $F_t\in\mathcal H_K(\ZZ)$; otherwise it recovers exactly the
observable lattice component and nothing from the orthogonal component.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:poisson-channel-rkhs} gives
$F_t=\mathcal U\widetilde\delta_t\in\mathcal H_K$ and the formula
\[
F_t(a)=\frac1t\int_{\RR}
(\gamma-\bar\gamma)\,
K\!\left(a,\frac{\gamma-\bar\gamma}{t}\right)\dd\nu(\gamma).
\]
Taking $a=n\in\ZZ$ gives the last expression in
\eqref{eq:poisson-observation-integer-samples}.  Since
$\mathcal U\Psi_n=K(\cdot,n)$ and evaluation in $\mathcal H_K$ is
represented by $K(\cdot,n)$,
\[
F_t(n)
=\langle F_t,K(\cdot,n)\rangle_{\mathcal H_K}
=\langle \widetilde\delta_t,\Psi_n\rangle_{L^2(\omega)}.
\]
This proves \eqref{eq:poisson-observation-integer-samples}.

For every $n\in\ZZ$, the vector $K(\cdot,n)$ belongs to
$\mathcal H_K(\ZZ)$.  Hence
$F_t-\Pi_{\ZZ}F_t$ is orthogonal to $K(\cdot,n)$, and therefore
\[
(\Pi_{\ZZ}F_t)(n)
=\langle \Pi_{\ZZ}F_t,K(\cdot,n)\rangle_{\mathcal H_K}
=\langle F_t,K(\cdot,n)\rangle_{\mathcal H_K}
=F_t(n)=\alpha_n(t).
\]
Applying the sampling inequality
\eqref{eq:lattice-sampling-HK} from
Proposition~\ref{prop:poisson-lattice-sampling} to
$\Pi_{\ZZ}F_t\in\mathcal H_K(\ZZ)$ proves that
$\alpha(t)\in\ell^2(\ZZ)$ and gives the first two inequalities in
\eqref{eq:poisson-observation-lattice-stability}.  The last inequality
follows from the contractivity of the orthogonal projection and the
unitarity of~$\mathcal U$.

By Corollary~\ref{cor:poisson-cardinal-reconstruction}, there is a unique
element of $\mathcal H_K(\ZZ)$ with lattice data $\alpha(t)$, and it is
the cardinal sum in
\eqref{eq:poisson-observation-cardinal-projection}.  Since
$\Pi_{\ZZ}F_t$ has exactly those samples, uniqueness identifies the
cardinal sum with $\Pi_{\ZZ}F_t$.  The convergence assertion and the norm
identity \eqref{eq:poisson-observation-cardinal-norm} are precisely the
convergence and norm statements of that corollary applied to
$\alpha=\alpha(t)$.

The residual belongs to $\mathcal H_K(\ZZ)^\perp$ by definition of
$\Pi_{\ZZ}$, and the displayed computation above already shows that it
vanishes at every integer.  If $F_t\in\mathcal H_K(\ZZ)$, then
$\Pi_{\ZZ}F_t=F_t$, so the cardinal reconstruction is the full channel.
Conversely, if the integer-orbit cardinal reconstruction is the full
channel, then $F_t=\Pi_{\ZZ}F_t\in\mathcal H_K(\ZZ)$.
\end{proof}

\begin{remark}
Proposition~\ref{prop:poisson-lattice-sampling}
and Corollary~\ref{cor:poisson-cardinal-reconstruction} fit the standard
sampling paradigm for principal shift-invariant spaces.
The sampling inequalities come from the bounded and invertible symbol
$\sigma(\theta)=\pi\cosh(2(\pi-|\theta|))/\sinh(2\pi)$ and the
frame characterization by Riesz bounds in
\cite{deBoorDeVoreRon1994,RonShen1995,AldroubiGrochenig2001,NashedWalter1991,NashedSun2010}.
The Galerkin sampling viewpoint in general reproducing-kernel spaces is
developed in \cite{ChengJiangSun2016}.

For this same paradigm, explicit nod-function formulae are known for many
generators, including Gaussian families
\cite{RomeroUlanovskiiZlotnikov2024,BaranovBelovGrochenig2022}
and recent derivative-sampling extensions
\cite{GrochenigRomeroStockler2020,RomeroUlanovskiiZlotnikov2024}.
The classical Cauchy/Lorentz integer-shift literature also contains Riesz
and interpolation statements of this type
\cite{KiselevMininNovikov2016,KiselevMininNovikov2019}.
The Poisson-strip representation used here makes the symbol and the
interpolation norm completely explicit, yielding the closed formula for
$\sigma$ and the cardinal norm identity
\eqref{eq:cardinal-norm-formula}.
\end{remark}
