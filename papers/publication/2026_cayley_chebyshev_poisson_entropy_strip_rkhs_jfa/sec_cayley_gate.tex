\subsection{The Cayley chart}\label{sec:cayley-gate}

We work with the Cayley coordinates
\begin{equation}\label{eq:phase-coord}
\upsilon(x):=\frac{1}{\pi}\arctan x\in \UU,
\end{equation}
\begin{equation}\label{eq:angular-var}
\vartheta(x):=2\pi\upsilon(x)=2\arctan x\in(-\pi,\pi),
\end{equation}
and
\begin{equation}\label{eq:cayley-embed}
\iota(x):=e^{\ii\vartheta(x)}=\frac{1+\ii x}{1-\ii x}\in\TT,
\qquad x\in\RR.
\end{equation}
Thus $\upsilon$ identifies $\RR$ with the interval $\UU$, while $\iota$
identifies $\RR$ with $\TT\setminus\{-1\}$.  We shall use these
coordinates repeatedly.

\begin{proposition}\label{prop:cayley-homeo}
The map $\iota$ extends to a homeomorphism
\[
\iota:\overline\RR\xrightarrow{\sim}\TT
\]
by $\iota(\infty)=-1$.  Its inverse is given by
\begin{equation}\label{eq:cayley-inverse}
\iota^{-1}(e^{\ii\theta})=\tan\frac{\theta}{2},
\qquad
\theta\in(-\pi,\pi),
\qquad
\iota^{-1}(-1)=\infty.
\end{equation}
Moreover, $\upsilon:\RR\to\UU$ and
$\iota:\RR\to\TT\setminus\{-1\}$ are $C^\infty$ diffeomorphisms.
\end{proposition}

\begin{proof}
The identity \eqref{eq:cayley-embed} shows that $\iota$ is a smooth
bijection from $\RR$ onto $\TT\setminus\{-1\}$ with inverse
$x=\tan(\theta/2)$.  Since $\iota(x)\to -1$ as $x\to\pm\infty$, the map
extends continuously to~$\overline\RR$ by $\iota(\infty)=-1$.
Compactness of~$\overline\RR$ and the Hausdorff property of~$\TT$
imply that the extension is a homeomorphism.  The formulas
$x=\tan(\pi u)$ and \eqref{eq:cayley-inverse} give the smooth inverses
of $\upsilon$ and~$\iota$ on the open parts.
\end{proof}
