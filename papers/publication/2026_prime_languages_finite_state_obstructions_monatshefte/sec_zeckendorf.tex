\section{Context: Zeckendorf Growth}
\label{sec:zeckendorf}

This section specialises the density dichotomy to Zeckendorf words. We
record only the minimal classical facts needed later---in particular the
Zeckendorf growth law---and then combine them with prime-counting
estimates to obtain the quantitative obstructions promised in
\Cref{sec:automata}.

Let
\[
  \mathcal Z
  :=\{x_1\cdots x_m\in\{0,1\}^\ast:
      x_ix_{i+1}\neq 11,\ x_m=1\}
\]
be the language of canonical Zeckendorf words. We write these words
least-significant digit first: for \(w=x_1\cdots x_m\in\mathcal Z\),
\[
  \val(w):=\sum_{k=1}^m x_kF_{k+1},
\]
where \(F_0=0\), \(F_1=1\), \(F_{n+2}=F_{n+1}+F_n\), and
\(\varphi=(1+\sqrt5)/2\). Thus \(x_m=1\) marks the most significant
nonzero digit. The map \(w\mapsto\val(w)\) is a bijection from
\(\mathcal Z\) onto \(\NN_{\ge 1}\) \cite{Zeckendorf1972}.
Consequently \(\mathcal Z\cap\{0,1\}^{\le m}\) parametrises the
interval \([1,F_{m+2})\), so words of length exactly \(m\) parametrise
\([F_{m+1},F_{m+2})\). Equivalently,
\(\abs{\mathcal Z_m}=F_m\) for every \(m\ge 1\).
We write
\[
  \mathcal Z_m:=\{w\in\mathcal Z:\abs{w}=m\}
  \qquad\text{and}\qquad
  \mathcal P_m^{(\mathrm Z)}
  :=\{w\in\mathcal Z_m:\val(w)\text{ is prime}\},
\]
and set
\[
  \mathcal P_{\mathcal Z}
  :=\bigcup_{m\ge 1}\mathcal P_m^{(\mathrm Z)}.
\]

\begin{proposition}[Known growth law for regular Zeckendorf languages]
  \label{prop:zeckendorf-regular-powerlaw}
  Let \(L\subseteq\mathcal Z\) be regular. Choose a DFA recognizing
  \(L\), delete states that are not reachable from the initial state or
  cannot reach an accepting state, and let \(\rho\) be the maximal
  spectral radius of a strongly connected component of the resulting
  automaton. Define
  \[
    N_L(T):=\#\{w\in L:\val(w)\le T\}.
  \]
  Then
  \[
    N_L(T)=T^{\alpha+o(1)},
    \qquad
    \alpha=
    \begin{cases}
      0, & \rho\le 1,\\[0.3em]
      \dfrac{\log \rho}{\log \varphi}, & \rho>1.
    \end{cases}
    \qquad (T\to\infty).
  \]
\end{proposition}

\begin{proof}
  This is the Zeckendorf specialization of the general growth theorem
  for \(S\)-recognizable sets in \cite[Theorem~7]{CharlierRampersad2011};
  see also \cite{LecomteRigo2001}. In the present numeration system the
  ambient counting function satisfies
  \[
    \#(\mathcal Z\cap\{0,1\}^{\le m})
    =F_{m+2}-1
    =\varphi^{m+o(m)},
  \]
  The trimmed automaton for \(L\) contributes the exponential factor
  \(\max(1,\rho)\). Substituting these two growth parameters into the
  theorem of Charlier and Rampersad yields exactly the displayed
  asymptotic
  \[
    N_L(T)=T^{\alpha+o(1)}
  \]
  with \(\alpha=0\) when \(\rho\le 1\) and
  \(\alpha=\log \rho/\log \varphi\) when \(\rho>1\).
\end{proof}

\begin{remark}\label{rem:zeckendorf-examples}
  The endpoint cases and the borderline case \(\rho=1\) already occur
  for simple regular sublanguages.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item For \(L=\mathcal Z\), one has \(N_L(T)=\lfloor T\rfloor\),
    \(\rho=\varphi\), and \(\alpha=1\).
    \item If \(L\) is finite, then \(N_L(T)\) is eventually constant,
    so \(\alpha=0\).
    \item For \(L=0^\ast1\subseteq\mathcal Z\), one has \(a_m=1\) for
    every \(m\), hence \(\rho=1\) and
    \(N_L(T)\asymp \log T=T^{o(1)}\).
  \end{enumerate}
\end{remark}

\begin{remark}
  The same transfer-matrix argument applies more generally to Parry
  numeration systems whose admissible digit language is regular. In that
  setting the regular syntax of admissible expansions replaces
  \(\mathcal Z\), and the dominant exponential growth constant of the
  numeration system replaces \(\varphi\); see
  \cite{Frougny1992,BruyereHansel1997,LecomteRigo2001}.
\end{remark}

\begin{lemma}[Prime counts in Zeckendorf slices]
  \label{lem:zeck-prime-count}
  There exist constants \(A_{\mathrm Z},B_{\mathrm Z}>0\) such that for
  all sufficiently large \(m\),
  \[
    A_{\mathrm Z}\,\frac{\varphi^m}{m}
    \le \abs{\mathcal P_m^{(\mathrm Z)}}
    \le B_{\mathrm Z}\,\frac{\varphi^m}{m}.
  \]
\end{lemma}

\begin{proof}
  Every word in \(\mathcal Z_m\) represents an integer in
  \([F_{m+1},F_{m+2})\), so
  \(\abs{\mathcal P_m^{(\mathrm Z)}}=\pi(F_{m+2})-\pi(F_{m+1})\), where
  \(\pi\) is the prime-counting function. The prime number theorem with
  a first-order error term
  \cite[Chapter~22]{HardyWright2008} gives
  \(\pi(x)=x/\log x+O(x/\log^2 x)\). Hence
  \[
    \pi(F_{m+2})-\pi(F_{m+1})
    =\frac{F_{m+2}}{\log F_{m+2}}-\frac{F_{m+1}}{\log F_{m+1}}
     +O\!\left(\frac{F_{m+2}}{\log^2 F_{m+2}}\right).
  \]
  Since \(F_{m+1}/F_{m+2}\to\varphi^{-1}<1\) and
  \(\log F_{m+1}=m\log\varphi+O(1)=\log F_{m+2}+O(1)\), the quantity in
  parentheses
  \[
    1-\frac{F_{m+1}}{F_{m+2}}
      \frac{\log F_{m+2}}{\log F_{m+1}}
  \]
  stays bounded between two positive constants for all large \(m\).
  Moreover \(F_{m+2}=\Theta(\varphi^m)\) and
  \(\log F_{m+2}=m\log\varphi+O(1)\), while the error term
  \(O(F_{m+2}/\log^2 F_{m+2})\) is \(o(\varphi^m/m)\). The displayed
  expression is therefore comparable to \(\varphi^m/m\), yielding
  constants \(A_{\mathrm Z},B_{\mathrm Z}>0\) with the stated property.
\end{proof}

For a deterministic automaton \(\mathcal A\) over the Zeckendorf digit
alphabet, write
\[
  a_m^{(\mathrm Z)}(\mathcal A)
  :=\abs{L(\mathcal A)\cap\mathcal Z_m}.
\]

\begin{corollary}[Zeckendorf symmetric-difference obstruction]
  \label{cor:zeck-symmetric-diff}
  Let \(\mathcal A\) be a deterministic automaton over the Zeckendorf
  alphabet. Then there exists \(c_{\mathcal A,\varphi}>0\) such that
  \[
    \abs{(L(\mathcal A)\cap\mathcal Z_m)
         \triangle\mathcal P_m^{(\mathrm Z)}}
    \ge c_{\mathcal A,\varphi}\,\frac{\varphi^m}{m}
  \]
  for all sufficiently large \(m\). If
  \(a_m^{(\mathrm Z)}(\mathcal A)\) lies in the positive-density
  alternative of \Cref{thm:parry-density}, then along the corresponding
  arithmetic progression of lengths one has the stronger bound
  \(\abs{(L(\mathcal A)\cap\mathcal Z_m)\triangle\mathcal P_m^{(\mathrm Z)}}
  \gg \varphi^m\).
\end{corollary}

\begin{proof}
  Apply \Cref{thm:parry-density} with \(\mathcal N=\mathcal Z\). Thus
  \[
    \frac{a_m^{(\mathrm Z)}(\mathcal A)}{F_m}
    =c_{m\bmod p}+O(\theta^m)
  \]
  for some integers \(p\ge 1\), \(\theta\in(0,1)\), and constants
  \(c_0,\dots,c_{p-1}\ge 0\). Treat each residue class separately.

  If \(c_r=0\), then along \(m=pk+r\) we have
  \(a_m^{(\mathrm Z)}(\mathcal A)=O(\theta^k\varphi^{pk+r})\). By
  \Cref{lem:zeck-prime-count},
  \(\abs{\mathcal P_m^{(\mathrm Z)}}\asymp \varphi^{pk+r}/(pk+r)\). For
  \(k\) large enough the difference is therefore bounded below by
  \(\tfrac{1}{2}A_{\mathrm Z}\varphi^{pk+r}/(pk+r)\).

  If \(c_r>0\), then along \(m=pk+r\) we have
  \(a_m^{(\mathrm Z)}(\mathcal A)\ge \tfrac{1}{2}c_r F_{pk+r}\). Since
  \(F_m\asymp \varphi^m\) and \(\abs{\mathcal P_m^{(\mathrm Z)}}\ll
  \varphi^m/m\), the symmetric difference is \(\gg \varphi^{pk+r}\) on
  that progression.

  Taking the minimum of the finitely many lower bounds obtained over all
  residue classes produces \(c_{\mathcal A,\varphi}>0\) with the stated
  properties.
\end{proof}

Define recall and precision for Zeckendorf slices by
\[
  \mathrm{Rec}_m^{(\mathrm Z)}
  :=\frac{\abs{L(\mathcal A)\cap\mathcal P_m^{(\mathrm Z)}}}
          {\abs{\mathcal P_m^{(\mathrm Z)}}},
  \qquad
  \mathrm{Prec}_m^{(\mathrm Z)}
  :=\frac{\abs{L(\mathcal A)\cap\mathcal P_m^{(\mathrm Z)}}}
          {a_m^{(\mathrm Z)}(\mathcal A)}
\]
whenever both \(\abs{\mathcal P_m^{(\mathrm Z)}}>0\) and
\(a_m^{(\mathrm Z)}(\mathcal A)>0\).

\begin{corollary}[No simultaneous recall and precision in Zeckendorf]
  \label{cor:zeck-recall-precision}
  Every deterministic automaton \(\mathcal A\) over the Zeckendorf
  alphabet satisfies one of the following:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item \(\mathrm{Rec}_m^{(\mathrm Z)}\) decays exponentially along
    every residue class of lengths.
    \item Along some arithmetic progression one has
    \(\mathrm{Prec}_m^{(\mathrm Z)}\le C_{\mathcal A,\varphi}/m\) for a
    constant depending only on \(\mathcal A\).
  \end{enumerate}
  Consequently no fixed DFA has both recall and precision bounded below
  on infinitely many Zeckendorf lengths.
\end{corollary}

\begin{proof}
  Apply the density dichotomy \Cref{thm:parry-density} with
  \(\mathcal N=\mathcal Z\) and argue as in the proof of
  \Cref{cor:dfa-prime-recall-precision}, using
  \Cref{lem:zeck-prime-count}. When \(c_r=0\) the numerator of
  \(\mathrm{Rec}_m^{(\mathrm Z)}\) is exponentially small relative to
  \(\abs{\mathcal P_m^{(\mathrm Z)}}\asymp \varphi^m/m\), so recall
  decays exponentially. When \(c_r>0\),
  \(a_m^{(\mathrm Z)}(\mathcal A)\gg \varphi^m\) along that progression,
  whereas the numerator is \(O(\varphi^m/m)\), yielding
  \(\mathrm{Prec}_m^{(\mathrm Z)}\le C_{\mathcal A,\varphi}/m\). The
  final statement follows as in the base-\(b\) case.
\end{proof}

\begin{corollary}[Prime Zeckendorf language is not regular]
  \label{cor:zeckendorf-prime-language-not-regular}
  Let
  \[
    \mathcal P_{\mathcal Z}
    :=\{\mathrm{Zeck}(p)\in\mathcal Z:p\text{ prime}\},
  \]
  where \(\mathrm{Zeck}(n)\) denotes the canonical Zeckendorf
  representation of \(n\). Then \(\mathcal P_{\mathcal Z}\) is not a
  regular language.
\end{corollary}

\begin{proof}
  If \(\mathcal P_{\mathcal Z}\) were regular, then
  \Cref{prop:zeckendorf-regular-powerlaw} would give
  \[
    \pi(T)=N_{\mathcal P_{\mathcal Z}}(T)=T^{\alpha+o(1)}
  \]
  for some \(\alpha\in[0,1]\). This is impossible because
  \(\pi(T)\sim T/\log T\): for every \(\alpha<1\) one has
  \(\pi(T)/T^\alpha\to\infty\), while \(\pi(T)/T\to 0\).
\end{proof}

\begin{remark}[Easy sofic corollary]
  \label{rem:zeckendorf-primes-not-sofic}
  The preceding non-regularity statement immediately excludes eventual
  sofic realization of the prime Zeckendorf length slices. We record
  only the short argument.
  Let \(\mathcal B_m(X)\) denote the admissible words of length
  \(m\) in a sofic shift \(X\). If
  \(\mathcal B_m(X)=\mathcal P_m^{(\mathrm Z)}\) for all \(m\ge N\),
  then
  \[
    \mathcal B(X)\triangle\mathcal P_{\mathcal Z}\subseteq \{0,1\}^{<N},
  \]
  a finite set. Since \(\mathcal B(X)\) is regular
  (\cite[Theorem~3.2.1]{LindMarcus1995}; see also
  \cite[Chapter~4]{PerrinPin2004}), this would make
  \(\mathcal P_{\mathcal Z}\) a finite symmetric difference away from a
  regular language, hence regular itself, contradicting
  \Cref{cor:zeckendorf-prime-language-not-regular}.
\end{remark}
