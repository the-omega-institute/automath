\section{Preliminaries and the Kernel}\label{sec:preliminaries}

\subsection{Primitive orbit counts}

Let \(M\) be a primitive non-negative matrix with Perron root
\(\lambda=\rho(M)\). The associated dynamical \(\zeta\)-function is
\[
  \zeta_M(z)
  =\exp\!\left(\sum_{n\ge 1}\frac{\Tr(M^n)}{n}z^n\right)
  =\det(I-zM)^{-1}.
\]
The number of primitive orbits of least period \(n\) is
\begin{equation}\label{eq:primitive-mobius}
  p_n(M)=\frac{1}{n}\sum_{d\mid n}\mu(d)\Tr(M^{n/d}).
\end{equation}

\begin{lemma}\label{lem:primitive-orbit-asymptotic}
If
\[
  \Lambda(M):=\max\{|\mu|:\mu\in\spec(M),\ \mu\neq\lambda\},
\]
then
\[
  p_n(M)=\frac{\lambda^n}{n}
  +O\!\left(\frac{\max\{\Lambda(M)^n,\lambda^{n/2}\}}{n}\right).
\]
\end{lemma}

\begin{proof}
By Perron--Frobenius theory,
\(
  \Tr(M^n)=\lambda^n+O(\Lambda(M)^n)
\).
Substituting this into \eqref{eq:primitive-mobius}, the divisor \(d=1\)
gives the main term and every divisor \(d\ge 2\) contributes at most
\[
  O\!\left(\frac{\lambda^{n/2}}{n}\right).
\]
\end{proof}

\subsection{Self-dual kernels and completion}

Let \(B(u)=B_0+uB_1\) with integer matrices \(B_0,B_1\), and suppose there is a permutation matrix \(\Pi\) such that \(\Pi^{-1}B_0\Pi=B_1\) (and hence \(\Pi^{-1}B_1\Pi=B_0\)). Write
\[\Delta(z,u):=\det(I-zB(u)).\]
The following observation packages the formal consequences of this self-duality.

\begin{lemma}[Self-dual compression]\label{lem:self-dual-compression}
With \(B(u)\) as above, one has \(\Pi^{-1}B(u)\Pi=u\,B(1/u)\), hence
\(\Delta(z,u)=\Delta(uz,u^{-1})\). After the change of variables
\(u=r^2\), \(w=zr\), and \(s=r+r^{-1}\), the expression
\(\widehat\Delta(w,s):=\Delta(w/r,r^2)\) belongs to \(\ZZ[s][w]\). Writing
\[
  \widehat\Delta(w,s)=\sum_{j=0}^d c_j(s)w^j,
\]
each \(c_j(s)\) satisfies \(\deg_s c_j\le j\) and
\(c_j(-s)=(-1)^j c_j(s)\). In particular,
\(\widehat\Delta(w,-s)=\widehat\Delta(-w,s)\).
\end{lemma}

\begin{proof}
The matrix identity follows from the hypotheses:
\[
  \Pi^{-1}B(u)\Pi
  =\Pi^{-1}(B_0+uB_1)\Pi
  =B_1+uB_0
  =u\,B(1/u).
\]
Taking determinants of both sides gives the functional equation for
\(\Delta(z,u)\).

Write
\[
  \Delta(z,u)=\sum_{j=0}^d a_j(u)z^j
  \qquad\text{with }a_j(u)\in\ZZ[u].
\]
Comparing coefficients in
\(\Delta(z,u)=\Delta(uz,u^{-1})\) gives
\[
  a_j(u)=u^j a_j(u^{-1})
  \qquad (0\le j\le d).
\]
Therefore \(r^{-j}a_j(r^2)\) is a Laurent polynomial in \(r\) that is
invariant under \(r\mapsto r^{-1}\), whose exponents lie in
\(\{-j,-j+2,\dots,j-2,j\}\). Hence there is a unique polynomial
\(c_j(s)\in\ZZ[s]\) with \(\deg_s c_j\le j\) such that
\[
  r^{-j}a_j(r^2)=c_j(r+r^{-1})=c_j(s).
\]
Substituting \(u=r^2\) and \(z=w/r\) yields
\[
  \widehat\Delta(w,s)=\Delta(w/r,r^2)=\sum_{j=0}^d c_j(s)w^j\in\ZZ[s][w].
\]
Replacing \(r\) by \(-r\) sends \((w,s)\) to \((-w,-s)\), so
\[
  \widehat\Delta(w,-s)=\widehat\Delta(-w,s).
\]
Comparing coefficients of \(w^j\) gives
\(c_j(-s)=(-1)^j c_j(s)\).
\end{proof}
For reference we keep the terminology of Section~\ref{sec:introduction}: the polynomial $\widehat\Delta(w,s)$ is the completed determinant, and $\Phi_m^+$ denotes the real cyclotomic polynomial of $2\cos(\pi/m)$.

\begin{lemma}[Parity normal form for the sign quotient]
  \label{lem:sign-quotient-normal-form}
  Let \(k\) be a field of characteristic different from \(2\), and let
  \[
    H(w,s)=\sum_{j=0}^d c_j(s)w^j\in k[w,s]
  \]
  satisfy \(\deg_s c_j\le j\) and \(c_j(-s)=(-1)^j c_j(s)\) for every
  \(j\). Then there is a unique polynomial \(G(x,y)\in k[x,y]\) such
  that
  \[
    H(w,s)=G(w^2,sw).
  \]
  If \(H\) is irreducible, then the quotient of the affine curve
  \(H(w,s)=0\) by the involution \((w,s)\mapsto(-w,-s)\) is birational
  to the affine plane curve \(G(x,y)=0\).
\end{lemma}

\begin{proof}
For each \(j\), the parity condition implies that \(c_j(s)\) is a
linear combination of monomials \(s^b\) with \(0\le b\le j\) and
\(b\equiv j\pmod 2\). Hence every monomial \(s^bw^j\) occurring in
\(H\) can be rewritten as
\[
  s^bw^j=(w^2)^{(j-b)/2}(sw)^b=x^{(j-b)/2}y^b,
\]
where \(x=w^2\) and \(y=sw\). This gives a polynomial
\(G(x,y)\in k[x,y]\) with \(H(w,s)=G(w^2,sw)\). Uniqueness follows
because the monomials \(x^ay^b\) are linearly independent in
\(k[w,s]\).

If \(H\) is irreducible, then \(k[w,s]/(H)\) is a domain. Let
\[
  K:=\operatorname{Frac}\bigl(k[w,s]/(H)\bigr)
\]
be its function field. The involution \((w,s)\mapsto(-w,-s)\) acts on
\(K\), and its invariant subfield is generated by \(x=w^2\) and
\(y=sw\). Hence the quotient is birational to the plane curve
\(G(x,y)=0\).
\end{proof}


\subsection{The ten-state weighted kernel}

\begin{definition}\label{def:sync-kernel}
We work throughout with the explicit ten-state weighted digraph shown in Figure~\ref{fig:sync-kernel}. Its state labels \(\st{000},\st{001},\dots,\st{11-1}\) are inherited from the two-track comparison automaton that originally motivated the example, but in this note they serve purely as names for the vertices of the digraph. Every edge and weight is displayed in the transition table below, so no external construction is required; we keep the synchronisation language only as a mnemonic for the involution that exchanges output-zero and output-one edges.
Let
\[
  \mathcal S=
  \{\st{000},\st{001},\st{002},\st{010},\st{100},\st{101},
    \st{0-12},\st{1-12},\st{01-1},\st{11-1}\}
\]
be the state set. The weighted synchronisation kernel is the
\(10\times 10\) matrix \(B(u)=B_0+uB_1\) obtained from the following
directed transitions:
\begin{align*}
  \st{000}&\to \st{000},\st{001},\st{002}
    &&\text{with output }0,\\
  \st{100}&\to \st{000},\st{001},\st{002}
    &&\text{with output }1,\\
  \st{001}&\to \st{010},\st{100},\st{101}
    &&\text{with output }0,\\
  \st{002}&\to \st{11-1},\st{000},\st{001}
    &&\text{with outputs }0,1,1,\\
  \st{010}&\to \st{100},\st{101},\st{0-12}
    &&\text{with outputs }0,0,1,\\
  \st{101}&\to \st{010},\st{100},\st{101}
    &&\text{with output }1,\\
  \st{0-12}&\to \st{01-1},\st{010},\st{100}
    &&\text{with output }0,\\
  \st{1-12}&\to \st{01-1},\st{010},\st{100}
    &&\text{with output }1,\\
  \st{01-1}&\to \st{001},\st{002},\st{1-12}
    &&\text{with output }0,\\
  \st{11-1}&\to \st{001},\st{002},\st{1-12}
    &&\text{with output }1.
\end{align*}
Each output-\(0\) edge contributes \(1\) to the corresponding matrix
entry, and each output-\(1\) edge contributes \(u\).

\medskip
\noindent\textit{Digraph structure.}\quad
The ten states partition into five dual pairs under the involution
\(\iota\) of \Cref{prop:sync-self-duality}:
\[
  \underbrace{\st{000}\leftrightarrow\st{101}}_{\text{pair 1}},\quad
  \underbrace{\st{001}\leftrightarrow\st{100}}_{\text{pair 2}},\quad
  \underbrace{\st{002}\leftrightarrow\st{010}}_{\text{pair 3}},\quad
  \underbrace{\st{0-12}\leftrightarrow\st{11-1}}_{\text{pair 4}},\quad
  \underbrace{\st{1-12}\leftrightarrow\st{01-1}}_{\text{pair 5}}.
\]
Each state has exactly three outgoing edges. The self-loop at
\(\st{000}\) (and dually at \(\st{101}\)) makes the digraph aperiodic.
The strong connectivity can be checked from the transitions: every state
reaches \(\st{000}\) in at most three steps via
\[
  \st{001}\to\st{010}\to\st{100}\to\st{000},\qquad
  \st{002}\to\st{000},\qquad
  \st{010}\to\st{100}\to\st{000},\qquad
  \st{101}\to\st{100}\to\st{000},
\]
\[
  \st{0-12}\to\st{100}\to\st{000},\qquad
  \st{1-12}\to\st{100}\to\st{000},\qquad
  \st{01-1}\to\st{002}\to\st{000},
\]
\[
  \st{11-1}\to\st{002}\to\st{000},
\]
and \(\st{000}\) reaches every state by following
\(\st{000}\to\st{001}\to\st{010}\to\st{0-12}\to\st{01-1}\to\st{1-12}\)
and the dual chain. The transition table below summarises the adjacency
structure, where each row lists the three target states and the
corresponding output label:
\[
  \begin{array}{c|ccc|c}
    \text{source} & \text{target 1} & \text{target 2} & \text{target 3}
      & \text{outputs} \\
    \hline
    \st{000} & \st{000} & \st{001} & \st{002} & 0,0,0 \\
    \st{001} & \st{010} & \st{100} & \st{101} & 0,0,0 \\
    \st{002} & \st{11-1} & \st{000} & \st{001} & 0,1,1 \\
    \st{010} & \st{100} & \st{101} & \st{0-12} & 0,0,1 \\
    \st{100} & \st{000} & \st{001} & \st{002} & 1,1,1 \\
    \st{101} & \st{010} & \st{100} & \st{101} & 1,1,1 \\
    \st{0-12} & \st{01-1} & \st{010} & \st{100} & 0,0,0 \\
    \st{1-12} & \st{01-1} & \st{010} & \st{100} & 1,1,1 \\
    \st{01-1} & \st{001} & \st{002} & \st{1-12} & 0,0,0 \\
    \st{11-1} & \st{001} & \st{002} & \st{1-12} & 1,1,1 \\
  \end{array}
\]
\end{definition}

\begin{proposition}[Self-duality]\label{prop:sync-self-duality}
Let \(\iota\) be the involution of \(\mathcal S\) defined by
\[
  \st{000}\leftrightarrow\st{101},\qquad
  \st{001}\leftrightarrow\st{100},\qquad
  \st{002}\leftrightarrow\st{010},
\]
\[
  \st{0-12}\leftrightarrow\st{11-1},\qquad
  \st{1-12}\leftrightarrow\st{01-1},
\]
and let \(\Pi\) be the associated permutation matrix. Then
\[
  \Pi^{-1}B_0\Pi=B_1,
  \qquad
  \Pi^{-1}B_1\Pi=B_0.
\]
Consequently,
\[
  \Pi^{-1}B(u)\Pi=u\,B(1/u),
  \qquad
  \Delta(z,u):=\det(I-zB(u))=\Delta(uz,u^{-1}).
\]
\end{proposition}

\begin{proof}
Inspection of the transition list shows that, whenever
\(x\to y\) is an edge with output \(e\), the transition
\(\iota(x)\to\iota(y)\) is an edge with output \(1-e\). This gives the
matrix identities for \(B_0\) and \(B_1\). The identity for \(B(u)\)
follows immediately:
\[
  \Pi^{-1}B(u)\Pi
  =\Pi^{-1}(B_0+uB_1)\Pi
  =B_1+uB_0
  =u\,B(1/u).
\]
Taking determinants of \(I-zB(u)\) and \(I-zu\,B(1/u)\) yields the
functional equation for \(\Delta(z,u)\).
\end{proof}

\begin{definition}\label{def:perron-residue-constant}
We write \(B:=B(1)\) for the unweighted kernel and
\[
  C_B:=\lim_{z\to 1/\rho(B)}(1-\rho(B)z)\det(I-zB)^{-1}
\]
for its Perron residue constant.
\end{definition}
