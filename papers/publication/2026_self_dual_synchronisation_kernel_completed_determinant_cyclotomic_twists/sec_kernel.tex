\section{The Completed Determinant and Cyclotomic Twists}
\label{sec:completion-twists}

\subsection*{Notation for smooth projective models}

Throughout this section, for an affine curve \(C_{\mathrm{aff}}\subset
\mathbb{A}^2\) defined by an irreducible polynomial equation we write
\(\overline{X}\) (resp.\ \(\overline{Y}\)) for the smooth projective curve
whose function field is \(k(C_{\mathrm{aff}})\) (resp.\ its degree-two subfield
under the relevant involution). All genus and Hurwitz statements are made for
these complete nonsingular models.

\subsection{The unweighted spectrum}

\begin{proposition}[Explicit determinant]\label{prop:kernel-delta-explicit}
One has
\[
  \Delta(z,u)
  =1-(1+u)z-5uz^2+3u(1+u)z^3-u(u^2-3u+1)z^4
\]
\[
  \qquad
  +u(u^3-3u^2-3u+1)z^5+u^2(u^2+u+1)z^6.
\]
\end{proposition}

\begin{proof}
A symbolic determinant computation from the explicit \(10\times 10\)
matrix \(B(u)\) of \Cref{def:sync-kernel} gives the displayed formula.
As a consistency check, the closed form satisfies the self-duality
identity of \Cref{prop:sync-self-duality}; in particular, comparing
\(\Delta(z,u)\) with \(\Delta(uz,u^{-1})\) pairs the coefficients of
\(z^k u^\ell\) and \(z^k u^{k-\ell}\) exactly as required.
\end{proof}

\begin{proposition}[Characteristic polynomial and residue constant]
  \label{prop:sync-kernel-spectrum}
  The matrix \(B=B(1)\) is primitive and satisfies
  \[
    \det(I-zB)=(z-1)(z+1)(3z-1)(z^3-z^2+z+1).
  \]
  In particular \(\rho(B)=3\), and the Perron residue constant
  \(C_B\) of \Cref{def:perron-residue-constant} satisfies
  \[
    C_B=\frac{243}{272}.
  \]
\end{proposition}

\begin{proof}
Every state reaches \(\st{000}\) in at most two steps, and
\(\st{000}\) reaches every state by iterating the transitions through
\(\st{001}\), \(\st{002}\), \(\st{010}\), and \(\st{0-12}\); hence the
digraph of \(B\) is strongly connected. Since there is also the self-loop
\(\st{000}\to\st{000}\), the period is \(1\), so \(B\) is primitive.
Set \(u=1\) in \Cref{prop:kernel-delta-explicit}. The factorisation is
then immediate. Since \((3z-1)\) is the unique factor whose reciprocal
lies outside the unit circle, \(\rho(B)=3\). Evaluating the cofactor at
\(z=1/3\) gives the residue constant. This proves part~\textup{(i)} of
\Cref{thm:intro-kernel-geometry}.
\end{proof}

Together with \Cref{lem:primitive-orbit-asymptotic}, this yields
\[
  p_n(B)=\frac{3^n}{n}
  +O\!\left(\frac{\max\{\Lambda(B)^n,3^{n/2}\}}{n}\right).
\]

\subsection{Completion and algebraic geometry}

Introducing
\[
  u=r^2,\qquad s=r+r^{-1},\qquad w=zr,
\]
the self-duality of \(\Delta(z,u)\) implies that
\(\Delta(w/r,r^2)\) is a polynomial in \(w\) and \(s\), which we denote
by \(\widehat\Delta(w,s)\).

\begin{proposition}[Explicit completed determinant]
  \label{prop:sync-hatdelta-completion}
  The completed determinant is
  \[
    \widehat\Delta(w,s)
    =1-sw-5w^2+3sw^3+(5-s^2)w^4+(s^3-6s)w^5+(s^2-1)w^6.
  \]
  It satisfies
  \[
    \widehat\Delta(w,-s)=\widehat\Delta(-w,s).
  \]
\end{proposition}

\begin{proof}
Substitute \(u=r^2\) and \(z=w/r\) into the explicit formula of
\Cref{prop:kernel-delta-explicit}. Terms in \(r\) and \(r^{-1}\)
combine to give a polynomial in \(s=r+r^{-1}\), and a direct expansion
produces the displayed formula. The parity relation is immediate. This
proves part~\textup{(ii)} of \Cref{thm:intro-kernel-geometry}.
\end{proof}

Let \(C_{\mathrm{aff}}:=\{(w,s)\in\mathbb{A}^2:\widehat\Delta(w,s)=0\}\) be
the affine curve, and let \(\overline{X}\) be the smooth projective curve with
function field \(k(C_{\mathrm{aff}})\). The involution
\[
  \iota:(w,s)\longmapsto(-w,-s)
\]
preserves \(C_{\mathrm{aff}}\), and induces an involution on \(\overline{X}\).
Write \(\overline{Y}\) for the smooth projective curve with function field
\(k(\overline{X})^{\langle\iota\rangle}\).

\begin{proposition}[The quotient is a smooth plane quartic]
  \label{prop:sync-hatdelta-quotient}
  The invariants
  \[
    x:=w^2,\qquad y:=sw
  \]
  identify \(\overline{Y}\) birationally with the smooth projective closure of
  the affine curve
  \[
    F(x,y):=
    1-y-5x+3xy+5x^2-xy^2+xy^3-6x^2y+x^2y^2-x^3=0.
  \]
  Consequently,
  \[
    g(\overline{Y})=3.
  \]
\end{proposition}

\begin{proof}
Since \(\iota\) changes the signs of both \(w\) and \(s\), the
quantities \(x=w^2\) and \(y=sw\) are invariant. Substituting
\(s=y/w\) and \(w^2=x\) into \(\widehat\Delta(w,s)=0\) yields
\(F(x,y)=0\). Conversely, adjoining \(w=\sqrt{x}\) recovers the
function field of \(\overline{X}\) from that of the curve \(F=0\),
showing that \(\overline{Y}\) is birationally the smooth projective closure
of \(\{F=0\}\).

We verify that this projective closure is smooth.  Homogenise by setting
\(X=x\), \(Y=y\), \(Z=1\) and clearing denominators to obtain
\[
  F_h(X,Y,Z) = Z^4 - YZ^3 - 5XZ^2 + 3XYZ + 5X^2Z^2
    - XY^2Z + XY^3 - 6X^2YZ + X^2Y^2 - X^3Z.
\]
On the affine chart \(Z=1\),
\[
  \partial_x F
  =-5+3y+10x-y^2+y^3-12xy+2xy^2-3x^2,
\]
\[
  \partial_y F
  =-1+3x-2xy+3xy^2-6x^2+2x^2y.
\]
A Gr\"obner-basis computation of the ideal
\(\langle F,\,\partial_x F,\,\partial_y F\rangle\subset\QQ[x,y]\)
gives \(\langle 1\rangle\), so there is no affine singular point.

The line at infinity \(Z=0\) meets the quartic where the leading form
\(XY^2(X+Y)=0\), giving the three points
\[
  [0:1:0],\qquad [1:0:0],\qquad [1:-1:0].
\]
At \([0:1:0]\) (chart \(Y=1\)): one computes
\[
  \partial_X F_h\big|_{(0,1,0)}
  = \bigl[-5Z^2+3YZ-Y^2Z+Y^3-12XYZ+2XY^2-3X^2Z\bigr]_{(0,1,0)}
  = 1\neq 0.
\]
At \([1:0:0]\) (chart \(X=1\)): \(\partial_Z F_h\big|_{(1,0,0)}=-1\neq 0\).
At \([1:-1:0]\) (chart \(X=1\)): \(\partial_Z F_h\big|_{(1,-1,0)}=1\neq 0\).

Hence the projective curve is smooth everywhere, so by the
genus formula for smooth plane curves \cite{Hartshorne1977},
\(g(\overline{Y})=(4-1)(4-2)/2=3\).
\end{proof}

\begin{proposition}[Two-sheeted cover of genus six]
  \label{prop:sync-hatdelta-double-cover}
  The covering map \(\overline{X}\to\overline{Y}\) is a degree-two map
  branched at exactly two points of \(\overline{Y}\). Hence
  \[
    g(\overline{X})=6.
  \]
\end{proposition}

\begin{proof}
The extension of function fields \(k(\overline{X})/k(\overline{Y})\) is
obtained by adjoining \(w=\sqrt{x}\), so it is quadratic of degree two.

We use the standard criterion for ramification in quadratic extensions
of function fields: let \(K\) be a function field of characteristic
\(\neq 2\), and let \(L=K(\sqrt{f}\,)\) with \(f\in K^\times\) not a
square. Then a place \(P\) of \(K\) ramifies in \(L/K\) if and only if
\(\operatorname{ord}_P(f)\) is odd; see \cite[III.7.3]{Stichtenoth2009}.
Here \(f=x\) and we must determine \(\operatorname{ord}_P(x)\) at each
place of \(\overline{Y}\).

On the affine chart, \(F(0,y)=1-y\), so the only place with \(x=0\) is
\(P=(0,1)\). Since \(\partial_y F(0,1)=-1\neq 0\), the line \(x=0\)
meets \(\overline{Y}\) transversely there, and therefore
\(\operatorname{ord}_P(x)=1\) (odd: ramified).

For the points at infinity, work on the chart \(X=1\) with
\(y=Y/X\) and \(z=Z/X\). The equation becomes
\[
  G(y,z)=z^4-yz^3-y^2z+y^3-3yz+y^2-z.
\]
At \(Q_0=[1:0:0]\), one has \(\partial_z G(0,0)=-1\), so \(z\) is an
analytic function of \(y\), and solving \(G(y,z)=0\) gives
\[
  z=y^2-2y^3+O(y^4).
\]
Hence \(x=X/Z=1/z\) has pole order \(2\) at \(Q_0\), so
\(\operatorname{ord}_{Q_0}(x)=-2\) (even: unramified). At
\(Q_\infty=[1:-1:0]\), writing \(y=-1+t\) gives
\[
  G(-1+t,z)=t-2t^2+t^3+z-tz-t^2z+z^3-tz^3+z^4,
\]
so \(\partial_z G(-1,0)=1\neq 0\), and
\[
  z=-t+t^2+O(t^3).
\]
Thus \(x=1/z\) has a simple pole at \(Q_\infty\), so
\(\operatorname{ord}_{Q_\infty}(x)=-1\) (odd: ramified).

At the remaining point \(Q_1=[0:1:0]\), use the chart \(Y=1\) with
\(u=X/Y\) and \(v=Z/Y\). The homogenised equation begins as
\[
  u-v^3+O(uv,v^4,u^2)=0,
\]
so \(u=v^3+O(v^4)\), and therefore \(x=X/Z=u/v\) has a zero of order
\(2\) at \(Q_1\), so \(\operatorname{ord}_{Q_1}(x)=2\) (even: unramified).

In summary:
\[
  \operatorname{ord}_P(x)=1,\quad
  \operatorname{ord}_{Q_0}(x)=-2,\quad
  \operatorname{ord}_{Q_\infty}(x)=-1,\quad
  \operatorname{ord}_{Q_1}(x)=2.
\]
Ramification occurs exactly at \(P\) and \(Q_\infty\), giving two simple
branch points. Hurwitz's formula \cite{Hartshorne1977} then gives
\[
  2g(\overline{X})-2=2(2g(\overline{Y})-2)+2=2(2\cdot 3-2)+2,
\]
hence \(g(\overline{X})=6\).
\end{proof}

\begin{proposition}[Generic Galois group]
  \label{prop:sync-hatdelta-galois-s6-generic}
  Viewed as a polynomial in \(w\), the completed determinant
  \(\widehat\Delta(w,s)\) has generic Galois group \(S_6\) over
  \(\QQ(s)\).
\end{proposition}

\begin{proof}
The polynomial has degree \(6\) in \(w\). Write
\(D(s):=\operatorname{disc}_w\widehat\Delta(w,s)\in\QQ[s]\). A direct
computation shows that \(D(s)\) has degree~\(20\) with leading
coefficient~\(-256\). Since the leading coefficient is negative,
\(D(s)\) takes negative values for \(s\gg 0\) and is therefore not a
square in~\(\QQ(s)\). It follows that the Galois group
\(G\le S_6\) is not contained in~\(A_6\).

To pin down~\(G\), we specialise and apply the following standard
principle \cite{Serre2008Topics}: for a monic separable polynomial
\(f(w,s_0)\in\QQ[w]\) obtained from \(\widehat\Delta(w,s)\) by
specialisation \(s\mapsto s_0\in\QQ\), the Galois group over \(\QQ\) of
the specialised polynomial embeds into \(G\). The specialised polynomial
is separable (and the embedding therefore valid) whenever
\(\gcd(f(w,s_0),\partial_w f(w,s_0))=1\) in \(\QQ[w]\).

We carry out three specialisations.

\medskip
\noindent\textit{Step 1 (\(s=3\), mod \(7\)).}\quad One has
\[
  \widehat\Delta(w,3)=8w^6+9w^5-4w^4+9w^3-5w^2-3w+1,
\]
which is separable over \(\QQ\) (verified by computing its discriminant).
One checks
\[
  \gcd\!\bigl(\widehat\Delta(w,3),\,\partial_w\widehat\Delta(w,3)\bigr)
  \equiv 1\pmod{7},
\]
so the mod-\(7\) reduction is squarefree. Reducing modulo~\(7\) gives
\[
  w^6+2w^5+3w^4+2w^3+2w^2+4w+1,
\]
which is irreducible over~\(\mathbb{F}_7\). Hence \(G\) contains a
\(6\)-cycle and is therefore transitive.

\medskip
\noindent\textit{Step 2 (\(s=2\), mod \(7\)).}\quad One checks
\[
  \gcd\!\bigl(\widehat\Delta(w,2),\,\partial_w\widehat\Delta(w,2)\bigr)
  \equiv 1\pmod{7},
\]
so the reduction is squarefree. Then
\[
  \widehat\Delta(w,2)\equiv
  (w+6)(w+5)(w+2)(w+1)(3w^2+3w+2)\pmod 7,
\]
where \(3w^2+3w+2\) is irreducible over \(\mathbb{F}_7\). This gives
cycle type \((1)(1)(1)(1)(2)\) in \(G\), so \(G\) contains a transposition.

\medskip
\noindent\textit{Step 3 (\(s=3\), mod \(19\)).}\quad One checks
\[
  \gcd\!\bigl(\widehat\Delta(w,3),\,\partial_w\widehat\Delta(w,3)\bigr)
  \equiv 1\pmod{19},
\]
so the reduction is squarefree. Then
\[
  \widehat\Delta(w,3)\equiv
  (w+16)\bigl(8w^5+14w^4+9w^2+3w+6\bigr)\pmod{19},
\]
with the quintic factor irreducible over \(\mathbb{F}_{19}\). This gives
cycle type \((1)(5)\) in \(G\), so \(G\) contains a \(5\)-cycle.

\medskip
A transitive subgroup of \(S_6\) containing a \(5\)-cycle is
primitive. Indeed, let \(\sigma\in G\) be a \(5\)-cycle. If \(\mathcal{B}\)
were a non-trivial block, then \(|\mathcal{B}|\) would divide \(6\). If
\(\mathcal{B}\) meets the support of \(\sigma\), then
\(\mathcal{B}\cap\sigma(\mathcal{B})\neq\varnothing\), so
\(\sigma(\mathcal{B})=\mathcal{B}\), and therefore \(\mathcal{B}\)
contains all five moved points of \(\sigma\). Thus
\(|\mathcal{B}|=5\) or \(|\mathcal{B}|=6\), which is impossible for a
proper block. Hence \(G\) is primitive \cite[Ch.~3]{Dixon1996}. A
primitive subgroup of \(S_6\) that contains a transposition must be the
full symmetric group \cite[Ch.~3]{Dixon1996}. Hence \(G=S_6\). This
completes part~\textup{(iii)} of \Cref{thm:intro-kernel-geometry}.
\end{proof}

\subsection{Cyclotomic resultants and twisted Perron roots}

For \(m\ge 2\), let
\[
  s_m:=2\cos(\pi/m),
\]
let \(\Phi_m^+(s)\) be the minimal polynomial of \(s_m\) over \(\QQ\), and
define
\[
  R_m(w):=\Res_s\bigl(\widehat\Delta(w,s),\Phi_m^+(s)\bigr)\in\ZZ[w].
\]
The notation \(\Phi_m^+\) is non-standard; it denotes the minimal polynomial
of the real part \(2\cos(\pi/m)\) of a primitive \(2m\)-th root of unity.
Concretely, the conjugates of \(s_m\) under \(\Gal(\QQ(s_m)/\QQ)\) are
\[
  A_m:=\Bigl\{2\cos\!\tfrac{\pi a}{m}:a\in(\ZZ/2m\ZZ)^\times/\{\pm 1\}\Bigr\},
\]
and \(\Phi_m^+(s)=\prod_{\alpha\in A_m}(s-\alpha)\). The degree is
\(\deg\Phi_m^+=|A_m|=\varphi(2m)/2\).

The corresponding twisted Perron root is
\[
  \rho_m:=\frac{1}{\min\{\abs{w}:R_m(w)=0\}}.
\]

\begin{theorem}[Cyclotomic resultant degree law]
  \label{thm:sync-cyclotomic-degree-law}
  For every \(m\ge 4\),
  \[
    \deg_w R_m=3\varphi(2m).
  \]
  Moreover, if \(m\) is even then \(R_m(w)\) is an even polynomial.
\end{theorem}

\begin{proof}
Let \(d_m:=\deg\Phi_m^+=\varphi(2m)/2\), and let \(A_m\) be the
conjugate set
\[
  A_m:=\Bigl\{2\cos\!\tfrac{\pi a}{m}:a\in(\ZZ/2m\ZZ)^\times/\{\pm 1\}\Bigr\}.
\]
Then \(\Phi_m^+(s)=\prod_{\alpha\in A_m}(s-\alpha)\), and the
product formula for resultants \cite{CoxLittleOShea2005} gives
\[
  R_m(w)
  =\Res_s\bigl(\widehat\Delta(w,s),\Phi_m^+(s)\bigr)
  =\prod_{\alpha\in A_m}\widehat\Delta(w,\alpha).
\]
For \(m\ge 4\), the leading coefficient of \(\widehat\Delta(w,\alpha)\)
as a polynomial in \(w\) is \(\alpha^2-1\). Since \(\alpha=2\cos(\pi a/m)\)
for \(a\in(\ZZ/2m\ZZ)^\times\), we have \(\alpha=\pm 1\) only if
\(\zeta+\zeta^{-1}=\pm 1\) for a primitive \(2m\)-th root of unity
\(\zeta\), which occurs exactly when \(2m=6\), i.e.\ \(m=3\). Hence
for \(m\ge 4\) one has \(\alpha^2-1\neq 0\) for every \(\alpha\in A_m\),
so each factor \(\widehat\Delta(w,\alpha)\) has degree exactly \(6\) in
\(w\). Therefore
\[
  \deg_w R_m=6\,|A_m|=6d_m=3\varphi(2m).
\]

For evenness when \(m\) is even: the set \(A_m\) is stable under
\(\alpha\mapsto-\alpha\), because \(\cos(\pi a/m)\) and
\(-\cos(\pi a/m)=\cos(\pi(m-a)/m)\) belong to the same conjugate set.
The symmetry \(\widehat\Delta(w,-s)=\widehat\Delta(-w,s)\) then gives
\[
  R_m(-w)
  =\prod_{\alpha\in A_m}\widehat\Delta(-w,\alpha)
  =\prod_{\alpha\in A_m}\widehat\Delta(w,-\alpha)
  =\prod_{\alpha\in A_m}\widehat\Delta(w,\alpha)
  =R_m(w).\qedhere
\]
\end{proof}

\begin{corollary}[The twist \(m=2\)]
  \label{cor:rho-m2-closed-form}
  One has
  \[
    R_2(w)=\widehat\Delta(w,0)
    =1-5w^2+5w^4-w^6
    =(1-w^2)(w^4-4w^2+1),
  \]
  and therefore
  \[
    \rho_2=\sqrt{2+\sqrt3}.
  \]
\end{corollary}

\begin{proof}
Since \(s_2=0\), the resultant is obtained by direct substitution. The
smallest-modulus zero of \(R_2\) is \(\sqrt{2-\sqrt3}\), whose
reciprocal is the stated value. Together with
\Cref{thm:sync-cyclotomic-degree-law}, this proves
parts~\textup{(i)} and~\textup{(ii)} of
\Cref{thm:intro-kernel-cyclotomic}.
\end{proof}

\subsection{Endpoint jets and large-\texorpdfstring{$m$}{m} asymptotics}

Let \(w(s)\) denote the simple branch of \(\widehat\Delta(w,s)=0\) near
\((w,s)=(1/3,2)\), and set
\[
  \rho(s):=\frac{1}{w(s)}.
\]

\begin{proposition}[Endpoint jet]
  \label{prop:sync-endpoint-jet}
  The branch \(w(s)\) is analytic near \(s=2\), with
  \[
    w(2)=\frac13,\qquad
    w'(2)=-\frac{11}{153},\qquad
    \rho'(2)=\frac{11}{17},\qquad
    \rho''(2)=\frac{1351}{9826}.
  \]
  Moreover, if \(\delta:=2-s\), then
  \begin{align*}
    \rho(2-\delta)
    &=3-\frac{11}{17}\delta+\frac{1351}{19652}\delta^2
      -\frac{17133}{45435424}\delta^3 \\
    &\quad
      -\frac{350037409}{105046700288}\delta^4
      -\frac{47614595705}{242867971065856}\delta^5 \\
    &\quad
      +\frac{141889658001627}{561510749104259072}\delta^6
      +O(\delta^7).
  \end{align*}
\end{proposition}

\begin{proof}
One has
\[
  \widehat\Delta(1/3,2)=0,\qquad
  \partial_w\widehat\Delta(1/3,2)\neq 0.
\]
Hence the implicit function theorem gives a unique analytic branch
\(w(s)\) near \(s=2\). Differentiating the identity
\(\widehat\Delta(w(s),s)=0\) repeatedly and solving for \(w^{(k)}(2)\)
at each order yields the stated values of \(w'(2)\), \(\rho'(2)\), and
\(\rho''(2)\). Continuing through sixth order by the same implicit
differentiation procedure (a finite algebraic computation verified by
computer algebra), one obtains the six-term jet of \(\rho\) displayed
above. The coefficient of \(\delta^k\) at order \(k\le 6\) is a rational
number determined solely by the partial derivatives of \(\widehat\Delta\)
up to order \(k+1\) at \((1/3,2)\).
\end{proof}

\begin{lemma}[Root separation]
  \label{lem:root-separation}
  For all sufficiently large \(m\), the smallest-modulus root of
  \(R_m(w)\) is \(w^*(s_m)\), the unique root of
  \(\widehat\Delta(w,s_m)=0\) near \(w=1/3\). In particular,
  \(\rho_m=\rho(s_m)=1/w^*(s_m)\).
\end{lemma}

\begin{proof}
By the product formula,
\[
  R_m(w)=\prod_{\alpha\in A_m}\widehat\Delta(w,\alpha).
\]
We must show that, for large \(m\), the smallest-modulus zero of
\(R_m\) comes from the factor \(\widehat\Delta(w,s_m)\) and not from any
factor with \(\alpha\neq s_m\).

\textit{Step 1: Isolation at the endpoint.}\quad
At \(s=2\), the polynomial \(\widehat\Delta(w,2)\) factorises as
\[
  \widehat\Delta(w,2)=(w-1)(w+1)(3w-1)(w^3-w^2+w+1).
\]
The root \(w=1/3\) is simple; all other roots satisfy \(|w|>0.54\).
Fix a closed disk \(D:=\{|w-1/3|<0.1\}\). By continuity of roots
of \(\widehat\Delta(\cdot,s)\) in \(s\), there exists \(\varepsilon>0\) such
that for \(|s-2|<\varepsilon\):
\begin{itemize}
  \item \(\widehat\Delta(\cdot,s)\) has exactly one simple root
        \(w^*(s)\in D\);
  \item every other root of \(\widehat\Delta(\cdot,s)\) lies outside
        \(\{|w|<0.45\}\).
\end{itemize}
Moreover, differentiating \(\widehat\Delta(w(s),s)=0\) at \(s=2\) gives
\(w^{*\prime}(2)=-11/153<0\), so \(w^*\) is strictly decreasing and
\(w^*(s)\in D\) satisfies \(|w^*(s)|<|w^*(2)|=1/3\) for \(s\) slightly
below \(2\).

\textit{Step 2: Uniform lower bound away from the endpoint.}\quad
The minimum modulus function \(\lambda(s):=\min\{|w|:\widehat\Delta(w,s)=0\}\)
is continuous on \([-2,2]\). The symmetry \(\widehat\Delta(w,-s)=\widehat\Delta(-w,s)\)
gives \(\lambda(-s)=\lambda(s)\), so the analysis near \(s=-2\) is
identical. On the compact set \([-2,2]\setminus\bigl((-2,-2+\varepsilon)
\cup(2-\varepsilon,2)\bigr)\), continuity gives a uniform lower bound
\[
  \lambda(s)\ge c_0>w^*(2-\varepsilon/2)
\]
for some constant \(c_0\).

\textit{Step 3: Comparing conjugates.}\quad
Every \(\alpha\in A_m\setminus\{s_m\}\) satisfies
\(|\alpha|\le 2\cos(3\pi/2m)<s_m\) for \(m\ge 4\) (with equality only
for \(\alpha=-s_m\) when \(m\) is even, which also achieves modulus
\(s_m\)). Since \(s_m\to 2\) and \(s_m=2-\pi^2/m^2+O(m^{-4})\), for
all sufficiently large \(m\) every conjugate \(\alpha\in A_m\) with
\(\alpha\neq\pm s_m\) lies in the interior of \([-2+\varepsilon,2-\varepsilon]\),
where the uniform bound of Step 2 applies. The factors
\(\widehat\Delta(\cdot,\pm s_m)\) contribute root \(w^*(s_m)\) (or
its negative by parity); factors \(\widehat\Delta(\cdot,\alpha)\) for
\(\alpha\neq\pm s_m\) all have minimum-modulus roots bounded below by
\(c_0>w^*(2-\varepsilon/2)>w^*(s_m)\).

Hence for all sufficiently large \(m\),
\[
  \min\{|w|:R_m(w)=0\}=|w^*(s_m)|,
\]
which gives \(\rho_m=1/|w^*(s_m)|=\rho(s_m)\).
\end{proof}

\begin{theorem}[Asymptotic expansion of the twisted Perron roots]
  \label{thm:rho-gap-m12}
  As \(m\to\infty\),
  \begin{align*}
    \rho_m
    &=3-\frac{11\pi^2}{17m^2}
      +\frac{1808\pi^4}{14739m^4}
      -\frac{27872249\pi^6}{2044594080m^6} \\
    &\quad
      -\frac{77645433997\pi^8}{33089710590720m^8}
      +\frac{2976802482983509\pi^{10}}
             {3442653489858508800m^{10}} \\
    &\quad
      +\frac{3832209193654464509\pi^{12}}
             {23878244605658617036800m^{12}}
      +O(m^{-14}).
  \end{align*}
  In particular,
  \[
    3-\rho_m\sim \frac{11\pi^2}{17m^2}.
  \]
\end{theorem}

\begin{proof}
By \Cref{lem:root-separation}, \(\rho_m=\rho(s_m)\) for all sufficiently
large \(m\). The Taylor series of \(2-s_m=2-2\cos(\pi/m)\) at infinity is
\[
  2-s_m
  =\frac{\pi^2}{m^2}-\frac{\pi^4}{12m^4}+\frac{\pi^6}{360m^6}
   -\frac{\pi^8}{20160m^8}+\frac{\pi^{10}}{1814400m^{10}}
   -\frac{\pi^{12}}{239500800m^{12}}+O(m^{-14}).
\]
Setting \(\delta=2-s_m\) and substituting into the sixth-order jet of
\Cref{prop:sync-endpoint-jet} yields the displayed asymptotic series,
where the coefficient of \(m^{-2k}\) is determined by the \(\delta^k\)
term of the jet together with the lower-order terms of the Taylor series
of \(\delta\). This proves part~\textup{(iii)} of
\Cref{thm:intro-kernel-cyclotomic}.
\end{proof}

\begin{remark}
The asymptotic expansion is already numerically accurate for moderate
values of \(m\). Solving the specialised equations \(R_m(w)=0\) for the
smallest-modulus root and comparing with the truncation of
\Cref{thm:rho-gap-m12} gives
\[
  \begin{array}{c|c|c|c}
    m & \rho_m & \text{12th-order approximation} & \text{absolute error} \\
    \hline
    4  & 2.644080451866 & 2.644083856227 & 3.40\times 10^{-6} \\
    6  & 2.831532258235 & 2.831532269934 & 1.17\times 10^{-8} \\
    8  & 2.903081377378 & 2.903081377585 & 2.07\times 10^{-10} \\
    10 & 2.937319429068 & 2.937319429077 & 9.09\times 10^{-12}
  \end{array}
\]
for \(m=4,6,8,10\).
\end{remark}

\subsection{Primitive orbit counts under cyclic lift}

\begin{proposition}[Cyclic-lift primitive asymptotics]
  \label{prop:cyclic-lift-primitive-asymptotics}
  Let \(M\) be a primitive matrix with Perron root \(\lambda\), and for
  \(q\ge 2\) let \(M^{\langle q\rangle}:=M\otimes P_q\). Then
  \[
    p_n(M^{\langle q\rangle})=0
    \qquad (q\nmid n),
  \]
  and for \(n=qm\),
  \[
    p_{qm}(M^{\langle q\rangle})
    =\frac{1}{m}\sum_{d\mid m}\mu(d)\Tr(M^{qm/d})
    =\frac{\lambda^{qm}}{m}
      +O\!\left(
        \frac{\max\{\Lambda(M)^{qm},\lambda^{qm/2}\}}{m}
      \right).
  \]
\end{proposition}

\begin{proof}
Since \((M\otimes P_q)^n=M^n\otimes P_q^n\) and
\[
  \Tr(P_q^n)=
  \begin{cases}
    q, & q\mid n,\\
    0, & q\nmid n,
  \end{cases}
\]
the trace sequence of \(M^{\langle q\rangle}\) vanishes outside the
multiples of \(q\). Applying M\"obius inversion and then
\Cref{lem:primitive-orbit-asymptotic} gives the result.
\end{proof}
