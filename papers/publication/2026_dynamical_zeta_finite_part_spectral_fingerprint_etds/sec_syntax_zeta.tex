\subsection{A density dichotomy for deterministic automata}

$\mathcal A=(Q,\Sigma,\delta,q_0,F)$ be a complete deterministic
finite automaton. For each $m\ge 0$, let
\[
  a_m(\mathcal A)
  :=\#\{w\in\Sigma^m:\mathcal A\text{ accepts }w\}.
\]

\subsection{Binary primes and finite automata}

For each $m\ge 1$, let $\mathcal P_m^{(2)}$ be the set of binary words
of length $m$ whose first digit is $1$ and whose binary value is
prime. Let $p_m^{(2)}:=\abs{\mathcal P_m^{(2)}}$. By the prime number
theorem,
\[
  p_m^{(2)}
  =\pi(2^m)-\pi(2^{m-1})
  \asymp \frac{2^m}{m}.
\]

\begin{corollary}[Symmetric-difference lower bound for binary primes]
  \label{cor:dfa-prime-symmetric-diff}
  See \cite[Corollary~2.5]{MaZhang2026PrimeLanguageObstructions}.
\end{corollary}

\begin{corollary}[No simultaneous recall and precision for fixed DFA]
  \label{cor:dfa-prime-recall-precision}
  Let
  \[
    \mathrm{Rec}_m
    :=\frac{\abs{L_m\cap\mathcal P_m^{(2)}}}{\abs{\mathcal P_m^{(2)}}},
    \qquad
    \mathrm{Prec}_m
    :=\frac{\abs{L_m\cap\mathcal P_m^{(2)}}}{\abs{L_m}}
  \]
  whenever $\abs{L_m}>0$. Then every DFA satisfies one of the
  following.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item $\mathrm{Rec}_m\to 0$ exponentially fast.
    \item $\mathrm{Prec}_m\to 0$ along an infinite arithmetic
    progression of lengths.
  \end{enumerate}
  In particular no fixed DFA has both recall and precision bounded
  away from zero for infinitely many lengths.
\leanverified{paper_etds_dfa_prime_recall_precision}
\end{corollary}

\begin{proof}
  In the sparse case,
  \[
    \mathrm{Rec}_m
    \le \frac{\abs{L_m}}{\abs{\mathcal P_m^{(2)}}}
    \ll \eta^m m
  \]
  for some $\eta\in(0,1)$, hence $\mathrm{Rec}_m\to 0$
  exponentially. In the positive-density case,
  \[
    \mathrm{Prec}_m
    \le \frac{\abs{\mathcal P_m^{(2)}}}{\abs{L_m}}
    \ll \frac{1}{m}
  \]
  along the relevant arithmetic progression.
\end{proof}

\subsection{Zeckendorf syntax}

Let
\[
  \mathcal Z
  :=\{x_1\cdots x_m\in\{0,1\}^\ast:
      x_ix_{i+1}\neq 11,\ x_m=1\}
\]
be the language of canonical Zeckendorf words. We assign to
$w=x_1\cdots x_m\in\mathcal Z$ the value
\[
  \mathrm{val}(w):=\sum_{k=1}^m x_kF_{k+1},
\]
where $F_n$ is the Fibonacci sequence \cite{Zeckendorf1972}. The map
$w\mapsto \mathrm{val}(w)$ is bijective from $\mathcal Z$ onto
$\NN_{\ge 1}$.

\begin{theorem}[Regular Zeckendorf languages have power-law growth]
  \label{thm:zeckendorf-regular-powerlaw}
  Let $L\subseteq\mathcal Z$ be regular and define
  \[
    N_L(T):=\#\{w\in L:\mathrm{val}(w)\le T\}.
  \]
  Then there exists $\alpha\in[0,1]$ such that
  \[
    N_L(T)=T^{\alpha+o(1)}
    \qquad (T\to\infty).
  \]
\leanverified{paper_etds_zeckendorf_regular_powerlaw}
\end{theorem}

\begin{proof}
  Let $a_m:=\#\{w\in L:\abs{w}=m\}$. Since $L$ is regular,
  $a_m=u^{\mathsf T}B^m v$ for some non-negative integer matrix $B$.
  Hence $a_m=\lambda^{m+o(m)}$ for a well-defined exponential growth
  rate $\lambda\in[0,\varphi]$; the upper bound comes from
  $\abs{\mathcal Z\cap\{0,1\}^m}=F_{m+1}=\varphi^{m+o(m)}$.

  If $F_{n+1}\le T<F_{n+2}$, then every Zeckendorf word of value at
  most $T$ has length at most $n$, while every word of length at most
  $n-1$ has value at most $T$. Therefore
  \[
    \sum_{m\le n-1}a_m\le N_L(T)\le \sum_{m\le n}a_m.
  \]
  Since $n=(\log T)/(\log\varphi)+O(1)$ and
  $\sum_{m\le n}a_m=\lambda^{n+o(n)}$, we obtain
  \[
    N_L(T)
    =\exp\!\left(\frac{\log\lambda}{\log\varphi}\log T+o(\log T)\right)
    =T^{\alpha+o(1)}
  \]
  with $\alpha:=\log\lambda/\log\varphi\in[0,1]$.
\end{proof}

\begin{corollary}[Prime Zeckendorf language is not regular]
  \label{cor:zeckendorf-prime-language-not-regular}
  Let
  \[
    \mathcal P_{\mathcal Z}
    :=\{\mathrm{Zeck}(p)\in\mathcal Z:p\text{ prime}\},
  \]
  where $\mathrm{Zeck}(n)$ denotes the canonical Zeckendorf
  representation of $n$. Then $\mathcal P_{\mathcal Z}$ is not a
  regular language.
\leanverified{paper_etds_zeckendorf_prime_language_not_regular}
\end{corollary}

\begin{proof}
  If $\mathcal P_{\mathcal Z}$ were regular, then
  \Cref{thm:zeckendorf-regular-powerlaw} would give
  \[
    \pi(T)=N_{\mathcal P_{\mathcal Z}}(T)=T^{\alpha+o(1)}
  \]
  for some $\alpha\in[0,1]$. This is impossible because
  $\pi(T)\sim T/\log T$: for every $\alpha<1$ one has
  $\pi(T)/T^\alpha\to\infty$, while $\pi(T)/T\to 0$.
\end{proof}

\begin{theorem}[Prime Zeckendorf shift is not sofic]
  \label{thm:zeckendorf-primes-not-sofic}
  Let
  \[
    \mathcal P_m^{(\mathrm Z)}
    :=\{w\in\mathcal Z:\abs{w}=m,\ \mathrm{val}(w)\text{ prime}\}.
  \]
  Then the length slices $\mathcal P_m^{(\mathrm Z)}$ cannot be the
  admissible words of any sofic shift.
\leanverified{paper_etds_zeckendorf_primes_not_sofic}
\end{theorem}

\begin{proof}
  Since $\mathcal Z\cap\{0,1\}^m$ parametrises the integers in
  $[1,F_{m+2})$, one has
  \[
    \abs{\mathcal P_m^{(\mathrm Z)}}
    =\pi(F_{m+2})+O(1)
    \asymp \frac{\varphi^m}{m}.
  \]
  If these length slices came from a sofic shift, then their counting
  sequence would be an exponential polynomial
  $\sum_j Q_j(m)\lambda_j^m$, equivalently a linear recurrence
  sequence \cite{LindMarcus1995,Kitchens1998}. Such a sequence has
  dominant asymptotic $m^k\Lambda^m$, never $\Lambda^m/m$. This
  contradicts the displayed estimate.
\end{proof}

\subsection{Analytic obstructions}

\begin{theorem}[Prime-supported Euler products have a natural boundary]
  \label{thm:euler-product-natural-boundary}
  Let $(b_n)_{n\ge 1}\subseteq\ZZ_{\ge 0}$ and assume that
  $b_p>0$ for infinitely many primes $p$. Suppose that
  \[
    Z_b(z):=\prod_{n\ge 1}(1-z^n)^{-b_n}
  \]
  converges to a holomorphic function on $\abs{z}<1$. Then the unit
  circle is a natural boundary for $Z_b$.
\end{theorem}

\begin{proof}
  If $b_p>0$, then the factor $(1-z^p)^{-b_p}$ has poles at all $p$th
  roots of unity. Since every other factor has non-negative exponent,
  no cancellation can occur there. Thus every prime $p$ with $b_p>0$
  contributes poles at the $p$th roots of unity. These roots are dense
  in the unit circle because the set of such primes is infinite. A
  holomorphic continuation across any boundary arc would therefore meet
  infinitely many poles, which is impossible. Hence $\abs{z}=1$ is a
  natural boundary.
\leanverified{paper_etds_euler_product_natural_boundary}
\end{proof}

\begin{proposition}[Imaginary-periodicity rigidity of finite determinants]
  \label{prop:finite-zeta-imaginary-periodicity}
  For every finite matrix $M$, the function
  \[
    F_M(s):=\det(I-e^{-s}M)^{-1}
  \]
  is $2\pi i$-periodic. In particular, the Riemann
  $\zeta$-function cannot coincide with $F_M$ on any open subset of
  $\Re(s)>1$.
\end{proposition}

\begin{proof}
  The identity $e^{-(s+2\pi i)}=e^{-s}$ implies
  $F_M(s+2\pi i)=F_M(s)$. On the other hand, for large real $\sigma>1$,
  \[
    \zeta(\sigma+2\pi i)-\zeta(\sigma)
    =\sum_{n\ge 2}n^{-\sigma}(e^{-2\pi i\log n}-1).
  \]
  The $n=2$ term has size comparable to $2^{-\sigma}$, while the tail
  is $O(3^{-\sigma})$, so the difference is non-zero for sufficiently
  large $\sigma$. Hence $\zeta(s)$ is not $2\pi i$-periodic on
  $\Re(s)>1$.
\leanverified{paper_etds_finite_zeta_imaginary_periodicity}
\end{proof}

\begin{remark}
  \Cref{thm:euler-product-natural-boundary,prop:finite-zeta-imaginary-periodicity}
  are complementary. The first obstructs finite-state models at the
  level of the boundary geometry of the Euler product. The second
  obstructs them already after the logarithmic change of variables.
\end{remark}
