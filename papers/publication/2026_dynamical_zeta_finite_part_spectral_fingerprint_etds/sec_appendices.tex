\section{Auxiliary cyclic formulas and cyclic detection}
\label{app:supplementary}

This appendix collects the auxiliary results used in
\Cref{sec:finite-part,sec:shadows}. The first subsection concerns
cyclic lifts, and the second records the orthogonal decomposition
underlying cyclic detection.

\subsection{Auxiliary cyclic-lift formulas}

\begin{proposition}[Multiple-sum expansion for cyclic lifts]
  \label{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}
  For every \(q\ge 1\),
  \[
    \Psi_q(A)=\sum_{m\ge 1}\frac{u_{qm}(A)}{m},
  \]
  and therefore
  \[
    \Psi_q(A)=u_q(A)+O\!\left(\Lambda_{\mathrm{rel}}(A)^{2q}\right)
    \qquad (q\to\infty).
  \]
  More precisely,
  \[
    \abs{\Psi_q(A)-u_q(A)}
    \le
    \frac{d-1}{2(1-\Lambda_{\mathrm{rel}}(A))}
    \Lambda_{\mathrm{rel}}(A)^{2q}.
  \]
\end{proposition}

\begin{proof}
  By \Cref{thm:finite-part-cyclic-lift-reduced-constant-closed},
  \[
    \Psi_q(A)
    =-\sum_{j=2}^d\log\!\left(1-\left(\frac{\mu_j}{\lambda}\right)^q\right).
  \]
  Expanding \(-\log(1-x)=\sum_{m\ge 1}x^m/m\) and exchanging the finite
  sum over \(j\) with the absolutely convergent series over \(m\) gives
  the first identity. The remainder estimate follows from
  \[
    |u_{qm}(A)|\le (d-1)\Lambda_{\mathrm{rel}}(A)^{qm}.
  \]
  Indeed,
  \[
    \begin{aligned}
      \abs{\Psi_q(A)-u_q(A)}
      &\le
      \sum_{m\ge 2}\frac{\abs{u_{qm}(A)}}{m} \\
      &\le
      \frac{d-1}{2}\sum_{m\ge 2}\Lambda_{\mathrm{rel}}(A)^{qm} \\
      &\le
      \frac{d-1}{2(1-\Lambda_{\mathrm{rel}}(A))}
      \Lambda_{\mathrm{rel}}(A)^{2q}.
    \end{aligned}
  \]
\end{proof}

\begin{proposition}[M\"obius inversion on multiples]
  \label{prop:finite-part-cyclic-lift-mobius-inversion}
  For every \(n\ge 1\),
  \[
    \frac{u_n(A)}{n}
    =\sum_{m\ge 1}\frac{\mu(m)}{mn}\,\Psi_{mn}(A).
  \]
\end{proposition}

\begin{proof}
  Set \(b_n:=u_n(A)/n\). Then
  \[
    \frac{\Psi_q(A)}{q}
    =\sum_{m\ge 1}b_{qm}
  \]
  by \Cref{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}. Since
  \[
    \sum_{s\ge 1}\abs{b_{ns}}
    \le \frac{d-1}{n}\sum_{s\ge 1}\Lambda_{\mathrm{rel}}(A)^{ns}
    <\infty,
  \]
  the rearrangement below is justified by absolute convergence:
  \begin{align*}
    \sum_{m\ge 1}\frac{\mu(m)}{mn}\Psi_{mn}(A)
    &=
    \sum_{m\ge 1}\mu(m)\sum_{r\ge 1}b_{mnr} \\
    &=
    \sum_{s\ge 1}b_{ns}\sum_{m\mid s}\mu(m)
    =b_n,
  \end{align*}
  since \(\sum_{m\mid s}\mu(m)=0\) unless \(s=1\).
\end{proof}

\begin{lemma}[Limsup of a finite exponential sum]
  \label{lem:finite-part-exponential-sum-limsup}
  Let
  \[
    f_q=\sum_{j=1}^r c_j\alpha_j^q
    \qquad (q\ge 1),
  \]
  where \(c_j\in\CC\setminus\{0\}\) and \(\alpha_j\in\CC\). Then
  \[
    \limsup_{q\to\infty}\abs{f_q}^{1/q}
    =\max_{1\le j\le r}\abs{\alpha_j}.
  \]
\end{lemma}

\begin{proof}
  Set \(R:=\max_j\abs{\alpha_j}\). The upper bound
  \(\limsup_{q\to\infty}\abs{f_q}^{1/q}\le R\) is immediate from
  \(\abs{f_q}\le (\sum_j\abs{c_j})R^q\).

  For the reverse inequality, collect equal frequencies on the boundary
  circle and write
  \[
    f_q
    =R^q\bigl(g_q+O(\vartheta^q)\bigr),
    \qquad
    g_q:=\sum_{k=1}^s d_k\beta_k^q,
  \]
  where \(\vartheta\in[0,1)\), the numbers \(\beta_1,\dots,\beta_s\in\TT\)
  are pairwise distinct, and \(d_k\neq 0\). Then
  \[
    \frac{1}{N}\sum_{q=1}^N\abs{g_q}^2
    =\sum_{k,\ell=1}^s d_k\overline{d_\ell}\,
      \frac{1}{N}\sum_{q=1}^N(\beta_k\overline{\beta_\ell})^q.
  \]
  As \(N\to\infty\), the inner Ces\`aro average tends to \(1\) when
  \(k=\ell\) and to \(0\) otherwise. Hence
  \[
    \frac{1}{N}\sum_{q=1}^N\abs{g_q}^2
    \longrightarrow
    \sum_{k=1}^s\abs{d_k}^2>0.
  \]
  Therefore \(\limsup_{q\to\infty}\abs{g_q}>0\), so there exist
  \(c>0\) and infinitely many \(q\) such that \(\abs{g_q}\ge c\). Along
  that subsequence,
  \[
    \abs{f_q}\ge \frac{c}{2}R^q
  \]
  for all sufficiently large \(q\), because \(\vartheta^q\to 0\). This
  yields
  \(\limsup_{q\to\infty}\abs{f_q}^{1/q}\ge R\), completing the proof.
\end{proof}

\begin{proposition}[Reconstruction from one root-of-unity layer]
  \label{thm:finite-part-cyclic-lift-single-q-torsion-reconstruction}
  Write
  \[
    F_A(t)=\sum_{n=0}^{d-1}a_n t^n
    \qquad\text{with } a_0=1.
  \]
  If \(q\ge d\), then the values
  \(\{F_A(\omega_q^j)\}_{0\le j\le q-1}\) determine every coefficient
  \(a_n\). More precisely,
  \[
    a_n=\frac{1}{q}\sum_{j=0}^{q-1}
      F_A(\omega_q^j)\omega_q^{-jn}
    \qquad (0\le n\le d-1).
  \]
\end{proposition}

\begin{proof}
  The polynomial identity
  \[
    F_A(\omega_q^j)=\sum_{n=0}^{d-1}a_n\omega_q^{jn}
  \]
  holds for \(0\le j\le q-1\). Averaging against
  \(\omega_q^{-jm}\) and using discrete orthogonality gives
  \[
    \frac{1}{q}\sum_{j=0}^{q-1}
      F_A(\omega_q^j)\omega_q^{-jm}
    =
    \sum_{n=0}^{d-1}a_n
      \frac{1}{q}\sum_{j=0}^{q-1}\omega_q^{j(n-m)}.
  \]
  Since \(q\ge d\), the inner sum equals \(1\) when \(n=m\) and \(0\)
  otherwise.
\end{proof}

\begin{corollary}[Root-of-unity values of \(\zeta_A\) determine the
  reduced spectrum]
  \label{cor:finite-part-cyclic-lift-zeta-torsion-determines-spectrum}
  For \(1\le j\le q-1\),
  \[
    F_A(\omega_q^j)
    =\frac{\det(I-(\omega_q^j/\lambda)A)}{1-\omega_q^j}
    =\frac{1}{1-\omega_q^j}\,
      \zeta_A(\omega_q^j/\lambda)^{-1},
  \]
  while
  \[
    F_A(1)=\det\nolimits'(I-A/\lambda)=C(A)^{-1}.
  \]
  Consequently, if \(q\ge d\), then the data
  \[
    \Bigl(C(A),\ \{\zeta_A(\omega_q^j/\lambda)\}_{j=1}^{q-1}\Bigr)
  \]
  determine the non-Perron spectrum of \(A\) with multiplicity.
\end{corollary}

\begin{proof}
  The displayed identities are immediate from the definitions of
  \(F_A\), \(\zeta_A\), and \(C(A)\). The reconstruction statement
  follows from
  \Cref{thm:finite-part-cyclic-lift-single-q-torsion-reconstruction}.
\end{proof}

\subsection{Orthogonal decomposition and cyclic detection}

\begin{proposition}[Orthogonal decomposition]
  \label{thm:abelian-shadow-defect}
  Equip the class functions on \(G\) with the inner product
  \[
    \langle F,H\rangle_G
    :=
    \sum_{C\subset G}\frac{|C|}{|G|}F(C)\overline{H(C)}.
  \]
  With respect to this inner product,
  \(\mathcal B_A^{\mathrm{ab}}\) and \(\mathcal B_A^{\mathrm{na}}\)
  are orthogonal, and
  \begin{equation}\label{eq:abelian-shadow-defect-energy}
    \norm{\mathcal B_A^{\mathrm{ab}}}_G^2
    =
    \sum_{\substack{\chi\in\Irr(G)\\ \dim\chi=1,\ \chi\neq \mathbf 1}}
      \abs{\Pi_\chi(\lambda^{-1})}^2,
    \qquad
    \norm{\mathcal B_A^{\mathrm{na}}}_G^2
    =
    \sum_{\substack{\varrho\in\Irr(G)\\ \dim\varrho>1}}
      \abs{\Pi_\varrho(\lambda^{-1})}^2.
  \end{equation}
  Moreover, the following are equivalent:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item \(\mathcal B_A\) factors through the abelianisation
      \(q_{\mathrm{ab}}:G\twoheadrightarrow G_{\mathrm{ab}}\);
    \item \(\mathcal B_A^{\mathrm{na}}\equiv 0\);
    \item \(\Pi_\varrho(\lambda^{-1})=0\) for every
      \(\varrho\in\Irr(G)\) with \(\dim\varrho>1\).
  \end{enumerate}
\end{proposition}

\begin{proof}
  The orthogonality of the irreducible characters in the class-function
  inner product immediately implies that the one-dimensional and
  higher-dimensional parts of the expansion
  \eqref{eq:abelian-shadow-defect-sum} are orthogonal, and gives
  \eqref{eq:abelian-shadow-defect-energy}.

  If \(\mathcal B_A\) factors through \(G_{\mathrm{ab}}\), then it
  belongs to the subspace spanned by the pullbacks of the irreducible
  characters of \(G_{\mathrm{ab}}\), that is, by the one-dimensional
  characters of \(G\). Hence all higher-dimensional Fourier coefficients
  in \eqref{eq:normalised-primitive-class-profile} vanish, which proves
  \((i)\Rightarrow(iii)\). The implication \((iii)\Rightarrow(ii)\) is
  immediate from \eqref{eq:abelian-shadow-defect}. Finally,
  \((ii)\Rightarrow(i)\) because a class function whose irreducible
  expansion involves only one-dimensional characters factors through the
  abelianisation.
\end{proof}

For later use, write \(\mathscr C_{\mathrm{cyc}}(G)\) for the linear
span of the pullbacks \(q^\ast f\), where \(q:G\twoheadrightarrow \ZZ/m\ZZ\)
ranges over all cyclic quotients and \(f\) ranges over the class
functions on \(\ZZ/m\ZZ\).

\begin{lemma}[Cyclic quotients and one-dimensional characters]
  \label{lem:cyclic-quotients-one-dimensional-characters}
  One has
  \[
    \mathscr C_{\mathrm{cyc}}(G)
    =
    \operatorname{span}\{\chi\in\Irr(G):\dim\chi=1\}.
  \]
\end{lemma}

\begin{proof}
  Let \(q:G\twoheadrightarrow \ZZ/m\ZZ\) be a cyclic quotient and let
  \(f\) be a class function on \(\ZZ/m\ZZ\). Since \(\ZZ/m\ZZ\) is
  abelian, its irreducible characters form a basis of its class
  functions, so
  \[
    f=\sum_{\psi\in\widehat{\ZZ/m\ZZ}}\widehat f(\psi)\psi.
  \]
  Pulling back gives
  \[
    q^\ast f
    =
    \sum_{\psi\in\widehat{\ZZ/m\ZZ}}
      \widehat f(\psi)(\psi\circ q),
  \]
  and each \(\psi\circ q\) is a one-dimensional character of \(G\).
  Hence
  \[
    \mathscr C_{\mathrm{cyc}}(G)
    \subset
    \operatorname{span}\{\chi\in\Irr(G):\dim\chi=1\}.
  \]

  Conversely, let \(\chi\in\Irr(G)\) be one-dimensional. Its image
  \(\chi(G)\subset\CC^\times\) is a finite subgroup of the multiplicative
  group of complex numbers, hence cyclic. Let
  \(q_\chi:G\twoheadrightarrow \chi(G)\) be the induced quotient map,
  and let \(\iota_\chi:\chi(G)\hookrightarrow \CC^\times\) be the
  tautological character. Then
  \[
    \chi=q_\chi^\ast\iota_\chi,
  \]
  so \(\chi\in\mathscr C_{\mathrm{cyc}}(G)\). This proves the reverse
  inclusion.
\end{proof}

\begin{proposition}[Exact boundary of cyclic detection]
  \label{thm:cyclic-detection-boundary}
  One has
  \[
    \mathscr C_{\mathrm{cyc}}(G)
    =
    \operatorname{span}\{\chi\in\Irr(G):\dim\chi=1\}.
  \]
  If \(P_{\mathrm{cyc}}\) denotes the orthogonal projection onto
  \(\mathscr C_{\mathrm{cyc}}(G)\), then
  \[
    P_{\mathrm{cyc}}\mathcal B_A=\mathcal B_A^{\mathrm{ab}},
    \qquad
    (I-P_{\mathrm{cyc}})\mathcal B_A=\mathcal B_A^{\mathrm{na}}.
  \]
  Equivalently, for every cyclic quotient \(q:G\twoheadrightarrow \ZZ/m\ZZ\)
  and every class function \(f\) on \(\ZZ/m\ZZ\),
  \[
    \langle \mathcal B_A^{\mathrm{na}},q^\ast f\rangle_G=0.
  \]
  In particular, the full primitive class profile is generated by
  cyclic quotient data if and only if
  \(\mathcal B_A^{\mathrm{na}}\equiv 0\).
\end{proposition}

\begin{proof}
  Set
  \[
    V_{\mathrm{ab}}(G)
    :=
    \operatorname{span}\{\chi\in\Irr(G):\dim\chi=1\}.
  \]
  \Cref{lem:cyclic-quotients-one-dimensional-characters} gives
  \(\mathscr C_{\mathrm{cyc}}(G)=V_{\mathrm{ab}}(G)\).
  \Cref{thm:abelian-shadow-defect} identifies
  \(\mathcal B_A^{\mathrm{ab}}\) and \(\mathcal B_A^{\mathrm{na}}\) as
  the orthogonal projections of \(\mathcal B_A\) onto the
  one-dimensional and higher-dimensional Peter--Weyl sectors. Since the
  first sector is exactly \(\mathscr C_{\mathrm{cyc}}(G)\), the
  projection identities and orthogonality follow immediately.
\end{proof}
