\section{Operator framework and Fredholm realisation}
\label{sec:operator}

The determinant identities of
\Cref{prop:prelim-trace-primitive} admit
an operator-theoretic reformulation that is useful for two reasons.
First, it isolates the formal trace--primitive--Euler mechanism from
any particular symbolic model. Second, it shows exactly where
finite-state rigidity enters: trace-class realisability is much more
flexible than finite-state realisability, and the difference is
measured by the integrality and divisibility constraints of the Witt
coordinates.

\subsection{The Fredholm--Witt bridge}

Let $H$ be a complex Hilbert space and let $T\in\mathcal S_1(H)$ be a
trace-class operator. Define
\[
  a_n(T):=\Tr(T^n),
  \qquad
  p_n(T):=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(T)
  \qquad (n\ge 1).
\]

\begin{theorem}[Fredholm--Witt bridge]
  \label{thm:fredholm-witt-bridge}
  For every trace-class operator $T$ and every $\abs{z}<\norm{T}^{-1}$
  one has
  \[
    \det(I-zT)^{-1}
    =\exp\!\left(\sum_{n\ge 1}\frac{a_n(T)}{n}z^n\right)
    =\prod_{n\ge 1}(1-z^n)^{-p_n(T)}.
  \]
  Equivalently, the trace sequence, the primitive Witt coordinates,
  and the Euler product form a strictly commutative diagram.
\end{theorem}

\begin{proof}
  The Fredholm determinant identity
  \[
    \det(I-zT)^{-1}
    =\exp\!\left(\sum_{n\ge 1}\frac{\Tr(T^n)}{n}z^n\right)
  \]
  is standard for trace-class operators. By definition of $p_n(T)$,
  M\"obius inversion gives
  \[
    a_n(T)=\sum_{d\mid n} d\,p_d(T).
  \]
  Hence
  \begin{align*}
    \sum_{n\ge 1}\frac{a_n(T)}{n}z^n
    &=\sum_{n\ge 1}\sum_{d\mid n}\frac{d\,p_d(T)}{n}z^n \\
    &=\sum_{m\ge 1}\sum_{r\ge 1}\frac{p_m(T)}{r}z^{mr}
      =-\sum_{m\ge 1}p_m(T)\log(1-z^m),
  \end{align*}
  and exponentiation yields the Euler product.
\end{proof}

\begin{remark}[Arithmetic restrictions in the finite-state case]
  \label{rem:witt-audit-constraints}
  \Cref{thm:fredholm-witt-bridge} is purely operator-theoretic. It does
  not force the coordinates $p_n(T)$ to be integers, nor even to be
  non-negative. In contrast, for finite-state weighted kernels the
  Witt coordinates are integral polynomials in the weight parameter and
  satisfy a necklace divisibility law. These arithmetic
  constraints are therefore specific to the finite-state setting and do
  not appear for general trace-class determinants.
\end{remark}

\subsection{A converse realisation by cyclic blocks}

For $n\ge 1$, let $\Pi_n$ denote the cyclic permutation matrix on
$\CC^n$. For $\alpha\in\CC$ define the cyclic block
\[
  C(n,\alpha):=\alpha \Pi_n.
\]
Since the eigenvalues of $\Pi_n$ are the $n$th roots of unity,
\begin{equation}
  \label{eq:cycle-block-det}
  \det(I-zC(n,\alpha))=1-\alpha^n z^n.
\end{equation}

\begin{theorem}[Cyclic-block Fredholm realisation]
  \label{thm:cyclic-fredholm-realisation}
  See \cite[Theorem~3.2]{MaZhang2026NormalTraceClassRealisations}.
\end{theorem}

\begin{remark}
  \Cref{thm:cyclic-fredholm-realisation} gives a converse to
  the Euler expansion of \Cref{thm:fredholm-witt-bridge}, but only in
  the trace-class category. The resulting operator need not come from a
  finite graph or a subshift of finite type. In particular, the
  construction preserves the Euler product while discarding the
  combinatorial constraints discussed in
  \Cref{rem:witt-audit-constraints}.
\end{remark}

\subsection{Spectral-gap limit laws}

Let $(X_A^+,\sigma)$ be the mixing subshift determined by a primitive
matrix $A$, let $\mu$ be its Parry measure, and let
$f:X_A^+\to\RR$ be a locally constant observable. For $\theta\in\CC$
near $0$, define the twisted transfer operator
\[
  (\mathcal L_\theta g)(x)
  :=\sum_{\sigma y=x} e^{\theta f(y)}g(y).
\]

\begin{theorem}[Spectral-gap central limit theorem]
  \label{thm:clt-spectral}
  Assume that $f$ is not a coboundary modulo constants. Then there
  exists $\sigma_f^2>0$ such that
  \[
    \frac{\sum_{k=0}^{n-1}f\circ\sigma^k
      -n\int f\,\dd\mu}{\sqrt n}
    \Longrightarrow \mathcal N(0,\sigma_f^2).
  \]
  Moreover, for every compact set $K\Subset\RR\setminus\{0\}$ there
  exist constants $C_K>0$ and $\vartheta_K\in(0,1)$ such that
  \[
    \sup_{t\in K}
    \abs{\int e^{it\sum_{k=0}^{n-1}f\circ\sigma^k}\,\dd\mu}
    \le C_K\vartheta_K^n
  \]
  for all $n\ge 1$.
\end{theorem}

\begin{proof}
  Because $f$ is locally constant, one may work on a
  finite-dimensional space of cylinder functions of sufficiently large
  range on which the operators $\mathcal L_\theta$ act analytically.
  Standard perturbation theory therefore yields a simple dominant
  eigenvalue $\lambda(\theta)$ and a spectral gap uniform for $\theta$
  near $0$. The logarithm
  $\log\lambda(\theta)$ is the cumulant generating function of the
  Birkhoff sums. Since $f$ is not a coboundary modulo constants, the
  variance
  \[
    \sigma_f^2=\lambda''(0)/\lambda(0)
  \]
  is strictly positive. The central limit theorem follows from the
  Nagaev--Guivarc'h argument, while the uniform spectral gap for
  $\theta=it$ with $t\in K$ gives the exponential bound for the
  characteristic functions.
\end{proof}

\begin{remark}
  The same perturbative mechanism governs both
  \Cref{thm:clt-spectral} and the holomorphic variation theorem for the
  finite-part constant in \Cref{thm:logM-holomorphic-variation}. In
  both cases the key input is the simplicity of the Perron eigenvalue
  and analytic control of the reduced resolvent.
\end{remark}
