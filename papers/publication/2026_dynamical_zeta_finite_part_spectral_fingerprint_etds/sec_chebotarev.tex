\section{Finite-group extensions and Adams--M\"obius inversion}
\label{sec:chebotarev}

In this section we study finite-group extensions. For cyclic extensions
the relevant channels are one-dimensional, and the discussion of
\Cref{sec:finite-part} reduces to Fourier analysis on roots of unity.
For a general finite group \(G\), the same passage from periodic data
to primitive data takes place through the full Peter--Weyl
decomposition. The periodic determinant factorizations and the leading
Chebotarev asymptotics for shifts of finite type are classical; see
Parry--Pollicott \cite{ParryPollicott1986Chebotarev},
Noorani--Parry \cite{NooraniParry1992ChebotarevShifts}, and also
\cite{PollicottSharp2007Chebotarev,BoyleSchmieding2017FiniteGroupExtensions,Parry1999LivsicNonAbelian}.
On the graph side, compare
\cite{KotaniSunada2000ZetaFiniteGraphs,StarkTerras1996ZetaGraphCoverings,StarkTerras2000ZetaGraphCoveringsII,TerrasStark2007ZetaGraphCoveringsIII,HasegawaSaito2016GraphMertens,HasegawaSaitoSato2015AbelianCovering,HasegawaSaitoSato2018GaloisCovering}.
Our concern here is the primitive level.

The new feature is that ordinary M\"obius inversion, applied to a single
twisted logarithm, does not remove repetitions of the group label. The
necessary correction is supplied by the Adams operations \(g\mapsto g^m\)
on class functions. With this additional input, the twisted determinant
logarithms recover the primitive irreducible channels directly. We work
throughout in the finite-dimensional twisted-matrix setting, using the
standard representation-theoretic formalism of Serre
\cite{Serre1977LinearRepresentations}.

\subsection{Class functions and primitive inversion}

For a conjugacy class \(C\subset G\), we write \(\chi_\varrho(C)\) for
the common value of \(\chi_\varrho\) on \(C\), and \(\mathbf 1_C\) for
the indicator function of \(C\). If \(x\in\operatorname{Fix}(\sigma^n)\)
corresponds to the closed edge path \(e_0\cdots e_{n-1}\), define
\[
  \tau_n(x):=\tau(e_0)\cdots\tau(e_{n-1})\in G.
\]
For a class function \(\phi:G\to\CC\), set
\begin{equation}\label{eq:class-logarithm-def}
  T_{\phi,n}:=\sum_{x\in\operatorname{Fix}(\sigma^n)}\phi(\tau_n(x)),
  \qquad
  \mathcal L_\phi(z):=\sum_{n\ge 1}\frac{T_{\phi,n}}{n}z^n,
\end{equation}
and
\begin{equation}\label{eq:primitive-class-series-def}
  \Pi_\phi(z):=\sum_{\gamma\in\cP}\phi(\tau(\gamma))z^{|\gamma|}.
\end{equation}
Since \(\phi\) is bounded and \(p_n(A)\ll \lambda^n/n\), the series
\(\Pi_\phi(z)\) converges absolutely for \(|z|<\lambda^{-1}\).

\begin{proposition}[Conjugacy-class expansion]
  \label{prop:class-indicator-expansion}
  For every conjugacy class \(C\subset G\) and every \(g\in G\),
  \[
    \mathbf 1_C(g)
    =\frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \overline{\chi_\varrho(C)}\,\chi_\varrho(g)
    =\frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho(C^{-1})\,\chi_\varrho(g).
  \]
\end{proposition}

\begin{proof}
  The irreducible characters form an orthonormal basis of the class
  functions on \(G\) for the inner product
  \[
    \langle f,h\rangle
    :=\frac{1}{|G|}\sum_{g\in G}f(g)\overline{h(g)}.
  \]
  Hence
  \[
    \mathbf 1_C
    =\sum_{\varrho\in\Irr(G)}
      \langle \mathbf 1_C,\chi_\varrho\rangle \chi_\varrho.
  \]
  Since \(\chi_\varrho\) is constant on \(C\),
  \[
    \langle \mathbf 1_C,\chi_\varrho\rangle
    =\frac{1}{|G|}\sum_{g\in C}\overline{\chi_\varrho(g)}
    =\frac{|C|}{|G|}\overline{\chi_\varrho(C)}
    =\frac{|C|}{|G|}\chi_\varrho(C^{-1}),
  \]
  which gives the claim.
\end{proof}

\begin{proposition}[Peter--Weyl determinant factorisation]
  \label{prop:kernel-peter-weyl-block-diagonalisation}
  Let \(\widetilde A\) be the adjacency matrix of the skew-product
  extension attached to \(\tau\). Then
  \[
    \det(I-z\widetilde A)
    =
    \prod_{\varrho\in\Irr(G)}
      \det(I-zA_\varrho)^{\dim\varrho}.
  \]
\end{proposition}

\begin{proof}
  The regular representation decomposes as
  \[
    \CC[G]
    \cong
    \bigoplus_{\varrho\in\Irr(G)}
      V_\varrho^{\oplus \dim\varrho}.
  \]
  Under this decomposition, \(\widetilde A\) becomes block diagonal with
  blocks \(A_\varrho\otimes I_{\dim\varrho}\). Taking determinants gives
  the stated factorisation.
\end{proof}

\begin{proposition}[Twisted spectral-radius bound]
  \label{prop:twisted-spectral-radius-bound}
  For every irreducible unitary representation
  \(\sigma\in\Irr(G)\),
  \[
    \rho(A_\sigma)\le \rho(\widetilde A)=\lambda.
  \]
\end{proposition}

\begin{proof}
  By \Cref{prop:kernel-peter-weyl-block-diagonalisation},
  \[
    \det(tI-\widetilde A)
    =
    \prod_{\varrho\in\Irr(G)}
      \det(tI-A_\varrho)^{\dim\varrho}.
  \]
  Every eigenvalue \(\alpha\) of \(A_\sigma\), including \(\alpha=0\),
  is a root of \(\det(tI-A_\sigma)\) and hence of the right-hand
  side. Therefore \(\alpha\in\spec(\widetilde A)\), and consequently
  \(\rho(A_\sigma)\le \rho(\widetilde A)\).

  To compute \(\rho(\widetilde A)\), let
  \(N_n(B):=\mathbf 1^{\mathsf T}B^n\mathbf 1\), the number of
  admissible length-\(n\) paths in the graph with adjacency matrix
  \(B\). Choosing the initial group element gives a bijection between
  admissible length-\(n\) paths in the extension and pairs
  \((g,\gamma)\), where \(g\in G\) and \(\gamma\) is an admissible
  length-\(n\) path in the base graph. Hence
  \(N_n(\widetilde A)=|G|\,N_n(A)\) for every \(n\ge 1\).

  For any \(d\times d\) non-negative matrix \(B\), one has
  \[
    \|B^n\|_1\le N_n(B)\le d\,\|B^n\|_1,
  \]
  so Gelfand's formula implies
  \[
    \rho(B)=\lim_{n\to\infty}N_n(B)^{1/n}.
  \]
  Applying this to \(A\) and \(\widetilde A\) yields
  \[
    \rho(\widetilde A)=\rho(A)=\lambda.
  \]
\end{proof}

The next proposition identifies the passage from periodic determinant
data to primitive orbit data.

\begin{proposition}[Class-function linearisation]
  \label{thm:class-function-linearisation}
  Let \(\phi:G\to\CC\) be a class function and write
  \[
    \phi=\sum_{\varrho\in\Irr(G)}\widehat\phi(\varrho)\chi_\varrho
  \]
  as in \eqref{eq:class-function-fourier}. Then, for \(|z|<\lambda^{-1}\),
  \begin{equation}\label{eq:class-logarithm-determinant-linearisation}
    \mathcal L_\phi(z)
    =
    \sum_{\varrho\in\Irr(G)}
      \widehat\phi(\varrho)\,
      \log\det(I-zA_\varrho)^{-1}.
  \end{equation}
\end{proposition}

\begin{proof}
  For \(n\ge 1\), expanding \(\phi\) in the irreducible character basis
  gives
  \[
    T_{\phi,n}
    =\sum_{x\in\operatorname{Fix}(\sigma^n)}
      \sum_{\varrho\in\Irr(G)}
        \widehat\phi(\varrho)\chi_\varrho(\tau_n(x))
    =\sum_{\varrho\in\Irr(G)}
      \widehat\phi(\varrho)\Tr(A_\varrho^n),
  \]
  because \(\Tr(A_\varrho^n)\) is the sum of \(\chi_\varrho\) over all
  period-\(n\) points. Therefore,
  \[
    \mathcal L_\phi(z)
    =\sum_{n\ge 1}\frac{z^n}{n}
      \sum_{\varrho\in\Irr(G)}
        \widehat\phi(\varrho)\Tr(A_\varrho^n).
  \]
  Moreover,
  \[
    \sum_{n\ge 1}\frac{|z|^n}{n}\abs{T_{\phi,n}}
    \le
    \norm{\phi}_\infty
    \sum_{n\ge 1}\frac{\Tr(A^n)}{n}|z|^n
    =
    \norm{\phi}_\infty \log\det(I-|z|A)^{-1}
    <\infty
  \]
  for \(|z|<\lambda^{-1}\). Since
  \[
    \abs{T_{\phi,n}}
    \le \norm{\phi}_\infty \#\operatorname{Fix}(\sigma^n)
    =\norm{\phi}_\infty \Tr(A^n),
  \]
  the series is absolutely convergent for \(|z|<\lambda^{-1}\), and we
  may exchange the order of summation to obtain
  \[
    \mathcal L_\phi(z)
    =
    \sum_{\varrho\in\Irr(G)}
      \widehat\phi(\varrho)
      \sum_{n\ge 1}\frac{\Tr(A_\varrho^n)}{n}z^n
    =
    \sum_{\varrho\in\Irr(G)}
      \widehat\phi(\varrho)\,
      \log\det(I-zA_\varrho)^{-1}.
  \]
  This proves \eqref{eq:class-logarithm-determinant-linearisation}.
\end{proof}

\begin{corollary}[Adams--M\"obius primitive inversion]
  \label{cor:class-function-adams-mobius}
  For every class function \(\phi:G\to\CC\) and every \(|z|<\lambda^{-1}\),
  \begin{equation}\label{eq:class-adams-mobius-inversion}
    \Pi_\phi(z)
    =
    \sum_{m\ge 1}\frac{\mu(m)}{m}\,
      \mathcal L_{\psi^m\phi}(z^m).
  \end{equation}
\end{corollary}

\begin{proof}
  For \(\gamma\in\cP\) with length \(\ell=|\gamma|\), the \(r\)-fold
  repetition of \(\gamma\) contributes \(\ell\) points to
  \(\operatorname{Fix}(\sigma^{r\ell})\), and each such point has group
  label conjugate to \(\tau(\gamma)^r\). Since \(\phi\) is a class
  function,
  \[
    \mathcal L_\phi(z)
    =
    \sum_{\gamma\in\cP}\sum_{r\ge 1}
      \frac{\phi(\tau(\gamma)^r)}{r}z^{r|\gamma|}.
  \]
  Hence
  \begin{align*}
    \sum_{m\ge 1}\frac{\mu(m)}{m}\mathcal L_{\psi^m\phi}(z^m)
    &=
    \sum_{m\ge 1}\frac{\mu(m)}{m}
      \sum_{\gamma\in\cP}\sum_{r\ge 1}
        \frac{\phi(\tau(\gamma)^{mr})}{r}z^{mr|\gamma|} \\
    &=
    \sum_{\gamma\in\cP}\sum_{s\ge 1}
      \frac{\phi(\tau(\gamma)^s)}{s}z^{s|\gamma|}
      \sum_{m\mid s}\mu(m),
  \end{align*}
  where \(s=mr\). Since \(\sum_{m\mid s}\mu(m)=1\) for \(s=1\) and
  \(0\) otherwise, only the primitive term survives, yielding
  \[
    \sum_{m\ge 1}\frac{\mu(m)}{m}\mathcal L_{\psi^m\phi}(z^m)
    =
    \sum_{\gamma\in\cP}\phi(\tau(\gamma))z^{|\gamma|}
    =\Pi_\phi(z).
  \]
  This proves \eqref{eq:class-adams-mobius-inversion}.
\end{proof}

\begin{proposition}[Direct M\"obius inversion does not suffice]
  \label{prop:naive-mobius-contamination}
  For \(\varrho\in\Irr(G)\), define the direct trace-level M\"obius
  transform
  \[
    \mathcal N_\varrho(z)
    :=\sum_{m\ge 1}\frac{\mu(m)}{m}\,
      \log\det(I-z^mA_\varrho)^{-1}.
  \]
  Then, for \(|z|<\lambda^{-1}\),
  \[
    \mathcal N_\varrho(z)
    =
    \Pi_\varrho(z)
    +\sum_{\gamma\in\cP}\sum_{s\ge 2}\frac{z^{s|\gamma|}}{s}
      \sum_{m\mid s}\mu(m)\,
        \chi_\varrho(\tau(\gamma)^{s/m}).
  \]
  Consequently, \(\mathcal N_\varrho=\Pi_\varrho\) holds for a fixed
  \(\varrho\) if and only if
  \[
    \sum_{m\mid s}\mu(m)\chi_\varrho(g^{s/m})=0
    \qquad
    (g\in G,\ s\ge 2).
  \]
  In general this condition fails: if \(\chi\) is a non-trivial
  quadratic character and \(\chi(g)=-1\), then the \(s=2\) coefficient
  is
  \[
    \chi(g^2)-\chi(g)=1-(-1)=2.
  \]
\end{proposition}

\begin{proof}
  Since
  \[
    \log\det(I-zA_\varrho)^{-1}
    =\mathcal L_{\chi_\varrho}(z)
    =\sum_{\gamma\in\cP}\sum_{r\ge 1}
      \frac{\chi_\varrho(\tau(\gamma)^r)}{r}z^{r|\gamma|},
  \]
  one has
  \begin{align*}
    \mathcal N_\varrho(z)
    &=
    \sum_{m\ge 1}\frac{\mu(m)}{m}
      \sum_{\gamma\in\cP}\sum_{r\ge 1}
        \frac{\chi_\varrho(\tau(\gamma)^r)}{r}z^{mr|\gamma|} \\
    &=
    \sum_{\gamma\in\cP}\sum_{s\ge 1}\frac{z^{s|\gamma|}}{s}
      \sum_{m\mid s}\mu(m)\,
        \chi_\varrho(\tau(\gamma)^{s/m}),
  \end{align*}
  where \(s=mr\). The \(s=1\) term is exactly \(\Pi_\varrho(z)\), and
  the remaining terms give the displayed correction. The criterion for
  equality is immediate. For the quadratic example, the \(s=2\)
  coefficient equals
  \(\chi(g^2)-\chi(g)=\chi(g)^2-\chi(g)=1-\chi(g)\), which is \(2\)
  when \(\chi(g)=-1\).
\end{proof}

The formulas above belong to the classical twisted
\(L\)-function and Chebotarev framework of
\cite{AdachiSunada1987TwistedPerronFrobenius,Sunada1989DynamicalLFunctions,ParryPollicott1986Chebotarev,KotaniSunada2000ZetaFiniteGraphs,StarkTerras1996ZetaGraphCoverings,StarkTerras2000ZetaGraphCoveringsII,TerrasStark2007ZetaGraphCoveringsIII}.
For primitive orbit questions one needs formulas in which repeated
labels have already been removed. The preceding proposition explains
why a direct M\"obius transform of a single twisted logarithm is not
sufficient, and leads to the Adams-corrected formulas below.

\subsection{Irreducible channels}

Applying \Cref{cor:class-function-adams-mobius} to irreducible
characters produces the primitive Peter--Weyl coefficients directly
from the twisted determinants.

For \(\varrho\in\Irr(G)\), define
\begin{equation}\label{eq:primitive-peter-weyl-series}
  \Pi_\varrho(z):=\Pi_{\chi_\varrho}(z)
  =\sum_{n\ge 1}\pi_{n,\varrho}(A)z^n,
  \qquad
  \pi_{n,\varrho}(A)
  :=\sum_{\substack{\gamma\in\cP\\ |\gamma|=n}}
    \chi_\varrho(\tau(\gamma)).
\end{equation}
For \(m\ge 1\), write
\begin{equation}\label{eq:adams-character-decomposition}
  \psi^m\chi_\varrho
  =\sum_{\sigma\in\Irr(G)}a_{\varrho,\sigma}^{(m)}\chi_\sigma
\end{equation}
for the Adams decomposition in the character ring.

\begin{lemma}\label{lem:adams-coefficients-bounded}
  For every \(\varrho,\sigma\in\Irr(G)\) and every \(m\ge 1\),
  \[
    \abs{a_{\varrho,\sigma}^{(m)}}\le \dim\varrho.
  \]
\end{lemma}

\begin{proof}
  Since the irreducible characters are orthonormal,
  \[
    a_{\varrho,\sigma}^{(m)}
    =\langle \psi^m\chi_\varrho,\chi_\sigma\rangle.
  \]
  Hence
  \[
    \abs{a_{\varrho,\sigma}^{(m)}}
    \le \norm{\psi^m\chi_\varrho}_2\norm{\chi_\sigma}_2
    =\norm{\psi^m\chi_\varrho}_2.
  \]
  Moreover,
  \[
    \norm{\psi^m\chi_\varrho}_2^2
    =\frac{1}{|G|}\sum_{g\in G}\abs{\chi_\varrho(g^m)}^2
    \le (\dim\varrho)^2,
  \]
  because \(\abs{\chi_\varrho(h)}\le \dim\varrho\) for all \(h\in G\).
  Taking the square root gives the claim.
\end{proof}

\begin{remark}[Periodicity of Adams coefficients]
  \label{rem:adams-coefficients-periodic}
  Let \(\exp(G)\) denote the exponent of \(G\). Since
  \(g^{m+\exp(G)}=g^m\) for every \(g\in G\), one has
  \[
    \psi^{m+\exp(G)}\chi_\varrho=\psi^m\chi_\varrho.
  \]
  Consequently
  \(a_{\varrho,\sigma}^{(m+\exp(G))}=a_{\varrho,\sigma}^{(m)}\) for all
  \(m\), so only finitely many Adams coefficient vectors occur. In the
  estimates below we use this finiteness only through the uniform bound
  of \Cref{lem:adams-coefficients-bounded}.
\end{remark}

\begin{corollary}[Irreducible-channel determinant formula]
  \label{cor:primitive-peter-weyl-determinant}
  For every \(\varrho\in\Irr(G)\) and every \(|z|<\lambda^{-1}\),
  \begin{equation}\label{eq:primitive-peter-weyl-determinant}
    \Pi_\varrho(z)
    =
    \sum_{m\ge 1}\frac{\mu(m)}{m}
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}
        \log\det(I-z^mA_\sigma)^{-1}.
  \end{equation}
\end{corollary}

\begin{proof}
  Apply \Cref{cor:class-function-adams-mobius} with
  \(\phi=\chi_\varrho\). Then
  \[
    \Pi_\varrho(z)
    =
    \sum_{m\ge 1}\frac{\mu(m)}{m}
      \mathcal L_{\psi^m\chi_\varrho}(z^m).
  \]
  By \eqref{eq:adams-character-decomposition} and the determinant
  linearisation \eqref{eq:class-logarithm-determinant-linearisation},
  \[
    \mathcal L_{\psi^m\chi_\varrho}(z^m)
    =
    \sum_{\sigma\in\Irr(G)}
      a_{\varrho,\sigma}^{(m)}
      \log\det(I-z^mA_\sigma)^{-1}.
  \]
  Substituting this identity proves
  \eqref{eq:primitive-peter-weyl-determinant}.
\end{proof}

\begin{remark}[The abelian case]
  \label{rem:abelian-toy-model-cyclic}
  Let \(G=\ZZ/q\ZZ\), and let \(\chi_j(h)=\omega_q^{jh}\) be its
  irreducible characters. For the canonical cyclic-lift cocycle one has
  \(A_{\chi_j}=\omega_q^jA\), and therefore
  \[
    \mathcal L_{\chi_j}(z)=\log\det(I-\omega_q^j zA)^{-1}.
  \]
  The Adams--M\"obius inversion
  \eqref{eq:class-adams-mobius-inversion} then reduces to the ordinary
  Fourier decomposition on the \(q\)-th roots of unity. This is the
  bridge between the abelian formulas of
  \Cref{sec:finite-part} and the present finite-group formalism. In
  particular, for every non-trivial \(\chi_j\) one has
  \(\rho(A_{\chi_j})=\rho(\omega_q^jA)=\lambda\). Thus canonical cyclic
  lifts do \emph{not} satisfy the strict inequality \(\eta<\lambda\),
  and the Perron-point formulas proved later in this section do not
  specialise directly to \Cref{sec:finite-part}.
\end{remark}

\begin{corollary}[Primitive Peter--Weyl trace formula]
  \label{cor:primitive-peter-weyl-trace}
  For every \(\varrho\in\Irr(G)\) and every \(n\ge 1\),
  \begin{equation}\label{eq:primitive-peter-weyl-trace}
    \pi_{n,\varrho}(A)
    =
    \frac{1}{n}\sum_{m\mid n}\mu(m)
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}
        \Tr(A_\sigma^{n/m}).
  \end{equation}
\end{corollary}

\begin{proof}
  Extract the coefficient of \(z^n\) in
  \eqref{eq:primitive-peter-weyl-determinant}. Since
  \[
    [z^n]\log\det(I-z^mA_\sigma)^{-1}
    =
    \begin{cases}
      \Tr(A_\sigma^{n/m})/(n/m), & m\mid n, \\
      0, & m\nmid n,
    \end{cases}
  \]
  the stated formula follows immediately.
\end{proof}

\subsection{Conjugacy classes and Mertens constants}

The primitive Peter--Weyl coefficients determine the conjugacy-class
counts by Fourier inversion on \(G\).

\begin{proposition}[Conjugacy-class primitive decomposition]
  \label{thm:class-primitive-decomposition}
  Let \(C\subset G\) be a conjugacy class. If \(p_{n,C}\) denotes the
  number of primitive orbits of length \(n\) whose \(G\)-label lies in
  \(C\), then for \(|z|<\lambda^{-1}\),
  \begin{equation}\label{eq:class-primitive-generating-function}
    \sum_{n\ge 1}p_{n,C}z^n
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho(C^{-1})\,\Pi_\varrho(z),
  \end{equation}
  and therefore
  \begin{equation}\label{eq:class-primitive-coefficients}
    p_{n,C}
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho(C^{-1})\,\pi_{n,\varrho}(A).
  \end{equation}
\end{proposition}

\begin{proof}
  By definition,
  \[
    \sum_{n\ge 1}p_{n,C}z^n
    =
    \sum_{\gamma\in\cP}\mathbf 1_C(\tau(\gamma))z^{|\gamma|}.
  \]
  Applying \Cref{prop:class-indicator-expansion} to the class function
  \(\mathbf 1_C\) and using the linearity of \(\Pi_\phi\) yields
  \eqref{eq:class-primitive-generating-function}. Comparing coefficients
  gives \eqref{eq:class-primitive-coefficients}.
\end{proof}

\begin{corollary}[Conjugacy-class Artin--M\"obius trace formula]
  \label{cor:class-artin-mobius-trace}
  For every conjugacy class \(C\subset G\) and every \(n\ge 1\),
  \begin{equation}\label{eq:class-artin-mobius-trace}
    p_{n,C}
    =
    \frac{|C|}{|G|\,n}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho(C^{-1})
        \sum_{m\mid n}\mu(m)
          \sum_{\sigma\in\Irr(G)}
            a_{\varrho,\sigma}^{(m)}
            \Tr(A_\sigma^{n/m}).
  \end{equation}
\end{corollary}

\begin{proof}
  Substitute \eqref{eq:primitive-peter-weyl-trace} into
  \eqref{eq:class-primitive-coefficients}.
\end{proof}

From this point on we assume the twisted spectral-gap hypothesis
\[
  \eta:=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda.
\]
This condition is used only to justify evaluation at the Perron point in
the non-trivial Peter--Weyl channels. We begin by separating the trivial
channel and estimating the non-trivial ones.

\begin{lemma}[Separation of the trivial channel]
  \label{lem:class-primitive-trivial-split}
  For every conjugacy class \(C\subset G\) and every \(n\ge 1\),
  \[
    p_{n,C}
    =
    \frac{|C|}{|G|}p_n(A)
    +\frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\,\pi_{n,\varrho}(A).
  \]
\end{lemma}

\begin{proof}
  Separate the trivial representation from
  \eqref{eq:class-primitive-coefficients}. Since
  \(\chi_{\mathbf 1}(C^{-1})=1\) and
  \(\pi_{n,\mathbf 1}(A)=p_n(A)\), the displayed identity follows.
\end{proof}

\begin{lemma}[Bounds for the non-trivial channels]
  \label{lem:nontrivial-peter-weyl-channel-bound}
  For every non-trivial \(\varrho\in\Irr(G)\),
  \[
    \pi_{n,\varrho}(A)
    =O\!\left(\frac{\max\{\eta^n,\lambda^{n/2}\}}{n}\right).
  \]
\end{lemma}

\begin{proof}
  Fix \(\varrho\neq\mathbf 1\). Since \(A_\varrho\) is finite
  dimensional and \(\rho(A_\varrho)\le \eta\), there is a constant
  \(K_1(\varrho)\) such that
  \[
    \abs{\Tr(A_\varrho^n)}\le K_1(\varrho)\eta^n
    \qquad (n\ge 1).
  \]
  By \Cref{prop:twisted-spectral-radius-bound} and the finiteness of
  \(\Irr(G)\), there is a constant \(K_2\) such that
  \[
    \abs{\Tr(A_\sigma^\ell)}\le K_2\lambda^\ell
    \qquad
    (\sigma\in\Irr(G),\ \ell\ge 1).
  \]
  Now \Cref{cor:primitive-peter-weyl-trace} gives
  \[
    \pi_{n,\varrho}(A)
    =
    \frac{1}{n}\Tr(A_\varrho^n)
    +\frac{1}{n}
      \sum_{\substack{m\mid n\\ m>1}}\mu(m)
        \sum_{\sigma\in\Irr(G)}
          a_{\varrho,\sigma}^{(m)}\Tr(A_\sigma^{n/m}).
  \]
  The first term is \(O(\eta^n/n)\). For the divisor sum, use
  \Cref{lem:adams-coefficients-bounded}:
  \[
    \abs{
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}\Tr(A_\sigma^{n/m})
    }
    \le
    K_2\,\dim\varrho\,\abs{\Irr(G)}\,\lambda^{n/m}.
  \]
  Since \(n/m\le n/2\) for \(m>1\),
  \[
    \sum_{\substack{m\mid n\\ m>1}}\lambda^{n/m}
    \le \sum_{k=1}^{\lfloor n/2\rfloor}\lambda^k
    =O(\lambda^{n/2}).
  \]
  Combining the two bounds proves the claim.
\end{proof}

\begin{lemma}[Absolute convergence at the Perron point]
  \label{lem:nontrivial-perron-point-convergence}
  For every non-trivial \(\varrho\in\Irr(G)\), the series
  \[
    \Pi_\varrho(\lambda^{-1})
    =\sum_{n\ge 1}\pi_{n,\varrho}(A)\lambda^{-n}
  \]
  converges absolutely.
\end{lemma}

\begin{proof}
  Let
  \[
    \vartheta:=\max\{\eta/\lambda,\lambda^{-1/2}\}\in(0,1).
  \]
  By \Cref{lem:nontrivial-peter-weyl-channel-bound},
  \[
    \abs{\pi_{n,\varrho}(A)}\lambda^{-n}
    =O\!\left(\frac{\vartheta^n}{n}\right),
  \]
  and the series \(\sum_{n\ge 1}\vartheta^n/n\) converges.
\end{proof}

\begin{lemma}[Geometric-harmonic tails]
  \label{lem:geometric-harmonic-tail}
  If \(0<\vartheta<1\), then for every \(N\ge 1\),
  \[
    \sum_{n>N}\frac{\vartheta^n}{n}
    \le
    \frac{\vartheta^{N+1}}{(N+1)(1-\vartheta)}.
  \]
  In particular, if \(b_n=O(\vartheta^n/n)\), then
  \[
    \sum_{n>N}b_n=O\!\left(\frac{\vartheta^N}{N}\right).
  \]
\end{lemma}

\begin{proof}
  Since \(n\ge N+1\) throughout the tail,
  \[
    \sum_{n>N}\frac{\vartheta^n}{n}
    \le
    \frac{1}{N+1}\sum_{n>N}\vartheta^n
    =
    \frac{\vartheta^{N+1}}{(N+1)(1-\vartheta)}.
  \]
  The second statement is immediate.
\end{proof}

\begin{lemma}[Contribution of the trivial channel]
  \label{lem:perron-channel-contribution}
  Assume that
  \[
    \Lambda_*:=\max\{\Lambda(A),\eta,\lambda^{1/2}\}.
  \]
  Then \(\vartheta_*:=\Lambda_*/\lambda\in(0,1)\), and
  \[
    \sum_{n\le N}\frac{p_n(A)}{\lambda^n}
    =
    H_N+\mathcal M(A)-\gamma
    +O\!\left(\frac{\vartheta_*^N}{N}\right).
  \]
\end{lemma}

\begin{proof}
  Since \(\Lambda(A)<\lambda\), \(\eta<\lambda\), and
  \(\lambda^{1/2}<\lambda\), one has \(\vartheta_*\in(0,1)\). By
  \Cref{lem:primitive-orbit-asymptotic},
  \[
    \frac{p_n(A)}{\lambda^n}
    =
    \frac{1}{n}
    +O\!\left(\frac{\vartheta_*^n}{n}\right).
  \]
  Hence the correction series
  \[
    \sum_{n\ge 1}\left(\frac{p_n(A)}{\lambda^n}-\frac{1}{n}\right)
  \]
  converges absolutely, and \Cref{cor:finite-part-mertens-asymptotic}
  identifies its value with \(\mathcal M(A)-\gamma\). Therefore
  \[
    \sum_{n\le N}\frac{p_n(A)}{\lambda^n}
    =
    H_N+\mathcal M(A)-\gamma
    -\sum_{n>N}\left(\frac{p_n(A)}{\lambda^n}-\frac{1}{n}\right).
  \]
  The tail is \(O(\vartheta_*^N/N)\) by
  \Cref{lem:geometric-harmonic-tail}.
\end{proof}

\begin{remark}[Historical note]
  \label{rem:chebotarev-known-vs-new}
  The leading Chebotarev asymptotic
  \(p_{n,C}\sim(|C|/|G|)\lambda^n/n\) in the next theorem is known
  from Noorani--Parry \cite{NooraniParry1992ChebotarevShifts} in the
  finite homogeneous SFT setting and from Parry--Pollicott
  \cite{ParryPollicott1986Chebotarev} in the Axiom~A context.
  The periodic determinant factorisations used here are classical as
  well, and closely related Mertens constants for Ihara and Artin--Ihara
  zeta functions of graph coverings, including abelian and finite Galois
  coverings, appear in the work of
  Stark--Terras, Kotani--Sunada, Hasegawa--Saito, and
  Hasegawa--Saito--Sato
  \cite{KotaniSunada2000ZetaFiniteGraphs,StarkTerras1996ZetaGraphCoverings,StarkTerras2000ZetaGraphCoveringsII,TerrasStark2007ZetaGraphCoveringsIII,HasegawaSaito2016GraphMertens,HasegawaSaitoSato2015AbelianCovering,HasegawaSaitoSato2018GaloisCovering}.
  The point of \Cref{thm:class-mertens-explicit} is therefore the
  description of the constant term in terms of the primitive
  Peter--Weyl channels \(\Pi_\varrho(\lambda^{-1})\). In
  \Cref{thm:class-mertens-fourier} these channel values are recovered
  from the classwise constants, and above the square-root barrier
  \Cref{thm:class-mertens-secondary-term} shows that the same channels
  govern the first spectral-edge correction to both the primitive class
  counts and the harmonic remainder.
\end{remark}

\begin{theorem}[Classwise Mertens asymptotics under
  \texorpdfstring{$\eta<\lambda$}{eta<lambda}]
  \label{thm:class-mertens-explicit}
  Assume that
  \[
    \eta:=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda,
    \qquad
    \Lambda_*:=\max\{\Lambda(A),\eta,\lambda^{1/2}\}.
  \]
  Then, for every conjugacy class \(C\subset G\),
  \begin{equation}\label{eq:class-primitive-asymptotic}
    p_{n,C}
    =
    \frac{|C|}{|G|}\frac{\lambda^n}{n}
    +O\!\left(\frac{\Lambda_*^n}{n}\right).
  \end{equation}
  In particular, for all sufficiently large \(n\),
  \begin{equation}\label{eq:class-chebotarev-density}
    \frac{p_{n,C}}{p_n(A)}
    =
    \frac{|C|}{|G|}
    +O\!\left(\left(\frac{\Lambda_*}{\lambda}\right)^n\right).
  \end{equation}
  Moreover, the series \(\Pi_\varrho(\lambda^{-1})\) converges
  absolutely for every non-trivial \(\varrho\), and
  \begin{equation}\label{eq:class-mertens-harmonic}
    \sum_{n\le N}\frac{p_{n,C}}{\lambda^n}
    =
    \frac{|C|}{|G|}H_N
    +\mathcal M_C(A)-\frac{|C|}{|G|}\gamma
    +O\!\left(\frac{(\Lambda_*/\lambda)^N}{N}\right),
  \end{equation}
  where \(H_N=\sum_{n\le N}1/n\) and
  \begin{equation}\label{eq:class-mertens-constant}
    \mathcal M_C(A)
    =
    \frac{|C|}{|G|}\mathcal M(A)
    +\frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\,\Pi_\varrho(\lambda^{-1}).
  \end{equation}
  In particular,
  \begin{equation}\label{eq:class-mertens-log}
    \sum_{n\le N}\frac{p_{n,C}}{\lambda^n}
    =
    \frac{|C|}{|G|}\log N+\mathcal M_C(A)+O(N^{-1}).
  \end{equation}
  The implied constants in
  \eqref{eq:class-primitive-asymptotic} and
  \eqref{eq:class-mertens-harmonic} depend on \(A\), \(G\), and
  \(\tau\), but are uniform in the conjugacy class \(C\) and in \(n\)
  or \(N\), respectively.
\end{theorem}

\begin{proof}
  Set \(\vartheta_*:=\Lambda_*/\lambda\in(0,1)\). By
  \Cref{lem:class-primitive-trivial-split},
  \[
    p_{n,C}
    =
    \frac{|C|}{|G|}p_n(A)
    +\frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\,\pi_{n,\varrho}(A).
  \]
  Combining \Cref{lem:primitive-orbit-asymptotic} with
  \Cref{lem:nontrivial-peter-weyl-channel-bound} yields
  \eqref{eq:class-primitive-asymptotic}.
  Since
  \[
    p_n(A)=\frac{\lambda^n}{n}+O\!\left(\frac{\Lambda_*^n}{n}\right),
  \]
  dividing \eqref{eq:class-primitive-asymptotic} by \(p_n(A)\) yields
  \eqref{eq:class-chebotarev-density} for all sufficiently large \(n\).

  By \Cref{lem:nontrivial-perron-point-convergence}, the series
  \(\Pi_\varrho(\lambda^{-1})\) is absolutely convergent for every
  non-trivial \(\varrho\). Summing the identity for
  \(p_{n,C}\lambda^{-n}\) from \(1\) to \(N\), the trivial
  representation contributes
  \[
    \frac{|C|}{|G|}\sum_{n\le N}\frac{p_n(A)}{\lambda^n}
    =
    \frac{|C|}{|G|}H_N
    +\frac{|C|}{|G|}\bigl(\mathcal M(A)-\gamma\bigr)
    +O\!\left(\frac{\vartheta_*^N}{N}\right)
  \]
  by \Cref{lem:perron-channel-contribution}. The non-trivial
  representations contribute
  \[
    \frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}\chi_\varrho(C^{-1})
      \sum_{n\le N}\pi_{n,\varrho}(A)\lambda^{-n}.
  \]
  For each non-trivial \(\varrho\),
  \Cref{lem:nontrivial-peter-weyl-channel-bound} gives
  \[
    \pi_{n,\varrho}(A)\lambda^{-n}
    =O\!\left(\frac{\vartheta_*^n}{n}\right),
  \]
  so \Cref{lem:geometric-harmonic-tail} implies
  \[
    \sum_{n\le N}\pi_{n,\varrho}(A)\lambda^{-n}
    =
    \Pi_\varrho(\lambda^{-1})
    +O\!\left(\frac{\vartheta_*^N}{N}\right).
  \]
  Summing over the finitely many non-trivial irreducibles yields
  \eqref{eq:class-mertens-harmonic} and
  \eqref{eq:class-mertens-constant}. Since
  \(H_N=\log N+\gamma+O(N^{-1})\), equation
  \eqref{eq:class-mertens-log} follows.
\end{proof}

For every finite matrix \(M\) and every eigenvalue
\(\alpha\in\spec(M)\), let \(m_M(\alpha)\) denote the algebraic
multiplicity of \(\alpha\). Assume throughout the next theorem that
\[
  R:=\max\{\Lambda(A),\eta\}>\lambda^{1/2}.
\]
Define the spectral edge sets
\[
  \begin{aligned}
    \mathcal E_{\mathbf 1}
    &:=\{\mu\in\spec(A)\setminus\{\lambda\}:|\mu|=R\}, \\
    \mathcal E_\varrho
    &:=\{\beta\in\spec(A_\varrho):|\beta|=R\}
    \quad (\varrho\neq\mathbf 1),
  \end{aligned}
\]
and
\[
  \Gamma_{\mathrm{sec}}
  :=
  \max\left\{
    \lambda^{1/2},
    \max_{\mu\in\spec(A)\setminus(\{\lambda\}\cup\mathcal E_{\mathbf 1})}
      |\mu|,
    \max_{\varrho\neq\mathbf 1}
      \max_{\beta\in\spec(A_\varrho)\setminus\mathcal E_\varrho}
        |\beta|
  \right\}.
\]
For each conjugacy class \(C\subset G\), set
\begin{equation}\label{eq:class-secondary-profile}
  \mathfrak s_C(n)
  :=
  \frac{|C|}{|G|}
  \left(
    \sum_{\mu\in\mathcal E_{\mathbf 1}}m_A(\mu)\mu^n
    +
    \sum_{\varrho\neq\mathbf 1}\chi_\varrho(C^{-1})
      \sum_{\beta\in\mathcal E_\varrho}m_{A_\varrho}(\beta)\beta^n
  \right)
\end{equation}
and
\begin{equation}\label{eq:class-secondary-tail}
  \mathfrak T_C(N)
  :=
  \sum_{n>N}\frac{\mathfrak s_C(n)}{n\lambda^n}.
\end{equation}

\begin{theorem}[Secondary term above the square-root barrier]
  \label{thm:class-mertens-secondary-term}
  Assume that
  \[
    \eta:=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda
    \qquad\text{and}\qquad
    R:=\max\{\Lambda(A),\eta\}>\lambda^{1/2}.
  \]
  Then \(\Gamma_{\mathrm{sec}}<R\), the series
  \(\mathfrak T_C(N)\) converges absolutely for every \(C\), and for
  every conjugacy class \(C\subset G\),
  \begin{equation}\label{eq:class-secondary-pointwise}
    p_{n,C}
    =
    \frac{|C|}{|G|}\frac{\lambda^n}{n}
    +\frac{\mathfrak s_C(n)}{n}
    +O\!\left(\frac{\Gamma_{\mathrm{sec}}^n}{n}\right).
  \end{equation}
  Moreover,
  \begin{equation}\label{eq:class-secondary-harmonic}
    \sum_{n\le N}\frac{p_{n,C}}{\lambda^n}
    =
    \frac{|C|}{|G|}H_N
    +\mathcal M_C(A)-\frac{|C|}{|G|}\gamma
    -\mathfrak T_C(N)
    +O\!\left(\frac{(\Gamma_{\mathrm{sec}}/\lambda)^N}{N}\right).
  \end{equation}
  The implied constants are uniform in the conjugacy class \(C\), in
  \(n\), and in \(N\).
\end{theorem}

\begin{proof}
  Since \(R>\lambda^{1/2}\) and the eigenvalue sets are finite, the
  definition of \(R\) implies \(\Gamma_{\mathrm{sec}}<R\).

  The trace of a finite matrix equals the sum of its eigenvalues with
  algebraic multiplicity. Therefore
  \[
    \Tr(A^n)
    =
    \lambda^n
    +\sum_{\mu\in\mathcal E_{\mathbf 1}}m_A(\mu)\mu^n
    +O(\Gamma_{\mathrm{sec}}^n),
  \]
  and, for every non-trivial \(\varrho\in\Irr(G)\),
  \[
    \Tr(A_\varrho^n)
    =
    \sum_{\beta\in\mathcal E_\varrho}m_{A_\varrho}(\beta)\beta^n
    +O(\Gamma_{\mathrm{sec}}^n).
  \]
  The implied constants are uniform in \(\varrho\) because \(\Irr(G)\)
  is finite.

  Inserting the first identity into \eqref{eq:primitive-mobius} yields
  \[
    p_n(A)
    =
    \frac{\lambda^n}{n}
    +\frac{1}{n}\sum_{\mu\in\mathcal E_{\mathbf 1}}m_A(\mu)\mu^n
    +O\!\left(\frac{\Gamma_{\mathrm{sec}}^n}{n}\right),
  \]
  because the divisor \(d=1\) contributes the displayed spectral-edge
  sum, whereas the divisors \(d>1\) contribute
  \[
    O\!\left(
      \frac{1}{n}\sum_{\substack{d\mid n\\ d>1}}\lambda^{n/d}
    \right)
    =
    O\!\left(\frac{\lambda^{n/2}}{n}\right)
    =
    O\!\left(\frac{\Gamma_{\mathrm{sec}}^n}{n}\right).
  \]

  Fix next \(\varrho\neq\mathbf 1\). By
  \Cref{cor:primitive-peter-weyl-trace},
  \[
    \pi_{n,\varrho}(A)
    =
    \frac{1}{n}\Tr(A_\varrho^n)
    +\frac{1}{n}
      \sum_{\substack{m\mid n\\ m>1}}\mu(m)
        \sum_{\sigma\in\Irr(G)}
          a_{\varrho,\sigma}^{(m)}\Tr(A_\sigma^{n/m}).
  \]
  The first term gives the contribution from \(\mathcal E_\varrho\). For
  the divisor sum, the same argument as in
  \Cref{lem:nontrivial-peter-weyl-channel-bound}, using
  \Cref{lem:adams-coefficients-bounded} and
  \Cref{prop:twisted-spectral-radius-bound}, yields
  \[
    \sum_{\substack{m\mid n\\ m>1}}
      \left|
        \sum_{\sigma\in\Irr(G)}
          a_{\varrho,\sigma}^{(m)}\Tr(A_\sigma^{n/m})
      \right|
    =
    O(\lambda^{n/2}),
  \]
  uniformly in \(\varrho\). Hence
  \[
    \pi_{n,\varrho}(A)
    =
    \frac{1}{n}
      \sum_{\beta\in\mathcal E_\varrho}m_{A_\varrho}(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\mathrm{sec}}^n}{n}\right)
  \]
  uniformly for all non-trivial \(\varrho\).

  Substituting the last two expansions into
  \Cref{lem:class-primitive-trivial-split} gives
  \eqref{eq:class-secondary-pointwise}. The uniformity in \(C\) follows
  from the finiteness of the conjugacy classes and the bounds
  \(|\chi_\varrho(C^{-1})|\le \dim\varrho\).

  Dividing \eqref{eq:class-secondary-pointwise} by \(\lambda^n\) yields
  \[
    \frac{p_{n,C}}{\lambda^n}
    =
    \frac{|C|}{|G|}\frac{1}{n}
    +\frac{\mathfrak s_C(n)}{n\lambda^n}
    +O\!\left(\frac{(\Gamma_{\mathrm{sec}}/\lambda)^n}{n}\right).
  \]
  Since \(R<\lambda\), the definition of \(\mathfrak s_C(n)\) shows
  that \(\mathfrak T_C(N)\) converges absolutely. Summing the preceding
  identity over \(n>N\) and applying
  \Cref{lem:geometric-harmonic-tail} gives
  \[
    \sum_{n>N}
      \left(
        \frac{p_{n,C}}{\lambda^n}
        -\frac{|C|}{|G|}\frac{1}{n}
      \right)
    =
    \mathfrak T_C(N)
    +O\!\left(\frac{(\Gamma_{\mathrm{sec}}/\lambda)^N}{N}\right).
  \]
  On the other hand, \Cref{thm:class-mertens-explicit} implies
  \[
    \mathcal M_C(A)-\frac{|C|}{|G|}\gamma
    =
    \sum_{n\ge 1}
      \left(
        \frac{p_{n,C}}{\lambda^n}
        -\frac{|C|}{|G|}\frac{1}{n}
      \right).
  \]
  Subtracting the tail beyond \(N\) from this identity yields
  \[
    \sum_{n\le N}\frac{p_{n,C}}{\lambda^n}
    =
    \frac{|C|}{|G|}H_N
    +\mathcal M_C(A)-\frac{|C|}{|G|}\gamma
    -\mathfrak T_C(N)
    +O\!\left(\frac{(\Gamma_{\mathrm{sec}}/\lambda)^N}{N}\right),
  \]
  which is \eqref{eq:class-secondary-harmonic}.
\end{proof}

\begin{corollary}[Classwise primitive remainders detect the dominant modulus]
  \label{cor:class-secondary-term-detects-spectrum}
  Under the hypotheses of \Cref{thm:class-mertens-secondary-term},
  \[
    R
    =
    \max_{C\subset G}
      \limsup_{n\to\infty}
      \left|
        n\,p_{n,C}-\frac{|C|}{|G|}\lambda^n
      \right|^{1/n},
  \]
  where the maximum is taken over the conjugacy classes of \(G\).
\end{corollary}

\begin{proof}
  By \eqref{eq:class-secondary-pointwise},
  \[
    n\,p_{n,C}-\frac{|C|}{|G|}\lambda^n
    =
    \mathfrak s_C(n)+O(\Gamma_{\mathrm{sec}}^n)
    \qquad (n\to\infty),
  \]
  so
  \[
    \limsup_{n\to\infty}
      \left|
        n\,p_{n,C}-\frac{|C|}{|G|}\lambda^n
      \right|^{1/n}
    \le R
  \]
  for every \(C\).

  Assume that the displayed maximum were strictly smaller than \(R\).
  Since the set of conjugacy classes is finite, there would then exist
  \(\Gamma_0<R\) such that
  \[
    n\,p_{n,C}-\frac{|C|}{|G|}\lambda^n
    =O(\Gamma_0^n)
    \qquad\text{uniformly in } C.
  \]
  Together with \eqref{eq:class-secondary-pointwise}, this would imply
  \[
    \mathfrak s_C(n)=O(\Gamma_1^n)
    \qquad\text{for every } C,
  \]
  where \(\Gamma_1:=\max\{\Gamma_0,\Gamma_{\mathrm{sec}}\}<R\).
  Multiplying by \(|G|/|C|\) gives the class function
  \[
    \mathfrak b_n(C)
    :=
    \frac{|G|}{|C|}\mathfrak s_C(n)
    =
    \sum_{\mu\in\mathcal E_{\mathbf 1}}m_A(\mu)\mu^n
    +
    \sum_{\varrho\neq\mathbf 1}\chi_\varrho(C^{-1})
      \sum_{\beta\in\mathcal E_\varrho}m_{A_\varrho}(\beta)\beta^n,
  \]
  which would also satisfy \(\mathfrak b_n(C)=O(\Gamma_1^n)\)
  uniformly in \(C\).

  Averaging over conjugacy classes yields
  \[
    \sum_{C\subset G}\frac{|C|}{|G|}\mathfrak b_n(C)
    =
    \sum_{\mu\in\mathcal E_{\mathbf 1}}m_A(\mu)\mu^n,
  \]
  because the average of every non-trivial irreducible character
  vanishes. Likewise, for every non-trivial \(\sigma\in\Irr(G)\),
  character orthogonality gives
  \[
    \sum_{C\subset G}\frac{|C|}{|G|}\mathfrak b_n(C)\chi_\sigma(C)
    =
    \sum_{\beta\in\mathcal E_\sigma}m_{A_\sigma}(\beta)\beta^n.
  \]
  The assumed bound \(\mathfrak b_n(C)=O(\Gamma_1^n)\) would therefore
  force each of these spectral-edge sums to be \(O(\Gamma_1^n)\).

  By definition of \(R\), at least one among the finitely many sums
  \[
    \sum_{\mu\in\mathcal E_{\mathbf 1}}m_A(\mu)\mu^n,
    \qquad
    \sum_{\beta\in\mathcal E_\sigma}m_{A_\sigma}(\beta)\beta^n
    \quad (\sigma\neq\mathbf 1),
  \]
  is a non-zero finite exponential sum whose frequencies all have
  modulus \(R\). \Cref{lem:finite-part-exponential-sum-limsup} then
  gives limsup \(R\), contradicting the bound \(O(\Gamma_1^n)\) with
  \(\Gamma_1<R\). Hence the maximum cannot be smaller than \(R\), and
  equality follows.
\end{proof}

\begin{proposition}[Evaluation at the Perron point]
  \label{prop:class-mertens-deviations}
  Under the hypotheses of \Cref{thm:class-mertens-explicit}, define
  \[
    \Delta_C(A)
    :=\mathcal M_C(A)-\frac{|C|}{|G|}\mathcal M(A).
  \]
  Then the classwise deviations \(\Delta_C(A)\) are well defined, and
  \begin{equation}\label{eq:class-mertens-fourier-expansion}
    \Delta_C(A)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\,\Pi_\varrho(\lambda^{-1}).
  \end{equation}
  Moreover, for every non-trivial \(\varrho\),
  \begin{equation}\label{eq:class-mertens-determinant}
    \Pi_\varrho(\lambda^{-1})
    =
    \sum_{m\ge 1}\frac{\mu(m)}{m}
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}
        \log\det(I-\lambda^{-m}A_\sigma)^{-1}.
  \end{equation}
\end{proposition}

\begin{proof}
  Subtract \((|C|/|G|)\mathcal M(A)\) from
  \eqref{eq:class-mertens-constant} to obtain
  \eqref{eq:class-mertens-fourier-expansion}.

  For \(\varrho\neq\mathbf 1\), one has
  \(a_{\varrho,\mathbf 1}^{(1)}=0\) because
  \(\psi^1\chi_\varrho=\chi_\varrho\) is itself non-trivial. Hence the
  \(m=1\) term in \eqref{eq:primitive-peter-weyl-determinant} involves
  only non-trivial \(\sigma\). By
  \Cref{prop:twisted-spectral-radius-bound} and the hypothesis
  \(\eta<\lambda\), each such \(A_\sigma\) satisfies
  \(\rho(A_\sigma)\le \eta<\lambda\), so
  \(\log\det(I-zA_\sigma)^{-1}\) is holomorphic at \(z=\lambda^{-1}\).
  For \(m\ge 2\), the point \(\lambda^{-m}\) lies strictly inside the
  Perron disc for every twisted determinant logarithm. Moreover, there
  is a constant \(K\) such that
  \(\abs{\Tr(A_\sigma^n)}\le K\lambda^n\) for all
  \(\sigma\in\Irr(G)\) and \(n\ge 1\), and therefore
  \[
    \abs{\log\det(I-\lambda^{-m}A_\sigma)^{-1}}
    \le
    K\sum_{n\ge 1}\frac{\lambda^{-(m-1)n}}{n}
    =O(\lambda^{1-m})
    \qquad (m\ge 2),
  \]
  uniformly in \(\sigma\). Together with the finiteness of \(\Irr(G)\)
  and \Cref{lem:adams-coefficients-bounded}, this proves the absolute
  convergence of \eqref{eq:class-mertens-determinant}. Evaluating
  \eqref{eq:primitive-peter-weyl-determinant} at \(z=\lambda^{-1}\)
  yields \eqref{eq:class-mertens-determinant}.
\end{proof}

\begin{remark}[Reality of the class constants]
  \label{rem:class-mertens-constants-real}
  Although \eqref{eq:class-mertens-constant} and
  \eqref{eq:class-mertens-fourier-expansion} are written with complex
  characters, the constants \(\mathcal M_C(A)\) and \(\Delta_C(A)\) are
  real. Indeed,
  \[
    \sum_{n\le N}\frac{p_{n,C}}{\lambda^n}\in\RR
  \]
  for every \(N\), so \Cref{thm:class-mertens-explicit} identifies a
  real constant term. Equivalently, the non-real contributions pair in
  complex-conjugate channels: if \(\overline{\varrho}\) denotes the
  contragredient irreducible, then
  \(\chi_{\overline{\varrho}}(C^{-1})
  =\overline{\chi_\varrho(C^{-1})}\) and
  \(\Pi_{\overline{\varrho}}(\lambda^{-1})
  =\overline{\Pi_\varrho(\lambda^{-1})}\).
\end{remark}

\subsection{The hypothesis \texorpdfstring{$\eta<\lambda$}{eta<lambda}}
\label{subsec:eta-gap}

\begin{remark}
  \label{rem:twisted-spectral-gap-role}
  The condition
  \[
    \eta=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda
  \]
  does not enter the determinant identities themselves: the formulas of
  \Cref{thm:class-function-linearisation,cor:class-function-adams-mobius,cor:primitive-peter-weyl-determinant}
  are valid throughout the disc \(|z|<\lambda^{-1}\) without it. Its
  only role is to permit evaluation at the Perron point
  \(z=\lambda^{-1}\) in the non-trivial channels and thereby convert the
  primitive Peter--Weyl series into explicit class Mertens constants.
\end{remark}

\begin{theorem}[Rigidity at the Perron boundary]
  \label{thm:eta-gap-rigidity}
  Let \(\varrho:G\to U(V_\varrho)\) be an irreducible unitary
  representation. Then
  \[
    \rho(A_\varrho)=\lambda
  \]
  if and only if there exist \(\omega\in\TT\) and unit vectors
  \(\xi_i\in V_\varrho\), indexed by the vertices \(i\in V\), such that
  \begin{equation}\label{eq:eta-gap-line-field}
    \varrho(\tau(e))\xi_j=\omega\,\xi_i
    \qquad
    \text{for every admissible edge } e:i\to j.
  \end{equation}
  Whenever \eqref{eq:eta-gap-line-field} holds, every closed path
  \(\gamma\) of length \(n\) based at \(i\) satisfies
  \begin{equation}\label{eq:eta-gap-loop-character}
    \varrho(\tau(\gamma))\xi_i=\omega^n\xi_i.
  \end{equation}
  In particular, the subgroup
  \[
    \Gamma_i(\tau)
    :=
    \bigl\langle \tau(\gamma):
      \gamma \text{ closed path based at } i
    \bigr\rangle
  \]
  acts on the line \(\CC\xi_i\) through a one-dimensional character.
\end{theorem}

\begin{proof}
  Assume first that \(\rho(A_\varrho)=\lambda\). Since
  \(A_\varrho\) is finite dimensional, some eigenvalue
  \(\alpha\in\spec(A_\varrho)\) satisfies \(|\alpha|=\lambda\). Let
  \(v=(v_i)_{i\in V}\in \CC^V\otimes V_\varrho\) be a corresponding
  eigenvector, so
  \[
    \alpha v_i=\sum_{e:i\to j}\varrho(\tau(e))v_j
    \qquad (i\in V).
  \]
  Set \(u_i:=\norm{v_i}\). Then
  \[
    \lambda u_i
    =
    \norm{\alpha v_i}
    =
    \left\|\sum_{e:i\to j}\varrho(\tau(e))v_j\right\|
    \le
    \sum_{e:i\to j}\norm{v_j}
    =(Au)_i
  \]
  for every \(i\in V\). Let \(\ell>0\) be a left Perron vector of
  \(A\), normalised by \(\ell^{\mathsf T}A=\lambda\ell^{\mathsf T}\).
  Then
  \[
    \lambda\,\ell^{\mathsf T}u
    \le
    \ell^{\mathsf T}Au
    =
    \lambda\,\ell^{\mathsf T}u,
  \]
  so every coordinate inequality is an equality because \(\ell_i>0\)
  for all \(i\). Hence \(Au=\lambda u\). Primitivity of \(A\) implies
  that \(u\) is a positive Perron eigenvector, so \(u_i>0\) for every
  \(i\in V\).

  Equality in the triangle inequality now holds in every vertex
  equation. Therefore, for each fixed \(i\), all non-zero vectors
  \(\varrho(\tau(e))v_j\) with \(e:i\to j\) lie on the same ray. Since
  each such vector has norm \(u_j\) and their sum is \(\alpha v_i\),
  one has
  \[
    \varrho(\tau(e))v_j
    =
    u_j\,\frac{\alpha v_i}{\lambda u_i}
    \qquad
    \text{for every admissible edge } e:i\to j.
  \]
  Write \(\omega:=\alpha/\lambda\in\TT\) and
  \(\xi_i:=v_i/u_i\in V_\varrho\). Dividing by \(u_j\) yields
  \eqref{eq:eta-gap-line-field}.

  Conversely, assume that \eqref{eq:eta-gap-line-field} holds. Let
  \(h=(h_i)_{i\in V}>0\) be a right Perron vector of \(A\), so
  \(Ah=\lambda h\). Define \(w=(w_i)_{i\in V}\) by
  \(w_i:=h_i\xi_i\). Then
  \begin{align*}
    (A_\varrho w)_i
    &=
    \sum_{e:i\to j}\varrho(\tau(e))w_j \\
    &=
    \sum_{e:i\to j}h_j\varrho(\tau(e))\xi_j \\
    &=
    \omega\sum_{e:i\to j}h_j\xi_i \\
    &=
    \omega(Ah)_i\xi_i \\
    &=
    \omega\lambda w_i.
  \end{align*}
  Thus \(\omega\lambda\in\spec(A_\varrho)\), so
  \(\rho(A_\varrho)\ge \lambda\). On the other hand,
  \Cref{prop:twisted-spectral-radius-bound} gives
  \(\rho(A_\varrho)\le \lambda\). Hence
  \(\rho(A_\varrho)=\lambda\).

  Finally, iterating \eqref{eq:eta-gap-line-field} along a closed path
  \(\gamma=e_1\cdots e_n\) based at \(i\) gives
  \eqref{eq:eta-gap-loop-character}. Hence the line \(\CC\xi_i\) is
  \(\Gamma_i(\tau)\)-invariant. The action of \(\Gamma_i(\tau)\) on
  this line defines a one-dimensional character.
\end{proof}

\begin{corollary}[Perfect loop groups force a strict twisted gap]
  \label{cor:eta-gap-perfect-group}
  Assume that \(\Gamma_{i_0}(\tau)=G\) for some vertex \(i_0\). If
  \(G\) is perfect, then
  \[
    \rho(A_\varrho)<\lambda
    \qquad
    \text{for every non-trivial } \varrho\in\Irr(G).
  \]
  In particular,
  \[
    \eta=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda.
  \]
  Consequently, the Perron-point formulas of
  \Cref{thm:class-mertens-explicit,thm:class-mertens-fourier} apply
  automatically to such extensions.
\end{corollary}

\begin{proof}
  Suppose that \(\rho(A_\varrho)=\lambda\) for some non-trivial
  \(\varrho\in\Irr(G)\). By \Cref{thm:eta-gap-rigidity}, the subgroup
  \(\Gamma_{i_0}(\tau)=G\) preserves a one-dimensional subspace of
  \(V_\varrho\). The induced action of \(G\) on that line is a
  one-dimensional character. Since \(G\) is perfect, every
  one-dimensional character of \(G\) is trivial. Therefore
  \(V_\varrho\) contains a non-zero \(G\)-fixed vector, which is
  impossible for a non-trivial irreducible representation.
\end{proof}

\begin{corollary}[One-dimensional equality criterion]
  \label{cor:eta-gap-character-classification}
  Let \(\chi:G\to\TT\) be a one-dimensional character. Then
  \[
    \rho(A_\chi)=\lambda
  \]
  if and only if there exist \(\omega\in\TT\) and a function
  \(b:V\to\TT\) such that, for every admissible edge \(e:i\to j\),
  \begin{equation}\label{eq:eta-gap-character-classification}
    \chi(\tau(e))=\omega\,b(i)^{-1}b(j).
  \end{equation}
  Equivalently, if \(D:=\diag(b(i))\), then
  \begin{equation}\label{eq:eta-gap-character-similarity}
    A_\chi=\omega\,D^{-1}AD.
  \end{equation}
\end{corollary}

\begin{proof}
  If \(\rho(A_\chi)=\lambda\), then
  \Cref{thm:eta-gap-rigidity} gives unit vectors
  \(\xi_i\in\CC\), hence numbers \(b(i)\in\TT\) with
  \(\xi_i=b(i)^{-1}\). The line-field relation
  \eqref{eq:eta-gap-line-field} becomes
  \[
    \chi(\tau(e))\,b(j)^{-1}=\omega\,b(i)^{-1},
  \]
  which is exactly \eqref{eq:eta-gap-character-classification}.

  Conversely, if \eqref{eq:eta-gap-character-classification} holds,
  then for each admissible pair \(i\to j\),
  \[
    (A_\chi)_{ij}
    =
    \sum_{e:i\to j}\chi(\tau(e))
    =
    \omega\,b(i)^{-1}
      \left(\sum_{e:i\to j}1\right)b(j)
    =
    \omega\,b(i)^{-1}A_{ij}b(j).
  \]
  This is \eqref{eq:eta-gap-character-similarity}, so \(A_\chi\) is
  similar to \(\omega A\). Therefore \(\rho(A_\chi)=\lambda\).
\end{proof}

\begin{remark}
  Canonical cyclic lifts furnish an explicit equality case:
  \Cref{rem:abelian-toy-model-cyclic} shows that
  \(A_{\chi_j}=\omega_q^jA\), so every non-trivial cyclic character has
  spectral radius \(\lambda\). In particular, primitivity of the base
  shift does not by itself imply \(\eta<\lambda\), even for finite
  abelian extensions.
\end{remark}

\begin{remark}
  The equality mechanism of
  \Cref{thm:eta-gap-rigidity,cor:eta-gap-character-classification}
  is the finite-state analogue of the cohomological obstructions studied by Parry
  \cite{Parry1999LivsicNonAbelian} and Boyle--Schmieding
  \cite{BoyleSchmieding2017FiniteGroupExtensions}. For the broader
  pressure and zeta-function setting of group extensions, compare
  Sharp's discussion of amenable group extensions of shifts
  \cite{Sharp2020AmenableGroupExtensions} and Stadlbauer's work on
  amenability, conformal measures, and harmonic functions for group
  extensions
  \cite{Stadlbauer2013KestenAmenability,Stadlbauer2019ConformalMeasures}.
\end{remark}

\subsection{Fourier reconstruction}

By \Cref{prop:class-mertens-deviations}, the deviations
\(\Delta_C(A)\) form a class function whose Fourier coefficients are the
non-trivial Perron-point values \(\Pi_\varrho(\lambda^{-1})\). The next
theorem makes this explicit.

\begin{theorem}[Fourier inversion for the class constants]
  \label{thm:class-mertens-fourier}
  Under the hypotheses of \Cref{thm:class-mertens-explicit}, let
  \(\Delta_C(A)\) be as in \Cref{prop:class-mertens-deviations}. Then
  for every non-trivial \(\sigma\in\Irr(G)\),
  \begin{equation}\label{eq:class-mertens-fourier-inversion}
    \Pi_\sigma(\lambda^{-1})
    =
    \sum_{C\subset G}\Delta_C(A)\chi_\sigma(C),
  \end{equation}
  where the sum runs over conjugacy classes. Moreover,
  \begin{equation}\label{eq:class-mertens-plancherel}
    \sum_{C\subset G}\frac{|G|}{|C|}\abs{\Delta_C(A)}^2
    =
    \sum_{\varrho\neq\mathbf 1}
      \abs{\Pi_\varrho(\lambda^{-1})}^2.
  \end{equation}
\end{theorem}

\begin{proof}
  Multiply \eqref{eq:class-mertens-fourier-expansion} by
  \(\chi_\sigma(C)\) and sum over conjugacy classes:
  \[
    \sum_{C\subset G}\Delta_C(A)\chi_\sigma(C)
    =
    \sum_{C\subset G}\frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\Pi_\varrho(\lambda^{-1})\chi_\sigma(C).
  \]
  The orthogonality relation
  \[
    \sum_{C\subset G}\frac{|C|}{|G|}
      \chi_\varrho(C^{-1})\chi_\sigma(C)
    =\delta_{\varrho,\sigma}
  \]
  yields \eqref{eq:class-mertens-fourier-inversion}.

  For the Parseval identity, multiply
  \eqref{eq:class-mertens-fourier-expansion} by \(|G|/|C|\), take the
  modulus squared, and sum over conjugacy classes. Since
  \[
    |\Delta_C(A)|^2
    =
    \left(\frac{|C|}{|G|}\right)^2
    \left|
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\Pi_\varrho(\lambda^{-1})
    \right|^2,
  \]
  one obtains
  \[
    \sum_{C\subset G}\frac{|G|}{|C|}|\Delta_C(A)|^2
    =
    \sum_{C\subset G}\frac{|C|}{|G|}
      \left|\sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\Pi_\varrho(\lambda^{-1})\right|^2.
  \]
  Expanding the modulus square with complex conjugation gives
  \[
    \sum_{C\subset G}\frac{|G|}{|C|}|\Delta_C(A)|^2
    =
    \sum_{\varrho,\sigma\neq\mathbf 1}
      \Pi_\varrho(\lambda^{-1})
      \overline{\Pi_\sigma(\lambda^{-1})}
      \sum_{C\subset G}\frac{|C|}{|G|}
        \chi_\varrho(C^{-1})
        \overline{\chi_\sigma(C^{-1})}.
  \]
  By \Cref{rem:character-inversion-conjugation},
  \(\overline{\chi_\sigma(C^{-1})}=\chi_\sigma(C)\), so the inner sum is
  \[
    \sum_{C\subset G}\frac{|C|}{|G|}
      \chi_\varrho(C^{-1})\chi_\sigma(C)
    =\delta_{\varrho,\sigma}.
  \]
  Therefore only the diagonal terms survive, and
  \eqref{eq:class-mertens-plancherel} follows.
\end{proof}

\begin{remark}
  \Cref{thm:class-mertens-fourier} is the non-abelian analogue of the
  root-of-unity reconstruction results from \Cref{sec:finite-part}. In
  the abelian case, one root-of-unity layer recovers the reduced
  spectral data by ordinary Fourier inversion. Here the full family of
  conjugacy-class Mertens constants recovers the non-trivial twisted
  primitive values \(\Pi_\varrho(\lambda^{-1})\) by non-abelian Fourier
  inversion.
\end{remark}

\subsection{A H\"older transfer-operator extension}

The primitive inversion formalism extends beyond finite matrices. Let
\((X_A^+,\sigma)\) be the mixing one-sided shift attached to the
primitive matrix \(A\), let \(\varphi:X_A^+\to\RR\) be a real
H\"older potential, and write \(P:=P(\varphi)\) for its topological
pressure. Let \(\tau:X_A^+\to G\) be a locally constant cocycle into a
finite group. Fix \(\alpha\in(0,1]\) such that \(\varphi\in C^\alpha\).
The same determinant formalism yields both Perron-point constants and,
above the square-root barrier, explicit spectral-window corrections
coming from the twisted Ruelle spectrum.
For \(\varrho\in\Irr(G)\), the twisted Ruelle operator on
\(C^\alpha(X_A^+,V_\varrho)\) is
\[
  (\mathscr L_{\varphi,\varrho}F)(x)
  :=
  \sum_{\sigma y=x}e^{\varphi(y)}\varrho(\tau(y))F(y).
\]
For a class function \(\phi:G\to\CC\), define
\[
  T_{\phi,\varphi,n}
  :=
  \sum_{x\in\operatorname{Fix}(\sigma^n)}
    e^{S_n\varphi(x)}\phi(\tau_n(x)),
  \qquad
  \mathcal L_{\phi,\varphi}(z)
  :=
  \sum_{n\ge 1}\frac{T_{\phi,\varphi,n}}{n}z^n.
\]
If \(\gamma\) is a primitive orbit of length \(|\gamma|=n\), then
\(S_\gamma\varphi:=S_n\varphi(x)\) is independent of the choice of
\(x\in\gamma\), and the weighted primitive series is
\[
  \Pi_{\phi,\varphi}(z)
  :=
  \sum_{\gamma\in\mathcal P}
    e^{S_\gamma\varphi}\phi(\tau(\gamma))z^{|\gamma|}.
\]
For \(\varrho\in\Irr(G)\), write
\[
  \Pi_{\varrho,\varphi}(z):=\Pi_{\chi_\varrho,\varphi}(z)
  =\sum_{n\ge 1}\pi_{n,\varrho}(\varphi)z^n.
\]

\begin{theorem}[H\"older transfer-operator Perron-point theorem]
  \label{thm:holder-transfer-perron-point}
  Assume that
  \[
    \eta_\varphi
    :=
    \max_{\varrho\neq\mathbf 1}\rho(\mathscr L_{\varphi,\varrho})
    <e^P.
  \]
  Then there exists \(\Theta_\varphi\in(0,e^P)\) with the following
  properties.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item For every class function \(\phi:G\to\CC\) and every
      \(|z|<e^{-P}\),
      \begin{equation}\label{eq:holder-transfer-adams-mobius}
        \Pi_{\phi,\varphi}(z)
        =
        \sum_{m\ge 1}\frac{\mu(m)}{m}\,
          \mathcal L_{\psi^m\phi,\varphi}(z^m).
      \end{equation}
    \item For every non-trivial \(\varrho\in\Irr(G)\), the series
      \[
        \Pi_{\varrho,\varphi}(e^{-P})
        =
        \sum_{n\ge 1}\pi_{n,\varrho}(\varphi)e^{-Pn}
      \]
      converges absolutely.
    \item If
      \[
        p_n(\varphi)
        :=
        \sum_{\substack{\gamma\in\mathcal P\\ |\gamma|=n}}
          e^{S_\gamma\varphi}
      \]
      and,
      for a conjugacy class \(C\subset G\),
      \[
        p_{n,C}(\varphi)
        :=
        \sum_{\substack{\gamma\in\mathcal P\\ |\gamma|=n\\
          \tau(\gamma)\in C}}
          e^{S_\gamma\varphi},
      \]
      then
      \begin{equation}\label{eq:holder-transfer-pointwise}
        p_{n,C}(\varphi)
        =
        \frac{|C|}{|G|}\frac{e^{Pn}}{n}
        +O\!\left(\frac{\Theta_\varphi^n}{n}\right),
      \end{equation}
      and
      \begin{equation}\label{eq:holder-transfer-mertens}
        \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
        =
        \frac{|C|}{|G|}H_N
        +\mathcal M_C(\varphi)-\frac{|C|}{|G|}\gamma
        +O\!\left(\frac{(\Theta_\varphi e^{-P})^N}{N}\right),
      \end{equation}
      where
      \[
        \mathcal M(\varphi)
        :=
        \gamma+\sum_{n\ge 1}
          \left(e^{-Pn}p_n(\varphi)-\frac{1}{n}\right)
      \]
      and
      \begin{equation}\label{eq:holder-transfer-constant}
        \mathcal M_C(\varphi)
        =
        \frac{|C|}{|G|}\mathcal M(\varphi)
        +\frac{|C|}{|G|}
          \sum_{\varrho\neq\mathbf 1}
            \chi_\varrho(C^{-1})\Pi_{\varrho,\varphi}(e^{-P}).
      \end{equation}
      In particular,
      \[
        \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
        =
        \frac{|C|}{|G|}\log N+\mathcal M_C(\varphi)+O(N^{-1}).
      \]
    \item If
      \[
        \Delta_C(\varphi)
        :=
        \mathcal M_C(\varphi)-\frac{|C|}{|G|}\mathcal M(\varphi),
      \]
      then for every non-trivial \(\sigma\in\Irr(G)\),
      \begin{equation}\label{eq:holder-transfer-fourier}
        \Pi_{\sigma,\varphi}(e^{-P})
        =
        \sum_{C\subset G}\Delta_C(\varphi)\chi_\sigma(C),
      \end{equation}
      where the sum runs over conjugacy classes.
  \end{enumerate}
\end{theorem}

\begin{proof}
  Classical dynamical determinant theory for H\"older potentials on
  mixing shifts of finite type, applied to the operators
  \(\mathscr L_{\varphi,\varrho}\), gives a number \(\Theta_0<e^P\) such
  that
  \[
    T_{\mathbf 1,\varphi,n}=e^{Pn}+O(\Theta_0^n),
    \qquad
    T_{\chi_\varrho,\varphi,n}=O(\Theta_0^n)
    \quad (\varrho\neq\mathbf 1);
  \]
  see \cite{Ruelle1978Thermodynamic,Baladi2000PositiveTransfer,%
  Baladi2018ZetaDeterminants,AdachiSunada1987TwistedPerronFrobenius}.
  Set
  \[
    \Theta_\varphi:=\max\{\Theta_0,e^{P/2}\}<e^P.
  \]
  The estimate
  \[
    |T_{\chi_\sigma,\varphi,n}|
    \le (\dim\sigma)\,T_{\mathbf 1,\varphi,n}
    =O(e^{Pn})
  \]
  holds uniformly in \(\sigma\in\Irr(G)\). Since every class function is
  a finite linear combination of irreducible characters, the same
  \(O(e^{Pn})\) bound holds for \(T_{\phi,\varphi,n}\). Hence
  \(\mathcal L_{\phi,\varphi}(z)\) converges absolutely for
  \(|z|<e^{-P}\).

  Part~\textup{(i)} is then purely orbit-theoretic. If \(\gamma\) is
  primitive of length \(\ell\), then its \(r\)-fold repetition
  contributes \(\ell\) points to \(\operatorname{Fix}(\sigma^{r\ell})\),
  each with weight \(e^{rS_\gamma\varphi}\) and label conjugate to
  \(\tau(\gamma)^r\). Therefore
  \[
    \mathcal L_{\phi,\varphi}(z)
    =
    \sum_{\gamma\in\mathcal P}\sum_{r\ge 1}
      \frac{e^{rS_\gamma\varphi}\phi(\tau(\gamma)^r)}{r}z^{r|\gamma|},
    \qquad |z|<e^{-P},
  \]
  and the same M\"obius calculation as in the proof of
  \Cref{cor:class-function-adams-mobius} yields
  \eqref{eq:holder-transfer-adams-mobius}.

  Coefficient extraction in
  \eqref{eq:holder-transfer-adams-mobius}, applied to
  \(\phi=\chi_\varrho\) and the Adams decomposition
  \eqref{eq:adams-character-decomposition}, gives
  \[
    \pi_{n,\varrho}(\varphi)
    =
    \frac{1}{n}\sum_{m\mid n}\mu(m)
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}T_{\chi_\sigma,\varphi,n/m}.
  \]
  Since \(a_{\varrho,\sigma}^{(1)}=\delta_{\varrho,\sigma}\), the
  divisor \(m=1\) contributes \(O(\Theta_\varphi^n/n)\) when
  \(\varrho\neq\mathbf 1\). For \(m>1\), the bounds above and
  \Cref{lem:adams-coefficients-bounded} give
  \[
    \frac{1}{n}
    \sum_{\substack{m\mid n\\ m>1}}
      \left|
        \sum_{\sigma\in\Irr(G)}
          a_{\varrho,\sigma}^{(m)}T_{\chi_\sigma,\varphi,n/m}
      \right|
    =
    O\!\left(\frac{e^{Pn/2}}{n}\right),
  \]
  because \(n/m\le n/2\) for \(m>1\). Hence
  \[
    \pi_{n,\varrho}(\varphi)
    =O\!\left(\frac{\Theta_\varphi^n}{n}\right)
    \qquad (\varrho\neq\mathbf 1).
  \]
  This proves the absolute convergence in part~\textup{(ii)} after
  multiplication by \(e^{-Pn}\).

  For the trivial representation,
  \[
    p_n(\varphi)=\pi_{n,\mathbf 1}(\varphi)
    =
    \frac{e^{Pn}}{n}+O\!\left(\frac{\Theta_\varphi^n}{n}\right).
  \]
  Therefore
  \[
    e^{-Pn}p_n(\varphi)
    =
    \frac{1}{n}+O\!\left(\frac{(\Theta_\varphi e^{-P})^n}{n}\right),
  \]
  Hence the defining series for \(\mathcal M(\varphi)\) converges
  absolutely. Summing this estimate and applying
  \Cref{lem:geometric-harmonic-tail} gives
  \[
    \sum_{n\le N}e^{-Pn}p_n(\varphi)
    =
    H_N+\mathcal M(\varphi)-\gamma
    +O\!\left(\frac{(\Theta_\varphi e^{-P})^N}{N}\right).
  \]

  The weighted class decomposition is
  \[
    p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho(C^{-1})\pi_{n,\varrho}(\varphi),
  \]
  by the same argument as in
  \Cref{thm:class-primitive-decomposition}. The trivial channel gives
  the main term \((|C|/|G|)e^{Pn}/n\), while the non-trivial channels are
  \(O(\Theta_\varphi^n/n)\), proving
  \eqref{eq:holder-transfer-pointwise}. Summing after multiplication by
  \(e^{-Pn}\) gives
  \[
    \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{n\le N}e^{-Pn}p_n(\varphi)
    +\frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}\chi_\varrho(C^{-1})
        \sum_{n\le N}\pi_{n,\varrho}(\varphi)e^{-Pn}.
  \]
  Part~\textup{(ii)} and \Cref{lem:geometric-harmonic-tail} imply
  \[
    \sum_{n\le N}\pi_{n,\varrho}(\varphi)e^{-Pn}
    =
    \Pi_{\varrho,\varphi}(e^{-P})
    +O\!\left(\frac{(\Theta_\varphi e^{-P})^N}{N}\right)
  \]
  for each non-trivial \(\varrho\). Substituting the scalar asymptotic
  and collecting the constant term gives
  \eqref{eq:holder-transfer-mertens} and
  \eqref{eq:holder-transfer-constant}. Since
  \(H_N=\log N+\gamma+O(N^{-1})\), the final logarithmic form follows.

  Finally,
  \[
    \Delta_C(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho(C^{-1})\Pi_{\varrho,\varphi}(e^{-P})
  \]
  by \eqref{eq:holder-transfer-constant}. Multiplying by
  \(\chi_\sigma(C)\), summing over conjugacy classes, and invoking
  character orthogonality yields
  \eqref{eq:holder-transfer-fourier}.
\end{proof}

The next theorem isolates the higher spectral windows of the twisted
Ruelle operators in the same manner as
\Cref{thm:class-mertens-secondary-term} does in the finite-state
setting.

\begin{theorem}[H\"older spectral-window expansion above the square-root barrier]\textbf{[New-3 | 2026-04-17]}
  \label{thm:holder-transfer-spectral-window}
  Assume that
  \[
    \eta_\varphi
    :=
    \max_{\varrho\neq\mathbf 1}\rho(\mathscr L_{\varphi,\varrho})
    <e^P.
  \]
  Let
  \[
    r_{\mathrm{ess}}^\ast
    :=
    \max_{\varrho\in\Irr(G)}
      r_{\mathrm{ess}}(\mathscr L_{\varphi,\varrho}),
  \]
  and fix \(R\) such that
  \[
    \max\{r_{\mathrm{ess}}^\ast,e^{P/2}\}<R<e^P,
  \]
  with the additional property that no eigenvalue of any
  \(\mathscr L_{\varphi,\varrho}\) has modulus \(R\).
  For the trivial channel set
  \[
    \mathcal E_{\mathbf 1}(R)
    :=
    \{\beta\in\spec(\mathscr L_{\varphi,\mathbf 1}):R<|\beta|<e^P\},
  \]
  and for every non-trivial \(\varrho\in\Irr(G)\) set
  \[
    \mathcal E_\varrho(R)
    :=
    \{\beta\in\spec(\mathscr L_{\varphi,\varrho}):|\beta|>R\}.
  \]
  For \(\beta\in\mathcal E_\varrho(R)\), let
  \(\mathscr P_{\varrho,\beta}\) be the corresponding Riesz projector,
  and define, for each conjugacy class \(C\subset G\),
  \begin{equation}\label{eq:holder-transfer-spectral-window-profile}
    \mathfrak s_{C,\varphi,R}(n)
    :=
    \frac{|C|}{|G|}
    \left(
      \sum_{\beta\in\mathcal E_{\mathbf 1}(R)}
        \Tr\bigl(\mathscr L_{\varphi,\mathbf 1}^n
          \mathscr P_{\mathbf 1,\beta}\bigr)
      +
      \sum_{\varrho\neq\mathbf 1}\chi_\varrho(C^{-1})
        \sum_{\beta\in\mathcal E_\varrho(R)}
          \Tr\bigl(\mathscr L_{\varphi,\varrho}^n
            \mathscr P_{\varrho,\beta}\bigr)
    \right).
  \end{equation}
  Then there exists \(\Gamma_{\varphi,R}\in(0,R)\) such that, for every
  conjugacy class \(C\subset G\),
  \begin{equation}\label{eq:holder-transfer-spectral-window-pointwise}
    p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}\frac{e^{Pn}}{n}
    +\frac{\mathfrak s_{C,\varphi,R}(n)}{n}
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right).
  \end{equation}
  Moreover, the series
  \begin{equation}\label{eq:holder-transfer-spectral-window-tail}
    \mathfrak T_{C,\varphi,R}(N)
    :=
    \sum_{n>N}
      \frac{\mathfrak s_{C,\varphi,R}(n)}{n\,e^{Pn}}
  \end{equation}
  converges absolutely, and
  \begin{equation}\label{eq:holder-transfer-spectral-window-harmonic}
    \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}H_N
    +\mathcal M_C(\varphi)-\frac{|C|}{|G|}\gamma
    -\mathfrak T_{C,\varphi,R}(N)
    +O\!\left(\frac{(\Gamma_{\varphi,R}e^{-P})^N}{N}\right).
  \end{equation}
  For every \(\varrho\in\Irr(G)\) and every
  \(\beta\in\mathcal E_\varrho(R)\),
  \[
    \Tr\bigl(\mathscr L_{\varphi,\varrho}^n\mathscr P_{\varrho,\beta}\bigr)
    =m_\varrho(\beta)\beta^n,
  \]
  where \(m_\varrho(\beta)\) denotes the algebraic multiplicity of
  \(\beta\).
\end{theorem}

\begin{proof}
  Standard dynamical determinant theory for H\"older Ruelle operators
  on mixing shifts of finite type implies that, for every
  \(\varrho\in\Irr(G)\), the determinant
  \[
    D_\varrho(z)
    :=
    \exp\!\bigl(-\mathcal L_{\chi_\varrho,\varphi}(z)\bigr)
  \]
  extends holomorphically to the disc
  \(|z|<r_{\mathrm{ess}}(\mathscr L_{\varphi,\varrho})^{-1}\), and that
  its zeros there are precisely the reciprocals of the eigenvalues of
  \(\mathscr L_{\varphi,\varrho}\), counted with algebraic multiplicity;
  see
  \cite{Ruelle1978Thermodynamic,Baladi2000PositiveTransfer,%
  Baladi2018ZetaDeterminants,AdachiSunada1987TwistedPerronFrobenius}.
  Since \(R>r_{\mathrm{ess}}^\ast\), each set
  \(\mathcal E_\varrho(R)\) is finite. The absence of eigenvalues on
  the circle \(|z|=R\), together with the fact that isolated
  eigenvalues can accumulate only at the essential spectrum, therefore
  permits the choice of a number
  \[
    \Gamma_{\varphi,R}\in(\max\{r_{\mathrm{ess}}^\ast,e^{P/2}\},R)
  \]
  such that no eigenvalue of any \(\mathscr L_{\varphi,\varrho}\) has
  modulus in the interval \([\Gamma_{\varphi,R},R]\).

  For each \(\varrho\in\Irr(G)\), determinant factorisation on the disc
  \(|z|<\Gamma_{\varphi,R}^{-1}\) gives
  \[
    \mathcal L_{\chi_\varrho,\varphi}(z)
    =
    \delta_{\varrho,\mathbf 1}\log(1-e^Pz)^{-1}
    +
    \sum_{\beta\in\mathcal E_\varrho(R)}
      m_\varrho(\beta)\log(1-\beta z)^{-1}
    +H_{\varrho,R}(z),
  \]
  where \(H_{\varrho,R}\) is holomorphic on
  \(|z|<\Gamma_{\varphi,R}^{-1}\); for the trivial channel,
  \(e^P\notin\mathcal E_{\mathbf 1}(R)\) by definition. Writing
  \[
    H_{\varrho,R}(z)=\sum_{n\ge 1}\frac{h_{\varrho,R}(n)}{n}z^n,
  \]
  Cauchy's estimates yield \(h_{\varrho,R}(n)=O(\Gamma_{\varphi,R}^n)\),
  uniformly in \(\varrho\), because \(\Irr(G)\) is finite. Comparing
  coefficients therefore gives
  \begin{equation}\label{eq:holder-transfer-spectral-window-traces}
    T_{\chi_\varrho,\varphi,n}
    =
    \delta_{\varrho,\mathbf 1}e^{Pn}
    +
    \sum_{\beta\in\mathcal E_\varrho(R)}
      m_\varrho(\beta)\beta^n
    +O(\Gamma_{\varphi,R}^n).
  \end{equation}
  If \(\mathscr P_{\varrho,\beta}\) is the Riesz projector for
  \(\beta\), then the restriction of \(\mathscr L_{\varphi,\varrho}\)
  to the finite-dimensional space
  \(\mathscr P_{\varrho,\beta}C^\alpha(X_A^+,V_\varrho)\) has the form
  \(\beta I+N\), with \(N\) nilpotent. Consequently,
  \[
    \Tr\bigl(\mathscr L_{\varphi,\varrho}^n\mathscr P_{\varrho,\beta}\bigr)
    =
    \Tr\bigl((\beta I+N)^n\bigr)
    =m_\varrho(\beta)\beta^n.
  \]

  Coefficient extraction in
  \eqref{eq:holder-transfer-adams-mobius}, applied to
  \(\phi=\chi_\varrho\) and the Adams decomposition
  \eqref{eq:adams-character-decomposition}, yields
  \[
    \pi_{n,\varrho}(\varphi)
    =
    \frac{1}{n}\sum_{m\mid n}\mu(m)
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}T_{\chi_\sigma,\varphi,n/m}.
  \]
  For the trivial representation, the divisor \(m=1\) contributes
  \[
    \frac{e^{Pn}}{n}
    +\frac{1}{n}
      \sum_{\beta\in\mathcal E_{\mathbf 1}(R)}
        m_{\mathbf 1}(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right)
  \]
  by \eqref{eq:holder-transfer-spectral-window-traces}. For \(m>1\),
  \Cref{lem:adams-coefficients-bounded} and the uniform estimate
  \(T_{\chi_\sigma,\varphi,\ell}=O(e^{P\ell})\) from the proof of
  \Cref{thm:holder-transfer-perron-point} give
  \[
    \frac{1}{n}
    \sum_{\substack{m\mid n\\ m>1}}
      \left|
        \sum_{\sigma\in\Irr(G)}
          a_{\mathbf 1,\sigma}^{(m)}
          T_{\chi_\sigma,\varphi,n/m}
      \right|
    =
    O\!\left(\frac{e^{Pn/2}}{n}\right)
    =
    O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right),
  \]
  because \(\Gamma_{\varphi,R}>e^{P/2}\). Hence
  \begin{equation}\label{eq:holder-transfer-spectral-window-trivial}
    p_n(\varphi)
    =
    \frac{e^{Pn}}{n}
    +\frac{1}{n}
      \sum_{\beta\in\mathcal E_{\mathbf 1}(R)}
        m_{\mathbf 1}(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right).
  \end{equation}

  For a fixed non-trivial \(\varrho\), the divisor \(m=1\) contributes
  \[
    \frac{1}{n}
      \sum_{\beta\in\mathcal E_\varrho(R)}
        m_\varrho(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right),
  \]
  again by \eqref{eq:holder-transfer-spectral-window-traces}, while the
  divisors \(m>1\) satisfy the same
  \(O(\Gamma_{\varphi,R}^n/n)\) bound as above. Therefore
  \begin{equation}\label{eq:holder-transfer-spectral-window-nontrivial}
    \pi_{n,\varrho}(\varphi)
    =
    \frac{1}{n}
      \sum_{\beta\in\mathcal E_\varrho(R)}
        m_\varrho(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right)
    \qquad (\varrho\neq\mathbf 1).
  \end{equation}

  The weighted class decomposition
  \[
    p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho(C^{-1})\pi_{n,\varrho}(\varphi)
  \]
  now yields \eqref{eq:holder-transfer-spectral-window-pointwise} after
  substituting
  \eqref{eq:holder-transfer-spectral-window-trivial} and
  \eqref{eq:holder-transfer-spectral-window-nontrivial}. The error term
  is uniform in \(C\) because the set of conjugacy classes is finite and
  \(|\chi_\varrho(C^{-1})|\le \dim\varrho\).

  Dividing \eqref{eq:holder-transfer-spectral-window-pointwise} by
  \(e^{Pn}\) gives
  \[
    e^{-Pn}p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}\frac{1}{n}
    +\frac{\mathfrak s_{C,\varphi,R}(n)}{n\,e^{Pn}}
    +O\!\left(\frac{(\Gamma_{\varphi,R}e^{-P})^n}{n}\right).
  \]
  Since every \(\beta\in\mathcal E_\varrho(R)\) satisfies
  \(|\beta|<e^P\), the series
  \eqref{eq:holder-transfer-spectral-window-tail} converges absolutely.
  Summing the preceding estimate over \(n>N\) and applying
  \Cref{lem:geometric-harmonic-tail} gives
  \[
    \sum_{n>N}
      \left(
        e^{-Pn}p_{n,C}(\varphi)
        -\frac{|C|}{|G|}\frac{1}{n}
      \right)
    =
    \mathfrak T_{C,\varphi,R}(N)
    +O\!\left(\frac{(\Gamma_{\varphi,R}e^{-P})^N}{N}\right).
  \]
  By \Cref{thm:holder-transfer-perron-point},
  \[
    \mathcal M_C(\varphi)-\frac{|C|}{|G|}\gamma
    =
    \sum_{n\ge 1}
      \left(
        e^{-Pn}p_{n,C}(\varphi)
        -\frac{|C|}{|G|}\frac{1}{n}
      \right).
  \]
  Subtracting the tail beyond \(N\) proves
  \eqref{eq:holder-transfer-spectral-window-harmonic}.
\end{proof}

\begin{theorem}[Pressure-boundary rigidity for twisted Ruelle operators]
  \label{thm:holder-transfer-boundary-rigidity}
  Let \(h_\varphi\in C^\alpha(X_A^+)\) be the positive eigenfunction of
  \(\mathscr L_{\varphi,\mathbf 1}\) satisfying
  \[
    \mathscr L_{\varphi,\mathbf 1}h_\varphi=e^P h_\varphi.
  \]
  For an irreducible unitary representation \(\varrho\in\Irr(G)\), the
  following are equivalent.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item \(\rho(\mathscr L_{\varphi,\varrho})=e^P\).
    \item There exist \(\omega\in\TT\) and a H\"older map
      \(\xi:X_A^+\to V_\varrho\) with \(\|\xi(x)\|=1\) for every
      \(x\in X_A^+\) such that
      \begin{equation}\label{eq:holder-transfer-line-field}
        \varrho(\tau(y))\xi(y)=\omega\,\xi(x)
        \qquad (\sigma y=x).
      \end{equation}
  \end{enumerate}
  Whenever these conditions hold, every periodic point
  \(x\in\operatorname{Fix}(\sigma^n)\) satisfies
  \begin{equation}\label{eq:holder-transfer-periodic-line}
    \varrho(\tau_n(x))\xi(x)=\omega^n\xi(x).
  \end{equation}
\end{theorem}

\begin{proof}
  The weighted norm
  \[
    \|F\|_{h_\varphi}
    :=
    \sup_{x\in X_A^+}\frac{\|F(x)\|}{h_\varphi(x)}
  \]
  satisfies
  \[
    \frac{\|(\mathscr L_{\varphi,\varrho}F)(x)\|}{h_\varphi(x)}
    \le
    \sum_{\sigma y=x}
      e^{\varphi(y)}\frac{h_\varphi(y)}{h_\varphi(x)}
      \frac{\|F(y)\|}{h_\varphi(y)}
    \le
    e^P\|F\|_{h_\varphi},
  \]
  because \(\mathscr L_{\varphi,\mathbf 1}h_\varphi=e^P h_\varphi\).
  Consequently
  \[
    \rho(\mathscr L_{\varphi,\varrho})\le e^P
    \qquad (\varrho\in\Irr(G)).
  \]

  Assume first that \(\rho(\mathscr L_{\varphi,\varrho})=e^P\). The
  operator \(\mathscr L_{\varphi,\varrho}\) is quasi-compact on
  \(C^\alpha(X_A^+,V_\varrho)\), so some eigenvalue
  \(\alpha\in\spec(\mathscr L_{\varphi,\varrho})\) satisfies
  \(|\alpha|=e^P\). Let \(H\neq 0\) be a corresponding H\"older
  eigenfunction:
  \[
    \mathscr L_{\varphi,\varrho}H=\alpha H.
  \]
  Set
  \[
    q(x):=\frac{\|H(x)\|}{h_\varphi(x)},
    \qquad
    M:=\max_{x\in X_A^+}q(x)>0.
  \]
  For every \(x\in X_A^+\),
  \[
    e^P q(x)
    =
    \frac{|\alpha|\,\|H(x)\|}{h_\varphi(x)}
    \le
    \sum_{\sigma y=x}
      e^{\varphi(y)}\frac{\|H(y)\|}{h_\varphi(x)}
    \le
    M\sum_{\sigma y=x}
      e^{\varphi(y)}\frac{h_\varphi(y)}{h_\varphi(x)}
    =e^P M.
  \]
  Hence \(q(x)\le M\) for all \(x\). Let
  \(E:=\{x\in X_A^+:q(x)=M\}\). If \(x\in E\), then the last chain of
  inequalities is an equality, so every predecessor \(y\) of \(x\)
  satisfies \(q(y)=M\). Thus \(E\) is non-empty, closed, and backward
  invariant.

  Fix \(x\in E\). Let \([a_0,\dots,a_{k-1}]\) be any non-empty cylinder.
  Primitivity of \(A\) supplies, for all sufficiently large \(n\), an
  admissible word of length \(n-k\) connecting \(a_{k-1}\) to the first
  symbol of \(x\). Appending the tail of \(x\) therefore produces
  \(y\in[a_0,\dots,a_{k-1}]\) with \(\sigma^n y=x\). Backward invariance
  gives \(y\in E\). Every cylinder thus meets \(E\), so \(E=X_A^+\).
  Therefore
  \[
    \|H(x)\|=M h_\varphi(x)
    \qquad (x\in X_A^+),
  \]
  and in particular \(H(x)\neq 0\) everywhere.

  Write \(\omega:=\alpha e^{-P}\in\TT\) and
  \[
    \xi(x):=\frac{H(x)}{M h_\varphi(x)}.
  \]
  Then \(\xi\) is H\"older and \(\|\xi(x)\|=1\). Since equality holds in
  the triangle inequality at every \(x\), all vectors
  \(\varrho(\tau(y))H(y)\) with \(\sigma y=x\) lie on the ray spanned by
  \(\alpha H(x)\). Their norms are
  \[
    \|\varrho(\tau(y))H(y)\|
    =\|H(y)\|
    =M h_\varphi(y),
  \]
  and the sum of these norms equals
  \[
    \sum_{\sigma y=x}e^{\varphi(y)}M h_\varphi(y)
    =e^P M h_\varphi(x).
  \]
  Consequently each summand is exactly
  \(e^{\varphi(y)}M h_\varphi(y)\omega\,\xi(x)\), which gives
  \eqref{eq:holder-transfer-line-field}.

  Conversely, assume that \eqref{eq:holder-transfer-line-field} holds.
  Define \(H:=h_\varphi\xi\). Then
  \begin{align*}
    (\mathscr L_{\varphi,\varrho}H)(x)
    &=
    \sum_{\sigma y=x}
      e^{\varphi(y)}h_\varphi(y)\varrho(\tau(y))\xi(y) \\
    &=
    \omega\,\xi(x)\sum_{\sigma y=x}e^{\varphi(y)}h_\varphi(y) \\
    &=
    \omega e^P h_\varphi(x)\xi(x)
    =\omega e^P H(x).
  \end{align*}
  Therefore \(\omega e^P\in\spec(\mathscr L_{\varphi,\varrho})\), so
  \(\rho(\mathscr L_{\varphi,\varrho})\ge e^P\). Together with the
  general upper bound, this proves
  \(\rho(\mathscr L_{\varphi,\varrho})=e^P\).

  Finally, if \(x\in\operatorname{Fix}(\sigma^n)\), repeated application
  of \eqref{eq:holder-transfer-line-field} along the orbit of \(x\)
  yields \eqref{eq:holder-transfer-periodic-line}.
\end{proof}
