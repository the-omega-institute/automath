\section{Scan--projection systems}
\label{sec:scan-projection}

\begin{definition}[Binary scan--projection system]
\label{def:binary-spg}
A binary scan--projection system is a tuple
\[
(X,\mathcal B,\mu,T,W),
\]
where \((X,\mathcal B,\mu)\) is a probability space, \(T:X\to X\) is a measurable map, and \(W\in\mathcal B\) is a binary observation window.  The scan readout is
\[
s_t(x)=\one_W(T^t x),
\]
and the length-\(m\) window word is
\[
\omega_m(x;t)=s_t(x)s_{t+1}(x)\cdots s_{t+m-1}(x)\in\Om_m=\{0,1\}^m.
\]
\end{definition}

For a word \(a=a_0\cdots a_{m-1}\in\Om_m\), the corresponding cylinder event is
\[
C_a(t)=\{x\in X:s_{t+j}(x)=a_j,\ 0\le j<m\}.
\]
The finite visible sigma-algebra at depth \(m\) is
\[
\mathcal F_{m,t}=\sigma\{C_a(t):a\in\Om_m\}.
\]
When the origin is fixed at \(t=0\), we write \(\mathcal F_m\).

The infinite readout map is
\[
\omega_\infty(x)=s_0(x)s_1(x)s_2(x)\cdots\in\Om_\infty=\{0,1\}^{\NN}.
\]
For \(\xi,\eta\in\Om_\infty\), define the common prefix length
\[
\ell(\xi,\eta)=\sup\{k:\xi_0\cdots \xi_{k-1}=\eta_0\cdots\eta_{k-1}\},
\]
and the prefix ultrametric
\[
d_{\rm pre}(\xi,\eta)=2^{-\ell(\xi,\eta)}.
\]

\begin{theorem}[Scan--projection interface]
\label{thm:scan-projection-interface}
For every \(m\), the length-\(m\) readout map \(x\mapsto\omega_m(x;0)\) factors all depth-\(m\) visible events through \(\Om_m\).  Equivalently, every \(A\in\mathcal F_m\) is a union of length-\(m\) cylinders.  Moreover,
\[
\mathcal F_1\subseteq \mathcal F_2\subseteq \cdots \subseteq \mathcal F_\infty
\]
whenever the scan is taken from a fixed origin.
\end{theorem}

\begin{proof}
The atoms of \(\mathcal F_m\) are the fibers of the map \(x\mapsto\omega_m(x;0)\).  Refining from \(m\) to \(m+1\) appends one coordinate, so each length-\(m\) cylinder is a union of length-\((m+1)\) cylinders.
\end{proof}

\begin{remark}
\Cref{thm:scan-projection-interface} defines only the observable input layer.  It does not assume stable types, arithmetic, Hilbert structure, spacetime geometry, or any physical interpretation.
\end{remark}
