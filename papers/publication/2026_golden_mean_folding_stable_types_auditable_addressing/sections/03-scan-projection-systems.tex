\section{Scan--projection systems}
\label{sec:scan-projection}

Let \((X,\mathcal B)\) be a measurable space, let
\[
T:X\to X
\]
be a measurable transformation, and let \(W\in\mathcal B\) be a measurable binary observation window.  The associated binary readout is
\[
s_t(x):=\one_W(T^t x)\in\{0,1\}.
\]
For a resolution \(m\ge1\), the length-\(m\) readout word at origin \(t\) is
\[
\omega_m(x;t):=
s_t(x)s_{t+1}(x)\cdots s_{t+m-1}(x)\in\Om_m,
\qquad
\Om_m:=\{0,1\}^m.
\]
Thus \(\Om_m\) is the raw microstate space visible at resolution \(m\), and
\[
|\Om_m|=2^m.
\]

For a word \(a=a_1\cdots a_m\in\Om_m\), define the cylinder event
\[
C_a(t):=\{x\in X:\omega_m(x;t)=a\}.
\]
Equivalently,
\[
C_a(t)=\bigcap_{j=1}^{m}T^{-(t+j-1)}W_{a_j},
\qquad
W_1=W,\quad W_0=X\setminus W.
\]
The finite visible sigma-algebra at resolution \(m\) and origin \(t\) is
\[
\mathcal F_m(t):=\sigma(C_a(t):a\in\Om_m).
\]
When \(t=0\) is fixed, we write \(\mathcal F_m:=\mathcal F_m(0)\).

The infinite readout map is
\[
\omega_\infty(x)=s_0(x)s_1(x)s_2(x)\cdots\in\Om_\infty=\{0,1\}^{\NN}.
\]
For \(\xi,\eta\in\Om_\infty\), define the common prefix length
\[
\ell(\xi,\eta)=\sup\{k\ge0:\xi_1\cdots\xi_k=\eta_1\cdots\eta_k\},
\]
with \(\ell=\infty\) when \(\xi=\eta\), and define the prefix ultrametric
\[
d_{\rm pre}(\xi,\eta)=2^{-\ell(\xi,\eta)},
\qquad 2^{-\infty}:=0.
\]
For \(a\in\Om_m\), the symbolic cylinder
\[
[a]=\{\xi\in\Om_\infty:\xi_1\cdots\xi_m=a\}
\]
is a clopen ball in this ultrametric, and
\[
C_a(0)=\omega_\infty^{-1}([a]).
\]

\begin{theorem}[Scan--projection cylinders]
\label{thm:scan-projection-interface}
Let \((X,\mathcal B)\) be a measurable space, \(T:X\to X\) measurable, and \(W\in\mathcal B\).  Then for every \(m\ge1\), \(t\ge0\), and \(a\in\Om_m\), the cylinder \(C_a(t)\) is measurable.  Consequently,
\[
\mathcal F_m(t)=\sigma(C_a(t):a\in\Om_m)
\]
is a finite sigma-algebra.  If \(t=0\) is fixed, then
\[
\mathcal F_m\subseteq\mathcal F_{m+1}.
\]
\end{theorem}

\begin{proof}
For \(a=a_1\cdots a_m\),
\[
C_a(t)=\bigcap_{j=1}^{m}T^{-(t+j-1)}W_{a_j}.
\]
Each factor is measurable, hence the finite intersection is measurable.  Since \(\Om_m\) is finite, \(\mathcal F_m(t)\) is finite.  Finally,
\[
C_a(0)=C_{a0}(0)\cup C_{a1}(0),
\]
so every generator of \(\mathcal F_m\) lies in \(\mathcal F_{m+1}\).  Hence \(\mathcal F_m\subseteq\mathcal F_{m+1}\).
\end{proof}

\begin{remark}
\Cref{thm:scan-projection-interface} establishes only the finite observable input layer.  It uses measurability of \(T\) and \(W\); invariance or ergodicity of \(\mu\) is needed only for later statistical or asymptotic claims.
\end{remark}
