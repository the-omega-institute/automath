\section{Scan error profiles and clarity bounds}
\label{sec:clarity}

The previous sections define stable types and their cross-resolution compatibility.  We now quantify how well a finite scan resolves an event.

Assume \((X,\mathcal B,\mu)\) is a probability space and \(\mathcal F_m\subseteq\mathcal B\) is the finite visible sigma-algebra generated by resolution-\(m\) cylinders.  For \(P\in\mathcal B\), define the resolution-\(m\) scan error by
\[
\boxed{
\varepsilon_m(P;\mu):=
\inf_{g\in L^\infty(\mathcal F_m),\ g(X)\subseteq\{0,1\}}
\mu(g\ne \one_P).
}
\]
This is the Bayes risk among all binary decisions that only use information visible at resolution \(m\).

Let \(\mathcal C_m\) be the finite atom partition of \(\mathcal F_m\).  For an atom \(C\in\mathcal C_m\), let
\[
p_C:=\mu(P\mid C)
\]
whenever \(\mu(C)>0\).  Atoms of measure zero do not affect the formulas below.

\begin{theorem}[Cylinder decomposition of scan error]
\label{prop:cylinder-decomposition}
For every \(P\in\mathcal B\),
\[
\varepsilon_m(P;\mu)=
\sum_{C\in\mathcal C_m}
\min\{\mu(P\cap C),\mu(C\setminus P)\}.
\]
Equivalently,
\[
\varepsilon_m(P;\mu)=
\sum_{C\in\mathcal C_m}\mu(C)\min\{p_C,1-p_C\}.
\]
\end{theorem}

\begin{proof}
Every \(\{0,1\}\)-valued \(\mathcal F_m\)-measurable function is constant on each atom \(C\in\mathcal C_m\).  On a fixed atom \(C\), there are only two decisions.  If the decision is \(1\), the local error is \(\mu(C\setminus P)\).  If the decision is \(0\), the local error is \(\mu(P\cap C)\).  Therefore the optimal local error is
\[
\min\{\mu(P\cap C),\mu(C\setminus P)\}.
\]
Summing over the disjoint atom partition gives the first formula.  If \(\mu(C)>0\), then \(\mu(P\cap C)=\mu(C)p_C\) and \(\mu(C\setminus P)=\mu(C)(1-p_C)\), giving the second formula.
\end{proof}

\begin{theorem}[Bayes-optimal prefix classifier]
\label{thm:bayes-prefix-classifier}
Let
\[
p_m(x):=\mu(P\mid\mathcal F_m)(x)
\]
and define
\[
P_m^\star:=\{x:p_m(x)\ge 1/2\}.
\]
Then \(P_m^\star\in\mathcal F_m\) and
\[
\mu(P\triangle P_m^\star)=\varepsilon_m(P;\mu).
\]
Moreover, for every \(Q\in\mathcal F_m\),
\[
\mu(P\triangle Q)\ge \mu(P\triangle P_m^\star).
\]
\end{theorem}

\begin{proof}
The conditional probability \(p_m\) is constant on each atom \(C\in\mathcal C_m\), with value \(p_C\).  On \(C\), the decision \(1\) is optimal exactly when \(\mu(C\setminus P)\le \mu(P\cap C)\), equivalently \(p_C\ge1/2\).  Thus \(P_m^\star\) chooses the locally optimal value on every atom.  \Cref{prop:cylinder-decomposition} then gives \(\mu(P\triangle P_m^\star)=\varepsilon_m(P;\mu)\).  Any other \(Q\in\mathcal F_m\) makes a binary choice on the same atoms and cannot improve the local optimum.
\end{proof}

\begin{corollary}[Observable stability]
\label{cor:observable-stability}
For every bounded measurable \(\varphi:X\to\RR\),
\[
\left|
\int \varphi\one_P\,d\mu
-
\int \varphi\one_{P_m^\star}\,d\mu
\right|
\le
\|\varphi\|_\infty\,\varepsilon_m(P;\mu).
\]
\end{corollary}

\begin{proof}
Since \(\one_P-\one_{P_m^\star}\) is supported on \(P\triangle P_m^\star\),
\[
\left|\int \varphi(\one_P-\one_{P_m^\star})\,d\mu\right|
\le
\|\varphi\|_\infty\mu(P\triangle P_m^\star).
\]
Apply \Cref{thm:bayes-prefix-classifier}.
\end{proof}

\begin{proposition}[Monotonicity]
\label{prop:scan-error-monotonicity}
If \(\mathcal F_m\subseteq\mathcal F_{m+1}\), then
\[
\varepsilon_{m+1}(P;\mu)\le \varepsilon_m(P;\mu).
\]
Moreover, \(\varepsilon_m(P;\mu)=0\) if and only if \(P\) agrees mod null sets with an \(\mathcal F_m\)-measurable event.
\end{proposition}

\begin{proof}
Every resolution-\(m\) binary decision rule is also admissible at resolution \(m+1\), so the infimum cannot increase.  If \(P\) is \(\mathcal F_m\)-measurable mod null sets, choosing \(g=\one_P\) gives zero error.  Conversely, zero error means that an \(\mathcal F_m\)-measurable binary rule agrees with \(\one_P\) almost everywhere.
\end{proof}

\begin{definition}[Boundary cylinders]
\label{def:boundary-cylinders}
The depth-\(m\) boundary cylinder set of \(P\) is
\[
\partial_mP:=
\{C\in\mathcal C_m:\mu(P\cap C)>0\text{ and }\mu(C\setminus P)>0\}.
\]
\end{definition}

\begin{theorem}[Boundary-cylinder clarity bound]
\label{thm:boundary-clarity}
Assume that for constants \(K<\infty\) and \(\lambda_{\rm cyl}>1\),
\[
\mu(C)\le K\lambda_{\rm cyl}^{-m}
\qquad(C\in\mathcal C_m).
\]
Then
\[
\varepsilon_m(P;\mu)\le K\lambda_{\rm cyl}^{-m}|\partial_mP|.
\]
If
\[
|\partial_mP|\le \lambda_{\rm cyl}^{dm+o(m)}
\]
with \(d<1\), then
\[
\varepsilon_m(P;\mu)
\le
K\lambda_{\rm cyl}^{-(1-d)m+o(m)}.
\]
\end{theorem}

\begin{proof}
Only atoms in \(\partial_mP\) contribute to \Cref{prop:cylinder-decomposition}.  For each such atom,
\[
\min\{\mu(P\cap C),\mu(C\setminus P)\}\le \mu(C)\le K\lambda_{\rm cyl}^{-m}.
\]
Summing over \(\partial_mP\) gives the first bound.  Substituting the assumed growth of \(|\partial_mP|\) gives the second.
\end{proof}

\begin{theorem}[Almost-sure eventual clarity under summable scan error]
\label{thm:eventual-clarity}
If
\[
\sum_{m=1}^{\infty}\varepsilon_m(P;\mu)<\infty,
\]
then \(1_{P_m^\star}(x)\to 1_P(x)\) for \(\mu\)-almost every \(x\).  More precisely, for almost every \(x\), there exists \(m_0(x)\) such that for all \(m\ge m_0(x)\),
\[
1_{P_m^\star}(x)=1_P(x).
\]
\end{theorem}

\begin{proof}
Let \(E_m=P\triangle P_m^\star\).  By \Cref{thm:bayes-prefix-classifier},
\[
\mu(E_m)=\varepsilon_m(P;\mu).
\]
Thus \(\sum_m\mu(E_m)<\infty\).  The Borel--Cantelli lemma gives \(\mu(E_m\text{ infinitely often})=0\).  Hence for almost every \(x\), only finitely many \(E_m\) occur, which is exactly the claimed eventual equality.
\end{proof}
