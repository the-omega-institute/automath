\section{Scan error profiles and clarity}
\label{sec:clarity}

The phrase ``deeper scans are clearer'' should be made quantitative.  We use optimal prefix classification error.

\begin{definition}[Scan error profile]
\label{def:scan-error-profile}
Let \(P\in\mathcal F_\infty\).  The depth-\(m\) scan error is
\[
\varepsilon_m(P;\mu)=
\inf\{\mu(g\ne \one_P):g\in L^\infty(\mathcal F_m),\ g(X)\subseteq\{0,1\}\}.
\]
\end{definition}

Let \(\mathcal C_m\) be the set of depth-\(m\) cylinders.

\begin{proposition}[Cylinder decomposition]
\label{prop:cylinder-decomposition}
For every \(P\in\mathcal F_\infty\),
\[
\varepsilon_m(P;\mu)=
\sum_{C\in\mathcal C_m}
\min\{\mu(P\cap C),\mu(C\setminus P)\}.
\]
The minimizer is the Bayes prefix classifier
\[
P_m^\star=\{x:\mu(P\mid\mathcal F_m)(x)\ge 1/2\}.
\]
\end{proposition}

\begin{proof}
On each cylinder \(C\), any \(\mathcal F_m\)-measurable classifier is constant.  The smaller of the two masses \(\mu(P\cap C)\) and \(\mu(C\setminus P)\) is the unavoidable error on \(C\).  Summing over all cylinders gives the formula.
\end{proof}

\begin{definition}[Boundary cylinders]
\label{def:boundary-cylinders}
The depth-\(m\) boundary of \(P\) is the set
\[
\partial_m P=\{C\in\mathcal C_m:\mu(P\cap C)>0\ \text{and}\ \mu(C\setminus P)>0\}.
\]
Let \(N_m(\partial P)=|\partial_mP|\).
\end{definition}

\begin{theorem}[Scan error profile and clarity bound]
\label{thm:boundary-clarity}
Assume every nonempty \(m\)-cylinder satisfies
\[
\mu(C)\le K\lambda_{\mathrm{cyl}}^{-m},\qquad \lambda_{\mathrm{cyl}}>1.
\]
Then
\[
\varepsilon_m(P;\mu)\le \frac K2\lambda_{\mathrm{cyl}}^{-m}N_m(\partial P).
\]
If
\[
N_m(\partial P)=\lambda_{\mathrm{cyl}}^{(d+o(1))m}
\quad\text{with }d<1,
\]
then
\[
\varepsilon_m(P;\mu)\le
\lambda_{\mathrm{cyl}}^{-(1-d+o(1))m}.
\]
\end{theorem}

\begin{proof}
By \Cref{prop:cylinder-decomposition}, only boundary cylinders contribute and each contributes at most \(\mu(C)/2\).  Apply the cylinder mass bound and substitute the assumed boundary growth.
\end{proof}

\begin{remark}
\Cref{thm:boundary-clarity} is one of the main quantitative results suitable for the first submission.  It translates boundary complexity into an auditable scan-depth error profile.
\end{remark}
