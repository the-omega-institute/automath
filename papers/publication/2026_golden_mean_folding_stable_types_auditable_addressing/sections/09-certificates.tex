\section{Certificates and finite graph interfaces}
\label{sec:certificates}

The certificate layer records what can be checked from finite windows without turning auxiliary spectra or statistics into new main theorems.  It belongs to the \(C3\) audit layer of \Cref{def:claim-hierarchy}.

\begin{definition}[Finite certificate package]
\label{def:finite-certificate-package}
At depth \(m\), a certificate package may contain
\[
\Cert_m=(X_m,\Fold_m,r_{m+1,m},A_m,h_m,\pi_m^{\mathrm{base}},\widehat\pi_{m,N},\epsilon_{m,N,\delta},R_{m,N},H_m),
\]
where \(A_m\) is a finite graph or adjacency matrix for the stable-type presentation, \(h_m\) is an entropy or pressure statistic, \(\pi_m^{\mathrm{base}}\) is a baseline distribution, \(\widehat\pi_{m,N}\) is an empirical histogram, \(\epsilon_{m,N,\delta}\) is a finite-sample uncertainty radius, \(R_{m,N}\) is a cross-resolution residual, and \(H_m\) denotes optional moment or Hankel data.
\end{definition}

\begin{definition}[Finite graph presentation]
\label{def:finite-graph-presentation}
A finite graph presentation of \(X_m\) is an adjacency matrix \(A_m\) or labeled automaton whose accepted language is the stable type language at depth \(m\).  Its Perron--Frobenius eigenvalue, entropy, and Parry measure may be used as baseline certificate data.
\end{definition}

\begin{definition}[Cross-resolution residual]
\label{def:cross-resolution-residual}
Given \(r_{m+1,m}:X_{m+1}\to X_m\), define
\[
R_{m,N}=
\TV\bigl((r_{m+1,m})_\#\widehat\pi_{m+1,N},\widehat\pi_{m,N}\bigr).
\]
\end{definition}

\begin{definition}[Total-variation envelope]
\label{def:tv-envelope}
For a finite state space \(X_m\) and effective sample size \(N_{\mathrm{eff}}\), a conservative total-variation radius is
\[
\epsilon_{\mathrm{TV}}(m,N,\delta)=
\frac12\sqrt{\frac{2}{N_{\mathrm{eff}}}\left(|X_m|\log2+\log\frac1\delta\right)}.
\]
\end{definition}

\begin{proposition}[Finite-sample audit semantics]
\label{prop:finite-sample-audit-semantics}
If
\[
\TV(\widehat\pi_{m,N},\pi_m^{\mathrm{ref}})
>
\epsilon_{\mathrm{TV}}(m,N,\delta)+\tau,
\]
then the reference distribution \(\pi_m^{\mathrm{ref}}\) is rejected at the stated tolerance.  If the left side is below the uncertainty radius, the correct conclusion is non-rejection, not proof of the reference mechanism.
\end{proposition}

\begin{remark}[Baseline discipline]
The Parry/Perron--Frobenius distribution of a finite graph is a natural maximum-entropy or Markov baseline.  Agreement with it is not mechanism identification.  Mechanism identification requires additional statistics, holdout predictions, or identifiability assumptions.
\end{remark}

\begin{theorem}[Auditable stable type generator]
\label{thm:auditable-stable-type-generator}
The data
\[
(\Om_m,X_m,\Fold_m,r_{m+1,m},\mathcal F_m,\mathcal G_L,\varepsilon_m,\Cert_m)
\]
define a stable type generator with computable normal forms, cross-resolution compatibility, recursive addressing without visible sigma-algebra expansion, and statistically auditable clarity profiles.
\end{theorem}

\begin{proof}
Computable normal forms follow from \Cref{thm:golden-fold-normal-form}; cross-resolution compatibility follows from \Cref{thm:cross-resolution-compatibility}; address non-expansion follows from \Cref{thm:recursive-address-nonexpansion}; and clarity profiles follow from \Cref{prop:cylinder-decomposition,thm:boundary-clarity}.  The certificate package records these finite data and their statistical audit quantities.
\end{proof}
