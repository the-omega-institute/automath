\section{Auditable stable type generators and certificate interfaces}
\label{sec:certificates}

The certificate layer records what can be checked from finite windows without turning auxiliary spectra or statistics into new main theorems.  It belongs to the \(C3\) audit layer of \Cref{def:claim-hierarchy}.

\begin{definition}[Finite certificate package]
\label{def:finite-certificate-package}
At depth \(m\), a certificate package may contain
\[
\Cert_m=(X_m,\Fold_m,\pi_{m+1\to m},\widetilde\rho_{m+1,m},\mathcal K_m,A_m,h_m,\zeta_{\mathcal K_m},d_m,p_m^{\mathrm{fib}},q_m,\widehat\pi_{m,N},\epsilon_{m,N,\delta},R_{m,N},\kappa_m,H_m),
\]
where \(\mathcal K_m\) is the right-resolving Markov cover of the folded block factor \(Y_m\) constructed in \Cref{app:finite-state-audit}, \(A_m\) is its adjacency matrix, \(h_m=\log\rho(A_m)\) is the cover entropy certificate, \(\zeta_{\mathcal K_m}\) is the cover zeta certificate, \(d_m\) records fold-fiber multiplicities, \(p_m^{\mathrm{fib}}\) is the IID-uniform fiber law, \(q_m\) is the golden-mean Parry baseline, \(\widehat\pi_{m,N}\) is an empirical histogram, \(\epsilon_{m,N,\delta}\) is a finite-sample uncertainty radius, \(R_{m,N}\) is a cross-resolution residual, \(\kappa_m\) is the hidden-volume statistic, and \(H_m\) denotes optional moment or Hankel data.
\end{definition}

The deterministic fiber ledger
\[
(d_m,D_m,p_m^{\mathrm{fib}},\kappa_m,M_{m,q})
\]
is constructed in \Cref{app:fiber-ledger}.  These quantities are exact outputs of the fold table and are separated from empirical histogram estimates.

\begin{definition}[Finite-state cover certificate]
\label{def:finite-graph-presentation}
A finite-state cover certificate for the folded factor
\[
Y_m=\Phi_m(\{0,1\}^{\ZZ})
\]
is a right-resolving labeled Markov cover \(\mathcal K_m\), its adjacency matrix \(A_m\), its Perron--Frobenius eigenvalue \(\rho(A_m)\), and the associated entropy, Parry, pressure, or zeta data.  The construction is finite and reproducible by \Cref{def:reference-finite-state-audit}.  Cover-level objects are kept separate from intrinsic labeled-system invariants.
\end{definition}

\begin{definition}[Cross-resolution residual]
\label{def:cross-resolution-residual}
Given stable prefix restriction \(\pi_{m+1\to m}:X_{m+1}\to X_m\), define
\[
R_{m,N}=
\TV\bigl((\pi_{m+1\to m})_\#\widehat\pi_{m+1,N},\widehat\pi_{m,N}\bigr).
\]
\end{definition}

\begin{definition}[Total-variation envelope]
\label{def:tv-envelope}
For a finite state space \(X_m\) and effective sample size \(N_{\mathrm{eff}}\), a conservative total-variation radius is
\[
\epsilon_{\mathrm{TV}}(m,N,\delta)=
\frac12\sqrt{\frac{2}{N_{\mathrm{eff}}}\left(|X_m|\log2+\log\frac1\delta\right)}.
\]
For overlapping windows or Markov-dependent folded processes, the declared envelope should instead use an explicit dependence correction such as the Markov-TV protocol in \Cref{app:statistical-audit}.
\end{definition}

\begin{proposition}[Finite-sample audit semantics]
\label{prop:finite-sample-audit-semantics}
If
\[
\TV(\widehat\pi_{m,N},\pi_m^{\mathrm{ref}})
>
\epsilon_{\mathrm{TV}}(m,N,\delta)+\tau,
\]
then the reference distribution \(\pi_m^{\mathrm{ref}}\) is rejected at the stated tolerance.  If the left side is below the uncertainty radius, the correct conclusion is non-rejection, not proof of the reference mechanism.
\end{proposition}

\begin{remark}[Baseline discipline]
The Parry/Perron--Frobenius distribution of a finite cover is a natural maximum-entropy or Markov baseline.  The golden-mean cylinder baseline \(q_m\) is recorded in \Cref{app:golden-parry-baseline}.  Agreement with a Parry baseline is not mechanism identification.  Mechanism identification requires additional statistics, holdout predictions, or identifiability assumptions.
\end{remark}

\begin{corollary}[Auditable stable type generator]
\label{thm:auditable-stable-type-generator}
The data
\[
(\Om_m,X_m,\Fold_m,\pi_{m+1\to m},\widetilde\rho_{m+1,m},\mathcal F_m,\mathcal G_{\le L},\varepsilon_m,\Cert_m)
\]
define a stable type generator with computable normal forms, cross-resolution compatibility, recursive addressing without visible sigma-algebra expansion, and statistically auditable clarity profiles.
\end{corollary}

\begin{proof}
Computable normal forms follow from \Cref{thm:golden-fold-normal-form}; cross-resolution compatibility follows from \Cref{thm:cross-resolution-compatibility,prop:stable-restrictions-inverse-system}; address non-expansion follows from \Cref{thm:recursive-address-nonexpansion}; and clarity profiles follow from \Cref{prop:cylinder-decomposition,thm:bayes-prefix-classifier,thm:boundary-clarity}.  The certificate package records these finite data and their statistical audit quantities.
\end{proof}
