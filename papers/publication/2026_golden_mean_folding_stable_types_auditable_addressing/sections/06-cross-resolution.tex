\section{Cross-resolution compatibility}
\label{sec:cross-resolution}

Stable type spaces should be comparable across adjacent resolutions.  The correct comparison is not raw truncation.  Raw truncation may discard a digit before a Fibonacci carry has been normalized.  Stable comparison must normalize first and restrict second.

For stable words, define the stable prefix restriction
\[
\pi_{m+1\to m}:X_{m+1}\to X_m
\]
by
\[
\pi_{m+1\to m}(z_1\cdots z_mz_{m+1})=z_1\cdots z_m.
\]
This is well-defined because every prefix of a Zeckendorf-legal word is again Zeckendorf-legal.

Define the fold-aware restriction
\[
\boxed{
\widetilde\rho_{m+1,m}:=
\pi_{m+1\to m}\circ\Fold_{m+1}:
\Om_{m+1}\to X_m.
}
\]
Thus a raw length-\((m+1)\) word is first folded at resolution \(m+1\), and only then restricted to resolution \(m\).

\begin{theorem}[Fold-aware restriction stability]
\label{thm:cross-resolution-compatibility}
For every \(m\ge1\),
\[
\Fold_m\circ\widetilde\rho_{m+1,m}
=
\widetilde\rho_{m+1,m}.
\]
\end{theorem}

\begin{proof}
For every \(\omega\in\Om_{m+1}\), \(\Fold_{m+1}(\omega)\in X_{m+1}\).  Since prefixes of legal words are legal,
\[
\widetilde\rho_{m+1,m}(\omega)=\pi_{m+1\to m}(\Fold_{m+1}(\omega))\in X_m.
\]
By \Cref{thm:golden-fold-normal-form}, \(\Fold_m\) fixes every element of \(X_m\).  Hence
\[
\Fold_m(\widetilde\rho_{m+1,m}(\omega))
=
\widetilde\rho_{m+1,m}(\omega).
\]
\end{proof}

\begin{proposition}[Naive truncation need not commute with folding]
\label{prop:naive-truncation-failure}
Let \(\tau_{m+1,m}:\Om_{m+1}\to\Om_m\) be raw prefix truncation.  In general,
\[
\tau_{m+1,m}\circ\Fold_{m+1}
\ne
\Fold_m\circ\tau_{m+1,m}.
\]
\end{proposition}

\begin{proof}
It suffices to give one example.  Let \(m=2\) and take \(\omega=011\in\Om_3\).  With weights \(F_2=1\), \(F_3=2\), and \(F_4=3\), we get
\[
N_3(011)=F_3+F_4=2+3=5.
\]
Since \(F_{3+2}=F_5=5\), we have
\[
N_3(011)\bmod F_5=0,
\]
so \(\Fold_3(011)=000\).  Therefore
\[
\tau_{3,2}(\Fold_3(011))=\tau_{3,2}(000)=00.
\]
On the other hand,
\[
\tau_{3,2}(011)=01.
\]
The word \(01\) is legal in length \(2\), so \(\Fold_2(01)=01\).  Hence
\[
\tau_{3,2}\Fold_3(011)=00
\quad\text{but}\quad
\Fold_2\tau_{3,2}(011)=01.
\]
Thus raw truncation and folding do not commute.
\end{proof}

\begin{proposition}[Stable restrictions form an inverse system]
\label{prop:stable-restrictions-inverse-system}
For \(n\ge m\), let \(\pi_{n\to m}:X_n\to X_m\) be stable prefix restriction.  Then
\[
\pi_{m\to m}=\id_{X_m},
\]
and for \(\ell\ge n\ge m\),
\[
\pi_{\ell\to m}=\pi_{n\to m}\circ\pi_{\ell\to n}.
\]
\end{proposition}

\begin{proof}
All maps are defined by deleting stable digits after a given position.  Deleting from length \(\ell\) directly to length \(m\) gives the same prefix as first deleting to length \(n\), and then deleting to length \(m\).  The identity property is immediate.
\end{proof}
