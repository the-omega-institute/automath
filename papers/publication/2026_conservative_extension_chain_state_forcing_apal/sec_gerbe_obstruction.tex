\subsection{Component gerbes and branched gluing obstructions}\label{sec:gerbe-obstruction}

The stackification $\mathfrak L_r$ may have several global connected components over $a$. Cohomological obstruction theory therefore belongs, in general, not to a single gerbe attached to $r$, but to the component gerbes indexed by the visible value classes of $r$.

The results of this subsection use the following hypotheses; individual theorems invoke only the subset they require.

\begin{enumerate}[label=\textup{(H\arabic*)},nosep]
\item\label{hyp:admitted-prestack} An admitted reference $r:\mathsf{Ref}_s$ at a state $p$ with $a=[r]_p$, equipped with a realization prestack $\mathfrak P_r \to \mathcal C_a$ and band $\Band$ in the sense of \Cref{def:realization-prestack}.
\item\label{hyp:band-identification} The band $\Band$ is identified with a fixed abelian sheaf on the slice site $\mathcal C_{p,s}/a$.
\item\label{hyp:global-conservativity} The realization prestack $\mathfrak P_r$ is globally conservative at $a$ in the sense of \Cref{def:global-conservativity}.
\item\label{hyp:cofinal-gerbe-adapted} For every visible branch $v$, gerbe-adapted covers satisfying the vanishing hypotheses of \Cref{thm:cech-bridge-compatible-realizations} form a cofinal family in $\Cov_{p,s}(a)$.
\end{enumerate}

\begin{definition}[Component gerbes and branch obstruction classes]\label{def:compatible-realization-stack}\label{def:cech-gluing-obstruction}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, write
\[
a=[r]_p,
\]
and assume that $r$ admits a realization prestack $\mathfrak P_r$ with stackification $\mathfrak L_r$ and band $\Band$ as in \Cref{def:realization-prestack}. For a visible value class
\[
v\in \Vis_{p,s}(r)\cong \pi_0(\mathfrak L_r)(a),
\]
define the \emph{component substack}
\[
\mathfrak G_v(U):=\{x\in \mathfrak L_r(U)\mid [x]=v|_U\in \pi_0(\mathfrak L_r)(U)\}
\]
for each object $U$ of $\mathcal C_{p,s}/a$. When \Cref{thm:component-gerbe-decomposition} applies, $\mathfrak G_v$ is called the \emph{component gerbe} of the visible branch $v$.

Assume moreover that, for every visible branch $v$, gerbe-adapted covers of $a$ satisfying the vanishing hypotheses of \Cref{thm:cech-bridge-compatible-realizations} form a cofinal family in $\Cov_{p,s}(a)$. Then the resulting filtered Cech class of the branch gerbe $\mathfrak G_v$ is denoted
\[
\omega_v\in \varinjlim_{\mathcal U}\Cech^2(\mathcal U,\Band)=:\Cech^2(\mathcal C_{p,s}/a,\Band)
\]
and called the \emph{branch obstruction class} of $v$.
\end{definition}

\begin{theorem}[Component gerbes of visible branches]\label{thm:component-gerbe-decomposition}
In the setting of \Cref{def:compatible-realization-stack}, for every
\[
v\in \Vis_{p,s}(r)\cong \pi_0(\mathfrak L_r)(a),
\]
the full substack $\mathfrak G_v$ is a banded $\Band$-gerbe over $\mathcal C_{p,s}/a$.
\end{theorem}

\begin{proof}
Since $\mathfrak L_r$ is a stack in groupoids and $\mathfrak G_v$ is a full substack, $\mathfrak G_v$ is again a stack in groupoids. Let $U$ be an object of $\mathcal C_{p,s}/a$. Because $v|_U\in \pi_0(\mathfrak L_r)(U)$ is a section of the sheaf of connected components, by the definition of $\pi_0(\mathfrak L_r)$ there exists a cover $\{U_i\to U\}$ and objects
\[
x_i\in \mathfrak L_r(U_i)
\]
such that $[x_i]=v|_{U_i}$. Hence $(\mathfrak G_v)_{U_i}\neq\varnothing$, so $\mathfrak G_v$ is locally nonempty.

Now let $x,y\in \mathfrak G_v(U)$. Then $[x]=[y]=v|_U\in \pi_0(\mathfrak L_r)(U)$. By the definition of the sheaf of connected components, there exists a cover $\{U_i\to U\}$ such that
\[
x|_{U_i}\cong y|_{U_i}
\]
for every $i$. Thus any two objects of $\mathfrak G_v(U)$ are locally isomorphic, so $\mathfrak G_v$ is a gerbe \cite[Tag 06NY]{StacksProject}. The banding is inherited from $\mathfrak L_r$: since $\mathfrak G_v$ is a full substack, the automorphism sheaf of any object $x\in \mathfrak G_v(U)$ coincides with its automorphism sheaf in $\mathfrak L_r(U)$, hence is identified with $\Band|_U$.
\end{proof}

\begin{definition}[Branch constancy]\label{def:branch-constancy}
In the setting of \Cref{def:compatible-realization-stack}, we say that $\pi_0(\mathfrak L_r)$ is \emph{branch-constant over $a$} if for every object $U$ of $\mathcal C_{p,s}/a$, the restriction map
\[
\rho_U:\pi_0(\mathfrak L_r)(a)\longrightarrow \pi_0(\mathfrak L_r)(U)
\]
is bijective.
\end{definition}

\begin{corollary}[Global branch decomposition under branch constancy]\label{cor:branch-decomposition}
Assume branch constancy over $a$ in the sense of \Cref{def:branch-constancy}. Then the canonical functor
\[
\coprod_{v\in \Vis_{p,s}(r)} \mathfrak G_v\longrightarrow \mathfrak L_r
\]
is an equivalence of stacks in groupoids.
\end{corollary}

\begin{proof}
Let $U\in \mathcal C_{p,s}/a$ and $x\in \mathfrak L_r(U)$. Put $\xi:=[x]\in \pi_0(\mathfrak L_r)(U)$. By surjectivity of $\rho_U$, there exists $v\in \pi_0(\mathfrak L_r)(a)$ such that $\rho_U(v)=\xi$. Hence $x\in \mathfrak G_v(U)$, so the displayed functor is essentially surjective on every fibre.

For full faithfulness, let $x\in \mathfrak G_v(U)$ and $y\in \mathfrak G_w(U)$, and suppose there is a morphism $f:x\to y$ in $\mathfrak L_r(U)$. Any morphism in a groupoid is an isomorphism, hence $[x]=[y]$ in $\pi_0(\mathfrak L_r)(U)$. Therefore $\rho_U(v)=\rho_U(w)$. By injectivity of $\rho_U$, one gets $v=w$. Thus there are no morphisms in $\mathfrak L_r(U)$ between objects of distinct summands, and within each summand the morphism sets coincide because $\mathfrak G_v$ is a full substack.
\end{proof}

\begin{theorem}[Constructing the Cech obstruction of a branch gerbe]\label{thm:cech-bridge-compatible-realizations}
Assume that $\mathfrak G_v$ is a banded component gerbe with band $\Band$ in the sense of \Cref{def:compatible-realization-stack}. Let
\[
\mathcal U=\{a_i\to a\}_{i\in I}\in \Cov_{p,s}(a)
\]
be a cover whose Cech nerve exists in $\mathcal C_{p,s}/a$, and write
\[
U_{i_0\cdots i_n}
\]
for the corresponding simplices. Assume
\[
H^1(U_i,\Band)=0
\qquad\text{and}\qquad
H^1(U_{ij},\Band)=0
\]
for all $i,j\in I$. Choose objects
\[
x_i\in \mathfrak G_v(U_i).
\]
Then:
\begin{enumerate}[label=(\roman*),nosep]
\item for every pair $(i,j)$ there exists an isomorphism
\[
\phi_{ij}:x_i|_{U_{ij}}\xrightarrow{\sim} x_j|_{U_{ij}};
\]
\item on each triple overlap $U_{ijk}$, the element
\[
g_{ijk}:=\phi_{ik}^{-1}|_{U_{ijk}}\circ \phi_{jk}|_{U_{ijk}}\circ \phi_{ij}|_{U_{ijk}}
\in \Band(U_{ijk})
\]
defines a Cech $2$-cocycle $g=(g_{ijk})\in \check Z^2(\mathcal U,\Band)$;
\item the class $[g]\in \Cech^2(\mathcal U,\Band)$ depends only on the branch gerbe $\mathfrak G_v$ and the chosen cover: changing the local objects or the isomorphisms changes $g$ by a Cech coboundary;
\item the class $[g]$ vanishes if and only if the branch gerbe $\mathfrak G_v$ is neutral over $a$.
\end{enumerate}
\end{theorem}

\begin{proof}
Because $\mathfrak G_v$ is a gerbe, the chosen cover is locally nonempty, so the objects $x_i\in \mathfrak G_v(U_i)$ exist by assumption. For every pair $(i,j)$, the sheaf
\[
\mathrm{Isom}(x_i|_{U_{ij}},x_j|_{U_{ij}})
\]
is a $\Band|_{U_{ij}}$-torsor. The vanishing
\[
H^1(U_{ij},\Band)=0
\]
therefore trivializes this torsor \cite[Chap.~III]{MacLaneMoerdijk1994}, \cite[Sec.~11]{StacksProject}, so we may choose a section
\[
\phi_{ij}:x_i|_{U_{ij}}\xrightarrow{\sim} x_j|_{U_{ij}}.
\]
This proves \textup{(i)}.

On a triple overlap $U_{ijk}$, the composite
\[
g_{ijk}:=\phi_{ik}^{-1}|_{U_{ijk}}\circ \phi_{jk}|_{U_{ijk}}\circ \phi_{ij}|_{U_{ijk}}
\]
is an automorphism of $x_i|_{U_{ijk}}$ and hence an element of $\Band(U_{ijk})$. On a quadruple overlap $U_{ijkl}$ one computes
\[
g_{jkl}\,g_{ijl}
=
\phi_{jl}^{-1}\phi_{kl}\phi_{jk}\,\phi_{il}^{-1}\phi_{jl}\phi_{ij}
=
\phi_{il}^{-1}\phi_{kl}\phi_{ik}^{-1}\phi_{ik}\phi_{jk}\phi_{ij}
=
g_{ikl}\,g_{ijk},
\]
where all restrictions to $U_{ijkl}$ are suppressed from the notation. This is exactly the \v{C}ech cocycle identity $\delta g=0$, so $g=(g_{ijk})$ lies in $\check Z^2(\mathcal U,\Band)$. This proves \textup{(ii)}.

For \textup{(iii)}, first keep the objects $x_i$ fixed and replace $\phi_{ij}$ by another choice $\phi'_{ij}$. Since $\mathrm{Isom}(x_i|_{U_{ij}},x_j|_{U_{ij}})$ is a $\Band|_{U_{ij}}$-torsor, there exist unique sections $h_{ij}\in \Band(U_{ij})$ with
\[
\phi'_{ij}=\phi_{ij}\circ h_{ij}^{-1}.
\]
A direct calculation on triple overlaps gives
\[
g'_{ijk}=h_{ik}\,g_{ijk}\,h_{jk}^{-1}h_{ij}^{-1},
\]
which is the multiplicative form of the statement that $g'$ and $g$ differ by the coboundary $\delta h$.

Next replace the objects $x_i$ by another family $x_i'\in \mathfrak G_v(U_i)$. Since $x_i$ and $x_i'$ lie in the same gerbe over $U_i$, the sheaf $\mathrm{Isom}(x_i,x_i')$ is a $\Band|_{U_i}$-torsor. The vanishing
\[
H^1(U_i,\Band)=0
\]
trivializes it, so choose isomorphisms
\[
\psi_i:x_i\xrightarrow{\sim} x_i'.
\]
Define
\[
\phi_{ij}':=\psi_j|_{U_{ij}}\circ \phi_{ij}\circ \psi_i^{-1}|_{U_{ij}}.
\]
Then the associated cocycle satisfies
\[
g'_{ijk}
=
\psi_i|_{U_{ijk}}\,g_{ijk}\,\psi_i^{-1}|_{U_{ijk}}
=
g_{ijk},
\]
under the band identification $\underline{\mathrm{Aut}}(x_i)\cong \Band|_{U_i}$. Thus changing local objects changes the cocycle only by the preliminary freedom in the choice of the $\psi_i$, hence only by a coboundary. Finally, a refinement of covers pulls $g$ back to the corresponding refinement cocycle, so the induced class is well defined in the filtered \v{C}ech system.

For \textup{(iv)}, suppose first that $\mathfrak G_v$ is neutral and choose a global object $x\in \mathfrak G_v(a)$. For each $i$, the sheaf $\mathrm{Isom}(x|_{U_i},x_i)$ is a $\Band|_{U_i}$-torsor and therefore trivial because $H^1(U_i,\Band)=0$. Choose
\[
\psi_i:x|_{U_i}\xrightarrow{\sim} x_i.
\]
If we set
\[
\phi_{ij}:=\psi_j|_{U_{ij}}\circ \psi_i^{-1}|_{U_{ij}},
\]
then the resulting cocycle is identically zero, so $[g]=0$.

Conversely, assume $[g]=0$. Then there is a $1$-cochain $h=(h_{ij})$ with
\[
g=\delta h.
\]
Replace $\phi_{ij}$ by
\[
\phi_{ij}':=\phi_{ij}\circ h_{ij}^{-1}.
\]
The defining equation $g=\delta h$ is exactly the statement that the $\phi_{ij}'$ satisfy the strict cocycle condition on triple overlaps. Hence $(x_i,\phi_{ij}')$ is a descent datum for the stack $\mathfrak G_v$, and since $\mathfrak G_v$ is a stack in groupoids, that descent datum is effective. The resulting glued object of $\mathfrak G_v(a)$ is a global object, so the gerbe is neutral.
\end{proof}

\begin{proposition}[Compatible local sections versus visible branches]\label{prop:compsecs-vs-visible-branches}
Assume the hypotheses of \Cref{def:cech-gluing-obstruction}. For an admitted reference
$r:\mathsf{Ref}_s$ at $p$,
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
\Vis_{p,s}(r)\neq\varnothing.
\]
\end{proposition}

\begin{proof}
By \Cref{thm:sheafification-characterization},
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
\bigl(a_{p,s}F_{p,s}\bigr)(a)\neq\varnothing.
\]
By \Cref{thm:visible-value-components}, this set is $\Vis_{p,s}(r)$.
\end{proof}

\begin{lemma}[Terminal-fibre nonemptiness and neutrality]\label{lem:terminal-fibre-neutrality}
Let $v\in \Vis_{p,s}(r)$. Then
\[
\mathfrak G_v(a)\neq\varnothing
\quad\Longleftrightarrow\quad
\mathfrak G_v\ \text{is neutral}.
\]
\end{lemma}

\begin{proof}
By definition of the gerbe notion used in this paper, neutrality means the existence of a global object over $a$. Therefore it is exactly the same as $\mathfrak G_v(a)\neq\varnothing$.
\end{proof}

\begin{proposition}[Global sections versus neutral component gerbes]\label{prop:global-sections-vs-neutral-components}
Assume the hypotheses of \Cref{def:cech-gluing-obstruction} and global conservativity at $a$ in the sense of \Cref{def:global-conservativity}. For an admitted reference
$r:\mathsf{Ref}_s$ at $p$,
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\exists v\in \Vis_{p,s}(r)\ \text{with}\ \mathfrak G_v(a)\neq\varnothing.
\]
Equivalently, by \Cref{lem:terminal-fibre-neutrality},
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\exists v\in \Vis_{p,s}(r)\ \text{with}\ \mathfrak G_v\ \text{is neutral}.
\]
\end{proposition}

\begin{proof}
From \Cref{prop:no-new-global-objects}, $M,p\Vdash \Sec_s(r)$ iff $\mathfrak L_r(a)\neq\varnothing$.

If $\mathfrak L_r(a)\neq\varnothing$, choose $x\in \mathfrak L_r(a)$ and let
\[
v=[x]\in \pi_0(\mathfrak L_r)(a)\cong \Vis_{p,s}(r).
\]
Then $x\in \mathfrak G_v(a)$, so $\mathfrak G_v(a)\neq\varnothing$.

Conversely, if $\mathfrak G_v(a)\neq\varnothing$ for some visible branch $v$, then $\mathfrak G_v(a)\subseteq \mathfrak L_r(a)$, so $\mathfrak L_r(a)\neq\varnothing$.
\end{proof}

\begin{theorem}[Conditional branched gerbe semantics of gluing-level absence]\label{thm:gerbe-null-semantics}
Assume the hypotheses of \Cref{def:cech-gluing-obstruction} and global conservativity at $a$ in the sense of \Cref{def:global-conservativity}. Then, for the admitted reference $r:\mathsf{Ref}_s$ under consideration,
\[
M,p\Vdash \CompSec_s(r)
\quad\Longleftrightarrow\quad
\Vis_{p,s}(r)\neq\varnothing.
\]
\[
M,p\Vdash \Sec_s(r)
\quad\Longleftrightarrow\quad
\exists v\in \Vis_{p,s}(r)\ \text{such that}\ \mathfrak G_v\ \text{is neutral}.
\]
Consequently,
\[
M,p\Vdash \Null^{\mathrm{glue}}_s(r)
\quad\Longleftrightarrow\quad
\Vis_{p,s}(r)\neq\varnothing
\ \text{and}\
\forall v\in \Vis_{p,s}(r),\ \mathfrak G_v\ \text{is non-neutral}.
\]

If, in addition, the chosen cofinal family of gerbe-adapted covers computes derived cohomology in degree $2$, then the last condition is equivalent to
\[
\Vis_{p,s}(r)\neq\varnothing
\ \text{and}\
\omega_v\neq 0\ \text{for every}\ v\in \Vis_{p,s}(r).
\]
In particular, the single obstruction class picture is recovered precisely when $|\Vis_{p,s}(r)|=1$.
\end{theorem}

\begin{proof}
The first equivalence is \Cref{prop:compsecs-vs-visible-branches}. The second is \Cref{prop:global-sections-vs-neutral-components}. The third is then immediate from the definition
\[
\Null^{\mathrm{glue}}_s(r):=\Adm_s(r)\land \CompSec_s(r)\land \neg\Sec_s(r).
\]
For the final claim, if the chosen cofinal family computes derived cohomology in degree $2$, then \Cref{thm:cech-bridge-compatible-realizations}(iv) identifies the vanishing of each $\omega_v$ with neutrality of the corresponding component gerbe, while the comparison with derived $H^2$ is the standard banded-gerbe classification \cite[Chap.~IV]{Giraud1971}, \cite[Tag 06NY]{StacksProject}.
\end{proof}

\subsection{Visible quotients and the information cost of strictification}

The previous subsection shows that a single admitted reference may carry several visible branches, each with its own component gerbe and obstruction class. The following analysis fixes one visible branch and determines how much of its obstruction remains semantically visible without enlarging the state space. The answer is a canonical quotient determined by the actual obstruction values on that branch, together with an exact lower bound on the auxiliary information needed to restore full resolution. The aggregate interaction of all visible branches will be treated afterwards.

\begin{definition}[Finite obstruction presentation]\label{def:finite-obstruction-presentation}
Fix an admitted reference $r:\mathsf{Ref}_s$ at a state $p$, and assume:
\begin{enumerate}[label=(\roman*),nosep]
\item the hypotheses of \Cref{def:cech-gluing-obstruction} hold;
\item a visible branch
\[
v\in \Vis_{p,s}(r)
\]
has been chosen;
\item the associated branch gerbe $\mathfrak G_v$ is banded by a finite abelian group $A$,
\item a finite covering family
\[
\mathcal U=\{a_i\to a\}_{i\in I},
\qquad
a=[r]_p,
\]
has been chosen together with a \v{C}ech $2$-cocycle
\[
\alpha=(\alpha_{ijk})\in Z^2(\mathcal U,A)
\]
representing the branch obstruction class $\omega_v$, where the indices $(i,j,k)$ range over the nonempty triple overlaps of $\mathcal U$.
\end{enumerate}
Define the \emph{obstruction subgroup}
\[
H_\alpha:=\langle \alpha_{ijk}\mid (i,j,k)\text{ ranges over the nonempty triple overlaps of }\mathcal U\rangle\le A,
\]
and the associated \emph{visible quotient}
\[
A_{\mathrm{vis}}:=A/H_\alpha.
\]
\end{definition}

\begin{theorem}[Initial visible quotient of a gluing obstruction]\label{thm:initial-visible-quotient}
In the setting of \Cref{def:finite-obstruction-presentation}, let
\[
\pi:A\twoheadrightarrow A_{\mathrm{vis}}
\]
be the canonical quotient map. Then:
\begin{enumerate}[label=(\roman*),nosep]
\item the pushed-forward cocycle $\pi_*(\alpha)$ is pointwise zero on every nonempty triple overlap;
\item if $\varphi:A\to B$ is any homomorphism into an abelian group $B$ such that $\varphi(\alpha_{ijk})=0$ for every nonempty triple overlap, then there exists a unique homomorphism
\[
\bar\varphi:A_{\mathrm{vis}}\to B
\]
with $\varphi=\bar\varphi\circ\pi$.
\end{enumerate}
Hence $A_{\mathrm{vis}}$ is the initial coefficient quotient through which the chosen obstruction presentation becomes strictly trivial.
\end{theorem}

\begin{proof}
By construction, every cocycle value $\alpha_{ijk}$ lies in the subgroup $H_\alpha$. Therefore $\pi(\alpha_{ijk})=0$ for every nonempty triple overlap, which proves (i).

For (ii), the hypothesis $\varphi(\alpha_{ijk})=0$ on every nonempty triple overlap implies that every generator of $H_\alpha$ lies in $\ker\varphi$. Hence $H_\alpha\subseteq\ker\varphi$. By the universal property of the quotient by $H_\alpha$, there exists a unique homomorphism $\bar\varphi:A/H_\alpha\to B$ satisfying $\varphi=\bar\varphi\circ\pi$.
\end{proof}

The strict quotient $A/H_\alpha$ depends on the chosen cocycle representative $\alpha$, not only on the cohomology class $\omega_v$. The next subsection removes this dependence: the homological evaluation map $\mathrm{ev}_\omega$ is defined from the class itself, and its cokernel---the intrinsic visible quotient $A_{\mathrm{vis}}^\omega$---is a canonical invariant. The comparison between the two quotients is given by \Cref{cor:strict-to-intrinsic-visible} and \Cref{prop:chain-vs-homological-visibility}: every strict visible quotient is a quotient of the intrinsic one, and the gap between them is exactly the gap between the image of the cocycle on all $2$-chains and its image on $2$-cycles.

\begin{theorem}[Definability barrier for character channels]\label{thm:character-definability-barrier}
In the setting of \Cref{def:finite-obstruction-presentation}, let
\[
\widehat A:=\mathrm{Hom}(A,\mathbb T),
\]
where $\mathbb T\subseteq\mathbb C^\times$ is the unit circle, and define
\[
H_\alpha^\perp:=\{\chi\in\widehat A\mid \chi(h)=1\text{ for every }h\in H_\alpha\}.
\]
Call a character $\chi\in\widehat A$ \emph{globally admissible} for the obstruction presentation if
\[
\chi(\alpha_{ijk})=1
\]
for every nonempty triple overlap of $\mathcal U$. Then:
\begin{enumerate}[label=(\roman*),nosep]
\item a character $\chi\in\widehat A$ is globally admissible if and only if $\chi\in H_\alpha^\perp$;
\item the quotient map $\pi:A\twoheadrightarrow A_{\mathrm{vis}}$ induces a natural isomorphism
\[
\widehat{A_{\mathrm{vis}}}\cong H_\alpha^\perp;
\]
\item consequently, every globally admissible character channel factors uniquely through $A_{\mathrm{vis}}$, while no character outside $H_\alpha^\perp$ can define a gauge-invariant global readout.
\end{enumerate}
\end{theorem}

\begin{proof}
Since $H_\alpha$ is generated by the cocycle values $\alpha_{ijk}$, a character $\chi$ is trivial on $H_\alpha$ if and only if it is trivial on each generator $\alpha_{ijk}$. This proves (i).

For (ii), the quotient map $\pi$ induces by precomposition an injective homomorphism
\[
\pi^*:\widehat{A_{\mathrm{vis}}}\hookrightarrow \widehat A.
\]
Its image consists exactly of the characters on $A$ that are trivial on $\ker\pi=H_\alpha$, namely $H_\alpha^\perp$. Thus $\pi^*$ identifies $\widehat{A_{\mathrm{vis}}}$ with $H_\alpha^\perp$.

For (iii), a character-level readout can be gauge-invariant only if the corresponding character kills the obstruction multipliers on all triple overlaps, which is the condition in (i). By (ii), this is equivalent to factorization through $A_{\mathrm{vis}}$. If $\chi\notin H_\alpha^\perp$, then some obstruction value survives in the $\chi$-channel, so the resulting would-be readout remains path-dependent and cannot descend to a global observable.
\end{proof}

\begin{corollary}[The visible quotient is exactly the character-visible quotient]\label{cor:visible-quotient-character-separation}
In the setting of \Cref{def:finite-obstruction-presentation}, for elements $x,y\in A$ the following are equivalent:
\begin{enumerate}[label=(\roman*),nosep]
\item $\pi(x)=\pi(y)$ in $A_{\mathrm{vis}}$;
\item $\chi(x)=\chi(y)$ for every globally admissible character $\chi\in\widehat A$.
\end{enumerate}
Equivalently, $A_{\mathrm{vis}}$ is the quotient of $A$ obtained by identifying exactly those elements that cannot be separated by globally admissible character channels.
\end{corollary}

\begin{proof}
Assume first that $\pi(x)=\pi(y)$. By \Cref{thm:character-definability-barrier}(ii), every globally admissible character $\chi$ factors as
\[
\chi=\bar\chi\circ\pi
\]
for a unique character $\bar\chi\in\widehat{A_{\mathrm{vis}}}$. Hence
\[
\chi(x)=\bar\chi(\pi(x))=\bar\chi(\pi(y))=\chi(y).
\]

Conversely, suppose $\pi(x)\neq\pi(y)$. Then
\[
g:=\pi(x)-\pi(y)\neq 0
\]
in the finite abelian group $A_{\mathrm{vis}}$. By the structure theorem for finite abelian groups \cite{DummitFoote2004}, there is an isomorphism
\[
A_{\mathrm{vis}}\cong \mathbb Z/n_1\mathbb Z\oplus\cdots\oplus \mathbb Z/n_t\mathbb Z
\]
under which the image of $g$ has some nonzero coordinate, say the $k$th. The coordinate character
\[
\psi(z_1,\dots,z_t):=\exp(2\pi i z_k/n_k)
\]
therefore satisfies $\psi(g)\neq 1$. By \Cref{thm:character-definability-barrier}(ii), the character
\[
\chi:=\psi\circ\pi
\]
is globally admissible. Since
\[
\chi(x)\chi(y)^{-1}=\psi(\pi(x)-\pi(y))=\psi(g)\neq 1,
\]
we have $\chi(x)\neq\chi(y)$, contradicting (ii). Thus $\pi(x)=\pi(y)$.
\end{proof}

\begin{definition}[Auxiliary strictification code]\label{def:auxiliary-strictification-code}
In the setting of \Cref{def:finite-obstruction-presentation}, an \emph{auxiliary strictification code} consists of a finite set $R$ and an injective map
\[
\iota:A\hookrightarrow A_{\mathrm{vis}}\times R
\]
such that the first projection recovers the canonical quotient:
\[
\mathrm{pr}_1\circ\iota=\pi.
\]
Equivalently, the visible quotient is kept fixed, while the auxiliary register $R$ resolves the hidden information inside the fibers of $\pi$.
\end{definition}

\begin{theorem}[Minimal information cost of strictification]\label{thm:minimal-information-cost-of-strictification}
In the setting of \Cref{def:auxiliary-strictification-code}, every auxiliary strictification code satisfies
\[
|R|\ge |H_\alpha|.
\]
Equivalently,
\[
\log_2 |R|\ge \log_2 |H_\alpha|.
\]
Moreover, this lower bound is sharp.
\end{theorem}

\begin{proof}
Let $\bar a\in A_{\mathrm{vis}}$. The fiber $\pi^{-1}(\bar a)$ is a coset of $\ker\pi=H_\alpha$, hence has cardinality $|H_\alpha|$. Since $\iota$ is injective and $\mathrm{pr}_1\circ\iota=\pi$, the restriction of $\iota$ to $\pi^{-1}(\bar a)$ lands inside the slice $\{\bar a\}\times R$ and remains injective. Therefore
\[
|H_\alpha|=|\pi^{-1}(\bar a)|\le |\{\bar a\}\times R|=|R|.
\]
This proves the lower bound.

To see that it is sharp, take $R:=H_\alpha$. Choose a set-theoretic section $s:A_{\mathrm{vis}}\to A$ of $\pi$. For each $x\in A$, the difference
\[
h(x):=x-s(\pi(x))
\]
lies in $H_\alpha$. Define
\[
\iota(x):=(\pi(x),h(x))\in A_{\mathrm{vis}}\times H_\alpha.
\]
Then $\mathrm{pr}_1\circ\iota=\pi$. If $\iota(x)=\iota(y)$, then $\pi(x)=\pi(y)$ and $h(x)=h(y)$, hence
\[
x=s(\pi(x))+h(x)=s(\pi(y))+h(y)=y.
\]
Thus $\iota$ is injective and $|R|=|H_\alpha|$.
\end{proof}

\begin{corollary}[Conservative recovery requires hidden-state budget]\label{cor:conservative-recovery-budget}
For a finite obstruction presentation, no conservative enrichment that leaves the visible quotient unchanged can restore full $A$-resolution with fewer than $\log_2|H_\alpha|$ bits of auxiliary local state. In particular, the subgroup $H_\alpha$ measures the exact hidden-state budget of strictification for the chosen gluing obstruction.
\end{corollary}

\begin{proof}
Any such enrichment induces an auxiliary strictification code in the sense of \Cref{def:auxiliary-strictification-code}. Apply \Cref{thm:minimal-information-cost-of-strictification}.
\end{proof}

\subsection{A worked finite example}

\begin{example}[A one-bit hidden obstruction]\label{ex:one-bit-hidden-obstruction}
Assume an admitted reference $r:\mathsf{Ref}_s$ together with a three-member cover
\[
\mathcal U=\{a_0,a_1,a_2\to a\},
\qquad
a=[r]_p,
\]
and let the band be the additive group
\[
A=\mathbb Z/4\mathbb Z.
\]
Suppose the chosen obstruction presentation is represented by a cocycle whose only nonzero value on a nonempty triple overlap is
\[
\alpha_{012}=2\in \mathbb Z/4\mathbb Z.
\]
Then
\[
H_\alpha=\langle 2\rangle=\{0,2\},
\qquad
A_{\mathrm{vis}}=A/H_\alpha\cong \mathbb Z/2\mathbb Z.
\]

Thus the visible quotient identifies $0$ with $2$ and identifies $1$ with $3$. By \Cref{thm:initial-visible-quotient}, this is the initial coefficient quotient through which the chosen obstruction presentation becomes trivial.

For $m\in\mathbb Z/4\mathbb Z$, define a character
\[
\chi_m(x):=\exp(2\pi i mx/4).
\]
Then
\[
\chi_m(\alpha_{012})=\chi_m(2)=(-1)^m,
\]
so \Cref{thm:character-definability-barrier} implies that the globally admissible characters are exactly
\[
\chi_0
\qquad\text{and}\qquad
\chi_2.
\]
Hence the only surviving linear observation is parity: $\chi_2$ distinguishes the two cosets of $H_\alpha$, while the phase-sensitive channels $\chi_1$ and $\chi_3$ do not descend to global observables. By \Cref{cor:visible-quotient-character-separation}, the visible quotient is therefore exactly the quotient detected by all globally admissible character channels.

Finally, \Cref{thm:minimal-information-cost-of-strictification} yields
\[
|R|\ge |H_\alpha|=2,
\]
so at least one auxiliary bit is required to recover the full $\mathbb Z/4\mathbb Z$-resolution while keeping the visible quotient fixed. This lower bound is attained by the code
\[
0\mapsto (\overline{0},0),\qquad
2\mapsto (\overline{0},1),\qquad
1\mapsto (\overline{1},0),\qquad
3\mapsto (\overline{1},1),
\]
where $\overline{0},\overline{1}\in A_{\mathrm{vis}}\cong \mathbb Z/2\mathbb Z$ and the second component is a one-bit register. This is the smallest nontrivial instance in which visible information and hidden information separate.
\end{example}
