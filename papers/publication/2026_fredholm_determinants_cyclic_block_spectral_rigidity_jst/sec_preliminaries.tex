\section{Finite-State \texorpdfstring{$\zeta$}{zeta}-Functions and Finite Parts}
\label{sec:preliminaries}

\subsection{Primitive orbit counts}

Let \(\mathcal A\) be a finite alphabet and let
\(A\in\{0,1\}^{\mathcal A\times\mathcal A}\) be a primitive adjacency
matrix. The associated one-sided subshift of finite type is
\[
  X_A^+
  =\bigl\{(x_n)_{n\ge 0}\in \mathcal A^{\NN}:
    A_{x_n,x_{n+1}}=1\ \text{for all }n\ge 0\bigr\},
\]
equipped with the left shift \(\sigma\). We write
\[
  a_n(A):=\Tr(A^n)
  =\#\Fix(\sigma^n)
\]
for the number of points of period dividing \(n\). The number of
primitive orbits of least period \(n\) is
\begin{equation}\label{eq:primitive-mobius}
  p_n(A)=\frac{1}{n}\sum_{d\mid n}\mu(d)\,a_{n/d}(A),
\end{equation}
and the inverse necklace relation is
\begin{equation}\label{eq:trace-primitive-identity}
  a_n(A)=\sum_{d\mid n}d\,p_d(A).
\end{equation}
These are the usual necklace/Witt inversion identities
\cite{MetropolisRota1983} in the finite symbolic setting.
The corresponding dynamical \(\zeta\)-function is the finite-state
instance of the Artin--Mazur periodic-point generating function
\cite{ArtinMazur1965}.

\begin{proposition}[Trace, primitive orbits, and Euler product]
  \label{prop:prelim-trace-primitive}
  For a primitive adjacency matrix \(A\) with Perron root \(\lambda\),
  \[
    \zeta_A(z)
    :=\exp\!\left(\sum_{n\ge 1}\frac{a_n(A)}{n}z^n\right)
    =\det(I-zA)^{-1}
    =\prod_{n\ge 1}(1-z^n)^{-p_n(A)}
  \]
  for every \(|z|<\lambda^{-1}\). The Euler product identity is an
  equality of holomorphic germs at \(z=0\); equivalently, it is an
  identity of formal power series. In particular,
  \[
    \log\det(I-zA)^{-1}
    =\sum_{n\ge 1}\frac{a_n(A)}{n}z^n
    \qquad (|z|<\lambda^{-1}).
  \]
\end{proposition}

\begin{proof}
  The determinant formula for shifts of finite type is classical; see
  \cite{BowenLanford1970Zeta,LindMarcus1995}. Since \(A\) is a finite
  matrix, we can also verify directly the exact identity needed here.
  Let \(\lambda_1,\dots,\lambda_d\) be the eigenvalues of \(A\), counted
  with algebraic multiplicity. Primitivity gives
  \(|\lambda_i|\le\lambda\) for every \(i\), so for \(|z|<\lambda^{-1}\)
  one has \(|z\lambda_i|<1\) and therefore
  \[
    -\log(1-z\lambda_i)=\sum_{n\ge1}\frac{(\lambda_i z)^n}{n},
  \]
  with the principal branch. Summing over \(i\) yields
  \begin{align*}
    \log\det(I-zA)^{-1}
    &=-\sum_{i=1}^d\log(1-z\lambda_i) \\
    &=\sum_{n\ge1}\frac{z^n}{n}\sum_{i=1}^d\lambda_i^n
      =\sum_{n\ge1}\frac{\Tr(A^n)}{n}z^n
      =\sum_{n\ge1}\frac{a_n(A)}{n}z^n.
  \end{align*}
  Exponentiating gives
  \[
    \zeta_A(z)=\det(I-zA)^{-1}
    =\exp\!\left(\sum_{n\ge1}\frac{a_n(A)}{n}z^n\right).
  \]

  For the Euler product, \eqref{eq:trace-primitive-identity} gives
  \(a_n(A)=\sum_{d\mid n} d\,p_d(A)\). Hence, for \(|z|<\lambda^{-1}\le1\),
  \begin{align*}
    \sum_{n\ge1}\frac{a_n(A)}{n}z^n
    &=\sum_{n\ge1}\sum_{d\mid n}\frac{d\,p_d(A)}{n}z^n \\
    &=\sum_{m\ge1}\sum_{r\ge1}\frac{p_m(A)}{r}z^{mr}
      =-\sum_{m\ge1}p_m(A)\log(1-z^m),
  \end{align*}
  again using the principal branch on \(|z|<1\). Exponentiating gives
  \[
    \det(I-zA)^{-1}
    =\prod_{m\ge1}(1-z^m)^{-p_m(A)}.
  \]
  This is precisely an identity of holomorphic germs at \(0\), hence also
  of formal power series.
\end{proof}

If \(\lambda=\rho(A)\) is the Perron root and
\[
  \Lambda(A):=\max\{|\mu|:\mu\in\spec(A),\ \mu\neq\lambda\},
\]
then the primitive orbit counts satisfy the usual Perron asymptotic.

\begin{lemma}\label{lem:primitive-orbit-asymptotic}
For a primitive matrix \(A\),
\[
  p_n(A)=\frac{\lambda^n}{n}
  +O\!\left(\frac{\max\{\Lambda(A)^n,\lambda^{n/2}\}}{n}\right).
\]
\end{lemma}

\begin{proof}
  Perron--Frobenius theory gives
  \(
    a_n(A)=\lambda^n+O(\Lambda(A)^n)
  \).
  In \eqref{eq:primitive-mobius}, the divisor \(d=1\) gives the main
  term. For \(d\ge 2\), one has \(n/d\le n/2\), hence
  \[
    \sum_{\substack{d\mid n\\ d\ge 2}}\frac{|a_{n/d}(A)|}{n}
    \ll \frac{\lambda^{n/2}}{n},
  \]
  where the implied constant depends only on the Jordan data of the
  non-Perron spectrum of \(A\).
  This proves the estimate.
\end{proof}
\leanverified{paper_fredholm_witt_primitive_orbit_asymptotic}

\subsection{Finite-part constants}

Let \(z_\ast=\lambda^{-1}\). Since \(\lambda\) is a simple eigenvalue,
\(\zeta_A\) has a simple pole at \(z_\ast\), and we define
\begin{equation}\label{eq:residue-constant-def}
  C(A):=\lim_{z\to z_\ast}(1-\lambda z)\zeta_A(z).
\end{equation}
If \(\mu_2,\dots,\mu_d\) are the non-Perron eigenvalues of \(A\), then
\begin{equation}\label{eq:reduced-det-poly}
  F_A(t):=\frac{\det(I-tA/\lambda)}{1-t}
  =\prod_{j=2}^d\left(1-\frac{\mu_j}{\lambda}t\right)
\end{equation}
and
\begin{equation}\label{eq:reduced-det-constant}
  C(A)=F_A(1)^{-1}.
\end{equation}

For the Abel finite part we assume from now on in this subsection that the
Perron root is expanding, \(\lambda>1\). This is not a consequence of
primitivity alone: the one-state one-loop matrix has \(\lambda=1\), and
the renormalisation below is not finite there. Define the primitive Abel
generating series
\[
  P_A(r):=\sum_{n\ge 1}p_n(A)\lambda^{-n}r^n,
  \qquad 0\le r<1.
\]
Because
\(
  p_n(A)\lambda^{-n}=1/n+O(\vartheta^n/n)
\)
for some \(\vartheta\in(0,1)\), the renormalised quantity
\[
  P_A(r)+\log(1-r)
\]
has a finite limit as \(r\uparrow 1\).

\begin{definition}[Abel finite-part constant]
  \label{def:abel-finite-part}
  We write
  \[
    \log\MM(A):=\lim_{r\uparrow 1}\bigl(P_A(r)+\log(1-r)\bigr).
  \]
  Indeed,
  \[
    P_A(r)+\log(1-r)
    =\sum_{n\ge 1}\left(p_n(A)\lambda^{-n}-\frac1n\right)r^n,
  \]
  and the estimate
  \(p_n(A)\lambda^{-n}=1/n+O(\vartheta^n/n)\) compares the difference
  termwise with the absolutely convergent series
  \(\sum_{n\ge 1}\vartheta^n r^n/n\).
\end{definition}

\begin{theorem}[Primitive-orbit closed form]
  \label{thm:finite-part-primitive-closed-form}
  Let \(A\) be primitive with Perron root \(\lambda>1\). Then
  \(\log\MM(A)\) exists and satisfies
  \[
    \log\MM(A)
    =\log C(A)+\sum_{k\ge 2}\frac{\mu(k)}{k}
      \log\det(I-\lambda^{-k}A)^{-1}.
  \]
  The logarithms in the secondary determinants are the trace-log
  branches determined by the absolutely convergent series
  \[
    \log\det(I-\lambda^{-k}A)^{-1}
    =\sum_{m\ge1}\frac{\Tr((\lambda^{-k}A)^m)}{m}
    \qquad (k\ge2).
  \]
  The expansion assumption \(\lambda>1\) is part of the theorem: it gives
  \(\rho(\lambda^{-k}A)=\lambda^{1-k}<1\) for \(k\ge2\), and this is the
  reason the secondary determinant branches are single-valued.
  Equivalently,
  \[
    \log\MM(A)
    =\log C(A)
      +\sum_{n\ge 1}p_n(A)
       \bigl(\lambda^{-n}+\log(1-\lambda^{-n})\bigr).
  \]
  \leanverified{paper_fredholm_witt_finite_part_primitive_closed_form}
\end{theorem}

\begin{proof}
  We first keep an Abel parameter \(0\le r<1\) in every determinant. By
  \Cref{prop:prelim-trace-primitive},
  \[
    \log\det(I-zA)^{-1}
    =\sum_{m\ge 1}\frac{\Tr(A^m)}{m}z^m
    \qquad (|z|<\lambda^{-1}).
  \]
  Since every eigenvalue of \(A\) has modulus at most \(\lambda\),
  \(\abs{\Tr(A^m)}\le d\lambda^m\). Therefore, for \(0\le r<1\),
  \[
    \sum_{k\ge 1}\sum_{m\ge 1}
    \frac{\abs{\mu(k)}}{km}\,
    \abs{\Tr(A^m)}\,\lambda^{-km}r^{km}
    \le d\sum_{k\ge 1}\sum_{m\ge 1}
    \frac{\lambda^{-(k-1)m}r^{km}}{km}<\infty .
  \]
  The \(k=1\) part is \(\sum_{m\ge1}r^m/m\). For \(k\ge2\), the factor
  \(\lambda^{-(k-1)m}\) gives a geometric majorant; this is the only place
  in the rearrangement where the expansion \(\lambda>1\) is used.
  Hence Tonelli--Fubini gives
  \begin{align*}
    \sum_{k\ge 1}\frac{\mu(k)}{k}
      \log\det(I-\lambda^{-k}r^kA)^{-1}
    &=\sum_{k\ge 1}\frac{\mu(k)}{k}
      \sum_{m\ge 1}\frac{\Tr(A^m)}{m}\lambda^{-km}r^{km} \\
    &=\sum_{n\ge 1}\frac{\lambda^{-n}r^n}{n}
      \sum_{k\mid n}\mu(k)\Tr(A^{n/k}) \\
    &=\sum_{n\ge 1}p_n(A)\lambda^{-n}r^n
      =P_A(r).
  \end{align*}

  Now isolate the only singular term as \(r\uparrow1\). Since
  \[
    \det(I-\lambda^{-1}rA)=(1-r)F_A(r),
    \qquad F_A(1)=C(A)^{-1},
  \]
  the principal branch satisfies
  \[
    \log\det(I-\lambda^{-1}rA)^{-1}
    =-\log(1-r)+\log C(A)+o(1).
  \]
  For \(k\ge2\) and \(0\le r\le1\), every eigenvalue of
  \(\lambda^{-k}r^kA\) has modulus at most \(\lambda^{1-k}<1\). Hence
  the secondary determinants are non-zero at \(r=1\), their logarithms are
  the trace-log branches in the statement. More explicitly,
  \[
    \left|
      \log\det(I-\lambda^{-k}r^kA)^{-1}
    \right|
    \le
    d\sum_{m\ge1}\frac{\lambda^{(1-k)m}}{m}
    \le C\,\lambda^{1-k}
    \qquad(k\ge2,\ 0\le r\le1),
  \]
  with \(C\) independent of \(k\) and \(r\). This geometric bound permits
  the limit \(r\uparrow1\) to pass through the secondary sum. Combining
  this with the definition of
  \(\log\MM(A)\) proves
  \[
    \log\MM(A)
    =\log C(A)+\sum_{k\ge 2}\frac{\mu(k)}{k}
      \log\det(I-\lambda^{-k}A)^{-1}.
  \]

  For the primitive-orbit form, use the Euler product from
  \Cref{prop:prelim-trace-primitive}. For \(|x|<1\),
  \[
    -\sum_{k\ge 1}\frac{\mu(k)}{k}\log(1-x^k)
    =\sum_{n\ge 1}\frac{x^n}{n}\sum_{k\mid n}\mu(k)
    =x,
  \]
  so
  \[
    -\sum_{k\ge 2}\frac{\mu(k)}{k}\log(1-x^k)
    =x+\log(1-x).
  \]
  The interchange of the \(k\)- and \(n\)-sums is absolute: from
  \(p_n(A)=O(\lambda^n/n)\) and
  \(\abs{\log(1-y)}\le C_\lambda y\) for
  \(0\le y\le\lambda^{-2}\),
  \[
    \sum_{n\ge1}\abs{p_n(A)}\sum_{k\ge2}
    \frac{\abs{\mu(k)}}{k}\,
    \abs{\log(1-\lambda^{-kn})}
    \ll
    \sum_{n\ge1}\frac{\lambda^n}{n}
    \sum_{k\ge2}\lambda^{-kn}
    <\infty .
  \]
  Setting \(x=\lambda^{-n}\) for the cycle-length index \(n\) gives
  \[
    \log\MM(A)=\log C(A)+\sum_{n\ge 1}
    p_n(A)\bigl(\lambda^{-n}+\log(1-\lambda^{-n})\bigr).
  \]
  This last series is absolutely convergent because
  \(p_n(A)=O(\lambda^n/n)\) and
  \(x+\log(1-x)=O(x^2)\) as \(x\to0\), so the rearrangement is legitimate.
\end{proof}
\begin{remark}[Finite-state arithmetic]
  \label{rem:finite-state-arithmetic}
  For primitive finite matrices, the numbers \(p_n(A)\) are
  non-negative integers because they count primitive periodic orbits.
  This arithmetic feature belongs to the finite-state category and will
  not survive after passage to general trace-class operators.
\end{remark}
