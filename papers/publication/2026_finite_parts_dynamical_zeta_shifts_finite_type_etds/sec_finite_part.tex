\section{Auxiliary finite-matrix inputs: scalar finite parts and cyclic
reconstruction}
\label{sec:finite-part}

Throughout this section, \(A\) denotes a primitive non-negative real
matrix with Perron root \(\lambda>1\). The symbols \(p_n(A)\) denote the
algebraic M\"obius coefficients of \eqref{eq:primitive-mobius}; they are
primitive orbit counts only in the integer multigraph case fixed in
\Cref{sec:preliminaries}. This section supplies the scalar fixed-matrix
inputs isolated in \Cref{prop:intro-scope-of-results}: the finite-part
formula giving \(\mathcal M(A)\), the cyclic-lift recovery of the
reduced spectrum from the full record \(\{\Psi_q(A)\}_{q\ge 1}\), and
the one-layer Fourier recovery for one fixed \(q\).

\medskip\noindent\textbf{Logarithm convention.}
All logarithms in this section are taken in the principal branch
\(\Log\) on \(\CC\setminus(-\infty,0]\). Since the non-Perron
eigenvalues satisfy \(|\mu_j|<\lambda\), each factor
\(1-(\mu_j/\lambda)^q\) has positive real part for all \(q\ge 1\), so
\(\Log(1-(\mu_j/\lambda)^q)\) is unambiguous. Likewise, the reduced
spectral polynomial satisfies \(F_A(e^{i\theta})\ne 0\) for all
\(\theta\in\RR\) (since \(|\mu_j|<\lambda\) implies \(|\mu_j/\lambda|<1\)),
so \(\Log F_A(e^{i\theta})\) is well defined and
\(-\Log F_A(e^{i\theta})=\sum_{n\ge 1}u_n(A)e^{in\theta}/n\) is the
Fourier series of a continuous function on the torus. All subsequent
power series and determinant logarithms use this branch consistently.

\subsection{Finite parts and reduced determinants}

\begin{proposition}[Primitive-orbit closed form for the finite part]
  \label{thm:finite-part-primitive-closed-form}
  Let \(A\) be a primitive non-negative matrix with Perron root
  \(\lambda>1\). Then
  \begin{equation}\label{eq:finite-part-determinant-mobius}
    \FP(A)
    =
    \log C(A)
    +\sum_{m\ge 2}\frac{\mu(m)}{m}
      \log\det(I-\lambda^{-m}A)^{-1}.
  \end{equation}
  Equivalently,
  \begin{equation}\label{eq:finite-part-primitive-closed}
    \FP(A)
    =
    \log C(A)
    +\sum_{n\ge 1}p_n(A)
      \bigl(\lambda^{-n}+\log(1-\lambda^{-n})\bigr).
  \end{equation}
  Both series converge absolutely. In particular, if
  \[
    \psi(x):=-x-\log(1-x)=\sum_{k\ge 2}\frac{x^k}{k},
    \qquad 0<x<1,
  \]
  then
  \begin{equation}\label{eq:finite-part-gap-series}
    \log C(A)-\FP(A)
    =
    \sum_{n\ge 1}p_n(A)\psi(\lambda^{-n}).
  \end{equation}
\end{proposition}

\begin{proof}
  Put \(\zeta_A(z)=\det(I-zA)^{-1}\). For \(|z|<\lambda^{-1}\), the
  Euler product and the determinant trace expansion give
  \[
    \log\zeta_A(z)
    =
    \sum_{n\ge 1}\frac{\Tr(A^n)}{n}z^n
    =
    \sum_{\ell\ge 1}p_\ell(A)\sum_{k\ge 1}\frac{z^{k\ell}}{k}.
  \]
  Therefore, still for \(|z|<\lambda^{-1}\),
  \[
    \sum_{m\ge 1}\frac{\mu(m)}{m}\log\zeta_A(z^m)
    =
    \sum_{\ell\ge 1}p_\ell(A)
      \sum_{q\ge 1}
        \frac{z^{q\ell}}{q}\sum_{m\mid q}\mu(m)
    =
    \sum_{\ell\ge 1}p_\ell(A)z^\ell.
  \]
  The last equality uses
  \(\sum_{m\mid q}\mu(m)=1\) for \(q=1\) and \(0\) otherwise.
  With \(z=r/\lambda\), this says
  \[
    P_A(r)
    =
    \log\zeta_A(r/\lambda)
    +\sum_{m\ge 2}\frac{\mu(m)}{m}
      \log\zeta_A\bigl((r/\lambda)^m\bigr).
  \]
  Since
  \[
    \log\zeta_A(r/\lambda)+\log(1-r)
    =
    \log\bigl((1-r)\zeta_A(r/\lambda)\bigr)
    \longrightarrow \log C(A)
  \]
  as \(r\uparrow 1\), it remains only to pass to the limit in the
  \(m\ge2\) sum. This is justified by the spectral-radius bound
  \[
    \log\zeta_A(\lambda^{-m})
    =
    \sum_{j\ge 1}\frac{\Tr(A^j)}{j}\lambda^{-mj}
    =
    O(\lambda^{1-m}),
    \qquad m\ge 2,
  \]
  so the determinant series is absolutely summable. Letting
  \(r\uparrow1\) gives \eqref{eq:finite-part-determinant-mobius}.

  For the primitive-orbit form, write the same Euler product at
  \(z=r/\lambda\) as
  \[
    \log\zeta_A(r/\lambda)
    =
    P_A(r)
    +\sum_{n\ge 1}p_n(A)\sum_{k\ge 2}
      \frac{r^{kn}\lambda^{-kn}}{k}.
  \]
  By \Cref{lem:primitive-orbit-asymptotic},
  \(\abs{p_n(A)}=O(\lambda^n/n)\). Since
  \(\sum_{k\ge 2}\lambda^{-kn}/k=O(\lambda^{-2n})\), the double tail is
  absolutely summable at \(r=1\). Hence
  \[
    \FP(A)
    =
    \log C(A)
    -\sum_{n\ge 1}p_n(A)\sum_{k\ge 2}\frac{\lambda^{-kn}}{k}.
  \]
  Finally,
  \[
    -\sum_{k\ge 2}\frac{x^k}{k}=x+\log(1-x),
    \qquad |x|<1,
  \]
  with \(x=\lambda^{-n}\), which proves
  \eqref{eq:finite-part-primitive-closed}. The displayed gap formula is
  the same identity rewritten with \(\psi\).
\end{proof}

\begin{proposition}[Reduced determinant identity]
  \label{prop:finite-part-residue-reduced-determinant}
  Let \(\mu_2,\dots,\mu_d\) be the non-Perron eigenvalues of \(A\). Then
  \[
    C(A)
    =\prod_{j=2}^d\left(1-\frac{\mu_j}{\lambda}\right)^{-1}
    =\det\nolimits'(I-A/\lambda)^{-1}.
  \]
\end{proposition}

\begin{proof}
  Factor
  \[
    \det(I-zA)
    =(1-\lambda z)\prod_{j=2}^d(1-\mu_j z).
  \]
  Multiplying by \(1-\lambda z\) and letting \(z\to\lambda^{-1}\) gives
  the claim.
\end{proof}

\begin{corollary}[Orbit-Mertens asymptotic]
  \label{cor:finite-part-mertens-asymptotic}
  \leavevmode\par\noindent
  Let \(A\) be a primitive non-negative matrix with Perron root
  \(\lambda>1\). Then
  \[
    \sum_{n\le N}\frac{p_n(A)}{\lambda^n}
    =\log N+\mathcal M(A)+o(1)
    \qquad (N\to\infty),
  \]
  where \(\mathcal M(A)=\gamma+\FP(A)\).
\end{corollary}

\begin{proof}
  By \Cref{lem:primitive-orbit-asymptotic},
  \[
    \sum_{n\ge 1}\left|\frac{p_n(A)}{\lambda^n}-\frac{1}{n}\right|
    <\infty.
  \]
  Hence
  \[
    \sum_{n\le N}\frac{p_n(A)}{\lambda^n}
    =
    \sum_{n\le N}\frac{1}{n}
    +\sum_{n\ge 1}
      \left(\frac{p_n(A)}{\lambda^n}-\frac{1}{n}\right)
    +o(1).
  \]
  The convergent correction term equals \(\FP(A)\), while
  \(\sum_{n\le N}1/n=\log N+\gamma+o(1)\).
\end{proof}

\subsection{Cyclic lifts and reduced spectral reconstruction}

Let
\[
  \Lambda_{\mathrm{rel}}(A):=\Lambda(A)/\lambda\in[0,1)
\]
and
\[
  u_n(A):=\sum_{j=2}^d\left(\frac{\mu_j}{\lambda}\right)^n
  \qquad (n\ge 1).
\]

\begin{proposition}[A formula for the reduced determinants of cyclic lifts]
  \label{thm:finite-part-cyclic-lift-reduced-constant-closed}
  For every \(q\ge 1\),
  \[
    C_\lambda(A^{\langle q\rangle})
    =\frac{1}{q}\prod_{j=2}^d
      \left(1-\left(\frac{\mu_j}{\lambda}\right)^q\right)^{-1}.
  \]
  Equivalently,
  \[
    \Psi_q(A):=\log q+\log C_\lambda(A^{\langle q\rangle})
    =-\sum_{j=2}^d
      \log\!\left(1-\left(\frac{\mu_j}{\lambda}\right)^q\right).
  \]
\end{proposition}

\begin{proof}
  The spectrum of \(A^{\langle q\rangle}=A\otimes P_q\) is
  \[
    \{\mu\,\omega_q^k:\mu\in\spec(A),\ 0\le k\le q-1\}.
  \]
  The eigenvalue \(\lambda\) of \(A^{\langle q\rangle}\) is simple.
  Applying \Cref{lem:simple-eigenvalue-reduced-constant} to this
  eigenvalue yields
  \[
    C_\lambda(A^{\langle q\rangle})
    =\prod_{k=1}^{q-1}(1-\omega_q^k)^{-1}
      \prod_{j=2}^d\prod_{k=0}^{q-1}
        \left(1-\frac{\mu_j}{\lambda}\omega_q^k\right)^{-1}.
  \]
  The first product equals \(1/q\), while
  \[
    \prod_{k=0}^{q-1}(1-x\omega_q^k)=1-x^q.
  \]
  Substituting \(x=\mu_j/\lambda\) proves the claim.
\end{proof}

\begin{proposition}[Multiple-sum expansion for cyclic lifts with
  explicit exponential remainder]
  \label{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}
  For every \(q\ge 1\),
  \[
    \Psi_q(A)=\sum_{m\ge 1}\frac{u_{qm}(A)}{m},
  \]
  and therefore
  \[
    \Psi_q(A)=u_q(A)+O\!\left(\Lambda_{\mathrm{rel}}(A)^{2q}\right)
    \qquad (q\to\infty).
  \]
  More precisely,
  \[
    \abs{\Psi_q(A)-u_q(A)}
    \le
    \frac{d-1}{2(1-\Lambda_{\mathrm{rel}}(A))}
    \Lambda_{\mathrm{rel}}(A)^{2q}.
  \]
\end{proposition}

\begin{proof}
  By \Cref{thm:finite-part-cyclic-lift-reduced-constant-closed},
  \[
    \Psi_q(A)
    =-\sum_{j=2}^d\log\!\left(1-\left(\frac{\mu_j}{\lambda}\right)^q\right).
  \]
  Expanding \(-\log(1-x)=\sum_{m\ge 1}x^m/m\) and exchanging the finite
  sum over \(j\) with the absolutely convergent series over \(m\) gives
  the first identity. The remainder estimate follows from
  \[
    |u_{qm}(A)|\le (d-1)\Lambda_{\mathrm{rel}}(A)^{qm}.
  \]
  Indeed,
  \[
    \begin{aligned}
      \abs{\Psi_q(A)-u_q(A)}
      &\le
      \sum_{m\ge 2}\frac{\abs{u_{qm}(A)}}{m} \\
      &\le
      \frac{d-1}{2}\sum_{m\ge 2}\Lambda_{\mathrm{rel}}(A)^{qm} \\
      &\le
      \frac{d-1}{2(1-\Lambda_{\mathrm{rel}}(A))}
      \Lambda_{\mathrm{rel}}(A)^{2q}.
    \end{aligned}
  \]
\end{proof}

\begin{proposition}[M\"obius inversion on multiples]
  \label{prop:finite-part-cyclic-lift-mobius-inversion}
  For every \(n\ge 1\),
  \[
    \frac{u_n(A)}{n}
    =\sum_{m\ge 1}\frac{\mu(m)}{mn}\,\Psi_{mn}(A).
  \]
\end{proposition}

\begin{proof}
  Set \(b_n:=u_n(A)/n\). Then
  \[
    \frac{\Psi_q(A)}{q}
    =\sum_{m\ge 1}b_{qm}
  \]
  by \Cref{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}. Hence
  \begin{align*}
    \sum_{m\ge 1}\frac{\mu(m)}{mn}\Psi_{mn}(A)
    &=
    \sum_{m\ge 1}\mu(m)\sum_{r\ge 1}b_{mnr} \\
    &=
    \sum_{s\ge 1}b_{ns}\sum_{m\mid s}\mu(m)
    =b_n,
  \end{align*}
  since \(\sum_{m\mid s}\mu(m)=0\) unless \(s=1\).
\end{proof}

\begin{lemma}[Limsup of a finite exponential sum]
  \label{lem:finite-part-exponential-sum-limsup}
  Let
  \[
    f_q=\sum_{j=1}^r c_j\alpha_j^q
    \qquad (q\ge 1),
  \]
  where \(c_j\in\CC\setminus\{0\}\) and \(\alpha_j\in\CC\). Then
  \[
    \limsup_{q\to\infty}\abs{f_q}^{1/q}
    =\max_{1\le j\le r}\abs{\alpha_j}.
  \]
\end{lemma}

\begin{proof}
  Set \(R:=\max_j\abs{\alpha_j}\). The upper bound
  \(\limsup_{q\to\infty}\abs{f_q}^{1/q}\le R\) is immediate from
  \(\abs{f_q}\le (\sum_j\abs{c_j})R^q\).
  If \(R=0\), then every \(\alpha_j=0\), so \(f_q=0\) for all \(q\ge 1\)
  and the asserted equality is immediate.

  For the reverse inequality, collect equal frequencies on the boundary
  circle and write
  \[
    f_q
    =R^q\bigl(g_q+O(\vartheta^q)\bigr),
    \qquad
    g_q:=\sum_{k=1}^s d_k\beta_k^q,
  \]
  where \(\vartheta\in[0,1)\), the numbers \(\beta_1,\dots,\beta_s\in\TT\)
  are pairwise distinct, and \(d_k\neq 0\). Then
  \[
    \frac{1}{N}\sum_{q=1}^N\abs{g_q}^2
    =\sum_{k,\ell=1}^s d_k\overline{d_\ell}\,
      \frac{1}{N}\sum_{q=1}^N(\beta_k\overline{\beta_\ell})^q.
  \]
  As \(N\to\infty\), the inner Ces\`aro average tends to \(1\) when
  \(k=\ell\) and to \(0\) otherwise. Hence
  \[
    \frac{1}{N}\sum_{q=1}^N\abs{g_q}^2
    \longrightarrow
    \sum_{k=1}^s\abs{d_k}^2>0.
  \]
  Therefore \(\limsup_{q\to\infty}\abs{g_q}>0\), so there exist
  \(c>0\) and infinitely many \(q\) such that \(\abs{g_q}\ge c\). Along
  that subsequence,
  \[
    \abs{f_q}\ge \frac{c}{2}R^q
  \]
  for all sufficiently large \(q\), because \(\vartheta^q\to 0\). This
  yields
  \(\limsup_{q\to\infty}\abs{f_q}^{1/q}\ge R\), completing the proof.
\end{proof}

\begin{proposition}[Fixed-matrix recovery of the reduced spectrum from the
  full cyclic-lift record with explicit remainder]
  \label{thm:finite-part-cyclic-lift-spectrum-identifiability}
  For one fixed primitive non-negative matrix \(A\), the full recorded
  \(\{\Psi_q(A)\}_{q\ge 1}\), equivalently
  \(\{C_\lambda(A^{\langle q\rangle})\}_{q\ge 1}\), determines the
  reduced spectral polynomial \(F_A\) of that same matrix and hence the
  non-Perron spectrum of \(A\) with algebraic multiplicity. Moreover,
  \[
    \Psi_q(A)=u_q(A)+O\!\left(\Lambda_{\mathrm{rel}}(A)^{2q}\right),
  \]
  with the explicit bound from
  \Cref{prop:finite-part-cyclic-lift-dirichlet-multiple-sum}, and
  \[
    \limsup_{q\to\infty}\abs{\Psi_q(A)}^{1/q}
    =\Lambda_{\mathrm{rel}}(A).
  \]
\end{proposition}

\begin{proof}
  By \Cref{prop:finite-part-cyclic-lift-mobius-inversion}, the cyclic
  data determine every power sum \(u_n(A)\). Since
  \[
    F_A(t)=\prod_{j=2}^d\left(1-\frac{\mu_j}{\lambda}t\right)
  \]
  has known degree \(d-1\), Newton's identities recover its
  coefficients from the first \(d-1\) power sums, and hence from the
  full sequence \(\{u_n(A)\}_{n\ge 1}\). This reconstructs the
  non-Perron spectrum with algebraic multiplicity. The argument is
  purely fixed-record: it recovers one polynomial \(F_A\) from the one
  sequence attached to the same matrix \(A\).

  For the exponential growth rate,
  \(u_q(A)=\sum_{j=2}^d(\mu_j/\lambda)^q\), so
  \Cref{lem:finite-part-exponential-sum-limsup} gives
  \[
    \limsup_{q\to\infty}\abs{u_q(A)}^{1/q}
    =\Lambda_{\mathrm{rel}}(A).
  \]
  The explicit remainder bound from
  \Cref{prop:finite-part-cyclic-lift-dirichlet-multiple-sum} shows that
  \[
    \Psi_q(A)-u_q(A)
    =O\!\left(\Lambda_{\mathrm{rel}}(A)^{2q}\right).
  \]
  If \(\Lambda_{\mathrm{rel}}(A)=0\), then both sequences are
  eventually zero. Otherwise the error term has strictly smaller
  exponential rate than \(u_q(A)\), so
  \[
    \limsup_{q\to\infty}\abs{\Psi_q(A)}^{1/q}
    =\Lambda_{\mathrm{rel}}(A).
  \]
\end{proof}

\subsection{One fixed root-of-unity layer}

This subsection isolates the elementary finite Fourier step from the
cyclic-lift reconstruction above. It is used only in the canonical
cyclic-lift model of
\Cref{rem:abelian-toy-model-cyclic,prop:canonical-cyclic-lift-root-layer},
where the original edge labels have been replaced by the constant label
\(1\). The argument is simply coefficient recovery for a polynomial
sampled on a large enough cyclic subgroup of the unit circle.

\begin{corollary}[Auxiliary fixed-layer discrete Fourier recovery]
  \label{thm:finite-part-cyclic-lift-single-q-torsion-reconstruction}
  Write
  \[
    F_A(t)=\sum_{n=0}^{d-1}a_n t^n
    \qquad\text{with } a_0=1.
  \]
  If \(q\ge d\), then for this fixed \(q\) the values
  \(\{F_A(\omega_q^j)\}_{0\le j\le q-1}\) determine every coefficient
  \(a_n\). More precisely,
  \[
    a_n=\frac{1}{q}\sum_{j=0}^{q-1}
      F_A(\omega_q^j)\omega_q^{-jn}
    \qquad (0\le n\le d-1).
  \]
\end{corollary}

\begin{proof}
  The polynomial identity
  \[
    F_A(\omega_q^j)=\sum_{n=0}^{d-1}a_n\omega_q^{jn}
  \]
  holds for \(0\le j\le q-1\). Averaging against
  \(\omega_q^{-jm}\) and using discrete orthogonality gives
  \[
    \frac{1}{q}\sum_{j=0}^{q-1}
      F_A(\omega_q^j)\omega_q^{-jm}
    =
    \sum_{n=0}^{d-1}a_n
      \frac{1}{q}\sum_{j=0}^{q-1}\omega_q^{j(n-m)}.
  \]
  Since \(q\ge d\), the inner sum equals \(1\) when \(n=m\) and \(0\)
  otherwise.
\end{proof}

\noindent
Thus this one-layer statement is exactly finite Fourier inversion for a
degree-\((d-1)\) polynomial. It is recorded to identify the canonical
cyclic-lift comparison case; it is not used in the Adams--M\"obius
construction or in the proof of the class-Mertens theorem or its
character-orthogonality readout. In particular, it is a recovery formula
for one fixed recorded layer, not an ambient inverse statement for
varying families of samples, even when those samples are taken at one
fixed radius \(q\).

\begin{corollary}[A fixed root-of-unity layer of \(\zeta_A\) recovers the
  reduced spectrum]
  \label{cor:finite-part-cyclic-lift-zeta-torsion-determines-spectrum}
  For \(1\le j\le q-1\),
  \[
    F_A(\omega_q^j)
    =\frac{\det(I-(\omega_q^j/\lambda)A)}{1-\omega_q^j}
    =\frac{1}{1-\omega_q^j}\,
      \zeta_A(\omega_q^j/\lambda)^{-1},
  \]
  while
  \[
    F_A(1)=\det\nolimits'(I-A/\lambda)=C(A)^{-1}.
  \]
  Consequently, if \(q\ge d\), then the data
  \[
    \Bigl(C(A),\ \{\zeta_A(\omega_q^j/\lambda)\}_{j=1}^{q-1}\Bigr)
  \]
  recover the non-Perron spectrum of the fixed matrix \(A\) with
  multiplicity.
\end{corollary}

\begin{proof}
  The displayed identities are immediate from the definitions of
  \(F_A\), \(\zeta_A\), and \(C(A)\). The reconstruction statement
  follows from
  \Cref{thm:finite-part-cyclic-lift-single-q-torsion-reconstruction}.
\end{proof}

\begin{remark}[A fixed-layer cyclic toy example]
  \label{rem:finite-part-fibonacci-example}
  Let
  \[
    A=\begin{pmatrix}
      1 & 1 \\
      1 & 0
    \end{pmatrix}.
  \]
  Then \(d=2\), the Perron root is
  \(\lambda=(1+\sqrt5)/2\), the non-Perron eigenvalue is
  \(\mu=-\lambda^{-1}\), and
  \[
    F_A(t)=1+\lambda^{-2}t.
  \]
  Since \(q=2\ge d\), the single layer \(\{F_A(1),F_A(-1)\}\) already
  reconstructs \(F_A\):
  \[
    F_A(1)=1+\lambda^{-2},
    \qquad
    F_A(-1)=1-\lambda^{-2},
  \]
  so discrete Fourier inversion gives
  \[
    a_0=\frac{F_A(1)+F_A(-1)}{2}=1,
    \qquad
    a_1=\frac{F_A(1)-F_A(-1)}{2}=\lambda^{-2}.
  \]
  Thus one order-\(2\) root-of-unity layer already recovers the reduced
  polynomial and hence the non-Perron eigenvalue \(\mu=-\lambda^{-1}\).
\end{remark}

\subsection{Selected scalar remarks}

\begin{remark}[Consequences retained only for orientation]
  The one-layer reconstruction above also decides any condition expressible
  only in terms of the roots of \(F_A\). For example, when \(q\ge d\), the
  values \(\{F_A(\omega_q^j)\}_{0\le j\le q-1}\) determine whether
  \[
    \Lambda(A)\le \lambda^{1/2},
    \qquad\text{equivalently}\qquad
    \max_{j\ge 2}\left|\frac{\mu_j}{\lambda}\right|
      \le \lambda^{-1/2}.
  \]
  This is a scalar spectral test for the canonical cyclic-lift comparison;
  it is not part of the finite-group Adams--M\"obius theorem chain.

  Similarly, when \(A\) is an integer adjacency matrix,
  \eqref{eq:finite-part-gap-series} gives
  \[
    \log C(A)-\FP(A)
    =\sum_{n\ge 1}p_n(A)\psi(\lambda^{-n}),
    \qquad
    \psi(x)=-x-\log(1-x).
  \]
  Since \(p_n(A)\ge0\), \(p_n(A)>0\) for all sufficiently large \(n\), and
  \(\psi(x)>0\) for \(0<x<1\), this gap is positive. The observation is a
  useful check on the scalar normalisation \(\mathcal M(A)=\gamma+\FP(A)\),
  not an independent primitive reconstruction result.
\end{remark}
