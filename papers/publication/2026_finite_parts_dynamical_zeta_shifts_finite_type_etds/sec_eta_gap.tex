\section{Rigidity of the Twisted Spectral Gap}
\label{sec:eta-gap}

The strict inequality
\(\eta=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda\) is the
input that permits evaluation of non-trivial Peter--Weyl channels at
the Perron point. The boundary case is rigid: equality is equivalent to
the existence of a cocycle-equivariant line field.

\begin{theorem}[Rigidity at the Perron boundary]
  \label{thm:eta-gap-rigidity}
  Let \(\varrho:G\to U(V_\varrho)\) be an irreducible unitary
  representation. Then
  \[
    \rho(A_\varrho)=\lambda
  \]
  if and only if there exist \(\omega\in\TT\) and unit vectors
  \(\xi_i\in V_\varrho\), indexed by the vertices \(i\in V\), such that
  \begin{equation}\label{eq:eta-gap-line-field}
    \varrho(\tau(e))\xi_j=\omega\,\xi_i
    \qquad
    \text{for every admissible edge } e:i\to j.
  \end{equation}
  Whenever \eqref{eq:eta-gap-line-field} holds, every closed path
  \(\gamma\) of length \(n\) based at \(i\) satisfies
  \begin{equation}\label{eq:eta-gap-loop-character}
    \varrho(\tau(\gamma))\xi_i=\omega^n\xi_i.
  \end{equation}
  In particular, the subgroup
  \[
    \Gamma_i(\tau)
    :=
    \bigl\langle \tau(\gamma):
      \gamma \text{ closed path based at } i
    \bigr\rangle
  \]
  acts on the line \(\CC\xi_i\) through a one-dimensional character.
\end{theorem}

\begin{proof}
  Assume first that \(\rho(A_\varrho)=\lambda\). Since
  \(A_\varrho\) is finite dimensional, some eigenvalue
  \(\alpha\in\spec(A_\varrho)\) satisfies \(|\alpha|=\lambda\). Let
  \(v=(v_i)_{i\in V}\in \CC^V\otimes V_\varrho\) be a corresponding
  eigenvector, so
  \[
    \alpha v_i=\sum_{e:i\to j}\varrho(\tau(e))v_j
    \qquad (i\in V).
  \]
  Set \(u_i:=\norm{v_i}\). Then
  \[
    \lambda u_i
    =
    \norm{\alpha v_i}
    =
    \left\|\sum_{e:i\to j}\varrho(\tau(e))v_j\right\|
    \le
    \sum_{e:i\to j}\norm{v_j}
    =(Au)_i
  \]
  for every \(i\in V\). Let \(\ell>0\) be a left Perron vector of
  \(A\), normalised by \(\ell^{\mathsf T}A=\lambda\ell^{\mathsf T}\).
  Then
  \[
    \lambda\,\ell^{\mathsf T}u
    \le
    \ell^{\mathsf T}Au
    =
    \lambda\,\ell^{\mathsf T}u,
  \]
  so every coordinate inequality is an equality because \(\ell_i>0\)
  for all \(i\). Hence \(Au=\lambda u\). Primitivity of \(A\) implies
  that \(u\) is a positive Perron eigenvector, so \(u_i>0\) for every
  \(i\in V\).

  Equality in the triangle inequality now holds in every vertex
  equation. Therefore, for each fixed \(i\), all non-zero vectors
  \(\varrho(\tau(e))v_j\) with \(e:i\to j\) lie on the same ray. Since
  each such vector has norm \(u_j\) and their sum is \(\alpha v_i\),
  one has
  \[
    \varrho(\tau(e))v_j
    =
    u_j\,\frac{\alpha v_i}{\lambda u_i}
    \qquad
    \text{for every admissible edge } e:i\to j.
  \]
  Write \(\omega:=\alpha/\lambda\in\TT\) and
  \(\xi_i:=v_i/u_i\in V_\varrho\). Dividing by \(u_j\) yields
  \eqref{eq:eta-gap-line-field}.

  Conversely, assume that \eqref{eq:eta-gap-line-field} holds. Let
  \(h=(h_i)_{i\in V}>0\) be a right Perron vector of \(A\), so
  \(Ah=\lambda h\). Define \(w=(w_i)_{i\in V}\) by
  \(w_i:=h_i\xi_i\). Then
  \begin{align*}
    (A_\varrho w)_i
    &=
    \sum_{e:i\to j}\varrho(\tau(e))w_j \\
    &=
    \sum_{e:i\to j}h_j\varrho(\tau(e))\xi_j \\
    &=
    \omega\sum_{e:i\to j}h_j\xi_i \\
    &=
    \omega(Ah)_i\xi_i \\
    &=
    \omega\lambda w_i.
  \end{align*}
  Thus \(\omega\lambda\in\spec(A_\varrho)\), so
  \(\rho(A_\varrho)\ge \lambda\). On the other hand,
  \Cref{prop:twisted-spectral-radius-bound} gives
  \(\rho(A_\varrho)\le \lambda\). Hence
  \(\rho(A_\varrho)=\lambda\).

  Finally, iterating \eqref{eq:eta-gap-line-field} along a closed path
  \(\gamma=e_1\cdots e_n\) based at \(i\) gives
  \eqref{eq:eta-gap-loop-character}. Hence the line \(\CC\xi_i\) is
  \(\Gamma_i(\tau)\)-invariant. The action of \(\Gamma_i(\tau)\) on
  this line defines a one-dimensional character.
\end{proof}

\begin{corollary}[Perfect loop groups force a strict twisted gap]
  \label{cor:eta-gap-perfect-group}
  Assume that \(\Gamma_{i_0}(\tau)=G\) for some vertex \(i_0\). If
  \(G\) is perfect, then
  \[
    \rho(A_\varrho)<\lambda
    \qquad
    \text{for every non-trivial } \varrho\in\Irr(G).
  \]
  In particular,
  \[
    \eta=\max_{\varrho\neq\mathbf 1}\rho(A_\varrho)<\lambda.
  \]
  Consequently, the Perron-point formulas of
  \Cref{thm:class-mertens-explicit,thm:class-mertens-fourier} apply
  automatically to such extensions.
\end{corollary}

\begin{proof}
  Suppose that \(\rho(A_\varrho)=\lambda\) for some non-trivial
  \(\varrho\in\Irr(G)\). By \Cref{thm:eta-gap-rigidity}, the subgroup
  \(\Gamma_{i_0}(\tau)=G\) preserves a one-dimensional subspace of
  \(V_\varrho\). The induced action of \(G\) on that line is a
  one-dimensional character. Since \(G\) is perfect, every
  one-dimensional character of \(G\) is trivial. Therefore
  \(V_\varrho\) contains a non-zero \(G\)-fixed vector, which is
  impossible for a non-trivial irreducible representation.
\end{proof}

\begin{corollary}[One-dimensional equality criterion]
  \label{cor:eta-gap-character-classification}
  Let \(\chi:G\to\TT\) be a one-dimensional character. Then
  \[
    \rho(A_\chi)=\lambda
  \]
  if and only if there exist \(\omega\in\TT\) and a function
  \(b:V\to\TT\) such that, for every admissible edge \(e:i\to j\),
  \begin{equation}\label{eq:eta-gap-character-classification}
    \chi(\tau(e))=\omega\,b(i)^{-1}b(j).
  \end{equation}
  Equivalently, if \(D:=\diag(b(i))\), then
  \begin{equation}\label{eq:eta-gap-character-similarity}
    A_\chi=\omega\,D^{-1}AD.
  \end{equation}
\end{corollary}

\begin{proof}
  If \(\rho(A_\chi)=\lambda\), then
  \Cref{thm:eta-gap-rigidity} gives unit vectors
  \(\xi_i\in\CC\), hence numbers \(b(i)\in\TT\) with
  \(\xi_i=b(i)^{-1}\). The line-field relation
  \eqref{eq:eta-gap-line-field} becomes
  \[
    \chi(\tau(e))\,b(j)^{-1}=\omega\,b(i)^{-1},
  \]
  which is exactly \eqref{eq:eta-gap-character-classification}.

  Conversely, if \eqref{eq:eta-gap-character-classification} holds,
  then for each admissible pair \(i\to j\),
  \[
    (A_\chi)_{ij}
    =
    \sum_{e:i\to j}\chi(\tau(e))
    =
    \omega\,b(i)^{-1}
      \left(\sum_{e:i\to j}1\right)b(j)
    =
    \omega\,b(i)^{-1}A_{ij}b(j).
  \]
  This is \eqref{eq:eta-gap-character-similarity}, so \(A_\chi\) is
  similar to \(\omega A\). Therefore \(\rho(A_\chi)=\lambda\).
\end{proof}
