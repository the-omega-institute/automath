\section{H\"older Transfer Operators}
\label{sec:holder-transfer}

The primitive inversion formalism extends beyond finite matrices. Let
\((X_A^+,\sigma)\) be the mixing one-sided shift attached to the
primitive matrix \(A\), let \(\varphi:X_A^+\to\RR\) be a real
H\"older potential, and write \(P:=P(\varphi)\) for its topological
pressure. Let \(\tau:X_A^+\to G\) be a locally constant cocycle into a
finite group. Fix \(\alpha\in(0,1]\) such that
\(\varphi\in C^\alpha\). For \(\varrho\in\Irr(G)\), the twisted Ruelle
operator on \(C^\alpha(X_A^+,V_\varrho)\) is
\[
  (\mathscr L_{\varphi,\varrho}F)(x)
  :=
  \sum_{\sigma y=x}e^{\varphi(y)}\varrho(\tau(y))F(y).
\]
For a class function \(\phi:G\to\CC\), define
\[
  T_{\phi,\varphi,n}
  :=
  \sum_{x\in\operatorname{Fix}(\sigma^n)}
    e^{S_n\varphi(x)}\phi(\tau_n(x)),
  \qquad
  \mathcal L_{\phi,\varphi}(z)
  :=
  \sum_{n\ge 1}\frac{T_{\phi,\varphi,n}}{n}z^n.
\]
If \(\gamma\) is a primitive orbit of length \(|\gamma|=n\), then
\(S_\gamma\varphi:=S_n\varphi(x)\) is independent of the choice of
\(x\in\gamma\), and the weighted primitive series is
\[
  \Pi_{\phi,\varphi}(z)
  :=
  \sum_{\gamma\in\cP}
    e^{S_\gamma\varphi}\phi(\tau(\gamma))z^{|\gamma|}.
\]
For \(\varrho\in\Irr(G)\), write
\[
  \Pi_{\varrho,\varphi}(z):=\Pi_{\chi_\varrho,\varphi}(z)
  =\sum_{n\ge 1}\pi_{n,\varrho}(\varphi)z^n.
\]

\begin{theorem}[H\"older transfer-operator Perron-point theorem]
  \label{thm:holder-transfer-perron-point}
  Assume that
  \[
    \eta_\varphi
    :=
    \max_{\varrho\neq\mathbf 1}\rho(\mathscr L_{\varphi,\varrho})
    <e^P.
  \]
  Then there exists \(\Theta_\varphi\in(0,e^P)\) with the following
  properties.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item For every class function \(\phi:G\to\CC\) and every
      \(|z|<e^{-P}\),
      \begin{equation}\label{eq:holder-transfer-adams-mobius}
        \Pi_{\phi,\varphi}(z)
        =
        \sum_{m\ge 1}\frac{\mu(m)}{m}\,
          \mathcal L_{\psi^m\phi,\varphi}(z^m).
      \end{equation}
    \item For every non-trivial \(\varrho\in\Irr(G)\), the series
      \[
        \Pi_{\varrho,\varphi}(e^{-P})
        =
        \sum_{n\ge 1}\pi_{n,\varrho}(\varphi)e^{-Pn}
      \]
      converges absolutely.
    \item If
      \[
        p_n(\varphi)
        :=
        \sum_{\substack{\gamma\in\cP\\ |\gamma|=n}}
          e^{S_\gamma\varphi}
      \]
      and, for a conjugacy class \(C\subset G\),
      \[
        p_{n,C}(\varphi)
        :=
        \sum_{\substack{\gamma\in\cP\\ |\gamma|=n\\
          \tau(\gamma)\in C}}
          e^{S_\gamma\varphi},
      \]
      then
      \begin{equation}\label{eq:holder-transfer-pointwise}
        p_{n,C}(\varphi)
        =
        \frac{|C|}{|G|}\frac{e^{Pn}}{n}
        +O\!\left(\frac{\Theta_\varphi^n}{n}\right),
      \end{equation}
      and
      \begin{equation}\label{eq:holder-transfer-mertens}
        \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
        =
        \frac{|C|}{|G|}H_N
        +\mathcal M_C(\varphi)-\frac{|C|}{|G|}\gamma
        +O\!\left(\frac{(\Theta_\varphi e^{-P})^N}{N}\right),
      \end{equation}
      where
      \[
        \mathcal M(\varphi)
        :=
        \gamma+\sum_{n\ge 1}
          \left(e^{-Pn}p_n(\varphi)-\frac{1}{n}\right)
      \]
      and
      \begin{equation}\label{eq:holder-transfer-constant}
        \mathcal M_C(\varphi)
        =
        \frac{|C|}{|G|}\mathcal M(\varphi)
        +\frac{|C|}{|G|}
          \sum_{\varrho\neq\mathbf 1}
            \chi_\varrho^\ast(C)\Pi_{\varrho,\varphi}(e^{-P}).
      \end{equation}
      In particular,
      \[
        \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
        =
        \frac{|C|}{|G|}\log N+\mathcal M_C(\varphi)+O(N^{-1}).
      \]
    \item If
      \[
        \Delta_C(\varphi)
        :=
        \mathcal M_C(\varphi)-\frac{|C|}{|G|}\mathcal M(\varphi),
      \]
      then for every non-trivial \(\sigma\in\Irr(G)\),
      \begin{equation}\label{eq:holder-transfer-fourier}
        \Pi_{\sigma,\varphi}(e^{-P})
        =
        \sum_{C\subset G}\Delta_C(\varphi)\chi_\sigma(C),
      \end{equation}
      where the sum runs over conjugacy classes.
  \end{enumerate}
\end{theorem}

\begin{proof}
  Classical dynamical determinant theory for H\"older potentials on
  mixing shifts of finite type, applied to the operators
  \(\mathscr L_{\varphi,\varrho}\), gives a number \(\Theta_0<e^P\) such
  that
  \[
    T_{\mathbf 1,\varphi,n}=e^{Pn}+O(\Theta_0^n),
    \qquad
    T_{\chi_\varrho,\varphi,n}=O(\Theta_0^n)
    \quad (\varrho\neq\mathbf 1);
  \]
  see \cite{Ruelle1978Thermodynamic,Baladi2000PositiveTransfer,%
  Baladi2018ZetaDeterminants,AdachiSunada1987TwistedPF}.
  Set
  \[
    \Theta_\varphi:=\max\{\Theta_0,e^{P/2}\}<e^P.
  \]
  The estimate
  \[
    |T_{\chi_\sigma,\varphi,n}|
    \le (\dim\sigma)\,T_{\mathbf 1,\varphi,n}
    =O(e^{Pn})
  \]
  holds uniformly in \(\sigma\in\Irr(G)\). Since every class function is
  a finite linear combination of irreducible characters, the same
  \(O(e^{Pn})\) bound holds for \(T_{\phi,\varphi,n}\). Hence
  \(\mathcal L_{\phi,\varphi}(z)\) converges absolutely for
  \(|z|<e^{-P}\).

  Part~\textup{(i)} is purely orbit-theoretic. If \(\gamma\) is
  primitive of length \(\ell\), then its \(r\)-fold repetition
  contributes \(\ell\) points to \(\operatorname{Fix}(\sigma^{r\ell})\),
  each with weight \(e^{rS_\gamma\varphi}\) and label conjugate to
  \(\tau(\gamma)^r\). Therefore
  \[
    \mathcal L_{\phi,\varphi}(z)
    =
    \sum_{\gamma\in\cP}\sum_{r\ge 1}
      \frac{e^{rS_\gamma\varphi}\phi(\tau(\gamma)^r)}{r}z^{r|\gamma|},
    \qquad |z|<e^{-P},
  \]
  and the same M\"obius calculation as in the proof of
  \Cref{cor:class-function-adams-mobius} yields
  \eqref{eq:holder-transfer-adams-mobius}.

  Coefficient extraction in
  \eqref{eq:holder-transfer-adams-mobius}, applied to
  \(\phi=\chi_\varrho\) and the Adams decomposition
  \eqref{eq:adams-character-decomposition}, gives
  \[
    \pi_{n,\varrho}(\varphi)
    =
    \frac{1}{n}\sum_{m\mid n}\mu(m)
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}T_{\chi_\sigma,\varphi,n/m}.
  \]
  Since \(a_{\varrho,\sigma}^{(1)}=\delta_{\varrho,\sigma}\), the
  divisor \(m=1\) contributes \(O(\Theta_\varphi^n/n)\) when
  \(\varrho\neq\mathbf 1\). For \(m>1\), the bounds above and
  \Cref{lem:adams-coefficients-bounded} give
  \[
    \frac{1}{n}
    \sum_{\substack{m\mid n\\ m>1}}
      \left|
        \sum_{\sigma\in\Irr(G)}
          a_{\varrho,\sigma}^{(m)}T_{\chi_\sigma,\varphi,n/m}
      \right|
    =
    O\!\left(\frac{e^{Pn/2}}{n}\right),
  \]
  because \(n/m\le n/2\) for \(m>1\). Hence
  \[
    \pi_{n,\varrho}(\varphi)
    =O\!\left(\frac{\Theta_\varphi^n}{n}\right)
    \qquad (\varrho\neq\mathbf 1).
  \]
  This proves the absolute convergence in part~\textup{(ii)} after
  multiplication by \(e^{-Pn}\).

  For the trivial representation,
  \[
    p_n(\varphi)=\pi_{n,\mathbf 1}(\varphi)
    =
    \frac{e^{Pn}}{n}+O\!\left(\frac{\Theta_\varphi^n}{n}\right).
  \]
  Therefore
  \[
    e^{-Pn}p_n(\varphi)
    =
    \frac{1}{n}+O\!\left(\frac{(\Theta_\varphi e^{-P})^n}{n}\right).
  \]
  Hence the defining series for \(\mathcal M(\varphi)\) converges
  absolutely. Summing this estimate and applying
  \Cref{lem:geometric-harmonic-tail} gives
  \[
    \sum_{n\le N}e^{-Pn}p_n(\varphi)
    =
    H_N+\mathcal M(\varphi)-\gamma
    +O\!\left(\frac{(\Theta_\varphi e^{-P})^N}{N}\right).
  \]

  The weighted class decomposition is
  \[
    p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho^\ast(C)\pi_{n,\varrho}(\varphi),
  \]
  by the same argument as in
  \Cref{thm:class-primitive-decomposition}. The trivial channel gives
  the main term \((|C|/|G|)e^{Pn}/n\), while the non-trivial channels are
  \(O(\Theta_\varphi^n/n)\), proving
  \eqref{eq:holder-transfer-pointwise}. Summing after multiplication by
  \(e^{-Pn}\) gives
  \[
    \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{n\le N}e^{-Pn}p_n(\varphi)
    +\frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}\chi_\varrho^\ast(C)
        \sum_{n\le N}\pi_{n,\varrho}(\varphi)e^{-Pn}.
  \]
  Part~\textup{(ii)} and \Cref{lem:geometric-harmonic-tail} imply
  \[
    \sum_{n\le N}\pi_{n,\varrho}(\varphi)e^{-Pn}
    =
    \Pi_{\varrho,\varphi}(e^{-P})
    +O\!\left(\frac{(\Theta_\varphi e^{-P})^N}{N}\right)
  \]
  for each non-trivial \(\varrho\). Substituting the scalar asymptotic
  and collecting the constant term gives
  \eqref{eq:holder-transfer-mertens} and
  \eqref{eq:holder-transfer-constant}. Since
  \(H_N=\log N+\gamma+O(N^{-1})\), the final logarithmic form follows.

  Finally,
  \[
    \Delta_C(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\neq\mathbf 1}
        \chi_\varrho^\ast(C)\Pi_{\varrho,\varphi}(e^{-P})
  \]
  by \eqref{eq:holder-transfer-constant}. Multiplying by
  \(\chi_\sigma(C)\), summing over conjugacy classes, and invoking
  character orthogonality yields
  \eqref{eq:holder-transfer-fourier}.
\end{proof}

The next theorem isolates the higher spectral windows of the twisted
Ruelle operators in the same manner as
\Cref{thm:class-mertens-secondary-term} does in the finite-state
setting.

\begin{theorem}[H\"older spectral-window expansion above the square-root barrier]
  \label{thm:holder-transfer-spectral-window}
  Assume that
  \[
    \eta_\varphi
    :=
    \max_{\varrho\neq\mathbf 1}\rho(\mathscr L_{\varphi,\varrho})
    <e^P.
  \]
  Let
  \[
    r_{\mathrm{ess}}^\ast
    :=
    \max_{\varrho\in\Irr(G)}
      r_{\mathrm{ess}}(\mathscr L_{\varphi,\varrho}),
  \]
  and fix \(R\) such that
  \[
    \max\{r_{\mathrm{ess}}^\ast,e^{P/2}\}<R<e^P,
  \]
  with the additional property that no eigenvalue of any
  \(\mathscr L_{\varphi,\varrho}\) has modulus \(R\).
  For the trivial channel set
  \[
    \mathcal E_{\mathbf 1}(R)
    :=
    \{\beta\in\spec(\mathscr L_{\varphi,\mathbf 1}):R<|\beta|<e^P\},
  \]
  and for every non-trivial \(\varrho\in\Irr(G)\) set
  \[
    \mathcal E_\varrho(R)
    :=
    \{\beta\in\spec(\mathscr L_{\varphi,\varrho}):|\beta|>R\}.
  \]
  For \(\beta\in\mathcal E_\varrho(R)\), let
  \(\mathscr P_{\varrho,\beta}\) be the corresponding Riesz projector,
  and define, for each conjugacy class \(C\subset G\),
  \begin{equation}\label{eq:holder-transfer-spectral-window-profile}
    \mathfrak s_{C,\varphi,R}(n)
    :=
    \frac{|C|}{|G|}
    \left(
      \sum_{\beta\in\mathcal E_{\mathbf 1}(R)}
        \Tr\bigl(\mathscr L_{\varphi,\mathbf 1}^n
          \mathscr P_{\mathbf 1,\beta}\bigr)
      +
      \sum_{\varrho\neq\mathbf 1}\chi_\varrho^\ast(C)
        \sum_{\beta\in\mathcal E_\varrho(R)}
          \Tr\bigl(\mathscr L_{\varphi,\varrho}^n
            \mathscr P_{\varrho,\beta}\bigr)
    \right).
  \end{equation}
  Then there exists \(\Gamma_{\varphi,R}\in(0,R)\) such that, for every
  conjugacy class \(C\subset G\),
  \begin{equation}\label{eq:holder-transfer-spectral-window-pointwise}
    p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}\frac{e^{Pn}}{n}
    +\frac{\mathfrak s_{C,\varphi,R}(n)}{n}
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right).
  \end{equation}
  Moreover, the series
  \begin{equation}\label{eq:holder-transfer-spectral-window-tail}
    \mathfrak T_{C,\varphi,R}(N)
    :=
    \sum_{n>N}
      \frac{\mathfrak s_{C,\varphi,R}(n)}{n\,e^{Pn}}
  \end{equation}
  converges absolutely, and
  \begin{equation}\label{eq:holder-transfer-spectral-window-harmonic}
    \sum_{n\le N}e^{-Pn}p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}H_N
    +\mathcal M_C(\varphi)-\frac{|C|}{|G|}\gamma
    -\mathfrak T_{C,\varphi,R}(N)
    +O\!\left(\frac{(\Gamma_{\varphi,R}e^{-P})^N}{N}\right).
  \end{equation}
  For every \(\varrho\in\Irr(G)\) and every
  \(\beta\in\mathcal E_\varrho(R)\),
  \[
    \Tr\bigl(\mathscr L_{\varphi,\varrho}^n\mathscr P_{\varrho,\beta}\bigr)
    =m_\varrho(\beta)\beta^n,
  \]
  where \(m_\varrho(\beta)\) denotes the algebraic multiplicity of
  \(\beta\).
\end{theorem}

\begin{proof}
  Standard dynamical determinant theory for H\"older Ruelle operators
  on mixing shifts of finite type implies that, for every
  \(\varrho\in\Irr(G)\), the determinant
  \[
    D_\varrho(z)
    :=
    \exp\!\bigl(-\mathcal L_{\chi_\varrho,\varphi}(z)\bigr)
  \]
  extends holomorphically to the disc
  \(|z|<r_{\mathrm{ess}}(\mathscr L_{\varphi,\varrho})^{-1}\), and that
  its zeros there are precisely the reciprocals of the eigenvalues of
  \(\mathscr L_{\varphi,\varrho}\), counted with algebraic multiplicity;
  see
  \cite{Ruelle1978Thermodynamic,Baladi2000PositiveTransfer,%
  Baladi2018ZetaDeterminants,AdachiSunada1987TwistedPF}.
  Since \(R>r_{\mathrm{ess}}^\ast\), each set
  \(\mathcal E_\varrho(R)\) is finite. The absence of eigenvalues on
  the circle \(|z|=R\), together with the fact that isolated
  eigenvalues can accumulate only at the essential spectrum, therefore
  permits the choice of a number
  \[
    \Gamma_{\varphi,R}\in(\max\{r_{\mathrm{ess}}^\ast,e^{P/2}\},R)
  \]
  such that no eigenvalue of any \(\mathscr L_{\varphi,\varrho}\) has
  modulus in the interval \([\Gamma_{\varphi,R},R]\).

  For each \(\varrho\in\Irr(G)\), determinant factorisation on the disc
  \(|z|<\Gamma_{\varphi,R}^{-1}\) gives
  \[
    \mathcal L_{\chi_\varrho,\varphi}(z)
    =
    \delta_{\varrho,\mathbf 1}\log(1-e^Pz)^{-1}
    +
    \sum_{\beta\in\mathcal E_\varrho(R)}
      m_\varrho(\beta)\log(1-\beta z)^{-1}
    +H_{\varrho,R}(z),
  \]
  where \(H_{\varrho,R}\) is holomorphic on
  \(|z|<\Gamma_{\varphi,R}^{-1}\); for the trivial channel,
  \(e^P\notin\mathcal E_{\mathbf 1}(R)\) by definition. Writing
  \[
    H_{\varrho,R}(z)=\sum_{n\ge 1}\frac{h_{\varrho,R}(n)}{n}z^n,
  \]
  Cauchy's estimates yield \(h_{\varrho,R}(n)=O(\Gamma_{\varphi,R}^n)\),
  uniformly in \(\varrho\), because \(\Irr(G)\) is finite. Comparing
  coefficients therefore gives
  \begin{equation}\label{eq:holder-transfer-spectral-window-traces}
    T_{\chi_\varrho,\varphi,n}
    =
    \delta_{\varrho,\mathbf 1}e^{Pn}
    +
    \sum_{\beta\in\mathcal E_\varrho(R)}
      m_\varrho(\beta)\beta^n
    +O(\Gamma_{\varphi,R}^n).
  \end{equation}
  If \(\mathscr P_{\varrho,\beta}\) is the Riesz projector for
  \(\beta\), then the restriction of \(\mathscr L_{\varphi,\varrho}\)
  to the finite-dimensional space
  \(\mathscr P_{\varrho,\beta}C^\alpha(X_A^+,V_\varrho)\) has the form
  \(\beta I+N\), with \(N\) nilpotent. Consequently,
  \[
    \Tr\bigl(\mathscr L_{\varphi,\varrho}^n\mathscr P_{\varrho,\beta}\bigr)
    =
    \Tr\bigl((\beta I+N)^n\bigr)
    =m_\varrho(\beta)\beta^n.
  \]

  Coefficient extraction in
  \eqref{eq:holder-transfer-adams-mobius}, applied to
  \(\phi=\chi_\varrho\) and the Adams decomposition
  \eqref{eq:adams-character-decomposition}, yields
  \[
    \pi_{n,\varrho}(\varphi)
    =
    \frac{1}{n}\sum_{m\mid n}\mu(m)
      \sum_{\sigma\in\Irr(G)}
        a_{\varrho,\sigma}^{(m)}T_{\chi_\sigma,\varphi,n/m}.
  \]
  For the trivial representation, the divisor \(m=1\) contributes
  \[
    \frac{e^{Pn}}{n}
    +\frac{1}{n}
      \sum_{\beta\in\mathcal E_{\mathbf 1}(R)}
        m_{\mathbf 1}(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right)
  \]
  by \eqref{eq:holder-transfer-spectral-window-traces}. For \(m>1\),
  \Cref{lem:adams-coefficients-bounded} and the uniform estimate
  \(T_{\chi_\sigma,\varphi,\ell}=O(e^{P\ell})\) from the proof of
  \Cref{thm:holder-transfer-perron-point} give
  \[
    \frac{1}{n}
    \sum_{\substack{m\mid n\\ m>1}}
      \left|
        \sum_{\sigma\in\Irr(G)}
          a_{\mathbf 1,\sigma}^{(m)}
          T_{\chi_\sigma,\varphi,n/m}
      \right|
    =
    O\!\left(\frac{e^{Pn/2}}{n}\right)
    =
    O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right),
  \]
  because \(\Gamma_{\varphi,R}>e^{P/2}\). Hence
  \begin{equation}\label{eq:holder-transfer-spectral-window-trivial}
    p_n(\varphi)
    =
    \frac{e^{Pn}}{n}
    +\frac{1}{n}
      \sum_{\beta\in\mathcal E_{\mathbf 1}(R)}
        m_{\mathbf 1}(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right).
  \end{equation}

  For a fixed non-trivial \(\varrho\), the divisor \(m=1\) contributes
  \[
    \frac{1}{n}
      \sum_{\beta\in\mathcal E_\varrho(R)}
        m_\varrho(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right),
  \]
  again by \eqref{eq:holder-transfer-spectral-window-traces}, while the
  divisors \(m>1\) satisfy the same
  \(O(\Gamma_{\varphi,R}^n/n)\) bound as above. Therefore
  \begin{equation}\label{eq:holder-transfer-spectral-window-nontrivial}
    \pi_{n,\varrho}(\varphi)
    =
    \frac{1}{n}
      \sum_{\beta\in\mathcal E_\varrho(R)}
        m_\varrho(\beta)\beta^n
    +O\!\left(\frac{\Gamma_{\varphi,R}^n}{n}\right)
    \qquad (\varrho\neq\mathbf 1).
  \end{equation}

  The weighted class decomposition
  \[
    p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}
      \sum_{\varrho\in\Irr(G)}
        \chi_\varrho^\ast(C)\pi_{n,\varrho}(\varphi)
  \]
  now yields \eqref{eq:holder-transfer-spectral-window-pointwise} after
  substituting
  \eqref{eq:holder-transfer-spectral-window-trivial} and
  \eqref{eq:holder-transfer-spectral-window-nontrivial}. The error term
  is uniform in \(C\) because the set of conjugacy classes is finite and
  \(|\chi_\varrho(C)|\le \dim\varrho\).

  Dividing \eqref{eq:holder-transfer-spectral-window-pointwise} by
  \(e^{Pn}\) gives
  \[
    e^{-Pn}p_{n,C}(\varphi)
    =
    \frac{|C|}{|G|}\frac{1}{n}
    +\frac{\mathfrak s_{C,\varphi,R}(n)}{n\,e^{Pn}}
    +O\!\left(\frac{(\Gamma_{\varphi,R}e^{-P})^n}{n}\right).
  \]
  Since every \(\beta\in\mathcal E_\varrho(R)\) satisfies
  \(|\beta|<e^P\), the series
  \eqref{eq:holder-transfer-spectral-window-tail} converges absolutely.
  Summing the preceding estimate over \(n>N\) and applying
  \Cref{lem:geometric-harmonic-tail} gives
  \[
    \sum_{n>N}
      \left(
        e^{-Pn}p_{n,C}(\varphi)
        -\frac{|C|}{|G|}\frac{1}{n}
      \right)
    =
    \mathfrak T_{C,\varphi,R}(N)
    +O\!\left(\frac{(\Gamma_{\varphi,R}e^{-P})^N}{N}\right).
  \]
  By \Cref{thm:holder-transfer-perron-point},
  \[
    \mathcal M_C(\varphi)-\frac{|C|}{|G|}\gamma
    =
    \sum_{n\ge 1}
      \left(
        e^{-Pn}p_{n,C}(\varphi)
        -\frac{|C|}{|G|}\frac{1}{n}
      \right).
  \]
  Subtracting the tail beyond \(N\) proves
  \eqref{eq:holder-transfer-spectral-window-harmonic}.
\end{proof}

\begin{theorem}[Pressure-boundary rigidity for twisted Ruelle operators]
  \label{thm:holder-transfer-boundary-rigidity}
  Let \(h_\varphi\in C^\alpha(X_A^+)\) be the positive eigenfunction of
  \(\mathscr L_{\varphi,\mathbf 1}\) satisfying
  \[
    \mathscr L_{\varphi,\mathbf 1}h_\varphi=e^P h_\varphi.
  \]
  For an irreducible unitary representation \(\varrho\in\Irr(G)\), the
  following are equivalent.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item \(\rho(\mathscr L_{\varphi,\varrho})=e^P\).
    \item There exist \(\omega\in\TT\) and a H\"older map
      \(\xi:X_A^+\to V_\varrho\) with \(\|\xi(x)\|=1\) for every
      \(x\in X_A^+\) such that
      \begin{equation}\label{eq:holder-transfer-line-field}
        \varrho(\tau(y))\xi(y)=\omega\,\xi(x)
        \qquad (\sigma y=x).
      \end{equation}
  \end{enumerate}
  Whenever these conditions hold, every periodic point
  \(x\in\operatorname{Fix}(\sigma^n)\) satisfies
  \begin{equation}\label{eq:holder-transfer-periodic-line}
    \varrho(\tau_n(x))\xi(x)=\omega^n\xi(x).
  \end{equation}
\end{theorem}

\begin{proof}
  The weighted norm
  \[
    \|F\|_{h_\varphi}
    :=
    \sup_{x\in X_A^+}\frac{\|F(x)\|}{h_\varphi(x)}
  \]
  satisfies
  \[
    \frac{\|(\mathscr L_{\varphi,\varrho}F)(x)\|}{h_\varphi(x)}
    \le
    \sum_{\sigma y=x}
      e^{\varphi(y)}\frac{h_\varphi(y)}{h_\varphi(x)}
      \frac{\|F(y)\|}{h_\varphi(y)}
    \le
    e^P\|F\|_{h_\varphi},
  \]
  because \(\mathscr L_{\varphi,\mathbf 1}h_\varphi=e^P h_\varphi\).
  Consequently
  \[
    \rho(\mathscr L_{\varphi,\varrho})\le e^P
    \qquad (\varrho\in\Irr(G)).
  \]

  Assume first that \(\rho(\mathscr L_{\varphi,\varrho})=e^P\). The
  operator \(\mathscr L_{\varphi,\varrho}\) is quasi-compact on
  \(C^\alpha(X_A^+,V_\varrho)\), so some eigenvalue
  \(\alpha\in\spec(\mathscr L_{\varphi,\varrho})\) satisfies
  \(|\alpha|=e^P\). Let \(H\neq 0\) be a corresponding H\"older
  eigenfunction:
  \[
    \mathscr L_{\varphi,\varrho}H=\alpha H.
  \]
  Set
  \[
    q(x):=\frac{\|H(x)\|}{h_\varphi(x)},
    \qquad
    M:=\max_{x\in X_A^+}q(x)>0.
  \]
  For every \(x\in X_A^+\),
  \[
    e^P q(x)
    =
    \frac{|\alpha|\,\|H(x)\|}{h_\varphi(x)}
    \le
    \sum_{\sigma y=x}
      e^{\varphi(y)}\frac{\|H(y)\|}{h_\varphi(x)}
    \le
    M\sum_{\sigma y=x}
      e^{\varphi(y)}\frac{h_\varphi(y)}{h_\varphi(x)}
    =e^P M.
  \]
  Hence \(q(x)\le M\) for all \(x\). Let
  \(E:=\{x\in X_A^+:q(x)=M\}\). If \(x\in E\), then the last chain of
  inequalities is an equality, so every predecessor \(y\) of \(x\)
  satisfies \(q(y)=M\). Thus \(E\) is non-empty, closed, and backward
  invariant.

  Fix \(x\in E\). Let \([a_0,\dots,a_{k-1}]\) be any non-empty cylinder.
  Primitivity of \(A\) supplies, for all sufficiently large \(n\), an
  admissible word of length \(n-k\) connecting \(a_{k-1}\) to the first
  symbol of \(x\). Appending the tail of \(x\) therefore produces
  \(y\in[a_0,\dots,a_{k-1}]\) with \(\sigma^n y=x\). Backward invariance
  gives \(y\in E\). Every cylinder thus meets \(E\), so \(E=X_A^+\).
  Therefore
  \[
    \|H(x)\|=M h_\varphi(x)
    \qquad (x\in X_A^+),
  \]
  and in particular \(H(x)\neq 0\) everywhere.

  Write \(\omega:=\alpha e^{-P}\in\TT\) and
  \[
    \xi(x):=\frac{H(x)}{M h_\varphi(x)}.
  \]
  Then \(\xi\) is H\"older and \(\|\xi(x)\|=1\). Since equality holds in
  the triangle inequality at every \(x\), all vectors
  \(\varrho(\tau(y))H(y)\) with \(\sigma y=x\) lie on the ray spanned by
  \(\alpha H(x)\). Their norms are
  \[
    \|\varrho(\tau(y))H(y)\|
    =\|H(y)\|
    =M h_\varphi(y),
  \]
  and the sum of these norms equals
  \[
    \sum_{\sigma y=x}e^{\varphi(y)}M h_\varphi(y)
    =e^P M h_\varphi(x).
  \]
  Consequently each summand is exactly
  \(e^{\varphi(y)}M h_\varphi(y)\omega\,\xi(x)\), which gives
  \eqref{eq:holder-transfer-line-field}.

  Conversely, assume that \eqref{eq:holder-transfer-line-field} holds.
  Define \(H:=h_\varphi\xi\). Then
  \begin{align*}
    (\mathscr L_{\varphi,\varrho}H)(x)
    &=
    \sum_{\sigma y=x}
      e^{\varphi(y)}h_\varphi(y)\varrho(\tau(y))\xi(y) \\
    &=
    \omega\,\xi(x)\sum_{\sigma y=x}e^{\varphi(y)}h_\varphi(y) \\
    &=
    \omega e^P h_\varphi(x)\xi(x)
    =\omega e^P H(x).
  \end{align*}
  Therefore \(\omega e^P\in\spec(\mathscr L_{\varphi,\varrho})\), so
  \(\rho(\mathscr L_{\varphi,\varrho})\ge e^P\). Together with the
  general upper bound, this proves
  \(\rho(\mathscr L_{\varphi,\varrho})=e^P\).

  Finally, if \(x\in\operatorname{Fix}(\sigma^n)\), repeated application
  of \eqref{eq:holder-transfer-line-field} along the orbit of \(x\)
  yields \eqref{eq:holder-transfer-periodic-line}.
\end{proof}
