\section{Deterministic Automata and Binary Primes}
\label{sec:automata}

\begin{definition}\label{def:dfa}
Let \(\Sigma=\{0,1\}\). A complete deterministic finite automaton is a
tuple \(\mathcal A=(Q,\Sigma,\delta,q_0,F)\), where \(Q\) is a finite
state set, \(q_0\in Q\) is the initial state, \(F\subseteq Q\) is the
set of accepting states, and
\[
  \delta:Q\times\Sigma\to Q
\]
is a total transition map.
\end{definition}

For \(m\ge 0\), write
\[
  a_m(\mathcal A)
  :=\#\{w\in\Sigma^m:\mathcal A\text{ accepts }w\}.
\]

\begin{theorem}[Density dichotomy for DFA languages]
  \label{thm:dfa-density-dichotomy}
  There exist an integer \(p\ge 1\), non-negative constants
  \(c_0,\dots,c_{p-1}\), and \(\theta\in(0,1)\) such that
  \[
    \frac{a_m(\mathcal A)}{2^m}=c_{m\bmod p}+O(\theta^m)
    \qquad (m\to\infty).
  \]
  Consequently exactly one of the following alternatives holds:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item There exists \(\eta\in(0,1)\) such that
    \(a_m(\mathcal A)=O((2\eta)^m)\).
    \item There exists a residue class \(r\bmod p\) and a constant
    \(c>0\) such that
    \[
      \frac{a_{r+kp}(\mathcal A)}{2^{r+kp}}\ge c
      \qquad\text{for all sufficiently large }k.
    \]
  \end{enumerate}
  In particular, no DFA language can satisfy
  \[
    a_m(\mathcal A)/2^m\asymp m^{-\alpha}
  \]
  for any \(\alpha>0\).
\end{theorem}

\begin{proof}
  Let \(T_0,T_1\in\{0,1\}^{Q\times Q}\) be the transition matrices for
  the input symbols \(0\) and \(1\), and let \(B:=T_0+T_1\). Since the
  automaton is complete and deterministic, every row sum of \(B\) is
  equal to \(2\). If \(u\) is the indicator vector of the initial state
  and \(v\) is the indicator of the accepting set, then
  \[
    a_m(\mathcal A)=u^{\mathsf T}B^m v.
  \]
  Set \(P:=B/2\). Then \(P\) is a stochastic matrix on the finite set
  \(Q\). Decompose \(Q\) into transient states and recurrent classes.
  On each recurrent class, finite-state Markov-chain theory gives an
  eventual cyclic decomposition with exponential convergence to the
  stationary cyclic projections
  \cite[Chapter~4]{LevinPeresWilmer2009MarkovMixing}. Taking the least
  common multiple of the class periods yields the required periodic
  term. More concretely, if a recurrent class has period \(d\), then
  the eigenvalues of the restriction of \(P\) with modulus \(1\) are
  precisely the \(d\)th roots of unity coming from that cyclic
  decomposition, while every other eigenvalue has modulus \(<1\). Their
  contribution is therefore \(O(\theta_C^m)\) for some \(\theta_C<1\),
  and the transient part is absorbed into the same type of estimate.
  Choosing \(p\) as the least common multiple of the class periods and
  \(\theta\) as the maximum of the resulting decay rates gives
  \[
    u^{\mathsf T}P^m v=c_{m\bmod p}+O(\theta^m)
  \]
  for some \(p\) and some \(\theta\in(0,1)\). Multiplying by \(2^m\)
  proves the expansion.

  If every \(c_r\) vanishes, then \(a_m(\mathcal A)/2^m=O(\theta^m)\),
  which is alternative \textup{(i)}. Otherwise some \(c_r\) is strictly
  positive, and along the corresponding residue class the density has a
  positive lower bound, which is alternative \textup{(ii)}.
\end{proof}

For each \(m\ge 1\), let \(\mathcal P_m^{(2)}\) be the set of binary
words of length \(m\) whose first digit is \(1\) and whose binary value
is prime. Let \(p_m^{(2)}:=\abs{\mathcal P_m^{(2)}}\). By the prime
number theorem \cite{HardyWright2008},
\[
  p_m^{(2)}
  =\pi(2^m)-\pi(2^{m-1})
  \asymp \frac{2^m}{m}.
\]
The condition that the first digit be \(1\) excludes leading zeros. One
may equivalently compare a DFA language first with
\(\{1\}\{0,1\}^{m-1}\); in particular, the scale
\(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\) is unchanged because every
prime in \([2^{m-1},2^m)\) already has binary length \(m\).

\begin{corollary}[Symmetric-difference lower bound for binary primes]
  \label{cor:dfa-prime-symmetric-diff}
  Let \(L(\mathcal A)\subseteq\{0,1\}^\ast\) be recognised by a fixed
  DFA \(\mathcal A\), and let
  \[
    L_m:=L(\mathcal A)\cap\{0,1\}^m.
  \]
  Then there exists \(c>0\) such that
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}
    \ge c\,\frac{2^m}{m}
  \]
  for all sufficiently large \(m\). If \(L_m\) falls in the
  positive-density alternative of \Cref{thm:dfa-density-dichotomy},
  then along an infinite arithmetic progression of lengths one has the
  stronger bound
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}\gg 2^m.
  \]
\end{corollary}

\begin{proof}
  If \(L_m\) is exponentially sparse, then
  \(\abs{L_m}=o(2^m/m)\), while
  \(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\). Hence
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}
    \ge \abs{\mathcal P_m^{(2)}}-\abs{L_m}
    \gg \frac{2^m}{m}.
  \]
  If \(L_m\) has positive density along an arithmetic progression of
  lengths, then \(\abs{L_m}\gg 2^m\) there, whereas
  \(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\). Thus
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}
    \ge \abs{L_m}-\abs{\mathcal P_m^{(2)}}
    \gg 2^m
  \]
  on that progression.
\end{proof}

\begin{corollary}[No simultaneous recall and precision for fixed DFA]
  \label{cor:dfa-prime-recall-precision}
  Let
  \[
    \mathrm{Rec}_m
    :=\frac{\abs{L_m\cap\mathcal P_m^{(2)}}}{\abs{\mathcal P_m^{(2)}}},
    \qquad
    \mathrm{Prec}_m
    :=\frac{\abs{L_m\cap\mathcal P_m^{(2)}}}{\abs{L_m}}
  \]
  whenever \(\abs{L_m}>0\). Then every DFA satisfies one of the
  following:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item \(\mathrm{Rec}_m\to 0\) exponentially fast.
    \item \(\mathrm{Prec}_m\to 0\) along an infinite arithmetic
    progression of lengths.
  \end{enumerate}
  In particular, no fixed DFA has both recall and precision bounded
  away from zero for infinitely many lengths.
\end{corollary}

\begin{proof}
  In the sparse case,
  \[
    \mathrm{Rec}_m
    \le \frac{\abs{L_m}}{\abs{\mathcal P_m^{(2)}}}
    \ll \eta^m m
  \]
  for some \(\eta\in(0,1)\), hence \(\mathrm{Rec}_m\to 0\)
  exponentially. In the positive-density case,
  \[
    \mathrm{Prec}_m
    \le \frac{\abs{\mathcal P_m^{(2)}}}{\abs{L_m}}
    \ll \frac{1}{m}
  \]
  along the relevant arithmetic progression.
\end{proof}
