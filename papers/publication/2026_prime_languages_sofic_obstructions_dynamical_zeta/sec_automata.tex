\section{Deterministic Automata and Binary Primes}
\label{sec:automata}

\begin{definition}\label{def:dfa}
Let \(\Sigma=\{0,1\}\). A complete deterministic finite automaton is a
tuple \(\mathcal A=(Q,\Sigma,\delta,q_0,F)\), where \(Q\) is a finite
state set, \(q_0\in Q\) is the initial state, \(F\subseteq Q\) is the
set of accepting states, and
\[
  \delta:Q\times\Sigma\to Q
\]
is a total transition map.
\end{definition}

For \(m\ge 0\), write
\[
  a_m(\mathcal A)
  :=\#\{w\in\Sigma^m:\mathcal A\text{ accepts }w\}.
\]

\begin{theorem}[Density dichotomy for DFA languages]
  \label{thm:dfa-density-dichotomy}
  There exist an integer \(p\ge 1\), non-negative constants
  \(c_0,\dots,c_{p-1}\), and \(\theta\in(0,1)\) such that
  \[
    \frac{a_m(\mathcal A)}{2^m}=c_{m\bmod p}+O(\theta^m)
    \qquad (m\to\infty).
  \]
  Consequently exactly one of the following alternatives holds:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item There exists \(\eta\in(0,1)\) such that
    \(a_m(\mathcal A)=O((2\eta)^m)\).
    \item There exists a residue class \(r\bmod p\) and a constant
    \(c>0\) such that
    \[
      \frac{a_{r+kp}(\mathcal A)}{2^{r+kp}}\ge c
      \qquad\text{for all sufficiently large }k.
    \]
  \end{enumerate}
  In particular, no DFA language can satisfy
  \[
    a_m(\mathcal A)/2^m\asymp m^{-\alpha}
  \]
  for any \(\alpha>0\).
\end{theorem}

\begin{proof}
  Let \(T_0,T_1\in\{0,1\}^{Q\times Q}\) be the transition matrices for
  the input symbols \(0\) and \(1\), and let \(B:=T_0+T_1\). Since the
  automaton is complete and deterministic, every row sum of \(B\) is
  equal to \(2\). If \(u\) is the indicator vector of the initial state
  and \(v\) is the indicator of the accepting set, then
  \[
    a_m(\mathcal A)=u^{\mathsf T}B^m v.
  \]
  Set \(P:=B/2\). Then \(P\) is a stochastic matrix on the finite set
  \(Q\). After permuting states, write \(P\) in block upper-triangular
  form with one transient block and recurrent-class blocks
  \(P_1,\dots,P_s\), where each \(P_i\) is an irreducible stochastic
  matrix. The transient block has spectral radius \(<1\), so its
  contribution is \(O(\theta_T^m)\) for some \(\theta_T\in(0,1)\).

  Fix \(i\), and let \(d_i\) be the period of \(P_i\). By the periodic
  Perron--Frobenius theorem for irreducible non-negative matrices, there
  exist spectral projectors \(E_{i,0},\dots,E_{i,d_i-1}\), a constant
  \(C_i>0\), and \(\theta_i\in(0,1)\) such that
  \[
    P_i^m
    =\sum_{\ell=0}^{d_i-1}\omega_{d_i}^{\ell m}E_{i,\ell}+R_{i,m},
    \qquad
    \|R_{i,m}\|\le C_i\theta_i^m,
  \]
  where \(\omega_{d_i}:=e^{2\pi i/d_i}\). Since the transient block has
  no eigenvalues on the unit circle, these recurrent-class expansions
  account for every peripheral eigenvalue of \(P\). Hence there exist
  matrices \(F_{i,\ell}\) on the full state space, a constant \(C>0\),
  and \(\theta\in(0,1)\) such that
  \[
    P^m
    =\sum_{i=1}^s\sum_{\ell=0}^{d_i-1}
      \omega_{d_i}^{\ell m}F_{i,\ell}+E_m,
    \qquad
    \|E_m\|\le C\theta^m.
  \]

  Let \(p:=\operatorname{lcm}(d_1,\dots,d_s)\). Grouping the peripheral
  terms by the residue class of \(m\bmod p\), we obtain matrices
  \(S_0,\dots,S_{p-1}\) such that
  \[
    P^m=S_{m\bmod p}+E_m,
    \qquad
    \|E_m\|\le C\theta^m.
  \]
  Therefore
  \[
    u^{\mathsf T}P^m v=u^{\mathsf T}S_{m\bmod p}v+O(\theta^m).
  \]
  For \(0\le r<p\), define
  \[
    c_r:=\lim_{k\to\infty}u^{\mathsf T}P^{kp+r}v=u^{\mathsf T}S_r v.
  \]
  The limit exists by the preceding expansion, and \(c_r\ge 0\) because
  every term \(u^{\mathsf T}P^{kp+r}v\) is non-negative. Thus
  \[
    \frac{a_m(\mathcal A)}{2^m}
    =u^{\mathsf T}P^m v
    =c_{m\bmod p}+O(\theta^m),
  \]
  as claimed.

  If every \(c_r\) vanishes, then
  \(a_m(\mathcal A)/2^m=O(\theta^m)\), which is alternative
  \textup{(i)}. Otherwise some \(c_r\) is strictly positive, and along
  the corresponding residue class the density has a positive lower
  bound, which is alternative \textup{(ii)}.
\end{proof}

For each \(m\ge 1\), let \(\mathcal P_m^{(2)}\) be the set of binary
words of length \(m\) whose first digit is \(1\) and whose binary value
is prime. Let \(p_m^{(2)}:=\abs{\mathcal P_m^{(2)}}\). By the prime
number theorem \cite{HardyWright2008},
\[
  p_m^{(2)}
  =\pi(2^m)-\pi(2^{m-1})
  \asymp \frac{2^m}{m}.
\]
The condition that the first digit be \(1\) excludes leading zeros. One
may equivalently compare a DFA language first with
\(\{1\}\{0,1\}^{m-1}\); in particular, the scale
\(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\) is unchanged because every
prime in \([2^{m-1},2^m)\) already has binary length \(m\).

We first establish the quantitative obstruction that drives both
corollaries below.

\begin{proposition}[Quantitative recall/precision obstruction]
  \label{prop:dfa-quantitative-obstruction}
  Let \(L_m:=L(\mathcal A)\cap\{0,1\}^m\), and define
  \[
    \mathrm{Rec}_m
    :=\frac{\abs{L_m\cap\mathcal P_m^{(2)}}}{\abs{\mathcal P_m^{(2)}}},
    \qquad
    \mathrm{Prec}_m
    :=\frac{\abs{L_m\cap\mathcal P_m^{(2)}}}{\abs{L_m}}
  \]
  whenever \(\abs{L_m}>0\).
  Suppose that there exist
  \(\delta>0\) and infinitely many integers \(m\) such that
  \(\mathrm{Rec}_m\ge \delta\) and \(\mathrm{Prec}_m\ge \delta\).
  Then on those lengths,
  \[
    \delta\,\abs{\mathcal P_m^{(2)}}
    \le \abs{L_m\cap\mathcal P_m^{(2)}}
    \le \abs{L_m}
    \le \delta^{-1}\abs{L_m\cap\mathcal P_m^{(2)}}
    \le \delta^{-1}\abs{\mathcal P_m^{(2)}}.
  \]
  In particular, \(\abs{L_m}\asymp 2^m/m\) along infinitely many
  lengths, contradicting \Cref{thm:dfa-density-dichotomy}.
\end{proposition}

\begin{proof}
  The lower bound follows from
  \(\abs{L_m\cap\mathcal P_m^{(2)}}=\mathrm{Rec}_m\,\abs{\mathcal P_m^{(2)}}\ge\delta\,\abs{\mathcal P_m^{(2)}}\),
  and the upper bound follows from
  \(\abs{L_m}=\abs{L_m\cap\mathcal P_m^{(2)}}/\mathrm{Prec}_m\le\delta^{-1}\abs{L_m\cap\mathcal P_m^{(2)}}\le\delta^{-1}\abs{\mathcal P_m^{(2)}}\).
  Since \(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\), the displayed chain
  gives \(\abs{L_m}\asymp 2^m/m\) on an infinite set of lengths. This
  is impossible by \Cref{thm:dfa-density-dichotomy}.
\end{proof}

\begin{corollary}[Symmetric-difference lower bound for binary primes]
  \label{cor:dfa-prime-symmetric-diff}
  Let \(L(\mathcal A)\subseteq\{0,1\}^\ast\) be recognised by a fixed
  DFA \(\mathcal A\), and let \(L_m:=L(\mathcal A)\cap\{0,1\}^m\).
  Then there exists \(c>0\) such that
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}
    \ge c\,\frac{2^m}{m}
  \]
  for all sufficiently large \(m\). If \(L_m\) falls in the
  positive-density alternative of \Cref{thm:dfa-density-dichotomy},
  then along an infinite arithmetic progression of lengths one has the
  stronger bound
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}\gg 2^m.
  \]
\end{corollary}

\begin{proof}
  If \(L_m\) is exponentially sparse, then
  \(\abs{L_m}=o(2^m/m)\), while
  \(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\). Hence
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}
    \ge \abs{\mathcal P_m^{(2)}}-\abs{L_m}
    \gg \frac{2^m}{m}.
  \]
  If \(L_m\) has positive density along an arithmetic progression of
  lengths, then \(\abs{L_m}\gg 2^m\) there, whereas
  \(\abs{\mathcal P_m^{(2)}}\asymp 2^m/m\). Thus
  \[
    \abs{L_m\triangle \mathcal P_m^{(2)}}
    \ge \abs{L_m}-\abs{\mathcal P_m^{(2)}}
    \gg 2^m
  \]
  on that progression.
\end{proof}

\begin{corollary}[No simultaneous recall and precision for fixed DFA]
  \label{cor:dfa-prime-recall-precision}
  With \(\mathrm{Rec}_m\) and \(\mathrm{Prec}_m\) as in
  \Cref{prop:dfa-quantitative-obstruction}, every DFA satisfies one of
  the following:
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item \(\mathrm{Rec}_m\to 0\) exponentially fast.
    \item \(\mathrm{Prec}_m\to 0\) along an infinite arithmetic
    progression of lengths.
  \end{enumerate}
  In particular, no fixed DFA has both recall and precision bounded
  away from zero for infinitely many lengths.
\end{corollary}

\begin{proof}
  In the sparse case of \Cref{thm:dfa-density-dichotomy},
  \[
    \mathrm{Rec}_m
    \le \frac{\abs{L_m}}{\abs{\mathcal P_m^{(2)}}}
    \ll \eta^m m
  \]
  for some \(\eta\in(0,1)\), hence \(\mathrm{Rec}_m\to 0\)
  exponentially. In the positive-density case,
  \[
    \mathrm{Prec}_m
    \le \frac{\abs{\mathcal P_m^{(2)}}}{\abs{L_m}}
    \ll \frac{1}{m}
  \]
  along the relevant arithmetic progression. If both recall and
  precision were bounded below by some \(\delta>0\) for infinitely many
  lengths, then \Cref{prop:dfa-quantitative-obstruction} would
  contradict \Cref{thm:dfa-density-dichotomy}.
\end{proof}
