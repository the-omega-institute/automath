\section{Zeckendorf Syntax and Sofic Obstructions}
\label{sec:zeckendorf}

Let
\[
  \mathcal Z
  :=\{x_1\cdots x_m\in\{0,1\}^\ast:
      x_ix_{i+1}\neq 11,\ x_m=1\}
\]
be the language of canonical Zeckendorf words. For
\(w=x_1\cdots x_m\in\mathcal Z\), define
\[
  \val(w):=\sum_{k=1}^m x_kF_{k+1},
\]
where \(F_0=0\), \(F_1=1\), \(F_{n+2}=F_{n+1}+F_n\), and
\(\varphi=(1+\sqrt5)/2\). The map \(w\mapsto\val(w)\) is a bijection
from \(\mathcal Z\) onto \(\NN_{\ge 1}\) \cite{Zeckendorf1972}.
Consequently \(\mathcal Z\cap\{0,1\}^{\le m}\) parametrises the
interval \([1,F_{m+2})\), so words of length exactly \(m\) parametrise
\([F_{m+1},F_{m+2})\).

\begin{theorem}[Regular Zeckendorf languages have power-law growth]
  \label{thm:zeckendorf-regular-powerlaw}
  Let \(L\subseteq\mathcal Z\) be regular. Choose a DFA recognizing
  \(L\), delete states that are not reachable from the initial state or
  cannot reach an accepting state, and let \(\rho\) be the maximal
  spectral radius of a strongly connected component of the resulting
  automaton. Define
  \[
    N_L(T):=\#\{w\in L:\val(w)\le T\}.
  \]
  Then
  \[
    N_L(T)=T^{\alpha+o(1)},
    \qquad
    \alpha=
    \begin{cases}
      0, & \rho\le 1,\\[0.3em]
      \dfrac{\log \rho}{\log \varphi}, & \rho>1.
    \end{cases}
    \qquad (T\to\infty).
  \]
\end{theorem}

\begin{proof}
  If \(L=\emptyset\), then \(N_L(T)=0\) for all \(T\), and the theorem
  holds trivially with \(\alpha=0\). Assume henceforth that
  \(L\neq\emptyset\).

  Let
  \[
    a_m:=\#\{w\in L:\abs{w}=m\},
    \qquad
    A_n:=\sum_{m\le n}a_m.
  \]
  Since \(L\) is regular, there is a DFA for \(L\). After deleting
  states that are unreachable from the initial state or from which no
  accepting state is reachable, we obtain a trim automaton with
  transfer matrix \(B\) such that \(a_m=u^{\mathsf T}B^m v\). Because
  \(L\neq\emptyset\), the trim automaton has at least one path to an
  accepting state, so the definition of \(\rho\) is meaningful.

  Let \(\rho\) be the maximum spectral radius of a strongly connected
  component of this trim automaton.

  \textit{Upper bound.} Standard Jordan-form bounds for
  powers of a finite matrix give constants \(C>0\) and \(d\ge 0\) such
  that
  \[
    a_m\le C m^d\max(1,\rho)^m
    \qquad (m\ge 1).
  \]
  Summing over \(m\le n\) yields
  \[
    A_n\le C' n^{d+1}\max(1,\rho)^n
  \]
  for some \(C'>0\).

  \textit{Lower bound and the inequality \(\rho\le\varphi\).}
  If \(\rho>1\), choose a reachable, co-reachable strongly connected
  component \(C\) of spectral radius \(\rho\). Let \(d_C\) be the
  period of \(C\). By Perron--Frobenius theory for irreducible
  non-negative matrices, there exist states \(i,j\in C\), a residue
  class \(r\bmod d_C\), and a constant \(c>0\) such that
  \[
    (B_C^{r+k d_C})_{ij}\ge c\,\rho^{r+k d_C}
  \]
  for all sufficiently large \(k\). Fix paths of lengths \(\ell_1\) and
  \(\ell_2\) from the initial state to \(i\) and from \(j\) to an
  accepting state. Then
  \[
    a_{\ell_1+\ell_2+r+k d_C}\ge c\,\rho^{r+k d_C}
  \]
  for all large \(k\), so
  \[
    \limsup_{m\to\infty}a_m^{1/m}\ge\rho.
  \]
  On the other hand, \(a_m\le\abs{\mathcal Z\cap\{0,1\}^m}=F_{m+1}
  =\varphi^{m+o(m)}\), which gives
  \[
    \limsup_{m\to\infty}a_m^{1/m}\le\varphi.
  \]
  Hence \(\rho\le\varphi\).

  Since every sufficiently large integer \(n\) lies
  within \(O(1)\) of some value in the arithmetic progression
  \(\ell_1+\ell_2+r+kd_C\), the monotonicity of \(A_n\) gives
  \[
    A_n\ge c'\rho^n
  \]
  for all large \(n\), with \(c'>0\). Combined with the upper bound
  above, this implies
  \[
    A_n=\rho^{n+o(n)}.
  \]

  If \(\rho\le 1\), then the upper bound gives
  \(A_n\le C'n^{d+1}\), so \(A_n=\exp(o(n))\), and \(\rho\le\varphi\)
  holds trivially.

  \textit{Conversion to \(N_L(T)\).}
  If \(F_{n+1}\le T<F_{n+2}\), then every Zeckendorf word of value at
  most \(T\) has length at most \(n\), while every word of length at
  most \(n-1\) has value at most \(T\). Therefore
  \[
    A_{n-1}\le N_L(T)\le A_n.
  \]
  Since \(n=(\log T)/(\log\varphi)+O(1)\), the case \(\rho>1\) gives
  \[
    N_L(T)
    =\exp\!\left(
      \frac{\log\rho}{\log\varphi}\log T+o(\log T)
    \right)
    =T^{\alpha+o(1)}
  \]
  with \(\alpha:=\log\rho/\log\varphi\in(0,1]\). If \(\rho\le 1\),
  then \(N_L(T)\le A_n=\exp(o(n))=\exp(o(\log T))=T^{o(1)}\), so the
  same conclusion holds with \(\alpha=0\).
\end{proof}

\begin{remark}\label{rem:zeckendorf-examples}
  The endpoint cases and the borderline case \(\rho=1\) already occur
  for simple regular sublanguages.
  \begin{enumerate}[label=\textup{(\roman*)}]
    \item For \(L=\mathcal Z\), one has \(N_L(T)=\lfloor T\rfloor\),
    \(\rho=\varphi\), and \(\alpha=1\).
    \item If \(L\) is finite, then \(N_L(T)\) is eventually constant,
    so \(\alpha=0\).
    \item For \(L=0^\ast1\subseteq\mathcal Z\), one has \(a_m=1\) for
    every \(m\), hence \(\rho=1\) and
    \(N_L(T)\asymp \log T=T^{o(1)}\).
  \end{enumerate}
\end{remark}

\begin{remark}
  The same transfer-matrix argument applies more generally to Parry
  numeration systems whose admissible digit language is regular. In that
  setting the regular syntax of admissible expansions replaces
  \(\mathcal Z\), and the dominant exponential growth constant of the
  numeration system replaces \(\varphi\); see
  \cite{Frougny1992,BruyereHansel1997}.
\end{remark}

\begin{corollary}[Prime Zeckendorf language is not regular]
  \label{cor:zeckendorf-prime-language-not-regular}
  Let
  \[
    \mathcal P_{\mathcal Z}
    :=\{\mathrm{Zeck}(p)\in\mathcal Z:p\text{ prime}\},
  \]
  where \(\mathrm{Zeck}(n)\) denotes the canonical Zeckendorf
  representation of \(n\). Then \(\mathcal P_{\mathcal Z}\) is not a
  regular language.
\end{corollary}

\begin{proof}
  If \(\mathcal P_{\mathcal Z}\) were regular, then
  \Cref{thm:zeckendorf-regular-powerlaw} would give
  \[
    \pi(T)=N_{\mathcal P_{\mathcal Z}}(T)=T^{\alpha+o(1)}
  \]
  for some \(\alpha\in[0,1]\). This is impossible because
  \(\pi(T)\sim T/\log T\): for every \(\alpha<1\) one has
  \(\pi(T)/T^\alpha\to\infty\), while \(\pi(T)/T\to 0\).
\end{proof}

\begin{proposition}[Sofic word counts are exponential polynomials]
  \label{prop:sofic-counts-exponential-polynomial}
  Let \(X\) be a sofic shift, and let \(a_m(X)\) denote the number of
  admissible words of length \(m\). Then
  \[
    a_m(X)=\sum_{j=1}^J Q_j(m)\lambda_j^m
  \]
  for some complex numbers \(\lambda_j\) and some polynomials \(Q_j\).
  Equivalently, the sequence \(a_m(X)\) satisfies a linear recurrence.
\end{proposition}

\begin{proof}
  By definition, a sofic shift \(X\) is the image of a shift of finite
  type under a one-block factor map. A classical theorem (see
  \cite[Theorem~3.2.1]{LindMarcus1995}) then implies that the block
  language \(\mathcal B(X)=\{w\in\Sigma^\ast:w\text{ appears in }X\}\)
  is regular. Let \(\mathcal A=(Q,\Sigma,\delta,q_0,F)\) be a complete
  DFA recognizing \(\mathcal B(X)\), with transition matrix
  \(B\in\ZZ_{\ge 0}^{Q\times Q}\) defined by \(B_{q,q'}:=1\) if
  \(\delta(q,a)=q'\) for some \(a\in\Sigma\) and \(0\) otherwise.
  Then \(a_m(X)=u^{\mathsf T}B^m v\), where \(u\) is the indicator of
  \(q_0\) and \(v\) is the indicator of \(F\). The Jordan normal form
  of \(B\) yields the displayed expansion into polynomial-exponential
  terms; compare \cite[Chapter~V]{FlajoletSedgewick2009}.
\end{proof}

\begin{lemma}[Residue-class asymptotics for non-negative exponential polynomials]
  \label{lem:residue-class-asymptotics}
  Let
  \[
    a_m=\sum_{j=1}^J Q_j(m)\lambda_j^m,
  \]
  where \(Q_j\in\CC[x]\), \(\lambda_j\in\CC\), and \(a_m\ge 0\) for all
  sufficiently large \(m\). Set
  \[
    \Lambda:=\max_{1\le j\le J}\abs{\lambda_j}.
  \]
  Assume that \(\lambda_j/\Lambda\) is a root of unity whenever
  \(\abs{\lambda_j}=\Lambda\). Then there exist an integer \(q\ge 1\)
  and a constant \(\eta\in(0,1)\) such that for every residue class
  \(r\bmod q\) one has
  \[
    a_{qk+r}
    =\Lambda^{qk+r}\bigl(P_r(k)+O(\eta^k)\bigr),
  \]
  where \(P_r\in\CC[k]\) is a polynomial. If \(P_r\not\equiv 0\), then
  \(P_r\) has positive leading coefficient, and hence
  \[
    a_{qk+r}
    =c_r k^{d_r}\Lambda^{qk+r}(1+o(1))
  \]
  for some \(c_r>0\) and \(d_r\in\ZZ_{\ge 0}\).
\end{lemma}

\begin{proof}
  If \(\Lambda=0\), then \(a_m=0\) for all sufficiently large \(m\), and
  the claim is trivial. Assume \(\Lambda>0\), and let
  \[
    D:=\{j: \abs{\lambda_j}=\Lambda\}.
  \]
  Choose \(q\) so that \((\lambda_j/\Lambda)^q=1\) for every \(j\in D\).
  Also choose \(\delta>0\) such that \(\abs{\lambda_j}\le \Lambda-\delta\)
  for every \(j\notin D\), and set
  \[
    \eta:=\left(\frac{\Lambda-\delta}{\Lambda}\right)^q\in(0,1).
  \]

  Fix \(r\in\{0,\dots,q-1\}\). For \(m=qk+r\),
  \[
    a_{qk+r}
    =\sum_{j\in D}Q_j(qk+r)\lambda_j^{qk+r}
     +O\bigl((\Lambda-\delta)^{qk+r}\bigr).
  \]
  Since \(\lambda_j=\Lambda\zeta_j\) with \(\zeta_j^q=1\) for \(j\in D\),
  this becomes
  \[
    a_{qk+r}
    =\Lambda^{qk+r}
      \left(
        \sum_{j\in D}\zeta_j^r Q_j(qk+r)
      \right)
      +O\bigl(\Lambda^{qk+r}\eta^k\bigr).
  \]
  The bracket is a polynomial in \(k\); call it \(P_r(k)\). This gives
  the first assertion.

  Suppose \(P_r\not\equiv 0\), and let \(d_r:=\deg P_r\). Then
  \[
    P_r(k)=\gamma_r k^{d_r}+O(k^{d_r-1})
  \]
  with \(\gamma_r\neq 0\). Because \(a_{qk+r}\) is real and
  non-negative for all sufficiently large \(k\), the polynomial
  \(P_r(k)\) must be real-valued on all sufficiently large integers.
  Hence \(\gamma_r\in\RR\), and \(\gamma_r\) cannot be negative, since
  otherwise the main term would force \(a_{qk+r}<0\) for all large
  \(k\). Therefore \(\gamma_r>0\), and
  \[
    a_{qk+r}
    =\gamma_r k^{d_r}\Lambda^{qk+r}(1+o(1)).
  \]
  Writing \(c_r:=\gamma_r\) proves the final claim.
\end{proof}

\begin{theorem}[Prime Zeckendorf slices are not sofic]
  \label{thm:zeckendorf-primes-not-sofic}
  Let
  \[
    \mathcal P_m^{(\mathrm Z)}
    :=\{w\in\mathcal Z:\abs{w}=m,\ \val(w)\text{ prime}\}.
  \]
  Then there is no sofic shift \(X\) over \(\{0,1\}\) such that
  \[
    \mathcal B_m(X)=\mathcal P_m^{(\mathrm Z)}
  \]
  for all sufficiently large \(m\), where \(\mathcal B_m(X)\) denotes
  the admissible words of length \(m\) in \(X\).
\end{theorem}

\begin{proof}
  Since words of length exactly \(m\) in \(\mathcal Z\) parametrise
  \([F_{m+1},F_{m+2})\), one has
  \[
    \abs{\mathcal P_m^{(\mathrm Z)}}
    =\pi(F_{m+2}-1)-\pi(F_{m+1}-1).
  \]
  By the prime number theorem \cite{HardyWright2008}, and because
  \(F_{m+2}/F_{m+1}\to\varphi>1\), this gives
  \[
    \abs{\mathcal P_m^{(\mathrm Z)}}\asymp\frac{\varphi^m}{m}.
  \]
  If a sofic shift \(X\) satisfied
  \(\mathcal B_m(X)=\mathcal P_m^{(\mathrm Z)}\) for all sufficiently
  large \(m\), then \Cref{prop:sofic-counts-exponential-polynomial}
  would imply that \(\abs{\mathcal P_m^{(\mathrm Z)}}\) is eventually an
  exponential polynomial \(\sum_jQ_j(m)\lambda_j^m\). Let
  \(\Lambda:=\max_j\abs{\lambda_j}\). For the non-negative matrix
  underlying the proof of
  \Cref{prop:sofic-counts-exponential-polynomial}, every eigenvalue of
  modulus \(\Lambda\) is of the form \(\Lambda\omega\) with \(\omega\) a
  root of unity. Thus
  \Cref{lem:residue-class-asymptotics} applies. Since
  \(\abs{\mathcal P_m^{(\mathrm Z)}}\asymp\varphi^m/m\) and is positive
  for all sufficiently large \(m\), every residue subsequence modulo
  the period \(q\) furnished by
  \Cref{lem:residue-class-asymptotics} must be non-zero and have the
  same exponential rate. Therefore \(\Lambda=\varphi\), and for every
  residue class \(r\bmod q\) one has
  \[
    \abs{\mathcal P_{qk+r}^{(\mathrm Z)}}
    =c_r k^{d_r}\varphi^{qk+r}(1+o(1))
  \]
  with \(c_r>0\) and \(d_r\ge 0\). On the other hand, the prime number
  theorem gives
  \[
    \abs{\mathcal P_{qk+r}^{(\mathrm Z)}}
    \asymp\frac{\varphi^{qk+r}}{k}
  \]
  along the same residue class. The factor \(k^{-1}\) is incompatible
  with \(k^{d_r}\) for any integer \(d_r\ge 0\). This contradiction
  proves the claim.
\end{proof}
