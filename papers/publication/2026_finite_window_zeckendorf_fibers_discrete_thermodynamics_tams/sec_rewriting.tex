\section{A confluent rewriting model}\label{sec:rewriting}

We consider the operator words generated by normalization and conditional
expectations.  This fragment is exact and confluent.  When additional
operators are introduced, exact equality is too rigid, and operator-norm error
provides a natural replacement.

\subsection{An exact rewriting system}

\begin{definition}[Exact rewrite system]
\label{def:rewrite-system}
Let \(\mathcal R\) be the rewriting system on the free monoid
\(\mathcal W_{Z,E}\) generated by the rules
\[
  ZZ\longrightarrow Z,
  \qquad
  E_mE_n\longrightarrow E_{\min\{m,n\}}
  \qquad (m,n\ge 1).
\]
We write \(W\to_{\mathcal R}W'\) for a one-step reduction and
\(\leftrightarrow_{\mathcal R}^*\) for the congruence generated by these
reductions.
\end{definition}

\begin{remark}\label{rem:rewrite-semantics}
The rule \(ZZ\to Z\) expresses idempotence of normalization.  The rule
\(E_mE_n\to E_{\min\{m,n\}}\) is the tower property of conditional expectation
along nested visible \(\sigma\)-algebras.
\end{remark}

\begin{lemma}[Termination]
\label{lem:rewrite-termination}
The system \(\mathcal R\) is terminating.
\leanverified{paper_projection_rewrite_termination}
\end{lemma}

\begin{proof}
Every reduction shortens the word length by one, so an infinite reduction
sequence is impossible.
\end{proof}

\begin{lemma}[Local confluence]
\label{lem:rewrite-local-confluence}
The system \(\mathcal R\) is locally confluent.
\leanverified{paper_projection_rewrite_local_confluence_seeds}
\end{lemma}

\begin{proof}
The only critical overlaps occur in words of length three.

For \(ZZZ\), reducing the first or the last pair leads to the same intermediate
word \(ZZ\), hence to \(Z\).

For \(E_aE_bE_c\), reducing the first pair gives
\[
  E_{\min\{a,b\}}E_c\to E_{\min\{a,b,c\}},
\]
while reducing the second pair gives
\[
  E_aE_{\min\{b,c\}}\to E_{\min\{a,b,c\}}.
\]
There are no mixed critical pairs because the left-hand sides of the two rules
involve disjoint alphabets.
\end{proof}

\begin{theorem}[Termination and confluence]
\label{thm:rewrite-confluence}
The exact rewrite system \(\mathcal R\) is terminating and confluent.
Consequently every word \(W\in\mathcal W_{Z,E}\) has a unique irreducible
normal form \(\mathrm{NF}(W)\).
\leanverified{paper_projection_rewrite_confluence_seeds}
\end{theorem}

\begin{proof}
Combine \Cref{lem:rewrite-termination,lem:rewrite-local-confluence} with
Newman's lemma.
\end{proof}

\begin{corollary}[Normal forms and decidability]
\label{cor:rewrite-word-problem}
A word is irreducible if and only if it contains no subword \(ZZ\) and no
adjacent pair \(E_mE_n\).  Equivalently, every normal form is an alternating
word
\[
  E_{m_0}ZE_{m_1}Z\cdots ZE_{m_r},
\]
with either endpoint \(E\)-letter allowed to be absent.  Moreover,
\[
  W_1\leftrightarrow_{\mathcal R}^* W_2
  \quad\Longleftrightarrow\quad
  \mathrm{NF}(W_1)=\mathrm{NF}(W_2),
\]
so the exact word problem is decidable in time linear in the total word length.
\leanverified{paper_projection_rewrite_word_problem_seeds}
\end{corollary}

\begin{proof}
Irreducibility is immediate from the form of the rules.  In a terminating and
confluent system two words are equivalent if and only if their normal forms
agree.  The normal form is obtained by collapsing each maximal \(Z\)-block to a
single \(Z\) and each maximal \(E\)-block to one letter indexed by the minimum
visible level appearing in that block.
\end{proof}

\begin{proposition}[Linear-time reduction algorithm]
\label{prop:rewrite-algorithm}
There is a deterministic linear-time procedure that maps every
\(W\in\mathcal W_{Z,E}\) to its normal form and simultaneously produces a
certificate consisting of exactly
\[
  \ell(W)-\ell(\mathrm{NF}(W))
\]
elementary contractions, where \(\ell(\cdot)\) denotes word length.
\end{proposition}

\begin{proof}
Scan the word from left to right.  Replace each maximal run of \(Z\)'s by one
copy of \(Z\), and replace each maximal run \(E_{m_1}\cdots E_{m_s}\) by the
single letter \(E_{\min\{m_1,\dots,m_s\}}\).  Every discarded letter records
one elementary contraction, and no backtracking is required.
\end{proof}

\subsection{Metric comparison beyond the exact system}

The exact confluence theorem applies only to the operator words above.  Once
additional operators are introduced, the relevant notion of comparison is small
semantic error rather than literal equality of words.

\begin{definition}[\texorpdfstring{$\varepsilon$}{epsilon}-sound rewrites]
\label{def:epsilon-sound}
Let \(W\) and \(W'\) be operator words with the same source and target state
simplices, and write \(\mathrm{Sem}(W)\) for the affine map represented by
\(W\).  We call a rewrite \(W\Rightarrow_{\varepsilon} W'\)
\emph{\(\varepsilon\)-sound} if
\[
  \|\mathrm{Sem}(W)-\mathrm{Sem}(W')\|_{1\to 1}\le \varepsilon.
\]
Exact rewrites are therefore \(0\)-sound.
\end{definition}

\begin{proposition}[Lawvere-type error calculus]
\label{prop:lawvere-distance}
If \(W\Rightarrow_{\varepsilon} W'\) and \(W'\Rightarrow_{\delta} W''\), then
\(W\Rightarrow_{\varepsilon+\delta} W''\).  Consequently the quantity
\[
  \begin{aligned}
  d_L(W,W')
  &:=\inf\Bigl\{\sum_{j=1}^N \varepsilon_j:
  W=W_0\Rightarrow_{\varepsilon_1}W_1\Rightarrow_{\varepsilon_2}\cdots \\
  &\hspace{5em}
  \Rightarrow_{\varepsilon_N}W_N=W'\Bigr\}
  \end{aligned}
\]
is a pseudo-metric on operator words with fixed source and target.
\end{proposition}

\begin{proof}
The triangle inequality for the operator norm gives
\[
  \|\mathrm{Sem}(W)-\mathrm{Sem}(W'')\|_{1\to1}
  \le
  \|\mathrm{Sem}(W)-\mathrm{Sem}(W')\|_{1\to1}
  +
  \|\mathrm{Sem}(W')-\mathrm{Sem}(W'')\|_{1\to1}
  \le \varepsilon+\delta.
\]
The pseudo-metric properties of \(d_L\) follow by taking infima over zigzags.
\end{proof}

\begin{remark}[Boundary of the exact theory]
\label{rem:rewrite-boundary}
The confluence theorem is specific to the exact system above.  More elaborate
operator languages should be compared in the pseudo-metric of
\Cref{prop:lawvere-distance}.  The finite-state kernels of
\Cref{sec:moment-kernel} are exact after compilation into matrix realizations,
but not necessarily at the level of naive word reduction.
\end{remark}
