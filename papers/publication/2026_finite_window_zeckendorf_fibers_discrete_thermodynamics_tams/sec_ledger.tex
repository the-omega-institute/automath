\section{Visible entropy rates and canonical bands}\label{sec:ledger}

Since \(p_m(x)=d_m(x)/2^m\), the visible law under uniform input is the
\(q=1\) canonical ensemble.  With the all-\(q\) asymptotics of \(S_q(m)\) in
hand, we obtain its integer-order R\'enyi entropy rates
\cite{CoverThomas2006} and sharp exponential tail estimates.

\begin{definition}[Visible law under uniform input]
\label{def:visible-law}
Let \(u_{\Omega_m}\) be the uniform law on \(\Omega_m\), and define
\[
  p_m:=P_mu_{\Omega_m}\in\Delta(X_m).
\]
Then
\[
  p_m(x)=\frac{d_m(x)}{2^m}.
\]
\end{definition}

\begin{corollary}[Typical visible thickness band]
\label{cor:visible-band}
For every \(\varepsilon>0\), define
\[
  V_m(\varepsilon)
  :=
  \left\{
    x\in X_m:
    \Delta_1-\varepsilon\le \frac1m\log d_m(x)\le \Delta_2+\varepsilon
  \right\}.
\]
Then
\[
  p_m\bigl(V_m(\varepsilon)\bigr)
  =
  1-\exp\bigl(-(\varepsilon+o(1))m\bigr).
\]
Equivalently,
\[
  V_m(\varepsilon)
  =
  \left\{
    x\in X_m:
    \begin{array}{c}
      \log 2-\log\varphi-\varepsilon
      \le
      \frac1m\log d_m(x) \\[2pt]
      \le
      \log\lambda_2-\log 2+\varepsilon
    \end{array}
  \right\}.
\]
\end{corollary}

\begin{proof}
Since \(S_1(m)=2^m\), one has \(\mu_{m,1}=p_m\).  Apply
\Cref{thm:gibbs-selection} with \(q=1\).  The displayed endpoint values follow
from \(\Delta_1=p_1-p_0=\log 2-\log\varphi\) and
\(\Delta_2=p_2-p_1=\log\lambda_2-\log 2\).
\end{proof}

\begin{theorem}[Integer-order R\'enyi entropy rates]
\label{thm:renyi-entropy-rate}
\leanverified{paper_projection_renyi_entropy_rate}
For every fixed integer \(q>1\), the visible law \(p_m\) satisfies
\[
  \sum_{x\in X_m} p_m(x)^q = 2^{-m q} S_q(m),
\]
and hence
\[
  h_q
  :=
  \lim_{m\to\infty}\frac{1}{m}H_q(p_m)
  =
  \frac{q\log 2-p_q}{q-1}.
\]
Moreover,
\[
  \lim_{q\to\infty} h_q
  =
  \log 2-\frac12\log\varphi.
\]
If \(q=q(m)\) is any integer sequence with \(q(m)\to\infty\), then
\[
  -\frac{1}{m\,q(m)}\log \sum_{x\in X_m} p_m(x)^{q(m)}
  \to
  \log 2-\frac12\log\varphi.
\]
\end{theorem}

\begin{proof}
For every integer \(q\ge 2\),
\[
  \sum_{x\in X_m} p_m(x)^q
  =
  \sum_{x\in X_m}\left(\frac{d_m(x)}{2^m}\right)^q
  =
  2^{-m q}S_q(m).
\]
Hence
\[
  H_q(p_m)
  =
  \frac{1}{1-q}\log\sum_{x\in X_m} p_m(x)^q
  =
  \frac{1}{1-q}\bigl(\log S_q(m)-m q\log 2\bigr).
\]
Divide by \(m\) and let \(m\to\infty\).  The fixed-\(q\) limit follows from
\Cref{thm:all-q-transfer}, and the \(q\to\infty\) limit from
\Cref{cor:renyi-endpoint}.  For \(q(m)\to\infty\), use
\[
  -\frac{1}{m\,q(m)}\log \sum_{x\in X_m} p_m(x)^{q(m)}
  =
  \log 2-\frac{1}{m\,q(m)}\log S_{q(m)}(m),
\]
and apply \Cref{thm:diagonal-high-moments}.
\end{proof}

\begin{corollary}[Visible-tail envelope for thick fibers]
\label{cor:visible-thick-tail}
\leanverified{paper_projection_visible_thick_tail}
For \(\alpha\ge 0\), let
\[
  \mathcal P_m(\alpha)
  :=
  p_m\bigl(\{x\in X_m:\ d_m(x)\ge e^{\alpha m}\}\bigr).
\]
Then
\[
  \limsup_{m\to\infty}\frac{1}{m}\log \mathcal P_m(\alpha)
  \le
  -\sup_{q\ge 1}\bigl((q-1)\alpha+\log 2-\log\lambda_q\bigr),
\]
where the supremum runs over positive integers \(q\).  In particular, if
\(\alpha>\frac12\log\varphi\), then \(\mathcal P_m(\alpha)=0\) for all
sufficiently large \(m\).  If \(0\le \alpha<\frac12\log\varphi\), then
\[
  \liminf_{m\to\infty}\frac{1}{m}\log \mathcal P_m(\alpha)
  \ge
  \frac12\log\varphi-\log 2.
\]
\end{corollary}

\begin{proof}
Since \(S_1(m)=2^m\) for all \(m\), \Cref{thm:all-q-transfer} forces
\(\lambda_1=2\).  Fix \(\alpha\ge 0\), and let \(q\ge 1\) be an integer.  On the
event \(\{d_m(x)\ge e^{\alpha m}\}\), one has
\(
d_m(x)\le e^{-(q-1)\alpha m}d_m(x)^q.
\)
Therefore
\begin{align*}
  \mathcal P_m(\alpha)
  &=
  2^{-m}\sum_{\substack{x\in X_m\\ d_m(x)\ge e^{\alpha m}}} d_m(x) \\
  &\le
  2^{-m}e^{-(q-1)\alpha m}
  \sum_{x\in X_m} d_m(x)^q \\
  &=
  2^{-m}e^{-(q-1)\alpha m}S_q(m).
\end{align*}
Applying \Cref{thm:all-q-transfer} and then taking the supremum over \(q\ge 1\)
gives the upper bound.

If \(\alpha>\frac12\log\varphi\), \Cref{thm:max-fiber} gives
\(D_m<e^{\alpha m}\) for all sufficiently large \(m\), so the tail is empty.
If \(0\le \alpha<\frac12\log\varphi\), then for all sufficiently large \(m\) the
maximal fiber belongs to the tail, hence
\[
  \mathcal P_m(\alpha)\ge \frac{D_m}{2^m}.
\]
Now apply \Cref{thm:max-fiber}.
\end{proof}

\begin{corollary}[Collision entropy rate]
\label{thm:collision-entropy-rate}
\leanverified{paper_projection_collision_entropy_rate}
For the visible law \(p_m\),
\[
  \sum_{x\in X_m}p_m(x)^2=\frac{T_2(m)}{4^m}.
\]
Hence
\[
  \lim_{m\to\infty}
  -\frac{1}{m}\log \sum_{x\in X_m}p_m(x)^2
  =
  2\log 2-\log \lambda_2.
\]
More precisely, with \(\rho=\lambda_2\) and with \(\lambda_-,\lambda_+\) as in
\Cref{thm:q2-alternating-second-order},
\[
  \sum_{x\in X_m}p_m(x)^2
  =
  c_\rho\left(\frac{\rho}{4}\right)^m
  +
  c_-\left(\frac{\lambda_-}{4}\right)^m
  +
  O\left(\left(\frac{\lambda_+}{4}\right)^m\right),
\]
so the second-order correction is eventually alternating.
\end{corollary}

\begin{proof}
The identity \(\sum_x p_m(x)^2=T_2(m)/4^m\) is immediate from the definition of
\(p_m\).  The asymptotic expansion is exactly
\Cref{thm:q2-alternating-second-order} divided by \(4^m\), and
\(\rho=\lambda_2\) by \Cref{thm:all-q-transfer} and the discussion after
\Cref{thm:cubic-recurrence}.
\end{proof}

\begin{corollary}[R\'enyi-2 asymptotics]
\label{cor:renyi2}
Let
\[
  H_2(p_m):=-\log \sum_{x\in X_m}p_m(x)^2.
\]
Then
\[
  H_2(p_m)
  =
  m(2\log 2-\log \rho)-\log c_\rho
  +
  O\left(\left|\frac{\lambda_-}{\rho}\right|^m\right).
\]
After subtracting the affine term
\(m(2\log 2-\log \rho)-\log c_\rho\), the remaining error is eventually
alternating in sign.
\end{corollary}

\begin{proof}
Factor the expansion of \Cref{thm:collision-entropy-rate} as
\[
  \sum_{x\in X_m}p_m(x)^2
  =
  c_\rho\left(\frac{\rho}{4}\right)^m
  \left(
    1+\frac{c_-}{c_\rho}\left(\frac{\lambda_-}{\rho}\right)^m
    +O\left(\left(\frac{\lambda_+}{\rho}\right)^m\right)
  \right).
\]
Taking logarithms yields the result, since
\(|\lambda_-|>\lambda_+\) and \(\lambda_-<0\).
\end{proof}
