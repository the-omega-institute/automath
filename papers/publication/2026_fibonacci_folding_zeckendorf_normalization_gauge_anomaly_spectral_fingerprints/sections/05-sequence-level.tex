\section{Sequence-level folding: sliding block codes, sofic models, and a conjugacy threshold}
\label{sec:sequence}

\subsection{A sliding block folding code and its image}
Let \(\eta=(\eta_t)_{t\in \ZZ}\in \set{0,1}^{\ZZ}\) be a two-sided microstate sequence. Fix \(m\ge 1\) and define the length-\(m\) sliding block folding code
\begin{equation}\label{eq:Phi}
(\Phi_m(\eta))_t
:=
\Fold_m(\eta_{t-m+1}\cdots \eta_t)\in X_m,
\qquad t\in \ZZ.
\end{equation}
The image shift is
\[
Y_m := \Phi_m(\set{0,1}^{\ZZ})\subset X_m^{\ZZ},
\]
equipped with the left shift on the \(X_m\)-alphabet.

\subsection{An explicit sofic presentation}
\begin{proposition}[Sofic model]\label{prop:sofic}
For each \(m\ge 1\), the image shift \(Y_m\) is sofic and admits an explicit labeled-graph presentation.
\end{proposition}

\begin{proof}[Construction]
Let \(V_m:=\set{0,1}^{m-1}\). For each \(v\in V_m\) and \(b\in \set{0,1}\), define the successor vertex \(v'\) by shifting and appending \(b\),
\[
v'= \text{suffix}_{m-1}(vb),
\]
and label the directed edge \(v\to v'\) by
\[
\lambda_m(vb):=\Fold_m(vb)\in X_m.
\]
Let \(\Sigma_m\) be the edge shift of this finite directed graph and let \(\Lambda_m:\Sigma_m\to X_m^{\ZZ}\) be the label map. By construction, \(\Lambda_m(\Sigma_m)=Y_m\), hence \(Y_m\) is sofic.
\end{proof}

\subsection{A conjugacy threshold at \texorpdfstring{$m\ge 3$}{m >= 3}}
\begin{theorem}[Conjugacy for \(m \ge 3\)]\label{thm:conjugacy}
For \(m\ge 3\), the sliding block code \(\Phi_m:\set{0,1}^{\ZZ}\to Y_m\) is a topological conjugacy: there exists a sliding block inverse \(\Psi_m:Y_m\to \set{0,1}^{\ZZ}\) such that
\[
\Psi_m\circ \Phi_m=\mathrm{id}_{\set{0,1}^{\ZZ}},
\qquad
\Phi_m\circ \Psi_m=\mathrm{id}_{Y_m}.
\]
\end{theorem}

\begin{proof}
Appendix~\ref{app:sofic} proves that the labeled graph of Proposition~\ref{prop:sofic} is right-resolving and, more importantly, that every admissible three-label block determines the appended bit on its third edge. This gives an explicit inverse sliding block code
\[
(\Psi_m(y))_t=\psi_m(y_{t-2},y_{t-1},y_t),
\]
with memory \(2\) and anticipation \(0\). Since the edge shift of \(G_m\) is canonically conjugate to \(\{0,1\}^{\ZZ}\) by reading appended bits, \(\Phi_m\) is bijective onto \(Y_m\), and \(\Psi_m\) is its inverse.
\end{proof}

\begin{proposition}[The degenerate case \(m=2\)]\label{thm:m2}
When \(m=2\), the only failure of injectivity is that the two fixed microstate sequences \(0^\ZZ\) and \(1^\ZZ\) have the same image \((00)^\ZZ\). Away from this pair, \(\Phi_2\) is injective.
\end{proposition}

\begin{proof}
Appendix~\ref{app:sofic} identifies the only off-diagonal cycles in the fiber product for \(m=2\): they are the two ordered lifts of the constant label sequence \((00)^\ZZ\), corresponding to the two preimages \(0^\ZZ\) and \(1^\ZZ\). Every other label sequence contains \(01\) or \(10\), and these labels determine the adjacent microstate bits uniquely, so the resulting path lifts uniquely.
\end{proof}

\begin{remark}
Appendix~\ref{app:sofic} supplies the structural proofs, the complete \(m=3\) label table, and the uniform decoder window for \(m\ge3\). The important point is that two labels may leave a finite ambiguity in the terminal vertex, but three labels already determine the appended microstate bit, so the inverse code has bounded memory independent of \(m\).
\end{remark}
