\section{Sequence-level folding: sliding block codes, sofic models, and a conjugacy threshold}
\label{sec:sequence}

\subsection{A sliding block folding code and its image}
Let \(\eta=(\eta_t)_{t\in \ZZ}\in \set{0,1}^{\ZZ}\) be a two-sided microstate sequence. Fix \(m\ge 1\) and define the length-\(m\) sliding block folding code
\begin{equation}\label{eq:Phi}
(\Phi_m(\eta))_t
:=
\Fold_m(\eta_{t-m+1}\cdots \eta_t)\in X_m,
\qquad t\in \ZZ.
\end{equation}
The image shift is
\[
Y_m := \Phi_m(\set{0,1}^{\ZZ})\subset X_m^{\ZZ},
\]
equipped with the left shift on the \(X_m\)-alphabet.

\subsection{An explicit sofic presentation}
\begin{proposition}[Sofic model]\label{prop:sofic}
For each \(m\ge 1\), the image shift \(Y_m\) is sofic and admits an explicit labeled-graph presentation.
\end{proposition}

\begin{proof}[Construction]
Let \(V_m:=\set{0,1}^{m-1}\). For each \(v\in V_m\) and \(b\in \set{0,1}\), define the successor vertex \(v'\) by shifting and appending \(b\),
\[
v'= \text{suffix}_{m-1}(vb),
\]
and label the directed edge \(v\to v'\) by
\[
\lambda_m(vb):=\Fold_m(vb)\in X_m.
\]
Let \(\Sigma_m\) be the edge shift of this finite directed graph and let \(\Lambda_m:\Sigma_m\to X_m^{\ZZ}\) be the label map. By construction, \(\Lambda_m(\Sigma_m)=Y_m\), hence \(Y_m\) is sofic.
\end{proof}

\subsection{A conjugacy threshold at \(m \ge 3\)}
\begin{theorem}[Conjugacy for \(m \ge 3\)]\label{thm:conjugacy}
For \(m\ge 3\), the sliding block code \(\Phi_m:\set{0,1}^{\ZZ}\to Y_m\) is a topological conjugacy: there exists a sliding block inverse \(\Psi_m:Y_m\to \set{0,1}^{\ZZ}\) such that
\[
\Psi_m\circ \Phi_m=\mathrm{id}_{\set{0,1}^{\ZZ}},
\qquad
\Phi_m\circ \Psi_m=\mathrm{id}_{Y_m}.
\]
Moreover, \(\Psi_m\) can be chosen with memory length \(m-1\).
\end{theorem}

\begin{proof}
Appendix~\ref{app:sofic} proves that the graph of Proposition~\ref{prop:sofic} is right-resolving and synchronizing with delay \(m-1\) for every \(m\ge 3\). Therefore each block of \(m-1\) consecutive labels determines the underlying vertex, and hence the next microstate bit, uniquely. The resulting decoder is a sliding block code \(\Psi_m\) of memory \(m-1\), and the two compositions are identities by construction.
\end{proof}

\begin{theorem}[The degenerate case m=2]\label{thm:m2}
When \(m=2\), the only failure of injectivity is that the two fixed microstate sequences \(0^\ZZ\) and \(1^\ZZ\) have the same image \((00)^\ZZ\). Away from this pair, \(\Phi_2\) is injective.
\end{theorem}

\begin{proof}
Appendix~\ref{app:sofic} gives the complete argument. The only collision occurs for the constant label sequence \((00)^\ZZ\), whose two preimages are the two fixed microstate sequences \(0^\ZZ\) and \(1^\ZZ\). Any label sequence containing \(01\) or \(10\) determines the neighboring microstate bits uniquely and therefore propagates to a unique bi-infinite preimage.
\end{proof}

\begin{remark}
Appendix~\ref{app:sofic} supplies the structural proofs, proves the synchronization statement on the labeled graph, and records an explicit inverse-code construction.
\end{remark}
