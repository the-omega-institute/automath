\section{Restriction discrepancy and the bulk pair shift}
\label{sec:gauge}

For \(\omega\in \Om_{m+1}\), set
\[
\Delta^{(m)}(\omega)
:=
\naive_{m+1\to m}(\omega)\oplus \aware_{m+1\to m}(\omega)\in \{0,1\}^m
\]
and define the \emph{restriction discrepancy}
\begin{equation}\label{eq:gauge-anomaly}
G_m(\omega):=\Hamming{\Delta^{(m)}(\omega)}.
\end{equation}
Thus \(G_m\) measures, coordinate by coordinate, the failure of naive truncation to agree with the restriction dictated by normalization.

\subsection{Worst cases and bounded repairs}

\begin{proposition}[Extensive worst case]\label{prop:worst}
For every \(m\ge 1\) there exists an explicit word \(\omega^\star\in \Om_{m+1}\) such that
\[
G_m(\omega^\star)=m.
\]
One may take \(\omega^\star\) to be
\[
\omega^\star=
\begin{cases}
(110)^n, & m+1=3n,\\
(110)^n0, & m+1=3n+1,\\
(110)^n11, & m+1=3n+2.
\end{cases}
\]
\end{proposition}

\begin{proof}
Write \(L=m+1\). The identity \(\Fib_{k+1}+\Fib_{k+2}=\Fib_{k+3}\) shows that each block \(110\) may be replaced by \(001\) without changing Fibonacci value. Repeating this blockwise gives
\[
\Fold_L\bigl((110)^n\bigr)=(001)^n,
\qquad
\Fold_L\bigl((110)^n0\bigr)=(001)^n0.
\]
Indeed, the words on the right already lie in \(X_L\), so uniqueness of the Zeckendorf normal form identifies them as the folds.

In the remaining case \(L=3n+2\), the terminal block \(11\) occupies positions \(L-1\) and \(L\) and contributes
\[
\Fib_L+\Fib_{L+1}=\Fib_{L+2},
\]
which is exactly the weight of a single \(1\) at the new position \(L+1\). Hence
\[
N\bigl((110)^n11\bigr)=N\bigl((001)^n00\bigr)+\Fib_{L+2},
\]
and the first \(L\) digits of \(\Fold_L((110)^n11)\) are \((001)^n00\).

Now delete the last coordinate. When \(L=3n\), the naive truncation is \((110)^{n-1}11\) and the compatible truncation is \((001)^{n-1}00\); when \(L=3n+1\), they are \((110)^n\) and \((001)^n\); when \(L=3n+2\), they are \((110)^n1\) and \((001)^n0\). In each case the two length-\(m\) words are coordinatewise complementary. Therefore \(G_m(\omega^\star)=m\).
\end{proof}

\begin{proposition}[No bounded tail patch]\label{prop:tail-patch}
Fix \(\ell\ge 1\). There is no family of maps
\[
P_{m,\ell}:\{0,1\}^{\ell+1}\to \{0,1\}^{\ell}
\]
such that, for all sufficiently large \(m\) and all \(\omega\in \Om_{m+1}\),
\begin{equation}\label{eq:tail-patch-form}
\aware_{m+1\to m}(\omega)
=
\omega_1\cdots \omega_{m-\ell}\,
P_{m,\ell}(\omega_{m-\ell+1}\cdots \omega_{m+1}).
\end{equation}
\end{proposition}

\begin{proof}
Assume that \eqref{eq:tail-patch-form} holds for some fixed \(\ell\) and all sufficiently large \(m\). Choose \(m>\ell\) in that range and let \(\omega^\star\) be the word from Proposition~\ref{prop:worst}. Since \(G_m(\omega^\star)=m\), the words \(\naive_{m+1\to m}(\omega^\star)\) and \(\aware_{m+1\to m}(\omega^\star)\) differ at every coordinate, in particular on the first \(m-\ell\) coordinates.

The right-hand side of \eqref{eq:tail-patch-form}, however, leaves those first \(m-\ell\) coordinates unchanged and modifies only the last \(\ell\) positions. This contradiction proves the claim.
\end{proof}

\subsection{The bulk pair process}
\label{subsec:joint-law}

To describe the limiting statistics, let \(\eta=(\eta_t)_{t\in\ZZ}\) be a bi-infinite Bernoulli-\(\frac12\) sequence, and let \((X_t,Y_t)_{t\in\ZZ}\) be the sitewise raw/normalized pair process encoded by the reversed pair shift of Appendix~\ref{app:pressure}. Thus \(X_t=\eta_t\), while \(Y_t\) is the normalized digit seen at the same site in the bulk limit. The law of this stationary process will be called the \emph{bulk law}.

\begin{theorem}[Interior language theorem]\label{thm:pair-language}
Let \(G_{\mathrm{pair}}\) be the labeled graph with state set \(\mathcal{S}=\{00,01,10,11\}\) and edges
\[
00 \xrightarrow{00} 00,\quad
00 \xrightarrow{01} 01,\quad
00 \xrightarrow{11} 11,\quad
01 \xrightarrow{10} 10,
\]
\[
10 \xrightarrow{10} 00,\quad
10 \xrightarrow{00} 01,\quad
10 \xrightarrow{11} 01,\quad
11 \xrightarrow{00} 00.
\]
For a finite word \(w\in(\{00,01,10,11\})^\ell\), the following are equivalent.
\begin{enumerate}
\item[(i)] \(w\) occurs as an interior factor of some reversed pair word \(\mathcal{R}_M(\omega)\).
\item[(ii)] \(w\) is the label of a path in \(G_{\mathrm{pair}}\).
\end{enumerate}
Moreover, every path label in \(G_{\mathrm{pair}}\) is realized at arbitrarily large distance from the boundary by a reversed pair word \(\mathcal{R}_M(\omega)\).
\end{theorem}

\begin{proof}
Suppose first that \(w\) occurs as an interior factor of \(\mathcal{R}_M(\omega)\). By Lemma~\ref{lem:zero-padding}, we may prepend arbitrarily many high-weight zeros and assume that the factor is preceded by a synchronized block \(00|00\). Appendix~\ref{app:pressure} classifies all synchronized length-\(3\) and length-\(4\) factors. The length-\(3\) classification shows that, once the synchronized regime has been entered, the future depends only on the current suffix state \(s\in\mathcal{S}\), namely on the last pair symbol in
\[
00|00,\qquad 00|01,\qquad 00|10,\qquad 00|11.
\]
The length-\(4\) classification then shows that the admissible one-step extensions are exactly the edges displayed above. Reading the factor from left to right therefore produces a path in \(G_{\mathrm{pair}}\) with label \(w\).

Conversely, every edge of \(G_{\mathrm{pair}}\) is realized by one of the synchronized length-\(4\) factors listed in Appendix~\ref{app:pressure}. Start with a long block \(00^L\), which keeps the presentation in state \(00\). We claim by induction on the path length that any labeled path in \(G_{\mathrm{pair}}\) can be realized after this synchronized prefix. The initial step is the realization of a single edge. For the induction step, suppose a synchronized factor ending in state \(s\) has been realized. By construction of the states as synchronized follower classes, the admissible continuations from that suffix are precisely the outgoing edges from \(s\), and Appendix~\ref{app:pressure} gives an explicit synchronized length-\(4\) witness for each such continuation. Appending that witness realizes the next edge and moves the suffix state to the prescribed target state. Iterating along the path realizes the full label word.

Finally, Lemma~\ref{lem:zero-padding} allows us to increase the initial high-weight padding arbitrarily, so the realized factor can be placed arbitrarily far from the boundary.
\end{proof}

Since the first coordinate of each pair symbol is the microstate digit itself, the map \(\omega\mapsto \mathcal{R}_M(\omega)\) is injective, and the uniform measure on \(\Om_M\) is exactly the uniform measure on admissible reversed pair words of length \(M\). Proposition~\ref{prop:bulk-limit} therefore identifies the bulk law as the local weak limit of the uniform measures on \(\Om_M\).

\begin{theorem}[Exact bulk joint law]\label{thm:joint}
Under the bulk law, the one-site joint distribution of \((X_0,Y_0)\) is
\[
\begin{array}{c|cc}
& Y_0=0 & Y_0=1\\\hline
X_0=0 & \frac{7}{18} & \frac{1}{9}\\
X_0=1 & \frac{1}{3} & \frac{1}{6}.
\end{array}
\]
Consequently,
\[
\PP(X_0\neq Y_0)=\frac{4}{9},
\qquad
\PP(Y_0=1)=\frac{5}{18}.
\]
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:pair-language}, the graph \(G_{\mathrm{pair}}\) is a primitive right-resolving presentation of the interior reversed-pair language. In the state order \((00,01,10,11)\), its adjacency matrix is
\[
A=\begin{pmatrix}
1&1&0&1\\
0&0&1&0\\
1&2&0&0\\
1&0&0&0
\end{pmatrix}.
\]
The graph is primitive, so its bulk measure is the Parry measure \cite{Parry1964,ParryPollicott1990,Ruelle2004}. Appendix~\ref{app:pressure} computes the Perron eigenvalue \(\lambda=2\), right/left eigenvectors \(r=(2,1,2,1)^{\mathsf T}\) and \(\ell=(2,2,1,1)^{\mathsf T}\), and the stationary state distribution
\[
\pi=\frac{1}{9}(4,2,2,1).
\]
The probability of a labeled edge \(i\to j\) is \(\pi_iP_{ij}/A_{ij}\), where \(P\) is the Parry transition matrix and the multiplicity-\(2\) edge \(10\to 01\) is split uniformly between the labels \(00\) and \(11\). Summing over the four labels gives
\[
\PP(00)=\frac{7}{18},\qquad
\PP(01)=\frac{1}{9},\qquad
\PP(10)=\frac{1}{3},\qquad
\PP(11)=\frac{1}{6},
\]
which is the stated table.
\end{proof}

The marginal identity \(\PP(Y_0=1)=5/18\) agrees with the classical value obtained for the normalized Fibonacci digit process in \cite{AlexanderZagier1991,SidorovVershik1998,KemptonDigits2014,BezhaevaOseledets2007,BezhaevaOseledets2010}. What is specific here is that the four-state presentation encodes the pair observable \((X_t,Y_t)\), and therefore the discrepancy process \(X_t\oplus Y_t\).

\subsection{Limit theorems for the discrepancy process}
\label{subsec:lln-clt}

Let \(\Delta_t:=X_t\oplus Y_t\), and for \(n\ge 1\) set
\[
S_n:=\sum_{t=1}^n \Delta_t.
\]

\begin{corollary}[LLN and CLT for anomaly density]\label{thm:llnclt}
Under the bulk law,
\[
\frac{S_n}{n}\xrightarrow{\ \mathrm{a.s.}\ }\frac{4}{9},
\]
and
\[
\frac{S_n-\frac{4}{9}n}{\sqrt n}\Rightarrow \mathcal{N}\!\left(0,\frac{118}{243}\right).
\]
\end{corollary}

\begin{proof}
Appendix~\ref{app:pressure} realizes \(\Delta_t\) as a bounded observable of an irreducible, aperiodic, finite-state Markov chain, namely the edge chain of the four-state presentation. The almost-sure law of large numbers is therefore the ergodic theorem for finite-state Markov chains \cite[Section~4]{Billingsley1961}. The central limit theorem follows from the standard Markov-chain CLT, for instance in Gordin's martingale-coboundary form \cite{Gordin1969}; see also \cite[Chapter~4]{ParryPollicott1990}. The asymptotic variance is the second derivative at \(0\) of the logarithmic pressure of the tilted matrix \(A_\theta\) computed in Appendix~\ref{app:pressure}, namely
\[
\Lambda''(0)=\frac{118}{243}.
\]
\end{proof}

Because \((X_t,Y_t)\) is the bulk limit of uniformly chosen finite windows, the constants in Theorem~\ref{thm:joint} and Corollary~\ref{thm:llnclt} govern the interior statistics of \(G_m\): boundary sites contribute only \(O(1)\) terms and do not affect the limiting density or variance.

\subsection{Autocovariances and power spectrum}
\label{subsec:spectrum}

\begin{theorem}[Autocovariance and spectrum]\label{thm:spectrum}
The autocovariances \(c_j:=\Cov(\Delta_0,\Delta_j)\) satisfy
\[
c_0=\frac{20}{81},
\]
and, for \(j\ge 1\),
\[
c_j=2^{-j}\!\left(\frac{1}{8}+(-1)^j\!\left(\frac{7}{648}+\frac{j}{108}\right)\right).
\]
The two-sided power spectral density
\[
S(\omega):=\sum_{j\in \ZZ} c_j\,e^{-i\omega j}
\]
is the rational function
\[
S(\omega)
=
\frac{2\bigl(32\cos^3\omega+12\cos^2\omega-116\cos\omega-105\bigr)}
{9\,(4\cos\omega-5)\,(4\cos\omega+5)^2}.
\]
In particular,
\[
S(0)=\sum_{j\in\ZZ} c_j=\frac{118}{243},
\]
in agreement with Corollary~\ref{thm:llnclt}.
\end{theorem}

\begin{proof}
Appendix~\ref{app:pressure} expresses the covariances as
\[
c_j=qP^{j-1}h-\left(\frac49\right)^2\qquad (j\ge1),
\]
for explicit vectors \(q\) and \(h\) attached to the Parry transition matrix \(P\). Because \(P\) is similar to \(A/2\), its nontrivial spectral modes are \(1/2\) and \(-1/2\), with a double root at \(-1/2\). Hence \(c_j\) is a linear combination of \(2^{-j}\), \((-2)^{-j}\), and \(j(-2)^{-j}\), and matching the first three values yields the displayed formula. The series is absolutely summable, so the power spectral density is obtained by summing the resulting cosine series. Appendix~\ref{app:pressure} performs that sum explicitly and obtains the stated rational function.
\end{proof}
