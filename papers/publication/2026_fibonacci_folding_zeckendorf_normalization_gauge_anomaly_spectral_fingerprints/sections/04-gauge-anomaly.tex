\section{Gauge anomaly: worst cases, limit laws, and spectral formulas}
\label{sec:gauge}

\subsection{Definition}
Consider the scale change \(m+1\to m\). For \(\omega\in \Om_{m+1}\), define the \emph{gauge anomaly} as the Hamming distance between naive truncation and fold-aware restriction:
\begin{equation}\label{eq:gauge-anomaly}
G_m(\omega)
:=
\Hamming{\naive_{m+1\to m}(\omega)\oplus \aware_{m+1\to m}(\omega)}.
\end{equation}
Equivalently, if
\[
\Delta^{(m)}(\omega)
:=
\naive_{m+1\to m}(\omega)\oplus \aware_{m+1\to m}(\omega)\in \set{0,1}^m,
\]
then \(G_m(\omega)=\sum_{k=1}^m \Delta^{(m)}_k(\omega)\).

\subsection{Worst-case extensive deviation and non-patchability}
\begin{proposition}[Extensive worst case]\label{prop:worst}
For every \(m\ge 1\) there exists an explicit microstate \(\omega^\star\in \Om_{m+1}\) such that
\[
G_m(\omega^\star)=m.
\]
\end{proposition}

\begin{remark}[An explicit worst-case family]\label{rem:explicit-worst}
Let \(L=m+1\) and write \(L=3n+r\) with \(r\in\{0,1,2\}\). Define a length-\(L\) word \(\omega^\star\) by
\[
\omega^\star=
\begin{cases}
(110)^n, & r=0,\\
(110)^n0, & r=1,\\
(110)^n11, & r=2,
\end{cases}
\]
where concatenation is in the window order \((\omega_1,\dots,\omega_{m+1})\).
\end{remark}

\begin{proof}
The identity \(\Fib_{k+1}+\Fib_{k+2}=\Fib_{k+3}\) shows that each block \(110\) may be replaced, independently of the neighboring blocks, by \(001\) without changing Fibonacci value. Therefore
\[
\Fold_L\bigl((110)^n\bigr)=(001)^n,
\qquad
\Fold_L\bigl((110)^n0\bigr)=(001)^n0.
\]
In the remaining case,
\[
(110)^n11
\]
has the same value as \((001)^n00\) on the first \(L\) coordinates together with one extra \(1\) at position \(L+1\), because the terminal block \(11\) contributes \(\Fib_{L+1}+\Fib_{L+2}=\Fib_{L+3}\). Hence the first \(L\) digits of \(\Fold_L((110)^n11)\) are \((001)^n00\).

Now delete the last coordinate. When \(r=0\), the naive truncation is \((110)^{n-1}11\) and the fold-aware truncation is \((001)^{n-1}00\); when \(r=1\), they are \((110)^n\) and \((001)^n\); when \(r=2\), they are \((110)^n1\) and \((001)^n0\). In each case the two length-\(m\) words are coordinatewise complementary. Therefore every coordinate contributes \(1\) to the Hamming distance, so \(G_m(\omega^\star)=m\).
\end{proof}

\begin{proposition}[No bounded tail patch]\label{prop:tail-patch}
Fix \(\ell\ge 1\). There is no family of maps
\[
P_{m,\ell}:\set{0,1}^{\ell+1}\to \set{0,1}^{\ell}
\]
such that, for all sufficiently large \(m\) and all \(\omega\in \Om_{m+1}\),
\begin{equation}\label{eq:tail-patch-form}
\aware_{m+1\to m}(\omega)
=
\omega_1\cdots \omega_{m-\ell}\,
P_{m,\ell}(\omega_{m-\ell+1}\cdots \omega_{m+1}).
\end{equation}
In particular, the anomaly cannot be uniformly removed by modifying only a bounded tail of the naive truncation.
\end{proposition}

\begin{proof}
Assume that \eqref{eq:tail-patch-form} holds for some fixed \(\ell\) and all sufficiently large \(m\). Choose \(m>\ell\) in that range and let \(\omega^\star\) be the word from Proposition~\ref{prop:worst}. Since \(G_m(\omega^\star)=m\), the words \(\naive_{m+1\to m}(\omega^\star)\) and \(\aware_{m+1\to m}(\omega^\star)\) differ at every coordinate. In particular, their first \(m-\ell\) coordinates differ.

But the right-hand side of \eqref{eq:tail-patch-form} leaves those first \(m-\ell\) coordinates unchanged: it agrees with \(\omega^\star_1\cdots \omega^\star_{m-\ell}\) there and only modifies the last \(\ell\) positions. This contradicts the previous paragraph. Hence no such bounded tail patch exists.
\end{proof}

\subsection{Bulk stationary model and the exact joint law}
\label{subsec:joint-law}
To state the limiting statistical results precisely, we pass from finite windows to the stationary bulk model. Let \(\eta=(\eta_t)_{t\in\ZZ}\) be a bi-infinite Bernoulli-\(\frac12\) sequence, and let \((X_t,Y_t)_{t\in\ZZ}\) be the input/output pair process produced by the boundary-state normalization transducer of Appendix~\ref{app:pressure}. Thus \(X_t=\eta_t\), while \(Y_t\) is the stable digit emitted at site \(t\). By construction, \((X_t,Y_t)\) is stationary. We refer to its law as the \emph{bulk law}. Standard path-counting for primitive right-resolving presentations implies that this bulk law is the local weak limit of the uniform measures on \(\Om_M\): if \(\omega^{(M)}\) is uniform on \(\Om_M\) and \(k_M\to\infty\) with \(M-k_M\to\infty\), then
\[
\bigl(\omega^{(M)}_{k_M},(\Fold_M(\omega^{(M)}))_{k_M}\bigr)
\Rightarrow
(X_0,Y_0);
\]
see \cite[Chapter~8]{LindMarcus1995} and \cite[Chapter~2]{Kitchens1998}.

\begin{theorem}[Exact bulk joint law]\label{thm:joint}
Under the bulk law, the one-site joint distribution of \((X_0,Y_0)\) is
\[
\begin{array}{c|cc}
& Y_0=0 & Y_0=1\\\hline
X_0=0 & \frac{7}{18} & \frac{1}{9}\\
X_0=1 & \frac{1}{3} & \frac{1}{6}.
\end{array}
\]
Consequently,
\[
\PP(X_0\neq Y_0)=\frac{4}{9},
\qquad
\PP(Y_0=1)=\frac{5}{18}.
\]
\end{theorem}

\begin{proof}
Appendix~\ref{app:pressure} derives the four-state boundary transducer associated with the local rewrite \(011\to 100\). In the state order \((00,01,10,11)\), its adjacency matrix is
\[
A=\begin{pmatrix}
1&1&0&1\\
0&0&1&0\\
1&2&0&0\\
1&0&0&0
\end{pmatrix},
\]
and the edges are labeled by \((X_t,Y_t)\in\{0,1\}^2\) as follows:
\[
00 \xrightarrow{00} 00,\quad
00 \xrightarrow{01} 01,\quad
00 \xrightarrow{11} 11,\quad
01 \xrightarrow{10} 10,
\]
\[
10 \xrightarrow{10} 00,\quad
10 \xrightarrow{00} 01,\quad
10 \xrightarrow{11} 01,\quad
11 \xrightarrow{00} 00.
\]
The graph is primitive, so its bulk measure is the Parry measure \cite{ParryPollicott1990,Ruelle2004}. Appendix~\ref{app:pressure} computes the Perron eigenvalue \(\lambda=2\), right/left eigenvectors \(r=(2,1,2,1)^{\mathsf T}\) and \(\ell=(2,2,1,1)^{\mathsf T}\), and the stationary state distribution
\[
\pi=\frac{1}{9}(4,2,2,1).
\]
The probability of a labeled edge \(i\to j\) is \(\pi_iP_{ij}/A_{ij}\), where \(P\) is the Parry transition matrix and the multiplicity-\(2\) edge \(10\to 01\) is split uniformly between the labels \(00\) and \(11\). Summing the corresponding edge probabilities over the four labels gives
\[
\PP(00)=\frac{7}{18},\qquad
\PP(01)=\frac{1}{9},\qquad
\PP(10)=\frac{1}{3},\qquad
\PP(11)=\frac{1}{6},
\]
which is exactly the stated table. The two displayed marginals follow by summation.
\end{proof}

\subsection{Law of large numbers and central limit theorem}
\label{subsec:lln-clt}
Let \(\Delta_t:=X_t\oplus Y_t\) be the stationary discrepancy indicator process in the bulk. For \(n\ge 1\), set
\[
S_n:=\sum_{t=1}^n \Delta_t.
\]

\begin{theorem}[LLN and CLT for anomaly density]\label{thm:llnclt}
Under the bulk law,
\[
\frac{S_n}{n}\xrightarrow{\ \mathrm{a.s.}\ }\frac{4}{9},
\]
and
\[
\frac{S_n-\frac{4}{9}n}{\sqrt n}\ \Rightarrow\ \mathcal{N}\!\left(0,\frac{118}{243}\right).
\]
\end{theorem}

\begin{proof}
Appendix~\ref{app:pressure} realizes \(\Delta_t\) as a bounded observable of an irreducible, aperiodic, finite-state Markov chain (equivalently, of the edge chain of the four-state presentation). The almost-sure law of large numbers is therefore the ergodic theorem for finite-state Markov chains \cite[Section~4]{Billingsley1961}. The central limit theorem follows from the standard Markov-chain CLT, for instance in Gordin's martingale-coboundary form \cite{Gordin1969}; see also \cite[Chapter~4]{ParryPollicott1990}. The asymptotic variance is the second derivative at \(0\) of the logarithmic pressure of the tilted matrix \(A_\theta\) computed in Appendix~\ref{app:pressure}, namely
\[
\Lambda''(0)=\frac{118}{243}.
\]
\end{proof}

\begin{remark}[Finite-window interpretation]
The process \((X_t,Y_t)\) is the bulk limit of the uniform measures on finite windows. Consequently, the constants in Theorems~\ref{thm:joint} and \ref{thm:llnclt} govern the interior statistics of the finite-window anomaly \(G_m\); boundary effects contribute only \(O(1)\) sites and do not alter the limiting density or variance.
\end{remark}

\subsection{Closed-form autocovariance and rational power spectrum}
\label{subsec:spectrum}
Let \((\Delta_t)_{t\in\ZZ}\) be the stationary bulk discrepancy process.

\begin{theorem}[Autocovariance and power spectral density]\label{thm:spectrum}
The autocovariances \(c_j:=\Cov(\Delta_0,\Delta_j)\) satisfy
\[
c_0=\frac{20}{81},
\qquad
c_j = 2^{-j}\!\left(\frac{1}{8}+(-1)^j\!\left(\frac{7}{648}+\frac{j}{108}\right)\right)\quad (j\ge 1).
\]
The two-sided power spectral density
\[
S(\omega):=\sum_{j\in \ZZ} c_j\,e^{-i\omega j}
\]
is the rational function
\[
S(\omega)
=
\frac{2\bigl(32\cos^3\omega+12\cos^2\omega-116\cos\omega-105\bigr)}
{9\,(4\cos\omega-5)\,(4\cos\omega+5)^2}.
\]
In particular,
\[
S(0)=\sum_{j\in\ZZ}c_j=\frac{118}{243},
\]
in agreement with Theorem~\ref{thm:llnclt}.
\end{theorem}

\begin{proof}
Appendix~\ref{app:pressure} expresses the covariances as
\[
c_j=qP^{j-1}h-\left(\frac49\right)^2\qquad (j\ge1),
\]
for explicit vectors \(q\) and \(h\) attached to the Parry transition matrix \(P\). Because \(P\) is similar to \(A/2\), its nontrivial spectral modes are \(1/2\) and \(-1/2\), with a double root at \(-1/2\). Hence \(c_j\) is a linear combination of \(2^{-j}\), \((-2)^{-j}\), and \(j(-2)^{-j}\), and matching the first three values yields the displayed formula. The series is absolutely summable, so the power spectral density is obtained by summing the resulting cosine series. Appendix~\ref{app:pressure} performs that sum explicitly and obtains the stated rational function. Evaluating at \(\omega=0\) gives \(S(0)=118/243\).
\end{proof}

\begin{remark}[Internal consistency]
The identity
\[
S(0)=c_0+2\sum_{j\ge 1} c_j
\]
is the Green--Kubo/Wiener--Khinchin consistency relation linking the CLT variance, the covariance sum, and the power spectrum. The formulas above therefore provide three equivalent descriptions of the same second-order statistic.
\end{remark}
