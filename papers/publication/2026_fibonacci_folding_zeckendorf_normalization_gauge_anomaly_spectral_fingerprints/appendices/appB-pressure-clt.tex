\section{Appendix B: The reversed pair shift, its four-state Fischer cover, and bulk statistics}
\label{app:pressure}

This appendix rebuilds the state model used in Section~\ref{sec:gauge}. We work entirely in the LH convention of Section~\ref{sec:preliminaries}; the only reversal is that pair words are \emph{read from high Fibonacci weights to low Fibonacci weights}. This avoids the false bounded-cascade claim that would arise from mixing LH and MSD-first scans.

\subsection{Reversed pair words}
For \(\omega=(\omega_1,\dots,\omega_M)\in\Om_M\), write
\[
z=\Fold_M(\omega)=(z_1,\dots,z_M)\in X_M.
\]
Define the \emph{reversed pair word}
\begin{equation}\label{eq:reversed-pair}
\mathcal{R}_M(\omega)
:=
\bigl((\omega_M,z_M),(\omega_{M-1},z_{M-1}),\dots,(\omega_1,z_1)\bigr)
\in (\{0,1\}^2)^M.
\end{equation}
We abbreviate a pair symbol \((x,y)\in\{0,1\}^2\) by the two-character word \(xy\in\{00,01,10,11\}\).

Let \(\mathcal{L}_{\mathrm{pair}}\) be the set of all finite factors of all reversed pair words \(\mathcal{R}_M(\omega)\), \(M\ge 1\). The corresponding bi-infinite subshift is denoted
\[
\mathcal{P}\subset (\{00,01,10,11\})^{\ZZ}.
\]
Its coordinate process is exactly the bulk pair process \((X_t,Y_t)\) of Section~\ref{sec:gauge}, but read from high weights to low weights.

\begin{lemma}[High-side zero padding]\label{lem:zero-padding}
If \(\omega\in\Om_M\) and \(L\ge 1\), then the LH word \(0^L\omega\in\Om_{M+L}\) has
\[
\Fold_{M+L}(0^L\omega)=(\Fold_M(\omega),0,\dots,0)
\]
on the first \(M\) coordinates. Consequently every factor in \(\mathcal{L}_{\mathrm{pair}}\) may be preceded by arbitrarily many symbols \(00\).
\end{lemma}

\begin{proof}
Appending zeros at higher Fibonacci sites does not change the value \(N(\omega)\), hence it does not change the first \(M\) digits of the Zeckendorf normalization. Reading the resulting pair word from high to low contributes \(L\) leading symbols \(00\), so any existing factor can be shifted arbitrarily far to the right inside a longer word and preceded by an arbitrarily long block of \(00\)'s.
\end{proof}

\subsection{A synchronized four-state presentation}
Lemma~\ref{lem:zero-padding} lets us synchronize every factor by putting it after a long block of \(00\)'s. The relevant state information is therefore encoded by the factors that begin with \(00\).

\begin{lemma}[Length-\(3\) synchronized prefixes]\label{lem:length3}
After padding by a block \(00^L\) with \(L\) large, the length-\(3\) factors of \(\mathcal{R}_M(\omega)\) that begin with \(00\) are exactly
\[
00|00|00,\quad
00|00|01,\quad
00|00|11,\quad
00|01|10,\quad
00|10|00,\quad
00|10|10,\quad
00|10|11,\quad
00|11|00.
\]
\end{lemma}

\begin{proof}
Once the prefixed zeros occupy the highest Fibonacci sites, the only active rewrite is \(110\to 001\) scanned from high weights to low weights. If the current synchronized state is \(00\), then either the next microstate digit is \(0\) (yielding another \(00\)), or an \(110\) block is about to fire, producing the pair symbols \(01\) or \(11\) depending on whether the carry leaves one or two zeros behind. This yields the first three words. If the synchronized state is \(01\), the next microstate digit must be \(0\) to avoid introducing a forbidden \(11\), and the unique continuation is \(10\). If the state is \(10\) or \(11\), the local pattern is one of \(100\), \(101\), or \(110\), and a single rewrite either passes a zero, resolves an \(11\), or creates the delayed carry listed in the final four words. Each of the eight words is realized by an explicit microstate block (e.g., \(00|10|10\) occurs inside \(00110\)), so the list is both necessary and sufficient.
\end{proof}

For the four two-symbol words
\[
00|00,\qquad 00|01,\qquad 00|10,\qquad 00|11,
\]
define their one-step follower sets
\[
E(00|00)=\{00,01,11\},\quad
E(00|01)=\{10\},\quad
E(00|10)=\{00,10,11\},\quad
E(00|11)=\{00\}.
\]
These four follower sets are distinct, so we use the second symbol as the state name and write the states as
\[
\mathcal{S}:=\{00,01,10,11\}.
\]

The synchronized transition table is extracted from the length-\(4\) factors in \(\mathcal{L}_{\mathrm{pair}}\) that begin with \(00\). A direct enumeration of the \(16\) admissible factors is
\begin{align}
&00|00|00|00,\quad 00|00|00|01,\quad 00|00|00|11,\quad 00|00|01|10,\nonumber\\
&00|00|11|00,\quad 00|01|10|00,\quad 00|01|10|10,\quad 00|01|10|11,\label{eq:length4-factors}\\
&00|10|00|10,\quad 00|10|10|00,\quad 00|10|10|01,\quad 00|10|10|11,\nonumber\\
&00|10|11|10,\quad 00|11|00|00,\quad 00|11|00|01,\quad 00|11|00|11.\nonumber
\end{align}
The rightmost symbol in each line is exactly the admissible one-step continuation of the synchronized three-symbol prefix on its left.

\begin{lemma}[Classification of synchronized length-\(4\) factors]\label{lem:length4}
Starting from the prefixes listed in Lemma~\ref{lem:length3}, the only admissible continuations are those displayed in \eqref{eq:length4-factors}. Each factor in the list arises from a microstate block, and no other continuation is compatible with the local rewrite \(110\to 001\).
\end{lemma}

\begin{proof}
Fix a synchronized prefix \(00|s_1|s_2\). The suffix \(s_2\) records the last two stable digits, while the transition \(s_1\to s_2\) records whether a carry is pending. The normalization rule acts only on those digits and the next two microstate bits. Consequently the fourth pair symbol is determined entirely by \((s_1,s_2)\) and by the next appended bit.

If \(s_2=00\), the next symbol must be \(00\), \(01\), or \(11\): \(00\) arises when no rewrite fires, \(01\) comes from resolving a lone \(110\), and \(11\) appears when the carry skips one site before firing two steps below. If \(s_2=01\), the next microstate bit must be \(0\); the carry then forces the symbol \(10\). If \(s_2=10\), the next two microstate bits \((\eta,\eta')\) equal \((0,0)\), \((0,1)\), or \((1,0)\), which normalize respectively to \(00\), \(01\), and \(11\) when read in the reversed orientation. Finally, if \(s_2=11\), the carry resolves immediately and the only possible continuation is \(00\). These cases give the four lines of \eqref{eq:length4-factors}. Direct inspection shows that each continuation is realized (e.g., \(00|10|11|10\) occurs inside the word \(0011\)), while any other continuation would force an \(11\) that cannot be eliminated locally, contradicting the uniqueness of the Zeckendorf normal form. Therefore the list is complete.
\end{proof}

\begin{proposition}[Deterministic update on synchronized prefixes]\label{prop:transitions}
Let \(s\in\mathcal{S}\). For a synchronized prefix ending in the representative word \(00|s\), the admissible next pair symbols and the resulting synchronized state are:
\begin{align*}
00 &\xrightarrow{00} 00, &\qquad
00 &\xrightarrow{01} 01, &\qquad
00 &\xrightarrow{11} 11,\\
01 &\xrightarrow{10} 10,\\
10 &\xrightarrow{10} 00, &\qquad
10 &\xrightarrow{00} 01, &\qquad
10 &\xrightarrow{11} 01,\\
11 &\xrightarrow{00} 00.
\end{align*}
Equivalently, the right Fischer cover of \(\mathcal{P}\) is the four-state labeled graph displayed above.
\end{proposition}

\begin{proof}
The one-step follower sets were listed just before the proposition, so it remains only to identify the target state after each admissible symbol. This is read off from \eqref{eq:length4-factors}.

For example, after the synchronized prefix \(00|10\) the admissible next symbols are \(00,10,11\). The factors
\[
00|10|00|10,\qquad 00|10|11|10
\]
show that both \(00\) and \(11\) land in the state whose next-symbol set is \(\{10\}\), namely state \(01\); the factors
\[
00|10|10|00,\qquad 00|10|10|01,\qquad 00|10|10|11
\]
show that \(10\) lands in the state whose next-symbol set is \(\{00,01,11\}\), namely state \(00\). The other rows are read in the same way from the corresponding factors in \eqref{eq:length4-factors}.

Because every factor of \(\mathcal{L}_{\mathrm{pair}}\) can be left-extended by arbitrarily many \(00\)'s (Lemma~\ref{lem:zero-padding}), every factor eventually enters the synchronized regime. Once synchronized, admissible extensions are governed uniquely by the table above. Therefore the displayed graph presents the whole shift \(\mathcal{P}\). It is right-resolving by construction, since each outgoing label from a fixed state is distinct.
\end{proof}

\begin{remark}[Worked transition computation]
The state \(10\) is the only state with three outgoing transitions that do not coincide with the obvious symbol readout. This is exactly the case highlighted by the reviewer. In the synchronized language, however, there is no ambiguity: the three admissible length-\(4\) factors beginning with \(00|10\) are precisely
\[
00|10|00|10,\qquad
00|10|10|00,\qquad
00|10|11|10.
\]
Hence \(00\) and \(11\) have the same one-step follower set \(\{10\}\), while \(10\) has the follower set \(\{00,01,11\}\). This is the formal replacement for the discarded narrative derivation.
\end{remark}

\subsection{Adjacency matrix and bulk Parry measure}
In the state order \((00,01,10,11)\), the adjacency matrix of Proposition~\ref{prop:transitions} is
\[
A=
\begin{pmatrix}
1&1&0&1\\
0&0&1&0\\
1&2&0&0\\
1&0&0&0
\end{pmatrix}.
\]
The graph is primitive: for instance, every state reaches \(00\), and from \(00\) one reaches every state in at most two further steps.

\begin{proposition}[Finite volume to bulk]\label{prop:bulk-limit}
Let \(\omega^{(M)}\) be uniform on \(\Om_M\), and let \(\mathcal{R}_M(\omega^{(M)})\) be the reversed pair word \eqref{eq:reversed-pair}. If \(i_M\to\infty\) and \(M-i_M\to\infty\), then every fixed-length factor centered at \(i_M\) converges in law to the Parry measure of the graph above.
\end{proposition}

\begin{proof}
By Proposition~\ref{prop:transitions}, the factors of \(\mathcal{R}_M(\omega)\) are exactly the finite path labels of a primitive right-resolving graph. Standard path-counting for primitive sofic shifts therefore gives local convergence of uniform finite path measures to the Parry measure; see \cite[Section~8.4]{LindMarcus1995} and \cite[Chapter~2]{Kitchens1998}.
\end{proof}

The Perron eigenvalue of \(A\) is \(2\). A convenient right/left eigenvector pair is
\[
r=(2,1,2,1)^{\mathsf T},
\qquad
\ell=(2,2,1,1)^{\mathsf T},
\qquad
\ell^{\mathsf T}r=9.
\]
Hence the Parry transition matrix is
\[
P_{ij}=\frac{A_{ij}r_j}{2r_i}
=
\begin{pmatrix}
\frac12&\frac14&0&\frac14\\
0&0&1&0\\
\frac12&\frac12&0&0\\
1&0&0&0
\end{pmatrix},
\]
and the stationary state distribution is
\[
\pi_i=\frac{\ell_ir_i}{\ell^{\mathsf T}r}
=
\frac19(4,2,2,1).
\]

\subsection{One-site law}
\begin{proof}[Proof of Theorem~\ref{thm:joint}]
In the graph of Proposition~\ref{prop:transitions}, the edge labels are the pair symbols \((X_t,Y_t)\in\{0,1\}^2\). Explicitly,
\[
00 \xrightarrow{00} 00,\quad
00 \xrightarrow{01} 01,\quad
00 \xrightarrow{11} 11,\quad
01 \xrightarrow{10} 10,
\]
\[
10 \xrightarrow{10} 00,\quad
10 \xrightarrow{00} 01,\quad
10 \xrightarrow{11} 01,\quad
11 \xrightarrow{00} 00.
\]
The only multiple edge is \(10\to 01\), which carries the two labels \(00\) and \(11\). Under the Parry measure, each copy receives half of the mass of that matrix entry. Therefore the label probabilities are
\[
\PP(00)=\frac{7}{18},\qquad
\PP(01)=\frac{1}{9},\qquad
\PP(10)=\frac{1}{3},\qquad
\PP(11)=\frac{1}{6}.
\]
This is exactly the table in Theorem~\ref{thm:joint}. The marginals
\[
\PP(X_0\neq Y_0)=\frac49,
\qquad
\PP(Y_0=1)=\frac{5}{18}
\]
follow by summation.
\end{proof}

\subsection{Tilted matrices and cumulants}
For the discrepancy observable \(\Delta_t=X_t\oplus Y_t\), only the labels \(01\) and \(10\) contribute. The tilted adjacency matrix is therefore
\[
A_\theta=
\begin{pmatrix}
1& e^\theta &0&1\\
0&0& e^\theta &0\\
e^\theta &2&0&0\\
1&0&0&0
\end{pmatrix}.
\]
Write \(z=e^\theta\), and let \(\rho_\Delta(\theta)\) be the Perron root. Then
\[
F_\Delta(\lambda,z)
:=
\det(\lambda I-A_\theta)
=
\lambda^4-\lambda^3-(2z+1)\lambda^2+(2z-z^3)\lambda+2z.
\]
Since \(F_\Delta(2,1)=0\), implicit differentiation of
\[
F_\Delta(\rho_\Delta(\theta),e^\theta)=0
\]
at \(\theta=0\) gives
\[
F_{\Delta,\lambda}(2,1)=9,\qquad
F_{\Delta,z}(2,1)=-8,
\]
so \(\rho_\Delta'(0)=8/9\). A second differentiation yields
\[
F_{\Delta,\lambda\lambda}(2,1)=30,\qquad
F_{\Delta,\lambda z}(2,1)=-9,\qquad
F_{\Delta,zz}(2,1)=-12,
\]
hence \(\rho_\Delta''(0)=332/243\). Therefore
\[
\Lambda_\Delta(\theta):=\log \rho_\Delta(\theta)-\log 2
\]
satisfies
\[
\Lambda_\Delta'(0)=\frac49,
\qquad
\Lambda_\Delta''(0)=\frac{118}{243}.
\]
These are the mean discrepancy density and asymptotic variance used in Corollary~\ref{thm:llnclt}.

For the output-weight observable \(Y_t\), the labels with \(Y_t=1\) are \(01\) and \(11\). Weighting each such edge by \(e^\theta\) gives
\[
M(e^\theta)=
\begin{pmatrix}
1& e^\theta &0& e^\theta\\
0&0&1&0\\
1&1+e^\theta&0&0\\
1&0&0&0
\end{pmatrix},
\]
the weighted matrix used in Section~\ref{sec:partition}. Its characteristic polynomial is
\[
F_Y(\lambda,z)
:=
\det(\lambda I-M(z))
=
\lambda^4-\lambda^3-(2z+1)\lambda^2+\lambda+z(z+1).
\]
Since \(F_Y(2,1)=0\), implicit differentiation of \(F_Y(\rho_Y(\theta),e^\theta)=0\) gives
\[
F_{Y,\lambda}(2,1)=9,\qquad
F_{Y,z}(2,1)=-5,
\]
so \(\rho_Y'(0)=5/9\). Differentiating again and using
\[
F_{Y,\lambda\lambda}(2,1)=30,\qquad
F_{Y,\lambda z}(2,1)=-8,\qquad
F_{Y,zz}(2,1)=2,
\]
gives \(\rho_Y''(0)=71/243\). Hence
\[
\Lambda_Y(\theta):=\log \rho_Y(\theta)-\log 2
\]
satisfies
\[
\Lambda_Y'(0)=\frac{5}{18},
\qquad
\Lambda_Y''(0)=\frac{67}{972}.
\]

\subsection{Autocovariances and power spectrum}
For the vertex chain with transition matrix \(P\), the discrepancy observable depends only on the current transition:
\[
g(i,j)=
\begin{cases}
1,& (i,j)\in\{(00,01),(01,10),(10,00)\},\\
0,& \text{otherwise}.
\end{cases}
\]
Define
\[
h_j:=\sum_k P_{jk}g(j,k)
=
\begin{pmatrix}
\frac14\\[2mm]
1\\[2mm]
\frac12\\[2mm]
0
\end{pmatrix},
\qquad
q_k:=\sum_j \pi_jP_{jk}g(j,k)
=
\begin{pmatrix}
\frac19 & \frac19 & \frac29 & 0
\end{pmatrix}.
\]
Then
\[
\EE[\Delta_0]=\pi h=\frac49,
\qquad
\EE[\Delta_0\Delta_j]=qP^{j-1}h\qquad (j\ge1).
\]
Because \(P=\tfrac12 D^{-1}AD\) with \(D=\mathrm{diag}(2,1,2,1)\), \(P\) is similar to \(A/2\). Therefore its characteristic polynomial is
\[
(\lambda-1)\left(\lambda-\frac12\right)\left(\lambda+\frac12\right)^2.
\]
The stationary covariance sequence \(c_j=\Cov(\Delta_0,\Delta_j)\) therefore has the form
\[
c_j=2^{-j}\bigl(a+(-1)^j(b+cj)\bigr)\qquad (j\ge1).
\]
Direct evaluation of \(qP^{j-1}h-(4/9)^2\) at \(j=1,2,3\) gives
\[
c_1=\frac{17}{324},\qquad
c_2=\frac{25}{648},\qquad
c_3=\frac{7}{648},
\]
from which
\[
a=\frac18,\qquad
b=\frac{7}{648},\qquad
c=\frac{1}{108}.
\]
Hence
\[
c_j = 2^{-j}\!\left(\frac{1}{8}+(-1)^j\!\left(\frac{7}{648}+\frac{j}{108}\right)\right)\qquad (j\ge1),
\]
while
\[
c_0=\Var(\Delta_0)=\frac49\left(1-\frac49\right)=\frac{20}{81}.
\]
Substituting these values into
\[
S(\omega)=c_0+2\sum_{j\ge1}c_j\cos(j\omega)
\]
and summing the resulting geometric series yields
\[
S(\omega)
=
\frac{2\bigl(32\cos^3\omega+12\cos^2\omega-116\cos\omega-105\bigr)}
{9\,(4\cos\omega-5)\,(4\cos\omega+5)^2}.
\]
At \(\omega=0\) this gives \(S(0)=118/243\), in agreement with the CLT variance.
