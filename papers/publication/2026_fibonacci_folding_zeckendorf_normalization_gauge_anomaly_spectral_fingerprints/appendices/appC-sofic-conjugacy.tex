\section{Appendix C: Sofic graph models and the conjugacy threshold}
\label{app:sofic}

This appendix supplies the detailed fiber-product bookkeeping used in Section~\ref{sec:sequence}. The main theorem there is Theorem~\ref{thm:core-resolution}; here we record the graph models, the explicit \(m=3\) and \(m=4\) ambiguity tables, and the core-reduction lemmas.

\subsection{The labeled graph presentation}
Proposition~\ref{prop:sofic} constructs a labeled directed graph
\[
G_m=(V_m,E_m,\lambda_m),
\qquad
V_m=\{0,1\}^{m-1},
\]
whose edges are the length-\(m\) words \(vb\) (\(v\in\{0,1\}^{m-1}\), \(b\in\{0,1\}\)), with
\[
v\longrightarrow v'=\mathrm{suffix}_{m-1}(vb),
\qquad
\lambda_m(vb)=\Fold_m(vb)\in X_m.
\]
The edge shift of \(G_m\) is canonically conjugate to the full two-sided shift \(\{0,1\}^{\ZZ}\): the underlying microstate sequence is recovered by reading off the appended bit on each edge.

\subsection{Right-resolving property}
\begin{proposition}\label{prop:right-resolving}
For every \(m\ge 1\), the graph \(G_m\) is right-resolving: for each fixed vertex \(v\in V_m\), the two outgoing edges \(v0\) and \(v1\) have distinct labels.
\end{proposition}

\begin{proof}
The two words \(v0\) and \(v1\) have values differing by \(\Fib_{m+1}\). On the other hand, a length-\(m\) word has value at most
\[
\sum_{k=1}^m \Fib_{k+1}=\Fib_{m+3}-2<\Fib_{m+3},
\]
so the full Zeckendorf normalization of any length-\(m\) word can have at most one nonzero digit beyond the first \(m\) coordinates, namely a possible \(1\) in position \(m+1\). Consequently, if two length-\(m\) words have the same fold label, then their values differ by a multiple of \(\Fib_{m+2}\). Since
\[
0<\Fib_{m+1}<\Fib_{m+2},
\]
the words \(v0\) and \(v1\) cannot have the same fold label. Hence \(\Fold_m(v0)\neq \Fold_m(v1)\).
\end{proof}

\subsection{Fiber product, passive coordinates, and core reduction}
For \(m\ge 1\), let \(\widehat{G}_m\) be the off-diagonal fiber product of \(G_m\) with itself: its vertices are ordered pairs \((v,v')\in V_m\times V_m\) with \(v\ne v'\), and there is an edge
\[
(v,v')\xrightarrow{y}(w,w')
\]
whenever \(v\to w\) and \(v'\to w'\) are edges of \(G_m\) carrying the same label \(y\). Thus an ambiguous two-label block is exactly a length-\(2\) path in \(\widehat{G}_m\).

\begin{definition}[Passive coordinate and ambiguous core]
Consider a length-\(2\) path in \(\widehat{G}_m\), equivalently two distinct lifts of the same two-label block in \(G_m\). A coordinate is called \emph{passive} if deleting that coordinate from both lifted paths still yields two distinct lifts of the same two-label block, now in \(G_{m-1}\). Iterating this deletion produces a minimal witness with no passive coordinates; such a witness is called an \emph{ambiguous core}.
\end{definition}

\begin{lemma}[Stripping lemma]\label{lem:stripping}
Every ambiguous two-label block reduces, after finitely many deletions of passive coordinates, to an ambiguous core. Reattaching the deleted coordinates preserves the third-label resolution of the core.
\end{lemma}

\begin{proof}
Each deletion shortens the window size by \(1\), so the process terminates. By construction, the resulting witness has no passive coordinates and is therefore an ambiguous core. Since the deleted coordinates agree in both lifts and contribute the same symbols to every later label, reattaching them does not change the set of admissible third labels and does not alter the decoded appended bit.
\end{proof}

\begin{lemma}[Core length bound]\label{lem:core-length}
Every ambiguous core has length at most \(4\).
\end{lemma}

\begin{proof}
Let the active interval of the core run from its leftmost differing coordinate to its rightmost differing coordinate. Equality of the first label and equality of the second label are both mediated by the local relation \(110\leftrightarrow 001\), whose support has length \(3\). Hence the left boundary of the active interval lies in a rewrite triple for one of the two labels, and the right boundary lies in a rewrite triple for the other label. If the active interval had length at least \(5\), then these two boundary triples would be disjoint and there would be at least one coordinate strictly between them. That coordinate would not participate in either boundary rewrite, so it would be passive, contradicting minimality. Therefore the active interval has length at most \(4\).
\end{proof}

\subsection{The base case \(m=3\): explicit ambiguous cores}
For \(m=3\) the edge-label table is
\[
00\xrightarrow{000}00,\quad 00\xrightarrow{001}01,\quad
01\xrightarrow{010}10,\quad 01\xrightarrow{000}11,
\]
\[
10\xrightarrow{100}00,\quad 10\xrightarrow{101}01,\quad
11\xrightarrow{001}10,\quad 11\xrightarrow{100}11.
\]

\begin{lemma}[The \(m=3\) core table]\label{lem:m3-decoder}
The only ambiguous two-label blocks in \(Y_3\) are
\[
(000,001)
\quad\text{and}\quad
(100,001).
\]
In both cases the competing lifts after the second label terminate at the vertices \(01\) and \(10\), and the third label resolves the ambiguity via
\[
\mathcal{L}^+(01)=\{000,010\},
\qquad
\mathcal{L}^+(10)=\{100,101\}.
\]
\end{lemma}

\begin{proof}
Enumerating the pairs of edges with equal labels in the table above shows that the only off-diagonal length-\(2\) paths in the fiber product are the two displayed pairs. Their terminal vertices are \(01\) and \(10\). Since the successor label sets of those vertices are disjoint, the third label determines which lift continues, and therefore determines the appended bit on the third edge.
\end{proof}

\begin{example}[A concrete \(m=3\) decoding step]\label{ex:m3-decoder}
The two-label block \((000,001)\) is ambiguous:
\[
00\xrightarrow{000}11\xrightarrow{001}10,
\qquad
00\xrightarrow{001}01\xrightarrow{?}\cdots.
\]
If the third label is \(000\) or \(010\), only the path through terminal vertex \(01\) can continue, so the current bit is \(1\). If the third label is \(100\) or \(101\), only the path through terminal vertex \(10\) can continue, so the current bit is \(0\).
\end{example}

\subsection{The base case \(m=4\): the last remaining core ambiguities}
\begin{lemma}[The \(m=4\) core table]\label{lem:m4-decoder}
The only ambiguous two-label blocks in \(Y_4\) are
\[
(0000,0001),\quad (1000,0001),\quad (0010,1001),\quad (0100,1001),
\]
\[
(0001,0000),\quad (0001,0010),\quad (1001,0000),\quad (1001,0010).
\]
In every case the two possible lifts after the second label terminate at the vertices \(001\) and \(110\). The common successor label set is
\[
\{0000,0010\},
\]
and the third label resolves the ambiguity: \(0000\) forces the current bit \(1\), while \(0010\) forces the current bit \(0\).
\end{lemma}

\begin{proof}
The length-\(4\) label table of \(G_4\) has finitely many entries, so the off-diagonal fiber-product paths of length \(2\) can be listed exhaustively. The eight pairs displayed above are exactly those paths. For each pair, the two possible lifts after the second label terminate at \(001\) and \(110\). The outgoing edges from these vertices carry the same two labels \(0000\) and \(0010\), but they append opposite current bits. Hence the third label resolves every \(m=4\) ambiguity.
\end{proof}

\subsection{Proof of the general decoder theorem}
\begin{proof}[Detailed proof of Theorem~\ref{thm:core-resolution}]
By Lemma~\ref{lem:stripping}, every ambiguous two-label block reduces to an ambiguous core. By Lemma~\ref{lem:core-length}, every such core has length \(3\) or \(4\). The complete length-\(3\) and length-\(4\) core lists are given by Lemmas~\ref{lem:m3-decoder} and \ref{lem:m4-decoder}. In every listed case the third label determines the current bit. Reattaching the stripped passive coordinates preserves that resolution by Lemma~\ref{lem:stripping}. Therefore every admissible three-label block in \(Y_m\) determines a unique current bit, which is exactly the statement of Theorem~\ref{thm:core-resolution}.
\end{proof}

\subsection{The degenerate case \texorpdfstring{$m=2$}{m = 2}}
For \(m=2\), the label table is
\[
00\mapsto 00,\qquad
01\mapsto 01,\qquad
10\mapsto 10,\qquad
11\mapsto 00.
\]
Thus the only ambiguity is the collision \(00\leftarrow\{00,11\}\).

\begin{proof}[Proof of Proposition~\ref{thm:m2}]
The only off-diagonal bi-infinite path in \(\widehat G_2\) is the constant label sequence \((00)^\ZZ\), lifted once by \(0^\ZZ\) and once by \(1^\ZZ\). Every other label sequence contains \(01\) or \(10\), and those labels have unique preimages. Hence \((00)^\ZZ\) is the unique non-injective fiber, with exactly the two preimages stated in Proposition~\ref{thm:m2}.
\end{proof}
