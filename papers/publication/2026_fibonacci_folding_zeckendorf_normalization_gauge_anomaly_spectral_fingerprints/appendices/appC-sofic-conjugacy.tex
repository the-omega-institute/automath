\section{Appendix C: Sofic graph models and the conjugacy threshold}
\label{app:sofic}

\subsection{The labeled graph presentation}
Proposition~\ref{prop:sofic} constructs a labeled directed graph
\[
G_m=(V_m,E_m,\lambda_m),
\qquad
V_m=\{0,1\}^{m-1},
\]
whose edges are the length-\(m\) words \(vb\) (\(v\in\{0,1\}^{m-1}\), \(b\in\{0,1\}\)), with
\[
v\longrightarrow v'=\mathrm{suffix}_{m-1}(vb),
\qquad
\lambda_m(vb)=\Fold_m(vb)\in X_m.
\]
The edge shift of \(G_m\) is canonically conjugate to the full two-sided shift \(\{0,1\}^{\ZZ}\): the underlying microstate sequence is recovered by reading off the appended bit on each edge.

\subsection{Right-resolving property}
\begin{proposition}\label{prop:right-resolving}
For every \(m\ge 1\), the graph \(G_m\) is right-resolving: for each fixed vertex \(v\in V_m\), the two outgoing edges \(v0\) and \(v1\) have distinct labels.
\end{proposition}

\begin{proof}
The two words \(v0\) and \(v1\) have Fibonacci values differing by \(\Fib_{m+1}\), because they agree in the first \(m-1\) positions and differ only at the final position. The Zeckendorf expansion of \(\Fib_{m+1}\) has a \(1\) in the \(m\)-th coordinate and zeros below, so a difference of \(\Fib_{m+1}\) cannot disappear after truncation to the first \(m\) digits. Therefore
\[
\Fold_m(v0)\neq \Fold_m(v1),
\]
which is exactly the right-resolving property.
\end{proof}

\subsection{Synchronization for \(m\ge 3\)}
For a label block \(B=(x^{(0)},\dots,x^{(L-1)})\in X_m^L\), let \(\mathcal V(B)\subset V_m\) denote the set of terminal vertices of paths in \(G_m\) carrying the label block \(B\).

\begin{proposition}\label{prop:sync}
If \(m\ge 3\) and \(L=m-1\), then \(\mathcal V(B)\) is a singleton for every admissible label block \(B\in Y_m^{m-1}\).
\end{proposition}

\begin{proof}
Fix an admissible label block \(B=(x^{(0)},\dots,x^{(j-1)})\) of length \(j\), with \(0\le j\le m-1\). A path in \(G_m\) carrying \(B\) is determined by an initial vertex
\[
v^{(0)}=a_1\cdots a_{m-1}\in\{0,1\}^{m-1}
\]
and by the successive appended bits \(b_1,\dots,b_j\), because the \(r\)-th edge has the form
\[
a_{r}\cdots a_{m-1}b_1\cdots b_{r-1}\ \xrightarrow{\ x^{(r-1)}\ }\ a_{r+1}\cdots a_{m-1}b_1\cdots b_r.
\]
By Proposition~\ref{prop:right-resolving}, once the current vertex and the label \(x^{(r-1)}\) are known, the outgoing edge is uniquely determined. Hence two paths carrying the same label block and starting from initial vertices that share the suffix \(a_{j+1}\cdots a_{m-1}\) must also have the same appended bits \(b_1,\dots,b_j\), and therefore the same terminal vertex
\[
v^{(j)}=a_{j+1}\cdots a_{m-1}b_1\cdots b_j.
\]
It follows that the terminal vertex depends on the initial vertex only through the suffix \(a_{j+1}\cdots a_{m-1}\), which has length \(m-1-j\). Consequently the set of possible terminal vertices after reading \(j\) labels has cardinality at most \(2^{m-1-j}\).

Applying this with \(j=m-1\) shows that an admissible block of \(m-1\) labels has at most one terminal vertex. Because \(B\in Y_m^{m-1}\) is admissible, at least one path with that label block exists, so \(\mathcal V(B)\) is in fact a singleton.
\end{proof}

For \(m\ge 3\), define \(v(B)\) to be the unique element of \(\mathcal V(B)\) for a label block \(B\in Y_m^{m-1}\). We then obtain an explicit inverse code
\begin{equation}\label{eq:Psi-def}
(\Psi_m(y))_t
:=
\text{the last bit of }v\bigl(y_t,\dots,y_{t+m-2}\bigr),
\qquad y=(y_t)_{t\in\ZZ}\in Y_m.
\end{equation}
This is a sliding block code of memory \(m-1\).

\begin{proof}[Proof of Theorem~\ref{thm:conjugacy}]
Let \(\eta\in\{0,1\}^{\ZZ}\), and consider the path in \(G_m\) generated by the successive \((m-1)\)-blocks
\[
\eta_{t-m+1}\cdots \eta_{t-1}\in V_m.
\]
Its edge labels are exactly \(\Phi_m(\eta)\). For each \(t\), the block
\[
\bigl(\Phi_m(\eta)_t,\dots,\Phi_m(\eta)_{t+m-2}\bigr)
\]
has terminal vertex \(\eta_t\cdots \eta_{t+m-2}\), so \eqref{eq:Psi-def} gives \((\Psi_m\circ\Phi_m)(\eta)_t=\eta_t\). Thus \(\Psi_m\circ\Phi_m=\mathrm{id}\).

Conversely, let \(y\in Y_m\). By definition of \(Y_m\), some path in \(G_m\) carries the label sequence \(y\). The singleton property of Proposition~\ref{prop:sync} shows that each length-\((m-1)\) label block determines the corresponding terminal vertex uniquely, so the path itself is unique. Reading off the appended bits along that path recovers a microstate sequence \(\eta=\Psi_m(y)\) whose label sequence is \(y\). Hence \(\Phi_m(\Psi_m(y))=y\), so \(\Phi_m\circ \Psi_m=\mathrm{id}_{Y_m}\). Therefore \(\Phi_m\) is a topological conjugacy with inverse \(\Psi_m\).
\end{proof}

\subsection{The degenerate case \(m=2\)}
For \(m=2\), the label table is particularly simple:
\[
00\mapsto 00,\qquad
01\mapsto 01,\qquad
10\mapsto 10,\qquad
11\mapsto 00.
\]
Thus the only ambiguity is the collision \(00\leftarrow\{00,11\}\).

\begin{proof}[Proof of Theorem~\ref{thm:m2}]
If a label sequence in \(Y_2\) contains either \(01\) or \(10\), then the corresponding length-\(2\) microstate block is uniquely determined, and overlapping windows propagate this uniqueness to the entire bi-infinite sequence. Therefore non-injectivity can occur only for the constant label sequence \((00)^\ZZ\).

The two fixed microstate sequences \(0^\ZZ\) and \(1^\ZZ\) both map to \((00)^\ZZ\), because \(\Fold_2(00)=00\) and \(\Fold_2(11)=00\). No other microstate sequence can map to \((00)^\ZZ\): any occurrence of \(01\) or \(10\) in the microstate would produce the label \(01\) or \(10\) at the corresponding site. Hence \((00)^\ZZ\) has exactly the two preimages \(0^\ZZ\) and \(1^\ZZ\), while every other point of \(Y_2\) has a unique preimage.
\end{proof}
