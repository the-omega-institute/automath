\section{Appendix C: Sofic graph models and the conjugacy threshold}
\label{app:sofic}

\subsection{The labeled graph presentation}
Proposition~\ref{prop:sofic} constructs a labeled directed graph
\[
G_m=(V_m,E_m,\lambda_m),
\qquad
V_m=\{0,1\}^{m-1},
\]
whose edges are the length-\(m\) words \(vb\) (\(v\in\{0,1\}^{m-1}\), \(b\in\{0,1\}\)), with
\[
v\longrightarrow v'=\mathrm{suffix}_{m-1}(vb),
\qquad
\lambda_m(vb)=\Fold_m(vb)\in X_m.
\]
The edge shift of \(G_m\) is canonically conjugate to the full two-sided shift \(\{0,1\}^{\ZZ}\): the underlying microstate sequence is recovered by reading off the appended bit on each edge.

\subsection{Right-resolving property}
\begin{proposition}\label{prop:right-resolving}
For every \(m\ge 1\), the graph \(G_m\) is right-resolving: for each fixed vertex \(v\in V_m\), the two outgoing edges \(v0\) and \(v1\) have distinct labels.
\end{proposition}

\begin{proof}
The two words \(v0\) and \(v1\) have values differing by \(\Fib_{m+1}\). On the other hand, a length-\(m\) word has value at most
\[
\sum_{k=1}^m \Fib_{k+1}=\Fib_{m+3}-2<2\Fib_{m+2},
\]
so the full Zeckendorf normalization of any length-\(m\) word can have at most one nonzero digit beyond the first \(m\) coordinates, namely a possible \(1\) in position \(m+1\). Consequently, if two length-\(m\) words have the same fold label, then their values differ by a multiple of \(\Fib_{m+2}\). Since
\[
0<\Fib_{m+1}<\Fib_{m+2},
\]
the words \(v0\) and \(v1\) cannot have the same fold label. Hence \(\Fold_m(v0)\neq \Fold_m(v1)\).
\end{proof}

\subsection{Fiber product and ambiguous cores}
For \(m\ge 1\), let \(\widehat{G}_m\) be the fiber product of \(G_m\) with itself: its vertices are ordered pairs \((v,v')\in V_m\times V_m\) with \(v\ne v'\), and there is a directed edge \((v,v')\xrightarrow{y} (w,w')\) whenever \(v\to w\) and \(v'\to w'\) are edges of \(G_m\) that carry the same label \(y\). Removing the diagonal ensures that edges of \(\widehat{G}_m\) parameterize precisely those label sequences that admit two distinct lifts in \(G_m\). In particular, a two-label block \((y_{t-1},y_t)\) is \emph{ambiguous} if and only if it extends to a path of length \(2\) in \(\widehat{G}_m\).

The normalization rule \(110\to 001\) can only move mass to higher Fibonacci weights. Therefore any disagreement between two lifts must lie in the highest-weight coordinates of the current window. Stripping all trailing coordinates on which the two lifts agree (equivalently, shifting the edge in \(\widehat{G}_m\) as far as possible until the head vertices differ at their lowest index) produces an \emph{ambiguous core}. Every ambiguous two-label block reduces uniquely to such a core, and every core lifts to an actual ambiguity by padding zeros at the higher weights. Consequently the list of ambiguous cores for the smallest window sizes controls the general case.

\subsection{The base case \(m=3\): an explicit decoder}
For \(m=3\) the decoder can be written down completely. The edge-label table is
\[
00\xrightarrow{000}00,\quad 00\xrightarrow{001}01,\quad
01\xrightarrow{010}10,\quad 01\xrightarrow{000}11,
\]
\[
10\xrightarrow{100}00,\quad 10\xrightarrow{101}01,\quad
11\xrightarrow{001}10,\quad 11\xrightarrow{100}11.
\]
If the third edge in a path is
\[
(\eta_{t-2},\eta_{t-1})\xrightarrow{\;y_t\;}(\eta_{t-1},\eta_t),
\]
then its appended bit is exactly the current microstate digit \(\eta_t\).

\begin{lemma}[The \(m=3\) three-block decoder]\label{lem:m3-decoder}
Every admissible label triple \((y_{t-2},y_{t-1},y_t)\in X_3^3\) determines a unique value of \(\eta_t\). Equivalently, there is a well-defined map
\[
\psi_3:\mathcal{B}_3(Y_3)\to\{0,1\},
\qquad
\eta_t=\psi_3(y_{t-2},y_{t-1},y_t),
\]
where \(\mathcal{B}_3(Y_3)\) denotes the admissible three-label blocks in \(Y_3\).
\end{lemma}

\begin{proof}
Most two-label blocks already determine the terminal vertex uniquely. The only ambiguous pairs are
\[
(000,001)
\quad\text{and}\quad
(100,001),
\]
for which the terminal vertex may be \(01\) or \(10\). These two candidates have disjoint successor label sets:
\[
\mathcal{L}^+(01)=\{000,010\},
\qquad
\mathcal{L}^+(10)=\{100,101\}.
\]
Hence the third label resolves the ambiguity. If \(y_t\in\{000,010\}\), the terminal vertex is \(01\) and \(\eta_t=1\); if \(y_t\in\{100,101\}\), the terminal vertex is \(10\) and \(\eta_t=0\). In every other case the terminal vertex, hence its last bit, is already fixed by \((y_{t-2},y_{t-1})\).
\end{proof}

\begin{example}[A concrete \(m=3\) decoding step]\label{ex:m3-decoder}
The two-label block \((000,001)\) is ambiguous:
\[
00\xrightarrow{000}11\xrightarrow{001}10,
\qquad
00\xrightarrow{001}01\xrightarrow{?}\cdots.
\]
If the third label is \(010\), only the path through terminal vertex \(01\) can continue, so \(\eta_t=1\). If the third label is \(100\), only the path through terminal vertex \(10\) can continue, so \(\eta_t=0\).
\end{example}

\subsection{The base case \(m=4\): the last remaining core ambiguity}
The fiber product \(\widehat{G}_4\) contains eight off-diagonal edges, and therefore eight ambiguous two-label blocks. Four of them place the ambiguity at the trailing end of the window:
\[
(0000,0001),\quad (1000,0001),\quad (0010,1001),\quad (0100,1001).
\]
The remaining four place it at the leading end (equivalently, they are the time reversals of the first four, written in the same left-to-right convention used in Section~\ref{sec:sequence}):
\[
(0001,0000),\quad (0001,0010),\quad (1001,0000),\quad (1001,0010).
\]
Direct inspection of \(G_4\) (or of \(\widehat{G}_4\)) shows that, in every case, the two possible lifts after the second label terminate at the vertices \(001\) and \(110\). Thus the appended bit on that edge is \(1\) along the \(001\)-lift and \(0\) along the \(110\)-lift, but the two lifts remain synchronized in the sense that their outgoing label sets agree:
\[
001\xrightarrow{0000}011,\qquad 001\xrightarrow{0010}010,
\]
\[
110\xrightarrow{0000}101,\qquad 110\xrightarrow{0010}100.
\]
Consequently a third label immediately distinguishes the two lifts, regardless of whether the ambiguity sits at the entering or exiting end of the window.

\begin{lemma}[The \(m=4\) three-block decoder]\label{lem:m4-decoder}
Every admissible label triple \((y_{t-2},y_{t-1},y_t)\in X_4^3\) determines a unique value of \(\eta_t\).
\end{lemma}

\begin{proof}
Outside the eight pairs displayed above, the terminal vertex after \((y_{t-2},y_{t-1})\) is already unique. For each displayed pair, the only possible terminal vertices are \(001\) and \(110\), and both have the same outgoing label set \(\{0000,0010\}\). Along the \(001\)-lift the edge labeled \(0000\) appends the bit \(1\) while the edge labeled \(0010\) appends the bit \(0\); the same statement holds for the \(110\)-lift. Hence the third label determines \(\eta_t\) even in the ambiguous cases: observing \(0000\) forces \(\eta_t=1\), while observing \(0010\) forces \(\eta_t=0\).
\end{proof}

\subsection{A uniform local decoder for \(m\ge 3\)}
For general \(m\ge 3\), the same phenomenon persists: two labels may leave a finite ambiguity in the terminal vertex, but three labels determine the appended bit uniquely.

\begin{lemma}[Three labels determine the current bit]\label{lem:three-block}
Fix \(m\ge 3\). Let
\[
e_{t-2},e_{t-1},e_t
\]
be three consecutive edges of \(G_m\) with labels
\[
y_{t-2},y_{t-1},y_t\in X_m.
\]
Then the appended bit on \(e_t\) depends only on the label triple \((y_{t-2},y_{t-1},y_t)\), not on the chosen lift in \(G_m\). Equivalently, there is a well-defined map
\[
\psi_m:\mathcal{B}_3(Y_m)\to\{0,1\},
\qquad
\eta_t=\psi_m(y_{t-2},y_{t-1},y_t).
\]
\end{lemma}

\begin{proof}
Work in the LH convention so that the rewrite rule is \(110\to001\) scanned from high to low weights. As observed above, any ambiguity propagates only toward higher weights. Therefore if two lifts of the same two-label block agree on their lowest-weight coordinates, that common prefix is passive and may be stripped without affecting the remainder of the lift. This stripping process is exactly the projection of a path in \(\widehat{G}_m\) to an ambiguous core as described above.

Lemma~\ref{lem:m3-decoder} shows that the only length-\(3\) cores are the two ambiguous pairs in the \(m=3\) decoder, and Lemma~\ref{lem:m4-decoder} lists all length-\(4\) cores (the eight pairs in the \(m=4\) analysis). No longer core can occur: if an ambiguity survived beyond four sites, the leftmost site of that core would be passive and could be stripped, contradicting minimality. In every core case, the third label separates the two potential lifts because the outgoing labels from the competing vertices have disjoint sets of appended bits: observing \(0000\) forces the bit \(1\), while observing \(0010\) forces the bit \(0\) (and for \(m=3\) the analogous statement holds with labels \(000\) and \(010/100\)). Reattaching the stripped passive prefix copies the third label verbatim, so the separation persists. Hence every admissible three-label block in \(Y_m\) yields a unique appended bit and the decoder \(\psi_m\) is well defined.
\end{proof}

\begin{remark}[Decoder window]
Lemma~\ref{lem:three-block} gives a uniform inverse with memory \(2\) and anticipation \(0\):
\[
(\Psi_m(y))_t:=\psi_m(y_{t-2},y_{t-1},y_t),\qquad y\in Y_m.
\]
The window length does not grow with \(m\); only the finite lookup table \(\psi_m\) changes.
\end{remark}

\begin{proof}[Proof of Theorem~\ref{thm:conjugacy}]
Read \(Y_m\) on the edge shift of \(G_m\). By Lemma~\ref{lem:three-block}, the local rule \(\psi_m\) recovers the appended bit on every edge from the output sequence alone. Hence the entire bi-infinite edge path is recovered uniquely, so the label map from the edge shift of \(G_m\) to \(Y_m\) is injective. Since the edge shift of \(G_m\) is canonically conjugate to \(\{0,1\}^{\ZZ}\) by reading appended bits, this proves that \(\Phi_m\) is bijective onto \(Y_m\).

The inverse is the sliding block code \(\Psi_m\) from the preceding remark. Therefore
\[
\Psi_m\circ \Phi_m=\mathrm{id}_{\{0,1\}^{\ZZ}},
\qquad
\Phi_m\circ \Psi_m=\mathrm{id}_{Y_m},
\]
so \(\Phi_m\) is a topological conjugacy for every \(m\ge 3\).
\end{proof}

\subsection{The degenerate case \texorpdfstring{$m=2$}{m = 2}}
For \(m=2\), the label table is
\[
00\mapsto 00,\qquad
01\mapsto 01,\qquad
10\mapsto 10,\qquad
11\mapsto 00.
\]
Thus the only ambiguity is the collision \(00\leftarrow\{00,11\}\).

\begin{proof}[Proof of Proposition~\ref{thm:m2}]
The only off-diagonal bi-infinite lifts are the two constant microstate sequences \(0^\ZZ\) and \(1^\ZZ\), both mapped to \((00)^\ZZ\). Every other label sequence contains either \(01\) or \(10\), and these labels have unique preimages. Once one such block appears, the overlapping-window structure forces the entire lift uniquely. Hence \((00)^\ZZ\) is the unique non-injective fiber, with exactly the two preimages stated in Proposition~\ref{thm:m2}.
\end{proof}
