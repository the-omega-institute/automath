\section{Introduction}\label{sec:introduction}

\subsection*{Problem}

Let \((X,\mathcal{B},\mu,T)\) be a measure-preserving system and let
\(W\in\mathcal{B}\) be a binary readout set.  The coding map
\(\omega_\infty(x)=(\mathbf{1}_W(T^t x))_{t\geq 0}\) defines a
measurable symbolic factor.  For a visible event \(\mathsf P\), the
depth-\(m\) Bayes error
\[
  \varepsilon_m(\mathsf P;\mu)
\]
is the smallest classification error achievable from the first \(m\)
visible symbols.  This paper studies the sequence
\(m\mapsto \varepsilon_m(\mathsf P;\mu)\) when \(\mathsf P\) is defined
by first entry into a symbolic hole, and shows that it encodes the
escape rate, quasistationary amplitudes, and cyclotomic resonance
structure of the associated open system.

\subsection*{Setting}

Fix a topologically mixing one-step subshift of finite type \(Y\), a
one-step potential \(\varphi\), and the equilibrium Markov measure
\(\nu_\varphi\).  Let \(H\) be a Markov hole with mixing survivor
subshift \(Y_H\), and write
\[
  \gamma_H(\varphi):=P_Y(\varphi)-P_{Y_H}(\varphi)
\]
for the escape rate.  For a non-empty proper subset
\(R\subsetneq \ZZ/q\ZZ\), let \(\mathsf P_R\) be the event that the
first entry time into \(H\) is finite and belongs to \(R\) modulo
\(q\).

\subsection*{Main results}

The paper is organized around one main theorem with subsequent
refinements and consequences.
\begin{enumerate}[label=(\roman*)]
  \item \textbf{Main theorem.}  An exact killed-matrix formula for the
    Bayes error of first-entry residue events of Markov holes,
    identifying its exponential decay with the escape rate and refining
    the leading term by residue-dependent quasistationary ambiguity
    amplitudes
    (Theorems~\ref{thm:main-amplitude}
    and~\ref{thm:first-entry-escape-rate}).
  \item \textbf{Algebraic corollary.}  A resolvent reformulation that
    places the error generating function in the \(q\)-th cyclotomic lift
    of the killed spectrum, with dominant real pole
    \(e^{\gamma_H(\varphi)}>1\)
    (Theorem~\ref{thm:main-resonance}).
  \item \textbf{Secondary consequence.}  A survivor R\'enyi pressure
    formalism for the conditioned survivor law, yielding the R\'enyi-\(2\)
    entropy rate and the resulting Poisson birthday thresholds
    (Theorem~\ref{thm:main-renyi}).
\end{enumerate}
The center of gravity is the escape-rate theorem together with its
quasistationary amplitude refinement in
Section~\ref{sec:open-system}; the resonance and survivor-R\'enyi
statements are recorded as subsequent consequences of the same
killed-operator formalism.

\begin{theorem}[Quasistationary ambiguity amplitudes]
\label{thm:main-amplitude}
Assume the setting above and the residue non-degeneracy hypothesis from
Section~\ref{sec:open-system}.  Let \(B_H\) be the killed transition
matrix on the safe alphabet, and let \(\ell_H\) be a non-negative left
eigenvector of \(B_H\) for \(\rho_H:=\rho(B_H)\), normalized by
\[
  \xi_H:=\frac{\ell_H}{\ell_H^{\mathsf T}\mathbf 1}.
\]
For \(r\in \ZZ/q\ZZ\), let \(b_r\) be the residue ambiguity vector from
Theorem~\ref{thm:first-entry-escape-rate}.  Then for every residue class
\(r\),
\[
  \lim_{n\to\infty}
  \frac{\varepsilon_{nq+r+1}(\mathsf P_R;\nu_\varphi)}
       {\nu_\varphi(\tau_H\ge nq+r+1)}
  = \xi_H^{\mathsf T}b_r.
\]
Equivalently,
\[
  \varepsilon_m(\mathsf P_R;\nu_\varphi)
  = \bigl(\xi_H^{\mathsf T}b_{m-1\bmod q}+o(1)\bigr)\,
    \nu_\varphi(\tau_H\ge m).
\]
In particular, finite observation records not only the escape exponent
\(\gamma_H(\varphi)\) but also a residue-dependent quasistationary
ambiguity amplitude.
\end{theorem}

\noindent
Theorem~\ref{thm:main-renyi} is a thermodynamic consequence of the same
path-sum calculation, now applied to power weights on survivor words.

\begin{theorem}[Survivor R\'enyi pressure spectrum]
\label{thm:main-renyi}
Assume again that \(Y\) is a mixing one-step subshift of finite type,
\(\nu_\varphi\) is the equilibrium Markov measure of a one-step
potential \(\varphi\), and \(H\) is a Markov hole with mixing survivor
subshift \(Y_H\).  For \(m\geq 1\), let \(\mathcal L_m^H\) be the set of
admissible length-\(m\) words that avoid \(H\) for the first \(m\)
steps, and for \(s>0\) define
\[
  \mathcal Z_{H,m}(s)
  := \sum_{a\in \mathcal L_m^H} \nu_\varphi([a])^s.
\]
Then
\[
  \lim_{m\to\infty}
  -\frac1m\log \mathcal Z_{H,m}(s)
  = sP_Y(\varphi)-P_{Y_H}(s\varphi).
\]

If \(\lambda_m^H\) denotes the law of the survivor word conditioned on
survival to depth \(m\),
\[
  \lambda_m^H(a)
  := \frac{\nu_\varphi([a])}{\nu_\varphi(\tau_H\ge m)},
  \qquad a\in\mathcal L_m^H,
\]
then
\[
  \lim_{m\to\infty}
  -\frac1m\log
  \sum_{a\in\mathcal L_m^H} \lambda_m^H(a)^s
  = sP_{Y_H}(\varphi)-P_{Y_H}(s\varphi).
\]

In particular, at \(s=2\) the conditioned survivor law has R\'enyi-\(2\)
entropy rate
\[
  h_{2,H}:=2P_{Y_H}(\varphi)-P_{Y_H}(2\varphi),
\]
while the ambient two-sample exponent for simultaneous survival and
visible collision is
\[
  \chi_H:=2P_Y(\varphi)-P_{Y_H}(2\varphi)
        = 2\gamma_H(\varphi)+h_{2,H}.
\]
\end{theorem}

The identity \(\chi_H = 2\gamma_H(\varphi)+h_{2,H}\) places the
collision story inside open-system thermodynamic formalism.  Combined
with a Chen--Stein argument (Section~\ref{sec:double-budget}), it yields
Poisson birthday thresholds for surviving samples on the scale
\(\binom{N_m}{2}e^{-m h_{2,H}}\asymp 1\).

\medskip
\noindent
The next theorem is a supplementary resolvent reformulation of the
preceding asymptotics.

\begin{theorem}[Cyclotomic resonance lift]
\label{thm:main-resonance}
For first-entry residue events of Markov holes, the Bayes error
generating function is a residue-twisted killed resolvent,
\[
  \mathcal E_{\mathsf P_R}(z)
  = z\,\widehat u^{\mathsf T}(I-z\widehat B_H)^{-1}\widehat b,
\]
and every pole belongs to the \(q\)-th cyclotomic lift of the killed
spectrum.  The positive real pole of smallest modulus is
\[
  z_H=\rho_H^{-1}=e^{P_Y(\varphi)-P_{Y_H}(\varphi)}=e^{\gamma_H(\varphi)},
\]
which is the reciprocal escape factor; since
\(\rho_H=e^{-\gamma_H(\varphi)}<1\), one has \(z_H>1\).
\end{theorem}

\begin{remark}[Expected H\"older extension]
After a sufficiently high block recoding, the same resolvent description
is expected to hold for finite unions of cylinders and H\"older
potentials, with poles in the reciprocal \(q\)-th cyclotomic lift of the
open Ruelle spectrum.  We develop the formal transfer-operator setup for
this extension in
Section~\ref{sec:open-system-resonance}
(Theorem~\ref{thm:holder-open-resonance}), but a complete
publication-level proof of the spectral-gap and regularity claims in the
infinite-state H\"older setting would require significant additional
operator-theoretic work and is deferred.
\end{remark}

\subsection*{Context and comparison}

Thermodynamic formalism for symbolic systems and equilibrium states is
classical; we use standard material from
\citet{Bowen2008,Ruelle1978,Walters1975GMeasures,WaltersErgodicTheory}.
The open-system viewpoint---escape through holes, killed transfer
operators, and quasistationary behaviour---goes back to
\citet{PianigianiYorke1979} and was developed for symbolic systems by
\citet{ColletMartinezSchmitt1997}, \citet{DemersYoung2006},
\citet{KellerLiverani2009}, \citet{BunimovichYurchenko2011},
\citet{DemersWrightYoung2012}, and
\citet{FergusonPollicott2012}; see also
\citet{DemersTodd2017} for pressure and equilibrium states in the
presence of holes for multimodal maps and
\citet{BruinDemers2022} for recent extensions to dispersing billiards.
The Poisson approximation input is the
Chen--Stein method of \citet{ArratiaGoldsteinGordon1989}.

\medskip
\noindent
\textbf{Comparison with prior work on subsystems and escape
statistics.}\quad
The pressure-gap formula
\(\gamma_H(\varphi)=P_Y(\varphi)-P_{Y_H}(\varphi)\) for the escape rate
through a Markov hole is standard in the theory of open dynamical
systems.  \citet{ChazottesCoelhoCollet2009} and
\citet{ChazottesCoelhoCollet2008} study pressure-difference asymptotics
for subsystems of finite type, including Poisson-process limits for
periodic orbits in the survivor subshift; related convergence-rate
questions are treated by \citet{FreitasFreitasTodd2015}.  The relation
between hitting statistics and escape statistics is developed further by
\citet{BruinDemersTodd2018}, and local escape-rate asymptotics beyond
the finite-state setting are studied by \citet{HaydnYang2020}.  What is
specific to the present paper is the identification of a new
\emph{observable}---the depth-\(m\) Bayes error of first-entry residue
events---that (i)~recovers the escape rate from finite observation
rather than asymptotic hitting or return statistics, (ii)~refines the
leading asymptotics by explicit residue-dependent quasistationary
amplitudes, and (iii)~packages the killed spectrum into a cyclotomic
resolvent accessible from the error generating function.  The Poisson
collision thresholds recorded in Section~\ref{sec:double-budget} are
classical in method but acquire their dynamical content from the survivor
R\'enyi pressure spectrum, which emerges from the same killed-matrix
calculation.

\subsection*{ETDS focus and hierarchy}

Prepared for \emph{Ergodic Theory and Dynamical Systems}, the manuscript
is organized around one dynamical theorem: the killed-matrix Bayes error
formula that identifies finite-observation exponents with the escape
rate.  The main result is
Theorem~\ref{thm:first-entry-escape-rate}, proved in
Section~\ref{sec:open-system}, which is the center of gravity.  Its
principal refinement is
Theorem~\ref{thm:quasistationary-ambiguity}, which resolves the
quasistationary amplitudes.  The resolvent reformulation
(Theorem~\ref{thm:error-resolvent-cyclotomic}) is an algebraic
corollary.  Section~\ref{sec:double-budget} records secondary
thermodynamic consequences of the same calculation.

\subsection*{Notation summary}

Throughout the paper: \(Y\) is a mixing one-step SFT over alphabet
\(A\); \(\varphi\) is a one-step potential with equilibrium Markov
measure \(\nu_\varphi\), stationary vector \(\pi\), and transition
matrix \(K\); \(H\) is a Markov hole with safe alphabet
\(\mathcal S:=A\setminus A_{\mathrm{hole}}\), survivor alphabet
\(\mathcal S_\infty\subseteq\mathcal S\), and mixing survivor subshift
\(Y_H\); \(B_H\) is the killed transition matrix on \(\mathcal S\) with
Perron eigenvalue \(\rho_H\); \(\gamma_H(\varphi):=P_Y(\varphi)-P_{Y_H}(\varphi)\)
is the escape rate; \(\varepsilon_m(\mathsf P;\mu)\) is the depth-\(m\)
Bayes error of an event \(\mathsf P\).

\subsection*{Organization}

Section~\ref{sec:preliminaries} introduces the visible symbolic factor
and finite observation \(\sigma\)-algebras.
Section~\ref{sec:spg} records visible words and a Sturmian remark.
Section~\ref{sec:recursive-addressing} summarizes the key fact that
recursive visible refinements reduce to base-level observations;
this material is compressed and is used only to justify reductions in
later sections.
Section~\ref{sec:clarity} develops the exact Bayes error formulas and
the weighted boundary transfer theorem.
Section~\ref{sec:open-system} proves
Theorems~\ref{thm:main-amplitude} and~\ref{thm:main-resonance}; this is
the core of the paper.
Section~\ref{sec:double-budget} proves
Theorem~\ref{thm:main-renyi} and derives the collision consequences.
Appendix~\ref{sec:injective-appendix} records a supplementary note on
injective refinements.
Section~\ref{sec:conclusion} collects brief final remarks.
