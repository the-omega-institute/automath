\section{Visible words and a brief Sturmian illustration}\label{sec:spg}

\subsection{Visible words}

\begin{definition}[Visible word map]\label{def:visible-word-map}
For \(m\ge 1\), the depth-\(m\) visible word map associated with
\((T,W)\) is the function
\[
  \omega_m:X\times\NNzero\to\Sigma^m,
  \qquad
  \omega_m(x;t)=\ind_W(T^t x)\cdots \ind_W(T^{t+m-1}x),
\]
for \(t\in\NNzero\).
\end{definition}

For \(a\in\Sigma^m\), the cylinder \(C_a\) is precisely the event that
the visible word equals \(a\) at time \(0\).  Finite words therefore
index both finite observations and the events determined by those
observations.

\subsection{Finite-resolution events}

\begin{proposition}[Clopen decidability]\label{prop:decidable-clopen}
Let $P\subseteq\Sigma^{\NNzero}$ be Borel and let $m\ge 1$. The following are
equivalent.
\begin{enumerate}[label=(\roman*)]
  \item $P\in\mathcal{A}_m$.
  \item There exists a set $A\subseteq\Sigma^m$ such that
  \[
    P=\bigcup_{a\in A}[a].
  \]
  \item $P$ is clopen and is a union of closed balls of radius $2^{-m}$ in the
  prefix ultrametric.
\end{enumerate}
Consequently, an event $\mathsf{P}\in\mathcal{F}_m$ is determined entirely by
its length-$m$ address prefixes.
\end{proposition}

\begin{proof}
The atoms of $\mathcal{A}_m$ are exactly the length-$m$ cylinders $[a]$, so (i)
and (ii) are equivalent.
By Lemma~\ref{lem:prefix-ball-cylinder}, each $[a]$ is a closed ball of radius
$2^{-m}$ and is therefore clopen. Hence (ii) implies (iii).
Conversely, if $P$ is a union of closed balls of radius $2^{-m}$, then each
such ball is a length-$m$ cylinder, again by
Lemma~\ref{lem:prefix-ball-cylinder}. Thus $P$ belongs to $\mathcal{A}_m$.
The final statement follows from Proposition~\ref{prop:symbolic-factor}(c).
\end{proof}

\subsection{A brief Sturmian illustration}

For a \(T\)-invariant probability measure \(\mu\), the information
content of a visible word \(a\in\Sigma^m\) is
\[
  \tau_\mu(a):=-\log\mu(C_a).
\]
The Sturmian coding of an irrational rotation gives a standard model in
which \(\tau_\mu(a)\) is logarithmic in the word length.

\begin{proposition}[Sturmian bounds]\label{prop:sturmian-cylinder-sandwich}
Let $\alpha\in(0,1)$ be irrational, with continued-fraction expansion
$\alpha=[0;a_1,a_2,\ldots]$ and convergent denominators $(q_n)_{n\ge 0}$.
Consider the rotation $R_\alpha(x)=x+\alpha\pmod 1$ on
$\TT=\RR/\ZZ$ with the binary partition
\[
  I_1=[1-\alpha,1),
  \qquad
  I_0=\TT\setminus I_1.
\]
Let $C(w)$ denote the length-$t$ cylinder of a Sturmian word $w$.
If $q_n\le t<q_{n+1}$, then every length-$t$ Sturmian word satisfies
\[
  \frac{1}{q_{n+1}+q_n}
  < \Leb(C(w))
  < \frac{1}{q_n}+\frac{1}{q_{n+1}}
  < \frac{2}{q_n}.
\]
Equivalently,
\[
  -\log\Leb(C(w))=\log t+O\bigl(\log(a_{n+1}+1)\bigr).
\]
If $\alpha$ is of bounded type, then the error term is $O(1)$.  In
particular, for the golden slope $\alpha=\varphi^{-1}$,
\[
  -\log\Leb(C(w))=\log t+O(1),
\]
and at the convergent lengths $t=F_{n+1}$ one has
\[
  -\log\Leb(C(w))=n\log\varphi+O(1).
\]
\end{proposition}

\begin{proof}
The endpoints of length-$t$ cylinders are the points
$\{1-\alpha-k\alpha\}_{k=0}^{t}$ on the circle. Thus each cylinder is one of
the gaps determined by this finite orbit.

For the lower bound, the shortest gap is
$\min_{1\le k\le t}\|k\alpha\|$, where $\|u\|$ denotes distance to the nearest
integer. If $q_n\le t<q_{n+1}$, the best-approximation property of continued
fractions gives
$\min_{1\le k\le t}\|k\alpha\|=\|q_n\alpha\|$ and
$\|q_n\alpha\|>1/(q_{n+1}+q_n)$.

For the upper bound, the three-gap theorem in this setting implies that the
largest gap among the first $q_n$ cutting points has length
$\|q_{n-1}\alpha\|+\|q_n\alpha\|$
\cite{Slater1950DistributionIntegers,Slater1967GapsSteps}. Since adding more
cutting points can only subdivide existing gaps, every length-$t$ cylinder has
measure at most this quantity, which is strictly smaller than
$1/q_n+1/q_{n+1}$.

Finally, the recurrence $q_{n+1}=a_{n+1}q_n+q_{n-1}$ and the condition
$q_n\le t<q_{n+1}$ imply
$\log q_n=\log t+O(\log(a_{n+1}+1))$.
Applying $-\log$ to the cylinder inequalities gives the stated
asymptotic.  The bounded-type claim is immediate when the
continued-fraction coefficients stay bounded.  For
$\alpha=\varphi^{-1}$, all continued-fraction coefficients are equal to
$1$ and the convergent denominators are the Fibonacci numbers
$F_{n+1}$, giving the final two displays.
\end{proof}

The Sturmian example is included only as a contrast.  Its cylinder
geometry is polynomial rather than exponential, so the weighted boundary
transfer theorem from Section~\ref{sec:clarity} does not apply directly;
the later open-system sections treat the complementary Markov regime in
which survivor asymptotics are governed by transfer-operator spectra.
