\section{Supplementary structural background on visible refinements}
\label{sec:recursive-addressing}

This section records, for completeness, that recursive visible
refinements remain inside ordinary factor theory: every recursive step
is a sliding-block factor, visible \(\sigma\)-algebras do not expand,
and no new entropy is created.  The material is supplementary to the
main open-system argument and is included only to justify reductions
from finite recursive codings back to the base visible process in
Sections~\ref{sec:open-system}--\ref{sec:double-budget}.

\subsection{Recursive schemes}

Let \(\mathcal{C}=\{C_1,\ldots,C_r\}\) be a finite family of measurable
sets in \(X\).  Its associated coded process is
\[
  \pi_{\mathcal{C}}:X\to (\{0,1\}^r)^{\NNzero},
  \qquad
  \pi_{\mathcal{C}}(x)_t
  := \bigl(\ind_{C_1}(T^t x),\ldots,\ind_{C_r}(T^t x)\bigr).
\]
We also write
\[
  \mathcal{G}(\mathcal{C})
  := \sigma\bigl(\ind_{C_j}(T^t\cdot):
      1\leq j\leq r,\ t\geq 0\bigr).
\]

\begin{definition}[Recursive addressing scheme]
\label{def:recursive-scheme}
A recursive addressing scheme consists of finite families
\(\mathcal{C}^{(L)}=\{C^{(L)}_1,\ldots,C^{(L)}_{r_L}\}\), \(L\geq 0\),
constructed as follows.
\begin{enumerate}
  \item The initial family \(\mathcal{C}^{(0)}\) is given.
  \item For each \(L\geq 0\) one chooses finitely many cylinder sets
    \(B^{(L+1)}_1,\ldots,B^{(L+1)}_{r_{L+1}}\) in the one-sided shift
    \((\{0,1\}^{r_L})^{\NNzero}\), and sets
    \[
      C^{(L+1)}_j
      := \bigl(\pi_{\mathcal{C}^{(L)}}\bigr)^{-1}
          \bigl(B^{(L+1)}_j\bigr),
      \qquad 1\leq j\leq r_{L+1}.
    \]
\end{enumerate}
We write
\[
  \pi^{(L)}:=\pi_{\mathcal{C}^{(L)}},
  \qquad
  \mathcal{G}^{(L)}:=\mathcal{G}(\mathcal{C}^{(L)}),
  \qquad
  \nu^{(L)} := \mu\circ(\pi^{(L)})^{-1}.
\]
For \(m\geq 1\) we also write
\[
  \mathcal{F}_m^{(L)}
  := \sigma\Bigl(
      (\pi^{(L)})^{-1}([a]):
      a\in(\{0,1\}^{r_L})^m
     \Bigr)
\]
for the depth-\(m\) observation \(\sigma\)-algebra of the level-\(L\)
process.
\end{definition}

\subsection{Sliding-block realization}

\begin{proposition}[Each recursive step is a sliding-block factor]
\label{prop:sliding-block-factor}
Let \(\mathcal{C}^{(L+1)}\) be obtained from \(\mathcal{C}^{(L)}\) as in
Definition~\ref{def:recursive-scheme}.  Then there exists a measurable
sliding-block code
\[
  \Phi_L:(\{0,1\}^{r_L})^{\NNzero}
  \to (\{0,1\}^{r_{L+1}})^{\NNzero}
\]
such that
\[
  \pi^{(L+1)} = \Phi_L\circ \pi^{(L)}
  \qquad \mu\text{-almost everywhere}.
\]
\leanverified{paper_scan_projection_address_sliding_block_factor_seeds}
\end{proposition}

\begin{proof}
For \(y\in (\{0,1\}^{r_L})^{\NNzero}\) and \(t\geq 0\), set
\[
  \Phi_L(y)_t
  := \bigl(
      \mathbf{1}_{B^{(L+1)}_1}(\sigma^t y),\ldots,
      \mathbf{1}_{B^{(L+1)}_{r_{L+1}}}(\sigma^t y)
     \bigr).
\]
Each \(B^{(L+1)}_j\) is a cylinder, so \(\Phi_L\) is a measurable
sliding-block code.  For \(x\in X\) and \(t\geq 0\),
\[
  (\Phi_L(\pi^{(L)}(x)))_t
  = \bigl(
      \mathbf{1}_{B^{(L+1)}_1}(\sigma^t\pi^{(L)}(x)),\ldots,
      \mathbf{1}_{B^{(L+1)}_{r_{L+1}}}(\sigma^t\pi^{(L)}(x))
     \bigr)
  = \pi^{(L+1)}(x)_t,
\]
because \(\pi^{(L)}\circ T=\sigma\circ \pi^{(L)}\) almost everywhere and
\(C_j^{(L+1)}=(\pi^{(L)})^{-1}(B_j^{(L+1)})\).  Thus
\(\pi^{(L+1)}=\Phi_L\circ\pi^{(L)}\) almost everywhere.
\end{proof}

\begin{corollary}[\(\sigma\)-algebra non-expansion]
\label{cor:sigma-nonexpansion}
For every recursive addressing scheme and every \(L\geq 0\),
\[
  \mathcal{G}^{(L+1)} \subseteq \mathcal{G}^{(L)}.
\]
\leanverified{paper_scan_projection_address_sigma_nonexpansion_seeds}
\end{corollary}

\begin{proof}
Each generator \(C^{(L+1)}_j\) is the preimage under \(\pi^{(L)}\) of a
level-\(L\) cylinder, hence belongs to \(\mathcal{G}^{(L)}\).  Since
\(\mathcal{G}^{(L)}\) is generated by the observables
\(\mathbf{1}_{C_i^{(L)}}\circ T^t\) with \(t\geq 0\), it is stable under
pullback by \(T\).  Hence every time translate
\(T^{-t}C^{(L+1)}_j\) also belongs to \(\mathcal{G}^{(L)}\), and
therefore so does the \(\sigma\)-algebra generated by all such
translates.
\end{proof}

\begin{corollary}[Entropy monotonicity]
\label{cor:entropy-monotonicity}
For every recursive addressing scheme and every \(L\geq 0\),
\[
  h_{\nu^{(L+1)}}(\sigma)\leq h_{\nu^{(L)}}(\sigma).
\]
\end{corollary}

\begin{proof}
By Proposition~\ref{prop:sliding-block-factor},
\((\supp\nu^{(L+1)},\sigma,\nu^{(L+1)})\) is a factor of
\((\supp\nu^{(L)},\sigma,\nu^{(L)})\).  Metric entropy does not increase
under measurable factor maps; see \citet[Chapter~4]{WaltersErgodicTheory}.
\end{proof}

\subsection{Finite-depth collapse to level \texorpdfstring{$0$}{0}}

\begin{definition}[Cumulative prefix budget]
\label{def:cumulative-prefix-budget}
For each \(q\geq 1\), let \(s_q\) denote the maximum cylinder length
among the defining cylinders \(B^{(q)}_1,\ldots,B^{(q)}_{r_q}\) used to
construct level \(q\) from level \(q-1\).  Set
\[
  R_0:=1,
  \qquad
  R_L:=1+\sum_{q=1}^{L}(s_q-1)
  \quad(L\geq 1).
\]
\end{definition}

\begin{theorem}[Finite-depth collapse theorem]
\label{thm:finite-depth-collapse}
For every \(L\geq 0\) there exists a sliding-block code
\[
  \Psi_L:(\{0,1\}^{r_0})^{\NNzero}
  \to (\{0,1\}^{r_L})^{\NNzero}
\]
such that
\[
  \pi^{(L)}=\Psi_L\circ\pi^{(0)}
  \qquad \mu\text{-almost everywhere},
\]
and the coordinate \(\Psi_L(y)_t\) depends only on the block
\[
  y_t,y_{t+1},\ldots,y_{t+R_L-1}.
\]
Consequently, for every \(m\geq 1\),
\[
  \mathcal{F}_m^{(L)}\subseteq \mathcal{F}_{m+R_L-1}^{(0)}.
\]

If \(P\in\mathcal{G}^{(L)}\), define the optimal depth-\(m\) decision
error at level \(L\) by
\[
  \varepsilon_m^{(L)}(P;\mu)
  := \inf\set{
      \mu(g\neq\mathbf{1}_P):
      g\in L^\infty(\mathcal{F}_m^{(L)}),\ g(X)\subseteq\{0,1\}
     }.
\]
Then
\[
  \varepsilon_{m+R_L-1}^{(0)}(P;\mu)
  \leq \varepsilon_m^{(L)}(P;\mu)
  \qquad\text{for every } m\geq 1.
\]
\end{theorem}

\begin{proof}
For \(L=0\) take \(\Psi_0=\mathrm{id}\).  For \(L\geq 1\), compose the
sliding-block codes from Proposition~\ref{prop:sliding-block-factor}:
\[
  \Psi_L:=\Phi_{L-1}\circ\cdots\circ\Phi_0,
  \qquad
  \pi^{(L)}=\Psi_L\circ\pi^{(0)}
  \quad \mu\text{-almost everywhere}.
\]
The block-length claim follows by induction on \(L\).  If
\(\Psi_L(y)_t\) depends on \(y_t,\ldots,y_{t+R_L-1}\) and the
time-\(t\) coordinate of \(\Phi_L(z)\) depends on
\(z_t,\ldots,z_{t+s_{L+1}-1}\), then
\(\Psi_{L+1}(y)_t=\Phi_L(\Psi_L(y))_t\) depends on
\[
  y_t,\ldots,y_{t+R_L+s_{L+1}-2},
\]
which has length \(R_{L+1}=R_L+s_{L+1}-1\).

Hence the first \(m\) output coordinates of \(\Psi_L\) are determined by
the first \(m+R_L-1\) input coordinates.  Pulling back \(m\)-cylinders
therefore yields
\(\mathcal{F}_m^{(L)}\subseteq \mathcal{F}_{m+R_L-1}^{(0)}\).  The
classifier inequality is immediate, since every
\(\mathcal{F}_m^{(L)}\)-measurable \(\{0,1\}\)-valued classifier is also
\(\mathcal{F}_{m+R_L-1}^{(0)}\)-measurable.
\end{proof}

\subsection{Address models and inverse limits}

Set
\[
  Y:=\supp\nu^{(0)}\subseteq(\{0,1\}^{r_0})^{\NNzero},
\]
equipped with the prefix ultrametric inherited from the ambient shift.

\begin{definition}[Address model over the visible factor]
\label{def:address-model}
An address model over \(Y\) consists of
\begin{enumerate}[label=(\roman*)]
  \item finite nonempty sets \(\mathcal{A}_b\) for \(b\geq 0\);
  \item surjective bonding maps
    \(\rho_{b+1,b}:\mathcal{A}_{b+1}\twoheadrightarrow\mathcal{A}_b\);
  \item nonempty closed sets \(F_b(a)\subseteq Y\) for \(a\in\mathcal{A}_b\),
\end{enumerate}
such that for every \(b\geq 0\) and \(a'\in\mathcal{A}_{b+1}\),
\[
  F_{b+1}(a')\subseteq F_b(\rho_{b+1,b}(a'))
\]
and, for every \(a\in\mathcal{A}_b\),
\[
  \diam(F_b(a))\leq 2^{-b}.
\]
\end{definition}

\begin{proposition}[Inverse-limit address space]
\label{prop:inverse-limit-space}
Let \((\mathcal{A}_b,\rho_{b+1,b},F_b)\) be an address model over \(Y\).
Define
\[
  \mathcal{A}_\infty
  := \varprojlim_b \mathcal{A}_b
  = \set{(a_b)_{b\geq 0}:
      \rho_{b+1,b}(a_{b+1})=a_b \text{ for all } b\geq 0 }.
\]
Equip \(\mathcal{A}_\infty\) with the inverse-limit ultrametric
\[
  d_\infty(a,a')
  :=
  \begin{cases}
    0, & a=a',\\[1mm]
    2^{-b_\ast(a,a')}, & a\neq a',
  \end{cases}
  \qquad
  b_\ast(a,a')
  := \sup\set{b\geq 0: a_j=a'_j\text{ for }0\leq j\leq b }.
\]
Then \(\mathcal{A}_\infty\) is non-empty, and \(d_\infty\) is an
ultrametric for which \(\mathcal{A}_\infty\) is compact, hence
complete, and totally disconnected.  For each \(b\geq 0\) and each
\(\alpha\in\mathcal{A}_b\), the cylinder
\[
  U_b(\alpha)
  := \set{a=(a_j)_{j\geq 0}\in\mathcal{A}_\infty:a_b=\alpha}
\]
is clopen, and these cylinders form a basis for the topology of
\(\mathcal{A}_\infty\).
\end{proposition}

\begin{proof}
Because each \(\mathcal{A}_b\) is finite and the bonding maps are
surjective, the inverse limit is a nonempty closed subset of the compact
product \(\prod_b\mathcal{A}_b\).  The definition of \(d_\infty\) is the
usual common-prefix ultrametric, so the ultrametric inequality is
immediate.  Closedness in the product topology gives compactness; hence
\(\mathcal{A}_\infty\) is complete.  The sets
\(U_b(\alpha)\) are coordinate-cylinder sets, so they are clopen and
form the standard basis of the inverse-limit topology.  Therefore
\(\mathcal{A}_\infty\) is totally disconnected.
\end{proof}

\begin{theorem}[Stable address determinacy on the visible factor]
\label{thm:inverse-limit-determinacy}
Let \((\mathcal{A}_b,\rho_{b+1,b},F_b)\) be an address model over \(Y\).
Then for every \(a=(a_b)_{b\geq 0}\in\mathcal{A}_\infty\) the
intersection
\[
  \bigcap_{b\geq 0}F_b(a_b)
\]
consists of exactly one point, denoted \(\xi(a)\in Y\).  The resulting
map
\[
  \xi:\mathcal{A}_\infty\to Y
\]
is \(1\)-Lipschitz:
\[
  d_{\mathrm{pre}}(\xi(a),\xi(a'))
  \leq d_\infty(a,a')
  \qquad\text{for all } a,a'\in\mathcal{A}_\infty.
\]
\leanverified{paper_scan_projection_address_inverse_limit_determinacy_seeds}
\end{theorem}

\begin{proof}
For \(a=(a_b)_{b\geq 0}\in\mathcal{A}_\infty\), the sets \(F_b(a_b)\)
are nested, nonempty, and closed in the compact space \(Y\), so their
intersection is nonempty.  The diameter bound
\(\diam(F_b(a_b))\le 2^{-b}\) forces that intersection to be a
singleton, denoted \(\xi(a)\).  If \(a\neq a'\) and \(b_\ast\) is their
last common index, then \(\xi(a)\) and \(\xi(a')\) both lie in
\(F_{b_\ast}(a_{b_\ast})\), hence
\[
  d_{\mathrm{pre}}(\xi(a),\xi(a'))
  \leq \diam(F_{b_\ast}(a_{b_\ast}))
  \leq 2^{-b_\ast}
  = d_\infty(a,a').
\]
This is the required \(1\)-Lipschitz estimate.
\end{proof}

\begin{definition}[Complete address model]
\label{def:complete-address-model}
An address model \((\mathcal{A}_b,\rho_{b+1,b},F_b)\) over \(Y\) is
called complete if, for every \(b\geq 0\), the family
\[
  \set{F_b(a):a\in\mathcal{A}_b}
\]
is a clopen partition of \(Y\), and if for every \(a\in\mathcal{A}_b\),
\[
  F_b(a)
  = \bigsqcup_{\rho_{b+1,b}(a')=a} F_{b+1}(a').
\]
\end{definition}

\begin{theorem}[Complete address reconstruction]
\label{thm:complete-address-reconstruction}
Let \((\mathcal{A}_b,\rho_{b+1,b},F_b)\) be a complete address model
over \(Y\), and let \(\xi:\mathcal{A}_\infty\to Y\) be the map from
Theorem~\ref{thm:inverse-limit-determinacy}.  Then the map
\[
  \alpha:Y\to\mathcal{A}_\infty,
  \qquad
  \alpha(y):=(a_b(y))_{b\geq 0},
\]
where \(a_b(y)\) is the unique element of \(\mathcal{A}_b\) with
\(y\in F_b(a_b(y))\), is well defined and continuous.  Moreover,
\[
  \xi\circ\alpha=\mathrm{id}_Y
  \qquad\text{and}\qquad
  \alpha\circ\xi=\mathrm{id}_{\mathcal{A}_\infty}.
\]
Hence \(\xi\) is a homeomorphism from \(\mathcal{A}_\infty\) onto \(Y\).

Assume in addition that for each \(b\geq 0\) there is a map
\[
  \tau_{b+1,b}:\mathcal{A}_{b+1}\to\mathcal{A}_b
\]
satisfying
\[
  \rho_{b+1,b}\circ\tau_{b+2,b+1}
  = \tau_{b+1,b}\circ\rho_{b+2,b+1}
\]
and
\[
  \sigma\bigl(F_{b+1}(a')\bigr)
  \subseteq F_b\bigl(\tau_{b+1,b}(a')\bigr)
  \qquad\text{for all } a'\in\mathcal{A}_{b+1}.
\]
Then
\[
  S(a)_b:=\tau_{b+1,b}(a_{b+1}),
  \qquad a=(a_b)_{b\geq 0}\in\mathcal{A}_\infty,
\]
defines a continuous map \(S:\mathcal{A}_\infty\to\mathcal{A}_\infty\),
and
\[
  \xi\circ S=\sigma\circ\xi.
\]
\end{theorem}

\begin{proof}
Because the level families form clopen partitions, each \(y\in Y\)
determines a unique compatible sequence \(\alpha(y)\in\mathcal A_\infty\),
and the coordinate maps \(a_b(\cdot)\) are locally constant; hence
\(\alpha\) is continuous.  By construction,
\(y\in\bigcap_b F_b(a_b(y))\), so
Theorem~\ref{thm:inverse-limit-determinacy} gives
\(\xi(\alpha(y))=y\).  Conversely, if \(a\in\mathcal A_\infty\) and
\(y=\xi(a)\), then \(y\in F_b(a_b)\) for every \(b\), so the uniqueness
of the level-\(b\) partition piece yields \(\alpha(y)=a\).  Thus
\(\alpha=\xi^{-1}\), and \(\xi\) is a homeomorphism.

For the dynamical statement, define \(S(a)_b:=\tau_{b+1,b}(a_{b+1})\).
Compatibility with the bonding maps implies \(S(a)\in\mathcal A_\infty\),
and continuity is immediate from the inverse-limit topology.  If
\(y=\xi(a)\), then \(y\in F_{b+1}(a_{b+1})\), so the shift-compatibility
hypothesis gives
\[
  \sigma(y)\in F_b\bigl(\tau_{b+1,b}(a_{b+1})\bigr)=F_b(S(a)_b)
  \qquad\text{for every } b.
\]
Therefore \(\sigma(y)\in\bigcap_b F_b(S(a)_b)\), and
Theorem~\ref{thm:inverse-limit-determinacy} yields
\(\xi(S(a))=\sigma(y)=\sigma(\xi(a))\).
\end{proof}

\begin{remark}
The results of this section show that recursive visible refinement is an
internal factor construction over the original visible process: every
recursive step is a sliding-block factor
(Proposition~\ref{prop:sliding-block-factor}), the visible
\(\sigma\)-algebra does not expand
(Corollary~\ref{cor:sigma-nonexpansion}), and no new entropy is created
(Corollary~\ref{cor:entropy-monotonicity}).  The prefix-site framework
developed here provides the categorical underpinning for the survivor
amplitudes and quasistationary measures analysed in
Sections~\ref{sec:open-system}--\ref{sec:double-budget}: the
finite-depth collapse theorem
(Theorem~\ref{thm:finite-depth-collapse}) guarantees that the Bayes
error inequalities driving the escape-rate and collision-threshold
results (Theorems~\ref{thm:first-entry-escape-rate}
and~\ref{thm:double-budget-poisson}) remain valid when one passes from
a higher-level recursive coding back to the base-level visible process.
\end{remark}
