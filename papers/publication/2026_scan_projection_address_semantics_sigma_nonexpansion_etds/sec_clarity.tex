\section{Finite observation and boundary transfer}\label{sec:clarity}

This section records the finite-observation identities used later, in a
compressed form.  Detailed proofs follow the classical martingale and
weighted-automaton arguments and we cite the relevant references inline.

\begin{definition}[Observation error]\label{def:scan-error}
Let \(\mathsf{P}\in\mathcal{F}_\infty\).  For \(m\geq 1\) define
\[
  \varepsilon_m(\mathsf{P};\mu)
  := \inf\set{\mu(g\neq \mathbf{1}_{\mathsf{P}}):
      g\in L^{\infty}(\mathcal{F}_m),\ g(X)\subseteq\{0,1\}}
\]
as the minimised Bayes classification error from the first \(m\) symbols.
\end{definition}

\begin{proposition}[Exact cylinder formula]\label{prop:scan-error-cylinder}
For every \(\mathsf{P}\in\mathcal{F}_\infty\) and \(m\geq 1\),
\[
  \varepsilon_m(\mathsf{P};\mu)
  = \sum_{a\in\Sigma^m} \min\{\mu(\mathsf{P}\cap C_a),\mu(C_a\setminus\mathsf{P})\}
  = \EE\bigl[\min\{p_m,1-p_m\}\bigr],
\]
where \(p_m:=\mu(\mathsf{P}\mid\mathcal{F}_m)\).
\end{proposition}

\begin{proposition}[Bayes rule and functional stability]
\label{prop:bayes-optimality}
Let \(p_m=\mu(\mathsf{P}\mid\mathcal{F}_m)\) and
\(\mathsf{P}_m^*:=\{x:p_m(x)\geq \tfrac12\}\).  Then
\(\mu(\mathsf{P}\triangle \mathsf{Q})\geq \varepsilon_m(\mathsf{P};\mu)\)
for every \(\mathsf{Q}\in\mathcal{F}_m\), with equality for
\(\mathsf{Q}=\mathsf{P}_m^*\); moreover
\(\bigl|\int \varphi(\mathbf{1}_{\mathsf{P}}-
\mathbf{1}_{\mathsf{P}_m^*})\,\mathrm{d}\mu\bigr|
\leq \Vert\varphi\Vert_\infty\,\varepsilon_m(\mathsf{P};\mu)\).
\end{proposition}

\begin{theorem}[Tanaka identity]\label{thm:tanaka-stokes}
Let \(p_m=\mu(\mathsf{P}\mid\mathcal{F}_m)\) and set
\(\varepsilon_0(\mathsf{P};\mu)=\min\{\mu(\mathsf{P}),1-\mu(\mathsf{P})\}\).
Define
\[
  L_m^{1/2}
  := \sum_{k=0}^{m-1}
      \Bigl(\lvert p_{k+1}-\tfrac12\rvert-\lvert p_k-\tfrac12\rvert
        - \operatorname{sgn}(p_k-\tfrac12)(p_{k+1}-p_k)\Bigr).
\]
Then \(L_m^{1/2}\geq 0\), the sequence is non-decreasing, and
\(\varepsilon_m(\mathsf{P};\mu)=\varepsilon_0(\mathsf{P};\mu)-
\EE[L_m^{1/2}]\).
If \(\mathbf{1}_{\mathsf{P}}\) is \(\mathcal{F}_\infty\)-measurable
then \(\EE[L_\infty^{1/2}]=\varepsilon_0(\mathsf{P};\mu)\).
\end{theorem}

\begin{proof}[Sketch]
Apply the discrete Tanaka formula to the martingale \((p_m)\) with the
threshold \(\tfrac12\); see \citet[Section~IV.4]{WaltersErgodicTheory}.
\end{proof}

\begin{corollary}\label{cor:clarity-monotone}
The sequence \(m\mapsto \varepsilon_m(\mathsf{P};\mu)\) is
non-increasing, and it vanishes if and only if \(\mathsf{P}\) agrees
almost surely with an \(\mathcal{F}_m\)-measurable event.
\end{corollary}

\subsection{Boundary cylinders and weighted boundary mass}

For \(\mathsf{P}\in\mathcal{F}_\infty\), let
\(\partial_m\mathsf{P}\) denote the collection of length-\(m\) words
whose cylinders meet both \(\mathsf{P}\) and its complement, and set
\(Z_m^\mu(\partial\mathsf{P})=
\sum_{a\in\partial_m\mathsf{P}}\mu(C_a)\).

\begin{proposition}[Weighted boundary decomposition]
\label{prop:weighted-boundary-decomposition}
For \(a\in\partial_m\mathsf{P}\), let
\(\beta_{\mathsf{P}}(a)=\min\{\mu(\mathsf{P}\mid C_a),
1-\mu(\mathsf{P}\mid C_a)\}\).  Then
\(\varepsilon_m(\mathsf{P};\mu)
 = \sum_{a\in\partial_m\mathsf{P}} \mu(C_a)\beta_{\mathsf{P}}(a)\),
so \(\theta Z_m^\mu(\partial\mathsf{P}) \leq
\varepsilon_m(\mathsf{P};\mu) \leq \tfrac12 Z_m^\mu(\partial\mathsf{P})\)
whenever \(\beta_{\mathsf{P}}(a)\geq \theta\).
\end{proposition}

\begin{theorem}[Upper boundary estimate]\label{thm:clarity-boundary-dimension}
If every non-empty \(m\)-cylinder has measure at most
\(K\lambda_{\mathrm{cyl}}^{-m}\) with \(\lambda_{\mathrm{cyl}}>1\),
then
\(\varepsilon_m(\mathsf{P};\mu)\leq 
\tfrac{K}{2}\lambda_{\mathrm{cyl}}^{-m}N_m(\partial\mathsf{P})\).
Hence \(\overline{\dim}_B(\partial\mathsf{P})<1\) implies
\(\varepsilon_m(\mathsf{P};\mu)=
\lambda_{\mathrm{cyl}}^{-(1-d)m+o(m)}\).
\end{theorem}

\subsection{Weighted boundary transfer}

Assume throughout that the visible factor law is a fully supported
stationary Markov measure on a mixing one-step SFT with adjacency matrix
\(M\), stationary vector \(\pi\), and transition matrix \(K\).

\begin{theorem}[Weighted boundary transfer]
\label{thm:weighted-boundary-transfer}
Let \(\mathcal{A}_\partial=(Q,q_*,\delta,F)\) be a trim DFA
recognising the boundary language of \(\mathsf{P}\).
Define \(\widehat Q=Q\times\Sigma\) and
\(B_{(q,i),(q',j)}:=K_{ij}\mathbf{1}_{\{M_{ij}=1\}}
\mathbf{1}_{\{q'=\delta(q,j)\}}\),
\(u_{(q,i)}:=\pi_i\mathbf{1}_{\{q=\delta(q_*,i)\}}\), and
\(v_{(q,i)}:=\mathbf{1}_{\{q\in F\}}\).  Then
\(Z_m^\mu(\partial\mathsf{P})=u^{\mathsf T}B^{m-1}v\) and the zeta
function
\(\zeta_{\partial\mathsf{P},\mu}(s)=
\sum_{m\geq 1}Z_m^\mu(\partial\mathsf{P})e^{-sm}\)
coincides with
\(e^{-s}u^{\mathsf T}(I-e^{-s}B)^{-1}v\),
extending meromorphically to
\(\CC\) as a rational function of \(e^{-s}\).  If \(B\) is primitive
with spectral radius \(\rho_\partial<1\) and the boundary weights are
non-degenerate, then
\(Z_m^\mu(\partial\mathsf{P}) = c_\partial\rho_\partial^{m-1}+O(\vartheta^m)\)
and 
\(\lim_{m\to\infty}-\tfrac1m\log\varepsilon_m(\mathsf{P};\mu)
= -\log\rho_\partial\).
\end{theorem}

\begin{proof}[Sketch]
Every boundary word determines a unique path in the product graph, which
produces the weighted path sum; the resolvent expression follows from the
geometric series and the Perron--Frobenius asymptotics, as in
\citet{Sakarovitch2009}.
\end{proof}

\begin{corollary}[Weighted boundary pole law]
\label{cor:weighted-boundary-pole}
Under the hypotheses above, \(\zeta_{\partial\mathsf{P},\mu}\) has a
simple pole at \(s_\partial=\log\rho_\partial\) of residue
\(c_\partial/\rho_\partial\).
\end{corollary}

\begin{corollary}[Pressure gap formula]\label{cor:pressure-gap}
If \(\nu\) is the equilibrium state of a one-step potential \(\varphi\)
with transfer matrix \(L_\varphi\) and Perron eigenvector \(h\), set
\(\widetilde B_\varphi:=M_{ij}e^{\varphi(i,j)}\mathbf{1}_{\{q'=\delta(q,j)\}}\).
Then \(B=e^{-P_Y(\varphi)}D^{-1}\widetilde B_\varphi D\) with
\(D_{(q,i),(q,i)}=h_i\), so
\(\rho_\partial=e^{P_\partial(\varphi)-P_Y(\varphi)}\) and, whenever
\(B\) is primitive with non-degenerate boundary weights,
\(\lim_{m\to\infty}-\tfrac1m\log\varepsilon_m(\mathsf{P};\mu)
= P_Y(\varphi)-P_\partial(\varphi)\).
\end{corollary}

\begin{remark}
The pressure-gap identity extends to H\"older potentials after a block
presentation, via the usual Ruelle--Perron--Frobenius theory
\cite{Bowen2008,Ruelle1978,Walters1975GMeasures}.
\end{remark}
