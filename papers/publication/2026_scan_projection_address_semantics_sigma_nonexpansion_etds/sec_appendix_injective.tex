\appendix
\section{Injective refinements and hidden conditional entropy}
\label{sec:injective-appendix}

This appendix records a supplementary observation: if a visible quotient
collapses the survivor collision spectrum, then any injective refinement
that restores the microscopic R\'enyi profile must pay exactly the hidden
conditional entropy.  All entropies and divergences are taken with
natural logarithms.

\begin{theorem}[Entropy ledger for finite refinements]
\label{thm:entropy-ledger}
Let \(f_m:\Sigma^m\to \mathcal{Y}_m\) be a map into a finite set.  Let \(U_m\) be
any \(\Sigma^m\)-valued random variable with law \(u_m\), and define
\[
  Y_m:=f_m(U_m).
\]
For \(y\in \mathcal{Y}_m\), write
\[
  d_m(y):=\card{f_m^{-1}(y)},
\]
and define the fibre-uniform lift
\[
  \widetilde u_m(\omega)
  := \frac{\PP(Y_m=f_m(\omega))}
           {d_m(f_m(\omega))}.
\]
Then
\[
  H(U_m)
  = H(Y_m)+\EE[\log d_m(Y_m)]-D(u_m\Vert \widetilde u_m),
\]
equivalently,
\[
  H(U_m\mid Y_m)
  = \EE[\log d_m(Y_m)]-D(u_m\Vert \widetilde u_m)
  \leq \EE[\log d_m(Y_m)].
\]
If \(r_m:\Sigma^m\to R_m\) is any map such that
\(\omega\mapsto (f_m(\omega),r_m(\omega))\)
is injective, then
\(H(r_m(U_m)\mid Y_m)=H(U_m\mid Y_m)\leq \log\card{R_m}\).
\end{theorem}

\begin{proof}
Expanding relative entropy and rearranging yields the first identity.
The conditional-entropy formula follows, and its upper bound by
non-negativity of relative entropy.  If the joint map is injective, then
\(U_m\) is determined by \((Y_m,r_m(U_m))\), so
\(H(r_m(U_m)\mid Y_m)=H(U_m\mid Y_m)\).
\end{proof}

\begin{theorem}[Injective refinements restore the microscopic collision profile]
\label{thm:collision-profile-restoration}
Let \(f_m:\Sigma^m\to \mathcal{Y}_m\), \(U_m\), \(Y_m:=f_m(U_m)\) be as
above.  Then \(S_r(Y_m)\geq S_r(U_m)\) for \(r\geq 1\), with equality
for some \(r>1\) iff \(f_m\) is injective on \(\supp u_m\).  If
\(\omega\mapsto(f_m(\omega),r_m(\omega))\) is injective and
\(Z_m:=(Y_m,r_m(U_m))\), then \(S_r(Z_m)=S_r(U_m)\) for every
\(r\geq 1\).  Thus every injective refinement simultaneously restores the
microscopic birthday scale and pays exactly the full hidden conditional
entropy.
\end{theorem}

\begin{proof}
Convexity of \(x\mapsto x^r\) gives the inequality.  Injectivity of the
joint map puts the atoms of \(Z_m\) in bijection with those of \(U_m\)
with the same probability weights, hence \(S_r(Z_m)=S_r(U_m)\).
\end{proof}
