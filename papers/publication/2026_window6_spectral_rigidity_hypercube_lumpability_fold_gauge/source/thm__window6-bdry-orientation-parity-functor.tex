\begin{proposition}[Sign reduction for fold gauges: binary jumps migrate from the center to the abelianization]\label{prop:bdry-fold-gauge-sign-abelianization}
For any finite set \(S\) with \(|S|=d\ge 2\), let
\[
\mathrm{Lab}(S):=\mathrm{Bij}([d],S).
\]
Then \(\mathrm{Lab}(S)\) is a principal \(\mathfrak S_d\)-torsor.
Writing \(\mathrm{sgn}:\mathfrak S_d\to\ZZ_2\) for the sign homomorphism, there is a canonical isomorphism
\[
\mathrm{Or}(S)\ \cong\ \mathrm{Lab}(S)\times_{\mathfrak S_d,\mathrm{sgn}}\ZZ_2.
\]
In particular, when \(d=2\), the group \(\mathfrak S_2\cong\ZZ_2\) is central and the binary jump can be realized as an internal sheet flip.
When \(d\ge 3\), however, \(Z(\mathfrak S_d)=\{e\}\) while \(\mathfrak S_d^{\mathrm{ab}}\cong\ZZ_2\), so the jump survives only through the abelianized sign representation, namely as an orientation torsor rather than a literal sheet label.
\end{proposition}

\begin{proof}
By definition, \(\mathrm{Or}(S)=\mathrm{Bij}([d],S)/\mathfrak A_d\).
On the other hand, the contracted product \(\mathrm{Lab}(S)\times_{\mathfrak S_d,\mathrm{sgn}}\ZZ_2\) identifies two labelings precisely when they differ by an even permutation.
This is exactly the quotient by \(\mathfrak A_d\), proving the canonical isomorphism.
The statements about the center and the abelianization are the standard structure theory of symmetric groups. \qedhere
\end{proof}

\begin{proposition}[Criterion for realizing character-twisted binary labels]\label{prop:bdry-chi-twisted-binary-label-existence}
Let a finite group \(G\) act transitively on a finite set \(S\), and let \(\chi:G\to\ZZ_2\) be a group homomorphism.
Call a map \(p:S\to\ZZ_2\) a \(\chi\)-twisted binary jump label if
\[
p(g\cdot s)=p(s)+\chi(g)\qquad(\mathrm{mod}\ 2)
\]
for every \(g\in G\) and \(s\in S\).
Then such a label exists if and only if, for some (equivalently every) base point \(s_0\in S\),
\[
\chi|_{G_{s_0}}\equiv 0,
\qquad
G_{s_0}:=\{g\in G:\ g\cdot s_0=s_0\}.
\]
Whenever this condition holds, the solution set is a \(\ZZ_2\)-torsor under the global flip \(p\mapsto p+1\).
\end{proposition}

\begin{proof}
Necessity is immediate: if \(p\) is a solution and \(h\in G_{s_0}\), then
\(
p(s_0)=p(h\cdot s_0)=p(s_0)+\chi(h)
\),
so \(\chi(h)=0\).
\par
Conversely, assume \(\chi|_{G_{s_0}}\equiv 0\) and define
\[
p(g\cdot s_0):=\chi(g).
\]
If \(g\cdot s_0=g'\cdot s_0\), then \(g^{-1}g'\in G_{s_0}\), hence \(\chi(g)=\chi(g')\); therefore \(p\) is well defined.
For \(g,g'\in G\),
\[
p(gg'\cdot s_0)=\chi(gg')=\chi(g)+\chi(g')=p(g'\cdot s_0)+\chi(g),
\]
which is exactly the twisted equivariance relation.
Finally, if \(p\) is one solution, then \(p+1\) is another, and the difference of any two solutions is constant by transitivity.
Thus the solution set is a \(\ZZ_2\)-torsor. \qedhere
\end{proof}

\begin{corollary}[Rigidity of sign-twisted binary labels for symmetric groups]\label{cor:bdry-symmetric-group-sign-twisted-label-d2}
Let \(\mathfrak S_d\) act naturally on \([d]=\{1,\dots,d\}\), and take \(\chi=\mathrm{sgn}:\mathfrak S_d\to\ZZ_2\).
Then a \(\mathrm{sgn}\)-twisted binary jump label \(p:[d]\to\ZZ_2\) exists if and only if \(d=2\).
When \(d=2\), this label is unique up to the global flip \(p\mapsto p+1\).
\end{corollary}

\begin{proof}
By Proposition \ref{prop:bdry-chi-twisted-binary-label-existence}, existence is equivalent to requiring that \(\mathrm{sgn}\) vanish on the stabilizer \((\mathfrak S_d)_1\cong \mathfrak S_{d-1}\).
If \(d\ge 3\), then \(\mathfrak S_{d-1}\) contains a transposition and hence an odd permutation, so this is impossible.
If \(d=2\), the stabilizer is trivial and existence follows.
Uniqueness up to global flip is exactly the torsor statement in Proposition \ref{prop:bdry-chi-twisted-binary-label-existence}. \qedhere
\end{proof}

\begin{proposition}[Orbit classification of binary layerings under permutation symmetry]\label{prop:bdry-binary-layering-orbit-classification}
Let \(F\) be a \(d\)-point set.
Define the quotient space of binary layerings by
\[
\mathcal P_d(F):=\{p:F\to\ZZ_2\}\big/\bigl(p\sim p+1\bigr),
\]
and let \(\mathfrak S(F)\cong\mathfrak S_d\) act on \(\mathcal P_d(F)\) by permutation.
Then the orbit space is completely classified by a single integer:
\[
\mathcal P_d(F)\big/\mathfrak S(F)\ \cong\ \{0,1,\dots,\lfloor d/2\rfloor\},
\qquad
k:=\min\bigl(|p^{-1}(0)|,\ |p^{-1}(1)|\bigr).
\]
Moreover, the orbit of parameter \(k\) has size
\[
|\mathcal O_k|=
\begin{cases}
\binom{d}{k}, & k<d/2,\\[4pt]
\binom{d}{d/2}/2, & d\ \text{even and}\ k=d/2.
\end{cases}
\]
\end{proposition}

\begin{proof}
For any \(p\), let \((a,d-a)\) be the sizes of the two level sets.
Modulo the global flip \(p\sim p+1\), this is equivalent to the ordered pair \((\min(a,d-a),\max(a,d-a))\), so \(k=\min(a,d-a)\) is an orbit invariant.
Conversely, given \(k\le d/2\), choose a \(k\)-subset \(A\subset F\) and let \(p=\mathbf 1_A\).
If \(k<d/2\), the orbit corresponds to the collection of all \(k\)-subsets, on which \(\mathfrak S(F)\) acts transitively, hence the orbit size is \(\binom{d}{k}\).
If \(k=d/2\), the subset \(A\) and its complement determine the same element of \(\mathcal P_d(F)\), so one divides by \(2\).
\end{proof}

\begin{proposition}[Determinant realization of the orientation torsor and the jump class]\label{prop:bdry-orientation-torsor-determinant-realization-jumpclass}
Let \(S\) be a finite set with \(|S|=d\ge 1\), and define
\[
V(S):=\RR^{S},
\]
the real vector space with basis indexed by \(S\).
Let \(\mathrm{Or}_{\det}(V(S))\) be the set of orientations of \(V(S)\), viewed as a \(\ZZ_2\)-torsor.
Then there is a canonical natural isomorphism of \(\ZZ_2\)-torsors
\[
\Phi_S:\ \mathrm{Or}(S)\ \xrightarrow{\ \sim\ }\ \mathrm{Or}_{\det}(V(S)).
\]
\par
Now let \(Y\) be connected and let \(\pi:\widetilde Y\to Y\) be a locally finite covering of constant degree \(d\), with fibers \(S_y=\pi^{-1}(y)\).
Then the assignment \(y\mapsto \mathrm{Or}(S_y)\) assembles into a \(\ZZ_2\)-local system on \(Y\), hence defines a cohomology class
\[
j(\pi)\in H^1(Y;\ZZ_2).
\]
If \(\pi_\ast\RR\) is viewed as the rank-\(d\) real local system induced by the fiber permutation action, then
\[
j(\pi)=w_1\bigl(\det(\pi_\ast\RR)\bigr)\in H^1(Y;\ZZ_2),
\]
where \(\det(\pi_\ast\RR)\) is the determinant line local system and \(w_1\) is the first Stiefel--Whitney class.
\end{proposition}

\begin{proof}
Choose a labeling \(b\in\mathrm{Bij}([d],S)\).
Then the ordered tuple \((b(1),\dots,b(d))\) is an ordered basis of \(V(S)\), hence determines an orientation.
Right multiplication of \(b\) by an even permutation preserves the orientation because the corresponding change-of-basis determinant is \(+1\).
Therefore the orientation depends only on the image of \(b\) in \(\mathrm{Bij}([d],S)/\mathfrak A_d\), giving a well-defined map \(\Phi_S\).
This map is bijective and natural in bijections \(S\to S'\), hence a canonical isomorphism.
\par
For the covering \(\pi\), analytic continuation along a loop \(\gamma\) produces a monodromy permutation \(\rho(\gamma)\in\mathfrak S_d\) of the fiber.
By definition, the local system \(\mathrm{Or}(S_y)\) has monodromy \(\mathrm{sgn}(\rho(\gamma))\in\ZZ_2\), which is precisely the class \(j(\pi)\).
On the other hand, the local system \(\pi_\ast\RR\) has monodromy given by the associated permutation matrices, whose determinants are again \(\mathrm{sgn}(\rho(\gamma))\).
Hence the determinant line has the same \(\ZZ_2\)-valued monodromy character, proving that
\(
j(\pi)=w_1(\det(\pi_\ast\RR))
\). \qedhere
\end{proof}

\begin{corollary}[Cohomological classification of the orientation local system and the \texorpdfstring{$\mathfrak A_d$}{Ad}-reduction criterion]\label{cor:bdry-orientation-local-system-w1-alternating-reduction}
Retain the notation of Proposition \ref{prop:bdry-orientation-torsor-determinant-realization-jumpclass}, and let \(\rho:\pi_1(Y,y_0)\to \mathfrak S_d\) be the monodromy representation of the covering \(\pi\).
Then the isomorphism class of the orientation local system \(y\mapsto \mathrm{Or}(S_y)\) is determined uniquely by \(\mathrm{sgn}\circ \rho:\pi_1(Y,y_0)\to\ZZ_2\), hence by the class \(j(\pi)\in H^1(Y;\ZZ_2)\).
In particular, the following are equivalent:
\begin{enumerate}
  \item \(j(\pi)=0\in H^1(Y;\ZZ_2)\);
  \item the local system \(y\mapsto \mathrm{Or}(S_y)\) admits a global section, equivalently the corresponding principal \(\ZZ_2\)-bundle is trivial;
  \item \(\rho(\pi_1(Y,y_0))\subseteq \mathfrak A_d\), i.e. the structure group of the covering reduces from \(\mathfrak S_d\) to the alternating group.
\end{enumerate}
\end{corollary}

\begin{proof}
Proposition \ref{prop:bdry-orientation-torsor-determinant-realization-jumpclass} already shows that the monodromy of the orientation local system is exactly \(\mathrm{sgn}\circ\rho\), hence its cohomology class is \(j(\pi)\).
Therefore \(j(\pi)=0\) if and only if \(\mathrm{sgn}\circ\rho\) is trivial, which is equivalent to \(\rho(\pi_1)\subseteq \ker(\mathrm{sgn})=\mathfrak A_d\).
On the other hand, a principal \(\ZZ_2\)-bundle over a connected base is trivial if and only if it has a global section.
This proves the equivalence of the three conditions. \qedhere
\end{proof}

\begin{proposition}[Multiplicativity of jump classes under fiber products and annihilation by even degree]\label{prop:bdry-orientation-jumpclass-fiberproduct-multiplicativity}
Let \(Y\) be connected, and let \(\pi_i:\widetilde Y_i\to Y\) (\(i=1,2\)) be finite coverings of constant degrees \(d_i\).
Assume the fiber product \(\widetilde Y:=\widetilde Y_1\times_Y\widetilde Y_2\) is again a covering over \(Y\), with induced covering \(\pi:\widetilde Y\to Y\).
Then
\[
j(\pi)\equiv (d_2\bmod 2)\,j(\pi_1)+(d_1\bmod 2)\,j(\pi_2)\qquad\text{in }H^1(Y;\ZZ_2).
\]
In particular, if \(d_1\) is even, then \(j(\pi)=(d_2\bmod 2)\,j(\pi_1)\); if both \(d_1\) and \(d_2\) are even, then \(j(\pi)=0\).
\end{proposition}

\begin{proof}
By Proposition \ref{prop:bdry-orientation-torsor-determinant-realization-jumpclass},
\(
j(\pi)=w_1(\det(\pi_\ast\RR))
\).
For coverings of constant degree there is a natural isomorphism
\[
\pi_\ast\RR\cong (\pi_{1\ast}\RR)\otimes(\pi_{2\ast}\RR).
\]
For finite-dimensional real local systems \(U\) and \(W\), one has
\[
\det(U\otimes W)\cong \det(U)^{\otimes \dim W}\otimes \det(W)^{\otimes \dim U}.
\]
Applying this with \(U=\pi_{1\ast}\RR\) and \(W=\pi_{2\ast}\RR\), and using
\(
w_1(L^{\otimes n})\equiv (n\bmod 2)\,w_1(L)
\)
over \(\ZZ_2\), gives the formula.
The even-degree special cases are immediate. \qedhere
\end{proof}

\begin{theorem}[Parity formula for orientation obstructions of iterated fiber products]\label{thm:bdry-orientation-iterated-fiberproduct-parity}
Let \(Y\) be connected, and let \(\pi_i:\widetilde Y_i\to Y\) (\(1\le i\le r\)) be finite coverings of constant degrees \(d_i\), with orientation obstruction classes
\[
j(\pi_i)\in H^1(Y;\ZZ_2).
\]
Let
\[
\Pi:=\widetilde Y_1\times_Y\widetilde Y_2\times_Y\cdots\times_Y\widetilde Y_r\longrightarrow Y
\]
be the induced iterated fiber-product covering.
Then
\[
j(\Pi)=\sum_{i=1}^r\Bigl(\prod_{\ell\neq i} d_\ell \bmod 2\Bigr)\,j(\pi_i)
\qquad\text{in }H^1(Y;\ZZ_2).
\]
In particular:
\begin{enumerate}
\normalfont
  \item if at least two of the \(d_i\) are even, then \(j(\Pi)=0\);
  \item if all \(d_i\) are odd, then \(j(\Pi)=\sum_{i=1}^r j(\pi_i)\);
  \item if exactly one \(d_k\) is even and all others are odd, then \(j(\Pi)=j(\pi_k)\).
\end{enumerate}
\end{theorem}

\begin{proof}
We proceed by induction on \(r\).
The case \(r=2\) is Proposition \ref{prop:bdry-orientation-jumpclass-fiberproduct-multiplicativity}.
Assume the claim has been proved for \(r-1\), and write
\[
\Pi_{r-1}:=\widetilde Y_1\times_Y\cdots\times_Y\widetilde Y_{r-1}\to Y,
\qquad
D_{r-1}:=\prod_{i=1}^{r-1}d_i.
\]
By the inductive hypothesis,
\[
j(\Pi_{r-1})
=
\sum_{i=1}^{r-1}\Bigl(\prod_{\substack{\ell\neq i\\ \ell\le r-1}} d_\ell \bmod 2\Bigr)\,j(\pi_i).
\]
Now write
\(
\Pi_r=\Pi_{r-1}\times_Y\widetilde Y_r
\)
and apply Proposition \ref{prop:bdry-orientation-jumpclass-fiberproduct-multiplicativity}:
\[
j(\Pi_r)
=(d_r\bmod 2)\,j(\Pi_{r-1})+(D_{r-1}\bmod 2)\,j(\pi_r).
\]
Substituting the expression for \(j(\Pi_{r-1})\) and absorbing the factor \((d_r\bmod 2)\) into each coefficient yields
\[
j(\Pi_r)
=
\sum_{i=1}^r\Bigl(\prod_{\ell\neq i} d_\ell \bmod 2\Bigr)\,j(\pi_i).
\]
The three special cases are immediate parity consequences of this closed formula. \qedhere
\end{proof}

\begin{proposition}[Orientation torsors of Cartesian products and the exponent law]\label{prop:bdry-orientation-cartesian-product-exponent-law}
For any finite sets \(S\) and \(T\), there is a canonical natural isomorphism
\[
\mathrm{Or}(S\times T)\ \cong\ \mathrm{Or}(S)^{\otimes |T|}\ \otimes\ \mathrm{Or}(T)^{\otimes |S|}.
\]
Here \(P^{\otimes k}\) denotes the \(k\)-fold tensor power in the Picard groupoid \(\mathbf{Tors}(\ZZ_2)\), which depends only on \(k\bmod 2\).
\end{proposition}

\begin{proof}
The vector-space identification \(V(S\times T)\cong V(S)\otimes V(T)\) together with the determinant identity
\[
\det(U\otimes W)\cong \det(U)^{\otimes \dim W}\otimes \det(W)^{\otimes \dim U}
\]
gives
\[
\mathrm{Or}_{\det}(V(S\times T))
\cong
\mathrm{Or}_{\det}(V(S))^{\otimes |T|}\otimes \mathrm{Or}_{\det}(V(T))^{\otimes |S|}.
\]
Applying Proposition \ref{prop:bdry-orientation-torsor-determinant-realization-jumpclass} to identify \(\mathrm{Or}\) with \(\mathrm{Or}_{\det}\) gives the stated isomorphism. \qedhere
\end{proof}

\begin{theorem}[Power law and parity collapse for Cartesian powers of orientation torsors]\label{thm:bdry-orientation-cartesian-power-parity-collapse}
Let \(S\) be a finite set of cardinality \(d\ge 1\).
Then for every integer \(r\ge 1\), there is a canonical natural isomorphism
\[
\mathrm{Or}(S^{\times r})\ \cong\ \mathrm{Or}(S)^{\otimes\, r d^{r-1}}.
\]
Consequently,
\[
\mathrm{Or}(S^{\times r})\cong
\begin{cases}
\mathrm{Or}(S), & r=1,\\[2pt]
\mathbf 1, & d\ \text{even and}\ r\ge 2,\\[2pt]
\mathbf 1, & d\ \text{odd and}\ r\ \text{even},\\[2pt]
\mathrm{Or}(S), & d\ \text{odd and}\ r\ \text{odd},
\end{cases}
\]
where \(\mathbf 1\) denotes the unit object in \(\mathbf{Tors}(\ZZ_2)\).
\end{theorem}

\begin{proof}
Let \(a_r\) be the integer such that
\[
\mathrm{Or}(S^{\times r})\cong \mathrm{Or}(S)^{\otimes a_r}.
\]
Clearly \(a_1=1\).
Using Proposition \ref{prop:bdry-orientation-cartesian-product-exponent-law},
\[
\mathrm{Or}(S^{\times(r+1)})
\cong
\mathrm{Or}(S^{\times r})^{\otimes d}\otimes \mathrm{Or}(S)^{\otimes d^r}
\cong
\mathrm{Or}(S)^{\otimes (d a_r+d^r)}.
\]
Hence \(a_r\) satisfies
\[
a_{r+1}=d a_r+d^r,\qquad a_1=1.
\]
The closed-form solution is \(a_r=r d^{r-1}\), proving the first displayed isomorphism.
\par
Tensor powers of torsors depend only on the exponent modulo \(2\).
If \(d\) is even and \(r\ge 2\), then \(r d^{r-1}\) is even, so \(\mathrm{Or}(S^{\times r})\cong \mathbf 1\).
If \(d\) is odd, then \(d^{r-1}\equiv 1\pmod 2\), hence
\(
r d^{r-1}\equiv r\pmod 2
\),
which gives the remaining cases. \qedhere
\end{proof}

\begin{proposition}[The orientation functor is symmetric monoidal with the Koszul sign rule]\label{prop:bdry-orientation-functor-symmetric-monoidal-koszul}
Let \(\mathbf{FinSet}^{\cong}\) be the groupoid of finite sets and bijections, equipped with the monoidal structure given by disjoint union \(\sqcup\).
Let \(\mathbf{Tors}(\ZZ_2)\) be the Picard groupoid of \(\ZZ_2\)-torsors, with tensor product \(\otimes\).
Then the orientation functor \(\mathrm{Or}:\mathbf{FinSet}^{\cong}\to \mathbf{Tors}(\ZZ_2)\) lifts to a symmetric monoidal functor: for any finite sets \(S\) and \(T\), there is a natural isomorphism
\[
\mu_{S,T}:\ \mathrm{Or}(S\sqcup T)\xrightarrow{\ \sim\ }\mathrm{Or}(S)\otimes\mathrm{Or}(T),
\]
and for the standard symmetry isomorphism \(\beta_{S,T}:S\sqcup T\to T\sqcup S\),
\[
\mu_{T,S}\circ \mathrm{Or}(\beta_{S,T})\circ \mu_{S,T}^{-1}
=(-1)^{|S||T|}\cdot\mathrm{id}_{\mathrm{Or}(S)\otimes \mathrm{Or}(T)},
\]
where \((-1)\) denotes the nontrivial automorphism in \(\mathbf{Tors}(\ZZ_2)\).
\end{proposition}

\begin{proof}
Write \(d=|S|\) and \(e=|T|\).
Choose labelings \(b_S:[d]\xrightarrow{\sim}S\) and \(b_T:[e]\xrightarrow{\sim}T\), and concatenate them to obtain a labeling
\(
b_{S\sqcup T}:[d+e]\xrightarrow{\sim}S\sqcup T
\).
This defines \(\mu_{S,T}\).
The symmetry \(\beta_{S,T}\) corresponds on \([d+e]\) to the block permutation that moves the first \(d\) elements past the last \(e\) elements, whose parity is \((-1)^{de}\).
Hence it acts on orientation torsors by the indicated sign.
Naturality and independence of the auxiliary labelings follow immediately from the quotient by even permutations. \qedhere
\end{proof}

\begin{proposition}[A determinant formula for block decompositions and the visibility of odd blocks]\label{prop:bdry-orientation-block-decomposition-odd-visibility}
Suppose a finite set \(S\) is decomposed into pairwise disjoint blocks
\(
S=\bigsqcup_{t\in T}S_t
\),
and define the index set of odd blocks by
\[
T_{\mathrm{odd}}:=\{t\in T:\ |S_t|\equiv 1\ (\mathrm{mod}\ 2)\}\subseteq T.
\]
Then there is a canonical natural isomorphism in \(\mathbf{Tors}(\ZZ_2)\)
\[
\mathrm{Or}(S)\ \cong\ \mathrm{Or}(T_{\mathrm{odd}})\ \otimes\ \bigotimes_{t\in T}\mathrm{Or}(S_t).
\]
Equivalently, the binary jump induced on \(\mathrm{Or}(S)\) by permuting entire blocks is controlled only by the odd blocks: swapping two blocks \(S_{t_1}\) and \(S_{t_2}\) acts nontrivially on \(\mathrm{Or}(S)\) if and only if both \(t_1\) and \(t_2\) lie in \(T_{\mathrm{odd}}\).
\end{proposition}

\begin{proof}
Choose a labeling \(\theta:[|T|]\xrightarrow{\sim}T\) of the index set and, for each \(t\in T\), a labeling \(b_t:[|S_t|]\xrightarrow{\sim}S_t\).
Concatenating the blocks in the order determined by \(\theta\) and using Proposition \ref{prop:bdry-orientation-functor-symmetric-monoidal-koszul} yields an isomorphism
\[
\mathrm{Or}(S)\ \cong\ \bigotimes_{i=1}^{|T|}\mathrm{Or}\!\bigl(S_{\theta(i)}\bigr).
\]
Changing the labeling \(\theta\) reduces to swapping adjacent blocks \(S_{t_1}\) and \(S_{t_2}\).
The corresponding permutation of the underlying ordered basis has sign \((-1)^{|S_{t_1}||S_{t_2}|}\), so it is nontrivial exactly when both blocks have odd cardinality.
\par
On the other hand, the symmetry on \(\mathrm{Or}(T_{\mathrm{odd}})\) records exactly this parity: exchanging two odd singletons contributes a sign \((-1)^{1\cdot 1}=-1\), whereas exchanges involving an even block do not affect \(T_{\mathrm{odd}}\).
Thus the additional factor \(\mathrm{Or}(T_{\mathrm{odd}})\) compensates precisely for the block-order ambiguity, giving a canonical isomorphism
\[
\mathrm{Or}(S)\cong \mathrm{Or}(T_{\mathrm{odd}})\otimes\bigotimes_{t\in T}\mathrm{Or}(S_t).
\]
\end{proof}

\begin{theorem}[Odd-block visibility and the complete character formula for wreath-type jumps]\label{thm:bdry-orientation-wreath-character-odd-visibility}
Let
\[
S=\bigsqcup_{t\in T}S_t,
\qquad
T_{\mathrm{odd}}:=\{t\in T:\ |S_t|\equiv 1\pmod 2\},
\]
and retain the canonical isomorphism
\[
\mathrm{Or}(S)\cong \mathrm{Or}(T_{\mathrm{odd}})\otimes \bigotimes_{t\in T}\mathrm{Or}(S_t)
\]
from Proposition \ref{prop:bdry-orientation-block-decomposition-odd-visibility}.
Let \(g\in \mathrm{Sym}(S)\) preserve the block decomposition, so that it induces a block permutation \(\tau\in \mathrm{Sym}(T)\) and bijections
\[
g_t:\ S_t\xrightarrow{\sim} S_{\tau(t)}
\]
for each \(t\in T\).
Write \(\varepsilon(g_t)\in\{\pm 1\}\) for the sign by which \(g_t\) acts on \(\mathrm{Or}(S_t)\).
Then the character induced by \(g\) on \(\mathrm{Or}(S)\) is
\[
\varepsilon_S(g)
=
\mathrm{sgn}\!\bigl(\tau|_{T_{\mathrm{odd}}}\bigr)\cdot
\prod_{t\in T}\varepsilon(g_t).
\]
In particular, if all blocks have the same cardinality \(d\), so that \(S\cong T\times S_d\) for a model \(d\)-point block \(S_d\), then for every wreath-product element
\[
(\sigma_t)_{t\in T}\rtimes \tau\in \mathrm{Sym}(S_d)\wr \mathrm{Sym}(T)
\]
one has
\[
\varepsilon_S\bigl((\sigma_t)\rtimes \tau\bigr)
=
\mathrm{sgn}(\tau)^d\cdot \prod_{t\in T}\mathrm{sgn}(\sigma_t).
\]
Hence, when \(d\) is even, the top-level permutation \(\tau\) is completely invisible at the level of orientation; when \(d\) is odd, its parity is retained in full.
\end{theorem}

\begin{proof}
By Proposition \ref{prop:bdry-orientation-block-decomposition-odd-visibility}, \(\mathrm{Or}(S)\) splits as the tensor product of the odd-block index torsor and the blockwise orientation torsors.
The block permutation \(\tau\) acts on \(\mathrm{Or}(T_{\mathrm{odd}})\) through the sign of its restriction to \(T_{\mathrm{odd}}\), contributing
\(
\mathrm{sgn}(\tau|_{T_{\mathrm{odd}}})
\).
Each map \(g_t:S_t\to S_{\tau(t)}\) acts on the factor \(\mathrm{Or}(S_t)\) by \(\varepsilon(g_t)\).
Multiplying these contributions yields
\[
\varepsilon_S(g)
=
\mathrm{sgn}\!\bigl(\tau|_{T_{\mathrm{odd}}}\bigr)\cdot
\prod_{t\in T}\varepsilon(g_t).
\]
\par
If all blocks have common size \(d\), then \(T_{\mathrm{odd}}=\varnothing\) when \(d\) is even and \(T_{\mathrm{odd}}=T\) when \(d\) is odd.
Thus the block-permutation contribution is \(1\) in the former case and \(\mathrm{sgn}(\tau)\) in the latter, which is uniformly \(\mathrm{sgn}(\tau)^d\).
The internal contribution is simply \(\prod_t\mathrm{sgn}(\sigma_t)\), proving the wreath-product formula. \qedhere
\end{proof}

\begin{proposition}[Transfer coefficient for the orientation torsor on the configuration cover of \texorpdfstring{$t$}{t}-subsets]\label{prop:bdry-orientation-tsubset-transfer-binomial}
Let \(S\) be a finite set with \(|S|=k\ge 2\), and write
\[
\binom{S}{t}:=\{A\subset S:\ |A|=t\}\qquad(1\le t\le k-1).
\]
Then there is a canonical natural isomorphism
\[
\mathrm{Or}\!\left(\binom{S}{t}\right)\ \cong\ \mathrm{Or}(S)^{\otimes\,\binom{k-2}{t-1}}.
\]
Equivalently, for the natural action of any \(\sigma\in \mathfrak S_k\),
\[
\mathrm{sgn}\bigl(\sigma\curvearrowright \tbinom{S}{t}\bigr)=\mathrm{sgn}(\sigma)^{\binom{k-2}{t-1}}.
\]
\end{proposition}

\begin{proof}
Since \(\mathfrak S_k^{\mathrm{ab}}\cong \ZZ_2\), the sign character of any permutation representation of \(\mathfrak S_k\) is either trivial or a power of \(\mathrm{sgn}\).
It therefore suffices to evaluate the action of a transposition \(\tau\in\mathfrak S_k\).
On \(t\)-subsets, \(\tau\) fixes those subsets containing both exchanged points or neither of them; all other subsets occur in transposed pairs.
The number of subsets containing the first exchanged point but not the second is \(\binom{k-2}{t-1}\), so \(\tau\) acts as exactly \(\binom{k-2}{t-1}\) disjoint transpositions on \(\binom{S}{t}\).
Hence
\[
\mathrm{sgn}(\tau\curvearrowright \binom{S}{t})=(-1)^{\binom{k-2}{t-1}},
\]
which proves the stated exponent.
The corresponding torsor identity follows from the uniqueness statement of Theorem \ref{thm:bdry-z2-jump-functor-uniqueness}. \qedhere
\end{proof}

\begin{corollary}[Lucas parity criterion for preservation or annihilation of the jump under the \texorpdfstring{$t$}{t}-subset operation]\label{cor:bdry-orientation-tsubset-lucas-parity}
Retain the notation of Proposition \ref{prop:bdry-orientation-tsubset-transfer-binomial}.
Then \(\binom{k-2}{t-1}\) is odd if and only if, in binary expansion, each bit of \(t-1\) is bounded above by the corresponding bit of \(k-2\), as given by Lucas' theorem modulo \(2\) \cite{lucas1878theorie}.
Consequently, for any constant-degree covering \(\pi:\widetilde Y\to Y\), if \(\pi^{(t)}\) denotes the induced covering of \(t\)-subset configurations, then
\[
j(\pi^{(t)})\equiv \binom{k-2}{t-1}\,j(\pi)\qquad\text{in }H^1(Y;\ZZ_2).
\]
Thus the induced jump class is annihilated whenever \(\binom{k-2}{t-1}\) is even, and it coincides with the original jump class whenever \(\binom{k-2}{t-1}\) is odd.
\end{corollary}

\begin{proof}
The parity criterion is the standard mod-\(2\) form of Lucas' theorem.
The cohomological statement follows immediately from Proposition \ref{prop:bdry-orientation-tsubset-transfer-binomial} together with Proposition \ref{prop:bdry-orientation-torsor-determinant-realization-jumpclass}: the induced configuration covering has monodromy sign character raised to the power \(\binom{k-2}{t-1}\), so the same coefficient multiplies the class \(j(\pi)\) in \(H^1(Y;\ZZ_2)\). \qedhere
\end{proof}

\begin{proposition}[Minimal equivariant register degree for explicit \texorpdfstring{$t$}{t}-layer selection]\label{prop:bdry-equvariant-register-min-degree-choose}
Fix \(k\ge 3\) and \(1\le t\le k-1\), and let \(C_t:=\binom{[k]}{t}\), viewed as a transitive \(\mathfrak S_k\)-set.
Suppose \(R\) is a finite transitive \(\mathfrak S_k\)-set and that there exists an \(\mathfrak S_k\)-equivariant map \(f:R\to C_t\).
Then
\[
|R|\ \ge\ |C_t|\ =\ \binom{k}{t},
\]
and this lower bound is sharp, attained by taking \(R=C_t\) and \(f=\mathrm{id}\).
Hence any mechanism that realizes an explicit \(t|(k-t)\) grading under permutation naturality requires at least \(\log_2\binom{k}{t}\) bits of auxiliary register capacity.
\end{proposition}

\begin{proof}
By transitivity one may write \(R\cong \mathfrak S_k/H\) and \(C_t\cong \mathfrak S_k/H_t\), where \(H\) is a stabilizer subgroup and \(H_t\cong \mathfrak S_t\times \mathfrak S_{k-t}\).
An equivariant map \(\mathfrak S_k/H\to \mathfrak S_k/H_t\) exists if and only if \(H\) is conjugate into \(H_t\).
Therefore
\[
[\mathfrak S_k:H]\ge [\mathfrak S_k:H_t]=\binom{k}{t},
\]
which is exactly \(|R|\ge \binom{k}{t}\).
Taking \(H=H_t\) shows sharpness, and the bit lower bound is the binary logarithm of this cardinality. \qedhere
\end{proof}
