\begin{theorem}[Existence criterion and exact count for boundary sheet parity in the zero-auxiliary-gating regime]\label{thm:window6-bdry-sheet-parity-nonfunctorial-extension}
Let \(X_m^{\mathrm{bdry}}\subset X_m\) denote the audited boundary sector, and define
\[
\widetilde X_m^{\mathrm{bdry}}
:=
\bigsqcup_{w\in X_m^{\mathrm{bdry}}}F_w,
\qquad
F_w:=
\bigl(\Fold^{\mathrm{bin}}_m\bigr)^{-1}(w),
\qquad
d_w:=|F_w|,
\]
together with the natural projection
\[
\mathrm{pr}_m:\widetilde X_m^{\mathrm{bdry}}\twoheadrightarrow X_m^{\mathrm{bdry}}.
\]
We say that boundary sheet parity extends in the ``zero auxiliary gating'' regime if there exists an involution
\[
\sigma:\widetilde X_m^{\mathrm{bdry}}\longrightarrow \widetilde X_m^{\mathrm{bdry}}
\]
satisfying \(\mathrm{pr}_m\circ \sigma=\mathrm{pr}_m\) and having no fixed points on any fiber \(F_w\).
Let \(N_{\mathrm{sheet}}(m)\) be the total number of such extensions.
Then
\[
N_{\mathrm{sheet}}(m)>0
\quad\Longleftrightarrow\quad
d_w\ \text{is even for every}\ w\in X_m^{\mathrm{bdry}}.
\]
Whenever such extensions exist, one has
\[
N_{\mathrm{sheet}}(m)
=
\prod_{w\in X_m^{\mathrm{bdry}}}(d_w-1)!!,
\]
where, if \(d_w=2n\), then
\[
(d_w-1)!!=(2n-1)!!=\frac{(2n)!}{2^n n!}.
\]
In particular, the audited boundary data give
\[
N_{\mathrm{sheet}}(6)=1,
\qquad
N_{\mathrm{sheet}}(7)=N_{\mathrm{sheet}}(8)=0.
\]
\end{theorem}

\begin{proof}
The condition \(\mathrm{pr}_m\circ \sigma=\mathrm{pr}_m\) is equivalent to a fiberwise decomposition of \(\sigma\) into involutions
\[
\sigma_w:F_w\to F_w
\qquad
\bigl(w\in X_m^{\mathrm{bdry}}\bigr),
\]
and \(\sigma\) is fixed-point-free exactly when each \(\sigma_w\) is fixed-point-free.
Thus the problem reduces completely to the following question: for a finite set \(F\), when does there exist a fixed-point-free involution \(F\to F\), and how many are there?
\par
If \(|F|\) is odd, then every involution on \(F\) decomposes \(F\) into two-cycles and fixed points, hence must have a fixed point.
So no fixed-point-free involution exists.
Conversely, if \(|F|=2n\) is even, partition \(F\) into \(n\) unordered pairs and swap the two elements in each pair; this gives a fixed-point-free involution.
This proves the existence criterion.
\par
When \(|F|=2n\), choosing a fixed-point-free involution is equivalent to partitioning \(F\) into \(n\) unordered pairs.
The number of such partitions is the classical perfect-matching count
\[
\frac{(2n)!}{2^n n!}=(2n-1)!!.
\]
Since distinct boundary fibers are independent, the total number of global extensions is the product of the fiberwise counts, yielding the displayed formula.
\par
Finally, in the audited data for \(m=6\), every boundary fiber has cardinality \(2\), so each factor equals \(1\), giving \(N_{\mathrm{sheet}}(6)=1\).
For \(m=7\) and \(m=8\), every boundary fiber has cardinality \(3\), and therefore no fixed-point-free involution exists on any of them; hence \(N_{\mathrm{sheet}}(7)=N_{\mathrm{sheet}}(8)=0\). \qedhere
\end{proof}

\begin{theorem}[Among the audited dyadic uplifts, boundary central charge appears only in window-\texorpdfstring{$6$}{6}]\label{thm:window6-boundary-central-charge-unique-among-audited-lifts}
For \(m\in\{6,7,8\}\), consider the direct-product decomposition
\[
\mathcal G_m^{\mathrm{bin}}
\cong
\prod_{w\in X_m}\mathfrak S_{d_m^{\mathrm{bin}}(w)},
\]
and let the subgroup supported on the boundary sector be
\[
\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}}
:=
\prod_{w\in X_m^{\mathrm{bdry}}}\mathfrak S_{d_m^{\mathrm{bin}}(w)}.
\]
Then
\[
\boxed{\;
Z(\mathcal G_m^{\mathrm{bin}})
\cap
\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}}
\cong
\begin{cases}
(\ZZ_2)^3, & m=6,\\[1mm]
1, & m=7,8.
\end{cases}
\;}
\]
In other words, among the audited dyadic boundary uplifts, only window \(6\) produces a nontrivial central charge on the boundary sector.
\end{theorem}
\begin{proof}
By the fiberwise direct-product decomposition, \(\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}}\) is a direct factor of \(\mathcal G_m^{\mathrm{bin}}\), and therefore
\[
Z(\mathcal G_m^{\mathrm{bin}})
\cap
\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}}
=
Z(\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}}).
\]
For \(m=6\), the audited boundary sector consists of three fibers, each of size \(2\).
Hence
\[
\mathcal G_{6,\mathrm{bdry}}^{\mathrm{bin}}\cong \mathfrak S_2^{\,3},
\]
and consequently
\[
Z(\mathcal G_{6,\mathrm{bdry}}^{\mathrm{bin}})
\cong
Z(\mathfrak S_2)^3
\cong
(\ZZ_2)^3.
\]
For \(m=7\) and \(m=8\), every boundary fiber has size \(3\), so
\[
\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}}
\cong
\mathfrak S_3^{\,|X_m^{\mathrm{bdry}}|}.
\]
Since \(Z(\mathfrak S_3)=1\), it follows that
\[
Z(\mathcal G_{m,\mathrm{bdry}}^{\mathrm{bin}})=1
\qquad(m=7,8).
\]
Combining these three cases proves the result. \qedhere
\end{proof}
