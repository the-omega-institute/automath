\begin{theorem}[\(\{2,4\}\)-thresholds completely determine low-degree homology: closed formulas for \texorpdfstring{$H_1,H_2,H^2(\cdot;\ZZ_2)$}{H1, H2, H^2}]\label{thm:window6-foldbin-gauge-low-degree-homology}
For each resolution \(m\ge 1\), let
\(
\mathcal{G}^{\mathrm{bin}}_m:=\Aut(\Fold^{\mathrm{bin}}_m)
\)
be the discrete fold-gauge group induced by binary interval folding, and write
\(
d^{\mathrm{bin}}_m(w):=\card{(\Fold^{\mathrm{bin}}_m)^{-1}(w)}
\)
for the fiber multiplicity at \(w\in X_m\).
Define the threshold counts
\[
N_2:=\#\{w\in X_m:\ d^{\mathrm{bin}}_m(w)\ge 2\},
\qquad
N_4:=\#\{w\in X_m:\ d^{\mathrm{bin}}_m(w)\ge 4\}.
\]
Then there are natural isomorphisms
\[
H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\ \cong\ (\ZZ_2)^{N_2},
\qquad
H_2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\ \cong\ (\ZZ_2)^{\,N_4+\binom{N_2}{2}},
\]
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)\ \cong\ (\ZZ_2)^{\,N_4+\binom{N_2+1}{2}}.
\]
Moreover, if
\[
C(t):=\sum_{w\in X_m}\min\{d^{\mathrm{bin}}_m(w),t\}\qquad(t\ge 0)
\]
is the capacity curve, then its jumps at integer points satisfy
\[
N_2=C(2)-C(1),
\qquad
N_4=C(4)-C(3),
\]
so the three groups above are completely recovered from \(C(1),\dots,C(4)\).
\end{theorem}
\begin{proof}
The fiber decomposition gives
\(
\mathcal{G}^{\mathrm{bin}}_m\cong \prod_{w\in X_m}\mathfrak{S}_{d^{\mathrm{bin}}_m(w)}
\).
For \(H_1\), we use \(H_1(G;\ZZ)\cong G^{\mathrm{ab}}\) together with compatibility of abelianization with finite products:
\[
H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)
\cong
\bigoplus_{w\in X_m}\mathfrak{S}_{d^{\mathrm{bin}}_m(w)}^{\mathrm{ab}}.
\]
Since \(\mathfrak{S}_d^{\mathrm{ab}}\cong\ZZ_2\) for \(d\ge 2\) and is trivial for \(d\le 1\), we obtain
\(
H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\cong (\ZZ_2)^{N_2}
\).
\par
For \(H_2\), the K\"unneth decomposition for group homology over \(\ZZ\) yields
\[
H_2\!\Bigl(\prod_{w}G_w;\ZZ\Bigr)
\cong
\Bigl(\bigoplus_{w}H_2(G_w;\ZZ)\Bigr)\ \oplus\
\Bigl(\bigoplus_{w<w'}H_1(G_w;\ZZ)\otimes H_1(G_{w'};\ZZ)\Bigr),
\]
where \(G_w=\mathfrak{S}_{d^{\mathrm{bin}}_m(w)}\).
For symmetric groups \cite{brown1982cohomology,nakaoka1961homology},
\(
H_2(\mathfrak{S}_d;\ZZ)=0
\)
for \(d\le 3\), whereas
\(
H_2(\mathfrak{S}_d;\ZZ)\cong\ZZ_2
\)
for \(d\ge 4\), and
\(
H_1(\mathfrak{S}_d;\ZZ)\cong\ZZ_2
\)
if and only if \(d\ge 2\).
Hence the first direct-sum term contributes \((\ZZ_2)^{N_4}\), and the tensor term contributes one copy of \(\ZZ_2\) for each unordered pair of fibers whose multiplicities satisfy \(d^{\mathrm{bin}}_m(\cdot)\ge 2\), namely \(\binom{N_2}{2}\) copies.
Therefore
\(
H_2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ)\cong (\ZZ_2)^{N_4+\binom{N_2}{2}}
\).
\par
Finally, the universal coefficient theorem gives
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)
\cong
\mathrm{Hom}\!\bigl(H_2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ),\ZZ_2\bigr)\ \oplus\
\mathrm{Ext}\!\bigl(H_1(\mathcal{G}^{\mathrm{bin}}_m;\ZZ),\ZZ_2\bigr).
\]
Because \(H_1\) and \(H_2\) are both \(\ZZ_2\)-vector spaces, each copy of \(\ZZ_2\) contributes one copy of \(\ZZ_2\) to both \(\mathrm{Hom}\) and \(\mathrm{Ext}\).
Hence
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)
\cong
(\ZZ_2)^{\,\dim_{\ZZ_2}H_2+\dim_{\ZZ_2}H_1}
=
(\ZZ_2)^{\,N_4+\binom{N_2}{2}+N_2}
=
(\ZZ_2)^{\,N_4+\binom{N_2+1}{2}}.
\]
\par
For the capacity curve, the identity
\(
\min(d,k)-\min(d,k-1)=\mathbf 1_{\{d\ge k\}}
\)
holds for every integer \(k\ge 1\) and every \(d\in\NN\).
Summing over \(w\in X_m\) gives
\[
C(k)-C(k-1)=\#\{w\in X_m:\ d^{\mathrm{bin}}_m(w)\ge k\}.
\]
Taking \(k=2\) and \(k=4\) yields the formulas for \(N_2\) and \(N_4\). \qedhere
\end{proof}

\begin{remark}[Three natural families of generators in \texorpdfstring{$H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)$}{H^2}]\label{rem:window6-foldbin-gauge-h2-generators}
Under the above decomposition, the second cohomology splits naturally as
\[
H^2(\mathcal{G}^{\mathrm{bin}}_m;\ZZ_2)
\cong
(\ZZ_2)^{N_4}\ \oplus\ (\ZZ_2)^{\binom{N_2}{2}}\ \oplus\ (\ZZ_2)^{N_2}.
\]
These three summands correspond respectively to single-fiber Schur-multiplier classes coming from symmetric-group factors with \(d^{\mathrm{bin}}_m(w)\ge 4\), cross-fiber pairing classes coming from the K\"unneth term \(H_1\otimes H_1\), and Bockstein-type classes arising from
\(
\mathrm{Ext}(H_1,\ZZ_2)
\).
\end{remark}
