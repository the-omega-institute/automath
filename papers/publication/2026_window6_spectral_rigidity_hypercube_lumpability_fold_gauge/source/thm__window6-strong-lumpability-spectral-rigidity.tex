\begin{theorem}[Spectral rigidity of strong lumpability: eigenvalues of an equitable quotient must come from the discrete hypercube spectrum]\label{thm:window6-strong-lumpability-spectral-rigidity}
Let \(Q_m\) be the simple random walk on the \(m\)-dimensional hypercube, with adjacency matrix \(A\) and transition operator \(T=\frac{1}{m}A\).
Given any surjection \(F:\{0,1\}^m\to X\), let \(M\) be the block-indicator matrix determined by the fibers of \(F\), whose columns are the indicator vectors of the fibers.
Assume that the partition is \emph{equitable}, namely that there exists a matrix \(B\) such that
\[
AM=MB.
\]
Then the spectrum of \(B\) is contained in the spectrum of \(A\).
Consequently, every eigenvalue of the induced quotient transition matrix \(P:=\frac{1}{m}B\) belongs to the set
\[
\Bigl\{1-\tfrac{2k}{m}: k=0,1,\dots,m\Bigr\}.
\]
Therefore, if a candidate transition kernel \(P\) obtained from a pushforward construction has an eigenvalue outside this discrete set, then the corresponding coarse-graining cannot arise from a first-order equitable quotient of \(Q_m\) without introducing additional hidden states.

\leanverified{spec\_inclusion\_of\_intertwine}
\leanverified{rescale\_eigenvalue}
\leanverified{paper\_window6\_strong\_lumpability\_spectral\_rigidity}
\end{theorem}

\begin{proof}
Let \(Bv=\lambda v\).
Then
\(
A(Mv)=M(Bv)=\lambda(Mv)
\).
Whenever \(Mv\neq 0\), it follows that \(\lambda\) is an eigenvalue of \(A\), and hence \(\mathrm{spec}(B)\subseteq \mathrm{spec}(A)\).
Since the adjacency spectrum of \(Q_m\) is \(\{m-2k\}_{k=0}^m\), we obtain
\[
\mathrm{spec}(P)=\frac{1}{m}\mathrm{spec}(B)\subseteq \Bigl\{1-\tfrac{2k}{m}\Bigr\}_{k=0}^m.
\]
The final statement is simply the contrapositive of this inclusion. \qedhere
\end{proof}

\begin{corollary}[A spectral falsifier for window-\texorpdfstring{$6$}{6} strong lumpability]\label{cor:window6-strong-lumpability-spectral-falsifier}
For the audited window-\(6\) pushforward kernel \(P_6\), the spectral certificate yields the values
\[
\lambda_2(P_6)\approx 0.4841,
\qquad
\lambda_\star(P_6)\approx 0.6031.
\]
But
\[
\Bigl\{1-\tfrac{2k}{6}: k=0,1,\dots,6\Bigr\}
=\Bigl\{1,\tfrac23,\tfrac13,0,-\tfrac13,-\tfrac23,-1\Bigr\},
\]
and neither of the above values belongs to this set.
Hence the stable-type coarse-graining in window \(6\) cannot be an equitable quotient of \(Q_6\) without auxiliary hidden states.
\end{corollary}
