\begin{theorem}[Chiral compression induced by an exchange-complement involution and dimension-reducing intertwining of the adjacency operator]\label{thm:window6-chiral-compression-hypercube-adjacency}
Let \(m\ge 2\), set \(\Omega_m:=\{0,1\}^m\), and write \(V_m:=\CC^{\Omega_m}\) for the space of complex-valued functions on \(\Omega_m\).
Choose two distinct coordinates \(i\neq j\), and define an affine involution of the hypercube \(Q_m\) by
\[
\sigma_{i,j}(\omega)_i:=1-\omega_j,\qquad
\sigma_{i,j}(\omega)_j:=1-\omega_i,\qquad
\sigma_{i,j}(\omega)_k:=\omega_k\ (k\ne i,j).
\]
Let \((\sigma_{i,j}\cdot f)(\omega):=f(\sigma_{i,j}^{-1}\omega)\) be the induced permutation action and define the \((-1)\)-eigenspace
\[
V_m^-:=\{f\in V_m:\ \sigma_{i,j}\cdot f=-f\}.
\]
Let \(A_m\) be the adjacency operator of \(Q_m\):
\(
(A_m f)(\omega):=\sum_{\omega'\sim\omega}f(\omega')
\).
Then:
\begin{enumerate}
  \item \(\dim_{\CC}V_m^-=2^{m-2}\).
  \item There exists a linear isomorphism
  \(
  \Phi_{i,j}:\CC^{\Omega_{m-2}}\xrightarrow{\sim}V_m^-
  \)
  determined uniquely by \((i,j)\) and the coordinate ordering, such that for every \(u\in\Omega_{m-2}\),
  \[
  \Phi_{i,j}(\mathbf 1_u)=\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)},
  \]
  where \(\omega^{00}(u)\in\Omega_m\) denotes the vertex whose \(i\)-th and \(j\)-th coordinates are \((0,0)\) and whose remaining coordinates are filled by \(u\) in the prescribed order, and \(\omega^{11}(u)\) is defined similarly.
  \item The adjacency operators intertwine on \(V_m^-\):
  \[
  A_m\circ \Phi_{i,j}=\Phi_{i,j}\circ A_{m-2},
  \qquad\text{hence}\qquad
  A_m|_{V_m^-}\ \cong\ A_{m-2}.
  \]
\end{enumerate}
\end{theorem}

\begin{proof}
The orbit sizes of the involution \(\sigma_{i,j}\) on \(\Omega_m\) are either \(1\) or \(2\).
If \((\omega_i,\omega_j)\in\{(0,1),(1,0)\}\), then \(\sigma_{i,j}(\omega)=\omega\).
If \((\omega_i,\omega_j)\in\{(0,0),(1,1)\}\), then \(\sigma_{i,j}\) exchanges the two vertices.
Therefore each \(u\in\Omega_{m-2}\) determines a unique two-cycle \(\{\omega^{00}(u),\omega^{11}(u)\}\), and there are \(2^{m-2}\) such two-cycles in total.
Each of them contributes one antisymmetric direction
\(
\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}
\),
which proves \(\dim V_m^-=2^{m-2}\) and shows that \(\Phi_{i,j}\) is an isomorphism.
\par
To prove the intertwining identity, it suffices to compute on basis vectors.
Fix \(u\in\Omega_{m-2}\) and consider
\(
\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}
\).
Flipping coordinate \(i\) or \(j\) sends \(\omega^{00}(u)\) to \((1,0,u)\) and \((0,1,u)\), while \(\omega^{11}(u)\) is sent to the same two vertices in the opposite order, so these contributions cancel in the difference.
The remaining \(m-2\) coordinate directions simply flip one coordinate of \(u\), sending the pair simultaneously to \(\omega^{00}(u\oplus e_t)\) and \(\omega^{11}(u\oplus e_t)\).
Hence
\[
A_m\bigl(\mathbf 1_{\omega^{00}(u)}-\mathbf 1_{\omega^{11}(u)}\bigr)
=\sum_{t=1}^{m-2}\bigl(\mathbf 1_{\omega^{00}(u\oplus e_t)}-\mathbf 1_{\omega^{11}(u\oplus e_t)}\bigr)
=\Phi_{i,j}(A_{m-2}\mathbf 1_u),
\]
which is the desired identity. \qedhere
\end{proof}

\begin{theorem}[Walsh-spectral chiral decomposition for an affine involution]\label{thm:window6-affine-chiral-walsh-decomposition}
Retain the notation of Theorem \ref{thm:window6-chiral-compression-hypercube-adjacency}, and let
\(
V_m^+:=\{f\in V_m:\ \sigma_{i,j}\cdot f=f\}
\).
For every \(S\subseteq[m]\), define the Walsh character
\[
\chi_S(\omega):=(-1)^{\sum_{k\in S}\omega_k}\qquad(\omega\in\Omega_m).
\]
Let \(R:=[m]\setminus\{i,j\}\).
Then for every \(J\subseteq R\):
\begin{enumerate}
  \item \(\chi_J\in V_m^+\) and \(\chi_{J\cup\{i,j\}}\in V_m^+\);
  \item if
  \[
  \psi_J^+:=\chi_{J\cup\{i\}}-\chi_{J\cup\{j\}},
  \qquad
  \psi_J^-:=\chi_{J\cup\{i\}}+\chi_{J\cup\{j\}},
  \]
  then \(\psi_J^+\in V_m^+\) and \(\psi_J^-\in V_m^-\).
\end{enumerate}
Moreover,
\[
\mathcal B^+:=\Bigl\{\chi_J,\ \chi_{J\cup\{i,j\}},\ \psi_J^+: \ J\subseteq R\Bigr\}
\]
is a Walsh-orthogonal basis of \(V_m^+\), while
\[
\mathcal B^-:=\Bigl\{\psi_J^-: \ J\subseteq R\Bigr\}
\]
is a Walsh-orthogonal basis of \(V_m^-\).
In particular,
\[
\dim V_m^+=3\cdot 2^{m-2},
\qquad
\dim V_m^-=2^{m-2},
\qquad
V_m=V_m^+\oplus V_m^-.
\]
\end{theorem}
\begin{proof}
For every \(J\subseteq R\), the involution \(\sigma_{i,j}\) leaves the coordinates in \(R\) unchanged, while the exchange-complement operation on coordinates \(i\) and \(j\) contributes an even number of sign changes to \(\chi_J\) and \(\chi_{J\cup\{i,j\}}\).
Thus
\(
\chi_J\circ\sigma_{i,j}=\chi_J
\)
and
\(
\chi_{J\cup\{i,j\}}\circ\sigma_{i,j}=\chi_{J\cup\{i,j\}}
\),
so both lie in \(V_m^+\).
Likewise,
\(
\chi_{J\cup\{i\}}\circ\sigma_{i,j}=-\chi_{J\cup\{j\}}
\)
and
\(
\chi_{J\cup\{j\}}\circ\sigma_{i,j}=-\chi_{J\cup\{i\}}
\),
which shows that \(\psi_J^\pm\in V_m^\pm\).
\par
The Walsh characters \(\{\chi_S\}_{S\subseteq[m]}\) form an orthogonal basis of \(\ell^2(\Omega_m)\), and \(\psi_J^\pm\) are linear combinations of characters of the same degree that remain orthogonal for distinct \(J\).
Counting basis vectors gives \(|\mathcal B^-|=2^{m-2}\) and \(|\mathcal B^+|=3\cdot 2^{m-2}\), which matches \(\dim V_m=2^m\).
Therefore these orthogonal sets are bases of \(V_m^\pm\), and \(V_m=V_m^+\oplus V_m^-\). \qedhere
\end{proof}

\begin{theorem}[Closed formulas for chiral heat traces and the \texorpdfstring{$\Phi_3$}{Phi3} factor]\label{thm:window6-chiral-heat-trace-phi3}
On \(\Omega_m\), define the unweighted hypercube Laplacian by
\[
(\Delta f)(\omega):=\sum_{r=1}^{m}\bigl(f(\omega)-f(\omega^{(r)})\bigr),
\]
where \(\omega^{(r)}\) denotes the vertex obtained by flipping the \(r\)-th bit of \(\omega\).
Since \(\sigma_{i,j}\) is an automorphism of \(Q_m\), the operators \(\Delta\) and \(\sigma_{i,j}\) commute, so one may define the chiral heat traces on \(V_m^\pm\) by
\[
Z^\pm(t):=\Tr_{V_m^\pm}\!\bigl(e^{-t\Delta}\bigr)\qquad(t>0).
\]
Then
\[
Z^-(t)=e^{-2t}(1+e^{-2t})^{m-2},
\qquad
Z^+(t)=(1+e^{-2t})^{m-2}\bigl(1+e^{-2t}+e^{-4t}\bigr).
\]
If \(q:=e^{-2t}\), this becomes
\[
Z^-(q)=q(1+q)^{m-2},
\qquad
Z^+(q)=(1+q)^{m-2}(1+q+q^2),
\]
where \(1+q+q^2=\Phi_3(q)\) is the third cyclotomic polynomial.
\end{theorem}
\begin{proof}
The Walsh characters diagonalize \(\Delta\), with
\(
\Delta\chi_S=2|S|\chi_S
\).
By Theorem \ref{thm:window6-affine-chiral-walsh-decomposition}, an orthogonal basis of \(V_m^-\) is given by \(\{\psi_J^-:J\subseteq R\}\), and each \(\psi_J^-\) is a linear combination of Walsh characters of degree \(|J|+1\).
Hence its eigenvalue is \(2(|J|+1)\), and therefore
\[
Z^-(t)=\sum_{J\subseteq R}e^{-2t(|J|+1)}
=e^{-2t}\sum_{J\subseteq R}e^{-2t|J|}
=e^{-2t}(1+e^{-2t})^{m-2}.
\]
Similarly, \(V_m^+\) has orthogonal basis \(\{\chi_J,\psi_J^+,\chi_{J\cup\{i,j\}}:J\subseteq R\}\), with eigenvalues \(2|J|\), \(2(|J|+1)\), and \(2(|J|+2)\), respectively.
Thus
\[
Z^+(t)=\sum_{J\subseteq R}\Bigl(e^{-2t|J|}+e^{-2t(|J|+1)}+e^{-2t(|J|+2)}\Bigr)
=(1+e^{-2t})^{m-2}(1+e^{-2t}+e^{-4t}).
\]
Setting \(q=e^{-2t}\) gives the stated \(q\)-form and the \(\Phi_3\) factor. \qedhere
\end{proof}

\begin{corollary}[The canonical \(16\)-dimensional chiral carrier of window-\(6\) has the exact spectral fingerprint of \(Q_4\)]\label{cor:window6-chiral-sector-q4-spectrum}
Let \(\sigma_{\mathrm{geo}}\) be the audited geometric stabilizer involution of window \(6\), acting on coordinates by exchanging bits \(1\) and \(5\) and complementing both simultaneously, and define
\[
V^-:=\{f\in\CC^{\Omega_6}: \ \sigma_{\mathrm{geo}}\cdot f=-f\}.
\]
If \(A_6\) denotes the adjacency operator of \(Q_6\), then there is an operator isomorphism
\[
A_6|_{V^-}\ \cong\ A_4,
\]
and hence
\[
\Spec(A_6|_{V^-})=\{4,2,0,-2,-4\},
\qquad
\text{with multiplicities }\binom40,\binom41,\binom42,\binom43,\binom44.
\]
\end{corollary}
\begin{proof}
The involution \(\sigma_{\mathrm{geo}}\) is precisely the special case \(\sigma_{1,5}\) of the exchange-complement involution considered above.
Therefore Theorem \ref{thm:window6-chiral-compression-hypercube-adjacency}, applied with \(m=6\) and \((i,j)=(1,5)\), gives \(A_6|_{V^-}\cong A_4\).
The adjacency spectrum of \(Q_4\) is \(\{4-2k:0\le k\le 4\}\), with multiplicities \(\binom{4}{k}\), so the claim follows. \qedhere
\end{proof}

\begin{proposition}[Exterior-algebra model and the adjacency action as a degree operator]\label{prop:window6-chiral-exterior-algebra-degree-operator}
Retain the notation of Corollary \ref{cor:window6-chiral-sector-q4-spectrum}.
Under the identification \(V^-\cong \CC^{\Omega_4}\), there exists a graded isomorphism
\[
\CC^{\Omega_4}\ \cong\ \Lambda^\bullet(\CC^4),
\]
unique up to relabeling of the four residual coordinates, such that on the common eigenspace decomposition one has
\[
A_6|_{V^-}\ \longleftrightarrow\ 4\,\mathrm{Id}-2N,
\qquad
N|_{\Lambda^k(\CC^4)}=k.
\]
\end{proposition}
\begin{proof}
Take the Walsh basis \(\chi_s(u)=(-1)^{s\cdot u}\) on \(\Omega_4\cong(\ZZ/2\ZZ)^4\).
Then \(A_4\chi_s=(4-2|s|)\chi_s\).
Sending \(\chi_s\) to the exterior basis vector \(e_{i_1}\wedge\cdots\wedge e_{i_{|s|}}\), where \(\mathrm{supp}(s)=\{i_1,\dots,i_{|s|}\}\), gives a graded isomorphism under which the eigenvalue dependence on \(|s|\) becomes the action of \(4-2k\) on \(\Lambda^k(\CC^4)\).
Since \(A_6|_{V^-}\cong A_4\), the result follows. \qedhere
\end{proof}

\begin{proposition}[Strict orthogonality of the chiral sector to stable-type observables]\label{prop:window6-chiral-orthogonality-to-type-observables}
Assume that the audited window-\(6\) folding map is invariant under \(\sigma_{\mathrm{geo}}\), i.e.
\[
\Fold_6^{\mathrm{bin}}\circ \sigma_{\mathrm{geo}}=\Fold_6^{\mathrm{bin}}.
\]
Let \(g\in\CC^{\Omega_6}\) be an observable that depends only on the stable type, so that \(g=\bar g\circ \Fold_6^{\mathrm{bin}}\) for some \(\bar g\).
Then \(g\) is invariant under \(\sigma_{\mathrm{geo}}\):
\(
\sigma_{\mathrm{geo}}\cdot g=g
\).
Consequently, for every \(f\in V^-\),
\[
\langle g,f\rangle=0,
\]
where the inner product is the \(\ell^2\) inner product induced by counting measure, equivalently by the uniform probability measure, on \(\Omega_6\).
\end{proposition}
\begin{proof}
Since \(\sigma_{\mathrm{geo}}\) stabilizes \(\Fold_6^{\mathrm{bin}}\), one has \(\Fold_6^{\mathrm{bin}}\circ\sigma_{\mathrm{geo}}=\Fold_6^{\mathrm{bin}}\).
Therefore \(g(\sigma_{\mathrm{geo}}^{-1}\omega)=g(\omega)\), i.e. \(\sigma_{\mathrm{geo}}\cdot g=g\).
On the other hand, every \(f\in V^-\) satisfies \(\sigma_{\mathrm{geo}}\cdot f=-f\).
Using the unitarity of permutation representations with respect to the counting inner product,
\[
\langle g,f\rangle=\langle \sigma_{\mathrm{geo}}\cdot g,\sigma_{\mathrm{geo}}\cdot f\rangle=\langle g,-f\rangle=-\langle g,f\rangle,
\]
and hence \(\langle g,f\rangle=0\). \qedhere
\end{proof}
