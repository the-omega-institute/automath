\section{Conditional shell equations in static KMS response models}\label{sec:kms-shells}

Having established the record-level thresholds, we now fix a stationary static-KMS response model and ask what level-set statements follow. This section is deliberately conditional. The record law supplies the target rates $\log\varphi/\Delta\tau$ and $\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$; a response model supplies the local excitation profile; only their substitution produces a shell equation. No claim is made here that KMS detailed balance alone determines the shell geometry.

Let $N(x)$ be the lapse function of a static spacetime and assume the field is in a global $\beta_\infty$-KMS state for Killing time. Along a static worldline at position $x$, Tolman scaling gives \cite{Tolman1930,TolmanEhrenfest1930}
\begin{equation}\label{eq:tolman-main}
\beta_{\mathrm{loc}}(x)=\beta_\infty N(x).
\end{equation}
This relation fixes the proper-time KMS parameter seen by a static detector. It does not say that every local response observable is a function of $N(x)$ alone; mode density, boundary conditions, detector profile, and greybody transmission can still enter the absolute transition rate.

For a static worldline, the weak-coupling transition rates are Fourier transforms of the pulled-back two-point function. We use this standard UDW response fact explicitly rather than absorbing it into a response ansatz. Let $\Phi_x(\tau)$ denote the field operator smeared with the detector profile and transported along the static orbit through $x$, and set
\[
W_x(s)=\omega\!\left(\Phi_x(\tau)\Phi_x(\tau-s)\right),
\qquad
\widehat W_x(\nu)=\int_{\RR}e^{-i\nu s}W_x(s)\,ds,
\]
with the Fourier transform understood distributionally whenever $W_x$ is not an $L^1$ function.

\begin{assumption}[Stationary response input and KMS strip regularity]\label{ass:long-time-switch-main}
Fix a static worldline through $x$, a detector gap $\Omega>0$, and a smooth nonnegative switching profile $\chi\in C_c^\infty(\mathbb R)$ with $\int_{\RR}\chi=1$. Let $\chi_T(t)=\chi(t/T)$ and let $H_{g\to e}^{(T)}(x,\Omega)$ and $H_{e\to g}^{(T)}(x,\Omega)$ be the corresponding first-order transition probabilities for a detector with fixed spatial profile and bare coupling $\lambda_c$. Assume:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.2em,itemsep=0.25em]
\item The pullback $W_x$ is a tempered distribution on $\RR$, and there is an open interval $I_x\subset\RR$ containing $\pm\Omega$ on which its Fourier transform $\widehat W_x$ is represented by a continuous function.
\item The long-time stationary detector theorem applies to this worldline and switching family: the limits
\[
\Gamma_+(x,\Omega):=\lim_{T\to\infty}\frac{1}{T}H_{g\to e}^{(T)}(x,\Omega),
\qquad
\Gamma_-(x,\Omega):=\lim_{T\to\infty}\frac{1}{T}H_{e\to g}^{(T)}(x,\Omega),
\]
exist and obey the standard smooth-switching stationary response formulas \cite{LoukoSatz2008,FewsterJuarezLouko2016}.
\item If the state is $\beta_\infty$-KMS with respect to static Killing time, then with $\beta_{\mathrm{loc}}(x)=\beta_\infty N(x)$ there exists a function $F_x$ holomorphic on the strip
\[
S_{\beta_{\mathrm{loc}}}^-=\{z\in\mathbb C:-\beta_{\mathrm{loc}}(x)<\Im z<0\}
\]
such that for every closed substrip $-\beta_{\mathrm{loc}}(x)+\delta\le \Im z\le -\delta$ there exist $m_\delta\in\mathbb N$ and $C_\delta>0$ with
\[
|F_x(z)|\le C_\delta(1+|\Re z|)^{m_\delta},
\]
and the boundary values satisfy
\[
\lim_{\epsilon\to0^+}F_x(s-i\epsilon)=W_x(s),\qquad
\lim_{\epsilon\to0^+}F_x(s-i\beta_{\mathrm{loc}}(x)+i\epsilon)=W_x(-s)
\]
in $\mathcal S'(\RR)$.
\end{enumerate}
Define the switched effective coupling\footnote{The original bare coupling is $\lambda_c$; the switching convention contributes the fixed factor $\|\chi\|_2^2:=\int_{\mathbb R}\chi(u)^2\,du>0$.} by
\[
\lambda_{\mathrm{sw}}^2:=\lambda_c^2\,\|\chi\|_2^2
\]
and use $\lambda_{\mathrm{sw}}$ in the long-time rate formulas below.
\end{assumption}

\begin{proposition}[Stationary UDW response and KMS spectral relation]\label{prop:stationary-udw-response-main}
Fix a static worldline through $x$ and a two-level detector with gap $\Omega>0$, coupling scale $\lambda_c$, and fixed spatial profile. Assume \Cref{ass:long-time-switch-main}. Then the stationary weak-coupling long-time excitation and de-excitation rates are
\begin{equation}\label{eq:fourier-rates-main}
\Gamma_+(x,\Omega)=\lambda_{\mathrm{sw}}^2\widehat W_x(\Omega),
\qquad
\Gamma_-(x,\Omega)=\lambda_{\mathrm{sw}}^2\widehat W_x(-\Omega).
\end{equation}
If the state is $\beta_\infty$-KMS with respect to static Killing time, then the static pullback is KMS in proper time with inverse temperature $\beta_\infty N(x)$ and
\begin{equation}\label{eq:kms-spectral-distribution-main}
\widehat W_x(\nu)=e^{-\beta_\infty N(x)\nu}\widehat W_x(-\nu)
\qquad\text{in }\mathcal S'(\RR).
\end{equation}
In particular, because $\widehat W_x$ is continuous at $\nu=\pm\Omega$,
\begin{equation}\label{eq:kms-detailed-balance-main}
\widehat W_x(-\Omega)=e^{\beta_\infty N(x)\Omega}\widehat W_x(\Omega).
\end{equation}
Consequently, whenever $\widehat W_x(\Omega)>0$, the positive spectral weight
\begin{equation}\label{eq:spectral-weight-def-main}
\mathcal S_x(\Omega)=
\bigl(e^{\beta_\infty N(x)\Omega}-1\bigr)\widehat W_x(\Omega)
\end{equation}
gives
\begin{equation}\label{eq:kms-rate-main}
\Gamma_+(x,\Omega)=
\lambda_{\mathrm{sw}}^2\,
\frac{\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1},
\qquad
\Gamma_-(x,\Omega)=
\lambda_{\mathrm{sw}}^2\,
\frac{e^{\beta_\infty N(x)\Omega}\mathcal S_x(\Omega)}
{e^{\beta_\infty N(x)\Omega}-1}.
\end{equation}
\end{proposition}

\begin{proof}
Assumption~\ref{ass:long-time-switch-main}(ii) is exactly the stationary detector input needed for the long-time rate formulas \eqref{eq:fourier-rates-main}; it is the standard smooth-switching stationary response theorem used in detector theory \cite{LoukoSatz2008,FewsterJuarezLouko2016}.

For the KMS identity, abbreviate $\beta_{\mathrm{loc}}=\beta_\infty N(x)$ and let $\phi\in C_c^\infty(\RR)$. Define the entire function
\[
h_\phi(z)=\int_{\RR}e^{-i\nu z}\phi(\nu)\,d\nu.
\]
Because $\phi$ has compact support, $h_\phi$ decays faster than any inverse power uniformly on closed horizontal strips. Together with the polynomial bounds in \Cref{ass:long-time-switch-main}(iii), Cauchy's theorem on rectangles in $S_{\beta_{\mathrm{loc}}}^-$ shows that
\[
\int_{\RR}F_x(s-i\eta)\,h_\phi(s-i\eta)\,ds
\]
is independent of $\eta\in(0,\beta_{\mathrm{loc}})$. Taking $\eta\downarrow0$ and $\eta\uparrow\beta_{\mathrm{loc}}$, and using the distributional boundary values from \Cref{ass:long-time-switch-main}(iii), yields
\[
\langle W_x,h_\phi\rangle
=
\langle W_x(-\cdot),h_\phi(\cdot-i\beta_{\mathrm{loc}})\rangle.
\]
Since
\[
h_\phi(s-i\beta_{\mathrm{loc}})
=\int_{\RR}e^{-i\nu s}e^{-\beta_{\mathrm{loc}}\nu}\phi(\nu)\,d\nu
=h_{e^{-\beta_{\mathrm{loc}}(\cdot)}\phi}(s),
\]
the preceding identity becomes
\[
\langle \widehat W_x,\phi\rangle
=
\langle \widehat{W_x(-\cdot)},e^{-\beta_{\mathrm{loc}}(\cdot)}\phi\rangle.
\]
Using $\widehat{W_x(-\cdot)}(\nu)=\widehat W_x(-\nu)$ distributionally, we obtain
\[
\langle \widehat W_x,\phi\rangle
=
\langle e^{-\beta_{\mathrm{loc}}(\cdot)}\widehat W_x(-\cdot),\phi\rangle
\]
for all $\phi\in C_c^\infty(\RR)$, hence \eqref{eq:kms-spectral-distribution-main}. Evaluating this identity at $\nu=\Omega$ is legitimate because $\widehat W_x$ is represented by a continuous function near $\pm\Omega$, and this gives \eqref{eq:kms-detailed-balance-main}. Substituting the definition of $\mathcal S_x(\Omega)$ then yields \eqref{eq:kms-rate-main}. For operator-algebraic formulations of the same KMS strip and spectral relation, see \cite{HaagHugenholtzWinnink1967,KayWald1991,BratteliRobinson1997}.
\end{proof}

Thus the KMS condition fixes the detailed-balance ratio
\[
\frac{\Gamma_-(x,\Omega)}{\Gamma_+(x,\Omega)}
=e^{\beta_{\mathrm{loc}}(x)\Omega},
\]
with our convention that $\Gamma_+$ is excitation and $\Gamma_-$ is de-excitation. It does not, by itself, fix the absolute magnitude of the excitation rate. The remaining positive factor is a local response spectral weight. We therefore keep a separate regularity and calibration hypothesis for this factor rather than treating it as a geometric identity.

\begin{assumption}[Static response regularity and calibration]\label{ass:kms-response-main}
Fix a static region $U$ and a detector with positive gap $\Omega$ and fixed spatial profile. For each static worldline $x\in U$, use the stationary weak-coupling rates supplied by \Cref{prop:stationary-udw-response-main}. Assume that the positive-frequency response $\widehat W_x(\Omega)$ is finite and nonzero, and define $\mathcal S_x(\Omega)$ by \eqref{eq:spectral-weight-def-main}. The function $x\mapsto\mathcal S_x(\Omega)$ is specified or calibrated as part of the response model and has the $C^1$ regularity stated in each shell-regularity or inverse result below.
\end{assumption}

\begin{proposition}[Exact homogeneous model from spatial symmetry]\label{prop:symmetry-homogeneous-main}
Assume, in addition to \Cref{ass:long-time-switch-main}, that the static region is an ultrastatic spacetime $(\mathbb R\times\Sigma,-dt^2+h)$ and that a group $G$ of spatial isometries acts transitively on a region $U\subset\Sigma$. Suppose the $\beta_\infty$-KMS state $\omega$ is invariant under the induced automorphisms $\alpha_g$ of the field algebra, and that the detector profile at $x=gx_0$ is obtained from one reference profile by rigid transport under $g$. Then
\[
W_x(s)=W_{x_0}(s),\qquad
\widehat W_x(\nu)=\widehat W_{x_0}(\nu),\qquad
\mathcal S_x(\Omega)=\mathcal S_{x_0}(\Omega)
\]
for all $x\in U$. Hence the homogeneous spectral-weight regime holds exactly on $U$.

In particular, for a thermal Klein--Gordon KMS state in Minkowski space with identical translated compactly supported smearing profiles, the one-gap spectral weight is exactly independent of the detector position.
\end{proposition}

\begin{proof}
Let $x=gx_0$ with $g\in G$. By construction of the transported detector profile, the smeared detector observable satisfies
\[
\Phi_x(\tau)=\alpha_g\!\bigl(\Phi_{x_0}(\tau)\bigr).
\]
Using the $G$-invariance of the state,
\[
\begin{aligned}
W_x(s)
&=\omega\!\left(\Phi_x(\tau)\Phi_x(\tau-s)\right) \\
&=\omega\!\left(\alpha_g(\Phi_{x_0}(\tau))\,\alpha_g(\Phi_{x_0}(\tau-s))\right) \\
&=\omega\!\left(\alpha_g(\Phi_{x_0}(\tau)\Phi_{x_0}(\tau-s))\right) \\
&=\omega\!\left(\Phi_{x_0}(\tau)\Phi_{x_0}(\tau-s)\right)
=W_{x_0}(s).
\end{aligned}
\]
Taking Fourier transforms gives $\widehat W_x(\nu)=\widehat W_{x_0}(\nu)$, and then \eqref{eq:spectral-weight-def-main} gives $\mathcal S_x(\Omega)=\mathcal S_{x_0}(\Omega)$. In Minkowski thermal equilibrium the spatial translation group is transitive, the thermal state is translation invariant, and rigidly translated detector profiles satisfy the stated covariance, so the same conclusion applies.
\end{proof}

Under \Cref{ass:kms-response-main}, \eqref{eq:kms-rate-main} is a consequence of the Fourier response formula and the KMS spectral relation. The implication is one-way for the purposes of shell geometry: first a response model supplies the positive-frequency pullback, then the spectral weight in \eqref{eq:spectral-weight-def-main} is computed from it, and only then are the record-selected target rates imposed. If instead one starts with an arbitrary positive stationary excitation profile satisfying KMS detailed balance at the same local inverse temperature, equation~\eqref{eq:kms-rate-main} can be read backwards as the definition of
\[
\mathcal S_x(\Omega)=
\lambda_{\mathrm{sw}}^{-2}\bigl(e^{\beta_\infty N(x)\Omega}-1\bigr)\Gamma_+(x,\Omega).
\]
That reverse reading is only a parametrization of the unknown absolute response, not an independent derivation of a detector shell. The content of \Cref{ass:kms-response-main} is therefore not a postulate replacing the transition-rate formula, but the regularity and calibration of the remaining positive factor after the standard response formula has been used. Thus $\mathcal S_x(\Omega)$ is part of the response model, not an inferred geometric invariant. It absorbs the local mode density, detector profile, and any greybody or boundary contribution that remains after the KMS occupation factor has been separated. The homogeneous regime of \Cref{ass:homogeneous-spectral-main} is therefore an explicit calibration regime, not a consequence of KMS thermality. Until that specialization is invoked, all shell statements retain the position dependence of $\mathcal S_x(\Omega)$.
Proposition~\ref{prop:symmetry-homogeneous-main} gives an explicit detector-field model in which this regime is exact. Section~\ref{sec:inverse} also records a quantitative slow-variation criterion showing that the same regime is recovered asymptotically when the shell separation is small compared with the spatial scale on which $\log \mathcal S_x(\Omega)$ varies.

\begin{unnumberedlemma}[One-gap KMS underdetermination]
At the fixed-gap rate level used in this section, Tolman scaling and KMS detailed balance do not determine the positive excitation-rate profile. Let $G:U\to(0,\infty)$ be any $C^1$ function. The rates
\[
\Gamma_+^G(x,\Omega)=G(x),
\qquad
\Gamma_-^G(x,\Omega)=e^{\beta_\infty N(x)\Omega}G(x)
\]
satisfy the one-gap detailed-balance relation
\[
\Gamma_-^G(x,\Omega)/\Gamma_+^G(x,\Omega)
=e^{\beta_\infty N(x)\Omega},
\]
and can be written in the form \eqref{eq:kms-rate-main} with
\[
\mathcal S_x^G(\Omega)=
\lambda_{\mathrm{sw}}^{-2}\bigl(e^{\beta_\infty N(x)\Omega}-1\bigr)G(x)>0 .
\]
Consequently, changing the positive spectral weight changes the target-rate loci while preserving the same one-gap detailed-balance ratio. The shell statements below are therefore conditional statements for a fixed response function $R_\Omega$, or for an additional calibration such as the homogeneous spectral-weight regime. This is a fixed-frequency underdetermination statement, not an existence theorem asserting that every such $G$ extends to a global QFT two-point function.
\end{unnumberedlemma}

\begin{proof}
The displayed ratio of $\Gamma_-^G$ and $\Gamma_+^G$ is immediate. Substituting the displayed $\mathcal S_x^G(\Omega)$ into \eqref{eq:kms-rate-main} gives $\Gamma_+^G(x,\Omega)=G(x)$. Since $G$ was arbitrary among positive $C^1$ functions, the equation $G(x)=\gamma$ has whatever regular level-set geometry the chosen response profile supplies, subject only to the ordinary range and regularity conditions. Thus detailed balance alone cannot select the shell.
\end{proof}

The point of the underdetermination lemma is methodological. From here through the end of \Cref{sec:inverse}, the manuscript is no longer deriving detector-theoretic consequences of static KMS theory alone; it is recording what follows for a fixed calibrated response function
\[
R_\Omega(x)=
\lambda_{\mathrm{sw}}^2\,
\frac{\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}.
\]
All shell and inverse statements below should be read in that conditional sense.

Define the positive response function
\begin{equation}\label{eq:response-function-main}
R_\Omega(x)=
\lambda_{\mathrm{sw}}^2\,
\frac{\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}.
\end{equation}
The record analysis supplies two target rates,
\begin{equation}\label{eq:target-rates-main}
\gamma_\varphi=\frac{\log\varphi}{\Delta\tau},
\qquad
\gamma_{\mathrm{mem}}=\Gamma_c(\kappa_{\mathrm r},\Delta\tau).
\end{equation}
The geometric problem is therefore not to derive new response rates, but to solve the two response-level equations
\[
R_\Omega(x)=\gamma_\varphi,
\qquad
R_\Omega(x)=\gamma_{\mathrm{mem}}.
\]
The following lemma records the geometric step used by both detectors. The word ``shell'' is used only after this regularity check; otherwise the same equation defines a response locus.

\begin{lemma}[Response loci and shell regularity]\label{lem:response-locus-main}
Let $U$ be the static region under consideration, and assume that $N$ and $x\mapsto\mathcal S_x(\Omega)$ are $C^1$ on $U$, with $N>0$ and $\mathcal S_x(\Omega)>0$. For any target rate $\gamma>0$, define
\[
L_\gamma=\{x\in U:R_\Omega(x)=\gamma\}.
\]
Then $L_\gamma$ is empty exactly when $\gamma$ is outside the range of $R_\Omega$ on $U$. If $x_0\in L_\gamma$ and $dR_\Omega(x_0)\neq0$, then $L_\gamma$ is, near $x_0$, a $C^1$ codimension-one hypersurface with normal covector $dR_\Omega(x_0)$.

In the calibrated homogeneous spectral-weight regime, $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$, the target equation is equivalent to
\[
\beta_\infty N(x)=\beta_\gamma,
\qquad
\beta_\gamma=\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}{\gamma}\right).
\]
In that regime $dR_\Omega(x_0)\neq0$ if and only if $dN(x_0)\neq0$.
\end{lemma}

\begin{proof}
The first assertion is the definition of the range. The local hypersurface statement is the regular level-set theorem applied to the $C^1$ function $R_\Omega$.

If $\mathcal S_x(\Omega)$ is constant, the equation
\[
\lambda_{\mathrm{sw}}^2\frac{\mathcal S(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}=\gamma
\]
is equivalent to
\[
e^{\beta_\infty N(x)\Omega}
=1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}{\gamma},
\]
which gives the displayed value of $\beta_\gamma$. Differentiating the homogeneous response function gives
\[
dR_\Omega
=
-\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)
\frac{\beta_\infty\Omega e^{\beta_\infty N\Omega}}
{(e^{\beta_\infty N\Omega}-1)^2}\,dN .
\]
The scalar prefactor is strictly negative on $U$, so $dR_\Omega(x_0)$ vanishes exactly when $dN(x_0)$ vanishes.
\end{proof}

The theorem below is therefore a level-set statement for the fixed response function $R_\Omega$.

\begin{theorem}[Conditional shell loci]\label{thm:shells-general-main}
Fix a detector design $(\Omega,\lambda_c,\Delta\tau,\kappa_{\mathrm r})$ in a static KMS background with lapse $N(x)>0$ everywhere in the region of interest, and assume the response regularity and calibration in \Cref{ass:kms-response-main}. Assume that $N$ and the specified spectral weight $x\mapsto\mathcal S_x(\Omega)$ are $C^1$ on that region. For this fixed response function $R_\Omega$, the two record benchmarks are equivalent to the following response-level loci.
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.2em,itemsep=0.25em]
\item The ideal dead-time detector realizes the Parry benchmark exactly on the locus
\begin{equation}\label{eq:shell-phi-general-main}
\Sigma_{\varphi}=
\left\{x:
\frac{\lambda_{\mathrm{sw}}^2\mathcal S_x(\Omega)\Delta\tau}{e^{\beta_\infty N(x)\Omega}-1}=\log\varphi
\right\}.
\end{equation}
This is the locus for the ideal benchmark in the specified response model. It is a smooth codimension-one hypersurface wherever the gradient of $x\mapsto\lambda_{\mathrm{sw}}^2\mathcal S_x(\Omega)\Delta\tau/(e^{\beta_\infty N(x)\Omega}-1)$ is nonvanishing; otherwise it may fail to be a manifold and we refer to it as the ideal-benchmark locus.

\item The finite-recovery detector reaches the memory-transition point exactly on the locus
\begin{equation}\label{eq:shell-mem-general-main}
\Sigma_{\mathrm{mem}}=
\left\{x:
\frac{\lambda_{\mathrm{sw}}^2\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}
=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)
\right\}.
\end{equation}
This is the memory-transition locus for the specified response model. It is a smooth codimension-one hypersurface under the analogous nonvanishing-gradient condition. On $\Sigma_{\mathrm{mem}}$, the sampled record is exact $1$-dependent.
\end{enumerate}
\end{theorem}

\begin{proof}
For the ideal model, Proposition~\ref{prop:ideal-main} says that the Parry benchmark is equivalent to $\Gamma_+(x,\Omega)\Delta\tau=\log\varphi$, i.e. $R_\Omega(x)=\gamma_\varphi$ with the notation of~\eqref{eq:response-function-main}--\eqref{eq:target-rates-main}. By \Cref{prop:stationary-udw-response-main,ass:kms-response-main}, $\Gamma_+(x,\Omega)$ is the specified function $R_\Omega(x)$, so this condition is exactly \eqref{eq:shell-phi-general-main}. Conversely, any point satisfying \eqref{eq:shell-phi-general-main} has $R_\Omega(x)=\gamma_\varphi$ and hence realizes the ideal record benchmark.

For finite recovery, \Cref{prop:recovery-main,lem:kernel-factorization-main} say that the memory transition is equivalent to $\Gamma_+(x,\Omega)=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$, i.e. $R_\Omega(x)=\gamma_{\mathrm{mem}}$. With the same fixed calibrated response function, this is exactly \eqref{eq:shell-mem-general-main}, and every point satisfying that equation has an exact $1$-dependent sampled record in the finite-recovery model. The regularity and possible nonexistence statements are then exactly \Cref{lem:response-locus-main} applied to these two target rates. The preceding underdetermination lemma explains why the theorem is not a derivation of $R_\Omega$ from KMS thermality alone: changing $\mathcal S_x(\Omega)$ changes the loci while preserving one-gap detailed balance.
\end{proof}

Theorem~\ref{thm:shells-general-main} is the main output of this section: once $R_\Omega$ is fixed, the record-selected thresholds become two level sets, and regularity determines whether those level sets are smooth hypersurfaces. The remaining results in this section record the calibrated homogeneous specialization and a simple existence criterion on monotone spherical regions.

\begin{corollary}\label{cor:homogeneous-main}
Assume that $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$ on the region under study. Then both shells are pure lapse level sets. Their associated local inverse temperatures are
\begin{equation}\label{eq:betaphi-betamem-main}
\beta_{\varphi}=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)\Delta\tau}{\log\varphi}\right),
\qquad
\beta_{\mathrm{mem}}=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}
{\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}\right),
\end{equation}
so that
\begin{equation}\label{eq:lapse-level-main}
\Sigma_{\varphi}=\{x:\beta_{\mathrm{loc}}(x)=\beta_{\varphi}\},
\qquad
\Sigma_{\mathrm{mem}}=\{x:\beta_{\mathrm{loc}}(x)=\beta_{\mathrm{mem}}\}.
\end{equation}
\end{corollary}

\begin{proof}
With $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$, the two shell equations contain $x$ only through $\beta_\infty N(x)$. Solving the ideal-benchmark equation for $\beta_\infty N(x)$ gives $\beta_\varphi$, and solving the memory equation gives $\beta_{\mathrm{mem}}$. Hence the loci are the lapse level sets in~\eqref{eq:lapse-level-main}.
\end{proof}

For later shell-pair comparisons we use one fixed denominator orientation. If
$r_\varphi$ and $r_{\mathrm{mem}}$ are homogeneous ideal-benchmark and
memory-transition radii, set
\[
D_\varphi=e^{\beta_\infty\Omega N(r_\varphi)}-1,
\qquad
D_{\mathrm{mem}}=e^{\beta_\infty\Omega N(r_{\mathrm{mem}})}-1,
\qquad
Q_{\varphi/\mathrm{mem}}=\frac{D_\varphi}{D_{\mathrm{mem}}}.
\]
Thus $Q_{\varphi/\mathrm{mem}}$ means ideal-benchmark denominator divided by
memory-transition denominator. The cancellation identity and the reciprocal
target-rate convention are proved once in \Cref{thm:ratio-main}; this section
uses only the level-set formulas above.

\begin{proposition}[Existence on monotone spherical regions]\label{prop:existence-main}
Assume the calibrated homogeneous spectral-weight regime in a static spherically symmetric region $r\in I$, and suppose $N:I\to N(I)$ is $C^1$ and strictly monotone. For either threshold $\star\in\{\varphi,\mathrm{mem}\}$, the corresponding shell exists in $I$ if and only if
\[
\frac{\beta_\star}{\beta_\infty}\in N(I),
\]
where $\beta_\varphi$ and $\beta_{\mathrm{mem}}$ are defined in~\eqref{eq:betaphi-betamem-main}. When it exists, the shell radius is unique and equals
\[
r_\star=N^{-1}\!\left(\frac{\beta_\star}{\beta_\infty}\right).
\]
If $N'(r_\star)\neq0$, this radius defines a regular spherical hypersurface.
\end{proposition}

\begin{proof}
By \Cref{cor:homogeneous-main}, the shell equation is exactly $\beta_\infty N(r)=\beta_\star$. Since $N$ is strictly monotone on $I$, it is one-to-one onto its range. Hence the equation has a solution precisely when $\beta_\star/\beta_\infty$ lies in $N(I)$, and the solution is then the displayed inverse image. The nonzero derivative condition is the one-dimensional form of the regular level-set condition.
\end{proof}

In this calibrated regime, both shells are simply lapse level sets selected by the two record thresholds. The same local response scale therefore supports two different detector-selected response hypersurfaces whenever the range condition in \Cref{prop:existence-main} is met. \Cref{fig:shell-radii} illustrates the corresponding lapse values and proper-distance ratio in Schwarzschild.

\begin{proposition}\label{prop:order-main}
Assume the calibrated homogeneous spectral-weight regime and define
\[
\Gamma_{\varphi}=\frac{\log\varphi}{\Delta\tau}.
\]
Let $N_{\varphi}$ and $N_{\mathrm{mem}}$ denote the lapse values of $\Sigma_{\varphi}$ and $\Sigma_{\mathrm{mem}}$. Then
\begin{equation}\label{eq:order-main}
N_{\varphi}<N_{\mathrm{mem}}
\iff
\Gamma_{\varphi}>\Gamma_c(\kappa_{\mathrm r},\Delta\tau),
\end{equation}
with equality if and only if $\Gamma_{\varphi}=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$. Thus the shell ordering is detector controlled.
\end{proposition}

\begin{proof}
Both shells are lapse level sets in the calibrated homogeneous spectral-weight regime. Writing $A=\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)>0$, the local excitation rate as a function of the lapse is
\[
\Gamma_+(N)=\frac{A}{e^{\beta_\infty\Omega N}-1}.
\]
For $\beta_\infty,\Omega,N>0$,
\[
\frac{d\Gamma_+}{dN}
=
-A\,\frac{\beta_\infty\Omega e^{\beta_\infty\Omega N}}
{(e^{\beta_\infty\Omega N}-1)^2}<0 .
\]
Hence the higher threshold rate corresponds exactly to the smaller lapse value, giving~\eqref{eq:order-main} and the equality statement.
\end{proof}
