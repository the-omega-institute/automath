\section{Conditional response loci and shell regularity in static KMS response models}\label{sec:kms-shells}

Table~\ref{tab:hypothesis-ledger} already separates the record theorem from the
response-model input. In this section we use only the relevant part of that
map: the record law supplies the target rates $\log\varphi/\Delta\tau$ and
$\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$, KMS detailed balance supplies only
the thermal ratio, and a fixed calibrated response function $R_\Omega$ supplies
the absolute excitation profile. We therefore use the word ``shell'' only for
regular level sets of that fixed $R_\Omega$, always within the sampled interval
instrument of \Cref{prop:instrument-kernels-main} and the closure
$\Gamma=\Gamma_+(x,\Omega)$.

Let $N(x)$ be the lapse function of a static spacetime and assume the field is in a global $\beta_\infty$-KMS state for Killing time. Along a static worldline at position $x$, Tolman scaling gives \cite{Tolman1930,TolmanEhrenfest1930}
\begin{equation}\label{eq:tolman-main}
\beta_{\mathrm{loc}}(x)=\beta_\infty N(x).
\end{equation}
This relation fixes the proper-time KMS parameter seen by a static detector. It does not say that every local response observable is a function of $N(x)$ alone; mode density, boundary conditions, detector profile, and greybody transmission can still enter the absolute transition rate.

For a static worldline, the weak-coupling transition rates are Fourier transforms of the pulled-back two-point function. We use this standard UDW response fact explicitly rather than absorbing it into a response ansatz. Let $\Phi_x(\tau)$ denote the field operator smeared with the detector profile and transported along the static orbit through $x$, and set
\[
W_x(s)=\omega\!\left(\Phi_x(\tau)\Phi_x(\tau-s)\right),
\qquad
\widehat W_x(\nu)=\int_{\RR}e^{-i\nu s}W_x(s)\,ds,
\]
with the Fourier transform understood distributionally whenever $W_x$ is not an $L^1$ function.

\begin{assumption}[Switched stationary response input and KMS strip regularity]\label{ass:long-time-switch-main}
Fix a static worldline through $x$, a detector gap $\Omega>0$, and a smooth nonnegative switching profile $\chi\in C_c^\infty(\mathbb R)$ with $\int_{\RR}\chi=1$. Let $\chi_T(t)=\chi(t/T)$ and let $H_{g\to e}^{(T)}(x,\Omega)$ and $H_{e\to g}^{(T)}(x,\Omega)$ be the corresponding first-order transition probabilities for a detector with fixed spatial profile and bare coupling $\lambda_c$. Assume:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.2em,itemsep=0.25em]
\item The pullback $W_x$ is a tempered distribution on $\RR$, and there is an open interval $I_x\subset\RR$ containing $\pm\Omega$ on which its Fourier transform $\widehat W_x$ is represented by a continuous function.
\item The switched first-order transition probabilities are given by the standard stationary UDW response pairing
\begin{equation}\label{eq:switched-pairing-main}
H_{g\to e}^{(T)}(x,\Omega)
=\lambda_c^2\langle W_x,K_{T,\Omega}\rangle,
\qquad
H_{e\to g}^{(T)}(x,\Omega)
=\lambda_c^2\langle W_x,K_{T,-\Omega}\rangle,
\end{equation}
where
\[
K_{T,\omega}(s)=e^{-i\omega s}\int_{\RR}\chi_T(u)\chi_T(u-s)\,du.
\]
Equivalently, whenever $W_x$ is represented by a function, this is the usual double integral
\begin{equation}\label{eq:switched-double-integral-main}
H_{g\to e}^{(T)}(x,\Omega)
=\lambda_c^2\int_{\RR^2}
\chi_T(\tau)\chi_T(\tau')e^{-i\Omega(\tau-\tau')}W_x(\tau-\tau')\,d\tau\,d\tau',
\end{equation}
with the de-excitation formula obtained by replacing $\Omega$ by $-\Omega$.
\item If the state is $\beta_\infty$-KMS with respect to static Killing time, then with $\beta_{\mathrm{loc}}(x)=\beta_\infty N(x)$ there exists a function $F_x$ holomorphic on the strip
\[
S_{\beta_{\mathrm{loc}}}^-=\{z\in\mathbb C:-\beta_{\mathrm{loc}}(x)<\Im z<0\}
\]
such that for every closed substrip $-\beta_{\mathrm{loc}}(x)+\delta\le \Im z\le -\delta$ there exist $m_\delta\in\mathbb N$ and $C_\delta>0$ with
\[
|F_x(z)|\le C_\delta(1+|\Re z|)^{m_\delta},
\]
and the boundary values satisfy
\[
\lim_{\epsilon\to0^+}F_x(s-i\epsilon)=W_x(s),\qquad
\lim_{\epsilon\to0^+}F_x(s-i\beta_{\mathrm{loc}}(x)+i\epsilon)=W_x(-s)
\]
in $\mathcal S'(\RR)$.
\end{enumerate}
Define the switched effective coupling\footnote{The original bare coupling is $\lambda_c$; the switching convention contributes the fixed factor $\|\chi\|_2^2:=\int_{\mathbb R}\chi(u)^2\,du>0$.} by
\[
\lambda_{\mathrm{sw}}^2:=\lambda_c^2\,\|\chi\|_2^2
\]
and use $\lambda_{\mathrm{sw}}$ in the long-time rate formulas below.
\end{assumption}

\begin{proposition}[Stationary UDW response and KMS spectral relation]\label{prop:stationary-udw-response-main}
Fix a static worldline through $x$ and a two-level detector with gap $\Omega>0$, coupling scale $\lambda_c$, and fixed spatial profile. Assume \Cref{ass:long-time-switch-main}. Then the stationary weak-coupling long-time excitation and de-excitation rates are
\begin{equation}\label{eq:long-time-kernel-main}
\frac{1}{T}H_{g\to e}^{(T)}(x,\Omega)
=\lambda_c^2\|\chi\|_2^2\,\langle \widehat W_x,\Phi_{T,\Omega}\rangle,
\qquad
\frac{1}{T}H_{e\to g}^{(T)}(x,\Omega)
=\lambda_c^2\|\chi\|_2^2\,\langle \widehat W_x,\Phi_{T,-\Omega}\rangle,
\end{equation}
where
\begin{equation}\label{eq:approx-identity-main}
\Phi_{T,\omega}(\nu)=
\frac{T}{2\pi\|\chi\|_2^2}\,
\bigl|\widehat\chi\bigl(T(\omega-\nu)\bigr)\bigr|^2 .
\end{equation}
The family $\Phi_{T,\omega}$ is a nonnegative approximate identity centred at $\omega$, so
\begin{equation}\label{eq:long-time-limit-main}
\begin{aligned}
\Gamma_+(x,\Omega)
&:=\lim_{T\to\infty}\frac{1}{T}H_{g\to e}^{(T)}(x,\Omega)
=\lambda_c^2\|\chi\|_2^2\,\widehat W_x(\Omega),\\
\Gamma_-(x,\Omega)
&:=\lim_{T\to\infty}\frac{1}{T}H_{e\to g}^{(T)}(x,\Omega)
=\lambda_c^2\|\chi\|_2^2\,\widehat W_x(-\Omega).
\end{aligned}
\end{equation}
Hence, with
\[
\lambda_{\mathrm{sw}}^2=\lambda_c^2\|\chi\|_2^2,
\]
one obtains
\begin{equation}\label{eq:fourier-rates-main}
\Gamma_+(x,\Omega)=\lambda_{\mathrm{sw}}^2\widehat W_x(\Omega),
\qquad
\Gamma_-(x,\Omega)=\lambda_{\mathrm{sw}}^2\widehat W_x(-\Omega).
\end{equation}
If the state is $\beta_\infty$-KMS with respect to static Killing time, then the static pullback is KMS in proper time with inverse temperature $\beta_\infty N(x)$ and
\begin{equation}\label{eq:kms-spectral-distribution-main}
\widehat W_x(\nu)=e^{-\beta_\infty N(x)\nu}\widehat W_x(-\nu)
\qquad\text{in }\mathcal S'(\RR).
\end{equation}
In particular, because $\widehat W_x$ is continuous at $\nu=\pm\Omega$,
\begin{equation}\label{eq:kms-detailed-balance-main}
\widehat W_x(-\Omega)=e^{\beta_\infty N(x)\Omega}\widehat W_x(\Omega).
\end{equation}
Consequently, whenever $\widehat W_x(\Omega)>0$, the positive spectral weight
\begin{equation}\label{eq:spectral-weight-def-main}
\mathcal S_x(\Omega)=
\bigl(e^{\beta_\infty N(x)\Omega}-1\bigr)\widehat W_x(\Omega)
\end{equation}
gives
\begin{equation}\label{eq:kms-rate-main}
\Gamma_+(x,\Omega)=
\lambda_{\mathrm{sw}}^2\,
\frac{\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1},
\qquad
\Gamma_-(x,\Omega)=
\lambda_{\mathrm{sw}}^2\,
\frac{e^{\beta_\infty N(x)\Omega}\mathcal S_x(\Omega)}
{e^{\beta_\infty N(x)\Omega}-1}.
\end{equation}
\end{proposition}

\begin{proof}
Write
\[
g_T(s)=\int_{\RR}\chi_T(u)\chi_T(u-s)\,du.
\]
Then \Cref{ass:long-time-switch-main}(ii) gives
\[
H_{g\to e}^{(T)}(x,\Omega)=\lambda_c^2\langle W_x,e^{-i\Omega(\cdot)}g_T\rangle.
\]
Because $g_T$ is the autocorrelation of $\chi_T$, its Fourier transform is
\[
\widehat g_T(\nu)=|\widehat{\chi_T}(\nu)|^2
=T^2\bigl|\widehat\chi(T\nu)\bigr|^2.
\]
Using the Fourier inversion convention
\[
g(s)=\frac{1}{2\pi}\int_{\RR}e^{i\nu s}\widehat g(\nu)\,d\nu,
\]
we obtain the exact spectral representation
\[
\frac{1}{T}H_{g\to e}^{(T)}(x,\Omega)
=\frac{\lambda_c^2}{2\pi T}
\left\langle \widehat W_x,\bigl|\widehat{\chi_T}(\Omega-\cdot)\bigr|^2\right\rangle
=\lambda_c^2\|\chi\|_2^2\,\langle \widehat W_x,\Phi_{T,\Omega}\rangle,
\]
which is \eqref{eq:long-time-kernel-main}; the de-excitation formula is the same
with $\Omega$ replaced by $-\Omega$.

The functions $\Phi_{T,\omega}$ are nonnegative and
\[
\int_{\RR}\Phi_{T,\omega}(\nu)\,d\nu
=\frac{1}{2\pi\|\chi\|_2^2}\int_{\RR}|\widehat\chi(\zeta)|^2\,d\zeta
=1
\]
by the change of variables $\zeta=T(\omega-\nu)$ and Plancherel. Thus
$\Phi_{T,\omega}$ is an approximate identity centred at $\omega$. To extract the
limit against the tempered distribution $\widehat W_x$, choose
$\eta\in C_c^\infty(I_x)$ with $\eta\equiv1$ on a smaller neighborhood of
$\omega\in\{\Omega,-\Omega\}$. Then
\[
\langle \widehat W_x,\Phi_{T,\omega}\rangle
=\int_{\RR}\eta(\nu)\widehat W_x(\nu)\Phi_{T,\omega}(\nu)\,d\nu
+\langle \widehat W_x,(1-\eta)\Phi_{T,\omega}\rangle.
\]
Because $\eta\widehat W_x$ is continuous near $\omega$, the first term tends to
$\widehat W_x(\omega)$. Because $\widehat\chi$ is Schwartz and $(1-\eta)$
vanishes near $\omega$, the second term tends to $0$ in every Schwartz
seminorm, hence also after pairing with the tempered distribution
$\widehat W_x$. Therefore
\[
\langle \widehat W_x,\Phi_{T,\omega}\rangle\longrightarrow \widehat W_x(\omega),
\qquad \omega\in\{\Omega,-\Omega\},
\]
which proves \eqref{eq:long-time-limit-main} and therefore
\eqref{eq:fourier-rates-main}. This is the exact long-time normalization used in
the standard smooth-switching stationary detector theorem
\cite{LoukoSatz2008,FewsterJuarezLouko2016}.

	For the KMS identity, abbreviate $\beta_{\mathrm{loc}}=\beta_\infty N(x)$ and let $\phi\in C_c^\infty(\RR)$. Define the entire function
\[
h_\phi(z)=\int_{\RR}e^{-i\nu z}\phi(\nu)\,d\nu.
\]
Because $\phi$ has compact support, $h_\phi$ decays faster than any inverse power uniformly on closed horizontal strips. Together with the polynomial bounds in \Cref{ass:long-time-switch-main}(iii), Cauchy's theorem on rectangles in $S_{\beta_{\mathrm{loc}}}^-$ shows that
\[
\int_{\RR}F_x(s-i\eta)\,h_\phi(s-i\eta)\,ds
\]
is independent of $\eta\in(0,\beta_{\mathrm{loc}})$. Taking $\eta\downarrow0$ and $\eta\uparrow\beta_{\mathrm{loc}}$, and using the distributional boundary values from \Cref{ass:long-time-switch-main}(iii), yields
\[
\langle W_x,h_\phi\rangle
=
\langle W_x(-\cdot),h_\phi(\cdot-i\beta_{\mathrm{loc}})\rangle.
\]
Since
\[
h_\phi(s-i\beta_{\mathrm{loc}})
=\int_{\RR}e^{-i\nu s}e^{-\beta_{\mathrm{loc}}\nu}\phi(\nu)\,d\nu
=h_{e^{-\beta_{\mathrm{loc}}(\cdot)}\phi}(s),
\]
the preceding identity becomes
\[
\langle \widehat W_x,\phi\rangle
=
\langle \widehat{W_x(-\cdot)},e^{-\beta_{\mathrm{loc}}(\cdot)}\phi\rangle.
\]
Using $\widehat{W_x(-\cdot)}(\nu)=\widehat W_x(-\nu)$ distributionally, we obtain
\[
\langle \widehat W_x,\phi\rangle
=
\langle e^{-\beta_{\mathrm{loc}}(\cdot)}\widehat W_x(-\cdot),\phi\rangle
\]
for all $\phi\in C_c^\infty(\RR)$, hence \eqref{eq:kms-spectral-distribution-main}. Evaluating this identity at $\nu=\Omega$ is legitimate because $\widehat W_x$ is represented by a continuous function near $\pm\Omega$, and this gives \eqref{eq:kms-detailed-balance-main}. Substituting the definition of $\mathcal S_x(\Omega)$ then yields \eqref{eq:kms-rate-main}. For operator-algebraic formulations of the same KMS strip and spectral relation, see \cite{HaagHugenholtzWinnink1967,KayWald1991,BratteliRobinson1997}.
\end{proof}

Thus the KMS condition fixes the detailed-balance ratio
\[
\frac{\Gamma_-(x,\Omega)}{\Gamma_+(x,\Omega)}
=e^{\beta_{\mathrm{loc}}(x)\Omega},
\]
with our convention that $\Gamma_+$ is excitation and $\Gamma_-$ is de-excitation. It does not, by itself, fix the absolute magnitude of the excitation rate. The remaining positive factor is a local response spectral weight. We therefore keep a separate regularity and calibration hypothesis for this factor rather than treating it as a geometric identity.

\begin{assumption}[Static response regularity and calibration]\label{ass:kms-response-main}
Fix a static region $U$ and a detector with positive gap $\Omega$ and fixed spatial profile. For each static worldline $x\in U$, use the stationary weak-coupling rates supplied by \Cref{prop:stationary-udw-response-main}. Assume that the positive-frequency response $\widehat W_x(\Omega)$ is finite and nonzero, and define $\mathcal S_x(\Omega)$ by \eqref{eq:spectral-weight-def-main}. The function $x\mapsto\mathcal S_x(\Omega)$ is specified or calibrated as part of the response model and has the $C^1$ regularity stated in each shell-regularity or inverse result below.
\end{assumption}

\begin{proposition}[Exact homogeneous model from spatial symmetry]\label{prop:symmetry-homogeneous-main}
Assume, in addition to \Cref{ass:long-time-switch-main}, that the static region is an ultrastatic spacetime $(\mathbb R\times\Sigma,-dt^2+h)$ and that a group $G$ of spatial isometries acts transitively on a region $U\subset\Sigma$. Suppose the $\beta_\infty$-KMS state $\omega$ is invariant under the induced automorphisms $\alpha_g$ of the field algebra, and that the detector profile at $x=gx_0$ is obtained from one reference profile by rigid transport under $g$. Then
\[
W_x(s)=W_{x_0}(s),\qquad
\widehat W_x(\nu)=\widehat W_{x_0}(\nu),\qquad
\mathcal S_x(\Omega)=\mathcal S_{x_0}(\Omega)
\]
for all $x\in U$. Hence the homogeneous spectral-weight regime holds exactly on $U$.

In particular, for a thermal Klein--Gordon KMS state in Minkowski space with identical translated compactly supported smearing profiles, the one-gap spectral weight is exactly independent of the detector position.
\end{proposition}

\begin{proof}
Let $x=gx_0$ with $g\in G$. By construction of the transported detector profile, the smeared detector observable satisfies
\[
\Phi_x(\tau)=\alpha_g\!\bigl(\Phi_{x_0}(\tau)\bigr).
\]
Using the $G$-invariance of the state,
\[
\begin{aligned}
W_x(s)
&=\omega\!\left(\Phi_x(\tau)\Phi_x(\tau-s)\right) \\
&=\omega\!\left(\alpha_g(\Phi_{x_0}(\tau))\,\alpha_g(\Phi_{x_0}(\tau-s))\right) \\
&=\omega\!\left(\alpha_g(\Phi_{x_0}(\tau)\Phi_{x_0}(\tau-s))\right) \\
&=\omega\!\left(\Phi_{x_0}(\tau)\Phi_{x_0}(\tau-s)\right)
=W_{x_0}(s).
\end{aligned}
\]
Taking Fourier transforms gives $\widehat W_x(\nu)=\widehat W_{x_0}(\nu)$, and then \eqref{eq:spectral-weight-def-main} gives $\mathcal S_x(\Omega)=\mathcal S_{x_0}(\Omega)$. In Minkowski thermal equilibrium the spatial translation group is transitive, the thermal state is translation invariant, and rigidly translated detector profiles satisfy the stated covariance, so the same conclusion applies.
\end{proof}

In this exact symmetry example the ultrastatic lapse satisfies $N\equiv 1$ on
$U$, so the fixed calibrated response function $R_\Omega$ is constant on $U$ as
well. The corresponding level sets are therefore either empty or all of $U$.
Proposition~\ref{prop:symmetry-homogeneous-main} verifies only the homogeneous
amplitude statement $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$; nontrivial
homogeneous shell geometry still requires a separate calibrated model with
nonconstant lapse.

Under \Cref{ass:kms-response-main}, \eqref{eq:kms-rate-main} is a consequence of the Fourier response formula and the KMS spectral relation. The implication is one-way for the purposes of response-locus analysis: first a response model supplies the positive-frequency pullback, then the spectral weight in \eqref{eq:spectral-weight-def-main} is computed from it, and only then are the record-selected target rates imposed. If instead one starts with an arbitrary positive stationary excitation profile satisfying KMS detailed balance at the same local inverse temperature, equation~\eqref{eq:kms-rate-main} can be read backwards as the definition of
\[
\mathcal S_x(\Omega)=
\lambda_{\mathrm{sw}}^{-2}\bigl(e^{\beta_\infty N(x)\Omega}-1\bigr)\Gamma_+(x,\Omega).
\]
That reverse reading is only a parametrization of the unknown absolute response, not an independent derivation of a detector shell. The content of \Cref{ass:kms-response-main} is therefore not a postulate replacing the transition-rate formula, but the regularity and calibration of the remaining positive factor after the standard response formula has been used. Thus $\mathcal S_x(\Omega)$ is part of the response model, not an inferred geometric invariant. It absorbs the local mode density, detector profile, and any greybody or boundary contribution that remains after the KMS occupation factor has been separated. The homogeneous regime of \Cref{ass:homogeneous-spectral-main} is therefore an explicit calibration regime, not a consequence of KMS thermality. Until that specialization is invoked, all shell statements retain the position dependence of $\mathcal S_x(\Omega)$.
Proposition~\ref{prop:symmetry-homogeneous-main} gives an explicit detector-field
model in which this amplitude homogeneity is exact, but because that
ultrastatic symmetry example has constant lapse it does not by itself generate
nontrivial shell geometry. Section~\ref{sec:inverse} instead studies the
abstract calibrated specialization in which one fixes a nonconstant lapse and a
homogeneous response amplitude directly at the response-function level; no
separate exact nonconstant-lapse UDW realization is proved in the present
paper. It also records a quantitative
slow-variation criterion showing that the same homogeneous amplitude regime is
recovered asymptotically when the shell separation is small compared with the
spatial scale on which $\log \mathcal S_x(\Omega)$ varies.

For the rest of the section we package the calibrated excitation rate as the
fixed response function
\begin{equation}\label{eq:response-function-main}
R_\Omega(x)=
\lambda_{\mathrm{sw}}^2\,
\frac{\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}.
\end{equation}
This is the response-model input that later supports response loci and, under a
regularity hypothesis, shells.

\begin{lemma}[One-gap KMS underdetermination]\label{lem:one-gap-kms-underdetermination-main}
At the fixed-gap rate level used in this section, Tolman scaling and KMS detailed balance do not determine the positive excitation-rate profile. Let $G:U\to(0,\infty)$ be any $C^1$ function. The rates
\[
\Gamma_+^G(x,\Omega)=G(x),
\qquad
\Gamma_-^G(x,\Omega)=e^{\beta_\infty N(x)\Omega}G(x)
\]
satisfy the one-gap detailed-balance relation
\[
\Gamma_-^G(x,\Omega)/\Gamma_+^G(x,\Omega)
=e^{\beta_\infty N(x)\Omega},
\]
and can be written in the form \eqref{eq:kms-rate-main} with
\[
\mathcal S_x^G(\Omega)=
\lambda_{\mathrm{sw}}^{-2}\bigl(e^{\beta_\infty N(x)\Omega}-1\bigr)G(x)>0 .
\]
Consequently, changing the positive spectral weight changes the target-rate loci while preserving the same one-gap detailed-balance ratio. The shell statements below are therefore conditional statements for a fixed response function $R_\Omega$, or for an additional calibration such as the homogeneous spectral-weight regime. This is a fixed-frequency underdetermination statement, not an existence theorem asserting that every such $G$ extends to a global QFT two-point function.
\end{lemma}

\begin{proof}
The displayed ratio of $\Gamma_-^G$ and $\Gamma_+^G$ is immediate. Substituting the displayed $\mathcal S_x^G(\Omega)$ into \eqref{eq:kms-rate-main} gives $\Gamma_+^G(x,\Omega)=G(x)$. Since $G$ was arbitrary among positive $C^1$ functions, the equation $G(x)=\gamma$ has whatever regular level-set structure the chosen response profile supplies, subject only to the ordinary range and regularity conditions. Thus detailed balance alone cannot select the shell.
\end{proof}

The underdetermination statement of
\Cref{lem:one-gap-kms-underdetermination-main} comes before any shell
interpretation. From here through the end of \Cref{sec:inverse}, we work with
the fixed calibrated response function \eqref{eq:response-function-main}. The
record analysis then supplies two target rates,
\begin{equation}\label{eq:target-rates-main}
\gamma_\varphi=\frac{\log\varphi}{\Delta\tau},
\qquad
\gamma_{\mathrm{mem}}=\Gamma_c(\kappa_{\mathrm r},\Delta\tau).
\end{equation}
The next step is therefore not to derive new response rates, but to solve the
two response-level equations
\[
R_\Omega(x)=\gamma_\varphi,
\qquad
R_\Omega(x)=\gamma_{\mathrm{mem}}.
\]
The following lemma isolates the regularity step used by both detectors after
\Cref{lem:one-gap-kms-underdetermination-main} has fixed the logical status of
the response profile. The word ``shell'' is used only after this regularity
check; otherwise the same equation defines only a response locus for the fixed
calibrated $R_\Omega$.

\begin{lemma}[Response loci for a fixed calibrated $R_\Omega$ and shell regularity]\label{lem:response-locus-main}
Let $R_\Omega$ be the fixed calibrated response function in
\eqref{eq:response-function-main}. Let $U$ be the static region under
consideration, and assume that $N$ and $x\mapsto\mathcal S_x(\Omega)$ are
$C^1$ on $U$, with $N>0$ and $\mathcal S_x(\Omega)>0$. For any target rate
$\gamma>0$, define
\[
L_\gamma=\{x\in U:R_\Omega(x)=\gamma\}.
\]
Then $L_\gamma$ is empty exactly when $\gamma$ is outside the range of $R_\Omega$ on $U$. If $x_0\in L_\gamma$ and $dR_\Omega(x_0)\neq0$, then $L_\gamma$ is, near $x_0$, a $C^1$ codimension-one hypersurface with normal covector $dR_\Omega(x_0)$.

In the calibrated homogeneous spectral-weight regime, $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$, the target equation is equivalent to
\[
\beta_\infty N(x)=\beta_\gamma,
\qquad
\beta_\gamma=\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}{\gamma}\right).
\]
In that regime $dR_\Omega(x_0)\neq0$ if and only if $dN(x_0)\neq0$.
\end{lemma}

\begin{proof}
The first assertion is the definition of the range. The local hypersurface statement is the regular level-set theorem applied to the $C^1$ function $R_\Omega$.

If $\mathcal S_x(\Omega)$ is constant, the equation
\[
\lambda_{\mathrm{sw}}^2\frac{\mathcal S(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}=\gamma
\]
is equivalent to
\[
e^{\beta_\infty N(x)\Omega}
=1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}{\gamma},
\]
which gives the displayed value of $\beta_\gamma$. Differentiating the homogeneous response function gives
\[
dR_\Omega
=
-\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)
\frac{\beta_\infty\Omega e^{\beta_\infty N\Omega}}
{(e^{\beta_\infty N\Omega}-1)^2}\,dN .
\]
The scalar prefactor is strictly negative on $U$, so $dR_\Omega(x_0)$ vanishes exactly when $dN(x_0)$ vanishes.
\end{proof}

\Cref{lem:one-gap-kms-underdetermination-main,lem:response-locus-main} are the
two ingredients that make shell language precise: the first fixes the
calibration dependence, and the second upgrades a response locus to a shell at
regular points.

\begin{lemma}[Abstract calibrated homogeneous specialization]\label{lem:abstract-homogeneous-specialization-main}
Fix a static region $U$, a gap $\Omega>0$, a Tolman parameter
$\beta_\infty>0$, a positive constant $A$, and a $C^1$ lapse
$N:U\to(0,\infty)$. Define the calibrated response function
\[
R_\Omega^{\mathrm{hom}}(x)=\frac{A}{e^{\beta_\infty N(x)\Omega}-1}.
\]
Then for any target rate $\gamma>0$ the response locus
\[
L_\gamma^{\mathrm{hom}}
=\{x\in U:R_\Omega^{\mathrm{hom}}(x)=\gamma\}
\]
is exactly the lapse level set
\[
\beta_\infty N(x)=\frac{1}{\Omega}\log\!\left(1+\frac{A}{\gamma}\right).
\]
If $dN(x_0)\neq0$ at $x_0\in L_\gamma^{\mathrm{hom}}$, then
$L_\gamma^{\mathrm{hom}}$ is near $x_0$ a $C^1$ codimension-one shell. This is
an abstract calibrated specialization of a fixed response function. The only
exact detector-field realization proved in this paper is
\Cref{prop:symmetry-homogeneous-main}, where the lapse is constant; whenever
later sections use homogeneous shell geometry with nonconstant lapse, they do
so through this level-set specialization rather than through a separate exact
UDW construction.
\end{lemma}

\begin{proof}
Solving $R_\Omega^{\mathrm{hom}}(x)=\gamma$ gives
\[
e^{\beta_\infty N(x)\Omega}=1+\frac{A}{\gamma},
\]
hence the displayed lapse level set. Differentiating yields
\[
dR_\Omega^{\mathrm{hom}}
=-A\frac{\beta_\infty\Omega e^{\beta_\infty N\Omega}}
{(e^{\beta_\infty N\Omega}-1)^2}\,dN,
\]
whose scalar prefactor is strictly negative. Therefore
$dR_\Omega^{\mathrm{hom}}(x_0)\neq0$ exactly when $dN(x_0)\neq0$, and the
regular level-set theorem gives the shell statement.
\end{proof}

\begin{theorem}[Conditional response loci and shell interpretation for a fixed calibrated response function $R_\Omega$]\label{thm:shells-general-main}
Assume the sampled classical interval instrument of
\Cref{prop:instrument-kernels-main}, with independent Poisson attempts and
fresh interval ancillas, and impose only the modelling closure
$\Gamma=\Gamma_+(x,\Omega)$. Fix a detector design
$(\Omega,\lambda_c,\Delta\tau,\kappa_{\mathrm r})$ in a static KMS background
with lapse $N(x)>0$ everywhere in the region of interest, and assume the
response regularity and calibration in \Cref{ass:kms-response-main}. Assume
that $N$ and the specified spectral weight $x\mapsto\mathcal S_x(\Omega)$ are
$C^1$ on that region. For this fixed response function $R_\Omega$, the two
record benchmarks of the UDW-calibrated classical sampled counter are equivalent to the
following response-level loci. Their nonemptiness is only a range condition for
the already calibrated function $R_\Omega$: if the relevant target rate lies
outside the range of $R_\Omega$ on the chosen region, the corresponding locus
is empty. If the target rate lies in the range, one obtains a shell only at
points where $dR_\Omega\neq0$.
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.2em,itemsep=0.25em]
\item The ideal one-step blanking counter mode reaches the Parry benchmark
exactly on the response locus
\begin{equation}\label{eq:shell-phi-general-main}
\Sigma_{\varphi}=
\left\{x:
\frac{\lambda_{\mathrm{sw}}^2\mathcal S_x(\Omega)\Delta\tau}{e^{\beta_\infty N(x)\Omega}-1}=\log\varphi
\right\}.
\end{equation}
This is the ideal-benchmark response locus for the specified response model
obtained after imposing $\Gamma=\Gamma_+(x,\Omega)$. Wherever
$dR_\Omega\neq0$ on this locus, \Cref{lem:response-locus-main} upgrades it to a
smooth codimension-one shell; otherwise it may fail to be a manifold and we
retain the response-locus terminology.

\item The sampled one-recovery-opportunity boundary-reset counter mode reaches
the memory-transition point exactly on the response locus
\begin{equation}\label{eq:shell-mem-general-main}
\Sigma_{\mathrm{mem}}=
\left\{x:
\frac{\lambda_{\mathrm{sw}}^2\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}
=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)
\right\}.
\end{equation}
This is the memory-transition response locus for the specified response model.
Wherever $dR_\Omega\neq0$ on this locus, it is a smooth codimension-one shell.
On the locus itself, the sampled record is exact $1$-dependent.

\item Equivalently, shell language is justified only after two checks on the
fixed calibrated response function: first the range check
\[
\gamma_\varphi,\gamma_{\mathrm{mem}}\in \operatorname{range}(R_\Omega),
\]
which guarantees nonempty response loci, and then the regularity check
$dR_\Omega\neq0$ at the point under discussion, which upgrades that response
locus to a shell. Neither check is generic KMS output; both are verifiable only
after the response model has been specified.
\end{enumerate}
\end{theorem}

\begin{proof}
For the ideal counter mode, Proposition~\ref{prop:ideal-main} says that the Parry benchmark is equivalent to $\Gamma\Delta\tau=\log\varphi$. Under the closure $\Gamma=\Gamma_+(x,\Omega)$ this becomes $\Gamma_+(x,\Omega)\Delta\tau=\log\varphi$, i.e. $R_\Omega(x)=\gamma_\varphi$ with the notation of~\eqref{eq:response-function-main}--\eqref{eq:target-rates-main}. By \Cref{prop:stationary-udw-response-main,ass:kms-response-main}, $\Gamma_+(x,\Omega)$ is the specified function $R_\Omega(x)$, so this condition is exactly \eqref{eq:shell-phi-general-main}. Conversely, any point satisfying \eqref{eq:shell-phi-general-main} has $R_\Omega(x)=\gamma_\varphi$ and hence realizes the ideal record benchmark.

For the sampled one-recovery-opportunity boundary-reset model,
\Cref{prop:recovery-main,lem:one-dep-criterion-main} say that the memory
transition is equivalent to $\Gamma=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$.
Under the same closure this becomes
$\Gamma_+(x,\Omega)=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$, i.e.
$R_\Omega(x)=\gamma_{\mathrm{mem}}$. With the same fixed calibrated response
function, this is exactly \eqref{eq:shell-mem-general-main}, and every point
satisfying that equation has an exact $1$-dependent sampled record in that
model. The regularity and possible nonexistence statements are then exactly
\Cref{lem:response-locus-main} applied to these two target rates. The
\Cref{lem:one-gap-kms-underdetermination-main} explains why the theorem is not
a derivation of $R_\Omega$ from KMS thermality alone: changing
$\mathcal S_x(\Omega)$ changes the loci while preserving one-gap detailed
balance.
\Cref{lem:one-gap-kms-underdetermination-main} is the reason this theorem is
only conditional on a fixed calibrated response function, and
\Cref{lem:response-locus-main} is exactly the range-plus-regularity check stated
in item~(iii).
\end{proof}

\Cref{thm:shells-general-main} is the main output of this section: after
\Cref{lem:one-gap-kms-underdetermination-main} and
\Cref{lem:response-locus-main}, the fixed calibrated function $R_\Omega$ turns
the record-selected thresholds into two response loci, and regularity
determines whether those level sets are smooth hypersurfaces. The remaining
results in this section record the calibrated homogeneous specialization and a
simple existence criterion on monotone spherical regions. Whenever that
specialization is used with nonconstant lapse, it is the abstract level-set
construction of \Cref{lem:abstract-homogeneous-specialization-main}, not a
separate exact detector-field model.

\begin{corollary}[Homogeneous specialization for a fixed calibrated response function $R_\Omega$]\label{cor:homogeneous-main}
Assume the sampled classical interval instrument of
\Cref{prop:instrument-kernels-main}, with independent Poisson attempts and
fresh interval ancillas, and impose only the modelling closure
$\Gamma=\Gamma_+(x,\Omega)$. Assume that
$\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$ on the region under study. For
this fixed calibrated homogeneous response function $R_\Omega$, both response
loci are pure lapse level sets. Their associated local inverse
temperatures are
\begin{equation}\label{eq:betaphi-betamem-main}
\beta_{\varphi}=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)\Delta\tau}{\log\varphi}\right),
\qquad
\beta_{\mathrm{mem}}=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}
{\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}\right),
\end{equation}
so that
\begin{equation}\label{eq:lapse-level-main}
\Sigma_{\varphi}=\{x:\beta_{\mathrm{loc}}(x)=\beta_{\varphi}\},
\qquad
\Sigma_{\mathrm{mem}}=\{x:\beta_{\mathrm{loc}}(x)=\beta_{\mathrm{mem}}\}.
\end{equation}
If $dN\neq0$ on one of these level sets, then \Cref{lem:response-locus-main}
identifies it as the corresponding shell. For nonconstant lapse this is an
abstract calibrated specialization in the sense of
\Cref{lem:abstract-homogeneous-specialization-main}.
\end{corollary}

\begin{proof}
With $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$, the two shell equations contain $x$ only through $\beta_\infty N(x)$. Solving the ideal-benchmark equation for $\beta_\infty N(x)$ gives $\beta_\varphi$, and solving the memory equation gives $\beta_{\mathrm{mem}}$. Hence the loci are the lapse level sets in~\eqref{eq:lapse-level-main}.
\end{proof}

For later shell-pair comparisons we use one fixed denominator orientation. If
$r_\varphi$ and $r_{\mathrm{mem}}$ are homogeneous ideal-benchmark and
memory-transition radii, set
\[
D_\varphi=e^{\beta_\infty\Omega N(r_\varphi)}-1,
\qquad
D_{\mathrm{mem}}=e^{\beta_\infty\Omega N(r_{\mathrm{mem}})}-1,
\qquad
Q_{\varphi/\mathrm{mem}}=\frac{D_\varphi}{D_{\mathrm{mem}}}.
\]
Thus $Q_{\varphi/\mathrm{mem}}$ means ideal-benchmark denominator divided by
memory-transition denominator. The cancellation identity and the reciprocal
target-rate convention are proved once in \Cref{thm:ratio-main}; this section
uses only the level-set formulas above.

\begin{proposition}[Existence on monotone spherical regions via the target-rate range condition for a fixed calibrated $R_\Omega$]\label{prop:existence-main}
Assume the sampled classical interval instrument of
\Cref{prop:instrument-kernels-main}, with independent Poisson attempts and
fresh interval ancillas, and impose only the modelling closure
$\Gamma=\Gamma_+(x,\Omega)$. Assume the calibrated homogeneous spectral-weight
regime in a static spherically symmetric region $r\in I$, and suppose
$N:I\to N(I)$ is $C^1$ and strictly monotone. For either threshold
$\star\in\{\varphi,\mathrm{mem}\}$, the corresponding response locus exists in $I$ if
and only if
\[
\frac{\beta_\star}{\beta_\infty}\in N(I),
\]
where $\beta_\varphi$ and $\beta_{\mathrm{mem}}$ are defined in~\eqref{eq:betaphi-betamem-main}. When it exists, its radius is unique and equals
\[
r_\star=N^{-1}\!\left(\frac{\beta_\star}{\beta_\infty}\right).
\]
If $N'(r_\star)\neq0$, this response locus is a regular spherical shell.
\end{proposition}

\begin{proof}
By \Cref{cor:homogeneous-main}, the shell equation is exactly
$\beta_\infty N(r)=\beta_\star$. Since $N$ is strictly monotone on $I$, it is
one-to-one onto its range. Hence the equation has a solution precisely when
$\beta_\star/\beta_\infty$ lies in $N(I)$, and the solution is then the
displayed inverse image. Thus shell existence on a monotone spherical region is
completely equivalent to the calibrated target lapse lying in the available
range of $N$. The nonzero derivative condition is the one-dimensional form of
the regular level-set condition.
\end{proof}

In this abstract calibrated homogeneous regime, both response loci are simply lapse level sets
selected by the two record thresholds, and they become shells when the
regularity condition holds. The same local response scale therefore supports
two different detector-selected response hypersurfaces whenever the range
condition in \Cref{prop:existence-main} is met. \Cref{fig:shell-radii}
illustrates the corresponding lapse values and proper-distance ratio in
Schwarzschild.

\begin{proposition}[Ordering for the fixed calibrated homogeneous response function $R_\Omega$]\label{prop:order-main}
Assume the sampled classical interval instrument of
\Cref{prop:instrument-kernels-main}, with independent Poisson attempts and
fresh interval ancillas, and impose only the modelling closure
$\Gamma=\Gamma_+(x,\Omega)$. Assume the calibrated homogeneous spectral-weight
regime and define
\[
\Gamma_{\varphi}=\frac{\log\varphi}{\Delta\tau}.
\]
Let $N_{\varphi}$ and $N_{\mathrm{mem}}$ denote the lapse values of the two
response loci $\Sigma_{\varphi}$ and $\Sigma_{\mathrm{mem}}$. Then
\begin{equation}\label{eq:order-main}
N_{\varphi}<N_{\mathrm{mem}}
\iff
\Gamma_{\varphi}>\Gamma_c(\kappa_{\mathrm r},\Delta\tau),
\end{equation}
with equality if and only if $\Gamma_{\varphi}=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$. Thus the ordering of the two homogeneous response loci is detector controlled; when the regularity condition holds, this is the shell ordering.
\end{proposition}

\begin{proof}
Both response loci are lapse level sets in the calibrated homogeneous
spectral-weight regime. Writing $A=\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)>0$,
the local excitation rate as a function of the lapse is
\[
\Gamma_+(N)=\frac{A}{e^{\beta_\infty\Omega N}-1}.
\]
For $\beta_\infty,\Omega,N>0$,
\[
\frac{d\Gamma_+}{dN}
=
-A\,\frac{\beta_\infty\Omega e^{\beta_\infty\Omega N}}
{(e^{\beta_\infty\Omega N}-1)^2}<0 .
\]
Hence the higher threshold rate corresponds exactly to the smaller lapse value, giving~\eqref{eq:order-main} and the equality statement.
\end{proof}
