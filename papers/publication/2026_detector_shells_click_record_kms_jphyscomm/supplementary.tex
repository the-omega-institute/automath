\paragraph{Related-work boundary.}
The nearby literature splits into four roles that the paper keeps separate.
On the stochastic side, the record-law results belong beside exact finite
dependence, hidden-Markov outputs, renewal processes, and dead-time counting
\cite{HoeffdingRobbins1948,Diananda1955,Blackwell1957,BaumPetrie1966,HolroydLiggett2016,Muller1973,CantorTeich1975,Hadfield2009,EisamanEtAl2011,Brandes2008}.
On the response side, the stationary UDW inputs and KMS ratio identities come
from the standard stationary-trajectory and regularized-detector literature
\cite{Letaw1981,Schlicht2004,Satz2007,LoukoSatz2006,LoukoSatz2008,FewsterJuarezLouko2016,JuarezAubryMoustos2022,PercheTherm2022,Bunney2024StationaryTraj,MistryPinzul2024DiamondTemp}.
On the measurement side, microscopic output-statistics and AQFT measurement
frameworks set the scope boundary
\cite{Glauber1963,SrinivasDavies1981,MandelWolf1995,Carmichael1993Book,WisemanMilburn2009,BarchielliGregoratti2009,Sorkin1993,FewsterVerch2020,Fewster2020,PoloGomezEtAl2022,PercheLocalized2023,PranziniKeskiVakkuri2025AQFT,Oeckl2025CausalMeasurement}.
Against those backgrounds, the theorem-level claim here remains narrow: one
explicit sampled boundary-reset record law is analyzed exactly, and the later
response-side statements are conditional level-set and inverse consequences
inside a fixed calibrated response model rather than new detector-dynamics or
general QFT-measurement theorems.

\section{Critical-rate branch and equal-rate limits}

Let
\[
p=e^{-\Gamma\Delta\tau},\qquad
c=e^{-\kappa_{\mathrm r}\Delta\tau},\qquad
s=\kappa_{\mathrm r}\Delta\tau,\qquad
x=\frac{\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}{\kappa_{\mathrm r}} .
\]
The non-removable critical-rate equation is
\[
\log x=s(x-1),
\qquad\text{equivalently}\qquad
x e^{-sx}=e^{-s}.
\]
Thus, away from the meeting point $s=1$,
\[
\Gamma_c(\kappa_{\mathrm r},\Delta\tau)
=-\frac{1}{\Delta\tau}
W_k\!\left(-\kappa_{\mathrm r}\Delta\tau\,e^{-\kappa_{\mathrm r}\Delta\tau}\right),
\qquad
k=\begin{cases}
-1,&0<s<1,\\
0,&s>1.
\end{cases}
\]
The other real branch gives the removable solution $x=1$ of the multiplied
equation. At $s=1$ the two branches meet and $\Gamma_c=\kappa_{\mathrm r}$.

When $\kappa_{\mathrm r}=\Gamma$ the distinct-root formulas for the renewal
law are interpreted by continuity. With $\alpha=\Gamma\Delta\tau$ and
$p=e^{-\alpha}$,
\[
\mathbb P(G=k)=
p^k\bigl[(1-p)+\alpha(k(1-p)-p)\bigr],
\qquad
\mathbb P(G\ge k)=p^k(1+\alpha k),
\]
and hence
\[
h_k=
\frac{(1-p)+\alpha(k(1-p)-p)}{1+\alpha k},
\qquad
(h_{k+1}-h_k)=\frac{\alpha^2p}{(1+\alpha k)(1+\alpha(k+1))}>0 .
\]
The hidden eigenvalue on the equal-rate line is
\[
\lambda_{\mathrm{hid}}=e^{-\Gamma\Delta\tau}(1-\Gamma\Delta\tau).
\]

For fast sampling at fixed $\kappa_{\mathrm r}$, the same critical equation
gives
\[
\Gamma_c(\kappa_{\mathrm r},\Delta\tau)
=
\frac{L+\log L+o(1)}{\Delta\tau},
\qquad
L=\log\frac{1}{\kappa_{\mathrm r}\Delta\tau},
\qquad
\Delta\tau\downarrow0 .
\]

\section{Finite-counting identity}

The main text proves the covariance tail
\[
\operatorname{Cov}(A_0,A_n)=\rho(a-\rho)\lambda_{\mathrm{hid}}^{\,n-1},\qquad n\ge1,
\]
with $a<\rho$ and $|\lambda_{\mathrm{hid}}|<1$. For the block count
$N_N=\sum_{t=0}^{N-1}A_t$, stationarity gives
\[
\operatorname{Var}(N_N)
=N\rho(1-\rho)+2\sum_{n=1}^{N-1}(N-n)\rho(a-\rho)\lambda_{\mathrm{hid}}^{\,n-1}.
\]
Evaluating the geometric sum yields the finite-$N$ identity
\[
\operatorname{Var}(N_N)
=
N\rho(1-\rho)
+
\frac{2\rho(a-\rho)}{(1-\lambda_{\mathrm{hid}})^2}
\bigl(N(1-\lambda_{\mathrm{hid}})-1+\lambda_{\mathrm{hid}}^N\bigr).
\]
Dividing by $\mathbb E N_N=N\rho$ and letting $N\to\infty$ gives the
sub-Poisson Fano limit displayed in the main text.

\section{Ambient residual graph for ratio inversion}

For ambient algebraic radius bookkeeping, write
\[
\mathsf e=(\eta,c),\qquad c=(c_1,c_2),
\]
and define
\begin{equation}\label{eq:two-mode-phi-main}
\Phi_m(\theta,\mathsf e)=R_m(\theta;\eta)-c_m,\qquad
\Phi=(\Phi_1,\Phi_2).
\end{equation}
For the physical detector settings, $c=C$ and
\[
\Phi(\theta,(\eta,C))=H(\theta,\eta).
\]
Exact generated data at the true parameter are denoted
\[
\eta_*=(r_{\varphi,1},r_{{\mathrm{mem}},1},
        r_{\varphi,2},r_{{\mathrm{mem}},2}),
\qquad
\mathsf e_*=(\eta_*,C),
\]
and on the fixed-detector slice we write
\[
\Phi_C(\theta,\eta)=\Phi(\theta,(\eta,C))=H(\theta,\eta).
\]

\begin{lemma}[Data-dependent residual graph]
Let $U_0\subset\RR^2$ and $W_0\subset\RR^q$ be open sets and let
$G:U_0\times W_0\to\RR^2$ be $C^1$. If $G(\theta_0,w_0)=0$ and
$D_\theta G(\theta_0,w_0)$ is nonsingular, then there are neighbourhoods
$U\subset U_0$ of $\theta_0$ and $W\subset W_0$ of $w_0$ and a unique
$C^1$ map $\Psi:W\to U$ such that $\Psi(w_0)=\theta_0$ and
\[
G(\Psi(w),w)=0\qquad(w\in W).
\]
After shrinking $U$ and $W$ if necessary, every solution of
$G(\theta,w)=0$ in $U\times W$ lies on this graph. Freezing the data first
would give only local uniqueness of a zero of $\theta\mapsto G(\theta,w_0)$;
the data-to-parameter map is the graph obtained with $w$ retained as the
parameter.
\end{lemma}

\begin{proof}
This is the parameter-dependent implicit function theorem \cite{KrantzParks2002},
applied with $\theta$ as the unknown and $w$ as the parameter.
\end{proof}

\begin{corollary}[Ambient residual graph, not a shell-data inverse]\label{cor:ambient-ratio-graph-app-main}
In the two-mode inverse notation of the main text, let
\[
\mathcal E\subset\mathcal D\times(0,\infty)^2
\]
be the open algebraic data set on which \eqref{eq:two-mode-phi-main} is
defined, and fix $(\theta_*,\mathsf e_*)$ with $\Phi(\theta_*,\mathsf e_*)=0$.
Assume $D_\theta\Phi(\theta_*,\mathsf e_*)$ is nonsingular.

Then there are neighborhoods $U\subset\Theta$ of $\theta_*$ and
$W\subset\mathcal E$ of $\mathsf e_*$ and a unique $C^1$ map
\[
\Psi:W\to U,\qquad \Psi(\mathsf e_*)=\theta_*,
\]
such that
\[
\Phi(\Psi(\mathsf e),\mathsf e)=0\qquad(\mathsf e\in W).
\]
After shrinking $U$ and $W$, every solution in $U\times W$ lies on this graph.
On the fixed-detector slice
\[
W_C=\{\eta\in\mathcal D:(\eta,C)\in W\},
\qquad
\Psi_C(\eta)=\Psi((\eta,C)),
\]
there is an identity
\[
\Phi_C(\Psi_C(\eta),\eta)=0\qquad(\eta\in W_C),
\]
and if $\theta\in U$ and $\Phi_C(\theta,\eta)=0$ for some $\eta\in W_C$, then
$\theta=\Psi_C(\eta)$. This is only an ambient residual graph for the two ratio
equations. It does not assert that the radius quadruples in $W_C$ are physical
common-core shell data.
\end{corollary}

\begin{proof}
Apply the preceding lemma with $G=\Phi$ and $w=\mathsf e$.
\end{proof}

\begin{corollary}[Fixed-record local full-shell subset]\label{thm:full-shell-subset-main}
In the setting of \Cref{cor:ambient-ratio-graph-app-main}, shrink $W_C$ if
necessary so that the four radii remain in the coordinate range where the
background family and the response factors are defined. For $\eta\in W_C$ set
\[
\theta(\eta)=\Psi_C(\eta),
\]
and define the two mode amplitudes along the ratio graph by
\[
\mathcal A_m(\eta)=
\mathcal A_{\varphi,m}(\theta(\eta);\eta)
=
\mathcal A_{{\mathrm{mem}},m}(\theta(\eta);\eta),
\qquad m=1,2,
\]
where the equality is the within-mode equality forced by
$\Phi_C(\theta(\eta),\eta)=0$. Let
\[
E_C(\eta)=\mathcal A_1(\eta)-\mathcal A_2(\eta).
\]
Then the nearby full common-core shell records in the fixed-detector slice are
exactly
\begin{equation}\label{eq:full-shell-subset-main}
\mathcal P_C
=
\{\eta\in W_C:E_C(\eta)=0,\ \mathcal A_1(\eta)>0,
\text{ and the four radii are exterior-admissible}\}.
\end{equation}
The function $E_C$ is $C^1$. In particular, if the exact datum $\eta_*$ is
exterior-admissible and
\[
dE_C(\eta_*)\neq0
\]
as a covector on the ambient radius-record space $W_C$, then
$\mathcal P_C$ is, near $\eta_*$, a nonempty $C^1$ codimension-one submanifold
of $W_C$. If this transversality fails, the theorem still gives the exact local
zero-set characterization \eqref{eq:full-shell-subset-main}, but no manifold
structure is asserted.
\end{corollary}

\begin{proof}
For every $\eta\in W_C$, the defining property of the ambient graph gives
\[
\Phi_C(\theta(\eta),\eta)=0,
\]
which is the pair of physical ratio equations $H_m(\theta(\eta),\eta)=0$.
\Cref{lem:ratio-admissibility-main} says that each of these two ratio equations
is equivalent to equality of the ideal-benchmark and memory amplitude estimates
within the same mode. Hence the displayed definitions of
$\mathcal A_m(\eta)$ are unambiguous along the graph.

A full common-core shell record requires exactly one positive homogeneous
hardware prefactor for all four undivided shell equations, together with the
exterior-range restrictions. Along the ratio graph the two estimates within
each mode are already equal, so the only remaining cross-mode amplitude
condition is
\[
\mathcal A_1(\eta)=\mathcal A_2(\eta),
\]
which is $E_C(\eta)=0$. Positivity of the common amplitude and exterior
admissibility are the other two conditions in
\Cref{lem:ratio-admissibility-main}. This proves the set identity
\eqref{eq:full-shell-subset-main}.

The maps $\Psi_C$, $N$, and $\beta_\infty$ are $C^1$ on the neighbourhoods in
question, and the amplitude estimates are compositions of $C^1$ functions with
the smooth exponential function. Therefore $\mathcal A_m$ and $E_C$ are $C^1$.
At exact generated data, the common-core hypothesis gives
$E_C(\eta_*)=0$, so $\eta_*\in\mathcal P_C$ after shrinking to preserve
exterior admissibility and positivity. If $dE_C(\eta_*)\neq0$, the regular
level-set theorem makes $E_C^{-1}(0)$ a $C^1$ codimension-one submanifold of
$W_C$ near $\eta_*$. Positivity and exterior admissibility are open conditions,
so intersecting with them leaves the same local submanifold structure. Without
the nonvanishing derivative, only the proved zero-set characterization remains.
\end{proof}

\begin{corollary}[Restriction to full common-core shell records]\label{cor:full-shell-restriction-main}
In the setting of \Cref{thm:full-shell-subset-main}, let
$\mathcal P_C\subset W_C$ be the full common-core subset characterized in
\eqref{eq:full-shell-subset-main}.
For every $\eta\in\mathcal P_C$, the parameter $\Psi_C(\eta)$ is the unique
local background parameter in $U$ compatible with the corresponding full
common-core shell data, and the common amplitude is obtained by evaluating any
one of the four equal undivided amplitude estimates. Conversely, if
$\eta\in W_C$ is compatible with full common-core shell equations at some
$\theta\in U$, then $\theta=\Psi_C(\eta)$ and $\eta\in\mathcal P_C$.
\end{corollary}

\begin{proof}
If $\eta\in\mathcal P_C$, then by definition the ratio equations hold at
$\Psi_C(\eta)$ and the undivided amplitude, positivity, and exterior-range
checks also hold there. Hence the full shell equations are satisfied with one
positive common amplitude. If another $\theta\in U$ satisfied the same full
shell equations, division within each mode would give
$\Phi_C(\theta,\eta)=0$, so the uniqueness part of
\Cref{cor:ambient-ratio-graph-app-main} gives $\theta=\Psi_C(\eta)$. The converse is
the same argument in reverse: full common-core shell equations imply the ratio
equations and the admissibility checks, hence the local root is $\Psi_C(\eta)$
and $\eta$ belongs to $\mathcal P_C$.
\end{proof}

\begin{remark}[Ratio admissibility and amplitude consistency]\label{rem:amplitude-consistency-main}
\Cref{thm:two-mode-main},
\Cref{cor:ambient-ratio-graph-app-main}, and
\Cref{lem:ratio-admissibility-main} separate checks that are sometimes
conflated. The fixed-radius map $\mathcal R_{\eta_*}$ contains the two
ratio-level observables and no unknown common amplitude; the main-text theorem
is the deterministic local inverse branch for those two values together with the fixed-record test
for whether the recovered parameter still yields one positive common amplitude.
If the radius record is also varied, the only general statement is the ambient residual graph in
\Cref{cor:ambient-ratio-graph-app-main}. \Cref{cor:full-shell-restriction-main}
is the step that restricts that graph to full shell data. A candidate
ratio-level solution is a full common-core solution only if the four undivided
amplitude estimates
\[
\mathcal A_{\varphi,1}(\theta;\eta),\quad
\mathcal A_{{\mathrm{mem}},1}(\theta;\eta),\quad
\mathcal A_{\varphi,2}(\theta;\eta),\quad
\mathcal A_{{\mathrm{mem}},2}(\theta;\eta)
\]
are positive and equal to one common value in the exact model. Thus
amplitude-free ratio inversion means that the amplitude is not needed to solve
the two ratio equations; it does not mean that arbitrary radius records
automatically come from one physical detector core.
\end{remark}

\[

\]

\[

Together with the small-gap linearized proper-distance ratio from

\begin{equation}\label{eq:area-ratio-main}

under the hypotheses of \Cref{thm:horizon-main}.

\begin{proposition}[Abstract calibrated toy-model corollary: Schwarzschild algebra]\label{prop:schw-main}
Assume a fixed calibrated $R_\Omega$. The homogeneous Schwarzschild toy
specialization yields the following algebraic identities.
Work under the standing convention for the fixed-radius inverse results, and
assume the calibrated homogeneous spectral-weight regime, so that the fixed
calibrated response function $R_\Omega$ is homogeneous. Define the detector
constants
\begin{equation}\label{eq:beta-constants-main}
\beta_{\varphi}(D)=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)\Delta\tau}{\log\varphi}\right),
\qquad
\beta_{\mathrm{mem}}(D)=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}
{\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}\right).
\end{equation}
If model-admissible shell data generated by the same response amplitude occur
at distinct radii $r_{\varphi}\neq r_{\mathrm{mem}}$ in Schwarzschild, then
\begin{equation}\label{eq:beta-infty-main}
\beta_\infty^2=
\frac{\beta_{\varphi}(D)^2r_{\varphi}-\beta_{\mathrm{mem}}(D)^2r_{\mathrm{mem}}}
{r_{\varphi}-r_{\mathrm{mem}}},
\end{equation}
and
\begin{equation}\label{eq:mass-main}
M=
\frac{r_{\varphi}r_{\mathrm{mem}}\bigl(\beta_{\varphi}(D)^2-\beta_{\mathrm{mem}}(D)^2\bigr)}
{2\bigl(\beta_{\varphi}(D)^2r_{\varphi}-\beta_{\mathrm{mem}}(D)^2r_{\mathrm{mem}}\bigr)}.
\end{equation}
\end{proposition}

\begin{proof}
In Schwarzschild,
\[
N(r)^2=1-\frac{2M}{r}.
\]
The two homogeneous shell equations are
\[
\frac{\beta_{\varphi}(D)^2}{\beta_\infty^2}=1-\frac{2M}{r_\varphi},
\qquad
\frac{\beta_{\mathrm{mem}}(D)^2}{\beta_\infty^2}=1-\frac{2M}{r_{\mathrm{mem}}}.
\]
Equivalently,
\[
\beta_\infty^2-\frac{2M\beta_\infty^2}{r_\varphi}=\beta_{\varphi}(D)^2,
\qquad
\beta_\infty^2-\frac{2M\beta_\infty^2}{r_{\mathrm{mem}}}=\beta_{\mathrm{mem}}(D)^2.
\]
This is a nonsingular linear system for $\beta_\infty^2$ and
$2M\beta_\infty^2$ when $r_\varphi\neq r_{\mathrm{mem}}$. Solving it gives
\eqref{eq:beta-infty-main} and \eqref{eq:mass-main}. The model-admissibility
hypothesis is what guarantees that the displayed right-hand sides are positive
and correspond to a shell outside the horizon; arbitrary pairs of radii and
detector constants need not do so.
\end{proof}

Proposition~\ref{prop:schw-main} is exact algebra only within this toy
calibrated model. It is not amplitude-free, because $\beta_\varphi(D)$ and
$\beta_{\mathrm{mem}}(D)$ still contain the common response amplitude.

\begin{corollary}[Abstract calibrated toy-model corollary: Schwarzschild branch inverse and residual check]\label{cor:schwarzschild-branch-main}
Assume a fixed calibrated $R_\Omega$. The Schwarzschild branch reduction has
the following one-parameter inverse and residual-check form inside the toy
calibrated model.
Assume the same fixed calibrated homogeneous response-model setting as in
\Cref{thm:two-mode-main}, and assume further that the background family
contains a Hartle--Hawking Schwarzschild branch
\[
\iota_{\mathrm{Schw}}:I_M\to\Theta,
\qquad
\iota_{\mathrm{Schw}}(M)=(8\pi M,M),
\]
defined on an open interval $I_M\subset(0,\infty)$. Fix an exact radius record
$\eta_*$ produced at some $M_*\in I_M$ by admissible full common-core shell
data on that branch with one positive shared amplitude $\mathcal A_*>0$, so
that
\[
\theta_*=\iota_{\mathrm{Schw}}(M_*),
\qquad
\mathcal R_{\eta_*}(\theta_*)=C
\]
and
\[
\mathcal A_{\varphi,1}(\theta_*;\eta_*)
=\mathcal A_{{\mathrm{mem}},1}(\theta_*;\eta_*)
=\mathcal A_{\varphi,2}(\theta_*;\eta_*)
=\mathcal A_{{\mathrm{mem}},2}(\theta_*;\eta_*)
=\mathcal A_*>0,
\]
with all four radii lying in the exterior coordinate range at $\theta_*$.
Choose one detector mode $j\in\{1,2\}$ as the estimation mode and let
$k\in\{1,2\}\setminus\{j\}$ be the remaining mode. Define the reduced
Schwarzschild branch ratio functions
\[
G_{m,\eta_*}(M)=R_m(\iota_{\mathrm{Schw}}(M);\eta_*),
\qquad m=1,2.
\]
Assume that the fixed radii in $\eta_*$ remain exterior-admissible on a
neighbourhood of $M_*$ in $I_M$ and that
\begin{equation}\label{eq:schwarzschild-branch-derivative-main}
\frac{d}{dM}G_{j,\eta_*}(M_*)\neq0.
\end{equation}
Then there are neighbourhoods $I_*\subset I_M$ of $M_*$ and
$Y_j\subset(0,\infty)$ of $C_j$, together with a unique $C^1$ inverse branch
\[
\mathcal J_{\eta_*,j}:Y_j\to I_*,
\qquad
\mathcal J_{\eta_*,j}(C_j)=M_*,
\]
such that
\[
G_{j,\eta_*}(\mathcal J_{\eta_*,j}(y_j))=y_j
\qquad(y_j\in Y_j).
\]
Hence nearby estimated values of the single ratio coordinate $y_j$ recover the
physical Schwarzschild parameter through
\[
\widehat M=\mathcal J_{\eta_*,j}(y_j),
\qquad
\widehat\beta_\infty=8\pi\,\widehat M.
\]
The second ratio does not provide an independent free coordinate on the
physical Schwarzschild branch; instead it gives the residual consistency check
\begin{equation}\label{eq:schwarzschild-branch-residual-main}
\mathfrak c_{\eta_*,k|j}(y_j,y_k)
=G_{k,\eta_*}(\mathcal J_{\eta_*,j}(y_j))-y_k.
\end{equation}
At the exact physical record one has
\[
\mathfrak c_{\eta_*,k|j}(C_j,C_k)=0.
\]
Moreover, if exact full fixed-setting shell data occur at the same detector
settings for a pair $(y_j,y_k)$ with $y_j\in Y_j$, then
\[
y_j=C_j,\qquad y_k=C_k,
\qquad
\mathfrak c_{\eta_*,k|j}(y_j,y_k)=0,
\]
by \Cref{thm:two-mode-main}. Thus on the physical
Hartle--Hawking Schwarzschild branch the inverse statement is genuinely
one-parameter, and the remaining ratio is a residual compatibility test rather
than a second independent inverse coordinate.

If, in addition, a scalar ratio estimator satisfies
\[
\sqrt{N}\,(\widehat y_{j,N}-C_j)\Longrightarrow \mathcal N(0,\sigma_j^2),
\]
then with
$
\widehat M_N=\mathcal J_{\eta_*,j}(\widehat y_{j,N})
$
and
$
\widehat\beta_{\infty,N}=8\pi\widehat M_N
$
one has
\[
\sqrt{N}\,(\widehat M_N-M_*)
\Longrightarrow
\mathcal N\!\left(0,
\frac{\sigma_j^2}{\bigl(G_{j,\eta_*}'(M_*)\bigr)^2}\right),
\]
and
\[
\sqrt{N}\,(\widehat\beta_{\infty,N}-8\pi M_*)
\Longrightarrow
\mathcal N\!\left(0,
\frac{(8\pi)^2\sigma_j^2}{\bigl(G_{j,\eta_*}'(M_*)\bigr)^2}\right).
\]
\end{corollary}

\begin{proof}
Because $\iota_{\mathrm{Schw}}$, $\beta_\infty$, $N$, and the exponential map
are $C^1$, each $G_{m,\eta_*}$ is a $C^1$ function of $M$ on the
exterior-admissible branch interval. The exact shell-data hypothesis gives
$G_{j,\eta_*}(M_*)=C_j$. The nonvanishing derivative in
\eqref{eq:schwarzschild-branch-derivative-main} is precisely the one-variable
inverse-function condition, so the ordinary one-dimensional inverse function
theorem yields the neighbourhoods $I_*,Y_j$ and the unique $C^1$ inverse
branch $\mathcal J_{\eta_*,j}$.

The residual formula \eqref{eq:schwarzschild-branch-residual-main} is the
remaining ratio equation evaluated along that recovered branch. At the exact
record it vanishes because $G_{k,\eta_*}(M_*)=C_k$. If exact full fixed-setting
shell data were to occur at the same detector settings for some
$(y_j,y_k)$, then the recovered parameter would be
$\theta=\iota_{\mathrm{Schw}}(\mathcal J_{\eta_*,j}(y_j))$ and
\Cref{thm:two-mode-main} would force the full ratio datum to
equal the unique fixed target $C$, proving the stated necessity of
$y_j=C_j,y_k=C_k$.

The final two limits are the one-dimensional delta method applied to the $C^1$
map $\mathcal J_{\eta_*,j}$ at $C_j$, together with the linear relation
$\beta_\infty=8\pi M$ on the Hartle--Hawking Schwarzschild branch.
\end{proof}

\subsection{Deterministic numerical reproduction recipe}\label{subsec:numerical-recipe}

The numerical items in the manuscript are deterministic evaluations of the
closed formulas already proved in the main text. No stochastic simulation or
external fit is used. This subsection records a paper-local recipe for
reproducing
\Cref{fig:hazard-cov,fig:spectrum-fano,fig:shell-radii,tab:worked}
directly from those formulas. Throughout, use a
root-solve tolerance of $10^{-12}$ and round displayed table entries to six
decimal places.

\paragraph{Step 1: solve the critical rate $\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$.}
Given positive $\kappa_{\mathrm r}$ and $\Delta\tau$, set
\[
s=\kappa_{\mathrm r}\Delta\tau,
\qquad
H_s(x)=\log x-s(x-1).
\]
If $|s-1|<10^{-12}$, use the equal-rate value
$\Gamma_c(\kappa_{\mathrm r},\Delta\tau)=\kappa_{\mathrm r}$. Otherwise solve
$H_s(x)=0$ for the non-removable root $x_c\neq1$ from
\Cref{lem:critical-rate-main}. A deterministic bisection scheme is:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.2em,itemsep=0.25em]
\item If $0<s<1$, set $x_{\mathrm{lo}}=1+10^{-12}$ and
$x_{\mathrm{hi}}=2$. While $H_s(x_{\mathrm{hi}})\ge0$, replace
$x_{\mathrm{hi}}$ by $2x_{\mathrm{hi}}$.
\item If $s>1$, set $x_{\mathrm{hi}}=1-10^{-12}$ and
$x_{\mathrm{lo}}=1/2$. While $H_s(x_{\mathrm{lo}})\ge0$, replace
$x_{\mathrm{lo}}$ by $x_{\mathrm{lo}}/2$.
\item Bisect the resulting bracket until both the bracket width and the
residual in \eqref{eq:gamma-c-main} are below $10^{-12}$, and then set
\[
\Gamma_c(\kappa_{\mathrm r},\Delta\tau)=\kappa_{\mathrm r}x_c.
\]
\end{enumerate}

The following Schwarzschild calculations are not claimed to arise from an
independently derived exact UDW response with

abstract calibrated homogeneous response function

Use the same abstract homogeneous Schwarzschild-lapse specialization as in the
table caption:
\[
r_{\mathrm s}=1,\qquad
\beta_\infty=4\pi,\qquad
\Omega=1,\qquad
\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)=2,

with detector modes
\[
D_1:\ (\Delta\tau,\kappa_{\mathrm r})=(0.5,8.0),
\qquad
D_2:\ (\Delta\tau,\kappa_{\mathrm r})=(1.2,3.0).
\]
Then:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.2em,itemsep=0.25em]
\item For each mode $m$, compute $\Gamma_c$ by Step~1,
