\section{Ideal dead time and the Parry threshold}\label{sec:supp-ideal}

\begin{proposition}\label{prop:supp-ideal-markov}
In the ideal one-step dead-time detector, the sampled output is a stationary Markov chain on the golden-mean shift with transition matrix
\[
P_\beta=
\begin{pmatrix}
 p_\beta & 1-p_\beta \\
 1 & 0
\end{pmatrix},
\qquad p_\beta=e^{-\Gamma_+\Delta\tau}.
\]
\end{proposition}

\begin{proof}
If $A_n=1$, the detector is refractory during the next sampling interval, so $A_{n+1}=0$ with probability one. Hence the block $11$ is forbidden. If $A_n=0$, the detector is ready at the next step, and the next output is $1$ with probability $q_\beta=1-p_\beta$ and $0$ with probability $p_\beta$. These are exactly the transition probabilities above.
\end{proof}

The Parry measure on the golden-mean shift has transition probabilities
\[
P(0,0)=\varphi^{-1},\qquad P(0,1)=\varphi^{-2},\qquad P(1,0)=1,
\]
and is the unique measure of maximal entropy on that shift \cite{Parry1964,LindMarcus1995}. Matching the detector law to the Parry matrix yields the ideal threshold.

\begin{corollary}\label{cor:supp-parry-threshold}
For the ideal dead-time detector,
\[
\text{exact symbolic thermality}
\iff p_\beta=\varphi^{-1}
\iff \Gamma_+\Delta\tau=\log\varphi.
\]
\end{corollary}

\begin{proof}
By \cref{prop:supp-ideal-markov}, the only free transition probability of the sampled output is $P_\beta(0,0)=p_\beta$. Matching the Parry matrix therefore requires $p_\beta=\varphi^{-1}$, or equivalently $\Gamma_+\Delta\tau=\log\varphi$.
\end{proof}

\section{Finite recovery as a renewal process}\label{sec:supp-renewal}

Write
\[
p=e^{-\Gamma\Delta\tau},\qquad q=1-p,\qquad c=e^{-\kappa_{\mathrm r}\Delta\tau},
\]
and define
\begin{equation}\label{eq:supp-a-b}
a=\int_0^{\Delta\tau}\kappa_{\mathrm r}e^{-\kappa_{\mathrm r}s}\bigl(1-e^{-\Gamma(\Delta\tau-s)}\bigr)\,ds,
\qquad
b=\int_0^{\Delta\tau}\kappa_{\mathrm r}e^{-\kappa_{\mathrm r}s}e^{-\Gamma(\Delta\tau-s)}\,ds.
\end{equation}
The output-labelled kernels are
\begin{equation}\label{eq:supp-T0T1}
T_0=
\begin{pmatrix}
 p & 0 \\
 b & c
\end{pmatrix},
\qquad
T_1=
\begin{pmatrix}
 0 & q \\
 0 & a
\end{pmatrix},
\end{equation}
with rows and columns indexed by the hidden states $(R,D)$.

\begin{proposition}\label{prop:supp-renewal}
Every observed click resets the next sampling interval to the hidden state $D$. Consequently the visible record is a renewal process and the zero-run length $G$ between successive clicks satisfies
\begin{equation}\label{eq:supp-gap-law}
\PP(G=k)=g_k=e_D^{\top}T_0^kT_1\mathbf 1,\qquad k\ge0,
\end{equation}
where $e_D=(0,1)^{\top}$ and $\mathbf 1=(1,1)^{\top}$.

If $\kappa_{\mathrm r}\neq\Gamma$, then
\begin{equation}\label{eq:supp-a-b-closed}
a=\frac{\kappa_{\mathrm r}(1-p)-\Gamma(1-c)}{\kappa_{\mathrm r}-\Gamma},
\qquad
b=\frac{\kappa_{\mathrm r}(p-c)}{\kappa_{\mathrm r}-\Gamma},
\end{equation}
and
\begin{equation}\label{eq:supp-gap-closed}
g_k=
\frac{\kappa_{\mathrm r}(1-p)p^k-\Gamma(1-c)c^k}{\kappa_{\mathrm r}-\Gamma},
\qquad k\ge0.
\end{equation}
\end{proposition}

\begin{proof}
Every observed click places the detector in the refractory state for the next interval, independently of the earlier history. Hence the post-click hidden state is deterministic, and the strong Markov property of the hidden two-state chain makes click times regeneration times. The resulting zero-run lengths are therefore i.i.d., and a gap of length $k$ consists of $k$ successive output-$0$ transitions followed by one output-$1$ transition, giving \cref{eq:supp-gap-law}.

To evaluate it explicitly, note that $T_0$ is lower triangular. Direct integration yields \cref{eq:supp-a-b-closed}. Since
\[
T_0^k=
\begin{pmatrix}
 p^k & 0 \\
 b\,\dfrac{p^k-c^k}{p-c} & c^k
\end{pmatrix},
\]
substituting into \cref{eq:supp-gap-law} gives
\[
g_k=q b\frac{p^k-c^k}{p-c}+ac^k,
\]
which simplifies to \cref{eq:supp-gap-closed}.
\end{proof}

\begin{remark}[Equal-rate limit]\label{rem:supp-equal-rate}
When $\kappa_{\mathrm r}=\Gamma$, equations~\eqref{eq:supp-a-b-closed}--\eqref{eq:supp-gap-closed} are $0/0$ forms. L'H\^opital's rule applied to~\eqref{eq:supp-gap-closed} in $\kappa_{\mathrm r}$ at $\kappa_{\mathrm r}=\Gamma$ gives
\[
g_k\big|_{\kappa_{\mathrm r}=\Gamma} = p^k\bigl[(1-p)+\Gamma\Delta\tau\bigl(k(1-p)-p\bigr)\bigr],
\]
and analogously for $a$, $b$, $\lambda$. All subsequent results extend by continuity to $\kappa_{\mathrm r}=\Gamma$.
\end{remark}

\begin{proposition}\label{prop:supp-hazard}
Assume $\kappa_{\mathrm r}\neq\Gamma$. Then
\begin{equation}\label{eq:supp-hazard}
h_k=\frac{g_k}{\sum_{j\ge k}g_j}
=\frac{\kappa_{\mathrm r}(1-p)p^k-\Gamma(1-c)c^k}{\kappa_{\mathrm r}p^k-\Gamma c^k},
\qquad k\ge0,
\end{equation}
and
\begin{equation}\label{eq:supp-hazard-diff}
h_{k+1}-h_k=
\frac{\kappa_{\mathrm r}\Gamma(p-c)^2p^kc^k}
{(\kappa_{\mathrm r}p^k-\Gamma c^k)(\kappa_{\mathrm r}p^{k+1}-\Gamma c^{k+1})}>0.
\end{equation}
Hence the visible process is not an $r$-step Markov chain for any finite $r$.
\end{proposition}

\begin{proof}
The tail of the gap law is
\[
\sum_{j\ge k}g_j=
\frac{\kappa_{\mathrm r}p^k-\Gamma c^k}{\kappa_{\mathrm r}-\Gamma}.
\]
This gives \cref{eq:supp-hazard}. Direct subtraction yields \cref{eq:supp-hazard-diff}. If the visible process were finite-order Markov, the next-click probability after a long run of zeros would eventually depend only on a fixed finite suffix, hence would be constant on sufficiently long zero runs. The strict monotonicity of $h_k$ excludes that possibility.
\end{proof}

\begin{proposition}\label{prop:supp-cov}
Let $(A_n)_{n\in\mathbb Z}$ be any stationary version of the finite-recovery record and let $\rho=\PP(A_0=1)$. Define
\begin{equation}\label{eq:supp-lambda}
\lambda=
\frac{\kappa_{\mathrm r}e^{-\kappa_{\mathrm r}\Delta\tau}-\Gamma e^{-\Gamma\Delta\tau}}
{\kappa_{\mathrm r}-\Gamma}.
\end{equation}
Then
\begin{equation}\label{eq:supp-cov}
\operatorname{Cov}(A_0,A_n)=\rho(a-\rho)\lambda^{n-1},\qquad n\ge1.
\end{equation}
Moreover $a<\rho<1-p$, so $\operatorname{Cov}(A_0,A_1)<0$. The critical line
\begin{equation}\label{eq:supp-critical}
\Delta\tau=\frac{\log\kappa_{\mathrm r}-\log\Gamma}{\kappa_{\mathrm r}-\Gamma}
\end{equation}
is the unique locus at which $\lambda=0$. At that locus, $\operatorname{Cov}(A_0,A_n)=0$ for all $n\ge2$.
\end{proposition}

\begin{proof}
Let
\[
P=T_0+T_1=
\begin{pmatrix}
 p & 1-p\\
 b & 1-b
\end{pmatrix},
\qquad
u=T_1\mathbf 1=\binom{1-p}{a}.
\]
The nontrivial eigenvalue of the two-state matrix $P$ is $\lambda=p-b$. If $\pi$ is the stationary distribution of $P$, then $\rho=\pi u$. Whenever $A_0=1$, the next hidden state is deterministically $D$, so
\[
\PP(A_0=1,A_n=1)=\rho\, e_D^\top P^{n-1}u,\qquad n\ge1.
\]
The two-state spectral decomposition gives
\[
P^{n-1}u=\rho\mathbf 1+\lambda^{n-1}(u-\rho\mathbf 1).
\]
Evaluating at $e_D$ yields
\[
e_D^\top P^{n-1}u=\rho+\lambda^{n-1}(a-\rho),
\]
which proves \cref{eq:supp-cov}. The inequality $a<\rho<1-p$ follows because $\rho$ is a strict convex combination of $a$ and the ready-state click probability $1-p$. Solving $\lambda=0$ gives \cref{eq:supp-critical}.
\end{proof}

\begin{unnumberedlemma}[Uniqueness and sign of the critical line]
For fixed $\kappa_{\mathrm r},\Delta\tau>0$, the equation $\lambda(\Gamma)=0$ has exactly one positive solution $\Gamma_c$. It is characterized by
\[
\Delta\tau=\frac{\log\kappa_{\mathrm r}-\log\Gamma_c}{\kappa_{\mathrm r}-\Gamma_c},
\]
with the quotient interpreted by continuity when $\Gamma_c=\kappa_{\mathrm r}$. In addition,
\[
\lambda(\Gamma)>0\quad\text{for }0<\Gamma<\Gamma_c,
\qquad
\lambda(\Gamma)<0\quad\text{for }\Gamma>\Gamma_c .
\]
\end{unnumberedlemma}

\begin{proof}
Set $s=\kappa_{\mathrm r}\Delta\tau$ and $x=\Gamma/\kappa_{\mathrm r}$. For $x\neq1$,
\[
\lambda(\Gamma)
=
\frac{e^{-s}-x e^{-sx}}{1-x}.
\]
The equation $\lambda=0$ is therefore equivalent to
\[
H_s(x):=\log x-s(x-1)=0,
\]
apart from the removable root $x=1$ introduced by the quotient. The equal-rate value is $\lambda=e^{-s}(1-s)$, so $x=1$ is critical only when $s=1$. Since $H_s''(x)=-x^{-2}<0$, $H_s$ is strictly concave. If $s<1$, the unique non-removable root lies in $(1,\infty)$; if $s>1$, it lies in $(0,1)$; if $s=1$, the equal-rate point is the unique critical solution. Rewriting $H_s(x)=0$ gives the displayed equation for $\Gamma_c$. The sign of $\lambda$ on either side follows from the same concavity analysis and the factor $1-x$ in the denominator.
\end{proof}

\begin{lemma}[Kernel factorization at the critical line]\label{lem:supp-1dep-factorization}
At the critical line \cref{eq:supp-critical}, let $\pi$ be the stationary row distribution of $P=T_0+T_1$. Then $P=\mathbf 1\pi$. Consequently, for finite words $u$ and $v$ supported respectively on a past block ending at time $0$ and a future block beginning at time $2$,
\[
\PP(A_{-m}^0=u,\ A_2^{2+\ell}=v)
=\PP(A_{-m}^0=u)\PP(A_2^{2+\ell}=v).
\]
\end{lemma}

\begin{proof}
At the critical line, the hidden-state transition matrix has rank one because its nontrivial eigenvalue vanishes, so $P=\mathbf 1\pi$. The labelled-kernel probability of the two words, with the output at time $1$ unobserved, is
\[
\PP(A_{-m}^0=u,\ A_2^{2+\ell}=v)
=\pi T_u P T_v\mathbf 1
=\pi T_u\mathbf 1\,\pi T_v\mathbf 1
=\PP(A_{-m}^0=u)\PP(A_2^{2+\ell}=v),
\]
where $T_u$ and $T_v$ are the corresponding products of output-labelled kernels.
\end{proof}

\begin{corollary}\label{cor:supp-1dep}
At the critical line \cref{eq:supp-critical}, the stationary finite-recovery output is exact $1$-dependent:
\[
\sigma(A_i:i\le0)\ \perp\ \sigma(A_j:j\ge2).
\]
\end{corollary}

\begin{proof}
Cylinder events generate the two sigma algebras, and \cref{lem:supp-1dep-factorization} factorizes all such cylinders. This yields independence.
\end{proof}

\begin{corollary}\label{cor:supp-no-exact}
No finite-recovery detector with positive parameters $(\Gamma,\kappa_{\mathrm r},\Delta\tau)$ realizes exact symbolic thermality within the full-support renewal class.
\end{corollary}

\begin{proof}
The full-support Bernoulli benchmark would require constant hazard. By \cref{prop:supp-hazard}, the finite-recovery hazard is strictly increasing. Therefore the finite-recovery renewal law cannot coincide with the zero-jitter benchmark.
\end{proof}

\section{Soft dead time and finite-window concentration}\label{sec:supp-soft}

\begin{proposition}\label{prop:supp-soft}
In the finite-recovery model,
\begin{equation}\label{eq:supp-soft}
\PP(A_{n+1}=1\mid A_n=1)=a
=
\frac12\,\kappa_{\mathrm r}\Gamma(\Delta\tau)^2+O((\Delta\tau)^3)
\qquad (\Delta\tau\downarrow0).
\end{equation}
Consequently, for the inadmissible set
\[
V_m=\{(a_0,\dots,a_{m-1})\in\{0,1\}^m:\exists\,k\le m-2,\ a_ka_{k+1}=11\},
\]
one has
\begin{equation}\label{eq:supp-window}
\PP\!\bigl((A_0,\dots,A_{m-1})\in V_m\bigr)
\le
\frac12\,\kappa_{\mathrm r}\Gamma(\Delta\tau)^2\,(m-1)
+O\!\bigl(m(\Delta\tau)^3\bigr)
\qquad(\Delta\tau\downarrow0).
\end{equation}
\end{proposition}

\begin{proof}
The identity $\PP(A_{n+1}=1\mid A_n=1)=a$ is immediate from the recovery model: after a click, the detector begins the next interval in the refractory state, so a second consecutive click requires recovery at time $s\in[0,\Delta\tau]$ followed by excitation during the remaining subinterval. This yields
\[
a=\int_0^{\Delta\tau}\kappa_{\mathrm r}e^{-\kappa_{\mathrm r}s}\bigl(1-e^{-\Gamma(\Delta\tau-s)}\bigr)\,ds.
\]
Using
\[
e^{-\kappa_{\mathrm r}s}=1+O(\Delta\tau),
\qquad
1-e^{-\Gamma(\Delta\tau-s)}=\Gamma(\Delta\tau-s)+O((\Delta\tau)^2),
\]
uniformly for $0\le s\le \Delta\tau$, gives
\[
a=
\kappa_{\mathrm r}\Gamma\int_0^{\Delta\tau}(\Delta\tau-s)\,ds
+O((\Delta\tau)^3)
=
\frac12\,\kappa_{\mathrm r}\Gamma(\Delta\tau)^2+O((\Delta\tau)^3).
\]
This proves \cref{eq:supp-soft}.

For the finite-window bound, use the union bound:
\[
\PP(V_m)\le \sum_{k=0}^{m-2}\PP(A_k=1,A_{k+1}=1).
\]
By stationarity,
\[
\PP(A_k=1,A_{k+1}=1)=\PP(A_0=1)\PP(A_1=1\mid A_0=1)\le a.
\]
Substituting the expansion for $a$ proves \cref{eq:supp-window}.
\end{proof}

\section{Spectral density and sub-Poisson counting}\label{sec:supp-spectrum}

\begin{proposition}\label{prop:supp-spectrum}
Let $(A_n)$ be a stationary finite-recovery record. Its spectral density
\[
f(\theta)=\rho(1-\rho)+2\sum_{n\ge1}\operatorname{Cov}(A_0,A_n)\cos(n\theta)
\]
has the closed form
\begin{equation}\label{eq:supp-spectrum}
f(\theta)=
\rho(1-\rho)+2\rho(a-\rho)\,
\frac{\cos\theta-\lambda}{1-2\lambda\cos\theta+\lambda^2},
\qquad \theta\in[-\pi,\pi],
\end{equation}
and is strictly increasing on $[0,\pi]$.
\end{proposition}

\begin{proof}
By \cref{prop:supp-cov},
\[
\operatorname{Cov}(A_0,A_n)=\rho(a-\rho)\lambda^{n-1},\qquad n\ge1.
\]
Hence
\[
f(\theta)=
\rho(1-\rho)+2\rho(a-\rho)\sum_{n\ge1}\lambda^{n-1}\cos(n\theta).
\]
Since $|\lambda|<1$,
\[
\sum_{n\ge1}\lambda^{n-1}e^{in\theta}
=
\frac{e^{i\theta}}{1-\lambda e^{i\theta}},
\]
so taking real parts gives \cref{eq:supp-spectrum}. To prove monotonicity, write $x=\cos\theta$ and
\[
R_\lambda(x)=\frac{x-\lambda}{1-2\lambda x+\lambda^2}.
\]
Then
\[
R_\lambda'(x)=\frac{1-\lambda^2}{(1-2\lambda x+\lambda^2)^2}>0.
\]
Since $a-\rho<0$, the affine function $\rho(1-\rho)+2\rho(a-\rho)R_\lambda(x)$ is strictly decreasing in $x$, while $\cos\theta$ is strictly decreasing on $[0,\pi]$. Hence $f$ is strictly increasing on $[0,\pi]$.
\end{proof}

\begin{proposition}\label{prop:supp-fano}
Let
\[
N_N=\sum_{t=0}^{N-1}A_t.
\]
Then
\begin{equation}\label{eq:supp-count-var}
\Var(N_N)=
N\rho(1-\rho)
+
\frac{2\rho(a-\rho)}{(1-\lambda)^2}
\Bigl(N(1-\lambda)-1+\lambda^N\Bigr).
\end{equation}
Consequently,
\begin{equation}\label{eq:supp-fano}
\lim_{N\to\infty}\frac{\Var(N_N)}{\EE N_N}
=
1-\rho+\frac{2(a-\rho)}{1-\lambda}
<1.
\end{equation}
\end{proposition}

\begin{proof}
For any stationary binary process,
\[
\Var(N_N)=N\gamma_0+2\sum_{n=1}^{N-1}(N-n)\gamma_n,
\qquad
\gamma_n=\operatorname{Cov}(A_0,A_n).
\]
Using $\gamma_0=\rho(1-\rho)$ and $\gamma_n=\rho(a-\rho)\lambda^{n-1}$ for $n\ge1$ gives
\[
\Var(N_N)=
N\rho(1-\rho)+2\rho(a-\rho)\sum_{n=1}^{N-1}(N-n)\lambda^{n-1}.
\]
The geometric sum evaluates to
\[
\sum_{n=1}^{N-1}(N-n)\lambda^{n-1}
=
\frac{N(1-\lambda)-1+\lambda^N}{(1-\lambda)^2},
\]
which proves \cref{eq:supp-count-var}. Since $\EE N_N=N\rho$, division by $N\rho$ and passage to the limit yield \cref{eq:supp-fano}. The strict inequality follows from $a-\rho<0$ and $1-\lambda>0$.
\end{proof}

\section{Shell equations, spectral transport, and inverse formulas}\label{sec:supp-kms}

The proper-time inverse temperature along a static worldline is
\[
\beta_{\mathrm{loc}}(x)=\beta_\infty N(x),
\]
and the weak-coupling response rate is
\begin{equation}\label{eq:supp-kms-rate}
\Gamma_+(x,\Omega)=\lambda_c^2\frac{\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}.
\end{equation}
Substituting the ideal and memory thresholds yields the shell equations
\begin{equation}\label{eq:supp-shell-phi}
\frac{\lambda_c^2\mathcal S_x(\Omega)\Delta\tau}{e^{\beta_\infty N(x)\Omega}-1}=\log\varphi,
\end{equation}
\begin{equation}\label{eq:supp-shell-mem}
\frac{\lambda_c^2\mathcal S_x(\Omega)}{e^{\beta_\infty N(x)\Omega}-1}=\Gamma_c(\kappa_{\mathrm r},\Delta\tau).
\end{equation}
If $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$, these rearrange to the pure-lapse formulas in the main text.

Dividing \cref{eq:supp-shell-phi,eq:supp-shell-mem} at shell points of the same detector design gives the general spectral-transport identity
\begin{equation}\label{eq:supp-spectral-transport}
\frac{e^{\beta_\infty\Omega N(x_{\varphi})}-1}
     {e^{\beta_\infty\Omega N(x_{\mathrm{mem}})}-1}
=
\frac{\Delta\tau\,\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}{\log\varphi}
\frac{\mathcal S_{x_{\varphi}}(\Omega)}{\mathcal S_{x_{\mathrm{mem}}}(\Omega)}.
\end{equation}
The homogeneous ratio law is the special case in which the spectral-weight ratio equals one.

For a nonextremal Hartle--Hawking horizon, $\beta_\infty=2\pi/\kappa_{\mathrm H}$ and
\[
N(\rho)=\kappa_{\mathrm H}\rho+O(\rho^3).
\]
Hence
\[
e^{\beta_\infty\Omega N(\rho)}-1
=
2\pi\Omega\rho\,\bigl(1+O(\rho)\bigr),
\]
which inserted into the homogeneous ratio law yields the proper-distance relation
\begin{equation}\label{eq:supp-rho-ratio}
\frac{\rho_{\varphi}}{\rho_{\mathrm{mem}}}
=
\frac{\Delta\tau\,\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}{\log\varphi}
\Bigl(1+O(\rho_{\varphi}+\rho_{\mathrm{mem}})\Bigr).
\end{equation}

For Schwarzschild, write $r_H=2M$ and $h=r-r_H$. The Gaussian normal distance from the horizon satisfies
\[
\rho=\int_{r_H}^{r_H+h}\frac{dr}{\sqrt{1-r_H/r}}
=2\sqrt{r_Hh}+O(h^{3/2}),
\]
so
\begin{equation}\label{eq:supp-schwarz-rho}
h=\frac{\rho^2}{4r_H}+O(\rho^4).
\end{equation}
The area excess of the areal sphere is therefore
\begin{equation}\label{eq:supp-area-excess}
A-A_H=4\pi\bigl((r_H+h)^2-r_H^2\bigr)
=2\pi\rho^2+O(\rho^4).
\end{equation}
Combining \cref{eq:supp-rho-ratio,eq:supp-area-excess} gives the Schwarzschild area-excess ratio in the main text.

For the inverse formulas in Schwarzschild,
\[
N(r)=\sqrt{1-\frac{2M}{r}},
\]
and under homogeneous spectral weight one has
\[
\beta_\infty N(r_{\varphi})=\beta_{\varphi}(D),
\qquad
\beta_\infty N(r_{\mathrm{mem}})=\beta_{\mathrm{mem}}(D).
\]
Squaring these identities produces two linear equations in $\beta_\infty^2$ and $2M\beta_\infty^2$, which solve to
\begin{equation}\label{eq:supp-beta-infty}
\beta_\infty^2=
\frac{\beta_{\varphi}(D)^2r_{\varphi}-\beta_{\mathrm{mem}}(D)^2r_{\mathrm{mem}}}
{r_{\varphi}-r_{\mathrm{mem}}},
\end{equation}
\begin{equation}\label{eq:supp-mass}
M=
\frac{r_{\varphi}r_{\mathrm{mem}}\bigl(\beta_{\varphi}(D)^2-\beta_{\mathrm{mem}}(D)^2\bigr)}
{2\bigl(\beta_{\varphi}(D)^2r_{\varphi}-\beta_{\mathrm{mem}}(D)^2r_{\mathrm{mem}}\bigr)}.
\end{equation}

For the two-mode self-calibrating inverse (Theorem~5.5 in the main text), fix one hardware design and operate it in two modes $D_1,D_2$ with distinct sampling parameters. By the homogeneous ratio law, the shell radii of mode $m$ satisfy $F_m(\theta)=0$, where $F_m$ is the shell-ratio function defined in the main text. Thus $F=(F_1,F_2)$ is a $C^1$ map from the two background parameters $\theta\in\RR^2$ to two shell-ratio residuals. If $F(\theta_*)=0$ and $DF(\theta_*)$ is nonsingular, the inverse function theorem \cite{KrantzParks2002} gives open neighbourhoods $U\ni\theta_*$ and $V\ni0$ such that $F:U\to V$ is one-to-one with a $C^1$ inverse. In particular, $\theta_*$ is the unique parameter in $U$ compatible with the observed shell data. The same parameter-dependent theorem also gives $C^1$ dependence of the recovered parameter on small perturbations of the measured shell radii, provided the Jacobian remains nonsingular. The Jacobian degenerates in particular when the two modes produce the same ratio constant $C_1=C_2$, which is a codimension-one condition in the mode parameter space and is avoided by choosing modes with distinct sampling intervals. No prior knowledge of the common amplitude $\lambda_c^2\mathcal S(\Omega)$ enters $F_m$, so the reconstruction is self-calibrating.

Finally, set
\[
s=\kappa_{\mathrm r}\Delta\tau,
\qquad
x=\frac{\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}{\kappa_{\mathrm r}}.
\]
The defining equation for $\Gamma_c$ is equivalent to
\[
s=\frac{\log x}{x-1}.
\]
For $x\neq1$, this is the same as
\[
\log x=s(x-1),
\qquad
x e^{-sx}=e^{-s}.
\]
Hence
\[
\Gamma_c(\kappa_{\mathrm r},\Delta\tau)
=-\frac{1}{\Delta\tau}\,
W_k\!\left(-\kappa_{\mathrm r}\Delta\tau\,e^{-\kappa_{\mathrm r}\Delta\tau}\right),
\qquad
k=
\begin{cases}
-1, & 0<s<1,\\
0, & s>1.
\end{cases}
\]
The other real branch gives the removable root $x=1$, which is a solution of the multiplied equation for all $s$ but solves the original quotient equation only at $s=1$. At $s=1$, both real branches meet and $\Gamma_c=\kappa_{\mathrm r}$.
As $s\downarrow0$, writing $L=\log(1/s)$ and $x=(L+u)/s$ yields $u=\log(L+u)+s$, hence $u=\log L+o(1)$. Therefore
\begin{equation}\label{eq:supp-fast-gammac}
\Gamma_c(\kappa_{\mathrm r},\Delta\tau)
=
\frac{L+\log L+o(1)}{\Delta\tau},
\qquad
L=\log\frac{1}{\kappa_{\mathrm r}\Delta\tau},
\end{equation}
which is the fast-sampling asymptotic used in the main text.
