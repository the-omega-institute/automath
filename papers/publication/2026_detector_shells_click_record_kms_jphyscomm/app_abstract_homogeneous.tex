\section{Abstract calibrated toy-model homogeneous shell specializations}\label{app:abstract-homogeneous-main}

The paper proves one exact homogeneous detector-field realization in
\Cref{prop:symmetry-homogeneous-main}, and that realization is ultrastatic with
constant lapse. Accordingly, every nonconstant-lapse homogeneous shell
statement is kept in this appendix and used only as abstract calibrated
toy-model algebra for one fixed response function. No separate exact
nonconstant-lapse UDW model with
\[
\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)
\]
is constructed here.

\subsection{Homogeneous specialization of the fixed calibrated response theorem}

From this point onward the appendix no longer adds new detector-field input.
The remaining statements keep the same fixed-response starting point from
\Cref{thm:response-locus-main,thm:shells-general-main} and impose extra
calibration or background hypotheses in a strictly one-way order:
\[
\text{fixed }R_\Omega
\Longrightarrow
\text{response loci}
\Longrightarrow
\text{pair transport}
\Longrightarrow
\text{homogeneous toy-model specializations}.
\]
Whenever a nonconstant lapse profile is inserted below, it is inserted only at
the last stage of this chain.

\begin{lemma}[Abstract calibrated homogeneous specialization]\label{lem:abstract-homogeneous-specialization-main}
\ModelAClause specialize the fixed calibrated response function $R_\Omega$ to the homogeneous form
below. Fix a static region $U$, a gap $\Omega>0$, a Tolman parameter
$\beta_\infty>0$, a positive constant $A$, and a $C^1$ lapse
$N:U\to(0,\infty)$. Define the calibrated response function
\[
R_\Omega^{\mathrm{hom}}(x)=\frac{A}{e^{\beta_\infty N(x)\Omega}-1}.
\]
Then for any target rate $\gamma>0$ the response locus
\[
L_\gamma^{\mathrm{hom}}
=\{x\in U:R_\Omega^{\mathrm{hom}}(x)=\gamma\}
\]
is exactly the lapse level set
\[
\beta_\infty N(x)=\frac{1}{\Omega}\log\!\left(1+\frac{A}{\gamma}\right).
\]
If $dN(x_0)\neq0$ at $x_0\in L_\gamma^{\mathrm{hom}}$, then
$L_\gamma^{\mathrm{hom}}$ is near $x_0$ a $C^1$ codimension-one shell. This is
an abstract calibrated toy-model specialization of a fixed response function.
The only exact detector-field realization proved in this paper is
\Cref{prop:symmetry-homogeneous-main}, where the lapse is constant; whenever
later appendices use homogeneous shell geometry with nonconstant lapse, they do
so through this level-set specialization rather than through a separate exact
UDW construction.
\end{lemma}

\begin{proof}
Solving $R_\Omega^{\mathrm{hom}}(x)=\gamma$ gives
\[
e^{\beta_\infty N(x)\Omega}=1+\frac{A}{\gamma},
\]
hence the displayed lapse level set. Differentiating yields
\[
dR_\Omega^{\mathrm{hom}}
=-A\frac{\beta_\infty\Omega e^{\beta_\infty N\Omega}}
{(e^{\beta_\infty N\Omega}-1)^2}\,dN,
\]
whose scalar prefactor is strictly negative. Therefore
$dR_\Omega^{\mathrm{hom}}(x_0)\neq0$ exactly when $dN(x_0)\neq0$, and the
regular level-set theorem gives the shell statement.
\end{proof}

\begin{corollary}[Abstract homogeneous corollary for a fixed calibrated response function $R_\Omega$]\label{cor:homogeneous-main}
\ModelAClause assume in addition the calibrated homogeneous spectral-weight regime.
Work under the standing convention for the response section, and assume in
addition that $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$ on the region under
study. Then both response loci are pure lapse level sets, with local inverse
temperatures
\begin{equation}\label{eq:betaphi-betamem-main}
\beta_{\varphi}=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)\Delta\tau}{\log\varphi}\right),
\qquad
\beta_{\mathrm{mem}}=
\frac{1}{\Omega}
\log\!\left(1+\frac{\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)}
{\Gamma_c(\kappa_{\mathrm r},\Delta\tau)}\right),
\end{equation}
so that
\begin{equation}\label{eq:lapse-level-main}
\Sigma_{\varphi}=\{x:\beta_{\mathrm{loc}}(x)=\beta_{\varphi}\},
\qquad
\Sigma_{\mathrm{mem}}=\{x:\beta_{\mathrm{loc}}(x)=\beta_{\mathrm{mem}}\}.
\end{equation}
If $dN\neq0$ on one of these level sets, then
\Cref{thm:response-locus-main,lem:abstract-homogeneous-specialization-main}
identify it as the corresponding shell; for nonconstant lapse this remains an
abstract calibrated toy-model specialization.
\end{corollary}

\begin{proof}
With $\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$, the two response-locus
equations contain $x$ only through $\beta_\infty N(x)$. Solving the
ideal-benchmark equation for $\beta_\infty N(x)$ gives $\beta_\varphi$, and
solving the memory equation gives $\beta_{\mathrm{mem}}$. Hence the loci are
the lapse level sets in~\eqref{eq:lapse-level-main}.
\end{proof}

For later response-locus-pair comparisons write $r_\varphi$ and
$r_{\mathrm{mem}}$ for the homogeneous ideal-benchmark and
memory-transition radii, and set
\[
D_\varphi=e^{\beta_\infty\Omega N(r_\varphi)}-1,
\qquad
D_{\mathrm{mem}}=e^{\beta_\infty\Omega N(r_{\mathrm{mem}})}-1,
\qquad
Q_{\varphi/\mathrm{mem}}=\frac{D_\varphi}{D_{\mathrm{mem}}}.
\]
The cancellation identity and the target-rate relation are proved later in
\Cref{thm:ratio-main}; this appendix uses only the level-set formulas above.

\subsection{Monotone spherical lapse bookkeeping in the abstract calibrated toy model}

Everything in this subsection stays at the level of the abstract calibrated
specialization from \Cref{lem:abstract-homogeneous-specialization-main}. No
new exact detector-field realization with
$\mathcal S_x(\Omega)\equiv\mathcal S(\Omega)$ is being claimed here.
In particular, any nonconstant-lapse insertion is used only as toy-model
bookkeeping for a fixed calibrated response function.

\begin{proposition}[Abstract calibrated existence on monotone spherical regions for a fixed calibrated $R_\Omega$]\label{prop:existence-main}
\ModelAClause assume in addition the homogeneous
specialization on a monotone spherical region.
Work under the standing convention for the response section, and assume the
homogeneous specialization of \Cref{cor:homogeneous-main} in a static
spherically symmetric region $r\in I$. Suppose
$N:I\to N(I)$ is $C^1$ and strictly monotone. For either threshold
$\star\in\{\varphi,\mathrm{mem}\}$, the corresponding response locus exists in $I$ if
and only if
\[
\frac{\beta_\star}{\beta_\infty}\in N(I),
\]
where $\beta_\varphi$ and $\beta_{\mathrm{mem}}$ are defined in~\eqref{eq:betaphi-betamem-main}. When it exists, its radius is unique and equals
\[
r_\star=N^{-1}\!\left(\frac{\beta_\star}{\beta_\infty}\right).
\]
If $N'(r_\star)\neq0$, this response locus is a regular spherical shell inside
the abstract calibrated toy model.
\end{proposition}

\begin{proof}
By \Cref{cor:homogeneous-main}, the shell equation is exactly
$\beta_\infty N(r)=\beta_\star$. Since $N$ is strictly monotone on $I$, it is
one-to-one onto its range. Hence the equation has a solution precisely when
$\beta_\star/\beta_\infty$ lies in $N(I)$, and the solution is then the
displayed inverse image. Thus shell existence on a monotone spherical region is
completely equivalent to the calibrated target lapse lying in the available
range of $N$. The nonzero derivative condition is the one-dimensional form of
the regular level-set condition.
\end{proof}

In this abstract calibrated homogeneous regime, both response loci are simply
lapse level sets selected by the two record thresholds, and they become shells
when the regularity condition holds. The same local response scale therefore
supports two different detector-selected response hypersurfaces whenever the
range condition in \Cref{prop:existence-main} is met. The physical comparison
kept in the main text is still the transport identity
\Cref{cor:spectral-transport-main}, with the spectral-weight ratio retained
unless exact homogeneity is added as a separate calibration hypothesis.
Appendix~\ref{app:schwarzschild-toy} records a purely algebraic
Schwarzschild-lapse toy specialization of this homogeneous model, together with
the associated range and regularity checks.

\begin{proposition}[Abstract calibrated ordering for the fixed homogeneous response function $R_\Omega$]\label{prop:order-main}
\ModelAClause assume in addition the homogeneous
specialization.
Work under the standing convention for the response section, assume the calibrated
homogeneous spectral-weight regime, and define
\[
\Gamma_{\varphi}=\frac{\log\varphi}{\Delta\tau}.
\]
Let $N_{\varphi}$ and $N_{\mathrm{mem}}$ denote the lapse values of the two
response loci $\Sigma_{\varphi}$ and $\Sigma_{\mathrm{mem}}$. Then
\begin{equation}\label{eq:order-main}
N_{\varphi}<N_{\mathrm{mem}}
\iff
\Gamma_{\varphi}>\Gamma_c(\kappa_{\mathrm r},\Delta\tau),
\end{equation}
with equality if and only if $\Gamma_{\varphi}=\Gamma_c(\kappa_{\mathrm r},\Delta\tau)$. Thus the ordering of the two homogeneous response loci is detector controlled; when the regularity condition holds, this is the shell ordering inside the abstract calibrated toy model.
\end{proposition}

\begin{proof}
Both response loci are lapse level sets in the calibrated homogeneous
spectral-weight regime. Writing $A=\lambda_{\mathrm{sw}}^2\mathcal S(\Omega)>0$,
the local excitation rate as a function of the lapse is
\[
\Gamma_+(N)=\frac{A}{e^{\beta_\infty\Omega N}-1}.
\]
For $\beta_\infty,\Omega,N>0$,
\[
\frac{d\Gamma_+}{dN}
=
-A\,\frac{\beta_\infty\Omega e^{\beta_\infty\Omega N}}
{(e^{\beta_\infty\Omega N}-1)^2}<0 .
\]
Hence the higher threshold rate corresponds exactly to the smaller lapse value, giving~\eqref{eq:order-main} and the equality statement.
\end{proof}
