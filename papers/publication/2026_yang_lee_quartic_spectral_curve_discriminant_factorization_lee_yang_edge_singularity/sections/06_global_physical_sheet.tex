\section{Positive-real dominant branch and endpoint asymptotics}\label{sec:physical-sheet}

Having classified all real branch points, we now parametrize the real branch over positive \(y\), prove by direct modulus inequalities that it is the dominant branch, derive the Puiseux expansion at the endpoint \(y=0\), and show that the two asymptotic regimes are controlled by two different mechanisms: a dominant square-root branch point at finite \(y\) and a double tangency at projective infinity.

\subsection{Explicit parametrization and strict monotonicity}

The following lemma records the only local facts used in this section:
simple roots vary analytically in the parameter, and a strict modulus gap persists locally by continuity.

\begin{lemma}[Local simplicity and persistence on $(0,\infty)$]\label{lem:local-simplicity-persistence}
Fix \(y_0>0\) with \(y_0\neq 1\).
Then the four roots of \(\Pi(\cdot,y_0)\) are simple.
Consequently:
\begin{enumerate}
\item\label{item:local-analyticity} Each root extends to an analytic function on a neighborhood of \(y_0\).
\item\label{item:local-persistence} If one of these local branches, say \(\lambda_1(y)\), satisfies
\[
\abs{\lambda_1(y_0)}>\abs{\lambda_j(y_0)}
\qquad (j\neq 1),
\]
then the same strict inequality holds for all \(y\) in a neighborhood of \(y_0\).
\end{enumerate}
At \(y_0=1\), the root \(\lambda=2\) is simple and therefore also admits a local analytic continuation.
\end{lemma}

\begin{proof}
The discriminant \(\Disc_\lambda\Pi(\lambda,y)\) was computed in Corollary~\ref{cor:disc-factor}:
\[
\Disc_\lambda\Pi(\lambda,y) = -y(y-1)(256y^3+411y^2+165y+32).
\]
This vanishes on \((0,\infty)\) only at \(y=1\), because the cubic factor has its unique real root at \(y_{\mathrm{sub}}<0\) by Theorem~\ref{thm:ep-classification}(3).
Therefore the four roots are simple at every \(y_0>0\) with \(y_0\neq1\).
The analytic continuation in \ref{item:local-analyticity} is the standard implicit-function theorem for simple roots of a polynomial with analytic coefficients \cite{Kato1976}.
The persistence statement \ref{item:local-persistence} is immediate from continuity of the local branches.
At \(y_0=1\), the factorization
\[
\Pi(\lambda,1)=(\lambda-2)(\lambda-1)(\lambda+1)^2
\]
shows directly that \(\lambda=2\) is simple, so the same local analyticity statement applies to that branch.
\end{proof}

\begin{remark}[Reducible boundary at $y=1$]\label{rem:y1-reducible}
At $y=1$ the polynomial factors as $\Pi(\lambda,1)=(\lambda-2)(\lambda-1)(\lambda+1)^2$.
The double root $\lambda=-1$ is subdominant (modulus $1$) while the simple root $\lambda=2$ is the dominant eigenvalue.
Since $\lambda=2$ is simple at $y=1$, Lemma~\ref{lem:local-simplicity-persistence}\ref{item:local-analyticity}---the first lemma in this section---applies to this root: the branch $\lambda_{\mathrm{dom}}$ extends analytically through $y=1$.
\end{remark}

\begin{theorem}[Positive-real dominant branch on $y > 0$]\label{thm:physical-sheet}
The real branch of the spectral curve connecting $(\lambda, y) = (1, 0)$ to $(\lambda, y) = (2, 1)$ is given by
\begin{equation}\label{eq:y-phys}
y_{\mathrm{dom}}(\lambda)
= \lambda^2 - \frac{1}{2} - \frac{1}{2}\sqrt{4\lambda^3 - 4\lambda + 1},
\qquad \lambda \ge 1.
\end{equation}
The function $y_{\mathrm{dom}}$ is strictly increasing on $(1, \infty)$ and maps $(1, \infty)$ bijectively onto $(0, \infty)$.
Consequently it has an inverse
\[
\lambda_{\mathrm{dom}}:(0,\infty)\to(1,\infty).
\]
\end{theorem}

\begin{proof}
From \eqref{eq:Pi-as-quadratic-in-y}, the spectral curve is quadratic in $y$ with solutions
\[
y = \lambda^2 - \frac{1}{2} \pm \frac{1}{2}\sqrt{4\lambda^3 - 4\lambda + 1}.
\]
At $(\lambda, y) = (1, 0)$ we have $4 \cdot 1 - 4 + 1 = 1$, and $1 - 1/2 - 1/2 = 0$, confirming that the minus-sign branch passes through $(1, 0)$.
Similarly, at $(\lambda, y) = (2, 1)$, one has $4 \cdot 8 - 8 + 1 = 25$ and $4 - 1/2 - 5/2 = 1$, confirming continuity to $(2, 1)$.

Differentiating \eqref{eq:y-phys} with respect to $\lambda$ gives
\begin{equation}\label{eq:dy-dlambda}
\frac{\dd y_{\mathrm{dom}}}{\dd\lambda}
= \frac{2\lambda\sqrt{4\lambda^3 - 4\lambda + 1} - (3\lambda^2 - 1)}
{\sqrt{4\lambda^3 - 4\lambda + 1}}.
\end{equation}
The critical points of $y_{\mathrm{dom}}$ on $(1, \infty)$ satisfy
\[
2\lambda\sqrt{4\lambda^3 - 4\lambda + 1} = 3\lambda^2 - 1.
\]
Squaring both sides and simplifying yields
\[
(\lambda - 1)^2(16\lambda^3 - 9\lambda^2 + 1) = 0.
\]
For $\lambda > 1$, both sides of
\[
2\lambda\sqrt{4\lambda^3 - 4\lambda + 1} = 3\lambda^2 - 1
\]
are strictly positive, so this squaring step introduces no spurious solution.
The polynomial $16\lambda^3 - 9\lambda^2 + 1$ has no zero on $[0, \infty)$: its derivative is $6\lambda(8\lambda - 3)$, so the unique interior minimum on $[0, \infty)$ is at $\lambda = 3/8$, where
\[
16\left(\frac{3}{8}\right)^3 - 9\left(\frac{3}{8}\right)^2 + 1 = \frac{37}{64} > 0.
\]
Hence the numerator of \eqref{eq:dy-dlambda} has no zero on $(1, \infty)$.
At $\lambda = 2$ it equals $2 \cdot 2 \cdot 5 - 11 = 9 > 0$, so continuity implies that it is positive throughout $(1,\infty)$.
Therefore $y_{\mathrm{dom}}$ is strictly increasing on $(1, \infty)$.

Since $y_{\mathrm{dom}}(1) = 0$ and $y_{\mathrm{dom}}(\lambda) \to +\infty$ as $\lambda \to +\infty$ (the leading term is $\lambda^2$), the map is a bijection $(1, \infty) \to (0, \infty)$.
\end{proof}

\begin{lemma}[Comparison of the two positive branches for \(y>1\)]\label{lem:ygt1-sheet-comparison}
For \(\lambda>1\), define
\[
y_-(\lambda)=\lambda^2-\frac12-\frac12\sqrt{4\lambda^3-4\lambda+1},
\qquad
y_+(\lambda)=\lambda^2-\frac12+\frac12\sqrt{4\lambda^3-4\lambda+1}.
\]
Then \(y_+(\lambda)>y_-(\lambda)\) for every \(\lambda>1\), and \(y_+\) is strictly increasing on \((1,\infty)\).
If \(y>1\) and
\[
y=y_-(q)=y_+(p)
\]
for positive roots \(0<p<q\) of \(\Pi(\cdot,y)\), then \(q>p\).
Consequently, for \(y>1\) the larger positive root of \(\Pi(\cdot,y)\) lies on the minus-sign branch \(y_-=y_{\mathrm{dom}}\).
\end{lemma}

\begin{proof}
The strict inequality \(y_+(\lambda)>y_-(\lambda)\) is immediate from the definition.
Moreover,
\[
y_+'(\lambda)=2\lambda+\frac{3\lambda^2-1}{\sqrt{4\lambda^3-4\lambda+1}}>0
\qquad (\lambda>1),
\]
because both summands are positive.
Hence \(y_+\) is strictly increasing on \((1,\infty)\).

Now suppose \(y=y_-(q)=y_+(p)\) with \(0<p<q\).
Then
\[
y_+(q)>y_-(q)=y=y_+(p).
\]
By the strict monotonicity of \(y_+\), this implies \(q>p\).
Therefore the root on the minus-sign sheet is the larger of the two positive roots for \(y>1\).
\end{proof}

\begin{proposition}[Global dominance on the positive axis]\label{prop:global-dominance}
For every $y>0$, the root $\lambda_{\mathrm{dom}}(y)$ is the spectral radius:
\[
r(y)=\lambda_{\mathrm{dom}}(y).
\]
Hence the branch parameterized by \eqref{eq:y-phys} is exactly the dominant branch over $y>0$.
\end{proposition}

\begin{proof}
The proof proceeds in two steps: an initial-point identification at $y=1$, and a direct modulus comparison covering each parameter regime.
The reducible boundary at $y=1$ is handled explicitly as Case 3.

\medskip
\noindent\textbf{Step 1: Initial-point identification at $y=1$.}
At $y=1$, Theorem~\ref{thm:ep-classification}(2) gives $\Pi(\lambda,1)=(\lambda-2)(\lambda-1)(\lambda+1)^2$.
The four roots are $\lambda=2,1,-1,-1$ with moduli $2,1,1,1$.
The spectral radius is $r(1)=2$.
By Theorem~\ref{thm:physical-sheet}, $\lambda_{\mathrm{dom}}(1)=y_{\mathrm{dom}}^{-1}(1)$.
Since $y_{\mathrm{dom}}(2)=4-\tfrac12-\tfrac12\sqrt{32-8+1}=\tfrac72-\tfrac{5}{2}=1$, we have $\lambda_{\mathrm{dom}}(1)=2$.
Therefore $\lambda_{\mathrm{dom}}(1)=r(1)=2$: the chosen branch coincides with the spectral radius at the explicit base point $y_0=1$.

\medskip
\noindent\textbf{Step 2: Direct comparison in each regime.}

\noindent\textit{Case 1: $0<y<1$.}
By Theorem~\ref{thm:real-phase-diagram}, the four roots are real and can be ordered as
\[
r_1<r_2<r_3<r_4,
\qquad
r_1<-1<r_2<0<r_3<1<r_4.
\]
The branch from Theorem~\ref{thm:physical-sheet} is the unique real root greater than \(1\), so \(\lambda_{\mathrm{dom}}(y)=r_4\).
It remains to compare \(\abs{r_1}\) with \(r_4\).
For \(x\ge 1\),
\[
\partial_\lambda\Pi(-x,y)
=
-4x^3-3x^2+2(2y+1)x+1
\le
-4x^3-3x^2+6x+1
=
-(x-1)(4x^2+7x+1)<0.
\]
Hence \(\lambda\mapsto \Pi(\lambda,y)\) is strictly decreasing on \((-\infty,-1]\).
Since \(r_4>1\) is a root,
\[
\Pi(-r_4,y)-\Pi(r_4,y)=2r_4(r_4^2-1)>0,
\]
and therefore \(\Pi(-r_4,y)>0\).
Because \(\Pi\) is strictly decreasing on \((-\infty,-1]\) and vanishes at \(r_1\), one must have \(r_1>-r_4\), hence \(\abs{r_1}<r_4\).
Since also \(\abs{r_2}<1<r_4\) and \(0<r_3<r_4\), the root \(r_4=\lambda_{\mathrm{dom}}(y)\) is the unique dominant root.

\noindent\textit{Case 2: $y>1$.}
By Theorem~\ref{thm:real-phase-diagram}, only the noncompact oval contributes for \(y>1\), so the two real roots are positive; write them as \(0<p<q\).
The remaining two roots form a nonreal conjugate pair \(z,\bar z\).
Lemma~\ref{lem:ygt1-sheet-comparison} shows that the larger positive root is the one on the minus-sign branch, so \(q=\lambda_{\mathrm{dom}}(y)\).
Factor
\[
\Pi(\lambda,y)=(\lambda-p)(\lambda-q)(\lambda^2+a\lambda+b),
\]
where \(b=\abs{z}^2>0\).
Comparing coefficients of \(\lambda^3\) and \(\lambda\):
\[
a=p+q-1,
\qquad
b=\frac{(p+q-1)pq-1}{p+q}=pq-\frac{pq+1}{p+q}<pq<q^2.
\]
Therefore \(\abs{z}=\sqrt{b}<q\), and since \(p<q\), the root \(q=\lambda_{\mathrm{dom}}(y)\) has strictly largest modulus.

\noindent\textit{Case 3: Reducible boundary $y=1$.}
Step 1 already established $r(1)=\lambda_{\mathrm{dom}}(1)=2$.
At $y=1$ the double root $\lambda=-1$ has modulus $1<2$, so $\lambda_{\mathrm{dom}}(1)$ is strictly dominant.
Since $\lambda=2$ is simple at $y=1$, Lemma~\ref{lem:local-simplicity-persistence}\ref{item:local-analyticity} applies to this root.

\medskip
\noindent\textbf{Conclusion.}
Cases 1, 2, and 3 jointly cover all of $(0,\infty)$ and show that $r(y)=\lambda_{\mathrm{dom}}(y)$ holds for every $y>0$.
In each case, the dominant root is identified by explicit inequality against all other roots, so the conclusion does not rely on any modulus-crossing or discriminant-based continuity argument.
This completes the proof.
\end{proof}

Accordingly,
\[
f_{\mathrm{dom}}(y)=\log \lambda_{\mathrm{dom}}(y)
\qquad (y>0).
\]

\subsection{Endpoint at \texorpdfstring{\(y=0\)}{y=0} and Puiseux expansion}\label{subsec:vacuum-puiseux}

The endpoint $y = 0$ is not an ordinary analytic point of the inverse function $\lambda_{\mathrm{dom}}(y)$, but a square-root ramification point.
The local expansion is a classical Puiseux series \cite{BrieskornKnorrer1986,Walker1950}.

\begin{proposition}[Endpoint Puiseux expansion at \(y=0\)]\label{prop:vacuum-puiseux}
Setting $\lambda = 1 + \varepsilon$, the dominant branch satisfies
\begin{equation}\label{eq:y-puiseux}
y_{\mathrm{dom}}(1 + \varepsilon)
= 2\varepsilon^2 - 5\varepsilon^3 + 21\varepsilon^4 - 94\varepsilon^5 + O(\varepsilon^6).
\end{equation}
Inverting, the dominant eigenvalue on the dominant branch over \(y>0\) expands as
\begin{equation}\label{eq:lambda-puiseux}
\lambda_{\mathrm{dom}}(y)
= 1 + \frac{1}{\sqrt{2}}\,y^{1/2}
+ \frac{5}{8}\,y
- \frac{43}{64\sqrt{2}}\,y^{3/2}
- \frac{1}{16}\,y^2
+ O(y^{5/2}).
\end{equation}
The logarithmic spectral radius
$f_{\mathrm{dom}}(y) = \log\lambda_{\mathrm{dom}}(y)$
satisfies
\begin{equation}\label{eq:f-puiseux}
f_{\mathrm{dom}}(y)
= \frac{1}{\sqrt{2}}\,y^{1/2}
+ \frac{3}{8}\,y
- \frac{217\sqrt{2}}{384}\,y^{3/2}
+ O(y^2),
\end{equation}
and the scaled logarithmic slope
\begin{equation}\label{eq:rho-vacuum}
\sigma(y):=y\,f_{\mathrm{dom}}'(y)
= \frac{1}{2\sqrt{2}}\,y^{1/2}
+ \frac{3}{8}\,y
- \frac{217\sqrt{2}}{256}\,y^{3/2}
+ O(y^2).
\end{equation}
\end{proposition}

\begin{proof}
At $\lambda = 1 + \varepsilon$,
\[
4\lambda^3 - 4\lambda + 1
= 1 + 8\varepsilon + 12\varepsilon^2 + 4\varepsilon^3.
\]
Taking the square root by the binomial series:
\[
\sqrt{4\lambda^3 - 4\lambda + 1}
= 1 + 4\varepsilon - 2\varepsilon^2 + 10\varepsilon^3 - 42\varepsilon^4 + 188\varepsilon^5 + O(\varepsilon^6).
\]
Substituting into \eqref{eq:y-phys}:
\begin{align*}
y_{\mathrm{dom}}(1 + \varepsilon)
&= (1 + \varepsilon)^2 - \tfrac{1}{2}
  - \tfrac{1}{2}\bigl(1 + 4\varepsilon - 2\varepsilon^2 + 10\varepsilon^3 - 42\varepsilon^4 + 188\varepsilon^5 + \cdots\bigr) \\
&= 1 + 2\varepsilon + \varepsilon^2 - \tfrac{1}{2}
  - \tfrac{1}{2} - 2\varepsilon + \varepsilon^2 - 5\varepsilon^3 + 21\varepsilon^4 - 94\varepsilon^5 - \cdots \\
&= 2\varepsilon^2 - 5\varepsilon^3 + 21\varepsilon^4 - 94\varepsilon^5 + O(\varepsilon^6).
\end{align*}
To invert the series, write
\[
\varepsilon = A y^{1/2} + B y + C y^{3/2} + D y^2 + O(y^{5/2}).
\]
Substituting this ansatz into \eqref{eq:y-puiseux} and matching the coefficients of
\(
y,\ y^{3/2},\ y^2,\ y^{5/2}
\)
gives
\[
2A^2=1,\qquad 4AB-5A^3=0,
\]
\[
4AC+2B^2-15A^2B+21A^4=0,
\]
\[
4AD+4BC-15A^2C-15AB^2+84A^3B-94A^5=0.
\]
Choosing the positive branch \(A=1/\sqrt2\) and solving successively yields
\[
A=\frac{1}{\sqrt2},\qquad
B=\frac58,\qquad
C=-\frac{43}{64\sqrt2},\qquad
D=-\frac{1}{16},
\]
which is \eqref{eq:lambda-puiseux}.
Then
\[
\log(1+\varepsilon)=\varepsilon-\frac{\varepsilon^2}{2}+\frac{\varepsilon^3}{3}+O(y^2)
\]
gives \eqref{eq:f-puiseux}, and term-by-term differentiation yields \eqref{eq:rho-vacuum}.
\end{proof}

Equation \eqref{eq:rho-vacuum} shows that the scaled logarithmic slope does not open linearly from the endpoint \(y=0\), but with a square-root singularity:
\[
\sigma(y) \sim \frac{1}{2\sqrt{2}}\,y^{1/2}
\qquad (y \to 0^+).
\]

\subsection{Standard square-root branching and local cosine law}\label{subsec:vacuum-quantization}

The same square-root branch point controls the first zeros of \(Z_m\) on the negative real axis.
There are two distinct ingredients.
First, the spectral polynomial itself forces a square-root splitting whenever \(\partial_\lambda P\) vanishes simply in the \(y\)-direction.
Second, once a dominant pair with symmetric amplitudes has been isolated and the remaining branches are shown to be uniformly smaller on the complex microscopic scale, the polynomial family reduces to a shifted cosine law.
The next two lemmas are standard local tools stated only in the specialized form needed later:
Lemma~\ref{lem:generic-ep2-branching} is the usual Puiseux square-root branching at a simple double point \cite{Walker1950,BrieskornKnorrer1986}, and Lemma~\ref{lem:ep2-cosine-law} is a localized two-branch version of the standard Beraha--Kahane--Weiss analysis of zeros of spectral sums \cite{BerahaKahaneWeiss1975,BerahaKahaneWeiss1978,Sokal2004}.

\begin{lemma}[Generic square-root branching at a simple double point]\label{lem:generic-ep2-branching}
Let \(P(\lambda,y)\) be holomorphic near \((\lambda_c,y_c)\in\CC^2\), and assume
\[
P(\lambda_c,y_c)=P_\lambda(\lambda_c,y_c)=0,
\qquad
P_{\lambda\lambda}(\lambda_c,y_c)P_y(\lambda_c,y_c)\neq 0.
\]
Then, after setting \(y=y_c+s^2\), there exist \(s_0>0\) and holomorphic branches \(\lambda_\pm(s)\) for \(|s|<s_0\) such that
\[
P(\lambda_\pm(s),y_c+s^2)=0,
\qquad
\lambda_\pm(0)=\lambda_c,
\]
and
\[
\lambda_\pm(s)=\lambda_c\pm \alpha s+\beta s^2+O(s^3),
\qquad
\alpha^2=-\frac{2P_y}{P_{\lambda\lambda}}(\lambda_c,y_c).
\]
After relabeling, one may choose the branches so that \(\lambda_-(s)=\lambda_+(-s)\).
\end{lemma}

\begin{proof}
Write \(h=\lambda-\lambda_c\).
Taylor expansion at \((\lambda_c,y_c)\) gives
\[
P(\lambda_c+h,y_c+s^2)
=
\frac{1}{2}P_{\lambda\lambda}(\lambda_c,y_c)h^2
+P_y(\lambda_c,y_c)s^2
+R(h,s),
\]
where
\[
R(h,s)=O\!\bigl(|h|^3+|h||s|^2+|s|^4\bigr).
\]
Set \(h=su\) and define
\[
G(u,s)=
\begin{cases}
s^{-2}P(\lambda_c+su,y_c+s^2), & s\neq 0,\\[4pt]
\dfrac{1}{2}P_{\lambda\lambda}(\lambda_c,y_c)u^2+P_y(\lambda_c,y_c), & s=0.
\end{cases}
\]
The remainder estimate shows that \(G\) is holomorphic near \((u,s)=(\pm\alpha,0)\), where \(\alpha\) is any square root of \(-2P_y/P_{\lambda\lambda}\).
Moreover,
\[
G(\pm\alpha,0)=0,
\qquad
\partial_u G(\pm\alpha,0)=P_{\lambda\lambda}(\lambda_c,y_c)(\pm\alpha)\neq 0.
\]
By the implicit-function theorem, there exist holomorphic functions \(u_\pm(s)=\pm\alpha+O(s)\) such that
\[
G(u_\pm(s),s)=0.
\]
Setting \(\lambda_\pm(s)=\lambda_c+s\,u_\pm(s)\) yields
\[
\lambda_\pm(s)=\lambda_c\pm\alpha s+\beta_\pm s^2+O(s^3).
\]
Replacing \(s\) by \(-s\) leaves \(y_c+s^2\) unchanged and interchanges the two roots issuing from \(\lambda_c\).
After relabeling, this gives \(\lambda_-(s)=\lambda_+(-s)\), and hence \(\beta_-=\beta_+=:\beta\).
\end{proof}

\begin{lemma}[Localized cosine law for a dominant two-branch decomposition]\label{lem:ep2-cosine-law}
Let \(\{Z_m(y)\}_{m\ge 0}\) be holomorphic in a neighborhood of \(y_c\).
Assume that there exists \(s_0>0\) such that, for \(|s|<s_0\),
\[
Z_m(y_c+s^2)=C_+(s)\lambda_+(s)^m + C_-(s)\lambda_-(s)^m + R_m(y_c+s^2),
\]
where \(\lambda_\pm\) and \(C_\pm\) are holomorphic on \(|s|<s_0\) and satisfy
\[
\lambda_\pm(s)=\lambda_c\exp\!\left(\pm as + bs^2 + O(s^3)\right),
\qquad
C_\pm(s)=A_*\mp B_*s+O(s^2),
\]
with real constants \(\lambda_c>0\), \(a>0\), \(b\in\RR\), \(A_*\neq 0\), and \(B_*\in\RR\).
Assume moreover that for every compact \(K\subset\CC\) there exists \(M_K>0\) such that
\[
\sup_{u\in K}
\left|
R_m\!\left(y_c-\frac{u^2}{m^2}\right)
\right|
\le M_K \lambda_c^m m^{-2}
\]
for all sufficiently large \(m\).
Define
\[
F_m(u):=
\frac{e^{b u^2/m}}{2A_*\lambda_c^m}
Z_m\!\left(y_c-\frac{u^2}{m^2}\right).
\]
Then, for every compact \(K\subset\CC\),
\begin{equation}\label{eq:universal-ep2-scaling}
F_m(u)
=
\cos\!\left(
a u-\frac{B_*}{A_*}\frac{u}{m}
\right)+O_K(m^{-2})
\end{equation}
uniformly for \(u\in K\).
Fix \(k\ge 1\) and set
\[
q_k:=\frac{(2k-1)\pi}{2},
\qquad
u_k^{(0)}:=\frac{q_k}{a}.
\]
Then there exist \(\eta_k>0\) and \(m_k\) such that, for every \(m\ge m_k\), the function \(F_m\) has exactly one zero \(u_{m,k}\) in the disc
\[
\{u\in\CC:\abs{u-u_k^{(0)}}<\eta_k\},
\]
and this zero satisfies
\begin{equation}\label{eq:universal-ep2-u}
u_{m,k}
=
\frac{q_k}{a}
\left(
1+\frac{B_*}{aA_*m}
\right)
+O_k(m^{-2}),
\end{equation}
whence
\begin{equation}\label{eq:universal-ep2-y}
y_{m,k}
=
y_c-\frac{q_k^2}{a^2m^2}
-\frac{2B_*q_k^2}{a^3A_*m^3}
+O_k(m^{-4}).
\end{equation}
\end{lemma}

\begin{proof}
Fix a compact set \(K\subset\CC\) and set \(s=iu/m\), so that \(y=y_c-u^2/m^2\).
Because the expansions are holomorphic in \(s\), the \(O(s^3)\) and \(O(s^2)\) remainders are uniform on compact subsets of \(\CC\) near \(0\).
Therefore,
\[
\lambda_\pm\!\left(\frac{iu}{m}\right)^m
=
\lambda_c^m
\exp\!\left(
\pm iau-\frac{bu^2}{m}+O_K(m^{-2})
\right),
\]
and similarly
\[
C_\pm\!\left(\frac{iu}{m}\right)
=
A_*\mp iB_*\frac{u}{m}+O_K(m^{-2}).
\]
Multiplying and adding the two dominant contributions gives
\begin{align*}
\frac{e^{b u^2/m}}{2A_*\lambda_c^m}
\bigl(C_+\lambda_+^m + C_-\lambda_-^m\bigr)
&=
\cos(au)+\frac{B_*}{A_*}\frac{u}{m}\sin(au)+O_K(m^{-2}).
\end{align*}
Using
\[
\cos(\theta-\delta)=\cos\theta+\delta\sin\theta+O(\delta^2),
\qquad
\delta=\frac{B_*}{A_*}\frac{u}{m},
\]
we obtain
\[
\frac{e^{b u^2/m}}{2A_*\lambda_c^m}
\bigl(C_+\lambda_+^m + C_-\lambda_-^m\bigr)
=
g_m(u)+O_K(m^{-2}),
\]
where
\[
g_m(u):=\cos\!\left(a u-\frac{B_*}{A_*}\frac{u}{m}\right).
\]
The assumed bound on \(R_m\) then gives \eqref{eq:universal-ep2-scaling}.

Now fix \(k\ge1\).
Choose \(\eta_k>0\) so small that \(\cos(au)\) has no zero on the circle \(\abs{u-u_k^{(0)}}=\eta_k\).
Then
\[
\delta_k:=\inf_{\abs{u-u_k^{(0)}}=\eta_k}\abs{\cos(au)}>0.
\]
Since \(g_m\to \cos(au)\) uniformly on that circle, there exists \(m_k\) such that for all \(m\ge m_k\),
\[
\abs{g_m(u)}\ge \frac{\delta_k}{2}
\qquad
\text{whenever }\abs{u-u_k^{(0)}}=\eta_k.
\]
Equation \eqref{eq:universal-ep2-scaling} gives a matching bound
\[
\abs{F_m(u)-g_m(u)}\le \frac{\delta_k}{4}
\qquad
\text{whenever }\abs{u-u_k^{(0)}}=\eta_k
\]
for all sufficiently large \(m\).
Therefore
\[
\abs{F_m(u)}\ge \abs{g_m(u)}-\abs{F_m(u)-g_m(u)}\ge \frac{\delta_k}{4}
\qquad
\text{whenever }\abs{u-u_k^{(0)}}=\eta_k.
\]
Rouch\'e's theorem applied to the positively oriented circle \(\abs{u-u_k^{(0)}}=\eta_k\) now implies that \(F_m\) and \(g_m\) have the same number of zeros in \(\abs{u-u_k^{(0)}}<\eta_k\).
Since \(g_m(u)=0\) exactly when
\[
au-\frac{B_*}{A_*}\frac{u}{m}=q_k
\]
and the affine phase has a unique solution
\[
u_{m,k}^{(0)}:=\frac{q_k}{a-\frac{B_*}{A_*m}}
\]
in that disc for \(m\) large, \(F_m\) has exactly one zero \(u_{m,k}\) there.

Moreover,
\[
g_m'(u_{m,k}^{(0)})
=
-\left(a-\frac{B_*}{A_*m}\right)\sin q_k
\]
stays uniformly away from \(0\).
Since \(F_m(u_{m,k})=0\) and \(F_m-g_m=O_k(m^{-2})\) uniformly on the disc, a first-order Taylor expansion about \(u_{m,k}^{(0)}\) gives
\[
u_{m,k}=u_{m,k}^{(0)}+O_k(m^{-2}).
\]
Expanding \(u_{m,k}^{(0)}\) yields \eqref{eq:universal-ep2-u}, and squaring \(u_{m,k}\) gives \eqref{eq:universal-ep2-y}.
\end{proof}

\begin{theorem}[Endpoint scaling and zero quantization]\label{thm:vacuum-zero-quantization}
For the quartic transfer family studied here, the double-root endpoint at $y_c=0$ satisfies the assumptions of Lemma~\ref{lem:ep2-cosine-law} with
\[
a=\frac{1}{\sqrt2},
\qquad
b=\frac38,
\qquad
A_*=\frac12,
\qquad
B_*=\frac{3\sqrt2}{16}.
\]
Consequently, uniformly for \(u\) in compact subsets of \(\CC\),
\begin{equation}\label{eq:vacuum-scaling}
Z_m\!\left(-\frac{u^2}{m^2}\right)
=
\exp\!\left(-\frac{3u^2}{8m}\right)
\left[
\cos\!\left(\frac{u}{\sqrt{2}}-\frac{3u}{4\sqrt{2}\,m}\right)
+ O_K(m^{-2})
\right],
\end{equation}
and for each fixed $k\ge 1$ the $k$th negative real zero nearest $y=0$ satisfies
\begin{equation}\label{eq:vacuum-zero-quantization}
y_{m,k}
=
-\frac{\pi^2(2k-1)^2}{2m^2}
-\frac{3\pi^2(2k-1)^2}{4m^3}
+ O_k(m^{-4}).
\end{equation}
\end{theorem}

\begin{proof}
Appendix~\ref{app:vacuum-edge} works on the uniform disc $D_r$ introduced there and proves that on the local scale \(y=s^2\) the dominant pair satisfies the square-root branching predicted by Lemma~\ref{lem:generic-ep2-branching}:
\[
\lambda_\pm(s)=1\pm \frac{1}{\sqrt2}s+\frac58 s^2+O(s^3),
\qquad
C_\pm(s)=\frac12\mp \frac{3\sqrt2}{16}s+O(s^2),
\]
and that the remaining two branches contribute only \(O_K(m^{-2})\) on compact complex sets \(K\) under the scaling \(y=-u^2/m^2\).
Thus the hypotheses of Lemma~\ref{lem:ep2-cosine-law} hold with the stated constants and \(\lambda_c=1\).
Applying \eqref{eq:universal-ep2-scaling} gives \eqref{eq:vacuum-scaling}, and applying \eqref{eq:universal-ep2-y} on discs around the points \(u_k^{(0)}=\pi(2k-1)/\sqrt2\) gives \eqref{eq:vacuum-zero-quantization}.
\end{proof}

\begin{corollary}[Edge-zero counting law]\label{cor:edge-zero-counting}
Define
\[
N_m(\xi)
:=
\#\left\{
k\ge 1:
y_{m,k}\ge -\frac{\xi}{m^2}
\right\},
\qquad
\xi>0.
\]
If $\xi$ does not belong to the threshold set
\[
\left\{
\frac{\pi^2(2k-1)^2}{2}:k\ge1
\right\},
\]
then for all sufficiently large $m$ one has
\begin{equation}\label{eq:edge-counting-law}
N_m(\xi)
=
\left\lfloor
\frac{\sqrt{\xi}}{\pi\sqrt2}+\frac12
\right\rfloor.
\end{equation}
\end{corollary}

\begin{proof}
By \eqref{eq:vacuum-zero-quantization},
\[
-m^2 y_{m,k}
=
\frac{\pi^2(2k-1)^2}{2}+O_k(m^{-1}).
\]
If $\xi$ is not a threshold value, then for each fixed $k$ the sign of
\(
\xi+m^2y_{m,k}
\)
agrees, for all sufficiently large $m$, with the sign of
\(
\xi-\pi^2(2k-1)^2/2.
\)
Thus the set of indices counted by $N_m(\xi)$ stabilizes and is exactly
\[
\left\{
k\ge1:
\frac{\pi^2(2k-1)^2}{2}<\xi
\right\},
\]
which yields \eqref{eq:edge-counting-law}.
\end{proof}

In continuum form, \eqref{eq:vacuum-zero-quantization} gives
\[
k \sim \frac{m}{\pi\sqrt2}\sqrt{|y|},
\qquad
\left|\frac{\dd k}{\dd y}\right|
\sim
\frac{m}{2\pi\sqrt{2|y|}}
\qquad
(y\uparrow 0,\ |y|=O(m^{-2})).
\]
Thus the negative zeros nearest \(y=0\) do not approach the endpoint with a smooth density: they compress with the same square-root exponent that governs the Puiseux opening of the dominant branch.

\subsection{Double tangency at infinity and the quarter-power splitting law}\label{subsec:large-y-splitting}

\begin{proposition}[Double tangency at projective infinity]\label{prop:saturation}
Let $Y:=y^{1/4}$.
Then the dominant homogeneous balance of the spectral polynomial at projective infinity is
\begin{equation}\label{eq:infinity-balance}
\Pi_\infty(\lambda,y)
:=
\lambda^4-2y\lambda^2+y^2
=(\lambda^2-y)^2.
\end{equation}
Hence the positive large-\(y\) branch is generated by a double tangency of the sheets $\lambda\sim \pm y^{1/2}$ at infinity.
On the dominant branch one has
\begin{equation}\label{eq:lambda-high-y}
\lambda_{\mathrm{dom}}(y)
= y^{1/2} + \frac{1}{2}\,y^{1/4} + \frac{1}{4}
+ \frac{7}{64}\,y^{-1/4} + O(y^{-1/2}),
\end{equation}
and the logarithmic branch satisfies
\begin{equation}\label{eq:f-high-y}
f_{\mathrm{dom}}(y)
= \frac{1}{2}\log y + \frac{1}{2}\,y^{-1/4}
+ \frac{1}{8}\,y^{-1/2} + O(y^{-3/4}).
\end{equation}
Writing $y = e^s$ and $\Lambda_{\mathrm{dom}}(s) = f_{\mathrm{dom}}(e^s) - \log 2$, the scaled logarithmic slope
\begin{equation}\label{eq:rho-saturation}
\varrho(s) = \Lambda_{\mathrm{dom}}'(s)
= \frac{1}{2} - \frac{1}{8}\,e^{-s/4}
- \frac{1}{16}\,e^{-s/2} + O(e^{-3s/4})
\end{equation}
satisfies
\[
\lim_{s \to +\infty} \varrho(s) = \frac{1}{2}.
\]
The exponent $1/4$ is the splitting exponent created by the double tangency at infinity.
\end{proposition}

\begin{proof}
Set $y=Y^4$ and seek the positive large-\(y\) branch in the form
\[
\lambda=Y^2+\alpha Y+\beta+\gamma Y^{-1}+O(Y^{-2}).
\]
Substituting into $\Pi(\lambda,Y^4)$ gives
\begin{align}
\Pi(\lambda,Y^4)
&=
Y^6(4\alpha^2-1) \notag\\
&\quad
+Y^5(4\alpha^3+8\alpha\beta-3\alpha) \notag\\
&\quad
+Y^4(\alpha^4+12\alpha^2\beta-3\alpha^2+8\alpha\gamma+4\beta^2-3\beta)
+O(Y^3). \label{eq:infinity-expansion}
\end{align}
The leading coefficient $4\alpha^2-1$ is precisely the first-order splitting of the double balance \eqref{eq:infinity-balance}, so the dominant branch must satisfy $\alpha=1/2$.
The vanishing of the $Y^5$ coefficient then gives
\[
4\left(\frac12\right)^3+8\left(\frac12\right)\beta-3\left(\frac12\right)=0,
\qquad\text{hence}\qquad
\beta=\frac14.
\]
Using these values in the $Y^4$ coefficient of \eqref{eq:infinity-expansion} yields
\[
\frac{1}{16}+12\cdot \frac14\cdot \frac14-\frac34+4\gamma+4\cdot \frac{1}{16}-\frac34=0,
\qquad\text{hence}\qquad
\gamma=\frac{7}{64}.
\]
This proves \eqref{eq:lambda-high-y}.
Equations \eqref{eq:f-high-y} and \eqref{eq:rho-saturation} then follow by logarithmic expansion and differentiation with respect to $s=\log y$.
\end{proof}

The point of Proposition~\ref{prop:saturation} is conceptual as well as computational.
The exponent \(1/4\) is not merely a quartic fingerprint; it is the quarter-power splitting law generated by a double tangency at projective infinity.
Combined with Theorem~\ref{thm:support-law}, which gives only polynomial growth of the top-degree coefficients, this shows that the top-degree sector is a rigid algebraic boundary rather than an extensive multiplicity regime.
