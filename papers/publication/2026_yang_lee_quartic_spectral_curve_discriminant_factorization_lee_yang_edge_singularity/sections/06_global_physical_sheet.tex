\section{Positive-real dominant sheet and edge asymptotics}\label{sec:physical-sheet}

Having classified all real branch points, we now construct the dominant sheet over positive real fugacity, derive the exact Puiseux expansion at the vacuum boundary $y = 0$, and show that the two asymptotic edges are controlled by two different double-touching mechanisms: a dominant EP2 at finite fugacity and a double tangency at projective infinity.

\subsection{Explicit parametrization and strict monotonicity}

\begin{theorem}[Dominant sheet on $y > 0$]\label{thm:physical-sheet}
The dominant branch of the spectral curve, connecting $(\lambda, y) = (1, 0)$ to $(\lambda, y) = (2, 1)$, is given by
\begin{equation}\label{eq:y-phys}
y_{\mathrm{phys}}(\lambda)
= \lambda^2 - \frac{1}{2} - \frac{1}{2}\sqrt{4\lambda^3 - 4\lambda + 1},
\qquad \lambda \ge 1.
\end{equation}
The function $y_{\mathrm{phys}}$ is strictly increasing on $(1, \infty)$ and maps $(1, \infty)$ bijectively onto $(0, \infty)$.
In particular, the positive-real dominant sheet has no additional real branch point; its unique boundary is $y = 0$.
\end{theorem}

\begin{proof}
From \eqref{eq:Pi-as-quadratic-in-y}, the spectral curve is quadratic in $y$ with solutions
\[
y = \lambda^2 - \frac{1}{2} \pm \frac{1}{2}\sqrt{4\lambda^3 - 4\lambda + 1}.
\]
At $(\lambda, y) = (1, 0)$ we have $4 \cdot 1 - 4 + 1 = 1$, and $1 - 1/2 - 1/2 = 0$, confirming that the minus-sign branch passes through $(1, 0)$.
Similarly, at $(\lambda, y) = (2, 1)$: $4 \cdot 8 - 8 + 1 = 25$, $4 - 1/2 - 5/2 = 1$, confirming continuity to $(2, 1)$.

Differentiating \eqref{eq:y-phys} with respect to $\lambda$:
\begin{equation}\label{eq:dy-dlambda}
\frac{\dd y_{\mathrm{phys}}}{\dd\lambda}
= \frac{2\lambda\sqrt{4\lambda^3 - 4\lambda + 1} - (3\lambda^2 - 1)}
{\sqrt{4\lambda^3 - 4\lambda + 1}}.
\end{equation}
The critical points of $y_{\mathrm{phys}}$ on $(1, \infty)$ satisfy
\[
2\lambda\sqrt{4\lambda^3 - 4\lambda + 1} = 3\lambda^2 - 1.
\]
Squaring both sides and simplifying yields
\[
(\lambda - 1)^2(16\lambda^3 - 9\lambda^2 + 1) = 0.
\]
For $\lambda > 1$, both sides of
\[
2\lambda\sqrt{4\lambda^3 - 4\lambda + 1} = 3\lambda^2 - 1
\]
are strictly positive, so this squaring step introduces no spurious solution.
The polynomial $16\lambda^3 - 9\lambda^2 + 1$ has no zero on $[0, \infty)$: its derivative is $6\lambda(8\lambda - 3)$, so the unique interior minimum on $[0, \infty)$ is at $\lambda = 3/8$, where
\[
16\left(\frac{3}{8}\right)^3 - 9\left(\frac{3}{8}\right)^2 + 1 = \frac{37}{64} > 0.
\]
Hence the numerator of \eqref{eq:dy-dlambda} has no zero on $(1, \infty)$.
At $\lambda = 2$ it equals $2 \cdot 2 \cdot 5 - 11 = 9 > 0$, so continuity implies that it is positive throughout $(1,\infty)$.
Therefore $y_{\mathrm{phys}}$ is strictly increasing on $(1, \infty)$.

Since $y_{\mathrm{phys}}(1) = 0$ and $y_{\mathrm{phys}}(\lambda) \to +\infty$ as $\lambda \to +\infty$ (the leading term is $\lambda^2$), the map is a bijection $(1, \infty) \to (0, \infty)$.
\end{proof}

\subsection{Vacuum boundary and Puiseux expansion}\label{subsec:vacuum-puiseux}

The endpoint $y = 0$ is not an ordinary analytic point of the inverse function $\lambda_{\mathrm{phys}}(y)$, but a square-root ramification point.
The local expansion is a classical Puiseux series \cite{BrieskornKnorrer1986,Walker1950}.

\begin{proposition}[Vacuum Puiseux expansion]\label{prop:vacuum-puiseux}
Setting $\lambda = 1 + \varepsilon$, the dominant branch satisfies
\begin{equation}\label{eq:y-puiseux}
y_{\mathrm{phys}}(1 + \varepsilon)
= 2\varepsilon^2 - 5\varepsilon^3 + 21\varepsilon^4 - 94\varepsilon^5 + O(\varepsilon^6).
\end{equation}
Inverting, the dominant eigenvalue on the positive-real sheet expands as
\begin{equation}\label{eq:lambda-puiseux}
\lambda_{\mathrm{phys}}(y)
= 1 + \frac{1}{\sqrt{2}}\,y^{1/2}
+ \frac{5}{8}\,y
- \frac{43}{64\sqrt{2}}\,y^{3/2}
- \frac{1}{16}\,y^2
+ O(y^{5/2}).
\end{equation}
The logarithmic branch
$f_{\mathrm{dom}}(y) = \log\lambda_{\mathrm{phys}}(y)$
satisfies
\begin{equation}\label{eq:f-puiseux}
f_{\mathrm{dom}}(y)
= \frac{1}{\sqrt{2}}\,y^{1/2}
+ \frac{3}{8}\,y
- \frac{217\sqrt{2}}{384}\,y^{3/2}
+ O(y^2),
\end{equation}
and the logarithmic slope $\rho(y) = y\,f_{\mathrm{dom}}'(y)$ has the form
\begin{equation}\label{eq:rho-vacuum}
\rho(y) = \frac{1}{2\sqrt{2}}\,y^{1/2}
+ \frac{3}{8}\,y
- \frac{217\sqrt{2}}{256}\,y^{3/2}
+ O(y^2).
\end{equation}
\end{proposition}

\begin{proof}
At $\lambda = 1 + \varepsilon$,
\[
4\lambda^3 - 4\lambda + 1
= 1 + 8\varepsilon + 12\varepsilon^2 + 4\varepsilon^3.
\]
Taking the square root by the binomial series:
\[
\sqrt{4\lambda^3 - 4\lambda + 1}
= 1 + 4\varepsilon - 2\varepsilon^2 + 10\varepsilon^3 - 42\varepsilon^4 + 188\varepsilon^5 + O(\varepsilon^6).
\]
Substituting into \eqref{eq:y-phys}:
\begin{align*}
y_{\mathrm{phys}}(1 + \varepsilon)
&= (1 + \varepsilon)^2 - \tfrac{1}{2}
  - \tfrac{1}{2}\bigl(1 + 4\varepsilon - 2\varepsilon^2 + 10\varepsilon^3 - 42\varepsilon^4 + 188\varepsilon^5 + \cdots\bigr) \\
&= 1 + 2\varepsilon + \varepsilon^2 - \tfrac{1}{2}
  - \tfrac{1}{2} - 2\varepsilon + \varepsilon^2 - 5\varepsilon^3 + 21\varepsilon^4 - 94\varepsilon^5 - \cdots \\
&= 2\varepsilon^2 - 5\varepsilon^3 + 21\varepsilon^4 - 94\varepsilon^5 + O(\varepsilon^6).
\end{align*}
To invert the series, write
\[
\varepsilon = A y^{1/2} + B y + C y^{3/2} + D y^2 + O(y^{5/2}).
\]
Substituting this ansatz into \eqref{eq:y-puiseux} and matching the coefficients of
\(
y,\ y^{3/2},\ y^2,\ y^{5/2}
\)
gives
\[
2A^2=1,\qquad 4AB-5A^3=0,
\]
\[
4AC+2B^2-15A^2B+21A^4=0,
\]
\[
4AD+4BC-15A^2C-15AB^2+84A^3B-94A^5=0.
\]
Choosing the positive branch \(A=1/\sqrt2\) and solving successively yields
\[
A=\frac{1}{\sqrt2},\qquad
B=\frac58,\qquad
C=-\frac{43}{64\sqrt2},\qquad
D=-\frac{1}{16},
\]
which is \eqref{eq:lambda-puiseux}.
Then
\[
\log(1+\varepsilon)=\varepsilon-\frac{\varepsilon^2}{2}+\frac{\varepsilon^3}{3}+O(y^2)
\]
gives \eqref{eq:f-puiseux}, and term-by-term differentiation yields \eqref{eq:rho-vacuum}.
\end{proof}

Equation \eqref{eq:rho-vacuum} shows that the logarithmic slope does not open linearly from the vacuum boundary, but with a square-root singularity:
\[
\rho(y) \sim \frac{1}{2\sqrt{2}}\,y^{1/2}
\qquad (y \to 0^+).
\]

\subsection{A universal dominant-EP2 edge normal form}\label{subsec:vacuum-quantization}

The same square-root exceptional point controls the first finite-volume zeros on the negative real axis.
The relevant mechanism is not peculiar to the present quartic family, but is the universal local standard form of a dominant square-root EP2.

\begin{theorem}[Universal dominant-EP2 edge normal form]\label{thm:ep2-normal-form}
Let $\{Z_m(y)\}_{m\ge 0}$ admit, in a punctured neighborhood of $y_c$, a decomposition
\[
Z_m(y)=C_+(y)\lambda_+(y)^m + C_-(y)\lambda_-(y)^m + R_m(y).
\]
Assume there exists a Puiseux coordinate $y=y_c+s^2$ in which
\[
\lambda_\pm(s)=\lambda_c\exp\!\left(\pm as + bs^2 + O(s^3)\right),
\qquad
C_\pm(s)=C_0\mp cs+O(s^2),
\]
with real constants $\lambda_c>0$, $a>0$, $b\in\RR$, $C_0\neq 0$, and $c\in\RR$.
Assume moreover that for every compact $U\subset[0,\infty)$ one has
\[
R_m\!\left(y_c-\frac{u^2}{m^2}\right)=O_U(\lambda_c^m m^{-2})
\]
uniformly for $u\in U$.
Then, uniformly for $u\in U$,
\begin{equation}\label{eq:universal-ep2-scaling}
Z_m\!\left(y_c-\frac{u^2}{m^2}\right)
=
2C_0\lambda_c^m
\exp\!\left(-\frac{b u^2}{m}\right)
\left[
\cos\!\left(
a u-\frac{c}{C_0}\frac{u}{m}
\right)+O_U(m^{-2})
\right].
\end{equation}
Consequently, if $q_k:=(2k-1)\pi/2$, then the $k$th edge zero satisfies
\begin{equation}\label{eq:universal-ep2-u}
u_{m,k}
=
\frac{q_k}{a}
\left(
1+\frac{c}{aC_0m}
\right)
+O_k(m^{-2}),
\end{equation}
and therefore
\begin{equation}\label{eq:universal-ep2-y}
y_{m,k}
=
y_c-\frac{q_k^2}{a^2m^2}
-\frac{2cq_k^2}{a^3C_0m^3}
+O_k(m^{-4}).
\end{equation}
\end{theorem}

\begin{proof}
Fix a compact set $U\subset[0,\infty)$ and set $s=iu/m$, so that $y=y_c-u^2/m^2$.
From the assumed local form of $\lambda_\pm$,
\[
\lambda_\pm\!\left(\frac{iu}{m}\right)^m
=
\lambda_c^m
\exp\!\left(
\pm iau-\frac{bu^2}{m}+O_U(m^{-2})
\right).
\]
Similarly,
\[
C_\pm\!\left(\frac{iu}{m}\right)
=
C_0\mp ic\frac{u}{m}+O_U(m^{-2}).
\]
Multiplying and adding the two dominant contributions gives
\begin{align*}
C_+\lambda_+^m + C_-\lambda_-^m
&=
2C_0\lambda_c^m
\exp\!\left(-\frac{bu^2}{m}\right)
\left[
\cos(au)+\frac{c}{C_0}\frac{u}{m}\sin(au)+O_U(m^{-2})
\right].
\end{align*}
Using
\[
\cos(\theta-\delta)=\cos\theta+\delta\sin\theta+O(\delta^2),
\qquad
\delta=\frac{c}{C_0}\frac{u}{m},
\]
we obtain
\[
C_+\lambda_+^m + C_-\lambda_-^m
=
2C_0\lambda_c^m
\exp\!\left(-\frac{bu^2}{m}\right)
\left[
\cos\!\left(
au-\frac{c}{C_0}\frac{u}{m}
\right)+O_U(m^{-2})
\right].
\]
The assumed bound on $R_m$ then gives \eqref{eq:universal-ep2-scaling}.

Now fix $k\ge1$.
The principal phase in \eqref{eq:universal-ep2-scaling} has a simple zero near $u_k^{(0)}=q_k/a$.
Since its derivative equals $a+O(m^{-1})$, a standard one-dimensional perturbation argument yields a unique zero $u_{m,k}$ in that neighborhood.
Solving
\[
au-\frac{c}{C_0}\frac{u}{m}=q_k
\]
gives
\[
u_{m,k}
=
\frac{q_k}{a-\frac{c}{C_0m}}
=
\frac{q_k}{a}
\left(
1+\frac{c}{aC_0m}
\right)
+O_k(m^{-2}),
\]
which is \eqref{eq:universal-ep2-u}.
Squaring and dividing by $m^2$ yields \eqref{eq:universal-ep2-y}.
\end{proof}

\begin{theorem}[Vacuum-edge scaling and zero quantization]\label{thm:vacuum-zero-quantization}
For the quartic transfer family studied here, the vacuum edge at $y_c=0$ satisfies the universal dominant-EP2 normal form with
\[
a=\frac{1}{\sqrt2},
\qquad
b=\frac38,
\qquad
C_0=\frac12,
\qquad
c=\frac{3\sqrt2}{16}.
\]
Consequently, uniformly for $u$ in compact subsets of $[0,\infty)$,
\begin{equation}\label{eq:vacuum-scaling}
Z_m\!\left(-\frac{u^2}{m^2}\right)
=
\exp\!\left(-\frac{3u^2}{8m}\right)
\left[
\cos\!\left(\frac{u}{\sqrt{2}}-\frac{3u}{4\sqrt{2}\,m}\right)
+ O_U(m^{-2})
\right],
\end{equation}
and for each fixed $k\ge 1$ the $k$th negative real zero nearest $y=0$ satisfies
\begin{equation}\label{eq:vacuum-zero-quantization}
y_{m,k}
=
-\frac{\pi^2(2k-1)^2}{2m^2}
-\frac{3\pi^2(2k-1)^2}{4m^3}
+ O_k(m^{-4}).
\end{equation}
\end{theorem}

\begin{proof}
Appendix~\ref{app:vacuum-edge} proves that on the vacuum scale $y=s^2$ the dominant pair satisfies
\[
\lambda_\pm(s)=1\pm \frac{1}{\sqrt2}s+\frac58 s^2+O(s^3),
\qquad
C_\pm(s)=\frac12\mp \frac{3\sqrt2}{16}s+O(s^2),
\]
and that the remaining two branches contribute only $O_U(m^{-2})$ on the scale $y=-u^2/m^2$.
Thus the hypotheses of Theorem~\ref{thm:ep2-normal-form} hold with the stated constants and $\lambda_c=1$.
Substituting these values into \eqref{eq:universal-ep2-scaling} and \eqref{eq:universal-ep2-y} gives \eqref{eq:vacuum-scaling} and \eqref{eq:vacuum-zero-quantization}.
\end{proof}

\begin{corollary}[Edge-zero counting law]\label{cor:edge-zero-counting}
Define
\[
N_m(\xi)
:=
\#\left\{
k\ge 1:
y_{m,k}\ge -\frac{\xi}{m^2}
\right\},
\qquad
\xi>0.
\]
If $\xi$ does not belong to the threshold set
\[
\left\{
\frac{\pi^2(2k-1)^2}{2}:k\ge1
\right\},
\]
then for all sufficiently large $m$ one has
\begin{equation}\label{eq:edge-counting-law}
N_m(\xi)
=
\left\lfloor
\frac{\sqrt{\xi}}{\pi\sqrt2}+\frac12
\right\rfloor.
\end{equation}
\end{corollary}

\begin{proof}
By \eqref{eq:vacuum-zero-quantization},
\[
-m^2 y_{m,k}
=
\frac{\pi^2(2k-1)^2}{2}+O_k(m^{-1}).
\]
If $\xi$ is not a threshold value, then for each fixed $k$ the sign of
\(
\xi+m^2y_{m,k}
\)
agrees, for all sufficiently large $m$, with the sign of
\(
\xi-\pi^2(2k-1)^2/2.
\)
Thus the set of indices counted by $N_m(\xi)$ stabilizes and is exactly
\[
\left\{
k\ge1:
\frac{\pi^2(2k-1)^2}{2}<\xi
\right\},
\]
which yields \eqref{eq:edge-counting-law}.
\end{proof}

In continuum form, \eqref{eq:vacuum-zero-quantization} gives
\[
k \sim \frac{m}{\pi\sqrt2}\sqrt{|y|},
\qquad
\left|\frac{\dd k}{\dd y}\right|
\sim
\frac{m}{2\pi\sqrt{2|y|}}
\qquad
(y\uparrow 0,\ |y|=O(m^{-2})).
\]
Thus the first negative zeros do not approach the vacuum edge with a smooth density: they compress with the same square-root exponent that governs the Puiseux opening of the dominant sheet.

\subsection{Double tangency at infinity and the dense-edge splitting law}\label{subsec:high-fugacity}

\begin{proposition}[Double tangency at projective infinity]\label{prop:saturation}
Let $Y:=y^{1/4}$.
Then the dominant homogeneous balance of the spectral polynomial at projective infinity is
\begin{equation}\label{eq:infinity-balance}
\Pi_\infty(\lambda,y)
:=
\lambda^4-2y\lambda^2+y^2
=(\lambda^2-y)^2.
\end{equation}
Hence the positive large-\(y\) branch is generated by a double tangency of the sheets $\lambda\sim \pm y^{1/2}$ at infinity.
On the dominant branch one has
\begin{equation}\label{eq:lambda-high-y}
\lambda_{\mathrm{phys}}(y)
= y^{1/2} + \frac{1}{2}\,y^{1/4} + \frac{1}{4}
+ \frac{7}{64}\,y^{-1/4} + O(y^{-1/2}),
\end{equation}
and the logarithmic branch satisfies
\begin{equation}\label{eq:f-high-y}
f_{\mathrm{dom}}(y)
= \frac{1}{2}\log y + \frac{1}{2}\,y^{-1/4}
+ \frac{1}{8}\,y^{-1/2} + O(y^{-3/4}).
\end{equation}
Writing $y = e^s$ and $\Lambda_{\mathrm{dom}}(s) = f_{\mathrm{dom}}(e^s) - \log 2$, the logarithmic slope
\begin{equation}\label{eq:rho-saturation}
\rho(s) = \Lambda_{\mathrm{dom}}'(s)
= \frac{1}{2} - \frac{1}{8}\,e^{-s/4}
- \frac{1}{16}\,e^{-s/2} + O(e^{-3s/4})
\end{equation}
satisfies
\[
\lim_{s \to +\infty} \rho(s) = \frac{1}{2}.
\]
The exponent $1/4$ is the splitting exponent created by the double tangency at infinity.
\end{proposition}

\begin{proof}
Set $y=Y^4$ and seek the positive dense branch in the form
\[
\lambda=Y^2+\alpha Y+\beta+\gamma Y^{-1}+O(Y^{-2}).
\]
Substituting into $\Pi(\lambda,Y^4)$ gives
\begin{align}
\Pi(\lambda,Y^4)
&=
Y^6(4\alpha^2-1) \notag\\
&\quad
+Y^5(4\alpha^3+8\alpha\beta-3\alpha) \notag\\
&\quad
+Y^4(\alpha^4+12\alpha^2\beta-3\alpha^2+8\alpha\gamma+4\beta^2-3\beta)
+O(Y^3). \label{eq:infinity-expansion}
\end{align}
The leading coefficient $4\alpha^2-1$ is precisely the first-order splitting of the double balance \eqref{eq:infinity-balance}, so the dominant branch must satisfy $\alpha=1/2$.
The vanishing of the $Y^5$ coefficient then gives
\[
4\left(\frac12\right)^3+8\left(\frac12\right)\beta-3\left(\frac12\right)=0,
\qquad\text{hence}\qquad
\beta=\frac14.
\]
Using these values in the $Y^4$ coefficient of \eqref{eq:infinity-expansion} yields
\[
\frac{1}{16}+12\cdot \frac14\cdot \frac14-\frac34+4\gamma+4\cdot \frac{1}{16}-\frac34=0,
\qquad\text{hence}\qquad
\gamma=\frac{7}{64}.
\]
This proves \eqref{eq:lambda-high-y}.
Equations \eqref{eq:f-high-y} and \eqref{eq:rho-saturation} then follow by logarithmic expansion and differentiation with respect to $s=\log y$.
\end{proof}

The point of Proposition~\ref{prop:saturation} is conceptual as well as computational.
The dense-edge correction exponent $1/4$ is not merely a quartic fingerprint; it is the quarter-power splitting law generated by a double tangency at projective infinity.
Combined with Theorem~\ref{thm:support-law}, which gives only polynomial growth of the top-degree coefficients, this shows that the top-degree sector is a rigid edge rather than an extensive multiplicity regime.
