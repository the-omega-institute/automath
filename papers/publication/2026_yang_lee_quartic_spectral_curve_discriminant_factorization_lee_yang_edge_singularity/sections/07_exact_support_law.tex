\section{Degree growth and top-degree coefficients}\label{sec:support-law}

The large-\(y\) asymptotics of Section~\ref{sec:physical-sheet} have an exact algebraic counterpart in the \(y\)-degree of the polynomials \(Z_m(y)\) and in their top-degree coefficients.

\begin{theorem}[Degree law and top-degree coefficients]\label{thm:support-law}
Let $d_m = \deg_y Z_m(y)$ and $L_m = [y^{d_m}]\,Z_m(y)$.
Then $d_1 = 1$, and for every $m \ge 2$,
\begin{equation}\label{eq:degree-formula}
d_m = \left\lfloor \frac{m+3}{2} \right\rfloor.
\end{equation}
The leading coefficients are given by
\begin{equation}\label{eq:leading-odd}
L_{2n+1} = n \qquad (n \ge 1),
\end{equation}
\begin{equation}\label{eq:leading-even}
L_{2n} = \frac{n(n^2 + 5)}{6} \qquad (n \ge 1).
\end{equation}
Consequently,
\begin{equation}\label{eq:half-filling}
\lim_{m \to \infty} \frac{d_m}{m} = \frac{1}{2},
\end{equation}
and the top-degree coefficients are subexponential:
\[
L_m = e^{o(m)}.
\]
\end{theorem}

\begin{proof}
The proof proceeds by induction on the recurrence \eqref{eq:recurrence},
\[
Z_m = Z_{m-1} + (2y+1)\,Z_{m-2} - Z_{m-3} - y(y+1)\,Z_{m-4}.
\]

\noindent\textbf{Initial values.}
Direct computation gives
\[
d_1 = 1, \qquad d_2 = 2, \qquad d_3 = 3, \qquad d_4 = 3.
\]
The formula \eqref{eq:degree-formula} is intended for $m \ge 2$ and matches the cases $m=2,3,4$:
$\lfloor 5/2 \rfloor = 2$, $\lfloor 6/2 \rfloor = 3$, $\lfloor 7/2 \rfloor = 3$.
The leading coefficients are $L_1 = 1$, $L_2 = 1$, $L_3 = 1$, $L_4 = 3$.
These match \eqref{eq:leading-odd}--\eqref{eq:leading-even}: $L_3 = 1$ and $L_4 = 2(4+5)/6 = 3$.

\medskip
\noindent\textbf{Degree formula for odd index $m = 2n+1$.}
Assume by induction that
\[
d_{2n} = n+1, \quad d_{2n-1} = n+1, \quad d_{2n-2} = n, \quad d_{2n-3} = n.
\]
In the recurrence for $Z_{2n+1}$, the four terms contribute degrees
\[
d_{2n},\quad 1 + d_{2n-1},\quad d_{2n-2},\quad 2 + d_{2n-3},
\]
respectively equal to $n+1$, $n+2$, $n$, $n+2$.
Hence $d_{2n+1} = n + 2 = \lfloor (2n+4)/2 \rfloor$.

\medskip
\noindent\textbf{Leading coefficient for odd index.}
At the top degree $n+2$, the terms of degree $n+2$ come from $(2y+1)Z_{2n-1}$ and $-y(y+1)Z_{2n-3}$.
Hence
\begin{equation}\label{eq:L-odd-recurrence}
L_{2n+1} = 2L_{2n-1} - L_{2n-3}.
\end{equation}
With $L_3 = 1$ and $L_5 = 2$, this recurrence has the solution $L_{2n+1} = n$.

\medskip
\noindent\textbf{Degree formula for even index $m = 2n$.}
Similarly, the inductive hypothesis gives degrees
\[
d_{2n-1},\quad 1 + d_{2n-2},\quad d_{2n-3},\quad 2 + d_{2n-4}
\]
equal to $n+1$, $n+1$, $n$, $n+1$.
Hence $d_{2n} = n+1 = \lfloor(2n+3)/2\rfloor$.

\medskip
\noindent\textbf{Leading coefficient for even index.}
At the top degree $n+1$, contributions come from $Z_{2n-1}$, $(2y+1)Z_{2n-2}$, and $-y(y+1)Z_{2n-4}$.
Hence
\begin{equation}\label{eq:L-even-recurrence}
L_{2n} = L_{2n-1} + 2L_{2n-2} - L_{2n-4}.
\end{equation}
Since $L_{2n-1}=n-1$, define $a_n = L_{2n}$.
Then
\begin{equation}\label{eq:a-recurrence}
a_n - 2a_{n-1} + a_{n-2} = n - 1,
\qquad a_1 = 1, \quad a_2 = 3.
\end{equation}
The homogeneous solution is $C_1 + C_2 n$.
A particular solution is $a_n^{(p)} = n^3/6$, so
\[
a_n = \frac{n^3}{6} + C_1 + C_2 n.
\]
Imposing $a_1 = 1$ and $a_2 = 3$ gives $C_1 = 0$ and $C_2 = 5/6$.
Therefore
\[
a_n = \frac{n^3}{6} + \frac{5n}{6} = \frac{n(n^2 + 5)}{6}.
\]
\end{proof}

\begin{remark}\label{rem:hard-support}
Theorem~\ref{thm:support-law} shows that the maximal \(y\)-support of \(Z_m(y)\) grows linearly with slope \(1/2\).
Together with the subexponential growth of \(L_m\), this identifies the top-degree sector as a rigid boundary sector rather than an exponentially large multiplicity regime.
\end{remark}
