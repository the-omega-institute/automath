\section{Genus-one reduction of the quartic spectral curve}\label{sec:genus-one}

The present spectral polynomial is not an isolated quartic curiosity.
Its reduction by the substitution \(\xi=y-\lambda^2\) is the specialization of a balanced family whose coefficients are exactly tuned so that the unwanted \(\lambda^4\), \(\xi\lambda^2\), and \(\lambda^2\) terms cancel simultaneously.
This section first identifies that family and then specializes it to the quartic transfer polynomial \(\Pi(\lambda,y)\).

\subsection{A balanced quartic family}

\begin{proposition}[Balanced quartic family]\label{prop:balanced-family}
For parameters \(a,c,d,g\in\CC\), define
\begin{equation}\label{eq:balanced-family}
P_{a,c,d,g}(\lambda,y)
=
\lambda^4+a\lambda^3+(c-2y)\lambda^2+d\lambda+y(y-c)+g.
\end{equation}
Set
\begin{equation}\label{eq:balanced-xi}
\xi=y-\lambda^2,
\qquad
Y=2\xi-c=2y-2\lambda^2-c.
\end{equation}
Then the affine curve \(P_{a,c,d,g}(\lambda,y)=0\) is isomorphic to the affine curve
\begin{equation}\label{eq:balanced-weierstrass}
Y^2=c^2-4a\lambda^3-4d\lambda-4g.
\end{equation}
The inverse map is
\begin{equation}\label{eq:balanced-inverse}
y=\lambda^2+\frac{Y+c}{2}.
\end{equation}
If \(a\neq 0\) and the cubic polynomial
\[
F_{a,c,d,g}(\lambda)=c^2-4a\lambda^3-4d\lambda-4g
\]
has no repeated root, then the smooth projective completion \(E_{a,c,d,g}\) is a genus-one curve, and the function \(y\) defines a degree-\(4\) map
\[
y:E_{a,c,d,g}\to\mathbf P^1_y.
\]
\end{proposition}

\begin{proof}
Using \(\xi=y-\lambda^2\), one rewrites \eqref{eq:balanced-family} as
\[
P_{a,c,d,g}(\lambda,y)
=
(y-\lambda^2)^2-c(y-\lambda^2)+a\lambda^3+d\lambda+g
=
\xi^2-c\xi+a\lambda^3+d\lambda+g.
\]
Thus
\[
P_{a,c,d,g}(\lambda,y)=0
\iff
\xi^2-c\xi+a\lambda^3+d\lambda+g=0.
\]
Multiplying by \(4\) and completing the square gives
\[
(2\xi-c)^2=c^2-4a\lambda^3-4d\lambda-4g.
\]
This is \eqref{eq:balanced-weierstrass} with \(Y=2\xi-c\).
Conversely, from \(Y=2\xi-c\) one gets \(\xi=(Y+c)/2\), hence \eqref{eq:balanced-inverse}.
Therefore the affine change of variables \((\lambda,y)\leftrightarrow(\lambda,Y)\) is polynomial and invertible, so the two affine curves are isomorphic.

Assume now that \(a\neq 0\) and \(F_{a,c,d,g}\) has no repeated root.
Then \eqref{eq:balanced-weierstrass} is a smooth cubic Weierstrass model.
The projection \((\lambda,Y)\mapsto \lambda\) is a double cover of \(\mathbf P^1_\lambda\) branched at the three finite roots of \(F_{a,c,d,g}\) and, because \(\deg F_{a,c,d,g}=3\), also at infinity.
Hence there are four simple branch points.
By Riemann--Hurwitz,
\[
2g-2=2(-2)+4,
\]
so \(g=1\).

Let \(O\) be the point at infinity on the smooth projective completion.
For a cubic Weierstrass model one has
\[
\operatorname{ord}_O(\lambda)=-2,
\qquad
\operatorname{ord}_O(Y)=-3.
\]
Using \eqref{eq:balanced-inverse},
\[
\operatorname{ord}_O(y)
=
\operatorname{ord}_O\!\left(\lambda^2+\frac{Y+c}{2}\right)
=
\min(-4,-3)
=
-4.
\]
Therefore \(y\) has a unique pole of order \(4\) at \(O\), so the map \(y:E_{a,c,d,g}\to\mathbf P^1_y\) has degree \(4\).
\end{proof}

\begin{corollary}[Smooth genus-one condition]\label{cor:balanced-smooth}
For \(a\neq 0\), the cubic \(F_{a,c,d,g}\) has no repeated root if and only if
\begin{equation}\label{eq:balanced-genus-condition}
64d^3+27a(c^2-4g)^2\neq 0.
\end{equation}
\end{corollary}

\begin{proof}
Write
\[
F_{a,c,d,g}(\lambda)
=
-4a\left(\lambda^3+\frac{d}{a}\lambda+\frac{4g-c^2}{4a}\right).
\]
For a monic cubic \(u^3+pu+q\), the discriminant is \(-4p^3-27q^2\).
Substituting
\[
p=\frac{d}{a},
\qquad
q=\frac{4g-c^2}{4a}
\]
gives the condition
\[
-4\left(\frac{d}{a}\right)^3-27\left(\frac{4g-c^2}{4a}\right)^2\neq 0,
\]
which is equivalent to \eqref{eq:balanced-genus-condition} after clearing denominators.
\end{proof}

\begin{corollary}[Finite ramification in the balanced family]\label{cor:balanced-ramification}
Let \(a\neq 0\) and assume \eqref{eq:balanced-genus-condition}.
Then the finite ramification points of the degree-\(4\) map \(y:E_{a,c,d,g}\to\mathbf P^1_y\) satisfy
\begin{equation}\label{eq:balanced-ramification-eq}
2\lambda Y=3a\lambda^2+d.
\end{equation}
Eliminating \(Y\) with \eqref{eq:balanced-weierstrass} yields the quintic equation
\begin{equation}\label{eq:balanced-ramification}
4\lambda^2\bigl(c^2-4a\lambda^3-4d\lambda-4g\bigr)-(3a\lambda^2+d)^2=0,
\end{equation}
equivalently
\begin{equation}\label{eq:balanced-quintic}
16a\lambda^5+9a^2\lambda^4+16d\lambda^3+(16g+6ad-4c^2)\lambda^2+d^2=0.
\end{equation}
\end{corollary}

\begin{proof}
Differentiating \(\xi^2-c\xi+a\lambda^3+d\lambda+g=0\) gives
\[
(2\xi-c)\,\dd\xi+(3a\lambda^2+d)\,\dd\lambda=0.
\]
Since \(Y=2\xi-c\),
\[
\dd\xi=-\frac{3a\lambda^2+d}{Y}\,\dd\lambda.
\]
Using \(y=\lambda^2+\xi\), we obtain
\[
\dd y
=
\left(2\lambda-\frac{3a\lambda^2+d}{Y}\right)\dd\lambda
=
\frac{2\lambda Y-(3a\lambda^2+d)}{Y}\,\dd\lambda.
\]
Therefore the finite ramification condition is \eqref{eq:balanced-ramification-eq}.
Squaring and substituting \(Y^2=c^2-4a\lambda^3-4d\lambda-4g\) gives \eqref{eq:balanced-ramification}, and expanding gives \eqref{eq:balanced-quintic}.
\end{proof}

\subsection{Real projection structure for the balanced family}

The following theorem establishes a general criterion, applicable to the entire balanced family \(P_{a,c,d,g}\) with real parameters, that determines whether the real branch locus of the degree-\(4\) map $y:E_{a,c,d,g}\to\mathbf P^1_y$ consists of one or two disjoint real intervals.
This is the structural statement of the paper that goes beyond the single specialization.

\begin{theorem}[Real projection criterion for the balanced family]\label{thm:balanced-real-projection}
Let $a,c,d,g\in\RR$ with $a\neq0$ and assume \eqref{eq:balanced-genus-condition}, so that $E_{a,c,d,g}$ is a smooth genus-one curve.
Let $F(\lambda)=c^2-4a\lambda^3-4d\lambda-4g$ be the cubic in \eqref{eq:balanced-weierstrass}.

\begin{enumerate}
\item The real locus $E_{a,c,d,g}(\RR)$ is connected (a single oval) if and only if $F$ has exactly one real root.
It has two connected components (two real ovals) if and only if $F$ has three distinct real roots.

\item The real $y$-projection $y(E_{a,c,d,g}(\RR))\subset\RR$ is a single closed interval if and only if $E_{a,c,d,g}(\RR)$ is connected and all finite real critical values of $y$ on that oval lie in the interior of the image interval.

\item The real $y$-projection consists of two disjoint closed intervals if and only if $E_{a,c,d,g}(\RR)$ has two real ovals and each oval maps surjectively onto one of the two intervals under $y$.
In this case the two intervals are separated by a gap of width equal to the difference between the largest critical value on the compact oval and the smallest critical value on the noncompact oval.
\end{enumerate}
\end{theorem}

\begin{proof}
The real locus of the Weierstrass model $Y^2=F(\lambda)$ with $F(\lambda)=-4a\lambda^3+c^2-4d\lambda-4g$ consists of those real $\lambda$ where $F(\lambda)\ge0$.
Since the leading coefficient of $F$ is $-4a$, the sign of $F$ at $+\infty$ is $-\operatorname{sgn}(a)$.

If $F$ has three distinct real roots $\beta_1<\beta_2<\beta_3$, the sign of $F$ alternates between consecutive roots.
If $-4a>0$ (i.e., $a<0$), then $F>0$ on $(-\infty,\beta_1)$ and on $(\beta_2,\beta_3)$; if $-4a<0$ (i.e., $a>0$), then $F>0$ on $(\beta_1,\beta_2)$ and on $(\beta_3,+\infty)$.
In either case the real locus of $Y^2=F(\lambda)$ has two connected components (including the point at infinity of the noncompact component), each homeomorphic to a circle.
If $F$ has one real root $\beta_1$ (and two complex conjugate roots), then $F$ is positive on one side of $\beta_1$ and negative on the other, so the real locus has exactly one component, the noncompact oval.
This proves~(1).

For~(2) and~(3), the real $y$-projection $y(\lambda,Y)=\lambda^2+\xi=\lambda^2+(Y+c)/2$ is a continuous real-valued function on each compact connected component of $E_{a,c,d,g}(\RR)$.
By the intermediate value theorem, the image of each component under $y$ is a closed interval (possibly degenerate), bounded by the minimum and maximum values of $y$ on that component.
These extrema occur either at finite critical points of $y$ on $E_{a,c,d,g}(\RR)$ (given by the real solutions of \eqref{eq:balanced-ramification-eq}) or at the boundary of the component (the branch points of $F$, where $Y=0$).
Since $y$ has a pole at the point at infinity on the noncompact oval, the noncompact component maps surjectively onto a ray $[y_{\min},+\infty)$ for some finite $y_{\min}$.

If the two components are separated by a gap in the $\lambda$-direction, and if the images of the two components under $y$ are disjoint (no critical value of the compact oval falls in the image of the noncompact oval and vice versa), then the total real $y$-projection consists of two disjoint intervals as described in~(3).
The width of the gap is the difference between the extremal $y$-values on the two components.
If instead the images overlap or share an endpoint, the total real projection is a single interval.
\end{proof}

\begin{corollary}[Two-interval structure for the present specialization]\label{cor:two-interval-specialization}
For the specialization $(a,c,d,g)=(-1,-1,1,0)$ the cubic $F(\lambda)=4\lambda^3-4\lambda+1$ has three distinct real roots $\beta_1<\beta_2<\beta_3$ (the discriminant $\Delta=37>0$ confirms this).
Thus $E_{-1,-1,1,0}(\RR)$ has two real ovals.
The compact oval maps under $y$ to the interval $[y_{\mathrm{sub}},1]$ and the noncompact oval maps to $[0,+\infty)$; these two intervals meet only at $y=0$ (a branch value) as their adjacent endpoints, which is the touching point described in Theorem~\ref{thm:real-phase-diagram}.
For $y<y_{\mathrm{sub}}$ the real $y$-fiber is empty, reflecting the two-interval structure predicted by Theorem~\ref{thm:balanced-real-projection}(3).
\end{corollary}

\begin{remark}\label{rem:balanced-family-generality}
Theorem~\ref{thm:balanced-real-projection} applies to any balanced family member with real parameters.
The criterion in~(1) depends only on the discriminant of the cubic $F(\lambda)$, which factors as \eqref{eq:balanced-genus-condition}.
The separation in~(3) is controlled by the real critical values of $y$ on $E_{a,c,d,g}$, which are the real branch values and can be computed explicitly from the ramification quintic \eqref{eq:balanced-quintic}.
The specialization $(a,c,d,g)=(-1,-1,1,0)$ is the simplest member of the balanced family with $g=0$ and an explicit two-interval structure; the general case records when the compact and noncompact oval images are disjoint, adjacent, or overlapping.
\end{remark}

\subsection{Canonical normalization inside the balanced family}

Two elementary operations preserve the balanced form: a shift \(y\mapsto y+\beta\) and a simultaneous scaling \((\lambda,y)\mapsto(L\lambda,L^2 y)\).
They allow us to normalize the dominant square-root endpoint and motivate the canonical choice of parameters.

\begin{lemma}[Shift and scaling]\label{lem:balanced-affine}
Let $P_{a,c,d,g}$ be given by \eqref{eq:balanced-family}.
\begin{enumerate}
\item For any $\beta\in\RR$, the polynomial
\[
\widehat P(\lambda,y):=P_{a,c,d,g}(\lambda,y+\beta)
\]
equals $P_{a,c',d,g'}(\lambda,y)$ with $c'=c-2\beta$ and $g'=g+\beta(\beta-c)$.
\item For any $L\neq 0$, the change of variables $(\lambda,y)=(L\widetilde\lambda,L^2\widetilde y)$ yields
\[
P_{a,c,d,g}(L\widetilde\lambda,L^2\widetilde y)=L^4 P_{\widetilde a,\widetilde c,\widetilde d,\widetilde g}(\widetilde\lambda,\widetilde y),
\]
where $\widetilde a=a/L$, $\widetilde c=c/L^2$, $\widetilde d=d/L^3$, and $\widetilde g=g/L^4$.
\end{enumerate}
\end{lemma}

\begin{proof}
Both statements follow from direct substitution.
For (1), expand $(y+\beta)((y+\beta)-c)+g=y(y-(c-2\beta))+\beta(\beta-c)+g$ and note that the remaining coefficients do not change.
For (2), divide by $L^4$ after substituting $(L\widetilde\lambda,L^2\widetilde y)$ into \eqref{eq:balanced-family}.
\end{proof}

Lemma~\ref{lem:balanced-affine} shows that we may shift to arrange that the dominant square-root branch sits at $y=0$ and scale so that the colliding positive eigenvalue equals $1$.
The following theorem solves the resulting algebraic constraints when the neighboring real branch is normalized at $y=1$.

\begin{theorem}[Canonical balanced member]\label{thm:balanced-classification}
Let $a,c,d,g\in\RR$.
Assume that the balanced polynomial $P_{a,c,d,g}$ has double roots at $(\lambda,y)=(1,0)$ and $(\lambda,y)=(-1,1)$.
Then $(a,c,d,g)=(-1,-1,1,0)$.
\end{theorem}

\begin{proof}
Set $Q(\lambda,y)=P_{a,c,d,g}(\lambda,y)$.
The two double-root conditions mean that $Q(1,0)=Q_\lambda(1,0)=Q(-1,1)=Q_\lambda(-1,1)=0$.
Direct evaluation gives four linear equations:
\[
1+a+c+d+g=0,\qquad
4+3a+2c+d=0,
\]
\[
-a-d+g=0,\qquad
3a-2c+d=0.
\]
Solving yields $c=-1$, $a=-1$, $d=1$, and $g=0$.
\end{proof}

\begin{remark}\label{rem:canonical-normalization}
Lemma~\ref{lem:balanced-affine} shows that any balanced family member whose dominant branch collides at a positive eigenvalue $\lambda_*>0$ over a real value $y_0$ can be normalized to the setup of Theorem~\ref{thm:balanced-classification} by shifting $y\mapsto y-y_0$ and rescaling $(\lambda,y)\mapsto (\lambda/\lambda_*,y/\lambda_*^2)$.
The assumption that the neighboring real branch is located at $y=1$ is therefore a canonical normalization: it fixes the only remaining affine freedom on the $y$-line.
Under this normalization the structural data are forced to coincide with the quartic treated in this paper, which explains why $(a,c,d,g)=(-1,-1,1,0)$ is the minimal balanced representative with the properties requested by the referee.
\end{remark}

\subsection{Explicit elimination example}

The symbolic elimination that produces the ramification quintic \eqref{eq:balanced-quintic} can be displayed concretely.
Starting from \eqref{eq:balanced-ramification-eq} and \eqref{eq:balanced-weierstrass}, eliminate $Y$ as follows.
Equation \eqref{eq:balanced-ramification-eq} gives $Y=(3a\lambda^2+d)/(2\lambda)$, provided $\lambda\neq 0$ (the $\lambda=0$ case is trivial and leads again to $d=0$, excluded here).
Substituting into $Y^2=c^2-4a\lambda^3-4d\lambda-4g$ yields
\[
\frac{(3a\lambda^2+d)^2}{4\lambda^2}=c^2-4a\lambda^3-4d\lambda-4g.
\]
Clearing denominators produces
\[
(3a\lambda^2+d)^2=4\lambda^2\bigl(c^2-4a\lambda^3-4d\lambda-4g\bigr).
\]
Expanding both sides and collecting powers of $\lambda$ gives the degree-$5$ polynomial
\[
16a\lambda^5+9a^2\lambda^4+16d\lambda^3+(16g+6ad-4c^2)\lambda^2+d^2,
\]
which is exactly \eqref{eq:balanced-quintic}.
The factorization used later (for example, in \eqref{eq:ramification-poly}) was verified algebraically and checked by the short SageMath script archived with the reproducibility bundle.
This explicit elimination is typical of the symbolic steps that appear elsewhere in the paper; whenever the resulting polynomial is too long to display, Appendix~\ref{app:pseudocode} references the auxiliary scripts used to confirm the computation.

\subsection{Specialization to the quartic transfer family}

\begin{theorem}[Affine elliptic model and branch geometry]\label{thm:elliptic-reduction}
The spectral polynomial \(\Pi(\lambda,y)\) from \eqref{eq:spectral-poly} is the specialization
\[
\Pi(\lambda,y)=P_{-1,-1,1,0}(\lambda,y).
\]
Under the substitution
\begin{equation}\label{eq:xi-def}
\xi = y - \lambda^2, \qquad Y = 2\xi + 1 = 2y + 1 - 2\lambda^2,
\end{equation}
the affine spectral curve \(\Pi(\lambda,y)=0\) is isomorphic to the affine elliptic model
\begin{equation}\label{eq:elliptic-xi}
\xi^2 + \xi = \lambda^3 - \lambda,
\end{equation}
or equivalently
\begin{equation}\label{eq:elliptic-weierstrass}
Y^2 = 4\lambda^3 - 4\lambda + 1.
\end{equation}
Its Weierstrass invariants are
\begin{equation}\label{eq:weierstrass-invariants}
g_2 = 4, \qquad g_3 = -1, \qquad
\Delta = g_2^3 - 27g_3^2 = 37, \qquad
j = \frac{1728\,g_2^3}{\Delta} = \frac{110592}{37}.
\end{equation}
The function
\begin{equation}\label{eq:y-from-E}
y = \lambda^2 + \xi = \lambda^2 - \tfrac{1}{2} + \tfrac{1}{2}Y
\end{equation}
defines a degree-\(4\) map \(E\to \mathbf P^1_y\).
Its finite branch values are exactly
\begin{equation}\label{eq:branch-values}
y = 0, \qquad y = 1, \qquad
\text{and the three roots of } c(y) = 256y^3 + 411y^2 + 165y + 32 = 0.
\end{equation}
\end{theorem}

\begin{proof}
Substituting \((a,c,d,g)=(-1,-1,1,0)\) into \eqref{eq:balanced-family} gives
\begin{equation}\label{eq:Pi-as-quadratic-in-y}
\Pi(\lambda,y)
=
y^2+(1-2\lambda^2)y+\lambda^4-\lambda^3-\lambda^2+\lambda.
\end{equation}
Equivalently,
\[
\Pi(\lambda,y)
=
(y-\lambda^2)^2+(y-\lambda^2)-\lambda^3+\lambda.
\]
Thus Proposition~\ref{prop:balanced-family} gives the affine isomorphism \eqref{eq:elliptic-xi}--\eqref{eq:elliptic-weierstrass} and the formula \eqref{eq:y-from-E}.

The smoothness condition follows from Corollary~\ref{cor:balanced-smooth}:
\[
64d^3+27a(c^2-4g)^2
=
64\cdot 1^3+27(-1)(1-0)^2
=
37\neq 0.
\]
Hence the projective completion is genus one.
The invariants in \eqref{eq:weierstrass-invariants} are read off directly from \(Y^2=4\lambda^3-4\lambda+1\).
The degree statement follows from Proposition~\ref{prop:balanced-family}.

For the finite ramification points, Corollary~\ref{cor:balanced-ramification} gives
\[
-16\lambda^5+9\lambda^4+16\lambda^3-10\lambda^2+1=0.
\]
Multiplying by \(-1\) and factoring yields
\begin{equation}\label{eq:ramification-poly}
(\lambda - 1)(\lambda + 1)(16\lambda^3 - 9\lambda^2 + 1) = 0.
\end{equation}
The corresponding branch values are obtained from
\[
y = \lambda^2 + \xi = \frac{4\lambda^3 - 3\lambda^2 - 2\lambda + 1}{4\lambda}.
\]
At \(\lambda=1\), this gives \(y=0\); at \(\lambda=-1\), it gives \(y=1\).
Eliminating \(\lambda\) between \(16\lambda^3-9\lambda^2+1=0\) and the displayed expression for \(y\) yields the cubic branch polynomial \(c(y)=256y^3+411y^2+165y+32\).
\end{proof}

\begin{lemma}[Projective normalization of the quartic]\label{lem:projective-normalization}
Let \(\overline C\subset \mathbf P^2_{[\Lambda:W:Z]}\) be the projective closure of the affine quartic \(\Pi(\lambda,y)=0\), namely
\[
\overline C=\{F(\Lambda,W,Z)=0\},
\]
where
\begin{equation}\label{eq:projective-quartic}
F(\Lambda,W,Z)
=
\Lambda^4-\Lambda^3Z-(2WZ+Z^2)\Lambda^2+\Lambda Z^3+W(W+Z)Z^2.
\end{equation}
Then \(\overline C\) has a unique point at infinity, namely \([0:1:0]\), and this point is singular.
The morphism
\begin{equation}\label{eq:normalization-map}
\nu:E\longrightarrow \overline C,
\qquad
(\lambda,Y)\longmapsto
\left[\lambda:\lambda^2-\tfrac12+\tfrac12Y:1\right]
\end{equation}
extends across the point at infinity \(O\in E\), satisfies
\[
\nu^{-1}([0:1:0])=\{O\},
\]
and is finite and birational.
Consequently, \(E\) is the normalization of \(\overline C\).
\end{lemma}

\begin{proof}
Homogenizing \(\Pi(\lambda,y)\) gives \eqref{eq:projective-quartic}.
On the line \(Z=0\), one has \(F(\Lambda,W,0)=\Lambda^4\), so the only point at infinity is \([0:1:0]\).
The partial derivatives are
\[
F_\Lambda=4\Lambda^3-3\Lambda^2Z-2(2WZ+Z^2)\Lambda+Z^3,
\]
\[
F_W=-2Z\Lambda^2+(2W+Z)Z^2,
\]
\[
F_Z=-\Lambda^3-2(W+Z)\Lambda^2+3\Lambda Z^2+2W^2Z+3WZ^2,
\]
and all three vanish at \([0:1:0]\), so this point is singular.

On the affine chart \(Z=1\), the map \(\nu\) is inverse to the rational map
\[
(\lambda,y)\longmapsto (\lambda,Y),\qquad Y=2y+1-2\lambda^2,
\]
between the affine quartic \(\Pi(\lambda,y)=0\) and the affine cubic \(E:Y^2=4\lambda^3-4\lambda+1\).
Hence \(\nu\) is birational on dense affine opens.

Let \(O\in E\) be the point at infinity on the smooth projective cubic.
For a Weierstrass cubic one has
\[
\operatorname{ord}_O(\lambda)=-2,
\qquad
\operatorname{ord}_O(Y)=-3
\]
\cite{Silverman2009,Miranda1995}.
Since
\[
y=\lambda^2-\frac12+\frac12Y,
\]
it follows that
\[
\operatorname{ord}_O(y)=\min\{-4,-3,0\}=-4.
\]
Therefore \(\lambda/y\) and \(1/y\) are regular at \(O\) and vanish there.
Writing
\[
\nu=[\lambda:y:1]=[\lambda/y:1:1/y]
\]
shows that \(\nu\) extends to \(O\) and sends \(O\) to \([0:1:0]\).
Conversely, any point of \(E\) mapping to \(Z=0\) must be a pole of \(y\), and \(y\) has its unique pole at \(O\).
Thus \(\nu^{-1}([0:1:0])=\{O\}\).

Because \(E\) is projective and \(\nu\) is a nonconstant morphism onto the irreducible curve \(\overline C\), the map \(\nu\) is finite.
It is already birational on the affine loci, so \(\nu\) is a finite birational morphism.
Since \(E\) is smooth, it is normal, and therefore it is the normalization of \(\overline C\).
\end{proof}

\subsection{Riemann--Hurwitz consistency}\label{subsec:riemann-hurwitz}

The specialization above admits a strong geometric consistency check via the Riemann--Hurwitz formula \cite{Miranda1995}.

\begin{corollary}\label{cor:riemann-hurwitz}
The total ramification index of the degree-\(4\) map \(E \to \mathbf{P}^1_y\) equals \(8\), in agreement with the Riemann--Hurwitz formula.
\end{corollary}

\begin{proof}
The Riemann--Hurwitz formula for a degree-\(d\) map from a curve of genus \(g\) to \(\mathbf P^1\) gives
\[
2g-2=d(-2)+\sum_P(e_P-1).
\]
With \(g=1\) and \(d=4\), the total ramification must be \(8\).
At infinity the map has a single totally ramified point, contributing \(4-1=3\).
From \eqref{eq:ramification-poly}, there are five finite simple ramification points, each contributing \(1\).
Hence the finite contribution is \(5\), and \(3+5=8\).
\end{proof}

\subsection{Discriminant factorization as a corollary}\label{subsec:disc-corollary}

The branch locus of the degree-\(4\) covering coincides with the vanishing locus of the \(\lambda\)-discriminant.

\begin{lemma}[Finite repeated roots and ramification]\label{lem:repeated-root-ramification}
Let \(P=(\lambda_0,Y_0)\in E\) be a point with finite \(y\)-value \(y_0\).
Then the following are equivalent:
\begin{enumerate}
\item \(\lambda_0\) is a repeated root of the quartic \(\Pi(\cdot,y_0)\).
\item \(P\) is a finite ramification point of the map \(y:E\to \mathbf P^1_y\).
\end{enumerate}
\end{lemma}

\begin{proof}
On the affine quartic one has
\begin{equation}\label{eq:Pi-y-equals-Y}
\Pi_y(\lambda,y)=2y+1-2\lambda^2=Y.
\end{equation}
Away from \(Y=0\), implicit differentiation of \(\Pi(\lambda,y)=0\) therefore gives
\[
\frac{\dd y}{\dd\lambda}
=
-\frac{\Pi_\lambda(\lambda,y)}{\Pi_y(\lambda,y)}
=
-\frac{\Pi_\lambda(\lambda,y)}{Y}.
\]
Thus, whenever \(Y_0\neq 0\), the conditions \(\Pi(\lambda_0,y_0)=0=\Pi_\lambda(\lambda_0,y_0)\) and \(\dd y/\dd\lambda(P)=0\) are equivalent.

It remains to rule out the case \(Y_0=0\) at a finite repeated root.
If \(Y_0=0\), then \(2y_0+1=2\lambda_0^2\), hence
\[
\Pi_\lambda(\lambda_0,y_0)
=
4\lambda_0^3-3\lambda_0^2-2(2y_0+1)\lambda_0+1
=
-3\lambda_0^2+1.
\]
Therefore \(\Pi_\lambda(\lambda_0,y_0)=0\) would force \(\lambda_0=\pm 1/\sqrt3\).
But then
\[
Y_0^2=4\lambda_0^3-4\lambda_0+1
=
1\mp \frac{8}{3\sqrt3}\neq 0,
\]
contradicting \(Y_0=0\).
So \(Y_0\neq 0\) at every finite repeated root, and the equivalence follows.
\end{proof}

\begin{corollary}[Discriminant factorization]\label{cor:disc-factor}
The discriminant of \(\Pi(\lambda,y)\) with respect to \(\lambda\) factors as
\begin{equation}\label{eq:disc-factor-new}
\Disc_\lambda \Pi(\lambda,y) = -y(y-1)\bigl(256y^3 + 411y^2 + 165y + 32\bigr).
\end{equation}
\end{corollary}

\begin{proof}
By Lemma~\ref{lem:repeated-root-ramification}, a finite value \(y_0\) is a branch value of \(y:E\to \mathbf P^1_y\) if and only if \(\Pi(\cdot,y_0)\) has a repeated root.
Equivalently, the finite branch values are exactly the zeros of \(\Disc_\lambda\Pi(\lambda,y)\).
Theorem~\ref{thm:elliptic-reduction} identifies them as \(y=0\), \(y=1\), and the roots of \(c(y)=256y^3+411y^2+165y+32\).
Appendix~\ref{app:disc} computes the resultant explicitly and shows that \(\Disc_\lambda\Pi\) is a degree-\(5\) polynomial in \(y\) with leading coefficient \(-1\).
These roots and the leading coefficient determine the factorization uniquely, giving \eqref{eq:disc-factor-new}.
\end{proof}

\begin{remark}\label{rem:genus-one-upgrade}
For the present transfer polynomial, the substitution \(\xi=y-\lambda^2\) is not an ad hoc trick.
It is the specialization of the balanced family \eqref{eq:balanced-family}, in which the same change of variables always removes the quartic obstruction and converts the curve to Weierstrass form.
This is the structural reason why the later branch-point and dominant-branch analysis can be carried out exactly.
\end{remark}
