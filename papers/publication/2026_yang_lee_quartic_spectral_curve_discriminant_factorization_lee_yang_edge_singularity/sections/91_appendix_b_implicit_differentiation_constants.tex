\section{Implicit differentiation and fluctuation constants}\label{app:implicit}

Let \(\lambda_+(y)\) denote the analytic branch of the spectral curve satisfying
\(\Pi(\lambda_+(y),y)=0\) and \(\lambda_+(1)=2\).
Write \(\Pi_\lambda=\partial_\lambda \Pi\) and \(\Pi_y=\partial_y \Pi\).
By implicit differentiation,
\begin{equation}\label{eq:implicit-first}
\lambda_+'(y)=-\frac{\Pi_y(\lambda_+(y),y)}{\Pi_\lambda(\lambda_+(y),y)}.
\end{equation}

From \eqref{eq:spectral-poly},
\begin{align*}
\Pi_\lambda(\lambda,y) &= 4\lambda^3-3\lambda^2-2(2y+1)\lambda+1,\\
\Pi_y(\lambda,y) &= -2\lambda^2+2y+1.
\end{align*}
At \((\lambda,y)=(2,1)\) we have \(\Pi_\lambda(2,1)=9\) and \(\Pi_y(2,1)=-5\), hence
\[
\lambda_+'(1)=\frac{5}{9}.
\]

\paragraph{Second derivative.}
Differentiating \(\Pi(\lambda_+(y),y)=0\) once more yields
\[
\Pi_{\lambda\lambda}(\lambda_+')^2+2\Pi_{\lambda y}\lambda_+'+\Pi_\lambda\lambda_+''+\Pi_{yy}=0,
\]
where
\begin{align*}
\Pi_{\lambda\lambda}(\lambda,y) &= 12\lambda^2-6\lambda-2(2y+1),\\
\Pi_{\lambda y}(\lambda,y) &= -4\lambda,\\
\Pi_{yy}(\lambda,y) &= 2.
\end{align*}
Substituting \((\lambda,y)=(2,1)\) and \(\lambda_+'(1)=5/9\) gives
\[
\lambda_+''(1)=-\frac{64}{243}.
\]

\paragraph{From \(\lambda_+(y)\) to \(\Lambda(s)\).}
Let \(\Lambda(s)=\log \lambda_+(e^s)-\log 2\) and set \(y=e^s\).
Then
\[
\frac{\dd}{\dd s}\lambda_+(e^s)=y\lambda_+'(y),
\qquad
\frac{\dd^2}{\dd s^2}\lambda_+(e^s)=y\lambda_+'(y)+y^2\lambda_+''(y).
\]
At \(s=0\) (i.e.\ \(y=1\)),
\[
\lambda_+(1)=2,\qquad
\left.\frac{\dd}{\dd s}\lambda_+(e^s)\right|_{s=0}=\frac{5}{9},\qquad
\left.\frac{\dd^2}{\dd s^2}\lambda_+(e^s)\right|_{s=0}=\frac{71}{243}.
\]
Therefore,
\[
\mu=\Lambda'(0)=\frac{(5/9)}{2}=\frac{5}{18},
\qquad
\sigma^2=\Lambda''(0)=\frac{(71/243)}{2}-\left(\frac{5}{18}\right)^2=\frac{67}{972}.
\]
