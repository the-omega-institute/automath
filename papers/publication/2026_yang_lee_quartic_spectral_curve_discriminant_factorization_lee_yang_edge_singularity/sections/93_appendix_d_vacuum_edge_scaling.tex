\section{Verification of the endpoint EP2 hypotheses at \texorpdfstring{\(y=0\)}{y=0}}\label{app:vacuum-edge}

This appendix verifies the hypotheses of Lemma~\ref{lem:ep2-cosine-law} for the quartic transfer family at the square-root endpoint $y_c=0$ and records the branch expansion predicted by Lemma~\ref{lem:generic-ep2-branching}.
The local computation produces the constants
\[
\lambda_c=1,\qquad
a=\frac{1}{\sqrt2},\qquad
b=\frac38,\qquad
C_0=\frac12,\qquad
c=\frac{3\sqrt2}{16},
\]
and yields the model-specific scaling law stated in Theorem~\ref{thm:vacuum-zero-quantization}.

\subsection{Partial fractions and local spectral branches}

For $y$ away from the branch locus, the quartic polynomial $\Pi(\lambda,y)$ has four simple roots $\lambda_1(y),\dots,\lambda_4(y)$, and the rational generating function admits the partial-fraction decomposition
\begin{equation}\label{eq:partial-fractions}
Z(t,y)=\sum_{j=1}^4 \frac{C_j(y)}{1-\lambda_j(y)t},
\qquad
C_j(y)=\frac{\lambda_j(y)^3\,N(1/\lambda_j(y),y)}{\partial_\lambda\Pi(\lambda_j(y),y)}.
\end{equation}
Consequently,
\begin{equation}\label{eq:spectral-decomposition}
Z_m(y)=\sum_{j=1}^4 C_j(y)\lambda_j(y)^m.
\end{equation}

Near $y=0$, write $y=s^2$.
Since
\[
\Pi(1,0)=\Pi_\lambda(1,0)=0,
\qquad
\Pi_{\lambda\lambda}(1,0)=4,
\qquad
\Pi_y(1,0)=-1,
\]
Lemma~\ref{lem:generic-ep2-branching} yields \(s_0>0\) and two holomorphic branches \(\lambda_\pm(s)\) for \(|s|<s_0\) with \(\lambda_\pm(0)=1\).
The simple roots at $\lambda=-1$ and $\lambda=0$ give analytic branches $\lambda_{-1}(y)$ and $\lambda_0(y)$ by the implicit function theorem.
Because the residue formula \eqref{eq:partial-fractions} is algebraic in $\lambda_j$ and $y$, the amplitudes $C_\pm(s)$ are holomorphic as well.
The Puiseux expansion from Proposition~\ref{prop:vacuum-puiseux} gives
\begin{equation}\label{eq:lambda-plus-minus}
\lambda_\pm(s)
=
1\pm \frac{1}{\sqrt2}s+\frac58 s^2 \mp \frac{43}{64\sqrt2}s^3+O(s^4).
\end{equation}
Substituting \eqref{eq:lambda-plus-minus} into \eqref{eq:partial-fractions} yields
\begin{equation}\label{eq:C-plus-minus}
C_\pm(s)
=
\frac12\mp \frac{3\sqrt2}{16}s+O(s^2).
\end{equation}
In particular,
\[
\lambda_\pm(s)
=
\exp\!\left(\pm \frac{1}{\sqrt2}s+\frac38 s^2+O(s^3)\right),
\]
so the two branches issuing from $\lambda_c=1$ have the square-root form used in Lemma~\ref{lem:ep2-cosine-law}.

For the remaining two branches, a direct Taylor expansion of $\Pi(\lambda,y)=0$ near $(\lambda,y)=(-1,0)$ and $(0,0)$ gives
\begin{equation}\label{eq:other-branches}
\lambda_{-1}(y)=-1-\frac{y}{4}+O(y^2),
\qquad
\lambda_0(y)=-y+O(y^2),
\end{equation}
while the residue coefficients satisfy
\begin{equation}\label{eq:other-amplitudes}
C_{-1}(y)=\frac{y}{8}+O(y^2),
\qquad
C_0(y)=-y+O(y^2).
\end{equation}

\subsection{Endpoint-scale remainder bound}

Fix a compact set \(K\subset\CC\) and let \(M_K:=\sup_{u\in K}\abs{u}\).
Set
\[
y=-\frac{u^2}{m^2},
\qquad u\in K.
\]
Then \(\abs{y}\le M_K^2m^{-2}\), so for all sufficiently large \(m\) the point \(y\) lies in a fixed small disc around \(0\) on which the expansions \eqref{eq:other-branches}--\eqref{eq:other-amplitudes} hold uniformly.

Define \(h(y):=-\lambda_{-1}(y)\).
Since \(h(0)=1\) and \(h(y)=1+y/4+O(y^2)\), the function \(h\) is nonvanishing near \(0\), and the principal logarithm gives
\[
\log h(y)=\frac{y}{4}+O(y^2).
\]
Hence, for the corresponding analytic branch of the logarithm of \(\lambda_{-1}\),
\begin{equation}\label{eq:log-lambda-minus-one}
\log \lambda_{-1}(y)=i\pi+\frac{y}{4}+O(y^2).
\end{equation}
Exponentiating,
\[
\lambda_{-1}(y)^m
=
(-1)^m\exp\!\left(\frac{my}{4}+O(my^2)\right).
\]
Since \(my=O_K(m^{-1})\) and \(my^2=O_K(m^{-3})\), the amplitude expansion \(C_{-1}(y)=y/8+O(y^2)\) yields
\[
\left|
C_{-1}\!\left(-\frac{u^2}{m^2}\right)
\lambda_{-1}\!\left(-\frac{u^2}{m^2}\right)^m
\right|
\le C_K m^{-2}e^{C_K/m}
=
O_K(m^{-2}).
\]

For the root near \(0\), the expansions \eqref{eq:other-branches}--\eqref{eq:other-amplitudes} give
\[
\lambda_0(y)=-y+O(y^2),
\qquad
C_0(y)=-y+O(y^2).
\]
Therefore
\[
\left|
C_0\!\left(-\frac{u^2}{m^2}\right)
\lambda_0\!\left(-\frac{u^2}{m^2}\right)^m
\right|
\le
C_K m^{-2}\bigl(C_K m^{-2}\bigr)^m
=
O_K(m^{-2m-2}).
\]

Writing
\[
R_m(y):=
C_{-1}(y)\lambda_{-1}(y)^m+C_0(y)\lambda_0(y)^m,
\]
we conclude that
\[
R_m\!\left(-\frac{u^2}{m^2}\right)=O_K(m^{-2})
\]
uniformly for \(u\in K\).
Since \(\lambda_c=1\), this is precisely the remainder estimate required by Lemma~\ref{lem:ep2-cosine-law}.
In particular, the same bound holds on every closed disc \(\{u\in\CC:\abs{u-u_k^{(0)}}\le \eta_k\}\) used in the Rouch\'e step.

\subsection{Specialization of the local normal form}

Fix again a compact set $K\subset\CC$.
The logarithm of \eqref{eq:lambda-plus-minus} has the expansion
\begin{equation}\label{eq:log-lambda-plus-minus}
\log \lambda_\pm(s)
=
\pm \frac{1}{\sqrt2}s+\frac38 s^2+O(s^3).
\end{equation}
With $s=iu/m$, one obtains
\begin{equation}\label{eq:lambda-power}
\lambda_\pm\!\left(\frac{iu}{m}\right)^m
=
\exp\!\left(
\pm \frac{iu}{\sqrt2}-\frac{3u^2}{8m}+O_K(m^{-2})
\right).
\end{equation}
Similarly, \eqref{eq:C-plus-minus} becomes
\begin{equation}\label{eq:amplitude-scale}
C_\pm\!\left(\frac{iu}{m}\right)
=
\frac12\mp i\frac{3\sqrt2}{16}\frac{u}{m}+O_K(m^{-2}).
\end{equation}
Therefore the constants in Lemma~\ref{lem:ep2-cosine-law} are
\[
a=\frac{1}{\sqrt2},
\qquad
b=\frac38,
\qquad
C_0=\frac12,
\qquad
c=\frac{3\sqrt2}{16}.
\]
Substituting \eqref{eq:lambda-power} and \eqref{eq:amplitude-scale} into \eqref{eq:spectral-decomposition} gives
\begin{align}
Z_m\!\left(-\frac{u^2}{m^2}\right)
&=
\exp\!\left(-\frac{3u^2}{8m}\right)
\Biggl[
\left(\frac12-i\frac{3\sqrt2}{16}\frac{u}{m}\right)e^{iu/\sqrt2}
+\left(\frac12+i\frac{3\sqrt2}{16}\frac{u}{m}\right)e^{-iu/\sqrt2}
+O_K(m^{-2})
\Biggr] \notag\\
&=
\exp\!\left(-\frac{3u^2}{8m}\right)
\left[
\cos\!\left(\frac{u}{\sqrt2}\right)
+\frac{3u}{4\sqrt2\,m}\sin\!\left(\frac{u}{\sqrt2}\right)
+O_K(m^{-2})
\right].\label{eq:appendix-scaling}
\end{align}
Using
\[
\cos(\theta-\delta)=\cos\theta+\delta\sin\theta+O(\delta^2),
\qquad
\delta=\frac{3u}{4\sqrt2\,m},
\]
one obtains the uniform scaling law
\begin{equation}\label{eq:appendix-shifted-cosine}
Z_m\!\left(-\frac{u^2}{m^2}\right)
=
\exp\!\left(-\frac{3u^2}{8m}\right)
\left[
\cos\!\left(\frac{u}{\sqrt2}-\frac{3u}{4\sqrt2\,m}\right)
+ O_K(m^{-2})
\right].
\end{equation}
This is \eqref{eq:vacuum-scaling}, namely the specialization of the local cosine law to the quartic family.

\subsection{Extraction of the first zeros}

Fix $k\ge1$ and define $q_k=(2k-1)\pi/2$.
Applying Lemma~\ref{lem:ep2-cosine-law} with
\[
a=\frac{1}{\sqrt2},
\qquad
C_0=\frac12,
\qquad
c=\frac{3\sqrt2}{16},
\]
gives a unique zero $u_{m,k}$ of the exact function $u\mapsto Z_m(-u^2/m^2)$ in a small neighborhood of \(\sqrt2\,q_k\), and
\[
u_{m,k}
=
\sqrt2\,q_k\left(1+\frac{3}{4m}\right)+O_k(m^{-2})
=
\frac{\pi(2k-1)}{\sqrt2}\left(1+\frac{3}{4m}\right)+O_k(m^{-2}).
\]
Finally,
\[
y_{m,k}=-\frac{u_{m,k}^2}{m^2}
=
-\frac{\pi^2(2k-1)^2}{2m^2}
-\frac{3\pi^2(2k-1)^2}{4m^3}
+O_k(m^{-4}),
\]
which is the quantization formula stated in Theorem~\ref{thm:vacuum-zero-quantization}.
