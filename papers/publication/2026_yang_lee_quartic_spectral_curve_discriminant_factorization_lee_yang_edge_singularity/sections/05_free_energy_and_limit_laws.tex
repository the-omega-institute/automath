\section{Free energy and limit laws}\label{sec:free}

\subsection{Free energy and the dominant spectral branch}

For \(y>0\), define the (per-step) free energy, whenever the limit exists,
\begin{equation}\label{eq:free-energy}
f(y)=\lim_{m\to\infty}\frac{1}{m}\log Z_m(y).
\end{equation}
By the order-\(4\) recurrence \eqref{eq:recurrence}, we have a spectral decomposition of the form
\[
Z_m(y)=\sum_{i=1}^4 a_i(y)\lambda_i(y)^m,
\]
where \(\lambda_i(y)\) are the roots of \(\Pi(\lambda,y)=0\) and \(a_i(y)\) are determined by the initial conditions.
When a unique dominant root (in modulus) exists and its coefficient does not vanish, it controls the exponential growth rate.

\begin{theorem}\label{thm:lambda-plus}
There exists an open neighborhood \(U\subset\CC\setminus\Branch\) of \(y=1\) and a function \(\lambda_+(y)\) analytic on \(U\) such that
\[
\Pi(\lambda_+(y),y)=0,\qquad \lambda_+(1)=2.
\]
Moreover, for \(y\in U\cap(0,\infty)\), the free energy exists and satisfies
\[
f(y)=\log \lambda_+(y).
\]
\end{theorem}

\begin{proof}
At \(y=1\) the spectral polynomial factors as
\[
\Pi(\lambda,1)=(\lambda-2)(\lambda-1)(\lambda+1)^2,
\]
so \(\lambda=2\) is a simple root.
By the analytic implicit function theorem, there is a unique analytic branch \(\lambda_+(y)\) defined near \(y=1\) with \(\lambda_+(1)=2\), and \(U\) can be chosen to avoid the branch locus \(\Branch\).
For \(y>0\) sufficiently close to \(1\), continuity of the roots implies \(\abs{\lambda_+(y)}\) is strictly larger than the moduli of the other roots.
The spectral expansion then gives \(Z_m(y)=a_+(y)\lambda_+(y)^m(1+o(1))\) with \(a_+(y)\neq 0\), hence \eqref{eq:free-energy} holds with \(f(y)=\log\lambda_+(y)\).
\end{proof}

\subsection{An exponential family and the limiting log-mgf}

Let \(H_m\) be a random variable taking values in \(\{0,1,\dots,m\}\) whose moment generating function is defined by
\begin{equation}\label{eq:mgf-def}
\mathbb{E}\bigl[e^{sH_m}\bigr]=\frac{Z_m(e^s)}{Z_m(1)}.
\end{equation}
Since \(Z_m(1)=2^m\), this defines an exponential tilting of the index \(m\) via the parameter \(s\).
Define the limiting log-moment generating function, when it exists,
\[
\Lambda(s)=\lim_{m\to\infty}\frac{1}{m}\log \mathbb{E}\bigl[e^{sH_m}\bigr].
\]

\begin{theorem}\label{thm:lambda-mgf}
There exists a complex neighborhood \(V\) of \(s=0\) such that for all \(s\in V\),
\[
\Lambda(s)=\log \lambda_+(e^s)-\log \lambda_+(1)=\log \lambda_+(e^s)-\log 2.
\]
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:lambda-plus} applied with \(y=e^s\), we have
\(\log Z_m(e^s)=m\log \lambda_+(e^s)+O(1)\) uniformly for \(s\) in a small neighborhood of \(0\).
The same holds at \(s=0\) with \(\lambda_+(1)=2\).
Subtracting and dividing by \(m\) in \eqref{eq:mgf-def} yields the claim.
\end{proof}

\subsection{Closed-form constants for LLN and CLT}

Let \(\mu=\Lambda'(0)\) and \(\sigma^2=\Lambda''(0)\).

\begin{theorem}\label{thm:lln-clt}
We have
\[
\mu=\frac{5}{18},\qquad \sigma^2=\frac{67}{972}.
\]
Moreover:
\begin{enumerate}
  \item \(m^{-1}H_m\to \mu\) in probability as \(m\to\infty\);
  \item \(\displaystyle \frac{H_m-\mu m}{\sqrt{m}}\Rightarrow \mathcal{N}(0,\sigma^2)\) in distribution.
\end{enumerate}
\end{theorem}

\begin{proof}
The constants follow from implicit differentiation of \(\Pi(\lambda_+(y),y)=0\) at \(y=1\) and the chain rule \(y=e^s\); see Appendix~\ref{app:implicit}.

The convergence statements follow from the analytic expansion \(\log \mathbb{E}[e^{sH_m}]=m\Lambda(s)+O(1)\) in a complex neighborhood of \(s=0\), a standard setting of the quasi-powers theorem (e.g.\ \cite{Hwang1998,FlajoletSedgewick2009}).
\end{proof}
